class Solution {
    public int tilingRectangle(int n, int m) {
        dp[11][13] =6;
        
        return gcd(n,m);
    }
    
    int dp[][]=  new int[15][15];
    
    int gcd(int n, int m) {
        
        if(n>m){
            int t = n;
            n = m;
            m = t;
        }
        if(dp[n][m]!=0) return dp[n][m];
        if(n == 0||m==0) return 0;
        if(n==m) return 1;
        int mx =100000;
        for(int i=1;i<=n;++i){
            mx = Math.min(1+gcd(m-i,i)+gcd(n-i,m),mx);
            mx = Math.min(1+gcd(m-i,n)+gcd(n-i,i),mx);
        }
        return dp[n][m] = mx;
    }
}
