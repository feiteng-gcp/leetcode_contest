class Solution {
    public int tilingRectangle(int n, int m) {
        if(n == m) return 1;
        if(n < m) {
            int temp = n;
            n = m;
            m = temp;
        }
        int min = Integer.MAX_VALUE;
        for(int i = m; i >= 0; i--) {
            int temp = helper(n, m, i);
            if(temp < min) min = temp;
        }
        return min;
    
}
    
    int helper(int n, int m, int s) {
        int count = 1;
        char[][] map = new char[n][m];
        for(char[] row : map) Arrays.fill(row, '1');
        
for(int i = 0; i < s; i++) {
            for(int j = 0; j < s; j++) {
                map[i][j] = '0';
            } 
        }
        while(!isFilled(map)) {
            count++;
            fillSquare(map);
        }
        return count;
    }
    
    boolean isFilled(char[][] map) {
        for(int i = 0; i < map.length; i++) {
            for(int j = 0; j < map[0].length; j++) {
                
if(map[i][j] == '1') return false;
            }
        }
        return true;
    }
      void fillSquare(char[][] matrix) {
        int rows = matrix.length, cols = rows > 0 ? matrix[0].length : 0;
        int maxsqlen = 0;
        int si = 0;
        int sj = 0;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (matrix[i][j] == '1') {
                    int sqlen = 1;
                    boolean flag = true;
                    while (sqlen + i < rows && sqlen + j < cols && flag) {
                        for (int k = j; k <= sqlen + j; k++) {
                            if (matrix[i + sqlen][k] == '0') {
                                flag = false;
                                break;
                            }
                        }
                        for (int k = i; k <= sqlen + i; k++) {
                            if (matrix[k][j + sqlen] == '0') {
                                flag = false;
                                break;
                            }
                        }
                        if (flag)
                            sqlen++;
                    }
                    if (maxsqlen < sqlen) {
                        maxsqlen = sqlen;
                        si = i;
                        sj = j;
    
                    }
                }
            }
        }
          
        
for(int i = si; i < si + maxsqlen; i++) {
            for(int j = sj; j < sj + maxsqlen; j++) {
                matrix[i][j] = '0';
            }
        }
          
    }

}
