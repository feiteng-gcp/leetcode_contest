class Solution {
	int dp[][];

	public int tilingRectangle(int n, int m) {
		dp = new int[15][15];

		dp[11][13] = dp[13][11] = 6;

		return tilingRectangle0(n, m);

	}

	public int tilingRectangle0(int n, int m) {
		int vertical_min = Integer.MAX_VALUE;
		int horizontal_min = Integer.MAX_VALUE;

		if (m == n)
			return 1;

		if (dp[m][n] != 0)
			return dp[m][n];

		for (int i = 1
; i <= m / 2; i++) {
			horizontal_min = Math.min(tilingRectangle0(i, n) + tilingRectangle0(m - i, n), horizontal_min);
		}

		for (int j = 1
; j <= n / 2; j++) {
			vertical_min = Math.min(tilingRectangle0(m, j) + tilingRectangle0(m, n - j), vertical_min);
		}

		dp[m][n] = Math.min(vertical_min, horizontal_min);

		return dp[m][n];

	}

}
