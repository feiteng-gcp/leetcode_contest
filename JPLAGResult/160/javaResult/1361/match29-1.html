class Solution {
   int n, m;

    int curMin;
    int[][] mat;

    public int tilingRectangle(int n, int m) {

        //dp = new int[n][1 << m];

        // first line

        this.n = n;
        this.m = m;
        this.mat = new int[n][m];

        if (n >= m) {
            curMin = 1 + (n - m) * m;
        } else {
            curMin = 1 + (m - n) * n;
        }

        dfs(0, 0, 0);
        return curMin;



    }

    void dfs(int x, int count, int step) {
        if (count == n * m) {
            curMin = Math.min(curMin, step);
            return;
        }

        if (step >= curMin) return;

        int i = 0;
        int j = 0;

        for (; i < m; i++) {
            if (mat[x][i] == 0) {
                break;
            }
        }

        if (i >= m) {
            dfs(x + 1, count, step);
            return;
        }

        for (j = i; j < m; j++) {
            if (mat[x][j] == 1) break;
        }

        for (int len = j - i; len >= 1; len--) {
            //int k = i + len - 1;

            if (x + len > n) continue;
            if (i + len > m) continue;

            
boolean ok = true;
            for (int r = x; r < x + len && r < n; r++) {
                for (int c = i; c < i + len && c < m; c++) {
                    if (mat[r][c] == 1) {
                        ok = false;
                        break;
                    }
                }
            }

            if (ok) {
                
for (int r = x; r < x + len && r < n; r++) {
                    for (int c = i; c < i + len && c < m; c++) {
                        mat[r][c] = 1;
                    }
                }

                int newCount =  count + len * len;

                dfs(x, newCount, step + 1);

                
for (int r = x; r < x + len && r < n; r++) {
                    for (int c = i; c < i + len && c < m; c++) {
                        mat[r][c] = 0;
                    }
                }

            }
        }
    }
}
