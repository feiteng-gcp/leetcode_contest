class Solution {
    
    int maxValue = 13*13;
    
    public int tilingRectangle(int n, int m) {
        int[][] arr = new int[n][m];
        
        int maxVal = Math.max(n,m);
        
        find(arr,n,m,maxVal,0,0);
        
        
return maxValue;
    }
    
    public void find(int[][] arr, int n, int m, int maxVal, int currSize, int steps){
        
        if(steps >= maxValue){
            return;
        }
        
        if(n*m==currSize){
            maxValue = steps;
            return;
        }
        
        for(int i=maxVal;i>=1;i--){
            int[] pos = canAdd(arr,n,m,i);
            if(pos[0]>=0 && pos[1]>=0){
                find(arr, n, m, maxVal, currSize + (i*i), steps+1);
                remove(arr,pos,i);
            }
        }
    }
    
    public void remove(int[][] arr, int[] pos, int k){
        
        
for(int i=pos[0];i<pos[0]+k;i++){
            for(int j=pos[1];j<pos[1]+k;j++){
                arr[i][j]=0;
            }
        }
        
    }
    
    public int[] canAdd(int[][] arr, int n, int m, int val){
        
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(arr[i][j]==0){
                    if(i+val>n || j+val>m){
                        return new int[]{-1,-1};
                    }
                    for(int k=i;k<i+val;k++){
                        for(int l=j;l<j+val;l++){
                            if(arr[k][l]==1){
                                return new int[]{-1,-1};
                            }
                        }
                    }
                    
                    
for(int k=i;k<i+val;k++){
                        for(int l=j;l<j+val;l++){
                            arr[k][l]=1;
                        }
                    }
                    
                    return new int[]{i,j};
                }
            }
        }
        return new int[]{-1,-1};
    }
    
}
