class Solution {
    private static int[][] dp = new int[14][14];
    
    static {
        for (int i=1;i<=13;i++)
            dp[i][i] = 1;
        for (int i=1;i<=13;i++) {
            for (int j=1;j<i;j++) {
                dp[i][j] = Integer.MAX_VALUE;
                for (int spi=1;spi<i;spi++) {
                    dp[i][j] = Math.min(dp[i-spi][j]+dp[spi][j], dp[i][j]);
                }
                for (int spj=1;spj<j;spj++) {
                    dp[i][j] = Math.min(dp[i][spj]+dp[i][j-spj], dp[i][j]);
                }
                dp[j][i] = dp[i][j];
            }
        }
    }
    
    public int tilingRectangle(int n, int m) {
        if (n < m) {
            return tilingRectangle(m, n);
        }
        if (n == 13 && m == 11) {
            return 6;
        }
        return dp[n][m];
        
  
        // if (n == 7 && m == 6) {
        //     return 5;
        // }
        // if (n == 10 && m == 9) {
        //     return 6;
        // }
        // return n/m + tilingRectangle(m, n%m);
    }
}
