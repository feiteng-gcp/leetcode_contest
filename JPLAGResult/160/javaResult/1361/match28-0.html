class Solution {
    	public int tilingRectangle(int n, int m) {
		int max = Math.max(n, m);
		int min = Math.min(n, m);
		int minCount = greedy(max, min);
		for (int fl = min - 2; fl >= min / 2; fl--) {
			boolean[][] cells = new boolean[max][min];
			for (int i = 0; i < fl; i++) {
				for (int j = 0; j < fl; j++) {
					cells[i][j] = true;
				}
			}
			int result = tilingRectangle(cells, max, min, 1, fl * fl, minCount);
			minCount = Math.min(minCount, result);
		}
		return minCount;
    }
	
	private int tilingRectangle(boolean[][] cells, int max, int min, int curCnt, int covered, int minCount) {
		if (curCnt == minCount) {
			// give up
			return minCount;
		}
		if (covered == max * min) {
			return curCnt;
		}
		int si = -1;
		int sj = -1;
		OUTER : for (int i = 0; i < max; i++) {
			for (int j = 0; j < min; j++) {
				if (cells[i][j]) {
					continue;
				} else {
					si = i;
					sj = j;
					break OUTER;
				}
			}
		}
		int maxLength = Math.min(max - si, min - sj);
		int minValue = minCount;
		for (int l = maxLength; l >= 1; l--) {
			// copy, fill, recurse
			boolean[][] newCells = new boolean[max][min];
			for (int i = 0; i < max; i++) {
				for (int j = 0; j < min; j++) {
					newCells[i][j] = cells[i][j];
				}
			}
			boolean success = true;
			OUTER : for (int i = 0; i < l; i++) {
				for (int j = 0; j < l; j++) {
					if (newCells[si + i][sj + j]) {
						success = false;
						break OUTER;
					}
					newCells[si + i][sj + j] = true;
				}
			}
			if (success) {
				minValue = Math.min(minValue, tilingRectangle(newCells, max, min, curCnt + 1, covered + l * l, minCount));
			}
		}
		return minValue;
	}
	
	private int greedy(int max, int min) {
		if (min == 0) {
			return 0;
		} else {
			return max / min + greedy(min, max % min);
		}
	}
}
