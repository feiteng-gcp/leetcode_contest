class Solution {
    boolean[][] vst;
    int res = Integer.MAX_VALUE;
    public int tilingRectangle(int n, int m) {
        if(n > m) return tilingRectangle(m, n);
        if(n == m) return 1;
        vst = new boolean[n][m];
        dfs(0, 0);
        return res;
    }
    void dfs(int id, int acc) {
        if(acc >= res) return;
        int n = vst.length;
        int m = vst[0].length;
        if(id == m * n) {
        //  System.out.println(id);
            if(acc < res) res = acc;
            return;
        }
        int cx = id / m;
        int cy = id % m;
        if(vst[cx][cy]) {
            dfs(id + 1, acc);
            return;
        }
        for(int i = 1; i <= Math.min(m, n); ++i) {
            if(helper(cx, cy, i)) {
                for(int a = cx; a < cx + i; ++a) {
                    for(int b = cy; b < cy + i; ++b) {
                        vst[a][b] = true;
                    }
                }
                dfs(id + 1, acc + 1);
                for(int a = cx; a < cx + i; ++a) {
                    for(int b = cy; b < cy + i; ++b) {
                        vst[a][b] = false;;
                    }
                }                
            } else break;
        }
    }
    boolean helper(int x, int y, int L) {
        int n = vst.length;
        int m = vst[0].length;
        for(int i = x; i < x + L; ++i) {
            for(int j = y; j < y + L; ++j) {
                if(i >= n || j >= m) return false;
                if(vst[i][j]) return false;
            }
        }
        return true;
    }
}
