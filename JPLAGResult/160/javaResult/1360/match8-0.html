class Solution {
    
    private int ans = 0;
    
    public int maxLength(List<String> arr) {
        int n = arr.size();
        help(arr, 0, n, new HashSet<>());
        return ans;
    }
    
    private void help(List<String> arr, int s, int n, Set<Character> set) {
        if (s >= n) {
            ans = Math.max(ans, set.size());
            return;
        }
        
        // add
        Set<Character> v = new HashSet<>();
        String cur = arr.get(s);
        
        boolean add = true;
        for (int i = 0; i < cur.length(); i++) {
            if (v.contains(cur.charAt(i))) {
                add = false;
            } else {
                v.add(cur.charAt(i));
            }
            if (set.contains(cur.charAt(i))) add = false;
        }
        
        if (!add) {
            help(arr, s + 1, n, set);
            return;
        }
        if (add) {
            help(arr, s + 1, n, set);
            
            for (int i = 0; i < cur.length(); i++) {
                set.add(cur.charAt(i));
            }
            help(arr, s + 1, n, set);
            
            for (int i = 0; i < cur.length(); i++) {
                set.remove(cur.charAt(i));
            }
        }
    }
}
