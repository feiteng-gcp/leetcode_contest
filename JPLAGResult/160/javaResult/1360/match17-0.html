class Solution {
    List<Integer> map = new ArrayList<>(), len = new ArrayList<>();
    int max = 0;
    public int maxLength(List<String> arr) {
        outer: for (String s : arr) {
            int appear = 0;
            for (char c : s.toCharArray()) {
                int mask = 1 << (c - 'a');
                if ((appear & mask) != 0) continue outer;
                appear |= mask;
            }
            map.add(appear);
            len.add(s.length());
        }
        dfs(0, 0, 0);
        return max;
    }
    private void dfs(int l, int idx, int cur) {
        if (idx == map.size()) return;
        for (int i = idx; i < map.size(); ++i) {
            int next = map.get(i);
            if ((next & cur) == 0) {
                max = Math.max(l + len.get(i), max);
                dfs(l + len.get(i), i + 1, next | cur);
            }
        }
    }
}
