class Solution {
    public int maxLength(List<String> arr) {
        
        int n = arr.size();
        int longest = 0;
        outer:
        for(int i=1; i<1<<n; ++i) {
            boolean[] f = new boolean[26];
            int length = 0;
            for(int j=0; j<n; ++j) {
                if(i<<~j < 0) {
                    length += arr.get(j).length();
                    for(int k=0; k<arr.get(j).length(); ++k) {
                        int c = arr.get(j).charAt(k) - 'a';
                        if(f[c]) {
                            continue outer;
                        }else {
                            f[c] = true;
                        }
                    }
                }
            
}
            
            longest = Math.max(longest, length);
        }
        return longest;
    }
}
