class Solution {
    int res = 0;
    public int maxLength(List<String> arr) {
        int n = arr.size();
        String[] A = arr.toArray(new String[n]);
        int[] map = new int[26];
        
dfs(A, 0, 0, map);
        return res;
    }
    
    void dfs(String[] A, int k, int cnt, int[] map) {
        int n = A.length;
        if (n == k) {
            res = Math.max(cnt, res);
            return;
        }
        dfs(A, k + 1, cnt, map); // not concatenate A[k]
        boolean canAdd = true;
        for (char c: A[k].toCharArray()) 
{
            if (map[c - 'a'] > 0) {
                canAdd = false;
            }
            map[c - 'a']++;
        }
        if (canAdd) {
            dfs(A, k + 1, cnt + A[k].length(), map);
        }
        for (char c: A[k].toCharArray()) {
            map[c - 'a']--;
        }
    }
}
