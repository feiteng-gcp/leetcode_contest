	class Solution {
	    public int maxLength(List<String> arr) {
	    	int n = arr.size();
	        int[] ptns = new int[n];
	        outer:
	        for(int i = 0;i < n;i++){
	        	int ptn = 0;
	        	for(char c : arr.get(i).toCharArray()){
	        		if(ptn<<~(c-'a')<0){
	        			ptns[i] = -1;
	        			continue outer;
	        		}

	        		ptn |= 1<<c-'a';
	        	}
	        	ptns[i] = ptn;
	        }
	        
	        int ret = 0;
	        outer:
	        for(int i = 0;i < 1<<n;i++){
	        	int my = 0;
	        	for(int j = 0;j < n;j++){
	        		if(i<<~j<0){
	        			if(ptns[j] == -1)continue outer;
	        			if((my&ptns[j]) != 0)continue outer;

	        			my |= ptns[j];
	        		}
	        	}
	        	ret = Math.max(ret, Integer.bitCount(my));
	        }
	        return ret;
	    }
	}
