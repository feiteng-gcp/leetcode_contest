class Solution {
    int max = 0;
    
public void recurse(List<String> newarr, int j, HashSet<Character> chk1){
        if(j>=newarr.size())
            return;
        for(int i=j;i<newarr.size();i++){
            HashSet<Character> chk2 = new HashSet<>();
            chk2.addAll(chk1);
            for(char ch : newarr.get(i).toCharArray())
                chk2.add(ch);
            if(chk2.size()==chk1.size()+newarr.get(i).length()){
                max = 
Math.max(max,chk2.size());
                recurse(newarr,i+1,chk2);
            }
        }
    }
    public int maxLength(List<String> arr) {
        
        List<String> newarr = new ArrayList<>();
        for(String s : arr){
            HashSet<Character> chk = new HashSet<>();
            for(char ch : s.toCharArray())
                chk.add(ch);
            if(s.length()==chk.size()){
                newarr.add(s);
                max = Math.max(max,chk.size());
            }
        }
        /*for(int i=0;i<newarr.size();i++){
            for(int j=i+1;j<newarr.size();j++){
                HashSet<Character> chk = new HashSet<>();
                for(char ch : newarr.get(i).toCharArray())
                    chk.add(ch);
                for(char ch : newarr.get(j).toCharArray())
                    chk.add(ch);
                if(chk.size()==newarr.get(i).length()+newarr.get(j).length()){
                    max = Math.max(max,chk.size());
                }
            }
        }*/
        for(int i=0;i<newarr.size();i++){
            HashSet<Character> chk = new HashSet<>();
            for(char ch : newarr.get(i).toCharArray())
                chk.add(ch);
            recurse(newarr, i+1, chk);
        }
        return max;
    }
}
