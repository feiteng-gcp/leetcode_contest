class Solution {
    public int maxLength(List<String> arr) {
        int sz = arr.size();
        
        int f[] = new int[sz];
        int g = 0;
        ot:for(String s:arr){
            int d[] = new int[26];
            for(char c:s.toCharArray()){
                if(d[c-'a']>0){
                    continue  ot;
                }
                d[c-'a']++;
            }
            int st = 0;
            for(int i=0;i<26;++i){
                if(d[i]>0){
                st |= 1<<i;
                }
            }
            f[g++] = st;
//System.out.println(f[g-1]);
        }
        int mx =0;
        
        ot:for(int j=0;j<1<<g;++j){
            int st = 0;
            for(int i=0;i<g;++i){
                if(j<<~i<0){
                    if((st&f[i])==0){
                        
                        st |= f[i];
                    }else{
                        
                        continue ot;
                    }  
                    
                    
                }
                
                
            }
            mx = Math.max(mx,Integer.bitCount(st));
            
            
            
        }
        return mx;
        
    }
}
