class Solution {
    public int maxLength(List<String> arr) {
        dfs(arr,0,"");
        return res;
    }
    public int res = 0;
    public void dfs(List<String> arr, int id, String cur){
        for(int i=id;i<arr.size();i++){
            String s = arr.get(i);
            if(judge(s,cur)){
                res = Math.max(res, s.length()+cur.length());
                dfs(arr,i+1,cur+s);
            }
        }    
    }
    
    public boolean judge(String a, String b){
        int[] map = new int[26];
        for(char c:a.toCharArray()){
            map[c-'a']++;
            if(map[c-'a']>1) return false;
        }
        for(char c:b.toCharArray()){
            map[c-'a']++;
            if(map[c-'a']>1) return false;
        }
        return true;
    }
}
