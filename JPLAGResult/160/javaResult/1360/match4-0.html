class Solution {
    public int maxLength(List<String> arr) {
        final int N = arr.size();
        ArrayList<Integer> H = new ArrayList<>();
        for (int i = 0; i < N; ++i) {
            int h = sig(arr.get(i));
            if (h == -1) {
                continue;
            }
            H.add(h);
        }
        HashSet<Integer> seen = new HashSet<>();
        seen.add(0);
        for (int v : H) {
            HashSet<Integer> add = new HashSet<>();
            for (int h : seen) {
                if ((h & v) == 0) {
                    add.add(h | v);
                }
            }
            seen.addAll(add);
        }
        int best = 0;
        for (int v : seen) {
            best = Math.max(best, Integer.bitCount(v));
        }
        return best;
    }
    
    private static int sig(String s) {
        int h = 0;
        for (char c : s.toCharArray()) {
            int b = c - 'a';
            if (((h >> b) & 1) == 1) {
                return -1;
            }
            h |= (1 << b);
        }
        return h;
    }
}
