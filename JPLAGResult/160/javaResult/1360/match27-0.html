class Solution {
    public int maxLength(List<String> arr) {
        int n = arr.size();
        HashSet<Integer> set = new HashSet<>();
        set.add(0);
        for(String s : arr) {
            Integer a = convert(s);
            if(a == null) continue;
            HashSet<Integer> next = new HashSet<>(set);
            for(int b : set) {
                Integer c = concate(a, b);
                if(c != null) {
                    next.add(c);
                }
            }
            set = next;
        }
        int ans = 0;
        for(int s : set) {
            ans = Math.max(ans, Integer.bitCount(s));
        }
        return ans;
    }
    private Integer convert(String s) {
        boolean[] cnt = new boolean[26];
        for(char c : s.toCharArray()) {
            if(cnt[c-'a']) return null;
            cnt[c-'a'] = true;
        }
        int ret = 0;
        for(int i = 0; i < 26; i++) {
            if(cnt[i]) ret |= (1 << i);
        }
        return ret;
    }
    private Integer concate(int a, int b) {
        if((a & b) != 0) return null;
        int c = a | b;
        return c;
    }
}
