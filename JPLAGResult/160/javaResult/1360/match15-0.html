class Solution {
    public int maxLength(List<String> arr) {
		Set<Integer> set = new HashSet<Integer>();
		set.add(0);
		for (String str : arr) {
			int v = getV(str);
			if (v == -1) {
				continue;
			}
			Set<Integer> newSet = new HashSet<Integer>();
			for (Integer s : set) {
				if ((s & v) > 0) {
					continue;
				} else {
					newSet.add(s + v);
				}
			}
			set.addAll(newSet);
		}
		int ans = 0;
		for (Integer s : set) {
			int v = getL(s);
			ans = Math.max(v, ans);
		}
		return ans;
	}

	private int getL(Integer s) {
		int ans = 0;
		while (s > 0) {
			ans += s % 2;
			s /= 2;
		}
		return ans;
	}

	private int getV(String str) {
		int ans = 0;
		for (int i = 0; i < str.length(); i++) {
			int v = 1 << (str.charAt(i) - 'a');
			if ((ans & v) > 0) {
				return -1;
			}
			ans += v;
		}
		return ans;
	}
}
