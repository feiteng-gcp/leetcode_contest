class Solution {
    
    int res = 0;
    public int maxLength(List<String> arr) {
        dfs(new int[26], arr, 0, 0);
        return res;
    }
    
    public void dfs(int[] visited, List<String> arr, int index, int len) {
        if(index >= arr.size()) return;

        String s = arr.get(index);
        boolean[] curVisited = new boolean[26];
        boolean hasDups = false;
        for(char c : s.toCharArray()) {
            if(visited[c - 'a'] != 0) {
                hasDups = true;
                break;
            }
            if(curVisited[c - 'a']) {
                hasDups = true;
                break;
            }
            curVisited[c - 'a'] = true;
        }
        
        if(!hasDups) {
            for(char c : s.toCharArray()) {
                visited[c - 'a']++;
            }
            res = Math.max(res, len + s.length());
            dfs(visited, arr, index + 1, len + s.length());
            for(char c : s.toCharArray()) {
                visited[c - 'a']--;
            }
        }
        
        dfs(visited, arr, index + 1, len);
    }
}
