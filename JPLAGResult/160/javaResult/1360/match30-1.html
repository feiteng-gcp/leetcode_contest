class Solution {
    int max;
    public int maxLength(List<String> arr) {
        Map<String, Set<String>> map = new HashMap<>();
        
List<String> swap = new ArrayList<String>(26);
        for(String curr : arr) {
            if(unique1(curr)) swap.add(curr);
        }
        arr = swap;
        for(String curr : arr) {
            map.put(curr, new HashSet<String>());
            
for(String next : arr) {
                if(unique(curr, next)) map.get(curr).add(next);
            }
        }
        max = 0;
        helper(arr, map, 0, new ArrayList<String>());
        return max;
         
    }
    
    void helper(List<String> arr, Map<String, Set<String>> map, int start, List<String> list) {
        int temp = 0;
        for(String exist : list) {
            temp = temp + exist.length();
            max = temp > max ? temp : max;
        }
        if(start == arr.size()) return;
        for(int i = start; i < arr.size(); i++) {
            boolean unique = true; 
            String addString = arr.get(i);
            for(String stored : list) {
                if(!map.get(stored).contains(addString)) {
                    unique = false;
                    break;
                }
            }
            if(!unique) continue;
            list.add(addString);
            helper(arr, map, i + 1, list);
            list.remove(list.size() - 1);
        }
    } 
    
    boolean unique1(String a) {
        int[] holder = new int[26];
        String temp = a;
        for(int i = 0; i < temp.length(); i++) {
            int ind = (int)(temp.charAt(i) - 'a');
            holder[ind]++;
            if(holder[ind] > 1) return false
;
        }
        return true;
    }
    
    boolean unique(String a, String b) {
        int[] holder = new int[26];
        String temp = a + b;
        for(int i = 0; i < temp.length(); i++) {
            int ind = (int)(temp.charAt(i) - 'a');
            holder[ind]++;
            if(holder[ind] > 1) return false;
        }
        return true;
    }
}
