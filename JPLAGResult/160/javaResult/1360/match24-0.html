class Solution {
    int res = 0;
    public int maxLength(List<String> arr) {
        int len = arr.size();
        int[][] nums = new int[len][26];
        for(int i=0 ; i<len ; i++){
            nums[i] = getNum(arr.get(i));
        }
        for(int i=0 ; i<len ; i++){
            dfs(nums[i],i,nums);
        }
        return res;
    }
    
    public void dfs(int[] num ,int index , int[][] nums){
        if(index==nums.length || !isValid(num)){
            return;
        }
        res = Math.max(res,get(num));
        
        for(int i=index+1 ; i<nums.length ; i++){
            for(int j=0 ; j<26 ; j++){
                num[j] += nums[i][j];
            }
            dfs(num,i,nums);
            for(int j=0 ; j<26 ; j++){
                num[j] -= nums[i][j];
            }
        }
        
    }
    public int get(int[] num){
        int x = 0;
        for(int i : num){
            if(i==1){
                x++;
            }
        }
        return x;
    }
    
    public boolean isValid(int[] num){
        for(int i : num){
            if(i>1){
                return false;
            }
        }
        return true;
    }
    
    public int[] getNum(String s){
        int[] num = new int[26];
        
        for(int i=0 ; i<s.length() ; i++){
            num[s.charAt(i)-'a'] +=1; 
        }
        return num;
    }
}
