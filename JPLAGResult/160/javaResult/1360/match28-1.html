class Solution {
    int max = 0;
    public int maxLength(List<String> arr) {
        if (arr == null || arr.size() == 0) return max;
        List<String> validArr = new ArrayList<>();
        for (String s : arr) {
            if (validWord(s)) {
                validArr.add(s);
            }
        }
        dfs(validArr, 0, "");
        return max;
    }
    
    private void dfs(List<String> arr, int start, String prev) {
        if (start == arr.size()) return;
        for (int i = start; i < arr.size(); i++) {
            String s = arr.get(i);
            if (overlap(prev, s)) continue;
            
if (prev.length() + s.length() > max) {
                max = prev.length() + s.length();
            }
            dfs(arr, i + 1, prev + s);
        }
    
}
    
    private boolean overlap(String prev, String s) {
        boolean[] chars = new boolean[26];
        for (int i = 0; i < prev.length(); i++) {
            chars[
prev.charAt(i) - 'a'] = true;
        }
        for (int i = 0; i < s.length(); i++) {
            if (chars[s.charAt(i) - 'a']) return true;
        }
        return false;
    }
    
    private boolean validWord(String s) {
        boolean[] chars = new boolean[26];
        for (int i = 0; i < s.length(); i++) {
            if (chars[s.charAt(i) - 'a']) return false;
            chars[s.charAt(i) - 'a'] = true;
        }
        return true;
    }
}
