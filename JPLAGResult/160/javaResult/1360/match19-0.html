class Solution {
    public int maxLength(List<String> arr) {
        int n = arr.size();
        int[][] memo = new int[n][2];
        int index = 0;
        for(String s : arr){
            int base = 0;
            for(char c : s.toCharArray()){
                int cur = 1 << (c - 'a');
                if((cur & base) != 0){
                    memo[index][0] = -1;
                    break;
                }
                base |= cur;
            }
            
            if(memo[index][0] != -1){
                memo[index][0] = base;
                memo[index][1] = s.length();
            }
            
            index++;
        }
        
        return dfs(memo, 0, 0);
    }
    
    private int dfs(int[][] memo, int loc, int cur){
        if(loc == memo.length){
            return 0;
        }
        
        int res = 0;
        if(memo[loc][0] != -1 && (cur & memo[loc][0]) == 0){
            res = dfs(memo, loc + 1, cur | memo[loc][0]) + memo[loc][1];
        }
        
        res = Math.max(res, dfs(memo, loc + 1, cur));
        
        return res;
    }
}
