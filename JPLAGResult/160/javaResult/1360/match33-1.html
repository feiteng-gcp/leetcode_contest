class Solution {
   public int maxLength(List<String> arr) {
        return recursion(arr, 0, new boolean[26], 0);
    }
    
    private int recursion(List<String> arr, int idx, boolean[] c, int cur) {
    	if(idx == arr.size()) return cur;
    	boolean[] g = gen(arr.get(idx));
    	
    	int ans = cur;
    	ans = Math.max(ans, recursion(arr, idx+1, c, cur));
    	
    	if(check(c, g, arr.get(idx))) {
    		boolean[] todel = new boolean[26];
    		for(int i = 0; i < 26; i++) {
    			if(!c[i] && g[i]) {
    				todel[i] = true;
    				c[i] = true;
    			}
    		}
    		ans = Math.max(ans, recursion(arr, idx+1, c, cur + arr.get(idx).length()));
    		
    		for(int i = 0; i < 26; i++) {
    			if(todel[i]) {
    				c[i] = false;
    			}
    		}
    	}
    	return ans;
    }
    
    private boolean[] gen(String s) {
    	boolean[] g = new boolean[26];
    	for(int i = 0; i < s.length(); i++) {
    		char cc = s.charAt(i);
    		if(g[cc-'a'] == true) return new boolean[0];
    		g[cc-'a'] = true;
    	}
    	return g;
    }
    
    private boolean check(boolean[] c, boolean[] g, String s) {
    	if(g.length == 0) return false;
    	for(int i = 0; i < 26; i++) {
    		if(g[i] && c[i]) return false;
    	}
    	return true;
    }
}
