class Solution {
    
    private int toInt(String arr) {
        int val = 0;
        int add ;
        for (int i=0;i<arr.length();i++) {
            add = (1<<(arr.charAt(i)-'a'));
            
if ((val & add) != 0) {
                return -1;
            }
            val |= add;
        }
        return val;
    }
    
    private int judge(List<String> arr, int[] list, int val) {
        int length = 0;
        int merge = 0;
        for (int i=0;i<list.length;i++) {
            if ((val&1) == 1 && list[i] == -1) {
                return 0;
            }
            if ((val&1) == 1 && (merge & list[i]) == 0) {
                merge |= list[i];
                length += arr.get(i).length();
            }
            val >>= 1;
        }
        return length;
    }
    
    public int maxLength(List<String> arr) {
        int n = arr.size();
        int[] list = new int[n];
        for (int i=0;i<n;i++) {
            list[i] = toInt(
arr.get(i));
        }
        int answer = 0;
        for (int i=(1<<n)-1;i>0;i--) {
            // System.out.println(i + " " + judge(arr, list, i));
            answer = Math.max(answer, judge(arr, list, i));
        }
        return answer;
    }
}
