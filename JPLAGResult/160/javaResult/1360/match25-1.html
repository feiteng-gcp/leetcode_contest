class Solution {
    public int maxLength(List<String> arr) {
        return maxLength(arr, 0, new int[26], 0);
    
}

    private int maxLength(List<String> arr, int i, int[] counts, int length) {
        if (i == arr.size()) return length;

        int ans = 
maxLength(arr, i + 1, counts, length);

        boolean available = true;
        String s = arr.get(i);
        for (char c : s.toCharArray()) {
            if (++counts[c - 'a'] == 2) available = false;
        }

        if (available) {
            ans = Math.max(ans, maxLength(arr, i + 1, counts, length + s.length()));
        }

        for (char c : s.toCharArray()) {
            --counts[c - 'a'];
        }
    
        return ans;
    }
}
