class Solution {
    int ans = 0;
    public int maxLength(List<String> arr) {
        int[] ck = new int[26];
        List<String> tmp = new ArrayList<>();
        for (String s : arr) {
            char[] ca = s.toCharArray();
            int[] cnt = new int[26];
            for (char c : ca) {
                cnt[c - 'a']++;
            }
            boolean flag = true;
            for (int i = 0; i < 26; i++) {
                if (cnt[i] > 1) {
                    flag = false;
                }
            }
            if (flag) {
                tmp.add(s);
            }
        }
        dfs(tmp, 0, ck, 0);
        return ans;
    }
    public void dfs(List<String> arr, int idx, int[] ck, int cur) {
        if (idx == arr.size()) {
            ans = Math.max(ans, cur);
            return;
        }
        for (int i = idx; i < arr.size(); i++) {
            String s = arr.get(i);
            char[] sa = s.toCharArray();
            boolean flag = true;
            for (char c : sa) {
                if (ck[c - 'a'] == 1) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                for (char c : sa) {
                    ck[c - 'a'] = 1;
                    cur++;
                }
                dfs(arr, i + 1, ck, cur);
                for (char c : sa) {
                    ck[c - 'a'] = 0;
                    cur--;
                }
            } else {
                dfs(arr, i + 1, ck, cur);
            }
        }
    }
}
