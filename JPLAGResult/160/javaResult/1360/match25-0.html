class Solution {
    private int max;
    public int maxLength(List<String> arr) {
        int[] table = new int[256];
        max = 0;
        helper(arr, 0, 0, table);
        return max;
    }
    private void helper(List<String> arr, int index, int curMax, int[] table) {
        if (index == arr.size()) {
            return;
        }
        // don't select indexTH string
        helper(arr, index+1, curMax, table);
        
        // select indexTH string
        boolean isUnique = true;
        String str = arr.get(index);
        for (int i = 0; i < str.length(); i++) {
            if (table[str.charAt(i)] != 0)
                isUnique = false;
            table[str.charAt(i)]++;
        }
        if (isUnique) {
            curMax += str.length();
            max = Math.max(max, curMax);
            helper(arr, index+1, curMax, table);
        }
        for (int i = 0; i < str.length(); i++)
            table[str.charAt(i)]--;
    }
}
