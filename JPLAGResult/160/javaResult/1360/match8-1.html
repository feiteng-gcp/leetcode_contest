class Solution {
    int maxLen = 0;
    
    public int maxLength(List<String> arr) {
        
backtrack(arr, 0, new HashSet<Character>());
        return maxLen;
    }
    
    public void backtrack(List<String> arr, int pos, HashSet<Character> set)
    {
        if(pos == arr.size())
        {
            maxLen = Math.max(maxLen, set.size());
            return;
        }
        
        boolean validToAdd = true;
        String s = arr.get(pos);
        
        HashSet<Character> tmpSet = new HashSet<Character>();
        
        for(int i=0;i<s.length();i++)
        {
            if(set.contains(s.charAt(i))
              || tmpSet.contains(s.charAt(i)))
            {
                validToAdd = false;
                break;
            }
            
            tmpSet.add(s.charAt(i));
        }
        
        if (validToAdd)
        {
            
for(int i=0;i<s.length();i++)
            {
                set.add(s.charAt(i));
            }
            
            backtrack(arr, pos+1, set);
            
            for(int i=0;i<s.length();i++)
            {
                set.remove(s.charAt(i));
            }
        }
        
        backtrack(arr, pos+1, set);
    }
}
