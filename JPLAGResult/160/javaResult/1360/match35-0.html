class Solution {
    public int maxLength(List<String> arr) {
        int n = arr.size();
        int[][] lens = new int[n][256];
        int ind = 0;
        for (String s : arr) {
            for (char c : s.toCharArray())
                lens[ind][c]++;
            ++ind;
        }
        boolean[][] tog = new boolean[n][n];
        for (int i = 0; i < n; ++i) {
            tog[i][i] = true;
            for (int j = i+1; j < n; ++j) {
                tog[i][j] = true;
                for (int k = (int)'a'; k <= 'z'; ++k) {
                    if (lens[i][k]+lens[j][k]>=2)
                        tog[i][j] = false;
                }
                tog[j][i] = tog[i][j];
                //System.out.println(i+" "+j+" "+tog[i][j]);
            }
        }
        int[] dp = new int[1<<n];
        int[] last = new int[1<<n];
        int ans = 0;
        ArrayDeque<Integer> q = new ArrayDeque<Integer>();
        for (int i = 0; i < n; ++i) {
            boolean add = true;
            for (int k = (int)'a'; k <= 'z'; ++k) {
                if (lens[i][k]>=2)
                    add = false;
            }
            if (add) {
                q.add(1<<i);
                dp[1<<i] = arr.get(i).length();
                last[1<<i] = i;
                ans = Math.max(ans, dp[1<<i]);
            }
        }
        while (!q.isEmpty()) {
            int u = q.removeFirst();
            int l = last[u];
            for (int i = l+1; i < n; ++i) {
                boolean add = true;
                for (int j = 0; j <= l; ++j) {
                    if (((1<<j)&u)>0&&!tog[i][j]) {
                        add = false;
                        break;
                    }
                }
                if (add) {
                    q.add(u+(1<<i));
                    dp[u+(1<<i)] = dp[u] + arr.get(i).length();
                    last[u+(1<<i)] = i;
                    ans = Math.max(ans, dp[u+(1<<i)]);
                }
            }
        }
        return ans;
    }
}
