class Solution {
    public int maxLength(List<String> arr) {
        int[] count = new int[arr.size()];
        int[] num = new int[arr.size()];
        for (int i = 0; i < count.length; i++) {
            count[i] = getCount(arr.get(i));
            num[i] = getNum(count[i]);
        
}
        return getAns(count, num, 0, 0);

    }
    public int getAns(int[] count, int[] num, int pos, int cur) {
        int res = 0;
        for (int i = pos; i < count.length; i++) {
            if ((cur & count[i]) == 0) {
                res = Math.max(getAns(count, num, i + 1, cur | count[i]) + num[i], res);
            }
        }
        return res;
    }
    public int getNum(int i) {
        int res = 0;
        while (i > 0) {
            res += (i & 1);
            i >>= 1;
        }
        return res;
    }
    public int getCount(String str) {
        int res = 0;
        for (char a : str.toCharArray()) {
            if ((res & (1 << (a - 'a'))) > 0) {
                return 0;
            }
            res |= (1 << (a - 'a'));
        }
        return res;
    }
}
