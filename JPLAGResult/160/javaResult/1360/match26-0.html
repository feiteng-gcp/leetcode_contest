class Solution {
    int max = 0;
    public int maxLength(List<String> arr) {
        helper(arr, 0, new HashSet<>());
        return max;
    }
    
    private void helper(List<String> arr, int idx, HashSet<Character> s) {
        if (idx == arr.size()) return;
        for (int i = idx; i < arr.size(); i++) {
            String cur = arr.get(i);
            boolean can = true;
            HashSet<Character> dup = new HashSet<>();
            for (char c : cur.toCharArray()) {
                if (s.contains(c) || dup.contains(c)) {
                    can = false;
                    break;
                }
                dup.add(c);
            }
            if (can) {
                for (char c : cur.toCharArray()) {
                    s.add(c);
                }
                max = Math.max(max, s.size());
                helper(arr, i + 1, s);
                for (char c : cur.toCharArray()) {
                    s.remove(c);
                }
            }
        }
    }
}
