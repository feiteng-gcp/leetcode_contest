class Solution {
    private int max = 0;
    
    public int maxLength(List<String> arr) {
        String[] arrs = new String[arr.size()];
        int i = 0;
        for (String s: arr) {
            arrs[i++] = s;
        }
        calc(arrs, 0, new int[255], 0);
        
return max;
    }
    
    private void calc(String[] arrs, int index, int[] cs, int num) {
        if (index == arrs.length) {
            if (num > max) {
                max = num;
            }
        }
        for (int i = index; i < arrs.length; i++) {
            char[] ss = arrs[i].toCharArray();
            boolean flag = false;
            int[] ttt = new int[255];
            
for (char c: ss) {
                if (cs[c] != 0 || ++ttt[c] > 1) {
                    flag = true;
                    break;
                }
            }
            if (!flag) {
                int nn = num;
                for (char c: ss) {
                    cs[c]++;
                    nn++;
                }
                calc(arrs, i + 1, cs, nn);
                for (char c: ss) {
                    cs[c]--;
                
}
            } else {
                calc(arrs, i + 1, cs, num);
            }
        }
    }
    
    
}
