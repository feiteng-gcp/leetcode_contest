class Solution {
     private int maxLen = 0;
    public int maxLength(List<String> arr) {
        int[] map = new int[26];
        dfs(0, arr, map, 0);
        return maxLen;
    }

    private void dfs(int start, List<String> arr, int[] map, int len) {
        maxLen = Math.max(maxLen, len);
        if (maxLen == 26) {
            return;
        }
        if (start >= arr.size())
            return;
        int[] newMap = new int[map.length];
        System.arraycopy(map, 0, newMap, 0, map.length);
        String cur = arr.get(start);
        boolean add = true;
        for (int  i = 0; i < cur.length(); i++) {
            int ch = cur.charAt(i) -'a';
            if (newMap[ch] > 0) {
                add = false;
                break;
            }
            newMap[ch] = 1;
        }
        if (add)
            dfs(start+1, arr, newMap, len+cur.length());
        dfs(start+1, arr, map, len);

    }
}
