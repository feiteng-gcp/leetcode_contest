class Solution {
    int max = 0;
        
    private int trans(String s){
        int[] count = new int[26];
        for(int i = 0; i < s.length(); i++){
            char c = s.charAt(i);
            count[c - 'a']++;
            if(count[c - 'a'] > 1) return -1;
        }
        
        int start = 1, res = 0, digits = 0;
        for(int i = 0; i < 26; i++){
            if(count[i] > 0){
                res += start;
                digits++;
            }
            start *= 2;
        }
        
        
        max = Math.max(max, digits);
        
return res;
    }
    
    private void dfs(List<Integer> list, int idx, int v){
        if(idx == list.size()){
            max = Math.max(max, countDigit(v));
            return;
        }
        dfs(list, idx + 1, v);
        
        if((v & list.get(idx)) == 0)
            dfs(list, idx + 1, (v | list.get(idx)));
        
    }
    
    private int countDigit(int n){
        int res = 0, v = 1;
        for(int i = 0; i < 26; i++){
            if( (n & v) != 0){
                res++;
            }
            v *= 2;
        }
        return res;
    }
    
    
public int maxLength(List<String> arr) {
        
        Set<Integer> set = new HashSet<>();
        for(String s: arr){
            int v = trans(s);
            if(v < 0) continue;
            set.add(v);
        }
        
        List<Integer> list = new ArrayList<>(set);
        dfs(list, 0, 0);
        return max;
    }
}
