class Solution {
    private int dfs(int i, List<String> words, Set<Character> uniq) {
        if (i >= words.size()) {
            return uniq.size();
        }
        int answer = 0;
        
String word = words.get(i);
        boolean isUniq = true;
        Set<Character> wordUniq = new HashSet<>();
        for (char c: word.toCharArray()) {
            if (uniq.contains(c) || wordUniq.contains(c)) {
                isUniq = false;
                break;
            }
            wordUniq.add(c);
        }
        if (isUniq) {
            uniq.addAll(wordUniq);
            answer = Math.max(answer, dfs(i + 1, words, uniq));
            uniq.removeAll(wordUniq);
        }
        answer = Math.max(answer, dfs(i + 1, words, uniq));
        return answer;
    }
    public int maxLength(List<String> arr) {
        return dfs(0, arr, new HashSet<>());
    }
}
