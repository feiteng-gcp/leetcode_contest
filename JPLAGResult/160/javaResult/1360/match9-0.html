class Solution {
    private static int res = 0;
    public int maxLength(List<String> arr) {
        res = 0;
        Set<Character> set = new HashSet<>();
        dfs(0, arr, set);
        return res;
    }
    
    private void dfs(int level, List<String> arr, Set<Character> set) {
        if(level == arr.size()) {
            return;
        }
        String str = arr.get(level);
        int i = 0;
        for(;i < str.length(); i++) {
            if(set.contains(str.charAt(i))) {
                break;       
            }
            set.add(str.charAt(i));
        }
        if(i != str.length()) {
            for(int j = 0; j < i; j++) {
                set.remove(str.charAt(j));
            }
        } else {
            res = Math.max(res, set.size());
            dfs(level + 1, arr, set);
            for(i = 0; i < str.length(); i++) {
                set.remove(str.charAt(i));
            }
        }
        
        dfs(level + 1, arr, set);
    }
}
