class Solution {
    public int maxLength(List<String> arr) {
        int[] bits = new int[arr.size()];
        boolean[] invalid = new boolean[arr.size()];
        for(int i=0;i<arr.size();i++){
            int val = 0;
            for(char c : 
arr.get(i).toCharArray()) {
                int idx = c-'a';
                if(((val >> idx) & 1) == 1) {
                    invalid[i] = true;
                    break;
                }
                val |= (1<<idx);
            }
            bits[i] = val;
        
}

        return dfs(0,bits,invalid,0,0, arr);
    }
    public int dfs(int idx, int[] bits, boolean[] invalid, int mask, int total, List<String> arr){
        if(idx == bits.length) {
            return total;
        }
        if(invalid[idx]) return dfs(idx+1,bits,invalid,mask,total,arr);
        int best = 0;
        best = dfs(idx+1,bits,invalid, mask, total, arr);
        if((bits[idx] & mask) == 0)
        best = Math.max(best,
                        dfs(idx+1,bits,invalid,mask|bits[idx],total+arr.get(idx).length(),arr));
        return best;
    }
}
