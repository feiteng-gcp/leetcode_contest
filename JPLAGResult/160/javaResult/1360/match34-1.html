class Solution {

    static int res = 0;
    public int maxLength(List<String> strs) {
        res = 0;
        if(hasCommonStrs(strs))
            return 0;
        boolean[] visited = new boolean[strs.size()];
        for(int i=0;i<strs.size();i++) {
            dfs(strs, visited, 0, "");
        }
        return res;
    }

    static void dfs(List<String> strs, boolean[] visited , int cur, String tmp) {
        res = Math.max(res, tmp.length());
        for(int i=cur;i<strs.size();i++) {
            if(!visited[i]) {
                if(hasCommon(tmp, strs.get(i)))
                    continue;
                else {
                    visited[i] = true;
                    dfs(strs, visited, i+1, tmp + strs.get(i));
                    visited[i] = false;
                }
            
}
        }
    }

    static boolean hasCommon(String s1, String s2) {
        int[] cnt = new int[26];
        for(char c : s1.toCharArray()) {
            cnt[c-'a']++;
        }
        for(char c : s2.toCharArray()) {
            cnt[c-'a']++;
        }
        for(int i=0;i<cnt.length;i++) 
{
            if(cnt[i] > 1)
                return true;
        }
        return false;
    }

    static boolean hasCommonStrs(List<String> strs) {
        int[] cnt = new int[26];
        for(int i = 0; i < strs.size(); i++) {
            String s = strs.get(i);
            for(char c : s.toCharArray()) {
                if(++cnt[c-'a'] > 1) {
                    strs.remove(i);
                    break;
                }
            }
            for(int k = 0; k < 26; k++) cnt[k] = 0;
        }
        return false;
    }
}
