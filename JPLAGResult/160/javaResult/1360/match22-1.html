class Solution {
    public int maxLength(List<String> arr) {
        return dfs(0, arr, new boolean[26], 0);
    
}

    int dfs(int cur, List<String> arr, boolean[] seen, int l) {
        if (cur == arr.size()) {
            return l;
        }
        int ret = l;
        for (int i = cur; i < arr.size(); i++) {
            boolean can = true;
            for (char c : arr.get(i).toCharArray()) {
                if (seen[c - 'a']) {
                    can = false;
                    break;
                }
            }
            if (!can) {
                continue;
            }
            boolean[] s2 = new boolean[26];
            for (char c : arr.get(i).toCharArray()) {
                
if (s2[c - 'a']) {
                    can = false;
                    break;
                }
                s2[c - 'a'] = true;
            }
            if (!can) {
                continue;
            }
            for (char c : arr.get(i).toCharArray()) {
                seen[c - 'a'] = true;
            
}
            ret = Integer.max(ret, dfs(i + 1, arr, seen, l + arr.get(i).length()));
            for (char c : arr.get(i).toCharArray()) {
                seen[c - 'a'] = false;
            }
        }
        return ret;
    }
}
