class Solution {
    private int max = 0;

    public int maxLength(List<String> arr) {
        List<Integer> list = 
new ArrayList<>();
        for (String str : arr) {
            int temp = helper(str);
            if (temp > 0) list.add(temp);
        }
        // System.out.println(list);
        if(list.size() == 0) return 0;
        
rec(0, 0, list);
        return max;
    }

    private int helper(String str) {
        int res = 0;
        for (char ch: str.toCharArray()) {
            int temp = 1 << (ch - 'a');
            if ((res & temp) > 0) return 0;
            res |= temp;
        }
        return res;
    
}

    private int count(int bit) {
        int c = 0;
        while (bit > 0) {
            bit &= (bit - 1);
            c++;
        }
        return c;
    }

    private void rec(int bit, int idx, List<Integer> list) {
        if (idx == list.size()) {
            max = Math.max(max, count(bit));
            return;
        }
        rec(bit, idx + 1, list);
        int cur = list.get(idx);
        if ((bit & cur) == 0) {
            rec(bit | cur, idx + 1, list);
        }
    }
}
