class Solution {
    List<Integer> lx;
    int maxL = 0;
    public int maxLength(List<String> arr) {
        lx = new ArrayList<>();
        for(String x : arr) {
            int t = helper(x);
            if(t != -1) lx.add(t);
        }
        dfs(0, 0);
        return maxL;
    }
    int helper(String x) {
        int t = 0;
        for(char c : x.toCharArray()) {
            if((t & (1 << (c - 'a'))) != 0) return -1;
            t |= (1 << (c - 'a'));
        }
        return t;
    }
    void dfs(int id, int cur) {
        if(id == lx.size()) return;
        int t = lx.get(id);
        if((t & cur) == 0) {
            t |= cur;
            int cnt = count(t);
            if(cnt > maxL) maxL = cnt;
            dfs(id + 1, t);
        }
        dfs(id + 1, cur);
    }
    int count(int t) {
        int res = 0;
        while(t != 0) {
            ++res;
            t &= t - 1;
        }
        return res;
    }
}
