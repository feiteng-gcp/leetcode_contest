class Solution {
    int maxLen;
    public int maxLength(List<String> arr) {
        maxLen = 0;
        
        int[] bitVal = new int[arr.size()];
        
boolean[] invalid = new boolean[arr.size()];
        int[] count = new int[arr.size()];
        for(int i=0;i<arr.size();i++){
            
for(char c: arr.get(i).toCharArray()){
                int bit = 1<<(c-'a');
                if((bitVal[i]&bit)!=0){
                    //System.out.println("DEBUG: "+ arr.get(i));
                    invalid[i] = true;
                    break;
                } else{
                    bitVal[i] = bitVal[i]|bit;
                }
            }
            count[i] = countSetBits(bitVal[i]);
        }
        
         helper(bitVal, invalid, count,0, 0, 0);
        
        return maxLen;
    
}
    private void helper(int[] bitVal, boolean[] invalid, int[] count, int start, int curLen, int curVal){
        maxLen = Math.max(curLen, maxLen);
        for(int i=start;i<bitVal.length;i++){
          if(invalid[i]==true){
              continue;
          }
          if((curVal^bitVal[i])==(curVal|bitVal[i])){
              helper(bitVal, invalid, count, i+1, curLen+count[i],curVal|bitVal[i]);
          }
        }
        return;
    }
    private int countSetBits(int n) 
    { 
        int count = 0; 
        while (n > 0) 
        { 
            count += n & 1; 
            n >>= 1; 
        } 
        return count; 
    } 
}
