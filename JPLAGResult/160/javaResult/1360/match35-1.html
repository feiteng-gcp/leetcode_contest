class Solution {
    public int maxLength(List<String> arr) {
        int n = arr.size();
        int[][] count = new int[n][26];
        int index = 0;
        for (String str : arr) {
            for (char ch : str.toCharArray())
                count[index][ch - 'a']++;
            index++;
        }
        int ans = 0;
        search: for (int i = 0; i < (1 << n); i++) {
            int[] sum = new int[26];
            for (int j = 0; j < n; j++)
                if (((1 << j) & i) != 0) {
                    for (int k = 0; k < 26; k++) {
                        sum[k] += count[j][k];
                        if (sum[k] > 1) continue search;
                    }
                }
            int psum = 0;
            for (int j = 0; j < 26; j++) psum += sum[j];
            ans = Math.max(ans, psum);
        }
        
        return ans;
    }
}
