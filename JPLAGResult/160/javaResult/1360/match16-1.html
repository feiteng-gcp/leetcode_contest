class Solution {
    int max = 0;
    int[] nums;
    public int maxLength(List<String> arr) {
        nums = new int[arr.size()];
        for (int i = 0; i < nums.length; i++) {
            int cur = 0;
            String str = arr.get(i);
            
for (int j = 0; j < str.length(); j++) {
                int mask = 1 << (str.charAt(j) - 'a');
                if ((cur & mask) != 0) {
                    cur = 0;
                    break;
                }
                cur |= mask;
            }
            nums[i] = cur;
        }
        
        // for (int i : nums) {
        //     System.out.print(Integer.toBinaryString(i) + ",");
        //     System.out.println();
        // }
        
dfs(arr, 0, 0, 0);
        return max;
    }
    
    private void dfs(List<String> arr, int idx, int cur, int len) {
        max = Math.max(max, len);
        if (idx == nums.length) return;
        if ((cur & nums[idx]) == 0 && nums[idx] != 0) {
            dfs(arr, idx + 1, cur | nums[idx], len + arr.get(idx).length());
        }
        
        dfs(arr, idx + 1, cur, len);
    }
}
