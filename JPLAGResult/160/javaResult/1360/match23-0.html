class Solution {
    public int maxLength(List<String> arr) {
        Set<Integer> cand = new HashSet<>();
        for (String a : arr) {
            int code = encode(a);
            if (code != 0) {
                cand.add(code);
            }
        }
        int[] res = new int[1];
         dfs(cand.toArray(new Integer[cand.size()]), 0, 0, res);
        return res[0];
    }
    
    void dfs(Integer[] arr, int index, int sum, int[] res) {
        if (res[0] == 26) return;
 
        int count = Integer.bitCount(sum);
        if (index == arr.length) {
            res[0] = Math.max(res[0], count);
            return;
        } 
       
        int cur = arr[index];
        if ((cur & sum) == 0) {
            dfs(arr, index + 1, sum | cur, res);
        }
        dfs(arr, index + 1, sum, res);
    }
    
    int encode(String word) {
        int[] count = new int[26];
        for (char c : word.toCharArray()) {
            if (++count[c - 'a'] > 1) return 0;
        }
        int res = 0;
        for (int i = 0; i < 26; i++) {
           if (count[i] != 0) {
               res += (1 << i);
           } 
        }
        return res;
    }
}
