class Solution {
    public int maxLength(List<String> arr) {
        int N=arr.size();
//        String dp[][]=new int[N+1][N+1];      
//         dp[0][0]="";
//         for(int i=1;i<=N;i++){
//             dp[i][0]=arr.get(i-1);

//         }
//         for(int i=1;i<=N;i++){
//             for(int j=1;j<=N;j++){
                
//             }
//         }
//         return dp[N][N];
        return recurse(arr, N, 0, "");
    }
    
    
    int recurse(List<String> arr, int N, int i, String S){
        if(i==N){
            return S.length();
        }
        
        int n=0;
        if(isValid(S, arr.get(i))){
            n=recurse(arr, N, i+1, S+arr.get(i));
        }
        int m=recurse(arr, N, i+1, S);
        
        return Math.max(m, n);
    }
    
    boolean isValid(String S1, String S2){
        int freq[]=new int[26];
        for(int i=0;i<S1.length();i++){
            freq[S1.charAt(i)-'a']++;
        }
        
        for(int i=0;i<S2.length();i++){
            if(freq[S2.charAt(i)-'a']>0){
                return false;
            }
            else{
                freq[S2.charAt(i)-'a']++;
            }
        }
        
        return true;
    }
}
