class Solution {
    
    List<List<Integer>> ans = new ArrayList<>();
    
    void findAns(int x) {
        if (x == 1) {
            List<Integer> first = new ArrayList<>();
            first.add(0);
            first.add(1);
            ans.add(first);
            return;
        }
        findAns(x-1);
        List<Integer> last = 
ans.get(x-2);
        List<Integer> cur = new ArrayList<>(last);
        for (int i=0; i<last.size(); i++) {
            cur.add(last.get(last.size()-i-1) + (1<<(x-1)));
        }
        ans.add(cur);
    }
    
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> ret = new ArrayList<>();
        findAns(n);
        List<Integer> last = ans.get(n-1);
        for (int i=0; i<last.size(); i++) {
            if (last.get(i) == start) {
                
for (int j=0; j<last.size(); j++) {
                    ret.add(last.get((j+i) % last.size()));
                }
                break;
            }
        }
        return ret;
    }
}
