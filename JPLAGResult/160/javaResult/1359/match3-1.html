// p[0] = start
// p[i] and p[i+1] differ by only one bit in their binary representation.
// p[0] and p[2^n -1] must also differ by only one bit in their binary representation.

class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> cur = new ArrayList<>(2);
        cur.add(0);
        cur.add(1);
        int curMulti = 2;
        
        for(int i=2;i<=n;i++){
            ArrayList<Integer> tmp = new ArrayList<>(curMulti*2);
            for(int j=0;j*2<cur.size();j++){
                tmp.add(cur.get(j*2));
                tmp.add(cur.get(j*2)+curMulti);
                tmp.add(cur.get(j*2+1)+curMulti);
                tmp.add(cur.get(j*2+1));
            }
            cur = tmp;
            curMulti*=2;
        
}
        
        int startIdx = -1;
        for(int i=0;i<cur.size();i++){
            if(cur.get(i)==start ){
                startIdx = i;
                break;
            }
        }
        
        List<Integer> ret = new ArrayList<>();
        for(int i=0;i<cur.size();i++){
            int idx = (i+startIdx)%
cur.size();
            ret.add(cur.get(idx));
        }
        
        return ret;
    }
}
