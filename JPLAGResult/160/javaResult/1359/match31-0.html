class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<String> A=new ArrayList<String>();
        A.add("0");  
        A.add("1");
        
        for(int i=2; i<(1<<n); i=i<<1){    
            for(int j=i-1; j>=0; j--){  
                A.add(A.get(j));  
            }
            for(int j=0; j<i; j++){  
                A.set(j, "0" + A.get(j));
            }
            for (int j=i; j<2*i; j++){  
                A.set(j, "1" + A.get(j));
            }
        }  
        
        int startIndex=0;
        List<Integer> decimal=new ArrayList();
        for(int i=0;i<A.size();i++){
            int d=Integer.parseInt(A.get(i), 2);
            decimal.add(d);
            if(d==start){
                startIndex=i;
            }
        }
        
        List<Integer> res=new ArrayList();
        for(int i=startIndex;i<A.size();i++){
            res.add(decimal.get(i));
        }
        for(int i=0;i<startIndex;i++){
            res.add(decimal.get(i));
        }
        
        return res;
    }  
}
