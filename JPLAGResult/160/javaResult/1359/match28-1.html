class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> list = new ArrayList<>();
        list.add(0);
        list.add(1);
        int round = 1;
        while (list.size() < (int)Math.pow(2, n)) {
            int size = list.size();
            for (int i = size - 1; i >= 0; i--) {
                int num = list.get(i);
                list.add(getNum(num, round));
            }
            round++;
        }
        
        List<Integer> res = new ArrayList<>();
        
int i = 0;
        for (; i < list.size(); i++) {
            if (list.get(i) == start) {
                break;
            }
        }
        
for (int j = i; j < list.size(); j++) {
            res.add(list.get(j));
        }
        for (int j = 0; j < i; j++) {
            res.add(list.get(j));
        }
        return res;
    }
    
    public int getNum(int num, int round) {
        return (1 << round) | num;
    }
}
