class Solution {
    int[] res;
    
public List<Integer> circularPermutation(int n, int start) {
        int size = 1 << n;
        int[] array = new int[size];
        for(int i = 0; i < size; i++){
            array[i] = i;
        }
        
        array[start] = 0;
        array[0] = start;
        
boolean[] use = new boolean[size];
        use[start] = true;
        
        dfs(array, 1, n, use);
        
        //System.out.println(Arrays.toString(res));
        List<Integer> out = new ArrayList<>(size);
        for(int i = 0; i < size; i++){
            out.add(res[i]);
        }
        return out;
    }
    
    private boolean dfs(int[] array, int loc, int n, boolean[] use){
        if(loc == array.length){
            //System.out.println(Arrays.toString(array));
            int out = array[loc - 1] ^ array[0];
            if ((out & (out - 1)) == 0){
                res = array;
                return true;
            }
        }
        
        for(int i = 0; i < n; i++){
            int diff = 1 << i;
            int next = array[loc - 1] ^ diff;
            if(!use[next]){
                
array[loc] = next;
                use[next] = true;
                if(dfs(array, loc + 1, n, use)){
                    return true;
                }
                use[next] = false;
            }
        }
        
        return false;
    }
}
