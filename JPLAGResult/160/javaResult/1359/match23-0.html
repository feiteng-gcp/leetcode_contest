class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> res = new ArrayList<>(), r = new ArrayList<>();
        int s = -1, id = 0, total = (int)Math.pow(2,n);
        for (int i = 0; i < total; ++i) {
            if(((i >> 1) ^ i) == start) s = id;
            res.add((i >> 1) ^ i);
            id++;
        }
        for (int i = s; i < total; ++i) {
            r.add(res.get(i));
        }
        for (int i = 0; i < s; ++i) {
            r.add(res.get(i));
        }
        return r;
    }
}
