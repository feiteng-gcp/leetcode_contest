class Solution {
     public List<Integer> circularPermutation(int n, int start) {
        int a[]=new int[1<<n];
        for (int i = 0; i <n ; i++) {

            int count=1<<(n-1-i),ind=0;
            for (int j = 0; j <count; j++) {
                if(j%2==0){
                    ind+=1<<i;
                    for (int k =0 ; k <1<<i ; k++) {
                        a[ind++]|=1<<i;
                       // ind++;
                    }
                }
                else{

                    for (int k =0 ; k <1<<i ; k++) {
                        a[ind++]|=1<<i;
                        //ind++;
                    }
                    ind+=1<<i;
                }
            }
        }
       
        ArrayList<Integer> ans=new ArrayList<>();
        for (int i = 0; i <a.length ; i++) {
       
            if(a[i]==start){
       
                for (int j = 0; j <a.length ; j++) {
                    ans.add(a[(i+j)%(1<<n)]);
                }
                break;
            }
        }
        return ans;
    }
}
