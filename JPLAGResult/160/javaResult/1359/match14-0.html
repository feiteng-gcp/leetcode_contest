class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> graylist = grayCode(n);
        int startindex = 0;
        for (int i = 0; i < graylist.size(); i++) {
            if (graylist.get(i) == start) {
                startindex = i;
                break;
            }
        }
        ArrayList<Integer> list = new ArrayList<Integer>();
        for (int i = startindex; i < graylist.size(); i++) {
            list.add(graylist.get(i));
        }
        for (int i = 0; i < startindex; i++) {
            list.add(graylist.get(i));
        }
        return list;
    }
    
    public List<Integer> grayCode(int n) {
        if (n == 0) {
            Integer[] arr = {0};
            return Arrays.asList(arr);
        }
        int length = 1 << n;
        Integer[] arr = new Integer[length];
        List<Integer> arrp = grayCode(n - 1);
        for (int i = 0; i < length >> 1; i++){
            arr[i] = arrp.get(i);
            arr[length - i - 1] = arrp.get(i);
        }
        int p = 1 << (n-1);
        for(int i = length >> 1; i <length; i++)
            arr[i] = arr[i] | p;
        return Arrays.asList(arr);
    }
}
