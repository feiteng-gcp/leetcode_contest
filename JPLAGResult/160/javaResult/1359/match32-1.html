class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> res = new ArrayList<>();
        for (int i = 0; i < Math.pow(2, n); i++) {
            res.add((i >> 1) ^ i);
        }
        List<Integer> ans = new ArrayList<>();
        int idx = 0;
        while (res.get(idx) != start) {
            idx++;
        }
        
for (int i = idx; i< res.size(); i++) {
            ans.add(res.get(i));
        }
        for (int j = 0; j < idx; j++) {
            ans.add(res.get(j));
        }
        return ans;
    }
}
