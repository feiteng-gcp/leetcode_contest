class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        // List<Integer> ans = new ArrayList<>();
        // if (n == 1) {
        //     ans.add(0);
        //     return ans;
        // }
        int size = 1 << n;
        int[] sofar = new int[size];
        for (int i = 0; i < size; ++i) {
            sofar[i] = i;
        }
        swap(sofar, 0, start);
        boolean[] seen = new boolean[size];
        seen[start] = true;
        dfs(n, 1, seen, sofar);
        List<Integer> ans = new ArrayList<>();
        for (int v : sofar) {
            ans.add(v);
        }
        return ans;
    }
    
    public static boolean dfs(int n, int pos, boolean[] seen, int[] sofar) {
        if (pos >= (1 << n)) {
            return Integer.bitCount(sofar[sofar.length - 1] ^ sofar[0]) == 1;
        }
        int curr = sofar[pos - 1];
        for (int i = 0; i < n; ++i) {
            int next = curr ^ (1 << i);
            if (seen[next]) {
                continue;
            }
            sofar[pos] = next;
            seen[next] = true;
            if (dfs(n, pos + 1, seen, sofar)) {
                return true;
            }
            seen[next] = false;
        }
        // for (int i = pos; i < sofar.length; ++i) {
        //     int next = sofar[i];
        //     swap(sofar, pos, i);
        //     if (dfs(n, pos + 1, sofar)) {
        //         return true;
        //     }
        //     swap(sofar, pos, i);
        // }
        return false;
    }
    
    public static void swap(int[] arr, int i, int j) {
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }
}
