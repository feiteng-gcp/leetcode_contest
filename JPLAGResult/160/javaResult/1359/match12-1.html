	class S
olution {
	    public List<Integer> circularPermutation(int n, int start) {
	        int[] a = new int[1<<n];
	        for(int i = 0;i < 1<<n;i++){
	        	a[i] = i^i>>>1;
	        }

    		List<Integer> ret = new ArrayList<>();
	        for(int i = 0;i < 1<<n;i++){
	        	if(a[i] == start){
	        		for(int j = 0;j < 1<<n;j++){
	        			ret.add(a[(i+j)%(1<<n)]);
	        		}
	        		return ret;
	        	}
	        }
	        return null;
	    }
	}
