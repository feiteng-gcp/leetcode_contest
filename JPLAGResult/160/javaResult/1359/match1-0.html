class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> res = new ArrayList<Integer>();
        // 加入初始值0
        res.add(0);
        for(int i = 0; i < n; i++){
            // 每一轮的最高位
            int highestBit = 1 << i;
            int size = res.size();
            // 逆序添加上一轮里出现的数，不过开头加上这一轮的最高位
            for(int j = size - 1; j >= 0; j--){
                int num = res.get(j);
                num += highestBit;
                res.add(num);
            }
        }
        
        List<Integer> rr = new ArrayList<Integer>();
        
        for(int i=0;i<res.size();++i){
            if(res.get(i)==start){
                
                for(int j=0;j<res.size();++j){
                    int idx = (j+i)%res.size();
                    rr.add(res.get(idx));
                }
                return rr;
                
            }
        }
        
        return null;
    }
}
