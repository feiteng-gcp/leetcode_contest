class Solution {
    private int[] grayCode(int n)
	{
		int[] result = new int[(int) Math.pow(2, n)];
		for (int i = 0; i < Math.pow(2, n); i++)
		{
			result[i] = ((i >> 1) ^ i);

		}
		return result;
	}

	public 
List<Integer> circularPermutation(int n, int start)
	{
		List<Integer> result = new ArrayList<Integer>();
		int[] grayCode = grayCode(n);
		int index = -1;
		for (int i = 0; i < grayCode.length; i++)
		{
			if (grayCode[i] == start)
			{
				index = i;
				break;
			}
		}


		for (int i = index; i < grayCode.length; i++)
			result.add(grayCode[i]);
		for (int i = 0; i < index; i++)
			result.add(grayCode[i]);

		return result;
	}
}
