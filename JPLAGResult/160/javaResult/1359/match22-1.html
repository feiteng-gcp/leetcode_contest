class Solution {
    public List<Integer> circularPermutation(int n, int start) {
           List<Integer> res =  new ArrayList<>();
        List<Integer> code = grayCode(n);
        int N = code.size();
        int i = 0;
        for (; i <  N; i++) 
{
            if (code.get(i) == start) break;
        }
        for (int j = i; j < N; j++) 
{
            res.add(code.get(j));
        }
        for (int j = 0; j < i; j++) {
            res.add(code.get(j));
        }
        return res;
    }
    
    List<Integer> grayCode(int n) {
        int size = 1 << n;
        List<Integer> res = new ArrayList<>(size);
        for (int i = 0; i < size; i++) {
            res.add((i >> 1) ^ i);
        }
        return res;
    }
}
