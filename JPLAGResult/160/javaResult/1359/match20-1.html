class Solution {
    List<Integer> returnlist = new ArrayList<>();
    
    
public List<Integer> circularPermutation(int n, int start) {
        List<Integer> cur = new ArrayList<>();
        HashSet<Integer> set = new HashSet<>();
        set.add(start);
        cur.add(start);
        
        permutate(cur, set, n);
        
        return returnlist;
    }
    
    public boolean permutate(List<Integer> cur, HashSet<Integer> set, int n)
    {
        if(cur.size() == Math.pow(2, n))
        {
            int first = cur.get(0);
            int last = cur.get(cur.size()-1);
                
            int count = 0; 

            // since, the numbers are less than 2^31 
            // run the loop from '0' to '31' only 
            for (int i = 0; i < n; i++) { 

                // right shift both the numbers by 'i' and 
                // check if the bit at the 0th position is different 
                if (((first >> i) & 1) != ((last >> i) & 1)) { 
                    count++; 
                } 
            }
            
            if(count == 1)
            {
                returnlist = cur;
            }
            
            return count == 1;
        }
        
        for(int i=0;i<n;i++)
        {
            int newval = cur.get(cur.size()-1)^(1<<i);
            if(!set.contains(newval))
            {
                set.add(newval);
                cur.add(newval);
                if (permutate(cur, set, n))
                {
                    return true;
                
}
                
                set.remove(newval);
                cur.remove(cur.size()-1);
            }
        }
        
        return false;
    }
    
}
