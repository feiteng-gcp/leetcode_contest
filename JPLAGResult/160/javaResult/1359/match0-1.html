class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> grayCodeList = grayCode(n);
        int size = grayCodeList.size();
        int startIndex = grayCodeList.indexOf(start);
        List<Integer> permutation = new ArrayList<Integer>();
        for (int i = startIndex; i < size; i++)
            permutation.add(grayCodeList.get(i));
        for (int i = 0; i < startIndex; i++)
            permutation.add(grayCodeList.get(i));
        return permutation;
    }

    public List<Integer> grayCode(int n) {
        if (n == 0) {  
            List<Integer> result = new ArrayList<Integer>();
            result.add(0);
            return result;
        }
        List<Integer> previous = grayCode(n - 1);
        int addNumber = 1 << (n - 1);
        List<Integer> result = new ArrayList<Integer>(previous);
        int size = previous.size();
        for (int i = size - 1; i >= 0; i--)
            result.add(addNumber + previous.get(i));
        return result;
    }
}
