class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        if (n == 1) {
            List<Integer> list = new ArrayList<>();
            list.add(0);
            list.add(1);
            return rotate(list, start);
        }
        
List<Integer> list = circularPermutation(n-1, 0);
        List<Integer> solution = new ArrayList<>();
        for (int i = 0; i < list.size(); i++) {
            solution.add(list.get(i));
        }
        
for (int i = 0; i < list.size(); i++) {
            solution.add(list.get(list.size() - i - 1) + (1 << (n-1)));
        }
        return 
rotate(solution, start);
    }
    
    private List<Integer> rotate(List<Integer> list, int start) {
        List<Integer> l = new ArrayList<>();
        int ind = list.indexOf(start);
        for (int i = 0; i < list.size(); i++) {
            l.add(list.get((ind + i) % list.size()));
        }
        return l;
    }
}
