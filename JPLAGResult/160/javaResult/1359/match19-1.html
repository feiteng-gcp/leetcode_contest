class Solution {
    
    public boolean dfs(int n, int start, List<Integer> ans, HashSet<Integer> seen) {
        if (ans.size() == Math.pow(2, n))
            return true;
        
        
for (int i = 0; i < n; ++i) {
            int nxt = start ^ (1 << i);
            if (!seen.contains(nxt)) {
                seen.add(nxt);
                ans.add(nxt);
                if (dfs(n, nxt, ans, seen))
                    return true;
                ans.remove(ans.size() - 1);
                seen.remove(nxt);
            }
        }
        
        return false;
    }
    
    
public List<Integer> circularPermutation(int n, int start) {
        HashSet<Integer> seen = new HashSet<>();
        List<Integer> ans = new LinkedList<>();
        ans.add(start);
        seen.add(start);
        
        dfs(n, start, ans, seen);
        
        return ans;
    }
}
