class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> tmp = new ArrayList<>();
        int s = grayCode(n, start, tmp);
        List<Integer> res = new ArrayList<>();
        int max = (int)Math.pow(2, n);
        for (int i = 0; i < max; ++i) {
            res.add(tmp.get((s + i) % max));
        }
        return res;
    }

    public int grayCode(int n, int start, List<Integer> res) {
        int t = 0;
        for (int i = 0; i < Math.pow(2, n); i++) {
            res.add((i >> 1) ^ i);
            if (((i >> 1) ^ i) == start) {
                t = i;
            }
        }
        return t;
    }
}
