class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> temp = util(n);
        int startIndex = 0;
        while (true) {
            if (start == temp.get(startIndex)) {
                break;
            }
            startIndex++;
        }
        List<Integer> ret = new ArrayList<>(temp.size());
        for (int i = 0; i < temp.size(); i++) {
            ret.add(
temp.get((startIndex + i) % temp.size()));
        }

        return ret;
    }

    List<Integer> util(int n) {
        if (n == 1) {
            List<Integer> ret = new ArrayList<>();
            ret.add(0);
            ret.add(1);
            return ret;
        }
        List<Integer> ret = util(n - 1);
        int L = ret.size();
        for (int i = L - 1; i >= 0; i--) 
{
            ret.add(ret.get(i) + L);
        }
        return ret;
    }
}
