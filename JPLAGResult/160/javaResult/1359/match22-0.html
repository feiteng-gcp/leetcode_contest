class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> result = new ArrayList<Integer>();
        if(n < 0) return result;
        if(n == 0) {
            result.add(0);
            return result;
        }
        List<Integer> tmp = circularPermutationRecursive(n);
        int i = 0;
        for(i = 0; i < tmp.size(); i++){
            if(tmp.get(i) == start) break;
        }
        for(int j = i; j < tmp.size(); j++){
            result.add(tmp.get(j));
        }
        for(int j = 0; j < i; j++){
            result.add(tmp.get(j));
        }
        return result;
    }
    
    public List<Integer> circularPermutationRecursive(int n) {
        List<Integer> result = new ArrayList<Integer>();
        if(n < 0) return result;
        if(n == 0) {
            result.add(0);
            return result;
        }
        List<Integer> tmp = circularPermutationRecursive(n - 1);
        for(int i = 0; i < tmp.size(); i++){
            result.add(tmp.get(i));
        }
        for(int i = tmp.size() - 1; i >= 0; i--){
            result.add(tmp.get(i) | (1 << (n - 1)));
        }
        return result;
    }
}
