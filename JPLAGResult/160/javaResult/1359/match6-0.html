class Solution {
    // gray code
    public List<Integer> circularPermutation(int n, int start) {
        int num = (int) Math.pow(2, n);
        List<Integer> preList = new ArrayList<>();
        List<Integer> curList = new ArrayList<>(num);
        boolean startVis = false;
        for (int i = 0; i < num; ++i) {
            int ans = i^(i>>1);
            if (ans == start) {
                startVis = true;
            }
            if (startVis) {
                curList.add(ans);
            } else {
                preList.add(ans);
            }
        }
        curList.addAll(preList);
        return curList;
    }
}
