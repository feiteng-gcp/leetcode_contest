class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        if (n == 1) {
            return Arrays.asList(start, start ^ 1);
        } else {
            int msb = start >>> (n - 1);
            List<Integer> temp = circularPermutation(n - 1, start - msb*(int)Math.pow(2, n - 1));
            List<Integer> rt = new ArrayList<>();
            for(int i = 0; i < temp.size(); i++) {
              rt.add(msb * (int)Math.pow(2,n-1) + temp.get(i));
            }
            for(int i = temp.size() - 1; i >= 0; i--) {
              rt.add((msb ^ 1) * (int)Math.pow(2,n-1) + temp.get(i));
            }
            return rt;
        }
    }
}
