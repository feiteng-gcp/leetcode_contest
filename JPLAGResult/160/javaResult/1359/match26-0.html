class Solution {
    List<Integer> re = new ArrayList<>(), list = new ArrayList<>();
    int n;
    Set<Integer> set = new HashSet<>(), vis = new HashSet<>();

    public List<Integer> circularPermutation(int n, int start) {
   
        // differ by 1 bit -> xor = 1
        // try dfs?
        this.n = n;
        list.add(start);
        set.add(start);
        dfs(start, 1);
        return re;
    }
    
    boolean dfs(int cur, int count)
    {
        if(re.size() > 0) return true;
        if(count == 1 << n) re = new ArrayList<>(list);
        for(int i = 0; i < n; i++) {
            // swap bit out
            int next = cur ^ (1 << i);
            if(set.contains(next)) continue;
            set.add(next);
            list.add(next);
            if(dfs(next, count + 1)) return true;
            set.remove(next);
            list.remove(list.size() - 1);
        }
        return false;
    }
}
