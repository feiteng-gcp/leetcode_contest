class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> temp = new ArrayList<>();
        temp.add(0);
        for(int i = 0; i < n; i++) {
            int len = temp.size(), shift = 1 << i;
            for(int j = len-1; j >= 0; j--) {
                temp.add(temp.get(j) + shift);
            }
        }
        int size = temp.size();
        List<Integer> ans = new ArrayList<>();
        boolean flag = false;
        for(int i = 0, j = 0; i < 2 * size; i++) {
            if(temp.get(i%size) == start) flag = true;
            if(flag && j < size) {
                ans.add(temp.get(i%size));
                j++;
            }
        }
        return ans;
    }
}
