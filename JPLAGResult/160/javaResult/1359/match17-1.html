class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        if(n<1) return new ArrayList<>();
        List<Integer> ini = recur(n);
        int startIndex = 0;
        for(int i=0;i<ini.size();i++){
            if(ini.get(i)==start) startIndex = i;
        }
        for(int i=0;i<startIndex;i++){
            ini.add(ini.get(i));
        }        
        for(int i=0;i<startIndex;i++){
            
ini.remove(0);
        }                
        
        return ini;
    }
    public List<Integer> recur(int n){
        if(n==1){
            List<Integer> res = new ArrayList<>();
            res.add(0);
            res.add(1);
            return res;
        }
        List<Integer> res = recur(n-1); 
        
for(int i=res.size()-1;i>=0;i--){
            res.add(res.get(i)+(1<<n-1));
        }
        return res;
    }
}
