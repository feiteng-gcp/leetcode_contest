class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> greyCode = new ArrayList<>();
        greyCode.add(0);
        int i = 0;
        int idx = -1;
        if(start == 0) {
            idx = 0;
        }
        while(i < n) {
            List<Integer> temp = new ArrayList<>(greyCode);
            for(int j = greyCode.size() - 1; j >= 0; j--) {
                int val = greyCode.get(j) ^ (1 << i);
                if(start == val) {
                    idx = temp.size();
                }
                temp.add(greyCode.get(j) ^ (1 << i));
            }
            greyCode = temp;
            i++;
        }
        
        if(idx == 0) {
            return greyCode;
        }
        
        List<Integer> res = new ArrayList<>();
        
for(int j = idx; j < greyCode.size(); j++) {
            res.add(greyCode.get(j));
        }
        for(int j = 0; j < idx; j++) {
            res.add(greyCode.get(j));
        }
        return res;
    }
}
