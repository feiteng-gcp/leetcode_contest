class Solution {
    public List<Integer> circularPermutation(int n, int start) {
        List<Integer> list = new ArrayList<>();
        list.add(0);
        int indexStart = 0;
        for(int i = 0,m = 1;i<n;i++){
            int len = list.size();
            for(int j = len - 1;j>=0;j--){
                int curr = list.get(j)+m;
                if(curr == start){
                    indexStart = list.size();
                }
                list.add(curr);
            }
            m = m*2;
        }
        List<Integer> ans = new LinkedList<>();
        for(int i = indexStart;i<list.size();i++){
            ans.add(list.get(i));
        }
        for(int i = 0;i<indexStart;i++){
            ans.add(list.get(i));
        }
        return ans;
    }
}
