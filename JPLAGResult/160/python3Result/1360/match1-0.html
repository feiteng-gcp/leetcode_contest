class Solution:
    def maxLength(self, arr: List[str]) -> int:
        bits = []
        for s in arr:
            nodup = set(s)
            if len(nodup) != len(s):
                continue
            bits.append(nodup)
            
        #print (bits)
            
        n = len(bits)
        self.ans = 0
        
        def dfs(index, used):
            if index == n:
                return
            
            
            dfs(index+1, used)
            if not (bits[index] & used):
                tmp = used | bits[index]
                self.ans = max(self.ans, len(tmp))
                dfs(index+1, tmp)
        
        dfs(0, set())
        return self.ans
