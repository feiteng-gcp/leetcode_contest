class Solution:
    def maxLength(self, arr: List[str]) -> int:
        
def uni(s):
            cur = set()
            for c in s:
                if c in cur: return False
                cur.add(c)
            return True

        dp = set()
        for s in arr:
            if not uni(s): continue
            new = set()
            new.add(s)
            for z in dp:
                if uni(s + z): new.add(s + z)
            dp = new | dp
        
        ret = 0
        for s in dp:
            ret = max(len(s), ret)
        
        return ret
