class Solution:
    def possibleStrings(self, arr, output = []):
        if len(arr) == 1:
            return ['', arr[0]]
        output = self.possibleStrings(arr[1:], output)
        l = len(output)
        for i in range(l):
            output.append(arr[0] + output[i])
        return output
            
    def maxLength(self, arr: List[str]) -> int:
        poss = self.possibleStrings(arr)
        max_len = 0
        for p in poss:
            if len(set(p)) == len(p):
                max_len = max(max_len, len(p))
        return max_len
