class Solution:
    def maxLength(self, arr: List[str]) -> int:
        pre = []
        for s in arr:
            t = 0
            f = True
            for c in s:
                g = 1 << (ord(c) - ord('a'))
                if t & g:
                    f = False
                    break
                else:
                    t = t | g
            if f:
                pre.append(t)

        self.ans = 0
        import functools
        @functools.lru_cache(None)
        def dfs(i, p):
            self.ans = max(self.ans, str(bin(p)).count("1"))
            for j in range(i, len(pre)):
                temp = p & pre[j]
                if temp == 0:
                    dfs(j + 1, p | pre[j])

        dfs(0, 0)
        return self.ans
