class Solution:
    def maxLength(self, arr: List[str]) -> int:
        tmp = []
        for a in arr:
            if len(set(a)) == len(a): tmp.append(a)
        arr = tmp
        n = len(arr)
        ma = 0
        def bt(idx, visited):
            nonlocal ma
            if idx == n:
                ma = max(ma, len(visited))
                return
            bt(idx + 1, visited)
            if all(c not in visited for c in arr[idx]):
                for c in arr[idx]: visited.add(c)
                bt(idx + 1, visited)
                for c in arr[idx]: 
                    if c in visited: visited.remove(c)
                        
        
        
        bt(0, set())
        return ma
