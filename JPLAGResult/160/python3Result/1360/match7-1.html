class Solution:
    def maxLength(self, arr: List[str]) -> int:
        sets = []
        for s in arr:
            st = set(s)
            if len(st) == len(s):
                sets.append(st)
        self.ans = 0
        def dfs(start, cur):
            self.ans = max(self.ans, len(cur))
            for i in range(start, len(sets)):
                if len(cur - sets[i]) == len(cur):                    
                    cur |= sets[i]                    
                    dfs(i + 1, cur)
                    cur -= sets[i]
        dfs(0, set())
        return self.ans
