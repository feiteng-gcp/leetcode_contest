class Solution:
    def maxLength(self, arr: List[str]) -> int:
        nums = [self.to_bits(s) for s in arr if self.ok(s)]
        if not nums:
            return 0
        legal = [nums[0]]
        for idx in range(1, len(nums)):
            legal.append(nums[idx])
            for num in legal:
                if num & nums[idx] == 0:
                    legal.append(num | nums[idx])
        return max([self.get_len(l) for l in legal])
        
    def ok(self, s):
        return len(s) == len(set(list(s)))
    
    
def to_bits(self, s):
        ans = 0
        for c in s:
            ans |= 1 << (ord(c) - ord('a'))
        return ans
    
    def get_len(self, a):
        ans = 0
        while a > 0:
            a = a & (a - 1)
            ans += 1
        return ans
