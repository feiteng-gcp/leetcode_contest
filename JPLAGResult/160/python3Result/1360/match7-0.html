class Solution:
    def maxLength(self, arr: List[str]) -> int:
        sets = []
        for s in arr:
            set_s = set(s)
            if len(set_s) == len(s):
                sets.append(set_s)
                
        overlaps = []
        for i in range(len(sets)):
            overlaps.append([])
            for j in range(i+1, len(sets)):
                if len(sets[i].union(sets[j])) == len(sets[i]) + len(sets[j]):
                    overlaps[i].append(j)
        
        solution = 0
        for i in range(len(sets)):
            solution = max(solution, self.recurse(sets, overlaps, len(sets[i]), overlaps[i]))
        return solution
        
    def recurse(self, sets, overlaps, size, options):
        highest = size
        for o in options:
            new_options = list(set(options).intersection(set(overlaps[o])))
            highest = max(highest, self.recurse(sets, overlaps, size+len(sets[o]), new_options))
        return highest
