import collections
import functools

class Solution:
    def maxLength(self, arr: List[str]) -> int:
        new_arr = []
        for a in arr:
            if len(set(a)) == len(a):
                new_arr.append(a)
        arr = new_arr
        
        
        D = {}
        E = collections.defaultdict(list)
        for i, a in enumerate(arr):
            D[i] = set(a)
            for letter in a:
                E[letter].append(i)
        
        @functools.lru_cache()
        def dp(arr, indices):
            if len(indices) == 0:
                return 0
            if len(indices) == 1:
                return len(arr[indices[0]])
            A = dp(arr, indices[:-1])
            letters_in_last = D[indices[-1]]
            conflicting = [indices[:-1]]
            for l in letters_in_last:
                conflicting.extend(E[l])
            new_indices = []
            for ind in indices:
                if ind not in conflicting:
                    new_indices.append(ind)
            B = dp(arr, tuple(new_indices)) + len(arr[indices[-1]])
            return max(A, B)
            
        return dp(tuple(arr), tuple(range(len(arr))))
