class Solution:
    def filt(self, word):
        s = set()
        for i in word:
            if i in s:
                return False
            s.add(i)
        return True
    def dfs(self, arr, curResult, result, curIdx):
        if curIdx == len(arr):
            result.append(curResult)
            return;
        else:
            self.dfs(arr, curResult+arr[curIdx], result, curIdx+1)
            self.dfs(arr, curResult, result, curIdx+1)
    def maxLength(self, arr: List[str]) -> int:
        filtarr = []
        for i in range(len(arr)):
            if self.filt(arr[i]):
                filtarr.append(arr[i])
        if len(filtarr) == 0:
            return 0
        elif len(filtarr) == 1:
            return len(filtarr[0])
        else:
            result = []
            
            curIdx = 0
            curResult = ""
            self.dfs(filtarr, curResult, result,curIdx)
            curLen = -1
            for i in result:
                if len(i) == len(set(i)):
                    curLen = max(curLen, len(i))
            return curLen
