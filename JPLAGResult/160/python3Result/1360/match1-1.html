class Solution:
    def maxLength(self, arr: List[str]) -> int:
        tmp = []
        for i in arr:
            seti = set(i)
            if len(seti) != len(i):
                continue
            tmp.append(seti)
        
        self.res = 0
        def dfs(lst, index, cur):
            self.res = max(len(cur), self.res)
            
            for i in range(index, len(lst)):
                if len(cur & lst[i]) == 0:
                    dfs(lst, i + 1, cur | lst[i])
        dfs(tmp, 0, set())
        return self.res
