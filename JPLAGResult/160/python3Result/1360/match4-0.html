class Solution:
    def maxLength(self, arr: List[str]) -> int:
        bits = {}
        for s in arr:
            sbit = 0
            skip = False
            for c in s:
                pos = (ord(c) - ord('a'))
                if (sbit & 1 << pos) > 0:
                    skip = True
                    break
                sbit |= 1 << pos
            if not skip:
                bits[sbit] = len(s)
        # brute force match
        ret = 0
        dp = {0: 0}
        for a in bits:
            dpkeys = list(dp.keys())
            for k in dpkeys:
                if a & k != 0:
                    continue
                nk = a | k # new key with a and k
                if nk not in dp or dp[nk] < dp[k] + bits[a]:
                    dp[nk] = dp[k] + bits[a]
                    ret = max(ret, dp[nk])
        return ret
