from collections import Counter
class Solution:
    def sToNum(self, s):
        ans = 0
        for c in s:
            ans |= 1<<(ord(c) - ord("a"))
        return ans
    def count(self, num):
        ans = 0
        while num:
            num -= num&(-num)
            ans += 1
        return ans
    def maxLength(self, A):
        A = [word for word in A if len(word) == len(set(word))]
        sA = [self.sToNum(word) for word in A]

        aSet = set([0])
        for word in sA:
            newS = set(aSet)
            for s in aSet:
                if not (s & word):
                    newS.add(s | word)
            aSet = newS
            # print (aSet)
        return max(self.count(word) for word in aSet)
