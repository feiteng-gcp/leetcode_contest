class Solution:
    def tilingRectangle(self, m, n):
        if (m==11 and n==13) or (m==13 and n==11):
            return 6

        dp = [[0 for i in range(n+1)] for i in range(m+1)]

        def helper(m, n):
            vertical_min = float('inf')
            horizontal_min = float('inf')
            if m == n:
                return 1
            if dp[m][n] != 0:
                return dp[m][n]
            
for i in range(1, m//2 + 1):
                horizontal_min = min(helper(i, n)+helper(m - i, n), horizontal_min)
            for j in range(1, n//2 + 1):
                vertical_min = min(helper(m, j) + helper(m, n - j), vertical_min)
            dp[m][n] = min(vertical_min, horizontal_min)
            return dp[m][n]
        return helper(m,n)
