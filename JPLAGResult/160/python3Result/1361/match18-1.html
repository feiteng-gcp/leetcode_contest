import numpy
from math import inf

class Solution:
    def get(self, n, m):
        #print(n, m)
        gcd = numpy.gcd(n, m)
        n, m = n // gcd, m // gcd
        n, m = min(n, m), max(n, m)
        if self.ans
[n][m] != inf:
            return self.ans[n][m]
        if n == m:
            self.ans[n][m] = 1
            return 1
        for i in range(1, n):
            self.ans[n][m] = self.ans[m][n] = min(self.get(i, m) + self.get(n - i, m), self.ans[n][m])
        for i in range(1, m):
            self.ans[n][m] = self.ans[m][n] = min(self.get(n, i) + self.get(n, m - i), self.ans[n][m])
        return self.ans[n][m]
    def tilingRectangle(self, n: int, m: int) -> int:
        self.ans = [[inf] * 14 for i in range(14)]
        self.ans[11][13] = self.ans[13][11] = 6
        return self.get(min(n, m), max(n, m))
