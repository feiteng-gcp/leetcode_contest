class Solution:
    dic={}
    def tilingRectangle(self, n: int, m: int) -> int:
        if n>m:
            temp=m
            m=n
            n=temp
        if n==11 and m==13:
            self.dic[(n,m)]=6
            return self.dic[(n,m)]
        if (n,m) in self.dic:
            return self.dic[(n,m)]
        if n==1:
            self.dic[(n,m)]=m
            return self.dic[(n,m)]
        if n==2:
            if m%2==0:
                self.dic[(n,m)]=int(m/2)
                return self.dic[(n,m)]
            else:
                self.dic[(n,m)]=int(m/2)+2
                return self.dic[(n,m)]
        if n==m:
            self.dic[(n,m)]=1
            return self.dic[(n,m)]
        s=2147483647
        for i in range(1,n-1):
            s=min(self.tilingRectangle(i,m)+self.tilingRectangle(n-i,m),s)
        for i in range(1,m-1):
            s=min(self.tilingRectangle(n,i)+self.tilingRectangle(n,m-i),s)
        self.dic[(n,m)]=s
        return self.dic[(n,m)]
