#import functools
class Solution:
    def tilingRectangle(self, n: int, m: int) -> int:
        
        dp = [[-1 for _ in range(14)] for _ in range(14)]
        #@functools.lru_cache(none)
        def compute(n, m):
            nonlocal dp
            if m > n:
                m, n = n, m
                
            
            if (dp[n][m] >= 0):
                return dp[n][m]
            if n == 13 and m == 11:
                return 6
            
            if n == m:
                ans = 1
            else:
                ans = n * m
                
for i in range(1, m):
                    ans = min(ans, compute(n, i) + compute(n, m - i))

                for i in range(1, n):
                    ans = min(ans, compute(i, m) + compute(n - i, m))
            
            dp[n][m] = ans
            return ans
        return compute(n, m)
