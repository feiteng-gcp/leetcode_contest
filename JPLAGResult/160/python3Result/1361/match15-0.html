class Solution:
    def tilingRectangle(self, n: int, m: int) -> int:
        
        def gcd(a, b):
            if b == 0:
                return a
            else:
                return gcd(b, a % b)
            
        def search(h, w):
            ans = 100000
            for i in range(h // 2, h):
                a1 = i
                a2 = w - a1
                a3 = h - a2
                a4 = h - a1
                if a2 > a1 or a3 > a2 or a3 <= 0:
                    continue
                cur = min(
                    f(a4, w - a3 - a4) + f(a2 - a3, a1 - a2),
                    f(a4, a1 - a4) + f(a3, a2 - a3),
                )
                ans = min(ans, 4 + cur)
                
            return ans
                    
                    
            
        def f(h, w):
            if h > w:
                return f(w, h)
            if h == w:
                return 1
            if h == 0:
                return 0
            if h == 1:
                return w
            if h == 2:
                ans = w // 2
                if w % 2 == 1:
                    ans += 2
                return ans
            if w // h >= 3:
                return (w // h - 1) + f(h, w % h + h)
            
            if w == 12 and h == 11:
                return 7
            if w == 13 and h == 12:
                return 7
            
            ans1 = 1 + f(h, w - h)
            
            ans2 = search(h, w)
            
            return min(ans1, ans2)

        g = gcd(n, m)
        return f(n // g, m // g)
