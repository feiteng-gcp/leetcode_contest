import re
class Solution:
    def tilingRectangle(self, n: int, m: int) -> int:
        mm = """
            (1,1) -> 1: split
2,1 -> 2 : horizontal split at 1
2,2 -> 1 : square
3,1 -> 3 : horizontal split at 1
3,2 -> 3 : horizontal split at 1
3,3 -> 1 : square
4,1 -> 4 : horizontal split at 1
4,2 -> 2 : scale 2
4,3 -> 4 : horizontal split at 1
4,4 -> 1 : square
5,1 -> 5 : horizontal split at 1
5,2 -> 4 : horizontal split at 1
5,3 -> 4 : horizontal split at 2
5,4 -> 5 : horizontal split at 1
5,5 -> 1 : square
6,1 -> 6 : horizontal split at 1
6,2 -> 3 : scale 2
6,3 -> 2 : scale 3
6,4 -> 3 : scale 2
6,5 -> 5 : vertical split at 2
6,6 -> 1 : square
7,1 -> 7 : horizontal split at 1
7,2 -> 5 : horizontal split at 1
7,3 -> 5 : horizontal split at 1
7,4 -> 5 : horizontal split at 3
7,5 -> 5 : horizontal split at 2
7,6 -> 5 : horizontal split at 3
7,7 -> 1 : square
8,1 -> 8 : horizontal split at 1
8,2 -> 4 : scale 2
8,3 -> 5 : horizontal split at 2
8,4 -> 2 : scale 4
8,5 -> 5 : horizontal split at 3
8,6 -> 4 : scale 2
8,7 -> 7 : vertical split at 3
8,8 -> 1 : square
9,1 -> 9 : horizontal split at 1
9,2 -> 6 : horizontal split at 1
9,3 -> 3 : scale 3
9,4 -> 6 : horizontal split at 1
9,5 -> 6 : horizontal split at 4
9,6 -> 3 : scale 3
9,7 -> 6 : horizontal split at 2
9,8 -> 7 : horizontal split at 4
9,9 -> 1 : square
10,1 -> 10 : horizontal split at 1
10,2 -> 5 : scale 2
10,3 -> 6 : horizontal split at 1
10,4 -> 4 : scale 2
10,5 -> 2 : scale 5
10,6 -> 4 : scale 2
10,7 -> 6 : horizontal split at 3
10,8 -> 5 : scale 2
10,9 -> 6 : vertical split at 4
10,10 -> 1 : square
11,1 -> 11 : horizontal split at 1
11,2 -> 7 : horizontal split at 1
11,3 -> 6 : horizontal split at 2
11,4 -> 6 : horizontal split at 3

11,6 -> 6 : horizontal split at 2
11,7 -> 6 : horizontal split at 4

11,9 -> 7 : horizontal split at 2
11,10 -> 6 : horizontal split at 5
11,11 -> 1 : square
12,1 -> 12 : horizontal split at 1

12,3 -> 4 : scale 3
12,4 -> 3 : scale 4
12,5 -> 6 : horizontal split at 2
12,6 -> 2 : scale 6
12,7 -> 6 : horizontal split at 5
12,8 -> 3 : scale 4
12,9 -> 4 : scale 3
12,10 -> 5 : scale 2
12,11 -> 7 : vertical split at 3
12,12 -> 1 : square
13,1 -> 13 : horizontal split at 1
13,2 -> 8 : horizontal split at 1
13,3 -> 7 : horizontal split at 1
13,4 -> 7 : horizontal split at 1
13,5 -> 6 : horizontal split at 3
13,6 -> 6 : horizontal split at 3
13,7 -> 6 : horizontal split at 6
13,8 -> 6 : horizontal split at 5
13,9 -> 7 : horizontal split at 4
13,10 -> 7 : horizontal split at 3
13,11 -> 6 : basic solution 1:4,1:4,2:7,2:1,1:5,1:6,
13,12 -> 7 : horizontal split at 4
13,13 -> 1
        """
        dd = {}
        for val in mm.split("\n"):
            xx = re.split("\D+", val)
            try:
                xx = [ii for ii in xx if ii.isdigit()]
          #      print(xx)
                u,v,w = int(xx[0]), int(xx[1]), int(xx[2])
                dd[(u, v)] = dd[(v, u)] = w
            except:
                pass
        return dd[(m, n)]
