class Solution:
    def tilingRectangle(self, n: int, m: int) -> int:
        max_n = 14
        dp = [[10**9]*max_n for _ in range(max_n)]
        dp[0][0] = 0
        for i in range(1,14):
            dp[i][0] = 0
            dp[0][i] = 0
            dp[i][1] = i
            dp[1][i] = i
            dp[i][i] = 1
        for i in range(1,14):
            for j in range(i+1,14):
                for k in range(1,j + 1):
                    dp[i][j] = min(dp[i][j], dp[i][k] + dp[i][j-k])
                for k in range(1, i + 1):
                    dp[i][j] = min(dp[i][j], dp[k][j] + dp[i-k][j])
                dp[j][i] = dp[i][j]
        
        n,m = min(n,m), max(n,m)
        if n == 11 and m == 13:
            return 6
        return dp[n][m]
