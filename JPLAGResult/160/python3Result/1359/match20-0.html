class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        
        a = [0]
        for i in range(n):
            m = len(a)
            for j in reversed(range(m)):
                a.append((1 << i) + a[j])
        ans = [(x ^ start) for x in a]
        return ans
