class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        # gray code?
        def changeOne(n):
            if n == 0:
                return [0]
            elif n == 1:
                return [0, 1]
            else:
                subset = changeOne(n - 1)
                tmp = [x | (1 << (n - 1)) for x in subset]
                return subset + tmp[
::-1]

        ls = changeOne(n)
        idx = ls.index(start)
        ls = ls[idx:]+ ls[:idx]
        return ls
