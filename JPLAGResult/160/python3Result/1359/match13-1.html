class Solution:
    def circularPermutation(self, n, start):
        def gc(n):
            a =[0]
            for i in range(n):
                for j in range(len(a)-1, -1, -1):
                    a.append(1<<i|a[j
])
            return a

        l = gc(n)
        s = l.index(start)

        return l[s:]+l[:s]
