class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        if n == 1:
            return [0, 1] if start == 0 else [1, 0]
        else:
            base = ['0', '1']
            for i in range(n-1):
                base2 = base[:]
                
                base = ['0'+s for s in base]
                base2 = ['1'+s for s in base2]
                # concat
                base = base + base2[::-1]
            data = [int(s, 2) for s in base]
            idx = data.index(start)
            return data[idx:] + data[:idx]
