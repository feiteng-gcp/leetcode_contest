class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        res = [0,1]
        for i in range(1, n):
            x = (1<<i)
            res.extend(e+x for e in res[::-1])
        idx = 0
        for i in range(len(res)):
            if res[i] == start:
                idx = i
        return res[idx:]+res[:idx]
