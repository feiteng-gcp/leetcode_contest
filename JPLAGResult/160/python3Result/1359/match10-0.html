class Solution:
    def helper(self, n): 
        if n == 1: 
            return [0, 1]
        
        temp = self.helper(n-1)
        temp1 = [i + 2**(n-1) for i in reversed(temp)]

        return temp + temp1

    def circularPermutation(self, n: int, start: int) -> 'List[int]':
        res = self.helper(n)
        idx = res.index(start)

        return res[idx:] + res[:idx]
