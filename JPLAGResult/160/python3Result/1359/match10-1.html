class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        gray_arr = self.gray(n)
        idx = gray_arr.index(start)
        return gray_arr[idx:] + gray_arr[:idx]
    
    
def gray(self, n):
        if n == 1:
            return [0, 1]
        left = self.gray(n - 1)
        right = [num | (1 << (n - 1)) for num in reversed(left)]
        return left + right
