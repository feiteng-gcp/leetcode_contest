class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        def generate(n):
            if n==1:
                return [0,1]
            else:
                a=generate(n-1)
                b=a.copy()
                b.reverse()
                for i in range(len(b)):
                    b[i]+=2**(n-1)
                return a+b
        arr=generate(n)
        #print(arr)
        ind=arr.index(start)
        #print(ind)
        ans=arr[ind:]+arr[:ind]
        return ans
