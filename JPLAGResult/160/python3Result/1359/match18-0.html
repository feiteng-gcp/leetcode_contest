class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        def gray_code(n):
            if n == 0:
                return []
            if n == 1:
                return [0,1]
            gc = gray_code(n-1)
            l = len(gc)
            for i in range(l-1, -1, -1):
                gc.append(gc[i]+(2**(n-1)))
            return gc
        gc = gray_code(n)
        ind = gc.index(start)
        return gc[ind:] + gc[:ind]
