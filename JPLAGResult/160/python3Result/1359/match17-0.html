class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        
        def ccp(n: int) -> List[int]:
            if n == 1:
                return [0, 1]
            sub = ccp(n - 1)
            ans = []
            for num in sub:
                ans.append(num * 2)
            for num in reversed(sub):
                ans.append(num * 2 + 1)
            return ans
        ans = ccp(n)
        i = 0
        for i in range(len(ans)):
            if ans[i] == start:
                return ans[i:] + ans[:i]
