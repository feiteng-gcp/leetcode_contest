class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        ans = [0,1]
        ss = 0
        while len(ans) < 2**n:
            k = len(ans)
            for i in range(len(ans)-1,-1,-1):
                ans.append(ans[i]+k)
        for i in range(len(ans)):
            if ans[i] == start:
                return ans[i:]+ans[:i]
