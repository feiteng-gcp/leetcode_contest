class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        res = [0, 1]
        for i in range(n-1):
            # reverse res, add 2^(i+1) to each element, and append to the res
            res += [num + 2**(i+1) for num in res[::-1]]
        idx = res.index(start)
        return res[idx:] + res[:idx]
