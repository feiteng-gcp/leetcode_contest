class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        
        
        res = [0]
        
        
for i in range(n):
            res += [2**i + x for x in res][::-1]
            
        index = res.index(start)
            
        return res[index: ] + res[:index]
