class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        result = [0, 1]
        for i in range(1, n):
            next = [x + 2**i for x in result]
            result.extend(next[::-1])
        pos = result.index(start)
        return result[pos:] + result[:pos]
