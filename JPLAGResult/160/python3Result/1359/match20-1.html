class Solution:
    def circularPermutation(self, n: int, start: int):
        ret, k 
= [0], 1
        for i in range(1, n + 1):
            m = len(ret)
            for j in reversed(range(m)):
                ret.append(ret[j] + k)
            k *= 2
        for i in range(len(ret)):
            a, b, c = start, ret[i], 1
            while a > 0 or b > 0:
                if a % 2 != 0:
                    if b % 2 == 1:
                        ret[i] -= c
                    else:
                        ret[i] += c
                a //= 2
                b //= 2
                c *= 2
        return ret
