class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        result = self.getPermuation(n)
        #print(result)
        result = [int(k,2) for k in result]
        #print(result)
        index = result.index(start)
        return result[index:] + result[:index]
        
    def getPermuation(self, n):
        if n==0:
            return ["0"]
        elif n==1:
            return ["0", "1"]
        else:
            result = self.getPermuation(n-1)
            ans = []
            for k in result:
                ans.append('0'+k)
            for k in range(len(result)-1, -1, -1):
                ans.append('1'+result[k])
            return ans
