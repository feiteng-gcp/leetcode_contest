class Solution:
    def generateGrayCodeOfLength(self,n):
        if n==1:
            return [0,1]
        else:
            smaller = self.generateGrayCodeOfLength(n-1)
            first = smaller
            second = [x+2**(n-1) for x in smaller[::-1]]
            return first+second
        
    def circularPermutation(self, n: int, start: int) -> List[int]:
        code = self.generateGrayCodeOfLength(n)
        startInd = code.index(start)
        return code[startInd:] + code[0:startInd]
