class Solution:
    def circularPermutation(self, n: int, start: int) -> List[int]:
        base_cases = [0, 1]
        for i in range(n - 1):
            new_cases = []
            current_lead = 0
            for base_case in base_cases:    
                new_cases.append(base_case * 2 + current_lead)
                current_lead += 1
                current_lead %= 2
                new_cases.append(base_case * 2 + current_lead)
            base_cases = new_cases
            
        return base_cases[base_cases.index(start):] + base_cases[:base_cases.index(start)]
