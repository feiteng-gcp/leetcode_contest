/*
 * // This is the custom function interface.
 * // You should not implement it, or speculate about its implementation
 * class CustomFunction {
 * public:
 *     // Returns f(x, y) for any given positive integers x and y.
 *     // Note that f(x, y) is increasing with respect to both x and y.
 *     // i.e. f(x, y) < f(x + 1, y), f(x, y) < f(x, y + 1)
 *     int f(int x, int y);
 * };
 */

class Solution {
    set<vector<int>>ret;
public:
    vector<vector<int>> findSolution(CustomFunction& customfunction, int z) 
    {
        int x1=1, x2=1000;
        int y1=1, y2=1000;
        
        DFS(x1,x2,y1,y2,customfunction,z);
        
        vector<vector<int>>ans(ret.begin(),ret.end());
        
        return ans;
        
    }
    
    void DFS(int x1, int x2, int y1, int y2, CustomFunction& customfunction, int z)
    {        
        if (x1==x2 && y1==y2)
        {
            if (customfunction.f(x1,y1)==z)
                ret.insert({x1,y1});
            return;
        }
        if (x1>x2 || y1>y2)
            return;
        
        int mid1 = x1+(x2-x1)/2;
        int mid2 = y1+(y2-y1)/2;
        
        int t = customfunction.f(mid1, mid2);
        
        if (t==z)
        {
            ret.insert({mid1,mid2});
            DFS(mid1+1,x2,y1,mid2-1,customfunction,z);
            DFS(x1,mid1-1,mid2+1,y2,customfunction,z);
        }
        else if (t<z)
        {
            DFS(mid1+1,x2,y1,y2,customfunction,z);
            DFS(x1,x2,mid2+1,y2,customfunction,z);
        }
        else
        {
            DFS(x1,mid1-1,y1,y2,customfunction,z);
            DFS(x1,x2,y1,mid2-1,customfunction,z);
        }
            
    }
};
