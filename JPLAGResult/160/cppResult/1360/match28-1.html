class Solution {
public:
    int ans=0;
    unordered_set<char> tree;
    void solve(vector<string>& arr,int loc){
        if(loc>=arr.size()) return;
        
        for(int i=0;i<arr[loc].size();i++){
            if(tree.find(arr[loc][i])!=tree.end()){
                for(int j=i-1;j>=0;j--){
                    tree.erase(arr[loc][j]);
                }
                solve(arr,loc+1);
                return;
            }else{
                tree.insert(arr[loc][i]);
            }
        }
        ans=ans>tree.size()?ans:tree.size();
        //ans=max(ans,tree.size());
        solve(arr,loc+1);
        
        for(int i=0;i<arr[loc].size();i++){
            tree.erase(arr[loc][i]);
        }
        solve(arr,loc+1);
        return;
        
    }
    
    int maxLength(vector<string>& arr) {
        solve(arr,0);
        return ans;
    }
};
