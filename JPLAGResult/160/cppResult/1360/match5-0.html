class Solution {
public:
    int maxLength(vector<string>& arr) {

        int n = arr.size();
        int ans = 0;
        //n = 3
        //8ge, (0, 1, 2, 3, 4, 5, 6, 7)
        for (int i = 0; i < (1 << n); i++)
        {
            bool flag = false;
            int a[153];
            memset(a, 0, sizeof(a));

            string ret = "";

            for (int j = 0; j < n; j++)
            {
                if (i & (1 << j))
                {
                    ret += arr[j];
                    for (auto c : arr[j])
                    {
                        if (a[c] >= 1)
                        {
                            flag = true;
                            break;
                        }
                        a[c] += 1;
                    }
                }
            }
            
            if (flag == false)
            {
                ans = max(ans, (int)ret.size());
                //cout << "i:" << i << " ret:" << ret << endl;
            }

        }

        cout << "ans:" << ans << endl;

        return ans;
    }
};
