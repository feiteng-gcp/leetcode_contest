int num[16],len[16];
class Solution {
public:
    int maxLength(vector<string>& arr) {
        int n=arr.size();
        for(int i=0;i<n;i++)
        {
            num[i]=0;
            len[i]=arr[i].size();
            for(int j=0;j<len[i];j++)
            {
                if(num[i]&(1<<(arr[i][j]-'a')))
                {
                    num[i]=-1;
                    break;
                }
                num[i]=num[i]|(1<<(arr[i][j]-'a'));
            }
        }
        int ans=0;
        for(int i=0;i<(1<<n);i++)
        {
            int tot=0,ok=1;
            for(int j=0;j<n;j++)
            {
                if(!(i&(1<<j)))continue;
                tot=tot+len[j];
                if(num[j]==-1)
                {
                    ok=0;
                    break;
                }
                for(int k=j+1;k<n;k++)
                {
                    if(!(i&(1<<k)))continue;
                    if(num[k]==-1||num[j]&num[k])
                    {
                        ok=0;
                        break;
                    }
                }
                if(!ok)break;
            }
            if(ok)ans=max(ans,tot);
        }
        return ans;
    }
};
