#define SIZE 26
#define MAX(x,y) ((x)>(y)?(x):(y))

class Solution {
public:
    int maxLength(vector<string>& arr) {
        int N = arr.size();
        
        int ans = 0;
        for(int i=0;i<(1<<N);i++){
            bitset<SIZE> charsFound;
            charsFound.reset();
            
            bool possible = true;
            for(int j=0;j<N;j++){
                if(i & (1<<j)){
                    for(char ch:arr[j]){
                        if(charsFound[ch-'a']){
                            possible = false;
                            break;
                        }
                        charsFound[ch-'a'] = true;
                    }
                }    
            }
            
            if(possible){
                int tmp = 0;
                for(int j=0;j<SIZE;j++)
                    if(charsFound[j])
                        tmp++;
                ans = MAX(ans,tmp);
            }
        }
        
        return ans;
    }
};
