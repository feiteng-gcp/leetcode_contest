class Solution {
public:
    int maxLength(vector<string>& arr) {
        int n = arr.size();
        int res = 0;
        vector<string> pr;
        vector<int> bitt;
        for (int i = 0; i < n; i++) {
            int bts= 0;
            for (int j = 0; j < arr[i].length(); j++) {
                int c = arr[i][j] - 'a';
                if((bts&(1<<c)) != 0) goto outer;
                bts |= (1<<c);
            }
            // cout<<bts<<endl;
            res = max(res,__builtin_popcountll(bts));
            bitt.push_back(bts);
            outer:;
        }
        n = bitt.size();
        for (int i = 0; i < pow(2,n); i++) {
            int cur=0;
            bool go = true;
            for (int j = 0; j < n; j++) {
                if (i&(1<<j)){
                    if((cur&bitt[j])!=0) {
                        go=false;
                        break;
                    }
                    cur |= bitt[j];
                }
                if(!go) break;
            }
            if(go){
                int tt = __builtin_popcountll(cur);
                if(tt == 5) cout<<i<<endl;
                res=max(res,tt);
            }
            inn:;
        }
        return res;
    }
};
