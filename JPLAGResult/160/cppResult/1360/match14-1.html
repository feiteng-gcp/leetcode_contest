class Solution {
public:
    int maxLength(vector<string>& arr)
    {
        vector<int> masks;
        for (string& str : arr) {
            int mask = 0;
            for (char c : str) {
                int bit = 1 << (c - 'a');
                if (mask & bit) mask = -1;

                mask |= bit;
            }
            masks.push_back(mask);
        }
        int max = 0;
        for (int i = 0; i < (1 << arr.size()); i++) {
            int mask = 0;

            int len = 0;
            for (int j = 0; j < masks.size(); j++) {
                if ((i & (1 << j)) == 0) continue;
                if ((mask & masks[j]) != 0) {
                    mask = -1;
                    break;
                }
                mask |= masks[j];
                len += arr[j].size();
            }
            if (mask < 0) continue;
            max = std::max(max, len);
        }
        return max;
    }
};
