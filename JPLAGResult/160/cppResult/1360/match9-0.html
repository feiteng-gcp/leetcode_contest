class Solution {
public:
    int maxLength(vector<string>& arr) {
        vector<pair<int, int>> a;
        for (auto &x : arr) {
            int mask = 0;
            for (char c : x) {
                int bit = 1 << (c - 'a');
                if (mask & bit) {
                    mask = -1;
                    break;
                }
                mask |= 1 << (c - 'a');
            }
            if (mask >= 0)
                a.push_back({x.size(), mask});
        }
        int ans = 0;
        for (int i = 0; i < (1 << a.size()); ++i) {
            int len = 0, mask = 0;
            for (int j = 0; j < a.size(); ++j) {
                if (i & (1 << j)) {
                    if (mask & a[j].second) {
                        len = -1;
                        break;
                    }
                    len += a[j].first;
                    mask |= a[j].second;
                }
            }
            ans = max(ans, len);
        }
        return ans;
    }
};
