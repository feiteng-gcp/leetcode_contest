class Solution {
public:
    int maxLength(vector<string>& arr) {
        int ans = 0;
        vector<int> V;
        for (auto s : arr) {
            int t = 0;
            bool ok = true;
            for (auto c : s) 
                if (!ok) break;
                else {
                    int g = (1 << (c - 'a'));
                    if ((t & g) > 0) ok = false;
                    t |= g;
                }

            if (ok) V.push_back(t);
        }
        int N = V.size();
        for (int z=0;z<(1<<N);++z) {
            int t = 0;
            bool ok = true;
            for (int i=0;i<N;++i)
                if (z & (1<<i)) {
                    if ((t & V[i]) == 0) t |= V[i];
                    else ok = false;
                }
            if (ok) ans = max(ans, __builtin_popcount(t));
        }
        return ans;
    }
};
