class Solution {
public:
    int maxLength(vector<string>& arr) {

        for (auto &s : arr) {
            int n = s.length(), t = 0;
            for (auto &c : s) {
                int j = 1 << (c-'a');
                if ((t & j) != 0) {
                    n = 0;
                    break;
                }
                t |= j;
            }
            if (n) v.push_back({t, n});
        }
        n = v.size(), res = 0;
        dfs(0, 0, 0);
        return res;
    }
private:
    int n, res;
    vector<pair<int, int>> v;
    
    void dfs(int cur, int k, int sum) {
        if (sum > res) res = sum;
        if (cur == n) return;
        dfs(cur+1, k, sum);
        if (!(v[cur].first & k)) dfs(cur+1, v[cur].first | k, sum + v[cur].second);
    }
};
