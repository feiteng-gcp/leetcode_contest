class Solution {
public:
    int maxLength(vector<string>& a) {
        vector<int> b(a.size());
        for (int j = 0; j < a.size(); j++) {
            string& s = a[j];
            for (int i = 0; i < s.size(); i++) {
                int x = (1 << (s[i] - 'a'));
                if (b[j] & x) {
                    b[j] = -1;
                    break;
                }
                b[j] |= x;
            }
        }
        int res = 0;
        dfs(0, 0, 0, b, a, res);
        return res;
    }
    void dfs(int i, int x, int s, vector<int>& b, vector<string>& a, int& res) {
        if (i == b.size()) {
            res = max(res, s);
            return;
        }
        dfs(i + 1, x, s, b, a, res);
        if (b[i] != -1 && (x & b[i]) == 0)
            dfs(i + 1, x | b[i], s + a[i].size(), b, a, res);
    }
};
