class Solution {
public:
    int maxLength(vector<string>& arr) {
        int res=0;
        vector<int> b;
        for(auto& s: arr) {
            int t=0;
            for(char c: s) t|=1<<(c-'a');
            if (__builtin_popcount(t)==s.size())
                b.push_back(t);
        }
        int n=b.size();
        for(int i=0; i<(1<<n); ++i) {
            int L=0, x=0;
            for(int j=0; j<n; ++j) if ((i>>j)&1) {
                x|=b[j];
                L+=__builtin_popcount(b[j]);
            }
            if (__builtin_popcount(x)==L) res=max(res, L);
        }
        return res;
    }
};
