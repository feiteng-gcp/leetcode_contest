class Solution {
public:
    int maxLength(vector<string>& arr) {
        int n = arr.size();
        vector<int> v(n);
        for(int i=0;i<n;i++)
        {
            int x = 0;
            for(auto e:arr[i])
            {
                if(((x>>(e-'a') )& 1) != 1)
                    x += (1<<(e-'a'));
                else
                {
                    v[i] = -1;
                    break;
                }
            }
            if(v[i] != -1)
                v[i] = x;
           // printf("%d\n",v[i]);
        }
        
        int m = (1<<n);
        int ans = 0;
        for(int i=0;i<m;i++)
        //for(int i=3;i<4;i++)
        {
            int cnt = 0;
            for(int k=0;k<n;k++)
            {
                if(((i>>k)&1) == 1 && v[k] != -1)
                {
                    //printf("%d %d %d %d\n",i,k,cnt,v[k]);
                    if(__builtin_popcount(cnt) + __builtin_popcount(v[k]) == __builtin_popcount(cnt ^ v[k]) )
                    {
                        //printf("%d %d %d\n",__builtin_popcount(cnt),__builtin_popcount(v[k]),__builtin_popcount(cnt ^ v[k]));
                        cnt ^= v[k];
                        //printf("%d\n",cnt);
                    }
                    else
                    {
                        break;
                    }
                }
            }
            //printf("%d\n",cnt);
            ans = max(ans,__builtin_popcount(cnt));
        }
        return ans;
    }
};
