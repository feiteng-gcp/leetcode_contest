class Solution {
public:
    int best = 0;
    set<char> now;
    void dfs(int i, vector<string>& arr) {
        if(best == 26) return;
        for(int j = 0 ; j < arr[i].size() ; ++j) {
            if(now.count(arr[i][j])) {
                for(int k = j - 1 ; k >= 0 ; --k) now.erase(arr[i][k]);
                return;
            }
            else now.insert(arr[i][j]);
        }
        best = max(best, (int)now.size());
        if(best == 26) return;
        for(int j = i + 1 ; j < arr.size() ; ++j) {
            dfs(j, arr);
        }
        for(int j = 0 ; j < arr[i].size() ; ++j) now.erase(arr[i][j]);
    }
    int maxLength(vector<string>& arr) {
        sort(arr.begin(), arr.end());
        for(int i = 0 ; i < arr.size() ; ++i) {
            dfs(i, arr);
        }
        return best;
    }
};
