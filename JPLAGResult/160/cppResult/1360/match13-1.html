class Solution {
public:
    int help(vector<string>&arr, int &val){
        vector<int>cnt(26, 0);

        int N = arr.size();
        int res = 0;
        for(int i=0;i<N;++i){
            if(val&(1<<i)){
                for(auto &ch:arr[i]){
                    if(cnt[ch-'a']){
                        return 0;
                    }
                    ++cnt[ch-'a'];
                }
                res += arr[i].size();
            }
        }
        return res;
    }
    int maxLength(vector<string>& arr) {
        int res = 0;

        int N = arr.size();
        int num = (1<<N);
        for(int i=0;i<num;++i){
            res = max(res, help(arr, i));
        }
        return res;
    }
};
