class Solution {
public:
    int maxLength(vector<string>& arr) {
        vector<pair<int, int>> v; 
        for ( auto s:arr ){
            int d = encode(s);
            if ( d>0 )
                v.push_back( {s.size(),d} );
        }
        
        int res = 0;
        dfs( res, v, 0, 0, 0 );
        return res;
    }
    
    void dfs( int & res, vector<pair<int, int>> & nums, int idx, int cur, int state ){
        if ( idx>=nums.size() ){
            res = max( res, cur );
            return ;
        }
        
        for ( int i=idx; i<nums.size(); ++i ){
            int code = state & nums[i].second;
            if ( (code == 0 )){
                dfs( res, nums, i+1, cur+nums[i].first, state | nums[i].second );
            }
        }
        res = max( res, cur );
    }
    
    int encode( string s ){
        int code = 0;
        unordered_set<char> t;
        for ( auto c:s ){
            if ( t.count(c)>0 ) return 0;
            t.insert(c);
            int d = c-'a';
            if ( (code & (1<<d) )==1 ) return 0; //出现重复字符
            code = ( code | (1<<d) );
        }
        return code;
    }
};
