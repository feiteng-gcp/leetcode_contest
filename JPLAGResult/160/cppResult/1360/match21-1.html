class Solution {
public:
    
    inline int count(int x) {
        int cnt = 0;
        while (x) {
            if (x&1) ++cnt;
            x >>= 1;
        }
        return cnt;
    }
    

    void dfs(vector<int> &bit, int cur, int now, int &res)
    {
        if (cur >= bit.size()) {
            res = max(res, count(now));
            return;
        }
        if ((now&bit[cur]) == 0) dfs(bit, cur+1, now|bit[cur], res);
        dfs(bit, cur+1, now, res);
    }
    
    int maxLength(vector<string>& arr) {
        vector<int> bit(arr.size(), 0);
        for (int i = 0; i < arr.size(); ++i) {
            for (char c : arr[i]) {

                if (bit[i]&(1<<(c-'a'))) {
                    bit[i] = 0;
                    break;
                }
                bit[i] |= 1<<(c-'a');
            }
        }
        int res = 0;
        dfs(bit, 0, 0, res);
        return res;
    }
};
