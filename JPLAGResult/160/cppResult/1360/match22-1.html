class Solution {
public:
    int maxLength(vector<string>& a) {
        int n = a.size();

        vector<int> m(n), w(n);
        for (int i = 0; i < n; ++i) {
            w[i] = a[i].length();
            for (int j = 0; j < a[i].length(); ++j) {
                int cur = (1 << (a[i][j] - 'a'));
                if (m[i] & cur) {
                    w[i] = 0;
                    break;
                }
                m[i] |= cur;
            }
        }
       
        return dfs(0, n, m, w, 0);
    }
    
    int dfs(int idx, int n, vector<int> &m, vector<int> &w, int mask) {
        if (idx == n)
            return 0;
        
        int x = dfs(idx+1, n, m, w, mask);
        if (mask & m[idx])
            return x;
        else
            return max(x, dfs(idx+1, n, m, w, mask | m[idx]) + w[idx]);
    }
};
