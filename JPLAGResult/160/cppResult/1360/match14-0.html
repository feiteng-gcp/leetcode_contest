class Solution {
public:
    int maxLength(vector<string>& arr) {
        int n = arr.size();
        vector<int> flag(n, 0);
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < arr[i].size(); j++)
            {
                if (flag[i] & (1 << arr[i][j] - 'a'))
                {
                    flag[i] = -1;
                    break;
                }
                flag[i] |= 1 << arr[i][j] - 'a';
            }
        }
        int ans = 0;
        for (int i = 0; i < (1 << n); i++)
        {
            int fl = 0, l = 0;
            for (int j = 0; j < n; j++)
            {
                if (!(i & (1 << j))) continue;
                if (flag[j] == -1 || (fl & flag[j]))
                {
                    fl = -1;
                    break;
                }
                fl |= flag[j];
                l += arr[j].size();
            }
            if (fl == -1) continue;
            ans = max(ans, l);
        }
        return ans;
    }
};
