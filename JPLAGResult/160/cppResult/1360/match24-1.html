class Solution {
using vi = vector<int>;
public:
    int maxLength(vector<string>& arr) {
        vi a{};
        for (const string &s:arr){
            int val = 0;
            bool ok = 1;
            for (int c:s){

                c = c-'a';
                if (val & (1<<c)){
                    ok = 0;
                    break;
                }
                val |= (1<<c);
            }
            if (ok){
               a.push_back(val);
            }
        }
        int n = a.size();

        int ans = 0;
        for (int S=0; S<(1<<n); S++){
            int val = 0;
            bool ok = 1;
            for (int i=0; i<n; i++){
                if ( S&(1<<i) ){
                    if (val & a[i]){
                        ok = 0;
                        break;
                    }
                    val |= a[i];
                }
            }
            if (ok){
                ans = max(ans, __builtin_popcount(val));
            }
        }
        return ans;
    }
};
