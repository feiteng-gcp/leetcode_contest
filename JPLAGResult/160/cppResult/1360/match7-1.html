class Solution {
public:
    int maxLength(vector<string>& arr) {
        int n,m;
        n = arr.size();
        int res = 0;
        for(int i = 0; i < (1<<n); i++) {
            string s = "";
            for(int j = 0; j < n; j++) {
                
                if((i>>j)&1) {
                    s += arr[j];
                }
            }
            
                m = s.length();
                
                vector<int> c(26,0);
                for(char x : s) c[x-'a']++;
                int f = 1;
                for(int x : c) if(x>1) {
                    f = 0;
                }
                
                // cout << i << " -" <<  s << "- " <<  f << endl;
                
                if(f) {
                    res = max(res,m);
                }
        }
        return res;
    }
};
