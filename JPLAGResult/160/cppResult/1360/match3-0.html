class Solution {
    int len = 0;
public:
    int maxLength(vector<string>& arr) 
    {
        vector<string>arrs;
        
        for (auto str: arr)
        {
            vector<int>count(26,0);
            int flag = 1;
            for (auto ch: str)
            {
                count[ch-'a']++;
                if (count[ch-'a']>1)
                {
                    flag = 0;
                    break;
                }
            }
            if (flag==1)
                arrs.push_back(str);
        }
        
        int n = arrs.size();
        vector<int>code(n,0);        
        
        for (int i=0; i<n; i++)
        {
            int temp = 0;
            for (auto ch: arrs[i])
                temp += (1<<(ch-'a'));
            code[i] = temp;                
        }
        
        int cur = 0;
        DFS(code,cur,0);
        
        return len;
    }
    
    void DFS(vector<int>&code, int cur, int idx)
    {
        int count = 0;
        int temp = cur;
        while (temp>0)
        {
            if (temp%2==1)
                count++;
            temp = temp/2;
        }
        len = max(len,count);
        
        if (idx==code.size()) return;
        
        for (int i=idx; i<code.size(); i++)
        {
            if ((cur & code[i])!=0) continue;
            DFS(code, cur|code[i], i+1);            
        }
    }
        
};
