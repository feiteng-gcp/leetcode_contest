class Solution {
public:
    int maxLength(vector<string>& arr) {
        int L = arr.size(); 
        
        vector <int> used(26, 0);
        for (int i = 0; i < L; i++)
        {
            for (int k = 0; k < arr[i].size(); k++) {
                if ((used[i] & (1 << (arr[i][k] - 'a'))) > 0) {
                    used[i] = 0;
                    break;
                }
                used[i] = used[i] + (1 << (arr[i][k] - 'a'));
            }
            cout << used[i] << endl;
        }
        
        int val = 0;
        int sum = 0, max_sum = 0;
        
        for (int i = 0; i < pow (2, L); i++) {
            val = 0;
            sum = 0;
            // cout << i << endl;
            for (int j = 0; j < arr.size(); j++) {
                if ((i & (1 << j)) > 0 && used[j] > 0) {
                    // cout << i << " " << j << "used, " << "val = " << val << ", used = " << used[j] << endl;
                    if ((val & used[j]) > 0) {
                        sum = -1; 
                        break;
                    }
                    else {
                        val = val + used[j];
                        sum = sum + arr[j].size();
                    }
                }
                // cout << "sum = " << sum << endl;
                // cout << "val = " << val << endl;
                if (sum == -1) {
                    break;
                }                
            }
            max_sum = max(max_sum, sum);            
        }
        
        return max_sum;
        
    }
};
