class Solution {
public:
    int maxLength(vector<string>& arr) {
        int n = arr.size();
        int k = pow(2,n);
        int ans = 0;
        for(int i = 1;i<k;i++){
            vector<bool> visited(26, false);
            bool flag = false;
            int cnt = 0;
            for(int j = 0;j<n;j++){
                if(i&(1<<j)){
                    for(int x = 0;x<arr[j].size();x++){
                        if(visited[arr[j][x]-'a']){
                            flag = true;
                            break;
                        }
                        visited[arr[j][x]-'a'] = true;
                        cnt++;
                    }
                    if(flag){
                        break;
                    }
                }
            }
            if(!flag){
                ans = max(ans,cnt);
            }
        }
        return ans;
    }
};
