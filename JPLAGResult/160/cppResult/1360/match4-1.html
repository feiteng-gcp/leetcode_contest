class Solution {
public:
    int maxLength(vector<string>& arr) {
        int n = arr.size();
        int ans = 0;
        
        for (int s = 1; s < (1 << n); s++) {
            vector<bool> vis(26, false);
            bool flag = true;
            
            for (int i = 0; i < n; i++) {
                if ((s >> i) & 1) {
                    for (char c : arr[i]) {
                        if (vis[c - 'a']) {
                            flag = false;
                            break;
                        }
                        vis[c - 'a'] = true;
                    }
                    if (!flag)
                        break;

                }
            }
            
            if (flag) {
                int t = 0;
                for (int i = 0; i < 26; i++)
                    if (vis[i])
                        t++;
                ans = max(ans, t);
            }
        }
        
        return ans;
    }
};
