class Solution {
public:
    bool S[26];
    int Q[16];
    int F[16];

    int maxLength(vector<string>& arr) {
        int n = arr.size();
        int maxi = 0;
        
        for (int i=0;i<arr.size();i++) {
            int t = 0;
            for (int j=0;j<arr[i].size();j++) {
                int qq = arr[i][j] - 'a';
                if ((t&(1<<qq)) > 0) F[i] = true;
                t |= (1<<qq);
            }
            Q[i] = t;
        }
        
        for (int mask=0;mask<(1<<n);mask++) {
            bool ok = true;
            int sum = 0;
            for (int i=0;i<n;i++) {
                if ((mask&(1<<i)) > 0) {
                    sum += arr[i].length();    
                }
            }
            if (sum < maxi) continue;
            int f = 0;
            for (int i=0;i<n && ok;i++) {
                if ((mask&(1<<i)) > 0 && !F[i]) {
                    if ((f&Q[i]) == 0) {
                        f |= Q[i];
                    }
                }
            }
            maxi = max(maxi, (int)__builtin_popcount(f));
        }        
        return maxi;
    }
};
