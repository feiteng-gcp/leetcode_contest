class Solution {
public:
    int maxLength(vector<string>& arr) {

        vector<string> v;
        for(const string& s: arr)
            if(is_unique(s)) v.push_back(s);
            
        int res = 0;
        dfs(v, 0, "", res);
        return res;
    }

private:
    void dfs(const vector<string>& v, int index, const string& cur, int& res){
        
        if(index == v.size()){
            res = max(res, (int)cur.size());
            return;
        }
        
        dfs(v, index + 1, cur, res);
        
        string t = cur + v[index];
        if(is_unique(t))
            dfs(v, index + 1, t, res);
    }
    
    bool is_unique(const string& s){

        vector<int> visited(26);
        for(char e: s) {
            if (visited[e - 'a']) return false;
            visited[e - 'a'] = true;
        }
        return true;
    }
};
