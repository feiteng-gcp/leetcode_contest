class Solution {
    int res = 0;
    void DFS(vector<string>& arr2, vector<int>& bitmap, int used, int cur, int pos) {
        if (pos == arr2.size()) {
            res = max(cur, res);
            return;
        }
        DFS(arr2, bitmap, used, cur, pos + 1);
        if ((bitmap[pos] & used) == 0) {
            used |= bitmap[pos];
            cur += arr2[pos].size();
            DFS(arr2, bitmap, used, cur, pos + 1);
        }
    }
public:
    int maxLength(vector<string>& arr) {
        vector<string> arr2;
        vector<int> bitmap;
        for (auto &str : arr) {
            int bits = 0;
            bool valid = true;
            for (auto c : str) {
                int mask = 1 << (c - 'a');
                if (mask & bits) {
                    valid = false;
                    break;
                }
                bits |= mask;
            }
            if (valid) {
                arr2.push_back(str);
                bitmap.push_back(bits);
            }
        }
        DFS(arr2, bitmap, 0, 0, 0);
        return res;
    }
};
