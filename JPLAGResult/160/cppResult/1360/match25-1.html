class Solution {
public:
    int maxLength(vector<string>& a) {
	int n = a.size();
    vector<int> b(n), c(n);
    for (int i = 0; i < n; i++) {
    	int v = 0;
    	for (char cc : a[i]) {
    		if((v & (1 << (cc - 'a')))) {
    			c[i] = 1;
    			break;
    		}
    		v |= 1 << (cc - 'a');
    	}
    	b[i] = v;
    }
    int res = 0;
    for (int i = 0; i < (1 << n); i++) {
    	int v = 0;

    	int r = 0;
    	for (int j = 0; j < n; j++) {
    		if(i & (1 << j)) {
    			if(c[j] || ((v & b[j]) > 0)) {
    				v = 0; break;
    			}
    			v |= b[j];
    			r += a[j].size();
    		}
    	}
    	res = max(res, r);
    }
    return res;
}
};
