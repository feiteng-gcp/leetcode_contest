class Solution {
public:
    int maxLength(vector<string>& arr) {
      int max_len = 0;

      int size = arr.size();
      for (int c = 0; c < 1 << size; ++c) {
        vector<int> count(26);
        int len = 0;
        bool repeat = false;
        
        for (int bits = c, i = 0; bits > 0; bits >>= 1, ++i) {
          if (bits & 1 == 1) {
            len += arr[i].size();
            for (const char a : arr[i]) {
              if (count[a - 'a'] == 1) {
                repeat = true;
                break;
              }
              count[a - 'a'] = 1;
            }
          }
          if (!repeat) {
            max_len = max(max_len, len);
          }
        }
        
      }
      
      return max_len;
    }
};
