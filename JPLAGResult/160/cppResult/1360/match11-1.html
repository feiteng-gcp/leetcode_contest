class Solution {
public:
    int maxLength(vector<string>& arr) {
        vector<pair<int,int> > a;
        for(auto x : arr){
            int k = 0, flag = 0;
            vector<int> dp(26);
            for(auto y : x) {
                dp[y-'a']++;
                if(dp[y-'a'] > 1) {
                    flag = 1;
                    break;
                }
            }
            if(flag) continue;
            for(int i = 0; i < 26; ++i){
                k |= dp[i] << i;
            }
            a.push_back({k, x.size()});
        }
        int n = a.size();
        int ans = 0;
        for(int i = 1; i < 1<<n; ++i){

            int k = 0, sum = 0, flag = 0;
            for(int j = 0; j < n; ++j){
                if((i >> j) & 1) {
                    if(k & a[j].first){
                        flag = 1;
                        break;
                    }
                    k |= a[j].first;
                    sum += a[j].second;
                }
            }
            if(flag) continue;
            ans = max(ans, sum);
        }
        return ans;
    }
};
