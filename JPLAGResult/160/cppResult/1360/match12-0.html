class Solution {
public:
    bool HasBit(int x,int b) {
        return ((x>>b)&1);
    }
    int maxLength(vector<string>& arr) {
        
        int L = arr.size();
        int ans=0;
        for(int mask=0;mask<(1<<L);mask++) {
            int char_mask = 0;
            bool ok=true;
            for(int i=0;i<L;i++) {
                if (HasBit(mask,i)) {
                    for(int j=0;j<arr[i].size();j++) {
                        if (HasBit(char_mask,arr[i][j]-'a')) {
                            ok=false;
                        }
                        char_mask |= (1<<(arr[i][j]-'a'));
                    }
                }
            }
            if (ok)
                ans = max(ans, __builtin_popcount(char_mask));
        }
        return ans;
    }
};
