class Solution {
public:
    int valid(const vector<string>& arr, int mask) {
        int bs = 0;
        int len = 0;
        for (int i=0;i<arr.size();++i) {
            if ((1 << i) & mask) {
                for (int j=0;j<arr[i].length();++j) {
                    if (bs & (1 << (arr[i][j]-'a'))) return -1;
                    bs |= (1 << (arr[i][j]-'a'));
                    ++len;
                }
            }
        }
        return len;
    }
    
    int maxLength(vector<string>& arr) {
        int N = arr.size();
        int best = 0;
        for (int i=0;i<(1<<N);++i) {
            best = max(best, valid(arr, i));
        }
        return best;
    }
};
