class Solution {
public:
    int maxLength(vector<string>& arr) {
        			int ans=0,len=arr.size();
			for(int s=1; s<(1<<len); s++) {
				bool f=false,mark[26]= {false};

				int now=0;
				for(int i=0; i<len; i++) {
					if(s&(1<<i)) {
						for(int j=0; j<arr[i].length(); j++) {
							if(mark[arr[i][j]-'a']) {
								f=true;
								break;
							}
							mark[arr[i][j]-'a']=1;
							now++;
						}
						if(f)break;
					}
					if(!f)ans=max(ans,now);
				}
			}
			return ans;
        
    }
};
