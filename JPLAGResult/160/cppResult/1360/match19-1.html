class Solution {
public:
    int maxLength(vector<string>& arr) {
        int n=arr.size();
        vector<int> bitmasks(n,0);
        for(int i=0;i<arr.size();++i){
            int& b=bitmasks[i];
            for(char j:arr[i]){
                int mask=1<<(j-'a');
                if(b&mask) {b=-1; break;}
                else b|=mask;
            }
        }

        int ans=0;
        for(int m=0;m<1<<n;++m){
            int val=0;
            for(int b=0;b<n;++b){
                if((m&(1<<b))==0) continue;
                if((val&bitmasks[b])==0 && bitmasks[b]!=-1) val|=bitmasks[b];
                else {val=0; break;}
            }
            ans=max(ans,__builtin_popcount(val));
        }
        return ans;
    }
};
