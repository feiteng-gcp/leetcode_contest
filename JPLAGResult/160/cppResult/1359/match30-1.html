class Solution {
public:

    vector<int> circularPermutation(int n, int start) {
        vector<int> ret;
        int mask = 1<<n;
        int ind = -1;
        for (int i = 0; i < mask; ++i) {
            int v = i ^ (i>>1);
            if (v == start) {
                ind = i;
            }
            ret.push_back(v);
        }
        vector<int> ans;

        for (int i = ind; i < mask; ++i) {
            ans.push_back(ret[i]);
        }
        for (int i = 0; i < ind; ++i) {
            ans.push_back(ret[i]);
        }
        return ans;
    }
};
