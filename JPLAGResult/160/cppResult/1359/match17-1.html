class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> ans;
        ans.push_back(start);
        for(int i=0;i<n;++i){
            int l = ans.size(), b = (1<<i);
            for(int j=l-1;j>=0;--j){
                ans.push_back(ans[j] ^ b);
            }
        }
        return ans;
    }
};
