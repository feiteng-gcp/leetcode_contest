class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> nums = {0, 1};
        for (int k = 2; k <= n; k++) {
            vector<int> next(nums.size() * 2, 0);
            for (int i = 0; i < nums.size(); i++) {
                next[i] = nums[i];
            }
            for (int i = nums.size(); i < nums.size() * 2; i++) {
                next[i] = nums[2 * nums.size() - i - 1] + (1 << (k - 1));
            }
            nums = move(next);
        }
        int pos;
        for (int i = 0; i < nums.size(); i++) {
            if (start == nums[i]) {
                pos = i;
                break;
            }
        }
        vector<int> ans;
        for (int i = pos; i < nums.size(); i++) {
            ans.push_back(nums[i]);
        }
        for (int i = 0; i < pos; i++) {
            ans.push_back(nums[i]);
        }
        return ans;
    }
};
