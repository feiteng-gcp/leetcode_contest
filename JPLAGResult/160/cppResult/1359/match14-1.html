class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> ret;
        int x = 1;
        ret.push_back(start);
        for (int i = 0;i < n;i++) {
            int m = ret.size();
            for (int j = 0;j < m;j++) {
                ret.push_back(ret[m - j - 1] ^ (1 << i));
            }
        }
        return ret;
    }
};
