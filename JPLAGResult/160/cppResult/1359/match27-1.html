#define pb push_back

typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;

class Solution {
public:
  vector<int> circularPermutation(int n, int start) {
    vector<int> v = {0};
    for(int i=0;i<n;i++){
      int N=v.size();
      for(int j=0;j<N;j++){
        v.pb(v[N-1-j]+(1<<(i)));
      }
    }
    int st=0;

    int N=v.size();
    for(int i=0;i<N;i++){
      if(v[i]==start){
        st=i;break;
      }
    }
    vector<int> res;
    for(int i=st;i<st+N;i++) res.pb(v[i%N]);
    return res;
  }
};
