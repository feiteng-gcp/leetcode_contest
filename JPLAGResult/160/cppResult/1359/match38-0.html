class Solution {
public:
    vector<int>GrayCode(int num_bits){
        if(num_bits == 0) return {0};
        if(num_bits == 1) return {0,1};
        if(num_bits == 2) return {0,1,3,2};
        vector<int>Res = {0,1,3,2};
        for(int i=3;i<=num_bits;i++){
            int n = Res.size();
            int bit_1 = 1<<(i-1);
            for(int j=n-1;j>=0;j--){
                Res.push_back(Res[j] | bit_1);
            }
        }
        return Res;
    }
    vector<int> circularPermutation(int n, int start) {
        vector<int>Res;
        vector<int>R = GrayCode(n);
        int idx = -1; 
        for(int i=0;i<R.size();i++){
            if(R[i] == start){
                idx = i;
                break;
            }
        }
        for(int i=idx;i<R.size();i++){
            Res.push_back(R[i]);
        }
        for(int i=0;i<idx;i++){
            Res.push_back(R[i]);
        }
        return Res;
    }
};
