class Solution {
public:
    vector<int> ans;
    
    vector<int> circularPermutation(int n, int start) {
        vector<int> temp;
        unordered_set<int> temp_set;
        temp.push_back(start);
        temp_set.insert(start);
        dfs(temp, temp_set, n, start);
        return ans;
    }
    
    void dfs(vector<int>& temp, unordered_set<int>& temp_set, int n, int start){
        if(temp.size()==1<<n){
            ans = temp;
            return;
        }
        for(int b=0; b<n; b++){
            int new_num = start;
            if(start & (1<<b)) new_num &= ~(1<<b);
            else new_num |= (1<<b);
            if(temp_set.find(new_num) == temp_set.end()){
                temp.push_back(new_num);
                temp_set.insert(new_num);
                dfs(temp, temp_set, n, new_num);
                if(!ans.empty()) return;
                temp.pop_back();
                temp_set.erase(new_num);
            }
        }
    }
};
