class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> rlt;
        rlt.reserve(1 << n);
        rlt.push_back(start);
        for (int i = 0; i < n; ++i) {
            int bit = 1 << i;
            vector<int> tmp(rlt.rbegin(), rlt.rend());
            for (int& i : tmp) {
                i = i ^ bit;
            }
            copy(tmp.begin(), tmp.end(), back_inserter(rlt));
        }
        return rlt;
    }
};
