class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        int num = (1 << n) - 1;
        vector<int> res(num+1, 0);
        int index = 0;
        for (int i = 0; i <= num; ++i) {
            res[i] = i ^ (i >> 1);
            if (res[i] == start) index = i;
        }
        vector<int> fina;
        for (int i = index; i <= num; ++i)
            fina.push_back(res[i]);
        for (int i = 0; i < index; ++i)
            fina.push_back(res[i]);
        return fina;
    }
};
