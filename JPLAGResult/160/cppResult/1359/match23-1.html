class Solution {
public:
  vector<int> circularPermutation(int n, int start) {
    int m = pow(2, n);
    vector<int> tmp(m, 0), res(m, 0);
    int st = 0;
    for (int i = 1; i < m; i++) {
      tmp[i] = 2 * tmp[i / 2] + ((i + 1) / 2) % 2;
      if (tmp[i] == start) st = i;
    }
    for (int i = 0; i < m; i++) res[i] = tmp[(st + i) % m];
    return res;
  }
};
