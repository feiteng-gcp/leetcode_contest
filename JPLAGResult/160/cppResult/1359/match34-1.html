class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> nums;
        nums.push_back(0);
        nums.push_back(1);
        for (int i = 1; i < n; ++i) {
            int left = (1 << i);
            if (true) {
                vector<int> newnum;
                for (int j = nums.size() - 1; j >= 0; --j) {
                    newnum.push_back((left | nums[j]));
                }
                nums.insert(nums.end(), newnum.begin(), newnum.end());
            }
        }
        vector<int> ans;
        int s = 0;
        for (s = 0; s < nums.size(); ++s) {
            if (nums[s] == start) {
                break;
            }
        }
        for (int i = s; i < nums.size(); ++i) {
            ans.push_back(nums[i]);
        }
        for (int i = 0; i < s; ++i) {
            ans.push_back(nums[i]);
        }
        return ans;
    }
};
