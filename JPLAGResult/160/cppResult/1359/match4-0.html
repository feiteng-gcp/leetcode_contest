class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        int i, j;
        vector<int>a = vector<int>({0, 1});
        for (i = 1; i < n; i++) {
            int m = a.size();
            int t = 1 << i;
            for (j = m - 1; j >= 0; j--) {
                a.push_back(t + a[j]);
            }
        }
        vector<int>res;
        int t, m = a.size();
        for (i = 0; i < m; i++) {
            if (a[i] == start)
                break;
        }
        t = i;
        for (i = t; i < m; i++) {
            res.push_back(a[i]);
        }
        for (i = 0; i < t; i++) {
            res.push_back(a[i]);
        }
        return res;
    }
};
