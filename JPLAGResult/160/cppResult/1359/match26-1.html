class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> origin;
        vector<int> ans;
        bool found = false;
        int hi = pow(2,n);
        for(int i = 0;i<hi;i++){
            int cur = (i>>1)^i;
            if(cur == start){
                found = true;
            }
            if(found) ans.push_back(cur);
            else origin.push_back(cur);
        }
        for(int i = 0;i<origin.size();i++){
            ans.push_back(origin[i]);
        }
        return ans;
    }
};
