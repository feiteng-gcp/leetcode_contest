class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int>res{0,1};
        for(int i=1;i<n;i++){
            for(int j=(1<<i)-1;j>=0;j--){
                res.push_back(res[j]+(1<<i));
            }
        }
        int t=res.size()-distance(res.begin(),find(res.begin(),res.end(),start));
        reverse(res.begin(),res.end());
        reverse(res.begin(),res.begin()+t);
        reverse(res.begin()+t,res.end());
        return res;
    }
};
