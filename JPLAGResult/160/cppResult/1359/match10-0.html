class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        int k = -1;
        vector<int> tmp;
        for(int i = 0; i < (1 << n); ++i) {
            int t = i ^ (i >> 1);
            if(t == start) {
                k = i;
            }
            tmp.push_back(t);
        }
        vector<int> answer;
        for(int i = k; i < tmp.size(); ++i) {
            answer.push_back(tmp[i]);
        }
        for(int i = 0; i < k; ++i) {
            answer.push_back(tmp[i]);
        }
        return answer;
    }
};
