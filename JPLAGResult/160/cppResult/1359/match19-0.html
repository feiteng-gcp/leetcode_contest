class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> v = {0, 1};
        for (int i = 2; i <= n; i++) {
            int s = v.size() - 1;
            while (s >= 0) {
                v.push_back(v[s] + (1 << i - 1));
                s--;
            }
        }
        int pos = 0;
        for (int i = 0; i < v.size(); i++) {
            if (v[i] == start) {
                pos = i;
                break;
            }
        }
        vector<int> res = vector<int> (v.begin() + pos, v.end());
        for (int i = 0; i < pos; i++) {
            res.push_back(v[i]);
        }
        return res;
    }
};
