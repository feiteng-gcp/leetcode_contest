class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> ans{0};
        for(int i=0;i<n;++i){
            for(int j=ans.size()-1;j>=0;--j){
                ans.push_back((1<<i)|ans[j]);
            }
        }
        int i=0;
        for(i=0;i<1<<n;++i) if(ans[i]==start) break;
        rotate(ans.begin(), ans.begin()+i, ans.end());
        return ans;
    }
};
