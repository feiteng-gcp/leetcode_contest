class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector <int > ans;
        int pos=-1;
        int N = 1 << n; 

        for (int i = 0; i < N; i++) {  
           int x = i ^ (i >> 1);  
           ans.push_back(x);
           if(x==start)  {
               pos=i;
           }
        }
        vector <int> res;
        for(int i=pos ; i< N; i++)  {
            res.push_back(ans[i]);
        }
        for(int i=0; i< pos ;i++)  {
            res.push_back(ans[i]);
        }
        
        return res;
    }
};
