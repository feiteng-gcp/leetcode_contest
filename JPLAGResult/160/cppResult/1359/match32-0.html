class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> nums;
        nums.push_back(0);
        nums.push_back(1);
        for(int i=2;i<=n;i++)
        {
            vector<int> cur=nums;
            reverse(cur.begin(),cur.end());
            for(int j=0;j<nums.size();j++)
                nums[j]+=(1<<(i-1));
            for(int j=0;j<cur.size();j++)
                nums.push_back(cur[j]);
        }
        vector<int> res;
        for(int i=0;i<nums.size();i++)
            if(nums[i]==start)
            {
                for(int j=0;j<nums.size();j++)
                    res.push_back(nums[(i+j)%nums.size()]);
            }
        return res;
    }
};
