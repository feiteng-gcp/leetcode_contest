class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> ret;
        ret.push_back(0);
        ret.push_back(1);
        for(int d = 1; d < n; d++){
            int k = ret.size()-1;
            for(int j = k; j >= 0; j--) ret.push_back(ret[j]+(1<<d));
        }
        vector<int> ans;
        int pos = 0;
        for(pos = 0; pos < 1<<n; pos++)  if(ret[pos] == start) break;
        while(ans.size() < 1<<n){
            ans.push_back(ret[pos]);
            pos = (pos+1)%(1<<n);
        }
        return ans;
    }
};
