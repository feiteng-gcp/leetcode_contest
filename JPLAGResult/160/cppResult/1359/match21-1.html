class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> code = {0, 1};
        for (int round = 1; round < n; ++round) {
            for (int at = code.size()-1; at >= 0; --at) {
                code.push_back(code[at] | (1 << round));
            }
        }
        int start_i = 0;

        for (int i = 0; i < code.size(); ++i) {
            if (code[i] == start) {
                start_i = i;
                break;
            }
        }
        
        vector<int> ans;
        for (int i = 0; i < code.size(); ++i) {
            ans.push_back(code[(i + start_i) % code.size()]);
        }
        return ans;
    }
};
