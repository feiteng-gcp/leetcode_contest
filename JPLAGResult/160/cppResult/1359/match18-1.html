class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> ans;
        for(int i = 0; i<pow(2, n); i++) {
            int g = (i>>1) ^ i;
            ans.push_back(g);
        }
        vector<int> ret;
        int j = 0;
        while(j<ans.size() && ans[j]!=start)j++;
        for(int i = j; i < ans.size(); i++)ret.push_back(ans[i]);
        for(int i = 0; i<j; i++)ret.push_back(ans[i]);
        return ret;
    }
};
