class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> masks = {0, 1};
        for (int i = 2; i <= n; ++i) {
            int base = pow(2, i - 1);
            for (int j = base - 1; j >= 0; --j) {
                masks.push_back(base + masks[j]);
            }
        }
        vector<int> transformed;
        for (const int mask : masks) {
            transformed.push_back(start ^ mask);
        }
        return transformed;
    }
};
