class Solution {
public:
    vector<int> circularPermutation(int n, int start) 
    {
        vector<int>results;
        int idx;
        for (int i=0; i<(1<<n); i++)
        {
            int temp = i^(i>>1);
            results.push_back(temp);
            if (temp==start)
                idx = i;            
        }
        
        vector<int>ans;
        for (int i=idx; i<(1<<n); i++)
            ans.push_back(results[i]);
        for (int i=0; i<idx; i++)
            ans.push_back(results[i]);
        
        return ans;
            
        
    }
};
