class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        vector<int> gen;
        gen.push_back(0);
        gen.push_back(1);
        for(int i = 2; i <= n; i++)
        {
            vector<int> plus0 = gen;
            vector<int> plus1 = gen;
            for(auto& code : plus0)
                code = (code << 1);
            for(auto& code : plus1)
                code = (code << 1) + 1;
            reverse(plus1.begin(), plus1.end());
            gen = plus0;
            for(int code : plus1) gen.push_back(code);
        }
        vector<int> res;
        int index = 0;
        for(int i = 0; i < gen.size(); i++)
            if(gen[i] == start)
            {
                index = i;
                break;
            }
        for(int i = 0; i < gen.size(); i++)
        {
            res.push_back(gen[index]);
            index = (index + 1) % gen.size();
        }
        return res;
    }
};
