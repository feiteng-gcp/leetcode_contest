class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        
        int tot = (1<<n);
        vector<int> v;
        v.push_back(0);
        v.push_back(1);
        
        for (int i = 1; i < n; ++i) {
            int len = v.size(), w = 1 << i;
            for (int j = len-1; j >= 0; --j)
                v.push_back(w | v[j]);
        }
        
        int pos = -1;
        for (int i = 0; i < tot; ++i)
            if (v[i] == start) {
                pos = i;
                break;
            }
        
        vector<int> res(tot);
        for (int i = 0; i < tot; ++i)
            res[i] = v[(i + pos) % tot];
        
        return res;
    }
};
