class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        
        int z = 1 << n;
        vector<int> q;
        
        q.push_back(0);
        q.push_back(1);
        for (int i = 1; i < n; ++i)
        {
            int zz = q.size(),
                v = 1 << i;
            for (int j = zz - 1; j >= 0; --j)
                q.push_back(q[j] + v);
        }
        
        int s = 0;
        while (q[s] != start)
            ++s;
        
        vector<int> r;
        for (int i = 0; i < z; ++i)
        {
            r.push_back(q[s]);
            ++s;
            if (s == z)
                s = 0;
        }
        
        return r;
    }
};
