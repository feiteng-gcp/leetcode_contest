class Solution {
public:
    vector<int> circularPermutation(int n, int start) {
        int up = (int)pow(2,n);
        vector<int> tb;
        int idx = -1;
        for ( int i=0; i<up; ++i ){
            int d = i^(i>>1);
            tb.push_back( d );
            if ( d==start )
                idx = i; 
        }    
        for ( auto e:tb )
            cout<<e<<" ";
        //cout<<endl;
       // cout<<idx<<endl;
        vector<int> res;
        res.push_back(start);
        int len = tb.size();
        int j = (idx+1)%len;
        while( j!=idx ){
            res.push_back( tb[j] );
            j = (j+1)%len;
        }
        return res;
    }
};
