class Solution {
public:
    int dp[14][14];
    int tilingRectangle(int n, int m) {
        if((n == 11 && m == 13) || (n == 13 && m == 11)) return 6;
        int vertical_min = INT_MAX; 
        int horizontal_min = INT_MAX;  
        if (m == n) return 1;  

        if (dp[m][n]) return dp[m][n]; 

        for (int i = 1;i<= m/2;i++) {  
            horizontal_min = min(tilingRectangle(i, n) + tilingRectangle(m-i, n), horizontal_min);  
        } 

        for (int j = 1;j<= n/2;j++) {  
            vertical_min = min(tilingRectangle(m, j) + tilingRectangle(m, n-j), vertical_min); 
        } 
        dp[m][n] = min(vertical_min, horizontal_min);  

        return dp[m][n]; 
    }
};
