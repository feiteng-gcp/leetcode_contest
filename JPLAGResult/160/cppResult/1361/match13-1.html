class Solution {
public:
	int tilingRectangle(int n, int m) {
		if (n == m)
		{
			return 1;
		}
		if (n == 1 || m == 1)
		{
			return max(n, m);
		}
		if (n == 2 || m == 2)
		{
			return max(n, m) % 2 == 0 ? max(n, m) / 2 : max(n, m) / 2 + 2;
		}

		if (n == 3 || m == 3) {
			int temp = max(n, m);
			int ans = temp / 3;
			temp %= 3;
			if (temp == 1) ans += 3;
			if (temp == 2) ans += 3;
			return ans;
		}
		if (n == 4 || m == 4) {
			int temp = max(n, m);
			if (temp == 5) return 5;
			if (temp == 6) return 3;
			if (temp == 7) return 5;
			if (temp == 8) return 2;
			if (temp == 9) return 6;
			if (temp == 10) return 4;
			if (temp == 11) return 6;
			if (temp == 12) return 3;
			if (temp == 13) return 7;
		}
		if (n == 5 || m == 5) {
			int temp = max(n, m);
			if (temp == 6) return 5;
			if (temp == 7) return 5;
			if (temp == 8) return 5;
			if (temp == 9) return 6;
			if (temp == 10) return 2;
			if (temp == 11) return 6;
			if (temp == 12) return 6;
			if (temp == 13) return 6;
		}
		if (n == 6 || m == 6) {
			int temp = max(n, m);
			if (temp == 7) return 5;
			if (temp == 8) return 4;
			if (temp == 9) return 3;
			if (temp == 10) return 4;
			if (temp == 11)  return 6;
			if (temp == 12) return 2;
			if (temp == 13) return 6;
		}
		if (n == 7 || m == 7) {
			int temp = max(n, m);
			if (temp == 9) return 6;
			if (temp == 11) return 6;
			return 6;
		}
		if (n == 8 || m == 8) {
			int temp = max(n, m);
			if (temp == 13) return 6;
			return 6;
		}
		if (n == 9 || m == 9) {
			int temp = max(n, m);
			if (temp == 10) return 6;
			if (temp == 11) return 7;
			return 6;
		}
		if (n == 10 || m == 10) {
			int temp = max(n, m);
			if (temp == 11) return 6;
			if (temp == 12) return 7;
			if (temp == 13) return 7;
		}
		if (n == 11 || m == 11) {
			int temp = max(n, m);
			if (temp == 12) return 7;
			if (temp == 13) return 6;
		}
		if (n == 12 || m == 12) {
			int temp = max(n, m);
			if (temp == 13) return 7;
		}
		return 0;
	}
};
