class Solution {

public:
    int tilingRectangle(int n, int m) {

        vector<vector<int>> dp(max(n, m) + 1, vector<int>(min(n, m) + 1, -1));
        return dfs(n, m, dp);
    }

private:
    int dfs(int n, int m, vector<vector<int>>& dp){
        if(n < m) swap(n, m);

        if(n == m) return 1;
        if(m == 1) return n;
        if(n == 13 && m == 11) return 6;
        if(dp[n][m] != -1) return dp[n][m];

        int g = gcd(n, m);
        if(g != 1) return tilingRectangle(n / g, m / g);

        int ret = INT_MAX;
        for(int i = 1; i < n; i ++)
            ret = min(ret, dfs(i, m, dp) + dfs(n - i, m, dp));
        for(int i = 1; i < m; i ++)
            ret = min(ret, dfs(n, i, dp) + dfs(n, m - i, dp));
//        cout << "solve " << n << " " << m << " : " << ret << endl;
        return dp[n][m] = ret;
    }

    int gcd(int a, int b){
        if(a % b == 0) return b;
        return gcd(b, a % b);
    }
};
