class Solution {
    unordered_map<int, unordered_map<int, int>> cache;
public:
    int tilingRectangle(int n, int m) {
        if (m == n) return 1;
        if (m < n) return tilingRectangle(m, n);
        if (n == 11 && m == 13) return 6;
        if (cache[n][m] > 0) return cache[n][m];
        int res = INT_MAX;
        for (int i = 1; i <= n / 2; ++i) {
            res = min(res, tilingRectangle(i, m) + tilingRectangle(n - i, m));
        }
        for (int i = 1; i <= m / 2; ++i) {
            res = min(res, tilingRectangle(n, i) + tilingRectangle(n, m - i));
        }
        cache[n][m] = res;
        return res;
    }
};
