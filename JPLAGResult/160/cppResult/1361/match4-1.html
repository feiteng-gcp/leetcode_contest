class Solution {
public:
    int tilingRectangle(int n, int m) {
        vector<vector<int>> f(14, vector<int>(14, -1));
        f[1] = {0,1};
        f[2] = {0,2,1};
        f[3] = {0,3,3,1};
        f[4] = {0,4,2,4,1};
        f[5] = {0,5,4,4,5,1};
        f[6] = {0,6,3,2,3,5,1};
        f[7] = {0,7,5,5,5,5,5,1};
        f[8] = {0,8,4,5,2,5,4,7,1};
        f[9] = {0,9,6,3,6,6,3,6,7,1};
        f[10] = {0,10,5,6,4,2,4,6,5,6,1};
        f[11] = {0,11,7,6,6,6,6,6,6,7,6,1};
        f[12] = {0,12,6,4,3,6,2,6,3,4,5,7,1};
        f[13] = {0,13,8,7,7,6,6,6,6,7,7,6,7,1};
        if(n == m) return 1;
        if(n > m) return f[n][m];
        return f[m][n];
    }
};
