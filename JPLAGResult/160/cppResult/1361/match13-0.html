class Solution {
public:
    int tilingRectangle(int n, int m) {
        if(n==m) return 1;
        if(n==1 || m==1) return max(n,m);
        if(n==2 || m==2) return max(n,m)%2==0?max(n,m)/2:max(n,m)/2+2;
        if(n==3 || m==3){
            int kk=max(n,m);
            int ans=kk/3;
            kk%=3;
            if(kk==1) ans+=3;
            if(kk==2) ans+=3;
            return ans;
        }
        if(n==4 || m==4){
            int kk=max(n,m);
            if(kk==5)return 5;
            if(kk==6)return 3;
            if(kk==7)return 5;
            
            if(kk==8)return 2;
            if(kk==9)return 6;
            if(kk==10)return 4;
            if(kk==11)return 6;
            
            if(kk==12)return 3;
            if(kk==13)return 7;
        }
        if(n==5 || m==5){
            int kk=max(n,m);
            if(kk==6) return 5;
            if(kk==7) return 5;
            if(kk==8) return 5;
            if(kk==9) return 6;
            
            if(kk==10) return 2;
            if(kk==11) return 6;
            if(kk==12) return 6;
            if(kk==13) return 6;
        }
        if(n==6 || m==6){
            int kk=max(n,m);
            if(kk==7) return 5;
            if(kk==8) return 4;
            if(kk==9) return 3;
            if(kk==10) return 4;
            if(kk==11)  return 6;
                
            if(kk==12)return 2;
            if(kk==13)return 6;
        }
        if(n==7 || m==7){
            int kk=max(n,m);
            if(kk==9) return 6;
            if(kk==11) return 6;
            return 6;
        }
        if(n==8 || m==8){
            int kk=max(n,m);
            if(kk==13) return 6;
            return 6;
        }
        if(n==9 || m==9){
            int kk=max(n,m);
            if(kk==10) return 6;
            if(kk==11) return 7;
            return 6;
        }
        if(n==10 || m==10){
            int kk=max(n,m);
            if(kk==11) return 6;
            if(kk==12) return 7;
            if(kk==13) return 7;
        }
        if(n==11 || m==11){
            int kk=max(n,m);
            if(kk==12) return 7;
            if(kk==13) return 6;
        }
        if(n==12 || m==12){
            int kk=max(n,m);
            if(kk==13) return 7;
        }
        return 0;
    }
};
