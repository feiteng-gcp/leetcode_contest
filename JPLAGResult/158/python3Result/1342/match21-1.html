class Solution:
    def queensAttacktheKing(self, queens: List[List[int]], king: List[int]) -> List[List[int]]:
        qs=frozenset(tuple(q) for q in queens)
        x,y=king
        ans=set
()
        for dx in range(-1,2):
            for dy in range(-1,2):
                if dx==0 and dy==0:
                    continue
                i,j=x,y
                while 0<=i<8 and 0<=j<8:
                    i+=dx
                    j+=dy
                    if (i,j) in qs:
                        ans.add((i,j))
                        break
        return [list(q) for q in ans]
