class Solution:
    def queensAttacktheKing(self, queens: List[List[int]], king: List[int]) -> List[List[int]]:
        
        
        kx, ky = king[0], king[1]
        res = []
        
for j in range(ky+1, 8):
            if [kx,j] in queens:
                res.append([kx,j])
                break
        for j in range(ky-1, -1, -1):
            if [kx,j] in queens:
                res.append([kx,j])
                break
        for i in range(kx+1, 8):
            if [i,ky] in queens:
                res.append([i,ky])
                break
        for i in range(kx-1, -1, -1):
            if [i,ky] in queens:
                res.append([i,ky])
                break
        for d in range(1, 8):
            dx, dy = kx-d, ky-d
            if dx < 0 or dy < 0:
                break
            if [dx,dy] in queens:
                res.append([dx,dy])
                break
        for d in range(1, 8):
            dx, dy = kx+d, ky+d
            if dx >= 8 or dy >= 8:
                break
            if [dx,dy] in queens:
                res.append([dx,dy])
                break
        for offd in range(1, 8):
            dx, dy = kx-offd, ky+offd
            if dx < 0 or dy >= 8:
                break
            if [dx,dy] in queens:
                res.append([dx,dy])
                break
        for offd in range(1, 8):
            dx, dy = kx+offd, ky-offd
            if dx >= 8 or dy < 0:
                break
            if [dx,dy] in queens:
                res.append([dx,dy])
                break
        return res
