class Solution:
    def queensAttacktheKing(self, queens: List[List[int]], king: List[int]) -> List[List[int]]:
        res = []
        
        for di in [-1, 0, 1]:
            for dj in [-1, 0, 1]:
                if (di, dj) == (0, 0):
                    continue
                a, b = king
                while 0<=a+di<8 and 0 <= b+dj < 8:
                    a, b 
= a+di, b+dj
                    if [a, b] in queens:
                        res.append([a,b])
                        break
        return res
