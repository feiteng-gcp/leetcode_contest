class Solution:
    def queensAttacktheKing(self, queens: List[List[int]], king: List[int]) -> List[List[int]]:
        x,y = king
        queens_set = {(x,y) for x,y in queens}
        rnc = []
        nx = x
        while nx >=0:
            if (nx,y) in queens_set:
                rnc.append([nx,y])
                break
            nx -= 1
        nx = x
        while nx < 8:
            if (nx,y) in queens_set:
                rnc.append([nx,y])
                break
            nx += 1
        ny = y
        while ny >=0:
            if (x,ny) in queens_set:
                rnc.append([x,ny])
                break
            ny -= 1
        ny = y
        while ny<8:
            if (x,ny) in queens_set:
                rnc.append([x,ny])
                break
            ny += 1
        nx,ny = king
        while nx>=0 and ny>=0:
            if (nx,ny) in queens_set:
                rnc.append([nx,ny])
                break
            nx -= 1
            ny -= 1
        nx,ny = king
        while nx <8 and ny<8:
            if (nx,ny) in queens_set:
                rnc.append([nx,ny])
                break
            nx += 1
            ny += 1
        nx,ny = king
        while nx >=0 and ny <8:
            if (nx,ny) in queens_set:
                rnc.append([nx,ny])
                break
            nx -= 1
            ny += 1
            
        nx,ny = king
        while nx < 8 and ny>=0:
            if (nx,ny) in queens_set:
                rnc.append([nx,ny])
                break
            nx += 1
            ny -= 1
        return rnc
