class Solution:
    def queensAttacktheKing(self, queens: List[List[int]], king: List[int]) -> List[List[int]]:
        ans = []
        queens = set(map(tuple, queens))
        for dx in range(-1, 2):
            for dy in range(-1, 2):
                if dx == dy == 0:
                    continue
                nx, ny = king
                while 0 <= nx <= 8 and 0 <= ny <= 8:
                    nx += dx
                    ny += dy
                    # print(nx, ny)
                    if (nx, ny) in queens:
                        ans.append((nx, ny))
                        break
        return ans
