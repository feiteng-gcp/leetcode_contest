class Solution:
    def queensAttacktheKing(self, queens: List[List[int]], king: List[int]) -> List[List[int]]:
        ret=[]
        ht={}
        for [i,j] in queens:
            if (i,j) not in ht:
                ht[(i,j)]=True
        i=king[0]
        while i>=0:
            if (i,king[1]) in ht:
                ret.append([i,king[1]])
                break
            i-=1
        i=king[0]
        while i<=7:
            if (i,king[1]) in ht:
                ret.append([i,king[1]])
                break
            i+=1
        j=king[1]
        while j>=0:
            if (king[0],j) in ht:
                ret.append([king[0],j])
                break
            j-=1
        j=king[1]
        while j<=7:
            if (king[0],j) in ht:
                ret.append([king[0],j])
                break
            j+=1            
        i=king[0]
        j=king[1]
        
while i>=0 and j>=0:
            if (i,j) in ht:
                ret.append([i,j])
                break
            i-=1
            j-=1
        i=king[0]
        j=king[1]
        
while i<=7 and j>=0:
            if (i,j) in ht:
                ret.append([i,j])
                break
            i+=1
            j-=1
        i=king[0]
        j=king[1]
        
while i>=0 and j<=7:
            if (i,j) in ht:
                ret.append([i,j])
                break
            i-=1
            j+=1
        i=king[0]
        j=king[1]
        
while i<=7 and j<=7:
            if (i,j) in ht:
                ret.append([i,j])
                break
            i+=1
            j+=1            
        return ret
