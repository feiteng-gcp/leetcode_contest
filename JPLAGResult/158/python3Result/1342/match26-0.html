class Solution:
    def queensAttacktheKing(self, queens: List[List[int]], king: List[int]) -> List[List[int]]:
        ki, kj = king
        ans = []
        queens = set((qi, qj) for qi, qj in queens)
        for j in range(kj+1, 8):
            if (ki, j) in queens:
                ans.append([ki, j])
                break
        for j in range(kj-1, -1, -1):
            if (ki, j) in queens:
                ans.append([ki, j])
                break
        for i in range(ki+1, 8):
            if (i, kj) in queens:
                ans.append([i, kj])
                break
        for i in range(ki-1, -1, -1):
            if (i, kj) in queens:
                ans.append([i, kj])
                break
        for di, dj in [[1, -1], [-1, 1], [1, 1], [-1, -1]]:
            ci, cj = ki + di, kj + dj
            while 0 <= ci < 8 and 0 <= cj < 8:
                if (ci, cj) in queens:
                    ans.append([ci, cj])
                    break
                ci += di
                cj += dj
        return ans
