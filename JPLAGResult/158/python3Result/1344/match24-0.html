class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        ans = 0
        Min = Max = 0
        freq = collections.Counter()  
        counter = collections.Counter()  # num appears counter[num] times
        for i, num in enumerate(nums):
            if counter[num]: 
                freq[counter[num]] -= 1
            if Min == counter[num] and freq[counter[num]] == 0:
                Min += 1
            counter[num] += 1
            Min = min(counter[num], Min)
            freq[counter[num]] += 1
            Max = max(counter[num], Max)
            if freq[Max] == 1 and (Max-Min <= 1 or Min == 0):
                ans = i+1
            if Min == 1 and freq[Max]*Max >= i:
                ans = i+1
            #print(freq, Min, Max)
        return ans
