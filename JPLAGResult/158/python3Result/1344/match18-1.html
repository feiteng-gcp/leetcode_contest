from collections import defaultdict


class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        counter, freq = defaultdict(int), defaultdict(int)
        res = 0
        for i, x in enumerate(nums):
            
if counter[x] in freq:
                freq[counter[x]] -= 1
                if freq[counter[x]] == 0:
                    del freq[counter[x]]
            counter[x] += 1
            freq[counter[x]] += 1
            if len(freq) == 2:
                for k, v in freq.items():
                    if v == 1 and (k == 1 or k - 1 in freq): 
                        res = i + 1
            elif len(freq) == 1:
                if 1 in freq or list(freq.values())[0] == 1:
                    res = i + 1
        return res
