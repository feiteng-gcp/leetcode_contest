class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        
        
        n2f = {}
        f2n = {}
        res = 1
        for i,num in enumerate(nums):
            if num not in n2f:
                n2f[num] = 1
                if 1 not in f2n:
                    f2n[1] = {num}
                else:
                    f2n[1].add(num)
            else:
                pf = n2f[num]
                n2f[num] += 1
                f2n[pf].remove(num)
                if len(f2n[pf]) == 0:
                    del f2n[pf]
                if pf+1 not in f2n:
                    f2n[pf+1] = {num}
                else:
                    f2n[pf+1].add(num)
            if len(f2n) == 1:
                if 1 in f2n:
                    res = i+1
                key = list(f2n.keys())[0]
                if len(f2n[key]) == 1:
                    res = i+1
            elif len
(f2n) == 2:
                small, large = min(f2n.keys()), max(f2n.keys())
                if large == small + 1 and len(f2n[large]) == 1:
                    res = i+1
                if small == 1 and len(f2n[small]) == 1:
                    res = i+1
        return res
