class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        num_counts = collections.defaultdict(int)
        count_counts = collections.defaultdict(int)
        
        result = 1
        for i, n in enumerate(nums):
            
if num_counts[n] > 0:
                count_counts[num_counts[n]] -= 1
                if count_counts[num_counts[n]] == 0:
                    del count_counts[num_counts[n]]
                    
            num_counts[n] += 1
            
            count_counts[num_counts[n]] += 1
            if len(count_counts) == 1 and 1 in count_counts:
                result = i + 1
            if len(count_counts) == 1 and 1 in count_counts.values():
                result = i + 1
                
            if len(count_counts) == 2:
                mi = min(count_counts)
                ma = max(count_counts)
                
                if 1 in (mi, ma) and count_counts[1] == 1:
                    result = i + 1
                if mi + 1 == ma and count_counts[ma] == 1:
                    result = i + 1
                    
        return result
