class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        freq = [0] *(10 ** 5 + 1)
        dic = {}
        ret = 0
        for idx, num in enumerate(nums):
            freq[num] += 1
            
            if freq[num] not in dic:
                dic[freq[num]] = set()
            
            if freq
[num] - 1 > 0:
                dic[freq[num] - 1].remove(num)
                if len(dic[freq[num] - 1]) == 0:
                    dic.pop(freq[num] - 1, None)
            
            dic[freq[num]].add(num)
            
            if len(dic) == 2:
                
                freq1, freq2 = list(dic.keys())
                if (abs(freq1-freq2)) == 1 and len(dic[max(freq1, freq2)]) == 1:
                    ret = idx + 1
                elif 1 in dic and len(dic[1]) == 1:
                    ret = idx + 1
                    #print("sdfaf")
                    
            elif len(dic) == 1:
                if 1 in dic:
                    ret = idx + 1
                elif len(dic[list(dic.keys())[0]]) == 1:
                    ret = idx + 1
                    
            #print(dic)
            #print(idx+1, ret)
        return ret
