class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        res = 0
        count = collections.Counter()
        freq = {}
        for i, n in enumerate(nums):
            count[n] += 1
            if count[n] not in freq:
                freq[count[n]] = 1
            else:
                freq[count[n]] += 1
            if count[n] != 1:
                freq[count[n]-1] -= 1
                if freq[count[n]-1] == 0:
                    freq.pop(count[n]-1)        
            if len(freq) == 1 and (1 in freq.values() or 1 in freq.keys()):
                res = i + 1
            elif len(freq) == 2:
                maxi, mini = max(freq.keys()), min(freq.keys())
                if (freq[maxi] == 1 and maxi - mini == 1) or (mini == 1 and freq[mini] == 1):
                    res = i + 1
        return res
