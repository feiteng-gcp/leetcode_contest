class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        ct = dict()
        counters = dict()
        
        best = 0
        for i,n in enumerate(nums):
            if n in counters:
                ct[counters[n]] = ct[counters[n]] - 1
                if not ct[counters[n]]:
                    del ct[counters[n]]
                counters[n] = counters[n] + 1
                ct[counters[n]] = ct.get(counters[n],0) + 1
            else:
                counters[n] = 1
                ct[1] = ct.get(1,0)+1
            if len(ct) == 2:
                items = list(ct.keys())
                if items[0] == items[1]+1 and ct[items[0]] == 1:
                    best = i+1
                elif items[1] == items[0]+1 and ct[items[1]] == 1:
                    best = i+1
                elif 1 in ct and ct[1] == 1:
                    best = i+1
            if len(ct) == 1 and (1 in ct.values() or 1 in ct):
                best = i+1
        return best
