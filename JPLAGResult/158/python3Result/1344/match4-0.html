class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        count = {}
        countNum = {}
        ans = 0
        for i, n in enumerate(nums):       
            if n not in count:
                count[n] = 0
            else:
                countNum[count[n]] = countNum[count[n]] - 1
                if countNum[count[n]] == 0:
                    countNum.pop(count[n])
            count[n] = count[n] + 1;
            if count[n] not in countNum:
                countNum[count[n]] = 0
            countNum[count[n]] = countNum[count[n]] + 1
            if len(countNum) == 1:
                for k,v in countNum.items():
                    if k == 1 or v == 1:
                        ans = i + 1
            if len(countNum) == 2:
                if 1 in countNum and countNum[1] == 1:
                    ans = i + 1
                    continue
                tmp = []
                for k, v in countNum.items():
                    tmp.append([k, v])
                tmp.sort()
                if tmp[0][0] + 1 == tmp[1][0] and tmp[1][1] == 1:
                    ans = i + 1
            # print(countNum, ans, n)
        return ans
