class Solution:
    def maxEqualFreq(self, nums):
        res = 0
        cnt1 = {}
        cnt2 = {}
        for i,a in enumerate(nums):
            if a in cnt1:
                b = cnt1[a]
                cnt2[b] -= 1
                if cnt2[b] == 0:cnt2.pop(b)
                cnt2[b+1] = cnt2.get(b+1,0) + 1
                cnt1[a] += 1
            else:
                cnt1[a] = 1
                cnt2[1] = cnt2.get(1,0) + 1
            if len(cnt2) == 1 and (1 in cnt2 or 1 in cnt2.values()):
                #print(res,cnt1,cnt2)
                res = i+1
            if len(cnt2) == 2:
                x,y = cnt2.keys()
                x,y = sorted([x,y])
                if y-x == 1: 
                    if cnt2[y] == 1:res = i+1
                if x == 1 and cnt2[x] == 1:
                    res = i + 1
        return res
