from collections import defaultdict, Counter
class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        cnt, ans = Counter(), 0
        inv_freq = Counter()
        for i, x in enumerate(nums, 1):
            f = cnt[x]
            if x in cnt:
                inv_freq[f] -= 1
                if inv_freq[f] == 0:
                    inv_freq.pop(f)
            cnt[x] += 1
            inv_freq[f + 1] += 1
            if len(inv_freq) == 1:
                c = list(inv_freq.values())[0]
                if c == 1 or c == i:
                    ans = i
            elif len(inv_freq) == 2:
                f1, f2 = sorted(inv_freq)
                if inv_freq[f1] == f1 == 1:
                    ans = i
                elif f1 + 1 == f2 and inv_freq[f2] == 1:
                    ans = i
        return ans
