class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        count = collections.defaultdict(int)
        sets = collections.defaultdict(set)
        res = 0
        
        for i,v in enumerate(nums):
            if count[v] in sets and v in sets[count[v]]:
                sets[count[v]].discard(v)
                if not sets[count[v]]:
                    sets.pop(count[v])
            count[v] += 1
            sets[count[v]].add(v)
            
            if len(count) == 1:
                res = i + 1
            elif len(sets) == 1 and 1 in sets:
                res = i + 1
            elif len(sets) == 2 and 1 in sets and len(sets[1]) == 1:
                res = i + 1
            elif len(sets) == 2:
                maxV, minV = max(sets.keys()), min(sets.keys())
                if maxV == (minV + 1) and len(sets[maxV]) == 1:
                    res = i + 1
        
        return res
