from collections import defaultdict
class Solution:
    def maxEqualFreq(self, nums):
        def is_valid(i):
            j = i+1 #we've processed j nums already
            # print(j, maxx, num_counts)
            if num_counts[maxx] == 1 and maxx-1 >= 0 and num_counts[maxx-1]*(maxx-1)==(j-maxx):
                return True
            if num_counts[maxx] == (j-1)/maxx and num_counts[1] == 1:
                return True
            if num_counts[1] == j:
                return True
            return False
        maxx = 0
        counts 
= defaultdict(int)
        num_counts = defaultdict(int)
        # ones = 0
        ans = 0
        for i, num in enumerate(nums):
            if counts[num] == 0:
                counts[num] = 1
                num_counts[counts[num]] += 1
                # ones +=1
            else:
                # if counts[num] == 1:
                #   ones -= 1
                num_counts[counts[num]] -= 1
                counts[num] += 1
                num_counts[counts[num]] += 1
            maxx = max(maxx, counts[num])
            if is_valid(i):
                # print("pass")
                ans = i+1
        return ans
