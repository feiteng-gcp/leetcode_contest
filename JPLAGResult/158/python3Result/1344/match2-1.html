class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        cnt = [0] * (len(nums) + 2)
        fre = [0] * (len(nums) + 2)
        res, minn, maxx = 1, 1, 1
        cnt[nums[0]] += 1
        fre[1] += 1
        for i in range(1, len(nums)):
            x = nums[i]
            cnt[x] += 1
            fre[cnt[x]] += 1
            fre[cnt[x]-1] -= 1
            if fre[minn] == 0:
                minn += 1
            if fre[1] != 0:
                minn = 1
            if fre[maxx+1] != 0:
                maxx += 1
            if (minn == maxx and (minn == 1 or fre[minn] == 1)) or (minn + 1 == maxx and fre[maxx] == 1) or (minn == 1 and fre[1] == 1 and maxx * fre[maxx] == i):
                res = i + 1
        return res
