class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        if len(nums)<=1:
            return len(nums)
        
        freqs = collections.defaultdict(int)
        for num in nums:
            freqs[num] += 1
            
        freqsfreqs = collections.defaultdict(int)
        for key in freqs:
            freqsfreqs[freqs[key]] += 1

        n = len(nums)
        for idx in reversed(range(n)):
            #print(dict(freqs), dict(freqsfreqs))
            if len(freqsfreqs) <= 2:
                keys = [key for key in freqsfreqs]
                s, l = min(keys), max(keys)
                if s+1 == l and freqsfreqs[l] == 1:
                    return idx + 1
                if s == 1 and freqsfreqs[s] == 1:
                    return idx + 1
                if s==l and s==1:
                    return idx + 1
                if s==l and freqsfreqs[s] == 1:
                    return idx + 1
            
            num = nums[idx]
            freq = freqs
[num] - 1
            freqs[num] = freq
            freqsfreqs[freq+1] -= 1
            if freqsfreqs[freq+1] == 0:
                del freqsfreqs[freq+1]
            if freq>0:
                freqsfreqs[freq] += 1
        return 1
