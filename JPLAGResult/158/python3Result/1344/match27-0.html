class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        count = collections.Counter(nums) # count[num] = freq
        freqsum = collections.defaultdict(set)
        for num, freq in count.items():
            freqsum[freq].add(num)
        n= len(freqsum)
        l = len(nums)
        if (n == 1 and count[nums[0]] == 1) or len(count) == 1:
            return l
        for i in range(l-1, -1, -1):
            x = nums[i]
            #print(i, x, n,[(a, b) for a, b in count.items()])
            if n == 2:
                #print('ready')
                a, b = sorted(set(count.values()))
                if (b == a + 1 and len(freqsum[b]) == 1) or (a == 1 and len(freqsum[a]) == 1):
                    return i + 1
            
            
            f = count[x]
            count[x] -= 1
            if not count[x]:
                del count[x]
            freqsum[f].remove(x)
            if not freqsum[f]:
                del freqsum[f]
                n -= 1
            if f - 1 != 0:
                if not freqsum[f-1]:
                    n += 1
                freqsum[f-1].add(x)
        return 2
