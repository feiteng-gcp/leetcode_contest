class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        
        tot = collections.Counter(nums)
        
        cnt_dic = collections.defaultdict(set)
        for k in tot.keys():
            cnt_dic[tot[k]].add(k)
            
        for i in range(len(nums)-1,-1,-1):
            maxkey = max(cnt_dic.keys())
            if (len(tot)==1) or (len(cnt_dic.keys())==1 and 1 in cnt_dic) or (len(cnt_dic.keys())==2 and ((1 in cnt_dic and len(cnt_dic[1])==1) or (maxkey-1 in cnt_dic and len(cnt_dic[maxkey])==1))):
                #print(maxkey,tot,cnt_dic,i+1)
                return i+1
                
            icnt = tot[nums[i]]
            cnt_dic[icnt].remove(nums[i])
            if not cnt_dic[icnt]:
                del cnt_dic[icnt]
                
            if icnt>1:
                cnt_dic[icnt-1].add(nums[i])
                tot[nums[i]] -= 1
            else:
                del tot[nums[i]]
                
        return 1
