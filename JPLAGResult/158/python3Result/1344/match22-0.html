class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        
        out = 0
        if len(nums) >= 2:
            out = 1
        
        number_occurences = dict()
        value_occurences = dict()
        
        for i, n in enumerate(nums):
            if n in number_occurences.keys():
                number_occurences[n] += 1
            else:
                number_occurences[n] = 1
                
            new_val_occ = number_occurences[n]
            
            if new_val_occ > 1:
                if value_occurences[new_val_occ-1] > 1:
                    value_occurences[new_val_occ-1] -= 1
                else:
                    value_occurences.pop(new_val_occ-1)
 
            if new_val_occ in value_occurences.keys():
                value_occurences[new_val_occ] += 1
            else:
                value_occurences[new_val_occ] = 1
                
            
            if len(value_occurences) == 1:
                
                k1 = list(value_occurences.keys())[0]
                if k1 == 1:
                    out = max(out, i)
                    
            if len(number_occurences) == 1:
                out = max(out, i)
                
                            
            if len(value_occurences) == 2:
                k1, k2 = list(value_occurences.keys())
                
                if k1 + 1 == k2 and value_occurences[k2]==1:
                    out = max(out, i)
                    
                if k2 + 1 == k1 and value_occurences[k1]==1:
                    out = max(out, i)
                    
                if k1 == 1 and value_occurences[k1]==1:
                    out = max(out, i)
                    
                if k2 == 1 and value_occurences[k2]==1:
                    out = max(out, i)
                    
        return out + 1
