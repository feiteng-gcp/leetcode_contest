class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        c = collections.Counter(nums)
        cc = collections.Counter(c.values())
        for i in range(len(nums) - 1, -1, -1):
            if len(c) == 1:
                return i + 1
            if len(cc) == 1 and cc[1] > 0:
                return i + 1
            if len(cc) == 2:
                a, b = cc.keys()
                a, b = min(a,b), max(a, b)
                if a == 1 and cc[a] == 1 or b - a == 1 and cc[b] == 1:
                    return i + 1
            oc = c[nums[i]] 
            c[nums[i]] -= 1
            cc[oc] -= 1
            if cc[oc] == 0:
                del cc[oc]
            if oc > 1:
                cc[oc - 1] += 1     
        return 1
