class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        c = collections.Counter()
        rc = collections.defaultdict(set)
        count = 0
        maxcount = 0
        for i in nums:
            count += 1
            c[i] += 1

            if c[i] == 1:
                rc[1].add(i)
            else:
                j = c[i]
                rc[j].add(i)
                rc[j - 1].discard(i)
                if len(rc[j - 1]) == 0:
                    rc.pop(j - 1)

            if len(rc) == 1:
                if 1 in rc:   # all members appeared once
                    maxcount = count
                j = max(rc)
                if len(rc[j]) == 1: # only one member exist and appeared multiple times
                    maxcount = count
            elif len(rc) == 2:
                j = max(rc)
                if len(rc[j]) == 1 and j - 1 in rc:   # 111222333[3]
                    maxcount = count
                elif 1 in rc and len(rc[1]) == 1:     # 111222333[4]
                    maxcount = count
            #print(i, rc)

        return maxcount
