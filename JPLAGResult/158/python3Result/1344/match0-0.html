class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        cnt = collections.Counter()
        freq = collections.Counter()
        f = 0
        c = 0
        res = 0
        for i in nums:
            c += 1
            cnt[i] += 1
            freq[cnt[i]-1] -= 1
            if freq[cnt[i]-1] == 0:
                del freq[cnt[i]-1]
            freq[cnt[i]] += 1
            f = max(f,cnt[i])
            if f == 1:
                res = c
            if f*freq[f] == c-1:
                res = max(res,c)
            if (f-1)*(freq[f-1]+1) == c-1:
                res = max(res,c)
        return res
