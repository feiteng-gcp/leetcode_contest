class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        '''
        maintain a data structure, such that update and query happens in O(1):
        * update: enter a number, maintain the frequency of such number (fn) and the frequency of frequencies (ff)
        * query: determine whether only two numbers in ff table and one value +1 of the other value
        
        fn is a Counter.
        ff is a Counter with key removal.
        '''
        
        from collections import Counter
        fn = Counter()
        ff = Counter()
        
        ans = 1
        for i, n in enumerate(nums):
            old = fn[n]
            fn[n] += 1
            new = fn[n]
            if old > 0:
                ff[old] -= 1
                if ff[old] == 0:
                    del ff[old]
            ff[new] += 1
