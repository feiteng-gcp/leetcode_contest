class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        cnter = collections.Counter()
        cnts = {}
        
        current = 2
        for i, v in enumerate(nums):
            c = cnter[v]
            if c > 0:
                cnts[c].remove(v)
                if len(cnts[c]) == 0:
                    cnts.pop(c)
            cnter[v] = c + 1
            cnts[c+1] = cnts.get(c+1, set())
            cnts[c+1].add(v)
            
            # print(cnts)
            
            if len(cnts) == 1:
                k = list(cnts.keys())[0]
                if k == 1 or len(cnts[k]) == 1:
                    current = i + 1
                    continue
                
            if len(cnts) == 2:
                if cnts.get(1) and len(cnts[1]) == 1:
                    current = i + 1
                    continue
                
                max_k = max(cnts.keys())
                min_k = min
(cnts.keys())
                if max_k == min_k + 1 and len(cnts[max_k]) == 1:
                    current = i + 1
        return current
