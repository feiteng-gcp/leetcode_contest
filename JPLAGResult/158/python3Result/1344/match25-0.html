class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        res = 0
        pq = []
        
        freq = collections.Counter()
        maxx = 0
        
        for i, n in enumerate(nums):
            freq[n] += 1
            maxx = max(maxx, freq[n])
            heapq.heappush(pq, [freq[n], n])
            while pq and freq[pq[0][1]] > pq[0][0]:
                heapq.heappop(pq)
            minn = pq[0][0]
            # print(i, n, minn, maxx)
            if minn == 1 and i == maxx * (len(freq) - 1) or i == minn * len(freq) or i == maxx - 1:
                res = max(res, i + 1)
        #print('\n')
        return res
