class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        c = collections.Counter()
        cc = collections.Counter()
        maxi = 0
        pos = 0
        for num in nums:
            pos += 1 
            oc = c[num];
            if(oc):
                cc[oc] -= 1
                if(not cc[oc]):
                    cc.pop(oc)
            c[num]+=1
            cc[oc+1] += 1
            #print(cc)
            istrue = False
            if(len(cc)<=2 and cc[1]==1):
                istrue = True
            elif(len(cc)==2 and cc[max(cc.keys())] == 1 and max(cc.keys()) == min(cc.keys())+1):
                istrue = True
            elif(len(cc) == 1 and (1 in cc or 1 in cc.values())):
                istrue = True
            if(istrue):
                maxi = pos
        return maxi
