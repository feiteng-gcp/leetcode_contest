class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        c1 = collections.Counter()
        c2 = collections.Counter()
        ans = 0
        for i, v in enumerate(nums):
            c1[v] += 1
            if c1[v] - 1 > 0:
                c2[c1[v] - 1] -= 1
                if c2[c1[v] - 1] == 0:
                    del c2[c1[v] - 1]
            c2[c1[v]] += 1
            if len(c2) == 1 and (list(c2.keys())[0] == 1 or list(c2.values())[0] == 1):
                ans = i + 1
            elif len(c2) == 2:
                cand = sorted(c2.items())
                if cand[0][0] + 1 == cand[1][0] and cand[1][1] == 1:
                    ans = i + 1
                elif cand[0][0] == 1 and cand[0][1] == 1:
                    ans = i + 1
        return ans
