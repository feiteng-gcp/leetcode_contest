class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        freq = [set() for i in range(100001)]
        num_freq = collections.defaultdict(int)
        not_zero = set()
        res = 1
        for i, num in enumerate(nums):
            # print(i, '--before', freq[num_freq[num]])
            if num in freq[num_freq[num]]:
                freq[num_freq[num]].discard(num)
                # print('removed', freq[num_freq[num]])
                if len(freq[num_freq[num]]) == 0:
                    not_zero.discard(num_freq[num])
            num_freq[num] += 1
            freq[num_freq[num]].add(num)
            # print('after', freq)
            not_zero.add(num_freq[num])
            # print(not_zero, num_freq)
            if len(not_zero)==2:
                arr = list(not_zero)
                if abs(arr[0]-arr[1])==1 and len(freq[max(arr)])==1:
                    res = max(res, i+1)
                if min(arr)==1 and len(freq[min(arr)]) == 1:    
                    res = max(res, i+1)
                if len(freq[max(arr)]) == 1 and len(freq[min(arr)]) == 1:    
                    res = max(res, i+1)
            if len(not_zero)==1:
                arr = list(not_zero)
                if arr[0] == 1 or len(freq[arr[0]])==1:
                    res = max(res, i+1)
        return res
