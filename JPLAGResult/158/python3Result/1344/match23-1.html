class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        res = 0
        cnt = collections.defaultdict(int)
        rcnt = collections.defaultdict(int)
        s = set()
        g = 0
        
        for i, num in enumerate(nums):
            if num not in cnt:
                g += 1
            cnt[num] += 1
            rcnt[cnt[num]] += 1
            
if cnt[num] > 1:
                rcnt[cnt[num] - 1] -= 1
                if rcnt[cnt[num] - 1] == 0:
                    s.remove(cnt[num] - 1)
            s.add(cnt[num])
            if g == 1:
                res = i + 1
            if len(s) == 1 and 1 in s:
                res = i + 1
            if len(s) == 2:
                if 1 in s and g > 1 and i // (g - 1) in s and i % (g - 1) == 0:
                    res = i + 1
                c = i // g
                if rcnt[c + 1] == 1:
                    res = i + 1
        
        return res
