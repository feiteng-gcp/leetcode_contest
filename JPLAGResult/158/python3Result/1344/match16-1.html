class Solution:
    def maxEqualFreq(self, nums: List[int]) -> int:
        D = {}
        C = {}
        t = 0
        res = 2
        for n in nums:
            if n in D:
                C[D[n]] -= 1
                if not C[D[n]]:
                    C.pop(D
[n])
                D[n] += 1
                C[D[n]] = C.get(D[n], 0) + 1
            else:
                D[n] = 1
                C[D[n]] = C.get(D[n], 0) + 1
            t += 1
            if len(C) == 1:
                if tuple(C.values()) == (1,) or 1 in C:
                    res = t
            elif len(C) == 2:
                mi = min(C.keys())
                if 1 in C and C[1] == 1:
                    res = t
                elif mi + 1 in C and C[mi + 1] == 1:
                    res = t
        return res
