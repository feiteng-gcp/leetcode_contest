class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        dp = [[0] * (n + 1) for _ in range(6)]
        for i in range(1, n+1):
            for j in range(0, 6):
                for k in range(1, min(i, rollMax[j]) + 1):
                    if k == i:
                        dp[j][i] += 1
                    else:    
                        dp[j][i] += sum(dp[m][i-k] for m in range(0, 6) if m != j)
                    dp[j][i] %= 10 ** 9 + 7
        return sum(dp[j][n] for j in range(0, 6)) % (10 ** 9 + 7)
