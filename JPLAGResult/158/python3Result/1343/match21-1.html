class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        st = [[0] * rollMax[d] for d in range(6)]
        for d in range(6):
            st[d][0] = 1
            
        cnt = 0
        last_sum = [1] * 6
        for _ in range(2, n+1):
            n_st = [[0] * rollMax[d] for d in range(6)]
            n_last_sum = [0] * 6
            total = 0
            for d in range(6):
                for v in range(rollMax[d]):
                    if v == 0:
                        c = sum(last_sum) - last_sum[d]
                    else:
                        c = st[d][v-1]
                    c %= 1000000007
                    
                    n_st[d][v] = c
                    n_last_sum[d] += c
                    total += c
            st = n_st
            last_sum = n_last_sum
            cnt = total % 1000000007
        return cnt
