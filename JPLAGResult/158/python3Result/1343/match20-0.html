class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        dp1 = [[1 for k in range(rollMax[j])] for j in range(6)]
        for i in range(6):
            for k in range(1, len(dp1[i])):
                dp1[i][k] = 0
        dp2 = [[0 for k in range(rollMax[j])] for j in range(6)]
        for _ in range(n - 1):
            s = [sum(v) for v in dp1]
            _s = sum(s)
            for i in range(6):
                dp2[i][0] = _s - s[i]
                for k in range(1, len(dp2[i])):
                    dp2[i][k] = dp1[i][k - 1]
            dp1, dp2 = dp2, dp1
        MOD = 10 ** 9 + 7
        return sum([sum(v) for v in dp1]) % MOD
