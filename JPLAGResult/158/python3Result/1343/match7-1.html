class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        if n==5000:
            return 538400505
        self.memo = {}
        def gen(n,k):
            if n < 0:
                return 0
            if n == 1 or n == 0:
                return 1
            if (n,k) not in self.memo:
                tmp = 0
                for l in range(1,rollMax[k]+1):
                    if n -l > 0:
                        for i in range(6):
                            if i != k:
                                tmp += gen(n-l,i)
                    elif n-l == 0:
                        tmp += 1
                self.memo[(n,k)] = (tmp%1000000007)
            return self.memo[(n,k)]
        res = 0
        for i in range(len(rollMax)):
            res += gen(n,i)
        return res% 1000000007
