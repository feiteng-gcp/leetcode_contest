from functools import lru_cache

class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        MAX = 10**9 + 7
        max_roll = max(rollMax)
        @lru_cache(None)
        def inner(prev_roll, prev_count, rem):
            if rem == 1:
                ans = 0
                for d in range(6):
                    if d != prev_roll:
                        ans += 1
                    elif prev_count + 1 <= rollMax[d]:
                        ans += 1
                
return ans
            ans = 0
            for d in range(6):
                if d != prev_roll:
                    ans += inner(d, 1, rem-1)
                elif d == prev_roll and prev_count + 1 <= rollMax[d]:
                    ans += inner(d, prev_count+1, rem-1)
            return ans % MAX
        return inner(-1, 0, n) % MAX
