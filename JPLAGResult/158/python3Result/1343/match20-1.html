class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        mod = 1000000007
        rnc = [[0]*i for i in rollMax]
        for i in range(6):
            rnc[i][0] = 1
        nn = 1
        while nn<n:
            tmp = [[0]*i for i in rollMax]
            sum_lis 
= [sum(n) for n in rnc]
            sum_t = sum(sum_lis)
            for i in range(6):
                tmp[i][0] = (sum_t - sum_lis[i])%mod
                for j in range(1,len(tmp[i])):
                    tmp[i][j] = rnc[i][j-1]
            nn += 1
            rnc = tmp
        return sum([sum(n) for n in rnc])% mod
