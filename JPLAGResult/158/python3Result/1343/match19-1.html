"""
dp[i][x][k]: 前i个合规则, 最后有k个x的合法数
ans = sum(dp[i][x][k], 0 < k <= rollMax[x])
"""

class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        INF = 1000000000 + 7
        dp = [[[0]*16 for x in range(6)] for i in range(n)]
        for x in range(6):
            dp[0][x][1] = 1
            
        for i in range(1, n):
            s = sum([sum(dp[i-1][x]) for x in range(6)]) % INF
            for x in range(6):
                sx = sum(dp[i-1][x]) % INF
                dp[i][x][1] = (s + INF - sx) % INF 
                
for k in range(2, rollMax[x]+1):
                    dp[i][x][k] = dp[i-1][x][k-1]
        
        ans = sum([sum(dp[n-1][x]) for x in range(6)]) % INF
        return ans
