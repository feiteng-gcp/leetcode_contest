class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        f = [[0] * 6 for _ in range(n + 1)]
        tot = [0] * (n + 1)
        for i in range
(1, n + 1):
            for j in range(6):
                for k in range(1, rollMax[j] + 1):
                    if i == k:
                        f[i][j] += 1
                    elif i > k:
                        f[i][j] += tot[i - k] - f[i - k][j]
            tot[i] = sum(f[i])

        return sum(f[n]) % (10**9 + 7)
