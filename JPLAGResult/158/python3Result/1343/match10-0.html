class Solution:
    def dieSimulator(self, n: int, rollMax) -> int:
        cnt = collections.defaultdict(int)
        cnt[(-1,0)] = 1
        for i in range(n):
            tmp = collections.defaultdict(int)
            for j,k in list(cnt.keys()):
                for l in range(6):
                    if j == l:
                        if k == rollMax[l]:continue
                        tmp[(j,k+1)] += cnt[(j,k)]
                    else:
                        tmp[(l,1)] += cnt[(j,k)]
            cnt = tmp
        return sum(cnt.values()) % (10**9 + 7)
