class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        cnt = [[0] * rollMax[i] for i in range(6)]
        for i in range(6):
            cnt[i][0] = 1
        total = [1] * 6
        m = 10 ** 9 + 7
        for _ in range(n - 1):
            tmp = [[0] * rollMax[i] for i in range(6)]
            tmp_t = [0] * 6
            sum_ = sum(total)
            for i in range(6):
                tmp[i][0] = (sum_ - total[i]) % m
                for j in range(1, rollMax[i]):
                    tmp[i][j] = cnt[i][j - 1]
            for i in range(6):
                tmp_t[i] = sum(tmp[i]) % m
            cnt = tmp
            total = tmp_t
        # print(cnt, total)
        return sum(total) % m
