from functools import lru_cache

class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        MOD = 10**9 + 7
        @lru_cache(maxsize=None)
        def impl(n, d, t):
            # for seq with length n, and the fact that previously dd...d(in total t) appears before current number.
            if n == 1:
                return 5 + (1 if t < rollMax[
d - 1] else 0)
            ans = 0
            for i in range(1, 7):
                if i != d:
                    ans += impl(n - 1, i, 1)
                elif t < rollMax[d - 1]:
                    ans += impl(n - 1, d, t + 1)
            return ans % MOD
        return impl(n, 1, 0)
