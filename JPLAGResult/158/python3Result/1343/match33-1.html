class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        nums 
= [[0] * 15 for _ in range(6)]
        total = [1]* 6
        for j in range(6):
            nums[j][0] = 1
        for i in range(n-1):
            T = sum(total)
            for j in range(6):
                for k in range(rollMax[j]-1, -1, -1):
                    if k == 0:
                        nums[j][0] = T - total[j]
                    else:
                        nums[j][k] = nums[j][k-1]
            for j in range(6):
                total[j] = sum(nums[j])
            
        return sum(total)%(10**9+7)
