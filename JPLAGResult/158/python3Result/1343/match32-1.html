class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        '''
        DP.
        
        State: f[n][p][o]
        * sequence length n
        * ending with point p (1-6)
        * consecutive #o p's (<=15)
        
        transition:
        * o>1: f[n-1][p][o-1]
        * o=1: f[n-1][~p][*]
        
        ans: sum_p,o{f[n][p][o]}
        
        n only relies on n-1 --> use rolling array
        '''
        
        def init_f():
            f = []
            f.append([0] + [0] + [0] * 15)
            for _ in range(6):
                f.append([0] + [1] + [0] * 15)
            return f
        
        mod = 10**9 + 7
            
        rm = [0] + rollMax
        f = init_f()
        for _ in range(1, n):
            nf = init_f()
            for p in range(1, 7):
                tmp = 0
                for pp in range(1, 7):
                    if pp != p:
                        tmp += sum(f[
pp])
                nf[p][1] = tmp % mod
                for o in range(2, rm[p] + 1):
                    nf[p][o] = f[p][o - 1]
            f = nf
        
        ans = sum(sum(p) for p in f) % mod
        return ans
