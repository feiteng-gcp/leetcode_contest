class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        maxNum = max(rollMax)
        dp = [[0 for _ in range(maxNum+1)] for _ in range(6)]
        for i in range(6):
            dp[i][0] = 1
        curSum = 6 
        for i in range(1, n):
            temp = [row[:] for row in dp]
            tempSum = 0
            for j in range(6):
                for k in range(min(i, rollMax[j]-1), -1, -1):
                    if k > 0:
                        temp[j][k] = dp[j][k-1]
                    else:
                        temp[j][k] = curSum - sum(dp[j])
                    tempSum += temp[j][k]
            curSum = tempSum
            dp = temp
        return curSum % (10**9 + 7)
