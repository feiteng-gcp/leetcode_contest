class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        mod = 10 ** 9 + 7
        
        dp 
= [[0] * 15 for _ in range(6)]
        for i in range(6):
            dp[i][0] = 1
            
        for _ in range(n - 1):
            new_dp = [[0] * 15 for _ in range(6)]
            for i in range(6):
                for j in range(6):
                    if i != j:
                        new_dp[i][0] += sum(dp[j])
                    else:
                        for k in range(1, rollMax[i]):
                            new_dp[i][k] += dp[i][k - 1]
            dp = new_dp
        
        return sum([sum(d) for d in dp]) % mod
