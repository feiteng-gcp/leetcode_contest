class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        mod = 1000000007
        stats = [[[0]*15 for _ in range(6)] for _ in range(n)]
        for i in range(6):
            stats[0][i][0] = 1
        preSum = 6
        for k in range(n-1):
            for j in range(6):
                stats[k+1][j][0] = preSum - sum(stats[k][j])
            for j in range(6):
                for m in range(rollMax[j]-1):
                    stats[k+1][j][m+1] = stats[k][j][m]
            preSum = sum(sum(stats[k+1][l]) for l in range(6)) % mod
        #print(stats[n-1])
        return sum(sum(stats[n-1][l]) for l in range(6)) % mod
