class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        mp = {(0,0):1}
        for i in range(n):
            newmp = collections.defaultdict(int)
            for (num,times),ct in mp.items
():
                for j in range(1,7):
                    if num!=j:
                        newmp[(j,1)]+=ct
                    else: #num==j
                        if times+1<=rollMax[j-1]:
                            newmp[(j,times+1)]+=ct
            for k in newmp:
                newmp[k] = newmp[k]%(10**9+7)
            mp = newmp
        res = 0
        for v in mp.values():
            res+=v
        return res%(10**9+7)
