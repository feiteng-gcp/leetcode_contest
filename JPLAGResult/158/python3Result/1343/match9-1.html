from collections import defaultdict
class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        modulo = 10 ** 9 + 7
        state = defaultdict(int)
        state[(0, 0)] = 1
            
        for i in range(n):
            next_state = defaultdict(int)
            for num, seq_l in state:
                if rollMax[num] > seq_l:
                    next_state[(num, seq_l + 1)] += state[(num, seq_l)]
                    # next_state[(num, seq_l + 1)] %= modulo
                for next_num in range(6):
                    if next_num != num:
                        next_state
[(next_num, 1)] += state[(num, seq_l)]
                        # next_state[(next_num, 1)] %= modulo
            state = next_state
        return sum(state.values()) % modulo
