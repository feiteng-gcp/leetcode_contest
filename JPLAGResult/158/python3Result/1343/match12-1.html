class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        
        #each one have maximum of rollmax chances - or we can just set it to 15.
        
        MOD = 10**9 + 7
        
        dp = [[0]*16 for _ in range(6)] #the 15th position is used as tot.
        for a in range(6):
            dp[a][0] = 1
            dp
[a][-1] = 1
        
        for i in range(1,n):
            new_dp = [[0]*15 for _ in range(6)]
            for a in range(6):
                #1st: add every others up.
                for b in range(6):
                    if b!=a:
                        new_dp[a][0] += (dp[b]
[-1]%MOD)
                
                #2nd: add itself up.
                for k in range(1,rollMax[a]):
                    new_dp[a][k] = dp[a][k-1]
                
                #3rd: set the sum bit.
                tot = sum(new_dp[a])%MOD
                new_dp[a][-1] = tot
                #print(a, tot)
            
            #print(new_dp)
            dp = new_dp
            
        return sum([dp[a][-1] for a in range(6)])%MOD
