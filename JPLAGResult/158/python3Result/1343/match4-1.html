from functools import lru_cache
class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        rollMax = [0] + rollMax
        @lru_cache(None)
        def helper(n, last=0):
            if n == 0: return 1
            return sum(
                helper(n - d, last=i)
                for i in range(1, 7) if i != last
                for d in range(1, min(n, rollMax[i]) + 1) 
            ) % int(1e9 + 7)
        return helper(n) % int(1e9 + 7)
