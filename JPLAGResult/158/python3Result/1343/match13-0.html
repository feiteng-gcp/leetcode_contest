class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        dp = [[0] * 15 for _ in range(6)]
        for i in range(6):
            dp[i][0] = 1
        for i in range(1, n):
            newDP = [[0] * 15 for _ in range(6)]
            for j in range(6):
                for k in range(6):
                    if k != j:
                        newDP[j][0] += sum(dp[k])
                for k in range(1, rollMax[j]):
                    newDP[j][k] += dp[j][k - 1]
            dp = newDP
        return sum(sum(row) for row in dp) % (10 ** 9 + 7)
