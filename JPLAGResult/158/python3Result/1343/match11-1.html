class Solution:
    def dieSimulator(self, n: int, rollMax: List[int]) -> int:
        maxroll = max(rollMax)
        dp = [[0 for y in range(maxroll)] for x in range(6)]
        for x in range(6):
            dp[x][0] = 1
        print(dp)
        
        for epoch in range(n-1):
            new_dp = [[0 for y in range(maxroll)] for x in range(6)]
            for numminus in range(6):
                for endcount in range(rollMax[numminus]):
                    if endcount == 0:
                        new_dp[numminus][endcount] = sum([y if x!=numminus else 0 for x in range(6) for y in dp
[x]])
                    else:
                        new_dp[numminus][endcount] = dp[numminus][endcount-1] 
            dp = new_dp
            #print(epoch)
            #print(dp)
            
        return sum([y for x in dp for y in x]) % (10**9 + 7)
