class Solution:
    def dieSimulator(self, n: int, rollMax) -> int:
        cnt = collections.defaultdict(int)
        cnt[(-1,0)] = 1
        for i in range(n):
            tmp = collections.defaultdict(int)
            s = [0] * 7
            for j,k in cnt:
                s[6] += cnt[(j,k)]
                if 0<= j < 6:
                    s[j] += cnt[(j,k)]
            # for j,k in cnt:
            #     for l in range(6):
            #         if j == l:
            #             if k == rollMax[l]:continue
            #             tmp[(j,k+1)] += cnt[(j,k)]
            #         else:
            #             tmp[(l,1)] += cnt[(j,k)]
            for l in range(6):
                for r in range(1, rollMax[l]+1):
                    if r == 1:
                        tmp[(l,1)] = s[6] - s[l]
                    elif (l,r-1) not in cnt:
                        continue
                    else:
                        tmp
[(l,r)] = cnt[(l,r-1)]                    
            cnt = tmp
        return sum(cnt.values()) % (10**9 + 7)
