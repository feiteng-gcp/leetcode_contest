class Solution {
    	public List<List<Integer>> queensAttacktheKing(int[][] queens, int[] king) {
		int[][] dp = new int[8][8];
		for (int [] q : queens) {
			int x = q[0];
			int y = q[1];
			dp[x][y] = 1;

		}
		List<List<Integer>> result = new ArrayList<>();
		for (int i = 1; king[0] + i < 8; i++) {
			if (dp[king[0] + i][king[1]] == 1) {
				result.add(Arrays.asList(king[0] + i, king[1]));
				break;
			}
		}
		for (int i = 1; king[0] - i >= 0; i++) {
			if (dp[king[0] - i][king[1]] == 1) {
				result.add(Arrays.asList(king[0] - i, king[1]));
				break;
			}
		}
		for (int i = 1; king[1] + i < 8; i++) {
			if (dp[king[0]][king[1] + i] == 1) {
				result.add(Arrays.asList(king[0], king[1] + i));
				break;
			}
		}
		for (int i = 1; king[1] - i >= 0; i++) {
			if (dp[king[0]][king[1] - i] == 1) {
				result.add(Arrays.asList(king[0], king[1] - i));
				break;
			}
		}
