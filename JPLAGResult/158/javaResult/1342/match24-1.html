class Solution {
   public List<List<Integer>> queensAttacktheKing(int[][] queens, int[] king) {
    	
    	Set<Integer> qs = new HashSet<Integer>();
    	for(int i = 0; i < queens.length; i++) {
    		qs.add(queens[i][0] * 10 + queens[i][1]);
    	}

    	
    	List<List<Integer>> ans = new ArrayList<>();
    	
    	int x = king[0];
    	int y = king[1];
    	
    	for(int i = x; i >= 0; i--) {
    		if(qs.contains(i * 10 + y)) {

    			List<Integer> hit = new ArrayList<>();
    			hit.add(i);
    			hit.add(y);
    			ans.add(hit);
    			break;
    		}
    	}
    	for(int i = x; i < 8; i++) {
    		if(qs.contains(i * 10 + y)) {

    			List<Integer> hit = new ArrayList<>();
    			hit.add(i);
    			hit.add(y);
    			ans.add(hit);
    			break;
    		}
    	}
    	for(int i = y; i >= 0; i--) {
    		if(qs.contains(x * 10 + i)) {

    			List<Integer> hit = new ArrayList<>();
    			hit.add(x);
    			hit.add(i);
    			ans.add(hit);
    			break;
    		}
    	}
    	for(int i = y; i < 8; i++) {
    		if(qs.contains(x * 10 + i)) {

    			List<Integer> hit = new ArrayList<>();
    			hit.add(x);
    			hit.add(i);
    			ans.add(hit);
    			break;
    		}
    	}
    	for(int i = x, j = y; i < 8 && j < 8; i++, j++) {
    		if(qs.contains(i * 10 + j)) {

    			List<Integer> hit = new ArrayList<>();
    			hit.add(i);
    			hit.add(j);
    			ans.add(hit);
    			break;
    		}
    	}
    	for(int i = x, j = y; i < 8 && j >= 0; i++, j--) {
    		if(qs.contains(i * 10 + j)) {

    			List<Integer> hit = new ArrayList<>();
    			hit.add(i);
    			hit.add(j);
    			ans.add(hit);
    			break;
    		}
    	}
    	for(int i = x, j = y; i >= 0 && j < 8; i--, j++) {
    		if(qs.contains(i * 10 + j)) {

    			List<Integer> hit = new ArrayList<>();
    			hit.add(i);
    			hit.add(j);
    			ans.add(hit);
    			break;
    		}
    	}
    	for(int i = x, j = y; i >= 0 && j >= 0; i--, j--) {
    		if(qs.contains(i * 10 + j)) {

    			List<Integer> hit = new ArrayList<>();
    			hit.add(i);
    			hit.add(j);
    			ans.add(hit);
    			break;
    		}
    	}    	
    	
        return ans;
    }
}
