class Solution {
    public List<List<Integer>> queensAttacktheKing(int[][] queens, int[] king) {
        int[][] grid=new int[8][8];
        for(int[] q:queens){
            grid[q[0]][q[1]]=1;
        }
        List<List<Integer>> ret=new ArrayList<>();
        int x=king[0],y=king[1];
        for(int i=x;i>=0;i--){
            if(grid[i][y]==1){
                List<Integer> list=new ArrayList<>();
                list.add(i);
                list.add(y);
                ret.add(list);
                break;
            }
        }
        for(int i=x;i<8;i++){
            if(grid[i][y]==1){
                List<Integer> list=new ArrayList<>();
                list.add(i);
                list.add(y);
                ret.add(list);
                break;
            }
        }
        for(int j=y;j>=0;j--){
            if(grid[x][j]==1){
                List<Integer> list=new ArrayList<>();
                list.add(x);
                list.add(j);
                ret.add(list);
                break;
            }
        }
        for(int j=y;j<8;j++){
            if(grid[x][j]==1){
                List<Integer> list=new ArrayList<>();
                list.add(x);
                list.add(j);
                ret.add(list);
                break;
            }
        }
        for(int i=x,j=y;i>=0&&i<8&&j>=0&&j<8;i--,j--){
            if(grid[i][j]==1){
                List<Integer> list=new ArrayList<>();
                list.add(i);
                list.add(j);
                ret.add(list);
                break;
            }
        }
        for(int i=x,j=y;i>=0&&i<8&&j>=0&&j<8;i++,j--){
            if(grid[i][j]==1){
                List<Integer> list=new ArrayList<>();
                list.add(i);
                list.add(j);
                ret.add(list);
                break;
            }
        }
        for(int i=x,j=y;i>=0&&i<8&&j>=0&&j<8;i--,j++){
            if(grid[i][j]==1){
                List<Integer> list=new ArrayList<>();
                list.add(i);
                list.add(j);
                ret.add(list);
                break;
            }
        }
        for(int i=x,j=y;i>=0&&i<8&&j>=0&&j<8;i++,j++){
            if(grid[i][j]==1){
                List<Integer> list=new ArrayList<>();
                list.add(i);
                list.add(j);
                ret.add(list);
                break;
            }
        }
        return ret;
    }
}
