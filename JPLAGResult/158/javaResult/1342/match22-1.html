class Solution {
    public List<List<Integer>> queensAttacktheKing(int[][] queens, int[] king) {
        boolean[][] board = new boolean[8][8];
        for(int[] q : queens){
            board[q[0]][q[1]] =true;
        }
        int x = king[0];
        int y = king[1];
        ArrayList<List<Integer>> res = new ArrayList<>();
        //row
        
for(int i=1;i+y<8;i++){
            if(board[x][y+i]) {
                ArrayList<Integer> pair = new ArrayList<>();
                pair.add(x);
                pair.add(y+i);
                res.add(pair);
                break;
            }
        }
        for(int i=1;y-i>=0;i++){
            if(board[x][y-i]) {
                ArrayList<Integer> pair = new ArrayList<>();
                pair.add(x);
                pair.add(y-i);
                res.add(pair);
                break;
            }
        }
        //col
        for(int i=1;i+x<8;i++){
            if(board[x+i][y]) {
                ArrayList<Integer> pair = new ArrayList<>();
                pair.add(x+i);
                pair.add(y);
                res.add(pair);
                break;
            }
        }
        for(int i=1;x-i>=0;i++){
            if(board[x-i][y]) {
                ArrayList<Integer> pair = new ArrayList<>();
                pair.add(x-i);
                pair.add(y);
                res.add(pair);
                break;
            }
        }
        //d1
        for(int i=1;i+y<8&&i+x<8;i++)
{
            if(board[x+i][y+i]) {
                ArrayList<Integer> pair = new ArrayList<>();
                pair.add(x+i);
                pair.add(y+i);
                res.add(pair);
                break;
            }
        }
        for(
int i=1;y-i>=0&&x-i>=0;i++){
            if(board[x-i][y-i]) {
                ArrayList<Integer> pair = new ArrayList<>();
                pair.add(x-i);
                pair.add(y-i);
                res.add(pair);
                break;
            }
        }
        //d2
        for(int i=1;i+y<8&&x-i>=0;i++)
{
            if(board[x-i][y+i]) {
                ArrayList<Integer> pair = new ArrayList<>();
                pair.add(x-i);
                pair.add(y+i);
                res.add(pair);
                break;
            }
        }
        for(
int i=1;y-i>=0&&x+i<8;i++){
            if(board[x+i][y-i]) {
                ArrayList<Integer> pair = new ArrayList<>();
                pair.add(x+i);
                pair.add(y-i);
                res.add(pair);
                break;
            }
        }
        return res;
    }
}
