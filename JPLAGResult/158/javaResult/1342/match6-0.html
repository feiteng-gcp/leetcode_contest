class Solution {
    public List<List<Integer>> queensAttacktheKing(int[][] qs, int[] k) {
        List<List<Integer>> r = new ArrayList<>();
        
        boolean f[][] = new boolean[8][8];
        
        
        for(int[] q:qs){
            f[q[0]][q[1]] = true;
            
        }
        
        int dx[] = {0,-1,-1,-1,0,1,1,1};
        int dy[] = {-1,-1,0,1,1,1,0,-1};
        
        
        for(int j=0;j<8;++j){
            
            int nx = k[0];
            int ny = k[1];
            
            while(nx>=0&&ny>=0&&nx<8&&ny<8){
                if(f[nx][ny]){
                    r.add(Arrays.asList(nx,ny));
                    break;
                }
                nx += dx[j];
                ny += dy[j];
                
            }
            
            
            
        }
        
        
        return r;
    }
}
