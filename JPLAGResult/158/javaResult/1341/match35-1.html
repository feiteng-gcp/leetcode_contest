class Solution {
    public int balancedStringSplit(String s) {
        
        int res  = 0;
        int offset = 0;
        for(int i=0; i<s.length(); i++) {
            if(s.charAt(i) == 'L') offset ++;
            else 
offset--;
            
            if(offset == 0)
                res++;
        }
        return res;
    }
}
