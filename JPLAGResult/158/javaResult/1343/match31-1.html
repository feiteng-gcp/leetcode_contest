class Solution {
    private int MOD = 1_000_000_000 + 7;
    public int dieSimulator(int n, int[] rollMax) {
        long[][][] dp = new long[n+1][16][7];
        
        for (int i = 1; i <= 6; i++) dp[1][1][i] = 1;
        
        for (int i = 2; i <= n; i++) {
            for (int j = 1; j <= 6; j++) {
                for (int k = 1; k <= 6; k++) {
                    if (k == j) continue;
                    
for (int x = 1; x < 16; x++) {dp[i][1][j] += dp[i-1][x][k]; dp[i][1][j] %= MOD;}
                }
                for (int x = 2; x <= rollMax[j-1]; x++) {
                    dp[i][x][j] += dp[i-1][x-1][j];
                    dp[i][x][j] %= MOD;
                }
            }
        
}
        long ans = 0;
        for (int i = 1; i <= 6; i++) {
            for (int k = 1; k <= rollMax[i-1]; k++) {
                ans += dp[n][k][i];
                ans %= MOD;
            }
        }
        return (int)ans;
    }
}
