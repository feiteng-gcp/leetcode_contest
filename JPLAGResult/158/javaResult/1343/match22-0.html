class Solution {
    private static final int MOD = 1000000007;
    
    public int dieSimulator(int n, int[] rollMax) {
        int[][][] dp = new int[n + 1][6][16];
        for (int d = 0; d < 6; d++) {
            dp[1][d][1] = 1;
        }
        
        for (int i = 1; i < n; i++) {
            for (int d = 0; d < 6; d++) {
                for (int c = 0; c < rollMax[d]; c++) {
                    dp[i + 1][d][c + 1] += dp[i][d][c]; // Consecutive
                    dp[i + 1][d][c + 1] %= MOD;
                }
                for (int c = 0; c < 16; c++) {
                    for (int d2 = 0; d2 < 6; d2++) {
                        if (d2 == d) {
                            continue;
                        }
                        dp[i + 1][d2][1] += dp[i][d][c];
                        dp[i + 1][d2][1] %= MOD;
                    }
                }
            }
        }
        
        int total = 0;
        for (int d = 0; d < 6; d++) {
            for (int c = 0; c < 16; c++) {
                total += dp[n][d][c];
                total %= MOD;
            }
        }
        
        return total;
    }
}
