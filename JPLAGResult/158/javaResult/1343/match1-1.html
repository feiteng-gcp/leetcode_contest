class Solution {
    public int dieSimulator(int n, int[] rollMax) {
        if (n == 0) return 0;
        int max = 0;
        for (int m : rollMax) max = Math.max(m, max);
        if (max == 0) return 0;
        int M = 1000000007;
        int[][] dp = new int[6][max + 2];
        for (int i = 0; i < 6; i++) dp[i][1] = 1;
        
for (int l = 1; l < n; l++) {
            int[][] next = new int[6][max + 2];
            for (int i = 0; i < 6; i++) {
                for (int j = 0; j <= rollMax[i]; j++) {
                    for (int k = 0; k < 6; k++) {
                        if (k == i) {
                            next[k][j + 1] += dp[i][j];
                            next[k][j + 1] %= M;
                        }
                        else {
                            next[k][1] += dp[i][j];
                            next[k][1] %= M;
                        
}
                    }
                }
            }
            dp = next;
        }
        int res = 0;
        for (int i = 0; i < 6; i++) {
            for (int j = 0; j <= rollMax[i]; j++) {
                res += dp[i][j];
                // System.out.printf("%d %d %d\n", i, j, dp[i][j]);
                res %= M;
            }
        }
        return res;
    }
}
