class Solution {
    public int dieSimulator(int n, int[] rollMax) {
        int divisor = 1000000007;
        int[][][] dp = new int[6][][];
        for (int number = 0; number < 6; number++) {
            dp[number] = new int[rollMax[number] + 1][];
            for (int consecutive = 0; consecutive <= rollMax[number]; consecutive++) {
                dp[number][consecutive] = new int[n + 1];
            }
        }
        for (int number = 0; number < 6; number++) {
            dp[number][1][1] = 1;
        }
        for (int roll = 2; roll <= n; roll++) {
            for (int number = 0; number < 6; number++) {
                
for (int nextNumber = 0; nextNumber < 6; nextNumber++) {
                    if (nextNumber == number) {
                        for (int consecutive = 1; consecutive < rollMax[number]; consecutive++) {
                            dp[number][consecutive + 1][roll] += dp[number][consecutive][roll - 1];
                            dp[number][consecutive + 1][roll] %= divisor;
                        }
                    } else {
                        for (int consecutive = 1; consecutive <= rollMax[number]; consecutive++) {
                            dp[nextNumber][1][roll] += dp[number][consecutive][roll - 1];
                            dp[nextNumber][1][roll] %= divisor;
                        
}
                    }
                }
            }
        }
        int sum = 0;
        for (int number = 0; number < 6; number++) {
            for (int consecutive = 1; consecutive <= rollMax[number]; consecutive++) {
                sum += dp[number][consecutive][n];
                sum %= divisor;
            }
        }
        return sum;
    }
}
