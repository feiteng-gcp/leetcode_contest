class Solution {
    public int dieSimulator(int n, int[] rollMax) {
        int MOD = 1000_000_007;
        int[][][] dp = new int[n+1][6][15]; // ith roll ends with dice d of j consecutive times;
        
        for (int i = 0; i < 6; ++i)
            dp[0][i][0] = 1;
        
        for (int i = 1; i < n; i ++) {
            for (int d = 0; d < 6; ++d) {
                for (int d1 = 0; d1 < 6; ++d1) {
                    if(d == d1) {
                        for (int j = 1; j < 15 && j < rollMax[d]; ++j) {
                            dp[i][d][j] += dp[i-1][d][j-1];
                            dp[i][d][j] %= MOD;
                        }
                    }
                    else {
                        for (int j = 0; j < 15; ++j) {
                            dp[i][d][0] += dp[i-1][d1][j];
                            dp[i][d][0] %= MOD;
                        }
                    }
                }
            }   
        }
        
        int ans = 0;
        for (int i = 0; i < 6; ++i) {
            for (int j = 0; j < 15; ++j) {
                ans += dp[n-1][i][j];
                ans %= MOD;
            }
        }
        
        return ans;
        
        
    }
}
