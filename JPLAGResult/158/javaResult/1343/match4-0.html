class Solution {
    public int dieSimulator(int n, int[] rollMax) {
        int m = 1000000007;
        long[][] prev = new long[7][16];
        for (int i = 1; i < 7; i ++) {
            prev[i][0] = 1;
        }
        for (int e = 1; e < n; e ++) {
            long[][] dp = new long[7][16];
            for (int i = 1; i < 7; i ++) {
                for (int j = 0; j < rollMax[i - 1]; j ++) {
                    if (j == 0) {
                        for (int k = 1; k < 7; k ++) {
                            if (k == i) {
                                continue;
                            }
                            for (int kk = 0; kk < 16; kk ++) {
                                dp[i][j] += prev[k][kk] % m;                            
                            }
                        }
                    } else {
                        dp[i][j] = prev[i][j - 1] % m;
                    }
                }
            }
            prev = dp;
        }
        long sum = 0;
        for (int i = 1; i < 7; i ++) {
            for (int j = 0; j < rollMax[i - 1]; j ++) {
                sum += prev[i][j] % m;
            }
        }
        return (int)(sum % m);
    }
}
