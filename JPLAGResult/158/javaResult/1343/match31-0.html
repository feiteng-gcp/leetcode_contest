public class Solution {
	private static final int MOD = 1_000_000_007;
	public int dieSimulator(int n, int[] rollMax) {
		long[][] dp = new long[6][15];
		for (int i = 0; i < 6; ++i) {
			dp[i][0] = 1;
		}
		for (int i = 0; i < n - 1; ++i) {
			long[][] nDP = new long[6][16];
			long[] sum = new long[6];
			for (int r = 0; r < 6; ++r) {
				for (long num : dp[r]) {
					sum[r] += num;
				}
			}
			for (int r = 0; r < 6; ++r) {
				int maxReps = rollMax[r];
				for (int c = 1; c < maxReps; ++c) {
					nDP[r][c] = dp[r][c - 1];
				}
				for (int r1 = 0; r1 < 6; ++r1) {
					if (r1 == r) continue;
					nDP[r][0] += sum[r1];
				}
				nDP[r][0] %= MOD;
			}
			dp = nDP;
		}
		long ans = 0;
		for (int r = 0; r < 6; ++r) {
			for (long num : dp[r]) {
				ans += num;
			}
		}
		return (int)(ans % MOD);
	}
}
