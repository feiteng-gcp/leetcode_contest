class Solution {
	private long MOD = 1000_000_007;	
	
	private long[][][] dp;	
	
    public int dieSimulator(int n, int[] rollMax) {
    	dp = new long[n][7][20];
    	long ans = 0;
    	for (int i = 0;i < 6;i ++) {    		
    		dp[0][i][1] = 1;    		
    	}
    	for (int i = 1;i < n;i ++) {    		
    		for (int roll = 0;roll < 6;roll ++) {    			
    			for (int count = 1;count <= 15;count ++) {
    				// same    				
    				if (count + 1 <= rollMax[roll]) {    					
    					dp[i][roll][count + 1] = (dp[i - 1][roll][count] + dp[i][roll][count + 1]) % MOD;
    				}
    				// diff
    				for (int r = 0;r < 6;r ++) {    					
    					if (r != roll) {
    						dp[i][r][1] = (dp[i - 1][roll][count] + dp[i][r][1]) % MOD;    						
    					}
    				}
    			}
    		}    		
    	}
    	for (int roll = 0;roll < 6;roll ++) {
    		for (int count = 1;count <= 15;count ++) {
    			ans = (ans + dp[n - 1][roll][count]) % MOD;
    		}
    	}
    	return (int) ans;
    }
}
