class Solution {
    public int dieSimulator(int n, int[] rollMax) {
        int mod = 1_000_000_007;
        int[][][] dp = new int[n + 1][6][16];
        dp[0][0][1] = 1;
        dp[0][1][1] = 1;
        dp[0][2][1] = 1;
        dp[0][3][1] = 1;
        dp[0][4][1] = 1;
        dp[0][5][1] = 1;
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < 6; j++) {
                for (int p = 0; p < 6; p++) {
                    if (p == j) {
                        continue;
                    }
                    for (int k = 1; k <= rollMax[p]; k++) {
                        dp[i][j][1] += dp[i - 1][p][k];
                        dp[i][j][1] %= mod;
                    }
                }
                for (int k = 2; k <= rollMax[j]; k++) {
                    dp[i][j][k] = dp[i - 1][j][k - 1];
                    dp[i][j][k] %= mod;
                }
            }
        }
        int ret = 0;
        for (int i = 0; i < 6; i++) {
            for (int j = 1; j <= rollMax[i]; j++) {
                ret += dp[n - 1][i][j];
                ret %= mod;
            }
        }
        return ret;
    }
}
