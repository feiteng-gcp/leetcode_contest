class Solution {
    public int dieSimulator(int n, int[] rollMax) {
        
        long dp[][] = new long[6][16];
        
        for(int i=0;i<6;++i){
            dp[i][1] = 1;
        }
        long p = 1000000007;
        
        for(int i=2;i<=n;++i){
            long ndp[][] = new long[6][16];
            for(int j=0;j<6;++j){
                for(int k=1;k<=15;++k){
                    for(int z=0;z<6;++z){
                        if(z!=j){
                            ndp[z][1]+= dp[j][k];
                            ndp[z][1] %= p;
                        }else{
                            if(k<rollMax[j]){
                                
                                ndp[z][k+1]+= dp[j][k];
                                ndp[z][k+1] %= p;
                            }    
                            
                            
                        }
                    }
                }
                
                
            }
            dp = ndp;
            
        }
        long s =0;
        for(int i=0;i<6;++i){
            for(int j=0;j<16;++j){
                s += dp[i][j];
                s %= p;
            }
        }
        return (int)s;
        
        
        
        
    }
}
