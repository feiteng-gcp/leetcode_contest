class Solution {
    public int dieSimulator(int n, int[] rollMax) {
        int mod = 1000000007;
        int[][][] dp = new int[n+1][6][16];
        
for (int i = 0; i < 6; ++i) {
            dp[1][i][1] = 1;
        }
        for (int i = 2; i < n + 1; ++i) {
            for (int j = 0; j < 6; ++j) {
                for (int k = 1; k <= 15; ++k) {
                    if (k > rollMax[j]) {
                        dp[i][j][k] = 0;
                    } else if (k == 1) {
                        int sum = 0;
                        for (int jj = 0; jj < 6; ++jj) {
                            
for (int kk = 1; kk <= 15; ++kk) {
                                if (jj != j) {
                                    sum = (sum + dp[i-1][jj][kk]) % mod;
                                }
                            }
                        }
                        dp[i][j][k] = sum;
                    } else {
                        dp[i][j][k] = dp[i-1][j][k-1];
                    }
                }
            }
        }
        int ret = 0;
        
for (int j = 0; j < 6; ++j) {
            for (int k = 1; k <= 15; ++k) {
                ret = (ret + dp[n][j][k]) % mod;
            }
        }
        return ret;
    }
}
