class Solution {
    long MOD = 1000000007;
    public int dieSimulator(int n, int[] rollMax) {
        long[][][] T = new long[n][6][15 + 1];
        
        for (int d = 0; d < 6; d++) {
            T[0][d][1] = 1;
        }
        
        for (int i = 1; i < n; i++) {
            for (int newNum = 0; newNum < 6; newNum++) {
                for (int inarow = 2; inarow <= 15; inarow++) {
                    if (inarow > rollMax[newNum]) 
                        T[i][newNum][inarow] = 0;
                    else
                        T[i][newNum][inarow] = (T[i-1][newNum][inarow - 1]);
                }
                // inarow = 1
                long total = 0;
                for (int d = 0; d < 6; d++) {
                    if (d == newNum) continue;
                    for (int j = 1; j <= 15; j++) {
                        total = (total + T[i-1][d][j]) % MOD;
                    }
                }
                T[i][newNum][1] = total;
            }
        }
        
        // for (int i = 0; i < n; i++) {
        //     for (int d = 0; d < 6; d++) {
        //         for (int j = 0; j <= 15; j++) {
        //             System.out.print(T[i][d][j] + " ");
        //         }
        //         System.out.print(" || ");
        //     }
        //     System.out.println();
        // }
        
        
        long result = 0;
        for (int d = 0; d < 6; d++) {
            for (int r = 1; r <= 15; r++) {
                result = (result + T[n-1][d][r]) % MOD;
            }
        }
        return (int)result;
    }
}
