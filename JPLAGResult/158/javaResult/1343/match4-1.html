class Solution {
    public int dieSimulator(int n, int[] rollMax) {
        long mod = (long) 1e9 + 7;
        
        long[][][] dp = new long[n][6][16];
        for (int j = 0; j < 6; j++) {
            dp[0][j][1] = 1;
        }
        for (int i = 1; i < n; i++) {
            
for (int j = 0; j < 6; j++) {
                for (int k = 1; k <= rollMax[j]; k++) {
                    if (k == 1) {
                        for (int jj = 0; jj < 6; jj++) {
                            if (jj != j) {
                                for (int kk = 1; kk < 16; kk++){
                                    dp[i][j][k] += dp[i-1][jj][kk];
                                    dp[i][j][k] %= mod;
                                }
                            }
                        
}
                    } else {
                        dp[i][j][k] = dp[i-1][j][k-1];
                    }
                }
            }
        
}
        long sum = 0;
        for (int j = 0; j < 6; j++) {
            for (int k = 0; k <= rollMax[j]; k++) {
                sum += dp[n-1][j][k];
                sum %= mod;
            }  
        }
        return (int) sum;
    }
}
