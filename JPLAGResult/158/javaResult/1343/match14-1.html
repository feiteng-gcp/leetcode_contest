class Solution {
 private static long MOD = 1_000_000_007L;

    public int dieSimulator(int n, int[] rollMax) {
        long[][][] dp = new long[n + 1][6][16];

        
for (int i = 0; i < 6; i++) {
            dp[0][i][1] = 1;
        }

        for (int i = 1; i < n; i++) {
            for (int j = 0; j < 6; j++) {
                for (int k = 1; k <= rollMax[j]; k++) {
                    if (k > 1) {
                        dp[i][j][k] += dp[i - 1][j][k - 1];
                    } else {
                        for (int x = 0; x < 6; x++) {
                            if (x == j) continue;

                            for (int y = 1; y <= rollMax[x]; y++) {
                                dp[i][j][k] += dp[i - 1][x][y];
                            }
                        
}
                    }

                    dp[i][j][k] %= MOD;
                }
            }
        }


        long ans = 0;
        for (int x = 0; x < 6; x++) {
            for (int y = 1; y <= 15; y++) {
                ans += dp[n - 1][x][y];
                ans %= MOD;
            }
        }
        return (int) ans;

    }

}
