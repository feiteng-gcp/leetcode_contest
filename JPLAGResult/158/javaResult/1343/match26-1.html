class Solution {
    public int dieSimulator(int n, int[] rollMax) {
        int dp[][][] = new int[n + 1][7][16];
        final int mod = 1000000007;
        
        
for (int i = 1; i <= 6; ++i) {
            dp[0][i][1] = 1;
        }
        
        for (int i = 1; i < n; ++i) {
            for (int j = 1; j <= 6; ++j) {
                for (int k = 1; k < rollMax[j - 1]; ++k) {
                    dp[i][j][k + 1] = (dp[i][j][k + 1] + dp[i - 1][j][k]) % mod;
                } 
                
                for (int o = 1; o <= 6; ++o) {
                    if (o == j) {
                        continue;
                    
}
                    
                    for (int k = 1; k <= 15; ++k) {
                        dp[i][o][1] = (dp[i - 1][j][k] + dp[i][o][1]) % mod;
                    }
                }
            
}
        }
        
        int total = 0;
        for (int j = 1; j <= 6; ++j) {
            for (int k = 1; k <= 15; ++k) {
                total = (total + dp[n - 1][j][k]) % mod;
            }
        }
        return total;
    }
}
