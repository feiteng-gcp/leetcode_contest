class Solution {
public int dieSimulator(int n, int[] rollMax) {
        int[][][] dp = new int[n+1][6][20];
        int mod = 1000000007;
        for (int i = 0; i < 6; i++) {
            dp[0][i][1] = 1;
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < 6; j++) {
                for (int k = 0; k < 6; k++) {
                    if (k != j) {
                        for (int l = 1; l <= rollMax[k]; l++) {
                            dp[i][j][1] += dp[i - 1][k][l];
                            dp[i][j][1] %= mod;
                        }
                    } else {
                        for (int l = 1; l < rollMax[k]; l++) {
                            dp[i][j][l+1] += dp[i - 1][k][l];
                            dp[i][j][l+1] %= mod;
                        }
                    }
                }
            }
        }
        
        int res = 0;
        for (int j = 0; j < 6; j++) {
            for (int l = 1; l <= rollMax[j]; l++) {
                res += dp[n-1][j][l];
                res %= mod;
            }
        }
        return res;
    }
}
