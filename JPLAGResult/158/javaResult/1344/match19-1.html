class Solution {
    
    public int maxEqualFreq(int[] nums) {
        HashMap<Integer,Integer> map=new HashMap<>();
        TreeMap<Integer,Integer> fr=new TreeMap<>();
        int ans=0;
        for(int i=0;i<nums.length;i++){
            map.put(nums[i],map.getOrDefault(nums[i],0)+1);
            int f=map.get(nums[i]);
            if(f-1!=0){
                fr.put(f-1,fr.get(f-1)-1);
                if(fr.get(f-1)==0){
                    fr.remove(f-1);
                }
            }
            fr.put(f,fr.getOrDefault(f,0)+1);
            if(fr.size()==2&&((fr.get(fr.lastKey())==1&&(fr.lastKey()-fr.firstKey())==1)||(fr.get(fr.firstKey())==1&&fr.firstKey()==1))){
                ans=i+1;
            
}
            if(fr.size()==1&&fr.firstKey()==1){
                ans=i+1;
            }
            if(fr.size()==1&&map.size()==1){
                ans=i+1;
            }
            // debug(i+1);
            // debug(map);
            // debug(fr);
        }
        return ans;
    }
    public static void debug(Object... o) {
        //if(!oj)
        System.out.println(Arrays.deepToString(o));
    }
}
