class Solution {
   public int maxEqualFreq(int[] nums) {

        //num-> freq
        HashMap<Integer,Integer> map =new HashMap<>();
        //freq -> num
        HashMap<Integer, HashSet<Integer>> freq=new HashMap<>();
        for (int num: nums){
            map.put(num,map.getOrDefault(num,0)+1);
        }
        for (int num: map.keySet()){
            freq.putIfAbsent(map.get(num),new HashSet<>());
            freq.get(map.get(num)).add(num);
        }

        // System.out.println(freq);
        // System.out.println(map);
        int res =1;
        for (int i=nums.length-1;i>=0;i--){
            if (freq.size()==1){
                int a=0;
                for (int val: freq.keySet()){
                    a = val;
                }
                if (a==1||freq.get(a).size()==1) return i+1;
            }
            if (freq.size()==2){
                int[] temp =new int[2];
                int j=0;
                for (int val: freq.keySet()){
                    temp[j++] = val;
                }
                int a =Math.max(temp[0],temp[1]);
                int b =Math.min(temp[0],temp[1]);
                // System.out.println("a="+a+" b="+b);
                // System.out.println(freq.get(a).size());
                // System.out.println(freq.get(b).size());
                if ((a-b==1&&freq.get(a).size()==1)||(b==1&&freq.get(b).size()==1))
                    return i+1;
            }
            int num = nums[i];
            int curfreq = map.get(num);
            freq.get(curfreq).remove(num);
            map.put(num,map.get(num)-1);
            if (freq.get(curfreq).size()==0)
                freq.remove(curfreq);
            if (curfreq==1) continue;
            freq.putIfAbsent(curfreq-1,new HashSet<>());
            freq.get(curfreq-1).add(num);
            //System.out.println(freq);

        }
        return res;
    }
}
