class Solution {
    public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer> freqByKey = new HashMap<>();
        TreeMap<Integer, Set<Integer>> keysByFreq = new TreeMap<>();
        int output = 0;
        for (int i = 0; i < nums.length; i++) {
            int key = nums[i];
            int freq = freqByKey.getOrDefault(key, 0);
            freqByKey.put(key, freq + 1);
            if (freq == 0) {
                keysByFreq.computeIfAbsent(1, k -> new HashSet<>()).add(key);
            } else {
                keysByFreq.get(freq).remove(key);
                if (keysByFreq.get(freq).isEmpty()) {
                    keysByFreq.remove(freq);
                }
                keysByFreq.computeIfAbsent(freq + 1, k -> new HashSet<>()).add(key);
            }

            if (freqByKey.size() == 1) {
                output = i + 1;
            } else 
if (keysByFreq.size() == 1 && keysByFreq.firstKey() == 1) {
                output = i + 1;
            } else if (keysByFreq.size() == 2) {
                int lowFreq = keysByFreq.firstKey();
                int highFreq = keysByFreq.lastKey();
                if (highFreq == lowFreq + 1 && keysByFreq.get(highFreq).size() == 1) {
                    output = i + 1;
                } else if (lowFreq == 1 && keysByFreq.get(lowFreq).size() == 1) {
                    output = i + 1;
                
}
            }
        }
        return output;
    }
}
