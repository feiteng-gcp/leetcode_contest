class Solution {
    public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer> counts = new HashMap<>();
        Map<Integer, Set<Integer>> buckets = new HashMap<>();

        int max = 0;

        for (int i = 0; i < nums.length; i++) {
            int bucket = counts.getOrDefault(nums[i], 0) + 1;
            Integer oldBucket = counts.put(nums[i], bucket);

            if (oldBucket != null) {
                if (buckets.get(oldBucket).size() == 1) 
{
                    buckets.remove(oldBucket);
                } else {
                    buckets.get(oldBucket).remove(nums[i]);
                }
            }

            Set<Integer> set = buckets.get(bucket);
            if (set == null) {
                set = new HashSet<>();
                buckets.put(bucket, set);
            }
            set.add(nums[i]);

            if (
check(buckets)) {
                max = i + 1;
            }
        }

        return max;
    }

    private boolean check(Map<Integer, Set<Integer>> buckets) {
        if (buckets.size() == 1 && (buckets.keySet().iterator().next() == 1 || buckets.values().iterator().next().size() == 1)) return true;

        if (buckets.size() != 2) return false;

        List<Integer> list = new ArrayList<>(buckets.keySet());
        int minBucket = Math.min(list.get(0), list.get(1));
        int maxBucket = Math.max(list.get(0), list.get(1));

        if (maxBucket - minBucket == 1 && buckets.get(maxBucket).size() == 1) {
            return true;
        }

        if (minBucket == 1 && buckets.get(minBucket).size() == 1) {
            return true;
        }

        return false;
    }
}
