class Solution {
    public int maxEqualFreq(int[] nums) {
        int numc = 0;
        
Map<Integer, Integer> map = new HashMap<>();
        TreeMap<Integer, Set<Integer>> count = new TreeMap<>();
        int ans = 0;
        int i = 0;
        for (int n : nums) {
            i++;
            map.putIfAbsent(n, 0);
            map.put(n, map.get(n) + 1);
            count.putIfAbsent(map.get(n), new HashSet<>());
            count.get(map.get(n)).add(n);
            if (map.get(n) > 1) {
                count.get(map.get(n) - 1).remove(n);
                if (count.get(map.get(n) - 1).size() == 0) count.remove(map.get(n) - 1)
;
            }
            if (count.size() == 1 && count.firstEntry().getValue().size() == 1) {
                ans = i;
                continue;
            }
            if (count.size() > 3) break;
            if (count.size() == 1 && count.get(1) != null) ans = i;
            if (count.size() == 2 && 
                ((count.firstEntry().getValue().size() == 1 && count.firstKey()==1) || (
count.lastEntry().getValue().size() == 1 && count.lastKey() - count.firstKey() == 1 ))) {
                ans = i;
            }
        }
        return ans;
    }
}
