class Solution {
   public int maxEqualFreq(int[] nums) {
        int[] num = new int[100005], cnt = new int[100005];
        
Set<Integer> set = new HashSet<>();
        int max = 1, res = 1;
        for(int i = 0; i < nums.length; i++){
            if(set.contains(nums[i]))
                cnt[num[nums[i]]] --;
            set.add(nums[i]);
            num[nums[i]] ++;
            cnt[num[nums[i]]] ++;
            // System.out.println(Arrays.toString(cnt));
            max = Math.max(max, num[nums[i]]);
            if(cnt[max - 1] == set.size() - 1 
               || (cnt[max] == set.size() - 1 && cnt[1] == 1)
                || (i == 1 && set.size() == 2)
                || (cnt[max] == 
set.size() && max == 1))
                res = Math.max(res, i + 1);
        }
        return res;
    }
}
