class Solution {
    public int maxEqualFreq(int[] nums) {
        int[] map = new int[100002];
        Map<Integer,Integer> count = new HashMap<> ();
        Set<Integer> appears = new HashSet<> ();

        int max = 0, t = 0;
        for (int i: nums) {
            t++;
            int curr = map[i];
            map[i]++;

            appears.add(i);
            if (curr != 0) {
                count.put(curr, count.get(curr) - 1);
                if (count.get(curr) == 0) {
                    count.remove(curr);
                }
            }

            count.put(curr+1, count.getOrDefault(curr+1, 0) + 1);

            int numSize = count.size();
            int tsize = appears.size();

            // Every ele appears once
            if (tsize == 1) {
                max = t;
            }
            else if (tsize == t) {
                max = t;
            }
            else {
                // if some element only appears once
                if (count.containsKey(1)) {
                    int div = (t - 1) / (tsize - 1);
                    if (div * (tsize - 1) == t - 1) {
                        if (count.containsKey(div) && count.get(div) == tsize - 1) {
                            max = t;
                        }
                    }
                }

                // t - 1 must total divide by tsize
                int div = (t - 1) / tsize;
                if (div * tsize == t - 1) {
                    if (count.containsKey(div) && count.get(div) == (tsize - 1) && count.containsKey(div+1) && count.get(div+1) == 1) {
                        max = t;
                    }
                }
            }
        }

        return max;
    }
}
