public class Solution {
    public int maxEqualFreq(int[] nums) {
        int maxFreq = 0, ans = 0;
        Map<Integer, Integer> map = new HashMap<>();
        Map<Integer, Integer> freqs = new HashMap<>();
        for (int i = 0; i < nums.length; ++i) {
            int origFreq = map.getOrDefault(nums[i], 0);
            int freq = origFreq + 1;
            maxFreq = Math.max(maxFreq, freq);
            map.put(nums[i], freq);

            if (origFreq > 0) {
                freqs.put(origFreq, freqs.get(origFreq) - 1);
                if (freqs.get(origFreq) == 0) {
                    freqs.remove(origFreq);
                
}
            }

            freqs.put(freq, freqs.getOrDefault(freq, 0) + 1);
            if (freqs.size() == 1) {
                if (freq == 1 || freqs.get(freq) == 1)
                    ans = i + 1;
            } else if (freqs.size() == 2) {
                List<Integer> fList = new ArrayList<>(freqs.keySet());
                Collections.sort(fList);
                if (freqs.get(fList.get(0)) == 1 && fList.get(0) == 1) ans = i + 1;
                else 
if (freqs.get(fList.get(1)) == 1 && fList.get(1) == fList.get(0) + 1) {
                    ans = i + 1;
                }
            }
//            if (i == 11) {
//                System.out.println(map);
//                System.out.println(freqs);
//            }
        }
        return ans;
    }
}
