class Solution {
    public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        Map<Integer, Integer> freq = new HashMap<>(); 
        int ret = 1; 
        for (int i = 0; i < nums.length; i++) {
            map.put(nums[i], map.getOrDefault(nums[i], 0) + 1);
            int f = map.get(nums[i]);
            freq.put(f, freq.getOrDefault(f, 0) + 1);
            if (freq.containsKey(f - 1)) {
                
freq.put(f - 1, freq.get(f - 1) - 1);
                if (freq.get(f - 1) == 0) {
                    freq.remove(f - 1);
                }
            }
            if (
freq.keySet().size() == 2) {
                int[] arr = new int[2]; 
                
                int idx = 0;
                for (int t : freq.keySet()) {
                    arr[idx++] = t;
                }
                
                Arrays.sort(arr);
                // System.out.println(i + " " + arr[0] + " " + arr[1]);
                if ((arr[0] == 1 && freq.get(1) == 1) || (arr[0] + 1 == arr[1] && freq.get(arr[1]) == 1)) {
                    ret = i + 1;
                }
            } else 
if (map.keySet().size() == 1) {
                ret = i + 1;
            } else if (freq.keySet().size() == 1) {
                for (int t : freq.keySet()) {
                    if (t == 1) {
                        ret = i + 1;
                    }
                }
            }
        }
        return ret;
    }
}
