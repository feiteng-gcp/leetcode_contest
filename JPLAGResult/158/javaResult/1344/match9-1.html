class Solution {
    public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        Map<Integer, Integer> cnt = new HashMap<>();
        
        int ret = 1;
        for(int i = 0; i < nums.length; i++) {
            int occur = 
cnt.containsKey(nums[i]) ? cnt.get(nums[i]) : 0;
            if (occur > 0) {
                map.put(occur, map.get(occur) - 1);
                if(map.get(occur) == 0) map.remove(occur);
            }
            
            cnt.put(nums[i], occur + 1);
            if(map.containsKey(occur + 1)) map.put(occur + 1, map.get(occur + 1) + 1);
            else map.put(occur + 1, 1);
            
            if(map.keySet().size() == 2) {
                Iterator<Integer> it = map.keySet().iterator();
                
                int cnt1 = it.next();
                int num1 = map.get(cnt1);
                
                int cnt2 = it.next();
                int num2 = map.get(cnt2);
                
                if(((cnt1 == cnt2 + 1 || cnt1 == 1) && num1 == 1) || ((cnt2 == cnt1 + 1 || cnt2 == 1) && num2 == 1)) ret = i + 1;
            } else if(map.keySet().size() == 1) {
                int cntt = map.keySet().iterator().next();
                int numm = map.get(cntt);
                if(cntt == 1 || numm == 1) ret = i + 1
;
            }
        }
        
        return ret;
    }
}
