class Solution {
    public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer> freq = new HashMap<>();
        Map<Integer, Set<Integer>> map = new HashMap<>();

        int res = 1;

        for(int i = 0; i < nums.length; i++) {
            int num = nums[i];

            freq.put(num, freq.getOrDefault(num, 0) + 1);

            int f = freq.get(num);
            if(f != 1) {
                map.get(f - 1).remove(num);
                if(map.get(f - 1).size() == 0) {
                    map.remove(f - 1);
                }
            }

            if(!map.containsKey(f)) {
                map.put(f, new HashSet<>());
            }
            map.get(f).add(num);

            if(isValid(map)) {
                res = i;
            }

        }

        return res + 1;
    }

    private boolean isValid(Map<Integer, Set<Integer>> map) {
        int size = map.size();
        if(size == 1) {
            int key = 0;
            for(int k : map.keySet()) {
                key = k;
            }

            return key == 1 || map.get(key).size() == 1;
        }
        if(size > 2) {
            return false;
        }

        int lowKey = -1;
        int highKey = -1;

        for(int k : map.keySet()) {
            if (lowKey == -1) {
                lowKey = k;
            } else {
                highKey = k;
            }
        }
        if(lowKey > highKey) {
            int tmp = lowKey;
            lowKey = highKey;
            highKey = tmp;
        }

        return (map.get(highKey).size() == 1 && highKey == lowKey + 1) || (lowKey == 1 && map.get(lowKey).size() == 1) ;
    }
}
