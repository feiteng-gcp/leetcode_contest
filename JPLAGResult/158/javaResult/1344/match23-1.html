class Solution {
    private boolean check(Map<Integer, Set<Integer>> reverseMap) {
        if (reverseMap.size() == 2) {
            List<Integer> l = new ArrayList<>(reverseMap.keySet());
            l.sort((a, b)->(a-b));
            int x = l.get(0);
            int y = l.get(1);
            if (x + 1 == y) {
                if (reverseMap.get(y).size() == 1) {
                    return true;
                }
            }
            if (x == 1) {
                if (reverseMap.get(x).size() == 1) {
                    return true;
                }
            }
        }
        if (reverseMap.size() == 1) {
            for (int k : reverseMap.keySet()) {
                if (k == 1) {
                    return true;
                }
                if (reverseMap.get(k).size() == 1) {
                    return true;
                }
            }
        }
        return false;
    }
    
    
public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer> countMap = new HashMap<>();
        Map<Integer, Set<Integer>> reverseMap = new HashMap<>();
        for (int n : nums) {
            int count = 
countMap.getOrDefault(n, 0) + 1;
            countMap.put(n, count);
        }
        for (int n : countMap.keySet()) {
            int count = countMap.get(n);
            Set<Integer> set = 
reverseMap.getOrDefault(count, new HashSet<>());
            set.add(n);
            reverseMap.put(count, set);
        }
        for (int i = nums.length - 1; i >= 0; --i) {
            if (check(reverseMap)) {
                return i + 1;
            }
            int n = nums[i];
            int count = countMap.get(n) - 1;
            countMap.put(n, count);
            
reverseMap.get(count + 1).remove(n);
            if (reverseMap.get(count + 1).size() == 0) {
                reverseMap.remove(count + 1);
            }
            if (count != 0) {
                Set<Integer> set = 
reverseMap.getOrDefault(count, new HashSet<>());
                set.add(n);
                reverseMap.put(count, set);
            }
        }
        return 1;
    }
}
