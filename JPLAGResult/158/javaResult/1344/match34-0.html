class Solution {
    public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        Map<Integer, Integer> countFreq = new HashMap<>();
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            int cnt = map.getOrDefault(nums[i], 0);
            map.put(nums[i], cnt + 1);
            Integer freq = countFreq.get(cnt);
            if (freq != null) {
                countFreq.put(cnt, freq - 1);
                if (countFreq.get(cnt) == 0) {
                    countFreq.remove(cnt);
                }
            }
            countFreq.put(cnt + 1, countFreq.getOrDefault(cnt + 1, 0) + 1);
            //System.out.println(countFreq);
            if (isValid(countFreq)) {
                res = Math.max(res, i + 1);
            }
        }
        return res;
    }
    
    private boolean isValid(Map<Integer, Integer> countFreq) {
        if (countFreq.size() == 1) {
            for (Map.Entry<Integer, Integer> e : countFreq.entrySet()) {
                if (e.getValue() == 1 || e.getKey() == 1) {
                    return true;
                }
                return false;
            }
        }
        if (countFreq.size() != 2) {
            return false;
        }
        if (countFreq.containsKey(1) && countFreq.get(1) == 1) {
            return true;
        }
        List<Map.Entry<Integer, Integer>> list = new ArrayList<>();
        for (Map.Entry<Integer, Integer> e : countFreq.entrySet()) {
            list.add(e);
        }
        Map.Entry<Integer, Integer> e1 = list.get(0);
        Map.Entry<Integer, Integer> e2 = list.get(1);
        if (e1.getValue() == 1 && e1.getKey() - e2.getKey() == 1) {
            return true;
        }
        if (e2.getValue() == 1 && e2.getKey() - e1.getKey() == 1) {
            return true;
        }
        return false;
    }
}
