class Solution {
    public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer> occurence = new HashMap<>();
        Map<Integer, Integer> count = new HashMap<>();
        int length = 1;
        for (int i = 0; i < nums.length; i++) {
            int prev = occurence.getOrDefault(nums[i], 0);
            int prevCount = prev == 0 ? 0 : count.get(prev);
            count.put(prev + 1, count.getOrDefault(prev + 1, 0) + 1);
            occurence.put(nums[i], prev + 1);
            if (prevCount == 1) {
                count.remove(prev);
            }
            else if (prevCount > 1) {
                count.put(prev, prevCount - 1);
            }
            
if (isValid(count)) {
                length = i + 1;
            }
        }
        return length;
    }

    boolean isValid(Map<Integer, Integer> count) {
        if (count.size() > 2) 
{
            return false;
        }
        int min = Integer.MAX_VALUE;
        int max = Integer.MIN_VALUE;
        for (int key: count.keySet()) {
            min = Math.min(min, key);
            max = Math.max(max, key);
        }
        if (min == max) {
            return count.get(min) == 1 || min == 1;
        }
        if (min == 1 && count.get(min) == 1) {
            return true;
        }
        return max == min + 1 && count.get(max) == 1;
    }
}
