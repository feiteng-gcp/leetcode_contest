class Solution {
    public int maxEqualFreq(int[] nums) {
        //num -> freq
        Map<Integer, Integer> count = new HashMap<>();
        //freq-1 -> count of num
        List<Integer> bucket = new ArrayList<>();
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            int f = count.getOrDefault(nums[i], 0);
            count.put(nums[i], f + 1);
            if (f > 0) {
                bucket.set(f-1, bucket.get(f-1) - 1);
            }
            if (f == bucket.size()) 
{
                bucket.add(1);
            } else {
                bucket.set(f, bucket.get(f) + 1);
            }
            //2种情况 1. 1个x+1其他全x  2. 1个1 其他全x
            int maxF = bucket.size();
            if (bucket.get(maxF - 1) == 1 && (maxF == 1 || bucket.get(maxF - 2) == count.size() - 1)) {
                res = i + 1;
            } else 
if (bucket.get(maxF - 1) == count.size() - 1 && bucket.get(0) == 1 || bucket.get(0) == count.size()) {
                res = i + 1;
            }
        }
        return res;
    }
}
