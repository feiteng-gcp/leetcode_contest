class Solution {
    public int maxEqualFreq(int[] nums) {
        HashMap<Integer, Integer> counter = new HashMap<>();
        HashMap<Integer, Integer> freq = new HashMap<>();
        int longest = 1;
        
        for (int i = 0; i < nums.length; ++i) {
            int n = nums[i];
            counter.put(n, counter.getOrDefault(n, 0) + 1);
            freq.put(counter.get(n), freq.getOrDefault(counter.get(n), 0) + 1);
            if (counter.get(n) - 1 != 0) {
                freq.put(counter.get(n) - 1, freq.get(counter.get(n) - 1) - 1);
                if (freq.get(counter.get(n) - 1) == 0)
                    freq.remove(counter.get(n) - 1);
            
}
            if (freq.size() == 2) {
                Iterator it = freq.keySet().iterator();
                int a = (Integer)it.next();
                int b = (Integer)it.next();
                if (a > b) {
                    int tmp = a;
                    a = b;
                    b = tmp;
                }
                if (b == a + 1 && freq.get(b) == 1) {
                    longest = Math.max(i + 1, longest);
                }
                if (a == 1 && freq.get(a) == 1) 
{
                    longest = Math.max(i + 1, longest);
                }
            } else if (freq.size() == 1) {
                int a = (Integer)freq.keySet().iterator().next();
                
if (freq.get(a) == 1 || a == 1) {
                    longest = Math.max(i + 1, longest);
                }
            }
        }
        
        return longest;
    }
}
