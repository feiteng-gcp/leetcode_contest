class Solution {
    public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer> nToCnt = new HashMap<>();
        Map<Integer, HashSet<Integer>> cntToN = new HashMap<>();
        int n = 0, res = 0, tmp = 0;
        
        for (int i = 0; i < nums.length; i++) {
            n = nums[i];
            if (nToCnt.containsKey(n)) {
                tmp = nToCnt.get(n);
                nToCnt.put(n, tmp + 1);
                HashSet<Integer> set = cntToN.get(tmp);
                set.remove(n);
                if (set.isEmpty()) {
                    cntToN.remove(tmp);
                }
            } else {
                nToCnt.put(n, 1);
                tmp = 0;
            }
            
            if (cntToN.containsKey(tmp + 1)) {
                cntToN.get(tmp + 1).add(n);
            } else {
                HashSet<Integer> set2 = new HashSet<>();
                set2.add(n);
                cntToN.put(tmp + 1, set2);
            }
            
            if (cntToN.size() == 2) {
                int[] keys = new int[2];
                int j = 0;
                for (int key: cntToN.keySet()) {
                    keys[j] = key;
                    j++;
                }
                int key1 = Math.min(keys[0], keys[1]);
                int key2 = Math.max(keys[0], keys[1]);
                if (key2 - key1 == 1 && cntToN.get(key2).size() == 1) {
                    res = i + 1;
                } else if (key1 == 1 && cntToN.get(1).size() == 1) {
                    res = i + 1;
                }
            } else if (cntToN.size() == 1) {
                for (int key: cntToN.keySet()) {
                    if (key == 1) {
                        res = i + 1;
                    } else if (key != 1 && cntToN.get(key).size() == 1) {
                        res = i + 1;
                    }
                }
            }
        }
        
        return res;
    }
}
