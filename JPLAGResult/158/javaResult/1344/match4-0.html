class Solution {
        public int maxEqualFreq(int[] nums) {
        int[] arr = new int[100000+1];
        int fq1 = 0;
        int fqmx = 0;
        int mx = 0;
        int count = 0;
        int diff = 0;
        int ans = 1;
        for(int n:nums){
            arr[n]++;
            if(arr[n]>mx){
                mx = arr[n];
                fqmx = 1;
            }else if(arr[n]==mx){
                ++fqmx;
            }
            if(arr[n]==1){
                ++fq1;
                ++diff;
            }
            if(arr[n]==2)
                --fq1;
            ++count;
            if(mx==1){
                ans = count;
                continue;
            }
            if(fq1==1&&fqmx==diff-1){
                ans=count;
                continue;
            }
            if(mx==2&&fqmx==1){
                ans=count;
                continue;
            }
            if(fqmx==1&&(diff*(mx-1))==(count-1)){
                ans=count;
                continue;
            }
        }
        return ans;
    }
}
