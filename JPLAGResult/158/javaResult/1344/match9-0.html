class Solution {
    public int maxEqualFreq(int[] nums) {
        HashMap<Integer, Integer> counts = new HashMap<Integer,Integer>();
        HashMap<Integer, HashSet<Integer>> values = new HashMap<Integer, HashSet<Integer>>();
        int res = 0;
        for (int i = 0; i < nums.length; i++) {
            int cnt = counts.getOrDefault(nums[i], 0) + 1;
            counts.put(nums[i], cnt);
            if (cnt > 1) {
                values.get(cnt-1).remove(nums[i]);
                if (values.get(cnt-1).isEmpty()) {
                    values.remove(cnt-1);
                }
            }
            HashSet<Integer> set = values.getOrDefault(cnt, null);
            if (set == null) {
                set = new HashSet<Integer>();
                values.put(cnt, set);
            }
            set.add(nums[i]);
            if (values.size() == 1) {
                if (values.get(cnt).size()==1 || cnt == 1)
                    res = i+1;
            } else if (values.size() == 2){
                if (values.containsKey(1) && values.get(1).size() == 1) {
                    res = i+1;
                } else if ((values.containsKey(cnt-1) && values.get(cnt).size()==1)
                           ||(values.containsKey(cnt+1) && values.get(cnt+1).size()==1)) {
                    res = i+1;
                }
            }
        }
        return res;
    }
}
