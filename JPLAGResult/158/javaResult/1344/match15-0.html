class Solution {
    public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer> map = new HashMap<>();
        Map<Integer, Integer> fre = new HashMap<>();
        Map<Integer, Integer> cntFre = new HashMap<>();
        int maxFre = 0, res = 0;
        for (int i = 0; i < nums.length; i++) {
            int num = nums[i];
            map.put(num, map.getOrDefault(num, 0) + 1);
            int f = map.get(num);
            cntFre.put(f, cntFre.getOrDefault(f, 0) + 1);
            cntFre.put(f - 1, cntFre.getOrDefault(f - 1, 1) - 1);
            
            if (map.get(num) > maxFre) maxFre = map.get(num);
            fre.put(f, fre.getOrDefault(f, 0) + map.get(num));
            if (fre.containsKey(f - 1)) {
                fre.put(f - 1, fre.get(f - 1) - f + 1);
            }
            if (fre.getOrDefault(1, 0) == 1 && fre.get(maxFre) == i) res = i;
            if (cntFre.get(maxFre) == 1 && fre.getOrDefault(maxFre - 1, 0) + fre.getOrDefault(maxFre, 0) == i + 1 && (fre.getOrDefault(maxFre, 0) - 1) * cntFre.getOrDefault(maxFre - 1, 0) == fre.getOrDefault(maxFre - 1, 0)) res = i;
            if (fre.getOrDefault(1, 0) == i + 1) res = i;
        }
        return res + 1;
    }
}
