import java.util.HashMap;
import java.util.TreeMap;

class Solution {

	public int maxEqualFreq(int[] nums) {
		HashMap<Integer, Integer> map = new HashMap<>();
		TreeMap<Integer, Integer> map2 = new TreeMap<>();
		for (int num : nums) {
			map.put(num, map.getOrDefault(num, 0) + 1);
		}
		for (int num : map.keySet()) {
			map2.put(map.get(num), map2.getOrDefault(map.get(num), 0) + 1);
		}
		for (int i = nums.length - 1; i >= 0; i--) {
			if (map2.size() == 1) {
				int a = map2.firstKey();
				if (map2.get(a) > 1 && a == 1 || a > 1 && map2.get(a) == 1) {
					return i + 1;
				}
			}
			if (map2.size() == 2) {
				int a = map2.firstKey(), b = map2.lastKey();
				if (map2.get(a) == 1 && a == 1) {
					return i + 1;
				}
				if (map2.get(b) == 1 && b - a == 1) {
					return i + 1;
				}
			}
			map2.put(map.get(nums[i]), map2.get(map.get(nums[i])) - 1);
			if (map2.get(map.get(nums[i])) == 0) {
				map2.remove(map.get(nums[i]));
			}
			map.put(nums[i], map.get(nums[i]) - 1);
			if (map.get(nums[i]) > 0) {
				map2.put(map.get(nums[i]), map2.getOrDefault(map.get(nums[i]), 0) + 1);
			}
		}
		return 0;
    }
}
