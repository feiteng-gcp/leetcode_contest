class Solution {
    public int maxEqualFreq(int[] nums) {
        TreeMap<Integer, Set<Integer>> map = new TreeMap<>();
        HashMap<Integer, Integer> all = new HashMap<>();
        int res = 1;
        int i = 0;
        // [2,2,1,1,5,3,3,5]
        for(int num: nums) {
            i++;
            if(all.containsKey(num)) {
                int cnt = all.get(num);
                all.put(num, cnt + 1);
                Set<Integer> set = map.get(cnt);
                set.remove(num);
                if(set.size() == 0)
                    map.remove(cnt);
                if(!map.containsKey(cnt+1))
                    map.put(cnt+1, new HashSet<>());
                map.get(cnt+1).add(num);
            } else {
                all.put(num, 1);
                if(!map.containsKey(1))
                    map.put(1, new HashSet<>());
                map.get(1).add(num);
            }
            if(isValid(map))
                res = Math.max(res, i);
        }
        return res;
    }
    
    private boolean isValid(TreeMap<Integer, Set<Integer>> map) {
        // System.out.println(map.size());
        int first = map.firstKey(), last = map.lastKey();
        if(map.size() == 1)
            return map.get(first).size() == 1 || first == 1;
        if(map.size() > 2)
            return false;
        if(first == 1 && map.get(first).size() == 1)
            return true;
        return first == last - 1 && map.get(last).size() == 1;
    }
}
