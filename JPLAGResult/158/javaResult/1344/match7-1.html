class Solution {
    public int maxEqualFreq(int[] nums) {
        int res = 1;
        int len = nums.length;
        //make all occurences equal except for one
        //sum up all occurences
        //remove one so that one more will
        
        //can either be one extra or one more
        
        //keep track of num to reach
        //
        
        
Map<Integer, Integer> map = new HashMap<>();
        
        Map<Integer, Integer> countMap = new HashMap<>();
        
        int tracker = 1;
        for (int i: nums) {
        	int count = map.getOrDefault(i, 0) + 1;
        	map.put(i, count);
        	
        	countMap.put(count, countMap.getOrDefault(count, 0) + 1);
        	
        	if (countMap.containsKey(count-1)) {
        		int prevCount = countMap.get(count-1);
        		if (prevCount == 1) {
        			countMap.remove(count-1);
        		} else {

        			countMap.put(count-1, prevCount-1);
        		}
        	}
        	
        	Set<Integer> set = countMap.keySet();
        	if (set.size()==1) {
        		if (set.contains(1))
        			res = tracker;
        		else {
        			List<Integer> lst = new ArrayList<>(set);
        			if (countMap.get(lst.get(0)) == 1)
        				res = tracker;
        		}

        	} else if (countMap.size() == 2) {
        		if (set.contains(1) && countMap.get(1) == 1) {
        			res = tracker;
        		} else {
        			List<Integer> lst = new ArrayList<>(set);
            		if( Math.abs (lst.get(0) - lst.get(1)) == 1) {
            			int greater = Math.max(lst.get(0), lst.get(1));
            			if (countMap.get(greater) == 1)
            				res = tracker;
            		}
        		}

        	}
        		
        	tracker++;
        }
        
        return res;
    }
}
