class Solution {
    public int maxEqualFreq(int[] nums) {
        //cnts.size() == 0 ok
        //cnts.size() == 1 ok
        //cnts.size() == 2 => 1, n + 1 = m and m size() == 1; 2, n == 1 and n size() == 1
        Map<Integer, Set<Integer>> cntsToNum = new HashMap<>();
        Map<Integer, Integer> numToCnt = new HashMap<>(); 
        for (int x : nums) {
            numToCnt.put(x, numToCnt.getOrDefault(x, 0) + 1);
        }
        
        for (int num : numToCnt.keySet()) {
            int cnt = numToCnt.get(num);
            cntsToNum.putIfAbsent(cnt, new HashSet<>());
            cntsToNum.get(cnt).add(num); 
        }
        
        for (int i = nums.length - 1; i >= 0; i--) {
            if (checkMap(cntsToNum)) {
                return i + 1; 
            }
            //update maps
            int num = nums[i]; 
            int cnt = numToCnt.get(num);
            numToCnt.put(num, cnt - 1);
            if (cnt == 1) numToCnt.remove(num);
            
cntsToNum.get(cnt).remove(num);
            if (cntsToNum.get(cnt).size() == 0) {
                cntsToNum.remove(cnt);
            }
            if (cnt > 1) {
                cntsToNum.putIfAbsent(cnt - 1, new HashSet<>());
                cntsToNum.get(cnt - 1).add(num);
            
}
        }
        
        
        return 0;
    }
    
    private boolean checkMap(Map<Integer, Set<Integer>> cntsToNum) {
        if (cntsToNum.size() > 2) {
            return false;
        } else if (cntsToNum.size() == 0) {
            return true;
        } else if (cntsToNum.size() == 1) 
{
            int cnt = -1;
            for (int key : cntsToNum.keySet()) {
                cnt = key; 
            }
            return (cnt == 1 || cntsToNum.get(cnt).size() == 1);
        } else {
            
int m = -1, n = -1; 
            for (int key : cntsToNum.keySet()) {
                if (m == -1) {
                    m = key;
                } else {
                    n = key; 
                }
            }
            if (n > m) { //make n < m
                int tmp = n; 
                n = m; 
                m = tmp;
            }
            //two conditions
            if (n + 1 == m) {
                if (cntsToNum.get(m).size() == 1) return true;
            } 
            if (n == 1) {
                if (cntsToNum.get(n).size() == 1) return true;
            }
            return false; 
        }
    }
}
