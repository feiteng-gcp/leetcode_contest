class Solution {
    public int maxEqualFreq(int[] nums) {
            int l = nums.length;
            int c = 0;
            int dp[] = new int[100001];
            
            int times[] = new int[100001];
            
        
            for(int i=0;i<=l-1;++i){
                if(dp[nums[i]]==0){
                    c++;
                    dp[nums[i]]++;
                    times[dp[nums[i]]]++;
                }else{
                    times[dp[nums[i]]]--;
                    dp[nums[i]]++;
                    times[dp[nums[i]]]++; 
                }
                
            }
            
            for(int i=l-1;i>=0;--i){
                int len = i+1;
                // if(i==6){
                //         System.out.println(len);
                //     int ck =  (len-1)/c;
                //         System.out.println(times[ck]);
                //         System.out.println(c);
                //     }
                if(c==1){
                    return len;
                }
                
                
                
                if((len-1)%c==0){
                    
                   int ck =  (len-1)/c;
                    if(c-1==times[ck]){
                        return len;
                    }
                    // if(i==6){
                    //     System.out.println(len);
                    //     System.out.println(times[ck]);
                    //     System.out.println(c);
                    // }
                    
                }
                if((len-1)%(c-1)==0){
                    
                   int ck =  (len-1)/(c-1);
                    
                    if(ck==1&&c==times[ck]){
                        return len;
                    }
                    
                    if(c-1==times[ck]){
                        return len;
                    }
                    // if(i==6){
                    //     System.out.println(len);
                    //     System.out.println(times[ck]);
                    //     System.out.println(c);
                    // }
                    //   if(i==1){
                    //     System.out.println(len);
                    //     System.out.println(times[ck]);
                    //     System.out.println(c);
                    // }
                    
                }
                
                
                
                
                
                
                times[dp[nums[i]]]--;
                dp[nums[i]]--;
                if(dp[nums[i]]==0)c--;
                times[dp[nums[i]]]++;
                
                
            }
            return 0;

        
    }
}
