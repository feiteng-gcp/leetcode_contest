class Solution {
    public int maxEqualFreq(int[] nums) {
        int res = 0;
        int number = (int) Math.pow(10, 6);
        int[] arr = new int[number];
        Map<Integer, Integer> myMap = new HashMap<>();
        
for (int i = 0; i < nums.length; i++) {
            arr[nums[i]]++;
            int times = arr[nums[i]];
            myMap.put(times, myMap.getOrDefault(times, 0) + 1);
            int curTimes = myMap.get(times);
            if (curTimes * times == i) {
                res = Math.max(res, i);
            }
            if (curTimes * times == i + 1 && i != nums.length - 1) 
{
                res = Math.max(res, i + 1);
            }
        }
        return res + 1;
    }
}
