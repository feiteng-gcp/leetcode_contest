class Solution {
    public int maxEqualFreq(int[] nums) {
        // num -> count
        HashMap<Integer, Integer> numCount = new HashMap<>();
        // count -> count number
        HashMap<Integer, Integer> count = new HashMap<>();
        int res = 2;
        int len = 0;
        
        for(int i : nums){
            len++;
            int curCount = numCount.getOrDefault(i, 0) + 1;
            numCount.put(i, curCount);
            int before = curCount - 1;
            if(before != 0){
                
int c = count.get(before);
                if(c == 1){
                    count.remove(before);
                }
                else{
                    count.put(before, c - 1);
                }
            }
            count.put(curCount, count.getOrDefault(curCount, 0) + 1);
            
            //System.out.println(len + " " + count);
            if(len != 1){
                if(count.size() == 2){
                    int[] key = new int[2];
                    int[] val = new int[2];
                    int index = 0;
                    for(int k : count.keySet()){
                        key[index] = k;
                        val[index] = count.get(k);
                        index++;
                    }
                    
                    if(key[0] == 1 && val[0] == 1){
                        res = len;
                    
}
                    if(key[1] == 1 && val[1] == 1){
                        res = len;
                    }
                    if(Math.abs(key[0] - key[1]) == 1){
                        if(key[0] > key[1] && val[0] == 1){
                             res = len;
                        }
                        if(key[0] < key[1] && val[1] == 1){
                             res = len;
                        }
                    }
                    
                }
                
                if(count.size() == 1){
                    for(int k : count.keySet()){
                        if(k == 1 || count.get(k) == 1){
                            res = len;
                        }
                    }
                }
            }
        }
        
        return res;
    }
}
