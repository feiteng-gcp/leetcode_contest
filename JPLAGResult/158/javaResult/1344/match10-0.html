class Solution {
    public int maxEqualFreq(int[] nums) {
        
        // hashmap + hashset?
        
        Map<Integer, Integer> numToOccurance = new HashMap<>();
        Map<Integer, Set<Integer>> occuranceToNum = new HashMap<>();
        
        // condition that gurantee we can delete one element and make all numbers have equal occurance
        // occuranceToNum 
        // 1) keyCount == 1
        // 2) keyCount == 2 && key2 == key1 + 1 && map.get(key2).size() == 1
        // 3) keyCount == 2 && keyX == 1 && map.get(1).size() == 1
       
        int index = 0;
        int maxLen = 0;
        for (int num : nums) {
            index++;
            numToOccurance.put(num, numToOccurance.getOrDefault(num, 0) + 1);
            int count = numToOccurance.get(num);
            if (count > 1) { 
                occuranceToNum.get(count - 1).remove(num);
                if (occuranceToNum.get(count - 1).size() == 0) {
                    occuranceToNum.remove(count - 1);
                }
            }
            occuranceToNum.putIfAbsent(count, new HashSet<>());
            occuranceToNum.get(count).add(num);
            
            if (occuranceToNum.keySet().size() == 1) {
                if (count == 1 || occuranceToNum.get(count).size() == 1)
                maxLen = index;
            } 
            
            if (occuranceToNum.keySet().size() == 2 && 
                     (occuranceToNum.containsKey(count - 1) 
                      || occuranceToNum.containsKey(count + 1))) {
                int otherKey = occuranceToNum.containsKey(count - 1) ? count - 1 : count + 1;
                if (otherKey > count && occuranceToNum.get(otherKey).size() == 1) {
                    maxLen = index;
                } else if (otherKey < count && occuranceToNum.get(count).size() == 1) {
                    maxLen = index;
                }        
            } 
            
            if (occuranceToNum.keySet().size() == 2 
                       && occuranceToNum.containsKey(1) && occuranceToNum.get(1).size() == 1) {
                maxLen = index;
            }
        }
        return maxLen;
        
    }
}
