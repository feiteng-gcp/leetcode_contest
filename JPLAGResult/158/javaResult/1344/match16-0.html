class Solution {
    public int maxEqualFreq(int[] nums) {
        Map<Integer, Integer>  frq = new HashMap();
        Map<Integer, Integer> mp = new HashMap(); // leng/ count
        int ans = 0;
        //if(nums.length==2) return 2;
        for(int i =0; i<nums.length; i++){
            int nb = nums[i];
            int nc = frq.getOrDefault(nb,0)+1;
            frq.put(nb, nc);
            mp.put(nc, mp.getOrDefault(nc,0)+1);
            if(nc!=1){
               int count = mp.get(nc-1);
               if(count==1) mp.remove(nc-1);
               else mp.put(nc-1, count-1);
            }

            if(mp.containsKey(1) && mp.size()==1){
                ans = i+1;
            }else if(mp.size()==1){
                Iterator<Integer> it = mp.keySet().iterator();
                Integer key1 = it.next();
                if(mp.get(key1)==1){
                    ans = i+1;
                }
            }else if(mp.size()==2){
                if(mp.getOrDefault(1,0)==1){
                    ans = i+1;
                }else{
                   Iterator<Integer> it = mp.keySet().iterator();
                   Integer key1 = it.next();
                   Integer key2 = it.next();
                   Integer max = Math.max(key1,key2);
                   Integer min = Math.min(key1, key2);
                   if(mp.get(max)==1 && max==min+1){
                       ans = i+1;
                   }
                }
            }
        }
        return ans;
    }
}
