class Solution {
    public int maxEqualFreq(int[] nums) {
        
Map<Integer, Integer> map = new HashMap();
        Map<Integer, Integer> counts = new HashMap();
        
        int res = 0, maxFreq = 0;
        for(int i=0; i<nums.length; i++) {
            int num = nums[i];
            int c = map.getOrDefault(num, 0);
            if(c > 0) {
                counts.put(c, counts.get(c)-1);
                if(counts.get(c) == 0)
                    counts.remove(c);
            }
            c++;
            maxFreq = Math.max(c, maxFreq);
            map.put(num, c);
            counts.put(c, counts.getOrDefault(c, 0)+1);
            
            if(map.size() == 1) res = i
;
            if(counts.size() == 1 && counts.containsKey(1)) res = i;
            if(counts.size() == 2 && (counts.containsKey(1) && counts.get(1) == 1)) res = i;
            
            if(counts.size() == 2 && counts.containsKey(maxFreq-1) && 
counts.get(maxFreq) == 1) res = i;
            
        }
        
        return res+1;
    }
}
