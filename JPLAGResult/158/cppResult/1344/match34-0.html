class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        vector<int> cnt(100005,0);
        vector<int> sz(100005,0);
    
        int ans = 0;
        for (int i = 0; i < nums.size(); i++)
        {
            int x = nums[i];
            cnt[x]++;
            sz[cnt[x]]++;
            if (sz[cnt[x]] * cnt[x] == i + 1)
            {
                if (i + 1 < nums.size())
                ans = max(ans, i + 2);
                else ans = max(ans, i);
            }
            else if (sz[cnt[x]] * cnt[x]  + 1 == i + 1)
            {
                ans = max(ans, i + 1);
            }
        }
        return ans;
       
    }
};
