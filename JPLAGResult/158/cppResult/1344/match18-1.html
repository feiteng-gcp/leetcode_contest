class Solution {
public:

    int maxEqualFreq(vector<int>& nums) {
        unordered_map<int, unordered_set<int>> mmap;
        unordered_map<int, int> freqs;
        int ans = 0;
        
        for (int i=0; i<nums.size(); i++) {
            int n = nums[i];
            int f = freqs[n]++;
            if (mmap.find(f) != mmap.end() && mmap[f].find(n) != mmap[f].end()) {
                mmap[f].erase(n);
                if (mmap[f].size() == 0)
                    mmap.erase(f);
            }
            mmap[freqs[n]].insert(n);
            // update ans

            if (mmap.size() == 1) {
                auto it = mmap.begin();
                if (it->first == 1 || it->second.size() == 1)
                    ans = i + 1;
            }
            else if (mmap.size() == 2) {
                auto it = mmap.begin();
                if (it->first == 1 && it->second.size() == 1)
                    ans = i + 1;
                else {
                    auto prev = it++;
                    if (it->first == 1 && it->second.size() == 1)
                        ans = i + 1;
                    else if (it->first == prev->first + 1 && it->second.size() == 1)
                        ans = i + 1;
                    else if (it->first == prev->first - 1 && prev->second.size() == 1)
                        ans = i + 1;
                }
            }
        }
        return ans;
    }
};
