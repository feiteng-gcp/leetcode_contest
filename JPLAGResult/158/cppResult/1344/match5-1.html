class Solution {
public:
	int maxEqualFreq(vector<int>& nums) {
		unordered_map<int, int> freq;
		map<int, int> fv;

		int ret = 1;
		for (int i = 0; i < nums.size();i++) {
			int v = nums[i];

			int f = freq[v];
			if (f > 0) {
				if (--fv[f] == 0)fv.erase(f);
			}
			++freq[v];
			f++;
			++fv[f];

			if (freq.size() == 1) {
				ret = i + 1;
			}

			else if (fv.size() == 1 && fv.count(1)) {
				ret = i + 1;
			}
			else if (fv.size() == 2 && fv.count(1) && fv[1] == 1) {
				ret = i + 1;
			}
			else if (fv.size() == 2 && fv.begin()->first + 1 == fv.rbegin()->first && fv.rbegin()->second == 1) {
				ret = i + 1;
			}
		}
		return ret;
	}
};
