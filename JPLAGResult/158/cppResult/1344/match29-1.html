class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        
        int best = 1;
        map<int, int> freq;
        
        map<int, int> freqfreq;
        
        for (int i = 0; i < nums.size(); i++) {
            int x = nums[i];
            if (freq[x] > 0) {
                freqfreq[freq[x]]--;
                if (freqfreq[freq[x]] == 0) {
                    freqfreq.erase(freq[x]);
                }
            }
            freq[x]++;

            freqfreq[freq[x]]++;
            
            /*
            std::cout << "after " << nums[i] << " at " << i << std::endl;
            for (auto [a, b] : freq) {
                std::cout << a << " occurs " << b << " times " << std::endl;
            }
            for (auto [a, b] : freqfreq) {
                std::cout << a << " is used " << b << " times " << std::endl;
            }
            */
            
            
            if (freqfreq.size() == 2) {
                auto it = freqfreq.begin();
                auto it2 = it;
                ++it2;
                
                if (it2->second == 1 && it2->first - 1 == it->first) {
                    // std::cout << "A " << i+1 << std::endl;
                    best = i+1;
                }
            }
            
            
            if (freqfreq.size() == 1 && freqfreq.begin()->first == 1) {
                // only one kind of thing!
                best = i+1;
                 // std::cout << "B " << i+1 << std::endl;
            }
            
            if (freq.size() == 1) {
                best =i+1;

            }
            
            if (freqfreq.size() == 2) {
                auto it = freqfreq.begin();
                if (it->first == 1 && it->second == 1) {
                    best = i+1;
                    // std::cout << "C " << i+1 << std::endl;
                }
            }
            
            
            
            // freq must have exactly one outlier, with more count than all others.
            // 0s are excluded
        }
        
        return best;
    }
};
