class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        map<int,int>Mg;
        map<int,int>Ng;
        int ret,Max = 0;
        for(int i = 0;i<nums.size();i++){
            Mg[nums[i]]++;
            Max = Max>Mg[nums[i]]?Max:Mg[nums[i]];
            if(Mg[nums[i]] > 1){
                Ng[Mg[nums[i]]]++;
                // Ng[Mg[nums[i]]-1]--;
            }else{
                Ng[1]++;
            }
            if(Max == 1 || (Ng[Max-1]*(Max-1) == i && Ng[Max] == 1) || (Ng[Max]*Max == i)){
                ret = i+1;
            }
        }
        return ret;
    }
};
