class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        unordered_map<int, int> counts;
        unordered_map<int, int> count_times;
        int result = 2;
        int maxt = 1;
        int mint = 1;
        int size = nums.size();
        
        for (int i = 0; i < size; i++)
        {
            int num = nums[i];
            if (counts[num])
                count_times[counts[num]]--;
            
            counts[num]++;
            count_times[counts[num]]++;
            maxt = max(maxt, counts[num]);
            while (count_times[mint] == 0)
                mint++;
            
            mint = min(mint, counts[num]);
            if (mint == 1 && count_times[1] == 1 && count_times[maxt] * maxt == i)
            {
                result = i + 1;
            }
            else if (mint == maxt - 1 && count_times[maxt] == 1)
            {
                result = i + 1;
            }
            else if (maxt == 1)
                result = i + 1;
            else if (maxt == mint && mint == i + 1)
                result = i + 1;
        }
        
        return result;
    }
};
