class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        unordered_map<int, int> count;
        unordered_map<int, unordered_set<int>> cand;
        
        int r = 1, m = 0;
        for (int i = 0; i < nums.size(); i++) {
            int n = nums[i];
            
            cand[count[n]].erase(n);
            if (cand[count[n]].size() == 0) {
                cand.erase(count[n]);
            }
            cand[count[n] + 1].insert(n);
            
            count[n]++;
            m = max(m, count[n]);
            
            if (cand.size() == 1 && cand.count(1)) {
                r = i + 1;
            }
            if (cand.size() == 1 && cand[m].size() == 1) {
                r = i + 1;
            }
            if (cand.size() == 2) {
                if (cand.count(m - 1) && cand[m].size() == 1) {
                    r = i + 1;
                }
                if (cand.count(1) && cand[1].size() == 1) {
                    r = i + 1;
                }                
            }
        }
        return r;
    }
};
