class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        unordered_map<int, unordered_set<int>> f2v;
        vector<int> v2f(100005);
        int ans = 1;
        for (int i = 0; i < nums.size(); i++) {
            int of = v2f[nums[i]];
            v2f[nums[i]]++;
            if (f2v.count(of)) {
                f2v[of].erase(nums[i]);
                if (f2v[of].empty())
                    f2v.erase(of);
            }
            f2v[v2f[nums[i]]].insert(nums[i]);
            //cout << "i=" << i << " num= " << nums[i] << " of= " << of << " " << f2v.size() << endl; 
            if (f2v.size() == 2) {
                vector<int> f;
                for (auto &p : f2v) {
                    f.push_back(p.first);
                }
                if (f[0] > f[1])
                    swap(f[0], f[1]);
                if (f[0] == 1 && f2v[1].size() == 1) {
                    ans = max(ans, i+1);
                }
                if (f[0] + 1 == f[1] && f2v[f[1]].size() == 1) {
                    ans = max(ans, i+1);
                }
            }
            if (f2v.size() == 1) {
                if (f2v.begin()->first == 1) {
                    ans = max(ans, i+1);
                } else if (f2v.begin()->second.size() == 1) {
                    ans = max(ans, i+1);
                }
            }
        }
        return (ans);
    }
};
