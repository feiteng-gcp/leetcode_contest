class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        // map v,times
        // map times n
        unordered_map<int,int> cnt;
        map<int,int> kinds;
        int ret = 0;
        for(int i=0;i<nums.size(); i++){
            int v = nums[i];
            // add cnt 
            cnt[v] ++;
            int t = cnt[v];
            if(t!=1){ // if not first apper
                if(kinds[t-1]-- == 1) {
                    kinds.erase(t-1); // remove kinds if last one
                }
            }
            kinds[t]++; // add kinds cnt 
            
            if(cnt.size() == 1) ret = max(ret, i+1);
            if(kinds.size() == 2) {
                if(kinds.count(1) && kinds[1] == 1) ret = max(ret, i+1);
                else {
                    // 2 2 2 2 3 2 
                    int p = kinds.begin()->first;
                    if(kinds.count(p+1) && kinds[p+1] == 1) ret = max(ret, i+1);
                }
            }else if(kinds.size() == 1 && kinds.count(1)) {
               ret = max(ret, i+1);
            }
        }
        return ret;
    }
};
