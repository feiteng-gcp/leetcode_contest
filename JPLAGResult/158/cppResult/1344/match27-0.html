class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        int ans = 1;
        unordered_map<int, int> cnt;
        vector<int> st(1e5+1);
        set<int> s;
        cnt[nums[0]]++;
        st[1] = 1;
        int l = 1, r = 1;
        for (int i = 1; i < nums.size(); ++i) {
            int raw = cnt[nums[i]];
            cnt[nums[i]]++;
            if (st[raw]) st[raw]--;
            if (!st[raw]) s.erase(raw);
            st[raw+1]++;
            s.insert(raw+1);
            l = min(l, raw+1);
            while (!st[l]) l++;
            r = max(r, raw+1);
            if ((l == 1 && r == 1) || (cnt.size() == 1) || (s.size() == 2 && ((l == 1 && st[l] == 1) || (l == r - 1 && st[r] == 1)))) ans = max(ans, i+1);
        }
        // cout << l << " " << st[l] << endl;
        // cout << r << " " << st[r] << endl;
        return ans;
    }
};
