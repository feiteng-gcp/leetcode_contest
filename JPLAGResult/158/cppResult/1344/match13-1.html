class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        

        multiset<int> st;
        map<int, int> freq;
        
        int ans = 0;
        
        for(int i = 0; i < nums.size(); i++) {
            if(freq.find(nums[i]) == freq.end()) {
                freq[nums[i]] = 1;
                st.insert(freq[nums[i]]);
            } else {
                st.erase(st.find(freq[nums[i]]));
                freq[nums[i]]++;
                st.insert(freq[nums[i]]);
            }
            
            if((int)st.size() == 1) {
                ans = i + 1;
            } else {
                multiset<int>::iterator it = st.begin();
                int s1 = (*it);
                it++;
                int s2 = (*it);
                
                it = st.end();
                it--;
                int e1 = (*it);
                it--;
                int e2 = (*it);
                

                if(s1 == 1 && s2 == e1) {
                    ans = i + 1;
                }
                if(s1 == e2 && (e1 == e2 + 1)) {
                    ans = i + 1;
                }
            }
            
        }
        
        return ans;
        
    }
};
