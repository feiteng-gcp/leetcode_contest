class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        unordered_map<int, int> cnts;
        unordered_map<int, int> lens;
        int ans = 1;
        int oneCnt = 0;
        for (int i = 0; i < nums.size(); ++i) {
            const int n = nums[i];
            const int l = cnts[n];
            ++cnts[n];
            if (lens.find(l) != lens.end()) {
                --lens[l];
                if (lens[l] == 0) {
                    lens.erase(l);                    
                }
            }
            ++lens[l + 1];
            if (cnts[n] == 1) {
                ++oneCnt;
            } else {
                --oneCnt;
            }
            if (lens.size() == 2) {
                vector<int> c;
                vector<int> l;
                for (auto p : lens) {
                    l.push_back(p.first);
                    c.push_back(p.second);
                }
                if (c[0] == 1 && (l[0] - 1 == l[1] || l[0] - 1 == 0)) {
                    ans = max(ans, i + 1);
                    continue;
                }
                if (c[1] == 1 && (l[1] - 1 == l[0] || l[1] - 1 == 0)) {
                    ans = max(ans, i + 1);
                    continue;
                }
            }

            if (lens.size() == 1 && oneCnt > 1) {
                ans = max(ans, i + 1);
            }
            if (lens.size() == 1 && lens.begin()->second == 1) {
                ans = max(ans, i + 1);
            }
        }
        return ans;
    }
};
