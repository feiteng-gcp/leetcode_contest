class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        int sol = 1;
        unordered_map<int, int> hash;
        unordered_map<int, int> count;
        for(int i = 0; i < nums.size(); i++){
            hash[nums[i]] = 0;
        }
        for(int i = 0; i < nums.size(); i++){
            int curr = hash[nums[i]];
            hash[nums[i]]++;
            if(curr != 0){
                if(count[curr] == 1){
                    count.erase(curr);
                }else{
                    count[curr]--;
                }
            }
            count[hash[nums[i]]]++;
            // cout<<i<<" "<<count.size()<<endl;
            if(count.size() == 2){
                int one = count.begin()->first;
                int valo = count.begin()->second;
                int two = (++count.begin())->first;
                int valt = (++count.begin())->second;
                // cout<<one<<" "<<two<<" "<<valo<<" "<<valt<<endl;
                if(one == two + 1 && valo == 1){
                    sol = i + 1;
                }else if(two == one + 1 && valt == 1){
                    sol = i + 1;
                }
                if(one == 1 && valo == 1){
                    sol = i + 1;
                }else if(two == 1 && valt == 1){
                    sol = i + 1;
                }
            }

            if(count.size() == 1 && count.begin()->first == 1){
                sol = i + 1;
            }
            if(count.size() == 1 && count.begin()->second == 1){
                sol = i + 1;
            }
        }
        return sol;
    }
};
