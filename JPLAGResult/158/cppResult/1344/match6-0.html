class Solution {
public:
    int maxEqualFreq(vector<int>& a) {
        const int n = a.size();
        unordered_map<int, int> m;
        map<int, int> f;
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            int c = ++m[a[i]];
            f[c]++;
            if (c > 1) {
                f[c - 1]--;
                if (f[c - 1] == 0) f.erase(c - 1);
            }
            if (f.size() == 2 && f.rbegin()->first == f.begin()->first + 1 && f.rbegin()->second == 1) {
                ans = i + 1;
            } else if (f.size() == 2 && f.count(1) && f[1] == 1) {
                ans = i + 1;
            } else if (f.size() == 1 && f.count(1)) {
                ans = i + 1;
            } else if (f.size() == 1 && f.begin()->second == 1) {
                ans = i + 1;
            }
        }
        return ans;
    }
};
