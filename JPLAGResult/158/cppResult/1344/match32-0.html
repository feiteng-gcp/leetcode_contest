class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        if (nums.size() == 1) return 0;
        
        unordered_map<int, int> oc;     // occurrence
        unordered_map<int, int> num;    // number appears for each value
        int ret = 1;
        
        for (int i = 0; i < nums.size(); ++i) {
            int n = nums[i];
            
            // occurence of number n
            int now = oc[n];
            int nxt = now + 1;
            oc[n]++;

            // dont care 0 case
            if (now) num[now]--;
            num[nxt]++;
            
            if (num[now]== 0) num.erase(now);
            
            // dbg
            // cout << "number: " << n << endl;
            // for (auto it: num) 
            // {
            //     cout << it.first << " = " << it.second << endl;
            // }
            
            
            // check if all numbers occurs only n and n+1 now
            if (oc.size() == 1) {
                // only one element
                ret = i + 1;
                    // cout << "1 record " << ret << endl;

            }
            if (oc.size() == 2) {
                // two elements and one of it occur only 1 time
                auto it1 = oc.begin();
                auto it2 = oc.begin();
                it2++;
                if (it1->second == 1 ||
                    it2->second == 1) {
                    ret = i + 1;
                    // if (ret > 2450) cout << "2 record " << ret << endl;

                }
            }
            if (num.size() == 1) {
                auto it = num.begin();
                if (it->first == 1) {
                    // all element occur only 1 time
                    ret = i + 1;
                                       // if (ret > 2400) cout << "3 record " << ret << endl;

                }
            } else if (num.size() == 2) {
                // one element occurs 1 more time than others
                auto it1 = num.begin();
                auto it2 = num.begin();
                it2++;
                if ((it1->first == 1 && it1->second == 1) ||
                    (it2->first == 1 && it2->second == 1)) {
                    ret = i + 1;
                    // if (ret > 2450) cout << "41 record " << ret << endl;
                }

                if (it1->first == it2->first + 1) {
                    if (it1->second == 1) {
                        ret = i + 1;
                        // if (ret > 2450) cout << "42 record " << ret << endl;
                    }
                }
                if (it1->first + 1 == it2->first) {
                    if (it2->second == 1) {
                        ret = i + 1;
                        // if (ret > 2450) cout << "43 record " << ret << endl;
                    }
                }
            }
        }
        return ret;
    }
};
