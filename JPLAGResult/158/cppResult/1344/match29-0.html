class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        map<int, int> mp, c;
        int n = nums.size();

        int ans = 1;
        for(int i = 0; i < n; i++) {
            int x = nums[i];
            mp[x] += 1;
            if(mp[x] != 1) {
                c[mp[x] - 1] -= 1;
                if(c[mp[x] - 1] == 0) {
                    c.erase(c.find(mp[x] - 1));
                }
            }
            c[mp[x]] += 1;
            if(c.size() == 2) {
                auto p = c.begin();
                auto q = c.end();
                q--;
                if(p->first + 1 == q->first && q->second == 1) { 
                    ans = i + 1;
                }
                if(p->first == 1 && p->second == 1) {
                    ans = i + 1;
                }
            }
            if(c.size() == 1) {
                auto p = c.begin();
                if(p->first == 1 || p->second == 1) {
                    ans = i + 1;
                }
            }
        }
        return ans;
    }
};
