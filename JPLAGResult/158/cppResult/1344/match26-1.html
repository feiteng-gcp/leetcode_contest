class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        int n = static_cast<int>(nums.size());

        map<int, int> cnt, cntcnt;
        int result = 0;
        for (int i = 1; i <= n; ++i) {
            int num = nums[i - 1];
            if (cnt.find(num) != cnt.end()) {
                if (--cntcnt[cnt[num]] == 0) {
                    cntcnt.erase(cnt[num]);
                }
            }
            ++cnt[num];
            ++cntcnt[cnt[num]];
            if (cnt.size() == 1) {
                result = i;
            } else if (cntcnt.size() == 1) {
                if (cntcnt.begin()->first == 1) {
                    result = i;
                }

            } else if (cntcnt.size() == 2) {
                auto it = cntcnt.begin();
                int fc = it->first, fn = it->second;
                ++it;
                int sc = it->first, sn = it->second;
                if (fc == 1 && fn == 1) {
                    result = i;
                } else if (sn == 1 && sc == fc + 1) {
                    result = i;
                }
            }
        }
        return result;
    }
};
