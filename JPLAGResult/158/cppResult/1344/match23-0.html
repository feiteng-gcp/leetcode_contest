class Solution {
public:
    int cnt[100000+5];
    int maxEqualFreq(vector<int>& nums) {
        memset(cnt, 0, sizeof(cnt));
        map<int, int> mp;
        int ans = 1, n = nums.size();
        cnt[nums[0]] = 1;
        mp[1] = 1;
        for (int i = 1; i < n; i++) {
            int a = nums[i];
            if (cnt[a] != 0) {
                mp[cnt[a]]--;
                if (mp[cnt[a]] == 0) mp.erase(cnt[a]);
            }
            cnt[a]++;
            mp[cnt[a]]++;
            if (mp.size() == 1) {
                if (mp.begin()->second == 1) ans = i + 1;
                if (mp.begin()->first == 1) ans = i + 1;
            } else if (mp.size() == 2) {
                if (mp.begin()->first == 1 && mp.begin()->second == 1) ans = i + 1;
                if (mp.begin()->first + 1 == next(mp.begin())->first && next(mp.begin())->second == 1) ans = i + 1;
            }
        }
        return ans;
    }
};
