class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        map<int, int> val;
        map<int, int> cnt;
        int res = 0;
        for(int i = 0; i < nums.size(); ++i) {
            int sum = ++val[nums[i]];
            auto it = cnt.find(sum - 1);
            if(it != cnt.end()) {
                --it->second;
                if(0 == it->second) cnt.erase(it);
            }
            ++cnt[sum];
            if(2 == cnt.size()) {
                auto it1 = cnt.begin();
                auto it2 = it1;
                ++it2;
                if(1 == (it2->first - it1->first) && 1 == it2->second) res = i + 1;
                else if((1 == it1->first && 1 == it1->second) || (1 == it2->first && 1 == it2->second)) res = i + 1;

            } else if(1 == cnt.size()) {
                bool flag = true;
                for(auto it = val.begin(); it != val.end(); ++it) {
                    if(1 != it->second) {
                        flag = false;
                        break;
                    }
                }
                if(1 == val.size()) flag = true;
                if(flag) res = i + 1;
                
            }
        }
        return res;
    }
};
