class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
		int ans = 0;
		unordered_map<int, int> num_cnt;
		unordered_map<int, int> cnt_cnt;
		// for (const auto& num : nums) {
		//     if (num_cnt.find(num) == num_cnt.end()) {
		//         ++ cnt_cnt[0];
		//         num_cnt[num] = 0;
		//     }
		// }
		int curr_num_cnt = 0;
		for (int i = 0; i < nums.size(); ++i) {
			if (curr_num_cnt == 0 || i % curr_num_cnt == 0 && cnt_cnt[i / curr_num_cnt] == curr_num_cnt) {
				ans = max(ans, i + 1);
			}

			if (num_cnt[nums[i]] == 0) ++curr_num_cnt;
			//cout << curr_num_cnt << " differenct number now" << endl;
			if (num_cnt[nums[i]] != 0) --cnt_cnt[num_cnt[nums[i]]];
			++num_cnt[nums[i]];
			//cout << num_cnt[nums[i]] << "times" << endl;
			++cnt_cnt[num_cnt[nums[i]]];
			//cout << cnt_cnt[num_cnt[nums[i]]] << " numbers with same time" << endl;
			int curr_cnt = num_cnt[nums[i]];
			
			// last number included
			if (cnt_cnt[curr_cnt] == curr_num_cnt - 1 && (i + 1 - curr_cnt * cnt_cnt[curr_cnt] == curr_cnt + 1 || i + 1 - curr_cnt * cnt_cnt[curr_cnt] == 1)) {
				ans = max(ans, i + 1);
			}

			// last number not included
			if (cnt_cnt[curr_cnt - 1] == curr_num_cnt - 1 && i + 1 - (curr_cnt - 1) * cnt_cnt[curr_cnt - 1] == curr_cnt) {
				ans = max(ans, i + 1);
			}
		}
		return ans;
	}
};
