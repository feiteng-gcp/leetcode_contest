class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        //len, types. freq = len / types
        unordered_map<int, unordered_set<int>> freq;
        freq[0] = unordered_set<int>();
        unordered_map<int, int> record;
        int max_freq = 0;
        int ans = 0;
        for(int i = 0; i < nums.size(); i++) {
            int num = nums[i];
            int old_freq = record[num];
            int new_freq = ++record[num];
            
            max_freq = max(max_freq, new_freq);
            
            if(old_freq != 0) freq[old_freq].erase(num);
            freq[new_freq].insert(num);
            
            
            int types = record.size();
            int len = i + 1;
            
            if(freq[max_freq].size() + freq[max_freq - 1].size() == types && freq[max_freq].size() == 1) {
                ans = max(ans, len);
            }            
            
            if(freq[max_freq].size() == types - 1 && freq[1].size() == 1) {
                ans = max(ans, len);
            }
            
            if(max_freq == 1) {
                ans = max(ans, len);
            }
        }
        return ans;
    }
};
