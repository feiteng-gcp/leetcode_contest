class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        map<int,int> cntNum; cntNum.clear();
        map<int,int> cnt; cnt.clear();
        int res = 1;
        for(int i=0; i<nums.size(); i++){            
            cntNum[nums[i]]++;
            if(cnt.size() == 1){
                res = i+1;
            }            

            int c = cntNum[nums[i]];
            
            cnt[c] ++;            
            if(c > 1){
                cnt[c-1]--;                
                if(cnt[c-1]==0){
                    cnt.erase(c-1);
                }                
            }
            if(cnt.size() == 2){
               
                if(cnt.find(1)!=cnt.end() && cnt[1]==1) res = i+1;
                int x[3], j = 0;                              
                for(map<int,int>::iterator it=cnt.begin(); it!=cnt.end(); it++){
                    x[j] = it->first;
                    j++;
                }                
                int lr = max(x[0],x[1]), sm=min(x[0],x[1]);
                /*cout << "Map=" << endl;
                for(map<int,int>::iterator it=cnt.begin(); it!=cnt.end(); it++){
                    cout << it->first <<" " << it->second << " " << res << " " << lr << " " <<  endl;
                }*/
                if(lr == sm+1 && cnt[lr]==1) res = i+1;
            }
        }        
        return res;
    }
};
