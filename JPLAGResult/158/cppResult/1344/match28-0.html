class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        multiset<int> ss;
        unordered_map<int, int> mm;
        int ans = 1;
        for(int i=0; i<nums.size(); i++){
            int todel = mm[nums[i]]++;
            if(todel){
                auto it = ss.find(todel);
                ss.erase(it);
            }
            ss.insert(todel+1);
            if(ss.size() == 1){
                ans = max(ans, *ss.begin());
                continue;
            }
            int last = *prev(ss.end());
            int sec = *next(ss.begin());
            if((*ss.begin() == 1 && (sec == last)) || (last-1 == *ss.begin() && last-1 == *prev(prev(ss.end()))))
                ans = max(ans, i+1);
        }
        return ans;
    }
};
