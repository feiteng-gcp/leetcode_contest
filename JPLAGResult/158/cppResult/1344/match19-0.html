class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        int result = 0;
        unordered_map<int, unordered_set<int>> countMap;
        unordered_map<int, int> numCount;
        for (int i = 0; i < nums.size(); i++) {
            int num = nums[i];
            if (numCount.find(num) == numCount.end()) {
                numCount[num] = 1;
                countMap[1].insert(num);
            } else {
                int count = numCount[num];
                countMap[count].erase(num);
                if (countMap[count].empty()) {
                    countMap.erase(count);
                }
                count++;
                countMap[count].insert(num);
                numCount[num] = count;
            }
            //
            if (countMap.size() == 1) {
                if (countMap.begin()->second.size() == 1) 
                    result = i + 1;
                
                if (numCount[*countMap.begin()->second.begin()] == 1) 
                    result = i + 1;
            }
            if (countMap.size() == 2) {
                auto temp = vector<pair<int, unordered_set<int>*>>();
                for (auto it = countMap.begin(); it != countMap.end(); it++) {
                    temp.push_back(make_pair(it->first, &(it->second)));
                }
                if (temp[0].second->size() == 1 && numCount[*temp[0].second->begin()] == 1) {
                    result = i + 1;
                }
                if (temp[0].second->size() == 1 && numCount[*temp[0].second->begin()] == 1 + numCount[*temp[1].second->begin()]) {
                    result = i + 1;
                }
                if (temp[1].second->size() == 1 && numCount[*temp[1].second->begin()] == 1) {
                    result = i + 1;
                }
                if (temp[1].second->size() == 1 && numCount[*temp[1].second->begin()] == 1 + numCount[*temp[0].second->begin()]) {
                    result = i + 1;
                }
            }
        }
        return result;
    }
};
