class Solution {
public:
    bool canIremove(map<int,set<int>> &count2num){
        if(count2num.size()==1 && (count2num.begin())->second.size()==1){
            return true;
        }
        if(count2num.size()==1 && (count2num.begin())->first==1){
            return true;
        }
        if(count2num.size()==2){
            auto it = count2num.begin();
            int small_count = it->first;
            ++it;
            int large_count = it->first;
            if(large_count==(small_count+1) && count2num[large_count].size()==1){
                //cout<<"here1 \n";
                //cout<<small_count<<" "<<large_count<<endl;
                return true;
            }
            if(small_count==1 && count2num[small_count].size()==1){
                //cout<<"here2 \n";
                return true;
            }
        }
        return false;
    }
    
    int maxEqualFreq(vector<int>& nums) {
        int res = 0;
        unordered_map<int, int> count;
        map<int,set<int>> count2num;
        for(int i=0; i<nums.size(); i++){
            //cout<<"i "<<i<<endl;
            int old_count = count[nums[i]];
            count[nums[i]]++;
            count2num[old_count].erase(nums[i]);
            if(count2num[old_count].empty()){
                count2num.erase(old_count);
            }
            count2num[old_count+1].insert(nums[i]);
            
            if(canIremove(count2num)){
                res = i+1;
            }
        }
        return res;
    }
};
