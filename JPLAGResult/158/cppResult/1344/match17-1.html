class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
      int visited = 0, num_of_one = 0, num_of_max = 0;

      vector<int> count(100001, 0);
      int max_count = 0;
      
      int longest = 0;
      for (int i = 0; i < nums.size(); ++i) {
        int n = nums[i];
        
        if (count[n] == 0) {
          if (visited * max_count == i) {
            longest = i + 1;
          }
          ++visited;
          ++num_of_one;
        }
        
        ++count[n];
        if (count[n] == 2) {
          --num_of_one;
        }
        
        if (count[n] > max_count) {
          max_count = count[n];
          num_of_max = 1;

        } else if (count[n] == max_count) {
          ++num_of_max;
        }
        
        if (num_of_max == 1 && visited * (max_count - 1) == i) {
          longest = i + 1;
        }
        
        if (num_of_one == 1 && (visited - 1) * max_count == i) {
          longest = i + 1;
        }
      }
      
      return longest;
    }
};
