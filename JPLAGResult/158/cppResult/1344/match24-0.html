class Solution {
public:
    int maxEqualFreq(vector<int>& v) {
        map<int, int> m;
        map<int, int> f;
        int res = 1;
        for (int i = 0; i < v.size(); i++) {
            m[ v[i] ]++;
            
            int nF = m[ v[i] ];
            
            f[ nF ]++;
            
            //cout << f[nF] << endl;

            
            if (nF > 1) {
                f[ nF - 1]--;
                if (f[ nF - 1] == 0) {
                    f.erase(nF - 1);
                }
            }
            
            if (i == 11) {
                map<int, int>::iterator I;
                vector<int> v;
                for (I = f.begin(); I != f.end(); I++) {
                    cout << I->first << " " << I->second << endl;
                }
            }
            
            if (i > 0) {
                if (m.size() == 1 || (f.size() == 1 && nF == 1)) {
                    res = i + 1;
                } else if (f.size() == 2) {
                    map<int, int>::iterator I;
                    vector<int> v;
                    for (I = f.begin(); I != f.end(); I++) {
                        v.push_back(I->first);
                    }
                    sort(v.begin(), v.end());
                    if (f[ v[1] ] == 1 && v[1] == (v[0] + 1)) {
                        res = i + 1;
                    }
                    
                    //if (f[v[0]] == 1 || f[v[1]] == 1) res = i + 1;
                    if (v[0] == 1 && f[v[0]] == 1) res = i + 1;
                    //cout << "-- " << v[0] << " " << v[1] << endl;
                }
            }
        }
        return res;
    }
};
