class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        return solve(nums);
    }
    
    int solve(vector<int> &vc) {
        map<int, int> cnt;
        map<int, set<int>> st;

        int ans = 0;

        for (int i = 0; i < vc.size(); ++i) {
            int v = vc[i];
            int n = cnt[v];

            auto cur = st.find(n);
            if (cur != st.end()) {
                cur -> second.erase(v);
                if (cur -> second.size() == 0) {
                    st.erase(n);
                }
            }

            cnt[v]++;
            ++n;

            st[n].insert(v);

            if (st.size() == 1){
                auto fs = st.begin();
                if (fs -> first == 1 || fs -> second.size() == 1) ans = i;
            }
            else if (st.size() == 2) {
                auto fs = st.begin();
                auto sc = st.rbegin();

                if (fs -> first == 1 && fs -> second.size() == 1) ans = i;

                if (sc -> first == fs -> first + 1 && sc -> second.size() == 1) ans = i;
            }

            //print(st);
        }

        return ans + 1;
    }

};
