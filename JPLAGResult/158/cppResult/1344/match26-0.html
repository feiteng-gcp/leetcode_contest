class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        unordered_map<int, int> cnt, num;
        int ret = 0;
        for (int k = 0;k < nums.size(); k++){
            int i = nums[k];
            if (cnt[i]){
                num[cnt[i]]--;
                if (num[cnt[i]]==0) num.erase(cnt[i]);
            }
            cnt[i]++;
            num[cnt[i]]++;
            if (num.size()==1 && (num.begin()->second==1 || num.begin()->first==1)) ret=k+1;
            else if (num.size()==2){
                auto it = num.begin();
                int k1 = it->first, v1 = it->second;
                it++;
                int k2 = it->first, v2 = it->second;
                if (v1==1 && (k1-1==k2 || k1==1) || v2==1 && (k2-1==k1 || k2==1)) ret=k+1;
            }
        }
        return ret;
    }
};
