class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        set<pair<int,int>> s;
        unordered_map<int,int> m;
        int ans = 0;
        for (int i=0; i<nums.size(); i++) {
            if (m.count(nums[i])) {
                s.erase({m[nums[i]], nums[i]});
            }
            s.insert({++m[nums[i]], nums[i]});
            auto ub = s.lower_bound({s.begin()->first + 1, 0});
            int idx = distance(s.begin(), ub);
            if (s.size() == 1 || (ub->first == s.begin()->first + 1 && idx + 1 == s.size()) || (s.begin()->first == 1 && (idx == s.size() || (idx == 1 && ub->first == s.rbegin()->first)))) ans = i + 1;
        }
        return ans;
    }
};
