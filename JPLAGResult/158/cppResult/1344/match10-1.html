class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {

      map<int, int> x_to_freq;
      map<int, set<int>> freq_to_x;
      int ans = 0;
      for (int i = 0; i < nums.size(); ++i) {
        int x = nums[i];
        int freq = x_to_freq[x];
        ++freq;
        x_to_freq[x] = freq;
        freq_to_x[freq - 1].erase(x);
        freq_to_x[freq].insert(x);
        if (freq_to_x[freq - 1].empty()) {
          freq_to_x.erase(freq - 1);
        }
        int mx_freq = freq_to_x.rbegin()->first;
        int mn_freq = freq_to_x.begin()->first;

        if (mn_freq + 1 == mx_freq && freq_to_x[mx_freq].size() == 1) {
          ans = max(ans, i + 1);
        } else if (mn_freq == 1 && freq_to_x.begin()->second.size() == 1 && freq_to_x.size() == 2) {
          ans = max(ans, i + 1);
        } else if (mn_freq == mx_freq && (mn_freq == 1 || freq_to_x.begin()->second.size() == 1)) {
          ans = max(ans, i + 1);
        } 
      }
      return ans;
    }
};
