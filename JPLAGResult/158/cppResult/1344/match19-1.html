class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        int res = 0;
        
        unordered_map<int, int> freq;
        unordered_map<int, int> cnts;
        for (int i = 0; i < nums.size(); ++i) {
            auto [it, inserted] = freq.emplace(nums[i], 1);
            if (!inserted) {
                if (--cnts[it->second] == 0)
                    cnts.erase(it->second);
                ++it->second;
                ++cnts[it->second];
            } else {
                ++cnts[1];
            }
            
            /*
            cout << i << " : " << endl;
            for (auto& [a, b] : cnts) {
                cout << " " << a << ": " << b << endl;
            }
            cout << endl;
            */
            if (cnts.size() == 1) {
                auto [a, ac] = *cnts.begin();
                if (a == 1 || ac == 1) {
                    res = max(res, i + 1);
                }
            }
            if (cnts.size() == 2) {
                auto [a, ac] = *cnts.begin();
                auto [b, bc] = *next(cnts.begin());

                if ((a == 1 && ac == 1) || (b == 1 && bc == 1)) {
                    res = max(res, i + 1);
                }
                if (a == b + 1 && ac == 1) {
                    res = max(res, i + 1);
                }
                if (a + 1 == b && bc == 1) {
                    res = max(res, i + 1);
                }
            }
        }
        
        return res;
    }
};
