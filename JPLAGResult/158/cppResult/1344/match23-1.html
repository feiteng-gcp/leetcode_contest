class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        map<int,int> m;
        int cnt[100005];
        fill(cnt,cnt+100005,0);
        int ans=0;
        for(int i=0;i<nums.size();i++){
            if(cnt[nums[i]]!=0){
                if(--m[cnt[nums[i]]]==0){
                    m.erase(cnt[nums[i]]);
                }

            }
            cnt[nums[i]]++;
            m[cnt[nums[i]]]++;
            if(m.size()==1){
                if(m.find(1)!=m.end())ans=i+1;
                if(m.begin()->second==1)ans=i+1;
            }
            else if(m.size()==2){
                if(m.find(1)!=m.end()&&m[1]==1)ans=i+1;
                if(m.begin()->first+1==m.rbegin()->first&&m.rbegin()->second==1)ans=i+1;
            }
        }
        return ans;
    }
};
