class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        unordered_map<int, unordered_set<int>> cntMap;
        int res = 0;
        for (int i = 0; i < nums.size(); i++) {
            int n = nums[i];
            bool met = false;
            for (unordered_map<int, unordered_set<int>>::iterator it = cntMap.begin(); it != cntMap.end(); ++it) {
                int cnt = it->first;
                if (it->second.count(n)) {
                    met = true;
                    it->second.erase(n);
                    cntMap[cnt + 1].insert(n);
                    break;
                }
            }
            for (unordered_map<int, unordered_set<int>>::iterator it = cntMap.begin(); it != cntMap.end();) {
                if (it->second.empty()) {
                    it = cntMap.erase(it);
                } else {
                    ++it;
                }
            }
            if (!met) {
                cntMap[1].insert(n);
            }
            if (cntMap.size() == 2) {
                unordered_map<int, unordered_set<int>>::iterator it = cntMap.begin();
                int cnt1 = it->first;
                int siz1 = it->second.size();
                ++it;
                int cnt2 = it->first;
                int siz2 = it->second.size();
                if ((cnt2 == cnt1 + 1 && siz2 == 1) || (cnt1 == cnt2 + 1 && siz1 == 1)
                    || (cnt1 == 1 && siz1 == 1) || (cnt2 == 1 && siz2 == 1)) {
                    res = i;
                }
            } else if (cntMap.size() == 1) {
                unordered_map<int, unordered_set<int>>::iterator it = cntMap.begin();
                int cnt1 = it->first;
                int siz1 = it->second.size();
                if (cnt1 == 1 || siz1 == 1) {
                    res = i;
                }                
            }
        }
        return res + 1;
    }
};
