class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        map<int, vector<int> > x;
        unordered_map<int,int> um;
        int ans=0;
        int max_freq=1;
        if(nums.size()<=2){
            return nums.size();
        }
        for(int i=0;i<nums.size();i++){
            if(um.find(nums[i])==um.end()){
                um[nums[i]]=1;
                x[1].push_back(nums[i]);
            } else {
                um[nums[i]]++;
                x[um[nums[i]]].push_back(nums[i]);
                max_freq=max(max_freq,um[nums[i]]);
            }
            if(x[max_freq].size()==1 && x[1].size()==x[max_freq-1].size()){
                ans=i;
            }
            if(x[1].size()==x[max_freq].size()+1 && (max_freq!=2 && x[1].size()!=x[2].size() || max_freq==2)){
                ans=i;
            }
            if(max_freq==1){
                ans=i;
            }
        }
        return ans+1;
    }
};
