class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        unordered_map<int,int> cnt;
        unordered_map<int,int> cnt_dis;
        int ans=1;
        for(int i=0;i<nums.size();i++){
            if(cnt[nums[i]]>0){
                cnt_dis[cnt[nums[i]]]--;
                if(cnt_dis[cnt[nums[i]]]==0){
                    cnt_dis.erase(cnt[nums[i]]);
                }
            }
            cnt[nums[i]]++;
            cnt_dis[cnt[nums[i]]]++;
            if(cnt_dis.size()==1&&(cnt_dis.begin()->second==1||cnt_dis.begin()->first==1)) ans=i+1;
            if(cnt_dis.size()==2){
                auto a=cnt_dis.begin();
                auto b=next(cnt_dis.begin());
                //cout<<a->first<<" "<<a->second<<endl;
                //cout<<b->first<<" "<<b->second<<endl;
                if(a->second==1&&a->first==1){

                    ans=i+1;
                }
                
                if(b->second==1&&b->first==1){
                    ans=i+1;
                }
                
                if(a->first==b->first+1&&a->second==1){
                    ans=i+1;
                }
                
                if(b->first==a->first+1&&b->second==1){
                    ans=i+1;
                }
            }
        }
        return ans;
    }
};
