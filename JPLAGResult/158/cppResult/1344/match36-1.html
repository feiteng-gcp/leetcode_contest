class Solution {
public:
    int maxEqualFreq(vector<int>& nums) {
        unordered_map<int, int> dict;
        map<int, int> cnt;
        
        int ans = 1;
        for(int i=0;i<nums.size();i++){
            if(dict.find(nums[i]) == dict.end()){
                dict[nums[i]] = 1;
                cnt[1] ++;
            }
            else{                
                int temp = dict[nums[i]];
                cnt[temp]--;
                if(cnt[temp] == 0)
                    cnt.erase(temp);
                
                dict[nums[i]]++;
                cnt[dict[nums[i]]]++;
            }
            if(i>0){
                int low = cnt.begin()->first;
                int high = cnt.rbegin()->first;
                if(dict.size() == 1)
                    ans = i + 1;
                else if(low + 1 == high && low*((int)dict.size()) == i)
                    ans = i + 1;
                else if(low == 1 && high*((int)dict.size()-1) == i)
                    ans = i + 1;
            }
        }
        return ans;
    }
};
