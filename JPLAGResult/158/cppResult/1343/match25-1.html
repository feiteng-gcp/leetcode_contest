const int N = 5020, M = 6, K = 16, MOD = 1000000007;
int dp[N][M][K];
void add(int &x, int y) {
  x += y;
  if (x >= MOD) {
    x %= MOD;
  }

}
class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
      memset(dp, 0, sizeof(dp));
      for (int i = 0; i < 6; ++i) {
        dp[1][i][1] = 1;
      }
      for (int i = 2; i <= n; ++i) {
        for (int j = 0; j < 6; ++j) {
          for (int k = 1; k <= rollMax[j]; ++k) {
            dp[i][j][k] = 0;     
            if (k == 1) {
              for (int t = 0; t < 6; ++t) {
                if (t == j) {
                  continue;
                }
                for (int kk = 0; kk <= rollMax[t]; ++kk) {
                  add(dp[i][j][k], dp[i - 1][t][kk]);
                }

              }
            }
            add(dp[i][j][k], dp[i - 1][j][k - 1]);
          }
        }
      } 
      int ans = 0;
      for (int j = 0; j < 6; ++j) {
        for (int k = 0; k <= rollMax[j]; ++k) {
          //cout << j << " " << k << " " << dp[n][j][k] << endl;
          add(ans, dp[n][j][k]);
        }
      }
      return ans;
    }
};
