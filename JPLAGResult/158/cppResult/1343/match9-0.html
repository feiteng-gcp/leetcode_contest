const int mod = 1e9 + 7;
const int N = 5010;
const int M = 20;
int dp[10][M][N];
class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        memset(dp, 0, sizeof(dp));
        for ( int i = 1; i <= 6; ++i ) dp[i][1][1] = 1;
        // 以j个i结尾
        for ( int k = 2; k <= n; ++k ) {
            for ( int i = 1; i <= 6; ++i ) {
                for ( int j = 1; j <= 15; ++j ) {
                    
                    for ( int l = 1; l <= 6; ++l ) {
                        if ( l == i ) {
                            if ( rollMax[i - 1] == j ) continue;
                            dp[l][j + 1][k] += dp[i][j][k - 1];
                            dp[l][j + 1][k] %= mod;
                        } else {
                            dp[l][1][k] += dp[i][j][k - 1];
                            dp[l][1][k] %= mod;
                        }
                    }
                }
            }
        }
        int res = 0;
        for ( int i = 1; i <= 6; ++i ) {
            for ( int j = 1; j <= 15; ++j ) {
                res += dp[i][j][n];
                res %= mod;
            }
        }
        return res;
    }
};
