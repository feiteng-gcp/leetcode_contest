class Solution {

public:
    int dieSimulator(int n, vector<int>& rollMax) {
        vector<vector<vector<int>>> dp(n, vector<vector<int>>(6, vector<int>(15))); 
        for (int i = 0; i < 6; ++i) {
            dp[0][i][0] = 1;
        }
        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < 6; ++j) {
                for (int k = 0; k <= min(rollMax[j] - 1, i); ++k) {
                    if (k != 0) {
                        dp[i][j][k] = dp[i - 1][j][k - 1];

                    } else {
                        for (int t = 0; t < 6; ++t) {
                            if (t != j) {
                                for (int l = 0; l < rollMax[t]; ++l) {
                                    dp[i][j][k] = (dp[i][j][k] + dp[i - 1][t][l]) % MODULO;
                                }
                            }
                        }
                    }

                }
            }
        }
        int ans = 0;
        for (int i = 0; i < 6; ++i) {
            for (int j = 0; j < rollMax[i]; ++j) {
                ans = (ans + dp[n - 1][i][j]) % MODULO;
            }
        }
        return ans;
    }
private:
    const int MODULO = 1e9 + 7;
};
