class Solution {
    const int P=1000000007;

    inline void add(int &x,int y){
        x+=y;
        if(x>=P)x-=P;
    }
public:
    int dieSimulator(int n, vector<int>& rollMax) {

        vector<array<array<int,15>,6>> dp(n);
        for(int j=0;j<6;j++)dp[0][j][0]=1;
        for(int i=1;i<n;i++)
            for(int j=0;j<6;j++)
                for(int k=0;k<15;k++)
                    for(int jp=0;jp<6;jp++){
                        if(k>=rollMax[j])continue;
                        if(j==jp){
                            if(k>0)add(dp[i][j][k],dp[i-1][j][k-1]);
                        }
                        else if(k==0)
                            for(int kp=0;kp<15;kp++)
                                add(dp[i][j][0],dp[i-1][jp][kp]);

                    }
        int ans=0;
        for(int j=0;j<6;j++)
            for(int k=0;k<15;k++)
                add(ans,dp[n-1][j][k]);
        return ans;
    }
};
