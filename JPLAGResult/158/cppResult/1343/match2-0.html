class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
                int mod = 1000000007;
        vector<int>ans[6];
        for(int i =0;i<n+10;i++){
            for(int j=0;j<6;j++){
                ans[j].push_back(0);
            }
        }
        for(int j = 0;j<6;j++){
            ans[j][0] = 1;
        }
        for(int i = 1;i<n;i++){
            for(int  j =0 ;j<6;j++){
                for(int k =1;k<=rollMax[j];k++){
                    if(i-k < 0)break;
                    for(int m=0;m<6;m++){
                        if(m==j)continue;
                        (ans[j][i]+=ans[m][i-k])%=mod;
                    }
                }
                if(i < rollMax[j]){
                    ans[j][i]++;
                }
            }
        }
        int ret  =0;
        for(int j = 0;j<6;j++){
            (ret+=ans[j][n-1])%=mod;
        }
        return ret%mod;
    }
};
