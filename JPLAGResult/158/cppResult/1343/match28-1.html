class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        long long inf = 1000000007;
        vector<vector<vector<long long>>> opt(n, vector<vector<long long>>(6, vector<long long>(15, 0)));
        for (int i = 0; i < 6; ++i) opt[0][i][0] = 1;

        for (int i = 0; i + 1 < n; ++i) {
            for (int j = 0; j < 6; ++j) {
                for (int k = 0; k < rollMax[j]; ++k) {
                    if (k + 1 < rollMax[j]) {
                        opt[i + 1][j][k + 1] = (opt[i + 1][j][k + 1] + opt[i][j][k]) % inf;

                    }
                    for (int l = 0; l < 6; ++l) {
                        if (j == l) continue;
                        opt[i + 1][l][0] = (opt[i + 1][l][0] + opt[i][j][k]) % inf;
                    }
                }
            }
        }

        long long ans = 0;
        for (int i = 0; i < 6; ++i) {
            for (int j = 0; j < rollMax[i]; ++j) {
                ans = (ans + opt[n - 1][i][j]) % inf;
            }
        }
        return ans;
    }
};
