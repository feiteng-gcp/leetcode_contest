#define MODM ((int)1e9 + 7)
class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {

        int dp[5001][7][16];
        memset(dp, 0, sizeof(dp));
        for(int i=1; i<=6; i++) dp[1][i][1] = 1;
        
        for(int i=2; i<=n; i++) {
            int c = i;
            int p = c-1;

            for(int j=1; j<=6; j++) { // cur dice
                for(int k=1; k<=6; k++) { // prev dice
                    for(int l=1; l<=rollMax[k-1]; l++) {
                        int cl;
                        if (j == k) cl = l+1;
                        else cl = 1;
                        
                        if (cl > rollMax[j-1]) break;
                        dp[c][j][cl] = (dp[c][j][cl] + dp[p][k][l])%MODM;
                    }
                }
            }

        }
        int ans = 0;
        for(int i=1; i<=6; i++) {
            for(int j=1; j<=rollMax[i-1]; j++) {
                ans = (ans + dp[n][i][j])%MODM;
            }
        }
        return ans;
    }
};
