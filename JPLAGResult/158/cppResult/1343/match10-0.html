class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        const int maxElement = *max_element(rollMax.begin(), rollMax.end());
        vector<vector<vector<int>>> dp(6, vector<vector<int>>(n+1, vector<int>(maxElement, 0)));
        for (int i = 0; i < 6; ++i) {
            dp[i][1][0] = 1;
        }
        for (int l = 2; l < n + 1; ++l) {
            for (int i = 0; i < 6; ++i) {
                for (int j = 0; j < 6; ++j) {
                    if (i == j) {
                        continue;
                    }
                    for (int k = 0; k < maxElement; ++k) {
                        dp[i][l][0] += dp[j][l - 1][k];
                        dp[i][l][0] %= m;
                    }
                }
                for (int k = 0; k < maxElement - 1; ++k) {
                    if (k + 1 == rollMax[i]) {
                        break;
                    }
                    dp[i][l][k + 1] += dp[i][l - 1][k];
                    dp[i][l][k + 1] %= m;
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < 6; ++i) {
            for (int j = 0; j < maxElement; ++j) {
                ans += dp[i][n][j];
                ans %= m;
            }
        }
        return ans;
    }
    
    const int m = 1000000007;
};
