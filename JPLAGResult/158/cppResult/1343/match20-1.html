class Solution {
    const int INF = 1e9 + 7;
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        int rm = *max_element(rollMax.begin(), rollMax.end());
        vector<vector<vector<int>>> f(n, vector<vector<int>>(6, vector<int>(rm + 1, 0))); //n, num, times

        for (int i = 0; i < 6; i++)
            f[0][i][1] = 1;
        
        for (int i = 1; i < n; i++)
            for (int j = 0; j < 6; j++)
                for (int k = 1; k <= min(rollMax[j], i + 1); k++) {
                    if (k == i + 1) {
                        f[i][j][k] = 1;
                        continue;
                    }
                    if (k > 1) {
                        f[i][j][k] = f[i - 1][j][k - 1];
                        continue;
                    }

                    for (int pre = 0; pre < 6; pre++) {
                        if (pre == j)
                            continue;
                        for (int times = 1; times <= rollMax[pre]; times++)
                        f[i][j][k] = (1LL * f[i][j][k] + f[i - 1][pre][times]) % INF;
                    }
                }
        

        int ans = 0;
        for (int j = 0; j < 6; j++)
            for (int k = 1; k <= rollMax[j]; k++)
                ans = (1LL * ans + f.back()[j][k]) % INF;
        return ans;
    }
};
