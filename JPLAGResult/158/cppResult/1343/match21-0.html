const int mod = 1e9 + 7;
typedef long long LL;
LL dp[6][16], tmp[6][16];
class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        memset(dp, 0, sizeof(dp));
        for (int i = 0; i < 6; i++) dp[i][1] = 1;
        n--;
        while (n--) {
            memset(tmp, 0, sizeof(tmp));
            for (int i = 0; i < 6; i++) {
                for (int j = 0; j <= 15; j++) {
                    for (int k = 0; k < 6; k++) {
                        if (k == i && rollMax[i] <= j) continue;
                        if (k != i) tmp[k][1] = (tmp[k][1] + dp[i][j]) % mod;
                        else tmp[k][j+1] = (tmp[k][j+1] + dp[i][j]) % mod;
                    }
                }
            }
            for (int i = 0; i < 6; i++) {
                for (int j = 0; j < 16; j++) {
                    dp[i][j] = tmp[i][j];
                }
            }
        }
        LL ret = 0;
        for (int i = 0; i < 6; i++) {
            for (int j = 0; j < 16; j++) {
                ret = (ret + dp[i][j]) % mod;
            }
        }
        return ret;
        
    }
};
