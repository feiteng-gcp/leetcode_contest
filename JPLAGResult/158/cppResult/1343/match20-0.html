typedef long long ll;
const ll modulo = 1e9+7;

class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        ll f[5005][7][16]{};
        for (int i = 1; i <= 6; ++i) f[1][i][1] = 1;
        for (int i = 2; i <= n; ++i)
            for (int j = 1; j <= 6; ++j)
                for (int k = 1; k <= rollMax[j-1]; ++k) {
                    if (k > 1) f[i][j][k] = f[i-1][j][k-1];
                    else for (int u = 1; u <= 6; ++u) {
                        if (u == j) continue;
                        for (int v = 1; v <= rollMax[u-1]; ++v)
                            f[i][j][k] = (f[i][j][k] + f[i-1][u][v]) % modulo;
                    }
                }
        
        ll ans = 0;
        for (int i = 1; i <= 6; ++i)
            for (int j = 1; j <= rollMax[i-1]; ++j)
                ans = (ans + f[n][i][j]) % modulo;
        
        return ans;
    }
};
