class Solution {
private:
    static const int MOD = 1000000007;
    int f[5050][6][20];

public:
    int dieSimulator(int n, vector<int>& rollMax) {
        memset(f, 0, sizeof(f));
        for (int i = 0; i < 6; i++)
        {
            f[1][i][1] = 1;
        }
        for (int i = 1; i < n; i++)
        {
            for (int j = 0; j < 6; j++)
            {
                for (int k = 1; k <= rollMax[j]; k++)
                {
                    for (int l = 0; l < 6; l++)
                    {
                        if (l == j) continue;
                        f[i + 1][l][1] = (f[i + 1][l][1] + f[i][j][k]) % MOD;
                    }
                    if (k != rollMax[j])
                    {
                        f[i + 1][j][k+1] = (f[i + 1][j][k+1] + f[i][j][k]) % MOD;

                    }
                }
            }
        }
        int res = 0;
        for (int i = 0; i < 6; i++)
        {
            for (int j = 1; j <= rollMax[i]; j++)
            {
                res = (res + f[n][i][j]) % MOD;
            }
        }
        return res;
    }
};
