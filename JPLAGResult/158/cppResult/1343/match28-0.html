#define LL long long

LL mod = 1e9 + 7;

LL dp[10000][6];
class Solution {
public:
    int dieSimulator(int n, vector<int>& r) {
        LL ans = 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < 6; j++) {
                dp[i][j] = 0;
            }
        }
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < 6; j++) {
                for(int k = 1; k <= r[j]; k++) {
                    if(i - k == -1) {
                        dp[i][j] = (dp[i][j] + 1) % mod;
                        break;
                    }
                    for(int l = 0; l < 6; l++) {
                        if(l == j) continue;
                        dp[i][j] = (dp[i][j] + dp[i - k][l]) % mod;
                    }
                }
            }
        }
        for(int i = 0; i < 6; i++) {
            ans = (ans + dp[n - 1][i]) % mod;
        }
        return ans;
    }
};
