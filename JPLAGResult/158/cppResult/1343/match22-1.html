class Solution {
#define MAX 5005
#define ll long long
#define P (ll)(1e9+7)
public:
    ll f[MAX][7][20];
    int dieSimulator(int n, vector<int>& a) {
        memset(f, 0, sizeof(f));
        for(int i = 1; i <= 6; i++){

            if(a[i-1] > 0) f[1][i][1] = 1;
        }
        for(int i = 1; i < n; i++){
            for(int j = 1; j <= 6; j++){
                for(int k = 1; k <= a[j-1]; k++){
                    if(k > i) continue;
                    for(int l = 1; l <= 6; l++){
                        if(l == j) continue;
                        (f[i+1][l][1] += f[i][j][k]) %= P;
                    }
                    if(k < a[j-1]) (f[i+1][j][k+1] += f[i][j][k])%=P;
                }
            }
        }

        ll ans = 0;
        for(int i = 1; i <= 6; i++){
            for(int j = 1; j <= a[i-1]; j++){
                ans = (ans + f[n][i][j])%P;
            }
        }
        return ans;
    }
};
