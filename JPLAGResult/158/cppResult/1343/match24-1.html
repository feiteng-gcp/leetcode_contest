#define ll long long
const ll mod=1e9+7;
inline void adjust(ll &x)
{
    if(x>=mod)
    {
        x-=mod;
    }
}

class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        rollMax.insert(rollMax.begin(),0);
        ll dp[n+1][7][16];
        memset(dp,0,sizeof(dp));
        for(int j=1;j<=6;j++)
        {

            dp[1][j][1]=1;
        }
        for(int i=2;i<=n;i++)
        {
            for(int j=1;j<=6;j++)
            {
                for(int k=2;k<=rollMax[j];k++)
                {
                    dp[i][j][k]+=dp[i-1][j][k-1];
                    adjust(dp[i][j][k]);
                }
                for(int l=1;l<=6;l++)
                {
                    if(l==j)
                    {
                        continue;
                    }
                    for(int num=1;num<=rollMax[l];num++)
                    {
                        dp[i][j][1]+=dp[i-1][l][num];
                        adjust(dp[i][j][1]);
                    }

                }
            }
        }
        ll ans=0;
        for(int j=1;j<=6;j++)
        {
            for(int k=1;k<=rollMax[j];k++)
            {
                ans+=dp[n][j][k];
                adjust(ans);
            }
        }
        int fans=ans;
        return fans;
    }
};
