class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        int dp[n][6][16],mod = 1000000007;
        memset(dp,0,sizeof(dp));

        for(int j = 0 ; j < 6 ; j ++)
            dp[0][j][1] = 1;
        for(int i = 1 ; i < n ; i ++)
        {
            for(int j = 0 ; j < 6 ; j ++)
            {
                for(int t = 1 ; t <= rollMax[j] ; t ++)
                {
                    if(t == 1)
                    {
                        for(int k = 0 ; k < 6 ; k ++)
                        {
                            if(k != j) 
                            {
                                for(int m = 1 ; m <= rollMax[k] ; m ++)

                                    dp[i][j][t] = (dp[i][j][t] + dp[i - 1][k][m]) % mod;
                            }
                        }
                    }else
                    {
                        dp[i][j][t] = dp[i - 1][j][t - 1];
                    }
                }
            }
        }
        int res = 0;
        for(int j = 0 ; j < 6 ; j ++)
        {
            for(int t = 1 ; t <= rollMax[j] ; t ++)
                res = (res + dp[n - 1][j][t]) % mod;
        }
        return res;
    }
};
