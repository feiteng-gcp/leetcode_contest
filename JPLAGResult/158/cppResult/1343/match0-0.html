class Solution {
    map<pair<int,int>,int> memo;
    int mod = 1E9+7;
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        return dfs(n, -1, rollMax);
    }
    
    int dfs(int n, int prev, vector<int> & num) {
        if(n==0)
            return 1;
        auto pr = make_pair(n, prev);
        if(memo.count(pr))
            return memo[pr];

        int cur = 0;
        for(int i=0; i<6; i++) {
            if(i==prev)
                continue;
            int this_n = n;
            for(int k=0; k<num[i]&&this_n>0; k++) {
                this_n--;
                cur+=dfs(n-k-1, i, num);
                cur%=mod;
            }
        }
        return memo[pr]=cur;
    }
    
};
