long dp[5005][7][16];
class Solution {
private:
    long mod = 1000000007;
    long add(long a, long b) {
        long ans = (a+b) % mod;
        return (ans);
    }
public:

    int dieSimulator(int n, vector<int>& rollMax) {
        memset(dp, 0, sizeof (dp));
        for (int i = 1; i <= 6; i++) {
            dp[1][i][1] = 1;
        }

        for (int i = 1; i < n; i++) {
            for (int j = 1; j <= 6; j++) {
                for (int k = 1; k <= rollMax[j-1]; k++) {
                    //cout << i << " " << j << " " << k << endl;
                    if (k < rollMax[j-1]) {
                        dp[i+1][j][k+1] = add(dp[i+1][j][k+1], dp[i][j][k]);    
                    }

                    for (int l = 1; l <= 6; l++) {
                        if (l != j) {
                            dp[i+1][l][1] = add(dp[i+1][l][1], dp[i][j][k]);
                        }
                    }
                }
            }
        }

        long ans = 0;
        for (int i = 1; i <= 6; i++) {
            for (int j = 1; j <= rollMax[i-1]; j++) {
                ans = add(ans, dp[n][i][j]);
            }
        }
        return (ans);
    }
};
