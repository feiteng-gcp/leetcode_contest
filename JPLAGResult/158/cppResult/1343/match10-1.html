class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        const int MOD = 1e9 + 7;
        int dp[2][6][16];
        memset(dp, 0, sizeof(dp));
        int prev = 1, curr = 0;

        for (int i = 0; i < 6; ++i) {
            dp[curr][i][1] = 1;
        }
        for (int step = 1; step < n; ++step) {
            swap(curr, prev);
            for (int i = 0; i < 6; ++i) {
                dp[curr][i][1] = 0;

                for (int j = 0; j < 6; ++j) {
                    if (i == j) {
                        continue;
                    }
                    for (int k = 1; k <= rollMax[j]; ++k) {
                        (dp[curr][i][1] += dp[prev][j][k]) %= MOD;
                    }
                }
                for (int k = 2; k <= rollMax[i]; ++k) {

                    dp[curr][i][k] = dp[prev][i][k - 1];
                }
            }
        }
        int result = 0;
        for (int i = 0; i < 6; ++i) {
            for (int k = 1; k <= rollMax[i]; ++k) {
                (result += dp[curr][i][k]) %= MOD;
            }
        }
        return result;
    }
};
