class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        const int M = 1e9+7;
        vector<vector<vector<int> > > dp(n, vector<vector<int> >(6, vector<int>(16, 0)));
        for(int i = 0; i<6; i++) {
            dp[0][i][1] = 1;
        }
        
        for(int i = 1; i<n; i++) {
            for(int j = 0; j<6; j++) {
                for(int k = 1; k<=rollMax[j] && k<=i+1; k++) {
                    if(k==1) {
                        for(int t = 0; t<6; t++) {
                            if(t==j) continue;
                            for(int q = 1; q<=rollMax[t]; ++q) {
                                dp[i][j][k] = (dp[i][j][k] + dp[i-1][t][q])%M;
                            }
                        }   
                    } else {
                        dp[i][j][k] = dp[i-1][j][k-1];
                    }
                }
            }
        }
        int ans = 0;
        for(int j = 0; j<6; j++) for(int k = 1; k<=rollMax[j]; k++) {
            ans = (ans + dp[n-1][j][k])%M;
        }
        return ans;
        
        
    }
};
