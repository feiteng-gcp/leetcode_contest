const int MAXN = 5e3 + 10;
const long long mod = 1e9 + 7;
long long dp[MAXN][10][20];
class Solution {
public:
    int dieSimulator(int n, vector<int>& rm) {
        memset(dp, 0, sizeof(dp));
        for(int j = 1; j <= 6; j++) {
            dp[0][j][1] = 1;
        }
        for(int i = 1; i < n; i++) {
            for(int j = 1; j <= 6; j++) {
                for(int jj = 1; jj <= 6; jj++)
                    if(jj != j) {
                        for(int kk = 1; kk <= rm[jj - 1]; kk++) {
                            dp[i][j][1] = (dp[i][j][1] + dp[i - 1][jj][kk]) % mod;
                        }
                    }
                for(int k = 2; k <= rm[j - 1]; k++) {
                    dp[i][j][k] = dp[i - 1][j][k - 1];
                }
            }
        }
        long long res = 0;
        for(int j = 1; j <= 6; j++) {
            for(int k = 1; k <= rm[j - 1]; k++) {
                res = (res + dp[n - 1][j][k]) % mod;
            }
        }
        return res;
    }
};
