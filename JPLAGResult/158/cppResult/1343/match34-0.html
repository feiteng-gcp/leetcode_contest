class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        long long roll[n+1][6][16];
        for(int i=0;i<=n;i++){
            for(int j=0;j<6;j++){
                for(int k=0;k<15;k++){
                    if(i==1 && k==0){
                        roll[i][j][k]=1;
                    } else {
                        roll[i][j][k]=0;
                    }
                }
            }
        }
        for(int i=2;i<=n;i++){
            for(int j=0;j<6;j++){
                for(int k=0;k<6;k++){
                    if(k==j){
                        for(int l=0;l<rollMax[k];l++){
                            roll[i][j][l+1]=(roll[i][j][l+1]+roll[i-1][k][l])%1000000007;
                        }
                    } else {
                        for(int l=0;l<rollMax[k];l++){
                            roll[i][j][0]=(roll[i][j][0]+roll[i-1][k][l])%1000000007;
                        }
                    }
                }
            }
        }
        long long ans=0;
        for(int i=0;i<6;i++){
            for(int j=0;j<rollMax[i];j++){
               ans=(ans+roll[n][i][j])%1000000007; 
            }
        }
        return ans;
    }
};
