const int mod = 1000000007;
int dp[5000][6][15];

class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        memset(dp,0,sizeof(dp));
        
        for(int i=0;i<6;i++)dp[0][i][0] = 1;
        for(int i=0;i<n - 1;i++){
            for(int j=0;j<6;j++){
                for(int k=0;k<rollMax[j];k++){
                    for(int x=0;x<6;x++){
                        if(j == x){
                            if(k + 1 < rollMax[j]){
                                (dp[i+1][x][k+1] += dp[i][j][k]) %=mod;
                            }

                        }else{
                            (dp[i+1][x][0] += dp[i][j][k] )%=mod;
                        }
                    }
                }
            }
        }
        int ret=0;
        for(int j=0;j<6;j++)
                for(int k=0;k<rollMax[j];k++) (ret += dp[n-1][j][k]) %=mod;
        return ret;
    }
};
