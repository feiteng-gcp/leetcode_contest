class Solution {
public:
    const static int MOD = 1e9 + 7;
    int dp[5005][6][20];
    

    int dieSimulator(int n, vector<int>& rollMax) {
        memset(dp, 0, sizeof(dp));
        
        for(int i = 0; i < 6; ++ i)
        {
            dp[0][i][1] = 1;
        }
        
        for(int i = 1; i < n; ++ i)
        {
            for(int j = 0; j < 6; ++ j)
            {
                for(int k = 0; k < 6; ++ k)
                {
                    if(k != j)
                    {
                        for(int l = 0; l < 20; ++ l)
                        {
                            dp[i][j][1] += dp[i - 1][k][l];
                            dp[i][j][1] %= MOD;
                        }
                    }
                }
                for(int k = 1; k <= rollMax[j]; ++ k)
                {
                    dp[i][j][k] += dp[i - 1][j][k - 1];
                    dp[i][j][k] %= MOD;

                }
            }
        }
        
        int ans = 0;
        for(int i = 0; i < 6; ++ i)
        {
            for(int j = 0; j < 20; ++ j)
            {
                ans += dp[n - 1][i][j];
                ans %= MOD;
            }
        }
        
        return ans;
    }
};
