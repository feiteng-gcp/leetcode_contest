class Solution {
    int mod = 1e9+7;

public:
    int dieSimulator(int n, vector<int>& rollMax) {
        vector<vector<vector<int> > > dp(n + 2, vector<vector<int> > (6, vector<int> (16)));
        for(int i = 0; i < 6; ++i) {
            dp[1][i][1] = 1;
        }
        for(int i = 1; i < n; ++i){
            for(int j = 0; j < 6; ++j){
                for(int l = 1; l <= rollMax[j]; ++l){
                    for(int k = 0; k < 6; ++k){
                        if(k != j) dp[i+1][k][1] = (dp[i+1][k][1] + dp[i][j][l]) % mod;
                        else {
                            if(l == rollMax[j]) continue;
                            dp[i+1][k][l+1] = (dp[i+1][k][l+1] + dp[i][j][l]) % mod;
                        }
                    }
                }
            }
        }

        long long ans = 0;
        for(int i = 0; i < 6; i++){
            //cout << "i : "<<i<<" : ";
            for(int j = 0; j <= rollMax[i]; ++j) {
                ans += dp[n][i][j];
                //cout << dp[n][i][j] << " ";
            }
            //cout << endl;
        }
        ans %= mod;
        return (int)ans;
    }
};
