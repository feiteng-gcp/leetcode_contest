class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        return solve(n, rollMax);
    }
    
    int solve(int n, vector<int> &rollMax) {
	int dp[5001][6][16];

	const int mod = 1000 * 1000 * 1000 + 7;

	for (int i = 0; i < 5001; ++i)
		for (int j = 0; j < 6; ++j)
			for (int k = 0; k < 16; ++k)
				dp[i][j][k] = 0;


	for (int j = 0; j < 6; ++j)
		dp[0][j][1] = 1;

	for (int i = 1; i < n; ++i) {
		for (int j = 0; j < 6; ++j) {
			for (int k = 1; k <= rollMax[j]; ++k) {
				if (k == 1) {
					for (int tj = 0; tj < 6; ++tj) {
                        if (tj == j) continue;
						for (int tk = 1; tk <= rollMax[tj]; ++tk) {
							dp[i][j][k] = (dp[i][j][k] + dp[i - 1][tj][tk]) % mod;
						}
					}
				}
				else {
					dp[i][j][k] = dp[i - 1][j][k - 1];
				}
			}
		}
	}

	int res = 0;
	for (int j = 0; j < 6; ++j)
		for (int k = 1; k <= rollMax[j]; ++k)
			res = (res + dp[n - 1][j][k]) % mod;

	return res;
}
};
