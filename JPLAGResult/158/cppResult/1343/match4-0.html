using A = vector<vector<vector<int>>>;
const int MOD = 1000000007;
class Solution {
public:
    int solve(int bef, int beg, int n, const vector<int> &roll, A &dp) {
        if(0 >= n) return 1;
        if(dp[bef][beg][n] >= 0) return dp[bef][beg][n];
        int res = 0;
        for(int i = 0; i < 6; ++i) {
            if(i == beg) {
                if(bef < roll[i]) res += solve(bef + 1, i, n - 1, roll, dp);
            } else {
                res += solve(1, i, n - 1, roll, dp);
            }
            res %= MOD;
        }
        return dp[bef][beg][n] = res;
    }
    int dieSimulator(int n, vector<int>& rollMax) {
        A dp(16, vector<vector<int>>(7, vector<int>(5005, -1)));
        int res = 0;
        for(int i = 0; i < 6; ++i) {
            res += solve(1, i, n - 1, rollMax, dp);
            res %= MOD;
        }
        return res;
    }
};
