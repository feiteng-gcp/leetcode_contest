const int MOD = 1000000007;

class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        vector<vector<int64_t>> dp(6, vector<int64_t>(15, 0));
        for (int x = 0; x < 6; ++x)
            dp[x][0] = 1;
        
        for (int i = 1; i < n; ++i) {
            vector<vector<int64_t>> dp2(6, vector<int64_t>(15, 0));
            for (int x = 0; x < 6; ++x) {
                for (int y = 0; y < 6; ++y) {
                    if (y == x)
                        continue;
                    for (int c = 0; c < rollMax[y]; ++c) {
                        dp2[x][0] += dp[y][c];
                    }
                }
                dp2[x][0] = dp2[x][0] % MOD;
                for (int c = 1; c < dp2[x].size(); ++c) {
                    dp2[x][c] = dp[x][c-1];
                }
            }
            swap(dp, dp2);
        }
        
        int64_t result = 0;
        for (int x = 0; x < 6; ++x) {
            for (int c = 0; c < rollMax[x]; ++c) {
                result += dp[x][c];
            }
        }
        return result % MOD;
    }
};
