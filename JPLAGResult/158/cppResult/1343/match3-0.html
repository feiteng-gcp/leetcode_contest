class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        long long MOD = 1e9 + 7, res = 0;
        vector<long long> sum(6, 1);
        vector<vector<int>> cnt(6);
        for (int i = 0; i < 6; ++i) {
            cnt[i].resize(rollMax[i], 0);
            cnt[i][0] = 1;
        }
        while (--n) { 
            vector<int> Sum(6, 0);
            long long temp_sum = 0;
            for (int j = 0; j < 6; ++j) {
                temp_sum = (temp_sum + sum[j]) % MOD;
            }
            for (int j = 0; j < 6; ++j) {
                Sum[j] = (temp_sum - sum[j] + MOD) % MOD; 
            }
            
            for (int i = 0; i < 6; ++i) {
                long long temp = Sum[i];
               
                vector<int> &v = cnt[i];
                for (int j = v.size()-1; j != 0; --j) {
                    v[j] = v[j-1];
                }
                v[0] = temp;
                sum[i] = 0;
                for (auto &j : v) {
                    sum[i] = (sum[i] + j) % MOD;
                }
               // cout << sum[i] << endl;
            }
        }
         
        for (int i = 0; i < 6; ++i) {
            res = (res + sum[i]) % MOD;
        }
        return res;
    }
};
