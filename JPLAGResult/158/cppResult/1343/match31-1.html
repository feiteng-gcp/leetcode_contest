class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        const int Z = 1e9+7;

        int dp[n+1][7];
        memset(dp, 0, sizeof(dp));

        dp[0][6] = 1;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < 6; j++) {
                for (int k = 1; k <= rollMax[j] && k <= i; k++) {
                    for (int x = 0; x < 7; x++) {
                        if (x != j) {

                            dp[i][j] = (dp[i][j] + dp[i-k][x]) % Z;
                        }
                    }
                }
            }
        }

        int ret = 0;
        for (int i = 0; i < 7; i++) {
            ret = (ret + dp[n][i]) % Z;
        }
        
        return ret;
    }
};
