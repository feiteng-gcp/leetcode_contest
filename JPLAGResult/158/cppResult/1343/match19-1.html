class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        // f[i][j][k] = 
        //              sum of f[i - 1][1 ~ 6 not include j][~] if k == 1
        //              f[i - 1][j][k - 1]                      if k != 1
        const int mod = 1000000007;
        vector<vector<int>> f1(6, vector<int>(15 + 1, 0));
        for (int i = 0;i < 6;++ i)
            f1[i][1] = 1;
        for (int i = 1;i < n;++ i) {
            vector<vector<int>> f2(6, vector<int>(15 + 1, 0));

            for (int j = 0;j < 6;++ j) {
                for (int k = 1;k <= rollMax[j];++ k) {
                    if (k == 1) {
                        int sum = 0;

                        for (int M = 0;M < 6;++ M) {
                            if (M == j) continue;
                            for (int N = 1;N <= 15;++ N) {
                                sum = ((long long)sum + f1[M][N]) % mod;
                            }
                        }
                        f2[j][k] = sum;

                    }
                    else {
                        f2[j][k] = f1[j][k - 1];
                    }
                }
            }
            swap(f1, f2);
        }
        int ans = 0;
        for (int i = 0;i < 6;++ i)
            for (int j = 1;j <= 15;++ j) {
                ans = ((long long)ans + f1[i][j]) % mod;
            }
        return ans;
    }
};
