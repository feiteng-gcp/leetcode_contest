class Solution {
public:
    int dieSimulator(int n, vector<int>& rollMax) {
        const int mod = 1e9 + 7;
        static int dp[6][16];
        memset(dp, 0, sizeof dp);
        for (int i = 0; i < 6; ++i)
            dp[i][1] = 1;
        for (int _ = 1; _ < n; ++_) {
            static int t[6][16];
            memset(t, 0, sizeof t);
            for (int i = 0; i < 6; ++i) {
                for (int j = 1; j <= 15; ++j) {
                    for (int k = 0; k < 6; ++k) {
                        if (i == k && rollMax[i] == j)
                            continue;
                        int &r = t[k][i == k ? j + 1 : 1];
                        r = (r + dp[i][j]) % mod;
                    }
                }
            }
            memcpy(dp, t, sizeof t);
        }
        int ans = 0;
        for (int i = 0; i < 6; ++i) {
            for (int j = 0; j < 16; ++j) {
                ans = (ans + dp[i][j]) % mod;
            }
        }
        return ans;
    }
};
