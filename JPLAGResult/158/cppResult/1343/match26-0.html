class Solution {
private:
    const int mod = 1000000007;
    void pd(int&ret,int val){
        ret+=val;
        if(ret>=mod)ret-=mod;
    }
public:
    int dieSimulator(int n, vector<int>& a) {
        vector<vector<vector<int>>>f=vector<vector<vector<int>>>(n, vector<vector<int>>(6, vector<int>(16, 0)));
        for(int i=0;i<6;i++)f[0][i][1] = 1;
        for(int i=0;i<n-1;i++)
            for(int j=0;j<6;j++)
                for(int k=0;k<=a[j];k++){
                    for(int r=0;r<6;r++){
                        if(r==j&&k==a[j])continue;
                        pd(f[i+1][r][1+(r==j?k:0)], f[i][j][k]);
                    }
                }
        int ret = 0;
        for(int i=0;i<6;i++)
            for(int j=0;j<=a[i];j++)pd(ret, f[n-1][i][j]);
        return ret;
    }
};
