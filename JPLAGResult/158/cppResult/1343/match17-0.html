class Solution {
public:
#define MOD 1000000007
	int dieSimulator(int n, vector<int> &rollMax) {
		vector<vector<vector<int>>>dp(n+1,vector<vector<int>>(6,vector<int>(17,0)));
		for (int i = 0; i < 6; i++) {
			dp[1][i][1]=1;
		}
		for (int i = 2; i <= n; i++) {
			for (int j = 0; j < 6; j++) {
				for (int k = 0; k < 6; k++) {
					for (int a = 1; a <= 15; a++) {
						if(j==k && a>=rollMax[j])break;
						if(j==k)dp[i][j][a + 1] = (dp[i][j][a + 1] + dp[i - 1][k][a]) % MOD;
						else dp[i][j][1] = (dp[i][j][1] + dp[i - 1][k][a]) % MOD;
					}
				}
			}
		}
		int ret=0;
		for (int i = 0; i < 6; i++) {
			for (int j = 0; j <= 15; j++) {
				ret=(ret+dp[n][i][j])%MOD;
			}
		}
		return ret;
	}
};
