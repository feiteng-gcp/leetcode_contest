typedef long long int64;
const int64 MOD = 1000000007;


class Solution {
public:
  int64 dp[5010][6][16];

  int dieSimulator(int n, vector<int>& rollMax) {
    memset(dp, 0, sizeof(dp));

    for (int d = 0; d < 6; ++d) dp[1][d][1] = 1;

    for (int i = 2; i <= n; ++i) {
      for (int d = 0; d < 6; ++d) {
        for (int k = 1; k <= rollMax[d]; ++k) {
          for (int d2 = 0; d2 < 6; ++d2) {
            for (int k2 = 0; k2 <= rollMax[d2]; ++k2) {
              if (d == d2 && k == k2 + 1) {
                dp[i][d][k] += dp[i-1][d2][k2];
                dp[i][d][k] %= MOD;
              }

              if (d != d2 && k == 1) {
                dp[i][d][k] += dp[i-1][d2][k2];
                dp[i][d][k] %= MOD;
              }
            }
          }
        }
      }
    }


    int64 ret = 0;
    for (int d = 0; d < 6; ++d) {
      for (int k = 1; k <= rollMax[d]; ++k) {
        ret += dp[n][d][k];
        ret %= MOD;
      }
    }
    return ret;
  }
};
