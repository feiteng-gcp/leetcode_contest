class Solution {
public:
    int dp[5010][6][16];
    int MOD = 1000000007;

    int dieSimulator(int n, vector<int>& rollMax) {
        memset(dp, 0, sizeof(dp));
        for(int i = 0; i < 6; ++i) {
            dp[1][i][1] = 1;
        }
        for(int i = 2; i <= n; ++i) {
            for(int j = 0; j < 6; ++j) {
                for(int k = 0; k < 6; ++k) {
                    for(int l = 1; l <= rollMax[k]; ++l) {
                        if(j == k && l + 1 > rollMax[k]) {
                            continue;
                        }
                        dp[i][j][ k == j ? l+1 : 1] += dp[i-1][k][l];

                        dp[i][j][ k == j ? l+1 : 1] %= MOD;
                    }
                }
            }
        }
        int re = 0;
        for(int k = 0; k < 6; ++k) {
            for(int l = 1; l <= rollMax[k]; ++l) {
                re += dp[n][k][l];
                re %= MOD;
            }
        }
        return re;
    }
};
