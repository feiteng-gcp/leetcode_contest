class Solution {
public:
    int mod = 1e9 + 7;
    int dp[5010][6][17];


    int solve(int n, vector<int>& A) {
        memset(dp, 0, sizeof(dp));

        for (int i = 0; i < 6; i++) {
            dp[1][i][1] = 1;
        }   

        int m = *max_element(A.begin(), A.end());

        for (int i = 1; i <= n - 1; i++) {
            for (int j = 0; j < 6; j++) {
                for (int k = 1; k <= m; k++) {
                    if (k + 1 <= A[j])
                        dp[i+1][j][k+1] = (dp[i+1][j][k+1] + dp[i][j][k]) % mod;
                    for (int l = 0; l < 6; l++) {
                        if (l != j) {
                            dp[i+1][l][1] = (dp[i+1][l][1] +  dp[i][j][k]) % mod;
                        }   

                    }   
                }   
            }   
        }   

        int ans = 0;
        for (int i = 0; i < 6; i++) {
            for (int j = 1; j <= m; j++) {
                ans = (ans + dp[n][i][j]) % mod;
            }   
        }   
        return ans;
    }
    int dieSimulator(int n, vector<int>& rollMax) {
        return solve(n, rollMax);
    }
};
