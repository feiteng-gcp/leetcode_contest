class Solution {
public:
    int balancedStringSplit(string s) {
        int res = 0;
        int count = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s[i] == 'R')
                count ++;
            else 
                count --;
            if (count == 0)
                res ++;
        }
        return res;
    }
};
