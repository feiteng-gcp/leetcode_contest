class Solution {
public:
    vector<vector<int>> ans;
    int matrix[8][8];
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        memset(matrix,0,sizeof(matrix));
        for(int i = 0; i < queens.size(); i++)
            matrix[queens[i][0]][queens[i][1]] = 1;
        for(int i = king[1]; i < 8; i++)
        {
           if(matrix[king[0]][i] == 1)
            {
                ans.push_back({king[0], i});
                break;
            } 
        }    
        for(int i = king[1]; i >= 0; i--)
        {
            if(matrix[king[0]][i] == 1)
            {
                ans.push_back({king[0], i});
                break;
            }
        }
        for(int i = king[0]; i >= 0; i--)
        {
            if(matrix[i][king[1]] == 1)
            {
                ans.push_back({i, king[1]});
                break;
            }
        }
        for(int i = king[0]; i < 8; i++)
        {
            if(matrix[i][king[1]] == 1)
            {
                ans.push_back({i, king[1]});
                break;
            }
        }
        for(int i = king[0], j = king[1]; i < 8 && j < 8; i++, j++)
        {
            if(matrix[i][j] == 1)
            {
                ans.push_back({i, j});
                break;
            }
        }
        for(int i = king[0], j = king[1]; i >= 0 && j >= 0; i--, j--)
        {
            if(matrix[i][j] == 1)
            {
                ans.push_back({i, j});
                break;
            }
        }
        for(int i = king[0], j = king[1]; i >= 0 && j < 8; i--, j++)
        {
            if(matrix[i][j] == 1)
            {
                ans.push_back({i, j});
                break;
            }
        }
        for(int i = king[0], j = king[1]; i < 8 && j >= 0; i++, j--)
        {
            if(matrix[i][j] == 1)
            {
                ans.push_back({i, j});
                break;
            }
        }
        return ans;   
       
    }
};
