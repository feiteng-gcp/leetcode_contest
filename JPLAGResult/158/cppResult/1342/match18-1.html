class Solution {
public:
	vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
		int x=king[0],y=king[1];
		set<pair<int,int>>all;
		for (auto it : queens) {
			all.insert({ it[0],it[1] });
		}

		vector<vector<int>>ret;
		for (int i = x; i >= 0; i--) {
			if (all.find({ i,y }) != all.end()) {
				ret.push_back({ i,y });
				break;
			}
		}
		for (int i = x; i <8; i++) {
			if (all.find({ i,y }) != all.end()) {
				ret.push_back({ i,y });
				break;
			}
		}
		for (int i = y; i >= 0; i--) {
			if (all.find({ x,i }) != all.end()) {
				ret.push_back({ x,i });
				break;
			}
		}
		for (int i = y; i <8; i++) {
			if (all.find({ x,i }) != all.end()) {
				ret.push_back({ x,i });
				break;
			}
		}
		for (int i = 1; i < 8; i++) {
			int xx=i+x,yy=i+y;
			if(xx>=8 || yy>=8)break;

			if (all.find({ xx,yy }) != all.end()) {
				ret.push_back({ xx,yy });
				break;
			}
		}
		for (int i = -1; i>=-8; i--) {
			int xx = i + x, yy = i + y;
			if (xx <0 || yy <0)break;

			if (all.find({ xx,yy }) != all.end()) {
				ret.push_back({ xx,yy });
				break;
			}
		}
		for (int i = 1; i < 8; i++) {
			int xx = i + x, yy = y-i;
			//if (xx >= 8 || yy >= 8)break;

			if (all.find({ xx,yy }) != all.end()) {
				ret.push_back({ xx,yy });
				break;
			}
		}
		for (int i = -1; i >= -8; i--) {
			int xx = i + x, yy = y-i;
			//if (xx < 0 || yy < 0)break;

			if (all.find({ xx,yy }) != all.end()) {
				ret.push_back({ xx,yy });
				break;
			}
		}
		return ret;
	}
};
