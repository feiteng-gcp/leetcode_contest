class Solution {
public:
  static bool attacks(int qr, int qc, int kr, int kc) {
    if (qr == kr || kc == kc)
      return true;
    for(int i = -8; i <= 8; ++i) {
      if (qr + i == kr && qc + i == kc)
        return true;
      if (qr + i == kr && qc - i == kc)
        return true;
    }
    return false;
  }
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
      vector<vector<int>> ret;
      ret.reserve(queens.size());
      int kr = king[0], kc = king[1];
      set<int> qs;
      for(auto& q : queens)
        qs.insert(q[0] *8 + q[1]);
      
      for(int c = kc  + 1; c < 8; ++c) {
        if (qs.count(kr * 8 + c)) {
          ret.push_back(vector<int>{kr, c});
          break;
        }
      }
      for(int c = kc - 1; c >= 0; --c) {
        if (qs.count(kr * 8 + c)) {
          ret.push_back(vector<int>{kr, c});
          break;
        }
      }
       for(int r = kr  + 1; r < 8; ++r) {
        if (qs.count(r * 8 + kc)) {
          ret.push_back(vector<int>{r, kc});
          break;
        }
      }
      for(int r = kr - 1; r >= 0; --r) {
        if (qs.count(r * 8 + kc)) {
          ret.push_back(vector<int>{r, kc});
          break;
        }
      }
      
      for(int r = kr + 1, c = kc + 1; r < 8 && c < 8; ++r, ++c) {
        if (qs.count(r * 8 + c)) {
          ret.push_back(vector<int>{r, c});
          break;
        }
      }
       
      for(int r = kr - 1, c = kc - 1; r >= 0 && c >= 0; --r, --c) {
        if (qs.count(r * 8 + c)) {
          ret.push_back(vector<int>{r, c});
          break;
        }
      }
        
      for(int r = kr + 1, c = kc - 1; r < 8 && c >= 0; ++r, --c) {
        if (qs.count(r * 8 + c)) {
          ret.push_back(vector<int>{r, c});
          break;
        }
      }
       
      for(int r = kr - 1, c = kc + 1; r >= 0 && c < 8; --r, ++c) {
        if (qs.count(r * 8 + c)) {
          ret.push_back(vector<int>{r, c});
          break;
        }
      }
                    
           
      
      
      return ret;
      
    }
};
