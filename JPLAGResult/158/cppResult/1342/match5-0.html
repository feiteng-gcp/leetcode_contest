class Solution {
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        vector<vector<int>> result;
        set<vector<int>> qs(queens.begin(), queens.end());
        for (int dx = -1; dx <= 1; ++dx) {
            for (int dy = -1; dy <= 1; ++dy) {
                if (dx == 0 && dy == 0)
                    continue;
                int next_x = king[0] + dx;
                int next_y = king[1] + dy;
                while (0 <= next_x && next_x < 8 &&
                       0 <= next_y && next_y < 8) {
                    if (qs.count({next_x, next_y}) > 0) {
                        result.push_back({next_x, next_y});
                        break;
                    }
                    next_x += dx;
                    next_y += dy;
                }
            }
        }
        return result;
    }
};
