class Solution {
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        set<vector<int>> s(queens.begin(), queens.end());
        int x = king[0], y = king[1];
        vector<vector<int>> ans;
        for (int i=0; i+x<8; i++) {
            if (s.find({i+x, y}) != s.end()) {
                ans.push_back({x+i, y});
                break;
            }
        }
        for (int i=0; x-i>=0; i++) {
            if (s.find({x-i, y}) != s.end()) {
                ans.push_back({x-i, y});
                break;
            }
        }
        for (int i=0; i+y<8; i++) {
            if (s.find({x, i+y}) != s.end()) {
                ans.push_back({x, y+i});
                break;
            }
        }
        for (int i=0; y-i>=0; i++) {
            if (s.find({x, y-i}) != s.end()) {
                ans.push_back({x, y-i});
                break;
            }
        }
        for (int i=0; i+y<8&&i+x<8; i++) {
            if (s.find({i+x, i+y}) != s.end()) {
                ans.push_back({x+i, y+i});
                break;
            }
        }
        for (int i=0; i+y<8&&x-i>=0; i++) {
            if (s.find({x-i, i+y}) != s.end()) {
                ans.push_back({x-i, y+i});
                break;
            }
        }
        for (int i=0; y-i>=0&&i+x<8; i++) {
            if (s.find({i+x, y-i}) != s.end()) {
                ans.push_back({x+i, y-i});
                break;
            }
        }
        for (int i=0; y-i>=0&&x-i>=0; i++) {
            if (s.find({x-i, y-i}) != s.end()) {
                ans.push_back({x-i, y-i});
                break;
            }
        }
        return ans;
    }
};
