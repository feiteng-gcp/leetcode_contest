class Solution {
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        int n = 10;
        vector<vector<int>>ans;
        vector<vector<int>> mp;
        for(int i =0;i<n;i++){
            vector<int> row;
            for(int j=0;j<n;j++){
                row.push_back(0);
            }
            mp.push_back(row);
        }
        for(auto item:queens){
            mp[item[0]][item[1]] = 1;
        }
        int di[] = {0,-1,-1,-1,0,1,1,1};
        int dj[] = {-1,-1,0,1,1,1,0,-1};
        for(int k =0;k<8;k++){
            int ni=di[k]+king[0];
            int nj=dj[k]+king[1];
            while(!(ni < 0 || nj < 0 || ni ==n || nj == n)){
                if(mp[ni][nj] == 1){
                    vector<int> anspos;
                    anspos.push_back(ni);
                    anspos.push_back(nj);
                    ans.push_back(anspos);
                    break;
                }
                ni+=di[k];
                nj+=dj[k];
            }
        }
        return ans;
    }
};
