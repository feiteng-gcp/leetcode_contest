class Solution {
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        vector<vector<int>> qs(8, vector<int>(8));
        for(auto& q : queens) {
            qs[q[0]][q[1]] = 1;
        }
        vector<vector<int>> res;
        int x = king[0], y = king[1];
        // down
        for(int i = x+1; i < 8; i++) {
            if(qs[i][y]) {
                res.push_back({i, y});
                break;
            }
        }
        // up
        for(int i = x-1; i >= 0; i--) {
            if(qs[i][y]) {
                res.push_back({i, y});
                break;
            }
        }
        // right
        for(int i = y+1; i < 8; i++) {
            if(qs[x][i]) {
                res.push_back({x, i});
                break;
            }
        }
        // left
        for(int i = y-1; i >= 0; i--) {
            if(qs[x][i]) {
                res.push_back({x, i});
                break;
            }
        }
        // left-up
        for(int i = x-1, j = y-1; i < 8 && i >= 0 && j < 8 && j >= 0; i--,j--) {
            if(qs[i][j]) {
                res.push_back({i, j});
                break;
            }
        }
        // left-down
        for(int i = x+1, j = y-1; i < 8 && i >= 0 && j < 8 && j >= 0; i++,j--) {
            if(qs[i][j]) {
                res.push_back({i, j});
                break;
            }
        }
        // right-up
        for(int i = x-1, j = y+1; i < 8 && i >= 0 && j < 8 && j >= 0; i--,j++) {
            if(qs[i][j]) {
                res.push_back({i, j});
                break;
            }
        }
        // right-down
        for(int i = x+1, j = y+1; i < 8 && i >= 0 && j < 8 && j >= 0; i++,j++) {
            if(qs[i][j]) {
                res.push_back({i, j});
                break;
            }
        }
        return res;
    }
};
