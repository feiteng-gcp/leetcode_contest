class Solution {
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        vector<vector<int>> res;
        unordered_map<int, unordered_set<int>> u;
        for (auto q : queens) {
            u[q[0]].insert(q[1]);
        }
        for (int i = king[0]; i < 8; i++) {
            if (u[i].find(king[1]) != u[i].end()) {
                res.push_back({i, king[1]});
                break;
            }
        }
        for (int i = king[0]; i >= 0; i--) {
            if (u[i].find(king[1]) != u[i].end()) {
                res.push_back({i, king[1]});
                break;
            }
        }
        for (int i = king[1]; i < 8; i++) {
            if (u[king[0]].find(i) != u[king[0]].end()) {
                res.push_back({king[0], i});
                break;
            }
        }
        for (int i = king[1]; i >= 0; i--) {
            if (u[king[0]].find(i) != u[king[0]].end()) {
                res.push_back({king[0], i});
                break;
            }
        }
        for (int i = king[0], j = king[1]; i >= 0 && i < 8 && j >= 0 && j < 8; i++, j++) {
            if (u[i].find(j) != u[i].end()) {
                res.push_back({i, j});
                break;
            }
        }
        
        for (int i = king[0], j = king[1]; i >= 0 && i < 8 && j >= 0 && j < 8; i--, j++) {
            if (u[i].find(j) != u[i].end()) {
                res.push_back({i, j});
                break;
            }
        }
        
        for (int i = king[0], j = king[1]; i >= 0 && i < 8 && j >= 0 && j < 8; i++, j--) {
            if (u[i].find(j) != u[i].end()) {
                res.push_back({i, j});
                break;
            }
        }
        
        for (int i = king[0], j = king[1]; i >= 0 && i < 8 && j >= 0 && j < 8; i--, j--) {
            if (u[i].find(j) != u[i].end()) {
                res.push_back({i, j});
                break;
            }
        }
        return res;
    }
};
