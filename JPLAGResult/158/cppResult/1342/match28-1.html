class Solution {
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        int x = king[0];
        int y = king[1];
        map<pair<int,int>,int > mp;
        for(int i = 0;i<queens.size();i++) {
            int xx = queens[i][0];
            int yy = queens[i][1];
            mp[make_pair(xx,yy)] = 1;
        }
        vector<vector<int> > ans;


        for(int i = x, j = y;i<8;i++) {
            if(mp[{i,j}]==1) {
                ans.push_back({i,j});
                break;
            }
        }
        for(int i = x, j = y;i>=0;i--) {

            if(mp[{i,j}]==1) {
                ans.push_back({i,j});
                break;
            }
        }

        for(int i = x, j = y;j<8;j++) {

            if(mp[{i,j}]==1) {
                ans.push_back({i,j});
                break;
            }
        }
        for(int i = x, j = y;j>=0;j--) {
            if(mp[{i,j}]==1) {
                ans.push_back({i,j});
                break;
            }
        }


        for(int i = x, j = y;i<8&&j<8;i++,j++) {
            if(mp[{i,j}]==1) {
                ans.push_back({i,j});
                break;
            }
        }
        for(int i = x, j = y;i>=0&&j>=0;i--,j--) {

            if(mp[{i,j}]==1) {
                ans.push_back({i,j});
                break;
            }
        }

        for(int i = x, j = y;j<8&&i>=0;j++,i--) {

            if(mp[{i,j}]==1) {
                ans.push_back({i,j});
                break;
            }
        }
        for(int i = x, j = y;j>=0&&i<8;j--,i++) {

            if(mp[{i,j}]==1) {
                ans.push_back({i,j});
                break;
            }
        }

        return ans;
    }
};
