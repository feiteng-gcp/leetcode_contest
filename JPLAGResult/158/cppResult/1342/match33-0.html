class Solution {
    const vector<int> d = {1, 0, -1, 0, 1, 1, -1, -1, 1};
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        set<vector<int>> q(queens.begin(), queens.end());
        vector<vector<int>> res;
        for (int i=0; i<8; i++) {
            for (int x=king[0], y=king[1]; x>-1 && y>-1 && x<8 && y<8; x+=d[i], y+=d[i+1]) {
                if (q.count({x, y})) {
                    res.push_back({x, y});
                    break;
                }
            }
        }
        return res;
    }
};
