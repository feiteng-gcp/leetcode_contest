class Solution {
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        /*
        vector<vector<int>> ret;
        for (int i = 0;i < queens.size();i++) {
            if (queens[i][0] == king[0] || queens[i][1] == king [1] || queens[i][0] + queens[i][1] == king[0] + king[1] || queens[i][0] - queens[i][1] == king[0] - king[1])
                ret.push_back({queens[i][0], queens[i][1]});
            
        }
        */
        set<pair<int, int>> s;
        for (int i = 0;i < queens.size();i++) {
            s.insert({queens[i][0], queens[i][1]});
        }
        vector<vector<int>> ret;
        for (int i = king[0]+1;i < 8;i++) {
            if (s.count({i, king[1]})) {
                ret.push_back({i, king[1]});
                break;
            }
        }
        for (int i = king[1]+1;i < 8;i++) {
            if (s.count({king[0], i})) {
                ret.push_back({king[0], i});
                break;
            }
        }
        for (int i = king[0]-1;i >= 0;i--) {
            if (s.count({i, king[1]})) {
                ret.push_back({i, king[1]});
                break;
            }
        }
        for (int i = king[1]-1;i >= 0;i--) {
            if (s.count({king[0], i})) {
                ret.push_back({king[0], i});
                break;
            }
        }
        int i, j;
        for (i = king[0]+1, j = king[1]+1;i < 8 && j < 8;i++, j++) {
            if (s.count({i, j})) {
                ret.push_back({i, j});
                break;
            }
        }
        for (i = king[0]+1, j = king[1]-1;i < 8 && j >= 0;i++, j--) {
            if (s.count({i, j})) {
                ret.push_back({i, j});
                break;
            }
        }
        for (i = king[0]-1, j = king[1]+1;i >= 0 && j < 8;i--, j++) {
            if (s.count({i, j})) {
                ret.push_back({i, j});
                break;
            }
        }
        for (i = king[0]-1, j = king[1]-1;i >= 0 && j >= 0;i--, j--) {
            if (s.count({i, j})) {
                ret.push_back({i, j});
                break;
            }
        }
        return ret;
        
    }
};
