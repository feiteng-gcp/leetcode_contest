class Solution {
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        std::set<std::pair<int, int>> qset;
        for (const auto& q : queens) {
            qset.insert({q[0], q[1]});
        }
        std::vector<vector<int>> ans;
        int x = king[0], y = king[1];
        for (int tx = x+1; tx < 8; ++tx) {
            if (qset.count({tx, y})) {
                ans.push_back({tx ,y});
                break;
            }
        }
        
        for (int tx = x-1; tx >= 0; --tx) {
            if (qset.count({tx, y})) {
                ans.push_back({tx ,y});
                break;
            }
        }
        
        for (int ty = y+1; ty < 8; ++ty) {
            if (qset.count({x, ty})) {
                ans.push_back({x, ty});
                break;
            }
        }
        for (int ty = y-1; ty >= 0; --ty) {
            if (qset.count({x, ty})) {
                ans.push_back({x, ty});
                break;
            }
        }
        
        for (int tx = x+1, ty = y+1; tx < 8 && ty < 8; ++tx, ++ty) {
            if (qset.count({tx, ty})) {
                ans.push_back({tx, ty});
                break;
            }
        }
        
        for (int tx = x-1, ty = y-1; tx >= 0 && ty >= 0; --tx, --ty) {
            if (qset.count({tx, ty})) {
                ans.push_back({tx, ty});
                break;
            }
        }
        
        for (int tx = x+1, ty = y-1; tx < 8 && ty >= 0; ++tx, --ty) {
            if (qset.count({tx, ty})) {
                ans.push_back({tx, ty});
                break;
            }
        }
        
        for (int tx = x-1, ty = y+1; tx >= 0 && ty < 8; --tx, ++ty) {
            if (qset.count({tx, ty})) {
                ans.push_back({tx, ty});
                break;
            }
        }
        return ans;
    }
};
