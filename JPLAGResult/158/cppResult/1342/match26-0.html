class Solution {
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        int n = queens.size();
        int row = king[0];
        int col = king[1];
        
        vector<vector<bool>> hasQ(8, vector<bool>(8, false));
        for (auto& q : queens) {
            hasQ[q[0]][q[1]] = true;
        }
        vector<vector<int>> res;
        for (int y = row + 1; y < 8; ++y) {
            if (hasQ[y][col]) {
                res.push_back({y, col});
                break;
            }
        }
        for (int y = row - 1; y >= 0; --y) {
            if (hasQ[y][col]) {
                res.push_back({y, col});
                break;
            }
        }
                        
        for (int x = col + 1; x < 8; ++x) {
            if (hasQ[row][x]) {
                res.push_back({row, x});
                break;
            }
        }
        for (int x = col - 1; x >= 0; --x) {
            if (hasQ[row][x]) {
                res.push_back({row, x});
                break;
            }
        }
        
        for (int y = row + 1, x = col + 1; y<8 && x<8; ++y, ++x) {
            if (hasQ[y][x]) {
                res.push_back({y, x});
                break;
            }
        }
        for (int y = row + 1, x = col - 1; y<8 && x>=0; ++y, --x) {
            if (hasQ[y][x]) {
                res.push_back({y, x});
                break;
            }
        }
        for (int y = row - 1, x = col + 1; y>=0 && x<8; --y, ++x) {
            if (hasQ[y][x]) {
                res.push_back({y, x});
                break;
            }
        }
        for (int y = row - 1, x = col - 1; y>=0 && x>=0; --y, --x) {
            if (hasQ[y][x]) {
                res.push_back({y, x});
                break;
            }
        }
        
        return res;
    }
};
