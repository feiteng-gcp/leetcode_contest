class Solution {
    vector<int> compare;
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        compare = king;
        vector<vector<int>> x(2);
        vector<vector<int>> y(2);
        vector<vector<int>> xpy(2);
        vector<vector<int>> xmy(2);
        for(auto & cur:queens) {
            if(cur[0]==king[0]) {
                if(cur[1]>king[1])
                    update(x[0], cur);
                else
                    update(x[1], cur);
            } else if(cur[1]==king[1]) {
                if(cur[0]>king[0])
                    update(y[0], cur);
                else
                    update(y[1], cur);
            } else if(cur[0]+cur[1]==king[0]+king[1]) {
                if(cur[0]>king[0])
                    update(xpy[0], cur);
                else
                    update(xpy[1], cur);
            } else if(cur[0]-cur[1]==king[0]-king[1]) {
                if(cur[0]>king[0])
                    update(xmy[0], cur);
                else
                    update(xmy[1], cur);

            }
        }
        vector<vector<int>> res;
        add(res, x);
        add(res, y);
        add(res, xpy);
        add(res, xmy);
        return res;
    }
    
    int dist(vector<int> & pos) {
        return abs(compare[0]-pos[0]) + abs(compare[1]-pos[1]);
    }
    
    void add(vector<vector<int>> & res, vector<vector<int>> & target) {
        if(!target[0].empty())
            res.push_back(target[0]);
        if(!target[1].empty())
            res.push_back(target[1]);
    }
    
    void update(vector<int> & left, vector<int> & right) {
        if(left.empty())
            left = right;
        else {
            if(dist(right)<dist(left))
                left = right;
        }
    }
};
