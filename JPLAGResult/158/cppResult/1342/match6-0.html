class Solution {
public:
    vector<vector<int>> queensAttacktheKing(vector<vector<int>>& queens, vector<int>& king) {
        set<vector<int>> q(queens.begin(), queens.end());
        vector<vector<int>> res;
        for (int i = -1; i <= 1; i++) {
            for (int j = -1; j <= 1; j++) {
                if (i == 0 && j == 0)
                    continue;
                int x = king[0] + i, y = king[1] + j;
                while (x >= 0 && x < 8 && y >= 0 && y < 8) {
                    if (q.count({x, y})) {
                        res.push_back({x, y});
                        break;
                    }
                    x += i;
                    y += j;
                }
            }
        }
        return res;
    }
};
