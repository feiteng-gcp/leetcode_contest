class Solution {
public:
    vector<int> prisonAfterNDays(vector<int> cells, int N) {
        vector<vector<int>> states;
        states.push_back(cells);

        int repeat = -1;
        while (true) {
            auto cur = states.back();
            vector<int> next(8, 0);
            for (int i = 1; i < 7; i++) {
                if (cur[i - 1] == cur[i + 1]) {
                    next[i] = 1;
                }
            }
            for (int i = 0; i < states.size(); i++) {
                if (next == states[i]) {
                    repeat = i;
                    break;
                }
            }
            if (repeat >= 0) break;
            states.push_back(next);
        }
        int idx = N < states.size() ? N : ((N - repeat) % (states.size() - repeat) + repeat);
        return states[idx];
    }
};
