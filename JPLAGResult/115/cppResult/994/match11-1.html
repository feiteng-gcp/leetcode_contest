class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        map<vector<int>,int >m;
        map<int,vector<int> >m2;
        
        m[cells]=0;        
        m2[0]=cells;
        int start=-1;

        int end=-1;
        
        for(int i=1;i<10000;i++){
            vector<int>x=cells;
            for(int j=1;j<7;j++){
                if(cells[j-1]==cells[j+1] )
                    x[j]=1;
                else
                    x[j]=0;
            }
            x[0]=x[7]=0;
            cells=x;
            if(i==N)return x;
            
            if(m.find(x)==m.end()){
                m[x]=i;   
                m2[i]=x;
            }else{
                start=m[x];
                end=i-1;
                break;
            }
            
        }
        
        int T=end-start+1;
                
        return m2[start+(N-start)%T];
    }
};
