class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        p = vector<int>(10);
        p[0] = 1;
        for (int i = 1; i < 10; ++i) p[i] = p[i - 1] * 2;
        
        vector<int> f(p[9], -1);
        int opt = getOpt(cells);
        f[opt] = 0;
        
        while (N > 0) {
            --N;

            vector<int> a = cells;
            for (int i = 1; i < 7; ++i) {
                if (cells[i - 1] == cells[i + 1]) a[i] = 1;
                else a[i] = 0;
            }
            a[0] = 0;
            a[7] = 0;
            
            int newOpt = getOpt(a);
            if (f[newOpt] > -1) {
                int len = f[opt] - f[newOpt] + 1;
                N %= len;
            }
            f[newOpt] = f[opt] + 1;
            cells = a;
            opt = newOpt;
        }
        return cells;
    }

private:
    vector<int> p;
    

    int getOpt(vector<int> &a) {
        int opt = 0;
        for (int i = 0; i < 8; ++i) opt = opt + p[i] * a[i];
        return opt;
    }    
};
