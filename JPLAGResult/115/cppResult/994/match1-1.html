class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        
        int s = 0;
        for(int i=0;i<8;i++)
            s += cells[i]*pow(2,i);
        vector<int> x{s};
        vector<vector<int>> res{cells};
        
        for(int i=0;i<N;i++)
        {
            vector<int> tt(8,0);
            for(int j=1;j<7;j++)
                tt[j] = !(res[i][j-1] ^ res[i][j+1]);
            s = 0;
            for(int i=0;i<8;i++)
                s += tt[i]*pow(2,i);
            if(i>0 && s == x[1])
            {
                int bb = (N-1) % (x.size()-1);
                
                return res[bb+1];
            }
            x.push_back(s);
            res.push_back(tt);
        }
        return res.back();
    }
};
