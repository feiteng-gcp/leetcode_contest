class Solution {
public:
vector<int> prisonAfterNDays(vector<int>& cells, int N) {
	vector<vector<int>> prisons;
	for (int n = 0; n < N; ++n)
	{
		vector<int> cur(cells.size());
		cur[0] = 0;
		cur[cells.size() - 1] = 0;
		for (int i = 1; i < cells.size() - 1; ++i)
		{
			if (cells[i - 1] == cells[i + 1])
			{
				cur[i] = 1;
			}
			else
			{
				cur[i] = 0;
			}
		}
		int index = -1;
		for (int i = 0; i < prisons.size(); ++i)
		{
			if (prisons[i] == cur)
			{
				index = i;
				break;
			}
		}
		if (index == -1)
		{
			prisons.push_back(cur);
		}
		else
		{
			cells = prisons[(N -1 - index) % prisons.size()];
			break;
		}
		cells = cur;
	}
	return cells;
}
};
