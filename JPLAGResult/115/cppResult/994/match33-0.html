class Solution {
    int digits[8] = {1,2,4,8,16,32,64,128};
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int count = 1;
        vector<int> pos(256);
        int idx = toint(cells);
        pos[idx] = 1;
        while(N){
            next(cells);
            --N;
            ++count;
            idx = toint(cells);
            if (pos[idx]) {
                count -= pos[idx];
                break;
            }
            pos[idx] = count;
        }
      //  cout << count;
        N %= count;
        while(N) {
            --N;
            next(cells);
        }
        return cells;
    }
    int toint(vector<int>& cells){
        int res = 0;
        
        for (int i = 0; i < 8; ++i){
            res |= cells[i] * digits[i];
        }
        return res;
    }
    void next(vector<int>& cells){
        int prev = cells[0];
        cells[0] = 0;
        for (int i = 1; i < 7; ++i){
            if (prev ^ cells[i + 1]) {
                prev = cells[i];
                cells[i] = 0;
            }
            else {
                prev = cells[i];
                cells[i] = 1;
            }
        }
        cells[7] = 0;
    }
};
