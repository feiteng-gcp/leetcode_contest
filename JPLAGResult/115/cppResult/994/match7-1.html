class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        N = N-1;
        N %= 14;


        for (int i = 0; i <= N; i++) {
            vector<int> t(8);
            t[0] = t[7] = 0;
            for (int j = 1; j <= 6; j++) {
                if (cells[j-1] == cells[j+1])
                    t[j] = 1;
                else
                    t[j] = 0;
            }
            cells[0] = cells[7] = 0;
            for (int j = 1; j <= 6; j++)
                cells[j] = t[j];
        }
        return cells;
    }
};
