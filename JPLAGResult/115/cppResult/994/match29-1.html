class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        unordered_map<int, int> map;
        vector<vector<int>> table;
        int step = 0;
        int cur = 0;
        while (step < N) {
            cur = 0;
            for (auto i : cells) {
                cur = (cur | i) << 1;
            }
            if (map.count(cur)) break;
            table.push_back(cells);
            map[cur] = table.size() - 1;
            vector<int> next(8, 0);
            for (int i = 1; i <= 6; ++i) {
                next[i] = cells[i - 1] == cells[i + 1];
            }
            cells = next;
            ++step;
        }
        if (step == N) return cells;
        int cycle = step - map[cur];
        return table[map[cur] + (N - step) % cycle];
    }
};
