class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        vector<vector<int>> states;
        
        states.push_back(cells);
        
        int n = cells.size();
        
        int beg = 0;
        while (true) {
            auto& cur = states.back();
            vector<int> nxt(n, 0);
            for (int i = 1; i < n - 1; ++i) {
                if ((cur[i - 1] ^ cur[i + 1]) == 0) {
                    nxt[i] = 1;
                }
            }
            
            bool newState = true;
            for (int i = 0; i < states.size(); ++i) {
                if (states[i] == nxt) {
                    beg = i;
                    newState = false;
                    break;
                }
            }
            
            if (newState) {
                states.emplace_back(nxt);
            } else {
                break;
            }
        }
        
        if (N < beg) {
            return states[N];
        }
        int pd = states.size() - beg;
        int idx = beg + (N - beg) % pd;
        return states[idx];
    }
};
