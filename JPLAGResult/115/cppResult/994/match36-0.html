typedef vector<int> VI;
typedef vector<VI> VVI;
typedef vector<bool> VB;

class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        map< vector<bool>, int > seen;

        VB cur(8);
        for (int i = 0; i < 8; ++i)
            cur[i] = cells[i];

        vector< VB > A;
        A.push_back(cur);
        seen[cur] = 0;

        /*
        cerr << "Day 0:"; 
        for (int x : cur)
            cerr << ' ' << x;
        cerr << endl;
        */

        for (int n = 1; ; ++n) {
            VB nxt(8);
            for (int i = 1; i < 7; ++i) {
                nxt[i] = (cur[i-1] && cur[i+1]) ||
                         (!cur[i-1] && !cur[i+1]);
            }

            /*
            cerr << "Day " << n << ":"; 
            for (int x : nxt)
                cerr << ' ' << x;
            cerr << endl;
            */
            
            if (n == N) {
                VI res(8);
                for (int i = 0; i < 8; ++i)
                    res[i] = nxt[i] ? 1 : 0;
                return res;
            }
            
            if (seen.count(nxt)) {
                int start = seen[nxt];
                int period = n - start;

                VI res(8);
                int idx = (N - start) % period + start;
            //  cerr << "idx: " << idx << endl;
                for (int i = 0; i < 8; ++i)
                    res[i] = A[idx][i] ? 1 : 0;

                return res;
            }

            A.push_back(nxt);
            seen[nxt] = n;
            cur = nxt;
        }
        
        return {};
    }
};
