class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        map<int,int> day2pat;
        map<int,int> pat2day;
        map<int,vector<int>> pat2cells;
        int pat = getNum(cells);
        day2pat[0] = pat;
        pat2day[pat] = 0;
        pat2cells[pat] = cells;
        
        vector<int> new_cells = cells;
        int begin = 0;
        int cycle = 0;
        for (int i = 1; i <= N; ++i) {
            new_cells = nextday(new_cells);
            int new_pat = getNum(new_cells);
            pat2cells[new_pat] = new_cells;
            if (pat2day.count(new_pat) > 0) {
                begin = i;
                cycle = i - pat2day[new_pat];
                break;
            } else {
                pat2day[new_pat] = i;
                day2pat[i] = new_pat;
            }
        }
        if (cycle > 0) {
            int d = (N-begin) % cycle;
            int pat = day2pat[begin-cycle+d];
            return pat2cells[pat];
        } else {
            return new_cells;
        }
    }
    

    vector<int> nextday(vector<int>& cells) {
        vector<int> ret = cells;
        for (int i = 1; i < cells.size()-1; ++i) {
            ret[i] = (cells[i-1] == cells[i+1]);
        }
        ret[0] = 0;
        ret.back() = 0;

        return ret;
    }
    
    int getNum(vector<int> &cells) {
        int ret = 0;
        for (int i = 0; i < 8; ++i) {
            ret = ret*2 + cells[i];
        }
        return ret;
    }
};
