class Solution {
public:
    vector<int> tran(int num) {
        vector<int> res;
        for(int i = 0; i < 8; i++) {
            if(num & (1<<i))
                res.push_back(1);
            else res.push_back(0);
        }
       // reverse(res.begin(), res.end());
        return res;
    }

    vector<int> prisonAfterNDays(vector<int>& cell, int N) {
        vector<int> nxt(8,0);
        vector<int> states;
        map<int, int> hash;
        int period,start;
        start = period = -1;

        for(int i = 0; i < N; i++) {
            int base = 1;
            int num = 0;
            for(int j = 0; j < 8; j++) {
                num += (base * cell[j]);
                base *= 2;
            }
            states.push_back(num);
            // cout << "num = " << num << endl;
            // cout << "i = " << i << endl;
            // for(int j = 0; j < 8; j++)
            //     cout << cell[j] << " ";
            // cout << endl << endl;
            if(hash.find(num) != hash.end()) {
                start = hash[num];
                period = i - hash[num];
                break;
            }
            else hash[num] = i;
            nxt[0] = nxt[7] = 0;
            for(int j = 1; j < 7; j++) 
                nxt[j] = !(cell[j-1] ^ cell[j+1]);
           cell = nxt;
        }
        if(start == -1) return cell;
      // cout << "start = " << start << " period = " << period << endl;
        int r = (N-start) % period;
        int ans = states[start + r];
       //cout << "r = " << r << " ans = " << ans << endl;
        return tran(ans);
    }
};
