class Solution {
public:
    int getState(vector<int>& cells) {
        int sum = 0;
        for (int i = cells.size()-1; i >= 0; --i) {
            sum = (sum << 1) + (cells[i] & 1);
        }
        return sum;
    }
    vector<int> getCells(int num) {
        vector<int> result(8, 0);
        for (int i = 7; i >= 0; --i) {
            result[i] = (num >> i) & 1;
        }
        return result;
    }
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        unordered_map<int, int> states; // state to nth day
        unordered_map<int, int> day2states;
        states.insert(make_pair(getState(cells), 0));
        day2states.insert(make_pair(0, getState(cells)));
        for (int i = 1; i <= N; ++i) {
            vector<int> tmp(cells.size(), 0);
            for (int j = 1; j < cells.size()-1; ++j) {
                if (cells[j-1] ^ cells[j+1]) tmp[j] = 0;
                else tmp[j] = 1;
            }
            int st = getState(tmp);
            auto got = states.find(st);
            if (got == states.end()) {
                states.insert(make_pair(st, i));
                day2states.insert(make_pair(i, st));
                cells = tmp;
            } else {
                // find repeating
                cout << i << " " << got->second << endl;
                int cycle = i - got->second;
                int mod = (N - got->second) % cycle;
                int target = day2states[got->second + mod];
                return getCells(target);
            }
        }
        return cells;
    }
};
