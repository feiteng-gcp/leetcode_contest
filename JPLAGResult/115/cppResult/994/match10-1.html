class Solution {
public:
	int convert(vector<int>& cells) {
		int ret = 0;
		for (int i = 0; i<cells.size(); i++) {
			ret += cells[i];
			ret *= 2;
		}
		return ret;
	}
	vector<int> rev(int num) {
		vector<int> ret;

		for(int i=0;i<8;i++) {
			ret.push_back(num % 2);
			num /= 2;
		}
		reverse(ret.begin(), ret.end());
		return ret;
	}
	vector<int> next(vector<int> cells) {
		vector<int> newvec(8, 0);
		newvec[0] = 0;
		newvec[7] = 0;
		for (int i = 1; i<7; i++) {
			if (!(cells[i - 1] ^ cells[i + 1]))
				newvec[i] = 1;
		}
		return newvec;
	}

	vector<int> prisonAfterNDays(vector<int>& cells, int N) {
		map<int, int> m;
		for (int i = 0; i<N; i++) {
			int k = convert(cells);
			if (m.find(k) == m.end())
				m[k] = i;
			else {
				int div = i - m[k];
				int rem = (N - m[k]) % div;
				vector<int> ans = cells;
				for (int j = 0; j < rem; j++) {
					ans = next(ans);
				}
				return ans;
			}			
			cells = next(cells);
		}
		return cells;
	}
};
