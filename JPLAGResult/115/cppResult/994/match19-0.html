class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int n = cells.size();
        N = N % 14 + 14;
        for (int i = 0; i < N; i++) {
            vector<int> tmp(n, -1);
            for (int j = 0; j < n; j++) {
                if (j == 0) {
                    tmp[j] = 0; 
                } else if (j == n - 1) {
                    tmp[j] = 0;
                } else {
                    tmp[j] = cells[j - 1] == cells[j + 1] ? 1 : 0;
                }
            }
            cells = tmp;
        }
        return cells;
    }
};
