class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int prev[8], now[8];
        for (int i = 0; i < 8; i++) {
            now[i] = prev[i] = cells[i];
        }
        
        N--;

        now[0] = now[7] = 0;
        for (int j = 1; j < 7; j++) {
            if (prev[j - 1] == prev[j + 1]) {
                now[j] = 1;
            }
            else {
                now[j] = 0;
            }
        }
        memcpy(prev, now, sizeof(now));
        
        vector<int> cache[64];
        for (int i = 0; i < 8; i++) {
            cache[0].push_back(prev[i]);
        }
        
        int i = 1;
        for (; i <= 64; i++) {
            now[0] = now[7] = 0;
            for (int j = 1; j < 7; j++) {
                if (prev[j - 1] == prev[j + 1]) {
                    now[j] = 1;
                }
                else {
                    now[j] = 0;
                }
            }
            memcpy(prev, now, sizeof(now));
            
            vector<int> ans;
            for (int i = 0; i < 8; i++) {
                ans.push_back(prev[i]);
            }
            cache[i] = ans;
            
            if (ans == cache[0]) {
                break;
            }
        }

        return cache[N % i];
    }
};
