class Solution {
public:
    int next(int cells){
        int nxt = 0;
        for(int i=1; i+1<8; i++){
            if( ((cells>>(i-1))&1) == ((cells>>(i+1))&1) ){
                nxt |= 1<<i;
            }
        }
        return nxt;
    }
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        vector<int> last(1<<8, -1);
        int s = 0;
        for(int i=0; i<cells.size(); i++){
            if(cells[i]){
                s |= 1<<i;
            }
        }
        last[s] = 0;

        for(int i=1; i<=N; i++){
            int t = next(s);
            if(last[t] == -1) last[t] = i;
            else{
                int r = N-i;
                int d = i - last[t];
                r %= d;
                N = i+r;
            }
            s = t;
        }
        
        vector<int> res(8, 0);
        for(int i=0; i<8; i++){
            res[i] = (s>>i)&1;
        }
        return res;
    }
};
