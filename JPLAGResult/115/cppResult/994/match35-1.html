class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int x = 0;
        map<int, int> times;
        for(auto i: cells) x = x*2+i;
        for(int i = 0; i<N; i++){
            x = ((~(x ^ (x<<2)) & 0xFF)>>2)<<1;
            if(times.find(x)!=times.end()){

                int cycle = i  - times[x];
                //printf("%d %d %d ", i, N, cycle);
                i += (N-i-1)/cycle * cycle;
                //printf("%d %d %d ", i, N, cycle);
            }
            times[x] = i;
        }
        vector<int> z(8, 0);
        for(int i = 0; i<8;i++){
            z[7-i] = (x & (1<<(i)))>>(i);
        }
        return z;
    }
};
