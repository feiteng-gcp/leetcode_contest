class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        vector<int> init = cells;
        vector<int> firstDay(8, 0);
        vector<int> tmp(8, 0);
        
        for (int j = 1; j <= 6; j++) {
            firstDay[j] = cells[j - 1] == cells[j + 1] ? 1 : 0;
        }
        cells = firstDay;

        int t = 0;
        for (int i = 2; i < (1 << 20); i++) {
            
            for (int j = 1; j <= 6; j++) {
                tmp[j] = cells[j - 1] == cells[j + 1] ? 1 : 0;
            }
            cells = tmp;
            if (cells == firstDay) {
                t = i - 1;
                break;
            }
        }
        cout << t << endl;
        tmp = vector<int>(8, 0);
        N = N % t;
        if (N == 0) N = t;
        for (int i = 1; i <= N; i++) {
            for (int j = 1; j <= 6; j++) {
                tmp[j] = (init[j - 1] == init[j + 1]) ? 1 : 0;
            }
            init = tmp;
            //cout << init[0] << endl;
        }
        
        
        return init;
    }
};
