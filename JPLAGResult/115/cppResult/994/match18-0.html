class Solution {
public:
    int getState(vector<int> &cells) {
        int res = 0;
        for (int i = 0; i < 8; i++) {
            res |= cells[i] << i;
        }
        return res;
    }
    
    vector<int> getCells(int state) {
        vector<int> res;
        for (int i = 0; i < 8; i++) {
            res.push_back(state & (1 << i) ? 1 : 0);
        }
        return res;
    }
    
    int nextState(int curr) {
        int next = 0;
        for (int i = 1; i < 7; i++) {
            next |= ((((curr >> (i-1)) & (curr >> (i+1)) & 1) | 
                     ((~curr >> (i-1)) & (~curr >> (i+1)) & 1))) << i;
        }
        return next;
    }
    
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int state = getState(cells);
        unordered_map<int, int> state_day, day_state;
        for (int i = 0; i < N; i++) {
            state = nextState(state);
            if (state_day.count(state) != 0) {
                int cycle = i - state_day[state];
                return getCells(day_state[(N-1) % cycle]);
            } else {
                state_day[state] = i;
                day_state[i] = state;
            }
        }
        return getCells(state);
    }
};
