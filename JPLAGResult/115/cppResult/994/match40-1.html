class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        vector<int> re(8,0);
        re.at(0)=0;
        re.at(7)=0;
        vector<int> check(cells);
        check.at(0)=0;
        check.at(7)=0;
        for(int i=1; i<7; ++i)
        {
            if(cells.at(i-1)==0&cells.at(i+1)==0)check.at(i)=1;
            else if(cells.at(i-1)==1&cells.at(i+1)==1)check.at(i)=1;
            else check.at(i)=0;
        }
        //N=N%256;
        int T(0);
        if(N>64){
        while(N-->0)
        {
            ++T;
            for(int i=1; i<7; ++i)
            {
                if(cells.at(i-1)==0&cells.at(i+1)==0)re.at(i)=1;
                else if(cells.at(i-1)==1&cells.at(i+1)==1)re.at(i)=1;
                else re.at(i)=0;
            }
            cells=re;
            bool haha(true);
            for(int i=0; i<8; ++i)
            {
                if(check.at(i)!=re.at(i))
                {
                    haha=false;
                    break;
                }
            }
            if(haha==true&T>1)break;
        }
        N=N%(T-1);
        }
        while(N-->0)
        {
            for(int i=1; i<7; ++i)
            {
                if(cells.at(i-1)==0&cells.at(i+1)==0)re.at(i)=1;
                else if(cells.at(i-1)==1&cells.at(i+1)==1)re.at(i)=1;
                else re.at(i)=0;
            }
            cells=re;
        }
        return re;
    }
};
