class Solution {
public:
	vector<int> prisonAfterNDays(vector<int>& cells, int N) {
		int i = 0;
		unordered_map<int, int> map, rMap;
		int curr = getInt(cells);
		while (i < N)
		{
			if (map.find(curr) == map.end())
			{
				rMap[i] = curr;
				map[curr] = i++;
				curr = getNext(curr);
			}
			else
				break;
		}

		if (i == N)return getArr(curr);
		int cycle = i - map[curr], remain = N - i, idx = map[curr] + remain % cycle;
		int state = rMap[idx];
		return getArr(state);
	}

private:
	int getNext(int state)
	{
		int next = 0;
		for (int i = 1; i < 7; ++i)
		{
			int left = (state >> (i - 1)) & 1, right = (state >> (i + 1)) & 1;
			if (!left && !right || left && right)
				next |= (1 << i);
		}
		return next;
	}

	int getInt(const vector<int>& cells)
	{
		int res = 0;
		for (int i = 0; i < 8; ++i)
			if (cells[i])
				res |= (1 << i);
		return res;
	}

	vector<int> getArr(int state)
	{
		vector<int> res(8, 0);
		for (int i = 0; i < 8; ++i)
		{
			if ((state >> i) & 1)
				res[i] = 1;
		}
		return res;
	}
};
