class Solution {
public:

    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        vector<int> res = cells;
        //print_seq(res);
        for (int i = 1; i <= (N-1)%14+1; ++i) {
            res = generation(res);
            //print_seq(res);
        }
        return res;
    }
private:
    vector<int> generation(vector<int>& cells) {
        vector<int> res(8, 0);
        for (int i = 1; i < 7; ++i) {
            if (cells[i-1] == cells[i+1]) {
                res[i] = 1;
            }
        }
        return res;
    }
};
