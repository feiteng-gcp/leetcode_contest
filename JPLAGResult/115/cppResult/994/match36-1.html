class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        //vector<int> orig(cells.size());
        //for(int i=0;i<8;i++) orig[i] = cells[i];
        N = N%14 + 14;
        vector<int> rep(cells.size());

        rep[0] = rep[7] = 0; 
        for(int i=1;i<=N;i++) {
            for(int x=1;x<=6;x++) {
                rep[x] = !(cells[x-1] ^ cells[x+1]);
            }
            cells = rep;
            /*bool eq = true;
            cout << "i: " << i << " "; 
            for(int a=0;a<8;a++) {
                cout << cells[a] << ", ";
                
            }
            cout << endl;*/
        }
        return cells;
        
    }
};
