class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& c, int n) {
        int k = 1;
        vector<int> cur = next(c);
        vector<int> res = next(cur);
        
        while (res != cur) {
            res = next(res);
            ++k;
        }
        
        k = (n-1)%k;
        res = cur;
        for (int i = 0; i < k; ++i) {
            res = next(res);
        }
        return res;
    }
    
    vector<int> next(vector<int>& cur) {
        vector<int> res(cur.size(), 0);
        res[0] = 0; res[cur.size()-1] = 0;
        for (int i = 1; i < cur.size()-1; ++i) {
            if (cur[i-1] == cur[i+1]) res[i] = 1;
        }
        return res;
    }
};
