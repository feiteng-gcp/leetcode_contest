/**
Day 0: [0, 1, 0, 1, 1, 0, 0, 1]
Day 1: [0, 1, 1, 0, 0, 0, 0, 0]
Day 2: [0, 0, 0, 0, 1, 1, 1, 0]
Day 3: [0, 1, 1, 0, 0, 1, 0, 0]
Day 4: [0, 0, 0, 0, 0, 1, 0, 0]
Day 5: [0, 1, 1, 1, 0, 1, 0, 0]
Day 6: [0, 0, 1, 0, 1, 1, 0, 0]
Day 7: [0, 0, 1, 1, 0, 0, 0, 0]

*/

class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        auto period = getPeriod(cells);
        int T = period.second;
        int offset = period.first;
        N = N > offset ? (N - offset) % T + offset : N;
        while (0 < N--) {
            cells = emulate(cells);
        }
        return cells;
        // return {offset, T};
    }
    
    pair<int, int> getPeriod(const vector<int>& cell) {
        int T = 0;
        int offset = 0;
        vector<int> currentCell = cell;
        unordered_map<int, int> track;
        do {
            int x = encode(currentCell);
            if (track.find(x) != track.end()) {
                T = offset - track[x];
            } else {
                track[x] = offset++;
                currentCell = emulate(currentCell);
            }
        } while (T == 0);
        return {offset, T};
    }
    

    vector<int> emulate(const vector<int>& cell) {
        vector<int> newCell(8, 0);
        for (int i = 1; i < 7; i++) {
            newCell[i] = (cell[i - 1] == cell[i + 1]);
        }
        return newCell;
    }
    

    int encode(const vector<int>& cells) {
        int x = 0;
        for (int i = 0; i < cells.size(); i++) {
            x |= cells[i] << i;
        }
        return x;
    }
};
