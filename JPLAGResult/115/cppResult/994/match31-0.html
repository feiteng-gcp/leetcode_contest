class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int s = 0;
        for (int i = 0; i < 8; ++i) {
            s |= cells[i] << i;
        }
        vector<int> visit(256, -1);
        //cout << "s=" << s << endl;
        int A, B;
        vector<int> a;
        visit[s] = a.size();
        a.push_back(s);
        while (true) {
            int t = 0;
            vector<int> cur(8);
            for (int i = 0; i < 8; ++i) cur[i] = (s & (1 << i)) ? 1 : 0;
            for (int i = 0; i < 8; ++i) {
                if (i - 1 >= 0 && i + 1 < 8 && cur[i - 1] == cur[i + 1]) t |= 1 << i;
            }
            //cout << s << " " << t << endl;
            if (visit[t] >= 0) {
                A = visit[t];
                B = a.size() - A;
                break;
            }
            visit[t] = a.size();
            a.push_back(t);
            s = t;
        }
        vector<int> ret(8);
        if (N < A) {
            s = a[N];
            for (int i = 0; i < 8; ++i) ret[i] = (s & (1 << i)) ? 1 : 0;
        }
        N -= A;
        N %= B;
        s = a[A + N];
        for (int i = 0; i < 8; ++i) ret[i] = (s & (1 << i)) ? 1 : 0;
        return ret;
    }
};
