class Solution {
public:
	vector<int> prisonAfterNDays(vector<int>& cells, int N) {
		map<vector<int>, int>t;
		t[cells] = 0;
		vector<int>curr = cells;
		bool found = false;
		for (int i = 1; i <= N; i++) {
			vector<int>temp(curr.size());
			for (int j = 1; j < curr.size() - 1; j++) {
				if (curr[j - 1] == curr[j + 1])temp[j] = 1;
				else temp[j] = 0;
			}
			curr = temp;
			if (!found && t.find(curr) != t.end()) {
				int cycle = i - t[curr];
				i = 0;
				N = (N - t[curr]) % cycle;
				found = true;
			}
			t[curr] = i;
		}
		return curr;
	}
};
