class Solution {
public:

  vector<int> next(vector<int> cells) {
    vector<int> res(cells.size(), 0);
    res[0] = res[cells.size() - 1] = 0;
    for (int i = 1; i < cells.size() - 1; ++i) {
      if (cells[i - 1] == cells[i + 1]) {
        res[i] = 1;
      } else {
        res[i] = 0;
      }
    }
    return res;
  }
  
  
  vector<int> prisonAfterNDays(vector<int>& cells, int N) {
    int l = 0;
    vector<int> res;
    if (N == 0) return cells;
    cells = next(cells);
    res = cells;
    while (1) {
      res = next(res);
      // for (int i = 0; i < res.size(); ++i) {
      //   cout << res[i] << " ";
      // }
      // cout << endl;
      ++l;
      if (res == cells) break;
    }
    --N;
    N %= l;
    
    for (int i = 0; i < N; ++i) {
      res = next(res);
    }
    return res;
  }
};
