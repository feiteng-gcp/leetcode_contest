class Solution {
private:
	int tra(const vector<int> &cells) {
		int ans = 0;
		int pow = 1;
		for (int i = 0; i < cells.size(); i++) {
			ans += cells[i] * pow;
			pow *= 2;
		}
		return ans;
	}
public:
	vector<int> prisonAfterNDays(vector<int>& cells, int N) {
		vector<int> temp(cells.size());
		vector<int> record(129, -1);
		int a, i;
		int loop;
		int loopstart;
		temp[0] = 0;
		temp.back() = 0;
		for (i = 0; i < N; i++) {
			for (int j = 1; j < cells.size() - 1; j++) {
				if (cells[j - 1] == cells[j + 1])
					temp[j] = 1;
				else
					temp[j] = 0;
			}
			cells = temp;
			a = tra(temp);
			if (record[a] == -1)
				record[a] = i;
			else {
				loop = i - record[a];
				i += (N - i-1) / loop * loop;
			}
		}
		return cells;
	}
};
