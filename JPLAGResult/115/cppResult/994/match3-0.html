class Solution {
public:
    int encode(vector<int>& cells) {
        int res = 0;
        for(int i=0; i<cells.size(); i++)
            res = res*2 + cells[i];
        return res;
    }
    
    void print(vector<int>& cells, int days) {
        cout << days << ": ";
        for(int i=0; i<cells.size(); i++)
            cout << cells[i] << " ";
        cout << endl;
    }
    
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        unordered_map<int, int> m;
        int val = encode(cells);
        for(int i=1; i<= N; i++) {
            vector<int> tmp = cells;
            tmp[0] = 0;
            tmp[cells.size()-1] = 0;
            for(int j=1; j<cells.size()-1; j++) {
                if(cells[j-1] == cells[j+1])
                    tmp[j] = 1;
                else
                    tmp[j] = 0;
            }
            val = encode(tmp);
            if(m.find(val) != m.end()) {
                print(tmp, i);
                //cout << val << " " << m[val] << endl;
                int days = i - m[val];
                N = N % days;
                if(N == 0)
                    N = days;
                i = 1;
                unordered_map<int, int> tmpm;
                m = tmpm;
            }    
            else {
                m[val] = i;
            }    
            cells = tmp;
            //print(cells, i);
        }

        vector<int> res = cells;
        return res;
    }
};
