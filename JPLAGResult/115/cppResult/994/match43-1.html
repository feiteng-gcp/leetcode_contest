class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        vector<vector<int>> order;
        map<vector<int>,int> seen;
        vector<int> current=cells;
        order.push_back(current);
        for(int i=0;i<N;i++)
        {
            current=next(current);
            if(seen.find(current)!=seen.end())
            {
                N-=order.size();
                N=N%(order.size()-1-seen[current]+1);
                return order[seen[current]+N];
            }
            order.push_back(current);
            seen[current]=order.size()-1;
        }

        return order.back();
    }
    
    vector<int> next(vector<int>& current)
    {
        vector<int> ans(current.size(),0);
        for(int i=1;i<current.size()-1;i++)
        {
            if(current[i-1]==0&&current[i+1]==0||current[i-1]==1&&current[i+1]==1)ans[i]=1;
        }
        return ans;
    }
};
