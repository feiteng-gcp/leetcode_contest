class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) 
    {
        map<int, vector<int>> m;
        
        for(int i = 0; i <= 13; i++)
        {
            int a[8];
            
            a[0] = a[7] = 0;
            
            for(int j = 1; j < 7; j++)
            {
                if((cells[j - 1] and cells[j + 1]) or (!cells[j - 1] and !cells[j + 1]))
                    a[j] = 1;
                else
                    a[j] = 0;
            }
            
            for(int j = 0; j < 8; j++)
                cells[j] = a[j];
            
            m[i] = cells;
        }
        
        return m[(N - 1) % 14];
    }
};
