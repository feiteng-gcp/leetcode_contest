class Solution {
public:
    int F(int x){
        return (~((x<<1) ^ (x>>1))) & (((1<<6) - 1)<<1);
    }
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int u=0;
        for(int i=0;i<8;++i){
            if(cells[i])u|=1<<i;
        }
        int seen[1<<8];
        for(int i=0;i<(1<<8);++i)seen[i]=-1;
        for(int i=1;i<=N;){
            /*for(int j=0;j<8;++j){
                printf("%d",!!(u&(1<<j)));
            }
            printf("\n");*/
            if(seen[u]>=0){
                int lam=(i-1)-seen[u];
                int num_skip=(N-i+1)/lam;
                if(num_skip>0){
                    i+=num_skip*lam;
                }else{
                    seen[u]=i-1;
                    u=F(u);
                    ++i;
                }
            }else{
                seen[u]=i-1;
                u=F(u);
                ++i;
            }
        }
        vector<int> R(8);
        for(int i=0;i<8;++i){
            R[i]=!!((1<<i)&u);
        }
        return R;
    }
};
