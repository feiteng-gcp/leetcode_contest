#include <vector>

class Solution {
public:

    vector<int> prisonAfterNDays(vector<int>& state, int N) {
        vector<vector<int>> statesArr;
        map<vector<int>, int> states;
        states[state] = 0;
        statesArr.push_back(state);
        for (int i = 1; i <= N; ++i) {
            vector<int> newState = state;
            newState[0] = newState[state.size() - 1] = 0;
            for (int j = 1; j + 1 < state.size(); ++j) {
                newState[j] = 1 ^ state[j - 1] ^ state[j + 1];
            }
            state = newState;
        
            if (states.count(state)) {
                int cycleLength = i - states[state];
                return statesArr[states[state] + (N - i) % cycleLength];
            }
            
            states[state] = i;
            statesArr.push_back(state);
        }
        return state;
    }
};
