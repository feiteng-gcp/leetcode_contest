class Solution {
public:
    set<int> bst;
    int val(vector<int>& arr) {
        int ans = 0;
        for (int i = 0; i < arr.size(); i++) {
            ans = ans * 10 + arr[i];
        }
        return ans;
    }
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        vector<int> pre = cells;
        for (int i = 1; i <= 6; i++) {
            if (pre[i - 1] == pre[i + 1])
                cells[i] = 1;
            else 
                cells[i] = 0;
        }
        cells[0] = cells.back() = 0;
        N --;
        vector<int> ori = cells;
        int tmp = val(cells);
        bst.insert(tmp);
        int num = 1;
        while (1) {
            vector<int> pre = cells;
            for (int i = 1; i <= 6; i++) {
                if (pre[i - 1] == pre[i + 1])
                    cells[i] = 1;
                else 
                    cells[i] = 0;
            }
            tmp = val(cells);
            if (bst.find(tmp) != bst.end()) break;
            num++;
            bst.insert(tmp);
        }
        
        N %= num;
        printf("%d..", N);
        cells = ori;
        for (int t = 0; t < N; t++) {
            vector<int> pre = cells;
            for (int i = 1; i <= 6; i++) {
                if (pre[i - 1] == pre[i + 1])
                    cells[i] = 1;
                else 
                    cells[i] = 0;
            }
            cells[0] = cells.back() = 0;
        }
        return cells;
    }
};
