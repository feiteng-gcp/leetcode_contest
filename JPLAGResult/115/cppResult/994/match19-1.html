class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        // N = N % 16 + 16 ;
        if(N > 16){
            N = (N - 2) % 14 + 2; 
        }

        for(int i = 0; i < N; i ++){
            vector<int> next(8, 0);
            for(int i = 1; i < 7; i ++){
                if(cells[i-1] ^ cells[i+1] == 0){
                    next[i] = 1;
                }else{
                    next[i] = 0;
                }
            }
            for(int i = 0; i < 8; i ++){
                cells[i] = next[i];
            }
        }
        return cells;
    }
};
