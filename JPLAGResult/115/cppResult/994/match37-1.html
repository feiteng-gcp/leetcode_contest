class Solution {
public:
    unordered_map<int,int> memo;
    unordered_map<int,int> reverse_memo;
    vector<int> primes={2,3,7,11,13,17,19,23};
    int get_hash(vector<int>& cells)
    {
        int hash=1;
        for(int i=0;i<cells.size();i++)
            hash=cells[i]?hash*primes[i]:hash;
        return hash;
    }
    vector<int> prisonAfterNDays(vector<int>& cells, int N) 
    {
        vector<int> cells_next(8);
        int hash=get_hash(cells);
        memo[hash]=0;
        reverse_memo[0]=hash;

        cells_next[0]=0,cells_next[7]=0;
        for(int i=1;i<=N;i++)
        {
            for(int j=1;j<cells.size()-1;j++)
            {
                if(!cells[j-1]&&!cells[j+1]||cells[j-1]&&cells[j+1])
                    cells_next[j]=1;
                else
                    cells_next[j]=0;
            }
            cells=cells_next;
            hash=get_hash(cells);
            if(memo.count(hash))
            {
                hash=reverse_memo[(N-i)%(i-memo[hash])+memo[hash]];
                for(int i=0;i<8;i++)
                    if(hash%primes[i]==0)
                        cells[i]=1;
                    else
                        cells[i]=0;
                return cells;
            }
            else
                memo[hash]=i,reverse_memo[i]=hash;
        }
        return cells;
    }
};
