class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& a, int N) {
        vector<vector<int>> dp(70,vector<int>(8));
        map<vector<int>,int> maps;
        dp[0]=a;
        int i,j,p,k,n;
        for (i=1;i<70;i++)
        {
            dp[i][0]=0; dp[i][7]=0;
            for (j=1;j<7;j++)
                if (dp[i-1][j-1]==dp[i-1][j+1]) dp[i][j]=1; else dp[i][j]=0;
            if (maps[dp[i]]!=0) break;
            maps[dp[i]]=i;
        }
        //return dp[1];
        p=maps[dp[i]];
        k=i-p;
        if (N<=i) return dp[N];
        n=(N-p)%k;
        return dp[p+n];
    }
};
