class Solution {
    int trans(int state) {
        int newState = 0;

        for (int i=1; i<7; i++) {
            newState |=  ( ((state>>(i-1))&1)^((state>>(i+1))&1)^1) << i;
        }
        return newState;
    }
    
    
    vector<int> to_vec(int state) {
        vector<int> cells;
        for (int i=0; i<8; i++) 
            cells.push_back((state>>i)&1);
        return cells;
    }
    
    void print(int state) {
        auto z = to_vec(state);
        for (int i=0; i<8; i++) {
            cout << z[i];
        }
        cout << endl;
    }
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int state = 0;
        for (int i=0;i<8; i++) state |= (cells[i] << i);
        vector<int> x(256,-1);
        int i = 0;
        for (;; i++) {
            //cout << i << " "; print(state);
            if (i==N) return to_vec(state);

            if (x[state] >= 0) {
                int loop_size = i - x[state];
                int head = x[state];
                int rem = (N - head) % loop_size;
                for (int j=0; j<rem; j++) {
                    state = trans(state);
                }                
                return to_vec(state);
            }
            x[state] = i;
            state = trans(state);
        } 
    }
};
