class Solution {
public:
    int next(int a) {
        int b = 0;
        for (int i = 1; i<= 6; i++ ) {
            int x = (a >> (i-1)) & 1;
            int y = (a >> (i+1)) & 1;
            if (x == y) {
                b |= (1<<i);
            }
        }
        return b;
    }
    

    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int v = 0;
        for (int i = 0 ;i< 8; i++) {
            v = v * 2 + cells[i];
        }
        
        int save_v = v;
        
        vector<int> visit(256, -1);
        int k = 0;
        while (visit[v] == -1) {
           visit[v] = k;
           v= next(v);
           k++;
        }
        
        if (N < visit[v]) {
          for (int i=0;i<N;i++) {
            save_v = next(save_v);
          }
          vector<int> r(8);
          for (int i = 0; i< 8;i++) {
            r[7-i] = (save_v >> i) & 1;
          }
          return r;    
        }
        
        
        N = N-visit[v];
        N= N % (k - visit[v]);
        
        for (int i=0;i<N;i++) {
            v = next(v);
        }
        
        vector<int> r(8);
        for (int i = 0; i< 8;i++) {
            r[7-i] = (v >> i) & 1;
        }
        return r;
    }
};
