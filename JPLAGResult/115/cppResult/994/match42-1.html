class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int cur = cells[0];
        bool same = true;
        for (int i = 1; i < 8; ++i) { // all the same
            if (cur != cells[i]) same = false;
        }
        if (same) {
            if (N == 1) {
                next(cells, 1);
            } else {
                N = (N - 1) % 7;
                next(cells, N + 1);
            }
        } else {
            if (N == 1) {
                next(cells, 1);
            } else {
                N = (N - 1) % 14;
                next(cells, N + 1);
            }
        }
        return cells;
    }

    void next(vector<int>& v, int n) {
    

    for (int i = 0; i < n; ++i) {
        vector<int> pre(v);
        for (int j = 1; j < 7; ++j) {
            if (pre[j + 1] == pre[j - 1]) v[j] = 1;
            else v[j] = 0;
        }
        v[0] = v[7] = 0;
    }
        return;
}
};
