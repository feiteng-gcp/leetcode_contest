class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        vector<int> states;
        unordered_map<int, int> m;
        int cur = 0;
        for (int i = 0; i < 8; i++)
        {
            cur |= (cells[i] << i);
        }
        states.push_back(cur);
        m[cur] = 0;
        for (int i = 1; i <= N; i++)
        {
            int next = 0;
            for (int j = 1; j < 7; j++)
            {
                next |= (((cur >> j-1) & 1) == ((cur >> j+1) & 1)) << j;
            }
            if (m.find(next) == m.end())
            {
                states.push_back(next);
                m[next] = states.size()-1;
            }
            else
            {
                int start = m[next];
                int end = m.size()-1;
                cur = states[ (N - start) % (end -start + 1) + start];
                break;
            }
            cur = next;
        }
        vector<int> res;
        for (int i = 0; i < 8; i++)
        {
            res.push_back(((cur & ( 1 << i)) != 0) ? 1 : 0);
        }
        return res;
    }
};
