class Solution {
public:
    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        vector<char> a, b;
        for (auto i : cells) {
            a.push_back('0'+i);
            b.push_back('0'+i);
        }
        unordered_map<string, int> m;
        for (int i = 0; i < N; i++) {
            string s(a.begin(), a.end());
            if (m.find(s) != m.end()) {
                int t = (N-1-i) % (i-m[s]);
                for (int k = 0; k <= t; k++) {
                    b[0] = b[7] = '0';
                    for (int j = 1; j < 7; j++) {
                        if (a[j-1] == a[j+1]) b[j] = '1';
                        else b[j] = '0';
                    }
                    a = b;
                }
                break;
            }
            m[s] = i;
            b[0] = b[7] = '0';
            for (int j = 1; j < 7; j++) {
                if (a[j-1] == a[j+1]) b[j] = '1';
                else b[j] = '0';
            }
            a = b;
        }
        vector<int> ans;
        for (auto i : a) {
            ans.push_back(i-'0');
        }
        return ans;
    }
};
