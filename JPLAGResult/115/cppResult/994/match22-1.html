class Solution {
public:
    void print(int mask){
        for(int i = 0; i < 8; i++){
            if(mask&(1<<i))
                cout << "1,";
            else
                cout << "0,";
        }
        cout << endl;
    }

    vector<int> prisonAfterNDays(vector<int>& cells, int N) {
        int cellMask = 0;
        for(int i = 0; i < 8; i++){
            cellMask |= (1<<i)*cells[i];
        }
        set <int> seen;

        int res[300], day = 0;
        // res[0] = cellMask;
        int curMask = cellMask;
        while(1){
            int newMask = 0;
            for(int i = 1; i < 7; i++){
                if(((curMask&(1<<(i-1))) == 0) == ((curMask&(1<<(i+1)))==0))
                    newMask |= 1<<i;
            }
            curMask = newMask;
            if(seen.find(curMask) != seen.end())
                break;
            res[++day] = curMask;
            seen.insert(curMask);
            // print(curMask);
        }
        N %= day;
        if(N == 0)
            N = day;
        vector <int> ret;
        for(int i = 0; i < 8; i++) {
            if(res[N]&(1<<i))
                ret.push_back(1);
            else
                ret.push_back(0);
        }
        return ret;
    }
};
