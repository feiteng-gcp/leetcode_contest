/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    vector < int > arr;
    void dfs(TreeNode* u, int val) {
        if (u == NULL) return ;
        arr.push_back(val);
        dfs(u->left, 2*val);
        dfs(u->right, 2*val+1);
    }
    bool isCompleteTree(TreeNode* root) {
        dfs(root, 1);
        sort(arr.begin(), arr.end());
        for (int i = 1; i < arr.size(); i++) if (arr[i] != arr[i-1] + 1) return false;
        return true;
    }
};
