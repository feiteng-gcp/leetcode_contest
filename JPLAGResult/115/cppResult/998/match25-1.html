/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        vector<string> seq;
        queue<TreeNode*> tq;
        tq.push(root);
        while(!tq.empty()) {
            if(tq.front()) {
                seq.push_back("valid");
                tq.push(tq.front()->left);
                tq.push(tq.front()->right);
            }
            else {
                seq.push_back("null");
            }
            tq.pop();
        }
        

        int flag = 0;
        for(int i = 0; i < seq.size(); ++i) {
            if(seq[i] == "null") {
                flag = 1;
            }
            
            if(flag == 1 && seq[i] == "valid")
                return false;
        }
        return true;
    }
};
