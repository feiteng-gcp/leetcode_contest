/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        if(root == nullptr)
            return true;
        queue<TreeNode*> q;
        q.push(root);
        bool flag = false;
        while(q.empty() == false) {
            TreeNode* n = q.front();
            q.pop();
            //cout << flag << " " << n->val << endl;
            if(flag == false) {
                if(n->right != nullptr && n->left == nullptr)
                    return false;
                //cout << "hello" << endl;
                if(n->left != nullptr)
                    q.push(n->left);
                else
                    flag = true;
                if(n->right != nullptr)
                    q.push(n->right);
                else
                    flag = true;
                
            }
            else {
                if(n->left != nullptr || n->right != nullptr)
                    return false;
                if(n->left != nullptr)
                    q.push(n->left);
                if(n->right != nullptr)
                    q.push(n->right);
            }

        }
        return true;      
    }
};
