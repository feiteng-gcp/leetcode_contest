/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        queue<TreeNode*> q;
        q.push(root);

        bool haveNull = false;
        while (q.size()) {
            TreeNode* p = q.front();
            q.pop();
            
            if (p->left != NULL) {
                if (haveNull) return false;
                q.push(p->left);
            } else {
                haveNull = true;
            }
            
            if (p->right != NULL) {
                if (haveNull) return false;
                q.push(p->right);
            } else {
                haveNull = true;
            }
            
        }
        
        return true;
    }
};
