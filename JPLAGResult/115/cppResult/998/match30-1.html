/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int get_depth(TreeNode *root) {
        if (root == NULL)
            return 0;
        return 1 + max(get_depth(root->left), get_depth(root->right));
    }  
    
    int recurse_sizes(TreeNode *root) {
        if (root == NULL)
           return 0;
        int l = recurse_sizes(root->left);
        int r = recurse_sizes(root->right);
        root->val = 1 + l + r;
        return root->val;
    }    
    
    bool check(TreeNode *root) {
        if (root == NULL)
            return true;
        if (root->left == NULL)
            return root->right == NULL;
        int l = root->left->val;
        int r = (root->right == NULL ? 0 : root->right->val);
        if (r > l || 2 * r + 1 < l)
            return false;
        
        if (l & (l + 1)) {
            if (r & (r + 1))
                return false;
            return check(root->left);
        } else {
            return check(root->right);
        }
    }

    bool isCompleteTree(TreeNode* root) {
        recurse_sizes(root);
        return check(root);
    }
};
