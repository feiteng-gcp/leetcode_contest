/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        queue<TreeNode *> q;
        q.push(root);
        bool vac = false;
        while (!q.empty()) {
          auto t = q.front();
          q.pop();
          if (t == NULL) {
            vac = true;
            continue;
          }
          if (vac && t != NULL) return false;
          if (t != NULL) {
            q.push(t->left);
            q.push(t->right);
          }
        }
        return true;
    }
};
