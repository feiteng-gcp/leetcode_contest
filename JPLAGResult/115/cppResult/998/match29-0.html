/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        queue<TreeNode *> q;
        if(!root) return true;
        if(root) q.push(root);
        
        bool flag = false;
        while(q.size()) {
            int l = q.size();
            while(l--) {
                auto front = q.front();
                if(flag && front) return false;
                if(front) {
                    q.push(front->left);
                    q.push(front->right);
                }
                else {
                    flag = true;
                }
                q.pop();
            }
        }
        return true;
    }
};
