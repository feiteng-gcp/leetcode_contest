/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:

    bool isCompleteTree(TreeNode* root) {
        vector<TreeNode*> dq = {root};
        bool last = false;
        while (!dq.empty()) {
            vector<TreeNode*> new_dq;
            for (auto node: dq) {
                if (node->left == nullptr) {
                    last = true;
                } else {
                    if (last == true)
                        return false;
                    new_dq.push_back(node->left);
                }
                if (node->right == nullptr) {
                    last = true;
                } else {
                    if (last == true)
                        return false;
                    new_dq.push_back(node->right);
                }
            }
            dq.swap(new_dq);
        }
        return true;
        
    }
};
