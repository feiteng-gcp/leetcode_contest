/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        queue<TreeNode*> q;
        if(!root)
            return false;
        q.push(root);
        bool foundLast = false;
        while(!q.empty()){
            int n = q.size();
            while(n > 0){
                auto cn = q.front();
                q.pop();
                if(cn->left){
                    if(foundLast)
                        return false;
                    q.push(cn->left);
                } else {
                    foundLast = true;
                }
                if(cn->right){
                    if(foundLast)
                        return false;
                    q.push(cn->right);
                } else {
                    foundLast = true;
                }
                n--;
            }
                
        }
        return true;

        
    }
};
