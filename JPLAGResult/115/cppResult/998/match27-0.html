/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) 
    {
        if(!root)
            return true;
        queue<TreeNode *> q;
        TreeNode * node;
        q.push(root);
        bool flag(true);
        while(!q.empty())
        {
            node = q.front();
            q.pop();
            auto l = node->left, r = node->right;
            if(flag)
            {
                if(l && r)
                {
                    q.push(l);
                    q.push(r);
                    continue;
                }
                else if(l && ! r)
                {
                    q.push(l);
                    flag = false;
                }
                else if(!l)
                {
                    if(r)
                        return false;
                    else
                        flag = false;
                }
            }
            else
            {
                if(l || r)
                    return false;
            }
                
        }
        return true;
    }
};
