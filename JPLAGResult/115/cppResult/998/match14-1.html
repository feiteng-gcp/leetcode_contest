/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    int numbers;
public:

    bool check(TreeNode* r, int id){
        if (!r)
            return true;
        if (id > numbers - 1)
            return false;
        return check(r->left, 2 * id + 1) && check(r->right, 2 * id + 2);
    }
    int get_count(TreeNode* r)
    {
        if (!r)
            return 0;
        return get_count(r->left) + get_count(r->right) + 1;
    }
    bool isCompleteTree(TreeNode* r) {
        numbers = get_count(r);
        printf("%d", numbers);
        return check(r, 0);
    }
};
