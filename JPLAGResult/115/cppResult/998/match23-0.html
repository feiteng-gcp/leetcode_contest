/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    int getHeight(TreeNode *root)
    {
        
        if (root)
        {
            return 1 + max(getHeight(root->left), getHeight(root->right));
        }
        return 0;
    }
    
    bool isCompleteTree(TreeNode* root) {
        if (!root) return true;
        int n = getHeight(root);
        if (n>10) return false;
        vector<TreeNode*> a;
        queue<TreeNode*> bfs;
        bfs.push(root);
        while (!bfs.empty())
        {
            auto head = bfs.front();
            a.push_back(head);
            bfs.pop();
            if (head)
            {
                bfs.push(head->left);
                bfs.push(head->right);
            }
        }

        int i = a.size()-1;
        while (i>0 && a[i] == nullptr)
        {
            i--;
        }
        while (i>0)
        {
            if (a[i] == nullptr)
                return false;
            i--;
        }
        return true;
    }
    
};
