/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        return isCompleteTree(root, treeSize(root));
    }

    int treeSize(TreeNode * root) {
        if (!root)
            return 0;
        return treeSize(root->left) + treeSize(root->right) + 1;
    }
    bool isCompleteTree(TreeNode * root, int size) {
        if (!root && size == 0)
            return true;
        if (!root || size <= 0)
            return false;
        
        int full = 1;
        while (full * 2 + 1 <= size) {
            full = full * 2 + 1;
        }
        int lcap = (full + 1) / 2;
        int nleft = std::min(size - full, lcap) + (full - 1) / 2;
        int nright = size - 1 - nleft;
        
        return isCompleteTree(root->left, nleft) && isCompleteTree(root->right, nright);
    }
};
