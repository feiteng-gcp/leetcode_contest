/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        queue<TreeNode *> a;
        bool nullOccur = false;
        
        a.push(root);
        while (!a.empty()) {
            auto node = a.front();
            if (node == NULL) {
                nullOccur = true;
            } else {
                if (nullOccur) return false;
            }
            
            if (node != NULL) {
                a.push(node->left);
                a.push(node->right);
            }
            a.pop();
        }
        
        return true;
    }
    
};
