/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        if (root == nullptr) return true;
        
        queue<TreeNode*> q;
        q.push(root);
        TreeNode* head = NULL;
        bool flag = false;
        while (!q.empty()) {
            TreeNode *cur = q.front();
            q.pop();
            if (cur != nullptr) {
                if (flag) return false;
                q.push(cur->left);
                q.push(cur->right);
            } else {
                flag = true;
            }
        }
        return true;
    }
};
