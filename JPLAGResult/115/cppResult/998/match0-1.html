/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        queue<TreeNode*> bfs;
        bfs.push(root);
        
        while(bfs.front() != nullptr) {
            TreeNode* node = bfs.front();
            bfs.pop();
            bfs.push(node->left);
            bfs.push(node->right);
        }
        
        while(!bfs.empty() && bfs.front() == nullptr) {
            bfs.pop();
        }
        
        return bfs.empty();
    }
};
