/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        if (!root) return true;
        queue<TreeNode*> bfs;
        bfs.push(root);
        vector<TreeNode*> nodes;
        while (!bfs.empty()) {
            auto cur = bfs.front();
            bfs.pop();
            nodes.push_back(cur);
            if (cur) {
                bfs.push(cur->left);
                bfs.push(cur->right);
            }
        }
        bool null_flag = false;
        for (auto itr = nodes.begin(); itr != nodes.end(); ++itr) {
            if (*itr == nullptr) {
                if (!null_flag) {

                    null_flag = true;
                }
            }
            else {
                if (null_flag) return false;
            }
        }
        return true;
    }
};
