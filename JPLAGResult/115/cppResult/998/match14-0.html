/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        int count = _count(root);
        int index = 0;
        return helper(root, index, count);
    }
    bool helper(TreeNode* root, int index, int number_nodes){
        if(!root) return  true;
        if(index >= number_nodes) return false;
        return (helper(root->left, 2*index + 1, number_nodes) && 
            helper(root->right, 2*index + 2, number_nodes));
    }
    int _count(TreeNode *r){
        if(!r) return 0;
        return 1 + _count(r->left) + _count(r->right);
    }
    int height(TreeNode *r){
        if(!r) return 0;
        return 1 + max(height(r->left), height(r->right));
    }
    
};
