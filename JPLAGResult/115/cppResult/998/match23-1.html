/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
  vector<int> bfs(TreeNode* root) {
    vector<int> ans;
    queue<TreeNode*> q;
    q.push(root);
    while (!q.empty()) {
      TreeNode* cur = q.front();
      q.pop();
      if (cur == NULL) ans.push_back(0);
      else {
        ans.push_back(cur->val);
        q.push(cur->left);
        q.push(cur->right);
      }
    }
    return ans;
  }
  
  
  bool isCompleteTree(TreeNode* root) {
    vector<int> vs = bfs(root);

    int p = vs.size() - 1;
    while (p >= 0 && vs[p] == 0) {
      --p;
    }
    while (p >= 0) {
      if (vs[p] == 0) return false;
      --p;
    }
    return true;
  }
};
