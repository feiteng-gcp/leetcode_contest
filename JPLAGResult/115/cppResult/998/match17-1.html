/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        queue<TreeNode*> q;
        if(root)
        {
            q.push(root);

            TreeNode* cur = NULL;
            bool flag = false;
            while(!q.empty())
            {
                cur = q.front();
                q.pop();
                if(cur)
                {
                    if(flag)
                        return false;
                    q.push(cur->left);
                    q.push(cur->right);
                }
                else
                    flag = true;
                
            }
            return true;
        }
        return true;
    }
};
