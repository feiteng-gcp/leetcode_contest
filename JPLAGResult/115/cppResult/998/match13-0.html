/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        vector<TreeNode*> q;
        if (!root) return true;
        bool isStop = false;
        q.push_back(root);
        
        while (true) {
            if (q.empty()) return true;
            vector<TreeNode*> new_q;
            for (TreeNode *node: q) {
                if (node->left && node->right) {
                    if (isStop) return false;
                    new_q.push_back(node->left);
                    new_q.push_back(node->right);
                } else if (node->left) {
                    if (isStop) return false;
                    new_q.push_back(node->left);
                    isStop = true;
                } else if (node->right) {
                    return false;
                } else {
                    isStop = true;
                }
            }
            q = new_q;
            if (q.empty()) return true;
        }
        return true;
    }
};
