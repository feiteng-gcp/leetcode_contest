class Solution {
  int cc;
  int c(TreeNode *x) {
    if (!x) return 0;
    return 1 + c(x->left) + c(x->right);
  }
  bool f(TreeNode *x, int i) {
    if (!x) return true;
    if (i >= cc) return false;
    return f(x->left, 2*i+1) && f(x->right, 2*i+2);
  }
public:
  bool isCompleteTree(TreeNode *root) {
    cc = c(root);
    return f(root, 0);
  }
};
