/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        
        if(!root) return true;
        queue<TreeNode *> q;
        q.push(root);
        bool flag = true;
        while(!q.empty()) {
            int size = q.size();
            while(size--) {
                TreeNode *curr = q.front();
                q.pop();
                if(curr->left) {
                    if(!flag) return false;
                    q.push(curr->left);
                }
                else {
                    flag = false;
                }
                if(curr->right) {
                    if(!flag) return false;
                    q.push(curr->right);
                }
                else {
                    flag = false;
                }
                
            }
            
            
            
        }
        return true;
        
    }
};
