/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:

    int Count;
    int count(TreeNode* &root)
    {
        if(!root)
            return 0;
        return 1+count(root->left)+count(root->right);
    }
    bool isCompleteTree(TreeNode* &root,int i)
    {
        if(!root)
            return true;
        if(i>=Count)
            return false;
        return isCompleteTree(root->left,2*i+1)&&isCompleteTree(root->right,2*i+2);
    }
    bool isCompleteTree(TreeNode* &root) 
    {
        Count=count(root);
        return isCompleteTree(root,0);
    }
};
