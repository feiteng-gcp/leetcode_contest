/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        if (!root) return false;
        queue<TreeNode *> q;
        q.push(root);

        bool f = 0;
        while (!q.empty())
        {
            int size = q.size();
            while (size--)
            {

                auto u = q.front(); q.pop();
                if (f)
                {
                    if (u->left || u->right) return false;
                }
                else
                {
                    if (u->left && u->right)
                    {
                        q.push(u->left);
                        q.push(u->right);

                    }
                    else if (u->left && !u->right) 
                    {
                        q.push(u->left);
                        f = 1;
                    }
                    else if (!u->left && u->right) 
                    {
                        return false;
                    }
                    else 
                    {
                        f = 1;
                    }
                }
            }
        }
        return true;
    }
};
