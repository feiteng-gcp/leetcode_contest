/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> v;
    
    void dfs(TreeNode* node, int val) {
        v.push_back(val);
        if (node == NULL) return;
        dfs(node->left, val * 2);
        dfs(node->right, val * 2 + 1);
    }
    
    bool isCompleteTree(TreeNode* root) {
        dfs(root, 1);
        sort(v.begin(), v.end());
        for (int i = 1; i < v.size(); i++) {
            if (v[i] != (v[i-1] + 1)) return false;
        }
        return true;
    }
};
