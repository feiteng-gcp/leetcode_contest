#include <bits/stdc++.h>
using namespace std;
using vi = vector<int>;
using vvi = vector<vi>;

class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        queue<TreeNode*> que;
        que.push(root);
        bool finish = 0;
        while (que.size()) {
            TreeNode* t = que.front();
            que.pop();
            if (finish) {
                if (t->left || t->right) {
                    return false;
                }
                continue;
            }
            if (t->left) {
                que.push(t->left);
            } else {
                finish = 1;
            }
            if (finish) {
                if (t->right) {
                    return false;
                }
            } else {

                if (t->right) {
                    que.push(t->right);
                } else {
                    finish = 1;
                }
            }
        }
        return true;
    }
};
