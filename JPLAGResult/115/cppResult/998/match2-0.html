/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        queue<TreeNode*> qu;
        qu.push(root);
        int f=0;
        while(!qu.empty())
        {
            TreeNode *u=qu.front(); qu.pop();
            if(u==NULL)
            {
                f=1;
                continue;
            }
            if(f) return 0;
            qu.push(u->left);
            qu.push(u->right);
        }
        return 1;
    }
};
