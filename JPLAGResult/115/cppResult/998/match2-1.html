/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        queue<TreeNode*> myq;
        myq.push(root);
        int flag=0;
        while (!myq.empty())
        {
          TreeNode* a = myq.front();
            myq.pop();
            if (a->left!=NULL)
            {   if (flag)
                    return false;
                myq.push(a->left);
            }
            else
            {
                flag=1;
            }
            if (a->right!=NULL)
            {   if (flag)
                    return false;
                myq.push(a->right);
            }
            else
            {
                flag=1;
            }
            
            
            
        }
        return true;
    }
};
