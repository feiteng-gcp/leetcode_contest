/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isCompleteTree(TreeNode* root) {
        vector<TreeNode*> nextLevel, currentLevel;
        bool terminate = false;
        int expectedCount = 1;
        currentLevel.push_back(root);
        while (!currentLevel.empty()) {
            terminate = false;
            for (const auto node: currentLevel) {

                if (node->left) {
                    if (terminate) return false;
                    nextLevel.push_back(node->left);
                } else {
                    terminate = true;
                }
                if (node->right) {
                    if (terminate) return false;
                    else nextLevel.push_back(node->right);
                } else {
                    terminate = true;
                }
            }
            if (!nextLevel.empty() && (expectedCount != currentLevel.size())) return false;
            nextLevel.swap(currentLevel);
            expectedCount <<= 1;
            nextLevel.clear();
        }
        return true;
    }
};
