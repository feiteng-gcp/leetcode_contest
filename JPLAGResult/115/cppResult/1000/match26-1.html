class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n = static_cast<int>(A.size());
        if (n == 0) {
            return 0;
        }
        int m = static_cast<int>(A[0].size());
        if (m == 0) {
            return 0;
        }
        vector<vector<bool>> valid(m, vector<bool>(m, true));

        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < m; ++j) {
                for (int k = 0; k < n; ++k) {
                    if (A[k][i] > A[k][j]) {
                        valid[i][j] = false;
                        break;
                    }
                }
            }
        }
        vector<vector<int>> dp(m, vector<int>(m, -1));

        dp[0][0] = 0;
        for (int i = 1; i < m; ++i) {
            dp[i][i] = i;
            for (int j = 0; j < i; ++j) {
                if (dp[i - 1][j] == -1) {
                    continue;
                }
                if (valid[j][i]) {
                    dp[i][i] = min(dp[i][i], dp[i - 1][j]);
                }
                if (dp[i][j] == -1 || dp[i][j] > dp[i - 1][j]) {
                    dp[i][j] = dp[i - 1][j] + 1;
                }
            }
        }
        /*for (int i = 0; i < m; ++i) {
            for (int j = 0; j < m; ++j) {
                cout << dp[i][j] << " ";
            }
            cout << endl;
        }*/
        int minDel = m;
        for (auto del : dp[m - 1]) {
            if (del != -1) {
                minDel = min(minDel, del);
            }
        }
        return minDel;
    }
};
