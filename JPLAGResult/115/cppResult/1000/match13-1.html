class Solution {
public:
    int minDeletionSize(vector<string>& A)
    {
        int m = A.size();
        if (m == 0)
        {
            return 0;
        }
        int n = A[0].size();
        if (n <= 1)
        {
            return 0;
        }
        vector<int> dp(n + 1, 1);

        int ret = 1;
        for (int i = 1; i < n + 1; ++i)
        {
            for (int j = 1; j < i; ++j)
            {
                bool found = true;
                for (int k = 0; k < m; ++k)
                {
                    if (A[k][j - 1] > A[k][i - 1])
                    {
                        found = false;
                        break;
                    }
                }
                if (found)
                {
                    dp[i] = max(dp[i], dp[j] + 1);
                }
            }
            ret = max(ret, dp[i]);
        }
        return n - ret;
    }
};
