class Solution {
public:
    bool check(vector<string>& A, int ij, int ii) {
        for (int i = 0; i < A.size(); ++i) {
            if (A[i][ij] > A[i][ii]) return false;
        }
        return true;
    }
    int minDeletionSize(vector<string>& A) {
        int n = A.size(), m = A[0].length();
        vector<int> dp(m + 1, 0);

        int maxp = 1;
        for (int i = 1; i <= m; ++i) {
            dp[i] = 1;
            for (int j = 1; j < i; ++j) {
                if (check(A, j - 1, i - 1)) {
                    //cout << j << '-' << i << ' ' << dp[j] + 1 << endl;
                    dp[i] = max(dp[j] + 1, dp[i]);
                }
            }
            maxp = max(maxp, dp[i]);
        }
        
        return m - maxp;
    }
};
