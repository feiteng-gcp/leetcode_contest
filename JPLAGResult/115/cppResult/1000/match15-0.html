class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n = A[0].size();
        int m = A.size();
        vector<int> dp;
        dp.resize(n);
        dp[0] = 0;
        for (int i = 1; i < n; i++)
            dp[i] = 1000;
        int ans = 1000;
        for (int i = 1; i < n; i++) {
            dp[i] = i;
            for (int j = 0; j < i; j++) {
                int diff = i - j - 1;
                int ok = 1;
                for (int k = 0; k < m; k++) {
                    if (A[k][j] > A[k][i]) {
                        ok = 0;
                        break;
                    }
                }
                if (ok) dp[i] = min(dp[i], dp[j] + diff);
            }
            ans = min(ans, dp[i] + n - i - 1);
        }
        return ans;
    }
};
