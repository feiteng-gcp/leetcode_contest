class Solution {
public:
    int minDeletionSize(vector<string>& A) {

        const int n = A.size(), m = A[0].size();
        vector<int> dp(m, 1);
        int max_len = 0;
        for (int i = 1; i < m; ++i) {
            for (int j = i - 1; j >= 0; --j) {
                bool flag = true;
                for (int k = 0; k < n; ++k) {
                    if (A[k][j] > A[k][i]) {
                        flag = false;
                        break;
                    }
                }
                if (flag) {
                    dp[i] = max(dp[i], dp[j] + 1);
                }
            }
            max_len = max(max_len, dp[i]);
        }
        return m - max_len;
    }
};
