class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int m = (int)A.size(), n = (int)A[0].size();
    vector<int> dp(n, 0);
    for(int i = 1; i < n; i++)
    {

        dp[i] = i;
        for(int j = 0; j < i; j++)
        {
            bool ok = true;
            for(int k = 0; k < m; k++)
                if(A[k][j] > A[k][i])
                {
                    ok = false;
                    break;
                }
            if(ok) dp[i] = min(dp[i], dp[j] + (i - j - 1));
        }
    }
    int ans = n - 1;
    for(int i = 0; i < n; i++) ans = min(ans, dp[i] + n - i - 1);
    return ans;
    }
};
