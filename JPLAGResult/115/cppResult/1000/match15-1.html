class Solution {
public:
  int minDeletionSize(vector<string> &A) {
    if (A.empty() || A[0].empty())
      return 0;

    int dp[110] = {}, ans = 0;
    int l = A[0].size();
    for (int i = 0; i < l; ++i) {
      dp[i] = 1;
      for (int j = 0; j < i; ++j) {
        if (dp[j] + 1 <= dp[i]) continue;

        bool ok = true;
        for (int k = 0; k < A.size(); ++k) {
          if (A[k][j] > A[k][i]) {
            ok = false;
            break;
          }
        }
        if (ok) dp[i] = dp[j] + 1;
      }
      ans = max(ans, dp[i]);
    }
    return l - ans;
  }
};
