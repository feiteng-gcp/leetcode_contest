int g[105][105];
int dis[105];
class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n=A.size(),m=A[0].size();
	memset(g,0,sizeof(g));

	for(int i=0;i<m;i++)
	{
		for(int j=i+1;j<m;j++)
		{
			int ok=1;
			for(int k=0;k<n;k++)
			{
				if(A[k][j]<A[k][i])
				{
					ok=0;
					break;
				}
			}
			if(ok)g[i][j]=1;
		}
	}
	memset(dis,0,sizeof(dis));
	for(int i=0;i<m;i++)
	{
		if(dis[i]==0)dis[i]=1;
		for(int j=i+1;j<m;j++)
		{
			if(g[i][j])dis[j]=max(dis[j],dis[i]+1);
		}
	}
	int ans=0;
	for(int i=0;i<m;i++)
	{
		ans=max(ans,dis[i]);
		//printf("%d:%d\n",i,dis[i]);
	}
	return m-ans;
    }
};
