class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n = A.size();
        int m = A[0].size();

        vector<int> dp(m+1, 1);

        int ans = m-1;

        for(int i=0; i<m; i++){
            for(int j=0; j<i; j++){
                bool ok = true;
                for(int k=0; k<n; k++){
                    if(A[k][j] <= A[k][i]) continue;
                    ok = false;
                    break;
                }
                if(ok){
                    dp[i] = max(dp[i], dp[j]+1);
                    ans = min(ans, m-dp[i]);
                }
            }
        }

        return ans;
    }
};
