class Solution {
public:
    
    int minDeletionSize(vector<string>& A) {
        int n=A.size(),m=A[0].size();
        vector<int> dp(m,1);
        for(int i=1;i<m;i++)
        {
            for(int j=0;j<i;j++)
            {
                int f=1;
                for(int k=0;k<n;k++)
                {
                    if(A[k][i]<A[k][j])
                    {
                        f=0;break;
                    }
                }
                if(f)
                    dp[i]=max(dp[i],dp[j]+1);
            }
        }
        int ans=m;
        for(int i=0;i<m;i++)
            ans=min(ans,m-dp[i]);
        return ans;
    }
};
