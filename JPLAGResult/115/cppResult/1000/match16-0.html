class Solution {
public:
    int minDeletionSize(vector<string>& s) {
        int n = s.size(), m = s[0].size();
        vector<int> dp(m);
        dp[m - 1] = 1;
        for (int i = m - 2; i >= 0; --i) {
            dp[i] = 1;
            for (int j = i + 1; j < m; ++j) {
                bool found = false;
                for (int k = 0; k < n; ++k) {
                    if (s[k][i] > s[k][j]) found = true;
                }
                if (found) continue;
                dp[i] = max(dp[i], dp[j] + 1);
            }
        }
        int ret = 0;
        for (int i = 0; i < m; ++i) ret = max(ret, dp[i]);
        return m - ret;
    }
};
