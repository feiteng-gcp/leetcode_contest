class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int N = A[0].length();
        vector<int> dp(N);
        int max_len = 1;
        dp[0] = 1;
        
        for (int i = 1; i < N; ++i) {
            dp[i] = 1;
            for (int j = i-1; j >= 0; --j) {
                if (dp[j] + 1 > dp[i] && valid(A, j, i)) {
                    dp[i] = dp[j] + 1;
                }
            }
            max_len = max(max_len, dp[i]);
        }
        
        return N - max_len;
    }
private:
    bool valid(vector<string>& A, int j, int i) {
        for (auto& s : A) {
            if (s[j] > s[i]) {
                return false;
            }
        }
        return true;
    }
};
