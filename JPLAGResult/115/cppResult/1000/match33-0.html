class Solution {
public:
#define INF 1000000
	int minDeletionSize(vector<string>& A) {
		int n = A.size();
		for (int i = 0; i < n; i++) {
			A[i] = "#" + A[i];
		}
		vector<int>dp(A[0].size() + 5, -INF);
		int maxx = 0; dp[0] = 0;
		for (int i = 1; i <= A[0].size()-1; i++) {
			for (int j = 0; j < i; j++) {
				bool remove = false;
				for (int a = 0; a < n; a++) {
					if (A[a][j] > A[a][i]) {
						remove = true;
						break;
					}
				}
				if (remove)continue;
				dp[i] = max(dp[j]+1, dp[i]);
			}
			maxx = max(maxx, dp[i]);
		}
		return A[0].size()-1 - maxx;
	}
};
