class Solution {
public:
    int dp[100];
    int minDeletionSize(vector<string>& a) {
        int ans = 0;
        for (int i = 0; i < a[0].size(); i++) {
            int best = 0;
            for (int j = 0; j < i; j++) {
                bool flag = true;
                for (int k = 0; k < a.size(); k++)
                    if (a[k][j] > a[k][i]) {
                        flag = false;
                        break;
                    }
                if (flag)
                    best = max(best, dp[j]);
            }
            dp[i] = best + 1;
            ans = max(ans, dp[i]);
        }
        return a[0].size() - ans;
    }
};
