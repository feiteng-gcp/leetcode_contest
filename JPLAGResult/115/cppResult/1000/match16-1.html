class Solution {
public:
    int dp[105] ; 
    int minDeletionSize(vector<string>& A) {
        memset( dp , 0 , sizeof( dp ) ) ; 
        for( int i=0 ; i< A[0].size() ; i++)
            dp[i] = 1;         
        for( int i=1 ; i< A[0].size() ; i++){

            for( int j=0 ; j < i ; j ++ ){
                bool flag = false ; 
                for( int k =0 ; k <A.size() ; k++){
                    if( A[k][j] > A[k][i] ) 
                        flag = true ; 
                }
                if( flag ) 
                    continue ; 
                dp[i] = max( dp[j] + 1 , dp[i] ) ; 
            }
        }
        int ans = 0 ; 
        for( int i=0 ; i< A[0].size() ; i++) 
            ans = max( ans , dp[i] ) ; 
        return A[0].size()- ans ; 
    }
};
