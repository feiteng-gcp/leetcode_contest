class Solution {
public:
    int minDeletionSize(vector<string>& a) {

        vector<int> dp(a[0].size(), 1);
        int res = 0;
        for (int i = 0; i < a[0].size(); ++i) {
            res = max(res, dp[i]);
            for (int j = i + 1; j < a[0].size(); ++j) {
                if (good(a, i, j)) {
                    dp[j] = max(dp[j], dp[i] + 1);
                }
            }
        }
        return a[0].size() - res;
    }
    

    bool good(const vector<string>& a, int last, int now) {
        for (int i = 0; i < a.size(); ++i) {
            if (a[i][now] < a[i][last]) return false;
        }
        return true;
    }
};
