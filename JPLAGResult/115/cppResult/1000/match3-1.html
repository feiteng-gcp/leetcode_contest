class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n = A.size();

        int m = A[0].length();
        
        vector<int> max_chain(m, 1);
        
        for (int i = 1; i < m; i++) {
            for (int j = 0; j < i; j++) {

                bool precedes = true;
                for (int k = 0; k < n; k++) {
                    if (A[k][j] > A[k][i]) {
                        precedes = false;
                        break;
                    }
                }
                if (!precedes)
                    continue;
                

                max_chain[i] = max(max_chain[i], 1 + max_chain[j]);
            }
        }
     
        int ans = 0;
        for (int i = 0; i < m; i++)
            ans = max(ans, max_chain[i]);
        return m - ans;
    }
};
