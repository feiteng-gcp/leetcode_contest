class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        bool unAdj[105][105];
        memset(unAdj,false, sizeof(unAdj));
        int n = A.size();
        int s = A[0].length();
        for(int ix=0;ix<n;ix++){
            for(int i=0;i<s;i++){
                for(int j=i+1;j<s;j++){
                    if(A[ix][i] > A[ix][j]) {
                        unAdj[i][j] = true;
                    }
                }
            }
        }
        int len[105];
        int ans = 1;
        len[s-1]=1;
        for(int i=s-2;i>=0;i--){
            len[i] = 1;
            for(int j=i+1;j<s;j++){
                if(unAdj[i][j]==false){
                    len[i] = max(len[i], 1+len[j]);
                    ans = max(ans, len[i]);
                }
            }
        }
        return s-ans;
    }
};
