class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int n = A.size();
        if (n == 0) {
            return 0;
        }
        int m = A[0].size();
        if (m == 0) {
            return 0;
        }
        const int INF = INT_MAX / 20;
        vector<int> dp(m, INF);
        dp[0] = 0;
        vector<int> nxt = dp;
        
        vector<vector<bool>> valid(m, vector<bool>(m, false));

        for (int i = 0; i < m; ++i) {
            for (int j = i + 1; j < m; ++j) {
                int cnt = 0;
                for (int k = 0; k < n; ++k) {
                    if (A[k][i] > A[k][j]) {
                        ++cnt;
                    }
                }
                if (cnt == 0) {
                    valid[i][j] = true;
                }
            }
        }
        
        for (int i = 1; i < m; ++i) {
            for (int j = 0; j <= i; ++j) {
                nxt[j] = i;
                
                for (int k = 0; k < j; ++k) {
                    if (valid[k][j]) {
                        nxt[j] = min(nxt[j], dp[k]);
                    }
                }
                // cout << i << " " << j << " : " << nxt[j] << endl;
            }
        
            swap(dp, nxt);
        }
        
        int res = INF;
        for (int i = 0; i < m; ++i) {
            res = min(res, dp[i]);   
        }
        return res;
    }
};
