class Solution {
public:
    vector<vector<int> > g;
    int n,m;
    int dx[4]={0,0,-1,1};
    int dy[4]={1,-1,0,0};
    void dfs(int nx,int ny)
    {
        if(g[nx][ny]) return;
        g[nx][ny]=1;
        for(int i=0;i<4;i++)
        {
            int x=nx+dx[i],y=ny+dy[i];
            if(x<0||x>=n*3||y<0||y>=m*3||g[x][y]) continue;
            dfs(x,y);
        }
    }
    int regionsBySlashes(vector<string>& grid) {
        n=grid.size(); m=grid[0].length();
        g.assign(n*3,vector<int>(m*3));
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
            {
                if(grid[i][j]=='\\')
                {
                    for(int k=0;k<3;k++)
                        g[i*3+k][j*3+k]=1;
                }
                if(grid[i][j]=='/')
                {
                    for(int k=0;k<3;k++)
                        g[i*3+k][j*3+2-k]=1;
                }
            }
            //for(int i=0;i<n*3;i++)
            //{
            //    for(int j=0;j<m*3;j++)
            //        cout<<g[i][j];
            //    cout<<endl;
            //}
        int ans=0;
        for(int i=0;i<n*3;i++)
            for(int j=0;j<m*3;j++)
                if(!g[i][j])
                {
                    ans++;
                    dfs(i,j);
                }
        return ans;
    }
};
