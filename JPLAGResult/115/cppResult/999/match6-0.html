class Solution {
public:
    int regionsBySlashes(vector<string>& grid) {
        int n = grid.size() * 2;
        vector<vector<int>> graph(grid.size() * 2, vector<int>(grid.size() * 2, 1));
        for(int i = 0; i < grid.size(); ++i) {
            for(int j = 0; j < grid.size(); ++j) {
                if(grid[i][j] == '\\') {
                    graph[i * 2][j * 2] = -1;
                    graph[i * 2 + 1][j * 2 + 1] = -1;
                }
                else if(grid[i][j] == '/') {
                    graph[i * 2][j * 2 + 1] = -2;
                    graph[i * 2 + 1][j * 2] = -2;
                }
            }
        }
        // for(auto v : graph) {
        //     for(auto item : v) cout << item << " ";
        //     cout << endl;
        // }
        int count = 0;
        int dir[4][2] = {{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
        for(int i = 0; i < n; ++i) {
            for(int j = 0; j < n; ++j) {
                if(graph[i][j] == 1) {
                    // for(auto v : graph) {
                    //     for(auto item : v) cout << item << " ";
                    //     cout << endl;
                    // }
                    // cout << endl;
                    count += 1;
                    queue<pair<int, int>> que;
                    que.push({i, j});
                    graph[i][j] = 0;
                    while(que.size()) {
                        auto tmp = que.front();
                        que.pop();
                        for(int i = 0; i < 4; ++i) {
                            int dx = tmp.first + dir[i][0], dy = tmp.second + dir[i][1];
                            if(isValid(dx, dy, n) && graph[dx][dy] == 1) {
                                graph[dx][dy] = 0;
                                que.push({dx, dy});
                            }
                        }
                        int x = tmp.first, y = tmp.second;
                        if(isValid(x - 1, y - 1, n) && (graph[x - 1][y] == -1 && graph[x][y - 1] == -1) && graph[x - 1][y - 1] == 1) {
                            graph[x - 1][y - 1] = 0;   
                            que.push({x - 1, y - 1});
                        }
                        if(isValid(x - 1, y + 1, n) && (graph[x - 1][y] == -2 && graph[x][y + 1] == -2) && graph[x - 1][y + 1] == 1) {
                            graph[x - 1][y + 1] = 0;
                            que.push({x - 1, y + 1});
                        }
                        if(isValid(x + 1, y - 1, n) && (graph[x][y - 1] == -2 && graph[x + 1][y] == -2) && graph[x + 1][y - 1] == 1) {
                            graph[x + 1][y - 1] = 0;
                            que.push({x + 1, y - 1});
                        }
                        if(isValid(x + 1, y + 1, n) && (graph[x][y + 1] == -1 && graph[x + 1][y] == -1) && graph[x + 1][y + 1] == 1) {
                            graph[x + 1][y + 1] = 0;
                            que.push({x + 1, y + 1});
                        }
                    }
                }
            }
        }
        return count;
    }
private:
    bool isValid(int x, int y, int n) {
        return (0 <= x && x < n && 0 <= y && y < n);
    }
};
