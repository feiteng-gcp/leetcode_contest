class Solution {
public:
    int regionsBySlashes(vector<string>& grid) {
        const int m = grid.size(), n = grid[0].size();
        vector<int> roots(m * n * 4, -1);
        auto index = [&](int i, int j, int k) -> int {
            return i * n * 4 + j * 4 + k;
        };
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if ('/' != grid[i][j]) {
                    merge(roots, index(i, j, 0), index(i, j, 1));
                    merge(roots, index(i, j, 2), index(i, j, 3));
                }
                if ('\\' != grid[i][j]) {
                    merge(roots, index(i, j, 0), index(i, j, 3));
                    merge(roots, index(i, j, 1), index(i, j, 2));
                }
                if (i > 0) {
                    merge(roots, index(i, j, 0), index(i - 1, j, 2));
                }
                if (j > 0) {
                    merge(roots, index(i, j, 3), index(i, j - 1, 1));
                }
            }
        }
        int result = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                for (int k = 0; k < 4; ++k) {
                    if (roots[index(i, j, k)] < 0) {
                        result++;
                    }
                }
            }
        }
        return result;
    }
private:
    void merge(vector<int> &roots, int i, int j) const {
        i = get_root(roots, i);
        j = get_root(roots, j);
        if (i == j) {
            return;
        } else if (roots[i] < roots[j]) {
            roots[i] += roots[j];
            roots[j] = i;
        } else {
            roots[j] += roots[i];
            roots[i] = j;
        }
    }
    int get_root(vector<int> &roots, int i) const {
        if (roots[i] < 0) {
            return i;
        } else {
            return roots[i] = get_root(roots, roots[i]);
        }
    }
};
