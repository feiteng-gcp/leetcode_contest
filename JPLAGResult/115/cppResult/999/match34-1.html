class Solution {
public:
    int vis[35][35][4];

void bfs(int i, int j, int k, int tp, vector<string>& G)
{
    queue<vector<int> > q;
    q.push({i, j, k});
    while(!q.empty())
    {
        vector<int> cur = q.front(); q.pop();
        i = cur[0]; j = cur[1]; k = cur[2];
        vector<vector<int> > nxt;
        int ni = i, nj = j, nk = k;
        switch(cur[2])
        {
            case 0:
                ni = i - 1; nj = j; nk = 2;
                if(G[i][j] != '\\' && !vis[i][j][1])
                    nxt.push_back({i, j, 1});
                if(G[i][j] != '/' && !vis[i][j][3])
                    nxt.push_back({i, j, 3});
                break;
            case 1:
                ni = i; nj = j - 1; nk = 3;
                if(G[i][j] != '\\' && !vis[i][j][0])
                    nxt.push_back({i, j, 0});
                if(G[i][j] != '/' && !vis[i][j][2])
                    nxt.push_back({i, j, 2});
                break;
            case 2:
                ni = i + 1; nj = j; nk = 0;
                if(G[i][j] != '\\' && !vis[i][j][3])
                    nxt.push_back({i, j, 3});
                if(G[i][j] != '/' && !vis[i][j][1])
                    nxt.push_back({i, j, 1});
                break;
            case 3:
                ni = cur[0]; nj = j + 1; nk = 1;
                if(G[i][j] != '\\' && !vis[i][j][2])
                    nxt.push_back({i, j, 2});
                if(G[i][j] != '/' && !vis[i][j][0])
                    nxt.push_back({i, j, 0});
                break;
        }
        if(ni >= 0 && ni < G.size() && nj >= 0 && nj < G.size() && !vis[ni][nj][nk])
            nxt.push_back({ni, nj, nk});
        for(int l = 0; l < nxt.size(); l++)
        {
            q.push(nxt[l]);
            vis[nxt[l][0]][nxt[l][1]][nxt[l][2]] = tp;
        }
    }
}

int regionsBySlashes(vector<string>& grid)
{
    memset(vis, 0, sizeof(vis));

    int len = (int)grid.size(), tp = 0;
    for(int i = 0; i < len; i++)
        for(int j = 0; j < len; j++)
            for(int k = 0; k < 4; k++)
                if(!vis[i][j][k])
                {
                    vis[i][j][k] = ++tp;
                    bfs(i, j, k, tp, grid);
                }
    return tp;
}
};
