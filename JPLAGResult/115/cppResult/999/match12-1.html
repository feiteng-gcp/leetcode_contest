class Solution {
public:
    int g[100][100];
    
    void flood(int x, int y, int val) {
        if (g[x][y] != 0)
            return;
        //cout << "flood " << x << " " << y << " " << val << endl;
        g[x][y] = val;
        flood(x - 1, y, val);
        flood(x + 1, y, val);
        flood(x, y - 1, val);
        flood(x, y + 1, val);
    }
    
    int regionsBySlashes(vector<string>& grid) {
        int n = grid.size();
        int m = 3 * n + 2;
        assert(m < 100);
        for (int i = 0; i < m; i++)

            g[i][0] = g[i][m-1] = g[0][i] = g[m-1][i] = -1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int x = 3 * i + 1;
                int y = 3 * j + 1;
                for (int ii = 0; ii < 3; ii++) {
                    for (int jj = 0; jj < 3; jj++) {
                        g[x + ii][y + jj] = 0;
                    }
                }
                if (grid[i][j] == '\\') {
                    g[x][y] = g[x+1][y+1] = g[x+2][y+2] = -1;   
                } else if (grid[i][j] == '/') {

                    g[x+2][y] = g[x+1][y+1] = g[x][y+2] = -1;          
                }
            }
        }
        
        int num_comps = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < m; j++) {
                if (g[i][j] == 0) {
                    num_comps++;
                    flood(i, j, num_comps);
                }
            }
        }
        return num_comps;
    }
};
