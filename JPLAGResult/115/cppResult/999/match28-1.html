class Solution {
public:
    struct dsa {
        dsa* parent = this;
        bool tag = false;
        int rank = 0;

        dsa* head(){
            if(parent != parent->parent){
                parent = parent->head();
            }

            return parent;
        }
        void merge(dsa* sec){
            dsa* fir = head();
            sec = sec->head();

            if(fir == sec){
                return;
            }

            if(fir->rank < sec->rank){
                fir->parent = sec;
            } else if(fir->rank > sec->rank){
                sec->parent = fir;
            } else {
                sec->parent = fir;
                fir->rank++;
            }
        }
    };
    
    int ay[4] = {1, -1, 0, 0};
    int ax[4] = {0, 0, 1, -1};
    dsa dsu[30][30][2];
    int coming[30][30][4];
    int going[30][30][4];
    int regionsBySlashes(vector<string>& grid) {

        for(int i = 0; i < grid.size(); i++){
            for(int j = 0; j < grid.size(); j++){
                if(grid[i][j] == '\\'){
                    coming[i][j][0] = 0;
                    coming[i][j][1] = 1;
                    coming[i][j][2] = 1;
                    coming[i][j][3] = 0;
                } else if(grid[i][j] == '/'){
                    coming[i][j][0] = 0;
                    coming[i][j][1] = 1;
                    coming[i][j][2] = 0;
                    coming[i][j][3] = 1;
                } else {
                    coming[i][j][0] = 0;
                    coming[i][j][1] = 0;
                    coming[i][j][2] = 0;
                    coming[i][j][3] = 0;
                }
            }
        }
        for(int i = 0; i < grid.size(); i++){
            for(int j = 0; j < grid.size(); j++){
                for(int k = 0; k < 4; k++){
                    int ny = i + ay[k];
                    int nx = j + ax[k];
                    
                    if(0 <= ny and ny < grid.size() and 0 <= nx and nx < grid.size()){
                        dsu[i][j][!coming[i][j][k]].merge(&dsu[ny][nx][coming[ny][nx][k]]);
                    }
                }
                
                if(grid[i][j] == ' '){
                    dsu[i][j][0].merge(&dsu[i][j][1]);
                }
            }
        }
        

        int res = 0;
        for(int i = 0; i < grid.size(); i++){
            for(int j = 0; j < grid.size(); j++){
                if(!dsu[i][j][0].head()->tag){
                    res++;
                    dsu[i][j][0].head()->tag = true;
                }
                if(!dsu[i][j][1].head()->tag){
                    res++;
                    dsu[i][j][1].head()->tag = true;
                }
            }
        }
        
        return res;
    }
};
