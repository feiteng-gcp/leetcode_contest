class Solution {
public:
    int regionsBySlashes(vector<string>& grid) {
        if(grid.size()==0)
            return 0;
        int N = grid.size();
        vector<vector<int>> graph(3*N, vector<int> (3*N,0));
        
        for(int i =0; i<N;i++){
            for(int j =0; j<N;j++){
                int tx= 3*i;
                int ty = 3*j;
                if (grid[i][j] == '/'){
                    graph[tx+2][ty]=1;
                    graph[tx+1][ty+1]=1;
                    graph[tx][ty+2]=1;
                }else if (grid[i][j] == '\\'){
                    graph[tx][ty] = 1;
                    graph[tx+1][ty+1]=1;
                    graph[tx+2][ty+2]=1;
                }
            }
        }
        
        DSU myuf(3*N*3*N);
        
        for(int i =0; i <3*N;i++){
            for(int j =0; j<3*N;j++){
                int nmap = i*3*N+j;
                if (graph[i][j] == 1){
                    myuf.parents[nmap]=-1;
                    continue;
                }
                if (i > 0 && graph[i-1][j] == 0)
                    myuf.Union(nmap,(i-1)*3*N+j);
                if (j > 0 && graph[i][j-1] == 0)
                    myuf.Union(nmap,i*3*N+j-1);
            }
        }
        
        int res=0;
        for(int i =0; i <myuf.parents.size();i++){
            if (myuf.parents[i] == i)
                res++;
        }
        return res;
        
    }
private:
    class DSU{
        public:
        vector<int> parents;
        vector<int> rank;
        DSU(int N){
            parents = vector<int> (N);
            rank = vector<int> (N,0);
            for(int i =0; i <N;i++)
                parents[i] = i;
        }
        
        void Union(int a, int b){
            int pa = Find(a);
            int pb = Find(b);
            if (pa == pb)
                return;
            if (rank[pa] > rank[pb])
                swap(pa,pb);
            if (rank[pa] == rank[pb])
                rank[pb]++;
            parents[pa]=pb;
        }
        
        int Find(int a){
            if (parents[a] !=a)
                parents[a] = Find(parents[a]);
            return parents[a];
        }
        private:

    };
};
