class Solution {
public:
    int regionsBySlashes(vector<string>& grid) {
        int n = static_cast<int>(grid.size());
        if (n == 0) {
            return 0;
        }
        int m = static_cast<int>(grid[0].size());
        if (m == 0) {
            return 0;
        }
        auto encode = [&m](int i, int j, int p) {
            return (i * m + j) * 4 + p;
        };
        init(encode(n - 1, m - 1, 3) + 1);
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (i > 0) {
                    combine(encode(i - 1, j, 2), encode(i, j, 0));
                }
                if (j > 0) {
                    combine(encode(i, j - 1, 1), encode(i, j, 3));
                }
                if (grid[i][j] != '/') {
                    combine(encode(i, j, 0), encode(i, j, 1));
                    combine(encode(i, j, 3), encode(i, j, 2));
                }
                if (grid[i][j] != '\\') {
                    combine(encode(i, j, 0), encode(i, j, 3));
                    combine(encode(i, j, 1), encode(i, j, 2));
                }
            }
        }
        unordered_set<int> groups;
        for (int i = 0; i < static_cast<int>(parent.size()); ++i) {
            groups.insert(find(i));
        }
        return static_cast<int>(groups.size());
    }
    
private:
    vector<int> parent;
    
    void init(int n) {
        parent = vector<int>(n);
        for (int i = 0; i < n; ++i) {
            parent[i] = i;
        }
    }
    
    int find(int x) {
        if (x == parent[x]) {
            return x;
        }
        return parent[x] = find(parent[x]);
    }
    
    void combine(int x, int y) {
        parent[find(x)] = find(y);
    }
};
