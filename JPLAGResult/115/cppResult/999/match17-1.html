class Solution {
public:
    vector<int> dx = {0,0,1,-1};
    vector<int> dy = {1,-1,0,0};
    void dfs(vector<vector<int>>& cells, int r, int c){
        cells[r][c] = 1;
        for(int k=0; k<4; k++){
            int rr = r + dy[k];
            int cc = c + dx[k];
            if(rr < 0 || rr >= cells.size()) continue;
            if(cc < 0 || cc >= cells.size()) continue;
            if(cells[rr][cc] == 0) dfs(cells, rr, cc);
        }
    }
    int regionsBySlashes(vector<string>& grid) {
        int n = grid.size();

        vector<vector<int>> cells(n*3, vector<int>(n*3, 0));
        for(int r=0; r<n; r++) for(int c=0; c<n; c++){
            if(grid[r][c] == '/'){
                cells[r*3  ][c*3+2] = 1;
                cells[r*3+1][c*3+1] = 1;
                cells[r*3+2][c*3  ] = 1;
            }
            if(grid[r][c] == '\\'){

                cells[r*3  ][c*3  ] = 1;
                cells[r*3+1][c*3+1] = 1;
                cells[r*3+2][c*3+2] = 1;
            }
        }
        int res = 0;
        for(int r=0; r<n*3; r++) for(int c=0; c<n*3; c++){
            if(cells[r][c] == 1) continue;
            res++;
            dfs(cells, r, c);
        }
        return res;
    }
};
