class Solution {
public:
    void flip(vector<vector<int>> &a, int i, int j) {
        int sz = a.size();
        a[i][j] = 1;
        if (i > 0 && a[i-1][j] == 0)
            flip(a, i-1, j);
        if (i < sz-1 && a[i+1][j] == 0)
            flip(a, i+1, j);
        if (j > 0 && a[i][j-1] == 0)
            flip(a, i, j-1);
        if (j < sz-1 && a[i][j+1] == 0)
            flip(a, i, j+1);
    }
    int regionsBySlashes(vector<string>& grid) {

        int sz = grid.size();
        vector<vector<int>> a(sz*3, vector<int>(sz*3, 0));
        for (int i = 0; i < sz; i++) {
            for (int j = 0; j < sz; j++) {
                if (grid[i][j] == '/') {
                    for (int k = 0; k < 3; k++)
                        a[i*3+k][j*3+2-k] = 1;
                } else if (grid[i][j] == '\\') {
                    for (int k = 0; k < 3; k++)
                        a[i*3+k][j*3+k] = 1;
                }
            }
        }

        int cnt = 0;
        for (int i = 0; i < sz*3; i++) {
            for (int j = 0; j < sz*3; j++) {
                if (a[i][j] == 0) {
                    flip(a, i, j);
                    cnt ++;
                }
            }
        }
        
        return cnt;
    }
};
