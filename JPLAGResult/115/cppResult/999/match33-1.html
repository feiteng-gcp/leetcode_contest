struct UF {
    vector<int> parent;
    int regions;
    
    UF(int N) {
        regions = N;
        parent.resize(N);
        for (int i = 0; i < N; i++) parent[i] = i;
    }
    
    int Find(int x) {
        if (parent[x] == x) return x;
        return parent[x] = Find(parent[x]);
    }
    
    void Union(int x, int y) {
        x = Find(x);
        y = Find(y);
        if (x != y) {
            regions--;
            parent[x] = y;
        }

    }
};

class Solution {
    int N;
    int P(int i, int j, int k) { return (i*N+j)*4+k; }
public:
    int regionsBySlashes(vector<string>& grid) {
        N = grid.size();
        UF uf(4*N*N);

        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                printf("i=%d j=%d c=[%c]\n", i, j, grid[i][j]);
                if (grid[i][j] == '/' || grid[i][j] == ' ') {
                    uf.Union(P(i,j,0), P(i,j,3));
                    uf.Union(P(i,j,1), P(i,j,2));
                }
                if (grid[i][j] == '\\' || grid[i][j] == ' ') {
                    uf.Union(P(i,j,0), P(i,j,1));
                    uf.Union(P(i,j,2), P(i,j,3));
                }
                
                if (i > 0)  uf.Union(P(i,j,0), P(i-1,j,2));
                if (j+1 < N) uf.Union(P(i,j,1), P(i,j+1,3));
                if (i+1 < N) uf.Union(P(i,j,2), P(i+1,j,0));
                if (j > 0)  uf.Union(P(i,j,3), P(i,j-1,1));
            }
        }
        
        return uf.regions;
    }
};
