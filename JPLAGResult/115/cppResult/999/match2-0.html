class Solution {
public:
	vector<int>p;
	int getroot(int x) {
		if (p[x] == x)return x;
		return p[x] = getroot(p[x]);
	}
	void join(int x,int y){
		int a = getroot(x), b = getroot(y);
		if (a == b)return;
		p[a] = b;
	}
	int regionsBySlashes(vector<string>& grid) {
		int n = grid.size();
		p.resize(4 * n*n);
		for (int i = 0; i < p.size(); i++) {
			p[i] = i;
		}
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				int index = j + i * n;
				char c = grid[i][j];
				if (j != n - 1) join(n*n * 2 + index, index + 1);
				if (i != n - 1)join(n*n * 3 + index, n*(n + 1) + index);
				
				if (c == '\\') {
					join(index, index + n * n*3);
					join(n*n * 2 + index, index + n*n);
				}
				if (c == ' '){
					join(index, n*n + index);
					join(n*n + index, n*n * 2 + index);
					join(n*n + index, 3 * n*n + index);
				}
				if (c == '/') {
					join(n*n + index, index);
					join(n*n * 3 + index, n*n * 2 + index);
				}

			}
		}
		set<int>ct;
		for (int i = 0; i < p.size(); i++) {
			ct.insert(getroot(i));
		}
		return ct.size();
	}
};
