class Solution {
public:
    int ufind(int a, vector<int>&uroot) {
        if(a == uroot[a]) return a;
        return (uroot[a] = ufind(uroot[a], uroot));
    }
    void uunion(int a, int b, vector<int>&uroot) {
        int ra, rb;
        ra = ufind(a, uroot);
        rb = ufind(b, uroot);
        uroot[ra] = rb;
    }
    int regionsBySlashes(vector<string>& grid) {
        int R, C;
        R = grid.size();
        C = grid[0].size();
        vector<int> uroot(R*C*4);

        for(int i=0; i<R*C*4; i++) {
            uroot[i] = i;
        }
        for(int r=0; r<R; r++) {
            for(int c=0; c<C; c++) {
                int id = r * C + c;
                printf("%d:[%c]\n", id, grid[r][c]);
                if(r != 0) {
                    // up
                    int idu = id - C;
                    uunion(id*4+0, idu*4+3, uroot);
                }
                if(r != R-1) {
                    // down
                    int idd = id + C;
                    uunion(id*4+3, idd*4+0, uroot);
                }
                if(c != 0) {
                    // left
                    int idl = id - 1;
                    uunion(id*4+1, idl*4+2, uroot);
                }
                if(c != C-1) {
                    // right
                    int idr = id + 1;
                    uunion(id*4+2, idr*4+1, uroot);
                }

                if(grid[r][c] == ' ') {
                    uunion(id*4+0, id*4+1, uroot);
                    uunion(id*4+1, id*4+2, uroot);
                    uunion(id*4+2, id*4+3, uroot);
                }
                if(grid[r][c] == '/') {
                    uunion(id*4+0, id*4+1, uroot);
                    uunion(id*4+2, id*4+3, uroot);
                }
                if(grid[r][c] == '\\') {
                    uunion(id*4+0, id*4+2, uroot);
                    uunion(id*4+1, id*4+3, uroot);
                }
            }
        }
        int ans = 0;
        for(int i=0; i<R*C*4; i++) {
            if(i == ufind(i, uroot)) {
                ans++;
            }
        }
        return ans;
    }
};
