class Solution {
    int lg;
public:
    int regionsBySlashes(vector<string>& grid) {
        lg = grid.size();

        vector<vector<int>> dp(lg*3, vector<int>(lg*3));

        for (int i = 0; i < lg; i++) {
            for (int j = 0; j < lg; j++) {
                if (grid[i][j] == ' ') continue;
                int si = i*3, sj = j*3;
                if (grid[i][j] == '\\') {
                    dp[si][sj] = dp[si+1][sj+1] = dp[si+2][sj+2] = 1;
                } else {
                    dp[si+2][sj] = dp[si+1][sj+1] = dp[si][sj+2] = 1;
                }
            }
        }

        lg *= 3;
        int ans = 0;
        for (int i = 0; i < lg; i++) {
            for (int j = 0; j < lg; j++) {
                if (dp[i][j] == 0) {
                    dfs(dp, i, j);
                    ans++;
                }
            }
        }

        return ans;
    }

    void dfs(vector<vector<int>>& dp, int y, int x) {
        dp[y][x] = 1;
        if (y > 0 && dp[y-1][x]==0) dfs(dp, y-1, x);
        if (y < lg-1 && dp[y+1][x]==0) dfs(dp, y+1, x);
        if (x > 0 && dp[y][x-1]==0) dfs(dp, y, x-1);
        if (x < lg-1 && dp[y][x+1]==0) dfs(dp, y, x+1);
    }
};
