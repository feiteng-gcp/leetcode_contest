class Solution {
    int n;
    int numIslands(vector<vector<bool>>& grid) {

        int count = 0;
        for (int i = 0; i < 3*n; i++){
            for (int j = 0; j < 3*n; j++){
                if (grid[i][j]) {
                    DFSMarking(grid, i, j);
                    ++count;
                }
            }
        }
        return count;
    }

    void DFSMarking(vector<vector<bool>>& grid, int i, int j) {
        if (i < 0 || j < 0 || i >= 3*n || j >= 3*n || grid[i][j] != 1) return;
        grid[i][j] = 0;
        DFSMarking(grid, i + 1, j);
        DFSMarking(grid, i - 1, j);
        DFSMarking(grid, i, j + 1);
        DFSMarking(grid, i, j - 1);
    }
public:
    int regionsBySlashes(vector<string>& grid) {
        n = grid.size();
        vector<vector<bool>> grid_b(3*n,vector<bool>(3*n,1));
        
        int i,j;
        for(i = 0;i<n;i++){
            for(j = 0;j<n;j++){
                if(grid[i][j]=='/'){
                    grid_b[3*i+2][3*j]=0;
                    grid_b[3*i+1][3*j+1]=0;
                    grid_b[3*i][3*j+2]=0;
                }else if(grid[i][j]=='\\'){
                    grid_b[3*i][3*j]=0;
                    grid_b[3*i+1][3*j+1]=0;
                    grid_b[3*i+2][3*j+2]=0;
                }
            }
        }
        // for(i = 0;i<3*n;i++){
        //     for(j = 0;j<3*n;j++){
        //         cout<<grid_b[i][j];
        //     }
        //     cout<<endl;
        // }
        // cout<<endl;
        
                         
        return numIslands(grid_b);
    }
};
