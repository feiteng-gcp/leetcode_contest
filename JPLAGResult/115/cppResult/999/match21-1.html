class Solution {
public:
    int regionsBySlashes(vector<string>& grid) {
        memset(vis, 0, sizeof(vis));
        this->grid = grid;
        
        n = grid.size();
        m = grid[0].size();

        int comp = 0;
        for (int i = 0; i < grid.size(); ++i) {
            for (int j = 0; j < grid[0].size(); ++j) {
                for (int k = 0; k < 2; ++k) {
                    if (!vis[i][j][k]) {
                        dfs(i, j, k);
                        ++comp;
                    }
                }
            }
        }
        return comp;
    }
    
    void dfs(int x, int y, int half) {
        if (vis[x][y][half]) return;
        vis[x][y][half] = true;
        if (half == 0) {
            if (x - 1 >= 0) {
                dfs(x - 1, y, 1);
            }
            if (grid[x][y] != '\\' && y - 1 >= 0) {
                dfs(x, y - 1, grid[x][y - 1] == '/' ? 1 : 0);
            }
            if (grid[x][y] != '/' && y + 1 < m) {
                dfs(x, y + 1, grid[x][y + 1] == '/' ? 0 : 1);
            }
        }
        if (half == 1) {
            if (x + 1 < n) {
                dfs(x + 1, y, 0);
            }
            if (grid[x][y] != '/' && y - 1 >= 0) {
                dfs(x, y - 1, grid[x][y - 1] == '/' ? 1 : 0);
            }
            if (grid[x][y] != '\\' && y + 1 < m) {
                dfs(x, y + 1, grid[x][y + 1] == '/' ? 0 : 1);
            }
        }
        if (grid[x][y] == ' ') {
            dfs(x, y, half ^ 1);
        }
    }
    
    int n, m;
    vector<string> grid;
    bool vis[50][50][2];
};
