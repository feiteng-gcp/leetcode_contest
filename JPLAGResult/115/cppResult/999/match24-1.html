class Solution {
public:
  
  struct UnionFind{
    int n;
    vector<int> r,p;
    UnionFind(){}
    UnionFind(int sz):n(sz),r(sz,1),p(sz,0){iota(p.begin(),p.end(),0);}
    int find(int x){
      return (x==p[x]?x:p[x]=find(p[x]));
    }
    bool same(int x,int y){
      return find(x)==find(y);
    }
    void unite(int x,int y){
      x=find(x);y=find(y);
      if(x==y) return;
      if(r[x]<r[y]) swap(x,y);
      r[x]+=r[y];
      p[y]=x;
    }
  };

  int regionsBySlashes(vector<string>& grid) {
    auto s=grid;
    int h=s.size();
    int w=s[0].size();
    UnionFind uf(h*w*4);

    auto idx=[&](int i,int j,int k){return i*w*4+j*4+k;};    
    for(int i=0;i<h;i++){
      for(int j=0;j<w;j++){
        if(s[i][j]=='/'){
          uf.unite(idx(i,j,0),idx(i,j,3));
          uf.unite(idx(i,j,1),idx(i,j,2));
        }
        if(s[i][j]=='\\'){
          uf.unite(idx(i,j,0),idx(i,j,1));
          uf.unite(idx(i,j,2),idx(i,j,3));
        }
        if(s[i][j]==' '){          
          uf.unite(idx(i,j,0),idx(i,j,3));
          uf.unite(idx(i,j,1),idx(i,j,2));
          uf.unite(idx(i,j,0),idx(i,j,1));
          uf.unite(idx(i,j,2),idx(i,j,3));
        }
        if(i+1<h){          
          uf.unite(idx(i,j,2),idx(i+1,j,0));
        }
        if(j+1<w){          
          uf.unite(idx(i,j,1),idx(i,j+1,3));
        }
      }
    }
    int ans=0;
    for(int i=0;i<h*w*4;i++){
      if(uf.find(i)==i) ans++;
    }
    return ans;
  }
};
