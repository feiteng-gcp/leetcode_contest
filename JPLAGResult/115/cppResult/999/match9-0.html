const int maxn = 30;
struct Cell {
    int u, r, d, l;
};
Cell c[maxn][maxn];
vector<int> edge[4 * maxn * maxn];
bool vis[4 * maxn * maxn];
class Solution {
public:
    int regionsBySlashes(vector<string>& grid) {
        int n = grid.size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                c[i][j].u = 4 * (i * n + j);
                c[i][j].d = 4 * (i * n + j) + 1;
                c[i][j].l = 4 * (i * n + j) + 2;
                c[i][j].r = 4 * (i * n + j) + 3;
            }
        }
        
        for (int i = 0; i < 4 * n * n; i++) {
            edge[i].clear();
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i - 1 >= 0) {
                    edge[c[i][j].u].push_back(c[i - 1][j].d);
                }
                if (i + 1 < n) {
                    edge[c[i][j].d].push_back(c[i + 1][j].u);
                }
                if (j - 1 >= 0) {
                    edge[c[i][j].l].push_back(c[i][j - 1].r);
                }
                if (j + 1 < n) {
                    edge[c[i][j].r].push_back(c[i][j + 1].l);
                }
                if (grid[i][j] != '/') {
                    edge[c[i][j].u].push_back(c[i][j].r);
                    edge[c[i][j].r].push_back(c[i][j].u);
                    edge[c[i][j].l].push_back(c[i][j].d);
                    edge[c[i][j].d].push_back(c[i][j].l);
                }
                if (grid[i][j] != '\\') {
                    edge[c[i][j].u].push_back(c[i][j].l);
                    edge[c[i][j].l].push_back(c[i][j].u);
                    edge[c[i][j].r].push_back(c[i][j].d);
                    edge[c[i][j].d].push_back(c[i][j].r);
                }
            }
        }
        
        int cnt = 0;
        memset(vis, false, sizeof(vis));
        for (int i = 0; i < 4 * n * n; i++) {
            if (!vis[i]) {
                queue<int> q;
                q.push(i);
                vis[i] = true;
                while (!q.empty()) {
                    int u = q.front(); q.pop();
                    for (int v: edge[u]) {
                        if (!vis[v]) {
                            q.push(v);
                            vis[v] = true;
                        }
                    }
                }
                cnt++;
            }
        }
        return cnt; 
    }
};
