class Solution {
private:
    int N;
    int diffColors;
    vector<string>* g;
    vector<vector<vector<int>>> allCells;
    
    vector<tuple<int,int,int>> getNeigh(int i, int j, int k)
    {
        vector<tuple<int,int,int>> res;
        vector<string> grid(*g);
        
        if (grid[i][j] == ' ')
        {
            for (int l = 0; l < 4; ++l) if (l != k) res.push_back({i,j,l});
        }
        else if (grid[i][j] == '/')
        {
            if (k == 0) res.push_back({i,j,1});
            if (k == 1) res.push_back({i,j,0});
            if (k == 2) res.push_back({i,j,3});
            if (k == 3) res.push_back({i,j,2});
        }
        else if (grid[i][j] == '\\') 
        {
            if (k == 0) res.push_back({i,j,3});
            if (k == 1) res.push_back({i,j,2});
            if (k == 2) res.push_back({i,j,1});
            if (k == 3) res.push_back({i,j,0});
        }
        if (k == 0 && j > 0) res.push_back({i, j - 1, 2});
        if (k == 2 && j < N - 1) res.push_back({i, j + 1, 0});
        if (k == 1 && i > 0) res.push_back({i - 1,j , 3});
        if (k == 3 && i < N - 1) res.push_back({i + 1, j, 1});
        return res;
    }
    void goGo(int i, int j, int k)
    {
        allCells[i][j][k] = diffColors;
        
        for (auto n :  getNeigh(i, j, k))
        {
            int ii = get<0>(n);
            int jj = get<1>(n);
            int kk = get<2>(n);
            if (allCells[ii][jj][kk] == -1)
            {
                goGo(ii, jj, kk);
            }
        }
    }
public:
    int regionsBySlashes(vector<string>& grid) {
        N = grid.size();
        allCells = vector<vector<vector<int>>>(N, vector<vector<int>>(N, vector<int>(4, -1))); 
        
        g = &grid;
        diffColors = 0;
        
        for (int i = 0; i < N; ++i)
        {
            for (int j = 0; j < N; ++j)
            {
                for (int k = 0; k < 4; ++k)
                {
                    if (allCells[i][j][k] == -1)
                    {
                        diffColors++;
                        goGo(i, j, k);
                    }
                }
            }
        }
        return diffColors;
    }
};
