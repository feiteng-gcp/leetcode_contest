class Solution {
    typedef vector<int> vi;
    typedef vector<string> vs;
    typedef vector<vector<int>> vii;
    vii visited;
    vs grid;
    int n;
    void dfs(int i, int j) {
        //cout << i << ' ' << j << endl;
        if(i < 0 || j < 0 || i >= n + 1 || j >= n + 1 || visited[i][j])
            return;
        visited[i][j] = 1;
        dfs(i - 1, j);
        dfs(i + 1, j);
        dfs(i, j - 1);
        dfs(i, j + 1);
        if(i > 0 && j > 0 && grid[i - 1][j - 1] != '/')
            dfs(i - 1, j - 1);
        if(i < n && j < n && grid[i][j] != '/')
            dfs(i + 1, j + 1);
        
        if(i > 0 && j < n && grid[i - 1][j] != '\\')
            dfs(i - 1, j + 1);
        if(i < n && j > 0 && grid[i][j - 1] != '\\')
            dfs(i + 1, j - 1);
    }
    void get_grid(vs & _grid) {
        int m = _grid.size();
        grid = vs(2 * m, string(2 * m, ' '));
        //cout << m << endl;
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < m; j++) {
                if(_grid[i][j] == '/') {
                    grid[2 * i][2 * j + 1] = '/';
                    grid[2 * i + 1][2 * j] = '/';
                }
                if(_grid[i][j] == '\\') {
                    grid[2 * i][2 * j] = '\\';
                    grid[2 * i + 1][2 * j + 1] = '\\';
                }
            }
        }
    }
public:
    int regionsBySlashes(vector<string>& input) {

        get_grid(input);
        n = grid.size();
        
        //cout << n << endl;
        visited = vii(n + 1, vi(n + 1, 0));

        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                if(grid[i][j] == '/') {
                    visited[i][j + 1] = 1;
                    visited[i + 1][j] = 1;
                }
                if(grid[i][j] == '\\') {
                    visited[i][j] = 1;
                    visited[i + 1][j + 1] = 1;
                }
            }
        }
        
        int res = 0;
        for(int i = 0; i <= n; i++) {
            for(int j = 0; j <= n; j++) {
                //cout << i << '-' << j << endl;
                if(!visited[i][j]) {
                    //cout << i << j << endl;
                    res++;
                    dfs(i, j);
                }
            }
        }
        return res;
    }
};
