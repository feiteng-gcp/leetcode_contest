class Solution {
public:
    int n;
    vector<string> g;
    bool h[30][30][4];
    void dfs(int r, int c, int d) {
        h[r][c][d] = true;
        if (d == 0 && r + 1 < n && !h[r + 1][c][2])
            dfs(r + 1, c, 2);
        if (d == 1 && c > 0 && !h[r][c - 1][3])
            dfs(r, c - 1, 3);
        if (d == 2 && r > 0 && !h[r - 1][c][0])
            dfs(r - 1, c, 0);
        if (d == 3 && c + 1 < n && !h[r][c + 1][1])
            dfs(r, c + 1, 1);
        if (g[r][c] == ' ') {
            for (int i = 0; i < 4; i++)
                if (!h[r][c][i])
                    dfs(r, c, i);
        } else if (g[r][c] == '/') {
            if (!h[r][c][3 - d])
                dfs(r, c, 3 - d);
        } else {
            if (!h[r][c][d / 2 * 2 + (1 - d % 2)])
                dfs(r, c, d / 2 * 2 + (1 - d % 2));
        }
    }
    int regionsBySlashes(vector<string>& grid) {
        g = grid;
        n = g.size();
        int ans = 0;
        for (int r = 0; r < n; r++)
            for (int c = 0; c < n; c++)
                for (int d = 0; d < 4; d++)
                    if (!h[r][c][d]) {
                        dfs(r, c, d);
                        ans++;
                    }
        return ans;
    }
};
