class Solution {
    int[] par, cnt;
    int find(int i) {
        if (par[i] != i) {
            par[i] = find(par[i]);
        }
        return par[i];
    }
    boolean union(int i, int j) {
        int pi = find(i);
        int pj = find(j);
        if (pi == pj) return false;
        par[pi] = par[pj];
        cnt[pj] += cnt[pi];
        return true;
    }
    public int regionsBySlashes(String[] grid) {
        int n = grid.length;
        par = new int[n*n*4];
        cnt = new int[n*n*4];
        for (int i = 0; i < n*n*4; i++) {
            cnt[i] = 1;
            par[i] = i;
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0, k = 0; j < n; j++, k++) {
                int b = i*n*4 + j*4;
                if (grid[i].charAt(k) == ' ') {
                    union(b, b+1); union(b+1, b+2); union(b+2, b+3);
                } else if (grid[i].charAt(k) == '/') {
                    union(b, b+1); union(b+2, b+3);
                } else {
                    union(b, b+2); union(b+1, b+3);
                    //k++;
                }
                if (j != n-1) union(b+3, b+4); 
                if (i != n-1) union(b+2, b+4*n+1);
            }
        }
        //System.out.println(Arrays.toString(par));
        //System.out.println(Arrays.toString(cnt));
        int res = 0;
        boolean[] count = new boolean[4*n*n];
        for (int i = 0; i < n*n*4; i++) count[find(i)] = true;
        for (boolean b: count) if (b) res++;
        return res;
    }
}
