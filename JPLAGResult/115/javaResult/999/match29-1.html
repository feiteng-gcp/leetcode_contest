class Solution {

	private boolean[][][][] graph = new boolean[40][40][4][4];
	private boolean[][][] vis = new boolean[40][40][4];
	
    public int regionsBySlashes(String[] grid) {
        
    	int i , j , k , n = grid.length , m = grid[0].length();
    	for (i = 0;i < n;i ++) {
    		for (j = 0;j < m;j ++) {
    			graph[i][j][0][1] = true;
    			graph[i][j][0][3] = true;
    			graph[i][j][1][0] = true;
    			graph[i][j][1][2] = true;
    			graph[i][j][2][1] = true;
    			graph[i][j][2][3] = true;
    			graph[i][j][3][0] = true;
    			graph[i][j][3][2] = true;
    		}
    	}
    	for (i = 0;i < n;i ++) {
    		for (j = 0;j < m;j ++) {
    			if (grid[i].charAt(j) == '/') {    				
    				graph[i][j][0][1] = false;
    				graph[i][j][1][0] = false;
    				graph[i][j][3][2] = false;
    				graph[i][j][2][3] = false;
    			} else if (grid[i].charAt(j) == '\\') {
    				graph[i][j][0][3] = false;
    				graph[i][j][3][0] = fals
e;
    				graph[i][j][1][2] = false;
    				graph[i][j][2][1] = false;
    			}
    		}
    	}
    	
    	int ans = 0;
    	for (i = 0;i < n;i ++) {
    		for (j = 0;j < m;j ++) {
    			for (k = 0;k < 4;k ++) {
    				if (!vis[i][j][k]) {
    					dfs(i , j , k , n , m);
    					ans ++;
    				}
    			}
    		}

    	}
    	return ans;
    	
    }
	
    private void dfs(int x , int y , int index , int n , int m) {
    	
    	vis[x][y][index] = true;
    	// internal traverse
    	for (int i = 0;i < 4;i ++) {
    		if (graph[x][y][index][i] && !vis[x][y][i]) {
    			dfs(x , y , i , n , m);
    		}
    	}
    	// outer traverse
    	if (index == 0) {
    		if (x - 1 >= 0 && !vis[x - 1][y][2]) {    			
    			dfs(x - 1 , y , 2 , n , m);
    		}
    	} else if (index == 1) {
    		if (y + 1 < m && !vis[x][y + 1][3]) {
    			dfs(x , y + 1 , 3 , n , m);
    		}
    	} else if (index == 2) {    		
    		if (x + 1 < n && !vis[x + 1][y][0]) {
    			dfs(x + 1 , y , 0 , n , m);
    		}
    	} else {
    		if (y - 1 >= 0 && !vis[x][y - 1][1]) {
    			dfs(x , y - 1 , 1 , n , m);
    		}
    	}
    	
    }
	
}
