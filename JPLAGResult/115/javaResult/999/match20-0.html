class Solution {
    String[] gs;
    int n;
    public void fill(int x, int y, boolean[][] mark) {
        if (x < 0 || x >= n || y < 0 || y >= n) return;
        if (mark[x][y]) return;
        mark[x][y] = true;
        int i = x/2, j = y/2;
        char c = gs[i].charAt(j);
        if (x%2 == 0 && y%2 == 0) {
            // 1
            fill (x + 1, y - 1, mark);
            if (c != '/') fill (x + 1, y, mark);
            if (c != '\\') fill (x, y + 1, mark);
        } else if (x%2 != 0 && y%2 == 0) {
            // 2
            fill (x + 1, y + 1, mark);
            if (c != '/') fill (x - 1, y, mark);
            if (c != '\\') fill (x, y + 1, mark);
        } else if (x%2 == 0 && y%2 != 0) {
            // 3
            fill (x - 1, y - 1, mark);
            if (c != '\\') fill (x, y - 1, mark);
            if (c != '/') fill (x + 1, y, mark);
        } else {
            // 4
            fill (x - 1, y + 1, mark);
            if (c != '\\') fill (x, y - 1, mark);
            if (c != '/') fill (x - 1, y, mark);
        }
        
    }
    public int regionsBySlashes(String[] grid) {
        n = grid.length*2;
        gs = grid;
        boolean[][] mark = new boolean[grid.length *2][grid.length * 2];
        int count = 0;
        for (int i = 0;i < n;i++) {
            for (int j = 0;j < n;j++) {
                if (!mark[i][j]) {
                    count++;
                    fill(i, j, mark);
                }
            }
        }
        return count;
    }
}
