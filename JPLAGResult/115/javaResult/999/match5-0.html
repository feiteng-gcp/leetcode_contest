class Solution {
    public int regionsBySlashes(String[] grid) {
            int len = grid.length;
            int[][] matrix = new int[len * 3][len * 3];
            for (int i = 0; i < grid.length; i++) {
                for (int j = 0; j < grid[i].length(); j++) {
                    char ch = grid[i].charAt(j);
                    int row = i * 3;
                    int col = j * 3;
                    if (ch == '\\') {
                        matrix[row][col] = 1;
                        matrix[row + 1][col + 1] = 1;
                        matrix[row + 2][col + 2] = 1;
                    } else if (ch == '/') {
                        matrix[row][col + 2] = 1;
                        matrix[row + 1][col + 1] = 1;
                        matrix[row + 2][col] = 1;
                    }
                }
            }
            int counter = 0;
            for (int i = 0; i < matrix.length; i++) {
                for (int j = 0; j < matrix.length; j++) {
                    if (matrix[i][j] == 0) {
                        counter++;
                        dfs(matrix, len * 3, i, j);
                    }
                }
            }
            return counter;
        }

        void dfs(int[][] matrix, int size, int i, int j) {

            if (i < 0 || i > size - 1 || j < 0 || j > size - 1 || matrix[i][j] == 1) {
                return;
            }
            matrix[i][j] = 1;
            dfs(matrix, size, i + 1, j);
            dfs(matrix, size, i - 1, j);
            dfs(matrix, size, i, j + 1);
            dfs(matrix, size, i, j - 1);
        }
}
