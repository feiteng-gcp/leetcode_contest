class Solution {
    int dim;
    public int regionsBySlashes(String[] grid) {
        int n = grid.length;
        List<Set<Integer>> graph = new ArrayList<>();

        dim = 2 * n + 1;
        for (int i = 0; i < dim * dim; i++) {
            graph.add(new HashSet<>());
        }

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int midr = 2 * i + 1;
                int midc = 2 * j + 1;
                connect(midr - 1, midc, midr, midc - 1, graph); //1 - 2
                connect(midr, midc - 1, midr + 1, midc, graph); //2 - 3
                connect(midr + 1, midc, midr, midc + 1, graph); //3 - 4
                connect(midr, midc + 1, midr - 1, midc, graph); //4 - 1
            }
        }

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                char c = grid[i].charAt(j);
                int midr = 2 * i + 1;
                int midc = 2 * j + 1;
                if (c == '/') {
                    disconnect(midr, midc - 1, midr + 1, midc, graph); //2 - 3
                    disconnect(midr, midc + 1, midr - 1, midc, graph); //4 - 1
                }
                else if (c == '\\') {
                    disconnect(midr - 1, midc, midr, midc - 1, graph); //1 - 2
                    disconnect(midr + 1, midc, midr, midc + 1, graph); //3 - 4
                }
            }
        }


        return findComp(graph);
    }

    void disconnect(int r1, int c1, int r2, int c2, List<Set<Integer>> graph) {
        int id1 = dim * r1 + c1;
        int id2 = dim * r2 + c2;
        graph.get(id1).remove(id2);
        graph.get(id2).remove(id1);
    }

    void connect(int r1, int c1, int r2, int c2, List<Set<Integer>> graph) {
        int id1 = dim * r1 + c1;
        int id2 = dim * r2 + c2;
        graph.get(id1).add(id2);
        graph.get(id2).add(id1);
    }

    int findComp(List<Set<Integer>> graph) {
        int dim = graph.size();
        int res = 0;
        boolean[] vis = new boolean[dim];
        for (int i = 0; i < dim; i++) {
            if (i % 2 == 0) {
                continue;
            }
            if (!vis[i]) {
                dfs(graph, vis, i);
                res++;
            }
        
}

        return res;
    }

    void dfs(List<Set<Integer>> graph, boolean[] vis, int cur) {
        if (cur % 2 == 0) {
            return;
        }

        if (vis[cur]) {
            return;
        }

        vis[cur] = true;
        for (int nxt : graph.get(cur)) {
            dfs(graph, vis, nxt);
        }
    }
}
