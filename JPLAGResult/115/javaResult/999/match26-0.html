class Solution {
    class Node implements Comparable<Node>{
		public int key;
		public Node left, right, up, down;

		public Node(int key) {
			this.key = key;
		}

		@Override
		public int compareTo(Node o) {
			return Integer.compare(this.key, o.key);
		}
	}

	public static int[] offsetX = new int[]{0,1,-1,0};
	public static int[] offsetY = new int[]{1,0,0,-1};

	public int regionsBySlashes(String[] grid) {
		int n = grid.length;
		if (n == 0) return 0;
		Map<Integer, Node> nodeMap = new HashMap<>();

		for(int i = 0; i < n; i++) {
			for(int j = 0; j < n; j++) {
				Node node = new Node(n * i + j);
				nodeMap.put(n * i + j, node);
			}
		}
		for(int i = 0; i < n; i++) {
			for(int j = 0; j < n; j++) {
				Node node = nodeMap.get(n * i + j);
				if (i > 0) {
					node.up = nodeMap.get(n * (i - 1) + j);
				}
				if (i < n - 1) {
					node.down = nodeMap.get(n * (i + 1) + j);
				}
				if (j > 0) {
					node.left = nodeMap.get(n * i + j - 1);
				}
				if (j < n - 1) {
					node.right = nodeMap.get(n * i + j + 1);
				}
			}
		}
		int top = 2 * n * n;
		for(int i = 0; i < n; i++) {
			for(int j = 0; j< n; j++) {
				char ch = grid[i].charAt(j);
				int parent = n * i + j;
				Node parentNode = nodeMap.get(parent);
				if (ch == '/') { //tach dinh trai phai
					nodeMap.remove(parent);
					top++;
					//left
					Node node1 = new Node(top);
					node1.up = parentNode.up;
					node1.left = parentNode.left;
					nodeMap.put(top, node1);
					//right
					top++;
					Node node2 = new Node(top);
					node2.right = parentNode.right;
					node2.down = parentNode.down;
					nodeMap.put(top, node2);

					for(int key: nodeMap.keySet()) {
						Node node = nodeMap.get(key);
						if (node.down == parentNode) node.down = node1;
						if (node.right == parentNode) node.right = node1;
						if (node.left == parentNode) node.left = node2;
						if (node.up == parentNode) node.up = node2;
					}
				} else if (ch == '\\') {
					nodeMap.remove(parent);
					top++;
					//left
					Node node1 = new Node(top);
					node1.left = parentNode.left;
					node1.down = parentNode.down;
					nodeMap.put(top, node1);
					//right
					top++;
					Node node2 = new Node(top);
					node2.up = parentNode.up;
					node2.right =parentNode.right;
					nodeMap.put(top, node2);

					for(int key: nodeMap.keySet()) {
						Node node = nodeMap.get(key);
						if (node.right == parentNode) node.right = node1;
						if (node.up == parentNode) node.up = node1;
						if (node.down == parentNode) node.down = node2;
						if (node.left == parentNode) node.left = node2;
					}

				}
			}
		}
		//bfs
		int result = 0;
		while(!nodeMap.isEmpty()) {
			List<Integer> keyList = new ArrayList<>(nodeMap.keySet());
			Node first = nodeMap.get(keyList.get(0));
			result++;
			dfs(first, nodeMap);
		}
		return result;
	}

	private void dfs(Node first, Map<Integer, Node> nodeMap) {
		if (!nodeMap.containsKey(first.key)) return;;
		nodeMap.remove(first.key);
		if (first.left != null) {
			dfs(first.left, nodeMap);
		}
		if (first.up != null) {
			dfs(first.up, nodeMap);
		}

		if (first.down != null) {
			dfs(first.down, nodeMap);
		}

		if (first.right != null) {
			dfs(first.right, nodeMap);
		}

	}
}
