class Solution {
    public int regionsBySlashes(String[] grid) {
        int m = grid.length;
        int n = grid[0].length();
        char[][] g = new char[m][0];
        for(int i = 0; i < m; ++i) {
            g[i] = grid[i].toCharArray();
        }
        boolean vis[][][] = new boolean[m][n][4];
        int res = 0;
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                for(int k = 0; k < 4; ++k) {
                    if (vis[i][j][k] == false) {
                        ++res;
                        dfs(g, vis, i, j, k);
                    }
                }
            }
        }
        return res;
    }
    public void dfs(char[][] grid, boolean vis[][][], int i, int j, int k) {
        int m = grid.length;
        int n = grid[0].length;
        if (k == 0) {
            if (vis[i][j][1] == false && grid[i][j] != '/') {
                vis[i][j][1] = true;
                dfs(grid, vis, i, j, 1);
            } 
            if (i > 0 && vis[i-1][j][2] == false) {
                vis[i-1][j][2] = true;
                dfs(grid, vis, i-1, j, 2);
            } 
            if (vis[i][j][3] == false && grid[i][j] != '\\') {
                vis[i][j][3] = true;
                dfs(grid, vis, i, j, 3);
            }
        } else if (k == 1) {
            if (vis[i][j][2] == false && grid[i][j] != '\\') {
                vis[i][j][2] = true;
                dfs(grid, vis, i, j, 2);
            } 
            if (j < n-1 && vis[i][j+1][3] == false) {
                vis[i][j+1][3] = true;
                dfs(grid, vis, i, j+1, 3);
            } 
            if (vis[i][j][0] == false && grid[i][j] != '/') {
                vis[i][j][0] = true;
                dfs(grid, vis, i, j, 0);
            }
        } else if (k == 2) {
            if (vis[i][j][3] == false && grid[i][j] != '/') {
                vis[i][j][3] = true;
                dfs(grid, vis, i, j, 3);
            } 
            if (i < m - 1 && vis[i+1][j][0] == false) {
                vis[i+1][j][0] = true;
                dfs(grid, vis, i+1, j, 0);
            } 
            if (vis[i][j][1] == false && grid[i][j] != '\\') {
                vis[i][j][1] = true;
                dfs(grid, vis, i, j, 1);
            }
        } else if (k == 3) {
            if (vis[i][j][0] == false && grid[i][j] != '\\') {
                vis[i][j][0] = true;
                dfs(grid, vis, i, j, 0);
            } 
            if (j > 0 && vis[i][j-1][1] == false) {
                vis[i][j-1][1] = true;
                dfs(grid, vis, i, j-1, 1);
            } 
            if (vis[i][j][2] == false && grid[i][j] != '/') {
                vis[i][j][2] = true;
                dfs(grid, vis, i, j, 2);
            }
        }
    }
}
