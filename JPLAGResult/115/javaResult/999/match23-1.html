class Solution {
    public int regionsBySlashes(String[] grid) {
        int n = grid.length;
        if (n == 0) {
            return 0;
        }
        int m = grid[0].length();
        DUS dus = new DUS(n * m * 4);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int key = (i * m + j) * 4;
                char ch = grid[i].charAt(j);
                if (ch == ' ') {
                    dus.merge(key + 0, key + 1);
                    dus.merge(key + 0, key + 2);
                    dus.merge(key + 0, key + 3);
                } else if (ch == '/') 
{
                    dus.merge(key + 0, key + 3);
                    dus.merge(key + 1, key + 2);
                } else {
                    dus.merge(key + 0, key + 1);
                    dus.merge(key + 2, key + 3);
                }
            }
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = 1; j < m; j++) {
                int key = (i * m + j) * 4;
                dus.merge(key - 4 + 1, key + 3);
            }
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int key = (i * m + j) * 4;
                dus.merge(key - 4 * m + 2, key);
            }
        }
        return dus.getBlock();
    }
}

class DUS {
    int[] f;
    int block;
    
    public DUS(int n) {
        f = new int[n];
        block = n;
        for (int i = 0; i < n; i++) {
            f[i] = i;
        }
    }
    
    
public int getF(int u) {
        if (f[u] != u) {
            f[u] = getF(f[u]);
        }
        return f[u];
    }
    
    public void merge(int u, int v) {
        u = getF(u);
        v = getF(v);
        if (u != v) {
            f[v] = u;
            block--;
        }
    }
    
    public int getBlock() {
        return block;
    }
}
