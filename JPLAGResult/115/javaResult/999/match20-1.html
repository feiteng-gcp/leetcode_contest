class Solution {
    static final int TOP = 0, LEFT = 1, RIGHT = 2, BOTTOM = 3;

    public int regionsBySlashes(String[] grid) {
        int len = grid.length;
        int[][][] g = new int[len][len][4];
        int n = 0;
        for (int i = 0; i < len; i++) {
            
for (int j = 0; j < len; j++) {
                for (int k = 0; k < 4; k++) {
                    if (g[i][j][k] == 0) {
                        mark(grid, g, i, j, k, ++n);
                    }
                }
            }
        }
        return n;
    }

    private void mark(String[] grid, int[][][] g, int i, int j, int k, int n) {
        int len = g.length;
        if (i < 0 || i >= len || j < 0 || j >= len || g[i][j][k] != 0) return;
        g[i][j][k] = n;
        char c = grid[i].charAt(j);
        if (c == '/') {
            if (k == LEFT || k == TOP) {
                mark(grid, g, i, j, LEFT, n);
                mark(grid, g, i, j, TOP, n);
            } else {
                mark(grid, g, i, j, BOTTOM, n);
                mark(grid, g, i, j, RIGHT, n);
            }
        } else if (c == '\\') {
            if (k == LEFT || k == BOTTOM) {
                mark(grid, g, i, j, LEFT, n);
                mark(grid, g, i, j, BOTTOM, n);
            } else {
                mark(grid, g, i, j, TOP, n);
                mark(grid, g, i, j, RIGHT, n);
            }
        } else {
            mark(grid, g, i, j, TOP, n);
            mark(grid, g, i, j, LEFT, n);
            mark(grid, g, i, j, RIGHT, n);
            mark(grid, g, i, j, BOTTOM, n);
        }

        if (k == LEFT) {
            mark(grid, g, i, j - 1, RIGHT, n);
        } else if (k == RIGHT) {
            mark(grid, g, i, j + 1, LEFT, n);
        } else if (k == BOTTOM) {
            mark(grid, g, i + 1, j, TOP, n);
        } else if (k == TOP) {
            mark(grid, g, i - 1, j, BOTTOM, n);
        }
    }
}
