import java.util.*;
import java.util.stream.*;
import java.math.*;

public class Solution {

    Map<String, String> parent = new HashMap<>();

    String find(String i) {
        parent.putIfAbsent(i, i);

        if (!parent.get(i).equals(i)) {
            parent.put(i, find(parent.get(i)));
        }
        return parent.get(i);
    }

    void union(String i, String j) {
        parent.put(find(i), find(j));
    }

    public int regionsBySlashes(String[] grid) {
        parent = new HashMap<>();

        int r = grid.length;
        int c = grid[0].length();

        char[][] g = new char[r][];
        for (int i = 0; i < r; i++) {
            g[i] = grid[i].toCharArray();
        }

        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                String up = key(i, j, 'u');
                String down = key(i, j, 'd');
                String left = key(i, j, 'l');
                String right = key(i, j, 'r');

                if (g[i][j] == ' ') {
                    union(up, down);
                    union(up, left);
                    union(up, right);
                } else if (g[i][j] == '/') {
                    union(left, up);
                    union(right, down);
                } else if (g[i][j] == '\\') {
                    union(right, up);
                    union(left, down);
                }

                if (j + 1 < c) {
                    union(right, key(i, j + 1, 'l'));
                }
                if (i + 1 < r) {
                    union(down, key(i + 1, j, 'u'));
                }
            }
        }

        Set<String> res = new HashSet<>();
        for (String s : parent.keySet()) {
            res.add(find(s));
        }
        return res.size();
    }

    String key(int i, int j, char dir) {
        return i + "," + j + "," + dir;
    }


    public static void main(String[] args) {
        Solution sol = new Solution();
        System.out.println(sol.regionsBySlashes(new String[]{
                " /",
                "/ ",
        }));
        System.out.println(sol.regionsBySlashes(new String[]{
                " /",
                "  ",
        }));
        System.out.println(sol.regionsBySlashes(new String[]{
                "\\/",
                "/\\",
        }));
        System.out.println(sol.regionsBySlashes(new String[]{
                "/\\",
                "\\/",
        }));
        System.out.println(sol.regionsBySlashes(new String[]{
                "//",
                "/ ",
        }));
        System.out.println(sol.regionsBySlashes(new String[]{
                "  ",
                "  ",
        }));
    }
}
