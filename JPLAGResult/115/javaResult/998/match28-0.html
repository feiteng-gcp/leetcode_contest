/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCompleteTree(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        int index = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            boolean hasEmptyRight = false;
            for (int i = 0; i < size; ++i) {
                TreeNode node = queue.poll();
                if (node.left != null) {
                    if (hasEmptyRight) return false;
                    queue.add(node.left);
                    if (node.right != null) {
                        queue.add(node.right);
                    } else {
                        hasEmptyRight = true;
                    }
                } else {
                    hasEmptyRight = true;
                    if (node.right != null) {
                        return false;
                    }
                }
            }
            if (size == Math.pow(2, index)) {
                index++;
            } else {
                if (queue.size() == 0) {
                    return true;
                } else {
                    return false;
                }
            }
        }
        return true;
    }
}
