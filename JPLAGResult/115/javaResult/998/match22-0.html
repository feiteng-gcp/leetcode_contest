/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCompleteTree(TreeNode root) {
        Deque<TreeNode> deque = new ArrayDeque<>();
        
        deque.addLast(root);
        boolean changed = false;
        while(!deque.isEmpty()){
            Deque<TreeNode> tmp = new ArrayDeque<>();
            for(TreeNode t: deque){
                if(t.left != null){
                    tmp.addLast(t.left);
                    if(changed)
                        return false;
                }
                else
                    changed = true;
                if(t.right != null){
                    tmp.addLast(t.right);
                    if(changed)
                        return false;
                }
                else
                    changed = true;
            }
            deque = tmp;
        }
        return true;
    }
}
