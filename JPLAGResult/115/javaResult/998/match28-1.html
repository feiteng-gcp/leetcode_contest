/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public boolean isCompleteTree(TreeNode root) {
        LinkedList<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        boolean over = false;
        while (!queue.isEmpty()) {
            TreeNode node = queue.removeFirst();
            if (over && (node.left != null || node.right != null)) {
                return false;
            } else {
                if (node.left != null) {
                    
queue.add(node.left);
                    if (node.right != null) {
                        queue.add(node.right);
                    } else {
                        over = true;
                    }
                } else {
                    over = true;
                    if (node.right != null) {
                        return false;
                    }
                }
            }
        }
        return true;
    }
}
