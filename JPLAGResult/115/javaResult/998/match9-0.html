/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCompleteTree(TreeNode root) {
        int num = cN(root);
        return getAns(root, 0, num);
    }
    
    public boolean getAns(TreeNode root, int index, int num){
        if(root == null) return true;
        if(index >= num) return false;
        return getAns(root.left, 2 * index + 1, num) && getAns(root.right, 2 * index + 2, num);
    }
    
    public int cN(TreeNode root){
        if(root == null) return 0;
        return 1 + cN(root.left) + cN(root.right);
    }
}
