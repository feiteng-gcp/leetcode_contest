/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCompleteTree(TreeNode root) {
        if (root == null) {
            return true;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        boolean flag = false;
        queue.offer(root);
        
while (!queue.isEmpty()) {
            TreeNode node = queue.poll();
            if (node.left != null) {
                if (flag) {
                    return false;
                }
                queue.offer(node.left);
            } else {
                flag = true;
            }
            if (node.right != null) {
                if (flag) {
                    return false;
                }
                
queue.offer(node.right);
            } else {
                flag = true;
            }
        }
        return true;
    }
}
