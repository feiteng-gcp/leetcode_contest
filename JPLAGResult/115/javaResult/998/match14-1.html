/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCompleteTree(TreeNode root) {
        Deque<TreeNode> q = new LinkedList<>();
        q.offer(root);
        
boolean missing = false;
        while (!q.isEmpty()) {
            TreeNode node = q.poll();
            if (node.left != null) {
                if (missing) {
                    return false;
                }
                q.offer(node.left);
            } else {
                missing = true;
            }
            if (node.right != null) {
                if (missing) {
                    return false;
                }
                
q.offer(node.right);
            } else {
                missing = true;
            }
        }
        return true;
    }
}
