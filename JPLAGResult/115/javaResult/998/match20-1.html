/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCompleteTree(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<TreeNode>();
        if (root == null) return true;
        q.add(root);
        boolean isConnect = true;
        while(q.isEmpty() == false && isConnect) {
            int size = q.size();
            isConnect = true;
            for(int i = 0; i < size; ++i) {
                TreeNode t = q.poll();
                
if (t.left != null) {
                    if (isConnect == false) {
                        return false;
                    }
                    q.add(t.left);
                } else {
                    isConnect = false;
                }
                if (t.right != null) {
                    if (isConnect == false) {
                        return false;
                    }
                    q.add(t.right);
                } else {
                    isConnect = false;
                }
            }
        }
        
int size = q.size();
        for(int i = 0; i < size; ++i) {
            TreeNode t = q.poll();
            if (t.left != null || t.right != null) {
                return false;
            }
        }
        return true;
    }
}
