/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCompleteTree(TreeNode root) {
        if (root == null)
            return true;
        if (root.left == null && root.right == null)
            return true;
        List<TreeNode> queue = new ArrayList<TreeNode>();
        queue.add(root);
        int currentLevelNodes = 1;
        boolean flag = false;
        while (!queue.isEmpty()) {
            int size = queue.size();
            if (size != currentLevelNodes)
                flag = true;
            if (!flag) {
                boolean levelFlag = false;
                for (int i = 0; i < size; i++) {
                    TreeNode node = queue.remove(0);
                    TreeNode leftNode = node.left;
                    TreeNode rightNode = node.right;
                    if (leftNode == null && rightNode != null)
                        return false;
                    if (levelFlag && (leftNode != null || rightNode != null))
                        return false;
                    if (!levelFlag && i < size - 1 && (leftNode == null || rightNode == null))
                        levelFlag = true;
                    if (leftNode != null)
                        queue.add(leftNode);
                    if (rightNode != null)
                        queue.add(rightNode);
                }
            } else {
                for (int i = 0; i < size; i++) {
                    TreeNode node = queue.remove(0);
                    TreeNode leftNode = node.left;
                    TreeNode rightNode = node.right;
                    if (leftNode != null || rightNode != null)
                        return false;
                }
            }
            currentLevelNodes *= 2;
        }
        return true;
    }
}
