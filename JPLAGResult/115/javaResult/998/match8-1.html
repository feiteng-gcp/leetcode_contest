/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public boolean isCompleteTree(TreeNode root) {
        if (root == null) return true;
        Queue<TreeNode> queue = new LinkedList<>();
        // the flag will be set true when a non full treenode is seen
        boolean flag = false;
        queue.add(root);
        
while (!queue.isEmpty()) {
            TreeNode temp = queue.poll();
            if (temp.left != null) {
                if (flag) return false;
                queue.add(temp.left);
            } else flag = true;
            if (temp.right != null) {
                if (flag) return false;
                queue.add(temp.right);
            } else flag = true;
        }
        return true;
    }
}
