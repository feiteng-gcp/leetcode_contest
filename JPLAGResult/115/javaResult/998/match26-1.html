/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCompleteTree(TreeNode root) {
        int n = count(root);
        return 
helper(root, 0, n);
    }
    
    private int count(TreeNode root) {
        if (root == null) {
            return 0;
        }
        
return 1 + count(root.left) + count(root.right);
    }
    
    private boolean helper(TreeNode root, int i, int n) {
        if (root == null) {
            return true;
        }
        if (root.left != null && 2 * i + 1 >= n || !helper(root.left, 2 * i + 1, n)) {
            return false;
        }
        if (root.right != null && 2 * i + 2 >= n || !helper(root.right, 2 * i + 2, n)) {
            return false;
        }
        return true;
    }
}
