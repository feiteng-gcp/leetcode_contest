/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public boolean isCompleteTree(TreeNode root) {
        List<TreeNode> list1 = new ArrayList<>();
        
        list1.add(root);
        boolean flag = true;
        while (!list1.isEmpty()) {
            List<TreeNode> list2 = new ArrayList<>();
            
            
            
            for (TreeNode node : list1) {
                
if (node.left == null) {
                    flag = false;
                } else {
                    if (!flag) {
                        return false;
                    }
                    list2.add(node.left);
                
}

                if (node.right == null) {
                    flag = false;
                } else {
                    if (!flag) {
                        return false;
                    }
                    list2.add(node.right);
                }
            }
            
            list1 = list2;
        }
        
        return true;
    }
}
