/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCompleteTree(TreeNode root) {
        if(root == null)
            return false;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        boolean flag = false;
        while(!q.isEmpty()) {
            TreeNode cur = q.poll();
            if(cur.left != null) {
                if(flag)
                    return false;
                q.offer(cur.left);
            }
            else
                flag = true;
            if(cur.right != null) {
                if(flag)
                    return false;
                q.offer(cur.right);
            }
            else
                flag = true;
        }
        return true;
    }
}
