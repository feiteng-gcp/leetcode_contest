/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public boolean isCompleteTree(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        boolean c = false;
        while (q.size() > 0) {
            int t = q.size();
            boolean flg = false;
            for (int i = 0; i < t; i++) {
                TreeNode cur = q.poll();
                if (cur != null && c) return false;
                if (cur != null && flg) return false;
                if (cur == null) {
                    flg = true;
                    c = true;
                }
                else {
                    
q.offer(cur.left);
                    q.offer(cur.right);
                }
            }
        }
        return true;
        
    }
    
}
