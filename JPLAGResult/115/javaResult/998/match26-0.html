/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isCompleteTree(TreeNode root) {
        return isCompleteTree(root, 0, numNodes(root));
    }
    
    private boolean isCompleteTree(TreeNode root, int index, int numNodes) {
        if (root == null) {
            return true;
        }
        if (index >= numNodes) {
            return false;
        }
        return isCompleteTree(root.left, 2 * index + 1, numNodes) && isCompleteTree(root.right, 2 * index + 2, numNodes);
    }
    
    private int numNodes(TreeNode root) {
        if (root == null) {
            return 0;
        }
        return 1 + numNodes(root.left) + numNodes(root.right);
    }
    
}
