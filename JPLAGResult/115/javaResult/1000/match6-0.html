class Solution {
    public int minDeletionSize(String[] A) {
        int n = A[0].length();
        int m = A.length;
        int dp[] = new int[n];
        dp[0] = 0;
        for(int i = 1; i < n; ++i) {
            dp[i] = i;
            for(int j = 0; j < i; ++j) {
                if(compare(A, j, i, m)) {
                    dp[i] = Math.min(dp[i], dp[j] + i - j - 1);
                }
            }
        }
        int res = n;
        for(int i = 0; i < n; ++i) {
            //System.out.print(dp[i] + " ");
            res = Math.min(res, n-1-i+dp[i]);
        }
        //System.out.println();
        return res;
    }
    private boolean compare(String[] A, int x, int y, int m) {
        for(int i = 0; i < m; ++i) {
            if (A[i].charAt(x) > A[i].charAt(y)) {
                return false;
            }
        }
        return true;
    }
}
