class Solution {
    public int minDeletionSize(String[] A) {
        int n = A.length;
        if (n == 0) {
            return 0;
        }
        int m = A[0].length();
        int[] ans = new int[m];
        ans[0] = 0;
        int ret = 0;
        for (int i = m - 1; i >= 0; i--) {
            int max = 0;
            
for (int j = i + 1; j <= m - 1; j++) {
                boolean small = true;
                for (int k = 0; k < n; k++) {
                    if (A[k].charAt(i) > A[k].charAt(j)) {
                        small = false;
                        break;
                    
}
                }
                if (small) {
                    max = Math.max(max, ans[j]);
                }
            }
            ans[i] = max + 1;
            ret = Math.max(ret, ans[i]);
        }
        return m - ret;
    }
}
