class Solution {
    public int minDeletionSize(String[] A) {
        
        int n = A[0].length();
        int[] f = new int[n];
        f[0] = 1;
        for (int i = 1; i < n; i++) {
            f[i] = 1;
            for (int j = 0; j < i; j++) {
                
                if (isValid(A, j, i)) f[i] = Math.max(f[i], f[j]+1);
            }
        }
        
        int max = 1;
        for (int a : f) {
            //System.out.print(a+" ");
            max = Math.max(a, max);
        }
        return n - max;
    }
    
    public boolean isValid(String[] A, int i, int j) {
        for (String a : A) {
            if (a.charAt(i) > a.charAt(j)) return false;
        }
        return true;
    }
}
