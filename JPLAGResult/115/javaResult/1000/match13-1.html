class Solution {
    public int minDeletionSize(String[] A) {
        int m = A.length;
        int n = A[0].length();
        
        int[] memo = new int[n];
        memo[0] = 1;
        int max = 1;
        
for (int i = 1; i < n; i++) {
            memo[i] = 1;
            for (int j = 0; j < i; j++) {
                boolean flag = true;
                
                for (int k = 0; k < m; k++) {
                    if (A[k].charAt(j) > A[k].charAt(i)) {
                        flag = false;
                        break;
                    }
                }
                
                if (flag) {
                    memo[i] = Math.max(memo[i], memo[j] + 1);
                    max = Math.max(memo[i], max);
                }
                
            }
        }
        
        return n - max;
    }
}
