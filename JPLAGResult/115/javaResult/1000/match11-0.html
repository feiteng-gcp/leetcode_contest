import java.util.*;
import java.util.stream.*;
import java.math.*;

public class Solution {

    public int minDeletionSize(String[] A) {
        int max = Integer.MAX_VALUE / 3;

        int M = A[0].length();
        int N = A.length;
        int ans = M - 1;
        for (int i = 0; i < N; i++) {
            A[i] = " " + A[i];
        }

        int[] f = new int[M + 1];
        f[1] = 0;
        for (int i = 1; i <= M; i++) {
            f[i] = i - 1;

            for (int j = 1; j < i; j++) {
                boolean ok = true;
                for (int k = 0; k < N; k++) {
                    if (A[k].charAt(j) > A[k].charAt(i)) {
                        ok = false;
                    }
                }
                if (ok) {
//                    System.out.println(i + " " + j);
                    f[i] = Math.min(f[i], f[j] + i - j - 1);
                }
            }
        }
//        System.out.println(Arrays.toString(f));

        for (int i = M; i >= 1; i--) {
            ans = Math.min(ans, f[i] + M - i);
        }

        return ans;
    }

    public static void main(String[] args) {
        Solution sol = new Solution();
        System.out.println(sol.minDeletionSize(new String[]{
                "babca", "bbazb",
        }));
        System.out.println(sol.minDeletionSize(new String[]{
                "edcba",
        }));
        System.out.println(sol.minDeletionSize(new String[]{
                "ghi",
                "def",
                "abc",
        }));
        System.out.println(sol.minDeletionSize(new String[]{
                "a",
        }));
    }
}
