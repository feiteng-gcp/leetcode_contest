class Solution {

	private boolean check(String[] A , int col1 , int col2) {
		
		int i , n = A.length;
		for (i = 0;i < n;i ++) {

			if (A[i].charAt(col1) > A[i].charAt(col2)) {
				return false;				
			}
		}
		return true;
		
	}
	
	private int[] dp;
	
    public int minDeletionSize(String[] A) {
        
    	int i , j , n = A.length , m = A[0].length() , ans = - 1;
    	dp = new int[m];
    	for (i = 0;i < m;i ++) {
    		dp[i] = i;
    		for (j = 0;j < i;j ++) {
    			if (check(A , j , i)) {
    				int temp = dp[j] + i - j - 1;
    				if (temp < dp[i]) {
    					dp[i] = temp;
    				}
    			}
    		}
    		int value = dp[i] + m - i - 1;
    		if (value < ans || ans < 0) {
    			ans = value;
    		}
    	}	
    	return ans;
    	
    }
}
