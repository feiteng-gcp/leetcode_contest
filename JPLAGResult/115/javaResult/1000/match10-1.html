class Solution {
    public int minDeletionSize(String[] A) {
        boolean[][] large = new boolean[A[0].length()][A[0].length()];
        for(int i = 0; i < A[0].length(); i++) {
            for(int j = i + 1; j <A[0].length(); j++) 
{
                boolean larger = true;
                for(int k = 0; k <A.length; k++) {
                    if(A[k].charAt(i) > A[k].charAt(j)) {
                        larger = false;
                        break;
                    }
                }
                large[i][j] = larger;
            }
        }
        int[] longest = new int[A[0].length()];
        longest[0] = 1;
        for(int i = 1; i < longest.length ;i++) {
            longest[i] = 1;
            for(int j = 0; j < i;j++) {
                if(large[j][i]) longest[i] = Math.max(longest[i], longest[j] + 1);
            }
        }
        int temp = 1;
        for(int i: longest) temp = Math.max(i, temp);
        return A[0].length() - temp;
    }
}
