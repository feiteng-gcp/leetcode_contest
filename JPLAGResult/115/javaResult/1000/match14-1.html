	class Solution {
	    public int minDeletionSize(String[] A) {
	        int n = A[0].length();
	        int[] dp = new int[n];
	        int ret = 0;
	        for(int i = 0;i < n;i++){
	        	dp[i] = 1;
	        	inner:
	        	for(int j = 0;
j < i;j++){
	        		for(int k = 0;k < A.length;k++){
	        			if(A[k].charAt(j) > A[k].charAt(i)){
	        				continue inner;
	        			}
	        		}
	        		dp[i] = Math.max(dp[i], dp[j] + 1);
	        	}
	        	ret = Math.max(ret, dp[i]);
	        }
	        return n-ret;
	    }
	}
