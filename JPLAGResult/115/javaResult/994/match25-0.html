class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        List<String> seen = new ArrayList<>();
        seen.add(conv(cells));
        int[] nxtState = cells;
        int matchIdx = -1;
        while (true) {
            nxtState = getNext(nxtState);
            matchIdx = idx(seen, conv(nxtState));
            if (matchIdx > -1) {
                break;
            } else {
                seen.add(conv(nxtState));
            }
        }
        //System.out.println(seen.size());
        //System.out.println(seen);
        //System.out.println(matchIdx);
        int mod = seen.size() - matchIdx;
        int idx = (N - matchIdx) % mod;
        //System.out.println("mod  = " + mod);
        //System.out.println("idx = " + idx);
        return inv(seen.get(idx + matchIdx));
    }
    
    int[] inv(String s) {
        int[] a = new int[s.length()];
        for (int i = 0; i < a.length; i++) {
            a[i] = s.charAt(i) - '0';
        }
        return a;
    }
    
    int idx(List<String> seen, String nxt) {
        for (int i = 0; i < seen.size(); i++) {
            if (seen.get(i).equals(nxt)) {
                return i;
            }
        }
        return -1;
    }
    
    String conv(int[] a) {
        String s = "";
        for (int x : a) {
            s += x;
        }
        return s;
    }
    
    int[] getNext(int[] a) {
        int[] ret = new int[a.length];
        for (int i = 1; i < a.length - 1; i++) {
            if ((a[i - 1] ^ a[i + 1]) == 0) {
                ret[i] = 1;
            }
        }
        return ret;
    }
}
