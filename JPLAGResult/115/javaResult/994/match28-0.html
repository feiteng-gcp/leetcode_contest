class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        int[] result = new int[8];
        Map<String, Integer> map = new HashMap<>();
        Map<Integer, String> table = new HashMap<>();
        for (int n = 0; n < N; n++) {
            for (int i = 1; i < 7; i++) {
                if (cells[i - 1] == cells[i + 1]) {
                    result[i] = 1;
                } else {
                    result[i] = 0;
                }
            }
            for (int i = 0; i < 8; i++) {
                cells[i] = result[i];
            }
            String s = toString(result);
            if (!map.containsKey(s)) {
                int index = map.size();
                map.put(s, index);
                table.put(index, s);
            } else {
                break;
            }
        }
        return toArray(table.get((N - 1) % table.size()));
    }
    private String toString(int[] arr) {
        StringBuilder sb = new StringBuilder();
        for (int num : arr) {
            sb.append(num);
        }
        return sb.toString();
    }
    private int[] toArray(String s) {
        int[] arr = new int[8];
        for (int i = 0; i < s.length(); i++) {
            arr[i] = s.charAt(i) - '0';
        }
        return arr;
    }
}
