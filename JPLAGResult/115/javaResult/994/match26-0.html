class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        int size = 1 << 8;
        int[] order = new int[size+1];
        Arrays.fill(order, -1);
        int k = getKey(cells);
        order[k] = 0;
        int tc[] = cells;
        int round = 0;
        for(int i = 1; i <= N; ++i) {
            tc = change(tc);
            k = getKey(tc);
            if (order[k] != -1) {
                round = (i - order[k]);
                int n = (N - i) % round;
                for(int j = 0; j < n; ++j) {
                    tc = change(tc);
                }
                break;
            } else {
                order[k] = i;
            }
        }
        return tc;
    }
    public int[] change(int[] input) {
        int n = input.length;
        int output[] = new int[n];
        for(int i = 1; i < n-1; ++i) {
            if (input[i-1] == input[i+1]) {
                output[i] = 1;
            } else {
                output[i] = 0;
            }
        }
        return output;
    }
    public int getKey(int[] input) {
        int res = 0;
        for(int i = 0; i < input.length; ++i) {
            res = (res << 1) + input[i];
        }
        return res;
    }
}
