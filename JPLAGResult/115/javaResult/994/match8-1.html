class Solution {
    
    private static final int MAX_LEN = 1 << 8;
    
    private int sequenceToNum(int[] cells) {
        int res = 0;
        for (int i = 0;i < cells.length;i++) {
            res += cells[i] * (1 << (7 - i));
        
}
        return res;
    }
    
    private int[] generateNewCells(int[] cells) {
        int[] newCells = new int[cells.length];
        newCells[0] = newCells[newCells.length - 1] = 0;
        for (int i = 1;i < cells.length - 1;i++) {
            if ((cells[i - 1] ^ cells [i + 1]) == 0) {
                newCells[i] = 1;
            } else {
                newCells[i] = 0;
            }
        }
        return newCells;
    }
    
    public int[] prisonAfterNDays(int[] cells, int N) {
        Map<Integer, Integer> numToIndex = new HashMap<>();
        List<int[]> sequence = new ArrayList<>(MAX_LEN);
        int loopStart = -1;
        int[] nowCells = cells;
        for (int i = 0;i < MAX_LEN;i++) {
            int num = sequenceToNum(nowCells);
            if (!numToIndex.containsKey(num)) {
                numToIndex.put(num, i);
                sequence.add(nowCells);
                nowCells = generateNewCells(nowCells);
            } else {
                loopStart = numToIndex.get(num);
                break;
            }
        }
        if (loopStart == -1) {
            throw new IllegalStateException("???");
        }
        
        int loopSize = sequence.size() - loopStart;
        if (N < loopStart) {
            return sequence.get(N);
        } else {
            return sequence.get(loopStart + ((N - loopStart) % loopSize));
        }
    }
}
