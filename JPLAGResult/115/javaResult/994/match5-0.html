class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        //int[] a = {0,1,0,1,1,0,0,1};
        int[] copy = Arrays.copyOf(cells, 8);
        int[] b = new int[8];
        Set<String> set = new HashSet<String>();
        String str = "";
        while(true){
            //set.add(str);
            for(int j = 1; j < 7; j++){
                if((cells[j - 1] == 1 && cells[j + 1] == 1) || (cells[j - 1] == 0 && cells[j + 1] == 0)) b[j] = 1;
                else b[j] = 0;
            }
            cells = Arrays.copyOf(b, 8);
            str = Arrays.toString(cells);
            if(set.contains(str)) break;
            else set.add(str);
        }
        int temp = set.size();
        N = N % temp;
        //if(N < temp)
        N += temp;
        cells = copy;
        while(N > 0){
            N--;
            for(int j = 1; j < 7; j++){
                if((cells[j - 1] == 1 && cells[j + 1] == 1) || (cells[j - 1] == 0 && cells[j + 1] == 0)) b[j] = 1;
                else b[j] = 0;
            }
            cells = Arrays.copyOf(b, 8);
        }
        return cells;
    }
}
