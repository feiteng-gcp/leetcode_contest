class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {

        if(N > 14) {
            N = N % 14;
            if(N == 0) {
                N = 14;
            }
        }

        int[] cache = new int[cells.length];
        cache = Arrays.copyOf(cells, cells.length);
        for (int i = 0; i < N; i++) {
            cache = change(cache);
        }
        return cache;
    }

    private int[] change(int[] cells) {
        int[] res = new int[cells.length];
        res[0] = 0;
        res[cells.length - 1] = 0;
        for (int i = 1; i < cells.length - 1; i++) {
            if (cells[i - 1] == cells[i + 1]) {
                res[i] = 1;
            } else {
                res[i] = 0;
            }
        }
        return res;
    }
}
