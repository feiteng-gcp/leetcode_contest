class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        int cnt = 0;
        HashSet<Integer> set = new HashSet<>();
        int ic = 0;
        while (!set.contains(ic = toInt(cells)) && N > 0) {
            set.add(ic);
            cnt++;
            cells = calc(cells);
            N--;
        }
        if (N == 0) {
            return cells;
        }
        set.clear();
        cnt=0;
        while (!set.contains(ic = toInt(cells)) && N > 0) {
            set.add(ic);
            cnt++;
            cells = calc(cells);
            N--;
        }
        if (N == 0) {
            return cells;
        }

        N %= cnt;
        while (N-- > 0) {
            cells = calc(cells);
        }
        return cells;
    }

    int[] calc(int[] c)
    {
        int ret[] = new int[8];
        for (int i = 1; i < 7; i++) {
            if ((c[i - 1] == 1 && c[i + 1] == 1)
                    || (c[i - 1] == 0 && c[i + 1] == 0)) {
                ret[i] = 1;
            }
        }
        return ret;
    }

    int toInt(int[] c)
    {
        int ret = 0;
        for (int v : c) {
            ret *= 2;
            ret += v;
        }
        return ret;
    }
}
