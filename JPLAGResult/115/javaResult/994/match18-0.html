class Solution {
    //   1 1 0
    //   _ 0 _
    public int[] prisonAfterNDays(int[] cells, int N) {
        int n = cells.length;
        int[] arr = helper(cells);
        N--;
        int[] origin = new int[n];
        for(int i = 0; i < n; i++) {
            origin[i] = arr[i];
        }
        
        for(int i = 1; i <= N; i++) {
            int[] next = helper(arr);
            arr = next;
            

            if(same(origin, arr)) {
                int pattern = i;

                N = N % pattern;
                i = 0;
            }
        }
        
        return arr;
    }
    
    private int[] helper(int[] arr) {
        int n = arr.length;
        int[] next = new int[n];
        for(int i = 1; i < n - 1; i++) {
            if( arr[i - 1] == arr[i + 1]) {
                next[i] = 1;
            }
        }
        
        return next;
    } 
    
    private boolean same(int[] arr1, int[] arr2) {
        int n = arr1.length;
        for(int i = 0; i < n; i++) {
            if(arr1[i] != arr2[i])  {
                return false;
            }
        }
        
        return true;
    }
}
