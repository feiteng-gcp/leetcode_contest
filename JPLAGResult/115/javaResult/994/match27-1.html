class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        int n = cells.length;
        int[] res = Arrays.copyOf(cells, n);
        Map<Integer, Integer> hm = new HashMap<>();
        int iter = 0;
        int prev = -1;
        int diff = -1;
        while (true) {
            int val = findInt(res);
            if (!hm.containsKey(val)) {
                hm.put(val, iter);
            }
            else {
                prev = hm.get(val);
                diff = iter - prev;
                break;
            }

            iter++;
            res = findNext(res);
        }

        res = Arrays.copyOf(cells, n);
        if (N <= prev) {
            for (int i = 0; i < N; i++) {
                res = findNext(res);
            }
        }
        else {
            N -= prev;
            N %= diff;
            int total = prev + N;
            for (int i = 0; i < total; i++) {
                res = findNext(res);
            }
        }

        
return res;
    }

    int[] findNext(int[] input) {
        int n = input.length;
        int[] res = new int[n];
        for (int i = 1; i < n - 1; i++) {
            if (input[i - 1] == 1 && input[i + 1] == 1) {
                res[i] = 1;
            }
            else if (input[i - 1] == 0 && input[i + 1] == 0) {
                res[i] = 1;
            
}
        }

        return res;
    }

    int findInt(int[] arr) {
        int res = 0;
        for (int i = 0; i < arr.length; i++) {
            res = res * 2 + arr[i];
        }

        return res;
    }
}
