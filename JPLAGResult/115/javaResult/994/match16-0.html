public class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        List<Configuration> history = new ArrayList<>(64);

        Configuration curr = new Configuration(cells);

        while (!history.contains(curr)) {
            history.add(curr);
            curr = new Configuration(transition(curr.cells));
        }
        int cycleLength = history.size() - history.indexOf(curr);
        
        int reducedN = N % cycleLength + cycleLength * 5;
        int[] newCells = cells;
        for (int i = 0; i < reducedN; i++) {
            newCells = transition(newCells);
        }
        return newCells;
    }

    private int[] transition(int[] cells) {
        int[] newCells = new int[cells.length];
        for (int i = 1; i < newCells.length - 1; i++) {
            if (cells[i-1] == cells[i+1]) {
                newCells[i] = 1;
            }
        }
        return newCells;
    }

    class Configuration {
        int[] cells;

        public Configuration(int[] cells) {
            this.cells = cells;
        }

        @Override
        public boolean equals(Object obj) {
            if (obj instanceof Configuration) {
                return Arrays.equals(((Configuration) obj).cells, this.cells);
            }
            return false;
        }

        @Override
        public int hashCode() {
            return Arrays.hashCode(cells);
        }
    }
}
