class Solution {
    Map<String, Integer> map = new HashMap<>();
    public int[] prisonAfterNDays(int[] cells, int N) {
        map = new HashMap<>();
        map.put(enCode(cells), 0);
        int period = 0;
        int[]endingState = new int[8];
        int firstTime = 0;
        for (int i = 1; i <= Math.min(N, 255); i++) {
            cells = generateNext(cells);
            String currentCode = enCode(cells);
            if (map.containsKey(currentCode)) {
                int lastTime = map.get(currentCode);
                firstTime = lastTime;
                period = i - lastTime;
                endingState = cells;
                break;
            }
            else {
                map.put(currentCode, i);
                endingState = cells;
            }
        }

        if (period == 0) return endingState;
        int remaining = (N - firstTime) % period;

        for (int i = 0; i < remaining; i++) {
            endingState = generateNext(endingState);
        }

        return endingState;
    }

    private String enCode(int []cells){
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 8; i++)
            sb.append(cells[i]);

        return sb.toString();
    }

    private int[] generateNext(int[]cells){
        int[]next = new int[8];
        for (int i = 0; i < 8; i++) {
            if (i == 0 || i == 7) next[0] = 0;
            else {
                int status = 1 - (cells[i-1]^cells[i+1]);
                next[i] = status;
            }
        }

        return next;
    }
}
