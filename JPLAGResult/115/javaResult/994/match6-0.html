class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        // first day
        cells = update(cells);
        
        // compute cycle (math later)
        int[] origin = Arrays.copyOf(cells, 8);
        int cycle = 0;
        for (int i=1; i<=32; i++) {
            cells = update(cells);
            boolean flag = false;
            for (int j=0; j<8; j++) {
                if (cells[j] != origin[j]) {
                    flag = true;
                    break;
                }
            }
            if (!flag) {
                cycle = i;
                break;
            }
        }
        
        N = (N-1) % cycle;
        for (int i=0; i<N; i++) {
            origin = update(origin);
        }
        
        return origin;
    }
    
    private int[] update(int[] cells) {
        int[] temp = new int[8];
        for (int j=1; j<7; j++) {
                if ((cells[j-1]==0 && cells[j+1]==0) || (cells[j-1]==1 && cells[j+1]==1)) {
                    temp[j] = 1;
                } else {
                    temp[j] = 0;
                }
        }
        return temp;
    }
}
