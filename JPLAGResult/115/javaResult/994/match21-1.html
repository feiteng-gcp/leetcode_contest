class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        List<Byte> days = new ArrayList<>();
        Map<Byte, Integer> prevStates = new HashMap<>();
        days.add(arrayToByte(cells));
        prevStates.put(days.get(0), 0);
        for (int i = 1; i <= N; i++) {
            int[] prevDay = byteToArray(days.get(days.size() - 1));
            
int[] thisDay = new int[8];
            for (int j = 1; j < 7; j++) {
                if (prevDay[j - 1] == prevDay[j + 1]) {
                    thisDay[j] = 1;
                }
            }
            byte thisDayAsByte = arrayToByte(thisDay);
            if (prevStates.containsKey(thisDayAsByte)) {
                int x = prevStates.get(thisDayAsByte);
                return byteToArray(days.get(x + ((N - i) % (i - x))));
            }
            days.add(thisDayAsByte);
            prevStates.put(thisDayAsByte, i);
        }
        return byteToArray(days.get(
days.size() - 1));
    }
    private static byte arrayToByte(int[] array) {
        byte result = 0;
        for (int i = 0; i < 8; i++) {
            result |= (array[8 - i - 1] << i);
        
}
        return result;
    }
    private static int[] byteToArray(byte b) {
        int[] result = new int[8];
        for (int i = 0; i < 8; i++) {
            int val = (b & (1 << i)) == 0 ? 0 : 1;
            result[8 - i - 1] = val;
        }
        return result;
    }
}
