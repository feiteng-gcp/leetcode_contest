import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;
import java.util.Set;

class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        int[] st = cells;
        Map<Integer, Integer> hashMap = new HashMap<>();
        int index = 0;
        hashMap.put(key(cells), index);
        int entry, length;
        while (true) {
            index++;
            int[] next = generate(cells);
            int k = key(next);
            cells = next;
            if (hashMap.containsKey(k)) {
                entry = hashMap.get(k);
                length = index - entry;
                break;
            }
            hashMap.put(k, index);
        }
        if (N > entry) {
            N = entry + (N - entry) % length;
        }
        for (int i = 0; i < N; i++) {
            st = generate(st);
        }
        return st;
    }

    public int[] generate(int[] cells) {
        int[] ans = new int[8];
        for (int i = 1; i < 7; i++) {
            ans[i] = cells[i - 1] == cells[i + 1] ? 1 : 0;
        }
        return ans;
    }

    public int key(int[] cells) {
        int ans = 0;
        for (int i = 0; i < 8; i++) {
            ans += cells[i] << i;
        }
        return ans;
    }
}
