class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        Set<Integer> set = new HashSet<>();
        List<int[]> list = new ArrayList<>();
        int i = 0;
        for (; i < N; i ++) {
            int[] newCells = nextDay(cells);
            cells = newCells;
            Integer key = convertInteger(cells);
            if (!set.contains(key)) {
                
set.add(key);
                list.add(cells);
            } else {
                break;
            }
        }
        if (i == N) {
            return cells;
        }
        int idx = (N-1) % i;
        return list.get(idx);
    
}

    private int[] nextDay(int[] cells) {
        int[] newCells = new int[8];
        for (int i = 1; i <= 6; i ++) {
            if ((cells[i-1] ^ cells[i+1]) == 0) {
                newCells[i] = 1;
            }
        }
        return newCells;
    
}

    private Integer convertInteger(int[] cells) {
        StringBuilder sb = new StringBuilder();
        for (int i: cells) {
            sb.append(i);
        }
        return Integer.valueOf(sb.toString(), 2);
    }

}
