class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
    	int n = cells.length;
    	Lis
t<int[]> circle = new ArrayList<>();
    	for (int i = 0; i < N; ++i) {
    		int[] aux = new int[n];
    		for (int j = 0; j < n; ++j) {
    			if (j == 0 || j == n - 1) aux[j] = 0;
    			else {
    				aux[j] = (cells[j - 1] ^ cells[j + 1]) == 1 ? 0 : 1;
    			}

    		}
    		if (circle.size() != 0 && Arrays.equals(aux, circle.get(0))) {
    			break;
    		}
    		circle.add(aux);
    		cells = aux;
//    		if (Arrays.equals(cells, aux)) return aux;
//    		else {
//    			cells = aux;
//    		}
    	}
    	return circle.get((N - 1) % circle.size());

    }
}
