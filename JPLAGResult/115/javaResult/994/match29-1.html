class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        int[] arr = new int[512];
        int[] pre = cells;
        int[] cur;
        arr[getSum(pre)] = 0;
        for (int i = 1; i <= N; ++i) {
            cur = new int[8];
            
for (int j = 1; j < 7; ++j) {
                if (pre[j - 1] == pre[j + 1]) {
                    cur[j] = 1;
                }
            }
            if (arr[getSum(cur)] == 0) {
                arr[getSum(cur)] = i;
            } else {
                int p = arr[getSum(cur)];
                int d = (N - i) % (i - p);
                for (int k = 0; k < 512; ++k) {
                    if (arr[k] == p + d) {
                        return getArr(k);
                    }
                }
            }
            
pre = cur;
        }
        return pre;
    }

    private int[] getArr(int n) {
        int[] result = new int[8];
        for (int i = 7; i >= 0; --i) {
            if (n >= Math.pow(2, i)) {
                result[i] = 1;
                n -= Math.pow(2, i);
            }
        }
        return result;
    
}

    private int getSum(int[] arr) {
        int result = 0;
        for (int i = 0; i < 8; ++i) {
            if (arr[i] == 1) {
                result += Math.pow(2, i);
            }
        }
        return result;
    }
}
