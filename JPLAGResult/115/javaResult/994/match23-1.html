class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < N; i++) {
            int key = getKey(cells);
            if (map.containsKey(key)) {
                int start = map.get(key);
                int m = i - start;
                int remain = (N - 1 - start) % m;
                N = start + m + remain + 1;
            }
            map.put(key, i);
            cells = getNext(cells);
        }
        
return cells;
    }
    
    private int[] getNext(int[] cells) {
        int[] next = new int[cells.length];
        for (int i = 1; i < cells.length - 1; i++) {
            if (!(cells[i - 1] == 0 ^ cells[i + 1] == 0)) {
                next[i] = 1;
            }
        }
        return next;
    }
    
    private int getKey(int[] cells) {
        int key = 0;
        for (int c : cells) {
            key = (key << 1) | c;
        }
        return key;
    }
}
