class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        N = 1 + (N - 1) % 14;
        for (int i = 0; i < N; i++) {
            // System.out.print(i + ": ");
            // int sum = 0;
            // for (int cell : cells) {
            //     sum *= 2;
            //     sum += cell;
            // }
            // System.out.println(sum);
            cells = next(cells);
        }
        return cells;
    }
    
    private int[] next(int[] cur) {
        int[] next = new int[cur.length];
        next[0] = next[cur.length - 1] = 0;
        for (int i = 1; i < cur.length - 1; i++) {
            if (cur[i - 1] == cur[i + 1]) {
                next[i] = 1;
            }
        }
        return next;
    }
}
