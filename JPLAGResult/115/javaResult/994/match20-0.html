class Solution {
    private String getString(int[] A) {
        StringBuilder sb = new StringBuilder();
        for (int a : A){
            sb.append(String.valueOf(a));
        }
        return sb.toString();
    }
    
    private int[] getNextCell(int[] cells) {
        int n = cells.length;
        int[] newcell = new int[n];
        newcell[0] = 0; newcell[n-1] = 0;
        for (int i = 1; i < n-1; i++) {
            newcell[i] = 1- Math.abs(cells[i-1]-cells[i+1]);
        }
        return newcell;
    }
    
    private int[] getIntArray(String str) {
        int n = str.length();
        int[] res = new int[n];
        for (int i=0; i<n; i++)
            res[i] = str.charAt(i)-'0';
        return res;
    }
    
    public int[] prisonAfterNDays(int[] cells, int N) {
        Map<String, Integer> map1 = new HashMap<>();
        Map<Integer, String> map2 = new HashMap<>();
        
        int[] curr = cells;
        map1.put(getString(curr), 0);
        map2.put(0, getString(curr));
        
        for (int day = 1; day <= N; day++) {
            int[] newcell = getNextCell(curr);
            String newcellStr = getString(newcell);
            if (!map1.containsKey(newcellStr)) {
                map1.put(newcellStr, day);
                map2.put(day, newcellStr);
            }
            else {
                int j = map1.get(newcellStr);
                return getIntArray(map2.get((N-j) % (day-j) +j));
            }
            curr = newcell;
        }
        
        return curr;
        
        
    }
}
