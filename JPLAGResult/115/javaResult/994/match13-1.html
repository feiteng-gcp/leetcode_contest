class Solution {
   public int[] prisonAfterNDays(int[] cells, int N) {
        Map<Integer, Integer> hasDoneAtStep = new HashMap();  // results to steps
        Map<Integer, int[]> result = new HashMap();  //steps to result
            
        int steps = 1;        
        cells = convert(cells);            
        
        while (!hasDoneAtStep.containsKey(getKey(cells))){
            hasDoneAtStep.put(getKey(cells), steps);
            result.put(steps, cells);
            
            cells = convert(cells);            
            steps ++ ;             
        }
        int repeating = steps - hasDoneAtStep.get(getKey(cells));
        
        return result.get( (N-1) % repeating + 1);
    
}
    
    public int[] convert(int[] cells){
        int[] res = new int[cells.length];
        res[0] = 0;
        res[cells.length-1] = 0;
        for (int i=1; i<cells.length-1; i++){
            if ((cells[i-1]+cells[i+1])%2 == 0){
                res[i] = 1;
            }
        }
        
        return res;
    }
    
    
public Integer getKey(int[] cells){
        int res = 0;
        for (int i: cells){
            res =  res*2 + i;
        }
        return res;
    }
}
