class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        if(N == 0) return cells;
        int[] next = new int[cells.length];
        
for(int i = 1; i < next.length-1; i++){
            if(cells[i-1] == cells[i+1]) next[i] = 1;
            else next[i] = 0;
        }
        cells = Arrays.copyOfRange(next, 0, cells.length);
        
        N--;
        N = N % 14;
        
        while(N > 0){
            Arrays.fill(next, 0);
            
for(int i = 1; i < next.length-1; i++){
                if(cells[i-1] == cells[i+1]) next[i] = 1;
                else next[i] = 0;
            }
            cells = Arrays.copyOfRange(next, 0, cells.length);
            N--;
        }
        return cells;
    }
}
