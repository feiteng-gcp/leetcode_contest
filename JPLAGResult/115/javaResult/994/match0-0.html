class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        Map<String, Integer> map1 = new HashMap<String, Integer>();
        Map<Integer, String> map2 = new HashMap<Integer, String>();
        map1.put(Arrays.toString(cells), 0);
        map2.put(0, Arrays.toString(cells));
        int periodStart = 0;
        int period = 0;
        for (int i = 1; i <= N; i++) {
            cells = prisonNextState(cells);
            if (i == N)
                return cells;
            String cellsStr = Arrays.toString(cells);
            if (map1.containsKey(cellsStr)) {
                periodStart = map1.get(cellsStr);
                period = i - periodStart;
                break;
            } else {
                map1.put(cellsStr, i);
                map2.put(i, cellsStr);
            }
        }
        int remainder = N % period;
        int periodCount = periodStart / period;
        int index = remainder;
        while (index < periodStart)
            index += period;
        String retStr = map2.get(index);
        retStr = retStr.substring(1, retStr.length() - 1);
        String[] strArray = retStr.split(", ");
        int[] ret = new int[strArray.length];
        for (int i = 0; i < strArray.length; i++)
            ret[i] = Integer.parseInt(strArray[i]);
        return ret;
    }

    public int[] prisonNextState(int[] cells) {
        int[] newCells = new int[8];
		newCells[0] = 0;
		newCells[7] = 0;
		for (int i = 1; i < 7; i++)
			newCells[i] = 1 - cells[i - 1] ^ cells[i + 1];
        return newCells;
    }
}
