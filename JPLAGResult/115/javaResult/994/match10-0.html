class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        Map<Integer,Integer> map=new HashMap<Integer,Integer>();
        List<Integer> list=new ArrayList<Integer>();
        int index=0;
        int first=convertToInt(cells);
        map.put(first,0);
        list.add(first);
        int t=-1;
        while(true){
            if(list.size()==N+1){
                return convertToArray(list.get(list.size()-1));
            }
            int[] newCell=new int[8];
            for(int i=1;i<7;i++){
                if(cells[i-1]==cells[i+1]){
                    newCell[i]=1;
                }
            }
            cells=newCell;
            int tracker=convertToInt(cells);
            if(map.containsKey(tracker)){
                t=tracker;
                break;
            }
            list.add(tracker);
            map.put(tracker,list.size()-1);
        }
        N=N-map.get(t);
        N=N%(list.size()-map.get(t));
        int target=list.get(map.get(t)+N);
        return convertToArray(target);
    }
    public int convertToInt(int[] nums){
        int res=0;
        for(int i=0;i<nums.length;i++){
            if(nums[i]==1){
                res=res|(1<<i);
            }
        }
        return res;
    }
    public int[] convertToArray(int num){
        int[] res=new int[8];
        for(int i=0;i<res.length;i++){
            if(((1<<i)&num)!=0){
                res[i]=1;
            }
        }
        return res;
    }
}
