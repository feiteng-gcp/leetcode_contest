class Solution {
    public int[] prisonAfterNDays(int[] cells, int N) {
        int[] times = new int[256];
        
        int i=1;
        int round=1;
        for(i=1;i<=N;i++){
            int[] tmp = prisonAfterOneDay(cells);
            cells = tmp;
            if(i==N){
                break;
            }
            
            int number = this.getNumber(cells);
            if(times[number]!=0){
                round=i-times[number];
                break;
            }else{
                times[number]=i;
            }
        }
        
        if(i==N){
            return cells;
        }else{
            //System.out.println(round);
            N=(N-i)%round;//继续
            
            for(i=1;i<=N;i++){
                int[] tmp = prisonAfterOneDay(cells);  
                cells = tmp;
            }
            
            return cells;
        }
    }
    
    private void display(int[] cells){
        
        for(int i=0;i<cells.length;i++){
            System.out.print(cells[i]);
            System.out.print(" ");
        }
        System.out.println();
    }
    
    
    private int getNumber(int[] cells){
        int init = 0;
        for(int i=0;i<cells.length;i++){
            init = init * 2 + cells[i];
        }
        return init;
    }
    
    private int[] prisonAfterOneDay(int[] cells){
        int[] tmp = new int[cells.length];
        tmp[0] = 0;
        tmp[7] = 0;

        for(int j=1;j<=6;j++){
            if(cells[j-1]==cells[j+1]){
                tmp[j]=1;
            }
        }
        return tmp;
    }
}
