class Solution {
    
    public int[] prisonAfterNDays(int[] cells, int n) {
        HashMap<Integer, Integer> h = new HashMap<>();
        h.put(k(cells), -1);
        for(int s = 0 ; s < n ; s++) {
            int [] next = next(cells);
            int p = k(next);
            if(h.containsKey(p)) {
                int diff = s - h.get(p);
                if(diff < n - s) {
                    int ppp = (n - s) / diff;
                    if(ppp > 0)
                    s += (ppp - 1) * diff; 
                }
                
            } else {
                h.put(p, s);
            }
            cells = next;
        }
        return cells;
    }
    int k (int [] kk) {
        int res = 0;
        int m = 1;
        for(int s : kk) {
            if(s == 1) {
                res += m;

            }
            m *= 2;
        }
        return res;
    }
    
    
    int [] next(int [] cur) {
        int [] next = new int[8];
        for(int s = 1 ; s< 7 ; s++) {
            if(cur[s - 1] == cur[s + 1]) {
                next[s] = 1;
            }
        }
        return next;
    }
}
