class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        if N == 0:
            return cells
        cs = dict()
        def f(k):
            res = [0]
            
for i in range(1,7):
                if k[i-1] == k[i+1]:
                    res.append(1)
                else:
                    res.append(0)
            res.append(0)
            return res
        kk = cells
        for i in range(N):
            tmp = f(kk)
            kk = tmp
            k = tuple(kk)
            if k in cs:
                a = cs[k]
                w = a+(N-1-i)%(i-a)
                for e in cs:
                    if cs[e] == w:
                        return list(e)
            cs[k] = i
            if i == N-1:
                return kk
