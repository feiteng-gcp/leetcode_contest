class Solution:
    def nextDay(self, cells):
        newCells 
= [0] * len(cells)
        for i in range(1, len(cells) - 1):
            if cells[i - 1] == cells[i + 1]:
                newCells[i] = 1
            else:
                newCells[i] = 0
        return newCells
    
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        idx = {}
        val = {}
        
        for j in range(0, N + 1):
            if j == N:
                return cells            
            
            s = ''.join(map(str, cells))
            if s in idx:
                offset = idx[s]
                loopLength = j - idx[s]
                break
            else:
                idx[s] = j
                val[j] = s            
            
            cells = self.nextDay(cells)
            
        N -= offset
        N %= loopLength
        return list(map(int, list(val[offset + N])))
