class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        hashs = {}
        hashs2 = {}
        cnt = 0
        for _ in range(N):
            tmp = [0]*8
            # print(cnt,cells)
            if tuple(cells) in hashs:
                t = hashs[tuple(cells)]
                # print(t)
                return hashs2[(N-cnt)%(cnt-t)+t]
            hashs[tuple(cells)] = cnt
            hashs2[cnt] 
=tuple(cells) 
            cnt+=1
            for i in range(1,len(cells)-1):
                if cells[i-1]^cells[i+1] == 0:
                    tmp[i] = 1
            for i in range(len(cells)):
                if tmp[i]:
                    cells[i] = 1
                else:
                    cells[i]=0
            
        return cells
