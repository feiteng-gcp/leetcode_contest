from collections import defaultdict
class Solution:
    def prisonAfterNDays(self, cells, N):
        s = set()
        l = []
        start = None
        p = []
        for _ in range(N):
            tuple_cell = tuple(cells)
            if tuple_cell in s:
                start = l.index(tuple_cell)
                p = l[start:]
                break
            s.add(tuple_cell)
            l.append(tuple_cell)

            _cells = [0] * 8
            
for i in range(1, 7):
                if cells[i-1] == cells[i+1]: _cells[i] = 1
                else: _cells[i] = 0
            cells = _cells

        if N > len(s): 
            k = len(p)
            return p[(N - start) % k]
        else: 
            return cells
