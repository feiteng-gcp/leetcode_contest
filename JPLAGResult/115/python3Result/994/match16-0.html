class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        def transform_cells(cells):
            new_cells = [0] * len(cells)
            
            for index in range(1, len(cells) - 1):
                if cells[index-1] == cells[index+1]:
                    new_cells[index] = 1
            
            return new_cells
        
        times = {}
        count = 0
        
        for count in range(N):
            as_str = ''.join(map(str, cells))
            
            if as_str in times:
                break
            
            cells = transform_cells(cells)
            times[as_str] = count
        else:
            return cells
        
        diff = count - times[as_str]
        count += ((N - count - 1) // diff) * diff
        
        for _ in range(count, N):
            cells = transform_cells(cells)
        
        return cells
