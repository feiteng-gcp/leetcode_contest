from copy import deepcopy
class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        
        
        first=deepcopy(cells)
        first[0]=0
        first[-1]=0
        for j in range(1,len(cells)-1):
            if cells[j-1]==cells[j+1]:
                first[j]=1
            else:
                first[j]=0
        cells=deepcopy(first)
        lis=[first]
        for i in range(1,N):
            temp=deepcopy(cells)
            for j in range(1,len(cells)-1):
                if cells[j-1]==cells[j+1]:
                    temp[j]=1
                else:
                    temp[j]=0
            if temp==first:
                break
            lis.append(temp)
            cells=deepcopy(temp)
        if N>1 and temp==first:
            cells=deepcopy(lis[N%len(lis)-1])
        return cells
