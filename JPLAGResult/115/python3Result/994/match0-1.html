class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        record = []
        for round in range(N):
            next_cells = [0] * len(cells)
            for i in range(1, len(cells)-1):
                if (cells[i-1] + cells[i+1]) in [0, 2]:
                    next_cells[i] = 1

            key = ''.join(map(str, cells))
            if key in record:
                i = record.index(key)
                rep_length = len(record) - i
                remain_len = (N - round) % rep_length
                return list(map(int, record[i+remain_len]))
            cells = next_cells
            record.append(key)

        return cells
