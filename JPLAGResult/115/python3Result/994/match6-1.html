class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        if N == 0:
            return cells
        calculate = N % 14
        if calculate == 0:
            calculate = 14
        tempCells = cells.copy()
        for i in range
(1, calculate+1): 
            for row in range(1,7):
                if cells[row-1] == cells[row+1]:
                    tempCells[row] = 1
                else:
                    tempCells[row] = 0
            tempCells[0] = 0
            tempCells[7] = 0
            cells = tempCells.copy()

        return cells
