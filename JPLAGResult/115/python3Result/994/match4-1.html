class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        def next_tick(state):
            new_state = [0]*len(state)
            
for i in range(1, len(state)-1):
                if state[i-1] + state[i+1] == 1:
                    new_state[i] = 0
                else:
                    new_state[i] = 1
            return tuple(new_state)
        seen = {tuple(cells): 0}
        tick = 0
        while True:
            new_cells = next_tick(cells)
            tick += 1
            if new_cells not in seen:
                seen[new_cells] = tick
                cells = new_cells
            else:
                period = tick - seen[new_cells]
                cells = new_cells
                break
        remainder = (N - tick) % period
        for i in range(remainder):
            cells = next_tick(cells)
        return list(cells)
