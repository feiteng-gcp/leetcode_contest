import copy
class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        # 暴力破解
        # 因为肯定会产生循环，所以我们只要找到循环节就行了
        circle = []
        array 
= [0 for _ in range(8)]
        for i in range(1, 7):
            if cells[i-1] + cells[i+1] == 1:
                array[i] = 0
            else:
                array[i] = 1
        circle.append(copy.copy(array))
        for _ in range(64):
            for i in range(1,7):
                if circle[-1][i-1] + circle[-1][i+1] == 1:
                    array[i] = 0
                else:
                    array[i] = 1
            if array == circle[0]:
                break
            circle.append(copy.copy(array))
        T = len(circle)
        N = N % T
        return circle[N-1]
