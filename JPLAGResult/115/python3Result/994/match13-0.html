class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        tm = [step(x) for x in range(2 ** 8)]
        cells = sum(cells[i] << i for i in range(8))
        while N:
            if N & 1:
                cells = tm[cells]
            tm = [tm[tm[i]] for i in range(2**8)]
            N //= 2
        return [(cells >> i) & 1 for i in range(8)]
def step(n):
    xs = [100] + [(n >> i) & 1 for i in range(8)] + [100]
    xp = [0] * 8
    for i in range(1, len(xs) - 1):
        if xs[i-1] == xs[i+1]:
            xp[i-1] = 1
    return sum(xp[i] << i for i in range(8))
