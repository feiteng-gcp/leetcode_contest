class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        table = set()
        index = []
        prev = cells
        for i in range(64):
            curr = self.trans(prev)
            curr_tuple = tuple(curr)
            if curr_tuple in table:
                break
            else:
                table.add(curr_tuple)
                index.append(curr)
                prev = curr
        return index[(N-1) % i]
        
    def trans(self, cells):
        res 
= [0] * 8
        for i, c in enumerate(cells):
            if 1 <= i <= 6 and cells[i-1] ^ cells[i+1] == 0:
                res[i] = 1
            else:
                res[i] = 0
        return res
