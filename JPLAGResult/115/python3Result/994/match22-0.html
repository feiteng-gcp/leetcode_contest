class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        visit, rec = dict(), dict()
        cnt = 1
        cells = tuple(cells)
        visit[cells] = cnt
        rec[cnt] = cells
        cnt += 1
        def trans(arr):
            new_arr = []
            for idx, bit in enumerate(arr):
                if 1 <= idx <=6 and arr[idx-1] == arr[idx+1]:
                    new_arr.append(1)
                else:
                    new_arr.append(0)
            return tuple(new_arr) 
        cycle = -1
        for t in range(N):
            cells = trans(cells)
            if cells in visit:
                first_repeat = visit[cells]
                cycle = cnt - first_repeat
                # print('st', visit[cells])
                # print('cycle', cycle)
                remainder = (N+1 - first_repeat) % cycle
                return list(rec[first_repeat + remainder])
            else:
                visit[cells] = cnt
                rec[cnt] = cells
                cnt += 1

        return list(cells)
