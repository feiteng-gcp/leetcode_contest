class Solution:
    def prisonAfterNDays(self, cells, N):
        dic={}
        dic[str(cells)]=0
        arr=[]
        arr.append(cells)
        c=0
        while True:
            c+=1
            temp=[]
            temp.append(0)
            for i in range(1,7):
                if cells[i-1]+cells[i+1]==1:
                    temp.append(0)
                else:
                    temp.append(1)
            temp.append(0)
            if c==N:
                return temp
            cells=temp
            if str(temp) in dic:
                break
            
            dic[str(cells)]=c
            arr.append(cells)
        
        na=len(arr)
        st=dic[str(cells)]
        if na==st+1:
            return cells
        #print(na,st)
        N=(N-na)%(na-st)
        return arr[st+N]
        
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
