class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        res = []
        n = N
        while N > 0:
            tmp = [c for c in cells]
            tmp[0] = 0
            for i in range(1, len(cells)-1):
                if cells[i-1] ^ cells[i+1] == 0:
                    tmp[i] = 1
                else:
                    tmp[i] = 0
            tmp[-1] = 0
            N -= 1
            cells = tmp
            if tmp in res:
                return res[N%(len(res))]
            else:
                res.append(tmp)
        return res[-1]
