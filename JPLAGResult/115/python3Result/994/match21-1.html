class Solution:
    def prisonAfterNDays(self, cells, N):
        """
        :type cells: List[int]
        :type N: int
        :rtype: List[int]
        """
        memo = {}
        dic = {}
        l = len(cells)
        
def update(cells):
            next = [0]*len(cells)
            next[0] = 0
            next[-1] = 0
            for i in range(1,len(cells)-1):
                if cells[i-1] == cells[i+1]:
                    next[i] = 1
            return next
        startkey,end = None,None
        for j in range(N):
            key = "".join([str(i) for i in cells])
            
            if key in memo:
                # print('exist')
                startkey = key
                end = j
                break
                # next = memo[key]
                
            else:
                next = update(cells)
                memo[key] = next
                dic[key] = j
            # if next == cells:
            #     break
            # print(next)
            cells = next
        # print(startkey,end,dic)
        # print(end-dic[startkey])
        if startkey:
            repeatlen = end-dic[startkey]
            idx = (N - dic[startkey] - 1)%repeatlen
            # print(idx)
            cells = memo[startkey]
            for _ in range(idx):
                key = "".join([str(i) for i in cells])
                next = memo[key]
                cells = next
            
                
            
        return cells
