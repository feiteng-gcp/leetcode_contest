# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCompleteTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        cur_level = [root]
        is_last = False
        while cur_level:
            tmp = []
            for node in cur_level:            
                if is_last:
                    if node:
                        return False                   
                else:
                    if node is None:
                        is_last = True
                    else:
                        tmp.append(node.left)
                        tmp.append(node.right)
            if is_last:
                return not any(tmp)
            else:
                cur_level = tmp
