# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCompleteTree(self, root):
        q=collections.deque()
        if not root:
            return True
        isnull=False
        q.append(root)
        while len
(q)>0:
            f=q.popleft()
            if f.left:
                if isnull:
                    return False
                q.append(f.left)
            else:
                isnull=True
            if f.right:
                if isnull:
                    return False
                q.append(f.right)
            else:
                isnull=True
        return True
        
        
        """
        :type root: TreeNode
        :rtype: bool
        """
