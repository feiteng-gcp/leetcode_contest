# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCompleteTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        maps = {}
        def count(root):
            if not root:return 0
            return 1 +count(root.left)+count(root.right)
        
        
def helper(root,idx,count):
            
            if not root: return True
            if idx >= count:return False
            return helper(root.left,2*idx+1,count) and helper(root.right,2*idx+2,count)
        
        
        
        return helper(root,0,count(root))
