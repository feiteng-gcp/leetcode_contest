# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCompleteTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        queue = []
        isComplete = False
        if root is None:
            return True
        queue.append(root)
        while len(queue) > 0:
            temp = queue.pop(0)
            
            
if temp.left:
                if isComplete == True:
                    return False
                queue.append(temp.left)
            else:
                isComplete = True
            if temp.right:
                if isComplete == True:
                    return False
                queue.append(temp.right)
            else:
                isComplete = True
        return True
