# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCompleteTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        lvl = [root]
        stage = 0
        
        while lvl:
            next_lvl = []
            missing = False
            
            for node in lvl:
                if node.left:
                    if missing:
                        return False
                    next_lvl.append(node.left)
                else:
                    missing = True
                    
                if node.right:
                    if missing:
                        return False
                    next_lvl.append(node.right)
                else:
                    missing = True
            
            if len(lvl) != 2 ** stage and next_lvl:
                return False
            
            stage += 1
            lvl = next_lvl
            
        return True
