# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def isCompleteTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        def count(root):
            if not root:
                return 0
            return 1 + count(root.left) + count(root.right)
        total = count(root)
        def dfs(root, index):
            if not root:
                return True
            if index >= total:
                return False
            left = dfs(root.left, (index << 1) + 1)
            right = dfs(root.right, (index << 1) + 2)
            return left and right
        return dfs(root, 0)
