class Solution:
    def isCompleteTree(self, root):
        nodes = list()
        nodes.append(root)
        flag = False
        while(len(nodes) > 0):
            next_nodes = list()
            for node in nodes:
                if node is None:
                    if not flag:
                        flag = True
                else:
                    if flag:
                        return False
                    next_nodes.append(node.left)
                    next_nodes.append(node.right)
            nodes = next_nodes
        return True
