# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCompleteTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        
        if root is None:
            return True
        
        q = [root]
        full = True
        while len(q) > 0:
            next_q = []
            for n in q:
                
                if not full and n:
                    return False
                
                if not n:
                    full = False
                else:
                    next_q.append(n.left)
                    next_q.append(n.right)
                            
            q = next_q
            
        return True
