# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCompleteTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        def check(root,index,countN):
            if not root:
                return True
            if index>=countN:
                return False
            return check(root.left,2*index+1,countN) and check(root.right,2*index+2,countN)
        return check(root,0,self.count(root))
    def count(self,root):
        if not root:
            return 0
        else:
            return (1+ self.count(root.left) + self.count(root.right))
