# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
from collections import deque


class Solution:
    def isCompleteTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        queue = deque()
        queue.append(root)
        is_end = False
        while len(queue) > 0:
            size = len(queue)
            for i in range(size):
                node = queue.popleft()
                if node is None:
                    is_end = True
                else:
                    if is_end:
                        return False
                    else:
                        queue.append(node.left)
                        queue.append(node.right)

        return True
