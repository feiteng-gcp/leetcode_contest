# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCompleteTree(self, root):
        if root == None: return True
        Q = [root]
        meetNone = False
        
        while Q:
            tempQ = Q
            Q = []
            for e in tempQ:
                
if meetNone and e != None: return False
                if e == None: meetNone = True
                else:
                    Q.append(e.left)
                    Q.append(e.right)
        return True
