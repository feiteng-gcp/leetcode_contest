# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isCompleteTree(self, root):
        if root is None:
            return True
        queue = []
        flag = False
        queue.append(root)
        while len(queue) > 0:
            temp_node = queue.pop(0)
            if temp_node.left:
                if flag == True:
                    return False
                queue.append(temp_node.left)
            else:
                flag = True
            
            if temp_node.right:
                if flag == True:
                    return False
                queue.append(temp_node.right)
            else:
                flag = True
        return True
