# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

from collections import deque

class Solution:
    def isCompleteTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        if not root:
            return True
        queue = deque([root]) 
        end = False
        while queue:
            level_len = len(queue)
            for i in range(level_len):
                top = queue.popleft()
                if top.left:
                    if end:
                        return False
                    queue.append(top.left)
                else:
                    end = True
                if top.right:
                    if end:
                        return False
                    queue.append(top.right)
                else:
                    end = True
        return True
