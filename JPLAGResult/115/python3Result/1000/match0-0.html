class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        m = len(A)
        n = len(A[0])
        dp = [1] * n
        for j in range(1, n):
            for k in range(j):
                compatible = True
                for i in range(m):
                    if A[i][k] > A[i][j]:
                        compatible = False
                if compatible:
                    dp[j] = max(dp[j], dp[k] + 1)
        return n - max(dp)
