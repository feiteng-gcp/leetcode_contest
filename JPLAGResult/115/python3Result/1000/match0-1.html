class Solution:
    def minDeletionSize(self, A):
        """
        :type A: List[str]
        :rtype: int
        """
        def compare(i, j):
            return all(s[i] <= s[j] for s in A)
        
        n = len(A[0])
        dp = [1 for _ in range(n)]

        for i in range(1, n):
            for j in range(0, i):
                if compare(j, i):
                    dp
[i] = max(dp[i], dp[j] + 1)
        return n - max(dp)
