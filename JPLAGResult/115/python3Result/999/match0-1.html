class Solution:
    def regionsBySlashes(self, grid):
        """
        :type grid: List[str]
        :rtype: int
        """
        m, n = len(grid), len(grid[0])
        nodes = []
        graph = {}
        def add_edge(e1, e2):
          graph.setdefault(e1, []).append(e2)
          graph.setdefault(e2, []).append(e1)

        for row in range(0, 2*m+1, 2):
          for col in range(1, 2*n, 2):
            nodes.append((row, col))
        for row in range(1, 2*m, 2):
          for col in range(0, 2*n+1, 2):
            nodes.append((row, col))
        for i in range(m):
          for j in range(n):
            top = (2*i, 2*j+1)
            bottom = (2*i+2, 2*j+1)
            left = (2*i+1, 2*j)
            right = (2*i+1, 2*j+2)
            c = grid[i][j]
            if c == ' ':
              add_edge(top, right)
              add_edge(right, bottom)
              add_edge(bottom, left)
              add_edge(left, top)
            elif c == '\\':
              add_edge(top, right)
              add_edge(bottom, left)
            else:
              add_edge(top, left)
              add_edge(bottom, right)
        #print(graph)
        regions = 0
        seen = set( )
        for node in nodes:
          if node not in seen:
            regions += 1
            seen.add(node)
            frontier = [node]
            while frontier:
              nextone = frontier.pop()
              for n in graph[nextone]:
                
if n not in seen:
                  seen.add(n)
                  frontier.append(n)
        return regions
