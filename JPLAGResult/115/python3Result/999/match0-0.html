from collections import deque

class Solution:
    def regionsBySlashes(self, grid):
        """
        :type grid: List[str]
        :rtype: int
        """
        N = len(grid)
        super_N = 2*N + 1
        graph = dict()
        
        for r in range(N):
            for c in range(N):
                sr = 2*r + 1
                sc = 2*c + 1
                
                edge_ids = [-1]*4
                
                dsr, dsc = 0, -1
                for k in range(4):
                    er, ec = sr+dsr, sc+dsc
                    if 0 <= er < super_N and 0 <= ec < super_N:
                        edge_id = er*super_N + ec
                        edge_ids[k] = edge_id
                    dsr, dsc = dsc, -dsr
                
                if grid[r][c] == ' ':
                    k_range = (0,1,2,3)
                elif grid[r][c] == '/':
                    k_range = (0, 2)
                else:
                    k_range = (1, 3)
                
                for k in k_range:
                    u_id = edge_ids[k]
                    v_id = edge_ids[(k+1)%4]
                    if u_id != -1 and v_id != -1:
                        if u_id not in graph:
                            graph[u_id] = set()
                        if v_id not in graph:
                            graph[v_id] = set()

                        graph[u_id].add(v_id)
                        graph[v_id].add(u_id)
                
        visited = set()
        component_num = 0
        
        for node, adjs in graph.items():
            if node not in visited:
                component_num += 1
                bfsq = deque()
                visited.add(node)
                bfsq.appendleft(node)
                
                while bfsq:
                    cur_node = bfsq.pop()
                    cur_adjs = graph[cur_node]
                    for adj in cur_adjs:
                        if adj not in visited:
                            bfsq.appendleft(adj)
                            visited.add(adj)
        
        return component_num
