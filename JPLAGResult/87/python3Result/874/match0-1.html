class Solution:
	def backspaceCompare(self, S, T):
		"""
		:type S: str
		:type T: str
		:rtype: bool
		"""
		def convert(s):
			import collections
			ret = collections.deque()
			
for e in s:
				if e == '#':
					if len(ret) > 0:
						ret.pop()
				else:
					ret.append(e)
			return ''.join([e for e in ret])

		return convert(S) == convert(T)
