class Solution:
    def backspaceCompare(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        s, t 
= [], []
        for i in S:
            if i == '#':
                if s:
                    s.pop()
            else:
                s.append(i)
        for i in T:
            if i == '#':
                if t:
                    t.pop()
            else:
                t.append(i)
        return ''.join(s) == ''.join(t)
