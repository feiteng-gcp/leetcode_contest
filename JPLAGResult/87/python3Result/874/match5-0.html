class Solution:
    def backspaceCompare(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        x=[]
        y=[]
        for s in S:
            if s=="#":
                if x!=[]:
                    x.pop(-1)
            else:
                x.append(s)
        for t in T:
            if t=="#":
                if y!=[]:
                    y.pop(-1)
            else:
                y.append(t)
        return x==y
