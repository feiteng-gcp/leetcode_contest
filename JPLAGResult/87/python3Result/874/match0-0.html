class Solution:
    def backspaceCompare(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        def convert(s):
            r = []
            for c in s:
                if c == '#':
                    if len(r) > 0:
                        r.pop()
                else:
                    r.append(c)
            return "".join(r)
        return convert(S) == convert(T)
