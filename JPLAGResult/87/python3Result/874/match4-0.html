class Solution:
    def backspaceCompare(self, S, T):
        """
        :type S: str
        :type T: str
        :rtype: bool
        """
        def helper(S):
            rList = []
            for ch in S:
                if ch=='#':
                    if rList:
                        rList.pop()
                else:
                    rList.append(ch)
            return ''.join(rList)

        s1 = helper(S)
        t1 = helper(T)
        print(s1, t1)
        return s1==t1
