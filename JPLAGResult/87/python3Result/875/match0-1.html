class Solution:
    def longestMountain(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        l = len(A)
        if l == 0:
            return 0
        d1 = [1]
        for i in range(1, l):
            if A[i] > A[i-1]:
                d1.append(d1[-1] + 1)
            else:
                d1.append(1)

        d2 = [0 for i in range(l)]
        d2[l-1] = 1
        for i in range(l-2, 0, -1):
            if A[i] > A[i + 1]:
                d2[i] = d2[i+1] + 1
            else:
                d2[i] = 1
        
        ans = 0
        for i in range(1, l-1):
            if d1[i] >=2 and d2[i] >=2:
                ans = max(ans, d1[i] + d2[i] - 1)
        return ans
