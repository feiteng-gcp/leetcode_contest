class Solution:
    def longestMountain(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        N = len(A)
        if N<3:
            return 0
        mList = []
        for i in range(1, N-1):
            if A[i-1]<A[i]>A[i+1]:
                mList.append(i)
        rMax = 0
        for m in mList:
            l = m-1
            while l>0 and A[l-1]<A[l]:
                l -= 1
            r = m+1
            while r<N-1 and A[r]>A[r+1]:
                r += 1
            if r-l+1>rMax:
                # print(m, l, r)
                rMax = r-l+1
        return rMax
