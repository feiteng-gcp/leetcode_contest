class Solution {
    public int longestMountain(int[] A) {
        if (A == null) {
            return 0;
        }
        int len = A.length;
        if (len < 3) {
            return 0;
        }
        int[] left = new int[len];
        int[] right = new int[len];
        left[0] = 1;
        right[len - 1] = 1;
        
        for (int i = 1; i < len; i++) {
            left[i] = A[i] > A[i - 1] ? left[i - 1] + 1 : 1;
        }
        for (int i = len - 2; i >= 0; i--) {
            right[i] = A[i] > A[i + 1] ? right[i + 1] + 1 : 1;
        }
        
        int ret = 0;
        for (int i = 0; i < len; i++) {
            if (left[i] > 1 && right[i] > 1) {
                ret = Math.max(ret, left[i] - 1 + right[i ] - 1 + 1);
            }
        }
        
        return ret;
    }
}
