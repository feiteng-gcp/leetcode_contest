import java.util.*;

class Solution {
    public int longestMountain(int[] A) {

        final int len = A.length;
        if (len < 3) { return 0; }
        int[] incRunLenLTOR = new int[len];
        int[] incRunLenRTOL = new int[len];

        incRunLenLTOR[0] = 1;
        for (int pos = 1; pos < len; pos++) {
            if (A[pos] > A[pos-1]) { incRunLenLTOR[pos] = incRunLenLTOR[pos-1]+1; }
            else { incRunLenLTOR[pos] = 1; }
        }

        incRunLenRTOL[len-1] = 1;
        for (int pos = len-2; pos >= 0; pos--) {
            if (A[pos] > A[pos+1]) { incRunLenRTOL[pos] = incRunLenRTOL[pos+1]+1; }
            else { incRunLenRTOL[pos] = 1; }
        }

        // Find longest
        int best = 0;
        for (int i = 1; i < len-1; i++) {  // any of these could be the i (peak)
            // viable peak?
            if (incRunLenLTOR[i] >= 2 && incRunLenRTOL[i] >= 2) {

                // potentially this is the best.
                int length = incRunLenLTOR[i] + incRunLenRTOL[i] - 1;
                best = Math.max(best, length);

            }
        }
            
        return best;
    }
}
