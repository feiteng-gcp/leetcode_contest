class Solution {
    public int longestMountain(int[] a) {
        int n = a.length;
        
int res = 0;
        for (int i = 1; i < n - 1;) {
            if (a[i] > a[i-1] && a[i] > a[i+1]) {
                int left = i-1;
                while (left - 1 >= 0 && a[left-1] < a[left]) left--;
                int right = i+1;
                while (right + 1 < n && a[right+1] < a[right]) right++;
                res = Math.max(res, right - left + 1);
                i = right+1;
            } else {
                i++;
            }
        }
        return res;
    }

}
