class Solution {
    public int longestMountain(int[] A) {
        if (A.length < 3) {
            return 0;
        }
        int left = 0;
        int top = -1;
        int right = -1;
        boolean goUp = false;
        
        int max = 0;
        for (int i = 1; i < A.length; i++) {
            if (A[i] == A[i - 1]) {
                if (top != -1 && right != -1 && right - left + 1 > max) {
                    max = right - left + 1;
                }
                left = i;
                top = -1;
                right = -1;
            } else if (A[i] > A[i - 1]) {
                if (right == -1) {
                    top = i;
                } else {
                    if (top != -1 && right - left + 1 > max) {
                        max = right - left + 1;
                    }
                    left = right;
                    top = i;
                    right = -1;
                }
            } else 
{
                if (top == -1) {
                    left = i;
                } else {
                    right = i;
                }
            }
        }
        if (top != -1 && right != -1) {
            if (right - left + 1 > max) {
                max = right - left + 1;
            }
        }
        return max;
    }
}
