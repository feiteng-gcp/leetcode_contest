class Solution {
    public int longestMountain(int[] A) {
        int n = A.length;
        if (n <= 2) return 0;
        
        int max = 0;
        for(int i = 1; i < n - 1; i++) {
            if (A[i] > A[i-1] && A[i] > A[i+1]) {
                int s = i - 1;
                while (s >= 0 && A[s] < A[s+1]) --s;
                int e = i + 1;
                while (e < n && A[e] < A[e - 1]) ++e;
                max = Math.max(max, e -s - 1);
            }
        }
        
        return max;
    }
}
