class Solution {
    public int longestMountain(int[] A) {
        
        int N = A.length;
        
        int[] left = new int[N];
        int[] right = new int[N];
        
        for (int i = 1; i < N; ++i) {
            if (A[i] > A[i - 1]) {
                left[i] = left[i - 1] + 1;
            }
        }
        
        for (int i = N - 2; i >= 0; --i) {
            if (A[i] > A[i + 1]) {
                right[i] = right[i + 1] + 1;
            }
        }
        
        int maxM = 0;
        
        for (int i = 1; i < N - 1; ++i) {
            if (left[i] > 0 && right[i] > 0) {
                maxM = Math.max(maxM, left[i] + 1 + right[i]);
            }
        }
        
        return maxM;
        
    }
}
