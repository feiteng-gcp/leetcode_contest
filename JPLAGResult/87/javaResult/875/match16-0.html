class Solution {
    public int longestMountain(int[] A) {
        if (A == null || A.length == 0) {
            return 0;
        }
        int len = A.length;
        int[] l2r = new int[len];
        int[] r2l = new int[len];
        l2r[0] = 1;
        for (int i = 1; i < len; i++) {
            if (A[i] > A[i - 1]) {
                l2r[i] = l2r[i - 1] + 1;
            } else {
                l2r[i] = 1;
            }
        }
        r2l[len - 1] = 1;
        for (int j = len - 2; j >= 0; j--) {
            if (A[j] > A[j + 1]) {
                r2l[j] = r2l[j + 1] + 1;
            } else {
                r2l[j] = 1;
            }
        }
        int maxVal = 0;
        for (int k = 1; k < len - 1; k++) {
            if (A[k] > A[k - 1] && A[k] > A[k + 1]) {
                maxVal = Math.max(maxVal, l2r[k] + r2l[k]);
            }
        }
        if (maxVal - 1 >= 3) return maxVal - 1;
        return 0;
    }
}
