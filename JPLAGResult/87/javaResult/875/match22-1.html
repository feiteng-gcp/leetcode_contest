class Solution {
    public int longestMountain(int[] A) {
    	if(A.length == 0) return 0;
    	int[] cArr = new int[A.length];
    	int combo = 1;
    	cArr[0] = 1;
    	int
 ans = 0;
    	for(int i = 1; i < A.length; i++) {
    		if(A[i] > A[i-1]) {
    			combo++;
    		}
    		else {
    			combo=1;
    		}
    		cArr[i] = combo;
    	}
    	
    	cArr[A.length-1] = 1;
    	combo = 1;
    	for
(int i = A.length-2; i >= 0; i--) {
    		if(A[i] > A[i+1]) {
    			combo++;
    		}
    		else {
    			combo=1;
    		}
    		
    		if(cArr[i] == 1 || combo == 1) cArr[i] = 0;
    		else cArr[i] = combo + cArr[i] - 1;
    		ans = Math.max(cArr[i], ans);
    	}
    	
    	return ans;
    }
	
}
