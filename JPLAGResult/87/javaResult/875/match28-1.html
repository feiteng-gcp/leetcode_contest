class Solution {
    public int longestMountain(int[] A) {
        int n = A.length;
        if (n<3)
            return 0;
        int[] up = new int[n];
        int[] down = new int[n];
        up[0] = 1;
        down[0] = 1;
        for (int i = 1; i < n; i++) {
            up[i] = 1;
            down[i] = 1;
            
if (A[i]>A[i-1])
                up[i] = up[i-1]+1;
        }
        for (int i = n-2; i >= 0; i--) {
            if (A[i]>A[i+1])
                down[i] = down[i+1]+1;
        }
        int max = 0;
        for (int i = 0; i < n; i++)
            if (up[i]>1&&down[i]>1)
                max = Math.max(max, up[i]+down[i]-1);
        if (max<3)
            return 0;
        return max;
    }
}
