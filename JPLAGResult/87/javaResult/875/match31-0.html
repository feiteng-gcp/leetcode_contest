class Solution {
    public int longestMountain(int[] A) {
        if (A.length == 0) return 0;
        int n = A.length, i, ans = 0;
        int[] dif = new int[n];
        for (i = 0; i < n - 1; i++) {
            if (A[i] < A[i + 1]) dif[i] = 1;
            else if (A[i] > A[i + 1]) dif[i] = -1;
            else dif[i] = 0;
        }
        int head = -1, tail = 0, downIdx = -1, goUp = -2,j;
        int res = 0;
        boolean goDown = false;
        for (i = 1; i < n - 1; i++) {
            if (A[i - 1] < A[i] && A[i] > A[i + 1]) {
                // System.out.println(i);
                res = 3;
                for (j = i - 2; j >= 0; j--) {
                    if (A[j] < A[j + 1]) {
                        res++;
                    } else break;
                }
                for (j = i + 2; j < n; j++) {
                    if (A[j - 1] > A[j]) {
                        res++;
                    } else break;
                }
                ans = Math.max(ans, res);
            }
        }
        return ans;
    }
}
