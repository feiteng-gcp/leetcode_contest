class Solution {
    int n;
    List<Integer>[] nbs;
    int[][] d;
    public int shortestPathLength(int[][] graph) {
        n = graph.length;
        if (n == 1) return 0;
        
        nbs = new List[n];
        d = new int[n][n];
        for (int i = 0; i < n; i++) {
            nbs[i] = new ArrayList<>();
        }
        
        for (int i = 0; i < n; i++) {
            for (int j : graph[i]) {
                nbs[i].add(j);
                nbs[j].add(i);
            }
        }
        int INF = 20000;
        Deque<Integer> q = new ArrayDeque<>();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) d[i][j] = INF;
            d[i][i] = 0;
            q.add(i);
            while (!q.isEmpty()){
                int cur = q.remove();
                for(int next : nbs[cur]){
                    if(d[i][next] == INF) {
                        d[i][next] = d[i][cur]+1;
                        q.add(next);
                    }
                }
            }
        }
        
        int nS = 1 << n;
        int[][] dp = new int[n][nS];
        State[] states = new State[nS - 1];
        for (int i = 1; i < nS; ++i) {
            states[i-1] = new State(i);
        }
        Arrays.sort(states);
        
        for (State state : states) {
            for (int s = 0; s < n; s++) {
                int len = INF;
                for (int next : state.t) {
                    int newState = state.v - (1 << next);
                    len = Math.min(len, dp[next][newState] + d[s][next]);
                }
                
                dp[s][state.v] = len;
            }
        }
        
        int minL = INF;
        for (int i = 0; i < n; i++){
            minL = Math.min(minL, dp[i][nS - 1]);
        }
        return minL;
    }
    
    class State implements Comparable<State>{
        int v;
        int nbit1;
        List<Integer> t = new ArrayList<>();
        public State(int v){
            this.v = v;
            nbit1 = 0;
            for (int i = 0; i < n; i++) {
                if (((v >>i) & 1) == 1){
                    t.add(i);
                    ++nbit1;
                }
            }            
            
        }
        
        public int compareTo(State o){
            return nbit1 - o.nbit1;
        }
        
        
    }
    
}
