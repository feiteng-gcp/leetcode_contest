class Solution {
    public int shortestPathLength(int[][] a) {
        int n = a.length;
        int res = 999_999_999;
        for (int i = 0; i < n; i++) {
            res = Math.min(res, go(a, i));
        }
        return res;
    }
    
    private int go(int[][] a, int start) {
        int n = a.length;
        Set<String> been = new HashSet<>();
        PriorityQueue<int[]> pq = new PriorityQueue<>(new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[2] - o2[2];
            }
        }); // 0: state, 1: where, 2: taken
        int[] init = {0, start, 0};
        init[0] |= 1 << start;
        pq.add(init);
        been.add(makeKey(init));

        int goal = 0;
        for (int i = 0; i < n; i++) goal |= 1 << i;

        while (!pq.isEmpty()) {
            int[] pop = pq.remove();
            if (pop[0] == goal) return pop[2];
            int u = pop[1];
            for (int v : a[u]) {
                int[] next = pop.clone();
                next[0] |= 1 << v;
                next[2]++;
                next[1] = v;
                if (next[0] == goal) return next[2];
                if (been.contains(makeKey(next))) continue;
                been.add(makeKey(next));
                pq.add(next);
            }
        }
        return 999_999_999;
    
}

    private String makeKey(int[] state) {
        return state[0] + "|" + state[1];
    }


}
