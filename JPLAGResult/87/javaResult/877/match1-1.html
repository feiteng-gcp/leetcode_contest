class Solution {
 
	
	private static int[][] dp = new int[(1 << 13)][13];
	private static int[][] dist = new int[15][15];
	
    public int shortestPathLength(int[][] graph) {
        	
    	int i , j , k , n = graph.length;
    	
    	for (i = 0;i < n;i ++) {
    		for (j = 0;j < n;j ++) {
    			dist[i][j] = - 1;
    		}
    	}
    	for (i = 0;i < n;i ++) {
    		dist[i][i] = 0;
    	}
    	for (i = 0;i < n;i ++) {
    		for (int v
 : graph[i]) {
    			dist[i][v] = 1;
    		}
    	}
    	
    	for (k = 0;k < n;k ++) {
    		for (i = 0;i < n;i ++) {
    			for (j = 0;j < n;j ++) {
    				if (k != i && k != j && i != j) {
    					if (dist[i][k] >= 0 && dist[k][j] >= 0) {
    						int temp = dist[i][k] + dist[k][j];
    						if (temp < dist[i][j] || dist[i][j] < 0) {
    							dist[i][j] = temp;
    						}
    					}
    				}

    			}
    		}
    	}
    	
    	int ans = - 1;
    	for (i = 0;i < n;i ++) {
    		
    		for (j = 0;j < dp.length;j ++) {
    			for (k = 0;k < dp[0].length;k ++) {
    				dp[j][k] = - 1;
    			}
    		}
    		
    		int result = dfs((1 << i) , n , i);
    		if (result < ans || ans < 0) {
    			ans = result;
    		}
    	}
    	return ans;
    	
    }
    
    private int dfs(int bitmap , int n , int last) {
    	
    	if (bitmap == (1 << n) - 1) {
    		return 0;
    	} else {
    		if (dp[bitmap][last] >= 0) {
    			return dp[bitmap][last];
    		} else {
    			int ans = - 1;
    			for (int i = 0;i < n;i ++) {
    				if ((bitmap & (1 << i)) == 0) {
    					int temp = dist[last][i] + dfs(bitmap | (1 << i) , n , i);
    					if (temp < ans || ans < 0) {
    						ans = temp;
    					}
    				}    				
    			}
    			return dp[bitmap][last] = ans;
    		}
    	}
    	
    }
    
}
