class Solution {
    
public boolean backspaceCompare(String S, String T) {
        return gao(S).equals(gao(T));
    }
    
String gao(String s) {
        StringBuilder sb = new StringBuilder();
        for (char ch : s.toCharArray()) {
            
if (ch == '#') {
                if (sb.length() != 0) sb.deleteCharAt(sb.length() - 1);
            }
            else {
                sb.append(ch);
            }
        }
        return sb.toString();
    }
}
