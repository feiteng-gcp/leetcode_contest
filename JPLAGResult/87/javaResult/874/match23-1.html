class Solution {
    public boolean backspaceCompare(String S, String T) {
        return getFinal(S).equals(getFinal(T));
    }
    
    String getFinal(String s) {
        Stack<Character> cs = new Stack<>();
        for (char c : s.toCharArray()) {
            if (c == '#') {
                if (!cs.isEmpty()) {
                    cs.pop();
                
}
            } else {
                cs.push(c);
            }
        }
        
        String res = "";
        while (!cs.isEmpty()) {
            res = cs.pop() + res;
        }
        return res;
    }
}
