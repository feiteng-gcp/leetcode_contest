class Solution {
    public boolean backspaceCompare(String S, String T) {
        
Stack<Character> P = new Stack<>();
        Stack<Character> Q = new Stack<>();
        for(int i=0; i<S.length(); i++) {
            char cur = S.charAt(i);
            if(cur == '#') {
                if(P.size() > 0) {
                    P.pop();
                }
            } else {
                P.push(cur);
            }
        }
        for(int i=0; i<T.length(); i++) {
            char cur = T.charAt(i);
            if(cur == '#') {
                if(Q.size() > 0) {
                    Q.pop();
                }
            } else {
                Q.push(cur);
            }
        }
        if(P.size() != Q.size()) {
            return false;
        }
        while(P.size() > 0) {
            if(P.pop() != Q.pop()) {
                return false;
            }
        }
        return true;
    }
}
