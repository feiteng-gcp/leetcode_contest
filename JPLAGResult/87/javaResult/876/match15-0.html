import java.util.*;

class Solution {

    class Group {
        int len;
    }

    public boolean isNStraightHand(int[] hand, int W) {

        int len = hand.length;
        if (W == 1) { return true; }
        if (len % W != 0) { return false; }

        int groupsCreated = 0;
        int groupsExpecting = len / W;
        int groupsFinished = 0;

        Arrays.sort(hand);

        Map<Integer,List<Group>> groupsNeedingNum = new HashMap<Integer,List<Group>>();

        for (int idx = 0; idx < hand.length; idx++) {

            int num = hand[idx];

            // if some group needs this number, then assign it and update the group
            if (groupsNeedingNum.containsKey(num)) {

                List<Group> groupsInterested = groupsNeedingNum.get(num);
                Group toJoin = groupsInterested.get(groupsInterested.size()-1);
                groupsInterested.remove(groupsInterested.size()-1);
                toJoin.len++;
                if (toJoin.len == W) {
                    // then this group is now finished.
                    groupsFinished++;
                }
                else {
                    groupsNeedingNum.putIfAbsent(num+1, new ArrayList<Group>());
                    groupsNeedingNum.get(num+1).add(toJoin);
                }

                // Clean up if no longer need that list.
                if (groupsInterested.size() == 0) {
                    groupsNeedingNum.remove(num);
                }

            }

            else {
                // Need to start a new group. First ensure this is legal.
                if (groupsCreated == groupsExpecting) { return false; } //adding another would be too many

                // Start a new group.
                Group group = new Group();
                group.len = 1;
                groupsNeedingNum.putIfAbsent(num+1, new ArrayList<Group>());
                groupsNeedingNum.get(num+1).add(group);
                groupsCreated++;
            }

        }
        return true;
    }
}
