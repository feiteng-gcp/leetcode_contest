class Solution {
    public boolean isNStraightHand(int[] hand, int W) {
        //Solution: Sort the array, then take out subsequences with the shortest one until they are all gone.
        if(hand.length%W != 0){
            return false;
        }
        Arrays.sort(hand);
        
        HashMap<Integer, Integer> h = new HashMap<>();
        for(int i = 0; i < hand.length; i++){
            if(!h.containsKey(hand[i])){
                h.put(hand[i], 1);
            }
            else{
                h.put(hand[i], h.get(hand[i]) + 1);
            }
        }
        
        Set<Integer> s = h.keySet();
        while(s.size() > 0){
            int min = Collections.min(s);
            for(int i = min; i < min + W; i++){
                if(!s.contains(i)){
                    return false;
                }
                else{
                    h.put(i, h.get(i) - 1);
                    if(h.get(i) == 0){
                        s.remove(i);
                    }
                }
            }
        }
        return true;
    }
}
