class Solution {
    
    public boolean isNStraightHand(int[] hand, int W) {
        int n = hand.length;
        if (n % W != 0) return false;
        
        int ng = n / W;
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int v : hand){
            if (map.containsKey(v)) map.put(v,map.get(v)+1);
            else map.put(v,1);
        }
        
        
        for (int i = 0;  i< ng; ++i){
            
            int first = map.firstKey();
            int cnt = map.get(first);
            if (cnt == 1) map.remove(first);
            else          map.put(first, cnt - 1);
            
            for (int j = 1; j < W; ++j){
                ++first;
                if (!map.containsKey(first)) return false;
                
                cnt = map.get(first);
                if (cnt == 1) map.remove(first);
                else          map.put(first, cnt - 1);
            }
        }
        return true;
    }
}
