class Solution {
    public boolean isNStraightHand(int[] hand, int W) {
        int n = hand.length;
        if(n%W != 0)
            return false;
        
        Arrays.sort(hand);
        boolean[] f = new boolean[n];
        for(int i =0;i<n/W;i++){
            int start = 0;
            int j =0;
            for(;j<n;j++){
                if(!f[j]){
                    start = hand[j];
                    f[j] = true;
                    break;
                }    
            }
            int C = W-1;
            for(;j<n&&C>0;j++){
                if(hand[j] == start || f[j])
                    continue;
                if(hand[j] - start ==1){
                    start++;
                    f[j] = true;
                    C--;
                }else
                    return false;
            }
            if(C > 0)
                return false;
        }
        return true;
    }
}
