class Solution {
    public boolean isNStraightHand(int[] hand, int W) {
        if(hand.length%W != 0)
            return false;
        boolean[] isUsed = new boolean[hand.length];
        Arrays.sort(hand);
        for(int i=0;i<hand.length/W;i++){
            int count = 0;
            int[] temp = new int[W];
            for(int j=0;j<hand.length;j++){
                if(!isUsed[j] && count==0){
                    //System.out.println("Here1"+count);
                    temp[count] = hand[j];
                    count++;
                    isUsed[j] = true;
                }else if(!isUsed[j] && count>0 && temp[count-1]+1 == hand[j]){
                    //System.out.println("Here2"+count);
                    temp[count] = hand[j];
                    count++;
                    isUsed[j] = true;
                }
                //System.out.println(count);
                if(count == W)
                    break;
            }
            // for(int k=0;k<temp.length;k++){
            //     System.out.print(temp[k]+" ");
            // }
            if(count != W)
                return false;
        }
        
        return true;
    }
}
