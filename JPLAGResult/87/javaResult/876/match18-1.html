class Solution {
    
    public boolean isNStraightHand(int[] hand, int W) {
    	if(hand.length % W != 0) return false;
    	Arrays.sort(hand);
    	int max = hand.length / W;
    	int startIdx = 0;
        for(int i = 0; i < max; i++) {
        	int combo = 0;
        	int last = -1;
        	int j = startIdx;
        	while(hand[startIdx] == -1) {
        		startIdx++;
        	}
        	for(j = startIdx; j < hand.length; j++) {
        		if(hand[j] == -1) continue;
        		if(last == -1 || last == hand[j] - 1) {
        			combo++;
        			last = hand[j];
        			//System.out.println(hand[j] + ", " + combo);
        			hand[j] = -1;
        			if(combo == W) {
        				break;
        			}
        		}

        	}
        	if(combo != W) return false;
        }
    	return true;
    	
    }
}
