class Solution {
    public boolean isNStraightHand(int[] hand, int W) {
        if(W == 1) {
            return true;
        }

        int len = hand.length;
        
        if(len % W != 0) {
            return false;
        }
        
        int min = hand[0];
        
        
        Map<Integer, Integer> map = new HashMap<>();
        List<Integer> list = new LinkedList<>();
        
        for (int i = 0; i < len; i++) {
            if(map.containsKey(hand[i])) {
                map.put(hand[i], map.get(hand[i]) + 1);
            }
            else {
                list.add(hand[i]);
                map.put(hand[i], 1);
            }
        }
        
        Collections.sort(list);
        
        int round = len / W;
        for(int i = 0; i < round; i++) {
            int start = list.get(0);
            int numOfStart = map.get(start);
            map.put(start, numOfStart - 1);
            if(numOfStart == 1) {
                list.remove(0);
            }
            int cur = start;
            for(int j = 1; j < W; j++) {
                if(!map.containsKey(++cur)) {
                    return false;
                }
                int numOfCur = map.get(cur);
                if(numOfCur < 1) {
                    return false;
                }
                map.put(cur, numOfCur - 1);
                if(numOfCur == 1) {
                    if(list.get(0) == cur) {
                        list.remove(0);
                    }
                    else {
                        return false;
                    }
                }
            }
        }
        return true;
        
    }
}
