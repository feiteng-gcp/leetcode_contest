class Solution {
public boolean isNStraightHand(int[] hand, int W) {
        if (hand.length / W * W != hand.length) return false;
        for (int i = 0; i < hand.length / W; i++) {
            int min = Integer.MAX_VALUE;
            int minIndex = -1;
            for (int k = i * W; k < hand.length; k++) {
                if (hand[k] < min) {
                    min = hand[k];
                    minIndex = k;
                }
            }
            swap(hand,i*W,minIndex);
            for (int j = 1; j < W; j++) {
                int index=find(hand,min+j,i*W+j);
                if(index<0) return false;
                else swap(hand,i*W+j,index);
            }
        }
        return true;
    }

    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
    private int find(int[] nums,int target,int begin){
        for (int i = begin; i < nums.length; i++) {
            
if(nums[i]==target) return i;
        }
        return -1;
    }
}
