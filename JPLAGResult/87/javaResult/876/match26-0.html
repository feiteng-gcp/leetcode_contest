class Solution {
    public boolean isNStraightHand(int[] hand, int W) {
        HashMap<Integer, Integer> hm = new HashMap<>();
        TreeSet<Integer> ts = new TreeSet<>();
        for (int card : hand) {
            if (!hm.containsKey(card))
                hm.put(card, 1);
            else
                hm.put(card, hm.get(card)+1);
            ts.add(card);
        }
        for (int card : ts) {
            int val = hm.get(card);
            if (val == 0)
                continue;
            for (int i = 0; i < W; i++) {
                if (!hm.containsKey(card+i)||hm.get(card+i)<val)
                    return false;
                hm.put(card+i, hm.get(card+i)-val);
            }
        }
        return true;
    }
}
