class Solution {

    public boolean isNStraightHand(int[] hand, int W) {
        
    	int i , n = hand.length;
    	Map<Integer , Integer> map = new HashMap<>();
    	Arrays.sort(hand);
    	for (i = 0;i < n;i ++) {
    		int value = hand[i];
    		if (!map.containsKey(value)) {
    			map.put(value , 1);
    		} else {
    			map.put(value , map.get(value) + 1);
    		}
    	}
    	for (i = 0;i < n;i ++) {
    		int value = hand[i];
    		if (map.containsKey(value)) {
    			for (int j = value;j <= value + W - 1;j ++) {
    				if (!map.containsKey(j)) {
    					return false;
    				} else {
    					map.put(j , map.get(j) - 1);
    					if (map.get(j) == 0) {
    						map.remove(j);
    					}
    				}
    			}	
    		}
    	}
    	return true;
    	
    }
    
}
