class Solution {
    public boolean isNStraightHand(int[] a, int w) {
        int n = a.length;
        if (n % w != 0) return false;

        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int i : a) {
            int get = map.getOrDefault(i, 0);
            map.put(i, get + 1);
        }
        while (!map.isEmpty()) {
            int smallest = map.firstKey();
            for (int i = 0; i < w; i++) {
                int key = smallest + i;
                Integer value = map.get(key);
                if (value == null) return false;
                if (value == 1) map.remove(key);
                else map.put(key, value - 1);
            }
        }
        return true;
    }

}
