class Solution {
    public boolean isNStraightHand(int[] hand, int W) {
        int n = hand.length;
        if (n % W != 0) {
            return false;
        }
        if (W == 1) {
            return true;
        }
        TreeMap<Integer, Integer> tmap = new TreeMap<>();
        for (int i = 0; i < n; i++) {
            int num = hand[i];
            
if (!tmap.containsKey(num)) {
                tmap.put(num, 1);
            } else {
                tmap.put(num, tmap.get(num) + 1);
            }
        }
        while (tmap.size() > 0) {
            int key = tmap.firstKey();
            boolean remove = removeGroup(W, tmap, key);
            if (!remove) {
                return false;
            }
        }
        return tmap.size() == 0;
    }
    
    private boolean removeGroup(int W, TreeMap<Integer, Integer> tmap, int key) {
        for (int num = key; num < key + W; num++) {
            if (!tmap.containsKey(num)) {
                return false;
            }
            if (tmap.get(num) == 1) {
                tmap.remove(num);
            } else {
                tmap.put(num, tmap.get(num) - 1);
            }
        }
        return true;
    } 
}
