class Solution {
    public boolean isNStraightHand(int[] hand, int W) {
        if (hand == null || W < 0) {
            return false;
        }
        int len = hand.length;
        if (len == 0) {
            return W == 0;
        }
        if (W == 0) {
            return false;
        }
        if (len % W != 0) {
            return false;
        }
        if (W == 1) {
            return true;
        }
        TreeMap<Integer, Integer> cnt = new TreeMap<>();
        for (int ele : hand) {
            cnt.put(ele, cnt.getOrDefault(ele, 0) + 1);
        }
            
        while (cnt.size() > 0) {
            int start = cnt.firstKey();
            for (int i = 0; i < W; i++) {
                int cur = start + i;
                if (cnt.get(cur) == null) {
                    return false;
                } 
                cnt.put(cur, cnt.get(cur) - 1);
            }
            for (int i = start; i < start + W; i++) {
                if (cnt.get(i) == 0) {
                    cnt.remove(i);
                }
            }
        }
        
        return true;
    }
}
