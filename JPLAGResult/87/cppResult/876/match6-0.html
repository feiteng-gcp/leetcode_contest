class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        map<int,int> m;
	int n=hand.size();
	if(n%W)return false;
	sort(hand.begin(),hand.end());
	for(int i=0;i<n;i++)
	{
		if(m[hand[i]]==0)m[hand[i]]=1;
		else m[hand[i]]++;
	}
	for(int i=0;i<n;i++)
	{
		if(m[hand[i]]==0)continue;
		for(int j=hand[i];j<hand[i]+W;j++)
		{
			if(m[j]==0)return false;
			m[j]--;
		}
	}
	return true;
    }
};
