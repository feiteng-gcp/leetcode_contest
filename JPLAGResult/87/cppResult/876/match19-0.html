class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int w) {
        int i,j,k,ln = hand.size();
        if(ln%w != 0)
        {
            return false;
        }
        set<int> s;
        map<int,int> m;
        for(i = 0;i<ln;i++)
        {
            m[hand[i]]++;
            s.insert(hand[i]);
        }
        while(!s.empty())
        {
            j = *(s.begin());
            //cout<<j<<"\n";
            m[j]--;
            if(m[j] == 0)
            {
                s.erase(j);
            }
            for(i = 2;i<=w;i++)
            {
                if(s.find(j+1) != s.end())
                {
                    m[j+1]--;
                    if(m[j+1] == 0)
                    {
                        s.erase(j+1);
                    }
                }
                else
                {
                    return false;
                }
                j++;
            }
        }
        return true;
    }
};
