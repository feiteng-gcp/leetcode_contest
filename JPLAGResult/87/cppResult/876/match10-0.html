const double EPS = 1e-10;
const double PI  = acos(-1.0);
const LL MOD = 1e9+7;

class Solution {
public:
  bool isNStraightHand(vector<int>& hand, int W) {
	int N = SZ(hand);
	if(N % W != 0) return false;

	map<int,int> cnt;
	for(int x: hand) ++cnt[x];
	SORT(hand);
	REP(i,N){
	  int x = hand[i];
	  if(cnt.count(x) && cnt[x] > 0){
		for(int i=x;i<x+W;++i){
		  if(!cnt.count(i) || cnt[i] == 0)
			return false;
		  --cnt[i];
		}
	  }
	}
	return true;
  }
};
