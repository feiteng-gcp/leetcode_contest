class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        int N=hand.size();
        if(N==0||N%W!=0)
            return false;
        if(W==1)
            return true;
        map<int,int> m;

        for(int i=0;i<N;i++){
            if(m.count(hand[i])!=0)
                m[hand[i]]++;
            else
                m[hand[i]]=1;
        }
        sort(hand.begin(),hand.end());
        for(int i=0;i<N;i++){
            int x=hand[i];
            if(m[x]>0){
                m[x]--;
                for(int j=1;j<W;j++){
                    if(m.count(x+j)<=0)
                        return false;
                    else{
                        m[x+j]--;
                    }
                }
            }
        }
        return true;
    }
};
