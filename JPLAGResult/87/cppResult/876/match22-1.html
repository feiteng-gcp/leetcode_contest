class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        int i, n;
        n = hand.size();
        
        if (n%W!=0)
            return false;
        
        vector<int> handSort = hand;
        std::sort(handSort.begin(), handSort.end());
        
        unordered_map<int,int> umap;

        for (i=0; i<n; i++) {
            if (umap.find(hand[i])==umap.end()) {
                umap.insert(make_pair(hand[i],1));
            }
            else {
                umap[hand[i]] = umap[hand[i]]+1;
            }
        }
        
        int val0, val;
        for (i=0; i<n; i++) {
            val0 = handSort[i];
            if (umap.find(val0)!=umap.end()) {
                for (val=val0; val<val0+W; val++) {
                    if (umap.find(val)==umap.end()) {
                        return false;
                    }
                    else {
                        if (umap[val]>1)
                            umap[val] = umap[val]-1;
                        else
                            umap.erase(val);
                    }
                }
            }
            
            if (umap.size()==0)
                return true;
        }
        
        return umap.size()==0;
    }
};
