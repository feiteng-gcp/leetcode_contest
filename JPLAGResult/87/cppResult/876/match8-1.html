class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        int N = hand.size(); if (N < W || N%W != 0) return false;
        unordered_map<int, int> hash;
        sort(hand.begin(), hand.end());
        for (int h : hand) ++hash[h];
        
        for (int i = 0; i < N; ++i) {
            int curr = hand[i];
            if (hash[curr] == 0) continue;
            
            for (int j = 0; j < W; ++j) {
                if (hash[curr + j] > 0) --hash[curr + j];
                else return false;
            }
        }
        
        return true;
    }
};
