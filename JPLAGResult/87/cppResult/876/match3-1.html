class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        if (hand.size()%W)
            return false;
        map<int,int> m;
        for (int i : hand)
            m[i]++;
        int cnt = 0;
        while (!m.empty()) {
            int n = m.begin()->first;
            while (cnt<W) {
                if (m.find(n)==m.end())
                    return false;
                m[n]--;
                if (m[n]==0)
                    m.erase(n);
                ++n;
                ++cnt;
            }
            cnt = 0;
        }
        return cnt==0;
    }
};
