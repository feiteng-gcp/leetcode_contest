class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        if (hand.empty() || hand.size() % W) return false;
        map<int, int> mp; // <card num, count>;
        for (int i = 0; i < hand.size(); i++) mp[hand[i]]++;
        
        while(mp.size()) {
            int num = mp.begin()->first;
            int w = 0;
            while(w < W) {
                if (mp.find(num) == mp.end()) return false;
                if (--mp[num] == 0) mp.erase(num);
                num++;
                w++;
            }
        }
        return true;
    }
};
