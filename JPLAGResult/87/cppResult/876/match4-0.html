class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) 
    {
        int size = hand.size();
        if(size % W != 0) return false;
        int group = size / W;
        map<int, int> counts;
        for(int e : hand)
            counts[e]++;
        
        auto it = counts.begin();
        
        for(int i = 0;i < group; i++)
        {
            while(it != counts.end() && it->second <= 0) ++it; 
            int x = it->first;
            for(int j = 0;j < W; j++, x++)
                if(counts[x] >= 1) counts[x]--;
                else
                    return false;        
        }
        
        return true;   
    }
};
