class Solution {
public:

    bool isNStraightHand(vector<int>& hand, int W) {
        int n = hand.size();
        
        if (n == 0 || W == 0 || n % W != 0) {
            return false;
        }
        
        map<int,int> cnts;
        
        for (auto num : hand) {
            ++cnts[num];
        }
        
        while (!cnts.empty()) {
            if (cnts.size() < W) {
                return false;
            }
            
            auto it = cnts.begin();
            int prev = it->first - 1;
            for (int i = 0; i < W; ++i) {
                if (prev + 1 != it->first) {
                    return false;
                }
                prev = it->first;
                it->second = it->second - 1;
                if (it->second == 0) {
                    it = cnts.erase(it);
                } else {
                    ++it;
                }
            }
        }
        
        return true;
    }
};
