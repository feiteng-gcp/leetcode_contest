class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        map<int, int> m;

        for (auto n: hand) {
            m[n]++;
        }
        while (!m.empty()) {
            int n = m.begin()->first;
            for (int i = 0; i < W; i++) {
                if (m.count(i + n) == 0) {
                    return false;
                }
                if(--m[i + n] == 0) {
                    m.erase(i + n);
                }
            }
        }
        return true;
    }
};
