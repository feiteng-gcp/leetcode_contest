class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        if (hand.size() % W != 0) {
            return false;
        }
        sort(hand.begin(), hand.end());
        vector<int> number;
        vector<int> remain;
        for (int i = 0; i < hand.size(); ++i) {
            bool found = false;
            for (int j = 0; j < number.size(); ++j) {
                if (number[j] == hand[i] - 1 && remain[j] > 0) {
                    found = true;
                    number[j] = hand[i];
                    -- remain[j];
                    break;
                }
            }
            if (!found) {
                if (number.size() == hand.size() / W) {
                    return false;
                }
                number.push_back(hand[i]);
                remain.push_back(W - 1);
            }
        }
        return true;
    }
};
