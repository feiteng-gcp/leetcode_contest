class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        map<int, int> val_cnt;
        for(int i=0; i<hand.size(); i++) {
            val_cnt[hand[i]] ++;
        }
        for(auto vc: val_cnt) {
            int val = vc.first;
            int cnt = vc.second;
            if(cnt == 0) continue;
            val_cnt[val] -= cnt;

            for(int i=val+1; i<val+W; i++) {

                if(val_cnt.find(i) == val_cnt.end() or val_cnt[i] < cnt)
                    return false;
                val_cnt[i] -= cnt;
            }
        }
        return true;
    }
};
