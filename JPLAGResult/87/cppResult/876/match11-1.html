class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        if (hand.size() % W != 0) {
            return false;
        }
        unordered_map<int, int> freq;
        sort(hand.begin(), hand.end());
        for (auto h: hand) {
            freq[h]++;
        }
        for (auto h: hand) {
            int r = freq[h];
            if (!r) {
                continue;
            }
            for (int i = 0; i < W; i++) {
                if (freq[h + i] < r) {
                    return false;
                }
                freq[h + i] -= r;
            }
        }
        return true;
    }
};
