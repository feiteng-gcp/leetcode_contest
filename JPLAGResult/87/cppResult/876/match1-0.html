class Solution {
public:
    bool isNStraightHand(vector<int>& hand, int W) {
        map<int, int> m;
        for (auto h : hand) {
            m[h]++;
        }
        auto it = m.begin();
        while (it != m.end()) {
            if (it->second == 0)
                ++it;
            else {
                it->second--;
                int next = it->first;
                for (int i = 1; i < W; ++i) {
                    next++;
                    if (m[next])
                        m[next]--;
                    else
                        return false;
                }
            }
        }
        return true;
    }
};
