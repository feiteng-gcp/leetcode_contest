class Solution {
public:
    bool backspaceCompare(string S, string T) {
        string tx,ty;

        for(auto c:S)
        {
            if(c=='#')
            {
                if(tx!="") tx.pop_back(); 
            }
            else tx.append(1,c);
        }
         for(auto c:T)
        {
            if(c=='#')
            {
                if(ty!="") ty.pop_back(); 
            }
            else ty.append(1,c);
        }
        cout<<tx<<endl<<ty<<endl;
        return tx.compare(ty)==0?1:0;
    }
};
