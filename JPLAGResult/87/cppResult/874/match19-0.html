class Solution {
public:
    bool backspaceCompare(string S, string T) {
        string ss;
        for(auto it:S){
            if(it=='#') {
                if(ss.length())ss.pop_back();
            }
            else{
                ss.push_back(it);
            }
        }
        string tt;
        for(auto it:T){
            if(it=='#') {
                if(tt.length())tt.pop_back();
            }
            else{
                tt.push_back(it);
            }
        }
        return ss==tt;
    }
};
