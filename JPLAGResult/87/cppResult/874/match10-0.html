class Solution {
public:
    bool backspaceCompare(string S, string T) 
    {
        stack<char> stk1, stk2;
        for(char ch : S)
        {
            if(ch == '#')
            {
                 if(stk1.size() > 0)
                     stk1.pop();
            }
            else
                stk1.push(ch);       
        }
        for(char ch : T)
        {
            if(ch == '#')
            {
                 if(stk2.size() > 0)
                     stk2.pop();
            }
            else
                stk2.push(ch);       
        }
        
        if(stk1.size() != stk2.size()) return false;
        while(stk1.size() > 0)
        {
            if(stk1.top() != stk2.top()) return false;
            stk1.pop();
            stk2.pop();
        }
        return true;        
    }
};
