class Solution {
public:
    bool backspaceCompare(string S, string T) {
        bool ret = false;
        stack<char> sp;
        stack<char> tp;
        for( auto c: S)
        {
            if( c == '#' )
            {
                if(!sp.empty())
                    sp.pop();
            }else
                sp.push(c);
        }
        for( auto c: T)
        {
            if( c == '#' )
            {
                if( !tp.empty())
                    tp.pop();
            }else
                tp.push(c);
        }
        string tans = "";
        string sans = "";
        while( !sp.empty())
        {
            sans += sp.top();
            sp.pop();
        }
        while( !tp.empty())
        {
            tans += tp.top();
            tp.pop();
        }
        
        return tans==sans;
    }
};
