class Solution {
public:
    bool backspaceCompare(string S, string T) {
        return convert(S) == convert(T);
    }
    string convert(string &s) {
        string ans;
        for (auto c: s) {
            if (c == '#') {
                if (ans.size()) ans.pop_back();
            } else {
                ans.push_back(c);
            }
        }
        return ans;
    }
};
