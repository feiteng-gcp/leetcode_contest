class Solution {
public:
    bool backspaceCompare(string S, string T) {
        string sx, sy;
        for(int i = 0; i < S.size(); i ++) {
            if('#' == S[i]) {
                if(!sx.empty()) sx.pop_back();
            }
            else sx.push_back(S[i]);
        }
        for(int i = 0; i < T.size(); i ++) {
            if('#' == T[i]) {
                if(!sy.empty()) sy.pop_back();
            }
            else sy.push_back(T[i]);
        }
        return (sx == sy) ? true : false;
    }
};
