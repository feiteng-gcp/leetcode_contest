class Solution {
public:
    bool backspaceCompare(string S, string T) {
        int i;
		string s;
		string t;
		
		for(i = 0; i < S.size(); i++)

		{
			if(S[i] == '#')
			{
				if(s.size())
				{
					s.pop_back();
				}
			}
			else
			{
				s += S[i];
			}
		}
		
			for(i = 0; i < T.size(); i++)

		{
			if(T[i] == '#')
			{
				if(t.size())
				{
					t.pop_back();
				}
			}
			else
			{
				t += T[i];
			}
		}

		return (s.compare(t) == 0);
    }
};
