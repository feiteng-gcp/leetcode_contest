class Solution {
public:
    string change(string S){
        string str;
        for(int i=0;i<S.length();i++){
            if(S[i]=='#'){
                if(!str.empty())
                str.pop_back();
            }else{
                str += S[i];
            }
        }
        return str;
    }
    bool backspaceCompare(string S, string T) {
        string str1=change(S),str2=change(T);
        return str1==str2;
    }
};
