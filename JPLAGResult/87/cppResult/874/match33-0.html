class Solution {
public:
    bool backspaceCompare(string S, string T) {
        string S_, T_;
        for (auto c: S) {
            if (c == '#') {
                if (S_.size() > 0) {
                    S_.pop_back();
                }
            } else {
                S_ += c;
            }
        }
        
        for (auto c: T) {
            if (c == '#') {
                if (T_.size() > 0) {
                    T_.pop_back();
                }
            } else {
                T_ += c;
            }
        }
        
        return S_ == T_;        
    }
};
