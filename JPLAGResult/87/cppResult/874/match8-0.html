class Solution {
public:
    bool backspaceCompare(string S, string T) {
        string s="",t="";
	int ls=S.size(),lt=T.size();
	for(int i=0;i<ls;i++)
	{
		if(S[i]=='#')
		{
			if(s.size())s.erase(s.end()-1);
		}
		else s=s+S[i];
	}
	for(int i=0;i<lt;i++)
	{
		if(T[i]=='#')
		{
			if(t.size())t.erase(t.end()-1);
		}
		else t=t+T[i];
	}
	return s==t;
    }
};
