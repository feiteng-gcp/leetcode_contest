class Solution {
    string update(string &str) {

        int n = str.size();
        string res;
        for (int i = 0; i < n; i += 1) {
            if (str[i] == '#') {
                if (!res.empty())
                    res.pop_back();
            }
            else {
                res.push_back(str[i]);
            }
        }
        return res;
    }
public:
    bool backspaceCompare(string S, string T) {
        string ns = update(S);
        string nt = update(T);
        return ns == nt;
    }
};
