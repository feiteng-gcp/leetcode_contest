class Solution {
    string processed(string S) {
        string ans;
        for (char ch : S) {
            if (ch == '#') {
                if (!ans.empty()) {
                    ans.erase(ans.size() - 1);
                }
            } else {
                ans += ch;
            }
        }
        return ans;
    }
public:
    bool backspaceCompare(string S, string T) {
        return processed(S) == processed(T);
    }
};
