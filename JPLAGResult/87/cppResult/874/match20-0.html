class Solution {
public:
    string getStr(string S)
    {
        string s="";
        int sn=S.length();
        for(int i=0;i<sn;++i)
        {
            if(S[i]=='#')
            {
                if(!s.empty())s.pop_back();
            }
            else
            {
                s.push_back(S[i]);
            }
        }
        return s;
    }
    bool backspaceCompare(string S, string T) {
        return getStr(S)==getStr(T);
    }
};
