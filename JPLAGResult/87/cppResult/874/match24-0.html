class Solution {
public:
    bool backspaceCompare(string S, string T) {
        stack<char> stk1, stk2;
        for (auto e : S) {
            if (e == '#') {
                if (!stk1.empty()) 
                    stk1.pop();
            }
            else 
                stk1.push(e);
            
        }
        for (auto e : T) {
            if (e == '#') {
                if (!stk2.empty())
                    stk2.pop();
            }
            else
                stk2.push(e);
        }
        if (stk1.size() != stk2.size()) return false;
        while (!stk1.empty()) {
            if (stk1.top() != stk2.top())
                return false;
            stk1.pop();
            stk2.pop();
        }
        return true;
    }
};
