class Solution {
public:
    bool backspaceCompare(string S, string T) {
        return convert(S) == convert(T);
    }
    
private:
    string convert(const string& s) {
        string res;
        

        for (auto c : s) {
            if (c == '#') {
                if (!res.empty()) {
                    res.pop_back();
                }
            } else {
                res += c;
            }
        }
        
        return res;
    }
};
