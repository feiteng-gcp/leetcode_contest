class Solution {
public:
    string removeBackspaces(string s) {
        string ans = "";
        for (auto c: s) {
            if (c == '#') {
                if (!ans.empty()) {
                    ans.pop_back();
                }
            } else {
                ans += c;
            }
        }
        return ans;
    }
    
    bool backspaceCompare(string S, string T) {
        string s = removeBackspaces(S);
        string t = removeBackspaces(T);
        return s == t;
    }
};
