class Solution {
public:
    bool backspaceCompare(string S, string T) {
        string tmp;
        for(int i=0; i<S.length(); i++) {
            if (S[i]=='#') {
                if (tmp.size()) tmp.pop_back();
            }
            else tmp.push_back(S[i]);
        }
        string t2;
        for(int i=0; i<T.length(); i++) {
            if (T[i]=='#') {
                if (t2.size()) t2.pop_back();
            }
            else t2.push_back(T[i]);
        }
        return tmp==t2;
    }
};
