class Solution {
public:
    bool backspaceCompare(string S, string T) {
        string S2,T2;

        for (int i=0;i<S.size();++i) {
            if (S[i]=='#') {
                if (!S2.empty()) S2.pop_back();
            } else {
                S2.push_back(S[i]);
            }
        }
        for (int i=0;i<T.size();++i) {
            if (T[i]=='#') {
                if (!T2.empty()) T2.pop_back();
            } else {
                T2.push_back(T[i]);
            }            
        }
        return S2==T2;
    }
};
