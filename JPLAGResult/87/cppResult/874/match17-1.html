class Solution {
public:
    string solve(string& S) {
        string ret;

        for (auto& it : S) {
            if (it == '#') {
                if (!ret.empty()) ret.pop_back();
            } else {
                ret.push_back(it);
            }
        }
        return ret;
    }
    bool backspaceCompare(string S, string T) {
        return solve(S) == solve(T);
        
    }
};
