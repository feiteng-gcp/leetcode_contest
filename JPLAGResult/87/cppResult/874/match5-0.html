class Solution {
public:
    bool backspaceCompare(string S, string T) {
        
        string a, b;
        for (int i = 0; i < S.size(); i++) {
            if (S[i] == '#') {
                if (a.size() > 0)
                    a.pop_back();
            } else
                a += S[i];
        }
        for (int i = 0; i < T.size(); i++) {
            if (T[i] == '#') {
                if (b.size() > 0)
                    b.pop_back();
            } else
                b += T[i];
        }
        cout << a << "; " << b << "\n";
        return a == b;
    }
};
