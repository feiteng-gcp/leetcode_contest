class Solution {
public:
    bool backspaceCompare(string s, string t) {
        string x1, x2;
        for(auto i: s) {
            if (i == '#') {
                if (!x1.empty()) {
                    x1.pop_back();
                }
            } else {
                x1 += i;
            }
        }
        for(auto i: t) {
            if (i == '#') {
                if (!x2.empty()) {
                    x2.pop_back();
                }
            } else {
                x2 += i;
            }
        }
        return x1 == x2;
    }
};
