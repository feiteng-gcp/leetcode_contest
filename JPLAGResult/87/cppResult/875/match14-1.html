class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n=A.size();
        vector<int> r(n,1),l(n,1);
        for(int i=1;i<n;i++)
        {
            l[i]=(A[i]>A[i-1]?l[i-1]+1:1);
        }
        for(int i=n-2;i>=0;i--)
        {
            r[i]=(A[i]>A[i+1]?r[i+1]+1:1);
        }
        int ans=0;
        for(int i=0;i<n;i++) 
        if(r[i]>=2&&l[i]>=2) ans=max(ans,r[i]+l[i]-1);
        return ans>=3?ans:0;
    }
};
