class Solution {
public:
    int longestMountain(vector<int>& A) {
        int i;
		
		if(A.size() < 3)
		{
			return 0;
		}
		
		int State = 0;
		int Result = 0;
		int Count = 1;
		for(i = 0; i < A.size() - 1; i++)
		{
			if(State == 0)
			{
				if(A[i] < A[(i + 1)])
				{
					Count++;
					State++;
				}
				else
				{
					Count = 1;
				}
			}
			else if(State == 1)
			{
				if(A[i] < A[(i + 1)])
				{
					Count++;
				}
				else if(A[i] > A[(i + 1)])
				{
					Count++;
					State++;
				}
				else
				{
					State = 0;
					Count = 1;
				}
			}
			else if(State == 2)
			{
				if(A[i] < A[(i + 1)])
				{
					State = 1;
					Count = 2;
				}
				else if(A[i] > A[(i + 1)])
				{
					Count++;
				}
				else
				{
					State = 0;
					Count = 1;
				}
			}
			
			if(State == 2)
			{
				Result = max(Result, Count);
			}
			//cout << Result << endl;
		}
		
		if(Result >= 3)
		{
			return Result;
		}
		else
		{
			return 0;
		}
    }
};
