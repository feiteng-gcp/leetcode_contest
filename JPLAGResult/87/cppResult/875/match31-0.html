class Solution {
public:
    int longestMountain(vector<int>& A) {
        int len = A.size();
        if(len<3) return 0;
        int res = 0;
        for(int i=1;i+1<len;i++){
            if(A[i]>A[i-1]&&A[i]>A[i+1]){
                int j = i-1;
                while(j>=0&&A[j]<A[j+1]){
                    j--;
                }
                int l = j;
                j = i+1;
                while(j<A.size()&&A[j]<A[j-1]){
                    j++;
                }
                int h = j;
                res = max(h-l-1, res);
                
            }
        }
        return res;
    }
};
