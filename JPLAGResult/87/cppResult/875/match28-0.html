class Solution {
public:
    int longestMountain(vector<int>& A) {
        int ans = 0;
        int up = 0;
        int down = 0;
        int dir = 0;
        for (int i = 1; i < A.size(); ++i) {
            if (A[i] == A[i - 1]) {
                dir = 0;
                up = 0;
                down = 0;
            } else {
                if (A[i] > A[i - 1]) {
                    if (dir == 1) {
                        ++ up;
                    } else {
                        dir = 1;
                        up = 1;
                    }
                } else {
                    if (dir == -1) {
                        ++ down;
                    } else {
                        dir = -1;
                        down = 1;
                    }
                    if (up > 0 && down > 0 && up + down + 1 > ans) {
                        ans = up + down + 1;
                    }
                }
            }
        }
        return ans;
    }
};
