class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n=A.size();
        vector<int> LR(n,0);
        vector<int> RL(n,0);
        
        for(int i=1;i<n;++i)
        {
            if(A[i]>A[i-1]) LR[i]=LR[i-1]+1;
            else LR[i]=0;
        }
        
        for(int i=n-2;i>=0;--i)
        {
            if(A[i]>A[i+1])RL[i]=RL[i+1]+1;
            else RL[i]=0;
        }
        
        int res=0;
        for(int i=1;i<n-1;++i)
        {
            if(LR[i]+RL[i]+1>=3 && LR[i]>0 && RL[i]>0)
            {
                res=max(res,LR[i]+RL[i]+1);
            }
        }
        return res;
    }
};
