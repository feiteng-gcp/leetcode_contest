class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n=A.size();
        int dev=0;
        for(int i=0;i+1<n;i++){
            int cont=0;
            if(A[i]<A[i+1]){
                i++;
                cont++;
                while(i<n && A[i-1]<A[i]){
                    cont++;
                    i++;
                }
                
                if(i==n)break;
                if( A[i-1]<=A[i]){
                    i--;
                    continue;
                }
                
                if(A[i-1]>A[i]){
                    while(i<n && A[i-1]>A[i]){
                        cont++;
                        i++;
                    }
                    
                    dev=max(dev,cont);
                    i-=2;
                }
                
            }
            
        }
        return dev;
    }
};
