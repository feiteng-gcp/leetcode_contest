class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n = A.size(); if (n == 0) return 0;
        int res = 0;
        for (int i = 1; i < n-1; ++i) {
            int left = 0;
            for (int j = i-1; j>=0; --j) {
                if (A[j] < A[j+1]) ++left;
                else break;
            }
            int right = 0;
            for (int j = i+1; j<n; ++j) {
                if (A[j] < A[j-1]) ++right;
                else break;
            }
            res = max(res, (left>0&&right>0) ? 1+left+right : 0);
        }
        return res;
    }
};
