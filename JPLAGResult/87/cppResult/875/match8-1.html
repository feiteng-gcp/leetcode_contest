class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n = A.size();

        vector<int> left(n, 0), right(n, 0);
        for(int i=1;i<n;i++){
            if(A[i]>A[i-1]) left[i] = left[i-1]+1;
        }
        for(int i=n-2;i>=0;i--){
            if(A[i]>A[i+1]) right[i] = right[i+1]+1;
        }
        int mx = 0;
        for(int i=1;i<n-1;i++){
            if(left[i]>0 && right[i]>0)
                mx = max(mx, left[i]+right[i]+1);
        }
        return mx;
    }
};
