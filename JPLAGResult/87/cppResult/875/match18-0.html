class Solution {
public:
    int longestMountain(vector<int>& A) {
        if (A.size()<=2) return 0;
        int d[10050];
        int ans=0;
        for(int i=0; i<A.size()-1; i++) {
            if (A[i]<A[i+1]) d[i]=1;
            else if (A[i]==A[i+1]) d[i]=0;
            else d[i]=-1;
        }
        int len=A.size()-1;
        cout<<len<<endl;
        for(int i=0; i<len; i++) cout<<d[i]<<' ';
        for(int i=0; i<len; i++) {
            if (d[i]!=1) continue;
            int j=i;
            while(d[j+1]==1&&j<len-1) j++;
            while(d[j+1]==-1&&j<len-1) j++;
            if (d[j]==-1) ans=max(ans, j-i+1+1);
            i=j;
        }
        return ans;
    }
};
