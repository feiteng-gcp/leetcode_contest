class Solution {
public:
	int longestMountain(vector<int>& A) {
		if (A.empty()) {
			return 0;
		}

		vector<int> longestUp(A.size(), 1), longestDown(A.size(), 1);
		for (int i = 1; i < A.size(); i++) {
			if (A[i] > A[i - 1]) {
				longestUp[i] = longestUp[i - 1] + 1;
			}
		}
		for (int i = A.size() - 2; i >= 0; i--) {
			if (A[i] > A[i + 1]) {
				longestDown[i] = longestDown[i + 1] + 1;
			}

		}

		int ans = 0;
		for (int i = 1; i < A.size() - 1; i++) {
			if (A[i] > A[i - 1] && A[i] > A[i + 1]) {
				ans = max(ans, longestUp[i - 1] + longestDown[i + 1] + 1);
			}
		}

		return ans;
	}
};
