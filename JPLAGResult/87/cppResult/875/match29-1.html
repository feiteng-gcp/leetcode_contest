class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n = A.size();
        int res = 0;
        bool inMountain = false;
        int Ms = INT_MIN;

        bool up = false;
        for (int i = 0; i < n; ++i) {
            if (!inMountain) {
                inMountain = true;
                up = true;
                Ms = i;
            } else {
                if (up) {
                    if (A[i] > A[i - 1])
                        ;
                    else if (A[i] < A[i - 1] && i - 1 - Ms > 0) {
                        up = false;
                    } else {
                        Ms = i;
                    }
                } else {
                    if (A[i] < A[i - 1])
                        ;
                    else {
                        res = max(res, i - 1 - Ms + 1 >= 3 ? i - 1 - Ms + 1 : 0);
                        up = true;
                        if (A[i] != A[i - 1])
                            Ms = i - 1;
                        else
                            Ms = i;
                    }
                }
            }
        }
        if (!up) {
            res = max(res, n - 1 - Ms + 1 >= 3 ? n - 1 - Ms + 1 : 0);
        }
        return res;
    }
};
