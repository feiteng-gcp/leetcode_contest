class Solution {
public:
    int longestMountain(vector<int>& A) {
        vector<int> dp1(A.size(), 1);
        vector<int> dp2(A.size(), 1);

        for(int i=1; i<A.size(); i++) {
            if(A[i] > A[i - 1]) dp1[i] = 1 + dp1[i - 1];
            // cout<<dp1[i]<<endl;
        }
        
        for(int i=A.size() - 2; i>=0; i--) {
            if(A[i] > A[i + 1]) dp2[i] = 1 + dp2[i + 1];
            // cout<<dp2[i];
        }
        int ans = 0;
        for(int i=0; i<A.size(); i++) {
            if(dp1[i] >= 2 and dp2[i] >= 2) {
                int tmp = dp1[i] + dp2[i] - 1;
                ans = max(ans, tmp);
            }
        }
        return ans >= 3 ? ans : 0;
    }
};
