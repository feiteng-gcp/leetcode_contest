class Solution {
    int f(vector<int>& A, int i) {
        for(int j = i + 1; j < A.size(); j++)
            if(A[j - 1] >= A[j]) return j - 1;
        return A.size() - 1;
    }
    int g(vector<int>& A, int i) {
        for(int j = i + 1; j < A.size(); j++)
            if(A[j - 1] <= A[j]) return j - 1;
        return A.size() - 1;
    }
public:
    int longestMountain(vector<int>& A) {
        int n = A.size();
        if(!n) return 0;
        int i = 0;
        int res = 0;
        while(i < n) {
            int j = f(A, i);
            int k = g(A, j);
            //cout << i << ' ' << j << ' ' << k << endl;
            if(i == k) i = k + 1;
            else {
                if(i < j && j < k)
                    res = max(res, k - i + 1);
                i = k;
            }
        }
        return res;
        
    }
};
