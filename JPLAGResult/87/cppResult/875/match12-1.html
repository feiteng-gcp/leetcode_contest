class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n, i, j, ans = 0;

        n = A.size();
        vector<int> l(n, 0);
        vector<int> r(n, 0);
        for (i = 1; i < n; i++) {
            if (A[i] > A[i - 1]) {
                l[i] = l[i - 1] + 1;
            }
        }
        for (i = n - 2; i >= 0; i--) {            
            if (A[i] > A[i + 1]) {
                r[i] = r[i + 1] + 1;
            }
        }
        for (i = 1; i < n - 1; i++) {
            //cout << i << endl;
            if (l[i] > 0 && r[i] > 0) {
                ans = max(ans, l[i] + r[i] + 1);
            }
        }
        return ans;
    }
};
