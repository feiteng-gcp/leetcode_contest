class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n = A.size();
        if (n<3)
            return 0;
        int res = 0;
        int up=0, down=0;
        for (int i=1; i<n; ++i) {
            if (A[i]>A[i-1]) {
                if (down>0) {                                             
                    res = max(res, up+down);
                    up = 0; down = 0;
                }
                ++up;                
            } else if (A[i]<A[i-1]) {
                if (up>0) {
                    ++down;
                }
            } else {
                if (up>0 && down>0)
                    res = max(res, up+down);
                up = 0; down = 0;
            }
        }
        if (up>0 && down>0)
            res = max(res, up+down);
        return res>0 ? res+1 : 0;
    }    
};
