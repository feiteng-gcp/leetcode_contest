class Solution {
public:
    int longestMountain(vector<int>& A) {
        int N = A.size();
        if (N <= 2) return 0;
        int ind = 0;
        int max_m = 0;
        
        while (ind < N - 1)
        {
            int i = ind + 1;
            if (A[i] <= A[ind]) ++ind; // Need to go up at least once
            else
            {
                // Go up as far as we can
                while (i < N - 1 && A[i + 1] > A[i]) ++i;
                if (i == N - 1) ind = N - 1; // Need to go down at least once
                else
                {
                    int j = i + 1;
                    if (A[j] == A[i]) ind = j; // Need to go down at least once
                    else
                    {
                        while (j < N && A[j] < A[j - 1]) ++j;
                    }
                    int mountain_len = j - ind;
                    if (mountain_len > max_m) max_m = mountain_len;
                    ind = j - 1;
                }
            }
        }
        return max_m;
    }
};
