class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n = A.size();
        vector<int> peaks;
        for (int i = 1; i < n - 1; ++i) {
            if (A[i-1] < A[i] && A[i+1] < A[i]) {
                peaks.push_back(i);
                ++i;
            }
        }
        if (peaks.empty()) return 0;
        vector<int> leftLen(n, 0);
        vector<int> rightLen(n, 0);
        for (int i = 1; i < n; ++i) {
            if (A[i] > A[i-1]) {
                leftLen[i] = leftLen[i-1] + 1;
            }
            // else leftLen[i] = 0;
        }
        
        for (int i = n - 2; i >= 0; --i) {
            if (A[i] > A[i+1]) {
                rightLen[i] = rightLen[i+1] + 1;
            }
        }
        
        int res = 0;
        for (int peak : peaks) {
            res = max(res, leftLen[peak] + rightLen[peak]);
        }
        return res + 1; // + 1 for peak itself
    }
};
