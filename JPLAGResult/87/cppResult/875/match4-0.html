class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n = A.size();
        vector<int> f(n), g(n);
        for (int i = 1; i < n; i++) {
            if (A[i] > A[i - 1]) {
                f[i] = f[i - 1] + 1;
            }
        }
        for (int i = n - 2; i >= 0; i--) {
            if (A[i] > A[i + 1]) {
                g[i] = g[i + 1] + 1;
            }
        }
        int res = 0;
        for (int i = 1; i < n - 1; i++) {
            if (f[i] > 0 && g[i] > 0) {
                res = max(res, f[i] + g[i] + 1);
            }
        }
        return res;
    }
};
