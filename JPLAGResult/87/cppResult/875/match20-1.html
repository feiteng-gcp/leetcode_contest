class Solution {
public:
    int longestMountain(vector<int>& A) {
        int res = 0, begin = 0;
        int st = 0; // 0 init 1 up 2 down
        for (int i = 1; i < A.size(); i++) {
            if (st == 0) {
                if (A[i] > A[i - 1]) {
                    begin = i - 1;

                    st = 1;
                }
            }
            else if (st == 1) {
                if (A[i] == A[i - 1]) {
                    st = 0;
                }
                else if (A[i] < A[i - 1]) { // met mid
                    st = 2;
                    res = max(res, i - begin + 1);
                }
            }
            else {
                if (A[i] < A[i - 1]) { // down

                    res = max(res, i - begin + 1);
                }
                else if (A[i] == A[i - 1]) {
                    st = 0;
                }
                else {
                    begin = i - 1;
                    st = 1;
                }
            }
        }
        return res;
    }
};
