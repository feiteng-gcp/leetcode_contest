class Solution {
public:
    int longestMountain(vector<int>& A) {
        int i,j,k,ln = A.size();
        if(ln<3)
        {
            return 0;
        }
        int ll[ln+1],rl[ln+1];
        ll[0] = 0;
        for(i = 1;i<ln;i++)
        {
            if(A[i]>A[i-1])
            {
                ll[i] = ll[i-1];
            }
            else
            {
                ll[i] = i;
            }
        }
        rl[ln-1] = ln-1;
        for(i = ln-2;i>=0;i--)
        {
            if(A[i+1]<A[i])
            {
                rl[i] = rl[i+1];
            }
            else
            {
                rl[i] = i;
            }
        }
        int ans = 0;
        for(i = 1;i<ln-1;i++)
        {
            if(rl[i]-ll[i]+1>ans&&ll[i]<i&&rl[i]>i)
            {
                ans = rl[i]-ll[i]+1;
            }
        }
        if(ans>=3)
        {
            return ans;
        }
        return 0;
    }
};
