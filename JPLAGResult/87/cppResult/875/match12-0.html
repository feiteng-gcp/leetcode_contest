class Solution {
public:
    int longestMountain(vector<int>& A) {
        int i,n=A.size();
        vector<int> leftCount(n,0) , rightCount(n,0);
        int maxCount, count;
        
        for (i=1; i<n; i++) {
            if (A[i]>A[i-1]) {
                leftCount[i] = leftCount[i-1]+1;
            }
        }
        for (i=n-2; i>=0; i--) {
            if (A[i]>A[i+1]) {
                rightCount[i] = rightCount[i+1]+1;
            }
        }
        
        maxCount = 0;
        for (i=1; i<n-1; i++) {
            if (leftCount[i]>0 && rightCount[i]>0) {
                if (leftCount[i]+rightCount[i]+1>maxCount)
                    maxCount = leftCount[i]+rightCount[i]+1;
            }
        }
        
        return maxCount;
    }
};
