class Solution {
public:
    int longestMountain(vector<int>& A) {
        if(A.size()==0)
            return  0;

        int res=0,left=0,right=0;        
        for(int i=1;i<A.size()-1;i++){
            if(A[i]>A[i-1]&&A[i]>A[i+1]){
                int l=i,r=i;
                while(l>0&&A[l]>A[l-1]){
                    left++;
                    l--;
                }
                while(r<A.size()-1&&A[r]>A[r+1]){
                    right++;
                    r++;
                }
                res=max(left+right+1,res);
                left=right=0;
            }
        }
        return res;
    }
};
