/***
[2,1,4,7,3,2,5]

[1,2,2,1,1,2]

*/

class Solution {
public:
    int longestMountain(vector<int>& A) {
        int n = A.size();
        int maxLen = 0, len = 0;
        int status = 0; // -1 down, 0 level, 1 up
        for (int i = 0; i < n - 1; i++) {
            if (A[i] < A[i+1]) {
                if (status == -1) {
                    maxLen = max(maxLen, len);
                }
                if (status != 1) {
                    status = 1;
                    len = 2;
                } else {
                    len++;
                }

            } else if (A[i] > A[i+1]) {
                if (status == 0) {
                    len = 0;
                } else {
                    status = -1;
                    len++;
                }
            } else {
                if (status == -1) {
                    maxLen = max(maxLen, len);
                }
                len = 0;
                status = 0;
            }
        }
        if (status == -1) maxLen = max(maxLen, len);
        return maxLen;
    }
};
