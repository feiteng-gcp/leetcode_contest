class Solution:
    def dietPlanPerformance(self, l: List[int], k: int, a: int, b: int) -> int:
        cur = sum(l[:k - 1])
        res = 0
        for i in range(k - 1, len(l)):
            cur += l[i]
            if cur < a: res -= 1
            elif cur > b: res += 1
            cur -= l[i - k + 1]
        return res
