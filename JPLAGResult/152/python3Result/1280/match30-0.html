class Solution:
    def dietPlanPerformance(self, cal: List[int], k: int, lower: int, upper: int) -> int:
        total = sum(cal[:k])
        points = 1 if total > upper else (-1 if total < lower else 0)
        for i in range(k, len(cal)):
            total += cal[i] - cal[i - k]
            points += 1 if total > upper else (-1 if total < lower else 0)
        return points
