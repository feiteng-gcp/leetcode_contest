class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        running_sum = 0
        points = 0
        for i in range(len(calories)):
            running_sum += calories[i]
            if i - k >= 0:
                running_sum -= calories[i - k]
            if i - k + 1 >= 0:
                if running_sum < lower:
                    points -= 1
                if running_sum > upper:
                    points += 1
        return points
