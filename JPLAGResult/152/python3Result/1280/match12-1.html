class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        score = 0
        window = sum(calories[:k])
        #print(window)
        
if window > upper:
            score = score + 1
        elif window < lower:
            score = score - 1
        for i in range(1,len(calories)-k+1):
            window = window - calories[i-1]
            window = window + calories[i+k-1]
            #print(window)
            if window > upper:
                score = score + 1
            elif window < lower:
                score = score - 1
        return score
