class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        n=len(calories)
        ps=[0]*(n+1)
        for i in range(1,n+1):
            ps[i]=ps[i-1]+calories[i-1]
        ret=0
        for i in range(n-k+1):
            tc=ps[i+k]-ps[i]
            if tc<lower:
                ret-=1
            if tc>upper:
                ret+=1
        return ret
