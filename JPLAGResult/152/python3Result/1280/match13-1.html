class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        points = 0
        SUM = 0
        if len(calories) < k:
            return 0
        for i in range(k):
            SUM += calories[i]
        
if SUM < lower:
            points = -1
        elif SUM > upper:
            points = 1
        for i in range(1, len(calories) - k + 1):
            SUM -= calories[i-1]
            SUM += calories[i + k - 1]
            if SUM < lower:
                points -= 1
            elif SUM > upper:
                points += 1
        return points
