class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        start=sum(calories[:k])
        ans=0
        if start<lower:
            ans-=1
        elif start>upper:
            ans+=1
            
        for i in range(k,len(calories)):
            start-=calories[i-k]
            start+=calories[i]
            if start<lower:
                ans-=1
            elif start>upper:
                ans+=1            
        return ans
