class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        s = 0
        c = 0
        N = len(calories)
        for i in range
(N):
            s += calories[i]
            if i >= k:
                s -= calories[i-k]
            if i >= k-1:
                #print(i, s)
                if s < lower:
                    c -= 1
                if s > upper:
                    c += 1
        return c
