class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        T, points = 0, 0
        for i, cal in enumerate(calories):
            T += cal
            
            
if i >= k:
                T -= calories[i-k]
            
            if i >= k - 1:
                if T < lower:
                    points -= 1
                if T > upper:
                    points += 1
        return points
