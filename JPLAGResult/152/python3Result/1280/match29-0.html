class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        pts = cal = 0
        for i in range(k-1):
            cal += calories[i]
        for i in range(k-1, len(calories)):
            cal += calories[i]
            if cal > upper:
                pts += 1
            elif cal < lower:
                pts -= 1
            cal -= calories[i-k+1]
        return pts
