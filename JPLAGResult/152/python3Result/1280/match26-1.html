class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        tot = 0
        point = 0
        for i, c in enumerate(calories):
            tot += c
            
if i >= k: tot -= calories[i - k]
            if i >= k - 1:
                if tot < lower: point -= 1
                if tot > upper: point += 1
        return point
