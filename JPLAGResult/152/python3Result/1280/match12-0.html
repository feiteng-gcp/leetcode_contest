class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        t = sum(calories[:k])
        
        points = 0
            
        if t < lower:
            points -= 1
        elif t > upper:
            points += 1

        for i in range(k, len(calories)):
            t += calories[i]
            t -= calories[i-k]
            
            if t < lower:
                points -= 1
            elif t > upper:
                points += 1
                
        return points
