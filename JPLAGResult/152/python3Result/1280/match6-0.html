class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        curr = sum(calories[:k])
        if curr < lower:
            points = -1
        elif curr > upper:
            points = 1
        else:
            points = 0
        
        for i in range(k, len(calories)):
            curr += -calories[i - k] + calories[i]
            
            if curr < lower:
                points -= 1
            elif curr > upper:
                points += 1
        
        return points
