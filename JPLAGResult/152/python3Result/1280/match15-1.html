class Solution:
    def dietPlanPerformance(self, calories: List[int], k: int, lower: int, upper: int) -> int:
        s = sum(calories[:k])
        ret = 0
        if s < lower:
            ret 
= -1
        elif s > upper:
            ret = 1
        for i in range(k, len(calories)):
            s += calories[i] - calories[i - k]
            if s < lower:
                ret -= 1
            elif s > upper:
                ret += 1
        return ret
