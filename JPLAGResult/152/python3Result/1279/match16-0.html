class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        
        def check(n):
            if n == 1:
                return False
            if n < 4:
                return True
            for i in range(2, n):
                if n % i == 0:
                    return False
            return True
        
        def f(n):
            if n == 0:
                return 1
            return n * f(n - 1)
        
        su = 0
        for i in range(1, n + 1):
            if check(i):
                su += 1
        return f(su) * f(n - su) % (10 ** 9 + 7)
