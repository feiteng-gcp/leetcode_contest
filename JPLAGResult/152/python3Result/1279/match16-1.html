class Solution:
    
def prime(self, x):
        if x==1:
            return False
        if x==2:
            return True
        
        for e in range(2, x):
            if x%e == 0:
                return False
        return True
            
    def fact(self, n):
        acc = 1
        while n>1:
            acc *= n
            n -= 1
        return acc
    
    def numPrimeArrangements(self, n: int) -> int:
        acc = 0
        for i in range(1, n+1):
            if self.prime(i):
                acc += 1
        return self.fact(acc)*self.fact(n-acc)%(10**9+7)
