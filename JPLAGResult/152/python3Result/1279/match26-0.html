'''
Figure out number of prime numbers. Then number of prime indices is also that number m.

Then m! ways to put in prime numbers, and (n-m)! ways to put in non-prime numbers

Test for 5: 2, 3, 5 are prime numbers
3! * 2! = 12

Since n is small, we can do a simple sieve.
'''
import math

class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        if n <= 2:
            return 1
        
        isPrime = [True]*(n+1)
        isPrime[0] = False
        isPrime[1] = False
        for i in range(2, n+1):
            if isPrime[i]:
                for j in range(2*i, n+1, i):
                    isPrime[j] = False
        
        # now count number of prime numbers
        numPrimes = sum(isPrime)
        numNonPrimes = n - numPrimes
        return (math.factorial(numPrimes)*math.factorial(numNonPrimes)) % (10**9 + 7)
