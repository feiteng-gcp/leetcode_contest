class Solution:
  def isPrime(self, n):
    if n <= 1:
      return False
    if n <= 3:
      return True

    
if n % 2 == 0 or n % 3 == 0:
      return False

    i = 5
    while i * i <= n:
      if (n % i == 0 or n % (i + 2) == 0):
        return False
      i = i + 6

    return True

  def numPrimeArrangements(self, n: int) -> int:
    prime_count = 0

    for i in range(1, n + 1):
      if self.isPrime(i):
        prime_count += 1

    rest_count = n - prime_count
    ans = 1
    for k in range(prime_count, 0, -1):
      ans *= k

    for k in range(rest_count, 0, -1):
      ans *= k

    return ans % (10**9 + 7)
