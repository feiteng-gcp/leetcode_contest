class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        mod = 10 ** 9 + 7

        def fac(n):
            ret = 1
            for i in range(n):
                ret = ret * (i + 1) % mod
            return ret

        p_cnt = 0
        for i in range(2, n + 1):
            flag = True
            j = 2
            while j * j <= i:
                if i % j == 0:
                    flag = False
                    break
                j += 1
            if flag:
                p_cnt += 1

        return fac(p_cnt) * fac(n - p_cnt) % mod
