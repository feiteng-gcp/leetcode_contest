mod = (int)(1e9+7)
class Solution:
    primes = []
    def numPrimeArrangements(self, n: int) -> int:
        nPrimes = 0
        for i in range(2,n+1):
            gd = True
            for j in range(2, i):                
                if i % j == 0:
                    gd = False
            if gd:
                nPrimes+=1
        ans = 1
        for i in range(2, nPrimes+1):
            ans = (ans * i) % mod
        z = n - nPrimes
        for i in range(2, z+1):
            ans = (ans * i) % mod
        return ans
