class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        def get_primes():
            arr = [True] * (n + 1)
            arr[0] = False
            arr[1] = False
            ret = 0
            idx = 2
            while idx <= n:
                if arr[idx]:
                    ret += 1
                    j = idx * 2
                    while j <= n:
                        arr[j] = False
                        j += idx
                idx += 1
            # print(arr)
            return ret
        a = get_primes()
        b = n - a
        ret = 1
        for i in range(1, a + 1):
            ret *= i
        for i in range(1, b + 1):
            ret *= i
        
        return ret % (10 ** 9 + 7)
