class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        MOD = 10**9 + 7
        
        def fac(n):
            f = 1
            for i in range(2, n+1):
                f = f*i%MOD
            return f
        
        def num_primes(n):
            count = 0
            known = []
            for i in range(2,n+1):
                isprime = True
                for a in known:
                    if i%a==0:
                        isprime = False
                        break
                if isprime: known.append(i)
            return len(known)
        
        p = num_primes(n)
        
        return (fac(p) * fac(n-p)) % MOD
