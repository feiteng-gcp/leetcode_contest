from math import factorial

def is_prime(n):
    if n == 1:
        return False
    i = 2
    while i**2 <= n:
        if n % i == 0:
            return False
        i += 1
        
    return True

def primes_up_to(n):
    count = 0
    for i in range(2, n+1):
        count += is_prime(i)
        
    return count

class Solution:
    def numPrimeArrangements(self, n):
        if n == 1:
            return 1
        
        ps = primes_up_to(n)
        return (factorial(n - ps)*factorial(ps)) % (10**9 + 7)
