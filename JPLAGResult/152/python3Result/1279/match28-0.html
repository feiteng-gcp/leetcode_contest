class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        
        
        
        def isPrime(n):
            for factor in range(2, int(n**0.5)+1):
                if n % factor == 0:
                    return False
            return True
        
        prime_count = 0
        for n in range(2, n+1):
            if isPrime(n) == True:
                prime_count += 1
        
        def factorial(n):
            res = 1
            for num in range(2, n+1):
                res *= num
            return res
        
        return factorial(prime_count) * factorial(n - prime_count) % (10**9 + 7)
