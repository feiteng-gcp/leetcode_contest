class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        MOD = 10**9 + 7
        cnt 
= 0
        for i in range(2, n+1):
            a = 0
            for x in range(2, i):
                if i % x == 0:
                    a = 1
            if a == 0:
                cnt += 1
        r = 1
        for i in range(1, cnt+1):
            r = r * i % MOD
        for i in range(1, n-cnt+1):
            r = r * i % MOD
        return r
