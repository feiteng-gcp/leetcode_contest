class Solution:
  def numPrimeArrangements(self, n: int) -> int:
    M = 1000000007
    prm 
= [1] * (n + 1)
    for i in range(2, n):
      if prm[i]:
        for j in range(i * i, n + 1, i):
          prm[j] = 0
    np = 0
    for i in range(2, n + 1):
      if prm[i]:
        np += 1
    ret = 1
    for i in range(1, np + 1):
      ret = ret * i % M
    for i in range(1, n - np + 1):
      ret = ret * i % M
    return ret
