class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        total_prime = 0
        for x in range(1, n+1):
            if self.isPrime(x):
                total_prime += 1
        result = 1
        for i in range(total_prime, 0, -1):
            result = (result * i) % (10**9 + 7)
            
        for i in range(n-total_prime, 0, -1 ):
            result = (result * i) % (10**9 + 7)
            
        return result % (10**9 + 7)
         
        
        
 
    
    def isPrime(self, n):
        if n == 1: return False
        for i in range(2, n):
            for j in range(2, i):
                if n % j == 0:
                    return False
        return True
