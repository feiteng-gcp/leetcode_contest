from math import factorial as f
class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        def check(n):
            for i in range(2,n):
                if n%i==0:
                    return False
            return True
        isPrime
= 0
        notPrime = 1
        for i in range(2,n+1):
            if check(i):
                isPrime+=1
            else:
                notPrime+=1
        return f(isPrime)*f(notPrime)%(10**9+7)
