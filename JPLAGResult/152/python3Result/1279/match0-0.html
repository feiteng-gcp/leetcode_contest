class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        mod = 1000000007
        num_prime = 0
        for i in range(1, n + 1):
            if self.is_prime(i):
                num_prime += 1 
        num_other = n - num_prime
        res = 1 
        for i in range(1, num_prime + 1):
            res *= i
            res = res % mod
        for i in range(1, num_other + 1):
            res *= i
            res = res % mod
        
        return res 
        
    def is_prime(self, n):
        if n < 2:
            return False

            if n % i == 0:
                return False
        return True
