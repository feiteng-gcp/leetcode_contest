import math
class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        mod = 10**9 + 7
        p 
= 0
        for i in range(2, n+1):
            d = 2
            good = True
            while d*d <= i:
                if i % d == 0:
                    good = False
                    break
                d += 1
            if good: p += 1

        return (math.factorial(p)*math.factorial(n-p))%mod
