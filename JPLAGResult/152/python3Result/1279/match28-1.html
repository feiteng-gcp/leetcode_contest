import math
class Solution:
    
    def numPrimeArrangements(self, n: int) -> int:
        NUM = 1000000007
        def isPrime(n):
            if n <= 1:
                return False
            
            for i in range(2, math.floor(math.sqrt
(n)) + 1):
                if n % i == 0:
                    return False
            return True
        
        cntNonPrime, cntPrime = 0, 0
        for i in range(1, n + 1):
            if isPrime(i):
                #print(i)
                cntPrime += 1
            else:
                cntNonPrime += 1
        
        #print(cntNonPrime, cntPrime)
        return ((math.factorial(cntPrime) % NUM) * (math.factorial(cntNonPrime) % NUM)) % NUM
