class Solution:
    def isPrime(self, x):
        if x == 1:
            return False
        for i in range(2, int(math.sqrt(x)) + 1):
            if x % i == 0:
                return False
        return True
    
    
    def fac(self, x):
        ret = 1
        for i in range(1,x+1):
            ret *= i
            ret %= 10**9+7
        
        return ret
    
    def numPrimeArrangements(self, n: int) -> int:
        cnt = 0
        for i in range(1, n+1):
            if self.isPrime(i):
                cnt += 1
        #print(cnt)
        return (self.fac(cnt) * self.fac(n-cnt)) % (10**9+7)
