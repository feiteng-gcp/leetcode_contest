class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        if n == 1:
            return 1
        primes 
= [2]
        for i in range(3, n + 1):
            flag = True
            for prime in primes:
                if i % prime == 0:
                    flag = False
                    break
            if flag:
                primes.append(i)
        
        def factorize(n):
            ans = 1
            for i in range(n, 1,-1):
                ans *= i
            return ans
        return (factorize(n - len(primes)) * factorize(len(primes))) % 1000000007
