class Solution:
    def isPrime(self, n):
        for i in range(2, int(n**0.5)+1):
            if n%i == 0:
                return False
        return True
    def numPrimeArrangements(self, n: int) -> int:
        if n <= 2:
            return 1
        count = 0
        for i in range(2, n+1):
            if self.isPrime(i):
                count += 1
        mod 
= 10**9 + 7
        ans = 1
        for i in range(1, count+1):
            ans = (ans*i)%mod
        for i in range(1, (n-count)+1):
            ans = (ans*i)%mod
        return ans
