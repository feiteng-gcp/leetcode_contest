class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        if n == 1: return 1
        if n == 2: return 1
        
        def isPrime(A):
            if A in [2, 3]: return True
            up = int(math.sqrt(A)) + 1
            for i in range(2, up):
                if A % i == 0: return False
            return True
        
        cnt = 0
        for i in range(2, n + 1):
            # print(i)
            if isPrime(i): cnt += 1
        
        mod = int(1e9) + 7
        x = cnt
        y = n - cnt
        # print(x, y, cnt)
        s1 = 1
        for i in range(1, x + 1):
            s1 *= i
            s1 %= mod
        s2 = 1
        for i in range(1, y + 1):
            s2 *= i
            s2 %= mod
        return (s1 * s2) % mod
