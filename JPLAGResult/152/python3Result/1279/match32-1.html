M=1000000007
def mul(x,y):
    return x*y%M
def getVal(x):
    ret=1
    for i in range(1,x+1):
        ret=mul(ret,i)
    return ret

def isprime(x):
    if x==1: return False
    for i in range(2,x):
        if x%i==0:
            return False
    return True
class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        pcount=0
        npcount=0
        for i in range(1,n+1):
            if isprime(i):
                pcount+=1
            else:
                npcount+=1
        
        return mul(getVal(pcount),getVal(npcount))
