from functools import lru_cache
class Solution:
    
    def numPrimeArrangements(self, n: int) -> int:
        @lru_cache(None)
        def fac(x):
            if x <= 1: return 1
            return x * fac(x - 1) % (10 ** 9 + 7)
        np = 0
        for i in range(2, n + 1):
            isPrime = True
            for j in range(2, i):
                if i % j == 0:
                    isPrime = False
            if isPrime: np += 1
        print(np)
        return fac(np) * fac(n - np) % (10 ** 9 + 7)
