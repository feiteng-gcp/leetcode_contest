class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        p 
= []
        for x in range(2, n+1):
            flag = True
            for k in p:
                if x % k == 0:
                    flag = False
                    break
            if flag:
                p.append(x)
        
        ans = 1
        m = len(p)
        for i in range(2, m+1):
            ans = (ans * i) % 1000000007
        for i in range(2, n-m+1):
            ans = (ans * i) % 1000000007
        return ans
