class Solution:
    def numPrimeArrangements(self, n: int) -> int:
        '''
        count number of primes
        '''
        def countPrimesNumbers(M):
            p = [1]*(M+1)
            p[0] = p[1] = 0
            for x in range(2, int(math.sqrt(M))+1):
                if p[x]:
                    for y in range(x*x, M+1, x):
                        p[y] = 0

            count = 0
            for i in range(2, M+1):
                if p[i]:
                    count += 1

            return count

        MAX = 10**9 + 7
        
        x = countPrimesNumbers(n)
        # print(x)
        
        y = n - x
        mm = 1
        for i in range(y, 0, -1):
            mm *= i
            mm %= MAX
            
        
        res = 1
        for i in range(x, 0, -1):
            res *= i
            res %= MAX
        
        return res * mm %MAX
