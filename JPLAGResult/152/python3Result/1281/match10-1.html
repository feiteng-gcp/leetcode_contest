class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        n = len(s)
        p = [[0 for j in range(n+1)] for i in range
(26)]
        for i, x in enumerate(s):
            for j in range(26):
                p[j][i+1] += p[j][i]
            p[ord(x)-ord('a')][i+1] += 1
        
        
        ans = []
        for q in queries:
            left, right, k = q[0], q[1], q[2]
            c = 0
            for i in range(26):
                d = p[i][right+1] - p[i][left]
                if d % 2 == 1:
                    c += 1
            l = q[1]+1 - q[0]
            k2 = c // 2
            ans.append(k2 <= k) 
        return ans
