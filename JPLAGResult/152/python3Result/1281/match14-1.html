class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        I = lambda c: ord(c) - ord('a')
        pre = [[0] * 26]
        for c in s:
            pre.append(pre[-1][:])
            pre[-1]
[I(c)] ^= 1
        # print(pre)
        res = []
        for l, r, k in queries:
            cnt = 0
            for i in range(26):
                cnt += pre[r + 1][i] ^ pre[l][i]
            # print(l, r, k, cnt)
            res.append(cnt - 1 <= 2 * k)
        return res
