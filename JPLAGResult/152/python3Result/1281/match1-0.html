class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        n = len(s)
        f = [[0] * 26 for _ in range(n + 1)]
        for i in range(n):
            for j in range(26):
                f[i + 1][j] = f[i][j]
            f[i + 1][ord(s[i]) - ord('a')] += 1
        ans = []
        for p, q, m in queries:
            t = 0
            for i in range(26):
                if (f[q + 1][i] - f[p][i]) % 2 != 0:
                    t += 1
            ans.append(m >= t // 2)
        return ans
