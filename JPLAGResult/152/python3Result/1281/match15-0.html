class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        n = len(s)
        ch_dict = [[0] * 26 for _ in range(n + 1)]
        for i in range(1, n + 1):
            ch = s[i - 1]
            for j in range(26):
                if ord(ch) - ord('a') == j:
                    ch_dict[i][j] = ch_dict[i - 1][j] + 1 
                else:
                    ch_dict[i][j] = ch_dict[i - 1][j]

        res = []
        for query in queries:
            mismatch = self.num_mismatch(ch_dict, query[0], query[1])
            if mismatch > query[2]:
                res.append(False)
            else:
                res.append(True)
        return res
        
    def num_mismatch(self, ch_dict, left, right):
        odd = 0
        for i in range(26):
            if (ch_dict[right + 1][i] - ch_dict[left][i]) % 2 == 1:
                odd += 1 
        return odd // 2
