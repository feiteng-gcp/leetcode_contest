from functools import lru_cache
class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        n, ans = len(s), []
        aux = {c: [0] * (n + 1) for c in string.ascii_lowercase}
        for i, c in enumerate(s, 1):
            for x in aux:
                aux[x][i] = aux[x][i - 1] + (c == x)
                
        @lru_cache(None)
        def num_odd(l, r):
            return sum(
                (v[r + 1] - v[l]) % 2
                for c, v in aux.items()
            )
                
        for l, r, k in queries:
            ans.append(num_odd(l, r) // 2 <= k)
        return ans
