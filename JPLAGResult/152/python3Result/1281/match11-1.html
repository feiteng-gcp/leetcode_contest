class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:

        ans = []
        count = collections.Counter()
        temp = [count.copy()]
        for c in s:
            count[c] += 1
            count[c] %= 2

            temp.append(count.copy())


        
        for l, r, k in queries:
            num = 0
            for ch in (set(temp[r+1]) | set(temp[l])):

                if (temp[r+1]
[ch] - temp[l][ch])%2 == 1:
                    num += 1
            ans.append(num//2 <= k)
                


        return ans
