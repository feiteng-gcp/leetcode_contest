class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        al = ['a','b','c','d','e','f','g','h','i','j','k','l','m',
              'n','o','p','q','r','s','t','u','v','w','x','y','z']
        maps = [[0] for i in range(26)] #REMEMBER THAT THERE IS A 0 AT THE BEGINNING FOR SOME REASON
        for x in s:
            c = al.index(x)
            maps[c].append((maps[c][-1]+1)%2)
            for l in range(26):
                if l == c:
                    continue
                maps[l].append(maps[l][-1])

        #for m in maps:
        #    print(m)

        #Let's just try a brute force kind of approach
        answers = []
        for q in queries:
            #print(s[q[0]:q[1]])
            errors = sum([(maps[i][q[0]]+maps[i][q
[1]+1])%2 for i in range(26)]) - (1+q[1]-q[0])%2
            if q[2] >= errors//2:
                answers.append(True)
            else:
                answers.append(False)
        return answers
