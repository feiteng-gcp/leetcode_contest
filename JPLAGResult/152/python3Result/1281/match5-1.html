class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        from copy import copy
        counter = collections.defaultdict(int)
        counters = []
        for i in s:
            counter = copy(counter)
            counter[i] += 1
            counters.append(counter)
        
        ret = []
        for left, right, k in queries:
            odd = 0
            if left > 0:
                left_hash = counters[left-1]
            else:
                left_hash = collections.defaultdict(int)
            for char in counters[right]:
                if (counters
[right][char] - left_hash[char]) % 2 == 1:
                    odd += 1
            ret.append(odd <= 2 * k + 1)
        return ret
