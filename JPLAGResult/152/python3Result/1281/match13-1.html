from collections import Counter

class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        
        res = [[0] * 26 for _ in range(len(s) + 1)]
        
        for i in range(26):
            cnt = 0
            for j in range(len(s)):
                if ord(s[j]) - 97 == i:
                    cnt += 1
                res[j + 1][i] = cnt
            
        # [0...1] -> res[2] - res[0]
        q = []
        for i, j, k in queries:
            n = [abs(q - r)%2 for q, r in zip(res[j + 1], res[i])]
            f = sum(n)//2
            q.append(f <= k)
        return q
