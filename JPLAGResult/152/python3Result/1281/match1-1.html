class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        n = len(s)
        l 
= len(queries)
        sta = [[0]*26 for _ in range(n+1)]
        for i in range(1, n+1):
            for j in range(26):
                sta[i][j] = sta[i-1][j]
            sta[i][ord(s[i-1])-ord('a')] += 1
        ans = []
        for i in range(l):
            temp = 0
            left, right, k = queries[i]
            
for j in range(26):
                if (sta[right+1][j]-sta[left][j])%2 == 1:
                    temp += 1
            ans.append((temp-2*k)<=1)
        return ans
