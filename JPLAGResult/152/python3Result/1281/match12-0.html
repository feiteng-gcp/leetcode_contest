from copy import copy
from pprint import pprint

class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        
        odd_chars = set()
        
        parity_history = [copy(odd_chars)]
        num_odd = 0
        for char in s:
            if char in odd_chars:
                odd_chars.remove(char)
            else:
                odd_chars.add(char)
            parity_history.append(copy(odd_chars))

        
        responses = []
        for low, high, edits in queries:
            # [closed, open)
            high += 1
            
            before = parity_history[low]
            after = parity_history[high]
            
            num_odd_range = len(before.symmetric_difference(after))
                    
            diffs = num_odd_range // 2
            responses.append(diffs <= edits)
        
        return responses
