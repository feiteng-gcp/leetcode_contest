class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        S = [[0 for _ in range(26)] for _ in range(len(s))]
        S[0][ord(s[0]) - ord('a')] += 1
        
        for i in range(1, len(s)):
            for j in range(26):
                S[i][j] = S[i - 1][j]
            
            S[i][ord(s[i]) - ord('a')] += 1

        ans = []
        
        for l, r, k in queries:
            c = 0
            for j in range(26):
                aux = S[l - 1][j] if l > 0 else 0
                if (S[r][j] - aux) % 2 == 1:
                    c += 1
            
            if (r - l + 1) % 2 == 0:
                ans.append(k >= c // 2)
            else:
                if k == 0:
                    ans.append(c == 1)
                else:
                    ans.append(k >= (c - 1) // 2)
        
        return ans
