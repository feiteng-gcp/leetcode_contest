from collections import defaultdict
class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        prefix_set = []
        ss = set()
        
        
for c in s:
            if c in ss:
                ss.remove(c)
            else:
                ss.add(c)
            prefix_set.append(set(ss))
        
        ret = []
        cache = {}
        for l, r, m in queries:
            if (l, r) not in cache:
                if l != 0:
                    cache[(l, r)] = len(prefix_set[r] ^ prefix_set[l - 1]) // 2
                else:
                    cache[(l, r)] = len(prefix_set[r]) // 2
            ret.append(cache[(l, r)] <= m)
                
        return ret
