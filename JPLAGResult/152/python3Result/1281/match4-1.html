class Solution:
    def canMakePaliQueries(self, s: str, queries: List[List[int]]) -> List[bool]:
        
        
        
        
        pre = collections.defaultdict(tuple)
        precount = [0]*26
        pre[-1] = tuple(precount)
        for i,ch in enumerate(s):
            precount[ord(ch)-97] += 1
            pre[i] = tuple(
precount)
        
        res = []
        for left, right, k in queries:
            odd_count = 0
            for i in range(26):
                if (pre[right][i] - pre[left-1][i]) % 2 == 1:
                    odd_count += 1
            
            if odd_count % 2 == 0:
                if k >= odd_count//2:
                    res.append(True)
                else:
                    res.append(False)
            else:
                if k >= (odd_count-1)//2:
                    res.append(True)
                else:
                    res.append(False)
        return res
