class Solution {
public:
    int dietPlanPerformance(vector<int>& calories, int k, int lower, int upper) {
        /*calories.clear();
        calories.push_back(6);
        calories.push_back(5);
        calories.push_back(0);
        calories.push_back(0);
        k = 2;
        lower = 1;
        upper = 5;*/
        int result = 0;
        int curCal = 0;
        for (int i=0; i<k; i++)
        {
            curCal += calories[i];
        }
        
        if (curCal < lower)
        {
            result --;
        }
        else if (curCal > upper)
        {
            result ++;
        }
        
        for (int i=k; i < calories.size(); i++)
        {
            curCal = curCal - calories[i-k] + calories[i];
            if (curCal < lower)
            {
                result --;
            }
            else if (curCal > upper)
            {
                result ++;
            }
        }
        return result;
    }
};
