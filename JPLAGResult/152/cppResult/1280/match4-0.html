class Solution {
public:
    int dietPlanPerformance(vector<int>& calories, int k, int lower, int upper) {
        int sum = 0;
        int res = 0;
        for (int i = 0; i < k; i++)
            sum += calories[i];
        if (sum < lower) res--;
        else if (sum > upper) res++;
        for (int i = k; i < calories.size(); i++)
        {
            sum += calories[i];
            sum -= calories[i-k];
            if (sum < lower) res--;
            else if (sum > upper) res++;
        }
        return res;
    }
};
