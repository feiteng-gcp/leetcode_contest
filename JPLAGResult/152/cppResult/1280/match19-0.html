class Solution {
public:
    int dietPlanPerformance(vector<int>& calories, int k, int lower, int upper) {
        int curr = 0;
        int points = 0;
        
        for(int i=0;i<k;i++) {
            curr+=calories[i];
        }
        
        if(curr < lower) points--;
        else if(curr > upper) points++;
        
        for(int i=k;i<calories.size();i++) {
            curr-=calories[i-k];
            curr+=calories[i];

            if(curr < lower) points--;
            else if(curr > upper) points++;
        }
        
        return points;
    }
};
