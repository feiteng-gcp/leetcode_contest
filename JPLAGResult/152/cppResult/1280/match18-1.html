class Solution {
public:
int dietPlanPerformance(vector<int>& c, int k, int lower, int upper) {
    int s = 0;
    int r = 0;
    int n = c.size();
    for (int i = 0; i < n; i++) {
    	s += c[i];
    	if(i >= k) {
    		s -= c[i - k];

    	}
    	if(i >= k - 1) {
    		if(s < lower) r -= 1;
    		else if(s > upper) r += 1;
    	}
    }
    return r;
}
};
