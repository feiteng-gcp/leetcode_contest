class Solution {
public:
    int dietPlanPerformance(vector<int>& calories, int k, int lower, int upper) {
        int s = 0;
        int ans=0;
        for(int i = 0; i < calories.size(); ++i) {
            s+=calories[i];
            if(i >= k) {
                s-=calories[i-k];
            }
            if(i>=k-1) {
                if(s < lower) {
                    ans--;
                } else if(s>upper) {
                    ans++;
                }
            }
        }
        return ans;
    }
};
