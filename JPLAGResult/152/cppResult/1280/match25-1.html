class Solution {
public:
    int dietPlanPerformance(vector<int>& calories, int k, int lower, int upper) {
        int sum=0;
        int ans=0;
        for(int i=0;i<k-1;i++)
            sum+=calories[i];
        int n=calories.size();

        for(int i=k-1;i<n;i++)
        {
            sum+=calories[i];
            if(sum<lower) ans--;
            if(sum>upper) ans++;
            sum-=calories[i-k+1];
        }
        return ans;
    }
};
