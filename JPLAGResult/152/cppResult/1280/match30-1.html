class Solution {
public:
  int dietPlanPerformance(vector<int>& calories, int k, int lower, int upper) {
    int tot = 0;
    vector<int> S{0};
    partial_sum(calories.begin(), calories.end(), back_inserter(S));

    for (int i = k; i < S.size(); i++) {
      int cur = S[i] - S[i - k];
      if (cur < lower) tot--;
      if (cur > upper) tot++;
    }
    return tot;
  }
};
