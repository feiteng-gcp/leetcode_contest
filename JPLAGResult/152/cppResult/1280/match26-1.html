class Solution {
public:
    int A[10]={6,13,8,7,10,1,12,11};

    int dietPlanPerformance(vector<int>& calories, int k, int lower, int upper) {
        int n=calories.size(),ans=0,tot=0;
        for(int i=0;i<k;i++){
            tot+=calories[i];
        }
        if(tot<lower)ans--;
        if(tot>upper)ans++;
        for(int i=k;i<n;i++){
            tot+=calories[i];
            tot-=calories[i-k];
            if(tot<lower)ans--;
        if(tot>upper)ans++;
        }
        return ans;
    }
};
