class Solution {
public:
    int dietPlanPerformance(vector<int>& a, int k, int l, int r) {
        int ans = 0;
        int n = a.size();
        int sum[n+10];
        for(int i = 1; i <= n; ++i){
            sum[i] = sum[i-1]+a[i-1];

        }
        for(int i = 1; i+k-1 <= n; i++){
            int cnt = sum[i+k-1]-sum[i-1];
            if(cnt < l) --ans;
            if(cnt > r) ++ans;
        }
        return ans;
    }
};
