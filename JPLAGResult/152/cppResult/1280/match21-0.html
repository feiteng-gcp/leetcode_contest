class Solution {
public:
    int dietPlanPerformance(vector<int>& a, int k, int lower, int upper) {
        int n = a.size();
        vector<int> s(n+1);
        for (int i = 1; i <= n; ++ i)
            s[i] = s[i-1]+a[i-1];
        int ret = 0;
        for (int i = k; i <= n; ++ i)
        {
            int cur = s[i]-s[i-k];
            if (cur < lower) ret --;
            if (cur > upper) ret ++;
        }
        return ret;
    }
};
