class Solution {
public:
    bool isprime(int a){
        for(int i =2;i<=sqrt(a);i++){
            if(a%i == 0)
                return false;

        }
        return true;
    }
    int getprimesum(int n){
        int ret = 0;
        for(int i =2;i<=n;i++){
            if(isprime(i))
                ret++;
        }
        return ret;
    }
    int numPrimeArrangements(int n) {
        int s1 = getprimesum(n);
        int s2 = n-s1;
        int m = 1000000007;

        long long ret = 1;
        for(int i=s2;i>=1;i--){
            ret *= i;
            ret %= m;
        }
        for(int i=s1;i>=1;i--){
            ret *= i;
            ret %= m;
        }
        return ret%m;
        
    }
};
