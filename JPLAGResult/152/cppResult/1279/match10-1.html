typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:

  inline bool pr(int n){
    if(n==1) return 0;
    for(int d=2;d*d<=n;d++)
      if(n%d==0) return 0;
    return 1;
  }

  int numPrimeArrangements(int n) {
    int p=0;
    for(int k=1;k<=n;k++) if(pr(k)) p++;

    ll res=1;
    for(int k=1;k<=p;k++) res=res*k%MOD;
    for(int k=1;k<=n-p;k++) res=res*k%MOD;
    return res;
  }
};
