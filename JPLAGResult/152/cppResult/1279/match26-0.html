class Solution {
public:
    int numPrimeArrangements(int n) {
        int count=0;
        for(int i=2;i<=n;i++)
        {
            int flag=0;
            for(int j=2;j<=sqrt(i);j++)
            {
                if(i%j==0)
                {
                    flag=-1;
                    break;
                }
            }
            if(flag==0) count++;
        }
       // return count;
        long long ans=1;
        int s=n-count;
        while(count>0)
        {
            ans*=count;
            count--;
            ans%=1000000007;
        }
        while(s>0)
        {
            ans*=s;
            s--;
            ans%=1000000007;
        }
        return ans;
    }
};
