class Solution {
public:
    int mod = 1e9+7;
    long long fact(int n) {
        long long ans = 1;
        for (int i=2; i<=n; i++) ans = (ans * i) % mod;

        return ans;
    }
    int numPrimeArrangements(int n) {
        vector<bool> v(n+1, 1);
        int prime = 0;
        for (int i=2; i<=n; i++) {

            for (int j=2; j*j<=i; j++) {
                if (i % j == 0) {
                    v[i] = 0;
                    break;
                }
            }
            if (v[i] == 1) prime++;
        }
        
        return fact(n-prime) * fact(prime) % mod;
    }
};
