class Solution {
public:
    const int M = 1000000007;
    bool prime(int x) {
        if (x == 1) return false;
        for (int i = 2; i * i <= x; i ++)
            if (x % i == 0) return false;
        return true;
    }
    int numPrimeArrangements(int n) {
        int c = 0;
        long long ans = 1;
        for (int i = 1; i <= n; i ++)
            if (prime(i)) c ++;
        for (int i = 2; i <= c; i ++)
            ans = ans * i % M;
        for (int i = 2; i <= n - c; i ++)
            ans = ans * i % M;
        return (int) ans;
    }
};
