class Solution {
public:
    const int mod = 1000000000+7;
    int numPrimeArrangements(int n) {
        int prime = 0;
        for(int i=2;i<=n;i++){
            bool flag = true;
            for(int j=2;j<i;j++){
                if(j*j>i)
                    break;
                if(i%j == 0){
                    flag = false;
                    break;
                }
            }
            if(flag)
                prime++;
        }
        int rem = n - prime;
        long long ans = 1, temp = 1;
        for(int i=2;i<=rem;i++){
            ans *= i;
            ans %= mod;
        }
        for(int i=2;i<=prime;i++){
            temp *= i;
            temp %= mod;
        }
        ans *= temp;
        ans %= mod;
        return ans;
    }
};
