const int MOD = 1e9 + 7;
using ll = long long;

class Solution {
public:
    int numPrimeArrangements(int n) {

        ll cnt = 0;
        for (int i = 2; i <= n; ++ i) {
            bool flag = true;
            for (int j = 2; j < i; ++ j) if (i % j == 0) {
                flag = false;
                break;
            }
            cnt += flag ? 1 : 0;
        }
        

        ll ans = 1;
        for (int i = 1; i <= cnt; ++ i) ans = ans * i % MOD;
        for (int i = 1; i <= n - cnt; ++ i) ans = ans * i % MOD;
        return ans;
    }
};
