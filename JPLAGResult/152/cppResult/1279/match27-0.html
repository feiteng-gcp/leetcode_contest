#define FOR(x,n) for(int x=0;x<n;x++)
typedef long long ll;

class Solution {
public:
    ll big = 1e9+7;
    int ct(int n) {
        int res = 1;
        for(int i = 3; i <= n; i++) {
            bool ok = true;
            for(int j = 2; j < i; j++) {
                if(i % j == 0) ok = false;
            }
            if(ok) res++;
        }
        return res;
    }
    
    int numPrimeArrangements(int n) {
        ll primes = ct(n);
        int ret = 1;
        for(int i = 1; i <= primes; i++) ret = (1ll*ret*i) % big;
        for(int i = 1; i <= n - primes; i++) ret = (1ll*ret*i) % big;
        return ret;
    }
};
