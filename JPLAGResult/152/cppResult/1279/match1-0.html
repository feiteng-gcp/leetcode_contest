class Solution {
public:
    int mod = 1e9+7;
    int jc[105];
    int numPrimeArrangements(int n) {
        jc[0]=1;
        for(int i=1;i<=n;i++)jc[i]=(1ll*jc[i-1]*i)%mod;
        int cnt=0;
        for(int i=2;i<=n;i++){
            bool flag=1;
            for(int j=2;j<i;j++){
                if(i%j==0){flag=0;break;}
            }
            if(flag)cnt++;
        }
        int ans=(1ll*jc[n-cnt]*jc[cnt])%mod;
        return ans;
    }
};
