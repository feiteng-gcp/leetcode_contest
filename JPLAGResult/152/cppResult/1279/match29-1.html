class Solution {
public:
    int p[101],m=0;
    int numPrimeArrangements(int n) {
        for(int i=2;i<=n;++i){

            bool f = true;
            for(int j=0;j<m;++j){
                if(i%p[j]==0){
                    f = false;
                    break;
                }
            }
            if(f) p[m++] = i;
        }

        long long ans = 1;
        for(int i=1;i<=m;++i){
            ans = (ans * i) % 1000000007;
        }
        for(int i=1;i<=n-m;++i){
            ans = (ans * i) % 1000000007;
        }
        return ans;
    }
};
