class Solution {
public:
    
    int mul(int a, int b, int m) {
        int ans = 0;
        while (b) {
            if (b & 1) {
                ans = (ans + a) % m;
            }
            b >>= 1;
            a = (a + a) % m;
        }
        return ans;
    }
    
    int numPrimeArrangements(int n) {
        bool isPrime[n + 1];
        memset(isPrime, true, sizeof(isPrime));
        isPrime[0] = isPrime[1] = false;
        for (int i = 2; i <= n; i ++) {
            if (isPrime[i]) {
                for (int j = i + i; j <= n; j += i) {
                    isPrime[j] = false;   
                }
            }
        }
        int num_prime = 0;
        for (int i = 2; i <= n; i ++) {
            if (isPrime[i]) {
                num_prime ++;
            }
        }
        int ans = 1;
        cout << num_prime;
        for (int i = 2; i <= num_prime; i ++) {
            ans = mul(ans, i, 1000000007);
        }
        for (int i = 2; i <= n - num_prime; i ++) {
            ans = mul(ans, i, 1000000007);
        }
        return ans;
    }
};
