#include <bits/stdc++.h>
using namespace std;


typedef long long ll;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;

const ll MOD = 1e9 + 7;


class Solution {
public:
	bool isprime(ll a){
		for(ll i=2; i<a; i++){
			if(a % i == 0) return false;
		}
		return true;
	}

	ll nf(ll a){
		ll r = 1;
		for(ll i=1; i<=a; i++){
			r *= i;
			r %= MOD;
		}
		return r;
	}

    int numPrimeArrangements(ll n) {

		ll np = 0;
		for(ll i=2; i<=n; i++){
			if(isprime(i)) np++;
		}

        cout<<np<<endl;
		return (nf(np) * nf(n - np)) % MOD;

    }
};
