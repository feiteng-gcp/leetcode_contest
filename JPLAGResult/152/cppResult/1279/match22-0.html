class Solution {
public:
    int numPrimeArrangements(int n) {
        int p=0,np=0;
        for(int i=1;i<=n;i++)
        {
            if(isp(i))p++;
            else np++;
        }
        //cout<<p<<" "<<np<<endl;
        return A(p)*A(np)%md;
    }
    
    bool isp(int n){
        if(n==1)return false;
        //cout<<n<<endl;
        for(int i=2;i<=n/2;i++){
            //cout<<n<<" "<<i<<" "<<(n%i==0)<<endl;
            if(n%i==0)return false;
        }
        return true;
    }
    
    int md=(1e9)+7;
    
    long long A(int n){
        if(n==0)return 1;
        if(n==1)return 1;
        return A(n-1)*n%md;
    }
};
