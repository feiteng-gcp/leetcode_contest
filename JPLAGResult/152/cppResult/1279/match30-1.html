class Solution {
public:
    const long long M = 1e9 + 7;
    

    bool is_prime(int x) {
        if (x == 1) return false;
        for (int i = 2; i <= sqrt(x); i++) {
            if (x % i == 0) return false;
        }
        return true;
    }
    
    long long func(int n) {
        long long res = 1;
        for (int i = 1; i <= n; i++) {
            res = (res * i) % M;
        }
        return res;
    }
    

    int numPrimeArrangements(int n) {
        int cnt_prime = 0;
        for (int i = 1; i <= n; i++) {
            if (is_prime(i)) cnt_prime++;
        }
        return (func(cnt_prime) * func(n - cnt_prime)) % M;
    }
};
