class Solution {
public:

    int M = 1000000000 + 7;
    bool isPrime(int n){
        if(n == 2)return true;
        for(int i = 2;i * i <= n;i++){
            if(n % i == 0)return false;
        }
        return true;
    }
    long jiecheng(int n){

        long res = 1;
        for(int i = 1;i <= n;i++){
            res = (res * i) % M;
        }
        return res;
    }
    int numPrimeArrangements(int n) {
        int cnt = 0;
        for(int i = 2;i <= n;i++){
            if(isPrime(i))cnt++;
        }
        return (jiecheng(cnt) * jiecheng(n - cnt)) % M;
        
    }
};
