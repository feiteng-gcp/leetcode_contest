class Solution {
public:
  const int MOD = 1e9+7;
  int numPrimeArrangements(int n) {
    vector<int> primes; 
    vector<bool> isPrime(n+1, true);
    isPrime[0] = isPrime[1] = false;
    for (int p = 2; p <= n; p++) {
      if (isPrime[p]) primes.push_back(p);
      for (auto t: primes) {
        int s = t * p;
        if (s > n) break;
        isPrime[s] = false;
        if (p%t == 0) break;
      }
    }
    int p = primes.size();
    long long s = 1;
    for (int i = 1; i <= p; i++) s = s * i % MOD;
    for (int i = 1; i <= n-p; i++) s = s * i % MOD;
    return s;
  }
};
