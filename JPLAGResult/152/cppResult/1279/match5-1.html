class Solution {
public:
    const int MOD = 1e9 + 7;
    int numPrimeArrangements(int n) {
        int num = 0;
        for (int i = 2; i <= n; ++i)
        {

            int flag = 1;
            for (int j = 2; j * j <= i; j++)
                if (i % j == 0)
                {
                    flag = 0;
                    break;
                }
            if (flag == 1) num++;
        }
        //printf("%d\n", num);
        long long res = 1;
        long long res2 = 1;

        for (int i = 1; i <= num; i++)
            res = (res * i) % MOD;
        for (int i = 1; i <= n - num; i++)
            res2 = (res2 * i) % MOD;
        return (res * res2) % MOD;
    }
};
