class Solution {
public:
    bool isPrime(int n){
        if(n == 1)
            return false;
        if(n == 2 || n == 3)
            return true;

        for(int i = 2;i < n; ++i){
            if(n % i == 0){
                //cout << n << i << endl;
                return false;
            }
        }
        return true;
    }
    int numPrimeArrangements(int n) {
        int primeCnt = 0;
        for(int i = 1;i <= n; ++i){
            if(isPrime(i))
                primeCnt += 1;
        }

        long long result = 1;
        for(int i = 1;i <= primeCnt; ++i){
            result *= i;
            result = result % (1000000000 + 7);
        }
        for(int i = 1;i <= (n-primeCnt); ++i){
            result *= i;
            result = result % (1000000000 + 7);
        }
        return result;
    }
};
