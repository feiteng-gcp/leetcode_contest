class Solution {
public:
    bool isZhishu(int a) {
        if (a<=1) return false;
        for (int i=2; i<a; i++) {
            if (a % i == 0) return false;
        }
        return true;
    }
    int numPrimeArrangements(int n) {
        int cnt = 0;
        for (int i=1;i<=n;i++) {
            if (isZhishu(i)) cnt++;
        }
        long long res = 1;
        int m = 1000000007;
        for(int i=1;i<=cnt;i++) {
            res = res*i % m;
        }
        for(int i=1;i<=n-cnt; i++) {
            res = res*i % m;
        }
        return res;
    }
};
