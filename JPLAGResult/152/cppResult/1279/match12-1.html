class Solution {
    bool ifprime(int a){
        bool flag = true;

        if(a == 1) return false;
        if(a < 4) return true;
        for(int i = 2; i < a; i++){
            if(a % i == 0){
                return false;
            }
        }
        return true;
    }
public:

    int numPrimeArrangements(int n) {
        int nums = 0;
        for(int i = 1; i <= n; i++){
            if(ifprime(i)) nums++;
        }
        long long int ans = 1;

        for(int i = 1; i <= nums; i++){
            ans *= i;
            ans = ans % 1000000007;
        }
        for(int i = 1; i <= n - nums; i++){
            ans *= i;
            ans = ans % 1000000007;
        }
        return ans;
    }
};
