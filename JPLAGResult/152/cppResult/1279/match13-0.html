class Solution {
public:
    bool is_prime(int num) {
        if (num <= 1) {
            return false;
        } else if (num == 2 || num == 3) {
            return true;
        }

        if ((num & 1) == 0 || num % 3 == 0) {
            return false;
        }

        for (int i = 5; i * i <= num; i += 6) {
            if (num % i == 0 || num % (i + 2) == 0) {
                return false;
            }
        }
        return true;
    }
    int countPrimes(int n) {
        int count = 0;
        for (int i = 0; i < n; ++i) {
            if (is_prime(i)) {
                count++;
            }
        }
        return count;
    }
    int const m = 1000000007;
    int numPrimeArrangements(int n) {
        int num_prime = countPrimes(n + 1);
        int num_not_prime = n - num_prime;
        long result = 1;
        for (int i = 2; i <= num_prime; ++i) {
            result *= i;
            result %= m;
        }
        for (int i = 2; i <= num_not_prime; ++i) {
            result *= i;
            result %= m;
        }
        return result;
    }
};
