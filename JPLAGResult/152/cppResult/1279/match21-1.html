class Solution {
public:
    int numPrimeArrangements(int n) {
        vector<int> prime(n + 1, 0);

        for(int i = 2; i <= n; i++) {
            if(prime[i] == 0) {
                for(int j = i + i; j <= n; j += i) {
                    prime[j] = 1;
                }
            }
        }
        prime[1] = 1;

        int cnt = 0;
        for(int i = 1; i <= n; i++) {
            if(prime[i] == 0) {
                cnt++;
            }
        }
        long long ans = 1;

        long long mod = 1e9 + 7;
        for(int i = 1; i <= n - cnt; i++) {
            ans = ans * i % mod; 
        }
        for(int i = 1; i <= cnt; i++) {
            ans = ans * i % mod;

        }
        return ans;
    }
};
