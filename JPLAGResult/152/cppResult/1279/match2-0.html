class Solution {
public:
    int numPrimeArrangements(int n) {
        long long int ans = 1, i, j, c = 0;
        for (i = 2; i <= n; i++) {
            for (j = 2; j <= sqrt(i); j++) {
                if (i % j == 0) {
                    break;
                }
            }
            if (i % j != 0 || i == j) {
                //cout << i << endl;
                c++;
            }
        }
        //cout << c << endl;
        for (i = 2; i <= c; i++) {
            ans *= i;
            ans %= 1000000007;
        }
        for (i = 2; i <= n - c; i++) {
            ans *= i;
            ans %= 1000000007;
        }
        return ans;
    }
};
