class Solution {
public:
    const int MOD = 1e9 + 7;
    bool is_prime(int x) {
        if (x <= 1) return false;
        for (int i = 2; i * i <= x; i++) if (x % i == 0) return false;
        return true;
    }
    
    long long cal(int x) {
        long long ret = 1;
        for (long long i = 1; i <= x; i++) ret = (ret * i) % MOD;
        return ret;
    }

    int numPrimeArrangements(int n) {
        int c = 0;
        for (int i = 1; i <= n; i++) if (!is_prime(i)) c++;
        return cal(c) * cal(n-c) % MOD;
    }
};
