const int mod = (int) 1e9 + 7;

class Solution {
public:
    bool isPrime(int x) {
        if (x <= 1) return false;

        for (int i=2;i<x;i++) {
            if (x%i == 0) return false;
        }
        return true;
    }
    int numPrimeArrangements(int n) {
        int cnt = 0;
        for (int i=1;i<=n;i++) {
            if (isPrime(i)) {
                cnt++;
            }
        }    

        int rest = n - cnt;
        int ans = 1;
        for (int i=1;i<=cnt;i++) {
            ans = ans *1ll* i % mod;
        }
        
        for (int i=1;i<=rest;i++) {
            ans = ans * 1ll * i % mod;
        }
        return ans;
        
    }
};
