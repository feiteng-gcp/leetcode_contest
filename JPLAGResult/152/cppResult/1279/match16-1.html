class Solution {
public:
    bool ok(int x){

        for(int i = 2; i < x; ++i){
            if(x % i == 0) return false;
        }
        return true;
    }
    int numPrimeArrangements(int n) {
        int cnt = 0 ;
        for(int i = 2; i <= n; ++i){
            if(ok(i)) cnt++;
        }
        long long t = 1, t1 = 1;

        int mod = 1e9 +7;
        for(int i = 1; i <= cnt; ++i){
            t = t * i % mod;
        }
        for(int i = 1; i <= n - cnt; ++i)
            t1 = t1 * i % mod;
        return t*t1%mod;
    }
};
