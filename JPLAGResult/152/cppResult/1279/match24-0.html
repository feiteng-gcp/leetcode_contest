class Solution {
    bool isprime(int n){
        for(int i=2;i*i<=n;++i){
            if(n%i==0) return false;
        }
        return true;
    }
public:
    int numPrimeArrangements(int n) {
        int prime=0, mod=1e9+7;
        for(int i=2;i<=n;++i)
            if(isprime(i)) ++prime;
        int ans=1;
        for(int i=1;i<=prime;++i) ans=(1LL*ans*i)%mod;
        for(int i=1;i<=n-prime;++i) ans=(ans*1LL*i)%mod;
        return ans;
    }
};
