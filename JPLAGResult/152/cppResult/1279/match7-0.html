class Solution {
public:
    const int mo = 1e9 + 7;
    
    int numPrimeArrangements(int n) {
        int p = 0;
        for (int i = 2; i <= n; ++i) {
            bool ok = true;
            for (int j = 2; j < i; ++j)
                if (i % j == 0) {
                    ok = false;
                    break;
                }
            
            if (ok)
                ++p;
        }
        
        long long ans = 1;
        for (int i = 2; i <= p; ++i)
            ans = ans * i % mo;
        
        for (int i = 2; i <= n-p; ++i)
            ans = ans * i % mo;
        
        return ans;
    }
};
