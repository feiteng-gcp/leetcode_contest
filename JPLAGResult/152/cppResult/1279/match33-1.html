class Solution {
public:

    int numPrimeArrangements(int n) {
        int prime = 0;
        for(int i = 2; i <= n; i++){
            bool p = true;
            for(int j = 2; j*j <= i; j++)   if(i%j == 0)    p = false;
            prime += p;
        }

        long long ans = 1;
        for(int i = 1; i <= prime; i++){
            ans *= i;
            ans %= 1000000007;
        }
        for(int i = 1; i <= n-prime; i++){
            ans *= i;
            ans %= 1000000007;
        }
        return (int)(ans);
    }
};
