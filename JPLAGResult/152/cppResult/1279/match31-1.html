class Solution {
public:
    long long gen(int i) {
        if (i <= 1) return 1;
        return i * gen(i - 1) % 1000000007;
    }

    int numPrimeArrangements(int n) {
        int cnt = 0;
        for (int i = 2; i <= n; i++) {
            bool flag = true;
            for (int j = 2; j < i; j++) {
                if (i % j == 0) {
                    flag = false;
                    break;
                }
            }
            if (flag) cnt++;
        }
        return gen(cnt) * gen(n - cnt) % 1000000007;
    }
};
