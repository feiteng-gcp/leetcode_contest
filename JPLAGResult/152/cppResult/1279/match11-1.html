#define MOD 1000000007
class Solution {
public:

    bool prime(int n)
    {
        if (n == 1)return false;
        for (int i = 2; i <= n/2; ++i)
            if (n%i == 0)
                return false;
        return true;
    }
    int numPrimeArrangements(int n) {
        int prime_count = 0;
        for (int i = 1;i <= n; ++i)
            if (prime(i))
                prime_count++;
        cout<<prime_count;

        long long ans = 1;
        for (int i = prime_count; i > 1; --i)
            ans = (ans % MOD * i);
        for (int i = n - prime_count; i > 1; --i)
            ans = (ans % MOD * i);
        return ans % MOD;
    }
};
