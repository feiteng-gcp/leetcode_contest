class Solution {
public:        
    int numPrimeArrangements(int n) {
        vector <bool> b(n+1, 0);
        vector <int> primes;        
        for (int i = 2; i < n+1; i++) {
            if (b[i] == 0) {
                for (int j = i; j < n+1; j+=i) {
                    b[j] = 1; 
                }
                primes.push_back(i);
            }
        }
        
        for (int i = 0; i < primes.size(); i++) 
            cout << primes[i] << endl;
        
        long long perm1=1, perm2=1; 

        long mod_val = 1e9 + 7;
        for (int i = 1; i <= n - primes.size(); i++) {
            perm1 = (perm1 * i) % mod_val;
        }
        for (int i = 1; i <= primes.size(); i++) {
            perm2 = (perm2 * i) % mod_val;
        }        
        long ans = (perm1*perm2) % mod_val;
        return ans;
        
    }
};
