class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector<array<int,26>> cnt(s.size()+1);
        cnt[0]={0};
        for(int i=0;i<s.size();i++){
            cnt[i+1]=cnt[i];
            cnt[i+1][s[i]-'a']++;

        }
        vector<bool> ans;
        for(auto &q:queries){
            int odd=0;
            for(int i=0;i<26;i++)
                if((cnt[q[1]+1][i]-cnt[q[0]][i])%2==1)odd++;
            ans.push_back(q[2]*2+1>=odd);
        }
        return ans;
    }
};
