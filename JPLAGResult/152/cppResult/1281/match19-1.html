class Solution {
public:
    int get_cnt(int l, int r, vector <vector<int> > &sum, int c) {
        return (sum[r + 1][c] - sum[l][c]) % 2;
    }
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector <bool> ans;
        int n = (int)s.size();

        vector <vector<int> > sum(n + 1, vector<int>(26, 0));
        for (int i = 0; i < 26; i++) {
            for (int j = 1; j <= n; j++) {
                sum[j][i] = sum[j - 1][i] + (s[j - 1] - 'a' == i);
            }

        }
        for (auto qr: queries) {
            int l = qr[0];
            int r = qr[1];
            int k = qr[2];
            int cnt_odd = 0;
            for (int i = 0; i < 26; i++) {
                cnt_odd += get_cnt(l, r, sum, i);
            }
            ans.emplace_back(k >= cnt_odd / 2);
        }
        return ans;
    }
};
