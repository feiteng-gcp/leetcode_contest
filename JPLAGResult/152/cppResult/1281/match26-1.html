class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int sum = 0;
        vector<int> prefix(s.length());
        for (int i = 0; i < s.length(); ++i) {
            sum ^= (1 << (s[i] - 'a'));
            prefix[i] = sum;
        }
        vector<bool> ret;
        for (auto& query : queries) {
            int s = prefix[query[1]];
            if (query[0]) s ^= prefix[query[0] - 1];
            int cnt = 0;
            for (int i = 0; i < 26; ++i) cnt += (s & (1 << i)) > 0;
            ret.push_back((cnt / 2) <= query[2]);
        }
        return ret;
    }
};
