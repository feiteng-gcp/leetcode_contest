class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector<vector<int>> cnt(s.size() + 1, vector<int>(26));
        for (int i = 0; i < s.size(); ++i) {

            for (int j = 0; j < 26; ++j) {
                cnt[i+1][j] = cnt[i][j] + (s[i] == 'a' + j);
            }
        }
        
        vector<bool> ans;
        for (auto q: queries) {
            int c = 0;
            for (int i = 0; i < 26; ++i) {
                int cnt_i = cnt[q[1] + 1][i] - cnt[q[0]][i];
                c += (cnt_i % 2);
            }
            int need = c / 2;
            ans.push_back(need <= q[2]);
        }
        return ans;
    }
};
