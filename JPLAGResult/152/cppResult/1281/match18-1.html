class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int N=s.size();
        vector<vector<int>> sums(26,vector<int>(N+1,0));
        for(int i=0;i<26;i++){
            for(int j=0;j<N;j++){
                if(s[j]==i+'a'){
                    sums[i][j+1]=sums[i][j]+1;
                }else sums[i][j+1]=sums[i][j];
            }
        }
        
        vector<bool> ans;
        for(int i=0;i<queries.size();i++){
            auto p=queries[i];

            int s=p[0];
            int e=p[1];
            int cnt=e-s+1;
            int odd_cnt=0;
            for(int j=0;j<26;j++){
                int a=sums[j][e+1]-sums[j][s];
                if(a%2==1) odd_cnt++;
            }
            
            bool res=false;
            if(cnt%2==0){
                if(p[2]*2>=odd_cnt&&(odd_cnt%2==0)) res=true;
            }else{
                if(p[2]*2>=odd_cnt-1&&(odd_cnt%2==1)) res=true;
            }
            
            ans.push_back(res);
        }
        return ans;
    }
};
