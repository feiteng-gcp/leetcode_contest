class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
      int n = s.length();

      vector<vector<int> > prefixSum(n + 1, vector<int>(26, 0));
      for (int i = 0; i < n; i++) {
        prefixSum[i + 1][s[i] - 'a']++;
      }
      for (int i = 0; i < n; i++) for (int j = 0; j < 26; j++) prefixSum[i + 1][j] += prefixSum[i][j];

      int low, upp, k, cntOdd, m = queries.size();
      vector<bool> res(m);
      for (int i = 0; i < m; i++) {
        vector<int> &q = queries[i];
        low = q[0], upp = q[1] + 1, k = q[2], cntOdd = 0;
        for (int j = 0; j < 26; j++) {
          cntOdd += (prefixSum[upp][j] - prefixSum[low][j]) % 2;
        }
        // if ((upp - low) % 2) cntOdd--;
        res[i] = (k >= cntOdd / 2);
      }
      return res;
    }
};
