class Solution {
public:
    int range_sum(int left, int right, int ch, const vector<vector<int>>& prefix_sum) {
        left++, right++;
        return prefix_sum[right][ch] - prefix_sum[left - 1][ch];
    }
    
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int n = s.size();
        vector<vector<int>> prefix_sum(n + 1, vector<int>(26, 0));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < 26; j++) prefix_sum[i + 1][j] = prefix_sum[i][j];
            prefix_sum[i + 1][s[i] - 'a']++;
        }
        
        // for (int ch = 0; ch < 26; ch++) cout << prefix_sum[n][ch] << " ";
        // cout << endl;
        
        vector<bool> res;
        for (auto& query : queries) {
            int left = query[0], right = query[1], k = query[2];
            int odd_cnt = 0;
            for (int ch = 0; ch < 26; ch++) {
                if (range_sum(left, right, ch, prefix_sum) % 2 == 1) odd_cnt++;
            }
            // cout << "odd_cnt = " << odd_cnt << endl;
            odd_cnt -= (2 * k);
            if (odd_cnt <= 1) res.push_back(true);
            else res.push_back(false);
        }
        
        return res;
    }
};
