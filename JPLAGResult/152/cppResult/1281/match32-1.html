class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector<int> color_cnts[26];
        int len = s.length(); 
        for (int i = 0; i < 26; i++) {
            int sum = 0;
            color_cnts[i].push_back(0);
            for (int j = 0; j < len; j++) {
                if (s[j] - 'a' == i) sum ++;
                color_cnts[i].push_back(sum);
            }
        }
        vector<bool> ret;
        for (int i = 0; i < queries.size(); i++) {
            int l = queries[i][0] + 1;
            int r = queries[i][1] + 1;

            int k = queries[i][2];
            int times[26];
            int odd_cnt = 0;
            for (int j = 0; j < 26; j++) {
                times[j] = color_cnts[j][r] - color_cnts[j][l - 1];
                if (times[j] & 1) odd_cnt++;
            }
            if (odd_cnt <= 1) ret.push_back(true);
            else {
                if (odd_cnt / 2 <= k) ret.push_back(true);
                else ret.push_back(false);
            }
        }  
        return ret;
    }
};
