class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int n = s.size();
        vector<array<int, 26>> F(n+1);
        for (int i = 1; i <= n; ++ i)
        {
            F[i] = F[i-1];
            F[i][s[i-1]-'a'] ++;
        }
        vector<bool> ret;
        for (auto &e : queries)
        {
            int x = e[0]+1, y = e[1]+1, k = e[2];
            array<int, 26> a;
            int cnt = 0;
            for (int i = 0; i < 26; ++ i)
            {
                a[i] = F[y][i]-F[x-1][i];
                cnt += a[i]&1;
            }
            if ((y-x+1)&1) cnt --;
            ret.push_back(cnt/2 <= k);
        }
        return ret;
    }
};
