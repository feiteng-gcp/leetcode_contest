class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector<vector<int>> prefix_cnts(s.size(), vector<int>(26, 0));
        
        prefix_cnts[0][s[0]-'a']++;
        for (int i = 1; i < s.size(); i++)
        {
            prefix_cnts[i] = prefix_cnts[i-1];
            prefix_cnts[i][s[i]-'a']++;
        }
        
        vector<bool> ans(queries.size());
        for (int qi = 0; qi < queries.size(); qi++)
        {
            const auto &q = queries[qi];
            vector<int> cnts(26);
            int num_odd_cnts = 0;
            for (int i = 0; i < 26; i++)
            {
                cnts[i] = prefix_cnts[q[1]][i];
                if (q[0] > 0)
                    cnts[i] -= prefix_cnts[q[0]-1][i];
                
                if (cnts[i] % 2)
                    num_odd_cnts++;
            }
            
            ans[qi] = (num_odd_cnts / 2) <= q[2];
        }
            
        return ans;
    }
};
