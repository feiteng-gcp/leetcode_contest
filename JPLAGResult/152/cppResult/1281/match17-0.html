class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector<bool> ans;
        
        int count[100005][26] = {0};
        for (int i=0; i<s.length(); i++) {
            if (i==0) {
                count[i+1][s[i]-'a']++;
            }
            else {
                for (int j=0; j<26; j++)
                    count[i+1][j] = count[i][j];
                
                count[i+1][s[i]-'a']++;
            }
        }        
        
        for (auto v : queries) {
            int diff = 0;
            for (int i=0; i<26; i++)
                if ((count[v[1]+1][i] - count[v[0]][i]) % 2 == 1)
                    diff++;
            if ((v[1]-v[0]+1) % 2 == 1)
                diff--;
            
            ans.push_back(diff <= v[2]*2);
        }
        
        return ans;
    }
};
