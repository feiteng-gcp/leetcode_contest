class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int n = s.size();
        vector<vector<int>> g(n+1, vector<int>(26));
        for(int i = 0; i < n; i++){
            g[i+1][s[i]-'a']++;
        }
        for(int i = 1; i <= n; i++){
            for(int j = 0; j < 26; j++){
                g[i][j] += g[i-1][j];
            }
        }
        vector<bool> ans;
        for(auto x : queries){
            int l = x[0], r = x[1] + 1, k = x[2];
            int cnt = 0;
            for(int j = 0; j < 26; j++){
                cnt += (g[r][j] - g[l][j]) & 1;
            }
            if(k * 2 + 1 >= cnt) ans.push_back(true);
            else ans.push_back(false);
        }
        return ans;
    }
};
