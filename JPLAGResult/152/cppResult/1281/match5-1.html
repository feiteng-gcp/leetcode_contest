class Solution {
public:
    int pre[100100][26],A[26];
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int n=s.size();
        for(int i=1;i<=n;i++){
            for(int j=0;j<26;j++)pre[i][j]=pre[i-1][j];
            pre[i][s[i-1]-'a']++;
        }
        vector<bool> ans;
        for(int i=0;i<queries.size();i++){
            int l=queries[i][0]+1,r=queries[i][1]+1,k=queries[i][2],cnt=0;
            int len=r-l+1,even=0,odd=0;
            for(int j=0;j<26;j++){
                A[j]=pre[r][j]-pre[l-1][j];
                if(A[j]>0){
                    cnt++;
                    if(A[j]&1)odd++;
                    else even++;
                }
            }
            if(len&1){
                if(odd/2>k)ans.push_back(false);
                else ans.push_back(true);
            }else{
                if(odd/2>k)ans.push_back(false);
                else ans.push_back(true);
            }
        }
        return ans;
    }
};
