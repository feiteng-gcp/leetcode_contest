class Solution {
public:
  vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
    int n = s.size();
    vector<vector<int> > cnt(n+1, vector<int>(26));
    for (int i = 0; i < s.size(); i++) {
      cnt[i+1] = cnt[i];
      cnt[i+1][s[i]-'a']++;
    }
    vector<bool> ans;
    for (auto q: queries) {
      int l = q[0], r = q[1], k = q[2];
      vector<int> R = cnt[r+1];
      vector<int> L = cnt[l];
      int c = 0;
      for (int i = 0; i < 26; i++) if (abs(R[i] - L[i])%2) c++;
      if (c - 2 * k > 1) ans.push_back(false);
      else ans.push_back(true);
    }
    return ans;
  }
};
