class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector<bool> ans;
        int i, j, k, n, t, l;
        l = s.size();
        vector<vector<int>> count(l + 1, vector<int>(26, 0));
        for (i = 1; i <= l; i++) {
            count[i] = count[i - 1];
            count[i][s[i - 1] -'a'] += 1;
        }
        n = queries.size();
        for (k = 0; k < n; k++) {
            t = 0;
            for (i = 0; i < 26; i++) {
                if ((count[queries[k][1] + 1][i] - count[queries[k][0]][i]) & 1) {
                    t++;
                }
            }
            //cout << t << " " << queries[k][2] << endl;
            if (t - 1 > queries[k][2] * 2) {
                ans.push_back(false);
            } else {
                ans.push_back(true);
            }
        }
        return ans;
    }
};
