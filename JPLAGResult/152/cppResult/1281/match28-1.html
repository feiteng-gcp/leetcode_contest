class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int len=s.size();

        vector<vector<int>>help(len+1,vector<int>(26,0));
        vector<int>cur(26,0);
        for(int i=0;i<len;i++)
        {
            cur[s[i]-'a']++;
            help[i+1]=cur;
        }
        vector<bool>res;
        for(auto t:queries)
        {
            int left=t[0],right=t[1],k=t[2];
            if(check(help[left],help[right+1],k))
                res.push_back(true);
            else
                res.push_back(false);
        }
        return res;
    }

    bool check(vector<int>&a,vector<int>&b,int k){
        int odd=0;
        for(int i=0;i<26;i++)
        {
            if((b[i]-a[i])%2==1)
                odd++;
        }
        return odd/2<=k;
    }
};
