class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector<bool> ans;
        vector<vector<int>> v;
        vector<int> cnt(26, 0);
        v.push_back(cnt);
        for (int i = 0 ; i < s.size(); i ++) {
            cnt[s[i] - 'a'] ++;
            v.push_back(cnt);
        }
        
        for (int i = 0; i < queries.size(); i ++) {
            int l = queries[i][0], r = queries[i][1], k = queries[i][2];
            int odd_cnt = 0;
            for (int j = 0; j < 26; j ++) {
                int n = v[r + 1][j] - v[l][j];
                if (n % 2 == 1) {
                    odd_cnt ++;
                }
            }
            if (odd_cnt / 2 <= k) {
                ans.push_back(true);
            } else {
                ans.push_back(false);
            }
        }
        return ans;
    }
};
