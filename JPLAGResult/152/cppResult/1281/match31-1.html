class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int cnt[100005][30];

        memset(cnt, 0, sizeof(cnt));
        int l = s.length();
        for (int i=1; i<=l; i++){
            for (int j=0; j<26; j++) cnt[i][j] = cnt[i-1][j];
            int pos = s[i-1]-'a';
            cnt[i][pos]++;
        }
        vector<bool> ret;
        int n = queries.size();
        for (int i=0; i<n; i++){
            vector<int> tmp = queries[i];
            int left = tmp[0], r = tmp[1], k = tmp[2];

            int l1 = r-left+1;
            r++;
            int odd = 0;
            for (int j=0; j<26; j++){
                if ((cnt[r][j] - cnt[left][j])%2) odd++;
            }
            if (l1%2) odd--;
            if (odd/2 > k) ret.push_back(false);
            else ret.push_back(true);
        }
        return ret;
    }
};
