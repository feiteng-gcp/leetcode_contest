class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& q) {
        vector<int> a[27];
        int n = s.size();
        for(int i=0;i<26;i++)
            a[i].resize(n+1);
        for(int i=0;i<n;i++)
        {
            int x = s[i] - 'a';
            a[x][i+1] = 1;
        }
        for(int i=0;i<26;i++)
        {
            for(int j=1;j<=n;j++)
            {
                a[i][j] += a[i][j-1];
            }
        }
        vector<bool> ans;
        for(auto e:q)
        {
            int L = e[0],R = e[1],k = e[2];
            int cnt = 0;
            for(int i=0;i<26;i++)
            {
                if((a[i][R+1] - a[i][L])%2 == 1)
                    cnt++;
                // if(L==0 && R == 4)
                // printf("%d\n",a[i][R+1] - a[i][L]);
            }
            if((cnt/2)>k)
                ans.push_back(false);
            else
                ans.push_back(true);
        }
        return ans;
    }
};
