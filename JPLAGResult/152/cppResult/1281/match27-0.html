class Solution {
public:
    int pre[100005][26];
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        memset(pre, 0, sizeof(pre));
        int n = s.size();
        vector<bool> ret;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < 26; j++) pre[i][j] = pre[i-1][j];
            pre[i][s[i-1]-'a']++;
        }
        for (auto q : queries) {
            q[0]++; q[1]++;
            int x = 0;
            for (int i = 0; i < 26; i++) {
                if ((pre[q[1]][i] - pre[q[0]-1][i]) & 1) x++;
            }
            ret.push_back(x / 2 <= q[2]);
        }
        return ret;
    }
};
