class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int n = queries.size();
        vector<bool> ans(n, false);
        
        int m = s.length();
        vector<vector<int>> cnt(m, vector<int>(26, 0));
        
        for (int i = 0; i < m; ++ i) {
            for (int j = 0; j < 26; ++ j) if (i > 0) cnt[i][j] = cnt[i - 1][j];
            ++ cnt[i][s[i] - 'a'];
        }
        
        for (int i = 0; i < n; ++ i) {
            int l = queries[i][0], r = queries[i][1], k = queries[i][2];
            int cc = 0;
            for (int j = 0; j < 26; ++ j) {
                cc += (cnt[r][j] - (l == 0 ? 0 : cnt[l - 1][j])) % 2;
            }
            if (cc % 2 == 1) -- cc;
            if (cc / 2 <= k) ans[i] = true;
        }
        
        return ans;
    }
};
