class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int n = s.length();
        vector<vector<int>> v(n+1, vector<int>(26, 0));
        for (int i=0; i<n; i++) {
            v[i+1] = v[i];
            v[i+1][s[i]-'a']++;
        }
        vector<bool> ans;
        for (auto& query: queries) {
            int odd = 0;
            for (int i=0; i<26; i++) {
                if ((v[query[1]+1][i] - v[query[0]][i]) % 2) odd++;
            }
            ans.push_back(odd / 2 <= query[2]);
        }
        return ans;
    }
};
