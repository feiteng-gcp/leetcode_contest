class Solution {
public:
  vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
    int m = queries.size();
    vector<bool> ret(m);
    int n = s.size();
    vector<vector<int>> c(26, vector<int>(n + 1));
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < 26; ++j) {
        c[j][i + 1] = c[j][i];
      }

      ++c[s[i] - 'a'][i + 1];
    }
    for (int i = 0; i < m; ++i) {
      int l = queries[i][0], r = queries[i][1], k = queries[i][2];
      int o = 0;
      for (int j = 0; j < 26; ++j) {
        int x = c[j][r + 1] - c[j][l];
        if (x % 2) {
          ++o;
        }
      }
      if (o <= k * 2 + 1) {
        ret[i] = true;
      }
    }
    return ret;
  }
};
