class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int n = s.size();
        vector<vector<int>> counter(n + 1, vector<int> (26, 0));
        for (int i = 0; i < n; ++i) {
            counter[i + 1] = counter[i];
            counter[i + 1][s[i] - 'a']++;
        }
        vector<bool> res;
        for (auto& q:queries) {
            int l = q[0], r = q[1], k = q[2];
            int diff_odd = 0;
            for (int i = 0; i < 26; ++i) {
                diff_odd += (counter[r + 1][i] - counter[l][i]) & 1;
            }
            if (diff_odd & 1) {
                res.push_back((diff_odd - 1) / 2 <= k);
            }
            else {
                res.push_back(diff_odd / 2 <= k);
            }
        }
        return res;
    }
};
