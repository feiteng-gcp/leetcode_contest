class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int mp[100010][30];
        memset(mp, 0, sizeof(mp));

        for (int i = 1; i <= s.length(); ++i)
        {
            for (int j = 0; j < 26; j++)
            {
                mp[i][j] = mp[i-1][j];
            }
            mp[i][s[i-1]-'a']++;
        }
        std::vector<bool> res;
        for (int i = 0; i < queries.size(); ++i)
        {
            int l = queries[i][0] + 1;
            int r = queries[i][1] + 1;

            int k = queries[i][2];
            int num = 0;
            for (int j = 0; j < 26; j++)
            {
                if ((mp[r][j] - mp[l-1][j]) % 2 == 1)
                    num++;
            }
            if (l != r && num / 2 > k)
                res.push_back(false);
            else
                res.push_back(true);
        }
        return res;
    }
};
