class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector<bool> result;
        int M = s.size();
        vector<vector<int>> preSum(M + 1, vector<int>(26, 0));
        for (int i = 0; i < M; ++i) {
            for (int j = 0; j < 26; ++j) {
                preSum[i + 1][j] = preSum[i][j];
            }
            ++preSum[i + 1][s[i] - 'a'];
        }
        for (auto &query : queries) {
            int l = query[0];
            int r = query[1];
            int k = query[2];
            int change = 0;
            for (int i = 0; i < 26; ++i) {
                int diff = preSum[r + 1][i] - preSum[l][i];
                if (diff % 2 == 1) ++change;
            }
            result.push_back(change / 2 <= k);
        }
        return result;
    }
};
