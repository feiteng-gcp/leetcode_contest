class Solution {
public:
    bool canMake(vector<int> mark, int count){
        int odd = 0;
        for(int i = 0; i < 26; ++i){
            if(mark[i] % 2 == 1) ++odd;
        }

        if(odd % 2 == 1) --odd;
        if((odd)/2 > count) return false;
        return true;
    }
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        vector<bool> ans;
        vector<vector<int>> dp(s.length()+1, vector<int>(26, 0));
        
        for(int i = 0;i<s.length(); i++){
            dp[i+1] = dp[i];
            dp[i+1][s[i]-'a']++;
        }
        
        for(auto& q : queries){
            // string subStr = s.substr(q[0], q[1]-q[0]+1);
            vector<int> tmp(26, 0);
            for(int i = 0; i<26; i++){
                tmp[i] = dp[q[1]+1][i] - dp[q[0]][i];
                // cout<<tmp[i]<<",";
            }
            // cout<<endl;
            ans.push_back(canMake(tmp, q[2]));
        }
        
        return ans;
    }
};
