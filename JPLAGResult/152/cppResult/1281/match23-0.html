class Solution {
public:
    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {
        int q=queries.size();
        int n=s.length();
        vector<vector<int>> freq(26,vector<int>(n+1,0));
        vector<int> cnt(26,0);
        for(int i=0; i<n; i++){
            cnt[s[i]-'a']++;
            for(int j=0; j<26; j++){
                freq[j][i+1]=cnt[j];
            }
        }
        
        vector<bool> ans;
        for(int i=0; i<q; i++){
            int l=queries[i][0]+1, r=queries[i][1]+1, k=queries[i][2];
            int odd=0;
            for(int j=0; j<26; j++){
                odd+=((freq[j][r]-freq[j][l-1])&1);
            }
            ans.push_back(odd/2<=k);
        }
        
        return ans;
    }
};
