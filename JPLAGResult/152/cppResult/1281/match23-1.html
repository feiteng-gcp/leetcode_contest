class Solution {
public:

    vector<bool> canMakePaliQueries(string s, vector<vector<int>>& queries) {

        vector < vector < int > > cnt(26, vector < int > (s.size(), 0));
        for (int i = 0; i < 26; i++) {
            cnt[i][0] = (s[0] == (i+'a'));
            for (int j = 1; j < s.size(); j++) {
                cnt[i][j] = cnt[i][j-1] + (s[j] == (i+'a'));
            }
        }

        vector < bool > ans;
        for (auto q : queries) {

            int l = q[0], r = q[1], m = q[2];
            int odd = 0;
            for (int i = 0; i < 26; i++) {
                int c = l == 0 ? cnt[i][r] : cnt[i][r] - cnt[i][l-1];
                odd += (c % 2);
            }
            int req = odd/2;
            if (req <= m) ans.push_back(true);
            else ans.push_back(false);
        }
        return ans;
    }
};
