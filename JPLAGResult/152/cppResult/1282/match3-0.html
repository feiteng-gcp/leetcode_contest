class Solution {
public:
    inline int lowbit(int x) {
        return x & (-x);
    }
    
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        vector<int> num_words;
        for (int i = 0; i < words.size(); i++) {
            int num = 0;
            for (int j = 0; j < words[i].length(); j++)
                num |= (1 << (words[i][j] - 'a'));
            int x = num, cnt = 0;
            while (x > 0) {
                x -= lowbit(x);
                cnt++;
            }
            if (cnt > 7) continue;
            num_words.push_back(num);
        }
        vector<int> contains_words[26];
        for (int i = 0; i < num_words.size(); i++) {
            for (int j = 0; j < 26; j++)
                if (num_words[i] & (1 << j)) {
                    contains_words[j].push_back(num_words[i]);
                }
        }
        vector<int> ret;
        for (int i = 0; i < puzzles.size(); i++) {
            vector<int> &cur = contains_words[puzzles[i][0] - 'a'];
            int hash = 0;
            for (int j = 0; j < puzzles[i].length(); j++)
                hash |= (1 << (puzzles[i][j] - 'a'));
            int tot = 0;
            for (int j = 0; j < cur.size(); j++) {
                if ((hash & cur[j]) == cur[j]) tot++;
            }
            ret.push_back(tot);
        }
        return ret;
    }
};
