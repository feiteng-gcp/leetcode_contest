class Solution {
public:
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        unordered_map<int, int> cnts[26];
        for (auto word : words) {
            int mask = getMask(word);
            for (int i = 0; i < 26; ++i) {
                if (mask & (1 << i)) {
                    ++cnts[i][mask];
                }
            }
        }
        vector<int> result;
        unordered_map<int, int> visited;
        for (auto puzzle : puzzles) {
            int mask = getMask(puzzle);
            int r = 0;
            visited.clear();
            search(visited, cnts[puzzle[0] - 'a'], mask, 0, 0, r);
            result.push_back(r);
        }
        return result;
    }
    
private:
    int getMask(const string &s) {
        int mask = 0;
        for (auto c : s) {
            mask |= (1 << (c - 'a'));
        }
        return mask;
    }
    
    void printMask(int mask) {
        for (int i = 0; i < 26; ++i) {
            if (mask & (1 << i)) {
                cout << 1;
            } else {
                cout << 0;
            }
        }
    }
    
    void search(unordered_map<int, int>& visited, unordered_map<int, int>& cnts, int mask, int index, int value, int& result) {
        if (!visited[value]) {
            visited[value] = 1;
            result += cnts[value];
        }
        if (index >= 26) {
            return;
        }
        search(visited, cnts, mask, index + 1, value, result);
        if (mask & (1 << index)) {
            search(visited, cnts, mask, index + 1, value + (1 << index), result);
        }
    }
};
