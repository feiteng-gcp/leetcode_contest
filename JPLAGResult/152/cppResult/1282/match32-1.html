class Solution {
public:
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        map<string, int> cnt;
        for (string &word : words) {
            sort(word.begin(), word.end());
            word.resize(unique(word.begin(), word.end()) - word.begin());
            
            cnt[word] ++;
        }
        vector<int> ans;
        for (string &p: puzzles) {
            char st = p[0];

            int c = 0;
            for (int state = 0; state < (1 << 6); ++state) {
                string x;
                x += st;
                for (int i = 0; i < 6; ++i) if (state >> i & 1) {
                    x += p[i + 1];
                }
                sort(x.begin(), x.end());
                
                c += cnt[x];
            }
            ans.push_back(c);
        }

        return ans;
    }
};
