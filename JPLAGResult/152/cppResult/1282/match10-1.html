class Solution {
public:
    int M=1<<6;
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        unordered_map<int,int> ma;
        for(string s:words)
        {
            int a=0;
            for(char c:s)
                a|=1<<(c-'a');
            ma[a]++;
        }
        vector<int> ans;
        for(string s:puzzles)
        {

            int nans=0;
            for(int i=0;i<M;i++)
            {
                int a=1<<(s[0]-'a');
                for(int j=0;j<6;j++)
                    if(i&(1<<j))
                        a|=1<<(s[j+1]-'a');
                if(ma.count(a)) nans+=ma[a];
            }
            ans.push_back(nans);
        }
        
        
        return ans;
    }
};
