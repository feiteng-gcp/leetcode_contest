class Solution {
public:
    
    int tran(string s) {
        int res = 0;
        for (int j = 0; j < s.length(); j++) {
            int idx = s[j] - 'a';
            res |= 1 << idx;
        }
        return res;
    }
    
    int countOne(int s) {
        int res = 0;
        while (s > 0) {
            res++;
            s &= s - 1;
        }
        return res;
    }
    

    bool contains(int s, int i) {
        
        return (s & (1 << i)) > 0;
    }
    
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& p) {
        vector<int> list[26];
        
        for (int i = 0; i < words.size(); i++) {
            int str = tran(words[i]);
            
            if (countOne(str) > 7) continue;

            for (int j = 0; j < 26; j++) if (contains(str, j)) {
                list[j].push_back(str);
            }
        }
        
        vector<int> ans;
        
        for (int i = 0; i < p.size(); i++) {
            int idx = p[i][0] - 'a';
            vector<int>& l = list[idx];
            
            int res = 0;
            int str = tran(p[i]);
            for (int w = 0; w < l.size(); w++) {
                int t = l[w];
                if ((t | str) == str)
                    res++;
            }
            ans.push_back(res);
        }
        return ans;
    }
    
};
