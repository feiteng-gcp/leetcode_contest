class Solution {
public:
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        vector<int> ret;
        vector<int> wordToInt;
        vector<int> wordContainChar[26];
        for (int i = 0; i < words.size(); i++) {
            int num = 0;
            int cnt = 0;
            for (int j = 0; j < words[i].length(); j++) {
                int which = words[i][j] - 'a';
                if ((num & (1<<which)) == 0) {
                    num ^= (1<<which);
                    cnt ++;
                }
            }
            if (cnt > 7) continue;
            wordToInt.push_back(num);
        }
        
        for (auto iter = wordToInt.begin(); iter != wordToInt.end(); iter ++) {
            int num = *iter;
            for (int i = 0; i < 26; i++) {
                int exist = num & (1 << i);
                if (exist != 0) {
                    wordContainChar[i].push_back(num);
                }
            }
        }
        
        for (int i = 0; i < puzzles.size(); i++) {
            string puzzle = puzzles[i];
            int cnt = 0;
            vector<int> tmpwords = wordContainChar[puzzle[0] - 'a'];
            int puzzlenum = 0;
            for (int j = 0; j < puzzle.length(); j++) {
                int which = puzzle[j] - 'a';
                if ((puzzlenum & (1<<which)) == 0) {
                    puzzlenum ^= (1<<which);
                }
            }
            
            for (int j = 0; j < tmpwords.size(); j++) {
                if ((tmpwords[j] & puzzlenum) == tmpwords[j]) cnt ++;
            }
            
            ret.push_back(cnt);
        }
        return ret;
    }
};
