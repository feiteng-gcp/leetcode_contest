#define pb push_back
#define fst first
#define snd second

typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:
  unordered_map<int,int> mp;
  vector<int> findNumOfValidWords(vector<string>& ws, vector<string>& ps) {
    for(auto& w:ws) {
      int k=0;
      for(char c:w) k|=(1<<(c-'a'));
      if(__builtin_popcount(k)<=7) mp[k]++;
    }
    vector<int> res;
    for(auto& p:ps) {

      int z=0;
      for(int m=1;m<(1<<7);m+=2) {
        int k=0;
        for(int i=0;i<7;i++){
          if(m&(1<<i)) k+=(1<<(p[i]-'a'));
        }
        z+=mp[k];
      }
      res.pb(z);
    }
    return res;
  }
};
