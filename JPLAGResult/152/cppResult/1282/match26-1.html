class Solution {
public:

    int hashString(string str){
        int ret = 0;
        for(int i = 0; i < str.length(); i++){
            ret |= (1 << (str[i]-'a'));
        }
        return ret;
    }
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        vector<int> ans;
        
        vector<int> hashW;
        vector<int> hashP;
        for(auto& w : words){
            int valW = hashString(w);
            hashW.push_back(valW);
        }
        
        for(auto& p : puzzles){
            int hashVal = hashString(p);
            int cnt = 0;
            for(int val : hashW){
                if((val & hashVal) == val){
                    
                    if(val & (1<<(p[0]-'a'))){
                        ++cnt;
                    }
                }
            }
            cout<<endl;
            ans.push_back(cnt);
        }
        return ans;
    }
};
