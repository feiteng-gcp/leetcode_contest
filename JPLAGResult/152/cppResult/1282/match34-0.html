class Solution {
public:
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        unordered_map<int, int> F;
        for (auto &s : words)
        {
            int flag = 0;
            for (auto c : s) flag |= 1<<(c-'a');
            F[flag] ++;
        }
        
        vector<int> ret;
        for (auto &t : puzzles)
        {
			int id = t[0]-'a';
            int flag = 0;
            for (auto c : t) flag |= 1<<(c-'a');
            
            int ans = 0;
            
            for (int x = flag; x; x = (x-1)&flag) if (x>>id&1)
            {
                auto it = F.find(x);
                if (it != F.end()) ans += it->second;
            }
            ret.push_back(ans);
        }
        return ret;
    }
};
