class Solution {
public:
    map<string, int> mp;
    
    string unique(string w) {
        string s = "";
        sort(w.begin(), w.end());
        s += w[0];
        for (int i = 1; i < w.size(); i++)
            if (w[i] != w[i-1]) s += w[i];
        return s;
        
    }
    
    vector<int> findNumOfValidWords(vector<string>& w, vector<string>& p) {
        for (int i = 0; i < w.size(); i++) {
            string u = unique(w[i]);
            if (u.size() <= 7) mp[u]++;
        }
            
        vector<int> res;
        for (int i = 0; i < p.size(); i++) {
            int cnt = 0;
            for (int k = 0; k < (1<<6); k++) {
                string s = "";
                s += p[i][0];
                for (int a = 0; a < 6; a++) {
                    if (k & (1<<a)) {
                        s += p[i][a+1];
                    }
                }
                sort(s.begin(), s.end());
                if (mp.find(s) != mp.end()) cnt += mp[s];
            }
            res.push_back(cnt);
        }
        return res;
    }
};
