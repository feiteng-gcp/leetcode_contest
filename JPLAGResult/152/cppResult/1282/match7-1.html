class Solution {
public:
	vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
		vector<int>result;
		vector<vector<int>>nums(26, vector<int>());
		for (int i = 0; i < words.size(); i++)
		{
			int temp = 0, sum = 0;
			vector<int>num(26, 0);
			for (int j = 0; j < words[i].size(); j++)
			{
				if (num[words[i][j] - 'a'] == 0)
				{
					num[words[i][j] - 'a'] = 1;
					sum++;
				}
				temp |= (1 << (words[i][j] - 'a'));
			}
			if (sum > 7) continue;
			for (int j = 0; j < 26; j++)
			{
				if (num[j]) nums[j].push_back(temp);
			}
		}
		for (int i = 0; i < puzzles.size(); i++)
		{
			int temp = 0, sum = 0;
			for (int j = 0; j < puzzles[i].size(); j++)
			{
				temp |= (1 << (puzzles[i][j] - 'a'));

			}
			for (int j = 0; j < nums[puzzles[i][0] - 'a'].size(); j++)
			{
				if ((temp | nums[puzzles[i][0] - 'a'][j]) == temp) sum++;
			}
			result.push_back(sum);
		}
		return result;
	}
};
