class Solution {
public:
    
    string f(const string &s) {
        string ans = s;
        std::sort(ans.begin(), ans.end());
        ans.resize(std::unique(ans.begin(), ans.end()) - ans.begin());
        return ans;
    }
    
    bool contain(const string &s, const string &p) {
        int i = 0;
        int j = 0;
        int len = p.size();
        while (i<7 && j<len) {
            if (s[i] == p[j]) {
                i++; j++;
            } else {
                i++;
            }
        }
        return j==len;
    }
    
    string foo(const string &s, int x) {
        string ret;
        ret.push_back(s[0]);
        for (int i=1; i<7; i++) {
            if (x&(1<<(i-1))) {
                ret.push_back(s[i]);
            }
        }
        std::sort(ret.begin(), ret.end());
        return ret;
    }
    
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        unordered_map<string, int> ref;
        for (auto &w: words) {
            // cout << f(w) << endl;
            ref[f(w)]++;
        }
        // cout << "------\n";
        int n = words.size();
        int m = puzzles.size();
        vector<int> ans(m, 0);
        for (int i=0; i<m; i++) {

            for (int j=0; j<64; j++) {
                string t = foo(puzzles[i], j);
                // cout << t << endl;
                if (ref.find(t) != ref.end()) {
                    // cout << t << " ok\n";
                    ans[i] += ref[t];
                }
            }
        }
        return ans;
    }
};
