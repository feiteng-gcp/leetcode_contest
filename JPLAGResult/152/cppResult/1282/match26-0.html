class Solution {
public:
    int  on(int n, int pos) {return n | (1<<pos);} //typecast 1 in case of int
    int  off(int n, int pos) {return n & ~(1<<pos);} //typecast 1 in case of int
    bool check(int n, int pos) {return (n & (1<<pos))>>pos;} //typecast 1 in case of int
    unordered_map < int, int > cnt;

    int create(string &a) {
        int mask = 0;
        for (int j = 0; j < a.size(); j++) {
            mask = on(mask, a[j]-'a');
        }
        return mask;
    }

    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        for (int i = 0; i < words.size(); i++) {
            cnt[create(words[i])]++;
        }

        vector < int > ans(puzzles.size(), 0);
        for (int i = 0; i < puzzles.size(); i++) {
            int bitmask = create(puzzles[i]);
            for ( int sub = 0 ; sub = sub - bitmask & bitmask ; ) {
                if (check(sub, puzzles[i][0]-'a')) {
                    ans[i] += cnt[sub];
                }
            }
        }
        return ans;
    }
};
