class Solution {
public:
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        int n=words.size(), m=puzzles.size();
        map<string,int> sw;
        for(auto x : words){
            vector<int> cnt(26,0);
            for(auto y : x) cnt[y-'a']++;
            string str="";
            for(int i=0; i<26; i++){
                if(cnt[i]) str+='a'+i;
            }
            sw[str]++;
        }
        
        // for(auto x : sw) cout<<x.first<<" "<<x.second<<endl;
        
        vector<int> ret;
        for(auto x : puzzles){
            vector<string> all;
            string str="";
            for(int j=1; j<x.length(); j++){
                int k=all.size();
                for(int i=0; i<k; i++){
                    all.push_back(all[i]+x[j]);
                }
                all.push_back(str+x[j]);
            }

            int ans=0;
            if(sw.find(str+x[0])!=sw.end()) ans+=sw[str+x[0]];
            
            for(auto y : all){
                str=x[0]+y;
                // cout<<str<<" ";
                sort(str.begin(),str.end());
                if(sw.find(str)!=sw.end()) ans+=sw[str];
            }
            // cout<<endl;
            ret.push_back(ans);
        }
        return ret;
    }
};
