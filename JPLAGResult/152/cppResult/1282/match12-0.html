class Solution {
public:
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        int n = puzzles.size();
        int m = words.size();
        vector<int> ans(n, 0);
        
        map<int, int> cnt;
        for (int i = 0; i < m; ++ i) {
            int num = 0;
            vector<bool> vis(26, false);
            for (char ch: words[i]) {
                if (!vis[ch - 'a']) num += (1<<(ch - 'a'));
                vis[ch - 'a'] = true;
            }
            ++ cnt[num];
        }
        
        for (int i = 0; i < n; ++ i) {
            for (int j = 0; j < (1<<6); ++ j) {
                int num = 1<<(puzzles[i][0] - 'a');
                for (int k = 0; k < 6; ++ k) if ((1<<k) & j) {
                    num += (1<<(puzzles[i][k + 1] - 'a'));
                }
                ans[i] += cnt[num];
            }
        }
        
        return ans;
    }
};
