class Solution {
public:
    map <int, int> cnt;
    int dfs(int b, int c, int bt) {
        if (c == 26) return cnt[b];
        int ans=0;
        if (b&(1<<c)) ans += dfs(b, c+1, bt);
        else {
            if (bt & (1 << c)) ans += dfs(b|(1<<c), c+1, bt);
            ans += dfs(b, c+1, bt);
        }
        return ans;
    }
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        cnt.clear();
        vector<int> ans;
        for (auto s : words) {
            int bt = 0;
            for (auto c : s) {
                bt |= (1 << (c-'a'));
            }
            cnt[bt]++;
        }
        for (auto s : puzzles) {
            int st = s[0] - 'a', bt = 0;
            for (auto c : s) {
                bt |= (1 << (c-'a'));
            }
            ans.push_back(dfs(1<<st, 0, bt));
        }
        return ans;
    }
};
