class Solution {
public:
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        int n = words.size(), m = puzzles.size();
        vector<int> masks(n, 0);
        for(int i=0;i<n;i++) {
            for(char c: words[i]) {
                masks[i] |= 1<<(c-'a');
            }
        }
        sort(masks.begin(), masks.end());
        vector<int> res(m, 0);
        for(int i=0;i<m;i++) {
            int mask = 0;
            for(char c: puzzles[i]) mask |= 1<<(c-'a');
            int r = upper_bound(masks.begin(), masks.end(), mask) - masks.begin();
            auto it = lower_bound(masks.begin(), masks.end(), 1<<(puzzles[i][0]-'a'));
            int l = it==masks.end()?0:(it-masks.begin());
            // cout<<i<<" "<<l<<" "<<r<<endl;
            for(int k = l; k<r; k++) {
                if((masks[k] | mask) == mask && (masks[k]&(1<<(puzzles[i][0]-'a')))) res[i]++;
            }
            // cout<<i<<endl;
        }
        return res;
    }
};
