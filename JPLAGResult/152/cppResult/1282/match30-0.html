class Solution {
public:
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        vector<int> a[26];
        for(auto w:words){
            int state = 0;
            for(auto c:w){
               state |= 1<<(c-'a');
            }
            vector<bool> ok(26,false);
            for(auto c:w) {
                if(ok[c-'a']== false) {
                    a[c - 'a'].push_back(state);
                    ok[c - 'a'] = true;
                }
            }
        }
        vector<int> res;
        for(auto p:puzzles){
            int ind = p[0] - 'a';
            int state = 0;
            for(auto c:p){
                state |= 1<<(c-'a');
            }
            int ans = 0;
            for(int j= 0;j < a[ind].size();j++){
                if((a[ind][j] | state) == state)
                    ans++;
            }
            res.push_back(ans);
        }
        return res;
    }
};
