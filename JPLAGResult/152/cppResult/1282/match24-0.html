map<int,int> cnt;

class Solution {
public:
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        int i,j,k,a;
        vector<int> ans;
        
        cnt.clear();
        
        for (i = 0;i < words.size();i++) {
            j = 0;
            for (char c : words[i])
                j |= (1 << (c - 'a'));
            cnt[j]++;
        }
        
        for (string p : puzzles) {
            a = 0;
            for (i = 0;i < (1 << 6);i++) {
                k = (1 << (p[0] - 'a'));
                for (j = 1;j < 7;j++) {
                    if (i & (1 << (j - 1)))
                        k |= (1 << (p[j] - 'a'));
                }
                a += cnt[k];
            }
            
            ans.push_back(a);
        }
        
        return ans;
    }
};
