int bit[100000000];
int temp[26];
vector<int> pos;
class Solution {
public:
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        long long int i,j,k,l,x,y,n=words.size(),m=puzzles.size();
        vector<int> res;
        for(i=0;i<26;i++)
            temp[i]=0;
         for(i=0;i<(1<<26);i++){
            bit[i] = 0;
        }
        for(i=0;i<n;i++){
            for(j=0;j<words[i].length();j++){
                temp[words[i][j]-'a']++;
            }
            long long int mask = 0;
            for(j=0;j<26;j++){
                if(temp[j])
                    mask |= (1<<j); 
                temp[j] = 0;
            }
            bit[mask]++;
        }
       
        for(i=0;i<m;i++)
        {   long long int ans=0;
            pos.clear();
            string s = puzzles[i];
            long long int mask = 0;
            x = s[0]-'a';
            mask = (1<<x);    //fixed
            for(j=1;j<7;j++){
                pos.push_back(s[j]-'a');
            }
            for(j=0;j<64;j++){
                mask = (1<<x);
                for(k=0;k<6;k++){
                    if(j&(1<<k))
                        mask|=(1<<pos[k]);
                }
                ans+=bit[mask];
            }
            res.push_back(ans);
        }
        return res;
        
    }
};
