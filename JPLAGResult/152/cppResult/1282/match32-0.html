class Solution {
public:
    unordered_map<string,int> mp[26];
    int cur_ans = 0;
    char fir;
    void make_combi(string s){
        set<char> p;
        for(auto x : s){
            p.insert(x);
        }
        string t = "";
        for(auto x : p){
            t += x;
        }
        int len = t.size();
        for(int i = 0 ; i < (1<<len); i++){
            string make = "";
            for(int j = 0 ; j < len; j++){
                if(i&(1 << j)){
                    make += t[j];
                }
            }
            cur_ans += mp[fir-'a'][make];
        }
        return;
    }
    vector<int> findNumOfValidWords(vector<string>& words, vector<string>& puzzles) {
        
        for(auto x : words){
            set<char> s;
            for(auto y : x){
                s.insert(y);
            }
            string t = "";
            for(auto y : s){
                t+=y;
            }
            for(auto x : t){
                mp[x-'a'][t]++;
            }
        }
        vector<int> ans;
        for(auto x : puzzles){
            string s2 = "";
            set<char> unq;
            for(auto y : x){
                unq.insert(y);
            }
            for(auto c : unq ){
                s2 += c;
            }
            // All possible combi of x
            // First letter of puzzle should be in the word
            fir = x[0];
            cur_ans = 0;
            make_combi(s2);
            ans.push_back(cur_ans);
        }
        return ans;   
    }
};
