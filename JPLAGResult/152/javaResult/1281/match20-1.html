class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        List<Boolean> res = new ArrayList<>();
        int N = s.length();
        int[][] cnt = new int[N][26];
        char[] ca = s.toCharArray();
        for (int i = 0; i < N; i++) {
            if (i > 0) {
                cnt[i] = Arrays.copyOf(cnt[i - 1], 26);
            }
            cnt[i][ca[i] - 'a']++;
        
}
        for (int i = 0; i < queries.length; i++) {
            int[] query = queries[i];
            int left = query[0];
            int right = query[1];
            int k = query[2];
            int[] charCnt = Arrays.copyOf(cnt[right], 26);
            int kNeeded = 0;
            for (int j = 0; j < 26; j++) {
                charCnt[j] -= left > 0 ? cnt[left - 1][j] : 0;
                kNeeded += charCnt[j] % 2;
            
}
            kNeeded /= 2;
            //boolean isOdd = (right - left + 1) % 2 == 1;
            res.add(kNeeded <= k);
        }
        return res;
    }
}
