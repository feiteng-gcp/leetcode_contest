class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        List<Boolean> res = new ArrayList<>();
        int n = s.length();
        int[][] sum = new int[n+1][26];
        for (int i = 0; i < n; i++) {
            sum[i+1][s.charAt(i)-'a'] = 1;
            for (char c = 'a'; c <= 'z'; c++) {
                sum[i+1][c-'a'] += sum[i][c-'a'];
            }
        }
        // for (char c = 'a'; c <= 'z'; c++) {
        //     System.out.println(sum[n][c-'a']);
        // }
        
        for (int[] q : queries) {
            int left = q[0], right = q[1], k = q[2];
            int[] map = new int[26];
            for (char c = 'a'; c <= 'z'; c++) {
                map[c-'a'] = sum[right+1][c-'a'] - sum[left][c-'a'];
            }
            // System.out.println("------");
            int odd = 0;
            for (char c = 'a'; c <= 'z'; c++) {
                if (map[c-'a'] % 2 == 1)
                    odd++;
            }
            res.add(odd / 2 <= k);
        }
        return res;
    }
}
