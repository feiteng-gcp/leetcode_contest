class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        char[] chs = s.toCharArray();
        List<Boolean> res = new ArrayList<>();
        for(int[] query: queries) {
            int left = query[0], right = query[1], k = query[2];
            if(k >= 26) {
                res.add(true);
                continue;
            }
            int[] cnt = new int[26];
            for(int i = left; i<= right; i++) {
                cnt[chs[i] - 'a']++;
            }
            int diff = 0;
            for(int i = 0; i < 26; i++) {
                if(cnt[i] % 2 == 1)
                    diff++;
            }
            res.add(diff <= k * 2 + 1);
        }
        return res;
    }
}
