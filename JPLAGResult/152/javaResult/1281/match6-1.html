class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        int n=s.length();
        int dp[][]=new int[n+1][26];
        for(int i=1;i<=n;i++){
            for(int j=0;j<26;j++){
                dp[i][j]=dp[i-1][j];
            }
            dp[i][s.charAt(i-1)-'a']++;
        }
        List<Boolean> list=new ArrayList<>();
        for(int q[]:queries){
            int odd=0;
            for(int i=0;i<26;i++){
                if((dp[q[1]+1][i]-dp[q[0]][i])%2==1){
                    odd++;
                }
            }
            if((q[1]-q[0]+1)%2==0){
                
if(odd/2<=q[2]){
                    list.add(true);
                }
                else{
                    list.add(false);
                }
            }
            else{
                if(odd/2<=q[2]){
                    list.add(true);
                }
                else{
                    list.add(false);
                }
            }
        }
        return list;
    }
}
