class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        int[][] grid = new int[s.length() + 1][26];
        for (int i = 0; i < s.length(); i++) {
            int idx = s.charAt(i) - 'a';
            
for (int j = 0; j < 26; j++) {
                if (j == idx) {
                    grid[i + 1][j] = grid[i][j] + 1;
                } else {
                    grid[i + 1][j] = grid[i][j];
                }
            }
        }
        List<Boolean> ans = new ArrayList<>();
        for (int[] query : queries) {
            ans.add(
isValid(grid, query));
        }
        return ans;
    }
    
    private boolean isValid(int[][] grid, int[] query) {
        int left = query[0], right = query[1], k = query[2];
        
int cnt = 0;
        for (int j = 0; j < 26; j++) {
            int tmp = grid[right + 1][j] - grid[left][j];
            if (tmp % 2 == 1) {
                cnt ++;
            }
        }
        return cnt - 1 <= 2 * k;
    }
}
