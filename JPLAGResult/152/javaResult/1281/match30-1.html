class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        final int n = s.length();
        
final char[] A = s.toCharArray();
        
        int[][] freq = new int[26][n + 1];
        for (int i = 0; i < n; ++i) {
            for (int c = 'a'; c <= 'z'; ++c) {
                freq[c - 'a'][i + 1] = freq[c - 'a'][i];
            }
            ++freq[A[i] - 'a'][i + 1];
        
}
        
        List<Boolean> lst = new ArrayList<>();
        for (int[] q : queries) {
            int lo = q[0];
            int hi = q[1] + 1;
            int k = q[2];
            int size = hi - lo;
            int[] f = getFreq(freq, lo, hi);
            // System.out.format("q = %s, changesNeeded = %d\n", Arrays.toString(q), changesNeeded(f, size));
            if (changesNeeded(f, size) <= k) 
{
                lst.add(true);
            } else {
                lst.add(false);
            }
        }
        return lst;
    }
    
    private static int[] getFreq(int[][] freq, int lo, int hi) {
        int[] f = new int[26];
        for (int c = 0; c < 26; ++c) {
            f[c] = freq[c][hi] - freq[c][lo];
        }
        return f;
    }
    
    private static int changesNeeded(int[] f, int size) {
        int ans = 0;
        if (size % 2 == 0) {
            for (int x : f) {
                ans += x % 2;
            }
        } else {
            boolean mid = false;
            for (int x : f) {
                if (x % 2 == 1) {
                    if (!mid) {
                        mid = true;
                    } else {
                        ++ans;
                    }
                }
            }
        }
        return ans >> 1;
    }
}
