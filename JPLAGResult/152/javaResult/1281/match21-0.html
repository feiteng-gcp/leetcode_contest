class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        int[][] count = new int[s.length() + 1][26];
        for (int i = 0; i < s.length(); i++) {
            for (int j = 'a'; j <= 'z'; j++) {
                count[i + 1][j - 'a'] = count[i][j - 'a'];
                if (s.charAt(i) == j) {
                    count[i + 1][j - 'a']++;
                }
            }
        }
        List<Boolean> res = new ArrayList<>();
        for (int[] q: queries) {
            int odd = 0, even = 0;
            for (int i = 'a'; i <= 'z'; i++) {
                int current = count[q[1] + 1][i - 'a'] - count[q[0]][i - 'a'];
                if (current % 2 == 0) even++;
                else odd++;
            }
            res.add(odd / 2 <= q[2]);
        }
        return res;
    }
}
