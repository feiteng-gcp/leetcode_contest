class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
      //[1,2,3,2,2,2,3,4,2,1]
      //a replace only makes sense to replace a,b where f[a] is odd && f[b] is odd
      // odd, even -> even,odd
      // odd, odd -> even, even
      // even, odd -> odd, even
      // even, even -> odd, odd
      // case 1: substring is even
      //   - we want odd freq characters to be even 
      // case 2: substring is odd
      //   - we want odd freq characters to be odd
      int[][] prefixFreq = new int[s.length() + 1][26];
      List<Boolean> rt = new ArrayList<>();
      
for(int i = 1; i < prefixFreq.length; i++){
          for(int z = 0; z < 26; z++){
            prefixFreq[i][z] = prefixFreq[i-1][z];
          }
          prefixFreq[i][s.charAt(i-1) - 'a']++;
      }
      for(int[] query: queries){
          int[] freqLeft = prefixFreq[query[0]];
          int[] freqRight = prefixFreq[query[1] + 1];
          
int len = query[1] - query[0] + 1;
          rt.add(canMake(freqLeft,freqRight, len, query[2]));
      }
      return rt;
    }

    private boolean canMake(int[] freqL, int[] freqR, int len, int chances){
      int numOfOddFreq = 0;
      for(int i = 0; i < 26; i++){
          int f = freqR[i] - freqL[i];
          if(f % 2 == 1)numOfOddFreq++;
      }
      // System.out.println(Arrays.toString(freqL));
      // System.out.println(Arrays.toString(freqR));
      return (len % 2 == numOfOddFreq % 2) && (numOfOddFreq / 2) <= chances;
    }
}
