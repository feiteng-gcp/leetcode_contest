class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        int n = s.length();
        int[][] cnt = new int[n + 1][26];
        for (int i = 1; i <= n; i++) {
            int v = s.charAt(i - 1) - 'a';
            for (int j = 0; j < 26; j++) {
                cnt[i][j] = cnt[i - 1][j];
            }
            cnt[i][v]++;
        }
        List<Boolean> list = new ArrayList<>();
        for (int[] q : queries) {
            int odd = 0;
            for (int i = 0; i < 26; i++) {
                if ((cnt[q[1] + 1][i] - cnt[q[0]][i]) % 2 == 1) odd++;
            }
            // System.out.println(odd);
            list.add((odd / 2) <= q[2]);
        }
        return list;
    }
}
