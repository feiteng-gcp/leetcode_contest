class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        
List<Boolean> ans = new ArrayList<>();
        for (int[] query : queries) {
            int begin = query[0], end = query[1], k = query[2];
            if (k >= 26) {
                ans.add(true);
                continue;
            }            
            int[] hash = new int[26];
            for (int i = begin; i <= end; ++i) {
                int idx = s.charAt(i) - 'a';
                ++hash[idx];
            
}                         
            int odds = 0;
            for (int i = 0; i < 26; ++i) {
                if (hash[i] % 2 != 0) ++odds;
            }                        
            int half = odds / 2;
            if (half <= k) {
                ans.add(true);                
            } else {
                ans.add(false);
            }
        }
        return ans;
    }
}
