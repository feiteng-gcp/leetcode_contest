class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        int n = s.length();
        int counts[][] = new int[n+1][26];
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < 26; j++) {
                counts[i+1][j] = counts[i][j];
            }
            counts[i+1][s.charAt(i)-'a']++;
        }
        
        List<Boolean> result = new ArrayList<Boolean>();
        for (int [] q : queries) {
            int l = q[0];
            int r = q[1];
            int k = q[2];
            int cnt = 0;
            for (int j = 0; j < 26; j++) {
                cnt += (counts[r+1][j] - counts[l][j]) % 2;
            }
            if (cnt > k * 2 + 1) {
                result.add(false);
            } else {
                result.add(true);
            }
        }
        return result;
    }
}
