class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        int[][] arr = new int[s.length()][26];
        for (int i = 0; i < s.length(); i++) {
            if (i != 0) {
                for (int j = 0; j < 26; j++) {
                    arr[i][j] = arr[i - 1][j];
                }
            }
            arr[i][s.charAt(i) - 'a']++;
        }
        List<Boolean> list = new ArrayList<>(queries.length);
        for (int[] query : queries) {
            int[] diff = new int[26];
            for (int i = 0; i < 26; i++) {
                diff[i] = arr[query[1]][i] - arr[query[0]][i];
            }
            diff[s.charAt(query[0]) - 'a']++;
            int mutate = find(diff);
            list.add(mutate <= query[2]);
        }
        return list;
    }
    int find(int[] str) {
        int count = 0;
        for (int i = 0; i < 26; i++) {
            str[i] %= 2;
            count += str[i];
        }
        if (count % 2 == 1) count--;
        return count / 2;
    }
}
