class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        char[] c=s.toCharArray();
        boolean[][] f=new boolean[c.length+1][26];
        for(int i=1;i<=c.length;i++){
            for(int j=0;j<26;j++)
                f[i][j]=f[i-1][j];
            f[i][c[i-1]-'a']=!f[i][c[i-1]-'a'];
        }
        List<Boolean> res=new LinkedList<Boolean>();
        for(int[] q:queries){
            int t=0;
            for(int j=0;j<26;j++)
                if(f[q[0]][j]!=f[q[1]+1][j])
                    t++;
            if((q[0]+q[1])%2==0)
                res.add((t-1)/2<=q[2]);
            else
                res.add(t/2<=q[2]);
        }
        return res;
    }
}
