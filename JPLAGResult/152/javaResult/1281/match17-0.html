class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        int n = s.length();
        int[][] cnt = new int[n][26];
        
        cnt[0][s.charAt(0) - 'a']++;
        for (int i = 1; i < n; ++i) {
            cnt[i][s.charAt(i) - 'a']++;
            for (int j = 0; j < 26; ++j) {
                cnt[i][j] += cnt[i - 1][j];
            }
        }
        
        List<Boolean> res = new ArrayList<>();
        for (int[] query: queries) {
            int l = query[0], r = query[1], k = query[2];
            if (k >= r - l + 1) {
                res.add(true);
                continue;
            }
            int odd = 0, even = 0;
            for (int i = 0; i < 26; ++i) {
                int p = l == 0 ?  cnt[r][i]: cnt[r][i] - cnt[l - 1][i];
                if (p > 0) {
                    if (p % 2 == 1) odd++;
                }
            }
            if ((odd / 2 <= k)||odd == 0 || (odd == 1 && k == 0 )) {
                res.add(true);
            }else {
                res.add(false);
            }
        }
        return res;
    }
}
