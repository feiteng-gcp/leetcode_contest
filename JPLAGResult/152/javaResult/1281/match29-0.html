class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        List<Boolean> sol = new ArrayList<>();
        int n = s.length();
        int[][] ct = new int[n][26];
        ct[0][s.charAt(0)-'a']++;
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < 26; j++) {
                ct[i][j] = ct[i-1][j];
            }
            ct[i][s.charAt(i)-'a']++;
        }
        
        for (int[] qr : queries) {
            int[] c1 = qr[0] == 0 ? new int[26] : ct[qr[0]-1];
            int[] c2 = ct[qr[1]];
            sol.add(qr[2] >= check(c1, c2));
        }
        return sol;
    }
    
    public int check(int[] c1, int[] c2) {
        int odd = 0;
        for (int i = 0; i < 26; i++) {
            if ((c2[i] - c1[i]) % 2 == 1) odd++;
        }
        return odd / 2;
    }
}
