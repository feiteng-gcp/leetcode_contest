class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        List<Boolean> ans = new ArrayList<>();
        int[][] cnt = new int[s.length() + 1][26];
        for (int i = 1; i <= s.length(); i++){
            for (int j = 0; j < 26; j++){
                cnt[i][j] = cnt[i - 1][j];
                if (s.charAt(i - 1) - 'a' == j){
                    cnt[i][j]++;
                }
            }
        }
        for (int[] query : queries){
            int[] charCnt = new int[26];
            for (int i = 0; i < 26; i++){
                charCnt[i] = cnt[query[1] + 1][i] - cnt[query[0]][i]; 
            }
            ans.add(can(charCnt, query[2]));
        }
        return ans;
    }
    boolean can(int[] charCnt, int k){
        int cnt = 0;
        for (int i = 0; i < charCnt.length; i++){
            if (charCnt[i] % 2 == 1){
                cnt++;
            }
        }
        if (cnt % 2 == 1){
            cnt--;
        }
        return cnt <= 2*k;
    }
}
