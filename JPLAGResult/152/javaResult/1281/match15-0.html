class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        char[] arr = s.toCharArray();
        int n = arr.length;
        int k = queries.length;
        int[][] pSum = new int[n + 1][26];
        for (int i = 0; i < n; i++) {
            char c = arr[i];
            for (int j = 0; j < 26; j++) {
                if (c - 'a' == j) {
                    pSum[i + 1][j] = pSum[i][j] + 1;
                } else {
                    pSum[i + 1][j] = pSum[i][j];
                }
            }
        }
        
        List<Boolean> ans = new ArrayList<>();
        for (int[] q : queries) {
            int diff = minReplace(pSum, q[0], q[1]);
            boolean b = (diff <= q[2]);
            ans.add(b);
        }
        return ans;
    }
    
    public int minReplace(int[][] pSum, int l, int r) {
        int count = 0;
        for (int j = 0; j < 26; j++) {
            int diff = pSum[r + 1][j] - pSum[l][j];
            if (diff % 2 == 1) {
                count++;
            }
        }
        return count / 2;
    }
}
