class Solution {
public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        int n = s.length();
        int[][] sum = new int[n + 1][26];
        
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < 26; j++) {
                sum[i][j] = sum[i - 1][j];
            }
            
            sum[i][s.charAt(i - 1) - 'a']++;
        }
        
        
        List<Boolean> ans = new ArrayList<>();
        for (int i = 0; i < queries.length; i++) {
            int count = 0;
            
            int l = queries[i][0] + 1;
            int r = queries[i][1] + 1;
            int k = queries[i][2];
            
            for (int j = 0; j < 26; j++) {
                int t = sum[r][j] - sum[l - 1][j];
                if (t % 2 == 1) count++;
            }
            
            count /= 2;
            if (count <= k) ans.add(true);
            else ans.add(false);
        }
        return ans;
    }

}
