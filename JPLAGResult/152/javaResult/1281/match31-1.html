class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        char[] chars = s.toCharArray();
        int[][] count = new int[chars.length][26];
        count[0][chars[0]-'a']++;
        for (int i = 1; i < chars.length; i++) {
            count[i] = 
Arrays.copyOf(count[i-1], count[i-1].length);
            count[i][chars[i]-'a']++;
        }
        List<Boolean> res = new ArrayList<>();
        for (int i = 0; i < queries.length; i++) {
            int left = queries[i][0];
            int right = queries[i][1];
            int k = queries[i][2];
            int[] aCount = Arrays.copyOf(count[right], count[right].length);
            // System.out.println("Length="+aCount.length);
            if (left > 0) {
                for (int j = 0; j < 26; j++) {
                    aCount[j] -= count[left-1][j];
                }
            }
            int odds = 0;
            for (int j = 0; j < 26; j++) {
                if (aCount[j] % 2 == 1) {
                    // System.out.print(aCount[j]+"-");
                    odds++;
                }
            }
            // System.out.println();
            // System.out.println(odds);
            if (odds / 2 <= k) {
                res.add(Boolean.TRUE);
            }
            else {
                res.add(Boolean.FALSE);
            }
        }
        return res;
    }
}
