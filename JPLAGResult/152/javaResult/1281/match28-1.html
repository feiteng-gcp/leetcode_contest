class Solution {
    private int[][] letterMap = null;
    
    
private boolean yesOrNo(String s, int[] q) {
        int odd = 0;
        for (int i = 0; i < 26; ++i) {
            if ((letterMap[q[1] + 1][i] - letterMap[q[0]][i]) % 2 != 0) {
                odd += 1;
            }
        }
        if (q[2] * 2 >= odd - 1) {
            return true;
        }
        return false;
    }
    
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        List<Boolean> ret = new ArrayList<>(queries.length);
        letterMap = 
new int[s.length() + 1][26];
        for (int i = 1; i <= s.length(); ++i) {
            for (int j = 0; j < 26; ++j) {
                char c = s.charAt(i-1);
                if (j == (c - 'a')) {
                    letterMap[i][j] = letterMap[i-1][j] + 1;
                } else {
                    letterMap[i][j] = letterMap[i-1][j];
                
}
            }
        }
        for (int[] q : queries) {
            ret.add(yesOrNo(s, q));
        }
        return ret;
    }
}
