class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        char[] cs = s.toCharArray();
        int n = cs.length;
        int[][] cnt = new int[26][n + 1];
        for (int i = 0; i < n; ++i) {
            int c = cs[i] - 'a';
            for (int j = 0; j < 26; ++j) {
                if (c == j) cnt[j][i + 1] = cnt[j][i] + 1;
                else cnt[j][i + 1] = cnt[j][i];
            }
        
}
        List<Boolean> res = new LinkedList();
        for (int[] query : queries) {
            int l = query[0], r = query[1] + 1, k = query[2], odd = 0;
            for (int i = 0; i < 26; ++i) {
                int j = cnt[i][r] - cnt[i][l];
                if (1 == (j & 1)) odd++;
            }
            if (odd <= 1 || (odd >> 1) <= k) res.add(true);
            else res.add(false);
        }
        return res;
    }
}
