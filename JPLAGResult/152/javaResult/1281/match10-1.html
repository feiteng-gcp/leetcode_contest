class Solution {
        public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
        int len = s.length();
        int[][] count =  new int[len][26];
        int[] temp = new int[26];
        for(int i=0;i<s.length();++i){
            char c = s.charAt(i);
            temp[c-'a']++;
            count[i]=
temp.clone();
        }
        List<Boolean> ls = new ArrayList<>();
        for(int i=0;i<queries.length;++i){
            int l = queries[i][0];
            int r = queries[i][1];
            int k = queries[i][2];
            int[] temparr = count[r].clone();
            int oddNum = 0;
            for(int j=0;j<26;++j){
                if(l>0)
                    temparr[j]-=count[l-1][j];
                
if((temparr[j]&1)>0)
                    ++oddNum;
            }
         //   System.out.println(i+"  "+oddNum);
            if(oddNum/2>k)
                ls.add(false);
            else 
                ls.add(true);
        }
        return ls;
    }
}
