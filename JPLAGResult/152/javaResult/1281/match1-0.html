class Solution {
    public List<Boolean> canMakePaliQueries(String s, int[][] qs) {
        int len = s.length();
        
        int d[][]=  new int[len+1][26];
        
        
        for(int i=0;i<len;++i){
            
            int c  =s.charAt(i)-'a';
            
            for(int j=0;j<26;++j){
                
                d[i+1][j] = d[i][j];
            }
            d[i+1][c]++;
            
        }
        
        
        
        
        
        List<Boolean> li = new ArrayList<>();
        for(int u[]:qs ){
            int l = u[0];
            int r = u[1];
            int k = u[2];
            int ct = 0;
            for(int i=0;i<26;++i){
                int v=  d[r+1][i]-d[l][i];
                
                if(v%2==1){
                    ct++;
                }
            }
            if(ct%2==1){
            ct--;
            ct = ct/2;}else{
                ct /= 2;
            }
            
            li.add(ct<=k);
            
        }
        return li;
    }
}
