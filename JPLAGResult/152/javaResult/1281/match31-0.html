// pre sum
// time complexity O(n)
class Solution {

  public List<Boolean> canMakePaliQueries(String s, int[][] queries) {
    int n = s.length();
    int[][] preSum = new int[n + 1][26];
    for (int i = 1; i <= n; i++) {
      System.arraycopy(preSum[i - 1], 0, preSum[i], 0, 26);
      preSum[i][s.charAt(i - 1) - 'a']++;
    }
    List<Boolean> res = new ArrayList<>();
    for (int[] query : queries) {
      int st = query[0] + 1;
      int e = query[1] + 1;
      int k = query[2];
      int diff = 0;
      for (int i = 0; i < 26; i++) {
        diff += ((preSum[e][i] - preSum[st - 1][i]) & 1) == 1 ? 1 : 0;
      }
      res.add(diff / 2 <= k);
    }
    return res;
  }
}
