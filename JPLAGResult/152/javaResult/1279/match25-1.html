class Solution {
    
    public static int mod = (int)Math.pow(10,9)+7;
    
    public int numPrimeArrangements(int n) {
        if(n <= 2)
            return 1;
        
int pC = 1;
        for(int i = 3; i <= n; i++) {
            if(isPrime(i))
                pC++;
        }
        long ret = (factorial(pC)*factorial(n-pC))%mod;
        return (int)ret;
    }
    
    public long factorial(int n) {
        long f = 1;
        
for(int i = 2; i <= n; i++) {
            f%=mod;
            f*=i;
            f%=mod;
        }
        return f%mod;
    }
    
    public boolean isPrime(int n) {
        for(int i = 2; i <= n/2; i++) {
            if(n % i == 0)
                return false;
        }
        return true;
    }
}
