class Solution {
    public int numPrimeArrangements(int n) {
        int mod = (int) 1e9+7;
        long ret = 1;
        
int c = 0;
        for (int i = 2; i <= n; i++) {
            boolean isPrime = true;
            for (int j = 2; j < i && isPrime; j++) {
                if (i % j == 0) {
                    isPrime = false;
                }
            }
            if (isPrime) c++;
        }
        
for (int i = 1; i <= c; i++) {
            ret = (ret * i) % mod;
        }
        for (int i = 1; i <= n - c; i++) {
            ret = (ret * i) % mod;
        }
        return (int) ret;
    }
}
