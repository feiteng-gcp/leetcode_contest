class Solution {
    public int numPrimeArrangements(int n) {
        long mod = 1_000_000_007;
        
int primes = 0;
        for (int i = 2; i <= n; ++i) {
            boolean isPrime = true;
            for (int j = 2; j < i; ++j) {
                if (i % j == 0)  {
                    isPrime = false;
                    break;
                }
            }
            if (isPrime) ++primes;
        }
        
        long fact1 = getFactorial(primes, mod);
        long fact2 = getFactorial(n - primes, mod);
        long ans = (fact1 * fact2) % mod;
        return (int)ans;
    }
    
    public long getFactorial(int n, 
long mod) {
        long ans = 1;
        for (int i = 2; i <= n; ++i) {
            ans = (ans * i) % mod;
        }
        return ans;
    }
 }
