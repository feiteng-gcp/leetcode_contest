class Solution {
    private int MOD = (int)1e9 + 7;
    public int numPrimeArrangements(int n) {
        if (n <= 1) {
            return 1;
        }
        
        int numOfPrime = 0;
        
long result = 1;
        for (int i = 2; i <= n; i ++) {
            if (isPrime(i)) {
                numOfPrime ++;
            }
        }
        
        for (int i = 1; i <= numOfPrime; i ++) {
            result = (result * i) % MOD;
        }
        
        
for (int i = 1; i <= n - numOfPrime; i ++) {
            result = (result * i) % MOD;
        }
        
        return (int)result;
    }
    
    private boolean isPrime(int num) {
        
for (int i = 2; i < num; i ++) {
            if (num % i == 0) {
                return false;
            }
        }
        
        return true;
    }
}
