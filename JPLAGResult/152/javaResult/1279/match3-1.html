class Solution {
    public int numPrimeArrangements(int n) {
        int cnt = 0;
        for (int i = 2; i <= n; i++) {
            boolean can = true;
            for (int j = 2; j < i; j++) {
                if (i % j == 0) can = false
;
            }
            if (can) cnt++;
        }
        long ans = 1, mod = (int) (1e9) + 7;
        for (int i = 1; i <= cnt; i++)
            ans = (ans * i ) % mod;
        for (int i = 1; i <= n - cnt; i++)
            ans = (ans * i ) % mod;
        return (int) ans;
    }
}
