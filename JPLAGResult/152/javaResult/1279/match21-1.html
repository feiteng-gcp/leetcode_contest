class Solution {
    int mod = 1000000007;
    public int numPrimeArrangements(int n) {
        int primeNum = numOfPrimes(n);
        int nonPrimeNum = n - primeNum;
        // System.out.println(primeNum);
        
        
return (int)((permutation(primeNum) * permutation(nonPrimeNum)) % mod);
    }
    
    private long permutation(int n) {
        long res = 1;
        for (int i = 2; i <= n; i++){
            res *= i;
            res %= mod;
        
}
        return res;
    }
    
    private int numOfPrimes(int n) {
        boolean[] mem = new boolean[n + 1];
        int cnt = 0;
        for (int i = 2; i <= n; i++) {
            if (mem[i]) continue;
            cnt++;
            for (int j = 1; j * i <= n; j++) {
                mem[i * j] = true;
            }
        }
        return cnt;
    }
}
