class Solution {
    public int numPrimeArrangements(int n) {
        int mod = (int)(1e9 + 7);
        int prime = 0;

        long res = 1;
        for(int i = 1; i <= n; ++i) {
            if(isPrime(i)) {
                prime++;
            }
        }
        // System.out.println(prime);
        for(int i = 1; i <= prime; ++i) {
            res *= i;
            res = res % mod;
        }
        
        for(int i = 1; i <= n - prime; ++i) {
            res *= i;
            res = res % mod;
        }
        
        return (int)res;
    }
    
    private boolean isPrime(int n) {
        if(n == 1) return false;
        for(int i = 2; i < n; ++i) {
            if(n % i == 0) return false;
        }
        return true;
    }
}
