class Solution {
    
public int numPrimeArrangements(int n) {
        int count = 0;
        for (int i = 2; i <= n; i++) {
            if (!isPrime(i)) continue
;
            count++;
        }
        
        return (int) ((permutation(count) * permutation(n - count)) % 1000000007);
        
    }
    
    private long permutation(int num) {
        
long ans = 1;
        for (int i = 2; i <= num; i++) {
            ans = (ans * i) % 1000000007;
        }
        return ans;
    }
    
    private boolean isPrime(int num) {
        for (int i = 2; i <= Math.sqrt(num); i++) 
{
            if (num % i == 0) return false;
        }
        return true;
    }
}
