class Solution {
    public int numPrimeArrangements(int n) {
        int prime_ct = 0;
        for(int i=2; i<=n; i++){
            boolean flag = true;
            
for(int j=2; j*j<=i;j++){
                if(i%j==0){
                    flag = false;
                    break;
                }
            }
            if(flag) prime_ct++;
        }
        // System.out.println(prime_ct);
        
int mod = 1000000007;
        long res = 1;
        for(int i=1; i<=prime_ct; i++){
            res *= i;
            res %= mod;
        }
        for(int i=1; i<=n-prime_ct; i++)
        {
            res *= i;
            res %= mod;
        }
        return (int)res;
    }
}
