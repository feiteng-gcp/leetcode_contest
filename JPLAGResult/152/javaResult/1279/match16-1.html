class Solution {
    boolean isPrime(int num) {
        
if (num < 2) return false;
        for (int i = 2; i <= num - 1; i++) {
            if ( num % i == 0) return false;
        }
        return true;
    }
    long factor(int num) {
        
long res = (long) 1;
        for (int i = 1; i <= num; i++) {
            res *= i;
            res = res % 1000000007;
        }
        return res;
    }
    public int numPrimeArrangements(int n) {
        int m = 0;
        for (int i = 1; i <= n; i++) {
            if (isPrime(i)) {
                m++;
            }
        }
        int mod = 1000000007;
        // System.out.println(m + " " + (n-m));
        long a = factor(m) % mod;
        long b = factor(n-m) % mod;
        long res = (a * b) % mod;
        // System.out.println(a + " " + b + " " + res);
        
        return (int)res;
    }
}
