class Solution {
    int MOD = 1000000007;
    public int numPrimeArrangements(int n) {
        int primeCnt = 0;
        for (int i = 1; i <= n; i++){
            if (isPrime(i)){
                primeCnt++;
            }
        }
        return (int)((fk(primeCnt) * fk(n - primeCnt)) % MOD);
    }
    public long fk(int n){
        long ans = 1;
        for (int i = 1; i <= n; i++){
            ans = (ans * i) % MOD; 
        }
        return ans;
    }
    public boolean isPrime(int n){
        if (n == 1){
            return false;
        }
        for (int i = 2; i < n; i++){
            if (n % i == 0){
                return false;
            }
        }
        return true;
    }
}
