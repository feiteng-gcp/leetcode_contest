class Solution {
    public int numPrimeArrangements(int n) {
        int primes = 0;
        for (int i = 2; i <= n; i++) {
            boolean p = true;
            
for (int j = 2; j < i; j++) {
                if(i % j == 0){
                    p = false;
                    break;
                }
            }
            if(p) primes++;
        }

        
long total = 1;
        for (int i = 1; i <= primes; i++) {
            total *= i;
            total %= 1000000007;
        }

        for (int i = 1; i <= n - primes; i++) {
            total *= i;
            total %= 1000000007;
        }
        return (int) total;
    }
}
