class Solution {
    int mod = 1000000007;
    
public int numPrimeArrangements(int n) {
        int cntPrime = 0;
        for (int i = 1; i <= n; i++) {
            if (isPrime(i)) {
                cntPrime++;
            }
        }
        return (int)(fac(cntPrime) * fac(n - cntPrime) % mod);
    
}
    
    private boolean isPrime(int a) {
        if (a == 1) {
            return false;
        }
        for (int i = 2; i < a; i++) {
            if (a % i == 0) {
                return false;
            }
        }
        return true;
    }
    
    private long fac(int a) {
        if (a == 0 || a == 1) {
            return 1;
        }
        long res = 1;
        for (int i = 2; i <= a; i++) {
            res *= i;
            res %= mod;
        }
        return res;
    }
}
