class Solution {
    public int numPrimeArrangements(int n) {
        
        boolean f[] = new boolean[n+1];
        f[1] = true;
        
        for(int i=2;i<=n;++i){
            for(int j=2;j<i;++j){
                if(i%j==0){
                    f[i] = true;
                    break;
                }
            }
            
        }
        int c =0;
        for(int i=1;i<=n;++i){
            if(!f[i]){
                c++;
            }
        }
        int nc = n-c;
        long p  = 1000000007L;
        long s = 1;
        for(long i=1;i<=c;++i){
            s *= i;
            s %= p;
        }
        for(long i=1;i<=nc;++i){
            s *= i;
            s %= p;
        }
        return (int)s;
        
    }
}
