class Solution {
    public int numPrimeArrangements(int n) {
        int count = 0;
        for (int i = 1; i <= n; i++) {
            if (isPrime(i)) {
                count++;
            }
        }
        //System.out.println(count);
        Long res = per(count) * per(n - count);
        return (int) (res % 1000000007);
    }
    
    private boolean isPrime(int num) {
        if (num == 1) {
            return false;
        }
        for (int i = 2; i * i <= num; i++) {
            if (num % i == 0) {
                return false;
            }
        }
        return true;
    }
    
    private long per(int i) {
        long res = 1;
        for (int k = 2; k <= i; k++) {
            res = res * k % 1000000007;
        }
        return res;
    }
}
