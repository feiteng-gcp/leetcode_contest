class Solution {
    public int numPrimeArrangements(int n) {
        int preNum = 0;
        for(int i=2; i<=n; i++) {
            if(isP(i)) preNum++;
        }
        int num1 = preNum, num2 = n-preNum;
        int res = 0;
        int MOD = 1_000_000_007;
        long t1 = 1, t2 = 1;
        for(int i=1; i<=num1; i++) {
            t1*=i;
            t1%=MOD;
        }
        for(int i=1; i<=num2; i++) {
            t2*=i;
            t2%=MOD;

        }
        //System.out.println(t1+" "+t2 +" "+preNum);
        return (int)((t1*t2)%MOD);
    }
    
    private boolean isP(int num) {
        for(int i=2; i<num; i++) {
            if(num%i ==0) return false;
        }
        
        return true;
    }
}
