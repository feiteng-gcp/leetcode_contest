class Solution {
    static final long MOD = 1_000_000_007;

    boolean isPrime(int n) {
        if (n <= 1) return false;
        if (n == 2 || n == 3) return true;

        for (int i = 2; i * i <= n; i++) {
            if (n % i == 0) return false;
        }
        return true;
    }

    long calP(int n) {
        if (n == 0) return 1;
        long ans = 1;
        for (int k = 1; k <= n; k++) {
            ans = ans * k % MOD;
        }
        return ans;
    }

    public int numPrimeArrangements(int n) {

        int count = 0;
        for (int i = 1; i <= n; i++) {
            if (isPrime(i)) {
                count++;
            }
        }

        return (int) (calP(count) * calP(n - count) % MOD);
    }
}
