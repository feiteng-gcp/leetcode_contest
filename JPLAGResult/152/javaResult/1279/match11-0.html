class Solution {
    long mod = 1000000007;
    
    public int numPrimeArrangements(int n) {
        //if (n == 1) return 0;
        int cnt = 0;
        for (int num = 1; num <= n; ++num) {
            if (isPrime(num)) {
                ++cnt;
            }
        }
        return (int)((fact(cnt) * fact(n - cnt)) % mod);
    }
    
    long fact(int x) {
        if (x == 0) return 1;
        long res = 1;
        for (long i = 2; i <= x; ++i) {
            res = (res * i) % mod;
        }
        return res;
    }
    
    boolean isPrime(int n) {
        if (n == 1) return false;
        if (n == 2) return true;
        for (int i = 2; i * i <= n; ++i) {
            if (n % i == 0) return false;
        }
        return true;
    }
}
