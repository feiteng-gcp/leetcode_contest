class Solution {
    public int numPrimeArrangements(int n) {
      int numOfPrime = 0;
      for(int i = 2; i <= n; i++){
          boolean isPrime = true;
          for(int j = 2; j * j <= i; j++){
              if(i % j == 0){
                isPrime = false;
                break;
              }
          }
          if(isPrime)numOfPrime++;
      }
      return (int)(factorialMod(numOfPrime) * factorialMod(n - numOfPrime) % (long)(1e9 + 7));
      
    }
    public long factorialMod(long n){
        long rt = 1;
        for(long i = 1; i <= n; i++){
          rt = (rt * i) % (long)(1e9 + 7);
        }
        return rt;
    }
}
