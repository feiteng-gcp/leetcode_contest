class Solution {
    public int numPrimeArrangements(int n) {
        int nprime = 0;
        for(int i = 2; i <= n; i ++) {
            boolean prime = true;
            
for(int j = 2; j <= i/2; j ++) {
                if(i%j ==0) {
                    prime = false;
                    break;
                }
            }
            if(prime) {
                nprime ++;
            }
        }
        int nonPrime = n - nprime;
        long base = 1000000000 + 7;
        
long ways = 1;
        for(int i = 1; i <= nprime; i ++) {
            ways = (ways * i) % base;
        }
        for(int i = 1; i <= nonPrime; i ++) {
            ways = (ways * i) % base;
        }
        //System.out.println(nprime);
        //System.out.println(nonPrime);
        return (int)ways;
    }
}
