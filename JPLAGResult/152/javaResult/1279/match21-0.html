class Solution {
    public int numPrimeArrangements(int n) {
        int n1 = countPrimes(n + 1);
        int n2 = n - n1;
        return solve(n1, n2);
    }
    
    public int solve(int n1, int n2) {
        return (int)((long)fac(n1) * fac(n2) % 1000000007);
    }
    
    public int fac(int n) {
        int mod = 1000000007;
        long sol = 1;
        for (int i = 1; i <= n; i++) {
            sol = sol * i % mod;
        }
        return (int)sol;
    }
    
    public int countPrimes(int n) {
        boolean[] noPr = new boolean[n];
        int ct = 0;
        for (int i = 2; i < n; i++) {
            if (noPr[i]) continue;
            ct++;
            for (int j = i; j <= (n - 1) / i; j++) {
                noPr[i*j] = true;
            }
        }
        
        return ct;
    }
}
