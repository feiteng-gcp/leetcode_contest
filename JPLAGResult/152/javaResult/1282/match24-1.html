class Solution {
    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        Trie trie = new Trie();
        
        for (String word : words) {
            trie.addWord(word);
        }
        List<Integer> result = new ArrayList<>();
        
for (String puzzle : puzzles) {
            Set<Character> puz = new HashSet<>();
            for (int j = 0; j < puzzle.length(); j++) {
                puz.add(puzzle.charAt(j));
            }
            int count = trie.find(puz, puzzle.charAt(0), new HashMap<>());
            
            result.add(count);
        }
        
        return result;
    }
    
    class Node {
        private Node[] alpha;
        private int isWord;
        
        Node() {
            alpha = new Node[26];
            isWord = 0;
        }
    }
    
    class Trie {
        private final Node node;
        
        Trie() {
            node = new Node();
        }
        
        int find(Set<Character> puz, Character first, Map<Character, Integer> curr) {
            return find(puz, first, curr, node, new StringBuilder());
            
        }
        
        int find(Set<Character> puz, Character first, Map<Character, Integer> curr, Node node, StringBuilder sb) {
            int count = 0;
            
            for (int i = 0; i < node.alpha.length; i++) {
                char c = (char)(i + 'a');
                Node nextNode = node.alpha[i];
                if (!puz.contains(c) || nextNode == null) {
                    continue;
                }
                
                curr.put(c, curr.getOrDefault(c, 0) + 1);
                sb.append(c);
                // System.out.println("SB:" + sb.toString() + ", " + nextNode.isWord + ", " + curr.containsKey(first) + " curr: " + curr + ", first: " + curr);
                if (nextNode.isWord > 0 && curr.containsKey(first)) {
                    count += nextNode.isWord ; // count the actual word
                    // System.out.println(sb.toString());
                }
                
                count += find(puz, first, curr, nextNode, sb);
                curr.put(c, curr.getOrDefault(c, 0) - 1);
                if (curr.get(c) == 0) {
                    curr.remove(c);
                }
                sb.deleteCharAt(
sb.length() - 1);
            }
            
            return count;
        }
            
        void addWord(String word) {
            Node curr = node;
            for (int i = 0; i < word.length(); i++) {
                int index = (word.charAt(i) - 'a');
                if (curr.alpha[index] == null) {
                    curr.alpha[index] = new Node();
                }
                curr = curr.alpha[index];
                
                if (i == word.length() - 1) {
                    curr.isWord += 1;
                }
            }
        }
    }
}
