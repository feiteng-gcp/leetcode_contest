class Solution {
    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        //List<Integer> wordSet = new ArrayList<>();
        HashMap<Integer, Integer> countMap = new HashMap<>();
        for(String word: words) {
            int set = 0;
            HashSet<Character> cset = new HashSet<>();
            for(char c: word.toCharArray()) {
                set = set | (1<<(c-'a'));
                cset.add(c);
                if(cset.size() > 7) {
                    break;
                }
            }
            if(cset.size() > 7) {
                continue;
            }
            //wordSet.add(set);
            countMap.put(set, countMap.getOrDefault(set, 0) + 1);
        }
        List<Integer> res = new ArrayList<>();
        for(String p: puzzles) {
            int set = 0;
            int first = p.charAt(0) - 'a';
            HashSet<Integer> idxes = new HashSet<Integer>();
            for(char c: p.toCharArray()) {
                if(c-'a' != first) {
                    idxes.add(c-'a');
                }
            }
            
            List<Integer> sets = new ArrayList<>();
            //sets.add(new ArrayList<>());
            helper(idxes, sets, first);
            int count = 0;
            for(int n: sets) {
               count += countMap.getOrDefault(n, 0);
            }
            res.add(count);
        }
        return res;
    }
    
    void helper(HashSet<Integer> idxes, List<Integer> sets, int first) {
        sets.add(0|(1<<first));
        for(int next: idxes) {
            for(int i = sets.size() - 1; i >=0; i --) {
                int n = sets.get(i) | (1<<next);
                sets.add(n);
            }
        }
    }
}
