class Solution {
    public int find(char[] c,int x,HashMap<String,Integer> map,StringBuilder sb,char cc){
        if(x==c.length){
            return map.getOrDefault(sb.toString(),0);
        }
        int res=0;
        if(c[x]!=cc) res+=find(c,x+1,map,sb,cc);
        sb.append(c[x]);
        res+=find(c,x+1,map,sb,cc);
        sb.delete(sb.length()-1,sb.length());
        return res;
    }
    
public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        HashMap<String,Integer> map=new HashMap<String,Integer>();
        for(String s:words){
            boolean[] f=new boolean[26];
            for(char c:s.toCharArray())
                f[c-'a']=true;
            StringBuilder sb=new StringBuilder(); 
            for(int i=0;i<26;i++)
                if(f[i])
                    sb.append((char)('a'+i));
            String ss=sb.toString();
            
map.put(ss,map.getOrDefault(ss,0)+1);
        }
        List<Integer> res=new LinkedList<Integer>();
        for(String s:puzzles){
            char[] c=s.toCharArray();
            Arrays.sort(c);
            char cc=s.charAt(0);
            int r=find(c,0,map,new StringBuilder(),cc);
            res.add(r);
        }
        return res;
    }
}
