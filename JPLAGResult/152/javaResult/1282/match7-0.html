class Solution {
    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        int wn = words.length;
        Map<Integer, List<Integer>> map = new HashMap<>();
        int[] bit = new int[wn];
        
        for (int i = 0; i < wn; i++) {
            boolean[] used = new boolean[26];
            for (int j = 0; j < words[i].length(); j++) {
                int k = words[i].charAt(j) - 'a';
                used[k] = true;
            }
            int temp = 0;
            for (int j = 0; j < 26; j++) {
                if (used[j]) {
                    temp += (1 << j);
                    List<Integer> l = map.getOrDefault(j, new ArrayList<>());
                    l.add(i);
                    map.put(j, l);
                }
            }
            bit[i] = temp;
        }
        
        List<Integer> ans = new ArrayList<>();
        for (String puzzle: puzzles) {
            int k = puzzle.charAt(0) - 'a';
            if (!map.containsKey(k)) {
                ans.add(0);
                continue;
            }
            int temp = 0;
            for (int i = 0; i < puzzle.length(); i++) {
                temp += (1 << (puzzle.charAt(i) - 'a'));
            }
            temp = ~temp;
            
            List<Integer> l = map.get(k);
            int total = 0;
            for (Integer i : l) {
                if ((bit[i] & temp) == 0) total++;
            }
            ans.add(total);
        }
        return ans;
    }
}
