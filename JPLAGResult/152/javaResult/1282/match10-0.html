class Solution {
    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
		List<Integer> rtn = new ArrayList<>();
		
        int puzzlesLen = puzzles.length;
        int[] intWords = new int[words.length];
        
        for (int i=0; i<words.length; i++) {
        	intWords[i] = convertWordToInt(words[i]);
        }
        
        for (int i=0; i<puzzlesLen; i++) {
        	String puzzle = puzzles[i];
        	
        	int intPuzzle = convertWordToInt(puzzle);
        	
        	int shiftedVal = 1 << ( puzzle.charAt(0) - 'a' );
        	
        	int total = 0;
        	for (int word : intWords) {
        		if ( ( (word | intPuzzle) > intPuzzle ) ||
        				( (word & shiftedVal) == 0 ) )
        			continue;
        		total++;
        	}
        	rtn.add(total);
        }
        return rtn;
    }
	
	private int convertWordToInt(String s) {
		int result = 0;
		
		for (int i=0; i<s.length(); i++) {
			char c = s.charAt(i);
			int charValue = c - 'a';
			
			result = result | ( 1 << charValue );
		}
		
		return result;
	}
}
