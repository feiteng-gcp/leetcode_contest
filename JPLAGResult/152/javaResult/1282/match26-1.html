class Solution {
    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        TrieNode root = new TrieNode();
        
        int n = words.length;
        for (int i = 0; i < n; ++i) {
            char[] sc  = words[i].toCharArray();
            
Arrays.sort(sc);
            insert(sc, root);
        }
        List<Integer> res = new ArrayList<>();
        
        for (String p: puzzles) {
            boolean[] cur = new boolean[26];
            for (int i = 0; i < p.length(); ++i) {
                cur[p.charAt(i) - 'a'] = true;
            }
            int[] t = {0};
            dfs(root, cur, t, 
p.charAt(0) - 'a', false);
            res.add(t[0]);
        }
        return res;
    }
    
    private void dfs(TrieNode cur, boolean[]exist, int[] t, int first, boolean k) {
        if (cur == null) return;
        if (k) {
            t[0] += cur.cnt;
        }
            
        for (int i = 0; i < 26; ++i) {
            if (cur.children[i] != null && exist[i]) {
                dfs(cur.children[i], exist, t, first, k || (i == first));
            }
        }
    }
    
    private void insert(char[] sc, TrieNode root) {
        TrieNode cur = root;
        for (int i = 0; i < sc.length; ++i) {
            if (i > 0 && sc[i] == sc[i - 1]) continue;
            if (cur.children[sc[i] - 'a'] == null) {
                cur.children[sc[i] - 'a'] = new TrieNode();
            }
            cur = cur.children[sc[i] - 'a'];
        }
        cur.cnt++;
    }
}

class TrieNode{
    TrieNode[] children;
    int cnt;
    TrieNode() {
        children = new TrieNode[26];
        cnt= 0 ;
    }
}
