class Solution {
    public List<Integer> findNumOfValidWords(String[] ws, String[] ps) {
        
            int len = ws.length;
            int s[] = new int[len];
            int i = 0;
        HashMap<Integer,Integer> st = new HashMap<>();
            for(String w:ws){
                for(char c:w.toCharArray()){
                    s[i] |= (1<<(c-'a'));
                }
                st.put(s[i],st.getOrDefault(s[i],0)+1);
                                i++;

            }
        
        List<Integer> li = new ArrayList<>();
        
        i = 0;
        for(String p:ps){
            int c = p.charAt(0)-'a';
            int all = 0;
            for(char t:p.toCharArray()){
                all |= 1<<(t-'a');
            }
            int ct =0;
            for (int x = all; x>0; x = (x-1)&all){
                if((x&(1<<c))>0){
                    
                    if(st.containsKey(x)){
                        ct+=st.get(x);
                    }
                    
                    
                }
            }
            
            

             li.add(ct);
            
            
        }
        return li;
        
        
    }
}
