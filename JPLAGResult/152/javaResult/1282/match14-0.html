class Solution {
   public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        int n = words.length;
        int[] bits = new int[n];

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < words[i].length(); j++) {
                bits[i] |= (1 << (int) (words[i].charAt(j) - 'a'));
            }
        }

        List<Integer>[] firstBits = new List[26];
        for (int i = 0; i < 26; i++) {
            firstBits[i] = new ArrayList<>();
        }

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < 26; j++) {
                if (((1 << j) & bits[i]) != 0) {
                    firstBits[j].add(bits[i]);
                }
            }
        }
        List<Integer> ans = new ArrayList<>();


        for (int i = 0; i < puzzles.length; i++) {
            int b = 0;
            for (int j = 0; j < puzzles[i].length(); j++) {
                b |= (1 << (int) (puzzles[i].charAt(j) - 'a'));
            }

            int count = 0;
            for (int bt : firstBits[puzzles[i].charAt(0) - 'a']) {
                if ((bt & b) == bt) count++;
            }
            ans.add(count);
        }
        return ans;
    }

}
