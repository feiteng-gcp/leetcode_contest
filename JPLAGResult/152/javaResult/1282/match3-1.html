class Solution {
    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        List<List<Integer>> wordMap = new ArrayList<>();
        for (int i = 0; i < 26; ++i) 
            wordMap.add(new ArrayList<Integer>());
        
        for (String word : words) {
            int v = convert(word);
            for (int i = 0; i < 26; ++i) {
                int mask = (1 << i);
                if ((v & mask) > 0) {
                    
wordMap.get(i).add(v);
                }
            }
        }
        
        List<Integer> ans = new ArrayList<>();
        for (String puzzle : puzzles) {
            int p = convert(puzzle);
            int idx = puzzle.charAt(0) - 'a';
            int c = 0;
            List<Integer> values = wordMap.get(idx);
            for (int v : values) 
{
                if ((p & v) == v) ++c;
            }
            ans.add(c);
        }
        
        return ans;
    }
    
    public int convert(String word) {
        int v = 0;
        for (int i = 0; i < word.length(); ++i) {
            int idx = word.charAt(i) - 'a';
            int mask = 1 << idx;
            v |= mask;
        }
        return v;
    }
}
