class Solution {
    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        int[] patterns = new int[words.length];
        for(int i = 0; i < words.length; i++) {
            String word = words[i];
            patterns[i] = getPattern(word);
        }

        List<Integer> res = new ArrayList<>();
        for(int i = 0; i < puzzles.length; i++) {
            int cnt = 0;
            int pattern = getPattern(puzzles[i]);
            for(int j = 0; j < words.length; j++) {
                // condition 2
                if((pattern | patterns[j]) != pattern)
                    continue;
                // condition 1
                int idx = puzzles[i].charAt(0) - 'a';
                if((patterns[j] & (1 << idx)) == 0)
                    continue;
                cnt++;
            }
            res.add(cnt);
        }

        return res;
    }

    private int getPattern(String word) {
        int pattern = 0;
        int[] cnt = new int[26];
        for(int j = 0; j < word.length(); j++) {
            char ch = word.charAt(j);
            cnt[ch - 'a']++;
        }
        for(int j = 0; j < 26; j++) {
            if(cnt[j] != 0) {
                pattern = (pattern | (1 << j));
            }
        }
        return pattern;
    }
}
