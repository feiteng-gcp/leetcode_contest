class Solution {
    Set<Integer> visited;
    int ans;

    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        int len = words.length;
        Map<Integer, Integer> record = new HashMap<>();
        for(int i = 0; i < len; i++){
            int cur = 0;
            for(char ch : words[i].toCharArray())
                cur |= (1 << (ch - 'a'));
            
record.put(cur, record.getOrDefault(cur, 0) + 1);
        }
        List<Integer> list = new ArrayList<>();
        for(String str : puzzles){
            int cur = 0;
            for(char ch : str.toCharArray())
                cur |= (1 << (ch - 'a'));
            int res = 0;
            int first = str.charAt(0) - 'a';
            visited = new HashSet<>();
            ans = 0;
            helper(cur, first, record);
            
list.add(ans);
        }
        return list;
    }

    public void helper(int cur, int first, Map<Integer, Integer> record){
        if(visited.contains(cur))
            return;
        //System.out.println(cur + " " + record.getOrDefault(cur, 0));
        ans += record.getOrDefault(cur, 0);
        for(int i = 0; i < 26; i++){
            if(i == first || (cur & (1 << i)) == 0)
                continue;
            int temp = cur ^ (1 << i);
            helper(temp, first, record);
        }
        visited.add(cur);
    }
}
