class Solution {
    // Trie
    class TrieNode {
        List<String> words = new ArrayList<>();
        TrieNode[] children = new TrieNode[256];
        char ch;
        public TrieNode() {
            
        }
        public TrieNode(char ch) {
            this.ch = ch;
        }
    };
    
    void insert(TrieNode root, char[] words, int index) {
        if (index == words.length) {
            root.words.add(new String(words));
            return;
        }
        char ch = words[index];
        if (root.children[ch - 'a'] == null) {
            root.children[ch - 'a'] = new TrieNode(ch);
        }
        root = root.children[ch - 'a'];
        insert(root, words, index + 1);
    }
    
    
public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        TrieNode root = new TrieNode();
        for (String word: words) {
            char[] chars = word.toCharArray();
            Arrays.sort(chars);
            StringBuffer buf = new StringBuffer();
            buf.append(chars[0]);
            for (int i = 1; i < chars.length; i++) {
                if (chars[i] != chars[i-1]) buf.append(chars[i]);
            }
            insert(root, 
buf.toString().toCharArray(), 0);
        }
        
        List<Integer> res = new ArrayList<>();
        for (String p: puzzles) {
            char[] chars = p.toCharArray();
            Set<Character> st = new HashSet<>();
            for (char ch: chars) {
                st.add(ch);
            }
            res.add(0);
            
helper(root, st, chars[0], res);
        }
        
        return res;
    }
    
    void helper(TrieNode root, Set<Character> dict, char firstChar, List<Integer> res) {
        for (String word: root.words) {
            if (word.indexOf(firstChar) != -1) {
                int idx = res.size() - 1;
                res.set(idx, res.get(idx) + 1);
            }
        }
        for (int i = 0; i < root.children.length; i++) {
            if (root.children[i] == null) {
                continue;
            }
            if (dict.contains(root.children[i].ch)) {
                helper(root.children[i], dict, firstChar, res);
            }
        }
    }
}
