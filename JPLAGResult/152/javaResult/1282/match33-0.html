class Solution {
    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        Map<String, Integer> normalized = new HashMap<>();
        for (String w : words) {
            String nm = norm(w);
            normalized.put(nm, normalized.getOrDefault(nm, 0) + 1);
        }
        List<Integer> r = new ArrayList<>();
        for (String p : puzzles) {
            char first = p.charAt(0);
            char[] cs = p.toCharArray();
            Arrays.sort(cs);
            String np = new String(cs);
            int rs = 0;
            List<String> allSubSets = getSub(np);
            for (String sub : allSubSets) {
                if (!sub.isEmpty()) {
                    if (sub.indexOf(first) != -1) {
                        Integer found = normalized.getOrDefault(sub, 0);
                        if(found!=0) {
                            rs += found;
                        }
                    }
                }
            }
            r.add(rs);
        }
        return r;
    }

    private List<String> getSub(String p) {
        return doSub(p, 0);
    }

    private List<String> doSub(String s, int i) {
        if (i == s.length()) {
            List<String> r = new ArrayList<>();
            r.add("");
            return r;
        }
        List<String> without = doSub(s, i + 1);
        List<String> cur = new ArrayList<>();
        for (String w : without) {
            cur.add(s.charAt(i) + w);
        }
        cur.addAll(without);
        return cur;
    }

    private String norm(String w) {
        int[] wm = new int[26];
        for (int i = 0; i < w.length(); i++) {
            wm[w.charAt(i) - 'a']++;
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < wm.length; i++) {
            if (wm[i] > 0) {
                char c = (char) (i + 'a');
                sb.append(c);
            }
        }
        return sb.toString();
    }
}
