class Solution {
    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        List<Integer> res = new ArrayList<>();
        List<Integer> wordsInt = new ArrayList<>();
        Map<Character, List<Integer>> map = new HashMap<>();
        for (char c = 'a'; c <='z'; c++) {
            map.put(c, new ArrayList<>());
        }
        for (int i = 0; i < words.length; i++) {
            int wordInt = word2Int(words[i]);
            wordsInt.add(wordInt);
            for (int j = 0; j < 26; j++) {
                if (((1 << j) & wordInt) != 0) {
                    map.get((char)('a' + j)).add(i);
                }   
            }
        }
        
        for (String p : puzzles) {
            int cnt = 0;
            int pInt = word2Int(p);
            List<Integer> idxes = map.get(p.charAt(0));
            for (int idx : idxes) {
                int wordInt = wordsInt.get(idx);
                if (((wordInt ^ pInt) & wordInt) == 0) {
                    cnt++;
                }
            }
            res.add(cnt);
        }
        
        return res;
    }
    
    private int word2Int(String word) {
        int res = 0;
        char[] ca = word.toCharArray();
        for (int i = 0; i < ca.length; i++) {
            res |= 1 << (ca[i] - 'a');
        }
        return res;
    }
}
