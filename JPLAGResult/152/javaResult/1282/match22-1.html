class Solution {
    public List<Integer> findNumOfValidWords(String[] words, String[] puzzles) {
        
FreqMap[] wMasks = new FreqMap[256];
        for (int c = 'a'; c <= 'z'; ++c) {
            wMasks[c] = new FreqMap();
        }
        for (String w : words) {
            HashSet<Character> charSet = new HashSet<>();
            for (char c0 : w.toCharArray()) {
                charSet.add(c0);
            }
            if (charSet.size() > 7) {
                continue;
            }
            int wm = makeMask(w);
            for (char c0 : charSet) {
                wMasks[c0].increment(wm);
            
}
        }
        
        List<Integer> ans = new ArrayList<>();
        for (String p : puzzles) {
            int pm = makeMask(p);
            char c0 = p.charAt(0);
            int count = 0;
            for (int s = pm; s != 0; s = (s - 1) & pm) {
                count += wMasks[c0].getOrDefault(s, 0);
            }
            ans.add(count);
        }
        return ans;

    private static int makeMask(String s) {
        int mask = 0;
        for (char c : s.toCharArray()) {
            int index = c - 'a';
            mask |= (1 << index);
        }
        return mask;
    }

    private static class FreqMap extends HashMap<Integer, Integer> {
        public void increment(int k) {
            put(k, 1 + getOrDefault(k, 0));
        }
