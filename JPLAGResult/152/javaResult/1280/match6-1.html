class Solution {
    public int dietPlanPerformance(int[] c, int k, int lower, int upper) {
        int res = 0;
        int n = c.length;
        int sum = 0;
        for (int i = 0; i < n; ++i) {
            sum += c[i];
            if (i >= k - 1) {
                if (sum < lower) res--;
                else if (sum > upper) res++
;
                sum -= c[i - k + 1];
            }
        }
        return res;
    }
}
