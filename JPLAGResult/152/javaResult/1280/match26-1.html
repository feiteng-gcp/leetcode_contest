class Solution {
    public int dietPlanPerformance(int[] nums, int k, int lower, int upper) {
        int sum = 0, res = 0;
        for (int i = 0; i < k; i++) 
            sum += nums[i]
;
        if (sum < lower) {
            res--;
        }
        else if (sum > upper) {
            res++;
        }
        for (int i = k; i < nums.length; i++) {
            sum -= nums[i-k];
            sum += nums[i];
            if (sum < lower) {
                res--;
            }
            else if (sum > upper) {
                res++;
            }
        }
        return res;
    }
}
