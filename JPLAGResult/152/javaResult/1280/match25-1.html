class Solution {
    public int dietPlanPerformance(int[] calories, int k, int lower, int upper) {
        int sum=0;
        for(int i=0;i<k;i++)
            sum+=calories[i];
        int res=0;
        if(sum>upper) res++;
        else
            if(sum<lower) res--;
        
for(int i=k;i<calories.length;i++){
            sum+=calories[i]-calories[i-k];
            if(sum>upper) res++;
            else
                if(sum<lower) res--;  
        }
        return res;
    }
}
