class Solution {
    public int dietPlanPerformance(int[] calories, int k, int lower, int upper) {
        int l = 0, r = 1;
        int T = calories[0];
        int ret = 0;
        for (; r < k && r < calories.length; r++) T += calories[r]
;
        if (T < lower) ret--;
        else if (T > upper) ret++;
        for (;r < calories.length; l++, r++) {
            T -= calories[l];
            T += calories[r];
            if (T < lower) ret--;
            else if (T > upper) ret++;
        }
        return ret;
    }
}
