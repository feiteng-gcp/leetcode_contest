class Solution {
    public int dietPlanPerformance(int[] calories, int k, int lower, int upper) {
        int points = 0, curSum = 0;
        for(int i = 0; i < calories.length; i++) {
            curSum += calories[i];
            if(i >= k - 1) {
                if(curSum > upper) points++;
                if(curSum < lower) points--
;
                curSum -= calories[i - k + 1];
            }
        }
        return points;
    }
}
