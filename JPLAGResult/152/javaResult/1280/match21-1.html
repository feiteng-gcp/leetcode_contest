class Solution {
    public int dietPlanPerformance(int[] calories, int k, int lower, int upper) {
        int ret = 0;
        int tmpSum = 0;
        for (int i = 0; i < k; ++i) {
            tmpSum += calories[i];
        }
        int n = calories.length;
        if (tmpSum > upper) {
            ret += 1;
        } else if (tmpSum < lower) {
            ret -= 1;
        }
        
for (int i = k; i < n; ++i) {
            tmpSum -= calories[i-k];
            tmpSum += calories[i];
            if (tmpSum > upper) {
                ret += 1;
            } else if (tmpSum < lower) {
                ret -= 1;
            }
        }
        return ret;
    }
}
