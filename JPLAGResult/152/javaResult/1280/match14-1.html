class Solution {
    public int dietPlanPerformance(int[] c, int k, int lower, int upper) {
        int res = 0;
        
        int n = c.length;
        
        int cal = 0;
        for (int i = 0; i < k; i++) cal += c[i]
;
        
        if (cal < lower) res--;
        else if (cal > upper) res++;
        
        for (int i = k; i < c.length; i++) {
            cal = cal - c[i-k] + c[i];
            if (cal < lower) res--;
            else if (cal > upper) res++;            
        }
        
        
        
        
        
        // int t = n/k;
        // for (int i = 0; i < t; i++) {
        //     int cal = 0;
        //     for (int j = 0; j < k; j++) {
        //         int idx = i * k + j;
        //         cal += c[idx];
        //     }
        //     if (cal < lower) res--;
        //     else if (cal > upper) res++;
        // }
        
        return res;
    }
}
