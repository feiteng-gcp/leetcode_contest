class Solution {
    public int dietPlanPerformance(int[] A, int k, int lower, int upper) {
        int n = A.length;
        int tot = 0;
        int pt = 0;
        for (int i = 0; i < n; i++) {
            tot += A[i];
            if (i >= k - 1) {
                if (tot > upper) {
                    pt++;
                } else if (tot < lower) {
                    pt--;
                }
                tot -= A[i - k + 1];
            }
        }
        return pt;
    }
}
