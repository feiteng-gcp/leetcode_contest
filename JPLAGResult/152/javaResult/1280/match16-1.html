class Solution {
    public int dietPlanPerformance(int[] calories, int k, int lower, int upper) {
        int res = 0, sum = 0;
        for (
int i = 0, j = 0; i < calories.length; ++i) {
            sum += calories[i];            
            if (++j == k) {
                if (sum < lower) res--;
                if (sum > upper) res++;
                --j;
                sum -= calories[i - j];
            }
        }
        return res;
    }
}
