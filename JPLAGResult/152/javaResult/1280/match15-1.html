class Solution {
    public int dietPlanPerformance(int[] calories, int k, int lower, int upper) {
        int ans = 0;
        
        int cur = 0;
        int i = 0;
        for(; i < k; i++) 
{
        	cur += calories[i];
        }
        	
        if(cur > upper) ans++;
        else if(cur < lower) ans--;
        
        for(; i < calories.length; i++) {
        	cur += calories[i];
        	cur -= calories[i-k];
        	if(cur 
> upper) ans++;
        	else if(cur < lower) ans--;
        	
        }
        return ans;
    }
}
