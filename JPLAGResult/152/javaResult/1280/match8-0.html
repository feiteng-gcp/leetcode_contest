class Solution {
    public int dietPlanPerformance(int[] calories, int k, int lower, int upper) {
        int n = calories.length, t = 0, p = 0;
        for (int i = 0; i < n; i++) {
            t += calories[i];
            if (i >= k - 1) {
                if (t < lower) p -= 1; else if (t > upper) p += 1;
                t -= calories[i - (k - 1)];
            }
        }
        return p;
    }
}
