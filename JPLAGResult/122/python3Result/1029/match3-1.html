from collections import defaultdict
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def verticalTraversal(self, root: 'TreeNode') -> 'List[List[int]]':
        x_records = [defaultdict(lambda: []) for _ in range(2003)]
        start_x = 1001
        
        queue = [(root, start_x)]
        
        y = 0
        while queue:
            level_len = len(queue)
            for idx in range(level_len):
                node, x = queue.pop(0)
                x_records[x]
[y].append(node.val)
                
                if node.left:
                    queue.append((node.left, x - 1))
                if node.right:
                    queue.append((node.right, x + 1))
            y += 1
        
        result = []
        for col in x_records:
            if len(col) > 0:
                tmp = []
                for y in sorted(col.keys()):
                    vals = col[y]
                    vals.sort()
                    tmp += vals
                result.append(tmp)
        
        return result
