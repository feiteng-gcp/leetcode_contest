# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def verticalTraversal(self, root: 'TreeNode') -> 'List[List[int]]':
        mapping = collections.defaultdict(list)
        queue = collections.deque([(root, 0)])
        minIdx, maxIdx = 0, 0
        while queue:
            size = len(queue)
            m = collections.defaultdict(list)
            for _ in range(size):
                node, idx = queue.popleft()
                minIdx = min(minIdx, idx)
                maxIdx = max(maxIdx, idx)
                m[idx].append(node.val)
                if node.left:
                    queue.append((node.left, idx - 1))
                if node.right:
                    queue.append((node.right, idx + 1))
            for idx, subList in m.items():
                subList.sort()
                mapping[idx].extend(subList)
        res = []
        for idx in range(minIdx, maxIdx + 1):
            res.append(mapping[idx])
        return res
