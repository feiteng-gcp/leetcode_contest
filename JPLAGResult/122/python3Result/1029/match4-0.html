# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
import collections
class Solution:
    def verticalTraversal(self, root: 'TreeNode') -> 'List[List[int]]':
        pos = collections.defaultdict(list)
        def scan(root, X, Y):
            if root is not None:
                pos[X].append((Y, root.val))
                if root.left is not None:
                    scan(root.left, X-1, Y+1)
                if root.right is not None:
                    scan(root.right, X+1, Y+1)
        scan(root, 0, 0)
        key = pos.keys()
        s = min(key)
        l = max(key)
        ans = [[] for _ in range(l - s + 1)]
        for k in range(l-s + 1):
            ans[k] = [x[1] for x in sorted(pos[k + s])]
        return ans
