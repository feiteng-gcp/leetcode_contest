# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def verticalTraversal(self, root: 'TreeNode') -> 'List[List[int]]':
        answer = {}
        def dfs(root, x, y):
            if x not in answer:
                answer[x] = [(y, -root.val)]
            else:
                answer[x].append((y, -root.val))
            if root.left:
                dfs(root.left, x - 1, y - 1)
                
            if root.right:
                dfs(root.right, x + 1, y - 1)
        dfs(root, 0, 0)
        result = []    
        for index in sorted(answer.keys()):
            #print(sorted(answer[index]))
            result.append(list(map(lambda x: -x[1], sorted(answer[index], reverse = True))))
            
        return result
