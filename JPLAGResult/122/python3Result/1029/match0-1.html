# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

from collections import defaultdict
class Solution:
    def verticalTraversal(self, root: 'TreeNode') -> 'List[List[int]]':
        d = {}
        
        def add_node(node, x_pos, y_pos):
            
            if x_pos in d:
                d[x_pos][y_pos].append(node.val)
            else:
                d[x_pos] = defaultdict(list)
                d[x_pos][y_pos].append(node
.val)
                
            if node.left:
                add_node(node.left, x_pos-1, y_pos-1)
            if node.right:
                add_node(node.right, x_pos+1, y_pos-1)
                
        add_node(root, 0, 0)
        res = []
        for x_pos, inner_d in sorted(d.items(), key = lambda x: x[0]):
            total_l = []
            for y_pos, l in sorted(inner_d.items(), key = lambda x: x[0], reverse = True):
                total_l += sorted(l)
            res.append(total_l)
        return res
