# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def verticalTraversal(self, root: 'TreeNode') -> 'List[List[int]]':
        d = dict()
        def f(root, k, h):
            if root == None:
                return
            if k not in d:
                d[k] = []
            d[k].append((h,root.val))
            f(root.left,k-1, h+1)
            f(root.right,k+1, h+1)
        f(root, 0, 0)
        res = []
        l = list(d)
        l.sort()
        for e in l:
            x = d[e]
            x.sort()
            res.append([])
            for ee in x:
                res[-1].append(ee[1])
        return res
