# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def verticalTraversal(self, root):
        if root == None:
            return []
        nodes = []
        def dfs(t, x, y):
            nodes.append((x, y, t.val))
            
if t.left != None:
                dfs(t.left, x - 1, y - 1)
            if t.right != None:
                dfs(t.right, x + 1, y - 1)
        dfs(root, 0, 0)
        ans = []
        dic = dict()
        for node in nodes:
            if not node[0] in dic:
                dic[node[0]] = []
            dic[node[0]].append(node)
        keys = sorted(k for k in dic)
        for k in keys:
            arr = dic[k]
            arr = [n[1:] for n in arr]
            arr.sort(key=lambda x: (x[0], -x[1]))
            arr.reverse()
            tmp = [i[1] for i in arr]
            ans.append(tmp)
        return ans
