class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        ans = []
        even = 0
        a = A
        for val in a:
            if val & 1 == 0:
                even += val
        for val, index in queries:
            temp = a[index]
            a[index] += val
            
if temp & 1 == 0:
                even -= temp
            if a[index] & 1 == 0:
                even += a[index]
            ans.append(even)
        return ans
