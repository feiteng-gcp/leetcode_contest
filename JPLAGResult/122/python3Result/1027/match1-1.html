class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        s = 0
        for a in A:
            if a % 2 == 0:
                s += a
        r = []
        
for a, b in queries:
            if A[b] % 2 == 0:
                s -= A[b]
            A[b] += a
            if A[b] % 2 == 0:
                s += A[b]
            r.append(s)
        return r
