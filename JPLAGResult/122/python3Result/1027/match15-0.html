class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        s = 0
        for num in A:
            if num % 2 == 0:
                s += num
        res = []
        for val, idx in queries:
            if A[idx] % 2 == 0:
                s -= A[idx]
            A[idx] += val
            if A[idx] % 2 == 0:
                s += A[idx]
            res.append(s)
        return res
