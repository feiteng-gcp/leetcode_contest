class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        esum = sum([i for i in A if i % 2 == 0])
        output = []
        for item in queries:
            if A[item[1]] % 2 == 0:
                if item[0] % 2 == 0:
                    A[item[1]] += item[0]
                    esum += item[0]
                    output.append(esum)
                else:
                    esum -= A[item[1]]
                    A[item[1]] += item[0]
                    output.append(esum)
            else:
                if item[0] % 2 == 0:
                    A[item[1]] += item[0]
                    output.append(esum)
                else:
                    A[item[1]] += item[0]
                    esum += A[item[1]]
                    output.append(esum)
        return output
