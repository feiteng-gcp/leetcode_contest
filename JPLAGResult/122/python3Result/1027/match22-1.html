class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        ret = []
        sm = sum(i for i in A if i%2==0)
        for i,j in queries:
            
if A[j]%2==0:
                if i%2==0:
                    sm+=i
                else:
                    sm-=A[j]
            else:
                if i%2!=0:
                    sm+=A[j]+i
            A[j]+=i
            ret.append(sm)
        return ret
