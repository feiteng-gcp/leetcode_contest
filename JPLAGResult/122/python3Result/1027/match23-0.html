class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        ans = []
        base = 0
        for e in A:
            if e % 2 == 0:
                base += e

        for q in queries:
            tmp = A[q[1]]
            A[q[1]] += q[0]
            if A[q[1]] % 2 == 0:
                if tmp % 2 == 0:
                    base += q[0]
                else:
                    base += A[q[1]]
            else:
                if tmp % 2 == 0:
                    base -= tmp
            ans.append(base)
            
        return ans
