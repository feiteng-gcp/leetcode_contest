class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        start = sum((v for v in A if v%2 == 0), 0)
        result = []
        for (v, idx) in queries:
            A[idx] += v
            if A[idx] % 2 == 0:
                if v % 2== 0:
                    start += v
                else:
                    start += A[idx]
            else:
                if v % 2 == 0:
                    pass
                else:
                    start -= A[idx] - v
            result.append(start)
        return result
