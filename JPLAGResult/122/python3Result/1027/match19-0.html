class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        sumi = sum([x for x in A if x % 2 == 0])
        #print("init", sumi)
        output = []
        for i in queries:
            if A[i[1]] % 2 == 0:
                if (A[i[1]] + i[0]) % 2 == 0:
                    sumi += i[0]
                else:
                    sumi -= A[i[1]]                    
            elif (A[i[1]] + i[0]) % 2 == 0:
                #print("enteere")
                sumi += A[i[1]] + i[0]
            A[i[1]] += i[0]
            #print("sumi", sumi)
            output.append(sumi)
        return output
