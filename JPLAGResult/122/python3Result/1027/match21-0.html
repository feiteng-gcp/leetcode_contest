class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        esum, ans = sum(a for a in A if not a & 1), []
        for d, i in queries:
            a, A[i] = A[i], A[i]+d
            if not a & 1:  esum -= a
            if not A[i] & 1: esum += A[i]
            ans.append(esum)
        return ans
