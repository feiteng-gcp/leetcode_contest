class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        sol=0
        for i in range(len(A)):
            if A[i]%2==0:
                sol+=A[i]
        
        sums=[]
        for query in queries:
            query[0] = A[query[1]]+ query[0]
            if A[query[1]]%2 == 0:
                sol-= A[query[1]]
            if query[0]%2 ==0:
                sol+= query
[0]
            A[query[1]]=query[0]
            sums.append(sol)
        return sums
