class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        res = sum(a if a % 2 == 0 else 0 for a in A)
        resp = []
        for q in queries:
            n1 = q[1]
            n2 = q[0]
            if A[n1] % 2 == 0:
                res -= A[n1]
            A[n1] += n2
            if A[n1] % 2 == 0:
                res += A[n1]
            resp.append(res)
        return resp
