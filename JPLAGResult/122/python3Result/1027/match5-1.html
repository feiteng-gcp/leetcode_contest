class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        
        even = 0
        for num in A:
            if num % 2 == 0:
                even += num
        
        res = []
        for q in queries:
            if A[q[1]] % 2 == 0:
                even -= A[q[1]]
            A[q[1]] += q[0]
            if A[q[1]] % 2 == 0:
                even += A[q[1]]
            res.append(even)
        return res
