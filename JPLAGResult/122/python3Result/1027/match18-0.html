class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        tmp = 0
        for a in A:
            if a % 2 == 0:
                tmp += a
        ans = []
        for q in queries:
            if A[q[1]] % 2 == 0:
                A[q[1]] += q[0]
                if A[q[1]] % 2 == 0:
                    tmp += q[0]
                    ans.append(tmp)
                else:
                    ans.append(tmp - A[q[1]] + q[0])
                    tmp = tmp - A[q[1]] + q[0]
            else:
                A[q[1]] += q[0]
                if A[q[1]] % 2 == 0:
                    tmp += A[q[1]]
                ans.append(tmp)
        return ans
