class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        ans = []
        res = sum(x for x in A if x % 2 == 0)
        
for val, idx in queries:
            if A[idx] % 2 == 0:
                res -= A[idx]
            A[idx] += val
            if A[idx] % 2 == 0:
                res += A[idx]
            ans.append(res)
        return ans
