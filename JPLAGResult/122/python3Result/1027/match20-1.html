class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        result = []
        even_sum = 0
        for val in A:
            if val % 2 == 0:
                even_sum += val
        
        #print(even_sum)
        for q in queries:
            val, idx = q
            
if A[idx] % 2 == 0:
                even_sum -= A[idx]
            A[idx] += val
            if A[idx] % 2 == 0:
                even_sum += A[idx]
            result.append(even_sum)
            
            #print(idx, val, even_sum, A)
        
        return result
