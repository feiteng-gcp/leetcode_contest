class Solution:
    def sumEvenAfterQueries(self, A: 'List[int]', queries: 'List[List[int]]') -> 'List[int]':
        results = []
        base = sum(a for a in A if a % 2 == 0)
        for val, index in queries:
            if A[index] % 2 == 0: base -= A[index]
            A[index] += val
            if A[index] % 2 == 0: base += A[index]
            results.append(base)
        return results
