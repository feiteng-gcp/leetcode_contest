# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def __init__(self):
        self.alpha = 'abcdefghijklmnopqrstuvwxyz'
        
    def smallestFromLeaf(self, root: 'TreeNode') -> 'str':
        if not root:
            return ''
        left = self.smallestFromLeaf(root.left)
        right = self.smallestFromLeaf(root.right)
        if left == '':
            return right + self.alpha[root.val]
        elif right == '':
            return left + self.alpha[root.val]
        elif left < right:
            return left + self.alpha[root.val]
        else:
            return right + self.alpha[root.val]
