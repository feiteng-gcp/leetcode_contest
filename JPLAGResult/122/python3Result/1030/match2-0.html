# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def smallestFromLeaf(self, root):
        strs = []
        def dfs(t, s):
            s = chr(ord('a') + t.val) + s
            if t.left == None and t.right == None:
                strs.append(s[:])
            else:
                if t.left != None:
                    dfs(t.left, s)
                if t.right != None:
                    dfs(t.right, s)
        dfs(root, '')
        return min(strs)
