# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def smallestFromLeaf(self, root: 'TreeNode') -> 'str':
        self.smallest = "z" * 1000
        
        def dfs(prev, node):
            if not node.left and not node.right:
                prev = prev[:]
                prev.append(node.val)
                chars = ''.join([chr(ord('a') + i) for i in prev[::-1]])
                if chars < self.smallest:
                    self.smallest = chars
            else:
                prev = prev
[:]
                prev.append(node.val)
                
                if node.left:
                    dfs(prev, node.left)
                
                if node.right:
                    dfs(prev, node.right)
        
        dfs([], root)
        return self.smallest
