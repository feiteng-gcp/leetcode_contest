# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def smallestFromLeaf(self, root: 'TreeNode') -> 'str':
        if root == None:
            return ''
        a = self.smallestFromLeaf(root.left)
        b = self.smallestFromLeaf(root.right)
        if a == '':
            return b + chr(97+root.val)
        elif b == '':
            return a + chr(97+root.val)
        return min(a,b)+chr(97+root.val)
