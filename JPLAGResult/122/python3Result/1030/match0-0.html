# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def smallestFromLeaf(self, root: 'TreeNode') -> 'str':
        if not root:
            return ''
        left_s = self.smallestFromLeaf(root.left)
        right_s = self.smallestFromLeaf(root.right)
        if not left_s:
            ans = right_s
        else:
            if not right_s:
                ans = left_s
            else:
                ans = min(left_s, right_s)
        char = chr(ord('a') + root.val)
        return ans + char
