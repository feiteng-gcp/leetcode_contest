# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def intervalIntersection(self, A: 'List[Interval]', B: 'List[Interval]') -> 'List[Interval]':
        m, n = len(A), len(B)
        i, j = 0, 0
        res = []
        while i < m and j < n:
            start = max(A[i].start, B[j].start)
            end = min(A[i].end, B[j].end)
            if start <= end:
                res.append([start, end])
            if A[i].end < B[j].end:
                i += 1
            else:
                j += 1
        return res
