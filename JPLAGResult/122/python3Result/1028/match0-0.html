# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def intervalIntersection(self, A: 'List[Interval]', B: 'List[Interval]') -> 'List[Interval]':
        points = []
        for pair in A:
            points.append((pair.start, 0))
            points.append((pair.end, 1))
        for pair in B:
            points.append((pair.start, 0))
            points.append((pair.end, 1))
        points.sort()
        # print(points)
        res = []
        count = 0
        start = None
        for point in points:
            # print('cur point',point)
            if point[1] == 0:
                count += 1
                if count > 1:
                    start = point[0]
                    # print('start',start)
            else:
                count -= 1
                if start != None:
                    # print('end',point[0])
                    ol = Interval(start, point[0])
                    res.append(ol)
                    start = None
                    
        return res
