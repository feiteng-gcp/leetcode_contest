# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def intervalIntersection(self, A: 'List[Interval]', B: 'List[Interval]') -> 'List[Interval]':
        a = A
        b = B
        if len(a) == 0 or len(b) == 0:
            return []
        
        pool 
= []
        for e in a:
            pool.append((e.start, 0))
            pool.append((e.end, 2))
        for e in b:
            pool.append((e.start, 1))
            pool.append((e.end, 3))

        pool.sort()
        # print(pool)

        marka = False
        markb = False
        ans = []
        l, r = None, None
        for pos, cate in pool:
            if cate == 0:
                marka = True
                if markb == True:
                    l = pos
            if cate == 1:
                markb = True
                if marka == True:
                    l = pos
            if cate == 2:
                if marka and markb:
                    r = pos
                    ans.append(Interval(l, r))
                marka = False
            if cate == 3:
                if marka and markb:
                    r = pos
                    ans.append(Interval(l, r))
                markb = False

        return ans
