# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e


class Solution:
    def intervalIntersection(self, A: 'List[Interval]', B: 'List[Interval]') -> 'List[Interval]':
        res = []
        a = 0
        b = 0
        while a<len(A) and b < len(B):
            x = A[a]
            y = B[b]
            if x.end < y.end:
                if x.end >= y.start:
                    res.append([max(x.start,y.start),x.end])
                a += 1
            else:
                if y.end >= x.start:
                    res.append([max(x.start,y.start),y.end])
                b += 1
        return res
