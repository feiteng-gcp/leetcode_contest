# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def intervalIntersection(self, A: 'List[Interval]', B: 'List[Interval]') -> 'List[Interval]':
        ans = []
        i = 0
        j = 0
        while i < len(A) and j < len(B):
            # 4 different situation
            if A[i].start < B[j].start:
                if A[i].end < B[j].start:
                    i += 1
                elif A[i].end < B[j].end:
                    ans.append(Interval(B[j].start, A[i].end))
                    i += 1
                else:
                    ans.append(Interval(B[j].start, B[j].end))
                    j += 1
            else:
                if B[j].end < A[i].start:
                    j += 1
                elif B[j].end < A[i].end:
                    ans.append(Interval(A[i].start, B[j].end))
                    j += 1
                else:
                    ans.append(Interval(A[i].start, A[i].end))
                    i += 1
        return ans
