# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def intervalIntersection(self, A: 'List[Interval]', B: 'List[Interval]') -> 'List[Interval]':
        ret = []
        if not (A and B):return ret
        i,j = 0,0
        n,m = len(A),len(B)
        while i<n and j<m:
            if A[i].start<=B[j].start<=A[i].end or B[j].start<=A[i].start<=B[j].end:
                ret.append(Interval(max(A[i].start,B[j].start),min(A
[i].end,B[j].end)))
                if A[i].end<B[j].end:
                    i+=1
                elif A[i].end>B[j].end:
                    j+=1
                else:
                    i+=1
                    j+=1
            elif A[i].start>B[j].end:
                j+=1
            elif A[i].end<B[j].start:
                i+=1
            else:
                print(A,B)
        return ret
