# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def intervalIntersection(self, A: 'List[Interval]', B: 'List[Interval]') -> 'List[Interval]':
        i = j = 0
        output = []
        while i < len(A) and j < len(B):
            a = A[i]
            b = B[j]
            if a.end < b.start:
                i += 1
            elif b.end < a.start:
                j += 1
            elif a.end < b.end:
                output.append(Interval(max(a.start, b.start), a.end))
                i += 1
            else:
                output.append(Interval(max(a.start, b.start), b.end))
                j += 1
        return output
