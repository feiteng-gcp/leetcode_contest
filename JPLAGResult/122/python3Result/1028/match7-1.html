# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def intervalIntersection(self, A, B):
        ans = []
        while len(A) > 0 and len(B) > 0:
            if A[0].end < B[0].start:
                A = A[1:]
            elif B[0].end < A[0].start:
                B = B[1:]
            else:
                s 
= max(A[0].start, B[0].start)
                e = min(A[0].end, B[0].end)
                ans.append(Interval(s, e))
                if A[0].end <= e:
                    A = A[1:]
                if B[0].end <= e:
                    B = B[1:]
        return ans
