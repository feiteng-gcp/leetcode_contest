# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e


class Solution:
    def intervalIntersection(self, A: 'List[Interval]', B: 'List[Interval]') -> 'List[Interval]':
        result = []
        a_idx = 0
        b_idx = 0
        
        while a_idx < len(A) and b_idx < len(B):
            cur_a = A[a_idx]
            cur_b = B[b_idx]
            
            if (cur_a.start <= cur_b.start and cur_a.end >= cur_b.start) or (cur_b.start <= cur_a.start and cur_b.end >= cur_a.start):
                start = max(cur_a.start, cur_b.start)
                end = min(cur_a.end, cur_b.end)
                result.append(Interval(start, end))
            
            if cur_a.end < cur_b.end:
                a_idx += 1
            else:
                b_idx += 1
        
        return result
