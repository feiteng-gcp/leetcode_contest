# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def intervalIntersection(self, A: 'List[Interval]', B: 'List[Interval]') -> 'List[Interval]':
        answer = []
        
        
        index_A = index_B = 0
        while index_A < len(A) and index_B < len(B):
            while index_A < len(A) and index_B < len(B) and A[index_A].start == B[index_B].start and A[index_A].end == B[index_B].end:
                answer.append(A[index_A])
                index_A += 1
                index_B += 1
                
            if not (index_A < len(A) and index_B < len(B)):
                break
            
            if A[index_A].start < B[index_B].start or A[index_A].start == B[index_B].start and A[index_A].end > B[index_B].end:
                while index_B < len(B) and B
[index_B].end < A[index_A].end:
                    answer.append(Interval(B[index_B].start, B[index_B].end))
                    index_B += 1
                if len(B) > index_B and B[index_B].start <= A[index_A].end:
                    answer.append(Interval(B[index_B].start, A[index_A].end))
                index_A += 1
                
            if not (index_A < len(A) and index_B < len(B)):
                break
                
            if A[index_A].start > B[index_B].start or A[index_A].start == B[index_B].start and A[index_A].end < B[index_B].end:
                
                while index_A < len(A) and A
[index_A].end < B[index_B].end:
                    answer.append(Interval(A[index_A].start, A[index_A].end))
                    index_A += 1
                if len(A) > index_A and A[index_A].start <= B[index_B].end:
                    answer.append(Interval(A[index_A].start, B[index_B].end))
                index_B += 1
            
            
        return answer
