# Definition for an interval.
# class Interval:
#     def __init__(self, s=0, e=0):
#         self.start = s
#         self.end = e

class Solution:
    def intervalIntersection(self, A: 'List[Interval]', B: 'List[Interval]') -> 'List[Interval]':
        ans = []
        a = A
        b= B
        m = len(a)
        n = len(b)
        index_a = 0
        index_b = 0
        def get_interval(in_a, in_b):
            if in_a.end < in_b.start or in_a.start > in_b.end:
                return False, None
            in_s = Interval(s=max(in_a.start, in_b.start), e=min(in_a.end, in_b.end))
            return True, in_s
        while index_a < m and index_b < n:
            flag, in_s = get_interval(a[index_a], b[index_b])
            if flag:
                ans.append(in_s)
            if a[index_a].end < b[index_b].end:
                index_a += 1
            elif a[index_a].end > b[index_b].end:
                index_b += 1
            else:
                index_a += 1
                index_b += 1
        return ans
