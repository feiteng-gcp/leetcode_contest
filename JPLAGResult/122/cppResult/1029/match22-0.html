/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    map<int, map<int, vector<int>, greater<int> > > mp;
    void goTree(TreeNode* node, int X, int Y){
        if(!node){
            return;
        }
        auto it=mp.find(X);
        if(it == mp.end()){
            map<int, vector<int>, greater<int> > mmp;
            vector<int> v;
            v.push_back(node->val);
            mmp.insert(make_pair(Y, v));
            mp.insert(make_pair(X, mmp));
        }else{
            auto & mmp = it->second;
            auto iit = it->second.find(Y);
            if(iit == it->second.end()){
                vector<int> v;
                v.push_back(node->val);
                it->second.insert(make_pair(Y, v));
            }else{
                iit->second.push_back(node->val);
            }
        }
        goTree(node->left, X-1, Y-1);
        goTree(node->right, X+1, Y-1);
    }
    
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        vector<vector<int>> ans;
       goTree(root, 0, 0);
        for(auto it=mp.begin();it!=mp.end();++it){
            vector<int> v;
            for(auto iit=it->second.begin();iit!=it->second.end();++iit){
                sort(iit->second.begin(), iit->second.end());
                for(int i=0;i<iit->second.size();i++){
                    v.push_back(iit->second[i]);
                }
            }
            ans.push_back(v);
        }
        return ans;
    }
};
