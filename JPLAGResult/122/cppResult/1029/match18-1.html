class Solution {
public:
  struct T{
    int x,y,v;
    T(){}
    T(int x,int y,int v):x(x),y(y),v(v){}

    bool operator<(const T& a)const{
      if(x!=a.x) return x<a.x;
      if(y!=a.y) return y>a.y;
      return v<a.v;
    }
  };
  vector<T> res;
  void dfs(TreeNode* v,int x,int y){
    if(!v) return;
    res.emplace_back(x,y,v->val);
    if(v->left) dfs(v->left,x-1,y-1);
    if(v->right) dfs(v->right,x+1,y-1);
  }
  vector<vector<int>> verticalTraversal(TreeNode* root) {
    res.clear();
    dfs(root,0,0);
    sort(res.begin(),res.end());
    map<int, vector<int> > mv;
    for(auto t:res)
      mv[t.x].emplace_back(t.v);
    vector<vector<int> > ans;
    for(auto p:mv) ans.emplace_back(p.second);
    return ans;
  }
};
