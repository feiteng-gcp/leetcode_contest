class Solution {
public:
    void tr(map<int, map<int, vector<int>>> & mp, TreeNode* n, int x, int y) {
        if (n == nullptr) {
            return;
        }
        mp[x][-y].push_back(n->val);
        tr(mp, n->left, x-1, y-1);
        tr(mp, n->right, x+1, y-1);
    }
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        map<int, map<int, vector<int>>> mp;
        tr(mp, root, 0, 0);
        vector<vector<int>> ans;
        for (auto it : mp) {
            vector<int> p;
            for (auto jt : it.second) {
                auto &v = jt.second;
                sort(v.begin(), v.end());
                for (auto w : v) {
                    p.push_back(w);
                }
            }
            ans.push_back(p);
        }
        return ans;
    }
};
