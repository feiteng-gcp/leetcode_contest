/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    map<int, map<int, vector<int>>> M;
public:

    vector<vector<int>> verticalTraversal(TreeNode* root) {
        dfs(root, 0, 0);
        vector<vector<int>> res;
        for (auto &it: M) {
            vector<int> now;
            for (auto &ot: it.second) {
                vector<int> &tmp = ot.second;
                sort(tmp.begin(), tmp.end());
                now.insert(now.end(), tmp.begin(), tmp.end());
            }
            res.push_back(now);
        }
        return res;
    }
    
    void dfs(TreeNode *tree, int x, int y) {
        if (!tree) {
            return;
        }
        M[x][-y].push_back(tree->val);
        dfs(tree->left, x - 1, y - 1);
        dfs(tree->right, x + 1, y - 1);
    }
};
