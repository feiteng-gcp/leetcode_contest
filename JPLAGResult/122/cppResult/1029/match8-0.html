/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        map<int,map<int,set<int>>> nodeMap;
        helper(root,0,0,nodeMap);
        vector<vector<int>> ans;
        for(auto &i:nodeMap)
        {
            ans.push_back(vector<int>());
            for(auto &j:i.second)
            {
                for(auto &k:j.second)
                {
                    ans.back().push_back(k);
                }
            }
        }
        return ans;
    }
    
    void helper(TreeNode* root,int x,int y,map<int,map<int,set<int>>>& nodeMap)
    {
        if(!root)return;
        helper(root->left,x-1,y+1,nodeMap);
        nodeMap[x][y].insert(root->val);
        helper(root->right,x+1,y+1,nodeMap);
    }
};
