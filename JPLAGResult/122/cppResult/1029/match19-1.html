/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    map<int, vector<pair<int, int>> > cache;
    void Rec(TreeNode* root, int x, int y) {
        if (!root)
            return;
        cache[x].push_back(make_pair(-y, root->val));
        Rec(root->left, x - 1, y - 1);
        Rec(root->right, x + 1, y - 1);
    }

    vector<vector<int>> verticalTraversal(TreeNode* root) {
        cache.clear();
        Rec(root, 0, 0);
        vector<vector<int>> solution;
        for (auto& entry: cache) {
            int x = entry.first;
            auto v = entry.second;
            sort(v.begin(), v.end());
            
            vector<int> new_list;
            for (auto el : v) {
                new_list.push_back(el.second);
            }
            solution.push_back(new_list);
        }
        return solution;
    }
};
