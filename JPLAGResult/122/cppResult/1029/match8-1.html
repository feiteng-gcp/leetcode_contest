/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        map<int,map<int,set<int>>> vals;
        helper(root, 0, 0, vals);
        vector<vector<int>> results;
        for(auto xItr = vals.begin(); xItr != vals.end(); xItr++){
            vector<int> xVec;
            for(auto yItr = xItr->second.begin(); yItr != xItr->second.end(); yItr++){
                for(auto val : yItr->second) xVec.push_back(val);

            }
            results.push_back(xVec);
        }
        return results;
    }
    
    void helper(TreeNode* root, int x, int y, map<int,map<int,set<int>>> & vals){
        if(!root) return;
        vals[x][y].insert(root->val);
        helper(root->left, x-1, y+1, vals);
        helper(root->right, x+1, y+1, vals);
    }
};
