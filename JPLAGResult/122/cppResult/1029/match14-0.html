/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<int,map<int,set<int>>> mp;
    void helper(TreeNode* root,int x,int y){
        if(root==NULL) return;
        mp[x][y].insert(root->val);
        helper(root->left,x-1,y-1);
        helper(root->right,x+1,y-1);
    }
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        helper(root,0,0);
        vector<vector<int>> ans;
        for(auto &it:mp){
            vector<int> tmp;
            map<int,set<int>> mpy=it.second;//mpy用倒序
            map<int,set<int>>::reverse_iterator ity=mpy.rbegin();
            for(;ity!=mpy.rend();ity++){
                for(auto &its:ity->second){
                    tmp.push_back(its);
                }
            }
            
            ans.push_back(tmp);
        }
        return ans;
    }
};
