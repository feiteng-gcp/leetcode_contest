/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<tuple<int,int,int>> v;
    map<int, vector<int>> mp;
    
    void build(TreeNode *now, int x, int y) {
        if(now == NULL) return;
        build(now->left, x-1, y+1);
        build(now->right, x+1, y+1);
        v.push_back(make_tuple(x, y, now->val));
    }
    
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        build(root, 0, 0);

        sort(v.begin(), v.end());
        for(auto emt: v) {
            mp[get<0>(emt)].push_back(get<2>(emt));
        }
        
        vector<vector<int>> ans;
        for(auto emt: mp) {
            ans.push_back(emt.second);
        }
        return ans;
    }
};
