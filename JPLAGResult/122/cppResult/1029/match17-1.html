/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector <pair <pair <int, int>, int> > l;
    
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        vector <vector <int> > ret;
        l.clear();
        dfs(root, 0, 0);
        sort(l.begin(), l.end());
        vector <int> cur;
        int now = l[0].first.first;
        cur.push_back(l[0].second);
        for (int i = 1; i < l.size(); i++) {
            //printf("%d %d %d\n", l[i].first.first, l[i].first.second, l[i].second);
            if (l[i].first.first == now) {
                cur.push_back(l[i].second);
            } else {
                ret.push_back(cur);
                cur.clear();
                cur.push_back(l[i].second);
                now = l[i].first.first;

            }
        }
        
        ret.push_back(cur);
        return ret;
    }
    
    void dfs(TreeNode *root, int x, int y) {
        if (root == NULL) {
            return;
        }
        
        l.push_back(make_pair(make_pair(x, -y), root->val));
        dfs(root->left, x - 1, y - 1);
        dfs(root->right, x + 1, y - 1);
    }
};
