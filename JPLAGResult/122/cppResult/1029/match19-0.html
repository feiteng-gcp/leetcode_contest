/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<int, vector<pair<int, int> > >mp;

    void rec(TreeNode *v, int x, int y) {
	if (!v) return;
	mp[x].push_back(make_pair(y, v->val));
	rec(v->left, x-1, y+1);
	rec(v->right, x+1, y+1);
    }

    vector<vector<int>> verticalTraversal(TreeNode* root) {
	mp.clear();
	rec(root, 0, 0);

	vector<VI> ans;
	EACH (e, mp) {
	    vector<pair<int, int> > &p = e->second;
	    sort(p.begin(), p.end());
	    VI h;
	    EACH (f, p) h.push_back(f->second);
	    ans.push_back(h);
	}
        
	return ans;
    }
};
