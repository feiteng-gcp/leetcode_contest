typedef long long ll;
typedef vector<int> VI;
typedef pair<int,int> PII;

#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))

const int INF = (int)1E9;
#define MAXN 100005

class Solution {
public:

  vector<pair<PII, int>> pts;
  void solve(TreeNode *r, int x, int y) {
    if (r == NULL) return;
    pts.push_back(make_pair(PII(x,y), r->val));
    solve(r->left, x - 1, y + 1);
    solve(r->right, x + 1, y + 1);
  }
  vector<vector<int>> verticalTraversal(TreeNode* root) {
    pts.clear();
    solve(root, 0, 0);
    sort(pts.begin(), pts.end());
    int n = pts.size();
    vector<VI> ans;
    REP(i,0,n) {
      int j = i;
      while (j < n && pts[j].first.first == pts[i].first.first) j++;
      VI vals;
      REP(k,i,j) vals.push_back(pts[k].second);
      ans.push_back(vals);
      i = j - 1;
    }
    return ans;
  }
};
