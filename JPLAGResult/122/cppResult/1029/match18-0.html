/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    struct mode{
        int x,y,v;
        bool operator<(const mode&a)const{
            if(x!=a.x) return x<a.x;
            if(y!=a.y) return y>a.y;
            return v<a.v;
        }
    };
    vector<mode> tmp;
    void dfs(TreeNode *root,int x,int y)
    {
        tmp.pb(mode{x,y,root->val});
        if(root->left) dfs(root->left,x-1,y-1);
        if(root->right) dfs(root->right,x+1,y-1);
    }
public:
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        vector<vector<int>> ans;
        tmp.clear();
        dfs(root,0,0);
        sort(ALL(tmp)),ans.pb(vector<int>(1,tmp[0].v));
        for(int i=1;i<tmp.size();++i)
            if(tmp[i].x!=tmp[i-1].x) ans.pb(vector<int>(1,tmp[i].v));
            else ans[ans.size()-1].pb(tmp[i].v);
        return ans;
    }
};
