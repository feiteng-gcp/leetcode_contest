/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {


private:
    vector<pair<pair<int, int>, int>> ord;
    
    void dfs(TreeNode* root, int x, int y) {
        if (!root) {
            return;
        }
        ord.push_back(make_pair(make_pair(x, y), root->val));
        if (root->left) {
            dfs(root->left, x-1, y+1);
        }
        if (root->right) {
            dfs(root->right, x+1, y+1);
        }
    }

public:
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        dfs(root, 0, 0);
        sort(ord.begin(), ord.end());
        
        vector<vector<int>> sol;
        if (ord.empty()) {
            return sol;
        }
        
        vector<int> lvl;
        int x = ord[0].first.first;
        lvl.push_back(ord[0].second);

        int n = ord.size();
        for (int i = 1; i < n; ++i) {
            if (ord[i].first.first == x) {
                lvl.push_back(ord[i].second);
            } else {
                sol.push_back(lvl);
                lvl.clear();
                lvl.push_back(ord[i].second);
                x = ord[i].first.first;
            }
        }
        
        if (!lvl.empty()) {
            sol.push_back(lvl);
        }
        return sol;
    }
};
