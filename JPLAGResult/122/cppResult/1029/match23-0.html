/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        map<int, vector<int>> result;
        vector<pair<int, TreeNode*>> q;
        vector<pair<int, TreeNode*>> next_q;
        
        if (root == nullptr) return vector<vector<int>>{};
        
        q.push_back(make_pair(0, root));
    
        while (q.size() > 0) {
            int end_index = q.size();
            map<int, set<int>> result_of_this_round;
            for (int i = 0; i < end_index; i++) {
                TreeNode* node = q[i].second;
                if (node->left != nullptr) {
                    next_q.push_back(make_pair(q[i].first-1, node->left));
                }
                if (node->right != nullptr) {
                    next_q.push_back(make_pair(q[i].first+1, node->right));
                }
                
                if (result_of_this_round.find(q[i].first) == end(result_of_this_round)) {
                    result_of_this_round[q[i].first] = set<int>{node->val};
                } else {
                    result_of_this_round[q[i].first].insert(node->val);
                }
            }
            
            for (auto& kvp : result_of_this_round) {
                if (result.find(kvp.first) == end(result)) {
                    result[kvp.first] = vector<int>(begin(kvp.second), end(kvp.second));
                } else {
                    result[kvp.first].insert(result[kvp.first].end(), begin(kvp.second), end(kvp.second));
                }
            }
            
            swap(q, next_q);
            next_q.clear();
        }
        
        vector<vector<int>> answer;
        for (auto& kvp : result) {
            answer.push_back(kvp.second);
        }
        return answer;
    }
};
