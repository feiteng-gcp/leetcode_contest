/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        if (!root) {
            return {};
        }
        
        queue <pair <TreeNode*, pair <int, int> > > q;
        q.push({root, {0, 0}});
        
        int size;
        TreeNode* node = NULL;
        int posx, posy;
        
        map <int, map <int, vector <int> > > data;
        
        while (!q.empty()) {
            size = q.size();
            
            while (size >= 1) {
                size--;
                
                node = q.front().first;
                posx = q.front().second.first;
                posy = q.front().second.second;
                q.pop();
                
                data[posx][posy].push_back(node->val);
                
                if (node->left) {
                    q.push({node->left, {posx - 1, posy + 1}});                    
                }
                if (node->right) {
                    q.push({node->right, {posx + 1, posy + 1}});
                }
            }
        }
        
        vector <vector <int> > ans;
        for (auto iter : data) {          
            vector <int> lev;
            for (auto val : iter.second) {
                sort(val.second.begin(), val.second.end());
                for (auto each : val.second) {
                    lev.push_back(each);
                }                
            }
            ans.push_back(lev);
        }
        
        return ans;
    }
};
