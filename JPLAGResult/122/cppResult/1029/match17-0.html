/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Pnt{
public:
    int x;
    int y;
    int val;
    Pnt(int _x, int _y, int _v)
    {
        x=_x;
        y=_y;
        val=_v;
    }
     bool operator<(const Pnt& other)  const 
    {
        if(y>other.y || (y==other.y && val<other.val)) 
            return true;
        return false;
    }
};
class Solution {
public:
    vector<vector<int>> verticalTraversal(TreeNode* r) {
        scan(r, 0, 0);
        vector<vector<int>> ret;
        for(auto is:mp)
        {
            vector<int> t;
            for(auto i:is.second)
            {
                t.emplace_back(i.val);
            }
            ret.emplace_back(t);
        }
        return ret;
    }
    void scan(TreeNode* t, int x, int y)
    {
        if(t == NULL)
        {
            return;
        }
        Pnt p(x,y,t->val);
        mp[x].insert(p);
        scan(t->left,x-1,y-1);
        scan(t->right,x+1,y-1);
    }
    map<int,set<Pnt>> mp;
};
