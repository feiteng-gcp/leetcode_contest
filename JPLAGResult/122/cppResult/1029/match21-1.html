#define INFL 0x3f3f3f3f3f3f3f3f
#define EPS (1e-10)
using namespace std;
typedef long long ll;
typedef pair<int,int>P;

#ifdef DEBUG
struct TreeNode{
	int val;
	TreeNode*left;
	TreeNode*right;
	TreeNode(int x):val(x),left(NULL),right(NULL){}
};
struct Interval {
    int start;
    int end;
    Interval() : start(0), end(0) {}
    Interval(int s, int e) : start(s), end(e) {}
};
#endif

class Solution{
public:
	vector<pair<P,int>>v;
	void dfs(TreeNode*t,int x,int y){
		v.push_back({P(x,y),t->val});
		if(t->left)dfs(t->left,x-1,y+1);
		if(t->right)dfs(t->right,x+1,y+1);
	}
	vector<vector<int>> verticalTraversal(TreeNode* root){
		dfs(root,0,0);
		sort(v.begin(),v.end());
		map<int,vector<int>>mp;

		for(auto p:v){
			mp[p.first.first].push_back(p.second);
		}
		vector<vector<int>>res;
		for(auto p:mp){
			res.push_back(p.second);
		}
		return res;
    }
};
