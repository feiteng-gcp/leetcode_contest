/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {
        vector<Interval> output;
        
        // Case1
        // [A.start,   A.end]
        //      [B.start, *]
        
        // Case2
        //    [A.start, *]
        // [B.start]  [B.end]
        
        // Case3
        // [A.start, A.end]
        // 
        
        int i = 0, j = 0;
        
        while (i < A.size() && j < B.size()) {
            if (A[i].start <= B[j].start) {
                if (B[j].start <= A[i].end) {
                    if (A[i].end <= B[j].end) {
                        output.push_back(Interval(B[j].start, A[i].end));
                        i++;
                    } else {
                        output.push_back(B[j]);
                        j++;
                    }
                } else {
                    i++;
                }
            } else { // A[i].start < B[j].start
                if (A[i].start <= B[j].end) {
                    if (A[i].end <= B[j].end) {
                        output.push_back(A[i]);
                        i++;
                    } else {
                        output.push_back(Interval(A[i].start, B[j].end));
                        j++;
                    }
                } else {
                    j++;
                }
            }
        }
        return output;
    }
};
