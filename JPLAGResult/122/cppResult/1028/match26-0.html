/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {
        vector<Interval> result;
        int aIdx = 0;
        int bIdx = 0;
        while (aIdx < A.size() && bIdx < B.size()) {
            int aStart = A[aIdx].start;
            int aEnd = A[aIdx].end;
            int bStart = B[bIdx].start;
            int bEnd = B[bIdx].end;
            int rStart = max(aStart, bStart);
            int rEnd = min(aEnd, bEnd);
            if (rStart <= rEnd) {
                result.push_back(Interval(rStart, rEnd));
            }
            
            if (aEnd < bEnd) {
                ++aIdx;
            } else {
                ++bIdx;
            }
        }
        
        return result;
    }
};
