/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {
        vector<Interval> result;
        int aIdx = 0;

        int bIdx = 0;
        while(aIdx < A.size() && bIdx < B.size()){
            if(A[aIdx].end < B[bIdx].start){
                aIdx++;
            } else if(B[bIdx].end < A[aIdx].start){
                bIdx++;
            } else {
                int start = max(A[aIdx].start, B[bIdx].start);
                int end = 0;
                if(A[aIdx].end < B[bIdx].end){
                    end = A[aIdx++].end;
                } else {
                    end = B[bIdx++].end;
                }
                result.push_back(Interval(start,end));
            }
        }
        return result;
    }
    
    int max(int a, int b){
        return a > b ? a : b;
    }
};
