/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
    
private:
    bool isIntersect(const Interval& a, const Interval& b) {
        return !(a.end < b.start || a.start > b.end);
    }
    
    Interval intersect(const Interval& a, const Interval& b) {
        return Interval(max(a.start, b.start), min(a.end, b.end));
    }
    
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {
        deque<Interval> a(A.begin(), A.end());
        deque<Interval> b(B.begin(), B.end());
        
        vector<Interval> ans;
        if (a.empty() || b.empty()) {
            return ans;
        }
        
        Interval pa = a.front(), pb = b.front();
        a.pop_front();
        b.pop_front();
        while (true) {
            if (isIntersect(pa, pb)) {
                ans.push_back(intersect(pa, pb));
            }
            
            bool cont = true;

            if (pa.end > pb.end) {
                if (!b.empty()) {
                    pb = b.front();
                    b.pop_front();
                } else {
                    cont = false;
                }
            } else if (pb.end > pa.end) {
                if (!a.empty()) {
                    pa = a.front();
                    a.pop_front();
                } else {
                    cont = false;
                }
            } else {
                if (!a.empty() && !b.empty()) {
                    pa = a.front();
                    a.pop_front();
                    pb = b.front();
                    b.pop_front();
                } else {
                    cont = false;
                }
            }
            if (!cont) break; 
        }
        
        return ans;
    }
};
