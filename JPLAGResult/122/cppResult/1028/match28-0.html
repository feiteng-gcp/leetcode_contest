/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {
        vector<Interval> res;
		if (A.size() == 0 || B.size() == 0) {
			return res;
		}
		int l1 = 0, l2 = 0;
		while (l1 < A.size() && l2 < B.size()) {
			if (A[l1].end < B[l2].start) {
				l1++;
			}
			else if (B[l2].end < A[l1].start) {
				l2++;
			}
			else if (B[l2].start >= A[l1].start && B[l2].start <= A[l1].end) {
				if (B[l2].end <= A[l1].end) {
					res.push_back(B[l2]);
					l2++;
				}
				else {
					Interval iv(B[l2].start, A[l1].end);
					res.push_back(iv);
					l1++;
				}
			}
			else if (A[l1].start >= B[l2].start && A[l1].start <= B[l2].end) {
				if (A[l1].end <= B[l2].end) {
					res.push_back(A[l1]);
					l1++;
				}
				else {
					Interval iv(A[l1].start, B[l2].end);
					res.push_back(iv);
					l2++;
				}
			}
		}
		return res;
    }
};
