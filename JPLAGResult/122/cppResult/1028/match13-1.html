/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {

        typedef pair<int,int>pii;
        vector<pii>e;
        for(auto a:A){
            e.push_back(pii(a.start,-1));
            e.push_back(pii(a.end,1));
        }
        for(auto a:B){
            e.push_back(pii(a.start,-2));
            e.push_back(pii(a.end,2));
        }
        sort(e.begin(),e.end());
        vector<Interval>a;
        int f=0;
        int bef=-1;
        for(pii k:e){
            int oldf=f;
            int t=k.first;
            int u=k.second;
            if(u<0)
                f|=(1<<(-u-1));
            else
                f&=~(1<<u-1);
            if(oldf!=3&&f==3)bef=t;
            if(oldf==3&&f!=3)a.emplace_back(bef,t);
        }
        return a;
    }
};
