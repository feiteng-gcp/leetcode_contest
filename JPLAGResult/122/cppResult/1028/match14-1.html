/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {

        vector<pair<int, int>> a, b;
        for (auto& ival : A) { 
            a.emplace_back(ival.end, ival.start);
        }
        for (auto& ival : B) { 
            b.emplace_back(ival.end, ival.start);
        }
        sort(begin(a), end(a));
        sort(begin(b), end(b));

        int p0 = 0;
        int p1 = 0;
        vector<Interval> res;
        while (p0 < a.size() and p1 < b.size()) {
            int l = max(a[p0].second, b[p1].second);
            int r = min(a[p0].first, b[p1].first);

            if (l <= r) {
                res.push_back(Interval(l, r));
            }
            if (a[p0].first == r) {
                ++p0;
            } else {
                ++p1;
            }
        }
        
        return res;
    }
};
