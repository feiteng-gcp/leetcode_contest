/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {
        int p1=0,p2=0;
        vector<Interval> ans;
        while(p1<A.size()&&p2<B.size())
        {
            if(A[p1].end<B[p2].start)
            {
                p1++;
            }
            else if(B[p2].end<A[p1].start)
            {
                p2++;
            }
            else
            {
                ans.push_back(Interval(max(A[p1].start,B[p2].start),min(A[p1].end,B[p2].end)));

                if(A[p1].end<B[p2].end)
                {
                    p1++;
                }
                else if(A[p1].end>B[p2].end)
                {
                    p2++;
                }
                else
                {
                    p1++,p2++;
                }
            }
        }
        return ans;
    }
};
