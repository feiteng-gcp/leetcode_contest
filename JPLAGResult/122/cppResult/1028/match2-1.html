/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {
        int i, j, m, n;
        vector<Interval> ans;
        i = j = 0;

        m = A.size();
        n = B.size();
        while (i < m && j < n) {
            if (A[i].end < B[j].start) {
                i++;
            } else if (A[i].start > B[j].end) {
                j++;
            } else {
                Interval t;
                t.start = max(A[i].start, B[j].start);
                t.end = min(A[i].end, B[j].end);
                ans.push_back(t);
                if (A[i].end < B[j].end) {
                    i++;
                } else {
                    j++;
                }
            }
        }
        return ans;
    }
};
