/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {
        int len1=A.size(),len2=B.size(),i=0,j=0;
        vector<Interval> ans;
        while(i<len1&&j<len2){
            if(A[i].end>B[j].start&&A[i].start<B[j].end){
                ans.push_back(Interval(max(A[i].start,B[j].start),min(A[i].end,B[j].end)));
            }
            if(A[i].end==B[j].start){
                ans.push_back(Interval(A[i].end,A[i].end));
            }
            if(A[i].start==B[j].end){
                ans.push_back(Interval(A[i].start,A[i].start));
            }
            if(A[i].end>B[j].end) j++;
            else if(A[i].end<B[j].end) i++;
            else{
                i++;
                j++;
            }
        }
        return ans;
    }
};
