/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {
        auto cmp = [&](Interval& l, Interval& r) {
            if (l.start == r.start) {
                return l.end < r.end;
            } else {
                return l.start < r.start;
            }
        };
        sort(A.begin(), A.end(), cmp);
        sort(B.begin(), B.end(), cmp);
        
        
        vector<Interval> res;
        res.reserve(20000);

        for (auto& p1 : A) {
            for (auto& p2 : B) {
                int l = max(p1.start, p2.start);
                int r = min(p1.end, p2.end);
                if (p2.start > p1.end) {
                    break;
                }
                
                if (l <= r) {
                    res.emplace_back(l, r);
                }
            }
        }
        sort(res.begin(), res.end(), cmp);
        
        return res;
    }
};
