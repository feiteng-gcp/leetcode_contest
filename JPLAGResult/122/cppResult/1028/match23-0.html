/**
 * Definition for an interval.
 * struct Interval {
 *     int start;
 *     int end;
 *     Interval() : start(0), end(0) {}
 *     Interval(int s, int e) : start(s), end(e) {}
 * };
 */
class Solution {
public:
    vector<Interval> intervalIntersection(vector<Interval>& A, vector<Interval>& B) {
        vector<Interval> ret;
        int pa = 0, pb = 0;
        while(pa < A.size() && pb < B.size()){
            Interval a = A[pa];
            Interval b = B[pb];
            if(a.end < b.start){
                pa++;
            }
            else if(b.end < a.start){
                pb++;
            }
            else{ // intersect
                ret.push_back(Interval(max(a.start, b.start), min(a.end, b.end)));
                if(a.end < b.end)
                    pa++;
                else
                    pb++;
            }
        }
        return ret;
    }
};
