class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int sum = 0;
        for (int i : A) if (abs(i) % 2 == 0) sum += i;
        vector<int> res;
        for (auto q : queries) {
            int i = q[1], v = q[0], origin = A[i];
            bool wasEven = (A[i] % 2 == 0);
            A[i] += v;
            bool isEven = (A[i] % 2 == 0);
            if (wasEven && isEven) sum += v;
            else if (!wasEven && isEven) sum += A[i];
            else if (wasEven && !isEven) sum -= origin;

            res.push_back(sum);
        }
        return res;
    }
};
