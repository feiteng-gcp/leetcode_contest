class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int sum = 0;
        for (int i = 0; i < A.size(); ++i)
            if ((A[i] & 1) == 0)
                sum += A[i];
        vector<int> ans;
        for (int j = 0; j < queries.size(); ++j) {
            int val = queries[j][0];
            int index = queries[j][1];
            if ((A[index] & 1) == 0)
                sum -= A[index];
            A[index] += val;
            if ((A[index] & 1) == 0)
                sum += A[index];
            
            ans.push_back(sum);
        } 
        return ans;
    }
};
