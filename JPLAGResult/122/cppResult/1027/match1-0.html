class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        vector<int> ans;
        int sum=0;
        for(int a:A)
        {
            if(a%2==0)sum+=a;
        }
        for(auto q:queries)
        {
            if(A[q[1]]%2==0)
            sum-=A[q[1]];
            A[q[1]]+=q[0];
            if(A[q[1]]%2==0)sum+=A[q[1]];
            ans.push_back(sum);
        }
        return ans;
        
    }
};
