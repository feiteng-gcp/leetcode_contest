class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int sum = 0;
        for (auto num : A) {
            if (num % 2 == 0) {
                sum += num;
            }
        }
        vector <int> ans;
        for (auto query : queries) {
            int val = query[0];
            int ind = query[1];

            if (A[ind] % 2 == 0) {
                sum -= A[ind];
            }
            A[ind] += val;
            if (A[ind] % 2 == 0) {
                sum += A[ind];
            }
            ans.push_back(sum);
        }
        return ans;
    }
};
