class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        vector<int> res;
        int sum = 0;
        for(auto a:A) {
            if(a%2==0)
                sum+=a;
        }
        for(auto q:queries) {
            int a = A[q[1]] + q[0];
            if(a%2==0) {
                if(A[q[1]]%2==0)
                    sum+= q[0];
                else
                    sum+= a;
            } else {
                if(A[q[1]]%2==0)
                    sum -= A[q[1]];
            }
            A[q[1]] = a;
            res.push_back(sum);
        }
        return res;
    }
};
