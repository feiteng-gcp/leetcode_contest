class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int sum = 0;
        for( auto n : A){
            if( n % 2 == 0 )
                sum += n;
        }
        vector<int> ans;
        for( auto query : queries){
            int diff = query[0];
            int idx = query[1];
            
            if( abs(A[idx]) % 2 == 0 )
                sum -= A[idx];
            A[idx] += diff;
            if( abs(A[idx]) % 2 == 0 )
                sum += A[idx];
            ans.push_back(sum);
        }
        return ans;
    }
};
