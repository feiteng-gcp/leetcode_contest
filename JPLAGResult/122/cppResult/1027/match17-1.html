typedef vector<int> vi;

class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& a, vector<vector<int>>& q) {
        int n = a.size();
        vi ans;

        int cur = 0;
        for (int i = 0; i < n; i++) {
            if ((a[i] & 1) == 0) {
                cur += a[i];
            }
        }
        n = q.size();
        for (int i = 0; i < n; i++) {
            vi tmp = q[i];

            int j = tmp[1];
            int val = tmp[0];
            if ((a[j] & 1) == 0) {
                cur -= a[j];
            }
            a[j] += val;
            if ((a[j] & 1) == 0) {
                cur += a[j];
            }
            ans.push_back(cur);
        }
        return ans;
    }
};
