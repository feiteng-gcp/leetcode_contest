class Solution {
public:
    int sum = 0;
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        vector<int> result;
        sum = 0;
        for_each(A.begin(), A.end(), [&](int a) { if (!(a % 2)) sum += a;});
        for (auto q : queries) {
            if (A[q[1]] % 2 == 0) sum -= A[q[1]];
            A[q[1]] += q[0];
            if (A[q[1]] % 2 == 0) sum += A[q[1]];
            result.push_back(sum);
        }
        return result;
    }
};
