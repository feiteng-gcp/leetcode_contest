class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        vector<int> res;   
        int sum = 0;
        for (auto val : A) sum += (val % 2 == 0 ? val : 0);
        for (auto q : queries) {
            int val = q[0]; int id = q[1];     
            if (abs(A[id] % 2) == abs(val % 2)) {
                sum += val;
                if (abs(A[id] % 2) == 1) sum += A[id];
            } else {
                if (abs(A[id] % 2) == 0) sum -= A[id];
            }
            A[id] += val;
            res.push_back(sum);
        }
        return res;
    }
};
