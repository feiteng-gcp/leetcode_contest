class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int ans = 0;
        for (auto i : A)
            if (i % 2 == 0)
                ans += i;
        vector<int> result;
        for (auto q : queries) {
            int val = q[0];
            int idx = q[1];
            if (A[idx] % 2 == 0)
                ans -= A[idx];
            A[idx] += val;
            if (A[idx] % 2 == 0)
                ans += A[idx];
            result.push_back(ans);
        }
        return result;
    }
};
