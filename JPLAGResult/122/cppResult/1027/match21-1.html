class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int sum = 0;
        for(int a : A) if(a % 2 == 0) sum += a;
        vector<int> result;
        for(auto q : queries){
            int val = q[0];
            int idx = q[1];
            if(A[idx] % 2 == 0) sum -= A[idx];
            A[idx] += val;
            if(A[idx] % 2 == 0) sum += A[idx];
            result.push_back(sum);
        }
        return result;
    }
};
