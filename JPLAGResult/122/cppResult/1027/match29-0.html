class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        vector<int> ret, AA(A);
        int sum = 0;
        for(int i = 0; i < A.size(); i++) {
            if (A[i] %2 ==  0)  {
                sum += A[i];
            }
        }
        
        for(int i = 0; i < queries.size(); i++) {
            int b = AA[queries[i][1]] + queries[i][0];
            if (AA[queries[i][1]] % 2 == 0)
                sum -= AA[queries[i][1]];
            if (b % 2 == 0)
                sum += b;
            ret.push_back(sum);
            AA[queries[i][1]] = b;
        }
        
        return ret;
    }
};
