class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int sum = 0;
        for (int &x : A) {
            if (x%2 == 0)
                sum += x;
        }
        vector<int> ans;
        for (auto &q : queries) {
            int val = q[0], idx = q[1];
            int bef = A[idx];
            A[idx] += val;
            int after = A[idx];
            if (bef%2 == 0) {
                if (val%2 == 0) sum += val;
                else sum -= bef;
            }
            else {
                if (val%2 == 0) {
                    int a = 1;
                }
                else {
                    sum += after;
                }
            }
            ans.push_back(sum);
        }
        return ans;
    }
};
