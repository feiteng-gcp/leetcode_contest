class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        vector<int> ans;
        int sum = 0;
        for(auto a: A) {
            if(a%2 == 0) sum += a;
        }
        for(auto q: queries) {
            int v, i;
            v = q[0];
            i = q[1];
            if(A[i]%2==0) sum -= A[i];
            A[i] += v;
            if(A[i]%2==0) sum += A[i];
            ans.push_back(sum);
        }
        return ans;
    }
};
