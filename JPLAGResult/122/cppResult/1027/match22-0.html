class Solution {
public:
   vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
      int evenSum = 0;
      for (int n : A)
      {
         if (!(n % 2))
            evenSum += n;
      }
      vector<int> result(queries.size());
      size_t i = 0;
      for (auto query : queries)
      {
         int index = query[1];
         int sum = A[index] + query[0];
         if (!(A[index]%2))
         {
            evenSum -= A[index];
         }
         if (!(sum%2))
         {
            evenSum += sum;
         }
         A[index] = sum;
         result[i++] = evenSum;
      }

      return result;
   }
};
