class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& a, vector<vector<int>>& q) {
        vector<int> ret(q.size(),0);
        for(int i=0;i<q.size();i++)
        {
            a[q[i][1]]+=q[i][0];
            ret[i]=getSum(a);
        }
        return ret;
    }
    int getSum(vector<int>& a)
    {
        int ret=0;
        for(int i=0;i<a.size();i++)
        {
            if(a[i]%2 == 0)
            {
                ret+=a[i];
            }
        }
        return ret;
    }
};
