class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int evenSum = 0;
        for (int x : A) {
            if (x % 2 == 0) {
                evenSum += x;
            }
        }
        vector<int> ans;
        for (auto& q : queries) {
            int val = q[0];
            int idx = q[1];
            int oldVal = A[idx];
            int newVal = oldVal + val;
            if (oldVal % 2 == 0) {
                evenSum -= oldVal;
            }
            if (newVal % 2 == 0) {
                evenSum += newVal;
            }
            A[idx] = newVal;
            ans.push_back(evenSum);
        }
        return ans;
    }
};
