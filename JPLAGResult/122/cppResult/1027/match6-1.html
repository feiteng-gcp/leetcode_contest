class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int s = 0;
        for (auto n : A) {
            if (n % 2 == 0) {
                s += n;
            }
        }
        
        vector<int> result;
        for (auto q : queries) {
            int i = q[1];
            int v = q[0];
            if (abs(A[i]) % 2 == 0) {
                if (abs(v) % 2 == 0) {
                    s += v;
                } else {
                    s -= A[i];
                }
            } else {
                if (abs(v) % 2 == 1) {
                    s += A[i] + v;
                }             
            }
            result.push_back(s);
            A[i]+= v;
        }
        return result;
    }
};
