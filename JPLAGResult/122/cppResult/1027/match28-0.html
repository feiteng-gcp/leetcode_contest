class Solution {
    private:
        bool isEven(int n) { return n % 2 == 0; }
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int even_sum = 0;
        for (auto a : A) if (isEven(a)) even_sum += a;
        
        vector<int> ret;
        for (auto &q : queries) {
            int val = q[0];
            int idx = q[1];
            if (isEven(A[idx])) {
                even_sum -= A[idx];
            }
            A[idx] += val;
            if (isEven(A[idx])) {
                even_sum += A[idx];
            }
            ret.push_back(even_sum);
        }
        return ret;
    }
};
