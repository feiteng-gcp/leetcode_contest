class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        vector<int> result;
        
        int answer = 0;
        for (int v : A) {
            if (v % 2 == 0) {
                answer += v;
            }
        }
        
        for (int i = 0; i < queries.size(); i++) {
            int j = queries[i][1];
            int original_value = A[j];
            
            A[j] += queries[i][0];
            
            if (A[j] % 2 == 0) {
                if (original_value % 2 == 0) {
                    answer += queries[i][0];
                } else {
                    answer += A[j];
                } 
            } else {
                if (original_value % 2 == 0) {
                    answer -= original_value;
                } 
            }
            result.push_back(answer);
        }
        return result;
    }
};
