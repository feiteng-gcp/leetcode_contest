class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        vector<int> sums;
        int sum = 0;
        for(auto num : A)
        {
            if(num % 2 == 0)
                sum += num;
        }
        
        for(auto &q : queries)
        {
            int add = q[0];
            int index = q[1];
            if(A[index] % 2 == 0)
                sum -= A[index];
            A[index] += add;
            if(A[index] % 2 == 0)
                sum += A[index];
            sums.push_back(sum);
        }
        return sums;
    }
};
