class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& Q) {
        int esum = 0;
        for(int x : A) if(x%2 == 0) esum += x;
        vector<int> ans;
        for(auto& q : Q) {
            int val = q[0];
            int idx = q[1];
            int prev = A[idx];
            A[idx] += val;
            if(A[idx]%2 == 0 && prev%2 != 0) esum += A[idx];
            if(A[idx]%2 == 0 && prev%2 == 0) esum += val;
            else if(A[idx]%2 != 0 && prev%2 == 0) esum -= prev;
            ans.push_back(esum);
        }
        return ans;
    }
};
