class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int s = 0;
        for (int a : A) if (a % 2 == 0) {
            s += a;
        }
        vector<int> ans;
        for (auto &q : queries) {
            int val = q[0];
            int index = q[1];
            if (A[index] % 2 == 0) {
                if (val % 2 == 0) {
                    s += val;
                } else {
                    s -= A[index];
                }
            } else {
                if (val % 2 == 0) {
                    
                } else {
                    s += A[index] + val;
                }
            }
            A[index] += val;
            ans.push_back(s);
        }
        return ans;
    }
};
