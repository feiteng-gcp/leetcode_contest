class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int sum = 0;

        for (int x: A) {
            if (x % 2 == 0) {
                sum += x;
            }
        }
        vector<int> res;
        for (auto &Q: queries) {
            int val = Q[0], idx = Q[1];

            if (A[idx] % 2 == 0) {
                sum -= A[idx];
            }
            A[idx] += val;
            if (A[idx] % 2 == 0) {
                sum += A[idx];
            }
            res.push_back(sum);
        }
        return res;
    }
};
