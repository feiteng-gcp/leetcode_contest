class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        vector<int> re;
        int ans=0;
        for(int i=0;i<A.size();++i) 
            if(A[i]&1^1) ans+=A[i];
        for(auto v:queries)
        {
            if(A[v[1]]&1^1) ans-=A[v[1]];
            A[v[1]]+=v[0];
            if(A[v[1]]&1^1) ans+=A[v[1]];
            re.pb(ans);
        }
        return re;
    }
};
