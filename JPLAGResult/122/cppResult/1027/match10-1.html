class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& A, vector<vector<int>>& queries) {
        int even = 0;
        for (int n: A) {
            if ((n & 1) == 0) {
                even += n;
            }
        }
        
        vector<int> ret;
        for (vector<int>& q: queries) {
            int v = q[0];

            int index = q[1];
            if ((A[index] & 1) == 0) {
                even -= A[index];
            }
            A[index] += v;
            if ((A[index] & 1) == 0) {
                even += A[index];
            }
            ret.push_back(even);
        }
        return ret;
    }
};
