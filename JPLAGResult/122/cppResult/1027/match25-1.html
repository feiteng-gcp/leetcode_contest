class Solution {
public:
    vector<int> sumEvenAfterQueries(vector<int>& a, vector<vector<int>>& q) {
        int sum=0;
        for(int i=0;i<a.size();i++)
        if(a[i]%2==0)
            sum+=a[i];
        vector<int>ret;
        for(int i=0;i<q.size();i++){
            int tmp=a[q[i][1]];
            a[q[i][1]]+=q[i][0];
            if(tmp%2==0){
                if(q[i][0]%2==0)
                    sum+=q[i][0];
                else
                    sum-=tmp;
            }else{
                if(q[i][0]%2!=0)
                    sum+=a[q[i][1]];
            }
            ret.push_back(sum);
        }
        return ret;
    }
};
