struct TreeNode{
	int val;
	TreeNode*left;
	TreeNode*right;
	TreeNode(int x):val(x),left(NULL),right(NULL){}
};
#endif

class Solution{
public:
	string ans;
	string s;
	void dfs(TreeNode*t){
		s+=t->val+'a';

		if((!t->left)&&(!t->right)){
			reverse(s.begin(),s.end());
			ans=min(ans,s);
			reverse(s.begin(),s.end());
		}
		if(t->left)dfs(t->left);
		if(t->right)dfs(t->right);
		s.pop_back();
	}
	string smallestFromLeaf(TreeNode* root) {
		ans=string(3000,'z');
        dfs(root);
        return ans;
    }
};
