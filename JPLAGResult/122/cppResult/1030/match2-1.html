/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    string smallestFromLeaf(TreeNode* root) {
        if (root == NULL) {
            return "";
        }
        vector<string> strs;
        strs.reserve(1100);
        string s;
        dfs(root, s, strs);
        for (auto& s : strs) {
            reverse(s.begin(), s.end());
        }
        sort(strs.begin(), strs.end());
        if (strs.empty()) {
            return "";
        }
        return strs[0];
    }
    
private:
    void dfs(TreeNode* root, string& prefix, vector<string>& res) {
        if (root == NULL) {
            res.push_back(prefix);
        }
        
        prefix.push_back('a' + root->val);
        if (root->left == NULL && root->right == NULL) {
            res.push_back(prefix);

        } else {
            if (root->left) {
                dfs(root->left, prefix, res);
            }
            if (root->right) {
                dfs(root->right, prefix, res);
            }
        }
        prefix.pop_back();
    }
};
