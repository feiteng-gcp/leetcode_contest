/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<string> all_paths;
    void Rec(TreeNode* root, string path) {
        string cur(1, 'a' + root->val);
        
        if (!root->left && !root->right) {
            auto new_path = path + cur;
            reverse(new_path.begin(), new_path.end());
            all_paths.push_back(new_path);
            return;

        }
        
        if (root->left) {
            Rec(root->left, path + cur);
        }
        if (root->right) {
            Rec(root->right, path + cur);
        }
    }
    
    string smallestFromLeaf(TreeNode* root) {
        all_paths.clear();
        Rec(root, "");
        sort(all_paths.begin(), all_paths.end());
        
        return all_paths[0];
    }
};
