template<class T> inline void amax(T &x, const T &y) { if (x<y) x=y; }

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    string ans;

    void rec(TreeNode*x, string &cur) {
	cur.push_back(x->val + 'a');
	if (x->left == 0 && x->right == 0) {
	    string t = cur;
	    reverse(t.begin(), t.end());
	    amin(ans, t);
	}
	if (x->left) rec(x->left, cur);
	if (x->right) rec(x->right, cur);
	cur.pop_back();
    }

    string smallestFromLeaf(TreeNode* root) {
	ans = "{";
	string cur;
	rec(root, cur);
	return ans;
    }
};
