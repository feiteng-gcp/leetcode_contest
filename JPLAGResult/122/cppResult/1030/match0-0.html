/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<string> set;
    vector<int> st;
    void goTree(TreeNode* t){
        st.push_back(t->val);
        if(!t->left && !t->right){
            string s;
            for(int i=st.size()-1;i>=0;--i){
                s.push_back(st[i]+'a');
            }
            set.push_back(s);
        }
        if(t->left){
            goTree(t->left);
        }
        if(t->right){
            goTree(t->right);
        }
        st.pop_back();
    }
    string smallestFromLeaf(TreeNode* root) {
        goTree(root);
        sort(set.begin(),set.end());
        return set[0];
    }
};
