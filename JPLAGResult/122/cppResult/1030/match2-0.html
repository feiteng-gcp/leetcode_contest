/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    string smallest;
public:
    string smallestFromLeaf(TreeNode* root) {
        if (root == nullptr) {
            return "";
        }
        
        string minStr;
        iterate(root, "", minStr);
        return minStr;
    }
    
    void iterate(TreeNode *node, const string &prefix, string &minStr) {
        cout << prefix << endl;
        string newPrefix = string(1, 'a' + node->val);
        newPrefix += prefix;
        
        if (node->left == nullptr && node->right == nullptr) {
            if (minStr.empty()) {
                minStr = newPrefix;
            } else {
                minStr = min(newPrefix, minStr);
            }
        } else {
            if (node->left != nullptr) {
                iterate(node->left, newPrefix, minStr);
            }
            if (node->right != nullptr) {
                iterate(node->right, newPrefix, minStr);
            }
        }
    }
};
