/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    String res = null;
    public String smallestFromLeaf(TreeNode root) {
        helper(new StringBuffer(), root);
        return res;
    }
    
    private void helper(StringBuffer sb, TreeNode node) {
        if(node.left == null && node.right == null) {
            String s = (char)(node.val + 'a') + sb.toString();
            res = compare(res, s);
            return;
        }
        
        sb.insert(0, (char)(node.val + 'a'));
        if(node.left == null) {
            helper(sb, node.right);
        } else if(node.right == null) {
            helper(sb, node.left);
        } else {
            helper(sb, node.left);
            helper(sb, node.right);
        }
        sb.deleteCharAt(0);
        
    }
    
    private String compare(String a, String b) {
        if(a == null)
            return b;
        for(int i = 0; i < a.length() && i < b.length(); i++) {
            if(a.charAt(i) < b.charAt(i)) {
                return a;
            } else if(a.charAt(i) > b.charAt(i)) {
                return b;
            }
        }
        return a.length() <= b.length() ? a : b;
    }
}
