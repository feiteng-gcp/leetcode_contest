	class S
olution {
		String ret = "~";
	    public String smallestFromLeaf(TreeNode root) {
	        dfs(root, "");
	        return ret;
	    }
	    
	    void dfs(TreeNode cur, String s)
	    {
	    	if(cur == null)return;
	    	s = (char)('a'+cur.val) + s;
	    	if(cur.left == null && cur.right == null){
	    		if(s.compareTo(ret) < 0){
	    			ret = s;
	    		}
	    	}
	    	dfs(cur.left, s);
	    	dfs(cur.right, s);
	    }
	}
