/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    String res;
    
    public String smallestFromLeaf(TreeNode root) {
        dfs(root, new ArrayDeque<>());
        
        return res;
    }
    
    private void dfs(TreeNode node, ArrayDeque<Integer> stack) {
        stack.push(node.val);
        if (node.left != null) dfs(node.left, stack);
        if (node.right != null) dfs(node.right, stack);

        if (node.left == null && node.right == null) {
            StringBuilder res = new StringBuilder();
            for (Integer i : stack) res.append((char) (i + 'a'));

            String resString = res.toString();
            if (this.res == null || this.res.compareTo(res.toString()) > 0) this.res = resString;
        }

        stack.poll();
    }
}
