/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public String smallestFromLeaf(TreeNode root) {
        return dfs(root, new StringBuilder());
    
}
    
    String dfs(TreeNode root, StringBuilder sb)
    {
        if(root == null) return null;
        sb.append((char) ('a' + root.val)); // append root
        if(root.left == null && root.right == null)
        {
            StringBuilder sb2 = new StringBuilder(sb);
            return sb2.reverse().toString();
        }
        int len = sb.length();
        String left = dfs(root.left, sb);
        sb.setLength(len);
        String right = dfs(root.right, sb);
        if(left == null || right == null) return left == null ? right : left;
        if(left.compareTo(right) < 0) return left;
        return right;
    }
}
