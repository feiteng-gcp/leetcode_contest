/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    String ret = "";
    public String smallestFromLeaf(TreeNode root) {
        ret = "";
        helper(root,"");
        return ret;
    
}
    private void helper(TreeNode root,String s){
        if(root==null) return;
        char c =(char)('a'+root.val);
        s = c+s;
        if(root.left==null&&root.right==null){
            if(ret.isEmpty()||s.compareTo(ret)<0){
                ret = s;
            }
        }
        helper(root.left,s);
        helper(root.right,s);
    }
}
