/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
  String res;
  public String smallestFromLeaf(TreeNode root) {
    if(root == null){
        return "";
    }
    
res = "";
    helper(root, "");
    return res;
  }
  
  private void helper(TreeNode node, String tmp){
    tmp =((char)('a' + node.val)) + tmp;
    if(node.left == null && node.right == null){
      res = resCmp(tmp);
      return;
    }
    if(node.left != null){
      helper(node.left, tmp);
    }
    if(node.right != null){
      helper(node.right, tmp);
    }
  }
  
  private String resCmp(String tmp){
    if(res.isEmpty()){
      return tmp;
    }
    char[] rChars = res.toCharArray();
    char[] tChars = tmp.toCharArray();
    
    for(int i = 0; i< Math.min(rChars.length, tChars.length); i++){
      if(rChars[i] < tChars[i]){
        return res;
      }
      else if(rChars[i] > tChars[i]){
        return tmp;
      }
    }
    return rChars.length < tChars.length ? res : tmp;
  }
}
