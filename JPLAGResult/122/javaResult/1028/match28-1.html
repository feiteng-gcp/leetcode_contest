/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public Interval[] intervalIntersection(Interval[] A, Interval[] B) {
        int n1 = A.length; 
        int n2 = B.length; 
        if (n1 == 0 && n2 == 0) {
            return new Interval[0]; 
        }
        
        List<Interval> l = new ArrayList<>(); 
        for (int i = 0; i < n1; i++) {
            l.add(A[i]);
        }
        for (int i = 0; i < n2; i++) {
            l.add(B[i]);
        }
        Collections.sort(l, new Comparator<Interval>() {
            public int compare(
Interval a, Interval b) {
                if (a.start == b.start) {
                    return a.end - b.end;
                } else {
                    return a.start - b.start;
                }
            }
        });
        
        // for (int i = 0; i < n1 + n2; i++) {
        //     System.out.println(l.get(i).start + " " + l.get(i).end);
        // }
        List<Interval> ret = new ArrayList<>(); 
        Interval cur = l.get(0);
        for (int i = 1; i < n1 + n2; i++) {
            if (cur.end >= l.get(i).start) {
                if (cur.end >= l.get(i).end) {
                    ret.add(l.get(i));
                } else {
                    ret.add(new Interval(l.get(i).start, cur.end));
                    cur.end = l.get(i).end;
                }
                
            } else {
                cur = l.get(i);
            
}
        }
        Interval[] res = new Interval[ret.size()]; 
        for (int i = 0; i < ret.size(); i++) {
            res[i] = ret.get(i);
        }
        return res; 
    }
}
