/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public Interval[] intervalIntersection(Interval[] A, Interval[] B) {
        if (A == null || A.length == 0 || B == null || B.length == 0) return new Interval[0];
        int l1 = 0, l2 = 0;
        List<Interval> res = new ArrayList<>();
        while (l1 < A.length && l2 < B.length) {
            Interval i1 = A[l1];
            Interval i2 = B[l2];
            if (Math.max(i1.start, i2.start) <= Math.min(i1.end, i2.end)) {
                res.add(new Interval(Math.max(i1.start, i2.start), Math.min(i1.end, i2.end)));
            }
            if (i1.end < i2.end) l1++;
            else l2++
;
        }
        Interval[] arr = new Interval[res.size()];
        for (int i = 0; i < res.size(); ++i) {
            arr[i] = res.get(i);
        }
        return arr;
    }
}
