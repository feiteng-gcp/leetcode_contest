/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public Interval[] intervalIntersection(Interval[] A, Interval[] B) {
        if (A == null || B == null) {
            return new Interval[0];
        }
        
        List<Interval> results = new ArrayList<>();
        int firstIndex = 0;
        int secondIndex = 0;
        while (firstIndex < A.length && secondIndex < B.length) {
            Interval first = A[firstIndex];
            Interval second = B[secondIndex];
            
            if (first.end < second.start) {
                firstIndex++;
            } else if (second.end < first.start) {
                secondIndex++;
            } else {
                int newStart = Math.max(first.start, second.start);
                int newEnd = Math.min(first.end, second.end);
                results.add(new Interval(newStart, newEnd));
                if (first.end < second.end) {
                    firstIndex++;
                } else {
                    secondIndex++;
                }
            }
        }
        
        Interval[] intervals = new Interval[results.size()];
        for (int i = 0; i < results.size(); i++) {
            intervals[i] = results.get(i);
        }
        
        return intervals;
    }
}
