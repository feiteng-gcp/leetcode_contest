/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */

class Solution {
    public Interval[] intervalIntersection(Interval[] A, Interval[] B) {
        int i = 0, j = 0;
        List<Interval> list = new ArrayList<>();
        while(i < A.length && j < B.length){
            Interval a = A[i];
            Interval b = B[j];
            if(a.start <= b.start){
                if(b.start > a.end){
                    i++;
                }else{
                        if(a.end <= b.end){
                            list.add(new Interval(b.start, a.end));
                            i++;
                        }else{
                            list.add(new Interval(b.start, b.end));
                            j++;
                        }
                }

            }else{
                if(a.start > b.end){
                    j++;
                }else{
                    if(b.end <= a.end){
                        list.add(new Interval(a.start, b.end));
                        j++;
                    }else{
                        list.add(new Interval(a.start, a.end));
                        i++;
                    }
                }
            }
        }
        Interval[] res = new Interval[list.size()];
        for(i = 0; i < list.size(); i++)
{
            res[i] = list.get(i);
        }
        return res;
    }
}
