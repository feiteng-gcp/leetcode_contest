/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public Interval[] intervalIntersection(Interval[] A, Interval[] B) 
{
        int indexA = 0, indexB = 0;
        List<int[]> ret = new ArrayList<>();
        while(indexA < A.length && indexB < B.length) {
            if(A[indexA].end < B[indexB].start) {
                indexA ++;
            }
            else if(A[indexA].start > B[indexB].end){
                indexB ++;
            }
            else {
                if(A[indexA].start >= B[indexB].start) {
                    ret.add(new int[]{A[indexA].start, Math.min(A[indexA].end, B[indexB].end)});
                }
                else {
                    ret.add(new int[]{B[indexB].start, Math.min(A[indexA].end, B[indexB].end)});
                }
                if(A[indexA].end < B[indexB].end) indexA ++;
                else indexB ++
;
            }
        }
        Interval[] retArr = new Interval[ret.size()];
        for(int i = 0; i<ret.size(); i++) retArr[i] = new Interval(ret.get(i)[0], ret.get(i)[1]);
        return retArr;
    }
}
