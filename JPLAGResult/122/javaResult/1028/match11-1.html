/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */

class Solution {
 public Interval[] intervalIntersection(Interval[] A, Interval[] B) {
    int i = 0, j = 0;
    List<Interval> interList = new ArrayList<>();
    while (i < A.length && j < B.length) {
      Interval a = A[i];
      Interval b = B[j];
      int as = a.start, ar = a.end, bs = b.start, br = b.end;
      if (as > br) {
        j++;
      } else if (ar < bs) {
        i++;
      } else if (as > bs) {
        if (ar < br) {
          interList.add(a);
          i++;
        } else {
          Interval tmp = new Interval(as, br);
          interList.add(tmp);
          j++;
        }
      } else {
        if (br < ar) {
          interList.add(b);
          j++;
        } else {
          Interval temp = new Interval(bs, ar);
          interList.add(temp);
          i++;
        }
      }
    }

    Interval[] res = new Interval[interList.size()];
    for (int k = 0; k < interList.size(); k++) 
{
      res[k] = interList.get(k);
    }
    return res;
  }
}
