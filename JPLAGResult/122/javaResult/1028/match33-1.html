/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public Interval[] intervalIntersection(Interval[] A, Interval[] B) {
        if (A == null || A.length == 0 || B == null || B.length == 0) {
            return new Interval[0];
        }
        
        Queue<int[]> pq = new PriorityQueue<int[]>((it1, it2) -> {
            if (it1[0] == it2[0]) {
                return it1[1] <= it2[1] ? -1 : 1;
            }
            return it1[0] <= it2[0] ? -1 : 1;
        });
        
        for (Interval it : A) {
            pq.offer(new int[]{it.start, 0, 0});
            pq.offer(new int[]{it.end, 1, 0});
        }
        for (Interval it : B) {
            pq.offer(new int[]{it.start, 0, 1});
            pq.offer(new int[]{it.end, 1, 1});
        }
        
        List<Interval> result = new ArrayList<Interval>();
        boolean[] statusInfo = new boolean[2];
        int[] posInfo = new int[2];
        while (!pq.isEmpty()) {
            int[] head = pq.poll();
            int group = head[2];
            int pos = head[0];
            int open = head[1];
            
            if (open == 0) {
                statusInfo[group] = true;
                posInfo[group] = pos;
            } else {
                if (statusInfo[0] && statusInfo[1]) {
                    int start = Math.max(posInfo[0], posInfo[1]);
                    result.add(new Interval(start, pos));
                    
                    statusInfo[group] = false;
                    
                    int other = (group == 1 ? 0 : 1);
                    
                    // break into different parts to avoid duplicates
                    posInfo[other] = pos;
                } else {
                    statusInfo[group] = false;
                }
            
}
        }
        
        Interval[] finalResult = new Interval[result.size()];
        for (int i = 0; i < finalResult.length; i++) {
            finalResult[i] = result.get(i);
        }
        return finalResult;
    }
}
