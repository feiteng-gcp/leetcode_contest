/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public Interval[] intervalIntersection(Interval[] A, Interval[] B) {
        if(A.length == 0 || B.length == 0){
            return new Interval[0];
        }else{
            int i = 0;
            int j = 0;
            ArrayList<Interval> arr = new ArrayList<>();
            while(i < A.length && j < B.length){
                
if(A[i].end < B[j].start){
                    i++;
                }else if(A[i].start > B[j].end){
                    j++;
                }else{
                    int start = Math.max(A[i].start, B[j].start);
                    int end = Math.min(A[i].end, B[j].end);
                    arr.add(new Interval(start, end));
                    // i++;
                    // j++;
                    if(A[i].end == B[j].end){
                        i++;
                        j++;
                        
                    }else if(A[i].end < B[j].end){
                        i++;
                    }else{
                        j++;
                    }
                
}
            }
            Interval[] newRes = new Interval[arr.size()];
            for(int k = 0; k < arr.size(); k++){
                newRes[k] = arr.get(k);
            }
            return newRes;
        }
        
    }
}
