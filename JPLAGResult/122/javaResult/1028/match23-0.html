/**
 * Definition for an interval.
 * public class Interval {
 *     int start;
 *     int end;
 *     Interval() { start = 0; end = 0; }
 *     Interval(int s, int e) { start = s; end = e; }
 * }
 */
class Solution {
    public Interval[] intervalIntersection(Interval[] A, Interval[] B) {
        int n = A.length;
        int m = B.length;
        int i = 0;
        int j = 0;
        List<Interval> res = new ArrayList<>();
        while (i < n && j < m) {
            int as = A[i].start;
            int ae = A[i].end;
            int bs = B[j].start;
            int be = B[j].end;
            if (as > be) {
                j++;
                continue;
            }
            if (bs > ae) {
                i++;
                continue;
            }
            if (be >= ae) {
                i++;
            } else {
                j++;
            }
            res.add(new Interval(Math.max(as, bs), Math.min(ae, be)));
        }
        Interval[] ans = new Interval[res.size()];
        for (int k = 0; k < res.size(); k++) {
            ans[k] = res.get(k);
        }
        return ans;
    }
}
