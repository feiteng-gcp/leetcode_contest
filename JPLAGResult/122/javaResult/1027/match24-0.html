class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int[] re = new int[queries.length];
        int sum = 0;
        for(int i: A) {
            if(i%2 == 0) sum += i;
        }
        for(int i = 0; i < queries.length; i++) {
            int v = queries[i][0], index = queries[i][1];
            if(v % 2 == 0) {
                if(A[index] % 2 == 0) sum += v;
            }
            else {
                if(A[index] % 2 == 0) sum -= A[index];
                else sum += (A[index] + v);
            }
            A[index] += v;
            re[i] = sum;
        }
        return re;
    }
}
