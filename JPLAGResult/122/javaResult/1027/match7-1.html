class Solution {
    
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int sum = 0;
        for (int i=0; i<A.length; i++) {
            if (A[i] % 2 == 0) {
                sum += A[i];
            }
        }
        int[] ans = new int[queries.length];
        for (int i=0; i<queries.length; i++) {
            
int old = A[queries[i][1]];
            if (old % 2 == 0) {
                sum -= old;
            }
            A[queries[i][1]] += queries[i][0];
            if (A[queries[i][1]] % 2 == 0) {
                sum += A[queries[i][1]];
            }
            ans[i] = sum;
        }
        return ans;
    }
}
