class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] qs) {
        
        int N = qs.length;
        int[] ans = new int[N];
        
        int sum =0;
        for(int i=0;i<A.length;i++){
            if(A[i]%2==0)
                sum+=A[i];
        }
        
        for(int i=0;i<N;i++){
            int d = qs[i][0], idx = qs[i][1];
            int orig = A[idx];
            int newV = A[idx]= orig+ d;
            
            if(orig%2==0){
                if(newV%2==0){
                    ans[i]= sum-orig+newV;
                }else{
                    ans[i]= sum-orig;
                }
            }else{
                if(newV%2==0){
                    ans[i]= sum+newV;
                }else{
                    ans[i]= sum;
                }
            }
            sum = ans[i];
        }
        return ans;
        
    }
}
