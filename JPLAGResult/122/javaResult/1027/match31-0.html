class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        if (A == null || A.length == 0) {
            return new int[0];
        }
        
        int[] result = new int[queries.length];
        int sum = 0;
        
        for (int i : A) {
            if (i % 2 == 0) {
                sum += i;
            }
        }
        
        for (int i = 0; i < queries.length; i++) {
            int diff = queries[i][0];
            int index = queries[i][1];
            
            int pre = A[index];
            
            if (pre % 2 == 0) {
                sum = sum - pre;
            }
            
            A[index] = A[index] + diff;
            
            if (A[index] % 2 == 0) {
                sum += A[index];
            }
            
            result[i] = sum;
        }
        
        return result;
    }
}
