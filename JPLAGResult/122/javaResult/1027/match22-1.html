class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int len = queries.length;
        int[] res = new int[len];
        int evenSum = 0;
        
        for(int a : A) {
            if(a % 2 == 0) {
                evenSum += a;
            
}
        }
        
        for(int i = 0; i < len; i++) {
            int[] q = queries[i];
            int v = q[0];
            int idx = q[1];
            
            
if(A[idx] % 2 == 0) {
                if(v % 2 == 0) {
                    res[i] = evenSum + v;
                    evenSum += v;
                } else {
                    res[i] = evenSum - A[idx];
                    evenSum -= A[idx];
                }
            } else{
                if(v % 2 == 0) {
                    res[i] = evenSum;
                } else {
                    res[i] = evenSum + A[idx] + v;
                    evenSum += A[idx] + v;
                }
            }
            
            A[idx] += v;
        }
        
        return res;
        
    }
}
