class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int[] ans = new int[queries.length];
        int sum = 0;
        for (int num : A){
            if (num%2 == 0)
                sum += num;
        }
        
        
        for (int i = 0; i < queries.length; i++){
            int[] query = queries[i];
            int prev = A[query[1]];
            int next = A[query[1]] + query[0];
            if (prev%2 == 0 && next%2 == 0){
                sum += query[0];
            }
            else if (prev%2 == 0 && next%2 != 0){
                sum -= prev;
            }
            else if (prev%2 != 0 && next%2 == 0){
                sum += next;
            }
            A[query[1]] = next;
            ans[i] = sum;
        }
        return ans;
    }
}
