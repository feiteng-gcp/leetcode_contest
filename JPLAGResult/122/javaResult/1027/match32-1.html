class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int[] res = new int[queries.length];
        int sum = 0;
        for(int a: A) {
            if(a % 2 == 0) sum += a
;
        }
        
        for(int i = 0; i < queries.length; i++){
            int[] query = queries[i];
            int idx = query[1];
            int val = query[0];
            int tmp = A[idx];
            int later = tmp + val;
            if(tmp %2 == 0){
                sum -= tmp;
            }
            if(later %2 == 0){
                sum += later;
            
}
            res[i] = sum;
            A[idx] = later;
            
        }
        return res;
    }
}
