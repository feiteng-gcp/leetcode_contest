class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        long sum = 0;
        for (int i = 0; i < A.length; ++i) {
            if (A[i]%2 == 0) {
                sum += A[i];
            }
        }
        int[] ret = new int[queries.length];
        
for (int i = 0; i < queries.length; ++i) {
            int ind = queries[i][1];
            int val = queries[i][0];
            if (A[ind]%2 == 0) {
                sum -= A[ind];
            }
            A[ind] += val;
            if (A[ind]%2 == 0) {
                sum += A[ind];
            }
            ret[i] = (int)sum;
        }
        return ret;
    }
}
