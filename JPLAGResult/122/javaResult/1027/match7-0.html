class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int even = 0;
        for (int x : A) {
            if (x % 2 == 0) {
                even += x;
            }
        }
        int[] res = new int[queries.length];
        int i = 0;
        for (int[] q : queries) {
            int val = q[0];
            int index = q[1];
            if (A[index] % 2 == 0) {
                even -= A[index];
            }
            A[index] += val;
            if (A[index] % 2 == 0) {
                even += A[index];
            }
            res[i++] = even;
        }
        return res;
    }
}
