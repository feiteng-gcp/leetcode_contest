class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int len = queries.length;
        
int[] res = new int[len];
        int even = 0;
        for (int i = 0; i < A.length; i ++) {
            if (A[i] % 2 == 0) even += A[i];
        }
        
        int idx = 0;
        for (int[] query : queries) {
            
if (A[query[1]] % 2 == 0)
                even -= A[query[1]];
            A[query[1]] += query[0];
            if (A[query[1]] % 2 == 0) even += A[query[1]];
            res[idx++] = even;
        }
        
        return res;
    }
}
