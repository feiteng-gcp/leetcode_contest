class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int sum = 0;
        for (int a : A) {
            if (a % 2 == 0) {
                sum += a;
            }
        }
        
        int n = queries.length;
        int[] res = new int[n];
        for (int q = 0; q < n; q++) {
            int m = A[queries[q][1]] + queries[q][0];
            if (A[queries[q][1]] % 2 == 0) {
                sum -= A[queries[q][1]];
            }
            if (m % 2 == 0) {
                sum += m;
            }
            res[q] = sum;
            A[queries[q][1]] = m;
        }
        
        return res;
    }
}
