class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int sum = 0;
        for(int x: A) {
            if(x%2 == 0) {
                sum += x;
            
}
        }
        int k = queries.length;
        int[] res = new int[k];
        for(int i=0; i<k; i++) {
            int index = queries[i][1];
            int val = queries[i][0];
            if(A[index]%2 == 0) {
                sum -= A[index];
            }
            A[index] += val;
            if(A[index]%2 == 0) {
                sum += A[index];
            }
            res[i] = sum;
        }
        return res;
    }
}
