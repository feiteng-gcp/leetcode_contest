class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int[] ret = new int[queries.length];
        int evenSum = 0;
        for(int v : A){
            if(v%2==0)
                evenSum += v;
        }
        for(int i=0;i<queries.length;i++){
            int index = queries[i][1];
            int val = queries[i][0];
            if(A[index]%2==0){
                evenSum -= A[index];
            }
            A[index] +=val;
            if(A[index]%2==0){
                evenSum += A[index];
            
}
            ret[i] = evenSum;
        }
        return ret;
    }
}
