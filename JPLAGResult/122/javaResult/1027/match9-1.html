class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int n = A.length;
        int m = queries.length;
        int ans[] = new int[m];
        
int sum = 0;
        for (int i = 0; i < n; i++) {
            if (A[i] % 2 == 0) {
                sum += A[i];
            }
        }
        
for (int i = 0; i < m; i++) {
            int a = queries[i][0];
            int b = queries[i][1];
            if (A[b] % 2 == 0) {
                sum -= A[b];
            }
            A[b] += a;
            if (A[b] % 2 == 0) {
                sum += A[b];
            }
            ans[i] = sum;
        }
        return ans;
    }
}
