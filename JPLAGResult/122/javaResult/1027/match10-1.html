class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int sum = 0;
        for(int a: A){
            if(a%2 == 0)
                sum += a
;
        }
        
        int[] result = new int[A.length];
        int i=0;
        for(int[] query: queries){
            int val = query[0];
            int index = query[1];
            int old = A[index];
            A[index] += val;
            if(A[index]%2 == 0){
                if(old %2 == 0){
                    sum -= old;
                } 
                sum += A[index];
            } else {
                if(old %2 == 0){
                    sum -= old;
                }
            
}
            result[i++] = sum;
        }
        
        return result;
    }
}
