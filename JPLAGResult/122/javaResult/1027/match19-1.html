class Solution
{
    public int[] sumEvenAfterQueries(int[] a, int[][] queries)
    {
        int[] ans = new int[queries.length];
        int evenSum = 0;
        for (int i : a)
        {
            if ((i & 1) == 0) evenSum += i;
        }

        int i = 0;
        
for (int[] q : queries)
        {
            int val = q[0], idx = q[1];
            if ((a[idx] & 1) == 0) evenSum -= a[idx];
            a[idx] += val;
            if ((a[idx] & 1) == 0) evenSum += a[idx];
            ans[i++] = evenSum;
        }
        return ans;
    }
}
