class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        
int q_len = queries.length;
        int[] res = new int[q_len];
        
        int evenSum = 0;
        for (int a : A) {
            if (a % 2 == 0) {
                evenSum += a;
            }
        }
        
        for (int i = 0; i < q_len; i++) {
            int[] q = queries[i];
            int a_prev = A[q[1]];
            A[q[1]] += q[0];
            int a_after = A[q[1]];
            
            
if (a_prev % 2 == 0) {
                evenSum -= a_prev;
            }
            if (a_after % 2 == 0) {
                evenSum += a_after;
            }
            
            res[i] = evenSum;
        }
        
        return res;
    }
}
