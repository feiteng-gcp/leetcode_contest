class Solution {
    public int[] sumEvenAfterQueries(int[] A, int[][] queries) {
        int n = queries.length; 
        int[] ret = new int[n]; 
        int val = 0;
        for (int i = 0; i < n; i++) {
            if (A[i] % 2 == 0) {
                val += A[i]; 
            }
        }
        for (int i = 0; i < n; i++) {
            int idx = queries[i][1];
            int add = queries[i][0]; 
            if (A[idx] % 2 == 0) {
                val -= A[idx];
            }
            A[idx] += add; 
            if (A[idx] % 2 == 0) {
                val += A[idx]; 
            }
            ret[i] = val;
        }
        return  ret; 
       
    }
}
