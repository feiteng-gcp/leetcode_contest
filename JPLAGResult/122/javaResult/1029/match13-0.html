/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<List<Integer>> verticalTraversal(TreeNode root) {
        if (root == null) {
            return new ArrayList<List<Integer>>();
        }
        
        // x -> [y, value]
        Map<Integer, List<int[]>> map = new HashMap<Integer, List<int[]>>();
        searchHelper(0, 0, root, map);
        
        List<List<Integer>> result = new ArrayList<List<Integer>>();
        List<Integer> keys = new ArrayList<Integer>(map.keySet());
        Collections.sort(keys);
        for (int key : keys) {
            List<int[]> value = map.get(key);
            Collections.sort(value, (x1, x2) -> {
                if (x1[0] == x2[0]) {
                    return x1[1] - x2[1];
                }
                return x1[0] >= x2[0] ? -1 : 1;
            });
            
            List<Integer> answer = new ArrayList<Integer>();
            for (int[] v : value) {
                answer.add(v[1]);
            }
            result.add(answer);
        }
        return result;
    }
    
    private void searchHelper(int x, int y, TreeNode current, Map<Integer, List<int[]>> map) {
        if (!map.containsKey(x)) {
            map.put(x, new ArrayList<int[]>());
        }
        map.get(x).add(new int[]{y, current.val});
        
        if (current.left != null) {
            searchHelper(x - 1, y - 1, current.left, map);
        }
        if (current.right != null) {
            searchHelper(x + 1, y - 1, current.right, map);
        }
    }
}
