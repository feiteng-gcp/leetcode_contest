/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    private Map<Integer, Map<Integer, List<Integer>>> map = new HashMap<>();
    
    public List<List<Integer>> verticalTraversal(TreeNode root) {
        map.clear();
        _verticalTraversal(root, 0, 0);
        return map.entrySet().stream()
            .sorted(Comparator.comparingInt(Map.Entry::getKey))
            .map(Map.Entry::getValue)
            .map(m -> m.entrySet().stream()
                 .sorted(Comparator.comparingInt(Map.Entry::getKey))
                 .map(Map.Entry::getValue)
                 .flatMap(l -> l.stream().sorted(Integer::compareTo))
                 .collect(Collectors.toList())
            ).collect(Collectors.toList());
    
}
    
    private void _verticalTraversal(TreeNode root, int X, int Y) {
        if (root == null) return;
        Map<Integer, List<Integer>> m = map.get(X);
        if (m == null) {
            m = new HashMap<>();
            map.put(X, m);
        }
        
        List<Integer> list = m.get(Y);
        if (list == null) {
            list = new ArrayList<>();
            m.put(Y, list);
        }
        list.add(root.val);
        _verticalTraversal(root.left, X - 1, Y + 1);
        _verticalTraversal(root.right, X + 1, Y + 1);       
    }
}
