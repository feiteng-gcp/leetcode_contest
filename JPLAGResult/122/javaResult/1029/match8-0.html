/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    TreeMap<Integer, TreeMap<Integer, TreeSet<Integer>>> map = new TreeMap<Integer, TreeMap<Integer, TreeSet<Integer>>>();

    public List<List<Integer>> verticalTraversal(TreeNode root) {
        locate(root, 0, 0);
        // System.out.println(map);
        List<List<Integer>> ret = new ArrayList<>();
        for (int x : map.keySet()) {
            Map<Integer, TreeSet<Integer>> xmap = map.get(x);
            List<Integer> col = new ArrayList<>();
            for (int y : xmap.keySet()) {
                Set<Integer> s = xmap.get(y);
                for (Integer v : s) {
                    col.add(v);
                }
            }
            ret.add(col);
        }
        return ret;
    }
    
    private void locate(TreeNode root, int x, int y) {
        if (root == null)  return;
        TreeMap<Integer, TreeSet<Integer>> ymap = map.getOrDefault(x, new TreeMap<>());
        TreeSet<Integer> nset = ymap.getOrDefault(y, new TreeSet<>());
        nset.add(root.val);
        ymap.put(y, nset);
        map.put(x, ymap);
        
        locate(root.left, x-1, y+1);
        locate(root.right, x+1, y+1);
    }
}
