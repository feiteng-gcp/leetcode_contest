/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    TreeMap<Integer, PriorityQueue<Point>> map;
    public List<List<Integer>> verticalTraversal(TreeNode root) {
        map = new TreeMap<>();
        
        dfs(root, 0, 0);
        
        List<List<Integer>> res = new ArrayList<>();
        while (!map.isEmpty()) { 
            PriorityQueue<Point> p = map.pollFirstEntry().getValue();
            List<Integer> next = new ArrayList<>();
            
            while (!p.isEmpty()) {
                next.add(p.poll().val);
            }
            
            res.add(next);
        }
        
        return res;
    }
    
    private void dfs(TreeNode node, int x, int y) {
        if (node == null) return;
        
        map.computeIfAbsent(x, k -> new PriorityQueue<>()).offer(new Point(y, node.val));
        dfs(node.left, x - 1, y + 1);
        dfs(node.right, x + 1, y + 1);
    }

    private class Point implements Comparable<Point> {
        int y;
        int val;

        public Point(int y, int val) {
            this.y = y;
            this.val = val;
        }

        @Override
        public int compareTo(Point o) {
            int res = Integer.compare(y, o.y);
            if (res == 0) return Integer.compare(val, o.val);
            else return res;
        }
    }
}
