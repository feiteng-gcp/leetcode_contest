/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    public List<List<Integer>> verticalTraversal(TreeNode root) {
        Map<Integer,List<int[]>> map = new TreeMap<>();
        helper(map,root,0,0);
        List<List<Integer>> ret = new ArrayList<>();
        for(List<int[]> v : map.values()){
            v.sort((a,b)->{
               if(a[0]!=b[0]) return a[0]-b[0];
               if(a[1]!=b[1]) return b[1]-a[1];
               return a[2]-b[2];
            });
            List<Integer> list = new ArrayList<>();
            for(int[] item : v){
                list.add(item[2]);
            }
            ret.add(list);
        }
        return ret;
    }
    private void helper(Map<Integer,List<int[]>> map,TreeNode root,int x,int y){
        if(root==null) return;
        if(!map.containsKey(x)){
            map.put(x,new ArrayList<int[]>());
        }
        map.get(x).add(new int[]{x,y,root.val});
        helper(map,root.left,x-1,y-1);
        helper(map,root.right,x+1,y-1);
    }
}
