/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    class Node{
        int x;
        int y;
        int val;
        public Node(int a, int b, int c) {
            x = a;
            y = b;
            val = c;
        }
    }
    public List<List<Integer>> verticalTraversal(TreeNode root) {
        List<List<Integer>> res = new ArrayList<>();
        List<Node> list = new ArrayList<>();
        helper(root, 0, 0, list);
        Collections.sort(list, new Comparator<Node>() {
            public int compare(Node a, Node b) {
                if(a.x == b.x) {
                    if(a.y == b.y) {
                        return a.val - b.val;
                    }
                    else return b.y - a.y;
                }
                else return a.x - b.x;
            }
        });
        List<Integer> temp = new ArrayList<>();
        for(int i = 0; i < list.size(); i++) {
            if(i == 0) {
                temp.add(list.get(i).val);
                continue;
            }
            Node cur = list.get(i);
            Node pre = list.get(i-1);
            if(cur.x != pre.x) {
                res.add(temp);
                temp = new ArrayList<>();
            }
            temp.add(cur.val);
        }
        if(temp.size() > 0) res.add(temp);
        return res;
    }
    public void helper(TreeNode root, int x, int y, List<Node> list) {
        if(root == null) return;
        list.add(new Node(x, y, root.val));
        helper(root.left, x - 1, y - 1, list);
        helper(root.right, x + 1, y - 1, list);
    }
}
