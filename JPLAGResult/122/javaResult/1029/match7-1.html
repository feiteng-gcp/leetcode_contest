/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<List<Integer>> verticalTraversal(TreeNode root) {
        List<Node> list = new ArrayList<>();
        dfs(root, 0, 0, list);
        
        Collections.sort(list, new Comparator<Node>() {
            @Override
            public int compare(Node o1, Node o2) {
                if(o1.x < o2.x) {
                    return -1;
                } else if(o1.x > o2.x) {
                    return 1;
                } else {
                    if(o1.y < o2.y) {
                        return 1;
                    } else if(o1.y > o2.y) {
                        return -1;
                    } else {
                        return o1.val - o2.val;
                    }
                }
            }
        });
        
        List<List<Node>> res = new ArrayList<>();
        List<List<Integer>> r = new ArrayList<>();
        
        for(Node node : list) {
            if(res.size() == 0 || res.get(res.size() - 1).get(0).x < node.x) {
                List<Node> newList = new ArrayList<>();
                newList.add(node);
                res.add(newList);
            } else {
                res.get(res.size() - 1).add(node);
            }
        }
        
        for(List<Node> l : res) {
            
List<Integer> lt = new ArrayList<>();
            for(Node node : l) {
                lt.add(node.val);
            }
            
            r.add(lt);
        }
        
        return r;
        
        
    }
    
    private void dfs(TreeNode node, int x, int y, List<Node> list) {
        if(node == null) {
            return;
        }
        
        list.add(new Node(node.val, x, y));
        dfs(node.left, x - 1, y - 1, list);
        dfs(node.right, x + 1, y - 1, list);
    }
    
    class Node {
        int val;
        int x;
        int y;
        Node(int val, int x, int y) {
            this.val = val;
            this.x = x;
            this.y = y;
        }
    }
}
