class Solution {
    private class NodePos{
        int x, y, val;
        NodePos(int x, int y, int val){
            this.x = x;
            this.y = y;
            this.val = val;
        }
    }

    private Map<Integer, List<NodePos>> map;

    public List<List<Integer>> verticalTraversal(TreeNode root) {
        if(root == null){
            return  new ArrayList<>();
        }
        map = new HashMap<>();
        helper(root, 0, 0);
        List<Integer> keys = new ArrayList<>(map.keySet());
        keys.sort(Comparator.comparingInt(i->i));
        List<List<Integer>> res = new ArrayList<>();
        for(int k : keys){
            List<Integer> tmp =
            map.get(k).stream().sorted((n1,n2)->{
                if(n1.y != n2.y){
                    return n2.y - n1.y;
                }
                else{
                    return n1.val-n2.val;
                }
            }).map(n->n.val).collect(Collectors.toList());
            res.add(tmp);
        }
        System.out.println(res);
        return res;

    }

    private void helper(TreeNode node, int x, int y){
        if(node.left != null){
            helper(node.left, x-1, y-1);
        }
        if(node.right != null){
            helper(node.right, x+1, y-1);
        }
        map.computeIfAbsent(x, k-> new ArrayList<>()).add(new NodePos(x, y , node.val));

    }
}
