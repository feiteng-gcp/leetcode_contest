/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<List<Integer>> verticalTraversal(TreeNode root) {
        List<List<Integer>> ret = new ArrayList<>(); 
        TreeMap<Integer, List<int[]>> map = new TreeMap<>(); 
        if (root == null) {
            return ret;
        }
        Pair r = new Pair(root, 0);
        Queue<Pair> q = new LinkedList<>(); 
        q.add(r);
        int level = 0; 
        while (!q.isEmpty()) {
            int size = q.size(); 
            for (int i = 0; i < size; i++) {
                Pair p = q.poll(); 
                if (!map.containsKey(p.idx)) {
                    map.put(p.idx, new ArrayList<>());
                }
                map.get(p.idx).add(new int[]{level, p.node.val});
                if (p.node.left != null) {
                    q.add(new Pair(p.node.left, p.idx - 1));    
                }
                
                if (p.node.right != null) {
                    q.add(new Pair(p.node.right, p.idx + 1));
                }
            }
            level++; 
        }
        
        for (int key : map.keySet()) {
            List<int[]> l = map.get(key);
            Collections.sort(l, new Comparator<int[]>() {
                
                public int compare(int[] a, int[] b) {
                    if (a[0] == b[0]) {
                        return a[1] - b[1]; 
                    } else {
                        return a[0] - b[0]; 
                    }
                }
            });
            List<Integer> tmp = new ArrayList<>(); 
            for (int i = 0; i < l.size(); i++) {
                tmp.add(l.get(i)[1]);
            }
            ret.add(tmp);
        }
        return ret;
    }
    
    private class Pair {
        TreeNode node; 
        int idx; 
        public Pair(TreeNode node, int idx) {
            this.node = node;
            this.idx = idx; 
        }
        
    }
}
