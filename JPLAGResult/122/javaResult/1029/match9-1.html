/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    TreeMap<Integer, TreeSet<int[]>> map;
    public List<List<Integer>> verticalTraversal(TreeNode root) {
        map = new TreeMap<>();
        helper(root, 0, 0);
        List<List<Integer>> ret = new ArrayList<>();
        for(int key : map.keySet()) {
            List<Integer> tmp = new ArrayList<>();
            for(int[] val : map.get(key)) tmp.add(val[0])
;
            ret.add(tmp);
        }
        return ret;
    }
    
    private void helper(TreeNode root, int x, int y) {
        if(root == null) return;
        TreeSet<int[]> set = map.get(x);
        if(set == null) {
            set = new TreeSet<>((a, b) -> 
                (a[1] == b[1] ? a[0] - b[0] : b[1] - a[1])
            );
            map.put(x, set);
        }
        
set.add(new int[]{root.val, y});
        helper(root.left, x - 1, y-1);
        helper(root.right, x + 1, y-1);
    }
}
