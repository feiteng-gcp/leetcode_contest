class Solution {
public:
    
    int s[10005];
    
    int tran(int x) {
        if (x == 0 || x == 1 || x == 8) return x;
        if (x == 2 || x == 5) return 7-x;
        if (x == 6 || x == 9) return 15 - x;
        return -1;
    }
    
    int change(int x) {
        int len = 0;
        int t = x;
        while(t) {
            s[len++] = t % 10;
            t /= 10;
        }
        
        for (int i = 0; i < len; i++) {
            s[i] = tran(s[i]);
            if (s[i] == -1) return -1;
        }
        int ans = 0;
        for (int i = len - 1; i >= 0; i--) {
            ans = ans * 10 + s[i];
        }
        
        if (ans == x) return -1;

        return ans;
    }
    
    int rotatedDigits(int N) {
        int ans = 0;
        for (int i = 0; i <= N; i++) {
            if (change(i) != -1) ans++;
        }
        return ans;
    }
};
