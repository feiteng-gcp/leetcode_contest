#define SZ 10005
static int good[SZ] = {0};
class Solution {
    bool isgood(int n) {
        if (good[n] != 0) {
            return good[n] == 1;
        }
        char buf[10];
        sprintf(buf, "%d", n);
        string s;
        for (int i = 0; i < strlen(buf); i++) {
            char ch = buf[i];
            if (ch == '0' || ch == '1' || ch == '8')
                s += ch;
            else if (ch == '2')
                s += '5';
            else if (ch == '5')
                s += '2';
            else if (ch == '6')
                s += '9';
            else if (ch == '9')
                s += '6';
            else {
                good[n] = 2;
                return false;
            }
        }
        good[n] = s == string(buf) ? 2 : 1;
        return good[n] == 1;
    }
public:
    int rotatedDigits(int N) {
        int cnt = 0;
        for (int i = 1; i <=N; i++) {
            if (isgood(i))
                cnt++;
        }
        return cnt;
    }
};
