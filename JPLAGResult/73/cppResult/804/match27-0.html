class Solution {
public:
	bool isgood(int n) {
		for (int m = n; m > 0; m /= 10) {
			int d = m % 10;
			if (d == 3 || d == 4 || d == 7) {
				return false;
			}
		}

		bool changed = false;
		for (int m = n; m > 0; m /= 10) {
			int d = m % 10;
			if (d == 2 || d == 5 || d == 6 || d == 9) {
				changed = true;
				break;
			}
		}
		return changed;
	}

	int rotatedDigits(int N) {
		int tot = 0;
		for (int i = 1; i <= N; i++) {
			tot += isgood(i);
		}
		return tot;
	}
};
