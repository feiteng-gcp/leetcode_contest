class Solution {
public:
    int check(int x){
        int y,f=0;
        while(x){
            y = x%10;
            if(y==3 or y==4 or y==7) return 0;

            if(y==2 or y==5 or y==6 or y==9) f=1;
            x/=10;
        }
        return f;
    }
    
    int rotatedDigits(int N) {
        int ans = 0;
        for(int i=1;i<=N;i++){
            ans += check(i);
        }
        return ans;
    }
};
