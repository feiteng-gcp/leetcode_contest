class Solution {

public:
  int rotatedDigits(int N) {
    int ret = 0;
    for (int i = 1; i <= N; ++i) {
      if (isGood(i)) ++ret;
    }
    return ret;
  }
  
  bool isGood(int x) {
    int q = 0;
    while (x > 0) {
      int r = x % 10;
      if (r == 2 || r == 5 || r == 6 || r == 9) {
        ++q;
      } else if (r == 3 || r == 4 || r == 7) {
        return false;
      }
      x /= 10;
    }
    return q > 0;
  }
};
