class Solution {
public:
    int rotatedDigits(int N) {
        int count = 0;
        for (int i = 1; i <= N; i++) {
            int temp = i;
            int isdiff = false;
            int isok = true;
            while (temp) {
                int a = temp%10;
                if (a == 2 || a == 5 || a == 6 || a == 9) {
                    isdiff = true;
                } else if (a == 3 || a == 4 || a == 7) {
                    isok = false;
                    break;
                }
                temp /= 10;
            }
            if (isok && isdiff)
                count++;
        }
        return count;
    }
};
