class Solution {
public:
    bool depend(int n) {
        bool flag = false;
        while(n>0) {
            int c = n %10;
            n = n / 10;
            if(c==3||c==4||c==7) return false;

            if(c==2||c==5||c==6||c==9) flag = true;
        }
        return flag;
    }
    int rotatedDigits(int N) {
        int total = 0;
        for(int i = 1; i <= N; ++i)
            if(depend(i))++total;
        return total;
    }
};
