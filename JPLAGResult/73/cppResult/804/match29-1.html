class Solution {
public:
    int rotatedDigits(int N) {
        int ans = 0;
        for (int i = 1; i <= N; i++) {
            string s = to_string(i);
            if (s.find_first_of("347") != string::npos) continue;
            string t = s;
            for (char &c : s) {
                if (c == '2') c = '5';
                else if (c == '5') c = '2';
                else if (c == '6') c = '9';
                else if (c == '9') c = '6';
            }
            if (s != t) ans++;
        }
        return ans;
    }
};
