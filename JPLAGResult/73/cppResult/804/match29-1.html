class Solution {
public:
    int rotatedDigits(int N) {

        set<char> good = {'0','1','8','2','5','6','9'};
        int ans = 0;
        for(int i=1;i<=N;i++) {
            string s = to_string(i);
            bool all_good = 1;
            int c6925 = 0;
            for(char ch:s) {
                all_good &= good.find(ch) != good.end();
                if(ch == '6' || ch == '9' || ch == '2' || ch == '5') c6925++;
            }
            if(all_good && c6925) ans++;
        }
        return ans;
    }
};
