class Solution {
public:
    int rotatedDigits(int N) {
        int valid = 0;
        set<int> validSet = {0, 1, 8, 2, 5, 6, 9};
        set<int> selfSet = {0, 1, 8};

        for (int i = 1; i <= N ; ++i) {
            if (isValid(i, validSet, selfSet)) {
                valid++;
            }
        }
        return valid;
    }
    

    bool isValid(int x, const set<int>& validSet, const set<int>& selfSet) {
        bool hasNumberInNonSelfSet = false;
        while ( x > 0) {
            int digit = x % 10;
            if (validSet.find(digit) == validSet.end()) {
                // Not found
                return false;
            }
            if (selfSet.find(digit) == selfSet.end()) {
                // Not Found;
                hasNumberInNonSelfSet = true;
            }
            x = x / 10;
        }
        return hasNumberInNonSelfSet;
    }
};
