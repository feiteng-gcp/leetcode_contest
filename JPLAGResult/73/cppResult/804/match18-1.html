class Solution {
public:
    int rotatedDigits(int N) {
        int ans = 0;
        for(int i = 1; i <= N; i++){
            int now = i;
            bool flag = 0;
            while(now){
                if(now%10==2||now%10==5||now%10==6||now%10==9)flag = 1;
                if(now%10==3||now%10==4||now%10==7){
                    flag = 0;
                    break;
                }
                now /= 10;
            }
            ans += flag;
        }
        return ans;
    }
};
