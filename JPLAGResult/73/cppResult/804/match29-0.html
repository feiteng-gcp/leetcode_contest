class Solution {
public:
    int rotatedDigits(int N) {
        int res = 0;
        for (int i = 1; i <= N; ++i) {
            bool valid = true;
            bool diff = false;
            
            for (int x = i; x > 0; x /= 10) {
                int y = x % 10;
                if (y == 3 || y == 4 || y == 7) valid = false;
                else
                if (y == 2 || y == 5 || y == 6 || y == 9) diff = true;
            }
            
            res += (valid && diff);
        }
        
        return res;
    }
};
