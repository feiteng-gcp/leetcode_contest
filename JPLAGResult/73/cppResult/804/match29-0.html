        set<int> gn = {0, 1, 2, 5, 6, 8, 9}; 
        set<int> cgn = { 2, 5, 6, 9 };
        int ans = 0;
        for (int i=1; i<=N; ++i) {
            string s = to_string(i);
            bool ok = true;;
            int ccnt = 0;
            for (auto c : s) {
                if (gn.find(c-'0') == gn.end()) {
                    ok = false;
                    break;
                }   
                if (cgn.find(c-'0') != cgn.end()) {
                    ++ccnt;
                }   
            }   

            if (ok && ccnt>0) { ++ans; }
        }   
        return ans;
    }   
};
