class Solution {
    bool isvalid(int x) {
        int flag = 0;
        while (x > 0) {
            if (x%10 == 3 || x%10 == 4 || x%10 == 7) return false;
            if (x%10 != 0 && x%10 != 1 && x%10 != 8) flag = 1;
            x/=10;
        }
        if (!flag) return false;
        return true;
    }
public:
    int rotatedDigits(int N) {
        int ans = 0;
        for (int i = 1; i <= N; i++) {
            if(isvalid(i)) ++ans;
        }
        return ans;
    }
};
