class Solution {
    bool check(int n) {
        string s = to_string(n);
        int ret = 0;
        for (char c : s) {
            if (c == '0' || c == '1' || c == '8') ret = ret*10 + c - '0';
            else if (c == '2') ret = ret*10 + 5;
            else if (c == '5') ret = ret*10 + 2;
            else if (c == '9') ret = ret*10 + 6;
            else if (c == '6') ret = ret*10 + 9;
            else return false;
        }

        return ret!=n;
    }
public:
    int rotatedDigits(int N) {
        int cnt = 0;
        for (int i = 1; i <= N; i++) {
            if (check(i)) cnt++;
        }
        return cnt;
        
        
    }
};
