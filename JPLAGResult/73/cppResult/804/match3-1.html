class Solution {
public:
    int rotatedDigits(int N) {
        int n = 0;
        for (int i=1; i<=N; i++) n += isGood(i);
        return n;
    }
private:
    int isGood(int n) {

        bool hasDistinct = false;
        while (n>0) {
            int a = n%10;
            if ((a==3)||(a==4)||(a==7)) return 0;
            if ((!hasDistinct)&&((a==2)||(a==5)||(a==6)||(a==9))) hasDistinct = true;
            n/=10;
        }
        return hasDistinct;
    }
};
