class Solution {
public:
    int rotatedDigits(int N) {
        vector<int> m{2,2,1,0,0,1,1,0,2,1};
        int ret = 0;
        for (int i = 2; i <= N ; i++){
            int temp = i;
            bool flag = true;
            bool flag2 = false;
            while (temp > 0){
                if (m[temp % 10] == 0){
                    flag = false;
                    break;
                    
                }
                else if(m[temp%10] == 1){
                    flag2 = true;
                }
                temp /= 10;
                
            }
            if (flag && flag2){
                ret++;
            }
        }
        return ret;
        
    }
};
