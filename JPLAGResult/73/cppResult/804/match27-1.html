class Solution {
public:
    
    bool isgood(int n) {
        int nn = n;
        int rn = 0;
        int r[] = {0,1,5,-1,-1,2,9,-1,8,6};
        int d = 1;
        while (nn) {
            int c = nn % 10;
            
            if (r[c] < 0) return false;
            rn += r[c] * d;
            
            nn /= 10;
            d *= 10;

        }
        
        
        return rn != n;
    }
    
    int rotatedDigits(int N) {
        int count =0;
       for (int i = 1; i <= N; i++) {
           count += isgood(i);
       }
        
        return count;
        
    }
};
