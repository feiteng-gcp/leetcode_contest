class Solution {
public:
    int rotatedDigits(int N) {
        int cnt = 0;
        for(int i = 1 ; i <= N ; i++){
            int temp = i;
            bool flag1 = false;
            bool flag2 = false;
            while(temp!=0){
                int rem = temp%10;
                if(rem==2 || rem == 5 || rem == 6 || rem == 9){
                    flag1 = true;
                }
                else if(rem==3 || rem==4 || rem==7){
                    flag2 = true;
                }
                temp = temp/10;
            }
            if(flag1 && !flag2){
                cnt+=1;
            }
        }
        return cnt;
    }
};
