class Solution {
public:
    int rotatedDigits(int N) {
        set<int> grot;
        set<int> brot;
        grot.insert(2);grot.insert(5);
        grot.insert(6);grot.insert(9);
        brot.insert(0);brot.insert(1);brot.insert(8);
        int count = 0;
        for(int i =1; i <= N; i++){
            int g = i;
            bool brot_found = false;
            bool bad = false;
            while(g > 0){
                int d = g % 10;
                g /= 10;
                if(grot.find(d) == grot.end()
                  && brot.find(d) == brot.end()){
                    bad = true;
                    break;
                }else if(grot.find(d) != grot.end()){
                    brot_found = true;
                }
            }
            if(!bad && brot_found)count++;
        }
        return count;
    }
};
