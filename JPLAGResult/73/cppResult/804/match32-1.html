class Solution {
public:
    int rotatedDigits(int N) 
    {
        int s = 0;
        for (int i = 1; i <= N; ++i)
        {
            int r = rotate(i);
            if (r != -1 && r != i)
                s++;
        }
        return s;
    }

    int rotate(int n)
    {
        int ret = 0;
        int base = 1;
        while (n)
        {
            int d = rotateOne(n % 10);
            if (d == -1)
                return -1;
            ret += d * base;
            base *= 10;
            n /= 10;
        }
        return ret;
    }

    int rotateOne(int n)
    {
        if (n == 0 || n == 1 || n == 8)
            return n;
        if (n == 2) return 5;
        if (n == 5) return 2;
        if (n == 6) return 9;
        if (n == 9) return 6;
        return -1;
    }
};
