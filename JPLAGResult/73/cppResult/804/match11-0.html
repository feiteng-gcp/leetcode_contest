class Solution {
public:
    int rotatedDigits(int N) {
        int count = 0;
        for (int i = 1; i <= N; ++i)
        {
            int val = i;
            bool good = false;
            bool valid = true;
            while (val)
            {
                int digit = val % 10;
                switch (digit)
                {
                    case 0:
                    case 1:
                    case 8:
                        break;
                        
                    case 2:
                    case 5:
                    case 6:
                    case 9:
                        good = true;
                        break;
                    default:
                        valid = false;
                        break;
                }
                if (!valid)
                    break;
                
                val /= 10;
            }
            
            if (good && valid)
                count += 1;
        }
        
        return count;
    }
};
