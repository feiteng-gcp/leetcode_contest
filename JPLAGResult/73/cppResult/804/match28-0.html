class Solution {
public:
    bool isRotate(int x){
        bool flag = false;
        while (x>0){
            int a = x % 10;
            if (a == 2 || a == 5 || a == 6 || a == 9){
                x = x / 10;
                flag = true;
                continue;
            }
            if (a == 0 || a == 1 || a == 8){
                x = x / 10;
                continue;
            }
            return false;
        }
        return flag;
    }
    int rotatedDigits(int N) {
        int res = 0;
        for (int i = 1; i <= N; i++){
            if (isRotate(i))
                res++;
        }
        return res;
    }
};
