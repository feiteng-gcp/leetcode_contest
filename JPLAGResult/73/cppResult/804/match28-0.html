int a[10] = {0, 1, 5, -1, -1, 2, 9, -1, 8, 6};

class Solution {
public:
    int rotate(int n) {
        int ret = 0;
        int p = 1;
        for (; n; n /= 10) {
            if (a[n % 10] < 0) return -1;
            ret += p * a[n % 10];
            p *= 10;
        }
        return ret;
    }
    int rotatedDigits(int N) {
        int ret = 0;
        for (int i = 1; i <= N; ++i) {
            int j = rotate(i);
            if (j != -1 && j != i) ++ret;
        }
        return ret;
        
    }
};
