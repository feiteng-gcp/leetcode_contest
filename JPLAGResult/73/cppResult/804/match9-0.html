class Solution {
public:
    bool good(int x) {
        string s = to_string(x);
        bool diff = false;
        for (char ch : s) {
            if (ch == '3' || ch == '4' || ch == '7')
                return false;
            if (ch == '2' || ch == '5' || ch == '6' || ch == '9')
                diff = true;
        }
        return diff;
    }
    int rotatedDigits(int n) {
        int res = 0;
        for (int i = 1; i <= n; i++)
            if (good(i))
                res++;
        return res;
    }
};
