class Solution {
public:
    int rotatedDigits(int N) {
        int res = 0;
        for(int i=1;i<=N;i++){
            int cur = i;
            bool valid = false;
            bool hasbad = false;
            while(cur){
                int lastbit = cur%10;
                if(lastbit==0||lastbit==1||lastbit==8){
                }
                else if(lastbit==2||lastbit==5||lastbit==6||lastbit==9){
                    valid = true;
                }
                else{
                    hasbad = true;
                    break;
                }
                cur = cur/10;
            }
            if(!hasbad&&valid){
                res++;
            }
        }
        return res;
    }
};
