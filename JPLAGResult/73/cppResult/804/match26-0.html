class Solution {
public:
    int rotatedDigits(int N) {
        int ret = 0;
        for (int i = 1; i <= N; i++) {
            if (isValidRotate(i)) {
                ret++;
            }
        }
        return ret;
    }
private:
    bool isValidRotate(int x) {
        string s = to_string(x);
        bool flag = false;
        for (auto c : s) {
            if (c == '3' || c == '4' || c == '7') {
                return false;
            }
            if (c != '0' && c != '1' && c != '8') {
                flag = true;
            }
        }
        return flag;
    }
};
