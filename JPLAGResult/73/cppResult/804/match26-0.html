class Solution {
public:
    int rotatedDigits(int N) {
        int res = 0;
        for(int i = 0; i <= N; i ++){
            if(valid(i))
                res ++;
        }
        return res;
    }
    
    bool valid(int n){
        bool other = false;
        while(n > 0){
            int digit = n % 10;
            n = n / 10;
            if(digit == 3 || digit == 4 || digit == 7)
                return false;
            
            if(digit == 2 || digit == 5 || digit == 6 || digit == 9)
                other = true;
        }
        return other;
    }
};
