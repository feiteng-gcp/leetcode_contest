class Solution {
public:
    int rotatedDigits(int N) {
        int ans = 0;
        for (int i = 1; i <= N; ++i) {
            if (c(i)) ans++;
        }
        return ans;
    }
    
    bool c(int n) {
        string s = to_string(n);
        
        int l = s.size();
        int ans = 0;
        for (int i = 0; i < l; ++i) {
            if (s[i] == '0' || s[i] == '1' || s[i] == '8') {
                ans = ans * 10 + (s[i] - '0');
            } else if (s[i] == '6') ans = ans * 10 + 9;
            else if (s[i] == '9') ans = ans * 10 + 6;
            else if (s[i] == '5') ans = ans * 10 + 2;
            else if (s[i] == '2') ans = ans * 10 + 5;
            else return false;
        
        }
        
        return ans != n;
    }
};
