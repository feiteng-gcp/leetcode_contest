class Solution {
    bool good(int i) {
        int n = 0;
        while (i > 0) {
            int j = i % 10;
            if (j == 3 || j == 4 || j == 7)
                return false;
            if (j == 2 || j == 5 || j == 6 || j == 9)
                n++;
            i /= 10;
        }
        return n > 0;
    }
public:

    int rotatedDigits(int N) {
        int n = 0;
        for (int i = 0; i <= N; i++)
            if (good(i))
                n++;
        return n;
    }
};
