class Solution {
public:

    int rotatedDigits(int N) {
        int res = 0;
        for (int i = 1; i <= N; ++i) {
            if (isValid(i)) ++res;
        }
        return res;
    }
    
    bool isValid(int N) {
        bool any = false;
        while (N > 0) {
            int tmp = N % 10;
            if (tmp == 0 || tmp == 1 || tmp == 8);
            else if (tmp == 2 || tmp == 5 || tmp == 6 || tmp == 9) any = true;
            else return false;
            
            N = N / 10;
        }
        return any;
    }
};
