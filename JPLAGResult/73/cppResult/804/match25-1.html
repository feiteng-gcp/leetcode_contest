class Solution {
public:
    int rotatedDigits(int N) {
        int result = 0;
        for(int i=1;i<=N;++i){
            if(rotate(i)){
                 result++;
                //cout << i << endl;
            }
              
        }
        return result;
    }
    
    bool rotate(int num){
        int n = num;
        int r = 0;
        int i = 1;
        while(n>0){
            int a = n % 10;
            n /= 10;
            switch(a){
                case 0:
                case 1:
                case 8:
                    r = r + a*i;
                    break;
                case 2:
                    r = r+5*i;
                    break;
                case 5:
                    r = r+2*i;
                    break;
                case 6:
                    r = r+9*i;
                    break;
                case 9:
                    r=r+6*i;
                    break;
                default:
                    return false;
            }
            i *= 10;
        }
        //cout <<"r:" << r << endl;
        return r!=num;
    }
};
