class Solution {
public:

unordered_set<int> mySet = {0, 1, 8, 2, 5, 6, 9};
unordered_set<int> my2569Set = {2, 5, 6, 9};
    
bool isGood(int x) {
    bool has2569 = false;
    while (x) {
        int digit = x % 10;
        if (mySet.count(digit) == 0) {
            return false;
        }
        
        if (my2569Set.count(digit) > 0) {
            has2569 = true;
        }
        
        x /= 10;
    }
    return has2569;
}
    
int rotatedDigits(int N) {
    int cnt = 0;
    for (int i = 1; i <= N; i++) {
        if (isGood(i)) {
            cnt++;
        }
    }
    return cnt;
}
};
