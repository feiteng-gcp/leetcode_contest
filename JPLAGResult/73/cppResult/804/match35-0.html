class Solution {
public:
    int rotatedDigits(int N) {
        int count = 0;
        for(int i = 1; i <= N; i++)
        {
            int j = i;
            int unvalid = 0;            
            int rotate = 0;
            while(j > 0)
            {
                if(j % 10 == 3 || j % 10 == 4 || j % 10 == 7)
                {
                    unvalid++;
                    break;
                }
                if(j % 10 == 2 || j % 10 == 5 || j % 10 == 6 || j % 10 == 9) rotate++;
                j /= 10;
            }
            if(unvalid == 0 && rotate > 0) count++;
        }
        return count;
    }
};
