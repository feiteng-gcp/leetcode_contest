class Solution {

public:

int check(int n)
{
    int ret=0;
    int f[]={0,0,1,-1,-1,1,1,-1,0,1};
    while(n)
    {
        if(f[n%10]==-1)return 0;

        ret+=f[n%10];
        n/=10;
    }
    return ret>0;
}
    int rotatedDigits(int N) {
        int i,ret=0;
        for(int i=1;i<=N;i++)if(check(i))ret++;
        return ret;
    }
};
