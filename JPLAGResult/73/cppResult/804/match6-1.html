class Solution {
public:

    int rotatedDigits(int N) {
        int num = 0;
        for (int i=1; i<=N; i++) {
            if (isGood(i)) {
                num ++;
                // cout << i << endl;
            }
        }
        return num;
    }
private:
    bool isGood(int n) {
        int count = 0;
        while (n > 0) {
            int num = n % 10;
            if (num == 3 || num == 4 || num == 7)
                return false;
            if (num == 2 || num == 5 || num == 6 || num == 9)
                count ++;
            n /= 10;
        }
        return (count > 0);
    }
};
