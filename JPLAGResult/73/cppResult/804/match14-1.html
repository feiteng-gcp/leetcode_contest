class Solution {
public:
    int rotatedDigits(int N) {
        int count = 0;

        for (int i = 0; i <= N; i++) {
            int flag = 0;

            int n = i;

            while (n > 0) {
                int d = n % 10;
                if (d == 2 || d == 5 || d == 6 || d == 9) {
                    flag = 1;
                }
                else if (d == 0 || d == 1 || d == 8) {
                    n /= 10;
                    continue;
                }
                else {
                    flag = 0;
                    break;
                }
                n /= 10;
            }

            if (flag) count++;
        }

        return count;
    }
};
