class Solution {
public:
    int rotatedDigits(int N) {
        int res = 0;
        for(int i = 1; i <= N; ++i){
            int k = i;
            int boo = 0;
            while(k){
                int d = k%10;
                if(d == 3 || d == 4 || d == 7){
                    boo = 0;
                    break;
                }
                if(d == 2 || d == 5 || d == 6 || d == 9){
                    boo = 1;
                }
                k /= 10;
            }
            res += boo;
        }
        return res;
    }
};
