class Solution {
public:
    
    bool rotate(int x) {
        bool met = false;
        
        while (x>0) {
            int y = x%10;
            x /= 10;
            if (y==3 || y==4 || y==7)
                return false;
            else if (y==2 || y==5 || y==6 || y==9)
                met = true;
        }
        return met;
    }
    
    int rotatedDigits(int N) {
        
        int ans = 0;
        for (int i=1;i<=N;i++) {
            if (rotate(i))
                ans++;
        }
        return ans;
    }
};
