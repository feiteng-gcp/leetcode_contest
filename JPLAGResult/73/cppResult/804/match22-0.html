class Solution {
public:
    int rotatedDigits(int N) {
        int after[10] ={0, 1, 5, -1, -1, 2, 9, -1, 8, 6};
        int count = 0;
        for (int i = 1; i <= N; i++) {
            string a = to_string(i);
            string b = "";
            bool notOk = false;
            for (int j = 0; j < a.size(); j++) {
                if (after[a[j] - '0'] == -1) {
                    notOk = true;
                    break;
                } else {
                    b += after[a[j] - '0'] + '0';
                }
            }
            if (!notOk && b.compare(a) != 0) {
                count++;
            }
        }
        return count;
    }
};
