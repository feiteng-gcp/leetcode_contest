class Solution {
public:
    int notValid[5] = {3,4,7};
    int need[5] = {2,5,6,9};
    bool check(int a){
        int tmp = a;
        bool flag = false;
        while(tmp > 0){
            int k = tmp % 10;
            for(int i=0;i<3;i++){
                if(k == notValid[i]){
                    return false;
                }
            }
            for(int i=0;i<4;i++){
                if(k == need[i]){
                    flag = true;
                }
            }

            tmp = tmp / 10;
        }
        
        return flag;
    }
    int rotatedDigits(int N) {
        int ans = 0;
        for(int i=1;i<=N;i++){
            if(check(i)){
                ans++;
            }
        }
        return ans;
    }
};
