class Solution {
public:
    bool escapeGhosts(vector<vector<int>>& ghosts, vector<int>& target) {
       int dis = abs(target[0]) + abs(target[1]);
        for (auto p : ghosts) {
            int x = p[0], y = p[1];
            int t = (abs(x - target[0]) + abs(y - target[1]));
            if (t <= dis) return false;
        }
        return true;
    }
};
