class Solution {
public:
	bool escapeGhosts(vector<vector<int>>& ghosts, vector<int>& target) {
		int i, j, g, n = ghosts.size(), t;

		g = abs(target[0]) + abs(target[1]);
		for (i = 0; i < n; i++) {
			t = abs(target[0] - ghosts[i][0] ) + abs(target[1] - ghosts[i][1]);
			if (t <= g) return false;
		}
		return true;
	}
};
