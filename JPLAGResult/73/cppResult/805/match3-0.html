class Solution {
public:
    bool escapeGhosts(vector<vector<int> >& ghosts, vector<int>& target) {
      int i,j,dis,dis1;
      dis=abs(target[0])+abs(target[1]);
      for (i=0;i<ghosts.size();i++)
      {
        dis1=abs(target[0]-ghosts[i][0])+abs(target[1]-ghosts[i][1]);
        if (dis1<=dis) return false;
      }
      return true;
    }
};
