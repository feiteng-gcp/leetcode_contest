class Solution {
public:
    bool escapeGhosts(vector<vector<int>>& ghosts, vector<int>& target) {
        int mins = 1000000000;
        for (auto g : ghosts) {
            int s = abs(target[0] - g[0]) + abs(target[1] - g[1]);
            if (s < mins ) mins = s;
        }
        int k = abs(target[0]) + abs(target[1]);
        if (k < mins) return true;
        return false;
    }
};
