#incluce <math>
#include <algorithm>

class Solution {
public:
    bool escapeGhosts(vector<vector<int>>& ghosts, vector<int>& target) {
        int dist=abs(target[0])+abs(target[1]);
        for(auto g:ghosts)
        {
            int d=abs(target[0]-g[0])+abs(target[1]-g[1]);
            if(d<=dist) return 0;
        }
        return 1;
    }
};
