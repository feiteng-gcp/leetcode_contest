class Solution {
public:
    bool escapeGhosts(vector<vector<int>>& ghosts, vector<int>& target) {
        int res = abs(target[0]) + abs(target[1]);
        for (auto other : ghosts) {
            if (abs(other[0] - target[0]) + abs(other[1] - target[1]) <= res) {
                return false;
            }
        }
        return true;
    }
};
