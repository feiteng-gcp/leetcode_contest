class Solution {
public:
    string customSortString(string S, string T) {
        int cmp[256]={0};
        for (int i=0;i<S.length();i++)
        {
            cmp[S[i]]=i;
        }
        int n=T.length();

        for (int i=0;i<n;i++)
        {
            for (int j=0;j<n-i-1;j++)
            {
                if (cmp[T[j]]>cmp[T[j+1]])
                {
                    swap(T[j],T[j+1]);
                }
            }
        }
        return T;
    }
};
