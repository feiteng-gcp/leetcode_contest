int val[30],a[210];

bool cmp(int x,int y){return val[x]<val[y];}
class Solution {
public:
    string customSortString(string S, string T) {
        for(int i=0;i<S.size();i++)val[S[i]-'a']=i+1;
        for(int i=0;i<T.size();i++)a[i]=T[i]-'a';
        int n=T.size();
        sort(a,a+n,cmp);
        string Z;
        for(int i=0;i<n;i++)Z+='a'+a[i];
        return Z;
    }
};
