class Solution {
    typedef long long LL;
public:
    int numTilings(int N) {
        vector<LL> dp(N+1, 0);
        int mod = 1e9 + 7;
        dp[0] = 1;
        for (int i = 1; i <= N; i++) {
            dp[i] = dp[i-1];
            if (i-2 >= 0) {
                dp[i] = (dp[i] + dp[i-2]) % mod;
            }
            if (i-3 >= 0) {
                dp[i] = (dp[i] + 2*dp[i-3]) % mod;
            }
            for (int j = i-4; j >= 0; j --) {
                dp[i] = (dp[i] + 2*dp[j]) % mod;
            }
        }
        return dp[N];
        
    }
};
