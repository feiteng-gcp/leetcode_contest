class Solution {
public:
    int numTilings(int N) {
        const int q = 1000000007;
        long long a, b, c, d;
        long long e, f, g, h;
        b = c = d = e = f = g = h = 0;
        a = 1;
        for (int i = 1; i <= N; i++) {
            e = (a + d) % q;
            f = (a + c) % q;
            g = (a + b) % q;
            h = (a + b + c) % q;
            a = e, b = f, c = g, d = h;
            // cout << a << " " << b << " " << c << " " << d << endl;
        }
        return a;
    }
};
