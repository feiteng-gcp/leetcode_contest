class Solution {
public:
    int numTilings(int N) {
        vector<long long> dp(4, 0);
        dp[0] = 1;
        for (int i = 1; i <= N; ++i) {
            vector<long long> ndp(4, 0);
            ndp[0] += dp[0];
            ndp[0] += dp[3];
            ndp[0] += dp[1] + dp[2];
            ndp[1] += dp[2] + dp[3];
            ndp[2] += dp[1] + dp[3];
            ndp[3] += dp[0];
            for (int j = 0; j < 4; ++j) {
                ndp[j] %= 1000000007;
            }
            dp = ndp;
        }
        return dp[0];
    }
};
