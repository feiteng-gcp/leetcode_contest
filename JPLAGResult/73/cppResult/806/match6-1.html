class Solution {
    int modulo = 1000000007;

public:
    int numTilings(int N) {
        int F[N + 1][3];
        fill(F[0], F[2], 0);
        F[0][0] = F[1][0] = 1;
        for (int i = 2; i <= N; ++i) {
            F[i][0] = ((long long)F[i - 1][0] + F[i - 2][0] + F[i - 1][1] + F[i - 1][2]) % modulo;
            F[i][1] = (F[i - 1][2] + F[i - 2][0]) % modulo;
            F[i][2] = (F[i - 1][1] + F[i - 2][0]) % modulo;
        }
        
        return F[N][0];
    }
};
