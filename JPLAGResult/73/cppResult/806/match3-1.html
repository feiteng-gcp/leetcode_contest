class Solution {
public:
    const int MOD=1000000007;
    long long f[1100][4];
    int numTilings(int N) {
        int i;

        f[0][3]=1;
        for(int i=1;i<=N;i++)
        {
            f[i][0]=f[i-1][3]%MOD;
            f[i][1]=(f[i-1][0]+f[i-1][2])%MOD;
            f[i][2]=(f[i-1][0]+f[i-1][1])%MOD;
            f[i][3]=(f[i-1][0]+f[i-1][1]+f[i-1][2]+f[i-1][3])%MOD;
        }
        return (int)f[N][3];
    }
};
