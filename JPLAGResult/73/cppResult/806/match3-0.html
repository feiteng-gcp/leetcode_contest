const int MOD = 1e9 + 7;
const int N = 1e3 + 10;
int dp[N][4];

class Solution {
public:
    int numTilings(int n) {
        memset(dp, 0, sizeof(dp));
                dp[n - 1][0] = dp[n - 1][3] = 1;

        for (int i = n - 2; i >= 0; --i) {
            dp[i][0] = ((dp[i + 1][0] + dp[i + 1][3]) % MOD + (dp[i + 1][1] + dp[i + 1][2]) % MOD) % MOD;
            dp[i][1] = (dp[i + 1][3] + dp[i + 1][2]) % MOD;
            dp[i][2] = (dp[i + 1][3] + dp[i + 1][1]) % MOD;
            dp[i][3] = dp[i + 1][0];
        }
        return dp[0][0];
        
    }
};
