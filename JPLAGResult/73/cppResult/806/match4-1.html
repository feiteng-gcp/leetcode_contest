const int MOD = 1e9 + 7;

class Solution {
public:
    int numTilings(int N) {
        int dp[10005][4];

        dp[0][3] = 1;
        dp[0][1] = dp[0][2] = dp[0][0] = 0;
        rep(i, 1, N + 1) rep(j, 0, 4) dp[i][j] = 0;
        for (int i = 1; i <= N; i++) {
            dp[i][0] = dp[i - 1][3];
            dp[i][1] = (dp[i - 1][2] + dp[i - 1][0]) % MOD;
            dp[i][2] = (dp[i - 1][1] + dp[i - 1][0]) % MOD;
            dp[i][3] = (dp[i - 1][3] + dp[i - 1][0]) % MOD;
            dp[i][3] += (dp[i - 1][1] + dp[i - 1][2]) % MOD;
            dp[i][3] %= MOD;
        }
        return dp[N][3];
    }
};
