class Solution {
public:
  int numTilings(int N) {
    const long M = 1000000007;
    vector<vector<long>> dp(3, vector<long>(N + 1)); 
    // dp[3][0]: XX XX XX OO
    // dp[3][1]: XX XX XX XO
    // dp[3][2]: XX XX XX OX
    dp[0][0] = 1; dp[1][0] = 0; dp[2][0] = 0; dp[0][1] = 1; dp[1][1] = 1; dp[2][1] = 1;
    for (int i = 2; i <= N; ++i) {
      dp[0][i] = dp[0][i - 1] + dp[0][i - 2] + dp[1][i - 2] + dp[2][i - 2];
      dp[1][i] = dp[0][i - 1] + dp[2][i - 1];
      dp[2][i] = dp[0][i - 1] + dp[1][i - 1];
      dp[0][i] %= M;
      dp[1][i] %= M;
      dp[2][i] %= M;
    }
    return static_cast<int>(dp[0][N]);
  }
};
