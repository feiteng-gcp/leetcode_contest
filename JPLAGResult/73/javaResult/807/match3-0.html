class Solution {
    int[] a=new int[100];
    public String customSortString(String S, String T) {
        for (int i=0;i<S.length();i++) a[S.charAt(i)-97]=i+1;
        int l=T.length();
        Integer[] b=new Integer[l];
        for (int i=0;i<l;i++) b[i]=new Integer(T.charAt(i)-97);
        Comparator cmp = new MyComparator();
        Arrays.sort(b,cmp);
        String ans="";
        for (int i=0;i<l;i++) ans+=(char)(b[i]+97);
        return ans;
    }
    class MyComparator implements Comparator<Integer>{
    @Override
    public int compare(Integer o1, Integer o2) {
         if(a[o1] > a[o2]) { 
             return 1;
         }else if(a[o1] < a[o2]) {
             return -1;
         }else {
             return 0;
         }
     }
     
 }
}
