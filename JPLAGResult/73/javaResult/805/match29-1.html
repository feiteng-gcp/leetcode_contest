class Solution {
    public boolean escapeGhosts(int[][] ghosts, int[] target) {
        int me = Math.abs(target[0]) + Math.abs(target[1]);
        int g = 200_000;
        for (int[] i : ghosts) g = Math.min(g, Math.abs(i[0] - target[0]) + Math.abs(i[1] - target[1]));
        return me < g;
    }
}
