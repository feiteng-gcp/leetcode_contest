class Solution {
    public boolean escapeGhosts(int[][] ghosts, int[] target) {
        for (int[] ghost : ghosts) {
            if(Math.abs(target[0] - ghost[0]) + Math.abs(target[1] - ghost[1]) <=
                    Math.abs(target[0]) + Math.abs(target[1])) {
                return false;
            }
        }
        return true;
    }
}
