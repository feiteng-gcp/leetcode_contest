class Solution {
    public boolean escapeGhosts(int[][] ghosts, int[] target) {
        int dis = Math.abs(target[0]) + Math.abs(target[1]);
        int min = Integer.MAX_VALUE;
        for(int[] g : ghosts){
            int cur = Math.abs(g[0] - target[0]) + Math.abs(g[1] - target[1]);
            min = Math.min(min, cur);
            if(min <= dis)
                return false;
        }
        return true;
    }
}
