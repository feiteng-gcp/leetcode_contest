class Solution {
    public boolean escapeGhosts(int[][] ghosts, int[] target) {
        int myTime = Math.abs(target[0]) + Math.abs(target[1]);
        int minGhostTime = Integer.MAX_VALUE;
        for (int[] ghost : ghosts) {
            minGhostTime = Math.min(minGhostTime, Math.abs(ghost[0] - target[0]) + Math.abs(ghost[1] - target[1]));
        }
        return myTime < minGhostTime;
    }
}
