class Solution {
    public boolean escapeGhosts(int[][] ghosts, int[] target) {
      int mysteps = numSteps (new int[] {0, 0}, target);
      for (int i = 0; i < ghosts.length; i++) {
        int gsteps = numSteps(ghosts[i], target);
        if (gsteps <= mysteps) {
          return false;  
        }
      }
      return true;  
    }
    
    private int numSteps (int[] pos, int[] target) {
      int steps = 0;
      steps = Math.abs(target[0] - pos[0]) + Math.abs (target[1] - pos[1]);
      return steps;  
    }
}
