class Solution {
    public int numTilings(int n) {
        long MOD = 1000000007;
        long[][] dp = new long[n + 1][4];
        dp[0][3] = 1;
        for (int i = 1; i <= n; i++) {
            dp[i][0] = dp[i - 1][3];
            
dp[i][1] = (dp[i - 1][0] + dp[i - 1][2]) % MOD;
            dp[i][2] = (dp[i - 1][0] + dp[i - 1][1]) % MOD;
            dp[i][3] = (dp[i - 1][3] + dp[i - 1][1] + dp[i - 1][2] + dp[i - 1][0]) % MOD;
        }
        return (int)dp[n][3];
    }
}
