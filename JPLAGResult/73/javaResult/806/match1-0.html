class Solution {
    public int numTilings(int N) {
        long MOD = (long)(1e9 + 7);
        long[] dp = new long[N+1];
        dp[0] = 1;
        for (int i = 1; i <= N; ++ i) {
            dp[i] = dp[i-1];
            if(i >= 2) {
                dp[i] += dp[i-2];
            }
            for(int j = i-3; j >= 0; j -= 1) {
                dp[i] += dp[j] * 2;
            }
            dp[i] %= MOD;
        }
        return (int)dp[N];
    }
}
