class Solution {
    public int numTilings(int N) {
        if (N < 3) return N;
        long[] tilings = new long[N+1];
        tilings[0] = 1;
        tilings[1] = 1;
        tilings[2] = 2;
        for (int i= 3; i <= N; i++) {
            tilings[i] = (tilings[i-3] + 2*tilings[i-1]) % 1000000007;
        }
        return (int)tilings[N];
    }
}
