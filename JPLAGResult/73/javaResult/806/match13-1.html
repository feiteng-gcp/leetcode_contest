class Solution {
    long mod = 1000000007L;
    
public int numTilings(int N) {
        long[][] dp = new long[1010][2];
        dp[0][0] = 1L;
        dp[1][0] = 1L;
        dp[2][0] = 2L;
        dp[2][1] = 2L;
        for (int i = 3; i <= N; i++) {
            dp[i][0] = (dp[i-1][1] + dp[i - 1][0] + dp[i - 2][0]) % mod;
            dp[i][1] = (dp[i-1][1] + dp[i - 2][0] * 2) % mod;
        }
        return (int)dp[N][0];
    }
}
