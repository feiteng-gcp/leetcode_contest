class Solution {
    public int numTilings(int N) {
        int[][] dp = new int[N + 1][3];
        dp[0][0] = 0;
        dp[0][1] = 0;
        dp[0][2] = 1;
        dp[1][0] = 1;
        dp[1][1] = 1;
        dp[1][2] = 2;
        int md = 1000000007;
        for (int i = 2; i < N; i++){
            dp[i][0] = (dp[i - 1][1] + dp[i - 2][2]) % md;
            dp[i][1] = (dp[i - 1][0] + dp[i - 2][2]) % md;
            dp[i][2] = ((dp[i - 1][2] + dp[i - 2][2]) % md + (dp[i - 1][0] + dp[i - 1][1]) % md ) % md;
        }
        return dp[N - 1][2];
    }
}
