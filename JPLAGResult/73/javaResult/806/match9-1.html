class Solution {
    int mo=1000000007;
    public int numTilings(int N) {
        long[][] f=new long[N+1][4];
        f[0][3]=1;
        for (int i=1;i<=N;i++)
        {
            f[i][0]=f[i-1][3];
            
f[i][1]=(f[i-1][0]+f[i-1][2])%mo;
            f[i][2]=(f[i-1][0]+f[i-1][1])%mo;
            f[i][3]=(f[i-1][3]+f[i-1][0]+f[i-1][1]+f[i-1][2])%mo;
        }
        return (int)f[N][3];
    }
}
