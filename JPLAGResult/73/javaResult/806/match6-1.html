import java.util.*;
import java.math.*;
import java.util.stream.*;


public class Solution {


    public int numTilings(int N) {
        int mod = 1000000007;
        long[][] f = new long[N + 1][2];

        f[0][0] = 1;
        for (int i = 1; i <= N; i++) {
            f[i][0] = f[i - 1][0];
            f[i][0] += f[i - 1][1] * 2;
            f[i][1] += f[i - 1][1];

            if (i - 2 >= 0) {
                f[i][0] += f[i - 2][0];
                f[i][1] += f[i - 2][0];
            }

            f[i][0] %= mod;
            f[i][1] %= mod;
        }
        return (int) (f[N][0]);
    }

    public static void main(String[] args) {
        Solution sol = new Solution();
        System.out.println(sol.numTilings(3));
    }
}
