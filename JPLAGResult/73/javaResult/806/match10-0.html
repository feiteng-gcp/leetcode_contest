class Solution {
    public int numTilings(int N) {
        if (N<=1) return 1;
        
        long[] a = new long[N+1], b = new long[N+1], c = new long[N+1];
        //a[i] = ways to fill 2*i, b[i] = ways to fill 2*i + single, c[i] = ways to fill 2*i + single + single
        long modulo = 1000000007L;
        a[0] = 1L;
        b[0] = 0L;
        c[0] = 1L;
        a[1] = 1L;
        b[1] = 1L;
        c[1] = 1L;
        for(int i=2;i<=N;i++) {
            a[i] = a[i-1] + c[i-2] + b[i-2] * 2;
            b[i] = b[i-1] + a[i-1];
            c[i] = a[i];
            a[i] %= modulo;
            b[i] %= modulo;
            c[i] %= modulo;
        }
        return (int)(a[N]);
    }
}
