class Solution {
    public int rotatedDigits(int N) {
        int cnt = 0;
        for (int i = 0; i <= N; i++){
            if (valid(
Integer.toString(i))){
                cnt++;
            }
        }
        return cnt;
    }
    
    boolean valid(String str){
        boolean found = false;
        for (char c : str.toCharArray()){
            if (c == '3' || c == '4' || c =='7'){
                return false;
            }
            if (c == '2' || c == '5' || c == '6' || c == '9'){
                found = true;
            }
        }
        return found;
    }
}
