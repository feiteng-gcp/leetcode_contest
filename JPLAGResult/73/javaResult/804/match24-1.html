class Solution {
    public int rotatedDigits(int N) {
        int count =0;        
        loop:
        for(int i=1;i<=N;i++)
        {
            boolean hasD=false;
            for(int q : digits(i) )
            {
                if(!isGood(q))
                    continue loop;
                hasD = hasD ||isGoodD(q);
            
}
            if(hasD)
                count++;
        }
        return count;
    }
    public List<Integer> digits(int n)
    {
        List<Integer> r= new ArrayList<Integer>();
            while(n>=10)
            {
                r.add(n%10);
                n/=10;
            }
        r.add(n);
        return r;
    }
    public boolean isGood(int d)
    {
        return (d!=3)&&(d!=4)&&(d!=7);
    }
    public boolean isGoodD(int d)
    {
        return isGood(d)&&(d!=0)&&(d!=1) &&(d!=8);
    }
}
