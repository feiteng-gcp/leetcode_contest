class Solution {
    
    //Question : Why does 2 rotates to 5?
    
    public int rotatedDigits(int N) {
        int ret = 0;
        for(int i=1;i<=N;i++) {
            if(isGoodNumber(Integer.toString(i))) {
                ret += 1;
            }
        }
        return ret;
    }
    
    private boolean isGoodNumber(String s) {
        return isGoodNumber(s.toCharArray());
    }
    
    private boolean isGoodNumber(char[] c) {
        boolean ret = false;
        for(int i=0;i<c.length;i++) {
            int t = getCategory(c[i]);
            if(t==0) continue;
            else if(t==1) {
                ret = true;
            } else {
                return false;
            }
        }
        return ret;
    }
    
    private int getCategory(char c) {
        if(c=='0' || c=='1' || c=='8') return 0;
        else if(c=='2' || c=='5' || c=='6' || c=='9') return 1;
        else return -1;
    }
}
