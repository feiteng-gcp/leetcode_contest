class Solution {
    public int rotatedDigits(int N) {
        Map<Integer,Integer> goodNumbers = new HashMap<>();
        goodNumbers.put(0,0);        
        goodNumbers.put(1,1);
        goodNumbers.put(2,5);
        goodNumbers.put(5,2);
        goodNumbers.put(6,9);
        goodNumbers.put(8,8);        
        goodNumbers.put(9,6);
        int count = 0 ;
        for(int i=1;i<=N;i++){
            int num = i;
            int rotNum = 0;
            int fact = 1;
            while(num>0){
                int rem = num%10;
                if(!goodNumbers.containsKey(rem)){
                    break;
                }
                int val = goodNumbers.get(rem);
                rotNum += fact*(val);
                fact =fact*10;
                num = num/10;
            }
            if(num==0 && i!=rotNum) count++;
        }
        return count;    
    }
}
