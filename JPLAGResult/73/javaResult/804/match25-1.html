class Solution {
    public int rotatedDigits(int N) {
        int ans = 0;
        for (int i = 0; i <= N; i++) {
            char[] str = String.valueOf(i).toCharArray();
            String newNum = "";
            boolean isOkay = true;
            for (int j =0;j<str.length;j++){
                if (str[j] == '2') {
                    str[j] = '5';
                } else if (str[j] == '5') {
                    str[j] = '2';
                } else if(str[j] == '6') {
                    str[j] = '9';
                } else if(str[j] == '9') {
                    str[j] = '6';
                }
                if (str[j] == '3' || str[j] == '7' || str[j] == '4') {
                    isOkay = false;
                    break;
                }
                newNum += str[j];
            }
            if(!isOkay) continue;
            if(
Integer.valueOf(newNum) == i) continue;
            // System.out.println(newNum + ", " + i);
            ans++;

        }
        return ans;
    }
}
