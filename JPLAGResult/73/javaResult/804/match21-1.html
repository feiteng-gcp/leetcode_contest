class Solution {
    public int rotatedDigits(int N) {
        int count = 0;
        for (int i = 1; i <= N; i++) {
            String cur = String.valueOf(i);
            boolean flag = false;
            for (int j = 0; j < cur.length(); j++) {
                char c = cur.charAt(j);
                if (c == '3' || c == '4' || c == '7') {
                    flag = false;
                    break;
                } else if (c == '2' || c == '5' || c == '6' || c == '9') {
                    flag = true;
                }
            }
            
if (flag) {
                count++;
            }
        }
        return count;
    }
}
