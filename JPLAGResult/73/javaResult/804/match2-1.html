class Solution {
    
    // Good number - After 180 deg, Valid number & Diff
    // Good numbers = (2, 5), (6, 9)
    // Non-Good numbers = (0, 1, 3, 4, 7, 8)
    
    public int rotatedDigits(int N) {
        int result = 0;
        
        for (int n = 1; n <= N; n++) {
            if (isGoodNumber(n))
                result++
;
        }
        
        return result;
    }
    
    public boolean isGoodNumber(int num) {
        boolean isGood = false;
        
        while (num > 0) {
            int digit = num % 10;
 
            if (digit == 2 || digit == 5 || digit == 6 || digit == 9)
                isGood = true;
            
            if (digit == 3 || digit == 4 || digit == 7)
                return false;
            
            num /= 10;
        }
        
        return isGood;
    }
}
