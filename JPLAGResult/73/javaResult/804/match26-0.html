class Solution {
    int total =0;
    public int rotatedDigits(int N) {
        
        for (int i=1;i<=N;i++){
            check(i+"");
        }
        return total;
    }
    
    public void check(String input){
        if (input.contains("3") || input.contains("4") || input.contains("7")){
            return;
        }
        if (!input.contains("2") && !input.contains("5") && !input.contains("6") && !input.contains("9")){
            return;
        }
        total++;
    }
    
   
}
