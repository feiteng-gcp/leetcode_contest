class Solution {
    public int rotatedDigits(int N) {
        int num = 0;
		for (int i = 1; i <= N; i++) {
			String val = String.valueOf(i);
			if (val.indexOf('3') > -1 || val.indexOf('4') > -1 || val.indexOf('7') > -1
					|| (val.indexOf('2') + val.indexOf('5') + val.indexOf('6') + val.indexOf('9')) < -3)
				continue;
			num++;
		}
		return num;
    }
}
