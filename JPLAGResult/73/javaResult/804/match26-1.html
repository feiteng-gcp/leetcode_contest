class Solution {
    private int res = 0;
    public int rotatedDigits(int N) {
        
        for (int i = 2; i <= N; i++) {
            
            check(i);
        }
        return res;
    }
    
    private void check(int cur) {
        char[] c = String.valueOf(cur).toCharArray();
        // int[] num = new int[c.length];

        boolean r = false;
        for (int i = 0; i < c.length; i++){
            int tmp = c[i] - '0';
            if (tmp == 2 || tmp == 5 || tmp == 6 || tmp == 9) {
                r = true;
            }
            if (tmp == 3 || tmp == 4 || tmp == 7) {
                r = false;
                break;
            }
        }
        if (r) {
            res++;
        }
        return;
    }
}
