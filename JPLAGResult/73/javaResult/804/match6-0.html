class Solution {
    
    public int rotatedDigits(int N) {
      int result = 0;
      for (int i = 1; i <= N; i++) {
        if (isGoodNumber(i)) {
          result++;  
        }
      }
      return result;  
    }
    
    private boolean isGoodNumber (int num) {
      String s = String.valueOf(num);
      boolean r = false;
      for (char c : s.toCharArray()) {
        if (c == '3' || c == '4' || c == '7') {
          return false;
        } else if (c == '2' || c == '5' || c == '6' || c == '9') {
          r = true;  
        }  
      }

      return r;  
    }
}
