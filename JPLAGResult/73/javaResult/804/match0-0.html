class Solution {
    //Map<Intger> map = new HashMap<>();
    public int rotatedDigits(int N) {
        int count = 0;
        for(int i=1;i<=N;i++){
            if(isValid(i))
                count++;
        }
        return count;
    }
    
    public boolean isValid(int x){
        int k = 0;
        boolean flag = false;
        while(x > 0){
            k = x % 10;
            if(k == 3 || k == 4 || k == 7)
                return false;
            if(k == 2 || k == 5 || k == 6 || k == 9)
                flag = true;           
            x /= 10;
            
        }
        return flag;
    }
}
