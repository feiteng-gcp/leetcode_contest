class Solution {
    public int rotatedDigits(int N) {
        int res = 0;
        for (int i = 1; i <= N; ++i){
            if (isGood(i)){
                // System.out.println(i);
                res++;
            }
        }
        return res;
    }
    
    public boolean isGood(int num){
        boolean diff = false;
        while (num > 0){
            int k = num % 10;
            if (k == 2 || k == 5 || k == 6 || k == 9){
                diff = true;
            }else if (k == 0 || k == 1 || k == 8){
                
            }else{
                return false;
            }
            num /= 10;
        }
        return diff;
    }
}
