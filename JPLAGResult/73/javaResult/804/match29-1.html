class Solution {
    public int rotatedDigits(int N) {
        Map<Integer, Integer> map = new HashMap();
        map.put(1, 1);
        map.put(0, 0);
        map.put(8, 8);
        map.put(2, 5);
        map.put(5, 2);
        map.put(6, 9);
        map.put(9, 6);
        // Set<Character> set = new HashSet();
        // set.add('1');
        // set.add('0');
        // set.add('8');
        // set.add('2');
        // set.add('5');
        // set.add('6');
        // set.add('9');
        
        int num = 0;
        for (int i = 1; i <= N; i++){
            String str = Integer.toString(i);
            //System.out.println(str.contains("2"));
            if (str.contains("3") || str.contains("4") || str.contains("7") || 
                (!str.contains("2") && !str.contains("5") && !str.contains("6") && !str.contains("9"))) {
                continue;
            }
            num++;
        }
        return num;
    }
}
