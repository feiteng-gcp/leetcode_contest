class Solution {
    public int rotatedDigits(int N) {
        int res = 0;
        List<Integer> list = Arrays.asList(0, 1, 8, 2, 5, 6, 9);
        List<Integer> list2 = Arrays.asList(0, 1, 8);
        
        for (int i = 1; i <= N; i++) {
            int tmp = i;
            int tmp2 = 0;
            boolean rotatable = true;
            boolean different = false;
            while (tmp > 0) {
                int digit = tmp % 10;
                if (!list.contains(digit)) {
                    rotatable = false;
                }
                if (!list2.contains(digit)) {
                    different = true;
                }
                tmp /= 10;
            }
            
            if (rotatable && different) {
                res++;
            }
        }
        
        return res;
    }
}
