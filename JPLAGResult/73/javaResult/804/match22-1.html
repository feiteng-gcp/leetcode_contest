class Solution {    
    public int rotatedDigits(int N) {
        int good = 0;
        for (int i = 1; i <= N; i++) {
            int num = i;
            boolean isGood = false;
            while (num > 0) {
                int digit = num % 10;
                if (digit == 3 || digit == 4 || digit == 7) {
                    isGood = false;
                    break;
                }
                if (digit == 2 || digit == 5 || digit == 6 || digit == 9) 
{
                    isGood = true;
                }
                num /= 10;
            }
            if (isGood) {
                good++;
            }
        }
        return good;
    }
}
