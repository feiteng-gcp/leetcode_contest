class Solution {
    public int rotatedDigits(int N) {
        int count = 0;

        for(int i = 1; i <= N; i++) {
            if(isValid(i)) {
                count++;
            }
        }
        
        return count;
    }
    
    public boolean isValid(int num) {
        boolean flag = false;
        
        while(num > 0) {
            int rem = num % 10;
            if (rem != 0 && rem != 1 && rem != 2 && rem != 5 && rem != 6 && rem != 8 && rem != 9) {
                return false;
            }
            
            else if (rem == 2 || rem == 5 || rem == 6 || rem == 9) {
                flag = true;
            }
            
            num /= 10;
        }
        
        return flag;
    }
}
