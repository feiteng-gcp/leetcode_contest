class Solution {
    
    private boolean isNumberGood(int n) {
        boolean hasDifferentDigit = false;
        while(n > 0) {
            int d = n%10;
            if(d == 3 || d == 4 || d == 7) return false;
            if(d == 2 || d == 5 || d == 6 || d == 9) {
                hasDifferentDigit = true;
            }
            n /= 10;
        }
        return hasDifferentDigit;
    }
    
    public int rotatedDigits(int N) {
        int count = 0;
        for(int i = 1; i <= N; i++) {
            if(isNumberGood(i)) {
                count++;
            }
        }
        return count;
    }
}
