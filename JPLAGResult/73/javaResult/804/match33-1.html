import java.util.*;
import java.math.*;
import java.util.stream.*;


public class Solution {


    public int rotatedDigits(int N) {
        int cnt = 0;
        for (int i = 1; i <= N; i++) {
            if (is(i)) {
                cnt++;
            
}
        }
        return cnt;
    }

    boolean is(int x) {
        int _x = x, base = 1;
        int xx = 0;
        while (x > 0) {
            int digit = x % 10;
            if (digit == 1 || digit == 0 || digit == 8) {
            } else if (digit == 2) {
                digit = 5;
            } else if (digit == 5) {
                digit = 2;
            } else if (digit == 6) {
                digit = 9;
            } else if (digit == 9) {
                digit = 6;
            } else {
                return false;
            }

            xx += digit * base;
            base *= 10;
            x /= 10;
        }
//        if (_x != xx) {
//            System.out.println(_x + " " + xx);
//        }
        return _x != xx;
    }


    public static void main(String[] args) {
        Solution sol = new Solution();
        System.out.println(sol.rotatedDigits(1000));
        System.out.println(sol.rotatedDigits(10));
    }
}
