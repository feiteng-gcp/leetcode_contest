class Solution {
    public int rotatedDigits(int N) {
        HashMap<Integer, Integer> valid = new HashMap<>();
        valid.put(0, 0);
        valid.put(1, 1);
        valid.put(2, 5);
        valid.put(5, 2);
        valid.put(6, 9);
        valid.put(8, 8);
        valid.put(9, 6);
        int ans = 0;
        for(int i = 1; i <= N; i++){
            int cur = i;
            boolean val = false;
            boolean add = true;
            while(cur > 0){
                int res = cur % 10;
                if(!valid.containsKey(res)){
                    add = false;
                    break;
                }
                if(res == 2 || res == 5 || res == 6 || res == 9)
                    val = true;
                
cur = cur / 10;
            }
            if(add && val)
                ans++;
        }
        return ans;
    }
}
