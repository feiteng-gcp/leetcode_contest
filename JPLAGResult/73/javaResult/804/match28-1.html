class Solution {
    public int rotatedDigits(int N) {
        int res = 0;
        for (int i = 1; i <= N; i++) {
            if (
helper(i)) {
                res++;
            }
        }

        return res;
    }

    boolean helper(int value) {
        String str = Integer.toString(value);
        int n = str.length();
        char[] arr = new char[n];
        for (int i = 0; i < n; i++) {
            char c = str.charAt(i);
            if (c == '0' || c == '1' || c == '8') {
                arr[i] = c;
            }
            else if (c == '2'){
                arr[i] = '5';
            }
            else if (c == '5') {
                arr[i] = '2';
            }
            else if (c == '6') {
                arr[i] = '9';
            }
            else 
if (c == '9') {
                arr[i] = '6';
            }
            else {
                return false;
            }
        }

        String rotate = new String(arr);
        return !rotate.equals(str);
    }
}
