class Solution {
    public int rotatedDigits(int N) {
        int count = 0;
        for (int i = 1; i <= N; i++) {
            if (isGood(i)) {
                count++;
            }
        }
        
        return count;
    }
    
    private boolean isGood(int num) {
        String str = "" + num;
        StringBuilder path = new StringBuilder();
        for (int i = 0; i < str.length(); i++) {
            char digit = getReversed(str.charAt(i));
            if (digit == 'a') {
                return false;
            }
            
            path.append(digit);
        }
        
        return !path.toString().equals(str);
    }
    
    private char getReversed(char digit) {
        if (digit == '1' || digit == '0' || digit == '8') {
            return digit;
        } else if (digit == '5') {
            return '2';
        } else if (digit == '2') {
            return '5';
        } else if (digit == '6') {
            return '9';
        } else if (digit == '9') {
            return '6';
        } else {
            return 'a';
        }
    }
}
