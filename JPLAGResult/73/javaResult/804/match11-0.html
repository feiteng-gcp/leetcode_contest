class Solution {
    public int rotatedDigits(int N) {
        
        //  (0)  (1)  2,  5,  6, (8)  9, 
        // (10) (11) 12, 15, 16, 18, 19
        //  20   21  22  25  26  28  29
        
        int count = 0;
        
        for (int i = 0; i <= N; i++) {
            if (isValid(i)) {
                count++;
            }
        }
        
        
        return count;
    }
    
    boolean isValid(int i) {
        
        boolean changed = false;
        
        while (i > 0) {
            
            int d = i % 10;
            if (d != 0 && d != 1 && d != 2 && d != 5 && d != 6 && d != 8 && d != 9) {
                return false;
            } else if (d == 2 || d == 5 || d == 6 || d== 9) {
                changed = true;
            }
            
            i = i / 10;
        }
        return changed;
    }
}
