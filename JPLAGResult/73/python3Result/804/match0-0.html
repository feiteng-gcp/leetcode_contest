class Solution:
    def rotatedDigits(self, N):
        """
        :type N: int
        :rtype: int
        """
        res = 0
        for x in range(N+1):
            xstr = str(x)
            theSame = True
            invalid = False
            for ch in xstr:
                if ch in ['3','4','7']:
                    invalid = True
                    break
                if theSame and ch not in ['0','1','8']:
                    theSame = False
            
            if not invalid and not theSame:
                res +=1
                #print(xstr)
                
        return res
