class Solution:
    def rotatedDigits(self, N):
        """
        :type N: int
        :rtype: int
        """
        ans = 0
        for i in range(N + 1):
            digits = get_digits(i)
            flag_dif = 0
            flag = 1
            for digit in digits:
                if digit in [2, 5, 6, 9]:
                    flag_dif = 1
                elif digit not in [0, 1, 8]:
                    flag = 0
            if flag and flag_dif:
                ans += 1
        return ans


def get_digits(N):
    ls = []
    while N != 0:
        ls.append(N % 10)
        N = N // 10
    return ls
