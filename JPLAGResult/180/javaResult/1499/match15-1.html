class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        long res = 0, sum = 0;
        Integer[] sorted = new Integer[n];
        for (int i = 0; i < n; ++i) {
            sorted[i] = i;
        }
        Arrays.sort(sorted, (
a, b) -> (efficiency[b] - efficiency[a]));
        PriorityQueue<Integer> queue = new PriorityQueue<>();
        for (int i = 0; i < n; ++i) {
            sum += speed[sorted[i]];
            queue.add(speed[sorted[i]]);
            if (queue.size() > k) sum -= 
queue.poll();
            res = Math.max(res, sum * efficiency[sorted[i]]);
        }
        return (int)(res % 1_000_000_007);
    }
}
