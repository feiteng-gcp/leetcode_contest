class Solution {
    public int maxPerformance(int n, int[] spd, int[] efy, int k) {
        
        int mod = 1_000_000_007;

        int[][] g = new int[n][];
        for(int  i = 0; i < n; i++)
        {
            g[i] = new int[]{spd[i], efy[i]};
        }
        Arrays.sort(g, (a, b)->(a[1] - b[1]));

        Queue<Integer> q = new PriorityQueue<>();
        
        long sum = 0, total = 0;
        for(int i = n - 1; i >= 0; i--)
        {
            sum += g[i][0];
            q.offer(g[i][0]);
            while(q.size() > k) sum -= q.poll();
            total = Math.max(total, sum * g[i][1]);
        }
        return (int)(total % mod);
    }
}
