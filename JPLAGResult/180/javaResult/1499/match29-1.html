class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        
        int[][] people = new int[n][2];
        
for (int i=0; i<n; i++)
        {
            people[i][0] = speed[i];
            people[i][1] = efficiency[i];
        }
        
        Arrays.sort(people, (i1, i2)->i2[1]-i1[1]);
        
        PriorityQueue<Integer> speeds = new PriorityQueue<>();
        long totalSpeed = 0;
        
        long res = 0;
        for (int i = 0; i < n; i++) {
            long curr = (long)people[i][1] * (long) (totalSpeed + people[i][0]);
            res = Math.max(res, curr);
            speeds.add(people[i][0]);
            totalSpeed += (long)people[i][0];
            if (speeds.size() == k)
                totalSpeed -= (long)speeds.poll(); // remove the smallest
        }
        return (int) (res%1_000_000_007);
    }
}
