class Solution {
    
public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        Integer[] engineer = new Integer[n];
        for(int i=0;i<n;i++){
            engineer[i]=i;
        
}
        Arrays.sort(engineer, new Comparator<Integer>(){
            @Override
            public int compare(Integer a, Integer b){
                return efficiency[b]-efficiency[a];
            }
        });
        
        int start = n-k-1;
        
Queue<Integer> queue = new PriorityQueue<>(new Comparator<Integer>(){
            @Override
            public int compare(Integer a, Integer b){
                return speed[a]-speed[b];
            
}
        });
        long max = 0;
        long sum=0;
        for(int i=0;i<engineer.length;i++){
            
            long eff = efficiency[engineer[i]];
            
            // for(Integer q : queue){
            //     sum+=speed[q];
            // }
            max = Math.max((sum+speed[engineer[i]])*eff,max);
            
            queue.offer(engineer[i]);
            sum+=speed[engineer[i]];
            if(queue.size()>=k){
                sum-=speed[queue.poll()];
            }
        }
        return (int)(max%1000000007);
    }
}
