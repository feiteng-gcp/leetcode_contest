class Solution {
    
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        
        int a[][]=new int[n][2];
        for(int i=0; i<n; i++) {
            a[i][0]=speed[i];
            a[i][1]=efficiency[i];
        }
        Arrays.sort(a, (a1, a2)->{
            if(a1[1]!=a2[1])
                return a2[1]-a1[1];
            return a2[0]-a1[0];
        });
        
        PriorityQueue<Integer> q=new PriorityQueue<>();
        
        long sum=0, max=0;
        for(int i=0; i<n; i++) {
            if(q.size()==k)
                sum-=q.poll();
            sum+=a[i][0];
            
q.add(a[i][0]);
            max=Math.max(max, sum*a[i][1]);
        }
        
        return (int)(max%1_000_000_007);
    }
}
