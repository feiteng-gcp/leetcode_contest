class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        
        int mod = 1_000_000_007;

        int[][] gg = new int[n][];
        for(int  i = 0; i < n; i++)
        {
            gg[i] = new int[]{speed[i], efficiency[i]};
        }
        Arrays.sort(gg, (a, b)->(a[1] - b[1]));

        Queue<Integer> qq = new PriorityQueue<>();
        
        long sum = 0, total = 0;
        
        for(int i = n - 1; i >= 0; i--)
        {
            sum += gg[i][0];
            qq.offer(gg[i][0]);
            while(qq.size() > k) sum -= qq.poll();
            total = Math.max(total, sum * gg[i][1]);
        }
        return (int)(total % mod);
    }
}
