class Solution {
    class Data implements Comparable<Data> {
        int s, e;
        public Data(int sp, int ef) {
            s = sp; e = ef;
        }
        public int compareTo(Data other) {
            return other.e - e;
        }
    }
    
long modu = 1000000007;
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        Data[] a = new Data[n];
        for (int i=0; i<n; i++)
            a[i] = new Data(speed[i], efficiency[i])
;
        Arrays.sort(a);
        PriorityQueue<Integer> h = new PriorityQueue<>();
        long sum = 0;
        long ans = 0;
        for (int i=0; i<k; i++) {
            h.add(a[i].s);
            sum += a[i].s;
            ans = Math.max(ans, sum * a[i].e);
        }
        for (int i=k; i<n; i++) {
            int x = 
h.poll();
            sum += Math.max(x, a[i].s) - x;
            h.add(Math.max(x, a[i].s));
            ans = Math.max(ans, sum * a[i].e);
        }
        return (int) (ans % modu);
    }
}
