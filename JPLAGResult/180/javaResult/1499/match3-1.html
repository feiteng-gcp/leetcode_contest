class Solution {
    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        int[][]speedEfficiency=new int[n][2];
        for(int i=0;i<n;i++){
            speedEfficiency[i]=new int[]{speed[i],efficiency[i]};
        }
        Arrays.sort(speedEfficiency,(a,b)->a[1]==b[1]?b[0]-a[0]:b[1]-a[1]);
        long curSum=0;
        PriorityQueue<Integer>maxKSpeed=new PriorityQueue<>();
        
long res=0;
        for(int i=0;i<k;i++){
            curSum+=speedEfficiency[i][0];
            maxKSpeed.add(speedEfficiency[i][0]);
            res=Math.max(res,curSum*speedEfficiency[i][1]);
        }
        for(int i=k;i<n;i++){
            if(maxKSpeed.peek()<speedEfficiency[i][0]){
                curSum-=maxKSpeed.poll();
                curSum+=speedEfficiency[i][0];
                maxKSpeed.add(speedEfficiency[i][0]);
                res=Math.max(res,curSum*speedEfficiency[i][1]);
            }
        }
        return (int) (res%1000000007);
    }
}
