class Solution {
    long mod = 1000000007;

    class Node {
        int s;
        int e;
        Node(int s, int e) {
            this.s = s;
            this.e = e;
        }
    }

    public int maxPerformance(int n, int[] speed, int[] efficiency, int k) {
        Node[] arr = new Node[n];
        for(int i=0; i < n; i++) {
            arr[i] = new Node(speed[i], efficiency[i]);
        }
        Arrays.sort(arr, new Comparator<Node>() {
            @Override
            public int compare(Node a, Node b) {
                return Integer.compare(b.e, a.e);
            }
        });

        PriorityQueue<Node> queue = new PriorityQueue<>(new Comparator<Node>() {
            @Override
            public int compare(Node a, Node b) {
                return Integer.compare(a.s, b.s);
            }
        });
        PriorityQueue<Node> queueE = new PriorityQueue<>(new Comparator<Node>() {
            @Override
            public int compare(Node a, Node b) {
                return Integer.compare(a.e, b.e);
            }
        });
        long sum = 0;
        long ans = 0;
        for(int i=0; i < n; i++) {
            Node node = arr[i];
            sum += node.s;
            queue.add(node);
            queueE.add(node);
            if(queue.size() > k) {
                Node top = queue.poll();
                sum -= top.s;
                queueE.remove(top);
            }
            int topE = queueE.peek().e;
            long val = sum * topE;
            if(val > ans) {
                ans = val;
            }
        }
        return (int) (ans % mod);
    }

}
