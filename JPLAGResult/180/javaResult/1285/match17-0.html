/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private void porder(TreeNode root,List<Integer> l){
        if(root == null) return;
        porder(root.left,l);
        l.add(root.val);
        porder(root.right,l);
    }

    private TreeNode cons(List<Integer> l,int from,int to){
        if(from > to) return null;
        int mid = (from + to) / 2;
        TreeNode node = new TreeNode(l.get(mid));
        node.left = cons(l,from,mid - 1);
        node.right = cons(l,mid + 1,to);
        return node;
    }

    public TreeNode balanceBST(TreeNode root) {
        List<Integer> l = new ArrayList<>();
        porder(root,l);
        return cons(l,0,l.size() - 1);
    }
}
