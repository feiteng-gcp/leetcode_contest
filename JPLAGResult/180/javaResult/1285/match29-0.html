/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    List<Integer> values;
    public TreeNode balanceBST(TreeNode root) {
        values = new ArrayList<>();
        add(root);
        return tree(values, 0, values.size());
        
    }
     
    void add(TreeNode root) {
        if (root == null) {
            return;
        }
        add(root.left);
        values.add(root.val);
        add(root.right);
    }
    
    TreeNode tree(List<Integer> values, int from, int to) {
        if (from == to) {
            return null;
        }
        int mid = (from + to) >> 1;
        TreeNode ans = new TreeNode(values.get(mid));
        ans.left = tree(values, from, mid);
        ans.right = tree(values, mid + 1, to);
        return ans;
    }
}
