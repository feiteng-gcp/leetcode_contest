/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    
    private List<Integer> arr;
    
    public TreeNode balanceBST(TreeNode root) {
        // Step 1: Convert the original BST into an array
        arr = new ArrayList<Integer>();
        order(root);
        // Step 2: Reconstruct the BST from the array
        return rebuild(0, arr.size() - 1);
    }
    
    private TreeNode rebuild(
int l, int r) {
        if(l > r) { return null; }
        int m = (l + r) / 2;
        TreeNode midNode = new TreeNode(arr.get(m));
        midNode.left = rebuild(l, m - 1);
        midNode.right = rebuild(m + 1, r);
        return midNode;
    }
    
    private void order(
TreeNode root) {
        if(root == null) { return; }
        order(root.left);
        arr.add(root.val);
        order(root.right);
    }
    
}
