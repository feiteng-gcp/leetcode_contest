/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    void preOrderList(TreeNode root, List<TreeNode> output) {
        if(root == null) return;
        preOrderList(root.left, output);
        output.add(root);
        preOrderList(root.right, output);
        root.left = null;
        root.right = null;
    }
    
    TreeNode buildTree(List<TreeNode> output, int start, int end) {
        if(start > end) return null;
        int mid = (start + end) / 2;
        TreeNode root = output.get(mid);
        root.left = buildTree(output, start, mid - 1);
        root.right = buildTree(output, mid + 1, end);
        return root;
    }
    
    public TreeNode balanceBST(TreeNode root) {
        List<TreeNode> output = new ArrayList<>();
        preOrderList(root, output);
        return buildTree(output, 0, output.size() - 1);
    }
}
