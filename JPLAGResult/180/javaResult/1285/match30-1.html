/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    List<Integer> vals;
    public TreeNode balanceBST(TreeNode root) {
        vals = new ArrayList();
        dfs(root);
        return buildBalanceBST(vals, 0, vals.size()-1);
    }
    
    public void dfs(TreeNode node){
        if(node==null){
            return;
        }
        dfs(node.left);
        vals.add(node.val);
        dfs(node.right);
    }
    
    public TreeNode buildBalanceBST(List<Integer> vals, int lo, int hi){
        if(lo>hi){
            return null;
        }
        if(lo==hi){
            return new TreeNode(vals.get(lo));
        }
        
int mid = lo + (hi-lo)/2;
        TreeNode node = new TreeNode(vals.get(mid));
        node.left = buildBalanceBST(vals, lo, mid-1);
        node.right = buildBalanceBST(vals, mid+1, hi);
        return node;
    }
}
