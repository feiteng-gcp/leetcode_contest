class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int N = matrix.length;
        int M = matrix[0].length;
        
        int[] rows = new int[N]; Arrays.fill(rows, Integer.MAX_VALUE);
        int[] cols = new int[M]; Arrays.fill(cols, Integer.MIN_VALUE);
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                rows[i] = Math.min(rows[i], matrix[i][j]);
                cols[j] = Math.max(cols[j], matrix[i][j]);
            }
        }
        
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (matrix[i][j] == rows[i] && matrix[i][j] == cols[j]) ans.add(matrix[i][j]);
            }
        }
        
        return ans;
    }
}
