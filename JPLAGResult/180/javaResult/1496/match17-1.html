class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        int row = matrix.length, col = matrix[0].length;
        int[] minRow = new int[row], maxCol = new int[col];
        Arrays.fill(minRow, Integer.MAX_VALUE);
        Arrays.fill(maxCol, Integer.MIN_VALUE);
        for (int i = 0; i < row; ++i) {
            for (int j = 0; j < col; ++j) {
                minRow[i] = Math.min(matrix[i][j], minRow[i]);
                maxCol[j] = Math.max(matrix[i][j], maxCol[j]);
            }
        }
        List<Integer> res = new ArrayList<>();
        for (int i = 0; i < row; ++i) {
            for (int j = 0; j < col; ++j) {
                if (matrix[i][j] == minRow[i] && matrix[i][j] == maxCol[j]) res.add(matrix[i][j])
;
            }
        }
        return res;
    }
}
