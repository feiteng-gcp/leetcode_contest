class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        List<Integer> result = new ArrayList<>();
        int m = matrix.length;
        int n = matrix[0].length;
        for (int i = 0; i < m; i++) {
            int rowMin = Integer.MAX_VALUE;
            int index = -1;
            for (int j = 0; j < n; j++) {
                if (matrix[i][j] < rowMin) {
                    rowMin = matrix[i][j];
                    index = j;
                }
            }
            boolean isColMax = true;
            for (int r = 0; r < m; r++) {
                if (matrix[r][index] > rowMin) {
                    isColMax = false;
                    break;
                }
            }
            if (isColMax) {
                result.add(rowMin);
            }
        }
        return result;
    }
}
