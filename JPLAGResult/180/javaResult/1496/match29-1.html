class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
            int m = matrix.length;
    int n = matrix[0].length;
    List<Integer> output = new ArrayList<>();
    
for (int i = 0; i < m; i++) {
      int minj = -1;
      int minv = Integer.MAX_VALUE;
      for (int j = 0; j < n; j++) {
        if (matrix[i][j] < minv) {
          minv = matrix[i][j];
          minj = j;
        }
      }
      boolean ok = true;
      for (int k = 0; k < m; k++) {
        if (matrix[k][minj] > minv) {
          ok = false;
          break;
        }
      }
      if (ok) {
        output.add(minv);
      }
    }
    return output;
    }
}
