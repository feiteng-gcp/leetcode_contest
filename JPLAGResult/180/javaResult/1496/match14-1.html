class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        
int[] rowmin = new int[matrix.length], colmax = new int[matrix[0].length];
        Arrays.fill(rowmin, 1000000);
        for (int i = 0; i < matrix.length; ++i) {
            for (int j = 0; j < matrix[0].length; ++j) {
                rowmin[i] = Math.min(matrix[i][j], rowmin[i]);
                colmax[j] = Math.max(matrix[i][j], colmax[j]);
            }
        }
        List<Integer> res = new ArrayList<>();
        for (int i = 0; i < matrix.length; ++i) {
            for (int j = 0; j < matrix[0].length; ++j) {
                if (matrix[i][j] == rowmin[i] && matrix[i][j] == colmax[j]) res.add(matrix[i][j]);
            }
        }
        return res;
    }
}
