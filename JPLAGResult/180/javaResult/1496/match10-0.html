class Solution {
    public List<Integer> luckyNumbers(int[][] matrix) {
		List<Integer> ans = new ArrayList<Integer>();

		int n = matrix.length;
		int m = matrix[0].length;

		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				boolean ok = true;

				for (int ii = 0; ii < n; ii++) {
					if ((ii != i) && (matrix[ii][j] > matrix[i][j])) {
						ok = false;
						break;
					}
				}

				for (int jj = 0; jj < m; jj++) {
					if ((jj != j) && (matrix[i][jj] < matrix[i][j])) {
						ok = false;
						break;
					}
				}

				if (ok) {
					ans.add(matrix[i][j]);
				}
			}
		}
		
		return ans;
	}
}
