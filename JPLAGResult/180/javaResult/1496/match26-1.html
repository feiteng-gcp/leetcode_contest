class Solution {
    public List<Integer> luckyNumbers (int[][] matrix) {
        List<Integer> res = new ArrayList<>();
        int n = matrix.length, m = matrix[0].length;
        
for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                boolean isLucky = true;
                for (int k = 0; k < m; k++) {
                    if (matrix[i][k] < matrix[i][j]) {
                        isLucky = false;
                    }
                }
                for (int k = 0; k < n; k++) {
                    if (matrix[k][j] > matrix[i][j]) {
                        isLucky = false;
                    }
                }
                if (isLucky) {
                    res.add(matrix[i][j]);
                }
            }
        }
        return res;
    }
}
