class CustomStack {

    int[] ary = new int[1010];
    int idx = 0;
    int size = 0;
    public CustomStack(int maxSize) {
        size = maxSize;
    }
    
    public void push(int x) {
        if(idx >= size) return;
        ary[idx++] = x;
    }
    
    public int pop() {
        if(idx == 0) return -1;
        int val = ary[idx - 1];
        idx--;
        return val;
    }
    
    public void increment(int k, int val) {
        for(int i = 0; i < Math.min(idx, k); i++) ary[i] += val;
    }
}
