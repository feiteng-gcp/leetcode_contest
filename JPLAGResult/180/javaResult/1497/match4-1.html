class CustomStack {

    int[] ar = new int[1010];
    int idx = 0;
    int maxSize = 0;
    public CustomStack(int maxSize) {
        this.maxSize = maxSize;
    }
    
    public void push(int x) {
        if(idx >= maxSize) return;
        ar[idx++] = x;
    }
    
    public int pop() {
        if(idx == 0) return -1;
        int val = ar[idx - 1];
        idx--;
        return val;
    }
    
    public void increment(int k, int val) {
        for(int i = 0; i < Math.min(idx, k); i++) ar[i] += val;
    }
}
