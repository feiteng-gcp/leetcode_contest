class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        vector<int> ans;
        int n = matrix.size(), m = matrix[0].size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int min_elem = *min_element(matrix[i].begin(), matrix[i].end());

                int max_elem = INT_MIN;
                for (int k = 0; k < n; k++) max_elem = max(max_elem, matrix[k][j]);
                if (matrix[i][j] == min_elem && matrix[i][j] == max_elem) ans.push_back(matrix[i][j]);
            }
        }
        return ans;
    }
};
