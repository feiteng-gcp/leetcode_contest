class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
        vector<int> min1(n);
        vector<int> max1(m);
        for(int i = 0; i<n; i++){
            int temp = INT_MAX;
            for(int j = 0; j<m; j++){
                temp = min(matrix[i][j],temp);
            }
            min1[i] = temp;
        }
        for(int i = 0; i<m; i++){
            int temp = INT_MIN;
            for(int j = 0; j<n; j++){
                temp = max(matrix[j][i],temp);
            }
            max1[i] = temp;
        }
        vector<int> ans;
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(matrix[i][j]==min1[i] && matrix[i][j]==max1[j]) ans.push_back(matrix[i][j]);
            }
        }
        return ans;
    }
};
