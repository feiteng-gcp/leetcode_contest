class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        vector<int> ret;
        int n = matrix.size(), m = matrix[0].size();
        for(int i = 0; i < n; i++){
            int minN = matrix[i][0];
            int minPos = 0;
            for(int j = 1; j < m; j++){
                if (matrix[i][j] < minN){
                    minN = matrix[i][j];
                    minPos = j;
                }
            }
            bool isMax = true;
            for (int j = 0; j < n; j++){
                if (j == i) continue;
                if (matrix[j][minPos] > minN){
                    isMax = false;
                    break;
                }
            }
            if (isMax){
                ret.push_back(minN);
            }
        }
        return ret;
    }
};
