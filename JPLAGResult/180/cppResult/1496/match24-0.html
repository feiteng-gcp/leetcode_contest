class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& a) {
        vector<int> ans;
        int n=a.size(),m=a[0].size();
        for(int i=0;i<n;++i) {
            for(int j=0;j<m;++j) {
                int mn=a[i][j],mx=a[i][j];
                for(int k=0;k<m;++k) mn=min(mn,a[i][k]);
                for(int k=0;k<n;++k) mx=max(mx,a[k][j]);
                if(mn == a[i][j] && mx == a[i][j]) ans.push_back(a[i][j]);
            }
        }
        return ans;
    }
};
