class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& m) {
        vector<int> a;
        for(int i =0;i<m.size();i++){
            int pos = 0;
            for(int j = 0;j<m[i].size();j++)if(m[i][j] < m[i][pos])pos = j;
            bool ok = 1;
            for(int j = 0;j<m.size();j++){
                if(m[j][pos] > m[i][pos])ok = 0;
            }
            if(ok)a.push_back(m[i][pos]);
        }
        return a;
    }
};
