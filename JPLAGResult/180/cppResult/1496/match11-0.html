class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        unordered_set<int> s;
        for (auto &v : matrix) {
            s.insert(*min_element(v.begin(), v.end()));
        }
        vector<int> res;
        int n = matrix.size(), m = matrix[0].size();
        for (int j = 0; j < m; ++j) {
            int Max = 0;
            for (int i = 0; i < n; ++i) {
                Max = max(Max, matrix[i][j]);
            }
            if (s.count(Max)) res.push_back(Max);
        }
        return res;
    }
};
