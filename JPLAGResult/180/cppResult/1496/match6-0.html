class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        vector<int> p(n);
        vector<int> t(m);
        for(int i=0;i<n;i++)
        {
            int ans = 1e9+5;
            for(int j=0;j<m;j++)
            {
                ans = min(ans,a[i][j]);
            }
            p[i] = ans;
        }
        for(int j=0;j<m;j++)
        {
            int ans = 0;
            for(int i=0;i<n;i++)
            {
                ans = max(ans,a[i][j]);
            }
            t[j] = ans;
        }
        vector<int> ans;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                if(a[i][j] == p[i] && a[i][j] == t[j])
                {
                    ans.push_back(a[i][j]);
                }
            }
        }
        return ans;
    }
};
