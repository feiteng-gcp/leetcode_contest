class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& dp) {
        int m = dp.size();
        int n = dp[0].size();
        vector<int> mm(m, 0);
        vector<int> nn(n, 0);
        for(int i=0;i<m;i++)
        {
            int tmp = INT_MAX;
            for(int j=0;j<n;j++)
            {
                tmp = min(tmp, dp[i][j]);
            }
            mm[i] = tmp;
        }
        for(int i=0;i<n;i++)
        {
            int tmp = 0;
            for(int j=0;j<m;j++)
            {
                tmp = max(tmp, dp[j][i]);
            }
            nn[i] = tmp;
        }
        vector<int> res;
        for(int i=0;i<m;i++)
        {

            for(int j=0;j<n;j++)
            {
                if(dp[i][j] == mm[i] && dp[i][j] == nn[j])
                {
                    res.push_back(dp[i][j]);
                }
            }

        }
        return res;
    }
};
