class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& M) {
        int r = M.size(), c = M[0].size();
        vector<int> row(r,INT_MAX), col(c,INT_MIN);
        for(int i=0;i<r;i++) {
            for(int j=0;j<c;j++) {
                row[i] = min(row[i],M[i][j]);
            }
        }
        for(int i=0;i<c;i++) {
            for(int j=0;j<r;j++) {
                col[i] = max(col[i],M[j][i]);
            }
        }
        vector<int> ans;
        for(int i=0;i<r;i++) {
            for(int j=0;j<c;j++) {
                if(M[i][j] == row[i] && M[i][j] == col[j]) ans.push_back(M[i][j]);
            }
        }
        return ans;
    }
};
