class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& A) {
        int m = A.size();
        int n = A[0].size();
        
        vector<int> row(m, 1000000);
        vector<int> col(n, 0);
        
        vector<int> ans;
        
        for(int i = 0; i < m; i++)
        {
            for(int j = 0; j < n; j++)
            {
                row[i] = min(A[i][j], row[i]);
            }
        }
        
        for(int j = 0; j < n; j++)
        {
            for(int i = 0; i < m; i++)
            {
                col[j] = max(A[i][j], col[j]);
            }
        }
        
        for(int i = 0; i < m; i++)
        {
            for(int j = 0; j < n; j++)
            {
                if(row[i] == col[j])
                {
                    ans.push_back(row[i]);
                }
            }
        }
        
        return ans;
    }
};
