class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& ma) {
        vector<int> ret;
        for (int i = 0; i < ma.size(); ++i) {
            for (int j = 0; j < ma[i].size(); ++j) {
                int x = ma[i][j];
                int mn = *min_element(ma[i].begin(), ma[i].end());
                int mx = x;
                for (int k = 0; k < ma.size(); ++k)
                    mx = max(mx, ma[k][j]);
                if (x == mn && x == mx)
                    ret.push_back(x);
            }
        }
        return ret;
    }
};
