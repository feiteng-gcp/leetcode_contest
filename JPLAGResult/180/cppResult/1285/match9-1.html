/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* balanceBST(TreeNode* root) {
        vector<int> vec;
        vectorizeBST(root, vec);
        
        return buildBST(vec, 0, vec.size() - 1);
    }
    
    void vectorizeBST(TreeNode* root, vector<int>& vec) {
        if (root == NULL) {
            return;
        }
        
        vectorizeBST(root->left, vec);
        vec.push_back(root->val);
        vectorizeBST(root->right, vec);
    }
    
    TreeNode* buildBST(vector<int>& vec, int lo, int hi) {
        if (hi < lo) {
            return NULL;
        }
        
        int mi = (lo + hi + 1) / 2;
        TreeNode* root = new TreeNode(vec[mi]);
        root->left = buildBST(vec, lo, mi - 1);
        root->right = buildBST(vec, mi + 1, hi);
        
        return root;
    }
};
