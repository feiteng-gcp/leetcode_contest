/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    vector<int> val;
    void getv(TreeNode* root){
        if(!root) return;
        val.push_back(root->val);
        getv(root->left);
        getv(root->right);
    }
    
    TreeNode* build(int l, int r){
        if(l == r) return new TreeNode(val[l]);
        else if(l>r) return nullptr;

        int mid = (l+r)>>1;
        TreeNode* rt = new TreeNode(val[mid]);
        rt->left = build(l, mid-1);
        rt->right = build(mid+1, r);
        return rt;
    }
    
    TreeNode* balanceBST(TreeNode* root) {
        getv(root);
        sort(val.begin(), val.end());
        return build(0, val.size()-1);
    }
};
