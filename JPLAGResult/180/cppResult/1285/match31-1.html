class Solution {
public:
  vector<int> vs;
  void dfs(TreeNode* v){
    if(!v) return;
    dfs(v->left);
    vs.emplace_back(v->val);
    dfs(v->right);
  }

  // [l, r]
  TreeNode* create(int l,int r){
    if(l==r) return new TreeNode(vs[l]);

    if(l+1==r){
      TreeNode* x=new TreeNode(vs[l]);
      TreeNode* y=new TreeNode(vs[r]);
      x->right=y;
      return x;
    }


    int m=(l+r)/2;
    TreeNode* z=new TreeNode(vs[m]);
    z->left=create(l,m-1);
    z->right=create(m+1,r);
    return z;
  }

  TreeNode* balanceBST(TreeNode* root) {
    vs.clear();
    dfs(root);

    sort(vs.begin(),vs.end());
    return create(0,vs.size()-1);
  }
};
