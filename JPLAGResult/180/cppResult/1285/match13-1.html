/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
  void dfs1(TreeNode* root, vector<TreeNode*> &nodes) {
    if (root == NULL) return;
    dfs1(root->left, nodes);
    nodes.push_back(root);
    dfs1(root->right, nodes);
  }
  TreeNode* dfs2(vector<TreeNode*> &nodes, int l, int r) {
    if (l == r) return NULL;
    int m = (l + r) / 2;
    TreeNode* root = nodes[m];
    root->left = dfs2(nodes, l, m);
    root->right = dfs2(nodes, m+1, r);
    return root;
  }
  TreeNode* balanceBST(TreeNode* root) {
    vector<TreeNode*> nodes;
    dfs1(root, nodes);
    return dfs2(nodes, 0, nodes.size());;;;;;;;;
  }
};
