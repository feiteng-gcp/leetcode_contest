/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* balanceBST(TreeNode* root) {
        vector<int> v;
        getAllVal(root, v);
        sort(v.begin(), v.end());
        return f(0, v.size()-1, v);
    }
private:
    void getAllVal(TreeNode* root, vector<int> &v) {
        if (!root) return;
        v.push_back(root->val);
        getAllVal(root->left, v);
        getAllVal(root->right, v);
    }
    
    TreeNode *f(int l, int r, vector<int> &v) {
        if (l > r) return nullptr;
        int mid = l + (r-l) / 2;
        auto t = new TreeNode(v[mid]);
        t->left = f(l, mid-1, v);
        t->right = f(mid+1, r, v);
        return t;
    }
};
