class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<pair<int,int>> member;
        for (int i = 0; i<n; ++i) {
            member.push_back({efficiency[i], speed[i]});
        }
        sort(member.begin(), member.end());
        
        priority_queue<int, vector<int>, greater<int>> heap;
        
        const long long M = 1000000000+7;

        long long res = 0;
        
        long long sumH = 0;
        for (int i = n-1; i>n-k; --i) {
            long long eff = member[i].first;
            long long spe = member[i].second;
            res = max(res, eff*(sumH+spe));
            sumH += member[i].second;
            heap.push(member[i].second);            
        }
        
        for (int i = n-k; i>=0; --i) {
            long long eff = member[i].first;
            long long spe = member[i].second;
            res = max(res, eff*(sumH+spe));
            
            if (heap.size() && spe>heap.top()) {
                sumH -= heap.top();
                heap.pop();
                heap.push(spe);
                sumH += spe;
            }
        }
        
        return res % M;
    }
};
