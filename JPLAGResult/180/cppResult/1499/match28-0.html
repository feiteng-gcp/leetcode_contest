class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<pair<int, int>> workers;
        for (int i = 0; i < n; ++i) {
            workers.push_back({efficiency[i], speed[i]});
        }
        sort(workers.begin(), workers.end());
        multiset<int> sums;
        long long sum = 0;
        long long maxP = 0;
        for (int i = n - 1; i >= 0; --i) {
            long long p = 1LL * workers[i].first * (sum + workers[i].second);
            maxP = max(maxP, p);
            sums.insert(workers[i].second);
            sum += workers[i].second;
            if (sums.size() > k - 1) {
                sum -= *sums.begin();
                sums.erase(sums.begin());
            }
        }
        return maxP % static_cast<int>(1e9 + 7);
    }
};
