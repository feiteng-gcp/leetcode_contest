bool cmp(const pair<int, int>&a, const pair<int, int>&b)
{
	return a.second > b.second;
}

const int Mod = 1e9 + 7;

class Solution {
public:
	int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) 
	{
		vector<pair<int, int>>temp;
		for (int i = 0; i < n; i++)
		{
			temp.push_back(make_pair(speed[i], efficiency[i]));
		}
		priority_queue<int, vector<int>, greater<int>>que;
		sort(temp.begin(), temp.end(), cmp);
		for (auto it : temp)
		{
			cout << it.first << " " << it.second << endl;
		}
		long long res = 0;
		long long sumcount = 0;
		for (int i = 0; i < k; i++)
		{
			sumcount += temp[i].first;
			que.push(temp[i].first);
			res = max(res, sumcount*temp[i].second);
		}
		for (int i = k; i < n; i++)
		{
			sumcount += temp[i].first;
			que.push(temp[i].first);
			sumcount -= que.top();
			que.pop();
			res = max(res, sumcount*temp[i].second);
		}
		res = res % Mod;
		return (int)res;
	}
};
