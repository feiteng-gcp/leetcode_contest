constexpr int mod = 1e9+7;
using ull = std::uint64_t;

class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<int> ids(n, 0);
        for (int i = 0; i < n; ++i) ids[i] = i;
        
        sort(ids.begin(), ids.end(), [&](int a, int b) { return efficiency[a] > efficiency[b]; });
        priority_queue<ull, vector<ull>, std::greater<ull>> heap;

        ull heap_sum = 0;
        ull ans = 0;
        
        // iter
        for (int i = 0; i < n; ++i) {
            int curr_id = ids[i];
            ull curr_speed = speed[curr_id];
            if (heap.size() < k) {
                heap.push(curr_speed);
                heap_sum += curr_speed;
            } else if (curr_speed > heap.top()) {
                ull top_speed = heap.top();
                heap.pop();
                heap.push(curr_speed);
                heap_sum = heap_sum - top_speed + curr_speed;
            }
            ans = max(ans, efficiency[curr_id] * heap_sum);
        }
        
        return ans % mod;
    }
};
