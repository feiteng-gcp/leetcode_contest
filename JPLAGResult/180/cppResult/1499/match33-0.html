typedef long long ll;

struct engineer
{
    int speed;
    int efficiency;
};

bool operator<(const engineer &lhs, const engineer &rhs)
{
    if(lhs.efficiency != rhs.efficiency) return lhs.efficiency < rhs.efficiency;
    return lhs.speed < rhs.speed;
}

class Solution {
public:
    engineer engineers[100000+5];
    ll max_selected_sum[100000+5];
    
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        for(int i=0;i<n;++i)
         {
            engineers[i].speed = speed[i];
            engineers[i].efficiency = efficiency[i];
        }
        sort(engineers, engineers + n);
        
        priority_queue<int> pq;
        ll cur_sum = 0LL;
        ll ans = 0LL;

        for(int i=n-1;i>=0;--i)
        {
            pq.push(-engineers[i].speed);
            cur_sum += engineers[i].speed;
            if(pq.size() == k)
            {
                cur_sum += pq.top();
                pq.pop();
            }
            max_selected_sum[i] = cur_sum;
        }

        for(int i=0;i<n;++i)
        {
            ll temp = 1LL * (engineers[i].speed + max_selected_sum[i+1]) * engineers[i].efficiency;
            if(temp > ans) ans = temp;
        }

        return ans%(1000000000+7);
    }
};
