class Solution {
public:
    int maxPerformance(int n, vector<int>& a, vector<int>& b, int k) {
        vector<pair<int,int>>c;
        for(int i=0;i<a.size();i++){
            c.push_back(make_pair(b[i],a[i]));
        }
        sort(c.begin(),c.end());
        reverse(c.begin(),c.end());
        multiset<int>s;
        long long sum=0;
        int mod=1e9+7;
        long long ret=0;
        for(int i=0;i<c.size();i++){
            sum+=c[i].second;
            s.insert(c[i].second);
            if(s.size()>k){
                int t=*s.begin();
                s.erase(s.begin());
                sum-=t;
            }
            ret=max(ret,sum*c[i].first);
        }
        return (int)(ret%mod);
    }
};
