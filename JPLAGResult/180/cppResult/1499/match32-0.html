class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<pair<int, int> > a;
        for(int i = 0; i<n; i++) {
            a.push_back({speed[i], efficiency[i]});
        }
        sort(a.begin(), a.end(), [](const pair<int, int> &b, const pair<int, int> &c){
            if (b.second!=c.second) return b.second > c.second;
            return b.first > c.first;
        });
        priority_queue<int, vector<int>, greater<int> > qe;
        long long sum = 0;
        long long ans = 0;
        for(int i = 0; i<k-1; i++) {
            qe.push(a[i].first);
            sum += a[i].first;
            ans = max(ans, sum*a[i].second);
        }
        
        for(int i = k-1; i<n; i++) {
            sum += a[i].first;
            qe.push(a[i].first);
            long long m = a[i].second;
            ans = max(ans, sum*m);
            sum -= qe.top();
            qe.pop();
        }
        const long long M = 1e9+7;
        return ans%M;
        
        
    }
};
