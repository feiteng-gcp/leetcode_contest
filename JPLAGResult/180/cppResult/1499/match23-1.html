typedef long long LL;
class Node{
public:
    LL sp;
    LL eff;
    Node(LL s,LL e){
        sp=s;
        eff=e;
    }
    bool operator<(const Node&other)const{
        return eff<other.eff;
    }
};

class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<Node> vNode;
        for(int i=0;i<n;i++){
            Node tmp(speed[i],efficiency[i]);
            vNode.push_back(tmp);
        }
        sort(vNode.begin(),vNode.end());
        LL ans=0;
        LL BASE=pow(10,9)+7;

        LL eff=INT_MAX;
        priority_queue<LL,vector<LL>,greater<LL>> pSpeed;
        LL add=0;
        for(int i=vNode.size()-1;i>=0;i--){
            eff=vNode[i].eff;  ///当前的效率值
            add+=vNode[i].sp;
            pSpeed.push(vNode[i].sp);
            if(pSpeed.size()>k){
                add-=pSpeed.top();
                pSpeed.pop();
            }
            ans=max(ans,eff*add);
        }
        
        return ans%BASE;
    }
};
