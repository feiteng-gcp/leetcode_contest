class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<pair<int, int>> data;
        for (int i = 0;i < n;i++) {
            data.push_back(make_pair(efficiency[i], speed[i]));
        }
        sort(data.begin(), data.end());
        reverse(data.begin(), data.end());
        priority_queue<int, vector<int>, greater<int>> Q;
        long long s = 0;
        long long ret = 0;
        for (int i = 0;i < n;i++) {
            if (Q.size() < k) {
                Q.push(data[i].second);
                s += data[i].second;
            } else {
                if (Q.top() < data[i].second) {
                    s -= Q.top();
                    Q.pop();
                    Q.push(data[i].second);
                    s += data[i].second;
                }
            }
            ret = max(ret, s * data[i].first);
        }
        return ret % 1000000007;
    }
};
