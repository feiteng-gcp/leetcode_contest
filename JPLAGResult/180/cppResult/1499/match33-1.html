#define ll long long
#define mod 1000000007ll

typedef pair<ll,ll> P;

bool cmp( P x, P y){return x.first>y.first;}
class Solution {
public:
    vector<P> v;
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        for(int i=0;i<n;i++) v.push_back(P((ll)efficiency[i],(ll)speed[i]));
        sort(v.begin(),v.end(),cmp);
        priority_queue<ll,vector<ll>,greater<ll> > pq;
        ll e=INT_MAX;

        ll tot=0;
        ll ans=0;
        for(int i=0;i<n;i++){
            e=min(e,v[i].first);
            if(pq.size()==k){
                tot-=pq.top();
                pq.pop();
            }
            tot+=v[i].second;
            pq.push(v[i].second);
            ans=max(ans,tot*e);
        }
        return ans%mod;
    }
};
