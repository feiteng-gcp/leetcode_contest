#define iPair pair<int,int>
typedef long long ll;

class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        vector<iPair> arr;
        for(int i = 0; i < n; i++) {
            arr.push_back(make_pair(efficiency[i], speed[i]));
        }
        sort(arr.rbegin(), arr.rend());
        multiset<int> pq;
        
        ll ans = 0;
        ll sm = 0;
        for(int i = 0; i < n; i++) {
            ll ef = arr[i].first;
            pq.insert(arr[i].second);
            sm += arr[i].second;
            
            if(i >= k) {
                sm -= *(pq.begin());
                pq.erase(pq.begin());
            }
            ans = max(ans, sm*ef);
        }
        
        
        return ans % 1000000007;
    }
};
