class CustomStack {
public:
    int cap;
    vector<int> stk;
    CustomStack(int maxSize) {
        cap = maxSize;
    }
    
    void push(int x) {
        if (stk.size() < cap)
            stk.push_back(x);
    }
    
    int pop() {
        if (stk.size() == 0)
            return -1;
        int v = stk.back();
        stk.pop_back();
        return v;
    }
    
    void increment(int k, int val) {
        for (int i=0; i < k && i < stk.size(); ++i)
            stk[i] += val;
    }
};

 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
