class CustomStack {
public:
    vector<int> dat;
    int top;
    int size;
    CustomStack(int maxSize) {
        dat.resize(maxSize);
        top = 0;
        size = maxSize;
    }
    
    void push(int x) {
        if(top < size) {
            dat[top++] = x;
        }
    }
    
    int pop() {
        if(top == 0) return -1;
        return dat[--top];

    }
    
    void increment(int k, int val) {
        for(int i = 0; i < min(k, top); ++i) {
            dat[i] += val;
        }
    }
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
