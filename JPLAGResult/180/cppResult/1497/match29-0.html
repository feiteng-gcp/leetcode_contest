class CustomStack {
public:
    int top,A[1010],maxx;
    CustomStack(int maxSize) {
        maxx = maxSize;
        top = 0;
    }
    
    void push(int x) {
        if(top<maxx)A[++top]=x;
    }
    
    int pop() {
        if(top==0)return -1;
        return A[top--];
    }
    
    void increment(int k, int val) {
        int x=min(top,k);
        for(int i=1;i<=x;i++)A[i]+=val;
    }
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
