class CustomStack {
public:
    CustomStack(int maxSize) {
        max_size_ = maxSize;
        arr_.clear();
    }
    
    void push(int x) {
        if (arr_.size() < max_size_) arr_.push_back(x);
    }
    
    int pop() {
        if (arr_.empty()) return -1;
        int x = arr_.back();
        arr_.pop_back();
        return x;
    }
    
    void increment(int k, int val) {
        int num_elems = min(k, (int)arr_.size());
        for(int i=0;i<num_elems;++i) arr_[i] += val;
        
    }
    private:
        vector<int> arr_;
        int max_size_;
};

 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
