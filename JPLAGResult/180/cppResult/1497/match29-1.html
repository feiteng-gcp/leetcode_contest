class CustomStack {
public:
    int mx;
    int a[1005];

    int p;
    CustomStack(int maxSize) {
        mx=maxSize;
        p=0;
    }
    
    void push(int x) {
        if(p<mx)a[p++]=x;
    }
    
    int pop() {
        if(p==0)return -1;
        return a[--p];
    }
    
    void increment(int k, int val) {
        int tmp=min(k,p);
        for(int i=0;i<tmp;i++)a[i]+=val;
    }
};

 * CustomStack* obj = new CustomStack(maxSize);

 * obj->increment(k,val);
