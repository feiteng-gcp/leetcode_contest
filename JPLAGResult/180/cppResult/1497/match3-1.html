class CustomStack {
public:
  int cap;
  vector<int> S;
  CustomStack(int maxSize) {
    cap = maxSize;
  }

  void push(int x) {
    if (S.size() >= cap) return;
    S.push_back(x);

  }

  int pop() {
    if (S.size() == 0) return -1;
    int x = S.back();
    S.pop_back();
    return x;
  }

  void increment(int k, int val) {
    for (int j = 0; j < k && j < S.size(); j++) S[j] += val;
  }
};

/**
 * Your CustomStack object will be instantiated and called as such:
 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
 */
