class CustomStack {
public:
    CustomStack(int maxSize): size(maxSize) {
        stk.reserve(size);
    }
    
    void push(int x) {
        if (stk.size() < size) {
            stk.emplace_back(x);
        }
    }
    
    int pop() {
        if (stk.empty()) {
            return -1;
        }
        int val = stk.back();
        stk.pop_back();
        return val;
    }
    
    void increment(int k, int val) {
        for (int i = 0; i < k && i < stk.size(); ++i) {
            stk[i] += val;
        }
    }
private:
    const int size;
    vector<int> stk;
};

 * CustomStack* obj = new CustomStack(maxSize);
 * obj->push(x);
 * int param_2 = obj->pop();
 * obj->increment(k,val);
