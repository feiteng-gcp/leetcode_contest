class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        M, N = len(matrix), len(matrix[0])
        mins = [0] * M
        maxs = [0] * N
        for i in range(M):
            mins[i] = matrix[i][0]
            for j in range(1, N):
                mins[i] = min(mins[i], matrix[i][j])
        for i in range(N):
            maxs[i] = matrix[0][i]
            for j in range(1, M):
                maxs[i] = max(maxs[i], matrix[j][i])
        ans = []
        for i in range(M):
            for j in range(N):
                tmp = matrix[i][j]
                if tmp == mins[i] and tmp==maxs[j]:
                    ans.append(tmp)
        return ans
