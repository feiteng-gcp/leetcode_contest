class Solution:
    def luckyNumbers (self, A: List[List[int]]) -> List[int]:
        M, N = len(A), len(A[0])
        MAX = 10**9
        row_min, col_max = [MAX]*M, [-MAX]*N
        
for i in range(M):
            for j in range(N):
                row_min[i] = min(row_min[i], A[i][j])
                col_max[j] = max(col_max[j], A[i][j])
        res = []
        for i in range(M):
            for j in range(N):
                if A[i][j] == row_min[i] == col_max[j]:
                    res.append(A[i][j])
        return res
