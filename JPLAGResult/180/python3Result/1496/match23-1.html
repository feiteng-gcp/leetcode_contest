class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        inf = 100000 + 1
        m = len(matrix)
        if m == 0:
            return []
        n 
= len(matrix[0])
        ret = []
        for idx, row in enumerate(matrix):
            row_min = row[0]
            min_idx = 0
            for col_idx, x in enumerate(row):
                if x < row_min:
                    min_idx = col_idx
                    row_min = x
            flag = True
            for row_idx in range(0, m):
                if matrix[row_idx][min_idx] > row_min:
                    flag = False
                    break
            if flag:
                ret.append(row_min)
        return ret
