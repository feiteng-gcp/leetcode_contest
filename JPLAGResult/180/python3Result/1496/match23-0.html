class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        if not matrix:
            return []
        numRow = len(matrix)
        numCol = len(matrix[0])
        res = []
        for i, row in enumerate(matrix):
            min_val = row[0]
            min_index = 0
            for j, num in enumerate(row):
                if num < min_val:
                    min_val = num
                    min_index = j
            inValid = False
            for j in range(numRow):
                if matrix[j][min_index] > min_val:
                    inValid = True
                    break
            if not inValid:
                res.append(min_val)
        return res
