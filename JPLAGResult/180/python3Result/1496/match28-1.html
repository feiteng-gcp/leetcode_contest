class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        m, n = len(matrix), len(matrix[0])
        rows = [math.inf] * m
        cols = [-math.inf] * n
        
        
for y in range(m):
            for x in range(n):
                rows[y] = min(rows[y], matrix[y][x])
                cols[x] = max(cols[x], matrix[y][x])
        ln=[]
        for y in range(m):
            for x in range(n):
                if rows[y] == cols[x]:
                    ln.append(rows[y])
        
        return ln
