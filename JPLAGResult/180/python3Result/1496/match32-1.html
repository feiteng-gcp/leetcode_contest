class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        n, m = len(matrix), len(matrix[0])
        rowmin = [min(row) for row in matrix]
        colmax = [0] * m
        for i in range(m):
            for j in range(n):
                colmax
[i] = max(matrix[j][i], colmax[i])
        res = []
        for i in range(n):
            for j in range(m):
                if rowmin[i] == colmax[j] == matrix[i][j]:
                    res.append(matrix[i][j])
        return res
