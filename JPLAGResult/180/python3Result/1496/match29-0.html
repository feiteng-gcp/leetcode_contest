import numpy as np
class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        m=len(matrix)
        n=len(matrix[0])
        A=np.array(matrix)
        ans=[]
        for i in range(m):
            for j in range(n):
                if A[i,j]==min(A[i,:]) and A[i][j]==max(A[:,j]):
                    ans.append(A[i,j])
        return ans
