class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        m,n=len(matrix),len(matrix[0])
        row=[min(r) for r in matrix]
        col=[0]*n
        for j in range(n):
            for i in range(m):
                col[j]=max(col[j],matrix[i][j])
        ans=[]
        for i in range(m):
            for j in range(n):
                if matrix[i][j]==row[i]==col[j]:
                    ans.append(matrix[i][j])
        return ans
