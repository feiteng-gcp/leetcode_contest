class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        m, n = len(matrix), len(matrix[0])
        minRow = [9999999] * m # min element in row i
        maxCol = [-9999999] * n # max element in col j
        for i in range(0, m):
            for j in range(0, n):
                val = matrix[i][j]
                minRow[i] = min(minRow[i], val)
                maxCol[j] = max(maxCol
[j], val)
        res = []
        for i in range(0, m):
            for j in range(0, n):
                val = matrix[i][j]
                if val == minRow[i] and val == maxCol[j]:
                    res.append(val)
        return res
