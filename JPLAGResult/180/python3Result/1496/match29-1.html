import numpy as np
class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        m = len(matrix)
        n = len(matrix[0])
        ans = []
        G = np.array(matrix)
        for i in range(m):
            for j in range(n):
                if G[i,j] == min(G
[i,:]) and G[i,j] == max(G[:,j]):
                    ans.append(G[i,j])
                    
        return ans
