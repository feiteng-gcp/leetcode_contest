class Solution:
    def luckyNumbers (self, matrix: List[List[int]]) -> List[int]:
        n, m = len(matrix), len(matrix[0])
        r = [float('inf')] * n
        c = [float('-inf')] * m
        for i in range(n):
            for j in range(m):
                r[i] = min(r[i], matrix[i][j])
                c
[j] = max(c[j], matrix[i][j])
        ret = []
        for i in range(n):
            for j in range(m):
                if matrix[i][j] == r[i] == c[j]:
                    ret.append(r[i])
        return ret
