import heapq


class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        arr = list(zip(efficiency, speed))
        arr.sort(reverse=True)
        # choose max k
        heap = []
        summ = 0
        ans = 0
        for i in range(len(arr)):
            min_eff = arr[i][0]
            # push out min
            if len(heap) >= k:
                evicted = heapq.heappop(heap)
                summ -= evicted
            heapq.heappush(heap, arr[i][1])
            summ += arr[i][1]
            ans = max(ans, min_eff * summ)
        return ans % (10**9 + 7)
