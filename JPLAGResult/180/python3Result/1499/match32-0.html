class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        
        vals = []
        for i in range(n):
            vals.append((efficiency[i], speed[i]))
        vals.sort(reverse = True)

        res = -1
        topK = []
        total = 0
        for val in vals:
            eff = val[0]
            sp = val[1]
            
            if len(topK) < k:
                topK.append(sp)
                total += sp
                if len(topK) == k:
                    topK.sort(reverse = True)
            else:
                if sp <= topK[-1]:
                    continue
                else:
                    total -= topK[-1]
                    total += sp
                    topK[-1] = sp
                    topK.sort(reverse=True)

            currMax = eff * total
            if currMax > res:
                res = currMax

        return res % (10**9 + 7)
