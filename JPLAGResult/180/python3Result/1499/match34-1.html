MOD = 10**9 + 7



class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        # binary search greedy
        
        # from highest efficiency down to lowest
        
        # add contributions
        # challenge the maximum
        
        a = list(zip(efficiency, speed))
        a.sort(reverse=True)
        
        # print(a)
        
        
        # heap actually. heap of the k speediest
        # minheap yes
        
        heap = []
        
        ret = 0
        
        tot = 0
        for ef,sp in a:
            # heappushpop I guess?
            # or heapreplace. I forget which one
            
            tot += sp
            heappush(heap, sp)
            
            if len(heap) > k:
                tot -= heappop(heap)
            
            ret = max(ret, tot * ef)
            
        
        return ret % MOD
