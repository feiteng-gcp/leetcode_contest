class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        if n == 0:
            return 0
        
        l = []
        for s,e in zip(speed, efficiency):
            l.append((s,e))
        l.sort(key=lambda x:(-x[1], -x[0]))
        
        ans = 0
        min_e = float('inf')
        max_speeds = []
        sum_speeds = 0
        
        for s,e in l:
            min_e = min(min_e, e)
            heapq.heappush(max_speeds, s)
            sum_speeds += s
            if len(max_speeds) > k:
                slow = heapq.heappop(max_speeds)
                sum_speeds -= slow
            ans = max(ans, sum_speeds * min_e)
            
        return ans % (10 ** 9 + 7)
