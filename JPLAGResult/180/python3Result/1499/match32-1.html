class Solution:
    def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
        import heapq
        eng
=[]
        for i in range(n):
            eng.append((efficiency[i],speed[i]))
        eng.sort(reverse=True)
        ret=0
        totalS=0
        spds=[]
        for eff,spd in eng:
            totalS+=spd
            heapq.heappush(spds,spd)
            if len(spds)>k:
                subt=heappop(spds)
                totalS-=subt
            ret=max(ret,totalS*eff)
        return ret%1000000007
