import bisect
from collections import deque



class Solution:
	def maxPerformance(self, n: int, speed: List[int], efficiency: List[int], k: int) -> int:
		arr = [(s, e) for s, e in zip(speed, efficiency)]
		arr.sort(key=lambda e: (e[1], e[0]))

		su = 0
		sd = deque()

		ans = 0

		for i in range(k):
			su += arr[n-1-i][0]
			bisect.insort(sd, arr[n-1-i][0])
			ans = max(ans, su * arr[n-1-i][1])

		for i in range(n-1-k, -1, -1):
			su += arr[i][0]
			bisect.insort(sd, arr[i][0])
			su -= sd.popleft()
			ans = max(ans, su * arr[i][1])

		return ans % (10 ** 9 + 7)
