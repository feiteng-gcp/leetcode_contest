# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None



class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        vals = []
        def search(node):
            if node is None:
                return
            vals.append(node.val)
            search(node.left)
            search(node.right)

        search(root)
        vals.sort()

        def create_balanced(i, j):
            if i == j:
                return None
            m 
= (i+j)//2
            node_mid = TreeNode(vals[m])
            node_mid.left = create_balanced(i, m)
            node_mid.right = create_balanced(m+1, j)
            return node_mid

        return create_balanced(0, len(vals))
