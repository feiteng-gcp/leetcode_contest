# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        def serialize(node):
            if not node:
                return []
            return serialize(node.left) + [node.val] + serialize(node.right)
        
        def buildTree(arr):
            if not arr:
                return None
            mid = len(arr) // 2
            treeRoot = TreeNode(arr[mid])
            treeRoot.left = buildTree(arr[:mid])
            treeRoot.right = buildTree(arr[mid+1:])
            return treeRoot
        
        vals = serialize(root)
        return buildTree(vals)
