# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def balanceBST(self, root: TreeNode) -> TreeNode:
        data = []
        def helper(node):
            if node is None: return
            helper(node.left)
            data.append(node.val)
            helper(node.right)
        helper(root)
        
def build(l, r):
            if l == r:
                return None
            if l + 1 == r:
                return TreeNode(data[l])
            m = (l + r) // 2
            node = TreeNode(data[m])
            node.left = build(l, m)
            node.right = build(m + 1, r)
            return node
        return build(0, len(data))
