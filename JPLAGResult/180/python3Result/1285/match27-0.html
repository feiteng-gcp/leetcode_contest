# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def dfs(self, n):
        if n is None:
            return
        self.dfs(n.left)
        self.vals.append(n.val)
        self.dfs(n.right)

    def make_tree(self, lb, ub):
        if lb == ub:
            return None
        if ub - 1 == lb:
            return TreeNode(self.vals[lb])
        mid = (lb + ub) // 2
        r = TreeNode(self.vals[mid])
        r.left = self.make_tree(lb, mid)
        r.right = self.make_tree(mid + 1, ub)
        return r
    
    def balanceBST(self, root: TreeNode) -> TreeNode:
        self.vals = []
        self.dfs(root)
        return self.make_tree(0, len(self.vals))
