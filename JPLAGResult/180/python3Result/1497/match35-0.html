class CustomStack:

    def __init__(self, maxSize: int):
        self._maxSize = maxSize
        self._stk = []

    def push(self, x: int) -> None:
        if len(self._stk) < self._maxSize:
            self._stk.append(x)

    def pop(self) -> int:
        if len(self._stk) == 0:
            return -1
        return self._stk.pop()
        

    def increment(self, k: int, val: int) -> None:
        for i in range(0, min(k, len(self._stk))):
            self._stk[i] += val
