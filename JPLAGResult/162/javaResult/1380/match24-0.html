class Solution {
    public int closedIsland(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        
        boolean[][] visited = new boolean[m][n];
        
        int result = 0;
        
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == 0 && !visited[i][j]) {
                    result += dfs(grid, i, j, visited) ? 1 : 0;
                }
            }
        }
        
        return result;
    }
    
    private boolean dfs(int[][] grid, int i, int j, boolean[][] visited) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length) {
            return false;
        }
        
        if (grid[i][j] == 1 || visited[i][j]) {
            return true;
        }
        
        visited[i][j] = true;
        
        boolean up    = dfs(grid, i + 1, j, visited);
        boolean down  = dfs(grid, i - 1, j, visited);
        boolean right = dfs(grid, i, j + 1, visited);
        boolean left  = dfs(grid, i, j - 1, visited);
        
        return up && down && right && left;
    }
}
