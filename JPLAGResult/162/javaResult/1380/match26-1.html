class Solution {
    private int[][] dirs = new int[][]{{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
    public int closedIsland(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        boolean[][] marked = new boolean[m][n];
        int ans = 0;
        for (int i = 1; i < m - 1; ++i) {
            for (int j = 1; j < n - 1; ++j) {
                if (grid[i][j] == 0 && !marked[i][j]) {
                    if (!dfs(grid, i, j, marked)) ++ans;
                }
            
}
        }
        return ans;
    }
    
    private boolean dfs(int[][] grid, int i, int j, boolean[][] marked) {
        boolean ans = false;
        
if (i == 0 || i == grid.length - 1 || j == 0 || j == grid[0].length - 1) ans = true;
        marked[i][j] = true;
        for (int[] dir : dirs) {
            int nexti = i + dir[0], nextj = j + dir[1];
            if (nexti < 0 || nexti >= grid.length || nextj < 0 || nextj >= grid[0].length) continue;
            if (grid[nexti][nextj] == 0 && !marked[nexti][nextj]) {
                ans |= dfs(grid, nexti, nextj, marked);
            }
        }
        return ans;
    }
    
    
}
