class Solution {
    int[][] direction = new int[][]{{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    public int closedIsland(int[][] grid) {
        if (grid.length == 0 || grid[0].length == 0) return 0;
        int res = 0;
        boolean[][] visited = new boolean[grid.length][grid[0].length];
        for (int i = 0 ; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 0 && !visited[i][j]) {
                    int[] flag = new int[1];
                    helper(grid, i, j, visited, flag);
                    if (flag[0] == 0) res++;
                }
            }
        }
        return res;
    }
    
    private void helper(int[][] grid, int i, int j, boolean[][] visited, int[] flag) {
        if (i == 0 || j == 0 || i == grid.length - 1 || j == grid[0].length - 1) flag[0]++;
        visited[i][j] = true;
        for (int[] d : direction) {
            int x = i + d[0];
            int y = j + d[1];
            if (x < 0 || y < 0 || x >= grid.length || y >= grid[0].length || visited[x][y] || grid[x][y] == 1) {
                continue;
            }
            helper(grid, x, y, visited, flag);
        }
    }
}
