class Solution {
    int res = 0;
    int xlen;
    int ylen;
    public int closedIsland(int[][] grid) {
        xlen = grid.length;
        ylen = grid[0].length;
        
        
for(int i=0 ; i<xlen ; i++){
            dfs1(grid,i,0);
            dfs1(grid,i,ylen-1);
        }
        for(int i=0 ; i<ylen ; i++){
            dfs1(grid,0,i);
            dfs1(grid,xlen-1,i);
        }
        
        
for(int i=0 ; i<xlen ; i++){
            for(int j=0 ; j<ylen ; j++){
                if(grid[i][j]==0){
                    dfs1(grid,i,j);
                    ++res;
                }
            }
        }
        
        return res;
    }
    
    public void dfs1(int[][] grid,int x,int y){
        
if(x<0 || x>=xlen || y<0 || y>=ylen || grid[x][y]==1){
            return;
        }
        grid[x][y] = 1;
        dfs1(grid,x+1,y);
        dfs1(grid,x-1,y);
        dfs1(grid,x,y+1);
        dfs1(grid,x,y-1);        
    }
    
}
