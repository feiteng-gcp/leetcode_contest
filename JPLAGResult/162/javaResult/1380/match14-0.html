class Solution {
    public int closedIsland(int[][] grid) {
        for (int i = 0; i < grid.length; i++) {
            if (grid[i][0] == 0) {
                delete(grid, i, 0);
            }
            if (grid[i][grid[0].length - 1] == 0) {
                delete(grid, i, grid[0].length - 1);
            }
        }

        for (int j = 0; j < grid[0].length; j++) {
            if (grid[0][j] == 0) {
                delete(grid, 0, j);
            }
            if (grid[grid.length - 1][j] == 0) {
                delete(grid, grid.length - 1, j);
            }
        }

        int count = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[i].length; j++) {
                if (grid[i][j] == 0) {
                    count++;
                    delete(grid, i, j);
                }
            }
        }
        return count;
    }

    private void delete(int[][] grid, int i, int j) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[i].length || grid[i][j] != 0) {
            return;
        }
        grid[i][j] = 2;
        delete(grid, i + 1, j);
        delete(grid, i - 1, j);
        delete(grid, i, j + 1);
        delete(grid, i, j - 1);
    }
}
