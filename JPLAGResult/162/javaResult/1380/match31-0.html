class Solution {
    private static final int[] X = new int[] {1, -1, 0,  0};
    private static final int[] Y = new int[] {0,  0, 1, -1};
    
    private boolean valid(int[][] grid, int n, int m, int x, int y) {
        if (x < 0 || x >= n  || y < 0 || y >= m || grid[x][y] != 0) {
            return false;
        }
        
        return true;
    }
    
    private void helper(int[][] grid, int n, int m, int x, int y) {
        grid[x][y] = 3;
        
        for (int i= 0; i < 4; i++) {
            int dX = x + X[i];
            int dY = y + Y[i];
            if (valid(grid, n, m, dX, dY)) {
                helper(grid, n, m, dX, dY);
            }
        }
    }
    
    public int closedIsland(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        
        for (int i = 0; i < n; i++) {
            if (grid[i][0] == 0) {
                helper(grid, n, m, i, 0);
            }
            
            if (grid[i][m - 1] == 0) {
                helper(grid, n, m, i, m - 1);
            }
        }
        
        for (int j = 0; j < m; j++) {
            if (grid[0][j] == 0) {
                helper(grid, n, m, 0, j);
            }
            
            if (grid[n - 1][j] == 0) {
                helper(grid, n, m, n - 1, j);
            }
        }
        
        int count = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 0) {
                    count++;
                    helper(grid, n, m, i, j);
                }
            }
        }
        
        return count;
    }
}
