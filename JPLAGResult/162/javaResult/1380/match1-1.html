class Solution {
    public int closedIsland(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        boolean[][] mark = new boolean[n][m];
        int res = 0;
        for(int i = 0 ; i < n ; i++){
            for(int j = 0 ; j < m ; j++){
                if(!mark[i][j] && grid[i][j] == 0){
                    if(dfs(grid, mark, i, j)){
                        res++;
                    }
                }
            }
        }
        return res;
    }
    
    private int[][] dirs = new int[][]{{1, 0},{-1,0},{0,1},{0,-1}};
    
    private boolean dfs(int[][] grid, boolean[][] mark, int r, int c){
        if(r < 0 || r >= grid.length || c < 0 || c >= grid[0].length){
           return false;
        }
        if(mark[r][c] || grid[r][c] == 1){
            return true;
        }
        mark[r][c] = true;
        boolean isIsLand = true;
        for(int[] dir:dirs){
            int nr = r + dir[0];
            int nc = c + dir[1];
            isIsLand &= dfs(grid, mark, nr,nc);
        }
        return isIsLand;
    }
}
