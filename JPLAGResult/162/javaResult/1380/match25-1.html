class Solution {
    public int closedIsland(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        for (int i = 0; i < n; i++) {
            if (grid[0][i] == 0) {
                turnBad(grid, 0, i);
            }
            if (grid[m - 1][i] == 0) {
                turnBad(grid, m - 1, i);
            
}
        }
        for (int i = 0; i < m; i++) {
            if (grid[i][0] == 0) turnBad(grid, i, 0);
            if (grid[i][n - 1] == 0) turnBad(grid, i, n - 1);
        }
        int res = 0;
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                if (grid[i][j] == 0) {
                    visit(grid, i, j);
                    res++;
                }
            }
        }
        return res;
    }
    
    public void visit(int[][] grid, int x, int y) {
        int m = grid.length, n = grid[0].length;
        if (x < 0 || x >= m || y < 0 || y >= n || grid[x][y] == 1 || grid[x][y] == 3) return;
        grid[x][y] = 3;
        visit(grid, x + 1, y);
        visit(grid, x - 1, y);
        visit(grid, x, y + 1);
        visit(grid, x, y - 1);
    }
    
    
    public void turnBad(int[][] grid, int x, int y) {
        int m = grid.length, n = grid[0].length;
        
if (x < 0 || x >= m || y < 0 || y >= n || grid[x][y] == 1 || grid[x][y] == 2) return;
        grid[x][y] = 2;
        turnBad(grid, x + 1, y);
        turnBad(grid, x - 1, y);
        turnBad(grid, x, y + 1);
        turnBad(grid, x, y - 1);
    }
}
