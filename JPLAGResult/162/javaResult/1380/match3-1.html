class Solution {
    public int closedIsland(int[][] grid) {
        int[][] dir4 = {{1,0}, {-1,0}, {0,1}, {0,-1}};
        for(int i=0;i<grid.length;++i){
            dfs5257(grid,dir4, i,0,2);
        }
        for(int i=0;i<grid.length;++i){
            dfs5257(grid,dir4, i,grid[i].length-1,2);
        }
        for(int i=0;i<grid[0].length;++i){
            dfs5257(grid,dir4, 0,i,2);
        }
        for(int i=0;i<grid[0].length;++i){
            dfs5257(grid,dir4, grid.length-1,i,2);
        }
        int ans = 0;
        for(int i=0;i<grid.length;++i){
            for(int j=0;j<grid[0].length;++j){
                if(grid[i][j]==0){
                    ++ans;
                    dfs5257(grid,dir4,i,j,1);
                }
            }
        }
        return ans;
    }
        private void dfs5257(int [][] grid,int[][] dir4, int x,int y,int v){
        if(x<0||y<0||x>=grid.length||y>=grid[x].length||grid[x][y]!=0)
            return;
        grid[x][y]=v;
        for(int i=0;i<4;++i){
            int newx = x+dir4[i][0];
            int newy = y+dir4[i][1];
            dfs5257(grid,dir4,newx,newy,v);
        }
    }
}
