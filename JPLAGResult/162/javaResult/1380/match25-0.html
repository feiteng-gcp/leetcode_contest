class Solution {
    public int closedIsland(int[][] grid) {
        int a = grid.length, b = grid[0].length;
        for (int i = 0; i < a; i++) {
            if(grid[i][0] == 0) parse(grid, i, 0);
            if(grid[i][b - 1] == 0) parse(grid, i, b - 1);
        }
        for (int i = 0; i < b; i++) {
            if (grid[0][i] == 0) parse(grid, 0, i);
            if (grid[a - 1][i] == 0) parse(grid, a - 1, i);
        }
        int res = 0;
        for (int i = 0; i < a; i++) {
            for (int j = 0; j < b; j++) {
                if (grid[i][j] == 0) {
                    res++;
                    parse(grid, i, j);
                }
            }
        }
        return res;
    }
    public void parse(int[][] grid, int i, int j) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length) return;
        if (grid[i][j] == 1) return;
        grid[i][j] = 1;
        parse(grid, i + 1, j);
        parse(grid, i - 1, j);
        parse(grid, i, j + 1);
        parse(grid, i, j - 1);
    }
}
