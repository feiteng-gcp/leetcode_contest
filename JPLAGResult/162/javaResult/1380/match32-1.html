class Solution {
    
    boolean componentInside = false;
    public int closedIsland(int[][] grid) {
        
        int r = grid.length;
        int c = grid[0].length;
        boolean visited[][] = new boolean[r][c];
        componentInside = true;
        
        
int answer = 0;
        for( int i=1 ; i<r ; i++ ){
            for( int j=1 ; j<c ; j++ ){
                
                if( grid[i][j] == 0 && !visited[i][j] ){
                    componentInside = true;
                    solve( i,j,visited,grid, r, c );
                    if( componentInside ){
                        answer++;
                    
}
                }
                
            }
        }
        
        return answer;
        
    }
    
    void solve( int i, int j, boolean visited[][], int grid[][] , int r, int c ){
        
        if( i>=r || j>=c || i<0 || j<0 ){
            return;
        }
        
        if( grid[i][j] == 1 || visited[i][j] )
            return;
        
        if( i==0 || i==r-1 )
            componentInside = false
;
        if( j==0 || j==c-1 )
            componentInside = false;
        
        visited[i][j] = true;
        
        solve( i+1,j, visited, grid, r, c );
        solve( i,j+1, visited, grid, r, c );
        solve( i-1,j, visited, grid, r, c );
        solve( i,j-1, visited, grid, r, c );
        
        
    }
    
}
