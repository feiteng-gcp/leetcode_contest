class Solution {
    public int closedIsland(int[][] grid) {
        if (grid.length <= 2) {
            return 0;
        }
        
int len = grid.length;
        int rLen = grid[0].length;
        for (int i = 0; i < rLen; i++) {
            clean(grid, 0, i);
            clean(grid, len - 1, i);
        }
        for (int i = 0; i < len; i++) {
            clean(grid, i, 0);
            clean(grid, i, rLen - 1);
        }
        int ans = 0;
        for (int i = 1; i < len - 1; i++) {
            for (int j = 1; j < rLen - 1; j++) {
                if (grid[i][j] == 0) {
                    ans++;
                    clean(grid, i, j);
                }
            }
        }
        return ans;
    }

    private void clean(int[][] grid, int x, int y) {
        if (x < 0 || x >= grid.length) {
            return;
        }
        if (y < 0 || y >= grid[0].length) {
            return;
        }
        
if (grid[x][y] == 1) {
            return;
        }
        grid[x][y] = 1;
        clean(grid, x - 1, y);
        clean(grid, x + 1, y);
        clean(grid, x, y - 1);
        clean(grid, x, y + 1);
    }
}
