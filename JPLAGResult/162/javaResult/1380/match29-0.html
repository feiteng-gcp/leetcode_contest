class Solution {
    int ans = 0, n, m;
    int [][] dir = new int[][]{{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
    public int closedIsland(int[][] grid) {
        n = grid.length; m = grid[0].length;
        for(int i = 0; i < n ; i ++)
        {
            if(grid[i][0] == 0)
                dfs(grid, i, 0);
            if(grid[i][m - 1] == 0)
                dfs(grid, i, m - 1);
        }
        for(int j = 0; j < m ; j ++)
        {
            if(grid[0][j] == 0)
                dfs(grid, 0, j);
            if(grid[n - 1][j] == 0)
                dfs(grid, n - 1, j);
        }
        //for(int i = 0 ; i < n; i ++) 
         //   System.out.println(Arrays.toString(grid[i]));
        for(int i = 0 ; i < n; i ++)       
            for(int j = 0; j < m; j ++)
                if(grid[i][j] == 0)
                {
                    ans++;
                    dfs(grid, i, j);
                }
        return ans;
    }
    void dfs(int [][] grid, int x, int y)
    {
        grid[x][y] = 1;
        for(int i = 0; i < 4; i ++)
        {
            int xx = x + dir[i][0], yy = y + dir[i][1];
            if(0 <= xx && xx < n && 0 <= yy && yy < m && grid[xx][yy] == 0)
                dfs(grid, xx, yy);
        }
    }
