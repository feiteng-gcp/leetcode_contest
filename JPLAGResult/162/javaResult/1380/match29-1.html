class Solution {
    
int n=0,m=0;
    int[][] dirs = new int[][]{{1,0},{0,1},{-1,0},{0,-1}};
    public int closedIsland(int[][] grid) {
        n = grid.length;
        m = grid[0].length;
        for(int i=0;i<n;i++) {
            if(grid[i][m-1] == 0) floodfill(i,m-1,grid);
            if(grid[i][0] == 0) floodfill(i,0,grid);
        }
        for(int i=0;i<m;i++) {
            if(grid[n-1][i] == 0) floodfill(n-1,i,grid);
            if(grid[0][i] == 0) floodfill(0,i,grid);
        }
        int res = 0;
        
for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
                if(grid[i][j] == 0){
                    res++;
                    floodfill(i,j,grid);
                
            }
        return res;
                    
    }
    public void floodfill(int x, int y, int[][] grid){
        if(x<0 || x>=n || y<0||y>=m) return;
        if(grid[x][y] != 0) return;
        grid[x][y] = 2;
        for(int[] dir : dirs){
            int nx = dir[0]+x;
            int ny = dir[1]+y;
            floodfill(nx,ny,grid);
        }
    }
}
