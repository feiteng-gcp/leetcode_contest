class Solution {
    public int closedIsland(int[][] grid) {
        if (grid == null || grid.length == 0) {
            return 0;
        }
        
        for (int i = 0; i < grid[0].length; i++) {
            if (grid[0][i] == 0) {
                mark(grid, 0, i);
            }
            
            if (grid[grid.length - 1][i] == 0) {
                mark(grid, grid.length - 1, i);
            }
        }
        
        for (int i = 0; i < grid.length; i++) {
            if (grid[i][0] == 0) {
                mark(grid, i, 0);
            }
            
            if (grid[i][grid[0].length - 1] == 0) {
                mark(grid, i, grid[0].length - 1);
            }
        }
        
        int result = 0;
        
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 0) {
                    result++;
                    mark(grid, i, j);
                }
            }
        }
        
        return result;
    }
    
    private void mark(int[][] grid, int i, int j) {
        if (i < 0 || j < 0 || i > grid.length - 1 || j > grid[0].length - 1) {
            return;
        }
        
        if (grid[i][j] != 0) {
            return;
        }
        
        grid[i][j] = 2;
        
        mark(grid, i - 1, j);
        mark(grid, i, j - 1);
        mark(grid, i + 1, j);
        mark(grid, i, j + 1);
    }
}
