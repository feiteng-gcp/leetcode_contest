class Solution {
    
    boolean closed = true;
    
    public int closedIsland(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        //boolean[][] visited = new boolean[m][n];
        int count = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 0) {
                    closed = true;
                    dfs(grid, i, j);
                    if (closed) count++;    
                }

            }
        }
        return count;
    }
    
    private void dfs(int[][] grid, int i, int j) {
        if(i < 0 || j < 0 || i >= grid.length || j >= grid[0].length || grid[i][j] == 1) return;
        if (i == 0 || i == grid.length - 1 || j == 0 || j == grid[0].length - 1) closed = false;
        grid[i][j] = 1;
        dfs(grid, i - 1, j); //up
        dfs(grid, i + 1, j); //down
        dfs(grid, i, j - 1); //left
        dfs(grid, i, j + 1); //right
    }
}
