class Solution {
    int[][] grid;
    boolean[][] vis;
    boolean reachEdge;
    
    public int closedIsland(int[][] grid) {
        if (grid.length == 0 || grid[0].length == 0) {
            return 0;
        }
        this.grid = grid;
        this.vis = new boolean[grid.length][grid[0].length];
        
        
int ans = 0;
        for (int i = 0; i < grid.length; ++i) {
            for (int j = 0; j < grid[0].length; ++j) {
                if (grid[i][j] == 0 && !vis[i][j]) {
                    reachEdge = false;
                    bfs(i, j);
                    if (!reachEdge) {
                        ++ans;
                    }
                }
            }
        }
        return ans;
    }
    
    
int[][] dirs = new int[][] {
        new int[]{-1, 0},
        new int[]{ 1, 0},
        new int[]{ 0,-1},
        new int[]{ 0, 1},
    };
    
    private void bfs(
int i, int j) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length) {
            reachEdge = true;
            return;
        }
        
        if (vis[i][j] || grid[i][j] != 0) {
            return;
        }
        
        vis[i][j] = true;

        for (int[] dir : dirs) {
            int ii = i + dir[0];
            int jj = j + dir[1];
            bfs(ii, jj);
        }
    }
}
