class Solution {
    public int closedIsland(int[][] grid) {
		int n = grid.length;
		int m = grid[0].length;
        boolean[][] seen = new boolean[n][m];
        for (int j=0; j<m; j++) {
        	if (!seen[0][j] && grid[0][j]==0) {
        		dfs(seen, grid, 0, j);
        	}
        	if (!seen[n-1][j] && grid[n-1][j]==0) {
        		dfs(seen, grid, n-1, j);
        	}
        }
        for (int i=1; i<n; i++) {
        	if (!seen[i][0] && grid[i][0]==0) {
        		dfs(seen, grid, i, 0);
        	}
        	if (!seen[i][m-1] && grid[i][m-1]==0) {
        		dfs(seen, grid, i, m-1);
        	}
        }
        
        int cnt = 0;
        for (int i=1; i<n-1; i++) {
        	for (int j=1; j<m-1; j++) {
        		if (!seen[i][j] && grid[i][j]==0) {
        			dfs(seen, grid, i, j);
        			cnt++;
        		}
        	}
        }
        return cnt;
    }
	void dfs(boolean[][] seen, int[][] grid, int i, int j) {
		if (i<0 || j<0 || i>=grid.length || j>=grid[0].length)
			return;
		if (seen[i][j] || grid[i][j]==1)
			return;
		seen[i][j] = true;
		dfs(seen, grid, i+1, j);
		dfs(seen, grid, i-1, j);
		dfs(seen, grid, i, j+1);
		dfs(seen, grid, i, j-1);
	}
}
