class Solution {
    public int closedIsland(int[][] grid) {
        //first change outer 0 to 1
        for (int i = 0; i < grid.length; i++) {
            if (grid[i][0] == 0) {
                water(grid, i, 0);
            }
            if (grid[i][grid[0].length-1] == 0) {
                water(grid, i, grid[0].length-1);
            }
        }
        
        for (int j = 0; j < grid[0].length; j++) {
            if (grid[0][j] == 0) {
                water(grid, 0, j);
            }
            if (grid[grid.length-1][j] == 0) {
                water(grid, grid.length-1, j);
            }
        }
        
        //for (int i = 0; i< grid.length; i++) {System.out.println(Arrays.toString(grid[i]));}
        
        //find island
        int ans = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 0) {
                    dfs(grid, i, j);
                    ans++;
                }
            }
        }
        return ans;
        
    }
    
    private void dfs(int[][] grid, int i, int j) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length || grid[i][j] != 0)
            return;
        grid[i][j] = -1;
        water(grid, i-1, j);
        water(grid, i+1, j);
        water(grid, i, j-1);
        water(grid, i, j+1);
    }   
    
    
    private void water(int[][] grid, int i, int j) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length || grid[i][j] == 1)
            return;
        grid[i][j] = 1;
        water(grid, i-1, j);
        water(grid, i+1, j);
        water(grid, i, j-1);
        water(grid, i, j+1);
    }
}
