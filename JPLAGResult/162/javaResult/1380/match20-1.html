class Solution {
    public int closedIsland(int[][] grid) {
        int result = 0;
        
for (int i = 0; i < grid.length; i++) {
            if (grid[i][0] == 0) {
                changeOne(grid, i, 0);
            }
            if (grid[i][grid[0].length - 1] == 0) {
                changeOne(grid, i, grid[0].length - 1);
            }
        }
        
        for (int i = 0; i < grid[0].length; i++) {
            if (grid[0][i] == 0) {
                changeOne(grid, 0, i);
            }
            if (grid[grid.length - 1][i] == 0) {
                changeOne(grid, grid.length - 1, i);
            }
        }
        
        
for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 0) {
                    result++;
                    changeOne(grid, i, j);
                }
            }
        }
        
        return result;
        
    }
    
    private void changeOne(int[][] grid, int row, int col) {
        if (row < 0 || col < 0 || row >= grid.length || col >= grid[0].length || grid[row][col] == 1) {
            return;
        
}
        grid[row][col] = 1;
        
        changeOne(grid, row + 1, col);
        changeOne(grid, row - 1, col);
        changeOne(grid, row, col + 1);
        changeOne(grid, row, col - 1);
    }
}
