import java.util.ArrayList;
import java.util.List;


class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        int count = 0;
        int countOne = 0;
        for (int i : colsum) {
            count += i;
            if (i == 1) {
                countOne++;
            }
        }
        List<List<Integer>> result = new ArrayList<>();
        if (upper + lower != count) {
            return result;
        }
        int countTwo = (count - countOne) >> 1;
        upper -= countTwo;
        lower -= countTwo;
        
if (upper < 0 || lower < 0) {
            return result;
        }

        List<Integer> rowOne = new ArrayList<>();
        List<Integer> rowTwo = new ArrayList<>();
        result.add(rowOne);
        result.add(rowTwo);
        
for (int i = 0; i < colsum.length; i++) {
            if (colsum[i] == 0) {
                rowOne.add(0);
                rowTwo.add(0);
            } else if (colsum[i] == 1) {
                if (upper > 0) {
                    rowOne.add(1);
                    rowTwo.add(0);
                    upper--;
                } else {
                    rowOne.add(0);
                    rowTwo.add(1);
                }
            } else {
                rowOne.add(1);
                rowTwo.add(1);
            }
        }
        return result;
    }
}
