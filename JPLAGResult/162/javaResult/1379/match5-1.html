class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        int len = colsum.length;
        int[][] arr = new int[2][len];
        List<List<Integer>> ans = new ArrayList<>();
        for(int i=0;i<len;++i){
            if(colsum[i]==0)
                continue;
            if(upper+lower<colsum[i])
                return ans;
            if(colsum[i]==2){
                if(upper==0||lower==0)
                    return ans;
                --upper;
                --lower;
                arr[0][i]=1;
                arr[1][i]=1;
            } else if(upper>=lower){
                arr[0][i]=1;
                --upper;
            }else {
                arr[1][i]=1;
                --lower;
            }
        }
        if(upper>0||lower>0)
            return ans;
        List<Integer> ls1 = new ArrayList<>();
        List<Integer> ls2 = new ArrayList<>();
        for(int i=0;i<len;++i){
            ls1.add(arr[0][i]);
            ls2.add(arr[1][i]);
            
        }
        ans.add(ls1);
        ans.add(ls2);
        return ans;
    }
}
