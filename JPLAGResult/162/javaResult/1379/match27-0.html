class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        List<List<Integer>> res = new ArrayList<>(2);
        List<Integer> up = new ArrayList<>();
        List<Integer> low = new ArrayList<>();
        
        for(int i : colsum){
            if(i == 2){
                upper--;
                lower--;
                up.add(1);
                low.add(1);
            }
            else if(i == 0){
                up.add(0);
                low.add(0);
            }
            else {
                if(upper >= lower){
                    upper--;
                    up.add(1);
                    low.add(0);
                }
                else{
                    lower--;
                    up.add(0);
                    low.add(1);
                }
            }
            
            if(upper < 0 || lower < 0){
                return res;
            }
        }
        
        if(upper > 0 || lower > 0){
            return res;
        }
        
        res.add(up);
        res.add(low);
        return res;
    }
}
