class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        int solid = 0, flex = 0;
        for(int num : colsum){
            if(num == 1)
                flex++;
            else if(num == 2)
                solid++
;
        }
        if(solid > upper || solid > lower || solid * 2 + flex != upper + lower)
            return new ArrayList<>();
        List<List<Integer>> res = new ArrayList<>();
        res.add(new ArrayList<>());
        res.add(new ArrayList<>());
        upper -= solid;
        lower -= solid;
        for(int num : colsum){
            if(num == 2){
                res.get(0).add(1);
                res.get(1).add(1);
            }
            else if(num == 1){
                
if(upper != 0){
                    res.get(0).add(1);
                    res.get(1).add(0);
                    upper--;
                }
                else{
                    res.get(0).add(0);
                    res.get(1).add(1);
                    lower--;
                }
            }
            else{
                res.get(0).add(0);
                res.get(1).add(0);
            }
        }
        return res;
    }
}
