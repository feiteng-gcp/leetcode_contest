class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        int ones = 0, twos = 0;
        for (int num : colsum) {
            if (num == 1) ++ones;
            if (num == 2) ++twos;
        }
        if (upper - twos < 0 || lower - twos < 0 || upper - twos + lower - twos != ones) return new ArrayList<>();
        
        List<List<Integer>> ans = new ArrayList<>();
        ans.add(new ArrayList<>());
        ans.add(new ArrayList<>());
        for (int num : colsum) {
            if (num == 2) {
                ans.get(0).add(1);
                ans.get(1).add(1);
            } else if (num == 0) {
                ans.get(0).add(0);
                ans.get(1).add(0);
            } else {
                if (upper - twos > 0) {
                    ans.get(0).add(1);
                    ans.get(1).add(0);
                    --upper;
                } else {
                    ans.get(0).add(0);
                    ans.get(1).add(1);
                    --lower;
                }
            }
        }
        return ans;
    }
}
