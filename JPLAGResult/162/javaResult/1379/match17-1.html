class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        int sum = 0, count2 = 0;
        for (int cs : colsum) {
            sum += cs;
            count2 += cs/2;
        
}
        
        if (upper + lower != sum || upper < count2 || lower < count2) return new ArrayList();
        
        List<List<Integer>> matrix = new ArrayList();
        matrix.add(new ArrayList());
        matrix.add(new ArrayList());
        
        for (int i = 0; i < colsum.length; i++) {
            if (colsum[i] == 2) {
                matrix.get(0).add(1);
                matrix.get(1).add(1);
            } else if (colsum[i] == 1) {
                matrix.get(0).add(upper > lower ? 1 : 0);
                matrix.get(1).add(upper > lower ? 0 : 1);
            
} else {
                matrix.get(0).add(0);
                matrix.get(1).add(0);
            }
            upper -= matrix.get(0).get(i);
            lower -= matrix.get(1).get(i);
        }
        return matrix;
    }
}
