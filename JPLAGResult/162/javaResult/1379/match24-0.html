class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        LinkedList <List<Integer>> ans = new LinkedList();
        LinkedList <Integer> r0 = new LinkedList(), r1 = new LinkedList();
        int cnt2 = 0, sum = 0;
        for(int i : colsum)
        {
            if(i == 2)
                cnt2++;
            sum += i;
        }
        if(upper + lower != sum || lower < cnt2 || upper < cnt2)
            return ans;
        upper -= cnt2;
        lower -= cnt2;
        for(int i : colsum)
        {
            if(i == 2)
            {
                r0.add(1);
                r1.add(1);
            }
            else if(i == 1)
            {
                if(upper > 0)
                {
                    r0.add(1);
                    r1.add(0);
                    upper--;
                }
                else
                {
                    r1.add(1);
                    r0.add(0);
                    lower--;
                }
            }
            else
            {
                r0.add(0);
                r1.add(0);
            }
        }
        ans.add(r0);
        ans.add(r1);
        return ans;
    }
}
