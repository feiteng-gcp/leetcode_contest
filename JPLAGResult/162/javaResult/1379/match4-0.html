class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        List<List<Integer>> res = new ArrayList<>();
        int sum = 0;
        for (int i : colsum) {
            sum += i;
        }
        if (upper + lower != sum) return res;
        List<Integer> first = new ArrayList<>();
        List<Integer> second = new ArrayList<>();
        for (int i = 0; i < colsum.length; i++) {
            first.add(0);
            second.add(0);
        }
        for (int i = 0; i < colsum.length; i++) {
            if (colsum[i] == 2) {
                first.set(i, 1);
                second.set(i, 1);
                upper--;
                lower--;
            } 
            if (lower < 0 || upper < 0) return res;
        }
        for (int i = 0; i < colsum.length; i++) {
            if (colsum[i] == 1) {
                if (upper > 0) {
                    first.set(i, 1);
                    upper--;
                } else {
                    if (lower > 0) {
                    second.set(i, 1);
                    lower--;
                    } else {
                        return res;
                    }
                }
            }
        }
        res.add(first);
        res.add(second);
        return res;
    }
}
