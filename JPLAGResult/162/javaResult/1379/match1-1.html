class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        int totalSum = 0;
        for(int col:colsum){
            totalSum += col;
            if(col == 2){
                upper--;
                lower--;
                totalSum -= 2;
            }
        }
        if(upper <0 || lower <0 || upper + lower != totalSum){
            return new ArrayList<>();
        }
        List<Integer> upRow = new ArrayList<>();
        List<Integer> lowRow = new ArrayList<>();
        for(int col: colsum){
            if(col == 0){
                upRow.add(0);
                lowRow.add(0);
            } else if(col == 2){
                upRow.add(1);
                lowRow.add(1);
            } else {
                if(upper != 0){
                    upRow.add(1);
                    lowRow.add(0);
                    upper--;
                } else {
                    lowRow.add(1);
                    upRow.add(0);
                    lower--;
                }
            }
        }
        List<List<Integer>> res = new ArrayList<>();
        res.add(upRow);
        res.add(lowRow);
        return res;
    }
}
