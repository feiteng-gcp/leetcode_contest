class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        List<List<Integer>> res = new ArrayList<>();
        int count1 = 0, count2 = 0;
        for (int i = 0; i < colsum.length; i++) {
            if (colsum[i] == 1)
                count1++;
            if (colsum[i] == 2)
                count2++;
        }
        if (count2 > upper || count2 > lower || (upper - count2 + lower - count2) != count1)
            return res;
        List<Integer> up = new ArrayList<>();
        List<Integer> low = new ArrayList<>();
        int upTarget = upper - count2, lowTarget = lower - count2;
        
for (int i = 0; i < colsum.length; i++) {
            if (colsum[i] == 0) {
                up.add(0);
                low.add(0);
            } else if (colsum[i] == 2) {
                up.add(1);
                low.add(1);
            } else {
                if (upTarget > 0) {
                    up.add(1);
                    low.add(0);
                    upTarget--;
                } else {
                    low.add(1);
                    lowTarget--;
                    up.add(0);
                }
            }
        }
        res.add(up);
        res.add(low);
        return res;
    }
}
