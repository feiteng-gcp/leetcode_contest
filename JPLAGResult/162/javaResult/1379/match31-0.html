class Solution {
    
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        List<Integer> uList = new ArrayList<>();
        List<Integer> lList = new ArrayList<>();
        List<List<Integer>> res = new ArrayList<>();
        
        for (int i = 0; i < colsum.length; i++) {
            int val = colsum[i];
            if (val == 2) {
                if (upper > 0 && lower > 0) {
                    uList.add(1);
                    lList.add(1);
                } else {
                    return res;
                }
                
                upper--;
                lower--;
            } else if (val == 1) {
                if (upper >= lower && upper > 0) {
                    uList.add(1);
                    lList.add(0);
                    
                    upper--;
                } else if (lower >= upper && lower > 0) {
                    uList.add(0);
                    lList.add(1);
                    
                    lower--;
                } else {
                    return res;
                }
            } else {
                uList.add(0);
                lList.add(0);
            }
        }
        
        if (upper != 0 || lower != 0) return res;
        
        res.add(uList);
        res.add(lList);
        return res;
    }
}
