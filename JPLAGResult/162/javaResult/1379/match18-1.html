class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        
        List<List<Integer>> result = new ArrayList<>();
        
        int ar[][] = new int[colsum.length][2];
        
        int colsumreq = 0;
        for( int i=0 ; i<colsum.length ; i++ ){
            
            
if( colsum[i] == 2 ){
                ar[i][0] = 1;
                ar[i][1] = 1;
                upper--;
                lower--;
            } 
            else if( colsum[i] == 1 ){
                colsumreq++;
            }
            
        }
        if( upper < 0 || lower < 0 )
            return result;
        
        
if( colsumreq != ( upper+lower ) ){
            return result;
        }
        
        for( int i=0 ; i<colsum.length ; i++ ){
            
            if( colsum[i] == 1 ){
                
                if( upper > 0 ){
                    ar[i][0] = 1;
                    upper--;
                }
                else{
                    ar[i][1] = 1;
                    lower--;
                }
                
            }
            
        }
        result.add( new ArrayList<>() );
        result.add( new ArrayList<>() );
        
        for( int i=0 ; i<colsum.length ; i++ ){
            
            result.get(0).add( ar[i][0] );
            result.get(1).add( ar[i][1] );
            
        }
        return result;
        
    }
}
