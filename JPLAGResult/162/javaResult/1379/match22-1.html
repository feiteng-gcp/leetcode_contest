class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        List<List<Integer>> result = new ArrayList<>();
        
        int countU = 0;
        int countL = 0;
        
        int[][] arr = new int[2][colsum.length];
        
        for (int i = 0; i < colsum.length; i++) {
            if (colsum[i] == 2) {
                arr[0][i] = 1;
                arr[1][i] = 1;
                countU++;
                countL++;
            }
            
            if (countU > upper || countL > lower) {
                return result;
            }
        }
        
        //System.out.println(countU + "m" + countL);
        for (int i = 0; i < colsum.length; i++) {
            //System.out.println(i);
            if (colsum[i] == 1) 
{
                if (countU < upper) {
                    arr[0][i] = 1;
                    countU++;
                } else {
                    arr[1][i] = 1;
                    countL++;
                }
            }
            
            if (countU > upper || countL > lower) {
                return result;
            
}
        }
        //System.out.println(countU + "m" + countL);
        if (countU != upper || countL != lower) {
            return result;
        }
        
        List<Integer> l1 = new ArrayList<>();
        
        for (int i : arr[0]) {
            l1.add(i);
        }
        
        List<Integer> l2 = new ArrayList<>();
        
        for (int i : arr[1]) {
            l2.add(i);
        }
        
        result.add(l1);
        result.add(l2);
        
        return result;
    }
}
