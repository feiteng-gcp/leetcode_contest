class Solution {
    
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colSum) {
        int totSum = 0;
        int n = colSum.length;
        for(int i=0; i<n; i++) {
            totSum+=colSum[i];
        }
        if(totSum!=(upper+lower)) return new ArrayList<>();
        List<Integer> top = new ArrayList<>();
        List<Integer> bot = new ArrayList<>();
        int topSum = upper;
        
int botSum = lower;
        for(int i=0; i<n; i++) {
            if(colSum[i]==2) {
                top.add(1);
                topSum--;
                bot.add(1);
                botSum--;
            } else if(colSum[i]==1) 
{
                if(topSum>botSum) {
                    top.add(1);
                    topSum--;
                    bot.add(0);
                } else {
                    bot.add(1);
                    botSum--;
                    top.add(0);
                }
            } else {
                top.add(0);
                bot.add(0);
            }
            
            if(topSum<0 || botSum<0) return new ArrayList<>()
;
        }
        List<List<Integer>> res= new ArrayList<>();
        res.add(top);
        res.add(bot);
        return res;
    }
}
