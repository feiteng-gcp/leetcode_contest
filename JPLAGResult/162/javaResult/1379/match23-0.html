class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        List<List<Integer>> res = new ArrayList<>();
        int sum = upper + lower;
        int sum2 = 0;
        int num2 = 0;
        for(int i = 0; i < colsum.length; i++){
            if(colsum[i] < 0 || colsum[i] > 2) return res;
            sum2+= colsum[i];
            if(colsum[i] == 2)num2 ++;
        }
        if(sum != sum2)return res;
        if(num2 > upper || num2 > lower)return res;
        int c1 = upper - num2, c2 = lower - num2;
        List<Integer> r1 = new ArrayList<>(), r2 = new ArrayList<>();
        for(int i = 0; i < colsum.length; i++){
            if(colsum[i] == 2){
                r1.add(1);
                r2.add(1);
            } else if (colsum[i] == 0){
                r1.add(0);
                r2.add(0);
            } else {
                if(c1 > 0){
                                    r1.add(1);
                r2.add(0);
                c1--;
                
                } else {
                                   r1.add(0);
                r2.add(1);
                
                c2--; 
                }
            }
        }
        res.add(r1);
        res.add(r2);
        return res;
    }
}
