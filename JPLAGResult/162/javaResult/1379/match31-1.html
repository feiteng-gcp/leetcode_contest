class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum)
    {
        List<List<Integer> > ans = new ArrayList<>();
        int tmp = 0;
        for(int i = 0; i < colsum.length;i++) tmp += colsum[i];
        if(tmp  != (upper + lower)) return ans;

        List<Integer> a = new ArrayList<>();
        List<Integer> b = new ArrayList<>();

        for(int i = 0;i < colsum.length;i++)
        
{
            if(upper >= colsum[i])
            {
                a.add(colsum[i]);
                b.add(0);
                upper -= colsum[i];
            }
            else if(lower >= colsum[i])
            {
                a.add(0);
                b.add(colsum[i]);
                lower -= colsum[i];
            }
            else
            {
                return ans;
            }
        
}

        if(upper != 0 || lower != 0) return ans;

        ans.add(a);
        ans.add(b);
        return  ans;
    }
}
