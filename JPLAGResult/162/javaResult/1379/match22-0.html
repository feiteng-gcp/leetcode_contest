class Solution {
    
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        List<List<Integer>> res = new ArrayList<>();
        
        int n = colsum.length;
        int[][] arr = new int[2][n];
        
        for(int j = 0; j < n; j++) {
            if(upper < 0 || lower < 0) {
                return res;
            }
            if(colsum[j] == 2 || colsum[j] == 0) {
                arr[0][j] = colsum[j] / 2;
                arr[1][j] = colsum[j] / 2;
                upper -= arr[0][j];
                lower -= arr[1][j];
            } else {
                if(upper > lower) {
                    upper--;
                    arr[0][j] = 1;
                } else {
                    lower--;
                    arr[1][j] = 1;
                }
            }
        }
        
        if(upper != 0 || lower != 0) {
            return res;
        }
        
        List<Integer> l1 = new ArrayList<>();
        for(int i : arr[0]) {
            l1.add(i);
        }
        List<Integer> l2 =  new ArrayList<>();
        for(int i : arr[1]) {
            l2.add(i);
        }
        res.add(l1);
        res.add(l2);
        
        return res;
    }

}
