class Solution {
    private int pushBit(List<Integer> list, int bit) {
        list.add(bit);
        return bit;
    }
    
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        List<List<Integer>> result = new ArrayList<List<Integer>>();
        List<Integer> upperList = new ArrayList<Integer>();
        List<Integer> lowerList = new ArrayList<Integer>();
        for (int i : colsum) {
            if (i == 2) {
                upper -= pushBit(upperList, 1);
                lower -= pushBit(lowerList, 1);
            } else if (i == 1) {
                if (upper >= lower) {
                    upper -= pushBit(upperList, 1);
                    lower -= pushBit(lowerList, 0);
                } else {
                    upper -= pushBit(upperList, 0);
                    lower -= pushBit(lowerList, 1);
                }
            } else {
                upper -= pushBit(upperList, 0);
                lower -= pushBit(lowerList, 0);
            }
        }
        if ((upper | lower) == 0) {
            result.add(upperList);
            result.add(lowerList);
        }
        return result;
    }
}
