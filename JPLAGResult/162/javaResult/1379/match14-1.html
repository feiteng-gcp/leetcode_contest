class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        List<List<Integer>> result=new ArrayList<>();
        result.add(new ArrayList<>());
        result.add(new ArrayList<>());
        
for (int col : colsum) {
            if (col == 0) {
                result.get(0).add(0);
                result.get(1).add(0);
            } else if (col == 2) {
                if (upper == 0 || lower == 0) {
                    return new ArrayList<>();
                }
                
result.get(0).add(1);
                upper--;
                result.get(1).add(1);
                lower--;
            } else {
                if (upper > lower) {
                    upper--;
                    result.get(0).add(1);
                    result.get(1).add(0);
                } else {
                    if (lower == 0) {
                        return new ArrayList<>();
                    }
                    lower--;
                    
result.get(1).add(1);
                    result.get(0).add(0);
                }
            }
        }
        if (lower!=0 || upper!=0){
            return new ArrayList<>();
        }
        return result;
    }
}
