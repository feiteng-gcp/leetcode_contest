class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        int all = 0;
        int num2 = 0;
        for (int i = 0; i < colsum.length; i++) {
            if (colsum[i] == 2) {
                num2++;
            }
            all += colsum[i];
        }
        List<List<Integer>> ans = new ArrayList<>();
        if (all != upper + lower) {
            return ans;
        }
        if (num2 > upper || num2 > lower) {
            return ans;
        
}
        List<Integer> upperList = new ArrayList<>();
        List<Integer> lowerList = new ArrayList<>();
        int key = upper - num2;
        for (int i = 0; i < colsum.length; i++) {
            if (colsum[i] == 2) {
                upperList.add(1);
                lowerList.add(1);
            } else if (colsum[i] == 1) {
                
if (key > 0) {
                    upperList.add(1);
                    lowerList.add(0);
                    key--;
                } else {
                    upperList.add(0);
                    lowerList.add(1);
                }
            } else {
                upperList.add(0);
                lowerList.add(0);
            }

        }
        ans.add(upperList);
        ans.add(lowerList);
        return ans;
    }
}
