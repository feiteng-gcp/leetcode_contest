class Solution {
    // 1 1 0 1
    // 1 1 1 0
    // 
public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
            List<List<Integer>> res = new ArrayList<>();
            int n = colsum.length;
            List<Integer> row0 = new ArrayList<Integer>();
            List<Integer> row1 = new ArrayList<Integer>();
            for (int i = 0; i < n; i++) {
                if (colsum[i] == 2) {
                    upper--;
                    lower--;
                }
            }

            
for (int i = 0; i < n; i++) {
                if (colsum[i] == 2) {
                    row0.add(1);
                    row1.add(1);
                } else if (colsum[i] == 1) 
{
                    if (upper > 0) {
                        row0.add(1);
                        row1.add(0);
                        upper--;
                    } else {
                        row0.add(0);
                        row1.add(1);
                        lower--;
                    }
                } else {
                    row0.add(0);
                    row1.add(0);
                
}

                if (upper < 0 || lower < 0) {
                    return res;
                }
            }
            if (upper == 0 && lower == 0) {
                res.add(row0);
                res.add(row1);
            }
            return res;
        }
}
