class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        int sum = 0;
        int count2 = 0;
        for (int num : colsum) {
            sum += num;
            if (num == 2) {
                count2++;
            }
        }
        
        if (upper + lower != sum || sum > 2 * colsum.length || sum < 0 || count2 > upper || count2 > lower) {
            return new ArrayList<>();
        }
        
        
        List<Integer> up = new ArrayList<>();
        List<Integer> low = new ArrayList<>();
        int up1 = upper - count2;
        
        //i choose to fill in the up
        for (int i = 0; i < colsum.length; i++) {
            if (colsum[i] == 2) {
                up.add(1);
                low.add(1);
            }
            else if (colsum[i] == 0) {
                up.add(0);
                low.add(0);
            }
            else {
                if (up1 > 0) {
                    up.add(1);
                    up1--;
                    low.add(0);
                }
                else {
                    up.add(0);
                    low.add(1);
                }
            }
        }
        List<List<Integer>> res = new ArrayList<>();
        res.add(up);
        res.add(low);
        return res;
    }
}
