class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        List<List<Integer>> ret = new ArrayList<List<Integer>>();
        int totalSum = upper + lower;
        int colSum = 0;
        for (int num : colsum)
            colSum += num;
        if (totalSum != colSum)
            return ret;
        int columns = colsum.length;
        int[][] matrix = new int[2][columns];
        int sum0 = 0, sum1 = 0;
        for (int i = 0; i < columns; i++) {
            if (colsum[i] == 2) {
                matrix[0][i] = 1;
                matrix[1][i] = 1;
                sum0++;
                sum1++;
            }
        }
        for (int i = 0; i < columns; i++) {
            if (colsum[i] == 1) {
                if (sum0 < upper) {
                    matrix[0][i] = 1;
                    sum0++;
                } else {
                    matrix[1][i] = 1;
                    sum1++;
                }
            }
        }
        if (sum0 != upper || sum1 != lower)
            return ret;
        for (int i = 0; i < 2; i++) {
            int[] row = matrix[i];
            List<Integer> rowList = new ArrayList<Integer>();
            for (int j = 0; j < columns; j++)
                rowList.add(row[j]);
            ret.add(rowList);
        }
        return ret;
    }
}
