class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
        int num2 = 0;
        
        for(int i : colsum){
            if(i==2){
                ++num2;
            }
        }
        int len = colsum.length;
        upper = upper - num2;
        lower = lower - num2;
        
if(upper<0 || lower < 0){
            return new ArrayList<>();
        }
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> u = new ArrayList<>();
        List<Integer> l = new ArrayList<>();
        res.add(u);
        res.add(l);
        
for(int i=0 ; i<len ; i++){
            if(colsum[i]==2){
                u.add(1);
                l.add(1);
            } else if(colsum[i]==0){
                u.add(0);
                l.add(0);
            } else{
                if(upper>0){
                    u.add(1);
                    l.add(0);
                    --upper;
                } else{
                    if(lower<=0){
                        return new ArrayList<>();
                    }
                    
l.add(1);
                    u.add(0);
                    --lower;
                }
            }
        }
        if(upper>0 || lower>0){
            return new ArrayList<>();
        } else{
            return res;
        }
    }
}
