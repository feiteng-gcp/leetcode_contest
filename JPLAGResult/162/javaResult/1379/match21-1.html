class Solution {
        public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
            List<List<Integer>> res = new ArrayList<>(2);
            List<Integer> lowerArr = new ArrayList<>(colsum.length);
            List<Integer> upperArr = new ArrayList<>(colsum.length);

            for (int i : colsum) {
                
if (i == 0) {
                    lowerArr.add(0);
                    upperArr.add(0);
                } else if (i == 1) {
                    if (lower > upper) {
                        --lower;
                        lowerArr.add(1);
                        upperArr.add(0);
                    } else {
                        --upper;
                        lowerArr.add(0);
                        upperArr.add(1);
                    }
                } else {
                    lowerArr.add(1);
                    upperArr.add(1);
                    --upper;
                    --lower;
                }
                if (lower < 0 || upper < 0) {
                    return new LinkedList<>();
                }
            }
            if (lower > 0 || upper > 0) {
                return new LinkedList<>();
            }

            res.add(upperArr);
            res.add(lowerArr);
            return res;
        }
    }
