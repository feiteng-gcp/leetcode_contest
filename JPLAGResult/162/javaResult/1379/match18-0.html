class Solution {
    public List<List<Integer>> reconstructMatrix(int upper, int lower, int[] colsum) {
       List<List<Integer>> list = new ArrayList<>();
        Integer[] upperArray = new Integer[colsum.length];
        Integer[] lowerArray = new Integer[colsum.length];
        int eitherCount = 0;
        for (int colIdx = 0; colIdx < colsum.length; colIdx++) {
            upperArray[colIdx] = 0;
            lowerArray[colIdx] = 0;
            if (colsum[colIdx] == 2) {
                upperArray[colIdx] = 1;
                lowerArray[colIdx] = 1;
                upper--;
                lower--;
            } else if (colsum[colIdx] == 1) {
                eitherCount++;
            }
        }
        if (upper < 0 || lower < 0 || (upper + lower) != eitherCount) {
            return list;
        }
        for (int colIdx = 0; colIdx < colsum.length; colIdx++) {
            if (colsum[colIdx] == 1) {
                if (upper > 0) {
                    upperArray[colIdx] = 1;
                    upper--;
                } else {
                    lowerArray[colIdx] = 1;
                    lower--;
                }
            }
        }
        list.add(Arrays.asList(upperArray));
        list.add(Arrays.asList(lowerArray));
        return list; 
    }
}
