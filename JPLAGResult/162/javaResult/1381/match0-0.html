class Solution {
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[][] wordsLettersCount = wordsLettersCount(words);
        int[] availableLettersCount = availableLettersCount(letters);
        List<List<Integer>> availableCombinations = availableCombinations(wordsLettersCount, availableLettersCount);
        int maxScore = 0;
        for (List<Integer> availableCombination : availableCombinations) {
            int curScore = 0;
            for (int index : availableCombination) {
                String word = words[index];
                int wordScore = getScore(word, score);
                curScore += wordScore;
            }
            maxScore = Math.max(maxScore, curScore);
        }
        return maxScore;
    }

    public int[][] wordsLettersCount(String[] words) {
        int length = words.length;
        int[][] wordsLettersCount = new int[length][26];
        for (int i = 0; i < length; i++) {
            String word = words[i];
            char[] array = word.toCharArray();
            for (char c : array)
                wordsLettersCount[i][c - 'a']++;
        }
        return wordsLettersCount;
    }

    public int[] availableLettersCount(char[] letters) {
        int[] availableLettersCount = new int[26];
        for (char c : letters)
            availableLettersCount[c - 'a']++;
        return availableLettersCount;
    }

    public List<List<Integer>> availableCombinations(int[][] wordsLettersCount, int[] availableLettersCount) {
        int length = wordsLettersCount.length;
        int maxCount = (int) Math.pow(2, length);
        List<List<Integer>> availableCombinations = new ArrayList<List<Integer>>();
        for (int i = 0; i < maxCount; i++) {
            int[] array = indexToArray(i, length);
            List<Integer> combination = new ArrayList<Integer>();
            for (int j = 0; j < length; j++) {
                if (array[j] == 1)
                    combination.add(j);
            }
            int[] actualLettersCount = new int[26];
            for (int index : combination) {
                int[] wordLettersCount = wordsLettersCount[index];
                for (int j = 0; j < 26; j++)
                    actualLettersCount[j] += wordLettersCount[j];
            }
            boolean flag = true;
            for (int j = 0; j < 26; j++) {
                if (actualLettersCount[j] > availableLettersCount[j]) {
                    flag = false;
                    break;
                }
            }
            if (flag)
                availableCombinations.add(combination);
        }
        return availableCombinations;
    }

    public int[] indexToArray(int index, int length) {
        int[] array = new int[length];
        int position = 0;
        while (index > 0) {
            int remainder = index % 2;
            array[position++] = remainder;
            index /= 2;
        }
        return array;
    }

    public int getScore(String word, int[] score) {
        int wordScore = 0;
        char[] array = word.toCharArray();
        for (char c : array)
            wordScore += score[c - 'a'];
        return wordScore;
    }
}
