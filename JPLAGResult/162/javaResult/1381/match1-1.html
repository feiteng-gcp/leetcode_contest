class Solution {

  public int maxScoreWords(String[] words, char[] letters, int[] score) {
    int res = 0;
    int n = 1 << words.length;
    int[] maxCnt = new int[26];
    for (char letter : letters) {
      maxCnt[letter - 'a']++;
    }
    int[][] dp = new int[n][26];
    int[] maxScore = new int[n];
    for (int i = 1; i < n; i++) {
      int pre = i & (i - 1);
      int index = 0;
      int tmp = pre ^ i;
      while ((tmp & 1) != 1) {
        tmp >>= 1;
        index++;
      }
      if (maxScore[pre] == -1) {
        maxScore[i] = -1;
      } else {
        System.arraycopy(dp[pre], 0, dp[i], 0, 26);
        int sum = 0;
        for (int j = 0; j < words[index].length(); j++) {
          dp[i][words[index].charAt(j) - 'a']++;
          sum += score[words[index].charAt(j) - 'a'];
        }
        if (!isValid(dp[i], maxCnt)) {
          maxScore[i] = -1;
        } else {
          maxScore[i] = maxScore[pre] + sum;
          res = Math.max(res, maxScore[i]);
        }
      }
    }
    return res;
  }

  private boolean isValid(int[] cnt, int[] maxCnt) {
    for (int i = 0; i < 26; i++) {
      if (cnt[i] > maxCnt[i]) {
        return false;
      }
    }
    return true;
  }
}
