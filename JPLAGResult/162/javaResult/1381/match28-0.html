class Solution {
public int maxScoreWords(String[] words, char[] letters, int[] score) {
            int n = words.length;
            int[] countLetter = new int[26];
            for (int i = 0; i < letters.length; i++) {
                countLetter[letters[i] - 'a']++;
            }

            int res = 0;
            for (int i = 0; i < (1 << n); i++) {
                int[] count = new int[26];
                for (int j = 0; j < n; j++) {
                    if (bit(i, j) == 1) {
                        for (int k = 0; k < words[j].length(); k++) {
                            int ch = words[j].charAt(k) - 'a';
                            count[ch]++;
                        }
                    }
                }

                int s = 0;
                for (int j = 0; j < 26; j++) {
                    if (count[j] > countLetter[j]) {
                        s = 0;
                        break;
                    } else {
                        s += (count[j] * score[j]);
                    }
                }
                res = Math.max(res, s);
            }
            return res;
        }

        public int bit(int i, int j) {
            return (i >> j) & 1;
        }
}
