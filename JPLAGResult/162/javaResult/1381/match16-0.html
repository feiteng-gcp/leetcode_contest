class Solution {
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] count = new int[26];
        for (char c: letters)
            count[c - 'a']++;
        return backTrack(words, 0, count, score);
    }
    
    private int backTrack(String[] words, int idx, int[] count, int[] score) {
        if (idx == words.length) return 0;
        int skip = backTrack(words, idx+1, count, score);
        
        char[] ch = words[idx].toCharArray();
        int breakPoint = -1;
        int thisWord = 0;
        for (int i=0; i<ch.length; ++i) {
            char c = ch[i];
            if (--count[c-'a'] >= 0) {
                thisWord += score[c-'a'];
            } else {
                breakPoint = i;
                break;
            }
        }
        
        if (breakPoint < 0) {
            skip = Math.max(skip, thisWord + backTrack(words, idx+1, count, score));
            breakPoint = ch.length-1;
        }
        
        for (int i=0; i<=breakPoint; ++i) {
            char c = ch[i];
            ++count[c-'a'];
        }
        return skip;
    }
}
