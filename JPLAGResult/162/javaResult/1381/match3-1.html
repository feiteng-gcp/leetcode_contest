class Solution {
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] fq = new int[26];
        for(char c:letters){
            fq[c-'a']++;
        }
        int[] maxScore = new int[1];
        dfs5258(words,fq,score,0,maxScore,0);
        return maxScore[0];
    }
    private void dfs5258(String[] words, int[] fq, int[] score, int curScore, int[] maxScore, int idx){
        if(idx>=words.length)
            return;
        char[] arr = words[idx].toCharArray();
        int[] fqArr = new int[26];
        for(char c:arr){
            fqArr[c-'a']++;
        }
        for(int i=0;i<26;++i){
            if(fqArr[i]>fq[i]){
                dfs5258(words,fq,score,curScore,maxScore,idx+1);
                return;
            }
        }
        int newScore = curScore;
        for(int i=0;i<26;++i){
            fq[i]-=fqArr[i];
            newScore+=score[i]*fqArr[i];
        }
        maxScore[0] = Math.max(maxScore[0],newScore);
        dfs5258(words,fq,score,newScore,maxScore,idx+1);
        for(int i=0;i<26;++i){
            fq[i]+=fqArr[i];
        }
        dfs5258(words,fq,score,curScore,maxScore,idx+1);
    }
}
