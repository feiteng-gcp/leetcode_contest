class Solution {
    private int ans = 0;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] cnt = new int[26];
        for (char c : letters) 
            cnt[c - 'a']++;
        backtracking(words, 0, 0, cnt, score, new boolean[words.length]);
        return ans;
    }
    
    private void backtracking(String[] words, int start, int sum, int[] cnt, int[] score, boolean[] marked) {
        ans = Math.max(ans, sum);
        for (int i = start; i < words.length; ++i) {
            if (marked[i]) continue;
            if (valid(words[i], cnt)) {
                int tmp = 0;
                for (char c : words[i].toCharArray()) {
                    cnt[c - 'a']--;
                    tmp += score[c - 'a'];
                }
                marked[i] = true;
                backtracking(words, i + 1, sum + tmp, cnt, score, marked);
                marked[i] = false;
                for (char c : words[i].toCharArray())
                    cnt[c - 'a']++;
            }
        }
    }
    
    private boolean valid(String word, int[] cnt) {
        int[] aux = new int[26];
        for (char c : word.toCharArray()) {
            aux[c - 'a']++;
            if (cnt[c - 'a'] < aux[c - 'a']) return false;
        }
        return true;
    }
}
