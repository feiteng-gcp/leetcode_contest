class Solution {
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] cnt = new int[26];
        for(char c:letters) cnt[c - 'a']++;
        return help(words,cnt,score,0);
    }
    
    int help(String[] words, int[] cnt, int[] score, int index) {
        if(index == words.length) return 0;
        int[] local = new int[26];
        for(char c : words[index].toCharArray()) {
            local[c-'a']++;
        }
        int res = help(words,cnt,score,index+1);
        int s = 0;
        int[] next = new int[26];
        for(int i = 0 ; i < 26;i++) {
            if(cnt[i] >= local[i]) {
                next[i] = cnt[i] - local[i];
                s += local[i] * score[i];
            
}
            else {
                next = null;
                break;
            }
        }
        if(next != null) res = Math.max(res, help(words,next,score,index+1) + s);
        return res;
    }
}
