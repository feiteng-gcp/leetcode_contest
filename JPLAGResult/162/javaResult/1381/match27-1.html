class Solution {
    
public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int n = words.length;
        int[] letNum = new int[26];
        for (int i = 0; i < letters.length; i++) {
            letNum[letters[i] - 'a']++;
        }
        int ans = 0;
        int[] nLetNum = new int[26];
        int[][] wordsNum = 
new int[words.length][26];
        int[] res = new int[words.length];
        for (int i = 0; i < words.length; i++) {
            for (int j = 0; j < words[i].length(); j++) {
                wordsNum[i][words[i].charAt(j) - 'a']++;
            }
            for (int j = 0; j < 26; j++) {
                res[i] += score[j] * wordsNum[i][j];
            }
        }
        for (int i = 1; i < (1 << n); i++) {
            copy(letNum, nLetNum);
            int tAns = 0;
            int key = 0;
            int k = i;
            while (k > 0) {
                if (k % 2 == 1) {
                    if (check(wordsNum[key], nLetNum)) {
                        tAns += res[key];
                    } else {
                        break;
                    }
                }
                k /= 2;
                key++;
            }
            ans = Math.max(ans, tAns);
        }
        return ans;
    }

    
private boolean check(int[] wordsNum, int[] nLetNum) {
        for (int i = 0; i < 26; i++) {
            if (wordsNum[i] > nLetNum[i]) {
                return false;
            }
            nLetNum[i] -= wordsNum[i];
        }
        return true;
    }

    private void copy(int[] letNum, int[] nLetNum) {
        for (int i = 0; i < 26; i++) {
            nLetNum[i] = letNum[i];
        }
    }
}
