class Solution {
    int res = 0;

    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] letterCount = new int[26];
        for (int i = 0; i < letters.length; i++)
            letterCount[letters[i] - 'a']++
;
        dfs(words, 0, letterCount, score, 0);
        return res;
    }

    private void dfs(String[] words, int point, int[] lettersCount, int[] score, int sum) {
        if (point == words.length) {
            if (sum > res)
                res = sum;
            return;
        }
        int[] tempCount = new int[26];
        for (int i = 0; i < words[point].length(); i++)
            tempCount[words[point].charAt(i) - 'a']++;
        boolean isOK = true;
        for (int i = 0; i < 26; i++)
            if (tempCount[i] > lettersCount[i]) {
                isOK = false;
                break;
            }
        if (isOK) {
            for (int i = 0; i < 26; i++) {
                sum += score[i] * tempCount[i];
                lettersCount[i] -= tempCount[i];
            }
            dfs(words, point + 1, lettersCount, score, sum);
            for (int i = 0; i < 26; i++) {
                sum -= score[i] * tempCount[i];
                lettersCount[i] += tempCount[i];
            }
        }
        dfs(words, point + 1, lettersCount, score, sum);
    }
}
