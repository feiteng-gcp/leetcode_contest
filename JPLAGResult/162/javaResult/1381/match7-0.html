class Solution {
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int fre[]=new int[26];
        for (int i = 0; i <letters.length ; i++) {
            fre[letters[i]-'a']++;
        }
        int n=words.length;
        int ans=0;
        for (int i = 0; i <1<<n ; i++) {
            int CloneFre[]=Arrays.copyOf(fre,fre.length);
            int temp=0;
            for (int j = 0; j <n ; j++) {
                if((i&(1<<j))>0){
                    int m=words[j].length();
                    for (int k = 0; k <m ; k++) {
                        int ind=words[j].charAt(k)-'a';
                        if(CloneFre[ind]>0){
                            temp+=score[ind];
                            CloneFre[ind]--;
                        }
                        else{
                            temp=0;
                            break;
                        }
                    }  
                }
            }
            ans=Math.max(ans,temp);
        }
        return ans;
    }
}
