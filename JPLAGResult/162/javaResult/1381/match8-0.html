class Solution {
    private int maxScore = 0;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        Map<Character, Integer> count = new HashMap<>();
        
        for (char c : letters) {
            count.put(c, count.getOrDefault(c, 0) + 1);
        }
        
        dfs(words, count, score, 0, 0);
        
        return maxScore;
    }
    
    private void dfs(String[] words, Map<Character, Integer> count, int[] score, int curScore, int index) {
        maxScore = Math.max(maxScore, curScore);
        
        for (int i = index; i < words.length; ++i) {
            if (isValid(count, words[i])) {
                int wordScore = 0;

                for (char c : words[i].toCharArray()) {
                    count.put(c, count.get(c) - 1);
                    wordScore += score[c - 'a'];
                }
                
                dfs(words, count, score, curScore + wordScore, i + 1);
                
                for (char c : words[i].toCharArray()) {
                    count.put(c, count.get(c) + 1);
                }
            }
        }
    }
    
    private boolean isValid(Map<Character, Integer> count, String word) {
        int[] hash = new int[26];
        
        for (char c : word.toCharArray()) {
            hash[c - 'a']++;
        }
        
        for (char c : word.toCharArray()) {
            if (count.getOrDefault(c, 0) < hash[c - 'a']) {
                return false;
            }
        }
        
        return true;
    }
}
