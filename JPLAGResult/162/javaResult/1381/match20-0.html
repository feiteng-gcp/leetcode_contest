class Solution {
    int res = 0;
    public  int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] letterNum = new int[26];
        for(char l : letters) {
            letterNum[l - 'a']++;
        }

        List<String> word = new ArrayList<>();

        recursion(words, word, letterNum, score, 0);

        return res;
    }
    
    private  void recursion(String[] words, List<String> wordList, int[] letterNum, int[] score, int index) {
        if(index == words.length) return;
        
        wordList.add(words[index]);
        int val = helper(wordList, letterNum, score);
        res = Math.max(res, val);
        recursion(words, wordList, letterNum, score, index + 1);
        wordList.remove(words[index]);
        recursion(words, wordList, letterNum, score, index + 1);
    }

    private  int helper(List<String> wordList, int[] letterNum, int[] score) {
        if(wordList.size() == 0) return 0;
        int[] wordLetter = new int[26];
        
        for(String word : wordList) {
            for(int i = 0 ; i < word.length(); i++) {
                wordLetter[word.charAt(i) - 'a']++;
            }    
        }

        int res = 0;
        for(int i = 0; i < wordLetter.length; i++) {
            if(wordLetter[i] > letterNum[i]) {
                return 0;
            }
            res += score[i] * wordLetter[i];
        }
        return  res;
    }
}
