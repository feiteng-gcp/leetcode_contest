class Solution {
    int[] s;
   
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        Map<Character, Integer> freq = new HashMap<>();
       
        for(char c : letters){
            freq.put(c, freq.getOrDefault(c, 0) + 1);
        }
       
        s = score;
       
        return dfs(words, 0, freq);
    }
   
    public int dfs(
String[] words, int index, Map<Character, Integer> freq){
        if(index == words.length){
            return 0;
        }
       
        Map<Character, Integer> curr_freq = new HashMap<>();
       
        
for(char c : freq.keySet()){
            curr_freq.put(c, freq.get(c));
        }

        int score = 0;
       
        for(char c : words[index].toCharArray()){
            if(!curr_freq.containsKey(c) || curr_freq.get(c) <= 0){
                return dfs(words, index+1, freq);
            } else {
                curr_freq.put(c, curr_freq.get(c) - 1);
                score += s[c - 'a'];
            }
        }
       
        int A = score + dfs(words, index+1, curr_freq);
        int B = dfs(words, index+1, freq);
        return Math.max(A, B);
    }
}
