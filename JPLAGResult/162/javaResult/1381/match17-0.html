class Solution {
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] num = new int[26];
        for (char c : letters) num[c - 'a']++;
        dfs(words, 0, num, score, 0);
        return res;
    }
    int res = 0;
    public void dfs(String[] words, int i, int[] num, int[] score, int sum) {
        if (i == words.length) {
            res = Math.max(res, sum);
            return;
        }
        boolean flag = false;
        int[] cur = new int[26];
        for (char c : words[i].toCharArray()) {
            cur[c - 'a']++;
            if (num[c - 'a'] < cur[c - 'a']) {
                flag = true;
            }
        }
        if (flag) dfs(words, i + 1, num, score, sum);
        else {
            int s = 0;
            dfs(words, i + 1, num, score, sum);
            for (char c : words[i].toCharArray()) {
                num[c - 'a']--;
                s += score[c - 'a'];
            }
            dfs(words, i + 1, num, score, s + sum);
            for (char c : words[i].toCharArray()) {
                num[c - 'a']++;
            }
        }
    }
}
