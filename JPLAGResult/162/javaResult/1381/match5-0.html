class Solution {

    int max = 0;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        Map<Character, Integer> let = new HashMap<>();
        for (char c: letters) {
            let.put(c, let.getOrDefault(c, 0) + 1);
        }
        backtrack(words, 0, let, score, 0);
        return max;
    }

    private void backtrack(String[] words, int index, Map<Character, Integer> let, int[] score, int current) {
        if (index == words.length) {
            return;
        }
        backtrack(words, index + 1, let, score, current);
        Map<Character, Integer> wd = new HashMap<>();
        String word = words[index];
        for (char c : word.toCharArray()) {
            wd.put(c, wd.getOrDefault(c, 0) + 1);
        }
        boolean canAdd = true;
        for (Map.Entry<Character, Integer> entry : wd.entrySet()) {
            if (let.getOrDefault(entry.getKey(), 0) < entry.getValue()) {
                canAdd = false;
            }
        }
        if (canAdd) {
            for (Map.Entry<Character, Integer> entry : wd.entrySet()) {
                let.put(entry.getKey(), let.get(entry.getKey()) - entry.getValue());
                current += entry.getValue() * (score[entry.getKey() - 'a']);
            }
            max = Math.max(max, current);
            backtrack(words, index + 1, let, score, current);
            for (Map.Entry<Character, Integer> entry : wd.entrySet()) {
                let.put(entry.getKey(), let.get(entry.getKey()) + entry.getValue());
            }
        }
    }
}
