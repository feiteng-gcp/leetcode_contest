class Solution {
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] count = new int[26];
        for(char c : letters){
            count[c - 'a']++;
        }
        
        return dfs(words, 0, count, score);
    }
    
    private int dfs(String[] words, int loc, int[] count, int[] score){
        if(loc == words.length){
            return 0;
        }
        
        int[] copy = new int[26];
        System.arraycopy(count, 0, copy, 0, 26);
        int res = 0;
        
        int use = helper(words[loc], copy, score);
        if(use > 0){
            res = use + dfs(words, loc + 1, copy, score);
        }
        
        return Math.max(res, dfs(words, loc + 1, count, score));
    }
    
    private int helper(String s, int[] count, int[] score){
        int res = 0;
        for(char c : s.toCharArray()){
            if(count[c - 'a'] == 0){
                return -1;
            }
            count[c - 'a']--;
            res += score[c - 'a'];
        }
        
        return res;
    }
}
