class Solution {
    int res = -1;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] counts = new int[26];
        for(char c : letters) {
            counts[c - 'a']++;
        }
        
        backtrack(words, 0, counts, score, 0);
        
        
return res;
    }
    
    private void backtrack(String[] words, int idx, int[] counts, int[] score, int acc) {
        if(idx == words.length) {
            res = Math.max(res, acc);
            return;
        }
        
        String cur = words[idx];
        
        backtrack(words, idx + 1, counts, score, acc);
        
        
if(isValid(cur, counts)) {
            int gain  = 0;
            for(char c : cur.toCharArray()) {
                counts[c - 'a']--;
                gain += score[c - 'a'];
            }
            
            backtrack(words, idx + 1, counts, score, acc + gain);    
            
            for(char c : cur.toCharArray()) {
                counts[c - 'a']++;
            }
        }
    
}
    
    private boolean isValid(String cur, int[] counts) {
        int[] sum = new int[26];
        for(char c : cur.toCharArray()) {
            sum[c - 'a']++;
        }
        for(int i = 0; i < 26; i++) {
            if(sum[i] > counts[i]) {
                return false;
            }
        }
        return true;
        
    }
}
