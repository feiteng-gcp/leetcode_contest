        int[] sc = new int[26];
        for (char c : letters) sc[c - 'a']++;
        
        
        int n = words.length;
        
int res = 0;
        outer:for (int i = 0; i < (1 << n); i++) {
            int[] cnt = new int[26];
            for (int j = 0; j < n; j++) {
                if (((1 << j) & i) != 0) {
                    for (char c : words[j].toCharArray()) if (++cnt[c - 'a'] > sc[c - 'a']) continue outer
;
                }
            }
            int temp = 0;
            for (int k = 0; k < 26; k++) {
                temp += score[k] * cnt[k];
            
}
            res = Math.max(res, temp);
            
            // map = new HashMap<>();
