class Solution {
    int maxScore = 0;
    public int maxScoreWords(String[] words, char[] letters, int[] scores) {
        Map<Character, Integer> letterCount = new HashMap<>();
        for (char l : letters) {
            letterCount.merge(l, 1, Integer::sum);
        }
        
        getMax(words, letterCount, scores, 0, 0);
        
        return maxScore;
    }
    
    private void getMax(String[] words, Map<Character, Integer> letterCount, int[] scores, int score, int pos) {
        if (pos >= words.length) {
            maxScore = Math.max(maxScore, score);
            return;
        }
        
        if (isWordPossible(words[pos], letterCount)) {
            for (char c : words[pos].toCharArray()) {
                letterCount.put(c, letterCount.get(c)-1);
                score += scores[c - 'a'];
            }
            
            getMax(words, letterCount, scores, score, pos+1);
            
            for (char c : words[pos].toCharArray()) {
                letterCount.put(c, letterCount.get(c)+1);
                score -= scores[c - 'a'];
            }
        }
        getMax(words, letterCount, scores, score, pos+1);
    }
    
    private boolean isWordPossible(String word, Map<Character, Integer> letterCount) {
        Map<Character, Integer> map = new HashMap<>();
        for (char c : word.toCharArray()) {
            map.merge(c, 1, Integer::sum);
        }
        
        for (Map.Entry<Character, Integer> entry : map.entrySet()) {
            if (!letterCount.containsKey(entry.getKey()) || entry.getValue() > letterCount.get(entry.getKey())) {
                return false;
            }
        }
        
        return true;
    }
}
