class Solution {
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] words_scores = new int[words.length];
        for (int i = 0; i < words.length; i++) {
            int tmp = 0;
            for (char c : words[i].toCharArray()) {
                tmp += score[c - 'a'];
            }
            words_scores[i] = tmp;
        }
        
        int[] curr = new int[26];
        for (char c : letters) {
            curr[c - 'a']++;
        
}
        
        return dfs(words, words_scores, curr, 0);
    }

    public int dfs(String[] words, int[] words_score, int[] curr, int pos) {
        if (pos >= words.length) return 0;
        int[] curr_s = count(words[pos]);
        int score = dfs(words, words_score, curr, pos + 1);
        if (sub(curr, curr_s)) {
            score = Math.max(score,
                    dfs(words, words_score, curr, pos + 1) + words_score[pos]);
            add(curr, curr_s);
        
}
        return score;
    }

    public void add(int[] curr, int[] s) {
        for (int i = 0; i < curr.length; i++) {
            curr[i] += s[i];
        }
    }

    public boolean sub(int[] curr, int[] s) {
        for (int i = 0; i < curr.length; i++) {
            if (curr[i] < s[i]) return false;
        }

        for (int i = 0; i < curr.length; i++) {
            curr[i] -= s[i];
        }
        return true;
    }

    public int[] count(
String word) {
        int[] ans = new int[26];
        for (char c : word.toCharArray()) {
            ans[c - 'a']++;
        }
        return ans;
    }
}
