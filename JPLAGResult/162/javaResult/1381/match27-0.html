class Solution {

    boolean legal(int a[], int b[]) {
        for (int i = 0; i < a.length; i++) {
            if (a[i] < b[i]) {
                return false;
            }
        }
        return true;
    }

    int dfs(int cnt[], int a[][], int val[], int cur, int idx) {
        if (idx >= a.length) {
            return cur;
        }
        int res = 0;
        if (legal(cnt, a[idx])) {
            int tmp[] = cnt.clone();
            for (int i = 0; i < 26; i++) {
                tmp[i] -= a[idx][i];
            }
            res = dfs(tmp, a, val, cur + val[idx], idx + 1);
        }
        return Math.max(res, dfs(cnt, a, val, cur, idx + 1));
    }

    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int n = words.length;
        int cnt[] = new int[26];
        for (int i = 0; i < letters.length; i++) {
            cnt[letters[i] - 'a']++;
        }
        int val[] = new int[n];
        int a[][] = new int[n][26];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < words[i].length(); j++) {
                val[i] += score[words[i].charAt(j) - 'a'];
                a[i][words[i].charAt(j) - 'a']++;
            }
        }
        return dfs(cnt, a, val, 0, 0);
    }
}
