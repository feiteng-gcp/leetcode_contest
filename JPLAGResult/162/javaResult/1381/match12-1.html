class Solution {
    
    
    int max = 0;
    
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] alphabet = new int[26];
        for (char letter : letters)
            alphabet[letter -'a']++
;
        
        backtrack(words, score, alphabet, 0, 0);
        return max;
    }
    
    private void backtrack(String[] words, int[] score, int[] alphabet, int index, int curScore) {
        if (index == words.length) {
            max = Math.max(curScore, max);
            return;
        }
        int[] freq = new int[26];
        String word = words[index];
        boolean cover = true;
        int preScore = curScore;
        for (int j = 0; j < word.length(); j++) {
            freq[word.charAt(j) - 'a']++;
            if (freq[word.charAt(j) - 'a'] > alphabet[word.charAt(j) - 'a']) 
{
                cover = false;
                break;
            }
        }
        if (cover) {
            for (int j = 0; j < 26; j++)
                alphabet[j] -= freq[j];
            for (int j = 0; j < 26; j++)
                curScore += score[j] * freq[j];
            //System.out.println("cover:" + words[index] +" with score: " + curScore + " this:" + (curScore - preScore));

            backtrack(words, score, alphabet, index + 1, curScore);
            for (int j = 0; j < 26; j++)
                alphabet[j] += freq[j];

        }
        backtrack(words, score, alphabet, index + 1, preScore);
    }
}
