class Solution {
    
    int answer = 0;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        
        List<String> wordsFiltered = new ArrayList<>();
        
        int count[] = new int[26];
        for( char ch: letters ){
            count[ch-'a']++;
        }
        
        for( int i=0 ; i<words.length ; i++ ){
            
            int temp[] = new int[26];
            for( int j=0 ; j<words[i].length() ; j++ ){
                temp[words[i].charAt(j) - 'a']++;
            }
            
            boolean flag = true;
            for( int j=0 ; j<26 ; j++ ){
                if( temp[j] > count[j] ){
                    flag = false;
                }
            }
            if( flag ){
                wordsFiltered.add( words[i] );
            }
            
        }
        
        answer = 0;
        for( int i=0 ; i<wordsFiltered.size() ; i++ ){
            
            solve( i, count, wordsFiltered, score, 0 );
            
        }
       
        
return answer;
        
    }
    
    void solve( int index , int count[], List<String> arr, int[] score, int total){
        
        // System.out.println(Arrays.toString(takens.toArray()));
        for( int i=index ; i<arr.size() ; i++ ){
            
            if( isPerf( count, arr.get(i) ) ){
                
                String s = arr.get(i);
                int currScore = 0;
                
for( int j=0 ; j<s.length() ; j++ ){
                    count[s.charAt( j )-'a']--;
                    currScore += score[s.charAt( j )-'a'];
                }
                
                solve( i+1, count, arr, score, total+currScore );
                
                for( int j=0 ; j<s.length() ; j++ ){
                    count[s.charAt( j )-'a']++;
                }
                
            }
            
        }
        
        answer = Math.max( answer, total );
        
    
}
    
    boolean isPerf( int count[], String s ){
        
        int temp[] = new int[26];
        
        for( int j=0 ; j<s.length() ; j++ ){
            temp[s.charAt(j) - 'a']++;
        }
        
        for( int i=0 ; i<s.length() ; i++ ){
            
            if( count[s.charAt( i )-'a'] < temp[
s.charAt(i)-'a']  )
                return false;
            
        }
        return true;
        
    }
    
    
}
