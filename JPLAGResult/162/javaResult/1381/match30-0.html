class Solution {
    
    private int max = 0;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] cnt = new int[26];
        for (int i = 0; i < letters.length; i++) {
            cnt[letters[i] - 'a']++;
        }
        
        dfs(words, 0, cnt, score, 0);
        return max;
    }
    private void dfs(String[] words, int start, int[] cnt, int[] score, int cur) {
        if (start >= words.length) {
            max = Math.max(cur, max);
            return;
        }
        
        String word = words[start];
        int[] cost = new int[26];
        for (int i = 0; i < word.length(); i++) {
            cost[word.charAt(i) - 'a']++;
        }
        
        boolean valid = true;
        int now = 0;
        for (int i = 0; i < 26; i++) {
            if (cost[i] > cnt[i]) {
                valid = false;
                break;
            }
            now += score[i] * cost[i];
        }
        
        if (!valid) {
            dfs(words, start + 1, cnt, score, cur);
            return;
        }
        
        dfs(words, start + 1, cnt, score, cur);
        
        for (int i = 0; i < 26; i++) {
            cnt[i] -= cost[i];
        }
        dfs(words, start + 1, cnt, score, cur + now);
        
        for (int i = 0; i < 26; i++) {
            cnt[i] += cost[i];
        }
    }
}
