class Solution {
    int res = 0;
    int[] score;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int []avail = new int[26];
        this.score = score;
        for(char c : letters){
            int index = c - 'a';
            avail[index]++;
        }
        findAllCombination(words, 0, avail, new ArrayList<>(), 0);
        return res;
    }

    private void findAllCombination(String[] words, int pos, int[] avail, List<String> temp, int value) {
        res = Math.max(res, value);
        for(int i = pos; i < words.length; i++){
            String word = words[i];
            int []count = new int[26];
            boolean ok = true;
            for(char c : word.toCharArray()){
                int index = c - 'a';
                count[index]++;
                if(count[index] > avail[index]){
                    ok = false;
                    break;
                }
            }
            if(ok){
                temp.add(word);
                int sum = 0;
                for(char c : word.toCharArray()) {
                    int index = c - 'a';
                    avail[index]--;
                    sum += score[index];
                }
                findAllCombination(words, i + 1, avail, temp, value + sum);
                temp.remove(word);
                for(char c : word.toCharArray()) {
                    int index = c - 'a';
                    avail[index]++;
                }
            }
        }
    }

    
}
