class Solution {
	private int score = 0;

	public int maxScoreWords(String[] words, char[] letters, int[] score) {
		int[] count = new int[26];
		for (char letter : letters) {
			count[letter - 'a']++;
		}
		maxScoreWords(0, 0, words, count, score);
		return this.score;
	}

	private void maxScoreWords(int i, int s, String[] words, int[] count, int[] score) {
		if (i == words.length) {
			this.score = Math.max(this.score, s);
		} else {
			maxScoreWords(i + 1, s, words, count, score);
			int[] clone = count.clone();
			for (char c : words[i].toCharArray()) {
				if (--clone[c - 'a'] < 0) {
					return;
				}
				s += score[c - 'a'];
			}
			maxScoreWords(i + 1, s, words, clone, score);
		}
	}
}
