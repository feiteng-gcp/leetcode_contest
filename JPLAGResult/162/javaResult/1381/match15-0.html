class Solution {
    int maxScore = 0;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] count = new int[26];
        for (char c : letters) {
            count[(int)(c-'a')]++;
        }
        
        check(words, 0, count, score, 0);
        return maxScore;
    }
    void check(String[] words, int index, int[] count, int[] score, int cur_score) {
        if (index == words.length) {
            if (cur_score > maxScore)
                maxScore = cur_score;
            return;
        }
        int[] word_cnt = new int[26];
        for (int i = 0; i < words[index].length(); i++)
            word_cnt[(int)(words[index].charAt(i)-'a')]++;
        
        int i = 0;
        for (; i < 26; i++) {
            if (count[i] < word_cnt[i]) break;
        }
        
        if (i == 26) {
            int[] new_count = new int[26];
            int point = 0;
            for (int j = 0; j < 26; j++) {
                new_count[j] = count[j] - word_cnt[j];
                point+=score[j]*word_cnt[j];
            }
            //System.out.println(point);
            check(words, index+1, new_count, score, cur_score+point);
        }
        check(words, index+1, count, score, cur_score);
    }
}
