class Solution {
    private int max;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        max = 0;
        
int[] count = new int[26];
        for (char c : letters) {
            count[c - 'a']++;
        }
        
        dfs(words, count, 0, 0, score);
        return max;
    }
    
    private void dfs(String[] words, int[] count, int idx, int cur, int[] score) {
        if (idx == words.length) {
            max = Math.max(max, cur);
            return;
        }
        
        // not pick
        dfs(words, count, idx + 1, cur, score);
        
        // try to pick
        int[] cnt = wordToCnt(words[idx]);
        
boolean canForm = true;
        for (int i = 0; i < cnt.length; i++) {
            if (count[i] < cnt[i]) {
                canForm = false;
                break;
            }
        }
        
        if (canForm) {
            
for (int i = 0; i < count.length; i++) {
                count[i] -= cnt[i];
                cur += cnt[i] * score[i];
            }
            
            dfs(words, count, idx + 1, cur, score);
            for (int i = 0; i < count.length; i++) {
                count[i] += cnt[i];
            }
        }
    }
    
    private int[] wordToCnt(String w) {
        int[] cnt = new int[26];
        for (char c : w.toCharArray()) {
            cnt[c - 'a']++;
        }
        
        return cnt;
    }
}
