class Solution {
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
		int[] cnt = new int[26];
		for (char c : letters) {
			cnt[c-'a']++;
		}
		List<int[]> wcnt = new ArrayList<>();
		for (String word : words) {
			int[] w = new int[26];
			for (char c : word.toCharArray()) {
				w[c-'a']++;
			}
			wcnt.add(w);
		}
		return maxScoreWords(wcnt, 0, cnt, score);
    }
	
	private int maxScoreWords(List<int[]> words, int idx, int[] ccnt, int[] score) {
		if (words.size() == idx) {
			return 0;
		}
		int[] word = words.get(idx);
		boolean canFit = canFit(word, ccnt);
		int max = maxScoreWords(words, idx+1, ccnt, score); 
		if(canFit) {
			int count = 0;
			for (int i = 0; i < 26; i++) {
				ccnt[i] -= word[i];
				count += word[i] * score[i];
			}
			max = Math.max(max, count + maxScoreWords(words, idx + 1, ccnt, score));
			for (int i = 0; i < 26; i++) {
				ccnt[i] += word[i];
			}
		}
		return max;
	}
	
	private boolean canFit(int[] word, int[] ccnt) {
		for (int i = 0; i < 26; i++) {
			if (word[i] > ccnt[i]) {
				return false;
			}
		}
		return true;
	}
}
