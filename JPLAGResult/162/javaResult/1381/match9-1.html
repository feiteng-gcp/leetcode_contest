class Solution {
    int result = 0;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        HashMap<Character, Integer> map = new HashMap<>();
        for (char c = 'a'; c <= 'z'; c++) map.put(c, 0);
        for (char c : letters) map.put(c, map.get(c) + 1)
;
        dfs(words, map, score, 0, 0);
        return result;
    }
    
    public void dfs(String[] words, HashMap<Character, Integer> map, int[] score, int val, int index) {
        result = Math.max(result, val);
        for (int i = index; i < words.length; i++) {
            
if (isValid(words[i], map)) {
                for (char c : words[i].toCharArray()) {
                    map.put(c, map.get(c) - 1);
                    val += score[c - 'a'];
                }
                dfs(words, map, score, val, i + 1);
                for (char c : words[i].toCharArray()) {
                    map.put(c, map.get(c) + 1);
                    val -= score[c - 'a'];
                }
            }
        }
    }
    
    public boolean isValid(String s, HashMap<Character, Integer> map) {
        int[] nums = new int[26];
        for (char c : s.toCharArray()) {
            nums[c - 'a']++;
        }
        for (char c = 'a'; c <= 'z'; c++) {
            if (map.get(c) < nums[c - 'a']) return false;
        }
        return true;
    }
}
