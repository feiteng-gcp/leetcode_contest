class Solution {
    private boolean canUseWord(int[][] wordsChars, int i, int[] lettersChars) {
        for (int j = 0; j < 26; j++) {
            if (wordsChars[i][j] > lettersChars[j]) {
                return false;
            }
        }
        
        return true;
    }
    
    private int helper(int[][] wordsChars, int n, int i, int sum, int[] lettersChars, int[] score) {
        if (i == n) {
            return sum;
        }
        
        int usingCurrentWord = 0;
        if (canUseWord(wordsChars, i, lettersChars)) {
            int wordSum = 0;
            for (int j = 0; j < 26; j++) {
                wordSum += wordsChars[i][j] * score[j];
                lettersChars[j] -= wordsChars[i][j]; 
            }
            
            usingCurrentWord = helper(wordsChars, n, i + 1, sum + wordSum, lettersChars, score);
            
            for (int j = 0; j < 26; j++) {
                lettersChars[j] += wordsChars[i][j]; 
            }
        }
        
        int notUsingCurrentWord = helper(wordsChars, n, i + 1, sum, lettersChars, score);
        
        return Math.max(usingCurrentWord, notUsingCurrentWord);
    }
    
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int n = words.length;
        int[][] wordsChars = new int[n][26];
        for (int i = 0; i < n; i++) {
            char[] a = words[i].toCharArray();
            for (char c : a) {
                wordsChars[i][c - 'a']++;
            }
        }
        
        int[] lettersChars = new int[26];
        for (char c : letters) {
            lettersChars[c - 'a']++;
        }
        
        return helper(wordsChars, n, 0, 0, lettersChars, score);
    }
}
