class Solution {
    int result = 0;
    public int maxScoreWords(String[] words, char[] letters, int[] score) {
        int[] arr = new int[26];
        for (int i = 0; i < letters.length; i++) {
            arr[letters[i] - 'a']++;
        }            
        dfs(words, 0, arr, score, 0);
        return result;
    }
    
    public void dfs(String[] words, int level, int[] arr, int[] score, int curScore) {
        if (level == words.length) {
            result = Math.max(result, curScore);
            return;
        }
        int valid = helper(words[level], arr, score);
        if (valid < 0) {
            dfs(words, level + 1, arr, score, curScore);    
        } else {            
            int calScore = update(words[level], arr, score);
            dfs(words, level + 1, arr, score, curScore + calScore);            
            recover(words[level], arr, score);
            dfs(words, level + 1, arr, score, curScore);              
        }                
    }
    
    public int helper(String str, 
int[] arr, int[] score) {
        int[] arrCopy = new int[26];
        for (int i = 0; i < str.length(); i++) {
            arrCopy[str.charAt(i) - 'a']++;
        }      
        for (int i = 0; i < 26; i++) {
            if (arr[i] >= arrCopy[i]) {
                ;
            } else 
{
                return -1;
            }
        }
        return 1;
    }
    
    public void recover(String str, int[] arr, int[] score) {
        for (int i = 0; i < str.length(); i++) {
            arr[str.charAt(i) - 'a']++;
        }        
    }
    
    public int update(String str, int[] arr, int[] score) {
        int result = 0;
        for (int i = 0; i < str.length(); i++) {
            result = result + score[str.charAt(i) - 'a'];
            arr[str.charAt(i) - 'a']--;
        }        
        return result;
    }
}
