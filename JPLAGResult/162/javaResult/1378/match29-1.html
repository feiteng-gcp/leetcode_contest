	class S
olution {
	    public int oddCells(int n, int m, int[][] indices) {
	    	int ret = 0;
	        for(int i = 0;i < n;i++){
	        	for(int j = 0;j < m;j++){
	        		int par = 0;
	        		for(int[] u : indices){
	        			if(i == u[0])par ^= 1;
	        			if(j == u[1])par ^= 1;
	        		}
	        		ret += par;
	        	}
	        }
	        return ret;
	    }
	}
