class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int[][] arr = new int[n][m];
        
        for(int[] ind : indices) {
            int r = ind[0];
            int c = ind[1];
            
            for(int i = 0; i < m; i++) {
                arr[r][i]++;
            }
            for(int i = 0; i < n; i++) {
                arr[i][c]++;
            }
        }
        
        int res = 0;
        for(int i = 0;i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(arr[i][j] % 2 == 1) {
                    res++;
                }
            }
        }
        
        return res;
    }
}
