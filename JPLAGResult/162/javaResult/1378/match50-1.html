class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        
        int ar[][] = new int[n][m];
        
        for( int i=0 ; i<indices.length ; i++ ){
            
            int r = indices[i][0];
            int c = indices[i][1];
            
            
for( int j=0 ; j<m ; j++ ){
                ar[r][j]++;
            }
            for( int j=0 ; j<n ; j++ ){
                ar[j][c]++;
            }
            
        }
        int answer = 0;
        for( int i=0 ; i<n ; i++ ){
            for( int j=0 ; j<m ; j++ ){
                if( ar[i][j] % 2 == 1 ){
                    answer++;
                
}
            }
        }
        
        return answer;
    }
}
