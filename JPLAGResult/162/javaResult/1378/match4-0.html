class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int g[][] = new int[n][m];
        for (int i = 0; i < indices.length; i++) {
            for (int j = 0; j < m; j++) {
                g[indices[i][0]][j]++;
            }
            for (int j = 0; j < n; j++) {
                g[j][indices[i][1]]++;
            }
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (g[i][j] % 2 == 1) {
                    res++;
                }
            }
        }
        return res;
    }
}
