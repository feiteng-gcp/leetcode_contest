class Solution {
    public int oddCells(int n, int m, int[][] indices) {
    	
    	int[][] arr = new int[n][m];
    	
    	for(int i = 0; i < indices.length; i++) {
    		for(int j = 0; j < m; j++) {
    			arr[indices[i][0]][j]++;
    		}
    		for(int j = 0; j < n; j++) {
    			arr[j][indices[i][1]]++;
    		}
    	}
        
    	int ans = 0;
    	for(int i = 0; i < arr.length; i++) {
    		for(int j = 0; j < arr[0].length; j++) {
    			if(arr[i][j] % 2 == 1) ans ++;
    		}
    	}
    	
    	return ans;
    }
}
