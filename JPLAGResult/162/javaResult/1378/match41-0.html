class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int a[][]=new int[n][m];
        for (int i = 0; i < indices.length; i++) {
            for (int j = 0; j <m ; j++) {
                a[indices[i][0]][j]++;
            }
            for (int j = 0; j <n ; j++) {
                a[j][indices[i][1]]++;
            }
        }
        int ans=0;
        for (int i = 0; i <n ; i++) {
            for (int j = 0; j <m ; j++) {
                if(a[i][j]%2==1)ans++;
            }
        }
        return ans;
    }

}
