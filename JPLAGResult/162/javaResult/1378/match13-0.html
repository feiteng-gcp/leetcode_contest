class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int[] lines = new int[n];
        int[] cols = new int[m];
        for(int[] id:indices){
            lines[id[0]]++;
            cols[id[1]]++;
        }
        
        int ans = 0;
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                if((lines[i]+cols[j])%2==1)
                    ++ans;
            }
        }
        return ans;
    }
}
