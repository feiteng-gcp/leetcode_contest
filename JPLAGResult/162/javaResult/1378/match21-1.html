class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int[][] matrix = new int[n][m];
        for(int[] indice:indices){
            for(int i = 0 ; i < m; i++){
                matrix[indice[0]][i]++;
            }
            for(int i = 0 ; i < n ; i++){
                matrix[i][indice[1]]++;
            }
        }
        int res = 0;
        for(int i = 0 ; i < n ; i++){
            for(int j = 0 ; j < m ; j++){
                if((matrix[i][j] & 1) == 1){
                    res++;
                }
            }
        }
        return res;
    }
}
