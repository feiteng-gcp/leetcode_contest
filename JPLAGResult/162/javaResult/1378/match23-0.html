class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int[][] mt = new int[n][m];
        for (int[] op : indices) {
            int r = op[0], c = op[1];
            for (int i = 0; i < m; i++) {
                mt[r][i]++;
            }
            for (int i = 0; i < n; i++) {
                mt[i][c]++;
            }
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (mt[i][j] % 2 == 1) res++;
            }
        }
        // System.out.println(Arrays.deepToString(mt));
        return res;
    }
}
