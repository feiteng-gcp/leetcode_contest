class Solution {
    public int oddCells(int n, int m, int[][] indices) {
		int[][] data = new int[n][m];
		for (int[] ind : indices) {
			int row = ind[0];
			int col = ind[1];
			for (int i = 0; i < m; i++) {
				data[row][i]++;
			}
			for (int i = 0; i < n; i++) {
				data[i][col]++;
			}
		}
		int count = 0;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (data[i][j] % 2 == 1) {
					count++;
				}
			}
		}
		return count;
    }
}
