class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int[][] mat = new int[n][m];
        for (int[] ind : indices) {
        	int row = ind[0];
        	int col = ind[1];
        	for (int i=0; i<m; i++) {
        		mat[row][i] += 1;
        	}
        	for (int i=0; i<n; i++) {
        		mat[i][col] += 1;
        	}
        }
        
        int cnt = 0;
        for (int i=0; i<n; i++) {
        	for (int j=0; j<m; j++) {
        		if (mat[i][j] % 2 ==1)
        			cnt++;
        	}
        }
        return cnt;
    }
}
