class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int[][] arr = new int[n][m];
        for (int[] index : indices) {
            int a = index[0];
            int b = index[1];
            
            for (int i = 0; i < m; i++) {
                arr[a][i]++;
            }
            for (int i = 0; i < n; i++) {
                arr[i][b]++;
            }
        }
        
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (arr[i][j] % 2 != 0) {
                    cnt++;
                }
            }
        }
        
        return cnt;
    }
}
