class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int[][] res = new int[n][m];
        
        for(int[] indice : indices){
            
for(int i=0 ; i<n ; i++){
                res[i][indice[1]] += 1;
            }
            for(int i=0 ; i<m ; i++){
                res[indice[0]][i] += 1;
            }
        }
        
        int num = 0;
        for(int i=0 ; i<n ; i++){
            for(int j=0 ; j<m ; j++){
                if(res[i][j]%2==1){
                    ++num;
                }
            }
        }
        
        return num;
    }
}
