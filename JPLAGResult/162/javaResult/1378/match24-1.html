class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int[][] arr = new int[n][m];
        for (int i = 0; i < indices.length; i++) {
            int row = indices[i][0], col = indices[i][1];
            for (int c = 0; c < m; c++) {
                arr[row][c]++;
            }
            for (int r = 0; r < n; r++) {
                arr[r][col]++;
            }
        }
        int count = 0;
        for (int r = 0; r < n; r++) {
            for (int c = 0; c < m; c++) {
                if (arr[r][c] % 2 == 1) count++;
            }
        }
        return count;
    }
}
