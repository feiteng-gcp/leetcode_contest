class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int[] row_count = new int[n];
        int[] col_count = new int[m];
        for(int[] ind : indices) {
            row_count[ind[0]]++;
            col_count[ind[1]]++;
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int row = row_count[i];
                int col = col_count[j];
                if((row + col) % 2 != 0) {
                    res++;
                }
            }
        }
        return res;
    }
}
