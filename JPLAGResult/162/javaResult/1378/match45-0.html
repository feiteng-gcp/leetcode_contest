class Solution {
    public int oddCells(int n, int m, int[][] indices) {
        int[][] board = new int[n][m];
        for (int [] idx : indices) {
            int r = idx[0], c = idx[1];
            for (int i = 0; i < m; i++) {
                board[r][i]++;
            }
            
            for (int i = 0; i < n; i++) {
                board[i][c]++;
            }
        }
        
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (board[i][j] % 2 != 0) cnt++;
            }
        }
        
        return cnt;
    }
}
