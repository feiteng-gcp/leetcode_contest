from collections import Counter
class Solution:
    ans = 0
    dc = None
    def dfs(self,words,x,n,s,score):
        if x==n:
            if s=="":
                return
            ds = Counter(s)
            ok = 1
            ss = 0
            for i in ds:
                if ds[i]>Solution.dc[i]:
                    ok = 0
                    break
                ss += score[ord(i)-97]*ds[i]
            if ok:
                Solution.ans = max(Solution.ans,ss)
            return
        self.dfs(words,x+1,n,s,score)
        self.dfs(words,x+1,n,s+words[x],score)
                    
        
        
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        Solution.ans = 0
        Solution.dc = Counter(letters)
        n = len(words)
        self.dfs(words,0,n,"",score)
        return Solution.ans
