from collections import Counter


class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        letter_count = Counter(letters)
        return self.solve(words, 0, letter_count, score)

    def solve(self, words, i, letters, score):
        if i == len(words):
            return 0
        s = 0
        take = 0
        wc = Counter(words[i])
        for k, v in wc.items():
            if letters[k] < v:
                break
            s += (score[ord(k) - ord('a')] * v)
        else:
            for k, v in wc.items():
                letters[k] -= v
            take = s + self.solve(words, i+1, letters, score)
            for k, v in wc.items():
                letters[k] += v
        no_take = self.solve(words, i+1, letters, score)
        return max(take, no_take)
