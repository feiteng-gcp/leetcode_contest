from collections import defaultdict
class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        def helper(i):
            if i >= m: return 0
            r = helper(i+1)
            if scw[i][0] > 0:
                done = False
                for j in range(0, 26):
                    c = chr(ord('a')+j)
                    if scw[i][1][c] > nl[c]:
                        done = True
                        break
                if not done:
                    for j in range(0, 26):
                        c = chr(ord('a') + j)
                        nl[c] -= scw[i][1][c]
                    t = scw[i][0]
                    t += helper(i+1)
                    r = max(r, t)
                    for j in range(0, 26):
                        c = chr(ord('a') + j)
                        nl[c] += scw[i][1][c]
            return r

        sc = dict()
        for i, s in enumerate(score):
            sc[chr(ord('a')+i)] = s
        nl = defaultdict(int)
        for c in letters:
            nl[c] += 1
        m = len(words)
        scw = []
        for w in words:
            nrl = defaultdict(int)
            done = False
            for c in w:
                nrl[c] += 1
                if nrl[c] > nl[c]:
                    scw.append((-1, nrl))
                    done = True
                    break
            if not done:
                r = 0
                for c in nrl:
                    r += nrl[c]*sc[c]
                scw.append((r, nrl))
        res = helper(0)
        return res
