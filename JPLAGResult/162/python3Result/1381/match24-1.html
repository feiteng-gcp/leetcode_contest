class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        count = [0]*26
        for c in letters:
            count[ord(c) - ord('a')] +=1
            
        new = []
        for w in words:
            res = 0
            temp = [0]*26
            for c in w:
                res += score[ord(c) - ord('a')]
                temp[ord(c) - ord('a')] += 1
            if res == 0:
                continue
            new.append((temp, res))
            #print((w, res))
        if not new:
            return 0
        n = len(new)
        dic = {}
        def helper(i = 0, s = tuple(count)):
            if i == n:
                return 0
            s0 = s
            if (i, s0) not in dic:
                cost, score = new[i]
                temp = helper(i + 1, s)
                if any([cost[x] >s[x] for x in range(26)]):
                    dic[(i, s)] = temp
                else:
                    s = list(s)
                    for x in range(26):
                        s[x] -= cost[x]
                    s = tuple(s)
                    dic[(i,s0)] = max(temp, score + helper(i + 1, s))
            return dic[(i, s0)]
        return helper()
