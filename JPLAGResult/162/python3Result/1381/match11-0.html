class Solution:
    def canMake(self, words, ls):
        for x in words:
            for y in x:
                if(y not in ls): return False
                ls[y] -= 1
                if(ls[y] < 0): return False
        return True
    
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        ls = {}
        for x in letters:
            ls[x] = ls.get(x,0)+1
        
        s = 0
        for x in range(2**len(words)):
            ws = [y for n,y in enumerate(words) if ((x>>n)&1)==0]
            
            if(self.canMake(ws, dict(ls))):
                s = max(s, sum(sum(score[ord(y)-ord('a')] for y in w) for w in ws))
                
        return s
