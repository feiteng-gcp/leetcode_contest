class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        def get_count(s):
            res = [0]*26
            i = ord('a')
            for cha in s:
                res[ord(cha)-i] += 1
            return res
        def add_count(l1, l2):
            l3 = [0]*26
            for i in range(26):
                l3[i] = l1[i]+l2[i]
            return l3
        def less(l1, l2):
            for i in range(26):
                if l1[i]>l2[i]:
                    return False
            return True
        def sum_(l1):
            res = 0
            for i in range(26):
                res += l1[i]*score[i]
            return res
        all_count = get_count(''.join(letters))
        word_counts = collections.defaultdict(list)
        for i, w in enumerate(words):
            word_counts[i] = get_count(w)
        n = len(words)
        res = 0
        def dfs(start, curr_count):
            # print(curr_count)
            nonlocal res
            nonlocal n
            if not less(curr_count, all_count):
                return
            if start == n:
                res = max(res, sum_(curr_count))
                return
            #not use
            # print('--', curr_count)
            dfs(start+1, curr_count)
            #use
            curr = add_count(word_counts[start], curr_count)
            # print('curr', curr)
            dfs(start+1, curr)
        dfs(0, [0]*26)
        return res
