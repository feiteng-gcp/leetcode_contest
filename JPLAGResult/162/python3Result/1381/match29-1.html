class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        
def get_score(word, score):
            ret = 0
            for c, freq in word.items():
                ret += score[ord(c) - ord('a')] * freq
            return ret
        
        def use_word(word, letters):
            letters = dict((i, j) for i, j in letters.items())
            for c, freq in word.items():
                letters[c] -= freq
                if letters[c] == 0:
                    del letters[c]
            return letters
        
        def is_legal(word, letters):
            for c, freq in word.items():
                if letters.get(c, 0) < freq:
                    return False
            return True
        
        def helper(words, letters, score):
            if len(words) == 1:
                if is_legal(words[0], letters):
                    return get_score(words[0], score)
                else:
                    return 0
            else:
                if is_legal(words[0], letters):
                    return max(
                        get_score(words[0], score) + helper(words[1:], use_word(words[0], letters), score),
                        helper(words[1:], letters, score)
                    )
                else:
                    return helper(words[1:], letters, score)
            
        words = [dict(collections.Counter(i)) for i in words]
        letters = dict(collections.Counter(letters))
        
        return helper(words, letters, score)
