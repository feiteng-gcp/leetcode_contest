class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        def check(word,letters):
            cnt=collections.Counter(word)
            for k in cnt:
                if k in letters and cnt[k]<=letters[k]:
                    continue
                else:
                    return False
            return True
        def getvalue(word,score):
            ret=0
            for w in word:
                ret+=score[ord(w)-ord('a')]
            return ret
        
        letters=collections.Counter(letters)
        value={}
        for w in words:
            if check(w,letters):
                value[w]=getvalue(w,score)
        ans=0
        tmpwords=[]
        for w in words:
            if w in value:
                tmpwords.append(w)
        words=tmpwords
        for i in range(1,len(words)+1):
            li=itertools.combinations(words,i)
            for l in li:
                tmp=0
                if not check(''.join(l),letters):
                    continue
                for k in l:
                    tmp+=value[k]
                ans=max(ans,tmp)
        return ans
