class Solution:#bitmask
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        scorehash = {}
        for i, c in enumerate("abcdefghijklmnopqrstuvwxyz"):
            scorehash[c] = score[i]
        
        res = 0
        counter = collections.Counter(letters)
        n = len(words)
        
        for mask in range(1, 2**n):
            tmp = ""
            sco = 0
            for i in range(n):
                if mask & (1 << i):
                    tmp += words[i]
            
            cnt = collections.Counter(tmp)
            flag = True
            for ch in cnt:
                if ch not in counter:
                    flag = False
                    break
                elif cnt[ch] > counter[ch]:
                    flag = False
                    break
                else:
                    sco += scorehash[ch] * cnt[ch]
            
            if flag:
                res = max(sco, res)
        
        return res
