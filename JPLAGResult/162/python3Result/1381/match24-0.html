class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        count = [0]*26
        for char in letters:
            count[ord(char)- ord('a')] += 1
        thislevel = set()
        thislevel.add(tuple([0]*26))
        for w in words:
            nextlevel = set()
            curr = [0]*26
            for char in w:
                curr[ord(char)-ord('a')] += 1
            
            for prev in thislevel:
                temp = curr.copy()
                flag = 0
                i = 0
                while i< 26:
                    temp[i] += prev[i]
                    if temp[i] > count[i]:
                        flag = 1
                        break
                    i += 1
                if flag == 0:
                    nextlevel.add(tuple(temp))
            thislevel |= nextlevel
        res = 0
        for curr in thislevel:
            temp = 0
            for i in range(26):
                temp += curr[i]*score[i]
            res = max(res, temp)
        return res
