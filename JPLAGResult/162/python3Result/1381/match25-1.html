class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        word_score = [0 for _ in range(len
(words))]
        
        for i, word in enumerate(words):
            for x in word:
                word_score[i] += score[ord(x)-ord('a')]
        
        c = collections.Counter(letters)
        
        dp = [(0, c)]
        
        def test(count_can, word):
            d = collections.Counter()
            
            for key in count_can:
                d[key] = count_can[key]
            
            for x in word:
                d[x] -= 1
            
            return d
        
        for i, word in enumerate(words):
            for can in dp[:]:
                # print(can)
                count_can = can[1]
                vd = test(count_can, word)
                valid = True
                for key in vd.keys():
                    if vd[key] < 0:
                        valid = False
                if valid:
                    if (can[0]+word_score[i], vd) not in dp:
                        dp.append((can[0]+word_score[i], vd))
            # print(dp)
        
        return max(x[0] for x in dp)
