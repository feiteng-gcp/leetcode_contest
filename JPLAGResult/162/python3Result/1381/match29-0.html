class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        from collections import Counter
        if not words: return 0
        candi = []
        memo = Counter(letters)
        for word in words:
            flag = True
            ct = Counter(word)
            for c, v in ct.items():
                if v > memo[c]:
                    flag = False
                    break
            if flag: candi.append(ct)
        if not candi: return 0
        def dfs(idx, path, count):
            nonlocal res

            if idx==len(candi):
                return
            if (count&candi[idx]) == candi[idx]:
                path += value(candi[idx])
                res = max(res, path )

                for i in range(idx+1, len(candi)):
                    dfs(i, path, (count-candi[idx]))
            
        
        def value(ct):
            count = 0
            for c, v in ct.items():
                count += score[ord(c)-ord('a')]*v
            return count
        
        
        res = 0
        for i in range(len(candi)):
            dfs(i, 0, memo)
        return res
