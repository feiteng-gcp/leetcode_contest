import collections

class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        d = collections.defaultdict(int)
        ord_a = ord('a')
        l = [0] * 26
        for c in letters:
            l[ord(c) - ord_a] += 1
        d[tuple(l)] = 0
        for w in words:
            l = [0] * 26
            for c in w:
                l[ord(c) - ord_a] += 1
            s = sum(c * s for c, s in zip(l, score))
            new = []
            for c, curr in d.items():
                new_c = tuple(x - y for x, y in zip(c, l))
                if all(x >= 0 for x in new_c):
                    new.append((new_c, curr + s))
            for new_c, new_s in new:
                d[new_c] = new_s
        return max(d.values())
