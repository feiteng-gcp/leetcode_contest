class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        char_score = {chr(i):score
[i-97] for i in range(97, 122+1)}
        word_score = [0] * len(words)
        for no, w in enumerate(words):
            for c in w:
                word_score[no]+=char_score[c]
        #print(word_score)
        char_count = [0] * 26
        for c in letters:
            char_count[ord(c)-97]+=1
                
        ans = 0
        for status in range(1<<len(words)):
            chars = [0]*26
            score = 0
            no = 0
            while(status):
                if status & 1:
                    score += word_score[no]
                    for c in words[no]:
                        chars[ord(c)-97]+=1
                no += 1
                status >>= 1
            if score <= ans:
                continue
            flag = True
            for i in range(26):
                if chars[i]>char_count[i]:
                    flag = False
            if flag:
                ans = score
                
        return ans
