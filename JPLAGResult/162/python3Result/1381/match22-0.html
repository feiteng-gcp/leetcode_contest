class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        
        def count_word(word):
            c = [0] * 26
            s = 0
            for ch in word:
                c[ord(ch)-ord('a')] += 1
                s += score[ord(ch)-ord('a')]
            return c, s
        
        def less(letters, word):
            for cw, cl in zip(word, letters):
                if cw > cl: return False
            return True
        
        def sub(letters, word):
            for i in range(len(letters)):
                letters[i] -= word[i]
        
        def add(letters, word):
            for i in range(len(letters)):
                letters[i] += word[i]
        
        words = [count_word(word) for word in words]
        letters, _ = count_word(letters)
        
        ans = 0
        def dfs(i, cur):
            nonlocal ans
            if i >= len(words):
                ans = max(ans, cur)
                return
            dfs(i+1, cur)
            if less(letters, words[i][0]):
                sub(letters, words[i][0])
                dfs(i+1, cur + words[i][1])
                add(letters, words[i][0])
        
        dfs(0, 0)
        return ans
