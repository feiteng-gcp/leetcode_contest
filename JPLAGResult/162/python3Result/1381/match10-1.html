from collections import defaultdict


class Solution:
    def maxScoreWords(self, words, letters, score) -> int:
        # backtrace/or backpack??
        d = defaultdict(int)
        for c in letters:
            d[c] += 1
        dw = []
        for w in words:
            dw.append(defaultdict(int))
            for c in w:
                dw[-1][c] += 1
        self
.res = 0

        def cal(i, curdict, curscore):
            if i == len(words):
                self.res = max(curscore, self.res)
                return
            cal(i+1, curdict, curscore)
            newd = defaultdict(int)
            for k in dw[i]:
                if dw[i][k] > curdict[k]:
                    return
                newd[k] = curdict[k] - dw[i][k]
                curscore += dw[i][k] * score[ord(k) - ord('a')]
            for k in curdict:
                if k not in newd:
                    newd[k] = curdict[k]
            cal(i+1, newd, curscore)
        cal(0, d, 0)
        return self.res
