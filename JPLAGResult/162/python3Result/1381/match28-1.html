class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        def to_profile(a):
            p = [0] * 26
            for e in a:
                i = ord(e) - ord('a')
                p[i] += 1
            return p
        
        for i in range(len(words)):
            words[i] = to_profile(words[i])
        
        letters = to_profile(letters)
        
        def profile_to_score(p):
            t = 0
            for i in range(len(score)):
                t += p[i] * score[i]
            return t
        
        self.max_score = 0
        
        def valid(p, cap):
            for i in range(len
(cap)):
                if p[i] > cap[i]:
                    return False
            return True
        
        def profile_sum(p1, p2):
            p = []
            for i in range(len(p1)):
                p.append(p1[i] + p2[i])
            return p
        
        def search(i, p):
            if not valid(p, letters):
                return
            if i == len(words):
                s = profile_to_score(p)
                self.max_score = max(self.max_score, s)
            else:
                search(i + 1, p)
                pp = profile_sum(p, words[i])
                search(i + 1, pp)
            
        search(0, [0] * 26)
        return self.max_score
