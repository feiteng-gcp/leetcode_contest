class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        from collections import defaultdict
        letter_dict = defaultdict(int)
        
        for l in letters:
            letter_dict[l] += 1
        
        score_word = [0] * len(words)
        for i, w in enumerate(words):
            for l in w:
                score_word[i] += score[ord(l) - ord('a')]
        
        res = 0
        for i in range(2 ** len(words)):
            s = bin(i)[2:]
            while len(s) < len(words):
                s = '0' + s
            
            flag = True
            count_letter = defaultdict(int)
            current_res = 0
            for j in range(len(words)):
                if s[j] == '1':
                    current_res += score_word[j]
                    for l in words[j]:
                        count_letter[l] += 1
                        if count_letter[l] > letter_dict[l]:
                            flag = False
                            break
                    if not flag:
                        break
            
            if flag:
                res = max(res, current_res)
        
        return res
