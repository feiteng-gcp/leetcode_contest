from collections import Counter
class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        letters = Counter(letters)
        n = len(letters)
        keys = sorted(letters.keys())
        dic = dict([(c, i) for i, c in enumerate(keys)])
        dp = Counter({tuple([0] * n):0})
        kmax = [letters[k] for k in keys]
        def w2k(w):
            w = Counter(w)
            key = [0] * n
            for k, v in w.items():
                if k not in letters:
                    return []
                key[dic[k]] += v
            return key
        def valid(k):
            for v1, vm in zip(k, kmax):
                if v1 > vm:
                    return False
            return True
        def w_score(w):
            total = 0
            for c in w:
                total += score[ord(c) - ord('a')]
            return total
        for w in words:
            kw = w2k(w)
            if not kw:
                continue
            tmp = {}
            sw = w_score(w)
            for k, v in dp.items():
                nk = [c1 + c2 for c1, c2 in zip(kw, k)]
                if valid(nk):
                    tmp[tuple(nk)] = v + sw
            for k, v in tmp.items():
                dp[k] = max(dp[k], v)
        return max(dp.values())
