class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        chs = {}
        for ch in letters:
            chs[ch] = chs.get(ch, 0) + 1
        n = len(words)
        dic = [{} for i in range(n)]
        scores = [0] * n
        self.res = 0
        for pos, word in enumerate(words):
            for ch in word:
                dic[pos][ch] = dic[pos].get(ch, 0) + 1
                scores[pos] += score[ord(ch)-ord('a')]
        def judge(dic1, dic2):
            for key in dic1:
                if dic1[key] > dic2.get(key, 0):
                    
return False
            return True
        def decrease(dic1, dic2):
            for key in dic1:
                dic2[key] -= dic1[key]
        def increase(dic1, dic2):
            for key in dic1:
                dic2[key] += dic1[key]
        def dfs(pos, cur_score):
            self.res = max(self.res, cur_score)
            for i in range(pos+1, n):
                if judge(dic[i], chs):
                    decrease(dic[i], chs)
                    dfs(i, cur_score + scores[i])
                    increase(dic[i], chs)
        for pos, word in enumerate(words):
            if judge(dic[pos], chs):
                decrease(dic[pos], chs)
                dfs(pos, scores[pos])
                increase(dic[pos], chs)
        return self.res
