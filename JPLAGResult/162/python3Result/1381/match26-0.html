class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        d = dict()
        for i in range(26):
            d[chr(ord('a') + i)] = score[i]
        
        visited = set()
        res = 0
        s = [(0, 0, "".join(letters))]
        
        while s:
            sco, pos, rest = s.pop()
            if (pos, rest) in visited:
                continue
            visited.add((pos, rest))
            res = max(res, sco)
            c = collections.Counter(rest)
            for i in range(pos, len(words)):
                word = words[i]
                cnow = collections.Counter(word)
                mark = True
                for key in cnow:
                    if cnow[key] > c[key]:
                        mark = False
                        break
                if mark:
                    newstr = ""
                    for key in c:
                        newstr += key * (c[key] - cnow[key])
                    s.append((sco + sum([d[ch] for ch in word]), i + 1, newstr))
        return res
