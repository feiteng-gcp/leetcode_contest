def maxScoreWords(words, letters, score):
    # have a helper function that moves one forward
    counts = [0] * 26
    for c in letters:
        counts[ord(c) - ord('a')] +
= 1
        
    scores = {}
    for w in words:
        s = 0
        for c in w:
            s += score[ord(c) - ord('a')]
        scores[w] = s
    
    return _maxScore(words, 0, counts, scores, {})

def _maxScore(words, i, counts, scores, cache):
    if i >= len(words):
        return 0
    key = (i, tuple(counts)) 
    if key in cache:
        return cache[key]
    
    word = words[i]
    new_counts = [x for x in counts]
    can_use = True
    for c in word:
        if new_counts[ord(c) - ord('a')] > 0:
            new_counts[ord(c) - ord('a')] -= 1
        else:
            can_use = False
        
    #use
    if can_use:
        use = scores[word] + _maxScore(words, i + 1, new_counts, scores, cache)
    else:
        use = 0
    not_use = _maxScore(words, i+1, counts, scores, cache)
    cache[key] = max(use, not_use)
    
    return cache[key]
