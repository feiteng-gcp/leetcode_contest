from string import ascii_lowercase as asc

class Solution:

    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        scores = {l: sc for l, sc in zip(asc, score)}
        ws = [0] * len(words)
        for i, word in enumerate(words):
            for ch in word:
                ws[i] += scores[ch]

        self.ls = collections.Counter(letters)
        self.max_ = 0
        
        def get_score(i, curr):
            if i >= len(words):
                self.max_ = max(self.max_, curr)
                return
            www = collections.Counter(words[i])
            if not www - self.ls:
                curr += ws[i]
                self.ls -= www
                self.max_ = max(self.max_, curr)
                get_score(i+1, curr)
                curr -= ws[i]
                self.ls += www
            get_score(i+1, curr)
        get_score(0, 0)
        return self.max_
