class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List
[int]) -> int:
        c = collections.Counter(letters)
        self.res = 0
        def bt(i, cnt, scores):
            if i == len(words):
                self.res = max(self.res, scores)
                return
            
            
            bt(i+1, cnt, scores)
            
            cur = collections.Counter(words[i])
            cur_score = 0
            if all(cur[k] <= cnt[k] for k in cur):
                for c in words[i]:
                    cur_score += score[ord(c) - ord('a')]
                cnt -= cur
                bt(i+1, cnt, scores + cur_score)
                cnt += cur
        bt(0, c, 0)
        return self.res
