class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        letterNum = {}
        for i in letters:
            letterNum[i] = letterNum.get(i, 0) + 1
        self.res = 0
        self.scoreTotal = score
        self.dfs(words, letterNum, 0, 0)
        return self.res
    
    def dfs(self, words, letterNum, score, level):
        if level == len(words):
            self.res = max(self.res, score)
            return
        count = {}
        temp = copy.deepcopy(letterNum)
        valid = True
        word = words[level]
        for i in word:
            if i not in temp or temp[i] == 0:
                valid = False
                break
            temp[i] -= 1
        if valid:
            wordScore = 0
            for i in word:
                wordScore += self.scoreTotal[ord(i) - ord('a')]
            self.dfs(words, temp, score + wordScore, level + 1)
        self.dfs(words, letterNum, score, level + 1)
