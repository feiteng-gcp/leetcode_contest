class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        self.max_score = 0
        self.scores = score
        counter = collections.defaultdict(int)
        for ch in letters:
            counter[ch] += 1
        self
.dfs(words, 0, 0, counter)
        return self.max_score
        
    def dfs(self, words, index, cur_score, num_letters):
        if index == len(words):
            self.max_score = max(self.max_score, cur_score)
            return
        self.dfs(words, index+1, cur_score, num_letters)
        new_num_letters = num_letters.copy()
        for ch in words[index]:
            if new_num_letters[ch] == 0:
                return
            new_num_letters[ch] -= 1
            cur_score += self.scores[ord(ch)-ord("a")]
        self.dfs(words, index + 1, cur_score, new_num_letters)
