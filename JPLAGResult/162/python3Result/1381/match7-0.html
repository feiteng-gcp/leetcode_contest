class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        n = len(words)
        from collections import Counter as C 
        
        b = C(letters)
        fen = [sum(score[ord(i)-ord('a')] for i in x) for x in words]
        a = [C(x) for x in words]
        
        def yes(x):
            for i in x:
                if b[i]<x[i]: return False
            return True
        
        def na(x): 
            for i in x: b[i] -= x[i]
        def nb(x): 
            for i in x: b[i] += x[i]
                
        self.ans = 0
        def dfs(x, f):
            self.ans = max(self.ans, f)
            if x>=n: return
            if yes(a[x]):
                na(a[x])
                dfs(x+1, f+fen[x])
                nb(a[x])
            dfs(x+1, f)
        
        dfs(0, 0)
        return self.ans
