class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        def dfs(move,total,n):
            res=[total]
            for i in range(move,n):
                string="";
                for j in words[i]:
                    if mydict[j]==0:
                        break
                    mydict[j]-=1
                    string+=j
                else:
                    res.append(dfs(i+1,total+weight[i],n))   
                for j in string:
                    mydict[j]+=1
            return max(res)
        mydict=dict()
        for i in "abcdefghijklmnopqrstuvwxyz":
            mydict[i]=0
        for i in letters:
                mydict[i]+=1
        weight=[]
        for i in words:
            val=0
            for j in i:
                val+=score[ord(j)-ord('a')]
            weight.append(val)
            
        return dfs(0,0,len(words))
