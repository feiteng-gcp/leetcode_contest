class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        lettercount = collections.Counter(letters)
        
        
def getMaxScore(i: int, sc: int, remain: dict):
            if i == len(words):
                self.ans = max(self.ans, sc)
                return
            
            wordcount = collections.Counter(words[i])
            if all(wordcount[i] <= remain.get(i, 0) for i in wordcount):
                new = 0
                temp = remain.copy()
                for k in wordcount:
                    temp[k] -= wordcount[k]
                    new += score[ord(k) - ord('a')] * wordcount[k]
                getMaxScore(i + 1, sc + new, temp)
            
            getMaxScore(i + 1, sc, remain)
        
        self.ans = 0
        getMaxScore(0, 0, lettercount)
        return self.ans
