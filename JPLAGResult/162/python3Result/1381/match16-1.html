from collections import Counter
class Solution:
    def maxScoreWords(self, words: List[str], letters: List[str], score: List[int]) -> int:
        letters = Counter(letters)
        return self._maxScoreWords(words, letters, score)
    
    def _maxScoreWords(self, words: List[str], letters: dict, score: List[int]) -> int:
        if not words or not letters:
            return 0
        rem_letters, canForm = self.canFormWord(words[0], letters)
        if canForm:
            s1 = self._maxScoreWords(words[1:], letters, score)
            s2 = self.wordScore(words[0], score) + self._maxScoreWords(words[1:], rem_letters, score)
            return max(s1, s2)
        return self._maxScoreWords(words[1:], letters, score)

    def canFormWord(self, word, letters):
        letters = letters.copy()
        for c in word:
            if letters[c] < 1:
                return None, False
            letters[c] -= 1
        return letters, True
    
    def wordScore(self, word, score):
        ans = 0
        for c in word:
            ans += score[ord(c) - ord('a')]
        return ans
