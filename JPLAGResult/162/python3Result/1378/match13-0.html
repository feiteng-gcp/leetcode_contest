class Solution:
    def oddCells(self, m: int, n: int, indices: List[List[int]]) -> int:
        matrix = [[0]*n for _ in range(m)]
        for i, j in indices:
            for k in range(n):
                matrix[i][k] += 1
            for k in range(m):
                matrix[k][j] += 1
        ret = 0
        for i in range(m):
            for j in range(n):
                if matrix[i][j] % 2:
                    ret += 1
        return ret
