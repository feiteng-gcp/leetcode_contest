class Solution:
    def oddCells(self, m: int, n: int, indices: List[List[int]]) -> int:
        arr = [[0] * n for _ in range(m)]
        for r, c in indices:
            for i in range(m):
                arr[i][c] += 1
            for j in range(n):
                arr[r][j] += 1
        print(arr)                
        total = 0
        for i in range(m):
            for j in range(n):
                if arr[i][j] % 2 == 1:
                    total += 1
        return total
