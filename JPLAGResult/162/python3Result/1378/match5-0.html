class Solution:
    def oddCells(self, n: int, m: int, indices: List[List[int]]) -> int:
        d = [[0 for _ in range(m)] for _ in range(n)]
        
        for r, c in indices:
            for j in range(m):
                d[r][j] += 1
            for i in range(n):
                d[i][c] += 1
        
        res = 0
        for i in range(n):
            for j in range(m):
                if d[i][j] % 2 == 1:
                    res += 1
        
        return res
