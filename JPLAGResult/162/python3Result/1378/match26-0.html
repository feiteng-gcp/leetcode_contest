class Solution:
    def oddCells(self, n: int, m: int, indices: List[List[int]]) -> int:
        
        mt = [ [0] * m for _ in range(n) ]
        
        for r, c in indices:
            
            for i in range(m):
                mt[r][i] += 1
                
            for j in range(n):
                mt[j][c] += 1
                
        ans = 0
        
        for i in range(n):
            for j in range(m):
                ans += mt[i][j]&1
                
        return ans
