class Solution:
    def oddCells(self, n: int, m: int, indices: List[List[int]]) -> int:
        arr=[[0 for _ in range(m)] for _ in range(n)]
        for x,y in indices:
            for i in range(n):
                arr[i][y]+=1
            for i in range(m):
                arr[x][i]+=1
        ret=0
        for i in range(n):
            for j in range(m):
                if arr[i][j]%2==1:
                    ret+=1
        return ret
