# n rows, m cols.

class Solution:
    def oddCells(self, n: int, m: int, indices: List[List[int]]) -> int:
        result = [[0 for _ in range(m)] for _ in range(n)]
        for a, b in indices:
            for i in range(m):
                result[a][i] += 1
            for i in range(n):
                result[i][b] += 1
        answer = 0
        for i in range(n):
            for j in range(m):
                if result[i][j] % 2 == 1:
                    answer += 1
        return answer
