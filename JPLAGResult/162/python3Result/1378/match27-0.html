class Solution:
    def oddCells(self, n: int, m: int, indices: List[List[int]]) -> int:
        memo = [[0] * m for _ in range(n)]
        for i, j in indices:
            for k in range(m):
                memo[i][k] += 1
            for k in range(n):
                memo[k][j] += 1
        res = 0
        for i in range(n):
            for j in range(m):
                res += memo[i][j]&1
        return res
