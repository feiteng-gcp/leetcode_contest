class Solution:
    def oddCells(self, n: int, m: int, indices: List[List[int]]) -> int:
        a = [[0] * m for _ in range(n)]
        
        for i, j in indices:
            for k in range(m): a[i][k] += 1
            for k in range(n): a[k][j] += 1
        
        ans = 0
        for i in range(n):
            for j in range(m):
                if a[i][j]&1==1:
                    ans += 1
        
        return ans
