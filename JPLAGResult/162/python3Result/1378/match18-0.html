class Solution:
    def oddCells(self, n: int, m: int, indices: List[List[int]]) -> int:
        mat = [[0]*m for _ in range(n)]
        for r, c in indices:
            for j in range(m):
                mat[r][j] += 1
            for i in range(n):
                mat[i][c] += 1
        cnt = 0
        for r in range(n):
            for c in range(m):
                if mat[r][c]%2:
                    cnt += 1
        return cnt
