class Solution:
    def oddCells(self, n: int, m: int, indices: List[List[int]]) -> int:
        cells = [[0] * m for _ in range(n)]
        for x, y in indices:
            for y1 in range(m):
                cells[x][y1] += 1
            for x1 in range(n):
                cells[x1][y] += 1
        ans = 0
        for x in range(n):
            for y in range(m):
                ans += (cells[x][y] % 2)
        return ans
