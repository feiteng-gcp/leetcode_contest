class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        sums = sum(colsum)
        if upper + lower != sums:
            return []
        two = 0
        for x in colsum:
            if x == 2:
                two += 1
        if two > min(upper, lower):
            return []
        n = len(colsum)
        res = [[0 for j in range(n)] for i in range(2)]
        for i in range(n):
            if colsum[i] == 2:
                res[0][i], res[1][i] = 1, 1
                upper -= 1
                lower 
-= 1
        for i in range(n):
            if colsum[i] == 1:
                if upper > 0:
                    upper -= 1
                    res[0][i] = 1
                else:
                    lower -= 1
                    res[1][i] = 1
        return res
