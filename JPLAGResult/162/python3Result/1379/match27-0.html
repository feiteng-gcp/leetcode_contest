class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        ret = [[], []]
        both = sum([e==2 for e in colsum])
        u, l = upper-both, lower-both
        for e in colsum:
            if e == 2:
                ret[0].append(1)
                ret[1].append(1)
            elif e == 0:
                ret[0].append(0)
                ret[1].append(0)
            elif e == 1:
                if u:
                    ret[0].append(1)
                    ret[1].append(0)
                    u -= 1
                else:
                    ret[0].append(0)
                    ret[1].append(1)
                    l -= 1
            else:
                return []
            if u < 0 or l < 0:
                return []
        if u != 0 or l != 0:
            return []
        return ret
