class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if sum(colsum)!=(upper+lower):
            return []
        
        m=len(colsum)
        ret=[[0 for j in range(m)] for i in range(2)]
        for i,x in enumerate(colsum):
            #print(i,x)
            if x==2:
                if upper>0 and lower>0:
                    ret[0][i]=1
                    ret[1][i]=1
                    upper-=1
                    lower-=1
                else:
                    return []
        for i,x in enumerate(colsum):
            if x==1:
                
if upper>0:
                    ret[0][i]=1
                    upper-=1
                elif lower>0:
                    ret[1][i]=1
                    lower-=1
                else:
                    return []
        #print(upper,lower)
        return ret
