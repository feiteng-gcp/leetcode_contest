class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if upper + lower != sum(colsum):
            return []
        
        ret = [[], []]
        
        for col in colsum:
            if col == 2:
                ret[0].append(1)
                ret[1].append(1)
                upper -= 1
                lower -= 1
                
            elif col == 1:
                if upper >= lower:
                    ret
[0].append(1)
                    ret[1].append(0)
                    upper -= 1
                else:
                    ret[0].append(0)
                    ret[1].append(1)
                    lower -= 1
            else:
                ret[0].append(0)
                ret[1].append(0)
            if upper < 0 or lower < 0:
                return []
        
        return ret
