class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if upper + lower != sum(colsum):
            return []
        
        res = [[-1] * len(colsum), [-1] * len(colsum)]
        
        for i in range(len(colsum)):
            if colsum[i] == 2:
                res[0][i] = 1
                res[1][i] = 1
                upper -= 1
                lower -= 1
            elif colsum[i] == 0:
                res[0][i] = 0
                res[1][i] = 0
        
        for i in range(len(colsum)):
            if colsum[i] == 1:
                if upper > 0:
                    upper -= 1
                    res[0][i] = 1
                    res[1][i] = 0
                else:
                    lower -= 1
                    res[0][i] = 0
                    res[1][i] = 1
        
        return res if upper >= 0 and lower >= 0 else []
