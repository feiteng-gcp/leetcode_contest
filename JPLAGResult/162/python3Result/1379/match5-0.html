class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if sum(colsum) != upper + lower:
            return []
        COL = len(colsum)
        res = [[0] * COL for _ in range(2)]
        
        for i in range(COL):
            if colsum[i] == 2:
                upper -= 1
                lower -= 1
                res[0][i] = res[1][i] = 1
        for i in range(COL):
            if colsum[i] == 1:
                if upper:
                    upper -= 1
                    res[0][i] = 1
                else:
                    lower -= 1
                    res[1][i] = 1
        return res if upper == 0 and lower == 0 else []
