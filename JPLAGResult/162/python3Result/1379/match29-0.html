class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        n = len(colsum)
        res = [[-1] * n, [-1] * n]
        unknowns = 0
        for j in range(n):
            if colsum[j] == 0:
                res[0][j] = 0
                res[1][j] = 0
            elif colsum[j] == 2:
                res[0][j] = 1
                res[1][j] = 1
                upper -= 1
                lower -= 1
            elif colsum[j] == 1:
                unknowns += 1
            else:
                return []
        if lower < 0 or upper < 0:
            return []
        if lower + upper != unknowns:
            return []
        for j in range(n):
            if res[0][j] == -1:
                if upper > 0:
                    upper -= 1
                    res[0][j] = 1
                    res[1][j] = 0
                else:
                    lower -= 1
                    res[0][j] = 0
                    res[1][j] = 1
        return res
