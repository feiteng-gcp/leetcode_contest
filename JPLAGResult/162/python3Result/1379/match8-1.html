class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        ans = [[0] * len(colsum), [0] * len(colsum)]
        for i, x in enumerate(colsum):
            if x == 2:
                upper -= 1
                lower -= 1
                ans[0][i] = 1
                ans[1][i] = 1
                colsum[i] = 0
        if upper < 0 or lower < 0 or lower + upper != sum(colsum):
            return []
        else:
            
for i, x in enumerate(colsum):
                if x == 1:
                    if lower > 0:
                        ans[1][i] = 1
                        lower -= 1
                    else:
                        ans[0][i] = 1
                        upper -= 1
        return ans
