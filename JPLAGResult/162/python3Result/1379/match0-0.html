class Solution:#O(n),注意特判！
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if sum(colsum) != (upper + lower): return []
        counter = collections.Counter(colsum)
        if counter[2] > min(upper, lower): return []
        
        upcnt = 0
        r1, r2 = [0 for _ in range(len(colsum))], [0 for _ in range(len(colsum))]
        for i, s in enumerate(colsum):
            if s == 2:
                r1[i] = 1
                upcnt += 1
                r2[i] = 1
        
        for i,s in enumerate(colsum):
            if s == 1:
                if upcnt < upper:
                    upcnt += 1
                    r1[i] = 1
                else:
                    r2[i] = 1
        
        return [r1, r2]
