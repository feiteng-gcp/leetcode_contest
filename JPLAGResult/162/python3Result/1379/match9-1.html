class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if upper < lower:
            return self.reconstructMatrix(lower, upper, colsum)[::-1]
        n = len(colsum)
        num_2 = sum([i == 2 for i in colsum])
        if upper + lower != sum(colsum) or num_2 > lower or num_2 > upper:
            return []
        matrix = [[0 for j in range(n)] for i in range(2)]
        for i, j in enumerate(colsum):
            if j == 2:
                matrix[0][i] = 1
                upper -= 1
                matrix[1][i] 
= 1
                lower -= 1
        for i, j in enumerate(colsum):
            if j == 1:
                if upper > 0:
                    matrix[0][i] = 1
                    upper -= 1
                else:
                    matrix[1][i] = 1
                    lower -= 1

        return matrix
