class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        count=[0,0,0]
        for num in colsum:
            count[num]+=1
        if count[2]>upper or count[2]>lower or upper+lower-(count[2])*2!=count[1]:
            return []
        ret=[[],[]]
        up=upper-count[2]
        
for num in colsum:
            if num==0:
                ret[0].append(0)
                ret[1].append(0)
            elif num==2:
                ret[0].append(1)
                ret[1].append(1)
            else:
                if up>0:
                    ret[0].append(1)
                    ret[1].append(0)
                    up-=1
                else:
                    ret[0].append(0)
                    ret[1].append(1)
        return ret
