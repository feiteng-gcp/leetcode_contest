class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if upper+lower!=sum(colsum):
            return []
        res_list = [[],[]]
        for i in colsum:
            if i==2:
                upper-=1
                lower-=1
                res_list[0].append(1)
                res_list[1].append(1)
            elif i==1:
                if upper>=lower:
                    upper-=1
                    res_list[0].append(1)
                    res_list[1].append(0)
                else:
                    lower-=1
                    res_list[0].append(0)
                    res_list[1].append(1)
            else:
                res_list[0].append(0)
                res_list[1].append(0)
        if upper==0 and lower==0:
            return res_list
        else:
            return []
