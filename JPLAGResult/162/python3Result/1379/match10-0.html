class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if upper + lower != sum(colsum):
            return []
        
        N = len(colsum)
        ret = [[0] * N, [0] * N]
        
        for i, s in enumerate(colsum):
            if s == 2:
                ret[0][i] = 1
                ret[1][i] = 1
                upper -= 1
                lower -= 1
            if upper < 0 or lower < 0:
                return []
            
        for i, s in enumerate(colsum):
            if s == 1:
                if upper:
                    ret[0][i] = 1
                    upper -= 1
                else:
                    ret[1][i] = 1
                    lower -= 1
        return ret
