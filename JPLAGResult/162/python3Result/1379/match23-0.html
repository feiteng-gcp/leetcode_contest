class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        n = len(colsum)
        if upper + lower != sum(colsum): return []
        ans = [[0] * n for _ in range(2)]
        for i, c in enumerate(colsum):
            if c == 2:
                ans[0][i] = ans[1][i] = 1
                upper -= 1
                lower -= 1
            elif c == 1:
                if upper > lower:
                    ans[0][i] = 1
                    upper -= 1
                else:
                    ans[1][i] = 1
                    lower -= 1
        return ans if upper == lower == 0 else []
