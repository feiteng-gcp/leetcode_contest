class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum):
        if upper + lower != sum(colsum):
            return []
        ans = [[-1] * len(colsum) for _ in range(2)]
        t = 0
        s 
= 0
        for i in range(len(colsum)):
            if colsum[i] == 0:
                ans[0][i] = 0
                ans[1][i] = 0
            elif colsum[i] == 2:
                ans[0][i] = 1
                ans[1][i] = 1
                s += 1
            elif colsum[i] == 1:
                t += 1
        u = upper - s
        l = lower - s
        if u >= 0 and l >= 0 and u + l == t:
            used_u = 0
            used_l = 0
            
for i in range(len(colsum)):
                if colsum[i] == 1:
                    if used_u < u:
                        ans[0][i] = 1
                        ans[1][i] = 0
                        used_u += 1
                    elif used_l  < l:
                        ans[1][i] = 1
                        ans[0][i] = 0
                        used_l += 1
            return ans
        else:
            return []
