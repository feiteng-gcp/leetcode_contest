class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if sum(colsum)!=upper+lower: return []
        l = len(colsum)
        res= [[0]*l for _ in range(2)]

        for i in range(l):
            if colsum[i] == 2:
                res[0][i] = 1
                res[1][i] = 1
                upper -= 1
                lower -= 1
        if upper<0 or lower<0: return [] 
        
        for i in range(l):
            if colsum[i] == 1:
                if upper>0:
                    res[0][i] = 1
                    upper-=1
                elif lower>0:
                    res[1][i] = 1
                    lower -= 1
                else:
                    return []
        return res if not upper and not lower else []
