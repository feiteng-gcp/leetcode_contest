class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: 'List[int]') -> 'List[List[int]]':
        cu,cl=upper,lower
        l=len(colsum)
        ans=[[0]*l for i in range(2)]
        for i in range(l):
            if colsum[i]==2:
                cu-=1
                cl-=1
                ans[0][i]+=1
                ans[1][i]+=1
        for i in range(l):
            if colsum[i]==1:
                if cu>0:
                    cu-=1
                    ans[0][i]+=1
                else:
                    cl-=1
                    ans[1][i]+=1
            if cu<0 or cl<0:
                return []
        if cu==0 and cl==0:
            return ans
        return []
