class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if upper + lower != sum(colsum):
            return []
        count2 = colsum.count(2)
        if count2 > upper or count2 > lower:
            return []

        n = len(colsum)
        res = [[0] * n for _ in range(2)]

        upper -= count2
        for i, x in enumerate(colsum):
            if x == 2:
                res[0][i] = res[1][i] = 1
            elif x == 1:
                if upper > 0:
                    res[0][i] = 1
                    upper -= 1
                else:
                    res[1][i] = 1
        return res
