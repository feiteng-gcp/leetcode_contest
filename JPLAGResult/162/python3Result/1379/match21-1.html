class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        
        if upper + lower != sum(colsum):
            return []
        
        n = len(colsum)
        res = [[0] * n for _ in range(2)]
        common = 0
        for i in range(n):
            if colsum[i] == 2:
                res[0][i] += 1
                res[1][i] += 1
                common += 1
                
        if upper < common or lower < common:
            return []
                
        for i in range(n):
            if colsum[i] == 1:
                if upper > common:
                    upper -= 1
                    res
[0][i] += 1
                else:
                    lower -= 1
                    res[1][i] += 1
        return res
