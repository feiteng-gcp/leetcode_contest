class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        c = collections.Counter(colsum)
        if upper + lower != sum(colsum) or c[2] > upper or c[2] > lower:
            return []
        
        res = [[0 for _ in range(len(colsum))] for _ in range(2)]
        
        u = 0
        for i in range(len(colsum)):
            if colsum[i] == 2:
                res[0][i] = 1
                res[1][i] = 1
                u += 1
            elif colsum[i] == 0:
                res[0][i] = 0
                res[1][i] = 0
                
        for i in range(len(colsum)):
            if colsum[i] == 1:
                if u < upper:
                    res[0][i] = 1
                    res[1][i] = 0
                    u += 1
                else:
                    res[1][i] = 1
                    res[0][i] = 0
        return res
