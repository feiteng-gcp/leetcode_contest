class Solution:
    def reconstructMatrix(self, upper: int, lower: int, colsum: List[int]) -> List[List[int]]:
        if upper + lower != sum(colsum): return []
        n = len(colsum)
        r = [[0] * n for _ in range(2)]
        for j, c in enumerate(colsum):
            if c == 2:
                r[0][j] = 1
                r[1][j] = 1
                upper -= 1
                lower -= 1
        if upper < 0 or lower < 0: return []
        
for j, c in enumerate(colsum):
            if c == 1:
                if upper > 0:
                    r[0][j] = 1
                    upper -= 1
                else:
                    r[1][j] = 1
        return r
