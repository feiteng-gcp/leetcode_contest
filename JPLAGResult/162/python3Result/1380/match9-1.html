class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        visited = [[0]*n for _ in range(m)]
        ds = [(-1,0),(0,-1),(1,0),(0,1)]
        def dfs(i, j):
            if i == 0 or j == 0 or i == m-1 or j == n-1:
                visited[i][j] = 1
                return False
            if visited[i][j]:
                return True
            visited[i][j] = 1
            ret = True
            for x, y in ds:
                i_, j_ = i+x, j+y
                if 0<=i_<m and 0<=j_<n and grid
[i_][j_] == 0:
                    ret &= dfs(i_, j_)
            return ret
        ret = 0
        for i in range(m):
            for j in range(n):
                if not visited[i][j] and grid[i][j] == 0:
                    ret += dfs(i, j)
        return ret
