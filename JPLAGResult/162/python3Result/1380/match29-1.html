class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        n_rows = len(grid)
        n_cols = len(grid[0])
        visited = [[False for j in range(n_cols)] for i in range(n_rows)]
        
        def dfs(grid, i, j, visited):
            if i < 0 or i > n_rows - 1 or j < 0 or j > n_cols - 1:
                return False
            elif visited[i][j]:
                return True
            elif grid[i][j] == 1:
                visited[i][j] = True
                return True
            else:
                visited[i][j] = True
                a = dfs(grid, i + 1, j, visited)
                b = dfs(grid, i - 1, j, visited)
                c = dfs(grid, i, j + 1, visited)
                d = dfs(grid, i, j - 1, visited)
                return a and b and c and d
        
        ret 
= 0
        for i in range(n_rows):
            for j in range(n_cols):
                if grid[i][j] == 0 and not visited[i][j]:
                    is_valid = dfs(grid, i, j, visited)
                    if is_valid:
                        ret += 1
                        # print(i, j)
        return ret
