class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        visited = [[0] * n for _ in range(m)]
        
        def dfs(i, j):
            visited[i][j] = 1
            flag = 0
            if i == 0 or i == m-1 or j == 0 or j == n-1:
                flag = 1
            for x, y in [(-1, 0), (0, 1), (1, 0), (0, -1)]:
                if 0 <= i+x < m and 0 <= j+y < n and visited[i+x][j+y] == 0 and grid[i+x][j+y] == 0:
                    if dfs(i+x, j+y) == 0:
                        flag = 1
            if flag:
                return 0
            else:
                return 1
        
        ans = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 0 and visited[i][j] == 0:
                    ans += dfs(i, j)
        return ans
