class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        res = 0
        m, n = len(grid), len(grid[0])

        visited = [[False for _ in range(n)] for _ in range(m)]
        
        def out(x, y):
            nonlocal m, n
            if x < 0 or x >= m or y < 0 or y >= n:
                return True
        
        moves = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1: continue
                if visited[i][j]: continue
                
                queue = collections.deque()
                queue.append((i, j))
                visited[i][j] = True
                flag = True
                while queue:
                    x, y = queue.popleft()
                    for dx, dy in moves:
                        nx, ny = x + dx, y + dy
                        
                        if out(nx, ny): flag = False
                        elif visited[nx][ny]: continue
                        else:
                            if grid[nx][ny] == 0:
                                queue.append((nx, ny))
                                visited[nx][ny] = True

                if flag:res += 1
            
        return res
