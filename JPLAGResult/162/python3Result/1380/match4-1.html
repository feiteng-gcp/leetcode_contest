class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        R, C = len(grid), len(grid[0])
        
        
def dfs(r, c):
            grid[r][c] = 2
            for nr, nc in [[r+1, c], [r, c+1], [r-1, c], [r, c-1]]:
                if 0 <= nr < R and 0 <= nc < C and grid[nr][nc] == 0:
                    dfs(nr, nc)
        
        for r in range(R):
            if grid[r][0] == 0:
                dfs(r, 0)
            if grid[r][C-1] == 0:
                dfs(r, C-1)
        for c in range(C):
            if grid[0][c] == 0:
                dfs(0, c)
            if grid[R-1][c] == 0:
                dfs(R-1, c)
        count = 0
        
for r in range(R):
            for c in range(C):
                if grid[r][c] == 0:
                    count += 1
                    dfs(r, c)
        return count
