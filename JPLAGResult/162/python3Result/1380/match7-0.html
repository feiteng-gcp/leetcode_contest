class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        row = len(grid)
        col = len(grid[0])

        def dfs(x, y):
            q = [[x, y]]
            while q:
                x0, y0 = q.pop(0)
                grid[x0][y0] = 1
                d = [-1, 0, 1, 0, -1]
                for k in range(4):
                    dx = x0 + d[k]
                    dy = y0 + d[k + 1]
                    if 0 <= dx < row and 0 <= dy < col and grid[dx][dy] == 0:
                        grid[dx][dy] = 1
                        q.append([dx, dy])

        for i in range(row):
            if grid[i][0] == 0:
                dfs(i, 0)
            if grid[i][col - 1] == 0:
                dfs(i, col - 1)
        for j in range(col):
            if grid[0][j] == 0:
                dfs(0, j)
            if grid[row - 1][j] == 0:
                dfs(row - 1, j)
        ans = 0
        for i in range(row):
            for j in range(col):
                if grid[i][j] == 0:
                    ans += 1
                    dfs(i, j)
        return ans
