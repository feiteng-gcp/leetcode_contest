class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        ans = 0
        def dfs(x, y):
            grid[x][y] = -1
            for a, b in ((1,0), (0,1), (-1,0), (0,-1)):
                if 0 <= x + a < m and 0 <= y + b < n and grid[x+a][y+b] == 0:
                    dfs(x+a, y + b)
        for i in (0, m - 1):
            for j in range(n):
                if grid[i][j] == 0:
                    dfs(i, j)
        for j in (0, n - 1):
            for i in range(1, m - 1):
                if grid[i][j] == 0:
                    dfs(i,j)
        for i in range(1, m-1):
            for j in range(1, n-1):
                if grid[i][j] == 0:
                    ans += 1
                    dfs(i, j)
        return ans
