class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        '''
        Completely surrounded by water --> can not reach from boundary.
        '''
        
        m, n = len(grid), len(grid[0])
        
        dirs 
= [
            (1, 0),
            (-1, 0),
            (0, -1),
            (0, 1),
        ]
        
        ans = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 0:
                    q = [(i, j)]
                    grid[i][j] = 2 # visited
                    closed = True
                    while q:
                        r, c = q.pop()
                        for dr, dc in dirs:
                            rr = r + dr
                            cc = c + dc
                            if rr < 0 or rr >= m or cc < 0 or cc >= n:
                                # touch the boundary
                                closed = False
                            else:
                                if grid[rr][cc] == 0:
                                    grid[rr][cc] = 2 # visited
                                    q
.append((rr, cc))
                    # end while
                    ans += closed
        return ans
