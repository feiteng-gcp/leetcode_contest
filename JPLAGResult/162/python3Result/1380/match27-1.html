class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        cnt = 0
        N = len(grid)
        M = len(grid[0])
        
        
def dfs(i, j):
            grid[i][j] = 2
            ret = 1
            if i == 0 or i == N - 1 or j == 0 or j == M - 1:
                ret = 0
            
            for ii, jj in [(i-1, j), (i+1, j), (i, j-1), (i, j+1)]:
                if 0 <= ii < N and 0 <= jj < M and grid[ii][jj] == 0:
                    ret &= dfs(ii, jj)
            return ret
                
        
        for i in range(N):
            for j in range(M):
                if grid[i][j] == 0:
                    cnt += dfs(i, j)
        return cnt
