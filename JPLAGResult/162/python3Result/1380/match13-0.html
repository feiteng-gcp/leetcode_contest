class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        def dfs(i, j):
            if 0 <= i < len(grid) and 0 <= j < len(grid[0]) and grid[i][j] == 0:
                grid[i][j] = -1
                dfs(i+1, j)
                dfs(i-1, j)
                dfs(i, j+1)
                dfs(i, j-1)
                
        for i in range(len(grid)):
            if grid[i][0] == 0:
                dfs(i, 0)
            if grid[i][-1] == 0:
                dfs(i, len(grid[0])-1)
        
        for j in range(len(grid[0])):
            if grid[0][j] == 0:
                dfs(0, j)
            if grid[-1][j] == 0:
                dfs(len(grid)-1, j)
        
        ans = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 0:
                    ans += 1
                    dfs(i, j)
        return ans
