class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        ans = 0
        m, n = len(grid), len(grid[0])
        visited = [[False] * n for _ in range(m)]
        
        def dfs(i, j):
            visited[i][j] = True
            border = i in (0, m - 1) or j in (0, n - 1)
            for x, y in [(i + 1, j), (i - 1, j), (i, j + 1), (i, j - 1)]:
                if 0 <= x < m and 0 <= y < n and not visited[x][y] and not grid[x][y]:
                    border = dfs(x, y) or border
            return border
        
        
for i in range(m):
            for j in range(n):
                if not visited[i][j] and not grid[i][j]:
                    ans += not dfs(i, j)
        return ans
