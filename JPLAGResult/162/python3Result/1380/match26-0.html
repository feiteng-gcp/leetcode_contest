class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:        
        def dfs1(i, j):
            grid[i][j] = 1
            for a, b in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                if 0 <= i + a < len(grid) and 0 <= j + b < len(grid[0]) and grid[i + a][j + b] == 0:
                    dfs1(i + a, j + b)
        
        for i in range(len(grid)):
            if grid[i][0] == 0:
                dfs1(i, 0)
            if grid[i][len(grid[0]) - 1] == 0:
                dfs1(i, len(grid[0]) - 1)
        
        for j in range(len(grid[0])):
            if grid[0][j] == 0:
                dfs1(0, j)
            if grid[len(grid) - 1][j] == 0:
                dfs1(len(grid) - 1, j)
        
        res = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 0:
                    res += 1
                    dfs1(i, j)
        return res
