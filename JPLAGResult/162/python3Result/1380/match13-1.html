class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
      c = 0
      if not grid or not grid[0]:
        return 0
      def dfs(i, j):
        if i < 0 or j < 0 or i >= len(grid) or j >= len(grid[0]):
          return
        if grid[i][j] != 0:
          return
        grid[i][j] = 1
        dfs(i, j+1)
        dfs(i, j-1)
        dfs(i+1, j)
        dfs(i-1, j)
      for j, _ in enumerate
(grid[0]):
        if grid[0][j] == 0:
          dfs(0, j)
        if grid[-1][j] == 0:
          dfs(len(grid)-1, j)
      for i, _ in enumerate
(grid):
        if grid[i][0] == 0:
          dfs(i, 0)
        if grid[i][-1] == 0:
          dfs(i, len(grid[0])-1)

      for i, row in enumerate(grid):
        for j, val in enumerate(row):
          if val == 0:
            c += 1
            dfs(i, j)
      return c
