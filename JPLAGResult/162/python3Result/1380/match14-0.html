class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        for i in range(len(grid)):
            if grid[i][0] == 0:
                self.fillWater(grid, i, 0)
            if grid[i][-1] == 0:   
                self.fillWater(grid, i, len(grid[0])-1)
        for j in range(len(grid[0])):
            if grid[0][j] == 0:
                self.fillWater(grid, 0, j)
            if grid[-1][j] == 0:
                self.fillWater(grid, len(grid)-1, j)
        count = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 0:
                    count += 1
                    self.fillWater(grid, i, j)
        return count
        
    def fillWater(self, grid, x, y):
        queue = collections.deque()
        queue.append((x,y))
        while queue:
            x,y = queue.popleft()
            if grid[x][y] == 1:
                continue
            grid[x][y] = 1
            for dx in [-1,1]:
                nx = x+dx
                if nx < 0 or nx >= len(grid):
                    continue
                queue.append((nx, y))
            for dy in [-1,1]:
                ny = y+dy
                if ny < 0 or ny >= len(grid[0]):
                    continue
                queue.append((x, ny))
