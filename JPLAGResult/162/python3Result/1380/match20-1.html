class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        offsets = [(0, 1), (0, -1), (1, 0), (-1, 0)]
        def dfs(r, c):
            if not (0 <= r < m and 0 <= c < n): return
            if grid[r][c] == 1: return
            grid[r][c] = 1
            for dr, dc in offsets:
                dfs
(r + dr, c + dc)
        for r in [0, m - 1]:
            for c in range(n):
                dfs(r, c)
        for r in range(m):
            for c in [0, n - 1]:
                dfs(r, c)
                
        ans = 0
        for r in range(m):
            for c in range(n):
                if grid[r][c] == 0:
                    ans += 1
                    dfs(r, c)
        return ans
