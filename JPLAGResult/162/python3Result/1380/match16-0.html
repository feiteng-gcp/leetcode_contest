class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        N = len(grid)
        M = len(grid[0])
        marked = set()
        dxdy = ((0, 1), (0, -1), (1, 0), (-1, 0))
        ans = 0
        for i in range(N):
            for j in range(M):
                if grid[i][j] == 0 and (i, j) not in marked:
                    marked.add((i, j))
                    s = [(i, j)]
                    isClosed = True
                    while s:
                        (x, y) = s.pop()
                        for dx, dy in dxdy:
                            u, v = x+dx, y + dy
                            if not (0 <= u < N and 0 <= v < M):
                                isClosed = False
                            else:
                                if grid[u][v] == 0 and (u, v) not in marked:
                                    marked.add((u, v))
                                    s.append((u, v))
                    ans += isClosed
        return ans
