class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        res = 0
        m = len(grid)
        if m == 0:
            return 0
        n = len(grid[0])
        if n ==0:
            return 0
        def dfs(i, j, m, n, grid):
            if i<0 or j<0 or i>=m or j>=n or grid[i][j] == 1:
                return
            grid[i][j] = 1
            dfs(i+1, j, m, n, grid)
            dfs(i-1, j, m, n, grid)
            dfs(i, j+1, m, n, grid)
            dfs(i, j-1, m, n, grid)
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 0:
                    res += 1
                    dfs(i,j,m,n,grid)
        return res
    def closedIsland(self, grid: List[List[int]]) -> int:
        def dfs(i, j, m, n, grid):
            if i<0 or j<0 or i>=m or j>=n or grid[i][j] == 1:
                return
            grid[i][j] = 1
            dfs(i+1, j, m, n, grid)
            dfs(i-1, j, m, n, grid)
            dfs(i, j+1, m, n, grid)
            dfs(i, j-1, m, n, grid)
        m = len(grid)
        if m == 0:
            return 0
        n = len(grid[0])
        if n ==0:
            return 0
        for i in [0, m-1]:
            for j in range(n):
                if grid[i][j] == 0:
                    dfs(i,j,m,n,grid)
        for j in [0, n-1]:
            for i in range(m):
                if grid[i][j] == 0:
                    dfs(i,j,m,n,grid)
        return self.numIslands(grid)
