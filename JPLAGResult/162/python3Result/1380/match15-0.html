class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        
        g = grid
        n, m = len(grid), len(grid[0])
        
        def dfs(u, v, t):
            
            nonlocal n
            nonlocal m
            
            if g[u][v] == 0:
                g[u][v] = t
            else:
                return
            
            for x0, y0 in [(-1, 0), (0, 1), (1, 0), (0, -1)]:
                
                x, y = u + x0, v + y0
                
                if 0 <= x < n and 0 <= y < m:
                    if g[x][y] == 0:
                        dfs(x, y, t)
                        
        
        for i in range(n):
            dfs(i, 0, -1)
            dfs(i, m - 1, -1)
            
        for j in range(m):
            dfs(0, j, -1)
            dfs(n - 1, j, -1)
            
        ans = 0
        
        for i in range(1, n - 1):
            for j in range(1, m - 1):
                if g[i][j] == 0:
                    ans += 1
                    dfs(i, j, 2)
                    
        return ans
