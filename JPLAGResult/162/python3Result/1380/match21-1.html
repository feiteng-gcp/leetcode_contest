class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        M, N = len(grid), len(grid[0])
        def mark(i: int, j: int, label: int) -> None:
            if i < 0 or i >= M or j < 0 or j >= N or grid[i][j] != 0:
                return 
            grid[i][j] = label
            for di, dj in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                mark(i + di, j + dj, label)
        for i in [0, M - 1]:
            for j in range(N):
                mark(i, j, 1)
        for i in range(M):
            for j in [0, N - 1]:
                mark(
i, j, 1)
        label = 2
        for i in range(M):
            for j in range(N):
                if grid[i][j] == 0:
                    mark(i, j, label)
                    label += 1
        return label - 2
