class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        M, N = len(grid), len(grid[0])
        visited = [[False] * N for _ in range(M)]
        directs = [(0, 1), (1, 0), (-1, 0), (0, -1)]
        def dfs(r, c):
            if visited[r][c] or grid[r][c] == 1:
                return False
            visited[r][c] = True
            surround = True
            for dr, dc in directs:
                nr, nc = r + dr, c + dc
                if nr < 0 or nr == M or nc < 0 or nc == N:
                    surround = False
                elif not visited[nr][nc] and grid[nr][nc] == 0:
                    surround &= dfs(nr, nc)
            return surround
        total = 0
        for r in range(M):
            for c in range(N):
                if dfs(r, c):
                    total += 1
        return total
