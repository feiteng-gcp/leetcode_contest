class Solution:
    def closedIsland(self, g: List[List[int]]) -> int:
        
        
        
        
        moves = [[1,0],[-1,0],[0,1],[0,-1]]
        n = len(g)
        m = len(g[0])
        def dfs(x,y):
            
            g[x][y] = 1
            for i, j in moves:
                nx = x + i
                ny = y + j
                if 0<=nx<n and 0<=ny<m and g[nx][ny] == 0:
                    dfs(nx,ny)
        
        for i in range(n):
            for j in range(m):
                if i == 0 or i == n-1 or j == 0 or j == m-1:
                    if g[i][j] == 0:
                        dfs(i,j)
        cnt = 0
        for i in range(n):
            for j in range(m):
                if g[i][j] == 0:
                    dfs(i,j)
                    cnt += 1
        return cnt
