class Solution:
    def closedIsland(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        dx, dy = [0,0,-1,1], [1,-1,0,0]
        def dfs(x, y):
            if grid[x][y] == 1:
                return 
            grid[x][y] = 1
            for i in range(4):
                nx,ny = x + dx[i], y + dy[i]
                if 0<=nx<m and 0<=ny<n:
                    dfs(
nx, ny)
        for i in range(n):
            dfs(0, i)
            dfs(m-1, i)
        for i in range(m):
            dfs(i, 0)
            dfs(i, n-1)
        res = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 0:
                    res += 1
                    dfs(i,j)
        return res
