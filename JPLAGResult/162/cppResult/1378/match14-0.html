class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<int> r(n, 0), c(m, 0);
        for (const auto& ind: indices) {
            r[ind[0]]++;
            c[ind[1]]++;
        }
        
        int cnt = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if ((r[i]+c[j]) % 2 == 1) cnt++;
            }
        }
        return cnt;
    }
};
