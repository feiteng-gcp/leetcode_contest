class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<vector<int>> vp(n, vector<int>(m));
        for(auto & pair:indices) {
            for(int i=0; i<m; i++)
                vp[pair[0]][i]+=1;
            for(int j=0; j<n; j++)
                vp[j][pair[1]]+=1;
        }
        int total = 0;
        for(auto & x:vp)
            for(auto & y:x)
                if(y&1)
                    total++;
        return total;
    }
};
