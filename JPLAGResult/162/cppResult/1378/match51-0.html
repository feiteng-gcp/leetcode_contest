class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<int> vRow(n);
        vector<int> vCol(m);
        for(auto ind:indices){
            vRow[ind[0]]++;
            vCol[ind[1]]++;
        }
        int ans=0;
        
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if((vRow[i]+vCol[j])&1!=0) ans++;
            }
        }
        
        return ans;

    }
};
