class Solution {
public:
    int oddCells(int m, int n, vector<vector<int>>& indices) {
        int cnt = 0;
        vector<vector<int>> t(m, vector<int> (n, 0));
        for(auto& v : indices) {
            int x = v[0];
            int y = v[1];
            for(int i = 0; i < n; ++i) {
                t[x][i]++;
            }
            for(int i = 0; i < m; ++i) {
                t[i][y]++;
            }
        }
        for(auto& row : t) {
            for(auto& c : row) {
                cnt += c%2;
            }
        }
        return cnt;
        
    }
};
