class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        
        vector<vector<int>> a(n, vector<int>(m, 0));
        
        for (auto ind : indices)
        {
            for (int i = 0; i < m; i++)
                a[ind[0]][i] += 1;
            for (int i = 0; i < n; i++)
                a[i][ind[1]] += 1;            
        }
        
        int ans = 0;
        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++)
                if (a[i][j] % 2 == 1)
                    ans += 1;
        
        
        return ans;
        
    }
};
