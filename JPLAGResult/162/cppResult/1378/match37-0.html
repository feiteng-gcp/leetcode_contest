class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<vector<int>> v(n, vector<int>(m, 0));
        for(vector<int>& t : indices){
            for(int h=0;h<n;h++)
                v[h][t[1]]++;
            for(int w=0;w<m;w++)
                v[t[0]][w]++;
        }
        
        int result = 0;
        for(int h=0;h<n;h++){
            for(int w=0;w<m;w++){
                result += (v[h][w]%2);
            }
        }
        
        return result;
    }
};
