class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<vector<int>> arr(n, vector<int>(m, 0));
        for (auto i: indices) {
            int r = i[0], c = i[1];
            for (int j = 0; j < m; j++) {
                arr[r][j] ++;
            }
            for (int j = 0; j < n; j++) {
                arr[j][c] ++;
            }
        }
        
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j ++) {
                if (arr[i][j] % 2 == 1) res ++;
            }
        }
        
        return res;
    }
};
