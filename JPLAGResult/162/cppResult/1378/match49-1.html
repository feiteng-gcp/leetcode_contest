class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
     //   n = 2, m = 2, indices = {{1, 1}, {0, 0}};
        vector<vector<int>> matrix(n, vector<int>(m, 0));
        
        for (auto &index: indices) {
            for (int j = 0; j < m; j++) matrix[index[0]][j]++;

            for (int i = 0; i < n; i++) matrix[i][index[1]]++;
        }
        
        int res = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (matrix[i][j] % 2 == 1)
                    res++;
            }
        }
        
        return res;
        
    }
};
