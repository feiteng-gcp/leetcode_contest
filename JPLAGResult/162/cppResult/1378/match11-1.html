class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<vector<int>> mat(n, vector<int>(m));
        for (auto i : indices) {
            int x = i[0], y = i[1];
            for (int j = 0; j < m; ++j) {
                ++mat[x][j];
            }
            for (int j = 0; j < n; ++j) {
                ++mat[j][y];
            }
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                ans += (mat[i][j] % 2);
            }
        }
        return ans;
    }
};
