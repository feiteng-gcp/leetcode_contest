class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<int> rows(n, 0), cols(m, 0);
        for (auto & v:indices) {
            rows[v[0]]++; cols[v[1]]++;
        }
        int count = 0;
        for (int i=0; i<n; i++)
            for (int j=0; j<m; j++)
                if ((rows[i] + cols[j]) % 2 == 1) count += 1;
        return count;
    }
};
