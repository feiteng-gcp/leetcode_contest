class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<vector<int>> matrix(n, vector<int>(m));
        for (auto & indice : indices) {
            for (int i = 0; i < m; i++) {
                matrix[indice[0]][i] += 1;
            }
            for (int i = 0; i < n; i++) {
                matrix[i][indice[1]] += 1;
            }
        }
        int count = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (matrix[i][j]%2) {
                    count++;
                }
            }
        }
        return count;
    }
};
