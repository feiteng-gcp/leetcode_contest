class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<vector<int>> data(n, vector<int>(m, 0));
        for (auto& indice : indices) {
            for (int i = 0;i < n;i++) {
                data[i][indice[1]]++;
            }
            for (int i = 0;i < m;i++) {
                data[indice[0]][i]++;
            }
        }
        int ret = 0;
        for (int i = 0;i < n;i++) {
            for (int j = 0;j < m;j++) {
                if (data[i][j] & 1) {
                    ret++;
                }
            }
        }
        return ret;
    }
};
