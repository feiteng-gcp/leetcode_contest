class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& i) {
        vector<int> r(n, 0), c(m, 0);
        for (auto v : i) {
            r[v[0]]++;
            c[v[1]]++;
        }
        int res = 0;
        for (int x = 0; x < n; ++x) for (int y = 0; y < m; ++y) if ((r[x] + c[y]) % 2) ++res;
        return res;
    }
};
