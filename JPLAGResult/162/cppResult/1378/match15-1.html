class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<int> x(n,0);
        vector<int> y(m,0);
        
        for(int i = 0; i < indices.size(); i++) {
            x[indices[i][0]]++;
            y[indices[i][1]]++;
        }
        int result = 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if((x[i]+y[j])%2==1) {
                    result++;
                }
            }
        }
        return result;
    }
};
