class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& b) {
        vector<vector<int>> a(n, vector<int>(m));
        for (auto& it : b) {
            int x = it[0], y = it[1];
            for (int j = 0; j < m; ++j) a[x][j]++;
            for (int i = 0; i < n; ++i) a[i][y]++;
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                ret += a[i][j] % 2;
            }
        }
        return ret;
    }
};
