class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<vector<int>> M(n, vector<int>(m, 0));
        
        for (auto &ind : indices) {
            for (int i = 0 ; i < m ; ++i)
                M[ind[0]][i]++;

            for (int i = 0 ; i < n ; ++i)
                M[i][ind[1]]++;            
        }
        
        int rtn = 0;
        for (int i = 0 ; i < n ; ++i) {
            for (int j = 0 ; j < m ; ++j) {
                if (M[i][j] % 2) rtn++;
            }
        }
        
        return rtn;
    }
};
