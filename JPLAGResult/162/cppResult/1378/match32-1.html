class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<vector<int>> mat(n,vector<int>(m,0));    
        for(auto x : indices){

            for(int i=0; i<m; i++){
                mat[x[0]][i]++;
            }
            for(int i=0; i<n; i++){
                mat[i][x[1]]++;
            }
        }
        
        int cnt=0;
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                cnt+=(mat[i][j]&1);
            }
        }
        
        return cnt;
    }
};
