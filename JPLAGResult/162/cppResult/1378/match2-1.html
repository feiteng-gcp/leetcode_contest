class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<vector<int>> matrix(n,vector<int>(m,0));
        for(auto v:indices){
            int row=v[0],col=v[1];
            for(int i=0;i<m;++i) matrix[row][i]++;
            for(int i=0;i<n;++i) matrix[i][col]++;
        }
        int count=0;
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                if(matrix[i][j]&1) ++count;
            }
        }
        return count;
    }
};
