class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        
        vector<vector<int>> A(n, vector<int>(m));
        
        for(auto& p : indices) {
            int r = p[0], c = p[1];
            for(int j = 0; j < m; j++) {
                ++A[r][j];
            }
            
            for(int i = 0; i < n; i++) {
                ++A[i][c];
            }
        }
        
        int ans = 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                ans += A[i][j] % 2;
            }
        }
        
        return ans;
        
    }
};
