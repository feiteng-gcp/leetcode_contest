class Solution {
public:
    int oddCells(int n, int m, vector<vector<int>>& indices) {
        vector<vector<int>> v(n,vector<int>(m,0));
        for(auto &e : indices){
            int r = e[0], c = e[1];
            for(int i = 0;i < m;i++) v[r][i]++;
            for(int i = 0;i < n;i++) v[i][c]++;
        }
        int cnt = 0;
        for(int i = 0;i < n;i++)
            for(int j = 0;j < m;j++)
                if(v[i][j] & 1)
                    cnt++;
        return cnt;
    }
};
