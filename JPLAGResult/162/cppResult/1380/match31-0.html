class Solution {
public:
    int dx[4] = {0, 1, 0, -1};
    int dy[4] = {1, 0, -1, 0};
    
    int closedIsland(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        
        for (int i = 0; i < n; i++) {
            DFS(i, 0, n, m, grid);
            DFS(i, m - 1, n, m, grid);
        }
        for (int j = 0; j < m; j++) {
            DFS(0, j, n, m, grid);
            DFS(n - 1, j, n, m, grid);
        }
        
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 0) {
                    ans++;
                    DFS(i, j, n, m, grid);
                }
            }
        }
        return ans;
    }
    
private:
    int is_in(int x, int y, int n, int m) {
        return 0 <= x && x < n && 0 <= y && y < m;
    }
    
    void DFS(int x, int y, int n, int m, vector<vector<int>>& grid) {
        if (grid[x][y] == 1) return;
        
        grid[x][y] = 2;
        for (int i = 0; i < 4; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            
            if (is_in(nx, ny, n, m) && grid[nx][ny] == 0) {
                DFS(nx, ny, n, m, grid);
            }  
        }
    }
};
