class Solution {
public:
    int m, n;
    int closedIsland(vector<vector<int>>& grid) {
        int rtn = 0;
        
        m = grid.size(), n = grid[0].size();
        for (int i = 0 ; i < m ; ++i) {
            for (int j = 0 ; j < n ; ++j) {
                if (grid[i][j] == 0) {
                    bool valid = true;
                    dfs(grid, i, j, valid);                                    
                    if (valid)
                        rtn++;
                }
            }
        }
        return rtn;
    }
    
    void dfs(vector<vector<int>>& grid, int x, int y, bool &valid) {
        if (x < 0 || x >= m || y < 0 || y >= n) {
            valid = false;
            return;
        }
        
        if (grid[x][y] == 1 || grid[x][y] == 2)
            return;
        
        grid[x][y] = 2;
        dfs(grid, x-1, y, valid);
        dfs(grid, x+1, y, valid);
        dfs(grid, x, y-1, valid);
        dfs(grid, x, y+1, valid);        
    }
};
