template<typename S, typename T>
ostream& operator<<(ostream& out, const pair<S, T> p) {
    out << "(" << p.first << "," << p.second << ")";
    return out;
}

ostream& operator<<(ostream& out, const vector<T>& v) {
    for (auto a: v)
        out << a << " ";
    return out;
}

template<typename S, typename T>
pair<S, T> operator+(pair<S, T> a, pair<S, T> b) {
    return make_pair(a.first + b.first, a.second + b.second);
}

template<typename S, typename T>
pair<S, T> operator-(pair<S, T> a, pair<S, T> b) {
    return make_pair(a.first - b.first, a.second - b.second);
}

class Solution {
public:
    vector<pair<int, int>> movs = {{0,1},{-1,0},{0,-1},{1,0}};
    vector<vector<int>> vis;
    void dfs(vector<vector<int>>& grid, pair<int,int> s) {
        int n = grid.size();
        int m = grid[0].size();
        vis[s.first][s.second] = 1;
        for (auto mov: movs) {
            auto t = s + mov;
            if (t.first >= 0 && t.first < n && t.second >= 0 && t.second < m) {
                if (grid[t.first][t.second] == 0 && !vis[t.first][t.second]) {
                    dfs(grid, t);
                }
            }
        }
    }
    int closedIsland(vector<vector<int>>& grid) {
        int n = grid.size();
        if (n == 0)
            return 0;
        int m = grid[0].size();
        if (m == 0)
            return 0;
        vis.resize(n);
        for (int i = 0; i < n; i++)
            vis[i].assign(m, 0);

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (i == 0 || i == n - 1 || j == 0 || j == m - 1) {
                    if (grid[i][j] == 0 && !vis[i][j]) {
                        dfs(grid, make_pair(i, j));
                    }

                }
            }
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 0 && !vis[i][j]) {
                    dfs(grid, make_pair(i, j));
                    ans++;
                }
            }
        }
        return ans;
    }
};
