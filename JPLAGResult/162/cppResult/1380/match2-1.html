class Solution {
public:
    int closedIsland(vector<vector<int>>& grid) 
    {
        int m = grid.size();
        int n = grid[0].size();
                
        for (int j=0; j<n; j++)
            if (grid[0][j]==0) 
                bfs(grid,0,j);
        
        for (int j=0; j<n; j++)
            if (grid[m-1][j]==0) 
                bfs(grid,m-1,j);
        
        for (int i=0; i<m; i++)
            if (grid[i][0]==0) 
                bfs(grid,i,0);
        
        for (int i=0; i<m; i++)
            if (grid[i][n-1]==0) 
                bfs(grid,i,n-1);
        
        int count = 0;
        for (int i=0; i<m; i++)
            for (int j=0; j<n; j++)
            {
                if (grid[i][j]!=0) continue;
                cout<<i<<" "<<j<<endl;
                bfs(grid, i, j);
                count++;
            }
        return count;
    }
    
    
    void bfs(vector<vector<int>>& grid, int x0, int y0)
    {
        if (grid[x0][y0]!=0) return;
        queue<pair<int,int>>q;
        q.push({x0,y0});
        grid[x0][y0] = 1;
        
        auto dir = vector<pair<int,int>>({{0,1},{0,-1},{1,0},{-1,0}});
        int m = grid.size();
        int n = grid[0].size();
        
        while (!q.empty())
        {
            int x = q.front().first;
            int y = q.front().second;
            q.pop();
            
            for (int k=0; k<4; k++)
            {
                int i = x+dir[k].first;
                int j = y+dir[k].second;
                if (i<0||i>=m||j<0||j>=n) continue;
                if (grid[i][j]!=0) continue;
                
                grid[i][j] = 1;
                q.push({i,j});
            }
        }
    }
    
};
