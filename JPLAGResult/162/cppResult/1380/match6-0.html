class Solution {
public:
    int closedIsland(vector<vector<int>>& grid) {
        int islands = 0;
        for(int i = 0; i < grid.size(); i++){
            for(int j = 0; j < grid[0].size(); j++){
                if(grid[i][j] == 0 && isClosed(grid, i, j)){
                    islands++;
                }
            }
        }
        return islands;
    }
    bool isClosed(vector<vector<int>>& grid, int i, int j){
        grid[i][j] = -1;
        bool isValid = true;
        if(i == 0 || (grid[i - 1][j] == 0 && !isClosed(grid, i - 1, j))){
            isValid = false;
        }
        if(j == 0 || (grid[i][j - 1] == 0 && !isClosed(grid, i, j - 1))){
            isValid = false;
        }
        if(i == grid.size() - 1 || (grid[i + 1][j] == 0 && !isClosed(grid, i + 1, j))){
            isValid = false;
        }
        if(j == grid[0].size() - 1 || (grid[i][j + 1] == 0 && !isClosed(grid, i, j + 1))){
            isValid = false;
        }
        return isValid;
    }
};
