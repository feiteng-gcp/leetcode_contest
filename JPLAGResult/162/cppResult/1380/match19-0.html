class Solution {
    int n, m;
public:
    int closedIsland(vector<vector<int>>& grid) {
        n = grid.size(); m = grid[0].size();
        for ( int i = 0; i < n; ++i ) {
            dfs(grid, i, 0);
            dfs(grid, i, m -1);
        }
        for ( int j = 0; j < m; ++j ) {
            dfs(grid, 0, j);
            dfs(grid, n - 1, j);
        }
        int res = 0;
        for ( int i = 0; i < n; ++i ) {
            for ( int j = 0; j < m; ++j ) {
                if ( grid[i][j] == 0 ) {
                    ++res;
                    dfs(grid, i, j);
                }
            }
        }
        return res;
        
    }
    
    void dfs(vector<vector<int>>& grid, int i, int j) {
        if ( i < 0 || i >= n || j < 0 || j >= m || grid[i][j] ) return;
        grid[i][j] = 1;
        dfs(grid, i + 1, j);
        dfs(grid, i - 1, j);
        dfs(grid, i, j + 1);
        dfs(grid, i, j - 1);
    }
};
