class Solution {
public:
    void closedIsland(int x, int y, vector<vector<int>>& grid) {
		if (x >= 0 && x < grid.size() && y >= 0 && y < grid[0].size() && grid[x][y] == 0) {
			grid[x][y] = 1;
			closedIsland(x - 1, y, grid);
			closedIsland(x, y - 1, grid);
			closedIsland(x, y + 1, grid);
			closedIsland(x + 1, y, grid);
		}

	}
    
    int closedIsland(vector<vector<int>>& grid) {
        for (int i = 0; i < grid.size(); i++) {
			for (int j = 0; j < grid[0].size(); j++) {
				if (i == 0 || j == 0 || i == grid.size() - 1 || j == grid[0].size() - 1) {
					closedIsland(i, j, grid);
				}
			}
		}
        int count = 0;
		for (int i = 0; i < grid.size(); i++) {
			for (int j = 0; j < grid[0].size(); j++) {
				if (grid[i][j] == 0) {
					closedIsland(i, j, grid);
					count++;
				}
			}
		}
		return count;
    }
};
