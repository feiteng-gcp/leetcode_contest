class Solution {
public:
    const int dx[4] = {1, 0, -1, 0};
    const int dy[4] = {0, 1, 0, -1};
    bool dfs(vector<vector<int>> &grid, int x, int y) {
        bool res = true;
        for(int i = 0; i < 4; ++i) {
            const int xx = x + dx[i];
            const int yy = y + dy[i];
            if(xx < 0 || xx >= grid.size() || yy < 0 || yy >= grid[0].size()) {
                res = false;
            } else if(0 == grid[xx][yy]){
                grid[xx][yy] = 2;
                res &= dfs(grid, xx, yy);
            }
        }
        return res;
    }
    int closedIsland(vector<vector<int>>& grid) {
        int res = 0;
        for(int i = 0; i < grid.size(); ++i) {
            for(int j = 0; j < grid[i].size(); ++j) {
                if(0 == grid[i][j] && dfs(grid, i, j)) {
                    cout << i << ", " << j << endl;
                    ++res;
                    /*
                    for(int ii = 0; ii < grid.size(); ++ii) {
                        for(int jj = 0; jj < grid[ii].size(); ++jj) {
                            cout << grid[ii][jj] << " ";
                        }
                        cout << endl;
                    }
                    */
                }
            }
        }
        return res;
    }
};
