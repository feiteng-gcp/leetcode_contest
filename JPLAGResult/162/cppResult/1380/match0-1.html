class Solution {
public:
    int DIR[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

    bool is_inland(vector<vector<int>> &grid, int r, int c) {
        for (int i = 0; i < 4; i++) {
            int dr = r + DIR[i][0];
            int dc = c + DIR[i][1];
            if (dr < 0 || dr >= grid.size() || dc < 0 || dc >= grid[0].size()) {
                return false;
            }
        }
        return true;
    }

    int closedIsland(vector<vector<int>>& grid) {
        int row = grid.size();
        int col = grid[0].size();
        if(row <= 2 || col <= 2) {
            return 0;
        }
        int res = 0;
        queue<pair<int, int>> q;
        for (int r = 0; r < row; r++) {
            for (int c = 0; c < col; c++) {
                if (grid[r][c] == 0) {
                    bool inland = true;
                    grid[r][c] = -1;
                    q.push({r, c});
                    while (!q.empty()) {
                        auto p = q.front();
                        q.pop();
                        int tmp_r = p.first;
                        int tmp_c = p.second;
                        if (!is_inland(grid, tmp_r, tmp_c)) {
                            inland = false;
                        }
                        for (int i = 0; i < 4; i++) {
                            int dr = tmp_r + DIR[i][0];
                            int dc = tmp_c + DIR[i][1];
                            if (dr < 0 || dr >= grid.size() || dc < 0 || dc >= grid[0].size() || grid[dr][dc] != 0) {
                                continue;
                            }
                            grid[dr][dc] = -1;
                            q.push({dr, dc});
                        }
                    }
                    if (inland) {
                        res++;
                    }
                }
            }
        }
        return res;
    }
};
