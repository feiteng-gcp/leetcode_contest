class Solution {
public:
    
    void helper(vector<vector<int>>& grid, vector<vector<bool>> &visited, int i, int j, int n, int m, bool &flag) {
        if (i < 0 || j < 0 || i >= n || j >= m) {
            flag = false;
            return;
        }
        if (grid[i][j] == 1 || visited[i][j]) return;
        visited[i][j] = true;
        helper(grid, visited, i + 1, j, n, m, flag);
        helper(grid, visited, i, j + 1, n, m, flag);
        helper(grid, visited, i - 1, j, n, m, flag);
        helper(grid, visited, i, j - 1, n, m, flag);
    }
    
    int closedIsland(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size(), res = 0;
        bool flag = true;
        vector<vector<bool>> visited(n, vector<bool>(m, false));
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == 0 && !visited[i][j]) {
                    flag = true;
                    helper(grid, visited, i, j, n, m, flag);
                    if(flag)res++;
                }
            }
        }
        
        return res;
    }
};
