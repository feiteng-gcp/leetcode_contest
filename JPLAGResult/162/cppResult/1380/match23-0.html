class Solution {
public:
    int dirs[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    bool dfs(vector<vector<int>>& grid, int i, int j, int R, int C) {
        grid[i][j] = 2;
        bool valid = true;
        for (int k = 0; k < 4; ++k) {
            int r = i + dirs[k][0];
            int c = j + dirs[k][1];
            if (r >= 0 && r < R && c >= 0 && c < C) {
                if (grid[r][c] == 0 && !dfs(grid, r, c, R, C)) {
                    valid = false;
                }
            } else {
                valid = false;
            }
        }
        return valid;
    }
    int closedIsland(vector<vector<int>>& grid) {
        if (grid.empty()) return 0;
        int R = grid.size();
        int C = grid[0].size();
        int res = 0;
        for (int i = 0; i < R; ++i) {
            for (int j = 0; j < C; ++j) {
                if (grid[i][j] == 0) {
                    res += dfs(grid, i, j, R, C);
                }
            }
        }
        return res;
    }
};
