class Solution {
public:
    int m, n;
    vector<vector<int>> visit;
    bool is (int x, int y) {
        return x >= 0 && x < m && y >= 0 && y < n;
    }
    
    void find(int x, int y, vector<vector<int>>& grid) {
        if (!is(x, y) || visit[x][y] || grid[x][y] == 1) return;
        //cout << x << " " << y << endl;;
        
        visit[x][y] = 1;
        find(x + 1, y, grid);
        find(x - 1, y, grid);
        find(x, y + 1, grid);
        find(x, y - 1, grid);
    }
    
    
    int closedIsland(vector<vector<int>>& grid) {
        m = grid.size();
        n = grid[0].size();
        
        visit = vector(m, vector<int>(n, 0));
        
        for (int i = 0; i < m; i ++) {
            find(i, 0, grid);
            find(i, n - 1, grid);
        }
        for (int j = 0; j < n; j ++) {
            find(0, j, grid);
            find(m-1, j, grid);
        }
        
        int res = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 0 && visit[i][j] == 0) {
                    res ++;
                    find(i, j, grid);
                }
            }
        }
        return res;
        
    }
};
