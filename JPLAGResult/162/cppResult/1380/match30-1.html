class Solution {
public:
    int closedIsland(vector<vector<int>>& grid) 
    {
        int m = grid.size();
        if (m == 0)
        {
            return 0;
        }
        int n = grid[0].size();
        if (n <= 2)
        {
            return 0;
        }
        
        //search corner
        //up down
        for (int i = 0; i < n; i++)
        {
            if (grid[0][i] == 0)
            {
                search(grid, 0, i);
            }
            
            if (grid[m-1][i] == 0)
            {
                search(grid, m-1, i);
            }
        }
        
        //left right
        for (int i = 0; i < m; i++)
        {
            if (grid[i][0] == 0)
            {
                search(grid, i, 0);
            }
            
            if (grid[i][n-1] == 0)
            {
                search(grid, i, n - 1);
            }
        }
        
        

        int result = 0;
        for (int i = 1; i+1 < m; i++)
        {
            for (int j = 1; j + 1 < n; j++)
            {
                if (grid[i][j] == 0)
                {
                    result ++;
                    search(grid, i, j);
                }
            }
        }
        return result;
    }
    
    void search(vector<vector<int>> &grid, int x, int y)
    {
        int m = grid.size();
        int n = grid[0].size();
        vector<int> xVec;
        vector<int> yVec;
        xVec.push_back(x);
        yVec.push_back(y);
        grid[x][y] = 1;
        while (xVec.size() > 0)
        {
            int tx = xVec[0];

            int ty = yVec[0];
            xVec.erase(xVec.begin());
            yVec.erase(yVec.begin());
            
            //up
            if (tx > 0 && grid[tx-1][ty] == 0)
            {
                grid[tx-1][ty] = 1;
                xVec.push_back(tx - 1);
                yVec.push_back(ty);
            }
            
            //down
            if (tx + 1 < m && grid[tx+1][ty] == 0)
            {
                grid[tx+1][ty] = 1;
                xVec.push_back(tx + 1);
                yVec.push_back(ty);
            }
            
            //left
            if (ty > 0 && grid[tx][ty-1] == 0)
            {
                grid[tx][ty-1] = 1;
                xVec.push_back(tx);
                yVec.push_back(ty - 1);
            }
            
            //right
            if (ty + 1 < n && grid[tx][ty+1] == 0)
            {
                grid[tx][ty+1] = 1;
                xVec.push_back(tx);
                yVec.push_back(ty + 1);
            }
        }
    }
};
