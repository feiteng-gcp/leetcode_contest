class Solution {
public:
    
    bool vis[105][105];
    

    bool dfs(vector<vector<int>>& A, int x, int y, int n, int m){
        bool f=1;
        vis[x][y] = 1;
        if(x+1<n and A[x+1][y] == 0 and vis[x+1][y] == 0){
            f = f & dfs(A,x+1,y,n,m);
        }
        if(x-1>=0 and A[x-1][y] == 0 and vis[x-1][y] == 0){
            f = f & dfs(A,x-1,y,n,m);
        }
        if(y+1<m and A[x][y+1] == 0 and vis[x][y+1] == 0){
            f = f & dfs(A,x,y+1,n,m);
        }
        if(y-1>=0 and A[x][y-1] == 0 and vis[x][y-1] == 0){
            f = f & dfs(A,x,y-1,n,m);
        }
        if(x==0 || x==n-1 || y==0 || y==m-1){
            f = 0;
        }
        return f;
    }
    
    int closedIsland(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size(); 
        // for(int i=0;i<n;i++)   for(int j=0;i<m;j++)    vis[i][j]=0;

        int cnt=0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]==0 and vis[i][j]==0){
                    if(dfs(grid,i,j,n,m))    cnt++;
                }
            }
        }
        return cnt;
    }
};
