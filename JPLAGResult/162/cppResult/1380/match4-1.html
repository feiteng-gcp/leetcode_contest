class Solution {
public:
    int directions[4][2]={{-1,0},{1,0},{0,-1},{0,1}};
    int closedIsland(vector<vector<int>>& grid) {

        int n=grid.size(),m=grid[0].size();
        int ans=0;
        for(int c=0;c<m;++c){
            if(grid[0][c]==0) dfs(grid,0,c);
            if(grid[n-1][c]==0) dfs(grid,n-1,c);
        }
        for(int r=0;r<n;++r){
            if(grid[r][0]==0) dfs(grid,r,0);
            if(grid[r][m-1]==0) dfs(grid,r,m-1);
        }
        for(int r=1;r<n-1;++r){
            for(int c=1;c<m-1;++c){
                if(grid[r][c]==0){
                    ++ans;
                    dfs(grid,r,c);
                }
            }
        }
        return ans;
    }
    void dfs(vector<vector<int>>& grid,int row,int col){
        if(0<=row&&row<grid.size() && 0<=col&&col<grid[0].size() && grid[row][col]==0){
            grid[row][col]=1;
            for(int i=0;i<4;++i){
                int nr=row+directions[i][0],nc=col+directions[i][1];
                dfs(grid,nr,nc);
            }
        }
    }
};
