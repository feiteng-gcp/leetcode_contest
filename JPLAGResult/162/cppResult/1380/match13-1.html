class Solution {
public:
    int closedIsland(vector<vector<int>>& board) {
        if(board.size() == 0 || board[0].size() == 0) return 0;

        int rows = board.size(), cols = board[0].size();
        for(int i = 0; i < rows; ++i)
        {
            dfs(board, i, 0);
            dfs(board, i, cols - 1);
        }
        
        for(int j = 0; j < cols; ++j)
        {
            dfs(board, 0, j);
            dfs(board, rows - 1, j);
        }
        
        int res = 0;
        for(int i = 0; i < rows; ++i)
        {
            for(int j = 0; j < cols; ++j)
            {
                if(board[i][j] == 0)
                {
                    res++;
                    dfs(board, i, j);
                }
            }
        }
        return res;
    }
    
    void dfs(vector<vector<int>>& board, int i, int j)
    {
        if(i < 0 || i >= board.size() || j < 0 || j >= board[0].size() || board[i][j] != 0)
        {
            return;
        }
        board[i][j] = 2;
        dfs(board, i + 1, j);
        dfs(board, i - 1, j);
        dfs(board, i, j + 1);
        dfs(board, i, j - 1);
    }
    
};
