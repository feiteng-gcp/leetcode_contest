class Solution {
public:
    int closedIsland(vector<vector<int>>& grid) {
        int total = 0;
        for (int i = 0; i < grid.size(); ++i)
        {
            for (int j = 0; j < grid[i].size(); ++j)
            {
                if (grid[i][j] == 0 && isIsland(grid, i, j))
                {
                    total++;
                }
            }
        }
        
        return total;
    }

private:
    bool isIsland(vector<vector<int>>& grid, int r, int c)
    {
        bool result = r > 0 && r + 1 < grid.size() && c > 0 && c + 1 < grid[r].size();
        grid[r][c] = 1;
        if (r > 0 && grid[r - 1][c] == 0)
        {
            result &= isIsland(grid, r - 1, c);
        }
        if (r + 1 < grid.size() && grid[r + 1][c] == 0)
        {
            result &= isIsland(grid, r + 1, c);
        }
        if (c > 0 && grid[r][c - 1] == 0)
        {
            result &= isIsland(grid, r, c - 1);
        }
        if (c + 1 < grid[r].size() && grid[r][c + 1] == 0)
        {
            result &= isIsland(grid, r, c + 1);
        }
        return result;
    }
};
