class Solution {
public:
    int mark[110][110] = {0};
    int m;
    int n;
    int closedIsland(vector<vector<int>>& grid) {
        m = grid.size();
        n = grid[0].size();
        int ans = 0;
        for (int i=0;i<m;i++)
        {
            if (!grid[i][0] && !mark[i][0])
                dfs(i, 0, grid);
            if (!grid[i][n-1] && !mark[i][n-1])
                dfs(i, n-1, grid);
        }
        for (int i=0;i<n;i++)
        {
            if (!grid[0][i] && !mark[0][i])
                dfs(0, i, grid);
            if (!grid[m-1][i] && !mark[m-1][i])
                dfs(m-1, i, grid);
        }
        for (int i=0;i<m;i++)
        {
            for (int j=0;j<n;j++)
            {
                if (!grid[i][j] && !mark[i][j])
                {
                    dfs(i, j, grid);
                    ++ans;
                }
            }
        }
        return ans;
    }
    void dfs(int x, int y, vector<vector<int>>& grid)
    {
        if (x < 0 || y < 0 || x >= m || y >= n || mark[x][y] || grid[x][y])
            return;
        mark[x][y] = 1;
        dfs(x, y-1, grid);
        dfs(x-1, y, grid);
        dfs(x+1, y, grid);
        dfs(x, y+1, grid);
    }
};
