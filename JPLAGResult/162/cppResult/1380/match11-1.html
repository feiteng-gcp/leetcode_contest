class Solution {
public:
    vector<pair<int,int>> dirs = {{1,0}, {-1,0}, {0, 1}, {0, -1}};
    
    int closedIsland(vector<vector<int>>& grid) {
        int m = grid.size(), n = m == 0 ? 0 : grid[0].size();
        if(m == 0)
            return 0;
        for(int i = 0; i < m; ++i) {
            if(grid[i][0] == 0) {
                flood(grid, i, 0, 1);
            }
            if(grid[i][n-1] == 0) {
                flood(grid, i, n-1, 1);
            }
        }
        for(int j = 0; j < n; ++j) {
            if(grid[0][j] == 0) {
                flood(grid, 0, j, 1);
            }
            if(grid[m-1][j] == 0) {
                flood(grid, m-1, j, 1);
            }
        }
        // print(grid);
        
        int count = 0;
        int color = 2;
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if(grid[i][j] == 0) {
                    ++count;
                    flood(grid, i, j, color++);
                }
            }
        }
        return count;
    }
    
    void flood(vector<vector<int>>& grid, int i, int j, int color) {
        int m = grid.size(), n = m == 0 ? 0 : grid[0].size();
        if(i < 0 || j < 0 || i >= m || j >= n || grid[i][j])
            return;
        grid[i][j] = color;
        flood(grid, i+1, j, color);
        flood(grid, i-1, j, color);
        flood(grid, i, j-1, color);
        flood(grid, i, j+1, color);
    }
    
    
};
