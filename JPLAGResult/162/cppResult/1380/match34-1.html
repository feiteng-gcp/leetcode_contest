bool visited[110][110];
int dirx[4] = {0,1,0,-1};
int diry[4] = {1,0,-1,0};
class Solution {
public:
    void dfs(int r, int c, vector<vector<int>>& grid){
        if(visited[r][c])
            return;
        visited[r][c] = true;
        int n = grid.size();
        int m = grid[0].size();
        for(int i=0;i<4;i++){
            int nextr = r+dirx[i];
            int nextc = c+diry[i];
            if(nextr>=0&&nextr<n&&nextc>=0&&nextc<m&&grid[nextr][nextc]==0){
                dfs(nextr, nextc, grid);
            }
        }
    }
    int closedIsland(vector<vector<int>>& grid) {
        memset(visited, 0, sizeof visited);

        int n = grid.size();
        int m = grid[0].size();
        for(int i=0;i<m;i++){
            if(grid[0][i]==0){
                dfs(0,i, grid);
                
            }
            if(grid[n-1][i]==0){
                dfs(n-1, i, grid);
            }
        }
        for(int i=0;i<n;i++){
            if(grid[i][0]==0){
                dfs(i,0, grid);
            }
            if(grid[i][m-1]==0){
                dfs(i, m-1, grid);
            }
        }
        int ans = 0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(grid[i][j]==0&&!visited[i][j]){
                    ans++;
                    dfs(i,j, grid);
                }
            }
        }
        return ans;
        
    }
};
