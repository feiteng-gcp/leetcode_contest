class Solution {
public:
    int closedIsland(vector<vector<int>>& grid) {

        int M = grid.size(), N = grid[0].size();

        for(int i = 0; i < M; ++i) {
            if(grid[i][0] == 0) {
                sink(grid, i, 0, M, N);
            }

            if(grid[i][N - 1] == 0) {
                sink(grid, i, N - 1, M, N);
            }
        }

        for(int j = 0; j < N; ++j) {
            if(grid[0][j] == 0) {
                sink(grid, 0, j, M, N);
            }

            if(grid[M - 1][j] == 0) {
                sink(grid, M - 1, j, M, N);
            }
        }

        int res = 0;

        for(int i = 0; i < M; ++i) {
            for(int j = 0; j < N; ++j) {
                if(grid[i][j] == 0) {
                    ++res;
                    sink(grid, i, j, M, N);
                }
            }
        }

        return res;
    }

private:
    void sink(vector<vector<int>>& grid, int x, int y, int M, int N) {
        if(x < 0 || x == M || y < 0 || y == N || grid[x][y] == 1) {
            return;
        }

        grid[x][y] = 1;
        sink(grid, x - 1, y, M , N);
        sink(grid, x + 1, y, M , N);
        sink(grid, x, y - 1, M , N);
        sink(grid, x, y + 1, M , N);
    }
};
