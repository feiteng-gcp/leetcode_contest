class Solution {
public:
    vector<vector<int>> A;
    int M;
    int N;
    bool dfs(int x, int y) {
        A[x][y] = 2; // visited;

        int d[4][2] = {{0,1},{0,-1},{-1,0},{1,0}};

        bool flag = true;
        for (int k = 0; k < 4; ++k) {
            int nx = x + d[k][0];
            int ny = y + d[k][1];
            if (nx >= 0 && nx < M && ny >= 0 && ny < N) {
                if (A[nx][ny] == 0) {
                    if (!dfs(nx, ny)) {
                        flag = false;
                    }

                }
            } else {
                flag = false;
            }
        }
        return flag;
    }
    int closedIsland(vector<vector<int>>& grid) {
        if (grid.empty()) return 0;
        A = grid;
        M = A.size();

        N = A[0].size();
        int ans = 0;
        for (int i = 0; i < M; ++i) {
            for (int j = 0; j < N; ++j) {
                if (A[i][j] == 0) {
                    ans += dfs(i, j) ? 1 : 0;
                }
            }
        }
        return ans;
    }
};
