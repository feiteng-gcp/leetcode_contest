class Solution {
public:
    int res = 0;
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        vector<int> count(26, 0);
        for(auto& c : letters) {
            count[c-'a']++;
        }
        dfs(words, 0, 0, count, score);
        return res;
    }
    
    void dfs(vector<string>& words, int index, int cur_score, vector<int>& count, vector<int>& scores) {
        //
        res = max(res, cur_score);
        if(index == words.size())
            return;
        
        vector<int> w(26);
        for(auto& c : words[index]) {
            w[c-'a']++;
        }
        if(bigger(count, w)) {
            int tmp = cur_score;
            for(auto& c : words[index]) {
                tmp += scores[c-'a'];
            }
            for(int i = 0; i < 26; ++i)
                w[i] = count[i] - w[i];
            res = max(res, tmp);
            dfs(words, index+1, tmp, w, scores);
        }
        dfs(words, index+1, cur_score, count, scores);
    }
    
    bool bigger(vector<int>& v1, vector<int>& v2) {
        for(int i = 0; i < v1.size(); ++i) {
            if(v1[i] < v2[i])
                return false;
        }
        return true;
    }
    
};
