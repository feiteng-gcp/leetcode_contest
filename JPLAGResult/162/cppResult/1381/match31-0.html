class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        vector<int> cnt(26, 0);
        for (auto& c : letters) cnt[c - 'a']++;
        int n = words.size();
        int maxstate = 1<<n;
        int ret = 0;
        for (int s = 1; s < maxstate; ++s) {
            vector<int> ct(26, 0);
            for (int i = 0; i < n; ++i) if ((1<<i) & s) {
                for (auto& c : words[i]) ct[c - 'a']++;
            }
            bool flag = true;
            int sum = 0;
            for (int k = 0; k < 26; ++k) {
                if (cnt[k] < ct[k]) {
                    flag = false;
                    break;
                }
                sum += ct[k] * score[k];
            }
            if (!flag) continue;
            ret = max(ret, sum);            
        }
        return ret;
    }
};
