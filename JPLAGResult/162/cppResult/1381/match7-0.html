class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        if (score.size() != 26)
            exit(-1);
        vector<int> max_letter_counts(26, 0);
        for(int i=0;i<letters.size();i++) {
            max_letter_counts[letters[i]-'a']++;
        }
        int ans=0;
        int W=words.size();
        for(int mask=0;mask<(1<<W);mask++) {
            vector<int> curr_letter_counts(26, 0);
            int total_score=0;
            for(int i=0;i<W;i++) {
                if ((mask>>i)&1) {
                    for(char c : words[i]) {
                        curr_letter_counts[c-'a']++;
                        total_score += score[c-'a'];
                    }
                }
            }
            bool ok=true;
            for(int i=0;i<26;i++) {
                ok &= curr_letter_counts[i]<=max_letter_counts[i];
            }
            if (ok) {
                ans = max(ans, total_score);
            }
        }
        return ans;
        
    }
};
