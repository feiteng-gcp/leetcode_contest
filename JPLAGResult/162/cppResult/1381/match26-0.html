int N;
vector<int> s;
unordered_map<char, int> m;
unordered_map<char, int> c;
 
class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        N = words.size();

        m.clear();
        for (auto x : letters) {
            m[x] += 1;
        }

        int best = 0;
        bool found = 0;
        int current;

        for (int i = 0; i < (1 << N); i++) {
            c.clear();
            found = 1;
            current = 0;
            for (int j = 0; j < N; j++) {
                if ((i & (1 << j)) > 0) {
                    for (auto k : words[j]) {
                        c[k] += 1;
                        if (c[k] > m[k]) found = 0;
                        current += score[k- 'a'];
                    }
                }
            }
            if (found) {
                //printf("%d %d\n", i, current);
                best = max(best, current);
            }
        }
        
        return best;
    }
};
