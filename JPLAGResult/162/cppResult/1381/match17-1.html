class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int cnt[26] = {};
        for(char c : letters)   cnt[c-'a']++;

        int ret = 0;
        int n = words.size();
        for(int i = 1; i < 1<<n; i++){
            int tmp[26] = {};
            int tmps = 0;
            for(int j = 0; j < n; j++){
                if((i>>j)&1){
                    for(char c : words[j])  tmp[c-'a']++, tmps += score[c-'a'];
                }
            }
            bool ok = true;
            for(int j = 0; j < 26; j++) ok &= tmp[j] <= cnt[j];
            if(ok)  ret = max(ret, tmps);
        }
        return ret;
    }
};
