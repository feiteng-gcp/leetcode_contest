class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int n = words.size();
        vector<int> cnt0(26);
        for(auto c : letters)
            cnt0[c - 'a']++;
        long long res = 0;
        for(int mask = 1; mask < (1 << n); mask++) {
            vector<int>cnt(26, 0);
            for(int i = 0; i < words.size(); i++) {
                if(mask & (1 << i)){
                    for(auto c : words[i])
                        cnt[c - 'a']++;
                }
            }
            int valid = 1;
            long long tmp = 0;
            for(int i = 0; i < 26; i++) {
                if(cnt[i] > cnt0[i])
                    valid = 0;
                tmp += score[i] * cnt[i];
            }
            if(valid)
            res = max(res, tmp);
        }
        return res;
        
    }
};
