int cmp(int a, int b){
    return a > b;
}

class Solution {
public:
    int cnt[26];
    int app[26];
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        memset(cnt, 0, sizeof(cnt));
        
        int n = words.size(), m = letters.size();
        for (int i = 0; i < m; ++i){
            ++cnt[letters[i] - 'a'];
        }
        
        int lim = 1<<n;
        int ans = 0, now = 0;
        for (int s = 0; s < lim; ++s){
            memset(app, 0, sizeof(app));
            now = 0;
            
            for (int i = 0; i < n; ++i){
                if ((s>>i)&1){
                    for (auto c: words[i]) ++app[c - 'a'];
                }
            }
            
            bool flag = true;
            for (int i = 0; i < 26 && flag; ++i){
                if (app[i] > cnt[i]) flag = false;
                now += score[i] * app[i];
            }
            
            if (flag && now > ans) ans = now;
        }
        
        return ans;
    }
};
