class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {

        int n=words.size();
        int m=letters.size();
        int ans=0;
        int x=1<<n;
        vector<string>arr;
        for(int i=0;i<x;i++)
        {
            int j=i;
            int cnt=0;
            string y="";
            while(j>0)
            {
                if(j&1)
                {
                    y+=words[cnt];
                }
                j=j>>1;
                cnt++;
            }
            arr.push_back(y);
        }
        int hh=arr.size();
        vector< vector<int> >val(hh, vector<int>(26,0));
        for(int i=0;i<arr.size();i++)
        {
            for(int j=0;j<arr[i].size();j++)
            {
                val[i][arr[i][j]-'a']++;
            }
        }
        vector<int>cnt(26,0);
        for(int i=0;i<letters.size();i++)
        {
            cnt[letters[i]-'a']++;
        }
        for(int i=0;i<arr.size();i++)
        {
            int sc=0;

            bool flag=true;
            for(int j=0;j<26;j++)
            {
                if(val[i][j]>cnt[j])
                {
                    flag=false;
                    break;
                }
                sc+=val[i][j]*score[j];
            }
            if(flag)
            {
                ans=max(ans,sc);
            }
        }
        return ans;
        
    }
};
