class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        const int n = words.size();
        vector<vector<int>> count(n, vector<int>(26, 0));
        vector<int> value(n, 0);
        vector<int> total(26, 0);
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < words[i].size(); ++j) {
                count[i][words[i][j] - 'a']++;
                value[i] += score[words[i][j] - 'a'];
            }
        }
        for (int i = 0; i < letters.size(); ++i) {
            total[letters[i] - 'a']++;
        }
        int ans = 0;
        for (int status = 0; status < (1 << n); ++status) {
            vector<int> current(26, 0);
            bool valid = true;
            int total_value = 0;
            for (int i = 0; i < n; ++i) {
                if ((status >> i) & 1) {
                    total_value += value[i];
                    for (int j = 0; j < 26; ++j) {
                        current[j] += count[i][j];
                        if (current[j] > total[j]) {
                            valid = false;
                            break;
                        }
                    }
                }
                if (!valid) break;
            }
            if (valid && total_value > ans) ans = total_value;
        }
        return ans;
    }
};
