const int ALPHABET = 26;


class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int W = words.size();
        vector<int> letter_count(ALPHABET, 0);

        for (char c : letters)

            letter_count[c - 'a']++;

        int best = 0;

        for (int mask = 0; mask < 1 << W; mask++) {
            vector<int> save = letter_count;
            int sum = 0;

            for (int i = 0; i < W; i++)
                if (mask >> i & 1)
                    for (int c = 0; c < ALPHABET; c++) {
                        int occur = count(words[i].begin(), words[i].end(), 'a' + c);
                        letter_count[c] -= occur;
                        sum += score[c] * occur;
                    }

            if (*min_element(letter_count.begin(), letter_count.end()) >= 0)
                best = max(best, sum);

            letter_count = save;
        }

        return best;
    }
};
