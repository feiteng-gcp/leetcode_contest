class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int n = words.size();
        vector<int> a = vector<int>(26, 0);
        for(auto ch: letters) {
            a[ch - 'a'] += 1;
        }
        
        int rt = 1 << n;

        int ans = 0;
        for(int i = 1; i < rt; i++) {
            vector<int> b = vector<int>(26, 0);
            int tmp = 0;
            for(int j = 0; j < n; j++) {
                if((i >> j) & 1) {
                    for(auto ch: words[j]) {
                        b[ch - 'a'] += 1;
                        tmp += score[ch - 'a'];
                    }
                }
            }
            bool flag = true;
            for(int j = 0; j < 26; j++) {
                if(b[j] > a[j]) {
                    flag = false;
                }
            }
            if(!flag) continue;
            else {
                ans = max(ans, tmp);
            }
        }
        return ans;
    }
};
