class Solution {
public:
    int cnt[30];
    int maxScoreWords(vector<string>& v, vector<char>& a, vector<int>& b) {
        memset(cnt, 0, sizeof(cnt));

        for(int i = 0; i < a.size(); i++){
            cnt[a[i]-'a']++;
        }
        int n = v.size();
        int c[30], ans = 0;
        for(int i = 0; i < (1<<n); i++){
            memset(c, 0, sizeof(c));
            int s = 0;

            for(int j = 0; j < n; j++){
                if(i&(1<<j)){
                    for(int k = 0; k < v[j].size(); k++){
                        c[v[j][k]-'a']++;
                        s += b[v[j][k]-'a'];
                    }
                }
            }
            int flag = 1;
            for(int j = 0; j < 26; j++){
                if(c[j] > cnt[j]) flag = 0;
            }
            if(flag) ans = max(ans, s);
        }
        return ans;
    }
};
