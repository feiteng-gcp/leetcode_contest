class Solution {
public:
    int solve(vector<vector<int>> &dp, int status, int k, vector<int> &ab, const vector<string>& words, const vector<int>& score) {
        if(k >= words.size()) return 0;
        if(dp[status][k] >= 0) return dp[status][k];
        
        int res = 0;
        if(0 == ((1 << k) & status)) {
            bool flag = true;
            int sum = 0;
            vector<int> abt = ab;
            for(int i = 0; i < words[k].size(); ++i) {
                const int wn = words[k][i] - 'a';
                if(abt[wn] > 0) {
                    --abt[wn];
                    sum += score[wn];
                } else {
                    flag = false;
                    break;
                }
            }
            if(flag) res = max(res, sum + solve(dp, status | (1 << k), k + 1, abt, words, score));
            res = max(res, solve(dp, status, k + 1, ab, words, score));
        }
        return dp[status][k] = res;
    }
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        vector<vector<int>> dp((1 << words.size()), vector<int>(words.size() + 1, -1));
        vector<int> ab(26, 0);
        for(int i = 0; i < letters.size(); ++i) ++ab[letters[i] - 'a'];
        return solve(dp, 0, 0, ab, words, score);
    }
};
