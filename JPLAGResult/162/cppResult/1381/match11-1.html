class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int n = words.size();
        vector<int> l(26);
        for(auto x : letters) l[x - 'a']++;

        int ans = 0;
        for(int i = 1; i < 1<<n; ++i){
            vector<int> t(26);
            for(int j = 0; j < n; ++j){
                if((i >> j) & 1) {

                    for(auto x : words[j]) t[x - 'a']++;
                }
            }
            int flag = 0, cnt = 0;
            for(int i = 0; i < 26; ++i){
                if(t[i] > l[i]){
                    flag = 1;
                    break;
                }
                cnt += t[i] * score[i];
            }
            if(!flag) ans = max(ans, cnt);
        }
        return ans;
    }
};
