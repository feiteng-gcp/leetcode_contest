class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        unordered_map<char, int> char_scores;
        unordered_map<char, int> char_count;
        for (int i = 0; i < letters.size(); i++) {
            char_count[letters[i]] ++;
            char_scores[letters[i]] = score[letters[i] - 'a'];
        }
        int max_res = 0;
        helper(words, char_scores, char_count, 0, 0, max_res);
        return max_res;
    }

    void helper(vector<string> &words, unordered_map<char, int> &char_scores, unordered_map<char, int> &char_count, int pos, int res, int &max_res)
    {
        if (pos == words.size()) {
            if (max_res < res) {
                max_res = res;
            }
            return;
        }
        string &word = words[pos];
        bool valid = true;
        int tmp_res = res;
        for (int i = 0; i < word.size(); i++) {
            char c = word[i];
            if (char_count[c] > 0) {
                char_count[c]--;
                tmp_res += char_scores[c];
            } else {
                for (int j = 0; j < i; j++) {
                    char_count[word[j]]++;
                }
                valid = false;
                break;
            }
        }
        
        if (valid) {
            helper(words, char_scores, char_count, pos + 1, tmp_res, max_res);
            for (char c : word) {
                char_count[c]++;
            }
        }
        helper(words, char_scores, char_count, pos + 1, res, max_res);

    }
};
