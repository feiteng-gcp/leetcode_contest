class Solution {
public:
    vector<int> cnt;
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        cnt.assign(26,0);
        for(char c:letters)
            cnt[c-'a']++;
        int n=words.size();
        int mask=1<<n;
        int ans=0;
        for(int i=1;i<mask;i++)
        {
            vector<int> ncnt(26,0);
            for(int j=0;j<n;j++)
            {
                if(!((1<<j)&i)) continue;
                for(char c:words[j])
                    ncnt[c-'a']++;
            }
            int nans=0;
            for(int j=0;j<26;j++)
            {
                if(ncnt[j]>cnt[j])
                {
                    nans=0;
                    break;
                }
                nans+=score[j]*ncnt[j];
            }
            ans=max(ans,nans);
        }
        return ans;
    }
};
