class Solution {
public:
    int cnt[27];
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int n=words.size(),tp=(1<<n);

        for(int i=0;i<letters.size();i++)cnt[letters[i]-'a']++;
        int ans=0;
        for(int i=1;i<tp;i++){
            int now[27]={0};
            for(int j=0;j<n;j++)if(i&(1<<j)){
                for(int k=0;k<words[j].size();k++)now[words[j][k]-'a']++;
            }
            bool flag=1;

            int tmp=0;
            for(int j=0;j<26;j++)if(cnt[j]<now[j]){flag=0;break;}
            if(flag){for(int j=0;j<26;j++)tmp+=score[j]*now[j];ans=max(ans,tmp);}
        }
        return ans;
    }
};
