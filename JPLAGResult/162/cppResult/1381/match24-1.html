class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int n = words.size();
        int cnt[26];
        int new_cnt[26];
        for (int i = 0; i < 26; i ++) cnt[i] = 0;

        for (int i = 0; i < letters.size(); i ++) cnt[letters[i] - 'a'] ++;
        int ans = 0;
        for (int s = 0; s < (1 << n); s ++) {
            for (int j = 0; j < 26; j ++) new_cnt[j] = 0;
            for (int j = 0; j < n; j ++) {
                if ((1 << j) & s) {
                    for (int i = 0; i < words[j].length(); i ++) new_cnt[words[j][i] - 'a'] ++;
                }

            }
            bool ok = true;
            int tmp = 0;
            for (int j = 0; j < 26; j ++) {
                if (new_cnt[j] > cnt[j]) {
                    ok = false;
                    break;
                }
                tmp += new_cnt[j] * score[j];
            }
            if (!ok) continue;
            ans = max(ans, tmp);
        }
        return ans;
    }
};
