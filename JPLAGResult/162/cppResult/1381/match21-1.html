class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int N=words.size();
        vector<int> cnt(26,0);
        for(char c:letters) cnt[c-'a']++;
        int T=1<<N;

        int ans=0;
        for(int i=0;i<T;i++){
            vector<int> cnt2(26,0);
            bool flag=false;
            int res=0;
            for(int j=0;j<N;j++){
                if((i&(1<<j))!=0){
                    for(char c:words[j]){
                        int a=c-'a';
                        cnt2[a]++;

                        res+=score[a];
                        if(cnt2[a]>cnt[a]){
                            flag=true;
                            break;
                        }
                    }
                }
                if(flag) break;
            }
            if(!flag) ans=max(ans,res);
        }
        return ans;
    }
};
