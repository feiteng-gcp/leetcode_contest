class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int n = words.size();
        int num[26];
        int ans = 0;
        for (int i=0;i<26;i++) num[i] = 0;
        for (int i=0;i<letters.size();i++) num[letters[i]-'a']++;
        for (int i=0;i<=(1<<n)-1;i++)
        {
        int num2[26];
        int temp = 0;
            for (int j=0;j<26;j++) num2[j]=0;
            for (int j=0;j<n;j++)
            {
                if (((1<<j)&i)==0) continue;
                for (int k=0;k<words[j].size();k++)
                {
                    num2[words[j][k]-'a']++;
                    temp+=score[words[j][k]-'a'];
                }
            }
            bool flag = true;
            // cout<<temp<<endl;
            for (int i=0;i<26;i++) if (num2[i]>num[i]) flag = false;
            if (flag) ans = max(ans,temp);
        }
        return ans;
    }
};
