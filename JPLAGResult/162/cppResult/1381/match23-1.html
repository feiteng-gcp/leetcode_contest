class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int nw=words.size(),ans=0;
        vector<int> let(26);

        for(int i=0;i<letters.size();i++) let[letters[i]-'a']++;
        for(int i=1;i<(1<<nw);i++){
            vector<int> V(26);int tscore=0;
            for(int j=0;j<nw;j++) if(i&(1<<j)){

                for(int k=0;k<words[j].size();k++){ 
                    V[words[j][k]-'a']++;
                    tscore+=score[words[j][k]-'a'];
                }
            }
            bool flag=true;
            //printf("\n");
            //for(int j=0;j<26;j++) printf("(%d,%d)  ",V[j],let[j]);
            for(int j=0;j<26;j++){
                if(V[j]>let[j]) flag=false;
            }
            //printf("%d %d %d\t",i,flag,tscore);
            if(flag) ans=max(ans,tscore);
        }
        return ans;
    }
};
