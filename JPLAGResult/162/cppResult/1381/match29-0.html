class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int nw = words.size(); 
        vector<vector<int>> wcs(nw, vector<int>(26, 0));
        vector<int> ws(nw, 0);
        vector<int> lcs(26, 0);
        
        for(int i=0; i<nw; i++){
            string w = words[i];
            for(char c : w){
                int ci = c - 'a';
                wcs[i][ci]++;
                ws[i] += score[ci];
                
            }
        }
        
        for(char c : letters){
            int ci = c - 'a';
            lcs[ci]++;
        }
        
        int res = 0;
        for(int a=1; a<(1<<nw); a++){
            int scur = 0;
            vector<int> cs(26, 0);
            bool f = true;
            for(int i=0; i<nw && f; i++){
                if(a & (1<<i)){
                    scur += ws[i];
                    for(char c : words[i]){
                        int ci = c - 'a';
                        cs[ci]++;
                        if(cs[ci] > lcs[ci]) f= false;
                    }
                }
            }
            
            if(f) res = max(res, scur);
        }
        
        return res;
    }
};
