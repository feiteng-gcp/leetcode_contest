class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& l, vector<int>& score) {
        int cnt[105] = {0}, cur[105] = {0};
        for (int i = 0; i < 100; i++) {
            cnt[i] = 0;
        }
        for (int i = 0; i < l.size(); i++) {
            cnt[l[i] - 'a']++;
        }
        
        int n = words.size();
        int ans = 0;
        for (int i = 0; i < (1 << n); i++) {
            int s = 0;
            int ok = true;
            for (int k = 0; k < 30; k++) {
                cur[k] = 0;
            }
            for (int j = 0; j < n; j++) {
                if (i & (1 << j)) {
                    for (int k = 0; k < words[j].size(); k++) {
                        cur[words[j][k] - 'a']++;
                        s += score[words[j][k] - 'a'];
                        if (cur[words[j][k] - 'a'] > cnt[words[j][k] - 'a']) {
                            ok = false;
                            break;
                        }
                    }
                }
                
                if (ok == false) {
                    break;
                }
            }
            
            if (ok) {
                ans = max(ans, s);
            }
        }
        
        return ans;
    }
};
