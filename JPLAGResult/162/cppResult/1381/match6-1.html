class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int m = words.size();
        int ret = 0;
        vector<int> cnt(26);
        for (auto& c : letters) {
            cnt[c - 'a']++;
        }
        for (int k = 1; k < (1 << m); ++k) {

            vector<int> cur(26);
            int sum = 0;
            for (int i = 0; i < m; ++i) {
                if (k & (1 << i)) {
                    for (auto& c : words[i]) {
                        cur[c - 'a']++;
                        sum += score[c - 'a'];
                    }
                }
            }
            bool found = false;
            for (int i = 0; i < 26; ++i) {
                if (cur[i] > cnt[i]) found = true;
            }
            if (!found) {
                ret = max(ret, sum);
            }
        }
        return ret;
    }
};
