class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        const int nw = words.size();
        vector<vector<int>> wv(nw, vector<int>(26, 0));
        for (int i = 0; i < nw; ++i) {
            for (char c: words[i]) {
                wv[i][c-'a']++;
            }
        }
        
        vector<int> lc(26, 0);
        for (const char c: letters) {
            lc[c-'a']++;
        }
        

        int max_score = 0;
        for (int b = 1; b < (1<<nw); ++b) {
            vector<int> total(26, 0);
            for (int i = 0; i < nw; ++i) {
                if (((1<<i) & b) == 0) continue;
                for (int ci = 0; ci < 26; ++ci) {
                    total[ci] += wv[i][ci];
                }
            }
            int sc = 0;

            bool valid = true;
            for (int ci = 0; ci < 26; ++ci) {
                if (total[ci] > lc[ci]) {
                    valid = false;
                    break;
                }
                sc += score[ci] * total[ci];
            }
            if (valid == false) continue;
            max_score = max(sc, max_score);
        }
        return max_score;
    }
};
