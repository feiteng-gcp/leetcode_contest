class Solution {
public:
    int ans=0;
    int len;
    int cnt[26];
    int sc[20];
    int c1[20][26];
    void dfs(int idx, int sum) {
        for (int i=0; i<26; ++i) {
            if (cnt[i]<0) return;
        }
        if (idx==len) {
            ans=max(ans,sum);
            return;
        }
        dfs(idx+1,sum);
        for (int i=0; i<26; ++i) {
            cnt[i]-=c1[idx][i];
        }
        dfs(idx+1,sum+sc[idx]);
        for (int i=0; i<26; ++i) {
            cnt[i]+=c1[idx][i];
        }
    }
    int maxScoreWords(vector<string>& words, vector<char>& d, vector<int>& score) {
        len=words.size();
        for (int i=0; i<d.size(); ++i) {
            cnt[d[i]-'a']++;
        }
        for (int i=0; i<words.size(); ++i) {
            int sum=0;
            for (int j=0; j<words[i].length(); ++j) {
                c1[i][words[i][j]-'a']++;
                sum+=score[words[i][j]-'a'];
            }
            sc[i]=sum;
        }
        dfs(0,0);
        return ans;
    }
};
