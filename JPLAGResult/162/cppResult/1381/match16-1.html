class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        int mp[30] = {0};
        for (int i = 0; i < letters.size(); ++i) mp[letters[i]-'a']++;
        int ans = 0;
        for (int bit = 0; bit < (1 << words.size()); ++bit)
        {
            int need[30] = {0};
            int scores = 0;
            for (int i = 0; (1 << i) <= bit; ++i)
                if (bit & (1 << i))
                {
                    for (int j = 0; j < words[i].size(); ++j)
                        need[words[i][j]-'a']++, scores += score[words[i][j]-'a'];
                }
            int flag = 1;
            for (int i = 0; i < 30; i++)
                if (need[i] > mp[i]) flag = 0;
            if (flag == 1) ans = max(ans, scores);
        }
        return ans;
    }
};
