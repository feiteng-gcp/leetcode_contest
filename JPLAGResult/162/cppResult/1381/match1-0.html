class Solution {
public:
    vector<int> count;
    unordered_map<string,vector<int>> word_count;
    vector<int> words_score;
    int max_score=0;
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        count.assign(26,0);
        for(string word:words){
            int word_score=0;
            vector<int> cnt(26,0);
            for(char letter:word) word_score+=score[letter-'a'],cnt[letter-'a']++;
            words_score.push_back(word_score);
            word_count[word]=cnt;
        }
        for(char letter:letters) count[letter-'a']++;
        dfs(words,0,0);
        return max_score;
        
    }
    void dfs(vector<string>& words,int ind,int cur_score){
        max_score=max(max_score,cur_score);
        for(int i=ind;i<words.size();++i){
            vector<int> curword=word_count[words[i]];
            bool can=true;
            for(int j=0;j<26;++j){
                count[j]-=curword[j];
                if(count[j]<0) can=false;
            }
            if(can) dfs(words,i+1,cur_score+words_score[i]);
            for(int j=0;j<26;++j){
                count[j]+=curword[j];
            }
        }
    }
};
