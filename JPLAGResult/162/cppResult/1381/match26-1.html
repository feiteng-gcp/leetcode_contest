class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        vector<int> w(26,0);
        for(auto c:letters)
            w[c-'a']++;
        int n = words.size(),ans = 0;
        for(int i = 0;i < (1<<n);i++){
            vector<int> w1(26,0);
            bool ok = true;

            int num = 0;
            for(int j = 0;j < n && ok;j++){
                if(i & (1<<j)){
                    for(auto c:words[j]){
                        w1[c-'a']++;
                        if(w1[c-'a'] > w[c-'a']){
                            ok = false;
                            break;
                        }

                        num += score[c-'a'];
                    }

                }
            }
            if(ok){
                ans = max(ans,num);
            }
        }
        return ans;
    }
};
