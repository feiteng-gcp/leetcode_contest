const int LTS = 26;
const int N = 20;

int dp[N][N];
int cnt[LTS];
int scores[N];

class Solution {
    int ans;
    bool ok(string &s) {
        int aux[LTS];
        memcpy(aux, cnt, sizeof(aux));
        for (int i = 0; i < s.length(); ++i) {
            if (--aux[s[i] - 'a'] < 0) { 
                return false;
            }
        }
        return true;
    }
    void solve(vector<string>& words, int d, int n, int cur) {
        if (d == n) { return; }
        solve(words, d + 1, n, cur);
        if (ok(words[d])) {
            ans = max(ans, cur + scores[d]);

            for (int i = 0; i < words[d].length(); ++i) {
                --cnt[words[d][i] - 'a'];
            }
            solve(words, d + 1, n, cur + scores[d]);
            for (int i = 0; i < words[d].length(); ++i) {
                ++cnt[words[d][i] - 'a'];
            }
        }
        return;
    }
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) {
        memset(dp, 0, sizeof(dp));
        memset(cnt, 0, sizeof(cnt));
        int n = words.size();

        int ln = letters.size();
        for (int i = 0; i < ln; ++i) { ++cnt[letters[i] - 'a']; }
        for (int i = 0; i < n; ++i) {
            int cur_score = 0;
            for (int j = 0; j < words[i].length(); ++j) {
                cur_score += score[words[i][j] - 'a'];
            }
            scores[i] = cur_score;
        }
        ans = 0;
        solve(words, 0, n, 0);
        return ans;
    }
};
