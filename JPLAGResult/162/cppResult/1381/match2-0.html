class Solution {
public:
    int maxScoreWords(vector<string>& words, vector<char>& letters, vector<int>& score) 
    {
        vector<vector<int>>statuses;
        map<vector<int>,int>Map;
        vector<int>temp(26,0);
        Map[temp] = 0;
        
        vector<int>count(26,0);
        for (auto ch:letters)
            count[ch-'a']++;
        
        for (auto word: words)
        {
            vector<int>count0(26,0);
            for (auto ch: word)
                count0[ch-'a']++;
            
            auto Map2 = Map;
            
            for (auto& x:Map)
            {
                auto status = x.first;
                vector<int>temp = status;
                int flag = 1;
                int cost = 0;
                
                for (int i=0; i<26; i++)
                {
                    temp[i] += count0[i];
                    cost += score[i]*count0[i];
                    if (temp[i]>count[i])
                    {
                        flag = 0;
                        break;
                    }                        
                }
                
                if (flag==0) continue;
                
                Map2[temp] = max(Map2[temp], Map[status]+cost);                
            }
            
            Map = Map2;
        }
        
        int ret = 0;
        for (auto& x:Map)
            ret = max(ret, x.second);
        return ret;
    }
};
