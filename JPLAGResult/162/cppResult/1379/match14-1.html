class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        int size = colsum.size();
        vector<vector<int>> res(2, vector<int>(size));
        int up_left = upper;
        int down_left = lower;
        vector<int> col_1;
        for (int i = 0; i < size; i++) {
            if (colsum[i] == 0) {
                continue;
            }
            if (colsum[i] == 2) {
                up_left -= 1;
                down_left -= 1;
                res[0][i] = res[1][i] = 1;
                continue;
            }
            col_1.push_back(i);
        }
        if (up_left + down_left != col_1.size() || up_left < 0 || down_left < 0) {
            return vector<vector<int>>();
        }
        int i = 0;
        while (i < col_1.size() && up_left > 0) {
            res[0][col_1[i]] = 1;
            up_left--;
            i++;
        }

        while (i < col_1.size()) {
            res[1][col_1[i]] = 1;
            i++;
        }
        return res;
    }
};
