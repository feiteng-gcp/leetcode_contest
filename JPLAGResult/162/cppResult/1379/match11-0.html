class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        vector<vector<int>> res(2, vector<int>(colsum.size(),0));
        for(int i=0;i<colsum.size();i++) {
            if(colsum[i] == 2) {
                res[0][i] = res[1][i] = 1;
                upper--, lower--;
            } else if(colsum[i] == 1) {
                // greedy
                if(upper > lower) {
                    res[0][i] = 1;
                    upper--;
                }
                else {
                    res[1][i] = 1;
                    lower--;
                }
            }
        }
        if(upper != 0 || lower !=0) return {};
        return res;
    }
};
