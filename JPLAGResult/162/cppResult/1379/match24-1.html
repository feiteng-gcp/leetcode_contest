    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        int m = colsum.size();
        vector<vector<int>> ans(2, vector<int>(m));
        for(int i = 0; i < m; ++i) {
            int c = colsum[i];

            if(c == 2) {
                ans[0][i] = 1;
                ans[1][i] = 1;
                --upper;
                --lower;
            } else if(c == 1) {
                if(upper > lower) {
                    ans[0][i] = 1;
                    --upper;
                } else {
                    ans[1][i] = 1;
                    --lower;
                }
            };
            if(upper < 0 || lower < 0) return {};
        };
        if(upper == 0 && lower == 0) return ans;
        return {};
    }
};
