class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        int N=colsum.size();
        vector<vector<int>> res(2,vector<int>(N,0));
        int a=0;
        int b=0;
        int cnt=0;
        for(int i=0;i<N;i++){
            if(colsum[i]==2){
                res[0][i]=1;
                res[1][i]=1;
                a++;
                b++;
            }else if(colsum[i]==1){
                cnt++;
            }
        }
        
        if(a>upper||b>lower) return vector<vector<int>>();
        if(upper-a+lower-b!=cnt) return vector<vector<int>>();
        for(int i=0;i<N;i++){
            if(colsum[i]==1){
                if(a<upper){
                    res[0][i]=1;
                    a++;
                }else{
                    res[1][i]=1;
                    b++;
                }
            }
        }
        return res;
    }
};
