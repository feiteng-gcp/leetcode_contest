class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        int n=colsum.size();
        vector<vector<int>> mat(2,vector<int>(n,0));

        for(int i=0; i<n; i++){
            if(colsum[i]==2){
                mat[0][i]=1;
                mat[1][i]=1;
                upper--;
                lower--;
            } else if(colsum[i]==1){

                if(upper>lower){
                    mat[0][i]++;
                    upper--;
                } else{
                    mat[1][i]++;
                    lower--;
                }
            }
        }
        
        if(upper!=0 || lower!=0) mat.clear();
        
        return mat;
    }
};
