class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        const int len = colsum.size();
        vector<vector<int>> mat(2, vector<int>(len, 0));
        for(int i = 0; i < len; ++i) {
            if(2 == colsum[i]) {
                --upper; --lower;
                mat[0][i] = 1; mat[1][i] = 1;
            }
        }
        for(int i = 0; i < len; ++i) {
            if(1 == colsum[i]) {
                if(upper > 0) {
                    mat[0][i] = 1;
                    --upper;
                } else {
                    mat[1][i] = 1;
                    --lower;
                }
            }
        }
        if(0 == upper && 0 == lower) return mat;
        return vector<vector<int>>();
    }
};
