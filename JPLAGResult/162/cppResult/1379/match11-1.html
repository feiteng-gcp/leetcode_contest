class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        vector<vector<int>> res(2, vector<int>(colsum.size()));
        for(int i=0; i<colsum.size(); i++) {
            if(colsum[i]==2) {
                res[0][i] = 1;
                res[1][i] = 1;
                upper--;
                lower--;
            }
        }
        if(upper<0||lower<0)
            return {};
        for(int i=0; i<colsum.size(); i++) {

            if(colsum[i]==1) {
                if(upper>0) {
                    upper--;
                    res[0][i] = 1;
                } else {
                    lower--;
                    res[1][i] = 1;
                }
            }
        }
        if(lower!=0||upper!=0)
            return {};
        return res;

    }
};
