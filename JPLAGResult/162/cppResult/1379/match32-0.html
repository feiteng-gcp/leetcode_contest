class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        
        vector<vector<int>> ans,anss;
        vector<int> ans1(colsum.size(),0),ans2(colsum.size(),0);
        ans.push_back(ans1);
        ans.push_back(ans2);
        
        for(int i=0;i<colsum.size();i++)
        {
            int tp=colsum[i];
            if(tp==0)
            {
                ans[0][i]=0;
                ans[1][i]=0;
            }
            else if(tp==2)
            {
                ans[0][i]=1;
                ans[1][i]=1;
                upper--;lower--;
            }
        }
        for(int i=0;i<colsum.size();i++)
        {
            int tp=colsum[i];
            if(tp==1)
            {
                if(upper>0)
                {
                    upper--;
                    ans[0][i]=1;
                }
                else
                {
                    lower--;
                    ans[1][i]=1;
                }
            }
        }
        
        if(upper==0 && lower==0)
            return ans;
        else
            return anss;
    }
};
