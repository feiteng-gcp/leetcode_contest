class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) 
    {
        int n = colsum.size();
        int count1=0, count2=0, count0=0;
        for (int i=0; i<n; i++)
        {
            if (colsum[i]==0)
                count0++;
            else if (colsum[i]==1)
                count1++;
            else
                count2++;
        }
                
        vector<int>p1(n,-1);
        vector<int>p2(n,-1);
        
        if (upper<count2 || lower<count2) return {};
        if (n-upper<count0 || n-lower<count0) return {};
        
        for (int i=0; i<n; i++)
        {
            if (colsum[i]==2)
            {
                p1[i]=1;
                p2[i]=1;
            }
            else if (colsum[i]==0)
            {
                p1[i]=0;
                p2[i]=0;
            }
            
        }
        
                
        upper -= count2;
        lower -= count2;
        if (upper+lower!=n-count2-count0) return {};
        
        
        for (int i=0; i<n; i++)
        {
            if (colsum[i]!=1) continue;
            if (upper>0)
            {
                p1[i]=1;
                p2[i]=0;
                upper--;
            }
            else
            {
                p1[i]=0;
                p2[i]=1;
            }
            
        }
            
        vector<vector<int>>rets({p1,p2});
        return rets;
        
    }
};
