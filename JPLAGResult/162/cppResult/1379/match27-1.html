class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        int len = colsum.size();
        vector<vector<int>> ans(2, vector<int>(len, 0));
        vector<vector<int>> def;
        for (int i = 0; i < len; i++) {
            if (colsum[i] == 2) {
                ans[0][i] = ans[1][i] = 1;

                upper--;
                lower--;
            }
        }
        for (int i = 0; i < len; i++) {
            if (colsum[i] == 1) {
                if (upper > 0) {
                    ans[0][i] = 1;
                    upper--;
                }
                else if (lower > 0) {
                    ans[1][i] = 1;
                    lower--;
                }
                else {
                    return def;
                }
            }
        }
        if (upper == 0 && lower == 0) {
            return ans;
        }
        return def;
    }
};
