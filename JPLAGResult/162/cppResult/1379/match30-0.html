class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        int n = colsum.size();
        vector<vector<int> > res;
        vector<vector<int> > ept;
        ept.clear();
        res.clear();
        res.push_back(vector<int>());
        res.push_back(vector<int>());
        for (int i = 0; i < n; ++i) {
            res[0].push_back(0);
            res[1].push_back(0);
        }
        int sum = 0;
        for (int i = 0; i < n; ++i) {
            sum += colsum[i];
        }
        if (sum != (upper + lower)) {
            return ept;
        }
        for (int i = 0; i < n; ++i) {
            if (colsum[i] == 2) {
                res[0][i] = 1;
                res[1][i] = 1;
                --lower;
                --upper;
            }
        }
        if (lower < 0 || upper < 0) {
            return ept;
        }
        for (int i = 0; i < n; ++i) {
            if (colsum[i] == 1) {
                if (lower > 0) {
                    --lower;
                    res[1][i] = 1;
                }
                else {
                    res[0][i] = 1;
                }
            }
        }
        return res;
    }
};
