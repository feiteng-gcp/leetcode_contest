class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        int n = colsum.size();
        vector<int> up_arr(n);
        vector<int> down_arr(n);
        vector<vector<int>> ans;
        for (int i = 0; i < n; i ++) {
            if (colsum[i] == 2) {
                upper --;
                lower --;
                up_arr[i] = 1;
                down_arr[i] = 1;
            } else if (colsum[i] == 0) {
                up_arr[i] = 0;
                down_arr[i] = 0;
            }
        }
        for (int i = 0; i < n; i ++) {
            if (colsum[i] == 1) {
                if (upper > 0) {
                    up_arr[i] = 1;
                    upper --;
                } else if (lower > 0) {
                    down_arr[i] = 1;
                    lower --;
                } else {
                    return ans;
                }
            }
        }
        if (upper != 0 || lower != 0) return ans;
        ans.push_back(up_arr);
        ans.push_back(down_arr);
        return ans;
    }
};
