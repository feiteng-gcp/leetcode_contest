class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        int COL = colsum.size();
        vector<vector<int>> ret(2, vector<int>(COL, 0));
        for (int j = 0; j < COL; ++j) {
            if (colsum[j] == 2) {
                --lower;
                --upper;
                ret[0][j] = ret[1][j] = 1;
            }
        }
        for (int j = 0; j < COL; ++j) {
            if (colsum[j] == 1) {
                if (upper > 0) {
                    --upper;
                    ret[0][j] = 1;
                }
                else {
                    --lower;
                    ret[1][j] = 1;
                }
            }
        }
        
        if (lower != 0 || upper != 0) return vector<vector<int>>();
        return ret;
    }
};
