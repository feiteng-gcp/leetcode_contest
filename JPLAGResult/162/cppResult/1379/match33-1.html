template <class T> ostream &operator <<(ostream &os, vector<T> V){
    os << "["; for (auto vv: V) os << vv << ","; return os << "]";
}


class Solution {
public:
    vector<vector<int>> reconstructMatrix(int u, int l, vector<int>& c) {
        int n = c.size();
        vector<vector<int>> res(2, vector<int>(n, 0));
        
        int sum = 0;
        for (int i : c) sum += i;
        if (sum != u + l) return {};
        if (u > n || l > n) return {};
        

        int s[2] = {0, 0};
        for (int i = 0; i < n; ++i) {
            if (c[i] == 2) {
                s[0] += 1;
                s[1] += 1;
                res[0][i] = res[1][i] = 1;
            } 
        }
        if (s[0] > u || s[1] > l) return {};
        for (int i = 0; i < n; ++i) {
            if (c[i] == 1) {
                if (s[0] < u) {
                    s[0] += 1;
                    res[0][i] = 1;
                } else {
                    s[1] += 1;
                    res[1][i] = 1;
                }
            }
        }
        return res;
    }
};
