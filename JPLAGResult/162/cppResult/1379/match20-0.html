class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
     
        int n = colsum.size();
        vector< vector<int> > res(2, vector<int>(n, 0));
        
        int U = upper;
        int L = lower;
        for (int i = 0; i < n; ++i)
        {
            if (colsum[i] == 2)
            {
                res[0][i] = 1;
                res[1][i] = 1;
                
                --U;
                --L;
            }
            else if (colsum[i] == 1)
            {
                if (U >= L)
                {
                    res[0][i] = 1;
                    --U;
                }
                else 
                {
                    res[1][i] = 1;
                    --L;
                }
            }
        }
        
        if (U != 0 || L != 0) return {};
        
        return res;
    }
};
