class Solution {
public:
    vector<vector<int>> reconstructMatrix(int upper, int lower, vector<int>& colsum) {
        int n = colsum.size();
        vector<vector<int>> res(2, vector<int>(n, 0));
        int sum = accumulate(colsum.begin(), colsum.end(), 0);
        if(sum != upper+lower)

            return {};
        for(int i = 0; i < n; ++i) {
            if(colsum[i] == 2) {
                res[0][i] = res[1][i] = 1;
                --upper;
                --lower;
            }
        }
        if(lower < 0)

            return {};
        for(int i = 0; i < n; ++i) {
            if(colsum[i] == 1) {
                if(upper) {
                    res[0][i] = 1;
                    upper--;
                }
                else {
                    res[1][i] = 1;
                }
            }
        }
        return res;
        
        
        
    }
};
