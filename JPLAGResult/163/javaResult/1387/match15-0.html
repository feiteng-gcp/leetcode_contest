/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class FindElements {
    HashSet<Integer> set;
    public FindElements(TreeNode root) {
        set = new HashSet<>();
        dfs(root, 0);
    }
    void dfs(TreeNode x, int v) {
        if(x == null) return;
        set.add(v);
        dfs(x.left, v * 2 + 1);
        dfs(x.right, v * 2 + 2);
    }
    
    public boolean find(int target) {
        return set.contains(target);
    }
}
