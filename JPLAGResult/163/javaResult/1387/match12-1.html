/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class FindElements {
    TreeNode binaryTree;
    Set<Integer> valuesSet;

    public FindElements(TreeNode root) {
        valuesSet = new HashSet<Integer>();
        binaryTree = root;
        binaryTree.val = 0;
        Queue<TreeNode> queue = new LinkedList<TreeNode>();
        queue.offer(binaryTree);
        while (!queue.isEmpty()) {
            TreeNode node = queue.poll();
            int val = node.val;
            valuesSet.add(val);
            TreeNode left = node.left, right = node.right;
            if (left != null) {
                int leftVal = val * 2 + 1;
                valuesSet.add(leftVal);
                left.val = leftVal;
                queue.offer(left);
            }
            if (right != null) {
                int rightVal = val * 2 + 2;
                valuesSet.add(rightVal);
                right.val = rightVal;
                queue.offer(right);
            }
        }
    }
    
    public boolean find(int target) {
        return valuesSet.contains(target);
    }
}
