/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class FindElements {

    private TreeNode root;

    public FindElements(TreeNode root) {
        this.root = root;
        this.root.val = 0;
        constructTree(this.root);
    }

    private void constructTree(TreeNode root) {
        if (root == null) {
            return;
        }
        if (root.left != null) {
            root.left.val = 2 * root.val + 1;
            constructTree(root.left);
        }
        if (root.right != null) {
            root.right.val = 2 * root.val + 2;
            constructTree(root.right);
        }
    }

    public boolean find(int target) {
        return find(target, this.root);
    }
    
    private boolean find(int target, TreeNode root) {
        if (root == null) {
            return false;
        }
        if (target == root.val) {
            return true;
        }
        return find(target, root.left) || find(target, root.right);
    }
}
