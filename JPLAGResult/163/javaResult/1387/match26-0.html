/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class FindElements {

    HashSet<Integer> set;
    
    public FindElements(TreeNode root) {
        set=new HashSet<>();   
        build(root, set, 0);
        // System.out.println(set);
    }
    
    public boolean find(int target) {
        return set.contains(target);
    }
    
    public void build(TreeNode root, HashSet<Integer> set, int val) {
        if(root==null)
            return;
        
        root.val=val;
        set.add(val);
        
        build(root.left, set, 2*val+1);
        build(root.right, set, 2*val+2);
    }
}

 * FindElements obj = new FindElements(root);
