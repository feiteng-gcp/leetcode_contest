/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
    class FindElements {

        private Set<Integer> set = new HashSet<>();

        public FindElements(TreeNode root) {
            addValueByTreeShape(root, 0);
        }

        private void addValueByTreeShape(TreeNode root, int rootValue) {
            if (Objects.isNull(root)) {
                return;
            }
            this.set.add(rootValue);
            addValueByTreeShape(root.left, 1 + 2 * rootValue);
            addValueByTreeShape(root.right, 2 + 2 * rootValue);
        }

        public boolean find(int target) {
            return this.set.contains(target);
        }
    }

 * FindElements obj = new FindElements(root);
 * boolean param_1 = obj.find(target);
