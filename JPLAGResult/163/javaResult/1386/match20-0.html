class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int n = grid.length;
        int m = grid[0].length;
        int downShift = (k / m) % n;
        int rightShift = k % m;
        List<List<Integer>> res = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            List<Integer> row = new ArrayList<>();
            for (int j = 0; j < m; j++) {
                int num = j >= rightShift ? grid[(i + n - downShift) % n][(j + m - rightShift) % m] : 
                grid[(i + n - downShift - 1) % n][(j + m - rightShift) % m];
                row.add(num);
            }
            res.add(row);
        }
        return res;
    }
}
