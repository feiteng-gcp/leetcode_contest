class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int m = grid.length, n = grid[0].length;
        int[][] ret = new int[m][n];
        
        int yOffset = k % n, xOffset = (k % (m * n)) / n;
        // System.out.println(yOffset + ", " + xOffset);
        
        for (int srcJ = 0; srcJ < n; ++srcJ) {
            int y = (yOffset + srcJ) % n;
            
            for (int srcI = 0; srcI < m; ++srcI) {
                int x = (xOffset + srcI + (y < yOffset ? 1 : 0)) % m;
                
                ret[x][y] = grid[srcI][srcJ];
            }
        }
        
        List<List<Integer>> retList = new ArrayList(m);
        for (int x = 0; x < m; ++x) {
            List<Integer> row = new ArrayList(n);
            for (int y = 0; y < n; ++y) {
                row.add(ret[x][y]);
            }
            retList.add(row);
        }
        
        // System.out.println("------");
        return retList;
    }
}
