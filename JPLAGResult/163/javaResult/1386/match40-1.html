class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int m = grid.length;
        int n = grid[0].length;
        
        List<List<Integer>> res = new ArrayList();
        for (int i = 0; i < m; i++) {
            List<Integer> row = new ArrayList();
            for (int j = 0; j < n; j++) {
                row.add(0);
            
}
            res.add(row);
        }
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int pos = xyToPos(i, j, n);
                pos += k;
                int[] xy = posToXy(pos, m, n);
                
res.get(xy[0]).set(xy[1], grid[i][j]); 
            }
        }
        return res;
    }
    
    public int xyToPos(int x, int y, int n) {
        return x * n + y;
    }
    
    public int[] posToXy(int pos, int m, int n) {
        pos %= m * n;
        
        int x = pos / n;
        int y = pos % n;
        return new int[] {x, y};
    }
}
