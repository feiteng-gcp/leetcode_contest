class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int m = grid.length;
        int n = grid[0].length;
        k %= (m * n);
        List<List<Integer>> result = new ArrayList();
        int start = m * n - k;
        start %= (m * n);
        for (int i = 0; i < m; i++) {
            List<Integer> list = new ArrayList<>();
            for (int j = 0; j < n; j++) {
                //System.out.println(start + " " + start / n + " " + start % n);
                list.add(grid[start / n][start % n]);
                start++;
                if (start >= m * n) start -= m * n;
            }
            result.add(list);
        }
        return result;
    }
}
