class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int rows = grid.length, columns = grid[0].length;
        List<Integer> list1D = new ArrayList<Integer>();
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < columns; j++)
                list1D.add(grid[i][j]);
        }
        int size = list1D.size();
        k %= size;
        int beginIndex = (size - k) % size;
        List<List<Integer>> ret = new ArrayList<List<Integer>>();
        for (int i = 0; i < rows; i++)
            ret.add(new ArrayList<Integer>());
        int retIndex = 0;
        for (int i = beginIndex; i < size; i++, retIndex++) {
            int num = list1D.get(i);
            ret.get(retIndex / columns).add(num);
        }
        for (int i = 0; i < beginIndex; i++, retIndex++) {
            int num = list1D.get(i);
            ret.get(retIndex / columns).add(num);
        }
        return ret;
    }
}
