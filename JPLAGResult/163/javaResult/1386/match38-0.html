class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int R = grid.length;
        int C = grid[0].length;
        Queue<Integer> q = new LinkedList<>();
        for (int i = R - 1; i >= 0; i--) {
            for (int j = C - 1; j >= 0; j--) {
                q.offer(grid[i][j]);
            }
        }
        for (int i = 0; i < k; i++) {
            q.offer(q.poll());
        }
        
        List<List<Integer>> ret = new ArrayList<>();
        for (int i = 0; i < R; i++) {
            List<Integer> part = new ArrayList<>();
            for (int j = 0; j < C; j++) {
                part.add(0, q.poll());
            }
            ret.add(0, part);
        }
        return ret;
    }
}
