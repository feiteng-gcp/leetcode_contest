class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int m = grid.length;
        int n = grid[0].length;
        List<List<Integer>> res = new ArrayList<>();
        
        k = k%(m*n);
        
        while(k>0){
            int[][] tmp = new int[m][n];
            for(int i=0 ; i<m ; i++){
                for(int j=0 ; j<n ; j++){
                    if(i==m-1 && j==n-1){
                        tmp[0][0] = grid[i][j];
                    } else if(j==n-1){
                        tmp[i+1][0] = grid[i][j];
                    } else{
                        tmp[i][j+1] = grid[i][j];
                    }
                }
            }
            grid = tmp;
            --k;
        }
        for(int i=0 ; i<m ; i++){
            List<Integer> list = new ArrayList<>();
            for(int j=0 ; j<n ; j++){
                list.add(grid[i][j]);
            }
            res.add(list);
        }
        return res;
    }
}
