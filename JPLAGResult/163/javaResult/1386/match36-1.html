class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int n = grid.length;
        int m = grid[0].length;
        int[][] curr = grid;
        int[][] next = new int[n][m];
        for (int t = 0; t < k; t++) {
            next = new int[n][m];
            next[0][0] = curr[n - 1][m - 1];
            for (int i = 0; i < n - 1; i++) {
                next[i + 1][0] = curr[i][m - 1];
            }
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < m - 1; j++) {
                    next[i][j + 1] = curr[i][j];
                }
            }
            curr = next;
        }
        
List<List<Integer>> ans = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            List<Integer> row = new ArrayList<>();
            for (int j = 0; j < m; j++) {
                
row.add(curr[i][j]);
            }
            ans.add(row);
        }
        return ans;
    }
}
