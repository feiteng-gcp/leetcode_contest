class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        final int N = grid.length;
        final int M = grid[0].length;
        final int X = N * M;
        int[] A = new int[X];
        int idx = 0;
        for (int[] row : grid) {
            for (int val : row) {
                A[idx++] = val;
            }
        }
        int[] S = new int[X];
        for (int i = 0; i < X; ++i) {
            int pos = (((i - k) % X) + X) % X;
            S[i] = A[pos];
        }
        
ArrayList<List<Integer>> ans = new ArrayList<>();
        for (int r = 0; r < N; ++r) {
            ArrayList<Integer> row = new ArrayList<>();
            for (int c = 0; c < M; ++c) {
                int pos = (r * M) + c;
                
row.add(S[pos]);
            }
            ans.add(row);
        }
        return ans;
    }
}
