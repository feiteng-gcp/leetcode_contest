class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        List<List<Integer>> res = new ArrayList<>();
        int m = grid.length, n = grid[0].length;
        k = k % (m * n);
        int s = (m * n - k) % (m * n);
        
for (int i = 0; i < m; i++) {
            List<Integer> temp = new ArrayList<>();
            for (int j = 0; j < n; j++) {
                temp.add(grid[s / n][s % n]);
                s = (s + 1) % (m * n);
            
}
            res.add(temp);
        }
        return res;
    }
}
