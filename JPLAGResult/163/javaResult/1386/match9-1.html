class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int m = grid.length, n = grid[0].length;
        k %= m * n;
        
        List<List<Integer>> res = new ArrayList<>();
        for(int i = 0; i < m; i++) {
            List<Integer> list = new ArrayList<>();
            for(int j = 0; j < n; j++) {
                int pos = (i * n + j - k + m * n) % (m * n);
                list.add(grid[pos / n][pos % n]);
            }
            res.add(list);
        }
        return res;
    }
}
