class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int n = grid.length;
        int m = grid[0].length;
        int total = n * m;
        k = k % total;
        
List<List<Integer>> result = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            result.add(new ArrayList<>());
            for (int j = 0; j < m; ++j) {
                result.get(i).add(0);
            }
        }
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                int next = (i * m + j + k) % total;
                int ni = next / m;
                int nj = next % m;
                
result.get(ni).set(nj, grid[i][j]);
            }
        }
        return result;
    }
}
