class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        for (
int i = 0; i < k; i++) {
            int last = grid[grid.length - 1][grid[0].length - 1];
            for (int j = 0; j < grid.length; j++) {
                int nlast = grid[j][grid[0].length - 1];
                for (int l = grid[0].length - 1; l >= 1; l--) {
                    grid[j][l] = grid[j][l - 1];
                }
                grid[j][0] = last;
                last = nlast;
            }
        
}
        List<List<Integer>> ret = new ArrayList<>();
        for(int[] r : grid){
            List<Integer> l = new ArrayList<>();
            for(int n : r) l.add(n);
            ret.add(l);
        }
        return ret;
    }
}
