class Solution {
     public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        List<List<Integer>> g = new ArrayList<>();
        for (int j = 0; j < grid[0].length; j++) {
            List<Integer> tmp = new ArrayList<>();
            for (int i = 0; i < grid.length; i++) {
                tmp.add(grid[i][j]);
            }
            g.add(tmp);
        }
        
        List<List<Integer>> res = shiftGrid(g, k);
        List<List<Integer>> ans = new ArrayList<>();
        for (int j = 0; j < res.get(0).size(); j++) {
            List<Integer> tmp = new ArrayList<>();
            for (int i = 0; i < res.size(); i++) {
                tmp.add(res.get(i).get(j));
            }
            ans.add(tmp);
        }
        return ans;
    }
    
    List<List<Integer>> shiftGrid(List<List<Integer>> grid, int k) {
        if (k == 0) {
            return grid;
        }
        List<Integer> end = grid.remove(grid.size() - 1);
        int endOfEnd = end.remove(end.size() - 1);
        end.add(0, endOfEnd);
        grid.add(0, end);
        return shiftGrid(grid, k - 1);
    }
}
