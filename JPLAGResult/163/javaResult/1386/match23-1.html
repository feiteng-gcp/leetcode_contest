class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        List<List<Integer>> res = new ArrayList<>();
        int n = grid[0].length, m = grid.length;
        int[][] copyGrid = new int[m][n];
        for (int i = 0; i < m * n; i++) {
            int cur = (i + k ) % (m * n);
            int row = cur / n, col = cur % n;
            copyGrid[row][col] = grid[i / n][i % n];
        
}
        for (int i = 0; i < m ; i++) {
            List<Integer> r = new ArrayList<>();
            for (int j = 0; j < n; j++) {
                r.add(copyGrid[i][j]);
            }
            res.add(r);
        }
        return res;
    }
}
