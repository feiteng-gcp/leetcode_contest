class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int m = grid.length;
        int n = grid[0].length;
        int[] flat = new int[m * n];
        for (int i = 0;i < m;i++) {
            for (int j = 0;j < n;j++) {
                flat[i * n + j] = grid[i][j];
            }
        }
        shiftK(flat, k);
        for (int i = 0;i < flat.length;i++) {
            int r = i / n;
            int c = i % n;
            grid[r][c] = flat[i];
        
}
        List<List<Integer>> res = new ArrayList<>();
        for (int i = 0;i < m;i++) {
            List<Integer> row = new ArrayList<>();
            for (int j = 0;j < n;j++) {
                row.add(grid[i][j]);
            }
            res.add(row);
        }
        return res;
    }
    
    private void shiftK(int[] arr, int k) {
        int n = arr.length;
        k = k % n;
        reverse(arr, 0, n - 1);
        reverse(arr, 0, k - 1);
        reverse(arr, k, n - 1);
    }
    
    private void reverse(int[] arr, int start, int end) {
        int l = start, r = end;
        while (l < r) {
            int temp = arr[l];
            arr[l] = arr[r];
            arr[r] = temp;
            l++;
            r--;
        }
    }
}
