class Solution {
    public List<List<Integer>> shiftGrid(int[][] grid, int k) {
        int n = grid.length;
        int m = grid[0].length;
        int all = n * m;
        List<List<Integer>> ret = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            ret.add(new ArrayList<>());
            for (int j = 0; j < m; j++) {
                int index = i * m + j;
                int pIndex = (index - k) % all;
                if (pIndex < 0) {
                    pIndex += all;
                }
                int x = pIndex / m;
                int y = pIndex % m;
                
ret.get(i).add(grid[x][y]);
            }
        }
        return ret;
    }
}
