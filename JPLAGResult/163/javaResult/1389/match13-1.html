class Solution {
    int R, C;
    int startX, startY, endX, endY;
    int beginX, beginY;
    
int[] d = {0, -1, 0, 1, 0};
    char[] P = {'N', 'S', 'E', 'W'};
    char[] M = {'n', 's', 'e', 'w'};
    Node f = new Node(0, 0, 0, 0, "");
    Node g = new Node(0, 0, 0, 0, "");
    node1 F = new node1(0, 0, "");
    node1 G = new node1(0, 0, "");
    boolean vis[][][];
    int res = -1;

    public boolean legal(int x, int y) {
        return x >= 0 && x < R && y >= 0 && y < C;
    }

    public boolean SToB(int bx, 
int by, int ex, int ey, char grid[][]) {

        int[][] Mark1 = new int[R][C];

        Queue<node1> P = new LinkedList<node1>();
        Mark1[bx][by] = 1;
        Mark1[f.bx][f.by] = 1;
        F.x = bx;
        F.y = by;
        F.ans = "";
        if (bx == ex && by == ey) {
            return true;
        
}
        P.offer(new node1(F.x, F.y, F.ans));
        while (!P.isEmpty()) {
            F = P.poll();
            for (int i = 0; i < 4; ++i) {
                G.x = F.x + d[i];
                G.y = F.y + d[i + 1];
                if (!legal(G.x, G.y) || grid[G.x][G.y] == '#') continue;
                if (Mark1[G.x][G.y] == 0) {
                    Mark1[G.x][G.y] = 1;
                    G.ans = F.ans + M[i];
                    if (G.x == ex && G.y == ey) {
                        F = G;
                        return true;
                    }
                    P.add(new node1(G.x, G.y, G.ans));
                }
            }
        }
        return false;
    }

    public boolean bfs(char grid[][]) {
        Queue<Node> Q = new LinkedList<>();
        f.bx = startX;
        f.by = startY;
        f.px = beginX;
        f.py = beginY;
        f.ans = "";
        Node node = new Node(f.bx, f.by, f.px, f.py, f.ans);
        Q.offer(node);
        Map<Node, Integer> cnt = new HashMap<>();
        cnt.put(node, 0);
        while (!Q.isEmpty()) 
{
            f = Q.poll();
            for (int i = 0; i < 4; ++i) {
                int newx = f.bx + d[i];
                int newy = f.by + d[i + 1];
                int tx = f.bx - d[i];
                int ty = f.by - d[i + 1];
                if (!legal(newx, newy) || grid[newx][newy] == '#' || !legal(tx, ty)
                        || grid[tx][ty] == '#' || vis[newx][newy][i]) continue;
                if (SToB(f.px, f.py, tx, ty, grid)) {
                    g.bx = newx;
                    g.by = newy;
                    g.px = f.bx;
                    g.py = f.by;
                    g.ans = f.ans + F.ans + P[i];
                    if (g.bx == endX && g.by == endY) {
                        res = cnt.get(f) + 1;
                        return true;
                    }
                    vis[newx][newy][i] = true;
                    node = new Node(g.bx, g.by, g.px, g.py, g.ans);
                    cnt.put(node, cnt.get(f) + 1);
                    Q.add(node);
                }
            }
        }
        return false;
    }


    
class node1 {
        int x;
        int y;
        String ans;

        public node1(int x, int y, String ans) {
            this.x = x;
            this.y = y;
            this.ans = ans;
        }
    }

    class Node {
        int bx;
        
int by;
        int px;
        int py;
        int len = 0;
        String ans;

        public Node(int bx, int by, int px, int py, String ans) {
            this.bx = bx;
            this.by = by;
            this.px = px;
            this.py = py;
            this.ans = ans;
        }
    }

    public int minPushBox(char[][] grid) {
        R = grid.length;
        C = grid[0].length;
        vis = new boolean[R][C][4];
        
for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                if (grid[i][j] == 'B') {
                    startX = i;
                    startY = j;
                }
                if (grid[i][j] == 'T') {
                    endX = i;
                    endY = j;
                }
                if (grid[i][j] == 'S') {
                    beginX = i;
                    beginY = j;
                }
            }
        }
        System.out.println(bfs(grid));
        System.out.println(g.ans);
        return res;
    }
}
