class Solution {

  public int minPushBox(char[][] g) {
    Queue<char[][]> q = new ArrayDeque<>();
    q.add(g);
    Set<String> set = new HashSet<>();
    for (int i = 0; !q.isEmpty(); i++)
      for (int j = q.size(); j > 0; j--) {
        char[][] h = q.remove();
        if (!set.contains(Arrays.deepToString(h))) {
          set.add(Arrays.deepToString(h));
          if (e(h))
            return i;
          else {
            int[] n = n(h);
            Arrays.copyOf(h, h.length);
            if (n[0] + 1 < g.length && g[n[0] + 1][n[1]] != '#' && p(c(h), n[0] - 1, n[1], n[2], n[3])) {
              char[][] c = c(h);
              c[n[0]][n[1]] = c[n[2]][n[3]] = '.';
              c[n[0] + 1][n[1]] = 'B';
              c[n[0] - 1][n[1]] = 'S';
              q.add(c);
            }
            if (n[1] + 1 < g[0].length && g[n[0]][n[1] + 1] != '#' && p(c(h), n[0], n[1] - 1, n[2], n[3])) {
              char[][] c = c(h);
              c[n[0]][n[1]] = c[n[2]][n[3]] = '.';
              c[n[0]][n[1] + 1] = 'B';
              c[n[0]][n[1] - 1] = 'S';
              q.add(c);
            }
            if (n[1] - 1 >= 0 && g[n[0]][n[1] - 1] != '#' && p(c(h), n[0], n[1] + 1, n[2], n[3])) {
              char[][] c = c(h);
              c[n[0]][n[1]] = c[n[2]][n[3]] = '.';
              c[n[0]][n[1] - 1] = 'B';
              c[n[0]][n[1] + 1] = 'S';
              q.add(c);
            }
            if (n[0] - 1 >= 0 && g[n[0] - 1][n[1]] != '#' && p(c(h), n[0] + 1, n[1], n[2], n[3])) {
              char[][] c = c(h);
              c[n[0]][n[1]] = c[n[2]][n[3]] = '.';
              c[n[0] - 1][n[1]] = 'B';
              c[n[0] + 1][n[1]] = 'S';
              q.add(c);
            }
          }
        }
      }
    return -1;
  }

  private boolean p(char[][] g, int a, int b, int c, int d) {
    if (a == c && b == d)
      return true;
    else if (a >= 0 && b >= 0 && a < g.length && b < g[0].length && (g[a][b] == '.' || g[a][b] == 'T')) {
      g[a][b] = '#';
      return p(g, a - 1, b, c, d) || p(g, a, b - 1, c, d) || p(g, a, b + 1, c, d) || p(g, a + 1, b, c, d);
    }
    return false;
  }

  private int[] n(char[][] g) {
    int[] ans = new int[4];
    for (int i = 0; i < g.length; i++)
      for (int j = 0; j < g[0].length; j++)
        if (g[i][j] == 'S') {
          ans[2] = i;
          ans[3] = j;
        } else if (g[i][j] == 'B') {
          ans[0] = i;
          ans[1] = j;
        }
    return ans;
  }

  private boolean e(char[][] grid) {
    for (char[] a : grid)
      for (char b : a)
        if (b == 'T')
          return false;
    return true;
  }

  private char[][] c(char[][] g) {
    char[][] clone = new char[g.length][g[0].length];
    for (int i = 0; i < clone.length; i++)
      for (int j = 0; j < clone[0].length; j++)
        clone[i][j] = g[i][j];
    return clone;
  }
}
