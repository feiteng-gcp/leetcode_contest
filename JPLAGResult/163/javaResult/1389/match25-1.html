class Solution {
    public int minPushBox(char[][] grid) {
        int n = grid.length, m = grid[0].length;
        int[] T, B, S;
        T = B = S = new int[]{-1, -1};
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 'T') T = new int[]{i, j};
                else if (grid[i][j] == 'B') B = new int[]{i, j};
                else if (grid[i][j] == 'S') S = new int[]{i, j};
            }
        }

        
BiPredicate<Integer, Integer> available = (i, j) -> {
            if (i < 0 || i >= n) return false;
            if (j < 0 || j >= m) return false;
            return grid[i][j] != '#';
        };

        int[][][][] steps = new int[n][m][n][m];
        steps[S[0]][S[1]][B[0]][B[1]] = 1;

        
int[][] directs = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{S[0], S[1], B[0], B[1]});
        int[] t;
        int sx, sy, bx, by;
        int ans = Integer.MAX_VALUE;
        while (!queue.isEmpty()) {
            t = 
queue.poll();
            if (t[2] == T[0] && t[3] == T[1]) {
                ans = Math.min(ans, steps[t[0]][t[1]][t[2]][t[3]] - 1);
                continue;
            }
            for (int[] d : directs) {
                sx = t[0] + d[0];
                sy = t[1] + d[1];
                if (!available.test(sx, sy)) continue;
                bx = t[2];
                by = t[3];
                if (sx == bx && sy == by) {
                    bx += d[0];
                    by += d[1];
                    if (!available.test(bx, by)) continue;
                    if (steps[sx][sy][bx][by] != 0 && steps[sx][sy][bx][by] <= steps[t[0]][t[1]][t[2]][t[3]] + 1) continue;
                    steps[sx][sy][bx][by] = steps[t[0]][t[1]][t[2]][t[3]] + 1;
                } else {
                    if (steps[sx][sy][bx][by] != 0 && steps[sx][sy][bx][by] <= steps[t[0]][t[1]][t[2]][t[3]]) continue;
                    steps[sx][sy][bx][by] = steps[t[0]][t[1]][t[2]][t[3]];
                }
                if (steps[sx][sy][bx][by] >= ans) continue;
                queue.offer(new int[]{sx, sy, bx, by});
            
}
        }
        return ans == Integer.MAX_VALUE ? -1 : ans;
    }
}
