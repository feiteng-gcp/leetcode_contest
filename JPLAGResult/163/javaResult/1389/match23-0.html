class Solution {
    private static final int INF = 111111111;
    
    private static final int[] DR = {-1, 0, 1, 0};
    private static final int[] DC = {0, 1, 0, -1};
    
    public int minPushBox(char[][] grid) {
        final int N = grid.length;
        final int M = grid[0].length;
        int[][][][] dp = new int[N][M][N][M];
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < M; ++j) {
                for (int k = 0; k < N; ++k) {
                    Arrays.fill(dp[i][j][k], INF);
                }
            }
        }
        ArrayDeque<int[]> q = new ArrayDeque<>();
        int rBox0 = -1, cBox0 = -1;
        int rGuy0 = -1, cGuy0 = -1;
        int rTarget = -1, cTarget = -1;
        for (int r = 0; r < N; ++r) {
            for (int c = 0; c < M; ++c) {
                if (grid[r][c] == 'B') {
                    rBox0 = r;
                    cBox0 = c;
                } else if (grid[r][c] == 'S') {
                    rGuy0 = r;
                    cGuy0 = c;
                } else if (grid[r][c] == 'T') {
                    rTarget = r;
                    cTarget = c;
                }
            }
        }
        q.offerFirst(new int[] {rBox0, cBox0, rGuy0, cGuy0});
        dp[rBox0][cBox0][rGuy0][cGuy0] = 0;
        while (!q.isEmpty()) {
            int[] args = q.pollFirst();
            int rBox = args[0];
            int cBox = args[1];
            int rGuy = args[2];
            int cGuy = args[3];
            int moves = dp[rBox][cBox][rGuy][cGuy];
            for (int d = 0; d < DR.length; ++d) {
                int nrGuy = rGuy + DR[d];
                int ncGuy = cGuy + DC[d];
                if (!valid(grid, nrGuy, ncGuy)) {
                    continue;
                }
                if (nrGuy == rBox && ncGuy == cBox) {
                    int nrBox = rBox + DR[d];
                    int ncBox = cBox + DC[d];
                    if (!valid(grid, nrBox, ncBox)) {
                        continue;
                    }
                    if (dp[nrBox][ncBox][nrGuy][ncGuy] == INF) {
                        q.offerLast(new int[] {nrBox, ncBox, nrGuy, ncGuy});
                        dp[nrBox][ncBox][nrGuy][ncGuy] = moves + 1;
                    }
                } else {
                    if (dp[rBox][cBox][nrGuy][ncGuy] == INF) {
                        q.offerFirst(new int[] {rBox, cBox, nrGuy, ncGuy});
                        dp[rBox][cBox][nrGuy][ncGuy] = moves;
                    }
                }
            }
        }
        int best = INF;
        for (int r = 0; r < N; ++r) {
            for (int c = 0; c < M; ++c) {
                best = Math.min(best, dp[rTarget][cTarget][r][c]);
            }
        }
        if (best == INF) {
            return -1;
        }
        return best;
    }
    
    private static boolean valid(char[][] grid, int r, int c) {
        if (r < 0 || r >= grid.length || c < 0 || c >= grid[r].length) {
            return false;
        }
        return grid[r][c] != '#';
    }
}
