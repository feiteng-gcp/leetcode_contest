class Solution {
    char[][] arr;
    int t;
    int co = 4000;
    public int minPushBox(char[][] grid) {
        HashSet<Integer> vst = new HashSet<>();
        arr = grid;
        int m = grid.length;
        int n = grid[0].length;
        State cur = new State();
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if(arr[i][j] == 'T') {
                    t = i * n + j;
                    arr[i][j] = '.';
                } else if(arr[i][j] == 'S') {
                    cur.player = i * n + j;
                    arr[i][j] = '.';
                } else if(arr[i][j] == 'B') {
                    cur.box = i * n + j;
                    arr[i][j] = '.';
                }
            }
        }
        PriorityQueue<State> pq = new PriorityQueue<>(11, new Comparator<State>(){
            public int compare(State a, State b) {
                return a.step - b.step;
            }
        });
        pq.add(cur);
        vst.add(cur.player * co + cur.box);
        int[] dx = {0, 0, 1, -1};
        int[] dy = {1, -1, 0, 0};
        while(!pq.isEmpty()) {
            State cx = pq.poll();
            int cp = cx.player;
            int cb = cx.box;
            int ax = cp / n;
            int ay = cp % n;
            int bx = cb / n;
            int by = cb % n;
            for(int k = 0; k < 4; ++k) {
                int nx = ax + dx[k];
                int ny = ay + dy[k];
                if(nx < 0 || ny < 0 || nx ==m || ny == n || arr[nx][ny] != '.') continue;
                int np = nx * n + ny;
                if(np != cb) {
                    if(vst.contains(np * co + cb)) continue;
                    State nst = new State();
                    nst.player = np;
                    nst.box = cb;
                    nst.step = cx.step;
                    vst.add(np * co + cb);
                    pq.add(nst);
                } else {
                    int nbx = bx + dx[k];
                    int nby = by + dy[k];
                    if(nbx < 0 || nby < 0 || nbx ==m || nby == n || arr[nbx][nby] != '.') continue;
                    int nbb = nbx * n + nby;
                    if(nbb == t) return cx.step + 1;
                    if(vst.contains(np * co + nbb)) continue;
                    State nst = new State();
                    nst.player = np;
                    nst.box = nbb;
                    nst.step = cx.step + 1;
                    vst.add(np * co + nbb);
                    pq.add(nst);
                }
            }
        }
        return -1;
    }
    class State {
        int player;
        int box;
        int step;
    }
}
