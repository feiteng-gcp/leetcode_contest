class Solution {
    int m, n;
    int dir[][] = {{0, 1}, {0, -1}, {-1, 0}, {1, 0}};
    private HashSet<String> visited = new HashSet<>();
    
    public int minPushBox(char[][] grid) {
        m = grid.length;
        n = grid[0].length;
        
int targetX = -1;
        int targetY = -1;
        int[] startPos = new int[5];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 'B') {
                    startPos[0] = i;
                    startPos[1] = j;
                    grid[i][j] = '.';
                } else if (grid[i][j] == 'S') {
                    startPos[2] = i;
                    
startPos[3] = j;
                    grid[i][j] = '.';
                } else if (grid[i][j] == 'T') {
                    targetX = i;
                    targetY = j;
                }
            }
        }

        LinkedList<int[]> q = new LinkedList<>();
        q.add(startPos);
        visited.add(encode(startPos));
        while (!q.isEmpty()) {
            int[] top = q.poll();
            int boxX = top[0];
            int boxY = top[1];
            int peopleX = top[2];
            int peopleY = top[3];
            int currentStep = top[4];
            if (boxX == targetX && boxY == targetY) {
                return currentStep;
            }
            for (int i = 0; i < 4; i++) {
                int nx = boxX + dir[i][0];
                int ny = boxY + dir[i][1];
                int nBoxX = boxX - dir[i][0];
                int nBoxY = boxY - dir[i][1];

                if (nx >= 0 && nx < m && ny >= 0 && ny < n && grid[nx][ny] != '#') {
                    if (nBoxX >= 0 && nBoxX < m && nBoxY >= 0 && nBoxY < n && grid[nBoxX][nBoxY] != '#') {
                        grid[boxX][boxY] = 'B';
                        if (canAccess(peopleX, peopleY, nx, ny, grid)) {
                            int[] next = new int[] {nBoxX, nBoxY, boxX, boxY, currentStep + 1};
                            String key = encode(next);
                            if (!visited.contains(key)) {
                                visited.add(key);
                                q.addLast(next);
                            }
                        }
                        
grid[boxX][boxY] = '.';
                    }
                }
            }
        }
        return -1;
    }

    private String encode(int[] pos) {
        StringBuilder sb = new StringBuilder();
        sb.append(pos[0]);
        sb.append(",");
        sb.append(pos[1]);
        sb.append(",");
        sb.append(pos[2]);
        sb.append(",");
        sb.append(pos[3]);
        return sb.toString();
    }

    private boolean canAccess(int startX, int startY, int endX, int endY, char[][] grid) {
        LinkedList<int[]> q = new LinkedList<>();
        boolean[][] visited = new boolean[m][n];
        
q.add(new int[] {startX, startY});
        visited[startX][startY] = true;
        while (!q.isEmpty()) {
            int[] top = q.pollFirst();
            if (top[0] == endX && top[1] == endY) {
                return true;
            }
            for (int i = 0; i < 4; i++) {
                int nx = top[0] + dir[i][0];
                int ny = top[1] + dir[i][1];
                
if (nx >= 0 && nx < m && ny >= 0 && ny < n && !visited[nx][ny] && grid[nx][ny] != '#' && grid[nx][ny] != 'B') {
                    q.add(new int[] {nx, ny});
                    visited[nx][ny] = true;
                }
            }
        }
        return false;
    }
}
