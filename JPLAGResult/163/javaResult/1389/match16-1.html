class Solution {
    int[][] dir = new int[][]{{0,1}, {0,-1}, {1,0}, {-1,0}};
    
    public int minPushBox(char[][] grid) {
        int m = grid.length, n = grid[0].length;
        
int tx = -1, ty = -1, bx = -1, by = -1, sx = -1, sy = -1;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                char c = grid[i][j];
                
if (c == 'T') {
                    tx = i;
                    ty = j;
                } else if (c == 'B') {
                    bx = i;
                    by = j;
                } else if (c == 'S') {
                    sx = i;
                    sy = j;
                }
            }
        }
        
        Set<Integer> visited = new HashSet<>();
        Queue<Integer> q = new ArrayDeque<>();
        q.add((bx * 22 + by) * 500 + (sx * 22 + sy));
        visited.add((bx * 22 + by) * 500 + (sx * 22 + sy));
        
        int lv = 0;
        while (!q.isEmpty()) {
            int cnt = q.size();
            while (cnt > 0) {
                cnt--;
                int t = q.poll();
                int l = t / 500, o = t % 500;
                int a = l / 22, b = l % 22, c = o / 22, d = o % 22;
                if (a == tx && b == ty) {
                    return lv;
                }
                for (int[] dr : dir) {
                    if (reachable(grid, c, d, a + dr[0], b + dr[1], a, b)) {
                        if (isValid(grid, a - dr[0], b - dr[1])) {
                            int code1 = (a - dr[0]) * 22 + b - dr[1], code2 = a * 22 + b;
                            int code = code1 * 500 + code2;
                            if (!visited.contains(code)) {
                                visited.add(code);
                                q.add(code);
                            }
                        }
                    }
                }
            }
            lv++;
        }
        return -1;
    }
    
    private boolean reachable(char[][] grid, int c, int d, int a1, int b1, int a, int b) {
        int m = grid.length, n = grid[0].length;
        if (a1 < 0 || a1 >= m || b1 < 0 || b1 >= n || grid[a1][b1] == '#') return false;
        if (a1 == a && b1 == b) return false;
        Set<Integer> set = new HashSet<>();
        Queue<Integer> q = new ArrayDeque<>();
        q.add(c * 22 + d);
        while (!q.isEmpty()) {
            int cur = q.poll();
            int x = cur / 22, y = cur % 22;
            if (x == a1 && y == b1) return true;
            for (int[] dr : dir) {
                if(valid(grid, x + dr[0], y + dr[1], a, b)) {
                    int code = (x + dr[0]) * 22 + y + dr[1];
                    if (!set.contains(code)) {
                        set.add(code);
                        q.add(code);
                    }
                }
            }
        }
        return false;
    }
    
    private boolean valid(char[][] grid, int x, int y, int a, int b) {
        if (x == a && y == b) return false;
        int m = grid.length, n = grid[0].length;
        if (x < 0 || x >= m || y < 0 || y >= n || grid[x][y] == '#') return false;
        return true;
    }
    
    private boolean isValid(char[][] grid, int x, int y) {
        int m = grid.length, n = grid[0].length;
        if (x < 0 || x >= m || y < 0 || y >= n || grid[x][y] == '#') return false;
        return true;
    }
    
    
}
