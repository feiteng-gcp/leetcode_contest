class Solution {
    public int maxSumDivThree(int[] nums) {
        int n = nums.length;
        int sum = Arrays.stream(nums).sum();
        int remain = sum % 3;
        if (remain == 0) {
            return sum;
        }
        Arrays.sort(nums);
        if (remain == 1) {
            // find the min 1
            int min = Integer.MAX_VALUE;
            for (int i = 0; i < n; i++) {
                if (nums[i] % 3 == 1) {
                    min = nums[i];
                    break;
                }
            }
            for (int i = 0; i < n - 1; i++) {
                for (int j = i + 1; j < n; j++) {
                    if ((nums[i] + nums[j]) % 3 == 1) {
                        if (nums[i] + nums[j] < min) {
                            min = nums[i] + nums[j];
                        }
                        break;
                    }
                }
            }
            if (min == Integer.MAX_VALUE) {
                return 0;
            } else {
                return sum - min;
            }
        } else {
            int min = Integer.MAX_VALUE;
            for (int i = 0; i < n; i++) {
                if (nums[i] % 3 == 2) {
                    min = nums[i];
                    break;
                }
            }
            for (int i = 0; i < n - 1; i++) {
                for (int j = i + 1; j < n; j++) {
                    if ((nums[i] + nums[j]) % 3 == 2) {
                        if (nums[i] + nums[j] < min) {
                            min = nums[i] + nums[j];
                        }
                        break;
                    }
                }
            }
            if (min == Integer.MAX_VALUE) {
                return 0;
            } else {
                return sum - min;
            }
        }
    }
}
