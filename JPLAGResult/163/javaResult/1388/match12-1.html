class Solution {
    public int maxSumDivThree(int[] nums) {
        int sum = 0, m11 = 10001, m12 = m11, m21 = m11, m22 = m11;
        for (int n : nums) {
            sum += n;
            if (n % 3 == 1) {
                if (n < m11) {
                    m12 = m11;
                    m11 = n;
                } else if (n < m12) m12 = n;
            } else if (n % 3 == 2) {
                if (n < m21) {
                    m22 = m21;
                    m21 = n;
                } else if (n < m22) m22 = n;
            }
        }
        if (sum % 3 == 1) {
            sum -= Math.min(m11, m21 + m22);
        } else if (sum % 3 == 2) {
            sum -= Math.min(m21, m11 + m12);
        }
        return sum;
    }
}
