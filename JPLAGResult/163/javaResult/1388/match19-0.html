class Solution {
    public int maxSumDivThree(int[] nums) {
		List<Integer> remind1 = new ArrayList<>();
		List<Integer> remind2 = new ArrayList<>();
		int sum = 0;
		for (int i = 0; i < nums.length; i++) {
			if (nums[i] % 3 == 1) {
				remind1.add(nums[i]);
			} else if (nums[i] % 3 == 2) {
				remind2.add(nums[i]);
			}

			sum += nums[i];
		}
		if (sum % 3 == 0) return sum;
		Collections.sort(remind1);
		Collections.sort(remind2);

		if (sum % 3 == 1) {
			int m1 = Integer.MAX_VALUE;
			if (remind1.size() > 0) m1 = remind1.get(0);
			if (remind2.size() > 1) {
				m1 = Math.min(m1, remind2.get(0) + remind2.get(1));
			}
			if (m1 == Integer.MAX_VALUE) return 0;
			return sum - m1;
		} else {
			int m1 = Integer.MAX_VALUE;
			if (remind2.size() > 0) m1 = remind2.get(0);
			if (remind1.size() > 1) {
				m1 = Math.min(m1, remind1.get(0) + remind1.get(1));
			}
			if (m1 == Integer.MAX_VALUE) return 0;
			return sum - m1;
		}
	}
}
