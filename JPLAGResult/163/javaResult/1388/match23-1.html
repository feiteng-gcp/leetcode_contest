class Solution {

	public int maxSumDivThree(int[] nums) {
		int sum = 0;
		PriorityQueue<Integer> queue1 = new PriorityQueue<>(), queue2 = new PriorityQueue<>();
		for (int num :
 nums) {
			sum += num;
			if (num % 3 == 1) {
				queue1.offer(num);
			} else if (num % 3 == 2) {
				queue2.offer(num);
			}
		}
		if (sum % 3 == 0) {
			return sum;
		} else if (sum % 3 == 1) {
			int max = 0;
			if (queue1.size() >= 1) {
				max = Math.max(max, sum - queue1.poll());
			}

			if (queue2.size() >= 2) {
				max = Math.max(max, sum - queue2.poll() - queue2.poll());
			}
			return max;
		} else {
			int max = 0;
			if (queue1.size() >= 
2) {
				max = Math.max(max, sum - queue1.poll() - queue1.poll());
			}
			if (queue2.size() >= 1) {
				max = Math.max(max, sum - queue2.poll());
			}
			return max;
		}
	}
}
