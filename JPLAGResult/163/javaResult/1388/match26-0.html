class Solution {
    public int maxSumDivThree(int[] nums) {
        int sum = 0;
        int min1a = 100000;
        int min1b = 100000;
        int min2a = 100000;
        int min2b = 100000;
        for (int num : nums) {
            sum += num;
            switch (num % 3) {
                case 1:
                    if (num < min1a) {
                        min1b = min1a;
                        min1a = num;
                    } else if (num < min1b) {
                        min1b = num;
                    }
                    break;
                case 2:
                    if (num < min2a) {
                        min2b = min2a;
                        min2a = num;
                    } else if (num < min2b) {
                        min2b = num;
                    }
                    break;
            }
        }
        
        switch (sum % 3) {
            case 1:
                sum -= Math.min(Math.min(min1a, min1b), min2a + min2b);
                break;
            case 2:
                sum -= Math.min(Math.min(min2a, min2b), min1a + min1b);
                break;
        }
        
        return sum;
    }
}
