class Solution {
    public int maxSumDivThree(int[] nums) {
        //dp[i][3] represent the largest sum for the first ith elemnt in the array, with the mod with 0 or 1 or 2;
        
        int l = nums.length;
        int[][] dp = new int[3][l + 1];
        
        for (int i = 1; i <= l; i++) {
            int mod = nums[i - 1] % 3;
            for (int k = 0; k < 3; k++) {
                dp[k][i] = dp[k][i - 1];
            }
            if (mod == 0) {
                dp[0][i] = dp[0][i - 1] + nums[i - 1];
                
                for (int k = 1; k < 3; k++) {
                    if (dp[k][i - 1] == 0) {
                        continue;
                    }
                    dp[k][i] = dp[k][i - 1] + nums[i - 1];
                }
            }
            else if (mod == 1) {
                if (dp[2][i - 1] != 0) {
                    dp[0][i] = Math.max(dp[0][i - 1], dp[2][i - 1] + nums[i - 1]);    
                }
                
                dp[1][i] = Math.max(dp[1][i - 1], dp[0][i - 1] + nums[i - 1]);
                if (dp[1][i - 1] != 0) {
                    dp[2][i] = Math.max(dp[2][i - 1], dp[1][i - 1] + nums[i - 1]);    
                }
                
            }
            else {
                if (dp[1][i - 1] != 0) {
                    dp[0][i] = Math.max(dp[0][i - 1], dp[1][i - 1] + nums[i - 1]);    
                }
                if (dp[2][i - 1] != 0) {
                    dp[1][i] = Math.max(dp[1][i - 1], dp[2][i - 1] + nums[i - 1]);    
                }
                dp[2][i] = Math.max(dp[2][i - 1], dp[0][i - 1] + nums[i - 1]);
            }
        }
        
//         for (int i = 0; i < dp.length; i++) {
//             for (int j = 0; j < dp[0].length; j++) {
//                 System.out.print(dp[i][j] + " ");
                
//             }
//             System.out.println();
//         }
        return dp[0][l];
    }
}
