class Solution {
    
public int maxSumDivThree(int[] nums) {
        final int N = nums.length;
        int[][] dp = new int[N + 1][3];
        for (int i = 0; i <= N; ++i) {
            Arrays.fill(dp[i], -400000000);
        }
        dp[0][0] = 0;
        for (int i = 0; i < N; ++i) {
            // int m = nums[i] % 3;
            // int prev = (((3 - m) % 3) + 3) % 3;
            for (int j = 0; j < 3; ++j) {
                int next = (j + nums[i]) % 3;
                dp[i + 1][next] = Math.max(dp[i + 1][next], dp[i][j] + nums[i]);
                
dp[i + 1][j] = Math.max(dp[i + 1][j], dp[i][j]);
                // System.out.format("dp[%d][%d] = %d\n", i + 1, next, dp[i + 1][next]);
            }
        }
        return dp[N][0];
        // IntList[] remainders = new IntList[3];
        // for (int i = 0; i < remainders.length; ++i) {
        //     remainders[i] = new IntList();
        // }
        // for (int v : nums) {
        //     remainders[v % 3].add(v);
        // }
        // for (IntList lst : remainders) {
        //     Collections.sort(lst, Collections.reverseOrder());
        // }
    }

}
