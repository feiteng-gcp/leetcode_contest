class Solution {
    public int maxSumDivThree(int[] nums) {
        int[] dp = new int[3];
        for(int num:nums){
            int[] next = dp.clone();
            for(int sum:dp){
                int nsum = sum+num;
                int index = nsum%3;
                next[index] = Math.max(nsum, next[index]);
            
}
            dp = next;
        }
        return dp[0];
    }
}
