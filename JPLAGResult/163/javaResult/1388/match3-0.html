class Solution {

  public int maxSumDivThree(int[] nums) {
    int n = nums.length;
    int[][] dp = new int[n + 1][3];
    dp[0][2] = dp[0][1] = -1;
    for (int i = 1; i <= n; i++) {
      for (int j = 0; j <= 2; j++) {
        dp[i][j] = dp[i - 1][j];
        if (dp[i - 1][(j + 3 - nums[i - 1] % 3) % 3] != -1) {
          dp[i][j] = Math.max(dp[i][j], dp[i - 1][(j + 3 - nums[i - 1] % 3) % 3] + nums[i - 1]);
        }
      }
    }
    return dp[n][0];
  }
}
