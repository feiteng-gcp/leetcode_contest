class Solution {
    public int maxSumDivThree(int[] nums) {
        int[][] max = new int[2][3];
        max[0][1] = -1;
        max[0][2] = -1;
        int curr = 0;
        for (int num : nums) {
            curr = 1-curr;
            int prev = 1-curr;
            for (int i=0; i<3; i++) {
                max[curr][i] = max[prev][i];
            }
            for (int i=0; i<3; i++) {
                if (max[prev][i] == -1) continue;
                int sum = (max[prev][i] + num);
                int index = sum % 3;
                if (sum > max[curr][index]) {
                    max[curr][index] = sum;
                }
            }
           /* for (int i=0; i<3; i++) {
                System.out.print(max[curr][i]+" ");
            }
            System.out.println();*/
        }
        return max[curr][0];
    }
}
