class Solution {
    public int maxSumDivThree(int[] nums) {
        int sum = 0;
        PriorityQueue<Integer> pq1 = new PriorityQueue<>();
        PriorityQueue<Integer> pq2 = new PriorityQueue<>();
        for(int num : nums){
            sum += num;
            if(num % 3 == 1)
                pq1.offer(num);
            else if(num % 3 == 2)
                pq2.offer(num);
        }
        if(sum % 3 == 0)
            return sum;
        int res1 = Integer.MIN_VALUE;
        if(sum % 3 == 1){
            if(!pq1.isEmpty())
                res1 = sum - pq1.poll();
            if(pq2.size() >= 2)
                res1 = Math.max(res1, sum - pq2.poll() - pq2.poll());
        }else{
            if(!pq2.isEmpty())
                res1 = sum - pq2.poll();
            if(pq1.size() >= 2)
                res1 = Math.max(res1, sum - pq1.poll() - pq1.poll());
        }
        if(res1 == Integer.MIN_VALUE)
            return 0;
        return res1;
    }
}
