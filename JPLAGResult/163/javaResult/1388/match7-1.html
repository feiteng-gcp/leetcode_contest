import java.util.PriorityQueue;

class Solution {
    public int maxSumDivThree(int[] nums) {

        PriorityQueue<Integer> pq1 = new PriorityQueue<>();
        PriorityQueue<Integer> pq2 = new PriorityQueue<>();
        int sum = 0;
        
for (int n : nums) {
            sum += n;
            if (n % 3 == 1) pq1.add(n);
            if (n % 3 == 2) pq2.add(n);
        }

        if (sum % 3 == 0) return sum;
        else if (sum % 3 == 1) {
            int candidate = Integer.MAX_VALUE;
            if (!pq1.isEmpty()) candidate = 
pq1.poll();
            if (pq2.size() > 1) candidate = Math.min(candidate, pq2.poll() + pq2.poll());
            return sum - candidate;
        } else {
            int candidate = Integer.MAX_VALUE;
            if (!pq2.isEmpty()) candidate = 
pq2.poll();
            if (pq1.size() > 1) candidate = Math.min(candidate, pq1.poll() + pq1.poll());
            return sum - candidate;
        }
    }
}
