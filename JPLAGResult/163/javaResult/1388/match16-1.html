class Solution {
    public int maxSumDivThree(int[] nums) {
        int[] max = new int[3];
        for(int num: nums) {
            int[] newMax = new int[3];
            for(int sum: max) {
                int newSum = sum + num;
                if (newSum % 3 == 0) {
                    newMax[0] = Math.max(newMax[0], Math.max(newSum, max[0]));
                }
                if (newSum % 3 == 1) {
                    newMax[1] = Math.max(newMax[1], Math.max(newSum, max[1]));
                }
                if (newSum % 3 == 2) {
                    newMax[2] = Math.max(newMax[2], Math.max(newSum, max[2]));
                }
            
}
            for(int i = 0; i < 3; i++) {
                max[i] = Math.max(max[i], newMax[i]);
            }
            //set.addAll(newSet);
        }
        return max[0];
    }
}
