class Solution {
    public int maxSumDivThree(int[] nums) {
        Queue<Integer> one = new PriorityQueue<>();
        Queue<Integer> two = new PriorityQueue<>();
        
        int ans = 0;
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] % 3 == 0) {
                ans += nums[i];
            }
            if (nums[i] % 3 == 1) {
                ans += nums[i];
                one.offer(nums[i]);
            }
            if (nums[i] % 3 == 2) {
                ans += nums[i];
                two.offer(nums[i]);
            }
        }
        
        if (ans % 3 == 0) return ans;
        if (ans % 3 == 1) {
            if (one.size() < 1 && two.size() < 2) return 0;
            int a = Integer.MAX_VALUE;
            int b = Integer.MAX_VALUE;
            if (one.size() >= 1) {
                a = one.poll();
            }
            if (two.size() >= 2) {
                b = two.poll() + two.poll();
            }
            return Math.max(ans - a, ans - b);
        }
        
        if (one.size() < 2 && two.size() < 1) return 0;
        int a = Integer.MAX_VALUE;
            int b = Integer.MAX_VALUE;
            if (one.size() >= 2) {
                a = one.poll() + one.poll();
            }
            if (two.size() >= 1) {
                b = two.poll();
            }
            return Math.max(ans - a, ans - b);
    }
}
