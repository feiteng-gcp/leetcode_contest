class Solution {
 public int maxSumDivThree(int[] nums) {
        PriorityQueue<Integer> q1 = new PriorityQueue<>();
        PriorityQueue<Integer> q2 = new PriorityQueue<>();
        int total = 0;
        for (int i = 0; i < nums.length;i++){
            if (nums[i] % 3 ==1){
                q1.offer(nums[i]);
            }
            if (nums[i] % 3 ==2){
                q2.offer(nums[i]);
            }
            total += nums[i];
        }
        if (total % 3 == 1){
            int min = 0;
            if (!q1.isEmpty()){
                min = q1.poll();
            
}
            if (q2.size() >= 2){
                min = Math.min(min, q2.poll() + q2.poll());
            }
            return total - min;
        }
        if (total % 3 == 2){
            int min = 0;
            if (!q2.isEmpty()){
                min = q2.poll();
            
}
            if (q1.size() >= 2){
                min = Math.min(min, q1.poll() + q1.poll());
            }
            return total - min;
        }
        return total;
    }
}
