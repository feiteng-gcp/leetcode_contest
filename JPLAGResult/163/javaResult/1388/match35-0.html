class Solution {
    public int maxSumDivThree(int[] nums) {
        int sum = 0;
        PriorityQueue<Integer> maxHeap1 = new PriorityQueue<>(Collections.reverseOrder());
        PriorityQueue<Integer> maxHeap2 = new PriorityQueue<>(Collections.reverseOrder());
        for (int n : nums) {
            sum += n;
            if (n % 3 ==1) {
                maxHeap1.add(n);
                if (maxHeap1.size()>2) {
                    maxHeap1.poll();
                }
            }
            if (n % 3 ==2) {
                maxHeap2.add(n);
                if (maxHeap2.size()>2) {
                    maxHeap2.poll();
                }
            }
        }
        
        int rem = sum % 3;
        if (rem == 0) {
            return sum;
        }
        
        if (rem == 1) {
            int min = sum;
            if (maxHeap1.size()>1) maxHeap1.poll();
            if (maxHeap1.size()>0) {
                min = Math.min(min, maxHeap1.poll());
            }
            if (maxHeap2.size()>1) {
                min = Math.min(min, maxHeap2.poll() + maxHeap2.poll());
            }
            return sum - min;
        }
        
        if (rem == 2) {
            int min = sum;
            if (maxHeap2.size()>1) maxHeap2.poll();
            if (maxHeap2.size()>0) {
                min = Math.min(min, maxHeap2.poll());
            }
            if (maxHeap1.size()>1) {
                min = Math.min(min, maxHeap1.poll() + maxHeap1.poll());
            }
            return sum - min;
        }
        
        return 0;
    }
}
