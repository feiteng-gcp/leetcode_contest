class Solution {
public int maxSumDivThree(int[] nums) {

        int[] dp = new int[3];

        int n  = nums.length;

        dp[1] = dp[2] = -1;
        dp[nums[0] % 3] = nums[0];
        //dp[]

        for (int i = 1; i < n; i++) {
            //System.out.println("dp[i] =  " + Arrays.toString(dp));
            int r = nums[i] % 3;

            int[] t = new int[3];
            
for (int j = 0; j < 3; j++) {
                t[j] = dp[j];
            }

            for (int j = 0; j < 3; j++) {
                if (dp[j] >= 0) {
                    dp[(j + r) % 3] = Math.max(t[(j + r) % 3], t[j] + nums[i]);
                
}
            }

            //dp = t;
        }

        return dp[0];
    }
}
