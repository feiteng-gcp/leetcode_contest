class Solution {
    public int maxSumDivThree(int[] nums) {
        Arrays.sort(nums);
        
int sum = 0, index = 0;
        
        List<Integer> ones = new ArrayList<>(), twos = new ArrayList<>();
        for(int i = 0; i < nums.length; i++) {
            if((i == 0 || nums[i - 1] < 0) && nums[i] >= 0) {
                index = i;
            }
            if(nums[i] >= 0) 
                sum += nums[i];
            int num = Math.abs(nums[i]);
            if(num % 3 == 1)
                ones.add(num);
            else if(num % 3 == 2)
                twos.add(num);
        }
        Collections.sort(ones);
        Collections.sort(twos);
        
        if(sum % 3 == 0) {
            return sum;
        } else if(sum % 3 == 1) {
            int max = 0;
            if(ones.size() > 0)
                max = 
Math.max(max, sum - ones.get(0));
            if(twos.size() > 1)
                max = Math.max(max, sum - twos.get(0) - twos.get(1));
            return max;
        } else {
            int max = 0;
            if(twos.size() > 0)
                max = Math.max(max, sum - 
twos.get(0));
            if(ones.size() > 1)
                max = Math.max(max, sum - ones.get(0) - ones.get(1));
            return max;
        }
    }
}
