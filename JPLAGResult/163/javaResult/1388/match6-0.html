class Solution {
    public int maxSumDivThree(int[] nums) {
        int[][] dp = new int[nums.length][3];
        
        dp[0][nums[0] % 3] = nums[0];
        for (int i = 1; i < nums.length; i++) {
            dp[i][0] = dp[i-1][0];
            dp[i][1] = dp[i-1][1];
            dp[i][2] = dp[i-1][2];

            for (int j = 0; j < 3; j++) {
                int cur = nums[i] + dp[i-1][j];
                dp[i][cur % 3] = Math.max(dp[i][cur % 3], cur);
            }
        }
        
        return dp[nums.length-1][0];
    }
}
