class Solution {
    public int maxSumDivThree(int[] nums) {
        Arrays.sort(nums);
        int length = nums.length;
        int totalSum = 0;
        for (int i = 0; i < length; i++) {
            int num = nums[i];
            totalSum += num;
        }
        int remainder = totalSum % 3;
        if (remainder == 0)
            return totalSum;
        List<Integer> ones = new ArrayList<Integer>();
        List<Integer> twos = new ArrayList<Integer>();
        int onesCount = 0, twosCount = 0;
        for (int i = 0; i < length; i++) {
            if (onesCount < 2 && nums[i] % 3 == 1) {
                ones.add(nums[i]);
                onesCount++;
            } else if (twosCount < 2 && nums[i] % 3 == 2) {
                twos.add(nums[i]);
                twosCount++;
            }
        }
        if (onesCount == 2 && twosCount == 2) {
            if (remainder == 1) {
                int difference = Math.min(ones.get(0), twos.get(0) + twos.get(1));
                return totalSum - difference;
            } else {
                int difference = Math.min(ones.get(0) + ones.get(1), twos.get(0));
                return totalSum - difference;
            }
        } else if (onesCount == 2) {
            if (twosCount == 1) {
                if (remainder == 1)
                    return totalSum - ones.get(0);
                else
                    return totalSum - Math.min(ones.get(0) + ones.get(1), twos.get(0));
            } else {
                if (remainder == 1)
                    return totalSum - ones.get(0);
                else
                    return totalSum - (ones.get(0) + ones.get(1));
            }
        } else if (twosCount == 2) {
            if (onesCount == 1) {
                if (remainder == 1)
                    return totalSum - Math.min(ones.get(0), twos.get(0) + twos.get(1));
                else
                    return totalSum - twos.get(0);
            } else {
                if (remainder == 1)
                    return totalSum - (twos.get(0) + twos.get(1));
                else
                    return totalSum - twos.get(0);
            }
        } else {
            if (remainder == 1) {
                if (onesCount == 1)
                    return totalSum - ones.get(0);
                else
                    return 0;
            } else {
                if (twosCount == 1)
                    return totalSum - twos.get(0);
                else
                    return 0;
            }
        }
    }
}
