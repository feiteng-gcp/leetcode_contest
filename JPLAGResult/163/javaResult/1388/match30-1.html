class Solution {
    public int maxSumDivThree(int[] nums) {
        int len = nums.length;
        int[][] dp = new int[3][len + 1];
        for(int i = 0; i < len; i++){
            int current = nums[i];
            for(int j = 0; j < 3; j++){
                dp[j][i + 1] = dp[j][i];
            }
            for(int j = 0; j < 3; j++){
                int sum = current + dp[j][i];
                dp[sum % 3][i + 1] = Math.max(dp[sum % 3][i + 1], sum);
                // System.out.println(sum % 3 + " " + i + " " + dp[sum % 3][i + 1]);
            }
        }
        return dp[0][len];
    }
}
