class Solution {
    public int maxSumDivThree(int[] nums) {
        PriorityQueue<Integer> pq1 = new PriorityQueue<>(3, (Integer a, Integer b)->(a-b));
        PriorityQueue<Integer> pq2 = new PriorityQueue<>(3, (Integer a, Integer b)->(a-b));
        
int n = nums.length, sum = 0;
        for (int i = 0; i < n; i++) {
            sum += nums[i];
            if (nums[i] % 3 == 1) {
                pq1.offer(nums[i]);
            } else if (nums[i] % 3 == 2) {
                pq2.offer(nums[i]);
            }
        }
        
        if (sum % 3 == 2) {
            int min = Integer.MAX_VALUE;
            
            
if (pq1.size() >= 2) {
                min = Math.min(min, pq1.poll() + pq1.poll());
            } 
            if (!pq2.isEmpty()) {
                min = Math.min(min, pq2.poll());
            }
            // System.out.println(min);
            sum -= min;
        } else if (sum % 3 == 1) {
            int min = Integer.MAX_VALUE;
            
if (pq2.size() >= 2) {
                min = Math.min(min, pq2.poll() + pq2.poll());
            } 
            if (!pq1.isEmpty()) {
                min = Math.min(min, pq1.poll());
            }
            sum -= min;
        }
        
        return sum;
    }
}
