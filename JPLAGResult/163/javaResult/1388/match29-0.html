class Solution {
        public int maxSumDivThree(int[] nums) {
         int[] a = {Integer.MAX_VALUE, Integer.MAX_VALUE};
         int[] b = {Integer.MAX_VALUE, Integer.MAX_VALUE};
         int sum = 0;
         for (int num : nums) {
             sum += num;
             if (num % 3 == 1) {
                 if (num <= a[0]) {
                     a[1] = a[0];
                     a[0] = num;
                 } else if (num <= a[1]) {
                     a[1] = num;
                 }
             } else if (num % 3 == 2) {
                 if (num <= b[0]) {
                     b[1] = b[0];
                     b[0] = num;
                 } else if (num <= b[1]) {
                     b[1] = num;
                 }
             }
         }

         if (sum % 3 == 0) return sum;
         else if (sum % 3 == 1) {
             int res = 0;
             if (a[0] != Integer.MAX_VALUE) res = sum - a[0];
             if (b[0] != Integer.MAX_VALUE  && b[1] != Integer.MAX_VALUE) res = Math.max(res, sum - b[0] - b[1]);
             return res;
         }
         else {
             int res = 0;
             if (b[0] != Integer.MAX_VALUE) res = sum - b[0];
             if (a[0] != Integer.MAX_VALUE  && a[1] != Integer.MAX_VALUE) res = Math.max(res, sum - a[0] - a[1]);
             return res;
         }
    }
}
