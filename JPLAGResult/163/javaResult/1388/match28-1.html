class Solution {
    public int maxSumDivThree(int[] nums) {
        int[][] dp = new int[nums.length + 1][3];
        for (int i = 0;i<nums.length;i++) {
            int n = nums[i];
            dp[i + 1][0] = dp[i][0];
            dp[i + 1][1] = dp[i][1];
            dp[i + 1][2] = dp[i][2];

            if(n % 3 == 0){
                dp[i + 1][0] += n;
                if(dp[i][1] > 0) dp[i + 1][1] += n;
                if(dp[i][2] > 0) dp[i + 1][2] += n;
            }else if(n % 3 == 1){
                
dp[i + 1][1] = Math.max(dp[i + 1][1],n + dp[i][0]);
                if(dp[i][1] > 0) dp[i + 1][2] = Math.max(dp[i + 1][2],n + dp[i][1]);
                if(dp[i][2] > 0) dp[i + 1][0] = Math.max(dp[i + 1][0],n + dp[i][2]);
            }else{
                if(dp[i][2] > 0) dp[i + 1][1] = Math.max(dp[i + 1][1],n + dp[i][2]);
                dp[i + 1][2] = Math.max(dp[i + 1][2],n + dp[i][0]);
                if(dp[i][1] > 0) dp[i + 1][0] = Math.max(dp[i + 1][0],n + dp[i][1]);
            }
        }
        return dp[nums.length][0];
    }
}
