class Solution {
    public int maxSumDivThree(int[] nums) {
        int sum = 0;
        int a=Integer.MAX_VALUE,b=Integer.MAX_VALUE,c=Integer.MAX_VALUE,d=Integer.MAX_VALUE;
        for (int i : nums){
            sum+=i;
            if (i%3==1){
                if (i<=b) {
                    a=b;
                    b=i;
                } else if (i<a) a=i;
            } else if (i%3==2){
                if (i<=d){
                    c=d;
                    d=i;
                } else if (i<c) c=i;                
            }            
        }
        if (sum%3==1){
            int min = Integer.MAX_VALUE;
            if (b<min) min = b;
            if (c<Integer.MAX_VALUE) min = Math.min(min,c+d);
            sum-=min;
        } else if (sum%3==2){
            int min = Integer.MAX_VALUE;
            if (d<min) min = d;
            if (a<Integer.MAX_VALUE) min = Math.min(min,a+b);
            sum-=min;
        }
            
        return sum;
        
    }
}
