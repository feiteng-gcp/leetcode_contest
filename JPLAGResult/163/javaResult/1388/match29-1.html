class Solution {
    public int maxSumDivThree(int[] nums) {
        int cs = 0;
        int[] one = new int[2];
        int[] two = new int[2];
        
for(int x : nums) {
            cs += x;
            if(x % 3 == 1) {
                if(one[0] == 0 || one[0] >= x) {
                    one[1] = one[0];
                    one[0] = x;
                } else if(one[1] == 0 || one[1] >= x) {
                    one[1] = x;
                } 
            } else if(x % 3 == 2) {
                if(two[0] == 0 || two[0] >= x) {
                    two[1] = two[0];
                    two[0] = x;
                } else if(two[1] == 0 || two[1] >= x) {
                    two[1] = x;
                }                 
            }
        }
      //  System.out.println(Arrays.toString(one));
      //  System.out.println(Arrays.toString(two));
        if(cs % 3 == 0) return cs;
        else if(cs % 3 == 1) {
            if(one[0] != 0) {
                if(two[1] != 0) {
                    cs -= Math.min(one[0], two[0] + two[1]);
                    return cs;
                }
                return cs - one[0];
            } else if(two[1] != 0) {
                return cs - two[0] - two[1];
            } else return 0;
        } else {
            if(two[0] != 0) {
                if(one[1] != 0) {
                    cs -= Math.min(two[0], one[0] + one[1]);
                    return cs;
                }
                return cs - two[0];
            } else if(one[1] != 0) {
                return cs - one[0] - one[1];
            } else return 0;
        }
    }
}
