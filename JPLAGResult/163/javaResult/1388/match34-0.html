class Solution {
    public int maxSumDivThree(int[] nums) {
        int min21 = Integer.MAX_VALUE;
        int min22 = Integer.MAX_VALUE;
        int min11 = Integer.MAX_VALUE;
        int min12 = Integer.MAX_VALUE;
        int sum = 0;
        for (int num : nums) {
            sum += num;
            int residual = num % 3;
            if (residual == 1) {
                if (num < min11) {
                    min12 = min11;
                    min11 = num;
                } else if (num < min12) {
                    min12 = num;
                }
            }
            if (residual == 2) {
                if (num < min21) {
                    min22 = min21;
                    min21 = num;
                } else if (num < min22) {
                    min22 = num;
                }
            }
        }
        int residual = sum % 3;
        
        if (residual == 0) {
            return sum;
        }
        
        if (residual == 1) {
            if (min11 == Integer.MAX_VALUE) {
                if (min21 != Integer.MAX_VALUE && min22 != Integer.MAX_VALUE) {
                    return sum - min21 - min22;
                } else {
                    return 0;
                }
            } else {
                if (min21 != Integer.MAX_VALUE && min22 != Integer.MAX_VALUE) {
                    return Math.max(sum - min21 - min22, sum - min11);
                } else {
                    return sum - min11;
                }
            }
        }
        
        if (residual == 2) {
            if (min21 == Integer.MAX_VALUE) {
                if (min11 != Integer.MAX_VALUE && min12 != Integer.MAX_VALUE) {
                    return sum - min11 - min12;
                } else {
                    return 0;
                }
            } else {
                if (min11 != Integer.MAX_VALUE && min12 != Integer.MAX_VALUE) {
                    return Math.max(sum - min11 - min12, sum - min21);
                } else {
                    return sum - min21;
                }
            }
        }
        return 0;
    }
}
