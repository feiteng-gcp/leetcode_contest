class Solution {
    public int maxSumDivThree(int[] nums) {
        int sum = 0;
        
        
            
        Integer m1min1 = null;
        Integer m1min2 = null;
        Integer m2min1 = null;
        Integer m2min2 = null;
        
        for (int n : nums) {
            sum += n;
            if (n % 3 == 1) {
                if (m1min1 == null || n < m1min1) {
                    m1min2 = m1min1;
                    m1min1 = n;
                } else if (m1min2 == null || n < m1min2) {
                    m1min2 = n;
                }
            } else if (n % 3 == 2) {
                if (m2min1 == null || n < m2min1) {
                    m2min2 = m2min1;
                    m2min1 = n;
                } else if (m2min2 == null || n < m2min2) {
                    m2min2 = n;
                }
            }
        }
        if (sum % 3 == 0) return sum;
            
        
        if (sum % 3 == 2) {
            int max = 0;
            if (m1min1 != null && m1min2 != null) {
                max = Math.max(max, sum - m1min1 - m1min2);
            }
            if (m2min1 != null) {
                max = Math.max(max, sum - m2min1);
            }
            return max;
        } else {
            int max = 0;
            if (m2min1 != null && m2min2 != null) {
                max = Math.max(max, sum - m2min1 - m2min2);
            }
            if (m1min1 != null) {
                max = Math.max(max, sum - m1min1);
            }
            return max;
        }
        
    }
}
