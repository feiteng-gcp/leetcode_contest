class Solution {
    public int maxSumDivThree(int[] nums) {
        List<Integer> r1 = new ArrayList<>();
        List<Integer> r2 = new ArrayList<>();
        
        int total = 0;
        for (int i = 0; i < nums.length; ++i) {
            int r = nums[i] % 3;
            if (r == 1) {
                r1.add(nums[i]);
            } else if (r == 2) {
                r2.add(nums[i]);
            }
            total += nums[i];
        }
        
        Collections.sort(r1);
        Collections.sort(r2);
        
        int r = total % 3;
        if (r == 0) {
            return total;
        } else 
if (r == 1) {
            int sum1 = r1.size() > 0 ? r1.get(0) : Integer.MAX_VALUE;
            int sum2 = r2.size() > 1 ? r2.get(0) + r2.get(1) : Integer.MAX_VALUE;
            return total - Math.min(sum1, sum2);
        } else 
{
            int sum1 = r1.size() > 1 ? r1.get(0) + r1.get(1) : Integer.MAX_VALUE;
            int sum2 = r2.size() > 0 ? r2.get(0) : Integer.MAX_VALUE;
            return total - Math.min(sum1, sum2);
        }
    }
}
