class Solution {
    public int maxSumDivThree(int[] nums) {
        int ans[] = new int[3];
        for(int i = 0;i < nums.length;i++){
            int temp[] = new int[3];
            temp[0] = ans[0];
            temp[1] = ans[1];
            temp[2] = ans[2];
            for(int j = 0; j < 3;j++){
                int sum = nums[i] + ans[j];
                if(sum > temp[sum%3])
                    temp[sum%3] = sum;
            
}
            ans = temp;
            // System.out.println(ans[0] + " " + ans[1] + " " + ans[2]);
        }
        return ans[0];
    }
}
