class Solution {
    public int maxSumDivThree(int[] nums) {
        return sln1(nums);
    }
    
    
private int sln1(int[] nums){
        int n = nums.length;
        int[][] r = new int[n+1][3];
        for(int i = 1;i< n+1;i++){
            
for(int j =0;j<3;j++){
                r[i][j] = r[i-1][j];
            }
            for(int j =0;j<3;j++){
                int sum = r[i-1][j]+nums[i-1];
                r[i][sum%3] = Math.max(r[i][sum%3], sum);
                
            }
        }
        
        return r[n][0];
    }
}
