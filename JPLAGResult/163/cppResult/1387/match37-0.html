/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class FindElements {
    TreeNode* r;
public:
    void fix(TreeNode* n){
        if(!n) return;
        if(n->left) n->left->val = 2*n->val + 1;
        if(n->right) n->right->val = 2*n->val + 2;
        fix(n->left);
        fix(n->right);
    }
    
    FindElements(TreeNode* root) {
        root->val = 0;
        fix(root);
        r = root;
    }
    bool fn(TreeNode* n, int x){
        if(!n) return false;
        if(n->val == x) return true;
        return fn(n->left, x) || fn(n->right, x);
    }
    
    bool find(int target) {
        return fn(r, target);
    }
};

 * FindElements* obj = new FindElements(root);
 * bool param_1 = obj->find(target);
