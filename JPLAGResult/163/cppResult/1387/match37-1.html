/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class FindElements {
public:
    TreeNode* r;
    void checkNode(TreeNode* root) {
        if (root == nullptr) return;
        if (root -> left != nullptr) root -> left -> val = root -> val * 2 + 1;
        checkNode(root -> left);
        if (root -> right != nullptr) root -> right ->val = root -> val * 2 + 2; 
        checkNode(root -> right);
    }
    FindElements(TreeNode* root) {
        r = root;
        if (r != nullptr)

            r ->val = 0;
        checkNode(r);
    }
    
    bool myfind(int target, TreeNode* root) {
        if (root == nullptr) return false;
        if (root -> val == target) return true;
        return myfind(target, root -> left) || myfind(target, root -> right);
    }
    
    bool find(int target) {
        return myfind(target, r);
    }
};
