/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class FindElements {
public:
    FindElements(TreeNode* root) {
        if(!root)
            return;
        src = root;
        root->val = 0;
        recover(root);
        // print(src);
    }
    
    void recover(TreeNode* root) {
        if(!root)
            return;
        int x = root->val;
        if(root->left) {
            root->left->val = 2*x+1;
            recover(root->left);
        }
        if(root->right) {
            root->right->val = 2*x+2;
            recover(root->right);
        }
    }
    
    bool find(int target) {
        return helper(src, target);
    }
    
    void print(TreeNode*  root) {
        if(!root)
            return;
        cout << root->val << endl;
        print(root->left);
        print(root->right);
    }
    
    bool helper(TreeNode* root, int target) {
        if(!root)
            return false;
        if(root->val == target)
            return true;
        if(helper(root->left, target) || helper(root->right, target))
            return true;
        return false;
    }
    
private:
    TreeNode* src;
};

 * FindElements* obj = new FindElements(root);
 * bool param_1 = obj->find(target);
