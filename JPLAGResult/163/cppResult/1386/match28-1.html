class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int n = grid.size();
        int m = grid[0].size();
        for(int i = 0; i<k; i++) {
            vector<vector<int> > tmp(n, vector<int>(m,0));

            for(int j = 0; j < m-1; j++) {
                for(int t = 0; t<n; t++) {
                    tmp[t][j+1] = grid[t][j];
                }
                
                
            }
            for(int t = 0; t<n-1; t++) {
                tmp[t+1][0] = grid[t][m-1];
            }
            tmp[0][0] = grid[n-1][m-1];
            grid.swap(tmp);
        }
        return grid;
    }
};
