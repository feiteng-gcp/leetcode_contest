class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<int>> cur = grid, nxt = grid;
        for(int t = 0; t < k; t++) {
            for(int i = 0; i < n; i++){
                for(int j = 0; j < m; j++) {
                    if(i == 0 && j == 0) nxt[i][j] = cur[n-1][m-1];
                    else if(j == 0) nxt[i][j] = cur[i-1][m-1];
                    else nxt[i][j] = cur[i][j-1];
                }
            }
            cur = nxt;
        }
        return cur;
    }
};
