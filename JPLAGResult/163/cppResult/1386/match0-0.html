class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<int>> res(n, vector<int> (m));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                int id = (i*m + j + k) % (n*m);
                res[id/m][id%m] = grid[i][j];
            }
        }
        return res;
    }
};
