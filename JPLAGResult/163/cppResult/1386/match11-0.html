class Solution {
public:
	vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
		vector<vector<int>>curr=grid,ret=grid;
		int n=grid.size(),m=grid[0].size();
		for (int i = 0;i < k;i++) {
			for (int x = 0;x < n;x++) {
				for (int y = 0;y < m;y++) {
					if(x==0 && y==0)ret[x][y]=curr[n-1][m-1];
					else if(y==0)ret[x][y]=curr[x-1][m-1];
					else ret[x][y]=curr[x][y-1];
				}
			}
			curr=ret;
		}
		return ret;
	}
};
