class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<int>> result(n, vector<int>(m));
        for (int i = 0; i < n; ++i)
        {
            for (int j = 0; j < m; ++j)
            {
                int t = i * m + j;
                int t1 = (t + k) % (m * n);
                int i1 = t1 / m;
                int j1 = t1 % m;
                result[i1][j1] = grid[i][j];
            }
        }
        
        return result;
    }
};
