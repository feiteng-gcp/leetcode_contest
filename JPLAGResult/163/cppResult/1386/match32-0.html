class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) 
    {
        int H = grid.size();
        int W = grid[0].size();
        vector<vector<int>> res(H, vector<int>(W, 0));
        for (int i=0; i<H; ++i)
            for (int j=0; j<W; ++j)
            {
                int next = i*W+j+k;
                res[(next/W)%H][next%W] = grid[i][j];                
            }
        return res;
    }
};
