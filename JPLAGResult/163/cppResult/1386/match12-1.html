class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        vector<vector<int>> new_g = grid;
        int n = grid.size();
        int m = grid[0].size();
        for(int f=0; f<k; f++) {
            for(int i=0; i<n; i++) {
                for(int j=0; j<m-1; j++)
                    new_g[i][j+1] = grid[i][j];
            }
            for(int i=0; i<n-1; i++)
                new_g[i+1][0] = grid[i][m-1];
            new_g[0][0] = grid[n-1][m-1];
            swap(new_g, grid);
        }
        return grid;
    }
};
