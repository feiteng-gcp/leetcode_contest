class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& g, int K) {
        int n=g.size();
        int m=g[0].size();
        vector<int>v;
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
                v.push_back(g[i][j]);
        K%=(n*m);
        vector<vector<int>>G=g;
        
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
            {
                int k=(i*m+j-K+n*m)%(n*m);
                //cout<<k<<endl;
                G[i][j]=v[k];
            }
        return G;
        
    }
};
