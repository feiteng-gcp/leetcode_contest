class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        if (grid.empty() || grid[0].empty()) return grid;
        int R = grid.size();
        int C = grid[0].size();
        int N = R * C;
        k %= N;
        vector<vector<int> > res(R, vector<int>(C, 0));

        for (int i = 0; i < R; ++i) {
            for (int j = 0; j < C; ++j) {
                int s = i * C + j + k;
                int r = (s / C) % R;
                int c = (s % C) % C;
                res[r][c] = grid[i][j];
            }
        }
        return res;
    }
};
