class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<int>> result (n, vector<int>(m, 0));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                int flatten = (i * m + j + k) % (n*m);
                int x = flatten / m;
                int y = flatten % m;
                result[x][y] = grid[i][j];
            }
        }
        return result;
    }
};
