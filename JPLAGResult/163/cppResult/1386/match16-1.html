class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        for(int i = 0; i < k; ++i) {
            shift(grid);
        }
        return grid;
        
    }
    
    void shift(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<int> f (m);
        f[0] = grid[m-1][n-1];
        for(int i = 1; i < m; ++i) {
            f[i] = grid[i-1][n-1];
        }
        for(int i = 0; i < m; ++i) {
            for(int j = n-1; j >= 1; --j) {
                grid[i][j] = grid[i][j-1];
            }
        }
        for(int i = 0; i < m; ++i) {
            grid[i][0] = f[i];
        }
    }
};
