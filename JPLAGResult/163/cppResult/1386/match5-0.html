class Solution {
public:
    /*
    Element at grid[i][j] becomes at grid[i][j + 1].
Element at grid[i][m - 1] becomes at grid[i + 1][0].
Element at grid[n - 1][m - 1] becomes at grid[0][0].
    */
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int n = grid.size();
        int m = grid[0].size();
        for (int kk = 0; kk < k; kk++) {
            vector<vector<int>> nextGrid(n, vector<int>(m));
            for (int i = 0; i < grid.size(); i++) {
                for (int j = 0; j < grid[0].size(); j++) {
                    /*
                    int t1 = grid[i][j+1];
                    int t2 = grid[i+1][0];
                    int t3 = grid[0][0];
                    grid[i][j] = t1;
                    grid[i][m-1] = t2;
                    grid[n-1][m-1] = t3;
                    */
                    //swap(grid[])
                    //int 
                    /*
                    nextGrid[i][j] = grid[i][j+1];
                    nextGrid[i][m-1] = grid[i+1][0];
                    nextGrid[n-1][m-1] = grid[0][0];
                    */
                    if (i == 0 && j == 0) {
                        nextGrid[0][0] = grid[n-1][m-1];
                    } else if (j == 0) {
                        nextGrid[i][j] = grid[i-1][m-1];
                    } else {
                        nextGrid[i][j] = grid[i][j-1];
                    }
                }
            }
            grid = nextGrid;
        }
        return grid;
    }
};
