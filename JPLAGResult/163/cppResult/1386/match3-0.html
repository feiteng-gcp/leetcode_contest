class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int N=grid.size(), M=grid[0].size();
        vector<vector<int>> R(N, vector<int>(M, 0));
        for(int i=0;i<N;++i){
            for(int j=0;j<M;++j){
                int x=grid[i][j];
                int down=max(0,(k-((M-1)-j)+M-1)/M);
                R[(i+(down%N)+N)%N][(j+k)%M]=x;
            }
        }
        return R;
    }
};
