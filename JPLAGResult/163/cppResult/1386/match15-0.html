class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int m = grid.size(), n = grid[0].size();
        vector<vector<int>> ret(m, vector<int>(n));
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int x = (i + (j + k) / n) % m;
                int y = (j + k) % n;
                ret[x][y] = grid[i][j];
            }
        }
        return ret;
    }
};
