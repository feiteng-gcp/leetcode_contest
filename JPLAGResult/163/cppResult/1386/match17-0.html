class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        vector<vector<int>> G;
        G = grid;
        int r = grid.size();
        int c = grid[0].size();
        for (int i = 0; i < r; i++)
            for (int j = 0; j < c; j++)
            {
                int t = i * c + j;
                int f = (t + k) % (r * c);
                G[f / c][f % c] = grid[i][j];
            }
        // for (int i = 0; i < r; i++)
        //     for (int j = 0; j < c; j++)
        //             cout << G[i][j] << " \n"[j == c - 1];
        return G;
    }
};
