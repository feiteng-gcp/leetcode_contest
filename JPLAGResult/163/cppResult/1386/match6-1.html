class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int m = grid.size();
        int n = grid[0].size();
        vector<vector<int>> ret(m,vector<int>(n,0));
        for(int i = 0;i<m;i++){
            for(int j = 0;j<n;j++){
                int x = i * n + j;
                int x_ = (x + k)%(m*n);
                int i_ = x_ / n;
                int j_ = x_ % n;
                //cout<<"i= "<<i<<" j= "<<j<<" i_= "<<i_<<" j_= "<<j_<<endl;
                ret[i_][j_] = grid[i][j];
            }
        }
        return ret;
    }
};
