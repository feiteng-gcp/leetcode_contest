class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<int> > ans(n, vector<int>(m, 0));
        for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                //ans[i][j] = grid[i][j];
                int id = (i * m + j + k) % (n * m);
                int ii = id / m;
                int jj = id % m;
                ans[ii][jj] = grid[i][j];
            }
        }
        return ans;
    }
};
