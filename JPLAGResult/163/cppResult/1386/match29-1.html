class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        vector<int> all;
        int n = grid.size(),m = grid[0].size();
        for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                all.push_back(grid[i][j]);
            }
        }
        k = m * n - k % (n * m);
        if(k % (n * m) == 0)return grid;

        for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                grid[i][j] = all[k];
                k = (k + 1) % (m * n);
            }
        }
        return grid;
    }
};
