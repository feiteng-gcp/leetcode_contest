int n,m,T;
int way[4][2]={{0,1},{0,-1},{1,0},{-1,0}};
int str[20][20];
class Solution {
public:
    struct Node{
        int x,y,step;
        int mp[20][20];
        bool check(){
            if(x>=0&&x<n&&y>=0&&y<m)
                return true;
            return false;
        }
    }s,e,u,v,ss,ee,uu,vv;
    bool bfs_(Node n1){
        queue<Node> q;
        ss=n1;
        bool flag[20][20];
        memset(flag,false,sizeof(flag));
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++)            
                if(n1.mp[i][j]==4){
                    ss.x=i;
                    ss.y=j;
                    ss.step=0;
                }
        }
        if(ss.x==ee.x&&ss.y==ee.y)
            return true;
        q.push(ss);
        flag[ss.x][ss.y]=true;
        while(!q.empty()){
            uu=q.front();
            q.pop();
            for(int i=0;i<4;i++){
                vv=uu;
                vv.step++;
                vv.x+=way[i][0];
                vv.y+=way[i][1];
                if(vv.check()&&flag[vv.x][vv.y]==false&&(n1.mp[vv.x][vv.y]!=1&&n1.mp[vv.x][vv.y]!=2)){
                    flag[vv.x][vv.y]=true;
                    if(vv.x==ee.x&&vv.y==ee.y)
                        return true;
                    q.push(vv);
                }
            }
        }
        return false;
    }

    int bfs(){
        int flag[20][20][4];
        queue<Node>q;
        q.push(s);
        memset(flag,false,sizeof(flag));
        while(!q.empty()){
            u=q.front();
            q.pop();
            for(int i=0;i<4;i++){
                v=u;
                v.x+=way[i][0];
                v.y+=way[i][1];
                v.step++;
                if(v.check()&&str[v.x][v.y]!=1&&flag[v.x][v.y][i]==false){
                    ee.x=u.x-way[i][0];
                    ee.y=u.y-way[i][1];
                    if(ee.check()==false)
                        continue;
                    if(bfs_(v)){
                        swap(v.mp[v.x][v.y],v.mp[u.x][u.y]);
                        swap(v.mp[ee.x][ee.y],v.mp[ss.x][ss.y]);
                        flag[v.x][v.y][i]=true;
                        if(str[v.x][v.y]==3)
                            return v.step;
                        q.push(v);
                    }
                }
            }
        }
        return -1;
    }
    int minPushBox(vector<vector<char>>& grid) {
        n = grid.size();m = grid[0].size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == '.')
                    str[i][j] = s.mp[i][j] = 0;
                else if (grid[i][j] == '#')
                    str[i][j] = s.mp[i][j] = 1;
                else if (grid[i][j] == 'T')
                    str[i][j] = s.mp[i][j] = 3;
                else if (grid[i][j] == 'B')
                    str[i][j] = s.mp[i][j] = 2;
                else if (grid[i][j] == 'S')
                    str[i][j] = s.mp[i][j] = 4;
                if(grid[i][j]=='B'){
                    s.x=i;
                    s.y=j;
                    s.step=0;
                }
            }
        }
        return bfs();
    }
};
