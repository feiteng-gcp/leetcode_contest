class Solution {
public:
    int dx[4] = {-1, 0, 1, 0};
    int dy[4] = {0, -1, 0, 1};
    int n, m;
    struct Node {
        int x, y;
    };
    int vis[25][25], vis2[25][25][25][25];
    struct Node2 {
        int x, y, bx, by, dist;
    };
    
    void bfs(int sx, int sy, int bx, int by, vector<vector<char>>& grid) {
        queue<Node> q;
        memset(vis, 0, sizeof vis);
        vis[sx][sy] = 1;
        q.push((Node){sx, sy});
        while (q.size()) {
            Node cur = q.front(); q.pop();
            int cx = cur.x, cy = cur.y;
            for (int k = 0; k < 4; ++k) {
                int nx = cx + dx[k], ny = cy + dy[k];
                if (nx < 0 || ny < 0 || nx >= m || ny >= n) continue;
                if (vis[nx][ny]) continue;
                if (grid[ny][nx] == '#') continue;
                if (nx == bx && ny == by) continue;
                vis[nx][ny] = 1;
                q.push((Node){nx, ny});
            }
        }
    }
    
    int bfs2(int sx, int sy, int bx, int by, int tx, int ty, vector<vector<char>>& grid) {
        queue<Node2> q;
        q.push((Node2){sx, sy, bx, by, 0});
        memset(vis2, 0, sizeof vis2);
        vis2[sx][sy][bx][by] = 1;
        while (q.size()) {
            Node2 cur = q.front(); q.pop();
            bx = cur.bx; by = cur.by; 

            int cx = cur.x, cy = cur.y;
            //cout << cx << ' ' << cy << ' ' << bx << ' ' << by << ';';
            if (bx == tx && by == ty) {
                return cur.dist;
            }
            bfs(cx, cy, bx, by, grid);
            for (int k = 0; k < 4; ++k) {
                int nbx = bx + dx[k], nby = by + dy[k], px = bx - dx[k], py = by - dy[k];
                //cout << " (" << bx << ' ' << by << ' ' << nbx << ' ' << nby << ") ";
                if (nbx < 0 || nby < 0 || nbx >= m || nby >= n) continue;
                if (px < 0 || py < 0 || px >= m || py >= n) continue;
                if (grid[nby][nbx] == '#' || !vis[px][py]) continue;
                //cout << " > " << bx << ' ' << by << ' ' << nbx << ' ' << nby << ';';
                if (vis2[bx][by][nbx][nby]) continue;
                vis2[bx][by][nbx][nby] = 1;
                q.push((Node2){bx, by, nbx, nby, cur.dist + 1});
            }
        }
        return -1;
    }
    
    int minPushBox(vector<vector<char>>& grid) {
        n = grid.size();
        m = grid[0].size();

        int cx, cy, tx, ty, sx, sy;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == 'B') { cx = j; cy = i; }
                if (grid[i][j] == 'T') { tx = j; ty = i; }
                if (grid[i][j] == 'S') { sx = j; sy = i; }
            }
        }
        return bfs2(sx, sy, cx, cy, tx, ty, grid);
    }
};
