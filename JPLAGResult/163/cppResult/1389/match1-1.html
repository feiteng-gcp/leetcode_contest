class Solution {
public:

    const int mov[4][2] = {
        {0, 1}, {1, 0}, {0, -1}, {-1, 0}
    };
    
    struct data {
        int sx, sy;
        int bx, by;
        
        data() {}
        
        data(int a, int b, int c, int d) :
            sx(a), sy(b), bx(c), by(d) {}
    };
    
    int dis[25][25][25][25];
    
    bool vis[25][25];
    
    void visit(data now, vector<vector<char>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        memset(vis, 0, sizeof vis);
        queue< pair<int, int> > que;
        que.push(make_pair(now.sx, now.sy));
        vis[now.sx][now.sy] = true;

        while (!que.empty()) {
            auto p = que.front(); 
            que.pop();
            for (int i = 0; i < 4; ++i) {
                int x = p.first + mov[i][0];
                int y = p.second + mov[i][1];
                if (x < 0 || x >= n)continue;
                if (y < 0 || y >= m)continue;
                if (grid[x][y] == '#')continue;
                if (x == now.bx && y == now.by)continue;
                if (!vis[x][y]) {
                    vis[x][y] = true;
                    que.push(make_pair(x, y));
                }
            }
        }
    }
    
    int minPushBox(vector<vector<char>>& grid) {
        data init;
        int n = grid.size();
        int m = grid[0].size();

        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == 'S') {
                    init.sx = i;
                    init.sy = j;
                } else if (grid[i][j] == 'B') {
                    init.bx = i;
                    init.by = j;
                }
            }
        }
        memset(dis, 0x3f, sizeof dis);
        queue< data > que;
        que.push(init);
        dis[init.sx][init.sy][init.bx][init.by] = 0;
        while (!que.empty()) {
            auto d = que.front();
            que.pop();
            if (grid[d.bx][d.by] == 'T') {
                return dis[d.sx][d.sy][d.bx][d.by];
            }
            visit(d, grid);
            for (int i = 0; i < 4; ++i) {
                int x = d.bx + mov[i][0];
                int y = d.by + mov[i][1];
                if (x < 0 || x >= n)continue;
                if (y < 0 || y >= m)continue;
                if (grid[x][y] == '#')continue;
                int sx = d.bx - mov[i][0];
                int sy = d.by - mov[i][1];
                if (sx < 0 || sx >= n)continue;
                if (sy < 0 || sy >= m)continue;
                if (grid[sx][sy] == '#')continue;
                if (!vis[sx][sy])continue;
                if (dis[d.bx][d.by][x][y] > dis[d.sx][d.sy][d.bx][d.by] + 1) {
                    dis[d.bx][d.by][x][y] = dis[d.sx][d.sy][d.bx][d.by] + 1;
                    que.push(data(d.bx, d.by, x, y));
                }
            }
        }
        return -1;
    }
};
