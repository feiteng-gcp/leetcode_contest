struct Item {
    int br, bc;
    int sr, sc;
    int dist;
    
    Item(int a, int b, int c, int d, int e) {
        br = a;
        bc = b;
        sr = c;
        sc = d;
        dist = e;
    }
    
    void print() {
        cout << "br,bc,sr,sc,d: " << br << " " << bc << " " << sr << " " << sc << " " << dist << endl;
    }
};

class Solution {
private:
    int dist[20][20][20][20];
    int part[20][20][20][20];
    
    int chr[4] = {1, 0, -1, 0};
    int chc[4] = {0, 1, 0, -1};
    bool in(const vector<vector<char>>& G, int r, int c) {
        return !(r < 0 || r >= G.size() || c < 0 || c >= G[0].size() || G[r][c] == '#');
    }
    
    void help(const vector<vector<char>>& G, int boxr, int boxc, int r, int c, int g) {
        if (!in(G, r, c) || (r == boxr && c == boxc)) {
            return;
        }
        part[boxr][boxc][r][c] = g;
        for (int i=0; i<4; i++) {
            if (in(G, r+chr[i], c+chc[i]) && part[boxr][boxc][r+chr[i]][c+chc[i]] == -1) {
                help(G, boxr, boxc, r+chr[i], c+chc[i], g);
            }
        }
    }
    void floodfill(const vector<vector<char>>& G, int boxr, int boxc) {
        int group = 1;
        for (int i=0; i<G.size(); i++) {
            for (int j=0; j<G[i].size(); j++) {
                if (G[i][j] == '#' || (i == boxr && j == boxc)) {
                    part[boxr][boxc][i][j] = 0;
                }
                else if (part[boxr][boxc][i][j] == -1) {
                    help(G, boxr, boxc, i, j, group);
                    group++;
                }
            }
        }
    }
public:
    int minPushBox(vector<vector<char>>& grid) {
        for (int i=0; i<20; i++) {
            for (int j=0; j<20; j++) {
                for (int k=0; k<20; k++) {
                    for (int l=0; l<20; l++) {
                        dist[i][j][k][l] = 1e9;
                        part[i][j][k][l] = -1;
                    }
                }
            }
        }
        
        int br, bc, sr, sc, tr, tc;
        for (int i=0; i<grid.size(); i++) {
            for (int j=0; j<grid[i].size(); j++) {
                if (grid[i][j] == 'B') {
                    br = i;
                    bc = j;
                }
                else if (grid[i][j] == 'S') {
                    sr = i;
                    sc = j;
                }
                else if (grid[i][j] == 'T') {
                    tr = i;
                    tc = j;
                }
            }
        }
        
        queue<Item> q;
        dist[br][bc][sr][sc] = 0;
        q.push(Item(br, bc, sr, sc, 0));
        
        while (q.size() > 0) {
            Item cur = q.front();
            q.pop();
            
            if (cur.br == tr && cur.bc == tc) {
                return cur.dist;
            }
            
            if (part[cur.br][cur.bc][0][0] == -1) {
                floodfill(grid, cur.br, cur.bc);
            }
            for (int i=0; i<4; i++) {
                if (in(grid, cur.br + chr[i], cur.bc + chc[i]) && in(grid, cur.br - chr[i], cur.bc - chc[i]) && 
                    part[cur.br][cur.bc][cur.sr][cur.sc] == part[cur.br][cur.bc][cur.br - chr[i]][cur.bc - chc[i]] &&
                    dist[cur.br + chr[i]][cur.bc + chc[i]][cur.br][cur.bc] > cur.dist+1) {
                    dist[cur.br + chr[i]][cur.bc + chc[i]][cur.br][cur.bc] = cur.dist+1;
                        q.push(Item(cur.br + chr[i], cur.bc + chc[i], cur.br, cur.bc, cur.dist+1));
                }
            }
        }
        
        return -1;
    }
};
