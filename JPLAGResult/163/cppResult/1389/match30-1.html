class Solution {
    
    bool got[21][21][21][21] = {false};
    
public:
    int minPushBox(vector<vector<char>>& grid) {
        int n = grid.size(), m = grid[0].size();
        

        int tx, ty, sx, sy, bx, by;
        for (int i=0; i<n; i++) {
            for (int j=0; j<m; j++) {
                if (grid[i][j] == 'B') {
                    bx = i;
                    by = j;
                }
                else if (grid[i][j] == 'T') {
                    tx = i;
                    ty = j;
                }
                else if (grid[i][j] == 'S') {
                    sx = i;
                    sy = j;
                }
            }
        }
        

        int diff[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        
        bool visited[25][25] = {false};
        stack<pair<int, int>> find;
        vector<pair<int, int>> found;
        find.push({sx, sy});
        visited[sx][sy] = true;
        while (find.size() > 0) {
            auto p = find.top();
            find.pop();
            
            if (abs(p.first - bx) + abs(p.second - by) == 1) 
                found.push_back(p);
            

            for (int k = 0; k < 4; k++) {
                int x = p.first + diff[k][0];
                int y = p.second + diff[k][1];
                if (x < 0 || y < 0 || x >= n || y >= m) continue;
                if (visited[x][y]) continue;
                if (grid[x][y] == '#') continue;
                if (x == bx && y == by) continue;
                
                visited[x][y] = true;
                find.push({x, y});
            }
        }
        
        vector<pair<pair<int, int>, pair<int, int>>> current;
        vector<pair<pair<int, int>, pair<int, int>>> temp;
        for (auto p : found) {
            got[p.first][p.second][bx][by] = true;
            current.push_back({{p.first, p.second}, {bx, by}});
        }
        
        int step = 0;
        while (1) {
            temp.clear();
            for (auto p : current) {
                int sx = p.first.first;
                int sy = p.first.second;
                int bx = p.second.first;
                int by = p.second.second;
                
                if (p.second.first == tx && p.second.second == ty) return step;
                
                for (int k=0; k<4; k++) {
                    int x = sx + diff[k][0];
                    int y = sy + diff[k][1];
                    int nx = bx;
                    int ny = by;
                    if (x == bx && y == by) {
                        nx += diff[k][0];
                        ny += diff[k][1];
                        if (nx < 0 || ny < 0 || nx >= n || ny >= m) continue;
                        if (grid[nx][ny] == '#') continue;
                        
                        bool visited[25][25] = {false};
                        stack<pair<int, int>> find;
                        vector<pair<int, int>> found;
                        find.push({x, y});
                        visited[x][y] = true;
                        while (find.size() > 0) {
                            auto p = find.top();
                            find.pop();

                            if (abs(p.first - nx) + abs(p.second - ny) == 1)
                                found.push_back(p);

                            for (int k = 0; k < 4; k++) {
                                int x = p.first + diff[k][0];
                                int y = p.second + diff[k][1];
                                if (x < 0 || y < 0 || x >= n || y >= m) continue;
                                if (visited[x][y]) continue;
                                if (grid[x][y] == '#') continue;
                                if (x == nx && y == ny) continue;

                                visited[x][y] = true;
                                find.push({x, y});
                            }
                        }
                        
                        for (auto next : found) {
                            if (got[next.first][next.second][nx][ny]) continue;
                            
                            got[next.first][next.second][nx][ny] = true;
                            temp.push_back({{next.first, next.second}, {nx, ny}});
                        }
                    }
                    
                }
            }
            
            if (temp.size() == 0) break;
            swap(current, temp);
            step++;
        }
        
        return -1;
    }
};
