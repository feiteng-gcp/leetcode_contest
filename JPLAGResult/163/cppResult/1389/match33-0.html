class Solution {
struct Node {
    int px , py , bx , by , step , bstep ;
     friend bool operator<(Node A,Node B){
        return A.bstep > B.bstep ;
    }

} ;
    
public:
    
    bool visited[20][20][20][20] ;
    int dx[5] = {-1 , 1 , 0 , 0} ;
    int dy[5] = {0 , 0 , -1 , 1} ;
    
    
    int minPushBox(vector<vector<char>>& grid) {
        int n = grid.size() , m = grid[0].size() ;
        int p1 , p2 , b1 , b2 ;
        int finalX , finalY ;
        for (int i = 0 ; i < n ; ++i) {
            for (int j = 0 ; j < m ; ++j) {
                if (grid[i][j] == 'B') {
                    b1 = i ; b2 = j ;
                }
                if (grid[i][j] == 'S') {
                    p1 = i ; p2 = j ;
                }
                if (grid[i][j] == 'T') {
                    finalX = i ; finalY = j ;
                }
            }
        }
        Node st , ns;
        st.step = 0 ;
        st.px = p1 ; st.py = p2 ; st.bx = b1 ; st.by = b2 ; st.bstep = 0 ;
        visited[p1][p2][b1][b2] = true ;
        priority_queue <Node> q ;
        q.push(st) ;
        while (!q.empty()) {
            st = q.top() ; q.pop() ;
            if (st.bx == finalX && st.by == finalY) {
                return st.bstep ;
            }
            //箱子在人上面
            if (st.bx == st.px - 1 && st.by == st.py) {
                //cout << st.px << " " << st.py << " " << st.bx << " " << st.by << " " << st.step << endl ;
                if (st.bx > 0 && grid[st.bx - 1][st.by] != '#') {
                    ns.step = st.step + 1 ;
                    ns.px = st.px - 1 ; ns.py = st.py ; ns.bx = st.bx - 1 ; ns.by = st.by ; ns.bstep = st.bstep + 1;
                    if (!visited[ns.px][ns.py][ns.bx][ns.by]) {
                        visited[ns.px][ns.py][ns.bx][ns.by] = true ;
                        q.push(ns) ;
                    }
                }//箱子在人下面
                for (int k = 0 ; k < 4 ; ++k) {
                    if (k == 0) continue ;
                    ns.step = st.step + 1 ;
                    ns.px = st.px + dx[k] ; ns.py = st.py + dy[k] ; ns.bx = st.bx ; ns.by = st.by ; ns.bstep = st.bstep ;
                    if (ns.px >= 0 && ns.py >= 0 && ns.px < n && ns.py < m && grid[ns.px][ns.py] != '#') {
                        if (!visited[ns.px][ns.py][ns.bx][ns.by]) {
                            visited[ns.px][ns.py][ns.bx][ns.by] = true ;
                            q.push(ns) ;
                        }
                    }
                }
            } else if (st.bx == st.px + 1 && st.by == st.py) {
                //cout << st.px << " " << st.py << " " << st.bx << " " << st.by << " " << st.step << endl ;
                if (st.bx + 1 < n && grid[st.bx + 1][st.by] != '#') {
                    ns.step = st.step + 1 ;
                    ns.px = st.px + 1 ; ns.py = st.py ; ns.bx = st.bx + 1 ; ns.by = st.by ; ns.bstep = st.bstep + 1;
                    if (!visited[ns.px][ns.py][ns.bx][ns.by]) {
                        visited[ns.px][ns.py][ns.bx][ns.by] = true ;
                        q.push(ns) ;
                    }
                }//箱子在人左边
                for (int k = 0 ; k < 4 ; ++k) {
                    if (k == 1) continue ;
                    ns.step = st.step + 1 ;
                    ns.px = st.px + dx[k] ; ns.py = st.py + dy[k] ; ns.bx = st.bx ; ns.by = st.by ; ns.bstep = st.bstep ;
                    if (ns.px >= 0 && ns.py >= 0 && ns.px < n && ns.py < m && grid[ns.px][ns.py] != '#') {
                        if (!visited[ns.px][ns.py][ns.bx][ns.by]) {
                            visited[ns.px][ns.py][ns.bx][ns.by] = true ;
                            q.push(ns) ;
                        }
                    }
                }
                
            } else if (st.bx == st.px && st.by == st.py - 1) {
                //cout << st.px << " " << st.py << " " << st.bx << " " << st.by << " " << st.step << endl ;
                if (st.by > 0 && grid[st.bx][st.by - 1] != '#') {
                    ns.step = st.step + 1 ;
                    ns.px = st.px ; ns.py = st.py - 1 ; ns.bx = st.bx ; ns.by = st.by - 1 ; ns.bstep = st.bstep + 1;
                    if (!visited[ns.px][ns.py][ns.bx][ns.by]) {
                        visited[ns.px][ns.py][ns.bx][ns.by] = true ;
                        q.push(ns) ;
                    }
                }//箱子在人右边
                for (int k = 0 ; k < 4 ; ++k) {
                    if (k == 2) continue ;
                    ns.step = st.step + 1 ;
                    ns.px = st.px + dx[k] ; ns.py = st.py + dy[k] ; ns.bx = st.bx ; ns.by = st.by ; ns.bstep = st.bstep ;
                    if (ns.px >= 0 && ns.py >= 0 && ns.px < n && ns.py < m && grid[ns.px][ns.py] != '#') {
                        if (!visited[ns.px][ns.py][ns.bx][ns.by]) {
                            visited[ns.px][ns.py][ns.bx][ns.by] = true ;
                            q.push(ns) ;
                        }
                    }
                }
            } else if (st.bx == st.px && st.by == st.py + 1) {
                //cout << st.px << " " << st.py << " " << st.bx << " " << st.by << " " << st.step << endl ;
                if (st.by + 1 < m && grid[st.bx][st.by + 1] != '#') {
                    ns.step = st.step + 1 ;
                    ns.px = st.px ; ns.py = st.py + 1 ; ns.bx = st.bx ; ns.by = st.by + 1 ; ns.bstep = st.bstep + 1 ;
                    if (!visited[ns.px][ns.py][ns.bx][ns.by]) {
                        visited[ns.px][ns.py][ns.bx][ns.by] = true ;
                        q.push(ns) ;
                    }
                }
                for (int k = 0 ; k < 4 ; ++k) {
                    if (k == 3) continue ;
                    ns.step = st.step + 1 ;
                    ns.px = st.px + dx[k] ; ns.py = st.py + dy[k] ; ns.bx = st.bx ; ns.by = st.by ; ns.bstep = st.bstep ;
                    if (ns.px >= 0 && ns.py >= 0 && ns.px < n && ns.py < m && grid[ns.px][ns.py] != '#') {
                        if (!visited[ns.px][ns.py][ns.bx][ns.by]) {
                            visited[ns.px][ns.py][ns.bx][ns.by] = true ;
                            q.push(ns) ;
                        }
                    }
                }
                
            } else {
                for (int k = 0 ; k < 4 ; ++k) {
                    ns.step = st.step + 1 ;
                    ns.px = st.px + dx[k] ; ns.py = st.py + dy[k] ; ns.bx = st.bx ; ns.by = st.by ; ns.bstep = st.bstep ;
                    if (ns.px >= 0 && ns.py >= 0 && ns.px < n && ns.py < m && grid[ns.px][ns.py] != '#') {
                        if (!visited[ns.px][ns.py][ns.bx][ns.by]) {
                            visited[ns.px][ns.py][ns.bx][ns.by] = true ;
                            q.push(ns) ;
                        }
                    }
                }
            }
        }
        return -1 ;
    }
};
