class Solution {

    int pos[4][2] = {{0,1}, {1,0}, {0,-1}, {-1,0}};

public:
    int minPushBox(vector<vector<char>>& grid) {
        set<vector<int>> visited;
        queue<vector<int>> q;

        vector<int> init(4);
        for(int i=0; i<grid.size(); i++) {
            for(int j=0; j<grid[0].size(); j++) {
                if(grid[i][j] == 'B') {
                    init[0] = i;
                    init[1] = j;
                } else if(grid[i][j]=='S')
                {
                    init[2] = i;
                    init[3] = j;
                }
            }
        }
        q.push(init);
        visited.insert(init);
        int level = 0;
        while(!q.empty()) {
            level++;
            int size = q.size();
            for(int i=0; i<size; i++) {
                auto front = q.front();
                q.pop();
                for(int k=0; k<4; k++) {
                    int x = front[0]+pos[k][0];
                    int y = front[1]+pos[k][1];
                    if(x<0||y<0||x>=grid.size()||y>=grid[0].size()||grid[x][y]=='#')
                        continue;
                    vector<int> new_front = {x,y,front[0], front[1]};
                    if(visited.count(new_front))
                        continue;
                    if(find_connect(grid, front, front[0]-pos[k][0], front[1]-pos[k][1])) {
                        if(grid[x][y]=='T')
                            return level;
                        q.push(new_front);
                        visited.insert(new_front);
                    }
                }
            }
            
        }
        return -1;
    }
    
    bool find_connect(vector<vector<char>> & grid, vector<int> & cur, int tx, int ty) {
        if(tx<0||ty<0||tx>=grid.size()||ty>=grid[0].size()||grid[tx][ty]=='#')
            return false;
        if(cur[2]==tx&&cur[3]==ty)
            return true;
        queue<pair<int,int>> q;
        set<pair<int,int>> visited;
        q.push(make_pair(tx, ty));
        visited.insert(make_pair(tx,ty));
        while(!q.empty()) {
            auto front = q.front();
            q.pop();

            for(int k=0; k<4; k++) {
                int x = front.first+pos[k][0];
                int y = front.second+pos[k][1];
                if(x<0||y<0||x>=grid.size()||y>=grid[0].size()||grid[x][y]=='#'||(x==cur[0]&&y==cur[1]))
                    continue;
                if(visited.count(make_pair(x,y)))
                    continue;
                if(cur[2]==x&&cur[3]==y)
                    return true;
                visited.insert(make_pair(x,y));
                q.push(make_pair(x,y));
            }
        }
        return false;
    }
};
