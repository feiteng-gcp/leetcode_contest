class Solution {
public:
    int minPushBox(vector<vector<char>>& grid) {
        int N=grid.size();
        int M=grid[0].size();
        vector<vector<bool>> vis(M*N,vector<bool>(M*N,false));
        int pa,pb,ba,bb;
        for(int i=0;i<N;i++)
            for(int j=0;j<M;j++){
                if(grid[i][j]=='S'){
                    pa=i;
                    pb=j;
                }else if(grid[i][j]=='B'){
                    ba=i;
                    bb=j;
                }
            }
        int pt=pa*M+pb;
        int bt=ba*M+bb;
        queue<vector<int>> qq;
        qq.push(vector<int>({pa,pb,ba,bb,0}));
        vis[pt][bt]=true;
        int dd[][2]={{1,0},{-1,0},{0,1},{0,-1}};
        int ans=INT_MAX;
        while(!qq.empty()){
            auto p=qq.front();
            qq.pop();
            if(grid[p[2]][p[3]]=='T'){
                ans=min(ans,p[4]);
                continue;
            }
            if((p[0]==p[2]&&abs(p[1]-p[3])==1)||(p[1]==p[3]&&abs(p[0]-p[2])==1)){
                int aaa=2*p[2]-p[0];
                int bbb=2*p[3]-p[1];
                if(bbb>=0&&bbb<M&&aaa>=0&&aaa<N&&grid[aaa][bbb]!='#'){
                    int bbt=aaa*M+bbb;
                    int ppt=p[0]*M+p[1];
                    if(!vis[ppt][bbt]){
                        qq.push(vector<int>{p[0],p[1],aaa,bbb,p[4]+1});
                        vis[ppt][bbt]=true;
                    }
                }

            }
            
            for(int i=0;i<4;i++){
                int ppa=p[0]+dd[i][0];
                int ppb=p[1]+dd[i][1];
                if(ppa==p[2]&&ppb==p[3]) continue;
                if(ppa<0||ppa>=N||ppb<0||ppb>=M) continue;
                if(grid[ppa][ppb]=='#') continue;
                int ppt=ppa*M+ppb;
                int bbt=p[2]*M+p[3];
                if(vis[ppt][bbt]) continue;
                qq.push(vector<int>{ppa,ppb,p[2],p[3],p[4]});
                vis[ppt][bbt]=true;
            }
        }
        return ans==INT_MAX?-1:ans;
    }
};
