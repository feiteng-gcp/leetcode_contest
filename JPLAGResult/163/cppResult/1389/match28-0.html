const int dx[4] = {-1, 0, 1, 0};
const int dy[4] = {0, -1, 0, 1};

struct Node {
    int pushes;
    int bx, by, sx, sy;
};

int encode(int bx, int by, int sx, int sy) {
    return bx * 8000 + by * 400 + sx * 20 + sy;
}

class Solution {
public:
    int minPushBox(vector<vector<char>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        int sx, sy, tx, ty, bx, by;
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == 'B') {
                    bx = j;
                    by = i;
                }
                if (grid[i][j] == 'T') {
                    tx = j;
                    ty = i;
                }
                if (grid[i][j] == 'S') {
                    sx = j;
                    sy = i;
                }
            }
        queue<Node> bfs;
        bfs.push(Node{0, bx, by, sx, sy});
        unordered_map<int, bool> visited;
        visited[encode(bx, by, sx, sy)] = true;
        while (!bfs.empty()) {
            Node f = bfs.front();
            // cout << "Step = " << f.pushes << ", " << f.by << " " << f.bx << ", " << f.sy << " " << f.sx << endl; 
            bfs.pop();
            if (f.bx == tx && f.by == ty)
                return f.pushes;
            for (int i = 0; i < 4; ++i) {
                int nx = f.bx + dx[i];
                int ny = f.by + dy[i];
                if (nx < 0 || nx >= m || ny < 0 || ny >= n) continue;
                if (grid[ny][nx] == '#') continue;
                int px = f.bx - dx[i];
                int py = f.by - dy[i];
                if (px < 0 || px >= m || py < 0 || py >= n) continue;
                if (grid[py][px] == '#') continue;
                int hash = encode(nx, ny, f.bx, f.by);
                if (visited[hash]) continue;
                
                unordered_map<int, bool> used;
                queue<pair<int, int>> q;
                q.push(make_pair(f.sx, f.sy));
                used[f.sx * 20 + f.sy] = true;
                bool can_reach = false;
                while (!q.empty()) {
                    auto ff = q.front();
                    q.pop();
                    if (ff.first == px && ff.second == py) {
                        can_reach = true;
                        break;
                    }
                    for (int j = 0; j < 4; ++j) {
                        int nnx = ff.first + dx[j];
                        int nny = ff.second + dy[j];
                        if (nnx < 0 || nnx >= m || nny < 0 || nny >= n) continue;
                        if (grid[nny][nnx] == '#') continue;
                        if (nnx == f.bx && nny == f.by) continue;
                        if (used[nnx * 20 + nny]) continue;
                        q.push(make_pair(nnx, nny));
                        used[nnx * 20 + nny] = true;
                    }
                }
                if (!can_reach) continue;
                visited[hash] = true;
                bfs.push(Node{f.pushes + 1, nx, ny, f.bx, f.by});
            }
        }
        return -1;
    }
};
