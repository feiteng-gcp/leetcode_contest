class Solution {
public:

    const int dir[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    int res = INT_MAX, tarx, tary, n, m, sx, sy, bx, by, tot;
    int getIdx(int x, int y) {
        return x * m + y;
    }
    
    int minPushBox(vector<vector<char>>& g) {
        n = g.size();
        m = g[0].size();
        tot = n * m;
        

        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j)
                if (g[i][j] == 'S') {
                    sx = i;
                    sy = j;
                } else if (g[i][j] == 'T') {
                    tarx = i;
                    tary = j;
                } else if (g[i][j] == 'B') {
                    bx = i;
                    by = j;
                }
        
        vector<vector<int>> f(tot, vector<int>(tot, -1));
        
        queue<vector<int>> qu;
        int a = getIdx(sx, sy), b = getIdx(bx, by);
        int tarb = getIdx(tarx, tary);
        
        f[a][b] = 0;
        qu.push({a, b, 0});
        
        while (!qu.empty()) {
            int a = qu.front()[0], b = qu.front()[1], dist = qu.front()[2];
            qu.pop();
            if (b == tarb)
                res = min(res, dist);
            
            if (dist != f[a][b])
                continue;
            
            sx = a / m;
            sy = a % m;
            bx = b / m;
            by = b % m;
            // printf("%d %d %d %d: %d\n", sx, sy, bx, by, dist);
            
            for (int i = 0; i < 4; ++i) {
                int tx = sx + dir[i][0], ty = sy + dir[i][1];
                if (tx < 0 || tx >= n || ty < 0 || ty >= m || g[tx][ty] == '#')
                    continue;
                
                int c = getIdx(tx, ty);
                if (tx == bx && ty == by) {
                    // push
                    int tx2 = bx + dir[i][0], ty2 = by + dir[i][1];
                    if (tx2 < 0 || tx2 >= n || ty2 < 0 || ty2 >= m || g[tx2][ty2] == '#')
                        continue;
                    
                    int d = getIdx(tx2, ty2);
                    if (f[c][d] == -1 || f[c][d] > dist + 1) {
                        f[c][d] = dist + 1;
                        qu.push({c, d, dist + 1});
                    }
                } else {
                    if (f[c][b] == -1 || f[c][b] > dist) {
                        f[c][b] = dist;
                        qu.push({c, b, dist});
                    }
                }
            }
        }
        
        return res == INT_MAX ? -1 : res;
    }
};
