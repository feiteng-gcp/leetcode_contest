#include <vector>
#include <queue>
using namespace std;

char grid[32][32];

int step[22][22][22][22];

struct State {
    int bx, by, px, py;
};


class Solution {
public:
    int minPushBox(vector<vector<char>>& g) {

        int N = g.size();
        int M = g[0].size();

        memset(grid, '#', sizeof(grid));
        memset(step, 0, sizeof(step));
        deque<State> que;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                grid[i + 1][j + 1] = g[i][j];
            }
        }
        N += 2;
        M += 2;

        State init;
        int gx, gy;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (grid[i][j] == 'S') {
                    init.px = i;
                    init.py = j;
                } else if (grid[i][j] == 'B') {
                    init.bx = i;
                    init.by = j;
                } else if (grid[i][j] == 'T') {
                    gx = i;
                    gy = j;
                }
            }
        }

        auto s = [&](State state) -> int& {
            return step[state.bx][state.by][state.px][state.py];
        };
        
        s(init) = 1;
        que.push_back(init);

        int ans = 0;
        while (que.size()) {
            State p = que.back();
            que.pop_back();

            if (p.bx == gx && p.by == gy) {
                ans = s(p);
                break;
            }

            for (int d = 0; d < 4; d++) {
                int dx = "1120"[d] - '1';
                int dy = "2011"[d] - '1';

                int x = p.px + dx;
                int y = p.py + dy;

                if (grid[x][y] == '#') {
                    continue;
                } else if (x == p.bx && y == p.by) {
                    if (grid[x + dx][y + dy] == '#')
                        continue;
                    State next = {x + dx, y + dy, x, y};
                    if (!s(next)) {
                        s(next) = s(p) + 1;
                        que.push_front(next);
                    }
                } else {
                    State next = {p.bx, p.by, x, y};
                    if (!s(next)) {
                        s(next) = s(p);
                        que.push_back(next);
                    }
                }
            }
        }






        return ans - 1;
        
    }
};
