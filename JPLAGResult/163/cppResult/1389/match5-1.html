class Solution {
public:
    
    vector<pair<int,int>> dirs = {{1,0}, {-1,0}, {0, 1}, {0,-1}};
    int visited[21][21][21][21] = {};
    int minPushBox(vector<vector<char>>& grid) {
        int x, y;
        int sx, sy;
        int m = grid.size(), n = grid[0].size();
        // 
        // set<vector<int>> seen;
        
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                if(grid[i][j] == 'B') {
                    x = i;
                    y = j;
                }
                if(grid[i][j] == 'S') {
                    sx = i;
                    sy = j;
                }
            }
        }
        queue<vector<int>> q;
        q.push({x, y, sx, sy});
        visited[x][y][sx][sy] = 1;
        int steps = 0;
        int xx, yy;
        while(q.size()) {
            for(int k = q.size(); k > 0; --k) {
                x = q.front()[0];
                y = q.front()[1];
                int si = q.front()[2];
                int sj = q.front()[3];
                q.pop();
                
                if(grid[x][y] == 'T')
                    return steps;
                for(auto& [dx, dy] : dirs) {
                    xx = x + dx;
                    yy = y + dy;
                    sx = x - dx;
                    sy = y - dy;
                    // printf("inorder to move box to %d %d, player need to move from %d %d to %d %d\n", xx, yy, si, sj, sx, sy);
                    if(xx >= 0 && xx < m && yy >=0 && yy < n && sx >= 0 && sx < m && sy >= 0 && sy < n 
                       && 
                       // !visited[xx][yy] 
                       !visited[x][y][sx][sy]
                       && grid[xx][yy] != '#'&& grid[sx][sy] != '#' && 
                       canWalk(grid, si, sj, sx, sy, x, y)) {
                        q.push({xx, yy, x, y});
                        // printf("get new state %d %d %d %d\n", xx, yy, x, y);
                        // visited[xx][yy] = true;
                        visited[x][y][sx][sy] = 1;
                    }
                }
            }
            ++steps;
            
        }
        return -1;
        
    }
    
    
    bool canWalk(vector<vector<char>>& grid, int si, int sj, int sx, int sy, int bx, int by) {
        int m = grid.size(), n = grid[0].size();
        if(si <  0 && sj < 0 || si >= m || sj >= n)
            return false;
        vector<vector<bool>> visited(m, vector<bool>(n, false));
        queue<pair<int,int>> q;
        q.push({si, sj});
        visited[si][sj] = true;
        while(q.size()) {
            int x = q.front().first;
            int y = q.front().second;
            q.pop();
            if(x == sx && y == sy)
                return true;
            for(auto& [dx, dy] : dirs) {
                int xx = x + dx;
                int yy = y + dy;
                if(xx >=0 && xx < m && yy >= 0 && yy < n && !visited[xx][yy] && !(xx == bx && yy == by) && grid[xx][yy] != '#') {
                    q.push({xx, yy});
                    visited[xx][yy] = true;
                }
            }
        }
        return false;

    }
    
    
};
