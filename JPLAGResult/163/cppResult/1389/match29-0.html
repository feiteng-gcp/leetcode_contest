class Solution {
public:
    int minPushBox(vector<vector<char>>& g) {
        vector < vector <vector <vector <int > > > > v(20, vector <vector <vector <int > > >(20, vector< vector <int> >(20, vector<int>(20, 1e6))));
       
        int I = g.size(); 
        int J = g[0].size();
        int si, sj, bi, bj, ti, tj;
        for (int i = 0; i < g.size(); i++) {
            for (int j = 0; j < g[i].size(); j++) {
                if (g[i][j] == 'S') {
                    si = i; 
                    sj = j; 
                }
                if (g[i][j] == 'B') {
                    bi = i; 
                    bj = j; 
                }
                if (g[i][j] == 'T') {
                    ti = i; 
                    tj = j; 
                }
            }
        }
        
        queue <vector <int> > q;
        vector <int> st(4, 0), st_new(4, 0);
        st[0] = si; st[1] = sj; st[2] = bi; st[3] = bj;
        q.push(st);
        v[si][sj][bi][bj] = 0;
        vector <vector <int> > dir = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        
        int min_steps = 1e6;
        
        while (q.size() > 0) {
            st = q.front();
            q.pop();
            
            si = st[0]; sj = st[1]; bi = st[2]; bj = st[3];            
            int steps = v[si][sj][bi][bj];
            if (steps >= min_steps) continue;
            
            //  cout << si << " " << sj << " " << bi << " " << bj << endl;
            
            for (vector <int> d : dir) {
                if (si + d[0] >= 0 && si + d[0] < I && sj + d[1] >= 0 && sj + d[1] < J) {
                    if (g[si + d[0]][sj + d[1]] == '#') {
                      //   cout << d[0] << " " << d[1] << " walkign into wall" << endl;
                        continue; // Wall
                    }
                    else if (si + d[0] == bi && sj + d[1] == bj && si + 2*d[0] >= 0 && si + 2*d[0] < I && sj +2* d[1] >= 0 && sj + 2*d[1] < J) {
                        // We're pushing the box
                        if (g[si + 2*d[0]][sj + 2*d[1]] == '#') {
                        //     cout << d[0] << " " << d[1] << " box into wall" << endl;
                            continue; // Pushing box into wall
                        }
                        else if (g[si + 2*d[0]][sj + 2*d[1]] == 'T') {
                        //     cout << d[0] << " " << d[1] << " found solo" << endl;
                            min_steps = min(min_steps, steps+1);
                            continue;
                        }
                        else {
                            if (v[si+d[0]][sj+d[1]][bi+d[0]][bj+d[1]] <= steps + 1) {
                        //         cout << d[0] << " " << d[1] << " box into wall better solu" << endl;
                                continue;
                            }
                            
                        //     cout << d[0] << " " << d[1] << " logging box push" << endl;
                            v[si+d[0]][sj+d[1]][bi+d[0]][bj+d[1]] = steps + 1;
                            st[0] = si+d[0]; st[1] = sj+d[1]; st[2] = bi+d[0]; st[3] = bj+d[1];
                            q.push(st);
                            continue;
                        }
                        
                    }
                    else {
                        if (v[si+d[0]][sj+d[1]][bi][bj] <= steps) {
                        //     cout << d[0] << " " << d[1] << " walking, better solu" << endl;
                            continue;
                        }
                        
                        // cout << d[0] << " " << d[1] << " logging walk" << endl;
                        v[si+d[0]][sj+d[1]][bi][bj] = steps;
                        st[0] = si+d[0]; st[1] = sj+d[1]; st[2] = bi; st[3] = bj;
                        q.push(st);
                        continue;
                    } // End check on move                        
                } // End check on bounds
            } // End loop on dir
        } // End loop on stack
        
        if (min_steps == 1e6) min_steps = -1;
        return min_steps;
    }
};
