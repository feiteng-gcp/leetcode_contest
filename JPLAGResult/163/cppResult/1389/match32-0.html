class Solution {
public:
    int dx[4]={1, -1, 0, 0};
    int dy[4]={0, 0, 1, -1};
    int minPushBox(vector<vector<char>>& grid) {
        unordered_set<int> used;
        int s, b, t;
        for (int i=0; i<grid.size(); i++) {
            for (int j=0; j<grid[0].size(); j++) {
                if (grid[i][j]=='S') {
                    s=i*20+j;
                    grid[i][j]='.';
                }
                if (grid[i][j]=='B') {
                    b=i*20+j;   
                    grid[i][j]='.';
                }
                if (grid[i][j]=='T') {
                    t=i*20+j;   
                    grid[i][j]='.';
                }
            }
        }
        int ret=0;
        vector<int> st;
        st.push_back(s*400+b);
        while(!st.empty()) {
            vector<int> st_next;
            while(!st.empty()) {
                int curr=st.back();
                st.pop_back();
                if (used.find(curr)==used.end()) {
                    used.insert(curr);
                    int ss=curr/400;
                    int bb=curr%400;
                    int sx=ss/20;
                    int sy=ss%20;
                    if (bb==t) {
                        return ret;
                    }
                    for (int k=0; k<4; k++) {
                        int x=sx+dx[k];
                        int y=sy+dy[k];
                        if (x>=0 && x<grid.size() && y>=0 && y<grid[0].size() && grid[x][y]!='#') {
                            if(x*20+y!=bb) {
                                st.push_back((x*20+y)*400+bb);
                            } else {
                                int bx=bb/20;
                                int by=bb%20;
                                bx+=dx[k];
                                by+=dy[k];
                                if (bx>=0 && bx<grid.size() && by>=0 && by<grid[0].size() && grid[bx][by]!='#') {
                                    st_next.push_back((x*20+y)*400+(bx*20+by));
                                }
                            }
                        }
                    }
                }
            }
            st=st_next;
            ret++;
        }
        return -1;
    }
};
