class Solution {
public:
    const int dirx[4] = {-1, 0, 1, 0};
    const int diry[4] = {0, -1, 0, 1};
    map<pair<pair<int,int>, int>, int> triedMoves;
    vector<vector<char>> grid;
    int Tx, Ty;
    int Bx, By;
    int Sx, Sy;
    int n, m;
    bool visited[20][20];
    void floodFill(int ax, int ay, int bx, int by) {
        queue<pair<int,int>> q;
        memset(visited, 0, sizeof(visited));
        q.push(make_pair(ax, ay));
        visited[ax][ay] = true;
        // printf("floodFill %d %d %d %d\n", ax, ay, bx, by);
        while (!q.empty()) {
            int cx = q.front().first;
            int cy = q.front().second;
            q.pop();
            for (int i = 0; i < 4; ++ i) {
                int nx = cx + dirx[i];
                int ny = cy + diry[i];
                if (validPos(nx, ny) && !visited[nx][ny] && grid[nx][ny] != '#' && !(bx == nx && by == ny)) {
                    // printf("%d %d is reachable\n", nx, ny);
                    visited[nx][ny] = true;
                    q.push(make_pair(nx, ny));
                }
            }
        }
    }
    inline bool validPos(int x, int y) {
        return x >= 0 && x < n && y >= 0 && y < m;
    }
    int minPushBox(vector<vector<char>>& grid) {
        this->grid = grid;
        n = grid.size();
        m = grid[0].size();
        for (int i = 0; i < n; ++ i) {
            for (int j = 0; j < m; ++ j) {
                if (grid[i][j] == 'T') {
                    Tx = i;
                    Ty = j;
                } else if (grid[i][j] == 'B') {
                    Bx = i;
                    By = j;
                } else if (grid[i][j] == 'S') {
                    Sx = i;
                    Sy = j;
                }
            }
        }
        queue<pair<pair<int,int>, int>> q;
        auto findNextSteps = [&](int bx, int by, int sx, int sy, int stepNumber) -> void {
            floodFill(sx, sy, bx, by);
            for (int i = 0; i < 4; ++ i) {
                int px = bx - dirx[i];
                int py = by - diry[i];
                int nx = bx + dirx[i];
                int ny = by + diry[i];
                if (validPos(px, py) && validPos(nx, ny) && visited[px][py] && grid[nx][ny] != '#') {
                    pair<pair<int,int>, int> key = make_pair(make_pair(nx, ny), i);
                    // printf("(%d, %d) from %d: %d\n", nx, ny, i, stepNumber);
                    if (triedMoves.find(key) != triedMoves.end()) continue;
                    q.push(key);
                    triedMoves[key] = stepNumber;
                }
            }
        };
        findNextSteps(Bx, By, Sx, Sy, 1);
        while (!q.empty()) {
            auto key = q.front();
            int bx = key.first.first;
            int by = key.first.second;
            int dir = key.second;
            q.pop();
            int steps = triedMoves[key];
            if (bx == Tx && by == Ty) {
                // printf("Done: %d\n", steps);
                return steps;
            }
            int sx = bx - dirx[dir];
            int sy = by - diry[dir];
            findNextSteps(bx, by, sx, sy, steps + 1);
        }
        return -1;
    }
};
