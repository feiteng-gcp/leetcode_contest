ll po(ll a, ll b, ll mod)
{
    ll res = 1;
    a %= mod;
    for (; b; b >>= 1)
    {
        if (b & 1)
            res = res * a % mod;
        a = a * a % mod;
    }
    return res;
}

{

    {

    {

void extgcd(ll a, ll b, ll &d, ll &x, ll &y)
{
    if (!b)
    {
        d = a;
        x = 1;
    }
    else
    {
        extgcd(b, a % b, d, y, x);
        y -= x * (a / b);
    }
}

{
    ll d, x, y;
    extgcd(a, n, d, x, y);
    return d == 1 ? (x + n) % n : -1;

class Solution {
public:
    std::vector<int> edge[5];
    int maxSumDivThree(vector<int>& nums) {
        REP(i,0,4) {
            edge[i].clear();
        } 
        int n = nums.size();
        REP(i,0,nums.size()) {
            edge[nums[i]%3].push_back(nums[i]);
        }
        REP(i,0,4) {
            sort(edge[i].begin(), edge[i].end());
            // reverse(edge[i].begin(), edge[i].end());
        } 
        int sum = 0;
        REP(i,0,n) {
            sum += nums[i];
        }
        if(sum % 3 == 0) {
            return sum;
        }
        if(sum % 3 == 1) {
            int t = 1e5;
            if(edge[1].size() >= 1) {
                t = min(t, edge[1][0]);
            }
            if(edge[2].size() >= 2) {
                t = min(t, edge[2][0] + edge[2][1]);
            }
            return sum - t;
        }
        if(sum % 3 == 2) {
            int t = 1e5;
            if(edge[2].size() >= 1) {
                t = min(t, edge[2][0]);
            }
            if(edge[1].size() >= 2) {
                t = min(t, edge[1][0] + edge[1][1]);
            }
            return sum - t;
        }

        return sum;

    }
