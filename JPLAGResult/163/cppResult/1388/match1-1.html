class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int n = nums.size();
        int dp[n+1][3] = {};
        for(int j = 1; j < 3; ++j)
            dp[0][j] = INT_MIN;
        for(int i = 1; i <= n; ++i) {
            for(int j = 0; j < 3; ++j) {
                int x = (j  - nums[i-1]%3 + 3) % 3;
                cout << x << endl;
                dp[i][j] = max(dp[i-1][x]+nums[i-1], dp[i-1][j]);
            }
        }
        int res = INT_MIN;
        for(int i = 0; i <= n; ++i)
            res = max(res, dp[i][0]);
        return res;
        
    }
};
