class Solution {
public:
    //unordered_map<int, int> memo;
    int maxSumDivThree(vector<int>& nums) {
        int n = nums.size();
        vector<int> mp1;
        vector<int> mp2;
        for(auto num : nums){
            if(num%3 == 1){
                mp1.emplace_back(num);
            }else if(num%3 == 2){
                mp2.emplace_back(num);
            }
        }
        long sum = accumulate(nums.begin(), nums.end(), 0L);
        if(sum % 3 == 0) return sum;
        //0, 1, 2
        sort(mp1.begin(), mp1.end());
        sort(mp2.begin(), mp2.end());
        if(sum % 3 == 1){
           int diff = -1;
            if(!mp1.empty()){
                diff = mp1[0];
            }
            if(mp2.size() > 1){
                diff = min(diff, mp2[0] + mp2[1]);
            }
            if(diff == -1) return -1;
            return sum - diff;
        }
        if(sum % 3 == 2){
           int diff = -1;
            if(!mp2.empty()){
                diff = mp2[0];
            }
            if(mp1.size() > 1){
                diff = min(diff, mp1[0] + mp1[1]);
            }
            if(diff == -1) return -1;
            return sum - diff;
        }
        return -1;
    }
};
