class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        vector<int> d1;
        vector<int> d2;
        int sum = 0;
        for (auto p : nums) {
            sum += p;
            if(p % 3 == 1) {
                d1.push_back(p);
            }
            else if(p % 3 == 2) {
                d2.push_back(p);
            }
        }
        sort(d1.begin(), d1.end());
        sort(d2.begin(), d2.end());
        if(sum % 3 == 1) {
            int xx = 1e7;
            if(d1.size() >= 1) {
                xx = min(xx, d1[0]);
            }
            if(d2.size() >= 2) {
                xx = min(xx, d2[0] + d2[1]);
            }
            return sum - xx;
        }
        else if(sum % 3 == 2) {
            int xx = 1e7;
            if(d1.size() >= 2) {
                xx = min(xx, d1[0] + d1[1]);
            }
            if(d2.size() >= 1) {
                xx = min(xx, d2[0]);
            }
            return sum - xx;
        }
        else return sum;
        
    }
};
