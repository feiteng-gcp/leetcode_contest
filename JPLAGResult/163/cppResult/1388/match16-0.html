class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int n = nums.size();
        vector<vector<int> > dp(n + 1, vector<int>(3, -1));
        dp[0][0] = 0;
        for(int i = 1;i <= n;i++){
            for(int j = 0;j < 3;j++){
                int to = (j + nums[i - 1]) % 3;
                if(dp[i - 1][to] != -1)
                    dp[i][to] = dp[i - 1][to];
                if(dp[i - 1][j] != -1)
                    dp[i][to] = max(dp[i][to], dp[i - 1][j] + nums[i - 1]);
            }
        }
        if(dp[n][0] == -1)
            return 0;
        else
            return dp[n][0];
    }
};
