class Solution {
public:
    int maxSumDivThree(vector<int>& nums)
    {
        int sum = 0;
        int min_r1 = INT_MAX, min_r1_prev = INT_MAX;
        int min_r2 = INT_MAX, min_r2_prev = INT_MAX;
        for (int num : nums) {

            sum += num;
            if (num % 3 == 1) {
                if (num < min_r1) {
                    min_r1_prev = min_r1;
                    min_r1 = num;
                }
                else if (num < min_r1_prev) {
                    min_r1_prev = num;
                }
            }
            if (num % 3 == 2) {
                if (num < min_r2) {
                    min_r2_prev = min_r2;
                    min_r2 = num;
                }
                else if (num < min_r2_prev) {
                    min_r2_prev = num;
                }
            }
        }
        if (sum % 3 == 1) {
            int ans1 = min_r1 != INT_MAX ? sum - min_r1 : 0;
            int ans2 = min_r2 != INT_MAX && min_r2_prev != INT_MAX ? sum - min_r2 - min_r2_prev : 0;
            return std::max(ans1, ans2);
        }
        if (sum % 3 == 2) {
            int ans1 = min_r2 != INT_MAX ? sum - min_r2 : 0;
            int ans2 = min_r1 != INT_MAX && min_r1_prev != INT_MAX ? sum - min_r1 - min_r1_prev : 0;
            return std::max(ans1, ans2);
        }
        return sum;
    }
};
