class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        vector<int> one, two;
        int ans = 0;
        for (const int &i : nums) {
            ans += i;
            if (i % 3 == 1) one.emplace_back(i);
            if (i % 3 == 2) two.emplace_back(i);
        }
        sort(one.begin(), one.end());
        sort(two.begin(), two.end());
        int mod = ans % 3;
        if (mod == 0)
            return ans;

        if (mod == 1) {
            int delta = INT_MAX;
            if (one.size() >= 1)
                delta = min(delta, one[0]);
            if (two.size() >= 2)
                delta = min(delta, two[0] + two[1]);
            return ans - delta;
        }
                      
        int delta = INT_MAX;
        if (one.size() >= 2)
            delta = min(delta, one[0] + one[1]);
        if (two.size() >= 1)
            delta = min(delta, two[0]);
        return ans - delta;
    }
};
