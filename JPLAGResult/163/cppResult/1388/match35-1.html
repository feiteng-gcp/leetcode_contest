class Solution {
public:
    int dp[2][3];
    
    int maxSumDivThree(vector<int>& nums) {
        dp[0][0] = 0;
        dp[0][1] = -0x3f3f3f3f;
        dp[0][2] = dp[0][1];
        
        for(int i = 0; i < nums.size(); ++ i)
        {
            for(int j = 0; j < 3; ++ j)
            {
                dp[(i + 1) % 2][(j + nums[i] % 3) % 3] = max(dp[i % 2][(j + nums[i] % 3) % 3], dp[i % 2][j] + nums[i]);
            }
        }
        
        return dp[nums.size() % 2][0];
    }
};
