class Solution {
public:
    int f[40010][3];
    int maxSumDivThree(vector<int>& nums) {
        memset(f,0,sizeof(f));
        f[0][nums[0] % 3] = nums[0];
        for (int i = 1; i < nums.size(); i++)
        {            
            for (int j = 0; j<3; j++)
            {
                f[i][j] = f[i-1][j];                
            }
            for (int j = 0; j<3; j++)
            {
                if (f[i-1][j] >0)
                    f[i][(j + nums[i]) % 3] = max(f[i][(j + nums[i]) % 3], f[i-1][j] + nums[i]);
                else
                    f[i][nums[i] % 3] = max(f[i][nums[i] % 3], nums[i]);
            }            
        }
        for (int i = 0; i<nums.size(); i++)
            cout<<f[i][0]<<' '<<f[i][1]<<' '<<f[i][2]<<endl;
        int res = f[nums.size()-1][0];
        return res;
    }
};
