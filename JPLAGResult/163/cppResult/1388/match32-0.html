class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int N = nums.size();
        vector<vector<int>> sum(N, vector<int>(3, -1));
        sum[0][nums[0] % 3] = nums[0];

        for(int i = 1; i < N; ++i) {
            for(int j = 0; j < 3; ++j) {
                sum[i][j] = sum[i - 1][j];
            }

            if(0 == (nums[i] % 3)) {
                sum[i][0] = max(sum[i][0], nums[i] + ((-1 == sum[i - 1][0]) ? 0 : sum[i - 1][0]));                    
    
                if(-1 != sum[i - 1][1]) {
                    sum[i][1] = max(sum[i][1], nums[i] + sum[i - 1][1]);                                        
                }

                if(-1 != sum[i - 1][2]) {
                    sum[i][2] = max(sum[i][2], nums[i] + sum[i - 1][2]);                                        
                }
            }
            else if(1 == (nums[i] % 3)) {
                if(-1 != sum[i - 1][2]) {
                    sum[i][0] = max(sum[i][0], nums[i] + sum[i - 1][2]);                                        
                }

                sum[i][1] = max(sum[i][1], nums[i] + ((-1 == sum[i - 1][0]) ? 0 : sum[i - 1][0]));                    
    
                if(-1 != sum[i - 1][1]) {
                    sum[i][2] = max(sum[i][2], nums[i] + sum[i - 1][1]);                                        
                }
            }
            else {
                if(-1 != sum[i - 1][1]) {
                    sum[i][0] = max(sum[i][0], nums[i] + sum[i - 1][1]);                                        
                }

                if(-1 != sum[i - 1][2]) {
                    sum[i][1] = max(sum[i][1], nums[i] + sum[i - 1][2]);                                        
                }

                sum[i][2] = max(sum[i][2], nums[i] + ((-1 == sum[i - 1][0]) ? 0 : sum[i - 1][0]));                    
    
            }
        }

        return (-1 == sum[N - 1][0]) ? 0 : sum[N - 1][0];
    }
};
