class Solution {
public:
    int sum[4];
    int maxSumDivThree(vector<int>& nums) {
        sum[0] = 0;
        sum[1] = sum[2] = -1e9;
        for(int i=0;i<nums.size();++i) {
            int s[4] = {0};
            for(int i=0;i<3;++i) s[i]=sum[i];
            for(int j=0;j<3;++j) 
                s[(nums[i]+j)%3] = max(s[(nums[i]+j)%3], sum[j] + nums[i]);
            for(int i=0;i<3;++i) sum[i]=s[i];
        }
        return max(0,sum[0]);
    }
};
