class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        vector<int> rem[3];
        int sum = 0;
        for (int n : nums) {
            sum += n;
            rem[n % 3].push_back(n);
        }
        // printf("sum = %d\n", sum);
        if (sum % 3 == 0) return sum;
        int ans = 0;
        for (int i = 0; i <= 2 && i <= rem[1].size(); ++ i) {
            for (int j = 0; j <= 2 && j <= rem[2].size(); ++ j) {
                int minus = 0;
                for (int k = 0; k < i; ++ k) minus += rem[1][k];
                for (int k = 0; k < j; ++ k) minus += rem[2][k];
                if ((sum - minus) % 3 == 0)
                    ans = max(ans, sum - minus);
            }
        }
        return ans;
    }
};
