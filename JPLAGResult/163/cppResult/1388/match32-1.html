const int INF = 1000000000;


class Solution {
public:
    int maxSumDivThree(vector<int>& a) {
        int n = a.size();
        vector<vector<int>> f(n+1, vector<int>(3, -INF));
        f[0][0] = 0;
        for (int i = 1; i <= n; ++ i)
        {
            for (int j = 0; j < 3; ++ j)
                f[i][(j+a[i-1])%3] = max(f[i-1][(j+a[i-1])%3], f[i-1][j]+a[i-1]);
        }
        return f[n][0];
    }
};
