class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        if(nums.size() == 1){
            if(nums[0] % 3 == 0) return nums[0];
            return 0;
        }
        long sum = 0;
        int rem1[2] ={INT_MAX, INT_MAX};
        int rem2[2] = {INT_MAX, INT_MAX};
        for(int num: nums){
            sum += num;
            if(num % 3 == 1){
                if(num < rem1[0]){
                    rem1[1] = min(rem1[0], rem1[1]);
                    rem1[0] = num;
                }
                else if(num < rem1[1]){
                    rem1[1] = num;
                }
            }
            else if(num % 3 == 2){
                if(num < rem2[0]){
                    rem2[1] = min(rem2[0], rem2[1]);
                    rem2[0] = num;
                }
                else if(num < rem2[1]){
                    rem2[1] = num;
                }
            }
        }
        
        if(sum % 3 == 0) return sum;
        
        if(sum % 3 == 1){
            sum = max(sum - rem1[0], sum - rem2[0] - rem2[1]);
        }
        if(sum % 3 == 2){
            sum = max(sum - rem2[0], sum - rem1[0] - rem1[1]);
        }
        
        if(sum < 0) return 0;
        return (int) sum;
    }
};
