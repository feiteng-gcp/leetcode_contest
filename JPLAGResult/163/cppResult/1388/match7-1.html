class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        vector<int> zero;
        vector<int> one;
        vector<int> two;
        for (auto num : nums) {
            if (num % 3 == 0) zero.push_back(num);
            if (num % 3 == 1) one.push_back(num);
            if (num % 3 == 2) two.push_back(num);
        }
        sort(zero.begin(), zero.end());
        sort(one.begin(), one.end());
        sort(two.begin(), two.end());
        
        int sum = 0;
        for (auto num : zero) sum += num;
        int part1 = 0;
        int part2 = 0;
        for (auto num : one) part1 += num;
        for (auto num : two) part2 += num;

        int p = part1 + part2;
        if (p % 3 == 0) return sum + part1 + part2;
        if (p % 3 == 1) {
            int best = sum;
            if (one.size() > 0) {
                best = sum + p - one[0];
            }
            if (two.size() > 1) {
                best = max(best, sum + p - two[0] - two[1]);
            }
            return best;
        }
        if (p % 3 == 2) {
            int best = sum;
            if (one.size() > 1) {
                best = sum + p - one[0] - one[1];
            }
            if (two.size() > 0) {
                best = max(best, sum + p - two[0]);
            }
            return best;
        }
        return sum;
    }
};
