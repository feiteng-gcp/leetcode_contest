class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {

        vector<int> ones, twos;
        for (auto& n : nums) {
            if (n % 3 == 1) {
                ones.emplace_back(n);
            } else if (n % 3 == 2) {
                twos.emplace_back(n);
            }
        }
        
        sort(ones.begin(), ones.end());
        sort(twos.begin(), twos.end());
        
        int total = accumulate(nums.begin(), nums.end(), 0);
        if (total % 3 == 0) {
            return total;

        } else if (total % 3 == 1) {
            int res = INT_MIN;
            if (ones.size() >= 1) {
                res = max(res, total - ones[0]);
            }
            if (twos.size() >= 2) {
                res = max(res, total - twos[0] - twos[1]);
            }
            if (res == INT_MIN) return 0;
            return res;
        } else {

            int res = INT_MIN;
            if (ones.size() >= 2) {
                res = max(res, total - ones[0] - ones[1]);
            }
            if (twos.size() >= 1) {
                res = max(res, total - twos[0]);
            }
            if (res == INT_MIN) return 0;
            return res;
        }
    }
};
