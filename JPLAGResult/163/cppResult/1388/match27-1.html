class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int f[3], g[3];
        f[0] = 0;
        f[1] = -2147483647;
        f[2] = -2147483647;
        for (int i = 0; i < nums.size(); i++)
        {
            int x = nums[i];

            for (int j = 0; j < 3; j++)
            {
                int k = (j + x) % 3;
                g[k] = f[j] + x;
            }
            for (int j = 0; j < 3; j++) f[j] = max(f[j], g[j]);
        }
        return f[0];
    }
};
