class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int mod0sum = 0, mod1sum = 0, mod2sum = 0;
        //int mod1min = 100000, mod2min = 100000;
        int totalsum = 0;
        vector<int> nums1, nums2;
        
        for(int i = 0; i < nums.size(); i++) {
            if(nums[i]%3 == 0) {
                mod0sum += nums[i];
            }
            else if(nums[i]%3 == 1) {
                mod1sum += nums[i];
                //mod1min = min(mod1min, nums[i]);
                nums1.push_back(nums[i]);
            }
            else {
                mod2sum += nums[i];
                //mod2min = min(mod2min, nums[i]);
                nums2.push_back(nums[i]);
            }
            totalsum += nums[i];
        }
        
        if(nums1.size() > 0)
            sort(nums1.begin(), nums1.end());
        if(nums2.size() > 0)
            sort(nums2.begin(), nums2.end());
        
        if(totalsum%3 == 0) {
            return totalsum;
        }
        else if(totalsum%3 == 1) {
            int sum = 0;
            if(nums1.size() > 0) {
                sum = max(sum, totalsum - nums1[0]);
            }
            if(nums2.size() > 1) {
                sum = max(sum, totalsum - nums2[0] - nums2[1]);
            }
            return sum;
        }
        else {
            int sum = 0;
            if(nums2.size() > 0) {
                sum = max(sum, totalsum - nums2[0]);
            }
            if(nums1.size() > 1) {
                sum = max(sum, totalsum - nums1[0] - nums1[1]);
            }
            return sum;
        }
    }
};
