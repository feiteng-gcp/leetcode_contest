class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int n = nums.size();
        int dp[n][3];
        int result = 0;
        
        for (int i = 0; i < n; i++)
            for (int j = 0; j < 3; j++)
                dp[i][j] = 0;
        
        if (nums[0] % 3 == 0)
            dp[0][0] = nums[0];
        if (nums[0] % 3 == 1)
            dp[0][1] = nums[0];
        if (nums[0] % 3 == 2)
            dp[0][2] = nums[0];
                
        for (int i = 1; i < n; i++) {
            int val = nums[i];
            
            for (int j = 0; j < 3; j++) {
                int prev = dp[i-1][j];
                int target = prev + val;
                
                dp[i][j] = max(dp[i][j], dp[i-1][j]);
                
                if (target % 3 == 0)
                    dp[i][0] = max(dp[i][0], target);
                if (target % 3 == 1)
                    dp[i][1] = max(dp[i][1], target);
                if (target % 3 == 2)
                    dp[i][2] = max(dp[i][2], target);
                result = max(result, dp[i][0]);
            }
        }
        
        return result;
    }
};
