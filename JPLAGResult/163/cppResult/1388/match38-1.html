class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        if (nums.empty()) return 0;
        int N = nums.size();
        vector<int> dp(3, 0);
        dp[nums[0] % 3] = nums[0];
        for (int i = 1; i < N; ++i) {
            vector<int> dp1(dp);
            int r = nums[i] % 3;
            dp1[r] = max(dp1[r], nums[i]);

            for (int j = 0; j < 3; ++j) {
                int t = (r + j) % 3;
                if (dp[j] > 0) {
                    dp1[t] = max(dp1[t], dp[j] + nums[i]);
                }
            }
            swap(dp, dp1);
        }
        return dp[0];
    }
};
