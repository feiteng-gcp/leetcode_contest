class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        vector<int> s1,s2;
        int sum =0,sum1= 0;
        for (int i = 0; i < nums.size(); i++) {
            if (nums[i]%3 == 0) {
                sum += nums[i];
            } else if (nums[i]%3 == 1) {
                s1.push_back(nums[i]);
            } else {
                s2.push_back(nums[i]);
            }
        }
        sort(s1.begin(),s1.end(),[](int a,int b){return a>b;});
        sort(s2.begin(),s2.end(),[](int a,int b){return a>b;});
        int i = 0,j=0,ii=0,jj=0;
        int ans1 = 0,ans2 = 0,ans3 = 0,ans4 = 0;
        while (i < s1.size()) {
            sum1 += s1[i++];
            if (sum1 %3 == 0) {
                ans1 += sum1;
                ii = i;
                sum1 = 0;
            }
        }
        sum1 = 0;
        while (j < s2.size()) {
            sum1 += s2[j++];
            if (sum1 %3 == 0) {
                ans1 += sum1;
                jj=j;
                sum1 = 0;
            }
        }
        for (; ii < s1.size() && jj < s2.size(); jj++,ii++) {
            ans1 += s1[ii] + s2[jj];
        }
        
        ii=0,jj=0;
        for (; ii < s1.size() && jj < s2.size(); jj++,ii++) {
            ans2 += s1[ii] + s2[jj];
        }
        i = ii,j = jj;
        sum1 = 0;
        while (i < s1.size()) {
            sum1 += s1[i++];
            if (sum1 %3 == 0) {
                ans2 += sum1;
                ii = i;
                sum1 = 0;
            }
        }
        sum1 = 0;
        while (j < s2.size()) {
            sum1 += s2[j++];
            if (sum1 %3 == 0) {
                ans2 += sum1;
                jj=j;
                sum1 = 0;
            }
        }

        ii=0,jj=0;i=0,j=0;
        sum1 = 0;
        while (i < s1.size()) {
            sum1 += s1[i++];
            if (sum1 %3 == 0) {
                ans3 += sum1;
                ii = i;
                sum1 = 0;
            }
        }
        for (; ii < s1.size() && jj < s2.size(); jj++,ii++) {
            ans3 += s1[ii] + s2[jj];
        }
        i = ii,j = jj;
        sum1 = 0;

        while (j < s2.size()) {
            sum1 += s2[j++];
            if (sum1 %3 == 0) {
                ans3 += sum1;
                jj=j;
                sum1 = 0;
            }
        }
        
        ii=0,jj=0;i=0,j=0;
        sum1 = 0;
        while (j < s2.size()) {
            sum1 += s2[j++];
            if (sum1 %3 == 0) {
                ans4 += sum1;
                jj=j;
                sum1 = 0;
            }
        }

        for (; ii < s1.size() && jj < s2.size(); jj++,ii++) {
            ans4 += s1[ii] + s2[jj];
        }
        i = ii,j = jj;
        sum1 = 0;
        while (i < s1.size()) {
            sum1 += s1[i++];
            if (sum1 %3 == 0) {
                ans4 += sum1;
                ii = i;
                sum1 = 0;
            }
        }

        return sum+max(ans1,max(ans2,max(ans3,ans4)));
    }
};
