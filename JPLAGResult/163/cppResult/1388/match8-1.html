class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int output = 0;
        vector<vector<int>> v(3, vector<int>());
        // for (int i = 0; i < 3; ++i) modl.push_back({});
        // cout << "gg";
        for (auto x: nums) {
            output += x;
            v[x % 3].push_back(x);
        }
        
        int remainder = output % 3;
        if (remainder == 0) return output;
        
        sort(v[1].begin(), v[1].end());
        sort(v[2].begin(), v[2].end());
        

        if (remainder == 1) {
            int tmp_output = 0;
            if (v[1].size() > 0) {
                tmp_output = output - v[1][0];
            } 
            if (v[2].size() > 1) {
                tmp_output = max(tmp_output, output - v[2][0] - v[2][1]);
            }
            return tmp_output;

        }
        
        if (remainder == 2) {
            int tmp_output = 0;
            if (v[2].size() > 0) {
                tmp_output = output - v[2][0];
            } 
            if (v[1].size() > 1) {
                tmp_output = max(tmp_output, output - v[1][0] - v[1][1]);
            }
            return tmp_output;
        }
        return 0;
        
    }
};
