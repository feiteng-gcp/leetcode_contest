class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        vector<int> memo(3, -1);
        memo[0] = 0;
        for (int i = 0; i < nums.size(); i++){
            vector<int> temp = memo;
            for (int j = 0; j < 3; j++){   
                if (temp[j] >= 0){
                    int x = (temp[j] + nums[i]) % 3;
                    memo[x] = max(memo[x], temp[j] + nums[i]);
                }
            }
        }
        return memo[0];
    }
};
