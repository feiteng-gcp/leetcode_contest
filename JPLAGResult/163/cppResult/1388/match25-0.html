class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int sum = accumulate(nums.begin(), nums.end(), 0);
        if (sum % 3 == 0)
            return sum;

        sort(nums.begin(), nums.end());
        vector<vector<int>> sums(3);
        for (int i : nums)
            sums[i % 3].push_back(i);

        if (sum % 3 == 1) {
            int mn = INT_MAX;
            if (sums[1].size() >= 1)
                mn = min(mn, sums[1][0]);
            if (sums[2].size() >= 2)
                mn = min(mn, sums[2][0] + sums[2][1]);
            return sum - mn;
        }

        int mn = INT_MAX;
        if (sums[1].size() >= 2)
            mn = min(mn, sums[1][0] + sums[1][1]);
        if (sums[2].size() >= 1)
            mn = min(mn, sums[2][0]);
        return sum - mn;
    }
};
