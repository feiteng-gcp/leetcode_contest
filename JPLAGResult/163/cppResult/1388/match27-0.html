class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        long pre[3]={0,0,0},cur[3]={0,0,0};
        for(auto n:nums){
            for(int i=0;i<3;i++){
                int j=(n+cur[i])%3;
                pre[j]=max(pre[j],n+cur[i]);
            }
            for(int i=0;i<3;i++)
                cur[i]=pre[i];
        }
        return pre[0];
    }
};
