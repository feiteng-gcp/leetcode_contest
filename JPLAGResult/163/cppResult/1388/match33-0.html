class Solution {
public:
    int maxSumDivThree(vector<int>& nums) 
    {
        int N = nums.size();
        vector<vector<int>> dp(N+1, vector<int>(3, -1)); 
        dp[0][0] = 0;
        for (int i=0; i<N; ++i)
            for (int j=0; j<3; ++j)
            if (dp[i][j] >= 0)
            {
                int now = dp[i][j] + nums[i];
                dp[i+1][now%3] = max(dp[i+1][now%3], now);
                dp[i+1][j] = max(dp[i+1][j], dp[i][j]); 
            }
        return dp[N][0];
    }
};
