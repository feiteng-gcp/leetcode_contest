class Solution {
public:
    //
    int maxSumDivThree(vector<int>& nums) {
        
        vector<int> mod;
        vector<int> st1;
        vector<int> st2;
        
        sort(nums.begin(),nums.end());
        
        int sum = 0;
        
        for(auto k:nums){
            mod.emplace_back(k%3);
            sum += k;
            if(k%3==1)
                st1.emplace_back(k);
            if(k%3==2)
                st2.emplace_back(k);
        }
        
        if(sum%3==0)
            return sum;
        
        if(sum%3==1){
            int sum1 = 1e6;
            if(st1.size()>=1){
                sum1 = min(sum1,st1[0]);
            }
            if(st2.size()>=2){
                sum1 = min(sum1,st2[0]+st2[1]);
            }
            
            if(sum1==1e6)
                return 0;
            else
                return sum-sum1;
        }
        
        else
        
        {
            int sum2 = 1e6;
            
            if(st1.size()>=2){
                sum2 = min(sum2,st1[0]+st1[1]);
            }
            
            //cout<<sum2<<endl;
            //cout<<st1[0]<<" "<<st1[1]<<endl;
            
            if(st2.size()>=1){
                sum2 = min(sum2,st2[0]);
                //cout<<st2[0]<<endl;
            }
            
            //cout<<sum2<<endl;
            //out<<st2[0]<<endl;
            
            if(sum2==1e6)
                return 0;
            else
                return sum-sum2;
        }
        return 0;
    }
};
