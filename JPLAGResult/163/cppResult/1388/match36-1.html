class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        vector<int> a[3];
        int sum = 0;
        for (int x : nums) {
            a[x % 3].push_back(x);
            sum += x;
        }
        vector<int> b;
        for (int i = 0; i < 3; ++i) {
            sort(a[i].begin(), a[i].end());
            for (int j = 0; j < 2; ++j) {
                if (j < a[i].size())
                    b.push_back(a[i][j]);
            }
        }
        if (sum % 3 == 0) return sum;
        sort(b.begin(), b.end());
        int ans = 0;
        for (int x : b) if ((sum - x) % 3 == 0) ans = max(ans, sum - x);
        for (int i = 0; i < b.size(); ++i) {

            for (int j = 0; j < i; ++j) {
                int t = sum - b[i] - b[j];
                if (t % 3 == 0) {
                    ans = max(ans, t);
                }
            }
        }
        return ans;
    }
};
