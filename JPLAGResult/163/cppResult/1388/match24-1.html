class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int mod21 = 10000;
        int mod22 = 10000;
        int mod11 = 10000;
        int mod12 = 10000;
        int sum = 0;

        for(auto n : nums){
            sum += n;
            if(n % 3 == 1){
                if(n <= mod11){
                    mod12 = mod11;
                    mod11 = n;
                }
                else if(n < mod12){
                    mod12 = n;
                }
            }
            else if(n % 3 == 2){
                if(n <= mod21){
                    mod22 = mod21;
                    mod21 = n;
                }
                else if(n < mod22){
                    mod22 = n;
                }
            }
        }
        if(sum % 3 == 0)
            return sum;
        if(sum % 3 == 1){
            return max(sum - mod11, sum - mod21 - mod22);
        }
        return max(sum - mod21, sum - mod11 - mod12);
    }
};
