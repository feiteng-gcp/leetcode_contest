class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        vector<int> dp(3, -1);
        sort(nums.begin(), nums.end());
        int ret = 0;
        for(int i = 0; i < nums.size(); i++) {
            int r = nums[i] % 3;
            if(dp[(3 - r) % 3] != -1) {
                ret = max(ret, dp[(3 - r) % 3] + nums[i]);
            }
            vector<int> next = dp;
            for(int j = 0; j < 3; j++) {
                if(dp[j] != -1) {
                    int r = (dp[j] + nums[i]) % 3;
                    next[r] = max(next[r], dp[j] + nums[i]);
                }
                next[nums[i] % 3] = max(next[nums[i] % 3], nums[i]);
            }
            dp = next;
            // for(int num: dp)
            //     cout << num << " " ;
            // cout << endl;
        }
        return ret;
    }
};
