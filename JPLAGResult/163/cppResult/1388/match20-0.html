class Solution {
public:
    int vis[10005];
    int maxSumDivThree(vector<int>& nums) {
        int sum = 0;
        int n = nums.size();
        vector<int> a;
        vector<int> b;
        int la = 0, lb = 0;
        sort(nums.begin(), nums.end());
        for(int i = 0; i < n; i++){
            if(nums[i] % 3 == 1){
                a.push_back(nums[i]);
                la ++;
            }
            if(nums[i] % 3 == 2){
                b.push_back(nums[i]);
                lb ++;
            }
            sum += nums[i];
        }
        if(sum % 3 == 0)
            return sum;
        if(sum % 3 == 1){
            int sum2 = 10000000;
            if(la >= 1)
                sum2 = min(sum2, a[0]);
            if(lb >= 2)
                sum2 = min(sum2, b[1] + b[0]);
            return sum - sum2;
        }
        if(sum % 3 == 2){
            int sum2 = 10000000;
            if(lb >= 1)
                sum2 = min(sum2, b[0]);
            if(la >= 2)
                sum2 = min(sum2, a[1] + a[0]);
            return sum - sum2;
        }
        return 0;
    }
};
