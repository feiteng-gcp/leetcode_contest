class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        //rem = 1, 2, 3
        vector<int> dp(3, 0);
        for (const auto& num : nums) {
            vector<int> next(dp);
            for (int i = 0; i < 3; ++i) {
                int cur = dp[i] + num;
                if (cur > next[cur % 3]) {
                    next[cur % 3] = cur;
                }
            }
            dp.swap(next);
        }
        return dp[0];
    }
};
