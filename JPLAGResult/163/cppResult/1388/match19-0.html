class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        vector<int> rone;
        vector<int> rtwo;
        
        int tot = 0;
        for(auto v : nums) {
            if(v % 3 == 1) rone.push_back(v);
            if(v % 3 == 2) rtwo.push_back(v);
            
            tot += v;
        }
        
        if(tot % 3 == 0) return tot;
        if(tot % 3 == 1) {
            int res = 0;
            
            if(rone.size() >= 1)
                res = max(res, tot - rone[0]);
            
            if(rtwo.size() >= 2) 
                res = max(res, tot - rtwo[0] - rtwo[1]);
            
            return res;
        }
        if(tot % 3 == 2) {
            int res = 0;
            
            if(rone.size() >= 2)
                res = max(res, tot - rone[0] - rone[1]);
            
            if(rtwo.size() >= 1) 
                res = max(res, tot - rtwo[0]);
            
            return res;
        }
        
        return 0;
    }
};
