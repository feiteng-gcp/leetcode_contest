class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int sum = 0;
        sort(nums.begin(), nums.end());
        vector<int> mod1, mod2;
        
        for (auto n: nums)
        {
            sum += n;
            if (n % 3 == 1)
            {
                mod1.push_back(n);
            }

            else if (n % 3 == 2)
            {
                mod2.push_back(n);
            }

        }
        
        if (sum % 3 == 0)
            return sum;
        
        if (sum % 3 == 1)
        {
            int diff = sum;
            if (mod1.size())
                diff = min(diff, mod1[0]);
            if (mod2.size() > 1)
                diff = min(diff, mod2[0] + mod2[1]);
            return sum - diff;
        }
        
        if (sum % 3 == 2)
        {
            int diff = sum;
            if (mod2.size())
                diff = min(diff, mod2[0]);
            if (mod1.size() > 2)
                diff = min(diff, mod1[0] + mod1[1]);
            return sum - diff;
        }
        
        return 0;
    }
};
