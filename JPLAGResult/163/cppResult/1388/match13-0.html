class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        vector<vector<int>> dp(3, vector<int>(nums.size()));
        dp[nums[0]%3][0] = nums[0];
        for(int i=1; i<nums.size(); i++) {
            for(int k=0; k<3; k++) {
                dp[(nums[i]+dp[k][i-1])%3][i] = max(max(dp[(nums[i]+dp[k][i-1])%3][i-1], dp[k][i-1] + nums[i]), dp[(nums[i]+dp[k][i-1])%3][i]);
                dp[k][i] = max(dp[k][i-1], dp[k][i]);
            }
        }
        return dp[0].back();
    }
};
