class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        vector<vector<int>> v(2);
        int res = 0;
        for (auto &i : nums) {
            if (i % 3 != 0) v[i % 3 - 1].push_back(i); 
            res += i;
        }
        for (int i = 0; i < 2; ++i) sort(v[i].begin(), v[i].end());
        int n = v[0].size() % 3, m = v[1].size() % 3;
        if (n == m) return res;
        else if (n < m){
            swap(n, m);
            swap(v[0], v[1]);
        }
           
        if (n == m+1) {
            if (v[1].size() > 1) {
                res -= min(v[0][0], v[1][0]+v[1][1]);
            }
            else res -= v[0][0];
        }
        else {
            if (v[1].size() > 1) {
                res -= min(v[0][0]+v[0][1], v[1][0]);
            }
            else res -= v[0][0]+v[0][1];
        }
        
        return res;
    }
};
