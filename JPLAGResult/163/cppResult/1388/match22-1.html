class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        int sum = 0;
        int n = nums.size();
        sort(nums.begin(), nums.end());
        vector<int> re1, re2;
        for (int i = 0; i < n; ++i) {
            sum += nums[i];
            if (nums[i] % 3 == 1 && re1.size() < 2) {
                re1.push_back(nums[i]);
            } else if (nums[i] % 3 == 2 && re2.size() < 2) {
                re2.push_back(nums[i]);
            }
        }

        if (sum % 3 == 0) {
            return sum;
        } else if (sum % 3 == 1) {
            int ans = 0;
            if (re1.size() > 0) {
                ans = max(ans, sum - re1[0]);
            } 
            if (re2.size() > 1) {
                ans = max(ans, sum - re2[0] - re2[1]);
            }
            return ans;
        } else {
            int ans = 0;
            if (re1.size() > 1) {
                ans = max(ans, sum - re1[0] - re1[1]);
            } 
            if (re2.size() > 0) {
                ans = max(ans, sum - re2[0]);
            }
            return ans;
        }
    }
};
