class Solution {
public:
    int maxSumDivThree(vector<int>& nums) {
        sort(nums.begin(), nums.end(), greater<int>());
        int sum = 0;
        for(auto n : nums) {
            sum += n;
        }
        int rem = sum % 3;
        if(rem == 0)
            return sum;
        
        int i = nums.size() - 1;
        for(; i >= 0; --i) {
            if(nums[i] % 3 == rem) {
                break;
            }
        }
        
        int n = (i >= 0) ? nums[i] : sum;
        int l = i + 1;
        for(int r = nums.size() - 1; r > l; --r) {
            for(int j = r - 1; j >= l; --j) {
                int tmp = nums[r] + nums[j];
                if(tmp < n && tmp % 3 == rem) {
                    n = tmp;
                }
            }
        }
        return sum - n;
    }
};
