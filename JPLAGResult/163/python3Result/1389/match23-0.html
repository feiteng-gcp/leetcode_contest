class Solution:
    def minPushBox(self, grid: List[List[str]]) -> int:
        
        def ps(x):
            if x[1:] not in seen:
                seen.add(x[1:])
                heapq.heappush(bq, x)
        
        dx, dy = [-1, 0, 1, 0], [0, -1, 0, 1]
        n, m = len(grid), len(grid[0])
        for i in range(n):
            for j in range(m):
                if grid[i][j] == 'S':
                    sx, sy = i, j
                    grid[i][j] = '.'
                elif grid[i][j] == 'T':
                    tx, ty = i, j
                elif grid[i][j] == 'B':
                    bx, by = i, j
                    grid[i][j] = '.'
        bq = [(0, sx, sy, bx, by)]
        seen = {(sx, sy, bx, by)}
        while bq:
            push, si, sj, bi, bj= heapq.heappop(bq)
            for k in range(4):
                nx, ny = si + dx[k], sj + dy[k]
                if 0 <= nx < n and 0 <= ny < m:
                    if (nx, ny) == (bi, bj):
                        mx, my = si + dx[k] * 2, sj + dy[k] * 2
                        if 0 <= mx < n and 0 <= my < m:
                            if grid[mx][my] in '.B':
                                ps((push + 1, nx, ny, mx, my))
                            elif grid[mx][my] == 'T':
                                return push + 1
                    elif grid[nx][ny] in '.TB':
                        ps((push, nx, ny, bi, bj))
        return -1
