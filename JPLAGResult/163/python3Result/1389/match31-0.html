import collections
import heapq


class Solution:
    def minPushBox(self, grid: List[List[str]]) -> int:
        m = len(grid)
        n = len(grid[0])
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 'S':
                    player = (i, j)
                elif grid[i][j] == 'T':
                    target = (i, j)
                elif grid[i][j] == 'B':
                    box = (i, j)

        visited = {}
        heap = []
        key = (box[0], box[1], player[0], player[1])
        visited[key] = 0
        heapq.heappush(heap, [0, key])
        while heap:
            steps, key = heapq.heappop(heap)
            bx, by, px, py = key
            if bx == target[0] and by == target[1]:
                return visited[key]
            
            for i, j in [(px + 1, py), (px - 1, py), (px, py + 1), (px, py - 1)]:
                if not (0 <= i < m and 0 <= j < n):
                    continue
                elif grid[i][j] == '#':
                    continue
                elif i == bx and j == by:
                    dx, dy = i - px, j - py
                    p, q = bx + dx, by + dy
                    if not (0 <= p < m and 0 <= q < n):
                        pass
                    elif grid[p][q] != '#':
                        if (p, q, i, j) not in visited:
                            visited[(p, q, i, j)] = visited[key] + 1
                            heapq.heappush(heap, [visited[key] + 1, (p, q, i, j)])
                elif (bx, by, i, j) not in visited:
                    visited[(bx, by, i, j)] = visited[key]
                    heapq.heappush(heap, [visited[key], (bx, by, i, j)])
        return -1
