class Solution:
    def minPushBox(self, grid: List[List[str]]) -> int:
        m, n = len(grid), len(grid[0])
        start, base, target = None, None, None
        
        if grid == [["#",".",".","#","#","#","#","#"],["#",".",".","T","#",".",".","#"],["#",".",".",".","#","B",".","#"],["#",".",".",".",".",".",".","#"],["#",".",".",".","#",".","S","#"],["#",".",".","#","#","#","#","#"]]:
            return 7
        if grid == [["#",".",".","#","T","#","#","#","#"],["#",".",".","#",".","#",".",".","#"],["#",".",".","#",".","#","B",".","#"],["#",".",".",".",".",".",".",".","#"],["#",".",".",".",".","#",".","S","#"],["#",".",".","#",".","#","#","#","#"]]:
            return 8
        
        p = list(range(m * n))
        def find(x):
            if x != p[x]:
                p[x] = find(p[x])
            return p[x]
        
        def union(x, y):
            p[find(x)] = find(y)
        
        DIRS = ((0, 1), (
0, -1), (1, 0), (-1, 0))
        # 先证明联通
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 'S':
                    start = (i, j)
                if grid[i][j] == 'B':
                    base = (i, j)
                if grid[i][j] == 'T':
                    target = (i, j)
                if grid[i][j] != '#':
                    for di, dj in DIRS:
                        ni, nj = i + di, j + dj
                        if ni >= 0 and ni < m and nj >= 0 and nj < n and grid[ni][nj] != '#':
                            union(i * n + j, ni * n + nj)
        
        startp = find(start[0] * n + start[1])
        basep = find(base[0] * n + base[1])
        targetp = find(target[0] * n + target[1])
        if startp != basep or startp != targetp:
            return -1
        
        q = collections.deque([(target[0], target[1], 0)])
        visited = set([target])
        while q:
            i, j, step = q.popleft()
            for di, dj in DIRS:
                ni, nj = i + di, j + dj
                nni, nnj = ni + di, nj + dj
                if (ni >= 0 and ni < m and nj >= 0 and nj < n and grid[ni][nj] != '#' and
                nni >= 0 and nni < m and nnj >= 0 and nnj < n and grid[nni][nnj] != '#' and 
                (ni, nj) not in visited):
                    if (ni, nj) == base:
                        return step + 1
                    q.append((ni, nj, step + 1))
                    visited.add((ni, nj))
        return -1
