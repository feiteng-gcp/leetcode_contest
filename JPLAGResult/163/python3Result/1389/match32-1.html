class Solution:
    def minPushBox(self, grid: List[List[str]]) -> int:
        n,m=len(grid),len(grid[0])
        seen=set()
        def reach(st,ed,Bpos):
            if st==Bpos:
                return False
            if st[0]<0 or st[0]>=n or st[1]<0 or st[1]>=m:
                return False
            if grid[st[0]][st[1]]=='#':
                return False
            if st==ed:
                return True
            seen.add(st)
            for i,j in [[-1,0],[1,0],[0,-1],[0,1]]:
                nst=(st[0]+i,st[1]+j)
                if nst not in seen:
                    tmp=reach(nst,ed,Bpos)
                    if tmp:
                        return True
            return False
        
for i in range(n):
            for j in range(m):
                if grid[i][j]=='B':
                    Bpos=(i,j)
                if grid[i][j]=='T':
                    Tpos=(i,j)
                if grid[i][j]=='S':
                    Ppos=(i,j)
        if Bpos==Tpos:
            return 0
        init=False
        for i,j in [[-1,0],[1,0],[0,-1],[0,1]]:
            seen=set()
            pos=(Bpos[0]+i,Bpos[1]+j)
            if 0<=pos[0]<n and 0<=pos[1]<m and grid[pos[0]][pos[1]]!='#':
                if reach(Ppos,pos,Bpos):
                    Ppos=pos
                    init=True
                    break
        if not init:
            return -1
        cur={(Ppos,Bpos)}
        Seen={(Ppos,Bpos)}
        ans=0
        while cur:
            nxt=set()
            ans+=1
            for Ppos,Bpos in cur:
                for i,j in [[-1,0],[1,0],[0,-1],[0,1]]:
                    seen=set()
                    pos=(Bpos[0]+i,Bpos[1]+j)
                    if 0<=pos[0]<n and 0<=pos[1]<m and grid[pos[0]][pos[1]]!='#':
                        if reach(Ppos,pos,Bpos):
                            d=(-i,-j)
                            nBpos=(Bpos[0]+d[0],Bpos[1]+d[1])
                            if 0<=nBpos[0]<n and 0<=nBpos[1]<m and grid[nBpos[0]][nBpos[1]]!='#':
                                nPpos=Bpos
                                if nBpos==Tpos:
                                    return ans
                                if (nPpos,nBpos) not in Seen:
                                    Seen.add((nPpos,nBpos))
                                    nxt.add((nPpos,nBpos))
            cur=nxt
        return -1
