class Solution:
    def minPushBox(self, grid: List[List[str]]) -> int:
        from collections import defaultdict
        visited = defaultdict(lambda: float("inf"))
        
        m, n = len(grid), len(grid[0])
        for i in range(m):
            for j in range(n):
                if grid[i][j] == "S":
                    xS, yS = i, j
                elif grid[i][j] == "B":
                    xB, yB = i, j
                elif grid[i][j] == "T":
                    xT, yT = i, j
                if grid[i][j] != "#":
                    grid[i][j] = "."
        
        visited[(xS, yS, xB, yB)] = 0
        
        from queue import Queue
        q = Queue()
        q.put((xS, yS, xB, yB, 0))
        
        while not q.empty():
            tmp = q.get()
            xS, yS, xB, yB = tmp[0], tmp[1], tmp[2], tmp[3]
            numPushes = tmp[4]
            
            if xS > 0:
                if xS == xB + 1 and yS == yB and xS > 1:
                    if grid[xS - 2][yS] == "." and visited[(xS - 1, yS, xB - 1, yB)] > numPushes + 1:
                        visited[(xS - 1, yS, xB - 1, yB)] = numPushes + 1
                        q.put((xS - 1, yS, xB - 1, yB, numPushes + 1))    
                else:
                    if grid[xS - 1][yS] == "." and visited[(xS - 1, yS, xB, yB)] > numPushes:
                        visited[(xS - 1, yS, xB, yB)] = numPushes
                        q.put((xS - 1, yS, xB, yB, numPushes))
            if xS < m - 1:
                if xS == xB - 1 and yS == yB and xS < m - 2:
                    if grid[xS + 2][yS] == "." and visited[(xS + 1, yS, xB + 1, yB)] > numPushes + 1:
                        visited[(xS + 1, yS, xB + 1, yB)] = numPushes + 1
                        q.put((xS + 1, yS, xB + 1, yB, numPushes + 1))    
                else:
                    if grid[xS + 1][yS] == "." and visited[(xS + 1, yS, xB, yB)] > numPushes:
                        visited[(xS + 1, yS, xB, yB)] = numPushes
                        q.put((xS + 1, yS, xB, yB, numPushes))
            if yS > 0:
                if yS == yB + 1 and xS == xB and yS > 1:
                    if grid[xS][yS - 2] == "." and visited[(xS, yS - 1, xB, yB - 1)] > numPushes + 1:
                        visited[(xS, yS - 1, xB, yB - 1)] = numPushes + 1
                        q.put((xS, yS - 1, xB, yB - 1, numPushes + 1))    
                else:
                    if grid[xS][yS - 1] == "." and visited[(xS, yS - 1, xB, yB)] > numPushes:
                        visited[(xS, yS - 1, xB, yB)] = numPushes
                        q.put((xS, yS - 1, xB, yB, numPushes))
            if yS < n - 1:
                if yS == yB - 1 and xS == xB and yS < n - 2:
                    if grid[xS][yS + 2] == "." and visited[(xS, yS + 1, xB, yB + 1)] > numPushes + 1:
                        visited[(xS, yS + 1, xB, yB + 1)] = numPushes + 1
                        q.put((xS, yS + 1, xB, yB + 1, numPushes + 1))    
                else:
                    if grid[xS][yS + 1] == "." and visited[(xS, yS + 1, xB, yB)] > numPushes:
                        visited[(xS, yS + 1, xB, yB)] = numPushes
                        q.put((xS, yS + 1, xB, yB, numPushes))
            
        res = float("inf")
        for i in range(m):
            for j in range(n):
                res = min(res, visited[(i, j, xT, yT)])
            
        return res if res < float("inf") else -1
