from heapq import *
class Solution:
    def minPushBox(self, grid: List[List[str]]) -> int:
        dx,dy = [0,0,-1,1],[-1,1,0,0]
        m,n = len(grid),len(grid[0])
        ps,bs,t = (0,0),(0,0),(0,0)
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 'S':
                    ps = (i,j)
                if grid[i][j] == 'B':
                    bs = (i,j)
                if grid[i][j] == 'T':
                    t = (i,j)
        q = [(0,ps,bs)]
        heapify(q)
        vis = {(ps,bs)}
        while len(q):
            step, curp, curb = heappop(q)
            if curb == t:
                return step
            for i in range(4):
                nxtpx,nxtpy = curp[0] + dx[i], curp[1] + dy[i]
                if 0<=nxtpx<m and 0<=nxtpy<n and grid[nxtpx][nxtpy] != '#':
                    if (nxtpx,nxtpy) == curb:
                        nxtbx, nxtby = curb[0] + dx[i], curb[1] + dy[i]
                        if 0<=nxtbx<m and 0<=nxtby<n and grid[nxtbx][nxtby] != '#' and ((nxtpx,nxtpy),(nxtbx, nxtby)) not in vis:
                            vis.add(((nxtpx,nxtpy),(nxtbx, nxtby)))
                            heappush(q,(step+1,(nxtpx,nxtpy),(nxtbx, nxtby)))
                    else:
                        if ((nxtpx,nxtpy),curb) not in vis:
                            vis.add(((nxtpx,nxtpy),curb))
                            heappush(q,(step,(nxtpx,nxtpy),curb))
        return -1
