class Solution:
    def minPushBox(self, grid: List[List[str]]) -> int:
        n = len(grid)
        m = len(grid[0])

        for i in range(n):
            for j in range(m):
                if grid[i][j] == 'S':
                    pos_man = i, j
                elif grid[i][j] == 'T':
                    pos_target = i, j
                elif grid[i][j] == 'B':
                    pos_box = i, j

        def neis(state):
            i_box, j_box, i_man, j_man = state
            for di, dj in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                ii_man = i_man + di
                jj_man = j_man + dj
                if 0 <= ii_man < n and 0 <= jj_man < m and grid[ii_man][jj_man] != '#':
                    if ii_man != i_box or jj_man != j_box:
                        yield (i_box, j_box, ii_man, jj_man), False
                    else:
                        ii_box= i_box + di
                        jj_box = j_box + dj
                        if 0 <= ii_box < n and 0 <= jj_box < m and grid[ii_box][jj_box] != '#':
                            yield (ii_box, jj_box, ii_man, jj_man), True

        def is_target(state):
            i_box, j_box, i_man, j_man = state
            return i_box==pos_target[0] and j_box == pos_target[1]

        q = [(0, pos_box+pos_man)]
        visited_dist = collections.defaultdict(int)
        while q:
            dist, state = heapq.heappop(q)
            if state not in visited_dist:
                visited_dist[state] = dist
                if is_target(state):
                    return dist
                for state2, pushed in neis(state):
                    if state2 not in visited_dist:
                        heapq.heappush(q, (dist+pushed, state2))
        return -1
