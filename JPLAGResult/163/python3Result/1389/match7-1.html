class Solution:
    def minPushBox(self, grid: List[List[str]]) -> int:
        n, m = len(grid), len(grid[0])
        R, T = [0, 0], (0, 0)
        P = [(0, 1), (0, -1), (1, 0), (-1, 0)]
        B = set()
        for i in range(n):
            for j in range(m):
                if grid[i][j] != '#':
                    B.add((i, j))
                if grid[i][j] == 'S':
                    R[0] = (i, j)
                if grid[i]
[j] == 'B':
                    R[1] = (i, j)
                if grid[i][j] == 'T':
                    T = (i, j)
        S = {(R[0], R[1])}
        R = [(R[0], R[1], 0)]
        def D(p, C):
            for d in P:
                np = (p[0] + d[0], p[1] + d[1])
                if np in B and np not in C:
                    C.add(np)
                    D(np, C)
        while R:
            r = R.pop(0)
            C = {r[0], r[1]}
            D(r[0], C)
            for d in P:
                np = (
r[1][0] + d[0], r[1][1] + d[1])
                bp = (r[1][0] - d[0], r[1][1] - d[1])
                if np in C and bp in B:
                    if bp == T:
                        return r[2] + 1
                    if (r[1], bp) not in S:
                        R.append((r[1], bp, r[2] + 1))
                        S.add((r[1], bp))
                #print(R, S)
        return -1
