class Solution:
    def minPushBox(self, g: List[List[str]]) -> int:
        m, n = len(g), len(g[0])
        p = [[True] * n for _ in range(m)]
        for i in range(m):
            for j in range(n):
                if g[i][j] == '#':
                    p[i][j] = False
                elif g[i][j] == 'S':
                    s = (i, j)
                elif g[i][j] == 'B':
                    b = (i, j)
                elif g[i][j] == 'T':
                    t = (i, j)
        # s move b to t
        if b == t: return 0
        
        def valid(a):
            return 0 <= a[0] < m and 0 <= a[1] < n and p[a[0]][a[1]]
        
        def path(s, ns, b):
            pq = collections.deque([s])
            pvisited = set([s, b])
            while pq:
                curs = pq.popleft()
                for dx, dy in [[-1, 0], [1, 0], [0, -1], [0, 1]]:
                    news = (curs[0] + dx, curs[1] + dy)
                    if not valid(news): continue
                    if news in pvisited: continue
                    if news == ns: return True
                    pq.append(news)
                    pvisited.add(news)
            return False
           
        q = collections.deque([(s, b, 0)])
        visited = set([(s, b)])
        
        while q:
            curs, curb, steps = q.popleft()
            for dx, dy in [[-1, 0], [1, 0], [0, -1], [0, 1]]:
                news = (curb[0] + dx, curb[1] + dy)
                newb = (curb[0] - dx, curb[1] - dy)
                if not valid(news) or not valid(newb): continue
                if (news, newb) in visited: continue
                if not path(curs, news, curb): continue
                if newb == t: return steps + 1
                q.append((news, newb, steps + 1))
                visited.add((news, newb))
        return -1
