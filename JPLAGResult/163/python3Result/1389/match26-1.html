class Solution:
    def minPushBox(self, grid: List[List[str]]) -> int:
        N = len(grid)
        M = len(grid[0])
        q = []
        
        start = None
        box = None
        end = None
        
for i in range(N):
            for j in range(M):
                if grid[i][j] == "S":
                    start = (i, j)
                    grid[i][j] = "."
                elif grid[i][j] == "B":
                    box = (i, j)
                    grid[i][j] = "."
                elif grid[i][j] == "T":
                    end = (i, j)
                    grid[i][j] = "."
        
        seen = {}
        seen[(start, box)] = 0
        heapq.heappush(q, (0, start, box))
        
        directions = [(1, 0), (0, 1), (-1, 0), (0, -1)]
        best = 1000000000
        count = 0
        last = -1
        while len(q) > 0:
            (d, pos, bpos) = heapq.heappop(q)
                
            for (dx, dy) in directions:
                nx = pos[0] + dx
                ny = pos[1] + dy
                
                if nx >= 0 and ny >= 0 and nx < N and ny < M and grid[nx][ny] != "#":
                    if nx == bpos[0] and ny == bpos[1]:
                        nbx = bpos[0] + dx
                        nby = bpos[1] + dy
                        
                        if nbx >= 0 and nby >= 0 and nbx < N and nby < M and grid[nbx][nby] != '#':
                            if nbx == end[0] and nby == end[1]:
                                best = min(best, d + 1)
                                return d + 1
                            
                            key = ((nx, ny), (nbx, nby))
                            if key in seen:
                                continue
                            
                            heapq.heappush(q, ((d + 1), (nx, ny), (nbx, nby)))
                            seen[key] = d + 1
                    else:
                        key = ((nx, ny), bpos)
                        if key in seen:
                            continue
                            
                        seen[key] = d
                        heapq.heappush(q, (d, (nx, ny), bpos))

        if best >= 1000000000:
            return -1
        return best
