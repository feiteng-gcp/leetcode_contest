class Solution:
    def minPushBox(self, grid: List[List[str]]) -> int:
        h, w = len(grid), len(grid[0])
        loc_s, loc_b, loc_t = None, None, None
        for i in range(h):
            for j in range(w):
                if grid[i][j] == 'S':
                    loc_s = (i, j)
                elif grid[i][j] == 'B':
                    loc_b = (i, j)
                elif grid[i][j] == 'T':
                    loc_t = (i, j)
        ans = 1
        layer = [(loc_b, loc_s)]
        visited = [[[False for k in range(4)] for j in range(w)] for i in range(h)]
        directions = ((0, 1), (1, 0), (0, -1), (-1, 0))
        while layer:
            next_layer = []
            for box, person in layer:
                for idx, dir in enumerate(directions):
                    stand_y, stand_x = box[0] + dir[0], box[1] + dir[1]
                    oppo_y, oppo_x = box[0] - dir[0], box[1] - dir[1]
                    if not (0 <= stand_y < h and 0 <= stand_x < w and 0 <= oppo_y < h and 0 <= oppo_x < w and grid[stand_y][stand_x] != '#' and grid[oppo_y][oppo_x] != '#' and not visited[oppo_y][oppo_x][idx]):
                        continue
                    walked = [[False for j in range(w)] for i in range(h)]
                    if self._reachable(grid, walked, person[0], person[1], box, stand_y, stand_x, directions):
                        visited[oppo_y][oppo_x][idx] = True
                        if loc_t[0] == oppo_y and loc_t[1] == oppo_x:
                            return ans
                        next_layer.append(((oppo_y, oppo_x), box))
            layer = next_layer
            ans += 1
        return -1
    
    def _reachable(self, grid, walked, y, x, box, tgt_y, tgt_x, directions):
        if y == tgt_y and x == tgt_x:
            return True
        walked[y][x] = True
        for dir in directions:
            ny, nx = y + dir[0], x + dir[1]
            if 0 <= ny < len(grid) and 0 <= nx < len(grid[0]) and grid[ny][nx] != '#' and not walked[ny][nx] and (ny, nx) != box:
                if self._reachable(grid, walked, ny, nx, box, tgt_y, tgt_x, directions):
                    return True
        return False
