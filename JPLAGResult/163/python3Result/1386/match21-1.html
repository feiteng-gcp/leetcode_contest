class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        n, m = len(grid), len(grid[0])
        k = k % (n * m)
        for _ in range(k):
            tail = grid[-1][-1]
            tmpList = []
            for i in range(n-1):
                tmpList.append(grid[i][-1])
            
for i in range(n):
                for j in range(m-1, 0, -1):
                    grid[i][j] = grid[i][j-1]
            for i in range(1, n):
                grid[i][0] = tmpList[i-1]
            grid[0][0] = tail
        return grid
