class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        h, w = len(grid), len(grid[0])
        new_grid = [[0]*w for _ in range(h)]
        for i in range(h):
            for j in range(w):
                new_pos = i*w + j + k
                new_pos %= h*w
                ni, nj = new_pos // w, new_pos % w
                new_grid[ni][nj] = grid[i][j]
        return new_grid
