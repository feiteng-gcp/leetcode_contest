class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        len1 = len(grid)
        len2 = len(grid[0])
        res = [[0 for i in range(len2)] for j in range(len1)]
        size = len2 * len1
        for i in range(len1):
            for j in range(len2):
                t = i * len2 + j
                print(t)
                t += k
                t %= size
                #print(grid[i][j])
                #print(t, t//len1, t%len2)
                res[t//len2][t%len2] = grid[i][j]
        return res
