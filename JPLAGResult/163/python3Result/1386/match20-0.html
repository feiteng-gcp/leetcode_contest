class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        
        n = len(grid)
        m = len(grid[0])
        
        for _ in range(k):
            g = [r.copy() for r in grid]
            
            for i in range(n):
                for j in range(m-1):
                    grid[i][j+1] = g[i][j]
                
            
            for i in range(n-1):
                grid[i+1][0] = g[i][m-1]
            
            grid[0][0] = g[n-1][m-1]
        
        return grid
