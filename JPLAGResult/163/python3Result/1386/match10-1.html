class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        n, m = len(grid), len(grid[0])
        ans = [[0]*m for _ in range(n)]
        for i in range(n):
            
for j in range(m):
                p = (i * m + j + k) % (m*n)
                ans[p//m][p%m] = grid[i][j]
        return ans
