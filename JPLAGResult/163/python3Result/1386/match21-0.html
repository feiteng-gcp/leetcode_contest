class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        
        n, m = len(grid), len(grid[0])
        
        numk = k
        
        for i in range(numk):
            
            t = [grid[j][m - 1] for j in range(n)]
            
            for k in range(n):
                for h in range(m - 1, 0, -1):
                    grid[k][h] = grid[k][h - 1]
                    
            for k in range(1, n):
                grid[k][0] = t[k - 1]
                
            grid[0][0] = t[n - 1]
            
        return grid
