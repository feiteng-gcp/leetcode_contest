class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        for _ in range(k):
            new = [row[:] for row in grid]
            for i in range(len(grid)):
                for j in range(len(grid[0])):
                    if j == 0:
                        new[i][j] = grid[-1][-1] if i == 0 else grid[i-1][-1]
                    else:
                        new
[i][j] = grid[i][j-1]
            grid = new
        return grid
