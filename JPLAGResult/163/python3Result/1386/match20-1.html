class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        m = len(grid)
        n = len(grid[0])
        for t in range(k):
            grid1 = [[0]*n for _ in range
(m)]
            for j in range(1,n):
                for i in range(m):
                    grid1[i][j] = grid[i][j-1]
            for i in range(m):
                grid1[i][0] = grid[(i-1+m)%m][n-1]
            grid = grid1
        return grid
