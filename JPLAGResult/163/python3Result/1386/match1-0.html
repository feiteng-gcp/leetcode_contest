class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        for _ in range(k):
            for i in range(len(grid)):
                grid[i] =  [grid[i][-1]] + grid[i][:-1]
            col = [row[0] for row in grid]
            col = [col[-1]] +col[:-1]
            for i in range(len(grid)):
                grid[i][0] = col[i]
        return grid
