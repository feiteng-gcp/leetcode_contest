class Solution:
    def shiftGrid(self, g: List[List[int]], k: int) -> List[List[int]]:
        m, n = len(g), len(g[0])
        q = []
        for i in range(m):
            for j in range(n):
                q.append(g[i][j])
        k = k %  (m * n)
        q = q[-k:] + q[:-k]
        ret = []
        for i in range(m):
            ret.append(q[i*n:(i+1)*n])
        return ret
