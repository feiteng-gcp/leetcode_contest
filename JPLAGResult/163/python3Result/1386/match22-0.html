class Solution:
    def shiftGrid(self, grid, k: int):
        n = len(grid)
        m = len(grid[0])
        
        for _ in range(k):
            new = [[0] * m for _ in range(n)]

            for i in range(n):
                for j in range(m):
                    if j == m - 1:
                        if i == n - 1:
                            new[0][0] = grid[n-1][m-1]
                        else:   
                            new[i+1][0] = grid[i][m-1]
                    else:
                        new[i][j+1] = grid[i][j]
            grid = new

        return grid
