class Solution:#O(mnk)
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        m, n = len(grid), len(grid[0])
        
        res = [[0 for _ in range(n)] for _ in range(m)]
        
        for i in range(m * n):
            idx = (m * n - k + i) % (m * n)
            res[i // n][i % n] = grid[idx // n][idx % n]
        
        
        return res
