class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        n = len(grid)
        m = len(grid[0])
        new_grid = []
        for x in grid:
            new_grid +
= x
        
        start_pos = k % (n*m)
        new_grid = new_grid[-start_pos:] + new_grid[:-start_pos]
        rotated_grid = []
        for i in range(n):
            rotated_grid.append(new_grid[m*i:m*(i+1)])
        return rotated_grid
