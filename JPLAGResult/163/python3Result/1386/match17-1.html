class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        n = len(grid)
        m = len(grid[0])
        final_grid = [[None for _ in range(m)] for _ in range(n)]
        
        for i in range(n):
            for j in range
(m):
                new_i = (i + (k + j) // m) % n
                new_j = (j + k) % m
                final_grid[new_i][new_j] = grid[i][j]
        return final_grid
