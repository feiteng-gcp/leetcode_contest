class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        m, n = len(grid), len(grid[0])
        ans = [[0] * n for i in range(m)]
        
        for i in range(m):
            for j in range(n):
                pos = i * n + j + k
                new_i = (pos//n) % m
                new_j = pos % n
                ans[new_i][new_j] = grid[i][j]
        return ans
