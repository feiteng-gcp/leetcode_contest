class Solution:
    def shiftGrid(self, grid, K):
        if K==0:
            return grid
        
        n = len(grid)
        m = len(grid[0])

        ngrid = []
        for k in range(K):
            ngrid = [[0 for x in range(m)] for y in range(n)]
            for i in range(n-1, -1, -1):
                for j in range(m-1, -1, -1):
                    if i == n-1 and j==m-1:
                        #swap
                        ngrid[0][0] = grid[i][j]
                    elif j==m-1:
                        #swap grid[i+1][0]
                        ngrid[i+1][0] = grid[i][j]
                    else:
                        ngrid[i][j+1] = grid[i][j]
            grid = ngrid
        return ngrid
