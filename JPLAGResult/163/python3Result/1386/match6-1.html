from collections import deque


class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        n, m = len(grid), len(grid[0])
        x, y = divmod(k , m)
        
        tmp = [[0] * m for _ in range(n)]
        for j in range(m):
            for i in range(n):
                tmp[i][(j + y) % m] = grid[i][j]
        
        fmove, smove = (x + 1) % n, x % n
        for j in range(y):
            for i in range(n):
                grid[(i + fmove) % n][j] = tmp[i][j]
        
        
for j in range(y, m):
            for i in range(n):
                grid[(i + smove) % n][j] = tmp[i][j]
                
        return grid
