class Solution:
    def shiftGrid(self, g: List[List[int]], k: int) -> List[List[int]]:
        n, m = len(g), len(g[0])
        
        ret = [[0] * m for _ in range(n)]
        for j in range(m):
            nj = (j + k) % m
            for i in range(n):
                ni = (i + (j+k)//m) % n
                # print (i, j, g[i][j], "-->", ni, nj, j+k//m)
                ret[ni][nj] = g[i][j]
        
        return ret
