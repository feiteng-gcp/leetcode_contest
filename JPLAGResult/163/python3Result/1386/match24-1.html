class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        m = len(grid)
        n = len(grid[0])
        nn = []
        for i in range(m):
            for j in range(n):
                nn.append(grid[i][j])
        k = k%(m*n)
        nn = nn[-k:]+nn[:-k]
        for i in range(m):
            for j in range(n):
                grid[i][j] = nn[i*n+j]
        return grid
