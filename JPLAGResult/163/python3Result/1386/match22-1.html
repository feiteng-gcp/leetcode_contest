class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        N = len(grid)
        M = len(grid[0])
        k %= (N * M)
        
for _ in range(k):
            new_grid = [[0] * M for _ in range(N)]
            for i in range(N):
                for j in range(M):
                    if j + 1 == M:
                        if i + 1 == N:
                            new_grid[0][0] = grid[i][j]
                        else:
                            new_grid[i+1][0] = grid[i][j]
                    else:
                        new_grid[i][j+1] = grid[i][j]
            grid = new_grid
        return grid
