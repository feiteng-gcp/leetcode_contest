class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        m, n = len(grid), len(grid[0])
        mod = m * n
        res = [[0 for j in range(n)] for i in range(m)]
        for i in range(m):
            for j in range(n):
                pos = (i * n + j + k) % mod
                x = pos // n
                y = pos % n
                res[x][y] = grid[i][j]
        return res
