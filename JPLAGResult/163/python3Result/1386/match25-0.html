class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        n=len(grid)
        m=len(grid[0])
        l=[]
        for i in range(n):
            for j in range(m):
                l.append(grid[i][j])
        k=k%(m*n)
        l=l[-k:]+l[:m*n-k]
        res=[]
        for i in range(n):
            res.append(l[i*m:(i+1)*m])
        return res
