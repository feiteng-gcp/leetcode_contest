class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        nrows = len(grid)
        ncols = len(grid[0])
        
        size = nrows * ncols
        
        k = k % size
        
        values = []
        for r in range(nrows):
            for c in range(ncols):
                values.append(grid[r][c])
        
        values_after_shift = values[-k:] + values[:size - k]
        
        res = []
        for r in range(nrows):
            res.append(values_after_shift[r * ncols : (r + 1) * ncols])
        
        return res
