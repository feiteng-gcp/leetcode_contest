class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        n, m = len(grid), len(grid[0])
        ans = [[0 for _ in range(m)] for _ in range(n)]
        for i in range(n):
            for j in range(m):
                index = i * m + j + k
                index %= (n * m)
                y = index // m
                x = index % m
                ans[y][x] = grid[i][j]
        return ans
