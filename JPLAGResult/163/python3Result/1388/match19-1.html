class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        sums = sum(nums)
        q1, q2 = [], []

        for num in nums:
            if num%3 == 1:
                q1
.append(num)
            elif num%3 == 2:
                q2.append(num)

        q1.sort()
        q2.sort()
        len1, len2 = len(q1), len(q2)
        # print(sums)
        # print(q1)
        # print(q2)
        # delta = min([sum(q1[:3]) if len1 >= 3 else float('inf'), 
        #     sum(q1[:2])+sum(q2[:2]) if len1 >= 2 and len2 >= 2 else float('inf'),
        #     sum(q1[:1])+sum(q2[:4]) if len1 >= 1 and len2 >= 4 else float('inf'),
        #     sum(q2[:3]) if len2 >= 3 else float('inf')])
        res = []
        for i in range(min(len1+1,7)):
            for j in range(min(len2+1,7)):
                tmp = sums - sum(q1[:i]) - sum(q2[:j])
                # print(i,j,tmp)
                if tmp%3 == 0:
                    res.append(tmp)
        if not res:
            return 0
        return max(res)
