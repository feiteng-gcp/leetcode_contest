class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        total = 0
        t = [[], [], []]
        for n in nums:
            remain = n % 3
            total += n
            if remain:
                t[remain].append(n)
        t[1].sort()
        t[2].sort()
        if total % 3 == 0:
            return total
        total1 = total2 = float('-inf')
        if total % 3 == 1:
            if t[1]:
                total1 = total - t[1][0]
            if len(t[2]) >= 2:
                total2 = total - sum(t[2][:2])
        else:
            if len(t[1]) >= 2:
                total1 = total - sum(t[1][:2])
            if t[2]:
                total2 = total - t[2][0]
        return max(total1, total2)
