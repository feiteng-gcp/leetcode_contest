class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        m = [[] for _ in range(3)]
        for num in nums:
            m[num % 3].append(num)
        
        m[1].sort()
        m[2].sort()

        res = sum(nums)
        if res % 3 != 0:
            p = res % 3
            q = 3 - p
            if len(m[p]) > 0:
                c = m[p][0]
            if len(m[q]) > 1:
                c = min(sum(m[q][:2]), c)
            res -= c
        
        return res
