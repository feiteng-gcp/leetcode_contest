class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        total = 0
        mod_1 = []
        mod_2 = []
        total_sum = 0
        for n in nums:
            total_sum += n
            
if n % 3 == 0:
                total += n
            elif n % 3 == 1:
                mod_1.append(n)
            elif n % 3 == 2:
                mod_2.append(n)
                
        mod_1.sort()
        mod_2.sort()
        
        if total_sum % 3 == 0:
            return total_sum
        elif total_sum % 3 == 1:
            # either remove 1 1 mod 3 or 2 2 mod 3
            if len(mod_1) >= 1:
                remove_mod_1 = mod_1[0]
            else:
                remove_mod_1 = total_sum
                
            if len(mod_2) >= 2:
                remove_mod_2 = mod_2[0] + mod_2[1]
            else:
                remove_mod_2 = total_sum
            
            return total_sum - min(remove_mod_1, remove_mod_2)
        elif total_sum % 3 == 2:
            # either remove 1 2 mod 3 or 2 1 mod 3
            if len(mod_1) >= 2:
                remove_mod_1 = mod_1[0] + mod_1[1]
            else:
                remove_mod_1 = total_sum
                
            if len(mod_2) >= 1:
                remove_mod_2 = mod_2[0]
            else:
                remove_mod_2 = total_sum
            
            return total_sum - min(remove_mod_1, remove_mod_2)
