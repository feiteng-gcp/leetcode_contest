class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        n = len(nums)
        dp = [[0 for _ in range(n)] for _ in range(3)]
        dp[nums[0] % 3][0] = nums[0]
        for i in range(1, n):
            for r in range(3):
                include = dp[(r - nums[i]) % 3][i-1] + nums
[i]
                if include % 3 == r:
                    dp[r][i] = max(dp[r][i-1], include)
                else:
                    dp[r][i] = dp[r][i-1]
        return dp[0][n-1]
