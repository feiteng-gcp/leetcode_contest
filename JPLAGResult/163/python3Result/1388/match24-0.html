from collections import defaultdict

class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        lut = defaultdict(list)
        total_sum = sum(nums)
        for num in nums:
            lut[num%3].append(num)
        lut[0].sort()
        lut[1].sort()
        lut[2].sort()

        remainder = total_sum%3
        
        if remainder == 0:
            return total_sum
        if remainder == 1:
            ans1 = total_sum - lut[1][0] if lut[1] else 0
            ans2 = total_sum - sum(lut[2][0:2]) if len(lut[2]) >= 2 else 0
            return max(ans1, ans2)
        if remainder == 2:
            ans1 = total_sum - lut[2][0] if lut[2] else 0
            ans2 = total_sum - sum(lut[1][0:2]) if len(lut[1]) >= 2 else 0
            return max(ans1, ans2)
