class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        prev = [float('-inf')] * 3
        prev[0] = 0
        for n in nums:
            cur = [float('-inf')] * 3
            rem = n % 3
            cur[0] = max(prev[0],prev[(3-rem)%3]+n)
            cur[1] = max(prev[1], prev[(3-rem+1)%3]+n)
            cur[2] = max(prev[2], prev[(3-rem+2)%3]+n)
            prev = cur
        return prev[0]
