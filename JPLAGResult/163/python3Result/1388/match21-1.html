from collections import defaultdict
class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        
        S = sum(nums)
        min1, sec1 = float('inf'), float('inf')
        min2, sec2 = float('inf'), float('inf')
        for num in nums:
            
            
if num % 3 == 1:
                if num < min1:
                    min1, sec1 = num, min1
                elif num < sec1:
                    sec1 = num
                    
            if num % 3 == 2:
                if num < min2:
                    min2, sec2 = num, min2
                elif num < sec2:
                    sec2 = num
                    
        if S % 3 == 1:
            sub = min(min2 + sec2, min1)
            return S - sub if sub != float('inf') else 0
        elif S % 3 == 2:
            sub = min(min1 + sec1, min2)
            return S - sub if sub != float('inf') else 0
        else:
            return S
