class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        tmp0 = []
        tmp1 = []
        tmp2 = []
        
        s = sum(nums)
        for i in nums:
            if i % 3 == 1:
                tmp1.append(i)
            if i % 3 == 2:
                tmp2.append(i)
            if i % 3 == 0:
                tmp0.append(i)
        tmp0.sort()
        tmp1.sort()
        tmp2.sort()
        
        if s % 3 == 0:
            return s
        if s % 3 == 1:
            x = 0;
            if len(tmp1) > 0:
                x = s - tmp1[0]
            if len(tmp2) > 1:
                x = max(x, s - tmp2[0] - tmp2[1])
            return x
        if s % 3 == 2:
            x = 0
            if len(tmp2) > 0:
                x = s - tmp2[0]
            if len(tmp1) > 1:
                x = max(x, s - tmp1[0] - tmp1[1])
            if len(tmp2) > 1 and len(tmp1) > 0:
                x = max(x, s - tmp2[0] - tmp2[1] - tmp1[0])
            return x
