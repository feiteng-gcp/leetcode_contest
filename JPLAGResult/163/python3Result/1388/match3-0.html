class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        b1, b2, b3 = [], [], []
        sum_all = 0
        for i in nums:
            if i % 3 == 0:
                b1.append(i)
            elif i % 3 == 1:
                b2.append(i)
            else:
                b3.append(i)
            sum_all += i
        if sum_all % 3 == 0:
            return sum_all
        elif sum_all % 3 == 1:
            t1, t2 = 0, 0
            if len(b2): t1 = sum_all - min(b2)
            if len(b3) > 1: t2 = sum_all - sum(sorted(b3)[:2])
            return max(t1, t2)
        else:
            t1, t2 = 0, 0
            if len(b3): t1 = sum_all - min(b3)
            if len(b2) > 1:t2 = sum_all - sum(sorted(b2)[:2])
            return max(t1, t2)
