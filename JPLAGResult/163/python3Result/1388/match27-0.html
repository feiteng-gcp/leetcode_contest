class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        nums1 = []
        nums2 = []
        total = sum(nums)
        for item in nums:
            if item % 3 == 1:
                nums1.append(item)
            if item % 3 == 2:
                nums2.append(item)
        nums1.sort()
        nums2.sort()
        l1 = len(nums1)
        l2 = len(nums2)
        if l1 == l2  or (l1 - l2) %3 == 0:
            return total
        elif l1 < l2:
            if (l2 - l1) % 3 == 1:
                if l1 >=2:
                    return max(total - nums2[0],total - nums1[0] - nums1[1])
                else:
                    return total - nums2[0]
            elif (l2 - l1) % 3 == 2:
                if l1>=1:
                    return max(total - nums2[0] - nums2[1],total - nums1[0])
                else:
                    return total - nums2[0] - nums2[1]
        else:
            if (l1 - l2) % 3 == 1:
                if l2 >=2:
                    return max(total - nums2[0] - nums2[1],total - nums1[0])
                else:
                    return total - nums1[0]
            elif (l1 - l2) % 3 == 2:
                if l2 >=1:
                    return max(total - nums2[0],total - nums1[0] - nums1[1])
                else:
                    return total - nums1[0] - nums1[1]
