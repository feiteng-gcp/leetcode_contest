class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        N = sum(nums)
        mod1 = []
        mod2 = []
        
        if N % 3 == 0:
            return N
        bst = N
        for x in nums:
            if x % 3 == 1:
                heapq.heappush(mod1, x)
            elif x % 3 == 2:
                heapq.heappush(mod2, x)
        
        if N % 3 == 1:
            if mod1 and mod1[0] < bst:
                bst = mod1[0]
            if len(mod2) > 1 and (mod2[0] + min(mod2[1], mod2[2] if len(mod2) > 2 else N)) < bst:
                bst = mod2[0] + min(mod2[1], mod2[2] if len(mod2) > 2 else N)
            return N - bst
        
        if N % 3 == 2:
            if mod2 and mod2[0] < bst:
                bst = mod2[0]
            if len(mod1) > 1 and (mod1[0] + min(mod1[1], mod1[2] if len(mod1) > 2 else N)) < bst:
                bst = mod1[0] + min(mod1[1], mod1[2] if len(mod1) > 2 else N)
            return N - bst
