class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        f0, f1, f2 = [],[],[]
        for num in nums:
            if num % 3 == 0:
                f0.append(num)
            elif num % 3 == 1:
                f1.append(num)
            else:
                f2.append(num)
        f1.sort()
        f2.sort()
        n1,n2 = len(f1),len(f2)
        res = sum(f1) + sum(f2)
        if res % 3 == 1:
            if n1 >= 1 and n2 >= 2:
                res = max(res-f1[0],res-f2[0]-f2[1])
            elif n1 >= 1:
                res -= f1[0]
            elif n2 >= 2:
                res -= (f2[0] + f2[1])
        elif res % 3 == 2:
            if n1 >= 2 and n2 >= 1:
                res = max(res-f1[0]-f1[1],res-f2[0])
            elif n1 >= 2:
                res -= (f1[0] + f1[1])
            elif n2 >= 1:
                res -= f2[0]
        return res + sum(f0)
