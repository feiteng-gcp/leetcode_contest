class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        
        
        
        
        res = sum(nums)
        if res%3 == 0:
            return res
        
        res0, res1, res2 = 0, [], []
        for num in nums:
            if num % 3 == 0:
                res0 += num
            elif num % 3 == 1:
                res1.append(num)
            else:
                res2.append(num)
        res1.sort()
        res2.sort()
        if res % 3 == 1:
            if len(res1) > 0 and len(res2) > 1:
                return res - min(res1[0], res2[0]+res2[1])
            else:
                if len(res1) <= 0:
                    if len(res2) > 1:
                        return res - res2[0] - res2[1]
                elif len(res2) <= 1:
                    if len(res2) == 0:
                        return res - res1[0]
                    elif len(res2) == 1:
                        return res - res1[0]
        elif res % 3 == 2:
            if len(res1) > 1 and len(res2) > 0:
                return res - min(res1[0]+res1[1], res2[0])
            else:
                if len(res1) <= 1:
                    if len(res1) == 0:
                        return res - res2[0]
                    elif len(res1) == 1:
                        return res - res2[0]
                elif len(res2) <= 0:
                    return res - res1[0] - res1[1]
