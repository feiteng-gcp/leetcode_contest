class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        totalSum = sum(nums)
        if totalSum % 3 == 0:
            return totalSum
        else:
            ro = []
            rt = []
            
for i in nums:
                if i % 3 == 1:
                    ro.append(i)
                if i % 3 == 2:
                    rt.append(i)
            ro.sort()
            rt.sort()
            if totalSum % 3 == 1:
                if len(rt) >= 2 and len(ro) >= 1:
                    toSub = min(rt[0] + rt[1], ro[0])
                    return totalSum - toSub
                elif len(rt) < 2 and len(ro) < 1:
                    return 0
                elif len(rt) < 2 and len(ro) >= 1:
                    return totalSum - ro[0]
                else:
                    return totalSum - rt[0] - rt[1]
            else:
                if len(rt) >= 1 and len(ro) >= 2:
                    toSub = min(ro[0] + ro[1], rt[0])
                    return totalSum - toSub
                elif len(rt) < 1 and len(ro) < 2:
                    return 0
                elif len(ro) < 2 and len(rt) >= 1:
                    return totalSum - rt[0]
                else:
                    return totalSum - ro[0] - ro[1]
