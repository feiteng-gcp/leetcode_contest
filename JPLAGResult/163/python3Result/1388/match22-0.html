class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        res = sum(nums)
        mod = [[] for _ in range(3)]
        nums.sort()
        for num in nums:
            mod[num % 3].append(num)
        mn = float('inf')
        if res % 3 == 1:
            if mod[1]:
                mn = mod[1][0]
            if len(mod[2]) >= 2:
                mn = min(mn, mod[2][0] + mod[2][1])
            res -= mn
        elif res % 3 == 2:
            if len(mod[1]) >= 2:
                mn = sum(mod[1][:2])
            if mod[2]:
                mn = min(mn, mod[2][0])
            res -= mn
        return res if res % 3 == 0 else 0
