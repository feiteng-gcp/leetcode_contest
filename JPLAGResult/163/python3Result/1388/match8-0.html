class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        sumt = 0
        listn = [[],[],[]]
        for t in nums :
            sumt += t
            listn[t%3].append(t)
        for i in range(3):
            listn[i] = sorted(listn[i])
        
        return_list = [0]
        if sumt % 3 == 0 :
            return sumt
        if sumt % 3 == 1 :
            if len(listn[1]) > 0 :
                return_list.append(sumt-listn[1][0])
            if len(listn[2]) > 1 :
                return_list.append(sumt-listn[2][0]-listn[2][1])
        else :
            if len(listn[2]) > 0 :
                return_list.append(sumt-listn[2][0])
            if len(listn[1]) > 1 :
                return_list.append(sumt-listn[1][0]-listn[1][1])
        return max(return_list)
