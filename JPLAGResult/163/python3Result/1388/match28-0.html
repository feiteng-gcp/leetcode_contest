class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        s = sum(nums)
        
        if s%3 == 0:
            
            return s
        nums.sort()
        
        if s%3 == 1:
            a = None
            for item in nums:
                if item%3 == 1:
                    a = item
                    break
            b = None
            c = None
            for item in nums:
                if b == None and item%3 == 2:
                    b = item
                    continue
                if b != None and item%3 == 2:
                    c = item
                    break
            if b == None or c == None:
                s-=a
                return s
            elif a == None:
                s -=(b+c)
                return s
            else:
                res = min(a,b+c)
                s-=res
                return s
        else:
            a = None
            for item in nums:
                if item%3 == 2:
                    a = item
                    break
            b = None
            c = None
            for item in nums:
                if b == None and item%3 == 1:
                    b = item
                    continue
                if b != None and item%3 == 1:
                    c = item
                    break
            if b == None or c == None:
                s-=a
                return s
            elif a == None:
                s -=(b+c)
                return s
           
            else:
                #print(a,b,c,s)
                res = min(a,b+c)
                s-=res
                return s
