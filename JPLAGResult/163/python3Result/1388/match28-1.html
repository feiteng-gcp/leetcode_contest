class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        ans = sum(nums)
        if ans % 3 == 0: return ans
        nums.sort()
        # print(nums)
        # print("ANS: ", ans)
        
        if ans % 3 == 1:
            ANS = 0
            total = 0
            for n in nums:
                if n % 3 == 1:
                    ANS = max(ANS, ans - n)
                if n % 3 == 2:
                    total += n
                if total % 3 == 1:
                    ANS = max(ANS, ans - total)
            return ANS
                    
        if ans % 3 == 2:
            ANS = 0
            total = 0
            for n in nums:
                if n % 3 == 2:
                    ANS = max(ANS, ans - n)
                if n % 3 == 1:
                    total += n
                if total % 3 == 2:
                    ANS = max(ANS, ans - total)
            return ANS
