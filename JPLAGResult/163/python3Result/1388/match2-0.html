class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        l1, l2, l3 = [], [], []
        for i in nums:
            if i%3 == 0:
                l1 += [i]
            if i%3 == 1:
                l2 += [i]
            if i %3 == 2:
                l3 += [i]
        res = sum(l1)
        l2.sort(reverse = True)
        l3.sort(reverse = True)
        tmp = sum(l2) +sum(l3)
        if tmp% 3 == 1:
            remove = float('inf')
            if len(l2):
                remove = l2[-1]
            if len(l3) > 1:
                remove = min(l3[-1]+l3[-2],remove)
        if tmp % 3 == 2:
            remove = float('inf')
            if len(l3):
                remove = l3[-1]
            if len(l2) > 1:
                remove = min(l2[-1]+l2[-2],remove)
        if tmp % 3 == 0:
            remove = 0
        return res + tmp - remove
