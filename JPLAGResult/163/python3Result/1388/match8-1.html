from collections import defaultdict
class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        ret = 0
        for n in nums:
            ret += n
                
        d = defaultdict(list)
        for n in nums:
            d[n % 3].append(n)

        d[1] = sorted(d[1])
        d[2] = sorted(d[2])
        mod = 0
        for k, v in d.items():
            mod += k * len(v)
            mod %= 3
        if mod == 0:
            return ret
        
        options = []
        print(d)
        subs = []
        
if mod == 1:
            if len(d[1]) >= 1:
                subs.append(d[1][0])
            if len(d[2]) >= 2:
                subs.append(d[2][0] + d[2][1])
        elif mod == 2:
            
            if len(d[1]) >= 2:
                subs.append(d[1][0] + d[1][1])
            if len(d[2]) >= 1:
                subs.append(d[2][0])
        return ret - min(subs)
            
        return ret
