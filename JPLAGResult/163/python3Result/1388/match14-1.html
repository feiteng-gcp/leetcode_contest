class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        groups = [[], [], []]
        
for num in nums:
            groups[num % 3].append(num)
            
        for i in range(3):
            groups[i].sort()
            
        ans, s = 0, sum(nums)
        if s % 3 == 0:
            return s
        elif s % 3 == 1:
            if len(groups[1]) >= 1:
                ans = s - groups[1][0]
            if len(groups[2]) >= 2:
                ans = max(ans, s - sum(groups[2][:2]))
        else:
            if len(groups[1]) >= 2:
                ans = s - sum(groups[1][:2])
            if len(groups[2]) >= 1:
                ans = max(ans, s - groups[2][0])
        return ans
