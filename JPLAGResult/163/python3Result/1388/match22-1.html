class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        cands = {0:[], 1:[], 2:[]}
        s = sum(nums)
        r = s%3
        if r == 0:
            return s
        for n in nums:
            bisect.insort(cands[n%3], n)
        
        cand = float("inf")
        if len(cands[r]) > 0:
            cand = min(cand, cands
[r][0])
        if len(cands[3-r]) > 1:
            cand = min(cand, cands[3-r][0] + cands[3-r][1])
            
        return s - cand if cand < float("inf") else 0
