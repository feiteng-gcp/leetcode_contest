def calc_k2(k1, max_k2):
    for k2 in range(max_k2, max(max_k2 - 3, -1), -1):
        if (k1 + 2*k2) % 3 == 0:
            return k2
    raise ValueError


class Solution:
        
    def maxSumDivThree(self, nums: List[int]) -> int:
        zeros = [n for n in nums if n % 3 == 0]
        ones = [n for n in nums if n % 3 == 1]
        twos = [n for n in nums if n % 3 == 2]
        ones.sort(reverse=True)
        twos.sort(reverse=True)
        best = 0
        for k1 in range(len(ones), max(len(ones) - 3, -1), -1):
            try:
                k2 = calc_k2(k1, len(twos))
            except ValueError:
                continue
            # print(ones[:k1], twos[:k2], sum(ones[:k1]) + sum(twos[:k2]))
            best = max(best, sum(ones[:k1]) + sum(twos[:k2]))
        return sum(zeros) + best
