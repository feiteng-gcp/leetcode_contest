class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        m10 = float('inf')
        m11 = float('inf')
        m20 = float('inf')
        m21 = float('inf')
        sm = 0
        for i in nums:
            sm += i
            if i % 3 == 1:
                if i < m10:
                    m11, m10 = m10, i
                elif i < m11:
                    m11 = i
            if i % 3 == 2:
                if i < m20:
                    m21, m20 = m20, i
                elif i < m21:
                    m21 = i
        if sm % 3 == 0:
            return sm
        
        if sm % 3 == 1:
            if m10 != float('inf') and m20 != float('inf') and m21 != float('inf'):
                return sm - min(m10, m20 + m21)
            elif m10 != float('inf'):
                return sm - m10
            elif m20 != float('inf') and m21 != float('inf'):
                return sm - (m20 + m21)
            else:
                return 0
            
        if sm % 3 == 2:
            if m10 != float('inf') and m11 != float('inf') and m20 != float('inf'):
                return sm - min(m20, m10 + m11)
            elif m20 != float('inf'):
                return sm - m20
            elif m10 != float('inf') and m11 != float('inf'):
                return sm - (m10 + m11)
            else:
                return 0
