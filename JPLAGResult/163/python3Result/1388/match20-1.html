class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        b0, b1, b2 = [], [], []
        for v in nums:
            if v % 3 == 0:
                b0.append(v)
            elif v % 3 == 1:
                b1.append(v)
            else:
                b2.append(v)
        b1.sort(reverse = True)
        b2.sort(reverse = True)
        res = sum(b0) if b0 else 0
        l1, l2 = len(b1), len(b2)
        s1, s2 = sum(b1) if b1 else 0, sum(b2) if b2 else 0
        s = res + s1 + s2
        # s11 = sum(b1[-1:])
        # s12 = sum(b1[-2:])
        # s13 = sum(b1[-3:])
        # s21 = sum(b2[-1:])
        # s22 = sum(b2[-2:])
        # s23 = sum(b2[-3:])
        if l1 % 3 == l2 % 3:
            return s1 + s2 + res
        if s % 3 == 1:
            res1 = s - b1[-1] if b1 else -sys.maxsize
            res2 = s - sum(b2[-2:]) if len(b2) >= 2 else -sys.maxsize
            return max(res1, res2) if max(res1, res2) > -sys.maxsize + 2 else 0
        elif s % 3 == 2:
            res1 = s - sum(b1[-2:]) if len(b1) >= 2 else -sys.maxsize
            res2 = s - b2[-1] if b2 else -sys.maxsize
        return max(res1, res2) if max(res1, res2) > -sys.maxsize + 2 else 0
