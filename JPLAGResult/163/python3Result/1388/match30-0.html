class Solution:
    def maxSumDivThree(self, nums) -> int:
        n = len(nums)
        dp0 = [0] * (n)
        dp1 = [0] * (n)
        dp2 = [0] * (n)

        if nums[0] % 3 == 0: dp0[0] = nums[0]
        elif nums[0] % 3 == 1: dp1[0] = nums[0]
        elif nums[0] % 3 == 2: dp2[0] = nums[0]

        # print(dp0, dp1, dp2)

        for i in range(1, n):
            if nums[i] % 3 == 0:
                if dp0[i-1] != 0:
                    dp0[i] = max(dp0[i-1], dp0[i-1] + nums[i])
                else:
                    dp0[i] = max(dp0[i-1], nums[i])
                if dp1[i-1] != 0:
                    dp1[i] = max(dp1[i-1], dp1[i-1] + nums[i])
                else:
                    dp1[i] = dp1[i-1]
                if dp2[i-1] != 0:
                    dp2[i] = max(dp2[i-1], dp2[i-1] + nums[i])
                else:
                    dp2[i] = dp2[i-1]
            elif nums[i] % 3 == 1:
                if dp2[i-1] != 0:
                    dp0[i] = max(dp0[i-1], dp2[i-1] + nums[i])
                else:
                    dp0[i] = dp0[i-1]
                if dp0[i-1] != 0:
                    dp1[i] = max(dp1[i-1], dp0[i-1] + nums[i])
                else:
                    dp1[i] = max(dp1[i-1], nums[i])
                if dp1[i-1] != 0:
                    dp2[i] = max(dp2[i-1], dp1[i-1] + nums[i])
                else:
                    dp2[i] = dp2[i-1]
            elif nums[i] % 3 == 2:
                if dp1[i-1] != 0:
                    dp0[i] = max(dp0[i-1], dp1[i-1] + nums[i])
                else:
                    dp0[i] = dp0[i-1]
                if dp2[i-1] != 0:
                    dp1[i] = max(dp1[i-1], dp2[i-1] + nums[i])
                else:
                    dp1[i] = dp1[i-1]
                if dp0[i-1] != 0:
                    dp2[i] = max(dp2[i-1], dp0[i-1] + nums[i])
                else:
                    dp2[i] = max(dp2[i-1], nums[i])

        # print(dp0, dp1, dp2)
        return dp0[-1]
