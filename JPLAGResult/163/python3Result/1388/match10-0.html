class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        N = len(nums)
        dp = [[float('-inf')]+[0]*N for _ in range(3)]
        dp[0][0] = 0
        for i in range(N):
            x = nums[i]
            for j in range(3):
                dp[j][i+1] = max(dp[j][i], dp[(j-x)%3][i] + x)
        return dp[0][N]
