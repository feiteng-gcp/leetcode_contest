class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        best = [0, -1000000000, -1000000000]
        
        for x in nums:
            new_best = [0, -1000000000, -1000000000]

            new_best
[0] = max(best[0], best[x % 3] + x)
            new_best[1] = max(best[1], best[(x + 1) % 3] + x)
            new_best[2] = max(best[2], best[(x + 2) % 3] + x)
            
            best = new_best
            #print(best)
        return best[0]
