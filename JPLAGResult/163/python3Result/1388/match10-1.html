class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        n = len(nums)
        dp = [[-1] * 3 for i in range
(n + 1)]
        dp[0][0] = 0
        for i in range(n):
            k = nums[i]
            for j in range(3):
                dp[i+1][j] = dp[i][j]
            for j in range(3):
                if dp[i][j] < 0: continue
                dp[i+1][(j+k)%3] = max(dp[i][(j+k)%3], dp[i][j] + k)
        print(dp)
        return dp[n][0]
