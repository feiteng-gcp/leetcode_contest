class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        s = sum(nums)
        if s%3 == 0:
            return s
        ones, twos 
= [], []
        for x in nums:
            if x%3 == 1:
                ones.append(x)
            elif x%3 == 2:
                twos.append(x)
        ones.sort()
        twos.sort()
        minus = s
        if s%3 == 1:
            if ones:
                minus = min(minus, ones[0])
            if len(twos) >= 2:
                minus = min(minus, twos[0] + twos[1])
        elif s%3 == 2:
            if twos:
                minus = min(minus, twos[0])
            if len(ones) >= 2:
                minus = min(minus, ones[0] + ones[1])
        return s - minus
