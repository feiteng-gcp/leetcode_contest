class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        if sum(nums) % 3 == 0: return sum(nums)
        
        pre_sums = [float('-Inf'),float('-Inf'),float('-Inf')]
        
for n in nums:
            new_pre = [float('-Inf'),float('-Inf'),float('-Inf')]
            for i in range(len(pre_sums)):
                new_pre[(i + n % 3) % 3] = max(new_pre[(i + n % 3) % 3], pre_sums[i] + n)
                
            if new_pre[n % 3] < n: new_pre[n % 3] = n
            for i in range(3): new_pre[i] = max(new_pre[i], pre_sums[i])
            pre_sums = new_pre

        return pre_sums[0] if pre_sums[0] != float('-Inf') else 0
