# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class FindElements:

    def __init__(self, root: TreeNode):
        root.val = 0
        self.data = set([0])
        def dfs(node):
            if node.left:
                node.left.val = 2*node.val  +1
                self.data.add(node.left.val)
                dfs(node.left)
            if node.right:
                node.right.val = 2*node.val + 2
                self.data.add(node.right.val)
                dfs(node.right)
        dfs(root)
        

    def find(self, target: int) -> bool:
        if target in self.data:
            return True
        return False
