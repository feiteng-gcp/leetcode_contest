# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class FindElements:

    def __init__(self, root: TreeNode):
        
        self.v = set()
        
        def fix(root, val):
            if not root: return
            root.val = val
            self.v.add(val)
            root.left = fix(root.left, 2 * val + 1)
            root.right = fix(root.right, 2 * val + 2)
        
        fix(root, 0)
        self.t = root

    def find(self, target: int) -> bool:
        return target in self.v
