# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class FindElements:

    def __init__(self, root: TreeNode):
        self.s = set()
        
        def dfs(u: TreeNode):
            self.s.add(u.val)
            if u.left:
                u.left.val = u.val * 2 + 1
                dfs(u.left)
            if u.right:
                u.right.val = u.val * 2 + 2
                dfs(u.right)
        
        root.val = 0
        dfs(root)
        

    def find(self, target: int) -> bool:
        return target in self.s
