# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class FindElements:

    def __init__(self, root: TreeNode):
        self.val_set = set()
        def dfs(root, val):
            if root is None:
                return
            root.val = val
            self.val_set.add(val)
            dfs(root.left, 2 * val + 1)
            dfs(root.right, 2 * val + 2)
        dfs(root, 0) 

    def find(self, target: int) -> bool:
        return target in self.val_set
