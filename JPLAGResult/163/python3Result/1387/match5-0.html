# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class FindElements:

    def __init__(self, root: TreeNode):
        self.root = root
        self.bag = set()
        def build(rt: TreeNode, val: int):
            if not rt:
                return
            rt.val = val
            self.bag.add(val)
            build(rt.left, val * 2 + 1)
            build(rt.right, val* 2 + 2)
        build(root, 0)

    def find(self, target: int) -> bool:
        
        return target in self.bag
