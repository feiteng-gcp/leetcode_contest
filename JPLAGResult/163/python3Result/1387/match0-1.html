# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class FindElements:

    def __init__(self, root: TreeNode):
        self.seen = set()
        
        def recover(node, val):
            if not node: return
            node.val = val
            self.seen.add(val)
            recover(node.left, val * 2 + 1)
            recover(node.right, val * 2 + 2)
        
        recover(root, 0)

    def find(self, target: int) -> bool:
        return target in self.seen
