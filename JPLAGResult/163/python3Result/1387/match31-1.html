        self.values = set()
        def assign(node, x):
            if not node:
                return
            node.val = x
            self.values.add(x)
            assign(node.left, 2 * x + 1)
            assign
(node.right, 2 * x + 2)
        assign(root, 0)

    def find(self, target: int) -> bool:
