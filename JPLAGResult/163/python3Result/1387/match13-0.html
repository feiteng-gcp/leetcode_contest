# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class FindElements:

    def __init__(self, root: TreeNode):
        self.vals = set([0])
        self.recover(root, 0)
        
    def recover(self, node: TreeNode, val: int):
        if node is None:
            return
        
        node.val = val
        self.vals.add(val)
        self.recover(node.left, val * 2 + 1)
        self.recover(node.right, val * 2 + 2)

    def find(self, target: int) -> bool:
        return target in self.vals
