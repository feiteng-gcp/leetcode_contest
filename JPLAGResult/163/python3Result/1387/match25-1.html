# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class FindElements:

    def __init__(self, root: TreeNode):
        self.memo = set()
        def recover(node):
            if node.val == -1:
                node.val = 0
                self.memo.add(node.val)
            if node.left:
                node.left.val = node.val * 2 +1
                recover(node.left)
                self.memo.add(node.left.val)
            if node.right:
                node.right.val = node.val*2 + 2
                recover(node.right)
                self.memo.add(node.right.val)
        recover(root)

    def find(self, target: int) -> bool:
        return target in self.memo
