# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class FindElements:

    def __init__(self, root: TreeNode):
        self.set = set()
        def dfs(node,val):
            if node is None:
                return
            node.val = val
            self.set.add(val)
            dfs(node.left, 2*val+1)
            dfs(node.right, 2*val+2)
        dfs(root, 0)
        

    def find(self, target: int) -> bool:
        return target in self.set
