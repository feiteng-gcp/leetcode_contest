# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class FindElements:

    def __init__(self, root: TreeNode):
        self.vals = set()
        self.dfs(root, 0)

        
    def dfs(self, node, ur_val):
        if not node: return
        node.val = ur_val
        self.vals.add(ur_val)
        self.dfs(node.left, 2*ur_val+1)
        self.dfs(node.right, 2*ur_val+2)
    
    def find(self, target: int) -> bool:
        return target in self.vals
