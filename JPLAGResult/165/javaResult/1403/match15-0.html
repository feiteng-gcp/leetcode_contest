class Solution {
    public int palindromePartition(String s, int k) {
        char[] chars = s.toCharArray();
        int n = chars.length;
        if (n <= k) {
            return 0;
        }
        int[][][] f = new int[n][n][k+1];
        
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                f[i][j][1] = getCount(chars, i, j);
            }
        }
        
        for (int split = 2; split <= k; split++) {
            for (int i = 0; i < n; i++) {
                for (int j = i + split - 1; j < n; j++) {
                    int min = Integer.MAX_VALUE;
                    for (int l = i; l < j; l++) {
                        min = Math.min(min, f[i][l][1] + f[l+1][j][split-1]);
                    }
                    f[i][j][split] = min;
                }
            }
        }
        
        return f[0][n-1][k];
    }
    
    private int getCount(char[] chars, int start, int end) {
        int count = 0;
        while (start < end) {
            if (chars[start++] != chars[end--]) {
                count++;
            }
        }
        return count;
    }
}
