class Solution {
    public int palindromePartition(String s, int k) {
        int[][][] dp = new int[s.length()][s.length()][k + 1];
        for (int[][] dd : dp) {
            for (int[] ddd: dd) {
                Arrays.fill(ddd, -1);
            }
        }
        return find(s.toCharArray(), 0, s.length()-1, k, dp);
    }
    
    private int find(char[] s, int start, int end, int k, int[][][] dp) {
        //System.out.println("start="+start + " end="+end + " k="+k);
        
        if (start >= s.length || end >= s.length || start > end || k == 0) { return 0; }
        if (dp[start][end][k] != -1) { return dp[start][end][k]; }
        if (k == 1) {
            dp[start][end][k] = getDiff(s, start, end);
            return dp[start][end][k];
        }
        
        int min = s.length;
        for (int i=start; i<=s.length-k && i<=end; i++) {
            //System.out.println("[forloop] start="+start + " i="+i);
            min = Math.min(min, getDiff(s, start, i) + find(s, i+1, end, k-1, dp));
        }
        dp[start][end][k] = min;
        //System.out.println("min="+min);
        return min;
    }
    
    private int getDiff(char[] s, int start, int end) {
        int d = 0;
        //System.out.println("[getDiff] start=" + start + " end="+end + " d=" + d);
        while (start < end) {
            if (s[start++] != s[end--]) {
                d++;
            }
        }
        //System.out.println("[getDiff] d=" + d);
        return d;
    }
}
