class Solution {
  public int palindromePartition(String s, int k) {
    int n = s.length();
    char[] sc = s.toCharArray();
    int[][] dis = new int[n][n];
    for(int i = 0 ; i < n ; i++){
      for(int j = i; j < n ; j++){
        for(int p = i, q = j; p < q; p++, q--){
          if(sc[p] != sc[q]){
            dis[i][j]++;
          }
        }
      }
      //System.out.println(Arrays.toString(dis[i]));
    }
    int[][] dp = new int[n + 1][k + 1];
    for(int i = 1; i <= n ; i++){
      dp[i][1] = dis[0][i-1];
      for(int j = 2; j <= Math.min(k, i); j++){
        dp[i][j] = n + 1;
        for(int m = j - 1; m < i; m++){
          dp[i][j] = Math.min(dp[i][j], dp[m][j-1] + dis[m][i - 1]);
        }
      }
      //System.out.println(Arrays.toString(dp[i]));
    }
    return dp[n][k];
  }
}
