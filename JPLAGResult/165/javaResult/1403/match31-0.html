class Solution {
    /*
    --0. def: int[][][] dp: dp[i][j][k] represents min char to change to divide s[i : j] into k parts
    1. def: int[][] dp: dp[i][j] represents min char to change to divide s[0 : i] into j parts
    2. base case: dp[x][x + 1] = 0, dp[x][y] = len for x < y - 1
    3. induction rule: dp[i][j] = min(change(p, i) + dp[p - 1][j - 1]) for p in [j - 1, i]
    */
    public int palindromePartition(String s, int k) {
        int len = s.length();
        int[][] change = new int[len][len];
        for (int gap = 0; gap < len; gap++) {
            for (int i = 0; i + gap < len; i++) {
                int j = i + gap;
                if (gap == 0) {
                    change[i][j] = 0;
                } else if (gap == 1) {
                    change[i][j] = s.charAt(i) == s.charAt(j) ? 0 : 1;
                } else {
                    change[i][j] = (s.charAt(i) == s.charAt(j) ? 0 : 1) + change[i + 1][j - 1];
                }
            }
        }
        int[][] dp = new int[len][k + 1];
        for (int i = 0; i < len; i++) {
            dp[i][1] = change[0][i];
            for (int j = 2; j <= k; j++) {
                /*
                if (i + 1 < j) {
                    break; // --> len
                }*/
                if (i + 1 == j) {
                    dp[i][j] = 0;
                    continue;
                }
                dp[i][j] = len;
                for (int p = j - 1; p <= i; p++) {
                    dp[i][j] = Math.min(dp[i][j], dp[p - 1][j - 1] + change[p][i]);
                }
            }
        }
        return dp[len - 1][k];
    }
}
