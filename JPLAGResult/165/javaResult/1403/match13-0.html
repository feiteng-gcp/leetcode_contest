class Solution {
    Map<Integer, Integer> map = new HashMap<>();
    
    public int palindromePartition(String s, int k) {
        int n = s.length();
        int[][][] dp = new int[n][n][k + 1];
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                dp[i][j][1] = convP(s, i, j);
            }
        }
        for (int l = 2; l <= k; l++) {
            for (int i = 0; i < n; i++) {
                for (int j = i; j < n; j++) {
                    dp[i][j][l] = Integer.MAX_VALUE;
                    for (int p = i; p < j && (j - p) >= l - 1; p++) {
                        dp[i][j][l] = Math.min(dp[i][p][1] + dp[p + 1][j][l - 1], dp[i][j][l]);
                    }
                }
            }
        }
        
        return dp[0][n - 1][k];
    }
    
    
    
    private int convP(String s, int l, int r) {
        int res = 0;
        while (l < r) {
            if (s.charAt(l) != s.charAt(r)) {
                res++;
            }
            l++;
            r--;
        }
        return res;
    }
}
