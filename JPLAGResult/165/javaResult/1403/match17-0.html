class Solution {
    public int palindromePartition(String s, int k) {
        int n = s.length();
        
        int[][] a = new int[n+1][n+1];            
        for (int i=n-1; i>=0; i--) {
            a[i][i] = 0;
            char ch = s.charAt(i);
            for (int j=i+1; j<n; j++) {
                a[i][j] = (j == i+1 ? 0 : a[i+1][j-1]) + (s.charAt(j) == ch ? 0 : 1);                
            }
        }
        
        int[][] f = new int[n+1][k+1];
        for (int i=0; i<=n; i++)
            for (int j=0; j<=k; j++) f[i][j] = n+1;
        f[n][0] = 0;
        for (int i=n-1; i>=0; i--) {
            for (int j=1; j<=k; j++) {
                int best = n + 1;
                for (int l=i+1; l<=n; l++) {
                    int val = a[i][l-1] + f[l][j-1];
                    best = Math.min(best, val);
                }
                f[i][j] = best;
            }
        }
        return f[0][k];
    }
}
