class Solution {
    public int palindromePartition(String ss, int k) {
        char cs[] = ss.toCharArray(); 
        int l = ss.length();
        int s[][] = new int[l][l];
        
        for(int i=0;i<l;++i){
            for(int j=i;j<l;++j){
                int lo = i;
                int hi = j;
                int ct =0;
                while(lo<hi){
                    if(cs[lo]!=cs[hi]){
                        ct++;
                    }
                    lo++;
                    hi--;
                }
                s[i][j] = ct;
            }
        }
        
        int dp[][] = new int[l][k+1];
        
        for(int v[]:dp){
            Arrays.fill(v,100000000);
        }
        
        for(int i=0;i<l;++i){
            dp[i][1] = s[0][i];
            
            for(int f=2;f<=k;++f){
                
                for(int j=0;j<i;++j){
                    
                    dp[i][f] = Math.min(dp[i][f], dp[j][f-1] + s[j+1][i]);
                    
                }
                
            }
            
            
            
        }
        
        return dp[l-1][k];
        
        
        
    }
}
