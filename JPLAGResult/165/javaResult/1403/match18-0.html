class Solution {
    public int palindromePartition(String s, int k) {
        int len = s.length();
        
        int[][] dp = new int[len+1][len]; // dp[l][i] is the number of changes needed for s[i..i+l-1]
        
        for(int l=2;l<=len;l++) {
            for(int i=0;i+l-1<len;i++) {
                dp[l][i] = dp[l-2][i+1];
                if(s.charAt(i)!=s.charAt(i+l-1)) dp[l][i]++;
            }
        }
        
        int[][] opt = new int[len][k+1]; // opt[i][j] (j<=i+1) is the min number if split s[0..i] into j parts 
        
        // opt[0][1] = 0;
        
        for(int i=1;i<len;i++) {
            for(int j=1;j<=i+1 && j<=k;j++) {
                
                if(j==1) {opt[i][j] = dp[i+1][0]; System.out.println("i: "+i+", j: "+j+", min: "+opt[i][j]); continue;}
                
                int min = Integer.MAX_VALUE;
                for(int t=i;t>=j-1;t--) { // consider s[t..i] as the last part
                    
                    // System.out.println("i: "+i+", j: "+j+", t: "+t);
                    min = Math.min(min, opt[t-1][j-1]+dp[i-t+1][t]);
                }
                
                opt[i][j] = min;
                // System.out.println("i: "+i+", j: "+j+", min: "+min);
            }
        }
        
        return opt[len-1][k];
    }
}
