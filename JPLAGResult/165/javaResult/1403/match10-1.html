class Solution {
    public int palindromePartition(String s, int k) {
        int N = s.length();
        int[][] dp = new int[N + 1][k + 1];
        
        for (int i = 0; i <= N; i++) {
            Arrays.fill(dp[i], Integer.MAX_VALUE);
        }
        
        dp[0][0] = 0;
        
        
for (int i = 1; i <= N; i++) {
            // System.out.println("start " + s.substring(0, i));
            for (int j = 1; j <= k; j++) {
                
                for (int l = 1; l <= i; l++) {
                    int cor = check(s, i - l, i - 1);
                    
                    // System.out.println(s.substring(i - l, i) + " " + cor);
                    
                    if (dp[i - l][j - 1] != Integer.MAX_VALUE) {
                        dp[i][j] = Math.min(dp[i][j], cor + dp[i - l][j - 1]);
                    }
                    
                    // System.out.println(i + " " + j + " " + dp[i][j]);
                
}
            }
        }
        
//         for (int i = 0; i <= N; i++) {
//             System.out.println(Arrays.toString(dp[i]));
//         }
        
        return dp[N][k];
    }
    
    private int check(String s, int i, int j) {
        int res = 0;
        while (i < j) {
            if (s.charAt(i) != s.charAt(j)) {
                res++;
            }
            i++;
            j--;
        }
        return res;
    }
}
