class Solution {
    public int palindromePartition(String s, int k) {
        int N = s.length();
        int[][] table = new int[N][N];
        for (int l = 2; l <= N; l++) {
            for (int start = 0; start + l <= N; start++) {
                int end = start + l - 1;
                if (s.charAt(start) == s.charAt(end)) {
                    if (l == 2) {
                        table[start][end] = 0;
                    } else {
                        table[start][end] = table[start + 1][end - 1];
                    }
                } else {
                    if (l == 2) {
                        table[start][end] = 1;
                    } else {
                        table[start][end] = table[start + 1][end - 1] + 1;
                    }
                }
            }
        }
        int[][] dp = new int[k + 1][N];
        for (int i = 0; i < N; i++) {
            dp[1][i] = table[0][i];
        }
        for (int i = 2; i <= k; i++) {
            for (int j = i - 1; j < N; j++) {
                dp[i][j] = j + 1;
                for (int m = i - 2; m < j; m++) {
                    dp[i][j] = Integer.min(dp[i][j], dp[i - 1][m] + table[m + 1][j]);
                }
            }
        }
        return dp[k][N - 1];
    }
}
