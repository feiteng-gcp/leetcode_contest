class Solution {
    public int palindromePartition(String s, int k) {
        int N = s.length();
        int[][] changes = new int[N][N];
        for (int i = 0; i < N; i++) {
            for (int j = i + 1; j < N; j++) {
                changes[i][j] = 
changesToPalindrome(s, i, j);
            }
        }
        int[][] dp = new int[k][N];
        for (int i = 0; i < N; i++) {
            dp[0][i] = changes[0][i];
        }
        for (int i = 1; i < k; i++) {
            for (int j = i; j < N; j++) {
                dp[i][j] = Integer.MAX_VALUE;
                for (int l = i - 1; l < j; l++) {
                    dp[i][j] = Math.min(dp[i][j], dp[i - 1][l] + changes[l + 1][j]);
                }
            }
        }
        return dp[k - 1][N - 1];
    }

    private int changesToPalindrome(String s, int lower, int upper) {
        int changes = 0;
        while (lower < upper) {
            if (s.charAt(lower) != s.charAt(upper)) {
                changes++;
            }
            lower++;
            upper--;
        }
        return changes;
    }
}
