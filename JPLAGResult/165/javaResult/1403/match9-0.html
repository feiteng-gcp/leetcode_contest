class Solution {
    public int palindromePartition(String s, int k) {
        int n = s.length();
        if (k >= n) return 0;
        
        int[][] f = new int[n + 1][k + 1];
        int[][] g = new int[n][n];
        
        for (int i = n - 1; i >= 0; i--) {
            for (int j = i; j < n; j++) {
                if (i == j) {
                    g[i][j] = 0;
                } else {
                    if (i == j - 1) {
                        g[i][j] = s.charAt(i) == s.charAt(j) ? 0 : 1;
                    } else {
                        g[i][j] = s.charAt(i) == s.charAt(j) ? 0 : 1;
                        g[i][j] += g[i + 1][j - 1];
                    }
                }
            }
        }
        
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= k; j++) {
                if (i == 0 && j == 0) {
                    continue;
                }
                if (i == 0) {
                    f[i][j] = -1;
                    continue;
                }
                if (j == 0) {
                    f[i][j] = -1;
                    continue;
                }
                f[i][j] = Integer.MAX_VALUE;
                for (int p = 0; p < i; p++) {
                    if (f[p][j - 1] >= 0) {
                        f[i][j] = Math.min(f[p][j - 1] + g[p][i - 1], f[i][j]);
                    } else {
                         continue;
                    }
                }
                if (f[i][j] == Integer.MAX_VALUE) {
                    f[i][j] = -1;
                }
            }
        }
        return f[n][k];
    }
}
