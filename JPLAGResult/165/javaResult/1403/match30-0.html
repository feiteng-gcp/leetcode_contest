class Solution {
    public int palindromePartition(String s, int k) {
        int n = s.length();
        int[][] pc = new int[n][n];
        for (int z = 1; z < n; ++z) {
            for (int i = 0; i+z<n; ++i) {
                int j = i+z;
                for (int x = 0; x < (z+1)/2; ++x) {
                    if (s.charAt(i+x)!=s.charAt(j-x))
                        ++pc[i][j];
                }
            }
        }
        int[][] dp = new int[k][n];
        for (int i = 0; i < n; ++i) {
            dp[0][i] = pc[0][i];
        }
        for (int i = 1; i < k; ++i) {
            for (int j = 0; j < n; ++j) {
                dp[i][j] = 1000000;
                for (int a = 0; a < j; ++a) {
                    dp[i][j] = Math.min(dp[i][j], dp[i-1][a]+pc[a+1][j]);
                }
            }
        }
        return dp[k-1][n-1];
    }
}
