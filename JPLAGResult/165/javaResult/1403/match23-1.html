class Solution {
    
public int palindromePartition(String s, int k) {
        int n = s.length();
        int[][] mc = new int[n + 1][n + 1];
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                mc[i][j] = minChange(s.substring(i, j + 1), j - i + 1);
            
}
        }
        int[][] dp = new int[n + 1][k + 1];
        for (int i = 0; i < n; i++) {
            dp[i][1] = mc[0][i];
        }
        for (int j = 2; j <= k; j++) {
            for (int i = 1; i < n; i++) {
                
dp[i][j] = 214748364;
                for (int r = 0; r < i; r++) {
                    dp[i][j] = Math.min(dp[i][j], dp[r][j - 1] + mc[r + 1][i]);
                }
            }
        }
        return dp[n - 1][k];
    }
    public int minChange(
String s, int len) {
        int ans = 0;
        for (int i = 0; i < len / 2; i++) {
            if (s.charAt(i) != s.charAt(len - i - 1)) {
                ans++;
            }
        }
        return ans;
    }
}
