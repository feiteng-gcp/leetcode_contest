class Solution {

    private final int INF = 0x3f3f3f3f;

    private int getDiff(String s, int l, int r) {
        int ans = 0;
        for (int i = 0; l + i <= r - i; ++i) {
            if (s.charAt(l + i) != s.charAt(r - i)) ans += 1;
        }
        return ans;
    }

    public int palindromePartition(String s, int k) {
        int N = s.length();
        int[][] chg = new int[N + 1][N + 1];
        for (int i = 0; i < N; ++i) {
            for (int j = i; j < N; ++j) {
                chg[i + 1][j + 1] = getDiff(s, i, j);
                // System.out.println("i = " + (i+1) + ", j = " + (j+1) + ", chg = " + chg[i+1][j+1]);
            }
        }

        int[][] dp = new int[N + 1][N + 1];
        for (int i = 1; i <= N; ++i) {
            for (int j = 0; j <= N; ++j) {
                if (j > i || j == 0) {
                    dp[i][j] = INF;
                    continue;
                }
                if (j == 1) {
                    dp[i][j] = chg[1][i];
                    continue;
                }
                dp[i][j] = INF;
                for (int t = 1; t <= i - j + 1; ++t) {
                    dp[i][j] = Math.min(dp[i][j], dp[i - t][j - 1] + chg[i - t + 1][i]);
                }
            }
            // System.out.println("dp[" + i + "] = " + Arrays.toString(dp[i]));
        }

        return dp[N][k];
    }
}
