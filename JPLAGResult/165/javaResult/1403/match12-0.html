class Solution {
    public int palindromePartition(String s, int k) {
        int len = s.length();
        int[][] dp = new int[len + 1][k + 1];
        for (int[] ints : dp) {
            Arrays.fill(ints, 100);
        }
        dp[1][1] = 0;
        dp[0][0] = 0;
        for (int i = 2; i <= len; i++) {
            for (int j = 0; j < i; j++) {
                for (int l = 1; l <= k; l++) {
                    int tmp = helper(s, i - 1, j);
                    dp[i][l] = Math.min(dp[i][l], dp[j][l - 1] + tmp);
                }
            }
        }
        return dp[len][k];
    }

    private int helper(String s, int i, int j) {
        int count = 0;
        while (j < i) {
            if (s.charAt(i--) != s.charAt(j++)) {
                count++;
            }
        }
        return count;
    }
}
