class Solution {
    public int palindromePartition(String s, int k) {
        int n = s.length();
        //number of change need to change s[i][j] to pali
        int[][] change = new int[n][n];
        
        for (int gap = 1; gap < n; gap++){
            for (int i = 0; i+gap < n; i++) {
                int j = i+gap;
                if (s.charAt(i) == s.charAt(j)) {
                    if (gap == 1)
                        change[i][j] = 0;
                    else
                        change[i][j] = change[i+1][j-1];
                }
                else {
                    if (gap == 1)
                        change[i][j] = 1;
                    else
                        change[i][j] = change[i+1][j-1] + 1;
                }
                
            }
        }
        
        //dp[i][j] -> until s[i], split into j part, how many
        int[][] dp = new int[n][k+1];
        for (int i = 0; i < n; i++) {
            Arrays.fill(dp[i], Integer.MAX_VALUE/2);
        }
        //dp[i][j] = dp[h][j-1] + change[h+1][i]
        
        //dp[i][1] -> 1 part -> change[0][i]
        
for (int i = 0; i < n; i++) {
            dp[i][1] = change[0][i];
        }
        
        for (int j = 2; j <= k; j++) {
            for (int i = 0; i < n; i++) {
                //k start from 0. start from dp[0][j'].
                //since there must be j' part from beginning -> no need to start k from -1
                
for (int h = 0; h+1 <= i; h++) {
                    dp[i][j] = Math.min(dp[i][j], dp[h][j-1] + change[h+1][i]);
                }
            }
        }
        
        return dp[n-1][k];
        
        
        
    }
}
