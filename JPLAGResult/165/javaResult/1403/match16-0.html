// dp[k + 1][i][j] => the minimum number of changes needed to convert s[i : j] to a k part palindrome

class Solution {
    public int palindromePartition(String s, int K) {
        int n = s.length();
        
        // Define dp
        int[][][] dp = new int[K + 1][n][n];
        
        // Init dp
        for (int l = 2; l <= n; l++) {
            for (int i = 0; i <= n - l; i++) {
                int j = i + l - 1;
                dp[1][i][j] = countChanges(s, i, j);
            }
        }
        
        // Build dp
        for (int k = 2; k <= K; k++) {
            for (int l = k + 1; l <= n; l++) {
                for (int i = 0; i <= n - l; i++) {
                    int j = i + l - 1;
                    dp[k][i][j] = Integer.MAX_VALUE / 2;
                    for (int l1 = 1; l1 < l; l1++) {
                        dp[k][i][j] = Math.min(dp[k][i][j], dp[k - 1][i][i + l1 - 1] + countChanges(s, i + l1, j));
                    }
                }
            }
        }
        
        return dp[K][0][n - 1];
    }
    
    public int countChanges(String s, int i, int j) {
        int cnt = 0, lo = i, hi = j;
        while (lo < hi) {
            if (s.charAt(lo) != s.charAt(hi))
                cnt++;
            lo++;    hi--;
        }
        return cnt;
    }
}
