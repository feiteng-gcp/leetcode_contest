class Solution {
    public int palindromePartition(String s, int group) {
        int n = s.length();
        int[] dp = new int[n];
        int[][] firstDp = new int[n][n];
        for(int i = 0; i < n; i++){
            dp[i] = checkRange(s, 0, i);
        }
        
        
        
for(int i = 0; i < n; i++){
            for(int j = i; j < n; j++){
                firstDp[i][j] = checkRange(s, i, j);
            }
        }
        
        
        for(int i = 2; i <= group; i++){
            int[] new_dp = new int[n];
            
for(int j = i - 1; j < n; j++){
                int res = Integer.MAX_VALUE;
                for(int k = i - 1; k <= j; k++){
                    res = Math.min(res, dp[k - 1] + firstDp[k][j]);
                }
                new_dp[j] = res;
            }
            
            dp = new_dp;
        
}
        
        return dp[n - 1];
    }
    
    private int checkRange(String s, int l, int r){
        int res = 0;
        while(l < r){
            if(s.charAt(l) != s.charAt(r))
{
                res++;
            }
            l++;
            r--;
        }
        
        return res;
    }
}
