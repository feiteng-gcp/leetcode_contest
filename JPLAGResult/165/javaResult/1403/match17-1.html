class Solution {
    public int palindromePartition(String s, int k) {
        int n = s.length();
        int[][] changes = new int[n][n];
        for (int len = 1; len < n; len++) {
            for (int i = 0; i + len < n; i++) {
                changes[i][i + len] = changes[i + 1][i + len - 1] + (s.charAt(i) == 
s.charAt(i + len) ? 0 : 1);
            }
        }
        int[][] partitions = new int[n][k + 1];
        for (int i = 0; i < n; i++) {
            partitions[i][1] = changes[0][i];
        }
        
for (int part = 2; part <= k; part++) {
            // System.out.println(part);
            for (int i = part - 1; i < n; i++) {
                int minChanges = Integer.MAX_VALUE;
                for (int j = i - 1; j >= part - 2; j--) {
                    // System.out.println(j + "->" + i + "   |   " + changes[j + 1][i] + "+" + partitions[j][part - 1]); 
                    
minChanges = Math.min(minChanges, changes[j + 1][i] + partitions[j][part - 1]);
                }
                partitions[i][part] = minChanges;
            }
        }
        return partitions[n - 1][k];
    }
}
