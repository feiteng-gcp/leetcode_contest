class Solution {
        public List<Integer> numOfBurgers(int tomatoSlices, int cheeseSlices) {
            List<Integer> ret = new ArrayList<>();
            int a = tomatoSlices - 2 * cheeseSlices;
            if (a % 2 != 0) return ret;
            a/=2;
            int b = cheeseSlices - a;
            if (a < 0 || b < 0) return ret;
            ret.add(a); ret.add(b);
            return ret;
        }
    }
