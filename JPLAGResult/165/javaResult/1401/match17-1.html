class Solution {
    public List<Integer> numOfBurgers(int t, int c) {
        List<Integer> res = new ArrayList<>();
        if ((t-2*c) % 2 != 0) return res;
        int a = (t-2*c) / 2;
        if (a < 0) return res;
        int b = c - a;
        if (b < 0) return res;
        res.add(a);
        res.add(b);
        return res;
    }
}
