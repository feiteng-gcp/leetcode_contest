class Solution {
    public List<Integer> numOfBurgers(int t, int c) {
        int v =t-2*c;
         List<Integer> li = new ArrayList<>();
        if(v>=0&&v%2==0){
            
            int a = v/2;
            int b = c-a;
            if(b>=0){
               
                li.add(a);
                li.add(b);
                return li;
            }
            
        }
        return li;
        
    }
}
