class Solution {
    public List<Integer> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        //ji tu tong long
        // x = (A-2B)/2
        // y = (4B - A)/2
        
        List<Integer> ans = new ArrayList<>();
        
        int A = tomatoSlices;
        int B = cheeseSlices;
        
        if ((A-2*B)%2 != 0 || (4*B - A)%2 != 0)
            return ans;
        
        int x = (A-2*B)/2;
        int y = (4*B - A)/2;
        
if (x < 0 || y < 0)
            return ans;
        ans.add(x);
        ans.add(y);
        return ans;
    }
}
