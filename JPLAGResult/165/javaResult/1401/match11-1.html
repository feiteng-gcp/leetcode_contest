class Solution {
    public List<Integer> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        List<Integer> res = new LinkedList<>();
        int a = tomatoSlices - 2 * cheeseSlices;
        if (a < 0 || a % 2 != 0) return res;
        int x = a / 2;
        int y = cheeseSlices - x;
        if (y < 0) return res;
        res.add(x);
        res.add(y);
        return res;
    }
}
