class Solution {
    public List<Integer> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        List<Integer> result=new ArrayList<>();
        int x2=tomatoSlices- 2 *cheeseSlices;
        if (x2<0 || x2 % 2==1){
            return result;
        }
        
if (cheeseSlices- x2/2<0){
            return result;
        }
        result.add(x2/2);
        result.add(cheeseSlices-x2/2);
        return result;
    }
}
