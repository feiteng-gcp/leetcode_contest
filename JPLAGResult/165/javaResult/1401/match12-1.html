class Solution {
    public List<Integer> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        List<Integer> ans = new ArrayList<>();
        int a = tomatoSlices - cheeseSlices * 2;
        if (a % 2 != 0)
            return ans;
        int x = a / 2;
        int y = cheeseSlices - x;
        
if (x < 0 || y < 0)
            return ans;
        ans.add(x);
        ans.add(y);
        return ans;
    }
}
