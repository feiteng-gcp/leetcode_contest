class Solution {
    public List<Integer> numOfBurgers(int ts, int cs) {
        List<Integer> ans = new ArrayList<>();
        if (ts%2!=0)
            return ans;
        
if (ts<cs*2||ts>cs*4)
            return ans;
        ts/=2;
        ans.add(ts-cs);
        ans.add(2*cs-ts);
        return ans;
    }
}
