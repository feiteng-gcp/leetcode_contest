class Solution {
    public List<Integer> numOfBurgers(int t, int c) {
        List<Integer> r = new ArrayList<>();
        int m = t - 2 * c;
        if (m < 0 || m % 2 != 0) {
            return r;
        }
        int jumbo = m / 2;
        int small = c - jumbo;
        if (small < 0) {
            return r;
        }
        r.add(jumbo);
        r.add(small);
        return r;
    }
}
