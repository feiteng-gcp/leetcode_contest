class Solution {
    public List<Integer> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        List<Integer> res = new ArrayList<>();
        if((tomatoSlices & 1) == 1){
            return res;
        }
        tomatoSlices >>= 1;
        
if(tomatoSlices < cheeseSlices || (cheeseSlices << 1) - tomatoSlices < 0){
            return res;
        }
        //System.out.println(tomatoSlices + " " + cheeseSlices);
        res.add(tomatoSlices - cheeseSlices);
        res.add((cheeseSlices << 1) - tomatoSlices);
        return res;
    }
}
