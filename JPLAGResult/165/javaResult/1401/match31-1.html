class Solution {
        public List<Integer> numOfBurgers(int tomatoSlices, int cheeseSlices) {
            List<Integer> res = new ArrayList<>();
            if(tomatoSlices < 2 * cheeseSlices || tomatoSlices > 4 * cheeseSlices) return res;
            int v1 = tomatoSlices - 2 * cheeseSlices, v2 = 4 * cheeseSlices - tomatoSlices;
            
if(v1 % 2 != 0 || v2 % 2 != 0) return res;
            res.add(v1 / 2);
            res.add(v2 / 2);
            return res;

        }
    }
