class Solution {
    public List<Integer> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        // 4x + 2y = tomato
        // x + y = cheese
        List<Integer> list = new ArrayList<>();
        int t = tomatoSlices, c = cheeseSlices;
        if (t < c + c || (t - c - c) % 2 != 0) return list;
        int x = (t - c - c) / 2;
        int y = c - x;
        if  (y < 0) return list;
        list.add(x);
        list.add(y);
        return list;
    }
}
