class Solution {
    public List<Integer> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        
        List<Integer> ans = new ArrayList<>();
        if(tomatoSlices % 2 == 1) return ans;
        
        int diff = tomatoSlices - 2 * cheeseSlices;
        if(diff < 0 || diff % 2 == 1) return ans;
        
        diff /= 2;
        
        int diff2 = cheeseSlices - diff;
        if(diff2 < 0) return ans;
        
        ans.add(diff);
        ans.add(diff2);
        
        return ans;
        
    }
}
