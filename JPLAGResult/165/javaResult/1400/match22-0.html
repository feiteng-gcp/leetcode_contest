class Solution {
    public String tictactoe(int[][] moves) {
        int step = 0;
        int player = 1;
        int[][] board = new int[3][3];
        for(int[] move: moves){
            board[move[0]][move[1]] = player;
            int wins = win(board);
            if(wins != 0) return wins == 1 ? "A" : "B";
            player = -player;
            step++;
        }
        return step == 9 ? "Draw" : "Pending";
    }
    
    
    private int win(int[][] board){
        for(int i = 0; i < 3; i++){
            if(board[i][0] != 0 && board[i][0] == board[i][1] && board[i][1] == board[i][2]) return board[i][0];
            if(board[0][i] != 0 && board[0][i] == board[1][i] && board[1][i] == board[2][i]) return board[0][i];
        }
        if(board[0][0] != 0 && board[0][0] == board[1][1] && board[1][1] == board[2][2]) return board[0][0];
        if(board[0][2] != 0 && board[0][2] == board[1][1] && board[1][1] == board[2][0]) return board[1][1];
        return 0;
    }
}
