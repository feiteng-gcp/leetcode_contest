class Solution {
    public String tictactoe(int[][] moves) {
        int[][] grid = new int[3][3];
        for(int i = 0 ; i < moves.length; i+=2){
            grid[moves[i][0]][moves[i][1]] = 1;
            if(i + 1 < moves.length){
                grid[moves[i + 1][0]][moves[i + 1][1]] = 2;
            }
            int res = judge(grid);
            if(res == 1){
                return "A";
            } else if(res == 2){
                return "B";
            }
        }
        return moves.length == 9 ? "Draw": "Pending";
    }
    
    private int judge(int[][] grid){
        for(int i = 0 ; i < 3; i++){
            if(grid[i][0] != 0 && grid[i][0] == grid[i][1] && grid[i][0] == grid[i][2]){
                return grid[i][0];
            }
            if(grid[0][i] != 0 && grid[0][i] == grid[1][i] && grid[0][i] == grid[2][i]){
                return grid[0][i];
            }
        }
        if(grid[0][0] != 0 && grid[1][1] == grid[0][0] && grid[2][2] == grid[0][0]){
            return grid[0][0];
        }
        if(grid[2][0] != 0 && grid[1][1] == grid[2][0] && grid[0][2] == grid[2][0]){
            return grid[2][0];
        }
        return 0;
    }
}
