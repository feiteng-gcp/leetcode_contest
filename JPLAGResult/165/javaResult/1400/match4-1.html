class Solution {
    public String tictactoe(int[][] moves) {
        int[][] grids = new int[3][3];
        int m = 1;
        for (int[] move : moves) {
            grids[move[0]][move[1]] = m;
            if (checkWin(grids, m)) {
                return m == 1 ? "A" : "B";
            }
            m = -m;
        }
        if (moves.length == 9) {
            return "Draw";
        }
        return "Pending";
    
}

    private boolean checkWin(int[][] grids, int value) {
        for (int i = 0; i < 3; i++) {
            boolean success = true;
            for (int j = 0; j < 3; j++) {
                if (grids[i][j] != value) {
                    success = false;
                    break;
                }
            }
            if (success) {
                return true;
            }
        }

        for (int i = 0; i < 3; i++) {
            boolean success = true;
            for (int j = 0; j < 3; j++) {
                if (grids[j][i] != value) {
                    success = false;
                    break;
                }
            }
            if (success) {
                return true;
            }
        }

        boolean success = true;
        for (int i = 0; i < 3; i++) {
            if (grids[i][i] != value) {
                success = false;
                break;
            }
        }

        if (success) {
            return true;
        }

        success = true;
        for (int i = 0; i < 3; i++) {
            if (grids[i][2 - i] != value) {
                success = false;
                break;
            }

        }
        return success;
    }

}
