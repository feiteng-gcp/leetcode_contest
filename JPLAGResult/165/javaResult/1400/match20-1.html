class Solution {
    public String tictactoe(int[][] moves) {
        int[] rowO = new int[3], rowX = new int[3], colO = new int[3], colX = new int[3];
        int digO = 0, digX = 0, antiDigO = 0, antiDigX = 0, total = 0;
        for (int i = 0; i < moves.length; i++) {
            int[] cur = moves[i];
            
if (i % 2 == 0) {
                if (++rowX[cur[0]] == 3) return "A";
                if (++colX[cur[1]] == 3) return "A";
                if (cur[0] == cur[1]) {
                    if (++digX == 3) return "A";
                }
                if (cur[0] + cur[1] == 2) {
                    if (++antiDigX == 3) return "A";
                }
            } else {
                if (++rowO[cur[0]] == 3) return "B";
                if (++colO[cur[1]] == 3) return "B";
                if (cur[0] == cur[1]) {
                    if (++digO == 3) return "B";
                }
                if (cur[0] + cur[1] == 2) {
                    if (++antiDigO == 3) return "B";
                }
            }
        }
        
        if (moves.length == 9) return "Draw";
        return "Pending";
    }
}
