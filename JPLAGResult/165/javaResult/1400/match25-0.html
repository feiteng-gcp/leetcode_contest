class Solution {
    public String tictactoe(int[][] moves) {
    char[][] board = new char[3][3];
    boolean turnA = true;
    for (int [] move: moves) {
      board[move[0]][move[1]] = turnA? 'X' : 'O';
        turnA = !turnA;
      if (check5275(board, 'X')) return "A";
      if (check5275(board, 'O')) return "B";
    }
    boolean full = true;
    for (int i = 0; i < 3; i++) {
      for (int j = 0; j < 3; j++) {
          if (board[i][j] != 'X' && board[i][j] != 'O') {
            full = false;
            break;
          }
      }
      if (!full) break;
    }
      return full? "Draw" : "Pending";
  }

  static boolean check5275(char[][] B, char c) {
    for (int i = 0; i < 3; i++) {
      if (B[i][0] == c && B[i][1] == c && B[i][2] == c) return true;
      if (B[0][i] == c && B[1][i] == c && B[2][i] == c) return true;
    }
    if (B[0][0] == c && B[1][1] == c && B[2][2] == c) return true;
    if (B[0][2] == c && B[1][1] == c && B[2][0] == c) return true;
    return false;
  }
}
