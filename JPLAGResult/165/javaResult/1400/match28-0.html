class Solution {
	int board[][];
	
    public String tictactoe(int[][] moves) {
        board = new int[3][3];
        for (int i = 0 ; i < 3 ; i++)
        	board[i] = new int[3];
        
        for (int i = 0 ; i < moves.length ; i++)
        {
        	int r = moves[i][0];
        	int c = moves[i][1];
        	
        	if (i % 2 == 0)
        		board[r][c] = 2;
        	else
        		board[r][c] = 1;
        }
        
        if (hasWinner(1))
        	return "B";

        if (hasWinner(2))
        	return "A";
        
        if (moves.length == 9)
        	return "Draw";
        
        return "Pending";
    }

	private boolean hasWinner(int a) {
		for (int r = 0 ; r < 3 ; r++)
			if ((board[r][0] == a) && (board[r][1] == a) && (board[r][2] == a))
				return true;

		for (int c = 0 ; c < 3 ; c++)
			if ((board[0][c] == a) && (board[1][c] == a) && (board[2][c] == a))
				return true;
		
		if ((board[0][0] == a) && (board[1][1] == a) && (board[2][2] == a))
			return true;
		
		if ((board[2][0] == a) && (board[1][1] == a) && (board[0][2] == a))
			return true;
		
		return false;
	}
}
