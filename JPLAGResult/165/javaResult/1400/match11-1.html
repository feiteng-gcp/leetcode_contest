class Solution {
    public String tictactoe(int[][] moves) {
        int[][] grid = new int[3][3];
        int now = 1;
        for(int[] move: moves) {
            grid[move[0]][move[1]] = now;
            if(isWin(grid))
                return now == 1 ? "A" : "B";
            now = 3 - now;
        }
        if(moves.length == 9)
            return "Draw";
        return "Pending";
    
}
    public boolean isWin(int[][] grid) {
        for(int i = 0; i < 3; ++i) {
            if(grid[i][0] == grid[i][1] && grid[i][1] == grid[i][2] && grid[i][1] != 0)
                return true;
            if(grid[0][i] == grid[1][i] && grid[1][i] == grid[2][i] && grid[1][i] != 0)
                return true;
        }
        if(grid[0][0] == grid[1][1] && grid[1][1] == grid[2][2] && grid[1][1] != 0)
            return true;
        if(grid[0][2] == grid[1][1] && grid[1][1] == grid[2][0] && grid[1][1] != 0)
            return true;
        return false;
    }
}
