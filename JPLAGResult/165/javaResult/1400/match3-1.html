class Solution {
    public String tictactoe(int[][] moves) {
        int[][] table = new int[3][3];
        
for (int i = 0; i < moves.length; i++) {
            int[] m = moves[i];
            int j = (i % 2) + 1;
            table[m[0]][m[1]] = j;
            if (win(table, j)) {
                if (j == 1) {
                    return "A";
                } else {
                    return "B";
                }
            }
        
}
        if (moves.length == 9) {
            return "Draw";
        } else {
            return "Pending";
        }
    }

    boolean win(int[][] table, int p) {
        for (int i = 0; i < 3; i++) {
            boolean f = true;
            for (int j = 0; j < 3; j++) {
                if (table[i][j] != p) {
                    f = false;
                    break;
                }
            }
            if (f) {
                return true;
            }
        }
        for (int i = 0; i < 3; i++) {
            boolean f = true;
            for (int j = 0; j < 3; j++) {
                if (table[j][i] != p) {
                    f = false;
                    break;
                }
            }
            if (f) {
                return true;
            }
        }
        boolean f = true;
        for (int i = 0; i < 3; i++) {
            if (table[i][i] != p) {
                f = false;
                break;
            }
        }
        if (f) {
            return true;
        }
        f = true;
        for (int i = 0; i < 3; i++) {
            if (table[2 - i][i] != p) {
                f = false;
                break;
            }
        }
        if (f) {
            return true;
        }
        return false;
    }
}
