class Solution {
    int[][] table = new int[3][3];

	public 
String tictactoe(int[][] moves) {
		for (int i = 0; i < moves.length; i++) {
			table[moves[i][0]][moves[i][1]] = 1 + i % 2;
			if (check(1 + i % 2)) {
				if (i % 2 == 0) return "A";
				else return "B";
			}

		}
		if (moves.length < 9) return "Pending";
		return "Draw";
	}

	private boolean check(int k) {
		for (int i = 0; i < 3; i++) {
			boolean ok = true;
			for (int j = 0; j < 3; j++) {
				if (table[i][j] != k) ok = false;
			}
			if (ok) return true;
		}
		for (int i = 0; i < 3; i++) {
			boolean ok = true;
			for (int j = 0; j < 3; j++) {
				if (table[j][i] != k) ok = false;
			}
			if (ok) return true;
		}
		if (table[0][0] == k && table[1][1] == k && table[2][2] == k) return true;
		if (table[0][2] == k && table[1][1] == k && table[2][0] == k) return true;
		return false;
	}
}
