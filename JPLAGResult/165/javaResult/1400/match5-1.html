class Solution {
    public String tictactoe(int[][] moves) {
        int[][] board = new int[3][3];
        for (int i = 0; i < moves.length; i++) {
            char c = (i % 2 == 0) ? 'X' : 'O';
            board[moves[i][0]][moves[i][1]] = c;
        }
        if (check('X', board)) {
            return "A";
        }
        if (check('O', board)) {
            return "B";
        }
        return (moves.length == 9) ? "Draw" : "Pending";
    
}

    private boolean check(char c, int[][] board) {
        for (int i = 0; i < 3; i++) {
            boolean win = true;
            for (int j = 0; j < 3; j++) {
                if (board[i][j] != c) {
                    win = false;
                
}
            }
            if (win) {
                return true;
            }
        }

        for (int j = 0; j < 3; j++) {
            boolean win = true;
            for (int i = 0; i < 3; i++) {
                if (board[i][j] != c) {
                    win = false;
                
}
            }
            if (win) {
                return true;
            }
        }

        return (board[0][0] == c && board[1][1] == c && board[2][2] == c) || (board[2][0] == c && board[1][1] == c && board[0][2] == c);
    }
}
