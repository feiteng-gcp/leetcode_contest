class Solution {
    public String tictactoe(int[][] moves) {
        int[][] cube = new int[3][3];
        int next = 1;
        for (int i = 0; i < moves.length; i++) {
            cube[moves[i][0]][moves[i][1]] = next;
            if (next == 1) {next = 2;}
            else next = 1;
        }
        int ans = check(cube);
        if (ans == 0) return "Pending";
        else if (ans == 1) return "A";
        else if (ans == 2) return "B";
        else return "Draw";
    }

    // 1: a win, 2: b win, 3: draw
    public int check(int[][] cube) {
        if (checkwin(cube, 1)) return 1;
        if (checkwin(cube, 2)) return 2;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (cube[i][j] == 0) return 0;
            }
        }

        return 3;
    }

    public boolean checkwin(int[][] cube, 
int num) {
        for (int i = 0; i < 3; i++) {
            if (cube[i][0] == num && cube[i][1] == num && cube[i][2] == num) return true;
        }

        for (int i = 0; i < 3; i++) {
            if (cube[0][i] == num && cube[1][i] == num && cube[2][i] == num) return true;
        }

        if (cube[0][0] == num && cube[1][1] == num && cube[2][2] == num) return true;
        if (cube[0][2] == num && cube[1][1] == num && cube[2][0] == num) return true;
        return false;
    }
}
