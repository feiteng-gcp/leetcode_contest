class Solution {
    public String tictactoe(int[][] moves) {
        int[][] board = new int[3][3];
        
        boolean isA = true;
        
        for(int[] move : moves) {
            if(isA) {
                board[move[0]][move[1]] = -1;
            } else {
                board[move[0]][move[1]] = 1;
            }
            
            isA = !isA;
        }
        
        if(isWin(board, -1)) {
            return "A";
        }
        
        if(isWin(board, 1)) {
            return "B";
        }
        
        
if(moves.length == 9) {
            return "Draw";
        }
        
        return "Pending";
    }
    
    boolean isWin(int[][] board, int play) {
        for(int i = 0; i < 3; i++) {
            int sum = 0;
            
            for(int j = 0; j < 3; j++) {
                sum +=board[i][j];
            
}
            
            if(sum == play * 3) {
                return true;
            }
            
            sum = 0;
            
            for(int j = 0; j < 3; j++) {
                sum +=board[j][i];
            
}
            
            if(sum == play * 3) {
                return true;
            }
        }
        
        return (board[0][0] + board[1][1] + board[2][2] == play * 3) || 
            (board[0][2] + board[1][1] + board[2][0] == play * 3);
    }
}
