class Solution {
    public String tictactoe(int[][] moves) {
        int[][] grid = new int[3][3];
        int turn = 1;
        for (int[] m : moves) {
            grid[m[0]][m[1]] = turn;
            if (check(grid, m[0], m[1])) {
                return turn == 1 ? "A" : "B";
            }
            
            turn = turn == 1 ? 2 : 1;
        }
        
        return full(grid) ? "Draw" : "Pending";
    }
    
    public boolean full(int[][] grid) {
        
for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (grid[i][j] == 0) return false;
            }
        }
        
        return true;
    }
    
    public boolean check(int[][] grid, int i, int j) {
        int val = grid[i][j];
        
if (grid[0][j] == val && grid[1][j] == val && grid[2][j] == val) return true;
        if (grid[i][0] == val && grid[i][1] == val && grid[i][2] == val) return true;
        if ((i == j && grid[0][0] == val && grid[1][1] == val && grid[2][2] == val)
           || (2 - i == j && grid[0][2] == val && grid[1][1] == val && grid[2][0] == val)) return true;
        return false;
    }
}
