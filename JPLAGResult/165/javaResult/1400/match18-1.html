class Solution {
    
private boolean check(int[][] a, int c, int x, int y) {
        int n = 0;
        for (int i = 0; i < 3; ++i) {
            if (a[y][i] == c) ++n
;
        }
        if (n == 3) return true;
        n = 0;
        for (int i = 0; i < 3; ++i) {
            if (a[i][x] == c) ++n
;
        }
        if (n == 3) return true;
        if (x == y) {
            n = 0;
            for (int i = 0; i < 3; ++i) {
                if (a[i][i] == c) ++n
;
            }
            if (n == 3) return true;
        }
        if (x + y == 2) {
            n = 0;
            for (int i = 0; i < 3; ++i) {
                if (a[i][2 - i] == c) ++n
;
            }
            if (n == 3) return true;
        }
        return false;
    }
    
public String tictactoe(int[][] moves) {
        int[][] a = new int[3][3];
        for (int i = 0; i < moves.length; ++i) {
            int c = (i % 2 == 0 ? 1 : 2);
            a[moves[i][0]][moves[i][1]] = c;
            if (check(a, c, moves[i][1], moves[i][0])) return c == 1 ? "A" : "B";
        }
        if (moves.length == 9) return "Draw";
        else return "Pending";
    }
}
