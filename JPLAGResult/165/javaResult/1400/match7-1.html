class Solution {
    public String tictactoe(int[][] moves) {
        boolean[][] a = new boolean[3][3];
        boolean[][] b = new boolean[3][3];
        for (int i = 0; i< moves.length; i+= 2) {
            a[moves[i][0]][moves[i][1]] = true;
        }
        
        for (int i = 1; i< moves.length; i+= 2) {
            b[moves[i][0]][moves[i][1]] = true;
        }
        
        if (isWinner(a)) {
            return "A";
        }
        
        if (isWinner(b)) {
            return "B";
        }
        
        return moves.length < 9 ? "Pending" : "Draw";
        
    }
    
    private boolean isWinner(
boolean[][] board) {
        for (int i = 0; i< board.length; i++) {
            
            boolean result = true;
            for (int j = 0; j < board[0].length; j++) {
                result &= board[i][j];
            
}
            if (result) {
                return true;
            }
        }
        
        
         for (int i = 0; i< board[0].length; i++) {
            
            boolean result = true;
            for (int j = 0; j < board.length; j++) {
                result &= board[j][i];
            
}
            if (result) {
                return true;
            }
        }
        
        boolean result = true;
        for (int i = 0; i< 3; i++) {
            result &= board[i][i];
        
}
        if (result) {
            return true;
        }
        
        result = true;
        
        for (int i = 0; i< 3; i++) {
            result &= board[2-i][i]; 
        
}
        
        if (result) {
            return true;
        }
        
        return false;
    }
    
    
    
}
