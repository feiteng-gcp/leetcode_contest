class Solution {
    
    public String tictactoe(int[][] moves) {
        char[][] board = new char[3][3];
        
        for(int i = 0; i < moves.length; i++) {
            board[moves[i][0]][moves[i][1]] = (i % 2 == 0) ? 'X' : 'O';
        }
        for(int i = 0; i < 3; i++) {
            if(board[i][0] == 'X' || board[i][1] == 'O') 
{
                boolean isSame = true;
                for(int j = 1; j < 3; j++) {
                    if(board[i][j] != board[i][0]) {
                        isSame = false;
                        break;
                    }
                }
                if(isSame) {
                    return (board[i][0] == 'X') ? "A" : "B";
                }
            }
            if(board[0][i] == 'X' || board[0][i] == 'O') 
{
                boolean isSame = true;
                for(int j = 1; j < 3; j++) {
                    if(board[j][i] != board[0][i]) {
                        isSame = false;
                        break;
                    }
                }
                if(isSame) {
                    return (board[0][i] == 'X') ? "A" : "B";
                
}
            }
        }
        if((board[0][0] == 'X' || board[0][0] == 'O') && board[1][1] == board[0][0] && board[2][2] == board[0][0]) {
            return (board[0][0] == 'X') ? "A" : "B";
        }
        if((board[0][2] == 'X' || board[0][2] == 'O') && board[1][1] == board[0][2] && board[2][0] == board[0][2]) {
            return (board[0][2] == 'X') ? "A" : "B";
        }
        if(moves.length == 9) return "Draw";
        return "Pending";
    }
}
