class Solution {

	public String tictactoe(int[][] moves) {
		int[][] c = new int[3][3];
		int idx = 0;
		for (int i = 0; i < 3; i++) {
			for (int j = 0; j < 3; j++) {
				c[i][j] = 10;
			}
		}
		for (int[] m : moves) {
			c[m[0]][m[1]] = idx % 2;
			idx++;
		}
		if (c[0][0] ==
 c[0][1] && c[0][1] == c[0][2] && c[0][0] != 10) {
			return c[0][0] == 1 ? "B" : "A";
		}
		if (c[1][0] == c[1][1] && c[1][1] == c[1][2] && c[1][2] != 10) {
			return c[1][0] == 1 ? "B" : "A";
		}
		if (c[2][0] == c[2][1] && c[2][1] == c[2][2] && c[2][2] != 10) {
			return c[2][0] == 1 ? "B" : "A";
		}
		if (c[0][0] == c[1][0] && c[1][0] == c[2][0] && c[2][0] != 10) {
			return c[0][0] == 1 ? "B" : "A";
		}
		if (c[0][1] == c[1][1] && c[1][1] == c[2][1] && c[2][1] != 10) {
			return c[0][1] == 1 ? "B" : "A";
		}
		if (c[0][2] == c[1][2] && c[1][2] == c[2][2] && c[2][2] != 10) {
			return c[0][2] == 1 ? "B" : "A";
		}
		if (c[0][0] == c[1][1] && c[1][1] == c[2][2] && c[2][2] != 10) {
			return c[0][0] == 1 ? "B" : "A";
		}
		if (c[2][0] == c[1][1] && c[1][1] == c[0][2] && c[1][1] != 10) {
			return c[2][0] == 1 ? "B" : "A";
		}
		if (moves.length == 9) {
			return "Draw";
		}
		return "Pending";
	}
}
