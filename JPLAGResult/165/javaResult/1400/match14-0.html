class Solution {
    public String tictactoe(int[][] moves) {
        int[][] bb = new int[3][3];
        for(int i=0; i<moves.length; i++){
            int[] m = moves[i];
            if(i%2==0){
                bb[m[0]][m[1]]=1;
            }else bb[m[0]][m[1]]=2;
            
            if(bb[0][0]==1 && bb[0][1]==1 && bb[0][2]==1) return "A";
            if(bb[1][0]==1 && bb[1][1]==1 && bb[1][2]==1) return "A";
            if(bb[2][0]==1 && bb[2][1]==1 && bb[2][2]==1) return "A";
            if(bb[0][0]==1 && bb[1][0]==1 && bb[2][0]==1) return "A";
            if(bb[0][1]==1 && bb[1][1]==1 && bb[2][1]==1) return "A";
            if(bb[0][2]==1 && bb[1][2]==1 && bb[2][2]==1) return "A";
            if(bb[0][0]==1 && bb[1][1]==1 && bb[2][2]==1) return "A";
            if(bb[0][2]==1 && bb[1][1]==1 && bb[2][0]==1) return "A";
            
            if(bb[0][0]==2 && bb[0][1]==2 && bb[0][2]==2) return "B";
            if(bb[1][0]==2 && bb[1][1]==2 && bb[1][2]==2) return "B";
            if(bb[2][0]==2 && bb[2][1]==2 && bb[2][2]==2) return "B";
            if(bb[0][0]==2 && bb[1][0]==2 && bb[2][0]==2) return "B";
            if(bb[0][1]==2 && bb[1][1]==2 && bb[2][1]==2) return "B";
            if(bb[0][2]==2 && bb[1][2]==2 && bb[2][2]==2) return "B";
            if(bb[0][0]==2 && bb[1][1]==2 && bb[2][2]==2) return "B";
            if(bb[0][2]==2 && bb[1][1]==2 && bb[2][0]==2) return "B";
        }
        
        if(moves.length==9) return "Draw";
        return "Pending";
    }
}
