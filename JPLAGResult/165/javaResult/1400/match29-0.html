class Solution {
    public String tictactoe(int[][] moves) {
        int[][] tt = new int[3][3];

        boolean first = true;
        for (int[] move: moves) {
            tt[move[0]][move[1]] = first ? 1 : 2;
            first = !first;
        }

        for (int i = 0; i < tt.length; i++) {
            if (tt[i][0] == 1 && tt[i][1] == 1 && tt[i][2] == 1) {
                return "A";
            }
            if (tt[i][0] == 2 && tt[i][1] == 2 && tt[i][2] == 2) {
                return "B";
            }

            if (tt[0][i] == 1 && tt[1][i] == 1 && tt[2][i] == 1) {
                return "A";
            }

            if (tt[0][i] == 2 && tt[1][i] == 2 && tt[2][i] == 2) {
                return "B";
            }
        }

        if (tt[0][0] == 1 && tt[1][1] == 1 && tt[2][2] == 1) return "A";
        if (tt[0][0] == 2 && tt[1][1] == 2 && tt[2][2] == 2) return "B";

        if (tt[2][0] == 1 && tt[1][1] == 1 && tt[0][2] == 1) return "A";
        if (tt[2][0] == 2 && tt[1][1] == 2 && tt[0][2] == 2) return "B";

        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (tt[i][j] == 0) return "Pending";
            }
        }

        return "Draw";
    }
}
