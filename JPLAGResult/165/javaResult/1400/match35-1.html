class Solution {
    public String tictactoe(int[][] moves) {
        int[][] b = new int[3][3];
        for (int i = 0; i < 9; i++) {
            if (i >= moves.length) {
                return "Pending";
            }
            int curr = i%2+1;
            int x = moves[i][0];
            int y = moves[i][1];
            b[moves[i][0]][moves[i][1]] = curr;
            
if (b[x][0]==curr&&b[x][1]==curr&&b[x][2]==curr) {
                return curr==1?"A":"B";
            }
            if (b[0][y]==curr&&b[1][y]==curr&&b[2][y]==curr) {
                return curr==1?"A":"B";
            }
            if (b[0][0]==curr&&b[1][1]==curr&&b[2][2]==curr) {
                return curr==1?"A":"B";
            }
            if (b[0][2]==curr&&b[1][1]==curr&&b[2][0]==curr) {
                return curr==1?"A":"B";
            }
        }
        return "Draw";
    }
}
