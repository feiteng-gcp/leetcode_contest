class Solution {
    
    public String tictactoe(int[][] moves) {
        char[][] board = new char[3][3];
        char cur = 'A';
        for (int[] m : moves) {
            int r = m[0];
            int c = m[1];
            board[r][c] = cur;
            if (won(board,cur)) {
                return String.valueOf(cur);
            }
            cur = cur == 'A' ? 'B' : 'A';
        }
        if (moves.length == 9) {
            return "Draw";
        } else {
            return "Pending";
        }
    }

    private boolean won(char[][] b, char cur) {
        for (int i = 0; i < 3; i++) {
            boolean same = true;
            for (int j = 0; j < 3; j++) {
                if (b[i][j] != cur) {
                    same = false;
                    break;
                }
            }
            if (same) {
                return true;
            }
        }
        for (int j = 0; j < 3; j++) {
            boolean same = true;
            for (int i = 0; i < 3; i++) {
                if (b[i][j] != cur) {
                    same = false;
                    break;
                }
            }
            if (same) {
                return true;
            }
        }
        boolean same = true;
        for(int i=0; i<3; i++){
            if(b[i][i]!=cur){
                same = false;
                break;
            }
        }
        if (same) {
            return true;
        }
        same = true;
        for(int i=0; i<3; i++){
            if(b[i][2-i]!=cur){
                same = false;
                break;
            }
        }
        if (same) {
            return true;
        }
        return false;
    }
}
