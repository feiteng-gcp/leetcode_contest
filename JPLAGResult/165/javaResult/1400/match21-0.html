class Solution {
        char[][] matrix = new char[3][3];
        char[] pos = new char[]{'A','B'};
        public String tictactoe(int[][] moves) {
            for(int i = 0; i < moves.length; i++){
                matrix[moves[i][0]][moves[i][1]] = pos[i % 2];
                if (hasLine('A')){
                    return "A";
                }
                if (hasLine('B')){
                    return "B";
                }
            }
            if (moves.length == 9) return "Draw";
            return "Pending";
        }

        boolean hasLine(char c){
            for(int i = 0; i <= 2; i++){
                boolean inline = true;
                for(int j = 0; j <= 2; j++){
                    if (matrix[i][j] != c) inline = false;
                }
                if (inline) return true;
            }
            for(int i = 0; i <= 2; i++){
                boolean inline = true;
                for(int j = 0; j <= 2; j++){
                    if (matrix[j][i] != c) inline = false;
                }
                if (inline) return true;
            }
            boolean inline = true;
            for(int i = 0,j = 0; i <= 2; i++,j++){
                if (matrix[i][j] != c) inline = false;
            }
            if (inline) return true;
            inline = true;
            for(int i = 0,j = 2; i <= 2; i++,j--){
                if (matrix[i][j] != c) inline = false;
            }
            if (inline) return true;
            return false;
        }
    }
