class Solution {
    public String tictactoe(int[][] moves) {
        int[][] board = new int[3][3];
        for (int i = 0; i < moves.length; i++) {
            int p = i % 2 + 1;
            int x = moves[i][0], y = moves[i][1];
            board[x][y] = p;
            if (check(board, x, y, p)) return p == 1 ? "A" : "B";
        }
        return moves.length == 9 ? "Draw" : "Pending";
    }
    private boolean check(int[][] board, int x, int y, 
int p) {
        boolean win = true;
        for (int i = 0; i < 3; i++) {
            if (board[i][y] != p) {
                win = false;
                break;
            }
        }
        if (win) return win
;
        win = true;
        for (int j = 0; j < 3; j++) {
            if (board[x][j] != p) {
                win = false;
                break;
            }
        }
        if (win) return win;
        if (x + y == 2) {
            win = true;
            
for (int j = 0; j < 3; j++) {
                if (board[2 - j][j] != p) {
                    win = false;
                    break;
                }
            }
            if (win) return true;
        }
        if (x == y) {
            win = true;
            
for (int j = 0; j < 3; j++) {
                if (board[j][j] != p) {
                    win = false;
                    break;
                }
            }
            if (win) return true;
        }
        return win;
    }
}
