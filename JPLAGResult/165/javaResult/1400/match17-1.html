class Solution {
    public String tictactoe(int[][] moves) {
        int[][] grid = new int[3][3];
        for (int i = 0; i < moves.length; i++) {
            int r = moves[i][0];
            int c = moves[i][1];
            int m = (i & 1) == 0 ? 1 : 2;
            grid[r][c] = m;
            if (grid[r][0] == m && grid[r][1] == m && grid[r][2] == m || grid[0][c] == m && grid[1][c] == m && grid[2][c] == m) {
                return m == 1 ? "A" : "B";
            }
            
if (r == c && grid[0][0] == m && grid[1][1] == m && grid[2][2] == m) {
                return m == 1 ? "A" : "B";
            }
            if ((r == 3 - c - 1 || c == 3 - r - 1) && grid[0][2] == m && grid[1][1] == m && grid[2][0] == m) {
                return m == 1 ? "A" : "B";
            }
        }
        return moves.length < 9 ? "Pending" : "Draw";
    }
}
