class Solution {
    public int countSquares(int[][] mp) {
        int n = mp.length;
        int m = mp[0].length;
        

         int f[][]=new int[n+2][m+2];
        
         int ans=0;
         for(int i=1;i<=n;i++){
             for(int j=1;j<=m;j++){
                 if(mp[i-1][j-1]==0)continue;
                  f[i][j]=Math.min(f[i-1][j],Math.min(f[i-1][j-1],f[i][j-1]))+mp[i-1][j-1];
                  ans +=f[i][j];
             }
         }
        return ans;
        
    }
}
