class Solution {
    public int countSquares(int[][] matrix) {
        int n=matrix.length, m=matrix[0].length;
        int ans=0;
        
        for (int i=0; i<Math.min(n,m); i++) {
            for (int j=0; j<n-i; j++) {
                for (int k=0; k<m-i; k++) {
                    boolean flag=true;
                    for (int x=j; x<=j+i; x++) {
                        for (int y=k; y<=k+i; y++) {
                            if (matrix[x][y]==0) {
                                flag=false;
                                break;
                            }
                        }
                        if (flag==false)
                            break;
                    }
                    if (flag) {
                        //System.out.printf("(%d,%d) size %d\n",j,k,i);
                        ans+=1;
                    }
                }
            }
        }
        return ans;
    }
}
