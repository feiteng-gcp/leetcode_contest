class Solution {
    public int countSquares(int[][] matrix) {
        int n = matrix.length, m = matrix[0].length;
        int[][] counts = new int[n][m];
        int sum = 0;
        for (int i = 0; i < n; i++) {
            counts[i][0] = matrix[i][0];
            sum += counts[i][0];
        }
        for (int j = 1; j < m; j++) {
            counts[0][j] = matrix[0][j];
            sum += counts[0][j];
        }
        for (int i = 1; i < n; i++) {
            for (int j = 1; j < m; j++) {
                if (matrix[i][j] == 1) {
                    counts[i][j] = 1 + Math.min(counts[i - 1][j - 1], Math.min(counts[i - 1][j], counts[i][j - 1]));
                }
                sum += counts[i][j];
            }
        }
        return sum;
    }
}
