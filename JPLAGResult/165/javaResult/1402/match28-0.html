class Solution {
    public int countSquares(int[][] matrix) {
        int m=matrix.length,n=matrix[0].length;
        int[][] dp=new int[m][n];
        int res=0;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(matrix[i][j]==0) continue;
                dp[i][j]=1;
                int top=i-1<0?0:dp[i-1][j];
                int left=j-1<0?0:dp[i][j-1];
                int dia=(i-1<0||j-1<0?0:dp[i-1][j-1]);
                dp[i][j]+=Math.min(top,Math.min(left,dia));
                res+=dp[i][j];
            }
        }
        return res;
    }
}
