class Solution {
    public int countSquares(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        
        //sum[i][j] -> sum util i,j
        int[][] sum = new int[m+1][n+1];
        
        // sum[0][0] = matrix[0][0];
        // for (int i = 1; i < m; i++) {
        //     sum[i][0] = sum[i-1][0] + matrix[i][0];
        // }
        // for (int j = 1; j < n; j++) {
        //     sum[0][j] = sum[0][j-1] + matrix[0][j];
        // }
        
        for (int i = 1; i <= m; i++) {
            int sumj = 0;
            
for (int j = 1; j <= n; j++) {
                sumj += matrix[i-1][j-1];
                sum[i][j] = sum[i-1][j] + sumj;
                //sum[i][j] = sum[i][j-1] + matrix[i-1][j-1];
            }
        }
        
        //for (int i = 0; i <= m; i++)
         //   System.out.println(Arrays.toString(sum[i]));
        
        int count = sum[m][n];
        
        for (int gap = 2; gap <= Math.min(m, n); gap++) {
            for (int i = 0; i+gap <= m; i++) {
                for (int j = 0; j+gap <= n; j++) {
                    int i1 = i + gap;
                    int j1 = j + gap;
                    
int sum1 = sum[i1][j1] - sum[i][j1] - sum[i1][j] + sum[i][j];
                    if (sum1 == gap * gap) {
                        count++;
                    }
                }
            }
        }
        return count;
        
        
        
    }
}
