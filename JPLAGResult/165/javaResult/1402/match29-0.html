class Solution {
    public int countSquares(int[][] matrix) {
        int result = 0;
        for (int i = 0; i < Math.min(matrix.length, matrix[0].length); i++) {
            result += helper(matrix, i);
            // System.out.println(result + " " + i);
        }
        return result;
    }
    private int helper(int[][] matrix, int len) {
        int result = 0;
        for (int i = 0; i < matrix.length; i++) {
            for (int j = 0; j < matrix[0].length; j++) {
                if (matrix[i][j] == 1 && valid(matrix, i, j, len)) {
                    // if (len == 1) {
                    //     System.out.println(i + " " + j);
                    // }
                    result ++;
                }
            }
        }
        return result;
    }
    private boolean valid(int[][] matrix, int row, int col, int len) {
        if (row + len >= matrix.length || col + len >= matrix[0].length) {
            return false;
        }
        for (int i = row; i <= row + len; i++) {
            for (int j = col; j <= col + len; j++) {
                if (matrix[i][j] == 0) {
                    return false;
                }
            }
        }
        return true;
    }
}
