class Solution {
    public int countSquares(int[][] matrix) {
        int M = matrix.length, N = matrix[0].length;
        int[][] sum = new int[M][N + 1];
        
        for (int i = 0; i < M; i++) {
            for (int j = 0; j < N; j++) {
                sum[i][j + 1] = sum[i][j] + matrix[i][j];
            }
            
            // System.out.println(Arrays.toString(sum[i]));
        }
        
        int res = 0;
        
        for (int i = 0; i < M; i++) {
            
            int[] preSum = new int[N + 1];
            
            for (int j = i; j < M; j++) {
                for (int l = 0; l <= N; l++) {
                    preSum[l] += sum[j][l];
                }
                
                // System.out.println(i + " " + j + " " + Arrays.toString(preSum));
                int L = j - i + 1; // length of sub matrics squares
                
                for (int k = L - 1; k < N; k++) {
                    // System.out.println(k);
                    int tmp = preSum[k + 1] - preSum[k + 1 - L];
                    // System.out.println(k + " " + tmp + " " + L);
                    if (tmp == L * L) {
                        res++;
                    }
                }
            }
        }
        
        return res;
        
    }
}
