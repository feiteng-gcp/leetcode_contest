class Solution {
    public int countSquares(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[][] down = new int[m][n];
        for (int i = m - 1; i >= 0; i--) {
            for (int j = 0; j < n; j++) {
                down[i][j] = matrix[i][j] == 0 ? 0 : 1 + (i == m - 1 ? 0 : down[i + 1][j]);
            
}
        }
        
        //System.out.println(Arrays.deepToString(down));
        int res = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int max = down[i][j];
                int seen = 0;
                for (int k = j; k >= Math.max(0, j - max + 1); k--) {
                    max = Math.min(max, down[i][k]);
                    if (k >= Math.max(0, j - max + 1)) seen++
;
                }
                
                //System.out.println(Math.min(seen, max) + " " + i + " " + j);
                res += seen;
            }
        }
        return res;
    }
}
