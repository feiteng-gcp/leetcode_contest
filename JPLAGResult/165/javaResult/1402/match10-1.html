class Solution {
    public int countSquares(int[][] mat) {
        int n = mat.length;
        int m = mat[0].length;
        int max = Math.min(n,m);
        int[][][] dp = new int[max][n][m];
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                dp[0][i][j] = mat[i][j];
                ans += dp[0][i][j];
            
}
        }
        for (int i = 1; i < max; ++i) {
            for (int j = 0; j < n-i; ++j) {
                for (int k = 0; k < m-i; ++k) {
                    dp[i][j][k] = (dp[i-1][j][k]+dp[i-1][j+1][k]+dp[i-1][j][k+1]+dp[i-1][j+1][k+1])>>2;
                    ans += dp[i][j][k];
                
}
            }
        }
        return ans;
    }
}
