class Solution {
    public int countSquares(int[][] matrix) {
        int r = matrix.length;
        int c = matrix[0].length;
        int[][] sums = new int[r+1][c+1];
        for (int i=0; i<r; i++) {
            for (int j=0; j<c; j++) {
                sums[i+1][j+1] = matrix[i][j] + sums[i][j+1] + sums[i+1][j] - sums[i][j];
            }
        }
        
        int total = 0;
        for (int i=0; i<r; i++) {
            for (int j=0; j<c; j++) {
                for (int k=0; k<=300; k++) {
                    if (i-k < 0 || j-k < 0) {
                        break;
                    }
                    if (sums[i+1][j+1] - sums[i-k][j+1] - sums[i+1][j-k] + sums[i-k][j-k] == (k+1)*(k+1)) {
                        total++;
                    } else {
                        break;
                    
}
                }
            }
        }
        return total;
    }
}
