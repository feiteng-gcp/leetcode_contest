class Solution {
    public int countSquares(int[][] m) {
        
int num = 0;
        for (int i = 1; i <= Math.min(m.length, m[0].length); i++) {
            for (int j = 0; j < m.length - i + 1; j++) {
                for (int k = 0; k < m[0].length - i + 1; k++) {
                    boolean flag = true;
                    for (int a = j; a < j + i; a++) {
                        for (int b = k; b < k + i; b++) {
                            if (m[a][b] != 1) {
                                flag = false;
                                break;
                            }
                        }
                        if (!flag) {
                            break;
                        
}
                    }
                    if (flag) {
                        num++;
                    }
                }
            }
        }
        return num;
    }
}
