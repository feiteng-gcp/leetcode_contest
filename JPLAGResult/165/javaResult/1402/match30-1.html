class Solution {
    public int countSquares(int[][] m) {
        int r = m.length;
        int c = m[0].length;
        int[][] dp = new int [r][c];
        int res = 0;
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                if (m[i][j] == 0) {continue;}
                dp[i][j] = Math.min(get(dp, i - 1, j - 1), Math.min(
get(dp, i - 1, j), get(dp, i, j - 1))) + 1;
                res += dp[i][j];
            }
        }
        return res;
    }
    
    int get(int[][] dp, int i, int j) {
        if (i < 0 || j < 0) return 0;
        return dp[i][j];
    }
}
