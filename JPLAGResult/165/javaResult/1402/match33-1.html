class Solution {
    public int countSquares(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[][] DP = new int[m][n];
        int res = 0;
        for(int i=0;i<m;i++){
            DP[i][0] = matrix[i][0]==1?1:0;
            res += DP[i][0];
        }
        for(int j=0;j<n;j++){
            DP[0][j] = matrix[0][j]==1?1:0;
            if(j==0){
                continue;
            }
            res += DP[0][j];
        }
        for(int i=1;i<m;i++){
            
for(int j=1;j<n;j++){
                if(matrix[i][j]==0){
                    DP[i][j] = 0;
                }else{
                    int min = Math.min(DP[i-1][j-1], Math.min(DP[i-1][j], DP[i][j-1]));
                    DP[i][j] = min+1;
                
}
                res += DP[i][j];
            }
        }
        return res;
    }
}
