class Solution {
    public int countSquares(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        int maxLen = Math.min(m, n);
        int[][] sum = new int[m+1][n+1];
        
for (int row=1;row<=m;row++) {
            for (int col=1;col<=n;col++) {
                sum[row][col] = sum[row][col-1] + sum[row-1][col] - sum[row-1][col-1] + matrix[row-1][col-1];
            }
        }
        int erow, ecol;
        int count = 0;
        
for (int i=1;i<=maxLen;i++) {
            for (int row=1;row<=m;row++) {
                for (int col=1;col<=n;col++) {
                    erow = row + i - 1;
                    ecol = col + i - 1;
                    if (erow > m || ecol > n) {
                        continue;
                    }
                    if (sum[erow][ecol] - sum[erow][col-1] - sum[row-1][ecol] + sum[row-1][col-1] == i*i) {
                        count++;
                    }
                
}
            }
        }
        return count;
    }

    public static void main(String[] args) {
        System.out.println(new Solution().countSquares(new int[][] {
                {0,1,1,1},
  {1,1,1,1},
  {0,1,1,1}
        }));
    }
}
