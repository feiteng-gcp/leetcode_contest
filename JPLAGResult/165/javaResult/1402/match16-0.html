class Solution {
    public int countSquares(int[][] matrix) {
        int rows = matrix.length;
        int cols = matrix[0].length;
        int[][] counts = new int[rows + 1][cols + 1];
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                counts[i + 1][j + 1] = counts[i + 1][j] + counts[i][j + 1] - counts[i][j] + matrix[i][j];
            }
        }
        int ret = 0;
        for (int l = 1; l <= rows && l <= cols; l++) {
            int s = l * l;
            for (int i = 0; i + l <= rows; i++) {
                for (int j = 0; j + l <= cols; j++) {
                    int ii = i + l;
                    int jj = j + l;
                    if (counts[ii][jj] - counts[ii][j] - counts[i][jj] + counts[i][j] == s) {
                        ret++;
                    }
                }
            }
        }
        return ret;
    }
}
