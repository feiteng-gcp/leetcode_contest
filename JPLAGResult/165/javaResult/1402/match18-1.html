class Solution {
    public int countSquares(int[][] matrix) {
        int n = matrix.length;
        int m = matrix[0].length;
        
        int[][] sums = new int[n+1][m+1];
        for (int y = 1; y <= n; y++) {
            int sum = 0;
            
for (int x = 1; x <= m; x++) {
                sum += matrix[y-1][x-1];
                sums[y][x] = sums[y-1][x] + sum;
            }
        }
        
        int ans = 0;
        for (int y = 1; y <= n; y++) {
            for (int x = 1; x <= m; x++) {
                int minyx = Math.min(y, x);
                for (int k = 1; k <= minyx; k++) {
                    int val = sums[y][x] - sums[y-k][x] - sums[y][x-k] + sums[y-k][x-k];
                    if (val == k*k) {
                        ans++;
                    } else {
                        break;
                    
}
                }
            }
        }
        return ans;
    }
}
