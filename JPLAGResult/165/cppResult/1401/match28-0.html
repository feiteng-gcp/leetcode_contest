class Solution {
public:
    vector<int> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        int to_divide = (tomatoSlices - 2*cheeseSlices);
        if(to_divide%2||to_divide<0)
            return {};
        int x = to_divide/2;
        int y = cheeseSlices - x;
        if(y<0)
            return {};
        return {x, y};
    }
};
