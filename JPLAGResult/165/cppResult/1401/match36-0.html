class Solution {
public:
    vector<int> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        int x = tomatoSlices - 2 * cheeseSlices;

        if (x % 2 != 0)
            return {};

        int jumbo = x / 2;
        int small = cheeseSlices - jumbo;

        if (jumbo < 0 || small < 0)
            return {};

        return {jumbo, small};
    }
};
