class Solution {
public:
    vector<int> numOfBurgers(int a, int b) {
        for (int k = 0; k <= b; ++k) {
            int r1 = 4*k;
            int r2 = 2*(b-k);
            if (r1 + r2 == a) {
                return {k, b-k};
            }
        }
        return {};
    }
};
