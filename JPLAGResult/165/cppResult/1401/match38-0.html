class Solution {
public:
    vector<int> numOfBurgers(int tomatoSlices, int cheeseSlices) {
        int xx = tomatoSlices - 2 * cheeseSlices;
        if (xx < 0 || xx % 2 != 0)
            return {};
        int x = xx / 2;
        int y = cheeseSlices - x;
        if (y < 0)
            return {};
        
        return {x, y};
    }
};
