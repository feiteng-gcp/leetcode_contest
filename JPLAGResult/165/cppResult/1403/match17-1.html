class Solution {
public:
    int palindromePartition(string s, int k) {
        int n =s.length();
        
        vector<vector<int> > dp(n+1);
        vector<vector<int> > tmp(n+1);
        for(int i=0;i<=n;i++){
            tmp[i].resize(n+1);
            for(int j=0;j<=n;j++){
                if(i==0&&j==0)dp[i].push_back(0);
                else
                dp[i].push_back(n+1);
                tmp[i][j]=0;
            }
        }
        for(int i=0;i<n;i++){
          for(int j=i;j<n;j++){
              if(i==j){
                  tmp[i+1][j+1]=0;
              }
              else{

                  for(int k=0;k<=(j-i)/2;k++){
                      if(s[i+k]!=s[j-k]){
                          tmp[i+1][j+1]++;
                      }
                  }
              }
          }
        }
        for(int i=1;i<=n;i++){

            for(int j=1;j<=min(n,k);j++){
                for(int a=j;a<=i;a++){
                    dp[i][j]=min(dp[a-1][j-1]+tmp[a][i],dp[i][j]);
                }
            }
        }
        return dp[n][k];
    }
};
