class Solution {
public:
    string ss;
    int kk;
    
    map<pair<int, int>, int> m;
    map<pair<int, int>, int> dp;
    
    int f(int b, int e) {
        if(m.count({b,e})) return m[{b,e}];
        int ans = 0;
        for(int i = b, j = e; i < j; i++, j--) {
            if(ss[i] != ss[j]) ans++;
        }
        return m[{b,e}] = ans;
    }
        
    int se(int n, int k) {
        if(dp.count({n,k})) return dp[{n,k}];
        if(k == 1) return dp[{n,k}] = f(0, n);
        
        int ret = 0x7ffffff;
        for(int i = 0; i < n; i++) {
          //  printf("%d %d %d %d\n", n,k,i, ret);
            ret = min(ret, se(i, k-1) + f(i+1, n));
        }
        return dp[{n,k}] = ret;
    }
    int palindromePartition(string s, int k) {
        ss = s;
        kk = k;
        return se(s.size() - 1, k);
    }
};
