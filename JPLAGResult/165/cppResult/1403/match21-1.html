class Solution {
public:
    int palindromePartition(string s, int k) {
        int len = s.length();
        vector<vector<int>> cost(len, vector<int>(len)); // 

        for (int i = 0; i < len; i++) {
            for (int j = i; j < len; j++) {
                int begin = i, end = j, cnt = 0;
                while(begin < end) {
                    if (s[begin] != s[end])
                        cnt++;
                    begin ++, end--;
                }
                cost[i][j-i] = cnt;
            }
        }
        /*
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < len; j++) {
                cout << cost[i][j] << " ";
            } cout << endl;
        }
        */
        vector<vector<int>> dp(len, vector<int>(k, INT_MAX));
        for (int i = 0; i < len; i++) {
            dp[i][0] = cost[i][len-1-i];
        }

        for (int kk = 1; kk < k; kk++) {
            for (int i = 0; i < len; i++) {
                for (int j = i+1; j+kk-1 < len; j++) {
                    dp[i][kk] = min(dp[i][kk], dp[j][kk-1] + cost[i][j-i-1]);
                }
            }
        }
        /*
        cout << "---" << endl;
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < k; j++) {
                cout << dp[i][j] << " ";
            } cout << endl;
        }
        */
        return dp[0][k-1];
    }
};
