class Solution {
public:

    int get(string &s, int l, int r) {
        int res = 0;
        while (l <= r) {
            if (s[l++] != s[r--]) res++;            
        }
        return res;
    }
    int palindromePartition(string s, int k) {
        int n = s.size();
        int pal[n][n];

        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                pal[i][j] = get(s, i, j);
                // cout << i << ' ' << j << ' ' << pal[i][j] << endl;
            }
        }
        long long dp[n + 1][k + 1];                
        for (int i = 0; i <= n; i++) for (int j = 0; j <= k; j++) dp[i][j] = INT_MAX;

        dp[0][0] = 0;
        // for (int i = 0; i <= n; i++) dp[i][0] = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= k; j++) {
                for (int l = 0; l < i; l++) {
                    dp[i][j] = min(dp[i][j], dp[l][j - 1] + pal[l][i - 1]);
                }
            }
        }
        return dp[n][k];
    }
};
