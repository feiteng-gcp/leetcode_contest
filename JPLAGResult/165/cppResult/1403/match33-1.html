class Solution {
public:

    int palindromePartition(string s, int k) {
      int len = s.size();
      vector<vector<int>> change(len, vector<int>(len, 0));
      
      for (int i = 0; i + 1 < len; ++i) {
        int j = i + 1;
        if (s[i] != s[j]) {
          change[i][j] = 1;
        }
      }
      
      for (int l = 3; l <= len; ++l) {
        for (int i = 0; i + l <= len; ++i) {
          int j = i + l - 1;
          if (s[i] == s[j]) {
            change[i][j] = change[i+1][j-1];
          } else {
            change[i][j] = change[i+1][j-1] + 1;
          }

        }
      }
      
      vector<vector<int>> dp(len + 1, vector<int>(k + 1, 0));
      for (int l = 2; l <= len; ++l) {
        dp[l][1] = change[0][l-1];
        
        for (int split = 2; split <= k; ++split) {
          dp[l][split] = dp[l][split - 1];

          for (int p = 1; p < l; ++p) {
            dp[l][split] = min(dp[l][split], dp[p][split-1] + change[p][l-1]);
          }
        }
      }
      
      return dp[len][k];
    }
};
