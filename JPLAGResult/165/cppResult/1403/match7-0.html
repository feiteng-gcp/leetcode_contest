class Solution {
public:
    int palindromePartition(string s, int k) {
        int n = s.size();
        vector<vector<int>> dp(k+1,vector<int>(n+1,1000000));
        dp[0][0]=0;
        // for(int j=0;j<n+1;j++){
        //     dp[0][j] = 0;
        // }
        for(int i=1;i<=k;i++){
            for(int j=i;j<=n;j++){
                for(int p=(i-1);p<j;p++){
                    int cnt = j-p;
                    int temp = 0;
                    for(int q=0;q<cnt/2;q++){
                        if(s[p+q] != s[p+cnt-1-q]) temp++;
                    }
                    // cout<<temp<<endl;
                    dp[i][j] = min(dp[i-1][p]+temp,dp[i][j]);
                }
            }
        }
        return dp[k][n];
    }
};
