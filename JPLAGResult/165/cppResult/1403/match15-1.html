class Solution {
public:
    
    int changeCount(string &s,int start,int end){
        int r =0;
        while(start < end){
            if(s[start] != s[end]){
                r++;
            }
            start++;
            end--;
        }
        
        return r;
    }
    
    int palindromePartition(string s, int k) {
        int sl = s.size();
        
        int dp[k+1][sl+1];
        
        
        for(int i=0;i < k+1;i++ ){
            for(int j=0;j < sl+1;j++){
                dp[i][j] = INT_MAX/2;
            }
        }
        dp[0][0] = 0;
        

        for(int stage = 1;stage <= k;stage++){
            for(int i=1;i <= sl;i++){
                for(int div = 0;div < i;div++){
                    int temp = changeCount(s,div+1-1,i-1);
                    dp[stage][i] = min(dp[stage][i],dp[stage-1][div]+temp);
                }
            }
        }
        return dp[k][sl];
        
    }
};
