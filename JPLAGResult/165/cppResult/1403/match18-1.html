class Solution {
public:
    int palindromePartition(string s, int k) {
        int n = s.length();
        vector<vector<int>> change(n, vector<int>(n, 0));
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                int len = j - i + 1;
                for (int l = 0; l < len / 2; l++) {
                    if (s[i + l] != s[i + len - l - 1]) {
                        change[i][j]++;
                    }
                }
            }
        }
        vector<vector<int>> dp(n, vector<int>(k + 1, -1));
        for (int i = 0; i < n; i++) {
            dp[i][1] = change[0][i];
            for (int j = 2; j <= k; j++) {
                int result = -1;
                bool init = true;
                for (int st = 1; st <= i; st++) {
                    if (dp[st - 1][j - 1] != -1) {
                        int cur = dp[st - 1][j - 1] + change[st][i];
                        if (init || cur < result) {
                            result = cur;
                            init = false;
                        }
                    }
                }
                dp[i][j] = result;
            }
        }
        return dp[n - 1][k];
    }
};
