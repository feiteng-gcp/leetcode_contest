class Solution {
public:
    int dp[110][110];
    int cost[110][110];
    int INF = 1e9+5;
    int palindromePartition(string s, int cnt) {
        int n = s.size();
        for(int i=0;i<=n;i++)
        {
            for(int j=0;j<=n;j++)
            {
                dp[i][j] = INF;
                cost[i][j] = 0;
            }   
        }
        dp[0][0] = 0;
        for(int len = 1;len<=n;len++)
        {
            for(int i=0;i<n;i++)
            {
                int j = i+len-1;
                if(j>=n)continue;
                int tmp = 0;
                for(int x = i;x<=(i+j)/2;x++)
                {
                    if(s[x] != s[i+j-x])
                        tmp++;
                }
                cost[i+1][j+1] = tmp;
            }
        }
        for(int i=1;i<=n;i++)
        {
            for(int k=1;k<=cnt;k++)
            {
                for(int j=0;j<i;j++)
                {
                    dp[i][k] = min(dp[j][k-1] + cost[j+1][i],dp[i][k]);
                }
            }
        }
        return dp[n][cnt];
    }
};
