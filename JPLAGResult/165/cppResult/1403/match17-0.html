class Solution {
public:
    int palindromePartition(string s, int k) {
        int L = s.length();
        vector<vector<int>> num(L, vector<int>(L, 0));
        for (int i = 0; i < L; ++i) {
            for (int j = i; j < L; ++j) {
                int N = (j - i + 1) / 2;
                for (int t = 0; t < N; ++t) {
                    if (s[i + t] != s[j - t]) {
                        ++num[i][j];
                    }
                }
            }
        }
        vector<vector<int>> min_num(L, vector<int>(k, L));
        for (int i = 0; i < L; ++i) {
            min_num[i][0] = num[0][i];
            for (int j = 1; j < k; ++j) {
                for (int start = 1; start <= i; ++start) {
                    min_num[i][j] = min(min_num[i][j], min_num[start - 1][j - 1] + num[start][i]);
                }
            }
        }
        return min_num[L - 1][k - 1];
    }
};
