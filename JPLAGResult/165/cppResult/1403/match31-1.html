class Solution {
public:
    const int INF = 10007;

    int palindromePartition(string s, int k) {
        int n = s.length();
        vector<vector<int>> need_change(n, vector<int>(n, 0));
        for (int i = n - 1; i >= 0; i--) {
            for (int j = i + 1; j < n; j++) {
                need_change[i][j] = need_change[i + 1][j - 1] + (s[i] != s[j]);
            }
        }
        
        vector<vector<int>> f(n, vector<int>(k + 1, INF));
        
        for (int i = 0; i < n; i++) {
            f[i][1] = need_change[0][i];

            for (int j = 2; j <= k; j++) {
                for (int first = 1; first <= i; first++) {
                    f[i][j] = min(f[i][j], f[first - 1][j - 1] + need_change[first][i]);
                }
            }
        }
        
        return f[n - 1][k];
    }
};
