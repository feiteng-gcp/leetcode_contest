class Solution {
public:
    int palindromePartition(string s, int k) {
        const int len = s.size();
        vector<vector<int>> count(len, vector<int>(len, 0));
        for (int i = 0; i < len; ++i) {
            for (int j = i; j < len; ++j) {
                for (int p = i; p <= j; ++p) {
                    const int pos = p - i;
                    if (s[p] != s[j - pos]) count[i][j]++;
                }
                count[i][j] /= 2;
            }
        }
        
        vector<vector<int>> opt(len, vector<int>(len + 1, 10000));
        for (int i = 0; i < len; ++i) opt[i][1] = count[0][i];
        for (int j = 2; j <= len; ++j) {
            for (int i = 0; i < len; ++i) {
                for (int p = 0; p < i; ++p) {
                    opt[i][j] = min(opt[i][j], opt[p][j - 1] + count[p + 1][i]);
                }
            }
        }
        return opt[len - 1][k];
    }
};
