class Solution {
public:
    int palindromePartition(string s, int k) 
    {
        int n = s.size();
        int K= k;
        auto dp = vector<vector<int>>(n+1,vector<int>(K+1,0));
        s = "#"+s;
        

        for (int i=1; i<=n; i++)
        {
            dp[i][1] = helper(s.substr(1,i-1+1));
            // cout<<i<<" "<<1<<" "<<dp[i][1]<<endl;
        }
            
        
        for (int i=2; i<=n; i++)
        {
            for (int k=2; k<=min(K,i); k++)
            {
                dp[i][k] = INT_MAX;

                for (int j=k; j<=i; j++)
                {
                    dp[i][k] = min(dp[i][k], dp[j-1][k-1]+helper(s.substr(j,i-j+1)));
                    // if (i==8 )
                    // {
                    //     cout<<i<<" "<<k<<" "<<j<<" "<<dp[j-1][k-1]<<" "<<helper(s.substr(j,i-j+1))<<endl;
                    // }
                }
                    
                
                
            }
        }
        
        return dp[n][K];
    }
    
    int helper(string s)
    {
        int i=0;

        int j=s.size()-1;
        int count = 0;
        while (i<j)
        {
            if (s[i]!=s[j])
            {
                count++;
            }
            i++;
            j--;
        }
        return count;
    }
};
