class Solution {
public:
    int need[105][105];
    int f[105][105];
    int INF = 1e9;
    
    int palindromePartition(string s, int k) {
        int n = s.size();
        for (int l = 0; l < n; l++) {
            for (int r = l; r < n; r++) {
                need[l][r] = 0;
                int i = l, j = r;
                for (; i <= j; i++, j--) {
                    if (s[i] != s[j]) {
                        need[l][r]++;
                    }
                }

            }
        }
        
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= 100; j++) {

                f[i][j] = INF;
            }
        }
        f[0][0] = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < k; j++) {
                if (f[i][j] == INF) {
                    continue;
                }
                
                //printf("%d %d %d\n", i, j, f[i][j]);

                for (int l = i + 1; l <= n; l++) {
                    f[l][j + 1] = min(f[l][j + 1], f[i][j] + need[i][l - 1]);
                    
                    //printf("get %d %d %d\n", l, j + 1, f[l][j + 1]);
                }
            }
        }
        
        return f[n][k];
    }
};
