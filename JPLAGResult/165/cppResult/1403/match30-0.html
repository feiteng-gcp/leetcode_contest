int g[105][105];
int dp[105][105];

class Solution {
public:
    int palindromePartition(string s, int p) {
        int n = s.size();
        for(int i = 0; i < n; i++) {
            for(int j = i; j < n; j++) {
                int cnt = 0;
                for(int k = 0; i + k < j - k; k++) {
                    if(s[i + k] != s[j - k]) {
                        cnt += 1;
                    }
                }
                g[i][j] = cnt;
            }
        }
        for(int i = 0; i < n; i++) {
            for(int j = 0; j <= p; j++) {
                dp[i][j] = 1e9;
            }
        }
        for(int i = 0; i < n; i++) {
            dp[i][1] = g[0][i];
            for(int j = 2; j <= p; j++) {
                for(int k = 0; k < i; k++) {
                    dp[i][j] = min(dp[i][j], dp[k][j - 1] + g[k + 1][i]);
                }
            }
        }
        return dp[n - 1][p];        
    }
};
