int cnt[101][101];
int dp[101][101];
class Solution {
public:
    int palindromePartition(string s, int k) {
        int len=s.size();
        for(int i=0;i<len;i++)
        {
            for(int j=i;j<len;j++)
            {
                int l=i,r=j;
                cnt[i][j]=0;
                while(l<r)
                {
                    if(s[l]!=s[r])cnt[i][j]++;
                    l++;
                    r--;
                }
            }
        }
        memset(dp,1,sizeof(dp));
        dp[0][0]=0;
        for(int c=1;c<=k;c++)
        {
            for(int i=1;i<=len;i++)
            {
                for(int j=0;j<i;j++)
                {
                    dp[c][i]=min(dp[c][i],dp[c-1][j]+cnt[j][i-1]);
                }
            }
        }
        return dp[k][len];
    }
};
