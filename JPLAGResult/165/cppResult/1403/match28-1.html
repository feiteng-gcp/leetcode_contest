class Solution {
public:
    int palindromePartition(string s, int k) {
        int n = s.size();
        if(n == k)
            return 0;
        

        vector<vector<int>> dist(n, vector<int>(n));
        // dist[i,j] = edit dist of s[i,j];
        for(int len = 2; len <= n; len++){
            for(int i=0; i+len <= n; i++){
                int j = i+len-1;
                dist[i][j] = (j > i+1 ? dist[i+1][j-1] : 0) + (s[i] != s[j]);

            }
        }
        
        vector<vector<int>> DP(n, vector<int>(k+1, INT_MAX));
        // DP[i][kk] = min divide s[0,i] into kk parts;
        // DP[i][1] = ?
        // DP[kk-1][kk] = 0
        
        for(int i=0; i<n; i++)
            DP[i][1] = dist[0][i];
        
        for(int kk = 2; kk <= k; kk++){
            DP[kk-1][kk] = 0;

            for(int i=kk; i<n; i++){
                for(int j = kk-2; j < i ; j++){
                    DP[i][kk] = min(DP[i][kk], DP[j][kk-1] + dist[j+1][i]);
                }
                //cout<<i<<"\t"<<kk<<"\t"<<DP[i][kk]<<endl;
            }
        }
        return DP[n-1][k];
    }
};
