class Solution {
public:
    int dp[200][200];
    int getx(const string &s, int l, int r) {
        int count = 0;
        while (l < r) {
            if (s[l] != s[r])
                ++count;
            ++l;
            --r;
        }
        return count;
    }
    int palindromePartition(string s, int k) {
        memset(dp, 0x3f3f3f3f, sizeof(dp));
        int len = s.length();
        for (int i = 0; i < len; i++) {
            dp[i][1] = getx(s, 0, i);
            for (int j = 1; j <= min(i + 1, k); j++) {
                for (int d = i; d >= 0; d--) {
                    int tmp = getx(s, d, i);
                    if (j - 1 > 0 && d >= j - 1)
                        dp[i][j] = min(dp[i][j], dp[d - 1][j - 1] + tmp);
                }
            }
            // for (int j = 1; j < 10; j++) {
            //     cout << dp[i][j] << " ";
            // }
            // cout << endl;
        }
        return dp[len - 1][k];
    }
};
