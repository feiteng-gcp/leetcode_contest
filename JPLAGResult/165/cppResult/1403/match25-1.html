class Solution {
 public:

  int palindromePartition(string s, int k) {
    int n = s.size();
    vector<vector<int>> cost(n, vector<int>(n + 1));
    for (int i = 0; i < n; ++i) {
      for (int j = 1; j <= n; ++j) {
        for (int x = i, y = j - 1; x < y; ++x, --y) {
          cost[i][j] += s[x] != s[y];
        }

      }
    }
    vector<vector<int>> dp(k + 1, vector<int>(n + 1, 0x3f3f3f3f));
    dp[0][0] = 0;
    for (int t = 0; t < k; ++t) {
      for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j <= n; ++j) {
          dp[t + 1][j] = min(dp[t + 1][j], dp[t][i] + cost[i][j]);
        }
      }
    }
    return dp[k][n];
  }
};
