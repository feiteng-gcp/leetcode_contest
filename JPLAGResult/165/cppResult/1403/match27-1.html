class Solution {
public:
    int palindromePartition(string s, int k) {
        int n = s.size();
        vector<vector<int>> cnts(n, vector<int>(n, 0));
        
        for (int i = 0; i < n; ++i) {
            for (int j = i; j < n; ++j) {
                cnts[i][j] = getCounts(s, i, j);
                //cout << "cnts: " << i << " " << j << " " << cnts[i][j] << endl;
            }
        }
        

        vector<vector<int>> dp(n + 1, vector<int>(k + 1, 0));
        
        for (int i = 1; i <= n; ++i) {
            dp[i][0] = n + 1;
            
            for (int j = max(1, k - n + i); j <= min(i, k); ++j) {

                dp[i][j] = n + 1;
                
                for (int l = max(0, j - 1); l < i; ++l) {
                    dp[i][j] = min(dp[i][j], dp[l][j - 1] + cnts[l][i - 1]);
                }
                
                //cout << i << " " << j << " " << dp[i][j] << endl;
            }
        }
        
        return dp[n][k];
    }
    

    int getCounts(string& s, int lhs, int rhs) {
        int cnt = 0;
        
        while (lhs < rhs) {
            if (s[lhs++] != s[rhs--])
                ++cnt;
        }
        
        return cnt;
    }
};
