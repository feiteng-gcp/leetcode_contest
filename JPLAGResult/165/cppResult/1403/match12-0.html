class Solution {
public:
    
    int get_cost(const string& s, int i, int j){
        int c = 0;
        while(i < j){
            if (s[i] != s[j]) c ++;
            i++;
            j--;
        }
        return c;
    }
    
    int palindromePartition(string s, int k) {
        
        int n = s.size();
        vector<vector<int>> cost(n + 1, vector<int>(n + 1, 0));
        for(int i = 1; i <= n; i++){
            for(int j = i; j <= n ; j ++){
                cost[i][j] = get_cost(s, i - 1, j - 1);
            }
        }
        
        vector<vector<int>> dp(n + 1, vector<int> (k + 1, n + 100));
        dp[0][0] = 0;
        
        for(int i = 1;i <= n; i ++){
            for(int j = 1; j <= k; j ++){
                for(int x = j - 1; x < i; x ++ ){
                    dp[i][j] = min(dp[i][j], dp[x][j-1] + cost[x + 1][i]);
                }
            }
        }
        
        return dp[n][k];
    }
};
