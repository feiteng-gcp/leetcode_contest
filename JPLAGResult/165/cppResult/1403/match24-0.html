class Solution {
public:
    int palindromePartition(string s, int k) {
        if(s.length() == k) {
            return 0;
        }
        
        vector<vector<int>> dp(s.length(), vector<int>(k + 1, INT_MAX));
        
        for(int i = 0; i < s.length(); i++) {
            dp[i][1] = cal(s.substr(0, i + 1));
        }
        
        for(int t = 2; t <= k; t++) {
            for(int i = t - 1; i < s.length(); i++) {
                for(int j = t - 2; j < i; j++) {
                    dp[i][t] = min(dp[i][t], dp[j][t - 1] + cal(s.substr(j + 1, i - j)));
                }
            }
        }
        
        return dp[s.length() - 1][k];
    }
    

    
    int cal(string str) {
        int start = 0, end = str.length() - 1;
        
        int res = 0;
        
        while(start < end) {
            if(str[start] != str[end]) {
                res++;
            }
            start++;
            end--;
        }
        
        return res;
    }
    
};
