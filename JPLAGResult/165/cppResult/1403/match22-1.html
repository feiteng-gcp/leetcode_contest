class Solution {
public:
    int diffCount(string & s, int i, int j) {
        int count = 0;
        while (i < j) {
            if (s[i] != s[j]) count++;

            i++;
            j--;
        }
        return count;
    }
    int palindromePartition(string s, int K) {
        vector<vector<int> > m(K+1, vector<int>(s.size(), 0));               
        
        for (int i = 0; i < s.size(); i++) {
            m[1][i] = diffCount(s, 0, i);
            for (int k = 2; k <= K; k++) {
                if (k > (i+1)) break;
                m[k][i] = 10000;

                for (int j = 0; j < i; j++) {
                    m[k][i] = min(diffCount(s, j+1, i) + m[k-1][j], m[k][i]);
                }                
            }
            /*
            cout << i << ": ";
            for (int k = 1; k <= K; k++) {
                cout << m[k][i] << " ";
            }
            cout << endl;
            */
            
        }
        return m[K][s.size()-1];
        
    }
};
