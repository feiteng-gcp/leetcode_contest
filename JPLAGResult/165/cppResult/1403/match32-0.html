class Solution {
public:
    int palindromePartition(string s, int k) {
        const int n = s.size();
        vector<vector<int>> terms(n, vector<int>(n, 0));
        for (int index = 0; index < n; ++index) {
            for (int i = index - 1, j = index + 1; i >= 0 && j < n; --i, ++j) {
                terms[i][j] = terms[i + 1][j - 1] + ((s[i] == s[j]) ? 0 : 1);
            }
            if (index == n - 1) {
                continue;
            }
            terms[index][index + 1] = (s[index] == s[index + 1]) ? 0 : 1;
            for (int i = index - 1, j = index + 2; i >= 0 && j < n; --i, ++j) {
                terms[i][j] = terms[i + 1][j - 1] + ((s[i] == s[j]) ? 0 : 1);
            }
        }
        vector<vector<int>> dp(n, vector<int>(k));
        for (int i = 0; i < n; ++i) {
            dp[i][0] = terms[0][i];
            for (int j = 1; j < k && j <= i; ++j) {
                dp[i][j] = INT_MAX;
                for (int p = j - 1; p < i; ++p) {
                    dp[i][j] = min(dp[i][j], dp[p][j - 1] + terms[p + 1][i]);
                }
            }
        }
        return dp[n - 1][k - 1];
    }
};
