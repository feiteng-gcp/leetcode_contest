class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        int t[3][3] = {0};
        for (int i = 0; i < moves.size(); i++){
            t[moves[i][0]][moves[i][1]] = 1 + (i&1);
        }
        for (int i = 0; i < 3; i++){
            if (t[i][0]==t[i][1] && t[i][1]==t[i][2]){
                if (t[i][0]==1) return "A";
                else if (t[i][0]==2) return "B";
            }
            if (t[0][i]==t[1][i] && t[1][i]==t[2][i]){
                if (t[0][i]==1) return "A";
                else if (t[0][i]==2) return "B";
            }
        }
        if (t[0][0]==t[1][1] && t[1][1]==t[2][2]){
            if (t[0][0]==1) return "A";
            else if (t[0][0]==1) return "B";
        }
        if (t[0][2]==t[1][1] && t[1][1]==t[2][0]){
            if (t[1][1]==1) return "A";
            else if (t[1][1]==2) return "B";
        }
        
        if (moves.size()==9) return "Draw";
        
        return "Pending";
    }
};
