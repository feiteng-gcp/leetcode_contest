class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        unordered_map<int, unordered_map<int, int>> um;
        int p = 1;
        for (auto m : moves) {
            um[m[0]][m[1]] = p;
            p *= -1;
            
            for (int i = 0; i < 3; i++) {
                if (um[i][0] == um[i][1] && um[i][1] == um[i][2] && um[i][0] != 0) {
                    if (um[i][1] == 1)
                        return "A";
                    else
                        return "B";
                }
                if (um[0][i] == um[1][i] && um[1][i] == um[2][i] && um[0][i] != 0) {
                    if (um[1][i] == 1)
                        return "A";
                    else
                        return "B";
                }
            }
            if (um[1][1] == um[0][0] && um[0][0] == um[2][2] && um[0][0] != 0) {
                if (um[1][1] == 1)
                        return "A";
                    else
                        return "B";
            }
            if (um[1][1] == um[2][0] && um[0][2] == um[1][1] && um[1][1] != 0) {
                if (um[1][1] == 1)
                        return "A";
                    else
                        return "B";
            }
        }
        if (moves.size() == 9)
            return "Draw";
        else
            return "Pending";
    }
};
