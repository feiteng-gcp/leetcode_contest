class Solution {
public:
    int mp[3][3];

    int judge(int x, int y) {
        bool flag = true;
        for (int i = 0; i < 3; i++) {
            if (mp[i][y] != mp[x][y]) {
                flag &= false;
            }
        }
        if (flag)
            return true;
        flag = true;
        for (int i = 0; i < 3; i++) {
            if (mp[x][i] != mp[x][y]) {
                flag &= false;
            }
        }
        if (flag)
            return true;
        if (mp[0][0] == mp[1][1] && mp[0][0] == mp[2][2] &&
            mp[0][0] == mp[x][y])
            return true;
        if (mp[0][2] == mp[1][1] && mp[1][1] == mp[2][0] &&
            mp[1][1] == mp[x][y])
            return true;
        return false;
    }
    string tictactoe(vector<vector<int>> &moves) {
        int e = 1;
        for (auto s : moves) {
            int x = s[0];
            int y = s[1];
            mp[x][y] = e;
            if (judge(x, y)) {
                return e == 1 ? "A" : "B";
            }
            e = -e;
        }
        int count = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (mp[i][j] != 0) {
                    count++;
                }
            }
        }
        
        if (count == 9)
            return "Draw";
        return "Pending";
    }
};
