class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        int a[3][3];
        int x = 1;
        memset(a, 0, sizeof(a));
        for (auto v : moves) {
            a[v[0]][v[1]] = x;
            x ^= 3;
        }

        for (int i = 0; i < 3; i++) {
            if (a[i][0] == a[i][1] && a[i][1] == a[i][2] && a[i][2] != 0) return a[i][0] == 1 ? "A" : "B";
        }
        for (int i = 0; i < 3; i++) {
            if (a[0][i] == a[1][i] && a[1][i] == a[2][i] && a[2][i] != 0) return a[0][i] == 1 ? "A" : "B";
        }
        if (a[0][0] == a[1][1] && a[1][1] == a[2][2] && a[2][2] != 0) return a[0][0] == 1 ? "A" : "B";
        if (a[0][2] == a[1][1] && a[1][1] == a[2][0] && a[2][0] != 0) return a[0][2] == 1 ? "A" : "B";
        return moves.size() == 9 ? "Draw" : "Pending";
    }
};
