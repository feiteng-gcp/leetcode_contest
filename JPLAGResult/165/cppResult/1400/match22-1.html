class Solution {
public:
    int sav[3][3];
    
    bool check(int x){
        for (int i = 0; i < 3; i++){
            bool flag = true;
            for (int j = 0; j < 3; j++){
                if (sav[i][j] != x) flag = false;
            }
            if (flag) return true;
        }
        
        for (int j = 0; j < 3; j++){

            bool flag = true;
            for (int i = 0; i < 3; i++){
                if (sav[i][j] != x) flag = false;
            }
            if (flag) return true;
        }
        
        if (sav[0][0] == sav[1][1] && sav[1][1] == sav[2][2] && sav[1][1] == x) return true;
        // printf("%d %d %d\n", sav[0][2], sav[1][1], sav[2][0]);
        if (sav[2][0] == sav[1][1] && sav[1][1] == sav[0][2] && sav[1][1] == x) return true;
        
        return false;
    }
    
    string tictactoe(vector<vector<int>>& moves) {
        memset(sav, 0, sizeof(sav));
        
        int cur = 0;
        for (auto V: moves){
            int x = V[0], y = V[1];
            sav[x][y] = cur % 2 + 1;
            // printf("%d %d %d\n", cur % 2 + 1, x, y);
            if (check(cur % 2 + 1)) {
                if (cur % 2 == 0) return "A"; else return "B";
            }
            ++cur;
        }
        
        if (cur == 9) return "Draw";
        return "Pending";
    }
};
