class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        vector<string> t(3, "   ");
        int id = 0;
        for (auto p : moves) {
            if(id) {
                t[p[0]][p[1]] = 'O';
            }
            else t[p[0]][p[1]] = 'X';
            id = 1 - id;
        }
        for (int i = 0; i < 3; i++) {
            int o = 0, x= 0;
            for (int j = 0; j < 3; j++) {
                if(t[i][j] == 'O') o++;
                if(t[i][j] == 'X') x++;
            }
            if(o == 3 ) return "B";
            if(x == 3) return "A";
        }
        for (int i = 0; i < 3; i++) {
            int o = 0, x= 0;
            for (int j = 0; j < 3; j++) {
                if(t[j][i] == 'O') o++;
                if(t[j][i] == 'X') x++;
            }
            if(o == 3 ) return "B";
            if(x == 3) return "A";
        }
        int o = 0, x= 0;
        for (int i = 0; i < 3; i++) {
            if(t[i][i] == 'O') o++;
            if(t[i][i] == 'X') x++;
        }
        
        if(o == 3 ) return "B";
        if(x == 3) return "A";
        o = 0, x= 0;
        for (int i = 0; i < 3; i++) {
            if(t[i][2 - i] == 'O') o++;
            if(t[i][2 - i] == 'X') x++;
        }
        if(o == 3 ) return "B";
        if(x == 3) return "A";
        if(moves.size() == 9) return "Draw";
        return "Pending";
    }
};
