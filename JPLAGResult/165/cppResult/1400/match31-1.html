class Solution {

    int check(vector<vector<int>>& grid){
        int cnt=0;
        for(int i=0;i<3;i++){
            for(int j=0;j<3;j++){
                if(grid[i][j]!=0) cnt++;
            }
        }
        

        for(int i=0;i<3;i++){
            bool f=true;
            for(int j=0;j<3;j++){
                if(grid[i][j]!=1){
                    f=false;
                    break;
                };
            }
            if(f){
                return 1;
            }
        }
        
        for(int i=0;i<3;i++){

            bool f=true;
            for(int j=0;j<3;j++){
                if(grid[j][i]!=1){
                    f=false;
                    break;
                };
            }
            if(f){
                return 1;
            }
        }
        
        
            bool f=true;
            for(int j=0;j<3;j++){
                if(grid[j][j]!=1){
                    f=false;
                    break;
                };
            }
            if(f){

                return 1;
            }
        
         f=true;
            for(int j=0;j<3;j++){
                if(grid[j][2-j]!=1){
                    f=false;
                    break;
                };
            }
            if(f){

                return 1;
            }
        
        for(int i=0;i<3;i++){
            bool f=true;
            for(int j=0;j<3;j++){
                if(grid[i][j]!=2){
                    f=false;
                    break;
                };
            }
            if(f){
                return 2;
            }
        }
        
        for(int i=0;i<3;i++){
            bool f=true;
            for(int j=0;j<3;j++){
                if(grid[j][i]!=2){
                    f=false;
                    break;
                };
            }
            if(f){
                return 2;
            }
        }
        
        f=true;
        for(int j=0;j<3;j++){
                if(grid[j][j]!=2){
                    f=false;
                    break;
                };
            }
            if(f){
                return 2;
            }
        
         f=true;
            for(int j=0;j<3;j++){
                if(grid[j][2-j]!=2){
                    f=false;
                    break;
                };
            }
            if(f){
                return 2;
            }
        
        if(cnt==9) return 3;
        else return 0;
    }
public:
    string tictactoe(vector<vector<int>>& moves) {
        vector<vector<int>> grid(3,vector<int>(3,0));
        bool flag=true;
        for(auto m:moves){
            if(flag){
                grid[m[0]][m[1]]=1;
            }else{
                grid[m[0]][m[1]]=2;
            }
            flag=!flag;
            int r=check(grid);
            if(r!=0){
                if(r==1) return "A";
                else if(r==2) return "B";
                else if(r==3) return "Draw";
            }
        }
        return "Pending";
    }
};
