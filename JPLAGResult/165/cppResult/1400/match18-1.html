class Solution {
public:
    int f[3][3];
    

    bool win(int t) {
        for (int i = 0; i < 3; i ++) {
            if (f[i][0] == t && f[i][1] == t && f[i][2] == t) return true;
            if (f[0][i] == t && f[1][i] == t && f[2][i] == t) return true;
        }
        if (f[0][0] == t && f[1][1] == t && f[2][2] == t) return true;
        if (f[0][2] == t && f[1][1] == t && f[2][0] == t) return true;
        return false;
    }
    

    string tictactoe(vector<vector<int>>& moves) {
        int n = moves.size();
        memset(f, -1, sizeof(f));
        for (int i = 0; i < n; i ++) {
            int o = i % 2;
            f[moves[i][0]][moves[i][1]] = o;
            if (win(o)) return o == 0 ? "A" : "B";
        }
        return n == 9 ? "Draw" : "Pending";
    }
};
