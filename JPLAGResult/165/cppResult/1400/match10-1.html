class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        string ret;
        vector<vector<int>> board(3,vector<int>(3,0));
        int i,n=moves.size();
        for(i=0;i<n;++i){
            board[moves[i][0]][moves[i][1]] = i%2+1;
        }
        // check win 1

        for(i=0;i<3;++i){
            if(board[i][0]==board[i][1]&&board[i][1]==board[i][2]){
                if(board[i][0]==1)
                    return "A";
                else if(board[i][0]==2)
                    return "B";
            }
            if(board[0][i]==board[1][i]&&board[1][i]==board[2][i]){
                if(board[0][i]==1)
                    return "A";
                else if(board[0][i]==2)
                    return "B";
            }
        }
        if(board[0][0]==board[1][1]&&board[1][1]==board[2][2]){
            if(board[0][0]==1)
                return "A";
            else if(board[0][0]==2)
                return "B";
        }
        if(board[0][2]==board[1][1]&&board[1][1]==board[2][0]){
            if(board[0][2]==1)
                return "A";
            else if(board[0][2]==2)
                return "B";
        }
        if(n==9)
            return "Draw";
        return "Pending";
    }
};
