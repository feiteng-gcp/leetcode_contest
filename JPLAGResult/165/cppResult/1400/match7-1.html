class Solution {
public:
    char ditu[3][3];

    bool judge1(){
        if(ditu[0][0]=='X' && ditu[0][1]=='X' && ditu[0][2]=='X') return true;
        if(ditu[1][0]=='X' && ditu[1][1]=='X' && ditu[1][2]=='X') return true;
        if(ditu[2][0]=='X' && ditu[2][1]=='X' && ditu[2][2]=='X') return true;
        
        if(ditu[0][0]=='X' && ditu[1][0]=='X' && ditu[2][0]=='X') return true;
        if(ditu[0][1]=='X' && ditu[1][1]=='X' && ditu[2][1]=='X') return true;
        if(ditu[0][2]=='X' && ditu[1][2]=='X' && ditu[2][2]=='X') return true;
        
        if(ditu[0][0]=='X' && ditu[1][1]=='X' && ditu[2][2]=='X') return true;
        if(ditu[0][2]=='X' && ditu[1][1]=='X' && ditu[2][0]=='X') return true;
        return false;
    }
    bool judge2(){
        if(ditu[0][0]=='O' && ditu[0][1]=='O' && ditu[0][2]=='O') return true;
        if(ditu[1][0]=='O' && ditu[1][1]=='O' && ditu[1][2]=='O') return true;
        if(ditu[2][0]=='O' && ditu[2][1]=='O' && ditu[2][2]=='O') return true;
        
        if(ditu[0][0]=='O' && ditu[1][0]=='O' && ditu[2][0]=='O') return true;
        if(ditu[0][1]=='O' && ditu[1][1]=='O' && ditu[2][1]=='O') return true;
        if(ditu[0][2]=='O' && ditu[1][2]=='O' && ditu[2][2]=='O') return true;
        
        if(ditu[0][0]=='O' && ditu[1][1]=='O' && ditu[2][2]=='O') return true;
        if(ditu[0][2]=='O' && ditu[1][1]=='O' && ditu[2][0]=='O') return true;
        return false;
    }
    string tictactoe(vector<vector<int>>& moves) {
        for(int i=0;i<moves.size();i++){

            if(i%2==0){
                ditu[moves[i][0]][moves[i][1]]='X';
                if(judge1()) return "A";
            }else{
                ditu[moves[i][0]][moves[i][1]]='O';
                if(judge2()) return "B";
            }
        }
        if(moves.size()<9) return "Pending";
        return "Draw";
    }
};
