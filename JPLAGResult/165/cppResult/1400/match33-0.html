class Solution {
public:
    int a[3][3];
    
    string checkWinner(){
        // horizontal
        
        for(int i = 0 ; i < 3;i ++){
            if(a[i][0] == a[i][1] && a[i][1] == a[i][2]){
                if(a[i][0] == 0) return "B";
                else if(a[i][0] == 1) return "A";
            }
        }
        
        for(int i = 0 ; i < 3;i ++){
            if(a[0][i] == a[1][i] && a[1][i] == a[2][i]){
                if(a[0][i] == 0) return "B";
                
                else if(a[0][i] == 1) return "A";
            }
        }
        
        if(a[0][0] == a[1][1] && a[1][1] == a[2][2]){
            if(a[0][0] == 0) return "B";
            else if(a[0][0] == 1) return "A";
        }
        
        if(a[0][2] == a[1][1] && a[1][1] == a[2][0]){
            if(a[0][2] == 0) return "B";
            else if(a[0][2] == 1) return "A";
        }
        
        for(int i = 0 ; i < 3;i ++){
            for(int j = 0 ; j < 3; j ++){
                if(a[i][j] == -1)return "Pending";
            }
        }
        
        return "Draw";
        
    }
    
    string tictactoe(vector<vector<int>>& moves) {
        for(int i = 0 ; i < 3; i ++){
            for(int j = 0 ; j < 3; j ++){
                a[i][j] = -1;
            }
        }
        
        for(int i = 0 ; i < moves.size(); i ++){
            if(i&1){
                a[moves[i][0]][moves[i][1]] = 0;
            }else{
                a[moves[i][0]][moves[i][1]] = 1;
            }
        }
        
        return checkWinner();
        
    }
};
