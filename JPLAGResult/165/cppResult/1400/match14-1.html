class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        vector<vector<int>> bo(3, vector<int>(3));
        for (int i = 0; i < moves.size(); i++) {
            if (i % 2 == 0) {
                bo[moves[i][0]][moves[i][1]] = 1;
            }
            else {
                bo[moves[i][0]][moves[i][1]] = 2;
            }

        }
        for (int i = 0; i < 3; i++) {
            int a = 0, b = 0;
            for (int j = 0; j < 3; j++) {
                if (bo[i][j] == 1) a++;

                else if (bo[i][j] == 2) b++;
            }
            if (a == 3) return "A";
            if (b == 3) return "B";
        }
        for (int i = 0; i < 3; i++) {
            int a = 0, b = 0;
            for (int j = 0; j < 3; j++) {
                if (bo[j][i] == 1) a++;

                else if (bo[j][i] == 2) b++;
            }
            if (a == 3) return "A";
            if (b == 3) return "B";
        }
        
        int a = 0, b = 0;
        for (int j = 0; j < 3; j++) {
            if (bo[j][j] == 1) a++;

            else if (bo[j][j] == 2) b++;
        }
        if (a == 3) return "A";
        if (b == 3) return "B";

        a = 0, b = 0;
        for (int j = 0; j < 3; j++) {
            if (bo[j][2 - j] == 1) a++;
            else if (bo[j][2 - j] == 2) b++;
        }
        if (a == 3) return "A";
        if (b == 3) return "B";
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (bo[i][j] == 0) return "Pending";
            }
        }
        return "Draw";
    }
};
