class Solution {
public:
    bool check(int x,int y,vector<vector<char>> &t){
        bool f = true;
        for(int j=0;j<3;j++){
            if(t[x][j]!=t[x][y]){
                f = false;
                break;
            }
        }
        if(f) return f;
        f = true;
        for(int j=0;j<3;j++){
            if(t[j][y]!=t[x][y]){
                f = false;
                break;
            }
        }
        if(f) return f;
        f = true;
        if(x==y){
            for(int j=0;j<3;j++){
                if(t[j][j]!=t[x][y]){
                    f = false;
                    break;
                }
            }
            if(f) return f;
        }
        f = true;
        if((x+y) == 2){
            for(int j=0;j<3;j++){
                if(t[j][2-j]!=t[x][y]){
                    f = false;
                    break;
                }
            }
            if(f) return f;
        }
        return false;
    }
    string tictactoe(vector<vector<int>>& moves) {
        vector<vector<char>> t(3,vector<char>(3,' '));
        int n = moves.size();
        for(int i=0;i<n;i++){
            int x = moves[i][0];
            int y = moves[i][1];
            if(i%2) t[x][y]='O';
            else t[x][y] = 'X';
            // cout<<t[x][y]<<endl;
            if(check(x,y,t)) {
                if(i%2) return "B";
                return "A";
            }
        }
        if(n<9) return "Pending";
        return "Draw";
    }
};
