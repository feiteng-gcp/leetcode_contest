class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        int g[4][4] ;
        memset(g , 0 , sizeof(g)) ;
        for (int i = 0 ; i < moves.size() ; ++i) {
            if (i % 2 == 0) {
                g[moves[i][0]][moves[i][1]] = 1 ;
            } else g[moves[i][0]][moves[i][1]] = 2 ;
        }
        int res = 0 ;
        //checkA
        for (int i = 0 ; i < 3 ; ++i) {
            int flag = 1 ;
            for (int j = 0 ; j < 3 ; ++j) {
                if (g[i][j] != 1) {
                    flag = 0 ; break ;
                }
            }
            if (flag) {
                return "A" ;
            }
        }
        for (int i = 0 ; i < 3 ; ++i) {
            int flag = 1 ;
            for (int j = 0 ; j < 3 ; ++j) {
                if (g[j][i] != 1) {
                    flag = 0 ; break ;
                }
            }
            if (flag) {
               return "A" ; 
            }
        }
        if (g[0][0] == 1 && g[1][1] == 1 && g[2][2] == 1) {
            return "A" ;
        }
        if (g[0][2] == 1 && g[1][1] == 1 && g[2][0] == 1) {
            return "A" ;
        }
        //checkB
        for (int i = 0 ; i < 3 ; ++i) {
            int flag = 1 ;
            for (int j = 0 ; j < 3 ; ++j) {
                if (g[i][j] != 2) {
                    flag = 0 ; break ;
                }
            }
            if (flag) {
                return "B" ;
            }
        }
        for (int i = 0 ; i < 3 ; ++i) {
            int flag = 1 ;
            for (int j = 0 ; j < 3 ; ++j) {
                if (g[j][i] != 2) {
                    flag = 0 ; break ;
                }
            }
            if (flag) {
               return "B" ; 
            }
        }
        if (g[0][0] == 2 && g[1][1] == 2 && g[2][2] == 2) {
            return "B" ;
        }
        if (g[0][2] == 2 && g[1][1] == 2 && g[2][0] == 2) {
            return "B" ;
        }
        if (moves.size() < 9) {
            return "Pending" ;
        }
        return "Draw" ;
    }
};
