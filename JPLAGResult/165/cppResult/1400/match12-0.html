int a[3][3];
class Solution {
private:
public:
    string tictactoe(vector<vector<int>>& b) {
        memset(a,-1,sizeof(a));
        for(int i=0;i<b.size();i++){
            a[b[i][0]][b[i][1]] = (i&1);
        }
        for(int i=0;i<3;i++){
            if(a[i][0]==a[i][1]&&a[i][1]==a[i][2]&&a[i][0]!=-1)return a[i][0]==0?"A":"B";
        }
        for(int i=0;i<3;i++){
            if(a[0][i]==a[1][i]&&a[1][i]==a[2][i]&&a[0][i]!=-1)return a[0][i]==0?"A":"B";
        }
        if(a[0][0]==a[1][1]&&a[1][1]==a[2][2]&&a[2][2]!=-1)return a[0][0]==0?"A":"B";
        if(a[0][2]==a[1][1]&&a[1][1]==a[2][0]&&a[2][0]!=-1)return a[0][2]==0?"A":"B";
        return b.size()==9?"Draw":"Pending";
    }
};
