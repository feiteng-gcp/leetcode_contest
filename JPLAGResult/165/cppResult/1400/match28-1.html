class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        vector<vector<int>> grid(3, vector<int>(3, -1));
        for (int i = 0; i < moves.size(); ++i) {
            int x = moves[i][0], y = moves[i][1];
            grid[x][y] = i % 2 == 0 ? 0 : 1;
            if (check(grid, x, y)) {
                return i % 2 == 0 ? "A" : "B";
            }
        }
        return moves.size() == 9 ? "Draw" : "Pending";
    }
private:
    bool check(vector<vector<int>>& grid, int x, int y) {
        int s = grid[x][y];

        bool valid = true;
        for (int i = 0; i < 3; ++i) {
            if (grid[i][y] != s) {
                valid = false;
                break;
            }
        }
        if (valid) {
            return true;
        }

        valid = true;
        for (int i = 0; i < 3; ++i) {
            if (grid[x][i] != s) {
                valid = false;
                break;
            }
        }
        if (valid) {
            return true;
        }
        if (x == y) {

            valid = true;
            for (int i = 0; i < 3; ++i) {
                if (grid[i][i] != s) {
                    valid = false;
                    break;
                }
            }
            if (valid) {
                return true;
            }
        }
        
        if (x + y == 2) {

            valid = true;
            for (int i = 0; i < 3; ++i) {
                if (grid[i][2 - i] != s) {
                    valid = false;
                    break;
                }
            }
            if (valid) {
                return true;
            }
        }
        return false;
    }
};
