class Solution {
public:
    int check(vector<int> &a)
    {
        for(int i = 0; i < 3; i++)
        {
            if (a[i*3] == a[i*3+1] && a[i*3+1] == a[i*3+2]) return a[i*3];
            if (a[i] == a[3+i] && a[3+i] == a[6+i]) return a[i];
        }
        if (a[0] == a[4] && a[4] == a[8]) return a[0];
        if (a[2] == a[4] && a[4] == a[6]) return a[2];
        return 0;
    }
    
    string tictactoe(vector<vector<int>>& moves) {
        vector<int> a(9, 0);
        for(int i = 0; i < moves.size(); i++)
            a[moves[i][0]*3+moves[i][1]] = (i%2?2:1);
        int c = check(a);
        if (c == 1) return string("A");
        if (c == 2) return string("B");
        if (c == 0 && moves.size() == 9) return string("Draw");
        return string("Pending");
    }
};
