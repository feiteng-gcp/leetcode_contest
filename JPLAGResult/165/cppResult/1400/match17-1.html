class Solution {
public:

    bool check(vector<vector<int>> &res){
        for(int i = 0; i < res.size(); i++){
            if(res[i][0] == res[i][1] && res[i][1] == res[i][2] && res[i][1] != -1){
                return true;
            }
        }
        for(int i = 0; i < res.size(); i++){
            if(res[0][i] == res[1][i] && res[1][i] == res[2][i]  && res[1][i] != -1){
                return true;
            }
        }
        if(res[0][0] == res[1][1] && res[1][1] == res[2][2] && res[1][1] != -1){
            return true;
        }
        if(res[2][0] == res[1][1] && res[1][1] == res[0][2] && res[1][1] != -1){
            return true;
        }
        return false;
    }
    
    string tictactoe(vector<vector<int>>& moves) {
        vector<vector<int>> res(3, vector<int>(3, -1));
        for(int i = 0; i < moves.size(); i++){
            res[moves[i][0]][moves[i][1]] = (i % 2);
            // cout<<i<<endl;
            if(check(res)){
                if(i % 2){
                    return "B";

                }else{
                    return "A";
                }
            }
        }
        if(moves.size() == 9){
            return "Draw";
        }
        return "Pending";
    }
};
