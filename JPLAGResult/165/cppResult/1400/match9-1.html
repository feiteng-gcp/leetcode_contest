class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        int d[3][3] = {0};
        int player = 1;
        for (auto &p : moves) {
            d[p[0]][p[1]] = player;
            player = player == 1 ? 2 : 1;
        }
        
        int winner = 0;
        // check lines
        for (int i=0; i<3; ++i) {

            winner = 1;
            for (int j=0; j<3; ++j) {
                if (d[i][j] != winner) {
                    winner = 0;
                    break;
                }
            }
            if (winner == 1) {
                return "A";
            }
            

            winner = 2;
            for (int j=0; j<3; ++j) {
                if (d[i][j] != winner) {
                    winner = 0;
                    break;
                }
            }
            if (winner == 2) {
                return "B";
            }
        }
        
        // check columns
        for (int i=0; i<3; ++i) {

            winner = 1;
            for (int j=0; j<3; ++j) {
                if (d[j][i] != winner) {
                    winner = 0;
                    break;
                }
            }
            if (winner == 1) {
                return "A";
            }
            

            winner = 2;
            for (int j=0; j<3; ++j) {
                if (d[j][i] != winner) {
                    winner = 0;
                    break;
                }
            }
            if (winner == 2) {
                return "B";
            }
        }
        
        // check diagonals
        winner = 1;
        for (int i=0; i<3; ++i) {
            if (d[i][i] != winner) {
                winner = 0;
                break;
            }
        }
        if (winner == 1) {
            return "A";
        }
        winner = 2;
        for (int i=0; i<3; ++i) {
            if (d[i][i] != winner) {
                winner = 0;
                break;
            }
        }
        if (winner == 2) {
            return "B";
        }
        
        winner = 1;
        for (int i=0; i<3; ++i) {
            if (d[2-i][i] != winner) {
                winner = 0;
                break;
            }
        }
        if (winner == 1) {
            return "A";
        }
        winner = 2;
        for (int i=0; i<3; ++i) {
            if (d[2-i][i] != winner) {
                winner = 0;
                break;
            }

        }
        if (winner == 2) {
            return "B";
        }
        
        if (moves.size() < 9) {
            return "Pending";
        }
        return "Draw";
    }
};
