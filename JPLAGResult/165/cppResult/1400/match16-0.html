class Solution {
public:
    bool Check(int b[3][3], int v) {
        for (int i = 0; i < 3; i++) {
            bool ok = true;
            for (int j = 0; j < 3; j++) {
                if (b[i][j] != v) ok = false;
            }
            if (ok) return true;
            
            ok = true;
            for (int j = 0; j < 3; j++) {
                if (b[j][i] != v) ok = false;
            }
            if (ok) return true;
        }
        if (b[0][0] == v && b[1][1] == v && b[2][2] == v) return true;
        if (b[0][2] == v && b[1][1] == v && b[2][0] == v) return true;
        return false;
    }
    string tictactoe(vector<vector<int>>& m) {
        int b[3][3];
        memset(b, 0, sizeof(b));
        for (int i = 0; i < m.size(); i++){
            if (i % 2)
                b[m[i][0]][m[i][1]] = 2;
            else
                b[m[i][0]][m[i][1]] = 1;
        }
        
        if (Check(b, 1)) return "A";
        if (Check(b, 2)) return "B";
        if (m.size() == 9) return "Draw";
        return "Pending";
        
    }
};
