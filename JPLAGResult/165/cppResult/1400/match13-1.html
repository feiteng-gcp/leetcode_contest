class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        int f[3][3];
        memset(f, 0, sizeof(f));
        int s = 0;
        for (auto each : moves)
        {
            s++;
            f[each[0]][each[1]] = s % 2 + 1;

            bool a = true, b= true;
            for (int i = 0; i < 3; ++i)
            {
                if (f[each[0]][i] != 2) a = false;
                if (f[each[0]][i] != 1) b = false;
            }
            if (a) return "A";
            if (b) return "B";
            a = true; b= true;
            for (int i = 0; i < 3; ++i)
            {
                if (f[i][each[1]] != 2) a = false;
                if (f[i][each[1]] != 1) b = false;
            }
            if (a) return "A";
            if (b) return "B";
            if (each[0] == each[1])
            {

                a = true; b= true;
                for (int i = 0; i < 3; ++i)
                {
                    if (f[i][i] != 2) a = false;
                    if (f[i][i] != 1) b = false;
                }
                if (a) return "A";
                if (b) return "B";
            }
            
            if (each[0] + each[1] == 2)
            {

                a = true; b= true;
                for (int i = 0; i < 3; ++i)
                {
                    if (f[i][2 - i] != 2) a = false;
                    if (f[i][2 - i] != 1) b = false;
                }
                if (a) return "A";
                if (b) return "B";
            }
        }
        if (s == 9) return "Draw";
        return "Pending";
    }
};
