class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        vector<vector<int>> board(3, vector<int>(3, -1));
        int N = moves.size();
        for (int i = 0; i < N; ++i) {
            board[moves[i][0]][moves[i][1]] = i % 2;
        }
        for (int i = 0; i < 3; ++i) {
            bool A = true, B = true;
            for (int j = 0; j < 3; ++j) {
                if (board[i][j] != 0) {
                    A = false;
                }
                if (board[i][j] != 1) {
                    B = false;
                }
            }
            if (A) {
                return "A";
            }
            if (B) {
                return "B";
            }
        }
        for (int j = 0; j < 3; ++j) {
            bool A = true, B = true;
            for (int i = 0; i < 3; ++i) {
                if (board[i][j] != 0) {
                    A = false;
                }
                if (board[i][j] != 1) {
                    B = false;
                }
            }
            if (A) {
                return "A";
            }
            if (B) {
                return "B";
            }
        }
        bool A, B;
        A = true;
        B = true;
        for (int i = 0; i < 3; ++i) {
            if (board[i][2 - i] != 0) {
                A = false;
            }
            if (board[i][2 - i] != 1) {
                B = false;
            }
        }
        if (A) {
            return "A";
        }
        if (B) {
            return "B";
        }
        A = true;
        B = true;
        for (int i = 0; i < 3; ++i) {
            if (board[i][i] != 0) {
                A = false;
            }
            if (board[i][i] != 1) {
                B = false;
            }
        }
        if (A) {
            return "A";
        }
        if (B) {
            return "B";
        }
        if (N == 9) {
            return "Draw";
        }
        return "Pending";
    }
};
