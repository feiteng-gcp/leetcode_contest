class Solution {
public:
	vector<vector<int>> field;
	
	bool check(int x){
		for(int i = 0; i < 3; i++){
			bool flag = true;
			for(int j = 0; j < 3; j++){
				if(field[i][j] != x) flag = false;
			}
			
			if(flag == true) return true;
		}
		
		for(int i = 0; i < 3; i++){
			bool flag = true;
			for(int j = 0; j < 3; j++){
				if(field[j][i] != x) flag = false;
			}
			
			if(flag == true) return true;
		}
		
		if(field[0][0] == x && field[0][0] == field[1][1] && field[0][0] == field[2][2]) return true;
		if(field[0][2] == x && field[0][2] == field[1][1] && field[0][2] == field[2][0]) return true;
		
		return false;
	}
	
    string tictactoe(vector<vector<int>>& moves) {
		field.resize(3, vector<int>(3, 0));
		
        for(int i = 0; i < moves.size(); i++){
			field[moves[i][0]][moves[i][1]] = i%2 + 1;
		}
		
		if(check(1)) return "A";
		if(check(2)) return "B";
		
		if(moves.size() == 9) return "Draw";
		
		else return "Pending";
    }
};
