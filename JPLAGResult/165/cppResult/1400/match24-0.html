class Solution {
public:
    char mp[3][3];
    string tictactoe(vector<vector<int>>& moves) {
        memset(mp, 0, sizeof mp);
        for (int i = 0; i < moves.size(); ++i) {
            int x = moves[i][0], y = moves[i][1];
            mp[x][y] = (i % 2) + 1;
        }
        for (int i = 0; i < 3; ++i) {
            if (mp[i][0] > 0 && mp[i][0] == mp[i][1] && mp[i][0] == mp[i][2]) {
                if (mp[i][0] == 1) return string("A");
                else return string("B");
            }
            if (mp[0][i] > 0 && mp[0][i] == mp[1][i] && mp[0][i] == mp[2][i]) {
                if (mp[0][i] == 1) return string("A");
                else return string("B");
            }
        }
        if (mp[1][1] > 0 && mp[1][1] == mp[0][0] && mp[2][2] == mp[0][0] ||
           mp[1][1] > 0 && mp[1][1] == mp[2][0] && mp[0][2] == mp[2][0]) {
            if (mp[1][1] == 1) return string("A");
            else return string("B");
        }
        if (moves.size() == 9) return string("Draw");
        return string("Pending");
    }
};
