class Solution {
public:
    bool judge(vector<vector<char>>& grid, char sym) {
        for (int i = 0; i < 3; i++) {
            if (grid[i][0] == sym && grid[i][1] == sym && grid[i][2] == sym) {
                return true;
            }
        }
        for (int i = 0; i < 3; i++) {
            if (grid[0][i] == sym && grid[1][i] == sym && grid[2][i] == sym) {
                return true;
            }
        }
        if (grid[0][0] == sym && grid[1][1] == sym && grid[2][2] == sym) {
            return true;
        }
        if (grid[0][2] == sym && grid[1][1] == sym && grid[2][0] == sym) {
            return true;
        }
        return false;
    }
    string tictactoe(vector<vector<int>>& moves) {
        vector<vector<char>> grid(3, vector<char>(3, ' '));
        for (int i = 0; i < moves.size(); i++) {
            if (i % 2 == 0) {
                grid[moves[i][0]][moves[i][1]] = 'X';
                if (judge(grid, 'X')) {
                    return "A";
                }
            } else {
                grid[moves[i][0]][moves[i][1]] = 'O';
                if (judge(grid, 'O')) {
                    return "B";
                }
            }
        }
        if (moves.size() == 9) {
            return "Draw";
        }
        return "Pending";
    }
};
