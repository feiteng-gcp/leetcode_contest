class Solution {
public:
    int dp[4][4], B[4][4] ;
    bool check(){
        if(dp[0][0] == 1 and dp[0][1] == 1 and dp[0][2] == 1)return 1 ;
        if(dp[1][0] == 1 and dp[1][1] == 1 and dp[1][2] == 1)return 1 ;
        if(dp[2][0] == 1 and dp[2][1] == 1 and dp[2][2] == 1)return 1 ;
        if(dp[0][0] == 1 and dp[1][0] == 1 and dp[2][0] == 1)return 1 ;
        if(dp[0][1] == 1 and dp[1][1] == 1 and dp[2][1] == 1)return 1 ;
        if(dp[0][2] == 1 and dp[1][2] == 1 and dp[2][2] == 1)return 1 ;
        if(dp[0][0] == 1 and dp[1][1] == 1 and dp[2][2] == 1)return 1 ;
        if(dp[0][2] == 1 and dp[1][1] == 1 and dp[2][0] == 1)return 1 ;
        return 0 ;
    }
    bool checkB(){
        if(B[0][0] == 1 and B[0][1] == 1 and B[0][2] == 1)return 1 ;
        if(B[1][0] == 1 and B[1][1] == 1 and B[1][2] == 1)return 1 ;
        if(B[2][0] == 1 and B[2][1] == 1 and B[2][2] == 1)return 1 ;
        if(B[0][0] == 1 and B[1][0] == 1 and B[2][0] == 1)return 1 ;
        if(B[0][1] == 1 and B[1][1] == 1 and B[2][1] == 1)return 1 ;
        if(B[0][2] == 1 and B[1][2] == 1 and B[2][2] == 1)return 1 ;
        if(B[0][0] == 1 and B[1][1] == 1 and B[2][2] == 1)return 1 ;
        if(B[0][2] == 1 and B[1][1] == 1 and B[2][0] == 1)return 1 ;
        return 0 ;
    }
    string tictactoe(vector<vector<int>>& moves) {
        int n = moves.size() ;
        for(int i = 0 ; i < n ; ++i){
            int x = moves[i][0], y = moves[i][1] ;
            if(i % 2 == 0){
                dp[x][y] = 1 ;
                if(check())return "A" ;
            }
            else {
                B[x][y] = 1 ;
                if(checkB())return "B" ;
            }
        }
        if(n < 9)return "Pending" ;
        else return "Draw" ;
    }
};
