int c[3][3];
class Solution {
public:
    int check()
    {
        int res=-1;
        for(int f=0;f<2;f++)
        {

            for(int i=0;i<3;i++)
            {
                res=f;
                for(int j=0;j<3;j++)
                {
                    if(c[i][j]!=f)
                    {
                        res=-1;
                        break;
                    }
                }
                if(res!=-1)return res;

            }
            for(int j=0;j<3;j++)
            {
                res=f;
                for(int i=0;i<3;i++)
                {
                    if(c[i][j]!=f)
                    {
                        res=-1;
                        break;
                    }
                }

                if(res!=-1)return res;
            }
            res=f;
            for(int i=0;i<3;i++)
            {
                if(c[i][i]!=f)
                {
                    res=-1;
                    break;
                }
            }
            if(res!=-1)return res;

            res=f;
            for(int i=0;i<3;i++)
            {
                if(c[i][2-i]!=f)
                {
                    res=-1;
                    break;
                }
            }
            if(res!=-1)return res;
        }
        return res;
    }
    string tictactoe(vector<vector<int>>& moves) {
        memset(c,-1,sizeof(c));
        int n=moves.size();
        for(int i=0;i<n;i++)
        {
            c[moves[i][0]][moves[i][1]]=i%2;
            int res=check();
            if(res!=-1)return res==0?"A":"B";
        }
        return n==9?"Draw":"Pending";
    }
};
