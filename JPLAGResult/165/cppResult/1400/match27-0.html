#include <bits/stdc++.h>
using namespace std;

class Solution {
public:

    string tictactoe(vector<vector<int>>& moves) {
        vector<vector<int>> G(3, vector<int>(3, -1));
        int pla = 0;
        for (auto& m : moves) {
        	G[m[0]][m[1]] = pla;
        	for (int i = 0; i < 3; ++i) {
        		bool all = 1;
        		for (int j = 0; j < 3; ++j) {
        			if (G[i][j]!=pla) {
        				all = false;
        				break;
        			}
        		}
        		if (all) return pla ? "B" : "A";
        		all = 1;
        		for (int j = 0; j < 3; ++j) {
        			if (G[j][i]!=pla) {
        				all = false;
        				break;
        			}
        		}
        		if (all) return pla ? "B" : "A";
        	}
        	if (G[0][0] == pla && G[1][1] == pla && G[2][2] == pla)
        		return pla ? "B" : "A";
        	if (G[0][2] == pla && G[1][1] == pla && G[2][0] == pla)
        		return pla ? "B" : "A";
        	pla = !pla;
        }
        return moves.size() == 9 ? "Draw" : "Pending";
    }
};
