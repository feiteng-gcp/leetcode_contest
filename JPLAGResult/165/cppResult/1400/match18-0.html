class Solution {
public:
    int g[3][3];
    string tictactoe(vector<vector<int>>& moves) {
        int n = moves.size();
        memset(g, 0, sizeof(g));
        
        for (int i = 0; i < n; ++i) {
            int cur = i % 2 == 0 ? 1 : 2;
            g[moves[i][0]][moves[i][1]] = cur;
            
            int w = findWinner();
            if (w != -1)
                return w == 1 ? "A" : "B";
        }
        
        return n == 9 ? "Draw" : "Pending";
    }
    
    int findWinner() {
        for (int i = 0; i < 3; ++i) {
            if (g[i][0] != 0 && g[i][1] == g[i][0] && g[i][2] == g[i][0])
                return g[i][0];
            if (g[0][i] != 0 && g[1][i] == g[0][i] && g[2][i] == g[0][i])
                return g[0][i];
        }
        
        if (g[0][0] != 0 && g[1][1] == g[0][0] && g[2][2] == g[0][0])
            return g[0][0];
        
        if (g[2][0] != 0 && g[1][1] == g[2][0] && g[0][2] == g[2][0])
            return g[2][0];
        
        return -1;
    }
};
