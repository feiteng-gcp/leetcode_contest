class Solution {
public:
    string tictactoe(vector<vector<int>>& moves) {
        vector<vector<int>> board(3, vector<int>(3,0));
        int turn = 1;
        for(auto m: moves) {
            board[m[0]][m[1]] = turn;
            
            /*
            for(int r=0;r<3;++r) {
                for(int c=0;c<3;++c) {
                    cout << board[r][c] << " ";
                }
                cout << endl;
            }
            cout << endl << endl;
*/

            bool win = false;
            for(int r=0;r<3;++r) {
                bool w = true;
                for(int c=0;c<3;++c) {
                    if(board[r][c] != turn) {
                        w = false;
                        break;
                    }
                }
                win = win || w;
            }
            for(int c=0;c<3;++c) {

                bool w = true;
                for(int r=0;r<3;++r) {
                    if(board[r][c] != turn) {
                        w = false;
                        break;
                    }
                }
                win = win || w;
            }
            if(board[0][0] == turn && board[1][1] == turn && board[2][2] == turn) {
                win = true;
            }
            if(board[0][2] == turn && board[1][1] == turn && board[2][0] == turn) {
                win = true;
            }
            if(win) {
                if(turn == 1) {
                    return "A";
                }
                else {
                    return "B";
                }
            } 

            turn *= -1;
        }
        if(moves.size() == 9) {
            return "Draw";
        }
        else {
            return "Pending";
        }
    }
};
