class Solution {
public:
    int countSquares(vector<vector<int>>& matrix) {
        int row = matrix.size(), col = matrix[0].size();
        vector<vector<int>> result(row+1, vector<int>(col+1, 0));
        int sum = 0;
        for (int i=1;i<=row; i++)
            for (int j=1; j<=col; j++) {
                if (matrix[i-1][j-1] == 0) continue;
                result[i][j] = 1 + min(result[i-1][j-1], min(result[i-1][j], result[i][j-1]));
                sum += result[i][j];
            }
        return sum;
    }
};
