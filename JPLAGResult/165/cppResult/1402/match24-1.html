class Solution {
public:
    int countSquares(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
        vector<vector<int> > pre(n + 1, vector<int>(m + 1, 0));
        for(int i = 0;i < n;i++){
            for(int j = 0;j < m;j++){
                pre[i + 1][j + 1] = pre[i][j + 1] + pre[i + 1][j] - pre[i][j] + (matrix[i][j] == 1);
            }

        }
        int ans = 0;
        for(int i = 1;i <= n;i++){
            for(int j = 1;j <= m;j++){
                for(int len = 1;len <= min(i, j);len++){
                    if(pre[i][j] + pre[i - len][j - len] - pre[i][j - len] - pre[i - len][j] == len * len){
                        ans++;
                    }
                }
            }
        }
        return ans;
    }
};
