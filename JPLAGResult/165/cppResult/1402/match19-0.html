class Solution {
public:
    int psum[310][310], sm[310];
    int countSquares(vector<vector<int>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        memset(psum, 0, sizeof psum);
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                psum[i+1][j+1] = - psum[i][j] + psum[i+1][j] + psum[i][j+1] + matrix[i][j];
            }
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                for (int k = 1; i + 1 >= k && j + 1 >= k; ++k) {
                    int l = j - k + 1, t = i - k + 1;
                    int sm = psum[i+1][j+1]-psum[i+1][l]-psum[t][j+1]+psum[t][l];
                    if (sm == k * k) ++ans;
                }
            }
        }
        return ans;
    }
};
