class Solution {
public:
    
    int countSquares(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        vector<vector<int>> t(m+1, vector<int>(n+1));
        for(int i=1;i<=m;i++){
            for(int j=1;j<=n;j++){
                t[i][j] = matrix[i-1][j-1] + t[i-1][j] + t[i][j-1] - t[i-1][j-1];
            }
        }
        int res = 0;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                int low = 0, high = min(n-j, m-i);
                while(low < high){
                    int mid = (high+low+1)/2;
                    if(t[i][j] + t[i+mid][j+mid] - t[i][j+mid] - t[i+mid][j]  == mid*mid) low = mid;
                    else high = mid-1;
                }
                res += low;
            }
        }
        return res;
    }
};
