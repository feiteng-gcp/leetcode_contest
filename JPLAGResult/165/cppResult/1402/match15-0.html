class Solution {
public:
    int countSquares(vector<vector<int>>& matrix) {
        int h = matrix.size(), w = matrix[0].size();
        
        vector<vector<int> > dp(h + 5 , vector<int>(w + 5 , 0));
        
        for( int i = 0 ; i < h ; ++i ){
            for( int j = 0 ; j < w ; ++j ){
                dp[i][j] = matrix[i][j];
            }
        }
        
        for( int i = 1 ; i < h ; ++i ){
            for( int j = 1 ; j < w ; ++j ){
                if( matrix[i][j] == 1 && matrix[i - 1][j - 1] == 1 && matrix[ i - 1 ][j] == 1 && matrix[i][j - 1] == 1 ){
                    dp[i][j] = 1 + min( dp[i - 1][j - 1], min(dp[i - 1][j], dp[i][j - 1]));
                }
            }
        }
        int res = 0;
        for( int i = 0 ; i < h ; ++i ){
            for( int j = 0 ; j < w ; ++j ){
                res += dp[i][j];
            }
        }
        return res;
    }
};
