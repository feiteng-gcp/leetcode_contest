class Solution {
public:
    int countSquares(vector<vector<int>>& m) {
        int r = m.size();
        int c = m[0].size();
        vector<vector<int>> cnt(r, vector<int>(c, 0));
        for (int i = 0 ; i < r; i++) {
            for (int j = 0; j < c; j++) {
                cnt[i][j] = m[i][j];
                if (i > 0)
                    cnt[i][j] += cnt[i - 1][j];
                if (j > 0)
                    cnt[i][j] += cnt[i][j - 1];
                if (i > 0 && j > 0)
                    cnt[i][j] -= cnt[i - 1][j - 1];
            }
        }
        int ret = 0;
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                for (int d = 1; ; d++) {
                    if (i + d - 1 >= r) break;
                    if (j + d - 1 >= c) break;
                    int sum = cnt[i + d - 1][j + d - 1];
                    if (i)
                    sum -= cnt[i - 1][j + d - 1];
                    if (j)
                    sum -= cnt[i + d - 1][j - 1];
                    if(i && j)
                    sum += cnt[i - 1][j - 1];
                    if (sum == d * d) ret++;
                }
            }
        }
        return ret;
    }
};
