class Solution {
public:
  int countSquares(vector<vector<int>>& matrix) {
    int n = matrix.size();
    int m = matrix[0].size();
    int tot = 0;
    for (int i = 1; i < n; i++) {
      for (int j = 1; j < m; j++) {
        if (matrix[i][j] == 1) {
          if (i == j && i == 0) continue;
          matrix[i][j] = 0x3f3f3f3f;
          matrix[i][j] = min(matrix[i][j], matrix[i-1][j]+1);
          matrix[i][j] = min(matrix[i][j], matrix[i][j-1]+1);
          matrix[i][j] = min(matrix[i][j], matrix[i-1][j-1]+1);
        }
      }
    }
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < m; j++) {
        tot += matrix[i][j];
      }
      cout << endl;
    }
    return tot;
  }
};
