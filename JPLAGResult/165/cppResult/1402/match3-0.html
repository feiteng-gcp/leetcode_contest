class Solution {
public:
    int countSquares(vector<vector<int>>& matrix) {
        int ans = 0;
        int m = matrix.size(),n = matrix[0].size();
        
        vector<vector<int>> dp(m+1,vector<int>(n+1,0));
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                dp[i+1][j+1] = dp[i][j+1]+dp[i+1][j]-dp[i][j]+matrix[i][j];
            }
        }
        
        
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(matrix[i][j]==0) continue;
                ans++;
                int tm = min(i,j);
                int k = 1;
                while(k<=tm){
                    int temp = (k+1)*(k+1);
                    int ttemp = dp[i+1][j+1]+dp[i-k][j-k]-dp[i-k][j+1]-dp[i+1][j-k];
                    if(temp == ttemp) ans++;
                    k++;
                }
            }
        }
        return ans;
    }
};
