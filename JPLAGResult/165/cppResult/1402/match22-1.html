class Solution {
public:
    int countSquares(vector<vector<int>>& matrix) {
        int M = matrix.size();
        int N = matrix[0].size();
        vector<vector<int>> S(M+1, vector<int>(N+1));
        
        for (int i=0;i<M;++i) {
            for (int j=0;j<N;++j) {
                S[i+1][j+1] = S[i+1][j] + S[i][j+1] - S[i][j] + matrix[i][j];
            }

        }
        
        int c = 0;
        for (int i=0;i<M;++i) {
            for (int j=0;j<N;++j) {
                for (int l=1;i+l<=M && j+l <= N;++l) {
                    int one = S[i+l][j+l] - S[i+l][j] - S[i][j+l] + S[i][j];
                    int area = l*l;
                    if (one == area) {
                        ++ c;
                    } else {
                        break;
                    }
                }
            }
        }
        return c;
        
    }
};
