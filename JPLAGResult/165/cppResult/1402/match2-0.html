class Solution {
public:
    int sum[400][400];
    int getx(int sx, int sy, int ex, int ey) {
        return sum[ex][ey] - sum[ex][sy - 1] - sum[sx - 1][ey] +
               sum[sx - 1][sy - 1];
    }
    int countSquares(vector<vector<int>> &matrix) {
        memset(sum, 0, sizeof(sum));
        int n = matrix.size();
        int m = matrix[0].size();
        for (int i = 0; i < n; i++) {
            sum[i + 1][1] = matrix[i][0];
            for (int j = 1; j < m; j++) {
                sum[i + 1][j + 1] = sum[i + 1][j] + matrix[i][j];
            }
        }
        for (int i = 0; i < m; i++) {
            for (int j = 1; j < n; j++) {
                sum[j + 1][i + 1] += sum[j][i + 1];
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                cout << sum[i + 1][j + 1] << " ";
            }
            cout << endl;
        }
        int ans = 0;
        for (int i = 1; i <= 300; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k < m; k++) {
                    int sx = j + 1;
                    int sy = k + 1;
                    int ex = j + i;
                    int ey = k + i;
                    if (ex > n || ey > m)
                        continue;
                    if (getx(sx, sy, ex, ey) == i * i)
                        ++ans;
                }
            }
        }
        return ans;
    }
};
