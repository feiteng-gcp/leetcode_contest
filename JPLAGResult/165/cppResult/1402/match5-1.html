class Solution {
public:
    int countSquares(vector<vector<int>>& m) {
       vector<vector<int>> s= m;
        for(int i = 0; i < m.size(); i++) {
            for(int j = 0; j < m[0].size(); j++) {
                if(i == 0) {
                    if(j > 0) s[i][j] += s[i][j-1];
                } else {
                    if(j == 0) s[i][j] += s[i-1][j];
                    else s[i][j] += s[i-1][j] + s[i][j-1] - s[i-1][j-1];
                }
            }
        }
        int ans = 0;
        
        for(int i = 0; i < m.size(); i++) {
            for(int j = 0; j < m[0].size(); j++) {
                if(m[i][j] == 0) continue;
                ans++;
                for(int k = 1; i+k < m.size() && j + k < m[0].size(); k++) {
                    int expected = (k+1)*(k+1);
                    int act = s[i+k][j+k];
                    if(i > 0) act -= s[i-1][j+k];
                    if(j > 0) act -= s[i+k][j-1];
                    if(i >0 && j > 0) {
                        act += s[i-1][j-1];
                    }
                 ans += ((expected == act)? 1:0);
                }
            }
        }
        return ans;
    }
};
