class Solution {
 public:
  int countSquares(vector<vector<int>>& matrix) {
    int n = matrix.size();
    int m = matrix[0].size();
    vector<vector<int>> dp(n, vector<int>(m));
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < m; ++j) {
        dp[i][j] = matrix[i][j];
        if (i) dp[i][j] += dp[i - 1][j];
        if (j) dp[i][j] += dp[i][j - 1];
        if (i && j) dp[i][j] -= dp[i - 1][j - 1];
      }
    }
    auto query = [&](int x1, int y1, int x2, int y2) {
      int u = dp[x2][y2];
      if (x1) u -= dp[x1 - 1][y2];
      if (y1) u -= dp[x2][y1 - 1];
      if (x1 && y1) u += dp[x1 - 1][y1 - 1];
      return u;
    };
    int cnt = 0;
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < m; ++j) {
        for (int k = 1; i + k <= n && j + k <= m; ++k) {
          cnt += query(i, j, i + k - 1, j + k - 1) == k * k;
        }
      }
    }
    return cnt;
  }
};
