class Solution {
public:
    int countSquares(vector<vector<int>>& matr) {
        int n = matr.size();
        int m = matr[0].size();
        vector<vector<int>> sums(n+1, vector<int>(m+1));
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                sums[i][j] = sums[i][j-1] + sums[i-1][j] - sums[i-1][j-1] + matr[i-1][j-1];
            }

        }
        int res = 0;
        for (int len = 1; len <= min(n, m); ++len) {
            for (int i = 0; i + len <= n; ++i) {
                for (int j = 0; j + len <= m; ++j) {
                    res += (sums[i+len][j+len] - sums[i][j+len] - sums[i+len][j] + sums[i][j]) == len*len;
                }
            }
        }
        return res;
    }
};
