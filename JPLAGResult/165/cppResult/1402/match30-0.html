int up1[310][310];
int left1[310][310];

int get(int i, int j){
    if(i<0||j<0)
        return 0;
    
    int ans = min(up1[i][j], left1[i][j]);
    return min(ans, get(i-1, j-1)+1);
}
class Solution {
public:
    int countSquares(vector<vector<int>>& matrix) {
        memset(up1, 0, sizeof up1);
        memset(left1, 0, sizeof left1);
        int n = matrix.size();
        int m = matrix[0].size();
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(i==0){
                    up1[i][j] = matrix[i][j]==1;
                }else{
                    if(matrix[i][j]==1){
                        up1[i][j] = up1[i-1][j]+1;
                    }else{
                        up1[i][j] = 0;
                    }
                }
            }
            
            for(int j=0;j<m;j++){
                if(j==0){
                    left1[i][j] = matrix[i][j]==1;
                }else{
                    if(matrix[i][j]==1){
                        left1[i][j] = left1[i][j-1]+1;
                    }else{
                        left1[i][j] = 0;
                    }
                }
            }
        }
        int ans = 0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                ans+=get(i,j);
            }
        }
        return ans;
    }
};
