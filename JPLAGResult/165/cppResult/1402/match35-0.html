class Solution {
public:
    int countSquares(vector<vector<int>>& matrix) {
        int n = matrix.size();
        if (n == 0) {
            return 0;             
        }
        int m = matrix[0].size();
        if (m == 0) {
            return 0;
        }
        vector<vector<int>> cnts(n + 1, vector<int>(m + 1, 0));
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                cnts[i + 1][j + 1] = cnts[i][j + 1] + cnts[i + 1][j] - cnts[i][j] + (matrix[i][j] == 1);
            }
        }
        
        auto count = [&](int t, int b, int l, int r) {
            return cnts[b + 1][r + 1] - cnts[b + 1][l] - cnts[t][r + 1] + cnts[t][l];
        };
        
        vector<int> squares(400, 0);
        for (int i = 0; i < squares.size(); ++i) {
            squares[i] = i * i;
        }
        
        int res = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                for (int k = 1; i + k <= n && j + k <= m; ++k) {
                    int t = i, b = i + k - 1, l = j, r = j + k - 1;
                    if (count(t, b, l, r) == squares[k]) {
                        ++res;
                    } else {
                        break;
                    }
                }
            }
        }
        
        return res;
    }
};
