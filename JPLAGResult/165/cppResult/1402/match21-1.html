class Solution {
public:
    int countSquares(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
        for (int i=0; i<n; i++){
            for (int j=0; j<m; j++){
                if (matrix[i][j] && j) matrix[i][j] += matrix[i][j-1];
            }
        }

        int ans = 0;
        /*
        for (int i=0; i<n; i++){
            for (int j=0; j<m; j++){
                cout<<matrix[i][j]<<' ';
            }
            cout<<endl;
        }
        */
        for (int k=1; k<=min(n, m); k++){
            for (int i=0; i<n; i++){
                for (int j=0; j<m; j++){
                    if (matrix[i][j] >= k && i+k-1<n){

                        int flag = 1;
                        for (int p = 0; p<k && i+p<n; p++){
                            if (matrix[i+p][j] < k){
                                flag = 0;
                                break;
                            }
                        }
                        ans += flag;
                    }
                }
            }
            //cout<<ans<<endl;
        }
        return ans;
    }
};
