class Solution {
public:
    
    int getSubSum(vector<vector<int>> &sum, int i, int j, int sz) {
        int ii = i - sz;
        int jj = j - sz;
        int result = sum[i][j];
        if (jj >= 0) {
            result -= sum[i][jj];
        }
        if (ii >= 0) {
            result -= sum[ii][j];
        }
        if (ii >= 0 and jj >= 0) {
            result += sum[ii][jj];
        }
        return result;
    }
    
    int countSquares(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
        vector<vector<int>> sum(n, vector<int>(m, 0));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                sum[i][j] = matrix[i][j];
                if (j > 0) {
                    sum[i][j] += sum[i][j - 1];
                }
            }
        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < m; j++) {
                sum[i][j] += sum[i - 1][j];
            }
        }
        int result = 0;
        for (int sz = 1; sz <= min(n, m); sz++) {
            for (int i = sz - 1; i < n; i++) {
                for (int j = sz - 1; j < m; j++) {
                    if (getSubSum(sum, i, j, sz) == sz * sz) {
                        
                        result++;
                    }
                }
            }
        }
        return result;
    }
};
