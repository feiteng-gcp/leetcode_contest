class Solution {
public:
    int countSquares(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        vector<vector<int>> f(n+1, vector<int>(m+1));
        for (int i = 1; i <= n; ++ i)
            for (int j = 1; j <= m; ++ j)
            {
                f[i][j] = a[i-1][j-1];
                f[i][j] = f[i-1][j]+f[i][j-1]-f[i-1][j-1]+f[i][j];
            }
        auto check = [&](int x1, int y1, int x2, int y2)
        {
            return f[x2][y2]-f[x1-1][y2]-f[x2][y1-1]+f[x1-1][y1-1];
        };
        int ans = 0;
        for (int i = 1; i <= n; ++ i)
            for (int j = 1; j <= m; ++ j)
            {
                if (a[i-1][j-1] == 0) continue;
                int len = 1;
                while (len <= i && len <= j && check(i-len+1, j-len+1, i, j) == len*len) ++ len;
                len --;
                ans += len;
                //printf("%d %d %d\n", i, j, len);
            }
        return ans;
    }
};
