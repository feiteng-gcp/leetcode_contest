class Solution {
public:
    int countSquares(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        vector<vector<int>> s(n, vector<int>(m, 0));
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                if(j > 0) s[i][j] = s[i][j - 1] + a[i][j];
                else s[i][j] = a[i][j];
            }
        }
        for(int j = 0; j < m; j++) {
            for(int i = 0; i < n; i++) {
                if(i > 0) s[i][j] += s[i - 1][j];
            }
        }
        int ans = 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                for(int k = 1; i + k <= n && j + k <= m; k++) {
                    int sum = s[i + k - 1][j + k -1];
                    if(i > 0) sum -= s[i - 1][j + k -1];
                    if(j > 0) sum -= s[i + k - 1][j - 1];
                    if(i > 0 && j > 0) sum += s[i - 1][j - 1];
                    if(sum == k * k) {
                        ans += 1;
                    }
                }
            }
        }
        return ans;
    
    }
};
