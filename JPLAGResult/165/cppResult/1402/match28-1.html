class Solution {
    bool hasSquare(vector<vector<int>>& sum, int x, int y, int len) {

        int res = sum[x + len - 1][y + len - 1];
        if (x > 0) {
            res -= sum[x - 1][y + len - 1];
        }
        if (y > 0) {
            res -= sum[x + len - 1][y - 1];
        }
        if (x > 0 && y > 0) {
            res += sum[x - 1][y - 1];
        }
        return (res == len * len);
    }
public:
    int countSquares(vector<vector<int>>& matrix) {

        int n = matrix.size(), m = matrix[0].size();
        vector<vector<int>> sum(n, vector<int>(m, 0));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                sum[i][j] = matrix[i][j];
                if (i > 0) sum[i][j] += sum[i - 1][j];
                if (j > 0) sum[i][j] += sum[i][j - 1];

                if (i > 0 && j > 0) sum[i][j] -= sum[i - 1][j - 1];
            }
        }
        int result = 0;
        for (int len = 1; len <= min(n, m); len++) {
            for (int i = 0; i <= n - len; i++) {
                for (int j = 0; j <= m - len; j++) {
                    if (hasSquare(sum, i, j, len)) {
                        result++;
                    }
                }
            }
        }
        return result;
    }
};
