class Solution {
public:
    int s[305][305];
    int countSquares(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= m; j++){
                s[i][j] = s[i-1][j]+s[i][j-1]-s[i-1][j-1]+a[i-1][j-1];
            }
        }
        int ans = 0;
        for(int l = 1; l <= 300; l++){
            for(int i = 1; i <= n; i++){
                for(int j = 1; j <= m; j++){
                    if(j+l-1 > m || i+l-1 > n){
                        continue;
                    }
                    int x = i+l-1, y = j+l-1;
                    if(s[x][y]-s[x][j-1]-s[i-1][y]+s[i-1][j-1] == l*l){
                        ans++;
                    }
                }
            }
        }
        return ans;
    }
};
