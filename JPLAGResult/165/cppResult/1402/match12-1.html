class Solution {
public:
    int countSquares(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        vector<vector<int>> pref(n+1, vector<int>(m+1, 0));
        for (int i = 1; i <= n; ++i) {
            int row = 0;
            for (int j = 1; j <= m; ++j) {
                row += a[i-1][j-1];

                pref[i][j] = pref[i-1][j] + row;
            }
        }
        int res = 0;
        for (int d = 1; d <= min(n, m); ++d) {
            for (int i = 1; i+d-1 <= n; ++i) {
                for (int j = 1; j+d-1 <= m; ++j) {
                    int ones = pref[i+d-1][j+d-1] - pref[i+d-1][j-1] - pref[i-1][j+d-1] + pref[i-1][j-1];
                    if (ones == d*d) {
                        ++res;
                    }
                }
            }
        }
        return res;
    }
};
