class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        n, m = len(matrix), len(matrix[0])
        rows, cols = [[[0] * m for i in range(n)] for j in range(2)]
        for i in range(n):
            for j in range(m):
                if matrix[i][j] == 1:
                    rows[i][j] = rows[i][j - 1] + 1 if j > 0 else 1
                    cols[i][j] = cols[i - 1][j] + 1 if i > 0 else 1
        ret = 0
        for i in range(n):
            for j in range(m):
                if matrix[i][j] > 0:
                    t = min(rows[i][j], cols[i][j])
                    if i > 0 and j > 0:
                        t = min(t, matrix[i - 1][j - 1] + 1)
                    matrix[i][j] = t
                    ret += t
        return ret
