class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        dp, ans = [[0 for i in range(n + 1)] for j in range(m + 1)], 0
        for row in range(m):
            for col in range(n):
                if matrix[row][col] == 1:
                    dp[row + 1][col + 1] = min(dp[row][col], dp[row][col + 1], dp[row + 1][col]) + 1
                    ans += dp[row + 1][col + 1]
        return ans
