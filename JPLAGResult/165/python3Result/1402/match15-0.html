class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        m,n = len(matrix), len(matrix[0])
        f = [[0 for j in range(n)] for i in range(m)]
        for i in range(n):
            f[0][i] = 1 if matrix[0][i] == 1 else 0
        for i in range(m):
            f[i][0] = 1 if matrix[i][0] == 1 else 0
        for i in range(1,m):
            for j in range(1,n):
                if matrix[i][j] == 1:
                    f[i][j] = min(f[i][j-1],f[i-1][j],f[i-1][j-1]) + 1
        ans = 0
        for i in range(m):
            ans += sum(f[i])
        return ans
