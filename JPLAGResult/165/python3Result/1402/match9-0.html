class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        m, n, ans = len(matrix), len(matrix[0]), 0
        aux = [[0] * n for _ in range(m)]
        for r in range(m):
            for c in range(n):
                if matrix[r][c] == 0: continue
                up = 0 if r == 0 else aux[r - 1][c]
                left = 0 if c == 0 else aux[r][c - 1]
                upleft = 0 if r * c == 0 else aux[r - 1][c - 1]
                aux[r][c] = 1 + min(up, left, upleft)
                ans += aux[r][c]
        return ans
