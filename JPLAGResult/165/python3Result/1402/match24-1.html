class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        ans = 0
        m, n = len(matrix), len(matrix[0]) 
        
for i in range(m-2, -1, -1):
            for j in range(n - 2, -1, -1):
                if matrix[i][j] == 0: continue
                matrix[i][j] = 1 + min(matrix[i+1][j], matrix[i][j+1], matrix[i+1][j+1])
        for i in range(m):
            for j in range(n):
                ans += matrix[i][j]
        return ans
