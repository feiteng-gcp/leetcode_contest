class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        f = [[0] * n for _ in range(m)]
        ans = 0
        for i in range(m):
            for j in range(n):
                if i == 0 or j == 0:
                    f[i][j] = matrix[i][j]
                else:
                    if matrix[i][j] == 0:
                        f[i][j] = 0
                    else:
                        f[i][j] = min(f[i - 1][j], f[i][j - 1])
                        if matrix[i - f[i][j]][j - f[i][j]] == 1:
                            f[i][j] += 1
                ans += f[i][j]
        return ans
