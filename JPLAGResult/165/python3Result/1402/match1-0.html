class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        m, n = len(matrix), len(matrix[0])
        dp = [[0 for _ in range(n)] for _ in range(m)]
        hp = [[0 for _ in range(n)] for _ in range(m)]
        dp[0][0] = hp[0][0] = matrix[0][0]
        for j in range(1, n):
            dp[0][j] = dp[0][j-1] + matrix[0][j]
            hp[0][j] = matrix[0][j]
        for i in range(1, m):
            dp[i][0] = dp[i-1][0] + matrix[i][0]
            hp[i][0] = matrix[i][0]
        for i in range(1, m):
            for j in range(1, n):
                dp[i][j] = dp[i-1][j] + dp[i][j-1] - dp[i-1][j-1]
                if matrix[i][j] == 0:
                    hp[i][j] = 0
                else:
                    hp[i][j] = 1 + min(hp[i-1][j], hp[i][j-1], hp[i-1][j-1])
                dp[i][j] += hp[i][j]
        return dp[m-1][n-1]
