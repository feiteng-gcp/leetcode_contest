class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        h, w = len(matrix), len(matrix[0])
        max_sq = [[0 for i in range(w)] for j in range(h)]
        for i in range(h):
            for j in range(w):
                if matrix[i][j] == 0:
                    continue
                max_sq[i][j] = 1
                if i > 0 and j > 0:
                    max_sq[i][j] += min(max_sq[i - 1][j - 1], max_sq[i - 1][j], max_sq[i][j - 1])
                
        ans = 0
        for i in range(h):
            for j in range(w):
                ans += max_sq[i][j]
        return ans
