class Solution:
    def countSquares(self, grid: List[List[int]]) -> int:
        n=len(grid)
        m=len(grid[0]) if n else 0
        dp=[[0]*m for _ in range(n)]
        res=0
        
for i in range(n):
            for j in range(m):
                if grid[i][j]==0:
                    continue
                l=dp[i][j-1] if j>0 else 0
                t=dp[i-1][j] if i>0 else 0
                tl=dp[i-1][j-1] if i>0 and j>0 else 0
                dp[i][j]=min(l,t,tl)+1
                res+=dp[i][j]
        return res
