class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        m = len(matrix)
        if m == 0:
            return 0
        n = len(matrix[0])
        dp = [[0 for i in range(n)] for j in range(m)]
        for i in range(m):
            for j in range(n):
                if i == 0 or j == 0:
                    if matrix[i][j] == 1:
                        dp[i][j] == 1
                if matrix[i][j] == 1:
                    # if dp[i-1][j-1] == dp[i-1][j] == dp[i][j-1]:
                    dp[i][j] = min(dp[i-1][j-1], dp[i-1][j], dp[i][j-1]) + 1
        # print(sum(dp))
        ans = sum([sum(dp[i]) for i in range(m)])
        # print(ans)
        return ans
