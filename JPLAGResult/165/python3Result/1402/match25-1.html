class Solution:
    def countSquares(self, A: List[List[int]]) -> int:
        ROW, COL = len(A), len(A[0])
        dp = [[0] * (COL+1) for _ in range(ROW+1)]
        
        res = 0
        for i in range(ROW):
            for j in range(COL):
                if A[i][j]:
                    a, b, c = dp[i+1][j], dp[i][j], dp[i][j+1]
                    dp[i+1][j+1] = min(a, b, c) + 1
                    res += dp[i+1][j+1]
        return res
