class Solution:
    def countSquares(self, M: List[List[int]]) -> int:
        R, C = len(M), len(M[0])
        S = [[0 for k in range(C)] for l in range(R)] 
        # here we have set the first row and column of S[][] 
        S[0] = M[0][:]
        # Construct other entries 
        count = sum(S[0])
        for i in range(1, R): 
            S[i][0] = M[i][0]
            count += S[i][0]
            
for j in range(1, C): 
                if (M[i][j] == 1): 
                    S[i][j] = min(S[i][j-1], S[i-1][j], 
                                S[i-1][j-1]) + 1
                    count += S[i][j]
                    
                else: 
                    S[i][j] = 0
        # for row in S:
        #     print(row)
        # print('')
        return count
