class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        from copy import deepcopy
        dp = deepcopy(matrix)
        m = len(matrix)
        n = len(matrix[0])
        for i in range(m):
            dp[i][0] = matrix[i][0]
        for i in range(n):
            dp[0][i] = matrix[0][i]

        for i in range(1, m):
            for j in range(1, n):
                if matrix[i][j] == 1:
                    dp[i][j] = 1
                    if dp[i-1][j-1] > 0:
                        for k in range(dp[i-1][j-1]):
                            if matrix[i-k-1][j] == 1 and matrix[i][j-k-1] == 1:
                                dp[i][j] = k+2
                            else:
                                break
                else:
                    dp[i][j] = 0
                    continue
        return sum([sum(i) for i in dp])
