class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        dp = [[0 for i in range(len(s))] for j in range(k)]
        for i in range(k):
            for j in range(i, len(s)):
                if i == 0:
                    dp[i][j] = self.change(s[:j+1])
                elif j == i:
                    dp[i][j] = 0
                else:
                    tmp = float('inf')
                    for t in range(i-1, j):
                        # print(t, dp[i][t], self.change(s[t+1:j+1]))
                        tmp = min(tmp, self.change(s[t+1:j+1]) + dp[i-1][t])
                    dp[i][j] = tmp
        # print(dp)
        return dp[k-1][-1]
        
    def change(self, s):
        # print(s)
        n = len(s)
        i, j = 0, n-1
        num = 0
        while i < j:
            if s[i] != s[j]:
                num += 1
            i += 1
            j -= 1
        return num
