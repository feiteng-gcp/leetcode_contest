class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        D = [[0]]
        def C(s):
            return sum(i != j for i, j in zip(s, s[::-1])) // 2
        for i in range(len(s)):
            D.append([float("inf")] + D[-1])
            for j in range(i):
                t = C(s[j: i + 1])
                for l in range(j + 1):
                    D[-1][l + 1] = min(D[-1][l + 1], D[j][l] + t)
                #print(i, j, t, s[j: i + 1], D)
        return D[-1][k]
