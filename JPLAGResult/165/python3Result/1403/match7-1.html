class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        import functools
        @functools.lru_cache(None)
        def get(i, j):
            res = 0
            while i < j:
                if s[i] != s[j]:
                    res += 1
                i += 1
                j -= 1
            return res
        
        n = len(s)
        dp = [[float('inf')] * k for _ in range(n)]
        for i in range(n):
            if i < k:
                dp[i][i] = 0
        for i in range(n):
            dp[i][0] = get(0, i)
            for kk in range(1, k):
                if kk > i:
                    continue
                
for j in range(i):
                    dp[i][kk] = min(dp[i][kk], dp[j][kk - 1] + get(j + 1, i))
        return dp[-1][-1]
