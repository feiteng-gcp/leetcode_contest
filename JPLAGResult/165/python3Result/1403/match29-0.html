class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        from functools import lru_cache
        
        length = len(s)
        
        def count_change(start, end):
            # end exclued
            count = 0
            while start<end:
                if s[start] != s[end]:
                    count += 1
                start +=1
                end -= 1
            return count            
        
        @lru_cache(None)
        def DP(idx, kk):
            if kk == length-idx:
                return 0
            if kk == 1:
                return count_change(idx, length-1)
            mini = float('inf')
            for i in range(idx+1, length-kk+2):
                candy = count_change(idx, i-1) + DP(i, kk-1)
                if candy < mini:
                    mini = candy
            return mini
        
        return DP(0, k)
