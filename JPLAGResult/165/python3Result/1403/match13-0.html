class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        n = len(s)
        dp = [[sys.maxsize] * n for _ in range(n)]
        for i in range(n):
            self.check(s, i, i, dp)
            self.check(s, i, i + 1, dp)
        print('dp', dp)
        result = [[sys.maxsize] * (n + 1) for _ in range(k + 1)]
        for i in range(k + 1):
            result[i][0] = 0
        for i in range(1, k + 1):
            for j in range(1, n + 1):
                for p in range(j):
                    result[i][j] = min(result[i][j], result[i - 1][p] + dp[p][j - 1])
        return result[k][n]
    
    def check(self, s, left, right, dp):
        change = 0
        while left >= 0 and right < len(s):
            if s[left] != s[right]:
                change += 1
            dp[left][right] = min(dp[left][right], change)
            left -= 1
            right += 1
