class Solution:
    def countToPalindrome(self, s: str):
        count = 0
        for i in range(len(s) // 2):
            if s[i] != s[-i-1]:
                count += 1
        return count
    
    def palindromePartition(self, s: str, k: int) -> int:
        dp = [[int(1e9)] * len(s) for _ in range(k + 1)]
        for i in range(1, k + 1):
            for j in range(len(s)):
                if i == 1:
                    dp[i][j] = self.countToPalindrome
(s[:j + 1])
                else:
                    for l in range(j):
                        dp[i][j] = min(dp[i][j], dp[i - 1][l] + self.countToPalindrome(s[l + 1:j + 1]))
        return dp[k][len(s) - 1]
