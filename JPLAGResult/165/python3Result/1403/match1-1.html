class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        def recursive(i, j, k):
            if k == j - i:
                return 0
            if k == 1:
                if (i, j, 1) in dp:
                    return dp[(i, j, 1)]
                p, q = i, j - 1
                res = 0
                while p < q:
                    if s[p] != s[q]:
                        res += 1
                    p, q = p + 1, q - 1
                dp[(i, j, k)] = res
                return dp[(i, j, k)]
            if (i, j, k) in dp:
                return dp[(i, j, k)]
            dp[(i, j, k)] = float('inf')
            for p in range(i + k - 1, j):
                dp[(i, j, k)] = min(dp[(i, j, k)], recursive(i, p, k - 1) + recursive(p, j, 1))
            return dp[(i, j, k)]
        
        dp = dict()
        return recursive(0, len(s), k)
