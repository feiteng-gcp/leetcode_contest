from functools import lru_cache

class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        def change(s, s_len):
            count = 0
            p, q 
= 0, s_len - 1
            while p < q:
                if s[p] != s[q]:
                    count += 1
                p += 1
                q -= 1
            return count
            
        @lru_cache(maxsize=None)
        def part(s, k):
            n = len(s)
            if k == 1:
                return change(s, n)
            best = float('inf')
            for i in range(1, n - k + 2):
                count = change(s, i) + part(s[i:], k - 1)
                best = min(best, count)
            return best
        
        return part(s, k)
