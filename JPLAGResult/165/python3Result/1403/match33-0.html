class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        dp = []
        for i in range(len(s)+1):
            dp.append([-1]*(k+1))

        def calc(ss: str):
            l = len(ss)
            result = 0
            for i in range(l//2):
                if ss[i] != ss[l-i-1]:
                    result += 1
            return result

        def solve(ll, lk):
            if dp[ll][lk] != -1: return dp[ll][lk]
            if ll==lk+1:return 0
            # print("sol", ll, lk)
            if lk == 0:
                dp[ll][lk] = calc(s[:ll])
                return dp[ll][lk]
            next_l = []
            # print(lk, ll)
            for i in range(lk, ll):
                next_l.append(solve(i, lk-1)+calc(s[i:ll]))
            dp[ll][lk] = min(next_l)
            return dp[ll][lk]
        # print(s)
        return solve(len(s), k-1)
