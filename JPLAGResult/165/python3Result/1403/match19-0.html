class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        
        def change(s):
            p, q = 0, len(s) - 1
            count = 0
            while p < q:
                if s[p] != s[q]:
                    count += 1
                p += 1
                q -= 1
            return count
        
        # dp[s_idx][k] = minimal changes
        dp = [[1000] * (k+1) for i in range(len(s))]
        # dp[0][1] = 1
        for j in range(1, k+1):
            dp[j-1][j] = 0
                
        for i in range(len(s)):
            dp[i][1] = change(s[:i+1])
        
        for i in range(1, len(s)):
            for j in range(2, k+1):
                min_change = dp[i][j]
                # j - 1 == final index of str of size j
                for p in range(j-2, i):
                    min_change = min(min_change, dp[p][j-1] + change(s[p+1:i+1]))
                dp[i][j] = min_change
        
        # for i in range(len(s)):
        #     print(dp[i][k-1], change(s[i+1:len(s)]))
        return dp[-1][k]
