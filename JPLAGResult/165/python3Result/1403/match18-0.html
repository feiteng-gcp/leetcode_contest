from functools import lru_cache
import math

class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        @lru_cache(maxsize=None)
        def nchange(i, j):
            res = 0
            while i < j:
                if s[i] != s[j]:
                    res += 1
                i,j = i+1, j-1
            return res
        
        @lru_cache(maxsize=None)
        def part(st, k):
            if k == 1: return nchange(st, len(s)-1)
            res = math.inf
            for i in range(st, len(s)):
                res = min(res, nchange(st, i) + part(i+1, k-1))
            return res
        return part(0, k)
