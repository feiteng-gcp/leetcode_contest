from functools import lru_cache

class Solution:
    @lru_cache(None)
    def changesNeeded(self, s: str, i: int, j: int):
        res = 0
        while i <= j:
            res += s[i] != s[j]
            i += 1
            j -= 1
        return res
    
    @lru_cache(None)
    def dp(self, s: str, end: int, k: int):
        if end == k:
            return 0
        if end == 1:
            return 0
        if k == 1:
            return self.changesNeeded(s, 0, end - 1)
        mi = float('inf')
        for i in range(1, end - k + 2):
            mi = min(self.dp(s, end - i, k - 1) + self.changesNeeded(s, end - i, end - 1), mi)
    
        return mi
    
    def palindromePartition(self, s: str, k: int) -> int:        
        return self.dp(s, len(s), k)
