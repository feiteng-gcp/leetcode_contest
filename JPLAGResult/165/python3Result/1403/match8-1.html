class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        l=len(s);kk=k
        dp=[[0]*(l+1) for i in range(l+1)]
        def calcost(st):
            l=len(st)
            l2=len(st)//2
            cost
=0
            for i in range(0,l2):
                if st[i]!=st[l-1-i]:
                    cost+=1
            return cost
        for k in range(1,l+1):
            for j in range(1,k+1):
                tmp=0
                if j==1:
                    tmp=calcost(s[:k])
                    dp[k][j]=tmp
                else:
                    tmp=min([dp[x][j-1]+calcost(s[x:k]) for x in range(j-1,k-1+1)])
                    dp[k][j]=tmp
        return dp[l][kk]
