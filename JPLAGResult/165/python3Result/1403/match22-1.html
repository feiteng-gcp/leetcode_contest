class Solution:
    
    
    
    import functools
    @functools.lru_cache(None)
    def palindromePartition(self, s: str, k: int) -> int:
        if k >= len(s):
            return 0
        if k == 1:
            res = 0
            for i in range(len(s)//2):
                res += s[i] != s[~i]
            return res
        else:
            res 
= len(s)
            for i in range(1, len(s)):
                res = min(res, self.palindromePartition(s[:i], 1)+ self.palindromePartition(s[i:], k-1))
            return res
