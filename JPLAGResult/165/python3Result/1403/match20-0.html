class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        
        def cal_k1(ss):
            ans_ss = 0
            for i in range(len(ss)//2):
                if ss[i] != ss[len(ss)-1-i]:
                    ans_ss += 1
            return ans_ss
        
        #for i in range(len(s)+1):
        #    print(s[0:i], cal_k1(s[0:i]))
            
            
            
        dp = {}
        
        def cal(end, k):
            
            if end < k:
                return 0
            
            if (end, k) in dp:
                return dp[(end, k)]
            
            if k == 1:
                dp[(end, 1)] =  cal_k1(s[0:end+1])
                #print("k=1", end, s[0:end+1], dp[(end, 1)])
                return dp[(end,1)]
            
            min_s = 1000
            
            for i in range(0, end):
                t1 = cal(i, k-1)
                t2 = cal_k1(s[i+1:end+1])
                #print("loop", t1, t2, i,k-1, s[i+1:])
                min_s = min(min_s, t1+t2)
                
            dp[(end,k)] = min_s
            
            return min_s
        
        
        ans = cal(len(s)-1, k)
        #print(dp)
        
        return ans
