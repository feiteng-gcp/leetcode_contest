class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        dp = {}
        def count(s):
            cnt = 0
            l = len(s) // 2
            ml = len(s) // 2 - 1
            mr = (len(s) + 1) // 2
            for _ in range(l):
                if s[ml] != s[mr]:
                    cnt += 1
                ml -= 1
                mr += 1
            return cnt
        def dfs(i, k):
            l = len(s) - i
            if i == len(s) and k == 0:
                return 0           
            if k > l or k == 0:
                return float('inf')
            if k == l:
                return 0
            if (i, k) in dp:
                return dp[(i, k)]
            min_ = float('inf')
            for j in range(i + 1, len
(s) + 1):
                min_ = min(count(s[i:j]) + dfs(j, k - 1), min_)
            dp[(i, k)] = min_
            return min_
        return dfs(0, k)
