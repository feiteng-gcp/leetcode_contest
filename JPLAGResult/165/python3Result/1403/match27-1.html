class Solution:
    def __init__(self):
        self.memo = {}
        self.fix = lambda s: sum(s[i] != s[~i] for i in range(len(s) // 2))

    def palindromePartition(self, s: str, k: int) -> int:
        #print(f"Called with {s},{k}")
        if k == 1:
            self.memo[s, k] = self.fix(s)
            return self.memo[s, k]
        if (s, k) in self.memo:
            return self.memo[s, k]
        if k == len(s):
            self.memo[s, k] = 0
            return 0
        answer 
= len(s)
        for i in range(1, len(s) - k + 2):
            answer = min(answer, self.fix(s[:i]) + self.palindromePartition(s[i:], k - 1))
        #print(f"{s}, {k} = {answer}")
        self.memo[s, k] = answer    
        return answer
