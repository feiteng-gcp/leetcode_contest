class Solution:
    #O(n^4)
    def palindromePartition(self, s: str, k: int) -> int:
        
        if k>len(s):
            return 0
        dp = [[0]*len(s) for _ in range(len(s))]
        memo = {}
        for a in range(len(s)):
            for b in range(a+1, len(s)):
                dp[a][b] = self.toPalindrome(s, a,b, memo)
        cost = [[float('inf')]*len(s) for _ in range(k)]
        for a in range(len(s)):
            cost[0][a] = dp[0][a]
        for split in range(1, k):
            #minv = cost[split-1][0]+dp[1][a]
            for a in range(len(s)):
                #cost[split][a] = minv
                for b in range(a):
                    cost[split][a] = min(cost[split][a], cost[split-1][b]+dp[b+1][a])
        return cost[-1][-1]
            
            
                    
        
        
    def toPalindrome(self, s, left, right, memo):
        if left>=right:
            return 0
        else:
            if (left, right) not in memo:
                if s[left]!=s[right]:
                    memo[(left, right)] = 1+self.toPalindrome(s, left+1, right-1, memo)
                else:
                    memo[(left, right)] = self.toPalindrome(s, left+1, right-1, memo)
            return memo[(left, right)]
