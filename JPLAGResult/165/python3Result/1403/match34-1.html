class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        N = len(s)
        ch = [[0] * (N+1) for _ in range(N)]
        
        for m in range(N):
            i, j = m - 1, m + 1
            while i >= 0 and j < N:
                if s[i] == s[j]:
                    ch[i][j+1] = ch[i+1][j]
                else:
                    ch[i][j+1] = ch[i+1][j] + 1
                i, j = i - 1, j + 1
            i, j = m, m + 1
            while i >= 0 and j < N:
                if s[i] == s[j]:
                    ch[i][j+1] = ch[i+1][j]
                else:
                    ch[i][j+1] = ch[i+1][j] + 1
                i, j = i - 1, j + 1
        
        dp = [[float('inf')] * (k+1) for i in range(N+1)]
        dp[0] = [0] * (k+1)
        for i in range(1, N+1):
            for x in range(1, k+1):
                for j in range(i):
                    dp[i][x] = min(dp[i][x], dp
[j][x-1] + ch[j][i])
        return dp[N][k]
