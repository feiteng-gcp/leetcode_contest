import functools


class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        n = len(s)
        
        @functools.lru_cache(None)
        def min_change(idx, partition):
            if n - idx <= partition:
                return 0
            if partition == 1:
                change, left, right = 0, idx, n - 1
                while left < right:
                    if s[left] != s[right]:
                        change += 1
                    left += 1
                    right -= 1
                return change
            candidate = float('inf')
            for i in range(idx + 1, n - partition + 2):
                change, left, right = 0, idx, i - 1
                while left < right:
                    if s[left] != s[right]:
                        change += 1
                    left += 1
                    right -= 1
                change += min_change(i, partition - 1)
                candidate = min(candidate, change)
            return candidate
        
        return min_change(0, k)
