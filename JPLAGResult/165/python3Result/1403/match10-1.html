from collections import defaultdict

class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        
        def check(s):
            head, tail = 0, len(s)-1
            res = 0
            while head<tail:
                if s[head]!=s[tail]:
                    res += 1
                head += 1
                tail -= 1
            return res
        
        res = [len(s)]
        
        memory = defaultdict(dict)
        def recursive(s, k):
            if s in memory and k in memory[s]:
                return memory[s][k]
            if k==len(s):
                return 0
            elif k==1:
                memory[s][k] = check(s)
                return memory[s][k]
            else:
                ans = len(s)
                for length in range(1, len(s)-k+2):
                    ans = min(ans, recursive(s[:length], 1) + recursive(s[length:], k-1))
                memory[s][k] = ans
                return ans
                
        
        return recursive(s,k)
