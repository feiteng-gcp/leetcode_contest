from functools import lru_cache
class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        
def calc(l, r):
            length = int((r - l + 1)/2)
            # print(length, l, r)
            tot = 0
            for i in range(length):
                if s[l + i] != s[r - i]:
                    tot += 1
            return tot
        @lru_cache(None)
        def dp(l, r, k):
            if k == 1:
                return calc(l, r)
            length = r - l + 1
            start = l + (k - 1)
            ans = float("inf")
            while start <= r:
                now = dp(l, start - 1, k - 1) + dp(start, r, 1)
                if now < ans:
                    ans = now
                start += 1
            return ans
        return dp(0, len(s)-1, k)
