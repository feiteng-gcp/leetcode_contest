class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        if k >= len(s) or k <= 0:
            return 0
        ans = [[0 for m in range(k + 1)] for j in range(len
(s))]
        def dist_to_palin(start, end):
            ans = 0
            while start < end:
                if s[start] != s[end]:
                    ans += 1
                start += 1
                end -= 1
            return ans
        
        for m in range(1, k + 1):
            for i in reversed(range(len(s) - m + 1)):
                if m == 1:
                    ans[i][m] = dist_to_palin(i, len(s) - 1)
                else:
                    ans[i][m] = 0x7fffffff
                    for j in range(i + 1, len(s)):
                        if len(s) - j < m - 1:
                            break
                        ans[i][m] = min(ans[i][m], dist_to_palin(i, j - 1) + ans[j][m - 1])
        return ans[0][k]
