class Solution:
    def palindromePartition(self, string: str, k: int) -> int:
        self.memo = {}
        def dp(s,e,k):
            if (s,e,k) not in self.memo:
                if k == 0: return float('inf')
                if e-s+1 < k: return float('inf')
                if e-s+1 == k: return 0
                if k == 1:
                    cnt = 0
                    for i in range((e-s)//2+1):
                        if string[s+i] != string[e-i]:
                            cnt += 1
                    self.memo[(s,e,k)] = cnt
                else:
                    self.memo[(s,e,k)] = min([dp(s,i,k-1)+dp(i+1,e,1) for i in range(s,e)])
            return self.memo[(s,e,k)]
        return dp(0,len(string)-1,k)
