class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        def find(string):
            i, j = 0, len(string)-1
            count = 0
            while i < j:
                if string[i] != string[j]:
                    count += 1
                i += 1
                j -= 1
            return count
        
        memo = {}
        def helper(i,k):
            if i == len(s):
                return 0
            
            if (i,k) in memo:
                return memo[i,k]
            
            if k == 1:
                ans = find(s[i:])
            else:
                ans = float("inf")
                for j in range(i+1, len(s)+1):
                    ans = min(ans, find(s[i:j]) + helper(j,k-1))
            memo[i,k] = ans
            return ans
        return helper(0,k)
