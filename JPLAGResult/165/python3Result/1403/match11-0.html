class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        
        def min_change_to_pal(st: str):
            l = len(st)
            i, j, res = 0, l - 1, 0
            while i < j:
                if st[i] != st[j]: res += 1
                i += 1
                j -= 1
            return res
        
        n = len(s)
        dp = [[-1] * (k + 1) for i in range(n + 1)]
        dp[0][0] = 0
        for i in range(1, n + 1):
            for q in range(0, i):
                min_change = min_change_to_pal(s[q:i])
                for j in range(1, k + 1):
                    if dp[q][j - 1] < 0: continue
                    t = dp[q][j - 1] + min_change
                    dp[i][j] = t if dp[i][j] < 0 or dp[i][j] > t else dp[i][j]
        return dp[n][k]
