class Solution:
    dp = {}
    def palindromePartition(self, s: str, k: int) -> int:
        if len(s) < k:
            return float('inf')
        if len(s) == k:
            return 0
        if (s, k) not in self.dp:
            if k == 1:
                l, r = 0, len(s) - 1
                cnt = 0
                while r > l:
                    if s[l] != s[r]:
                        cnt += 1
                    l += 1
                    r -= 1
                self.dp[(s, k)] = cnt
            else:
                min_mod = float('inf')
                for i in range(1, len(s)):
                    min_mod = min(min_mod, self.palindromePartition(s[:i], 1) + self.palindromePartition(s[i:], k - 1))
                self.dp[(s, k)] = min_mod
        return self.dp[(s, k)]
