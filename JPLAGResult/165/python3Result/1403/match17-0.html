class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        n = len(s)
        cache = {}
        def count(s):
            left, right = 0, len(s)-1
            ret = 0
            while left<right:
                if s[left] != s[right]:
                    ret += 1
                left += 1
                right -= 1
            # print('ret', s, ret)
            return ret
        def dfs(curr, parts, s):
            # print(s[:curr+1], parts)
            # if curr == 0:
            #     return 1
            if curr+1<parts:
                # print('er')
                return float('inf')
            if (curr, parts) in cache:
                # print('hit')
                return cache[(curr, parts)]
            if parts == 1:
                cache[(curr, parts)] = count(s[:curr+1])
                return cache[(curr, parts)]
            res = float('inf')
            for i in range(curr-1, -1, -1):
                # print(s[:i+1],s[i+1:curr+1])
                res = min(res, count(s[i+1:curr+1])+dfs(i, parts-1, s))
            cache[(curr, parts)] = res
            # print(curr, parts, res)
            return res
        
        return dfs(n-1, k, s)
