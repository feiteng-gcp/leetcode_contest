import functools
class Solution:
    def palindromePartition(self, s: str, k: int) -> int:
        @functools.lru_cache(None)
        def search(s, k):
            N = len(s)
            if N == k:
                return 0
            if k == 1:
                l = 0
                r = N - 1
                cnt 
= 0
                while l < r:
                    if s[l] != s [r]:
                        cnt += 1
                    l += 1
                    r -= 1
                return cnt
            rnt = N
            for j in range(N - k + 1):
                # print(s[:j], ' ', s[j:])
                rnt = min(rnt, search(s[:j + 1], 1) + search(s[j + 1:], k - 1))
                if rnt == 0:
                    return 0
            # for i in range(k - 1): # i + 1, k - i - 1
            #     for j in range(N - k + 1):
            #         rnt = min(rnt, search(s[:j + i + 1], i + 1) + search(s[j + i + 1:], k - i - 1))
            #         if rnt == 0:
            #             return 0
            return rnt
        return search(s, k)
