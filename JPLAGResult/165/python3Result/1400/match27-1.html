class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        rowCnt = [[0, 0] for _ in range(3)]
        colCnt = [[0, 0] for _ in range(3)]
        
        leftDiag = [0, 0]
        rightDiag = [0, 0]
        
        for idx, move in enumerate(moves):
            x, y = move
            rowCnt
[x][idx % 2] += 1
            colCnt[y][idx % 2] += 1
            if x == y:
                leftDiag[idx % 2] += 1
            if x + y == 2:
                rightDiag[idx % 2] += 1
            
            if rowCnt[x][idx % 2] == 3 or colCnt[y][idx % 2] == 3 or ((x == y) and (leftDiag[idx % 2] == 3)) or ((x + y) == 2 and (rightDiag[idx % 2] == 3)):
                return 'A' if idx % 2 == 0 else 'B'
            if idx == 8:
                return 'Draw'
        return 'Pending'
