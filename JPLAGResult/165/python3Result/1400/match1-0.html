class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        A, B = [], []
        for i in range(len(moves)):
            if i%2 == 0:
                A += [moves[i]]
            else:
                B += [moves[i]]
        A = set([tuple(pair) for pair in A])
        B = set([tuple(pair) for pair in B])
        win = [(0,0),(0,1),(0,2)]
        for i in range(3):
            tmp1 = []
            tmp2 = []
            for j in range(3):
                tmp1 += [(i,j)]
                tmp2 += [(j,i)]
            win += [tmp1,tmp2]
        win += [[(0,0),(1,1),(2,2)],[(0,2),(1,1),(2,0)]]
        for g in win:
            flag = True
            for t in g:
                if t not in A:
                    flag = False
            if flag: return 'A'
            flag = True
            for t in g:
                if t not in B:
                    flag = False
            if flag: return 'B'
        if len(moves) == 9: return 'Draw'
        return 'Pending'
