class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        g = [['0' for j in range(3)] for i in range(3)]
        for i,move in enumerate(moves):
            if i & 1:
                g[move[0]][move[1]] = 'B'
            else:
                g[move[0]][move[1]] = 'A'
        def check(g):
            for i in range(3):
                if g[i][0] != '0' and g
[i][0] == g[i][1] and g[i][1] == g[i][2]:
                    return g[i][0]
            for i in range(3):
                if g[0][i] != '0' and g[0][i] == g[1][i] and g[1][i] == g[2][i]:
                    return g[0][i]
            if g[0][0] != '0' and g
[0][0] == g[1][1] and g[1][1] == g[2][2]:
                return g[0][0]
            if g[0][2] != '0' and g[0][2] == g[1][1] and g[1][1] == g[2][0]:
                return g[0][2]
            return '0'
        end = check(g)
        if end == 'A' or end == 'B':
            return end
        if len(moves) == 9:
            return "Draw"
        return "Pending"
