class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        def test(a):
            d = dict({0:0,1:0,2:0})
            for e in a:
                d[e[0]] += 1
            if d[0] == 3 or d[1] == 3 or d[2] == 3:
                return True
            d = dict({0:0,1:0,2:0})
            for e in a:
                d[e[1]] += 1
            if d[0] == 3 or d[1] == 3 or d[2] == 3:
                return True
            d = dict({-2:0,-1:0,0:0,1:0,2:0})
            for e in a:
                d[e[0] - e[1]] += 1
            if d[0] == 3:
                return True
            d = dict({0:0,1:0,2:0,3:0,4:0})
            for e in a:
                d[e[0] + e[1]] += 1
            if d[2] == 3:
                return True
            return False
        if test(moves[0::2]):
            return "A"
        elif test(moves[1::2]):
            return "B"
        elif len(moves) == 9:
            return "Draw"
        else:
            return "Pending"
