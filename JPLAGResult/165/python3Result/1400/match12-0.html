class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        grid = [['']*3 for _ in range(3)]
        
        for i in range(len(moves)):
            c = 'A'
            if i % 2 == 1:
                c = 'B'
            move = moves[i]
            grid[move[0]][move[1]] = c
            #print(grid)
            
            for x in range(3):
                if grid[0][x] == c and grid[1][x] == c and grid[2][x] == c:
                    return c
                if grid[x][0] == c and grid[x][1] == c and grid[x][2] == c:
                    return c
            
            if grid[0][0] == c and grid[1][1] == c and grid[2][2] == c:
                return c
            
            if grid[0][2] == c and grid[1][1] == c and grid[2][0] == c:
                return c
            
                
        
        if len(moves) ==9 :
            return 'Draw'
        
        return "Pending"
