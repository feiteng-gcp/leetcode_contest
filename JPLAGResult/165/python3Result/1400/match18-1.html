class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:

        
        grid = [[0] * 3 for _ in range(3)]
        for i in range(len(moves)):
            r, c = moves[i]
            if i % 2 == 0:
                grid[r][c] = 'A'
            else:
                grid[r][c] = 'B'
        
        
for r in range(3):
            if grid[r][0] == grid[r][1] == grid[r][2] and grid[r][0] != 0:
                return grid[r][0]
            if grid[0][r] == grid[1][r] == grid[2][r] and grid[0][r] != 0:
                return grid[0][r]
        if grid[0][0] == grid[1][1] == grid[2][2] and grid[0][0] != 0:
            return grid[0][0]
        if grid[0][2] == grid[1][1] == grid[2][0] and grid[0][2] != 0:
            return grid[0][2]
        
        if len(moves) == 9:
            return 'Draw'
        return 'Pending'
