class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        grid = [["" for i in range(3)] for i in range(3)]
        cnt = 0
        for i, j in moves:
            if not cnt % 2:
                grid[i][j] = "X"
                if self.isfinish(i, j, grid, "X"):
                    return "A"
            else:
                grid[i][j] = "O"
                if self.isfinish(i, j, grid, "O"):
                    return "B"
            cnt += 1
        return "Pending" if len(moves) < 9 else "Draw"
    
    def isfinish(self, i, j, grid, target):
        
if grid[i][j] == grid[i-1][j] == grid[i-2][j] == target:
            return True
        if grid[i][j] == grid[i][j-1] == grid[i][j-2] == target:
            return True
        
if grid[0][0] == grid[1][1] == grid[2][2] == target:
            return True
        if grid[1][1] == grid[0][-1] == grid[-1][0] == target:
            return True
        return False
