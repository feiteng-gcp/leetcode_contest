def get_winner(b):
    char_to_winner = {'X': 'A', 'O': 'B'}
    for i in range(3):
        if b[i][0] == b[i][1] and b[i][0] == b[i][2] and b[i][0] != ' ':
            return char_to_winner[b
[i][0]]
    for i in range(3):
        if b[0][i] == b[1][i] and b[0][i] == b[2][i] and b[0][i] != ' ':
            return char_to_winner[b
[0][i]]
    if b[0][0] == b[1][1] and b[0][0] == b[2][2] and b[0][0] != ' ':
        return char_to_winner[b
[0][0]]
    if b[2][0] == b[1][1] and b[2][0] == b[0][2] and b[2][0] != ' ':
        return char_to_winner[b[2][0]]
    return None

class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        b = [[' ' for _ in range(3)] for i in range(3)]
        who = 'X'
        for move in moves:
            x, y = move
            b[x][y] = who
            who = 'X' if who == 'O' else 'O'
            winner = get_winner(b)
            if winner:
                return winner
        if len(moves) < 9:
            return 'Pending'
        return 'Draw'
