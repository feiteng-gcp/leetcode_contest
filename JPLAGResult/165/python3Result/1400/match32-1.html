class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        def win(grid, player):
            
if grid[0][0] == grid[0][1] == grid[0][2] == player:
                return True
            if grid[1][0] == grid[1][1] == grid[1][2] == player:
                return True
            if grid[2][0] == grid[2][1] == grid[2][2] == player:
                return True
            
            if grid[0][0] == grid[1][0] == grid[2][0] == player:
                return True
            if grid[0][1] == grid[1][1] == grid[2][1] == player:
                return True
            if grid[0][2] == grid[1][2] == grid[2][2] == player:
                return True
            
            
if grid[0][0] == grid[1][1] == grid[2][2] == player:
                return True
            if grid[0][2] == grid[1][1] == grid[2][0] == player:
                return True
            return False
        player = 'A'
        grid = [["_"]*3 for _ in range(3)]
        
        for x, y in moves:
            grid[x][y] = player
            if win(grid, player):
                return player
            if player == 'A':
                player = 'B'
            elif player == 'B':
                player = 'A'
        if len(moves) == 9:
            return "Draw"
        return "Pending"
