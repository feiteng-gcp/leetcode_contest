from itertools import cycle
class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        B = [['.'] * 3 for _ in range(3)]
        for m, (r, c) in zip(cycle('XO'), moves):
            B[r][c] = m
        
        for r in range(3):
            if B[r][0] == B[r][1] == B[r][2] and B[r][0] != '.':
                return 'A' if B[r][0] == 'X' else 'B'
            
        for c in range(3):
            if B[0][c] == B[1][c] == B[2][c] and B[0][c] != '.':
                return 'A' if B[0][c] == 'X' else 'B'
            
        if B[0][0] == B[1][1] == B[2][2] and B[1][1] != '.':
            return 'A' if B[1][1] == 'X' else 'B'
        
        if B[0][2] == B[1][1] == B[2][0] and B[1][1] != '.':
            return 'A' if B[1][1] == 'X' else 'B'
        
        return 'Draw' if len(moves) == 9 else 'Pending'
