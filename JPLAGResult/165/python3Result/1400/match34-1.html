class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        mat = [[None]*3 for _ in range(3)]
        for i,(x,y) in enumerate(moves):
            mat[x][y]=i%2
        # row
        
for i in range(3):
            if mat[i][0]==mat[i][1]==mat[i][2]:
                if mat[i][0] is not None:
                    return 'B' if mat
[i][0]%2 else 'A'

            if mat[0][i]==mat[1][i]==mat[2][i]:
                if mat[0][i] is not None:
                    return 'B' if mat[0][i]%2 else 'A'
        # dia
        if mat[0][0]==mat[1][1]==mat[2][2] and mat[0][0] is not None:
            return 'B' if mat
[0][0]%2 else 'A'
        if mat[0][2]==mat[1][1]==mat[2][0] and mat[2][0] is not None:
            return 'B' if mat[0][2]%2 else 'A'
        
        return 'Draw' if len(moves)==9 else 'Pending'
