class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        grid=[['','',''],['','',''],['','','']]
        for i in range(len(moves)):
            x,y=moves[i]
            v='X' if i%2==0 else 'O'
            grid
[x][y]=v
            for i in range(3):
                if grid[i][0]==grid[i][1]==grid[i][2]==v:
                    return 'A' if v=='X' else 'B'
            for j in range(3):
                
if grid[0][j]==grid[1][j]==grid[2][j]==v:
                    return 'A' if v=='X' else 'B'
            if grid[0][0]==grid[1][1]==grid[2][2]==v:
                return 'A' if v=='X' else 'B'
            if grid[0][2]==grid[1][1]==grid[2][0]==v:
                return 'A' if v=='X' else 'B'
        return 'Draw' if len(moves)==9 else 'Pending'
