class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        grid = [[' ' for _ in range(3)] for _ in range(3)]
        player = 'A'
        status = 'Pending'
        for x, y in moves:
            grid[x][y] = player
            for i in range(3):
                if len(set(grid[i])) == 1 and grid[i][0] != ' ':
                    return grid[i][0]
            for j in range(3):
                if len(set([grid[i][j] for i in range(3)])) == 1 and grid[0][j] != ' ':
                    return grid[0][j]
            if len(set([grid[i][i] for i in range(3)])) == 1 and grid[1][1] != ' ':
                return grid[1][1]
            if len(set([grid[i][2-i] for i in range(3)])) == 1 and grid[1][1] != ' ':
                return grid[1][1]
            player = 'B' if player == 'A' else 'A'
        if ' ' in set([grid[i][j] for i in range(3) for j in range(3)]):
            return 'Pending'
        else:
            return 'Draw'
