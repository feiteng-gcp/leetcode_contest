class Solution:
    def tictactoe(self, moves) -> str:
        mp = [[-1] * 3 for _ in range(3) ]
        turn = 1
        for x, y in moves:
            mp[x][y] = turn
            turn = 1 - turn


        def check(a):
            for i in range(3):
                if mp[i][0] == a and mp[i][1] == a and mp[i][2] == a:
                    return True
                if mp[1][i] == a and mp[2][i] == a and mp[0][i] == a:
                    return True
            if mp[1][1] == a and mp[2][2] == a and mp[0][0] == a:
                return True
            if mp[2][0] == a and mp[1][1] == a and mp[0][2] == a:
                return True
            return False

        if check(1):
            return "A"
        if check(0):
            return "B"
        if len(moves) < 9:
            return "Pending"
        return "Draw"
