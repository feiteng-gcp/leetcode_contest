class Solution:
    def check(self, A):
        for b in range(3):
            if A[b][0] == A[b][1] == A[b][2] and A[b][0] != ".":
                return A[b][0]
        for b in range(3):
            if A[0][b] == A[1][b] == A[2][b] and A[0][b] != ".":
                return A[0][b]
        if A[0][0] == A[1][1] == A[2][2] and A[0][0] != ".":
            return A[0][0]
        if A[0][2] == A[1][1] == A[2][0] and A[0][2] != ".":
            return A[0][2]
        return "."
        
    def tictactoe(self, moves: List[List[int]]) -> str:
        A = [["."] * 3 for _ in range(3)]
        for i in range(len(moves)):
            this = "A" if i%2 == 0 else "B"
            x, y = moves[i]
            A[x][y] = this
        result = self.check(A)
        if result != ".":
            return result
        if len(moves) < 9:
            return "Pending"
        return "Draw"
