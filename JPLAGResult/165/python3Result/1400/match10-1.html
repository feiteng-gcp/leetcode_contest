class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        board = [['T' for i in range(3)] for j in range(3)]
        p1 = 'X'
        p2 = 'O'
        player = p1
        for i in range(len(moves)):
            posx, posy = moves[i][0], moves[i][1]
            board[posx][posy] = player
            player = p1 if player == p2 else p2
        n = len(moves)
        
for i in range(3):
            if board[i][0] == board[i][1] == board[i][2] == p1 or board[0][i] == board[1][i] == board[2][i] == p1:
                return "A"
            if board
[i][0] == board[i][1] == board[i][2] == p2 or board[0][i] == board[1][i] == board[2][i] == p2:
                return 'B'
        if board[0][0] == board[1][1] == board[2][2] == p1 or board[0][2] == board[1][1] == board[2][0] == p1:
            return "A"
        if board
[0][0] == board[1][1] == board[2][2] == p2 or board[0][2] == board[1][1] == board[2][0] == p2:
            return "B"
        if n == 9:
            return "Draw"
        else:
            return "Pending"
