class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        m = [["" for j in range(3)] for i in range(3)]
        flag = "X"
        for move in moves:
            m[move[0]][move[1]] = flag
            flag = "O" if flag == "X" else "X"
        #print(m)
        for i in range(3):
            if (m[i][0] == "X" and m[i][1] == "X" and m[i][2] == "X") or (m[0][i] == "X" and m[1][i] == "X" and m[2][i] == "X"):
                return "A"
            elif (m[i][0] == "O" and m[i][1] == "O" and m[i][2] == "O") or (m[0][i] == "O" and m[1][i] == "O" and m[2][i] == "O"):
                return "B"
        if (m[0][0] == "X" and m[1][1] == "X" and m[2][2] == "X") or (m[0][2] == "X" and m[1][1] == "X" and m[2][0] == "X"):
            return "A"
        elif (m[0][0] == "O" and m[1][1] == "O" and m[2][2] == "O") or (m[0][2] == "O" and m[1][1] == "O" and m[2][0] == "O"):
            return "B"
        if len(moves) < 9:
            return "Pending"
        return "Draw"
