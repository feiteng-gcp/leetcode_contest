class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        if len(moves)==9:
            finish = True
        else:
            finish = False
        board = [[-1]*3 for _ in range(3)]
        role = 1
        for x,y in moves:
            board[x][y] = role
            role ^= 1
        win = -1
        for k in range(3):
            if board[k][0] != -1 and board[k][0]==board[k][1]:
                if board[k][2]==board[k][1]:
                    win = board[k][0]
            if board[0][k] != -1 and board[0][k]==board[1][k]:
                if board[1][k]==board[2][k]:
                    win = board[0][k]
        if win==-1:
            if board[0][0] != -1 and board[0][0]==board[1][1] and board[1][1]==board[2][2]:
                win = board[0][0]
            if board[1][1] != -1 and board[0][2]==board[1][1] and board[1][1]==board[2][0]:
                win = board[1][1]
        if finish:
            if win==-1:
                return "Draw"
            elif win==1:
                return "A"
            else:
                return "B"
        else:
            if win ==-1:
                return "Pending"
            elif win ==1:
                return "A"
            else:
                return "B"
