class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        board = [[-1] * 3 for _ in range(3)]
        
        turn = 0
        for move in moves:
            board[move[0]][move[1]] = turn
            turn = (turn + 1) % 2
            
        def win(x):
            if x == 0:
                return "A"
            else:
                return "B"
            
        for x in range(3):
            if board[x][0] >= 0 and board[x][0] == board[x][1] and board[x][0] == board[x][2]:
                return win(board
[x][0])
            if board[0][x] >= 0 and board[0][x] == board[1][x] and board[0][x] == board[2][x]:
                return win(board
[0][x])
        if board[0][0] >= 0 and board[0][0] == board[1][1] and board[0][0] == board[2][2]:
            return win(board[0][0])
        if board[0][2] >= 0 and board[0][2] == board[1][1] and board[0][2] == board[2][0]:
            return win(board[0][2])
        
        if len(moves) == 9:
            return "Draw"
        
        return "Pending"
