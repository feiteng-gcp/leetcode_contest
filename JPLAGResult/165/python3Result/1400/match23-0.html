class Solution:
    def tictactoe(self, moves: List[List[int]]) -> str:
        board = [[-1] * 3 for i in range(3)]
        user = 0
        for move in moves:
            board[move[0]][move[1]] = user
            user = 1 - user
        print(board)
        for i in range(3):
            if board[i][0] == board[i][1] == board[i][2] and board[i][0] != -1:
                return "A" if board[i][0] == 0 else "B"
            if board[0][i] == board[1][i] == board[2][i] and board[0][i] != -1:
                return "A" if board[0][i] == 0 else "B"
        if (board[0][0] == board[1][1] == board[2][2] or board[0][2] == board[1][1] == board[2][0]) and board[1][1] != -1:
            return "A" if board[1][1] == 0 else "B"
        for i in range(3):
            for j in range(3):
                if board[i][j] == -1:
                    return "Pending"
        return "Draw"
