class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        d 
= tomatoSlices - 2 * cheeseSlices
        if d < 0 or d % 2 != 0:
            return []
        x = d // 2
        y = cheeseSlices - x
        if y < 0:
            return []
        return[x, y]
