class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        diff = tomatoSlices - 2 * cheeseSlices
        if diff < 0 or diff & 1 == 1:
            return []
        x = diff // 2
        y = cheeseSlices - x
        if y >= 0:
            return [x, y]
        else:
            return []
