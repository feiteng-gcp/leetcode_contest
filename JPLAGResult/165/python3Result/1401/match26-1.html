class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        
if tomatoSlices % 2 != 0:
            return []
        a = tomatoSlices // 2
        a -= cheeseSlices
        b = cheeseSlices - a
        if a < 0 or b < 0:
            return []
        return [a, b]
