class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        if tomatoSlices % 2:
            return []
        
        x = tomatoSlices // 2 - cheeseSlices
        y = 2 * cheeseSlices - tomatoSlices // 2
        
        if x < 0 or y < 0:
            return []
        
        return [x, y]
