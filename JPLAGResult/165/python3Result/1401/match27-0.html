class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        t = tomatoSlices
        c = cheeseSlices
        if t % 2:
            return []
        t = t//2
        if t > 2 * c or t < c:
            return []
        return [t - c, 2 * c - t]
