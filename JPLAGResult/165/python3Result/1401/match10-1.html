class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        # 4x + 2y = tomatoSlices
        # x + y = cheeseSlices
        _2x = tomatoSlices - 2 * cheeseSlices
        if _2x % 2: return []
        x = _2x // 2
        y = cheeseSlices - x
        
        return [x, y] if x >= 0 and y >= 0 else []
