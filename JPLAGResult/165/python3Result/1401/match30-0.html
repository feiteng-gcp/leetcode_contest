class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        # 4J + 2S == t
        # J + S == c
        # 2 J == t - 2c
        t = tomatoSlices
        c = cheeseSlices
        
        num1 = t - 2 * c
        if num1 < 0 or num1 % 2 != 0:
            return []
        J = num1 // 2
        S = c - J
        if S < 0:
            return []
        return [J, S]
