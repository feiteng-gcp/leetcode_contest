class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        if tomatoSlices % 2 != 0:
            return []
        tomatoSlices //= 2
        jumbo = tomatoSlices - cheeseSlices
        if jumbo < 0:
            return []
        small = cheeseSlices*2 - tomatoSlices
        if small < 0:
            return []
        return [jumbo, small]
