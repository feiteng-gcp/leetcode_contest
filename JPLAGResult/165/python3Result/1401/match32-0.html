class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        temp = (tomatoSlices-2*cheeseSlices)
        if temp % 2 != 0:
            return []
        jumbo = temp//2
        if jumbo < 0:
            return []
        small = cheeseSlices - jumbo
        if small < 0:
            return []
        return [jumbo, small]
