class Solution:
    def numOfBurgers(self, tomatoSlices: int, cheeseSlices: int) -> List[int]:
        if tomatoSlices % 2 != 0:
            return []
        j, s = tomatoSlices // 2 - cheeseSlices, 2 * cheeseSlices - tomatoSlices // 2
        if j < 0 or s < 0:
            return []
        return [j, s]
