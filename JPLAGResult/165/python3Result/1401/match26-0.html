class Solution:
    def numOfBurgers(self, tomatoSlices: int, N: int):
        if tomatoSlices % 2 == 1:
            return []
        tomatoSlices //= 2
        jumbo = tomatoSlices - N
        small = N - jumbo
        if jumbo < 0 or small < 0:
            return []
        return [jumbo, small]
