# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    # self no cam, has cam
    def f(self, root):
        if (root is None):
            return 0, 0, 0
        if (not root.left) and (not root.right):
            return 1, 1, 0
        if (root.left):
            l0, l1, l2 = self.f(root.left)
        if (root.right):
            r0, r1, r2 = self.f(root.right)
        if (not root.right):
            return l1, min([l0, l1, l2])+1, min([l0, l1])
        if (not root.left):
            return r1, min([r0, r1, r2])+1, min([r0, r1])
        
        z0 = min([l1 + r0, l0 + r1, l1 + r1])
        z1 = min([l0, l1, l2]) + min([r0, r1, r2]) + 1
        z2 = min([l0, l1]) + min([r0, r1])
        
        return z0, z1, z2



    def minCameraCover(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        r0, r1, r2 = self.f(root)

        return min(r0, r1)
