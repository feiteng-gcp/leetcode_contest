# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    
def helper(self, root):
        if root is None:
            return 0, 0, 10000
        
        la, lb, lc = self.helper(root.left)
        ra, rb, rc = self.helper(root.right)
        a = lb + rb
        b = min(lb + rc, rb + lc)
        c = 1 + la + ra
        b = min(b, c)
        a = min(a, b)
        #print(a, b, c)
        #assert a <= b <= c
        return a, b, c
    
    def minCameraCover(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        a, b, c = self.helper(root)
        return min(b, c)
