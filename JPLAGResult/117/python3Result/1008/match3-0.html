# Definition for a binary tree node.
class TreeNode:
    def __init__(self, x):
        self.val = x
        self.left = None
        self.right = None
        
class Solution:
    def minCameraCover(self, root):
        if root.left==None and root.right == None:
            return 1
        res = [0]
        dic = {}
        def dfs(node,parent):
            if node == None:
                return
            dic[node] = False
            dfs(node.left,node)
            dfs(node.right,node)
            if not node.left and not node.right:
                return
            
            if (node.left and not dic[node.left]) or (node.right and not dic[node.right]):
                
                dic[node] = True
                if node.left !=None:
                    dic[node.left] = True
                if node.right !=None:
                    dic[node.right] = True
                if parent!=None:
                    dic[parent] = True
                res[0]+=1
                
        dfs(root,None)
        if dic[root] == False:
            res[0]+=1
        return res[0]
