# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    num=1
    dicset={}
    dicnotset={}
    dicundefine={}
    def minCameraCover(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.num=1
        self.dic={}
        self.dicnotset={}
        self.dicundefine={}
        self.bl(root)
        s=self.findundefine(root)
        return s
        
    def bl(self,root):
        root.val=self.num
        self.num+=1
        if not root.left==None:
            self.bl(root.left)
        if not root.right==None:
            self.bl(root.right)
    
    def findundefine(self,root):
        try:
            self.dicundefine[root.val]
        except:
            self.dicundefine[root.val]=-1
        if self.dicundefine[root.val]>-1:
            return self.dicundefine[root.val]
        s=0
        if root.right==None and root.left==None:
            self.dicundefine[root.val]=1
            return 1
        elif root.left==None:
            s1=0
            s2=0
            s3=0
            s4=0
            sl=0
            sr=0
            if not root.right.left==None:
                sl=self.findundefine(root.right.left)
                if not root.right.left.left==None:
                    s1=self.findundefine(root.right.left.left)
                if not root.right.left.right==None:
                    s2=self.findundefine(root.right.left.right)
            if not root.right.right==None:
                sr=self.findundefine(root.right.right)
                if not root.right.right.left==None:
                    s3=self.findundefine(root.right.right.left)
                if not root.right.right.right==None:
                    s4=self.findundefine(root.right.right.right)
            sl=min(sl,s1+s2)
            sr=min(sr,s3+s4)
            s=1+sl+sr
            self.dicundefine[root.val]=s
            return s
        elif root.right==None:
            s1=0
            s2=0
            s3=0
            s4=0
            sl=0
            sr=0
            if not root.left.left==None:
                sl=self.findundefine(root.left.left)
                if not root.left.left.left==None:
                    s1=self.findundefine(root.left.left.left)
                if not root.left.left.right==None:
                    s2=self.findundefine(root.left.left.right)
            if not root.left.right==None:
                sr=self.findundefine(root.left.right)
                if not root.left.right.left==None:
                    s3=self.findundefine(root.left.right.left)
                if not root.left.right.right==None:
                    s4=self.findundefine(root.left.right.right)
            sl=min(sl,s1+s2)
            sr=min(sr,s3+s4)
            s=1+sl+sr
            self.dicundefine[root.val]=s
            return s
        else:
            s1=0
            s2=0
            s3=0
            s4=0
            s5=0
            sl=0
            sr=0
            sl=self.findundefine(root.left)
            if not root.left.left==None:
                s1=self.findundefine(root.left.left)
            if not root.left.right==None:
                s2=self.findundefine(root.left.right)
            sr=self.findundefine(root.right)
            if not root.right.left==None:
                s3=self.findundefine(root.right.left)
            if not root.right.right==None:
                    s4=self.findundefine(root.right.right)
            sl=min(sl,s1+s2)
            sr=min(sr,s3+s4)
            s=1+sl+sr
            
            s1=0
            s2=0
            s3=0
            s4=0
            s5=0
            sl=0
            sr=0
            if not root.left.left==None:
                sl=self.findundefine(root.left.left)
                if not root.left.left.left==None:
                    s1=self.findundefine(root.left.left.left)
                if not root.left.left.right==None:
                    s2=self.findundefine(root.left.left.right)
            if not root.left.right==None:
                sr=self.findundefine(root.left.right)
                if not root.left.right.left==None:
                    s3=self.findundefine(root.left.right.left)
                if not root.left.right.right==None:
                    s4=self.findundefine(root.left.right.right)
            sl=min(sl,s1+s2)
            sr=min(sr,s3+s4)
            s5=self.findundefine(root.right)
            s=min(s,1+sl+sr+s5)
            
            s1=0
            s2=0
            s3=0
            s4=0
            s5=0
            sl=0
            sr=0
            if not root.right.left==None:
                sl=self.findundefine(root.right.left)
                if not root.right.left.left==None:
                    s1=self.findundefine(root.right.left.left)
                if not root.right.left.right==None:
                    s2=self.findundefine(root.right.left.right)
            if not root.right.right==None:
                sr=self.findundefine(root.right.right)
                if not root.right.right.left==None:
                    s3=self.findundefine(root.right.right.left)
                if not root.right.right.right==None:
                    s4=self.findundefine(root.right.right.right)
            sl=min(sl,s1+s2)
            sr=min(sr,s3+s4)
            s5=self.findundefine(root.left)
            s=min(s,1+sl+sr+s5)
            self.dicundefine[root.val]=s
            return s
