# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def go(self, root, cam, watched):
        if root is None:
            return cam if cam == watched else 200000
        
        if root.left is None and root.right is None:
            return cam if watched else 20000 
        
        if root.val == 0:
            root.val = len(self.dp)
            self.dp.append([None] * 4)
            
        dp = self.dp[root.val]
            
        i = cam * 2 + watched
        if dp[i] is None:
            if cam:
                x = 0
                if root.left is not None:
                    x = min(self.go(root.left, 0, 1) , self.go(root.left, 1, 1))
                y = 0
                if root.right is not None:
                    y = min(self.go(root.right, 0, 1), self.go(root.right, 1, 1))
                dp[i] = x + y + 1
            else:
                if watched:
                    x = 0
                    if root.left is not None:
                        x = min(self.go(root.left, 0, 0) , self.go(root.left, 1, 1))
                    if root.right is not None:
                        x += min(self.go(root.right, 0, 0) , self.go(root.right, 1, 1))
                    
                    dp[i] = x;
                    
                else:
                    x = 20000
                    x = min(x, self.go(root.left, 0, 0) + self.go(root.right, 1, 1))
                    x = min(x, self.go(root.left, 1, 1) + self.go(root.right, 0, 0))
                    x = min(x, self.go(root.left, 1, 1) + self.go(root.right, 1, 1))
                    dp[i] = x;
                        
                        
                
        #print(dp[i])
            
        return dp[i]
                
    dp = []           
    
        
    def minCameraCover(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        return 0 if root is None else min(self.go(root, 0, 0), self.go(root, 1, 1))
