# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def __init__(self):
        self.count = 0
    
    def minCameraCover(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        
        self.dfs(root, 0)
        return self.count
        
    def dfs(self, root, dep):
        if not root:
            return 0
        
        left = self.dfs(root.left, dep+1)
        right = self.dfs(root.right, dep+1)

        if left == 1 or right == 1:
            self.count += 1
            return 2
        elif left == 2 or right == 2:
            return 0

        if dep == 0:
            self.count += 1
            
        return 1
