# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def minCameraCover(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        def f(node):
            if (node.left==None) and (node.right==None):
                node.val=[1,1,0]
                return None
            if node.left==None:
                f(node.right)
                node.val=[1+node.right.val[2],min(node.right.val[0],1+node.right.val[2]),min(node.right.val[0],1+node.right.val[2],node.right.val[1])]
                return None
            elif node.right==None:
                f(node.left)
                node.val=[1+node.left.val[2],min(node.left.val[0],1+node.left.val[2]),min(node.left.val[0],1+node.left.val[2],node.left.val[1])]
                return None
            f(node.left)
            f(node.right)
            node.val=[1+node.left.val[2]+node.right.val[2],min(1+node.left.val[2]+node.right.val[2],node.left.val[0]+node.right.val[1],node.left.val[1]+node.right.val[0]),min(node.left.val[1]+node.right.val[1],1+node.left.val[2]+node.right.val[2],node.left.val[0]+node.right.val[1],node.left.val[1]+node.right.val[0])]
            
        import sys
        sys.setrecursionlimit(2000)
        f(root)
        return root.val[1]
