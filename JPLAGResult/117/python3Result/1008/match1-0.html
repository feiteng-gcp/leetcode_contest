# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def minCameraCover(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """

        def magic(u, is_root):
            if not u or (not u.left and not u.right):
                return 0
            l_ans = magic(u.left, False)
            r_ans = magic(u.right, False)
            if (u.left and u.left.val == 0) or (u.right and u.right.val == 0):
                u.val = 2
            elif (u.left and u.left.val == 2) or (u.right and u.right.val == 2):
                u.val = 1
            else:
                if is_root:
                    u.val = 2
            return l_ans + r_ans + (1 if u.val == 2 else 0)
        if root and (not root.left) and (not root.right):
            return 1
        return magic(root, True)
