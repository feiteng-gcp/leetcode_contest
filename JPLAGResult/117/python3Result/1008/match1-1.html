# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def minCameraCover(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.installCamera(root)
        return self.countCamera(root) + int(root.val == 0)
    
    def installCamera(self, root):
        if not root.left and not root.right:
            return
        if root.left:
            self.installCamera(root.left)
        if root.right:
            self.installCamera(root.right)
        
if (root.left and root.left.val == 0) or (root.right and root.right.val == 0):
            root.val = 1
        elif (root.left and root.left.val == 1) or (root.right and root.right.val == 1):
            root.val = -1
    
    def countCamera(self, root):
        if root is None:
            return 0
        return self.countCamera(root.left) + self.countCamera(root.right) + int(root.val == 1)
