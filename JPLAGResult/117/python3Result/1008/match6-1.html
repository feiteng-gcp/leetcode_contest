# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
import heapq
class Solution:
    def connectP(self, node, level):
        node.nums = 0
        node.level = level
        node.visit = False
        if node.left != None:
            node.left.p = node
            self.connectP(node.left, level - 1)
            node.nums += 1
        if node.right != None:
            node.right.p = node
            self.connectP(node.right, level - 1)
            node.nums += 1
        if node.nums == 0:
            heapq.heappush(self.lowest, (node.level, self.id))
            self.nodemap[self.id] = node
            self.id += 1
            # self.lowest.append(node)
    
def minCameraCover(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        root.p = None
        self.lowest = []
        self.id = 0
        self.nodemap = {}
        self.connectP(root, 1)
        ans = 0
        while len(self.lowest) > 0:
            _, nodeid = heapq.heappop(self.lowest)
            node = self.nodemap[nodeid]
            if node.visit:
                # print("ojbk")
                continue
            node.visit = True
            node = node.p
            if node != None:
                node.visit = True
                if node.left:
                    node.left.visit = True
                if node.right:
                    node.right.visit = True
                if node.p:
                    node.p.visit = True
                    if node.p.p:
                        node.p.p.nums -= 1
                        if node.p.p.nums == 0:
                            heapq.heappush(self.lowest, (node.p.p.level, self.id))
                            # heapq.heappush(self.lowest, (node.level, self.id))
                            self.nodemap[self.id] = node.p.p
                            self.id += 1
                            # self.lowest.append(node.p.p)
                            # print("asd")
            # print(len(self.lowest))
            ans += 1
        return ans
