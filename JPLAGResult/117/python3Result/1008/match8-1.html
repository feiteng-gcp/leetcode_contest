# class TreeNode:
#     def __init__(self, x):
#         self.val = x

    def minCameraCover(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        if not root:
            return 0
        root.p = None
        self.e = []
        self.init(root, 0)
        res = 0
        self.e.sort(key=lambda x: x[0])
        while self.e:
            e1 = set()

                n, ptr = self.e.pop()

                    continue
                if ptr.left and ptr.left.val == 1:
                    continue
                if ptr.right and ptr.right.val == 1:
                    continue
                if (n + 1, ptr.left) in self.e:
                    self.e.remove((n + 1, ptr.left))
                    res += 1
                    ptr.val = 1
                    continue
                if (n + 1, ptr.right) in self.e:
                    self.e.remove((n + 1, ptr.right))
                    res += 1
                    ptr.val = 1
                    continue

                    ptr.val = 1

                        self.e.remove((n + 1, ptr.left))

                        self.e.remove((n + 1, ptr.right))
                    res += 1
                elif ptr.p.val != 1:

                    ptr.p.val = 1

        root.n = n
