# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isUnivalTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        if not root:
            return True
        self.val = root.val
        return self.rec(root)
        
    
    def rec(self, node):
        if not node:
            return True
        if node.val != self.val:
            return False
        return self.rec(node.left) and self.rec(node.right)
