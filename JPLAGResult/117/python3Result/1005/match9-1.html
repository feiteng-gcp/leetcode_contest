# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def isUnivalTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        def univalue(node, i):
            if node is None:
                return True
            return node.val == i and univalue(node.left, i) and univalue(node.right, i)
        return univalue(root, root.val)
