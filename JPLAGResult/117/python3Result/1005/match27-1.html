from collections import deque

class Solution:
    def isUnivalTree(self, root):
        QUE= deque()
        QUE.append(root)
        V=root.val

        while QUE:
            x
=QUE.pop()
            if x.val!=V:
                return False

            if x.left!=None:
                QUE.append(x.left)
            if x.right!=None:
                QUE.append(x.right)

        return True
