# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isUnivalTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        if root is None: return True
        val
=[root.val]
        def dfs(cur):
            if cur is None: return True
            return cur.val==val[0] and dfs(cur.left) and dfs(cur.right)
        
        return dfs(root)
