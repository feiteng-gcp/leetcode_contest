# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isUnivalTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        val = root.val
        return self.isUniVal(root.left, val) and self.isUniVal(root.right, val)
    
    
def isUniVal(self, root, val):
        if root is None:
            return True
        if root.val != val:
            return False
        return self.isUniVal(root.left, val) and self.isUniVal(root.right, val)
