# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isUnivalTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        val = set()
        
        def dfs(node):
            val.add(node.val)
            if len(val) >= 2: return False
            if node.left and not dfs(node.left):
                return False
            if node.right and not dfs(node.right):
                return False
            return True
        
        return dfs(root)
