# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isMatchTree(self, val, node):
        if node is None:
            return True
        if node.val != val:
            return False
        return self.isMatchTree(val, node.left) and self.isMatchTree(val, node.right)
        
        
    def isUnivalTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        return self.isMatchTree(root.val, root)
