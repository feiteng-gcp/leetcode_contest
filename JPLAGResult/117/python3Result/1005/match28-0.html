# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isUnivalTree(self, root, val=None):
        """
        :type root: TreeNode
        :rtype: bool
        """
        if root is None:
            return True
        if val is not None and root.val != val:
            return False
        return self.isUnivalTree(root.left, root.val) and self.isUnivalTree(root.right, root.val)
