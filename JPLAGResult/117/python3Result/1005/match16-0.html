# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isUnivalTree(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """        
        return self.check(root, root.val)
    
    def check(self, root, val):
        if not root:
            return True
        
        if root.val != val:
            return False
        
        return self.check(root.left, val) and self.check(root.right, val)
