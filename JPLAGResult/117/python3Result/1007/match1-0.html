class Solution:
    def numsSameConsecDiff(self, N, K):
        """
        :type N: int
        :type K: int
        :rtype: List[int]
        """
        if N < 1:
            return []
        elif N == 1:
            return [i for i in range(10)]

        vals = set([str(i) for i in range(1, 10)])
        
        for n in range(N - 1):
            new_vals = set()
            for v in vals:

                # try to add K and subtract K
                
                if int(v[-1]) + K < 10:
                    new_vals.add(v + str(int(v[-1]) + K))
                
                if int(v[-1]) - K >= 0:
                    new_vals.add(v + str(int(v[-1]) - K))
            vals = new_vals
    
        return sorted([int(i) for i in vals])
