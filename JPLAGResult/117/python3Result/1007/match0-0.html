class Solution:
    def numsSameConsecDiff(self, N, K):
        """
        :type N: int
        :type K: int
        :rtype: List[int]
        """
        if N == 1:
            return list(range(10))
        prevRes = self.numsSameConsecDiff(N - 1, K)
        res = []
        for num in prevRes:
            if num == 0:
                continue
            string = str(num)
            lastDigit = int(string[-1])
            if lastDigit + K < 10:
                res.append(int(string + str(lastDigit + K)))
            if lastDigit - K >= 0:
                res.append(int(string + str(lastDigit - K)))
        return list(set(res))
