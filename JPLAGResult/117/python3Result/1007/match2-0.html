class Solution:
    def numsSameConsecDiff(self, N, K):
        """
        :type N: int
        :type K: int
        :rtype: List[int]
        """
        ans = []
        for i in range(10):
            if N != 1 and i == 0:
                continue
            q = [i]
            count = 0
            while q:
                count += 1
                if count == N:
                    ans += q
                    break
                tmpQ = []
                for num in q:
                    last = num % 10
                    if last + K < 10:
                        newNum = num * 10 + (last + K)
                        tmpQ.append(newNum)
                    if last - K >= 0:
                        newNum = num * 10 + (last - K)
                        tmpQ.append(newNum)
                q = tmpQ
        return list(set(ans))
