class Solution:
    def numsSameConsecDiff(self, N, K):
        """
        :type N: int
        :type K: int
        :rtype: List[int]
        """
        def getNums(num):
            nums = []
            if num + K < 10:
                nums.append(num + K)
            if num - K >= 0 and num - K != num + K:
                nums.append(num - K)
            return nums
        
        if N == 1:
            return [n for n in range(10)]
        
        digits = [n for n in range(1, 10)]
        res = []
        while digits:
            d = digits.pop(0)
            print(d)
            if len(str(d)) == N:
                res.append(d)
            else:
                nextds = getNums(d % 10)
                print('%d got back %s' % (d % 10, str(nextds)))
                if len(nextds) > 0:
                    digits += [int(str(d) + str(nd)) for nd in nextds]
                
        return list(set(res))
