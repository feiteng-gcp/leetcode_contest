import copy

class Solution:
    def numsSameConsecDiff(self, N, K):
        """
        :type N: int
        :type K: int
        :rtype: List[int]
        """
        if N == 1:
            return [k for k in range(10)]
        first_nums = [k for k in range(1, 10)]
        res = []
        for first in first_nums:
            num_set = set([first])
            for _ in range(1, N):
                new_num_set = copy.copy(num_set)
                for num in num_set:
                    new_num_set.remove(num)
                    tmp 
= num % 10
                    if tmp - K >= 0:
                        new_num = num * 10 + tmp - K
                        new_num_set.add(new_num)
                    if tmp + K < 10:
                        new_num = num * 10 + tmp + K
                        new_num_set.add(new_num)
                num_set = new_num_set
            res += list(num_set)
        return res
