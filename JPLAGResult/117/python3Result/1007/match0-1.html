class Solution:
    def numsSameConsecDiff(self, N, K):
        """
        :type N: int
        :type K: int
        :rtype: List[int]
        """
        if N == 1:
            return list(range(10))
        t = self.numsSameConsecDiff(N-1, K)
        ans = []
        for x in t:
            if x == 0:
                continue
            xx = x % 10
            if (xx + K < 10):
                ans.append(x*10+xx+K)
            if (xx - K >= 0):
                ans.append(x*10+xx-K)
        return list(set(ans))
