class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        def ff(w):
            x2 = ""
            for c in w.upper():
                if c not in "AEIOUaeiou":
                    x2 += c
                else:
                    x2 += '*'
            return x2
        d1 = {}
        d2 = {}
        stx = set()
        for w in wordlist:
            stx.add(w)
            x1 = w.upper()
            if x1 not in d1:
                d1[x1] = w
            x2 = ff(w)
            if x2 not in d2:
                d2[x2] = w
        ans = []
        
for q in queries:
            if q in stx:
                ans.append(q)
                continue
            if q.upper() in d1:
                ans.append(d1[q.upper()])
                continue
            xxx = ff(q)
            
if xxx in d2:
                ans.append(d2[xxx])
                continue
            ans.append("")
        return ans
