class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        set_word = set(wordlist)
        dic_word = {}
        word_list_lower = {}
        res = []
        n = len(wordlist)
        for i in range(n):
            lower = wordlist[i].lower()
            if lower not in word_list_lower: word_list_lower[lower] = i
            curr_word = ''
            for letter in lower:
                if letter in ['a','e','i','o','u']: curr_word += '*'             
                else: curr_word += letter
            if curr_word not in dic_word: dic_word[curr_word] = i
        for query in queries:
            if query in set_word: 
                res.append(query)
                continue
            query_lower = query.lower()
            if query_lower in word_list_lower: 
                res.append(wordlist[word_list_lower[query_lower]])
                continue
            query_upd = ''
            for s in query_lower:
                if s in ['a','e','i','o','u']: query_upd += '*'
                else: query_upd += s
            if query_upd in dic_word:
                res.append(wordlist[dic_word[query_upd]])
                continue
            res.append('')
        return res
