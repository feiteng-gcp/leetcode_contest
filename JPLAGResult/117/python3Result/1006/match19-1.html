from collections import defaultdict

class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        
        words = set(wordlist)
        bank1 = defaultdict(list)
        bank2 = defaultdict(list)
        
        def devowel(x):
            for c in 'aeiou':
                x = x.replace(c, '.')
            return x
        
        for w in wordlist:
            lowered = w.lower()
            bank1[lowered].append(w)
            bank2[devowel(lowered)].append(w)
            
        ret = []
        
for q in queries:
            if q in words:
                ret.append(q)
                continue
            lower_q = q.lower()
            if lower_q in bank1:
                ret.append(bank1[lower_q][0])
                continue
            
            qq = devowel(lower_q)
            
if qq in bank2:
                ret.append(bank2[qq][0])
                continue
                
            ret.append("")
        return ret
