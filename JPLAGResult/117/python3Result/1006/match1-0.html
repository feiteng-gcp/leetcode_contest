class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        ans = []
        # First Level
        s = set(wordlist)
        # Second Level
        d = {}
        for w in wordlist:
            if w.lower() not in d:
                d[w.lower()] = w
        # Third Level
        ad = {}
        for w in wordlist:
            aw = w.lower()
            for ch in 'eiou':
                aw = aw.replace(ch, 'a')
            if aw not in ad:
                ad[aw] = w
        for q in queries:
            # I
            if q in s:
                ans.append(q)
                continue
            # II
            if q.lower() in d:
                ans.append(d[q.lower()])
                continue
            # III
            aq = q.lower()
            for ch in 'eiou':
                aq = aq.replace(ch, 'a')
            if aq in ad:
                ans.append(ad[aq])
                continue
            # IV
            ans.append('')
        return ans
