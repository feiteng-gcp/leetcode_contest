class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        words_case_sensitive = set(wordlist)
        words_case_insensitive = {}
        words_vowel_errors = {}
        
        def get_ci(s):
            return s.lower()
        
        def get_ve(s):
            ve = ''
            for ch in s.lower():
                if ch in 'aeiou': ch = '_'
                ve += ch
            return ve
        
        for word in wordlist:
            ci_word = get_ci(word)
            if ci_word not in words_case_insensitive:
                words_case_insensitive[ci_word] = word
            
            ve_word = get_ve(word)
            if ve_word not in words_vowel_errors:
                words_vowel_errors[ve_word] = word
        
        ans = []
        for word in queries:
            if word in words_case_sensitive:
                ans.append(word)
                continue
            
            ci_word = get_ci(word)
            if ci_word in words_case_insensitive:
                ans.append(words_case_insensitive[ci_word])
                continue
            
            ve_word = get_ve(word)
            if ve_word in words_vowel_errors:
                ans.append(words_vowel_errors[ve_word])
                continue
            
            ans.append('')
        
        return ans
