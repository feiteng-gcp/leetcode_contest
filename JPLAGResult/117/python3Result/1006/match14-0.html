class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        h1 = {}
        h2 = {}
        h3 = {}
        for s in wordlist:
            h1[s] = s
            t = s
            t = t.lower()
            if t not in h2:
                h2[t] = s
            r = t
            r = r.replace('a', 'a')
            r = r.replace('e', 'a')
            r = r.replace('i', 'a')
            r = r.replace('o', 'a')
            r = r.replace('u', 'a')
            if r not in h3:
                h3[r] = s
        def conv(s):
            if s in h1:
                return h1[s]
            t = s
            t = t.lower()
            if t in h2:
                return h2[t]
            r = t
            r = r.replace('a', 'a')
            r = r.replace('e', 'a')
            r = r.replace('i', 'a')
            r = r.replace('o', 'a')
            r = r.replace('u', 'a')
            if r in h3:
                return h3[r]
            return ""
        return [conv(s) for s in queries]
