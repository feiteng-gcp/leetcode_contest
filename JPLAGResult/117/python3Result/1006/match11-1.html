class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        let_set = set(['a', 'e', 'i', 'o', 'u'])
        ex_w = set()
        w_low = {}

        wordlist_low = [s.lower() for s in wordlist]
        wordlist_low_val = {}

        for w in wordlist:
            ex_w.add(w)

            lw = w.lower()
            if lw not in w_low:
                w_low[lw] = w

            llw = self.replace
(lw)

            if llw not in wordlist_low_val:
                wordlist_low_val[llw] = w

        res = []
        for q in queries:
            if q in ex_w:
                res.append(q)
                continue

            ql = q.lower()
            if ql in w_low:
                res.append(w_low[ql])
                continue

            ql = self.replace(ql)

            if ql in wordlist_low_val:
                res.append(wordlist_low_val[ql])
            else:
                res.append("")

        return res

    def replace(self, s):
        for c in ['a', 'e', 'i', 'o', 'u']:
            s = s.replace(c, '_')
        return s
