import re

class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        
        ans = []
        same_words = set(wordlist)
        cap = {}
        vowel = {}
        
        for word in reversed(wordlist):
            l = word.lower()
            cap[l] = word
            v = re.sub(r'[aeiou]', '*', l)
            print(l, v)
            vowel[v] = word
        
        for query in queries:
            if query in same_words:
                ans.append(query)
            elif query.lower() in cap:
                ans.append(cap[query.lower()])
            else:
                v = re.sub(r'[aeiou]', '*', query.lower())
                if v in vowel:
                    ans.append(vowel[v])
                else:
                    ans.append('')
                    
        return ans
