class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        valid, lower, vowel, ans = set(wordlist), {}, {}, []
        for w in wordlist:
            temp = "".join(W if 97 <= ord(W) <= 122 else chr(ord(W) + 32) for W in w)
            if temp not in lower:
                lower[temp] = w
            temp = []
            for W in w:
                if W in {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'}:
                    temp.append(".")
                else:
                    temp.append(W if 97 <= ord(W) <= 122 else chr(ord(W) + 32))
            temp = "".join(temp)
            if temp not in vowel:
                vowel[temp] = w
        for q in queries:
            if q in valid:
                ans.append(q)
            else:
                temp = "".join(Q if 97 <= ord(Q) <= 122 else chr(ord(Q) + 32) for Q in q)
                if temp in lower:
                    ans.append(lower[temp])
                    continue
                temp = []
                for Q in q:
                    if Q in {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'}:
                        temp.append(".")
                    else:
                        temp.append(Q if 97 <= ord(Q) <= 122 else chr(ord(Q) + 32))
                temp = "".join(temp)
                if temp in vowel:
                    ans.append(vowel[temp])
                    continue
                ans.append("")
        return ans
