class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """

        wds = set
(wordlist)
        lowerwds = [x.lower() for x in wordlist]
        cap = collections.defaultdict(list)
        rmvowelwds = collections.defaultdict(list)
        vowels = {'a', 'e', 'i', 'o', 'u'}

        def rmvowel(val):
            return ''.join([x if x not in vowels else '*' for x in val.lower()])

        for i, v in enumerate(lowerwds):
            cap[v].append(i)
            rmvowelwds[rmvowel(v)].append(
i)

        ans = []
        for q in queries:
            if q in wds:
                ans.append(q)
            elif q.lower() in cap:
                ans.append(wordlist[cap[q.lower()][0]])
            else:
                u = rmvowel(q)
                if u in rmvowelwds:
                    ans.append(wordlist[rmvowelwds[u][0]])
                else:
                    ans.append('')

        return ans
