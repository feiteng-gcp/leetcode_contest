class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        wordset = set(wordlist)
        worddict = {}
        vowels = set(['a', 'e', 'i', 'o', 'u'])
        
        for w in wordlist:
            w_lower = w.lower()
            if w_lower not in worddict:
                worddict[w_lower] = w
                
            w_chars = tuple( [c if c not in vowels else ' ' for c in list(w_lower) ] )
            if w_chars not in worddict:
                worddict[w_chars] =
 w
        
        ret = []
        for w in queries:
            if w in wordset:
                ret.append(w)
            else:
                w_lower = w.lower()
                if w_lower in worddict:
                    ret.append(worddict[w_lower])
                else:
                    w_chars = tuple( [c if c not in vowels else ' ' for c in list
(w_lower) ] )
                    if w_chars in worddict:
                        ret.append(worddict[w_chars])
                    else:
                        ret.append("")
                        
        return ret
