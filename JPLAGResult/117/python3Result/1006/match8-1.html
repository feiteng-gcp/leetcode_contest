class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        res = []
        
        os = set(wordlist)
        cs, vs = {}, {}
        
        for word in wordlist:
            low_word = word.lower()
            if low_word not in cs:
                cs[low_word] = word
                
            pro = list(low_word)
            for i in range(len(pro)):
                if pro[i] in 'aeiou':
                    pro[i] = '*'
            pro_word = ''.join(pro)
            if pro_word not in vs:
                vs[pro_word] = word

        
for q in queries:
            if q in os:
                res.append(q)
                continue
            
            q = q.lower()
            if q in cs:
                res.append(cs[q])
                continue
            
            w = list(q)
            for i in range(len(w)):
                if w[i] in 'aeiou':
                    w[i] = '*'
            q = ''.join
(w)
            if q in vs:
                res.append(vs[q])
                continue
            
            res.append('')
            
        return res
