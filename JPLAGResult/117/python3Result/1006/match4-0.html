class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        wordSet = set(wordlist)
        # convert wordlist to a dictionary
        # key: lower-case work
        # value: original wordlist
        wordDict = collections.defaultdict(list)
        for word in wordlist:
            lower_word = word.lower()
            wordDict[lower_word].append(word)
            
        # vowel replace dict
        wordDictVow = collections.defaultdict(list)
        for word in wordlist:
            newword = word.lower()
            for char in 'aeiou':
                newword = newword.replace(char, ' ')
            wordDictVow[newword].append(word)
        
        res = []
        for query in queries:
            # exactly match
            if query in wordSet:
                res.append(query)
            # capitlization
            elif query.lower() in wordDict:
                res.append(wordDict[query.lower()][0])
            # replace vowel 
            else:
                new_query = query.lower()
                for char in 'aeiou':
                    new_query = new_query.replace(char, ' ')
                if new_query in wordDictVow:
                    res.append(wordDictVow[new_query][0])
                else:
                    res.append('')

        return res
