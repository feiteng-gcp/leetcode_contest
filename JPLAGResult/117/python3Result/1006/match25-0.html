class Solution:
    def spellchecker(self, wordlist, queries):
        exactly = set(wordlist)
        lower = {}
        vowel = {}
        vlist = "aeiou"
        def dfs(w,cw,ind):
            if ind == len(cw):
                if cw not in vowel:
                    vowel[cw] = w
                return
            if cw[ind] in vlist:
                for v in vlist:
                    wnew = cw[:ind]+v+cw[ind+1:]
                    dfs(w,wnew,ind+1)
            else:
                dfs(w,cw,ind+1)
        for w in wordlist:
            lw = w.lower()
            if lw not in lower:
                lower[lw] = w
                dfs(w,lw,0)
        #print(vowel)
        res = []
        #print(res)
        for q in queries:
            
            #print(q,res)
            if q in exactly:
                res.append(q)
                continue
            lq = q.lower()
            if lq in lower:
                res.append(lower[lq])
                continue
            if lq in vowel:
                res.append(vowel[lq])
                continue
            res.append("")
        #print(res)
        return res
