class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        wordset = set(wordlist)
        lowercase = {}
        for i in range(len(wordlist)):
            temp = wordlist[i].lower()
            if temp not in lowercase:
                lowercase[temp] = []
            lowercase[temp].append(i)
        vowel = {'a','e','i','o','u'}
        v = {}
        for index in range(len(wordlist)):
            ans = ''
            for i in wordlist[index].lower():
                if i not in vowel:
                    ans += i
                else:
                    ans += '#'
            if ans not in v:
                v[ans] = []
            v[ans].append(index)
        answer = []
        for word in queries:
            if word in wordset:
                answer.append(word)
            elif word.lower() in lowercase:
                answer.append(wordlist[lowercase[word.lower()][0]])
            else:
                string = ''
                for i in word.lower():
                    if i not in vowel:
                        string += i
                    else:
                        string += '#'
                if string in v:
                    answer.append(wordlist[v[string][0]])
                else:
                    answer.append('')
        return answer
