class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        tt = 'aeiou'
        dic_ori = set(wordlist)
        dic_low = {}
        dic_fu = {}
        res = []
        for word in wordlist:
            tword = word
            tword = word.lower()
            if tword not in dic_low.keys():
                dic_low[tword] = [word]
            else:
                dic_low[tword].append(word)
            
            lword = list(tword)
            for i, c in enumerate(lword):
                if c in tt:
                    lword[i] = '*'
            tword = ''.join(lword)
            if tword not in dic_fu.keys():
                dic_fu[tword] = [word]
            else:
                dic_fu[tword].append(word)
                
        for query in queries:
            if query in dic_ori:
                res.append(query)
                continue
            tq = query.lower()
            if tq in dic_low.keys():
                res.append(dic_low[tq][0])
                continue
            lq = list(tq)
            for i, c in enumerate(lq):
                if c in tt:
                    lq[i] = '*'
            tq = ''.join(lq)
            if tq in dic_fu.keys():
                res.append(dic_fu[tq][0])
                continue
            res.append('')
        
        
        return res
