class Solution:
    
    def normalize(self, word):
        word = word.upper()
        word = word.replace("A", "*")
        word = word.replace("E", "*")
        word = word.replace("I", "*")
        word = word.replace("O", "*")
        word = word.replace("U", "*")
        return word
        
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        
        words = set()
        capits = {}
        vowels = {}
        
        for w in wordlist:
            words.add(w)
            cap_norm = w.upper()
            if cap_norm not in capits:
                capits[cap_norm] = w

            vowel_norm = self.normalize(w)
            if vowel_norm not in vowels:
                vowels[vowel_norm] = w
                
        for (k,v) in vowels.items():
            print(k + ": " + v)
                
        results = []
        for q in queries:
            if q in words:
                results.append(q)
            elif q.upper() in capits:
                results.append(capits[q.upper()])    
            elif self.normalize(q) in vowels:
                results.append(vowels[self.normalize(q)])
            else:
                results.append("")    
                
        return results
