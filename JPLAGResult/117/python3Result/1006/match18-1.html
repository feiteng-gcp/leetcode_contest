class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        org_wordset = set(wordlist)
        cap_wordset = {}
        vow_wordset = {}
        for word in wordlist:
            if word.lower() not in cap_wordset:
                cap_wordset[word.lower()] = word
            new_word 
= self.maskVowels(word)
            if new_word not in vow_wordset:
                vow_wordset[new_word] = word
        ans = []
        for word in queries:
            if word in org_wordset:
                ans.append(word)
            elif word.lower() in cap_wordset:
                ans.append(cap_wordset[word.lower()])
            else:
                new_word = self.maskVowels(word)
                if new_word in vow_wordset:
                    ans.append(vow_wordset[new_word])
                else:
                    ans.append('')
        return ans

    def maskVowels(self, word):
        vowels = {'a', 'e', 'i', 'o', 'u'}
        return ''.join(['*' if c in vowels else c for c in word.lower()])
