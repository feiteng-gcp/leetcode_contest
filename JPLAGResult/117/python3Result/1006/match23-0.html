class Solution:
    def removevowel(self, word):
        copy = []
        for i in range(0, len(word)):
            if word[i] in self.vowel:
                copy.append('_')
            else:
                copy.append(word[i])
        return ''.join(copy)
        
        
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        self.vowel = ['a', 'e', 'i', 'o', 'u']
        dic = {x:x for x in wordlist}
        dicLower = {}
        for i in range(0, len(wordlist)):
            lower = wordlist[i].lower()
            if lower not in dicLower:
                dicLower[lower]= wordlist[i]
        dicVowel = {}
        for i in dicLower:
            vowel = self.removevowel(i)
            if vowel not in dicVowel:
                dicVowel[vowel] = dicLower[i]
        #print(dic)
        #print(dicLower)
        #print(dicVowel)
        result = []
        for query in queries:
            if query in dic:
                result.append(query)
            else:
                l = query.lower()
                if l in dicLower:
                    result.append(dicLower[l])
                else:
                    v = self.removevowel(l)
                    if v in dicVowel:
                        result.append(dicVowel[v])
                    else:
                        result.append("")
        return result
