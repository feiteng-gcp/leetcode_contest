# from collections import defaultdict
class Solution:
    def spellchecker(self, wordlist, queries):
        """
        :type wordlist: List[str]
        :type queries: List[str]
        :rtype: List[str]
        """
        wordset = set(wordlist)
        cpt_map = {}
        remove_map = {}
        for word in wordlist:
            tmp = word.lower()
            if tmp not in cpt_map:
                cpt_map[tmp] = word
            tmp = tmp.replace("a", "@")
            tmp = tmp.replace("e", "@")
            tmp = tmp.replace("i", "@")
            tmp = tmp.replace("o", "@")
            tmp = tmp.replace("u", "@")
            if tmp not in remove_map:
                remove_map[tmp] = word
        ans = []
        for q in queries:
            if q in wordset:
                ans.append(q)
            elif q.lower() in cpt_map:
                ans.append(cpt_map[q.lower()])
            else:
                tmp = q.lower()
                # print(tmp)
                tmp = tmp.replace("a", "@")
                tmp = tmp.replace("e", "@")
                tmp = tmp.replace("i", "@")
                tmp = tmp.replace("o", "@")
                tmp = tmp.replace("u", "@")
                # print(tmp)
                if tmp in remove_map:
                    ans.append(remove_map[tmp])
                else:
                    ans.append("")
        return ans
