class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        Set<String> unique = new HashSet();
        Map<String, String> captilize = new HashMap();
        Map<String, String> vowel = new HashMap();
        for (String word: wordlist) {
            unique.add(word);
            String lower = word.toLowerCase();
            if (!captilize.containsKey(lower)) {
                captilize.put(lower, word);
            }
            String v = replaceVowel(lower);
            if (!vowel.containsKey(v)) {
                vowel.put(v, word);
            }
        }
        String[] res = new String[queries.length];
        int index = 0;
        for (String query: queries) {
            String lower = query.toLowerCase();
            String v = replaceVowel(lower);
            if (unique.contains(query)) {
                res[index++] = query;
            } else if (captilize.containsKey(lower)) {
                res[index++] = captilize.get(lower);
            } else if (vowel.containsKey(v)) {
                res[index++] = vowel.get(v);
            } else {
                res[index++] = "";
            }
        }
        return res;
    }
    
    // input is lower case
    private String replaceVowel(String word) {
        StringBuilder sb = new StringBuilder();
        Set<Character> vowels = new HashSet();
        vowels.add('a');
        vowels.add('e');
        vowels.add('i');
        vowels.add('o');
        vowels.add('u');
        for (int i = 0; i < word.length(); i++) {
            if (vowels.contains(word.charAt(i))) {
                sb.append('*');
            } else {
                sb.append(word.charAt(i));
            }
        }
        return sb.toString();
    }
}
