class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        int n = queries.length;
        
Map<String, String> directMap = new HashMap<>();
        Map<String, String> caseInsMap = new HashMap<>();
        Map<String, String> vowelInsMap = new HashMap<>();
        
        for (String w: wordlist) {
            
directMap.put(w, w);
            String wLower = w.toLowerCase();
            if (!caseInsMap.containsKey(wLower)) {
                caseInsMap.put(wLower, w);
            }
            String wNorm = normalizeVowels(w);
            if (!vowelInsMap.containsKey(wNorm)) {
                vowelInsMap.put(wNorm, w);
            }
        }
        String[] res = new String[n]; 
        for (int i = 0; i < n; ++i) {
            
if (directMap.containsKey(queries[i])) {
                res[i] = queries[i];
                continue;
            } 
            String wLower = queries[i].toLowerCase();
            if (caseInsMap.containsKey(wLower)) {
                res[i] = caseInsMap.get(wLower);
                continue;
            }
            String wNorm = normalizeVowels(queries[i]);
            if (vowelInsMap.containsKey(wNorm)) {
                res[i] = vowelInsMap.get(wNorm);
                continue;
            }
            res[i] = "";
        }
        
        return res;
    }
    
    String normalizeVowels(String s) {
        return s.toLowerCase().replaceAll( "[aeiouAEIOU]", "#" );
        //StringBuilder sb = new StringBuilder();
        //for (int i = 0; i < s.length(); ++i) {
        //}
    }
    
}
