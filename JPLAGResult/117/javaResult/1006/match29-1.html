class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        
Set<String> isSame = new HashSet<>();
        
        // Map<String,List<String>> isCaseSense = new HashMap<>();
        Map<String,String> isCaseSense = new HashMap<>();
        Map<String,String>  vowels = new HashMap<>();
        
        Set<Character> letters = new HashSet<>();
        letters.add('a');
        letters.add('e');
        letters.add('i');
        letters.add('o');
        letters.add('u');
        
        
        for(String word:wordlist){
            isSame.add(word);
            
            String lowcase = word.toLowerCase();
            // List<String> list = isCaseSense.getOrDefault(lowcase,new LinkedList<String>());
            // list.add(word);
            // isCaseSense.put(lowcase,list);
            if(!isCaseSense.containsKey(lowcase)) {
                isCaseSense.put(lowcase,word);
            }
            
            
            
StringBuilder sb = new StringBuilder(lowcase);
            for(int i=0;i<lowcase.length();i++){
                if(letters.contains(lowcase.charAt(i))){
                    sb.replace(i,i+1,"*");
                }
            }
            // List<String> list_vowels  = vowels.getOrDefault(sb.toString(),new LinkedList<String>());
            //   list_vowels.add(word);
            // list_vowels.add();
            if(!vowels.containsKey(sb.toString())) {
                vowels.put(sb.toString(),word);
            }
          
            
        }
        
        String []res = new String[queries.length];
        int count = 0;
        
for(String s:queries){
            if(isSame.contains(s)){
                res[count] = s;
            }else{
                String lowcase = s.toLowerCase();
                if(isCaseSense.containsKey(lowcase)){
                    res[count] = isCaseSense.get(lowcase);
                
}else{
                    StringBuilder sb = new StringBuilder(lowcase);
                    for(int i=0;i<lowcase.length();i++){
                        if(letters.contains(lowcase.charAt(i))){
                            sb.replace(i,i+1,"*");
                        }
                    }
                    
                    if(vowels.containsKey(sb.toString())){
                        res[count] = vowels.get(sb.toString());
                        
                    
}else{
                        
                        res[count] ="";
                    }
                    
                }
                
            }
            
            count++;
            
        }
        
        
        return res;
    }
}
