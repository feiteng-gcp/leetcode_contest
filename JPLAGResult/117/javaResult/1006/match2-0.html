class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        Map<String, String> correct = new HashMap();
        Map<String, String> capital = new HashMap();
        Set<String> exact = new HashSet();
        for (String word : wordlist) {
            exact.add(word);
            String lowerWord = word.toLowerCase();
            if (!capital.containsKey(lowerWord)) {
                capital.put(lowerWord, word);
            }
            String key = key(lowerWord);
            if (!correct.containsKey(key)) {
                correct.put(key, word);
            }
        }
        int index = 0;
        String[] ans = new String[queries.length];
        for (String q : queries) {
            String key = q.toLowerCase();
            String vowelKey = key(key);
            
            if (exact.contains(q)) {
                ans[index++] = q;
            } else if (capital.containsKey(key)) {
                ans[index++] = capital.get(key);
            } else if (correct.containsKey(vowelKey)) {
                ans[index++] = correct.get(vowelKey);
            } else {
                ans[index++] = "";
            }
        }
        return ans;
        
    }
    private String key(String lowerWord) {
        char[] vowels = {'a', 'e', 'i', 'o', 'u'};
        Set<Character> vowelsSet = new HashSet();
        for (char c : vowels) {
            vowelsSet.add(c);
        }
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < lowerWord.length(); i++) {
                if (!vowelsSet.contains(lowerWord.charAt(i))) {
                    sb.append(lowerWord.charAt(i));
                } else {
                    sb.append('a');
                }
            }
        return sb.toString();
        
    }
}
