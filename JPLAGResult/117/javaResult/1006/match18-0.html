class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        Set<Character> vow = new HashSet<>(Arrays.asList('a', 'e', 'i', 'o', 'u'));
        
        Set<String> wordSet = new HashSet();
        Map<String, String> caseMap = new HashMap();
        Map<String, String> signatureMap = new HashMap();
        for (String word : wordlist){
            wordSet.add(word);
            
            String lowercaseWord = word.toLowerCase();
            if (!caseMap.containsKey(lowercaseWord)){
                caseMap.put(lowercaseWord, word);
            }
            
            String signature = unify(lowercaseWord, vow);
            if (!signatureMap.containsKey(signature)){
                signatureMap.put(signature, word);
            }
        }
        
        String[] res = new String[queries.length];
        
        for (int i=0; i<queries.length; i++){
            String query = queries[i];
            
            if (wordSet.contains(query)) {res[i]=query; continue;}
            
            String lowerQuery = query.toLowerCase();
            if (caseMap.containsKey(lowerQuery)) {res[i] = caseMap.get(lowerQuery); continue;}
            
            String signature = unify(lowerQuery, vow);
            if (signatureMap.containsKey(signature)) {res[i] = signatureMap.get(signature); continue;}
            
            res[i] = "";
        }        
        return res;        
    }
    
   //change String to lowercase and all vows to number 1
   private String unify(String str, Set<Character> vow){                
        for (char ch : vow){
            str = str.replace(ch, '1');
        }
        return str;      
    }
}
