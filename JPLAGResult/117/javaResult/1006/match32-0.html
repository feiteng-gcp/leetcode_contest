class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        // build three tables
        Set<String> set = new HashSet<String>(); // all the words in the list
        Map<String, String> map1 = new HashMap<String, String>(); // all the words to lower case 
        Map<String, String> map2 = new HashMap<String, String>(); // all the words with vowels replaced with *
        
        for (String word: wordlist) {
            
            set.add(word);
            String tmp1 = lower(word);
            if (!map1.containsKey(tmp1)) map1.put(tmp1, word);
            String tmp2 = vowel(word);
            if (!map2.containsKey(tmp2)) map2.put(tmp2, word);
            
        }
        
        int n = queries.length;
        String[] res = new String[n];
        
        for (int i = 0; i < n; ++i) {
            String query = queries[i];
            if (set.contains(query)) {
                res[i] = query;
                continue;
            }
            String tmp1 = lower(query);
            if (map1.containsKey(tmp1)) {
                res[i] = map1.get(tmp1);
                continue;
            }
            String tmp2 = vowel(query);
            if (map2.containsKey(tmp2)) {
                res[i] = map2.get(tmp2);
                continue;
            }
            res[i] = "";
        }
        return res;
        
    }
    
    private String lower(String str) {
        // covert to lower case
        return str.toLowerCase();
    }
    
    private String vowel(String str) {
        // replace the vowels
        str = lower(str);
        StringBuilder sb = new StringBuilder();
        int n = str.length();
        for (int i = 0; i < n; ++i) {
            char ch = str.charAt(i);
            if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') sb.append('*');
            else sb.append(ch);
        }
        return sb.toString();
    }
}
