class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        int n = queries.length;
        String[] res = new String[n];
        Map<String, String> map = new HashMap<>();
        Map<String, String> map2 = new HashMap<>();
        Map<String, String> map3 = new HashMap<>();
        preprocess(wordlist, map, map2, map3);
        for (int i = 0; i < n; i++) {
            String query = queries[i];
            if (map3.containsKey(query)) {
                res[i] = query;
                continue;
            }
            String lower = query.toLowerCase();
            if (map.containsKey(lower)) {
                res[i] = map.get(lower);
                continue;
            }
            char[] chars = lower.toCharArray();
            for (int j = 0; j < chars.length; j++) {
                if ("aeiou".contains(String.valueOf(chars[j]))) {
                    chars[j] = '*';
                }
            }
            String re = String.valueOf(chars);
            if (map2.containsKey(re)) {
                res[i] = map2.get(re);
                continue;
            }
            res[i] = "";
        }
        return res;
    }
    private void preprocess(String[] wordlist, Map<String, String> map, Map<String, String> map2, Map<String, String> map3) {
        for (String word : wordlist) {
            map3.put(word, "");
            String lowerWord = word.toLowerCase();
            char[] chars = lowerWord.toCharArray();
            for (int i = 0; i < chars.length; i++) {
                if ("aeiou".contains(String.valueOf(chars[i]))) {
                    chars[i] = '*';
                }
            }
            if (!map.containsKey(lowerWord)) {
                map.put(lowerWord, word);
            }
            String re = String.valueOf(chars);
            if (!map2.containsKey(re)) {
                map2.put(re, word);
            }
        }
    }
    
}
