class Solution {
    Set<Character> set = new HashSet<>(Arrays.asList('a', 'e', 'i', 'o', 'u'));
    
    public String[] spellchecker(String[] wordlist, String[] queries) {
        int n = wordlist.length;
        int m = queries.length;
        System.out.println(n);
        System.out.println(m);
        
Set<String> wordSet = new HashSet<>();
        Map<String, String> caseSet = new HashMap<>();
        Map<String, String> vowelSet = new HashMap<>();
        
        for (String s: wordlist) {
            wordSet.add(s);
            if (!caseSet.containsKey(s.toLowerCase())) {
                caseSet.put(
s.toLowerCase(), s);
            }
            
            String tmp = toVowel(s);
            if (!vowelSet.containsKey(tmp)) {
                vowelSet.put(tmp, s);
            }
        }
        
        
        String[] result = new String[m];
        for (int i = 0; i < m; i++) {
            if (wordSet.contains(queries[i])) {
                result[i] = queries[i];
                continue;
            }
            
            if (caseSet.containsKey(queries[i].toLowerCase())) {
                result[i] = caseSet.get(
queries[i].toLowerCase());
                continue;
            }
            
            String tmp = toVowel(queries[i]);
            if (vowelSet.containsKey(tmp)) {
                result[i] = vowelSet.get(tmp);
                continue;
            }
            result[i] = "";
        }
        return result;
    }
    
  
    
    private String toVowel(String s) {
        s = s.toLowerCase();
        String result = "";
        for (int i = 0; i < s.length(); i++) {
            if (set.contains(s.charAt(i))) {
                result += "#";
            } else {
                result += s.charAt(i);
            }
        }
        return result;
    }
    
    private boolean matchVowel(String s, String t) {
        if (s.length() != t.length()) return false;
        s = s.toLowerCase();
        t = t.toLowerCase();
        for (int i = 0; i < s.length(); i++) {
            if (set.contains(s.charAt(i))) {
                if (set.contains(t.charAt(i))) {
                    continue;
                } else {
                    return false;
                }
            } else {
                if (s.charAt(i) != t.charAt(i)) {
                    return false;
                }
            }
            
        }
        return true;
    }
}
