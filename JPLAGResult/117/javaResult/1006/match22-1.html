class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        int wl = wordlist.length;
        int ql = queries.length;
        String[] ans = new String[ql];
        
HashSet<String> exact = new HashSet<String>();
        HashMap<String, String> cs = new HashMap<String, String>();
        HashMap<String, String> vw = new HashMap<String, String>();
        for (String w : wordlist) {
            exact.add(w);
            String wlo = w.toLowerCase();
            if (!cs.containsKey(wlo)) {
                cs.put(wlo, w);
            }
            wlo = wlo.replaceAll("e","a");
            wlo = wlo.replaceAll("i","a");
            wlo = wlo.replaceAll("o","a");
            wlo = 
wlo.replaceAll("u","a");
            if (!vw.containsKey(wlo)) {
                vw.put(wlo, w);
            }
        }
        
for (int i = 0; i < ql; i++) {
            String w = queries[i];
            if (exact.contains(w)) {
                ans[i] = w;
                continue;
            }
            w = 
w.toLowerCase();
            if (cs.containsKey(w)) {
                ans[i] = cs.get(w);
                continue;
            }
            w = w.replaceAll("e","a");
            w = w.replaceAll("i","a");
            w = w.replaceAll("o","a");
            w = 
w.replaceAll("u","a");
            if (vw.containsKey(w)) {
                ans[i] = vw.get(w);
                continue;
            }
            ans[i] = "";
        }
        return ans;
    }
}
