class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        
        HashSet<String> set = new HashSet<>();
        HashMap<String,String> map = new HashMap<>();
        HashMap<String,String> mapVow = new HashMap<>();
        
        int size= queries.length;
        
        for(int i = 0 ; i< wordlist.length ; i++){
            set.add(wordlist[i]);
            
            if(map.get(wordlist[i].toLowerCase()) == null){
                map.put(wordlist[i].toLowerCase(),wordlist[i]);
            }
            String v = mapVowel(wordlist[i]);
            if(mapVow.get(v) == null){
                mapVow.put(v,wordlist[i]);
            }
        }
        
        String[] res = new String[size];
        for(int i = 0 ; i< size ; i++){
            String w = queries[i];
            res[i] = "";
            
            if(set.contains(w)){
                res[i] = w;    
            }else if(map.get(w.toLowerCase()) != null){
                res[i] = map.get(w.toLowerCase());
            }else if(mapVow.get(mapVowel(w)) != null){
                res[i] = mapVow.get(mapVowel(w));
            }
        }
        
        return res; 
    }
    
    public String mapVowel(String s){
        char[] chars = s.toLowerCase().toCharArray();
        for(int i = 0 ; i < s.length() ; i++){
            if(chars[i] == 'a' || chars[i] == 'A' || chars[i] == 'o' || chars[i] == 'O' || chars[i] == 'e' || chars[i] == 'E' ||
               chars[i] == 'i' || chars[i] == 'I' || chars[i] == 'u' || chars[i] == 'U'){
                chars[i] = '$';
            }
        }
        
        return new String(chars);
    }
}
