class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        Set<String> set = new HashSet<>();
        Map<String, String> lowerMap = new HashMap<>();
        Map<String, String> vowMap = new HashMap<>();
        for (String word : wordlist) {
            set.add(word);
            
            String lower = word.toLowerCase();
            if (!lowerMap.containsKey(lower)) {
                lowerMap.put(lower, word);
            }
            
            String vow = removeVow(lower);
            if (!vowMap.containsKey(vow)) {
                vowMap.put(vow, word);
            }
        }
        
        int n = queries.length;
        String[] ans = new String[n];
        for (int i = 0; i < n; i++) {
            String word = queries[i];
            if (set.contains(word)) {
                ans[i] = word;
            } else {
                String lower = word.toLowerCase();
                if (lowerMap.containsKey(lower)) {
                    ans[i] = lowerMap.get(lower);
                } else {
                    String vow = removeVow(lower);
                    
if (vowMap.containsKey(vow)) {
                        ans[i] = vowMap.get(vow);
                    } else {
                        ans[i] = "";
                    }
                }
            
}
        }
        return ans;
    }
    
    private String removeVow(String word) {
        StringBuilder sb = new StringBuilder();
        for (char ch : word.toCharArray()) {
            if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {
                
sb.append('-');
            } else {
                sb.append(ch);
            }
        }
        return sb.toString();
    }
}
