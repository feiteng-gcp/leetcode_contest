class Solution {
    
    Set<String> st = new HashSet<>();
    
    public void dfs(String n,int i,String cur){
        if(i==n.length()){
            st.add(cur);
            return;
        }
        char c = n.charAt(i);
        if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u'){
            dfs(n,i+1,cur+'a');
            dfs(n,i+1,cur+'e');
            dfs(n,i+1,cur+'i');
            dfs(n,i+1,cur+'o');
            dfs(n,i+1,cur+'u');
        }else{
            dfs(n,i+1,cur+c);
        }

    }
    
    
    
    public String[] spellchecker(String[] ws, String[] qs) {
        Map<String,String> mp = new HashMap<>();
        HashMap<String,Integer> d = new HashMap<>();
       // int f = 0;
        for(int f = 0;f<ws.length;f++){
            String w = ws[f];
            if(!d.containsKey(w)){
                d.put(w,f);
            }
            String a = w.toLowerCase();
            if(mp.containsKey(a)){
                //mp.get(a).add(w);
            }else{
                mp.put(a,w);
               // mp.get(a).add(w);
            }
        }
        int i = 0;
        String l[] = new String[qs.length];
        for(String q:qs){
            if(d.containsKey(q)){
                l[i++] = q;
                continue;
            }
        
            String now = q.toLowerCase();
            if(mp.containsKey(now)){
                l[i++] = mp.get(now);
                continue;
            }
            st.clear();
            dfs(now,0,"");
            int cur = 99999999;
            for(String m:st){
                if(mp.containsKey(m)){
                    String chk = mp.get(m);
                    if(d.get(chk)<cur){
                        cur = d.get(chk);
                    }
                }
            }
            if(cur!=99999999){
                l[i++] = ws[cur];
            }else{
                l[i++] = "";
            }
            
            
            
        }
        return l;
        
        
    }
}
