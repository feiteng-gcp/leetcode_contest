class Solution {
    
    private static char[] vowels = "AEIOUaeiou".toCharArray();
    
    public String[] spellchecker(String[] wordlist, String[] queries) {
        
HashSet<String> exactWords = new HashSet();
        HashMap<String, String> lowerCaseWords = new HashMap();
        HashMap<String, String> faultToerantWords = new HashMap();
        for(String word : wordlist) {
            exactWords.add(word);
            String lowerCaseWord = getLowerCase(word);
            if (!lowerCaseWords.containsKey(lowerCaseWord)) {
                lowerCaseWords.put(lowerCaseWord, word);
            }
            String faultToerantWord = getFaultTolerantString(word);
            if (!faultToerantWords.containsKey(faultToerantWord)) {
                faultToerantWords.put(faultToerantWord, word);
            }
        }
        
        int n = queries.length;
        
        String[] ret = 
new String[n];
        for(int i=0;i<n;i++) {
            if (exactWords.contains(queries[i])) {
                ret[i] = queries[i];
                continue;
            }
            
            String lowerCaseWord = lowerCaseWords.get(
                getLowerCase(queries[i])
            );
            if (lowerCaseWord != null) {
                ret[i] = lowerCaseWord;
                continue;
            }
            
            String faultToerantWord = faultToerantWords.get(
                getFaultTolerantString(queries[i])
            );
            
            if (faultToerantWord != null) {
                ret[i] = faultToerantWord;
                
continue;
            }
            
            ret[i] = "";
        }
        
        return ret;
    }
    
    private String getLowerCase(String original) {
        char[] c = original.toCharArray();
        for(int i=0;i<c.length;i++) {
            c[i] = lowerCase(c[i]);
        }
        return new String(c);
    }
    
    private String getFaultTolerantString(String original) {
        char[] c = original.toCharArray();
        for(int i=0;i<c.length;i++) {
            if (isVowel(c[i])) {
                c[i] = 'a';
            } else {
                c[i] = lowerCase(c[i]);
            }
        }
        return new String(c);
    }
    
    private char lowerCase(char c) {
        if (c >= 'A' && c <= 'Z') {
            return (char) (c - 'A' + 'a');
        } else {
            return c;
        }
    }
    
    private boolean isVowel(char c) {
        for(char vowel : vowels) {
            if (c == vowel) {
                return true;
            }
        }
        return false;
    }
}
