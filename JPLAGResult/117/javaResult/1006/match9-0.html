	class Solution {
	    public String[] spellchecker(String[] wordlist, String[] queries) {
	    	String[] ret = new String[queries.length];
	    	int p = 0;
	    	Set<String> w = new HashSet<>();
	    	Map<String, Integer> cs = new HashMap<>();
	    	Map<String, Integer> vs = new HashMap<>();
	    	for(String wl : wordlist)w.add(wl);
	    	for(int i = 0;i < wordlist.length;i++) {
	    		String o = wordlist[i].toLowerCase();
	    		if(!cs.containsKey(o)) {
	    			cs.put(o, i);
	    		}
	    	}
	    	for(int i = 0;i < wordlist.length;i++) {
	    		String o = wordlist[i].toLowerCase();
	    		char[] u = o.toCharArray();
	    		for(int j = 0;j < u.length;j++) {
	    			if("aiueo".indexOf(u[j]) >= 0) {
	    				u[j] = '*';
	    			}
	    		}
	    		o = new String(u);
	    		if(!vs.containsKey(o)) {
	    			vs.put(o, i);
	    		}
	    	}
	        for(String q : queries) {
	        	if(w.contains(q)) {
	        		ret[p++] = q;
	        		continue;
	        	}
	    		String o = q.toLowerCase();
	        	if(cs.containsKey(o)) {
	        		ret[p++] = wordlist[cs.get(o)];
	        		continue;
	        	}
	    		char[] u = o.toCharArray();
	    		for(int j = 0;j < u.length;j++) {
	    			if("aiueo".indexOf(u[j]) >= 0) {
	    				u[j] = '*';
	    			}
	    		}
	    		o = new String(u);
	        	if(vs.containsKey(o)) {
	        		ret[p++] = wordlist[vs.get(o)];
	        		continue;
	        	}
	        	ret[p++] = "";
	        }
		    return ret;
	    }
	}
