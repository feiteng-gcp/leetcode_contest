    public String[] spellchecker(String[] wordlist, String[] queries) {
        if (wordlist == null || queries == null) {
            return new String[0];
        }
        String[] res = new String[queries.length];
        String[] insensitive = new String[wordlist.length];
        String[] vowelerrors = new String[wordlist.length];
        Set<String> set1 = new HashSet<> ();
        Set<String> set2 = new HashSet<> ();
        Set<String> set3 = new HashSet<> ();
        Set<Character> vowels = new HashSet<> (Arrays.asList(new Character[] {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'}));
        
        for (int i = 0; i < wordlist.length; i++) {
            set1.add(wordlist[i]);
            insensitive[i] = wordlist[i].toLowerCase();
            set2.add(insensitive[i]);
            StringBuilder sb = new StringBuilder (insensitive[i]);
            for (int j = 0; j < insensitive[i].length(); j++) {
                if (vowels.contains(insensitive[i].charAt(j))) {
                    sb.setCharAt(j, 'a');
                }
            }
            vowelerrors[i] = sb.toString();
            set3.add(vowelerrors[i]);
        }
        
        for (int i = 0; i < queries.length; i++) {
            if (set1.contains(queries[i])) {
                res[i] = queries[i];
            } else {
                String lowercase = queries[i].toLowerCase();
                if (set2.contains(lowercase)) {
                    for (int j = 0; j < insensitive.length; j++) {
                        if (lowercase.equals(insensitive[j])) {
                            res[i] = wordlist[j];
                            break;
                        }
                    }
                } else {
                    StringBuilder sb = new StringBuilder (lowercase);
                    for (int j = 0; j < lowercase.length(); j++) {
                        if (vowels.contains(lowercase.charAt(j))) {
                            sb.setCharAt(j, 'a');
                        }
                    }
                    String vowelerror = sb.toString();
                    if (set3.contains(vowelerror)) {
                        for (int j = 0; j < vowelerrors.length; j++) {
                            if (vowelerror.equals(vowelerrors[j])) {
                                res[i] = wordlist[j];
                                break;
                            }
                        }
                    } else {
                        res[i] = "";
                    }
                }
            }
        }
        
        return res;
    }
