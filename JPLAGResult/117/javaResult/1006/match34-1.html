class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        String[] answers = new String[queries.length];
        HashSet<String> set = new HashSet<>();
        HashMap<String,Integer> map = new HashMap<>();
        HashMap<String,Integer> vowelMap = new HashMap<>();
        for(int i=0;i<wordlist.length;i++){
            set.add(wordlist[i]);
            
String word = wordlist[i].toLowerCase();
            if(!map.containsKey(word)){
                map.put(word,i);
            }
            StringBuilder sb = new StringBuilder();
            for(char c:word.toCharArray()){
                if(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u'){
                    sb.append('*');
                }else{
                    sb.append(c);
                }
            }
            word = 
sb.toString();
            if(!vowelMap.containsKey(word)){
                // System.out.println(word);
                vowelMap.put(word,i);
            }
        } 
        for(int i=0;i<queries.length;i++){
            String query = queries[i].toLowerCase();
            
if(set.contains(queries[i])){
                answers[i] = queries[i];
            }else if(map.containsKey(query)){
                answers[i] = wordlist[map.get(query)];
            }else{
                StringBuilder sb = new StringBuilder();
                for(char c:query.toCharArray()){
                    if(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u'){
                        sb.append('*');
                    }else{
                        sb.append(c);
                    }
                }
                String word = sb.toString();
                // System.out.println(query + " " + word);
                if(vowelMap.containsKey(word)){
                    answers[i] = wordlist[vowelMap.get(word)];
                }else{
                    answers[i] = "";
                }
            }
        }
        return answers;
    }
}
