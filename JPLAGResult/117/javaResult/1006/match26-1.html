class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        Set<String> eSet = new HashSet<String>();
        Map<String, List<String>> cMap = new HashMap<String, List<String>>();
        Map<String, List<String>> vMap = new HashMap<String, List<String>>();
        for (String word : wordlist) {
            if (eSet.contains(word)) {
                continue;
            }
            
            eSet.add(word);
            
            String lowerWord = word.toLowerCase();
            if (!cMap.containsKey(lowerWord)) {
                cMap.put(lowerWord, new ArrayList<String>());
            }
            
cMap.get(lowerWord).add(word);
            
            StringBuilder sb = new StringBuilder();
            for (int i = 0; i < word.length(); i++) {
                char c = lowerWord.charAt(i);
                if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
                    sb.append("-");
                } else {
                    sb.append(c);
                }
            }
            
            String vWord = sb.toString();
            if (!vMap.containsKey(vWord)) {
                vMap.put(vWord, new ArrayList<String>());
            }
            vMap.get(vWord).add(word);
        }
        
        String[] result = new String[queries.length];
        
for (int i = 0; i < queries.length; i++) {
            if (eSet.contains(queries[i])) {
                result[i] = queries[i];
                continue;
            }
            
            String cq = queries[i].toLowerCase();
            if (cMap.containsKey(cq)) {
                result[i] = 
cMap.get(cq).get(0);
                continue;
            }
            
            StringBuilder sb = new StringBuilder();
            for (int j = 0; j < queries[i].length(); j++) {
                char c = cq.charAt(j);
                if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
                    sb.append("-");
                } else {
                    sb.append(c);
                }
            }
            String vq = sb.toString();
            if (vMap.containsKey(vq)) {
                result[i] = vMap.get(vq).get(0);
                
continue;
            }
            
            result[i] = "";
        }
        return result;
    }
}
