class Solution {
//     Set<Character> vowels = new HashSet<>(Arrays.asList('a','e','i','o','u'));
    
    public String[] spellchecker(String[] wordlist, String[] queries) {
        Set<String> set = new HashSet<>(); // exact the same.
        Map<String, String> map1 = new HashMap<>(); // all lower case, value is the first string that matches;
        Map<String, String> map2 = new HashMap<>(); // replace all vowels with 'a';
        
        for (String word : wordlist) {
            set.add(word);
            map1.putIfAbsent(word.toLowerCase(), word);
            map2.putIfAbsent(transform(word), word);
        }
        
        String[] output = new String[queries.length];
        for (int i = 0 ; i < queries.length ; i++ ) {
            String query = queries[i];
            
if (set.contains(query)) {
                output[i] = query;
            } else if (map1.containsKey(query.toLowerCase())) {
                output[i] = map1.get(query.toLowerCase());
            } else if (map2.containsKey(transform(query))) {
                output[i] = map2.get(transform(query));
            } else {
                output[i] = "";
            }
        }
        return output;
    }
    
    private String transform(String word) {
        char[] array = word.toLowerCase().toCharArray();
        char[] result = new char[array.length];
        for (int i = 0 ; i < array.length ; i++) {
            if (array[i] == 'a' || array[i] == 'e' || array[i] == 'i' || array[i] == 'o' || array[i] == 'u') {
                result[i] = 'a';
            } else {
                result[i] = array[i];
            }
        }
        return new String(result);
    }
}
