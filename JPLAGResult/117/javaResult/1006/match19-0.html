class Solution {
       public String toVin(String word) {
        StringBuilder sb = new StringBuilder(word);
        for (int j = 0; j < word.length(); j++) {
            char c = word.charAt(j);
            if (c == 'e' || c == 'i' || c == 'o' || c == 'u') {
                sb.setCharAt(j, 'a');
            }
        }
        return sb.toString();
    }

    public String[] spellchecker(String[] wordlist, String[] queries) {
        HashSet<String> set = new HashSet<>();
        Map<String, Integer> map = new HashMap<>();
        Map<String, Integer> vmap = new HashMap<>();
        for (int i = 0 ; i < wordlist.length; i++) {
            String w = wordlist[i];
            set.add(w);
            String win = w.toLowerCase();
            if (!map.containsKey(win)) {
                map.put(win, i);
            }
            String wv = toVin(win);
            if (!vmap.containsKey(wv)) {
                vmap.put(wv, i);
            }
        }
        String[] res = new String[queries.length];
        for (int i = 0 ; i < queries.length; i++) {
            String q = queries[i];
            if (set.contains(q)) {
                res[i] = q;
                continue;
            }
            String ql = q.toLowerCase();
            if (map.containsKey(ql)) {
                res[i] = wordlist[map.get(ql)];
                continue;
            }
            String vq = toVin(ql);
            if (vmap.containsKey(vq)) {
                res[i] = wordlist[vmap.get(vq)];
                continue;
            }
            res[i] = "";
        }
        return res;
    }
}
