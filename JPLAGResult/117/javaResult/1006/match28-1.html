class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        Set<String> exactMatch = new HashSet<>();
        for (String word : wordlist) exactMatch.add(word);

        Map<String, String> capitalMatch = new HashMap<>();
        for (String word : wordlist) {
            String lowerCase = word.toLowerCase();
            if (capitalMatch.containsKey(lowerCase)) continue;
            capitalMatch.put(lowerCase, word);
        
}

        Map<String, String> vowelMatch = new HashMap<>();
        for (String word : wordlist) {
            String wildCard = getWildCard(word);
            if (vowelMatch.containsKey(wildCard)) continue;
            vowelMatch.put(wildCard, word);
        
}

        String[] res = new String[queries.length];
        for (int i = 0; i < queries.length; i++) {
            String query = queries[i];
            if (exactMatch.contains(query)) {
                res[i] = query;
                continue;
            }

            String lowerCase = query.toLowerCase();
            if (capitalMatch.containsKey(lowerCase)) {
                res[i] = capitalMatch.get(lowerCase);
                continue;
            }

            String wildCard = getWildCard(query);
            if (vowelMatch.containsKey(wildCard)) {
                res[i] = vowelMatch.get(wildCard);
                continue;
            }

            res[i] = "";
        
}

        return res;
    }
    
    private String getWildCard(String str) {
        StringBuilder sb = new StringBuilder();
        for (char c : str.toLowerCase().toCharArray()) {
            
if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
                sb.append('?');
            } else {
                sb.append(c);
            }
        }
        return sb.toString();
    }
}
