class Solution {
    
public String[] spellchecker(String[] wordlist, String[] queries) {
         Map<String, Integer> map1 = new HashMap<>();
        Map<String, Integer> map2 = new HashMap<>();
        Map<String, Integer> map3 = new HashMap<>();
        
        for (int i = 0; i < wordlist.length; i++) {
            
map1.put(wordlist[i], i);
            String lowerCase = wordlist[i].toLowerCase();
            if (!map2.containsKey(lowerCase)) {
                map2.put(lowerCase, i);
            }
            String s = replace(wordlist[i]);
            if (!map3.containsKey(s)) {
                map3.put(s, i);
            }
        }
        String[] res = new String[queries.length];
        for (int i = 0; i < queries.length; i++) {
            if (map1.containsKey(queries[i])) {
                res[i] = wordlist[map1.get(queries[i])];
                
continue;
            }
            String lowerCase = queries[i].toLowerCase();
            if (map2.containsKey(lowerCase)) {
                res[i] = wordlist[map2.get(lowerCase)];
                continue;
            }
            String s = replace(queries[i]);
            if (map3.containsKey(s)) {
                res[i] = wordlist[map3.get(s)];
                continue;
            }
            res[i] = "";
        }
        return res;
    }


    String replace(String word) {
        char[] aeiou = new char[]{'a', 'e', 'i', 'o', 'u','A','E','I','O','U'};
        StringBuffer sb = new StringBuffer();
        for (int j = 0; j < word.length(); j++) {
            boolean match = false;
            for (int k = 0; k < 10; k++) {
                if (word.charAt(j) == aeiou[k]) {
                    match = true;
                    break;
                }
            }
            if (match) {
                sb.append('-');
            } else {
                sb.append(word.charAt(j));
            }
        }
        return sb.toString().toLowerCase();
    }
}
