class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        HashMap<String, String> originMap = new HashMap<>();
        HashMap<String, String> capMap = new HashMap<>();
        HashMap<String, String> vowelMap = new HashMap<>();
        for (int i=0; i<wordlist.length; i++) {
            String word = wordlist[i];
            originMap.put(word, word);
            
            String lowWord = word.toLowerCase();
            if (!capMap.containsKey(lowWord)) {
                capMap.put(lowWord, word);
            }
            
            String vowelWord = vowel(word);
            if (!vowelMap.containsKey(vowelWord)) {
                vowelMap.put(vowelWord, word);
            }
        }
        
        int n = queries.length;
        String[] res = new String[n];
        for (int i=0; i<n; i++) {
            String q = queries[i];
            if (originMap.containsKey(q)) {
                res[i] = originMap.get(q);
            } else if (capMap.containsKey(q.toLowerCase())) {
                res[i] = capMap.get(q.toLowerCase());
            } else if (vowelMap.containsKey(vowel(q))) {
                res[i] = vowelMap.get(vowel(q));
            } else {
                res[i] = "";
            }
        }
        return res;
    }
    
    private String vowel(String str) {
        char[] ca = str.toLowerCase().toCharArray();
        for (int i=0; i<ca.length; i++) {
            char c = ca[i];
            if (c=='a' || c=='e' || c=='i' || c=='o' || c=='u') {
                ca[i] = 'a';
            }
        }
        return new String(ca);
    } 
}
