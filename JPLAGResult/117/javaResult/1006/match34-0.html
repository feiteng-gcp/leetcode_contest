class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        String[] solutions = new String[queries.length];
        HashMap<String, String> perfect = new HashMap<String, String>();
        HashMap<String, String> capitalization = new HashMap<String, String>();
        HashMap<String, String> vowels = new HashMap<String, String>();
        for(int i = 0; i<wordlist.length; i++){
            String key = wordlist[i];
            String value = wordlist[i];
            if(!perfect.containsKey(key)){
                perfect.put(key, value);

            }
            String capital = key.toLowerCase();
            if(!capitalization.containsKey(capital)){
                capitalization.put(capital,value);
            }
            String vowel = capital.replaceAll("[aeiou]", "!");
           
            if(!vowels.containsKey(vowel)){
                vowels.put(vowel,value);
            }
            

            
        }
      
        for(int i = 0; i<queries.length; i++){
            String query = queries[i];
            String query2 = query.toLowerCase();
            String query3 = query2.replaceAll("[aeiou]","!");
            String solution = "";
            if(perfect.containsKey(query)){
                solution = query;
            }
            
            else if(capitalization.containsKey(query2)){
                solution = capitalization.get(query2);
            }
           
            else if(vowels.containsKey(query3)){
                solution = vowels.get(query3);

            }
            solutions[i] = solution;
            
            
        }
        return solutions;
    }
}
