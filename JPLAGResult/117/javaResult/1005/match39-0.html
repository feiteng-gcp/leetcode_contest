/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isUnivalTree(TreeNode root) {
        if (root == null ) {
            return true;
        }
        int val = root.val;
        return same(root.left, val) && same (root.right, val);
    }
    
    private boolean same(TreeNode node, int val) {
        if (node == null) {
            return true;
        }
        if (node.val != val ) {
            return false;
        }
        return same(node.left, val) && same(node.right,val);
        
    }
}
