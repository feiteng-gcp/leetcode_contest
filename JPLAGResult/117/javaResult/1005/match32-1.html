/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public boolean isUnivalTree(TreeNode root) {
        if (root == null) 
            return true;
        int value = root.val;
        return matches(root.left, value) && 
matches(root.right, value);
    }
    
    public boolean matches(TreeNode node, int value) {
        if (node == null) 
            return true;
        if (node.val != value)
            return false;
        return matches(node.left, value) && matches(node.right, value);
    }
}
