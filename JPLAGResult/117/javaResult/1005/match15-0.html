	class Solution {
	    public boolean isUnivalTree(TreeNode root) {
	        int my = root.val;
	        return ok(root, my);
	    }
	    
	    boolean ok(TreeNode cur, int v)
	    {
	    	if(cur == null)return true;
	    	if(cur.val != v)return false;
	    	return ok(cur.left, v) && ok(cur.right, v);
	    }
	}
