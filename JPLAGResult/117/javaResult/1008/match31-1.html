/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
   Map<String, Integer> dp;

    public int minCameraCover(TreeNode root) {
        dp = new HashMap<>();
        return getMin(dfs(root, 1), dfs(root, 2));
    }

    int getMin(int a, int b, int c) {
        return Math.min(a, Math.min(b, c));
    }

    int getMin(int a, int b) {
        return Math.min(a, b);
    
}

    private int dfs(TreeNode root, int use) {
        if (root == null) return 0;
        String key = root.hashCode() + "," + use;
        if (dp.containsKey(key)) {
            return dp.get(key);
        }
        // 0 -> root is not covered
        // 1 -> root is covered, but not install
        // 2 -> root is covered, and install
        if (use == 0) {
            int t = getMin(
                    dfs(root.left, 1),
                    dfs(root.left, 2));

            t += getMin(
                    dfs(root.right, 1),
                    dfs(root.right, 2));

            dp.put(key, t);
        } else if (use == 1) {
            if (root.left == null && root.right == null) {
                dp.put(key, 1);
            } else if (root.left == null) {
                int t = dfs(root.right, 2);
                dp.put(key, t);
            } else if (root.right == null) {
                int t = dfs(root.left, 2);
                dp.put(key, t);
            } else {
                int a = dfs(root.left, 1) + dfs(root.right, 2);
                int b = dfs(root.left, 2) + dfs(root.right, 1);
                int c = dfs(root.left, 2) + dfs(root.right, 2);
                dp.put(key, getMin(a, b, c));
            }
        } else {
            if (root.left == null && root.right == null) {
                dp.put(key, 1);
            } else if (root.left == null) {
                int t = getMin(
                        dfs(root.right, 0),
                        dfs(root.right, 1),
                        dfs(root.right, 2));
                dp.put(key, t + 1);
            } else if (root.right == null) {
                int t = getMin(
                        dfs(root.left, 0),
                        dfs(root.left, 1),
                        dfs(root.left, 2));
                dp.put(key, t + 1);
            } else {
                int l = getMin(
                        dfs(root.left, 0),
                        dfs(root.left, 1),
                        dfs(root.left, 2));

                int r = getMin(
                        dfs(root.right, 0),
                        dfs(root.right, 1),
                        dfs(root.right, 2));

                dp.put(key, l + r + 1);
            }
        }
        return dp.get(key);
    }
}
