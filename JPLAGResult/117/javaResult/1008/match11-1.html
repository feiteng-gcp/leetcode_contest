/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minCameraCover(TreeNode root) {
        int[] rs = new int[1];
        if (minCameraCoverEx(root,rs) == 1)
            rs[0]++;
        
return rs[0];
    }

    private int minCameraCoverEx(TreeNode root, int[] rs) {
        if (root == null)
            return 0;
        int r1 = minCameraCoverEx(root.left,rs);
        int r2 = minCameraCoverEx(root.right,rs);
        if (r1 == 0 && r2 == 0)
        {
            return  1;
        }
        else if (r1 == 1 || r2 == 1)
        {
            rs[0]++;
            return 2;
        }
        else if (r1 == 2 || r2 == 2)
        {
            return 3;
        }
        else if (r1 == 3 || r2 == 3)
        {
            return 1;
        }
        return 0;
    }
}
