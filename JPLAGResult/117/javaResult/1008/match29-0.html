/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Map<TreeNode, Integer> minMap = new HashMap();
    public int minCameraCover(TreeNode root) {
        if (root == null) return 0;
        if (minMap.containsKey(root)) return minMap.get(root);
        int ans = Integer.MAX_VALUE;
        // put at self then go to grand children
        int self = 1;
        int leftGrand = 0;
        if (root.left != null) {
            leftGrand += minCameraCover(root.left.left);
            leftGrand += minCameraCover(root.left.right);
        }
        self += leftGrand;
        int rightGrand = 0;
        if (root.right != null) {
            rightGrand += minCameraCover(root.right.left);
            rightGrand += minCameraCover(root.right.right);
        }
        self += rightGrand;
        ans = Math.min(ans, self);
        
        int both = 0;
        int allThree = 0;
        int leftGrandGrand = 0;
        int rightGrandGrand = 0;
        // put at left, check left grand grand child and right
        if (root.left != null) {
            int left = 1;
            if (root.left.left != null) {
                int thisAns1 = 0;
                thisAns1 += minCameraCover(root.left.left.left);
                thisAns1 += minCameraCover(root.left.left.right);
                thisAns1 = Math.min(thisAns1, minCameraCover(root.left.left));
                left += thisAns1;
                allThree += thisAns1;
                leftGrandGrand += thisAns1;
            }
            if (root.left.right != null) {
                int thisAns1 = 0;
                thisAns1 += minCameraCover(root.left.right.left);
                thisAns1 += minCameraCover(root.left.right.right);
                thisAns1 = Math.min(thisAns1, minCameraCover(root.left.right));
                left += thisAns1;
                allThree += thisAns1;
                leftGrandGrand += thisAns1;
            }
            both += left;
            left += minCameraCover(root.right);
            ans = Math.min(ans, left);
        }
        
        
        // opposite
        if (root.right != null) {
            int left = 1;
            if (root.right.left != null) {
                int thisAns1 = 0;
                thisAns1 += minCameraCover(root.right.left.left);
                thisAns1 += minCameraCover(root.right.left.right);
                thisAns1 = Math.min(thisAns1, minCameraCover(root.right.left));
                left += thisAns1;
                allThree += thisAns1;
                rightGrandGrand += thisAns1;
            }
            if (root.right.right != null) {
                int thisAns1 = 0;
                thisAns1 += minCameraCover(root.right.right.left);
                thisAns1 += minCameraCover(root.right.right.right);
                thisAns1 = Math.min(thisAns1, minCameraCover(root.right.right));
                left += thisAns1;
                allThree += thisAns1;
                rightGrandGrand += thisAns1;
            }
            both += left;
            left += minCameraCover(root.left);
            ans = Math.min(ans, left);
        }
        
        // put at left and right
        if (both != 0) {
            ans = Math.min(ans, both);
        }
        
        // put at root and left/right
        if (root.left != null) ans = Math.min(ans, 2 + rightGrand + leftGrandGrand);
        if (root.right != null) ans = Math.min(ans, 2 + leftGrand + rightGrandGrand);
        
        // put at three
        ans = Math.min(ans, 3 + allThree);
        
        
        minMap.put(root, ans);
        return ans;
        
    }
}
