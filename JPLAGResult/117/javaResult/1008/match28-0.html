/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class NodeInfo {
	boolean isCam;
	boolean isTouched;
	NodeInfo(boolean cam, boolean touched) {
		this.isCam = cam;
		this.isTouched = touched;
	}
}
class Solution {
    public int count;
	public int minCameraCover(TreeNode root) {
        count = 0;
        if (!helper(root).isTouched) count++;
        return count;
    }
	private NodeInfo helper(TreeNode root) {
		if (root == null) {
			return new NodeInfo(false, true);
		}
		NodeInfo left = helper(root.left);
		NodeInfo right = helper(root.right);
		boolean isCam = false;
		boolean isTouched = false;
		if (left.isCam || right.isCam) {
			isCam = false;
			isTouched = true;
		} else {
			isTouched = false;
		}
		if (!left.isTouched || !right.isTouched) {
			isCam = true;
            isTouched = true;
			count++;
		}
        //if(isCam) System.out.println(root.val);
		return new NodeInfo(isCam, isTouched);
	}
}
