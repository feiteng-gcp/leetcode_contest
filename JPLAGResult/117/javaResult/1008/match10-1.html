/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minCameraCover(TreeNode root) {
        int[] res = dfs(root);
        return Math.min(res[1], res[2]);
    }
    
    public int[] dfs(TreeNode cur) {
        int[] res = new int[3];
        Arrays.fill(res, 1001);
        res[1] = 0;
        
        if (cur == null) {
            return res;
        }
        
        int[] left = dfs(cur.left);
        int[] right = dfs(cur.right);
        
        res[0] = left[1] + right[1];
        res[1] = minOfThree(left[2] + right[2], left[2] + right[1], right[2] + left[1]);
        
res[2] = minOfThree(left[0], left[1], left[2]) + minOfThree(right[0], right[1], right[2]) + 1;
        return res;
    }
    
    private int minOfThree(int i1, int i2, int i3) {
        return Math.min(Math.min(i1, i2), i3);
    }
}
