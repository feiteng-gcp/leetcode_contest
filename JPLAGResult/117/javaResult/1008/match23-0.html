/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minCameraCover(TreeNode root) {
        Integer[] re = get(root);
        if(re[1] == null) return re[0];
        return Math.min(re[0], re[1]);
    }
    Integer[] get(TreeNode root) {
        if(root.left == null && root.right== null) return new Integer[]{1, null, 0};
        if(root.left ==null) {
            root.left = root.right;
            root.right = null;
        }
        if(root.right == null) {
            Integer[] re = get(root.left);
            Integer a = mymin(re[0], re[1], re[2]) + 1;
            int b = re[0];
            Integer c = null;
            if(re[1] != null) c = re[1];
            return new Integer[]{a, b, c};
        }
        else {
            Integer[] l = get(root.left), r = get(root.right);
            Integer a = 1 + mymin(l[0], l[1], l[2]) + mymin(r[0], r[1], r[2]);
            Integer b = Math.min(l[0] + (r[1] == null? r[0]: Math.min(r[0], r[1])), r[0] + (l[1] == null? l[0]: Math.min(l[0], l[1])));
            Integer c = null;
            if(l[1] != null && r[1] != null) c = l[1] + r[1];
            return new Integer[]{a, b,c};
        }
    }
    Integer mymin(Integer a, Integer b, Integer c) {
        Integer re = a;
        if(b != null && re > b) re = b;
        if(c != null && re > c) re = c;
        return re;
    }
}
