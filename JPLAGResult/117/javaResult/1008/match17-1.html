/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minCameraCover(TreeNode root) {
        int[] res = dfs(root);
        
return Math.min(res[0],res[1]);
    }
    int[] dfs(TreeNode root){
        if(root==null) return new int[]{0,0,0};
        if(root.left==null&&root.right==null){
            return new int[]{1,1,0};
        }
        int[] le = dfs(root.left), ri = dfs(root.right);
        if(root.left==null) return new int[]{1+Math.min(ri[2],Math.min(ri[0],ri[1])),ri[0],Math.min(ri[0],ri[1])};
        if(root.right==null) return new int[]{1+Math.min(le[2],Math.min(le[0],le[1])),le[0],Math.min(le[0],le[1])};
        int a=10000;
        for(int i=0;i<3;i++){
            for(int j=0;j<3;j++) a = Math.min(a,le[i]+ri[j]);
        }
        a += 1; // a legal one, self install camera
        int leftIn = le[0]+Math.min(ri[0],ri[1]), rightIn = ri[0]+Math.min(le[0],le[1]);
        int b = Math.min(leftIn,rightIn);// legal one, self dont install
        int c = Math.min(le[0],le[1])+Math.min(ri[0],ri[1]);// not necessaruly legal self dont install;
        return new int[]{a,b,c};
    }
}
