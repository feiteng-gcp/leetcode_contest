/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    
    public int minCameraCover(TreeNode root) {
        long[] value = minCamera(root);
        // System.out.println(String.format("%d, %d, %d", value[0],value[1],value[2]));
        return (int) Math.min(value[0],value[1]);
    }
    
    // camera, detect, there are 3 states: 
    // {true, true}, {false, true}, {false, false}
    private long[] minCamera(TreeNode root) {
        long[] nums = new long[3];
        if (root == null) {
            nums[0] = Integer.MAX_VALUE;
            return nums;
        }
        long[] left = minCamera(root.left);
        long[] right = minCamera(root.right);
        long allMin = Integer.MAX_VALUE;
        for (int i = 0 ; i < 3 ; i++) {
            allMin = Math.min(allMin, left[i]);
            allMin = Math.min(allMin, right[i]);
        }
        nums[0] = 1 + Math.min(left[0], Math.min(left[1], left[2])) + Math.min(right[0], Math.min(right[1], right[2]));
        nums[1] = Math.min(left[0] + Math.min(right[0], right[1]), right[0] + Math.min(left[0], left[1]));
        nums[2] = left[1] + right[1];
        return nums;
    }
}
