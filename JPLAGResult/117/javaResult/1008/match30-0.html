/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minCameraCover(TreeNode root) {
        Map<TreeNode,int[]> map=new HashMap<TreeNode,int[]>();
        return Math.min(findMin(root,true,true,map),findMin(root,false,false,map));
    }
    public int findMin(TreeNode curr,boolean currCovered,boolean currIsCamera,Map<TreeNode,int[]> map){
        if(curr==null){
            return 0;
        }
        if(map.containsKey(curr)){
            int[] res=map.get(curr);
            if(currCovered&&currIsCamera&&res[0]!=-1){
                return res[0];
            }
            if(currCovered&&!currIsCamera&&res[1]!=-1){
                return res[1];
            }
            else if(!currCovered&&!currIsCamera&&res[2]!=-1){
                return res[2];
            }
            
        }
        if(!map.containsKey(curr)){
            map.put(curr,new int[]{-1,-1,-1});
        }
        int res=0;
        if(currIsCamera){
            res++;
        }
        int extra=5000;
        if(currCovered){
            if(currIsCamera){
                int left=Math.min(findMin(curr.left,true,false,map),findMin(curr.left,true,true,map));
                int right=Math.min(findMin(curr.right,true,false,map),findMin(curr.right,true,true,map));
                extra=left+right;
                map.get(curr)[0]=extra+res;
            }
            else{
                int left=Math.min(findMin(curr.left,false,false,map),findMin(curr.left,true,true,map));
                int right=Math.min(findMin(curr.right,false,false,map),findMin(curr.right,true,true,map));
                extra=left+right;
                map.get(curr)[1]=extra+res;
            }
        }
        else{
            // doesn't exist
            /*
            if(currIsCamera){
                int left=Math.min(findMin(curr.left,true,false),findMin(curr.left,true,true));
                int right=Math.min(findMin(curr.right,true,false),findMin(curr.right,true,true));
                extra=left+right
            }
            else{
            */
            // cover by left child
            int left1=curr.left==null?5000:findMin(curr.left,true,true,map);
            int right1=Math.min(findMin(curr.right,false,false,map),findMin(curr.right,true,true,map));
            
            // cover by right child
            int left2=Math.min(findMin(curr.left,false,false,map),findMin(curr.left,true,true,map));
            int right2=curr.right==null?5000:findMin(curr.right,true,true,map);
                //int left=Math.min(findMin(curr.left,false,false),findMin(curr.left,false,true));
                //int right=Math.min(findMin(curr.right,false,false),findMin(curr.right,false,true));
                extra=Math.min(left1+right1,left2+right2);
            map.get(curr)[2]=extra+res;
            //}
        }
        return res+extra;
    }
}
