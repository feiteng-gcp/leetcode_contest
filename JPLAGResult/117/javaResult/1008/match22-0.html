/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minCameraCover(TreeNode root) {
        // A: with current node installed
        // B: with current node not installed(but children installed), covered
        // C: with current node not installed(and children not installed), not covered
        // int[] {A, B, C}
        int[] result = search(root);
        int ret = Math.min(result[0], result[1]);
        return ret == 0? 1 :ret;
    }
    
    public int[] search(TreeNode node) {
        if (node == null) {
            return new int[] {0, 0, 0};
        }
        if (node.left == null && node.right == null) {
            return new int[] {1, 1, 0};
        }
        int[] left = search(node.left);
        int[] right = search(node.right);
        int[] result = new int[3];
        result[0] = Math.min(left[0], Math.min(left[1], left[2])) + Math.min(right[0], Math.min(right[1], right[2])) + 1;
        if (node.left == null) {
            result[1] = right[0];
        } else if (node.right == null) {
            result[1] = left[0];
        } else {
            result[1] = Math.min(Math.min(left[0] + right[1], right[0] + left[1]), left[0] + right[0]);
        }
        result[2] = left[1] + right[1];
        //System.out.println(result[0] + "," + result[1] + "," + result[2]);
        return result;
    }
}
