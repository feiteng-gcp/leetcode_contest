/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int res = 0;
    public int minCameraCover(TreeNode root) {
        rr(root, false);
        return res;
    }
    
    void rr(TreeNode k, boolean hasParent) {
        if(k == null) return;
        if(k.left == null && k.right == null) {
            if(hasParent) {
                k.val = 1;
            } else {
                res++;
            }
            return;
        }
        rr(k.left, true);
        rr(k.right, true);
        boolean need = (k.left != null && k.left.val == 1) || (k.right != null && k.right.val == 1);
        if(need) {
            k.val = 2;
            res++;
        } else {
            boolean ch = (k.left != null && k.left.val == 2) || (k.right != null && k.right.val == 2);    
            if(!ch) {
                if(hasParent) {
                    k.val = 1;
                } else {
                    res++;
                }
                
            }
        }
    }
    
    
    
    
}
