/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minCameraCover(TreeNode root) {
        int[] ans = mh(root);
        return Math.min(ans[1], ans[2]);
    }
    
    public int[] mh(TreeNode ro) {
        if (ro==null) {
            int[] ans = new int[3];
            ans[1] = 1000000;
            return ans;
        }
        int[] l = mh(ro.left);
        int[] r = mh(ro.right);
        int[] ans = new int[3];
        ans[0] = l[2]+r[2];
        ans[1] = 1000000;
        ans[2] = 1000000;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                ans[1] = Math.min(ans[1], l[i]+r[j]+1);
            }
            if (i==0)
                continue;
            ans[2] = Math.min(ans[2], l[1]+r[i]);
            ans[2] = Math.min(ans[2], l[i]+r[1]);
        }
        //System.out.println(ans[0]+" "+ans[1]+" "+ans[2]);
        return ans;
    }
}
