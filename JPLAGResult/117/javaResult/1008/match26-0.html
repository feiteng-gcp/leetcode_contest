/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minCameraCover(TreeNode root) {
        
        int n = helper(root, 0);
        Integer[][][] cache = new Integer[n][2][2];
        
        return dp(root, 1, 0, cache);
        
    }
    
    private int helper(TreeNode root, int id) {
        // change val to id
        if (root == null) return id;
        root.val = id;
        ++id;
        id = helper(root.left, id);
        return helper(root.right, id);
    }
    
    private int dp(TreeNode root, int state1, int state2, Integer[][][] cache) {
        // return the number of cameras needed to monitor the tree
        // state1: 0: the root not needs to be covered, 1: the root has to be covered
        // state2: 0: no camera already on, 1: has camera on
        if (root == null) return 0;
        int id = root.val;
        
        if (cache[id][state1][state2] == null) {
            // 
            if (state1 == 0) {
                // the root not needs to be covered
                // not placing camera at the root
                int min = 0; 
                if (state2 == 0) min = dp(root.left, 1, 0, cache) + dp(root.right, 1, 0, cache);
                else min = dp(root.left, 0, 0, cache) + dp(root.right, 0, 0, cache);
                // placing camera at the root
                if (state2 == 0) {
                    int tmp = 1 + dp(root.left, 0, 0, cache) + dp(root.right, 0, 0, cache);
                    if (tmp < min) min = tmp;
                }
                cache[id][state1][state2] = min;
            }
            else {
                // the root has to be covered, state2 = 0
                // place camera at the root
                int min = 1 + dp(root.left, 0, 0, cache) + dp(root.right, 0, 0, cache);
                // not place at root, has to be placed at either child
                // place on left
                int tmp = 1 + dp(root.left, 0, 1, cache) + dp(root.right, 1, 0, cache);
                if (tmp < min) min = tmp;
                // place on right
                tmp = 1 + dp(root.left, 1, 0, cache) + dp(root.right, 0, 1, cache);
                if (tmp < min) min = tmp;
                cache[id][state1][state2] = min;
            }
            
        }
        
        return cache[id][state1][state2];
    }
}
