class Solution{
    Map<String, Integer> dp;
	int res;
	
    public int minCameraCover(TreeNode root) {
        dp = new HashMap<>();
        res = Integer.MAX_VALUE;
        res = Math.min(res, dfs(root, true, false));
        return res;
    
}
    
    private int dfs(TreeNode root, boolean last, boolean cur) {
    	if (root == null) {
    		return 0;
    	}
    	String key = root.toString() + " " + last + " " + cur;
    	if (dp.containsKey(key)){
    		return dp.get(key);
    	}else if (!last){
    		int res = dfs(root.left, true, true) + dfs(root.right, true, true) + 1;
    		dp.put(key, res);
    		return res;
    	}else {
    		int res = Integer.MAX_VALUE;
    		if (cur) {
    			res = dfs(root.left, true, false) + dfs(root.right, true, false);
    			res = Math.min(res, dfs(root.left, true, true) + dfs(root.right, true, true) + 1);
    		}else {
    			res = dfs(root.left, tr
ue, true) + dfs(root.right, true, true) + 1;
    			if(root.left != null) {
    				res = Math.min(res, dfs(root.left, false, false) + dfs(root.right, true, false));
    			}
    			if(root.right != null) {
    				res = Math.min(res, dfs(root.left, true, false) + dfs(root.right, false, false));
    			}
    			dp.put(key, res);
    		}
    		return res;
    	}
    }
}
