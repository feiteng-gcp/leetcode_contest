/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int count = 0;
    public int minCameraCover(TreeNode root) {
        if (root == null)
            return 0;
        else if (root.left == null && root.right == null)
            return 1;
        
        
        //check if root need one extra camera
        boolean[] rootStatus = travel(root);
        if (rootStatus[1] == false){
            count++;
        }
        return count;
    
}
    
    //boolean[0] -> true if this node has camera
    //boolean[1] -> true if this node is watched
    private boolean[] travel(TreeNode node){
        //if leaf node -> no camera
        //null -> same as being watched
        if (node == null)
            return new boolean[] {false, true};
        
        if (node.left == null && node.right == null){
            return new boolean[] {false, false};
        }
        
        
        boolean[] left = travel(node.left);
        boolean[] right = travel(node.right);
        //if left or right is not watched -> this have to put camera
        //this is also the only condition to have a camera
        if (left[1] == false || right[1] == false){
            count++;
            return new boolean[] {true, true};
        }
        
        //if left has or right has -> this has no camera, but watched
        else if (left[0] || right[0]){
            return new boolean[] {false, true};
        }
        
        //if left and right is watched -> this has no camera, and not watched
        else{
            return new boolean[] {false, false};
        }
        
    }
}
