/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    public class EnhancedNode {
        int val;
        EnhancedNode left, right;
        
        
        // minimal number of camera covering the subtree, and also must have a camera on this node
        int minCameraWithCameraOn;
        
        // minimal number of camera covering the subtree, may or may not have a camera installed on this node
        int minCamera;
        
        // minimal number of camera covering the subtree except that this node may or may not be covered at all
        int minCameraWithoutCoveringSelf;
        
        public EnhancedNode(int x) {
            val = x;
        }
        
    }
    
    
public int minCameraCover(TreeNode root) {
        if (root == null) {
            return 0;
        }
        EnhancedNode enhancedRoot = rebuildTree(root);
        backfillMinCamera(enhancedRoot);
        
return enhancedRoot.minCamera;
    }
    
    private EnhancedNode rebuildTree(TreeNode root) {
        if (root == null) {
            return null;
        }
        EnhancedNode ret = new EnhancedNode(root.val);
        ret.left = rebuildTree(root.left);
        ret.right = rebuildTree(root.right);
        return ret;
    }
    
    private int getMinCameraWithoutCoveringSelf(EnhancedNode root) {
        if (root == null) {
            return 0;
        }
        int minCameraLeft = (root.left == null ? 0 : root.left.minCamera);
        int minCameraRight = (root.right == null ? 0 : root.right.minCamera);
        return minCameraLeft + minCameraRight;
    }
    
    private void backfillMinCamera(EnhancedNode root) {
        if (root == null) {
            return;
        }
        backfillMinCamera(root.left);
        backfillMinCamera(root.right);
        
        int minCameraLeft = (root.left == null ? 0 : root.left.minCamera);
        int minCameraRight = (root.right == null ? 0 : root.right.minCamera);
        int minCameraLeftNoCover = (root.left == null ? 0 : root.left.minCameraWithoutCoveringSelf);
        int minCameraRightNoCover = (root.right == null ? 0 : root.right.minCameraWithoutCoveringSelf);
        
        root.minCameraWithoutCoveringSelf = minCameraLeft + minCameraRight;
        root.minCameraWithCameraOn = 1 + minCameraLeftNoCover + minCameraRightNoCover;
        
        root.minCamera = root.minCameraWithCameraOn;
        
        if (root.left != null) {
            int tmp = root.left.minCameraWithCameraOn + minCameraRight;
            if (root.minCamera > tmp) {
                root.minCamera = tmp;
            }
        }
        
        if (root.right != null) {
            int tmp = root.right.minCameraWithCameraOn + minCameraLeft;
            if (root.minCamera > tmp) {
                root.minCamera = tmp;
            }
        }
        
        if (root.minCameraWithoutCoveringSelf > root.minCamera) {
            root.minCameraWithoutCoveringSelf = root.minCamera;
        }
        
    }
}
