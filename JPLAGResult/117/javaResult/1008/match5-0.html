/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int minCameraCover(TreeNode root) {
        if (root == null) {
            return 0;
        }
        int[] ans = dfs(root);
        int a = ans[0];
        if (ans[2] != -1) {
            a = Math.min(a, ans[2]);
        }
        return a;
    }
    
    private int[] dfs(TreeNode root) {
        if (root == null) {
            return new int[]{-1, 0, 0};
        }
        if (root.left == null && root.right == null) {
            return new int[]{1, 0, -1};
        }
        int[] left = dfs(root.left);
        int[] right = dfs(root.right);
        int ii = Integer.MAX_VALUE;
        int no0 = Integer.MAX_VALUE;
        int no1 = Integer.MAX_VALUE;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                int a = left[i];
                int b = right[j];
                if (a == -1 || b == -1 || a == Integer.MAX_VALUE || b == Integer.MAX_VALUE) {
                    continue;
                }
                ii = Math.min(ii, a + b + 1);
                if ((i == 0 || j == 0) && (i != 1 && j != 1)) {
                    no1 = Math.min(no1, a + b);
                }
                if (i == 2 && j == 2) {
                    no0 = Math.min(no0, a + b);
                }
            }
        }
        System.out.println(Arrays.toString(new int[]{ii, no0, no1}));
        return new int[]{ii, no0, no1};
        
    }
}
