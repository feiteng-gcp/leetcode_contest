/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Map<TreeNode, Integer> memo = new HashMap<>(); 
   
    
    public int minCameraCover(TreeNode root) {
        if (root == null) return 0;
        if (memo.containsKey(root)) return memo.get(root);
        int ans = 1;
        TreeNode n;
        if (root.left != null) {
            n = root.left;
            ans += Math.min(minCameraCover(n), minCameraCover(n.left) +  minCameraCover(n.right));   
        }
        if (root.right != null) {
            n = root.right;
            ans += Math.min(minCameraCover(n), minCameraCover(n.left) +  minCameraCover(n.right));           
        }
        
        if (root.left != null) {
            int ans2 = 1;
            ans2 += minCameraCover(root.right);
            if (root.left.left != null) {
                n = root.left.left;
                ans2 += Math.min(minCameraCover(n), minCameraCover(n.left) +  minCameraCover(n.right));   
            }
            if (root.left.right != null) {
                n = root.left.right;
                ans2 += Math.min(minCameraCover(n), minCameraCover(n.left) +  minCameraCover(n.right));
            }
            ans = Math.min(ans, ans2);
        }
        
        if (root.right != null) {
            int ans2 = 1;
            ans2 += minCameraCover(root.left);
            if (root.right.left != null) {
                n = root.right.left;
                ans2 += Math.min(minCameraCover(n), minCameraCover(n.left) +  minCameraCover(n.right));   
            }
            if (root.right.right != null) {
                n = root.right.right;
                ans2 += Math.min(minCameraCover(n), minCameraCover(n.left) +  minCameraCover(n.right));
            }
            ans = Math.min(ans, ans2);
        }
        
        memo.put(root, ans);
        return ans;
    }
}
