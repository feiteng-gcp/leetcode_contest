/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    
    Map<String, Integer> cache = new HashMap<>();
    public int minCameraCover(TreeNode root) {
        if (root == null) return 0;
        
        return helper(root, false, true);
    }
    
    private int helper(TreeNode root, boolean parent, boolean curr) {
        String key = "" + root + parent + curr;
        if (cache.containsKey(key)) {
            return cache.get(key);
        }
        if (root == null) {
            return 0;
        }
        
        if (root.left == null && root.right == null) {
            if (curr) {
                return 1;
            } else {
                return 0;
            }
        }
        
        
        int min = Integer.MAX_VALUE;
    // if (curr) {
            if (parent) { // must place camera at curr
                min = Math.min(min, helper(root.left, false, false) + helper(root.right, false, false) + 1);
            } else {
                // if place camera at curr
                min = Math.min(min, helper(root.left, false, false) + helper(root.right, false, false) + 1);

                // if not
                if (!curr) {
                    if (root.left != null) {
                        min = 
Math.min(min, helper(root.left, false, true) + helper(root.right, false, true));
                    }

                    if (root.right != null) {
                        min = Math.min(min, helper(root.left, false, true) + helper(root.right, false, true));
                    }
                } else {
                    if (root.left != null) {
                        min = 
Math.min(min, helper(root.left, true, true) + helper(root.right, false, true));
                    }

                    if (root.right != null) {
                        min = Math.min(min, helper(root.left, false, true) + helper(root.right, true, true));
                    }
                }
                
            }
        // } else {
        //     min = Math.min(min, helper(root.left, false) + helper(root.right, false) + 1);
        // }
        
         // System.out.println("" + root.val + parent + curr + min);
        cache.put(key, min);
        return min;
    }
}
