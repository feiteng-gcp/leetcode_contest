/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    int ans = 0;

    int dfs(TreeNode node) {
        if (null == node) {
            return 1;
        }
        if (null == node.left && null == node.right) {
            return 0;
        }
        int L = dfs(node.left);
        int R = dfs(node.right);
        if (L == 1 && R == 1
                && (null != node.left && node.left.val == 1 || null != node.right && node.right.val == 1)) {
            return 1;
        }

        if (L == 0 || R == 0) {
            ans++;
            node.val = 1;
            return 1;
        }
        return 0;
    }

    public int minCameraCover(TreeNode root) {
        if (null == root) {
            return 0;
        }
        if (null == root.left && null == root.right) {
            return 1;
        }
        dfs(root);
        boolean ok = true;
        if (root.val == 1) {
            ok = false;
        } else {
            if (null != root.left && root.left.val == 1) {
                ok = false;
            }
            if (null != root.right && root.right.val == 1) {
                ok = false;
            }
        }
        if (ok) ans++;
        return ans;
    }
}
