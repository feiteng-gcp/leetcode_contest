	class Solution {
		int ans = 0;
		
	    public int minCameraCover(TreeNode root) {
	    	ans = 0;
	    	dfs(root, true);
	        return ans;
	    }
	    
	    int dfs(TreeNode cur, boolean isr)
	    {
	    	if(cur == null)return -99;
	    	
	    	int l = dfs(cur.left, false);
	    	int r = dfs(cur.right, false);
	    	if(l == -99 && r == -99) {
	    		if(isr)ans++;
	    		return 1;
	    	}
	    	if((l == 0 || l == -99) && (r == 0 || r == -99) && isr) {
	    		ans++;
	    	}
	    	if(Math.max(l, r) >= 1) {
	    		ans++;
	    		return -1;
	    	}
	    	if(l == -1 || r == -1)return 0;
	    	return Math.max(l, r) + 1;
	    }
	}
