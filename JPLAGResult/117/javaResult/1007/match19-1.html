class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        List<Integer> result = 
new LinkedList<>();
        for (int i = 1; i < 10; i++) {
            helper(N, K, 1, i, result);
        }
        if (N == 1) {
            result.add(0);
        }
        int[] arr = new int[result.size()];
        int index = 0;
        for (Integer i: result) {
            arr[index++] = i;
        
}
        return arr;
    }
    
    private void helper(int N, int K, int x, int val, List<Integer> result) {
        if (x == N) {
            result.add(val);
            return;
        }
        
        int prev = val % 10;
        
if (prev + K <= 9) {
            helper(N, K, x + 1, val * 10 + prev + K, result);
        }
        
        if (prev - K >= 0 && K != 0) {
            helper(N, K, x + 1, val * 10 + prev - K, result);
        }
    }
}
