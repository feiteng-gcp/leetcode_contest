class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        List<int[]> list = new ArrayList<>();
        dfs(list, new int[N], 0, K);
        int[] ans = new int[list.size()];
        for (int i = 0; i < list.size(); i++) {
            StringBuilder sb = new StringBuilder();
            for (int n : list.get(i)) {
                sb.append(n);
            }
            ans[i] = Integer.parseInt(sb.toString());
        }
        return ans;
    }
    
    private void dfs(List<int[]> list, int[] cur, int pos, int K) {
        if (pos == cur.length) {
            
            list.add(cur.clone());
            return;
        }
        if (pos == 0) {
            for (int i = 1; i <= 9; i++) {
                cur[pos] = i;
                dfs(list, cur, pos + 1, K);
            }
            if (cur.length == 1) {
                cur[pos] = 0;
                dfs(list, cur, pos + 1, K);
            }
        } else {
            if (cur[pos - 1] + K <= 9) {
                cur[pos] = cur[pos - 1] + K;
                dfs(list, cur, pos + 1, K);
            }
            if (cur[pos - 1] - K >= 0 && cur[pos - 1] - K != cur[pos - 1] + K) {
                cur[pos] = cur[pos - 1] - K;
                dfs(list, cur, pos + 1, K);
            }
        }
    }
}
