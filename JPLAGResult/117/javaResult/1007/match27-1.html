class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        List<Integer> list = new ArrayList<>();
        search(list, N, K, 0, 0);
        if (N == 1) {
            list.add(0);
        }
        int[] result = new int[list.size()];
        for (int i = 0; i < result.length; i++) 
{
            result[i] = list.get(i);
        }
        return result;
    }
    
    public void search(List<Integer> list, int N, int K, int count, int current) {
        if (count == 0) {
            for (int i = 1; i <= 9; i++) {
                current = i;
                search(list, N, K, 1, current);  
            }
        } else {
            if (count == N) {
                list.add(current);
                return;
            }
            if (current % 10 >= K) {
                search(list, N, K, count + 1, current * 10 + (current % 10 - K));
            }
            
if (K != 0 && current % 10 + K <= 9) {
                search(list, N, K, count + 1, current * 10 + (current % 10 + K));
            }
        }
    }
}
