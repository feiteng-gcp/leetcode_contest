class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        if (N == 0) {
            return new int[0];
        }
        List<Integer> list = new ArrayList();
        int start = 1;
        if (N == 1) {
            start = 0;
        }
        
for (int i = start; i <= 9; i++) {
            dfs(list, i, i, N - 1, K);
        }
        int[] res = new int[list.size()];
        for (int i = 0 ; i < list.size(); i++) {
            res[i] = list.get(i);
        }
        return res;
    }
    
    public void dfs(List<Integer> list, int curInt, int lastDigit, int leftN, int K) {
        if (leftN == 0) {
            list.add(curInt);
            return;
        }
        
        if (lastDigit - K >= 0) {
            int cur = lastDigit - K;
            dfs(list, curInt * 10 + cur, cur, leftN - 1, K);
        }
        
        if (K != 0 && lastDigit + K < 10) {
            int cur = lastDigit + K;
            dfs(list, curInt * 10 + cur, cur, leftN - 1, K);
        }
    }
    
}
