class Solution {
    
    public int[] numsSameConsecDiff(int N, int K) {
        List<String> res = new ArrayList<>();
        
        numsSame(N, K, new StringBuilder(), res);
        
        if (N == 1) {
            res.add("0");
        }
    
        int[] r = new int[res.size()];
        for (int i = 0; i < res.size(); i++) {
            r[i] = Integer.valueOf(res.get(i));
        }
        
        return r;
    }
    
    private void numsSame(int len, int K, StringBuilder cur, List<String> res) {
        if (cur.length() == len) {
            res.add(cur.toString());
            return;
        }
        
        if (cur.length() == 0) {
            // try from 1 to 9
            for (int i = 1; i <= 9; i++) {
                cur.append(i);
                numsSame(len, K, cur, res);
                cur.deleteCharAt(cur.length() - 1);
            }
            return;
        }
        
        int num = Character.getNumericValue(cur.charAt(cur.length() - 1));
        
        int minus = num - K;
        int plus = num + K;
        
        if (minus >= 0) {
            cur.append(minus);
            numsSame(len, K, cur, res);
            cur.deleteCharAt(cur.length() - 1);
        }
        
        if (K != 0 && plus <= 9) {
            cur.append(plus);
            numsSame(len, K, cur, res);
            cur.deleteCharAt(cur.length() - 1);
        }
    }
    
}
