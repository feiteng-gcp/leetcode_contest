	class Solution {
    	List<Integer> ret = new ArrayList<>();
    	
	    public int[] numsSameConsecDiff(int N, int K) {
	    	for(int i = 1;i <= 9;i++) {
	    		dfs(i, N-1, K);
	    	}
	    	if(N == 1) {
	    		ret.add(0);
	    	}
	    	
	    	int[] re = new int[ret.size()];
	    	for(int i = 0;i < ret.size();i++) {
	    		re[i] = ret.get(i);
	    	}
	    	return re;
	    }
	    
	    public void dfs(int cur, int n, int K)
	    {
	    	if(n == 0) {
	    		ret.add(cur);
	    		return;
	    	}
	    	int L = cur % 10;
	    	{
	    		int t = L-K;
	    		if(t >= 0) {
	    			dfs(cur*10 + t, n-1, K);
	    		}
	    	}
	    	if(K > 0){
	    		int t = L+K;
	    		if(t <= 9) {
	    			dfs(cur*10 + t, n-1, K);
	    		}
	    	}
	    }
	}
