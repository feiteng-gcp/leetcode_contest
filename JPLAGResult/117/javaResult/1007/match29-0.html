class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        List<Integer> temp = new ArrayList();
        List<Integer> res = new ArrayList();
        for (int i=(N==1?0:1); i<=9; i++){
            temp.add(i);
            process(temp, N-1, K, res);
            temp.remove(temp.size()-1);
        }
        
        int[] res2 = new int[res.size()];
        for (int i=0; i<res2.length; i++){
            res2[i] = res.get(i);
        }
        return res2;
    }
    
    private void process(List<Integer> cur, int left, int K, List<Integer> res){
        if (left==0){
            int val = 0;
            for (int i=0; i<cur.size(); i++){
                val = val*10 + cur.get(i);
            }
            res.add(val);            
        }else{
            int prev = cur.get(cur.size()-1);
            if (prev + K <= 9){
                cur.add(prev+K);
                process(cur, left-1, K, res);
                cur.remove(cur.size()-1);
            }
            if (K!=0 && prev-K >= 0){             
                cur.add(prev-K);
                process(cur, left-1, K, res);
                cur.remove(cur.size()-1);
            }            
        }
    }
}
