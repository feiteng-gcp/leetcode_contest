class Solution {
    public int[] numsSameConsecDiff(int N, int K) {

        if (N == 1) {
            int[] ans = new int[10];
            for (int i = 0; i <= 9; i++) {
                ans[i] = i;
            }
            return ans;
        }

        ArrayList<Integer> list = new ArrayList<>();

        for (int i = 1; i <= 9; i++) {
            makeNumber(N - 1, i, K, list);
        }

        HashSet<Integer> set = new HashSet<>(list);

        int ans[] = new int[set.size()];
        int i = 0;
        for (int num : set) {
            ans[i] = num;
            i++;
        }

        return ans;

    }

    private void makeNumber(int len, int num, int k, ArrayList<Integer> list) {

        if (len == 0) {
            list.add(num);
            return;
        }

        int x = num % 10;
        int y = x + k;
        if (y >= 0 && y <= 9) {
            makeNumber(len - 1, num * 10 + y, k, list);
        }
        y = x - k;
        if (y >= 0 && y <= 9) {
            makeNumber(len - 1, num * 10 + y, k, list);
        }

    }
}
