class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        List<Integer> res = new ArrayList<>();
        for (int i = 1; i <= 9; i++)
        helper(res, N-1, i, K);
        if (N == 1) res.add(0);
        int len = res.size();
        int[] result = new int[len];
        for (int i = 0; i < len; i++) result[i] = res.get(i);
        return result;
    }
    
    public void helper(
List<Integer> res, int N, int curr, int k) {
        if (N == 0) {
            res.add(curr);
            return;
        }
        int temp = curr%10;
        
        
if (temp >= k) helper(res, N-1, curr*10 + temp - k, k);
        if (temp + k < 10 && k != 0) helper(res, N-1, curr*10 + temp + k, k);
    }
}
