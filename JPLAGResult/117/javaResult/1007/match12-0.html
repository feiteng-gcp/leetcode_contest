class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        Set<Integer> result = new HashSet<>();
        for(int i = 1;i<=9;i++)
            helper(N,K,result,i,1);
        if(N==1)result.add(0);
        int[] ans = new int[result.size()];
        int i = 0;
        for(Integer num:result)
            ans[i++] = (int)num;
            
        return ans;
    }
    private void helper(int N, int K, Set<Integer> result, int currInt,int digit){
        //System.out.println(currInt);
        if(digit==N){
            result.add(currInt);
            return;
        }
        if((currInt%10+K)<10)
            helper(N,K,result,currInt*10+(currInt%10+K),digit+1);
        if((currInt%10-K)>=0)
            helper(N,K,result,currInt*10+(currInt%10-K),digit+1);
        
    }
}
