class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        List<Integer> ans = new ArrayList<>();
        if (N == 1) {
            int[] res = new int[10];
            for (int i = 0; i< 10; i++) {
                res[i] = i;
            }
            return res;
        }
        
for (int i = 1; i <= 9; i++) {
            dfs(i, i, N - 1, K, ans);
        }
        int[] res = new int[ans.size()];
        for (int i = 0; i < ans.size(); i++) 
{
            res[i] = ans.get(i);
        }
        return res;
    }

    private void dfs(
int val, int d, int N, int K, List<Integer> ans) {
        if (N == 0) {
            ans.add(val);
            return;
        }
        for (int i = 0; i <= 9; i++) {
            if (Math.abs(d - i) == K) {
                dfs(val * 10 + i, i, N - 1, K, ans);
            }
        }
    }
}
