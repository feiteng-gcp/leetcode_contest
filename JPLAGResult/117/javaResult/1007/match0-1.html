class Solution {
    
    
    Set<Integer> l = new HashSet<>();
    public int[] numsSameConsecDiff(int N, int K) {
        if(N==1){
            int a[] = new int[10];
            for(int i=0;i<=9;i++){
                a[i] = i;
            }
            return a;
        }
        
        for(int i=1;i<=9;i++){
            dfs(1,K,N,i);
        }
        int i=0;
        int a[] = new int[l.size()];
        for(int v:l){
            a[i++] = v;
        }
        return a;
        
    }
    
    public void dfs(int cur,int K,int N,int num){
        if(cur==N){
            l.add(num);
            return;
        }
        int cc = num%10;
        if(cc+K<=9){
            dfs(cur+1,K,N,num*10+(cc+K));
        }
        if(cc-K>=0){
            dfs(cur+1,K,N,num*10+(cc-K));
        }
        
        
    }
    
}
