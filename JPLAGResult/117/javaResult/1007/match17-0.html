class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        List<Integer> result = new ArrayList<>();
        helper(N, K, new StringBuilder(), result, true, 0);
        int[] ret = new int[result.size()];
        for (int i = 0; i < ret.length; i++) {
            ret[i] = result.get(i);
        }
        return ret;
    }
    
    private void helper(int N, int K, StringBuilder sb, List<Integer> result, boolean isBegin, int start) {
        if (sb.length() == N) {
            result.add(Integer.parseInt(sb.toString()));
            return;
        }
        
        if (start == N) {
            return;
        }
        
        if (isBegin) {
            for (int i = N == 1 ? 0 : 1; i <= 9; i++) {
                sb.append(i);
                helper(N, K, sb, result, false, start + 1);
                sb.setLength(sb.length() - 1);
            }
        } else {
            int lastDight = sb.charAt(sb.length() - 1) - '0';
            if (lastDight - K >= 0) {
                sb.append(lastDight - K);
                helper(N, K, sb, result, isBegin, start + 1);
                sb.setLength(sb.length() - 1);
            }
            
            if (K != 0 && lastDight + K <= 9) {
                sb.append(lastDight + K);
                helper(N, K, sb, result, isBegin, start + 1);
                sb.setLength(sb.length() - 1);
            }
        }
        
    }
}
