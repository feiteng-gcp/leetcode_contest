class Solution {
    private int N, K;
    private List<Integer> ans;
    
    public int[] numsSameConsecDiff(int N, int K) {
        this.N = N;
        this.K = K;
        this.ans = new ArrayList<>();
        
for (int i = 1; i <= 9; i++) {
            dfs(i, i, 1);
        }
        if (N == 1) {
            ans.add(0);
        }
        int[] ret = new int[ans.size()];
        for (int i = 0; i < ans.size(); i++) {
            ret[i] = ans.get(i);
        }
        return ret;
    }
    
    private void dfs(int cur, int val, int d) {
        if (d == N) {
            ans.add(val);
        } else {
            if (cur + K <= 9) {
                dfs(cur + K, val * 10 + cur + K, d + 1);
            }
            
if (cur - K >= 0 && K != 0) {
                dfs(cur - K, val * 10 + cur - K, d + 1);
            }
        }
    }
}
