class Solution {
	Set<String> tmp;
	
    public int[] numsSameConsecDiff(int N, int K) {
        tmp = new HashSet<>();
        for (char i = (N == 1 ? '0' : '1'); i <= '9'; ++i) {
        	StringBuilder tmp = new StringBuilder();
        	tmp.append(i);
        	DFS(N - 1, K, tmp);
        }
        int[] res = new int[tmp.size()]; int p = 0;
        
for (String i : tmp) {
        	res[p++] = Integer.valueOf(i);
        }
        return res;
    }
    
    private void DFS(int N, int k, StringBuilder last) {
    	if (N == 0) {
    		tmp.add(last.toStr
ing());
    	}else {
    		int n = last.charAt(last.length() - 1) - '0';
    		if (n + k < 10) {
    			last.append(n + k);
    			DFS(N - 1, k, last);
    			last.deleteCharAt(last.length()-1);
    		}
    		if (n - k >= 0) {
    			last.append(n - k);
    			DFS(N - 1, k, last);
    			last.deleteCharAt(last.length()-1);
    		}
    	}
    }
}
