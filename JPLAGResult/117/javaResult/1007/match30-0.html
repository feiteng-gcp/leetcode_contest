class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        Map<Integer, List<Integer>> map = new HashMap<Integer, List<Integer>>();
        for (int i = 0; i <= 9; i++) {
        	List<Integer> list = new ArrayList<Integer>();
        	if (i - K >= 0) {
        		list.add(i - K);
        	}
        	if (K != 0 && i + K <= 9) {
        		list.add(i + K);
        	}
        	map.put(i, list);
        }
        List<Integer> res = new ArrayList<Integer>();
        helper(0, 0, 0, N, map, res);
        int[] resArr = new int[res.size()];
        for (int i = 0; i < res.size(); i++) {
        	// System.out.println(res.get(i));
        	resArr[i] = res.get(i);
        }
        return resArr;
    }
	private void helper(int cur, int lastDigit, int numOfDigits, int N, Map<Integer, List<Integer>> map, List<Integer> res) {
		if (numOfDigits == N) {
			res.add(cur);
			return;
		}
		List<Integer> avai = new ArrayList<Integer>();
		if (numOfDigits != 0) {
			avai = map.get(lastDigit);
		} else {
            if (N == 1) {
				avai.add(0);
			}
			for (int i = 1; i <= 9; i++) {
				avai.add(i);
			}
		}
		for (int d: avai) {
			helper(cur*10 + d, d, numOfDigits+1, N, map, res);
		}
	}
}
