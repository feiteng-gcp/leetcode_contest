class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        int[] tracker=new int[N];
        List<Integer> res=new ArrayList<Integer>();
        
        findNum(tracker,0,res,K,N);
        int[] array=new int[res.size()];
        for(int i=0;i<array.length;i++){
            array[i]=res.get(i);
        }
        return array;
    }
    public void findNum(int[] tracker,int index,List<Integer> res,int K,int N){
        if(index==tracker.length){
            res.add(makeNum(tracker));
            return;
        }
        if(index==0){
            int low=N==1?0:1;
            for(int i=low;i<=9;i++){
                tracker[index]=i;
                findNum(tracker,index+1,res,K,N);
            }
        }
        else{
            int up=tracker[index-1]+K;
            int down=tracker[index-1]-K;
            if(up<10){
                tracker[index]=up;
                findNum(tracker,index+1,res,K,N);
            }
            if(K>0&&down>=0){
                tracker[index]=down;
                findNum(tracker,index+1,res,K,N);
            }
        }
    }
    public int makeNum(int[] tracker){
        int num=0;
        for(int i=0;i<tracker.length;i++){
            num=num*10+tracker[i];
        }
        return num;
    }
}
