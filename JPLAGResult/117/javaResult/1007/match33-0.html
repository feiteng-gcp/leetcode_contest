class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        List<Integer> list = new ArrayList<Integer>();
        
        if(N == 1) {
            list.add(0);
        }
        
        for(int i = 1; i < 10; i++) {
            helper(i, 1, N, K, list);
        }
        
        int[] res = new int[list.size()];
        
        for(int i = 0; i < list.size(); i++) {
            res[i] = list.get(i);
        }
        
        return res;
    }
    
    private void helper(int number, int index, int N, int K, List<Integer> list) {
        if(index == N) {
            list.add(number);
            return;
        }
        
        int digit = number%10;
        if(digit + K < 10) {
            helper(number*10 + (digit + K), index + 1, N, K, list);
        }
        
        if(digit - K >= 0 && (digit - K) != (digit + K)) {
            helper(number*10 + (digit - K), index + 1, N, K, list);
        }
    }
}
