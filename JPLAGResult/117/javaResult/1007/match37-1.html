class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        List<Integer> res = new ArrayList<>();
        backtrack(N, K, -1, 0, new StringBuilder(), res);
        int[] ret = new int[res.size()];
        for (int i = 0; i < res.size(); i++) 
{
            ret[i] = res.get(i);
        }
        return ret;
    }
    private void backtrack(int N, int K, int pre, int digit, StringBuilder sb, List<Integer> res) {
        if (digit == N) {
            if (sb.length() > 1 && sb.charAt(sb.length() - 1) == '0')
                return;
            res.add(Integer.valueOf(new StringBuilder(sb).reverse().toString()));
            return;
        }
        for (int i = 0; i <= 9; i++) {
            
if (pre == -1 || pre - i == K || i - pre == K) {
                sb.append(i);
                backtrack(N, K, i, digit + 1, sb, res);
                sb.deleteCharAt(sb.length()-1);
            }
        }
    }
}
