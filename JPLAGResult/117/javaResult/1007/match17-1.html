class Solution {
   public int[] numsSameConsecDiff(int N, int K) {
        if (N == 1) {
            int[] ans = new int[10];
            for (int i = 0; i < 10; i++) {
                ans[i] = i;
            }
            return ans;
        } else 
{
            List<Integer> list = new ArrayList<>();
            dfs(N, K, new StringBuilder(), list);
            int[] ans = new int[list.size()];
            for (int i = 0; i < ans.length; i++) {
                ans[i] = list.get(i);

            }
            return ans;
        }
    }

    private void dfs(int N, int K, StringBuilder cur, List<Integer> ans) {
        if (cur.length() == N) {
            int value = Integer.valueOf(cur.toString());
            ans.add(value);
            return;
        }

        if (cur.length() == 0) {
            for (int i = 1; i <= 9; i++) 
{
                cur.append((char) (i + '0'));
                dfs(N, K, cur, ans);
                cur.deleteCharAt(cur.length() - 1);
            }
        } else {
            int lastDigit = cur.charAt(cur.length() - 1) - '0';
            if (lastDigit - K >= 0) {
                cur.append((char) (lastDigit - K + '0'));
                dfs(N, K, cur, ans);
                cur.deleteCharAt(cur.length() - 1);
            }

            if (K > 0 && lastDigit + K <= 9) {
                cur.append((char) (lastDigit + K + '0'));
                dfs(N, K, cur, ans);
                cur.deleteCharAt(cur.length() - 1);
            }
        }
    }
}
