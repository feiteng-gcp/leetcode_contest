class Solution {

    public int[] numsSameConsecDiff(int N, int K) {
        char[] tmp = new char[N];

        List<String> res = new ArrayList<>();
        if (N == 1) {
            res.add("0");
        }

        for (char i = '1'; i <= '9'; i++) {
            tmp[0] = i;
            helper(res, tmp, N, K, 1);
        }

        int[] ret = new int[res.size()];
        for (int i = 0; i < res.size(); i++) {
            ret[i] = Integer.parseInt(res.get(i));
        }
        return ret;
    }

    public void helper(List<String> res, char[] tmp, int N, int K, int i) {
        if (i == N) {
            res.add(String.valueOf(tmp));
            return;
        }

        char next = (char) (tmp[i - 1] + K);
        if (next >= '0' && next <= '9') {
            tmp[i] = next;
            helper(res, tmp, N, K, i + 1);
        }

        if (K > 0) {
            next = (char) (tmp[i - 1] - K);
            if (next >= '0' && next <= '9') {
                tmp[i] = next;
                helper(res, tmp, N, K, i + 1);
            }
        }

    }
}
