class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        if(N == 1) {
            return new int[] {1, 2, 3, 4, 5, 6, 7, 8, 9, 0};
        }
        List<Integer> list = new ArrayList<>();
        for(int i = 1; i <= 9; i++) {
            help(i, 1, N, K, list);
        }
        int[] result = new int[list.size()];
        for(int i = 0; i < result.length; i++) result[i] = list.get(i);
        return result;
    }
    
    private void help(int val, int index, int len, int dif, List<Integer> list) {
        if(index == len) {list.add(val); return;}
        int digit = val % 10;
        val *= 10;
        if(dif == 0) help(val+digit, index+1, len, dif, list);
        else {
            if(digit + dif >= 0 && digit + dif <= 9) help(val+digit+dif, index+1, len, dif, list);
            if(digit - dif >= 0 && digit - dif <= 9) help(val+digit-dif, index+1, len, dif, list);
        }
    }
}
