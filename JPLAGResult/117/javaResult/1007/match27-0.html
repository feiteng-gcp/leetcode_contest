class Solution {
    public int[] numsSameConsecDiff(int n, int k) {
        List<Integer> list = new ArrayList<>();
        dfs(n, "", k, list);
        if(n == 1) {
            list.add(0);
        }
        int[] ret = new int[list.size()];
        for(int i = 0; i < list.size(); ++i) {
            ret[i] = list.get(i);
        }
        return ret;
    }
    
    void dfs(int n, String cur, int k, List<Integer> list) {
        if(cur.length() == n) {
            list.add(Integer.valueOf(cur));
            return;
        }
        if(cur.length() == 0) {
            for(int i = 1; i <= 9; ++i) {
                dfs(n, String.valueOf(i), k, list);
            }
        }
        else {
            int lastd = cur.charAt(cur.length()-1) - '0';
            
            int prevd = lastd-k;
            if(prevd >= 0) {
                dfs(n, cur + prevd, k, list);
            }
            
            int nextd = lastd+k;
            if(nextd != prevd && nextd <= 9) {
                dfs(n, cur + nextd, k, list);    
            }
            
        }
        
    }
}
