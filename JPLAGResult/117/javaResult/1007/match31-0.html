class Solution {
    public int[] numsSameConsecDiff(int N, int K) {
        if (N == 1) {
            int[] res = new int[10];
            for (int i = 0; i < res.length; i++) {
                res[i] = i;
            }
            return res;
        } else {
            List<Integer> res = new ArrayList();
            for (int i = 1; i <= 9; i++) {
                helper(res, 1, i + "", N, K);
                // helper(res);
            }
            int[] ans = new int[res.size()];
            for (int i = 0; i < ans.length; i++) {
                ans[i] = res.get(i);
            }
            return ans;
        }
    }
    
    private void helper(List<Integer> res, int index, String cur, int N, int k) {
        if (index == N) {
            res.add(Integer.parseInt(cur));
            return;
        }
        int val = cur.charAt(cur.length() - 1) - '0';
        if (val - k >= 0) {
            helper(res, index + 1, cur + (val - k), N, k);
        }
        if (k != 0 && val + k <= 9) {
            helper(res, index + 1, cur + (val + k), N, k);
        }
    }
}
