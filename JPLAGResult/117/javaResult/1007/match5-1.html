class Solution {
    // backtracking
    public int[] numsSameConsecDiff(int N, int K) {
        
List<Integer> list = new ArrayList<>();
        if (N == 1) {
            list.add(0);
        }
        for (int i = 1 ;i <= 9 ; i++) {
            
create(list, i, i, N - 1, K);
        }
        int[] array = new int[list.size()];
        for (int i = 0 ; i < list.size() ; i++) {
            array[i] = 
list.get(i);
        }
        return array;
    }
    
    private void create(List<Integer> list, int pre, int last, int nLeft, int K) {
        if (nLeft == 0) {
            list.add(pre);
            return;
        }
        if (last + K <= 9) {
            create(list, pre * 10 + last + K, last + K, nLeft - 1, K);
        }
        if (K!= 0 && last - K >= 0) {
            create(list, pre * 10 + last - K, last - K, nLeft - 1, K);
        }
    }
}
