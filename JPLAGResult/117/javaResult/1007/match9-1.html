class Solution {

    public int[] numsSameConsecDiff(int N, int K) {
        Set<String> sublist = new HashSet<>();
        dfsHelp(N, K, sublist, new StringBuilder());
        // System.out.println(sublist);
        int[] anw = new int[sublist.size()];
        Iterator<String> it = sublist.iterator();
        for (int i = 0; i < anw.length; i++) {
            anw[i] = Integer.valueOf(
it.next());
        }
        return anw;
    }
    private void dfsHelp(int N, int K, Set<String> sublist, StringBuilder sb) {
        if (sb.length() == N) {
            sublist.add(sb.toString());
            return;
        }
        if (sb.length() == 0) {
            if (N == 1) {
                sb.append(0);
                dfsHelp(N, K, sublist,sb);
                sb.deleteCharAt(sb.length() - 1);
            } 
            for (int i = 1; i <= 9; i++) 
{
                sb.append(i);
                dfsHelp(N, K, sublist,sb);
                sb.deleteCharAt(sb.length() - 1);
            }
        } else {
            char c = sb.charAt(sb.length() - 1);
            char r = (char)(c + K);
            char l = (char) (c - K);
            
if (r <= '9') {
                sb.append(r);
                dfsHelp(N, K, sublist,sb);
                sb.deleteCharAt(sb.length() - 1);
            }
            if (l >= '0') {
                sb.append(l);
                dfsHelp(N, K, sublist,sb);
                sb.deleteCharAt(sb.length() - 1);
            }
        }
    }



}
