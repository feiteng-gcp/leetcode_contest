class Solution {
public:
    void helper(int cur, int num, int N, int K, vector<int> &ret)
    {
        if (num==N)
        {
            ret.push_back(cur);
            return;
        }
        int last=cur%10;
        int next=last+K;
        if (next<=9)
        {
            int temp=cur*10+next;
            helper(temp, num+1, N, K, ret);
        }
        if (K!=0 && (next=last-K)>=0)
        {
            int temp=cur*10+next;
            helper(temp, num+1, N, K, ret);
        }
    }
    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> ret;
        if (N==1)
            ret.push_back(0);
        for (int i=1; i<=9; i++)
        {
            int cur=i;
            helper(cur, 1, N, K, ret);
        }
        return ret;
    }
};
