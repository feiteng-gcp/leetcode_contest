class Solution {
public:
    int num;
    vector<int> a;

    void dfs(int x, int y, int sum, int N, int K){
        if (x > N){
            a.insert(a.end(), sum);
            return;
        }
        if (y - K >= 0)
            dfs(x + 1, y - K, sum * 10 + y - K, N, K);
        if (K > 0 && y + K <= 9)
            dfs(x + 1, y + K, sum * 10 + y + K, N, K);
    }
    vector<int> numsSameConsecDiff(int N, int K) {
        num = 0;
        for (int i = 1; i <= 9; i++)
            dfs(2, i, i, N, K);
        if (N==1) dfs(2, 0, 0, N, K);
        return a;
    }
};
