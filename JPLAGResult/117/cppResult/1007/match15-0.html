class Solution {
public:
    vector<int> ret;

    vector<int> numsSameConsecDiff(int N, int K) {
        if (N == 0)
            return {0};

        ret.clear();
        if (N == 1) {
            for (int i = 0; i <= 9; ++i)
                ret.push_back(i);
            return ret;
        }


        for (int i = 1; i <= 9; ++i)
            helper(i, N - 1, K, i);// 1

        return ret;
    }

    void helper(int first, int N, int K, int x) { // 1 1 1 1
        if (N <= 0) {
            ret.push_back(x);
            return;
        }

        int temp = first; // 1
        if (temp + K < 10)
            helper(temp + K, N - 1, K, x * 10 + temp + K);
        if (temp - K >= 0 && K != 0)
            helper(temp - K, N - 1, K, x * 10 + temp - K);
    }
};
