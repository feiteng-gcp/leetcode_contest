class Solution {
public:
    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> ans;
        if(N==1){
            for(int i=0;i<10;i++){
                ans.push_back(i);
            }
            return ans;
        }
        queue<int> que;
        for(int i=1;i<10;i++){
            if(i+K<=10||i-K>=0) que.push(i);
        }
        if(K==0&&N==1) ans.push_back(0);
        while(--N){
            int size=que.size();
            for(int i=0;i<size;i++){
                int n=que.front();
                que.pop();
                if(n%10+K<10){
                    int num=n*10+n%10+K;
                    que.push(num);
                }
                if(K&&(n%10-K>=0)){
                    int num=n*10+n%10-K;
                    que.push(num);
                }
            }
        }
        while(!que.empty()){
            ans.push_back(que.front());
            que.pop();
        }
        
        return ans;
    }
};
