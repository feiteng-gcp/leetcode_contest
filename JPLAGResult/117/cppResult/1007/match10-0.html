class Solution {
public:
vector<int> numsSameConsecDiff(int N, int K) {
    vector<int> v;
    for (int i=1;i<=9;++i)v.push_back(i);
    if (N==1)v.push_back(0);
    for (int i=2;i<=N;++i) {
        vector<int> prev;
        prev.swap(v);
        for (int p: prev) {
            int r=p%10;
            if (r+K<=9)v.push_back(p*10+r+K);
            if (K!=0&&r>=K)v.push_back(p*10+r-K);
        }
    }
    return v;
}


};
