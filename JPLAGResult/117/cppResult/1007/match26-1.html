class Solution {
public:

    void dfs(vector<int> &st, int lv, int& N, int k, int now){
        //cout<<lv<<','<<N<<","<<k<<','<<now<<" @ ";
        if (lv>=N) {st.push_back(now); return;}
        int p = now % 10;
        if (p+k<10) dfs(st, lv+1, N, k, now*10+p+k);
        if (k && p-k>=0) dfs(st, lv+1, N, k, now*10+p-k);
    }
    
    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> st;
        if (N==1) st.push_back(0);
        for(int i = 1; i<10;i++)
            dfs(st, 1, N, K, i);    
        return st;
    }
};
