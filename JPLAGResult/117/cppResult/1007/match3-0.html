class Solution {
public:
    void rec(vector<int>& ans, string& cur, int N, int K) {
        if (cur.size() == N) {
            ans.push_back(atoi(cur.c_str()));
            return;
        }
        int val = cur.back() - '0';
        if (K == 0) {
            cur.push_back(val+'0');
            rec(ans, cur, N, K);
            cur.pop_back();
            return;
        }
        if (val+K <= 9) {
            cur.push_back(val+K+'0');
            rec(ans, cur, N, K);
            cur.pop_back();
        }
        if (val-K >= 0) {
            cur.push_back(val-K+'0');
            rec(ans, cur, N, K);
            cur.pop_back();
        }
    }
    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> ans;
        for (int i = (N == 1) ? 0 : 1; i <= 9; ++i) {
            char input = '0'+i;
            string tmp(1, input);
            rec(ans, tmp, N, K);
        }
        return ans;
    }
};
