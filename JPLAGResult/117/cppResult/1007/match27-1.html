class Solution {
public:
    void append(int t, int l, int N, int K, vector<int>& v) {
        if (N == 1) {
            if (l + K <= 9) v.push_back(t * 10 + l + K);
            if (K != 0 && l - K >= 0) v.push_back(t * 10 + l - K);
            return;
        }
        if (l + K <= 9) append(t*10+l+K, l+K, N-1, K, v);
        if (K != 0 && l - K >= 0) append(t*10+l-K, l-K, N-1, K, v);
    }

    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> v;
        if (N == 1) {
            for (int i = 0; i < 10; ++i) v.push_back(i);
            return v;
        }
        for (int i = 1; i <= 9; ++i) append(i, i, N-1, K, v);
        return v;
    }
};
