class Solution {
public:
    void gen(int val, int cur, int i, int N, int K, vector < int > &res) {

        if (i == N + 1) {
            res.push_back(val);
            return;
        }
        if (cur + K <= 9) {
            gen(val * 10 + (cur + K), cur + K, i + 1, N, K, res);
        }
        if (K != 0 && cur - K >= 0) {
            gen(val * 10 + (cur - K), cur - K, i + 1, N, K, res);
        }
    }
    vector<int> numsSameConsecDiff(int N, int K) {
        vector < int > res;
        for (int st = 1; st <= 9; st ++) {
            gen(st, st, 2, N, K, res);
        }
        if (N == 1) {
            res.push_back(0);
        }
        return res;
    }
};
