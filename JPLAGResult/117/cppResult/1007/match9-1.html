typedef vector<int> vi;
class Solution {
public:
    vector<int> numsSameConsecDiff(int N, int K) {
        vi ret;
        if (N == 0) return ret;
        if (N == 1) ret.push_back(0);

        for (int i = 1; i < 10; i++) {
            rek(ret, i, N-1, K);
        }
        return ret;
    }
    void rek(vi &ret, int cur, int depth, int diff) {
        if (depth == 0) {
            ret.push_back(cur);
            return;
        }
        int last = cur % 10;
        cur *= 10;

        if (last - diff >= 0) {
            int tmp = last - diff;
            rek(ret, cur + tmp, depth-1, diff);
        }
        if (diff > 0 && last + diff < 10) {
            int tmp = last + diff;
            rek(ret, cur+tmp, depth-1, diff);
        }
    }
};
