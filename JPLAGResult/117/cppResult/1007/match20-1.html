class Solution {
private:

    vector<int>ret;
    void dfs(int p,int n,int k,int num,int last){
        if(p>=n){
            ret.push_back(num);
            return;
        }
        if(last+k<10){
            dfs(p+1,n,k,num*10+last+k,last+k);
        }
        if(last-k>=0){
            dfs(p+1,n,k,num*10+last-k,last-k);

        }
    }
public:
    vector<int> numsSameConsecDiff(int N, int K) {
        ret.clear();
        if(N==1){
            for(int i=0;i<10;i++)
                ret.push_back(i);
            return ret;
        }
        for(int i=1;i<10;i++){
            dfs(1,N,K,i,i);
        }
        set<int>s;
        for(int i=0;i<ret.size();i++)
            s.insert(ret[i]);
        ret.clear();
        for(auto it=s.begin();it!=s.end();it++)
            ret.push_back(*it);
        return ret;
    }
};
