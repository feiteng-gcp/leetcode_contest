class Solution {
    vector<int> res;
    void go(int x, int n, int lastd, int N, int K) {
        if (n >= N) {
            res.push_back(x);
            return;
        }
        for (int d = n > 0 ? 0 : 1; d < 10; ++d) {
            if (n > 0) {
                if (abs(d-lastd) != K)
                    continue;
            }
            go(x*10 + d, n+1, d, N, K);
        }
    }
public:
    vector<int> numsSameConsecDiff(int N, int K) {
        res.clear();
        if (N == 1) {
            for (int x = 0; x < 10; ++x)
                res.push_back(x);
            return res;
        }
        go(0, 0, -1, N, K);
        return res;
    }
};
