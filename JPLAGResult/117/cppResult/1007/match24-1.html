class Solution {
public:
	vector<int> numsSameConsecDiff(int n, int k) {
		vector<int> res;
		if (n == 1) {
			for (int i = 0; i < 10; i++) {
				res.emplace_back(i);
			}
		}
		else {
			backtrack(res, 0, n, k);
		}
		return res;
	}
private:
	void backtrack(vector<int>& res, int current, int n, int k) {
		if (n == 0) {
			res.emplace_back(current);
			return;
		}

		n--;
		if (current == 0) {
			for (int i = 1; i < 10; i++) {
				backtrack(res, i, n, k);
			}
		}
		else {
			int last_digit = current % 10;
			current *= 10;

			if (k == 0) {
				backtrack(res, current + last_digit, n, k);
			}
			else {
				if (last_digit - k >= 0) {
					backtrack(res, current + last_digit - k, n, k);
				}
				if (k != 0 && last_digit + k <= 9) {
					backtrack(res, current + last_digit + k, n, k);
				}
			}
		}
	}
};
