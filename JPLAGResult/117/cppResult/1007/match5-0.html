class Solution {
public:
    vector<int> sol;
    void dfs(int idx, int last, int k, int n) {
        if (idx == 0) {
            sol.push_back(n);
            return;
        }
        if (last - k >= 0 && n != 0) 
            dfs(idx - 1, last - k, k, n * 10 + last - k);
        if (k && last + k <= 9 && n != 0) 
            dfs(idx - 1, last + k, k, n * 10 + last + k);
    }
    vector<int> numsSameConsecDiff(int N, int K) {
        for (int i = 0; i <= 9; ++i)
            dfs(N - 1, i, K, i);
        return sol;
    }
};
