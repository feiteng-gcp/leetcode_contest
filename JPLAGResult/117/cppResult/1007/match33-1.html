class Solution {
public:
    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> ret;
        if(N==1){
            for(int i=0;i<10;i++){
                ret.push_back(i);                
            }
            return ret;
        }
        
        vector<int> cache = numsSameConsecDiff(N-1, K);
            for(auto n:cache){
                if(n==0)  continue;
                int one = n%10;
                if(one>=K)
                    ret.push_back(n*10+(one-K));
                if(K!=0 && one+K<10)
                    ret.push_back(n*10+(one+K));            
            }
            return ret;
    }
};
