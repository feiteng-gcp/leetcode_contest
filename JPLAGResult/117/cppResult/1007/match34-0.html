class Solution {
public:
    vector<int> numsSameConsecDiff(int N, int k) {
        unordered_set<int> s;
        for(int i = 0;i<=9;i++)
            s.insert(i);
        for(int i = 1;i<N;i++){
            unordered_set<int> p;
            int j = 1;
            int ten = 1;
            while(j<i){
                ten*= 10;
                j++;
            }
            for(auto e:s){
                int r = e/ten;
                if((r-k)>=0){
                    p.insert((r-k)*ten*10 + e);
                }
                if((r+k)<=9){
                    p.insert((r+k)*ten*10 + e);
                }
            }
            s = p;
        }
        int i = 1;
        int ten = 1;
        while(i<N){
            ten*= 10;
            i++;
        }
        vector<int> v;
        for(auto e:s){
            if(e/ten >0)
                v.push_back(e);
        }
        if(N==1)
            v.push_back(0);
        return v;
    }
};
