#include <vector>

using namespace std;

class Solution {
public:
    vector<int> numsSameConsecDiff(int N, int K) {
        //2^9*9
        vector<int> res;
        if(N == 1){ //1 9
            for(int i = 0; i < 10; i++){
                res.push_back(i);
            }
            return res;
        }
        if(K == 0){ // N > 1. 0...0 is not solution
            for(int h = 1; h < 10; h++){
                int s = 0;
                for(int b = 0; b < N; b++){
                    s = s * 10 + h;
                }
                res.push_back(s);
            }
            return res;
        }
        for(int h = 1; h < 10; h++){
            int maske = 1 << N;
            for(int mask = 0; mask < maske; mask++){
                if(mask & 1){
                    continue; //first must be +
                }
                int s = h, x = h; // s <= 10^9 - 1
                for(int b = 1; b < N; b++){
                    if((mask >> b) & 1){
                        x -= K;
                    }else{
                        x += K;
                    }
                    if(x < 0 || x > 9){
                        s = -1;
                        break;
                    }
                    s = s * 10 + x;
                }
                if(s > 0){
                    res.push_back(s);
                }
            }
        }
        return res;
    }
};
