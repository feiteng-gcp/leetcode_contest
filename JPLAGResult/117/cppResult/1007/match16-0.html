class Solution {
public:
    vector<int> res;
    void dfs(int num, int len1, int N, int K)
    {
        if(len1 == N)
        {
            res.push_back(num);
            return;
        }
        int c = num%10;
        
        if(K==0)
        {
            dfs(num*10+c, len1+1, N, K);
        }
        else
        {   
            if(c-K>=0)
                dfs(num*10+c-K, len1+1, N, K);
            if(c+K < 10)
                dfs(num*10+c+K, len1+1, N, K);
        }
        return;
    }
    vector<int> numsSameConsecDiff(int N, int K) {
        if(N==1)
        {
            for(int i=0;i<10;i++) res.push_back(i);
            return res;
        }
        for(int i=1;i<=9;i++)
        {
            dfs(i, 1, N, K);
        }
        return res;
    }
};
