class Solution {
public:
    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> ret;
        for (int i = 1; i <= 9; ++i) {
            dfs(ret, i, i, 1, N, K);
        }
        if (N == 1) {
            ret.push_back(0);
        }
        return ret;
    }
    
    void dfs(vector<int>& ret, int val, int lastNum, int idx, int N, int K) {
        if (idx == N) {
            ret.push_back(val);
            return;
        }
        
        if (lastNum >= K) {
            dfs(ret, val * 10 + lastNum - K, lastNum - K, idx + 1, N, K);
        }
        if (K != 0 && lastNum + K < 10) {
            dfs(ret, val * 10 + lastNum + K, lastNum + K, idx + 1, N, K);
        }
    }
};
