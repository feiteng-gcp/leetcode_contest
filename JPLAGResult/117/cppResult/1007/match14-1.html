vector<int> res;

class Solution {
public:
    void search(int cur,int tot,int last,int dif,int num)
{
	if(cur==tot)
	{
		res.push_back(num);
		return;
	}
	if(cur==0)
	{
		for(int i=1;i<10;i++)
		{
			search(cur+1,tot,i,dif,i);
		}
	}
	else
	{
		if(last-dif>=0)search(cur+1,tot,last-dif,dif,num*10+last-dif);
		if(dif!=0&&last+dif<10)search(cur+1,tot,last+dif,dif,num*10+last+dif);
	}
}
    vector<int> numsSameConsecDiff(int N, int K) {
        res.clear();
	if(N==1)
	{
		for(int i=0;i<10;i++)
		{
			res.push_back(i);
		}
		return res;
	}
	search(0,N,-1,K,0);
	return res;
    }
};
