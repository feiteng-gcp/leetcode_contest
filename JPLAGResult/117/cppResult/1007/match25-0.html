class Solution {
public:
    vector<int> helper(int n, int k) {
        vector<int> res;
        if(n == 1) {
            for(int i=1; i<=9; i++) {
                res.push_back(i);
            }
            return res;
        }
        vector<int> r = helper(n-1, k);
        for(int i=0; i<r.size(); i++) {
            if(k == 0) {
                res.push_back(r[i]*10 + (r[i]%10));
            } else {
                if(r[i]%10-k >= 0) {
                    res.push_back(r[i]*10 + (r[i]%10) - k);
                }
                if(r[i]%10+k < 10) {
                    res.push_back(r[i]*10 + (r[i]%10) + k);
                }
            }
        }
        return res;
    }
    
    
    vector<int> numsSameConsecDiff(int n, int k) {
        vector<int> res;
        if(n == 1) {
            for(int i=0; i<=9; i++) {
                res.push_back(i);
            }
            return res;
        }
        res = helper(n, k);
        return res;
    }
};
