class Solution {
public:
    vector<int> ans;
    int n, k;
    void dfs(int cur, int dep) {
        if (dep==n+1) {
            ans.push_back(cur);
            return;
        }
        int t=cur%10;
        if (t+k<=9) dfs(cur*10+t+k, dep+1);
        if (t-k>=0) dfs(cur*10+t-k, dep+1);
    }
    vector<int> numsSameConsecDiff(int N, int K) {
        n=N;
        k=K;
        ans.clear();
        for(int i=1; i<=9; i++) {
            dfs(i, 2);
        }
        if (N==1) ans.push_back(0);
        sort(ans.begin(), ans.end());
        vector<int>::iterator iter = unique(ans.begin(),ans.end());
        ans.erase(iter, ans.end());
        return ans;
    }
};
