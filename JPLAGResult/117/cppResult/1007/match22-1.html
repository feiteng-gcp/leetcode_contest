class Solution {
public:
    void dfs(vector<int>& res, vector<int> sample, int N, int K, int step)
    {
        if(step == N)
        {
            if(sample[N - 1] == 0 && N > 1)
                return;
            else {
                int x = 0;
                int s = 1;
                for(int i = 0; i < N; i++)
                {
                    x = x + sample[i] * s;
                    s = s * 10;
                }
                res.push_back(x);
            }
        }
        else{
            if(sample[step - 1] + K <= 9 && sample[step - 1] + K >= 0)
            {
                sample[step] = sample[step - 1] + K;
                dfs(res, sample, N, K, step + 1);
            }
            if(K != 0 && sample[step - 1] - K <= 9 && sample[step - 1] - K >= 0)
            {
                sample[step] = sample[step - 1] - K;
                dfs(res, sample, N, K, step + 1);

            }
        }
    }
    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> res;
        for(int i = 0; i < 10; i++)
        {
            vector<int> sample(N, 0);
            sample[0] = i;
            dfs(res, sample, N, K, 1);
        }
        return res;
    }
};
