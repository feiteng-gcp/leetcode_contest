class Solution {
public:
    
    void bfs(int n, int k, int l, int num, vector<int>& sol) {
        if (l == n) {
            sol.push_back(num);
            return;
        }
        for (int i = 0; i < 10; ++i) {
            int diff = abs((num % 10) - i);
            if (diff == k) {
                bfs(n, k, l + 1, num*10+i, sol);
            }
        }
    }
    
    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> sol;
        for (int i = 1; i < 10; ++i) {
            bfs(N, K, 1, i, sol);
        }
        if (N == 1) {
            sol.push_back(0);
        }
        return sol;
    }
};
