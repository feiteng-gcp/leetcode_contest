class Solution {
public:
    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> res;
        if(N == 1)
        {
            for(int i = 0; i <= 9; i++)
                res.push_back(i);
            return res;
        }
        for(int i = 1; i <= 9; i++)
            search(i, N-1, K, res);
        return res;
    }
    void search(int s, int n, int k, vector<int>& res)
    {
        if(n == 0)
            res.push_back(s);
        else
        {
            int d = s % 10;
            if(valid(d+k))
            {
                search(s*10 + d + k, n-1, k, res);
            }
            if(k != 0 && valid(d - k))
                search(s * 10 + d - k, n - 1, k, res);
        }
    }
    bool valid(int n)
    {
        return n >= 0 && n <= 9;
    }
};
