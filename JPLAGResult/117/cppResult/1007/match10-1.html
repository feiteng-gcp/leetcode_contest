class Solution {
public:
    vector<int> numsSameConsecDiff(int N, int K) {
        vector<int> res;
        for (int i = 1; i <= 9; i++) {
            res.push_back(i);
        }

        for (int i = 0; i < N - 1; i++) {
            vector<int> curr;
            for (int j : res) {
                int mod = j % 10;
                if (mod + K < 10) {
                    curr.push_back(j*10+mod+K);
                }
                if (K != 0 && mod - K >= 0) {
                    curr.push_back(j*10+mod-K);
                }
            }
            res = curr;
        }
        if (N == 1) res.push_back(0);
        return res;
    }
};
