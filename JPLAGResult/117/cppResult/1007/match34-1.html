class Solution {
public:
    vector<int> numsSameConsecDiff(int N, int K) {
        unordered_set<int> tmp, tmp1;
        for (int i=0; i<10; i++)
            tmp.insert(i);
        
        for (int i=2; i<=N; i++) {
            for (int t : tmp) {
                if (t % 10 + K < 10)
                    tmp1.insert(t * 10 + t % 10 + K);
                if (t % 10 - K >= 0)
                    tmp1.insert(t * 10 + t % 10 - K);
            }
            tmp = tmp1;
            tmp1.clear();
        }
        
        vector<int> res;
        for (int t : tmp)
            if (t >= pow(10, N-1))
                res.push_back(t);
        if (N == 1)
            res.push_back(0);
        
        return res;
    }
};
