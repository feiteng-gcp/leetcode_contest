class Solution {
public:
    vector<int> ans;
    void dfs(int numlen, int number, int length, int diff) {
        if (numlen == length) {
            ans.push_back(number);
            return;
        }
        if (number % 10 - diff >= 0)
            dfs(numlen + 1, number * 10 + number % 10 - diff, length, diff);
        if (number % 10 + diff <= 9 && number % 10 - diff != number % 10 + diff)
            dfs(numlen + 1, number * 10 + number % 10 + diff, length, diff);
    }
    vector<int> numsSameConsecDiff(int N, int K) {
        if (N == 1)
            ans.push_back(0);
        for (int i = 1; i <= 9; ++i)
            dfs(1, i, N, K);
        return ans;
    }
};
