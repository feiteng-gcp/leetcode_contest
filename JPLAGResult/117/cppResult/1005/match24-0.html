/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isUnivalTree(TreeNode* root) {
        if(!root) return true;
        return uni(root->left, root->val) && uni(root->right, root->val);
    }
    bool uni(TreeNode* root, int val) {
        if(!root) return true;
        return (root->val == val) && uni(root->left, val) && uni(root->right, val);
    }
};
