/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    
    bool findRes(TreeNode* node, int temp){
        if(!node) return true;
        if(node->val!=temp) return false;
        return findRes(node->left, node->val) && findRes(node->right, node->val);
    }
    
    bool isUnivalTree(TreeNode* root) {
        if(!root) return true;
        bool res=false;
        return findRes(root->left, root->val) && findRes(root->right, root->val);
        // if(res==false) return false;
        // return findRes(root->right, root->val);
    }
};
