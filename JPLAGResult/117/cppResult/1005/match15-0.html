/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    bool check(TreeNode *r, int v) {
        if (!r) return true;
        if (r->val != v) return false;
        return check(r->left, v) && check(r->right, v);
    }
public:
    bool isUnivalTree(TreeNode* r) {
        if (!r) return true;
        return check(r->left, r->val) && check(r->right, r->val);
    }
};
