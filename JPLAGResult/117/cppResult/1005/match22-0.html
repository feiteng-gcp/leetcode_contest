/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    set<int> st;
    void traverse(TreeNode *root) {
        if(root == NULL) return;
        st.insert(root->val);
        traverse(root->left);
        traverse(root->right);
    }
    
    bool isUnivalTree(TreeNode* root) {
        st.clear();
        traverse(root);
        if(st.size() == 1) return true;
        return false;
    }
};
