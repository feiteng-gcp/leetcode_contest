/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isUnivalTree(TreeNode* root) {
        if (!root) return false;
        return go(root, root->val);
    }
    
    bool go(TreeNode* root, int v) {
        if (!root) return true;
        if (root->val != v) return false;
        return go(root->left, v) && go(root->right, v);
    }
};
