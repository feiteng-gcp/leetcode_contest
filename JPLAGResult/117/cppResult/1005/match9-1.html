/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    bool helper(TreeNode* tn,int &val){
        if(!tn) return true;
        if(tn->val!=val) return false;
        return helper(tn->left,val)&&helper(tn->right,val);
    }
    bool isUnivalTree(TreeNode* root) {
        int val=root->val;
        return helper(root,val);
    }
    
};
