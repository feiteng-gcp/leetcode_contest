class Solution {
public:
	bool dfs(TreeNode *t) {
		if (t == NULL) return true;
		if (t->left != NULL && t->val != t->left->val) return false;
		if (t->right != NULL && t->val != t->right->val) return false;
		return dfs(t->left) & dfs(t->right);
	}
    bool isUnivalTree(TreeNode* root) {
		return dfs(root);    
    }
};
