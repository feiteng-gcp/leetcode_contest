/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    set<int>S;
    
    void dfs(TreeNode* root){
        if(root==NULL)return ;
        S.insert(root->val);
        dfs(root->left);
        dfs(root->right);
    }
    
    bool isUnivalTree(TreeNode* root) {
        S.clear();
        dfs(root);
        if(S.size()==1)return true;
        return false;
    }
};
