/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isUnivalTree(TreeNode* root) {

        queue<TreeNode*> Q; 
        Q.push(root); 
        int val = root->val; 
        
        while(!Q.empty()) {
            
            TreeNode* node = Q.front(); 
            Q.pop(); 
            
            if(node->val != val)
                return false; 
            
            if(node->left) 
                Q.push(node->left); 
            
            
            if(node->right)
                Q.push(node->right); 
            
        }
        
        return true; 
    }
};
