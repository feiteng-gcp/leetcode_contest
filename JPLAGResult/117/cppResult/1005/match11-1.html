/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    set<int> s;
    void func(TreeNode* a)
    {
        if (a==NULL) return;
        s.insert(a->val);
        func(a->left);
        func(a->right);
    }
    bool isUnivalTree(TreeNode* root) {
        func(root);
        return s.size()<=1;
    }
};
