/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    set<int> s;
    void dfs(TreeNode *node) {
        if (node == NULL)
            return;
        s.insert(node->val);
        dfs(node->left);
        dfs(node->right);
    }
    bool isUnivalTree(TreeNode* root) {
        dfs(root);
        return s.size() == 1;
    }
};
