class Solution {
public:
  string replaceVowel(string s) {
    for (char& c : s) {
      if (c == 'a' || c == 'i' || c == 'e' || c == 'o' || c == 'u') {
        c = '_';
      }
    }
    return s;
  }
  
  vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
    set<string> fullMatch;
    map<string, string> caseMatch;
    map<string, string> vowelMatch;
    for (string& s : wordlist) {
      fullMatch.insert(s);
      string lowercase = s;
      transform(lowercase.begin(), lowercase.end(), lowercase.begin(), ::tolower);
      if (caseMatch.find(lowercase) == caseMatch.end()) {
        caseMatch[lowercase] = s;
      }
      string novowel = replaceVowel(lowercase);
      if (vowelMatch.find(novowel) == vowelMatch.end()) {
        vowelMatch[novowel] = s;
      }
    }
    vector<string> res;
    for (string& s : queries) {
      string lowercase = s;
      transform(lowercase.begin(), lowercase.end(), lowercase.begin(), ::tolower);
      string novowel = replaceVowel(lowercase);
      if (fullMatch.find(s) != fullMatch.end()) {
        res.push_back(s);
      } else if (caseMatch.find(lowercase) != caseMatch.end()) {
        res.push_back(caseMatch[lowercase]);
      } else if (vowelMatch.find(novowel) != vowelMatch.end()) {
        res.push_back(vowelMatch[novowel]);
      } else {
        res.push_back("");
      }
    }
    return res;
  }
};
