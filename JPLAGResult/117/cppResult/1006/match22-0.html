class Solution {
public:
    vector<string> spellchecker(vector<string>& words, vector<string>& queries) {
        unordered_set<string> sames;
        unordered_map<string, string> capts;
        unordered_map<string, string> mutes; 
        for (const string& w : words) {
            sames.insert(w);
            string w1;
            for (char ch : w) w1 += tolower(ch);
            if (capts.find(w1) == capts.end()) {
                capts[w1] = w;
            }
            string w2;
            for (char ch: w1) {
                if ('a' == ch || 'e' == ch || 'i' == ch || 'o' == ch || 'u' == ch) {
                    w2 += '*';
                } else {
                    w2 += ch;
                }
            }
            if (mutes.find(w2) == mutes.end()) {
                mutes[w2] = w;
            }
        }
        int n = queries.size();
        vector<string> ret(n);
        for (int i = 0; i < n; ++i) {
            if (sames.find(queries[i]) != sames.end()) {
                ret[i] = queries[i];
            } else {
                string w1;
                for (char ch : queries[i]) w1 += tolower(ch);
                if (capts.find(w1) != capts.end()) {
                    ret[i] = capts[w1];
                } else {
                    string w2;
                    for (char ch : w1) {
                        if ('a' == ch || 'e' == ch || 'i' == ch || 'o' == ch || 'u' == ch) {
                            w2 += '*';
                        } else {
                            w2 += ch;
                        }
                    }
                    if (mutes.find(w2) != mutes.end()) {
                        ret[i] = mutes[w2];
                    }
                }
            }
        }
        return ret;
    }
};
