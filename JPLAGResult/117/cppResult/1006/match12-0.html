class Solution {
public:
    string toLower(string w)
{
    string s = w;
    int wLen = (int)w.size();
    for(int j = 0; j < wLen; j++)
        if(s[j] >= 'A' && s[j] <= 'Z') s[j] += 'a' - 'A';
    return s;
}

string deVowel(string w)
{
    string s = "";
    int wLen = (int)w.size();
    for(int j = 0; j < wLen; j++)
    {
        if(w[j] != 'a' && w[j] != 'e' && w[j] != 'i' && w[j] != 'o' && w[j] != 'u')
            s.push_back(w[j]);
        else s.push_back('a');
    }
    return s;
}

vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries)
{
    map<string, int> d1, d2, d3;
    int len = (int)wordlist.size();
    for(int i = len - 1; i >= 0; i--)
    {
        d1[wordlist[i]] = i;
        string s1 = toLower(wordlist[i]);
        d2[s1] = i;
        string s2 = deVowel(s1);
        d3[s2] = i;
    }
    vector<string> ans;
    for(int i = 0; i < queries.size(); i++)
    {
        if(d1.count(queries[i]))
        {
            ans.push_back(wordlist[d1[queries[i]]]);
            continue;
        }
        string q1 = toLower(queries[i]);
        if(d2.count(q1))
        {
            ans.push_back(wordlist[d2[q1]]);
            continue;
        }
        string q2 = deVowel(q1);
        if(d3.count(q2)) ans.push_back(wordlist[d3[q2]]);
        else ans.push_back("");
    }
    return ans;
}
