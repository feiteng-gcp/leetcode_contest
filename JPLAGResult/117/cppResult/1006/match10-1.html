class Solution {
public:
    map<string,string> mpp;
    map<string,string> mp1;
    map<string,string> mp2;
    string encode1(string s){
        for(int i=0;i<s.size();i++){
                s[i]=tolower(s[i]);
        }
        return s;
    }
    string encode2(string s){
        for(int i=0;i<s.size();i++){
            char c=tolower(s[i]);
            s[i]=c;
            if(c=='a'||c=='i'||c=='u'||c=='e'||c=='o'){
                s[i]='a';
            }
        }
        return s;
    }
    vector<string> ans;
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        for(auto s2:wordlist){
            string t=s2;

            mpp[s2]=s2;
            string s=encode1(t);
            if(mp1.count(s)==0){
                mp1[s]=t;
            }
            s=encode2(t);
            if(mp2.count(s)==0){
                mp2[s]=t;
            }
        }
        for(auto s2:queries){
            if(mpp.count(s2)){
                ans.push_back(s2);
                continue;
            }
            string t=s2;
            string s=encode1(t);

            if(mp1.count(s)){
                ans.push_back(mp1[s]);
                continue;
            }
            s=encode2(t);
            if(mp2.count(s)){
                ans.push_back(mp2[s]);
                continue;
            }
            ans.push_back("");
        }
        return ans;
    }
};
