class Solution {
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        unordered_set<string> table;
        unordered_map<string, vector<int>> lower_table;
        unordered_map<string, vector<int>> devowel_table;
        
        for(int i = 0; i < wordlist.size(); ++i) {
            table.insert(wordlist[i]);
            lower_table[lower(wordlist[i])].push_back(i);
            devowel_table[devowel(wordlist[i])].push_back(i);
        }
        
        vector<string> res(queries.size());
        for (int i = 0; i < res.size(); ++i) {
            string& q = queries[i];
            if (table.count(q) > 0) {
                res[i] = q;
            }
            else if (lower_table.count(lower(q)) > 0) {
                res[i] = wordlist[lower_table[lower(q)][0]];
            }
            else if (devowel_table.count(devowel(q)) > 0) {
                res[i] = wordlist[devowel_table[devowel(q)][0]];
            }
            else {
                res[i] = "";
            }
        }
        
        return res;
    }
private:
    string lower(string s) {
        for (int i = 0; i < s.length(); ++i) {
            s[i] = tolower(s[i]);
        }
        return s;
    }
    string devowel(string s) {
        for (int i = 0; i < s.length(); ++i) {
            s[i] = tolower(s[i]);
            if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u') {
                s[i] = '*';
            }
        }
        return s;
    }
};
