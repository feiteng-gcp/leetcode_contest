class Solution {
    unordered_set<string> words;
    unordered_map<string, vector<string>> captl;
    unordered_map<string, vector<string>> vowel;

    string getCaptl(string w) {
        for (char &c : w) {
            if (c >= 'A' && c <= 'Z')
                c += 'a' - 'A';
        }
        return w;
    }
    string getVowel(string w) {
        for (char &c : w) {
            if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u')
                c = '#';
        }
        return w;
    }
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        for (string w : wordlist) {
            words.insert(w);
            string cw = getCaptl(w);
            captl[cw].push_back(w);
            string vw = getVowel(cw);
            vowel[vw].push_back(w);
        }
        
        vector<string> res;
        for (string q : queries) {
            if (words.find(q) != words.end())
                res.push_back(q);
            else {
                string cq = getCaptl(q);
                if (captl.find(cq) != captl.end())
                    res.push_back(captl[cq][0]);
                else {
                    string vq = getVowel(cq);
                    if (vowel.find(vq) != vowel.end())
                        res.push_back(vowel[vq][0]);
                    else
                        res.push_back("");
                }
            }
        }
        return res;
    }
};
