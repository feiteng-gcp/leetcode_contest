class Solution {
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        std::unordered_map<string, string> h1, h2, h3;
        
        for (auto & word : wordlist) {
            h1[word] = word;
            
            string l = lower(word);
            if (!h2.count(l)) {
                h2[l] = word;
            }
            
            string nv = novowel(word);
            if (!h3.count(nv)) {
                h3[nv] = word;
            }
        }
        
        vector<string> result;
        for (auto & q : queries) {
            if (h1.count(q)) {
                result.push_back(h1[q]);
                continue;
            }
            
            string l = lower(q);
            if (h2.count(l)) {
                result.push_back(h2[l]);
                continue;
            }
            
            string nv = novowel(q);
            if (h3.count(nv)) {
                result.push_back(h3[nv]);
                continue;
            }
            
            result.push_back("");
        }
        
        return result;
    }
    
    string lower(string str) {
        for (char & ch : str) {
            if (ch >= 'A' && ch <= 'Z')
                ch = ch - 'A' + 'a';
        }
        return str;
    }
    
    string novowel(string str) {
        str = lower(str);
        
        string r;
        for (char ch : str) {
            if (ch != 'a' && ch != 'e' && ch != 'i' && ch != 'o' && ch != 'u') {
                r.push_back(ch);
            } else {
                r.push_back('*');
            }
        }
        return r;
    }
};
