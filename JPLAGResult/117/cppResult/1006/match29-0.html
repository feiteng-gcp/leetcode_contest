#include <vector>
#include <string>
#include <cctype>
#include <map>
#include <set>

using namespace std;

class Solution {
public:
    string ToLower(const string &s)
    {
        string res;
        for(int i = 0; i < s.size(); i++){
            res.push_back(tolower(s[i]));
        }
        return res;
    }
    
    string ToVow(const string &s){
        string res;
        for(int i = 0; i < s.size(); i++){
            int c = s[i];
            if(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u'){
                c = 'a';
            }
            res.push_back(c);
        }
        return res;
    }
    
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        int m = wordlist.size(), n = queries.size(); //m 1 5e3 n 1 5e3
        map<string, string> mpc, mpe; //trie?
        set<string> st;
        for(int i = 0; i < m; i++){
            st.insert(wordlist[i]);
            string wl = ToLower(wordlist[i]);
            if(mpc.count(wl) == 0){
                mpc[wl] = wordlist[i];
            }
            string wv = ToVow(wl);
            if(mpe.count(wv) == 0){
                mpe[wv] = wordlist[i];
            }
        }
        vector<string> res;
        for(int i = 0; i < n; i++){
            const string &s = queries[i];
            string sl = ToLower(s);
            string sv = ToVow(sl);
            if(st.count(s)){
                res.push_back(s);
            }else if(mpc.count(sl)){
                res.push_back(mpc.at(sl));
            }else if(mpe.count(sv)){
                res.push_back(mpe.at(sv));
            }else{
                res.push_back("");
            }
        }
        return res;
    }
};
