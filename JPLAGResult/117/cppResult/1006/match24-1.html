class Solution {
    public:
        vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
            unordered_set<string> exact_word;
            unordered_map<string, vector<string>> capitalization;
            unordered_map<string, vector<string>> vowel;

            for(string& str : wordlist) {
                exact_word.insert(str);
                string lower = tolower(str);
                capitalization[lower].push_back(str);
                vowel[delvowel(lower)].push_back(str);
            }

            vector<string> ans;
            for(string& str : queries) {
                string lower = tolower(str);
                string novowel = delvowel(lower);
                if(exact_word.count(str) > 0) {
                    ans.push_back(str);
                }
                else if(capitalization.count(lower) > 0) {
                    ans.push_back(capitalization[lower].front());
                }
                else if(vowel.count(novowel) > 0) {
                    ans.push_back(vowel[novowel].front());
                }
                else {
                    ans.push_back("");
                }
            }
            return ans;
        }
    private:

        string tolower(string& str) {
            string ans;
            for(char in : str) {
                if(in <= 'Z' && in >= 'A')
                     in -= ('Z' - 'z');
                ans.push_back(in);
            }
            return ans;
        }
        string delvowel(string& str) {
            string ans(str);
            for(char& ch : ans) {
                if(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u')
                    ch = '*';
            }
            return ans;
        }

};
