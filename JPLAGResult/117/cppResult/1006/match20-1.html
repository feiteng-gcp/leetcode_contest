class Solution {
	string ToLower(string S)
	{
		int j;
		int N = S.size();
		string Result = S;
		
		for(j = 0; j < N; j++)
		{
			Result[j] += (Result[j] < 'a') ? ('a' - 'A') : 0;
		}
		
		return Result;
	}
	
	string ToLowerA(string S)
	{
		int j;
		int N = S.size();
		string Result = ToLower(S);
		
		for(j = 0; j < N; j++)
		{
			if((Result[j] == 'e') || (Result[j] == 'i') || (Result[j] == 'o') || (Result[j] == 'u'))
			{
				Result[j] = 'a';
			}
		}
		
		return Result;
	}
	
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        int i, j;
		int N = wordlist.size();
		int M = queries.size();
		vector<string> Result;
		unordered_set<string> Set;
		unordered_map<string, string> Map;
		unordered_map<string, string> Map1;
		

		for(i = 0; i < N; i++)
		{
			Set.insert(wordlist[i]);
			
			string Temp = ToLower(wordlist[i]);
			
			if(Map.count(Temp) == 0)
			{
				Map[Temp] = wordlist[i];
			}
			
			Temp = ToLowerA(wordlist[i]);
			
			if(Map1.count(Temp) == 0)
			{
				Map1[Temp] = wordlist[i];
			}
		}
		
		for(i = 0; i < M; i++)
		{

			if(Set.count(queries[i]))
			{
				Result.push_back(queries[i]);
			}
			else if(Map.count(ToLower(queries[i])))
			{
				Result.push_back(Map[ToLower(queries[i])]);
			}
			else if(Map1.count(ToLowerA(queries[i])))
			{
				Result.push_back(Map1[ToLowerA(queries[i])]);
			}
			else
			{
				Result.push_back("");
			}
		}
		
		return Result;
    }
};
