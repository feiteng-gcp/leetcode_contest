class Solution {
public:
    string to_little(string str){
        for(char &c:str){
            if(c>='A'&&c<='Z')
                c+=('a'-'A');
        }
        return str;
    }
    string to_yuan(string str){
        for(char &c:str){
            if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u')
                c='*';
        }
        return str;
    }
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        vector<string> ans;
        unordered_set<string> dict(wordlist.begin(),wordlist.end());
        unordered_map<string,vector<string>> mp;
        for(auto str:wordlist){
            string s=to_little(str);
            mp[s].push_back(str);
            string ss=to_yuan(s);
            mp[ss].push_back(str);
        }
        for(auto str:queries){
            if(dict.find(str)!=dict.end()) ans.push_back(str);
            else {
                string s=to_little(str);
                if(mp.find(s)!=mp.end()){
                    ans.push_back(mp[s][0]);
                    continue;
                }
                string ss=to_yuan(s);
                if(mp.find(ss)!=mp.end()){
                    ans.push_back(mp[ss][0]);
                }
                else ans.push_back("");
            }
        }
        return ans;
    }
};
