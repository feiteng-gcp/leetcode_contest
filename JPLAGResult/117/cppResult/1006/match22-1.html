class Solution {
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        unordered_set<string> map1;
        unordered_map<string, string> map2;
        unordered_map<string, string> map3;
        for (auto& s : wordlist) {
            map1.insert(s);
            string lower = s;
            transform(lower.begin(), lower.end(), lower.begin(),::tolower);
            if (map2.count(lower) == 0) {
                map2[lower] = s;
            }
            string t;

            for (auto c : lower) {
                if (c != 'a' && c != 'e' && c != 'i' && c != 'o' && c != 'u') {
                    t += c;
                }
                else {
                    t += '#';
                }
            }
            if (map3.count(t) == 0) {
                map3[t] = s;
            }
        }
        for (auto& s : queries) {
            if (map1.count(s) == 0) {
                string lower = s;
                transform(lower.begin(), lower.end(), lower.begin(),::tolower);
                if (map2.count(lower)) {
                    s = map2[lower];
                }
                else {
                    string t;

                    for (auto c : lower) {
                        if (c != 'a' && c != 'e' && c != 'i' && c != 'o' && c != 'u') {
                            t += c;
                        }
                        else {
                            t += '#';
                        }
                    }
                    if (map3.count(t)) {
                        s = map3[t];
                    }
                    else s = "";
                }
            }
        }
        return queries;
    }
};
