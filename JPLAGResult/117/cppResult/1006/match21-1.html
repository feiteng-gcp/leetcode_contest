class Solution {
        string tocap (string w) {
            string s = "";
            for (auto& c: w) {
                s += toupper(c);
            }
            return s;
        }
    string devow (string s) {
        s = tocap(s);
        string s1 = "";
            for (auto& c: s) {
                if (c == 'A' || 
                   c == 'E' || 
                   c == 'I' || 
                   c == 'O' || 
                   c == 'U') {
                    s1 += ".";
                } else {
                    s1 += c;
                }
            }
        return s1;
    }
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        set<string> truth;
        map<string, string> cap;
        map<string, string> vow; 

        for (auto &w : wordlist) {
            truth.insert(w);
            //cout << w << endl;
            string s = tocap(w);            //cout << s << endl;

            if (cap.find(s) == cap.end()) {
                
            cap[s] = w;
            }
            string s1 = devow(w);            //cout << s1 << endl;
            if (vow.find(s1) == vow.end()) {

                        vow[s1] = w;
                        }
        }
        vector<string> ret;
        for (auto &w : queries) {

            if (truth.find(w) != truth.end()) {
                ret.push_back(w);
            } else if (cap.find(tocap(w)) != cap.end()) {
                ret.push_back(cap[tocap(w)]);
            } else if (vow.find(devow(w)) != vow.end()) {
                ret.push_back(vow[devow(w)]);
            } else 
                ret.push_back("");
        }
        return ret;
    }
};
