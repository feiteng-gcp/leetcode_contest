class Solution {
  string capitalize(string word) {
    for (char& c : word) {
      if (c >= 'A' && c <= 'Z')
        c += 'a' - 'A';
    }
    return word;
  }
  
  string removevowel(string word) {
    for (char& c : word) {
      if (c == 'a'  || c == 'e' || c == 'i' || c == 'o' || c == 'u')
        c = '_';
    }
    return word;
  }
  
public:
  vector<string> spellchecker(vector<string>& wordList, vector<string>& queries) {
    unordered_set<string> uset(wordList.begin(), wordList.end());
    map<string, string> umap1;
    map<string, string> umap2;
    for (int i = wordList.size() - 1; i >= 0; i--) {
      string word = wordList[i];
      string capitalized = capitalize(word);
      string vowelremoved = removevowel(capitalized);
      umap1[capitalized] = word;
      umap2[vowelremoved] = word;
    }
    vector<string> ans;
    for (string& word : queries) {
      if (uset.count(word) == 1) {
        ans.push_back(word);
        continue;
      }
      string capitalized = capitalize(word);
      if (umap1.find(capitalized) != umap1.end()) {
        ans.push_back(umap1[capitalized]);
        continue;
      }
      string vowelremoved = removevowel(capitalized);
      if (umap2.find(vowelremoved) != umap2.end()) {
        ans.push_back(umap2[vowelremoved]);
        continue;
      }
      ans.push_back("");
    }
    return ans;
  }
};
