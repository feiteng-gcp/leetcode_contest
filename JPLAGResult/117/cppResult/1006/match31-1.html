class Solution {
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        unordered_set<string> st;
        unordered_map<string, vector<string>> capMap, volMap;
        for (auto str : wordlist) {
            st.insert(str);
            string cap = str;
            for (int i = 0; i < cap.length(); ++i) {
                if (cap[i] >= 'A' && cap[i] <= 'Z') {
                    cap[i] = cap[i] - 'A' + 'a';
                }
            }
            capMap[cap].push_back(str);
            string vol = cap;
            for (int i = 0; i < vol.length(); ++i) {
                if (vol[i] == 'e' || vol[i] == 'i' || vol[i] == 'o' || vol[i] == 'u') {
                    vol[i] = 'a';
                }
            }
            volMap[vol].push_back(str);
        }
        
        vector<string> ret;
        for (auto query : queries) {
            if (st.find(query) != st.end()) {
                ret.push_back(query);
                continue;
            }
            
            string cap = query;

            for (int i = 0; i < cap.length(); ++i) {
                if (cap[i] >= 'A' && cap[i] <= 'Z') {
                    cap[i] = cap[i] - 'A' + 'a';
                }
            }
            if (capMap.find(cap) != capMap.end()) {
                ret.push_back(capMap[cap][0]);
                continue;
            }

            string vol = cap;

            for (int i = 0; i < vol.length(); ++i) {
                if (vol[i] == 'e' || vol[i] == 'i' || vol[i] == 'o' || vol[i] == 'u') {
                    vol[i] = 'a';
                }
            }
            if (volMap.find(vol) != volMap.end()) {
                ret.push_back(volMap[vol][0]);
                continue;
            }
            ret.push_back("");
        }
        
        return ret;
    }
};
