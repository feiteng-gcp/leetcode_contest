class Solution {
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        unordered_set<string> WLset(wordlist.begin(), wordlist.end());

        for(int i=0;i<wordlist.size();i++){
            string word = wordlist[i];
            string lower_word = to_lower(word);
            WL[lower_word].push_back(i);
            vowWL[remove_vow(lower_word)].push_back(i);
        }
        vector<string> res;
        for(auto & Query: queries){
            if(WLset.find(Query)!=WLset.end())
                res.push_back(Query);
            else {
                string query = to_lower(Query);
                if(WL.find(query)!=WL.end()) {res.push_back(wordlist[WL[query][0]]); continue;}
                string query_no_vowel = remove_vow(query);
                if(vowWL.find(query_no_vowel)!=vowWL.end()) {res.push_back(wordlist[vowWL[query_no_vowel][0]]); continue;}
                res.push_back("");
            }
        }
        return res;
    }
private:
    
    unordered_map<string, vector<int>> WL;
    unordered_map<string, vector<int>> vowWL;
    unordered_map<string, bool> wl_exist;
    const unordered_set<char> vowels = {'a','e','i','o','u'};
    string to_lower(string STR){
        std::transform(STR.begin(), STR.end(), STR.begin(), ::tolower);
        return STR;
    }
    string remove_vow(string str){
        for(int i=0;i<str.length();i++)
        if(vowels.find(str[i])!=vowels.end()) str[i]='*';
        return str;
    }
    
};
