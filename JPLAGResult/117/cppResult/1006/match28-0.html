class Solution {
public:
    vector<string> spellchecker(vector<string>& w, vector<string>& q) {
        map<string,string>m1,m2,m3;
        for(int i=0;i<w.size();i++){
            if(m1.find(w[i])==m1.end())
                m1[w[i]]=w[i];
            string tmp="";
            for(int j=0;j<w[i].size();j++){
                if(w[i][j]>='A'&&w[i][j]<='Z'){
                    tmp+=(char)(w[i][j]-'A'+'a');
                }else{
                    tmp+=w[i][j];
                }
            }
            if(m2.find(tmp)==m2.end())
                m2[tmp]=w[i];
            string temp="";
            for(int j=0;j<tmp.size();j++){
                if(tmp[j]=='a'||tmp[j]=='e'||tmp[j]=='i'||tmp[j]=='o'||tmp[j]=='u')
                    temp+="*";
                else
                    temp+=tmp[j];
            }
            if(m3.find(temp)==m3.end())
                m3[temp]=w[i];
        }
        vector<string>ret;
        for(int i=0;i<q.size();i++){
            if(m1.find(q[i])!=m1.end()){
                ret.push_back(q[i]);
                continue;
            }
            string tmp="";
            for(int j=0;j<q[i].size();j++){
                if(q[i][j]>='A'&&q[i][j]<='Z'){
                    tmp+=(char)(q[i][j]-'A'+'a');
                }else
                    tmp+=q[i][j];
            }
            if(m2.find(tmp)!=m2.end()){
                ret.push_back(m2[tmp]);
                continue;
            }
            string temp="";
            for(int j=0;j<tmp.size();j++){
                if(tmp[j]=='a'||tmp[j]=='e'||tmp[j]=='i'||tmp[j]=='o'||tmp[j]=='u')
                    temp+="*";
                else
                    temp+=tmp[j];
            }
            if(m3.find(temp)!=m3.end()){
                ret.push_back(m3[temp]);
                continue;
            }
            string ans="";
            ret.push_back(ans);
        }
        return ret;
    }
};
