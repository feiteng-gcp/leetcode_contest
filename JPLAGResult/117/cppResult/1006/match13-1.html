class Solution {
public:
    
    string toLowerCase(string s) {
        for (int i=0;i<s.length();i++) s[i] = tolower(s[i]);
        return s;
    }
    
    string toVowelCase(string s) {
        for (int i=0;i<s.length();i++) {
            s[i] = tolower(s[i]);
            if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u') {
                s[i] = '*';
            }
        }
        return s;
    }
    
    vector<string> spellchecker(vector<string>& w, vector<string>& queries) {
        unordered_set<string> S;
        for (string x: w) S.insert(x);
        
        map<string, string> M;
        for (string x: w) {
            string lowerx = toLowerCase(x);
            if (M.find(lowerx) == M.end()) {
                M[lowerx] = x;
            }
        }
        
        map<string, string> V;
        for (string x: w) {
            string vowelx = toVowelCase(x);
            if (V.find(vowelx) == V.end()) {
                V[vowelx] = x;
            }

        }
        
        vector < string > ans;
        for (string q: queries) {
            if (S.find(q) != S.end()) {
                ans.push_back(q);
                continue;
            }
            string lowerq = toLowerCase(q);
            if (M.find(lowerq) != M.end()) {
                ans.push_back(M[lowerq]);
                continue;
            }
            
            string vowelq = toVowelCase(q);
            if (V.find(vowelq) != V.end()) {
                ans.push_back(V[vowelq]);
                continue;
            }
            
            ans.push_back("");
        }
        return ans;
    }
};
