class Solution {
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        unordered_set<string> word;
        for(int i = 0;i<wordlist.size();i++)
            word.insert(wordlist[i]);
        unordered_map<string,int> lword;
        for(int i = 0;i<wordlist.size();i++){
            string s = wordlist[i];
                transform(s.begin(), s.end(), s.begin(), ::tolower);
            if(lword.find(s)==lword.end()){
                lword[s] = i;
            }
        }
        unordered_map<string,int> cword;
        for(int i = 0;i<wordlist.size();i++){

            string s = to_string(wordlist[i].length());
            for(int j = 0;j<wordlist[i].length();j++){
                if(wordlist[i][j]=='a'||wordlist[i][j]=='e'||wordlist[i][j]=='i'||wordlist[i][j]=='o'||wordlist[i][j]=='u'||wordlist[i][j]=='A'||wordlist[i][j]=='E'||wordlist[i][j]=='I'||wordlist[i][j]=='O'||wordlist[i][j]=='U')
                    s = s+'0';
                else
                s+= tolower(wordlist[i][j]);
            }
            if(cword.find(s)==cword.end()){
                cword[s] = i;
            }
        }
        vector<string> v;
        for(int i = 0;i<queries.size();i++){
            if(word.find(queries[i])!=word.end()){
                v.push_back(queries[i]);
                continue;
            }
            string s = queries[i];
                transform(s.begin(), s.end(), s.begin(), ::tolower);

            if(lword.find(s)!=lword.end()){
                v.push_back(wordlist[lword[s]]);
                continue;
            }
            s = to_string(queries[i].length());
            for(int j = 0;j<queries[i].length();j++){
                if(queries[i][j]=='a'||queries[i][j]=='e'||queries[i][j]=='i'||queries[i][j]=='o'||queries[i][j]=='u'||queries[i][j]=='A'||queries[i][j]=='E'||queries[i][j]=='I'||queries[i][j]=='O'||queries[i][j]=='U')
                    s = s+'0';
                else
                s+= tolower(queries[i][j]);
            }
            if(cword.find(s)!=cword.end()){
                v.push_back(wordlist[cword[s]]);
                continue;
            }
            v.push_back("");
        }
        return v;
    }
};
