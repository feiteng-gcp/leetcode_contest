class Solution {
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        //我们用哈希来装和原来一样的单词
        unordered_set<string> stringSet;
        for (int index = 0; index < wordlist.size(); index ++) {
            stringSet.insert(wordlist[index]);
        }
        //我们倒序装大小写的
        unordered_map<string, string> bigSmallMap;
        unordered_set<string> bigSmallSet;
        for (int index = 0; index < wordlist.size(); index ++) {
            //单词转换为小写
            string toSamllString = wordlist[index];
            transform(toSamllString.begin(), toSamllString.end(), toSamllString.begin(), ::tolower);
            //如果已经转换过了，直接跳过
            if(bigSmallSet.count(toSamllString)) {
                continue;
            }
            //否则装进去
            bigSmallMap[toSamllString] = wordlist[index];
            cout<<toSamllString<<"->"<<wordlist[index]<<endl;
            bigSmallSet.insert(toSamllString);
        }
        cout<<"----"<<endl;
        //我们装元音字母替换的，把元音字母替换为a
        unordered_map<string, string> yuanYinMap;
        unordered_set<string> yuanYinSet;
        for (int index = 0; index < wordlist.size(); index ++) {
            //单词转换为小写
            string yuanyinString = wordlist[index];
            transform(yuanyinString.begin(), yuanyinString.end(), yuanyinString.begin(), ::tolower);
            //把元音字母替换为a

            for (int i = 0; i < yuanyinString.length(); i ++) {
                if(yuanyinString[i] == 'a' || yuanyinString[i] == 'e' || yuanyinString[i] == 'i' || yuanyinString[i] == 'o' || yuanyinString[i] == 'u') {
                    yuanyinString[i] = 'a';
                }
            }
            
            //如果已经转换过了，直接跳过
            if(yuanYinSet.count(yuanyinString)) {
                continue;
            }
            //否则装进去
            yuanYinMap[yuanyinString] = wordlist[index];
            yuanYinSet.insert(yuanyinString);
            cout<<yuanyinString<<"->"<<wordlist[index]<<endl;
        }
        cout<<"----"<<endl;
       //我们来组装答案
        vector<string> result;
        for (int index = 0; index < queries.size(); index ++) {
            string currStr = queries[index];
            if(stringSet.count(currStr)) {
                result.push_back(currStr);
                continue;
            }
            
            string toSamllString = currStr;
            transform(toSamllString.begin(), toSamllString.end(), toSamllString.begin(), ::tolower);
            if(bigSmallSet.count(toSamllString)) {
                result.push_back(bigSmallMap[toSamllString]);
                continue;
            }
            
            //单词转换为小写
            string yuanyinString = toSamllString;
            //把元音字母替换为a

            for (int i = 0; i < yuanyinString.length(); i ++) {
                if(yuanyinString[i] == 'a' || yuanyinString[i] == 'e' || yuanyinString[i] == 'i' || yuanyinString[i] == 'o' || yuanyinString[i] == 'u') {
                    yuanyinString[i] = 'a';
                }
            }
            if(yuanYinSet.count(yuanyinString)) {
                result.push_back(yuanYinMap[yuanyinString]);
                continue;
            }
            
            result.push_back("");
        }
        return result;
    }
};
