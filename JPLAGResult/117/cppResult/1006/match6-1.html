class Solution {
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        set<string> A(begin(wordlist), end(wordlist));
        set<string> B, C;
        map<string, int> bi, ci;
        for(int i=0; i<wordlist.size(); ++i) {
            auto s=wordlist[i];

            for(auto& c: s) c=tolower(c);
            B.insert(s);
            if (!bi.count(s)) bi[s]=i;
            for(auto& c: s) if (c=='a' or c=='e' or c=='i' or c=='o' or c=='u') c='?';
            C.insert(s);
            if (!ci.count(s)) ci[s]=i;
        }

        vector<string> res;
        for(auto& q: queries) {
            if (A.count(q)) {
                res.push_back(q);
                continue;
            }
            string s=q;
            for(auto& c: s) c=tolower(c);

            if (B.count(s)) {
                res.push_back(wordlist[bi[s]]);
                continue;
            }
            for(auto& c: s) if (c=='a' or c=='e' or c=='i' or c=='o' or c=='u') c='?';
            if (C.count(s)) {
                res.push_back(wordlist[ci[s]]);
                continue;
            }
            res.push_back("");
        }
        return res;
    }
};
