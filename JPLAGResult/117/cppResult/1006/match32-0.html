class Solution {
public:
    
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        set<string> exact;
        map<string, string> word_lo;
        map<string, string> word_vowel;
        
        set<char> vowels = {'a', 'e', 'i', 'o', 'u'};
        vector<string> ans;
        
        for( auto it = wordlist.begin(); it != wordlist.end(); it ++){
            string ori = *it;
            string lo = *it;
            exact.insert(ori);
            
            std::transform(lo.begin(), lo.end(), lo.begin(), ::tolower);
            if( word_lo.find(lo) == word_lo.end() ){
                word_lo[lo] = ori;
            }
            
            for( int i = 0; i< lo.size(); i++){
                if( vowels.find(lo[i]) != vowels.end() ){
                    lo[i] = 'a';
                }
            }
            
            if( word_vowel.find(lo) == word_vowel.end() ){
                word_vowel[lo] = ori;
            }
        }
        
        for( auto it = queries.begin(); it != queries.end(); it ++){
            string lo = *it;
            string match;
            std::transform(lo.begin(), lo.end(), lo.begin(), ::tolower);
            
            if( exact.find(*it) != exact.end() ){
                match = *it;
            }
            else if( word_lo.find(lo) != word_lo.end() ){
                match = word_lo[lo];
            }
            else{
                for( int i = 0; i< lo.size(); i++){
                    if( vowels.find(lo[i]) != vowels.end() ){
                        lo[i] = 'a';
                    }
                }
            
                if( word_vowel.find(lo) != word_vowel.end() ){
                    match = word_vowel[lo];
                }
                else match = "";
            }
            
            ans.push_back(match);
            
        }
        return ans;
    }
};
