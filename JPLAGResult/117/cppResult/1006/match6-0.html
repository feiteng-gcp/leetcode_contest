class Solution {
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        unordered_map<string, string> vst;
        unordered_map<string, string> spl;
        unordered_map<string, string> ane;
        for (auto &word : wordlist) {
            vst[word] = word;
            string x = word;
            for (auto &ch : x) if ('a' <= ch && ch <= 'z') ch = ch - 'a' + 'A';
          //  cout << x << "- " << word << endl;
            if (spl.find(x) == spl.end()) spl[x] = word;        
            for (auto &ch : x) if ('A' == ch || 'E' == ch || 'I' == ch || 'O' == ch || 'U' == ch) ch = '#';
            if (ane.find(x) == ane.end()) ane[x] = word;                      
        }
        vector<string> ans;
        for (auto &word : queries) {
            //cout << word << endl;
            if (vst.find(word) != vst.end()) {                      
                ans.push_back(vst[word]);
                continue;
            }
            string x = word;
            for (auto &ch : x) if ('a' <= ch && ch <= 'z') ch = ch - 'a' + 'A';
            //cout << word << " : " << x << endl;
            if (spl.find(x) != spl.end()) {
                ans.push_back(spl[x]);
                continue;
            }            
            for (auto &ch : x) if ('A' == ch || 'E' == ch || 'I' == ch || 'O' == ch || 'U' == ch) ch = '#';
            //cout << word << " : " << x << endl;
            if (ane.find(x) != ane.end()) {
                ans.push_back(ane[x]);                                
                continue;
            } 
            ans.push_back("");
        }
        return ans;
    }
};
