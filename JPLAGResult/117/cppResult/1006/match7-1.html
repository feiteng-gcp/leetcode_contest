class Solution {
public:
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        unordered_set<string>m;
        unordered_map<string,int>lower, modified;
        for(auto str: wordlist) m.insert(str);
        for(int i=0;i<wordlist.size();i++){
            string t=wordlist[i];
            transform(t.begin(), t.end(), t.begin(), ::tolower);
            // cout<<"string : "<<wordlist[i]<<" and lower: "<<t<<endl;
            if(lower.count(t)==0) lower[t]=i;
        }
        for(int i=0;i<wordlist.size();i++){
            string t=wordlist[i];
            transform(t.begin(), t.end(), t.begin(), ::tolower);

            string temp="";
            for(int j=0;j<t.length();j++){
                if(t[j]=='a' || t[j]=='e' || t[j]=='i' || t[j]=='o' || t[j]=='u') temp+="#";
                else temp+=t[j];
            }
            // cout<<"string: "<<wordlist[i]<<" and modified: "<<temp<<endl;
            if(modified.count(temp)==0) modified[temp]=i;
        }
        
        vector<string>res;

        for(auto str:queries){
            if(m.count(str)==1){
                res.push_back(str);
                continue;
            }
            string t=str;
            transform(t.begin(), t.end(), t.begin(), ::tolower);
            // cout<<"query. str: "<<str<<" and lower: "<<t<<endl;
            if(lower.count(t)==1){
                res.push_back(wordlist[lower[t]]);
                continue;
            }
            string temp="";
            string x=str;
            transform(x.begin(), x.end(), x.begin(), ::tolower);

            for(int j=0;j<x.length();j++){
                if(x[j]=='a' || x[j]=='e' || x[j]=='i' || x[j]=='o' || x[j]=='u') temp+="#";
                else temp+=x[j];
            }
            if(modified.count(temp)==1){
                res.push_back(wordlist[modified[temp]]);
                continue;
            }
            res.push_back("");
            
        }
        return res;
    }
};
