class Solution {
public:
    string cap(string&w) {
        string ret;
        for(auto c: w) {
            if(c >= 'A' && c <= 'Z') {
                c = c - 'A' + 'a';
            }
            ret.push_back(c);
        }
        //std::cout << w << " -> " << ret << std::endl;
        return ret;
    }
    string conv(string&w) {
        string ret;
        for(auto c: w) {
            if(c >= 'A' && c <= 'Z') {
                c = c - 'A' + 'a';
            }
            if(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
                c = 'a';
            }
            ret.push_back(c);
        }
        //std::cout << w << " -> " << ret << std::endl;
        return ret;
    }
    vector<string> spellchecker(vector<string>& wordlist, vector<string>& queries) {
        vector<string> ret;
        map<string, string> dcap;
        map<string, string> dict;
        set<string> appear;

        for(auto w: wordlist) {
            appear.insert(w);
            string wc = cap(w);
            if(dcap.find(wc) == dcap.end()) {
                dcap[wc] = w;
            }
            string w2 = conv(w);
            if(dict.find(w2) == dict.end()) {
                dict[w2] = w;
            }
        }
        for(auto q: queries) {

            string qc = cap(q);
            string q2 = conv(q);
            if(appear.find(q) != appear.end()) {
                ret.push_back(q);
            }
            else if(dcap.find(qc) != dcap.end()) {
                ret.push_back(dcap[qc]);
            }
            else if(dict.find(q2) != dict.end()) {
                ret.push_back(dict[q2]);
            }
            else {
                ret.push_back("");
            }
        }
        return ret;
    }
};
