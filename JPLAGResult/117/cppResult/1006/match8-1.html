typedef pair<int,int>P;

class Solution {
public:
	bool isvowel(char c){
		return c=='a'||c=='i'||c=='u'||c=='e'||c=='o';
	}
    vector<string> spellchecker(vector<string> wordlist, vector<string> queries) {
		map<string,string>mp1,mp2,mp3;
		for(auto s:wordlist){
			if(!mp1.count(s))mp1[s]=s;
			string s2;
			for(char c:s){
				s2+=tolower(c);
			}
			if(!mp2.count(s2))mp2[s2]=s;
			string s3;

			for(char c:s2){
				if(isvowel(c))s3+='a';
				else s3+=c;
			}
			if(!mp3.count(s3))mp3[s3]=s;
		}
		vector<string>ans;
		for(auto s:queries){
			if(mp1.count(s)){
				ans.push_back(mp1[s]);
				continue;
			}
			string s2;
			for(char c:s){
				s2+=tolower(c);
			}
			if(mp2.count(s2)){
				ans.push_back(mp2[s2]);
				continue;
			}
			string s3;

			for(char c:s2){
				if(isvowel(c))s3+='a';
				else s3+=c;
			}
			if(mp3.count(s3)){
				ans.push_back(mp3[s3]);
				continue;
			}
			ans.push_back("");
		}
		return ans;
    }
};
