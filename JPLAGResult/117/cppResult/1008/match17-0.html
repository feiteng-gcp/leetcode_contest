/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int minCameraCover(TreeNode* root) {
        auto res=getmin(root);
        return min(res[1],res[2]);
    }
    vector<int> getmin(TreeNode* p){
        if(p==NULL) return {0,0,2000};
        auto lc=getmin(p->left);
        auto lr=getmin(p->right);
        int t00=lc[0]+lr[0];
        int t11=lc[1]+lr[1];
        int t22=lc[2]+lr[2];
        int t01=min(lc[0]+lr[1],lc[1]+lr[0]);
        int t02=min(lc[0]+lr[2],lc[2]+lr[0]);
        int t12=min(lc[1]+lr[2],lc[2]+lr[1]);
        int x=t11;
        int y=min(t22,t12);
        int z=min(t00,min(t01,t02))+1;
        return {x,y,z};
    }
};
