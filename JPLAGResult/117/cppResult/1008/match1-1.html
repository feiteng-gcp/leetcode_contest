/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int minCameraCover(TreeNode* root) {
        if (NULL == root) return 0;
        TreeNode* parent = new TreeNode(0);
        int result = minCameraCover(root, parent);
        if (parent->val >= 3) result++;
        delete parent;
        return result;
    }
    
private:
    int minCameraCover(TreeNode* current, TreeNode* parent) {
        if (current == NULL) return 0;
        current->val = 0;
        int leftDegree;
        if (current->left == NULL) leftDegree = 0;
        else if (current->left->left == NULL && current->left->right == NULL) leftDegree = 1;
        else leftDegree = 2;
        int rightDegree;
        if (current->right == NULL) rightDegree = 0;
        else if (current->right->left == NULL && current->right->right == NULL) rightDegree = 1;
        else rightDegree = 2;
        

        if (leftDegree == 0 && rightDegree == 0)
        {
            parent->val += 3;
            return 0;
        }
        else if (leftDegree < 2 && rightDegree < 2)
        {
            parent->val++;
            return 1;
        }
        else if (leftDegree == 1 || rightDegree == 1)
        {
            int leftValue = leftDegree >= 2 ? minCameraCover(current->left, current) : 0;
            int rightValue = rightDegree >= 2 ? minCameraCover(current->right, current) : 0;
            parent->val++;
            return leftValue + rightValue + 1;
        }
        else 
        {
            int leftValue = minCameraCover(current->left, current);
            int rightValue = minCameraCover(current->right, current);
            if (current->val > 0)
            {
                if (current->val < 3)
                {
                    current->val = 0;
                    return leftValue + rightValue;
                }
                else
                {
                    current->val = 0;
                    parent->val += 1;
                    return leftValue + rightValue + 1;
                }
            }
            else
            {
                parent->val += 3;
                return leftValue + rightValue;
            }
        }
    }
};
