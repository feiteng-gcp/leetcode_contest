/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int ret;

    void dfs(TreeNode* cur) {
        if (cur->left) dfs(cur->left);
        if (cur->right) dfs(cur->right);

        if (!cur->left && !cur->right) {
            cur->val = -1;
        } else if (cur->left && !cur->right) {
            if (cur->left->val == -1) {
                cur->val = 1;
            } else if (cur->left->val == -2) {
                cur->val = -1;
            } else if (cur->left->val == 1) {
                cur->val = -2;
            }
        } else if (!cur->left) {
            if (cur->right->val == -1) {
                cur->val = 1;
            } else if (cur->right->val == -2) {
                cur->val = -1;
            } else if (cur->right->val == 1) {
                cur->val = -2;
            }
        } else {
            if ((cur->left->val == -1) || (cur->right->val == -1)) {
                cur->val = 1;
            } else if ((cur->left->val == 1) || (cur->right->val == 1)) {
                cur->val = -2;
            } else {
                cur->val = -1;
            }
        }
        if (cur->val == 1) ++ret;
    }
    // 1, -1, -2

public:
    int minCameraCover(TreeNode* root) {
        if (!root) return 0;
        ret = 0;
        dfs(root);
        if (root->val == -1) ++ret;
        return max(ret, 1);
    }
};
