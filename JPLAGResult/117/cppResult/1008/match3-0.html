/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
	int cnt;
	int L[10007], R[10007];
	int dp[10007][3];
	const int INF = 0x3f3f3f3f;
	int mi(int id) {
		return min(min(dp[id][0], dp[id][1]), dp[id][2]);
	}
    
	void solve(int cur) {
		int left = L[cur], right = R[cur];
		if (left && right) {
			solve(L[cur]);
			solve(R[cur]);

			dp[cur][0] = dp[left][2] + dp[right][2];

			dp[cur][1] = 1 + mi(left) + mi(right);

			dp[cur][2] = min( min(dp[left][1], dp[left][2]) + dp[right][1], dp[left][1] + min(dp[right][1], dp[right][2]));
			// dp[cur][2] = min(dp[cur][0], dp[left][1] + dp[right][1]);
		}
		else if (left && !right) {
			solve(L[cur]);
			dp[cur][0] = dp[left][2];
			dp[cur][1] = min(min(dp[left][0], dp[left][1]), dp[left][2]) + 1;
			dp[cur][2] = dp[left][1];
		}
		else if (!left && right) {
			solve(R[cur]);
			dp[cur][0] = dp[right][2];
			dp[cur][1] = min(min(dp[right][0], dp[right][1]), dp[right][2]) + 1;
			dp[cur][2] = dp[right][1];
		}
		else {
			dp[cur][0] = 0;
			dp[cur][1] = 1;
			dp[cur][2] = INF;
		}
	}

	int build(TreeNode* root) {
		int cur = ++cnt;
		if (root->left) L[cur] = build(root->left);
		if (root->right) R[cur] = build(root->right);
		return cur;
	}

	int minCameraCover(TreeNode* root) {
		memset(dp, 0x3f, sizeof(dp));
		build(root);
		solve(1);
		return min(dp[1][1], dp[1][2]);
	}
};
