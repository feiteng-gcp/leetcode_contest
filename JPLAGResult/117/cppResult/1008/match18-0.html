/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int tcnt,dp[1005][3];
    void dfs(TreeNode* x) {
        x->val=++tcnt;
        dp[x->val][0]=0;
        dp[x->val][1]=0;
        dp[x->val][2]=1;
        if (x->left==NULL && x->right==NULL) {
            dp[x->val][0]=1005;
            return;
        }
        if (x->left==NULL) {
            dfs(x->right);
            dp[x->val][0]+=dp[x->right->val][2];
            dp[x->val][1]+=min(dp[x->right->val][0],dp[x->right->val][2]);
            dp[x->val][2]+=min(dp[x->right->val][1],dp[x->right->val][2]);
            return;
        }
        if (x->right==NULL){
            dfs(x->left);
            dp[x->val][0]+=dp[x->left->val][2];
            dp[x->val][1]+=min(dp[x->left->val][0],dp[x->left->val][2]);
            dp[x->val][2]+=min(dp[x->left->val][1],dp[x->left->val][2]);
            return;
        }
        dfs(x->left);
        dfs(x->right);
        int tmp1=min(dp[x->left->val][0],dp[x->left->val][2]);
        int tmp2=min(dp[x->right->val][0],dp[x->right->val][2]);
        dp[x->val][0]+=min(tmp1+dp[x->right->val][2],dp[x->left->val][2]+tmp2);
        dp[x->val][1]+=tmp1+tmp2;
        dp[x->val][2]+=min(dp[x->left->val][1],dp[x->left->val][2])+min(dp[x->right->val][1],dp[x->right->val][2]);
    }
    int minCameraCover(TreeNode* root) {
        tcnt=0;
        dfs(root);
        return min(dp[1][0],dp[1][2]);
    }
};
