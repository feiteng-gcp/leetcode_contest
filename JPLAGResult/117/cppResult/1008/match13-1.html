/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    unordered_map<TreeNode*, int> hm, mm;
    int helper(TreeNode* root) {
        if (hm.find(root) != hm.end()) return hm[root];
        if (!root) return 0;
        int res = 1;
        if (root->left) res += min(helper(root->left), minCameraCover(root->left->left) + minCameraCover(root->left->right));
        if (root->right) res += min(helper(root->right), minCameraCover(root->right->left) + minCameraCover(root->right->right));
        hm[root] = res;
        return res;
    }
public:
    int minCameraCover(TreeNode* root) {
        if (mm.find(root) != mm.end()) return mm[root];
        if (!root) return 0;
        if (!root->left && !root->right) return 1;
        if (!root->left) return helper(root->right);
        if (!root->right) return helper(root->left);

        int a = helper(root);
        int b = helper(root->left) + minCameraCover(root->right);
        int c = helper(root->right) + minCameraCover(root->left);
        int res = min(a, min(b, c));
        mm[root] = res;
        return res;
    }
};
