class Solution {
public:
  using P = pair<int, int>;
  P dfs(TreeNode* v,int is_root){
    if(!v) return P(0,0);
    P vl=dfs(v->left,0);
    P vr=dfs(v->right,0);
    int type=0,cnt=vl.second+vr.second;    
    if(vl.first==0&&vr.first==0){
      type=3;
    }else if(vl.first==3||vr.first==3){
      type=1;
      cnt++;
    }else if(vl.first==1||vr.first==1){
      type=2;
    }else{
      type=3;
    }
    if(is_root&&type==3){
      type=1;
      cnt++;
    }    
    return P(type,cnt);
  }
  int minCameraCover(TreeNode* root) {
    return dfs(root,1).second;
  }
};
