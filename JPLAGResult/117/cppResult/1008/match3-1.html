/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int minCameraCover(TreeNode* root) {
        if(!root) return 0;
        int leftWithoutCam, leftWithCam, leftWithCover;
        minCoverHelper(root, leftWithoutCam, leftWithCam, leftWithCover);
        return leftWithCover;
    }
    
    void minCoverHelper(TreeNode* root, int & withoutRootCamera, int & withRootCamera, int & withRootCoverNoCamera){
        withoutRootCamera = 0;
        withRootCamera = 0;
        withRootCoverNoCamera = 0;
        if(!root) return;
        int leftWithoutCam, leftWithCam, leftWithCover;
        int rightWithoutCam, rightWithCam, rightWithCover;
        minCoverHelper(root->left, leftWithoutCam, leftWithCam, leftWithCover);
        minCoverHelper(root->right, rightWithoutCam, rightWithCam, rightWithCover);

        if(!root->left && !root->right){
            withoutRootCamera = 0;
            withRootCamera = 1;
            withRootCoverNoCamera = 1;
        } else if(!root->right){
            withoutRootCamera = leftWithCover;
            withRootCamera = 1 + leftWithoutCam;
            withRootCoverNoCamera = leftWithCam;
        } else if(!root->left){
            withoutRootCamera = rightWithCover;
            withRootCamera = 1 + rightWithoutCam;
            withRootCoverNoCamera = rightWithCam;
        } else {
            withRootCamera = 1 + leftWithoutCam + rightWithoutCam;
            withoutRootCamera = min(leftWithCover + rightWithCover, withRootCamera);
            withRootCoverNoCamera = min(min(withRootCamera, leftWithCam + rightWithCover), rightWithCam + leftWithCover);
        }
        //cout << "In here at " << root->val << endl;
        //cout << "without/with/cover: " << withoutRootCamera << "/" << withRootCamera << "/" << withRootCoverNoCamera << endl;
    }
    
    int min(int a, int b){
        return a < b ? a : b;
    }
    
    int max(int a, int b){
        return a > b ? a : b;
    }
};
