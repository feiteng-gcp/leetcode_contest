/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int cover(TreeNode* root, bool* covered, bool* must) {
        if (root->left == NULL && root->right == NULL) {
            *covered = false;
            *must = true;
            return 0;   
        }
        int sum = 0;
        bool b = false;
        bool b2 = false;
        bool c = false;
        bool m = false;
        if (root->left != NULL) {
            sum+=cover(root->left, &c, &m);
        }
        if (root->right != NULL) {
            sum+=cover(root->right, &b2, &b);
            m = m || b;
            c = c || b2;
        }
        if (m) {
            sum++;
            *must = false;
            *covered = true;
        } else if (c) {
            *must = false;
            *covered = false;
        } else {
            *must = true;
            *covered = false;
        }
        return sum;
    }
    int minCameraCover(TreeNode* root) {
        if (root == NULL) return 0;
        if (root->left == NULL && root->right == NULL) return 1;
        bool b = false;

        bool must = false;
        bool covered = false;
        bool b2 = false;
        int sum = 0;
        if (root->left != NULL) sum+=cover(root->left, &covered, &must);
        if (root->right != NULL) sum+=cover(root->right, &b2, &b);
        must = must || b;
        covered = covered || b2;
        if (must || !covered) sum++;
        return sum;
    }
};
