/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int minCameraCover(TreeNode* root) {
        queue<TreeNode*> leaves;
        unordered_map<TreeNode*, TreeNode*> parent;
        unordered_map<TreeNode*, int> outs;
        int res = 0;
        parent[nullptr] = nullptr;
        traverse(root, leaves, parent, outs, nullptr);
        while(leaves.size()){
            TreeNode* f = leaves.front();
            leaves.pop();
            TreeNode* p = parent[f];
            if(f->val == 0){
                ++res;

                if(p){
                    p->val = 1;
                    if(p->left){
                        p->left->val = 1;
                    }
                    if(p->right){
                        p->right->val = 1;
                    }
                    if(parent[p]){
                        parent[p]->val = 1;
                    }
                }
                else{
                    f->val = 1;
                }
            }
            if(p){
                --outs[p];
                if(outs[p] == 0){
                    leaves.push(p);
                }
            }
        }
        return res;
    }
    void traverse(TreeNode* root, queue<TreeNode*> &leaves, unordered_map<TreeNode*, TreeNode*> &parent, unordered_map<TreeNode*, int> &outs, TreeNode* p){
        if(!root){
            return;
        }
        int od = 0;
        parent[root] = p;
        if(root->left){
            ++od;
        }
        if(root->right){
            ++od;
        }
        outs[root] = od;
        if(od == 0){
            leaves.push(root);
        }
        else{
            traverse(root->left, leaves, parent, outs, root);
            traverse(root->right, leaves, parent, outs, root);
        }
    }
};
