/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    map<TreeNode*, int> dp[4];
    
    int go(TreeNode* t, int parentHas, int should) {
        if (dp[parentHas*2+should].count(t)) return dp[parentHas*2+should][t];
        
        int res = 1<<30;
        if (!t) return 0;
        
        for (int p = should; p < 2; ++p) {
            for (int l = 0; l < 2; ++l) {
                for (int r = 0; r < 2; ++r) {
                    if (l && !t->left) continue;
                    if (r && !t->right) continue;
                    if (parentHas || p || l || r) {
                        int v = go(t->left, p, l) + go(t->right, p, r) + p;
                        if (v < res) {
                            res = v;
                        }
                    }
                }
            }
        }
        dp[parentHas*2+should][t] = res;
        return res;
    }
    
    int minCameraCover(TreeNode* root) {
        for (int i = 0; i < 4; ++i) dp[i].clear();
        return go(root,0,0);
    }
};
