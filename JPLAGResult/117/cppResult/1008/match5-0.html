/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    map<TreeNode*, int> dp;
public:
    int getNodeOrTwoKids(TreeNode* node) {
        return min(dfs(node), dfs(node->left)+dfs(node->right));
    }
   int dfs(TreeNode* root) {
       if (!root)return 0;
       auto it=dp.find(root);
       if (it!=dp.end())return it->second;
       int bests=INT_MAX, bestl=INT_MAX,bestr=INT_MAX;
       {
           bests=1;
           if (root->left) {
               bests+=getNodeOrTwoKids(root->left);
           }
           if (root->right) {
               bests+=getNodeOrTwoKids(root->right);
           }
       }
       if (root->left) {
           bestl=1;
           if (root->left->left) {
               bestl+=getNodeOrTwoKids(root->left->left);
           }
           if (root->left->right) {
               bestl+=getNodeOrTwoKids(root->left->right);
           }
           if (root->right) {
               bestl+=dfs(root->right);
           }
       }
       if (root->right) {
           bestr=1;
           if (root->right->left) {
               bestr+=getNodeOrTwoKids(root->right->left);
           }
           if (root->right->right) {
               bestr+=getNodeOrTwoKids(root->right->right);
           }
           if (root->left) {
               bestr+=dfs(root->left);
           }
       }
       return dp[root]=min(min(bests, bestl), bestr);


    }
    int minCameraCover(TreeNode* root) {
        if (!root)return 0;
        dp.clear();
        return dfs(root);
    }
};
