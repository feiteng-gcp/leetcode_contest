/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int minCameraCover(TreeNode* root) {
        inc_table[nullptr] = 0;
        exc1_table[nullptr] = 0;
        exc2_table[nullptr] = 0;
        
        return min(inc(root), exc1(root));
    }
private:
    unordered_map<TreeNode*, int> inc_table;
    unordered_map<TreeNode*, int> exc1_table;
    unordered_map<TreeNode*, int> exc2_table;
    
    int inc(TreeNode* node) {
        if (inc_table.count(node) > 0) {
            return inc_table[node];
        }
        if (node != nullptr) {
            inc_table[node] = 1 + min(inc(node->left), exc2(node->left)) + min(inc(node->right), exc2(node->right));
            return inc_table[node];
        }
        else {
            inc_table[node] = 0;
            return inc_table[node];
        }
    }
    
    int exc1(TreeNode* node) {
        if (exc1_table.count(node) > 0) {
            return exc1_table[node];
        }
        if (node != nullptr) {
            if (node->left == nullptr && node->right == nullptr) {
                exc1_table[node] = 1;
                return exc1_table[node];
            }
            else if (node->right == nullptr) {
                exc1_table[node] = inc(node->left);
                return exc1_table[node];
            }
            else if (node->left == nullptr) {
                exc1_table[node] = inc(node->right);
                return exc1_table[node];
            }
            else {
                exc1_table[node] = min(inc(node->left) + min(inc(node->right), exc1(node->right)), inc(node->right) + min(inc(node->left), exc1(node->left)));
                return exc1_table[node];
            }
        }
        else {
            exc1_table[node] = 0;
            exc1_table[node] = 0;
        }
    }
    
    int exc2(TreeNode* node) {
        if (exc2_table.count(node) > 0) {
            return exc2_table[node];
        }
        if (node != nullptr) {
            exc2_table[node] = min(inc(node->right), exc1(node->right)) + min(inc(node->left), exc1(node->left));
            return exc2_table[node];
        }
        else {
            exc2_table[node] = 0;
            return exc2_table[node];
        }
    }
};
