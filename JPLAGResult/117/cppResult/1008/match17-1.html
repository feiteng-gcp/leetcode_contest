/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void helper(TreeNode* root, int& bySelf, int& byParent, int& byChild) {
        bySelf = 1;
        byParent = 0;

        byChild = 0;
        
        int lBySelf = 0;
        int lByParent = 0;
        int lByChild = 0;
        int rBySelf = 0;
        int rByParent = 0;
        int rByChild = 0;
        if (root == NULL) return;
        TreeNode* l = root->left;
        TreeNode* r = root->right;
        if (l == NULL && r == NULL) {
            byChild = 1;
            return;
        }
        if (l != NULL) {
            helper(l, lBySelf, lByParent, lByChild);
        }
        if (r != NULL) {
            helper(r, rBySelf, rByParent, rByChild);
        }
        if (l != NULL && r != NULL) {
            bySelf += min(lBySelf, min(lByParent, lByChild));
            bySelf += min(rBySelf, min(rByParent, rByChild));
            byParent += min(lBySelf, lByChild);
            byParent += min(rBySelf, rByChild);
            // use left
            int useLeft = lBySelf;
            useLeft += min(rBySelf, rByChild);
            // use right
            int useRight = rBySelf;
            useRight += min(lBySelf, lByChild);
            byChild = min(useLeft, useRight);
            return;
        }
        if (l != NULL) {
            bySelf += min(lBySelf, min(lByParent, lByChild));
            byParent = min(lBySelf, lByChild);
            byChild = lBySelf;
        }
        if (r != NULL) {
            bySelf += min(rBySelf, min(rByParent, rByChild));
            byParent += min(rBySelf, rByChild);
            byChild = rBySelf;
        }
    }
    int minCameraCover(TreeNode* root) {
        int bySelf = 0;
        int byParent = 0;
        int byChild = 0;
        helper(root, bySelf, byParent, byChild);
        return min(bySelf, byChild);
    }
};
