#include<bits/stdc++.h>
using namespace std;

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
  int left[1001], right[1001],par[1001];
  int level[1001];
  bool colored[1001];
  int cnt;
  vector<int> leaves;

  int dfs(TreeNode* cur) {
    int rn=-1, ln=-1;
    int curn = cnt;
    
    if (cur->right !=NULL) {
      cnt ++;
      rn = dfs(cur->right);
      par[rn] = curn;
    }
    if (cur->left !=NULL) {
      cnt ++;
      ln = dfs(cur->left);
      par[ln] = curn;
    }

    right[curn] = rn;
    left[curn] = ln;

    if (cur->left == NULL && cur->right == NULL) {
      leaves.push_back(curn);
    }
    return curn;
  }

  void bfs() {
    queue<int> q;
    q.push(right[0]);
    q.push(left[0]);
    
    while(!q.empty()) {
      
      int here = q.front();
      q.pop();
      if (here == -1) continue;
      level[here] = level[par[here]] + 1;
      q.push(right[here]);
      q.push(left[here]);
    }
  }

  bool shouldBeColored(int x) {
    int rt = right[x];
    int lt = left[x];
    int pr = par[x];

    if (rt != -1 && colored[rt]) return false;
    if (lt != -1 && colored[lt]) return false;
    if (x != 0 && colored[pr]) return false;
    if (colored[x]) return false;
    return true;
  }
  
  int solve() {
    priority_queue<pair<int,int> > pq;
    for(int i=0;i<leaves.size();i++) {
      int xx = leaves[i];
      pq.push(make_pair(level[xx], xx));
    }

    while(!pq.empty()) {
      int cr = pq.top().second;
      pq.pop();
      if (shouldBeColored(cr)) colored[par[cr]] = true;

      if (cr != 0) {
        int nxt = par[cr];
        pq.push(make_pair(level[nxt], nxt));
      }
    }

    int ret = 0;

    for(int i=0;i<=cnt;i++) if (colored[i]) ret ++;
    // for(int i=0;i<=cnt;i++) cout << i << ' ' << colored[i] << endl;
    return ret;
  }
  
  int minCameraCover(TreeNode* root) {
    cnt = 0;
    leaves.clear();
    dfs(root);
    bfs();

    
    // for(int i=0;i<=cnt;i++) {
    //   cout << par[i] << endl;
    // }
    // cout << endl;


    return solve();
  }
};
