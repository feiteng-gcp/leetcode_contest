/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int helper(TreeNode* root){
        if(!root) return 0;
        int num=0;
        if(root->left) num+=helper(root->left);
        if(root->right) num+=helper(root->right);
        if((root->left&&root->left->val==0)
           ||(root->right&&root->right->val==0)){
            root->val=2;
            num++;
        }
        else if((root->left&&root->left->val==2)
           ||(root->right&&root->right->val==2)){
            root->val=1;
        }
        return num;
    }
    int minCameraCover(TreeNode* root) {
        int num=helper(root);
        if(root->val==0) return num+1;
        return num;
    }
};
