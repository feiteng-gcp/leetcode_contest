/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int count(TreeNode* now, int&ans) {
        if(now == NULL) return 0;
        int vr, vl;
        vr = count(now->left, ans);
        vl = count(now->right, ans);
        if(vr == 1 || vl == 1) {
            ans++;
            return 2;
        }
        if(vr == 2 || vl == 2) {
            return 0;
        }
        return 1;
    }
    int minCameraCover(TreeNode* root) {
        int ans = 0, rt;
        rt = count(root, ans);
        if(rt == 1) ans++;
        return ans;
    }
};
