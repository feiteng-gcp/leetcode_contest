/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

const int INF = 0x3f3f3f3f;

class Solution {
public:
    vector<int> dp(TreeNode* root) {
        if (root == nullptr) return vector<int>(4, INF);
        vector<int> l = dp(root->left), r = dp(root->right);
        vector<int> ret(4, INF);
        // 0 -> 0 0
        if (root->left != nullptr && root->right != nullptr) {
            ret[0] = min(l[2] + min(r[0], r[2]), r[2] + min(l[0], l[2]));
        } else if (root->left != nullptr) {
            ret[0] = l[2];
        } else if (root->right != nullptr) {
            ret[0] = r[2];
        }
        
        // 1 -> 0 1
        if (root->left != nullptr && root->right != nullptr) {
            ret[1] = min(l[0], l[2]) + min(r[0], r[2]);
        } else if (root->left != nullptr) {
            ret[1] = min(l[0], l[2]);
        } else if (root->right != nullptr) {
            ret[1] = min(r[0], r[2]);
        } else ret[1] = 0;
        
        // 2 -> 1 0
        if (root->left != nullptr && root->right != nullptr) {
            ret[2] = min(l[1], l[3]) + min(r[1], r[3]) + 1;
        } else if (root->left != nullptr) {
            ret[2] = min(l[1], l[3]) + 1;
        } else if (root->right != nullptr) {
            ret[2] = min(r[1], r[3]) + 1;
        } else ret[2] = 1;
        
        // 3 -> 1 1
        if (root->left != nullptr && root->right != nullptr) {
            ret[3] = min(l[1], l[3]) + min(r[1], r[3]) + 1;
        } else if (root->left != nullptr) {
            ret[3] = min(l[1], l[3]) + 1;
        } else if (root->right != nullptr) {
            ret[3] = min(r[1], r[3]) + 1;
        } else ret[3] = 1;
        return ret;
    }
    
    int minCameraCover(TreeNode* root) {
        vector<int> ret = dp(root);
        return min(ret[0], ret[2]);
    }
};
