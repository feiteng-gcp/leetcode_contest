/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
const int INF = 1 << 29;
class Solution {
public:
    map<pair<TreeNode*, int>, int> dp;
    int solve(TreeNode* p, int k) {
        if (!p) return 0;
        int put = k & 1, cover = k & 2;
        if (dp.count(make_pair(p, k))) return dp[{p, k}];
        int ret = INF;
        if (!put) {
            if (!cover) {
                if (p->left) {
                    ret = min(ret, solve(p->left, 3) + solve(p->right, 0));
                }
                if (p->right) {
                    ret = min(ret, solve(p->left, 0) + solve(p->right, 3));
                }
                if (p->left && p->right) {
                    ret = min(ret, solve(p->left, 3) + solve(p->right, 3));
                }

            } else {
                ret = min(ret, solve(p->left, 0) + solve(p->right, 0));
                if (p->left) {
                    ret = min(ret, solve(p->left, 3) + solve(p->right, 0));
                }
                if (p->right) {
                    ret = min(ret, solve(p->left, 0) + solve(p->right, 3));
                }
                if (p->left && p->right) {
                    ret = min(ret, solve(p->left, 3) + solve(p->right, 3));
                }
            }

        } else {
            ret = min(ret, solve(p->left, 2) + solve(p->right, 2));
            if (p->left) {
                ret = min(ret, solve(p->left, 3) + solve(p->right, 2));
            }
            if (p->right) {
                ret = min(ret, solve(p->left, 2) + solve(p->right, 3));
            }
            if (p->left && p->right) {
                ret = min(ret, solve(p->left, 3) + solve(p->right, 3));
            }
            ret += 1;
        }
        //cout << p->val << " " << k << " " << ret << endl;
        dp[{p, k}] = ret;
        return ret;
    }
    int minCameraCover(TreeNode* root) {
        dp.clear();
        int ret = solve(root, 0);
        ret = min(ret, solve(root, 1));
        return ret;
    }
};
