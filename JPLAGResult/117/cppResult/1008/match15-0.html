typedef long long llong;
typedef vector<int> VI;
typedef vector<VI> VVI;

#define UNCOVERED 0
#define COVERED 1
#define SET 2

const int INF = 10000000;
class Solution {
    int id;
    void number(TreeNode* cur) {
        if (cur == nullptr) return;
        cur->val = id++;
        number(cur->left);
        number(cur->right);
    }

    int getId(TreeNode* x) {
        return x == nullptr ? -1 : x->val;
    }

    VVI memo;
    int go(TreeNode* cur, int node_id, int parent_state) {
        if (cur == nullptr)
            return parent_state != UNCOVERED ? 0 : INF;

        int& res = memo[parent_state][node_id];

        if (res < 0) {
            res = INF;
            if (parent_state != UNCOVERED) {
                // do not place camera
                if (parent_state == SET) {
                    int cur0 = go(cur->left, getId(cur->left), COVERED) +
                               go(cur->right, getId(cur->right), COVERED);
                    res = min(res, cur0);
                }
                else {
                    // parent is COVERED
                    int cur1 = go(cur->left, getId(cur->left), UNCOVERED) +
                               go(cur->right, getId(cur->right), COVERED);
                    res = min(res, cur1);
                    
                    int cur2 = go(cur->left, getId(cur->left), COVERED) +
                               go(cur->right, getId(cur->right), UNCOVERED);
                    res = min(res, cur2);
                }
            }
            // place camera
            {
                int cur3 = go(cur->left, getId(cur->left), SET) +
                           go(cur->right, getId(cur->right), SET) + 1;
                res = min(res, cur3);
            }
        }

        return res;
    }

public:
    int minCameraCover(TreeNode* root) {
        id = 0;
        number(root);

        memo = VVI(3, VI(id+2, -1));

        int res = go(root, root->val, COVERED);
        return res;
    }
};
