/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

const int inf = 1000010000;

int mem[1024][2][2];

class Solution {
    int id;
    
    void mark(TreeNode* r) {
        if (r == nullptr) return;
        r->val = id++;
        mark(r->left);
        mark(r->right);

    }
    
    int go(TreeNode* r, int me, int up) {
        if (r == nullptr) return me ? inf : 0;
        int& ret = mem[r->val][me][up];
        if (ret < 0) {
            ret = inf;
            for (int i = 0; i < 2; ++i) for (int j = 0; j < 2; ++j) {
                if (me || up || i || j) {
                    ret = min(ret, me + go(r->left, i, me) + go(r->right, j, me));
                }
            }
            // cout << r->val << " " << me << " " << up << " " << ret << endl;
        }
        return ret;
    }
    
public:
    int minCameraCover(TreeNode* root) {
        id = 0;
        mark(root);
        memset(mem, 0xff, sizeof(mem));
        return min(go(root, 0, 0), go(root, 1, 0));
    }
};
