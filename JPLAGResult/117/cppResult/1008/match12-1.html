/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
 private:
  map<TreeNode*, array<int, 2>> dp;

  array<int, 2> do_dp(TreeNode* root) {
    if (dp.find(root) != dp.end()) {
      return dp[root];
    }
    dp[root][0] = 1;
    if (root->left) {
      int gch = 0;
      if (root->left->left) {
        gch += do_dp(root->left->left)[1];
      }
      if (root->left->right) {
        gch += do_dp(root->left->right)[1];
      }
      dp[root][0] += min(gch, do_dp(root->left)[1]);
    }
    if (root->right) {
      int gch = 0;
      if (root->right->left) {
        gch += do_dp(root->right->left)[1];
      }
      if (root->right->right) {
        gch += do_dp(root->right->right)[1];
      }
      dp[root][0] += min(gch, do_dp(root->right)[1]);
    }

    if (root->left && root->right) {
      dp[root][1] = min(do_dp(root->left)[0] + do_dp(root->right)[1],
                        do_dp(root->left)[1] + do_dp(root->right)[0]);
    } else if (root->left) {
      dp[root][1] = do_dp(root->left)[0];
    } else if (root->right) {
      dp[root][1] = do_dp(root->right)[0];
    } else {
      dp[root][1] = 1;
    }
    dp[root][1] = min(dp[root][0], dp[root][1]);
    return dp[root];
  }

 public:
  int minCameraCover(TreeNode* root) {
    return do_dp(root)[1];
  }
};
