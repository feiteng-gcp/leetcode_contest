/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<TreeNode*,map<int,int>> maps;
    int dfs(TreeNode* root, int k)
    {
        if (maps[root][k]) return maps[root][k];
        if (!root) if (k==0) return INT_MAX/3; else return 0;
        if (k==0) {
            maps[root][k]=dfs(root->left,2)+dfs(root->right,2)+1;
            return maps[root][k];
        }
        if (k==1) {
            int p=dfs(root->left,0)+dfs(root->right,0);
            int q=dfs(root->left,0)+dfs(root->right,1);
            int r=dfs(root->left,1)+dfs(root->right,0);
            maps[root][k]=min(p,min(q,r));
            return maps[root][k];
        }
        if (k==2) {
            int p=dfs(root->left,0)+dfs(root->right,0);
            int q=dfs(root->left,0)+dfs(root->right,1);
            int r=dfs(root->left,1)+dfs(root->right,0);
            int s=dfs(root->left,1)+dfs(root->right,1);
            int t=dfs(root->left,2)+dfs(root->right,2)+1;
            maps[root][k]=min(t,min(min(p,s),min(q,r)));
            return maps[root][k];
        }
     }
    int minCameraCover(TreeNode* root) {
        return min(dfs(root,0),dfs(root,1));
    }
};
