/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<pair<TreeNode *, int>, int> dp;
    int f(TreeNode *node, int st) {
        pair<TreeNode *, int> pr(node, st);
        if (dp.find(pr) != dp.end())
            return dp[pr];
        int ans, tmp;
        ans = 1;
        if (node->left)
            ans += f(node->left, 2);
        if (node->right)
            ans += f(node->right, 2);
        if (st >= 1) {
            if (node->left && node->right) {
                tmp = min(f(node->left, 0) + f(node->right, 1), f(node->left, 1) + f(node->right, 0));

                ans = min(ans, tmp);
            } else if (node->left) {
                tmp = f(node->left, 0);
                ans = min(ans, tmp);
            } else if (node->right) {
                tmp = f(node->right, 0);
                ans = min(ans, tmp);
            }
        }
        if (st == 2) {
            if (node->left && node->right) {
                tmp = f(node->left, 1) + f(node->right, 1);
                ans = min(ans, tmp);
            } else if (node->left) {
                tmp = f(node->left, 1);
                ans = min(ans, tmp);
            } else if (node->right) {
                tmp = f(node->right, 1);
                ans = min(ans, tmp);
            } else
                ans = 0;
        }
        dp[pr] = ans;
        return ans;
    }
    int minCameraCover(TreeNode* root) {
        if (root == NULL)
            return 0;
        return f(root, 1);
    }
};
