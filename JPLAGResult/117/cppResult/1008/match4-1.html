/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
int dfs(TreeNode* root, int &total)
{
	if (!root)
	{
		return 4;
	}
	if (!root->left && !root->right)
	{
		return 3;
	}
	int nLeft = dfs(root->left, total);

	int nRight = dfs(root->right, total);
	if (nLeft == 3 || nRight == 3)
	{
		total++;
		return 1;
	}
	if (nLeft == 1 || nRight == 1)
	{
		return 2;
	}
	return 3;
}

int minCameraCover(TreeNode* root) {
	int total = 0;
	int result = dfs(root, total);
	if (result == 3)
	{
		total++;
	}
	if (total == 0 && root)
	{
		total = 1;
	}
	return total;
}

};
