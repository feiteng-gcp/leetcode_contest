/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int minCameraCover(TreeNode* root) {
        f.clear();
        g.clear();
        z.clear();
        
        dp(root);
        return min(f[root], g[root]);
    }
    
private:
    map<TreeNode*, int> f, g, z;
    
    int dp(TreeNode* t) {
        if (t == NULL) return 0;
        
        TreeNode* l = t->left;
        TreeNode* r = t->right;
        int tot = 1;
        tot += dp(l);
        tot += dp(r);
        
        //put in t
        int sonV = 0;
        if (l != NULL) sonV += min(min(f[l], g[l]), z[l]);
        if (r != NULL) sonV += min(min(f[r], g[r]), z[r]);
        f[t] = sonV + 1;
        
        //dont put, but t covered
        g[t] = tot;
        sonV = 0;
        if (l != NULL) {
            sonV = f[l];
            if (r != NULL) sonV += min(f[r], g[r]);
            g[t] = min(g[t], sonV);
        }
        sonV = 0;
        if (r != NULL) {
            sonV = f[r];
            if (l != NULL) sonV += min(f[l], g[l]);
            g[t] = min(g[t], sonV);
        }
        
        //dont put, t not covered
        sonV = 0;
        if (l != NULL) sonV += min(f[l], g[l]);
        if (r != NULL) sonV += min(f[r], g[r]);
        z[t] = sonV;
        
        return tot;
    }
};
