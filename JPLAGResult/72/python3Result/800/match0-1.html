class Solution:
    def letterCasePermutation(self, S):
        """
        :type S: str
        :rtype: List[str]
        """
        ans = []
        if S == '':
            return ['']
        for letter in S:
            if ans == []:
                if letter.isalpha():
                    ans.append(letter.lower())
                    ans.append(letter.upper())
                else:
                    ans.append(letter)
                continue
            if letter.isalpha():
                ans = ans + ans
                for idx in range(0, len(ans) // 2):
                    ans[idx] += letter.lower()
                    ans[idx + len(ans) // 2] 
+=letter.upper()
            else:
                for idx in range(len(ans)):
                    ans[idx] += letter
        return ans
