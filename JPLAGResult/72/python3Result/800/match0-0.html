class Solution:
    def letterCasePermutation(self, S):
        """
        :type S: str
        :rtype: List[str]
        """
        res = [""]
        for c in S:
            if not c.isdigit():
                res = sorted(res + res)
                for i in range(0, len(res), 2):
                    res[i] += c.lower()
                    res[i+1] += c.upper()
            else:
                for i in range(len(res)):
                    res[i] += c
        
        return res
