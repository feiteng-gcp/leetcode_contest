class Solution {
    List<String> result;
    public List<String> letterCasePermutation(String S) {
        result=new ArrayList<>();
        //recursion
        helper(S,0,new StringBuilder());
        return result;
    }
    public void helper(String S,int index, StringBuilder sb)
    {
        if (index==S.length()) 
        {
            result.add(sb.toString());
            return;
        }
        if (S.charAt(index)<'0' || S.charAt(index)>'9')
        {
            char lower=Character.toLowerCase(S.charAt(index)),upper=Character.toUpperCase(S.charAt(index));
            StringBuilder upSB=new StringBuilder(sb).append(upper);
            StringBuilder lowSB=new StringBuilder(sb).append(lower);
            helper(S,index+1,upSB);
            helper(S,index+1,lowSB);
        }else
        {
            sb.append(S.charAt(index));
            helper(S,index+1,sb);
        }
    }
}
