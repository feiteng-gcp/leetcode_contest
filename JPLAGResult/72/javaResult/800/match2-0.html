class Solution {
    public List<String> letterCasePermutation(String S) {
		List<String> aList = new ArrayList<>();

		char[] arr = S.toCharArray();
		helper(arr, aList, 0);
		return aList;

	}

	public void helper(char[] arr, List<String> aList, int k) {

		if (k == arr.length) {
			aList.add(new String(arr));
			return;
		}

		if (Character.isAlphabetic(arr[k])) {
			arr[k] = Character.toLowerCase(arr[k]);
			helper(arr, aList, k + 1);
			arr[k] = Character.toUpperCase(arr[k]);
			helper(arr, aList, k + 1);
		}else{
			helper(arr, aList, k + 1);
		}

	}
}
