import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {

    
void dfs(String s, int t, String current, List<String> res) {
        if (t >= s.length()) {
            res.add(current);
        } else {
            
char ch = s.charAt(t);
            if (Character.isAlphabetic(ch)) {
                dfs(s, t + 1, current + Character.toLowerCase(ch), res);
                dfs(s, t + 1, current + Character.toUpperCase(ch), res);

            } else {
                dfs(s, t + 1, current + ch, res);
            }
        }
    }

    public List<String> letterCasePermutation(String S) {
        List<String> res = new ArrayList<>();
        if (S.isEmpty()) {
            res.add(S);
        } else {
            dfs(S, 0, "", res);

        }
        return res;
    }

    public static void main(String[] args) {
        Solution sol = new Solution();


    }
}
