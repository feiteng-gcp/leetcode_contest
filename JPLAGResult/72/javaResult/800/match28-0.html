class Solution {
    public char a = 'a';
    public char z = 'z';
    public char A = 'A';
    public char Z = 'Z';
    public List<String> letterCasePermutation(String S) {
        List<String> re = new ArrayList<String>();
        if(S == null) {
            return re;
        }
        if( S.length() == 0) {
            re.add(S);
            return re;
        }
        char[] c = new char[S.length()];
        process(0, S, re, c);
        
        return re;
    }
    
    private void process(int i, String s,List<String> re, char[] c) {
        char cur = s.charAt(i);
        if(i == s.length() -1) {
            if( a<=cur && z >= cur ) {
                c[i] = cur;
                re.add(new String(c));
                c[i] -=32;
                re.add(new String(c));
            }
            else if(A<=cur && Z >= cur) {
                c[i] = cur;
                re.add(new String(c));
                c[i] +=32;
                re.add(new String(c));
            }
            else {
                c[i] = cur;
                re.add(new String(c));
            }
        }
        else {
            if( a<=cur && z >= cur ) {
                c[i] = cur;
                process(i+1, s, re, c);
                c[i] -=32;
                process(i+1, s, re, c);
            }
            else if(A<=cur && Z >= cur) {
                c[i] = cur;
                process(i+1, s, re, c);
                c[i] +=32;
                process(i+1, s, re, c);
            }
            else {
                c[i] = cur;
                process(i+1, s, re, c);
            }
        }
        
        return;
    }
}
