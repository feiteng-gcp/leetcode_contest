class Solution {
    public List<String> letterCasePermutation(String S) {
        List<String> res = new ArrayList<>();
        helper(S.toCharArray(), 0, res);
        return res;
    }
    
    public void helper(char[] S, int i, List<String> res) {
        if (i == S.length) {
            res.add(new String(S));
            return;
        }
        char ch = S[i];
        if (Character.isDigit(ch)) {
            helper(S, i+1, res);
            return;
        }
        if (Character.isLowerCase(ch)) {
            helper(S, i+1, res);
            S[i] = Character.toUpperCase(ch);
             helper(S, i+1, res);            
        } else {
            helper(S, i+1, res);
            S[i] = Character.toLowerCase(ch);
            helper(S, i+1, res);            
        }
    }
}
