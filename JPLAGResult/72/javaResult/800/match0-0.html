class Solution {
  public void generatePerms(String s, int index, String prefix, List<String> result) {
    if (index == s.length()) {
      result.add(prefix);
      return;
    }

    char c = s.charAt(index);
    if (Character.isAlphabetic(c)) {
      generatePerms(s, index + 1, prefix + Character.toLowerCase(c), result);
      generatePerms(s, index + 1, prefix + Character.toUpperCase(c), result);
    } else {
      generatePerms(s, index + 1, prefix + c, result);
    }

  }

  public List<String> letterCasePermutation(String s) {
    List<String> result = new ArrayList<>();
    generatePerms(s, 0, "", result);
    return result;
  }
}
