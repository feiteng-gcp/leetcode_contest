class Solution {

    List<String> ans = new ArrayList<>();
    public void travese(
String s, int pos) {
        if(pos == s.length()) {
            ans.add(s);
            return;
        }
        char c = s.charAt(pos);
        if((c <= 'z' && c >= 'a') || (c <= 'Z' && c >= 'A')) {
            char[] copy = s.toCharArray();
            copy[pos] = Character.toLowerCase(c);
            travese(new String(copy), pos + 1);
            copy[pos] = Character.toUpperCase(c);
            travese(new String(copy), pos + 1);
        
} else {
            travese(s, pos + 1);
        }
    }
    public List<String> letterCasePermutation(String S) {

        travese(S, 0);
        return ans;
    }

}
