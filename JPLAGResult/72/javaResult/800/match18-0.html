class Solution {
    public List<String> letterCasePermutation(String S) {
        if(S==null){
            return new ArrayList<>();
        }
        if(S.length()==0){
            List<String> list = new ArrayList<>();
            list.add("");
            return list;
        }

        List<String> list = find(S);

        return list;
    }

    private List<String> find(String s){

        List<String> list = new ArrayList<>();

        List<String> result = new ArrayList<>();

        for(int i=0;i<s.length();i++){
            char c = s.charAt(i);
            if(Character.isLetter(c)){
                String pre = s.substring(0, i+1);
                list.add(pre);
                if(Character.isUpperCase(c)){
                    pre = pre.substring(0, pre.length()-1)+ Character.toLowerCase(c);
                    list.add(pre);
                }else {
                    pre = pre.substring(0, pre.length()-1)+ Character.toUpperCase(c);
                    list.add(pre);
                }

                List<String> rest = find(s.substring(i+1));
                for(int j=0; j<list.size();j++){
                    for(int k =0; k<rest.size();k++){
                        result.add(list.get(j)+rest.get(k));
                    }
                }

                break;
            }
        }

        if(result.isEmpty()){
            result.add(s);
        }
        return result;
    }
}
