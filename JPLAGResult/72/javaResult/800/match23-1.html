class Solution {
    
void helper(int idx, String s, List<String> lst, StringBuilder sb) {
        if(idx >= s.length()) {
            lst.add(sb.toString());
            return;
        }
        if(s.charAt(idx) >= '0' && s.charAt(idx) <= '9') {
            
sb.append(s.charAt(idx));
            helper(idx + 1, s, lst, sb);
        }
        else {
            sb.append(s.charAt(idx));
            helper(idx + 1, s, lst, sb);
            sb.delete(sb.length() - 1, sb.length());
            if(s.charAt(idx) >= 'A' && s.charAt(idx) <= 'Z') sb.append((char)(s.charAt(idx) + 32));
            else sb.append((char)(s.charAt(idx) - 32));
            helper(idx + 1, s, lst, sb);
        }
        sb.delete(sb.length() - 1, sb.length());
    }
    
    public List<String> letterCasePermutation(String S) {
        List<String> lst = new LinkedList<>();
        StringBuilder sb = new StringBuilder();
        helper(0, S, lst, sb);
        return lst;
    }
}
