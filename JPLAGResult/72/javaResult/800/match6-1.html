class Solution {
    public List<String> letterCasePermutation(String S) {
        List<String> res = new ArrayList<>();
        
        res.add("");
        if(S == null || S.length() == 0){return res;}
        int diff = 'a' - 'A';
        
for(char c : S.toCharArray()){
            List<String> temp = new ArrayList<>();
            for(String s : res){
                if(c >= 'a' && c <= 'z'){
                    temp.add(s + (char)(c - diff));
                } else if (c >= 'A' && c <= 'Z'){
                    temp.add(s + (char)(c + diff));
                } 
                temp.add(s + c);
            
}
            res = temp;
        }
        
        return res;
    }
}
