class Solution {
    public List<String> letterCasePermutation(String S) {
        ArrayList<String> arr = new ArrayList();
        arr.add(S);
        for(int index=0; index<S.length();index++){
            if((S.charAt(index) >='a' && S.charAt(index) <='z') || (S.charAt(index) >='A' && S.charAt(index) <='Z')){
                ArrayList<String> append = new ArrayList();
                for(String s:arr) {   
                    String ss= new String();
                    if(s.charAt(index) >='a' && s.charAt(index) <='z')
                        ss = s.substring(0,index)+s.substring(index,index+1).toUpperCase()+s.substring(index+1,s.length());    
                    else
                        ss = s.substring(0,index)+s.substring(index,index+1).toLowerCase()+s.substring(index+1,s.length());                           append.add(ss);    
                }
                arr.addAll(append);
            }
        }
        return arr;
    }
}
