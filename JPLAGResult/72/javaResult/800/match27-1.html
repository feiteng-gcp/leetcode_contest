class Solution {
    public List<String> letterCasePermutation(String S) {
        List<String> permutations = new ArrayList<String>();
        if (S == null)
            return permutations;
        char[] cArray = 
S.toCharArray();
        permutations.add(S);
        letterCasePermutation(0, cArray, permutations);
        return permutations;
        
    }
    
    private void letterCasePermutation(int idx, char[] cArray, List<String> permutations) {
        if (idx == cArray.length)
            return;
        letterCasePermutation(idx+1, cArray, permutations);
        char c = cArray[idx];
        if (c <= 'z' && c >= 'a') {
            cArray[idx] = (char)('A' + c - 'a');
            permutations.add(new String(cArray));
            letterCasePermutation(idx+1, cArray, permutations);
            cArray[idx] = c;
        } else if (c <= 'Z' && c >= 'A') {
            cArray[idx] = (char)('a' + c - 'A');
            permutations.add(new String(cArray));
            letterCasePermutation(idx+1, cArray, permutations);
            cArray[idx] = c;
        }
        
    }
}
