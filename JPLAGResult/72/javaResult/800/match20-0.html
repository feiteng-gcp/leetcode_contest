class Solution {
    public List<String> letterCasePermutation(String S) {
        
        ArrayList<String> list = new ArrayList<>();
        
        recurse(S.toCharArray(), 0, list);
        
        return list;
    }
    
    void recurse(char c[], int i, ArrayList<String> list) {
        // System.out.println("" + i);
        if(i >= c.length) {
            // System.out.println(new String(c));
            list.add(new String(c));
            return;
        }
        
        char c_new[] = new char[c.length];
        for(int j=0; j<c.length; j++) {
            c_new[j] = c[j];
        }
        
        if(c_new[i] >= 65 && c_new[i] <= 90) {
            c_new[i] = (char)(c_new[i] + 32);
            recurse(c_new, i+1, list);
            recurse(c, i+1, list);
        } else if(c_new[i] >= 97 && c_new[i] <= 122) {
            c_new[i] = (char)(c_new[i] - 32);
            recurse(c_new, i+1, list);
            recurse(c, i+1, list);
        } else {
            recurse(c, i+1, list);
        }
    }
}
