class Solution {
    public List<String> letterCasePermutation(String S) {
        
        List<String> result = new ArrayList<>();
        if (S == null) {
            return result;
        }
        
        StringBuilder cur = new StringBuilder();
        dfs(S, 0, result, cur);
        
return result;
    }
    
    private void dfs(String input, int index, List<String> result, StringBuilder cur) {
        while (index < input.length() && Character.isDigit(input.charAt(index))) {
            cur.append(input.charAt(index));
            index++;
        }
        if (index == input.length()) {
            result.add(new String(cur));
            return;
        }
        int originalLength = cur.length();
        cur.append(Character.toLowerCase(
input.charAt(index)));
        dfs(input, index + 1, result, cur);
        cur.setLength(originalLength);
        
        cur.append(Character.toUpperCase(input.charAt(index)));
        dfs(input, index + 1, result, cur);
        cur.setLength(originalLength);
    }
}
