class Solution {
    public List<String> letterCasePermutation(String S) {
        List<String> ret = new ArrayList();
        ret.add(S);
        helper(ret, S.toCharArray(), 0);
        return ret;
    }
    
    private void helper (List<String> ret, char[] chrs, int idx) {
        // if (idx == chrs.length) return;
        int diff = 'A' - 'a';
        for (int i = idx; i < chrs.length; i++) {
            if (chrs[i] >= 'a' && chrs[i] <= 'z') {
                chrs[i] += diff;
                ret.add(String.valueOf(chrs));
                helper(ret, chrs, i+1);
                chrs[i] -= diff;
            }
            if (chrs[i] >= 'A' && chrs[i] <= 'Z') {
                chrs[i] -= diff;
                ret.add(String.valueOf(chrs));
                helper(ret, chrs, i+1);
                chrs[i] += diff;
            }
        }
    }
}
