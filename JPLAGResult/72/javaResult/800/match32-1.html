class Solution {
    public List<String> letterCasePermutation(String S) {
        List<String> result=new ArrayList();
        
        permute(
S.toCharArray(),0,S.length()-1,result);
        return result;
        
    }
    void permute(char[] c,int i,int end , List<String> result){
        if(i>end){
            result.add(String.valueOf(c));
            return;
        }
        
            if(c[i]>=65 && c[i]<=90){
                permute(c,i+1,end,result);
                c[i]=(char)(c[i]+32);
                permute(c,i+1,end,result);
                c[i]=(char)(c[i]-32);
            }else if(c[i]>=97 && c[i]<=122){
                permute(c,i+1,end,result);
                c[i]=(char)(c[i]-32);
                permute(c,i+1,end,result);
                c[i]=(char)(c[i]+32);
            }else{
                permute(c,i+1,end,result);
            }
        }
    
    
}
