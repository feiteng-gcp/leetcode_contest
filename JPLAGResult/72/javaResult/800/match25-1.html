class Solution {
    public List<String> letterCasePermutation(String S) {
        List<String> res = new ArrayList();
        res.add("");
        for(int i=0;i<S.length();i++) {
            char c = S.charAt(i);
            List<String> newRes = new ArrayList();
            for(int j=0;j<res.size();j++) {
                if(Character.isLetter(c)) {
                    newRes.add(res.get(j)+String.valueOf(c).toLowerCase());
                    newRes.add(res.get(j)+String.valueOf(c).toUpperCase());
                }
                else
                    newRes.add(res.get(j)+String.valueOf(c));
            }
            res = newRes;
        }
        return res;
    }
}
