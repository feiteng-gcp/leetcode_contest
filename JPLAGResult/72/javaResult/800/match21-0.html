class Solution {
    public List<String> letterCasePermutation(String S) {
        char[] T = new char[S.length()];
        List<String> ans = new ArrayList<String>();
        re(S.toCharArray(), T, 0, ans);
        return ans;
    }
    
    public void re(char[] S, char[] T, int i, List<String> ans) {
        if (i >= T.length) {
            ans.add(new String(T));
            return;
        }
        if (S[i] >= '0' && S[i] <= '9') {
            T[i] = S[i];
            re(S, T, i + 1, ans);
        } else {
            T[i] = Character.toLowerCase(S[i]);
            re(S, T, i + 1, ans);
            T[i] = Character.toUpperCase(S[i]);
            re(S, T, i + 1, ans);
        }
    }
}
