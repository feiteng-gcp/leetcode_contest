class Solution {
    public List<String> letterCasePermutation(String S) {
        List<String> result = new ArrayList<>();
        dfs(S.toCharArray(), -1, result);
        return result;
    }
    
    private void dfs(char[] arr, int lastIndex, List<String> result) {
        result.add(new String(arr));
        for (int i = lastIndex + 1; i < arr.length; ++i) {
            char ch = arr[i];
            if (ch >= '0' && ch <= '9') {
                continue;
            }
            if (Character.isUpperCase(ch)) {
                arr[i] = Character.toLowerCase(ch);    
            } else {
                arr[i] = Character.toUpperCase(ch);    
            }
            dfs(arr, i, result);
            arr[i] = ch;
        }
    }
}
