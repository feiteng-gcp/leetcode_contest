class Solution {
    public List<String> letterCasePermutation(String S) {
        List<String> ans = new ArrayList<>();
        char[] s = S.toLowerCase().toCharArray();
        for(char ch : s)
            System.out.print(ch);
        
System.out.println();
        perm(s, 0, ans);
        return ans;
    }
    
    private void perm(char[] s, int i, List<String> ans) {
        if (i == s.length) {
            ans.add(new String(s));
            return;
        }
        //for(int j = i; j < s.length; j++) {
            char ch = s[i];
            if (ch >='0' && ch <= '9') {
                perm(s, i+1, ans);
            } else {
            perm(s, i+1, ans);
            s[i] = (char)('A' + ch - 'a');
            perm(s, i+1, ans);
            s[i] = ch;
            }
        //}
    }
}
