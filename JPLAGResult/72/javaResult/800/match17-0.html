class Solution {
    public List<String> letterCasePermutation(String S) {
        List<String> res = new ArrayList<String>();
        collect(res, S, "", 0);
        return res;
        
    }
    
    public void collect(List<String> res, String inp, String cur, int depth) {
        if (depth == inp.length()) {
            res.add(cur);
            return;
        }
        
        
            char c = inp.charAt(depth);
            if (Character.isDigit(c)) {
                collect(res, inp, cur+c, depth+1);
                return;
            }
            
            collect(res, inp, cur+Character.toLowerCase(c), depth+1);
            collect(res, inp, cur+Character.toUpperCase(c), depth+1);
        
    }
}
