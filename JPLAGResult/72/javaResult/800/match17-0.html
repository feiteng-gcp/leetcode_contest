class Solution {
    public List<String> letterCasePermutation(String s) {
        List<String> res = new ArrayList<>();
        go(res, s, 0, new StringBuilder());
        return res;
    }

    private void go(List<String> res, String s, int i, StringBuilder sb) {
        if (i == s.length()) {
            res.add(sb.toString());
            return;
        }
        char c = s.charAt(i);
        if (Character.isDigit(c)) {
            sb.append(c);
            go(res, s, i + 1, sb);
            sb.deleteCharAt(sb.length() - 1);
        } else {
            if (Character.isUpperCase(c)) {
                sb.append(c);
                go(res, s, i + 1, sb);
                sb.deleteCharAt(sb.length() - 1);
                sb.append((char)(c + 32));
                go(res, s, i + 1, sb);
                sb.deleteCharAt(sb.length() - 1);
            } else {
                sb.append(c);
                go(res, s, i + 1, sb);
                sb.deleteCharAt(sb.length() - 1);
                sb.append((char)(c - 32));
                go(res, s, i + 1, sb);
                sb.deleteCharAt(sb.length() - 1);
            }
        }
    }

}
