class Solution {
    List<String> ans = new ArrayList<>();
    public List<String> letterCasePermutation(String S) {
        if(S== null || S.length() == 0){
            ans.add(S);
            return ans;
        }
        traverse(S,"",0);
        return ans;
    }
    
    public void traverse(String S,String gen, int index){
        if(index == S.length())
            ans.add(gen);
        else{
            char curr = S.charAt(index);
            String X = gen, Y = gen;
            int i = index;
            if(Character.isLetter(curr)){
                traverse(S,X+String.valueOf(Character.toLowerCase(curr)),i+1);
                traverse(S,Y+String.valueOf(Character.toUpperCase(curr)),i+1);
            }
            else{
                traverse(S,gen+String.valueOf(curr),i+1);
            }
        }
    }
}
