class Solution {
    private Set<String> result;
    public List<String> letterCasePermutation(String S) {
        result = new HashSet<>();
        backtrack(0, S);
        List<String> res = new ArrayList<>();
        for (String s : result) {
            res.add(s);
        }
        return res;
    }
    private void backtrack(int pos, String s) {
        if (pos >= s.length()) {
            result.add(s);
        } else {
            for (int i = pos; i < s.length();) {
                while (i < s.length() && Character.isDigit(s.charAt(i))) {
                    i++;
                }
                result.add(s);
                if (i == s.length()) {
                    break;
                }
                if (Character.isLowerCase(s.charAt(i))) {
                    // result.add(s.substring(0, i) + Character.toUpperCase(s.charAt(i)) + s.substring(i + 1, s.length()));
                    
backtrack(i + 1, s.substring(0, i) + Character.toUpperCase(s.charAt(i)) + s.substring(i + 1, s.length()));
                } else {
                    // result.add(s.substring(0, i) + Character.toLowerCase(s.charAt(i)) + s.substring(i + 1, s.length()));
                    backtrack(i + 1, s.substring(0, i) + Character.toLowerCase(s.charAt(i)) + s.substring(i + 1, s.length()));
                }
                i++;
            }
        }
        
    }
}
