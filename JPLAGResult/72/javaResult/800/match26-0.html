class Solution {
    List<String> res = new ArrayList<>();
    public List<String> letterCasePermutation(String S) {
        char[] s = S.toCharArray();
        helper(s, res, 0);
        return res;
    }
    
    void helper(char[] s, List<String> res, int c) {
        if(c == s.length) {
            res.add(new String(s));
            return;
        }
        char cur = s[c];
        helper(s, res, c + 1);
        if(!Character.isDigit(cur)) {
            if(Character.isLowerCase(cur)) {
                s[c] = Character.toUpperCase(cur);
                helper(s, res, c + 1);
            } else {
                s[c] = Character.toLowerCase(cur);
                helper(s, res, c + 1);
            }
            s[c] = cur;
        }
    }
}
