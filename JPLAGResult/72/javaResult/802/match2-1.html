class Solution {
    public int[] kthSmallestPrimeFraction(int[] A, int K) {
        
        // use binary search
        int n = A.length;
        double lo = 0.0;
        double hi = 1.0; // the lo < target <= hi
        
        while (true) {
            
            // shrink the interval until there is only one unique value in that interval
            double mid = (lo + hi)/2.0;
            int count = 0; // the number of values smaller than or equal to mid
            //boolean flag = true;
            int m = 0; // the number of value in (lo, mid]
            
            
            for (int i = 1; i < n; ++i) {
                
                // find the interval
                int start = bs(A, 0, i, lo*(double)A[i]);
                int end = bs(A, 0, i, mid*(double)A[i]);
                
                count += end+1;
                m += end - start;
                
            }
            
            if (count >= K) {
                
                hi = mid;
                // check whether to end the loop
                if (m == 1) break;
                
            }
            if (count < K) {
                
                lo = mid;
                
            }
            
            
        }
        
        int[] res = new int[2];
        
         for (int i = 1; i < n; ++i) {
                
                // find the interval
                int start = bs(A, 0, i, lo*(double)A[i]);
                int end = bs(A, 0, i, hi*(double)A[i]);
                
                if (end - start == 1) {
                    
                    res[0] = A[end];
                    res[1] = A[i];
                    
                }                 
        }
    
        return res;   
        
    }
    
    private int bs(int[] A, int start, int end, double target) {
        
        // find the index of the largest number that is smaller than target
        // if not, return -1
        if (end <= start) return start-1;
        
        int mid = (start + end)/2;
        
        if ((double)A[mid] <= target) return bs(A, mid+1, end, target);
        else return bs(A, start, mid, target);
        
    }
    
}
