class Solution {
	public int[] kthSmallestPrimeFraction(int[] A, int K) {
		int n = A.length;
//		int m = n * (n - 1) / 2;
//		int[][] f = new int[m][2];
//		
//		int len = 0;
//		for (int i = 0; i < n; i++) {
//			for (int j = n - 1; j > i; j--) {
//				f[len][0] = A[i];
//				f[len][1] = A[j];
//				len++;
//			}
//		}
		
		PriorityQueue<int[]> q =  new PriorityQueue<>((a, b) -> {
			return A[a[0]] * A[b[1]] - A[b[0]] * A[a[1]]; 
		});
		
		for (int i = 0; i < n - 1; i++) {
			q.add(new int[] {i, n - 1});
		}
		
		for (int k = 0; k < K; k++) {
			int[] node = q.poll();
			int u = node[0];
			int v = node[1];
			
			if (u < v - 1) {
				q.add(new int[] {u, v - 1});
			}
			
			if (k == K - 1) {

				return new int[] {A[u], A[v]};
			}
		}
		return null;
	}
}
