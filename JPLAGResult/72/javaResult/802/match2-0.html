class Solution {

    public int[] kthSmallestPrimeFraction(int[] A, int K) {
        double lo = 0.0;
        double hi = 1.0;
        int n = A.length;

        while (hi - lo > 1e-8) {
            double mi = (hi + lo) / 2;
            int count = 0;
            for(int l = 0; l < n; l++) {
                for(int r = l + 1; r < n; r++) {
                    if((Double.valueOf(A[l]) / Double.valueOf(A[r])) <= mi) {
                        count++;
                    }
                }
            }
            if(count == K) {
                lo = mi;
                hi = mi;
            }
            if(count > K) {
                hi = mi;
            } else {
                lo = mi;
            }
        }

        int[] ans = {A[0], A[n - 1]};
        Double pre = null;
        for(int l = 0; l < n; l++) {
            for (int r = l + 1; r < n; r++) {
                Double tmp = Double.valueOf(A[l]) / Double.valueOf(A[r]);
                if (tmp <= lo) {
                    if (pre == null) {
                        pre = tmp;
                        ans = new int[]{A[l], A[r]};
                    } else {
                        if (tmp > pre) {
                            pre = tmp;
                            ans = new int[]{A[l], A[r]};
                        }
                    }
                }
            }
        }
        return ans;
    }



}
