class Solution {
    Map<Integer, Map<Integer, Integer>> prices = new HashMap<>();
    int [][] dp;
    int k;
    public void visit(int cur, int round, int val) {
        if(round > k) {
            return;
        }
        if(round >= 0) {
            for(int i = 0; i <= round; i++) {
                if(val > dp[cur][i]) {
                    return;
                }
            }
            dp[cur][round] = val;
        }
        for(int next : prices.get(cur).keySet()) {
            visit(next, round + 1, val + prices.get(cur).get(next));
        }
    }

    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int K) {
        k = K;
        for(int i = 0; i < n; i++) {
            prices.put(i, new HashMap<>());
        }
        dp = new int[n][K + 1];
        for(int i = 0; i < n; i++) {
            for(int j = 0; j <= K; j ++) {
                dp[i][j] =  Integer.MAX_VALUE;
            }
        }
        for(int[] flight : flights) {
            int start = flight[0];
            int end = flight[1];
            int p = flight[2];
            prices.get(start).put(end, p);
        }

        visit(src, -1, 0);


        int min = Integer.MAX_VALUE;
        for(int i = 0; i <= K; i++) {
            min = Math.min(min, dp[dst][i]);
        }

        if(min == Integer.MAX_VALUE) {
            return -1;
        } else {
            return min;
        }
    }
}
