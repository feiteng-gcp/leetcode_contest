class Solution {
    
    class Pair{
        int node;
        int value;
        int steps;
        Pair(int node, int value, int steps){
            this.node = node;
            this.value = value;
            this.steps = steps;
        }
    }
    
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int K) {
        
        int[][] arr = new int[n][n];
        
        for(int i=0;i<flights.length;i++){
            arr[flights[i][0]][flights[i][1]]=flights[i][2];
        }
        
        Queue<Pair> queue = new PriorityQueue<Pair>(1000, new Comparator<Pair>(){
            public int compare(Pair p1, Pair p2){
                if(p1.value==p2.value){
                    if(p1.steps<p2.steps){
                        return -1;
                    }
                }else if(p1.value<p2.value){
                    return -1;
                }
                return 1;
            }
        });
        queue.add(new Pair(src,0,0));
        
        while(!queue.isEmpty()){
            Pair p = queue.poll();
            
            //System.out.println("p.node: " + p.node + " p.val: " + p.value + " p.steps: " + p.steps);
            
            if(p.node == dst){
                return p.value;
            }
            
            for(int i=0;i<n;i++){
                int val = arr[p.node][i];
                if(val>0 && p.steps<=K){
                    queue.add(new Pair(i,val + p.value,p.steps+1));
                }
            }
            
        }
        return -1;
    }
}
