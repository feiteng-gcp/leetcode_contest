class Solution {
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
        Queue<Integer> q = new LinkedList<>();
        int[][] c = new int[n][n];
        for(int[] flight : flights){
            c[flight[0]][flight[1]] = flight[2];
        }
        Map<Integer, Integer> cost = new HashMap<>();
        for(int i = 0 ; i < n; i++){
            cost.put(i, Integer.MAX_VALUE);
        }
        cost.put(src, 0);
        int size = 1;
        q.offer(src);
        Set<Integer> visited = new HashSet<>();
        while(!q.isEmpty() && k >= 0){
            int cur = q.poll();
            visited.add(cur);
            size--;
            
            for(int j = 0; j < n; j++){
                if(c[cur][j] > 0){
                    int curCost = cost.get(cur) + c[cur][j];
                    if(curCost < cost.get(j)){                                        
                        cost.put(j, curCost);
                        q.offer(j);
                    } else if(!visited.contains(j)){
                        q.offer(j);
                    }
                }
            }
            
            if(size == 0){
                k--;
                size = q.size();
            }
        }
        int res = (cost.get(dst) == Integer.MAX_VALUE) ? -1 : cost.get(dst);
        return res;
    }
}
