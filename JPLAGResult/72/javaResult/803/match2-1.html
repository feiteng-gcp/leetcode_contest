class Solution {
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int K) {
        int[][] dp = new int[n][n];
        for(int f=0; f<flights.length;f++)
            dp[flights[f][0]][flights[f][1]] = flights[f][2];
        
        while(K>0){
            for(int from = 0; from < n; from++) {
                for(int to = 0; to < n; to++){
                    for(int x=0; x<n; x++) {
                        if(dp[from][x] > 0 && dp[x][to] > 0 && (dp[from][to] == 0 || dp[from][to] > (dp[from][x]+dp[x][to])))
                            dp[from][to] = dp[from][x]+dp[x][to];
                    }                
                }
            }
            K--;
        }    
        return (dp[src][dst]==0)?-1:dp[src][dst];
    }
}
