import java.util.Arrays;

class Solution {
	final static int MAX = 100000000;
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int K) {
        int[][] cost = new int[n][n];
        for (int[] d : cost) {
        	Arrays.fill(d, -1);
        }
        for (int i = 0; i < flights.length; ++i) {
        	int s = flights[i][0];
        	int t = flights[i][1];
        	int p = flights[i][2];
        	cost[s][t] = p;
        }
        ++K;
        int[][] minCost = new int[n][K + 1];
        for (int[] d : minCost) {
        	Arrays.fill(d, MAX);
        }
        minCost[src][0] = 0;
        for (int i = 0; i < K; ++i) {
        	for (int j = 0; j < n; ++j) {
        		if (minCost[j][i] < MAX) {
	        		for (int t = 0; t < n; ++t) {
	        			if (cost[j][t] >= 0) {
	        				minCost[t][i + 1] = Math.min(minCost[t][i + 1], minCost[j][i] + cost[j][t]);
	        			}
	        		}
        		}
        	}
        }
        int res = MAX;
        for (int i = 0; i <= K; ++i) {
        	res = Math.min(res, minCost[dst][i]);
        }
        return res == MAX ? -1 : res;
    }
}
