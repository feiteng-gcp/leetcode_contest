class Solution {
 	public
 int findCheapestPrice(int n, int[][] flights, int src, int dst, int K) {
		int[][] dist = new int[n][n];
		final int INF = 1 << 28;

		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				if (i != j) {
					dist[i][j] = INF;
				}
			}
		}


		for (int i = 0; i < flights.length; i++) {
			int u = flights[i][0];
			int v = flights[i][1];
			int w = flights[i][2];
			dist[u][v] = w;
		}

		int[] dp = new int[n];

		for (int i = 0; i < n; i++) {
			if (dist[src][i] < INF) {
				dp[i] = dist[src][i];
			} else {
				dp[i] = INF;
			}

		}
		for (int k = 1; k <= K; k++) {
			int[] dp2 = new int[n];
			for (int i = 0; i < n; i++) {
				dp2[i] = INF;
			}
			for (int j = 0; j < n; j++) {
				for (int i = 0; i < n; i++) {
					if (dp[i] >= INF)
						continue;
					if (dist[i][j] >= INF)
						continue;
					dp2[j] = Math.min(dp2[j], dp[i] + dist[i][j]);
				}
			}
			dp = dp2;
		}
		if (dp[dst] >= INF) {
			return -1;
		} else {
			return dp[dst];
		}
	}
}
