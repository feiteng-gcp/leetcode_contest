import java.util.*;
class Solution {
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int K) {
        Integer[] memo = new Integer[n];
        memo[src] = 0;
        
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < flights.length; i++) {
            int s = flights[i][0];
            map.putIfAbsent(s, new ArrayList<>());
            map.get(s).add(i);
        }
        dfs(n, flights, src, dst, K, memo, map);
        // System.out.println();
        if (memo[dst] == null) return -1;
        return memo[dst];
    }
    
    private void dfs(int n, int[][] flights, int src, int dst, 
int K, Integer[] memo, Map<Integer, List<Integer>> map) {
        // System.out.print(src + " cost: " + memo[src]+" ");
        if (src == dst || K < 0) {
            // System.out.println();
            return;
        }
        List<Integer> next = map.get(src);
        if (next == null)  return;
        for (int i = 0; i < next.size(); i++) {
            int index = next.get(i);
            int child = flights[index][1];
            int cost = flights[index][2];
            if (memo[child] != null && memo[src] + cost >= memo[child])  continue;
            memo[child] = memo[src] + cost;
            dfs(n, flights, child, dst, K-1, memo, map);
        }
    }
}
