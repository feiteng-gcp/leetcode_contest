class Solution {
    boolean flag = false;
    
public int findCheapestPrice(int n, int[][] flights, int src, int dst, int K) {
        int[] low = new int[n];
        for(int i=0;i<low.length;i++)
            low[i] = Integer.MAX_VALUE;
        low[src] = 0;
        helper(flights, low, src, K, 0, dst);
        return flag?low[dst]:-1;
    }
    private void helper(int[][] flights, int[] low, int src, int K, int sum, int dst) {
        if(K<0)
            return;
        for(int i=0;i<flights.length;i++) {
            if(flights[i][0]==src) {
                if(flights[i][1]==dst)
                    flag = true;
                int newSum = sum + flights[i][2];
                if(newSum<low[flights[i][1]]) {
                    low[flights[i][1]] = newSum;
                    helper(flights, low, flights[i][1], K-1, newSum, dst);
                }
            }
        }
    }
}
