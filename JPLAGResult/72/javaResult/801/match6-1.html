class Solution {
	boolean DFS(int at, int[] depth, boolean[] visited, int[][] graph, int d) {
		if (visited[at]) return (d - depth[at]) % 2 == 0;
		visited[at] = true;
		depth[at] = d;
		for (int i = 0; i < graph[at].length; ++i) {
			int next = graph[at][i];
			if (!DFS(next, depth, visited, graph, d + 1)) {
				return false;
			}

		}
		return true;
	}
	
    public boolean isBipartite(int[][] graph) {
    	int N = graph.length;
        int[] depth = new int[N];
        
boolean[] visited = new boolean[N];
        for (int i = 0; i < N; ++i) {
        	if (!visited[i]) {
        		if (!DFS(i, depth, visited, graph, 0)) {
        			return false;
        		}
        	}
        }
        return true;
    }
}
