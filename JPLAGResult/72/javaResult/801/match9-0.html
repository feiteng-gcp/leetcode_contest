class Solution {

    int[] visited;
    int n = 0;
    boolean isBipartite = true;
    public void traverse(int[][] graph, int cur, int color) {
        if(!isBipartite) {
            return;
        }
        if(visited[cur] != 0) {
            if(visited[cur] != color) {
                isBipartite = false;
            }
            return;
        }
        visited[cur] = color;
        for(int next : graph[cur]) {
            if(visited[next] == color) {
                isBipartite = false;
                // System.out.println("Here" + " " + visited[next] + " " + cur);
                return;
            }
        }
        for(int next : graph[cur]) {
            traverse(graph, next, -color);
        }
    }

    public boolean isBipartite(int[][] graph) {
        n = graph.length;
        visited = new int[n];
        for(int i = 0; i < n; i++) {
            if(visited[i] == 0)
            traverse(graph, i, 1);
        }
        for(int i = 0; i < n; i++) {
            if(visited[i] == 0) {
                System.out.println("DSF " + i);
                return false;
            }
        }
        return isBipartite;
    }


}
