class Solution {
    public boolean isBipartite(int[][] graph) {
        int N = graph.length;
        int[] colors = new int[N];
        Arrays.fill(colors, -1);
        for (int i = 0; i < N; ++i) {
            if (colors[i] < 0 && !colorable(graph, colors, 0, i)) {
                return false;
            }
        }
        return true;
    }
    
    public boolean colorable(int[][] graph, int[] colors, int col, int curr) {
        if (colors[curr] >= 0) {
            return colors[curr] == col;
        }
        
        colors[curr] = col;
        
        int N = graph.length;
        for (int v : graph[curr]) {
            if (!colorable(graph, colors, 1 - col, v)) {
                return false;
            }
        }
        return true;
    }
}
