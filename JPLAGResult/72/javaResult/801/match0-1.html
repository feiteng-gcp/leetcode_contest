class Solution {
    public boolean isBipartite(int[][] graph) {

        Queue<Integer> visiting = new ArrayDeque();

        int size = graph.length;
        int[] colors = new int[size];
        int[] status = new int[size];

        // colors[0] = 1;
        // status[0] = 1;
        // visiting.add(0);

        int visited = 0;
        while (visited < size) {
            if (visiting.isEmpty()) {
                // try to find the next one to visit
                for (int i = 0; i < size; i++) {
                    if (status[i] == 0) {
                        colors[i] = 1; // index 0 is color 1. we have colors 1 and 2 only
                        status[i] = 1; // 1 means inside queue. -1 means processed
                        visiting.add(i);
                        break;
                    }
                }
            }

            while (!visiting.isEmpty()) {
                Integer curr = visiting.poll();

                int color = colors[curr];
                status[curr] = -1; // we're done with it
                visited++;

                for (Integer neigh: graph[curr]) {
                    if (status[neigh] == -1 || status[neigh] == 1) {
                        if (colors[neigh] == color) {
                            return false;
                        }
                    } else {
                        visiting.add(neigh);
                        if (color == 1) {
                            colors[neigh] = 2;
                        } else {
                            colors[neigh] = 1;
                        }
                        status[neigh] = 1;
                    
}
                }
            }
        }

        return true;
    }
}
