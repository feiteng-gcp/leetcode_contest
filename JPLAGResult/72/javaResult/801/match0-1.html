class Solution {
	int[] flag;

	public boolean isBipartite(int[][] graph) {
		int n = graph.length;
		flag = new int[n];


		for (int i = 0; i < n; i++) {
			if (flag[i] == 0) {
				if (!dfs(i, graph, 1)) {

					return false;
				}
			}
		}
		return true;
	}
