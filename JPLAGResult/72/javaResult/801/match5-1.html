import java.util.*;

class Solution {
    public boolean isBipartite(int[][] graph) {
        int n = graph.length;
        int[] v = new int[n];
        for (int i = 0; i < n; i++) {
            if (v[i] == 0) {
                
if (!dfs(graph, v, i, -1))  return false;
            }
        }
        return true;
    }
    
    private boolean dfs(int[][] graph, int[] v, int i, int pre) {
        if (pre == 1)  v[i] = -1;
        else v[i] = 1;
        int[] next = graph[i];
        for (int j = 0; j < next.length; j++) {
            if (v[next[j]] == 0) {
                if (!dfs(graph, v, next[j], v[i]))  return false;
            } else if (v[next[j]] == v[i])  return false;
        }
        return true;
    }
}
