class Solution {
    
    public int getv(Set<Integer> vis) {
        for (int v:vis) {
            return v;
        }
        return -1;
    }
    
    public boolean isBipartite(int[][] graph) {
        int n = graph.length;
        boolean [][] G = new boolean[n][n];
        for (int i =0;i<n;i++) {
            for (int j: graph[i]) {
                G[i][j] = true;
            }
        }
        Boolean [] colour = new Boolean[n];
        for (int i =0;i<n;i++) {colour[i]=null;}
        
        Set<Integer> vis = new HashSet<>();
        for (int i =0;i<n;i++) {vis.add(i);}
        
        while(!vis.isEmpty()) {
            int curr = getv(vis);
            colour[curr] = true;
            List<Integer> queue = new LinkedList<>();
            queue.add(curr);
            vis.remove(curr);
            while (!queue.isEmpty()) {
                curr = queue.remove(0);
                for (int nbr: graph[curr]) {
                    if (colour[nbr]!=null && colour[nbr]==colour[curr]) {
                        return false;
                    }
                    if (colour[nbr]==null) {
                        colour[nbr]= !colour[curr];
                        queue.add(nbr);
                        vis.remove(nbr);
                    }
                }
            }
        }
        return true;
    }
}
