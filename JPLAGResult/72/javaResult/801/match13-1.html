class Solution {
    public boolean isBipartite(int[][] graph) {
        
        // use a bfs
        int n = graph.length;
        
        Integer[] mark = new Integer[n]; // the set mark
        
        for (int i = 0; i < n; ++i) {
            
            // bfs each node;
            if (mark[i] != null) continue;
            mark[i] = 0;
            
            List<Integer> queue = new LinkedList<Integer>();
            queue.add(i);
            int id = 1;
            
            while (queue.size() > 0) {
                
                int m = queue.size();
                for (int j = 0 ; j < m; ++j) {
                    int cur = queue.remove(0);
                    // find the connected nodes
                    for (int k: graph[cur]) {
                        // the connected nodes
                        if (mark[k] == null) {
                            mark[k] = id;
                            queue.add(k);
                        }
                        else if (mark[k] != id) return false;
                        
                    }
                    
                }
                
                id = 1 - id;
                
                
            }
            
        }
        
        return true;
        
    }
}
