class Solution {
public:
    const int inf=0x3f3f3f3f;
    vector<vector<int>> initDist(int n,int K,int val)
    {
        vector<vector<int>> ret;
        for(int i=0;i<n;i++)
        {
            vector<int> tmp(K+1,val);
            ret.push_back(tmp);
        }
        return ret;
    }
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int K) {
        K++;
        vector<vector<pair<int,int>>> g;
        for(int i=0;i<n;i++)
        {
            vector<pair<int,int>> tmp;
            g.push_back(tmp);
        }
        for(auto p:flights)
        {
            int cur=p[0];
            pair<int,int> destPric=make_pair(p[1],p[2]);
            g[p[0]].push_back(destPric);
        }
        if(src==dst) return 0;
        vector<vector<int>> dist=initDist(n,K,inf),vis=initDist(n,K,0);
        queue<pair<int,int>> q;
        q.push(make_pair(src,K));
        vis[src][K]=1;
        dist[src][K]=0;
        while(q.size())
        {
            auto curPair=q.front();q.pop();
            int now=curPair.first,left=curPair.second;
            if(left==0)continue;
            for(auto nextPair:g[now])
            {
                int next=nextPair.first,price=nextPair.second;
                if(dist[next][left-1]>dist[now][left]+price)
                {
                    dist[next][left-1]=dist[now][left]+price;
                    if(!vis[next][left-1])
                    {
                        vis[next][left-1]=1;
                        q.push(make_pair(next,left-1));
                    }
                }
            }
            vis[now][left]=0;
        }
        int ans=inf;
        for(int i=0;i<=K;i++) ans=min(ans,dist[dst][i]);
        return ans==inf?-1:ans;
    }
};
