vector<pair<int, int>> g[110];
#define F first
#define S second
int d[110][110];
bool inq[110][110];
class Solution {
public:
    int findCheapestPrice(int n, vector<vector<int>>& fs, int s, int t, int k) {
        if (s == t)
            return 0;
        for (int i = 0; i < n; ++i) g[i].clear();
        for (auto x : fs) {
            g[x[0]].emplace_back(x[1], x[2]);
        }
        memset(d, 0x7f, sizeof d);
        const int inf = d[0][0];
        for (int i = 0; i < 110; ++i)
            d[s][i] = 0;
        queue<pair<int, int>> q;
        q.emplace(s, 0);
        while (q.size()) {
            int x = q.front().F;
            int y = q.front().S;
            q.pop();
            inq[x][y] = false;
            if (y == k + 1)
                continue;
            for (auto e : g[x]) {
                int &val = d[e.F][y + 1];
                int nd = d[x][y] + e.S;
                if (nd < val) {
                    val = nd;
                    if (!inq[e.F][y + 1]) {
                        inq[e.F][y + 1] = true;
                        q.emplace(e.F, y + 1);
                    }
                }
            }

        }
        int ans = inf;
        for (int i = 0; i <= k + 1; ++i)
            ans = min(ans, d[t][i]);
        return ans >= inf ? -1 : ans;
    }
};
