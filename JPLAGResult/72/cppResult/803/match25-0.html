class Solution {
public:
    int dp[102][102];
    priority_queue<pair<int,pair<int,int> > , vector<pair<int,pair<int,int> > > , greater<pair<int,pair<int,int> > > > q;
    vector<pair<int,int> > vv[102];
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int K) {
        K++;
        memset(dp,-1,sizeof(dp));
        dp[src][0]=0;
        q.push(make_pair(0,make_pair(src,0)));
        for(int i=0;i<n;i++){
            vv[i].clear();
        }
        for(int i=0;i<flights.size();i++){
            vv[flights[i][0]].push_back(make_pair(flights[i][1],flights[i][2]));
        }
        while(!q.empty()){
            int cost=q.top().first;
            int a=q.top().second.first;
            int b=q.top().second.second;
            q.pop();
            if(dp[a][b]!=cost){
                continue;
            }
            for(int i=0;i<vv[a].size();i++){
                int go=vv[a][i].first;
                int cost=dp[a][b]+vv[a][i].second;
                int bb=b+1;
                if(bb>K)continue;
                if(dp[go][bb]==-1||dp[go][bb]>cost){
                    dp[go][bb]=cost;
                    q.push(make_pair(cost,make_pair(go,bb)));
                }
            }
        }
        int ans=INT_MAX;
        for(int i=0;i<=K;i++){
            if(dp[dst][i]!=-1){
                ans=min(ans,dp[dst][i]);
            }
        }
        if(ans==INT_MAX)ans=-1;
        return ans;
    }
};
