class Solution {
public:
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int K) {
        int ret = INT_MAX;
        int k = K + 1;
        vector<vector<int>> f(n, vector<int>(k + 1, INT_MAX));
        vector<vector<pair<int, int>>> arr(n, vector<pair<int, int>>(0, make_pair(0, 0)));
        for (auto& p : flights)
            arr[p[0]].push_back(make_pair(p[1], p[2]));
        f[src][0] = 0;
        deque<pair<int, int>> dq;
        dq.push_back(make_pair(src, 0));
        while (!dq.empty()) {
            int x = dq.front().first, t = dq.front().second;
            for (int i = t; i <= min(t, k - 1); ++ i)
                if (f[x][i] < INT_MAX) {
                    for (auto& p : arr[x])
                        if (f[x][i] + p.second < f[p.first][i + 1]) {
                            f[p.first][i + 1] = f[x][i] + p.second;
                            dq.push_back(make_pair(p.first, t + 1));
                        }
                }
            dq.pop_front();
        }
        for (int i = 0; i <= k; ++ i)
            ret = min(f[dst][i], ret);
        return ret == INT_MAX ? -1 : ret;
    }
};
