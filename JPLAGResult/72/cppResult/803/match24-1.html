class Solution {
public:
    
    bool ok(int n, vector<vector<pair<int,int>>>& edges, int src, int dst, int k, int maxlen) {
        if (src==dst) return true;
        ++k;
        vector<int> dis(n, INT_MAX);
        dis[src] = 0; 
        queue<pair<int,int>> q;
        q.push({src, 0});
        
        while (!q.empty()) {
            int now = q.front().first;
            int stp = q.front().second;
            q.pop();
            for (auto& e : edges[now]) {
                int nxt = e.first, w = e.second;
                if (dis[now]+w<dis[nxt] && dis[now]+w<=maxlen) {
                    if (nxt==dst) return true;
                    dis[nxt] = dis[now] + w;
                    if (stp+1<k) q.push({nxt, stp+1});
                }
            }

        }
        return false;
    }
    
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int K) {
        vector<vector<pair<int,int>>> edges(n+1, vector<pair<int,int>>{});
        
        int l = 0, r = 0;
        
        for (auto& line : flights) {
            edges[line[0]].push_back({line[1], line[2]});
            r += line[2];
        }
        
        while (l<r) {
            int mid = (l+r) >> 1;
            if (ok(n, edges, src, dst, K, mid)) 
                r = mid;
            else
                l = mid + 1;
        }
        return ok(n, edges, src, dst, K, l) ? l : -1;
        
    }
};
