class Solution {
public:
    int g[105][105];
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int K) {
        int dis[105][105],vis[105][105];
	memset(dis,-1,sizeof(dis));
	memset(vis,0,sizeof(vis));
	dis[src][0]=0;
	memset(g,-1,sizeof(g));
	int m=flights.size();
	for(int i=0;i<m;i++)
	{
		int s=flights[i][0],d=flights[i][1],c=flights[i][2];
		if(g[s][d]==-1)g[s][d]=c;
		else g[s][d]=min(g[s][d],c);
	}
	for(int i=0;i<n*n;i++)
	{
		int cur=-1,curc=-1;
		for(int j=0;j<n;j++)
		{
			for(int k=0;k<n;k++)
			{
				if(dis[j][k]!=-1&&!vis[j][k]&&(cur==-1||dis[cur][curc]>dis[j][k]))
				{
					cur=j;
					curc=k;
				}
			}
		}
		if(cur==-1)break;
		//printf("%d %d:%d\n",cur,curc,dis[cur][curc]);
		vis[cur][curc]=1;
		for(int j=0;j<n;j++)
		{

			if(g[cur][j]==-1)continue;
			if(dis[j][curc+1]==-1||dis[cur][curc]+g[cur][j]<dis[j][curc+1])
			{
				dis[j][curc+1]=dis[cur][curc]+g[cur][j];
				//printf("*%d %d:%d\n",j,curc+1,dis[j][curc+1]);
			}
		}
	}
	int ans=-1;
	for(int i=0;i<=K+1;i++)
	{
		if(dis[dst][i]!=-1)
		{
			if(ans==-1||ans>dis[dst][i])ans=dis[dst][i];
		}
	}
	return ans;
    }
};
