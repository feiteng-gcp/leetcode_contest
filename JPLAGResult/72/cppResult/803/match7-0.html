#include<string.h>
class Solution {
    int dis[128],edg[128][128];
    int dis2[128];
public:
    int findCheapestPrice(int n, vector<vector<int>>& f, int src, int dst, int k) {
        memset(dis,-1,sizeof(dis));
        memset(edg,-1,sizeof(edg));
        int m=f.size();
        for(int i=0;i<m;i++)
        {
            edg[f[i][0]][f[i][1]]=f[i][2];
        }
        dis[src]=0;
        k++;
        while(k--)
        {
            for(int i=0;i<n;i++)
            {
                dis2[i]=dis[i];
            }
            for(int i=0;i<n;i++)
            {
                if(dis2[i]<0)
                {
                    continue;
                }
                for(int j=0;j<n;j++)
                {
                    if(edg[i][j]<0)
                    {
                        continue;
                    }
                    if(dis[j]<0||dis[j]>dis2[i]+edg[i][j])
                    {
                        dis[j]=dis2[i]+edg[i][j];
                    }
                }
            }
        }
        return dis[dst];
    }
};
