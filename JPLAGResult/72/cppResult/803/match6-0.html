class Solution {
public:
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int K) {
        int m = flights.size();

        vector<vector<int>> cost(n,vector<int> (n,INT_MAX));
        cost[src][src] =0; 
        
        for(int i =0; i<n; i++)
        {
            cost[i][i] =0; 
        }
        
        for(int e =0; e<m ; e++)
        {
            int s = flights[e][0];
            int v = flights[e][1];
            int c = flights[e][2];
            
            cost[s][v] = c;
        }
        
        
        
        if(K <1) return (cost[src][dst] == INT_MAX)?-1:cost[src][dst];
        
        while(K)
        {
            for(int e =0; e<m ; e++)
            {
                int s = flights[e][0];
                int v = flights[e][1];
                int c = flights[e][2];

                if((cost[src][s] ) < (cost[src][v]-c))
                {
                    cost[src][v] = cost[src][s] + c; 
                }
            }
        
            K--; 
        }
        
        
        return (cost[src][dst] == INT_MAX)?(-1):(cost[src][dst]);
        
    }
};
