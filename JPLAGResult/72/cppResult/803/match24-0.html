#define inf (1<<28)

class Solution {
public:
    int calc(const vector<vector<pair<int,int>>>& graph, const int dst, vector<vector<int>>& dp, int now, int rem) {
        if (now == dst)
                return 0;
        if (rem == 0) 
            return inf;
        if (dp[now][rem] != -1)
            return dp[now][rem];
        dp[now][rem] = inf;
        for (int i = 0; i < graph[now].size(); i++) {
            int v = graph[now][i].second + calc(graph, dst, dp, graph[now][i].first, rem - 1);
            if (v < dp[now][rem])
                dp[now][rem] = v;
        }
        return dp[now][rem];
    }
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int K) {
        vector<vector<pair<int,int>>> graph(n, vector<pair<int,int>>());
        vector<vector<int>> dp(n + 1, vector<int>(K + 2, -1));
        for (int i = 0; i < flights.size(); i++)
            graph[flights[i][0]].push_back(make_pair(flights[i][1], flights[i][2]));
        int ret = calc(graph, dst, dp, src, K + 1);
        if (ret == inf)
            return -1;
        return ret;
    }
};
