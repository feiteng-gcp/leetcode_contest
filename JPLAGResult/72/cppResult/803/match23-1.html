class Solution {
public:
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int K) {
        vector<vector<pair<int, int>>> graph(n);
        for (const auto& edge : flights) {
            graph[edge[0]].emplace_back(edge[1], edge[2]);
        }
        
        int dis[n];
        memset(dis, 255, sizeof(dis));
        dis[src] = 0;
        int hop[n];
        memset(hop, 0, sizeof(hop));
        
        bool inque[n];
        memset(inque, 0, sizeof(inque));
        inque[src] = true;
        
        queue<int> que;
        que.push(src);
        while (!que.empty()) {
            int cur = que.front();
            que.pop();
            inque[cur] = false;
            
            for (const auto& edge : graph[cur]) {
                int next = edge.first;
                int w = edge.second;
                
                if ((dis[cur] + w <= dis[next] || dis[next] == -1) && hop[cur] <= K) {
                    if (dis[cur] + w < dis[next] || dis[next] == -1) {
                        dis[next] = dis[cur] + w;
                        hop[next] = hop[cur] + 1;
                        if (!inque[next]) {
                            que.push(next);
                            inque[next] = true;
                        }
                    } else {
                        if (hop[cur] + 1 < hop[next]) {
                            hop[next] = hop[cur] + 1;
                            if (!inque[next]) {
                                que.push(next);
                                inque[next] = true;
                            }
                        }
                    }
                }
            }
        }
        
        return dis[dst];
    }
};
