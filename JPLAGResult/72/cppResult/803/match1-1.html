class Solution {
public:
    int findCheapestPrice(int n, vector<vector<int>>& flights, int src, int dst, int K) {
        vector<int> dp(n, -1);
        dp[src] = 0;
        while (K-- >= 0) {
            auto tmp = dp;
            for (auto & item : flights) {
                if (dp[item[0]] == -1) continue;
                int val = dp[item[0]] + item[2];
                if (tmp[item[1]] == -1 || tmp[item[1]] > val) tmp[item[1]] = val;
            }
            swap(tmp, dp);
        }
        return dp[dst];
        
        
    }
};
