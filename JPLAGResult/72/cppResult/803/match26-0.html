class Solution {
public:
    int findCheapestPrice(int n, vector<vector<int> >& flights, int src, int dst, int K)
{
    vector<vector<pair<int, int> > > lnk(n);
    int len = (int)flights.size();
    for(int i = 0; i < len; i++)
    {
        int u = flights[i][0], v = flights[i][1], w = flights[i][2];
        lnk[v].push_back({u, w});
    }
    const int maxP = 1e9;
    vector<int> dis(n, maxP);
    dis[src] = 0;
    K++;
    while(K--)
    {
        vector<int> tem = dis;
        for(int i = 0; i < n; i++)
            for(int j = 0; j < lnk[i].size(); j++)
            {
                int u = lnk[i][j].first, w = lnk[i][j].second;
                if(dis[u] + w < tem[i]) tem[i] = dis[u] + w;
            }
        dis = tem;
    }
    return dis[dst] >= maxP ? -1 : dis[dst];
}
};
