#include<queue>
#include<string.h>
using namespace std;
struct rr{
    int p;
    int q;
    int wz1;
    int wz2;

    bool operator<(rr b) const
    {
        return p*b.q>q*b.p;
    }
};
priority_queue<rr> qq;
bool vis[2048][2048];
class Solution {
public:
    vector<int> kthSmallestPrimeFraction(vector<int>& a, int k) {
        while(!qq.empty())
        {
            qq.pop();
        }
        memset(vis,0,sizeof(vis));
        int n=a.size();
        rr s;
        s.wz1=0;
        s.wz2=n-1;
        s.p=1;
        s.q=a[n-1];
        qq.push(s);
        vis[0][n-1]=true;
        while(k--)
        {
            s=qq.top();
            qq.pop();
            if(s.wz2==s.wz1+1)
            {
                continue;
            }
            rr v1=s;
            rr v2=s;
            v1.wz1++;
            v2.wz2--;
            v1.p=a[v1.wz1];
            v2.q=a[v2.wz2];
            if(!vis[v1.wz1][v1.wz2])
            {
                vis[v1.wz1][v1.wz2]=true;
                qq.push(v1);
            }
            if(!vis[v2.wz1][v2.wz2])
            {
                vis[v2.wz1][v2.wz2]=true;
                qq.push(v2);
            }
        }
        vector<int> v;
        v.push_back(s.p);
        v.push_back(s.q);
        return v;
    }
};
