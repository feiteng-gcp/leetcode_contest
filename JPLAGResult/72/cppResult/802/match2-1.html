class Solution {
    struct Item{
      int a, b, id;
      Item(int _a, int _b, int _id) {
          a = _a;
          b = _b;
          id = _id;
      }
    };
    struct Cmp{
        bool operator() (const Item & x, const Item & y) const {
            if (x.a*y.b != y.a*x.b) return x.a*y.b > y.a*x.b;

            return x.id > y.id;
        }
    };
public:
    vector<int> kthSmallestPrimeFraction(vector<int>& A, int K) {
        int n = A.size();
        sort(A.begin(), A.end());
        vector<int> ind(n, n-1);
        Item ret(-1, -1, -1);
        priority_queue<Item, vector<Item>, Cmp> pq;
        for (int i = 0; i < n; i++) {
            if (i < ind[i]) {
                pq.push(Item(A[i], A[ind[i]--], i));
            }
        }
        while (K-- > 0) {
            ret = pq.top(); pq.pop();
            int i = ret.id;
            if (ind[i] > i) {
                pq.push({A[i], A[ind[i]--], i});
            }
        }
        return vector<int>{ret.a, ret.b};
        
        
    }
};
