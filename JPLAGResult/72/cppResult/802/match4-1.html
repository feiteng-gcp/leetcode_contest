typedef pair<int, int> ii;
struct Node {
    int p, q;
    int idx;
};

bool operator <(const Node& a, const Node& b) {
    return a.p * b.q > b.p * a.q;
}

class Solution {
public:
    vector<int> kthSmallestPrimeFraction(vector<int>& a, int k) {
        vector<ii> A;
        sort(a.begin(), a.end());
        int n = a.size();
        priority_queue<Node> Q;
        for (int i = 0; i < n - 1; ++i) {
            Q.push(Node{a[i], a[n - 1], n - 1});
        }
        while (k--) {
            auto cur = Q.top();
            Q.pop();
            //cout << cur.p << " " << cur.q << " " << cur.idx << endl;
            if (cur.idx - 1 >= 0) {
                Q.push(Node{cur.p, a[cur.idx - 1], cur.idx - 1});
            }
            if (k == 0) {
                return {cur.p, cur.q};
            }
        }
        return {};
    }
};
