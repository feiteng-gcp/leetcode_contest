class Solution {
public:
    vector<int> kthSmallestPrimeFraction(vector<int>& A, int K) {
        int l = A.size();
        vector<int> heap(l);
        vector<int> q(l);
        
        for (int i = 0; i < l; ++i) {
            heap[i] = i;
            q[i] = l - 1;
        }
        heap.pop_back();
        
        for (int i = 0; i < K-1; ++i) {
            int now = heap[0];
            //printf("%d %d\n", A[now], A[q[now]]);
            
            // pop
            heap[0] = heap[heap.size() - 1];
            heap.pop_back();
            int p = 0;
            while (true) {
                int p2 = p;
                if (p + p + 1 < heap.size() && check(A[heap[p2]], A[q[heap[p2]]], A[heap[p + p + 1]], A[q[heap[p + p + 1]]])) p2 = p + p + 1;
                if (p + p + 2 < heap.size() && check(A[heap[p2]], A[q[heap[p2]]], A[heap[p + p + 2]], A[q[heap[p + p + 2]]])) p2 = p + p + 2;
                if (p2 == p) break;
                int tmp = heap[p2]; heap[p2] = heap[p]; heap[p] = tmp;
                p = p2;
            }
            
            q[now] -= 1;
            if (q[now] > now) {
                // insert
                heap.push_back(now);
                int p = heap.size() - 1;
                while (p > 0) {
                    int p2 = (p-1) / 2;
                    if (check(A[heap[p2]], A[q[heap[p2]]], A[heap[p]], A[q[heap[p]]])) {
                        int tmp = heap[p2]; heap[p2] = heap[p]; heap[p] = tmp;
                        p = p2;
                    } else break;
                }
            }
        }
        
        vector<int> res(2);
        res[0] = A[heap[0]]; res[1] = A[q[heap[0]]];
        return res;
    }
    
    bool check(int p1, int q1, int p2, int q2) {
        return p1 * q2 > p2 * q1;
    }
};
