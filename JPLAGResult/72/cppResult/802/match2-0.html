class Solution {
private:
    // returns true if first <= second
    bool compar(pair<int, int> first, pair<int, int> second) {
        long left = (long) first.first * (long) second.second;
        long right = (long) first.second * (long) second.first;
        return left <= right;
    }
    
    // search inside [start, end)
    vector<int> kth(vector<int> &A, vector<int> &start, vector<int> &end, int K) {
        int n_elem = 0;
        int N = A.size();
        vector<int> active_idx;
        for (int i = 0; i < N; ++i) {
            if (start[i] < end[i]) {
                n_elem += end[i] - start[i];
                active_idx.push_back(i);
            }
        }
        int pivot_start_idx = active_idx[rand() % active_idx.size()];
        int pivot_end_idx = start[pivot_start_idx] + (rand() % (end[pivot_start_idx] - start[pivot_start_idx]));
        pair<int, int> pivot = {A[pivot_start_idx], A[pivot_end_idx]};
        vector<int> middle(N, 0);
        int n_less_than = 0;
        for (int i : active_idx) {
            int left = start[i], right = end[i] - 1;
            // Search for minimum middle, s.t. A[i] / A[middle] <= pivot
            if (!compar({A[i], A[right]}, pivot)) {
                middle[i] = end[i];
            } else {
                // binary search
                while (left < right) {
                    int mid = left + (right - left) / 2;
                    if (compar({A[i], A[mid]}, pivot)) {
                        right = mid;
                    } else {
                        left = mid + 1;
                    }
                }
                n_less_than += end[i] - left;
                middle[i] = left;
            }
        }
        if (n_less_than == K) return {pivot.first, pivot.second};
        else if (n_less_than < K) {
            return kth(A, start, middle, K - n_less_than);
        } else {
            middle[pivot_start_idx] = pivot_end_idx + 1;
            return kth(A, middle, end, K);
        }
    }

public:
    vector<int> kthSmallestPrimeFraction(vector<int>& A, int K) {
        int N = A.size();
        vector<int> start, end;
        for (int i = 0; i < N; ++i) {
            start.push_back(i + 1);
            end.push_back(N);
        }
        return kth(A, start, end, K);
    }
};
