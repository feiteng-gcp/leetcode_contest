class Solution {
public:
    struct node{
        int x,y,z;
        node(){}
        node(int a,int b,int c){
            x=a;
            y=b;
            z=c;
        }
        friend bool operator < (node a,node b){
            return a.x*b.y>a.y*b.x;
        }
    };

    vector<int> kthSmallestPrimeFraction(vector<int>& A, int K) {
        int n=A.size();
        vector<int>co;
        co.resize(n);
        int a,b;
        priority_queue<node>que;
        for(int i=1;i<n;++i) que.push(node(A[0],A[i],i)),co[i]=0;
        node t;
        while(K--){
            t=que.top();
            que.pop();
            ++co[t.z];
            //cout<<t.x<<' '<<t.y<<endl;
            if(co[t.z]>=t.z) continue;
            que.push(node(A[co[t.z]],A[t.z],t.z));
        }
        vector<int>ans;
        ans.push_back(t.x);
        ans.push_back(t.y);
        return ans;
    }
};
