struct fraction {
    int lob, hor;
    fraction(int l, int h) { lob = l; hor = h;}
};

bool operator<(fraction A, fraction B) {
    return A.lob*B.hor < A.hor*B.lob;
}

bool operator==(fraction A, fraction B) {
    return A.lob*B.hor == A.hor*B.lob;
}

bool operator>(fraction A, fraction B) {
    return A.lob*B.hor > A.hor*B.lob;
}

class Solution {
public:
    vector<int> kthSmallestPrimeFraction(vector<int>& A, int K) {
        priority_queue<fraction> pq;

        for (int i = 0; i < A.size(); i++) {
            for (int j = A.size() - 1; j > i; j--) {
                fraction f(A[i], A[j]);
                if (pq.size() < K) {
                    pq.push(f);
                    continue;
                }
                
                fraction now = pq.top();
                if (now < f) 
                    break;
                else {
                    pq.pop();
                    pq.push(f);
                }
            }
        }
        
        fraction now = pq.top();
        return vector<int> {now.lob, now.hor};
    }
};
