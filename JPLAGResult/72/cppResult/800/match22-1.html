class Solution {
public:
	vector<string> res; 
	void util(string S, string curRes, int idx)
	{
		if(idx == S.size())
		{
			res.push_back(curRes);
			return;
		}

		util(S, curRes + S[idx], idx + 1);
		if(S[idx] >= 65 && S[idx] <= 90)
		{
			 
			 util(S, curRes + static_cast<char>((S[idx] + ('a' -'A'))), idx + 1);
		}
		else if(S[idx] >= 97 && S[idx] <= 122)
		{
			 util(S, curRes + static_cast<char>((S[idx] + ('A' -'a'))), idx + 1);
		}

	}

    vector<string> letterCasePermutation(string S) 
    {
		util(S, "", 0);		    		    
		return res;
    }
};
