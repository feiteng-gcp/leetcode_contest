class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> result;
        result.push_back(S);
        letter(S,0,result);
        return result;
    }
    
    void letter(string S, int index, vector<string> &result){
        int n = S.size();
        for(int i=index;i<n;i++){

            if((S[i] >='a' && S[i]<='z')){
                S[i] = S[i] - 'a' + 'A';
                result.push_back(S);
                letter(S, i + 1, result);
                S[i] = S[i] - 'A' + 'a';
            }
            else if((S[i] >='A' && S[i]<='Z')){
                S[i] = S[i] - 'A' + 'a';
                result.push_back(S);
                letter(S, i + 1, result);
                S[i] = S[i] - 'a' + 'A';
            }
        }
    }
};
