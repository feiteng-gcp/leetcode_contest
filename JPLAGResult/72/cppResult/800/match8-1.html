class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> res;
        string ans;
        dfs(res, ans, S, 0);
        return res;
    }
    void dfs(vector<string>& res, string& ans, string& S, int i) {
        if(i == S.size()){
            res.push_back(ans);
            return;
        }
        if(isdigit(S[i])) {
            ans.push_back(S[i]);
            dfs(res, ans, S, i+1);
            ans.pop_back();
        }
        else if(isupper(S[i])){
            ans.push_back(S[i]);
            dfs(res, ans, S, i+1);
            ans.pop_back();
            ans.push_back(S[i]-'A'+'a');
            dfs(res, ans, S, i+1);
            ans.pop_back();
        }
        else {
            ans.push_back(S[i]+'A'-'a');
            dfs(res, ans, S, i+1);
            ans.pop_back();
            ans.push_back(S[i]);
            dfs(res, ans, S, i+1);
            ans.pop_back();            
        }
    }
};
