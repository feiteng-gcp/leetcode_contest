class Solution {
public:
    string curS;
    vector<string> rv;
    void cal(string &s, int pos) {
        if (pos >= s.size()) {
            rv.push_back(curS);
            return;
        }
        if (s[pos] <= 'z' && s[pos] >= 'a') {
            curS += s[pos];
            cal(s, pos + 1);
            curS.pop_back();
            curS += s[pos] - ('a' - 'A');
            cal(s, pos + 1);
            curS.pop_back();
        }
        else if (s[pos] <= 'Z' && s[pos] >= 'A') {
            curS += s[pos];
            cal(s, pos + 1);
            curS.pop_back();
            curS += s[pos] + ('a' - 'A');
            cal(s, pos + 1);
            curS.pop_back();
        }
        else {
            curS += s[pos];
            cal(s, pos + 1);
            curS.pop_back();
        }
    }
    vector<string> letterCasePermutation(string S) {
        cal(S, 0);
        return rv;
    }
};
