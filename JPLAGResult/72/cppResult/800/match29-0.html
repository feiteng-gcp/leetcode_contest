class Solution {
public:
vector<string> letterCasePermutation(string S) {
    vector<string> ans;
    unordered_set<string> s;
    int len = S.length();
    for(int i = 0; i < (1<<len); ++i){
        string tmp = S;
        for(int j = 0; j < len; ++j){
            if(tmp[j] >= 'a' && tmp[j] <= 'z' && (i & (1<<j))){
                tmp[j] = tmp[j] - 'a' + 'A';
            }
            if(tmp[j] >= 'A' && tmp[j] <= 'Z' && !(i & (1<<j))){
                tmp[j] = tmp[j] + 'a' - 'A';
            }
        }
        s.insert(tmp);
    }
    for(auto k : s){
        ans.push_back(k);
    }
    return ans;
}
};
