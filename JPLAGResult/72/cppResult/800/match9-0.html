class Solution {
public:
    vector<string> res;
    string cur;
    vector<string> letterCasePermutation(string S) {
        Dfs(S, 0);
        return res;
    }
    void Dfs(string &s, int loc) {
        if (loc == s.length()) {
            res.push_back(cur);
            return;
        }
        if (!isalpha(s[loc])) {
            cur.push_back(s[loc]);
            Dfs(s, loc + 1);
            cur.pop_back();
        }
        else {
            cur.push_back(tolower(s[loc]));
            Dfs(s, loc + 1);
            cur.pop_back();
            cur.push_back(toupper(s[loc]));
            Dfs(s, loc + 1);
            cur.pop_back();
        }
    }
};
