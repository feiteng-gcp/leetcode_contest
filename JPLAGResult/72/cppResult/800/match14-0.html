class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> ans;
        work(0, "", S, ans);
        return ans;
    }
    
    void work(int i, string temp, string& S, vector<string> &ans){
        if (i >= S.length()){
            ans.push_back(temp);
            return;
        }
        
        char cc = tolower(S[i]);
        
        work(i+1, temp + cc, S, ans);
        if ('a' <= cc && cc <= 'z')
            work(i+1, temp + ((char)(cc-'a'+'A')), S, ans);
    }
};
