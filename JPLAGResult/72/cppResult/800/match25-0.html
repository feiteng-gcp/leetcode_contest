class Solution {
public:
    void helper_bk(string& s, int idx, vector<string>& res) {
        if(idx == s.size()) {
            res.push_back(s);
            return;
        }
        helper_bk(s, idx+1, res);
        if(s[idx] >= 'a' && s[idx] <= 'z') {
            s[idx] = toupper(s[idx]);
            helper_bk(s, idx+1, res);
            s[idx] = tolower(s[idx]);
        }
    }
    vector<string> letterCasePermutation(string S) {
        if(S.size() == 0) return {""};
        vector<string> res;
        for(int i = 0; i < S.size(); i++)
            S[i] = tolower(S[i]);
        helper_bk(S, 0, res);
        return res;
    }
};
