class Solution {
private:
    void helper(string S, int idx, vector<string> &res) {
        if (idx == S.size()) {
            res.push_back(S);
            return;
        }
        if (isalpha(S[idx])) {
            S[idx] = tolower(S[idx]);
            helper(S, idx + 1, res);
            S[idx] = toupper(S[idx]);
            helper(S, idx + 1, res);
        } else {
            helper(S, idx + 1, res);
        }
    }
    
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> res;
        helper(S, 0, res);
        return res;
    }
};
