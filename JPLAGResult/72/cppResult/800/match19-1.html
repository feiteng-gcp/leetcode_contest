class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> res;
        dfs(S, 0, res);
        return res;
    }

    void dfs(string& S, int cur, vector<string>& res){
        locale loc;
        if(cur == S.size()){
            res.push_back(S);
            return;
        }
        if(isalpha(S[cur])){
            string next = S.substr(0,cur)+toupper(S[cur], loc)+S.substr(cur+1,S.size()-cur-1);
            dfs(next, cur+1, res);
            next = S.substr(0,cur)+tolower(S[cur], loc)+S.substr(cur+1,S.size()-cur-1);
            dfs(next, cur+1, res);
        }
        else{
            dfs(S, cur+1, res);
        }
    }
};
