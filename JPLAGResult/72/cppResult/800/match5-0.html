class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> res;

        work(S, 0, res);
        return res;
    }
    
    void work(string S, int p, vector<string> &res) {
        if (p == S.size()) {
            res.push_back(S);
            return;
        }
        
        work(S, p+1, res);
        if (S[p] >= 'a' && S[p] <= 'z') {
            S[p] = S[p] - 32;
            work(S, p+1, res);
        } else if (S[p] >= 'A' && S[p] <= 'Z') {
            S[p] = S[p] + 32;
            work(S, p+1, res);
        }
    }
};
