class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> ans;
        build(S, 0, ans);
        return ans;
    }
    
    void build(string& S, int i, vector<string>& ans) {
        if (i == S.size()) {
            ans.push_back(S);
            return;
        }
        
        build(S, i + 1, ans);
        if (!isAlpha(S[i])) {
            return;
        }
        
        if (isLower(S[i])) {
            S[i] = S[i] - 'a' + 'A';
        } else {
            S[i] = S[i] - 'A' + 'a';
        }
        build(S, i + 1, ans);
    }
    
    inline bool isLower(const char c) {
        return c >= 'a' && c <= 'z';
    }
    
    inline bool isUpper(const char c) {
        return c >= 'A' && c <= 'Z';
    }
    
    inline bool isAlpha(const char c) {
        return isLower(c) || isUpper(c);
    }
};
