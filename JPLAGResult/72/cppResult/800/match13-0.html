class Solution {
public:
    vector <string> res;
    
    void helper(string s, int index) {
        if (index >= s.size()) {
            res.push_back(s);
            return;
        }
        if (isalpha(s[index])) {
            s[index] = toupper(s[index]);
            helper(s, index + 1);
            s[index] = tolower(s[index]);
            helper(s, index + 1);
        } else helper(s, index + 1);
    }
    
    
    vector<string> letterCasePermutation(string s) {
        helper(s, 0);
        return res;
    }
};
