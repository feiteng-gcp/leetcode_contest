class Solution {
public:
vector<string> res;
string s;
int N;

void recur(int i){
  if(i==N){
    res.push_back(s);
    return;
  }
  if(isalpha(s[i])){
    s[i]=tolower(s[i]);
    recur(i+1);
    s[i]=toupper(s[i]);
    recur(i+1);
  }else recur(i+1);
}
vector<string> letterCasePermutation(string S) {
  N=S.size();
  s=S;
  recur(0);
  return res;
}


};
