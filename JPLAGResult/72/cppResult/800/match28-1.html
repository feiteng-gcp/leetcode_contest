class Solution {
    vector<string> ans;
public:
    vector<string> letterCasePermutation(string S) {
        impl(S, 0);
        return ans;
    }

    void impl(string S, int index) {
        if (index == S.length()) {
            ans.push_back(S);
            return ;
        }
        int i;
        for (i = index; i < S.length(); i++) {
            if (isalpha(S[i])) {
                S[i] = toupper(S[i]);
                impl(S, i + 1);
                S[i] = tolower(S[i]);
                impl(S, i + 1);
                break;
            }
        }
        if (i == S.length())
            ans.push_back(S);
    }
};
