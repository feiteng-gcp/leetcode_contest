class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> sol;
        permute(S,0,sol);
        return sol;
    }
    
    void permute(string S, int x, vector<string>& sol){
        if(x==S.length()) {
            sol.push_back(S);
            return;
        }
        
        if(isNumber(S[x])){
            permute(S,x+1,sol);
        }
        else{
            permute(S,x+1,sol);
            if(isLowerCase(S[x])) S[x] = S[x]-'a'+'A';
            else S[x] = S[x]-'A'+'a';
            permute(S,x+1,sol);
        }
    }
    
    bool isNumber(char ch){
        return (ch>='0' and ch<='9');
    }
    
    bool isLowerCase(char ch){
        return (ch>='a' and ch<='z');
    }
};
