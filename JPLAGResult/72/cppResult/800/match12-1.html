class Solution {
    vector<string>ans;

    void recu(string &s, string &temp, int curr)
    {
        if(curr == s.size())
        {
            ans.push_back(temp);
            return ;
        }
        
        if(isalpha(s[curr]))
        {
            temp.push_back(tolower(s[curr]));
            recu(s, temp, curr+1);
            temp.pop_back();
            
            temp.push_back(toupper(s[curr]));
            recu(s, temp, curr+1);
            temp.pop_back();
        }
        else
        {
            temp.push_back(s[curr]);
            recu(s, temp, curr+1);
            temp.pop_back();
        }
    }
public:
    vector<string> letterCasePermutation(string S) {
        ans.clear();
        string temp;
        recu(S, temp, 0);
        return ans;
    }
};
