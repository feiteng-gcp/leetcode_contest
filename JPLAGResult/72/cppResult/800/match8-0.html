class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> res;
        helper(S, 0, res);
        return res;
    }
    void helper(string S, int start, vector<string>& res) {
        if (start >= S.size()) {
            res.push_back(S);
            return;
        }
        helper(S, start + 1, res);    
        if (S[start] >= 'a' && S[start] <= 'z') {
            string t = S;
            t[start] = S[start] - 'a' + 'A';
            helper(t, start + 1, res);
        } else if (S[start] >= 'A' && S[start] <= 'Z') {
            string t = S;
            t[start] = S[start] - 'A' + 'a';
            helper(t, start + 1, res);
        }
    }
};
