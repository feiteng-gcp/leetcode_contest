class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> res;
        // if(!S.size())
        //     return res;
        string tuple = "";
        
        helper(res, S, tuple, 0);
            
        return res;
    }
    void helper(vector<string> &res, string &S, string &tuple, int pos) {
        if(pos == S.size()) {
            res.push_back(tuple);
            return;
        }
        
        
        if(islower(S[pos])) {
            tuple += S[pos];
            helper(res, S, tuple, pos+1);
            tuple.pop_back();
            tuple += toupper(S[pos]);
            helper(res, S, tuple, pos+1);
            tuple.pop_back();
        }
        else if(isupper(S[pos])) {
            tuple += S[pos];
            helper(res, S, tuple, pos+1);
            tuple.pop_back();
            tuple += tolower(S[pos]);
            helper(res, S, tuple, pos+1);
            tuple.pop_back();
        }
        else {
            tuple += S[pos];
            helper(res, S, tuple, pos+1);
            tuple.pop_back();
        }
            
    }
};
