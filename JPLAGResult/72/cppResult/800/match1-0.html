class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> res;
        dfs(S, 0, res);
        return res;
    }
    
    void dfs(string &s, int i, vector<string>& res){
        if(i>=s.size()){
            res.push_back(s);
            return;
        }
        int gap = 'a' - 'A';
        dfs(s, i+1, res);
        if(s[i]>='a'&&s[i]<='z'){
            s[i] -= gap;
            dfs(s, i+1, res);
            s[i]+=gap;
        }
        else if(s[i]>='A'&&s[i]<='Z'){
            s[i] += gap;
            dfs(s, i+1, res);
            s[i] -= gap;
        }
    }
};
