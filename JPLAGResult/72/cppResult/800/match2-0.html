class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> results;
        DFS(results, S, 0);
        return results;
    }
    
    void DFS(vector<string>& results, string S,int k) {
        if (k == S.length()) {
            //cout << "got it" << endl;
            results.push_back(S);
            return;
        }
        //cout << "hi" << " " << k << endl;
        int i = k;
        if (S[i] >= 'a' && S[i] <= 'z') {
            S[i] += ( 'A' - 'a' );
            DFS(results, S, k + 1);
            S[i] -= ( 'A' - 'a' );

        } else if (S[i] >= 'A' && S[i] <= 'Z') {
            S[i] += ('a' - 'A' );
            DFS(results, S, k + 1);
            S[i] -= ('a' - 'A' );
        }
        DFS(results, S, k + 1);
    }
};
