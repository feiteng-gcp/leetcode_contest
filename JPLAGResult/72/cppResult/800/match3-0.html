class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        vector<string> res;
        dfs(S, 0 , res);
        return res;
    }
    void dfs(string s, int index, vector<string> &res) {
        if (index == s.size()) {
            res.push_back(s);
            return;
        }
        dfs(s, index + 1, res);
        if ('a' <= s[index] && s[index] <= 'z') {
            s[index] = s[index] + 'A' - 'a';
            dfs(s, index + 1, res);
            s[index] = s[index] + 'a' - 'A';
        } else if ('A' <= s[index] && s[index] <= 'Z') {
            s[index] = s[index] + 'a' - 'A';
            dfs(s, index + 1, res);
            s[index] = s[index] + 'A' - 'a';
        }
    }
};
