class Solution {
    vector<string> result;
public:
    void solve(string S, int i){
        if (i==S.length()) {
            result.push_back(move(S));
            return;
        }
        if (isalpha(S[i])) {
            S[i] = tolower(S[i]);
            solve(S, i+1);
            S[i] = toupper(S[i]);
            solve(S, i+1);
        } else {
            solve(S, i+1);
        }
    }

    vector<string> letterCasePermutation(string S) {
        solve(S, 0);
        return move(result);
    }
};
