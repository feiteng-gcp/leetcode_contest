class Solution {
public:
    vector<string> letterCasePermutation(string S) {
        for(int i=0;i<S.size();i++){
            if(S[i]>='a' && S[i]<='z'){
                S[i]-=32;    
            }
        }
        vector<string> res;
        dfs(S, res, 0);
        return res;
    }
    void dfs(string& S, vector<string>& res, int pos){
        if(pos == S.size()){
            res.push_back(S);
            return;
        }
        
        if(S[pos]>='A' && S[pos]<='Z'){
            dfs(S, res, pos+1);
            S[pos]+=32;
            dfs(S, res, pos+1);
            S[pos]-=32;
        }
        else dfs(S, res, pos+1);
    }
};
