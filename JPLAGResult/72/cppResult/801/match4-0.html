class Solution {
public:
    bool helper_DFS(int node, vector<vector<int>>& graph, int part, vector<int>& partition) {
        //cout << "node: " << node << " part: " << part << " partition[node]: " << partition[node] << endl;
        if(partition[node] != -1) {
            if(partition[node] != part) return false;
            else return true;
        }
        //cout << "****node: " << node << endl;
        partition[node] = part;
        for(int i = 0; i < graph[node].size(); i++) {
            if(!helper_DFS(graph[node][i], graph, part^1, partition)) return false;
        }
        return true;
    }
    bool isBipartite(vector<vector<int>>& graph) {
        int n = graph.size();
        vector<int> partition(n, -1);
        for(int i = 0; i < n; i++) {
            if(partition[i] != -1) continue;
            if(!helper_DFS(i, graph, 0, partition)) return false;
        }
        return true;
    }
};
