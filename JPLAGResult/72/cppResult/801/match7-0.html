class Solution {
public:
    bool isBipartite(vector<vector<int>>& graph) {
        set<int> par1, par2;
        int n = graph.size();
        vector<bool> visited(n, false);
        for(int i = 0; i < n; ++i) {
            if(visited[i]) continue;
            if(!dfs(par1, par2, graph, i, n, true, visited)) return false;
        }
        return true;
    }
    
    bool dfs(set<int>& par1, set<int>& par2, vector<vector<int>>& graph, int cur, int n, bool first, vector<bool>& visited) {
        if(visited[cur]) return true;
        visited[cur] = true;
        if(first) {
            if(par2.find(cur) != par2.end()) return false;
            par1.insert(cur);
            for(int i = 0; i < graph[cur].size(); ++i) {
                int next = graph[cur][i];
                if(par1.find(next) != par1.end()) return false;
                if(!dfs(par1, par2, graph, next, n, false, visited)) return false;
            }
        } else {
            if(par1.find(cur) != par1.end()) return false;
            par2.insert(cur);
            for(int i = 0; i < graph[cur].size(); ++i) {
                int next = graph[cur][i];
                if(par2.find(next) != par2.end()) return false;
                if(!dfs(par1, par2, graph, next, n, true, visited)) return false;
            }
        }
        return true;
    }
};
