class Solution {
public:
    bool isBipartite(vector<vector<int>>& graph) {
        vector<char> col(graph.size(), -1);
        for (int i = 0; i < graph.size(); i++) if (col[i] == -1) {
            queue<int> q;
            col[i] = 0;
            q.push(i);
            
            while(!q.empty()) {

                int u = q.front();
                q.pop();
                
                for (int j = 0; j < graph[u].size(); j++) {
                    int v = graph[u][j];
                    if (col[v] >= 0 && col[v] == col[u])
                        return false;
                    if (col[v] == -1)
                        q.push(v);
                    col[v] = 1 - col[u];
                    
                }
            }
        }
        return true;
    }
};
