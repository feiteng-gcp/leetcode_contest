class Solution {
public:
    bool isBipartite(vector<vector<int>>& graph) {
        vector<bool> visited(graph.size(), false);
        vector<int> color(graph.size(), 0);
        queue<int> q;
        for(int i = 0; i < graph.size(); i++){
            if(!visited[i]){
                visited[i] = true;
                color[i] = 1;
                q.push(i);
                int c = 1;
                while(!q.empty()){
                    int cur = q.front();

                    c = 1-color[cur];
                    visited[cur] = true;
                    q.pop();
                    for(int j = 0; j < graph[cur].size(); j++){
                        int next = graph[cur][j];
                        if(!visited[next]){
                            color[next] = c;
                            q.push(next);
                        }
                        else if(color[next] != c){
                            return false;
                        }
                    }
                }
            }
        }
        return true;
    }
};
