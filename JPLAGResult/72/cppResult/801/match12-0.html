class Solution {
public:
    bool isBipartite(vector<vector<int>>& graph) {
        int l = graph.size();
        vector<int> res(l, 0);
        
        for (int i = 0; i < l; ++i) {
            if (res[i]) continue;
            
            vector<int> list;
            list.push_back(i);
            res[i] = 1;
            for (int j = 0; j < list.size(); ++j) {
                int now = list[j];
                for (int k = 0; k < graph[now].size(); ++k) {
                    int target = graph[now][k];
                    if (res[target] == res[now]) return false;
                    if (res[target] == 0) {
                        res[target] = 3 - res[now];
                        list.push_back(target);
                    }
                }
            }
        }
        return true;
    }
};
