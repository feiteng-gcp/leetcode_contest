class Solution {
public:
set<int> s[2];
bool fll(vector<vector<int>>& g,int v,int i){
  if(s[1-i].count(v)) return 0;
  s[i].insert(v);
  for(int w:g[v]){
    if(!s[1-i].count(w)) {
      if(!fll(g,w,1-i)) return 0; 
    }
  }
  return 1;
}
bool isBipartite(vector<vector<int>>& graph) {

  for(int i=0;i<graph.size();i++){
    if(!s[0].count(i) && !s[1].count(i)){
      if(!fll(graph,i,0)) return 0;
    }
  }
// cout<<s[0].size()<<' '<<s[1].size()<<'\n';
  return 1;
}


};
