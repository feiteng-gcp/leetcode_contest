class Solution {
public:
int color[105];
int V;

vector<vector<int> > G;
bool dfs(int v,int c)
{
    color[v] = c;
    for (int i = 0; i < G[v].size(); i++)
    {
        if (color[G[v][i]] == c)
            return false;
        if (color[G[v][i]] == 0 && !dfs(G[v][i], -c))
            return false;
    }
    return true;
}

bool solve()
{

    for (int i = 0; i < V; i++)
    {
        if (color[i] == 0)
        {
            if (!dfs(i, 1))
            {
                return false;
            }
        }
    }
    return true;
}
    bool isBipartite(vector<vector<int>>& graph) {
        G=graph;
        V=graph.size();
        memset(color,0,sizeof(color));
        return solve();
    }
};
