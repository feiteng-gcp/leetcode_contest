#include <bits/stdc++.h>

using namespace std;

#define VERA(i, a, b) for(ll i=(ll)(a);i<(ll)(b);++i)
#define ANDY(i, a, b) for(ll i=(ll)(a);i>(ll)(b);--i)
#define RARA(x, seq) for(auto& x:seq)
#define SIZE(x) ((ll)(x.size()))

typedef long long ll;
typedef long double dd;

class Solution {
 public:
  bool isBipartite(vector<vector<int>>& graph) {
    const ll n = SIZE(graph);
    vector<ll> col(n, -1);
    VERA(st, 0, n) {
      if (~col[st]) continue;
      col[st] = 0;
      queue<ll> que;
      que.push(st);
      while (not que.empty()) {

        const ll at = que.front();
        que.pop();
        RARA(to, graph[at]) {
          if (col[to] == -1) {
            col[to] = col[at] ^ 1;
            que.push(to);
          } else if (col[to] == col[at])
            return false;
        }
      }
    }
    return true;
  }
};
