class Solution {
public:
            int deg[105];
	int col[105];
	int que[105];
    bool isBipartite(vector<vector<int>>& graph) {
	int n=graph.size();
	for(int i=0;i<n;i++)
	{
		deg[i]=graph[i].size();
	}
	memset(col,0,sizeof(col));
	int front=0,rear=0;
	que[rear++]=0;
	col[0]=1;
	while(front<rear)
	{
		while(front<rear)
		{
			int now=que[front++];
			for(int i=0;i<deg[now];i++)
			{
				int noww=graph[now][i];
				if(!col[noww])
				{
					col[noww]=3-col[now];
					que[rear++]=noww;
				}
			}
		}
		for(int i=0;i<n;i++)
		{
			if(!col[i])
			{
				que[rear++]=i;
                col[i]=1;
				break;
			}
		}
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<deg[i];j++)
		{
			if(col[i]==col[graph[i][j]])return false;
		}
	}
	return true;
    }
};
