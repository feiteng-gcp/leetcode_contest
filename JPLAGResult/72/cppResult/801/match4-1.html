class Solution {
public:
    bool dfs(vector<int>&mark, vector<vector<int>> &g, int pos, int val) {
        mark[pos] = val;
        for(auto i: g[pos]) {
            if (mark[i] != 0 && mark[i] == val) return false;
            if (mark[i] != 0 && mark[i] != val) continue;

            if (!dfs(mark, g, i, val == 1 ? 2 : 1)) return false;
        }
        return true;
    }
    bool isBipartite(vector<vector<int>>& g) {
        int n = g.size();
        vector<int> mark(n, 0);
        for(int i=0; i<n; ++i) {
            if (mark[i] != 0) continue;
            if (!dfs(mark, g, i, 1)) return false;
        }
        return true;
    }
};
