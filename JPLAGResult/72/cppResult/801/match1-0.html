class Solution {
public:
    bool go(int p, vector<vector<bool>> &g, vector<int> &clr){
        queue <int> q;
        q.push(p);
        clr[p] = 1;
        
        while(!q.empty()){
            int u = q.front(); q.pop();
            if(g[u][u] == 1) return false;
            for(int v=0;v<g.size();++v){
                if(g[u][v] && clr[v] < 0){
                    clr[v] = 1 - clr[u];
                    q.push(v);
                }
                else if(g[u][v] && clr[v] == clr[u])
                    return false;
            }
        }
        return true;
    }
    bool check(vector<vector<bool>> &g){
        vector<int> clr(g.size(), -1);
        for(int i=0;i<g.size();++i)
            if(clr[i] < 0 && !go(i, g, clr))
                return false;
        return true;
    }
    bool isBipartite(vector<vector<int>>& graph) {
        vector<vector<bool>> g(graph.size(), vector<bool>(graph.size(), false));
        for(int i=0;i<graph.size();++i)
            for(int j=0;j<graph[i].size();++j)
                g[i][graph[i][j]] = g[graph[i][j]][i] = true;
        return check(g);
    }
};
