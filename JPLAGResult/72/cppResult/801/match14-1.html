class Solution {
public:
    bool isBipartite(vector<vector<int>>& graph) {
        vector<char> colors(graph.size(), 0);

        for (int i = 0; i < graph.size(); ++i) {
            if (colors[i] == 0) {
                if (!paint(graph, colors, i, 1)) {
                    return false;
                }
            }
        }
        return true;
    }
    
    bool paint(vector<vector<int>>& graph, vector<char>& colors, int i, int c) {
        if (colors[i] != 0) {
            return colors[i] == c;
        }
        
        colors[i] = c;
        for (int j : graph[i]) {
            if (!paint(graph, colors, j, -c)) {
                return false;
            }
        }
        return true;
    }
};
