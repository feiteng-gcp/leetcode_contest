class Solution {
    vector<int> color;
    int n;
    bool dfs(int cur, int c, vector<vector<int>>& graph) {
        color[cur] = c;
        for (int nxt : graph[cur]) {
            if (color[nxt] == c) return false;
            if (color[nxt] == -1) {
                if (!dfs(nxt, 1-c, graph)) return false;
            }
        }
        return true;
    }
public:
    bool isBipartite(vector<vector<int>>& graph) {
        n = graph.size();
        color = vector<int>(n, -1);
        for (int i = 0; i < n; i++) {
            if (color[i] == -1) {
                if (!dfs(i, 0, graph)) return false;
            }
        }
        return true;
        
    }
};
