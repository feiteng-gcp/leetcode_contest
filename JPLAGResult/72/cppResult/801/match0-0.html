class Solution {
public:
    bool isBipartite(vector<vector<int>>& graph) {
        int n = graph.size();
        bool check = false;
        int val = 0;
        vector <int> vec(n, -1);
        queue<int> q;
        
        for (int i = 0; i < graph.size(); ++i) {
            if (vec[i] != -1) continue;
            q.push(i);
            vec[i] = 5;
            while (!q.empty()) {
                int u = q.front();
                q.pop();
                for (int i = 0; i < graph[u].size(); ++i) {
                    if (vec[graph[u][i]] == -1) {
                        vec[graph[u][i]] = -vec[u];
                        q.push(graph[u][i]);
                    } else if (vec[graph[u][i]] == vec[u]) return false;
                }
            }
            
        }
        
        return true;
    }
};
