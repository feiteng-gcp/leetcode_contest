class Solution {
public:
    bool isBipartite(vector<vector<int> >& graph)
{
    int len = (int)graph.size();
    vector<int> id(len, 0);
    for(int i = 0; i < len; i++)
    {
        if(id[i]) continue;
        queue<int> q;
        q.push(i);
        id[i] = 1;
        while(!q.empty())
        {
            int cur = q.front(); q.pop();
            int pid = 3 - id[cur];
            for(int j = 0; j < graph[cur].size(); j++)
            {
                int nxt = graph[cur][j];
                if(id[nxt] && id[nxt] != pid) return false;
                if(!id[nxt]) q.push(nxt);
                id[nxt] = pid;
            }
        }
    }
    return true;
}
};
