class Solution {
public:
    bool isBipartite(vector<vector<int>>& graph) {
        int N = graph.size();
        unordered_map<int, int> group;
        for (int i = 0; i < N; ++i) {
            if (group.find(i) == group.end()) {
                // BFS from i
                vector<int> level = {i};
                group[i] = 0;
                int n_level = 0;
                while (!level.empty()) {
                    n_level = 1 - n_level;
                    vector<int> new_level;
                    for (int x : level) {
                        for (int y : graph[x]) {
                            if (group.find(y) != group.end() && group[y] == 1 - n_level) {
                                return false;
                            }
                            if (group.find(y) == group.end()) {
                                group[y] = n_level;
                                new_level.push_back(y);
                            }                            
                        }
                    }
                    level = new_level;
                }
            }
        }
        return true;
    }
};
