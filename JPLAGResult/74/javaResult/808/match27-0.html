import java.util.Arrays;

class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int n = words.length;
        int m = S.length();
        int[][] next = new int[m + 1][26];
        for (int[] d : next) {
        	Arrays.fill(d, -1);
        }
        for (int i = m - 1; i >= 0; --i) {
        	for (int j = 0; j < 26; ++j) {
        		next[i][j] = next[i + 1][j];
        	}
        	next[i][S.charAt(i) - 'a'] = i;
        }
        int res = 0;
        for (int i = 0; i < n; ++i) {
        	int index = 0;
        	boolean good = true;
        	for (int j = 0; j < words[i].length(); ++j) {
        		int x = words[i].charAt(j) - 'a';
        		if (next[index][x] == -1) {
        			good = false;
        			break;
        		} else {
        			index = next[index][x] + 1;
        		}
        	}
        	if (good) ++res;
        }
        return res;
    }
}
