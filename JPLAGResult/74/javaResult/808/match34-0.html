class Solution {
    public int numMatchingSubseq(String S, String[] words) {
		
		int count = 0;
		for(String word : words) {
			if(word.length() > S.length())
				continue;
			int fromIndex = -1;
			boolean stat = true;
			for(char ch : word.toCharArray()) {
				fromIndex = S.indexOf(ch, fromIndex + 1);
				if(fromIndex == -1) {
					stat = false;
					break;
				}
			}
			if(stat)
				count++;
		}
		return count;
	}
}
