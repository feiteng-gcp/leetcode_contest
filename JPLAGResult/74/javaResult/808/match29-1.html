class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int[][] nextOccur = new int[S.length()][26];
        int[] next = new int[26];
        for (int i = 0; i < 26; i++) next[i] = S.length();
        for (int c = S.length() - 1; c >= 0; c--) {
            next[S.charAt(c) - 'a'] = c;
            for (int j = 0; j < 26; j++) nextOccur[c][j] = next[j];
        }
        
int ans = 0;
        for (String word : words) {
            ans++;
            int where = 0;
            for (char c : word.toCharArray()) {
                if (where >= S.length()) {
                    ans--;
                    break;
                }
                where = nextOccur[where][c - 'a'];
                if (where >= S.length()) {
                    ans--;
                    break;
                
}
                where++;
            }
        }
        return ans;
    }
}
