class Solution {
    
    private static boolean isSubSequence(char[] sChars, 
String word) {
        char[] chars = word.toCharArray();
        if (chars.length == 0) {
            return false;
        }
        int index = 0;
        for (char c : chars) {
            while (true) {
                if (index >= sChars.length) {
                    return false;
                }
                if (c == sChars[index++]) {
                    break;
                }
            }
        
}
        return true;
    }
    
    public int numMatchingSubseq(String S, String[] words) {
        char[] sChars = S.toCharArray();
        return (int) Arrays.stream(words).parallel()
                .filter(word -> isSubSequence(sChars, word))
                .count();
    }
}
