class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        List<Integer>[] count = new ArrayList[26];

        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            if (count[c - 'a'] == null) {
                count[c - 'a'] = new ArrayList<>();
            }

            count[c - 'a'].add(i);
        }

        
int total = 0;

        for (String word : words) {
            if (isSub(count, word)) {
                total++;
                //System.err.println(word);
            }
        }
        return total;
    }

    boolean isSub(List<Integer>[] count, String word) {
        int[] index = new int[26];
        Arrays.fill(index, -1);

        int pre = -1;
        int i = 0;
        while (i < word.length()) {
            char c = word.charAt(i);
            int in = index[c - 'a'];

            if (count[c - 'a'] == null || count[c - 'a'].size() <= in + 1) {
                return false;
            }

            int cur = count[c - 'a'].get(in + 1);
            index[c - 'a']++;

            if (cur > pre) {
                i++;
                pre = cur;
            }

        }

        return true;
    }
}
