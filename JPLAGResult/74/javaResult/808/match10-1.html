class Solution {
public int numMatchingSubseq(String S, String[] words) {
        Map<Character, TreeSet<Integer>> tree = new HashMap<>();
        gao(S, tree);
        int ans = 0;
        for (String word : words) {
            if (ok(word, tree)) ans++
;
        }
        return ans;
    }
    boolean ok(String word, Map<Character, TreeSet<Integer>> tree) {
        int pos = -1;
        for (char ch : word.toCharArray()) {
            TreeSet<Integer> set = tree.get(ch);
            if (set == null) return false;
            
Integer next = set.higher(pos);
            if (next == null) return false;
            pos = next;
        }
        return true;
    }
    void gao(String s, Map<Character, TreeSet<Integer>> tree) {
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            TreeSet<Integer> set = tree.getOrDefault(ch, new TreeSet<>());
            set.add(i);
            tree.putIfAbsent(ch, set);
        }
    }
}
