class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int total = 0;
        Map<String, Boolean> dp = new HashMap<>();
        for (String word : words){
            if (isSubsequence(word, S, dp)) total++
;
        }
        return total;
    }
    private boolean isSubsequence(String word, String s, Map<String, Boolean> dp){
        if (word.length() > s.length()) return false;
        if (dp.containsKey(word)) return dp.get(word);
        boolean result = false;
        
int i = 0, j = 0;
        while (j < s.length() && i < word.length()){
            if (s.charAt(j++) == word.charAt(i)) i++;
        }
        if (i == word.length()) result = true;
        dp.put(word, result);
        return result;
    }
}
