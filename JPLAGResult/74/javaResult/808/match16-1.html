import java.util.*;

class Solution {
    public int numMatchingSubseq(String S, String[] words) {

        int[] lookingFor = new int[words.length]; // inits 0
        
        
Map<Character, List<Integer>> lookingForCharacter = new HashMap<Character, List<Integer>>();
        for (char c = 'a'; c <= 'z'; c++) {
            lookingForCharacter.put(c, new ArrayList<Integer>());
        }
        for (int wordIdx = 0; wordIdx < words.length; wordIdx++) {
            String word = words[wordIdx];
            char needFirst = word.charAt(0);
            lookingForCharacter.get(needFirst).add(wordIdx);
        
}

        int sum = 0;
        
        for (int pos = 0; pos < S.length(); pos++) {

            char c = S.charAt(pos);

            List<Integer> wordsNeedingThisChar = lookingForCharacter.get(c);
            List<Integer> wordsNeedingThisCharNext = new ArrayList<Integer>();
            
            for(int wordIdx : wordsNeedingThisChar) {
                
                // last character that was needed by this word?
                if (lookingFor[wordIdx] == words[wordIdx].length()-1) { sum++; }
                else {  // the word still needs at least one more character coming later
                    int charNextWantIdx = ++lookingFor[wordIdx];
                    
char charNextWant = words[wordIdx].charAt(charNextWantIdx);
                    if (charNextWant == c) {
                        wordsNeedingThisCharNext.add(wordIdx);
                    }
                    else {
                        lookingForCharacter.get(charNextWant).add(wordIdx);
                    }
                }
            }
            lookingForCharacter.put(c, wordsNeedingThisCharNext);

        }
        return sum;
    }
}
