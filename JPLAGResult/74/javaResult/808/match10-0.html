class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        
        int count = 0;
        for (int i = 0; i < words.length; i++) {
            if (subseq(words[i], S)) {
                count ++;
            }
        }
        return count;
    }
    
    private static boolean subseq(String word, String S) {
        char[] chars = word.toCharArray();
        int i = 0;
        for (int j = 0; j < chars.length; j++) {
            int idx = S.indexOf(chars[j], i);
            if (idx == -1)
                return false;
            i = idx + 1;
        }
        return true;
    }
}
