class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        Map<Character, TreeSet<Integer>> map = new HashMap<>();
        int m = S.length();
        for (int i = 0; i < m; i++) {
            char c = S.charAt(i);
            if (!map.containsKey(c)) {
                map.put(c, new TreeSet<>());
            }
            map.get(c).add(i);
        }
        
        int res = 0;
        
        for (String word: words) {
            Integer i = -1;
            boolean flag = true;
            for (char c: word.toCharArray()) {
                if (!map.containsKey(c)) {
                    flag = false;
                    break;
                }
                i = map.get(c).higher(i);
                if (i == null) {
                    flag = false;
                    break;
                }
            }
            
            if (flag) res++;
            
        }
        
        return res;
    }
}
