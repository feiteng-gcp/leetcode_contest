class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int c = 0;
        for (String w: words) {
            if (isSubseq(S, w)) {
                ++c;
            }
        }
        return c;
    }
    
    public boolean isSubseq(String s, String p) {
        int i = 0, j = 0;
        while (i < s.length() && j < p.length()) {
            if (s.charAt(i) == p.charAt(j)) {
                ++i;
                ++j;
            } else {
                ++i;
            }
        }
        
        return j == p.length();
    }
}
