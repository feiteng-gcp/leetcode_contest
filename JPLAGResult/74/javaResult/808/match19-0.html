class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int count = 0;
        for (String seq : words)
            if (isSubSeq(S, seq))
                count++;
        return count;
    }
    
    boolean isSubSeq(String word, String seq) {
        int pos = -1;
        for (char c : seq.toCharArray()) {
            pos = word.indexOf(c, pos + 1);
            if (pos < 0)
                return false;
        }
        return true;
    }
    
}
