class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        //List<String> dic = new ArrayList<>();
        // findSub(dic, S, "", 0);
        
        int count = 0;
        for (String str : words) {
            if (findSub(S, str)) count++;
        }
        return count;
    }
    
    public boolean findSub(String a, String b) {
        int cur = 0;
        for (Character chr : b.toCharArray()) {
            int newc = a.indexOf(chr, cur);
            if (newc == -1) return false;
            cur = newc + 1;
        } 
        return true;
    }
