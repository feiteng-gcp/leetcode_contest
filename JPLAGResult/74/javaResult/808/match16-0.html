class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        List<List<Pair>> p = new ArrayList();
        for (int i = 0; i < 26; ++i) {
            p.add(new ArrayList<Pair>());
        }
        for (int i = 0; i < words.length; ++i) {
            p.get(words[i].charAt(0) - 'a').add(new Pair(0, words[i]));
        }
        int result = 0;
        for (int i = 0; i < S.length(); ++i) {
            int target = S.charAt(i) - 'a';
            List<Pair> pt = new ArrayList();
            for (int j = 0; j < p.get(target).size(); ++j) {
                Pair pair = p.get(target).get(j);
                int newid = pair.index + 1;
                if (newid == pair.s.length()) {
                    result++;
                } else {
                    int newtarget = pair.s.charAt(newid) - 'a';
                    pair.index++;
                    if (newtarget == target) {
                        pt.add(pair);
                    } else {
                      p.get(newtarget).add(pair);
                    }
                }
            }
            p.get(target).clear();
            p.get(target).addAll(pt);
        }
        return result;
    }
    
    public static class Pair {
        public int index;
        public String s;
        public Pair(int id, String s) {
            this.index = id;
            this.s = s;
        }
    }
}
