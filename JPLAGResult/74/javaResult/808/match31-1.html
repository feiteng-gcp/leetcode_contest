class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        
        int L = S.length();
        
        int[][] dp = new int[L+1][26];
        
        for (int i=0; i<26; i++) {
            dp[L][i] = -1;
        }
        
        for (int i=L-1; i>=0; i--) {
            dp[i] = Arrays.copyOf(dp[i+1], 26);
            dp[i][S.charAt(i) - 'a'] = i;
        }
        
        
int ct = 0;
        
//         for (int i=0; i<dp.length; i++) {
//             for(int j=0; j<dp[0].length; j++) {
//                 System.out.print(dp[i][j] + ", ");
//             }
            
//             System.out.println("");
//         }
        for (String word : words) {
            if (isSub(dp, S, word)) {
                ct ++;
            }
        }
        return ct;
        
    }
    
    private boolean isSub(int[][] dp, String S, String word) {
        int i = 0;
        int j=0;
        while (j < word.length()) {
            i = dp[i][word.charAt(j) - 'a'];
            if (i < 0) {
                return false;
            }
            i++;
            j++;
        }
        
        return true;
    }
}
