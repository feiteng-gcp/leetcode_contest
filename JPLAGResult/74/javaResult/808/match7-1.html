class Solution {
    public int numMatchingSubseq(String s, String[] words) {
        int count = 0;
        for(String word : words){
            if(isSubsequence(word, s)){
                count++;
            
}
        }
        return count;
    }
    
    public boolean isSubsequence(String s, String t) {
        
int p1 = 0;
        int p2 = 0;
        while(p1 < s.length() && p2 < t.length()){
            if(s.charAt(p1) == t.charAt(p2)){
                p1++;
                p2++;
            }else{
                p2++;
            }
        }
        return p1 == s.length();
    }
}
