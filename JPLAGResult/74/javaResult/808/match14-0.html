class Solution {
	private List<Integer>[] lists = new ArrayList[26];
	
    public int numMatchingSubseq(String S, String[] words) {
        
    	int i , length = S.length();
    	for (i = 0;i < 26;i ++) {
    		lists[i] = new ArrayList<>();
    	}
    	for (i = 0;i < length;i ++) {
    		int index = S.charAt(i) - 'a';
    		lists[index].add(i);
    	}
    	int ans = 0;
    	for (String word : words) {
    		if (check(word)) {
    			ans ++;
    		}
    	}
    	return ans;
    	
    }
    
    private boolean check(String word) {
    	
    	int i , length = word.length() , prev = 0;
    	for (i = 0;i < length;i ++) {
    		int index = word.charAt(i) - 'a';
    		int pos = lowerBound(index , prev);
    		if (pos >= 0) {
    			prev = pos + 1;
    		} else {
    			return false;
    		}
    	}
    	return true;
    	
    }
    
    private int lowerBound(int index , int prev) {
    	
    	int low = 0 , high = lists[index].size();
    	while (low < high) {
    		int mid = (low + high) / 2;
    		if (lists[index].get(mid) < prev) {
    			low = mid + 1;
    		} else {
    			high = mid;
    		}
    	}
    	if (low < lists[index].size() && lists[index].get(low) >= prev) {
    		return lists[index].get(low);
    	} else {
    		return - 1;
    	}
    	
    }
}
