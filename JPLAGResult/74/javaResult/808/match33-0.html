class Solution
{
    HashMap<Character, ArrayList<Integer>> map;
    public int binary_search(ArrayList<Integer> list, int target)
    {
        int start = 0;
        int end = list.size() - 1;
        while(start <= end)
        {
            int mid = (start + end)/2;
            if(target == list.get(mid)) return list.get(mid);
            else if(target > list.get(mid)) start = mid + 1;
            else end = mid - 1;
        }
        return list.get(start);
    }
    
    
    
    public boolean helper(String word)
    {
        if(word.length() == 0) return true;
        if(!map.containsKey(word.charAt(0)) || map.get(word.charAt(0)).size() == 0) return false;
        int index = map.get(word.charAt(0)).get(0)+1;
        for(int i = 1; i < word.length(); i++)
        {
            //System.out.println(index);
            ArrayList<Integer> list = map.get(word.charAt(i));
            if(list.size() == 0 || index > list.get(list.size()-1)) return false;
            index = binary_search(list, index)+1;
        }
        return true;
    }
    
    
    
    public int numMatchingSubseq(String S, String[] words) 
    {
        map = new HashMap<Character, ArrayList<Integer>>();
        for(char i = 'a'; i <= 'z'; i++)
        {
            map.put(i, new ArrayList<Integer>());
        }
        for(int i = 0; i < S.length(); i++)
        {
            map.get(S.charAt(i)).add(i);
        }
        
        
        int res = 0;
        for(String word:words)
        {
            
            if(helper(word)) res++;
        }
        return res;
        
        
    }
}
