class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        
        Map<Character, List<Integer>> map = new HashMap<>();
        
        for (int i = 0; i < S.length(); i++) {
            char cur = S.charAt(i);
            if (!map.containsKey(cur)) {
                List<Integer> list = new LinkedList<>();
                list.add(i);
                map.put(cur, list);
            } else {
                List<Integer> list = 
map.get(cur);
                list.add(i);
            }
        }
        
        int total = 0;
        
        for (String word : words) {
            int idx = 0;
            int m = -1;
            while (idx < word.length()) {
                boolean flag = true;
                if (!map.containsKey(word.charAt(idx))) {
                    break;
                } else {
                    List<Integer> list = map.get(word.charAt(idx));
                    for (int i = 0; i < list.size(); i++) {
                        if (list.get(i) > m) {
                            //System.out.println(list.get(i) + ", "+ m);
                            m = list.get(i);
                            break;
                        }
                        if (i == list.size() - 1) {
                            flag = false;
                        }
                    }
                }
                if (flag == false) {
                    break;
                }
                idx++;
                if (idx == word.length()) {
                    total++;
                }
            }
        }
        
        return total;
    }
}
