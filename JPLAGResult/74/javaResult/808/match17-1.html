class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int count = 0;
        if (S.length() == 0)
            return 0;
        if (words.length == 0)
            return 0;
        
        HashMap<String, Boolean> map = new HashMap();
        for (String s : words) {
            if (map.containsKey(s)) {
                boolean res = map.get(s);
                if (res)
                    count++;
            
}
            else {
                if (isSubSeq(S, s)) {
                    count++;
                    map.put(s, true);
                }
                
                else 
                    map.put(s, false);
            }
        }
        
        return count;
    }
    
    public boolean isSubSeq(String S, String t) {
        if (S.length() == 0)
            return false;
        if (t.length() == 0)
            return false;
        int j = 0;
        for (int i = 0;i<
S.length() && j < t.length();i++) {
            if (S.charAt(i) == t.charAt(j))
                j++;
        }
        
        return j == t.length();
    }
}
