class Solution {
        
    public boolean isMatchingSubseq(String S, String word) {
        int len = S.length();
        int curr = 0;
        int curr_char = word.charAt(0);
        for (int i = 0; i < len; i++) {
            if (S.charAt(i) == curr_char) {
                curr++;
                if (curr == word.length()) {
                    return true;
                }
                curr_char = word.charAt(curr);
            }
        }
        return false;
    }

    
public int numMatchingSubseq(String S, String[] words) {
        int count = 0;
        for (String word : words) {
            if (isMatchingSubseq(S, word)) {
                count++;
            }
        }
        return count;
    }
}
