class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int result = 0;
        for (int i = 0 ; i < words.length; i++){
            String word = words[i];
            int index = -1;
            boolean hasError = false;
                // System.out.println("word=" + word);
            for (int j = 0; j < word.length(); j++){     
                // System.out.println("c=" + word.charAt(j));           
                
index = S.indexOf(word.charAt(j), index+1);
                // System.out.println("index=" + index);
                if (index < 0) {
                    hasError = true;
                    break;
                }
            }
            if (!hasError) result++;
        }
        return result;
    }
}
