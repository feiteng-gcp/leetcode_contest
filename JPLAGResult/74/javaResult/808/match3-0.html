class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int count=0;
        for(int i=0;i<words.length;i++){
            if(lcs(S,words[i]))
                count++;
        }
        return count;
    }
    public boolean lcs(String f,String s){
        if(s.length() > f.length())
            return false;
        int i=0,j=0;
        for(;i<f.length()&&j<s.length();){
            if(f.charAt(i) == s.charAt(j)){
                i++;
                j++;
            }
            else{
                i++;
            }
                
        }
        if(j == s.length()){
            return true;
        }
        return false;
    }
}
