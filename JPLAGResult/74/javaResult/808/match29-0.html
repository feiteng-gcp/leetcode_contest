class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int count = 0;
        if (S.length() == 0 || words.length == 0) {
            return 0;
        }
        for (String word : words) {
            if (helper(word, S)) {
                count++;
            }
        }
        return count;
    }
    
    public boolean helper(String word, String S) {
        int index = 0;
        char[] cc = word.toCharArray();
        for (char c : cc) {
            int temp = S.indexOf(c , index);
            if (temp < 0) {
                return false;
            }
            index = temp + 1;
        }
        return true;
    }
}
