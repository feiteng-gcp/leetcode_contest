class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int m = S.length();
        int[][] dp = new int[26][m];
        int[] len = new int[26];
        
        for (int i = 0; i < m; i++) {
            int k = S.charAt(i) - 'a';
            dp[k][len[k]++] = i;
        
}
        
        int res = 0;
        
        for (String word : words) {
            if (isSubsequence(dp, len, word)) res++;
        }
        
        return res;
    }
    
    private boolean isSubsequence(int[][] dp, int[] len, String word) {
        int[] idx = new int[26];
        int min = -1;
        
        for (char c : word.toCharArray()) {
            int i = c - 'a';
            while (idx[i] < len[i] && dp[i][idx[i]] <= min) idx[i]++;
            if (idx[i] >= len[i]) 
return false;
            min = Math.max(min, dp[i][idx[i]]);
        }
        
        return true;
    }
}
