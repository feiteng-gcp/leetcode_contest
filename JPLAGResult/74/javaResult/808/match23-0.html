class Solution {
    public int numMatchingSubseq(String S, String[] words) {
        int c = 0;
        for (String word : words) {
            if (isSub(S, word)) c++;
        }
        return c;
    }
    
    public boolean isSub(String str1, String str2) {
        int m = str1.length();
        int n = str2.length();
        
        if (n > m ) return false;
        
        int i = 0, j = 0;
        while (i < m && j < n) {
            if (str1.charAt(i) == str2.charAt(j)) {
                i++;
                j++;
            } else {
                i++;
            }
        }
        
        return j == n;
    }
}
