class Solution {
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        int[] right=new int[A.length];
        int res=0;
        // val+index
        Stack<int[]> higher=new Stack<int[]>();
        for(int i=A.length-1;i>=0;i--){
            int curr=A[i];
            while(!higher.isEmpty()&&higher.peek()[0]<=curr){
                higher.pop();
            }
            right[i]=higher.isEmpty()?A.length:higher.peek()[1];
            higher.push(new int[]{curr,i});
        }
        higher.clear();
        for(int i=0;i<A.length;i++){
            int curr=A[i];
            while(!higher.isEmpty()&&higher.peek()[0]<curr){
                higher.pop();
            }
            if(L<=curr&&curr<=R){
                res=res+(higher.isEmpty()?(i+1):(i-higher.peek()[1]))*(right[i]-i);
            }
            higher.push(new int[]{curr,i});
        }
        return res;
    }
}
