class Solution {
    static int left;
    static int right;
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        left = L;
        right = R;
        int result = 0;
        for (int i = 0; i < A.length; i ++){
            int max = A[i];
            for (int j = i; j < A.length; j++){
                int num = A[j];
                if (num > max) max = num;
                if (inRange(max)) result++;
                else if (max > R) break;
            }
        }
        return result;
    }
    
    private boolean inRange(int num){
        return num >= left && num <= right;
    }
}
