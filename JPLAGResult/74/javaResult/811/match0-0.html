class Solution {
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        int[][] buffer = new int[A.length][A.length];
        
        int res = 0;
        for (int i = 0; i < A.length; i++) {
            buffer[i][i] = A[i];
            if (isValid(A[i], L, R)) {
                res++;
            }
        }
        
        for (int i = 0; i < A.length; i++) {
            for (int j = i + 1; j < A.length; j++) {
                buffer[i][j] = Math.max(buffer[i][j - 1], A[j]);
                if (isValid(buffer[i][j], L, R)) {
                    res++;
                }
            }
        }
        
        return res;
    }
    
    private boolean isValid(int num, int left, int right) {
        return num >= left && num <= right;
    }
}
