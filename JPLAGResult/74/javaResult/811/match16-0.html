class Solution {
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        int count = 0;
        for(int i=0;i<A.length;i++) {
            int max = -1;
            for(int j=i;j<A.length;j++) {
                max = Math.max(max, A[j]);
                if(max > R) continue;
                if(L <= max) count++;
            }
        }
        return count;
    }
}
