class Solution {
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        int[] a = new int[A.length + 1];
        int[] b = new int[A.length + 1];
        for (int i = 0; i < A.length; i++) {
            if (A[i] < L) {
                a[i + 1] = a[i] + 1;
                b[i + 1] = b[i];
            } else if (A[i] >= L && A[i] <= R) {
                b[i + 1] = a[i] + b[i] + 1;
            }
        }
        return Arrays.stream(b).sum();
    }
}
