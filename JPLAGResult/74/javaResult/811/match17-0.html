class Solution {
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        int res = 0;
        int l = A.length;
        
        for(int i = 0; i < l; i++) {
            int max = A[i];
            for(int j = i; j < l; j++) {
                if(A[j] > max)
                    max = A[j];
                
                if(max >= L && max <= R)
                    res++;
                
                if(max > R)
                    break;
            }
        }
        
        return res;
    }
}
