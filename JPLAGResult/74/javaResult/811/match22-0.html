class Solution {
	private long solve(int[] A , int l , int r , int lower) {
		
		long total = (r - l + 1);
		long ans = total * (total + 1) / 2;
		for (int i = l;i <= r;i ++) {
			if (A[i] < lower) {
				int from = i , to = i;
				while (i <= r && A[i] < lower) {
					to = i;
					i ++;
				}
				long temp = (to - from + 1);
				ans -= (temp * (temp + 1) / 2);
			}
		}
		return ans;
		
	}
	
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
    
    	long ans = 0;
    	int i , n = A.length;
    	for (i = 0;i < n;i ++) {
    		if (A[i] <= R) {
    			int from = i , to = i;
    			while (i < n && A[i] <= R) {
    				to = i;
    				i ++;
    			}
    			ans += solve(A , from , to , L);
    		}
    	}
    	return (int) ans;
    	
    }
 
}
