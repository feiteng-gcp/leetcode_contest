class Solution {
    public int nc2(int n) {
        return n * (n + 1)/2;
    }
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        if (A.length == 0)
            return 0;
        
        int less = 0;
        int greater = 0;
        int res = 0;
        for (int i = 0; i< A.length;i++) {
            
            
if (A[i] >R) {
                res += nc2(greater) - nc2(less);
                less = 0;
                greater = 0;
            }
            
            else if (A[i] < L) {
                less++;
                greater ++;
            }
            
            else {
                res -= nc2(less);
                less = 0;
                greater ++;
            }
        }
        
        res += nc2(greater) - nc2(less);
        
        return res;
    }
}
