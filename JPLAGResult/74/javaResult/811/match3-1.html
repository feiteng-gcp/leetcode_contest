class Solution {
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        long r = 0, out = 0, in = 0;
        for (int a : A) {
            if (a > R) {
                r = r + N(in) - N(out);
                in = 0;
                out = 0;
            } else if (a < L) {
                in++;
                out++;
            } else {
                r -= N(out);
                out = 0;
                in++;
            }
        }
        return (int) (r + N(in) - N(out));
    
}

    long N(long n) { return n * (n + 1) / 2;}
}
