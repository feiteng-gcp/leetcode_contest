class Solution {
    long f( long n ) {
        return n*(n+1)/2;
    }
    int []last,flast;
    long []z;
    int left, right;
    long count( int i, int j, int []x, int n ) {
        long ans = f((j-1)-(i+1)+1);
        int ii,jj;
        for ( ii = i+1; ii <= j-1 && x[ii] >= left; ++ii );
        if ( ii == j )
            return ans;
        do {
            for ( jj = ii+1; jj < j && x[jj] < left; ++jj ) ;
            ans -= f(jj-1 - ii+1);
            for ( ii = jj; ii < j && x[ii] >= left; ++ii ) ;
        } while ( ii < j ) ;
        return ans;
    }
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        int i,j,k,n=A.length;
        left = L;
        right = R;
        for ( i = 0; i < n && A[i] <= R; ++i ) ;
        long ans = 0;
        if ( i == n ) {
            return (int)count(-1,n,A,n);
        }
        else {
            ans += count(-1,i,A,n);
            do {
                for ( j = i + 1; j < n && A[j] <= R; ++j ) ;
                assert( j == n || A[j] > R );
                ans += count(i,j,A,n);
            } while ( (i=j) < n ) ;
        }
        return (int)ans;
    }
}
