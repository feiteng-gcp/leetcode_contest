	class Solution {
	    public int numSubarrayBoundedMax(int[] a, int L, int R) {
	        int n = a.length;
	        return count(a, R) - count(a, L-1);
	    }
	    
	    int count(int[] a
, int R)
	    {
	    	int my = 0, ret = 0;
	    	for(int v : a){
	    		if(v <= R){
	    			my++;
	    		}else{
	    			my = 0;
	    		}
	    		ret += my;
	        }
	    	return ret;
	    }
	}
