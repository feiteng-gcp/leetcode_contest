class Solution {
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        int len = A.length;
        if (len == 0) {
            return 0;
        }
        int count = 0;
        for (int i = 0; i < len; i++) {
            int max = A[i];
            if (max >= L && max <= R) {
                count++;
            }
            for (int j = i + 1; j < len; j++) {
                max = Math.max(max, A[j]);
                if (max >= L && max <= R) {
                    count++;
                
}
            }
        }
        return count;
    }
}
