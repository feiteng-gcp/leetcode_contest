class Solution {
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        int res = 0, n = A.length, inc = 0, exc = 0;

        for (int i = 0 ; i < n; ++i) {
            int cur = A[i];
            if (A[i] > R) {
                res += countSubarrys(inc) - countSubarrys(exc);
                inc = exc = 0;
            } else if (cur < L) {
                ++exc;
                ++inc;
            } else {
                res -= countSubarrys(exc);
                exc = 0;
                inc++;
            }
        }

        res += countSubarrys(inc) - countSubarrys(exc);

        return res;
    }

    private long countSubarrys(long n) {
        return n * (n + 1) / 2;
    }
}
