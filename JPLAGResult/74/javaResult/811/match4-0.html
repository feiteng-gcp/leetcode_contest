class Solution {
    public int numSubarrayBoundedMax(int[] A, int L, int R) {
        int n = A.length;
        int res = 0;
        int len = 0;
        int invalid = 0;
        int invalidSum = 0;
        boolean isValid = false;
        for (int i = 0; i < n; i++) {
            if (A[i] <= R) {
                len++;
                if (A[i] >= L) {
                    invalidSum += (invalid + 1) * invalid / 2;
                    isValid = true;
                    invalid = 0;
                } else {
                    invalid++;
                }
            } else {
                if (isValid) {
                    res += (len + 1) * len / 2;  
                    invalidSum += (invalid + 1) * invalid / 2;
                }
                len = 0;
                invalid = 0;
            }
        }
        if (len > 0 && isValid) {
            res += (len + 1) * len / 2; 
            invalidSum += (invalid + 1) * invalid / 2;
        }
        return res - invalidSum;
    }
}
