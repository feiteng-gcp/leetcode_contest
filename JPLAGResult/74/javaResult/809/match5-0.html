class Solution {
    public int preimageSizeFZF(int k) {
        if(k == 0) {
            return 5;
        }
        long result = 0;
        ArrayList<Long> zeroes = new ArrayList<>();
        long x = 5;
        long zero = 1;
        while(x <= k * 5L) {
            zeroes.add(zero);
            zero = zero * 5 + 1;
            x = x * 5;
        }
        
        while(k > 0) {
            long nextZeroes = findMax(zeroes, k);
            if(k / nextZeroes >= 5) {
                return 0;
            }
            k -= nextZeroes;
        }
        return 5; 
    }
    
    private long findMax(ArrayList<Long> zeroes, long k) {
        int i = 0;
        for(; zeroes.get(i) * 5 < k; ++i);
        return zeroes.get(i);
    }
}
