import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {


    public int preimageSizeFZF(int K) {
        if (K == 0) return 5;

        long a = bs(K);
        long b = bs1(K);
//        System.out.println(a + " " + f(a));
//        System.out.println(b + " " + f(b));
//        System.out.println(f(a + 1));
//        System.out.println(f(b - 1));
        return (int) Math.max((b - 1) - a, 0);
    
}

    long bs(long K) {
        long left = 0, right = 1000L * Integer.MAX_VALUE;
        while (left + 1 < right) {
            long mid = (left + right) / 2;
            if (f(mid) < K) 
{
                left = mid;
            } else {
                right = mid;
            }
        }
        
return left;
    }

    long bs1(long K) {
        long left = 0, right = 100L * Integer.MAX_VALUE;
        while (left + 1 < right) {
            long mid = (left + right) / 2;
            if (f(mid) > K) 
{
                right = mid;
            } else {
                left = mid;
            }
        }
        return right;
    }

    long f(long n) {
        long cnt = 0;
        while (n > 0) {
            cnt += n / 5;
            n /= 5;
        }
        return cnt;
    }

    public static void main(String[] args) {
        Solution sol = new Solution();
        System.out.println(sol.preimageSizeFZF(1000071010));
    }
}
