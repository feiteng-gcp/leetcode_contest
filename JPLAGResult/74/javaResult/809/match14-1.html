class Solution {
    public int preimageSizeFZF(int K) {
        long lower = 0;
        long higher = (long) 1e14;
        
while (lower < higher) {
            long mid = (lower + higher) / 2;
            if (f(mid) < K) lower = mid + 1;
            else higher = mid;
        }
        int ans = 0;
        for (long i = Math.max(0, lower - 10); i <= lower + 10; i++) if (f(i) == K) ans++;
        return ans;
    }
    
    public long f(long x) {
        long m = 5;
        
long ans = 0;
        while (x >= m) {
            ans += (x / m);
            m *= 5;
        }
        return ans;
    }
}
