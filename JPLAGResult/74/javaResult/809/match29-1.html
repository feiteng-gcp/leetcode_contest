class Solution {
    public int preimageSizeFZF(int K) {
        return binarySearch(K);
    }

    private int trailingZeroes(int n) {
        int cnt = 0;
        while (n > 0) {
            n /= 5;
            cnt += n;
        
}
        return cnt;
    }

    private int binarySearch(int n) {
        int low = 0;

        // range of numbers
        int high = 1000000000;

        // binary search for first number
        // with n trailing zeros
        while (low < high) {
            int mid = (low + high) / 2;
            int count = trailingZeroes(mid);
            if (count < n)
                low = mid + 1;
            else
                high = mid;
        }

        int res = 0;

        while (trailingZeroes(low) == n) {
            ++res;
            low++;
        }

        return res;
    }
}
