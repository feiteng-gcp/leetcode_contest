class Solution {
	private long getZero(long value) {
		
		long i , ans = 0;
		for (i = 5;i <= value;i *= 5) {
			ans += value / i;
		}
		return ans;
		
	}
	
	private long lowerValue(long K) {
		
		long low = 0 , high = 10000000000L;
		while (low < high) {
			long mid = (low + high) / 2;
			long total = getZero(mid);
			if (total < K) {
				low = mid + 1;
			} else {
				high = mid;
			}
		}
		if (getZero(low) == K) {
			return low;
		} else {
			return - 1;
		}
		
	}
	
	private long upperValue(long K) {
		
		long low = 0 , high = 10000000000L;
		while (low < high) {
			long mid = (low + high) / 2;
			long total = getZero(mid);
			if (total <= K) {
				low = mid + 1;
			} else {
				high = mid;
			}
		}
		high --;
		if (high >= 0 && getZero(high) == K) {			
			return high;
		} else {
			return - 1;
		}
		
	}
	
    public int preimageSizeFZF(int K) {
     
    	long low = lowerValue(K) , up = upperValue(K);
    	if (low >= 0 && up >= 0) {
    		return (int) (up - low + 1);    
    	} else {
    		return 0;
    	}
    	
    }
}
