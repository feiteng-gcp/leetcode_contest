class Solution {
    static int[] cache = new int[13];
    
    
public int preimageSizeFZF(int K) {
        int low = 1, high = 1000000000;
        while (low <= high) {
            int mid = low + (high-low)/2;
            int zeros = findTrailingZeros(mid);
            if (zeros == K) {
                break;
            } else 
if (zeros < K) {
                low = mid + 1;
            } else {
                high = mid -1;
            }
        }
        
        if (high < low) {
            return 0;
        }
        
        int candidate = low + (high-low)/2;
        
        System.out.print("candidate: " + candidate);
        
        int res = 0;
        for (int i = candidate - 5; i < candidate + 5; ++i) {
            if (i >= 0 && findTrailingZeros(i) == K) {
                ++res;
            }
        }
        
        return res;
    }
    
    int findTrailingZeros(int  n)
    {
        // Initialize result
        int count = 0;
 
        // Keep dividing n by powers of 5 and update count
        for (int i=5; n/i>=1; i *= 5)
          count += n/i;
 
        return count;
    }
}
