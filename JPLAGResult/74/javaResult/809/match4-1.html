class Solution {
    public int preimageSizeFZF(int K) {
        return (int)(fun(K) - fun(K - 1));
    }
    long fun(long K) {
        if (K == -1) return -1;
        long left = 0, right = get(K);
        while (left < right) {
            long mid = left + (right - left + 1) / 2;
            if (z(mid) > K) right = mid - 1;
            else left = mid
;
        }
        return left;
    }
    long get(long k) {
        long x = 1;
        while (z(x) <= k) x *= 2
;
        return x;
    }
    long z(long x) {
        long ans = 0;
        while(x != 0) {
            ans += x / 5;
            x /= 5;
        }
        return ans;
    }
}
