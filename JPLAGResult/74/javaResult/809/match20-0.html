	class Solution {
	    public int preimageSizeFZF(int K) {
	        return (int)(count(K) - count(K-1));
	    }
	    public long count(int n)
	    {
	    	long low = -1, high = 5000000000L;
	    	while(high - low > 1){
	    		long h = high+low>>1;
	    		long s = 0;
	    		for(long k = h/5;k > 0;k /= 5)s += k;
	    		if(s <= n){
	    			low = h;
	    		}else{
	    			high = h;
	    		}
	    	}
	    	return low+1;
	    }
	}
