class Solution {
    public int preimageSizeFZF(int K) {
        int left = findMin(K), right = findMax(K), ret = right - left + 1;
        if(ret<0) return 0;
        return ret * 5;
    }
    
    private int getSum(int v) {
        // v<=getSum(v)<=(5/4)*v, 
        int ret = 0;
        while(v!=0) {
            ret += v;
            v /= 5;
        }
        return ret;
    }
    
    private int pow(int base, int exp) {
        int ret = 1;
        for(int i=0;i<exp;i++) ret *= base;
        return ret;
    }
    
    private int findMin(int K) {
        //return min {v:getSum(v)>=K}
        int left = K*4/5, right = K;
        while(right-left>1) {
            int middle = left + (right-left) / 2;
            if(getSum(middle)<K) {
                left = middle;
            } else {
                right = middle;
            }
        }
        if(getSum(left)>=K) return left;
        else return right;
    }
    
    private int findMax(int K) {
        //return max {v:getSum(v)<=K}
        int left = K*4/5, right = K;
        while(right-left>1) {
            int middle = left + (right-left) / 2;
            if(getSum(middle)>K) {
                right = middle;
            } else {
                left = middle;
            }
        }
        if(getSum(right)<=K) return right;
        else return left;
    }
    
    private int numberOfZerosInFactorial(int n) {
        int ret = 0;
        while(n!=0) {
            n /= 5;
            ret += n;
        }
        return ret;
    }
}
