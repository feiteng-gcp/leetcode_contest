class Solution {
    public int preimageSizeFZF(int K) {
        long left = findLeft(K);
        if (left == -1) {
            return 0;
        }
        long right = findRight(K);
        return (int) (right - left + 1);
    }

    long findLeft(long K) {
        long l = 0;
        long h = Integer.MAX_VALUE;

        while (l < h) {
            long mid = (l + h) / 2;
            if (findTrailingZeros(mid) >= K) {
                h = mid;
            }
            else {
                l = mid + 1;
            }
        }
        if (findTrailingZeros(l) != K) {
            return -1;
        }
        else {
            return l;
        }
    }

    long findRight(long K) {
        long l = 0;
        long h = Integer.MAX_VALUE;

        while (l < h) {
            long mid = (l + h) / 2;
            if (findTrailingZeros(mid) <= K) {
                l = mid + 1;
            }
            else {
                h = mid;
            
}
        }
        return l - 1;
    }

    int findTrailingZeros(long n) {
        int res = 0;
        for (int i = 5; n / i >= 1; i *= 5) {
            res += n / i;
        }

        return res;
    }
}
