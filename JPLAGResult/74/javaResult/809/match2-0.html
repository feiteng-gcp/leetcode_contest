class Solution {
    public int preimageSizeFZF(int K) {
        
        // use binary search
        long target = (long) K;
        long right = target;
        long left = target / 2;
        while (calculateK(left) > target && left > 0) left /= 2;
        // ck left < k, ck right > k, binary search ck mid == k, if find return 5, otherwise return 0
        while (left <= right) {
            long mid = left + (right - left) / 2;
            long ckm = calculateK(mid);
            if (ckm == target) return 5;
            if (ckm > target) right = mid - 1;
            else left = mid + 1;
        }
        return 0;
    }
    
    long calculateK(long x) {
        long sum = x;
        long fac = 5;
        while (fac <= x) {
            sum += x / fac;
            fac *= 5;
        }
        return sum;
    }
}
