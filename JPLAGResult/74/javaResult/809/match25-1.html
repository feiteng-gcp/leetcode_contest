class Solution {
   public static int preimageSizeFZF(int K) {
        return findNum(K + 1) - findNum(K);
    }

    static boolean check(int p, int n) {
        int temp = p, count = 0, f = 5;
        while (f <= temp) {
            count += temp / f;
            f = f * 5;
        }
        return (count >= n);
    }

    static int findNum(int n) {
        if (n == 0)
            return 0;

        if (n == 1)
            return 5;

        int low = 0;
        int high = 5 * n;

        while (low < high) {
            int mid = (low + high) / 2;

            if (check(mid, n))
                high = mid;
            else
                low = mid + 1;
        }

        return low;
    }
}
