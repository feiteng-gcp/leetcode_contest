class Solution {
    private long numOfFive(double d) {
        long sum = 0;
        long l = (long) d;
        while (l > 0) {
            sum += l / 5;
            l = l / 5;
        }
        return sum;
    }
    
    private long firstHasTZero(int T) {
        double l = 0, r = T * 5;
        while (r - l >= 1) {
            double mid = l + (r - l) / 2;
            if (numOfFive(mid) > T) {
                r = mid;
            } else {
                l = mid;
            }
        }
        if (numOfFive(l) == T) return ((long) (l / 5)) * 5;
        if (numOfFive(r) == T) return ((long) (r / 5)) * 5;
        else return -1;
    }
    
    public int preimageSizeFZF(int K) {
        long min = firstHasTZero(K);
        if (min == -1) return 0;
        long max;
        while ((max = firstHasTZero(++K)) == -1);
        return (int) (max - min);
    }
}
