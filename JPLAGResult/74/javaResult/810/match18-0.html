class Solution {
    public boolean validTicTacToe(String[] board) {
        if (board == null || board.length == 0) {
            return false;
        }
        
        int[] rows = new int[3];
        int[] cols = new int[3];
        int cross = 0;
        int antiCross = 0;
        
        int xNum = 0;
        int oNum = 0;
        
        for (int j = 0; j < 3; j++) {
            for (int i = 0; i < 3; i++) {
                char ch = board[j].charAt(i);
                if (ch == ' ') {
                    continue;
                }
                
                if (ch == 'X') {
                    rows[i] += 1;
                    cols[j] += 1;
                    
                    if (i == j) {
                        cross += 1;
                    }
                    
                    if (i + j == 2) {
                        antiCross += 1;
                    }
                    xNum++;
                } else {
                    rows[i] -= 1;
                    cols[j] -= 1;
                    
                    if (i == j) {
                        cross -= 1;
                    }
                    
                    if (i + j == 2) {
                        antiCross -= 1;
                    }
                    oNum++;
                }
            }
        }
        
        if (oNum > xNum || (xNum - oNum > 1)) {
            return false;
        }
        
        int count = 0;
        boolean xWins = false;
        boolean oWins = false;
        for (int i = 0; i < 3; i++) {
            if (rows[i] == 3) {
                count++;
                xWins = true;
            }
            if (rows[i] == -3) {
                count++;
                oWins = true;
            }
            if (cols[i] == 3) {
                count++;
                xWins = true;
            }
            if (cols[i] == -3) {
                count++;
                oWins = true;
            }
        }
        
        if (cross == 3) {
            count++;
            xWins = true;
        }
        if (cross == -3) {
            count++;
            oWins = true;
        }
        if (antiCross == 3) {
            count++;
            xWins = true;
        }
        if (antiCross == -3) {
            count++;
            oWins = true;
        }
        
        if (xWins && ((xNum - oNum) != 1)) {
            return false;
        }
        
        if (oWins && (oNum != xNum)) {
            return false;
        }
        
        return count > 1 ? false : true;
    }
}
