class Solution {
    static int[][] win = new int[][] {{0, 1, 2}, // Check first row.
        {3, 4, 5}, // Check second Row
        {6, 7, 8}, // Check third Row
        {0, 3, 6}, // Check first column
        {1, 4, 7}, // Check second Column
        {2, 5, 8}, // Check third Column
        {0, 4, 8}, // Check first Diagonal
        {2, 4, 6}};
    
    public boolean validTicTacToe(String[] board) {
        int xCount = 0;
		int oCount = 0;

		String b = board[0] + board[1] + board[2];

		System.out.println(b);

		for (int i = 0; i < 9; i += 1) {

			if (b.charAt(i) == 'X') {
				xCount += 1;

			} else if (b.charAt(i) == 'O') {
				oCount += 1;
			}

		}

		if (xCount == oCount || xCount == oCount + 1) {

			if (isWinner(b, 'O')) {

				if (isWinner(b, 'X')) {
					return false;
				}

				return xCount == oCount;

			}

			if (isWinner(b, 'X') && xCount != oCount + 1) {
				return false;
			}

			return true;
			
		}

		return false;
        
    }
    
    public static boolean isWinner(String board, char piece) {
		
		for(int i=0; i<8; i++) {
			
			if(board.charAt(win[i][0]) == piece &&
					board.charAt(win[i][1]) == piece &&
					board.charAt(win[i][2])== piece) {
				return true;
			}
			
			
			
		}
		
		return false;
		
		
		
	}
    
    
}
