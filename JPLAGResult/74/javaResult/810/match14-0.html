class Solution {
    public boolean validTicTacToe(String[] board) {
        int countX = 0;
        int countO = 0;
        int colX[] = new int[] {0, 0, 0};
        int colO[] = new int[] {0, 0, 0};
        int diagX1 = 0, diagX2 = 0;
        int diagO1 = 0, diagO2 = 0;
        boolean gameOverX = false;
        boolean gameOverO = false;
        for (int i = 0; i < board.length; i++) {
            int rowX = 0;
            int rowO = 0;
            for (int j = 0; j < board[i].length(); j++) {
                char c = board[i].charAt(j);
                if (c == 'X') {
                    countX++;
                    colX[j]++;
                    rowX++;
                }
                if (c == 'O') {
                    countO++;
                    colO[j]++;
                    rowO++;
                }
                if ((i == 0 && j == 0) || (i == 2 && j == 2)) {
                    if (c == 'X') diagX1++;
                    if (c == 'O') diagO1++;
                }
                if (i == 1 && j == 1) {
                    if (c == 'X') { diagX1++; diagX2++; }
                    if (c == 'O') { diagO1++; diagO2++; }
                }
                if ((i == 0 && j == 2) || (i == 2 && j == 0)) {
                    if (c == 'X') diagX2++;
                    if (c == 'O') diagO2++;
                }
            }
            if (rowX == 3) {
                gameOverX = true;
            }
            if (rowO == 3) {
                gameOverO = true;
            }
        }

        if (countX < countO || countX - countO > 1) return false;

        if (diagX1 == 3 || diagX2 == 3) gameOverX = true;
        if (diagO1 == 3 || diagO2 == 3) gameOverO = true;

        for (int i = 0; i < colX.length; i++) {
            if (colX[i] == 3) {
                gameOverX = true;
            }
            if (colO[i] == 3) {
                gameOverO = true;
            }
        }

        if (gameOverX && gameOverO) return false;            
        
        if (gameOverX && countX - countO != 1) return false;

        if (gameOverO && countX != countO) return false;
        
        return true;
    }
}
