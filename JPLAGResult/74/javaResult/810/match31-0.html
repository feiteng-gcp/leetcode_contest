import javax.swing.plaf.metal.MetalInternalFrameTitlePane;
import java.util.*;
import java.io.*;
import java.util.regex.*;

class Solution {
    int n=3;
    char [][]g;
    public boolean validTicTacToe(String[] b) {
        int i,j,k,cx=0,co=0;
        g = new char[n][];
        for ( i = 0; i < n; ++i )
            g[i] = b[i].toCharArray();
        for ( i = 0; i < n; ++i )
            for ( j = 0; j < n; ++j )
                if ( b[i].charAt(j) == 'X' )
                    ++cx;
                else if ( b[i].charAt(j) == 'O' )
                    ++co;
        if ( !(cx == co || cx-co == 1) )
            return false ;
        if ( wins('X') && wins('O') )
            return false;
        if ( wins('X') && cx == co )
            return false ;
        if ( wins('O') && cx == co+1 )
            return false ;
        return true ;
    }

    int up( int x, int y, char ch ) {
        int k=0;
        for ( int nx=x; nx >= 0 && g[nx][y] == ch; --nx, ++k ) ;
        return k;
    }
    int down( int x, int y, char ch ) {
        int k=0;
        for ( int nx=x; nx < n && g[nx][y] == ch; ++nx, ++k ) ;
        return k;
    }
    int left( int x, int y, char ch ) {
        int k=0;
        for ( int ny=y; ny >= 0 && g[x][ny] == ch; --ny, ++k ) ;
        return k;
    }
    int right( int x, int y, char ch ) {
        int k=0;
        for ( int ny=y; ny < n && g[x][ny] == ch; ++ny, ++k ) ;
        return k;
    }
    int diag0( int x, int y, char ch ) {
        int nx,ny,k=0;
        for ( nx=x,ny=y; nx >= 0 && ny >= 0 && nx < n && ny < n && g[nx][ny]==ch; --nx, ++ny, ++k ) ;
        return k;
    }
    int diag1( int x, int y, char ch ) {
        int nx,ny,k=0;
        for ( nx=x,ny=y; nx >= 0 && ny >= 0 && nx < n && ny < n && g[nx][ny]==ch; ++nx, ++ny, ++k ) ;
        return k;
    }
    private boolean wins(char x) {
        int i,j,k;
        for ( i = 0; i < n; ++i )
            for ( j = 0; j < n; ++j )
                if ( g[i][j] == x ) {
                    if ( up(i,j,x) == 3 )
                        return true ;
                    if ( down(i,j,x) == 3 )
                        return true ;
                    if ( left(i,j,x) == 3 )
                        return true ;
                    if ( right(i,j,x) == 3 )
                        return true ;
                    if ( diag0(i,j,x) == 3 )
                        return true ;
                    if ( diag1(i,j,x) == 3 )
                        return true ;
                }
        return false ;
    }
}
