class Solution {
    public boolean validTicTacToe(String[] board) {
        int countx = 0;
        int counto = 0;
        int winx = 0;
        int wino = 0;
        char[][] bb = new char[3][3];
        
        
int k = 0;
        for (String str : board) {
            for (int i = 0; i < 3; i++) {
                char c = str.charAt(i);
                bb[k][i] = c;
                if (c == 'X') countx++;
                else if (c == 'O') counto++;
            }
            k++;
        }
        
        if (countx < counto || countx - counto > 1) return false;
        
        for (String str : board) {
            if (str.equals("XXX")) winx++;
            else if (str.equals("OOO")) wino++;
        }
        
        
        for (int i = 0; i < 3; i++) {
            String str = "" + bb[0][i] + bb[1][i] + bb[2][i];
            if (str.equals("XXX")) winx++;
            else if (str.equals("OOO")) wino++;
        }
        
        String str1 = "" + bb[0][0] + bb[1][1] + bb[2][2];
        String str2 = "" + bb[0][2] + bb[1][1] + bb[2][0];
        
        if (str1.equals("XXX")) winx++;
        else if (str1.equals("OOO")) wino++;
        
        if (str2.equals("XXX")) winx++;
        else if (str2.equals("OOO")) wino++;
        //System.out.println(winx + " " + wino + " " + countx  + " " + counto);
        
        if (winx == 0 && wino == 0 || (winx == 0 && wino == 1 && countx == counto) || (winx == 1 && wino == 0 && countx - counto == 1)) return true;
        return false;
    }
}
