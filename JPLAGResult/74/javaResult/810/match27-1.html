class Solution {

    public boolean validTicTacToe(String[] board) {
        if (board == null) {
            return false;
        }

        int numXWins = getNumWins(board, 'X');
        if (numXWins > 1) {
            return false;
        }

        int numOWins = getNumWins(board, 'O');
        
if (numOWins > 1) {
            return false;
        }

        if (numXWins == 1 && numOWins == 1) {
            return false;
        }

        int numXs = getNum(board, 'X');
        int numOs = getNum(board, 'O');

        if (numXWins == 1) {
            return numXs == (numOs + 1);
        }

        if (numOWins == 1) {
            return numXs == numOs;
        }

        return (numXs == numOs) || (numXs == (numOs + 1));
    }

    private int getNumWins(String[] board, char player) {
        int count = 0;

        // rows
        for (String row : board) {
            if (allEqual(row, player)) {
                count += 1;
            }
        }

        // columns
        for (int i = 0; i < 3; i++) {
            if (allEqual(board, player, i)) {
                count += 1;
            }
        }

        // diagonals
        if (topLeftDiagonalEqual(board, player)) {
            count += 1;
        }

        if (topRightDiagonalEqual(board, player)) {
            count += 1;
        }

        return count;
    }

    private boolean topRightDiagonalEqual(String[] board, char player) {
        if (
board[0].charAt(2) != player) {
            return false;
        }
        if (board[1].charAt(1) != player) {
            return false;
        }
        if (board[2].charAt(0) != player) {
            return false;
        }
        return true;
    }

    private boolean topLeftDiagonalEqual(String[] board, char player) {
        if (board[0].charAt(0) != player) {
            return false;
        }
        if (board[1].charAt(1) != player) {
            return false;
        }
        if (board[2].charAt(2) != player) {
            return false;
        }
        return true;
    }

    private boolean allEqual(String[] board, char player, int column) {
        
for (String row : board) {
            if (row.length() >= column) {
                return false;
            }

            if (row.charAt(column) != player) {
                return false;
            }
        }

        return true;
    }

    private boolean allEqual(String row, char player) {
        for (char r : row.toCharArray()) {
            if (player != r) {
                return false;
            }
        }

        return true;
    }

    private int getNum(String[] board, char player) {
        int count = 0;
        for (String row : board) {
            for (int i = 0; i < row.length(); i++) {
                char move = row.charAt(i);
                if (move == player) {
                    count += 1;
                }
            }
        }
        return count;
    }

}
