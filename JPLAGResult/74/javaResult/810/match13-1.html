class Solution {
    public boolean validTicTacToe(String[] board) {
        int countX = 0, countO = 0;
        for (int i = 0; i < 3; i++){
            
for (int j = 0; j < 3; j++){
                if (board[i].charAt(j) == 'X'){
                    countX++;
                }else if (board[i].charAt(j) == 'O')
                    countO++;
            }
        }
        if (countX-countO > 1 || countO - countX > 0)
            return false;
        if (board[0].equals("XXX") || board[1].equals("XXX") || board[2].equals("XXX") || (board[0].charAt(0) == 'X' && board[1].charAt(0) == 'X' && board[2].charAt(0) == 'X') || (board[0].charAt(1) == 'X' && board[1].charAt(1) == 'X' && board[2].charAt(1) == 'X') || (board[0].charAt(2) == 'X' && board[1].charAt(2) == 'X' && board[2].charAt(2) == 'X') || (board[0].charAt(0) == 'X' && board[1].charAt(1) == 'X' && board[2].charAt(2) == 'X') || (board[0].charAt(2) == 'X' && board[1].charAt(1) == 'X' && board[2].charAt(0) == 'X')){
            if (countX == countO)
                return false;
        }
        if (board[0].equals("OOO") || board[1].equals("OOO") || board[2].equals("OOO") || (board[0].charAt(0) == 'O' && board[1].charAt(0) == 'O' && board[2].charAt(0) == 'O') || (board[0].charAt(1) == 'O' && board[1].charAt(1) == 'O' && board[2].charAt(1) == 'O') || (board[0].charAt(2) == 'O' && board[1].charAt(2) == 'O' && board[2].charAt(2) == 'O') || (board[0].charAt(0) == 'O' && board[1].charAt(1) == 'O' && board[2].charAt(2) == 'O') || (board[0].charAt(2) == 'O' && board[1].charAt(1) == 'O' && board[2].charAt(0) == 'O')){
            if (countX > countO)
                return false;
        }
        return true;
    }
}
