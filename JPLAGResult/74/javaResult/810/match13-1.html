class Solution {
    public boolean validTicTacToe(String[] board) {
        int[] rows = new int[3];
        int[] cols = new int[3];
        int diagonal = 0;
        int antiDiagonal = 0;
        
        
int x = 0;
        int o = 0;
        
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                char c = board[i].charAt(j);
                int diff = c == 'X' ? 1 : c == 'O' ? -1 : 0;
                x+= diff;
                rows[i] += diff;
                cols[j] += diff;
                if (i == j) {
                    diagonal += diff;
                } 
                if (i + j == 2) {
                    antiDiagonal += diff;
                }
            }
        }
        
        //System.out.println(Arrays.toString(rows));
        //System.out.println(Arrays.toString(cols));
        //System.out.println(x);
        
        if (x != 0 && x != 1) return false;
        int checkx = check(3, rows);
        checkx += check(3, cols);
        checkx += check(3, diagonal);
        checkx += check(3, antiDiagonal);
        int checko = check(-3, rows);
        checko += check(-3, cols);
        checko += check(-3, diagonal);
        checko += check(-3, antiDiagonal);
        //System.out.println(checkx);
        //System.out.println(checko);
        if (checkx > 1) return false;
        if (checkx > 0 && checko > 0) return false;
        if (checko > 1) return false;
        
        if (checkx == 1) return x == 1;
        if (checko == 1) return x == 0
;
        
        return true;
        
    }
    
    
    private int check(int val, int... inputs) {
        int res = 0;
        for (int i : inputs) {
            if (i == val) res++;
        }
        return res;
    }
}
