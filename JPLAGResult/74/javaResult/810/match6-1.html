class Solution {
    public boolean validTicTacToe(String[] board) {
        char[] a = new char[9];
        int i = 0;
        for (String s : board) {
            for (char c : s.toCharArray()) {
                a[i++] = c;
            }
        }

        int x = 0, o = 0;
        for (char c : a) {
            
if (c == 'X') {
                ++x;
            } else if (c == 'O') {
                ++o;
            }
        }

        if (x != o && x != o + 1) {
            return false;
        }

        if (hasWon(a, 'O')) {
            if (hasWon(a, 'X')) {
                return false;
            }
            return x == o;
        }

        if (hasWon(a, 'X') && x != o + 1) {
            return false;
        }
        
        return true;
    }

    
private int[][] win = new int[][] {
        {
            0, 1, 2
        }, // Check first row.
        {
            3, 4, 5
        }, // Check second Row
        {
            6, 7, 8
        }, // Check third Row
        {
            0, 3, 6
        }, // Check first column
        {
            1, 4, 7
        }, // Check second Column
        {
            2, 5, 8
        }, // Check third Column
        {
            0, 4, 8
        }, // Check first Diagonal
        {
            2, 4, 6
        }
    };

    private boolean hasWon(char[] board, char c) {
        for (int i = 0; i < 8; i++)
            if (board[win[i][0]] == c &&
                    board[win[i][1]] == c &&
                    board[win[i][2]] == c)
                return true;
        return false;
    }
}
