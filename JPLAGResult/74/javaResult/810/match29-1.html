class Solution {
    public boolean validTicTacToe(String[] board) {
        int countX = 0, countO = 0, wins = 0;
        char winner = ' ';

        for(int i = 0; i < 3; i++) {
            char c = board[i].charAt(0);
            if(c != ' ' && board[i].charAt(1) == c && board[i].charAt(2) == c) {
                wins++;
                winner = c;
            }
            
            c = board[0].charAt(i);
            if(c != ' ' && board[1].charAt(i) == c && board[2].charAt(i) == c) {
                wins++;
                winner = c;
            }
            
            for(int j = 0; j < 3; j++) {
                if(board[i].charAt(j) == 'X')
                    countX++;
                else if(board[i].charAt(j) == 'O')
                    countO++;
            }
        }
        
        char c = board[0].charAt(0);
        if(c != ' ' && board[1].charAt(1) == c && board[2].charAt(2) == c) {
            winner = c;            
            wins++;
        }
        
        c = board[0].charAt(2);
        if(c != ' ' && board[1].charAt(1) == c && board[2].charAt(0) == c) {
            winner = c;
            wins++;
        
}
        // System.out.println(countX + " " + countO + " " + wins);
        if(countX < countO || countX - countO > 1)
            return false;
        
        if(wins > 1)
            return false;
        
        if(wins == 1) {
            if(winner == 'X' && countX == countO)
                return false;
            
            if(winner == 'O' && countX > countO)
                return false;
        }
        
        return true;

    }
}
