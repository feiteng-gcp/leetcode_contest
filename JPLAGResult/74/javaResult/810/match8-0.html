class Solution {
    public boolean validTicTacToe(String[] board) {
        int xs=0, os=0;
        char[][] board1= new char[3][3];
        for(int i=0;i<3;i++){
            for(int j=0;j<3;j++){
                char c=board[i].charAt(j);
                if(c=='O') os++;
                else if(c=='X') xs++;
            board1[i][j]=c;
            }
        }
      //  System.out.println(xs+" "+os);
        if(xs<os || xs>os+1) return false;
       // if(xs==os) return checkValid(board1,xs);
        int[] ans=checkValid(board1,xs);
     //   System.out.println(xs+" "+os+" "+ans[0]+" "+ans[1]);
        if(xs==os && ans[0]>0) return false;
        if(xs>os && ans[1]>0) return false;
        return true;
    }
    
    public int[] checkValid(char[][] board,int count){
        int xs=0,os=0;
        for(int i=0;i<3;i++){
            if(board[i][0]==board[i][1] && board[i][1]==board[i][2]){
                if(board[i][0]=='O') os++;
                else if(board[i][0]=='X') xs++;
            }
            if(board[0][i]==board[1][i] && board[1][i]==board[2][i]){
                if(board[0][i]=='O') os++;
                else if(board[0][i]=='X') xs++;
            }
        }
        if(board[0][0]==board[1][1] && board[1][1]==board[2][2]){
             if(board[0][0]=='O') os++;
                else if(board[0][0]=='X') xs++;
        }
        
        if(board[0][2]==board[1][1] && board[1][1]==board[2][0]){
             if(board[0][2]=='O') os++;
                else if(board[0][2]=='X') xs++;
        }
        int[] res= new int[2];
        res[0]=xs;res[1]=os;
        return res;
    }
}
