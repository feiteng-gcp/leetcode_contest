class Solution {
    public boolean validTicTacToe(String[] board) {
        boolean x = false;
        boolean o = false;
        int count = 0;
        char[][] b = new char[3][3];
        for (int i = 0; i < 3; i++) {
            String temp = board[i];
            for (int j = 0; j < 3; j++) {
                b[i][j] = temp.charAt(j);
            }
        }
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                char temp = b[i][j];
                if (temp == ' ') {
                    continue;
                }
                if (temp == 'X') {
                    count++;
                }
                else {
                    count--;
                }
                if (temp == 'X' && helper(b, i, j)) {
                    x = true;
                }
                if (temp == 'O' && helper(b, i, j)) {
                    o = true;
                }
            }
        }
        if (count < 0 || count > 1) {
            return false;
        }
        if (count == 0 && x) {
            return false;
        }
        // System.out.println(count);
        // System.out.println(o);
        if (count == 1 && o) {
            return false;
        }
        if (x && o) {
            
return false;
        }
        return true;
    }
    
    public boolean helper(char[][] board, int x, int y) {
        char cur = board[x][y];
        if (y == 2) {
            if (board[x][y - 1] == cur && board[x][y - 2] == cur) {
                return true;
            }
        }
        if (x == 2) {
            if (board[x - 1][y] == cur && board[x - 2][y] == cur) {
                return true;
            }
        }
        if (x == 2 && y == 2) {
            if (board[x - 1][y - 1] == cur && board[x - 2][y - 2] == cur) {
                return true;
            }
        }
        if (x == 0 && y == 2) {
            if (board[x + 1][y - 1] == cur && board[x + 2][y - 2] == cur) {
                return true;
            }
        }
        return false;
    }
}
