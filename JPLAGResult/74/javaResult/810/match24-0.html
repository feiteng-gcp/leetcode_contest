class Solution {
    private static int[][] win = {{0,1,2}, {3,4,5}, {6,7,8}, {0,3,6}, {1,4,7}, {2,5,8}, {0,4,8}, {2,4,6}};
    public boolean validTicTacToe(String[] board) {
        String b = String.join("", board);
        int winO = winCount(b, 'O');
        int winX = winCount(b, 'X');
        int countO = b.length() - b.replaceAll("O", "").length();
        int countX = b.length() - b.replaceAll("X", "").length();
        int diff = countX - countO;
        if (winO == 0 && winX == 0) {
            return (diff == 0 || diff == 1);
        } else if (winO == 1 && winX == 0) {
            return diff == 0;
        } else if (winO == 0 && winX == 1) {
            return diff == 1;
        } else {
            return false;
        }
    }

    private int winCount(String b, char c) {
        int output = 0;
        for (int[] i : win) {
            boolean isWin = true;
            for (int j : i) {
                if (b.charAt(j) != c) {
                    isWin = false;
                }
            }
            if (isWin) {
                output++;
            }
        }
        return output;
    }

    public static void main(String[] args) {
        Solution s = new Solution();

        System.out.println(s.validTicTacToe(new String[] {"O  ", "   ", "   "}));
        System.out.println(s.validTicTacToe(new String[] {"XOX", " X ", "   "}));
        System.out.println(s.validTicTacToe(new String[] {"XXX", "   ", "OOO"}));
        System.out.println(s.validTicTacToe(new String[] {"XOX", "O O", "XOX"}));
        System.out.println(s);
    }
}
