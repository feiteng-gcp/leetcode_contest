class Solution {
    public boolean validTicTacToe(String[] board) {
        if (board == null || board.length != 3
           || board[0] == null || board[0].length() != 3) {
            return false;
        }
        
        
int xCnt = 0;
        int oCnt = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                char c = board[i].charAt(j);
                if (c == 'X') {
                    xCnt++;
                } else if (c == 'O') {
                    oCnt++;
                }
            }
        }
        
        if (oCnt > xCnt || (xCnt - oCnt) > 1) {
            return false;
        }
        
        int xWonCnt = wonGameCnt('X', board);
        int oWonCnt = wonGameCnt('O', board);
        if (xWonCnt > 0 && oWonCnt > 0) {
            return false;
        }
        
        if (xWonCnt > 1|| oWonCnt > 1) {
            return false;
        }
        
        if (xWonCnt > 0 && (xCnt - oCnt != 1)) {
            return false;
        }
        
        if (oWonCnt > 0 && xCnt != oCnt) {
            return false;
        
}
        
        return true;
    }
    
    private int wonGameCnt(char c, String[] board) {
        int cnt = 0;
        for (int i = 0; i < 3; i++) {
            int rowCnt = 0;
            for (int j = 0; j < 3; j++) {
                if (board[i].charAt(j) == c) {
                    rowCnt++;
                }
            }
            
            if (rowCnt == 3) {
                cnt++;
            }
        }
        
        for (int j = 0; j < 3; j++) {
            int colCnt = 0;
            for (int i = 0; i < 3; i++) {
                if(board[i].charAt(j) == c) {
                    colCnt++;
                }
            }
            
            if (colCnt == 3) {
                cnt++;
            }
        }
        
        if (board[0].charAt(0) == c
           && board[1].charAt(1) == c
           && board[2].charAt(2) == c) {
            cnt++;
        }
        
        if (board[0].charAt(2) == c
           && board[1].charAt(1) == c
           && board[2].charAt(0) == c) {
            cnt++;
        }
        
        return cnt;
    }
}
