import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {

    public boolean validTicTacToe(String[] board) {
        char[][] a = new char[3][];
        for (int i = 0; i < 3; i++)
            a[i] = board[i].toCharArray();


        int x = cnt(a, 'X');
        int o = cnt(a, 'O');
        if (x == o) {
            return !win(a, 'X');
        } else if (x - 1 == o) {
            return !win(a, 'O');
        } else {
            return false;
        }
    }

    private int cnt(char[][] a, char x) {
        int cnt = 0;
        for (int i = 0; i < a.length; i++)
            for (int j = 0; j < a[i].length; j++)
                if (a[i][j] == x) {
                    cnt++;
                }
        return cnt;
    }

    private boolean win(char[][] a, char x) {

        for (int i = 0; i < 3; i++) {
            if (a[i][0] == x && a[i][1] == x && a[i][2] == x) return true;
        }
        for (int j = 0; j < 3; j++) {
            if (a[0][j] == x && a[1][j] == x && a[2][j] == x) return true;
        }

        if (a[0][0] == x && a[1][1] == x && a[2][2] == x) return true;
        if (a[0][2] == x && a[1][1] == x && a[2][0] == x) return true;

        return false;
    }


    public static void main(String[] args) {
        Solution sol = new Solution();
        System.out.println(sol.validTicTacToe(new String[]{"O  ", "   ", "   "}));
        System.out.println(sol.validTicTacToe(new String[]{"XOX", " X ", "   "}));
        System.out.println(sol.validTicTacToe(new String[]{"XXX", "   ", "OOO"}));
        System.out.println(sol.validTicTacToe(new String[]{"XOX", "O O", "XOX"}));
    }
}
