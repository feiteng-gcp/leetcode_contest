class Solution {
  public boolean validTicTacToe(String[] board) {
    if (board==null || board.length!=3)
      return false;
    
    for (String row : board) {
      if (row==null || row.length()!=3)
        return false;
    }
    
    char[] newBoard = new char[9];
    for (int i=0; i<3; i++) {
      for (int j=0; j<3; j++)
        newBoard[i*3 + j] = board[i].charAt(j);
    }
    
    int x=0, o=0;
    for (int i=0; i<9; i++) {
      if (newBoard[i]=='X')
        x++;
      if (newBoard[i]=='O')
        o++;
    }
    
    if (x==o || x==o+1) {
      if (checkWin(newBoard, 'O')) {
        if (checkWin(newBoard, 'X')) 
          return false;
        return x==o;
      }
      
      if (checkWin(newBoard, 'X'))
        return x==o+1;
      
      return true;
    }
    
    return false;
  }
  
  private boolean checkWin(char[] board, char c) {
    int[][] winCase = {{0,1,2}, {3,4,5}, {6,7,8},
                       {0,3,6}, {1,4,7}, {2,5,8},
                       {0,4,8}, {2,4,6}};
    for (int i=0; i<8; i++) {
      if (board[winCase[i][0]]==c && board[winCase[i][1]]==c && board[winCase[i][2]]==c)
        return true;
    }
    
    return false;
  }
}
