class Solution {
    public boolean validTicTacToe(String[] board) {
        String total = String.join("", board);
        int xNum = 0, oNum = 0;
        for (char c : total.toCharArray()) {
            if (c == 'X') xNum++;
            if (c == 'O') oNum++;
        }
        if (xNum == oNum || xNum == oNum + 1) {
        } else return false;
        int endNum = 0;
        char endChar = ' ';
        for (String row : board) {
            if (row.charAt(0) != ' ' && row.charAt(0) == row.charAt(1) && row.charAt(1) == row.charAt(2)) {
                endNum++;
                endChar = row.charAt(0);
            }
        }
        for (int i = 0; i < 3; i++) {
            if (board[0].charAt(i) != ' ' && board[0].charAt(i) == board[1].charAt(i) && board[1].charAt(i) == board[2].charAt(i)) { 
                endNum++;
                endChar = board[0].charAt(i);
            }
        }
        if (board[0].charAt(0) != ' ' && board[0].charAt(0) == board[1].charAt(1) && board[2].charAt(2) == board[0].charAt(0)) {
            endNum++;
            endChar = board[0].charAt(0);
        }
        if (board[0].charAt(2) != ' ' && board[0].charAt(2) == board[1].charAt(1) && board[0].charAt(2) == board[2].charAt(0)) {
            endNum++;
            endChar = board[0].charAt(2);
        }
        if (endNum > 1) return false;
        if ((endChar == 'X' && xNum == oNum) || (endChar == 'O' && xNum == oNum + 1)) return false;
        return true;
    }
}
