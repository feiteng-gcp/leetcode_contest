class Solution {
    public boolean validTicTacToe(String[] board) {
        
        int cx = 0;
        
        int co = 0;
        
        for (int i=0; i<board.length; i++) {
            for (int j=0; j<3; j++) {
                char c = board[i].charAt(j);
                if (c == 'X') {
                    cx ++;
                } else if (c == 'O') {
                    co ++;
                } else if (c != ' ') {
                    return false;
                }
            }
        }
        
        if (cx < co) {
            return false;
        }
        
        // System.out.println("the cx: " + cx);
        // System.out.println("the co: " + co);
        
        if (cx > co + 1) {
            return false;
        }
        
        if (doesWin(board, 'X') && doesWin(board, 'O')) {
            return false;
        }
        
        if (doesWin(board, 'X') && (cx == co)) {
            return false;
        }
        
        if (doesWin(board, 'O') && (cx != co)) {
            return false;
        }
        
        // if (cx + co == 9 && !doesWin(board, 'X') && !doesWin(board, 'O')) {
        //     return false;
        // }
        
        return true;      
    }
    
    private boolean doesWin(String[] board, char symbol) {
        
        String pattern = symbol + "" + symbol + symbol;
        

        for (int i=0; i<board.length; i++) {
            if (board[i].equals(pattern)) {
                return true;
            }
        }
        
        for (int i=0; i<3; i++) {
            if (board[0].charAt(i) == symbol && board[1].charAt(i) == symbol && board[2].charAt(i) == symbol) {
                return true;
            }
        }
        
        if (board[0].charAt(0) == symbol && board[1].charAt(1) == symbol && board[2].charAt(2) == symbol) {
            return true;
        }
        
        if (board[0].charAt(2) == symbol && board[1].charAt(1) == symbol && board[2].charAt(0) == symbol) {
            return true;
        }
        
        return false;
    }
}
