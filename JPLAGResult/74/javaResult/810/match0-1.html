class Solution {
    public boolean validTicTacToe(String[] board) {
		int xs = 0, os = 0;

		for (String row : board) {
			for (char ox : row.replace(" ", "").toCharArray())
				if (ox == 'O')
					os++;
				else
					xs++;
		}
		if(xs == 0 && os == 0) return true;
		if(xs < os) return false;
		if(xs-os > 1) return false;

		
		int win = 0;
		String winner = "";
		
		for(int i = 0; i < 3; i++) {
			if(board[i].charAt(0) != ' ' && board[i].charAt(0) == board[i].charAt(1) && board[i].charAt(0) == board[i].charAt(2)) {
				win ++;
				winner = board[i].charAt(0) + "";
			}
			if(board[0].charAt(i)
 != ' ' && board[0].charAt(i) == board[1].charAt(i) && board[0].charAt(i) == board[2].charAt(i)) {
				win ++;
				winner = board[0].charAt(i) + "";
			}
		}
		
		if(board[0].charAt(0) != ' ' && board[0].charAt(0) == board[1].charAt(1) && board[0].charAt(0) == board[2].charAt(2)) {
			win ++;
			winner = board[0].charAt(0) + "";
		}
        
		if(board[0].charAt(2) != ' ' && board[0].charAt(2) == board[1].charAt(1) && board[1].charAt(1) == board[2].charAt(0)) {
			win ++;
			winner = board[0].charAt(2) + "";
		}
		if(win > 1
				|| (win == 1 && winner.equals("O") && xs > os)
				|| (win == 1 && winner.equals("X") && xs - os != 1)) return false;

		return true;
	}
}
