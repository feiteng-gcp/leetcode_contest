class Solution {
   public static boolean validTicTacToe(String[] board) {
        int x = 0;
        int o = 0;

        char[] b = new char[9];
        int k = 0;
        for (int i = 0; i < 3; i++)
            for (int j = 0; j < 3; j++) {
                b[k++] = board[i].charAt(j);
            }


        for (int i = 0; i < 9; i++) {
            char z = b[i];
            if (z == 'X')
                x++;
            else if (z == 'O')
                o++;
        }

        if (x == o || x == o + 1) {

            if (win(b, 'O')) {
                if (win(b, 'X'))
                    return false;
                return (x == o);
            }

            if (win(b, 'X') && x != o + 1)
                return false;

            return true;
        }

        return false;
    }

    public static boolean win(char[] b, 
char x) {

        int[][] c = new int[][]{{0, 1, 2},
                {3, 4, 5},
                {6, 7, 8},
                {0, 3, 6},
                {1, 4, 7},
                {2, 5, 8},
                {0, 4, 8},
                {2, 4, 6}};

        for (int i = 0; i < c.length; i++) {
            if (b[c[i][0]] == x && b[c[i][1]] == x && b[c[i][2]] == x)
                return true;
        }
        return false;
    }
}
