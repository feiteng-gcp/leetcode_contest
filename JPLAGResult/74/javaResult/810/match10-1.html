class Solution {
    public boolean validTicTacToe(String[] board) {
        int countX=0;
        int countO=0;
        for(int i=0;i<3;i++){
            for(int j=0;j<3;j++){
                char ch=board[i].charAt(j);
                if(ch=='X'){
                    countX++;
                }
                else if(ch=='O'){
                    countO++;
                }
            }
        }
        if(countX<countO||countX>countO+1){
            return false;
        }
        int[] resX=countCompleted(board,'X');
        int[] resO=countCompleted(board,'O');
        if(resX[0]>1||resO[0]>1||resX[1]>1||resO[1]>1){
            return false;
        }
        int totalX=resX[0]+resX[1]+resX[2];
        int totalO=resO[0]+resO[1]+resO[2];
        if(totalX>0&&totalO>0){
            return false;
        }
        if(totalX>0&&countX==countO){
            return false;
        }
        if(totalO>0&&countX>countO){
            
return false;
        }
        return true;
    }
    // horizon,veritical,diag
    public int[] countCompleted(String[] board,char ch){
        int[] res=new int[3];
        String target=""+ch+ch+ch;
        for(int i=0;i<3;i++){
            if(board[i].equals(target)){
                res[0]++;
            }
        }
        for(int i=0;i<3;i++){
            String temp=""+board[0].charAt(i)+board[1].charAt(i)+board[2].charAt(i);
            if(temp.equals(target)){
                res[1]++;
            }
        }
        if(board[0].charAt(0)==ch&&board[1].charAt(1)==ch&&board[2].charAt(2)==ch){
            res[2]++;
        }
        if(board[0].charAt(2)==ch&&board[1].charAt(1)==ch&&board[2].charAt(0)==ch){
            res[2]++;
        }
        return res;
    }
}
