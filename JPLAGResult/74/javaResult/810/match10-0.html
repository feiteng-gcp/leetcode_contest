class Solution {
    public boolean validTicTacToe(String[] board) {
        char[] values = new char[9];
        int x = 0;
        int o = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                values[i * 3 + j] = board[i].charAt(j);
                if (values[i * 3 + j] == 'X') {
                    x++;
                } else if (values[i * 3 + j] == 'O') {
                    o++;
                }
            }
        }
        if (o > x || x > o + 1) {
            return false;
        }
        
        int[][] answers = new int[][] {{0, 1, 2}, {3, 4, 5}, {6, 7, 8}, {0, 3, 6}, {1, 4, 7}, {2, 5, 8}, {0, 4, 8}, {2, 4, 6}};
        int ones = 0;
        int zeros = 0;
        for (int[] ans: answers) {
            if (values[ans[0]] == values[ans[1]] && values[ans[1]] == values[ans[2]]) {
                if (values[ans[0]] == 'X') {
                    ones++;
                } else if (values[ans[0]] == 'O') {
                    zeros++;
                }
            }   
        }
        if (zeros + ones > 1 || (zeros == 1 && x > o) || (ones == 1 && x == o)) {
            return false;
        }
        return true;
    }
}
