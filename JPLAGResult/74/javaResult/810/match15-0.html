class Solution {
    public boolean validTicTacToe(String[] board) {
        int numX = 0;
        int numO = 0;
        for(int i = 0; i < 3; i++){
            String cur = board[i];
            for(int j = 0; j< 3; j++ ){
                char ch = cur.charAt(j);
                if(ch == 'X'){
                    numX++;
                }else if(ch == 'O'){
                    numO++;
                }
            }
        }
        if(!(numO==numX || numO+1==numX)) return false;
        int success = 0;
        char winner = ' ';
        for(int i = 0; i < 3; i++){
            if(board[i].charAt(0)!=' ' && board[i].charAt(0)==board[i].charAt(1) && board[i].charAt(1)==board[i].charAt(2)){
                success++;
                winner = board[i].charAt(0);
            }
            if(board[0].charAt(i)!=' ' && board[0].charAt(i)==board[1].charAt(i) && board[1].charAt(i)==board[2].charAt(i)){
                success++;
                winner = board[0].charAt(i);
            }
        }
        if(board[0].charAt(0)!=' ' && board[0].charAt(0)==board[1].charAt(1) && board[1].charAt(1)==board[2].charAt(2)){
            success++;
            winner = board[0].charAt(0);
        }
        if(board[0].charAt(2)!=' ' && board[0].charAt(2)==board[1].charAt(1) && board[1].charAt(1)==board[2].charAt(0)){
            success++;
            winner = board[0].charAt(2);
        }
        if(success>1) return false;
        if(success == 1){
            if(winner == 'X' && numX-numO!=1) return false;
            else if(winner == 'O' && numX!=numO) return false;
        }
        
        return true;
    }
}
