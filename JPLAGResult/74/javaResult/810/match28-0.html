class Solution {
    public boolean validTicTacToe(String[] board) {
        if(board == null || board.length == 0){
            return false;
        }
        int arr[] =count(board);
        if(arr[0]-arr[1] == 0 ){
           boolean f1 = verify(board,'X');
            boolean f2 = verify(board,'O');
            if(f1){
                return false;
            }
            return true;
        }
        else if(arr[0]-arr[1] == 1){
           boolean f1 = verify(board,'X');
            boolean f2 = verify(board,'O');
            if((!f2 && f1) ||(!f1 && !f2)){
                return true;
            }
        }
        return false;
    }
    public int[] count(String[] board){
        int res[] = new int[2];
        for(String st:board){
            for(char ch:st.toCharArray()){
                if(ch == 'X')
                    res[0]++;
                else if(ch == 'O')
                    res[1]++;
            }
        }
        return res;
    }
    public boolean verify(String[] board,char ch){
        //boolean flag = true;
        for(int i=0;i<board.length;i++){
            int count =0;
            for(int j=0;j<board[i].length();j++){
                if(board[i].charAt(j) == ch)
                    count++;
            }
            if(count == board.length){
                return true;
            }
        }
        for(int j=0;j<3;j++){
            int count =0;
            for(int i=0;i<3;i++){
                if(board[i].charAt(j) == ch){
                    count++;
                }
            }
            if(count == 3)
                return true;
        }
        int count =0;
        for(int i=0;i<3;i++){
            if(board[i].charAt(i)==ch)
                count++;
        }
        if(count == 3)
            return true;
        count =0;
        for(int i=0;i<3;i++){
            if(board[i].charAt(2-i) ==ch)
                count++;
        }
        if(count == 3){
            return true;
        }
        return false;
    }
}
