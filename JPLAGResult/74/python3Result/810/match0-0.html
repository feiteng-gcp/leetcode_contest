class Solution:
    def validTicTacToe(self, board):
        """
        :type board: List[str]
        :rtype: bool
        """
        def check_win(char):
            for i in range(3):
                check_row = True
                check_col = True
                for j in range(3):
                    if board[i][j] != char:
                        check_row = False
                    if board[j][i] != char:
                        check_col = False
                if check_row or check_col:
                    return True
            check1 = True
            check2 = True
            for i in range(3):
                if board[i][i] != char:
                    check1 = False
                if board[i][2 - i] != char:
                    check2 = False
            if check1 or check2:
                return True
            return False
        O = 0
        X = 0
        for i in range(3):
            for j in range(3):
                if board[i][j] == "O":
                    O += 1
                elif board[i][j] == "X":
                    X += 1
        if O == X:
            return not check_win("X")
        elif X == O + 1:
            return not check_win("O")
        else:
            return False
