class Solution {
public:
    
    long long sum(long long mid){
        long long ans = 0;
        while(mid > 0){
            ans += mid / 5;
            mid /= 5;
        }
        
        return ans;
    }
    
    int preimageSizeFZF(int K) {
        
        long long lo = 0, hi = LLONG_MAX;
        while(lo <= hi){
            long long mid = lo + (hi - lo) / 2;
            long long t = sum(mid);
            if(t > K) hi = mid - 1;
            else if(t < K) lo = mid + 1;
            else return 5;
        }
        
        return 0;
    }
};
