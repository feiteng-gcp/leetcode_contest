typedef long long LL;typedef long double LD;
int DEBUG=0,MULTI=0,GCJ=0;

class Solution {
  public:
  #define LL long long
  
  LL cnt(LL val) {
    LL res = 0;
    while (val) {
      res += val / 5;
      val /= 5;
    }
    return res;
  }
  
  int preimageSizeFZF(int K) {
    LL res = 0;
    {
      LL le = 0, ri = (LL)1e9;
      while (le < ri) {
        LL mid = (le + ri) / 2;
        if (cnt(mid) >= K) {
          ri = mid;
        }
        else le = mid + 1;
      }
      DB(le);
      res -= le;
    }
    {
      LL le = 0, ri = (LL)1e9;
      while (le < ri) {
        LL mid = (le + ri) / 2;
        if (cnt(mid) > K) {
          ri = mid;
        }
        else le = mid + 1;
      }
      DB(le);
      res += le;
    }
    return res;
  }
};
