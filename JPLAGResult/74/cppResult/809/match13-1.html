class Solution {
public:
    int preimageSizeFZF(int K) 
    {
        long long left=0;
        long long right=INT_MAX;
        long long mid;
        
        while (left<right)
        {
            mid = left+(right-left)/2;
            int N = findTrailingZeros(mid);
            //cout<<left<<" "<<right<<" "<<mid<<" "<<N<<endl;
            
            if (N<K)            
                left = mid+1;                            
            else
                right = mid;
        }
        
        if (findTrailingZeros(left)!=K)            
            return 0;
        

        int count=1;
        while (findTrailingZeros(left+1)==K)
        {
            left++;
            count++;
        }
        return count;
        
        
    }
    
    int findTrailingZeros(int n)
    {    
        int count = 0;
     
        for (int i=5; n/i>=1; i *= 5)
          count += n/i;
 
        return count;
    }
};
