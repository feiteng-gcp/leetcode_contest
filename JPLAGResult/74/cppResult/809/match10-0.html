class Solution {
public:
    int cal(int x)
    {
      int ret;
      ret=0;
      while (x>0)
      {
        x/=5;
        ret+=x;
      }
      return ret; 
    }
    int preimageSizeFZF(int k) {
      int left,right,mid,ll,rr;
      left=-1;right=1000000000;
      while (right-left>1)
      {
        mid=(left+right)/2;
        if (cal(mid)<k) left=mid;
        else right=mid;
      }
      ll=right;
      left=-1;right=1000000000;
      while (right-left>1)
      {
        mid=(left+right)/2;
        if (cal(mid)<=k) left=mid;
        else right=mid;
      }
      rr=left;
      return rr-ll+1;
    }
};
