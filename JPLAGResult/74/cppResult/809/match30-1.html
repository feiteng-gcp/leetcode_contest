class Solution {
public:
    long long check(long long n) {
        long long ret = 0;
        while (n) {
            ret += n / 5;
            n /= 5;
        }
        return ret;
    }

    int preimageSizeFZF(int K) {
      long long low = 0, high = 1LL << 60;
        while (low != high) {
            long long mid = (low + high + 1) / 2;
            if (check(mid) >= K + 1) {
              high = mid - 1;
            } else {
              low = mid;
            }
        }
        long long B = high;
        low = 0, high = 1LL << 60;
        while (low != high) {
            long long mid = (low + high) / 2;
            if (check(mid) <= K - 1) {
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        if (check(high) > K - 1) --high;
        long long A = high;
        cout  << A << " " << B << endl;
        return max(0LL, B - A);
    }
};
