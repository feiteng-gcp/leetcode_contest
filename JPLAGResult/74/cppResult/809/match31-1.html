class Solution {

    int cnt(int x) {
        int res = 0;
        while (x) {
            res += x / 5;
            x /= 5;
        }
        return res;
    }
public:
    int preimageSizeFZF(int K) {
        int lower, upper;

        int l = 0, r = (int)1e9;
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (cnt(mid) >= K) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        lower = l;

        l = 0, r = (int)1e9;
        while (l < r) {
            int mid = r - (r - l) / 2;
            if (cnt(mid) <= K) {
                l = mid;
            } else {
                r = mid - 1;
            }
        }
        upper = l;


        return upper - lower + 1;
    }
};
