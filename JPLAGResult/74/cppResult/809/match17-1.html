class Solution {
public:

    long long f(int k) {
        long long ans = k;
        while (k > 0) {
            k /= 5;
            ans += k;
        }
        return ans;
    }

    int preimageSizeFZF(int K) {
        long long left = 0;
        long long right = 10000000000;
        while (left <= right) {
            long long mid = (left + right) / 2;
            long long res = f(mid);
            if (res == K) return 5;
            if (res > K) right = mid - 1;
            else left = mid + 1;
        }
        return 0;
    }
};
