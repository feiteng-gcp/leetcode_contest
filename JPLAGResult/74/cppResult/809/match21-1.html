class Solution {
public:
    long long cal(int x)

{
	long long res=0;
	while(x)
	{
		res=res+x/5;
		x=x/5;
	}
	return res;
}
    int preimageSizeFZF(int K) {
        int cur=0;
	int l=0,r=200000000;
	while(l<=r)
	{
		int m=(l+r)/2;
		if(cal(5*m)<K)l=m+1;
		else r=m-1;
	}
	if(cal(5*l)==K)return 5;
	else return 0;
    }
};
