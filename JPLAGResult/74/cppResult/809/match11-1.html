class Solution {
public:
    
    int count(int n) {
        int ret = 0;
        while (n) {
            n /= 5;
            ret += n;
        }
        return ret;
    }
    
    int preimageSizeFZF(int K) {
        if (K == 0) return 5;
        int start = 5;
        int end = 1000000000;
        bool found = false;
        while (start < end) {
            int mid = start + (end - start) / 2;
            int zeroes = count(mid);
            if (zeroes == K) {
                return 5;
            }
            else if (zeroes > K) {
                end = mid-1;
            }
            else {
                start = mid+1;
            }
        }
        return 0;
    }
};
