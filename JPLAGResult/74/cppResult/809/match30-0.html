class Solution {
public:
    int preimageSizeFZF(int K) {   
        long long l = 0, r = 1e18;
        while(l <= r) {
            long long mid = (l + r) / 2;
            long long cnt = 0;
            for(long long i = 5; i <= mid; i *= 5) cnt += mid / i;
            if(cnt < K) l = mid + 1;
            else if(cnt > K) r = mid - 1;
            if(cnt == K) return 5;
        }
        return 0;
    }
};
