class Solution {
public:
    int preimageSizeFZF(int K) {
        int i = 0;
        int j = INT_MAX;
        while (i <= j) {
            int mid = i + (j - i) / 2;
            int temp = findCount(mid);
            if (temp < K) i = mid + 1;
            else j = mid - 1;
        }
        if (findCount(i) != K) return 0;
        int lowerbound = i;
        j = INT_MAX;
        while (i <= j) {
            int mid = i + (j - i) / 2;
            int temp = findCount(mid);
            if (temp > K) j = mid - 1;
            else i = mid + 1;
        }
        return j - lowerbound + 1;
    }
    

    int findCount(int val) {
        int result = 0;
        while (val > 0) {
            result = result + val / 5;
            val = val / 5;
        }
        return result;
    }
};
