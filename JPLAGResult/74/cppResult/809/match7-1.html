class Solution {
public:
  int preimageSizeFZF(int K) {
    if (K < 0) return 0;
    if (K < 5) return 5;
    int lo = 0, hi = K * 6;
    while (hi > lo) {
      int mid = (lo + hi) / 2;
      int n = numZeros(mid);
      if (n == K) return 5;
      if (n > K) {
        hi = mid;
      } else {
        lo = mid + 1;
      }
    }
    return 0;
  }
  

  int numZeros(int n) {
    int ret = 0;
    while (n > 0) {
      n /= 5;
      ret += n;
    }
    return ret;
  }
};
