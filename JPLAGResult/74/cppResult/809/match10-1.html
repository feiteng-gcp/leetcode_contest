class Solution {
public:
    typedef long long LL;
LL check5(LL n)
{
    LL ans = 0;
    while(n)
    {
        ans += n / 5;
        n /= 5;
    }
    return ans;
}
int preimageSizeFZF(int K)
{
    LL L = 0, R = 0;

    LL l = 0, r = 5e9;
    while(l < r)
    {
        LL m = l + (r - l) / 2;
        if(check5(m) < K) l = m + 1;
        else r = m;
    }
    L = l;
    l = 0; r = 5e9;
    while(l < r)
    {
        LL m = l + (r - l) / 2;
        if(check5(m) <= K) l = m + 1;
        else r = m;
    }
    R = l;
    return (int)(R - L);
}
};
