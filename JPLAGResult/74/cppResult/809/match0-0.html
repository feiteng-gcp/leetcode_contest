class Solution {
public:
    
    long long getZeros(long long x){
        long long ans = 0;
        for(long long f = 5 ; f <= x; f*=5){
            ans += x / f;
        }
        return ans;
    }
    int preimageSizeFZF(int K) {
        
        
        long long lower = 0 , higher = 1000000000000000LL;
        while(lower + 1 < higher){
            long long middle = (lower + higher) / 2;
            if(getZeros(middle) <= K){
                lower = middle;
            }else{
                higher = middle;
            }
        }
        long long y = lower;
        if(getZeros(y) != K){
            return 0;
        }
        
        lower = -1 , higher = 1000000000000000LL;
        while(lower + 1 < higher){
            long long middle = (lower + higher) / 2;
            if(getZeros(middle) >= K){
                higher = middle;
            }else{
                lower = middle;
            }
        }
        long long x = higher;
        
        cout << x<<" "<<y<<endl;
        return y - x + 1;
        
        
    }
};
