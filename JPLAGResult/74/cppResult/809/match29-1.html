class Solution {
public:
    long long count5(long long x){
        return (x < 5) ? 0 : ((x/5) + count5(x/5));
    }
    
    long long getFirst(long long k){
        long long le = 0LL;
        long long ri = LONG_MAX;
        
        while (le < ri){
            long long mid = (le+ri) >> 1LL;
            
            if (count5(mid) >= k) ri = mid;

            else le = mid+1LL;
        }
        
        return ri;
    }
    
    int preimageSizeFZF(int K) {
        long long a = getFirst(K);
        long long b = getFirst(K+1);
        
        if (count5(a) != K)
            return 0;
        
        return b-a;
    }
};
