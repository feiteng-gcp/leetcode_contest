typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T> >;

const ll MOD=1e9+7;

class Solution {
public:
    int preimageSizeFZF(int K) {
      ll l=0,r=1e10;
      while(l<r){
	ll m=(l+r+1)/2;
	ll s=0,x=m;

	while(x) x/=5,s+=x;
	if(s<=K){
	  l=m;
	}else{
	  r=m-1;
	}
      }
      ll k1=l;
      l=0,r=1e10;
      while(l<r){
	ll m=(l+r)/2;
	ll s=0,x=m;

	while(x) x/=5,s+=x;
	if(s<K){
	  l=m+1;
	}else{
	  r=m;
	}
      }
      ll k2=l;
      cout<<k2<<' '<<k1<<'\n';
      return k1-k2+1;
    }
};
