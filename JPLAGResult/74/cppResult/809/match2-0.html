class Solution {
public:
    int getZero(long long x){
        int ans = 0;
        long long p = 5;
        while(x >= p){
            ans += x / p;
            p = p * 5;
        }
        return ans;
    }
    
    int preimageSizeFZF(int K) {
        long long hi = 1e10;
        long long lo = 0;
        
        while(lo < hi-1){
            long long mid = lo + (hi - lo) / 2;
            if(getZero(mid) > K){
                hi = mid-1;
            } else {
                lo = mid;
            }
        }
        //cout << lo << endl;
        
        int ans= 0;
        for(long long a = max((long long)0, lo - 5); a <= lo + 5; a++){
            if(getZero(a) == K) ans++;
        }
        return ans;
    }
};
