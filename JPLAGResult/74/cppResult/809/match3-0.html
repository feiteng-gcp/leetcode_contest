class Solution {
public:
    long long zeros(long long n) {
        long long two = 0, five = 0;
        for (long long i = 2; i <= n; i *= 2) {
            two += n / i;
        }
        for (long long i = 5; i <= n; i *= 5) {
            five += n / i;
        }
        return min(two, five);
    }
    long long cnt(int K) {
        long long l = 0, r = 1ll << 50;
        while (l < r) {
            long long m = (l + r) / 2;
            if (zeros(m) < K) {
                l = m + 1;
            } else {
                r = m;
            }
        }
        return l;
    }
    int preimageSizeFZF(int K) {
        return cnt(K + 1) - cnt(K);
    }
};
