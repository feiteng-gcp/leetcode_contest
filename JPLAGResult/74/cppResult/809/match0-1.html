class Solution {
public:
    int countzeros(int n) {
        int count = 0;
        while(n) {
            count += n / 5;
            n /= 5;
        }
        return count;
    }
    int preimageSizeFZF(int K) {
        int l = 0, r = 1 << 30;
        while(l < r) {
            int mid = l + r >> 1;
            int midc = countzeros(mid);
            if(midc == K) {
                int midl = mid - 1, midr = mid + 1;
                while(midl >= 0 && countzeros(midl) == K) midl--;
                while(countzeros(midr) == K) midr++;
                return midr - midl - 1;
            } else if(midc < K) {
                l = mid + 1;
            } else
                r = mid;
        }
        return 0;
    }
};
