class Solution {
public:
    bool validTicTacToe(vector<string>& board) {
        if (board.size() != 3 || board[0].size() != 3)
        {
            return false;
        }
        int x = 0, o= 0;
        for (int i = 0; i < 3; i++)
        {
            if (board[i].size() != 3)
            {
                return false;
            }
            for (int j = 0; j < 3; j++)
            {
                if (board[i][j] == 'X')
                {
                    x++;
                }
                else if (board[i][j] == 'O')
                {
                    o++;
                }
                else if (board[i][j] != ' ')
                {
                    return false;
                }
            }
        }
//        cout << x << " " << o << endl;
        if (x < o || x > o+1)
        {
            return false;
        }
        int X = three(board, 'X');
        int O = three(board, 'O');
//        cout << x << " " << o << endl;
        if (X > 1 || O > 1 || (X == 1 && O == 1))
        {
            return false;
        }
        if (X == 1 )
        {
            return x == o + 1;
        }
        if (O == 1)
        {
            return x == o;
        }
        return true;
    }
    int three(vector<string>& board, char c)
    {
        int res = 0;
        for (int i = 0; i < 3; i++)
        {
            bool column = true, row = true;
            for (int j = 0; j < 3; j++)
            {
                if (board[i][j] != c)
                {
                    row = false;
                }
                if (board[j][i] != c)
                {
                    column = false;
                }
            }
            if (column) res ++;
            if (row) res ++;
        }
        bool f1 = true, f2 = true;
        for (int i = 0; i < 3; i++)
        {
            if (board[i][i] != c)
            {
                f1 = false;
            }
            if (board[i][2-i] != c)
            {
                f2 = false;
            }
        }
        if (f1) res++;
        if (f2) res++;
        return res;
    }
};
