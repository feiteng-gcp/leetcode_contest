class Solution {
public:
	char b[3][3];

	bool check(int x, int y, int dx, int dy, int c) {
		for (int i = 0; i < 3; ++ i) {
			if (b[x][y] != c) {
				return 0;
			}
			x += dx, y += dy;
		}
		return 1;
	}
	int win(char a) {
		for (int i = 0; i < 3; ++ i) {
			if (check(0, i, 1, 0, a)) {
				return 1;
			}
			if (check(i, 0, 0, 1, a)) {
				return 1;
			}
		}
		if (check(0, 0, 1, 1, a)) {
			return 1;
		}
		if (check(0, 2, 1, -1, a)) {
			return 1;
		}
		return 0;
	}

    bool validTicTacToe(vector<string>& board) {
		int c0(0), c1(0);
		for (int i = 0; i < 3; ++ i) {
			for (int j = 0; j < 3; ++ j) {
				b[i][j] = board[i][j];
				if (b[i][j] == 'O') {
					++ c0;
				}
				if (b[i][j] == 'X') {
					++ c1;
				}
			}
		}
		if (c0 != c1 && c0 + 1 != c1) {
			return 0;
		}
		int wx = win('X');
		int wo = win('O');
		if (win('X') + win('O') == 2) {
			return 0;
		}
		if (wx) {
			return c1 == c0 + 1;
		}
		if (wo) {
			return c0 == c1;
		}
		return 1;
	}
};
