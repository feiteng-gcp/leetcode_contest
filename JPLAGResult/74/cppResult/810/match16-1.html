class Solution {
public:
    bool validTicTacToe(vector<string>& board) {
        int x = 0, o = 0;
        for (string s: board) {
            for (char c : s) {
                if (c == 'X') {
                    x++;
                }
                if (c == 'O') {
                    o++;
                }
            }
        }
        if ((x - o) > 1 || (x - o) < 0) {
            return false;

        }
        bool flagX = false;
        bool flagO = false;
        for (int i = 0; i < 3; ++i) {
            if (board[0][i] == board[1][i] && board[1][i] == board[2][i] && board[0][i] == 'X') {
                flagX = true;
            }
            if (board[i][0] == board[i][1] && board[i][1] == board[i][2] && board[i][0] == 'X') {
                flagX = true;
            }
            if (board[0][i] == board[1][i] && board[1][i] == board[2][i] && board[0][i] == 'O') {
                flagO = true;
            }
            if (board[i][0] == board[i][1] && board[i][1] == board[i][2] && board[i][0] == 'O') {
                flagO = true;
            }   
        }
        if (board[0][0] == board[1][1] && board[1][1] == board[2][2] && board[0][0] == 'X') {
                flagX = true;
            }
        if (board[0][2] == board[1][1] && board[1][1] == board[2][0] && board[1][1] == 'X') {
                flagX = true;
        }
        if (board[0][0] == board[1][1] && board[1][1] == board[2][2] && board[0][0] == 'O') {
                flagO = true;
            }
        if (board[0][2] == board[1][1] && board[1][1] == board[2][0] && board[1][1] == 'O') {
                flagO = true;
        }

        if (flagX && flagO) {
            return false;
        }
        if (flagX && x == o) {
            return false;
        }
        if (flagO && x > o) {
            return false;
        }
        return true;
    }
};
