class Solution {
public:
    bool isWin(vector<string>& board, char c) 
    {
        for (int i = 0; i < 3; i++)
        {
            if (board[i][0] == c && board[i][1] == c && board[i][2] == c)
                return true;
            if (board[1][i] == c && board[2][i] == c && board[0][i] == c)
                return true;
        }
        
        if (board[0][0] == c && board[1][1] == c && board[2][2] == c)
            return true;
        
        if (board[0][2] == c && board[1][1] == c && board[2][0] == c)
            return true;
        
        return false;
    }
    bool validTicTacToe(vector<string>& board) 
    {
        int cnt1 = 0, cnt2 = 0;
        for (int i = 0; i < 3; i++)
        {
            for (int j = 0; j < 3; j++)
            {
                if (board[i][j] == 'X')
                    cnt1++;
                else if (board[i][j] == 'O')
                    cnt2++;
            }
        }
        
        if (cnt2 > cnt1 || cnt1 > cnt2 + 1) 
            return false;
        
        if (cnt2 == cnt1 && isWin(board, 'X'))
            return false;
        
        if (cnt2 < cnt1 && isWin(board, 'O'))
            return false;
        
        return true;
    }
};
