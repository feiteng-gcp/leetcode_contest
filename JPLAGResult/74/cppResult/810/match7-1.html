class Solution {
public:
    bool validTicTacToe(vector<string>& board) {
        B = board;
        int nx = getCount('X'), no = getCount('O');
        bool xWin = isWin('X'), oWin = isWin('O');
        
        if (no > nx) return false;
        if (nx > no + 1) return false;
        if (nx == no && xWin) return false;
        if (nx > no && oWin) return false;
        return true;
    }
    
private:    
    vector<string> B;
    
    int getCount(char ch) {

        int res = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (B[i][j] == ch) res++;
            }
        }
        return res;
    } 
    

    bool isWin(char ch) {
        for (int i = 0; i < 3; i++) {
            if (B[i][0] == ch && B[i][1] == ch && B[i][2] == ch) return true;
            if (B[0][i] == ch && B[1][i] == ch && B[2][i] == ch) return true;
        }
        
        if (B[0][0] == ch && B[1][1] == ch && B[2][2] == ch) return true;
        if (B[0][2] == ch && B[1][1] == ch && B[2][0] == ch) return true;
        return false;
    }
};
