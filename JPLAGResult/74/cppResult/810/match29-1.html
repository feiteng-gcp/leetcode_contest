class Solution {
public:
    bool validTicTacToe(vector<string>& board)
{
    int cntX = 0, cntO = 0;
    for(int i = 0; i < 3; i++)
        for(int j = 0; j < 3; j++)
        {
            if(board[i][j] == 'O') cntO++;
            else if(board[i][j] == 'X') cntX++;
        }
    if(cntX != cntO && cntX != cntO + 1) return false;
    int winX = 0, winO = 0;
    if(board[0][0] == board[0][1] && board[0][0] == board[0][2])
    {winX += board[0][0] == 'X'; winO += board[0][0] == 'O';}
    if(board[1][0] == board[1][1] && board[1][0] == board[1][2])
    {winX += board[1][0] == 'X'; winO += board[1][0] == 'O';}
    if(board[2][0] == board[2][1] && board[2][0] == board[2][2])
    {winX += board[2][0] == 'X'; winO += board[2][0] == 'O';}
    if(board[0][0] == board[1][0] && board[0][0] == board[2][0])
    {winX += board[0][0] == 'X'; winO += board[0][0] == 'O';}
    if(board[0][1] == board[1][1] && board[0][1] == board[2][1])
    {winX += board[0][1] == 'X'; winO += board[0][1] == 'O';}
    if(board[0][2] == board[1][2] && board[0][2] == board[2][2])
    {winX += board[0][2] == 'X'; winO += board[0][2] == 'O';}
    if(board[0][0] == board[1][1] && board[0][0] == board[2][2])
    {winX += board[0][0] == 'X'; winO += board[0][0] == 'O';}
    if(board[0][2] == board[1][1] && board[0][2] == board[2][0])
    {winX += board[0][2] == 'X'; winO += board[0][2] == 'O';}
    if((winX && winO) || (winX && cntX == cntO) || (winO && cntX == cntO + 1)) return false;
    return true;
}
};
