class Solution {
public:
    bool validTicTacToe(vector<string>& board) {

        int ocount = 0, xcount = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (board[i][j] == 'O') ocount++;
                else if (board[i][j] == 'X') xcount++;
            }
        }
        if (xcount != ocount && xcount != ocount + 1) return false;
        bool oline = false, xline = false;

        for (int i = 0; i < 3; i++) {
            if (board[i][0] == 'O' && board[i][0] == board[i][1] && board[i][0] == board[i][2]) oline = true;
            if (board[i][0] == 'X' && board[i][0] == board[i][1] && board[i][0] == board[i][2]) xline = true;
            if (board[0][i] == 'O' && board[0][i] == board[1][i] && board[0][i] == board[2][i]) oline = true;
            if (board[0][i] == 'X' && board[0][i] == board[1][i] && board[0][i] == board[2][i]) xline = true;
        }
        if (board[0][0] == 'O' && board[0][0] == board[1][1] && board[1][1] == board[2][2]) oline = true;
        if (board[0][0] == 'X' && board[0][0] == board[1][1] && board[1][1] == board[2][2]) xline = true;
        if (board[0][2] == 'O' && board[0][2] == board[1][1] && board[1][1] == board[2][0]) oline = true;
        if (board[0][2] == 'X' && board[0][2] == board[1][1] && board[1][1] == board[2][0]) xline = true;
        if (oline && xline) return false;
        if (xline && xcount != ocount + 1) return false;
        if (oline && xcount != ocount) return false;
        return true;
    }
};
