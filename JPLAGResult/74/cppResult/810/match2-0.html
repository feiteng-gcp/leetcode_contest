class Solution {
public:
    bool validTicTacToe(vector<string>& board) {
        int xcnt = 0;
        for(auto str : board){
            for(auto c : str){
                if(c == 'X') xcnt++;
                if(c == 'O') xcnt--;
            }
        }
        
        if(xcnt != 1 && xcnt != 0) return false;
        
        bool xwin = false, owin = false;
        for(auto str: board){
            if(str == "XXX") xwin = true;
            if(str == "OOO") owin = true;
        }
        
        for(int k=0; k<3; k++){
            if(board[0][k] == 'X' && board[1][k] == 'X' && board[2][k] == 'X') xwin = true;
            if(board[0][k] == 'O' && board[1][k] == 'O' && board[2][k] == 'O') owin = true;
        }
        if(board[0][0] == 'X' && board[1][1] == 'X' && board[2][2] == 'X') xwin = true;
        if(board[0][2] == 'X' && board[1][1] == 'X' && board[2][0] == 'X') xwin = true;
        if(board[0][0] == 'O' && board[1][1] == 'O' && board[2][2] == 'O') owin = true;
        if(board[0][2] == 'O' && board[1][1] == 'O' && board[2][0] == 'O') owin = true;
        
        if(xwin && owin) return false;
        if(xwin && xcnt == 0) return false;
        if(owin && xcnt == 1) return false;
        return true;
    }
};
