class Solution {
public:
    bool validTicTacToe(vector<string>& board) {

        int cnto=0,cntx=0;
	for(int i=0;i<3;i++)
	{
		for(int j=0;j<3;j++)
		{
			if(board[i][j]=='O')cnto++;
			else if(board[i][j]=='X')cntx++;
		}
	}
	if(cnto>cntx||cntx-cnto>1)return false;
	bool yeso=0,yesx=0;
	int mido=0,midx=0;
	int cnt=0;
	//1
	if(board[0][1]==board[0][0]&&board[0][2]==board[0][0])cnt++;
	if(board[1][1]==board[0][0]&&board[2][2]==board[0][0])cnt++;
	if(board[1][0]==board[0][0]&&board[2][0]==board[0][0])cnt++;
	if(cnt)
	{
		if(board[0][0]=='O')
		{
			yeso=true;
			if(cnt>1)mido++;
		}
		else if(board[0][0]=='X')
		{
			yesx=true;
			if(cnt>1)midx++;
		}
	}
	//2
	cnt=0;
	if(board[0][0]==board[0][1]&&board[0][2]==board[0][1])cnt++;
	if(board[1][1]==board[0][1]&&board[2][1]==board[0][1])cnt++;
	if(cnt)
	{
		if(board[0][1]=='O')
		{
			yeso=true;
			if(cnt>1)mido++;
		}
		else if(board[0][1]=='X')
		{
			yesx=true;
			if(cnt>1)midx++;
		}
	}
	//3
	cnt=0;
	if(board[0][0]==board[0][2]&&board[0][1]==board[0][2])cnt++;
	if(board[1][1]==board[0][2]&&board[2][0]==board[0][2])cnt++;
	if(board[1][2]==board[0][2]&&board[2][2]==board[0][2])cnt++;
	if(cnt)
	{
		if(board[0][2]=='O')
		{
			yeso=true;
			if(cnt>1)mido++;
		}
		else if(board[0][2]=='X')
		{
			yesx=true;
			if(cnt>1)midx++;
		}
	}
	//4
	cnt=0;
	if(board[1][1]==board[1][0]&&board[1][2]==board[1][0])cnt++;
	if(board[0][0]==board[1][0]&&board[2][0]==board[1][0])cnt++;
	if(cnt)
	{
		if(board[1][0]=='O')
		{
			yeso=true;
			if(cnt>1)mido++;
		}
		else if(board[1][0]=='X')
		{
			yesx=true;
			if(cnt>1)midx++;
		}
	}
	//5
	cnt=0;
	if(board[0][1]==board[1][1]&&board[2][1]==board[1][1])cnt++;
	if(board[1][0]==board[1][1]&&board[1][2]==board[1][1])cnt++;
	if(board[0][0]==board[1][1]&&board[2][2]==board[1][1])cnt++;
	if(board[0][2]==board[1][1]&&board[2][0]==board[1][1])cnt++;
	if(cnt)
	{
		if(board[1][1]=='O')
		{
			yeso=true;
			if(cnt>1)mido++;
		}
		else if(board[1][1]=='X')
		{
			yesx=true;
			if(cnt>1)midx++;
		}
	}
	//6
	cnt=0;
	if(board[1][0]==board[1][2]&&board[1][1]==board[1][2])cnt++;
	if(board[0][2]==board[1][2]&&board[2][2]==board[1][2])cnt++;
	if(cnt)
	{
		if(board[1][2]=='O')
		{
			yeso=true;
			if(cnt>1)mido++;
		}
		else if(board[1][2]=='X')
		{
			yesx=true;
			if(cnt>1)midx++;
		}
	}
	//7
	cnt=0;
	if(board[2][1]==board[2][0]&&board[2][2]==board[2][0])cnt++;
	if(board[1][1]==board[2][0]&&board[0][2]==board[2][0])cnt++;
	if(board[1][0]==board[2][0]&&board[0][0]==board[2][0])cnt++;
	if(cnt)
	{
		if(board[2][0]=='O')
		{
			yeso=true;
			if(cnt>1)mido++;
		}
		else if(board[2][0]=='X')
		{
			yesx=true;
			if(cnt>1)midx++;
		}
	}
	//8
	cnt=0;
	if(board[2][0]==board[2][1]&&board[2][2]==board[2][1])cnt++;
	if(board[1][1]==board[2][1]&&board[0][1]==board[2][1])cnt++;
	if(cnt)
	{
		if(board[2][1]=='O')
		{
			yeso=true;
			if(cnt>1)mido++;
		}
		else if(board[2][1]=='X')
		{
			yesx=true;
			if(cnt>1)midx++;
		}
	}
	//9
	cnt=0;
	if(board[2][0]==board[2][2]&&board[2][1]==board[2][2])cnt++;
	if(board[1][1]==board[2][2]&&board[0][0]==board[2][2])cnt++;
	if(board[1][2]==board[2][2]&&board[0][2]==board[2][2])cnt++;
	if(cnt)
	{
		if(board[2][2]=='O')
		{
			yeso=true;
			if(cnt>1)mido++;
		}
		else if(board[2][2]=='X')
		{
			yesx=true;
			if(cnt>1)midx++;
		}
	}
	if((yeso&&yesx)||mido+midx>1)return false;
	else if(yesx&&cnto==cntx)return false;
    else if(yeso&&cntx==cnto+1)return false;
	return true;
    }
};
