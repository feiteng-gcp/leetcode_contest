class Solution {
public:
    bool validTicTacToe(vector<string>& board) {
        int x_num_row[3] = {0};
        int x_num_col[3] = {0};
        int x_num = 0;
        
        int o_num_row[3] = {0};
        int o_num_col[3] = {0};
        int o_num = 0;
        
        for(int i = 0; i < 3; i++)
            for(int j = 0; j < 3; j++)
            {
                if (board[i][j] == 'X')
                {
                    x_num++;
                    x_num_row[i]++;
                    x_num_col[j]++;
                }
                else if (board[i][j] == 'O')
                {
                    o_num++;
                    o_num_row[i]++;
                    o_num_col[j]++;
                }
            }
        
        if (o_num > x_num) return false;
        if(std::abs(x_num - o_num) >= 2) return false;
        

        for (int i = 0; i < 3; i++)
        {
            if ((x_num_row[i] == 3 || x_num_col[i] == 3) && o_num - x_num >= 0) return false;
            if ((o_num_row[i] == 3 || o_num_col[i] == 3) && x_num - o_num >= 1) return false;
        }
        
        if ((board[0][0] == 'X' && board[1][1] == 'X' && board[2][2] == 'X') && o_num - x_num >= 0) return false;
        if ((board[0][2] == 'X' && board[1][1] == 'X' && board[2][0] == 'X') && o_num - x_num >= 0) return false;
        if ((board[0][0] == 'O' && board[1][1] == 'O' && board[2][2] == 'O') && x_num - o_num >= 1) return false;
        if ((board[0][2] == 'O' && board[1][1] == 'O' && board[2][0] == 'O') && x_num - o_num >= 1) return false;
        
        return true;
    }
};
