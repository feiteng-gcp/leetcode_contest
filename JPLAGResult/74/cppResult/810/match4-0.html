class Solution {
public:
bool validTicTacToe(vector<string>& board) {
        int xnum = 0, ynum = 0;
        for(int i = 0; i < 3;  i ++)
        {
            for(int j = 0; j < 3; j ++)
            {
                if(board[i][j] == 'X')
                    xnum ++;
                if(board[i][j] == 'O')
                    ynum ++;
            }
        }
        if(xnum - ynum > 1 || xnum - ynum < 0)
            return false;
        int ax = 0, ay = 0;
        for(int i = 0; i < 3; i ++) // 同行
        {
            if(board[i][0] == board[i][1] && board[i][1] == board[i][2] && board[i][2] != ' ')
                if(board[i][2] == 'X')
                  ax ++;
                else
                  ay ++;
            if(board[0][i] == board[1][i] && board[1][i] == board[2][i] && board[2][i] != ' ')
                if(board[2][i] == 'X')
                    ax ++;
                else
                    ay ++;
        }
        if(board[0][0] == board[1][1] && board[1][1] == board[2][2] && board[2][2] != ' ')
                if(board[2][2] == 'X')
                    ax ++;
                 else
                    ay ++;
        if(board[2][0] == board[1][1] && board[1][1] == board[0][2] && board[0][2] != ' ')
        {
            if(board[0][2] == 'X')
                ax ++;
            else
                ay ++;
        }
          
        if(ax + ay == 0)
            return true;
        if(ax + ay == 1)
        {
            
            if(ax == 1 && xnum - ynum == 1)
                return true;
            if(ay == 1 && xnum == ynum)
                return true;
            else
                return false;
        }
        return false;
    }
};
