class Solution {
public:
    bool validTicTacToe(vector<string>& board) {
        int nXs = 0;
        int nOs = 0;
        for (int i=0; i < 3; ++i) {
            for (int j=0; j < 3; ++j) {
                if (board[i][j] == 'X')
                    ++nXs;
                else if (board[i][j] == 'O')
                    ++nOs;
            }
        }
        if (nOs > nXs)
            return false;
        if (nXs >= (nOs+2))
            return false;
        char someoneWin = 0;
        for (int i=0; i < 3; ++i) {
            if (board[i][0] == 'X' && board[i][1] == 'X' && board[i][2] == 'X') {
                if (someoneWin)
                    return false;
                someoneWin = 'X';
            }
            if (board[i][0] == 'O' && board[i][1] == 'O' && board[i][2] == 'O') {
                if (someoneWin)
                    return false;
                someoneWin = 'O';
            }
            if (board[0][i] == 'X' && board[1][i] == 'X' && board[2][i] == 'X') {
                if (someoneWin)
                    return false;
                someoneWin = 'X';
            }    
            if (board[0][i] == 'O' && board[1][i] == 'O' && board[2][i] == 'O') {
                if (someoneWin)
                    return false;
                someoneWin = 'O';
            }   
        }
        for (int i=0; i < 2; ++i) {
            char c = (i == 0) ? 'X' : 'O';
            if (board[0][0] == c && board[1][1] == c && board[2][2] == c) {
                if (someoneWin)
                    return false;
                someoneWin = c;
            }
            if (board[0][2] == c && board[1][1] == c && board[2][0] == c) {
                if (someoneWin)
                    return false;
                someoneWin = c;
            }
        }  
        if (someoneWin == 'X' && nXs == nOs)
            return false;
        if (someoneWin == 'O' && nXs == (nOs+1))
            return false;
        return true;
    }
};
