class Solution {
public:
    bool validTicTacToe(vector<string>& brd) {
      int x=0,o=0;

      int xw=0,ow=0;
      for(int i=0;i<3;i++){
	for(int j=0;j<3;j++){
	  if(brd[i][j]=='X') x++;
	  else if(brd[i][j]=='O') o++;
	}
      }
      for(int i=0;i<3;i++){
	if(brd[i][0]==brd[i][1] && brd[i][0]==brd[i][2]){
	  if(brd[i][0]=='X') xw++;
	  else if(brd[i][0]=='O') ow++;
	}
	if(brd[0][i]==brd[1][i] && brd[0][i]==brd[2][i]){
	  if(brd[0][i]=='X') xw++;
	  else if(brd[0][i]=='O') ow++;
	}
      }
      if(brd[0][0]==brd[1][1] && brd[0][0]==brd[2][2]){
	if(brd[0][0]=='X') xw++;
	else if(brd[0][0]=='O') ow++;
      }
      if(brd[0][2]==brd[1][1] && brd[0][2]==brd[2][0]){
	if(brd[0][2]=='X') xw++;
	else if(brd[0][2]=='O') ow++;
      }
      if(x==o){
	if(xw>0) return 0;
	return 1;
      }else if(x==o+1){
	if(ow>0) return 0;
	return 1;
      }
      return 0;
    }
};
