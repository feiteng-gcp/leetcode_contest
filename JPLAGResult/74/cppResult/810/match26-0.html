class Solution {
public:
    
    int getWins(vector<string> &board,char type){
        
        int total = 0;
        for(int i = 0; i < 3; ++i){
            int c = 0;
            for(int j = 0; j < 3; ++j){
                if(board[i][j] == type){
                    c++;
                }
            }
            if(c == 3){
                total++;
            }
            
            c = 0;
            for(int j = 0; j < 3; ++j){
                if(board[j][i] == type){
                    c++;
                }
            }
            if(c == 3){
                total++;
            }
        }
        
        if(board[0][0] == type && board[1][1] == type && board[2][2] == type){
            total++;
        }
        if(board[0][2] == type && board[1][1] == type && board[2][0] == type){
            total++;
        }
        return total;
    }
    
    bool validTicTacToe(vector<string>& board) {
        
        int x = 0 , o = 0;
        for(int i = 0; i < 3; ++i){
            for(int j = 0; j < 3; ++j){
                if(board[i][j] == 'X'){
                    x++;
                }
                if(board[i][j] == 'O'){
                    o++;
                }
            }
        }
        cout << x <<" "<<o<<endl;
        if( x >= o && x - o <=1 ){
            int xx = getWins(board , 'X');
            int oo = getWins(board , 'O');
            if(xx + oo <= 0){
                return true;
            }
            if(xx == 1 && oo == 0 && x - o == 1){
                return true;
            }
            if(oo == 1 && xx == 0 && x - o == 0){
                return true;
            }
            return false;
        }
        return false;
    }
};
