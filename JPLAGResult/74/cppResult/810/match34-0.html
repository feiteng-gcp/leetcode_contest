class Solution {
public:
    bool validTicTacToe(vector<string>& board) {
        int xs = 0;
        int os = 0;
        
        for (auto s : board) {
            for (auto c : s) {
                if (c == 'X') {
                    ++xs;
                } else if (c == 'O') {
                    ++os;
                } else if (c != ' ') {
                    return false;
                }
            }
        }
        

        if (xs < os) {
            return false;
        }
        
        if (xs - os > 1) {
            return false;
        }
        
        vector<string> rows = board;
        vector<string> cols = board;
        
        for (int i = 0; i < 3; ++i) {
            for (int j = 0; j < 3; ++j) {
                cols[i][j] = rows[j][i];
            }
        }
        
        bool xFilled = false;
        bool oFilled = false;
        for (auto s : rows) {
            if (s == "XXX") {
                xFilled = true;
            } else if (s == "OOO") {
                oFilled = true;
            } 
        }
        
        if (xFilled && oFilled) {
            return false;
        }
        
        if (xFilled) {
            return xs == os + 1;
        } 
        if (oFilled) {
            return xs == os;
        }
        
        xFilled = false;
        oFilled = false;
        for (auto s : cols) {
            if (s == "XXX") {
                xFilled = true;
            } else if (s == "OOO") {
                oFilled = true;
            }
        }
        
        if (xFilled && oFilled) {
            return false;
        }
        
        if (xFilled) {
            return xs == os + 1;
        } 
        if (oFilled) {
            return xs == os;
        }
        
        xFilled = false;
        oFilled = false;
        string s1;
        for (int i = 0; i < 3; ++i) {
            s1 += board[i][i];
        }
        
        string s2;
        for (int i = 0; i < 3; ++i) {
            s2 += board[i][2 - i];
        }
        
        if (s1 == "XXX") {
            xFilled = true;
        } else if (s1 == "OOO") {
            oFilled = true;
        }
        
        if (s2 == "XXX") {
            xFilled = true;
        } else if (s2 == "OOO") {
            oFilled = true;
        }
        
        if (xFilled) {
            return xs == os + 1;
        } else if (oFilled) {
            return xs == os;
        }
        return true; 
    }
};
