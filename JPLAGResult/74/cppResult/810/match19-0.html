class Solution {
public:
    bool validTicTacToe(vector<string>& board) {
        int xcount = 0;
        int ocount = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if(board[i][j] == 'X') xcount++;
                else if(board[i][j] == 'O') ocount++;
            }
        }
        if (xcount == ocount) {
            //x cannot win
            if (checkWin(board, 'X')) return false;
        } else if (xcount == ocount + 1) {
            //o cannot win
            if (checkWin(board, 'O')) return false;
        } else {
            return false;
        }
        return true;
    }
    
    bool checkWin(vector<string>& board, char a) {
        if (board[1][1] == a) {
            if (board[0][0] == a && board[2][2] == a) return true;
            if (board[2][0] == a && board[0][2] == a) return true;
            if (board[0][1] == a && board[2][1] == a) return true;
            if (board[1][0] == a && board[1][2] == a) return true;
        } else {
            if (board[0][0] == a && board[0][1] == a && board[0][2] == a) return true;
            if (board[2][0] == a && board[2][1] == a && board[2][2] == a) return true;
            if (board[0][0] == a && board[1][0] == a && board[2][0] == a) return true;
            if (board[0][2] == a && board[1][2] == a && board[2][2] == a) return true;
        }
        return false;
    }
};
