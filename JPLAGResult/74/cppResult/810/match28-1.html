class Solution {
public:
    bool validTicTacToe(vector<string>& a) {
        int n = a.size();
        vector<int> cnt(2);

        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (a[i][j] == 'X') {
                    ++cnt[0];
                } else if (a[i][j] == 'O') {
                    ++cnt[1];
                }
            }
        }
        if (!(cnt[0] >= cnt[1] && cnt[0] <= cnt[1] + 1)) return false;
        vector<bool> win(2);
        for (int i = 0; i < n; ++i) {
            int j;
            for (j = 0; j < n && a[i][j] == 'X'; ++j);
            if (j == n) win[0] = true;
            for (j = 0; j < n && a[i][j] == 'O'; ++j);
            if (j == n) win[1] = true;
        }
        for (int j = 0; j < n; ++j) {
            int i;
            for (i = 0; i < n && a[i][j] == 'X'; ++i);
            if (i == n) win[0] = true;
            for (i = 0; i < n && a[i][j] == 'O'; ++i);
            if (i == n) win[1] = true;
        }
        int i;
        for (i = 0; i < n && a[i][i] == 'X'; ++i);
        if (i == n) win[0] = true;
        for (i = 0; i < n && a[i][i] == 'O'; ++i);
        if (i == n) win[1] = true;
        for (i = 0; i < n && a[i][n - 1 - i] == 'X'; ++i);
        if (i == n) win[0] = true;
        for (i = 0; i < n && a[i][n - 1 - i] == 'O'; ++i);
        if (i == n) win[1] = true;
        if (win[0] && win[1]) return false;
        if (win[0] || win[1]) {
            if (!(win[0] && cnt[0] == cnt[1] + 1) && !(win[1] && cnt[0] == cnt[1])) return false;
        }
        return true;
    }
};
