class Solution {
    bool check(vector<string>& board, char ch) {
        for (int i = 0; i < 3; ++i) {
            bool ok = true;
            for (int j = 0; j < 3; ++j) {
                if (board[i][j] != ch) {
                    ok = false;
                    break;
                }
            }
            if (ok) return true;
        }

        for (int i = 0; i < 3; ++i) {
            bool ok = true;
            for (int j = 0; j < 3; ++j) {
                if (board[j][i] != ch) {
                    ok = false;
                    break;
                }
            }
            if (ok) return true;
        }

        bool ok = true;
        for (int i = 0; i < 3; ++i) {
            if (board[i][i] != ch) {
                ok = false;
                break;
            }
        }
        if (ok) return true;


        ok = true;
        for (int i = 0; i < 3; ++i) {
            if (board[i][2 - i] != ch) {
                ok = false;
                break;
            }
        }
        return ok;
    }
public:

    bool validTicTacToe(vector<string>& board) {
        int cntX = 0, cntO = 0;
        for (int i = 0; i < 3; ++i) {
            for (int j = 0; j < 3; ++j) {
                if (board[i][j] == 'X') {
                    ++cntX;
                } else if (board[i][j] == 'O') {
                    ++cntO;
                }
            }
        }
        bool x = check(board, 'X');
        bool o = check(board, 'O');
        if (x && o) return false;
        if (x) {
            return cntX == cntO + 1;
        }
        if (o) {
            return cntX == cntO;
        }
        if (cntX == cntO || cntX == cntO + 1) return true;
        return false;
    }
};
