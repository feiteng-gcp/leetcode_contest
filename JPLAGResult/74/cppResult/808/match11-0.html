typedef long long ll;
typedef vector<int> VI;
typedef pair<int,int> PII;

#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))

const int INF = (int)1E9;
#define MAXN 100005

VI pos[26];
class Solution {
public:
  int numMatchingSubseq(string S, vector<string>& words) {
    REP(i,0,26) pos[i].clear();
    REP(i,0,(int)S.size()) pos[S[i] - 'a'].push_back(i);

    int ans = 0;
    REP(k,0,(int)words.size()) {
      string w = words[k];
      int n = w.size();
      int p = 0, fail = 0;
      REP(i, 0, n) {
        int c = w[i] - 'a';
        int k = lower_bound(pos[c].begin(), pos[c].end(), p) - pos[c].begin();
        if (k == pos[c].size()) {
          fail = 1;
          break;
        }
        p = pos[c][k] + 1;
      }
      if (!fail) ans++;
    }
    return ans;
  }
};
