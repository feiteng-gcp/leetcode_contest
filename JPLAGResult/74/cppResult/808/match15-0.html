class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        vector<int> pos[30];
        for(int i = 0; i < S.size(); i++) {
            pos[S[i]-'a'].push_back(i);
        }
        int ans = 0;
        for(string &word : words) {
            int curr = 0;
            bool check = true;
            // cout<<word<<endl;
            for(char c : word) {
                if(pos[c-'a'].size() == 0) {
                    check = false;
                    break;
                }
                auto it = lower_bound(pos[c-'a'].begin(), pos[c-'a'].end(), curr);
                if(it == pos[c-'a'].end()) {
                    check = false;
                    break;
                }
                curr = (*it);
                curr++;
                // cout<<curr<<endl;
            }
            if(check) ans++;
        }
        return ans;
    }
};
