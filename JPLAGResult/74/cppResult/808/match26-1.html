class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        vector<vector<int>> pos(26);
        
        int n = S.length();

        for (int i = 0; i < n; ++i) {
            pos[S[i] - 'a'].push_back(i);
        }
        
        
        int res = 0;
        for (auto &w : words) {
            int p = 0;
            int m = w.length();
            bool bSubseq = true;
            for (int i = 0; i < m; ++i) {
                auto &v = pos[w[i] - 'a'];
                auto it = lower_bound(v.begin(), v.end(), p);
                if (it == v.end()) {
                    bSubseq = false;
                    break;
                }
                p = *it + 1;
            }
            
            if (bSubseq) {
                ++res;
            }
        }
        
        return res;
    }
    
    
};
