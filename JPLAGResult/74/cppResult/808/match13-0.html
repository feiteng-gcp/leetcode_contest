int pt[26], nxt[100000][26];

class Solution {
public:
    bool check(string word, int n) {
        int cur = 0;
        int len = word.length();
        for (int i = 0; i < len; ++i) {
            cur = nxt[cur][word[i] - 'a'];
            if (cur == -1) return false;
        }
        return true;
    }
    int numMatchingSubseq(string S, vector<string>& words) {
        int n = S.length();
        int m = words.size();
        memset(pt, -1, sizeof(pt));
        memset(nxt, -1, sizeof(nxt));
        for (int i = n; i >= 0; --i) {
            for (int j = 0; j < 26; ++j) nxt[i][j] = pt[j];
            if (i == 0) break;
            char ch = S[i - 1];
            pt[ch - 'a'] = i;
        }
        int ans = 0;
        for (int i = 0; i < m; ++i) {
            if (check(words[i], n)) ans ++;
            
        }
        return ans;

    }
};
