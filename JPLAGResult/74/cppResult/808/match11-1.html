class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        vector<vector<int>> V(26, vector<int>());
        for (int i = 0; i < S.size(); i++) V[S[i] - 'a'].push_back(i);
        

        int res = 0, n = S.size();
        for (auto& w : words) {
            int i = -1;
            bool valid = true;
            
            for (auto& ch : w) {
                int idx = ch - 'a';
                auto it = upper_bound(V[idx].begin(), V[idx].end(), i);
                if (it == V[idx].end()) {
                    valid = false;
                    break;
                }
                i = *it;
            }
            
            if (valid) res++;
        }
        return res;
    }
};
