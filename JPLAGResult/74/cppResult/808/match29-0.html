class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        int res = 0;
        vector<vector<int>> loc(26);
        for (int i = 0; i < S.length(); i++)
        {
            loc[S[i]-'a'].push_back(i);
        }
        cout << " 1" << endl;
        for (auto word : words)
        {
            cout << " 1" << endl;
            if (isSubsequence(loc, word))
            {
                res++;
            }
        }
        return res;
    }
    bool isSubsequence(vector<vector<int>>& loc, string word)
    {
        int i = 0;
        for (auto c: word)
        {
            auto it = lower_bound(loc[c-'a'].begin(), loc[c-'a'].end(), i);
            int idx = it - loc[c-'a'].begin();
            if (idx == loc[c-'a'].size())
            {
                return false;
            }
            else
            {
                i = loc[c-'a'][idx] + 1;
            }
        }
        return true;
    }
};
