class Solution {
public:

  int numMatchingSubseq(string S, vector<string>& words) {
    vector<unsigned short> lookup[26];
    for (int i=0; i<26; ++i) lookup[i].resize(S.length()+1);    
    for(int i=0; i<26;++i) {
      int cur = S.length();
      char c = 'a'+i;
      for(int j=S.length()-1; j>=0; --j) {
        lookup[i][j+1] = cur+1;
        if (S[j] == c) {
          cur = j;
        }
      }
      lookup[i][0] = cur+1;
    }
    int ans = 0;
    for (auto &w: words) {
      int i = 0;
      bool found = true;
      for (char c: w) {
        if (lookup[c-'a'][i] > S.length()) {
          found = false;
          break;
        }
        i = lookup[c-'a'][i];
      }
      if(found) ans++;
    }
    return ans;
  }
};
