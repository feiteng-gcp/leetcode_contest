class Solution {
public:
  int numMatchingSubseq(string S, vector<string>& words) {
    int n = S.size();
    vector<set<int>> pos(26);
    for (int i = 0; i != n; ++i)
      pos[S[i] - 'a'].insert(i);
    int ret = 0;
    for (const auto &word: words) {
      int p = 0;
      bool found = true;
      for (int i = 0; i != word.size(); ++i) {
        int c = word[i] - 'a';

        auto ite = pos[c].lower_bound(p);
        if (ite == pos[c].end()) {
          found = false;
          break;
        } 
        p = *ite + 1;
      }
      if (found) ++ret;
    }
    return ret;
  }
};
