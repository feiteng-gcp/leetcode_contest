class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        vector<vector<int>> letterPos(26, vector<int>());
        for (int i=0; i<S.size(); ++i) {
            letterPos[S[i]-'a'].push_back(i);
        }
        int cnt = 0;
        for (int i=0; i<words.size(); ++i) {
            if (isSub(letterPos, words[i])) ++cnt;
        }
        return cnt;
    }
    
    bool isSub(vector<vector<int>> &letterPos, string word ) {
        int target = 0;
        for (int i=0; i<word.size(); ++i) {
            vector<int> curr = letterPos[word[i]-'a'];
            auto low = lower_bound(curr.begin(), curr.end(), target);
            if (low ==curr.end()) return false;
            target = curr[low-curr.begin()]+1;
        }
        return true;
    }
};
