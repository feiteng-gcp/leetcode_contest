class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        vector<int> index[26];
        int n = S.size();
        for (int i = 0; i < n; i++) {
            index[S[i] - 'a'].push_back(i);
        }
        int cnt = 0;
        for (string word: words) {
            int p = 0;
            for (char c: word) {
                auto it = lower_bound(index[c - 'a'].begin(), index[c - 'a'].end(), p);
                if (it == index[c - 'a'].end()) {
                    p = -1; break;
                }
                p = *it + 1;
            }
            if (p != -1) cnt++;
        }
        return cnt;
    }
};
