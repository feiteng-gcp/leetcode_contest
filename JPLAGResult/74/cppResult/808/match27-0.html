class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        int len = S.length(), n = words.size(), ans = 0;
        vector<int> pos(n, 0);
        vector< queue<int> > next(26, queue<int>());
        for (int i = 0; i < n; i++) {
            next[words[i][0] - 'a'].push(i);
        }
        for (int i = 0; i < len; i++) {
            int cur = S[i] - 'a', size = next[cur].size();
            for (int iter = 0; iter < size; iter++) {
                int cur_pos = next[cur].front();
                next[cur].pop();
                if (pos[cur_pos] + 1 >= words[cur_pos].size()) { 
                    ans++; 
                    continue;
                }
                pos[cur_pos]++;
                int temp = words[cur_pos][pos[cur_pos]] - 'a';
                next[temp].push(cur_pos);
            }
        }
        return ans;
    }
};
