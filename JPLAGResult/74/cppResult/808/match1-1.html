class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words)
    {
        vector<vector<int>> s(26, vector<int>());
        for (int i = 0; i < S.size(); ++i)
        {
            s[S[i] - 'a'].push_back(i);
        }
        
        int cnt = 0;
        for (auto &str: words)
        {
            int pos = -1;
            vector<int> idx(26, 0);
            for (int i = 0; i < str.size(); ++i)
            {
                int p = str[i] - 'a';
                while (idx[p] < s[p].size() && s[p][idx[p]] <= pos) 
                    ++idx[p];
                if (idx[p] == s[p].size()) break;
                pos = s[p][idx[p]];
                if (i == str.size() - 1) ++cnt;
            }
        }
        
        
        return cnt;
    }
};
