int dp[50010][26];

class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        int len = (int)S.length();
        memset(dp, -1, sizeof(dp));
        for (int i = len - 1; i >= 0; --i) {
            for (int j = 0; j < 26; ++j) {
                dp[i][j] = dp[i + 1][j];
            }
            dp[i][S[i] - 'a'] = i;
        }

        int res = 0;
        for (string s : words) {
            int cur = 0, pos = 0;
            bool ok = true;
            while (cur < (int)s.length()) {

                if (dp[pos][s[cur] - 'a'] == -1) {
                    ok = false;
                    break;
                }
                pos = dp[pos][s[cur] - 'a'] + 1;
                ++cur;
            }
            if (ok) ++res;
        }
        return res;
    }
};
