class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        int len = S.length();
        vector<vector<int>> chIdxes(26); 
        for (int i=0; i < len; ++i) {
            int chIdx = S[i] - 'a';
            chIdxes[chIdx].push_back(i);
        }
        int ans = 0;
        int sz = words.size();
        for (int i=0; i < sz; ++i) {
            int l = words[i].length();
            int pos = -1;
            for (int j=0; j < l; ++j) {
                ++pos;
                if (pos >= len) {
                    pos = -1;
                    break;
                }
                int chIdx = words[i][j] - 'a';

                vector<int>::iterator itr = lower_bound(chIdxes[chIdx].begin(), chIdxes[chIdx].end(), pos);
                if (itr == chIdxes[chIdx].end()) {
                    pos = -1;
                    break;
                }
                pos = (*itr);
            }
            if (pos >= 0)
                ++ans;
        }
        return ans;
    }
};
