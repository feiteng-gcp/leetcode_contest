class Solution {
public:
	static const int maxn = 50003;
	std::vector<int> p[26];
    int numMatchingSubseq(string S, vector<string>& words) {
		int les = S.length();
		for (int i = 0; i < les; ++ i) {
			p[S[i] - 97].push_back(i);
		}
		int ans(0);
		int wl = words.size();
		for (int i = 0; i < wl; ++ i) {
			int k = -1, li = words[i].length();
			for (int j = 0; j < li; ++ j) {
				int c = words[i][j] - 97;
				std::vector<int>::iterator it = upper_bound(p[c].begin(), p[c].end(), k);
				if (it == p[c].end()) {
					k = -1;
					break;
				} else {
					k = *it;
				}
			}
			if (k > -1) {
				++ ans;
			}
		}
		return ans;
    }
};
