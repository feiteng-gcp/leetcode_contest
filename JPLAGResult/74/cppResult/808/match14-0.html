int dp[26][50000 + 10];
class Solution {
    typedef vector<int> vi;
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        int n = S.size();
        memset(dp, -1, sizeof(dp));
        for (int i = n-1; i >= 0; i--) {
            for (int j = 0; j < 26; j++) {
                dp[j][i] = dp[j][i+1];
            }
            dp[S[i]-'a'][i] = i;
        }
        int cnt = 0;
        for (string & word : words) {
            int idx = 0, tmp = 0;
            for (char c : word) {
                idx = dp[c-'a'][idx];
                // cout << idx << endl;
                if (idx == -1) break;
                tmp++;
                idx++;
                if (idx == n) break;
                
            }
            if (tmp == word.size()) cnt++;
        }
        return cnt;
    }
};
