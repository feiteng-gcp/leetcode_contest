class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        int count = 0;
        unordered_set<string> dict(words.begin(), words.end());
        unordered_set<string> occurred;
        for(auto w: dict){
            if(checkMatch(S, w)) occurred.insert(w);
        }
        for(int i = 0; i < words.size(); i++){
            if(occurred.count(words[i])) count++;
        }
        return count;
    }
    bool checkMatch(string& S, string& T){
        int j = 0;
        int count = 0;
        for(int i = 0; i < T.size(); i++){
            while(j < S.size() && S[j] != T[i]) {
                j++;
            }
            if(j == S.size()) break;
            count++;
            j++;
        }
        return T.size() == count;
    }
};
