class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        vector<int> arr[26];
        int last_index,flag,res=0;
        for(int i=0;i<S.size();i++)
        {
            arr[S[i]-'a'].push_back(i);
        }
        
        for(int i=0;i<words.size();i++)
        {
            last_index=-1;
            flag=1;
            for(int j=0;j<words[i].size();j++)
            {
                last_index=biSearchLT(arr[words[i][j]-'a'],last_index);
                if(last_index==-1){
                    flag=0;
                    break;
                }
                
            }
            
            if(flag)
                res++;
        }
        
        return res;
        
    }
    
    int biSearchLT(vector<int> arr,int val)
    {
        int len = arr.size();
        if(len==0)
            return -1;
        int l=0,r=len-1;
        if(val>=arr[r])return -1;
        int mid =0;
        while(r>l)
        {
            mid=(r+l)/2;
            if(val>=arr[mid])
                l=mid+1;
            else
                r=mid;
        }
        return arr[l];
    }
};
