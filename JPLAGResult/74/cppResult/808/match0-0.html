class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        vector<vector<int>> table(26, vector<int>());
        for (int i = 0; i < S.size(); ++i) {
            table[S[i] - 'a'].push_back(i);
        }
        
        int count = 0;
        for (int i = 0; i < words.size(); ++i) {
            string word = words[i];
            int pos = -1;
            bool flag = true;
            for (int j = 0; j < word.size(); ++j) {
                auto it = upper_bound(table[word[j] - 'a'].begin(), table[word[j] - 'a'].end(), pos);
                if (it == table[word[j] - 'a'].end()) {
                    flag = false;
                    break;
                } else 
                    pos = *it;
            }
            if (flag) count++;
        }
        return count;
    }
};
