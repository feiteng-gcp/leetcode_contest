 int a[50010][26];
class Solution {
 public:
	 

	 int numMatchingSubseq(string S, vector<string>& words) {

		 
		 for (int i = 0; i <= S.size(); i++) {
			 if (i == 0) {
				 for (int j = 0; j < 26; j++) a[i][j] = -1;
			 }
			 else {
				 for (int j = 0; j<26; j++) a[i][j] = a[i - 1][j];
				 a[i][S[i-1] - 'a'] = i-1;
			 }
			 

			 // for (int j = 0; j < 26; j++) cout << a[i][j] << " ";
			 // cout << endl;
		 }


		 


		 int ret = 0;
		 for (int i = 0; i < words.size(); i++) {
			 int cur = S.size();
			 bool flag = true;
			 for (int j = words[i].size() - 1; j >= 0; j--) {
				 char c = words[i][j];
				 if (a[cur][c - 'a'] == -1) {
					 flag = false;
					 break;
				 }
				 cur = a[cur][c - 'a'];

			 }
			 if (flag) ret++;
		 }
		 return ret;

	 }
 };
