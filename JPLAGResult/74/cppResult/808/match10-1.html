class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        unordered_map<string,int> m;
        
        int total = 0;
        for(int i=0;i<words.size();i++)
        {
            if(m[words[i]]){total++; continue;}
            int j = 0;
            int k = 0;
            while(j < words[i].length())
            {
                
                while(words[i][j] != S[k] && k < S.length())
                {
                    k++;
                }
                if(k == S.length()){break;}
                j++;
                k++;
                if(j == words[i].length()){total++; m[words[i]] = 1;}
            }
        }
        return total;
    }
};
