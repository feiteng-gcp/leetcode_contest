    int nxt[50010][26];
class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        int n = words.size();
        int len = S.size(), res = 0;
        int lpos[26], fpos[26];
        for (int i = 0; i < 26; i++)
            lpos[i] = len;
        for (int i = len - 1; i >= 0; i--) {
            for (int j = 0; j < 26; j++) {
                nxt[i][j] = lpos[j];
            }
            lpos[S[i]-'a'] = i;
        }
        for (int j = 0; j < 26; j++)
            fpos[j] = lpos[j];
        for (int i = 0; i < n; i++) {
            int cur = 0;
            for (int j = fpos[words[i][0]-'a']; j < len && cur < words[i].size(); j = nxt[j][words[i][cur]-'a']) {
                cur++;
                if (cur >= words[i].size())
                    break;
            }
            if (cur >= words[i].size())
                res++;
        }
        return res;
    }
};
