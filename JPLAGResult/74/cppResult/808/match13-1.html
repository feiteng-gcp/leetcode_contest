#define pb push_back
#define fst first
#define snd second

typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T> >;

const ll MOD=1e9+7;

class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& wd) {
      int N=wd.size();
      vector<vector<int>> dp(26);
      vector<int> ix(N);
      for(int i=0;i<N;i++){
	dp[wd[i][0]-'a'].pb(i);
      }
      for(char c:S){
	vector<vector<int>> tmp(26);
	for(int i:dp[c-'a']){
	  ix[i]++;
	  if(ix[i]<wd[i].size()){
	    tmp[wd[i][ix[i]]-'a'].pb(i);
	  }
	}
	dp[c-'a'].clear();
	for(int i=0;i<26;i++){
	  for(int n:tmp[i]) dp[i].pb(n);
	}

      }
      int res=0;
      for(int i=0;i<wd.size();i++){
	if(ix[i]==wd[i].size()) res++;
      }
      return res;
    }
};
