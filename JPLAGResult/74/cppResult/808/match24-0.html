class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        int n = S.length();
        vector<array<int, 26>> nxt(n + 1);
        for(int i = 0; i < 26; ++i) nxt[n][i] = n + 1;
        for(int i = n - 1; i >= 0; --i)
        {
            for(int j = 0; j < 26; ++j) nxt[i][j] = nxt[i + 1][j];
            nxt[i][S[i] - 'a'] = i;
        }
        int ans = 0;
        for(string word: words)
        {
            int i = -1;
            for(char c: word)
            {
                i = nxt[i + 1][c - 'a'];
                if(i > n) goto fail;
            }
            ++ans;
        fail:;
        }
        return ans;
    }
};
