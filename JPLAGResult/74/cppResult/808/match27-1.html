class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {
        int count = 0;

        for (int i = 0; i < words.size(); i++)
        { 
          int start_pos = 0;
          bool is_seq = true;
          for(int j = 0; j < words[i].size(); j++)
          {
              if((start_pos = S.find(words[i][j], start_pos)) != string::npos)
              {
                  start_pos += 1;
              }
              else
              {
                  is_seq = false;
                  break;
              }
          }
            
          if (is_seq) count++;
        }
       
        return count;
    }
};
