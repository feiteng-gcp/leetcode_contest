class Solution {
public:
    int numMatchingSubseq(string s, vector<string>& words) {
        int slen = s.size();
        int wlen = words.size();
        if (wlen == 0) {
            return 0;
        }
        int ret = 0;
        if (slen == 0) {
            for (auto w : words) {
                if (w == s) {
                    ret++;
                }
            }
            return ret;
        }
        
        vector<set<int>> bst(26);
        for (int i = 0; i < slen; i++) {
            bst[s[i] - 'a'].insert(i);
        }
        
        for (auto w : words) {
            int ws = w.size();
            int icur = -1;
            bool flag = true;
            for (int i  = 0; i < ws; i++) {

                auto next = bst[w[i] - 'a'].upper_bound(icur);
                if (next == bst[w[i] - 'a'].end()) {
                    flag = false;
                    break;
                } else {
                    icur = *next;
                }
            }
            if (flag == true) {
                ret++;
            }
        }
        
        return ret;

    }
};
