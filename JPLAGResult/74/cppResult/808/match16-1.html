class Solution {
public:
    int numMatchingSubseq(string S, vector<string>& words) {

        unordered_map<char, set<int>> rank;
        for (int i = 0; i < S.size(); i++) {
            rank[S[i]].insert(i);
        }
        int result = 0;
        for (int i = 0; i < words.size(); i++) {
            if (helper(words[i], rank)) result++;
        }
        return result;
    }
    
    bool helper(string& word, unordered_map<char, set<int>>& rank) {
        int curIndex = -1;
        for (int i = 0; i < word.size(); i++) {
            if (rank.find(word[i]) == rank.end()) return false;
            auto it = rank[word[i]].upper_bound(curIndex);
            if (it == rank[word[i]].end()) return false;
            curIndex = *it;
        }
        return true;
    }
};
