class Solution {
private:
	vector<int> ve[26]; int ans = 0;
public:
	int numMatchingSubseq(string S, vector<string>& words) {
        for(int i = 0; i < S.size(); ++i) {
        	ve[S[i] - 'a'].push_back(i);
        }
        for(auto i : words) {
        	int p = -1; bool flag = true;
        	for(char j : i) {
        		auto q = upper_bound(ve[j - 'a'].begin(), ve[j - 'a'].end(), p);
        		if(q == ve[j - 'a'].end()) { flag = false; break; }
        		p = ve[j - 'a'][q - ve[j - 'a'].begin()];
        	}
        	if(flag)++ans;
        }
        return ans;
    }
};
