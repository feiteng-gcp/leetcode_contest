class Solution {
public:
    
    vector<int> positions[26];
    
    bool valid(string s){
        int sz = s.size();
        int last = -1;
        for(char c : s){
            int idx = c - 'a';
            int pos = upper_bound(positions[idx].begin() , positions[idx].end() , last) - positions[idx].begin();
            if(pos == positions[idx].size()){
                return false;
            }
            last = positions[idx][pos];
        }
        return true;
    }
    
    int numMatchingSubseq(string S, vector<string>& words) {
        
        for(int i = 0; i < 26; ++i){
            positions[i].clear();
        }

        for(int i = 0; i < S.size(); ++i){
            positions[S[i] - 'a'].push_back(i);
        }
       
        int ans = 0;
        for(string e : words){
            if(valid(e)){
                ans++;
            }
        }
        return ans;
    }
};
