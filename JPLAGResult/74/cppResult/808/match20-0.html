class Solution {
public:
    bool match(vector<vector<int>> &dict, string &s, string &word) {
        vector<int> idx(26, -1);
        int curIdx = 0;
        for (char c : word) {
            int wIdx = idx[c - 'a'] + 1;
            while (wIdx < dict[c - 'a'].size() && dict[c - 'a'][wIdx] < curIdx) wIdx++;
            if (wIdx >= dict[c - 'a'].size()) return false;
            curIdx = dict[c - 'a'][wIdx] + 1;
            idx[c - 'a'] = wIdx;
        }
        return true;
    }
    int numMatchingSubseq(string S, vector<string>& words) {
        vector<vector<int>> dict = vector<vector<int>>(26);
        for (int i = 0; i < S.size(); i++) {
            dict[S[i] - 'a'].push_back(i);
        }
        int count = 0;
        for (string &word : words) {
            if (match(dict, S, word)) {
                count++;
            }
        }
        return count;
    }
};
