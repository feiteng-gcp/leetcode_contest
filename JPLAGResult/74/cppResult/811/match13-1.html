class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int cnt = 0, ans = 0, cnt2 = 0;
        for(int x : A) {
            if(x >= L && x <= R) cnt++, cnt += cnt2, cnt2 = 0;
            if(x > R) cnt = 0, cnt2 = 0;
            ans += cnt;
            if(x < L) cnt2++;
        }
        return ans;
    }
};
