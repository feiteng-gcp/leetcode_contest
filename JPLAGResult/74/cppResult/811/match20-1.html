class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        long long ans=0;

	int n=A.size();
	vector<int> v;
	v.push_back(-1);
	int tot=1,cur=1;
	for(int i=0;i<n;i++)
	{
		if(A[i]>=L&&A[i]<=R)
		{
			v.push_back(i);
			tot++;
		}
		else if(A[i]>R)
		{
			v.push_back(i);
			while(cur<tot)
			{
				ans=ans+(long long)(v[cur]-v[cur-1])*(i-v[cur]);
				cur++;
			}
			tot++;
			cur++;
		}
	}
	while(cur<tot)
	{
		ans=ans+(long long)(v[cur]-v[cur-1])*(n-v[cur]);
		cur++;
	}
	return ans;
    }
};
