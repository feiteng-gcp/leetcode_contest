class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int lastPos = -1;
        int n = A.size();
        
        int res = 0;
        vector<int> pos;
        for (int i = 0; i < n; ++i) {
            if (A[i] >= L && A[i] <= R) {
                pos.push_back(i);
            } else if (A[i] > R) {
                for (auto j : pos) {
                    res += (j - lastPos) * (i - j);
                    lastPos = j;
                }
                lastPos = i;
                pos.clear();
            }
        }
        
        if (!pos.empty()) {
            for (auto j : pos) {
                res += (j - lastPos) * (n - j);
                lastPos = j;
            }
        }
        
        return res;
    }
};
