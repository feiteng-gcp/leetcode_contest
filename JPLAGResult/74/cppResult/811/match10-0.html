class Solution {
public:

    
    int helper(vector<int>&A, int lo, int hi, int L, int R){
        int ans = (hi - lo + 1) * (hi - lo + 2) / 2;
        int i = lo;
        while(i <= hi){
            if(A[i] < L){
                int j = i;
                while(A[j] < L && j <= hi){
                    j++;
                }
                ans -= (j - i) * (j-i+1) / 2;
                i = j;
            } else i++;
        }
        return ans;
    }
    
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int lo = 0;
        int ans = 0;
        for(int hi = 0; hi <=A.size(); hi++){
            if(hi == A.size() || A[hi] > R) {
                ans += helper(A, lo, hi-1, L, R);
                lo = hi + 1;
            }
        }
        return ans;
    }
};
