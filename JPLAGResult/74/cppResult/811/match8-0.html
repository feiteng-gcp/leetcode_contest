class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int n = A.size();
        bool large[n+2];
        bool small[n+2];
        for(int i=0;i<n;i++)
        {
            large[i+1]=A[i]>R;
            small[i+1]=A[i]<L;
        }
        large[0]=true;
        small[0]=false;
        large[n+1]=true;
        small[n+1]=false;
        int tot=0;
        int count1=0;
        int count2=0;
        for(int i=0;i<n+2;i++)
        {
            if(!large[i])
            {
                count1++;
            }
            else if (count1>0)
            {
                tot+=count1*(count1+1)/2;
                count1=0;
            }
            if(small[i])
            {
                count2++;
            }
            else if(count2>0)
            {
                tot-=count2*(count2+1)/2;
                count2=0;
            }
        }
        return tot;
    }
};
