class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int len = A.size();
        int ans = 0;
        vector<int> left(len, 0);
        vector<int> right(len, 0);
        int tmp = 0;
        for (int i = 0; i < len; ++i) {
            if (A[i] < L) {
                tmp++;
                continue;
            }
            if (A[i] > R) {
                tmp = 0;
                continue;
            }
            tmp++;
            left[i] = tmp;
        }
        tmp = 0;
        for (int i = len - 1; i >= 0; --i) {
            if (A[i] < L) {
                tmp++;
                continue;
            }
            if (A[i] > R) {
                tmp = 0;
                continue;
            }
            tmp++;
            right[i] = tmp;
            tmp = 0;
        }
        for (int i = 0; i < len; ++i) {
            if (left[i] > 0 && right[i] > 0) {
                ans += (left[i] * right[i]);
                continue;
            }
            if (left[i] > 0 && right[i] == 0) {
                ans += left[i];
                continue;
            }
            if (right[i] > 0 && left[i] == 0) {
                ans += right[i];
                continue;
            }
        }
        
        return ans;
    }
};
