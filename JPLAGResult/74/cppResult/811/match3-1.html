class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R)
    {

        int len = A.size();
        int cnt = 0;
        int curMaxPos = -1;
        int curMaxValid = -1;
        for (int i = 0; i < len; ++i)
        {
            if (A[i] > R)
                curMaxPos = i;
            else
            {
                if (A[i] < L)
                {
                    if (curMaxValid > curMaxPos)
                    {
                        cnt += curMaxValid - curMaxPos;
                    }
                }
                else
                {
                    curMaxValid = i;
                    cnt += curMaxValid - curMaxPos;
                }
            }
        }
        return cnt;
    }
};
