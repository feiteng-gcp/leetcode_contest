class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int res = 0, i = 0, n = A.size();
        
        while (i < n) {
            int j = i;
            while (j < n && A[j] <= R) j++;
            res += getCount(j - i);
            i = j + 1;
        }
        
        i = 0;
        while (i < n) {
            if (A[i] >= L) {
                i++;
            }
            else {
                int j = i;
                while (j < n && A[j] < L) j++;
                res -= getCount(j - i);
                i = j + 1;
            }
        }
        
        return res;
    }
    
private:    
    int getCount(int n) {
        if (n == 0) return 0;
        return (1 + n) * n / 2;
    }
};
