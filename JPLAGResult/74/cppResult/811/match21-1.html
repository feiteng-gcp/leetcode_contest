class Solution {
    int cnt(int x) {
        return (x + 1) * x / 2;
    }
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int len = (int)A.size();


        int except = 0;
        int l = 0;
        for (int i = 0; i < len; ++i) {
            if (A[i] >= L) {
                except += cnt(i - l);
                l = i + 1;
            }
        }
        except += cnt(len - l);

        int tot = 0;
        l = 0;
        for (int i = 0; i < len; ++i) {
            if (A[i] > R) {
                tot += cnt(i - l);
                l = i + 1;
            }
        }
        tot += cnt(len - l);

        return tot - except;
    }
};
