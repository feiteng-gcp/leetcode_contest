class Solution {
public:
  int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
    vector<int> larger_pos;
    for (int i = 0; i != A.size(); ++i)
      if (A[i] > R) larger_pos.push_back(i);
    int ret = 0;
    int start = 0;
    for (const auto &p: larger_pos) {
      ret += subsum(A, start, p, L);
      start = p + 1;
    }
    ret += subsum(A, start, A.size(), L);
    return ret;
  }
  
  int subsum(vector<int> &A, int start, int end, int L) {
    int n = end - start;
    int total = n * (n + 1) / 2;
    int invalid = 0;
    int p = start, q = p;
    while (1) {
      while (p < end && A[p] < L) ++p;
      int l = p - q;
      invalid += l * (l + 1) / 2;
      if (p == end) return total - invalid;
      ++p;
      q = p;
    }
  }
};
