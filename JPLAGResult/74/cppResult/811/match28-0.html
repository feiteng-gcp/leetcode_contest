class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
		int ans(0), l(A.size());
		for (int i = 0, j; i < l; i = j + 1) {
			for (j = i; j < l && A[j] <= R; ++ j);
			int c(j - i);
			int d((long long)c * (c + 1) / 2);
			for (int k = i, m; k < j; k = m + 1) {
				for (m = k; m < j && A[m] < L; ++ m);
				int e = m - k;
				d -= ((long long)e * (e + 1) / 2);
			}
			ans += d;
		}
		return ans;
    }
};
