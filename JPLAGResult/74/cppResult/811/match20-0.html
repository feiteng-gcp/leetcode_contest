class Solution {
public:
    int ans;
    void helper(vector<int>& A, int beg, int end, vector<int>& pos) {
        int sz = pos.size();
        //ans ++;
        for (int i=0; i < sz; ++i) {
            int l = 1;
            if (pos[i] > beg) 
                l = pos[i] - beg + 1;
            int r = 1;
            int nxt = end;
            if (i < (sz-1))
                nxt = pos[i+1] - 1;
            if (nxt > pos[i])
                r = nxt - pos[i] + 1;
            ans += l * r;
            cout << "beg=" << beg << " end=" << end << " l=" << l << " r=" << r << " ans=" << ans << endl;
        }
    }
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int sz = A.size();
        vector<int> pos;
        int beg = 0;
        ans = 0;
        for (int i=0; i < sz; ++i) {
            if (A[i] > R) {
                if (pos.size() > 0) {
                    helper(A, beg, i-1, pos);
                    pos.clear();
                    //beg = i+1;
                }
                beg = i+1;
            } else if (A[i] >= L)
                pos.push_back(i);
        }
        if (pos.size() > 0)
            helper(A, beg, sz-1, pos);
        return ans;
    }
};
