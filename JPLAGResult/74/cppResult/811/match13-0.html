class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int i = 0, j = 0, result = 0, n = A.size();
        while (i < n && j < n) {
            while (i < n && A[i] > R) i++;
            j = i;
            while (j < n && A[j] <= R) j++;
            result = result + getCount(A, i, j, L);
            i = j;
        }
        return result;
    }
    
    int getCount(vector<int>& A, int i, int j, int L) {
        if (j == i) return 0;
        int n = j - i;
        int count = (n + 1) * n / 2;
        int ii = i, jj = i;
        while (ii < j && jj < j) {
            while (ii < j && A[ii] >= L) ii++;
            jj = ii;
            while (jj < j && A[jj] < L) jj++;
            count = count - (jj - ii) * (jj - ii + 1) / 2;
            ii = jj;
        }
        return count;
    }
};
