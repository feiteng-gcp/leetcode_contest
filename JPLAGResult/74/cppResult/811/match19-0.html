class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int count = 0;
        for (int i = 0; i < A.size(); ++i) {
            int curMax = INT_MIN;
            for (int j = i; j < A.size(); ++j) {
                if (A[j] > curMax) curMax = A[j];
                if (curMax >= L && curMax <= R) ++count;
                else if (curMax > R) break;
            }
        }
        return count;
    }
};
