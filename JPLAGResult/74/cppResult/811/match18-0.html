class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& A, int L, int R) {
        int n = A.size();
        int prev = 0, ans = 0, r = n, l = n;
        for(int i=n-1;i>=0;i--){
            if(A[i] > R){
                r = l = i;
            }else if(A[i] >= L){
                l = i;
                ans += (r - l);
            }else{
                ans += (r - l);
            }
        }
        
        ans += prev;
        return ans;
    }
};
