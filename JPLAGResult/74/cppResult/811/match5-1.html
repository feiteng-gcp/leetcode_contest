class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& a, int L, int R) {
      int N=a.size();
      int l=0,r=0;
      ll res=0;
      while(l<N){
	while(l<N && a[l]>R) l++;
	r=l;
	while(r<N && a[r]<=R) r++;
	ll tmp=1ll*(r-l)*(r-l+1)/2;
        // cout<<l<<' '<<r<<'\n';

          int ll=l,rr=l;
	while(ll<r){
	  while(ll<r && a[ll]>=L) ll++;
	  rr=ll;
	  while(rr<r && a[rr]<L) rr++;
	  tmp-=1ll*(rr-ll)*(rr-ll+1)/2;
        // cout<<tmp<<
	  ll=rr;
        // cout<<'\t'<<ll<<' '<<rr<<'\n';
	}
	res+=tmp;
	l=r;
      }
      return res;
    }
};
