class Solution {
public:
    int numSubarrayBoundedMax(vector<int>& a, int low, int high) {
        int n = a.size();
        vector<int> L(n), R(n);
        L[0] = 0;
        vector<int> A = {0};
        for (int i = 1; i < n; ++i) {
            while (!A.empty() && a[i] >= a[A.back()]) A.pop_back();
            L[i] = A.empty() ? 0 : A.back() + 1;
            A.push_back(i);
        }
        R[n - 1] = n - 1;
        A = {n - 1};
        for (int i = n - 2; i >= 0; --i) {
            while (!A.empty() && a[i] > a[A.back()]) A.pop_back();
            R[i] = A.empty() ? n - 1 : A.back() - 1;
            A.push_back(i);
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            if (a[i] >= low && a[i] <= high) {
                ret += (R[i] - i + 1) * (i - L[i] + 1);
            }
        }
        return ret;
        
    }
};
