class Solution {
    public String pushDominoes(String dominoes) {
        int n = dominoes.length();
        int[] lastl1 = new int[n];
        int[] lastr1 = new int[n];
        int[] lastl2 = new int[n];
        int[] lastr2 = new int[n];
        
int lr = -1, ll = -1;
        for(int i = 0; i<n; i++) {
            if(dominoes.charAt(i) == 'R'){
                lr = i;
            }
            lastr1[i] = lr;
            if(dominoes.charAt(i) == 'L'){
                ll = i;
            }
            lastl2[i] = ll;
        }
        ll = -1; lr = -1;
        for(int i = n-1; i>=0; i--) {
            if(dominoes.charAt(i) == 'L'){
                ll = i;
            }
            lastl1[i] = ll;
            if(dominoes.charAt(i) == 'R'){
                lr = i;
            
}
            lastr2[i] = lr;
        }
        StringBuilder sb = new StringBuilder();
        // for(int i : lastl2) System.out.println(i);
        // System.out.println(Arrays.asList(lastl2).toString());
        for(int i = 0; i<n; i++){
            ll = lastl1[i];
            lr = lastr1[i];
            if(lastr2[i]!=-1 && lastl1[i]!=-1 && lastr2[i]<lastl1[i] && lastl1[i]!=i) ll = -1;
            if(lastl2[i]!=-1 && lastr1[i]!=-1 && lastl2[i]>lastr1[i] && lastr1[i]!=i) lr = -1;

            if(ll == -1 && lr == -1) sb.append('.');
            else if(ll != -1 && lr == -1) sb.append('L');
            else if(ll == -1 && lr != -1) sb.append('R');
            else if(ll - i == i - lr) sb.append('.');
            else if(i - lr < ll - i) sb.append('R');
            else sb.append('L')
;
        }
        return sb.toString();
    }
}
