class Solution {
    public String pushDominoes(String s) {
        if (s == null || s.length() == 0) {
            return s;
        }
        StringBuilder sb = new StringBuilder();
        char[] tmp = new char[s.length()];
        for (int i = 0; i < s.length(); i++) {
            char cur = s.charAt(i);
            if (cur == 'L' || cur == 'R') {
                sb.append(cur);
                continue;
            }
            int j = i;
            while (i < s.length() && s.charAt(i) == '.') {
                i++;
            }
            char left = (j == 0) ? ' ' : s.charAt(j - 1);
            char right = (i == s.length() ? ' ' : s.charAt(i));
            if (left == 'R' && right == 'L') {
                for (int k = 0; j + k <= i - 1 - k; k++) {
                    tmp[j + k] = 'R';
                    tmp[i - 1 - k] = 'L';
                    if (j + k == i - 1 - k) {
                        tmp[j + k] = '.';
                    }
                }
                for (int k = j; k < i; k++) {
                    sb.append(tmp[k]);
                }
            
} else if (left == 'R') {
                for (int k = j; k < i; k++) {
                    sb.append('R');
                }
            } else if (right == 'L') {
                for (int k = j; k < i; k++) {
                    sb.append('L');
                }
            } else {
                for (int k = j; k < i; k++) {
                    sb.append('.');
                }
            }
            i--;
        }
        return sb.toString();
    }
}
