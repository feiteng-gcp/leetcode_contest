class Solution {
    public String pushDominoes(String dominoes) {
        char[] ca = dominoes.toCharArray();
        int[] mem = new int[ca.length];
        char[] ret = new char[ca.length];
        for (int i=0; i<ca.length; i++)
            ret[i] = ca[i];
        
        Arrays.fill(mem, -100001);
        
        int last = -100001;
        for (int i=0; i<ca.length; i++) 
{
            if (ca[i]=='R') {
                last = i;
            } else if (ca[i]=='L') {
                last = -100001;
            } else {
                if (last>=0) 
                    mem[i] = last;
            }
        }
        
        last = -100001;
        
for (int i=ca.length-1; i>=0; i--) {
            if (ca[i]=='L') {
                last = i;
            } else if (ca[i]=='R') {
                last = -100001;
            } else if (last>=0) {
                if (last - i < i - mem[i]) {
                    mem[i] = last;
                } else if (last - i == i - mem[i]) {
                    
mem[i] = -100001;
                }
            }
        }
        
        for (int i=0; i<ca.length; i++) {
            if (mem[i]>=0)
                ret[i] = ca[mem[i]];
        }
        
        return new String(ret);
    }
}
