class Solution {
    public String pushDominoes(String dominoes) {
        
char[] array = dominoes.toCharArray();
        int slow = -1;
        for (int i = 0; i < array.length; i++) {
            if (array[i] != '.') {
                //process
                process(slow, i, array);
                slow = i;
            }
        }
        //post processing
        if (slow != -1 && array[slow] == 'R') {
            for (int i = slow + 1; i < array.length; i++) {
                array[i] = 'R';
            }
        }
        return new String(array);
    
}

    //process
    private void process(int slow, int fast, char[] array) {
        if (slow == -1) {
            if (array[fast] == 'L') {
                //do something
                for (int i = slow + 1; i < fast; i++) {
                    array[i] = 'L';
                }
            }
        
} else if (array[slow] == array[fast]) {
            for (int i = slow + 1; i < fast; i++) {
                array[i] = array[slow];
            }
        } else if (array[slow] == 'R' && array[fast] == 'L') {
            int i = slow + 1;
            int j = fast - 1;
            while (i < j) {
                array[i] = array[slow];
                array[j] = array[fast];
                i++;
                j--;
            }
        }
    }
}
