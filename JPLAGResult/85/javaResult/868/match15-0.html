class Solution {
    public String pushDominoes(String dominoes) {
        int n = dominoes.length();
        char[] res = dominoes.toCharArray();
        
        int start = 0, p = 0, R = -1;
        while (p < n) {
            char c = dominoes.charAt(p);
            if (c == '.') p++;
            else if (c == 'L') {
                if (R < start) {
                    for (int i = start; i < p; i++) {
                        res[i] = 'L';
                    }
                    start = p + 1;
                    p = start;
                }
                else {
                    int lo = R + 1, hi = p - 1;
                    while (lo < hi) {
                        res[lo++] = 'R';
                        res[hi--] = 'L';
                    }
                    start = p + 1;
                    p = start;
                }
            }
            else {
                if (start == R) {
                    for (int i = start + 1; i < p; i++)
                        res[i] = 'R';
                }
                start = p;
                R = start;
                p++;
            }
        }
        
        
        if (start == R) {
            for (int i = start + 1; i < n; i++)
                res[i] = 'R';
        }
        return new String(res);
    }
}
