class Solution {
   public String pushDominoes(String dominoes) {
        int n = dominoes.length();
        int[] left = new int[n];
        int[] right = new int[n];
        
        
int cur = -1;
        for (int i = 0; i < n; i++) {
            if (dominoes.charAt(i) == 'R') {
                cur = i;
            } else if (dominoes.charAt(i) == 'L') {
                cur = -1;
            }
            right[i] = cur;
        }
        
        cur = -1;
        
for (int i = n - 1; i >= 0; i--) {
            if (dominoes.charAt(i) == 'L') {
                cur = i;
            } else if (dominoes.charAt(i) == 'R') {
                cur = -1;
            
}
            left[i] = cur;
        }
        
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (right[i] < 0 && left[i] < 0) {
                sb.append('.');
            } else if (right[i] < 0) {
                sb.append('L');
            } else if (left[i] < 0) {
                sb.append('R');
            } else {
                if (i - right[i] > left[i] - i) {
                    sb.append('L');
                } else if (i - right[i] < left[i] - i) {
                    sb.append('R');
                } else {
                    sb.append('.');
                }
            
}
        }
        return sb.toString();
    }

}
