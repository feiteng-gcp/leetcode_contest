class Solution {
    public String pushDominoes(String dominoes) {
        int N = dominoes.length();
        if (N == 0) return dominoes;
        
        int[] closestLeftAction = new int[N];
        int[] closestRightAction = new int[N];
        char[] ca = dominoes.toCharArray();
        
        int lastAction = -1;
        for (int i = 0; i < N; ++i) {
            
if (ca[i] == '.') {
                closestLeftAction[i] = lastAction;
            } else {
                lastAction = i;
            }
        
}
        
        lastAction = -1;
        for (int i = N-1; i >= 0; --i) {
            if (ca[i] == '.') {
                closestRightAction[i] = lastAction;
            } else {
                lastAction = i;
            }
        }
        
        for (int i = 0; i < N; ++i) {
            if (ca[i] != '.') continue;
            int rightPush = -1, leftPush = -1;
            
if (closestLeftAction[i] >= 0 && ca[closestLeftAction[i]] == 'R') {
                rightPush = i - closestLeftAction[i];
            }
            if (closestRightAction[i] >= 0 && ca[closestRightAction[i]] == 'L') {
                leftPush = closestRightAction[i] - i;
            }
            if (rightPush >= 0 && leftPush < 0)
                ca[i] = 'R';
            else if (rightPush < 0 && leftPush >= 0) 
                ca[i] = 'L';
            else if (rightPush >= 0 && leftPush >= 0) {
                if (rightPush != leftPush) {
                    ca[i] = rightPush < leftPush ? 'R' : 'L';
                }
            }
        }
        
        // System.out.print(new String(ca));
        
        return new String(ca);
    }
}
