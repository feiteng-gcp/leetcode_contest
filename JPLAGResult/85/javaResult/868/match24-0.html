class Solution {
    public String pushDominoes(String dominoes) {
        if (dominoes == null || dominoes.length() == 0) {
            return dominoes;
        }
        int n = dominoes.length();
        int[] left = new int[n];
        int[] right = new int[n];
        Arrays.fill(left, -1);
        Arrays.fill(right, -1);
        
        int lastR = -1;
        for (int i = 0; i < n; i++) {
            if (dominoes.charAt(i) == 'R') {
                lastR = i;
            } else if (dominoes.charAt(i) == 'L') {
                lastR = -1;
            }
            if (lastR < 0) {
                continue;
            }
            right[i] = i - lastR;
        }
        
        int lastL = -1;
        for (int i = n - 1; i >= 0; i--) {
            if (dominoes.charAt(i) == 'L') {
                lastL = i;
            } else if (dominoes.charAt(i) == 'R') {
                lastL = -1;
            }
            if (lastL < 0) {
                continue;
            }
            left[i] = lastL - i;
        }
        
        //System.out.println(Arrays.toString(left));
        //System.out.println(Arrays.toString(right));
        
        StringBuilder result = new StringBuilder();
        for (int i = 0; i < n; i++) {
            if (left[i] == right[i]) {
                result.append(".");
            } else if (right[i] == -1 || (left[i] != -1 && left[i] < right[i])) {
                result.append("L");
            } else {
                result.append("R");
            }
        }
        
        return result.toString();
    }
}
