class Solution {
    public String pushDominoes(String dominoes) {
        if (dominoes.length() == 0) return "";
        int last = 0;
        int index = -1;
        StringBuilder result = new StringBuilder();
        char[] chars = dominoes.toCharArray();
        for (int i = 0; i < dominoes.length(); i ++) {
            char c =chars[i];
            if (c == 'L') {
                if (last != 1) {
                    for (int j = 0; j < i - index; j ++)
                    result.append("L");
                    
                } else {
                    int len = i - index - 1;
                    for (int j = 0; j < len / 2; j ++) result.append("R");
                    if (len % 2 != 0) result.append(".");
                    for (int j = 0; j < len / 2; j ++) result.append("L");
                    result.append("L");
                    
                }
                last = -1;
                index = i;
            } else if (c == 'R') {
                if (last != 1) {
                    for (int j = 0; j < i - index - 1; j ++)
                    result.append(".");
                    result.append("R");
                } else {
                    for (int j = 0; j < i - index; j ++)
                    result.append("R");
                }
                last = 1;
                index = i;
            }
        }
        
        if (last != 1) {
                    for (int j = 0; j < dominoes.length() - index - 1; j ++)
                    result.append(".");
                } else {
                    for (int j = 0; j < dominoes.length() - index - 1; j ++)
                    result.append("R");
                }
        return result.toString();
    }
}
