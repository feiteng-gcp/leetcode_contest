class Solution {
    public String pushDominoes(String dominoes) {
        int change = 0;
        
StringBuilder sb = new StringBuilder(dominoes);
        for(int i = 1;i < dominoes.length();i++)
        {
            if(i == 1 && dominoes.charAt(0) == '.' && dominoes.charAt(1) == 'L')
            {
                change++;
                sb.replace(0,1,"L");
                // dominoes.charAt(i) = 'L';
            }
                
            else if(dominoes.charAt(i) == 'L' && dominoes.charAt(i - 1) == '.' && dominoes.charAt(i-2) != 'R')
            {
                change++;
                // dominoes.charAt(i-1) = 'L';
                sb.replace(i-1,i,"L");
            
}
                
        }
        for(int i = dominoes.length() - 2;i >= 0;i--)
        {
            if(i == dominoes.length() - 2 && dominoes.charAt(i+1) == '.' && dominoes.charAt(i) == 'R')
            {
                change++;
                // dominoes.charAt(i) = 'R';
                sb.replace(i+1,i+2,"R");
            }
                
            else if(dominoes.charAt(i) == 'R' && dominoes.charAt(i + 1) == '.' && dominoes.charAt(i + 2) != 'L')
            {
                change++;
                // dominoes.charAt(i-1) = 'R';
                sb.replace(i+1,i+2,"R");
            }
                
        }
        
        dominoes = sb.toString();
        if(change > 0)
            return pushDominoes(dominoes);
        else
            return dominoes;
    }
}
