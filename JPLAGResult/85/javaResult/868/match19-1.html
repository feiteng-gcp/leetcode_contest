class Solution {
    
    public String pushDominoes(String dominoes) {
        if (dominoes == null || dominoes.length() == 0) {
            return dominoes;
        }
        
        StringBuilder res = new StringBuilder();
        int index = 0;
        int lastUpdatedUntil = 0;
        boolean lastSeenR = false;
        for (int i = 0; i < dominoes.length(); i++) {
            char curt = dominoes.charAt(i);
            if (curt == '.') {
                res.append('.');
            } else if (curt == 'R') {
                
res.append('R');
                if (lastSeenR) {
                    for (int j = lastUpdatedUntil; j < i; j++) {
                        res.setCharAt(j, 'R');
                    }
                }
                lastUpdatedUntil = i + 1;
                lastSeenR = true;
            } else {
                res.append('L');
                if (lastSeenR) {
                    for (int l = lastUpdatedUntil, r = i - 1; l < r; l++, r--) {
                        res.setCharAt(l, 'R');
                        
res.setCharAt(r, 'L');
                    }
                    
                } else {
                    for (int j = lastUpdatedUntil; j < i; j++) {
                        res.setCharAt(j, 'L');
                    } 
                }
                
                lastUpdatedUntil = i + 1;
                lastSeenR = false;
            }
        }
        
        if (lastSeenR && lastUpdatedUntil < dominoes.length()) {
            for (int j = lastUpdatedUntil; j < dominoes.length(); j++) {
                res.setCharAt(j, 'R');
            }
        }
        
        return res.toString();
    }
}
