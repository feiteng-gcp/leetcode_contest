class Solution {
    public String pushDominoes(String dominoes) {
        // char[] right = dominoes.toCharArray();
        
        // char[] left = do
        int len = dominoes.length();
        int[] right = new int[len];
        int[] left = new int[len];
        int force = 0;
        for (int i = 0; i < len; i++) {
            char cur = dominoes.charAt(i);
            if (cur == 'R') {
                force = 1;
            } else if (force != 0 && cur == '.') {
                force += 1;
            } else 
{
                force = 0;
            }
            right[i] = force;
        }
        
        force = 0;
        for (int i = len - 1; i >= 0; i--) {
            char cur = dominoes.charAt(i);
            if (cur == 'L') {
                force = 1;
            } else if (force != 0 && cur == '.') {
                force += 1;
            } else 
{
                force = 0;
            }
            left[i] = force;
        }
        
        StringBuilder result = new StringBuilder();
        
        for (int i = 0; i < len; i++) {
            if (right[i] == left[i]) {
                result.append('.');
            } else if (left[i] == 0 || right[i] != 0 && right[i] < left[i]) {
                result.append('R');
            } else if (right[i] == 0 || left[i] !=0 && left[i] < right[i] ) {
                result.append('L');
            
}
        }
        
        return result.toString();
    }
}
