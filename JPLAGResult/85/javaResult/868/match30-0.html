class Solution {
    public String pushDominoes(String dominoes) {
        if (dominoes == null || dominoes.length() == 0) {
            return dominoes;
        }
        StringBuilder resSB = new StringBuilder();
        int dotCount = 0;
        char left = ' ';
        boolean lookForRight = false;
        for (int i = 0; i < dominoes.length(); i++) {
            if (dominoes.charAt(i) == '.') {
                if (lookForRight) {
                    dotCount ++;
                    continue;
                } else {
                    dotCount = 1;
                    lookForRight = true;
                    left = i == 0 ? ' ' : dominoes.charAt(i - 1);
                }
            } else {
                if (!lookForRight) {
                    resSB.append(dominoes.charAt(i));
                    continue ;
                } else {
                    lookForRight = false;
                    char right = dominoes.charAt(i);
                    if (left == ' ' && right == 'L') {
                        for (int j = 0; j <= dotCount; j++) {
                            resSB.append('L');
                        }
                    } else if (left == ' ' && right == 'R') {
                        for (int j = 0; j < dotCount; j++) {
                            resSB.append('.');
                        }
                        resSB.append('R');
                    } else if (left == 'L' && right == 'R') {
                        for (int j = 0; j < dotCount; j++) {
                            resSB.append('.');
                        }
                        resSB.append('R');
                    } else if (left == 'L' && right == 'L') {
                        for (int j = 0; j <= dotCount; j++) {
                            resSB.append('L');
                        }
                    } else if (left == 'R' && right == 'R') {
                        for (int j = 0; j <= dotCount; j++) {
                            resSB.append('R');
                        }
                    } else {//left == 'R' && right == 'L'
                        if (dotCount % 2 == 0) {
                            for (int j = 1; j <= dotCount / 2; j++) {
                                resSB.append('R');
                            }
                            for (int j = 1; j <= dotCount / 2; j++) {
                                resSB.append('L');
                            }
                            resSB.append('L');
                        } else {
                            for (int j = 1; j <= dotCount / 2; j++) {
                                resSB.append('R');
                            }
                            resSB.append('.');
                            for (int j = 1; j <= dotCount / 2; j++) {
                                resSB.append('L');
                            }
                            resSB.append('L');
                        }
                    }
                }
            }
        }
        
        if (lookForRight) {
            if (left == 'L' || left == ' ') {
                for (int j = 0; j < dotCount; j++) {
                    resSB.append('.');
                }
            } else{
                for (int j = 0; j < dotCount; j++) {
                    resSB.append('R');
                }
            }
        }
        
        return resSB.toString();
        
    }
}
