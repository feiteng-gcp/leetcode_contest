class Solution {
    public String pushDominoes(String dominoes) {
        char[] domi = dominoes.toCharArray();
        int slow = -1;
        int fast = 0;
        int peng = 0;
        while (fast < domi.length) {
            if (domi[fast] == 'L' || domi[fast] == 'R') {
                peng++;
            }
            if (peng == 1 && slow == -1) {
                slow = fast;
            }
            if (peng == 2) {
                break;
            }
            fast++;
        }
        
        if (fast >= domi.length) {
            if (slow == -1) {
                return dominoes;
            
} else {
                if (domi[slow] == 'L') {
                    for (int i = 0; i < slow; i++) domi[i] = 'L';
                } else 
{
                    for (int i = slow + 1; i < domi.length; i++) domi[i] = 'R';
                }
                return new String(domi);
            }
        }
        int former = 0;
        while (fast < domi.length && slow < domi.length) {
            // System.out.println(slow + "  " + fast);
            if (domi[slow] == 'L') {
                for (int i = former; i < slow; i++) domi[i] = 'L';
                former = slow + 1;
                slow = fast;
                fast++;
                while (fast < domi.length) {
                    if (domi[fast] != '.') break;
                    fast++;
                }
            } else {
                if (domi[fast] == 'R') {
                    for (int i = slow + 1; i < fast; i++) domi[i] = 'R';
                    former = slow + 1;
                    slow = fast;
                    fast++;
                    while (fast < domi.length) {
                        if (domi[fast] != '.') break;
                        fast++;
                    }
                
} else {
                    int i = slow + 1;
                    int j = fast - 1;
                    while (i < j) {
                        domi[i++] = 'R';
                        domi[j--] = 'L';
                    }
                    former = fast + 1;
                    slow++;
                    while (slow < domi.length) {
                        if (domi[slow] != '.') break;
                        slow++;
                    }
                    fast = slow + 1;
                    while (fast < domi.length) {
                        if (domi[fast] != '.') break;
                        fast++;
                    }
                }
            }
        }
        if (slow >= domi.length) return new String(domi);
            if (domi[slow] == 'R') for (int i = slow + 1; i < domi.length; i++) domi[i] = 'R';
            else for (int i = former; i < slow ; i++) domi[i] = 'L';
            return new String(domi);
    }
}
