class Solution {
    public String pushDominoes(String dominoes) {
        char[] domi = dominoes.toCharArray();
        char preStatus = 'L';
        int preIndex = 0;
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < domi.length; i++) {
            if(domi[i] == '.') continue;
            if(domi[i] == preStatus) {
                buildString(sb, i - preIndex, preStatus);
            } else {
                if(domi[i] == 'R'){
                    buildString(sb, i - preIndex, '.');
                } else {
                    if((i - preIndex) % 2 == 1) {
                        buildString(sb, (i - preIndex)/2, 'R');
                        if (i - preIndex > 0)
                            sb.append(".");
                        buildString(sb, (i - preIndex)/2, 'L');
                    } else {
                        buildString(sb, (i - preIndex)/2, 'R');
                        buildString(sb, (i - preIndex)/2, 'L');
                    }
                }
            }
            sb.append(domi[i]);
            preIndex = i+1;
            preStatus = domi[i];
        }
        if(preStatus == 'R') {
            buildString(sb, domi.length - preIndex , 'R');
        } else {
            buildString(sb, domi.length - preIndex , '.');
        }
        return sb.toString();
    }
    
    private void buildString(StringBuilder sb, int n, char status) {
        for(int i = 0; i < n; i++) {
            sb.append(status);
        }
    }
}
