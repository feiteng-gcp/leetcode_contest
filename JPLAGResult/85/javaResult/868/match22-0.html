class Solution {
    public String pushDominoes(String dominoes) {
        int n = dominoes.length();
        int[] dir = new int[n];
        Map<Integer, Integer>  m  = new HashMap<>();
        for (int i = 0; i < dominoes.length(); ++i) {
            char ch = dominoes.charAt(i);
            if (ch == 'L') {
                m.put(i, -1);
                dir[i] = -1;
            } else if (ch == 'R') {
                m.put(i, 1);
                dir[i] = 1;
            }
        }
        
        while (!m.isEmpty()) {
            Map<Integer, Integer> next = new HashMap<>();
            for (Map.Entry<Integer, Integer> entry: m.entrySet()) {
                int p = entry.getKey() + entry.getValue();
                if (0 <= p && p < n) {
                    if (dir[p] == 0) {
                        dir[p] += entry.getValue();
                        next.put(p, entry.getValue());
                    } else if (next.containsKey(p)) {
                        dir[p] += entry.getValue();
                        next.remove(p);
                    }
                }
            }
            m = next;
        }
        
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; ++i) {
            if (dir[i] == -1) {
                sb.append('L');
            } else if (dir[i] == 1) {
                sb.append('R');
            } else {
                sb.append('.');
            }
        }
        return sb.toString();
    }
}
