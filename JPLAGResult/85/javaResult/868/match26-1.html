class Solution {
    public String pushDominoes(String dominoes) {
        char[] c = dominoes.toCharArray();
        int[] loc = new int[c.length];
        
        
int dis = 0;
        for(int i = 0; i < c.length; i++) {
            if(c[i] == 'L') {
                dis = 0;
            }
            else if (c[i] == 'R') {
                dis = 1;
            }
            else if(dis > 0) {
                dis++;
            }
            
            loc[i] = dis;
        
}
        
        // System.out.println(Arrays.toString(loc));
        
        dis = 0;
        for(int i = c.length - 1; i >= 0; i--) {
            if(c[i] == 'R') {
                dis = 0;
            }
            else if (c[i] == 'L') {
                dis = -1;
            }
            else if(dis < 0) {
                dis--;
            }
            if(dis == 0)
                continue;
            if(loc[i] == 0)
                loc[i] = dis;
            else if(Math.abs(loc[i]) > Math.abs(dis))
                loc[i] = dis;
            else if(Math.abs(loc[i]) == Math.abs(dis))
                loc[i] += dis;
        }
        
        // System.out.println(Arrays.toString(loc));
        
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < c.length; i++) {
            if(loc[i] == 0)
                sb.append(".");
            else if(loc[i] > 0) 
                sb.append("R");
            else 
                sb.append("L");
        }
        
        return sb.toString();
    }
}
