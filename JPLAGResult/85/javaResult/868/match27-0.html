class Solution {
    public String pushDominoes(String dominoes) {
        if (dominoes == null || dominoes.length() == 0) {
            return "";
        } else {
            
            StringBuffer sb = new StringBuffer(dominoes);
            sb.append('R');
            sb.insert(0, 'L');
            
            char[] chars = sb.toString().toCharArray();
            
            
            int prev = 0;
            for(int i = 1; i < chars.length; i ++) {
                
                if (chars[i] != '.') {
                    convert(chars, prev, i);    
                    
                    prev = i;
                }
                
            }
            
                     return new String(chars).substring(1, chars.length-1);

            
            
        }
    }
    
    private void convert(char[] chars, int start, int end) {
        
        if (start >= end) {
            return;
        }
        
        if (start +1 == end) {
            return;
        }
        
        char first = chars[start];
        char last = chars[end];
        
        if (first == 'L' && last == 'L') {
            
            for(int i = start + 1; i <= end - 1; i ++) {
                chars[i] = 'L';
            }
            
        } else  if (first == 'L' && last == 'R') {

            

        } else  if (first == 'R' && last == 'L') {
            
            int runnerLeft = start + 1;
            int runnerRight = end -1;
            
            while (runnerLeft < runnerRight) {
                chars[runnerLeft] = 'R';
                chars[runnerRight] = 'L';
                
                runnerLeft += 1;
                runnerRight -= 1;
            }
            
        } else  if (first == 'R' && last == 'R') {
            for(int i = start + 1; i <= end - 1; i ++) {
                chars[i] = 'R';
            }
        }
        
        
    }
}
