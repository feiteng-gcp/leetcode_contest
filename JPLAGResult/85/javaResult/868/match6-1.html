class Solution {
    public String pushDominoes(String dominoes) {
        int n = dominoes.length();
        int[] rs = new int[n];
        for (int i = 0, l = -1; i < n; i++) {
            char c = dominoes.charAt(i);
            if (c == 'R') {
                rs[i] = 0;
                l = i;
            } else if (c == '.') {
                if (l == -1) {
                    rs[i] = -1;
                } else {
                    rs[i] = i - l;
                }
            } else {
                l = -1;
            }
        }
        int[] ls = new int[n];
        for (int i = n - 1, l = n; i >= 0; i--) {
            char c = dominoes.charAt(i);
            if (c == 'L') {
                ls[i] = 0;
                l = i;
            } else if (c == '.') {
                if (l == n) {
                    ls[i] = -1;
                } else {
                    ls[i] = l - i;
                }
            } else {
                l = n;
            }
        }
        
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            //System.out.println(i + "," + ls[i] + ", " + rs[i]);
            if (dominoes.charAt(i) == 'L') {
                sb.append("L");
            } else if (dominoes.charAt(i) == 'R') {
                sb.append("R");
            } else 
{
                if (ls[i] == -1 && rs[i] == -1) {
                sb.append(".");
            } else if (ls[i] == -1) {
                sb.append("R");
            } else if (rs[i] == -1) {
                sb.append("L");
            } else {
            if (ls[i] < rs[i]) {
                sb.append('L');
            } else if (ls[i] > rs[i]) {
                sb.append('R');
            } else {
                sb.append('.');
            }
            }
            }
        }
        
        return sb.toString();
    }
}
