class Solution {
    public double new21Game(int N, int K, int W) {
        double p[] = new double[K + W + 10];
        double sP = 1.0 / W; //single draw
        
        p[0] = 1;
        
        for (int i = 0 ; i < K ; i++)
        {
        	for (int j = i + 1 ; j <= i + W ; j++){
        		p[j] = p[j] + p[i] * sP;
        	}
        }
        
        double ans = 0;
        
for (int i = K ; i <= N ; i++)
        {
        	ans = ans + p[i];
        }
        
        return ans;
    }
}
