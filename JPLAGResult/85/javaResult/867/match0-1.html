class Solution {
    public double new21Game(int N, int K, int W) {
        if (W == 1) {
            return 1;
        }
        if (K == 0) {
            return 1;
        }
        int top = Math.max(K + W, N + 1);
        double[] sum = new double[top];
        double[] dp = new double[top];
        dp[0] = 1;
        double ans = 0;
        for (int i = 0; i < 128; ++i) {
            // System.out.format("i = %d, dp = %s\n", i, Arrays.toString(dp));
            double[] sums = makeCumSum(dp);
            double[] next = new double[top];
            for (int j = i + 1; j < top; ++j) {
                int lo = Math.max(0, j - W);
                next[j] = (sums[j] - sums[lo]) / W;
                // double p = (sums[j] - sums[lo]) / W;
                // if (j < K) {
                //     next[j] = p;
                // } else {
                //     if (j <= N) {
                //         ans += p;
                //     }
                // }
            }
            for (int j = K; j <= N; ++j) {
                ans += next[j];
                next[j] = 0;
            }
            for (int j = N + 1; j < top; ++j) {
                next[j] = 0;
            }
            dp = next;
        }
        return ans;
    }
    
    public double[] makeCumSum(double[] arr) {
        int N = arr.length;
        double[] cs = new double[N + 1];
        for (int i = 0; i < N; ++i) {
            cs[i + 1] = cs[i] + arr[i];
        }
        return cs;
    }
}
