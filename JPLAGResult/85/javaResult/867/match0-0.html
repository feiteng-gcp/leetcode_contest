class Solution {
    public double new21Game(int N, int K, int W) {
        if (K == 0) {
            return 1.0;
        }
        double[] dp = new double[K + W + 1];
        dp[0] = 1.0;
        
        /* for (int i = 0; i < K; i++) {
            for (int move = 1; move <= W; move++) {
                dp[i + move] += 1.0 * dp[i] / W;
            }
        } */
        
        int l = -W, r = 0;
        double sum = 1.0;
        
        for (int i = 1; i < K + W; i++) {
            dp[i] = sum / W;
            r++;
            l++;
            if (r < K)
                sum += dp[r];
            if (l >= 0)
                sum -= dp[l];
                
        }
        
        /*for (int i = 0; i < K + W; i++){
            System.out.printf("%d %f\n", i, dp[i]);
        }*/
        
        double good = 0.0, total = 0.0;
        for (int i = K; i < K + W; i++) {
            total += dp[i];
            if (i <= N)
                good += dp[i];
        }
        
        return good / total;
    }
}
