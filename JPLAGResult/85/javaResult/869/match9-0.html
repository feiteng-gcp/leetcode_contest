class Solution {
    public int numSimilarGroups(String[] A) {
        Set<String> set = new HashSet<>();
        for (String s : A) {
            set.add(s);
        }
        int res = 0;
        Queue<String> queue = new LinkedList<>();
        while (!set.isEmpty()) {
            String source = set.iterator().next();
            set.remove(source);
            res++;
            queue.offer(source);
            while (!queue.isEmpty()) {
                String cur = queue.poll();
                for (String tmp : A) {
                    if (!set.contains(tmp)) {
                        continue;
                    }
                    if (isSimilar(tmp, cur)) {
                        set.remove(tmp);
                        queue.offer(tmp);
                    }
                }
            }
        }
        return res;
    }
    
    private boolean isSimilar(String a, String b) {
        int cnt = 0;
        for (int i = 0; i < a.length(); i++) {
            if (a.charAt(i) == b.charAt(i)) {
                cnt++;
            }
        }
        return cnt == a.length() - 2;
    }
}
