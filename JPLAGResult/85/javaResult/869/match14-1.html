class Solution {
     public int numSimilarGroups(String[] A) {
        if (A == null || A.length == 0) {
            return 0;
        } else {

            int size = A.length;
            int[] arr = 
new int[size];

            for (int i = 0; i < size; i++) {
                arr[i] = i;
            }


            for (int i = 0; i < size; i++) {
                for (int j = i + 1; j < size; j++) {

                    if (isSameGroup(A[i], A[j])) {
                        group(arr, i, j);
                    }
                }
            }
            
            Set<Integer> set = new HashSet<>();
            for(int i = 0; i <size; i ++) {
                set.add(root(arr, i));
            }
            
            return set.size();

        }
    }

    private int root(int[] arr, int x) {
        while (arr[x] != x) {
            arr[x] = arr[arr[x]];
            x = arr[arr[x]];
        }
        
        return x;
    }

    private void group(int[] arr, int x, int y) {
        int pX = root(arr,x);
        int py = root(arr, y);
        
        arr[pX] = py;
    }
    
    private boolean isSameGroup(String s1, String s2) {
        if (s1.length() != s2.length()) {
            return false;
        } else {
            List<Integer> diffs = new ArrayList<>();
            for (int i = 0; i < s1.length(); i++) {
                if (s1.charAt(i) != s2.charAt(i)) {
                    diffs.add(i);

                    if (diffs.size() >= 3) {
                        return false;
                    }
                }
            }

            return diffs.size() == 2;
        }
    }
}
