class Solution {
    public int numSimilarGroups(String[] A) {
        int n = A.length;
        if (n == 0) {
            return 0;
        }

        int m = A[0].length();
        if (m == 0) {
            return 1;
        }

        boolean[][] connected = new boolean[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                connected[i][j] = isConnected(A[i], A[j]);
            }
        }

        boolean[] vis = new boolean[n];
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            if (!vis[i]) {
                cnt++;
                Stack<Integer> stack = new Stack<>();
                stack.push(i);
                while (!stack.isEmpty()) {
                    int cur = stack.pop();
                    vis[cur] = true;
                    for (int j = 0; j < n; j++) {
                        int min = Math.min(cur, j);
                        int max = Math.max(cur, j);
                        if (connected[min][max] && !vis[j]) {
                            stack.push(j);
                        }
                    }
                }
            }
        }

        return cnt;
    
}

    boolean isConnected(String s1, String s2) {
        int cnt = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1.charAt(i) != s2.charAt(i)) {
                cnt++;
            }
        }

        return cnt == 2;
    }
}
