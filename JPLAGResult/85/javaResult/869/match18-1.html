class Solution {
Map<Integer, Integer> map = new HashMap<>();
    public int numSimilarGroups(String[] A) {
        int[] uf = new int[A.length];
        for(int i = 0 ; i < A.length; i++) {
            uf[i] = i;
            map.put(i, 1);
        }

        for(int i = 0 ; i < A.length-1; i++)
            for(int j = i + 1 ; j < A.length; j++)
                if(uf[j] != uf[i] && isSim(A[i], A[j])){
                    union(uf[i], uf[j], uf);
                }
        return map.size();
    
}
    void union(int i, int j, int[] uf){
        int iC = map.get(i);
        int jC = map.get(j);
        if(iC < jC) union(j,i,uf);
        else{
            map.put(i, iC + jC);
            map.remove(j);
            for(int t = 0 ; t < uf.length; t++){
                if(uf[t] == j) uf[t] = i;
            }
        }
    }
    boolean isSim(String a, String b)
{
        List<Integer> diff = new ArrayList<>();
        for(int i = 0 ; i < a.length(); i ++){
            if(a.charAt(i) != b.charAt(i)) diff.add(i);
        }
        if(diff.size() == 2){
            int f = diff.get(0); int s = diff.get(1);
            if(a.charAt(f) == b.charAt(s) && a.charAt(s) == b.charAt(f)) return true;
        }
        return false;
    }
    
}
