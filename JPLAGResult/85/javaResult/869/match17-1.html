class Solution {
   private class DSU {
        int[] pnt;
        
        public DSU(int n) {
            pnt = new int[n];
            for (int i = 0; i < n; i++) {
                pnt[i] = i;
            }
        }
        
        public int find(int x) {
            if (pnt[x] != x) {
                pnt[x] = find(pnt[x]);
            }
            return pnt[x];
        }
        
        public void merge(int x, int y) {
            int px = find(x);
            int py = find(y);
            if (px != py) {
                pnt[py] = px;
            }
        }
        
        public int numOfGroup() {
            int sum = 0;
            for (int i = 0; i < pnt.length; i++) {
                if (pnt[i] == i) {
                    sum++;
                }
            }
            return sum;
        }
    }
    
    private boolean isSimilar(String a, String b) {
        if (a.length() != b.length()) return false;
        
List<Integer> pos = new ArrayList<>();
        for (int i = 0; i < a.length(); i++) {
            if (a.charAt(i) != b.charAt(i)) {
                pos.add(i);
                if (pos.size() > 2) return false;
            }
        }
        
        if (pos.size() != 2) return false;
        return true;
    }
    public int numSimilarGroups(String[] A) {
        int n = A.length;
        DSU dsu = new DSU(n);
        
        
for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (isSimilar(A[j], A[i])) {
                    dsu.merge(i, j);
                }
            }
        }
        return dsu.numOfGroup();
    }
}
