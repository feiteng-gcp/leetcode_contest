class Solution {
    public int numSimilarGroups(String[] A) {
        int n = A.length;
        int[][] g = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = i +1; j < n; j++) {
                if (match(A[i], A[j])) {
                    g[i][j] = 1; g[j][i] = 1;
                }
            }
        }
        
        boolean[] visited = new boolean[n];
        int count = 0;    
        for (int i = 0; i < n; i++) {
            if (!visited[i]) {
                dfs(g, visited, i);
                count++;
            }
        }
        return count;
    }
    
    private void dfs(int[][] graph, boolean[] visited, int curr) {
        visited[curr] = true;
        for (int i = 0; i < graph.length; i++) {
            if (!visited[i] && graph[curr][i] == 1) {
                dfs(graph, visited, i);
            }
        }
    }
    
    private boolean match(String A, String B) {
        int count = 0;
        for (int i = 0; i < A.length(); i++) {
            if (A.charAt(i) != B.charAt(i)) count++;
            if (count > 2) return false;
        }
        return count == 2;
    }
}
