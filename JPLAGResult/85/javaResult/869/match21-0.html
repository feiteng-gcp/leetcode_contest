class Solution {

	
	private int findSet(int x) {
		if (x == set[x]) {
			return x;
		} else {
			return set[x] = findSet(set[x]);
		}
	}
	
	private boolean check(String s1 , String s2) {
		
		int i , length = s1.length() , cnt = 0;
		for (i = 0;i < length;i ++) {
			if (s1.charAt(i) != s2.charAt(i)) {
				cnt ++;
			}
		}
		return cnt == 2;
		
	}
	
	private int[] set = new int[10000];
	
    public int numSimilarGroups(String[] A) {
    
    	int i , j , n = A.length;
    	for (i = 0;i < n;i ++) {
    		set[i] = i;
    	}
    	for (i = 0;i < n;i ++) {
    		for (j = i + 1;j < n;j ++) {
    			if (check(A[i] , A[j])) {
    				int f1 = findSet(i) , f2 = findSet(j);
    				if (f1 != f2) {
    					set[f1] = f2;
    				}
    			}
    		}
    	}
    	Set<Integer> s = new HashSet<>();
    	for (i = 0;i < n;i ++) {
    		s.add(findSet(i));
    	}
    	return s.size();
    	
    }
	
}
