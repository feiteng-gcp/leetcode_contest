class Solution {
    public int numSimilarGroups(String[] A) {
        HashMap<String, String> parent = new HashMap<>();
        for(int i = 0; i < A.length; i++) {
            parent.put(A[i], A[i]);
        }
        
        for(int i = 0; i < A.length; i++) {
            for(int j = i + 1; j < A.length; j++) {
                if(isSimilar(A[i], A[j])) {
                    String parentA = findParent(parent, A[i]);
                    String parentB = findParent(parent, A[j]);
                    if(parentA.equals(parentB))
                        continue;
                    
                    parent.put(parentA, parentB);
                }
            }
        }
        
        HashSet<String> set = new HashSet<>();
        for(int i = 0; i < A.length; i++) {
            set.add(findParent(parent, A[i]));
        }
        
        return set.size();
    }
    
    private String findParent(HashMap<String, String> map, String target) {
        while(target != map.get(target)) {
            map.put(target, map.get(map.get(target)));
            target = map.get(target);
        
}
        
        return target;
    }
    
    private boolean isSimilar(String a, String b) {
        int index1 = -1, index2 = -1;
        char[] ca = a.toCharArray();
        char[] cb = b.toCharArray();
        for(int i = 0; i < a.length(); i++) {
            if(ca[i] == cb[i])
                continue;
            
            if(index1 == -1) {
                index1 = i;
            }
            else if(index2 == -1) {
                index2 = i;
            }
            else 
                return false;
        }
        
        if(ca[index1] == cb[index2] && ca[index2] == cb[index1])
            return true;
        
        return false;
    }
}
