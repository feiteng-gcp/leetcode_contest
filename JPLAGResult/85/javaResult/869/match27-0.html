class Solution {
    public boolean similar (String a, String b) {
        int count = 0;
        for (int i = 0; i < a.length(); i ++)if (a.charAt(i) != b.charAt(i)) count ++;
        return count == 2;
    }
    
    class UF {
        private int[] roots;
        public UF (int n) {
            roots = new int[n];
            for (int i = 0; i < n; i ++) roots[i] = i;
        }
        public void union (int a, int b) {
            roots[this.find(b)] = roots[this.find(a)];
        }
        public int find (int a) {
            while (roots[a] != a) a = roots[a];
            return a;
        }
    }
    public int numSimilarGroups(String[] A) {
        if (A.length == 0 ) return 0;
        if (A.length == 1) return 1;
        UF uf = new UF(A.length);
        for (int i = 0; i < A.length; i ++) {
            for (int j = i + 1; j < A.length; j ++) {
                
                if (similar(A[i], A[j])) uf.union(i, j);
                
            }
        }
        Set<Integer> parents = new HashSet<>();
        for (int i = 0; i < A.length; i ++ ) parents.add(uf.find(i));
        return parents.size();
        
    }
}
