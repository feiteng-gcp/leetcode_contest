class Solution {
    public int numSimilarGroups(String[] A) {
        int n = A.length;
        boolean[] seen = new boolean[n];
        
        int numGroups = 0;
        for (int i = 0; i < n; ++i) {
            if (!seen[i]) {
                ++numGroups;
                dfs(A, seen, i);
            }
        }
        
        return numGroups;
        
    }
    
    public void dfs(String[] A, boolean[] seen, int curr) {
        seen[curr] = true;
        
for (int i = 0; i < A.length; ++i) {
            if (!seen[i] && isSimilar(A[i], A[curr])) {
                dfs(A, seen, i);
            }
        }
    }
    
    public boolean isSimilar(String s, String t) {
        int diff = 0;
        for (int i = 0; i < s.length(); ++i) {
            if (s.charAt(i) != t.charAt(i)) {
                ++diff;
                if (diff > 2) 
{
                    return false;
                }
            }
        }
        
        return diff == 2;
    }
}
