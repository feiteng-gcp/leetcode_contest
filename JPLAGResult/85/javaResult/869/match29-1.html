class Solution {
    public int numSimilarGroups(String[] A) {
        List<List<String>> groups = new ArrayList<List<String>>();
        for(String s : A)
        {
            boolean find = false;
            boolean merge = false;
            List<String> mergeList = null;
            for (Iterator<List<String>> iter = groups.listIterator(); iter.hasNext(); ) {
                List<String> list = iter.next();
                for(String l : list)
                {
                    if(isSameGroup(s, l))
                    {
                        if(find)
                        {
                            merge = true;                        
                            break;                            
                        }
                        find = true;
                        list.add(s);
                        mergeList = list;
                        break;
                    }
                }
                if(merge)
                {
                    merge = false;
                    mergeList.addAll(list);
                    iter.remove();
                }
            }
            if(!find)
            {
                List<String> newList = new ArrayList<String>();
                newList.add(s);
                groups.add(newList);
            }
        }
        return groups.size();
    }
    
    
private boolean isSameGroup(String a, String b)
    {
        int count = 0;
        for(int i=0;i<a.length();i++)
            if(a.charAt(i) != b.charAt(i))
                count++;
        return count==2;
    }
}
