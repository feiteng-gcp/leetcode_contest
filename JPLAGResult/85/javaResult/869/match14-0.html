class Solution {
    int dsu[];
    Random rand = new Random();
    private int getParent(int x) {
        if (dsu[x] == x) {
            return x;
        }
        dsu[x] = getParent(dsu[x]);
        return dsu[x];
    }
    private void unite(int a, int b) {
        a = getParent(a);
        b = getParent(b);
        if (rand.nextInt(2) == 0) {
            dsu[a] = b;
        } else {
            dsu[b] = a;
        }
    }
    private boolean similar(String a, String b) {
        int diff1 = -1;
        int diff2 = -1;
        for (int i = 0; i < a.length(); i++) {
            if (a.charAt(i) == b.charAt(i))
                continue;
            if (diff2 != -1)
                return false;
            if (diff1 == -1) {
                diff1 = i;
            } else {
                diff2 = i;
            }
        }
        return diff1 != -1 && diff2 != -1 && a.charAt(diff1) == b.charAt(diff2) && a.charAt(diff2) == b.charAt(diff1);
    }
    public int numSimilarGroups(String[] A) {
        if (A == null || A.length == 0) {
            return 0;
        }
        int n = A.length;
        dsu = new int[n];
        for (int i = 0; i < n; i++) {
            dsu[i] = i;
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (similar(A[i], A[j])) {
                    unite(i, j);
                }
            }
        }
        
        Set<Integer> result = new HashSet<Integer>();
        for (int i = 0; i < n; i++) {
            result.add(getParent(i));
        }
        
        return result.size();
    }
}
