class Solution {
    public boolean isRectangleOverlap(int[] rec1, int[] rec2) {
        // int[] lt1 = new int[]{rec1[0],rec1[3]};
        // int[] rt1 = new int[]{rec1[2],rec1[3]};
        // int[] lb1 = new int[]{rec1[0],rec1[1]};
        // int[] rb1 = new int[]{rec1[2],rec1[1]};
        // int[] lt2 = new int[]{rec2[0],rec2[3]};
        // int[] rt2 = new int[]{rec2[2],rec2[3]};
        // int[] lb2 = new int[]{rec2[0],rec2[1]};
        // int[] rb2 = new int[]{rec2[2],rec2[1]};
        if(rec2[2]<=rec1[0] || rec2[3] <= rec1[1] || rec2[0]>=rec1[2] || rec2[1]>=rec1[3]) return false;
        return true;
    }
}
