#define VI vector<int>
#define MAP unordered_map<int, VI>

class Solution {
public:
    bool sim(string a, string b, int len){
        int dist = 0;
        for(int i=0; i<len; i++){
            if(a[i] != b[i]){
                dist++;
            }
            if(dist > 2){
                return false;
            }
        }
        return dist == 2;
    }
    
    int numSimilarGroups(vector<string>& A) {
        int sz = A.size();
        if(sz <= 1){
            return sz;
        }
        int len = A[0].size();
        MAP mp;
        for(int i=0; i<sz; i++){
            for(int j=i+1; j<sz; j++){
                if(sim(A[i], A[j], len) == true){
                    if(mp.find(i) == mp.end())
                        mp[i] = {j};
                    else
                        mp[i].push_back(j);
                    if(mp.find(j) == mp.end())
                        mp[j] = {i};
                    else
                        mp[j].push_back(i);
                }
            }
        }
        
        int ids[sz];
        bool visited[sz];
        memset(visited, false, sizeof(bool)*sz);
        for(int i=0; i<sz; i++){
            ids[i] = i;
        }
        int res = 0;
        for(int i=0; i<sz; i++){
            if(visited[i] == true){
                continue;
            }else{
                res++;
                visited[i] = true;
                stack<int> stk;
                stk.push(i);
                while(stk.empty() == false){
                    int _id = stk.top();
                    stk.pop();
                    for(int next : mp[_id]){
                        if(visited[next] == false){
                            visited[next] = true;
                            ids[next] = i;
                            stk.push(next);
                        }
                    }
                }
            }
        }
        return res;
        
    }
};
