const int N = 2000 + 10;
int f[N];
int cnt[26];

int find(int x) {
    if (f[x] != f[f[x]]) f[x] = find(f[x]);
    return f[x];
}

class Solution {
public:
    vector<int> A;
    bool similar(string& s, string& t) {
        int n = s.size(), m = t.size();
        if (n != m) return false;
        A.clear();
        fill(cnt, cnt + 26, 0);
        for (int i = 0; i < n; ++i) {
            if (s[i] != t[i]) {
                A.push_back(i);
            }
            cnt[s[i] - 'a']++;
        }
        if (A.size() == 2 && s[A[0]] == t[A[1]] && s[A[1]] == t[A[0]]) return true;
        if (A.size() == 0) {
            for (int k = 0; k < 26; ++k) {
                if (cnt[k] > 1) return true;
            }
        }
        return false;
    }
    int numSimilarGroups(vector<string>& A) {
        int n = A.size();
        for (int i = 0; i < n; ++i) f[i] = i;
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                if (similar(A[i], A[j])) {
                    int rx = find(i), ry = find(j);
                    if (rx != ry) {
                        f[ry] = rx;
                    }
                }
            }
        }
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            if (find(i) == i) ++ret;
        }
        return ret;
    }
};
