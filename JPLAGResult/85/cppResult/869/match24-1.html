public:

    vector<int> adj[MAX];
    bool seen[ MAX];
    
    void solve(int cur){
        seen[ cur ] = 1;
        for( int i = 0 ; i < adj[cur].size() ; ++i  ){
            if( !seen[ adj[cur][i]] ){
                solve( adj[cur][i] );
            }
        }
    }
    
    int numSimilarGroups(vector<string>& A) {
        
        int n = A.size();
        //makeSet(n);
        map<string, int> hash;
        
        for( int i = 0 ; i < n ; ++i ){
            hash[A[i]] = i;
        }
    
        for( int i = 0 ; i < MAX; ++i ) adj[i].clear(), seen[i] = 0;
        
        for( int i = 0 ; i < n ; ++i ){    
            string s = A[i];
            for( int j = 0 ; j < s.size(); ++j ){
                for( int k = j + 1 ; k < s.size() ; ++k ){
                    swap(s[j], s[k]);
                    if( hash.find(s) != hash.end() ){
                        adj[i].push_back( hash[s] );
                        adj[hash[s]].push_back(i);
                    }
                    swap(s[j], s[k]);

                }
            }
        }
        /*
        for( int i = 0 ; i < n ; ++i ){
            for( int j = i + 1 ; j < n ; ++j ){
                int dif = 0;
                for( int k = 0 ; k < A[i].size() ; ++k ){
                    if( A[i][k] == A[j][k] ) continue;
                    dif++;
                }
                if( dif == 2 ){
                	adj[i].push_back( hash[s] );
                    adj[hash[s]].push_back(i);
                }

        }*/

        int cnt = 0;
        
        for( int i = 0 ; i < n ; ++i ){
            if( !seen[i] ){
                cnt++;
                solve(i);
            }
        }
        return cnt;
        
    }
};
