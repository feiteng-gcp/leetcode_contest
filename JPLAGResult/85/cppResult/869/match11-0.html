class Solution {
    
    vector<int> fa;
    
    int get_fa(int x) {
        if (fa[x] < 0) return x;
        fa[x] = get_fa(fa[x]);
        return fa[x];
    }
    
    bool merge(int x, int y) {
        int fx = get_fa(x);
        int fy = get_fa(y);
        if (fx != fy) {
            fa[fx] = fy;
            return true;
        }
        return false;
    }
    
    
public:
    int numSimilarGroups(vector<string>& A) {
        int group = 0;
        int N = A.size();
        int L = A[0].size();
        
        fa = vector<int>(N, -1);
        
        for (int i = 0; i < N - 1; ++i) {
            for (int j = i + 1; j < N; ++j) {
                int cnt = 0;
                for (int k = 0; k < L; ++k) {
                    if (A[i][k] != A[j][k]) {
                        ++cnt;
                        if (cnt > 2) break;
                    }
                }
                
                if (cnt <= 2) {
                    merge(i, j);
                }
            }
        }
        
        
        vector<bool> vis(N, false);
        int groups = 0;
        for (int i = 0; i < N; ++i) {
            int fx = get_fa(i);
            if (!vis[fx]) {
                vis[fx] = true;
                ++groups;
            }
        }
        return groups;
    }
    
};
