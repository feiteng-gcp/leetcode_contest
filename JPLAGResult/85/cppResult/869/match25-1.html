int dp[2005];
int find(int a){
    return a==dp[a]?a:dp[a]=find(dp[a]);
}
void merge(int a, int b){
    dp[find(a)] = find(b);
}
class Solution {
public:
    
    int numSimilarGroups(vector<string>& A) {
        int len  = A.size();
        if(!len) return 0;
        int m = A[0].size();

        for(int i=0;i<len;i++){
            dp[i]=i;
        }
        for(int i=0;i<len;i++){
            for(int j=i+1;j<len;j++){
                if(find(i)==find(j)) continue;
                int sum = 0;
                for(int k=0;k<m;k++){
                    if(A[i][k]!=A[j][k]) sum++;
                    if(sum>2) break;
                }
                if(sum<=2) merge(i, j);

            }
        }
        int res = 0;
        for(int i=0;i<len;i++){
            if(dp[i]==i) res++;
        }
        return res;
    }
};
