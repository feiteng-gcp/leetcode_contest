class Solution {
public:
    int numSimilarGroups(vector<string>& A) {
        vector<int> isGrouped(A.size(), 0);
        queue<int> next;
        int ans = 0;
        
        for (int i = 0; i < isGrouped.size(); ++i)
        {
            if (isGrouped[i] == 0)
            {
                ++ans;
                next.push(i);
                isGrouped[i] = 1;
                
                while (!next.empty())
                {
                    int current = next.front();
                    next.pop();
                    
                    for (int j = 0; j < isGrouped.size(); ++j)
                    {
                        if (isGrouped[j] != 0)
                            continue;
                        
                        if (isSimilar(A[current], A[j]))
                        {
                            isGrouped[j] = 1;
                            next.push(j);
                        }
                    }
                }
            }
        }
        
        return ans;
    }
    
    bool isSimilar(string& a, string& b)
    {
        int diffCount = 0;
        
        for (int i = 0; i < a.size(); ++i)
        {
            if (a[i] != b[i])
            {
                diffCount++;
                
                if (diffCount > 2)
                    return false;
            }
        }
        
        return true;
    }
};
