class Solution {
public:
    int g(vector<int> & group, int i) {
        if (group[i] == -1)
            return i;
        group[i] = g(group, group[i]);
        return group[i];
    }
    
    void merge(vector<int> & group, int a, int b, int & N) {
        int ga = g(group, a);
        int gb = g(group, b);
        if (ga == gb)
            return;
        group[gb] = ga;
        N--;
    }
    
    bool sim(string & a, string & b) {
        int count = 0;
        for (int i = 0; i < a.length(); i++) {
            if (a[i] != b[i])
                count++;
            if (count > 2)
                return false;
        }
        return true;
    }
    
    int numSimilarGroups(vector<string>& A) {
        int N = A.size();
        vector<int> group(N, -1);
        int NN = N;

        for (int i = 0; i < NN; i++) {
            for (int j = 0; j < NN; j++) {
                if (sim(A[i], A[j])) {
                    merge(group, i, j, N);
                }
            }
        }
        return N;
    }
};
