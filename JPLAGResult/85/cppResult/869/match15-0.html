class Solution {
public:
    int numSimilarGroups(vector<string>& A) {
        int size=A.size();
        vector<vector<int>>g(size,vector<int>());
        for(int i=0;i<size-1;i++){
            for(int j=i+1;j<size;j++){
                if(check(A[i],A[j])){
                    g[i].push_back(j);
                    g[j].push_back(i);
                }
                //cout<<i<<'\t'<<j<<endl;
            }
        }
        int res=0;
        vector<int>mark(size,0);
        for(int i=0;i<size;i++){
            if(!mark[i]){
                dfs(i,g,mark);
                res++;
            }
        }
        return res;
    }
private:
    bool check(string&a,string&b){
        if(a.size()!=b.size())return false;
        int len=(int)a.size();
        vector<int>diff;
        int counter=0;
        for(int i=0;i<len;i++){
            if(a[i]!=b[i]){
                counter++;
                diff.push_back(i);
            }
        }
        if(counter>2)return false;
        else if(counter==2&&(a[diff[0]]==b[diff[1]]&&a[diff[1]]==b[diff[0]]))return true;
        return false;
    }
    void dfs(int now,vector<vector<int>>&g,vector<int>&mark){
        mark[now]=1;
        for(int&i:g[now]){
            if(!mark[i])dfs(i,g,mark);
        }
    }
};
