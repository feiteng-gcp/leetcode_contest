class Solution {
public:
    int numSimilarGroups(vector<string>& A) {
        if (A.size() < 2) return A.size();
        
        vector<int> relations(A.size());
        for (int i = 0; i < A.size(); i++) {
            relations[i] = i;
        }


        for (int i = 0; i < A.size() - 1; i++) {
            for (int j = i + 1; j < A.size(); j++) {
                if (find(relations, i) != find(relations, j) && isSimilar(A[i], A[j])) {
                    relations[find(relations, i)] = find(relations, j);
                }
            }
        }

        unordered_set<int> groups;
        for (int i : relations) {
            groups.insert(find(relations, i));
        }

        return groups.size();
    }

    bool isSimilar(string& a, string& b) {

        int diff = 0;
        for (int i = 0; i < a.size(); i++) {
            if (a[i] != b[i]) diff++;
        }

        return diff == 2;
    }

    int find(vector<int>& relations, int target) {
        while (relations[target] != relations[relations[target]]) {
            relations[target] = relations[relations[target]];
        }
        return relations[target];
    }
};
