class Solution {
    void dfs(int s, unordered_map<int, vector<int>> &edge, vector<bool> &flag) {
        flag[s] = false;
        for (auto node : edge[s]) {
            if (flag[node]) dfs(node, edge, flag);
        }
    }
public:
    int numSimilarGroups(vector<string>& A) {
        int n = A.size();
        unordered_map<int, vector<int>> edge;
        
        for (int i = 0; i < n; i += 1) {
            for (int j = i+1; j < n; j += 1) {
                int size = A[i].size();
                vector<int> pos;
                for (int k = 0; k < size; k += 1) {
                    if (A[i][k] != A[j][k]) {
                        pos.push_back(k);
                        if (pos.size() > 2) break;
                    }
                }
                if (pos.size() == 2) {
                    edge[i].push_back(j);
                    edge[j].push_back(i);
                }
            }
        }
        
        vector<bool> flag(n, true);
        int res = 0;
        for (int i = 0; i < n; i += 1) {
            if (flag[i]) {
                res += 1;
                dfs(i, edge, flag);
            }
        }
        return res;
    }
};
