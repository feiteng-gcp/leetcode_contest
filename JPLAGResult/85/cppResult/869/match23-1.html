class Solution {
public:
    int numSimilarGroups(vector<string>& A) {

        vector<list<int>> con(A.size(), list<int>());
        for (int i = 0; i < A.size(); ++i) {
            for (int j = i + 1; j < A.size(); ++j) {
                int count = 0;
                for (int k = 0; k < A[i].size(); ++k) {
                    if (A[i][k] != A[j][k]) {
                        count++;

                        if (count > 2) break;
                    }
                }
                if (count <= 2) {
                    con[i].push_back(j);
                    con[j].push_back(i);
                }
            }
        }
        
        int nrGroup = 0;
        vector<bool> visited(A.size(), false);
        
        for (auto it = visited.begin();;) {
            it = std::find(it, visited.end(), false);
            if (it == visited.end()) break;
            nrGroup++;
            int start = distance(visited.begin(), it);
            list<int> q, q2;
            q.push_back(start);
            while (!q.empty()) {
                for (auto n: q) {
                    visited[n] = true;
                    for (auto s: con[n]) {
                        if (!visited[s])
                            q2.push_back(s);
                    }
                }
                swap(q, q2);
                q2.clear();
            }
        }
        return nrGroup;
    }
};
