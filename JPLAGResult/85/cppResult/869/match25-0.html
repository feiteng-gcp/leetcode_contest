class Solution {
    vector<int> uf;
public:
    int numSimilarGroups(vector<string>& A) {
        int len = A.size();
        uf = vector<int>(len);
        for (int i = 0; i < len; i++) {
            uf[i] = i;
        }
        for (int i = 1; i < len; i++) {
            for (int j = 0; j < i; j++) {
                if (uf[i] == uf[j])
                    continue;
                if (similar(A[i], A[j])) {
                    merge(i, j);
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < len; i++) {
            if (uf[i] == i)
                ans++;
        }
        return ans;
    }

    void merge (int n1, int n2) {
        int h1 = head(n1), h2 = head(n2);
        if (h1 == h2)
            return;
        uf[h2] = h1;
    }

    int head (int n) {
        int p = n;
        while (uf[p] != p) {
            p = uf[p];
        }
        return p;
    }

    bool similar (const string &s1, const string &s2) {
        vector<int> pos;
        int n = 0;
        for (int i = 0; i < s1.length(); i++) {
            if (s1[i] != s2[i]) {
                if (++n > 2) {
                    return false;
                }
                pos.push_back(i);
            }
        }
        if (n != 2)
            return false;
        int p1 = pos[0], p2 = pos[1];
        if (s1[p1] == s2[p2] && s1[p2] == s2[p1])
            return true;
        return false;
    }
};
