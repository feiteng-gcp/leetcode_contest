class Solution {
public:
    int numSimilarGroups(vector<string>& A) {
        unordered_map<string, vector<string>> m;
        for (int i = 0; i < A.size() - 1; ++i) {
            for (int j = i+1; j < A.size(); ++j) {
                if (distance2(A[i], A[j])) {
                    m[A[i]].push_back(A[j]);
                    m[A[j]].push_back(A[i]);
                }
            }
        }
        
        unordered_set<string> visited;
        int ret = 0;
        for (string& s: A) {
            if (visited.count(s)) continue;
            
            ++ret;
            dfs(s, m, visited);
        }
        return ret;
    }
    
    void dfs(string& s, unordered_map<string, vector<string>>& m, unordered_set<string>& visited) {
        if (visited.count(s)) return;
        
        visited.insert(s);
        for (string& next: m[s]) {
            dfs(next, m, visited);
        }
    }
    
    bool distance2(string& s1, string& s2) {
        int distance = 0;
        for (int i = 0; i < s1.size() && distance <= 2; ++i) {
            if (s1[i] != s2[i]) ++distance;
        }
        return distance <= 2;
    }
};
