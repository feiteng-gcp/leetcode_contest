class Solution {
public:
    bool isSimilar(string a, string b) {
        int cnt = 0;
        int x = -1, y = -1;
        for (int i = 0; i < a.length(); i++) {
            if (a[i] != b[i]) {
                cnt++;
                if (x == -1) {
                    x = i;
                } else {
                    y = i;
                }
            }
            if (cnt > 2) {
                break;
            }
        }
        swap(a[x], a[y]);
        return a == b;
    }
    string find(unordered_map<string, string>& head, string x) {
        if (head.find(x) == head.end()) {
            head[x] = x;
        }
        if (head[x] == x) {
            return x;
        } else {
            head[x] = find(head, head[x]);
            return head[x];
        }
    }
    void merge(unordered_map<string, string>& head, string x, string y) {
        string hx = find(head, x), hy = find(head, y);
        if (hx != hy) {
            head[hx] = hy;
        }
    }
    int numSimilarGroups(vector<string>& A) {
        unordered_map<string, string> head;
        for (int i = 0; i < A.size(); i++) {
            find(head, A[i]);
            for (int j = i + 1; j < A.size(); j++) {
                if (isSimilar(A[i], A[j])) {
                    merge(head, A[i], A[j]);
                }
            }
        }
        int cnt = 0 ;
        for (auto it: head) {
            if (it.first == it.second) {
                cnt++;
            }
        }
        return cnt;
    }
};
