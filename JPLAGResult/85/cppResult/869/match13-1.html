class Solution {
public:
    int numSimilarGroups(vector<string>& A) {

        int a[3000];
        for(int i=0;i<A.size();i++){
            a[i]=i;
        }

        for(int i=1;i<A.size();i++){
            for(int j=0;j<i;j++){
                if(a[i]==a[j]){
                    continue;
                }
                if(A[i].size()!=A[j].size()){
                    continue;
                }

                int x=0;
                int x1;
                int x2;
                int f=0;
                for(int k=0;k<A[i].size();k++){
                    if(A[i][k]!=A[j][k]){
                        x++;
                        if(x==1){
                            x1=A[i][k];
                            x2=A[j][k];
                        }
                        if(x==2){
                            if(A[i][k]==x2&&A[j][k]==x1){
                                continue;
                            }
                            f=1;
                            break;
                        }
                        if(x>2){
                            f=1;
                            break;
                        }
                    }
                }
                if(f==0&&x==2){
                    int ii=i;
                    int jj=j;
                    while(a[ii]!=ii){
                        ii=a[ii];
                    }
                    while(a[jj]!=jj){
                        jj=a[jj];
                    }
                    int min=a[jj];
                    if(a[ii]<a[jj]){
                        min=a[ii];
                    }
                    a[ii]=min;
                    a[jj]=min;
                }
            }
        }
        int asc=0;

        for(int i=0;i<A.size();i++){
            if(a[i]==i){
                asc++;
            }
            //cout<<a[i]<<endl;
        }
        return asc;
    }
};
