class Solution {
public:
    vector<int> parent;
    

    bool similiar(string a, string b) {
        int c = 0;
        for (int i = 0; i < a.length(); ++i) {
            if (a[i] != b[i]) {
                c++;
                if (c > 2) return false;
            }
        }
        return c == 2;
    }
    
    int find(int index) {
        if (parent[index] != index) 
            parent[index] = find(parent[index]);
        return parent[index];
    }
    
    int numSimilarGroups(vector<string>& A) {
        parent.clear();
        for (int i = 0; i < A.size(); i++) {
            parent.push_back(i);
        }
        
        for (int i = 0; i < A.size(); i++) {
            int root = i;
            for (int j = 0; j < i; ++j) {
                if (find(i) == find(j)) {
                    continue;
                }
                if (similiar(A[i], A[j])) {
                    if (root == i) {
                        root = find(j);
                    }
                    parent[find(j)] = root;
                    parent[i] = root;
                }
            }
        }
        
        unordered_set<int> usi;
        for (int i = 0; i < A.size(); ++i) {
            usi.insert(find(i));
        }
        return usi.size();
    }
};
