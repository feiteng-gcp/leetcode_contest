class Solution {
public:
    bool isSimilar(string& a, string& b) {
        int len = a.length();
        vector<int> pos;
        for (int i = 0; i < len; i++) {
            if (a[i] != b[i]) pos.push_back(i);
        }
        if (pos.size() != 2) return false;
        if (a[pos[0]] == b[pos[1]] && a[pos[1]] == b[pos[0]]) return true;
        return false;
    }
    void addEdge(vector<int>& father, int i, int j) {
        int fi = getFather(father, i);
        int fj = getFather(father, j);
        father[fj] = fi;
    }
    int getFather(vector<int>& father, int i) {
        if (father[i] == i) return i;
        int f = getFather(father, father[i]);
        father[i] = f;
        return f;
    }
    int numSimilarGroups(vector<string>& A) {
        int len = A.size();
        vector<int> father(len);
        for (int i = 0; i < len; i++) father[i] = i;

        for (int i = 0; i < len; i++) {
            for (int j = i + 1; j < len; j++) {
                if (isSimilar(A[i], A[j])) {
                    addEdge(father, i, j);
                }
            }
        }
        unordered_set<int> s;
        for (int i = 0; i < len; i++) s.insert(getFather(father, i));
        return s.size();
    }
};
