class Solution {
public:
    vector<int> uf ; 
    unordered_map< string , int > memo ; 
    int length ;  
    int find( int u ){
        if( uf[u] == u )
            return u ;
        return uf[u] = find( uf[u] ) ; 
    } 
    void Union( int u , int v ){
        int fu = find( u ) ; 
        int fv = find( v ) ; 
        if( fu == fv )
            return ;
        uf[fv] = fu;
    }
    void similar( string s1, string s2 ){ 
        int count = 0 ; 
        for( int i=0 ; i<length ; i++){
            if( s1[i] != s2[i] ){
                count ++;
            }
        }
        if( count == 2 ){
            Union( memo[s1] , memo[s2] );
        } 
    }
    int numSimilarGroups(vector<string>& A) {
        int n = A.size() ; 
        length = A[0].size() ; 
        uf.resize( n+1 , 0 ) ; 
        for( int i=1 ; i<= n  ; i ++){
            uf[i]=i; 
        }

        for( int i=0 ; i<A.size() ; i++){
            memo[A[i]]= i+1; 
        }
        for( int i=1 ; i< n ; i++){
            for( int j=0 ; j<i ; j++){
                if( find( i+1 ) == find( j+1 ) ) {
                    continue ; 

                }   
                similar( A[i] , A[j] ) ; 
            }
        }
        int count =  0 ;
        for( int i=1 ; i<=n ; i++){
            if( uf[i] == i )count++;
            //cout << uf[i] <<' ' ; 
        }
        return count  ; 
    }
};
