class Solution {
public:
    bool ov(int a1, int a2, int b1, int b2) {
        return max(a1,b1) < min(a2, b2);
    }
    bool isRectangleOverlap(vector<int>& rec1, vector<int>& rec2) {
        return ov(rec1[0], rec1[2], rec2[0], rec2[2]) && ov(rec1[1], rec1[3], rec2[1], rec2[3]);
    }
};
