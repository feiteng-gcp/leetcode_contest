class Solution {
public:
    bool isRectangleOverlap(vector<int>& rec1, vector<int>& rec2) {
        
        // total conditions:
		// 9 regions from rec1, for the top control points of rec2, total we have 9x9=81
		// 1 2 3
		// 4 5 6
		// 7 8 9
		// first rule out not intersection:
		// p1 in 3/6/9/7/8, total 11 conditions
		// p2 in 1/2/3/4/7, total 11 conditions
		// otherwise they intersect
		if ( rec2[ 0 ] >= rec1[2] ||
			rec2[1] >= rec1[3] || 
			rec2[3] <= rec1[ 1] ||
			rec2[2] <= rec1[ 0 ] )
		{
			return false;
		}
		return true;
    }
};
