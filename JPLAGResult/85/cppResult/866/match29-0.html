class Solution {
public:
    bool isRectangleOverlap(vector<int>& rec1, vector<int>& rec2) {
        int l1 = rec1[0];
        int b1 = rec1[1];
        int r1 = rec1[2];
        int t1 = rec1[3];
        
        int l2 = rec2[0];
        int b2 = rec2[1];
        int r2 = rec2[2];
        int t2 = rec2[3];
        
        if(l1 >= r2 || l2 >= r1 || b1 >= t2 || b2 >= t1) return false;
        return true;
    }
};
