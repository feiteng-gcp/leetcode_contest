class Solution {
public:
    bool isRectangleOverlap(vector<int>& rec1, vector<int>& rec2) {
        if (!test(rec1[0], rec1[2], rec2[0], rec2[2]))
            return false;
        if (!test(rec1[1], rec1[3], rec2[1], rec2[3]))
            return false;
        return true;
    }
    
    bool test(int a1, int a2, int b1, int b2)
    {
        if (a1 <= b1 && a2 <= b1)
            return false;
        if (b1 <= a1 && b2 <= a1)
            return false;
        return true;
    }
};
