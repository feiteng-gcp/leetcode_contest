class Solution {
public:
    bool isRectangleOverlap(vector<int>& rec1, vector<int>& rec2) {
        // 1 is to right of 2
        int Ax2=rec1[2];
        int Ay2=rec1[3];
        int Ax1=rec1[0];
        int Ay1=rec1[1];
        
        int Bx2=rec2[2];
        int By2=rec2[3];
        int Bx1=rec2[0];
        int By1=rec2[1];
        
        return  !(Bx1>=Ax2 || By1>=Ay2 || Ax1>=Bx2 || Ay1>=By2);
    }
};
