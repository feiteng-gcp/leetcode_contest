class Solution {
public:
    bool rangeOverlap(int a1, int b1, int a2, int b2) {
        return !(b2 <= a1 || a2 >= b1);
    }
    bool isRectangleOverlap(vector<int>& rec1, vector<int>& rec2) {
        return rangeOverlap(rec1[0], rec1[2], rec2[0], rec2[2])
            && rangeOverlap(rec1[1], rec1[3], rec2[1], rec2[3]);
    }
};
