class Solution {
public:
    bool isRectangleOverlap(vector<int>& rec1, vector<int>& rec2) {
        int maxleft_x = max(rec1[0],rec2[0]); //cout<<maxleft_x<<endl;
        int minright_x = min(rec1[2],rec2[2]);
        int maxleft_y = max(rec1[1],rec2[1]);
        int minright_y = min(rec1[3],rec2[3]);
        if((maxleft_x < minright_x) && (maxleft_y < minright_y))
            return true;
        return false;
    }
};
