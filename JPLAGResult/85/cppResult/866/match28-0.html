class Solution {
public:
    bool isRectangleOverlap(vector<int>& rec1, vector<int>& rec2) {
        //(l,b,r,t)
        int a_l = rec1[0];
        int a_b = rec1[1];
        int a_r = rec1[2];
        int a_t = rec1[3];
       
        int b_l = rec2[0];
        int b_b = rec2[1];
        int b_r = rec2[2];
        int b_t = rec2[3];
        
        if (a_l >= b_r || a_r <= b_l || a_b >= b_t || a_t <= b_b) {
            return false;
        }
        return true;
    }
};
