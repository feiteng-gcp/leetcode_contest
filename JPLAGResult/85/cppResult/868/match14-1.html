class Solution {
public:
    string pushDominoes(string dominoes) {
        int i;
        int dir;
        int dis;
        vector<int> left(dominoes.length(), 0);
        vector<int> right(dominoes.length(), 0);
        string res = dominoes;
        
        dir = -1;

        for(i = 0; i < dominoes.length(); ++i) {
            if(dominoes[i] == 'L') {
                dir = -1;
                dis = 0;
            } else if(dominoes[i] == 'R') {
                dir = 1;
                dis = 0;
            } else ++dis;
            left[i] = dir * dis;
        }
        
        dir = 1;

        for(i = dominoes.length() - 1; i >= 0; --i) {
            if(dominoes[i] == 'L') {
                dir = -1;
                dis = 0;
            } else if(dominoes[i] == 'R') {
                dir = 1;
                dis = 0;
            } else ++dis;
            right[i] = dir * dis;
        }
        
        for(i = 0; i < dominoes.length(); ++i) {
            if(left[i] > 0 && right[i] < 0) {
                if(left[i] + right[i] > 0) res[i] = 'L';
                else if(left[i] + right[i] < 0) res[i] = 'R';
                else res[i] = '.';
            }
            else if(left[i] < 0 && right[i] < 0) res[i] = 'L';
            else if(left[i] > 0 && right[i] > 0) res[i] = 'R';
        }
        
        return res;
    }
};
