class Solution {
public:
    string pushDominoes(string dominoes) {
        int n = dominoes.size();
        vector<int> l(n, -1000000);
        vector<int> r(n, -1000000);
        if (dominoes[0] == 'R') r[0] = 0;
        for (int i = 1; i < n; i++)
        {
            if (dominoes[i] == 'R') r[i] = 0;
            if (dominoes[i] == '.') r[i] = r[i-1] + 1;
        }
        
        if (dominoes[n - 1] == 'L') l[n-1] = 0;
        for (int i = n - 2; i >= 0; i--)
        {

            if (dominoes[i] == 'L') l[i] = 0;
            if (dominoes[i] == '.') l[i] = l[i + 1] + 1;
        }
        
        for (int i = 0; i < n; i++)
        {
            if (dominoes[i] == '.')
            {
                if (l[i] >= 0 || r[i] >= 0)
                {
                    if (l[i] == r[i]) continue;
                    if (l[i] < 0) dominoes[i] = 'R';
                    else if (r[i] < 0) dominoes[i] = 'L';
                    else if (l[i] > r[i]) dominoes[i] = 'R';
                    else dominoes[i] = 'L';
                }
            }
        }
        return dominoes;
    }
};
