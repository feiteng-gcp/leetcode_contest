class Solution {
public:
    string pushDominoes(string dominoes) {
        vector<int> res;
        int lastRight = -1;
        for (int i = 0; i < dominoes.length(); i++) {
            if (lastRight != -1 && dominoes[i] == 'L') {
                res.push_back(lastRight);
                res.push_back(i);
                lastRight = -1;
            }
            
            if (dominoes[i] == 'R') {
                lastRight = i;
            }
        }
        
        int lastI = 0;
        string aa;
        for (int i = 0; i < res.size() / 2; i++) {
            int left = res[2*i];
            int right = res[2*i + 1];
            for (int j = lastI; j < left; j++) {
                aa += dominoes[j];
            }
            for (int j = left; j <= right; j++) {
                if (2 * j < left + right) {
                    aa += 'R';
                } else if (2 * j == left + right) {
                    aa += '.';
                } else {
                    aa += 'L';
                }
            }
            lastI = right + 1;
        }
        for (int i = lastI; i < dominoes.length(); i++) {
            aa += dominoes[i];
        }
        
        int lastOne = -1;

        int lastIndex = 0;
        for (int i = 0; i < aa.length(); i++) {
            if (aa[i] == 'L') {
                if (lastOne == -1) {
                    for (int k = 0; k < i; k++) {
                        aa[k] = 'L';
                    }
                } else if (lastOne == 0) {
                    for (int k = lastIndex; k < i; k++) {
                        aa[k] = 'L';
                    }
                }
                lastOne = 0;
                lastIndex = i;
            }
            

            if (aa[i] == 'R') {
                if (lastOne == 1) {
                    for (int k = lastIndex; k < i; k++) {
                        aa[k] = 'R';
                    }
                }
                
                lastOne = 1;

                lastIndex = i;
            }
        }
        if (lastOne == 1) {
            for (int i = lastIndex; i < aa.length(); i++) {
                aa[i] = 'R';
            }
        } 
        return aa;
        
    }
        
        char choss(int k) {
            if (k == 0) {
                return '.';
            }
            if (k == 1) {
                return 'R';
            }
            if (k == 2) {
                return 'L';
            }
        }

};
