class Solution {
public:
    string pushDominoes(string dominoes) {
        int n = dominoes.size();
        vector<int> time(n, 0);
        for (int i = 0; i < n;)
        {
            if (dominoes[i] == '.')
            {
                i++;
            }
            else if (dominoes[i] == 'L')
            {
                int curtime = 0;
                int pos = i - 1;
                while (pos >= 0)
                {
                    curtime += 1;

                    if (dominoes[pos] == '.')
                    {
                        time[pos] = curtime;
                        dominoes[pos] = 'L';
                    }
                    else if (time[pos] > curtime)
                    {
                        time[pos] = curtime;
                        dominoes[pos] = 'L';
                    }
                    else if (time[pos] == curtime && dominoes[pos] == 'R')
                    {
                        dominoes[pos] = '.';
                    }
                    else
                        break;
                    pos--;
                }
                i++;
            }
            else
            {
                int curtime = 0;
                int pos = i + 1;
                while (pos < n)
                {
                    curtime += 1;
                    if (dominoes[pos] == '.')
                    {
                        time[pos] = curtime;
                        dominoes[pos] = 'R';
                        pos++;
                    }
                    else
                        break;
                }
                i = pos;
            }
        }
        return dominoes;
    }
};
