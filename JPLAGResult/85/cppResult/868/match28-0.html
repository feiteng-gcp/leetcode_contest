#include<bits/stdc++.h>

#define sz(x) ((int)(x).size())
#define all(x) (x).begin(),(x).end()
#define rep(i,l,r) for(int i=(l);i<(r);++i)
#define per(i,l,r) for(int i=(r)-1;i>=(l);--i)
//-------
class Solution {
public:
	int pr[100007], pl[100007];
    string pushDominoes(string s) {
    	int n = sz(s);
    	pl[n] = pr[n] = n;
    	per(i, 0, n) {
    		pl[i] = pl[i + 1];
    		pr[i] = pr[i + 1];
    		if (s[i] == 'L') {
    			pl[i] = i;
    		} else if (s[i] == 'R') {
    			pr[i] = i;
    		}
    	}
    	int last = -1;
    	string ret = s;
    	rep(i, 0, n) ret[i] = '.';
    	rep(i, 0, n) if (s[i] == 'L') {
    		rep(j, last + 1, i + 1) ret[j] = 'L';
    	} else if (s[i] == 'R') {
    		if (pr[i + 1] < pl[i + 1]) {
      			rep(j, i, pr[i + 1]) ret[j] = 'R';
        			i = pr[i + 1] - 1;
    		} else if (pl[i + 1] < pr[i + 1]) {
    			int l = i, r = pl[i + 1];
    			while (l < r) {
    				ret[l++] = 'R';
    				ret[r--] = 'L';
    			}
    			last = pl[i + 1];
    			i = pl[i + 1];
    		} else {
    			rep(j, i, n) ret[j] = 'R';
    			break;
    		}
    	}
    	return ret;
    }
};
