class Solution {
public:
    string pushDominoes(string dominoes) {
        vector<int> lhs(dominoes.length()), rhs(dominoes.length());
        string ans = dominoes;
        for (int i = 0; i < lhs.size(); i++) {
            lhs[i] = dominoes[i] != '.' ? i : i ? lhs[i - 1] : -1;
        }
        for (int i = int(rhs.size()) - 1; i >= 0; i--) {
            rhs[i] = dominoes[i] != '.' ? i : i + 1 == rhs.size() ? -1 : rhs[i + 1];
        }
        // for (int i = 0; i < dominoes.size(); i++) {
        //     cout << lhs[i] << " ";
        // }
        // cout << endl;
        // for (int i = 0; i < dominoes.size(); i++) {
        //     cout << rhs[i] << " ";
        // }
        // cout << endl;
        for (int i = 0; i < ans.length(); i++) {
            int l = lhs[i], r = rhs[i];
            // cout << i << " " << dominoes[r] - i << " " << i - dominoes[l] << endl;
            if (l != -1 && dominoes[l] == 'R' && (r == -1 || dominoes[r] != 'L' || r - i > i - l)) {
                ans[i] = 'R';
            } else if (r != -1 && dominoes[r] == 'L' && (l == -1 || i - l > r - i || dominoes[l] != 'R')) {
                ans[i] = 'L';
            } else {
                ans[i] = '.';
            }
        }
        return ans;
    }
};
