class Solution {
public:
    string pushDominoes(string dominoes) {
                int N = dominoes.length();
        string t = "";
        while (t != dominoes)
        {
            t = dominoes;
            if (N > 1)
            {
                if (t[0] == '.' && t[1] == 'L')
                {
                    dominoes[0] = 'L';
                }
                if (t[N - 1] == '.' && t[N - 2] == 'R')
                {
                    dominoes[N - 1] = 'R';
                }
            }
            for (int i = 1; i < N - 1; i++)
            {
                if (t[i] == '.')
                {
                    if (t[i - 1] == 'R')
                    {
                        if (t[i + 1] == 'L')
                        {

                        }
                        else
                        {
                            dominoes[i] = 'R';
                        }
                    }
                    else
                    {
                        if (t[i + 1] == 'L')
                        {
                            dominoes[i] = 'L';
                        }
                    }
                }
            }
        }

        return dominoes;
    }
};
