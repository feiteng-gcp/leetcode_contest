class Solution {
public:
    string pushDominoes(string dominoes) {
        bool change = true;
        int n = dominoes.size();
        string S = dominoes;
        while(change) {
            change = false;
            string S1 = S;
            for(int i=0;i<n;++i){
                if(i == 0 && S[i] == '.'&&S[i+1] == 'L'){
                    S1[i] = 'L';
                    change = true;
                } else if(i==n-1 && S[i] == '.' && S[i-1] == 'R'){
                    S1[i] = 'R';
                    change = true;
                } else if(S[i] == '.') {
                    if(S[i-1] == 'R' && S[i+1] == 'L'){
                        //nochange
                    } else if(S[i-1] == 'R'){
                        S1[i] = 'R';
                        change = true;
                    } else if(S[i+1] == 'L'){
                        S1[i] = 'L';
                        change = true;
                    }
                }
            }
            S = S1;
        }
        return S;
        
    }
};
