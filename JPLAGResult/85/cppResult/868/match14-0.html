class Solution {
public:
    struct Data{
        //char c;
        int k;
        int index;
        Data( int i, int kk): index(i), k(kk){}
        //Data(char cc, int i): c(cc) , index(i){}
        Data(){}
    };
    //#define pii pair<int,int>
    //#define mp ma
    string pushDominoes(string s) {
        int n = s.size();
        string result = s;
        vector<int> d(n , -1);
        queue<Data> Q;
        for( int i = 0 ; i < n ; ++i ){
            if( s[i] == '.' )continue;
            d[i] = 0;
            Q.push(Data(i,0));
        }
        
        while( !Q.empty() ){
            Data cur = Q.front(); Q.pop();
            int index = cur.index, dist = cur.k;
            if( s[index] == 'R' ){
                if( index + 1 < n ){
                    if( s[index + 1] == '.' ){
                        s[ index + 1 ] = 'R';
                        d[ index + 1 ] = dist + 1;
                        Q.push( Data( index + 1, dist + 1) );
                    }else if( s[index + 1] == 'L' && d[index + 1] == dist + 1){
                        s[ index + 1 ] = '.';
                        d[ index + 1 ] = 1<<30;
                    }
                }
            }else if( s[index] == 'L' ){
                if( index - 1 >= 0 ){
                    if( s[index - 1] == '.' ){
                        s[ index - 1 ] = 'L';
                        d[ index - 1 ] = dist + 1;
                        Q.push( Data( index - 1, dist + 1) );
                    }else if( s[index - 1] == 'R' && d[index - 1] == dist + 1){
                        s[ index - 1 ] = '.';
                        d[ index - 1 ] = 1<<30;
                    }
                }
            }
            
        }
        
        return s;
    }
};
