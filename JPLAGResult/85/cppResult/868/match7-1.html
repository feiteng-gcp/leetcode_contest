class Solution {
 public:
  string pushDominoes(string dominoes) {
    int idx   = 0;
    int total = dominoes.size();

    while (idx < total) {
      while (idx < total && dominoes[idx] == '.') {
        idx++;
      }

      if (idx < total) {

        if (dominoes[idx] == 'L') {
          int r = idx - 1;
          while (r >= 0 && dominoes[r] == '.') {
            dominoes[r] = 'L';
            r--;
          }
          idx++;
        } else if (dominoes[idx] == 'R') {
          int l = idx + 1;
          while (l < total && dominoes[l] == '.') {
            l++;
          }

          if (l < total) {

            if (dominoes[l] == 'R') {
              for (int i = idx; i <= l; i++) {
                dominoes[i] = 'R';
              }
              idx = l;
            } else if (dominoes[l] == 'L') {
              int r = idx + 1;
              int t = l - 1;
              while (r < t) {
                dominoes[r] = 'R';
                dominoes[t] = 'L';

                r++;
                t--;
              }

              idx = l + 1;
            }
          } else {
            for (int i = idx; i < total; i++) {
              dominoes[i] = 'R';
            }
            break;
          }
        }
      }
    }

    return dominoes;
  }
};
