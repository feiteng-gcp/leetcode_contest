class Solution {
public:
    string pushDominoes(string dominoes) {
        string ans=dominoes;
        
        bool stateChanged=true;
        while (stateChanged) {
            dominoes=ans;
            
            stateChanged=false;
            for (int i=0; i<dominoes.size(); ++i) {
                //cout << i << endl;
                if (dominoes[i]=='.') {
                    bool left=false;
                    bool right=false;
                    if (i>0 && dominoes[i-1]=='R') {
                        left=true;
                    }
                    if (i<(dominoes.size()-1) && dominoes[i+1]=='L') {
                        right=true;
                    }
                    if (left && (!right)) {
                        stateChanged=true;
                        ans[i]='R'; // push from left only
                        //cout << "R" << endl;
                    }
                    if ((!left) && right) {
                        stateChanged=true;
                        ans[i]='L'; // push from right only
                        //cout << "L" << endl;
                    }
                    // push from both side or no push, do nothing
                }
            }
        }
        return ans;
    }
};
