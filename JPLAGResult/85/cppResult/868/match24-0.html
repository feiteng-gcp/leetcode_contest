class Solution {
public:
    string pushDominoes(string dominoes) {
        int dlen = dominoes.size();
        if (dlen <= 1) return dominoes; // no change;
        string curr = dominoes;
        string next = "";
        while (true) {
            //cout << "==" << endl;
            //cout << "curr = " << curr << endl;
            next = curr;            
            for (int i = 0; i < dlen; i++) {
                if (curr[i] == '.') {
                    char l = (i == 0) ? '.' : curr[i - 1];
                    char r = (i == dlen - 1) ? '.' : curr[i + 1];
                    if (l == 'L') {
                        if (r == 'L') {
                            next[i] = 'L'; continue;
                        }
                    } else if (l == 'R'){
                        if (r == 'L') {
                            continue;
                        } else {
                            next[i] = 'R';
                        }
                    } else {
                        if (r == 'L') {
                            next[i] = 'L';
                        }
                    }
                    
                }
            }
            //cout << "next = " <<  next << endl;
            if (next == curr) {
                break;
            } else {
                curr = next;
            }
           
        }
        return next;
    }
};
