class Solution {
public:
    string pushDominoes(string dominoes) {
        int n = dominoes.size();
        vector<double> vec(n, 0);
        for(int i = 0; i < n; ++i){
            if(dominoes[i] == 'L'){
                vec[i] = -1;
                double force = -0.5;
                int p = i - 1;
                while(p >= 0 && dominoes[p] == '.') {vec[p--] += force; force = force/2;}
            }else if(dominoes[i] == 'R'){
                vec[i] = 1;
                double force = 0.5;
                int p = i + 1;
                while(p < n && dominoes[p] == '.') {vec[p++] += force; force = force/2;}
            }
        }
        for(int i = 0; i < n; ++i){
            if(dominoes[i] != '.') continue;

            if(vec[i] > 0){
                dominoes[i] = 'R';
            }else if(vec[i] < 0 ){
                dominoes[i] = 'L';
            }else{
                dominoes[i] = '.';
            }
        }
        return dominoes;
    }
};
