class Solution {
public:
    string pushDominoes(string s) {
        int n = s.size();
        string ret(n, '.');
        for (int i = 0, j; i < n; i = j) {
            for (; i < n && s[i] != '.'; ++i) ret[i] = s[i];
            if (i == n) break;
            for (j = i; j < n && s[j] == '.'; ++j);
//            cout << i << " " << j << " " << s[i - 1] << " " << s[j] << endl;
            if (i - 1 >= 0 && j < n) {
                if (s[i - 1] == s[j]) {
                    for (int k = i; k < j; ++k) ret[k] = s[i - 1];
                } else if (s[i - 1] == 'R' && s[j] == 'L') {
                    int len = j - i;
                    for (int k = 0; k < len / 2; ++k) {
                        ret[i + k] = s[i - 1];
                        ret[j - 1 - k] = s[j];
                    }
                }
            } else if (i - 1 >= 0) {
                if (s[i - 1] == 'R') {
                    for (int k = i; k < j; ++k) ret[k] = s[i - 1];
                }
            } else {
                if (s[j] == 'L') {
                    for (int k = i; k < j; ++k) ret[k] = s[j];
                }
            }
        }
        return ret;
        
    }
};
