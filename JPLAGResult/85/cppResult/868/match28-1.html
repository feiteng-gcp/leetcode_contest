class Solution {
public:
    string pushDominoes(string dominoes) {
        int len = dominoes.size(), i = 0;
        while(i < len){
            while(i < len && dominoes[i] == '.') i++;
            if(i == len) return dominoes;
            
            if(dominoes[i] == 'L'){
                int left = i - 1;
                while(left >= 0 && dominoes[left] == '.'){
                    dominoes[left] = 'L';
                    left--;
                }
                i++;
            }else if(dominoes[i] == 'R'){
                int right = i + 1;
                while(right < len && dominoes[right] == '.') right++;
                if(right == len || dominoes[right] == 'R'){
                    for(int a = i + 1; a < right; a++){
                        dominoes[a] = 'R';
                    }

                    i = right;
                }else if(dominoes[right] == 'L'){
                    int a = i + 1, b = right - 1;
                    while(a < b){
                        dominoes[a++] = 'R';
                        dominoes[b--] = 'L';
                    }
                    i = right + 1;
                }
            }
        }
        return dominoes;
    }
};
