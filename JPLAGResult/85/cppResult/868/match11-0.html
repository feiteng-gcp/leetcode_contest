class Solution {
public:
    string pushDominoes(string dominoes) {
        int f=-1;
        string asc="";
        int a=0;
        for(int i=0;i<dominoes.size();i++){
            if(f==-1&&dominoes[i]=='.'){
                asc+=".";
            }
            else if(f==-1&&dominoes[i]=='L'){
                for(int j=asc.size()-1;j>=a;j--){
                    asc[j]='L';
                }
                asc+="L";
            }
            else if(f==-1&&dominoes[i]=='R'){
                asc+=".";
                f=1;
                a=i;
            }
            else if(f==1&&dominoes[i]=='R'){
                for(int j=a;j<i;j++){
                    asc[j]='R';
                }
                asc+=".";
                f=1;
                a=i;
            }
            else if(f==1&&dominoes[i]=='.'){
                asc+=".";
            }
            else if(f==1&&dominoes[i]=='L'){
                asc+='.';
                int j1=a;
                int j2=i;
                for(;j1<j2;j1++,j2--){
                    asc[j1]='R';
                    asc[j2]='L';
                }
                f=-1;
                a=i+1;
            }
        }
        if(f==1){
            for(int j=a;j<asc.size();j++){
                asc[j]='R';
            }
        }
        return asc;
    }
};
