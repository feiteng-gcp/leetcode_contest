class Solution {
public:
    string pushDominoes(string dominoes) {
        string &d = dominoes;
        int len = dominoes.length();
        vector<int> vl(len, 0), vr(len, 0);
        string sl(len, '.'), sr(len, '.');
        int stat = 0;
        for(int i = 0; i < len; i++) {
            if(d[i] == '.') {
                if(stat != 0) {
                    vr[i] = stat;
                    stat++;
                } else {
                    vr[i] = 999999;
                }
            } else if(d[i] == 'L') {
                stat = 0;
                vr[i] = 999999;
            } else {
                stat = 1;
            }
        }
        stat = 0;
        for(int i = len-1; i >= 0; i--) {
            if(d[i] == '.') {
                if(stat != 0) {
                    vl[i] = stat;
                    stat++;
                } else {
                    vl[i] = 999999;
                }
            } else if(d[i] == 'R') {
                stat = 0;
                vl[i] = 999999;
            } else {
                stat = 1;
            }
        }
        for(int i = 0; i < len; i++) {
           // cout << vl[i] << ' ' << vr[i] << endl;
            if(vl[i] == 0 && vr[i] == 0) {
                // do nothing

            } else if (vl[i] < vr[i]) {
                d[i] = 'L';
            } else if (vl[i] > vr[i]) {
                d[i] = 'R';
            } else {
                d[i] = '.';
            }
        }
        return d;
    }
};
