class Solution {
public:
    string pushDominoes(string dominoes) {
        int prevdir;
        int prevpos = -1;
        string result = dominoes;
        int i = 0;
        while (dominoes[i] == '.' && i < dominoes.size()) {
            i++;
        }
        if (i == dominoes.size()) {
            return result;
        }
        if (dominoes[i] == 'L') {
            for (int j = 0; j < i; j++) {
                result[j] = 'L';
            }
            prevdir = 0;
        }
        else {
            prevdir = 1;
        }
        prevpos = i;
        i++;
        for (; i < dominoes.size(); i++) {
            if (dominoes[i] == 'L') {
                if (prevdir == 0) {
                    for (int j = prevpos + 1; j < i; j++) {
                        result[j] = 'L';
                    }
                }

                else {
                    int j = i-1;
                    prevpos++;
                    while (prevpos < j) {
                        //cout << "j" << j << " " << "prevpos " << prevpos << endl; 
                        result[prevpos++] = 'R';
                        result[j--] = 'L';
                    }
                }
                prevdir = 0;
                prevpos = i;
            }

            else if (dominoes[i] == 'R') {
                if (prevdir) {
                    for (int j = prevpos + 1; j < i; j++) {
                        result[j] = 'R';
                    }
                }
                prevdir = 1;
                prevpos = i;
            }
        }
        if (prevdir == 1) {
            prevpos++;
            while(prevpos < dominoes.size()) {
                result[prevpos++] = 'R';
            }
        }
        //cout << "finished \n";
        return result;
    }
};
