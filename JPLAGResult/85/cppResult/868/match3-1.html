class Solution {
public:
    string pushDominoes(string dominoes) {
        int n = dominoes.size();
        vector<int> vec(n, n+1);
        

        int count = n+1;
        for (int i = 0; i < n; i++) {
            if (dominoes[i] == 'R') {
                count = 0;
                vec[i] = count;
            }
            else if (dominoes[i] == 'L') {
                count = n + 1;
                vec[i] = count;
            }
            else {
                vec[i] = ++count;
            }
        }
        
        count = n + 1;
        for (int i = n - 1; i >= 0; i--) {
            if (dominoes[i] == 'L') {
                count = 0;
            }
            else if (dominoes[i] == 'R') count = n+1;
            else count++;
            
            if (count == vec[i]) vec[i] = 0;
            else if (count < min(vec[i], n)) vec[i] = -1;
        }
        
        for (int i = 0; i < n; i++) {
            if (vec[i] < 0) dominoes[i] = 'L';
            else if (vec[i] > 0 && vec[i] < n) dominoes[i] = 'R';
        }
        
        return dominoes;
    }
};
