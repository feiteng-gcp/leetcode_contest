class Solution {
public:
    string pushDominoes(string s) {
        for (int i = 0, j = 0; j <= s.size(); ++j) {
            if (j == s.size() || s[j] != '.') {
                if (i < j) {
                    if (!i) {
                        if (j == s.size())
                            return s;
                        if (s[j] == 'L') {
                            for (int k = i; k < j; ++k)
                                s[k] = 'L';

                        }
                    } else if (j == s.size()) {
                        if (s[i - 1] == 'R') {
                            for (int k = i; k < j; ++k)
                                s[k] = 'R';
                        }
                    } else {
                        if (s[i - 1] == 'L' && s[j] == 'R') {
                        } else if (s[i - 1] == 'R' && s[j] == 'L') {
                            for (int k = i, t = j - 1; k < t; ++k, --t)
                                s[k] = 'R', s[t] = 'L';
                        } else {
                            for (int k = i; k < j; ++k)
                                s[k] = s[i - 1];
                        }
                    }
                }
                i = j + 1;
            }
        }
        return s;
    }
};
