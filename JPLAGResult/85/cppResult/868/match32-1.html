class Solution {
public:
    string pushDominoes(string dominoes) 
    {
        if (dominoes.size()==1) return dominoes;
        
        int N=dominoes.size();
        while(1)
        {
            string temp=dominoes;
            for (int i=0; i<N; i++)
            {
                if (dominoes[i]!='.')
                    continue;
                
                if (i==0)
                {
                    if (dominoes[i+1]=='L')
                        temp[i]='L';                    
                }
                else if (i==N-1)
                {

                    if (dominoes[i-1]=='R')
                        temp[i]='R';
                }
                else
                {                    
                    if (dominoes[i-1]=='R' && dominoes[i+1]!='L')
                        temp[i]='R';
                    else if (dominoes[i+1]=='L' && dominoes[i-1]!='R')
                        temp[i]='L';
                }
            }
            
            if (temp==dominoes) break;
            dominoes = temp;
        }
        
        return dominoes;
    }
};
