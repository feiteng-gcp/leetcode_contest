class Solution {
public:
    string pushDominoes(string d) {
        int stop = false;
        while(!stop) {
            bool changed = false;
            string temp = d;
            for(int i = 0;i < d.size();i++) {
                if(d[i] == 'L' || d[i] == 'R') {
                    continue;
                }
                if(i == 0) {
                    if(d[i + 1] == 'L') {
                        temp[i] = 'L';
                        changed = true;
                    }
                }
                else if(i == d.size() - 1) {
                    if(d[i - 1] == 'R') {
                        temp[i] = 'R';
                        changed = true;
                    }
                }
                else {
                    if(d[i - 1] == 'R' && d[i + 1] == 'L') {
                        
                    }  

                    else if(d[i + 1] == 'L') {
                        temp[i] = 'L';
                        changed = true;
                    }
                    else if(d[i - 1] == 'R') {
                        temp[i] = 'R';
                        changed = true;
                    }
                }
            }
            d = temp;
            stop = !changed;
        }
        return d;
    }
};
