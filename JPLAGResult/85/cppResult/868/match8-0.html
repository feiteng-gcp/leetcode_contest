class Solution {
public:
    string pushDominoes(string dominoes) {
        int first = 0;
        int second = 0;
        int length = dominoes.length();
        while (first < length && second < length) {
            while (first < length && dominoes[first] == '.') first++;
            second = first+1;
            while (second < length && dominoes[second] == '.') second++;
            // cout << first << endl;
            if (first < length) {
                if (dominoes[first] == 'L') {
                    int i = first-1;
                    while (i >= 0 && dominoes[i] == '.') {
                        dominoes[i] = 'L';
                        i--;
                    }
                    first = second;
                } else if (dominoes[first] == 'R') {
                    if (second < length) {
                        if (dominoes[second] == 'L') {
                            int i = first+1;
                            int j = second-1;
                            while (i < j) {
                                dominoes[i++] = 'R';
                                dominoes[j--] = 'L';
                            }
                            first = second+1;
                        } else {
                            int i = first + 1;
                            while (i < length && dominoes[i] == '.') {
                                dominoes[i] = 'R';
                                i++;
                            }
                            first = i;                            
                        }
                    } else {
                        int i = first + 1;
                        while (i < length && dominoes[i] == '.') {
                            dominoes[i] = 'R';
                            i++;
                        }
                        first = i;
                    }
                }
            } 
            // cout << first << endl;
        }
        return dominoes;
    }
};
