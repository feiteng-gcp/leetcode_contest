class Solution {
public:
    string pushDominoes(string dominoes) {
        int sz = dominoes.size();
        int i=0, j = 0;
        bool go_right = false;
        while(i < sz){
            while(j < sz && dominoes[j] == '.'){
                j++;
            }
            if(j < sz){
                if(dominoes[j] == 'L'){
                    if(go_right == true){
                        int l = i, r = j;
                        while(l < r){
                            dominoes[l++] = 'R';
                            dominoes[r--] = 'L';
                        }
                    }else{
                        for(int k = i; k<j; k++){
                            dominoes[k] = 'L';
                        }
                    }
                    j++;
                    i = j;
                    go_right = false;
                }else{
                    if(go_right == true){
                        for(int k = i; k<j; k++){
                            dominoes[k] = 'R';
                        }
                    }
                    go_right = true;
                    i = j;
                    j++;
                }
            }else{
                if(go_right == true){
                    while(i < sz){
                        dominoes[i++] = 'R';
                    }
                }else{
                    i = sz;
                    break;
                }
            }
        }
        return dominoes;
    }
};
