class Solution {
public:
    string pushDominoes(string dominoes) {
        int index=0;
        char before='.';
        char after='.';
        while(index<dominoes.size()){
            if(dominoes[index]!='.'){
                before=dominoes[index++];
                continue;                
            }else {
                int pos=index;
                while(index<dominoes.size() && dominoes[index]=='.'){
                    index++;
                }
                if(index>=dominoes.size()){
                    after='.';                   
                }else {
                    after=dominoes[index];
                }
                if(before=='.'||before=='L'){
                    if(after=='L'){
                        while(pos<index){
                            dominoes[pos++]='L';
                        }
                    }
                }else {
                    if(after=='.'||after=='R'){
                        while(pos<index){
                            dominoes[pos++]='R';
                        }
                    }else{
                        int key=index-1;
                        while(pos<key){
                            dominoes[pos++]='R';
                            dominoes[key--]='L';
                        }
                    }
                }
            }
            
        }
        return dominoes;
    }
};
