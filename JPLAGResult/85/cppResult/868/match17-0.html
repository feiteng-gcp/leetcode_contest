class Solution {
public:
    string pushDominoes(string d) {
        int n=d.length();
        vector<int> l(n,0),r(n,0);
        int tmp=0;
        for(int i=0;i<n;i++)
        {
            if(d[i]=='.'&&tmp)
                r[i]=++tmp;
            if(d[i]=='L') tmp=0;
            if(d[i]=='R') tmp=1;
        }
        tmp=0;
        for(int i=n-1;i>=0;i--)
        {
            if(d[i]=='.'&&tmp)
                l[i]=++tmp;
            if(d[i]=='L') tmp=1;
            if(d[i]=='R') tmp=0;
        }
        for(int i=0;i<n;i++)
        {
            if(d[i]=='.')
            {
                if(l[i]&&r[i])
                {
                    if(l[i]>r[i]) d[i]='R';
                    else if(l[i]<r[i]) d[i]='L';
                }
                else
                {
                    if(l[i]) d[i]='L';
                    else if(r[i]) d[i]='R';
                }
            }
        }
        return d;
    }
};
