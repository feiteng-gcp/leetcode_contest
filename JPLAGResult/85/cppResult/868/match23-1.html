class Solution {
public:
    string pushDominoes(string dominoes) {
        bool done;
        do{
            done = true;
            for (int i = 0; i < dominoes.length(); i++) {
                switch (dominoes[i]) {
                    case 'L':
                        if (i > 0 && dominoes[i - 1] == '.') {
                            dominoes[i - 1] = 'L';
                        }
                        dominoes[i] = 'l';
                        done = false;
                        break;
                    case 'R':
                        int j = i;

                        if (i < dominoes.length() - 1 && dominoes[i + 1] == '.') {
                            if (i + 1 == dominoes.length() - 1) {
                                dominoes[i + 1] = 'R';
                                i += 1;
                            } else if (dominoes[i + 2] == 'L') {
                                dominoes[i + 2] = 'l'; 
                                i += 2;
                            } else {
                                dominoes[i + 1] = 'R'; 
                                i += 1;
                            }
                        }
                        dominoes[j] = 'r';
                        done = false;
                        break;
                }
            }
            //puts(dominoes.c_str());
        } while (!done);
        for (int i = 0; i < dominoes.length(); i++)
            if (dominoes[i] == 'l')
                dominoes[i] = 'L';
            else if (dominoes[i] == 'r')
                dominoes[i] = 'R';
        return dominoes;
    }
};
