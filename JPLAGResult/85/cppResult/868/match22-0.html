class Solution {
public:
    string pushDominoes(string&dominoes) {
        bool change=true;
        int len=dominoes.size();
        vector<int>mark(len,0);
        while(change){
            change=false;
            for(int i=0;i<len;i++){
                if(dominoes[i]=='.'){
                    if((i+1<len&&dominoes[i+1]=='L')&&(i-1<0||dominoes[i-1]!='R')){mark[i]=-1;change=true;}
                    else if((i-1>=0&&dominoes[i-1]=='R')&&(i+1>=len||dominoes[i+1]!='L')){mark[i]=1;change=true;}
                }
            }
            for(int i=0;i<len;i++){
                if(mark[i]==-1){dominoes[i]='L';mark[i]=0;}
                if(mark[i]==1){dominoes[i]='R';mark[i]=0;}
            }
        }
        return dominoes;
    }
};
