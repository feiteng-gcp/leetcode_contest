const ll MOD=1e9+7;

class Solution {
public:
    double new21Game(int N, int K, int W) {
        vector<double> p(K+W+1);
        double s=0.0;
        for(int i=K;i<K+W;i++) p[i]=1.0*(i<=N),s+=p[i];
        for(int i=K-1;i>=0;i--){
            p[i]=s/W;
            s+=p[i]-p[i+W];
            // cout<<i<<' '<<p[i]<<'\n';
        }
        return p[0];
    }
};
