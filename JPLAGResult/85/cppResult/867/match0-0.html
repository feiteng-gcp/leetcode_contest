class Solution {
public:
    double new21Game(int N, int K, int W) {
        vector<double> p(1, 1);
        double a = 0;

        for (int i = 0; i < K + W; ++i) {
            double tmp = 0;
            if (i < K)      tmp += p.back();
            if (i - W >= 0) tmp -= p[i - W];
            p.push_back(a + tmp / W);
            a = p.back();
            // cout << i << " " << a << " " << p.back() << endl;
        }

        double sol = 1;
        for (int i = N + 1; i < p.size(); ++i) {
            sol -= p[i];
        }
        // cout << sol << endl;
        return sol;
    }
};
