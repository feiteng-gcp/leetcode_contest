        :type dominoes: str
        :rtype: str

        count = -1
        for i in range(len(dominoes)):
            if dominoes[i] == 'R':
                count = 0
            if dominoes[i] == 'L':
                count = -1
            if dominoes[i] == '.' and count >= 0:
                count += 1
                result[i] = count
        count = -1

            if dominoes[i] == 'L':

                count = -1
