class Solution:
    def pushDominoes(self, dominoes):
        """
        :type dominoes: str
        :rtype: str
        """
        dp1 = [0] * len(dominoes)
        dp2 = [0] * len(dominoes)
        fall = False
        for i in range(len(dominoes)):
            if dominoes[i] == '.' and fall:
                dp1[i] = dp1[i-1] +1
            if dominoes[i] == 'R':
                dp1[i] = 1
                fall = True
            if dominoes[i] == 'L':
                fall = False
        fall = False
        for i in range(len(dominoes)-1, -1, -1):
            if dominoes[i] == '.' and fall:
                dp2[i] = dp2[i+1] +1 
            if dominoes[i] == 'L':
                dp2[i] = 1
                fall = True
            if dominoes[i] == 'R':
                fall = False
        res = ""
        for i in range(len(dominoes)):
            if dp1[i] == 0 and dp2[i] == 0:
                res += '.'
            elif dp1[i] == 0 and dp2[i] != 0:
                res += 'L'
            elif dp1[i] != 0 and dp2[i] == 0:
                res += 'R'
            elif dp1[i] > dp2[i]:
                res += 'L'
            elif dp1[i] < dp2[i]:
                res += 'R'
            else:
                res += '.'
        return res
