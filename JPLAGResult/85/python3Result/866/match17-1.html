class Solution:
    def isRectangleOverlap(self, rec1, rec2):
        """
        :type rec1: List[int]
        :type rec2: List[int]
        :rtype: bool
        """

        def overlap(rec1, rec2):
            lx 
= max(rec1[0], rec2[0])
            ly = max(rec1[1], rec2[1])

            rx = min(rec1[2], rec2[2])
            ry = min(rec1[3], rec2[3])



            return lx < rx and ly < ry
        # print(overlap(rec2, rec1))

        return overlap(rec1, rec2)
