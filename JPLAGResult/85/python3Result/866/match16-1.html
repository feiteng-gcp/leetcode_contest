class Solution:
    def isRectangleOverlap(self, rec1, rec2):
        """
        :type rec1: List[int]
        :type rec2: List[int]
        :rtype: bool
        """
        if rec1[0] > rec2[0]:
            rec1, rec2 = rec2, rec1
        if rec1
[2] > rec2[0]:
            if rec1[1] >= rec2[1] and rec1[1] < rec2[3]:
                return True 
            if rec1[1] <= rec2[1] and rec1[3] > rec2[1]:
                return True
        return False
