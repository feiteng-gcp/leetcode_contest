class Solution:
    def isRectangleOverlap(self, rec1, rec2):
        """
        :type rec1: List[int]
        :type rec2: List[int]
        :rtype: bool
        """
        x1 = sorted([rec1[0], rec1[2]])
        y1 = sorted([rec1[1], rec1[3]])
        
        x2 = sorted([rec2[0], rec2[2]])
        y2 = sorted([rec2[1], rec2[3]])
        
        if x2[0] >= x1[1] or y2[0] >= y1[1]:
            return False
        if x2[1] <= x1[0] or y2[1] <= y1[0]:
            return False
        
        return True
