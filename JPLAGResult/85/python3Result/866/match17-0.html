class Solution:
    def isRectangleOverlap(self, rec1, rec2):
        """
        :type rec1: List[int]
        :type rec2: List[int]
        :rtype: bool
        """
        center1 = [(rec1[0]+rec1[2])/2,(rec1[1]+rec1[3])/2]
        center2 = [(rec2[0]+rec2[2])/2,(rec2[1]+rec2[3])/2]
        
        w1 = abs(rec1[0]-rec1[2])
        h1 = abs(rec1[1]-rec1[3])
        
        w2 = abs(rec2[0]-rec2[2])
        h2 = abs(rec2[1]-rec2[3])
        
        if abs(center1[0]-center2[0])< w2/2+w1/2 and abs(center1[1]-center2[1])< h2/2+h1/2:
            return True
        else:
            return False
