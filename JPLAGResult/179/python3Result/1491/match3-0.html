class Solution:
    def numTimesAllBlue(self, a: List[int]) -> int:
        # check number of times the prefix is filled up.
        
        # n == max(idx)
        
        ret = 0
        hi = -1
        
        for i,x in enumerate(a):
            hi = max(hi, x)
            if hi == i+1:
                ret += 1
                
        return ret
