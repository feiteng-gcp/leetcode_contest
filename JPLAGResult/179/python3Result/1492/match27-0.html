class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        subord = collections.defaultdict(list)
        for i in range(n):
            if i != headID:
                subord[manager[i]].append(i)
        
        ans = 0
        queue = [(headID, 0)]
        while queue:
            now, time = queue.pop(0)
            newtime = time + informTime[now]
            ans = max(ans, newtime)
            for i in subord[now]:
                queue.append((i, newtime))
        return ans
