class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        r = collections.defaultdict(list)
        for i, j in enumerate(manager):
            r[j].append(i)
        b = [(0, headID)]
        ret = 0
        while b:
            time, staff = heapq.heappop(b)
            ret = max(ret, time)
            time += informTime[staff]
            for i in r[staff]:
                heapq.heappush(b, (time, i))
        return ret
