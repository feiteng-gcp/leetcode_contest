class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], it: List[int]) -> int:
        g = [[] for _ in range(n)]
        for i, m in enumerate(manager):
            if m != -1:
                g[m].append(i)
        def dfs(r):
            if it[r] == 0:
                return 0
            ti = max(dfs(ch) for ch in g[r])
            ti += it[r]
            return ti
        return dfs(headID)
