from collections import defaultdict

class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        tree = defaultdict(list)
        for v, p in enumerate(manager):
            if p != -1:
                tree[p].append(v)
                
        def get_max_time(v: int, p: int):
            if not tree[v]:
                return 0
            max_time = 0
            for u in tree[v]:
                max_time = max(max_time, get_max_time(u, v))
            return max_time + informTime[v]
        return get_max_time(headID, -1)
