import heapq
import collections
class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        heap=[]
        cnt=1
        time=0
        child
=collections.defaultdict(list)
        for i, p in enumerate(manager):
            if p==-1:
                continue
            child[p].append(i)
            
        heap=[
(0,headID)]
        while heap:
            newtime, curr = heapq.heappop(heap)
            time=max(time,newtime)
            for ch in child[curr]:
                heapq.heappush(heap, (newtime+informTime[curr], ch))
        return time
