class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        d = collections.defaultdict(list)
        for i, c in enumerate(manager):
            d[c].append(i)
        q = [(headID, 0)]

        t = 0
        while q:
            h, st = q.pop(0)
            t = max(t, st)
            for c in d[h]:
                q.append((c, st + informTime[h]))
        return t
