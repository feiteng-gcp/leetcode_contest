class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        mp 
= collections.defaultdict(list)
        for i, m in enumerate(manager):
            if m == -1:
                continue
            mp[m].append(i)
        queue = [(headID, 0)]
        res = 0
        
while queue:
            i, t = queue.pop()
            res = max(res, t)
            for sub in mp[i]:
                queue.append((sub, t+informTime[i]))
        return res
