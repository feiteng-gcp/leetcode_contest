class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        ans = 0
        graph = collections.defaultdict(list)
        
        for sub, ma in enumerate(manager):
            if ma != -1:
                graph[ma].append(sub)
        
        queue = [(headID, 0)]
        
        while queue:
            ma, time = queue.pop(0)
            ans = max(ans, time)
            for sub in graph[ma]:
                queue.append((sub, time + informTime[ma]))
        return ans
