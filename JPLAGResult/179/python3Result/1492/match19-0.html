class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        adj = [[] for _ in range(n)]
        for i, x in enumerate(manager):
            if x == -1:
                continue
            adj[x].append(i)
        def dfs(x):
            if len(adj[x]) == 0:
                return 0
            return informTime[x] + max(dfs(y) for y in adj[x])
        return dfs(headID)
