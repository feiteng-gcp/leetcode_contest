class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        # bfs
        D 
= collections.defaultdict(list)
        for i, u in enumerate(manager):
            if u != -1:
                D[u].append(i)
            
        queue = collections.deque([(headID, 0)])
        ans = 0
        while queue:
            for _ in range(len(queue)):
                idx, time 
= queue.popleft()
                ans = max(ans, time)
                for to in D[idx]:
                    queue.append((to, time+informTime[idx]))
        return ans
