class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        
        children = [[] for _ in range(n)]
        for i, x in enumerate(manager):
            if (x == -1): continue
            children
[x].append(i)
        
        def r(i):
            if len(children[i]) == 0:
                return 0
            return max(r(x) for x in children[i]) + informTime[i]
        
        return r(headID)
