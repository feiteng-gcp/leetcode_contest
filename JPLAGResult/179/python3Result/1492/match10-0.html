from collections import defaultdict
class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int],
                     informTime: List[int]) -> int:
        # DFS
        xiashu = defaultdict(list)
        for i, m in enumerate(manager):
            if m != -1:
                xiashu[m].append(i)
        self.res = 0

        def dfs(node, cnt):
            if node not in xiashu:
                self.res = max(self.res, cnt)
                return
            for x in xiashu[node]:
                dfs(x, cnt + informTime[node])

        dfs(headID, 0)
        return self.res
