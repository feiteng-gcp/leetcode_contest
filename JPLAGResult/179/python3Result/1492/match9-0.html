class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        tree = [[] for i in range(n)]
        for i, m in enumerate(manager):
            if m > -1:
                tree[m].append(i)
                
        def DFS(u):
            time = 0
            for v in tree[u]:
                time1 = DFS(v)
                time = max(time,time1)
            if tree[u]:
                time += informTime[u]
            return time
        
        result = DFS(headID)
        return result
