class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        mydict = collections.defaultdict(list)
        for idx, val in enumerate(manager):
            if idx == headID:
                continue
            mydict[val].append(idx)
        ans = 0
        myq = collections.deque()
        myq.append([headID, informTime[headID]])
        while myq:
            cur, time = myq.popleft()
            ans = max(time, ans)
            for nxt in mydict[cur]:
                myq.append([nxt, time + informTime[nxt]])
        return ans
