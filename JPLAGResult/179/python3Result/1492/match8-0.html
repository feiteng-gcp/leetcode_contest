class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        
        subord = [[] for _ in range(n)]
        for i, mi in enumerate(manager):
            if mi != -1:
                subord[mi].append(i)

        def solve(idx, t):
            res = t
            for s in subord[idx]:
                res = max(res, solve(s, t + informTime[idx]))
        
            return res
        
        return solve(headID, 0)
