from collections import defaultdict

class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        subs = defaultdict
(list)
        
        for i, m in enumerate(manager):
            if i == headID:
                continue
                
            subs[m].append(i)
        
        times = [0] * n
        
        res = 0
        bfs = [(headID, 0)]
        while bfs:
            i, t = bfs.pop(0)
            times[i] = t
            res 
= max(res, t)
            
            for u in subs[i]:
                bfs.append((u, t+informTime[i]))
        
        return res
