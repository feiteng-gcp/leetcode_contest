class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        descendents = [[] for _ in range(n)]
        for empID in range(n):
            if manager[empID] == -1:
                pass
            else:
                descendents[manager[empID]].append(empID)
        timeToReceive = [float('inf') for _ in range(n)]
        timeToReceive[headID] = 0
        stack 
= [headID]
        while stack:
            empID = stack.pop(0)
            for desc in descendents[empID]:
                timeToReceive[desc] = timeToReceive[empID] + informTime[empID]
                stack.append(desc)
        return max(timeToReceive)
