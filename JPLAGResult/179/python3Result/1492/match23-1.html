class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        g = collections.defaultdict(list)
        for i in range(n):
            if manager[i] == -1:
                continue
            g[manager
[i]].append(i)
        
        # print(g)
        q = [(headID, 0)]
        final = 0
        while q:
            top = q.pop(0)
            head = top[0]
            headTime = top[1]
            final = max
(final, headTime)
            for r in g[head]:
                q.append((r, headTime+informTime[head]))
        return final
