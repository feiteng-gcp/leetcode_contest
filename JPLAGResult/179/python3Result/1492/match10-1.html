from collections import defaultdict
class Solution:
    ans = 0
    
def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        d = defaultdict(list)
        for idx,i in enumerate(manager):
            if i!=-1:
                d[i].append(idx)
        Solution.ans = 0
        def dfs(x,su):
            if len(d[x])==0:
                Solution
.ans=max(Solution.ans,su)
                return
            for i in d[x]:
                dfs(i,su+informTime[x])
        dfs(headID,0)
        return Solution.ans
