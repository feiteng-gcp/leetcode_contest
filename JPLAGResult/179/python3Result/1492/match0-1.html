class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        children = collections.defaultdict(set)
        for index, value in enumerate(manager):
            if value == -1:
                continue
            children[value].add(index)
        res = 0
        q = collections.deque()
        q.append((headID, 0))
        
while q:
            i, time = q.popleft()
            res = max(res, time)
            for child in children[i]:
                q.append((child, time + informTime[i]))
        return res
