class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        subs = collections.defaultdict(set)
        for s, m in enumerate(manager):
            if m != -1:
                subs[m].add(s)
        
        self.ans = -1
        def dfs(id_, curtime):
            if not subs[id_]:
                self.ans = max(self.ans, curtime)
            
            for sub in subs[id_]:
                dfs(sub, curtime + informTime[id_])
        
        dfs(headID, 0)
        return self.ans
