class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        children = [[] for _ in range(n)]
        for i in range(n):
            if headID != i:
                children[manager[i]].append(i)
        return self.maxLen(headID, children, informTime)

    def maxLen(self, now, children, informTime):
        res = 0
        for v in children[now]:
            res = max(res, self.maxLen(v, children, informTime) + informTime[now])
        return res
