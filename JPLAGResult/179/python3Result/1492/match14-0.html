class Solution:
    def numOfMinutes(self, n: int, headID: int, manager: List[int], informTime: List[int]) -> int:
        edge = collections.defaultdict(list)
        for i, j in enumerate(manager):
            if j == -1:
                continue
            edge[j].append((i, informTime[j]))
        
        queue = collections.deque([(0, headID)])
        time = 0
        while queue:
            t, m = queue.popleft()
            time = max(time, t)
            for adj, w in edge[m]:
                queue.append((t + w, adj))
        return time
