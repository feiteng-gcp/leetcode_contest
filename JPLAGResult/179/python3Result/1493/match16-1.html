from functools import lru_cache

class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        adj = collections.defaultdict(list)
        for v, w in edges:
            adj[v].append(w)
            adj[w].append(v)
        visited = [False] * (n+1)
        visited[1] = True
        
        @lru_cache(maxsize=None)
        def backtrack(node, remain):
            if remain == 0:
                return node == target
            possible = [nei for nei in adj[node] if not visited[nei]]
            if not possible:
                return node == target
            res = 0
            for nei in possible:
                visited[nei] = True
                res += backtrack(nei, remain-1)
                visited[nei] = False
            return res / len(possible)
        
        return backtrack(1, t)
