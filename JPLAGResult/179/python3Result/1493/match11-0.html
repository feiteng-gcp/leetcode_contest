from collections import defaultdict
class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        graph = defaultdict(list)
        for a, b in edges:
            graph[a].append(b)
            graph[b].append(a)
        dp, seen = {1: 1}, {1}
        for _ in range(t):
            nxt = {}
            for cur, prob in dp.items():
                cands = set(graph[cur]) - seen
                if not cands:
                    nxt[cur] = prob
                else:
                    prob /= len(cands)
                    for cand in cands:
                        seen.add(cand)
                        nxt[cand] = prob
            dp = nxt
        return dp.get(target, 0)
