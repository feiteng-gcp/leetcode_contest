class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        connect = collections.defaultdict(list)
        for i, j in edges:
            connect[i].append(j)
            connect[j].append(i)
        
        queue = [(1, 1, 0)] # (vertex, prob, earliest time)
        visited = {1}
        while queue:
            now, prob, time = queue.pop(0)
            avail = list(filter(lambda x: x not in visited, connect[now]))
            if now == target:
                return prob if time == t or (time < t and not avail) else 0
            
            lena = len(avail)
            for k in avail:
                queue.append((k, prob / lena, time + 1))
                visited.add(k)
