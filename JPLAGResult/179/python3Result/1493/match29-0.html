class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        nei = collections.defaultdict(list)
        for u, v in edges:
            nei[u].append(v)
            nei[v].append(u)
        prob = [0] * (n + 1)
        prob[1] = 1
        act = {1}
        v = set()
        for i in range(t):
            na = set()
            for j in act:
                s = set(k for k in nei[j] if k not in v)
                t = len(s)
                if t > 0:
                    t = prob[j] / t
                    for k in s:
                        prob[k] = t
                        na.add(k)
                    prob[j] = 0
                v.add(j)
            act = na
        return prob[target]
