class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        
        adj = collections.defaultdict(list)
        for start, end in edges:
            adj[start].append(end)
            adj[end].append(start)
        
        odds = [0.0] * (n + 1)  # becomes zero if we never see target
        odds[1] = 1.0
        
        # BFS on all the options on each step
        cur_on = [1]
        seen = set([1])
        while t and cur_on:
            new_opts = []
            for opt in cur_on:
                next_opts = [n for n in adj[opt] if n not in seen]
                if not next_opts:
                    continue  # stuck on this vertex
                
                cur_odds = odds[opt]
                next_odds = cur_odds / len(next_opts)
                for nv in next_opts:
                    new_opts.append(nv)
                    odds[nv] = next_odds
                    seen.add(nv)
                    
                odds[opt] = 0.0  # jumped off current vertex
                    
            cur_on = new_opts
            
            t -= 1
            
        print(odds)
        
        return odds[target]
