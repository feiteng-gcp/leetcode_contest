class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        tree = {i: [] for i in range(1, n+1)}
        for e in edges:
            tree[e[0]].append(e[1])
            tree[e[1]].append(e[0])
        
        visit = [False] * (n + 1)
        q = collections.deque([(1, 0, 1.0)])  # id, depth, prob
        visit[1] = True
        while q:
            v, depth, prob = q.pop()
            children = [i for i in tree[v] if not visit[i]]
            if target == v:
                # leaf node
                if t < depth or (t > depth and children):
                    return 0.0
                return prob
            if children:
                prob = prob / len(children)
                for i in children:
                    visit[i] = True
                    q.append((i, depth + 1, prob))
        return 0
