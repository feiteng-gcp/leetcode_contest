class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], tt: int, target: int) -> float:
        to = collections.defaultdict(set)
        for a, b in edges:
            to[a].add(b)
            to[b].add(a)
        
        self.res = 0
        def dfs(i, p, t):
            if i == target and ((t < tt and len(to[i]) == 0) or t == tt):
                self.res = p
            for a in to[i]:
                to[a].remove(i)
                dfs(a, p / len(to[i]), t + 1)
        
        dfs(1, 1, 0)
        
        return self.res
