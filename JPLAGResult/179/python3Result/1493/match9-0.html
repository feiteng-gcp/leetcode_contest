from collections import Counter
class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        graph = [[] for _ in range(n + 1)]
        for n1, n2 in edges:
            graph[n1].append(n2)
            graph[n2].append(n1)
        # print(graph)
        prob = {(1, -1): 1}
        for _ in range(t):
            tmp = Counter()
            for k, v in prob.items():
                node, p = k
                # print(node, p, graph[node])
                if len(graph[node]) == 0 or (len(graph[node]) == 1 and graph[node][0] == p):
                    tmp[k] += v
                else:
                    nn = [n1 for n1 in graph[node] if n1 != p]
                    for n1 in nn:
                        tmp[(n1, node)] += v / len(nn)
            prob = tmp
            # print(prob)
        return sum([prob[k] for k in prob if k[0] == target])
