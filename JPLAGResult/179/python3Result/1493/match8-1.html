class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        neighbor = collections.defaultdict(set)
        for x, y in edges:
            neighbor[x].add(y)
            neighbor[y].add(x)
        level = [(1, 1, {1, },)]
        for _ in range(t):
            next_level = []
            for vertex, prob, visited in level:
                next_possible = neighbor[vertex] - visited
                if not next_possible:
                    next_level.append((vertex, prob, visited))
                else:
                    new_prob = prob / len(next_possible)
                    for next_vertex in next_possible:
                        next_level.append((next_vertex, new_prob, visited | {next_vertex, }))
            level = next_level
        return sum(p for v,p,_ in level if v == target)
