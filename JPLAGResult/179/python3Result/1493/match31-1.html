class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        gr = collections.defaultdict(list)
        for u, v in edges:
            gr[u].append(v)
            gr[v].append(u)
      #  print(gr)
        self.aa = []
        def helper(cur, aa, par=None, se=t):
            if se < 0:
                return
            if cur == target:
                self.aa = list(aa)
                return
            for v in gr[cur]:
                if v != par:
                    aa.append((v, (len(gr[v]) - 1) or 1))
                    helper(v, aa, cur,se - 1)
                    aa.pop()
        helper(1, [(1, len(gr[1]))], None, t)
        if not self.aa: return 0.0
        if len(self.aa) != t + 1:
            last = self.aa[-1]
            if (len(gr[last[0]]) > 1 and last[0] != 1) or (last[0] == 1 and len(gr[last[0]]) >= 1):
                return 0.0
        ans = 1.0
        for (u, length) in self.aa:
            ans *= 1.0 / (length or 1)
        return ans
