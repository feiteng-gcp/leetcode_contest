class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        adj = [[] for _ in range(n + 1)]
        for u, v in edges:
            adj[u].append(v)
            adj[v].append(u)
        
        seen = {1}
        q = [0.0] * (n + 1)
        q[1] = 1.0
        for _ in range(t):
            newq = [0.0] * (n + 1)
            for u, p in enumerate(q):
                if p == 0:
                    continue
                l = sum(v not in seen for v in adj[u])
                if l == 0:
                    newq[u] = p
                else:
                    for v in adj[u]:
                        if v not in seen:
                            seen.add(v)
                            newq[v] = p / l
            q = newq
        return q[target]
