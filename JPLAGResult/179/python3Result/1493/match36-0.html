from collections import defaultdict
class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        next_dict = defaultdict(list)
        for e in edges:
            next_dict[e[0]].append(e[1])
            next_dict[e[1]].append(e[0])
        
        visited_dict = defaultdict(int)
        
        visited_dict[1] = 1
        
        cur_probs = [(1, 1.0)]
        
        for i in range(t):
            next_probs = []
            for cur_p in cur_probs:
                node = cur_p[0]
                prob = cur_p[1]
                
                if next_dict[node]:
                    #next_prob = prob/len(next_dict[node])
                    next_steps = []
                    
                    for next_node in next_dict[node]:
                        if visited_dict[next_node] ==0:
                            next_steps.append(next_node)
                            visited_dict[next_node] = 1
                    if next_steps:
                        next_prob = prob/len(next_steps)
                        for next_node in next_steps:
                            next_probs.append((next_node, next_prob))
                    else:
                        next_probs.append((node, prob))
                            
                            
                        #next_probs.append((next_node, next_prob))
                else:
                    next_probs.append((node, prob))
            cur_probs = next_probs
        
        for tp in cur_probs:
            if tp[0] == target:
                return tp[1]
        
        return 0.0
