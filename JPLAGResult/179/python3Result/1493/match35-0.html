class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        edge=[[] for i in range(n+1)]
        for e in edges:
            edge[e[0]].append(e[1])
            edge[e[1]].append(e[0])
        deep=[-1 for i in range(n+1)]
        def dfs(u):
            for v in edge[u]:
                if deep[v]==-1:
                    deep[v]=deep[u]+1
                    dfs(v)
        deep[1]=0
        dfs(1)
        last=[0 for i in range(n+1)]
        for u in range(1,n+1):
            for v in edge[u]:
                if deep[v]==deep[u]+1:
                    last[u]+=1
        dp=[0.0 for i in range(n+1)]
        dp[1]=1.0
        for _ in range(t):
            ndp=[0.0 for i in range(n+1)]
            for u in range(1,n+1):
                if last[u]==0:
                    ndp[u]+=dp[u]
                else:
                    for v in edge[u]:
                        if deep[v]==deep[u]+1:
                            ndp[v]+=dp[u]/last[u]
            dp=ndp
        return dp[target]
