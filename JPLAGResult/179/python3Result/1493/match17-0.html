class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        graph=collections.defaultdict(list)
        for a,b in edges:
            graph[a].append(b)
            graph[b].append(a)
        que=[[1,1]]
        visited=set()
        for _ in range(t):
            tmp=[]
            for q,p in que:
                if q in visited:
                    continue
                visited.add(q)
                graph2=[nxt for nxt in graph[q] if nxt not in visited]
                l=len(graph2)
                if not l:
                    if q==target:
                        tmp.append([q,p])
                        visited.remove(q)
                else:
                    p2=p/l
                    for nxt in graph2:
                        if nxt!=q:
                            tmp.append([nxt,p2])
            que=tmp
        res=[p for q,p in que if q==target]
        return 0 if not res else res[0]
