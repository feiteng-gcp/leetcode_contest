class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        G = collections.defaultdict(list)
        for i, j in edges:
            G[i].append(j)
            G[j].append(i)
        
        pro = collections.defaultdict(float)
        pro[1] = 1.0
        que = [1]
        seen = {1}
        while t > 0 and que:
            next_que = []
            for i in que:
                to_visit = [j for j in G[i] if j not in seen]
                if len(to_visit) != 0:
                    for j in to_visit:
                        pro[j] = pro[i]/len(to_visit)
                        next_que.append(j)
                        seen.add(j)                        
                    pro[i] = 0
            que = next_que
            t -= 1
        return pro[target]
