class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        graph = collections.defaultdict(set)
        for a,b in edges:
            graph[a].add(b)
            graph[b].add(a)
        queue = [[1,0,set(),1]]
        res = 0
        while queue:
            curr, time, visit, poss = queue.pop(0)
            visit.add(curr)
            change = False
            if time < t:
                left = graph[curr] - visit
                for nei in left:
                    v = visit.copy()
                    v.add(nei)
                    change = True
                    queue.append([nei, time+1, v, poss/len(left)])
                if not change and curr == target:
                    res += poss
            elif time == t and target == curr:
                res += poss
                
        return res
