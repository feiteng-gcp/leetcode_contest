# possible optimization: coalesce (sum probs of) paths with same set of visited nodes and same last node

from collections import defaultdict

class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        edge_map = defaultdict(set)
        for a, b in edges:
            edge_map[a].add(b)
            edge_map[b].add(a)
        frontier = [([1], 1.0)] # start with path beginning at 1, with probability 1
        for _ in range(t):
            frontier = self.simulate(frontier, edge_map)
        total_prob = 0.0
        for path, prob in frontier:
            if path[-1] == target:
                total_prob += prob
        return total_prob
    
    def simulate(self, frontier, edge_map):
        new_frontier = []
        for path, prob in frontier:
            visited = set(path)
            current = path[-1] # node
            neighbors = edge_map[current] # set
            unvisited_neighbors = [n for n in neighbors if n not in visited]
            if len(unvisited_neighbors) == 0:
                # stay on current node, no need to make modifications
                new_frontier.append((path, prob))
                continue
            for neighbor in unvisited_neighbors:
                new_prob = prob / len(unvisited_neighbors)
                new_path = path + [neighbor]
                new_frontier.append((new_path, new_prob))
        return new_frontier
