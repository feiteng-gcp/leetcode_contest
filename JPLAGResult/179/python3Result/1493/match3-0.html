import collections
class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        graph=collections.defaultdict(list)
        for x,y in edges:
            graph[x].append(y)
            graph[y].append(x)
        # curr, pb, s
        queue=[[1, 1, set([1])]]
        
        for time in range(t):
            nxt=[]
            for curr, pb, s in queue:
                '''
                if time==t-1 and curr==target:
                    ret+=pb
                '''
                nbrs=[]
                for nbr in graph[curr]:
                    if nbr not in s:
                        nbrs.append(nbr)
                if len(nbrs)==0:
                    nxt.append([curr, pb, s])
                else:
                    nnbr=len(nbrs)
                    for nbr in nbrs:
                        nxt.append([nbr, pb/nnbr, s|{nbr}])
            queue=nxt
        
        ret=0
        for curr, pb, s in queue:
            if curr==target:
                ret+=pb
        return ret
