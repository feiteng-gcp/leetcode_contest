class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        nei = [[] for _ in range(n + 1)]
        for a, b in edges:
            nei[a].append(b)
            nei[b].append(a)
        res = self.dfs(1, 1, 0, t, nei, target)
        return res if res > 0 else 0.0

    def dfs(self, prob, now, pre, ti, nei, target):
        w = len(nei[now]) - (1 if pre > 0 else 0)
        # print(prob, ti, now, w)
        if ti == 0 or w == 0:
            return prob if now == target else -1.0
        for nxt in nei[now]:
            if nxt != pre:
                res = self.dfs(prob / w, nxt, now, ti - 1, nei, target)
                if res > 0:
                    return res
        return -1
