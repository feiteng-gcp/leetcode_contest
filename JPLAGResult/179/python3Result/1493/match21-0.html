class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        G = [[] for _ in range(n)]
        for u, v in edges:
            G[u-1].append(v-1)
            G[v-1].append(u-1)
        seen = [False]*n
        self.res = -1
        def dfs(u, h, p):
            d = len(G[u]) - (h != 0)
            if u == target-1:
                if d == 0 and t >= h:
                    self.res = p
                elif d != 0 and t == h:
                    self.res = p
                else:
                    self.res = 0
            seen[u] = True
            for v in G[u]:
                if not seen[v]:
                    dfs(v, h+1, p/d)
        dfs(0, 0, 1.0)
        return self.res
