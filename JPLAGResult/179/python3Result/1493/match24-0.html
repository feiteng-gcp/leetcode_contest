from typing import List

class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        adjList = [[] for _ in range(n + 1)]
        for e in edges:
            adjList[e[0]].append(e[1])
            adjList[e[1]].append(e[0])
        
        parent = [0 for _ in range(n + 1)]
        parent[1] = -1
        queue = [1]
        while queue:
            top = queue.pop(0)
            for child in adjList[top]:
                if parent[child] == 0:
                    parent[child] = top
                    queue.append(child)
        
        current = target
        p = 1
        time = 0
        while current != 1:
            current = parent[current]
            num_children = len(adjList[current])
            if current != 1:
                num_children -= 1
            p /= num_children
            time += 1
        target_children = len(adjList[target])
        if target != 1:
            target_children -= 1
        if time > t or (target_children != 0 and time != t):
            return 0
        else:
            return p
