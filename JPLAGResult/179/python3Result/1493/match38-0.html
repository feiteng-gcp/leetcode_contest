class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        graph = [[] for i in range(n+1)]
        for a, b in edges:
            graph[a].append(b)
            graph[b].append(a)
        prob = [0]*(n+1)
        prob[1] = 1
        ok = []
        def dfs(node, prev, s):
            if s == t or len(graph[node]) == 0 or (len(graph[node]) == 1 and graph[node][0] == prev): 
                ok.append(node)
                return
            cnt = 0
            for i in graph[node]:
                if i != prev:
                    cnt += 1
            div = prob[node] / cnt
            for i in graph[node]:
                if i != prev:
                    prob[i] = div
                    dfs(i, node, s+1)
            
        dfs(1, -1, 0)
        if target in ok:
            return prob[target]
        return 0
