from collections import defaultdict

class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:

        connected = defaultdict(list)
        for f, to in edges:
            connected[f].append(to)
            connected[to].append(f)

        seen = {1}
        children = defaultdict(list)
        level = [1]
        while len(seen) < n:
            newlevel = []
            for x in level:
                for c in connected[x]:
                    if c not in seen:
                        children[x].append(c)
                        seen.add(c)
                        newlevel.append(c)

            level = newlevel


        level = [(1, 1)]
        for i in range(t):
            newlevel = []
            for x, n in level:
                sub = children[x]
                if sub:
                    for s in sub:
                        newlevel.append((s, n * len(sub)))
                else:
                    newlevel.append((x, n * 1))

            level = newlevel

        for x, t in level:
            if x == target:
                return 1 / t

        return 0
