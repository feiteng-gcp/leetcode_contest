class Solution(object):
    def frogPosition(self, n, edges, t, target):
        conn = collections.defaultdict(set)
        for u,v in edges:
            conn[u].add(v)
            conn[v].add(u)
        cur = {1:[[set([1]),1]]}
        res = 0
        for _ in range(t):
            tmp = collections.defaultdict(list)
            for node in cur:
                for his,p in cur[node]:
                    flag = False
                    for nei in conn[node]:
                        if nei not in his:
                            flag = True
                            new_his = set([i for i in his])
                            new_his.add(nei)
                            new_p = 1/len(conn[node] - his) * p
                            tmp[nei] += [[new_his,new_p]]
                    if not flag and node == target: res += p
            cur = tmp
        for his,p in cur[target]: res += p
        return res
