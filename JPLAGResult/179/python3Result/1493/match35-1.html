from collections import defaultdict


class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        # concern about timing out - 
        # if you visit same spot, but it has to be different since you are eliminating visited edges.
        # but no guarantee that it's eliminated with regard to that reality.
        
        # wait, this is an undirected tree
        # am i dumb?
        
        
        g_edges = defaultdict(set)
        for edge in edges:
            g_edges[edge[0]].add(edge[1])
            g_edges[edge[1]].add(edge[0])
            
        visited = set()
        
        
        def recursive_helper(base_prob, current, time):    
            visited.add(current)
            
            if time == t:
                return base_prob if current == target else 0
            
            current_prob = 0
            
            edges = []
            for edge in g_edges[current]:
                if edge not in visited:
                    edges.append(edge)
                    
            if len(edges) == 0:
                return base_prob if current == target else 0
            
            for edge in edges:
                current_prob += recursive_helper(base_prob / len(edges), edge, time + 1)
            
            return current_prob
        
        return recursive_helper(1, 1, 0)
