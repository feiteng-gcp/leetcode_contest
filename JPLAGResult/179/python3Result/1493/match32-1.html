class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        eg = collections.defaultdict(set)
        for fr, to in edges:
            eg[fr].add(to)
            eg[to].add(fr)
        cur = [(1, set((1,)), 1)]
        while t > 0:
#             print(cur)
            nxt = []
            for pos, visted, p in cur:
                if visted is None:
                    nxt.append((pos, None, p))
                    continue
                con = eg[pos]
                candi = con.difference(visted)
                if candi:
                    for c in candi:
                        nxt.append((c, visted | set((c,)), p/len(candi)))
                else:
                    nxt.append((pos, None, p))
            t -= 1
            cur = nxt
        res = 0
        for pos, visted, p in cur:
            if pos == target:
                res += p
        return res
