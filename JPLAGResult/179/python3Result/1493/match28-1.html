class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        g=collections.defaultdict(list)
        for u,v in edges:
            g[u].append(v)
            g[v].append(u)
        self.ans=0
        visited={1}
        def bt(i,p,tx):
            if tx==t:
                if i==target:
                    self.ans+=p
                return
            c=0
            res=[]
            visited.add(i)
            for v in g[i]:
                if v not in visited:
                    c+=1
                    res.append(v)
            if not res:
                if i==target:
                    self.ans+=p
            for v in res:
                bt(v,p/c,tx+1)
            visited.remove(i)
        bt(1,1,0)
        return self.ans
