class Solution:
    def frogPosition(self, n: int, edges: List[List[int]], t: int, target: int) -> float:
        d = collections.defaultdict(list)
        for e in edges:
            d[e[0]].append(e[1])
            d[e[1]].append(e[0])
        
        res, walked = 0.0, set()
        
        def helper(n, t, p):
            nonlocal res, walked
            if t == 0:
                if n == target:
                    res += p
                return 
            
            nextNodes = []
            for x in d[n]:
                if x not in walked:
                    nextNodes.append(x)
            if len(nextNodes) == 0:
                if n == target:
                    res += p
            else:
                if n == target:
                    return
                for x in nextNodes:
                    walked.add(x)
                    helper(x, t-1, p*(1/len(nextNodes)))
                    walked.remove(x)
        
        walked.add(1)
        helper(1, t, 1.0)
        return res
