class Solution {
    public String generateTheString(int n) {
        if(n == 1){
            return "a";
        } else if(n == 2){
            return "ab";
        }
        StringBuilder sb = new StringBuilder();
        if((n & 1) == 0){
            sb.append('a');
            while(n-- > 1){
                sb.append('b');
            }
        } else {
            while(n-- > 0){
                sb.append('b');
            }
        }
        return sb.toString();
    }
}
