class Solution {
    public int numTimesAllBlue(int[] light) {
        long sum = 0;
        int count = 0;
        long max = 0;
        for (int i : light) {
            sum += i;
            max = Math.max(max, i);
            long target = (1 + max) * max / 2;
            if (sum == target) {
                count++;
            }
        }
        return count;
    }
}
