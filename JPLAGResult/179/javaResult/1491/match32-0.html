class Solution {
    public int numTimesAllBlue(int[] light) {
        
        int ma  = 0;
        int mi = 10000000;int c=  0;
        for(int i=0;i<light.length;++i){
            ma = Math.max(ma,light[i]);
            mi = Math.min(mi,light[i]);
            if(ma-mi+1==i+1&&ma==i+1){
                c++;
            }
        }
        return c;
    }
}
