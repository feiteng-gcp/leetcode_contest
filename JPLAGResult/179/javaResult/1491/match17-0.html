class Solution {
    public int numTimesAllBlue(int[] light) {
        int max = 0, cnt = 0, ans = 0;
        for (int k = 0 ; k < light.length; k++) {
            max = Math.max(max, light[k]);
            cnt++;
            if (cnt == max) {
                ans++;
            }
        }
        return ans;

    }
}
