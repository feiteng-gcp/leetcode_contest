class Solution {
    private double prob;
    public double frogPosition(int n, int[][] edges, int t, int target) {
        List<Set<Integer>> tree = new ArrayList<>();
        for (int i = 0; i <= n; i++) tree.add(new HashSet<>());
        for (int[] e : edges) {
            tree.get(e[0]).add(e[1]);
            tree.get(e[1]).add(e[0]);
        }
        
        prob = 0;
        dfs(1, tree, t, target, 1.0);
        return prob;
    }
    
    private void dfs(int cur, List<Set<Integer>> tree, int t, int target, double p) {
        if (cur == target && t == 0) {
            this.prob = p;
            return;
        }
        
        if (t == 0) return;
        int total = tree.get(cur).size();
        boolean hasUn = false;
        for (int child : tree.get(cur)) {
            hasUn = true;
            tree.get(child).remove(cur);
            dfs(child, tree, t - 1, target, p * 1.0 / total);
        }
        
        if (!hasUn) {
            dfs(cur, tree, t - 1, target, p);
        }
    }
}
