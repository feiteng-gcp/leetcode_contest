class Solution {
    public double frogPosition(int n, int[][] edges, int t, int target) {
        
List<List<Integer>> tree = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            tree.add(new ArrayList<>());
        }
        
        
for (int[] e : edges) {
            tree.get(e[0]-1).add(e[1]-1);
            tree.get(e[1]-1).add(e[0]-1);
        }
        
        double[] p = new double[n];
        p[0] = 1.0;
        
go(tree, 0, p, -1, t);
        return p[target-1];
    }
    
    void go(List<List<Integer>> tree, int i, double[] p, int back, int time) {
        int size = tree.get(i).size() - ((i == 0) ? 0 : 1);
        if (time == 0) return;
        if (size == 0) return;
        double portion = p[i] / (double) size;
        p[i] = 0;
        for (int j : tree.get(i)) {
            if (j == back) continue;
            p[j] = portion;
            go(tree, j, p, i, time-1);
        }
        
    }
}
