class Solution {
    int target;
    ArrayList<Integer>[] graph;
    int d;
    boolean f;
    double ret;
    public double frogPosition(int n, int[][] edges, int t, int target) {
        this.target=target-1;
        graph=(ArrayList<Integer>[])new ArrayList[n];
	    for(int i=0;i<n;i++)graph[i]=new ArrayList<>();
        for(int[] e:edges){
            graph[e[0]-1].add(e[1]-1);
            graph[e[1]-1].add(e[0]-1);
        }
        dfs(0,1d,0,new boolean[n]);
        if(d>t)return 0;
        if(d<t&&f)return 0;
        // System.out.println(d+","+f+","+ret);
        return ret;
    }
    void dfs(int node,double pro,int dep,boolean[] vis){
        vis[node]=true;
        if(node==target){
            d=dep;
            ret=pro;
            for(int nn:graph[node]){
                if(!vis[nn]){
                    f=true;
                }
            }
            return;
        }
        int cnt=0;
        for(int nn:graph[node]){
            if(!vis[nn]){
                cnt++;
            }
        }
        if(cnt==0)return;
        for(int nn:graph[node]){
            if(vis[nn])continue;
            dfs(nn,pro/cnt,dep+1,vis);
        }
    }
}
