class Solution {
    public double frogPosition(int n, int[][] edges, int t, int target) {
        List<Integer>[] adj = new List[n + 1];
        for(int i = 1; i <= n ; i++){
            adj[i] = new ArrayList<>();
        }
        for(int[] edge: edges){
            adj[edge[0]].add(edge[1]);
            adj[edge[1]].add(edge[0]);
        }
        boolean[] seen = new boolean[n + 1];
        return dfs(adj, seen, n, t, target, 1);
    
}
    
    private double dfs(List<Integer>[] adj, boolean[] seen, int n, int t, int target, int cur){
        if(t == 0){
            return cur == target ? 1: 0;
        }
        seen[cur] = true;
        int cnt = 0;
        for(int next: adj[cur]){
            if(seen[next]){
                continue;
            }
            cnt++;
        }
        if(cnt == 0){
            return cur == target ? 1: 0;
        } else {
            for(int next: adj[cur]){
                if(seen[next]){
                    continue;
                }
                double sub = dfs(adj, seen, n, t - 1, target, next);
                if(sub != 0){
                    return sub / cnt;
                }
            }
        }
        return 0;
    }
}
