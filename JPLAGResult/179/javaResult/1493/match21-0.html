class Solution {
    double[] count;
    List<List<Integer>> g;
    boolean[] visited;
    public double frogPosition(int n, int[][] edges, int t, int target) {
        count = new double[n+1];
        visited = new boolean[n+1];
        g = buildG(edges, n);
        dfs(1, t, 1.0);
        //for (double c : count) System.out.print(c + " ");
        return count[target];
    }
    
    private void dfs(int i, int left, double p) {
        if (left == 0) {
            count[i] = p;
            return;
        }
        visited[i] = true;
        int nov = 0;
        for (int nb : g.get(i)) {
            if (!visited[nb]) nov ++;
        }
        
        for (int nb : g.get(i)) {
            if (!visited[nb]) {
                dfs(nb, left - 1, p/nov);
            } 
        }
        if (nov == 0) count[i] = p;
        visited[i] = false;
    }
    
    
    
    
    
    
    private List<List<Integer>> buildG(int[][] edges, int n) {
        List<List<Integer>> g = new ArrayList<>();
        for (int i = 0; i <= n; i++) g.add(new LinkedList<>());
        for (int[] e : edges) {
            g.get(e[0]).add(e[1]);
            g.get(e[1]).add(e[0]);
        }
        return g;
    }
}
