class Solution {
    boolean[] visited;
    
    
    public double frogPosition(int n, int[][] edges, int t, int target) {
        ArrayList<ArrayList<Integer>> side = new ArrayList<>();
        visited = new boolean[n + 1];
        
for(int i = 0; i < n + 1; i++){
            side.add(new ArrayList<>());
        }
        
        for(int[] edge : edges){
            side.get(edge[0]).add(edge[1]);
            side.get(edge[1]).add(edge[0]);
        }
        
        return dfs(1, target, side, 1, 0, t);
    
}
    
    private double dfs(int cur, int target, ArrayList<ArrayList<Integer>> side, double possible, int step, int limit){
        if(step > limit){
            return 0;
        }
        
        if(cur == target){
            if(step == limit || (cur == 1 && side.get(cur).size() == 0) ||(cur != 1 && side.get(cur).size() == 1)){
                return possible;
            }
            return 0;
        }
        
        visited[cur] = true;
        double count = side.get(cur).size() - 1;
        if(cur == 1){
            count++;
        }
        double nextPossible = possible / count;
        for(int next : side.get(cur)){
            if(!visited[next]){
                double res = dfs(next, target, side, nextPossible, step + 1, limit);
                if(res != 0){
                    return res;
                }
            }
        }
        
        visited[cur] = false;
        return 0;
    }
}
