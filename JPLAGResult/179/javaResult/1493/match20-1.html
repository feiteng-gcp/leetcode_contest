class Solution {
    
public double frogPosition(int n, int[][] edges, int t, int target) {
        Map<Integer, List<Integer>> graph = new HashMap<>();
        for (int i = 1; i <= n; i++) {
            graph.put(i, new ArrayList<>());
        }
        graph.get(1).add(0);
        
for (int[] e : edges) {
            graph.get(e[0]).add(e[1]);
            graph.get(e[1]).add(e[0]);
        }
        return prob(1, 0, t, target, graph);
    
}
    private double prob(int cur, int prev, int time, int target, Map<Integer, List<Integer>> graph) {
        if (cur == target && (time == 0 || graph.get(cur).size() == 1)) {
            return 1;
        }
        if (cur == target || graph.get(cur).size() == 1 || time == 0) {
            return 0;
        }
        double p = 0;
        for (int child : graph.get(cur)) {
            if (p > 0) {
                break;
            }
            if (child == prev) {
                continue;
            }
            p = prob(child, cur, time - 1, target, graph);
        }
        return p / (graph.get(cur).size() - 1);
    }
}
