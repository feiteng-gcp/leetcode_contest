class Solution {
    List<Integer> edges[];
    double res = 0;

    public void dfs(int root, int dst, int f, int target, double cur) {
        List<Integer> es = edges[root];
        
if (dst < 0) {
            if (root == target) {
                res = cur;
            }
            return;
        }
        boolean flag = false;
        for (int i = 0; i < es.size(); i++) {
            int t = es.get(i);
            if (t == f) {
                continue;
            }
            dfs(t, dst - 1, root, target, cur * (1.0 / (es.size() - (f == -1 ? 0 : 1))));
            flag = true;
        
}
        if (!flag && root == target) {
            res = cur;
        }
    }

    public double frogPosition(int n, int[][] g, int t, int target) {
        edges = new List[n + 1];
        for (int i = 1; i <= n; i++) {
            edges[i] = new ArrayList<>();
        }
        for (int i = 0; i < n - 1; i++) {
            edges[g[i][0]].add(g[i][1]);
            edges[g[i][1]].add(g[i][0]);
        }
        if (t > 0) {
            dfs(1, t - 1, -1, target, 1.0);
        } else {
            return target == 1 ? 1 : 0;
        }
        return res;
    }
}
