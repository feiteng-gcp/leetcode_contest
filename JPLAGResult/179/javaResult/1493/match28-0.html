class Solution {
    public double frogPosition(int n, int[][] edges, int t, int target) {
        HashMap<Integer, List<Integer>> map = new HashMap<>();
        for(int[] edge : edges) {
            int a = edge[0], b = edge[1];
            map.computeIfAbsent(a, k -> new ArrayList<>()).add(b);
            map.computeIfAbsent(b, k -> new ArrayList<>()).add(a);
        }
        HashSet<Integer> visited = new HashSet<>();
        visited.add(1);
        HashMap<String, Double> memo = new HashMap<>();
        double res = dfs(1, 0, t, target, map, visited, memo);
        return res;
    }
    
    private double dfs(int cur, int k, int t, int target, HashMap<Integer, List<Integer>> map, HashSet<Integer> visited, HashMap<String, Double> memo) {
        if(k == t) {
            if(cur == target) return 1.0;
            else return 0.0;
        }
        String status = "";
        for(int i : visited) status += Integer.toString(i) + " ";
        status += Integer.toString(cur) + " " + Integer.toString(k);
        if(memo.get(status) != null) return memo.get(status);
        double res = 0.0;
        List<Integer> list = map.getOrDefault(cur, new ArrayList<>());
        int cnt = 0;
        for(int next : list) {
            if(visited.contains(next)) continue;
            else cnt += 1;
        }
        if(cnt == 0) {
            if(cur == target) res = 1.0;
            else res = 0.0;
        }
        else {
            double prob = 1.0 / (double)cnt;
            for(int next : list) {
                if(visited.contains(next)) continue;
                else {
                    visited.add(next);
                    res += prob * dfs(next, k + 1, t, target, map, visited, memo);
                    visited.remove(next);
                }
            }
        }
        memo.put(status, res);
        return res;
    }
}
