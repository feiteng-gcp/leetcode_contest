class Solution {
    public double frogPosition(int n, int[][] edges, int t, int target) {
        // assume tree structure
        target -= 1;
        List<List<Integer>> neighbors = new ArrayList<>();
        for (int i = 0; i < n ; i++) {
            neighbors.add(new ArrayList<>());
        }
        for (int[] e : edges) {
            neighbors.get(e[0]-1).add(e[1]-1);
            neighbors.get(e[1]-1).add(e[0]-1);
        }
        
        boolean[] visited = new boolean[n];
        Queue<long[]> queue = new LinkedList<>();
        Queue<long[]> saveQueue = new LinkedList<>();
        queue.add(new long[] {0, 1});
        visited[0] = true;
        
        while (!queue.isEmpty() && t-- > 0) {
            saveQueue = new LinkedList<>(queue);
            int size = queue.size();
            while (size-- > 0) {
                long[] curr = queue.remove();
                List<Integer> neibs = neighbors.get((int)curr[0]);
                boolean hasChild = false;
                for (int nxt : neibs) {
                    if (!visited[nxt]) {
                        hasChild = true;
                        visited[nxt] = true;
                        queue.add(new long[] {nxt, curr[1]*(neibs.size()-(curr[0]==0 ? 0 : 1))});
                    }
                }
                if (!hasChild) {
                    queue.add(curr);
                }
            }
        }
        
        if (queue.isEmpty()) {
            queue = saveQueue;
        }
        
        while (!queue.isEmpty()) {
            long[] curr = queue.remove();
            if (curr[0] == target) {
                return 1./curr[1];
            }
        }
        return 0;
        
    }
}
