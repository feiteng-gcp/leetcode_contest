import java.util.ArrayList;
import java.util.Arrays;

class Solution {
    public double frogPosition(int n, int[][] edges, int t, int target) {


        double[][] memo = new double[n + 1][n + 1];

        for (double[] row : memo) {
            Arrays.fill(row, -1);
        }

        ArrayList<ArrayList<Integer>> adj = new ArrayList<>();

        for (int i = 0; i < n + 1; i++) {
            adj.add(new ArrayList<>());
        }
        for (int[] e : edges) {
            int v = e[0], w = e[1];
            adj.get(v).add(w);
            adj.get(w).add(v);
        }


        return findProb(adj, t, 1, 1, target, memo);
    }

    private double findProb(ArrayList<ArrayList<Integer>> adj, int t,
                            int curr, int prev, int target, double[][] memo) {

        if (curr == target) {
            if (t == 0) {
                return 1.0;
            } else {
                if (adj.get(curr).size() == 0 || adj.get(curr).size() == 1 && prev != curr) {
                    return 1;
                } else {
                    return 0;
                }
            }

        } else if (t == 0) {
            return 0;
        }

        if (memo[curr][prev] >= 0) {
            return memo[curr][prev];
        }

        double possibleRoute = adj.get(curr).size();
        double p = 0;
        if (curr != prev) {
            possibleRoute -= 1;
        }

        for (int w : adj.get(curr)) {
            if (w == prev) {
                continue;
            }
            p += findProb(adj, t - 1, w, curr, target, memo) / possibleRoute;
        }

        memo[curr][prev] = p;
        return p;


    }
}
