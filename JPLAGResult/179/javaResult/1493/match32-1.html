class Solution {
    class T {
        int v;
        double p; 
        public T(int i, double d) {
            v = i;
            p = d;
        }
    }
    
    
public double frogPosition(int n, int[][] edges, int t, int target) {
        // how many possible at each vertex and the distance 
        List<Integer>[] list = new List[n + 1];
        for (int i = 0; i <= n; i++) {
            list[i] = new ArrayList<>();
        }
        for (int[] edge : edges) {
            list[edge[0]].add(edge[1]);
            list[edge[1]].add(edge[0]);
        }
        
        boolean[] visited = new boolean[n + 1];
        visited[1] = true;
        Queue<T> q = new LinkedList<>();
        q.offer(new T(1, 1.0));
        
        double res = 0.0;
        while (!q.isEmpty()) {
            int s = q.size();
            for (int k = 0; k < s; k++) {
                T cur = q.poll();
                if (cur.v == target) {
                    if (t == 0) {
                        return cur.p;
                    }
                    for (int next : list[target]) {
                        if (!visited[next]) {
                            return 0.0;
                        }
                    }
                    return cur.p;
                }
                
int count = 0;
                for (int next : list[cur.v]) {
                    if (!visited[next]) {
                        count++;
                    }
                }
                for (int next : list[cur.v]) {
                    if (!visited[next]) {
                        q.offer(new T(next, cur.p / count));
                        visited[next] = true;
                    }
                }
            }
            t--;
            if (t < 0) {
                return 0.0;
            }
        }
        return 0.0;
    }
}
