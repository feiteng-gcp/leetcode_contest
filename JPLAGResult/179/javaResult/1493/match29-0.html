class Solution {
    double[] p;
    boolean[][] e;
    int target;
    int tt;
    double pp;
    int n;
    
    public void visit(int nid, int t) {
        if (t == tt) {
            if (nid == target) {
                pp = p[nid];
            }
            return;
        }
        int vc = 0;
        for (int i = 1; i <= n; ++i) {
            if (e[nid][i] && p[i] == 0) {
                vc++;
            }
        }
        for (int i = 1; i <= n; ++i) {
            if (e[nid][i] && p[i] == 0) {
                p[i] = p[nid] / vc;
                //System.out.println(nid + " " + i + " " + p[i]);
                visit(i, t + 1);
            }
        }
        if (vc == 0 && nid == target) {
            pp = p[nid];
        }
    }
    
    public double frogPosition(int n, int[][] edges, int t, int target) {
        p = new double[n + 1];
        p[1] = 1;
        e = new boolean[n + 1][n + 1];
        for (int i = 0; i < edges.length; ++i) {
            e[edges[i][0]][edges[i][1]] = true;
            e[edges[i][1]][edges[i][0]] = true;
        }
        this.target = target;
        this.tt = t;
        this.n = n;
        pp = 0;
        visit(1, 0);
        return pp;
    }
}
