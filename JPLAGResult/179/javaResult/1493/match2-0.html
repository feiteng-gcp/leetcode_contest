class Solution {
    public double frogPosition(int n, int[][] edges, int t, int target) {
        class item
        {
            int pos;
            double prob;

            public item(int p, double pp)
            {
                pos = p;
                prob = pp;
            }
        }

        Map<Integer, Set<Integer>> map = new HashMap<>();
        for(int i = 1; i <= n; i++) map.put(i, new HashSet<>());
        for(int[] e : edges)
        {
            int u = e[0], v = e[1];
            map.get(u).add(v);
            map.get(v).add(u);
        }
        
        boolean[] visited = new boolean[n + 1];
        
        Queue<item> q = new LinkedList<>();
        q.offer(new item(1, 1.0));
        visited[1] = true;

        while(t-- > 0)
        {
            int size = q.size();
            while(size-- > 0)
            {
                item p = q.poll();
                int cur = p.pos;
                double p1 = p.prob;
                boolean flag = false;
                int count = 0;
                for(int next : map.get(cur))
                {
                    if(visited[next]) continue;
                    // visited[next] = true;
                    flag = true;
                    count++;
                    // q.offer(new int[]{next, v});
                }
                if(!flag)
                {
                    q.offer(p);
                }
                else
                {
                    for(int next : map.get(cur))
                    {
                        if(visited[next]) continue;
                        double p2 = 1.0 / count;
                        visited[next] = true;
                        q.offer(new item(next, p1 * p2));
                    }
                }
            }
        }
        
        
        double ans = 0.0;
        for(item qq : q)
        {
            if(qq.pos == target) ans += qq.prob;
        }

        return ans;
    }    

}
