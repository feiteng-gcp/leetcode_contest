class Solution {
    List<Integer> g[];
    double dp[];
    void go(int root,double p,int time,int fa){
        if(time==0){
            dp[root] += p;            
            return;
        }
        
        
        int f = g[root].size();
        if(root!=0){
            f--;
        }
        
        if(f>0){
        double ff = 1.0/f;
        for(int c:g[root]){
            if(c== fa) continue;
            go(c,p*ff,time-1,root);
        }
        }else{
            dp[root] = p;
        
}
        
        
    }
    
    
    public double frogPosition(int n, int[][] es, int t, int gt) {

        g = new ArrayList[n];
        dp = new double[n];

        for(int i=0;i<n;++i){
            g[i] = new ArrayList<>();
        }
        for(int i=0;i<n-1;++i){
            g[es[i][0]-1].add(es[i][1]-1);
            g[es[i][1]-1].add(es[i][0]-1);
        }
        
        go(0,1.0,t,-1);
        
        return dp[gt-1];
    }
}
