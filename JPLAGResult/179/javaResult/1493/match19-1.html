class Solution {
    boolean[] visited;
    
double ans = 0;
    public double frogPosition(int n, int[][] edges, int t, int target) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for(int i = 1; i <= n; i++)
            map.put(i, new ArrayList<>());
        for(int[] e: edges) {
            map.get(e[0]).add(e[1]);
            map.get(e[1]).add(e[0]);
        }
        visited = new boolean[n + 1];
        
dfs(map, 1, 1, 0, t, target, -1);
        return ans;
//         double[][] dp = new double[t][n + 1];
//         dp[0][1] = 1;
        
    }
    public void dfs(Map<Integer, List<Integer>> map, int cur, double p, int step, int t, int target, int par) {
        if(step == t) {
            if(cur == target)
                ans += p;
            return;
        }
        List<Integer> nexts = map.get(cur);
        int N = nexts.size();
        if(N == 0)
            dfs(map, cur, p, step + 1, t, target, par);
        else if(N == 1) {
            if(par == -1)
                dfs(map, nexts.get(0), p, step + 1, t, target, cur);
            else 
                dfs(map, cur, p, step + 1, t, target, par);
        } else
            for(int next: nexts)
                if(next != par) {
                    if(par == -1)
                        dfs(map, next, p * 1 / N, step + 1, t, target, cur);
                    else
                        dfs(map, next, p * 1 / (N - 1), step + 1, t, target, cur);
                }
    }
}
