class Solution {
    public double frogPosition(int n, int[][] edges, int t, int target) {
        ArrayList<ArrayList<Integer>> adj = new ArrayList<>();
        for(int i = 0; i <= n; i++){
            adj.add(new ArrayList<>());
        }
        int[] degree = new int[n+1];
        degree[1]++;
        
        for(int i = 0; i < edges.length; i++){
            int x = edges[i][0];
            int y = edges[i][1];
            adj.get(x).add(y);
            adj.get(y).add(x);
            degree[x]++;
            degree[y]++;
        }
        
        int[] pred = new int[n+1];
        LinkedList<Integer> queue = new LinkedList<>();
        queue.offer(1);
        while(queue.size() > 0){
            int x = queue.poll();
            for(Integer y: adj.get(x)){
                if (y != pred[x]){
                    pred[y] = x;
                    queue.offer(y);
                }
            }
            if (pred[target] > 0) break;
        }
        
        double prod = 1;
        int x = target;
        int step = 0;
        while(pred[x] > 0){
            step++;
            x = pred[x];
            prod *= degree[x] - 1;
        }
        
        if (t == step || (t > step && degree[target] == 1)){
            return 1 / prod;   
        }else{
            return 0;
        }
    }
}
