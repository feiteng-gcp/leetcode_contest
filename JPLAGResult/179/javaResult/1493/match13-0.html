class Solution {
    public double frogPosition(int n, int[][] edges, int t, int target) {
        if (n == 1) {
            return 1.0;
        }
        double[] prob = new double[1];
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 1; i <= n; i++) {
            map.put(i, new ArrayList<>());
        }
        map.get(1).add(0);
        for (int[] edge : edges) {
            map.get(edge[0]).add(edge[1]);
            map.get(edge[1]).add(edge[0]);
        }
        dfs(map, 1, prob, 1.0, t, target, 0);
        return prob[0];
    }
    
    private void dfs(Map<Integer, List<Integer>> map, int currNode, double[] prob, double currprob, int t, int target, int parentNode) {
        if (currNode == target && (t == 0 || map.get(currNode).size() == 1)) {
            prob[0] = currprob;
            return;
        }
        if (t == 0) {
            return;
        }
        List<Integer> next = map.get(currNode);
        double nextprob = 1.0 / (next.size() - 1);
        for (Integer n : next) {
            if (n != parentNode) {
                dfs(map, n, prob, currprob * nextprob, t - 1, target, currNode);   
            }
        }
    }
}
