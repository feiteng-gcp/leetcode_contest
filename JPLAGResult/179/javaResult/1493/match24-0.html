class Solution {

	public double frogPosition(int n, int[][] edges, int t, int target) {
		ArrayList<Integer>[] graph = new ArrayList[n];
		for (int i = 0; i < graph.length; i++) {
			graph[i] = new ArrayList<>();
		}
		for (int[] edge : edges) {
			graph[edge[0] - 1].add(edge[1] - 1);
			graph[edge[1] - 1].add(edge[0] - 1);
		}
		double[] prob = new double[n];
		boolean[] visited = new boolean[n];
		visited[0] = true;
		frogPosition(0, 1, t, graph, prob, visited);
		return prob[target - 1];
	}

	private void frogPosition(int i, double p, int t, ArrayList<Integer>[] graph, double[] prob, boolean[] visited) {
		if (t == 0) {
			prob[i] += p;
		} else {
			int count = 0;
			for (int j : graph[i]) {
				if (!visited[j]) {
					count++;
				}
			}
			if (count == 0) {
				prob[i] += p;
			} else {
				for (int j : graph[i]) {
					if (!visited[j]) {
						visited[j] = true;
						frogPosition(j, p / count, t - 1, graph, prob, visited);
						visited[j] = false;
					}
				}
			}
		}
	}
}
