class Solution {
    
double res = 0;
    int target;
    List<Integer>[] next;
    public double frogPosition(int n, int[][] edges, int t, int target) {
        next = new List[n + 1];
        this.target = target;
        
for (int i = 1; i <= n; ++i) next[i] = new ArrayList<>();
        for (int[] e : edges) {
            next[e[0]].add(e[1]);
            next[e[1]].add(e[0]);
        }
        dfs(1, t, new boolean[n + 1], (double) 1);
        
return res;
    }
    private void dfs(int cur, int t, boolean[] visit, double prob) {
        if (t == 0) {
            if (cur == target) res += prob;
            
return;
        }
        int count = 0;
        for (int n : next[cur]) {
            if (!visit[n]) { ++count; }
        }
        if (count == 0) {
            if (cur == target) res += prob;
        } else {
            visit[cur] = true;
            for (int n : next[cur]) {
                if (!visit[n]) { dfs(n, t - 1, visit, (double) (prob / count)); }
            }
            visit[cur] = false;
        }
    }
}
