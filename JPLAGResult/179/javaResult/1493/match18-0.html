class Solution {
    public void dfs(int t, int v, int p, List<Integer>[] g, double[] dp) {
        if (t == 0) return;
        int nei = p == -1? g[v].size() : g[v].size() - 1;
        if (nei == 0) return;
        for (int a : g[v]) {
            if (a != p) {
                dp[a] = dp[v] / nei;
                dfs(t - 1, a, v, g, dp);
            }
        }
        dp[v] = 0.0;
        
    }
    public double frogPosition(int n, int[][] edges, int t, int target) {
        List[] g = new List[n];
        for (int i = 0; i < n; i++) {
            g[i] = new ArrayList<>();
        }
        for (int[] a : edges) {
            g[a[0] - 1].add(a[1] - 1);
            g[a[1] - 1].add(a[0] - 1);
        }
        // System.out.println(Arrays.toString(g));
        double[] dp = new double[n];
        Arrays.fill(dp, 0.0);
        dp[0] = 1.0;
        dfs(t, 0, -1, g, dp);
        return dp[target - 1];
    }
}
