class Solution {
    double probability = 0;
    public double frogPosition(int n, int[][] edges, int t, int target) {
        Map<Integer, List<Integer>> map = buildMap(edges);
        Set<Integer> set = new HashSet<>();
        set.add(1);
        
dfs(1, 1, 0, t, target, map, set);
        return probability;
    }
    private void dfs(int vertice, double prob, int curTime, int totalTime, int target, Map<Integer, List<Integer>> map, Set<Integer> visited) {
        if (curTime > totalTime)
            return;
        Set<Integer> nexts = new HashSet<>(map.getOrDefault(vertice, new ArrayList<>()));
        nexts.removeAll(visited);
        if (vertice == target) {
            if (nexts.size() == 0 || curTime == totalTime) {
                probability += prob;
                return;
            } else {
                return;
            }
        }

        
for (int next : nexts) {
            visited.add(next);
            dfs(next, prob / nexts.size(), curTime + 1, totalTime, target, map, visited);
            visited.remove(next);
        }
    }
    private Map<Integer, List<Integer>> buildMap(int[][] edges) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int[] edge : edges) {
            int v1 = edge[0];
            int v2 = edge[1];
            List<Integer> list = map.getOrDefault(v1, new ArrayList<>());
            list.add(v2);
            map.put(v1, list);
            list = map.getOrDefault(v2, new ArrayList<>());
            list.add(v1);
            map.put(v2, list);
        }
        return map;
    }
}
