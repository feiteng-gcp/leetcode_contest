class Solution {
    double pos = 0;
    public double frogPosition(int n, int[][] edges, int t, int target) {
        Set<Integer>[] gra = new Set[n + 1];
        for (int i = 0; i < n + 1; i++) gra[i] = new HashSet<>();
        for (int[] e : edges) {
            gra[e[0]].add(e[1]);
            gra[e[1]].add(e[0]);
        }
        boolean[] seen = new boolean[n + 1];
        seen[1] = true;
        dfs(gra, 1, t, target, 1.0, seen);
        return pos;
    }
    void dfs(Set<Integer>[] gra, int cur, int t, int tar, double poss, boolean[] seen) {
        if (t < 0) return;
        int tot = 0;
        for (int chd : gra[cur]) {
            if (seen[chd]) continue;
            tot++;
        }
        if (tar == cur) {
            if (t == 0) {
                pos = poss;
            } else if (tot == 0) {
                pos = poss;
            } else {
                pos = 0;
            }
            return;
        }
        
        for (int chd : gra[cur]) {
            if (seen[chd]) continue;
            seen[chd] = true;
            dfs(gra, chd, t - 1, tar, poss * (1.0 / tot), seen);
        }
    }
}
