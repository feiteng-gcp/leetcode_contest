class Solution {
    public double frogPosition(int n, int[][] edges, int t, int target) {
        Map<Integer, Set<Integer>> graph = new HashMap<>();
        for (int i = 0; i < n; i++) {
            graph.put(i, new HashSet<Integer>());
        }
        for (int[] e: edges) {
            graph.get(e[0]-1).add(e[1]-1);
            graph.get(e[1]-1).add(e[0]-1);
        }
        int[] parent = new int[n];
        Arrays.fill(parent, -1);
        parent[0] = 0;
        Queue<Integer> q = new LinkedList<>();
        q.offer(0);
        
graph.get(0).add(0);
        while (!q.isEmpty()) {
            int x = q.poll();
            for (int nei: graph.get(x)) {
                if (parent[nei] == -1) {
                    parent[nei] = x;
                    q.offer(nei);
                }
            }
        }
        for (int i = 0; i < n; i++) System.out.print(parent[i] + " ");
        System.out.println(graph);
        
        int depthCount = 0;
        int prod = 1;
        int temp = target-1;
        while (temp != 0) {
            temp = parent[temp];
            depthCount++;
            prod *= graph.get(temp).size()-1;
        }
        
        if (depthCount < t && graph.get(target-1).size() > 1 || depthCount > t) return 0;
        return 1.0 / prod;
    }
}
