class Solution {
    
    private class Node {
        List<Node> childs = new ArrayList<>();
        int time;
    }
    
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        Node[] nodes = new Node[n];
        for(int i = 0 ; i < n ; i++){
            nodes[i] = new Node();
            nodes[i].time = informTime[i];
        }
        for(int i = 0 ; i < n; i++){
            if(manager[i] != -1){
                nodes[manager[i]].childs.add(nodes[i]);
            }
        }
        Deque<Node> queue = new ArrayDeque<>();
        int res = 0;
        queue.add(nodes[headID]);
        while(!queue.isEmpty()){
            int size = queue.size();
            while(size-- > 0){
                Node cur = queue.poll();
                res = Math.max(res, cur.time);
                for(Node node: cur.childs){
                    node.time += cur.time;
                    queue.add(node);
                }
            }
        }
        return res;
    }
}
