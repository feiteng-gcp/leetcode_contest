class Solution {
    Map<Integer, List<Integer>> adj = new HashMap<>();
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        if(n == 1){
            return 0;
        }
        for(int i = 0; i < n; i++){
            adj.put(i, new ArrayList<>());
        }
        for(int i = 0; i < n; i++){
            int par = manager[i];
            int child = i;
            if(par == -1){
                continue;
            }
            adj.get(par).add(child);
        }
        return dfs(headID, informTime);
    }
    public int dfs(int src, int [] informTime){
        int level = 0;
        for(int next: adj.get(src)){
            level = Math.max(level, dfs(next, informTime) + informTime[src]);
        }
        return level;
    }
}
