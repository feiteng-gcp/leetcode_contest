import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;


class Solution {
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int leader = manager[i], employer = i;
            if (!map.containsKey(leader)) {
                map.put(leader, new ArrayList<>());
            }
            map.get(leader).add(employer);
        }
        System.out.println(map);
        
return dfs(map, headID, informTime);
    }

    public int dfs(Map<Integer, List<Integer>> map, int curr, int[] informTime) {
        if (!map.containsKey(curr)) {
            return 0;
        }
        List<Integer> list = map.get(curr);
        int ans = 0;
        for (int i : list) 
{
            ans = Math.max(dfs(map, i, informTime), ans);
        }
        ans += informTime[curr];
        return ans;
    }
}
