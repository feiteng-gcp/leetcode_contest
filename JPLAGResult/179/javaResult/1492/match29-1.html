class Solution {
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        List<List<Integer>> workers = new ArrayList<>();
        for (int i = 0; i < manager.length; i++) {
            workers.add(new ArrayList<>());
        }
        for (int i = 0; i < manager.length; i++) {
            if (manager[i] == -1) continue;
            workers.get(manager[i]).add(i);
        }
        
        int maxTime = 0;
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[] {headID, 0});
        
        
while (!queue.isEmpty()) {
            int[] info = queue.remove();
            maxTime = Math.max(maxTime, (info[1]+informTime[info[0]]));
            for (int subID : workers.get(info[0])) 
{
                queue.add(new int[] {subID, info[1]+informTime[info[0]]});
            }
        }
        return maxTime;
    }
}
