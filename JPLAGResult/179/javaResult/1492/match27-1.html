class Solution {
    
    
    List<Integer> g[];
    int[] ts;
    int rs = 0;
    void go(int root,int sum){
        rs = Math.max(sum,rs);
        for(int c:g[root]){
            go(c,sum+ts[root]);
        }
        
    
}
    
    public int numOfMinutes(int n, int head, int[] fa, int[] time) {
        g = new ArrayList[n];
        ts = time;
        
for(int i=0;i<n;++i){
            g[i] = new ArrayList<>();
        }
        for(int i=0;i<fa.length;++i){
            if(i==head) continue
;
            g[fa[i]].add(i);
        }
        
        go(head,0);
        return rs;
        
        
        
    }
}
