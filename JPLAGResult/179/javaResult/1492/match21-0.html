class Solution {
    
    // build tree then bfs
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        HashMap<Integer, ArrayList<Integer>> graph = new HashMap<>();
        for(int i = 0; i < n; i++){
            if(i == headID) continue; // head has no supervisor
            
            ArrayList<Integer> sub = graph.get(manager[i]);
            if(sub == null){
                sub = new ArrayList<>();
                graph.put(manager[i], sub);
            }
            sub.add(i);
        }
        
        Deque<Integer> queue = new ArrayDeque<>();
        Deque<Integer> time = new ArrayDeque<>();
        queue.add(headID);
        time.add(0);
        
        int res = 0;
        while(!queue.isEmpty()){
            int curE = queue.poll();
            int curT = time.poll();
            
            res = Math.max(res, curT);
            if(!graph.containsKey(curE)) continue;
            for(Integer sub : graph.get(curE)){
                queue.add(sub);
                time.add(curT + informTime[curE]);
            }
        }
        return res;
    }
}
