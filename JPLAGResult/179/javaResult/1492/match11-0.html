class Solution {
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (n != headID) {
                map.putIfAbsent(manager[i], new ArrayList<>());
                map.get(manager[i]).add(i);
            }
        }
        return helper(headID, map, informTime);
    }
    
    private int helper(int current, Map<Integer, List<Integer>> map, int[] informTime) {
        if (!map.containsKey(current)) {
            return 0;
        }
        int maxTime = 0;
        for (int next : map.get(current)) {
            maxTime = Math.max(helper(next, map, informTime), maxTime);
        }
        return maxTime + informTime[current];
    }
}
