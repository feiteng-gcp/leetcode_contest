class Solution {
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        ArrayList<ArrayList<Integer>> workers = new ArrayList<>();
        for(int i = 0; i < n; i++){
            workers.add(new ArrayList<>());
        }
        
        for(int i = 0; i < n; i++){
            int m = manager[i];
            if(m != -1){
                workers.get(m).add(i);
            }
        
}
        
        Queue<Integer> time = new LinkedList<>();
        Queue<Integer> next = new LinkedList<>();
        
        time.add(0);
        next.add(headID);
        int max = 0;
        while(!time.isEmpty()){
            int cur = next.poll();
            int curTime = time.poll();
            max = Math.max(curTime, max);
            
for(int worker : workers.get(cur)){
                time.add(curTime + informTime[cur]);
                next.add(worker);
            }
        }
        
        
        return max;
    }
}
