class Solution {
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        Map<Integer, Map<Integer, Integer>> map = new HashMap<>();
        for(int i = 0; i < n; i++) map.put(i, new HashMap<>());
        for(int i = 0; i < n; i++)
        {
            int head = manager[i], employee = i, time = informTime[i];
            if(head == -1) continue;
            map.get(head).put(employee, time);
        }
        
        return informTime[headID] + dfs(headID, map);
    }
    
    int dfs(int root, Map<Integer, Map<Integer, Integer>> map)
    {
        int ans = 0;
        for(int next : map.get(root).keySet())
        {
            ans = Math.max(ans, map.get(root).get(next) + dfs(next, map));
        }
        return ans;
    }
}
