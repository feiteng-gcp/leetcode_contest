import java.util.ArrayList;
import java.util.Iterator;

class TreeNode{
    int id;
    ArrayList<Integer>list =new ArrayList<>();

    public TreeNode(int id) {
        this.id = id;
    }
}

class Solution {

    TreeNode[] treeNodes;
    int max = 0;
    int[] informTime;

    private void dfs(int root,int time){
        max=Math.max(time,max);
        Iterator<Integer>it = treeNodes[root].list.iterator();
        while (it.hasNext()){
            int p = it.next();
            dfs(p,time+informTime[root]);
        }
    
}

    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        this.informTime=informTime;
        int ans = 0;
        treeNodes = new TreeNode[n];
        
for(int i=0;i<n;i++){
            treeNodes[i]=new TreeNode(i);
        }
        for(int i=0;i<n;i++){
            if(manager[i]==-1){
                continue;
            
}
            treeNodes[manager[i]].list.add(i);
        }

        dfs(headID,0);

        return max;
    }
}
