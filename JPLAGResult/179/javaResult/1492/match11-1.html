class Solution {
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        Map<Integer, List<Integer>> subs = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (manager[i] == -1) {
                continue;
            }
            subs.computeIfAbsent(manager[i], key -> new LinkedList<>()).add(i);
        
}
        
        return dfs(subs, informTime, headID);
    }
    
    private int dfs(Map<Integer, List<Integer>> subs, int[] time, int cur) {
        int max = 0;
        for (int sub : subs.getOrDefault(cur, new LinkedList<>())) 
{
            max = Math.max(max, dfs(subs, time, sub));
        }
        
        return max + time[cur];
    }
}
