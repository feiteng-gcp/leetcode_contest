class Solution {
    Map<Integer,Set<Integer>> child = new HashMap<>();
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        for(int i=0;i<manager.length;i++) {
            child.putIfAbsent(manager[i], new HashSet<>());
            
child.get(manager[i]).add(i);
        }
        return dfs(headID, informTime);
    }
    public int dfs(int n, int[] informTime) {
        if(!child.containsKey(n)) return 0
;
        int max = 0;
        for(int c : child.get(n)) {
            max = Math.max(max, dfs(c, informTime));
        }
        return max+informTime[n];
    }
}
