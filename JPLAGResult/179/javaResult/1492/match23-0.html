class Solution {
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        HashMap<Integer, HashSet<Integer>> tree = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (manager[i] >= 0) {
                if (!tree.containsKey(manager[i])) {
                    tree.put(manager[i], new HashSet<>());
                }
                tree.get(manager[i]).add(i);
            }
        }
        return search(headID, 0, tree, informTime);
    }
    
    public int search(int i, int t, HashMap<Integer, HashSet<Integer>> tree, int[] informTime) {
        if (!tree.containsKey(i)) {
            return t;
        }
        int max = 0;
        for (int other : tree.get(i)) {
            max = Math.max(max, search(other, t+informTime[i], tree, informTime));
        }
        return max;
    }
}
