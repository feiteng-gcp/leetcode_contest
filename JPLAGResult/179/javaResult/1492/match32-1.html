class Solution {
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int m = manager[i];
            map.putIfAbsent(m, new HashSet<>());
            map.get(m).add(i);
        }
        
        PriorityQueue<int[]> pq = new PriorityQueue<>((o1, o2)->(o2[1]-o1[1]));
        
pq.offer(new int[]{headID, 0});
        
        int ans = 0;
        while (!pq.isEmpty()) {
            int[] cur = pq.poll();
            ans = Math.max(ans, cur[1]);
            for (int e : map.getOrDefault(cur[0], new HashSet<>())) 
{
                pq.offer(new int[]{e, cur[1]+informTime[cur[0]]});
            }
        }
        return ans;
    }
}
