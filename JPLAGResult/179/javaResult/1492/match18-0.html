class Solution {

    public int dfs(int h, Map<Integer, List<Integer>> child, int[] t) {
        if (!child.containsKey(h)) return 0;
        int ans = 0;
        for (int a : child.get(h)) {
            ans = Math.max(ans, dfs(a, child, t));
        }
        return ans + t[h];
    }
    
    public int numOfMinutes(int n, int headID, int[] manager, int[] informTime) {
        Map<Integer, List<Integer>> child = new HashMap<>();
        for (int i = 0; i < n; i++) {
            child.computeIfAbsent(manager[i], key -> new ArrayList<>()).add(i);
        }
        return dfs(headID, child, informTime);
    }
}
