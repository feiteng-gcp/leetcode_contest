class Solution {
  vector<int>sub[100005];
  
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
      for(int i=0;i<manager.size();++i){
        if(i==headID) continue;
        sub[manager[i]].push_back(i);
      }
      return f(n,headID,informTime);
    }
    int f(int n, int now, vector<int>& informTime){
      if(sub[now].empty()) return 0;
      int ans=0;
      for(auto& t:sub[now]){
        ans=max(ans,f(n,t,informTime));
      }
      return ans+informTime[now];
    }
};
