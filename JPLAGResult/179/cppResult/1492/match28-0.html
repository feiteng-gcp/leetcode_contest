class Solution {
public:
    // const int mxN=1e5;
    vector<int> adj[100000];
    int dfs(int u, vector<int> &t) {
        int r=0;
        for(int v : adj[u])
            r=max(r, dfs(v, t)+t[u]);
        return r;
    }
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        for(int i=0; i<n; ++i)
            adj[i].clear();
        for(int i=0; i<n; ++i)
            if(i^headID)
                adj[manager[i]].push_back(i);
        return dfs(headID, informTime);
    }
};
