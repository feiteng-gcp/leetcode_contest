class Solution {
    int f( unordered_map<int, vector<int>>& d, int cur, const vector<int> & informTime)
    {
        int m = 0;
        for (int sub : d[cur])
        {
            m = max(m, f(d, sub, informTime));
        }
        return m + informTime[cur];
    }
    
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        unordered_map<int, vector<int>> d;
        for (int i = 0; i < n; i++)
        {
            d[manager[i]].push_back(i);
        }
        return f(d, headID, informTime);
        
    }
};
