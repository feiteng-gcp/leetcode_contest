class Solution {
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<vector<int>> graph = vector<vector<int>>(n, vector<int>{});
        for(int i = 0; i < manager.size(); ++i)
        {
            if(manager[i] == -1) continue;
            graph[manager[i]].push_back(i);
        }
        
        return helper(graph, headID, 0, informTime);
    }
    
    int helper(vector<vector<int>> &graph, int id, int cost, vector<int>& informTime)
    {
        int res = cost;
        for(int i = 0; i < graph[id].size(); ++i)
        {
            res = max(res, helper(graph, graph[id][i], cost + informTime[id], informTime));
        }
        return res;
    }
};
