class Solution {
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        unordered_map<int, vector<int>> adjs;
        for(int i = 0; i < n; i++) adjs[i] = vector<int>();
        for(int i = 0; i < n; i++){
            if(manager[i] != -1) adjs[manager[i]].push_back(i);
        }
        queue<pair<int, int>> q;
        q.push({headID, 0});
        int maxT = 0;

        while(!q.empty()){
            auto node = q.front();
            q.pop();
            maxT = max(maxT, node.second);
            for(int sub : adjs[node.first]){
                q.push({sub, node.second + informTime[node.first]});
            }
        }
        return maxT;
    }
};
