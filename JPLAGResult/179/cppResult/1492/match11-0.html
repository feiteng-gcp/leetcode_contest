class Solution {
public:
    map<int, std::vector<int>> go;
    int ans;
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        ans = 0;
        go.clear();
        for (int i = 0; i < manager.size(); ++i) {
            int fa = manager[i];
            if (fa == -1) continue;
            go[fa].push_back(i);
        }
        
        dfs(headID, 0, informTime);
        return ans;
    }
    
    void dfs(int u, int cost, vector<int> &time) {
        ans = max(ans, cost);
        for (auto v : go[u]) {
            dfs(v, cost + time[u], time);
        }
    }
};
