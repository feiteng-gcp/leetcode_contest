class Solution {
public:
    int helper(int headID, vector<vector<int>>& vt, vector<int>& informTime)
    {
        int opt=0;
        for(auto i: vt[headID])
        {
            opt=max(opt, helper(i, vt, informTime));   
        }
        return opt+informTime[headID];
    }
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<vector<int>> vt(n, vector<int>());
        
        for(int i=0; i<n; i++)
        {
            if(i != headID)
               vt[manager[i]].push_back(i);
        }
        
        return helper(headID, vt, informTime);
    }
};
