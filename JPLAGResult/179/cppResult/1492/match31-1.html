class Solution {
public:
    #define N 100010
    #define MP make_pair
    vector<pair<int,int> >G[N];
    int dfs(int u,int f){
        int sz=0;
        for (int i=0;i<(int)G[u].size();++i){
            int v=G[u][i].first,w=G[u][i].second;
            if (v==f) continue;

            sz=max(sz,w+dfs(v,u));
        }
        return sz;
    }
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        for (int i=0;i<n;++i) G[i].clear();
        int rt=-1;
        for (int i=0;i<(int)manager.size();++i){
            if (manager[i]==-1) rt=i;
            else G[manager[i]].push_back(MP(i,informTime[manager[i]]));            
        }
        return dfs(rt,-1);
    }
};
