class Solution {
public:
    int dfs(int u, int t, vector<vector<int>> &e, vector<int>& informTime) {
      int ans = t;
      for(auto v: e[u]) {
        ans = max(ans, dfs(v, t + informTime[u], e, informTime));
      }
      return ans;
    }
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
      vector<vector<int>> e(n, vector<int>());
      for(int i = 0; i < n; i++) {
        if(manager[i] == -1) continue;
        e[manager[i]].push_back(i);
      }
      return dfs(headID, 0, e, informTime);
    }
};
