class Solution {
public:
    vector<int> son[110000];
    
    int get_ans(int root, vector<int>& informTime ){
    int tmp=0;
    for (int i=0;i<son[root].size();i++)
        tmp = max( tmp , get_ans( son[root][i] , informTime   ));
    return tmp+informTime[root];
    }
    
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
    for (int i=0;i<n;i++)
        son[i].clear();
    for (int i=0;i<n;i++)
        if ( i !=headID )
            son[ manager[i] ].push_back( i );
    return get_ans( headID , informTime );
    }
};
