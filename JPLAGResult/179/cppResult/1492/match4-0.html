#define pb push_back
#define fst first
#define snd second

typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;
class Solution {
public:
  int numOfMinutes(int n, int head, vector<int>& man, vector<int>& time) {
    vector<vector<int>> g(n);
    for(int i=0;i<n;i++){
      if(man[i]!=-1) g[man[i]].pb(i);
    }

    int res=0;
    queue<pii> q;
    q.push({head,0});
    while(!q.empty()){
      auto pr=q.front();
      q.pop();
      int k=pr.fst,d=pr.snd;
      res=max(res,d);
      for(int l:g[k]){
        q.push({l,d+time[k]});
      }
    }
    return res;
  }
};
