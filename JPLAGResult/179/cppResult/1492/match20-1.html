class Solution {
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<unordered_set<int>> children(n);
        for (int i = 0; i < n; ++i) {
            if (manager[i] != -1) children[manager[i]].emplace(i);
        }
        
        // <-time take to get noticed, employee-id>
        priority_queue<pair<int, int>> pq;
        pq.emplace(-0, headID);
        

        int mn_time = 0;
        
        while (!pq.empty()) {
            auto [t, id] = pq.top();
            pq.pop();

            mn_time = max(mn_time, -t);
            
            for (auto c : children[id]) {
                pq.emplace(t - informTime[id], c);
            }
        }
        
        return mn_time;
    }
};
