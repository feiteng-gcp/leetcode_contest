class Em {
public:
  int id = -1;
  int time = -1;
  Em() = default;
  Em(int i, int t) : id(i), time(t) {};
};

class Solution {
public:
  int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
    vector<vector<int>> sub(n);
    for (int i = 0; i < manager.size(); ++i) {
      // manager[i] is the direct manager of the i-th employee
      if (manager[i] != -1) {
        sub[manager[i]].push_back(i);
      }
    }
    queue<Em> q;
    q.emplace(headID, 0);
    int ans = 0;
    while (!q.empty()) {
      auto curr = q.front();
      q.pop();
      ans = max(ans, curr.time);
      for (auto next_id: sub[curr.id]) {
        q.emplace(next_id, curr.time + informTime[curr.id]);
      }
    }
    return ans;
  }
};
