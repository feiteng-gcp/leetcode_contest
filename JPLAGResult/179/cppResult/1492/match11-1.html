class Solution {
public:

    unordered_map<int, vector<int>> child;
    int time;
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        time = 0;
        
        for(int i=0;i<n;i++) {
            child[manager[i]].push_back(i);

        }
        dfs(informTime, headID, 0);
        return time;
    }
    void dfs(vector<int>& informTime, int cur, int totaltime) {
        time = max(time, totaltime);
        for(auto c: child[cur]) {
            dfs(informTime, c, totaltime+informTime[cur]);
        }
    }
};
