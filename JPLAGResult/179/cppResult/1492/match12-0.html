class Solution {
public:
    unordered_map<int,vector<int>> note;
    int ans;
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        int head;
        for(int i=0;i<n;i++) {
            note[manager[i]].push_back(i);
            if (manager[i] == -1)
                head = i;
        }
        dfs(informTime,head,0);
        return ans;
    }
    void dfs(vector<int>& informTime,int cur,int t){
        ans=max(ans,t);
        for(auto &x:note[cur]){
            dfs(informTime,x,t+informTime[cur]);
        }
    }
};
