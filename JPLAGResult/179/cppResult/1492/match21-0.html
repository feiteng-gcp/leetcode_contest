class Solution {
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<vector<int>> children(n);
        for (int i = 0; i < n; i++) {
            if (i == headID) {
                continue;
            }
            children[manager[i]].push_back(i);
        }
        return rec(headID, children, informTime);
    }
    
    int rec(int cur, vector<vector<int>>& children, vector<int>& informTime) {
        int ans = 0;
        for (int i = 0; i < children[cur].size(); i++) {
            ans =  max(ans, rec(children[cur][i], children, informTime));
        }
        ans += informTime[cur];
        return ans;
    }
};
