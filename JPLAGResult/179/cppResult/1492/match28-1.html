class Solution {
public:
    vector<int> arr[110000];
    int dfs(int id, vector<int>& informTime)
    {
         int ans = 0;
        for (int nid : arr[id])
        {
            ans = max(ans, dfs(nid, informTime));
        }

        return ans + informTime[id];
    }
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        for (int i = 0; i < n; ++i)
            if (manager[i] != -1)
                arr[manager[i]].push_back(i);
        int ans = dfs(headID, informTime);
        for (int i = 0; i < n; ++i) arr[i].clear();
        return ans;
    }
};
