class Solution {
public:
    vector<vector<int>> g;
    int dfs(int u,int sum,vector<int>& it)
    {
        int ans=sum;
        for(int v:g[u])
        {
            ans=max(ans,dfs(v,sum+it[u],it));
        }
        return ans;
    }
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        g.assign(n,vector<int>());
        for(int i=0;i<n;i++)
            if(manager[i]!=-1) g[manager[i]].push_back(i);
        return dfs(headID,0,informTime);
    }
};
