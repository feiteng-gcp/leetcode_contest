class Solution {
private:
    int dfs(int i, vector<vector<int>>& subor, vector<int>& informTime) {
        int mx = 0;
        for (auto it:subor[i]) {
            mx = max(mx,dfs(it,subor,informTime));
        }
        return informTime[i]+mx;
    }
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<vector<int>> subor(n);
        for (int i=0; i<n; i++) {
            if (i!=headID) {
                subor[manager[i]].push_back(i);
            }
        }
        return dfs(headID, subor, informTime);
    }
};
