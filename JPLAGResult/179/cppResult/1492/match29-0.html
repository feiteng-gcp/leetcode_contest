class Solution {
public:
    
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<vector<int>> next(n);
        vector<int> dis(n, INT_MAX);
        int root = -1;
        for (int i=0; i<manager.size(); ++i) {
            if (manager[i] == -1) root = i;
            else next[manager[i]].push_back(i);
        }
        if (root == -1) return 0;
        dis[root] = 0;
        deque<int> que;
        que.push_back(root);
        int ans = 0;
        while (!que.empty()) {
            int s = que.front();
            que.pop_front();
            for (auto t: next[s]) {
                dis[t] = dis[s] + informTime[s];
                que.push_back(t);
                ans = max(ans, dis[t]);
            }
        }
        return ans;
    }
};
