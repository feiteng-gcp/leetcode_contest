class Solution {
public:
    unordered_map<int, vector<int>> f;
    int res;
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        res = 0;
        for (int i = 0; i < n; i++) {
            f[manager[i]].push_back(i);
        }
        dfs(headID, informTime, 0);
        return res;
    }
    void dfs(int h, vector<int>& t, int cur) {
        if (t[h] == 0) {
            res = max(res, cur);
            return;
        }
        cur = cur + t[h];
        for (auto m: f[h]) {
            dfs(m, t, cur);
        }
    }
};
