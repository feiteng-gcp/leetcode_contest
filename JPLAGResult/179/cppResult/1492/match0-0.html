class Solution {
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        unordered_map<int, vector<int>> manager2child;
        for (int i = 0; i < manager.size(); i++) { 
            manager2child[manager[i]].push_back(i);
        }
        vector<int> minDist(n, INT_MAX);
        queue<int> q;
        vector<bool> visited(n, false);
        q.push(headID);
        minDist[headID] = 0;
        while (!q.empty()) { 
            int len = q.size();
            for (int i = 0; i < len; i++) { 
                int id = q.front();
                q.pop();
                if (visited[id] == true) { 
                    continue;
                }
                else { 
                    visited[id] = true;
                    for (int child : manager2child[id]) { 
                        minDist[child] = minDist[id] + informTime[id];
                        q.push(child);
                    }
                }
            }
        }
        int ans = 0;
        for (int i : minDist) { 
            ans = max(ans, i);
        }
        return ans;
        
    }
};
