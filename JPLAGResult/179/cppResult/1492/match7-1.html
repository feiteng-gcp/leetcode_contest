class Solution {
private:

    int calc(int now, vector<vector<int>>& child, vector<int>& informTime) {
        int res = 0;
        for (auto nex: child[now]) {
            res = max(res, calc(nex, child, informTime));
        }
        return res + informTime[now];
    }
public:

    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<vector<int>> child(n, vector<int>());
        for (int i = 0; i < n; i++) {
            if (manager[i] == -1) continue;
            child[manager[i]].push_back(i);
        }
        return calc(headID, child, informTime);
    }
};
