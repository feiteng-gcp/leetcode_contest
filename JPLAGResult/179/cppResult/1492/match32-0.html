class Solution {
public:
    map<int, vector<int>> m_worker;
    int informAll(int workerID, vector<int>& informTime)
    {
        int max_time = 0;
        for (auto it = m_worker[workerID].begin(); it != m_worker[workerID].end(); ++it)
        {
            max_time = max(max_time, informAll(*it, informTime));
        }
        return max_time + informTime[workerID];
    }
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        m_worker.clear();
        for (int i = 0; i < manager.size(); ++i)
        {
            m_worker[manager[i]].push_back(i);
        }
        return informAll(headID, informTime);
    }
};
