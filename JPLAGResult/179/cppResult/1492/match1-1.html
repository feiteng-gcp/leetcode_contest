class Solution {
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<vector<int>> m(n);
        
        for (int i = 0; i < manager.size(); ++i) {
            if (manager[i] >= 0) m[manager[i]].push_back(i);
        }
        
        vector<int> t(n, 0);
        list<int> q, q2;
        q.push_back(headID);
        while (!q.empty()) {
            for (auto n: q) {
                for (auto l: m[n]) {
                    t[l] = t[n] + informTime[n];
                    q2.push_back(l);
                }
            }
            swap(q, q2);
            q2.clear();
        }
        
        int ret = 0;
        for (auto t1: t) ret = max(t1, ret);
        return ret;
    }
};
