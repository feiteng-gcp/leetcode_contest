class Solution {
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<vector<int> > Sub(n);
		vector<int> Time(n);
		int Result = 0;
		int i;
		
		for(i = 0; i < n; i++)
		{
			if(manager[i] != -1)
			{
				//cout << manager[i] << " " << i << endl;
				Sub[manager[i]].push_back(i);
			}
		}
		
		queue<int> Q;
		Q.push(headID);
		

		while(Q.size())
		{
			int T = Q.front();
			Q.pop();
			
			for(auto it : Sub[T])
			{
				Time[it] = Time[T] + informTime[T];
				//cout << it << "" << Time[it] << endl;
				Result = max(Result, Time[it]);
				Q.push(it);
			}
		}
		
		return Result;
    }
};
