class Solution {
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<vector<int>> adj(n);
        for(int i=0;i<n;i++) {
            if(manager[i]!=-1) adj[manager[i]].push_back(i);
        }

        queue<vector<int>> q;
        int res = 0;
        q.push({headID, 0});
        while(!q.empty()) {
            auto v = q.front(); q.pop();
            // cout<<v[0]<<" "<<v[1]<<endl;
            res = max(res, v[1]);
            for(int j: adj[v[0]]) {
                q.push({j, informTime[v[0]]+v[1]});
            }
        }
        return res;
    }
};
