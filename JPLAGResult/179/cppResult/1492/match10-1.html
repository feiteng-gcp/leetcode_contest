class Solution {
public:

    vector<vector<int>> e;
    int res=0;
    void dfs(int r, int t, vector<int>& informTime){
        if(t>res)
            res=t;
        for(int x:e[r]){
            dfs(x,t+informTime[r],informTime);
        }
    }

    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        e.resize(n);
        for(int i=0;i<n;++i)
            if(i!=headID)
                e[manager[i]].push_back(i);
        dfs(headID,0,informTime);
        return res;
    }
};
