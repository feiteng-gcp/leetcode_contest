class Solution {
public:
    

    vector<int> adj[100005];
    
    int rec(int cur, vector<int>& time) {
        if (adj[cur].size() == 0) return 0;
        int mn = 0;
        for (auto child: adj[cur]) {
            mn = max(mn, rec(child, time));
        }
        return mn+time[cur];
    }
    

    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        for (int i = 0; i < n; i++) {
            if (manager[i] != -1) {
                adj[manager[i]].push_back(i);
            }
        }
        
        return rec(headID, informTime);
    }
};
