class Solution {
public:
    int dfs(int u, vector<vector<int>>& children, vector<int>& time) {
        int res = 0;
        for (int v : children[u]) {
            res = max(res, dfs(v, children, time));
        }
        return res + time[u];
    }
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        vector<vector<int>> children(n);
        for (int i = 0; i < n; ++i) {
            if (i != headID)
                children[manager[i]].push_back(i);
        }
        int res = dfs(headID, children, informTime);
        return res;
    }
};
