#define pb push_back

typedef long long ll;
#define REP(a,b) for(long long (a) = 0;(a) < (b);++(a))
#define ALL(x) (x).begin(),(x).end()


class Solution {
public:
    vector<int> vertexs[200];
    long double prob[200] = {};
    int dist[200] = {};
    int calc_dist(int now,int back) {
        long double hoge = prob[now];
        int nya = vertexs[now].size();
        if (back != -1) {
            nya--;
        }
        if (nya == 0) return 0;
        hoge /= (long double)nya;
        for (auto x : vertexs[now]) {
            if (x == back) continue;
            dist[x] = dist[now] + 1;
            prob[x] = hoge;
            calc_dist(x, now);
        }
        return 0;
    }
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        REP(i, n) {
            vertexs[i+1].clear();
            dist[i + 1] = 0;
            prob[i + 1] = 0;
        }
        if (n == 1) {
            return 1.0L;
        }

        prob[1] = 1.0;
        REP(i, edges.size()) {
            REP(q, 2)
                vertexs[edges[i][q]].push_back(edges[i][(q + 1) % 2]);
        }
        calc_dist(1,-1);
        if (vertexs[target].size() == 1&&dist[target] != 0) {
            if (dist[target] > t) {
                return 0;
            }
        }
        else {
            if (dist[target] != t) {
                return 0;
            }
        }
        return prob[target];
    }
};
