class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<unordered_set<int>> tos(n);
        for (auto& e : edges) {
            tos[e[0] - 1].emplace(e[1] - 1);
            tos[e[1] - 1].emplace(e[0] - 1);
        }
        
        vector<double> p(n, 0);
        
        queue<int> q;
        q.emplace(0);
        p[0] = 1.0;
        
        unordered_set<int> visited = {0};

        int cur_time = 0;
        while (cur_time < t && !q.empty()) {
            for (int i = q.size(); i > 0; --i) {
                auto e = q.front();
                q.pop();

                vector<int> to_add;
                for (auto& to : tos[e]) {
                    if (visited.emplace(to).second) {
                        to_add.emplace_back(to);
                    }
                }
                    
                if (to_add.size() > 0) {
                    double pp = p[e] / to_add.size();
                    for (auto id : to_add) {
                        p[id] = pp;
                        q.emplace(id);
                    }
                    p[e] = 0.0;
                }
            }
            
            ++cur_time;
        }
        
        return p[target - 1];
    }
};
