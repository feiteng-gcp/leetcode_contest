class Solution {
public:
  double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
    vector<vector<int>> g(n + 1);
    for (const auto& e : edges) {
      g[e[0]].push_back(e[1]);
      g[e[1]].push_back(e[0]);
    }
    vector<int> d(n + 1);
    vector<double> f(n + 1);
    vector<bool> leaf(n + 1, true);
    d[1] = 0;
    f[1] = 1.0;
    function<void(int, int)> go = [&](int x, int fr) {
      int sz = g[x].size();
      if (x != 1) --sz;
      for (const auto y : g[x]) {
        if (y != fr) {
          d[y] = d[x] + 1;
          f[y] = f[x] / sz;
          leaf[x] = false;
          go(y, x);
        }
      }
    };
    go(1, 0);
    if (leaf[target]) return t >= d[target] ? f[target] : 0.0;
    return t == d[target] ? f[target] : 0.0;
  }
};
