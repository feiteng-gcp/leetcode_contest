class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        if(n == 1)return 1.0;
        vector<double> prob(n+1,0);
        prob[1] = 1;
        vector<vector<int>> cs(n+1, vector<int>());
        for(auto e : edges){
            cs[e[0]].push_back(e[1]);
            cs[e[1]].push_back(e[0]);
        }
        vector<int> vis(n+1, 0);
        vis[1] = 1;
        int vis_count = 1;
        for(int i = 0; i < t; i++){
            vector<double> new_prob(n+1, 0);
            for(int j = 1; j <= n; j++){
                if(prob[j] > 0 && !cs[j].empty() ){
                    int ct = 0;
                    for(int c : cs[j]){
                        if(!vis[c]){
                            ct++;
                        }
                    }
                    for(int c : cs[j]){
                        if(!vis[c]){
                            new_prob[c] = prob[j] / ct;
                            vis[c] = 1;
                            vis_count++;
                        }    
                    }
                    if(ct)prob[j] = 0;
                }
            }
            for(int j = 1; j <= n; j++)new_prob[j] = max(prob[j], new_prob[j]);
            prob = new_prob;
            if(vis_count == n)break;
            //for(int k = 1; k <= n; k++)cout << prob[k] << " ";
            //cout << endl;
        }
        
        return prob[target];
    }
};
