class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        
        vector<vector<int>>graph(n);
        
        for(int i=0;i<edges.size();i++)
        {
            graph[edges[i][0] - 1].push_back( edges[i][1] -1);
            graph[edges[i][1] - 1].push_back( edges[i][0] -1);
        }
        
        vector<bool> isVisited(n,false);
        vector<bool> isParent(n,false);
        vector<pair<int,double>> prob(n);
        
        findProb(graph, isParent, isVisited, prob,0,0,1);
        
        if( isParent[target-1])
        {
            if(prob[target-1].first == t)
                return prob[target-1].second;
            else
                return 0;
        }
        else
        {
            if(prob[target-1].first <=t)
                return prob[target-1].second;
            else
                return 0;
        }
        
        return 0;
    }
    
    void findProb(vector<vector<int>> &graph, vector<bool> &isParent, vector<bool> &isVisited, vector<pair<int,double>> &prob, int src, int time, double pb)
    {
        isVisited[src] = true;
        prob[src] = make_pair(time,pb);
        
        int child = 0;
        for(int i=0;i<graph[src].size();i++)
        {
            if(!isVisited[graph[src][i]])
                child++;
        }
        
        if(child > 0)
            isParent[src] = true;
        
        for(int i = 0; i< graph[src].size();i++)
        {
            if(!isVisited[graph[src][i]])
                findProb(graph, isParent,isVisited, prob, graph[src][i], time+1, pb/child);
        }
    }
};
