class Solution {
public:
    void dfs(int u, int f, vector<vector<int>> &G, int step, int target, double &ans, double cur) {
        if(step == 0 || (G[u].size() == 1 && f != 0) || (G[u].size() == 0)) {
            if(u == target) ans = cur;
            return;
        }
        for(auto v: G[u]) {
            if(v == f)continue;
            dfs(v, u, G, step-1, target, ans, cur*1.0/(G[u].size()-(f!=0)));
        }
    }
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> G(n+1);
        for(auto es: edges) {
            G[es[0]].push_back(es[1]);
            G[es[1]].push_back(es[0]);
        }
        double ans = 0;
        dfs(1, 0, G, t, target, ans, 1.0);
        return ans;
    }
};
