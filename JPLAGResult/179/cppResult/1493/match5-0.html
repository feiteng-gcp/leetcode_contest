class Solution {
    map<int, set<int>> tree;
    vector<int> path;
    bool dfs(int root, int pre, int target){
        path.push_back(root);
        if(root == target){
            path.pop_back();
            return true;
        }
        
        for(auto& s: tree[root]){
            if(s == pre) continue;
            
            if(dfs(s, root, target)){
                return true;
            }
        }
        path.pop_back();
        return false;
    }
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        for(auto& v: edges){
            tree[v[0]].insert(v[1]);
            tree[v[1]].insert(v[0]);
        }
        dfs(1, -1, target); 
        
        //less
        if(t < path.size()){
            return 0;
        }
        
        
        //more, no leaf
        if(tree[target].size() > 1 && t > path.size()){
            return 0;
        }
        
        //root
        if(target == 1 && tree[target].size() > 0 && t > path.size()){
            return 0;
        }
        
        double ans = 1.0;
        for(auto v: path){
            int n = tree[v].size();
            if(v != 1){
                n--;
            }
            ans = ans / n;
        }
        
        return ans;
        
    }
};
