class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> nxt(n + 1, vector<int>());
        for (auto edge: edges) {
            nxt[edge[0]].push_back(edge[1]);
            nxt[edge[1]].push_back(edge[0]);
        }
        vector<int> father(n + 1, 1), vis_time(n + 1, 0), cnt(n + 1, 1);
        queue<int> q;
        q.push(1);
        while (!q.empty()) {
            int id = q.front(); q.pop();
            int child_num = (id == 1? nxt[id].size(): nxt[id].size() - 1);
            for (int child: nxt[id]) {
                if (child == father[id]) continue;
                father[child] = id;
                vis_time[child] = vis_time[id] + 1;
                cnt[child] = child_num * cnt[id];
                // cout << "cnt " << child << ": " << cnt[child] << endl;
                q.push(child);
            }
        }
        
        if (t > vis_time[target]) {
            if ((target == 1 && nxt[1].size() == 0) || (target != 1 && nxt[target].size() == 1)) return 1.0 / cnt[target];
            else return 0.0;
        } else if (t < vis_time[target]) {
            return 0.0;
        } else {
            return 1.0 / cnt[target];
        }
    }
};
