class Solution {
public:
    double dfs(unordered_map<int, unordered_set<int>>& m, int node, int t, int target, double prob) {
      if (node == target && t == 0) {
        return prob;
      }
      if (t == 0) return 0;
      std::vector<int> ivec(m[node].begin(), m[node].end());
      if (ivec.empty() && node == target) {
        return prob;
      }
      double re = 0;
      prob /= m[node].size();
      for (auto n : ivec) {
        m[node].erase(n);
        m[n].erase(node);
        re += dfs(m, n, t - 1, target, prob);
        m[node].insert(n);
        m[n].insert(node);

      }
      cout << node << " " << re << endl;
      return re;
    }
      
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
      unordered_map<int, unordered_set<int>> map;
      for(auto & pr : edges) {
        map[pr[0]].emplace(pr[1]);
        map[pr[1]].emplace(pr[0]);
      }
      return dfs(map,1, t, target, 1);
    }
};
