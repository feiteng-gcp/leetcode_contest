class Solution {
public:
    unordered_map<int, vector<int>> node2neighs;
    
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        for (vector<int>& edge : edges) { 
            node2neighs[edge[0]].push_back(edge[1]);
            node2neighs[edge[1]].push_back(edge[0]);
        }
        vector<bool> visited(n + 1, false);
        queue<pair<int, double>> q;
        q.push({1, 1});
        visited[1] = true;
        
        int time = 0;
        vector<pair<int, double>> savedState;
        while (!q.empty() && time < t) { 
            int len = q.size();
            for (int i = 0; i < len; i++) { 
                pair<int, double> p = q.front();
                q.pop();
                int pos = p.first;
                double prob = p.second;
                int countUnVisited = 0;
                for (int i = 0; i < node2neighs[pos].size(); i++) { 
                    if (visited[node2neighs[pos][i]] == false) { 
                        countUnVisited++;
                    }
                }
                if (countUnVisited == 0) { 
                    savedState.push_back(p);
                }
                else { 
                    for (int neigh : node2neighs[pos]) { 
                        if (visited[neigh]) { 
                            continue;
                        }
                        visited[neigh] = true;
                        q.push({neigh, prob * 1 / countUnVisited});
                    }
                    visited[pos] = true;
                }
            }
            time++;
        }
        while (!q.empty())  {
            savedState.push_back(q.front());
            q.pop();
        }
        for (auto p : savedState) { 
            if (p.first == target) { 
                return p.second;
            }
        }
        return 0;
    }
};
