class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        if (edges.size() == 1) {
            return 1.0;
        }
        vector<vector<int>> nns(n);
        for (auto e : edges) {
            nns[e[0] - 1].push_back(e[1] - 1);
            nns[e[1] - 1].push_back(e[0] - 1);
        }
        vector<vector<int>> children(n);
        vector<int> parents(n, -1);
        {
            vector<int> q;
            q.push_back(0);
            vector<bool> used(n, false);
            int pos = 0;
            while (pos < q.size()) {
                int x = q[pos];
                used[q[pos]] = true;
                for (auto y : nns[q[pos]]) {
                    if (!used[y]) {
                        children[x].push_back(y);
                        parents[y] = x;
                        q.push_back(y);
                        //cout << x << ' ' << y << endl;
                    }
                }
                ++pos;
            }
        }
        vector<double> probs(n, 0);
        probs[0] = 1.0;
        for (int i = 0; i < t; ++i) {
            vector<double> next_probs(n, 0);
            for (int x = 0; x < n; ++x) {
                if (children[x].empty()) {
                    next_probs[x] += probs[x];
                } else {
                    for (auto y : children[x]) {
                        next_probs[y] += probs[x] / children[x].size();
                    }
                }
            }
            probs = next_probs;
            /*for (auto x : probs) {
                cout << x << ' ';
            }
            cout << endl;*/
        }
        return probs[target - 1];
    }
};
