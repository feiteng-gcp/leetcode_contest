class Solution {
public:
    vector<int> nxt[102];
    bool flag = false;
    double ans = 0.0;
    void dfs(int now, int dep, int t, int target, long long tmp, int pre){
        //cout<<now<<'\t'<<dep<<endl;
        if(now == target){
            flag = true;
            if(dep == t || (pre==-1 && nxt[now].size()==0) || (pre!=-1 && nxt[now].size()==1)){
                ans = 1.0 / tmp;
            }else{
                ans = 0.0;
            }
            return;
        }    
        if(flag || dep >= t) return;
        for(auto &x: nxt[now]) if(x!=pre){
            dfs(x, dep+1, t, target, tmp * (nxt[now].size() - (now == 1 ? 0 : 1)), now);
        }
    }
    
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        for(auto &x: edges){
            nxt[x[0]].push_back(x[1]);
            nxt[x[1]].push_back(x[0]);
        }
        flag = false;
        dfs(1, 0, t, target, 1LL, -1);
        return !flag ? 0: ans;
    }
};
