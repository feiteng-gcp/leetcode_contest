class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> tree(n + 1);
        for (auto &e: edges) {
            tree[e[1]].push_back(e[0]);
            tree[e[0]].push_back(e[1]);
        }
        
        vector<double> p(n + 1, 0);
        p[1] = 1;
        list<int> q, q2;
        q.push_back(1);
        for (int i = 0; i < t; ++i) {
            for (auto n: q) {
                vector<int> nm;
                for (auto l: tree[n]) {
                    if (p[l] > 0) continue;
                    nm.push_back(l);
                }
                if (nm.empty()) q2.push_back(n);
                else {
                    for (auto l: nm) {
                        if (p[l] > 0) continue;
                        p[l] = (double)p[n] / nm.size();
                        //printf("%d %d %lf\n", n, l, p[l]);
                        q2.push_back(l);
                    }
                    p[n] = 2;
                } 
            }
            swap(q, q2);
            q2.clear();
        }
        
        return p[target] == 2?0:p[target];
    }
};
