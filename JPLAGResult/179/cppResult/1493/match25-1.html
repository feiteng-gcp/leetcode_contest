class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        // simulation
        // loop
        // it's a tree

        unordered_map<int, vector<int>> neighbor;
        
        for(const auto &edge: edges) {
            neighbor[edge[0]].push_back(edge[1]);
            neighbor[edge[1]].push_back(edge[0]);
        }
        
        queue<tuple<int,int, double>> q; // {node, parent, prob}
        q.push({1, -1, 1});
        

        for(int step=0;step<t;step++) {
            int qsize = q.size();
            for(int i=0;i<qsize;i++) {
                auto [node, parent, prob] = q.front();
                q.pop();
                
                int countnei = 0;
                if(node!=parent) {
                    for(auto nei:neighbor[node]) {
                        if(nei!=parent) {
                            if(parent==-1) {
                                q.push({nei, node, prob/(neighbor[node].size())});
                            } else {
                                q.push({nei, node, prob/(neighbor[node].size()-1)});
                            }
                            countnei++;
                        }
                    }
                }
                if(countnei==0) {
                    // stay in same node
                    q.push({node, node, prob});
                }
            }
        }
        
        unordered_map<int, double> m;
        while(!q.empty()) {
            auto [node, parent, prob] = q.front();
            q.pop();
            m[node]+=prob;
        }
        
        return m[target];
    }
};
