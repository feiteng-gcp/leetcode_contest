class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        if( n == 1){
            return 1;
        }
        
        vector < vector < int> > adj(n+1);
        for( auto x: edges){
            adj[x[0]].push_back(x[1]);
            adj[x[1]].push_back(x[0]);
        }
        
        stack < int > stk;
        stk.push(1);
        vector< bool > vis( n+1, false);
        vector< int > h(n+1, 0);
        vector< int > leaf(n+1, 0);
        vector< double > prob(n+1);
        

        prob[1] = 1;
        
        while(!stk.empty()){
            int x = stk.top();
            stk.pop();
            vis[x] = true;
            for( auto y: adj[x]){
                if(vis[y] == false){
                    stk.push(y);
                    h[y] = h[x] + 1;
                    if( x == 1 )
                        prob[y] = prob[x]*(1.0/double(adj[x].size()));
                    else
                        prob[y] = prob[x]*(1.0/double(adj[x].size()-1));
                    if( adj[y].size() == 1){
                        leaf[y] = 1;
                    }
                }
            }
        }
        
        if(leaf[target] == true){
            if( t >= h[target]){
                return prob[target];
            }
            else
                return 0;
        }
        else{
            if( t == h[target]){
                return prob[target];
            }
            else
                return 0;
        }
        return 0;
        
    }
};
