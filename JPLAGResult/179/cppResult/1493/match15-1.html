class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> g(n + 1);
        for (auto& e : edges) {
            g[e[0]].push_back(e[1]);
            g[e[1]].push_back(e[0]);
        }
        return dfs(t, 1, g, target, -1);
    }
    
    double dfs(int left_time, int cur, vector<vector<int>>& g, int target, int from) {
        if (left_time == 0) {
            if (cur == target) {
                return 1.0;
            }
            return 0.0;
        }
        int next_count = 0;
        double prob = 0.0;
        for (int next : g[cur]) {
            if (next == from) continue;
            next_count++;
            prob += dfs(left_time - 1, next, g, target, cur);
        }
        if (next_count == 0) {
            return cur == target ? 1.0 : 0.0;
        }
        return prob / next_count;
    }
};
