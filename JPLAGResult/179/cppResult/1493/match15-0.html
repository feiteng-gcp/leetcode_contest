class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> ng(n + 1);
        vector<bool> visited(n + 1, 0);
        for (auto &e : edges)
        {
            ng[e[0]].push_back(e[1]);
            ng[e[1]].push_back(e[0]);
        }
        //for (int i = 1; i <= n; ++i)
            //cout << ng[i].size() << endl;
        return dfs(1, visited, ng, 1, t, target);
    }
    
    double dfs(int node, vector<bool>& visited, vector<vector<int>>& ng, double prob, int t, int target) {
        int count = 0;
        for (auto &n : ng[node])
        {
            if (!visited[n]) ++count;
        }
        if (node == target)
        {
            if (t == 0 || count == 0) return prob;
            else return 0;
        }
        if (t == 0) return 0;
        visited[node] = 1;
        //cout << node << endl;
        for (auto &n : ng[node])
        {
            if (visited[n]) continue;
            //cout << n << " " << node << " " << prob << " " << ng[node].size() << endl;
            double p = dfs(n, visited, ng, prob / count, t - 1, target);
            if (p > 0) return p; 
        }
        return 0;
    }
};
