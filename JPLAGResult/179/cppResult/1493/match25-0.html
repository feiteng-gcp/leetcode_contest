class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        unordered_map<int, vector<int>> adjs;
        for(auto edge : edges){
            adjs[edge[0]].push_back(edge[1]);
            adjs[edge[1]].push_back(edge[0]);
        }
        vector<bool> vi(n + 1, false);
        queue<pair<int, double>> q;
        q.push({1, 1.0});
        vi[1] = true;
        int time = 0;
        while(!q.empty()){
            int n = q.size();
            for(int i = 0; i < n; i++){
                auto node = q.front();
                q.pop();
                int remain = 0;
                for(int c : adjs[node.first]) if(!vi[c]) remain++;
                if(node.first == target){
                    if(t == time) return node.second;
                    else if(t > time){
                        return remain == 0 ? node.second : 0.0;
                    }
                    else 0.0;
                }
                if(remain > 0){
                    double sub = node.second / remain;
                    for(int c : adjs[node.first]){
                        if(!vi[c]){
                            q.push({c, sub});
                            vi[c] = true;
                        }
                    }
                }
            }
            time++;
        }
        return 0.0;
    }
};
