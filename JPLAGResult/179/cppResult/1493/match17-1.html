class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> nexts(n + 1);
        for(auto& v : edges)
        {
            nexts[v[0]].push_back(v[1]);
            nexts[v[1]].push_back(v[0]);
        }
        vector<double> gailv(n + 1, 0.0);
        vector<int> vis(n + 1, 0);
        gailv[1] = 1.0;
        vector<int> v1, v2;
        v1.push_back(1);
        while(t--)
        {
            for(int node : v1)
            {
                vis[node] = 1;
                double childs = 0.0;
                for(int ne : nexts[node]) if(!vis[ne]) childs += 1.0;
                if(childs < 1e-5) continue;
                double nextp = gailv[node]/childs;
                gailv[node] = 0.0;
                for(int ne : nexts[node])
                {
                    if(!vis[ne])
                    {
                        gailv[ne] = nextp;
                        v2.push_back(ne);
                    }
                }
            }
            swap(v1, v2);
            v2.clear();
        }
        return gailv[target];
    }
};
