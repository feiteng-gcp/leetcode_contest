const int N = 111;
class Solution {
public:

   double prob = 0;
   vector<int> v[N];
   int target;
   int t;
   void dfs(int s, int p, int lvl, double time){
      int child = 0;
      for(auto i: v[s]){
         if(i != p){
            child++;
         }
      }
      if(s == target){
          if(child == 0){
              if(t >= lvl){
                  prob = time;
                  return;
              }
          }
          else{
              if(t == lvl){
                  prob = time;
                  return;
              }
          }

          prob = 0;
          return;
      }
       // cout << "S = " << s << " lvl = " << lvl << " time = " << time << " child = " << child << endl;
      for(auto i: v[s]){
         if(i == p)continue;
         dfs(i, s, lvl+1, time * (1.0000 / (child)));
      }
   }

    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        for (int i = 0; i < n; ++i)
        {
            v[i].clear();  
        }

        for (int i = 0; i < (int)(edges.size()); ++i)
        {
           int x = edges[i][0], y = edges[i][1];
            --x, --y;
           v[x].push_back(y);
           v[y].push_back(x);
        }

        --target;
        prob = 0;
        this->target = target;
        this->t = t;
        dfs(0, -1, 0, 1);
        return prob;
    }
};
