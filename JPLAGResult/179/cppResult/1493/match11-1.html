class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& e, int tmw, int target) {
        vector<vector<int>> g(n + 1);
        for (int i = 0; i < n - 1; ++i) {
            g[e[i][0]].push_back(e[i][1]);
            g[e[i][1]].push_back(e[i][0]);
        }
        vector<double> pr(n + 1, 1);
        vector<bool> vis(n + 1);
        queue<pair<int, int>> q;
        q.push({1, 0});
        vis[1] = 1;
        while (q.size()) {
            int u, t;
            tie(u, t) = q.front();
            q.pop();
            
            int cnt = 0;
            for (int v : g[u]) if (!vis[v]) ++cnt;
            
            if (u == target) {
                if (t > tmw)
                    return 0;
                else if (t == tmw)
                    return pr[u];
                if (cnt == 0)
                    return pr[u];
                else
                    return 0;
            }
            
            for (int v : g[u]) if (!vis[v]) {
                vis[v] = 1;
                q.push({v, t + 1});
                pr[v] = pr[u] / cnt;
            }
        }
        return 0;
    }
};
