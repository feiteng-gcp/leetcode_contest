#define FOR(i,a,b) for(int i=(a);i<=(b);i++)
#define FORD(i,a,b) for(int i=(a);i>=(b);i--)
#define FOREACH(i,c) for(__typeof((c).begin())i = (c).begin();i!=(c).end(); ++i)

class Solution {
public:
  vector<vector<int>> v;
  vector<double> prob;
  vector<bool> used;
  vector<int> depth;
  int T;

  void dfs(int x, int tm) {
    if (tm >= T) {
      return;
    }
    used[x] = true;
    for (auto y : v[x]) {
      if (!used[y]) {
        depth[y] = depth[x] + 1;
        prob[y] = prob[x] * 1.0 / (v[x].size() - (x != 0));
        dfs(y, tm + 1);
      }
    }
  }

  double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
    if (n == 1) {
      return 1.0;
    }
    v = vector<vector<int>>(n, vector<int>{});
    prob = vector<double>(n, 0.0);
    used = vector<bool>(n, false);
    depth = vector<int>(n, 0);
    target--;
    T = t;
    for (auto x : edges) {
      x[0]--;
      x[1]--;
      v[x[0]].push_back(x[1]);
      v[x[1]].push_back(x[0]);
    }
    prob[0] = 1.0;
    dfs(0, 0);
    if (t == depth[target] || (target != 0 && v[target].size() == 1)) {
      return prob[target];
    }
    return 0.0;
  }
};
