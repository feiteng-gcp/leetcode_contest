class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> next(n);
        for (int i=0; i<edges.size(); ++i) {
            next[edges[i][0]-1].push_back(edges[i][1]-1);
            next[edges[i][1]-1].push_back(edges[i][0]-1);
        }
        
        vector<int> vis(n);
        vector<int> dis(n, INT_MAX); 
        vector<double> possibility(n, 0); 
        deque<int> que;
        que.push_back(0);
        possibility[0] = 1;
        dis[0] = 0;
        while (!que.empty()) {
            int s = que.front();
            que.pop_front();
            int num = s == 0 ? next[s].size() : next[s].size() - 1;
            if (target -1 == s) {
                if (t < dis[s]) return 0;
                if (t == dis[s]) return possibility[s];
                if (t > dis[s]) return num == 0 ? possibility[s] : 0;
            }
            
            for (auto ts: next[s]) {
                if (dis[ts] > dis[s] + 1) {
                    dis[ts] = dis[s] + 1;
                    que.push_back(ts);
                    possibility[ts] = possibility[s] * (1.0 / num);
                }
            }
        }
        return 0;
    }
};
