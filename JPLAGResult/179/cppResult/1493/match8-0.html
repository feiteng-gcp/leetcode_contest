class Solution {
public:
    void adjust(int node, queue<int> &q, vector<int>& visited, vector<vector<int>> &graph, vector<double> &pos)
    {
        int n=0;
        for(auto i: graph[node])
        {
            if(!visited[i])
                n++;
        }
        
        if(n)
        {
            for(auto i: graph[node])
            {
                if(!visited[i])
                {
                   visited[i]=1;
                   q.push(i);
                   pos[i]+=pos[node]/n; 
                }
            }
            pos[node]=0;
        }
    }
    
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> graph(n+1, vector<int>());
        for(auto e:edges)
        {
            graph[e[0]].push_back(e[1]);
            graph[e[1]].push_back(e[0]);
        }
        
        vector<int> visited(n+1, 0);
        vector<double> pos(n+1, 0); //possiblility 
        
        pos[1]=1;
        visited[1]=1;
        
        queue<int> q;
        q.push(1);
        int sz,tmp, tmp1;
        
        while(!q.empty())
        {
            if(t==0)
                return pos[target];
            sz=q.size();
            
            while(sz--)
            {
               tmp=q.front();
               q.pop();
               adjust(tmp, q, visited, graph, pos);
            }
            t--;
        }
        return pos[target];
    }
};
