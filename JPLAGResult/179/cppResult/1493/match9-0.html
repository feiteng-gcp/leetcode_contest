class Solution {
public:
    double dfs(int u, int f, vector<vector<int>> &e, int t, int tu, double prob) {
      int son = 0;
      for(auto v : e[u]) {
        if(v == f) continue;
        son += 1;
      }
      if(tu == u) {
        if(t == 0) return prob;
        if(t < 0) return 0;
        if(son > 0) return 0;
        else return prob;
      }
      for(auto v : e[u]) {
        if(v == f) continue;
        double p = dfs(v, u, e, t - 1, tu, prob / son);
        if(p > -1) return p;
      }
      return -2;
    }
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
      vector<vector<int>> e(n + 1, vector<int>());
      for(auto p: edges) {
        e[p[0]].push_back(p[1]);
        e[p[1]].push_back(p[0]);
      }
      return dfs(1, 1, e, t, target, 1.0);
    }
};
