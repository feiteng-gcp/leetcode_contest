class Solution {
public:
    void DFS(int u, vector<vector<int>> &E, vector<bool> &visited, double probability, double &ans, int t, int target, bool &found) {
        visited[u] = true;
        if (u == target) {
            found = true;
        }
        if (t == 0) {
            if (u == target) {
                ans = probability;
            }
            return;
        }
        int count = 0;
        for (int v : E[u]) {
            if (!visited[v]) {
                ++count;
            }
        }
        if (u == target) {
            if (count == 0) {
                ans = probability;
            }
            return;
        }
        if (found) {
            return;
        }
        for (int v : E[u]) {
            if (!visited[v]) {
                DFS(v, E, visited, probability / count, ans, t - 1, target, found);
            }

        }
    }
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> E(n);
        for (auto V : edges) {
            E[V[0] - 1].push_back(V[1] - 1);
            E[V[1] - 1].push_back(V[0] - 1);
        }
        double ans = 0.0;
        vector<bool> visited(n, false);
        bool found = false;
        DFS(0, E, visited, 1.0, ans, t, target - 1, found);
        return ans;
    }
};
