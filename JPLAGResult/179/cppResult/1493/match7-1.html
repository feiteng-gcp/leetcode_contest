class Solution {
public:
    
    vector<int> adj[101];
    bool vis[101];
    
    double prob[101][101] = {0};
    int tt;
    int tar;
    double ans = 0.0;
    
    void rec(int cur, int k, double pr) {
        // cout << cur << " " << k << " "  << pr << endl; 
        
        prob[cur][k] += pr;
        
        vis[cur] = true;

        int cnt = 0;
        
        for (auto nxt: adj[cur]) 
            if (!vis[nxt]) cnt++;
        
        if (cnt == 0) {
            if (cur == tar && k < tt) ans += pr;
            vis[cur] == false;
            return;
        }
        
        pr /= (double)cnt;
        for (auto nxt: adj[cur]) 
            if (!vis[nxt]) {
                rec(nxt, k+1, pr);
            }
        
        vis[cur] = false;
        return;

    }
    
    
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        tt = t;
        tar = target;
        for (auto e: edges) {
            adj[e[0]].push_back(e[1]);
            adj[e[1]].push_back(e[0]);
        }
        memset(vis, false, sizeof vis);
        rec(1, 0, 1.0);
        return ans + prob[target][t];
    }
};
