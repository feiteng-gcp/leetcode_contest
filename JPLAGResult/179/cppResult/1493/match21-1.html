class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<int> v[n+1];
        for(auto x : edges){
            v[x[0]].push_back(x[1]);
            v[x[1]].push_back(x[0]);
        }
        vector<vector<double>> dp(n+1, vector<double>(n+1, 0));
        dp[1][1] = 1;
        while(t--){
            vector<vector<double>> ndp(n+1, vector<double>(n+1, 0));
            for(int i = 1; i <= n; i++){
                for(int j = 1; j <= n; j++){
                    if(dp[i][j] == 0)   continue;
                    double mom = (double)v[j].size()-(i!=j);
                    for(int k : v[j]){
                        if(k == i)  continue;
                        ndp[j][k] += dp[i][j]/mom;
                    }
                    if(mom == 0)    ndp[i][j] += dp[i][j];
                }
            }
            dp = ndp;
        }
        double ret = 0;
        for(int i = 1; i <= n; i++) ret += dp[i][target];
        return ret;
    }
};
