class Solution {
public:
    map<int, vector<int>> go;
    long long ans;
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        go.clear();
        for (auto edge : edges) {
            go[edge[0]].push_back(edge[1]);
            go[edge[1]].push_back(edge[0]);
        }
        ans = -1;
        dfs(1, 1ll, 0, -1, t, target);
        if (ans == -1) return 0;
        else 
            return 1.0 / ans;
    }
    
    void dfs(int u, long long y, int level, int fa, const int t, const int target) {
        int child_num = go[u].size() - (level > 0);
        if (u == target)  {
            if ((t > level && child_num > 0) || t < level) {
                ans = -1;
            } else {
                ans = y;                
            }
            return;
        }
        if (y >= 1000000) return;
        for (int v : go[u]) {
            if (v == fa) continue;
            dfs (v, y * child_num, level + 1, u, t, target);
        }
    }
};
