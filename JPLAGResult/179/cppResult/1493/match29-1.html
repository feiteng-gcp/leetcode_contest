class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        unordered_map<int, unordered_set<int>> block;
        unordered_map<int, unordered_set<int>> path;
        for (auto x:edges) {
            path[x[0]].insert(x[1]);
            path[x[1]].insert(x[0]);
        }
        vector<int> todo(1, 1);
        unordered_set<int> used;
        while(!todo.empty()) {
            vector<int> todo_next;
            for (int i:todo) {
                block[i]=used;
                used.insert(i);
                for (int j:path[i]) {
                    if (used.find(j)==used.end()) {
                        todo_next.push_back(j);
                    }
                }
            }
            todo=todo_next;
        }
        vector<double> st(n+1, 0.0);

        st[1]=1.0;
        for (int i=0; i<t; i++) {
            vector<double> st_next(n+1, 0.0);
            for (int i=1; i<=n; i++) {
                if (st[i]!=0) {
                    int av=0;
                    for (int n:path[i]) {
                        if (block[i].find(n)==block[i].end()) {
                            av++;
                        }
                    }
                    if (av!=0) {
                        for (int n:path[i]) {
                            if (block[i].find(n)==block[i].end()) {
                                st_next[n]+=st[i]/(double)av;
                            }
                        }
                    } else {
                        st_next[i]+=st[i];
                    }
                }
            }
            st=st_next;
        }
        return st[target];
    }
};
