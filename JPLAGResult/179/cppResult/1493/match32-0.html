class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<int> son(n+1, 0), dep(n+1, 0);
        vector<double> p(n+1, 0);
        vector<int> e[n+1];
        queue<int> q;
        
        for (auto v : edges) {
            e[v[0]].emplace_back(v[1]);
            e[v[1]].emplace_back(v[0]);
        }
        q.push(1);
        p[1] = 1;
        dep[1] = 1;
        while (q.size()) {
            int u = q.front();
            q.pop();
            for (int v : e[u]) if (dep[v] == 0) {
                q.push(v);
                ++son[u];
            }
            for (int v : e[u]) if (dep[v] == 0) {
                dep[v] = dep[u]+1;
                p[v] = p[u]/son[u];
            }
        }
        if (dep[target]-1 > t || (dep[target]-1 < t && son[target])) return 0;
        return p[target];
    }
};
