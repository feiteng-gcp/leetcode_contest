#define FOR(i, begin, end) for(int i=(begin),i##_end_=(end);i<i##_end_;i++)
#define REP(i, n) FOR(i,0,n)
class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> to(n + 1);
        for (auto p : edges) {
            to[p[0]].push_back(p[1]);
            to[p[1]].push_back(p[0]);
        }
        vector<vector<double>> prob(n + 1, vector<double>(n + 1));
        prob[1][0] = 1;
        REP(time, t) {
            vector<vector<double>> prob_nxt(n + 1, vector<double>(n + 1));
            REP(now, n + 1) {

                double p_now = accumulate(prob[now].begin(), prob[now].end(), 0.0);
                int nb_nxt = (time == 0 ? to[now].size() : to[now].size() - 1);
                
                if (!nb_nxt) {
                    REP(prv, n + 1) prob_nxt[now][prv] += prob[now][prv];
                    continue;
                }

                for (auto nxt : to[now]) {
                    double p = p_now - prob[now][nxt];
                    prob_nxt[nxt][now] += p / nb_nxt;
                }
                    
                    /*
                    int c = 0;
                    for (auto nxt : to[now]) if (nxt != prv) c++;
                    if (c == 0) {
                        prob_nxt[now][prv] += prob[now][prv];
                        continue;
                    }
                    for (auto nxt : to[now]) if (nxt != prv) {
                        prob_nxt[nxt][now] += prob[now][prv] / c;
                    }
                    */
            }
            prob = prob_nxt;
        }
        return accumulate(prob[target].begin(), prob[target].end(), 0.0);
    }
};
