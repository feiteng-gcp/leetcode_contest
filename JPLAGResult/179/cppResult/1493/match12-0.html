class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<unordered_set<int>> A(n);
        for (int i = 0; i < edges.size(); i++) {
            A[edges[i][0]-1].insert(edges[i][1]-1);
            A[edges[i][1]-1].insert(edges[i][0]-1);
        }
        rec(0, A);
        //for (int i = 0; i < n; i++) {
        //    for (int j = 0; j < A[i].size(); j++) {
        //        cout << A[i][j] << " ";
        //    }
        //    cout << "\n";
        //}
        
        vector<double> v(n);
        vector<double>& p = v;
        p[0] = 1;
        for (int i = 0; i < t; i++) {
            vector<double> p1(n);
            for (int j = 0; j < n; j++) {
                int len = A[j].size();
                for (auto k : A[j]) {
                    p1[k] += p[j]/len;
                }
                if (len == 0) {
                    p1[j] += p[j];
                }
            }
            p = p1;
            //for (int i = 0; i < n; i++) {
            //    cout << p[i] << " ";
            //}
            //cout << "\n";
        }
        return p[target - 1];
    }
    
    void rec(int cur, vector<unordered_set<int>>& A) {
        for (auto child: A[cur]) {
            A[child].erase(cur);
            rec(child, A);
        }
    }
};
