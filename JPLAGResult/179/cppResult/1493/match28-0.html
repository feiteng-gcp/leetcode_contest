vector<vector<int>> v;

double dfs(int a, int b, double c, int d, int t) {
    int hv = 0;
    for (int i : v[a])
        if (i != d)
            hv++;
    if (a == t) {
        if (b > 0)
            return hv ? 0.0 : c;
        return b == 0 ? c : 0.0;
    }
    double ans = 0;
    for (int i : v[a])
        if (i != d)
            ans += dfs(i, b - 1, c / hv, a, t);
    return ans;
}

class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        v.clear();
        v.resize(n);
        target--;
        for (auto& i : edges) {
            v[i[0] - 1].push_back(i[1] - 1);
            v[i[1] - 1].push_back(i[0] - 1);
        }
        return dfs(0, t, 1, -1, target);
    }
};
