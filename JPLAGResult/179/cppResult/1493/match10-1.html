class Solution {
public:
    vector<int>g[104];
    int num,dx;
    double ans = 0;
    void dfs(int u,int f,double p,int fa){
        if(f > dx) return;

        bool b = 1;
        for(int i = 0;i < g[u].size();i++){
            int v = g[u][i];
            if(v == fa) continue;
            dfs(v,f + 1,p / (g[u].size() - (u != 1)),u);
            b = 0;
        }
        if(u == num){
            if(f == dx || b) ans = p;
            else ans = 0;
        }
        //printf("%d %f\n",u,p);

    }
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        for(int i = 0;i < edges.size();i++){
            int x = edges[i][0],y = edges[i][1];
            g[x].push_back(y);
            g[y].push_back(x);
        }

        num = target;
        dx = t;
        dfs(1,0,1.0,-1);
        return ans;
    }
};
