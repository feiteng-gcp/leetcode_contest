class Solution {
public:    
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        queue<pair<int, double>> q;
        vector<bool> visit(n + 1, false);
        vector<vector<int>> g(n + 1, vector<int>(n + 1, 0));
        for (auto& e : edges) {
            g[e[0]][e[1]] = 1; g[e[1]][e[0]] = 1;
        }
        q.push({1, 1}); visit[1] = true;
        while (!q.empty() && t--) {
            int s = q.size();
            while (s--) {
                auto cur = q.front(); q.pop();
                int num = 0;
                for (int i = 1; i <= n; i++) {
                    if (i != cur.first && g[cur.first][i] && !visit[i]) {
                        num++;
                    }
                }
                for (int i = 1; i <= n; i++) {
                    if (i != cur.first && g[cur.first][i] && !visit[i]) {
                        if (i == target && t == 0) return cur.second / num;
                        q.push({i, cur.second / num}); visit[i] = true;
                    }
                }
                if (num == 0) {
                    if (cur.first == target && t == 0) return cur.second;
                    q.push(cur);
                }
            }
        }
        return 0;
    }
};
