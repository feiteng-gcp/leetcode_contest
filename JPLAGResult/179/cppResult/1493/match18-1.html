class Solution {
public:
  vector<vector<int>> G;
  vector<double> W;
  vector<int> D;
  void dfs(int s, int f, double p, int d) {
    D[s] = d;
    W[s] = p;

    for (auto t: G[s]) if (t != f) {
      dfs(t, s, p/(G[s].size() - (f == 0?0:1)), d+1);
    }
  }
  double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
    G.clear();
    G.resize(n+1);
    W.clear();
    W.resize(n+1);
    D.clear();
    D.resize(n+1);
    for (auto e: edges) G[e[0]].push_back(e[1]), G[e[1]].push_back(e[0]);
    dfs(1, 0, 1, 0);
    if (D[target] > t) return 0;
    if (D[target] < t && ((target == 1&&G[target].size() > 0) || G[target].size() > 1)) return 0;
    return W[target];
  }
};
