class Solution {
public:
    
double frogPosition(int n, vector<vector<int>>& ed, int t, int tar) {
    vector<vector<int>> e(n + 1);
    for (const auto&v : ed) {
    	e[v[0]].push_back(v[1]);
    	e[v[1]].push_back(v[0]);
    }
    vector<int> tm(n + 1, -1);
    vector<double> p(n + 1);
    vector<int> son(n + 1);
    queue<int> q;
    q.push(1);
    tm[1] = 0;
    p[1] = 1;
    while(!q.empty()) {
    	int u = q.front(); q.pop();
    	int c = 0;
    	for (int x : e[u]) {
    		if(tm[x] == -1)
    			++c;
    	}
        son[u] = c;
    	if(c == 0) continue;
    	double pp = p[u] / c;
    	for (int x : e[u]) {
    		if(tm[x] != -1) continue;
    		q.push(x);
            //cout << x << " " << c << endl;
    		tm[x] = tm[u] + 1;
    		p[x] = pp;
    	}
    }
    if(tm[tar] == t || (tm[tar] < t && son[tar] == 0))
    	return p[tar];
    return 0;
}
};
