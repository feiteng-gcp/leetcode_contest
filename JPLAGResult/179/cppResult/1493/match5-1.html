class Solution {
private:
    bool DFS(vector<vector<int>>& graph, int from, int target, int curr, vector<int>& path) {
      if (curr == target) {
        return true;
      }
      
      path.push_back(curr);
      for (int to : graph[curr]) {
        if (to == from) {
          continue;
        }
        

        if (DFS(graph, curr, target, to, path)) {
          return true;
        }
      }
      
      path.pop_back();
      return false;
    }
  
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
      vector<vector<int>> graph(n + 1);
      for (auto& e : edges) {
        int a = e[0];
        int b = e[1];
        
        graph[a].push_back(b);
        graph[b].push_back(a);
      }
      
      vector<int> path;
      DFS(graph, -1, target, 1, path);
      

      if (path.size() > t || 
         (path.size() < t && 
          ((target == 1 && graph[1].size() > 0) || 
           (target > 1  && graph[target].size() > 1)))) {
        return 0.0;
      }
      
      double prop = 1.0;
      for (auto p : path) {
        int child = graph[p].size();
        if (p != 1) {
          --child;
        }
        prop *= 1.0 / child;
      }
      
      return prop;
    }
};
