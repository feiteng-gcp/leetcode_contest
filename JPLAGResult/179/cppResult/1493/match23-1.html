class Solution {
public:
    
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<vector<int>> g(n);
        for(auto item:edges){
            g[item[0]-1].push_back(item[1]-1);
            g[item[1]-1].push_back(item[0]-1);
        }
        
        vector<double> dp(n,0.0);
        dp[0] = 1.0;
        vector<int> dis(n,-1);
        vector<bool> isleaf(n,true);
        dis[0] = 0;
        deque<int> que;
        que.push_back(0);
        while(!que.empty()){
            int tm = que.size();
            while(tm--){
                auto p = que.front();
                isleaf[p] = true;
                que.pop_front();
                for(auto item:g[p]){
                    if(dis[item]!=-1) continue;
                    dis[item] = dis[p]+1;
                    int cnt = g[p].size();
                    if(p) cnt--;
                    dp[item] = 1.0 / (cnt) * dp[p];
                    que.push_back(item);
                    isleaf[p] = false;
                }
            }
        }
        if(dis[target-1] != t && !isleaf[target-1]){
            return 0.0;
        }
        // for(int i=0;i<n;i++){
        //     cout<<dp[i]<<endl;
        // }
        if(t < dis[target-1]) return 0.0;
        return dp[target-1];
    }
};
