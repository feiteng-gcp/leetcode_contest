class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        vector<set<int>> tr(n+1);
        for (const auto& e: edges) {
            tr[e[0]].insert(e[1]);
            tr[e[1]].insert(e[0]);
        }
        
        unordered_map<int, double> p;
        unordered_map<int, int> depth;
        unordered_set<int> ends;
        int max_depth;
        p[1] = 1;
        depth[1] = 0;
        vector<pair<int, vector<int>>> next;
        next.push_back({-1, {1}});
        for (int i = 0; i <= t; ++i) {
            if (next.empty()) {
                break;
            }
            
            max_depth = i;
            vector<pair<int, vector<int>>> nnext;
            for (const auto& batch: next) {
                const auto& head = batch.first;
                const auto& nodes = batch.second;
                
                for (const auto node : nodes) {
                    if (head != -1) {
                        depth[node] = depth[head]+1;
                        p[node] = p[head]/nodes.size();
                    }
                    vector<int> chs;
                    for (const auto ch: tr[node]) {
                        if (!depth.count(ch)) {
                            chs.push_back(ch);
                        }
                    }
                    if (!chs.empty()) {
                        nnext.emplace_back(node, std::move(chs));
                    } else {
                        ends.insert(node);
                    }
                }
            }
            
            next = nnext;
        }
                
        t = min(max_depth, t);
        if ((ends.count(target) && t >= depth[target]) || depth[target] == t) {
            return p[target];
        } else {
            return 0;
        }
        
    }
};
