class Solution {
public:
  double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
    set<int> visited;
    vector<set<int>> next(n + 1);
    for (auto edge: edges) {
      next[edge[0]].emplace(edge[1]);
      next[edge[1]].emplace(edge[0]);
    }
    map<int, double> p;
    p[1] = 1;
    visited.emplace(1);
    for (int i = 0; i < t; i++) {
      map<int, double> nextP;
      set<int> nextVisited;
      for (auto cur: p) {
        for (auto nextX: next[cur.first]) if (visited.count(nextX)) next[cur.first].erase(nextX);
        for (auto nextX: next[cur.first]) {
          nextP[nextX] += cur.second / next[cur.first].size();
          nextVisited.emplace(nextX);
        }
        if (next[cur.first].size() == 0) nextP[cur.first] += cur.second;
      }
      p = nextP;
      for (auto x: nextVisited) visited.emplace(x);
    }
    return p[target];
  }
};
