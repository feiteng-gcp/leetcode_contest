class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        for(auto &a : edges)
        {
            if(a[0] > a[1])
                swap(a[0], a[1]);
        }
        
        vector<vector<int>> graph = vector<vector<int>>(n, vector<int>{});
        for(auto &e : edges)
        {
            graph[e[0] - 1].push_back(e[1] - 1);
            //graph[e[1] - 1].push_back(e[0] - 1);
        }
        
        
        vector<double> dp(n, 0);
        dp[0] = 1.0;
        for(int i = 0; i < t; ++i)
        {
            vector<double> new_dp = vector<double>(n, 0);
            for(int j = 0; j < n; ++j)
            {
                if(graph[j].size() == 0) 
                    new_dp[j] += dp[j];
                else
                {
                    for(int k = 0; k < graph[j].size(); ++k)
                    {
                        new_dp[graph[j][k]] += (dp[j] / double(graph[j].size()));
                    }
                }
                
            }
            dp = new_dp;
            // cout << "time" << i << endl;
            // for(auto &a : dp)
            //     cout << a << endl;
        }
        return dp[target - 1];
    }
};
