class Solution {
public:
    double frogPosition(int n, vector<vector<int>>& edges, int t, int target) {
        map<int, double> p;
     
        map<int, vector<int>> m;
        
        for(auto& e: edges) {
            m[e[0]].push_back(e[1]);
            m[e[1]].push_back(e[0]);

        }
        p[1] = 1.0;
        set<int> vis;
        vis.insert(1);
        
        queue<int> q;
        q.push(1);
        int iter = 0;
        while(q.size()) {
            for (int j = q.size(); j > 0; j--) {
            int cur = q.front(); q.pop();
            int si  = 0;
            for(int i: m[cur]) if(vis.count(i) == 0) { si++;}
            double each = p[cur] / si;
            for(int i: m[cur]) if(vis.count(i) == 0) { vis.insert(i); q.push(i); p[i] = each;}
            
            if (si != 0) p[cur] = 0;
            }
            iter++;
            if(iter >= t) break;
        }
        
        return p[target];
    }
};
