class Solution {
public:
    double ans=0.0;
    unordered_set <int> adj[111];
    unordered_set <int> visited;
    int target=-1;
    int timelimit=0;
    
    void dfs(int node, double prob=1.0, int time=0){
        if(time>timelimit)
            return;
        
        int children=0;
        for(auto c: adj[node])
            if(visited.find(c)==visited.end())
                children++;
        
        if(node==target){
            if(children==0 || time==timelimit)
              ans=prob;
            return;
        }
        
        
        if(!children)
            return;
        for(auto c: adj[node]){
            if(visited.find(c)==visited.end()){
                visited.insert(c);
                dfs(c, prob*1.0/(double)children, time+1);
                visited.erase(visited.find(c));
            }
        }
    }
    
    double frogPosition(int n, vector<vector<int>>& edges, int t, int targ) {
        target=targ;
        timelimit=t;
        for(auto edge: edges){
            adj[edge[0]].insert(edge[1]);
            adj[edge[1]].insert(edge[0]);
        }
        visited.insert(1);
        dfs(1);
        return ans;
    }
};
