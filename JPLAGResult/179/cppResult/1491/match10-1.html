class Solution {
public:
    int numTimesAllBlue(vector<int>& light) {
        int ans = 0;
        int maxLight = 0;
        int size = light.size();
        for (int i = 0; i < size; i++) {
            maxLight = max(maxLight, light[i]);
            if (maxLight == i + 1) {
                ans++;
            }
        }
        return ans;
    }
};
