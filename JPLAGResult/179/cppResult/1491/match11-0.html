class Solution {
public:
    int numTimesAllBlue(vector<int>& light) {
        int n=light.size();
        int max_turned_on = 0;
        int ans=0;
        for(int i=0;i<n;i++) {
            max_turned_on = max(max_turned_on, light[i]);
            if (max_turned_on == i+1) {
                ans++;
            }
        }
        return ans;
        
        
    }
};
