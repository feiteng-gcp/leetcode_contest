class Solution {
public:
    int numTimesAllBlue(vector<int>& light) {
        int latest = light[0];
        int cnt = 0;
        for (int i = 0; i < light.size(); i++) {
            latest = max(light[i], latest);
            if (latest == i + 1) {
                cnt++;
            }
        }
        return cnt;
    }
};
