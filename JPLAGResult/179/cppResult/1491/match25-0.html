class Solution {
public:
    int numTimesAllBlue(vector<int>& l) {
        int n = l.size();
        int res = 0;
        int mx = 0;
        for (int i = 0; i < n; ++i) {
            mx = max(mx, l[i]);
            if (i + 1 == mx)
                ++res;
        }
        return res;
    }
};
