class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        unordered_map<string, int> mp;
        for (auto & str : req_skills) {
            mp[str] = mp.size();
        }
        vector<int> masks;
        for (auto & v : people) {
            int mask = 0;
            for (auto & str : v) {
                if (mp.find(str) != mp.end()) {
                    mask = (mask | (1 << mp[str]));
                }
            }
            masks.push_back(mask);
        }
        int num = mp.size();
        int all = (1 << num);
        int num_dp[all];
        long team_dp[all];
        for (int i = 1; i < all; i++)
            num_dp[i] = INT_MAX;
        num_dp[0] = 0;
        team_dp[0] = 0;
        int len = masks.size();
        for (int mask = 0; mask < all; mask++) {
            if (num_dp[mask] == INT_MAX)
                continue;

            for (int i = 0; i < len; i++) {
                int team_mask = masks[i];
                if (num_dp[mask | team_mask] > num_dp[mask] + 1) {
                    num_dp[mask | team_mask] = num_dp[mask] + 1;
                    team_dp[mask | team_mask] = (team_dp[mask] | (1ll << i));
                }
            }
        }
        vector<int> ans;
        for (int i = 0; i < len; i++) {
            if (team_dp[all - 1] & (1ll << i)) {
                ans.push_back(i);
            }
        }
        return ans;
    }
};
