class Solution {
 public:
  vector<int> smallestSufficientTeam(vector<string>& req_skills,
                                     vector<vector<string>>& people) {
    int n = (int)req_skills.size();
    vector<vector<int>> dp(1 << n);
    for (int k = 0; k < (int)people.size(); ++k) {
      const auto& p = people[k];

      int mask = 0;
      for (int i = 0; i < n; ++i) {
        if (find(p.begin(), p.end(), req_skills[i]) != p.end()) {
          mask |= 1 << i;
        }
      }
      if (mask == 0) {
        continue;
      }
      for (int i = (1 << n) - 1; i >= 0; --i) {
        if ((i == 0 || !dp[i].empty()) &&
            (dp[i | mask].empty() || dp[i | mask].size() > dp[i].size() + 1)) {
          dp[i | mask] = dp[i];
          dp[i | mask].push_back(k);
        }
      }
    }
    return dp[(1 << n) - 1];
  }
};
