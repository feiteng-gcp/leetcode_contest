int dp[1<<16];
vector<int> rec[1<<16];
class Solution {
public:
    map<string, int> skills;
    
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        int m=req_skills.size(), n=people.size();
        for (int i=0; i<m; i++) skills[req_skills[i]]=i;
        
        memset(dp, -1, sizeof(dp));
    
        int limit=1<<m;
        dp[0]=0; rec[0].clear();
        
        for (int i=0; i<n; i++){
            int sk=0;
            for (string& str: people[i]) sk|=(1<<skills[str]);
            for (int st=0; st<limit; st++){
                if (dp[st]==-1) continue;

                int nxt=st|sk;
                if (dp[nxt]==-1 || dp[st]+1<dp[nxt]){
                    dp[nxt]=dp[st]+1;
                    rec[nxt]=rec[st];
                    rec[nxt].push_back(i);
                }
            }
        }
        
        return rec[limit-1];
    }
};
