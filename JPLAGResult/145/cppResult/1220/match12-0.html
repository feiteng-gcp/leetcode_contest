class Solution {
	int n, m, g;
	map<string, int> id;
	vector<int> s;
public:
	vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
		n = people.size();
		m = req_skills.size();
		g = (1 << (m));
		int i, j, k;
		for (i = 0; i < m; i++) id[req_skills[i]] = i;
		s.assign(n, 0);
		for (i = 0; i < n; i++) {
			for (auto p : people[i]) s[i] |= (1 << id[p]);
		}

		vector<vector<int>> dp(n + 1, vector<int>(g, 1000));
		vector<vector<long long>> ans(n + 1, vector<long long>(g, 0));
		dp[0][0] = 0;
		for (i = 0; i < n; i++) {
			for (j = 0; j < g; j++) {
				int r = j | s[i];

				if (dp[i + 1][r] > dp[i][j] + 1) {
					dp[i + 1][r] = dp[i][j] + 1;
					ans[i + 1][r] = ans[i][j] | (1LL << i);
				}

				if (dp[i + 1][j] > dp[i][j]) {
					dp[i + 1][j] = dp[i][j];
					ans[i + 1][j] = ans[i][j];
				}
			}
		}

		vector<int> alist;
		for (i = 0; i < n; i++) {
			if ((1LL << i) & (ans[n][g - 1])) alist.push_back(i);
		}

		return alist;
	}
};
