class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        int n = req_skills.size();
        int f[1 << n], g[1 << n], h[1 << n];
        memset(f, 0x7f, sizeof(f));
        memset(g, 0, sizeof(g));
        memset(h, 0, sizeof(h));
        f[0] = 0;
        
        unordered_map<string, int> skills;
        for (int i = 0; i < req_skills.size(); ++i) {
            skills[req_skills[i]] = i;
        }
        
        for (int i = 0; i < people.size(); ++i) {
            int cur = 0;
            for (int j = 0; j < people[i].size(); ++j) {
                cur += (1 << skills[people[i][j]]);
            }

            for (int j = 0; j < (1 << n); ++j) {
                if (f[j] + 1 < f[j | cur]) {
                    f[j | cur] = f[j] + 1;
                    g[j | cur] = i;
                    h[j | cur] = j;
                }
            }
        }
        
        vector<int> ans;
        int pos = (1 << n) - 1;
        while (pos > 0) {
            // cout << "p = " << pos << " " << f[pos] << " " << g[pos] << " " << h[pos] << "\n";
            ans.push_back(g[pos]);
            pos = h[pos];
        }
        return ans;
    }
};
