class Solution {
  public:
    unsigned dp[1 << 16][61];
    int from[1 << 16][61];
    unordered_map<string, int> X;
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
      memset(dp, -1, sizeof(dp));
      dp[0][0] = 0;

      int k = req_skills.size();
      for (int i = 0; i < req_skills.size(); i++) {
        X[req_skills[i]] = i;
      }
      for (int i = 0; i < people.size(); i++) {
        int s = 0;
        for (auto& x: people[i]) {
          if (X.count(x)) {
            s |= 1 << X[x];
          }
        }
        for (int j = 0; j < (1 << k); j++) {
          if (dp[j][i] == 0xffffffffU) {
            continue;
          }
          if (dp[j][i] < dp[j][i + 1]) {
            dp[j][i + 1] = dp[j][i];
            from[j][i + 1] = j;
          }
          if (dp[j][i] + 1 < dp[j | s][i + 1]) {
            from[j | s][i + 1] = j;
            dp[j | s][i + 1] = dp[j][i] + 1;
          }
        }
      }
      vector<int> ans;
      int cx = (1 << k) - 1, cy = people.size();
      while (cy > 0) {
        if (cx != from[cx][cy]) {
          ans.push_back(cy - 1);
        }
        cx = from[cx][cy];
        cy--;
      }
      return ans;
    }
};
