class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& s, vector<vector<string>>& p) {
        map <string,int> mp;
        int i,j,n=s.size(),m=p.size();
        int sz[105],t[105],dp[100005],cs[100005],lst[100005];
        for (i=0;i<n;i++) mp[s[i]]=i+1;
        for (i=0;i<m;i++)
        {sz[i]=p[i].size();t[i]=0;
         for (j=0;j<sz[i];j++)
         {if (!mp[p[i][j]]) continue;
          t[i]|=(1<<(mp[p[i][j]]-1));
         }
        }
        dp[0]=0;
        for (i=1;i<(1<<n);i++) dp[i]=100000000;
        for (i=0;i<(1<<n);i++)
        {for (j=0;j<m;j++)
        {if (dp[i|t[j]]>dp[i]+1)
        {dp[i|t[j]]=dp[i]+1;cs[i|t[j]]=j;lst[i|t[j]]=i;}
        }
        }
        int now=(1<<n)-1;
        vector<int> ans;
        while (now)
        {ans.push_back(cs[now]);now=lst[now];}
        return ans;
        
        
        
    }
};
