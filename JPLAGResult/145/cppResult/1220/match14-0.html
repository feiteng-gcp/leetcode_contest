class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        unordered_map<string, int> no;
        int m = req_skills.size();
        if (m == 0) return {};
        vector<vector<int>> f (1 << m, vector<int> ());
        vector<int> v (1 << m, false);
        for (int i = 0; i < m; ++i) no[req_skills[i]] = i;
        v[0] = true;
        f[0] = {};
        int n = people.size();
        for (int i = 0; i < n; ++i) {
            int mask = 0;
            for (auto skill: people[i]) if (no.count(skill)) mask |= 1 << no[skill];
            for (int j = (1 << m) - 1; ~j; --j) if (v[j]) {
                if (v[j | mask]) {
                    if (f[j].size() + 1 < f[j | mask].size()) {
                        f[j | mask] = {};
                        for (auto p: f[j]) f[j|mask].push_back(p);
                        f[j|mask].push_back(i);
                    }
                } else {
                    v[j | mask] = true;
                    f[j | mask] = {};
                    for (auto p: f[j]) f[j|mask].push_back(p);
                    f[j|mask].push_back(i);
                }
            }
        }
        return f[(1 << m) - 1];
    }
};
