class Solution {
public:
  vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {    

    const int n = req_skills.size();
    map<string, int> idx;
    for (int i = 0; i < n; ++i)
      idx[req_skills[i]] = i;
    int target = (1 << n) - 1;        
    vector<int> ps;
    for (auto& p : people) {
      int mask = 0;

      for (const string& s : p) {
        if (!idx.count(s)) continue;
        mask |= (1 << idx[s]);
      }
     // cout << mask << endl;
      ps.push_back(mask);      
    }
    
   // cout << n << " " << ps.size();
    int P = ps.size();
    
    //vector<vector<int>> dp((1 << n), vector<int>(ps.size(), INT_MAX / 2));
    vector<unordered_map<int, int>> dp(P);
    vector<vector<pair<int, int>>> parent((1 << n), vector<pair<int, int>>(ps.size(), {-1, -1}));
    
    int c = INT_MAX / 2;
    int p = -1;
    int s = target;
    
    for (int i = 0; i < ps.size(); ++i) {
      dp[i][ps[i]] = 1;
      parent[ps[i]][i] = {0, -1};      
        for (int j = 0; j < ps.size(); ++j) {
          if (i == j) continue;
          for (const auto& pa : dp[j]) {
            int k = pa.first;            
            int s = k | ps[i];
            if ((dp[i].count(s) && dp[j][k] + 1 < dp[i][s]) || !dp[i].count(s)) {
              dp[i][s] = dp[j][k] + 1;
              parent[s][i] = {k, j};
          }
        }        
      }
    }
    
    
    for (int i = 0; i < P; ++i) {
      if (dp[i].count(target) && dp[i][target] < c) {
        c = dp[i][target];
        p = i;
      }
    }
        
    vector<int> ans;
    while (p != -1) {
      // cout << p << endl;
      ans.push_back(p);
      const auto& pa = parent[s][p];
      s = pa.first;
      p = pa.second;      
    }
    return ans;
  }
};
