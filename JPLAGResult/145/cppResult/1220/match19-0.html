class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        map<string, int> m;
        int si = req_skills.size();
        for(int i=0;i<si;i++){
            m[req_skills[i]]=i;
        }
        int n=people.size();
        vector<int> np(n);
        vector<int> res(1<<(si), INT_MAX);
        for(int i=0;i<n;i++){
            int tmp=0;
            for(auto item:people[i]){
                if(m.find(item)!=m.end()){
                    tmp += (1<<m[item]);
                }
            }
            np[i]=tmp;
        }
        res[0]=0;
        vector<int> parent(1<<si,-1);
        vector<int> use(1<<si,-1);
        for(int i=0;i<(1<<si);i++){
            if(res[i]==INT_MAX) continue;
            for(int j=0;j<n;j++){
                if(res[i|np[j]] > 1+res[i]){
                    parent[i|np[j]] = i;
                    use[i|np[j]] = j;
                }
                res[i|np[j]] = min(res[i|np[j]], 1+res[i]);
            }
        }
        vector<int> mm;
        int tt=(1<<si)-1;
        while(tt>0){
            mm.push_back(use[tt]);
            tt=parent[tt];
        }
        sort(mm.begin(),mm.end());
        return mm;
    }
};
