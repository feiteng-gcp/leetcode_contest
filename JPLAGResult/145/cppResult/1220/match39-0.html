int dp[100][1<<16];
int f[100][1<<16];
class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        int m = req_skills.size();
        int n = people.size();
        map<string, int> skmp;
        int cnt = 0;
        for(auto sk: req_skills) {
            skmp[sk] = cnt++;
        }
        vector<int> a(n, 0);
        int rt = 1 << m;
        int idx = 0;
        for(auto p: people) {
            a[idx] = 0;
            for(auto sk: p) {
                if(skmp.find(sk) != skmp.end()) {
                    a[idx] |= (1 << skmp[sk]);
                }
            }
            idx++;
        }
        for(int i = 0; i <= n; i++) for(int j = 0; j < rt; j++) dp[i][j] = 1e9;
        dp[0][0] = 0;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < rt; j++) {
                dp[i + 1][j] = dp[i][j];
                f[i + 1][j] = j;
            }
            for(int j = 0; j < rt; j++) {
                if(dp[i][j] < 1e9) {
                    if(dp[i][j] + 1 < dp[i + 1][j | a[i]]) {
                        dp[i + 1][j | a[i]] = dp[i][j] + 1;
                        f[i + 1][j | a[i]] = j;
                    }
                }
            }
        }
        vector<int> ret;
        for(int i = n, j = rt - 1; i > 0; i--) {
            if(f[i][j] != j) {
                j = f[i][j];
                ret.push_back(i - 1);
            }
        }
        return ret;
    }
};
