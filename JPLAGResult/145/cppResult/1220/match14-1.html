class Solution {
public:
    vector<int> grps[1<<16];
    int dp[1<<16];
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        int K = req_skills.size(), N = people.size();
        map<string,int> M;
        memset(dp,-1,sizeof(dp));
        dp[0] = 0;

        for (int i=0;i<K;++i) M[req_skills[i]] = i;
        for (int i=0;i<N;++i) {
            int t = 0;
            for (auto w : people[i])
                if (M.find(w) != M.end()) t |= (1<<M[w]);
            for (int z=0;z<(1<<K);++z)
                if (dp[z] != -1) {
                    if (dp[z|t] == -1 || dp[z] + 1 < dp[z|t]) {
                        dp[z|t] = dp[z] + 1;
                        int r = z|t;
                        grps[r].clear();
                        for (auto v : grps[z]) grps[r].push_back(v);
                        grps[r].push_back(i);
                    }
                }
        }
        return grps[(1<<K)-1];
    }
};
