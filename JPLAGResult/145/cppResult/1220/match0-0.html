class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        map<string, int> index;
        for (int i = 0;i < req_skills.size();i++) {
            index[req_skills[i]] = i;
        }
        vector<int> dp(1 << req_skills.size(), -1);
        vector<pair<int, int>> last(1 << req_skills.size(), make_pair(-1, -1));
        dp[0] = 0;
        for (int i = 0;i < people.size();i++) {
            int x = 0;
            for (int j = 0;j < people[i].size();j++) {
                x |= 1 << (index[people[i][j]]);
            }
            for (int j = 0;j < dp.size();j++) {
                if (dp[j] != -1) {
                    int temp = j | x;
                    if (dp[temp] == -1 || dp[temp] > dp[j] + 1) {
                        dp[temp] = dp[j] + 1;
                        last[temp].first = j;
                        last[temp].second = i;
                    }
                }
            }
        }
        vector<int> ret;
        int x = (1 << req_skills.size()) - 1;
        while (x != 0) {
            ret.push_back(last[x].second);
            x = last[x].first;
        }
        return ret;
    }
};
