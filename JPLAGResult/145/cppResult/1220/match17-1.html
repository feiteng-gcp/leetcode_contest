class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
      int inf = 0x3f3f3f3f;
      int m = req_skills.size(), n = people.size();
      int all = 1 << m;
      map<string, int> skill_id;
      int k = 0;
      for (const auto &skill : req_skills) {
        skill_id[skill] = k++;
      }
      vector<int> skillset(n);
      for (int i = 0; i < n; ++i) {
        int st = 0;
        for (const auto &skill : people[i]) {
          st |= 1 << skill_id[skill];
        }
        skillset[i] = st;
      }
      vector<vector<int>> dp(n + 1, vector<int>(all, inf)), last(n + 1, vector<int>(all, 0));

      dp[0][0] = 0;
      for (int i = 1; i <= n; ++i) {
        for (int j = 0; j < all; ++j) {
          if (dp[i][j] > dp[i - 1][j]) {
            dp[i][j] = dp[i - 1][j];
            last[i][j] = j;
          }
          if (dp[i - 1][j] + 1 < dp[i][j | skillset[i - 1]]) {
            dp[i][j | skillset[i - 1]] = dp[i - 1][j] + 1;
            last[i][j | skillset[i - 1]] = j;
          }
        }
      }
      int i = n, j = all - 1;
      vector<int> res;
      for (; i; --i) {
        if (last[i][j] != j) {
          res.push_back(i - 1);
        }
        j = last[i][j];
      }
      return res;
    }
};
