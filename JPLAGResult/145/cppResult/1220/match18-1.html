int dp[65540],par[65540],chosen[65540],ppl[65];
map<string,int> m;

class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        int i,j,n,p,sk;
        vector<int> ans;
        
        n = req_skills.size();
        p = people.size();
        
        memset(dp,0x3f,sizeof(dp));
        memset(par,-1,sizeof(par));
        memset(chosen,-1,sizeof(chosen));
        m.clear();
        
        for (i = 0;i < n;i++)
            m[req_skills[i]] = i;
        for (i = 0;i < p;i++) {
            sk = 0;
            for (j = 0;j < people[i].size();j++)
                sk |= (1 << m[people[i][j]]);
            ppl[i] = sk;
        }
        

        dp[0] = 0;
        for (i = 0;i < (1 << n);i++) {
            for (j = 0;j < p;j++) {
                if (dp[i] + 1 < dp[i | ppl[j]]) {
                    dp[i | ppl[j]] = dp[i] + 1;
                    par[i | ppl[j]] = i;
                    chosen[i | ppl[j]] = j;
                }
            }
        }
        
        i = (1 << n) - 1;
        while (i > 0) {
            ans.push_back(chosen[i]);
            i = par[i];
        }
        
        return ans;
    }
};
