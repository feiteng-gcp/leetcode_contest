class Solution {
public:
    
    vector<int> v;
    int dp[1<<16];
    int par[1<<16];
    int n;
    
    int memo(int state) {
        //cout << state << endl;
        if (state == 0) return 0;
        if (dp[state] != -1) return dp[state];
        
        int res = -1;
        for (int i = 0; i < v.size(); i++) {

            int ss = state;
            for (int j = 0; j < n; j++) {
                if ( (ss & (1<<j)) && (v[i] & (1<<j))) {
                    ss ^= (1<<j);
                }
            }
            if (ss != state) {
                int k = 1 + memo(ss);
                if (res == -1) {
                    res = k;
                    par[state] = i;
                } else if (res > k) {
                    res = k;
                    par[state] = i;
                }
            }
        }
        return dp[state] = res;
    }
    
    vector<int> smallestSufficientTeam(vector<string>& rs, vector<vector<string>>& people) {
        map<string, int> m;
        for (int i = 0; i < rs.size(); i++)
            m[ rs[i] ] = i;
        v.clear();
        n = rs.size();
        for (int i = 0; i < people.size(); i++) {
            int state = 0;
            for (int j = 0; j < people[i].size(); j++) {
                if (m.find(people[i][j]) != m.end()) {
                    state |= (1<< m[people[i][j]]);
                }
            }
            v.push_back(state);
        }
        
        for (int j = 0; j < (1<<n); j++) 
            dp[j] = -1, par[j];

        memo((1<<n) - 1);
        
        vector<int> res;
        int state = (1<<n) - 1;
        while(state) {
            int p = par[state];
            res.push_back(p);

            for (int i = 0; i < n; i++) {
                if ( (state & (1<<i)) && (v[p] & (1<<i))) {
                    state ^= (1<<i);
                }
            }
        }
        sort(res.begin(), res.end());
        return res;
    }
};
