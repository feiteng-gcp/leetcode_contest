typedef long long ll;
typedef vector<int> VI;
typedef pair<int,int> PII;

#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))

const int INF = (int)1E9;
#define MAXN 100005

map<string, int> skills;
int dp[66][1<<16], pre[66][1<<16];
class Solution {
public:
  vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
    skills.clear();
    int m = req_skills.size();
    REP(i,0,m) skills[req_skills[i]] = i;

    int n = people.size();
    FILL(pre, -1);
    REP(i,0,n+1) REP(j,0,1<<m) dp[i][j] = INF;

    dp[0][0] = 0;
    REP(i,1,n+1) {
      int mask = 0;
      REP(j,0,people[i-1].size()) {
        string s = people[i-1][j];
        if (skills.find(s) == skills.end()) continue;
        mask |= 1 << skills[s];
      }
      REP(pm,0,1<<m) {
        if (dp[i-1][pm] == INF) continue;


        int nmask = pm | mask;
        if (dp[i-1][pm] + 1 < dp[i][nmask]) {
          dp[i][nmask] = dp[i-1][pm] + 1;
          pre[i][nmask] = pm;
        }

        if (dp[i-1][pm] < dp[i][pm]) {
          dp[i][pm] = dp[i-1][pm];
          pre[i][pm] = pm;
        }
      }
    }

    int cur = (1<<m) - 1, i = n;
    assert(dp[n][cur] != INF);
    VI ans;
    while (cur > 0) {
      int pm = pre[i][cur];
      if (pm < cur) ans.push_back(i - 1);
      cur = pm;
      i--;
    }

    sort(ans.begin(), ans.end());
    return ans;
  }
};
