class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        if (req_skills.empty()) {
            return vector<int>();
        }
        // skill, people
        unordered_map<int, vector<int>> dp;
        dp[0]=vector<int>();
        unordered_map<string, int> sk;
        for (int i=0; i<req_skills.size(); i++) {
            sk[req_skills[i]]=i;
        }
        for (int i=0; i<people.size(); i++) {
            unordered_map<int, vector<int>> dp_next;
            dp_next[0]=vector<int>();
            int curr_sk=0;
            for (int j=0; j<people[i].size(); j++) {
                if(sk.find(people[i][j])!=sk.end()) {
                    curr_sk = (curr_sk | (0x1<<sk[people[i][j]]));
                }
            }
            dp_next[curr_sk]=vector<int>({i});
            for (auto it:dp) {
                if (dp_next.find(it.first)==dp_next.end() || it.second.size()<dp_next[it.first].size()) {
                    dp_next[it.first]=it.second;
                }
                int new_sk=(it.first | curr_sk);
                if (dp_next.find(new_sk)==dp_next.end() || it.second.size()+1<dp_next[new_sk].size()) {
                    dp_next[new_sk]=it.second;
                    dp_next[new_sk].push_back(i);
                }
            }
            dp=dp_next;
        }
        vector<int> ret;
        for (auto it:dp) {
            if (it.first==(0x1<<req_skills.size())-1 && (ret.empty() || it.second.size()<ret.size())) {
                ret=it.second;
            }
        }
        return ret;
    }
};
