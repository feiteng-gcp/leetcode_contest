class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& a, vector<vector<string>>& b) {
        vector<int>c;
        vector<int>ret;
        int n=(int)a.size();
        for(int i=0;i<b.size();i++)
        {
            set<string>s;
            for(int j=0;j<b[i].size();j++)
                s.insert(b[i][j]);
            int num=0;
            for(int j=0;j<a.size();j++)
                if(s.find(a[j])!=s.end()){
                    num|=(1<<j);
                }
            c.push_back(num);
        }
        int dp[70005],p[70005],l[70005];
        for(int i=0;i<(1<<16);i++)
            dp[i]=-1;
        dp[0]=0;
        for(int i=0;i<c.size();i++){
            for(int j=0;j<(1<<n);j++){
                if(dp[j]>=0&&(dp[j|c[i]]==-1||dp[j|c[i]]>dp[j]+1)){
                    dp[j|c[i]]=dp[j]+1;
                    p[j|c[i]]=i;
                    l[j|c[i]]=j;
                }
            }
        }
        int num=(1<<n)-1;
        while(num){
            ret.push_back(p[num]);
            num=l[num];
        }
        return ret;
    }
};
