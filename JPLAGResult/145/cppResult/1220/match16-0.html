class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        int N = req_skills.size(), M = people.size();
        map<string,int> skill_index;
        for (int i = 0; i < N; i++) {
            skill_index[req_skills[i]] = i;
        }
        int covered[M];
        for (int i = 0; i < M; i++) {
            covered[i] = 0;
            for (const string &skill : people[i]) {
                if (skill_index.count(skill)) {
                    covered[i] |= (1 << skill_index[skill]);
                }
            }
        }
        int covered_state_num = (1 << N);
        int min_num[covered_state_num];
        pair<int,int> prev[covered_state_num];
        min_num[0] = 0;
        for (int i = 1; i < covered_state_num; i++) {
            min_num[i] = M;
        }
        for (int i = 0; i < M; i++) {
            for (int covered_state = covered_state_num - 1; covered_state >= 0; covered_state--) {
                int next_covered_state = (covered_state | covered[i]);
                if (min_num[covered_state] + 1 < min_num[next_covered_state]) {
                    min_num[next_covered_state] = min_num[covered_state] + 1;
                    prev[next_covered_state] = {covered_state, i};
                }
            }
        }
        vector <int> ans;
        int cur = covered_state_num - 1;
        while (cur != 0) {
            auto P = prev[cur];
            ans.push_back(P.second);
            cur = P.first;
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
