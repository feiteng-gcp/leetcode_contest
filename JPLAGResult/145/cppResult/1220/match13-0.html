class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        int K = req_skills.size(), N = people.size();
        unordered_map<string, int> id;
        for (int i=0; i<K; i++) id[req_skills[i]] = i;
        vector<int> skill(N, 0);
        for (int i=0; i<N; i++) {
            for (string& s : people[i]) skill[i] |= (1<<id[s]);
        }
        
        // bfs
        vector<int> dist(1<<K, INT_MAX), last(1<<K, -1), prestate(1<<K, 0);
        queue<int> q({0});
        dist[0] = 0;
        while (q.size()) {
            int n = q.front(); q.pop();
            if (n == (1<<K)-1) break;
            for (int i=0; i<N; i++) {
                int m = n|skill[i];
                if (m != n) {
                    if (dist[m] > dist[n] + 1) {
                        dist[m] = dist[n]+1;
                        last[m] = i;
                        prestate[m] = n;
                        q.push(m);
                    }
                }
            }
        }
        vector<int> res;
        for (int state=(1<<K)-1; state!=0; state = prestate[state])
            res.push_back(last[state]);
        return res;
    }
};
