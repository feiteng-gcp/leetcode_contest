class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& skills, vector<vector<string>>& people) {
        vector<vector<int>> v((1<<skills.size()));
        map<int, int> m;
        for (int i = 0; i < people.size(); ++i) {
            for (int j = 0; j < people[i].size();j++)
                for (int k = 0; k < skills.size(); k++)
                    if (skills[k] == people[i][j]) m[i] += 1 << k;
        }

        for (int i = 0; i < v.size(); ++i) {
            if (i > 0 && v[i].size() == 0) continue;
            for (int j = 0; j < people.size(); ++j) {
                int next = m[j] | i;
                if (next == i) continue;
                if (v[next].size() == 0 || v[next].size() > v[i].size() + 1) {
                    v[next] = vector<int>(v[i].begin(), v[i].end()); 
                    v[next].push_back(j);
                }
            }
        }
        
        return v[v.size() - 1];
    }
};
