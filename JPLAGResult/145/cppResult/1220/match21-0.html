class Solution {
public:
  vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
    int n=req_skills.size();
    map<string, int> idx;
    for(int i=0;i<n;i++) idx[req_skills[i]]=i;


    int m=people.size();
    vector<int> ss(m,0);
    for(int i=0;i<m;i++){
      for(string s:people[i]){
        if(!idx.count(s)) continue;
        ss[i]|=1<<idx[s];
      }
    }

    int s=1<<n;
    vector< vector<int> > dp(m+1,vector<int>(s,m+1));
    vector< vector<int> > pr(m+1,vector<int>(s,-1));
    dp[0][0]=0;
    for(int i=0;i<m;i++){
      for(int b=0;b<s;b++)
        dp[i+1][b]=dp[i][b];

      for(int b=0;b<s;b++){
        if(dp[i][b]>m) continue;
        int nb=b|ss[i];
        if(dp[i+1][nb]<=dp[i][b]+1) continue;
        dp[i+1][nb]=dp[i][b]+1;
        pr[i+1][nb]=b;
      }
    }
    int bit=s-1;
    vector<int> res;
    for(int i=m;i;i--){
      if(pr[i][bit]<0) continue;
      res.emplace_back(i-1);
      bit=pr[i][bit];
    }
    reverse(res.begin(),res.end());
    return res;
  }
};
