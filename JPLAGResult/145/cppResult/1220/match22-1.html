int dp[1 << 16];
vector<int> rec[1 << 16];

class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        int ns = req_skills.size();

        map<string, int> mp;
        for (int i = 0; i < ns; i++) {
            mp[req_skills[i]] = 1 << i;
        }
        int n = people.size();
        vector<int> pp(n);
        for (int i = 0; i < n; i++) {
            pp[i] = 0;
            for (string j: people[i]) {
                pp[i] |= mp[j];
            }
        }
        memset(dp, 1, sizeof(dp));
        dp[0] = 0;
        rec[0].clear();
        for (int msk = 0; msk < (1 << ns); msk++) {
            for (int i = 0; i < n; i++) {
                int nmsk = msk | pp[i];
                if (dp[nmsk] > dp[msk] + 1) {
                    dp[nmsk] = dp[msk] + 1;
                    rec[nmsk] = rec[msk];
                    rec[nmsk].push_back(i);
                }
            }
        }
        return rec[(1 << ns) - 1];
    }
};
