class Solution {
public:
  vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
    int R = req_skills.size();
    int P = people.size();
    map<string, int> M;
    for (int i = 0; i < R; i++) M[req_skills[i]] = i;
    vector<int> PP(P);
    for (int i = 0; i < P; i++) {
      for (auto s: people[i]) PP[i] |= 1 << M[s]; 
    }
    vector<vector<int>> dp(P + 1, vector<int>(1<<R, P + 1));
    vector<vector<long long>> used(P + 1, vector<long long>(1<<R, 0));

    dp[0][0] = 0;
    for (int i = 0; i < P; i++) {
      for (int s = 0; s < 1 << R; s++) {
        dp[i+1][s] = dp[i][s];
        used[i+1][s] = used[i][s];
      }
      for (int s = 0; s < 1 << R; s++) {
        if (dp[i+1][s|PP[i]] > dp[i][s] + 1) {
          dp[i+1][s|PP[i]] = dp[i][s] + 1;

          used[i+1][s|PP[i]] = used[i][s] | 1ll << i;
        }
      }
    }
    vector<int> ans;
    for (int i = 0; i < P; i++) {
      if ((1ll << i) & used[P][(1<<R) - 1]) ans.push_back(i);
    }
    return ans;
  }
};
