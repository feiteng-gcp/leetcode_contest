const int N = 16;
const int M = 60 + 1;
int dp[M][1 << N];
pair<int, int> path[M][1 << N];

class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        map<string, int> idx;
        int n = req_skills.size();
        for (int i = 0; i < n; ++i) idx[req_skills[i]] = i;
        int m = people.size();
        vector<int> mask(m);
        for (int i = 0; i < m; ++i) {

            for (auto& s : people[i]) {
                mask[i] |= 1 << idx[s];
            }
        }
        memset(dp, 255, sizeof(dp));
        dp[0][0] = 0;
        for (int i = 0; i < m; ++i) {
            copy(dp[i], dp[i] + (1 << n), dp[i + 1]);
            copy(path[i], path[i] + (1 << n), path[i + 1]);
            for (int k = 0; k < (1 << n); ++k) {
                if (dp[i][k] < 0) continue;
                int nxt_k = k | mask[i];
                if (dp[i + 1][nxt_k] < 0 || dp[i + 1][nxt_k] > dp[i][k] + 1) {
                    dp[i + 1][nxt_k] = dp[i][k] + 1;
                    path[i + 1][nxt_k] = {i, k};
                }
            }
        }
        cout << (int)dp[m][(1 << n) - 1] << endl;
        vector<int> ret;
        for (int i = m, k = (1 << n) - 1; k; ) {
            int j = path[i][k].first;
            int nxt_k = path[i][k].second;
            ret.push_back(j);
            //cout << i << " " << k << " " << dp[i][k] << " " << j << " " << nxt_k << dp[j][nxt_k] << endl;
            i = j;
            k = nxt_k;
        }
        return ret;
    }
};
