class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        int n = static_cast<int>(req_skills.size());
        int m = static_cast<int>(people.size());

        unordered_map<string, int> indice;
        for (int i = 0; i < n; ++i) {
            indice[req_skills[i]] = i;
        }
        vector<int> skills(m, 0);
        for (int i = 0; i < m; ++i) {
            for (auto skill : people[i]) {
                if (indice.count(skill)) {
                    skills[i] |= 1 << indice[skill];
                }
            }
        }
        vector<vector<int>> dp(m, vector<int>(1 << n, m + 1));
        vector<vector<pair<int, int>>> path(m, vector<pair<int, int>>(1 << n, {-1, -1}));
        for (int i = 0; i < m; ++i) {
            if (i > 0) {
                for (int j = 0; j < (1 << n); ++j) {
                    dp[i][j] = dp[i - 1][j];
                    path[i][j] = {-1, j};
                }
            }
            dp[i][skills[i]] = 1;
            path[i][skills[i]] = {i, -1};

            if (i > 0) {
                for (int j = 0; j < (1 << n); ++j) {
                    int masked = j | skills[i];
                    if (dp[i][masked] > dp[i - 1][j] + 1) {
                        dp[i][masked] = dp[i - 1][j] + 1;
                        path[i][masked] = {i, j};
                    }
                }
            }
        }
        
        /*for (int i = 0; i < m; ++i) {
            cout << skills[i] << ":\t";
            for (int j = 0; j < (1 << n); ++j) {
                cout << dp[i][j] << " ";
            }
            cout << endl;
        }
        cout << endl;
        
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < (1 << n); ++j) {
                cout << "{" << path[i][j].first << " " << path[i][j].second << "}" << " ";
            }
            cout << endl;
        }
        cout << endl;*/
        
        vector<int> workers;
        int last = (1 << n) - 1;
        for (int i = m - 1; i >= 0; --i) {
            if (path[i][last].first != -1) {
                workers.push_back(path[i][last].first);
            }
            if (path[i][last].second == -1) {
                break;
            }
            last = path[i][last].second;
        }
        return workers;
    }
};
