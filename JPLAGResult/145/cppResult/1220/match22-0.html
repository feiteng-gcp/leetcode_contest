class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        unordered_map<string, int> skill_map;
        for (int i = 0; i < req_skills.size(); i++) {
            skill_map[req_skills[i]] = i;
        }
        int n = people.size();
        vector<int> p(n, 0);
        for (int i = 0; i < n; i++) {
            int mask = 0;
            for (auto& s : people[i]) {
                int id = skill_map[s];
                mask |= 1 << id;
            }
            p[i] = mask;
        }
        int bits = skill_map.size();
        int max_mask = (1 << bits) - 1;
        vector<vector<int>> dp(max_mask + 1, vector<int>());
        dp[p[0]].push_back(0);
        
        for (int i = 1; i < n; i++) {
            dp[p[i]] = {i};
            for (int m = 0; m <= max_mask; m++) {
                if (!dp[m].empty()) {
                    int next_m = m | p[i];
                    int next_count = dp[m].size() + 1;
                    if (dp[next_m].empty() || dp[next_m].size() > next_count) {
                        dp[next_m] = dp[m];
                        dp[next_m].push_back(i);
                    }
                }
            }
        }
        return dp[max_mask];
    }

};
