class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        unordered_map<string, int> skills;

        for (int i = 0, x = 1; i < req_skills.size(); ++i, x *= 2) {
            skills[req_skills[i]] = x;
        }
        int full = (1 << req_skills.size()) - 1;
        vector<int> ps;
        for (int i = 0; i < people.size(); ++i) {
            int x = 0;
            for (auto& s: people[i]) {
                x |= skills[s];
            }
            if (x == full) {
                return {i};
            }
            ps.push_back(x);
        }
        unordered_map<int, vector<int>> m;
        for (int i = 0; i < ps.size(); ++i) {
            unordered_map<int, vector<int>> mm;
            for (auto& p: m) {
                int x = p.first | ps[i];
                if (mm.count(x) == 0) {
                    mm[x] = p.second;
                    mm[x].push_back(i);
                } else if (p.second.size() + 1 < mm[x].size()) {
                    mm[x] = p.second;
                    mm[x].push_back(i);
                }
                if (mm.count(p.first) == 0) {
                    mm[p.first] = p.second;
                } else if (p.second.size() < mm[p.first].size()) {
                    mm[p.first] = p.second;
                }
            }
            mm[ps[i]] = {i};
            m.swap(mm);
        }
        return m[full];
    }
};
