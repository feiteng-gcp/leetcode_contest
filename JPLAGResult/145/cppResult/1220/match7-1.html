class Solution {
public:
    vector<int> smallestSufficientTeam(vector<string>& req_skills, vector<vector<string>>& people) {
        int dp[1<<16];
        vector<int> res[1<<16];
        memset(dp, -1, sizeof dp);
        vector<int> p;
        for(auto v : people)
        {
            int tmp = 0;
            for(auto s : v)
            {
                for(int i = 0; i < req_skills.size(); ++i)
                    if(s == req_skills[i])
                    {
                        tmp += 1<<i;
                        break;
                    }
            }
            p.push_back(tmp);
        }

        dp[0] = 0;
        for(int j = 0; j < p.size(); ++j)
        {
            int s = p[j];
            for(int i = 0; i < (1<<req_skills.size()); ++i)
                if(dp[i] != -1)
                {
                    if(dp[i|s] == -1 || dp[i]+1 < dp[i|s])
                    {
                        dp[i|s] = dp[i]+1;
                        res[i|s].assign(res[i].begin(), res[i].end());
                        res[i|s].push_back(j);
                    }
                    
                }
        }
        return res[(1<<req_skills.size())-1];
    }
};
