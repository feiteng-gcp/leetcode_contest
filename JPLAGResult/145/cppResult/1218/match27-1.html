/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int a;
    int b;
    void help(TreeNode* root, int cur, map<int, int>& dp)
    {
        if(!root) return;
        dp[root->val] = cur;
        help(root->left, cur+1, dp);
        help(root->right, cur+1, dp);
    }
    
    TreeNode* res;

    int dfs(TreeNode* root,map<int, int>& dp)
    {
        if(!root) return 0;
        if(dp[root->val]==a)
        {
            if(b==1) res=root;
            return 1;
        }
        int cur = dfs(root->left, dp) + dfs(root->right, dp);
        if(cur == b&&!res)
        {
            res=root;
        }
        return cur;
    }
    
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        map<int, int> dp;
        help(root, 0, dp);
        a=-1;
        b=0;
        for(auto i:dp)
        {
            if(i.second>a)
            {
                a=i.second;
                b=1;
            }
            else if(i.second==a)
            {
                b++;
            }
        }
        res = nullptr;
        dfs(root, dp);
        return res;
        
        
        
    }
};
