/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        int maxD = 0;
        dfs(root, 0, maxD);
        TreeNode* res = root;
        dfs2(root, 0, maxD, res);
        // if (res != NULL)  { cout << res->val << endl; }
        return res;
    }
    
private:

    int dfs(TreeNode* root, int d, int& maxD) {
        if (root == NULL) {
            return d;
        }
        int L = dfs(root->left, d + 1, maxD);
        int R = dfs(root->right, d + 1, maxD);
        
        maxD = max(maxD, max(L, R));
        
        return max(L, R);
    }
    
    int dfs2(TreeNode* root, int d, int maxD, TreeNode* & res) {
        if (root == NULL) {
            return d;
        }
        if (d == maxD) {
            res = root;
        }

        int L = dfs2(root->left, d + 1, maxD, res);
        int R = dfs2(root->right, d + 1, maxD, res);
        if (L == maxD && R == maxD) {
            res = root;
        }
        return max(L, R);
    }
};
