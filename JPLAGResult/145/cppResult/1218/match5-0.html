/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int mx;
    int dfs(TreeNode* root, int d, TreeNode* &ans) {
        if(root == NULL) return d;
        else d++;
        int d1 = dfs(root->left, d, ans);
        int d2 = dfs(root->right, d, ans);
        if(d1 == d2 && d1 >= mx) {
            ans = root;
            mx = d1;
        }
        return max(d1, d2);
    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        mx = 0;
        TreeNode* ans;
        dfs(root, 0, ans);
        return ans;
    }
};
