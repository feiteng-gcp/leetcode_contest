/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        if (!root || (!root->left && !root->right)) {
            return root;
        }
        unordered_map<int, int> m;
        getDepth(root, m, 0);
        int d = m[root->val];
        return f(root, m, d);
    }
    
    int getDepth(TreeNode* t, unordered_map<int, int>& m, int d) {
        if (!t) {
            return d;
        }
        int ld = getDepth(t->left, m, d + 1);
        int rd = getDepth(t->right, m, d + 1);
        m[t->val] = max(ld, rd);
        return m[t->val];
    }
    
    TreeNode* f(TreeNode* t, unordered_map<int, int>& m, int d) {
        if (!t || (!t->left && !t->right)) {
            return t;
        }
        if (!t->left) {
            return f(t->right, m, d);
        } else if (!t->right) {
            return f(t->left, m, d);

        }
        if (m[t->left->val] == d && m[t->right->val] == d) {
            return t;
        } else if (m[t->left->val] == d) {
            return f(t->left, m, d);
        } else {
            return f(t->right, m, d);
        }
    }
};
