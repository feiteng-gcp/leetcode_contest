/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int best_dep;
    set<int> A;
    void traverse(TreeNode* p, int dep) {
        if (!p) return;
        if (!p->left && !p->right) {
            if (best_dep == dep) {
                A.insert(p->val);
            } else if (best_dep < dep) {
                best_dep = dep;
                A.clear();
                A.insert(p->val);
            }
        }
        traverse(p->left, dep + 1);
        traverse(p->right, dep + 1);
    }
    TreeNode* ret;
    int solve(TreeNode* p) {
        if (!p) return 0;
        if (!p->left && !p->right && A.count(p->val)) {
            if (A.size() == 1 && !ret) ret = p;
            return 1;
        }
        int L = solve(p->left);
        int R = solve(p->right);
        if (L + R == A.size() && !ret) ret = p;
        return L + R;
    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        best_dep = 0;
        A.clear();
        traverse(root, 0);
        for (auto& it : A) cout << it << " ";
        cout << endl;
        ret = nullptr;
        solve(root);
        return ret;
    }
};
