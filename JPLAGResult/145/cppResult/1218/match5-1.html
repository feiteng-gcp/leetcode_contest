/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int ans;
    TreeNode* ret = nullptr;
    int dfs(TreeNode* root, int deep) {
        int dp_s = 0;
        if (!root) return 0;

        int ld = dfs(root->left, deep + 1);
        int rd = dfs(root->right, deep + 1);
        if (ld == rd && deep + ld >= ans) {
            ans = deep + ld;
            ret = root;
        }
        return max(ld, rd) + 1;
        
    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        ans = 0;
        dfs(root, 0);
        return ret;
    }
};
