/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<TreeNode*, int> dis;
    map<TreeNode*, TreeNode*> pre;
    int depth;
    vector<TreeNode*> leaf;
    void dfs(TreeNode *x)
    {
        if (!x) return;
        if (x->left)
        {
            dis[x->left] = dis[x]+1;
            pre[x->left] = x;
            dfs(x->left);
        }
        if (x->right)
        {
            dis[x->right] = dis[x]+1;
            pre[x->right] = x;
            dfs(x->right);
        }
        if (!x->left && !x->right)
        {
            if (dis[x] > depth)
            {
                depth = dis[x];
                leaf.clear();
            }
            if (dis[x] == depth)
                leaf.push_back(x);
        }
    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        dis.clear();
        dis[root] = 0;
        pre.clear();
        depth = -1;
        leaf.clear();
        dfs(root);
        while (1)
        {
            int flag = 0;
            for (int i = 1; i < leaf.size(); ++ i)
                if (leaf[i] != leaf[i-1])
                {
                    flag = 1;
                    break;
                }
            if (!flag) break;
            for (int i = 0; i < leaf.size(); ++ i)
                leaf[i] = pre[leaf[i]];
        }
        return leaf[0];
    }
};
