/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* res = NULL;
    int dep = 0;
    
    int helper(TreeNode* root, int cur_dep){
        if(root == NULL)
            return 0;

        int l = helper(root->left, cur_dep + 1);
        int r = helper(root->right, cur_dep + 1);
        if(l == r && cur_dep + r + 1 >= dep){
            res = root;
            dep = cur_dep + r + 1;
        }
        return max(r, l) + 1;
    }
    
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        helper(root, 0);
        return res;
    }
};
