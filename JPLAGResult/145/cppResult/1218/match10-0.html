class Solution {
	map<int, set<TreeNode*>> vlist;
	map<TreeNode*, TreeNode*> parent;
	void dfs(TreeNode *root, int d) {
		if (root == NULL) return;
		if (root->left != NULL) {
			dfs(root->left, d + 1);
			parent[root->left] = root;
		}
		if (root->right != NULL) {
			dfs(root->right, d + 1);
			parent[root->right] = root;
		}

		if ((root->left == NULL) && (root->right == NULL)) vlist[d].insert(root);
	}

public:
	TreeNode* lcaDeepestLeaves(TreeNode* root) {
		vlist.clear();
		dfs(root, 0);

		set<TreeNode*> s1 = vlist.rbegin()->second, s2;

		while (s1.size() > 1) {
			s2.clear();
			for (auto p : s1) s2.insert(parent[p]);
			s1 = s2;
		}

		return *s1.begin();
	}
};
