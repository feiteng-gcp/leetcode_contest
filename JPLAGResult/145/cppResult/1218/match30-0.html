/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int max_dep;
    
    void get_dep(TreeNode* root, int dep) {
        if (!root)
            return;
        max_dep = max(max_dep, dep);
        get_dep(root->left, dep + 1);
        get_dep(root->right, dep + 1);
    }
    
    vector<vector<TreeNode*>> vv;
    
    void get_path(TreeNode* node, int dep, vector<TreeNode*>& v) {
        if (!node)
            return;
        v.push_back(node);
        if (dep == max_dep) {
            vv.push_back(v);
        } else {
            get_path(node->left, dep + 1, v);
            get_path(node->right, dep + 1, v);
        }
        v.pop_back();
    }
    
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        max_dep = 0;
        get_dep(root, 0);
        vv.clear();
        vector<TreeNode*> v;
        get_path(root, 0, v);
        TreeNode* ans = nullptr;
        for (int i = 0; i <= max_dep; i++) {
            bool same = true;
            for (auto & v : vv) {
                if (v[i] != vv[0][i]) {
                    same = false;
                    break;
                }
            }
            if (!same)
                break;
            ans = vv[0][i];
        }
        return ans;
    }
};
