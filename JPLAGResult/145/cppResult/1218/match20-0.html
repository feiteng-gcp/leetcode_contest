/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int max_h = 0;
    int num_max_h = 0;
    int low = -1;
    TreeNode* res = NULL;
    void getHeight(TreeNode* root, int h) {
        if (!root) return;
        getHeight(root->left, h+1);
        getHeight(root->right, h+1);
        if (!root->left && !root->right) max_h = max(max_h, h);
    }
    void getNum(TreeNode* root, int h, int& res) {
        if (!root) return;
        if (h == max_h) res ++;
        getNum(root->left, h+1, res);
        getNum(root->right, h+1, res);
    }
    int dfsNum(TreeNode* root, int h) {
        if (!root) return 0;
        int num = 0, l=0, r=0;
        if (h == max_h) num = 1;
        else {
            l = dfsNum(root->left, h+1), r = dfsNum(root->right, h+1);
            num = l + r;
        }
        if (num == num_max_h) {
            if (h > low) {
                low = h;
                res = root;
            }
        }
        return num;
    }
public:
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        getHeight(root, 0);
        cout << max_h << '\n';
        getNum(root, 0, num_max_h);
        cout << num_max_h << '\n';
        dfsNum(root, 0);
        return res;
    }
};
