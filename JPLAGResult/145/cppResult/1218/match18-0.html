/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    pair<int,int> ret;
    TreeNode* ans;
    pair<int,int> dfs(TreeNode* root,int d){
        if(root->left==NULL&&root->right==NULL) {
            if(ret.first<d){
                ret=make_pair(d,1);
                ans=root;
            }
            return make_pair(d,1);
        }
        pair<int,int>l=make_pair(-1,-1);
        pair<int,int>r=make_pair(-1,-1);
        if(root->left!=NULL)l=dfs(root->left, d+1);
        if(root->right!=NULL)r=dfs(root->right,d+1);
        if(l.first==-1||l.first<r.first){
            if(ret.first<r.first||(ret.first==r.first&&ret.second<r.second)){
                ret = r;
                ans = root;
            }
            return r;
        } else {
            if(r.first==-1||l.first>r.first){
                if(ret.first<l.first||(ret.first==l.first&&ret.second<l.second)){
                ret = l;
                    ans=root;
                }
                return l;
            }
            l.second+=r.second;
            if(ret.first<l.first||(ret.first==l.first&&ret.second<l.second)){
                ret = l;
                ans=root;
                }
                return l;
        }
    }
public:
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        ret = make_pair(-1,-1);
        dfs(root,0);
        return ans;
    }
};
