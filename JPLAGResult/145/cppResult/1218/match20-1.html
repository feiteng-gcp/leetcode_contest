/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        int depth=findDepth(root);
        addLeaf(root,depth);
        //for(auto &i:dl)cout<<i->val<<endl;
        helper(root,0);
        return aans;
    }
    
    int dp=INT_MIN;
    TreeNode* aans;
    unordered_set<TreeNode*> dl;
    int num=0;
    
    int findDepth(TreeNode* root){
        if(!root)return 0;
        return max(findDepth(root->left),findDepth(root->right))+1;
    }
    
    void addLeaf(TreeNode* root,int remDep){
        if(!root)return;
        if(remDep==1){
            dl.insert(root);
            num++;
        }
        addLeaf(root->left,remDep-1);
        addLeaf(root->right,remDep-1);

    }
    
    int helper(TreeNode* root,int dep){
        if(!root)return 0;
        int ans=0;
        ans+=helper(root->left,dep+1);
        ans+=helper(root->right,dep+1);
        if(dl.find(root)!=dl.end())ans++;
        if(ans==num){
            if(dep>dp){
                dp=dep;
                aans=root;
            }
        }
        return ans;
    }
};
