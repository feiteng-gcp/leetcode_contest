/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    /*

    int get_dep(TreeNode *cur)
    {
        if(!cur) return 0;
        return max(get_dep(cur->left), get_dep(cur->right))+1;
    }
    
    bool my_f(TreeNode *cur, int dep, set<pair<int, TreeNode*>> &v)
    {
        if(!cur) return false;
        int l = my_f(cur->left, dep-1, v);
        int r = my_f(cur->right, dep-1, v);
        if(dep == 1 || l || r)
        {
            v.insert(make_pair(dep, cur));
            return true;
        }
        return false;
    }
    
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        int dep = get_dep(root);
        set<pair<int, TreeNode*>> v;
        my_f(root, dep, v);
        map<int, int> m;
        int res = 0;
        for(auto i : v) m[i.first]++;
        for(auto i : m)
        {
            if(i.second == 1)
            {
                res = i.first;
                break;
            }
        }
        for(auto i : v)
            if(i.first == res)
                return i.second;
        return nullptr;
    }
};
