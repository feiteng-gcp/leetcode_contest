/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int get_dep(TreeNode* root) {
        if (root == NULL) {
            return 0;
        }
        return max(get_dep(root->left), get_dep(root->right)) + 1;
    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        TreeNode* la = NULL;
        TreeNode* ra = NULL;
        // root->val = 0;
        int dl = 0;
        int dr = 0;

        if (root->left) {
            la = lcaDeepestLeaves(root->left);
            dl = get_dep(root->left);
            // dl = root->left->val;
            // root->val = max(root->val, dl);
        }
        if (root->right) {
            ra = lcaDeepestLeaves(root->right);
            dr = get_dep(root->right);
            // dr = root->right->val;
            // root->val = max(root->val, dr);
        }
        // root->val += 1;
        if (dl > dr) {
            // cout << root->val << " L" << endl;
            return la;
        }
        if (dr > dl) {
            // cout << root->val << " R" << endl;
            return ra;
        }
        // cout << root->val << " self" << endl;
        return root;
    }
};
