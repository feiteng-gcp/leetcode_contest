/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<TreeNode*, int> deep;
    map<TreeNode*, TreeNode*> parent;
    int maxDeep = 0;
    void dfs(TreeNode* root, int _deep) {
        if (maxDeep < _deep) {
            maxDeep = _deep;
        }
        deep[root] = _deep;
        if (root->left) {
            dfs(root->left, _deep + 1);
            parent[root->left] = root;
        } 
        if (root->right) {
            dfs(root->right, _deep + 1);
            parent[root->right] = root;
        }
    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        deep.clear();
        parent.clear();
        maxDeep = 0;
        dfs(root, 0);
        set<TreeNode*> leaf;
        for (map<TreeNode*, int>::iterator it = deep.begin();it != deep.end();it++) {
            if (it->second == maxDeep) {
                leaf.insert(it->first);
            }
        }
        while (leaf.size() > 1) {
            set<TreeNode*> temp;
            for (set<TreeNode*>::iterator it = leaf.begin();it != leaf.end();it++) {
                temp.insert(parent[*it]);
            }
            leaf = temp;
        }
        return *(leaf.begin());
    }
};
