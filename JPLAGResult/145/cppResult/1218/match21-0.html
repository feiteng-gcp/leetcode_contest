/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> d=vector<int>(1001, -1);
    int helper(TreeNode* root) {
        if(root) {
            int l=helper(root->left);
            int r=helper(root->right);
            if (l<r) {
                d[root->val]=1+r;
                return 1+r;
            } else {
                d[root->val]=1+l;
                return 1+l;
            }
        }
        return 0;
    }
    TreeNode* helper2(TreeNode* root) {
        if (root) {
            if (!root->left && !root->right) {
                return root;
            }
            if (!root->left) {
                return helper2(root->right);
            }
            if (!root->right) {
                return helper2(root->left);
            }
            if (d[root->left->val]==d[root->right->val]){
                return root;
            }
            if (d[root->left->val]>d[root->right->val]){
                return helper2(root->left);
            } else {
                return helper2(root->right);    
            }
        }
        return nullptr;
    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        helper(root);
        return helper2(root);
    }
};
