/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        map<int, int> depth;
        map<TreeNode*, TreeNode*> parent;
        map<int, TreeNode*> node;
        depth[root->val] = 0;
        calculateDepths(root, depth, parent, node);
        int max_depth = 0;
        set<TreeNode*> leaves;
        for (map<int, int>::iterator it = depth.begin(); it != depth.end(); ++it) {
            if (it->second > max_depth) {
                max_depth = it->second;
                leaves.clear();
                leaves.insert(node[it->first]);
                // cout << "NEW: " << it->first << endl;

            } else if (it->second == max_depth) {
                leaves.insert(node[it->first]);
                // cout << "ADDED: " << it->first << endl;
            }
        }
        // cout << leaves.size() << endl;
        
        while (leaves.size() > 1) {
            set<TreeNode*> newLeaves;
            for (set<TreeNode*>::iterator it = leaves.begin(); it != leaves.end(); ++it) {
                newLeaves.insert(parent[*it]);
            }
            leaves = newLeaves;
        }
        
        return *(leaves.begin());
    }
    
    void calculateDepths(TreeNode* root,  map<int, int>& depth,  map<TreeNode*, TreeNode*>& parent,  map<int, TreeNode*>& node) {
        node[root->val] = root;
        if (root->left != NULL) {
            depth[root->left->val] = depth[root->val] + 1;
            parent[root->left] = root;
            calculateDepths(root->left, depth, parent, node);
        }
        
        if (root->right != NULL) {
            depth[root->right->val] = depth[root->val] + 1;
            parent[root->right] = root;
            calculateDepths(root->right, depth, parent, node);
        }
    }
};
