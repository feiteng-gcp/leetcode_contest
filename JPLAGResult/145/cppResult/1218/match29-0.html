/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    int depth;
    int cnt;
    int isFind;
    TreeNode* res;
    
    void dfs(TreeNode* root, int d) {
        if (root == NULL) return;
        if (depth == d) {
            cnt ++;
        } else if (depth < d) {
            cnt = 1;
            depth = d;
        }
        dfs(root->left, d + 1);
        dfs(root->right, d + 1);
    }
    
    int dfs2(TreeNode* root, int d) {
        if (root == NULL) return 0;
        int c = 0;
        c += dfs2(root->left, d + 1);
        c += dfs2(root->right, d + 1);
        if (d == depth) c++;
        if (c == cnt && !isFind) {
            res = root;
            isFind = true;
        }
        return c;
    }
    
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        depth = 0;
        cnt = 0;
        dfs(root, 0);
        isFind = false;
        dfs2(root, 0);
        return res;
    }
};
