/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
         map<TreeNode*, vector<TreeNode*>> m;
        queue<TreeNode*> q;
        queue<int> dep;
        q.push(root);
        dep.push(0);
        m[root].push_back(root);
        map<TreeNode* ,int> lm;
        
        while(!q.empty()) {
            auto t = q.front(); q.pop();
            auto d = dep.front(); dep.pop();
            if(t->left == nullptr && t->right == nullptr) {
               lm[t] = d;
            }
            else {
                if(t->left != nullptr) {
                    auto v = m[t];
                    v.push_back(t->left);
                    m[t->left] = v;
                    q.push(t->left);
                    dep.push(d+1);
                }
                if(t->right != nullptr) {
                    auto v = m[t];
                    v.push_back(t->right);
                    m[t->right] = v;
                    q.push(t->right);
                    dep.push(d+1);
                }
            }
        }
        
        int max_d = -1;
        for(auto llm: lm) {
            if(llm.second > max_d) {
                max_d = llm.second;
            }
        }
        set<TreeNode*> l;
        for(auto llm: lm) {
            if(llm.second == max_d) {
                l.insert(llm.first);
            }
        }
        
        for(auto iter = m[*l.begin()].rbegin(); iter != m[*l.begin()].rend(); iter++) {
            int v = (*iter)->val;
            bool fail = false;

            for(auto ll: l) {
                bool found = false;
                for(auto mm: m[ll]) {
                    if(mm->val == v) {
                        found = true;
                        break;
                    }
                }
                if(!found) {
                    fail = true;
                    goto loop;
                }
            }
            loop:
            if(!fail) return (*iter);
        }
        return nullptr;

    }
};
