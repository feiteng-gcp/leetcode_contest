/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    pair<TreeNode *, int> ancestor_depth(TreeNode *root) {
        if (root->left == NULL && root->right == NULL) {
            return {root, 0}; 
        }
        if (root->left == NULL) {
            auto P = ancestor_depth(root->right);
            P.second += 1;
            return P;
        }
        if (root->right == NULL) {
            auto P = ancestor_depth(root->left);
            P.second += 1;
            return P;
        }
        else {
            auto P1 = ancestor_depth(root->left);
            auto P2 = ancestor_depth(root->right);
            if (P1.second == P2.second) {
                return {root, P1.second + 1};
            }
            else if (P1.second > P2.second) {
                P1.second += 1;
                return P1;
            }
            else {
                P2.second += 1;
                return P2;
            }
        }
    }
    TreeNode* lcaDeepestLeaves(TreeNode *root) {
        auto P = ancestor_depth(root);
        return P.first;
    }
};
