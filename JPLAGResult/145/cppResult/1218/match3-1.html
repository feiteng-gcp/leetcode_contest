/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int _depth = -1;
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        if(root==NULL) return NULL;
        _depth = deep(root);
        return help(root, 0);
    }
    
    TreeNode* help(TreeNode* root, int len){// len is the cur depth
        if(len==_depth){
            return root;
        }
        if(root==NULL){
            return NULL;
        }
        TreeNode* tempL = help(root->left, len+1);
        TreeNode* tempR = help(root->right, len+1);
        if(tempL!=NULL && tempR!=NULL){
            return root;
        }
        if(tempL!=NULL){
            return tempL;
        }
        return tempR;
    }
    
    int deep(TreeNode* root){
        if(root==NULL){
            return -1;
        }
        return max(deep(root->left), deep(root->right)) + 1;
    }
    
    
};
