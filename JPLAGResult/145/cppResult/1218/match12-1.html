/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        ans = root;
        dep = 0;
        dfs(root, 0);
        

        return ans;
    }
    
    int dfs(TreeNode *n, int d) {
        if (!n) return d;
        
        int l = dfs(n->left, d + 1);
        int r = dfs(n->right, d + 1);
        
        if (l == r) {
            if (l >= dep) {
                ans = n;
                dep = l;
            }
        }
        
        return max(l, r);
    }
    
    TreeNode *ans;
    int dep;
};
