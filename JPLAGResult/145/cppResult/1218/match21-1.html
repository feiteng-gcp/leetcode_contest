/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    TreeNode* solve(TreeNode* R, int& depth) {
        if (!R) {
            depth = 0;
            return NULL;
        }
        int depthLeft = 0, depthRight = 0;
        TreeNode* P = solve(R->left, depthLeft);
        TreeNode* Q = solve(R->right, depthRight);
        depth = 1 + max(depthLeft, depthRight);

        if (depthLeft == depthRight) {
            return R;
        }
        if (depthLeft < depthRight) {
            return Q; 
        } else {
            return P;
        }
        
    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        int depth = 0;
        return solve(root, depth);
    }
};
