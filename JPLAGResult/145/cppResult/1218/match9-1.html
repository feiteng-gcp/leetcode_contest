TreeNode *ans;

int dep, maxd, num;

class Solution {
public:
    void getmaxd(TreeNode *node, int cd) {
            if(cd > maxd) {
                maxd = cd;
                num = 1;
            }
            else if(cd == maxd) {
                num += 1;
            }
            if(node->left != NULL) {
                getmaxd(node->left, cd + 1);
            }
            if(node->right != NULL) {
                getmaxd(node->right, cd + 1);
            }
        }

    int dfs(TreeNode *node, int cd) {
        int sum = 0;
        if(cd == maxd) sum += 1;
        if(node->left != NULL) {
            sum += dfs(node->left, cd + 1);
        }
        if(node->right != NULL) {
            sum += dfs(node->right, cd + 1);
        }
        if(sum == num && ans == NULL) {
            ans = node;
        } 
        return sum;
    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        ans = NULL;
        dep = -1;
        maxd = -1;
        num = 0;
        getmaxd(root, 0);
        dfs(root, 0);
        return ans;
    }
};
