/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    pair<TreeNode*, int> lcaDeepestLeaves(TreeNode* u, int d) {
        
        
        if (u->left == nullptr && u->right == nullptr) {
            return {u, d};
        }
        
        pair<TreeNode*, int> lca1 = {nullptr, -1};
        
        if (u->left != nullptr) {
            lca1 = lcaDeepestLeaves(u->left, d+1);
        }
        
        
        pair<TreeNode*, int> lca2 = {nullptr, -1};
        if (u->right != nullptr) {
            lca2 = lcaDeepestLeaves(u->right, d+1);
        }
        
        if (lca1.second > lca2.second) {
            return lca1;
        } else if (lca1.second < lca2.second) {
            return lca2;
        } else {
            return {u, lca1.second};
        }
        
        
        
    }
    
    
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        
        pair<TreeNode*, int> lca = lcaDeepestLeaves(root, 0);
        return lca.first;
    }
};
