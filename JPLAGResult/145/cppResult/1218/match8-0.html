/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

struct ReturnVl{
    int depth;
    TreeNode * node;
    ReturnVl(int d, TreeNode * n){
        depth = d;
        node = n;
    }
};

class Solution {
public:
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        auto res = find(root);
        return res.node;
    }
    
    ReturnVl find(TreeNode * root) {
        if(root == nullptr)
            return ReturnVl(0, nullptr);
        auto l = find(root->left);
        auto r = find(root->right);
        if(l.depth==r.depth) {
            return ReturnVl(l.depth+1, root);
        }
        else
            return ReturnVl(max(l.depth, r.depth)+1, r.depth>l.depth?r.node:l.node);
    }
};
