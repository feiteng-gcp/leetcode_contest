/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int numLeaves;
    int maxDepth;
    TreeNode *res;
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        numLeaves = 0, maxDepth = -1;
        queue<TreeNode*> q;
        q.push(root);
        while(!q.empty()) {
            int n = numLeaves = q.size();
            while(n--) {
                TreeNode *cur = q.front(); q.pop();
                if(cur->left) q.push(cur->left);
                if(cur->right) q.push(cur->right);
            }
            maxDepth++;
        }
        dfs(root, 0);
        return res;
    }
    

    int dfs(TreeNode *root, int depth) {
        if(res) return 0;
        if(depth == maxDepth) {
            if(numLeaves == 1) res = root;
            return 1;
        }
        int l = 0, r = 0;
        if(root->left) l = dfs(root->left, depth + 1);
        if(res) return 0;
        if(root->right) r = dfs(root->right, depth + 1);
        if(res) return 0;
        int num = l + r;
        if(num == numLeaves) res = root;
        return num;
    }
    
    
};
