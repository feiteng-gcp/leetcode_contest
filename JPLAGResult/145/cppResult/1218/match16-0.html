/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    unordered_map<int,int> mm;
    int dp(TreeNode* root){
        if(root==NULL){
            return 0;
        }
        int l=dp(root->left);
        int r=dp(root->right);
        int ans=max(l,r)+1;
        mm[root->val]=ans;
        return ans;
    }
    
    TreeNode* dp2(TreeNode* root){
        int l=root->left==NULL?0:mm[root->left->val];
        int r=root->right==NULL?0:mm[root->right->val];
        if(l==r) return root;
        if(l>r) return dp2(root->left);
        else return dp2(root->right);
    }
public:
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        dp(root);
        return dp2(root);
    }
};
