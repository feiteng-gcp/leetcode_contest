/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
	#define xl x->left
	#define xr x->right
	#define xv x->val
	int ans=-1;int deep[1005],son[1005];
	int maxd=-1;int tot=0;
    TreeNode *ansT;
	void dfs1(TreeNode *x)
	{
		maxd=max(maxd,deep[xv]);
		if(xl!=NULL)
		{
			deep[xl->val]=deep[xv]+1;
			dfs1(xl);
		}
		if(xr!=NULL)
		{
			deep[xr->val]=deep[xv]+1;
			dfs1(xr);
		}
	}
	void dfs2(TreeNode *x)
	{

		if(deep[xv]==maxd)
			son[xv]=1;
		if(xl!=NULL)
		{
			dfs2(xl);
			son[xv]+=son[xl->val];
		}
		if(xr!=NULL)
		{
			dfs2(xr);
			son[xv]+=son[xr->val];
		}
		if(son[xv]==tot&&ans==-1)
        {
			ans=xv;
            ansT=x;
        }
	}
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
    	memset(deep,0,sizeof(deep));
    	memset(son,0,sizeof(son));
    	deep[root->val]=0;
        dfs1(root);
        for(int i=0;i<=1000;++i)
        	tot+=(deep[i]==maxd);
        dfs2(root);
        return ansT;
    }
};
