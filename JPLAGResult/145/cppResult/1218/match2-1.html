/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
  public:
    unordered_set<TreeNode*> dl;
    unordered_map<TreeNode*, int> d;
    int md;
    void calc(TreeNode* x, int t) {
      if (!x) {
        return;
      }
      d[x] = t;
      md = max(md, t);
      calc(x->left, t + 1);
      calc(x->right, t + 1);
    }
    TreeNode* ans;
    int find(TreeNode* x) {
      if (!x) {
        return 0;
      }
      int ansl = find(x->left);
      int ansr = find(x->right);
      int ansx = dl.count(x);
      if (ansl < dl.size() && ansr < dl.size() && ansl + ansr + ansx == dl.size()) {
        ans = x;
      }
      return ansl + ansr + ansx;

    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
      md = 0;
      calc(root, 0);

      for (unordered_map<TreeNode*, int>::iterator it = d.begin(); it != d.end(); it++) {
        if (it->second == md) {
          dl.insert(it->first);
        }
      }
      find(root);
      d.clear();
      dl.clear();
      return ans;
    }
};
