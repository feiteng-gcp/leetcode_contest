/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> nodes[1005];
    TreeNode* P[1005];
    int d = 0, C[1005], cur = 0, dep[1005];
    
    void recur(TreeNode* root, int val) {
        d = max(d, val);
        nodes[val].push_back(root);

        dep[root->val] = val;
        if (root->left != NULL) {
            P[root->left->val] = root;
            recur(root->left, val+1);
        }
        if (root->right != NULL) {
            P[root->right->val] = root;
            recur(root->right, val+1);
        }
    }
    
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        memset(C,0,sizeof(C));
        P[root->val] = NULL;
        recur(root, 0);
        for (auto x : nodes[d]) {
            while (x != NULL) {
                C[x->val]++;
                x = P[x->val];
            }
        }
        TreeNode* ret = NULL;
        int t = d;
        while (t >= 0) {
            for (auto x : nodes[t]) {
                //printf("Node %d has dep %d and val %d\n", x->val, t, C[x->val]);
                if (C[x->val] == nodes[d].size()) return x;
            }
            t--;
        }
        return root;
    }
};
