/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

int depth, cnt;
TreeNode* ans;
int ansdep;

class Solution {
public:
    void dfs1(TreeNode* r, int d) {
        if (d > depth) {
            depth = d;
            cnt = 1;
        } else if (d == depth) {
            cnt++;
        }
        if (r->left) dfs1(r->left, d + 1);
        if (r->right) dfs1(r->right, d + 1);
    }
    int dfs(TreeNode* r, int d) {
        int t = (d == depth);
        if (r->left) t += dfs(r->left, d + 1);
        if (r->right) t += dfs(r->right, d + 1);
        if (t == cnt) {
            if (ansdep < d) {
                ansdep = d;
                ans = r;
            }
        }
        return t;
    }
    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        depth = -1;
        cnt = 0;
        ansdep = -1;
        dfs1(root, 0);
        dfs(root, 0);
        return ans;
    }
};
