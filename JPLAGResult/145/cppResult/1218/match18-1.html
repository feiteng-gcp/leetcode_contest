/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* ancestor;
    int maxDepth = 0;


    int findAncestor(TreeNode* root, int depth)
    {
        if (root->left == NULL && root->right == NULL)
        {
            if (maxDepth < depth)
            {
                ancestor = root;
                maxDepth = depth;
            }
            return depth;
        }
        int dep;
        if (root->left == NULL)
            return(findAncestor(root->right, depth+1));
        if (root->right == NULL)
            return(findAncestor(root->left, depth+1));
        int left = findAncestor(root->left, depth+1);
        int right = findAncestor(root->right, depth+1);
        if (left == maxDepth && right == maxDepth)
            ancestor = root;
        dep = left > right ? left : right;
        return dep;
    }

    TreeNode* lcaDeepestLeaves(TreeNode* root) {
        ancestor = root;
        findAncestor(root, 0);
        return ancestor;
    }
};
