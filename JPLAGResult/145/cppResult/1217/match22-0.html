class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        unordered_map<int, int> hash;
        for(int num: arr2) {
            hash[num] = 0;
        }
        vector<int> ret;
        vector<int> last;
        for(int num: arr1) {
            if(hash.find(num) != hash.end()) {
                hash[num]++;
            }else{
                last.push_back(num);
            }
        }
        for(int num: arr2) {
            int count = hash[num];
            for(int i = 0; i < count; i++) {
                ret.push_back(num);
            }
        }
        sort(last.begin(), last.end());
        for(int num: last) {
            ret.push_back(num);
        }
        return ret;
    }
};
