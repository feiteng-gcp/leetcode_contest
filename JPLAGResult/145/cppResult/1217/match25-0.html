class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        int len1 = arr1.size();
        int len2 = arr2.size();
        vector<int> ff(2000, 0);
        vector<int> ans;
        for (int i = 0; i < len1; i ++)
            ff[arr1[i]] += 1;
        for (int i = 0; i < len2; i ++) {
            for (int j = 0; j < ff[arr2[i]]; j ++) {
                ans.push_back(arr2[i]);
            }
            ff[arr2[i]] = 0;
        }
        for (int i = 0; i < 2000; i ++) {
            if (ff[i] > 0)
                for (int j = 0; j < ff[i]; j ++)
                    ans.push_back(i);
        }
        return ans;
    }
};
