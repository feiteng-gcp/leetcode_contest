#define ALL(x) (x).begin(), (x).end()

class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        map<int,int> pos;
        int n2 = arr2.size();
        for (int i = 0; i < n2; ++i) {
            pos[arr2[i]] = i;
        }
        sort(ALL(arr1), [&](int lhs, int rhs) {
            if (pos.count(lhs) && pos.count(rhs)) {
                return pos.at(lhs) < pos.at(rhs);
            }
            if (pos.count(lhs) || pos.count(rhs)) {
                return !!pos.count(lhs);
            }
            return lhs < rhs;
        });
        return arr1;
    }
};
