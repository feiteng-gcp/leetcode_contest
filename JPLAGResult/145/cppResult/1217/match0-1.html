class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        unordered_map<int, int> m;
        for(int i = 0;i<arr2.size();i++){
            m[arr2[i]]=i;
        }
        sort(arr1.begin(), arr1.end(), [&](int x, int y)->bool{
            if(m.find(x)!=m.end()&&m.find(y)!=m.end()){
                return m[x]<m[y];
            }else if(m.find(x)!=m.end()){
                return true;
            }else if(m.find(y)!=m.end()){
                return false;
            }else{
                return x<y;
            }
        });
        return arr1;
    }
};
