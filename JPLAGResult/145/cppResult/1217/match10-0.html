class Solution {
public:
  vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
    unordered_map<int, int> idx;
    for (int i = 0; i < arr2.size(); ++i) {
      idx[arr2[i]] = i;
    }
    sort(begin(arr1), end(arr1), [&](int x, int y) {
      if (idx.count(x) && idx.count(y)) {
        return idx[x] < idx[y];
      }
      if (idx.count(x)) return true;
      if (idx.count(y)) return false;
      return x < y;
    });
    
    return arr1;
  }
};
