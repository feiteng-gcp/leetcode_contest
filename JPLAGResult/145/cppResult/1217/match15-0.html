class Solution {
public:
    vector<int> relativeSortArray(vector<int>& a, vector<int>& b) {
        map<int, int> A;
        for (int i = 0; i < b.size(); ++i) A[b[i]] = i;
        sort(a.begin(), a.end(),
             [&](int u, int v) {
                 if (A.count(u) && A.count(v)) {
                     return A[u] < A[v];
                 } else if (A.count(u)) {
                     return true;
                 } else if (A.count(v)) {
                     return false;
                 }
                 return u < v;
             });
        return a;
    }
};
