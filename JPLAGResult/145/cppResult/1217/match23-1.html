class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        vector<int>v;
        int a[1005];
        for(int i=0;i<1005;i++)a[i]=0;
        for(int i:arr1)a[i]++;

        for(int i:arr2){
            while(a[i]>0){
                v.push_back(i);
                a[i]--;
            }
        }
        for(int i=0;i<1005;i++){
            while(a[i]>0){
                v.push_back(i);
                a[i]--;
            }
        }
        return v;
    }
};
