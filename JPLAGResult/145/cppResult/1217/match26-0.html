class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        int n1 = arr1.size(), n2 = arr2.size();
        vector<int> ans;
        vector<int> used(n1);
        for (int j = 0; j < n2; j++) {
            for (int i = 0; i < n1; i++) {
                if (arr1[i] == arr2[j]) {
                    ans.push_back(arr1[i]);
                    used[i] = 1;
                }
            }
        }
        vector<int> tmp;
        for (int i = 0; i < n1; i++) {
            if (!used[i]) {
                tmp.push_back(arr1[i]);
            }
        }
        sort(tmp.begin(), tmp.end());
        for (int x: tmp) ans.push_back(x);
        return ans;
    }
};
