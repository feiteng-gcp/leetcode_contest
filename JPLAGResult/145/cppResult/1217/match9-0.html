class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        unordered_map<int, int> m;
        for (int i = 0; i < arr2.size(); ++i) {
            m[arr2[i]] = i;
        }
        
        sort(arr1.begin(), arr1.end(), [&](auto a, auto b){
            auto at = m.find(a);
            auto bt = m.find(b);
            // a < b
            if (bt == m.end() && at == m.end()) {
                return a < b;
            }
            
            if (bt == m.end() && at != m.end()) {
                return true;
            }
            
            if (bt != m.end() && at == m.end()) {
                return false;
            }
            
            return at->second < bt->second;
        });
        
        return arr1;
    }
};
