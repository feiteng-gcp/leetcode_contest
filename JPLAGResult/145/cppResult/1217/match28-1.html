class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        unordered_map<int, int> hm;
        for(auto x : arr2) hm[x] = 0;

        vector<int> ed;
        for(auto x : arr1) {
            if(hm.find(x) != hm.end()) {
                hm[x]++;
            } else {
                ed.push_back(x);
            }
        }
        vector<int> res;

        for(auto x : arr2) {
            for(int i = 0; i < hm[x]; i++) {
                res.push_back(x);
            }
        }
        sort(ed.begin(), ed.end());
        for(auto x : ed) res.push_back(x);
        return res;
    }
};
