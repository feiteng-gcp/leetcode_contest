class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        vector<int> ret;
        set<int> inArr2;
        for (int i = 0;i < arr2.size();i++) {
            inArr2.insert(arr2[i]);
        }
        map<int, int> cal;
        for (int i = 0;i < arr1.size();i++) {
            cal[arr1[i]]++;
        }
        for (int i = 0;i < arr2.size();i++) {
            int num = cal[arr2[i]];
            for (int j = 0;j < num;j++) {
                ret.push_back(arr2[i]);
            }
        }
        for (map<int, int>::iterator it = cal.begin();it != cal.end();it++) {
            if (inArr2.find(it->first) == inArr2.end()) {
                for (int j = 0;j < it->second;j++) {
                    ret.push_back(it->first);
                }
            }
        }
        return ret;
    }
};
