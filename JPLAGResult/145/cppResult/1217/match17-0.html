class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        int N = arr2.size();
        unordered_map<int, int> um;
        for (int i=0; i<N; i++) um[arr2[i]] = i;
        sort(arr1.begin(), arr1.end(), [&](const int& a, const int& b) {
            if (um.count(a) && um.count(b)) return um[a] < um[b];
            else if (um.count(a)) return true;
            else if (um.count(b)) return false;
            else return a < b;
        });
        return arr1;
    }
};
