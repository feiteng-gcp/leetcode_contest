class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
      map<int, int> map_to;
      for (int i = 0; i < arr2.size(); ++i) {
        map_to[arr2[i]] = i;
      }
      sort(arr1.begin(), arr1.end(), [&] (int x, int y) {
        if (map_to.count(x) && map_to.count(y)) {
          return map_to[x] < map_to[y];
        }
        if (!map_to.count(x) && !map_to.count(y)) {
          return x < y;
        }
        return map_to.count(x) > map_to.count(y);
      });
      return arr1;
    }
};
