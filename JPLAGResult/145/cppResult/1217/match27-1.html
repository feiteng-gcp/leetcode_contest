class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        
        int N=arr1.size();
        int M=arr2.size();
        vector<bool> done(N,false);
        
         

        vector<int> ret;
        for(int i=0;i<M;i++) {
            for(int j=0;j<N;j++) {
                if (arr1[j]==arr2[i]) {
                    ret.push_back(arr2[i]);
                    done[j]=true;
                }
            }
        }
        
        vector<int> ret_left;
        for(int i=0;i<N;i++)
            if (!done[i]) {
                ret_left.push_back(arr1[i]);
            }
        
        sort(ret_left.begin(),ret_left.end());
        
        ret.insert(ret.end(), ret_left.begin(),ret_left.end());
        
        return ret;
        
        
        
    }
};
