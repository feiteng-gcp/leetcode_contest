class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        vector<int> res;
        set<int> ar;
        map<int,int> m;
        multiset<int> ar2;
        for(int i:arr2){
            ar.insert(i);
        }

        for(int i:arr1){
            if(ar.find(i)!=ar.end()){
                m[i]++;
            }
            else{
                ar2.insert(i);
            }
        }
        for(int i:arr2){

            for(int j=0;j<m[i];j++){
                res.push_back(i);
            }
        }
        for(int i:ar2){
            res.push_back(i);
        }
        return res;
    }
};
