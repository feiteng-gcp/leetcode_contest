class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        int N = 1010;
        vector<int> cnts(N, 0);
        
        vector<int> inArr2(N, false);
        for (auto num : arr2) {
            inArr2[num] = true;
        }
        
        vector<int> res;
        vector<int> arr3;
        for (auto num : arr1) {
            if (inArr2[num]) {
                ++cnts[num];
            } else {
                arr3.push_back(num);
            }
        }
        sort(arr3.begin(), arr3.end());
        
        for (auto num : arr2) {
            for (int i = 0; i < cnts[num]; ++i) {
                res.push_back(num);
            }
        }
        for (auto num : arr3) {
            res.push_back(num);
        }
        
        return res;
    }
};
