class Solution {
public:
vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
    set<int> arr2_s;
    map<int, int> count;
    vector<int> extra;
    vector<int> res;
    for(int i=0; i<arr2.size(); i++){
        arr2_s.insert(arr2[i]);
    }
    
    for(int i=0; i<arr1.size(); i++){
        if(arr2_s.find(arr1[i])!=arr2_s.end()){
            count[arr1[i]]++;
        }
        else{
            extra.push_back(arr1[i]);
        }
    }
    
    for(int i=0; i<arr2.size(); i++){
        int cnt = count[arr2[i]];
        for(int j=0; j<cnt; j++){
            res.push_back(arr2[i]);
        }
    }
    
    sort(extra.begin(),extra.end());
    for(int i=0; i<extra.size(); i++){
        res.push_back(extra[i]);
    }
    return res;
}
};
