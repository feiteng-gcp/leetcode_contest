class Solution {
    int search(int* searchspace, int s, int e, int key) 
{ 
    int ans = -1; 
  
    // Binary search 
    while (s <= e) { 
        int mid = (s + e) / 2; 
  
        if (searchspace[mid] - key > 0) { 
            ans = mid; 
            e = mid - 1; 
        } 
        else { 
            s = mid + 1; 
        } 
    } 
  
    return ans; 
} 
   
int longestSubarray(vector<int> a, int n) 
{ 
    int SuffixSum[n + 1]; 
    SuffixSum[n] = 0; 
  
    for (int i = n - 1; i >= 0; --i) { 
        SuffixSum[i] = SuffixSum[i + 1] + a[i]; 
    } 
  
    int ans = 0; 
   
    int searchspace[n]; 
   
    int index[n]; 
  
    int j = 0; 
  
    for (int i = 0; i < n; ++i) { 
  
        if (j == 0 or SuffixSum[i] > searchspace[j - 1]) { 
            searchspace[j] = SuffixSum[i]; 
            index[j] = i; 
            j++; 
        } 
  
        int idx = search(searchspace, 0, j - 1, SuffixSum[i + 1]); 
  
        if (idx != -1) 
            ans = max(ans, i - index[idx] + 1); 
    } 
  
    return ans; 
} 
    
    public:
    int longestWPI(vector<int>& hours) {
        for(long int i = 0; i < hours.size(); i++)
                if(hours[i] > 8)
                    hours[i] = 1;
                else
                    hours[i] = -1;
        return longestSubarray(hours,hours.size());
    }
};
