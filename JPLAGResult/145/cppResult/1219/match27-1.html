class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int sum = 0;
        int res = 0;
        map<int, int> dp;
        dp[0] = -1;
        for(int i=0;i<hours.size();i++)
        {
            int cur = (hours[i] > 8 ? 1:-1);
            sum += cur;
            if(sum>0)
            {
                res = max(res, i+1);
            }
            else
            {
                if(dp.find(sum-1)!=dp.end())
                {
                    res = max(res, i-dp[sum-1]);
                }
                else if(dp.find(sum)==dp.end())
                {
                    dp[sum] = i; 
                }
            }
            
        }
        return res;
    }
};
