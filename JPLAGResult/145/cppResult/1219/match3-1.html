class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int best = 0;
        vector<int> sum(hours.size() + 1);
        for (int i = 0; i < hours.size(); ++i) {
            int k = hours[i] > 8 ? 1 : -1;
            sum[i+1] = sum[i] + k;

        }
        int l = hours.size();
        for (int i = l; i > 0; --i) {
            for (int j = 0; j < sum.size() - i; ++j) {
                if (sum[i + j] - sum[j] > 0) {
                    return i;
                }
            }
        }
        
        return 0;
    }
};
