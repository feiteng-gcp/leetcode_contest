class Solution {
public:
    int longestWPI(vector<int>& hours) {
        vector<int> mem;
        int N = hours.size();
        mem.push_back(0);
        for(int i = 0; i < N; i++){
            if (hours[i] >8) mem.push_back(mem[i] + 1);
            else mem.push_back(mem[i]);
        }
        for(int i = N - 1; i >=0; i--){
            for(int j=0; j < N - i; j ++){
                if((mem[j + i + 1] - mem[j]) > ((i + 1) / 2)){
                    return i + 1;
                }
            }
        }
        return 0;
    }
};
