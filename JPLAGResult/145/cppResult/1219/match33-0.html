int Tree[100000];

void Set(int Node, int L, int R, int pos, int x) {
    // printf("SET %d %d %d %d %d\n", Node, L, R, pos, x);
    if (L == R) {
        Tree[Node] = x;
        return;
    }
    int Mid = (L + R) / 2;
    if (pos <= Mid) {
        Set(Node * 2, L, Mid, pos, x);
    } else {
        Set(Node * 2 + 1, Mid + 1, R, pos, x);
    }
    Tree[Node] = min(Tree[Node * 2], Tree[Node * 2 + 1]);
}

int Get(int Node, int L, int R, int l, int r) {
    if (L == l && R == r) {
        return Tree[Node];
    }
    int Mid = (L + R) / 2;
    int ret;
    if (r <= Mid) {
        ret = Get(Node * 2, L, Mid, l, r);
    } else if (l > Mid) {
        ret = Get(Node * 2 + 1, Mid + 1, R, l, r);
    } else {
        ret = min(Get(Node * 2, L, Mid, l, Mid), Get(Node * 2 + 1, Mid + 1, R, Mid + 1, r));
    }
    // fprintf(f, "%d %d %d %d %d = %d\n", Node, L, R, l, r, ret);
    // fflush(f);
    return ret;
}

class Solution {
public:
    int longestWPI(vector<int>& hours) {
        memset(Tree, 1, sizeof(Tree));
        map<int, int> mp;
        int n = hours.size();
        int cnt = 0;
        mp[0] = 0;
        Set(1, -10000 + 10005, 10000 + 10005, 0 + 10005, 0);
        int ans = 0;
        for (int i = 1; i <= n; i++) {
            if (hours[i - 1] > 8) {
                cnt++;
            } else {
                cnt--;
            }
            // printf("***%d\n", cnt);
            if (mp.find(cnt) == mp.end()) {
                mp[cnt] = i;
                Set(1, -10000 + 10005, 10000 + 10005, cnt + 10005, i);
            }
            int ppp = Get(1, -10000 + 10005, 10000 + 10005, -10000 + 10005, cnt - 1 + 10005);
            // printf("%d\n", ppp);
            ans = max(ans, i - ppp);
        }
        return ans;
    }
};
