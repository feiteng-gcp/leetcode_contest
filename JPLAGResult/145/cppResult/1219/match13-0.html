class Solution {
public:
    int longestWPI(vector<int>& hours) {
        if(hours.empty())
            return 0;
        vector<int> sum;
        int cur = 0;
        sum.push_back(cur);
        for(int i = 0; i < hours.size(); i++){
            if(hours[i]>8)
                cur += 1;
            else
                cur -= 1;
            sum.push_back(cur);
        }
        int longestlast = 0;
        int longestcur = 0;
        for(int i = 1; i <= hours.size(); i++){
            // use last
            for(int j = 0; j < i; j++){
                if(sum[i]-sum[j] > 0){
                    longestcur = max(i-j, longestlast);//i-j use last, longestlast not use
                    break;
                }
            }
            longestlast = longestcur;
        }
        return longestcur;
    }
};
