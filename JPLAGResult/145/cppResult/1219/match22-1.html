class Solution {
public:
  int longestWPI(vector<int>& hours) {
    int n = hours.size();
    for (int& x : hours) x = x > 8 ? 1 : -1;
    vector<int> suffix(hours.size() + 1);
    for (int i = n - 1; i >= 0; --i)
      suffix[i] = suffix[i + 1] + hours[i];
    vector<int> a(n);
    vector<int> idx(n);
    int j = 0;

    int ans = 0;
    for (int i = 0; i < n; ++i) {
      if (j == 0 || suffix[i] > a[j - 1]) {
        a[j] = suffix[i];
        idx[j++] = i;
      }      
      int k = search(a, 0, j, suffix[i + 1]);
      if (k == -1) continue;
      ans = max(ans, i - idx[k] + 1);
    }
    return ans;
  }
private:

  int search(const vector<int>& a, int l, int r, int key) {
    int ans = -1;
    while (l < r) {
      int mid = l + (r - l) / 2;
      if (a[mid] - key > 0) {
        ans = mid;
        r = mid;
      } else {
        l = mid + 1;
      }      
    }
    return ans;
  }
};
