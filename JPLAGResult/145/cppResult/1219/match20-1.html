class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int n = hours.size();
        if (n == 0) {
            return 0;
        }
        
        vector<int> helper(n + 1, 0);
        helper[0] = 0;

        for (int i = 0; i < n; i++) {
            if (hours[i] > 8) {
                helper[i + 1] = helper[i] + 1;
            }
            else {
                helper[i + 1] = helper[i] - 1;
            }
        }
        
        //for (int i = 0; i < n; i++) {
         //   cout<<helper[i]<<endl;
        //}
        int res = 0;
        // find the largest interval to have positive sum
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                if (helper[i] - helper[j] > 0) {
                    res = max(res, i - j);
                    break;
                }
            }
        }
        return res;
        
    }
};
