class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int large[hours.size()+1];
        for(int i = 0; i < hours.size(); i++) {
            if(hours[i] > 8) {
                large[i+1] = large[i] + 1;
            }else{
                large[i+1] = large[i];
            }
        }
        int interval = 0;
        for(int i = 1; i < hours.size()+1; i++) {
            for(int j = i+interval; j < hours.size()+1; j++) {
                int total = j - i + 1;
                int tired = large[j] - large[i-1];
                int untired = total - tired;
                if(tired > untired) {
                    interval = max(interval, j - i + 1);
                }
            }
            
        }
        return interval;
        
    }
};
