class Solution {
public:
    int longestWPI(vector<int>& hours) {
        vector<int> v;
        v.push_back(0);
        for (int i = 0; i < hours.size(); ++i) {
            v.push_back(hours[i] > 8 ? v[i] + 1 : v[i] - 1);
            
        }
       
        

        int result = 0;
        for (int i = 0; i < hours.size(); i++)
            for (int j = hours.size() - 1; result < j - i + 1; j--) {
                if (v[j+1] - v[i] > 0)  { result = j - i + 1; }
            }
        return result;
    }
};
