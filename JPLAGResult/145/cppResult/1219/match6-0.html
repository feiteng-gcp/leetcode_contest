class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int n = hours.size();
        vector<int> s(n + 1, 0);
        for (int i = 1; i <= n; i++) {
            s[i] = s[i-1] + (hours[i-1] > 8 ? 1 : -1);
        }
        vector<pair<int, int>> decrease_prefix;
        decrease_prefix.push_back(make_pair(0, 0));
        int max_len = 0;
        for (int i = 1; i <= n; i++) {
            int value = s[i];
            auto it = upper_bound(decrease_prefix.begin(), decrease_prefix.end(), make_pair(value, 0), greater<pair<int, int>>());
            if (it != decrease_prefix.end()) {
                int length = i - it->second;
                max_len = max(max_len, length);
            }
            if (value < decrease_prefix.back().first) {
                //printf("i = %d, push prefix (%d, %d)\n", i, value, i);
                decrease_prefix.push_back(make_pair(value, i));
            }
        }
        return max_len;
    }
};
