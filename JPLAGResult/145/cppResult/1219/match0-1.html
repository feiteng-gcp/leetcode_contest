class Solution {
public:
    struct Point {
        int value, pos;
        Point(int _value = 0, int _pos = 0) {
            value = _value;
            pos = _pos;
        }
        bool operator<(const Point &r) const {
            return value > r.value;
        }
    };
    int longestWPI(vector<int>& hours) {
        vector<Point> dp;
        dp.push_back(Point(0, -1));
        int x = 0;
        int y = 0;
        int ret = 0;
        for (int i = 0;i < hours.size();i++) {
            if (hours[i] > 8) {
                x++;
            } else {
                y++;
            }
            int temp = x - y;
            int index = upper_bound(dp.begin(), dp.end(), Point(temp, i)) - dp.begin();
            if (index != dp.size()) {
                if (ret < i - dp[index].pos) {
                    ret = i - dp[index].pos;
                }
            } else {
                dp.push_back(Point(temp, i));   
            }
        }
        return ret;
    }
};
