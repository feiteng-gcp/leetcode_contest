class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int n = hours.size();
        vector<int> sum(n + 1, 0);
        for (int i = 0; i < n; i++) {
            int is_tired = hours[i] > 8 ? 1 : -1;
            sum[i + 1] = sum[i] + is_tired;
        }
        
        vector<int> haha(n + 1);
        haha[n] = sum[n];
        for (int i = n - 1; i >= 0; i--) {
            haha[i] = max(haha[i + 1], sum[i]);
        }
        
        int left = 0;
        int ans = 0;
        for (int right = 0; right <= n; right++) {
            if (haha[right] > sum[left])
                ans = max(ans, right - left);
            else
                left++;
        }
        
        return ans;
    }
};
