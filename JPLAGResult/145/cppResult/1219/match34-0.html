class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int n=hours.size();
        vector<int> sums(n+1,0);
        for(int i=0;i<n;i++){
            sums[i+1]=sums[i]+(hours[i]>8?1:-1);
        }
        
        int maxs=0;
        map<int,int> mm;
        for(int i=n;i>=0;i--){
            auto it=mm.lower_bound(sums[i]+1);
            if(it!=mm.end()){
                maxs=max(it->second-i,maxs);
            }
            
            it=mm.lower_bound(sums[i]);
            if(it==mm.end()){
                mm[sums[i]]=i;
            }
        }
        return maxs;
    }
};
