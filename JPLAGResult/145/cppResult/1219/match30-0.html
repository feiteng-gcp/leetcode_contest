class Solution {
public:
    int longestWPI(vector<int>& hours) {
        if (hours.size() == 0) return 0;
        
        vector<int> numTiredDays(hours.size() + 1); // the sum of tired days for day < index
        numTiredDays[0] = 0;
        bool anyTired = false;
        for (int i = 1; i <= hours.size(); ++i) {
            numTiredDays[i] = numTiredDays[i - 1] + (hours[i - 1] > 8 ? 1 : 0);
            if (hours[i - 1]  > 8) anyTired = true;
        }
        
        if (!anyTired) return 0;
        
        vector<int> numNonTiredDays(hours.size() + 1); // the sum of non tired days for day < index
        numTiredDays[0] = 0;
        for (int i = 1; i <= hours.size(); ++i) {
            numNonTiredDays[i] = numNonTiredDays[i - 1] + (hours[i - 1] <= 8 ? 1 : 0);
        }
        
        // // stupid solution might work?
        // int start = 0;
        // int length = 1;
        // while (start + length <= hours.size()) {
        //     int numTired = numTiredDays[start + length] - numTiredDays[start];
        //     int numNotTired = numNonTiredDays[start + length] - numNonTiredDays[start];
        //     if (numTired > numNotTired) ++length;
        //     else ++start;
        // }
        
        int length = 1;
        for (int start = 0; start < hours.size(); ++start) {
            for (int end = start + length; end <= hours.size(); ++end) {
                int numTired = numTiredDays[end] - numTiredDays[start];
                int numNotTired = numNonTiredDays[end] - numNonTiredDays[start];
                // cout << start << " " << end << " " << numTired << " " << numNotTired << endl;
                if (numTired > numNotTired && end - start > length) length = end - start; 
            }
        }
        
        return length;
    }
    
    int isTired (int hours) {
        return hours > 8;
    }
};
