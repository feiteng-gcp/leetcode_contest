class Solution {
public:
    int longestWPI(vector<int>& hours) {
        
        int ans = 0;
        int n = hours.size();
        vector<int> prefixS(n + 1);
        int tmp = 0;
        for(int i = 1; i <= n; ++i){
            tmp += hours[i - 1] > 8 ? 1 : -1;
            prefixS[i] = tmp;
        }
        if(prefixS[n] > 0) return n;
        
        for(int i = 0; i < n; ++i){
            for(int j = i + 1; j <= n; ++j){
                if(prefixS[j] - prefixS[i] > 0) ans = max(ans, j - i);
            }
        }
        return ans;
    }
};
