class Solution {
    int ans = 0;
public:
    int longestWPI(vector<int>& hours) {
        int n = hours.size();
        vector<pair<int, int>> dp(n + 1, make_pair(0, 0));
        
        dp[0].second = -1;
        for (int i = 0; i < n; ++i) {
            if (hours[i] > 8) dp[i + 1].first = dp[i].first + 1;
            else dp[i + 1].first = dp[i].first - 1;
            dp[i + 1].second = i;
        }
        
        mergesort(dp, 0, n);
        
        return ans;
    }
    
    void mergesort(vector<pair<int, int>> &dp, int x, int y) {
        if (x == y) return;
        int mid = (x + y) / 2;
        
        mergesort(dp, x, mid);
        mergesort(dp, mid + 1, y);
        
        for (int i = mid + 1; i <= y; ++i) {
            for (int j = x; j <= mid; ++j) {
                if (dp[i].first > dp[j].first) ans = max(ans, dp[i].second - dp[j].second);
                else break;
            }
        }
        
        int a = x, b = mid + 1;
        vector<pair<int, int>> tmp;
        while (a <= mid && b <= y) {
            if (dp[a].first < dp[b].first) {
                tmp.push_back(dp[a]);
                a++;
            }
            else {
                tmp.push_back(dp[b]);
                b++;
            }
        }
        
        while (a <= mid) {
            tmp.push_back(dp[a]);
            a++;
        }
        
        while (b <= y) {
            tmp.push_back(dp[b]);
            b++;
        }
        
        for (int i = x; i <= y; ++i) dp[i] = tmp[i - x];
    }
};
