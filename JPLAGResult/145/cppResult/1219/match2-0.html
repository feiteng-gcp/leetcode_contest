class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int len = hours.size();
        vector<int> f(len + 1, 0);
        for (int i = 0; i < len; i ++) {
            f[i] = (i > 0?f[i - 1]: 0) + (hours[i] > 8? 1 : -1);
        }
        int ans = 0;
        for (int i = 0; i < len; i ++) {
            if (f[i] > 0) {
                ans = max(ans, i + 1);
                continue;
            }
            for (int j = 0; j < i; j ++) {
                if (f[i] - f[j] > 0) {
                    ans = max(ans, i - j);
                    break;
                }
            }
        }
        return ans;
    }
};
