class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int n = static_cast<int>(hours.size());
        vector<int> sums(n + 1);
        for (int i = 0; i < n; ++i) {
            hours[i] = hours[i] > 8;
            sums[i + 1] = sums[i] + hours[i] * 2;
        }
        int maxInterval = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = i + maxInterval; j <= n; ++j) {
                if (sums[j] - sums[i] > j - i) {
                    maxInterval = max(maxInterval, j - i);
                }
            }
        }
        return maxInterval;
    }
};
