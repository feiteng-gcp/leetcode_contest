class Solution {
public:
    int longestWPI(vector<int>& hours) {
        unordered_map<int, int> memo; // sum, index

        int s = 0;
        memo[0] = -1;
        int ret = 0;
        for (int i = 0; i < hours.size(); ++i) {
            if (hours[i] > 8) {
                ++s;
            } else {
                --s;
            }
            if (memo.find(s) == memo.end()) {
                memo[s] = i;
            }
            if (s - 1 <= 0 && memo.find(s - 1) != memo.end()) {
                ret = max(ret, i - memo[s - 1]);
            }
            if (s - 1 > 0) {
                ret = max(ret, i - memo[0]);
            }
        }
        return ret;
    }
};
