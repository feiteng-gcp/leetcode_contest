class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int N=hours.size();
        vector<int> cumul(N+1,0);
        for(int i=0;i<hours.size();i++) {
            cumul[i+1]=cumul[i]+(hours[i]>8);
        }
        
        // smallest j such that v[i] > v[j]
        
        int ans=0;
        for(int i=1;i<=N;i++) {
            for(int j=1;j<=i;j++) {
                // from j to i
                // upto i
                // cumul[i],i-cumul[i]
                // cumul[j-1],j-1-cumul[j-1]
                if (((cumul[i]-cumul[j-1])<<1) > i-j+1) {
                    ans=max(ans,i-j+1);
                    break;
                }
            }
            
        }
        return ans;
        
    }
};
