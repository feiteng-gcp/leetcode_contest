class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int n = hours.size();
        vector<int> acc(n+1, 0);
        for(int i=0;i<n;++i) {
            acc[i+1]=acc[i]+((hours[i]>8)?1:-1);
        }
        // for(auto it:acc) {
        //     cout<<it<<' ';
        // }
        
        int ans=0;
        for(int i=0;i<n;++i) {

            for(int j=n;j>i;--j) {
                if(j-i>ans && acc[j]-acc[i]>0) {
                    ans = j-i;
                    break;
                }
            }
        }
        return ans;
    }
};
