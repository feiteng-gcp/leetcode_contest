class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int n = hours.size();
        if(n == 1)
        {
            if(hours[0] > 8)
                return 1;
            return 0;
        }
        vector<int> sums(n, 0);
        int sum = 0;
        for(int i = 0; i < n; i++)
        {
            if(hours[i] > 8)
                sum += 1;
            else 
                sum += -1;
            sums[i] = sum;
        }
        
        vector<int> dp(n, 0);
        int ans = 0;
        for(int i = 0; i < n; i++)
        {
            if(sums[i] > 0)
            {
                dp[i] = i + 1;
            }
            else 
            {
                for(int j = 0; j < i; j++)
                {
                    if(sums[i] - sums[j] > 0)
                    {
                        dp[i] = i - j;
                        break;
                    }
                }
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
