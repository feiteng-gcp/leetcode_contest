class Solution {
public:
    int longestWPI(vector<int>& v) {
        int res = 0;
        for (int i = 0; i < v.size(); i++) {
            int cnt = 0;
            if (i > 0 && v[i] > 8 && v[i-1] > 8) continue;
            if ( (v.size() - i) <= res) break;
            for (int j = i; j < v.size(); j++) {
                if (v[j] > 8) cnt++;
                else cnt--;
                
                if (cnt > 0) {
                    res = max(res, j - i + 1);
                }
            }
        }
        return res;
        
    }
};
