class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int n = hours.size();
        vector<int> s(n + 1, 0);
        for (int i = 1; i <= n; i++) {
            s[i] = s[i - 1] + (hours[i - 1] > 8);
        }
        vector<pair<int, int> > tmp;
        for (int i = 1; i <= n; i++) {
            tmp.push_back(make_pair(2 * s[i] - i, i));
        }
        sort(tmp.begin(), tmp.end());
        vector<int> max_j(n);
        max_j[n - 1] = tmp[n - 1].second;

        for (int i = n - 2; i >= 0; i--) {
            max_j[i] = max(tmp[i].second, max_j[i + 1]);
        }
        /*
        for (int i = 0; i < n; i++) {
            cout << tmp[i].first << " ";
        }
        cout << endl;
        for (int i = 0; i < n; i++) {
            cout << tmp[i].second << " ";
        }
        cout << endl;
        for (int i = 0; i < n; i++) {
            cout << max_j[i] << " ";
        }
        cout << endl;
        */
        int ans = 0;
        for (int i = 1; i <= n; i++) {
            int key = 2 * s[i - 1] - (i - 1);
            int l = 0, r = n - 1;
            while (l < r) {
                int mid = (l + r) / 2;
                if (tmp[mid].first > key) {
                    r = mid;
                } else {
                    l = mid + 1;
                }
            }
            if (tmp[l].first > key) {
                int j = max_j[l];
                // cout << "i j " << i << " " << j << endl;
                if (j >= i) {
                    ans = max(ans, j - i + 1);
                }
            }
        }
        return ans;
        
    }
};
