const int MAXN = 20010;

int v[MAXN];

void add(int x, int p)
{
    for (; x < MAXN; x += x&-x)
        v[x] = min(v[x], p);
}

int get(int x)
{
    int p = MAXN;
    for (; x; x -= x&-x)
        p = min(p, v[x]);
    return p;
}

class Solution {
public:
    int longestWPI(vector<int>& a) {
        int n = a.size();
        vector<int> s(n+1);
        for (int i = 0; i < n; ++ i)
        {
            if (a[i] > 8) s[i+1] = 1; else s[i+1] = -1;
        }
        for (int i = 1; i <= n; ++ i)
        {
            s[i] += s[i-1];
        }
        for (int i = 0; i < MAXN; ++ i)
        {
            v[i] = MAXN;
        }
        add(0+n+1, 0);
        int ret = 0;
        for (int i = 1; i <= n; ++ i)
        {
            int j = get(s[i]+n);
            ret = max(ret, i-j);
            add(s[i]+n+1, i);
        }
        return ret;
    }
};
