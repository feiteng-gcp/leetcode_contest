class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int n = hours.size();
        vector<int> cnts(n + 1, 0);
        for (int i = 0; i < n; ++i) {
            cnts[i + 1] = cnts[i] + (hours[i] > 8 ? 1 : -1);
        }
        
        int N = n + 100;
        vector<int> data(N, N);
        auto put = [&](int x, int v) {
            for (; x < N; x |= x + 1) {
                data[x] = min(data[x], v);
            }
        };
        auto get = [&](int x) {
            int res = N;
            for (; x >= 0; x = (x & (x + 1)) - 1) {
                res = min(res, data[x]);
            }
            return res;
        };
        
        vector<int> sorted(cnts);
        sort(sorted.begin(), sorted.end());
        sorted.resize(unique(sorted.begin(), sorted.end()) - sorted.begin());
        auto index = [&](int x) -> int {
            auto it = lower_bound(sorted.begin(), sorted.end(), x);
            return it - sorted.begin();
        };
        
        int res = 0;
        put(index(0), -1);
        for (int i = 0; i < n; ++i) {
            int x = cnts[i + 1];
            int idx =index(x);
            res = max(res, i - get(idx - 1));
            put(idx, i);
        }
        
        return res;
        
    }
    
    
};
