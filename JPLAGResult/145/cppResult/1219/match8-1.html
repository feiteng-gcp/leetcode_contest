class Solution {
public:
    int longestWPI(vector<int>& hours) {
        int n=hours.size();
        vector<int> sum(n+1,0);
        for(int i=0;i<n;i++){
            sum[i+1]=sum[i] + (hours[i]>8?1:0);
        }
        // vector<pair<double, int>> v(n);
        // for(int i=0;i<n;i++){
            // v[i].first=sum[i]-double(i)/2;
            // v[i].second=i;
        // }
        // sort(v.begin(),v.end());
        set<int> jj;
        for(int i=0;i<=n;i++)jj.insert(i);
        map<double,vector<int>> m;
        for(int i=0;i<=n;i++){
            m[sum[i]-double(i)/2].push_back(i);
        }
        int res=0;
        for(auto & item:m){
            for(int i:item.second){
                jj.erase(i);
            }
            if(jj.empty()) break;
            res = max(res,*jj.rbegin() - item.second[0]);
        }
        return res;
    }
};
