class Solution {
public:
    int pre[10001];
    int longestWPI(vector<int>& h) {
        int n = h.size();
        for (int i = 0; i < n; ++i)
            pre[i + 1] = pre[i] + (h[i] > 8 ? 1  : -1);
        int ans = 0;
        for (int i = 1; i <= n; ++i)
            for (int j = n; j >= i; --j)
            {
                if (pre[j] - pre[i - 1] > 0) {
                    ans = max(ans, j - i + 1);
                    break;
                }
            }
        return ans;
    }
};
