class Solution {
public:
    int longestWPI(vector<int>& hours) {
        for(int &h: hours)
            h = ((h >= 9) ? 1 : 0);
        
        int n = hours.size();
        int i;
 
        int sum = 0; 

        unordered_map<int, int> prevSum; 

        int maxlen = 0; 

        int currlen; 

        for (i = 0; i < n; i++) { 
            if (hours[i] == 1) 
                sum++; 
            else
                sum--; 

            if (sum > 0) { 
                maxlen = i + 1; 
            } 
            else if (sum <= 0) { 
                if (prevSum.find(sum - 1) != prevSum.end()) { 
                    currlen = i - prevSum[sum - 1]; 
                    maxlen = max(maxlen, currlen); 
                } 
            } 
            if (prevSum.find(sum) == prevSum.end()) 
                prevSum[sum] = i; 
        } 

        return maxlen;        
    }
};
