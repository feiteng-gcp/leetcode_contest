class Solution {
    public int longestWPI(int[] hours) {
        int[] diff = new int[hours.length + 1];
        for(int i = 1; i <= hours.length; i++) {
            if(hours[i - 1] > 8) {
                diff[i] = diff[i - 1] + 1;
            } else {
                diff[i] = diff[i - 1] - 1;
            }
        
}
        // for(int i = 0; i < diff.length; i++) {
        //     System.out.println(diff[i]);
        // }
        int longest = 0;
        for(int i = 0; i < diff.length; i++) {
            for(int j = 0; j < i; j++) {
                
if(diff[i] - diff[j] > 0) {
                    longest = Math.max(i - j, longest);
                    break;
                }
            }
        }
        return longest;
    }
}
