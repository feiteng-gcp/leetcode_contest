class Solution {
    public int longestWPI(int[] hours) {
        int[] tired = new int[hours.length + 1];
        for (int i = 0; i < hours.length; i++) {
            if(hours[i]>8)
                tired[i+1]=tired[i]+1;
            else
                tired[i+1]=tired[i]-1;
        
}
        int res = 0;
        for (int i = 0; i < tired.length; i++) {
            for (int j = 0; j < i; j++) {
                if (tired[i] - tired[j] > 0)
                    res = Math.max(res, i - j)
;
            }
        }
        return res;
    }
}
