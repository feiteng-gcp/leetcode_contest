class Solution {
    public int longestWPI(int[] hours) {
        
int max = 0;
        
        for (int i = 0; i < hours.length; i++) {
            int tiring = 0;
            for (int j = i; j < hours.length; j++) {
                if (hours[j] > 8)
                    tiring++
;
                if (tiring > j - i + 1 - tiring) {
                    max = Math.max(max, j - i + 1);
                }
            }
        }
        return max;
    }
}
