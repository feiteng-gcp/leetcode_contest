class Solution {
    public int longestWPI(int[] hours) {
        int[] currArr = new int[hours.length];
        int curr = 0;
        for(int i =0; i <hours.length;i++){
            curr += hours[i] > 8 ? 1 : 0;
            
currArr[i] = curr;
        }
        int res = 0;
        for(int i =0; i< hours.length;i++){
            for(int j = i; j < hours.length; j++){
                curr = currArr[j] - currArr[i] + (hours[i] > 8 ? 1 : 0);
                
if(curr > (j - i + 1)/2){
                    res = Math.max(res,j - i + 1);
                }
            }
        }
        return res;
    }
}
