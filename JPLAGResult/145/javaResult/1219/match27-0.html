class Solution {
    public int longestWPI(int[] hours) {
        int [] cum = new int[hours.length];
        
        for (int i=0; i<hours.length; i++) {
            if (i>0) cum[i] = cum[i-1];
            cum[i] += hours[i] > 8? 1 : -1;
        }
        
        int ans = 0;
        for (int i=0; i<hours.length; i++) {
            for (int j=hours.length-1; j>=0; j--) {
                int sum = cum[j] - cum[i] + (hours[i] > 8? 1 : -1);
                
                if (sum > 0) {
                    ans = Math.max(ans, j-i+1);
                    break;
                }
            }
        }
        
        return ans;
    }
}
