//O(n^2)
class Solution {
    public int longestWPI(int[] hours) {
        int[] sum = new int[hours.length + 1];
        for(int i = 0; i < hours.length; i++){
            sum[i + 1] = sum[i] + (hours[i] > 8 ? 1 : -1);
        }
        
        int res = 0;
        for(int i = 0; i < hours.length; i++){
            for(int j = i; j < hours.length; j++){
                int len = sum[j + 1] - sum[i];
                if(len > 0) res = Math.max(res, j - i + 1);
            }
        }
        return res;
    }
}
