class Solution {
    public int longestWPI(int[] hours) {
        int res = 0;
        int N = hours.length;
        for (int i = 0; i < N; i++) {
            int cntTiring = 0;
            for (int j = i; j < N; j++) {
                if (hours[j] > 8) {
                    cntTiring++;
                }
                if (cntTiring > j - i + 1 - cntTiring) {
                    res = Math.max(res, j - i + 1);
                }
            }
        }
        return res;
    }
}
