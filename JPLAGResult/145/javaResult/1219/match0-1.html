class Solution {
        public int longestWPI(int[] hours) {
        TreeMap<Integer,Integer> treeMap = new TreeMap<>();
        int countH = 0;
        int countL = 0;
        for(int i=0;i<hours.length;++i){
            if(hours[i]<=8){
                ++countL;
            }else{
                ++countH;
            }
            int diff = countL-countH;
            if(diff<0||treeMap.containsKey(diff)||treeMap.higherKey(diff)!=null)
                continue;
            treeMap.put(diff,i);
        }
        countH = 0;
        countL = 0;
        int ans = 0;
      //  showTreeMap(treeMap);
        for(int i=0;i<hours.length;++i){
            if(hours[i]>8)
                ++countH;
            else
                ++countL;
            if(countL<countH){
                ans = i+1;
            //    System.out.println(i+"  "+ans);
                continue;
            }
            int diff = countL-countH;
            Integer pos = treeMap.higherKey(diff);
          //  System.out.println("diff = "+diff+"  pos = "+pos);
            if(pos==null||pos>i)
                continue;
            ans = Math.max(ans,i-treeMap.get(pos));
          //  System.out.println(i+"  "+ans+" "+pos+"  "+treeMap.get(pos));
        }
        return ans;
    }
}
