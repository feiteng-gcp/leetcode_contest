class Solution {
    public int longestWPI(int[] hours) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        int total=0, res = 0;
        map.put(0, -1);
        for(int i=0; i<hours.length; i++){
            int h = hours[i];
            if(h>8) {
                total++;
            }
            else {
                total--;
            }
            Map<Integer,Integer> sm = map.subMap(Integer.MIN_VALUE, total);
            if(!sm.isEmpty()){
                for(int v : sm.values()){
                    res = Math.max(res, i-v);
                }
            }
            map.put(total, map.getOrDefault(total, i));
        }
        return res;
    }
}
