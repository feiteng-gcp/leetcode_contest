class Solution {
    public int longestWPI(int[] hours) {
        int n = hours.length;
        int[] tiring = new int[n+1];
        
        for(int i=1;i<=n;i++){
            if(hours[i-1] > 8){
                tiring[i]++;
            }
            tiring[i] += tiring[i-1];
        
}
        
        int maxLen = 0;
        for(int i=1;i<=n;i++){
            for(int j=0;j<i;j++){
                int totalDays = i-j;
                if(totalDays < maxLen)    break;
                
                int non_tiring = totalDays- (tiring[i]-tiring[j]);
                
if(tiring[i]-tiring[j] > non_tiring){
                    maxLen = Math.max(totalDays, maxLen);
                }
                
            }
        }
        
        return maxLen;
    }
}
