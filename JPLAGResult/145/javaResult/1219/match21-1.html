class Solution {
    public int longestWPI(int[] hours) {
        int n = hours.length;
        int[] stp = new int[n+1];
        for (int i = 0; i < n; i++) {
            stp[i+1] = stp[i]+(hours[i]>8?1:-1);
        }
        int ans = 0;
        
for (int i = 0; i <= n; ++i) {
            for (int j = i+ans+1; j <= n; ++j) {
                if (stp[j]>stp[i]) {
                    ans = j-i;
                
}
            }
        }
        return ans;
    }
}
