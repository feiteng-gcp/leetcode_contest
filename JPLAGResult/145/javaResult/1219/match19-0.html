class Solution {
    public int longestWPI(int[] hours) {
        int[] dp = new int[hours.length+1];
        int sum = 0;
        for(int i=0;i<hours.length;i++)
        {
            if(hours[i]>8)
                ++sum;
            else
                --sum;
            dp[i+1]=sum;
        }
        int max = 0;
        for(int i=0;i<dp.length;i++)
        {
            for(int j=i+1;j<dp.length;j++)
            {
                if(dp[j]-dp[i]>0)
                {
                    max=Math.max(max,j-i);
                }
            }
        }
        return max;
    }
}
