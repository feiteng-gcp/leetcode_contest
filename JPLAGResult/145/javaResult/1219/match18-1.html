class Solution {
    public int longestWPI(int[] hours) {
        int n = hours.length;
        int[] ts = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            ts[i] = ts[i - 1] + (hours[i - 1] > 8 ? 1 : 0);
        
}
        int max = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j <= n; j++) {
                int t = ts[j] - ts[i];
                
if (t > (j - i) / 2) {
                    max = Math.max(max, j - i);
                }
            }
        }
        return max;
    }
}
