class Solution {
    public int longestWPI(int[] hours) {
        int out = 0;

        for(int i = 0; i < hours.length; i++) {
            
            int runningSum = 0;
            
            for(int j = i; j < hours.length; j++) {
                runningSum += hours[j] > 8 ? 1 : -1;
                
                
if(runningSum > 0) {
                    out = Math.max(out, j - i + 1);
                }
            }
        }
        
        return out;
    }
}
