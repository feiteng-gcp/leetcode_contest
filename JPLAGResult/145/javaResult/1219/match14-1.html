class Solution {
    public int longestWPI(int[] hours) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        int s=0, res = 0;
        map.put(0, -1);
        for(int i=0; i<hours.length; i++){
            int h = hours[i];
            if(h>8) s++;
            else s--
;
            Map<Integer,Integer> sub = map.subMap(Integer.MIN_VALUE, s);
            if(!sub.isEmpty()){
                for(int v : sub.values()){
                    res = Math.max(res, i-v);
                }
            }
            map.put(s, map.getOrDefault(s, i));
        }
        return res;
    }
}
