class Solution {
    public int longestWPI(int[] hours) {
        int len = hours.length;
        int[] sum = new int[len + 1];
        for (int i = 0; i < len; i++) {
            sum[i + 1] = sum[i] + (hours[i] > 8 ? 1 : 0);
        }
        for (int k = len; k > 0; k--) {
            for (int i = len; i - k >= 0; i--) {
                if (sum[i] - sum[i - k] > k / 2) return k;
            }
        }
        return 0;
    }
}
