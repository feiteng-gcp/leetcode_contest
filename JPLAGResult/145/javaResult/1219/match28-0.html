class Solution {
    public int longestWPI(int[] hours) {
        int N = hours.length;
        int[] tirings = new int[N + 1];
        int[] non_tirings = new int[N + 1];
        int tiring = 0;
        int non_tiring = 0;
        for (int i = 0; i < N; i++) {
            if (hours[i] > 8) {
                tiring++;
            } else {
               non_tiring++; 
            }
            tirings[i + 1] = tiring;
            non_tirings[i + 1] = non_tiring;
        }
        int max = 0;
        for (int i = 0; i < N; i++) {
            for (int j = i; j < N; j++) {
                int delta_tiring = tirings[j + 1] - tirings[i];
                int delta_non = non_tirings[j + 1] - non_tirings[i];
                if (delta_tiring > delta_non) {
                    max = Math.max(max, j - i + 1);
                }
            }
        }
        return max;
        
    }
}
