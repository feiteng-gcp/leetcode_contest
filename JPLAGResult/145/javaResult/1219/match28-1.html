class Solution {
    public int longestWPI(int[] hours) {
        //end at.
        
int[][] summary = new int[hours.length+1][2];
        int tiringDaySoFar = 0;
        int nonTiringDaySoFar = 0;
        for(int i=0;i<hours.length;i++){
            if(hours[i]>8){
                tiringDaySoFar++;
            }else{
                nonTiringDaySoFar++;
            }
            summary[i+1][0] = tiringDaySoFar;
            summary[i+1][1] = nonTiringDaySoFar;
        }
        
       
for(int len = hours.length; len > 0; len--){
           for(int s = 0; s+len<=hours.length;s++){
               int deltaTiring = summary[s+len][0] -summary[s][0];
               int deltaNonTiring = summary[s+len][1] -summary[s][1];
               if(deltaTiring>deltaNonTiring){
                   return len;
               
}
           }
       }
        
        return 0;
        
    }
}
