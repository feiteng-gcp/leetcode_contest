9, 9, 6, 0,9,9,9,9

class Solution {
    public int longestWPI(int[] hours) {
        int[] sum = new int[hours.length + 1];
        
        for (int i = 1; i <= hours.length; i++) {
            int s = 0;
            if (hours[i - 1] > 8) s = 1;
            sum[i] = s + sum[i - 1];
        }
        int res = -1;
        for (int l = hours.length; l >= 1; l--) {
            
            for (int i = 0; i + l <= hours.length; i++) {
                
                if (sum[i + l] - sum[i] > (l / 2)) return l;
            }
        }
        
        return 0;
    }
}
