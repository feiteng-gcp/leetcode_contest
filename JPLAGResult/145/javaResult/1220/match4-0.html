class Solution {
    int[][] dp;
    String[] skills;
    List<List<String>> people;
    int[] pset;
    int sl;
    
    public int[] smallestSufficientTeam(String[] skills, List<List<String>> people) {
        sl = skills.length;
        int n = people.size();
        dp = new int[n][1 << sl];
        for (int i = 0; i < n; i++) for (int j = 0; j < 1 << sl; j++) dp[i][j] = -1;
        this.skills = skills;
        this.people = people;
        pset = new int[n];
        
        for (int i = 0; i < people.size(); i++) {
            int set = 0;
            for (int j = 0; j < skills.length; j++) {
                for (int k = 0; k < people.get(i).size(); k++) {
                    if (skills[j].equals(people.get(i).get(k))) {
                       set |= 1 << j; 
                    }
                }
            }
            pset[i] = set;
        }
        int num = solve(0, 0);
        int[] res = new int[num];
        int idx = 0;
        int set = 0;
        for (int i = 0; i < n; i++) if (idx < num) {
            if ((set | pset[i]) == (1 << sl) - 1) {
                res[idx++] = i;
                set |= pset[i];
                continue;
            }
            if (dp[i][set] == dp[i+1][set | pset[i]] + 1) {
                res[idx++] = i;
                set |= pset[i];
            }
        }
        return res;
    }
    
    int solve(int idx, int set) {
        if (set+1 == (1 << sl)) return 0;
        if (idx == people.size()) return Integer.MAX_VALUE;
        if (dp[idx][set] != -1) return dp[idx][set];
        
        int res = solve(idx+1, set);
        int tmp = solve(idx+1, set | pset[idx]);
        if (tmp != Integer.MAX_VALUE) {
            res = Math.min(res, tmp + 1);
        }
        dp[idx][set] = res;
        return res;
    }
    
}
