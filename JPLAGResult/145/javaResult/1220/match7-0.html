class Solution {
    int[] res;
    Map<String, List<Integer>> map;
    List<List<String>> people;
    public int[] smallestSufficientTeam(String[] req_skills, List<List<String>> people) {
        this.people = people;
        map = new HashMap<>();
        for (String skill : req_skills) {
            map.put(skill, new ArrayList<Integer>());
        }
        for (int i = 0; i < people.size(); i++) {
            List<String> personSkill = people.get(i);
            for (String skill : personSkill) {
                if (map.containsKey(skill)) {
                    map.get(skill).add(i);
                }
            } 
        }
        Set<String> toDo = new HashSet(Arrays.asList(req_skills));
        dfs(toDo, new ArrayList<Integer>());
        return res;
    }
    
    private void dfs(Set<String> toDo, List<Integer> chosen) {
        //System.out.println("toDo=" + toDo);
        //System.out.println("chose=" + chosen);
        if (toDo.isEmpty()) {
            if (res == null || chosen.size() < res.length) {
                res = new int[chosen.size()];
                int idx = 0;
                for (int i : chosen) {
                    res[idx++] = i;
                }
            }
            return;
        }
        //String nextJob = toDo.iterator().next();
        String nextJob = chooseNextJob(toDo);
        for (int nextPerson : map.get(nextJob)) {
            Set<String> toDoNext = new HashSet<>(toDo);
            for (String job : people.get(nextPerson)) {
                toDoNext.remove(job);
            }
            chosen.add(nextPerson);
            dfs(toDoNext, chosen);
            chosen.remove(chosen.size() - 1);
        }
    }
    
    private String chooseNextJob(Set<String> toDo) {
        int min = Integer.MAX_VALUE;
        String minJob = "";
        for (String job : toDo) {
            if (map.get(job).size() < min) {
                min = map.get(job).size();
                minJob = job;
            }
        }
        return minJob;
    }
}
