class Solution {
    public int[] smallestSufficientTeam(String[] req_skills, List<List<String>> people) {
        final int M = req_skills.length;
        HashMap<String, Integer> skillID = new HashMap<>();
        for (int i = 0; i < M; ++i) {
            skillID.put(req_skills[i], i);
        }
        final int N = people.size();
        int[] masks = new int[N];
        int[] best = new int[1 << M];
        int[] prevBest = new int[1 << M];
        int[] currVal = new int[1 << M];
        Arrays.fill(best, N + 1);
        best[0] = 0;
        for (int i = 0; i < people.size(); ++i) {
            for (String skill : people.get(i)) {
                masks[i] |= (1 << skillID.get(skill));
            }
            // System.out.format("====== i = %d ======= \n", i);
            // System.out.format("i = %d, skillMask = %s\n", i, Integer.toString(masks[i], 2));
            for (int prevMask = 0; prevMask < best.length; ++prevMask) {
                // if (best[prevMask] > N) {
                //     continue;
                // }
                int comboMask = prevMask | masks[i];
                if (best[prevMask] + 1 < best[comboMask]) {
                    best[comboMask] = best[prevMask] + 1;
                    prevBest[comboMask] = prevMask;
                    currVal[comboMask] = i;
                }
            }
            
            // for (int mask = 0; mask < best.length; ++mask) {
            //     System.out.format("mask = %s, best = %d, prevBest = %s, currVal = %d\n", Integer.toString(mask, 2), best[mask], Integer.toString(prevBest[mask], 2), currVal[mask]);
            // }
        }
        // int bestMask = -1;
        // int bestVal = N + 1;
        // for (int mask = 0; mask < best.length; ++mask) {
        //     if (best[mask] < bestVal) {
        //         bestMask = mask;
        //         bestVal = best[mask];
        //     }
        // }
        
        int currMask = (1 << M) - 1;
        int ansSize = best[(1 << M) - 1];
        int[] ans = new int[ansSize];
        for (int i = ansSize - 1; i >= 0; --i) {
            // System.out.println(currMask);
            ans[i] = currVal[currMask];
            currMask = prevBest[currMask];
        }
        return ans;
        // return best[(1 << M) - 1];
    }
}
