class Solution {
    private Set<Integer> res = null;
    private Map<Integer, Integer> cache = new HashMap<>();

    private int[][] groups;
    private int[] peopleSkills;
    private int targetSkill;

    public int[] smallestSufficientTeam(String[] req_skills, List<List<String>> people) {
        targetSkill = (1 << req_skills.length) - 1;
        Map<String, Integer> skillValues = IntStream.range(0, req_skills.length).boxed().collect(Collectors.toMap(i -> req_skills[i], i -> 1 << i));
        peopleSkills = people.stream().mapToInt(skills -> skills.stream().mapToInt(skillValues::get).reduce((a, b) -> a | b).orElse(0)).toArray();
        groups = IntStream.range(0, req_skills.length)
                .map(i -> 1 << i)
                .boxed()
                .map(i -> IntStream.range(0, peopleSkills.length).filter(j -> (peopleSkills[j] & i) != 0).toArray())
                .toArray(int[][]::new);
        _smallest(0, 0, new HashSet<>());
        return res.stream().mapToInt(Integer::intValue).toArray();
    }

    private void _smallest(int i, int skills, Set<Integer> people) {
        if (skills == targetSkill) {
            if (res == null || people.size() < res.size()) res = new HashSet<>(people);
            return;
        }
        if (i >= groups.length) return;

        Integer cacheKey = (skills | (i << 16));
        if (cache.getOrDefault(cacheKey, Integer.MAX_VALUE) <= people.size()) return;
        cache.put(cacheKey, people.size());

        for (int n : groups[i]) {
            if (people.contains(n)) {
                _smallest(i + 1, skills, people);
            } else {
                people.add(n);
                _smallest(i + 1, skills |  peopleSkills[n], people);
                people.remove(n);
            }
        }
    }
}
