class Solution {
    public int[] smallestSufficientTeam(String[] skills, List<List<String>> people) {
        int sl = skills.length;
        int pl = people.size();
        HashMap<String, Integer> rshm = new HashMap<>();
        for (int i = 0; i < sl; i++)
            rshm.put(skills[i], i);
        int[] pbits = new int[pl];
        for (int i = 0; i < pl; i++) {
            for (int j = 0; j < people.get(i).size(); j++) {
                if (rshm.containsKey(people.get(i).get(j))) {
                    pbits[i] |= (1<<rshm.get(people.get(i).get(j)));
                }
            }
            //System.out.println(i+" "+pbits[i]);
        }
        int[] dists = new int[1<<sl];
        int[] lp = new int[1<<sl];
        int[] prev = new int[1<<sl];
        for (int j = (1<<sl)-1; j > 0; --j) {
            dists[j] = 1000;
        }
        for (int i = 0; i < pl; ++i) {
            int bit = pbits[i];
            for (int j = (1<<sl)-1; j >= 0; --j) {
                int nd = dists[j]+1;
                
int nn = j|bit;
                if (dists[nn] > nd) {
                    dists[nn]=nd;
                    lp[nn] = i;
                    prev[nn] = j;
                }
            }
        }
        int d = dists[(1<<sl)-1];
        int[] ans = new int[d];
        int num = (1<<sl)-1;
        
for (int i = d-1; i >= 0; i--) {
            ans[i] = lp[num];
            num = prev[num];
            //System.out.println(num);
        }
        return ans;
    }
}
