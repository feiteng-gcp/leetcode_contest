class Solution {
    List<Integer> res = new LinkedList<>();
    public int[] smallestSufficientTeam(String[] req_skills, List<List<String>> people) {
        Map<String, Integer> map = new HashMap<>();
        for (int i = 0; i < req_skills.length; i++)
            map.put(req_skills[i], i);
        Map<Integer, List<Integer>> knows = new HashMap<>();
        
for (int i = 0; i < people.size(); i++) {
            List<String> peo = people.get(i);
            for (String s : peo) {
                if (map.containsKey(s)) {
                    int index = map.get(s);
                    knows.putIfAbsent(index, new LinkedList<>());
                    
knows.get(index).add(i);
                }
            }
        }
        int [] cover = new int[req_skills.length];
        helper(cover, 0, map, knows, people, new LinkedList<>());
        int[] ans = new int[res.size()];
        for (int i = 0; i < ans.length; i++)
            ans[i] = res.get(i);
        return ans;
    }
    
    public void helper(int[] cover, int index, Map<String, Integer> map, Map<Integer, List<Integer>> knows, List<List<String>> people, List<Integer> temp) {
        // System.out.println(index);
        if (index == cover.length) {
            if (res.size() == 0 || res.size() > temp.size()) {
                res = new LinkedList<>(temp);
            }
        } else {
            if (res.size() != 0 && res.size() <= temp.size())
                return;
            if (cover[index] != 0) 
                helper(cover, index+1, map, knows, people, temp);
            for (int p : knows.get(index)) {
                temp.add(p);
                List<String> list = people.get(p);
                for (String s : list) 
                    cover[map.get(s)]++;
                helper(cover, index+1, map, knows, people, temp);
                temp.remove(temp.size()-1);
                for (String s : list)
                    cover[map.get(s)]--;
            }
        }
    }
}
