class Solution {
    public int[] smallestSufficientTeam(String[] r, List<List<String>> people) {
        Map<String,Integer> map = new HashMap<>();
        int reqskill = 0;
        for(int i =0; i<r.length; i++){
            map.put(r[i],i);
            reqskill |= (1<<i);
        }
        int[] p = new int[people.size()];
        for(int i = 0;i<people.size(); i++){
            for(String s : people.get(i)){
                if(map.containsKey(s)){
                    p[i] |=( 1 << map.get(s));
                }
            }
        }
        for(int i =1; i <= p.length; i++){
            List<Integer> res = new ArrayList<>();
            if(dfs(p,reqskill,res, i,0,0)){
                int[] ret = new int[res.size()];
                for(int j = 0;j<res.size();j++){
                    ret[j] = res.get(j);
                }
                return ret;
            }
        }
        return null;// never call
    }
    private boolean dfs(int[] p, int reqskill, List<Integer> res, int cnt, int prev, int prev_idx){
        for(int i = prev_idx; i<p.length; i++){
            if(p[i]==0) continue;
            int curr = p[i]|prev;
            if(curr == reqskill){
                res.add(i);
                return true;
            }
            if(cnt > 1){
                res.add(i);
                boolean next = dfs(p,reqskill,res,cnt-1,curr, i+1);
                if(next){
                    return true;
                }
                res.remove(res.size()-1);
            }
        }
        return false;
    }
}
