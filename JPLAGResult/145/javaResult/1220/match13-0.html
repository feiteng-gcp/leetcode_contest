class Solution {
    public int[] smallestSufficientTeam(String[] req_skills, List<List<String>> people) {
        Map<String, Integer> skillToBit = new HashMap<>();
        int allSkillsBit = 0;
        for (int i = 0; i < req_skills.length; i++) {
            int bit = 1 << i;
            skillToBit.put(req_skills[i], bit);
            allSkillsBit |= bit;
        }
        Map<Integer, Integer> min = new HashMap<>();
        Map<Integer, Long> minTeam = new HashMap<>();
        min.put(0, 0);
        minTeam.put(0, 0L);
        for (int i = 0; i < people.size(); i++) {
            int bit = 0;
            long personBit = 1L << i;
            for (String skill : people.get(i)) {
                bit |= skillToBit.get(skill);
            }
            Map<Integer, Integer> next = new HashMap<>(min);
            Map<Integer, Long> nextTeam = new HashMap<>(minTeam);
            for (Map.Entry<Integer, Integer> entry : min.entrySet()) {
                int k = bit | entry.getKey();
                int v = entry.getValue() + 1;
                if (!next.containsKey(k) || v < next.get(k)) {
                    next.put(k, v);
                    nextTeam.put(k, minTeam.get(entry.getKey()) | personBit);
                }
            }
            min = next;
            minTeam = nextTeam;
        }
        long team = minTeam.get(allSkillsBit);
        List<Integer> indexes = new ArrayList<>();
        for (int i = 0; i < people.size(); i++) {
            if ((team & (1L << i)) != 0) {
                indexes.add(i);
            }
        }
        return indexes.stream().mapToInt(Integer::intValue).toArray();
    }
}
