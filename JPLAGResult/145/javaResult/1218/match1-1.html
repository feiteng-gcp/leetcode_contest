/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        return dfs(root, 0).node;
    }
    private Ans dfs(TreeNode root, int deep){
        if(root==null) return new Ans();
        if(root.left==null && root.right==null){
            Ans ans = new Ans();
            ans.deep = deep;
            ans.node = root;
            return ans;
        }
        Ans left = dfs(root.left, deep+1), right = dfs(root.right, deep+1);
        if(left.deep==right.deep){
            left.node = root;
            return left;
        }else if(left.deep>right.deep){
            return left;
        }else {
            return right;
        }
    }
    
    class Ans{
        TreeNode node;
        int deep=-1;
    }
}
