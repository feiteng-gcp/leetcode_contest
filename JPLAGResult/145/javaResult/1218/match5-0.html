/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private class ResultType {
        public int depth;
        public TreeNode lca;
        
        public ResultType(int d, TreeNode n) {
            depth = d;
            lca = n;
        }
    }
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        ResultType res = find(root);
        return res.lca;
    }
    private ResultType find(TreeNode root) {
        if(root == null) {
            return new ResultType(0, null);
        }
        ResultType lRes = find(root.left);
        ResultType rRes = find(root.right);
        
        if(lRes.depth == rRes.depth) {
            return new ResultType(lRes.depth+1, root);
        } else {
            return new ResultType(Math.max(lRes.depth, rRes.depth)+1, rRes.depth > lRes.depth ? rRes.lca : lRes.lca);
        }
    }
}
