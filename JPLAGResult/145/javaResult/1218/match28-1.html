/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class ReturnType {
    TreeNode node;
    int deep;
    public ReturnType(TreeNode node, int deep) {
        this.node = node;
        this.deep = deep;
    }
}
class Solution {
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        return helper(root, 0).node;
    }
    ReturnType helper(TreeNode root, int deep) {
        if (root == null) {
            return new ReturnType(null, -1);
        }
        
if (root.left == null && root.right == null) {
            return new ReturnType(root, deep);
        }
        ReturnType left = helper(root.left, deep + 1);
        ReturnType right = helper(root.right, deep + 1);
        if (left.node != null && right.node != null) {
            if (left.deep > right.deep) {
                return left;
            } else if (left.deep < right.deep) {
                return right;
            }
            
return new ReturnType(root, left.deep);
        }
        if (left.node != null) {
            return left;
        }
        if (right.node != null) {
            return right;
        }
        return new ReturnType(null, -1);
    }
}
