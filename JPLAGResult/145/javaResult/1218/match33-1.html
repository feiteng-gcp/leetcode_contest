/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        if (root == null)
            return null;
        if (root.left == null && root.right == null)
            return root;
        List<TreeNode> nodesList = new ArrayList<TreeNode>();
        List<Integer> depthsList = new ArrayList<Integer>();
        Queue<TreeNode> leavesQueue = new LinkedList<TreeNode>();
        Queue<Integer> leavesDepthsQueue = new LinkedList<Integer>();
        Queue<TreeNode> queue = new LinkedList<TreeNode>();
        Queue<Integer> depthsQueue = new LinkedList<Integer>();
        queue.offer(root);
        depthsQueue.offer(0);
        int maxDepth = 0;
        while (!queue.isEmpty()) {
            TreeNode node = queue.poll();
            int depth = depthsQueue.poll();
            nodesList.add(node);
            depthsList.add(depth);
            if (isLeaf(node)) {
                leavesQueue.offer(node);
                leavesDepthsQueue.offer(depth);
            } else {
                if (node.left != null) {
                    queue.offer(node.left);
                    depthsQueue.offer(depth + 1);
                    maxDepth = Math.max(maxDepth, depth + 1);
                }
                if (node.right != null) {
                    queue.offer(node.right);
                    depthsQueue.offer(depth + 1);
                    maxDepth = Math.max(maxDepth, depth + 1);
                }
            }
        }
        if (leavesQueue.size() == 1)
            return leavesQueue.poll();
        int size = leavesQueue.size();
        for (int i = 0; i < size; i++) {
            TreeNode node = leavesQueue.poll();
            int depth = leavesDepthsQueue.poll();
            if (depth == maxDepth) {
                leavesQueue.offer(node);
                leavesDepthsQueue.offer(depth);
            }
        }
        while (!leavesQueue.isEmpty() && leavesQueue.size() > 1) {
            TreeNode node1 = leavesQueue.poll();
            TreeNode node2 = leavesQueue.poll();
            TreeNode commonAncestor = lowestCommonAncestor(root, node1, node2);
            if (!leavesQueue.contains(commonAncestor))
                leavesQueue.offer(commonAncestor);
        }
        return leavesQueue.poll();
    
}

    private boolean isLeaf(TreeNode node) {
        if (node == null)
            return false;
        return node.left == null && node.right == null;
    }

    public TreeNode lowestCommonAncestor(TreeNode root, TreeNode p, 
TreeNode q) {
        if (root == null || root == p || root == q)
            return root;
        TreeNode left = lowestCommonAncestor(root.left, p, q);
        TreeNode right = lowestCommonAncestor(root.right, p, q);
        if (left != null && right != null)
            return root;
        return left == null ? right : left;
    }
}
