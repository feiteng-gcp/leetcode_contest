/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        if(root == null) return null;
        
if(root.left == null && root.right == null) return root;
        
        return helper(root).node;
    }
    
    
    private HelperClass helper(TreeNode node){
        if(node == null) return new HelperClass(null, 0);
        
        if(node.left == null && node.right == null) return new HelperClass(node,1);
        
        HelperClass left = helper(node.left), right = helper(node.right);
        
        if(node.right == null){
            left.height = left.height+1;
            return left;
        }
        if(node.left == null){
           right.height = right.height +1;
            return right;
        } 
        

        
        if(left.height > right.height){
            left.height = left.height+1;
            return left;
        }else if(left.height < right.height){
           right.height = right.height +1;
            return right;
        
}else{
            
            return new HelperClass(node, left.height+1); 
        }
    }
    
    
    class HelperClass{
        TreeNode node;
        int height;
        
        public HelperClass(TreeNode node, int height){
            this.node = node;
            this.height = height;
        }
    }
}
