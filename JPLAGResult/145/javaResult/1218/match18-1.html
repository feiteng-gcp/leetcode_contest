/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Map<TreeNode, Integer> map = new HashMap<>();
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        // get max depth leaf;
        int max = maxDepth(root);
        System.out.println(max);
        return find(root, max);
    }
    public TreeNode find(TreeNode root, int max){
        if(root == null) return root;
        if (maxDepth(root.left) == max-1 && maxDepth(root.right) == max-1) return root;
        if (maxDepth(root.left) == max-1) 
return find(root.left, max-1);
        else return find(root.right, max-1);
    }
    
    // depth
    public int maxDepth(TreeNode root) {
    if (root == null) return -1;
    return Math.max(maxDepth(root.left),maxDepth(root.right)) + 1;
    }
}
