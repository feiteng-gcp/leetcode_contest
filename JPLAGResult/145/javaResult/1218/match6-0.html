/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        if (root == null)  return null;
        return helper(root).commonAncestor;
    }
    
    private ReturnType helper(TreeNode node) {
        if (node == null) return new ReturnType(-1, null); 
        if (node.left == null && node.right == null) return new ReturnType(1, node);
        
        ReturnType leftRes = helper(node.left);
        ReturnType rightRes = helper(node.right);
        
        if (leftRes.deepest == rightRes.deepest) {
            return new ReturnType(leftRes.deepest + 1, node); 
        } else if (leftRes.deepest >= rightRes.deepest) {
            return new ReturnType(leftRes.deepest + 1, leftRes.commonAncestor);
        } else {
            return new ReturnType(rightRes.deepest + 1, rightRes.commonAncestor);
        }
    }
    
    class ReturnType {
        int deepest;
        TreeNode commonAncestor;
        public ReturnType(int d, TreeNode node) {
            this.deepest = d;
            this.commonAncestor = node;
        }
    }
}
