/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class pair {
    
    int d;
    TreeNode lca;
    public pair(int de,TreeNode l) {
        d=de;
        lca=l;
    }
    
}

class Solution {
    
    public pair fun(TreeNode root,int d) {
        
        if(root==null) {
            return new pair(-1,root);
        }
        
        if(root.left==null && root.right==null) {
            
            return new pair(d,root);
            
        }
        
        pair pL=fun(root.left,d+1);
        pair pR=fun(root.right,d+1);
        
        if(pL.d>pR.d) {
            return pL;
        }
        if(pL.d<pR.d) {
            return pR;
        }
        
        return new pair(pL.d,root);
        
    }
    
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        
        return fun(root,0).lca;
    
    }
}
