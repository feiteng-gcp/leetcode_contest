/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        if(root ==null)
            return root;
        int maxDeep  = getMaxDeep(root);
        int[] numMax = new int[1];
        getNumOfMaxDeep(root,maxDeep,1,numMax);
        TreeNode[] lca = new TreeNode[1];
        getLca(root,maxDeep,1,numMax[0],lca);
        return lca[0];
        
    
}
    private int getMaxDeep(TreeNode root){
        if(root==null)
            return 0;
        return 1+Math.max(getMaxDeep(root.left),getMaxDeep(root.right));
    }
    private void getNumOfMaxDeep(TreeNode root,int maxDeep,int level,int[] count){
        if(root==null)
            return ;
        if(level==maxDeep){
            ++count[0];
            return;
        }
        getNumOfMaxDeep(root.left,maxDeep,level+1,count);
        getNumOfMaxDeep(root.right,maxDeep,level+1,count);
    }
    private int getLca(TreeNode root,int maxDeep,int level, int numMax, TreeNode[] lca){
        if(root==null)
            return 0;
        if(lca[0]!=null)
            return 0;
        int ans = 0;
        if(level == maxDeep)
            ++ans;
        ans+=getLca(root.left,maxDeep,level+1,numMax,lca);
        ans+=getLca(root.right,maxDeep,level+1,numMax,lca);
        if(ans == numMax){
            lca[0] = root;
            return 0;
        }
        return ans;
    }
}
