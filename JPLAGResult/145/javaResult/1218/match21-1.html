/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
public TreeNode lcaDeepestLeaves(TreeNode root) {
        Data data = lca(root, 0);
        return data.root;
    }

    private Data lca(TreeNode root, int d) {
        if (root == null) {
            return null;
        
}
        Data left = lca(root.left, d + 1);
        Data right = lca(root.right, d + 1);
        if (left == null && right == null) {
            return new Data(root, d);
        }
        if (left == null) return right;
        if (right == null) return left;
        if (left.d > right.d) return left;
        if (left.d < right.d) return right;
        return new Data(root, left.d);
    }


    
class Data {
        TreeNode root;
        int d;

        public Data(TreeNode root, int d) {
            this.root = root;
            this.d = d;
        }
    }
}
