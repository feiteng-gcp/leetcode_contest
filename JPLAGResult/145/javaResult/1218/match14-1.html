/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
class LNode {
        TreeNode lca;
        int depth;
        public LNode(TreeNode node, int d) {
            this.lca = node;
            this.depth = d;
        }
    }
    
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        return dfs(root).lca;
    
}
    
    private LNode dfs(TreeNode root) {
        if (root == null) return new LNode(null, 0);
        LNode l = dfs(root.left);
        LNode r = dfs(root.right);
        if (l.depth > r.depth) return new LNode(l.lca, l.depth + 1);
        else if (l.depth < r.depth) return new LNode(r.lca, r.depth + 1);
        return new LNode(root, l.depth + 1);
    }
}
