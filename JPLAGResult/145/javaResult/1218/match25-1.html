/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
class Pair {
        TreeNode root;
        int depth;
        public Pair(TreeNode root, int depth) {
            this.root = root;
            this.depth = depth;
        }
    }
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        return findLca(root, 0).root;
    
}
    private Pair findLca(TreeNode root, int d) {
        if (root == null) {
            return null;
        }
        Pair left = findLca(root.left, d + 1);
        Pair right = findLca(root.right, d + 1);
        if (left == null && right == null) {
            return new Pair(root, d);
        } else if (left == null) {
            return right;
        } else if (right == null) {
            return left;
        } else if (left.depth > right.depth) {
            return left;
        } else if (left.depth < right.depth) {
            return right;
        } else {
            return new Pair(root, left.depth);
        }
    }
}
