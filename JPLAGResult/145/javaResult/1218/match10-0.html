/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public class Ret{
        TreeNode node;
        int depth;
        Ret(TreeNode node, int depth){
            this.node = node;
            this.depth = depth;
        }
    }
    
    public Ret  rec(TreeNode root, int depth){
        
        if(root == null){
            return null;
        }
        
        if(root.left == null && root.right ==null){
            return new Ret(root, depth);
        }
        
        Ret left, right;
        left = rec(root.left, depth + 1);
        right = rec(root.right, depth + 1);
        
        if(left == null){
            return right;
        }
        if(right == null){
            return  left;
        }
        
        if(left.depth == right.depth ){
            return new Ret(root, left.depth);
        }
        
        return left.depth> right.depth? left: right;
        
    }
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        
        if(root == null){
            return null;
        }
        return rec(root,0).node;
        
    }
}
