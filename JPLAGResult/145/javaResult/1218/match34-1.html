/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        int dep = depth(root) - 1;
        // System.out.println(dep);
        return dfs(root, 0, dep);
        
    
}
    public TreeNode dfs(TreeNode root, int level, int depth) {
        if(root == null || level == depth) return root;
        TreeNode left = dfs(root.left, level + 1, depth);
        TreeNode right = dfs(root.right, level + 1, depth);
        return left == null ? right : right == null ? left : root;
    
}
    public int depth(TreeNode root) {
        if(root == null) return 0;
        return Math.max(depth(root.left), depth(root.right)) + 1;
    }
