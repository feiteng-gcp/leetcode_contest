/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public class DepthNode{
        TreeNode node;
        int depth;
        public DepthNode(TreeNode _node, int _depth){
            node = _node;
            depth = _depth;
        }
    }
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        DepthNode dn = dfs(root, 0);
        return dn.node;
    }
    private DepthNode dfs(TreeNode node, int d){
        if(node == null) return null;
        DepthNode left = dfs(node.left, d + 1);
        DepthNode right = dfs(node.right, d + 1);
        
        if(left == null && right == null) return new DepthNode(node, d);
        else if(left == null) return right;
        else if(right == null) return left;
        else{
            if(left.depth == right.depth) return new DepthNode(node, left.depth);
            else if(left.depth > right.depth) return left;
            else return right;
        }
    }
}
