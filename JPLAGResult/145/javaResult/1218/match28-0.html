/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        if (root == null) {
            return null;
        }
        return helper(root).node;
    }
    private Pair helper(TreeNode root) {
        if (root == null) {
            return new Pair(null, 0);
        }
        Pair left = helper(root.left), right = helper(root.right);
        // System.out.println(root.val + " " + left.len + " " + right.len);
        if (left.len == right.len) {
            return new Pair(root, left.len + 1);
        }
        if (left.len > right.len) {
            left.len ++;
            return left;
        }
        right.len ++;
        return right;
    }
    class Pair {
        TreeNode node;
        int len;
        public Pair(TreeNode node, int len) {
            this.node = node;
            this.len = len;
        }
    }
}
