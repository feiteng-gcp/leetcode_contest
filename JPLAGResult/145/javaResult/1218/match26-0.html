/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode lcaDeepestLeaves(TreeNode root) {
        Helper helper = helper(root, 0);
        return helper.node;
    }
    
    Helper helper(TreeNode node, int depth){
        if(node == null){
            return null;
        }
        Helper left = helper(node.left, depth+1);
        Helper right = helper(node.right, depth+1);
        if(left!=null && right !=null){
            if(left.depth == right.depth){
                return new Helper(left.depth, node);
            }
            else if (left.depth>right.depth){
                return left;
            }
            else {
                return right;
            }
        }
        else if(left == null && right == null){
            return new Helper(depth, node);
        }
        else if(left == null){
            return right;
        }
        return left;
    }
    static class Helper{
        int depth;
        TreeNode node;
        public Helper(int depth, TreeNode node){
            this.depth = depth;
            this.node = node;
        }
    }
}
