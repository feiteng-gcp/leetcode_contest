/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
public TreeNode lcaDeepestLeaves(TreeNode root) {
        Detail ans = find(root);
        return ans.node;
    }
    
    public Detail find(TreeNode root){
        if (root == null) return new Detail()
;
        
        Detail left = find(root.left);
        Detail right = find(root.right);
        
        left.dist++;
        right.dist++;
        if (left.dist == right.dist) return new Detail(root, left.dist);
        else if (left.dist > right.dist) return left;
        else return right;
    }
}


class Detail {
    TreeNode node;
    int dist;
    public Detail(TreeNode node, int dist){
        this.node = node;
        this.dist = dist;
    }
    public Detail(){
        dist = 0;
    }
}
