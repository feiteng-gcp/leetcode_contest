/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    // Testcase 2 ?????
    
    
class NodeDepth {
        int depth;
        TreeNode commonAncestor;
        
        public NodeDepth(int d, TreeNode node){
            this.depth = d;
            this.commonAncestor = node;
        }
    }
    
    
public TreeNode lcaDeepestLeaves(TreeNode root) {
        NodeDepth result = DFS(root);
        return result.commonAncestor;
    }
    
    private NodeDepth DFS(TreeNode node){
        // Terminate
        
if(node == null){
            return new NodeDepth(0, null);
        }
        
        NodeDepth left = DFS(node.left);
        NodeDepth right = DFS(node.right);
        
        // Root is a CA.
        if(left.depth == right.depth){
            return new NodeDepth(left.depth+1, node);
        } else if(left.depth > right.depth){
            return new NodeDepth(left.depth+1, left.commonAncestor);
        
} else {
            return new NodeDepth(right.depth+1, right.commonAncestor);
        }
    }
}
