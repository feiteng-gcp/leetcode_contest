class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        int[] count = new int[1001];
        for(int n: arr1) {
            count[n]++;
        }
        int[] res = new int[arr1.length];
        
int i = 0;
        for(int n: arr2) {
            while(count[n]-- > 0) {
                res[i++] = n;
            }
        }
        for(int j = 0; j < count.length; j++) {
            if(count[j] == 0) continue;
            
while(count[j]-- > 0) {
                res[i++] = j;
            }
        }
        return res;
        
    }
}
