class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        Map<Integer, Integer> map = new HashMap<>();
        
        for (int i = 0 ; i < arr2.length; i++) {
            map.put(arr2[i], i);
        }
        
        List<Integer> l1 = new ArrayList<>();
        List<Integer> l2 = new ArrayList<>();
        
        for (int n : arr1) {
            if (map.containsKey(n)) l1.add(n);
            else l2.add(n);
        }
        
        Collections.sort(l1, (n1, n2)->(map.get(n1) - map.get(n2)));
        Collections.sort(l2);
        
        int[] res = new int[arr1.length];
        int i = 0;
        for (int n : l1) {
            res[i++] = n;
        }
        
        for (int n : l2) {
            res[i++] = n;
        }
        
        return res;
    }
}
