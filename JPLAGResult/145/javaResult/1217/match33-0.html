class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        Map<Integer,Integer> replacedNum = new HashMap<>();
        for(int i=0;i<=1000;++i)
            replacedNum.put(i,i);
        for(int i=0;i<arr2.length;++i)
            replacedNum.put(arr2[i],i-100000);
        for(int i=0;i<arr1.length;++i){
            for(int j=arr1.length-1;j>i;--j){
                if(replacedNum.get(arr1[j]) < replacedNum.get(arr1[j-1])){
                    int temp = arr1[j];
                    arr1[j] = arr1[j-1];
                    arr1[j-1] = temp;
                }
            }
        }
        return arr1;
    }
}
