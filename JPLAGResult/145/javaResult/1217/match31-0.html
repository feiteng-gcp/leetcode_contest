class Solution {
    Map<Integer, Integer> map;
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        map = getMap(arr2);
        Integer[] A1 = new Integer[arr1.length];
        for (int i = 0; i < arr1.length; i++) {
            A1[i] = arr1[i];
        }
        Arrays.sort(A1, new Comparator<Integer>() {
            @Override
            public int compare(Integer a1, Integer a2) {
                Integer i = map.get(a1);
                Integer j = map.get(a2);
                if (i == null && j == null) return a1.compareTo(a2);
                else if (i == null) return 1;
                else if (j == null) return -1;
                else return i.compareTo(j);
            }
        });
        for (int i = 0; i < arr1.length; i++) {
            arr1[i] = A1[i];
        }
        return arr1;
    }
    private Map<Integer, Integer> getMap(int[] a) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < a.length; i++) {
            map.put(a[i], i);
        }
        return map;
    }
}
