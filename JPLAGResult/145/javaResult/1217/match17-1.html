class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        int[] result = new int[arr1.length];
        TreeMap<Integer, Integer> a1Map = new TreeMap<>();
        for (int n : arr1) {
            a1Map.put(n, a1Map.getOrDefault(n, 0) + 1);
        
}
        int index = 0;
        for (int i = 0; i < arr2.length; i++) {
            int count = a1Map.get(arr2[i]);
            while (count > 0) {
                result[index++] = arr2[i];
                count--;
            }
            a1Map.remove(arr2[i]);
        }
        for (Map.Entry<Integer, Integer> entry : a1Map.entrySet()) {
            int key = entry.getKey();
            int value = entry.getValue();
            
while (value > 0) {
                result[index++] = key;
                value--;
            }
        }
        return result;
    }
}
