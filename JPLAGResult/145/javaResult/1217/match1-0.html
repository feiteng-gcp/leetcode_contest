class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        // int[] rank = new int[1111];
        // for (int i = 0; i < arr2.length; ++i) {
        //     rank[arr2[i]] = i;
        // }
        int[] freq = new int[1111];
        for (int v : arr1) {
            ++freq[v];
        }
        int[] ans = new int[arr1.length];
        int idx = 0;
        for (int v : arr2) {
            while (freq[v] > 0) {
                ans[idx++] = v;
                --freq[v];
            }
        }
        for (int i = 0; i < freq.length; ++i) {
            while (freq[i] > 0) {
                ans[idx++] = i;
                --freq[i];
            }
        }
        return ans;
    }
}
