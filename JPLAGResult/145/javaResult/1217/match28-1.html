class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        Integer[] arr1C = new Integer[arr1.length];
        for(int i = 0; i < arr1.length; i++) {
            arr1C[i] = arr1[i];
        }
        HashMap<Integer, Integer> freq = new HashMap<>();
        for(int i = 0; i < arr2.length; i++) {
            freq.put(arr2[i], i);
        
}
        Arrays.sort(arr1C, (o1, o2)-> {
            int o1i = freq.getOrDefault(o1, Integer.MAX_VALUE);
            int o2i = freq.getOrDefault(o2, Integer.MAX_VALUE);
            if(o1i != Integer.MAX_VALUE && o2i != Integer.MAX_VALUE) {
                return o1i - o2i;
            } else if(o1i != Integer.MAX_VALUE) {
                return -1;
            } else if(o2i != Integer.MAX_VALUE) {
                return 1;
            } else {
                return o1 - o2;
            }
        });
        
for(int i = 0; i < arr1.length; i++) {
            arr1[i] = arr1C[i];
        }
        return arr1;
    }
}
