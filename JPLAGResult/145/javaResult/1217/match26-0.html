class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        if (arr1.length == 0 || arr2.length == 0) {
            Arrays.sort(arr1);
            return arr1;
        }

        Map<Integer, Integer> numToIdx = new HashMap<>();
        for (int i = 0; i < arr2.length; i++) {
            numToIdx.put(arr2[i], i);
        }
        
        Integer[] temp = new Integer[arr1.length];
        for (int i = 0; i < temp.length; i++) {
        	temp[i] = arr1[i];
        }
        
        
        Arrays.sort(temp, new Comparator<Integer>() {
            @Override
            public int compare(Integer num1, Integer num2) {
                if (numToIdx.containsKey(num1) && numToIdx.containsKey(num2)) {
                    return Integer.compare(numToIdx.get(num1), numToIdx.get(num2));
                } else if (numToIdx.containsKey(num1) || numToIdx.containsKey(num2)) {
                    return numToIdx.containsKey(num1) ? -1 : 1;
                } else {
                    return Integer.compare(num1, num2);
                }
            }
        });
        
        for (int i = 0; i < arr1.length; i++) {
        	arr1[i] = temp[i];
        }
        
        return arr1;
    }
}
