class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
       Map<Integer, Integer> posMap = new HashMap<>();
        for (int i = 0; i < arr2.length; i++) {
            posMap.put(arr2[i], i);
        }
        int N = arr1.length;
        
for (int i = 0; i < N; i++) {
            for (int j = i + 1; j < N; j++) {
                if ((!posMap.containsKey(arr1[i])) && (!posMap.containsKey(arr1[j]))) {
                    if (arr1[i] > arr1[j]) {
                        int temp = arr1[i];
                        arr1[i] = arr1[j];
                        arr1[j] = temp;
                    }
                } else if (!posMap.containsKey(arr1[i])) {
                    int temp = arr1[i];
                    arr1[i] = arr1[j];
                    arr1[j] = temp;
                } else if (!posMap.containsKey(arr1[j])) {
                    continue;
                } else if (posMap.get(arr1[i]) > posMap.get(arr1[j])) {
                    int temp = arr1[i];
                    arr1[i] = arr1[j];
                    arr1[j] = temp;
                
}
            }
        }
        return arr1;
    }
}
