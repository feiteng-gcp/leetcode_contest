class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        int[] memo = new int[1001];
        for (int i : arr1) {
            memo[i]++;
        }
        int j = 0;
        for (int i : arr2) {
            while (memo[i] != 0) {
                arr1[j] = i;
                j++;
                memo[i]--;
            }
        }
        for (int i = 0; i < 1001; i++) {
            while (memo[i] != 0) {
                arr1[j] = i;
                j++;
                memo[i]--;
            }
        }
        return arr1;
    }
}
