class Solution {
    public int[] relativeSortArray(int[] arr1, int[] arr2) {
        ArrayList<Integer> np = new ArrayList<>();
        HashMap<Integer, Integer> hm = new HashMap<>();
        for (int i : arr2)
            hm.put(i, 0);
        
for (int i : arr1) {
            if (hm.containsKey(i)) {
                hm.put(i, hm.get(i)+1);
            } else {
                np.add(i);
            }
        }
        Collections.sort(np);
        
int[] ans = new int[arr1.length];
        int ind = 0;
        for (int i : arr2) {
            for (int j = 0; j < hm.get(i); j++) {
                ans[ind] = i;
                ind++;
            }
        }
        
for (int i : np) {
            ans[ind] = i;
            ind++;
        }
        return ans;
    }
}
