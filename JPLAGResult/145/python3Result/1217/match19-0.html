class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        d = collections.defaultdict(int)
        for i, val in enumerate(arr2):
            d[val] = i
        ret = []
        ret1 = []
        for val in arr1:
            if val in d:
                ret.append(val)
            else:
                ret1.append(val)
        ret.sort(key=lambda x: d[x])
        ret1.sort()
        return ret + ret1
