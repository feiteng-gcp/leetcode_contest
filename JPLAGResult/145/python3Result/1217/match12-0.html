import collections


class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        arr2_set = set(arr2)
        arr1_1 = []
        arr1_2 = []
        for num in arr1:
            if num not in arr2_set:
                arr1_2.append(num)
            else:
                arr1_1.append(num)
        arr1_2.sort()

        arr1_1_count = collections.Counter(arr1_1)
        result = []
        for num in arr2:
            result.extend([num] * arr1_1_count[num])
        result.extend(arr1_2)
        return result
