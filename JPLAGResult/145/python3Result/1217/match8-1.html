class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        from collections import Counter
        c 
= Counter(arr1)
        result = []
        for key in arr2:
            result.extend([key]*c[key])
            del c[key]
        for k, v in sorted(c.items()):
            result.extend([k]*v)
        return result
