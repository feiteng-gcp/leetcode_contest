class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        arr2_set = set(arr2)
        appears = collections.Counter()
        not_appears = []
        
for num in arr1:
            if num in arr2_set:
                appears[num] += 1
            else:
                not_appears.append(num)
        output = []
        for num in arr2:
            to_add = [num] * appears[num]
            output.extend(to_add)
        not_appears.sort()
        output.extend(not_appears)
        return output
