from typing import List
from collections import Counter



class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        arr1_count = Counter(arr1)
        res = []

        for element2 in arr2:
            if element2 in arr1_count:
                res.extend([element2] * arr1_count[element2])
                del arr1_count[element2]

        remaining_arr = []
        for k, v in arr1_count.items():
            remaining_arr.extend([k] * v)
        remaining_arr.sort()
        res.extend(remaining_arr)

        return res
