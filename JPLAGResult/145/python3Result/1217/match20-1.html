class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        index = {}
        for i in range(len(arr2)):
            index[arr2[i]] = i
        pairs = []
        inf = 1 << 29
        for i in arr1:
            idx = index[i] if i in index else inf
            pairs.append((idx, i))
        pairs.sort()
        return [val for _, val in pairs]
