class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        s = {}
        for i in arr2:
            s[i] = 0
        right = []
        for i in arr1:
            if i in s:
                s[i] += 1
            else:
                right.append(i)
        right.sort()
        left = []
        for i in arr2:
            for j in range(s[i]):
                left.append(i)
        for i in right:
            left.append(i)
        return left
