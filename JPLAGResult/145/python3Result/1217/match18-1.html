class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        counts = [0] * 1001
        for a in arr1:
            counts[a] += 1
        result = []
        for a in arr2:
            result += [a] * counts[a]
            counts[a] = 0
        for i in range(1001):
            result += [i] * counts[i]
        return result
