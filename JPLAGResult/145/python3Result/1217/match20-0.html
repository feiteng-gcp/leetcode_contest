class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        dic={}
        for i in range(len(arr2)):
            dic[arr2[i]]=i
        for i in arr1:
            if i not in dic:
                dic[i]=10000+i
        arr=list(map(lambda x:dic[x],arr1))
        arr.sort()
        L=[]
        for i in arr:
            if i<len(arr2):
                L.append(arr2[i])
            else:
                L.append(i-10000)
        return L
