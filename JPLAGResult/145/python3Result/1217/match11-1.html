class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        if not arr2:
            return sorted(arr1)
        if not arr1:
            return []
        hashmap = {arr2[i]:i for i in range(len
(arr2))}
        temp = []
        other = []
        for ele in arr1:
            if ele in hashmap:
                temp.append(ele)
            else:
                other.append(ele)
        temp.sort(key = lambda x:hashmap[x])
        return temp + sorted(other)
