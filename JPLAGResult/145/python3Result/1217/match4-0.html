class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        
        
        set_arr2 = set(arr2)
        residue = []
        d = collections.defaultdict(int)
        for num in arr1:
            if num in set_arr2:
                d[num] += 1
            else:
                residue.append(num)
        
        res = []
        for num in arr2:
            res += [num]*d[num]
        return res + sorted(residue)
