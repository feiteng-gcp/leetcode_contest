class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        s2 = set(arr2)
        arr3 = []
        for a in arr1:
            if a not in s2:
                arr3.append(a)
        arr2.extend(sorted(arr3))
        arr1.sort(key=arr2.index)
        print(arr1)
        return arr1
