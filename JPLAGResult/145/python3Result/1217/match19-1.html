class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        
        i_map = {a: i for i, a in enumerate(arr2)}
        arr2 = set(arr2)
        first 
= []
        last = []
        for a in arr1:
            if a in arr2: first.append(a)
            else: last.append(a)
        
        last.sort()
        first.sort(key=lambda x: i_map[x])
        return first + last
