import collections
class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        a = collections.Counter(arr1)
        res= []
        for i in arr2:
            p = [i]*a[i]
            res.extend(p)
        
        p2 = []
        for i in a.keys():
            if i not in arr2:
                p = [i]*a[i]
                p2.extend(p)
        p2.sort()
        res.extend(p2)
        return res
