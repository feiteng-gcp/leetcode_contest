class Solution:
    def longestWPI(self, hours: List[int]) -> int:
        A=[]
        for x in hours:
            if x>8:
                A+=[1]
            else:
                A+=[-1]
        temp=0
        
        pre=[]
        for x in A:
            temp+=x
            pre+=[temp]
        #print(pre)
        dic={}
        MAX=0
        for i in range(len(pre)):
            if A[i]>0:
                MAX=max(MAX,1)
            
            
            if pre[i]>0:
                MAX=i+1
                continue
            
            if pre[i]<=0:
                temp=pre[i]-1
                if temp in dic:
                    position=dic[temp]
                    MAX=max(
MAX,i-position)
                if pre[i] not in dic:
                    dic[pre[i]]=i
        return MAX
