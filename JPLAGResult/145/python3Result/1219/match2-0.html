class Solution:
    def longestWPI(self, hours: List[int]) -> int:
        s = {0 : -1}
        acc = 0
        ret = 0
        for i, h in enumerate(hours):
            if h > 8:
                acc += 1
            else:
                acc -= 1
            if acc - 1 in s:
                ret = max(ret, i - s[acc - 1])
            if acc not in s:
                s[acc] = i
                if acc - 1 in s:
                    s[acc] = min(s[acc], s[acc - 1])
        return ret
