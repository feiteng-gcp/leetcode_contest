class Solution:
    def longestWPI(self, hours: List[int]) -> int:
        
        d_first = {0: -1}
        d_last = {0: -1}
        cur = 0
        for i, hour in enumerate(hours):
            
            if hour > 8:
                cur += 1
            else:
                cur -= 1
                
            if cur not in d_first:
                d_first[cur] = i
            d_last[cur] = i
            
        prev_far = float('inf')
        res = 0
        for cur in sorted(d_last):
            last = d_last[cur]
            if prev_far != float('inf'):
                res = max(res, last - prev_far)
            prev_far = min(d_first[cur], prev_far)
        return res
