class Solution:

    def longestWPI(self, hours):
        if not hours:
            return 0
        A = [(i>8)*1 for i in hours]
        prev = {}
        curSum = 0
        ans = 0

        for i in range(len(A)):
            if A[i]:
                curSum += 1
            else:
                curSum -= 1
            
if curSum > 0:
                ans = i + 1
            else:
                if curSum - 1 in prev:
                    ans = max(ans, i - prev[curSum-1])
            if curSum not in prev:
                prev[curSum] = i

        return ans
