class Solution:
    def longestWPI(self, hours: List[int]) -> int:
        m_hours=[i>8 for i in hours]
        cumsum=[0]
        for i in m_hours:
            cumsum.append(cumsum[-1]+(2*int(i)-1))
        dic={}
        for i in range(len(cumsum)):
            dic[cumsum[i]]=i
        Max=0
        for i in range(len(cumsum)):
            if cumsum[-1]>cumsum[i]:
                Max=max(Max,len(cumsum)-1-i)
            elif cumsum[i]+1 in dic:
                j=dic[cumsum[i]+1]
                Max=max(Max,j-i)
        return Max
