class Solution:
    def longestWPI(self, hours: List[int]) -> int:
        hours = [-1 if h <= 8 else 1 for h in hours]
        #print(hours)
        prefix = [0]
        for h in hours:
            prefix.append(h + prefix[
-1])
        appear = {}
        #print(prefix)
        
        for i in range(len(prefix)-1, -1, -1):
            appear[prefix[i]] = i
        #print(appear)
        res = 0
        for i, v in enumerate(prefix):
            if v > 0: 
                res = i
            else:
                res = max(res, i - appear.get(v-1, i))
        return res
