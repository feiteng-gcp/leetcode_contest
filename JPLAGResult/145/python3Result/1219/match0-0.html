class Solution:
    def longestWPI(self, hours: List[int]) -> int:
        daylist = [1 if i > 8 else -1 for i in hours]
        for i in range(1, len(daylist)): daylist[i] = daylist[i - 1] + daylist[i]
        maxday = 0
        daymap = {}
        for i, v in enumerate(daylist):
            if v > 0: maxday = i + 1
            else:
                if v - 1 in daymap:
                    maxday = max(maxday, i - daymap[v - 1])
            if v not in daymap: daymap[v] = i
        return maxday
