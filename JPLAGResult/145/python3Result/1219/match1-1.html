class Solution:
    def longestWPI(self, hours: List[int]) -> int:
        
        length = 0
        tiring = 0
        memo 
= {0: -1}
        last = {0:-1}
        start = 0
        
        for i, h in enumerate(hours):
            if h > 8: tiring += 1
            else: tiring -= 1
            if tiring not in memo: memo[tiring] = i
            last[tiring] = i
        
        for tir in memo:
            if tir-1 in last:
                length = max(length,last[tir]-memo[tir-1])
        
        return len(hours) if tiring > 0 else length
