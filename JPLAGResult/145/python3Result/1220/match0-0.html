class Solution:
    def smallestSufficientTeam(self, req_skills: List[str], people: List[List[str]]) -> List[int]:
        whocan = collections.defaultdict(list)
        cando = collections.defaultdict(list)
        
        for i, skill in enumerate(req_skills):
            for j, worker in enumerate(people):
                if skill in worker:
                    whocan[i].append(j)
                    cando[j].append(i)
        
        ans = [0] * (len(people) + 1)
        
        def dfs(todo, path):
            nonlocal ans
            if len(todo) == 0:
                if len(path) < len(ans):
                    ans = path
                return
            
            for w in whocan[todo[0]]:
                nxt = []
                for job in todo:
                    if job not in cando[w]:
                        nxt.append(job)
                nxt.sort(key = lambda x: len(whocan[x]))
                dfs(nxt, path + [w])
        
        todo = sorted(whocan.keys(), key = lambda x: len(whocan[x]))
        
        dfs(todo, [])
        return ans
