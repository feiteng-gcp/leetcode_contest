class Solution:
    def smallestSufficientTeam(self, req_skills: List[str], people: List[List[str]]) -> List[int]:
        mp = {k:2 ** i for i, k in enumerate(req_skills)}
        ss = {2 ** i: [] for i in range(len(req_skills))}
        for i, p in enumerate(people):
            for s in p:
                ss[mp[s]].append(i)
        def helper(state, dp):
            if not state:
                return 0, []
            if state in dp:
                return dp[state]
            ret = 10000
            rpp = []
            for p in ss[state & (-state)]:
                s2 = state
                for sk in people[p]:
                    s2 -= s2 & mp[sk]
                rr, pp = helper(s2, dp)
                if rr < ret - 1:
                    ret = rr + 1
                    rpp = pp + [p]
            dp[state] = [ret, rpp]
            return ret, rpp
        dp = {}
        return helper(2 ** len(req_skills) - 1, dp)[1]
