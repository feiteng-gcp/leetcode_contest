# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        self.ancestor = root
        def dfs(node, d):
            if not node:
                return (None, -1)
            d += 1
            if not node.left and not node.right:
                return (node, d)
            l = dfs(node.left, d)
            r = dfs(node.right, d)
            
if l[1] == r[1]:
                return (node, l[1])
            elif l[1] > r[1]:
                return l
            else:
                return r
        return dfs(root, 0)[0]
