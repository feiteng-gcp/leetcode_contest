# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        def function(root,deep=0):
            if root.left is None and root.right is None:
                return [root,deep]
            elif root.left is not None and root.right is not None:
                tmp1=function(root.left,deep+1)
                tmp2=function(root.right,deep+1)
                if tmp1[1]==tmp2[1]:
                    return [root,tmp1[1]]
                elif tmp1[1]>tmp2[1]:
                    return [tmp1[0],tmp1[1]]
                else:
                    return [tmp2[0],tmp2[1]]
            elif root.left is not None:
                return function(root.left,deep+1)
            else:
                return function(root.right,deep+1)
        return function(root)[0]
