# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        level = [root]
        num = 0
        temp = []
        while level:
            temp = level[:]
            level = []
            for node in temp:
                if node.left:
                    level.append(node.left)
                if node.right:
                    level.append(node.right)
            num += 1
        n = len(temp)
        if n == 1:
            return temp[0]
        a = self.lowestCommonAncestor(root, temp[0], temp[1])
        for i in range(2,n):
            a = self.lowestCommonAncestor(root, a, temp[i])
        return a

        
    def lowestCommonAncestor(self, root, A, B):
        if root is None:
            return None
        
        
if root == A or root == B:
            return root
        
        left_result = self.lowestCommonAncestor(root.left, A, B)
        right_result = self.lowestCommonAncestor(root.right, A, B)
        
        if left_result and right_result: 
            return root
        if left_result:
            return left_result
        if right_result:
            return right_result

        return None
