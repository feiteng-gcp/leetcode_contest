# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        def solve(node):
            if node is None:
                return node, 0
            lc, lh = solve(node.left)
            rc, rh = solve(node.right)
            if lh == rh:
                return node, lh + 1
            else:
                n = lc if lh > rh else rc
                return n, max(lh, rh) + 1
        n, h = solve(root)
        return n
