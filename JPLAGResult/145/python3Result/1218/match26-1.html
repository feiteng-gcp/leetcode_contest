# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        def findDeep(root):
            if not root:
                return 0
            left = findDeep(root.left)
            right = findDeep(root.right)
            return max(left,right)+1
        left = findDeep(root.left)
        right = findDeep(root.right)
        if left==right:
            return root
        if left>right:
            return self.lcaDeepestLeaves(root.left)
        else:
            return self.lcaDeepestLeaves(root.right)
