# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
                
        def lca(node):
            if not node: return (0, node)
            
            ldepth, llca = lca(node.left)
            rdepth, rlca = lca(node.right)
        
            if ldepth > rdepth:
                return ldepth + 1, llca
            elif ldepth < rdepth:
                return rdepth + 1, rlca
            else:
                return ldepth + 1, node
        
        _, ans = lca(root)
        
        return ans
