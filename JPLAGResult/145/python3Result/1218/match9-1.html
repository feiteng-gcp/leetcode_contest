# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        if root == None:
            return None
        par = {}
        q = [root]
        leave = set()
        leave2 = set()
        last = set()
        while q:
            curL = len(q)
            last = set()
            for _ in range(curL):
                cur = q.pop(0)
                last.add(cur)
                
if cur.left != None:
                    par[cur.left] = cur
                    q.append(cur.left)
                if cur.right != None:
                    par[cur.right] = cur
                    q.append(cur.right)
        
        
                
        while len(last) > 1:
            leave2 = set()
            for l in last:
                leave2.add(par[l])
            last = leave2
        for i in last:
            return i
