class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        def dfs(node):
            if not node:
                return None, -1
            n1, d1 = dfs(node.left)
            n2, d2 = dfs(node.right)
            d = max(d1, d2) + 1
            n = node if d1 == d2 else n1 if d1 > d2 else n2
            return n, d

        ans, _ = dfs(root)
        return ans
