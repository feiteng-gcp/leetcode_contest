# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        def recur(node):
            if not node: return 0,None
            dL,L = recur(node.left)
            dR,R = recur(node.right)
            if dL==dR:
                return dL+1, node
            elif dL>dR:
                return dL+1, L
            else:
                return dR+1, R
        d, N = recur(root)
        return N
