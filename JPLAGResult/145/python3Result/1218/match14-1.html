# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        def dep(node):
            if node is None:
                return 0
            dl = dep(node.left)
            dr = dep(node.right)
            return max(dl, dr) + 1

        def find(node, d, r, cd=1):
            if node is not None:
                if d == cd:
                    return node
                else:
                    l = find(node.left, d, r, cd + 1)
                    r = find(node.right, d, r, cd + 1)
                    if l and r:
                        return node
                    elif l or r:
                        return l or r
                    else:
                        return None

        res = []
        return find(root, dep(root), res)
