class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        def help(root, deep):
            # print(root.val, deep)
            if root.left == None:
                l = [root, deep]
            else:
                l = help(root.left, deep + 1)

            if root.right == None:
                r = [root, deep]
            else:
                r = help(root.right, deep + 1)

            if l[1] == r[1]:
                return [root, l[1]]
            elif l[1] > r[1]:
                return l
            else:
                return r
        return help(root, 0)[0]
