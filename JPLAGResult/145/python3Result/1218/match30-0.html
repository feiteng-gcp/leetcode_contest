# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lcaDeepestLeaves(self, root: TreeNode) -> TreeNode:
        dep, node = self.helper(root)
        return node
        
        
    def helper(self, node):
        if not node: return 0, None
        ldep, lnode = self.helper(node.left)
        rdep, rnode = self.helper(node.right)
        if ldep == rdep:
            return ldep + 1, node
        return max(ldep, rdep) + 1, lnode if ldep > rdep else rnode
