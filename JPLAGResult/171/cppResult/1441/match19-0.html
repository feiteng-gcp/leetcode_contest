class Solution {
public:
    int minFlips(int a, int b, int c) {
        int ret = 0;
        for (int k = 0; k < 30; ++k) {
            int u = (a >> k) & 1;
            int v = (b >> k) & 1;
            int w = (c >> k) & 1;
            if (w == 0) {
                ret += u + v;
            } else {
                if (u == 0 && v == 0) ret += 1;
            }
        }
        return ret;
    }
};
