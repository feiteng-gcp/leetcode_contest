class Solution {
public:
    int minFlips(int a, int b, int c) {
        int res = (a | b);
        int ans = 0;
        for (int digit = 0; digit < 31; digit++) {
            int res_value = (res & (1 << digit));
            int c_value = (c & (1 << digit));
            if (res_value != c_value) {
                if (c_value > 0) {
                    ans++;
                } else {
                    if (a & (1 << digit)) { ans++; }
                    if (b & (1 << digit)) { ans++; }
                }
            }
        }
        return ans;
    }
};
