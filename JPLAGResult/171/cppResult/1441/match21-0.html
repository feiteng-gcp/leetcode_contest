class Solution {
public:
    int minFlips(int a, int b, int c) {
        int res = 0;
        
        int n = 30;
        
        for (int i = n; i >= 0; --i) {
            int x = ((a | b) >> i) & 0x1;
            int y = (c >> i) & 0x1;
            if (x != y) {
                if (x == 0 && y == 1) {
                    ++res;
                } else {
                    if ((a >> i) & 0x1) {
                        ++res;
                    }
                    if ((b >> i) & 0x1) {
                        ++res;
                    }
                }
            }
        }
        
        return res;
    }
};
