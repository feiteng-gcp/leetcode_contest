class Solution {
public:
    int minFlips(int a, int b, int c) {
        int ans = 0;
        for (int i = 0; i < 32; ++i) {
            int bit = (int)(1u << i);
            int ai = a & bit;
            int bi = b & bit;
            int ci = c & bit;

            if (ci != 0) {
                if (ai == 0 && bi == 0) {
                    ans++;
                }
            } else {
                if (ai != 0) {
                    ans++;                    
                }
                if (bi != 0) {
                    ans++;
                }
            }
        }
        return ans;
    }
};
