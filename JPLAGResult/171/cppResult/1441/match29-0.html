class Solution {
public:
	int minFlips(int a, int b, int c) {
		int ct = 0;
		for (int i = 0; i < 32; i++) {
			int x = a & (1 << i), y = b & (1 << i), z = c & (1 << i);
			if (z > 0) {
				if (x > 0 || y > 0)continue;
				ct++;
			}
			else {
				if (x != 0)ct++;
				if (y != 0)ct++;
			}
		}
		return ct;
	}
};
