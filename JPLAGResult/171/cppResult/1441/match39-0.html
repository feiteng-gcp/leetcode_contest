class Solution {
public:
    int minFlips(int a, int b, int c) {
        int ret=0;
        while(a>0||b>0||c>0){
            if(c%2==0){
                if(a%2==1) ret++;
                if(b%2==1) ret++;
            }
            else{
                if(a%2==0&&b%2==0) ret++;
            }
            c/=2;
            a/=2;
            b/=2;
        }
        return ret;
    }
};
