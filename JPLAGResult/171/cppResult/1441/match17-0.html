class Solution {
public:
    int minFlips(int a, int b, int c) {
        int ret = 0;
        for (int i = 0; i < 30; ++ i)
        {
            int x = a>>i&1, y = b>>i&1;
            int z = c>>i&1;
            if ((x|y) != z)
            {
                if (z == 0)
                    ret += x+y;
                else
                    ret += 1;
            }
        }
        return ret;
    }
};
