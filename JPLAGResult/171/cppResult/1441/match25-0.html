class Solution {
public:
    int minFlips(int a, int b, int c) {
        int ans = 0;
        for (int i = 0; i < 30; i++) {
            int x = (((1 << i)&a) > 0);
            int y = (((1 << i)&b) > 0);
            int z = (((1 << i)&c) > 0);
            if ((x | y) == z) continue;
            if (z == 1) ans++;
            else ans += x + y;
        }
        return ans;
    }
};
