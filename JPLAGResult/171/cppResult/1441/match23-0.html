class Solution {
public:
    int minFlips(int a, int b, int c) 
    {
        vector<int>A = convert(a);
        vector<int>B = convert(b);
        vector<int>C = convert(c);
        int len = max(A.size(), max(B.size(), C.size()));
        while (A.size()<len)
            A.push_back(0);
        while (B.size()<len)
            B.push_back(0);
        while (C.size()<len)
            C.push_back(0);
        
 
        
        int cost = 0;
        for (int i=0; i<len; i++)
        {
            // cout<<A[i]<<" "<<B[i]<<" "<<C[i]<<" "<<A[i]||B[i]==C[i]<<endl;
            if ((A[i]||B[i])==C[i])
                continue;
            if (C[i]==0)
            {
                cost+= (A[i]!=0);
                cost+= (B[i]!=0);
            }
            else
            {
                cost+=1;
            }        
            
        }
        return cost;        
    }
    
    vector<int> convert(int x)
    {
        vector<int>ret;
        while (x>0)
        {
            ret.push_back(x%2);
            x = x/2;
        }
        return ret;
            
    }
};
