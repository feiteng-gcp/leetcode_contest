class Solution {
public:
    int minFlips(int a, int b, int c) {
        int ans = 0;
        
        while (a || b || c) {
            int ba = a & 1;
            int bb = b & 1;
            int bc = c & 1;
            if ((ba | bb) != bc) {
                if (bc == 0)
                    ans += (ba == 1) + (bb == 1);
                else
                    ans += 1;
            }
            a >>= 1;
            b >>= 1;
            c >>= 1;
        }
        return ans;
    }
};
