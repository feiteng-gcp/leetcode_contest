class Solution {
public:
    int minFlips(int a, int b, int c) {
        int res = 0;
        while (a || b || c) {
            if (1 & c) {
                if (!(1 & a) && !(1 & b))
                    ++res;
            }
            else {
                if (1 & a)
                    ++res;
                if (1 & b)
                    ++res;
            }
            a >>= 1;
            b >>= 1;
            c >>= 1;
        }
        return res;
    }
};
