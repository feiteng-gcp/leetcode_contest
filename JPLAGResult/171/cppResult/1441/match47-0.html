class Solution {
public:
    int minFlips(int a, int b, int c) {
        int ret=0;
        
        for (int i=0;i<31;i++)
        {
            int aim=(c>>i)&1;
            int t1,t2;
            if (((a>>i)&1)==1)   t1=1;
            else t1=0;
            if (((b>>i)&1)==1) t2=1;
            else t2=0;
            if (aim==1)
            {
                if (t1==0 && t2==0) ret++;
            }
            else
            {
                if (t1==1) ret++;
                if (t2==1) ret++;
            }
        }
        return ret;
        
    }
};
