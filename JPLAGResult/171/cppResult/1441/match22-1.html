class Solution {
public:
    int minFlips(int a, int b, int c) {
        int cnt = 0;
        while (a > 0 || b > 0 || c > 0) {
            int da = a > 0 ? a % 2 : 0;
            int db = b > 0 ? b % 2 : 0;
            int dc = c > 0 ? c % 2 : 0;
            
            if (dc == 1) {
                if (da == 0 && db == 0)    cnt++;
            }
            else if (dc == 0) {
                if (da == 1)    cnt++;
                if (db == 1)    cnt++;
            }
            
            a = a >> 1;    b = b >> 1;    c = c >> 1;
        }
        return cnt;
    }
};
