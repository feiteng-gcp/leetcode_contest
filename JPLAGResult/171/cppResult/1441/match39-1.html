class Solution {
public:
    int minFlips(int a, int b, int c) {
        int ans=0;
        while(!(a==0&&b==0&&c==0)){
            int ra=a%2;
            int rb=b%2;
            int rc=c%2;

            if(rc==0){
                if(ra!=0)ans++;
                if(rb!=0)ans++;
            }
            else{
                if(ra==0&&rb==0)ans++;
            }
            a/=2,b/=2,c/=2;
        }
        return ans;
    }
};
