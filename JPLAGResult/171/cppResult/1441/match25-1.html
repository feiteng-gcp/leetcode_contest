class Solution {
public:
    int minFlips(int a, int b, int c) {
        int cnt = 0;
        for (int i = 0; i <= 30; i++) {
            int mask = 1<<i;
            int x = a&mask;
            int y = b&mask;
            if ((x|y) == (c&mask)) continue;
            if (x != 0 && y != 0) cnt += 2;
            else cnt++;
        }
        return cnt;
        
    }
};
