class Solution {
public:
    int minFlips(int a, int b, int c) {
        int res = 0;
        while (a || b || c) {
            if (c % 2 == 0) { // 0 => 都要为0
                if (a % 2 == 1) ++res;
                if (b % 2 == 1) ++res;
            } else { // 1 => 随便一个为1即可
                if (a % 2 == 0 && b % 2 == 0) ++res;
            }

            a /= 2;
            b /= 2;
            c /= 2;
        }
        return res;
    }
};
