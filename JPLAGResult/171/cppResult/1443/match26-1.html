int dp[27][27][301];
class Solution {
public:
    int abs(int x)
    {
        return x>0?x:-x;
    }
    int dis(int x,int y)
    {
        if(x==26||y==26)return 0;
        int xc=x%6,xr=x/6,yc=y%6,yr=y/6;
        return abs(xr-yr)+abs(xc-yc);
    }
    int minimumDistance(string word) {

        memset(dp,1,sizeof(dp));
        dp[26][26][0]=0;
        int n=word.size();
        for(int i=1;i<=n;i++)
        {
            int tg=word[i-1]-'A';
            for(int j=0;j<=26;j++)
            {
                for(int k=0;k<=26;k++)
                {
                    dp[tg][k][i]=min(dp[tg][k][i],dp[j][k][i-1]+dis(j,tg));

                    dp[j][tg][i]=min(dp[j][tg][i],dp[j][k][i-1]+dis(k,tg));
                }
            }
        }
        int ans=9999999999;
        for(int i=0;i<=26;i++)
        {
            for(int j=0;j<=26;j++)
            {
                ans=min(ans,dp[i][j][n]);
            }
        }
        return ans;
    }
};
