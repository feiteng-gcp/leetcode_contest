class Solution {
public:
    void init_keybroad() {
        pos.clear();
        int x = -1;
        int y = 0;
        for (int i = 0; i < 26; ++i) {
            if (i % 6 == 0) {
                ++x;
                y = 0;
            }
            pos.push_back(pair<int, int>(x, y));
            ++y;
        }
    }
    int distance(int a, int b) {
        //printf("%d %d %d %d %d %d\n", a, b, pos[a].first, pos[a].second, pos[b].first, pos[b].second);
        return abs(pos[a].first - pos[b].first) + abs(pos[a].second - pos[b].second);
    }
    int minimumDistance(string word) {
        init_keybroad();
        int len = word.length();
        for (int i = 0; i <= len; ++i) {
            for (int p = 0; p < 26; ++p) {
                for (int q = 0; q < 26; ++q) {
                    if (i == 0) dp[i][p][q] = 0;
                    else dp[i][p][q] = (1 << 29);
                }
            }
        }
        for (int i = 0; i < len; ++i) {
            for (int p = 0; p < 26; ++p) {
                for (int q = 0; q < 26; ++q) {
                    int foo = (word[i] - 'A');
                    dp[i + 1][word[i] - 'A'][q] = min(dp[i + 1][word[i] - 'A'][q], dp[i][p][q] + distance(word[i] - 'A', p));
                    dp[i + 1][p][word[i] - 'A'] = min(dp[i + 1][p][word[i] - 'A'], dp[i][p][q] + distance(word[i] - 'A', q));
                }
            }
        }
        int ans = (1 << 29);
        for (int p = 0; p < 26; ++p) {
            for (int q = 0; q < 26; ++q) {
                ans = min(ans, dp[len][p][q]);
            }
        }
        return ans;
    }
    vector<pair<int, int> > pos;
    int dp[305][27][27];
};
