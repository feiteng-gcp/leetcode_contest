struct cmp {
    bool operator()(vector<int>& a, vector<int>& b) {
        return a[2] > b[2];
    }
};

class Solution {
public:
    int minimumDistance(string word) {
        int n = word.length();
        
        vector<int> arr;
        for (char c : word) {
            arr.push_back(c - 'A');
        }
        int na = arr.size();
        vector<vector<vector<int>>> memo(na, vector<vector<int>>(27, vector<int>(27, 0)));
        return dfs(arr, 1, arr[0], 26, memo);
    }
    
private:
    int dfs(vector<int>& arr, int s, int pa, int pb, vector<vector<vector<int>>>& memo) {
        // cout<<s<<" "<<pa<<" "<<pb<<endl;
        if (s == arr.size())    return 0;
        
        if (memo[s][pa][pb] != 0)    return memo[s][pa][pb];
        int dis = INT_MAX;
        if (pb == 26)    dis = min(dis, dfs(arr, s + 1, pa, arr[s], memo));
        else    dis = min(dis, dfs(arr, s + 1, pa, arr[s], memo) + getDis(pb, arr[s]));
        dis = min(dis, dfs(arr, s + 1, arr[s], pb, memo) + getDis(pa, arr[s]));
        
        // cout<<s<<" "<<dis<<endl;
        memo[s][pa][pb] = dis;
        return dis;
    }
    
    vector<int> getCor(int i) {
        return {i / 6, i % 6};
    }
    
    int getDis(int A, int B) {
        vector<int> a = getCor(A), b = getCor(B);
        int x1 = a[0], x2 = b[0], y1 = a[1], y2 = b[1];
        return abs(x1 - x2) + abs(y1 - y2);
    }
};
