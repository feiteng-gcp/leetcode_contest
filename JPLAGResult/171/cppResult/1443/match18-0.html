template<typename S, typename T>
pair<S, T> operator+(pair<S, T> a, pair<S, T> b) {
    return make_pair(a.first + b.first, a.second + b.second);
}

template<typename S, typename T>
pair<S, T> operator-(pair<S, T> a, pair<S, T> b) {
    return make_pair(a.first - b.first, a.second - b.second);
}
class Solution {
public:
    pair<int,int> query_pos(int a) {
        return {a / 6, a % 6};
    }
    int query_distance(int a, int b) {
        pair<int,int> p = query_pos(a) - query_pos(b);
        return abs(p.first) + abs(p.second);
    }
    int dp[305][26][26];
    int minimumDistance(string word) {
        int n = word.size();
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j < 26; j++) {
                for (int k = 0; k < 26; k++) {
                    if (i == 0)
                        dp[i][j][k] = 0;
                    else
                        dp[i][j][k] = 1e9;
                }
            }
        }
        for (int i = 1; i <= n; i++) {
            int a = word[i-1] - 'A';
            for (int j = 0; j < 26; j++) {
                for (int k = 0; k < 26; k++) {
                    dp[i][a][k] = min(dp[i][a][k], dp[i-1][j][k] + query_distance(j, a));
                    dp[i][j][a] = min(dp[i][j][a], dp[i-1][j][k] + query_distance(k, a));
                }
            }
        }
        /*
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < 26; j++) {
                for (int k = 0; k < 26; k++) {
                    if (dp[i][j][k] != 1e9) {
                        cout << "dp[" << i << "][" << j << "][" << "][" << k << "]: " << dp[i][j][k] << endl;
                    }
                }
            }
        }
        */
        int min_val = 1e9;
        for (int i = 0; i < 26; i++) {
            for (int j = 0; j < 26; j++) {
                min_val = min(min_val, dp[n][i][j]);
            }   
        }
        return min_val;
    }
};
