class Solution {
public:
    int minimumDistance(string word) {
        int n = 5, m = 6;
        
        // const int N = 6;
        const int INF = 1 << 20;

        const int A = 26;
        vector<vector<int>> dp(A, vector<int>(A, INF));
        for (int i = 0; i < A; ++i) {
            for (int j = 0; j < A; ++j) {
                if (i != j) {
                    dp[i][j] = 0;
                }
            }
        }
        auto ndp = dp;
        
        vector<pair<int,int>> pos(128);
        for (int i = 0; i < A; ++i) {
            int p = i;
            pos[i] = {p / m, p % m};
        };
        
        // auto distance = [&](int x1, int y1, int x2, int y2) {
        //     return abs(x1 - x2) + abs(y1 - y2);
        // };
        
        auto distance = [&](int a, int b) {
            int x1 = pos[a].first, y1 = pos[a].second;
            int x2 = pos[b].first, y2 = pos[b].second;
            return abs(x1 - x2) + abs(y1 - y2);
        };
        
        for (auto c : word) {
            for (auto& V : ndp) {
                fill(V.begin(), V.end(), INF);
            }
            

            for (int i = 0; i < A; ++i) {
                for (int j = 0; j < A; ++j) {
                    int k = c - 'A';
                    

                    ndp[k][j] = min(ndp[k][j], dp[i][j] + distance(i, k));
                    ndp[i][k] = min(ndp[i][k], dp[i][j] + distance(j, k));
                }
            }
            
            swap(ndp, dp);
        }
        
        int res = INF;
        for (int i = 0; i < A; ++i) {
            for (int j = 0; j < A; ++j) {
                res = min(res, dp[i][j]);
            }
        }
        
        return res;
    }
};
