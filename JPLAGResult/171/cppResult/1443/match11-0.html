class Solution {
public:
    int minimumDistance(string word) {
        std::vector<std::vector<std::vector<int>>> steps(word.length() + 1, std::vector<std::vector<int>>(26, std::vector<int>(26, 1e9 + 7)));
        for (int i = 0; i < 26; i++) {
            for (int j = 0; j < 26; j++) {
                steps[0][i][j] = 0;
            }
        }
        for (int i = 0; i < word.length(); i++) {
            int next_value = word[i] - 'A';
            for (int x = 0; x < 26; x++) {
                for (int y = 0; y < 26; y++) {
                    steps[i + 1][next_value][y] = std::min(steps[i + 1][next_value][y], steps[i][x][y] + dist(x, next_value));
                    steps[i + 1][x][next_value] = std::min(steps[i + 1][x][next_value], steps[i][x][y] + dist(y, next_value));
                }
            }
        }
        int ans = 1e9 + 7;
        for (int i = 0; i < 26; i++) {
            for (int j = 0; j < 26; j++) {
                ans = std::min(ans, steps[word.length()][i][j]);
            }
        }
        return ans;
    }
private:
    int dist(int x, int y) {
        int x1 = x / 6, x2 = x % 6;
        int y1 = y / 6, y2 = y % 6;
        return std::abs(x1 - y1) + std::abs(x2 - y2);
    }
};
