class Solution {
public:
    inline int dist(int a, int b)
    {
        //cout<<a<<' '<<b<<endl;
        int x=abs(a/6-b/6);
        int y=abs(a%6-b%6);
        //cout<<x+y<<endl;
        return x+y;
    }
    int minimumDistance(string word) {
        
        if (word=="") return 0;
        
        int f[310][30][30];
        
        int first=word[0]-'A';
        for (int j=0;j<30;j++)
            for (int k=0;k<30;k++)
            {
                if (j==first||k==first) f[0][j][k]=0;
                else f[0][j][k]=10000000;
            }
        
        for (int i=1;i<=word.length();i++)
        {
            int c = word[i-1]-'A';
            //cout<<c<<endl;

            for (int j=0;j<26;j++)
            {
                for (int k=0;k<26;k++)
                {
                    /*
                    if (j!=c && k!=c)
                    {
                        f[i][j][k]=10000000;
                        continue;
                    }
                    */
                    int min=0x7fffffff;
                    for (int m=0;m<26;m++)
                    {
                        
                        int d1=dist(c,j);
                        int d2=dist(c,k);
                        int d3=dist(m,j);
                        int d4=dist(m,k);
                        if (f[i-1][c][m]+d1+d4<min) min=f[i-1][c][m]+d1+d4;
                        if (f[i-1][m][c]+d2+d3<min) min=f[i-1][m][c]+d2+d3;
                        //if (c==10) cout<<m<<' '<<f[i-1][c][m]<<' '<<d1<<' '<<f[i-1][m][c]<<' '<<d2<<endl;
                    }
                    f[i][j][k]=min;
                }
            }
            
        }
        int last=word[word.length()-1]-'A';
        int ret=100000000;
        for (int i=0;i<26;i++)
        {
            for (int j=0;j<26;j++)
                if (f[word.length()][i][j]<ret) ret=f[word.length()][i][j];
        }
        return ret;
    }
};
