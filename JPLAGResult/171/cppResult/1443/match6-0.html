class Solution {
public:
    
    
    int minimumDistance(string word) {
        unordered_map <int, pair<int,int>> grid;
        int x = 0,  y = 0;
        for(int i = 0; i <= 25; ++i){
            grid[i] = {x,y};
            if(y == 5){
                y = 0;
                x++;
            } else
                y++;
        }
        
        
        int res = INT_MAX;
        int dp[27][27][word.size() + 1];
        memset(dp, 0, sizeof(dp));
        for(int z = word.size() - 1; z >= 0; --z){
        for(int x = 0; x < 26; ++x){
            for(int y = 0; y < 26; ++y){
                    int zz = word[z] - 'A';
                    dp[x][y][z] = min(dp[zz][y][z + 1] + cost(x,zz),
                                      dp[x][zz][z + 1] + cost(y,zz));
                    if(z == 0)
                        res = min(res,dp[x][y][z]);
                }
            }
        }
        return res;
    }
    
    int cost(int x, int y){
        return abs(x/6-y/6)+ abs(x%6-y%6);
    }
    
    
};
