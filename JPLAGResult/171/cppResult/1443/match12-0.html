int inf = 1e9;

class Solution {
public:
    int dis(int s, int t) {
        if(s == 26) return 0;
        int x0 = s / 6, y0 = s % 6;
        int x1 = t / 6, y1 = t % 6;
        return abs(x0 - x1) + abs(y0 - y1);
    }
    int minimumDistance(string word) {
        vector<vector<int>> a = vector<vector<int>>(27, vector<int>(27, inf));
        a[26][26] = 0;
        for(auto ch: word) {
            vector<vector<int>> b = vector<vector<int>>(27, vector<int>(27, inf));

            int w = ch - 'A';
            for(int i = 0; i < 27; i++) {
                for(int j = 0; j < 27; j++) {
                    b[i][w] = min(b[i][w], a[i][j] + dis(j, w));
                    b[w][j] = min(b[w][j], a[i][j] + dis(i, w));
                }
            }
            a = b;
        }
        int ans = inf;
        for(int i = 0; i < 27; i++) {
            for(int j = 0; j < 27; j++) {
                ans = min(ans, a[i][j]);
            }
        }

        return ans;
    }
};
