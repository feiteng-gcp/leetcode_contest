class Solution {
public:
    
    int dp[301][26][26];
    
    int minimumDistance(string word) {
        int x[26], y[26];
        for(int i = 0; i < 26; i ++){
            x[i] = i / 6;
            y[i] = i % 6;
        }
        int n = word.size();
        for(int i = 0; i <= n; i ++){
            for(int j = 0; j < 26; j ++){
                for(int k = 0; k < 26; k ++){
                    dp[i][j][k] = 1e9;
                }
            }
        }
        for(int i = 0; i < 26; i ++){
            for(int j = 0; j < 26; j ++){
                dp[0][i][j] = 0;
            }
        }
        
        for(int i = 0; i < n; i ++){
            int c = word[i] - 'A';
            for(int j = 0; j < 26; j ++){
                for(int k = 0; k < 26; k ++){
                    dp[i+1][c][k] = min(dp[i+1][c][k], dp[i][j][k] + std::abs(x[c] - x[j]) + std::abs(y[c] - y[j]));
                    dp[i+1][j][c] = min(dp[i+1][j][c], dp[i][j][k] + std::abs(x[c] - x[k]) + std::abs(y[c] - y[k]));
                }
            }
        }
        
        int res = 1e9;
        for(int i = 0; i < 26; i ++){
            for(int j = 0; j < 26; j ++){
                res = min(res, dp[n][i][j]); 
            }
        }
        return res;
    }
};
