const int N = 27;
const int M = 333;
int dis[N][N];
int dp[M][N][N];


void init() {
    
    for(int i = 0; i < N; i++) {
        for(int j = 0; j < N; j++) {
            dis[i][j] = 0;
        }
    }

    for(int i = 0; i < N-1; i++) {
        for(int j = 0; j < N-1; j++) {
            int a,b,x,y;
            a = i/6;
            b = i%6;
            x = j/6;
            y = j%6;
            dis[i][j] = abs(a-x) + abs(b-y);
        }
    }
    for(int i = 0; i < M; i++) {
        for(int j = 0; j < N; j++) {
            for(int k = 0; k < N; k++) {
                dp[i][j][k] = 1e9;
            }
        }
    }
    dp[0][26][26] = 0;
}

class Solution {
public:
    int minimumDistance(string word) {
        init();
        int n = word.length();
        for(int i = 0; i < n; i++) {
            int z = word[i] - 'A';

            for(int x = 0; x < N; x++) {
                for(int y = 0; y < N; y++) {
                    dp[i+1][x][z] = min(dp[i+1][x][z],dp[i][x][y]+dis[y][z]);
                    dp[i+1][z][y] = min(dp[i+1][z][y],dp[i][x][y]+dis[x][z]);
                }
            }
        }
        int res = 1e9;
        for(int x = 0; x < N; x++) {
            for(int y = 0; y < N; y++) {
                res = min(res,dp[n][x][y]);
            }
        }
        return res;
    }
};
