class Solution {
public:
	

	int dis(int a, int b){
		int ax = a / 6;
		int ay = a % 6;
		
		int bx = b / 6;
		int by = b % 6;
		
		return abs(ax - bx) + abs(ay - by);
	}
	
    int minimumDistance(string word) {
		const int inf = (1<<30);
		int res = inf;

		vector<vector<vector<int>>> dp(word.size()+1, vector<vector<int>>(27, vector<int>(27, inf)));
		
		dp[0][26][26] = 0;
		
		for(int i = 0; i < word.size(); i++){
			for(int j = 0; j < 27; j++){
				for(int k = 0; k < 27; k++){
					if(j == 26) {
						dp[i+1][word[i]-'A'][k] = min(dp[i+1][word[i]-'A'][k], dp[i][j][k]);
					} else {
						
						dp[i+1][word[i]-'A'][k] = min(dp[i+1][word[i]-'A'][k], dp[i][j][k] + dis(j, word[i]-'A'));
					}
					
					if(k == 26) {
						dp[i+1][j][word[i]-'A'] = min(dp[i+1][j][word[i]-'A'], dp[i][j][k]);
					} else {
						
						dp[i+1][j][word[i]-'A'] = min(dp[i+1][j][word[i]-'A'], dp[i][j][k] + dis(k, word[i]-'A'));
					}
				}
			}
		}
		

		for(int j = 0; j < 27; j++){
			for(int k = 0; k < 27; k++){
				res = min(dp[word.size()][j][k], res);	
			}
		}
		
		return res;
		
    }
};
