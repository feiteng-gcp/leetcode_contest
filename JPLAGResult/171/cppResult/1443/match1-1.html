class Solution {
public:
    int minimumDistance(string word) 
    {
        int n = word.size();
        auto dp = vector<vector<vector<int>>>(n,vector<vector<int>>(27,vector<int>(27,INT_MAX/2)));
        
        dp[0][word[0]-'A'][26] = 0;
        dp[0][26][word[0]-'A'] = 0;
        
        for (int k=1; k<n; k++)
        {
            int cur = word[k]-'A';
            
            for (int j=0; j<=26; j++)
            {
                for (int i=0; i<=26; i++)
                {
                    dp[k][cur][j] = min(dp[k][cur][j], dp[k-1][i][j] + dis(i,cur));
                    // if (k==1 && cur==0 && j==26)
                    // {                        
                    //     cout<<"test:"<<dp[1][0][26]<<" "<<dp[0][2][26]<<" "<<dis(2,0)<<endl;
                    // }
                }
                    
            }
            for (int i=0; i<=26; i++)
            {
                for (int j=0; j<=26; j++)
                    dp[k][i][cur] = min(dp[k][i][cur], dp[k-1][i][j] + dis(j,cur));
            }
        }
        
        // cout<<dp[1][0][26]<<" "<<dp[0][2][26]<<" "<<dis(2,0)<<endl;
        // cout<<dp[1][2][0]<<endl;
        
        int ret = INT_MAX;
        int cur = word[n-1]-'A';
        for (int i=0; i<=26; i++)
            ret = min(ret, dp[n-1][i][cur]);
        for (int j=0; j<=26; j++)
            ret = min(ret, dp[n-1][cur][j]);
        
        return ret;
    }
    
    int dis(int a, int b)
    {
        if (a==26) return 0;
        int x1 = a/6;
        int y1 = a%6;
        int x2 = b/6;
        int y2 = b%6;
        return abs(x1-x2)+abs(y1-y2);
    }
};
