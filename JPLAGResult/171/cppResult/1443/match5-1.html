class Solution {
public:
    int minimumDistance(string word) {
        int n = word.size();

        int res = INT_MAX;
        vector<vector<vector<int>>> dp(26, vector<vector<int>>(26, vector<int>(n + 1, 0)));
        for(int p = n - 1; p >= 0; --p){
            for(int i = 0; i < 26; ++i){
                for(int j = 0; j < 26; ++j){

                    dp[i][j][p] = min(
                        getCost(i, word[p] - 'A') + dp[word[p] - 'A'][j][p+1],
                        getCost(j, word[p] - 'A') + dp[i][word[p] - 'A'][p+1]
                    );
                    if(p == 0){
                        res = min(res, dp[i][j][0]);
                    }
                }
            }
        }
        return res;
    }
    
    int getCost(int i, int j){
        return abs(i/6 - j/6) + abs(i%6 - j%6);
    }
};
