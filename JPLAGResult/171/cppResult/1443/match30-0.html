class Solution {
public:
    int minimumDistance(string word) {
        vector<vector<int>> distance(27, vector<int>(27, 0)); // 26 : free
        for (int i = 0; i < 26; ++i) {
            for (int j = 0; j < 26; ++j) {
                int dx = abs(i / 6 - j / 6);
                int dy = abs(i % 6 - j % 6);
                distance[i][j] = dx + dy;
            }
        }
        vector<vector<int>> DP(27, vector<int>(27, 1E6)); // p1, p2
        DP[26][26] = 0;
        for (char c : word) {
            int ci = c - 'A';
            vector<vector<int>> next(27, vector<int>(27, 1E6));
            for (int i = 0; i <= 26; ++i) {
                for (int j = 0; j <= 26; ++j) {
                    next[i][ci] = min(next[i][ci], DP[i][j] + distance[j][ci]);
                    next[ci][j] = min(next[ci][j], DP[i][j] + distance[i][ci]);
                }
            }
            DP.swap(next);
        }
        int ans = 1E6;
        for (int i = 0; i <= 26; ++i) {
            for (int j = 0; j <= 26; ++j) {
                ans = min(ans, DP[i][j]);
            }
        }
        return ans;
    }
};
