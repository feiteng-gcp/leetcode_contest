class Solution {
public:
    int dp[310][26][26];
    pair<int,int> getpos(int x){
        return {x / 6,x % 6};
    }
    int minimumDistance(string word) {
        // memset(dp, 0, sizeof(dp));

        for(int i = 1;i < 310;i++){
            for(int j = 0;j < 26;j++){
                for(int k = 0;k < 26;k++){
                    dp[i][j][k] = 10e7;
                }
            }
        }
        for(int i = 0;i < word.size();i++){
            int now = word[i] - 'A';
            cout<<"now = "<<now<<endl;

            for(int p = 0;p < 26;p++){
                for(int q = 0;q < 26;q++){
                    //left is p
                    //right is q
                    
                    //use left:
                    int dist1 = abs(getpos(now).first - getpos(p).first) + abs(getpos(now).second - getpos(p).second);
                    dp[i + 1][now][q] = min(dp[i + 1][now][q],dp[i][p][q] + dist1);
                    
                    //use right:

                    int dist2 = abs(getpos(now).first - getpos(q).first) + abs(getpos(now).second - getpos(q).second);
                    dp[i + 1][p][now] = min(dp[i + 1][p][now],dp[i][p][q] + dist2);
                }
            }
        }
        int res = INT_MAX;
        for(int p = 0;p < 26;p++){
            for(int q = 0;q < 26;q++){
                res = min(res,dp[word.size()][p][q]);
            }
        }
        return res;
        
    }
};
