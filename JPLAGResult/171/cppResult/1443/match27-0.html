class Solution {
public:
    int x[105], y[105], dis[105][105];
    int f[305][55][55];
    
    int minimumDistance(string s) {
        int n = s.size();
        int curx = 0, cury = 0;
        for (int i = 0; i < 26; i++) {
            x[i] = curx, y[i] = cury;
            cury++;
            if (cury > 5) {
                cury = 0;
                curx++;
            }
        }
        for (int i = 0; i < 27; i++) {
            for (int j = 0; j < 27; j++) {
                if (i == 26 || j == 26) {
                    dis[i][j] = 0;
                } else {
                    dis[i][j] = abs(x[i] - x[j]) + abs(y[i] - y[j]);
                }
            }
        }
        
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j < 27; j++) {
                for (int k = 0; k < 27; k++) {
                    f[i][j][k] = 1e9 + 1;
                }
            }
        }
        f[0][26][26] = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < 27; j++) {
                for (int k = 0; k < 27; k++) {
                    int cur = f[i][j][k];
                    int c = s[i] - 'A';
                    f[i + 1][c][k] = min(f[i + 1][c][k], cur + dis[j][c]);
                    f[i + 1][j][c] = min(f[i + 1][j][c], cur + dis[k][c]);
                }
            }
        }
        int ans = f[n][0][0];
        for (int i = 0; i < 27; i++) {
            for (int j = 0; j < 27; j++) {
                ans = min(ans, f[n][i][j]);
            }
        }
        return ans;
    }
};
