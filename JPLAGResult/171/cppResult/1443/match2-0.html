class Solution {
public:
    int minimumDistance(string word) {

        vector<vector<vector<int>>> dp(27, vector<vector<int>>(27, vector<int>(word.size(), -1)));
        return dfs(word, 26, 26, 0, dp);
    }

private:
    int dfs(const string& word, int left, int right, int index,
            vector<vector<vector<int>>>& dp){

        if(index == word.size()) return 0;
        if(dp[left][right][index] != -1) return dp[left][right][index];

        int a = dis(left, word[index] - 'A') + dfs(word, word[index] - 'A', right, index + 1, dp);
        int b = dis(right, word[index] - 'A') + dfs(word, left, word[index] - 'A', index + 1, dp);
        return dp[left][right][index] = min(a, b);
    }

    int dis(int a, int b){
        if(a == 26) return 0;
        int x1 = a / 6, y1 = a % 6, x2 = b / 6, y2 = b % 6;
        return abs(x1 - x2) + abs(y1 - y2);
    }
};
