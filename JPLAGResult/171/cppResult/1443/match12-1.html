class Solution {
public:

    int dis(char a, char b) {
        if (b == -1) return 0;
        int xa = (a - 'A') / 6, ya = (a - 'A') % 6;
        int xb = (b - 'A') / 6, yb = (b - 'A') % 6;
        return abs(xa - xb) + abs(ya - yb);
    }
    
    int minimumDistance(string word) {
    //    word = "HAPPY";
    //    word = "NEW";
    //    word = "YEAR";
    //    word = string(300, 'A');
    //    srand(NULL);for (int i = 0; i < 300; ++i) {word[i] = (rand() % 26) + 'A';}printf("%s\n", word.c_str());
        
        vector<vector<vector<int>>> dp(word.size() + 1, vector<vector<int>>(27, vector<int>(27, INT_MAX)));

        dp[0][0][0] = 0;
        
        for (int i = 0; i < word.size(); ++i) {
            for (int j = 0; j < 27; ++j) {
                for (int k = 0; k < 27; ++k) {
                    if (dp[i][j][k] != INT_MAX) {
                        dp[i + 1][word[i] - 'A' + 1][k] = min(dp[i + 1][word[i] - 'A' + 1][k], dp[i][j][k] + dis(word[i], (j == 0)?-1:(j - 1 + 'A')));
                        dp[i + 1][j][word[i] - 'A' + 1] = min(dp[i + 1][j][word[i] - 'A' + 1], dp[i][j][k] + dis(word[i], (k == 0)?-1:(k - 1 + 'A')));
                    }
                }
            }

        }
        
        int minDis = INT_MAX;
        for (int i = 0; i < 27; ++i) {
            for (int j = 0; j < 27; ++j) {
                minDis = min(minDis, dp[word.size()][i][j]);
            }
        }
        return minDis;
    }
};
