class Solution {
public:
    int dp[26][26][301];
    int minimumDistance(string word) {
        memset(dp, 0x3f, sizeof(dp));

        int n = word.size();
        for(int i = 0; i < 26; i++){
            for(int j = 0; j < 26; j++){
                dp[i][j][0] = 0;
            }
        }
        auto dist = [](int i, int j, int k, int l)->int{
            return abs(k-i)+abs(j-l);
        };
        for(int k = 0; k < n; k++){
            int tind = word[k]-'A';

            int ti = tind/6, tj = tind%6;
            for(int i = 0; i < 26; i++){
                for(int j = 0; j < 26; j++){
                    dp[tind][j][k+1] = min(dp[tind][j][k+1], dp[i][j][k]+dist(i/6,i%6,ti,tj));
                    dp[i][tind][k+1] = min(dp[i][tind][k+1], dp[i][j][k]+dist(j/6,j%6,ti,tj));

                }
            }
        }
        int ret = 1<<30;
        for(int i = 0; i < 26; i++){
            for(int j = 0; j < 26; j++){
                ret = min(ret, dp[i][j][n]);
            }
        }
        return ret;
    }
};
