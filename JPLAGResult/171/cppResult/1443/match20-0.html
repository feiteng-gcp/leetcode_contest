class Solution {
public:
#define INF 100000000
	unordered_map<int, pair<int, int>>pos;
	int dist(int x, int y) {
		return abs(pos[x].first - pos[y].first) + abs(pos[x].second - pos[y].second);
	}
	int minimumDistance(string word) {
		int n = word.size();
		int curr = 0;
		for (int i = 0; i < 5; i++) {
			for (int j = 0; j < 6; j++) {
				pos[curr] = { i,j };
				curr++;
				if (curr == 26)break;
			}
		}
		vector<vector<vector<int>>>dp(n + 1, (vector<vector<int>>(27, vector<int>(27, INF))));
		dp[0][word[0] - 'A'][26] = 0;
		dp[0][26][word[0] - 'A'] = 0;
		for (int i = 1; i < n; i++) {
			for (int j = 0; j <= 26; j++) {
				for (int k = 0; k <= 26; k++) {
					if (dp[i - 1][j][k] == INF)continue;
					int p = word[i] - 'A';
					if (j == 26) {
						dp[i][p][k] = min(dp[i][p][k], dp[i - 1][j][k]);
						dp[i][j][p] = min(dp[i][j][p], dp[i - 1][j][k] + dist(k, p));
					}
					else if (k == 26) {
						dp[i][j][p] = min(dp[i][j][p], dp[i - 1][j][k]);
						dp[i][p][k] = min(dp[i][p][k], dp[i - 1][j][k] + dist(j, p));
					}
					else {
						dp[i][p][k] = min(dp[i][p][k], dp[i - 1][j][k] + dist(j, p));
						dp[i][j][p] = min(dp[i][j][p], dp[i - 1][j][k] + dist(k, p));
					}
					
				}
			}
		}
		int minn = INF;
		for (int i = 0; i <= 26; i++) {
			for (int j = 0; j <= 26; j++) {
				minn = min(minn, dp[n - 1][i][j]);
			}
		}
		return minn;
	}
};
