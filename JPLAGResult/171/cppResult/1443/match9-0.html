class Solution {
public:
    struct node{
        int x,y;
    }A[27];
    int dp[303][27][27];
    int cal(int x,int y){
        if(x==0)return 0;
        return abs(A[x-1].x-A[y-1].x)+abs(A[x-1].y-A[y-1].y);
    }
    int minimumDistance(string word) {
        for(int i=0;i<26;i++){
            A[i].x=i/6;
            A[i].y=i%6;
        }
        memset(dp,0x3f,sizeof(dp));
        dp[0][0][0]=0;
        int n=word.size();
        for(int i=1;i<=n;i++){
            int t=word[i-1]-'A'+1;
            for(int j=0;j<=26;j++){
                for(int k=0;k<=26;k++){
                    dp[i][t][k]=min(dp[i][t][k],dp[i-1][j][k]+cal(j,t));
                }
            }
            for(int j=0;j<=26;j++){
                for(int k=0;k<=26;k++){
                    dp[i][k][t]=min(dp[i][k][t],dp[i-1][k][j]+cal(j,t));
                }
            }
        }
        int ans=0x3f3f3f3f;
        for(int i=0;i<=26;i++){
            for(int j=0;j<=26;j++){
                ans=min(ans,dp[n][i][j]);
            }
        }
        return ans;
    }
};
