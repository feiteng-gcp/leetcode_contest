class Solution {
public:
    int minimumDistance(string word) {
        vector<vector<int>> dis(26, vector<int>(26, 0));
        vector<vector<int>> dp(26, vector<int>(26, INT_MAX));
        for (int i = 0; i < 26; i++) {
            int ri = i / 6;
            int ci = i % 6;
            for (int j = i + 1; j < 26; j++) {
                int rj = j / 6;
                int cj = j % 6;
                int d = abs(ri - rj) + abs(ci - cj);
                dis[i][j] = d;
                dis[j][i] = d;
            }
        }
        for (int i = 0; i < 26; i++) {
            dp[i][word[0] - 'A'] = 0;
            dp[word[0] - 'A'][i] = 0;
        }

        auto tmp = dp;
        int len = word.size();
        for (int p = 1; p < len; p++) {
            int idx = word[p] - 'A';
            for (int i = 0; i < 26; i++) {
                for (int j = 0; j < 26; j++) {
                    tmp[i][j] = INT_MAX;
                }
            }
            for (int i = 0; i < 26; i++) {
                for (int j = 0; j < 26; j++) {
                    if (dp[i][j] == INT_MAX)
                        continue;

                    tmp[i][idx] = min(tmp[i][idx], dp[i][j] + dis[idx][j]);
                    tmp[idx][j] = min(tmp[idx][j], dp[i][j] + dis[idx][i]);
                }
            }
            swap(tmp, dp);
        }
        int ans = INT_MAX;
        for (int i = 0; i < 26; i++) {
            for (int j = 0; j < 26; j++) {
                ans = min(ans, dp[i][j]);
            }
        }
        return ans;
    }
};
