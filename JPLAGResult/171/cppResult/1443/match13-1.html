class Solution {
public:
  int minimumDistance(string word) { 
    vector<int> X(26);
    vector<int> Y(26);
    for (int i = 0; i < 26; i++) {
      X[i] = i / 6;
      Y[i] = i % 6;
    }
    int N = word.size();
    vector<vector<int>> cur(26, vector<int>(26));

    for(auto c: word) {
      int i = c - 'A';
      vector<vector<int>> nxt(26, vector<int>(26, 0x3f3f3f3f));
      for (int x = 0; x < 26; x++) {
        for (int j = 0; j < 26; j++) {
          nxt[x][i] = min(nxt[x][i], cur[x][j] + abs(X[i]-X[j]) + abs(Y[i]-Y[j]));
        }
      }

      for (int y = 0; y < 26; y++) {
        for (int j = 0; j < 26; j++) {
          nxt[i][y] = min(nxt[i][y], cur[j][y] + abs(X[i]-X[j]) + abs(Y[i]-Y[j]));
        }
      }
      cur = nxt;
    }
    int ans = 0x3f3f3f3f;
    for (auto t: cur) ans = min(ans, *min_element(t.begin(), t.end()));
    return ans;
  }
};
