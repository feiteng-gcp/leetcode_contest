class Solution {
public:
    
    int dp[302][26][26] ;
    int minimumDistance(string word) {
        vector<vector<int>> ch ;
        int x = 0 , y = 0 ;
        for (int i = 0 ; i < 26 ; ++i) {
            ch.push_back({x , y}) ;
            ++y ;
            if (y == 6) {
                ++x ; y = 0 ;
            }
        }
        int dist[26][26] ;
        memset(dist , 0 , sizeof(dist)) ;
        for (int i = 0 ; i < 26 ; ++i) {
            for (int j = 0 ; j < 26 ; ++j) {
                dist[i][j] = abs(ch[i][0] - ch[j][0]) + abs(ch[i][1] - ch[j][1]) ;
            }
        }
        int n = word.length() ;
        memset(dp , 0x7f7f7f , sizeof(dp)) ;
        for (int i = 0 ; i < 26 ; ++i) {
            for (int j = 0 ; j < 26 ; ++j) {
                dp[0][i][j] = 0 ;
            }
        }
        for (int i = 1 ; i <= n ; ++i) {
            for (int j = 0 ; j < 26 ; ++j) {
                for (int k = 0 ; k < 26 ; ++k) {
                    dp[i][word[i - 1] - 'A'][j] = min(dp[i][word[i - 1] - 'A'][j] , dp[i - 1][k][j] + dist[k][word[i - 1] - 'A']) ;
                    dp[i][j][word[i - 1] - 'A'] = min(dp[i][j][word[i - 1] - 'A'] , dp[i - 1][j][k] + dist[k][word[i - 1] - 'A']) ;
                }
                //
            }
        }
        int res = 0x7f7f7f ;
        for (int i = 0 ; i < 26 ; ++i) {
            for (int j = 0 ; j < 26 ; ++j) {
                res = min(res , dp[n][i][j]) ;
            }
        }
        return res ;
    }
};
