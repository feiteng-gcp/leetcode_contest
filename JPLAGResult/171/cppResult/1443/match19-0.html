class Solution {
private:
    const int INF = 0x3f3f3f3f;
    int dp[2][30][30];
    int getDis(int a, int b) {
        int xa = a / 6; int ya = a % 6;
        int xb = b / 6; int yb = b % 6;
        return abs(xa - xb) + abs(ya - yb);
    }
public:
    int minimumDistance(string word) {
        int fl = 0;
        memset(dp[fl], 0, sizeof(dp[fl]));
        memset(dp[fl^1], INF, sizeof(dp[fl^1]));
        
        for(int i = 0, len = word.length(); i < len; ++i) {
            int tt = word[i] - 'A';
            for(int j = 0; j < 26; ++j) {
                for(int k = 0; k < 26; ++k) {
                    int dis1 = getDis(tt, j);
                    dp[fl^1][tt][k] = min(dp[fl^1][tt][k], dp[fl][j][k] + dis1);
                    
                    int dis2 = getDis(tt, k);
                    dp[fl^1][j][tt] = min(dp[fl^1][j][tt], dp[fl][j][k] + dis2);
                }
            }
            fl ^= 1;
            memset(dp[fl^1], INF, sizeof(dp[fl^1]));
        }
        int ans = INF;
        for(int i = 0; i < 26; ++i) {
            for(int j = 0; j < 26; ++j) {
                ans = min(ans, dp[fl][i][j]);
            }
        }
        return ans;
    }
};
