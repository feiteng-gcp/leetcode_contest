class Solution {
public:
    int minimumDistance(string str) {
        int N=str.length(), res=INT_MAX;
        vector<vector<vector<int>>> dp(26, vector<vector<int>>( 26, vector<int> (N+1)));

        for(int i= N-1; i!=-1; --i){
            for(int a=0; a!=26; a++){   // first finger
                for(int b=0; b!= 26; b++){
                    int c= str[i]-'A';  // curr character
                    dp[a][b][i]=min(cost(a,c) + dp[c][b][i+1], cost(b,c)+ dp[a][c][i+1]);
                    if (i == 0){
                        res = min(res, dp[a][b][i]);
                    }

                }
            }
        }
        return res;
    }
    
    int cost(int i, int j){
        return abs( i/6-j/6)+ abs(i % 6 - j % 6);
    }
};
