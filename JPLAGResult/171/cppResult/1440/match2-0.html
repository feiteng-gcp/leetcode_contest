class Solution {
public:
    bool nonzero(int n) {
        while (n) {
            if (n%10==0) return false;
            n=n/10;
        }
        return true;
    }

    vector<int> getNoZeroIntegers(int n) {
        for (int i=1; i<n; i++) {
            if (nonzero(i) && nonzero(n-i)) {
                return {i,n-i};
            }
        }
        return {1,n-1};
    }
};
