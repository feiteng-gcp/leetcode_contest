class Solution {
public:
    bool isContainZero(int num)
    {
        if(num==0) return true;
        while(num>0)
        {
            if(num%10==0) return true;
            num/=10;
        }
        return false;
    }
    vector<int> getNoZeroIntegers(int n) {
        vector<int> res;
        for(int i=1;i<n;i++)
        {
            if(isContainZero(i)==false&&isContainZero(n-i)==false)
            {
                res.push_back(i);
                res.push_back(n-i);
                break;
            }
        }
        return res;
    }
};
