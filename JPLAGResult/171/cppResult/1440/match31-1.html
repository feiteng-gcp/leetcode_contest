class Solution {
public:
    bool isOK(int x){
        while(x>0){
            if((x%10)==0){
                return false;
            }
            x/=10;
        }
        return true;
    }
    vector<int> getNoZeroIntegers(int n) {
        for(int a=1;a<n;a++){

            if(isOK(a)&&isOK(n-a)){
                return {a,n-a};
            }
        }
        return {};
    }
};
