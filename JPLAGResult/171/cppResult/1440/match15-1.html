#include <bits/stdc++.h>
using namespace std;


class Solution {
	public:
		bool gotZero(int x) {
			while (x > 0) {
				if (x % 10 == 0) return true;
				x /= 10;
			}
			return false;
		}
		vector<int> getNoZeroIntegers(int n) {
			for (int i = 1; i < n; i++) {
				if (!gotZero(i) && !gotZero(n-i)) {
					vector<int> ans;
					ans.push_back(i);
					ans.push_back(n-i);
					return ans;
				}
			}
			vector<int> ans;
			return ans;
		}
};
