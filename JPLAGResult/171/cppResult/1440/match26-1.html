class Solution {
public:
    bool no_z(int x){
        if(x == 0)
            return false;
        while(x > 0){
            if(x % 10 == 0)
                return false;
            x/=10;
        }
        return true;        
    }
    vector<int> getNoZeroIntegers(int n) {
        for(int i = 1;i < n;i++){
            if(no_z(i) && no_z(n-i))
                return {i,n-i};
        }
        return {0,n};
    }
};
