class Solution {
public:
    int comp = 0;
    int makeConnected(int n, vector<vector<int>>& connections) {
        vector <bool> visited(n, false);
        
        if(connections.size() < n - 1) return -1;
        
        unordered_map <int , vector <int>> graph;
        for(auto i : connections){
            graph[i[0]].push_back(i[1]);
            graph[i[1]].push_back(i[0]);
        }
        
        for(int i = 0; i < n; ++i){
            if(!visited[i]){
                dfs(i,graph,visited);
                comp++;
            }
        }
        
        
        return comp - 1;
    }
    
    void dfs(int i, unordered_map <int , vector <int>> &graph, vector <bool> &v){
        if(v[i])
            return;
        v[i] = true;
        for(auto z : graph[i])
            dfs(z,graph,v);
        return;
    }
};
