class Solution {
public:
    vector<vector<int>> edges;
    void connect(vector<int> &res, int node, int val){
        if(res[node] != -1){
            return;
        }

        res[node] = val;
        for(int i = 0; i < edges[node].size(); i++){
            connect(res, edges[node][i], val);
        }
    }
    
    int makeConnected(int n, vector<vector<int>>& connections) {
        if(connections.size() < n - 1){
            return -1;
        }
        edges.resize(n, vector<int>());
        for(int i = 0; i < connections.size(); i++){
            int a = connections[i][0];
            int b = connections[i][1];
            edges[a].push_back(b);
            edges[b].push_back(a);
        }
        vector<int> res(n, -1);

        int count = 0;
        for(int i = 0; i < n; i++){
            if(res[i] == -1){
                connect(res, i, count++);
            }
        }
        return count - 1 ;
    }
};
