class Solution {
public:
    vector<vector<int>> g;
    vector<bool> vis;
    void dfs(int x) {
        vis[x] = 1;
        for(auto &y : g[x]) if(!vis[y]) dfs(y);
    }
    int makeConnected(int n, vector<vector<int>>& e) {

        if(e.size() < n - 1) return -1;
        g.resize(n);
        vis.resize(n);
        for(auto &x : e) {
            g[x[0]].push_back(x[1]);
            g[x[1]].push_back(x[0]);
        }
        int com = 0;
        for(int i = 0; i < n; i++) {
            if(!vis[i]) {
                com++;
                dfs(i);
            }
        }
        return com - 1;
    }
};
