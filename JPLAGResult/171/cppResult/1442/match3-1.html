class Solution {
public:
    vector<int>fa;
    
    int getFather(int p)
    {
        if (p==fa[p]) return p;
        else
        {
            fa[p]=getFather(fa[p]);
            return fa[p];
        }
    }
    
    int makeConnected(int n, vector<vector<int>>& connections) {
        int ret=0;
        if (connections.size()<n-1) return -1;
        for (int i=0;i<n;i++) fa.push_back(i);
        
        for (auto& p:connections)
        {
            int f1=getFather(p[0]);
            int f2=getFather(p[1]);
            if (f1!=f2)
            {
                if (f1<f2) fa[f2]=f1;
                else fa[f1]=f2;
            }
        }
        
        for (int i=0;i<n;i++)
        {
            //cout<<i<<' '<<getFather(i)<<endl;
            if (getFather(i)==i) ret++;
        }
        return ret-1;
    }
};
