class Solution {
public:
    void dfs(int i,vector<vector<int> > &m,vector<bool> &isV)
    {
        isV[i]=true;
        for(int j=0;j<m[i].size();j++)
        {
            int e=m[i][j];
            if(isV[e]==false)
            {
                dfs(e,m,isV);
            }
        }
    }
    int makeConnected(int n, vector<vector<int>>& connections) {
        if(connections.size()<n-1) return -1;
       vector<vector<int> > m(n);
       vector<bool> isV(n,false);
        for(auto connect:connections)
        {
            int s=connect[0],e=connect[1];
            m[s].push_back(e);
            m[e].push_back(s);
        }
        int res=0;
        for(int i=0;i<n;i++)
        {
            if(isV[i]==false)
            {
                dfs(i,m,isV);
                res++;
            }
        }
        return res-1;
    }
};
