int p[100010];
int f(int x) { return x == p[x] ? x : p[x] = f(p[x]); }
class Solution {
public:
    int makeConnected(int n, vector<vector<int>>& connections) {
        if (connections.size() < n - 1) {
            return -1;
        }
        for (int i = 0; i < n; ++i) p[i] = i;
        int comp = n;
        for (auto &x : connections) {
            int fa = f(x[0]);
            int fb = f(x[1]);
            if (fa != fb) {
                comp--;
                p[fa] = fb;
            }
        }
        return comp - 1;
    }
};
