class Solution {
public:
    vector < vector < int > > G;
    vector < int > vis;

    void dfs(int u) {
        vis[u] = 1;
        for (int v : G[u]) {
            if (vis[v] == false) {
                dfs(v);
            }
        }
    }

    int makeConnected(int n, vector<vector<int>>& connections) {
        int m = connections.size();
        if (m < n-1) return -1;
        G.resize(n);
        vis.resize(n);
        for (auto v : connections) {
            G[v[0]].push_back(v[1]);
            G[v[1]].push_back(v[0]);
        }
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            if (vis[i] == 0) {
                dfs(i);
                cnt++;
            }
        }
        return cnt-1;
    }
};
