class Solution {
public:
    int vis[100005] = {0};vector<int>edge[100006];
    
    void dfs(int s){
        if(vis[s])return ;
        vis[s] = 1;
        for(auto it : edge[s]){
            dfs(it);

        }
    }
    
    int makeConnected(int n, vector<vector<int>>& connections) {
        int len = connections.size();
        if(len<n-1)return -1;
        for(int i = 0;i<len;i++){
            edge[connections[i][0]].push_back(connections[i][1]);
            edge[connections[i][1]].push_back(connections[i][0]);
        }

        int cnt = 0;
        for(int i = 0;i<n;i++){
            if(vis[i]==0){
                cnt ++;
                dfs(i);
            }
        }
        return cnt - 1;
    }
};
