class Solution {
public:
    vector<int> pre;
    int find(int x){
        if(x == pre[x]) return x;
        else return find(pre[x]);
    }
    
    void join(int x, int y){
        int fx = find(x), fy = find(y);
        if (fx != fy) pre[fx] = fy;
    }
    
    bool same(int x, int y) {
        return find(x) == find(y);
    }
    
    int makeConnected(int n, vector<vector<int>>& connections) {
        pre = vector<int>(n, 0);
        for(int i = 0; i < pre.size(); ++i) pre[i] = i;
        

        int line = 0;
        for(int i = 0; i < connections.size(); ++i) {
            int x = connections[i][0];
            int y = connections[i][1];
            if(same(x, y)) ++line;
            else join(x, y);
        }
        set<int> s;
        for(int i = 0; i < n; ++i) s.insert(find(i));
        int res = s.size() - 1;
        if(line >= res) return res;
        else return -1;
        
    }
};
