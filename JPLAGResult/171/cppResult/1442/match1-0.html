class Solution {
    vector<int>Father;
public:
    int makeConnected(int n, vector<vector<int>>& connections) 
    {
        Father.resize(n);
        for (int i=0; i<n; i++)
            Father[i] = i;
        
        for (auto con: connections)
        {
            if (FindFather(con[0])!=FindFather(con[1]))
                Union(con[0],con[1]);
        }
        
        unordered_map<int,int>Group;
        for (int i=0; i<n; i++)
        {
            Father[i] = FindFather(i);
            Group[Father[i]]+=1;
        }
        
        unordered_map<int,int>Wire;
        for (auto con: connections)
        {
            Wire[Father[con[0]]]++;
        }
        
        int count = 0;
        for (auto x: Group)
        {
            int p = x.first;
            count += Wire[x.first] - (x.second-1);
        }
        
        //cout<<count<<" "<<Group.size()<<endl;
        
        if (count >= Group.size()-1)
            return Group.size()-1;
        else
            return -1;
    }
    
    int FindFather(int x)
    {
        if (Father[x]!=x)
            Father[x] = FindFather(Father[x]);
        return Father[x];
    }
    
    void Union(int x, int y)
    {
        x = Father[x];
        y = Father[y];
        if (x<y)
            Father[y] = x;
        else
            Father[x] = y;
    }    
};
