class Solution {
public:
    

    void dfs(vector<vector<int>>& edges, vector<bool>& visit, int u){
        visit[u] = true;
        for(int v: edges[u]){
            if (!visit[v]){
                dfs(edges, visit, v);
            }
        }
    }
    
    int makeConnected(int n, vector<vector<int>>& connections) {
        if (connections.size() < n - 1) return -1;
        vector<vector<int>> edges(n);
        for(int i = 0; i < connections.size(); i ++){
            int u = connections[i][0], v = connections[i][1];
            edges[u].push_back(v);
            edges[v].push_back(u);

        }
        vector<bool> visit(n, false);
        int res = 0;
        for(int i = 0; i < n; i ++){
            if (!visit[i]){
                res ++;
                dfs(edges, visit, i);
            }
        }
        return res - 1;
    }
};
