int par[100005];

int find(int a) {
    return par[a] == a ? a : par[a] = find(par[a]);
}

class Solution {
public:

    int makeConnected(int n, vector<vector<int>>& x) {
        if (x.size() < n - 1)
            return -1;
        int ans = n - 1;
        for (int i = 0; i < n; i++)
            par[i] = i;
        for (auto& i : x) {
            int ta = find(i[0]);
            int tb = find(i[1]);
            if (ta != tb) {
                ans--;
                par[ta] = tb;
            }
        }
        return ans;
    }
};
