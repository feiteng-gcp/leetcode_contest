class Solution {
public:
  int makeConnected(int n, vector<vector<int>>& connections) {
    int n_conn = connections.size();
    if (n_conn < n - 1) {
      return -1;
    }
    vector<vector<int>> edges(n);
    for (auto &conn : connections) {
      edges[conn[0]].push_back(conn[1]);
      edges[conn[1]].push_back(conn[0]);
    }
    vector<int> connected(n, 0);
    int not_conn = 0;
    for (int i = 0; i < n; i++) {
      if (connected[i] == 0) {
        not_conn++;
        dfs(edges, i, connected);  
      }
    }
    return not_conn - 1;
  }
 private:
  void dfs(vector<vector<int>> &edges, int u, vector<int> &connected) {
    if (connected[u] == 1) {
      return;
    }
    connected[u] = 1;
    for (int v : edges[u]) {
      dfs(edges, v, connected);
    }
    return;
  }
};
