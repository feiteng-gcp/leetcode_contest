class Solution {
public:
    int makeConnected(int n, vector<vector<int>>& c) {
        int m = c.size();
        if (m < n-1) {
            return -1;
        }
        
        vector<int> p(n);
        for (int i = 0; i < n; ++i)
            p[i] = i;
        
        int num = n;
        for (int i = 0; i < m; ++i) {
            int x = c[i][0];
            int y = c[i][1];
            int px = get_p(x, p);
            int py = get_p(y, p);
            if (px != py) {
                --num;
                p[px] = py;
            }
        }
        
        return num - 1;
    }
    
    int get_p(int x, vector<int> &p) {
        if (p[x] == x)
            return x;
        p[x] = get_p(p[x], p);
        return p[x];
    }
};
