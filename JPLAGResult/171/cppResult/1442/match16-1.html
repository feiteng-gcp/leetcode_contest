class Solution {
public:

    void DFS(const vector<vector<int>>& g, const int u, vector<bool>& vis) {
        vis[u] = true;
        for (auto v : g[u]) {
            if (!vis[v])
                DFS(g, v, vis);
        }
    }
    int makeConnected(int n, vector<vector<int>>& connections) {
        if (connections.size() < n-1)
            return -1;
        vector<vector<int>> g(n);
        for (const auto& c : connections) {
            int u = c[0];

            int v = c[1];
            g[u].push_back(v);
            g[v].push_back(u);
        }
        vector<bool> vis(n, false);
        int cnt = 0;
        for (int i = 0; i < n; ++i) {
            if (!vis[i]) {
                cnt++;
                DFS(g, i, vis);
            }
        }
        return cnt-1;
    }
};
