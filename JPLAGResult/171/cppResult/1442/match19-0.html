class Solution {
public:
    void dfs(int j, vector<vector<int>>& mat, vector<bool>& vi) {
        vi[j] = true;
        for (int i = 0; i < mat[j].size(); i++) {
            if (!vi[mat[j][i]]) {
                dfs(mat[j][i], mat, vi);
            }
        }
    }
    int makeConnected(int n, vector<vector<int>>& connections) {
        if (connections.size() < n-1) return -1;
        vector<bool> vi(n, false);
        vector<vector<int>> mat(n, vector<int>());
        for (auto& c : connections) {
            mat[c[0]].push_back(c[1]);
            mat[c[1]].push_back(c[0]);
        }
        int res = -1;
        for (int i = 0; i < n; i++) {
            if (!vi[i]) {
                res++;
                dfs(i, mat, vi);
            }
        }
        return res;
    }
};
