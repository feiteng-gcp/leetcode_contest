class Solution {
public:
    int makeConnected(int n, vector<vector<int>>& connections) {
        if(n>connections.size()+1)return -1;
        vector<int> parent(n,0);
        for(int i=0;i<n;i++)parent[i]=i;
        for(auto &v:connections){
            
            parent[findParent(v[0],parent)]=findParent(v[1],parent);
            //for(auto &i:parent)cout<<i<<" ";cout<<endl;

        }
        int ans=0;
        //for(auto &i:parent)cout<<i<<endl;
        for(int i=0;i<n;i++){
            if(parent[i]==i)ans++;
        }
        return ans-1;
    }
    
    int findParent(int p,vector<int>& parent){
        if(parent[p]==p)return p;
        return parent[p]=findParent(parent[p],parent);
    }
};
