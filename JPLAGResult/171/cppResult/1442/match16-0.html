class Solution {
public:
    vector<int> G[102000];
    int V,E;
    bool vis[102000];

    void dfs(int u) {
        vis[u] = 1;
        for(auto v:G[u]) {
            if(!vis[v]) dfs(v);
        }
    }

    int makeConnected(int n, vector<vector<int>>& connections) {
        V = n;
        E = connections.size();
        if(E+1<V) return -1;
        memset(vis,0,sizeof(vis));
        for(auto &t: connections) {
            int u = t[0]+1, v = t[1]+1;
            G[u].push_back(v);
            G[v].push_back(u);
        }
        int cc=0;
        for(int i=1;i<=V;i++) {
            if(!vis[i]) {
                dfs(i);
                cc++;
            }
        }
        return cc-1;
    }
};
