class Solution {
public:
    int len;
    int fa[100005];
    int gf(int x)
    {
        if(x!=fa[x])
            fa[x]=gf(fa[x]);
        return fa[x];
    }
    
    int makeConnected(int n, vector<vector<int>>& connections) {
        len=connections.size();
        if(len<n-1)
            return -1;
        else
        {

            for(int i=0;i<n;i++)
                fa[i]=i;
            int ans=0;
            for(auto itm: connections)
            {
                int f1 = gf(itm[0]);
                int f2 = gf(itm[1]);
                if(f1!=f2)
                {
                    ans++;
                    fa[f2]=f1;
                }
            }
            return (n-1-ans);
        }
    }
};
