class Solution {
    vector<int> fa;
    int findf(int v) {
        while (fa[v] != v) v = findf(fa[v]);
        return v;
    }
public:
    int makeConnected(int n, vector<vector<int>>& con) {
        if (con.size() + 1 < n) return -1;

        fa = vector<int>(n, -1);
        for (int i = 0; i < n; i++)fa[i] = i;
        int tot = n;
        for (int i = 0; i < con.size(); i++) {
            int a = con[i][0];
            int b = con[i][1];
            int aa = findf(a);
            int bb = findf(b);
            if (aa != bb) {
                fa[bb] = aa;
                tot--;
            }
        }
        return tot - 1;
    }
};
