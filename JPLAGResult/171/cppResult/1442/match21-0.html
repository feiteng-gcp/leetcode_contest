vector<int> v[100001];
bool vis[100001];
class Solution {
public:
    void dfs(int n)
    {
        vis[n]=true;
        int l=v[n].size();
        for(int i=0;i<l;i++)
        {
            if(!vis[v[n][i]])dfs(v[n][i]);
        }
    }
    int makeConnected(int n, vector<vector<int>>& connections) {
        int m=connections.size();
        if(m<n-1)return -1;
        for(int i=0;i<n;i++)
        {
            v[i].clear();
        }
        for(int i=0;i<m;i++)
        {
            v[connections[i][0]].push_back(connections[i][1]);
            v[connections[i][1]].push_back(connections[i][0]);
        }
        memset(vis,0,sizeof(vis));
        int cnt=0;
        for(int i=0;i<n;i++)
        {
            if(!vis[i])
            {
                cnt++;
                dfs(i);
            }
        }
        return cnt-1;
    }
};
