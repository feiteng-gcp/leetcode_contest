class Solution {
public:

    void dfs(int u) {
        vist[u] = true;
        for (int i = 0; i < graph[u].size(); ++i) {
            int v = graph[u][i];
            if (!vist[v]) {
                dfs(v);
            }
        }
    }
    int makeConnected(int n, vector<vector<int>>& conn) {
        if (conn.size() + 1 < n) {
            return -1;
        }
        graph.clear();
        for (int i = 0; i < n; ++i) {
            graph.push_back(vector<int>());
        }
        for (int i = 0; i < conn.size(); ++i) {
            int u = conn[i][0];
            int v = conn[i][1];
            graph[u].push_back(v);
            graph[v].push_back(u);
        }
        memset(vist, 0, sizeof(vist));

        int cnt = 0;
        for (int i = 0; i < n; ++i) {
            if (!vist[i]) {
                ++cnt;
                dfs(i);
            }
        }
        return cnt - 1;
    }
    bool vist[100011];
    vector<vector<int> > graph;
};
