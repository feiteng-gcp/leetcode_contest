class UF {
public:
    vector<int> fa;
    vector<int> sz;
    int n;
    
public:
    UF(int _n): n(_n), fa(vector<int>(_n)), sz(vector<int>(_n, 1)) {
        for (int i = 0; i < n; ++i) {
            fa[i] = i;
        }
    }
    
    int findset(int x) {
        return fa[x] == x ? x : fa[x] = findset(fa[x]);
    }
    
    void unite(int x, int y) {
        if (sz[x] < sz[y]) {
            swap(x, y);
        }
        fa[y] = x;
        sz[x] += sz[y];
    }
    
    void findunite(int x, int y) {
        int x0 = findset(x);
        int y0 = findset(y);
        if (x0 != y0) {
            unite(x0, y0);
        }
    }

};


class Solution {
public:
    int makeConnected(int n, vector<vector<int>>& connections) {
        if (connections.size() + 1 < n) {
            return -1;
        }
        
        UF uf(n);

        int cnt = 0;
        for (auto&& c: connections) {
            int p = uf.findset(c[0]);
            int q = uf.findset(c[1]);
            if (p != q) {
                ++cnt;
                uf.unite(p, q);
            }
        }
        
        return n - 1 - cnt;
    }
};
