class Solution {
public:
    int f[100005];
    int Find(int x){
        if(f[x]==x)return x;
        return f[x]=Find(f[x]);
    }
    int makeConnected(int n, vector<vector<int>>& connections) {
        if(connections.size()<n-1)return -1;
        int cnt=0;
        for(int i = 1;i<=n;i++)f[i]=i;
        for(auto it:connections){
            int a=Find(it[0]);
            int b=Find(it[1]);
            if(a!=b){
                cnt++;
                f[a]=b;
            }
        }
        return n-1-cnt;
    }
};
