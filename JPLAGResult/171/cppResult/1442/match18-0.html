class Solution {
public:
    int makeConnected(int n, vector<vector<int>>& connections) {
        if(connections.size()<n-1)return -1;
        vector<int> p(n);
        for(int i = 0;i<n;++i){
            p[i]=i;
        }
        for(auto &conn:connections){
            merge(p,conn[0],conn[1]);
        }
        int num=0;
        for(int i = 0;i<n;++i){
            if(i==findR(p,i))num++;
        }
        return num-1;
    }
    void merge(vector<int> &p,int x,int y){
        int  rx=findR(p,x);
        int  ry=findR(p,y);
        if(rx==ry)return;
        p[ry]=rx;
    }
    int findR(vector<int> &p,int x){
        if(p[x]==x)return x;
        return p[x]=findR(p,p[x]);
    }
};
