class Solution {
private:
    vector<int>f;
    int pd(int x){
        if(f[x]!=x)f[x]=pd(f[x]);
        return f[x];
    }
public:
    int makeConnected(int n, vector<vector<int>>& connections) {
        if(connections.size()<n-1)return -1;
        f = vector<int>(n);
        for(int i=0;i<n;i++){
            f[i] = i;
        }
        int cnt = n;
        for(int i=0;i<connections.size();i++){
            int x = connections[i][0];
            int y = connections[i][1];
            int px = pd(x);
            int py = pd(y);
            if(px != py){
                f[px] = py;
                cnt--;
            }
        }
        return cnt - 1;
    }
};
