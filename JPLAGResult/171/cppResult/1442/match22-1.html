class Solution {

    vector<int> ss;
    int find(int a){
        if(ss[a]!=a) ss[a]=find(ss[a]);
        return ss[a];
    }
    void merge(int a,int b){
        ss[find(a)]=find(b);
    }
public:

    int makeConnected(int n, vector<vector<int>>& connections) {
        if(connections.size()<n-1) return -1;
        ss=vector<int>(n);
        for(int i=0;i<n;i++) ss[i]=i;
        for(auto c:connections){
            merge(c[0],c[1]);
        }
        int cnt=0;
        for(int i=0;i<n;i++){
            if(find(i)==i) cnt++;
        }
        return cnt-1;
    }
};
