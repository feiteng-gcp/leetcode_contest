class Solution:
    def minimumDistance(self, word: str) -> int:
        def distance(a, b):
            if a == '*' or b == '*':
                return 0
            index_a, index_b = ord(a) - ord('A'), ord(b) - ord('A')
            r_a, c_a = index_a // 6, index_a % 6
            r_b, c_b = index_b // 6, index_b % 6
            return abs(r_a - r_b) + abs(c_a - c_b)
            
        mem = {}
        def dp(pos1, pos2, index):
            if index == len(word):
                return 0
            
            if (pos1, pos2, index) in mem:
                return mem[pos1, pos2, index]
            
            res = min(dp(pos1, word[index], index + 1) + distance(pos2, word[index]), \
                      dp(word[index], pos2, index + 1) + distance(pos1, word[index]))
            
            mem[pos1, pos2, index] = res
            return res
            
        return dp('*', '*', 0)
