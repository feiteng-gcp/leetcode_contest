class Solution:
    def minimumDistance(self, word: str) -> int:
        def distance(a, b):
            if a == 0 or b == 0: return 0
            x1, y1 = (a - 1) // 6, (a - 1) % 6
            x2, y2 = (b - 1) // 6, (b - 1) % 6
            return abs(x1 - x2) + abs(y1 - y2)

        n = len(word)
        MAXD = 10**8
        dp = [[[MAXD] * 27 for _ in range(27)] for _ in range(n + 1)]
        dp[0][0][0] = 0
        for i in range(n):
            c = ord(word[i]) - ord('A') + 1
            for x in range(27):
                for y in range(27):
                    if dp[i][x][y] == MAXD: continue
                    dp[i + 1][x][c] = min(dp[i + 1][x][c], dp[i][x][y] + distance(y, c))
                    dp[i + 1][c][y] = min(dp[i + 1][c][y], dp[i][x][y] + distance(x, c))

        ret = MAXD
        for i in range(27):
            for j in range(27):
                ret = min(ret, dp[n][i][j])
        return ret
