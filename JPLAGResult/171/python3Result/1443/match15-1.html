import functools

class Solution:
    def minimumDistance(self, word: str) -> int:
        
        def coord(a: str) -> [int, int]:
            diff = abs(ord(a) - ord('A'))
            q, r = divmod(diff, 6)
            return q, r
        
        def distance(a: str, b: str) -> int:
            if a == '' or b == '':
                return 0
            ax, ay = coord(a)
            bx, by = coord(b)
            return abs(ax - bx) + abs(ay - by)
        
        N = len(word)
        
        @functools.lru_cache(maxsize=None)
        def recurse(i: int, one: str, two: str) -> int:
            if i == N:
                return 0
            target = word
[i]
            return min(recurse(i+1, target, two) + distance(one, target), recurse(i+1, one, target) + distance(two, target))
        
        return recurse(0, '', '')
