        def dist(a,b):
            if b==26: return 0
            return abs(a//6 - b//6) + abs(a%6 - b%6)
        for i,c in enumerate(temps):
            cd[c] = i
        dp[0][cd[word[0]]][26] =  0

            ans = 10**10

                for k in range(27):
                    if dp[i-1][j][k]!=-1:
                        dp[i]
[cd[word[i]]][k] = dp[i-1][j][k] + dist(cd[word[i]], j) if dp[i][cd[word[i]]][k]==-1 else min(dp[i-1][j][k] + dist(cd[word[i]], j), dp[i][cd[word[i]]][k])
                        dp[i]
[j][cd[word[i]]] = dp[i-1][j][k] + dist(cd[word[i]], k) if dp[i][j][cd[word[i]]]==-1 else min(dp[i-1][j][k] + dist(cd[word[i]], k), dp[i][j][cd[word[i]]])
