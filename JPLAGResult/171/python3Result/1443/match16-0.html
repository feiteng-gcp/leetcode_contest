from collections import defaultdict
class Solution:
    def minimumDistance(self, word: str) -> int:
        def get_idx(ch):
            idx = ord(ch) - ord('A')
            return idx
        def pos(idx):
            return (idx // 6, idx % 6)
        def dist(i1, i2):
            if i1 == -1:
                return 0
            r1, c1 = pos(i1)
            r2, c2 = pos(i2)
            return abs(r1 - r2) + abs(c1 - c2)
        dp = defaultdict(lambda: float('inf'))
        dp[(-1, -1)] = 0
        for ch in word:
            tmp = defaultdict(lambda: float('inf'))
            idx = get_idx(ch)
            for (l, r), cost in dp.items():
                d1 = dist(l, idx)
                tmp[(idx, r)] = min(tmp[(idx, r)], cost + d1)
                d2 = dist(r, idx)
                tmp[(l, idx)] = min(tmp[(l, idx)], cost + d2)
            dp = tmp
        return min(dp.values())
