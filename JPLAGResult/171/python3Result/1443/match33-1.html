class Solution:
    def minimumDistance(self, word: str) -> int:
        n = len(word)
        mp = {}
        dp = [[[-1 for k in range(27)] for j in range(27)] for i in range(n)]
        #print(dp[0][0][0])
        g = collections.defaultdict(list)
        for i in range(26):
            g[chr(i+65)] = [i//6, i%6]
        
        
def dist(a, b):
            if not a or not b:
                return 0
            return abs(a[0]-b[0]) + abs(a[1]-b[1])
        
        def val(tup):
            if not tup:
                return 0
            return tup[0]*6+tup[1]+1
        
        def go(idx, first, second):
            if idx == n:
                return 0
            if idx == 0:
                return go(idx+1, g[word[idx]], second)
            if dp[idx][val(first)][val(second)] != -1:
                return dp[idx][val(first)][val(second)]
            res = min(dist(g[word[idx]], first) + go(idx+1, g[word[idx]], second), dist(g[word[idx]], second) + go(idx+1, first, g[word[idx]]))
            dp[idx][val(first)][val(second)] = res
            return res
        
        return go(0, None, None)
