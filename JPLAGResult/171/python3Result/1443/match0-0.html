        curpos1 = pos[word[0]]
        from functools import lru_cache
        @lru_cache(None)
        def dfs(pos1, pos2, ind):
            # print(pos1, pos2, ind)
            if ind >= n: return 0
            p = pos[word[ind]]
            
            if pos2:
                return min(getDist(pos1, p) + dfs(p, pos2, ind + 1), getDist(pos2, p) + dfs(pos1, p, ind + 1))
            else:
                return min(getDist(pos1, p) + dfs(p, pos2, ind + 1), dfs(pos1, p, ind + 1))
            

        return dfs(curpos1, None, 1)
