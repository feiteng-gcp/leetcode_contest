class Solution:
    def minimumDistance(self, word: str) -> int:
        
        loc_map = {}
        chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
        for i in range(5):
            for j in range(6):
                idx = j + 6*i
                if idx >= len(chars): break
                loc_map[chars[idx]] = (i,j)
        #print(loc_map)
        if len(word) <= 2: return 0
        
        curr = {}
        for w in word:
            if w != word[0]:
                curr[(loc_map[word[0]], loc_map[w])] = 0
        
        for w in word[1:]:
            new_curr = {}
            loc = loc_map[w]
            for ((a,b), (c,d)), moves in curr.items():
                key = (loc, (c,d))
                new_curr[key] = min(new_curr.get(key, float('Inf')), moves + abs(a-loc[0]) + abs(b-loc[1]))
                key = ((a,b), loc)
                new_curr[key] = min(new_curr.get(key, float('Inf')), moves + abs
(c-loc[0]) + abs(d-loc[1]))                
            #print(curr)

            curr = new_curr
        
        return min(curr.values())
