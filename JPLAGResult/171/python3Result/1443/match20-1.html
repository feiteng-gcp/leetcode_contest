class Solution:
    def minimumDistance(self, word: str) -> int:
        def distance(c1, c2):
            x1, y1 = divmod(ord(c1)-ord('A'), 6)
            x2, y2 = divmod(ord(c2)-ord('A'), 6)
            return abs(x1-x2)+abs(y1-y2)
        
        dp1 = distance(word[0], word[1])
        dp2 = {tuple(sorted(word[0:2])): 0}
                
        for cp, c in zip(word[1:], word[2:]):
            dp1_ = dp1+distance(cp, c)
            
            dp2_ = {}
            for f1, f2 in dp2:
                d = dp2[(f1, f2)]
                if c == f1 or c == f2:
                    dp2_[(f1, f2)] = min(dp2_.get((f1, f2), float('inf')), d)
                else:
                    fs = tuple(sorted((f1, c)))
                    dp2_[fs] = min(dp2_.get(fs, float('inf')), d+distance(f2, c))
                    fs = tuple(sorted((f2, c)))
                    dp2_[fs] = min(dp2_.get(fs, float('inf')), d+distance(f1, c))
            if cp != c:
                fs = tuple(sorted((cp, c)))
                dp2_[fs] = min(dp2_.get(fs, float('inf')), dp1)
            
            dp1, dp2 = dp1_, dp2_
        
        return min(dp1, min(dp2.values()))
