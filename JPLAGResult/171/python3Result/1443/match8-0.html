import functools
class Solution:
    def minimumDistance(self, word: str) -> int:
        
        def cost(idx1, idx2):
            if idx1 == -1:
                return 0
            else:
                letter1, letter2 = word[idx1], word[idx2]
                r1, c1 = (ord(letter1) - ord('A')) // 6, (ord(letter1) - ord('A')) % 6
                r2, c2 = (ord(letter2) - ord('A')) // 6, (ord(letter2) - ord('A')) % 6
                
                return abs(r1 - r2) + abs(c1 - c2)
        
        @functools.lru_cache(None)
        def doit(idx1, idx2):
            if idx1 == len(word) - 1 or idx2 == len(word) - 1:
                return 0
            
            idx = max(idx1, idx2)
            cost1 = doit(idx + 1, idx2) + cost(idx1, idx + 1)
            cost2 = doit(idx1, idx + 1) + cost(idx2, idx + 1)
            return min(cost1, cost2)
        
        return doit(-1, -1)
