from functools import lru_cache
class Solution:
    def minimumDistance(self, word: str) -> int:
        pos_map 
= {
            'A': (0, 0), 'B': (0, 1), 'C': (0, 2), 'D': (0, 3), 'E': (0, 4),'F': (0, 5),
            'G': (1, 0), 'H': (1, 1), 'I': (1, 2), 'J': (1, 3), 'K': (1, 4),'L': (1, 5),
            'M': (2, 0), 'N': (2, 1), 'O': (2, 2), 'P': (2, 3), 'Q': (2, 4),'R': (2, 5),
            'S': (3, 0), 'T': (3, 1), 'U': (3, 2), 'V': (3, 3), 'W': (3, 4),'X': (3, 5),
            'Y': (4, 0), 'Z': (4, 1)
          }
        def cost(pos1, pos2):
            if pos1 is None or pos2 is None:
                return 0
            x1, y1 = pos1
            x2, y2 = pos2
            return abs(x1-x2) + abs(y1-y2)
        @lru_cache(None)
        def inner(p1, p2, i):
            if i >= len(word):
                return 0
            ch = word[i]
            pos = pos_map[ch]
            cost1 = cost(pos, p1) + inner(pos, p2, i+1)
            cost2 = cost(pos, p2) + inner(p1, pos, i+1)
            return min(cost1, cost2)
        return inner(None, None, 0)
