class Solution:
    def minimumDistance(self, word: str) -> int:
        self.d = {'A': (0, 0), 'B': (0, 1), 'C': (0, 2), 'D': (0, 3), 'E': (0, 4), 'F': (0, 5), 'G': (1, 0), 'H': (1, 1), 'I': (1, 2), 'J': (1, 3), 'K': (1, 4), 'L': (1, 5), 'M': (2, 0), 'N': (2, 1), 'O': (2, 2), 'P': (2, 3), 'Q': (2, 4), 'R': (2, 5), 'S': (3, 0),'T': (3, 1), 'U': (3, 2), 'V': (3, 3), 'W': (3, 4), 'X': (3, 5), 'Y': (4, 0), 'Z': (4, 1)}
        def distance(a, b):
            if a == -1 or b == -1:
                return 0
            else:
                return abs(self.d[a][0] - self.d[b][0]) + abs(self.d[a][1] - self.d[b][1])
        finger_dic = dict()
        finger_dic[tuple([word[0], -1])] = 0
        finger_dic[tuple([-1, word[0]])] = 0
        for i in range(1, len(word)):
            new_finger_dic = dict()
            for key in finger_dic.keys():
                [left, right] = key
                value = finger_dic[key]
                new_key = tuple([word[i], right])
                new_distance = value + distance(left, word[i])
                if new_key in new_finger_dic:
                    new_finger_dic[new_key] = min(new_distance, new_finger_dic[new_key])
                else:
                    new_finger_dic[new_key] = new_distance
                new_key = tuple([left, word[i]])
                new_distance = value + distance(right, word[i])
                if new_key in new_finger_dic:
                    new_finger_dic[new_key] = min(new_distance, new_finger_dic[new_key])
                else:
                    new_finger_dic[new_key] = new_distance
            finger_dic = new_finger_dic
        return min(finger_dic.values())
