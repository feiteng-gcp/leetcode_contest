class Solution:
    def minimumDistance(self, word: str) -> int:
        dp = {(-1,-1) : 0 }
        for ch in word:
            c = ord(ch) - ord('A')
            tmp = {}
            for a, b in dp:
                preMoves = dp[(a,b)]
                a2c = self.dist(a,c) + preMoves
                tmp[(c,b)] = min(tmp.get((c,b), 5000), a2c)
                b2c = self.dist(b,c) + preMoves
                tmp[(a,c)] = min(tmp.get((a,c), 5000), b2c)
            dp = tmp
        return min(dp.values())
    
    def dist(self, a, b):
        if a < 0 or b < 0:
            return 0
        return abs(a//6 - b//6) + abs(a%6 -b%6)
