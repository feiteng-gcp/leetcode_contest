from functools import lru_cache

class Solution:
    def minimumDistance(self, word: str) -> int:
        n = len(word)
        
        loc = {chr(i + ord('A')): (i // 6, i % 6) for i in range(26)}
        
        def dis(a, b):
            if a is None:
                return 0
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        @lru_cache(maxsize=None)
        def f(a, b, i):
            if i == n:
                return 0
            next_loc = loc[word[i]]
            return min(f(next_loc, b, i + 1) + dis(a, next_loc),
                       f(a, next_loc, i + 1) + dis(b, next_loc))
        
        return f(None, None, 0)
