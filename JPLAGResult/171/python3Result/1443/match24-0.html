class Solution:
    def minimumDistance(self, word: str) -> int:
        key = [
            "ABCDEF",
            "GHIJKL",
            "MNOPQR",
            "STUVWX",
            "YZ    "
        ]

        def c(x):
            return ord(x) - ord('A')
        
        dist = {}
        for x1 in range(len(key)):
            for y1 in range(len(key[x1])):
                if key[x1][y1] != " ":
                    for x2 in range(len(key)):
                        for y2 in range(len(key[x2])):
                            if key[x2][y2] != " ":
                                dist[(c(key[x1][y1]), c(key[x2][y2]))] = abs(x1 - x2) + abs(y1 - y2)
        #print(dist)
        seen = {}
        q = []
        N = len(word)
        
        for a in range(26):
            for b in range(26):
                seen[(a, b, 0)] = 0
                heapq.heappush(q, (0, a, b, 0))
                
        while len(q) > 0:
            (d, left, right, index) = heapq.heappop(q)
            
            if index == N:
                return d
            
            if d > seen[(left, right, index)]:
                continue
            
            nf = c(word[index])
            nkey = (nf, right, index + 1)
            if nkey not in seen or d + dist[(left, nf)] < seen[nkey]:
                seen[nkey] = d + dist[(left, nf)]
                heapq.heappush(q, (d + dist[(left, nf)], nf, right, index + 1))
            
            nkey = (left, nf, index + 1)
            if nkey not in seen or d + dist[(right, nf)] < seen[nkey]:
                seen[nkey] = d + dist[(right, nf)]
                heapq.heappush(q, (d + dist[(right, nf)], left, nf, index + 1))
            
        return -1
