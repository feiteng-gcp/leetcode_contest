class Solution:
    def minimumDistance(self, word: str) -> int:
        dpos={}
        for i in range(26):
            lett=chr(i+65)
            dpos[lett]=[i//6,i%6]
        def calcdis(c1,c2):
            if not c1 or not c2:
                return 0
            t1,t2=dpos[c1],dpos[c2]
            return abs(t1[0]-t2[0])+abs(t1[1]-t2[1])
        cur={}
        cur[(word[0],None)]=0
        for i in range(1,len(word)):
            nextcur={}
            ch=word[i]
            for k in cur:
                if (k[0],ch) not in nextcur:
                    nextcur[(k[0],ch)]=calcdis(k[1],ch)+cur[k]
                else:
                    nextcur[(k[0],ch)]=min(nextcur[(k[0],ch)],calcdis(k[1],ch)+cur[k])
                if (ch,k[1]) not in nextcur:
                    nextcur[(ch,k[1])]=calcdis(k[0],ch)+cur[k]
                else:
                    nextcur[(ch,k[1])]=min(nextcur[(ch,k[1])],calcdis(k[0],ch)+cur[k])
            cur=nextcur
        dis=[cur[k] for k in cur]
        return min(dis)
