class Solution:
    def minimumDistance(self, word: str) -> int:
        def getPos(num):
            row, col = num//6, num%6
            
            return (row, col)
        
        def getDist(p1, p2):
            return abs(p1[0]-p2[0]) + abs(p1[1]-p2[
1])
        
        
        dp = {}
        
        def solve(index, f1, f2):
            if index == len(word): return 0
            
            if (index, f1, f2) in dp: return dp[(index, f1, f2)]
            
            ret = 10**17
            
            if f1: c1 = ord(f1)-65
            else: c1 = -1
                
            if f2: c2 = ord(f2)-65
            else: c2 = -1
            
            c = ord(word[index])-65
            
            if not f1: ret = min(ret, solve(index+1, word[index], f2))
            else: ret = min(ret, solve(index+1, word[index], f2) + getDist(getPos(c1), getPos(c)))
                
            if not f2: ret = min(ret, solve(index+1, f1, word[index]))
            else: ret = min(ret, solve(index+1, f1, word[index]) + getDist(getPos(c2), getPos(c)))
                
            dp[(index, f1, f2)] = ret
            return ret
        
        return solve(0, "", "")
