class Solution:
    def minimumDistance(self, word: str) -> int:
        
        n = len(word)
        self.res = float('inf')
        
        def getpos(x):
            x = ord(x)-ord('A')
            return [x//6,x%6]
        def getdis(a,b, i,j):
            if (a,b)==(-1,-1):
                return 0
            return abs(a-i)+abs(b-j)
        import functools
        @functools.lru_cache(None)
        def dfs(i,x0,y0,x1,y1):
            #print(i,x0,y0,x1,y1)
            if i==n:
                return 0
            curx,cury = getpos(word
[i])
            return min( getdis(x0,y0,curx,cury)+dfs(i+1,curx,cury,x1,y1),  getdis(x1,y1,curx,cury)+dfs(i+1,x0,y0,curx,cury))
        
        return dfs(0,-1,-1,-1,-1)
