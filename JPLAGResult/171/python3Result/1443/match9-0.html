from functools import lru_cache
class Solution:
    def minimumDistance(self, word: str) -> int:
        ### f1 on first letter
        ### f2 could be on any of the remaining letter
        
        N = len(word)
        
        loc = {}
        for i in range(26):
            loc[chr(i+65)] = divmod(i,6)
        #print(loc)
        #return
            
        def dis(a,b):
            return abs(loc[a][0]-loc[b][0]) + abs(loc[a][1]-loc[b][1])
        
        @lru_cache(None)
        def min_step(f1,f2,index):
            if index>=N: return 0
            nc = word[index]
            a = dis(f1,nc) + min_step(nc,f2,index+1)
            b = dis(f2,nc) + min_step(f1,nc,index+1)
            return min(a,b)
            
        res = float('inf')
        for f2 in word[1:]:
            res = min(res, min_step(word[0],f2,1))
        return res
