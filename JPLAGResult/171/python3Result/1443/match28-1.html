class Solution:
    def minimumDistance(self, word: str) -> int:
        def dis(i, j):
            return abs(i//6-j//6) + abs(i%6-j%6)
        
        dp = {(i,j):0 for i in range(26) for j in range(26)}

            dp = ndp
        return min(dp.values())
