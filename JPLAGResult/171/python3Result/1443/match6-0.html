class Solution:
    def minimumDistance(self, word: str) -> int:
        def find(c):
            row, col = divmod(ord(c) - ord('A'), 6)
            return row, col
        
        q = [(0, 0, -1, -1)]
        seen = set()
        while q:
            cost, i, state1, state2 = heapq.heappop(q)
            if i == len(word):
                return cost
            
            if (i, state1, state2) in seen:
                continue
            seen.add((i, state1, state2))
            
            if state1 == -1:
                heapq.heappush(q, (cost, i+1, i, state2))
            else:
                ri, ci = find(word[i])
                rp, cp = find(word[state1])
                heapq.heappush(q, (cost + abs(ri - rp) + abs(ci - cp), i+1, i, state2))
                
            if state2 == -1:
                heapq.heappush(q, (cost, i+1, state1, i))
            else:
                ri, ci = find(word[i])
                rp, cp = find(word[state2])
                heapq.heappush(q, (cost + abs(ri - rp) + abs(ci - cp), i+1, state1, i))
