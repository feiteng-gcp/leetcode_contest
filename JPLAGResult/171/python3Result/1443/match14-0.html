class Solution:
    def cordinate(self, letter):
        order = ord(letter) - ord('A')
        return [order // 6 , order % 6]
    
    def distance(self, point1, point2):
        if point1[0] == -1 or point2[0] == -1:
            return 0
        return abs(point1[0]-point2[0]) + abs(point1[1] - point2[1])
    
    def minimumDistanceUtil(self, word , firstFinger , secondFinger , cp , dp):
        if cp == len(word):
            return 0
        key = (firstFinger[0],firstFinger[1],secondFinger[0],secondFinger[1],cp)
        if key in dp:
            return dp[key]
        nextCordinate = self.cordinate(word[cp])
        result1 = self.distance(firstFinger,nextCordinate) + self.minimumDistanceUtil(word,nextCordinate,secondFinger,cp+1,dp)
        result2 = self.distance(secondFinger,nextCordinate) + self.minimumDistanceUtil(word,firstFinger,nextCordinate,cp+1,dp)
        dp[key] = min(result1,result2)
        return dp[key]
        
    
    def minimumDistance(self, word: str) -> int:
        firstFinger , secondFinger , dp  = [-1,-1] , [-1,-1] , {}
        return self.minimumDistanceUtil(word , firstFinger , secondFinger , 0 , dp)
