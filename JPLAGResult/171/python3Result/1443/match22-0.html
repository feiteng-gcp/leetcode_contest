class Solution:
    def minimumDistance(self, word: str) -> int:
        def dist(c1, c2):
            x1, y1 = divmod(ord(c1) - ord('A'), 6)
            x2, y2 = divmod(ord(c2) - ord('A'), 6)
            return abs(x1 - x2) + abs(y1 - y2)
        
        memo = {}
        
        def dfs(c1, c2, idx):
            if idx == len(word) - 1:
                return 0
            if (c1, c2, idx) in memo:
                return memo[(c1, c2, idx)]
            memo[(c1, c2, idx)] = min(
                dist(c1, word[idx + 1]) + dfs(word[idx + 1], c2, idx + 1),
                dist(c2, word[idx + 1]) + dfs(c1, word[idx + 1], idx + 1)
            )
            return memo[(c1, c2, idx)]
        
        res = float('inf')
        uniqChars = set(word)
        for c2 in uniqChars:
            res = min(res, dfs(word[0], c2, 0))

        return res
