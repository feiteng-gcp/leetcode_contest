import heapq
class Solution:
    def minimumDistance(self, word: str) -> int:
        dists = [[[float('inf') for i in range(26)] for j in range(26)] for k in range(len(word)+5)]
        word = [ord(i) - 65 for i in word]
        arr = [[] for j in range(5)]
        location = {}


        def dist(x, y, x1, y1):
            return abs(x - x1) + abs(y - y1)


        for i in 'ABCDEFGHIJKLMNOPQRSTUVWXYZ':
            val = ord(i) - 65
            arr[val // 6].append(val)
            location[val] = (len(arr[val // 6]) - 1, val // 6 - 1)
        q = [[0, -1, -1, 0]]
        heapq.heapify(q)
        lens = len(word)
        while q:
            dis, a, b, idx = heapq.heappop(q)

            if idx == lens:
                break
            if a == -1:
                far = 0
            else:
                far = dist(location[a][0], location[a][1], location[word[idx]][0], location[word[idx]][1])

            if dis + far < dists
[idx][word[idx]][b]:
                dists[idx][word[idx]][b] = dis + far
                heapq.heappush(q, [dis + far, word[idx], b, idx + 1])
            if b == -1:
                far = 0
            else:
                far = dist(location[b][0], location[b][1], location[word[idx]][0], location[word[idx]][1])
            if dis + far < dists
[idx][a][word[idx]]:
                dists[idx][a][word[idx]] = dis + far
                heapq.heappush(q, [dis + far,a, word[idx], idx + 1])
        res = float('inf')
        for i in range(26):
            for j in range(26):
                res = min(res, dists[lens-1][i][j])
        return res
