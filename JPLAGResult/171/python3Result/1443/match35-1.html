class Solution:
    def minimumDistance(self, word: str) -> int:
        
def dist(a, b):
            na = ord(a)-ord("A")
            nb = ord(b)-ord("A")
            nax, nay = divmod(na, 6)
            nbx, nby = divmod(nb, 6)
            return abs(nax-nbx)+abs(nay-nby)

                l, r = finger
                d = tot_dist[finger]
                new_tot_dist[(c,r)].append(d + dist(l, c))
                new_tot_dist[(l,c)].append(d + (dist(r, c) if r else 0))
            tot_dist = {k:min(new_tot_dist[k]) for k in new_tot_dist}
        return min(tot_dist.values())
