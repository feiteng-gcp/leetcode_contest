from functools import lru_cache
class Solution:
    def minimumDistance(self, word: str) -> int:
        def dis(ch1, ch2):
            x1, y1 = divmod(ord(ch1)-ord('A'), 6)
            x2, y2 = divmod(ord(ch2)-ord('A'), 6)
            return abs(x2-x1)+abs(y2-y1)
        @lru_cache(None)
        def dp(n, l, r):
            # print(n, l, r)
            if n==-1:
                return 0
            return min(dp(n-1, word[n], r) + dis(l, word[n]), dp(n-1, l, word[n]) + dis(r, word[n]))
        ans=float('inf')
        for i in range(26):
            for j in range(26):
                l = chr(ord('A')+i)
                r = chr(ord('A')+j)
                ans = min(ans, dp(len(word)-1, l, r))
        return ans
