class Solution:
    def minimumDistance(self, word: str) -> int:
        mat = ["ABCDEF", "GHIJKL", "MNOPQR", "STUVWX", "YZ"]
        # n, m = len(mat), len(mat[0])
        pos = collections.defaultdict(tuple)
        n = len(word)
        # for i in range(n)
        
        for i, ch in enumerate("".join(mat)):
            pos[ch] = divmod(i, 6)
            
        def getDist(p1, p2):
            return abs(p1[0] - p2[0]) + abs(p1[1] - p2[1])

        curpos1 = pos[word[0]]
        from functools import lru_cache
        @lru_cache(None)
        def dfs(pos1, pos2, ind):
            # print(pos1, pos2, ind)
            if ind >= n: return 0
            p = pos[word[ind]]
            
            if pos2:
                return min(getDist(pos1, p) + dfs(p, pos2, ind + 1), getDist(pos2, p) + dfs(pos1, p, ind + 1))
            else:
                return min(getDist(pos1, p) + dfs(p, pos2, ind + 1), dfs(pos1, p, ind + 1))
            

        return dfs(curpos1, None, 1)
