class Solution:
    def minimumDistance(self, word: str) -> int:
        from functools import lru_cache
        @lru_cache(None)
        def getMinDist(a: int, b: int, word: str) -> int:
            if not word:
                
return 0
            
            n = ord(word[0]) - ord('A')
            return min(abs(a // 6 - n // 6) + abs(a % 6 - n % 6) + getMinDist(min(b, n), max(b, n), word[1:]), abs(b // 6 - n // 6) + abs(b % 6 - n % 6) + getMinDist(min(a, n), max(a, n), word[1:]))
        
        return min(getMinDist(a, b, word) for a in range(26) for b in range(a, 26))
