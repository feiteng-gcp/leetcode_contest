class Solution:
    def minimumDistance(self, word: str) -> int:
        
        def o(c):
            return ord(c) - ord('A')
        
        def dis(c, p):
            return abs(c % 6 - p % 6) + abs(c // 6 - p // 6) if p != 26 else 0
            
        n = len(word)
        dd = 0        
        dp = [[[float('inf')] * 27 for i in range(27)] for j in range(n + 1)]
        dp[0][26] = [0] * 27
        for i in range(26):
            dp[0][i][26] = 0
        oa = o(word[0])
        dp[1][oa] = [0] * 27
        for i in range(26):
            dp[1][i][oa] = 0
        for i in range(1, n):
            od = o(word[i])
            ood = o(word[i - 1])
#            if i > 0:
#                print(od,o(word[i - 1]),dis(15,15))
            dd += dis(od, ood)
#                print(i,dd)
            for j in range(26):
#                for k in range(27):
                for k in range(27) if j == ood else [ood, 26]:
                    dp
[i + 1][od][j] = min(dp[i + 1][od][j], dp[i][k][j] + dis(od, k))
            dp[i + 1][od][26] = dd
            for j in range(26):
                for k in range(27) if j == ood else [ood, 26]:
                    dp
[i + 1][j][od] = min(dp[i + 1][j][od], dp[i][j][k] + dis(od, k))      
            dp[i + 1][26][od] = dd
