from math import inf

class Solution:
    def minimumDistance(self, word: str) -> int:
        dp = {((-1, -1),(-1, -1)): 0}
        
        def distance(p1, p2):
            return abs(p2[1]-p1[1]) + abs(p2[0]-p1[0])
        
        def position(c):
            return  divmod(ord(c) - ord('A'), 6)
        
        for c in word:
            new_dp = {}
            pos = position(c)
            for k, v in dp.items():
                l, r = k
                o1 = tuple(sorted((pos, r)))
                o2 = tuple(sorted((pos, l)))
                if l[0] == -1:
                    v1 = v
                else:
                    v1 = v+distance(l, pos)
                if r[0] == -1:
                    v2 = v
                else:
                    v2 = v+distance(r, pos)
                new_dp.setdefault(o1, v1)
                new_dp[o1] = min(new_dp[o1], v1)
                new_dp.setdefault(o2, v2)
                new_dp[o2] = min(new_dp[o2], v2)
            dp = new_dp
        return min(dp.values())
