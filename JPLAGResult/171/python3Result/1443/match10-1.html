import functools

class Solution:
    def minimumDistance(self, word: str) -> int:
        ord_A = ord('A')
        ords = [ord(c) - ord_A for c in word]
        
        
def dist(o1, o2):
            a, b = divmod(o1, 6)
            c, d = divmod(o2, 6)
            return abs(a - c) + abs(b - d)
        
        @functools.lru_cache(None)
        def dp(o1, o2, i):
            if i == len(word):
                return 0
            if o2 is None:
                return min(dp(o1, ords[i], i + 1), dist(o1, ords[i]) + dp(ords[i], None, i + 1))
            else:
                return min(dist(o2, ords[i]) + dp(o1, ords[i], i + 1), dist(o1, ords[i]) + dp(ords[i], o2, i + 1))
        return dp(ords[0], None, 1)
