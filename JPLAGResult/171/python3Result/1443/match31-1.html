class Solution:
    def minimumDistance(self, word: str) -> int:
        def cost(a,b):
            return abs(a//6 - b//6) + abs(a%6 - b%6)
        def conv(char):
            return ord(char) - ord('A')
        
        N = len(word)
        dp = [[[0 for _ in range(N+1)] for _ in range(26)] for _ in range(26)]
        for k in range(N-1,-1,-1):
            m = conv(word[k])
            for i in range(26):
                first_finger = cost(i,m)
                for j in range(26):
                    dp[i][j][k] = min(first_finger + dp[m][j][k+1],
                                       cost(j,m) + dp[i][m][k+1])
                   
        ret = min([dp[i][j][0] for i in range(26) for j in range(26)])
        return ret
