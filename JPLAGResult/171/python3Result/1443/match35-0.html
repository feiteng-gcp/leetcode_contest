class Solution:
    def minimumDistance(self, word: str) -> int:
        visited = {}
        def dp(i,j,k):
            if (i,j,k) in visited:
                return visited[(i,j,k)]
            if k == len(word)-1:
                return min(self.distance(i,word[k]),self.distance(j,word[k]))
            visited[(i,j,k)] = min(self.distance(i,word[k])+dp(word[k],j,k+1),self.distance(j,word[k])+dp(i,word[k],k+1))
            return visited[(i,j,k)]
        res = float('inf')
        for i in range(26):
            for j in range(26):
                res = min(res,dp(chr(ord('A')+i),chr(ord('A')+j), 0))
        return res
                
    
    def distance(self,c1,c2):
        o1 = ord(c1) - ord('A')
        o2 = ord(c2) - ord('A')
        dy = abs(o1%6 - o2%6)
        dx = abs(o1//6 - o2//6)
        return dx + dy
