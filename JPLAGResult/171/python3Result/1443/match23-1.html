class Solution:
    def minimumDistance(self, word: str) -> int:
        
        alphs = 'abcdefghijklmnopqrstuvwxyz'.upper()
        
        kb = {c: (i//6, i % 6) for i, c in enumerate(alphs)}
        
def dis(a, b):
            if a == None or b == None:
                return 0
            
            a, b = kb[a], kb[b]
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        w = [None] + list(word)
        dp = {}
        def f(i, o):
            nonlocal dp, w
            if i == len(w):
                return 0
            if (i, o) in dp:
                return dp[(i, o)]
            else:
                v = min([
                    f(i + 1, o) + dis(w[i-1], w[i]),
                    f(i + 1, w[i-1]) + dis(o, w[i])
                ])
                dp[(i, o)] = v
                return v
        return f(1, None)
