class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        a, b, c = bin(a)[2:], bin(b)[2:], bin(c)[2:]
        # print(a, b, c)
        res = 0
        for i in range(1, len(c) + 1):
            if c[-i] == '0':
                if len(a) >= i and a[-i] == '1':
                    res += 1
                if len(b) >= i and b[-i] == '1':
                    res += 1
            else:
                has = False
                if len(a) >= i and a[-i] == '1':
                    has = True
                if len(b) >= i and b[-i] == '1':
                    has = True
                    
                if not has:
                    res += 1
                    
        res += a[:-i].count('1') + b[:-i].count('1')
        return res
