class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        if a | b == c: return 0
        s1, s2, s3 = bin(a)[2:], bin(b)[2:], bin(c)[2:]
        length = max(len(s1), len(s2), len(s3))
        s1 = max(0, length-len(s1))*"0"+s1
        s2 = max(0, length-len(s2))*"0"+s2
        s3 = max(0, length-len(s3))*"0"+s3
        res = 0
        for i in range(length):
            if s3[i] == "0":
                if s1[i] == "1":
                    res += 1
                if s2[i] == "1":
                    res += 1
            if s3[i] == "1":
                if s1[i] == "0" and s2[i] == "0":
                    res += 1
        return res
