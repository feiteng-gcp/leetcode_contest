class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        a, b, c = bin(a)[2:], bin(b)[2:], bin(c)[2:]
        max_len = max(len(a), len(b), len(c))
        a, b, c = "0"*(max_len - len(a)) + a, "0"*(max_len - len(b)) + b, "0"*(max_len - len(c)) + c
        count = 0
        
for i in range(max_len):
            if c[i] == '0':
                count += int(a[i] == '1') + int(b[i] == '1')
            else:
                count += int(a[i] == '0' and b[i] == '0')
        return count
