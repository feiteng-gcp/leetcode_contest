class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        flips = 0
        while a!=0 or b!=0 or c!=0:
            if c%2 == 0:
                flips += a%2 + b%2
            else:
                flips += 1 if (a%2==0 and b%2==0) else 0
            a//=2
            b//=2
            c//=2
        return flips
