class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        res = 0
        while a != 0 or b != 0 or c != 0:
            ra = a %2
            rb 
= b %2
            rc = c %2
            #print (ra,rb,rc)
            if ra != rb:
                if rc == 0:
                    res += 1
            else:
                if ra == 1 and rc == 0:
                    res += 2
                elif ra == 0 and rc == 1:
                    res += 1
            a, b, c = a//2, b//2, c//2
        return res
