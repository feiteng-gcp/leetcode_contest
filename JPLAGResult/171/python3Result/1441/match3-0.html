class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        aorb = a|b
        flip = 0
        
        baorb = bin(aorb)[2:]
        bc = bin(c)[2:]
        ba = bin(a)[2:]
        bb = bin(b)[2:]
        
        baorb = '0'*(33-len(baorb)) + baorb
        bc = '0'*(33-len(bc)) + bc
        ba = '0'*(33-len(ba)) + ba
        bb = '0'*(33-len(bb)) + bb
        
        for i in range(len(bc)):
            if bc[i] == baorb[i]:
                continue
            if baorb[i] == '0' and bc[i] == '1':
                flip += 1
            elif baorb[i] == '1' and bc[i] == '0':
                if ba[i] == bb[i] == '1':
                    flip += 2
                else:
                    flip += 1
        return flip
