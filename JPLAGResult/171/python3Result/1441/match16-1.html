class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        ret = 0
        while a > 0 or b > 0 or c > 0:
            x = a % 2
            y = b % 2
            z = c % 2
            if z == 0:
                ret += x + y
            else:
                if x == 0 and y == 0:
                    ret += 1
            # print(x, y, z, ret)
            a //= 2
            b //= 2
            c //= 2
        return ret
