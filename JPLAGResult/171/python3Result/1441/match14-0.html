class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        n = 0
        while not all([c == 0, a == 0, b == 0]) :
            if a % 2 | b % 2 == c % 2 :
                pass
            elif a % 2 + b % 2 == 0 and c % 2 == 1 :
                n += 1
            elif a % 2 + b % 2 == 1 and c % 2 == 0 :
                n += 1
            elif a % 2 + b % 2 == 2 and c % 2 == 0 :
                n += 2
            a = a // 2
            b = b // 2
            c = c // 2
        return n
