class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        result = 0
        for idx in range(0, 32):
            if c & (1 << idx):
                if not ((a & (1 << idx)) or (b & (1 << idx))):
                    result += 1
            else:
                if a & (1 << idx): 
                    result += 1
                if b & (1 << idx): 
                    result += 1
        return result
