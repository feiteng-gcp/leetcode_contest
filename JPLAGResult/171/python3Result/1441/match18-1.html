class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        x=bin(a)[2:]
        y=bin(b)[2:]
        z=bin(c)[2:]
        length = max(len(x), len(y), len(z))
        x=x.zfill(length)
        y=y.zfill(length)
        z=z.zfill(length)
        ans=0
        # print(x,y,z)
        for i in range(length):
            if z[i]=='0':
                ans += (x[i]=='1') + (y[i]=='1')
            else:
                ans += (x[i]=='0' and y[i]=='0')
        return ans
