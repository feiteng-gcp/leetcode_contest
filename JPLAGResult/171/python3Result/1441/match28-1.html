class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        a = bin(a)[2:][::-1]
        b = bin(b)[2:][::-1]
        c = bin(c)[2:]
[::-1]
        
        maxl = max(len(a), len(b), len(c))
        a = a + '0'*(maxl-len(a))
        b = b + '0'*(maxl-len(b))
        c = c + '0'*(maxl-len(c))
        
        res = 0
        for i in range(maxl):
            if c[i] == '1':
                res += a[i] != '1' and b[i] != '1'
            else:
                res += int(a[i]) + int(b[i])
        return res
