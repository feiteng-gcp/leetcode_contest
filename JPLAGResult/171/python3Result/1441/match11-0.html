class Solution:
    def convert(self, num):
        arr = []
        for i in range(32):
            arr.append(num % 2)
            num //= 2
        return arr
    
    def minFlips(self, a: int, b: int, c: int) -> int:
        aarr = self.convert(a)
        barr = self.convert(b)
        carr = self.convert(c)

        print(aarr, barr, carr)
        
        cnt = 0
        for i in range(32):
            if carr[i]:
                if not aarr[i] and not barr[i]:
                    cnt += 1
            else:
                if aarr[i]:
                    cnt += 1
                if barr[i]:
                    cnt += 1
        return cnt
