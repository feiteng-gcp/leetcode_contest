class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        a = bin(a)[2:][::-1]
        b = bin(b)[2:][::-1]
        c = bin(c)[2:][::-1]
        # print(a, b, c)
        l = max([len(x) for x in [a, b, c]])
        res = 0
        for i in range(l):
            a1 = b1 = c1 = 0
            if i < len(a) and a[i] == "1":
                a1 = 1
            if i < len(b) and b[i] == "1":
                b1 = 1
            if i < len(c) and c[i] == "1":
                c1 = 1
            
            if c1 == 0:
                if a1 == 1:
                    res += 1
                if b1 == 1:
                    res += 1
            if c1 == 1 and b1 != 1 and a1 != 1:
                res += 1
        return res
