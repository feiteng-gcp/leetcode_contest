class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        A = list(bin(a))[2:]
        B = list(bin(b))[2:]
        C = list(bin(c))[2:]
        n = max(len(A),len(B),len(C))
        A = ['0']*(n-len(A))+A
        B = ['0']*(n-len(B))+B
        C = ['0']*(n-len(C))+C
        
        ans = 0
        for a,b,c in zip(A,B,C):
            if a==b==c=='0' or (c=='1' and '1' in (a,b)):
                continue
            if c=='0' and a==b=='1':
                ans+=2
            else:
                ans+=1
        return ans
