class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        a_bin = bin(a)[2:][::-1]
        b_bin = bin(b)[2:][::-1]
        c_bin = bin(c)[2:][::-1]
        solution = 0
        for i in range(len(c_bin)):
            count 
= 0
            if i < len(a_bin) and a_bin[i] == '1':
                count += 1
            if i < len(b_bin) and b_bin[i] == '1':
                count += 1
            if c_bin[i] == '1' and count == 0:
                solution += 1
            elif c_bin[i] == '0':
                solution += count
        for i in range(len(c_bin), len(a_bin)):
            solution += a_bin[i]=='1'
        for i in range(len(c_bin), len(b_bin)):
            solution += b_bin[i] =='1'
        return solution
