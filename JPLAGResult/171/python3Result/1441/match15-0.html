class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        ret = 0
        while a + b + c > 0:
            x, y, z = a % 2, b % 2, c % 2
            if x | y != z:
                if z == 0: ret += (x == 1) + (y == 1)
                if z == 1: ret += 1
            a, b, c = a // 2, b // 2, c // 2
        return ret
