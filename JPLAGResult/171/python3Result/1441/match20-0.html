class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        a, b, c = bin(a)[2:], bin(b)[2:], bin(c)[2:]
        L = max(len(a), len(b), len(c))
        a, b, c = '0'*(L-len(a))+a, '0'*(L-len(b))+b, '0'*(L-len(c))+c
        
        count = 0
        for i in range(L): 
            if c[i] == '0': 
                if a[i] != b[i]:
                    count += 1
                elif a[i] == '1' and b[i] == '1':
                    count += 2
            elif a[i] == '0' and b[i] == '0'  and c[i] == '1': 
                count += 1
        return count
