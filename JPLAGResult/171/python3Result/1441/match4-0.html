class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        mask = 1
        res = 0
        while mask <= max(a, b, c):
            target = (c & mask) == 0
            x = (a & mask) == 0
            y = (b & mask) == 0
            
            if not target:
                if x and y:
                    res += 1
            else:
                if not x or (not y):
                    res += 2 - x - y
            # print(mask, target, x, y, res)
            mask <<= 1
            
        return res
