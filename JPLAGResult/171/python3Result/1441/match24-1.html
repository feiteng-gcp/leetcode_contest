class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        a = bin(a)[2:]
        b = bin(b)[2:]
        c = bin(c)[2:]
        n =max(len(a), len(b), len(c))
        a = '0' * (n - len(a)) + a
        b = '0' * (n - len(b)) + b
        c = '0' * (n - len(c)) + c
        
        steps = {
            '000': 0,
            '001': 1,
            '010': 1,
            '011': 0,
            '100': 1,
            '101': 0,
            '110': 2,
            '111': 0
        }
        
        ans = 0
        for i in range(n):
            pattern = a[i] + b[i] + c[i]
            ans += steps[pattern]
        
        return ans
