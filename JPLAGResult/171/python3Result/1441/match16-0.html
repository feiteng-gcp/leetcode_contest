class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        count = 0
        while c > 0:
            c1 = c % 2
            a1 = a % 2
            b1 = b % 2
            
            if c1 == 0:
                count += a1 + b1
            else:
                if a1 + b1 == 0:
                    count += 1
            
            c //= 2
            a //= 2
            b //= 2
        
        while a > 0:
            if a % 2 == 1:
                count += 1
            a //= 2
        while b > 0:
            if b % 2 == 1:
                count += 1
            b //= 2
        return count
