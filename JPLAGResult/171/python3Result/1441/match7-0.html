class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        ans = 0
        mask = 1
        while a >= mask or b >= mask or c >= mask:
            k1 = (a & mask != 0)
            k2 = (b & mask != 0)
            if c & mask:
                if k1 or k2:
                    pass
                else:
                    ans += 1
            else:
                if k1 or k2:
                    ans += k1
                    ans += k2
                else:
                    pass
            mask *= 2
        return ans
