class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        def binary(num):
            s = "{0:b}".format(num)
            return '0' * (64 - len(s)) + s
        a = binary(a)
        b = binary(b)
        c = binary(c)
        # print(a)
        total = 0
        for ia, ib, ic in zip(a, b, c):
            if ic == '1':
                if ia == '0' and ib == '0':
                    total += 1
            else:
                if ia == '1':
                    total += 1
                if ib == '1':
                    total += 1
        return total
