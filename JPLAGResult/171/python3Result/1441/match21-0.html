class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        a = format(a, 'b')[::-1]
        b = format(b, 'b')[::-1]
        c = format(c, 'b')[::-1]
        print(a, b, c)
        n = len(c)
        ans = 0
        if len(a) > n:
            for d in a[n:]:
                if d == '1':
                    ans += 1
        else:
            a += '0' * (n - len(a))
        if len(b) > n:
            for d in b[n:]:
                if d == '1':
                    ans += 1
        else:
            b += '0' * (n - len(b))
        for i in range(n):
            if c[i] == '0':
                ans += int(a[i] == '1') + int(b[i] == '1')
            else:
                ans += int(a[i] == '0' and b[i] == '0')
        return ans
