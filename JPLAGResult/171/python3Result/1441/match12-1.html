class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        def convert(x):
            return bin(x)[2:][::-1]
        a,b,c = convert(a),convert(b),convert(c)
        MAX 
= max(len(a),len(b),len(c))
        a += '0' * (MAX-len(a))
        b += '0' * (MAX-len(b))
        c += '0' * (MAX-len(c))
        count = 0
        for i in range(MAX):
            if c[i] == '1':
                if b[i] == '0' and  a[i] == '0':
                    count += 1
            else:
                if b[i] == '1':
                    count += 1
                if a[i] == '1':
                    count += 1
        return count
