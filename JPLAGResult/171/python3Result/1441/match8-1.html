class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        result = 0
        for i in range(30):
            mask = 1 << i
            
if (c & mask):
                if not ((a & mask) or (b & mask)):
                    result += 1
            else:
                if a & mask:
                    result += 1
                if b & mask:
                    result += 1
        return result
