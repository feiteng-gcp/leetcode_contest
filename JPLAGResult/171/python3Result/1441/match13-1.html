class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        ret=0
        while a>0 or b>0 or c>0:
            a1,b1,c1
=a%2,b%2,c%2
            if c1==1 and a1==0 and b1==0:
                ret+=1
            elif c1==0:
                ret+=a1+b1
            a//=2
            b//=2
            c//=2
        return ret
