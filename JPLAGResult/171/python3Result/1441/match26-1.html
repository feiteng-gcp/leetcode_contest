class Solution:
    def binary(self,n):
        ans = []
        for i in range(40):
            ans.append(n%2)
            n //= 2
        return ans
        
    def minFlips(self, a: int, b: int, c: int) -> int:
        lst_a,lst_b,lst_c = self.binary(a),self.binary(b),self.binary(c)
        
        ans = 0
        
        for i in range(40):
            if lst_c[i] == 1:
                if lst_a[i] == 0 and lst_b[i] == 0:
                    ans += 1
            else:
                ans += lst_a[i] + lst_b[i]
        
        return ans
