class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        def get_2(n):
            two=[]
            while n>0:
                two=[n%2]+two
                n=n//2
            return two
        a=get_2(a)
        b=get_2(b)
        c=get_2
(c)
        maxl=max(len(a),len(b),len(c))
        a=[0]*(maxl-len(a))+a
        b=[0]*(maxl-len(b))+b
        c=[0]*(maxl-len(c))+c
        ans=0
        for i in range(maxl):
            if c[i]==1:
                if a[i]==0 and b[i]==0:
                    ans+=1
            else:
                if a[i]==1:
                    ans+=1
                if b[i]==1:
                    ans+=1
        return ans
