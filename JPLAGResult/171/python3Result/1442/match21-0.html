class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        if len(connections) < n - 1: return -1
        g = collections.defaultdict(set)
        for a, b in connections:
            g[a].add(b)
            g[b].add(a)
        visited = [False] * n
        ret = -1
        def dfs(u):
            if visited[u]: return
            visited[u] = True
            for v in g[u]: dfs(v)
        for i in range(n):
            if not visited[i]:
                ret += 1
                dfs(i)
        return ret
