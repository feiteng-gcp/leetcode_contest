class DSU:
    def __init__(self, n):
        self.par = list(range(n))
    
    def find(self, x):
        if self.par[x] != x:
            self.par[x] = self.find(self.par[x])
        return self.par[x]
    
    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        self.par[px] = self.par[py]
        

class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        N = len(connections)
        if N < n-1:
            return -1
        
        dsu = DSU(n)
        for edge in connections:
            dsu.union(*edge)
        return sum(dsu.find(x) == x for x in range(n))-1
