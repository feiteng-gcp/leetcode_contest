class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        adj = collections.defaultdict(list)
        
        for x, y in connections:
            adj[x].append(y)
            adj[y].append(x)
            
        edges = len(connections)
        
        seen = set()
        count = 0
        def bfs(x):
            seen.add(x)
            q = [x]
            
            while len(q) > 0:
                now = q.pop()
                
                for y in adj[now]:
                    if y not in seen:
                        seen.add(y)
                        q.append(y)
            
        if edges < n - 1:
            return -1
        
        for i in range(n):
            if i not in seen:
                bfs(i)
                count += 1

        return count - 1
