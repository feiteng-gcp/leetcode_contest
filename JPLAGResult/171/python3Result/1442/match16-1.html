class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        UF = {}
        if len(connections) < n - 1:
            return -1
        def find(i):
            UF.setdefault(i,i)
            if UF
[i] != i:
                UF[i] = find(UF[i])
            return UF[i]
        def union(i,j):
            UF.setdefault(i,i)
            UF.setdefault(j,j)
            UF[find(i)] = find(j)
            
        for i,j in connections:
            union(
i,j)
        visited  = set()
        for i in range(n):
            visited.add(find(i))
        return len(visited) - 1
