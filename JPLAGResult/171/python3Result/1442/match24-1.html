class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        if len(connections) < n-1:
            return -1
        self.visited = [0] * n
        adj = [[] for i in range(n)]
        for a,b in connections:
            adj[a].append(b)
            adj[b].append(a)
        self.adj = adj
        components 
= 0
        for i in range(n):
            if not self.visited[i]:
                self.bfs(i)
                components += 1
        return components - 1
                
    def bfs(self, start):
        q = [start]
        while q:
            i = q.pop()
            if self.visited[i]: continue
            self
.visited[i] = 1
            for j in self.adj[i]:
                if not self.visited[j]:
                    q.append(j)
