class Solution:
    def makeConnected(self, n: int, edges: List[List[int]]) -> int:
        if len(edges) < n-1: return -1
    
        visited = [0] * n
        graph = collections.defaultdict(list)
        for u, v in edges:
            graph[u].append(v)
            graph[v].append(u)
            
        def dfs(node):
            if visited[node]: return
            visited[node] = 1
            for nei in graph[node]: dfs(nei)
        
        count = 0
        for i in range(len(visited)):
            if not visited[i]:
                dfs(i)
                count += 1
        return count-1
