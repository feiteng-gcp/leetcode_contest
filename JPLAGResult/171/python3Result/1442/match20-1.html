class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        if len(connections) < n - 1:
            return -1
        
        # how many connected groups
        connect 
= collections.defaultdict(list)
        for a, b in connections:
            connect[a].append(b)
            connect[b].append(a)
        
        visited = set()
        group = 0
        for i in range(n):
            if i not in visited:
                group += 1
                visited.add(i)
                queue = [i]
                while queue:
                    now 
= queue.pop(0)
                    for k in connect[now]:
                        if k not in visited:
                            visited.add(k)
                            queue.append(k)
        return group - 1
