class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        if len(connections) < n - 1:
            return -1
        
        parents = {}

        def find(p): 

            if p not in parents:
                parents[p] = p
            if parents[p] != p:
                parents[p] = find(parents[p])
            return parents[p]

        def union(p, q):
            i, j = find(p), find(q)
            parents[i] = j

        def connected(p, q):
            return find(p) == find(q)
        
        for p, q in connections:
            union(
p, q)
        
        res = set()
        for p in range(n):
            res.add(find(p))
        return len(res) - 1
