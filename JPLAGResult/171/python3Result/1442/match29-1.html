class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        if len(connections) < n-1:
            return -1
        parents = list(range(n))
        def find(x):
            while x != parents[x]:
                x = parents[x]
            return x
        for x, y in connections:
            px, py = find(x), find(y)
            
if px != py:
                parents[px] = py
        p = set()
        for i in range(n):
            p.add(find(i))
        return len(p) - 1
