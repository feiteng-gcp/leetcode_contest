class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        pa = [i for i in range(n)]
        rank = [1 for i in range(n)]
        
        def Find(x):
            return x if pa[x] == x else Find(pa[x])
        
        def Union(x, y):
            rootx, rooty = Find(x), Find(y)
            if rootx == rooty: return False
            if rank[rootx] > rank[rooty]: rootx, rooty = rooty, rootx
            pa[rootx] = rooty
            rank[rooty] += rank[rootx]
            return True
        
        seen = set()
        redun = 0
        for x, y in connections:
            seen.add(x)
            seen.add(y)
            if not Union(x, y): redun += 1
        
        for i in range(n):
            pa[i] = Find(i)
        
        parts = len(set(pa))
        # print(pa)
        need = parts - 1
        # need = n - len(seen)
        # print(need, redun)
        if need > redun: return -1
        return need
