class DSU:
    def __init__(self, n):
        self.p = list(range(n))
    def find(self, x):
        if x != self.p[x]:
            self.p[x] = self.find(self.p[x])
        return self.p[x]
    def union(self, x, y):
        xp = self.find(x)
        yp = self.find(y)
        self.p[xp] = yp
class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        l = len(connections)
        if l < n - 1:
            return -1
        dsu = DSU(n)
        for n1, n2 in connections:
            dsu.union(n1, n2)
        clusters = set()
        for i in range(n):
            clusters.add(dsu.find(i))
        return len(clusters) - 1
