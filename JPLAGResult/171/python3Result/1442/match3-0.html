class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        clen = len(connections)
        if clen < n - 1: return -1
        graph = dict()
        for c in connections:
            if c[1] in graph:
                graph[c[1]].append(c[0])
            else:
                graph[c[1]] = [c[0]]
        
            if c[0] in graph:
                graph[c[0]].append(c[1])
            else:
                graph[c[0]] = [c[1]]
            
        
        componets = 0
        visited = set()
        
        
        for i in range(n):
            if i in visited: continue
            componets += 1
            if i not in graph: continue

            q.append(i)
            visited.add(i)
            while q:
                cnode = q.popleft()
                for nei in graph[cnode]:
                    if nei not in visited:
                        visited.add(nei)
                        q.append(nei)
        return componets - 1
