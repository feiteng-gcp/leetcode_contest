class Solution:
    def dfs(self, t):
        self.vis[t] = True
        for nt in self.edges[t]:
            if not self.vis[nt]:
                self.dfs(nt)
    
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        m = len(connections)
        if n > m + 1:
            return -1
        
        self.n = n
        self.edges = [[] for _ in range(n)]
        for a, b in connections:
            self.edges[a].append(b)
            self.edges[b].append(a)
        
        self.vis = [False for _ in range(n)]
        cnt = 0
        for i in range(n):
            if not self.vis[i]:
                self.dfs(i)
                cnt += 1
        return cnt - 1
