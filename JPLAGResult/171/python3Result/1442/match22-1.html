class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        # count the number of connected components, nbr_components, return nbr_components - 1
        if len(connections) < n - 1:
            return -1
        nbr_components = 0
        
        adj 
= collections.defaultdict(list)
        
        for a, b in connections:
            adj[a].append(b)
            adj[b].append(a)
        
        visited = set()
        
        def bfs(s: int) -> None:
            queue = collections.deque()
            queue.append(s)
            visited.add(s)
            while queue:
                for _ in range(len
(queue)):
                    cur = queue.popleft()
                    for nei in adj[cur]:
                        if nei not in visited:
                            queue.append(nei)
                            visited.add(nei)
        
        for node in range(n):
            if node not in visited:
                bfs(node)
                nbr_components += 1
        
        return nbr_components - 1
