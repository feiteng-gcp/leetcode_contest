class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        # 并查集
        if len(connections)<n-1:
            return -1
        pre = {}
        def find(a):
            if a not in pre:
                pre[a] = a
            elif pre[a] != a:
                pre[a] = find(pre[a])
            return pre[a]
        def union(a, b):
            pre[find(a)] = find(b)
        for c in connections:
            union(c[0],c[1])
        s = set()
        for i in range(n):
            s.add(find(i))
        return len(s)-1
