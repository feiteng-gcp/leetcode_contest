class Solution:
    def makeConnected(self, n: int, connections: 'List[List[int]]') -> int:
        if len(connections)<n-1:
            return -1
        anc=[i for i in range(n)]
        def findanc(n):
            topanc=n
            while anc[topanc]!=topanc:
                topanc=anc[topanc]
            cur=n
            while anc[cur]!=topanc:
                nx=anc[cur]
                anc[cur]=topanc
                cur=nx
            return topanc
        for c in connections:
            an1,an2=findanc(c[0]),findanc(c[1])
            if an1!=an2:
                anc[an1]=an2
        ss=set()
        for i in range(n):
            ss.add(findanc(i))
        return len(ss)-1
