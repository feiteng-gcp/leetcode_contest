class Solution:
    def makeConnected(self, n: int, connections: List[List[int]]) -> int:
        if len(connections)<n-1: return -1
        graph = [[] for i in range(n)]
        for a,b in connections:
            graph[a].append(b)
            graph[b].append(a)
        visited = [False] * n
        def DFS(now):
            visited[now] = True
            for v in graph[now]:
                if visited[v]==False:
                    DFS(v)
        ans = 0
        for i in range(n):
            if not visited[i]:
                DFS(i)
                ans += 1
        return ans-1
