class Solution {
    public int minFlips(int a, int b, int c) {
        int[] aBin = toBinary(a);
        int[] bBin = toBinary(b);
        int[] cBin = toBinary(c);
        
        int total = 0;
        for (int i =0; i < 31; i++) {
            if (cBin[i] == 1) {
                if (!(aBin[i] == 1 || bBin[i] == 1)) {
                    total += 1;
                }
            } else {
                if (aBin[i] == 1 && bBin[i] == 1) {
                    total += 2;
                } else if ((aBin[i] == 1 || bBin[i] == 1)) {
                    total += 1;
                }
            }
        }
        return total;
    }
    
    public int[] toBinary(int a) {
        int[] binary = new int[32];
        int i = 0;
        while(a > 0) {
            binary[i++] = a%2;
            a = a/2;
        }
        return binary;
    }
}
