class Solution {
    public int minFlips(int a, int b, int c) {
        int res = 0;
        for (int i = 0; i < 31; i++) {
            int x = (a >> i) & 1, y = (b >> i) & 1, z = (c >> i) & 1;
            if (z == 0) {
                if (x == 1) res++;
                if (y == 1) res++;
            } else {
                if (x == 0 && y == 0) res++;
            }
        }
        return res;
    }
}
