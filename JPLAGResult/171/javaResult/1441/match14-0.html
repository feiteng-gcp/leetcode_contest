class Solution {
    public int minFlips(int a, int b, int c) {
        int res = 0;

        while (a != 0 || b != 0 || c != 0) {
            int abit = a & 1;
            int bbit = b & 1;
            int cbit = c & 1;

            if (cbit != 0) {
                if (abit == 0 && bbit == 0) res++;
            } else {
                if (abit != 0) res++;
                if (bbit != 0) res++;
            }

            a >>= 1;
            b >>= 1;
            c >>= 1;
        }

        return res;
    }
}
