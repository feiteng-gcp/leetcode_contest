class Solution {
    public int minFlips(int a, int b, int c) {
        int counter = 0;
        
        for(int i = 0; i <= 30; i++){
            int da = (a >> i) & 1;
            int db = (b >> i) & 1;
            int dc = (c >> i) & 1;
                
            if (dc == 0){
                if (da != 0) counter++;
                if (db != 0) counter++;
            } else {
                if (da == 0 && db == 0) counter++;
            }
            // System.out.println(i+" "+da+" "+db+" "+dc+" "+counter);
        }
        return counter;
    }
}
