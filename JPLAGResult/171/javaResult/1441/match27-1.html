class Solution {
    public int minFlips(int a, int b, int c) {
        int res = 0;
        while(a != 0 || b != 0 || c != 0) {
            int la = a & 0x1;
            int lb = b & 0x1;
            int lc = c & 0x1;
            
if(lc == 1) {
                if(la == 0 && lb == 0) ++res;
            } else {
                if(la == 1) ++res;
                if(lb == 1) ++res;
            }
            a >>= 1;
            b >>= 1;
            c >>= 1;
        }
        return res;
    }
}
