class Solution {
    public int minFlips(int a, int b, int c) {
        int res = 0, id = 0;
        while(id < 32){
            int d1 = (a & (1 << id)) == 0 ? 0 : 1;
            int d2 = (b & (1 << id)) == 0 ? 0 : 1;
            int tar = (c & (1 << id)) == 0 ? 0 : 1;
            if((d1 | d2) != tar) {
                if(tar == 0 && d1 == 1 && d2 == 1){
                    res += 2;
                } else {
                    res++;
                }
            }
            id++;
        }
        return res;
    }
}
