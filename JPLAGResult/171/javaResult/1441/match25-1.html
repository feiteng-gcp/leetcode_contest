class Solution {
    public int minFlips(int a, int b, int c) {
        int[] at = toTwo(a);
        int[] bt = toTwo(b);
        int[] ct = toTwo(c);
        int tot = 0;
        for(int i = 0; i < 32; ++i) {
            if (ct[i] == 1) {
                if (at[i] == 0 && bt[i] == 0) {
                    ++tot;
                }
            } else {
                if (at[i] == 1 || bt[i] == 1) {
                    tot += at[i] + bt[i];
                
}
            }
        }
        return tot;
    }
    
    private int[] toTwo(int n) {
        int[] arr = new int[32];
        int l = 0;
        while(n != 0) {
            arr[l++] = n % 2;
            n = n / 2;
        }
        return arr;
    }
}
