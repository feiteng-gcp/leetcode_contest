class Solution {
    
    public int minFlips(int a, int b, int c) {
        int r = 0;
        for (int i = 0; i < 32; i++) {
            int ia = (a >> i) & 1;
            int ib = (b >> i) & 1;
            int ic = (c >> i) & 1;
            if (ic == (ia | ib)) {
                continue;
            } else if (ic == 1) {
                // must be 0 | 0; flip any of them
                r++;
            } else {
                // c==0
                // either 1|1 or 1|0
                if (ia == 1 && ib == 1) {
                    r += 2;
                } else {
                    r++;
                }
            }
        }
        return r;
    }
}
