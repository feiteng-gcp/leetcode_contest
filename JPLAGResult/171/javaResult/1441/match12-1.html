class Solution {
    public int minFlips(int a, int b, int c) {
        StringBuilder A =  new StringBuilder(Integer.toBinaryString(a)),B =  new StringBuilder(Integer.toBinaryString(b)),C =  new StringBuilder(Integer.toBinaryString(c));
        A.reverse(); B.reverse(); C.reverse();
        // System.out.println(A +" "+B +" "+C);
        int i = 0, res = 0;
        for( i = 0; i < A.length() && i < B.length() && i < C.length() ; i++){
            if(C.charAt(i) == '0'){
                if(A.charAt(i) == '1') res++;
                if(B.charAt(i) == '1') res++;
            }
            
            if(C.charAt(i) == '1'){
                if(A.charAt(i) == '0' && B.charAt(i) == '0') res++;
            }
        }
        // System.out.println(res);
        if(i == C.length()){
            for( int j = C.length(); j < A.length(); j++) if(A.charAt(j) == '1') res++;
            for( int j = C.length(); j < B.length(); j++) if(B.charAt(j) == '1') res++;
        }
        if(A.length() > B.length()){
            StringBuilder temp = A;
            A = B;
            B = temp;            
        }
        if(i == A.length() && i != C.length()){
            int j = A.length();
            
            for(; j < Math.min(B.length(), C.length()); j++) if(B.charAt(j) != C.charAt(j)) {
                // System.out.println("t1");
                res++;
            }
            // System.out.println(res);
            if(j == C.length() && j < B.length()) for(; j < B.length(); j++) if(B.charAt(j) == '1') {
                // System.out.println("t2");
                res++;
            }
            // System.out.println(res);
            if(j < C.length()) for(; j < C.length(); j++) if(C.charAt(j) == '1') {
                // System.out.println("t3");
                res++;
            }
            
        }
        return res;
        
    }
}
