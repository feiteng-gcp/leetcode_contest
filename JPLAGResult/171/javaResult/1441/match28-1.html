class Solution {
    public int minFlips(int a, int b, int c) {
        int res = 0;
        for (int i = 0; i <= 31; i++) {
            int x = (1 << i);
            int p1 = (a & x), p2 = (b & x), p3 = (c & x);
            
if (p3 == 0) {
                if (p1 != 0) res++;
                if (p2 != 0) res++;
            } else {
                if (p1 == 0 && p2 == 0) res++;
            }
            
        }
        return res;
    }
}
