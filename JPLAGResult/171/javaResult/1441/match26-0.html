class Solution {
    public int minFlips(int a, int b, int c) {
        int count = 0;
        while (c > 0 || a > 0 || b > 0) {
            int r1 = a % 2;
            int r2 = b % 2;
            int r3 = c % 2;
            
            if (r3 == 0) {
                if (r1 == 1) count++;
                if (r2 == 1) count++;
            } else { // r3 == 1
                if (r1 == 0 && r2 == 0) count++;
            }
            
            a /= 2;
            b /= 2;
            c /= 2;
        }
        return count;
    }
}
