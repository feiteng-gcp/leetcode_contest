class Solution {
    public int minFlips(int a, int b, int c) {
        int res = 0;
        for(int i=0;i<31;i++)
        {
            // System.out.println("i=" + i);
            int mask = 1<<i;
            
if((c&mask) == 0)
            {
                // System.out.println("0");
                if((a&mask) != 0)
                {
                    res++;
                }
                if((b&mask) != 0)
                {
                    res++;
                }
            }
            else
            {
                if(((a&mask) == 0)&&((b&mask) == 0))
                    res++
;
            }
        }
        
        return res;
    }
}
