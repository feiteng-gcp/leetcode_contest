class Solution {
    public int minFlips(int a, int b, int c) {
        int count = 0;
        for (int i = 0; i < 32; i++) {
            int as = a & 1;
            int bs = b & 1;
            int cs = c & 1;
            if (cs == 1) {
                if (as == 0 && bs == 0) {
                    count++;
                }
            } else {
                if (as == 1) {
                    count++;
                }
                if (bs == 1) {
                    count++;
                }
            }
            a >>= 1;
            b >>= 1;
            c >>= 1;
        }
        return count;
    }
}
