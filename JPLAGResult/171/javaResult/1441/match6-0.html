class Solution {
    public int minFlips(int a, int b, int c) {
        int res = 0;
        String target = convert(c);
        String A = convert(a);
        String B = convert(b);
        // System.out.println(target);
        // System.out.println(A);
        // System.out.println(B);
        int index = 0;
        while (index < Math.max(target.length(), Math.max(A.length(), B.length()))) {
            int t = index < target.length() ? target.charAt(index) - '0': 0;
            int f = index < A.length() ? A.charAt(index) - '0' : 0;
            int s = index < B.length() ? B.charAt(index) - '0': 0;
            
            if (t == 1) {
                if (f == 1 || s == 1) {
                    index++;
                    continue;
                }
                res++;
            }
            else {
                res += f + s;
            }
            index++;
        }
        return res;
    }
    
    private String convert(int i) {
        StringBuilder sb = new StringBuilder();
        while (i > 0) {
            sb.append(i % 2);
            i /= 2;
        }
        return sb.toString();
    }
}
