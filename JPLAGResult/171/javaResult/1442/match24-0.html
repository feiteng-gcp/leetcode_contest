class Solution {

    class UF {
      int n;
      int cc;
      int[] p;

      UF(int n) {
        this.n = n;
        this.cc = n;
        p = new int[n];
        for(int i=0; i<n; i++) p[i] = i;
      }

      int find(int a) {
        if(p[a] == a) return a;
        int r = find(p[a]);
        p[a] = r;
        return r;
      }

      void union(int a, int b) {
        int ap = find(a);
        int bp = find(b);
        if(ap != bp) {
          p[ap] = bp;
          cc--;
        }
      }
    }

    public int makeConnected(int n, int[][] connections) {
      if(connections.length < n-1) return -1;
      UF uf = new UF(n);
      for(int[] c : connections) {
        uf.union(c[0], c[1]);
      }
      return uf.cc-1;
    }
  }
