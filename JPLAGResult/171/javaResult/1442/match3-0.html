class Solution {
    public int makeConnected(int n, int[][] connections) {
        UnionFind uf = new UnionFind();
        for(int[] con : connections) uf.union(con[0], con[1]);
        Set<Integer> set = new HashSet<>();
        for(int i = 0; i < n; i++) set.add(uf.find(i));
        int parts  = set.size();
        
        if(parts -1 <= uf.left) return parts-1;
        return -1;
    }
    
    class UnionFind{
        Map<Integer, Integer> map = new HashMap<>();
        int left = 0;
        int find(int n){
            map.putIfAbsent(n,n);
            while(n != map.get(n)){
                map.put(n, map.get(map.get(n)));
                n = map.get(n);
            }
            return n;
        }
        
        void union(int a, int b){
            if(find(a) == find(b)) left ++;
            else map.put(find(a), find(b));
        }
    }
}
