class Solution {
   public int makeConnected(int n, int[][] connections) {
        if(connections.length<n-1){
            return -1;
        }
        int[] parent=new int[n];
        for(int i=0;i<n;i++){
            parent[i]=i;
        }
        for(int[] con:connections){
            int a=con[0];
            int b=con[1];
            int aPa= getRoot(a,parent);
            int bPa=getRoot(b,parent);
            if(aPa!=bPa){
                parent[aPa]=bPa;
            }
        
}
        int cnt=0;
        for(int i=0;i<n;i++){
            if(parent[i]==i){
                cnt++;
            }
        }
        return cnt-1;
    }

    private int getRoot(int a, int[] parent) {
        if(parent[a]==a){
            return a;
        }
        int root= getRoot(parent[a],parent);
        parent[a]=root;
        return root;
    }
}
