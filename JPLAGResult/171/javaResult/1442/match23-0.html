class Solution {
    public int makeConnected(int n, int[][] connections) {

        if (connections.length < n - 1) {
            return -1;
        }

        UnionFind uf = new UnionFind(n);
        for (int[] e : connections) {
            int v = e[0];
            int w = e[1];
            uf.unite(v, w);
        }

        return uf.cnt - 1;

    }

    private class UnionFind {
        private int[] root;
        private int cnt;

        private UnionFind(int cap) {
            root = new int[cap];
            for (int i = 0; i < cap; i++) {
                root[i] = i;
            }
            cnt = cap;
        }

        private int find(int s) {
            while (root[s] != s) {
                root[s] = root[root[s]];
                s = root[s];
            }
            return s;
        }

        private void unite(int s, int t) {
            int rs = find(s);
            int rt = find(t);
            if (rs != rt) {
                cnt--;
                root[rs] = rt;
            }
        }
    }
}
