class Solution {
    class UnionFind {
        int[] f;
        int[] cnts;
        int cnt;
        public UnionFind(int N) {
            f = new int[N];
            Arrays.fill(f, -1);
            cnts = new int[N];
            Arrays.fill(cnts, 1);
            cnt = N;
        }
        
        public int find(int n) {
            int cur = n;
            while (f[cur] != -1) {
                cur = f[cur];
            }
            if (cur != n) f[n] = cur;
            
return cur;
        }
        
        public void union(int a, int b) {
            int pa = find(a);
            int pb = find(b);
            
            if (pa == pb) return;
            
            if (cnts[pa] > cnts[pb]) {
                f[pb] = pa;
                cnts[pa] += cnts[pb];
                cnts[pb] = 0;
            } else {
                f[pa] = pb;
                cnts[pb] += cnts[pa];
                
cnts[pa] = 0;
            }
            cnt--;
        }
    }
    public int makeConnected(int n, int[][] connections) {
        if (n - 1 > connections.length) return -1;
        
        UnionFind uf = new UnionFind(n);
        for (int[] edge : connections) {
            uf.union(edge[0], edge[1]);
        }
        
        return uf.cnt - 1;
    }
}
