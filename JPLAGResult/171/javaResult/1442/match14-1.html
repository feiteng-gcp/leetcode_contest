class Solution {
    
private void dfs(ArrayList<List<Integer>> G, int i, boolean[] visit){
        visit[i] = true;
        for(Integer j: G.get(i)){
            if (!visit[j]){
                dfs(G, j, visit);
            }
        }
    }
    
    public int makeConnected(int n, int[][] connections) {
        
ArrayList<List<Integer>> G = new ArrayList<>();
        for(int i = 0; i < n; i++){
            G.add(new ArrayList<Integer>());
        }
        for(int[] cable: connections){
            G.get(cable[0]).add(cable[1]);
            G.get(cable[1]).add(cable[0]);
        }
        
        // count number of SCCs
        int scc = 0;
        boolean[] visit = 
new boolean[n];
        for(int i = 0; i < n; i++){
            if (!visit[i]){
                dfs(G, i, visit);
                scc++;
            }
        }
        
        // extra cables
        int extra = connections.length - (n - scc);
        
        if (extra >= scc - 1){
            return scc - 1;
        }else{
            return -1;
        }
    }
}
