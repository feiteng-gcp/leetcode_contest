class Solution {
    public int makeConnected(int n, int[][] connections) {
        if(n > connections.length + 1) return -1;
        LinkedList<Integer>[] list = new LinkedList[n];
        for(int[] connection: connections){
            if(list[connection[0]] == null) list[connection[0]] = new LinkedList();
            list[connection[0]].add(connection[1]);
            if(list[connection[1]] == null) list[connection[1]] = new LinkedList();
            list[connection[1]].add(connection[0]);
        }
        
        boolean[] visited = new boolean[n];
        int clusters = 0;
        for(int i=0;i<n;i++){
            if(!visited[i]){
                dfs(i, list, visited);
                clusters += 1;
            }
        }
        
        return clusters -1;
    }
    
    
    void dfs(int i, LinkedList<Integer>[] list, boolean[] visited){
        if(visited[i] || list[i] == null) return;
        visited[i] = true;
        for(int j: list[i]) dfs(j, list, visited);       
    }
}
