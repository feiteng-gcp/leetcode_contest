class Solution {
    public int makeConnected(int n, int[][] edges) {
        if (edges.length < n - 1) {
            return -1;
        }

        UnionFind uf = new UnionFind(n);
        for (int[] edge : edges) {
            uf.union(edge[0], edge[1]);
        }

        return uf.getGroups() - 1;
    }

    static class UnionFind {
        private int[] father;
        private int groups;

        public UnionFind(final int n) {
            groups = n;
            father = new int[n];
            for (int i = 0; i < n; ++i) {
                father[i] = i;
            }
        }

        public int getGroups() {
            return groups;
        }

        public int find(int x) {
            if (x == father[x]) {
                return x;
            }
            father[x] = find(father[x]);
            return father[x];
        }

        public void union(int x, int y) {
            int xRoot = find(x);
            int yRoot = find(y);
            if (xRoot != yRoot) {
                father[xRoot] = yRoot;
                groups--;
            }
        }
    }
}
