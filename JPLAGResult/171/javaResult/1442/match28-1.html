class Solution {
    public int makeConnected(int n, int[][] connections) {
        int cableCnt = connections.length;
        if (n>cableCnt+1) {
            return -1;
        }

        DSU dsu = new DSU(n);
        for (int i=0; i<cableCnt; i++) {
            dsu.union(connections[i][0], connections[i][1]);
        }
        
        return dsu.size-1;

    }

    class DSU {
        int[] parent;
        int size;

        public DSU(int N) {
            parent = new int[N];
            size = N;
            for (int i = 0; i < N; ++i)
                parent[i] = i;
        }

        public int find(int x) {
            if (parent[x] != x) parent[x] = find(parent[x])
;
            return parent[x];
        }

        public void union(int x, int y) {
            int r1 = find(x);
            int r2 = find(y);
            if (r1 != r2) {
                parent[r1] = r2;
                size--;
            }
        }
    }
}
