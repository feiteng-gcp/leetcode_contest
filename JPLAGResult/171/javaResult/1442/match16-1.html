class Solution {
    public int makeConnected(int n, int[][] connections) {
        if (connections.length < n - 1) {
            return -1;
        }
        
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < connections.length; i++) {
            map.computeIfAbsent(connections[i][0], k -> new ArrayList<>()).add(connections[i][1]);
            map.computeIfAbsent(connections[i][1], k -> new ArrayList<>()).add(connections[i][0]);
        }
        
        boolean[] vis = new boolean[n];
        int comp = 0;
        
for (int i = 0; i < n; i++) {
            if (!vis[i]) {
                dfs(i, map, vis);
                comp++;
            }
        }
        
        return comp - 1;
    }
    
    private static void dfs(int start, Map<Integer, List<Integer>> map, boolean[] vis) {
        vis[start] = true;
        
        for (int next : map.getOrDefault(start, new ArrayList<>())) 
{
            if (!vis[next]) {
                dfs(next, map, vis);    
            }
        }
    }
}
