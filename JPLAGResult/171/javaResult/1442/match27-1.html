class Solution {
    public int makeConnected(int n, int[][] connections) {
        int totalWires = connections.length;
        
if(n-1>totalWires) {
            return -1;
        }
        
        UnionFind unionFind = new UnionFind(n);
        
        for(int[] connection : connections) {
            unionFind.union(connection[0], connection[1]);
        }
        
        return unionFind.disjoinSets-1;
    
}
    
    
    class UnionFind {
        int[] uf;
        int disjoinSets;
        public UnionFind(int n) {
            disjoinSets = n;
            uf = new int[n];
            for(int i=0; i<n; i++) {
                uf[i] = i;
            }
        }
        
        public int find(int i) {
            if(uf[i] != i) {
                uf[i] = find(uf[i]);
            }
            
return uf[i];
        }
        
        public boolean union(int i, int j) {
            int pi = find(i);
            int pj = find(j);
            if(pi == pj) {
                return false;
            } else {
                uf[pj] = pi;
                disjoinSets--;
                return true;
            }
        }
    }
    
    //Do union-find to find disjoint sets
}
