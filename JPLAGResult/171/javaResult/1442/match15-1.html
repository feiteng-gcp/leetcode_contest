import java.util.ArrayList;
import java.util.List;


class Solution {
    public int makeConnected(int n, int[][] connections) {
        if (connections.length < n - 1) {
            return -1;
        }

        
List<List<Integer>> cons = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            cons.add(new ArrayList<>());
        }
        for (int[] connection : connections) {
            cons.get(connection[0]).add(connection[1]);
            cons.get(connection[1]).add(connection[0]);
        }
        boolean visit[] = new boolean[n];
        int ans = -1;
        
for (int i = 0; i < n; i++) {
            if (!visit[i]) {
                ans++;
                dfs(cons, visit, i);
            }
        }
        return ans;
    }

    private void dfs(List<List<Integer>> cons, boolean[] visit, int current) {
        if (!visit[current]) {
            visit[current] = true;
            for (Integer integer : cons.get(current)) {
                dfs(cons, visit, integer);
            }
        }
    }
}
