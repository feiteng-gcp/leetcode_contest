class Solution {
    public int makeConnected(int n, int[][] connections) {
        if (connections.length < n - 1) return -1;
        List<Integer>[] la = new List[n];
        for (int i = 0; i < n; i++) {
            la[i] = new ArrayList<>();
        }
        
        for (int[] connection : connections) {
            int a = connection[0], b = connection[1];
            la[a].add(b);
            la[b].add(a);
        }
        
        boolean[] arr = new boolean[n];
        int res = 0;
        for (int i = 0; i < n; i++) {
            if (!arr[i]) {
                dfs(la, i, arr);
                res++;
            }
        }
        return res - 1;
    }
    
    private void dfs(List<Integer>[] la, int index, boolean[] arr) {
        if (arr[index]) return;
        arr[index] = true;
        for (int next : la[index])
            dfs(la, next, arr);
    }
}
