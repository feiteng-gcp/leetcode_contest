class Solution {
    public void dfs(List<Integer>[] g, int node, int color, int[] visit) {
        if (visit[node] != 0) return;
        visit[node] = color;
        for (int nei : g[node])
            dfs(g, nei, color, visit);
    }
    
    public int makeConnected(int n, int[][] connections) {
        if (connections.length < n - 1) return -1;
        int[] v = new int[n];
        int cc = 1;
        List<Integer>[] g = new List[n];
        for (int i = 0; i < n; i++) g[i] = new ArrayList<>();
        for (int[] a : connections) {
            g[a[0]].add(a[1]);
            g[a[1]].add(a[0]);
        }
        
        for (int i = 0; i < n; i++) {
            if (v[i] == 0) {
                dfs(g, i, cc++, v);
            }
        }
        
        return cc - 2;
    }
}
