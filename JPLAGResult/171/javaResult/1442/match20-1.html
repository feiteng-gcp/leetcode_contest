class Solution {
    public int makeConnected(int n, int[][] connections) {
        if (connections.length < n - 1) {
            return -1;
        }
        List<List<Integer>> out = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            out.add(new ArrayList<>());
        }
        for (int[] e: connections) {
            out.get(e[0]).add(e[1]);
            out.get(e[1]).add(e[0]);
        }
        int[] color = new int[n];
        for (int i = 0; i < n; i++) {
            color[i] = -1;
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (color[i] == -1) {
                color[i] = ans++;
                
dfs(out, color, i);
            }
        }
        return ans - 1;
    }
    
    private void dfs(List<List<Integer>> out, int[] color, int d) {
        for (int t : out.get(d)) {
            if (color[t] == -1) {
                color[t] = color[d];
                dfs(out, color, t);
            }
        }
    }
}
