class Solution {
     boolean[] v;
    List<Integer>[] g;

    public int makeConnected(int n, int[][] es) {
        int en = es.length;
        if (en < n - 1) {
            return -1;
        }
        g = new ArrayList[n];
        for (int i = 0; i < n; i++) {
            g[i] = new ArrayList<>();
        }
        for (int[] e : es) {
            g[e[0]].add(e[1]);
            g[e[1]].add(e[0]);
        }
        v = new boolean[n];
        int comps = 0;
        for (int i = 0; i < n; i++) {
            if (!v[i]) {
                comps++;
                dfs(i);
            }
        }
        return comps - 1;
    }

    private void dfs(int i) {
        v[i] = true;
        for (int j : g[i]) {
            if (!v[j]) {
                dfs(j);
            }
        }
    }
}
