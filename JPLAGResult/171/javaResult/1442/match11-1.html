class Solution {
public int makeConnected(int n, int[][] connections) {
        int m = connections.length;
        if (m < n - 1) return -1;

        boolean[] vst = new boolean[n];
        
List<Integer>[] g = new List[n];
        for (int i = 0; i < n; i++) {
            g[i] = new ArrayList<>();
        }

        for (int i = 0; i < connections.length; i++) {
            int x = connections[i][0];
            int y = connections[i][1];
            g[x].add(y);
            g[y].add(x);
        }

        
int ans = 0;

        for (int i = 0; i < n; i++) {
            if (!vst[i]) {
                ans++;
                dfs(i, g, vst);
            }
        }

        return ans - 1;

    }

    void dfs(int x, List<Integer>[] g, boolean[] vst) {
        if (vst[x]) return;
        vst[x] = true;

        for (int y : g[x]) {
            dfs(y, g, vst);
        }
    }

}
