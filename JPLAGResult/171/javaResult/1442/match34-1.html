class Solution {
	int f[];

	public int makeConnected(int n, int[][] connections) {
		if (connections.length < n - 1)
			return -1;
		f = new int[n];
		for (int i = 0; i < n; i++)
			f[i] = i;

		for (int conn[] : connections) {
			merge(conn[0], conn[1]);
		}
		int cnt = 0;
		for (int i = 0; i < n; i++)
			if (f[i] == i)
				cnt++;

		return cnt - 1;
	}

	private void merge(int u, int v) {
		int pu = find(u);
		int pv = find(v);
		if (pu == pv)
			return;
		f[pu] = pv;

	}


	private int find(int u) {
		if (f[u] != u) {
			f[u] = find(f[u]);
		}
		return f[u];
	}
}
