class Solution {
    public int makeConnected(int n, int[][] connections) {
        if (n <= 1) {
            return 0;
        }
        
        if (connections.length < n - 1) {
            return -1;
        }
        
        int count = n;
        
int[] origins = new int[n];
        for (int i = 0; i < origins.length; i++) {
            origins[i] = i;
        }
        
        for (int[] c : connections) {
            int left = c[0], right = c[1];
            int originLeft = findHelper(left, origins);
            int originRight = findHelper(right, origins);
            if (originLeft != originRight) {
                count--;
                origins[originLeft] = originRight;
            }
        }
        
return count - 1;
    }
    
    private int findHelper(int current, int[] origins) {
        if (origins[current] == current) {
            return current;
        }
        
        int result = findHelper(origins[current], origins);
        origins[current] = result;
        return result;
    }
}
