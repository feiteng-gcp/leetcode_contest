class Solution {
    
    private int[] p;
    
    public int makeConnected(int n, int[][] connections) {
        int m = connections.length;
        if(n > m + 1){
            return -1;
        }
        p = new int[n];
        for(int i = 0 ; i < n ; i++){
            p[i] = i;
        }
        for(int[] con: connections){
            union(con[0], con[1]);
        }
        int res = 0;
        for(int i = 0 ; i < n ; i++){
            if(i == find(i)){
                res++;
            }
        }
        return res - 1;
    }
    
    private void union(int u, int v){
        int pu = find(u);
        int pv = find(v);
        if(pu != pv){
            p[pu] = pv;
        }
    }
    
    private int find(int u){
        if(u != p[u]){
            p[u] = find(p[u]);
        }
        return p[u];
    }
}
