class Solution {
    int[] id;
    int N;
    int count;
    int extra;
    public int makeConnected(int n, int[][] connections) {
        N = n;
        count = N;
        extra = 0;
        id = new int[N];
        
for(int i = 0; i < id.length; i++) {
            id[i] = i;
        }
        for(int[] conn: connections) {
            union(conn[0], conn[1]);
            // System.out.println(count);
        }
        if(count == 1) return 0;
        if(count - extra <= 1) return count - 1;
        return -1;
        
        
    
}
    
    private int find(int i) {
        if(id[i] != i) {
            id[i] = find(id[i]);
        }
        return id[i];
    }
    private void union(int i, int j) {
        int ir = find(i);
        int jr = find(j);
        if(ir == jr) {
            extra++;
            return;
        } 
        id[ir] = jr;
        count--;
    }
}
