class Solution {
    public int makeConnected(int n, int[][] connections) {
        if(connections.length<n-1){
            return -1;
        }
        
        HashMap<Integer, List<Integer>> graph = new HashMap();
        
        for(int[] e : connections){
            graph.putIfAbsent(e[0], new ArrayList());
            graph.putIfAbsent(e[1], new ArrayList());
            graph.get(e[0]).add(e[1]);
            graph.get(e[1]).add(e[0]);
        }
        
        int count = 0;
        boolean[] visited = new boolean[n];
        
for(int c=0; c<n; c++){
            if(!visited[c]){
                count++;
                visit(graph, c, visited);
            }
        }
        return count-1;
    }
    
    public void visit(HashMap<Integer, List<Integer>> graph, int computer, boolean[] visited){
        if(visited[computer]){
            return;
        }
        visited[computer] = true;
        if(graph.containsKey(computer)){
            
for(int next : graph.get(computer)){
                visit(graph, next, visited);
            }            
        }
    }
}
