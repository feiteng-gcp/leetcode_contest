class Solution {
    // uf, need to record extra cables.
    private int[] father;
    private int[] rank;
    private int[] extra;
    private int groupCnt;
    public int makeConnected(int n, int[][] connections) {
        father = new int[n];
        rank = new int[n];
        extra = new int[n];
        groupCnt = n;
        for(int i = 0; i < n; i++) {
            father[i] = i;
        }
        for(int[] con : connections) {
            connect(con[0], con[1]);
        }
        int extraSum = 0;
        for(int i = 0; i < n; i++) {
            // unify all extra cables
            extraSum += extra[i];
        }
        return extraSum >= groupCnt - 1 ? (groupCnt - 1) : -1;
    }
    
    private int find(int a) {
        if(father[a] != a) {
            father[a] = find(father[a]);
        }
        return father[a];
    }
    
    private void connect(int a, int b) {
        int roota = find(a);
        int rootb = find(b);
        if(roota != rootb) {
            if(rank[roota] < rank[rootb]) {
                father[roota] = rootb;
            } else if(rank[roota] < rank[rootb]) {
                father[rootb] = roota;
            } else {
                father[roota] = rootb;
                rank[rootb]++;
            }
            groupCnt--;
        } else {
            extra[roota]++;
        }
    }
}
