class Solution {
    Map<Integer, List<Integer>> graph = new HashMap<>();
    int group[];
    
    public int makeConnected(int n, int[][] connections) {
        group = new int[n];
        
for (int i = 0; i < n; i++) {
            graph.put(i, new ArrayList<>());
        }
        for (int [] c : connections) {
            graph.get(c[0]).add(c[1]);
            graph.get(c[1]).add(c[0]);
        }
        
        int currentGroup = 1;
        
for (int i = 0; i < n; i++) {
            if (group[i] == 0) {
                dfs(i, currentGroup);
                currentGroup++;
            }
        }
        currentGroup--;
        
        
        
        int unconnected = 0;
        for (int i = 0; i < n; i++) {
            if (group[i] == 0) {
                unconnected++;
            }
        }
        
        int remaining = connections.length - (n - unconnected - 1);
        if (remaining >= unconnected) {
            return currentGroup - 1 + unconnected;
        }
        return -1;
    }
    
    private void dfs(
int i, int color) {
        group[i] = color;
        for (int next : graph.get(i)) {
            if (group[next] == 0) {
                dfs(next, color);
            }
        }
    }
}
