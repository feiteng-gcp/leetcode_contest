class Solution {
    private int[] father;
    
    public int makeConnected(int n, int[][] connections) {
        if (connections.length + 1 < n) {
            return -1;
        }
        father = new int[n];
        for(int i = 0; i < n; ++i) {
            father[i] = i;
        }
        
        for(int[] connection : connections) {
            union(connection[0], connection[1]);
        }
        
        Set<Integer> set = new HashSet<>();
        for(int i = 0; i < n; ++i) {
            set.add(findFather(i));
        }
        
        return set.size() - 1;
    }
    
    private int findFather(int x) {
        if (father[x] != x) {
            father[x] = findFather(father[x]);
        } 
        return father[x];
    }
    
    private void union(int x, int y) {
        int xp = findFather(x);
        int yp = findFather(y);
        if (xp != yp) {
            father[xp] = yp;
        }
    }
}
