class Solution {
    public int minimumDistance(String word) {
        int n = word.length(), ans = 10000000;
        char [] c = word.toCharArray();
        int [][] dist = new int[26][26];
        for(int i = 0 ; i < 26; i ++)        
            for(int j = 0 ; j < 26; j ++)
            {
                if(i == j) dist[i][j] = 0;
                else dist[i][j] = Math.abs(i / 6 - j / 6) + Math.abs(i % 6 - j % 6);
            }
        int [][] dp = new int[n + 1][n + 1]; // dp[i][j] means two finger at char[i] & char[j]
        /*for(int i = 0; i <= n; i ++)
            Arrays.fill(dp[i], 10000000);*/
        dp[0][0] = 0;
        dp[1][0] = 0;
        for(int i = 2; i <= n; i ++)
            dp[i][0] = dp[i - 1][0] + dist[c[i - 1] - 'A'][c[i - 2] - 'A'];
        for(int i = 1; i <= n; i ++)
        {
            for(int j = 1; j < i - 1 ; j ++)
            {
                dp[i][j] = dp[i - 1][j] + dist[c[i - 1] - 'A'][c[i - 2] - 'A'];
            }
            dp[i][i - 1] = dp[i - 1][0];
            for(int j = 1; j < i - 1 ; j ++)
            {
                dp[i][i - 1] = Math.min(dp[i][i - 1], dp[i - 1][j] + dist[c[j - 1] - 'A'][c[i - 1] - 'A']);
            }            
        }
        /*for(int i = 0; i <= n ; i ++)
            System.out.println(Arrays.toString(dp[i]));*/
        for(int i = 0; i < n ; i ++)
            ans = Math.min(ans, dp[n][i]);
        return ans;
    }
}
