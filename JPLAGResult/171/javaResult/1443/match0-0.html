class Solution {
    public int minimumDistance(String str) {        
        char[] S = str.toCharArray();
        int len = S.length;
        int res = Integer.MAX_VALUE;
        int dp[][][] = new int[26][26][len + 1];
        for(int fir = len - 1; fir >= 0; fir--){
            for(int x = 0; x < 26; x++){
                for(int y = 0; y < 26; y++){
                    dp[x][y][fir] = Math.min(cost(x,S[fir]-'A') + dp[S[fir]-'A'][y][fir+1], cost(y, S[fir]-'A')+dp[x][S[fir]-'A'][fir+1]);
                    if (fir == 0) {
                        res = Math.min(res, dp[x][y][fir]);
                    }
                }
            }
        }
        return res;
    }
    public int cost(int x, int y){
        return Math.abs(x / 6-y/6) + Math.abs(x % 6-y%6);
    }    
}
