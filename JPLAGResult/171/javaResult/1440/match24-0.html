class Solution {
    public int[] getNoZeroIntegers(int n) {
        int[] res = new int[2];
        for (int i = 1; i < n; i++) {
            int temp = n - i;
            if (vaild(temp) && vaild(i)) {
                res[0] = i;
                res[1] = temp;
            }
        }
        return res;
    }
    private boolean vaild(int num) {
        while (num > 0) {
            if(num % 10 == 0) {
                return false;
            }
            num /= 10;
        }
        return true;
    }
}
