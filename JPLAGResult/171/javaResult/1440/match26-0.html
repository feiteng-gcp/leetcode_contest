class Solution {
    public int[] getNoZeroIntegers(int n) {
       for(int i = 1; i < n; i++){
           if(noZero(i) && noZero(n-i))
               return new int[]{i, n-i};
       }   
        return null;
    }
    boolean noZero(int k){
        while(k > 0){
            if(k %10 == 0)
                return false;
            k/=10;
        }
        return true;
    }
}
