class Solution {
    public int[] getNoZeroIntegers(int n) {
        for (int i = 1; i < n; i++) {
            if (check(i) && check(n - i)) {
                return new int[] {i, n - i};
            }
         }
        return new int[]{};
    
}
    
    private static boolean check(int i) {
        return String.valueOf(i).indexOf('0') == -1; 
    }
}
