class Solution {
    public int[] getNoZeroIntegers(final int n) {
        for (int i = 1; i < n; ++i) {
            if (isValid(i) && isValid(n - i)) {
                return new int[] {i, n - i};
            }
        }
        return new int[] {-1, -1};
    
}

    private boolean isValid(int num) {
        while (num > 0) {
            if (num % 10 == 0) {
                return false;
            }
            num /= 10;
        }
        return true;
    }
}
