class Solution {
    public int[] getNoZeroIntegers(int n) {
        int[] res = new int[2];
        for(int i=1; i<=n/2; i++){
            int j = n-i;
            if(isNonZero(i) && isNonZero(j)){
                res[0] = i;
                res[1] = j;
                break;
            
}
        }
        return res;
    }
    
    public boolean isNonZero(int num){
        while(num>0){
            if(num%10==0){
                return false;
            }
            num/=10;
        }
        return true;
    }
}
