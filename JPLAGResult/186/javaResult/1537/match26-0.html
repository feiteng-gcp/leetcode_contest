class Solution {
    public int maxScore(String s) {
        int n = s.length();
        int[] ones = new int[n+1];
        int[] zeros = new int[n+1];
        for (int i = 0; i < n; i++) {
            if (s.charAt(i) == '0') {
                zeros[i+1] = zeros[i] + 1;
                ones[i+1] = ones[i];
            } else {
                ones[i+1] = ones[i] + 1;
                zeros[i+1] = zeros[i];
            }
        }
        // res = 0;
        int res = 0;
        for (int i = 0; i < n - 1; i++) {
            res = Math.max(res, zeros[i+1] + ones[n] - ones[i+1]);
        }
        return res;
    }
}
