class Solution {
    public int maxScore(String s) {
        int max = 0;
        for (int i = 0; i < s.length() - 1; i++) {
            int left = 0;
            for (int j = 0; j <= i; j++) {
                if (s.charAt(j) == '0') {
                    left++;
                }
            }
            int right = 0;
            for (int p = i + 1; p < s.length(); p++) {
                if (s.charAt(p) == '1') {
                    right++;
                }
            }
            max = Math.max(max, left + right);
        }
        return max;
    }
}
