	class Solution {
	    public int maxScore(String S) {
	        char[] s = S.toCharArray();
	        int one = 0;
	        for(char c : s){
	        	if(c == '1')one++;
	        }
	        int zero = 0;
	        int ans = 0;
	        for(int i = 1;i <= s.length-1;i++){
	        	if(s[i-1] == '0')zero++;
	        	if(s[i-1] == '1')one--;
	        	ans = Math.max(ans, zero+one);
	        }
	        return ans;
	    }
	}
