class Solution {
    public int maxScore(String s) {
        int ans = 0;
        int left = 0;
        int right = 0;
        
        if(s.charAt(0) == '0') left++;
        for(int i = 1; i < s.length(); i++){
            if(s.charAt(i) == '1') right++;
        }
        ans = left + right;
        
        for(int i = 1; i < s.length() - 1; i++){
            if(s.charAt(i) == '1') right--;
            else left++;
            
            ans = Math.max(ans, left + right);
        }
        return ans;
    }
}
