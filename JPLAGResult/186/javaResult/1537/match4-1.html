class Solution {
    public int maxScore(String s) {
        int max = 0;
        for (int i = 1; i < s.length(); i++) {
            int score = 0;
            for (int j = 0; j < i; j++) {
                if (s.charAt(j) == '0') {
                    score++;
                }
            }
            for (int j = i; j < s.length(); j++) {
                if (s.charAt(j) == '1') {
                    score++;
                }
            }
            max = Math.max(max, score);
        }
        return max;
    }
}
