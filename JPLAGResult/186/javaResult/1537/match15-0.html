class Solution {
    public int maxScore(String s) {
      int o0 = 0, o1 = 0;
      for(int i=0; i<s.length(); i++) {
        if(s.charAt(i) == '0') o0++;
        else o1++;
      }
      int res = 0, c0 = 0, c1 = 0;
      for(int i=0; i<s.length()-1; i++) {
        if(s.charAt(i) == '0') c0++;
        else c1++;
        res = Math.max(res, c0 + o1-c1);
      }
      return res;
    }
  }
