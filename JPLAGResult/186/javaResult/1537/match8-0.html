class Solution {
    public int maxScore(String s) {
        int res = 0;
        for(int i = 0; i < s.length() - 1; i++){
            int left = 0;
            for(int j = 0; j <= i; j++){
                if(s.charAt(j) == '0')
                    left++;
            }
            int right = 0;
            for(int j = i + 1; j < s.length(); j++){
                if(s.charAt(j) == '1')
                    right++;
            }
            res = Math.max(res, left + right);
        }
        return res;
    }
}
