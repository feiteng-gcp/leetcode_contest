class Solution {
    public int maxScore(String s) {
        int ret = 0;
        int one = 0;
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1') {
                one++;
            }
        }
        int zero = 0;
        for (int i = 0; i < s.length() - 1; i++) {
            if (s.charAt(i) == '0') {
                zero++;
            } else {
                one--;
            }
            ret = Math.max(ret, zero + one);
        }
        return ret;
    }
}
