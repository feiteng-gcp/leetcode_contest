class Solution {
    public int maxScore(String s) {
        int[] left = new int[s.length()];
        int[] right = new int[s.length()];
        int sum = 0;
        for (int i = 0;i < s.length(); i++) {
            if (s.charAt(i) == '0') {
                sum++;
            }
            left[i] = sum;
        }
        sum = 0;
        for (int i = s.length() - 1; i >= 0; i--) {
            right[i] = sum;
            if (s.charAt(i) == '1') {
                sum++;
            }
        
}
        int max = 0;
        for (int i = 0; i < s.length() - 1; i++) {
            max = Math.max(max, left[i] + right[i]);
        }
        return max;
    }
}
