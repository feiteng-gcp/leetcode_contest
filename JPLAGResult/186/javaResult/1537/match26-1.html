class Solution {
    public int maxScore(String s) {
        int n = s.length();
        int zeros[] = new int[n+2];
        int ones[] = new int[n+2];
        for(int i = 0; i<n; i++){ if(s.charAt(i)=='0') zeros[i+1] = zeros[i] + 1; else zeros[i+1] = zeros[i];}
        for(int i = n-1; i>=0; i--) {if(s.charAt(i) == '1') ones[i] = ones[i+1]+1;else ones[i] = ones[i+1]
;}
        int ans = 0;
        for(int i = 1; i<n ;i++){
            ans = Math.max(ans, zeros[i] + ones[i]);
        }
        return ans;
    }
}
