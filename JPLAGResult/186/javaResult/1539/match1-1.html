class Solution {
    public int[] findDiagonalOrder(List<List<Integer>> nums) {
        ArrayList<ArrayList<Integer>> sort = new ArrayList<>();
        int size = 0;
        for (int i = 0; i < nums.size(); i++) {
            for (int j = 0; j < nums.get(i).size(); j++) {
                while (sort.size() <= i + j) {
                    sort.add(new ArrayList<>());
                }
                
sort.get(i + j).add(0, nums.get(i).get(j));
                size++;
            }
        }
        int[] res = new int[size];
        int index = 0;
        for (ArrayList<Integer> arr : sort) {
            for (int i : arr) 
{
                res[index++] = i;
            }
        }
        return res;
    }
}
