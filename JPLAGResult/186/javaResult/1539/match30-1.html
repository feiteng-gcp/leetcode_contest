class Solution {
    public int[] findDiagonalOrder(List<List<Integer>> nums) {
        List<List<Integer>> map = new ArrayList<>();
        int max = 0;
        for(int i = 0; i < nums.size(); i++)
            max = Math.max(max, nums.get(i).size());
        for(int i = 0; i < max + nums.size(); i++)
            map.add(new ArrayList<Integer>());
        int total = 0;
        for(int i = 0; i < nums.size(); i++){
            for(int j = 0; j < nums.get(i).size(); j++){
                
map.get(i + j).add(nums.get(i).get(j));
                total++;
            }
        }
        
        int[] res = new int[total];
        int idx = 0;
        for(int i = 0; i < map.size(); i++){
            for(int j = map.get(i).size() - 1; j >= 0; j--){
                res[idx] = map.get(i).get(j);
            idx++;
            }
        }
        return res;
    }
    
    int get(List<List<Integer>> nums, int r, int c){
        if(r < 0 || r >= nums.size()) return -1;
        List<Integer> l = nums.get(r);
        if(c < 0 || c >= l.size()) return -1;
        return l.get(c);
    }
}
