class Solution {
       class Item{
        int r;
        int c;
        int v;
        
        Item(int r, int c, int v) {
            this.r = r+c;
            this.c = c;
            this.v = v;
        }
    }

    public int[] findDiagonalOrder(List<List<Integer>> nums) {
        List<Item> items = new ArrayList<>();
        int r = 0;
        for(List<Integer> ls: nums){
            int c = 0;
            for(int n: ls){
                items.add(new Item(r, c, n));
                ++c;
            
}
            ++r;
        }
        items.sort(new Comparator<Item>() {
            @Override
            public int compare(Item t0, Item t1) {
                if(t0.r!=t1.r) return t0.r-t1.r;
                return t0.c-t1.c;
            }
        });
        return items.stream().mapToInt(i->i.v).toArray();
    }
}
