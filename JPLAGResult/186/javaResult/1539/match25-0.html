class Solution {
    public int[] findDiagonalOrder(List<List<Integer>> nums) {
        List<int[]> record = new ArrayList<>();
        for (int i = 0; i < nums.size(); i++) {
            for (int j = 0; j < nums.get(i).size(); j++) {
                record.add(new int[]{i, j, nums.get(i).get(j)});
            }
        }
        Collections.sort(record, (o1, o2) -> o1[0] + o1[1] == o2[0] + o2[1] ? Integer.compare(o2[0], o1[0]) : Integer.compare(o1[0] + o1[1], o2[0] + o2[1]));
        int[] res = new int[record.size()];
        for (int i = 0; i < record.size(); i++) res[i] = record.get(i)[2];
        return res;
    }
}
