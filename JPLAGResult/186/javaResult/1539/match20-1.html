class Solution {
    public int[] findDiagonalOrder(List<List<Integer>> nums) {
        List<Integer> ans = new ArrayList<>();
        int K = 0;
        for (int i = 0; i < nums.size(); ++i) {
            K = Math.max(K, i + nums.get(i).size());
        }
        List<Integer>[] lists = new List[K + 1];
        for (int i = 0; i <= K; ++i) {
            lists[i] = new ArrayList();
        }
        int N = 0;
        for (int i = 0; i < nums.size(); ++i) {
            for (int j = 0; j < nums.get(i).size(); ++j) {
                
lists[i + j].add(nums.get(i).get(j));
                N++;
            }
        }
        int[] ret = new int[N];
        for (
int i = 0, ii = 0; i <= K; ++i) {
            for (int j = lists[i].size() - 1; j >= 0; --j) {
                ret[ii++] = lists[i].get(j);
            }
        }
        return ret;
    }
}
