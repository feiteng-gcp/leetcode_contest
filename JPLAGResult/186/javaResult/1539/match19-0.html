class Solution {
    class Pair {
        int i, j;
        public Pair(int i, int j) {
            this.i = i;
            this.j = j;
        }
    }
    
    public int[] findDiagonalOrder(List<List<Integer>> nums) {
        TreeMap<Integer, List<Pair>> diags = new TreeMap<>();
        
        int numElems = 0;
        for (int i = 0; i < nums.size(); i++) {
            for (int j = 0; j < nums.get(i).size(); j++) {
                diags.putIfAbsent(i+j, new ArrayList<>());
                diags.get(i+j).add(new Pair(i, j));
                numElems++;
            }
        }
        
        for (int k : diags.keySet()) {
            Collections.sort(diags.get(k), (a, b)->(a.j - b.j));
        }
        
        int[] res = new int[numElems];
        int ix  = 0;
        for (int k : diags.keySet()) {
            for (Pair p : diags.get(k)) {
                res[ix++] = nums.get(p.i).get(p.j);
            }
        }
        return res;
    }
}
