class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n + 1];
        TreeSet<Integer> st = new TreeSet<Integer>((a, b) -> dp[a] - dp[b]);
        for (int i = 1; i <= n; i++) {
            Integer max = null;
            if (st.size() != 0) {
                max = st.last();    
            }
            
            dp[i] = nums[i - 1]; 
            if (max != null) {
                dp[i] = Math.max(dp[i], dp[i] + dp[max]);
            }
            st.add(i);
            if (i - k >= 0) {
                st.remove(i - k);
            }
        
}
        
        int ret = -(int)1e9;
        for (int i = 1; i < n + 1; i++) {
            ret = Math.max(ret, dp[i]);
        }
        return ret;
    }
}
