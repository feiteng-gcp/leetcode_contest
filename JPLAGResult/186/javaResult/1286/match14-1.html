class Solution {
  public int constrainedSubsetSum(int[] nums, int k) {
    int n = nums.length;
    int[] dp = new int[n];
    int ans = Integer.MIN_VALUE;
    TreeMap<Integer, Integer> map = new TreeMap<>();
    for (int i = 0; i < n; ++i) {
      
if (i >= k + 1) {
        int cnt = map.get(dp[i - k - 1]);
        if (cnt > 1) {
          map.put(dp[i - k - 1], cnt - 1);
        } else {
          map.remove(dp[i - k - 1]);
        }
      }
      ans = Math.max(ans, dp[i] = nums[i] + Math.max(map.isEmpty() ? 0 : map.lastKey(), 0));
      map.put(dp[i], map.getOrDefault(dp[i], 0) + 1);
    }
    return ans;
  }
}
