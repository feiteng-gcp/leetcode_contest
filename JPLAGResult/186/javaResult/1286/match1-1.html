class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        TreeMap<Integer, Integer> map = new TreeMap<Integer, Integer>();
        map.put(nums[0], 0);
        int ans = nums[0];
        for(int i = 1; i < nums.length; i++) {
            while(!map.isEmpty()) {
                int key = map.lastKey();
                if(key + nums[i] < nums[i]) {
                    map.put(nums[i], i);
                    break;
                }
                if(i - map.get(key) <= k) {
                    map.put(key + nums[i], i);
                    break;
                } else {
                    map.remove(key);
                
}
            }
            ans = Math.max(ans, map.lastKey());
            // System.out.println(map);
        }
        return ans;
    }
}
