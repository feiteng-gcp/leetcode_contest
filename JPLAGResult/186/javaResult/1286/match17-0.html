class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        
        
        
        // heap + dp
        int[] dp = new int[n];
        PriorityQueue<Integer> heap = new PriorityQueue<>(
            (a, b) -> dp[b] - dp[a]
        );
        Queue<Integer> q = new ArrayDeque<>();
        int res = Integer.MIN_VALUE;
        for (int i = 0; i < n; i++) {
            // System.out.println("i=" + i);
            dp[i] = nums[i];
            if (!heap.isEmpty()) {
                int top = heap.peek();
                // only remain k elements
                if (i - top == k) {
                    heap.remove(top);
                }
                // System.out.println("top=" + top);
                // System.out.println("dp[top]=" + dp[top]);
                int localMax = nums[i] + dp[top];
                dp[i] = localMax;
                if (localMax >= 0) {
                    heap.add(i);
                }
                
            } else {
                if (nums[i] > 0) {
                    heap.add(i);
                }
            }
            // System.out.println("dp[i]=" + dp[i]);
            res = Math.max(res, dp[i]);
        }
    
        return res;
