import java.util.PriorityQueue;

class Solution {
  Integer[] dp;
  int[] nums;
  int k;
  
  public static void main(String args[]) {
    Solution sol = new Solution();
    int[] nums = new int[] {-1, -2, -3};
    int k = 1;
    System.out.println(sol.constrainedSubsetSum(nums, k));
  }
  
  public int constrainedSubsetSum(int[] nums, int k) {
    PriorityQueue<IndexValue> values = new PriorityQueue<IndexValue>(10, (iv1, iv2) -> iv2.value - iv1.value);
    int[] dp = new int[nums.length];
    
    dp[nums.length - 1] = nums[nums.length - 1];
    values.add(new IndexValue(nums.length - 1, dp[nums.length - 1]));
    values.add(new IndexValue(nums.length, 0));
    for (int i = nums.length - 2; i >= 0; i--) {
      while (!values.isEmpty() && values.peek().index - i > k) {
        values.poll();
      }
      int sum = nums[i] + values.peek().value;
      dp[i] = Math.max(sum, nums[i]);
      values.add(new IndexValue(i, dp[i]));
    }
    
    int max = Integer.MIN_VALUE;
    for (int x : dp) {
      max = Math.max(x, max);
    }
    return max;
  }
  
  
  static class IndexValue {
    int index;
    int value;
    
    public IndexValue(int i, int v) {
      this.index = i;
      this.value = v;
    }
  }
}
