import java.util.Map.Entry;
import java.util.TreeMap;

class Solution {
	public int constrainedSubsetSum(int[] nums, int k) {
		TreeMap<Integer, Integer> m = new TreeMap<Integer, Integer>();
		m.put(-1, 0);
		int ans = Integer.MIN_VALUE;
		for (int i = 0; i < nums.length; i++) {
			Entry<Integer, Integer> en = m.firstEntry();
			int cur =Math.max(en.getValue(),0) + nums[i];
			if (i - en.getKey() >= k)
				m.remove(en.getKey());
			while ((en = m.floorEntry(i)) != null) {
				if (en.getValue() <= cur)
					m.remove(en.getKey());
                else
                    break;
			}
			ans = Math.max(ans, cur);
			m.put(i, cur);
		}
		return ans;
	}
}
