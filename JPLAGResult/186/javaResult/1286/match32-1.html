class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int [] dp = new int[nums.length];
        TreeMultiSet set = new TreeMultiSet();
        dp[0] = nums[0];
        set.insert(dp[0]);
        for(int i=1;i<nums.length;i++){
            if(set.set.last()>=0)
             dp[i] = nums[i]+set.set.last();
            else{
                dp[i] = nums[i];
            }
            if(i>=k){
                set.delete(dp[i-k]);
            }
            set.insert(dp[i]);
        
}
        int ret = Integer.MIN_VALUE;
        for(int i=0;i<nums.length;i++){
            ret = Math.max(ret,dp[i]);
        }
            return ret;
    }
    
}
public class TreeMultiSet{
		static TreeSet<Integer> set; //can change the type of treeset
		static Map<Integer, Integer> occur; //can change the type of map;
		public TreeMultiSet() {
			set = new TreeSet<Integer>();
			occur = new HashMap<Integer,Integer>();
		}

		public void insert(int n) {
			occur.put(n,occur.getOrDefault(n,0)+1);
			set.add(n);
		}

		public void delete(int n) {
			if(!set.contains(n)) {
				return;
			}
			occur.put(n,occur.get(n)-1);
			if(occur.get(n)==0) {
				set.remove(n);
			}
		}

		public int first() {
			return set.first();
		}
		public int last() {
			return set.last();
		}
	}
