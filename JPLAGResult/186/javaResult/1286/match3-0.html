class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int dp[] = new int[nums.length];
        PriorityQueue<Integer> q = new PriorityQueue<>((a, b)->dp[b]-dp[a]);
        int max = nums[0];
        dp[0] =nums[0];
        q.add(0);
        for(int i=1;i<nums.length;i++){
            while(q.peek()<i-k)q.poll();
            dp[i] = Math.max(nums[i], nums[i] + dp[q.peek()]);
            max = Math.max(dp[i], max);
            q.add(i);
        }
        return max;
    }
}
