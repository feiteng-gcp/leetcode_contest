class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        TreeMap<Integer, Integer> map = new TreeMap<>();
        int[] res = new int[n];
        
        int out = nums[0];
        map.put(nums[0], 1);
        res[0] = nums[0];
        for(int i = 1; i < n; i++){
            int max = map.lastKey();
            int val = Math.max(nums[i] + max, nums[i]);
            out = Math.max(out, val);
            res[i] = val;
            map.put(val, map.getOrDefault(val, 0) + 1);
            
            // evact
            int old = i - k;
            if(old >= 0){
                int count = map.get(res[old]);
                if(count == 1){
                    map.remove(res[old]);
                }
                else{
                    map.put(res[old], count - 1);
                }
            }
        }
        
        return out;
    }
    
}
