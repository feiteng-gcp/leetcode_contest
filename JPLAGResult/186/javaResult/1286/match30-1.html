import java.util.TreeMap;


class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int[] dp = new int[nums.length];
        TreeMap<Integer, Integer> lastK = new TreeMap<>((e1, e2) -> e2 - e1);
        int sum = 0;
        int max = Integer.MIN_VALUE;
        for (int i = 0; i <= k; i++) {
            max = Math.max(max, nums[i]);
            int v = sum + nums[i];
            dp[i] = v;
            lastK.put(v, lastK.getOrDefault(v, 0) + 1);
            sum = Math.max(sum, v);
        }

        for (int i = k + 1; i < nums.length; i++) {
            max = Math.max(max, nums[i]);
            int last = dp[i - k - 1];
            int v = lastK.get(last);
            if (v == 1) {
                lastK.remove(last);
            } else {
                lastK.put(last, v - 1);
            }
            int m = lastK.firstKey();
            if (m < 0) {
                m = 0;
            }
            m += nums[i];
            dp[i] = m;
            lastK.put(m, lastK.getOrDefault(m, 0) + 1);
            sum = Math.max(sum, m);
        }
        return max <= 0 ? max : sum;
    }
}
