class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int[] d = new int[nums.length];
        TreeSet<Integer> tree = new TreeSet<>(); 
        int res = -2147483648;
        for(int i=nums.length-1; i>=0; i--){
            if(tree.size()==0){
                tree.add(-nums[i]);
                d[i] = -nums[i];
            }else{
                if(i+k+1<nums.length){
                    tree.remove(d[i+k+1]);
                }
                int a = -tree.first();
                int b = Math.max(nums[i], nums[i] + a);
                tree.add(-b);
                d[i] = -b;
            }
            
res = Math.max(res, -tree.first());
            
        }
        return res;
        
    }
}
