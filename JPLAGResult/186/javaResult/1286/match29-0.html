class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
      int[] mem = new int[nums.length];
      Deque<Integer> q = new ArrayDeque<>();
      int res = 0;
      for(int i=nums.length-1; i>=0; i--) {
        if(i == nums.length-1) {
          mem[i] = nums[i];
          q.add(i);
          res = mem[i];
        }
        else {
          if(q.getLast()-i > k) q.pollLast();
          int cres;
          if(mem[q.getLast()] < 0) cres = nums[i];
          else cres = nums[i] + mem[q.getLast()];
          mem[i] = cres;
          res = Math.max(res, cres);
          while(!q.isEmpty()) {
            if(mem[q.peekFirst()] <= cres) q.pollFirst();
            else break;
          }
          q.addFirst(i);
        }
      }
      return res;
    }
  }
