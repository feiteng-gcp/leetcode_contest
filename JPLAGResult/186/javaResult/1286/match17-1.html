class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
         int n = nums.length;
          final int[] dyn = new int[n];
          TreeMap<Integer, Integer> prevs = new TreeMap<>();
          
for(int i = 0; i<n; ++i){
              dyn[i] = nums[i];
              if(prevs.size()>0) {
                  int t = prevs.lastKey();
                  if(t>0) dyn[i]+=t;
              }
              if(prevs.containsKey(dyn[i])) {
                  prevs.put(dyn[i], 1+ prevs.get(dyn[i]));
              }else{
                  prevs.put(dyn[i], 1);
              }
              
              if(i>=k){
                  int t = dyn[i-k];
                  int c = prevs.get(t);
                  if(c==1) {
                      prevs.remove(t);
                  }else {
                      prevs.put(t, c-1);
                  }
              }
          }
          return IntStream.range(0, n).map(i->dyn[i]).max().getAsInt(); 
    }
}
