class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        TreeSet<Integer> set = new TreeSet<>();
        dp[0] = nums[0];
        set.add(nums[0]);
        int res = dp[0];
        for (int i = 1; i < n; i++) {
            if (i > k) set.remove(dp[i - k - 1]);
            dp[i] = Math.max(nums[i], nums[i] + set.last());
            res = Math.max(res, dp[i]);
            set.add(dp[i]);
        }
        return res;
    }
}
