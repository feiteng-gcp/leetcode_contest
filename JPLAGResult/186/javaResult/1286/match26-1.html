class Solution {
    
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        dp[0] = nums[0];
        TreeMap<Integer, Integer> valToFreq = new TreeMap<>();
        int max = nums[0];
        
        for (int i = 1; i < n; i++) {
            // 1. add fast
            valToFreq.put(dp[i-1], 
valToFreq.getOrDefault(dp[i-1], 0) + 1);
            
            // 2. remove slow
            if (i >= k+1) {
                valToFreq.put(dp[i-1-k], valToFreq.get(dp[i-1-k])-1);
                if (valToFreq.get(dp[i-1-k]) == 0) {
                    valToFreq.remove(dp[i-1-k]);
                }
            }
            
            // 3. update result
            dp[i] = Math.max(0, valToFreq.lastKey()) + nums[i];
            max = Math.max(max, dp[i]);
        }
        
        
        return max;
    }
}
