class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int[] dp = new int[nums.length];
        int maxSum = nums[0];
        Deque<Integer> dq = new LinkedList<Integer>();   // Non-increasing queue
        for(int j = 0; j < nums.length; j++) {
            dp[j] = nums[j];
            if(!dq.isEmpty() && dq.getFirst() > 0) { dp[j] += dq.getFirst(); }
            maxSum = Math.max(maxSum, dp[j]);
            while(!dq.isEmpty() && dq.getLast() < dp[j]) { dq.removeLast(); }
            dq.addLast(dp[j]);
            if(j - k >= 0 && dq.getFirst().equals(dp[j - k])) { dq.removeFirst(); }
        }
        return maxSum;
    }
}
