class Solution {
    public int constrainedSubsetSum(int[] nums, int k) {
        int n = nums.length;
        int[] dp = new int[n];
        TreeMap<Integer, Integer> map = new TreeMap();
        map.put(0, 1);
        int count = 0;
        for(int i=0; i<n; i++){
            int prevMax = map.lastKey();
            int curMax = prevMax+nums[i];
            dp[i] = curMax;
            if(count==k){
                int toRemove = dp[i-k];
                int freq = map.get(toRemove);
                if(freq==1){
                    map.remove(toRemove);
                }
                else{
                    map.put(toRemove, freq-1);
                }
                count--;
            }
            map.put(curMax, map.getOrDefault(curMax, 0)+1);
            count++;
        }
        int res = Integer.MIN_VALUE;
        for(int i=n-1; i>=0; i--){
            res = Math.max(res, dp[i]);
        }
        return res;
    }
}
