class Solution {
    public int maxScore(int[] p, int k) {
        int n = p.length;
        int[] l = new int[n + 1];
        int[] r=  new int[n + 1]; 
        for (int i = 1; i <= n; i++) {
            l[i] = l[i - 1] + p[i - 1];
        }
        for (int i = n - 1; i >= 0; i--) {
            r[i] = r[i + 1] + p[i];
        }
        int ret = 0;
        for (int i = 0; i <= k; i++) {
            ret = Math.max(ret, l[i] + r[n - (k - i)]);
        }
        return ret;
    }
}
