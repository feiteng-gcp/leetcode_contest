class Solution {
    public int maxScore(int[] cardPoints, int k) {
      int res = 0;
      for(int i=0; i<k; i++) {
        res += cardPoints[i];
      }
      int rr = 1;
      int cres = res;
      for(int i=k-1; i>=0; i--) {
        cres -= cardPoints[i];
        cres += cardPoints[cardPoints.length-rr];
        rr++;
        res = Math.max(res, cres);
      }
      return res;
    }
  }
