class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int max = 0;
        int leftSum = 0;
        int rightSum = 0;
        for (int i = 0; i < k; i++) {
            rightSum += cardPoints[cardPoints.length - 1 - i];
        }
        max = rightSum;
        for (int i = 0; i < k; i++) {
            rightSum -= cardPoints[cardPoints.length - k + i];
            leftSum += cardPoints[i];
            max = Math.max(max, leftSum + rightSum);
        }
        return max;
    }
}
