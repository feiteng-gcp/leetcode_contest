class Solution {
    public int maxScore(int[] cards, int k) {
        int n = cards.length;
        // prefix sum
        int[] sum = new int[n+1];
        for (int i = 0; i < n; i++) {
            sum[i+1] = sum[i] + cards[i];
        
}
        int res = 0;
        for (int i = 0; i <= k; i++) {
            res = Math.max(res, sum[i] - sum[0] + sum[n] - sum[n-(k-i)]);
        }
        return res;
    }
}
