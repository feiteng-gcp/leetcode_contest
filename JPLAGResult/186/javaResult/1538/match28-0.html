class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int res = 0;
        for(int i = 0; i < k; ++i) res += cardPoints[i];
        int max = res;
        for(int i = cardPoints.length - 1; i >= cardPoints.length - k; --i) {
            res += cardPoints[i];
            res -= cardPoints[k - cardPoints.length + i];
            if(res > max) max = res;
        }
        return max;
    }
}
