	class Solution {
	    public int maxScore(int[] a, int k) {
	    	int n = a.length;
	        int[] cum = new int[n+1];
	        for(int i = 0;i < n;i++)cum[i+1] = cum[i] + a[i];
	        int ans = 0;
	        for(int i = 0;i <= k;i++){
	        	ans = Math.max(ans, cum[i] + cum[n] - cum[n-k+i]);
	        }
	        return ans;
	    }
	}
