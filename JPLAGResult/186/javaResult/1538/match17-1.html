class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int n = cardPoints.length;
        int maxi = 0, run = 0;

        for(int i = n - 1; i >= n - k; i--)
        {  
            run += cardPoints[i];
        }
        maxi = run;
        // System.out.println("..." + run);
        for(int i = 0, shift = n - k; i < k; i++)
        
{
            run += cardPoints[i] - cardPoints[shift];
            shift++;
            maxi = Math.max(maxi, run);
        }
        return maxi;
    }
}
