class Solution {
public:
    int maxScore(string s) {
        vector<int> zeros(s.size());
        int pre = 0;
        for (int i = 0; i < s.size(); i++) {
            zeros[i] = pre;
            if (s[i] == '0') pre++;
        }
        pre = 0;
        int res = 0;
        for (int i = s.size() - 1; i > 0; i--) {
            if (s[i] == '1') pre++;
            res = max(res, pre + zeros[i]);
            // cout << res;
        }
        return res;
    }
};
