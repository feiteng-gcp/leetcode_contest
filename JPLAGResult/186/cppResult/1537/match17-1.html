class Solution {
public:
    int maxScore(string s) {
        int ans = 0, cnt_zero = 0, cnt_one = 0;
        for(auto c : s) {
            if(c == '1') ++cnt_one;
        }
        for(int i = 0; i < s.size()-1; ++i ) {
            if(s[i] == '1')  {
                --cnt_one;
            }else {
                ++cnt_zero;
            }
            ans = max(ans, cnt_zero + cnt_one);
        }
        return ans;
    }
};
