class Solution {
public:
    int maxScore(string s) {
        vector<int> l(s.size(),0);
        int cnt=0;
        for (int i=0;i<s.size();i++)
        {
            if (s[i]=='0')
                cnt++;

            l[i]=cnt;
        }
        cnt=0;
        int res=0;
        for (int i=s.size()-1;i>=1;i--)
        {
            if (s[i]=='1')
                cnt++;
            //if (i!=s.size()-1)
                res=max(res,cnt+l[i-1]);
        }
        return res;
    }
};
