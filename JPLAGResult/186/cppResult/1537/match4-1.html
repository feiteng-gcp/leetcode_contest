class Solution {
public:
    int maxScore(string s) {
        int ans=0;
        for (int i=1; i<s.length(); i++) {
            int c1=0,c2=0;

            for (int j=i-1; j>=0; j--) {
                if (s[j]=='0') c1++;
            }
            for (int j=i; j<s.length(); j++) {
                if (s[j]=='1') c2++;
            }
            ans=max(ans,c1+c2);
        }
        return ans;
    }
};
