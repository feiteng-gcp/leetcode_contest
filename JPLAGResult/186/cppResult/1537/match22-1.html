class Solution {
public:
    int maxScore(string s) {
        int res=0;
        for(int mid=1;mid<s.size();mid++){

            int tres=0;
            for(int i=0;i<mid;i++)if(s[i]=='0')tres++;
            for(int i=mid;i<s.size();i++)if(s[i]=='1')tres++;
            res=max(res,tres);
        }
        return res;
    }
};
