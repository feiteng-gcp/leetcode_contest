class Solution {
public:
    
    int maxScore(string s) {
        int ans = 0;
        int cnt0 = 0, cnt1;
        for (int i = 0 ; i < s.length() - 1 ; i++){

            cnt1 = 0;
            cnt0 += s[i] == '0';
            for (int j = i + 1; j < s.length() ; j++)
                cnt1 += s[j] == '1';
            ans = max(ans , cnt0 + cnt1);
        }
        return ans;
    }
};
