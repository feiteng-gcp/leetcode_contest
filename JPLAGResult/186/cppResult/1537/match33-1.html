#include <bits/stdc++.h>
using namespace std;

typedef int INT;
typedef long long ll;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;


class Solution {
public:
    int maxScore(string s) {
        INT n = s.size();
		INT res = 0;
		for(int l=1; l<=n-1; l++){
			int cur = 0;
			for(int i=0; i<l; i++){
				cur += s[i] == '0';
			}
			for(int i=l; i<n; i++){
				cur += s[i] == '1';
			}
			res = max(res, cur);
		}

		return res;
    }
};
