class Solution {
public:
    int maxScore(string s) {
        int ans = 0;
        for( int i = 1; i < s.size(); i++ ) {
            int res = 0;
            for( int j = 0; j < i; j++ ) {
                if( s[ j ] == '0' ) res++;
            }
            for( int j = i; j < s.size(); j++ ) {
                if( s[ j ] == '1' ) res++;
            }
            ans = max( ans, res );
        }
        return ans;
    }
};
