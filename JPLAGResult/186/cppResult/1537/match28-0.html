class Solution {
public:
    int maxScore(string s) {
        int n = s.size();
        int ans = 0;
        vector<int> one(n + 1);
        for (int i = n - 1; i >= 0; --i)
            one[i] = one[i + 1] + (s[i] == '1');
        int zero = 0;
        for (int i = 0; i < n - 1; ++i) {
            zero += s[i] == '0';
            ans = max(ans, zero + one[i + 1]);
        }
        return ans;
    }
};
