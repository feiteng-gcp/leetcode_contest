class Solution {
public:
    int maxScore(string s) {
        int mx = 0;
        for (int i = 1; i < s.size(); i++) {
            int score = 0;
            for (int j = 0; j < i; j++)
                score += s[j] == '0';
            for (int j = i; j < s.size(); j++)
                score += s[j] == '1';
            mx = max(mx, score);
        }
        return mx;
    }
};
