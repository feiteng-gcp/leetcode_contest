class Solution {
public:

    int constrainedSubsetSum(vector<int>& nums, int k) {
        deque<pair<int, int>> q;
        int ans = INT_MIN;
        for (int i = 0; i < nums.size(); ++i) {
            while (q.size() > 0 && (q.front().first < (i - k) || q.front().second < 0) ) {
                q.pop_front();
            }
            // if (q.size() > 0){
            // cout << "first" << q.front().first << endl;
            // }
            
            int now = nums[i];
            if (q.size() > 0) {
                now += q.front().second;
            }
            
            //cout << now << "i; " << i << endl;
            if (now > ans) {
                ans = now;
            }
            
            while (q.size() > 0 && q.back().second <= now) {
                q.pop_back();
            }
            
            q.push_back(make_pair(i,now));
        }
        
        return ans;
    }
};
