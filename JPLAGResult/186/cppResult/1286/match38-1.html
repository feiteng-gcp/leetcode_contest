bool operator<(const pair<int, int> &a, const pair<int, int> &b){
    return a.first < b.first;
}


class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> dyn(n);
        priority_queue<pair<int, int>> pq;
        int ans = nums[0];
        for(int i=0;i<n;i++){
            int bef = 0;
            while(!pq.empty()){
                auto cur = pq.top();            
                if(cur.second < i-k){
                    pq.pop();
                    continue;
                }
                
                bef = max(bef, cur.first);
                break;
            }
            
            dyn[i] = nums[i];
            if(bef > 0)
                dyn[i] += bef;
            pq.push({dyn[i], i});
            ans = max(ans, dyn[i]);
        }
        
        return ans;
    }
};
