class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> dp(n);
        multiset<int, greater<int>> win;
        dp[0] = nums[0];
        win.insert(nums[0]);
        int result = nums[0];   //!1 nums[0] not INT_MIN
        for (int i = 1; i < n; i++) {
            int m = *win.begin();
            dp[i] = max(nums[i], nums[i] + m);
            win.insert(dp[i]);
            if (win.size() > k) {
                win.erase(win.find(dp[i - k]));
            }
            result = max(result, dp[i]);
        }
        return result;
    }
};
