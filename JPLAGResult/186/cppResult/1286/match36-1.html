class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        queue<int> q;
        deque<int> maxQ;
        int ans=-1e8;
        for(int i=0;i<nums.size();i++){
            int tmp=(maxQ.empty())?0:maxQ.front();
            tmp+=nums[i];
            tmp=max(nums[i],tmp);
            ans=max(ans,tmp);
            
            if(q.empty()){
                q.push(tmp);
                maxQ.push_back(tmp);
            }else{
                q.push(tmp);
                while(!maxQ.empty() && tmp>maxQ.back())
                    maxQ.pop_back();
                maxQ.push_back(tmp);
                while(q.size()>k){
                    int top=q.front();
                    q.pop();
                    if(top==maxQ.front())
                        maxQ.pop_front();
                }
            }
        }
        return ans;
    }
};
