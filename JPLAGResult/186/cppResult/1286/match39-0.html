struct Node{
    int pos;
    int value;
    Node(int a, int b):pos(a), value(b){}
};
struct NodeCmp{
 bool operator()(const Node a, const Node b)
 {
     return a.value < b.value;
 }
};
class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> dp(n + 1, 0);
        priority_queue<Node, vector<Node>, NodeCmp> max_dp;
        max_dp.push(Node(0,0));
        int ans = nums[0];
        for (int i = 1; i <= n; ++i)
        {
            Node max_node = max_dp.top();
            while (i - max_node.pos > k)
            {
                max_dp.pop();
                max_node = max_dp.top();
            }
            int now_ans = max(0, max_node.value) + nums[i - 1];
            ans = max(ans, now_ans);
            max_dp.push(Node(i, now_ans));
        }
        return ans;
    }
};
