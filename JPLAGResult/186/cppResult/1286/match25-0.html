class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        // res[i] means the max sub arrary end at i 
        // res[i] could be the lastest sub array end with i-1, i-2, ... i-k, plus current one,
        // need maitain the last largest k
        multiset<int> s;
        int n = nums.size();
        vector<int> res(n, INT32_MIN);
        int ret = INT32_MIN;
        for(int i = 0; i < n; ++i) {
            if(s.empty()) {
                s.insert(nums[i]);
                ret = max(ret, nums[i]);
                res[i] = nums[i];
                continue;
            }
            // find the max possible
            int cur = (*s.rbegin()) + nums[i];
            res[i] = max(cur, nums[i]);
            s.insert(res[i]);
            if(s.size() > k) {
                s.erase(res[i - k]);
            }
            
            ret = max(ret, res[i]);
        }
        
        return ret;
    }
};
