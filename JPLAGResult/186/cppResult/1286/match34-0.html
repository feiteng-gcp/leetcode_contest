class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        multiset<int> ms;
        vector<int> dp(nums.size());
        int ans = -100000;
        for (int i = nums.size() - 1; i >= 0; i--) {
            int tmp = nums[i];
            if (!ms.empty() && *ms.rbegin() > 0) {
                tmp += *ms.rbegin();
            }
            ans = max(ans, tmp);
            dp[i] = tmp;
            ms.insert(tmp);
            if (ms.size() > k) {
                ms.erase(ms.find(dp[i + k]));
            }
        }
        return ans;
    }
};
