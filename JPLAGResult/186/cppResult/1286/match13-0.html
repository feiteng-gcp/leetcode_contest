class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int ans = INT_MIN;
        priority_queue<pair<int, int>> q;
        for (int i = 0; i < nums.size(); i++) {
            int tmp = 0;
            while (!q.empty() && i - q.top().second > k)
                q.pop();
            if (!q.empty())
                tmp = max(0, q.top().first);
            tmp += nums[i];
            ans = max(ans, tmp);
            q.emplace(tmp, i);
        }
        return ans;
    }
};
