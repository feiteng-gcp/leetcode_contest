class Solution {
public:
    int constrainedSubsetSum(vector<int>& v, int k) {
        int n = v.size();
        vector<int> F(n+1);
        F[0] = v[0];
        multiset<int> S;
        S.insert(F[0]);
        
        for(int i=1;i<n;i++)
        {
            int val = *S.rbegin();
            F[i] = max(v[i]+val,v[i]);
            S.insert(F[i]);
            if(S.size()>k)
            {
                int x = F[i-k];
                auto it = S.lower_bound(x);
                S.erase(it);
            }
        }
        int ans = -1e9;
        for(int i=0;i<n;i++)
        {
            ans = max(ans,F[i]);
        }
        return ans;
        
    }
};
