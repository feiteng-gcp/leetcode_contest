class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        map<int, int> index;
        int n = nums.size();
        vector<int> dp = nums;
        for (int i = 0; i < n; ++i) {
            if (index.empty()) {
            } else {
                dp[i] = nums[i] + max(0, index.rbegin()->first);
            }
            index[dp[i]]++;
            if (i >= k) {
                index[dp[i-k]]--;
                if (index[dp[i-k]] == 0) {
                    index.erase(dp[i-k]);
                }
            }
        }
        return *max_element(dp.begin(), dp.end());
    }
};
