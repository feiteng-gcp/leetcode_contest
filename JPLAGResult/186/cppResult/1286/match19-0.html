class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n=nums.size();
        int dp[n];multiset<int>s;
        
        dp[n-1]=nums[n-1];
        s.insert(dp[n-1]);
        for(int i=n-2;i>=n-k;i--){
            dp[i]=max(nums[i],nums[i]+(*(--s.end())));
            s.insert(dp[i]);
        }
        
        for(int i=n-k-1;i>=0;i--){
            dp[i]=max(nums[i],nums[i]+(*(--s.end())));
            s.erase(dp[i+k]);
            s.insert(dp[i]);
        }
        
        int ans;
        ans=*max_element(dp,dp+n);
        
        return ans;
    }
};
