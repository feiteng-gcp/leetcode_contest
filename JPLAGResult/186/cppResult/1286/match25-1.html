class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        multiset<int> s;
        int n = nums.size();
        vector<int> dp(n + 1, 0);
        int res = -2e9; 
        
        for (int i = 0; i < n; i++){
            if (!s.empty())
                dp[i] = max(dp[i], *s.rbegin());
            dp[i] += nums[i];
            res = max(res, dp[i]);
            s.insert(dp[i]);
            if (s.size() > k)
                s.erase(s.find(dp[i - k]));
        }
        
        return res;
            
    }
};
