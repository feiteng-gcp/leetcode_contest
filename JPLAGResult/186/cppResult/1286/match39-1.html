class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int dp[100005];
        dp[0]=0;
        multiset<int> s;
        s.insert(0);
        int ans=-1e9;
        for(int i = 0;i<nums.size();i++){
            dp[i+1]=max(max(*s.rbegin(),0)+nums[i],nums[i]);
            s.insert(dp[i+1]);
            ans=max(ans,dp[i+1]);
            if(i+1>=k){
                s.erase(s.find(dp[i+1-k]));
            }
        }
        return ans;
    }
};
