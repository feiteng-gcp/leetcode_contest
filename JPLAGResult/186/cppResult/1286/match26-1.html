class Solution {
public:
    int constrainedSubsetSum(vector<int>& a, int k) {
        int n = a.size(), ans = a[0];
        // vector<int> f(n);
        deque<pair<int, int>> dq;
        
        for (int i = 0; i < n; ++i) {
            int cur = a[i];
            
            while (!dq.empty() && dq.front().first < i-k)
                dq.pop_front();
            
            if (dq.front().second > 0)
                cur += dq.front().second;
            
            ans = max(ans, cur);
            
            while (!dq.empty() && dq.back().second <= cur) {
                dq.pop_back();
            }
            dq.push_back({i, cur});
        }
        
        return ans;
    }
};
