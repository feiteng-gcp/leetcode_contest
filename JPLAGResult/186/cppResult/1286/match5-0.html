class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int n = nums.size() ;
        vector<int> dp(n + 1 , -0x7f7f7f) ;
        dp[0] = 0 ;
        deque<int> dq ;
        dq.push_front(0) ;
        for (int i = 1 ; i <= n ; ++i) {
            while (!dq.empty() && i - dq.front() > k)
                dq.pop_front() ;
            dp[i] = max(0 , dp[dq.front()]) + nums[i - 1] ;
            
            while (!dq.empty() && dp[dq.back()] <= dp[i]) 
                dq.pop_back() ;
            dq.push_back(i) ;
        }
        int res = -0x7f7f7f ;
        for (int i = 1 ; i <= n ; ++i) {
            
            res = max(res , dp[i]) ;
        }
            
        return res ;
    }
};
