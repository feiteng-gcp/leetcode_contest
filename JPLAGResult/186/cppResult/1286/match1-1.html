#include <bits/stdc++.h>
using namespace std;

typedef int INT;

typedef long long ll;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;

class Solution {
public:
    INT constrainedSubsetSum(vector<INT>& nums, INT k) {
        INT n = nums.size();
		vector<ll> dp(n, 0);
		multiset<ll> k_max;

		ll res = INT_MIN;
		dp[0] = nums[0];k_max.insert(nums[0]);
		res = max(dp[0], res);
		for(INT i=1; i<n; i++){
			INT a = nums[i];
			dp[i] = a;
			ll mx_k = *k_max.rbegin();
			dp[i] = max(dp[i], mx_k + a);
			k_max.insert(dp[i]);
			if(k_max.size() > k){
				INT j = i - k;
				ll v = dp[j];
				k_max.erase(v);
			}
	
			res = max(res, dp[i]);
		}

		return res;
		
    }
};
