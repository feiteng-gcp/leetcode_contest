class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        //dp[i]表示0到i 并且必须包含有nums[i]能求到的最好结果
        //dp[i] = nums[i] + max(dp[i - 1], dp[i - 2], ... dp[i - k]);
        int res = -(1 << 30);
        int n = nums.size();
        map<int, int> counter;
        vector<int> dp(n);
        for ( int i = 0; i < n; ++i ) {
            dp[i] = nums[i];
            if ( !counter.empty() ) dp[i] += max(0, (*counter.rbegin()).first);
            res = max(res, dp[i]);
            counter[dp[i]]++;
            if ( i >= k ) {
                if ( --counter[dp[i - k]] == 0 ) counter.erase(dp[i - k]);
            }
        }
        return res;
    }
};
