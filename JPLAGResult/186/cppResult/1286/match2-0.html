class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        vector<int> dp(nums);
        int mx = nums[0];
        multiset<int, greater<int>> ms = {nums[0]};
        for (int i = 1; i < nums.size(); i++) {
            dp[i] = max(nums[i], nums[i] + *ms.begin());
            mx = max(dp[i], mx);
            ms.insert(dp[i]);
            if (i >= k)
                ms.erase(ms.find(dp[i-k]));
        }
        return mx;
    }
};
