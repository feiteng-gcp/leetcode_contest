class Solution {
public:
    int constrainedSubsetSum(vector<int>& nums, int k) {
        int l = nums.size();
        vector<int> dp(l);
        int res = INT_MIN;
        deque<int> q;
        dp[l-1] = nums[l-1];
        q.push_back(l-1);
        for (int i = l-2; i >= 0; --i) {
            int t = nums[i];
            while (q.front() > i+k) q.pop_front();
            if (dp[q.front()] > 0) t += dp[q.front()];
            res = max(res, t);
            dp[i] = t;
            while (q.size() > 0 && dp[q.back()] <= t) q.pop_back();
            q.push_back(i);
        }
        return res;
    }
};
