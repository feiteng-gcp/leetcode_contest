class Solution {
public:
    int constrainedSubsetSum(vector<int>& N, int k) {
        multiset<int> can;
        N.insert(N.begin(), 0);
        int n = N.size();
        can.insert(0);
        vector<int> f(n, -(1<<30));
        for(int i = 1; i < n; i++){
            if(i > k + 1){
                can.erase(can.find(f[i - k - 1]));
            }
            f[i] = *prev(can.end()) + N[i];
            can.insert(f[i]);
        }
        return *max_element(f.begin(), f.end());
    }
};
