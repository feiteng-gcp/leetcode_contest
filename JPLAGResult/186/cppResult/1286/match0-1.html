class Solution {
public:
    int constrainedSubsetSum(vector<int>& ARR, int k) {
        int ANS_W = ARR[0];
        int MEMO_ARR[500000];
        auto cmp = [](vector<int>&x, vector<int>&y){
            return x[0]<y[0];    
        };
        priority_queue<vector<int>,vector<vector<int>>, decltype(cmp)> HEAP(cmp);
        MEMO_ARR[0] = ARR[0];
        HEAP.push({ARR[0], 0});
        for(int i=1; i<ARR.size(); ++i){
            MEMO_ARR[i] = ARR[i];
            while( !HEAP.empty() ){
                auto TEMP_VAR = HEAP.top(); 
                if ( i-TEMP_VAR[1]>k || TEMP_VAR[0]<0 ) HEAP.pop();
                else break;
            }
            if (HEAP.size()>0)
                MEMO_ARR[i]+=HEAP.top()[0];
            HEAP.push({MEMO_ARR[i], i});
            ANS_W = max(ANS_W, MEMO_ARR[i]);
        }
        return  ANS_W;
    }
};
