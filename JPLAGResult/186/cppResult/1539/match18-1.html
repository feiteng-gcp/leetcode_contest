class Solution {
 public:
  vector<int> findDiagonalOrder(vector<vector<int>>& nums) {
    map<int, vector<int>> mp;
    for (int i = nums.size() - 1; i >= 0; --i) {
      for (int j = 0; j < (int)nums[i].size(); ++j) {

        mp[i + j].push_back(nums[i][j]);
      }
    }
    vector<int> ans;
    for (auto& it : mp) {
      for (int& v : it.second) ans.push_back(v);
    }
    return ans;
  }
};
