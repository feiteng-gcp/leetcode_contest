typedef long long ll;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;



class Solution {
public:
    vector<int> findDiagonalOrder(vector<vector<int>>& nums) {
        int R = nums.size(), C = 0;
		for(int i=0; i<R; i++){
			C = max(C, (int)nums[i].size());
		}

		vector<vector<pii>> nums2(C);
		for(int i=0; i<R; i++){
			for(int j=0; j<nums[i].size(); j++){
				nums2[j].push_back(pii(i, nums[i][j]));
			}
		}

		vector<vector<int>> res(max(R, C) * 2);
		for(int j=0; j<C; j++){
			for(int i=0; i<nums2[j].size(); i++){
				pii p = nums2[j][i];
				int i2 = p.first, v = p.second;

				int i3 = i2 + j;
				res[i3].push_back(v);
			}
		}

		vector<int> ans;
		for(int i=0; i<res.size(); i++){
			for(int j=0; j<res[i].size(); j++){
				ans.push_back(res[i][j]);
			}
		}
		return ans;
    }
};
