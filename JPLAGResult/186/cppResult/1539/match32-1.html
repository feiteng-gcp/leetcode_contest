class Solution {
public:
    vector<int> findDiagonalOrder(vector<vector<int>>& a) {
        int n = (int)(a.size());

        vector<vector<int>> M;
        for (int i = 0; i < n; ++i)
        {
            for (int j = 0; j < (int)(a[i].size()); ++j)

            {
               M.push_back({i+j, j, a[i][j]});
            }
        }

      sort(M.begin(), M.end());
        vector<int> ans;
        for(auto it: M){
            ans.push_back(it[2]);
        }

        return ans;
    }
};
