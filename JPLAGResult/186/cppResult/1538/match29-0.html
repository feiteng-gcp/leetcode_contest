class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int sum = 0;
        for (int i = 0; i < k; i++) sum += cardPoints[i];
        int res = sum;
        for (int i = k - 1, j = cardPoints.size() - 1; i > -1; i--, j--) {
            sum -= cardPoints[i] - cardPoints[j];
            res = max(res, sum);
        }
        return res;
    }
};
