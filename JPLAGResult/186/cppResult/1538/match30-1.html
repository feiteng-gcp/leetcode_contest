class Solution {
public:
    
    int maxScore(vector<int>& a, int k) {
        
        
        int n = a.size();
        vector<int> s1(n + 100, 0), s2(n + 100, 0);
        
        for (int i = 1; i <= n; i++)
        {
            s1[i] = s1[i - 1] + a[i - 1];
        }
        
        int cnt = 0;
        for (int i = n; i >= 1; i--)
        {
            cnt += 1;
            s2[cnt] = s2[cnt - 1] + a[i - 1];
        }
        

        int ans = 0;
        for (int i = 0; i <= k; i++)
        {
            ans = max(ans, s1[i] + s2[k - i]);
        }
        
        return ans;
        
        
    }
};
