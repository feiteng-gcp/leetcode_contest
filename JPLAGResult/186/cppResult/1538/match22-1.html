class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n=cardPoints.size();
        int pref[100000];
        int suff[100000];
        for (int i=1; i<=k; ++i) {
            pref[i]=pref[i-1]+cardPoints[i-1];
            suff[i]=suff[i-1]+cardPoints[n-i];
        }

        int ans=0;
        for (int i=0; i<=k; ++i) {
            ans=max(ans, pref[i]+suff[k-i]);
        }
        return ans;
    }
};
