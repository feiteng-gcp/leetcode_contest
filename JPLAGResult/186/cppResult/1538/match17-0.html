class Solution {
public:
    int maxScore(vector<int>& x, int k) {
        int n = x.size();
        vector<int> v(n + 1);
        for (int i = 0; i < n; i++)
            v[i + 1] = v[i] + x[i];
        int ans = 0;
        for (int i = 0; i <= k; i++)
            ans = max(ans, v[k - i] + v.back() - v[n - i]);
        return ans;
    }
};
