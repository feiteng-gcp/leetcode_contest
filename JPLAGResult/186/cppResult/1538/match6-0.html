class Solution {
public:
    int maxScore(vector<int>& v, int k) {
        int n = v.size();
        vector<vector<int> > sum(n+1,vector<int>(2,0));
        for(int i = 0;i < n;i++){
            sum[i+1][0] = sum[i][0] + v[i];
            sum[n-i-1][1] = sum[n-i][1] + v[n-i-1];
        }
        if(k >= n) return sum[n][0];
        int ans = 0;
        for(int i = 0;i <= k;i++){
            //cout<<sum[i][0]+sum[n-k-i][1]<<endl;
            ans = max(ans,sum[i][0]+sum[n-k+i][1]);
        }
        return ans;
    }
};
