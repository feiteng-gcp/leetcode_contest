class Solution {
public:
    int maxScore(vector<int>& c, int k) {
        int n = c.size();
        vector<int> v(n+1, 0);
        for(int i=1;i<=n;i++){
            v[i] = v[i-1] + c[i-1];
        }
        int m = 0;
        for(int i=0;i<=k;i++){
            m = max(m, v[i] + v[n] - v[n-k+i]);
        }
        return m;
    }
};
