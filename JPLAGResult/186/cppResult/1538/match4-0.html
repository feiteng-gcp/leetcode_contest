class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int size = cardPoints.size();
        vector<int> left(size + 1, 0);
        vector<int> right(size + 1, 0);
        for (int i = 1; i <= k; i++) {
            left[i] = left[i - 1] + cardPoints[i - 1];
            right[size - i] = right[size - i + 1] + cardPoints[size - i];
        }
        int ret = 0;
        for (int i = 0; i <= k; i++) {
            // cout << i << " " << left[i] << " " << right[size - k + i] << endl;
            ret = max(ret, left[i] + right[size - k + i]);
        }
        return ret;
    }
};
