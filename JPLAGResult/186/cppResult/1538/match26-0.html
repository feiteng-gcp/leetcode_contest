typedef long long ll;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;

class Solution {
public:
	ll query(int b, int e, vector<ll>& sc){
		if(b > e) return 0;
		int n = sc.size();
		ll res = sc[e];
		if(b > 0){
			res -= sc[b-1];
		}
		return res;
	}

    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size();
		vector<ll> sc = vector<ll>(n, 0);
		for(int i=0; i<n; i++){
			sc[i] = cardPoints[i];
			if(i > 0){
				sc[i] += sc[i-1];
			}
		}
		ll res = 0;
		for(int l=0; l<=k; l++){
			int r = k - l;
			ll cur = query(0, l-1, sc) + query(n-r, n-1, sc);
			res = max(res, cur);
		}

		return res;
    }
};
