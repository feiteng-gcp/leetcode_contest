#include <bits/stdc++.h>
using namespace std;

typedef int INT;

typedef long long ll;
typedef pair<INT,INT> pii;
typedef pair<ll,ll> pll;

class Solution {
public:
	ll query(INT b, INT e, vector<ll>& sc){
		if(b > e) return 0;
		INT n = sc.size();
		ll res = sc[e];
		if(b > 0){
			res -= sc[b-1];
		}
		return res;
	}

    INT maxScore(vector<INT>& cardPoINTs, INT k) {
        INT n = cardPoINTs.size();
		vector<ll> sc = vector<ll>(n, 0);
		for(INT i=0; i<n; i++){
			sc[i] = cardPoINTs[i];
			if(i > 0){
				sc[i] += sc[i-1];
			}
		}
		ll res = 0;
		for(INT l=0; l<=k; l++){
			INT r = k - l;
			ll cur = query(0, l-1, sc) + query(n-r, n-1, sc);
			res = max(res, cur);
		}

		return res;
    }
};
