class Solution {
public:
    int maxScore(vector<int>& a, int k) {
        int n=a.size();
        vector<int> sum(n+1);
        for(int i=1;i<=n;++i) sum[i]=sum[i-1]+a[i-1];
        int ret=0;
        for(int i=0;i<=k;++i) {
            ret=max(ret,sum[i]+sum[n]-sum[n-(k-i)]);
        }
        return ret;
    }
};
