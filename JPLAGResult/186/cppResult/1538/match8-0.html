class Solution {
public:
    int maxScore(vector<int>& A, int k) {
        int n = A.size();
        vector<int> presum(n + 1, 0);
        for(int i = 1; i <= n; ++i) presum[i] = presum[i - 1] + A[i - 1];
        int res = 0;
        for(int i = 0; i <= k; ++i) {
            res = max(res, presum[i] + presum[n] - presum[n - (k - i)]);
        }
        return res;
    }
};
