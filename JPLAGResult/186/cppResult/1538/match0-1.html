class Solution {
public:
  int maxScore(vector<int>& p, int k) {
    int sum = 0;
    for (int i = 0; i < k; ++i) sum += p[i];
    
    int j = (int) p.size() - 1, res = sum;
    for (int i = k - 1; i >= 0; --i) {
      sum = sum - p[i] + p[j--];
      res = max(res, sum);
    }

    return res;
  }
};
