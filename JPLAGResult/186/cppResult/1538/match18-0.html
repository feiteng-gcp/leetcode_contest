class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size();
        vector<int> presum(n+1, 0);
        for (int i = 1; i <= n; i++) {
            presum[i] = presum[i-1] + cardPoints[i-1];
        }
        int ans = 0;
        for (int i = 0; i <= k; i++) {
            ans = max(ans, presum[i] + presum[n] - presum[n-k+i]);
        }
        return ans;
    }
};
