class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size();
        vector<int> pref(n+1, 0);
        for(int i = 0 ;i < n; i++){
            pref[i+1] = pref[i] + cardPoints[i];
        }
        int ans = 0;
        for(int i = 0; i <= k; i++){
            int lef = k-i;
            //cout<<pref[n] - pref[n-lef-1]<<" "<<n-lef-1<<"\n";
            ans = max(ans, pref[i] + pref[n] - pref[n-lef]);
        }
        return ans;
    }
};
