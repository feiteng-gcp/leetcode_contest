class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int len = cardPoints.size() - k;
        int sum = 0, subsum = 0;
        for (int i = 0; i < cardPoints.size(); ++i) sum += cardPoints[i];

        for (int i = 0; i < len; ++i) subsum += cardPoints[i];
        int ret = sum - subsum;
        for (int i = len; i < cardPoints.size(); ++i) {
            subsum += cardPoints[i] - cardPoints[i - len];
          //  printf("%d %d %d\n", i, i - len, subsum);
            ret = max(ret, sum - subsum);
        }
        return ret;
    }
};
