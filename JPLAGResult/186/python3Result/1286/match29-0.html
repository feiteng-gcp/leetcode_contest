from collections import deque

class Solution:
	def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
		mxHere = -1
		dp = [float('inf')]*len(nums)
		queue = deque() 

		# mx, mxi = -float('inf'), -1
		for i in range(k): 
			if i == 0: 
				dp[i] = nums[i]
			else:
				dp[i] = max(0,dp[queue[0]]) + nums[i]
			while queue and dp[i] >= dp[queue[-1]] : 
				queue.pop() 
			queue.append(i)

			# print('i={}, num={}, mx={}, mxHere={}, queue={}'.format(i, nums[i], mx,mxHere,queue))

		for i in range(k,len(nums)): 
			mxi = queue[0]
			dp[i] = max(0,dp[mxi]) + nums[i]
			 
			while queue and queue[0] <= i-k: 
				queue.popleft()  
			  
			while queue and dp[i] >= dp[queue[-1]] : 
				queue.pop() 
			  
			queue.append(i)
			# print('i={}, num={}, mx={}, mxHere={}, queue={}'.format(i, nums[i], mx,mxHere,queue))


		res = max(dp)
		return res if res < float('inf') else -1
