class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        # [10,2,-10,5,20], k=2
        que = collections.deque([(nums[0], 0)])
        max_sum = nums[0]
        for i in range(1, len(nums)):
            new_sum = max(que[0][0]+nums[i], nums[i])
            max_sum = max(max_sum, new_sum)
            while que and que[-1][0] <= new_sum:
                que.pop()
            que.append((new_sum, i))
            while que and que[0][1] <= i-k:
                que.popleft()
            # print(que, new_sum, max_sum)
                
        return max_sum
