import heapq

from typing import List


class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        dp=[]
        heap=[] #[max, idx]

        for i, num in enumerate(nums):
            while heap and i-heap[0][1]>k:
                heapq.heappop(heap)

            if heap:
                dp.append(max(-heap[0][0]+num, num))
            else:
                dp.append(num)

            heapq.heappush(heap, (-dp[i], i))

        return max(dp)
