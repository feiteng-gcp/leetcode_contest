from collections import deque
class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        f = [float('-inf')] * len(nums)
        res = float('-inf')
        q = deque()
        for i in range(len(nums) - 1, -1, -1):
            f[i] = nums[i]
            while q and q[0][0] > i + k:
                q.popleft()
            if q and q[0][1] > 0:
                f[i] += q[0][1]
            while q and q[-1][1] <= f[i]:
                q.pop()
            q.append((i, f[i]))
            res = max(res, f[i])
        return res
