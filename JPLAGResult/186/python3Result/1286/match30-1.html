class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        import heapq
        curr=[]
        ret=-100000
        for i in range(len(nums)):
            prev=0
            while len(curr)>0:
                total,ind = heapq.heappop(curr)
                if i-ind<=k:
                    heapq.heappush(curr,(total,ind))
                    prev=max(prev,-total)
                    
break
            heapq.heappush(curr,(-(prev+nums[i]),i))
            ret=max(ret,(prev+nums[i]))
        return ret
