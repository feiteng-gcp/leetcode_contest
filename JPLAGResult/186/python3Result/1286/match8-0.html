class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        import collections
        n = len(nums)
        dp = [nums[i] for i in range(n)]
        q = collections.deque([(dp[0],0)])
        for i in range(1,n):
            dp[i] = max(dp[i],q[0][0]+nums[i])
            if q[0][1] == i-k:
                q.popleft()
            while q and q[-1][0] <= dp[i]:
                q.pop()
            q.append((dp[i],i))
        return max(dp)
