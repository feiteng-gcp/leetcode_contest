from collections import deque
class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        dp 
= [-float('inf') for i in range(n+1)]
        dq = deque()
        for i in range(n):
            if not dq:
                dp[i] = nums[i]
            else:
                dp[i] = max(nums[i], dp[dq[0]] + nums[i])
            while dq and dp[i] >= dp[dq[-1]]:
                dq.pop()
            while dq and dq[0] <= i-k:
                dq.popleft()
            dq.append(i)
        return max(dp)
