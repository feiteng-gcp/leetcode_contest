from collections import deque
class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        res = max(nums)
        d = deque()
        dp = [0] * n
        for i in range(n):
            if d and d[0] + k < i:
                d.popleft()
            num = nums[i]
            # print(d, num)
            dp[i] = max(0, dp[d[0]]) + num if d else num
            res = max(res, dp[i])
            while d and dp[d[-1]] <= dp[i]:
                d.pop()
            d.append(i)
            
        return res
