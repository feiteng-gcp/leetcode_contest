from collections import deque
class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        queue = deque()
        dp = [num for num in nums]
        ans = max(dp)
        for i in range(0, len(nums)):
            if len(queue) > 0:
                val, u = queue.popleft()
                if i - u <= k:
                    dp[i] = max(dp[i], nums[i] + val)
                    queue.appendleft((val, u))
                else:
                    if len(queue) > 0:
                        val, u = queue.popleft()
                        dp[i] = max(dp[i], nums[i] + val)
                        queue.appendleft((val, u))
            while len(queue) > 0:
                val, u = queue.pop()
                if val > dp[i]:
                    queue.append((val, u))
                    break
            queue.append((dp[i], i))
            ans = max(dp[i], ans)
        return ans
