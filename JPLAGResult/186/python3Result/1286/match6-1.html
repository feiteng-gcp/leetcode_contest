import heapq
class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        
        dp = [x for x in nums]
        
        sum_lst = []
        
        for i,x in enumerate(nums):
            while sum_lst and sum_lst[0][1] + k < i:
                heapq.heappop(sum_lst)
            if not sum_lst:
                heapq.heappush(sum_lst,(-x,i))
                dp[i] = x
                continue
                
            dp[i] = x -sum_lst[0]
[0] if -sum_lst[0][0] > 0 else x
            heapq.heappush(sum_lst,(-dp[i],i))
            
        return max(dp)
