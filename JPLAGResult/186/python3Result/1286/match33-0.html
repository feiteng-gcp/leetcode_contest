# 带滑动窗口的dp
import bisect 

class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        dp = [0] * n
        dp[0] = nums[0]
        heap = [(-dp[0], 0)]
        heapify(heap)
        ans = dp[0]
        for i in range(1, n):
            num = nums[i]
            j = max(0, i-k)
            while heap:
                dp_j, j = heapq.heappop(heap)
                if j >= i - k:
                    heapq.heappush(heap, (dp_j, j))
                    break
            
            dp[i] = max(0, -dp_j) + num
            
            # dp[i] = max(0, max(dp[j: i])) + num
            if dp[i] > 0:
                heapq.heappush(heap, (-dp[i], i))
            
            ans = max(ans, dp[i])
       
            
        return(ans)
