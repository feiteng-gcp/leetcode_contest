class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        dq = deque()
        f 
= []
        for i in range(0, len(nums)):
            x = 0
            while (len(dq) > 0):
                x = dq.popleft()
                if x >= i - k:
                    dq.appendleft(x)
                    break   
            temp = 0
            if len(dq) > 0: 
                x = dq.popleft()
                temp = f[x]
                dq.appendleft(x)
            f.append(max(nums[i], nums[i] + temp))
            while len(dq) > 0:
                x = dq.pop()
                if f[x] > f[i]:
                    dq.append(x)
                    break
            dq.append(i)
            
            #print(dq)
       # print(f)
        return max(f)
