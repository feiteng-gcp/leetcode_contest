import heapq

class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        # dp, dp[j] is the maximum when ends with nums[j], have to search the max in the pervious k places, priority queue to maintain maximum place
        lenn = len(nums)
        dp = [0] * lenn
        dp[0] = nums[0]
        q = [(-dp[0], 0)]
        heapq.heapify(q)
        max_rst, place = q[0]
        for i in range(1, lenn):
            # find the maximum from q
            max_rst, place = q[0]
            while place < i - k:
                max_rst, place = heapq.heappop(q)
            dp[i] = max(nums[i] - max_rst, nums[i])
            heapq.heappush(q, (-dp[i], i))
        print(dp)
        return max(dp)
