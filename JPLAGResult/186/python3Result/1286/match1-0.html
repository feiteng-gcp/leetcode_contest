import heapq


class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        pq = [(-nums[0], 0)]
        ans = nums[0]
        for i in range(1, len(nums)):
            while i-pq[0][1] > k:
                heapq.heappop(pq)
            cur = max(-pq[0][0]+nums[i], nums[i])
            if cur > ans:
                ans = cur
            heapq.heappush(pq, (-cur, i))
            
        return ans
