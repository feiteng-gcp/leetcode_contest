from collections import deque

class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        res = -float('inf')
        q = deque()
        for i, num in enumerate(nums):
            while q and i - q[0][1] > k:
                q.popleft()
            if q:
                t = max(0, q[0][0]) + num
            else:
                t = num
            res = max(res, t)
            
while q and q[-1][0] <= t:
                q.pop()
            q.append((t, i))
        return res
