class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        hqueue = [(-nums[0], 0)]
        T = [nums[0]]
        
        for i in range(1, len(nums)):
            while hqueue[0][1] < i - k:
                heapq.heappop(hqueue)
                
            maxval = -hqueue[0][0]
            T.append(max(nums[i], maxval+nums[i]))
            heapq.heappush(hqueue, (-T[-1], i))
                    
        return max(T)
