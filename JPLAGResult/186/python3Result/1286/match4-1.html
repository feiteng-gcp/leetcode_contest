class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        dp = []
        ans = -float('inf')
        
for i in range(n):
            while dp and i-dp[0][-1] > k:
                heapq.heappop(dp)
            if not dp:
                heapq.heappush(dp,(-nums[i],i))
            else:
                heapq.heappush(dp,(-max(-dp[0][0],0)-nums[i],i))
            ans = max(ans,-dp[0][0])
            # print(dp, ans)
        return ans
