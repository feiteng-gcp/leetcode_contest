from collections import deque

class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n=len(nums)
        dp=[0]*n
        dp[0]=nums[0]
        q=deque()
        q.append((0,nums[0]))
        for i in range(1,n):
            while q[0][0]<i-k:q.popleft()
            dp
[i]=max(nums[i],q[0][1]+nums[i])
            while q and q[-1][1]<=dp[i]:q.pop()
            q.append((i,dp[i]))
        return max(dp)
