import sortedcontainers

class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        dp = [0] * n
        dp[0] = nums[0]
        s = sortedcontainers.SortedList([dp[0]])
        ans = dp[0]
        for i in range(1,n):
            # print(s,dp)
            dp[i] = max(nums[i], s[-1] + nums[i])
            s.add(dp[i])
            if i-k >= 0: s.remove(dp[i-k])
            ans = max(ans, dp[i])
        return ans
