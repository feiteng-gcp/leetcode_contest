class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        n = len(nums)
        c = [0] * n
        c[-1] = nums[-1]
        
        q = []
        q.append(c[-1])
        m = c[-1]
        for i in range(1, n):
            idx = n-i-1
            s = max(nums[idx], nums[idx] + m)
            c[idx] = s
            q.append(s)
            m = max(m, s)
            if len(q) > k:
                p = q.pop(0)
                if p == m:
                    m = max(q)
        return max(c)
