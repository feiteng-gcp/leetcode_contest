class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        dp = nums[:]
        wd = []
        for i, n in enumerate(nums):
            if wd and wd[0][1] < i - k:
                wd.pop(0)
            
if wd and wd[0][0] > 0:
                dp[i] += wd[0][0]
            while wd and wd[-1][0] <= dp[i]:
                wd.pop()
            wd.append((dp[i], i))
        return max(dp)
