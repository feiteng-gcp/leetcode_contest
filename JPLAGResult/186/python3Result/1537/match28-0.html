class Solution:
    def maxScore(self, s: str) -> int:
        res = 0
        l0, r1 = 0, s.count('1')
        for i in range(len(s) - 1):
            if s[i] == '0':
                l0 += 1
            else:
                r1 -= 1
            res = max(res, l0 + r1)
        return res
