class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        total = sum(cardPoints)
        if k == len(cardPoints):
            return total
        p = len(cardPoints) - k
        current = sum(cardPoints[:p])
        remain = current
        for i in range(p, len(cardPoints)):
            current += cardPoints[i]
            current -= cardPoints[i - p]
            remain = min(remain, current)
        return total - remain
