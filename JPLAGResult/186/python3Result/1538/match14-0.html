class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        n, sums = len(cardPoints), sum(cardPoints)
        presum = sum(cardPoints[:n-k])
        res = presum
        for i in range(n-k,n):
            presum += cardPoints[i]-cardPoints[i-(n-k)]
            res = min(res, presum)
        return sums-res
