class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        n = len(cardPoints)
        size = n - k

        cur = tot = sum(cardPoints[:size]) # for n-k consecutive arr
        for i in range(size, n):
            cur += cardPoints[i]
            cur -= cardPoints[i - size]
            tot = min(cur, tot)

        return sum(cardPoints) - tot
