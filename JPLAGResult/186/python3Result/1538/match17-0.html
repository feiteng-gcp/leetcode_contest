class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        n = len(cardPoints)
        minus = sum(cardPoints[:-k])
        min_minus = minus

        for i in range(k):
            minus = minus - cardPoints[i] + cardPoints[n - k + i]
            min_minus = min(minus, min_minus)

        return sum(cardPoints) - min_minus
