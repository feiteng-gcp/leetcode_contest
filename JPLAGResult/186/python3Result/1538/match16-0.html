class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        
        n = len(cardPoints)
        res = window = sum(cardPoints[-k:])
        
        cards = cardPoints * 2
        for i in range(n-k, n):
            window -= cards[i]
            window += cards[i+k]
            res = max(res, window)
        return res
