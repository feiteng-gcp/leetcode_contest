class Solution:
    def maxScore(self, cardPoints, k: int) -> int:
        s = sum(cardPoints[:k])
        ans = s
        for i in range(1, k + 1):
            s -= cardPoints[k - i]
            s += cardPoints[len(cardPoints) - i]
            ans = max(ans, s)
            # print(s)
        return ans
