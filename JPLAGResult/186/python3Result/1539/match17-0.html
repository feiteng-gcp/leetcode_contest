class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        cur = []
        
        for i in range(len(nums)):
            for j in range(len(nums[i])):
                if len(cur) < i+j+1:
                    cur.append([])
                cur[i+j].append(nums[i][j])
        
        res = []
        for x in cur:
            res += x[::-1]
            
        return res
