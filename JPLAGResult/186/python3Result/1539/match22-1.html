class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        res = []
        m = len(nums)
        heap = []
        for i in range(m):
            for j in range(len(nums[i])):
                heapq.heappush(heap, [i+j,-i,nums[i][j]])
        
        while heap:
            res.append(heapq.heappop(heap)[2])
        return res
