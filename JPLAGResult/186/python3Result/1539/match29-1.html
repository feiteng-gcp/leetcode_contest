class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        trans=[]
        for i in range(len(nums)):
            for j in range(len(nums[i])):
                
if len(trans)<=i+j:
                    trans.append([])
                trans[i+j].append(nums[i][j])
        ret=[]
        for i in range(len(trans)):
            for j in range(len(trans[i])-1,-1,-1):
                ret.append(trans[i][j])
        return ret
