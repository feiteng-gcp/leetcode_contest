class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        arr = []
        for i, row in enumerate(nums):
            for j, val in enumerate(row):
                arr.append([i+j-1, j, val])
        arr.sort()
        ans = []
        for i in arr:
            ans.append(i[2])
        return ans
