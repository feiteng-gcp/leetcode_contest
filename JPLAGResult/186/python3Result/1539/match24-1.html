class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        diago_nums = collections.defaultdict(list)
        
        for i in range(len
(nums)):
            for j in range(len(nums[i])):
                diago_nums[i+j].append(nums[i][j])
        
        res = []
        for diago in sorted(diago_nums.keys()):
            res += diago_nums[diago][::-1]
            
        return res
