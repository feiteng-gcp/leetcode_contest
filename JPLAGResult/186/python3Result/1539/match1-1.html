class Solution:
    def findDiagonalOrder(self, nums: List[List[int]]) -> List[int]:
        ans = []
        for i, line in enumerate(nums):
            for j, num in enumerate(line):
                ans.append((i+j,j,num))
                

        ans.sort()
        return [x[2] for  x in ans]
