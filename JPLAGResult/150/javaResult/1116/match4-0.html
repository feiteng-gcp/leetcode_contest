/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int maxLevelSum(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        int max = Integer.MIN_VALUE;
        int maxLevel = 1;
        int level = 0;
        
        while(!q.isEmpty()){
            int size = q.size();
            level++;
            int sum = 0;
            while(size-- > 0){
                TreeNode top = q.poll();
                sum += top.val;
                if(top.left != null)
                    q.offer(top.left);
                if(top.right != null)
                    q.offer(top.right);
            }
            if(sum > max){
                max = sum;
                maxLevel = level;
            }
        }
        return maxLevel;
    }
}
