/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int maxLevelSum(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList();
        queue.offer(root);
        int level = 0, max = Integer.MIN_VALUE, curL = 0;;
        while (!queue.isEmpty()) {
            curL++;
            int size = queue.size();
            int sum = 0;
            while (size > 0) {
                TreeNode now = queue.poll();
                sum += now.val;
                if (now.left != null)
                    queue.offer(now.left);
                if (now.right != null)
                    queue.offer(now.right);
                size--;
            }
            if (sum > max) {
                max = sum;
                level = curL;
            }
        }
        return level;
    }
}
