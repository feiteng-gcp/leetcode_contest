/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int maxLevelSum(TreeNode root) {
        long result = 0;
        int level = 1;
        int curLevel = 1;
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            int size = q.size();
            long curSum = 0;
            for (int i = 0; i < size; i++) {
                TreeNode node = q.poll();
                curSum += node.val;
                if (node.left != null) {
                    q.add(node.left);
                }
                if (node.right != null) {
                    q.add(node.right);
                }
            }
            if (curSum > result) {
                result = curSum;
                level = curLevel;
            }
            curLevel ++;
        }
        return level;
    }
}
