class Solution {
    List<Long> list = new ArrayList<>();
    public int maxLevelSum(TreeNode root) {
        help(root, 1);
        long max = list.get(0);
        int res = 1;
        for (int i=0;i<list.size();i++) {
            if (max < list.get(i)) {
                max = list.get(i);
                res = i+1;
            }
        }
        return res;
    }
    private void help(TreeNode root, int depth) {
        if (root == null) return;
        if (list.size() < depth) {
            list.add((long)root.val);
        }else {
            list.set(depth-1, list.get(depth-1)+root.val);
        }
        help(root.left, depth+1);
        help(root.right, depth+1);
    }
}
