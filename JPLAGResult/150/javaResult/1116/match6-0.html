/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public int maxLevelSum(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        int max = 1 << 31;
        int sol = 0;
        int lev = 1;
        while (!q.isEmpty()) {
            int sum = 0;
            int len = q.size();
            for (int i = 0; i < len; i++) {
                TreeNode cur = q.poll();
                sum += cur.val;
                if (cur.left != null) q.add(cur.left);
                if (cur.right != null) q.add(cur.right);
            }
            if (sum > max) {
                max = sum;
                sol = lev;
            }
            lev++;
        }
        return sol;
    }
}
