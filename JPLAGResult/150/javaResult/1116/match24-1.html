/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public int maxLevelSum(TreeNode root) {
        int max = root.val;
        int midx = 1;
        
        int level = 1;
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        
        while(!queue.isEmpty()) {
            int size = queue.size();
            int sum = 0;
            while(size-- > 0) 
{
                TreeNode node = queue.poll();
                sum += node.val;
                
                if(node.left != null) {
                    queue.add(node.left);
                }
                
                if(node.right != null) {
                    queue.add(node.right);
                }
            }
            
            
if(sum > max) {
                max = sum;
                midx = level;
            }
            ++level;
        }
        
        return midx;
    }
}
