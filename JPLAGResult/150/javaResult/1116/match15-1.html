/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    
    public int maxLevelSum(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        int max = Integer.MIN_VALUE, level = -1;
        int l = 0;
        while(!q.isEmpty()){
            int size = q.size();
            int curSum = 0;
            for(int i=0;i<size;i++){
                TreeNode out = q.poll();
                curSum += out.val;
                if(out.left != null){
                    q.offer(out.left);
                }
                if(out.right != null){
                    q.offer(out.right);
                }
            }
            if(max < curSum){
                max = curSum;
                level = l;
            }
            l++;
        }
        
        return level+1;
    }
}
