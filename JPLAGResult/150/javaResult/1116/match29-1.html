/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public int maxLevelSum(TreeNode root) {
        
        Queue<TreeNode> q = new LinkedList<>();
        int level = 1;
        int max = Integer.MIN_VALUE;
        int res = 0;
        
        q.offer(root);
        while (!q.isEmpty()) {
            int tmp = 0;
            for (int k = q.size(); k > 0; k--) 
{
                TreeNode cur = q.poll();
                tmp += cur.val;
                
                if (cur.left != null) {
                    q.offer(cur.left);
                }
                
                if (cur.right != null) {
                    q.offer(cur.right);
                }
            }
            
            // System.out.println(tmp);
            if (tmp > max) {
                res = level; 
                max = tmp;
            }
            level++;
        }
        
        return res;
    }
}
