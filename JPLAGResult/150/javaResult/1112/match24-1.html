class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] count = new int[26];
        for(char c : chars.toCharArray()) {
            count[c - 'a']++;
        }
        
        int res = 0;
        for(String s : words) {
            int[] temp = new int[26];
            for(char c : s.toCharArray()) {
                temp[c - 'a']++;
            }
            boolean toAdd = true;
            for(int i = 0; i < 26; i++) {
                if(temp[i] > count[i]) {
                    toAdd = false;
                    break;
                
}
            }
            if(toAdd) res += s.length();
        }
        return res;
    }
}
