class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] alpha = new int[26];
        for(char ch : chars.toCharArray()) {
            alpha[ch - 'a']++;
        }
        int res = 0;
        for(String word : words) {
            if(can(word, alpha)) {
                res += word.length();
            }
        }
        return res;
        
    }
    private boolean can(String word, int[] alpha) {
        int[] wordsC = new int[26];
        for(char ch : word.toCharArray()) {
            wordsC[ch - 'a']++;
        }
        for(int i = 0; i < 26; i++) {
            if(wordsC[i] > alpha[i]) {
                return false;
            }
        }
        return true;
    }
}
