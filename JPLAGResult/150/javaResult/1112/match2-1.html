class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] count = new int[26];
        for (int i = 0; i < chars.length(); ++i) {
            int idx = chars.charAt(i) - 'a';
            ++count[idx];
        
}
        
        int ans = 0;
        for (String word : words) {
            int[] curr = new int[26];
            for (int i = 0; i < word.length(); ++i) {
                int idx = word.charAt(i) - 'a';
                ++curr[idx];
            
}
            boolean flag = true;
            for (int i = 0; i < 26; ++i) {
                if (curr[i] > count[i]) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                ans += word.length();
            }
        }
        return ans;
    }
}
