class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] chararr = new int[26];
        for (int i = 0; i < chars.length(); i++) {
            chararr[chars.charAt(i) - 'a']++;
        
}
        
        int res = 0; 
        for (String word : words) {
            if (isFormable(word, chararr)) {
                res += word.length();
            }
        }
        return res; 
    }
    
    private boolean isFormable(String word, int[] chararr) {
        int[] letters = new int[26];
        for (int i = 0; i < word.length(); i++) {
            letters[word.charAt(i) - 'a']++;
        
}
        for (int i = 0; i < 26; i++) {
            if (letters[i] > chararr[i]) {
                return false;
            }
        }
        return true;
    }
}
