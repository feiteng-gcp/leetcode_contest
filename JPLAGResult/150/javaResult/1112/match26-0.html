class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] cnt = count(chars);
        int ans = 0;
        for (String word : words) {
            int[] temp = count(word);
            if (ok(cnt, temp)) {
                ans += word.length();
            }
        }
        return ans;
    }

    private boolean ok(int[] cnt, int[] temp) {
        for (int i = 0; i < 26; i++) {
            if (cnt[i] < temp[i]) {
                return false;
            }
        }
        return true;
    }

    private int[] count(String chars) {
        int[] cnt = new int[26];
        for (char c : chars.toCharArray()) {
            cnt[c - 'a'] ++;
        }
        return cnt;
    }
}
