class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] cnt = new int[26];
        for(char x : chars.toCharArray()) ++cnt[x - 'a'];
        int res = 0;
        for(String x : words) {
            int[] nc = new int[26];
            
boolean f = true;
            for(char y : x.toCharArray()) {
                ++nc[y - 'a'];
                if(nc[y - 'a'] > cnt[y - 'a']) {
                    f = false;
                    break;
                }
            }
            if(f) res += x.length();
        }
        return res;
    }
}
