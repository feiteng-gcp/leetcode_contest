class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] pattern = getCount(chars);
        int len = 0;
        for (String word : words) {
            int[] need = getCount(word);
            if(canForm(pattern, need))
                len += word.length()
;
        }
        return len;
    }
    
    public boolean canForm(int[] pattern, int[] word) {
        for (int i = 0; i < 26; i++) {
            if(word[i] > pattern[i]) return false
;
        }
        return true;
    }
    
    public int[] getCount(String word) {
        int[] pattern = new int[26];
        for (char c : word.toCharArray()) {
            pattern[(int)(c-'a')]++;
        }
        return pattern;
    }
}
