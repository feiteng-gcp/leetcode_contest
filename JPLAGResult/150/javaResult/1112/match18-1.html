class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] chs = new int[26];
        for (char ch: chars.toCharArray()) {
            chs[ch - 'a']++;
        
}
        int total = 0;
        for (String word : words) {
            int[] temp = Arrays.copyOf(chs, chs.length);
            boolean f = true;
            for (char ch: word.toCharArray()) {
                if (temp[ch - 'a'] > 0) {
                    temp[ch - 'a']--;
                } else 
{
                    f = false;
                    break;
                }
            }
            if (f) total += word.length();
        }
        return total;
    }
}
