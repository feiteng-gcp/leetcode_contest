class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] count = new int[26];
        for(char c : chars.toCharArray()){
            count[c - 'a']++;
        }
        
        int res = 0;
        for(String word : words){
            if(isGood(count, word)){
                res += word.length();
            }
        }
        
        return res;
    }
    
    private boolean isGood(int[] count, String word){
        int[] num = new int[26];
        for(char c : word.toCharArray()){
            num[c - 'a']++;
        }
        
        for(int i = 0; i < 26; i++){
            if(num[i] > count[i]){
                return false;
            }
        }
        
        return true;
    }
}
