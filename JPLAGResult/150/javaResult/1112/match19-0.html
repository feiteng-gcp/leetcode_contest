class Solution {
    public int countCharacters(String[] words, String chars) {
        int count = 0;
        int[] counts = new int[26];
        for (char c : chars.toCharArray()) {
            counts[c - 'a']++;
        }
        for (String word : words) {
            if (validate(counts, word)) {
                count += word.length();
            }
        }
        return count;
    }
    
    private boolean validate (int[] counts, String word) {
        int[] curr = new int[26];
        for (char c : word.toCharArray()) {
            int index = (int)(c - 'a');
            curr[index]++;
            if (curr[index] > counts[index]) {
                return false;
            }
        }
        return true;
    }
}
