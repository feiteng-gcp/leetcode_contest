class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] count = new int[26];
        for (char c : chars.toCharArray()) {
            count[c - 'a']++;
        }
        
        int res = 0;
        
        
        for (String word : words) {
            int[] wc = new int[26];
            for (char c : word.toCharArray()) {
                wc[c - 'a']++;
            }
            
            boolean canForm = true;
            for (int i = 0; i < 26; i++) {
                if (wc[i] > count[i]) {
                    canForm = false;
                    break;
                }
            }
            
            if (canForm) {
                res += word.length();
            }
        }
        
        return res;
        
    }
}
