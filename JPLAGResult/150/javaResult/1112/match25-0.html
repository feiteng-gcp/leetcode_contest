class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] count = new int[26];
        
        for(char c : chars.toCharArray()){
            count[c - 'a']++;
        }
        
        int re = 0;
        
        for(String word : words){
            int[] curr = new int[26];
            
            for(int i = 0; i < 26; i++){
                curr[i] = count[i];
            }
            for(char c : word.toCharArray()){
                curr[c - 'a']--;
            }
            boolean f = true;
            for(int i = 0; i < 26; i++){
                if(curr[i] < 0)
                    f = false;
            }
            if(f)
                re += word.length();
        }
        
        return re;
    }
}
