class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] arr = new int[26];
        for (int i = 0; i < chars.length(); i++) {
            arr[chars.charAt(i) - 'a']++;
        
}
        
        int sum = 0;
        for (String word : words) {
            int[] tempArray = new int[26];
            boolean flag = true;
            for (int i = 0; i < word.length(); i++) {
                tempArray[word.charAt(i) - 'a']++;
                if (tempArray[word.charAt(i) - 'a'] > arr[word.charAt(i) - 'a']) 
{
                    flag = false;
                    break;
                }
            }
            if (flag)
                sum += word.length();
        }
        return sum;
    }
}
