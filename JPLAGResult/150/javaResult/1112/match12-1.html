class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] total = new int[26];
        for(char c : chars.toCharArray()) {
            total[c - 'a']++;
        }
        int result = 0;
        for(String word: words) {
            int[] tmp = new int[26];
            for(char c : word.toCharArray()) {
                tmp[c - 'a']++;
            }
            boolean success = true;
            for(int i = 0 ; i < 26;i++) {
                if(tmp[i] > total[i]) {
                    success = false;
                    break;
                
}
            }
            if(success) result += word.length();
        }
        return result;
    }
}
