class Solution {
    public int countCharacters(String[] words, String chars) {
        int[] map = new int[26];
        for(char c : chars.toCharArray())
            map[c-'a']++;
        int result = 0;
        for(String word : words){
            int[] copy = Arrays.copyOf(map, map.length);
            boolean valid = true;
            for(char c : word.toCharArray()){
                if(copy[c-'a'] <= 0){
                    valid = false;
                    break;
                }
                copy[c-'a']--;
            }
            if(valid)
                result += word.length();
        }
        return result;
    }
}
