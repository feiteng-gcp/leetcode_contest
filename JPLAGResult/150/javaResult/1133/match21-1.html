class Solution {
    public String lastSubstring(String s) {
        char[] arr = s.toCharArray();
        int[] cnts = new int[26];
        for (char c : arr) {
            cnts[c - 'a']++;
        }
        int t = 0;
        for (int cnt : cnts) {
            if (cnt > 0)    t++;
        }
        if (t == 1) return s;
        char c = arr[0];
        List<Integer> idxs = new ArrayList<>();
        
for (int i = 0; i < s.length(); i++) {
            char d = s.charAt(i);
            if (d > c) {
                idxs.clear();
                idxs.add(i);
                c = d;
            } else if (d == c) {
                idxs.add(i);
            }
        }
        int l = 1, n = s.length();
        while (idxs.size() > 1) {
            List<Integer> tmp = new ArrayList<>();
            char maxC = arr[idxs.get(0) + l];
            for (int idx : idxs) {
                if (idx + l < n)    maxC = (char)Math.max(maxC, arr[idx + l]);
            }
            for (int idx : idxs) {
                if (idx + l < n && arr[idx + l] == maxC)   tmp.add(idx)
;
            }
            idxs = tmp;
            l++;
        }
        return s.substring(idxs.get(0));
    }
}
