class Solution {
    public String lastSubstring(String s) {
        if(s.length() == 0){
            return s;
        }
        int[] counts = new int[26];
        char[] ch = s.toCharArray();
        int count = 0;
        for(int i = 0; i < ch.length; i++){
            counts[ch[i] - 'a']++;
            if(counts[ch[i] - 'a'] == 1){
                count++;
            }
        }
        
        if(count == 1){
            return s;
        }
        int index = 0;
        for(int i = 0; i < 26; i++){
            if(counts[i] > 0){
                index = i;
            }
        }
        String str = "";
        for(int i = 0; i < ch.length; i++){
            if(ch[i]-'a' == index){
                String sub = s.substring(i);
                if(sub.compareTo(str) > 0){
                    str = sub;
                }
            }
        }
        return str;
        
    }
}
