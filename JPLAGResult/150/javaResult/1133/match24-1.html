class Solution {
    public String lastSubstring(String s) {
        Queue<int[]> q = new LinkedList<>();
        int largest = 'a'-1;
        for (int i = 0; i < s.length(); i++) {
            largest = Math.max(largest, s.charAt(i));
        }
        // System.out.println((char) largest);
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == largest) 
{
                q.add(new int[]{i, i});
            }
        }
        while (!q.isEmpty()) {
            int size = q.size();
            if (size == 1) {
                break;
            }
            int max = 'a'-1;
            for (int i = 0; i < size; i++) {
                int[] cur = q.poll();
                if (cur[1] + 1 < s.length()) {
                    max = Math.max(max, s.charAt(cur[1] + 1));
                }
                q.add(cur);
            }
            int end = -1;
            for (int i = 0; i < size; i++) {
                int[] cur = q.poll();
                // System.out.println(cur[0] + "," + cur[1]);
                if (cur[1] + 1 < s.length() && s.charAt(cur[1] + 1) == max && cur[0] > end) {
                    cur[1] += 1;
                    end = Math.max(end, cur[1]);
                    q.add(cur);
                }
            }
        }
        int[] pos = q.poll();
        StringBuilder temp = new StringBuilder();
        for (int i = pos[0]; i < s.length(); i++) {
            temp.append(s.charAt(i));
        }
        return temp.toString();
    }
}
