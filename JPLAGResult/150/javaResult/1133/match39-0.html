class Solution {
    public String lastSubstring(String s) {
        int maxchar = -1;
        for(char c: s.toCharArray()){
            int index = (int)(c - 'a');
            if(index > maxchar){
                maxchar = index;
            }
        }
        List<int[]> indexes = new ArrayList<>();
        Map<Integer, Integer> counts = new HashMap<>();
        for(int i = 0; i < s.length(); i++){
            int index = (int)(s.charAt(i) - 'a');
            if(index == maxchar){
                indexes.add(new int[]{maxchar, i, i, 1});
                counts.put(i, 1);
            }
        }
        Set<Integer> visited = new HashSet<>();
        while(true){
            int size = indexes.size();
            List<int[]> next = new ArrayList<>();
            if(size == 1){
                int [] cur = indexes.get(0);
                return s.substring(cur[1]);
            }
            for(int i = 0; i < size; i++){
                int [] cur = indexes.get(i);
                if(cur[2] == s.length() - 1){
                    continue;
                }else{
                    int nextindex = cur[2] + 1;
                    nextindex = (int)(s.charAt(nextindex) - 'a');
                    cur[0] = nextindex;
                    cur[2]++;
                    cur[3]++;
                    counts.put(cur[2], counts.getOrDefault(cur[2], 0) + 1);
                }
            }
            Collections.sort(indexes, (a, b) -> b[3] == a[3] ? b[0] - a[0] : b[3] - a[3]);
            int curmax = indexes.get(0)[0];
            int curlen = indexes.get(0)[3];
            for(int i = 0; i < size; i++){
                int [] cur = indexes.get(i);
                if(cur[0] != curmax || cur[3] != curlen || counts.get(cur[1]) > 1){
                    continue;
                }else{
                    next.add(cur);
                }
            }
            indexes = new ArrayList<>();
            indexes.addAll(next);
        }
    }
}
