class Solution {
    public int getIndex(List<Integer> list, String s){
        int len = 1, n = s.length();
        Set<Integer> indexes = new HashSet<>();
        for(int i : list){
            indexes.add(i);
        }
        while(list.size() != 1){
            List<Integer> tmp = new ArrayList<>();
            char c1 = 'a';
            for(int i=0;i<list.size();i++){
                int x = list.get(i);
                if(x+len < n){
                    char c = s.charAt(x+len);
                    if(c == c1){
                        tmp.add(x);
                    }else if(c > c1){
                        tmp = new ArrayList<>();
                        tmp.add(x);
                        c1 = c;
                    }
                }
            }
            len ++;
            list = tmp;
        }
        
        return list.get(0);
    }
    public String lastSubstring(String s) {
        String ans = new String();
        int n = s.length();
        List<Integer> maxIndex = new ArrayList<>();
        char c = 'a';
        for(int i=0;i<n;i++){
            char c1 = s.charAt(i);
            if(c1 == c){
                maxIndex.add(i);
            }else if(c1-'a' > c - 'a'){
                c = c1;
                maxIndex = new ArrayList<>();
                maxIndex.add(i);
            }
        }
        List<Integer> list = new ArrayList<>();
        for(int i=0;i<maxIndex.size();i++){
            if(i == 0 || maxIndex.get(i) -1 != maxIndex.get(i-1)){
                list.add(maxIndex.get(i));
            }
        }
        int last = getIndex(list, s);
     
        return s.substring(last);
    }
}
