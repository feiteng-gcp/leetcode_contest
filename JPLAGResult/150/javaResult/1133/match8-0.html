class Solution {
    public String lastSubstring(String s) {
        int maxStart = s.length() - 1;
        int maxCh = s.charAt(maxStart) - 'a';
        for (int idx = s.length() - 2; idx >= 0; idx--) {
            int curCh = s.charAt(idx) - 'a';
            if (curCh == maxCh) {
                if (idx + 1 == maxStart) {
                    maxStart = idx;
                } else {
                    boolean flag = true;
                    for (int subIdx = 1; subIdx <= s.length() - 1 - maxStart; subIdx++) {
                        if (s.charAt(idx + subIdx) > s.charAt(maxStart + subIdx)) {
                            break;
                        } else if (s.charAt(idx + subIdx) < s.charAt(maxStart + subIdx)) {
                            flag = false;
                            break;
                        }
                    }
                    if (flag) {
                        maxStart = idx;
                    }
                }
            } else if (curCh > maxCh) {
                maxCh = curCh;
                maxStart = idx;
            }
        }
        return s.substring(maxStart);
    }
}
