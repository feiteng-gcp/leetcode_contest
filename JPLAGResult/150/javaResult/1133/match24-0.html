class Solution {
    public String lastSubstring(String s) {
        LinkedList<int[]> candidates = new LinkedList<int[]>();
        char maxChar = 'a';
        for (int i=0; i<s.length(); i++) {
            char c = s.charAt(i);
            if (c>maxChar) {
                candidates.clear();
                maxChar = c;
            }
            if (c == maxChar) {
                candidates.add(new int[]{i, i});
            }
        }
        if (candidates.size() == s.length()) return s;
        int n = s.length();
        while (candidates.size() > 1) {
            LinkedList<int[]> next = new LinkedList<int[]>();
            maxChar = 'a';
            for(int[] candidate : candidates) {
                if (candidate[1]+1>=n) continue;
                char c = s.charAt(candidate[1]+1);
                if (c > maxChar) {
                    next.clear();
                    maxChar = c;
                }
                if (c == maxChar) {
                    next.add(new int[]{candidate[0], candidate[1]+1});
                }
            }
            candidates = next;
        }
        return s.substring(candidates.get(0)[0]);
    }
}
