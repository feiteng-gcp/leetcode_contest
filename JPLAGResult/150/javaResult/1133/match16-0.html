class Solution {
    public String lastSubstring(String s) {
        int maxc = s.charAt(0);
        boolean test = true;
        for(char c : s.toCharArray()){
            if(c != maxc)
                test = false;
            maxc = Math.max(maxc, c);
        }
        if(test)
            return s;
        String result = "";
        for(int i = 0; i < s.length(); i++){
            if(s.charAt(i) == maxc && s.substring(i).compareTo(result) > 0)
                result = s.substring(i);
        }
        return result;
    }
}
