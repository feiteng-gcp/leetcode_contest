class Solution {
    public String lastSubstring(String s) {
        List<Integer> indexs = new ArrayList<>();
        findLargestChar(s, indexs);
        if(indexs.isEmpty() || indexs.size() == s.length()) return s;
        
        char[] chars = s.toCharArray();
        int start = indexs.get(0);
        for(int i = 1; i < indexs.size(); i++) {
            int nextS = indexs.get(i);
            
if(nextIsGreater(start, nextS, chars)) {
                start = nextS;

            }
        }
        
        return s.substring(start);
    }
    
    private boolean nextIsGreater(int firstS, int nextS, char[] chars) {
        for(int i = nextS + 1; i < chars.length; i++) {
            int x = chars[i] - chars[i - nextS + firstS];
            if(x > 0) return true;
            if( x < 0) return false;
        }
        return false;
    }
    
    private void findLargestChar(String s, List<Integer> indexs) {
        char cand = 'a';
        for(int i = 0;i < s.length(); i++) {
            char ch = s.charAt(i);
            if(ch == cand) indexs.add(i);
            else if(ch < cand) continue;
            else {
                indexs.clear();
                indexs.add(i);
                cand = ch;
            }
        }
    }
}
