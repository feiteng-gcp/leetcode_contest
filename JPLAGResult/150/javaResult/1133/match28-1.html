class Solution {
    public String lastSubstring(String s) {
		List<Integer> set = new ArrayList<Integer>();
		char[] cs = s.
toCharArray();
		char max = 'a';
		for (int i = 0; i < cs.length; i++) {
			if (cs[i] > max) {
				max = cs[i];
				set.clear();
				set.add(i);
			} else {
				if (cs[i] == max) {
					set.add(i);
				}
			}
		}
		if (set.size() == 0) {
			return s.substring(set.iterator().next());
		}
		if(set.size()==cs.length){
			return s;
		}
		int key = 0;
		while (set.size() > 1) {
			key++;
			List<Integer> newSet = new ArrayList<Integer>();
			max = 'a';
			for (Integer i : set)
 {
				if (i + key >= cs.length) {
					continue;
				}
				if (cs[i + key] > max) {
					max = cs[i + key];
					newSet.clear();
					newSet.add(i);
				} else {
					if (cs[i + key] == max) {
						newSet.add(i);
					}
				}
			}
			set = newSet;
		}
		return s.substring(set.iterator().next());
	}
}
