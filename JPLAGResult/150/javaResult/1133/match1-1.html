class Solution {
    public String lastSubstring(String s) {
        int[] chars = new int[26];
        for (int i = 0; i < s.length(); i++) {
            chars[s.charAt(i) - 'a']++;
        }
        char big = 'a';
        for (int i = 25; i >= 0; i--) {
            if (chars[i] > 0) {
                big = (char)('a' + i);
                break;
            
}
        }
        
        // System.out.println(big);
        String mx = ""; 
        for (int i = 0; i < s.length(); ++i) { 
            if (s.charAt(i) != big || (i > 0 && s.charAt(i) == s.charAt(i-1))) {
                continue; 
            }
            String sub = 
s.substring(i);
            if (sub.compareTo(mx)>0) { 
                mx = sub; 
            } 
        } 
  
        return mx; 
    }
}
