class Solution {
   public String lastSubstring(String s) {
        List<Integer> state = new ArrayList<>();
        StringBuilder sb = new StringBuilder();

        Set<Integer> set = new HashSet<>();

        for(int i=0; i<s.length(); i++){
            state.add(i);
            set.add(s.charAt(i) - 'a');
        }
        int cent = 0;
        if(set.size() == 1){
            return s;
        }

        while(state.size() > 1){
            List<Integer> tmpState = new ArrayList<>();
            int tmpMax = Integer.MIN_VALUE;
            for(int i=0; i<state.size(); i++){
                int c = 0;
                if(state.get(i) + cent < s.length()) {
                    c = s.charAt(state.get(i) + cent) - 'a';
                }else{
                    c = -1;
                }

                if(c == tmpMax){
                    tmpState.add(state.get(i));
                }
                if(c > tmpMax){
                    tmpMax = c;
                    tmpState = new ArrayList<>();
                    tmpState.add(state.get(i));
                }
            }
            state = tmpState;
            cent++;
        }
        return s.substring(state.get(0), s.length());
    }
}
