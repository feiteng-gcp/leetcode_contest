import java.util.ArrayList;
import java.util.Iterator;
import java.util.List;

class Solution {
	public String lastSubstring(String s) {
		char[] chars = s.toCharArray();
		char bestChar = 'A';
		int count = 0;
		List<Integer> positions = new ArrayList<>();
		for (int i = 0; i < chars.length; i++) {
			if (chars[i] > bestChar) {
				count++;
				bestChar = chars[i];
				positions.clear();
				positions.add(i);
			} else if (chars[i] == bestChar) {
				positions.add(i);
			} else {
				count++;
			}
		}
		if (count == 1) return s;

		int advance = 0;
		while (positions.size() > 1) {
			advance++;
			bestChar = 'a';
			Iterator<Integer> iterator = positions.iterator();
			while (iterator.hasNext()) {
				int pos = iterator.next() + advance;
				if (pos >= chars.length) {
					iterator.remove();
				} else if (chars[pos] > bestChar) {
					bestChar = chars[pos];
				}
			}
			iterator = positions.iterator();
			while (iterator.hasNext()) {
				int pos = iterator.next() + advance;
				if (chars[pos] < bestChar) {
					iterator.remove();
				}
			}
		}
		return s.substring(positions.get(0));
	}

	public String badLastSubString(String s) {
		String best = "";
		String test = "";
		for (int i = s.length() - 1; i >= 0; i--) {
			test = s.charAt(i) + test;
			if (test.compareTo(best) > 0) {
				best = test;
			}
		}
		return best;
	}
}
