class Solution {
    // BF get all substring (n^2)
    // compare lexicographical order (len)
    
    // largest must start with the largest char
    public String lastSubstring(String s) {
        Set<Integer> candidates = new HashSet<>();
        Character max = null;
        for (int i = 0; i < s.length(); ++i){
            char ch = s.charAt(i);
            if (max == null || max < ch){
                candidates.clear();
                candidates.add(i);
                max = ch;
            
} else if (max == ch){
                candidates.add(i);
            }
        }
        int offset = 1;
        // special case
        if (candidates.size() == s.length()){
            return s;
        }
        while (candidates.size() > 1){
            candidates = nextRound(candidates, offset++, s);
        }
        return s.substring((int)candidates.toArray()[0]);
    }
    private Set<Integer> nextRound(Set<Integer> set, int offset, String s){
        Character max = null;
        Set<Integer> next = new HashSet<>();
        for (Integer i : set){
            if (i + offset >= s.length()) continue;
            char ch = s.charAt(i + offset);
            if (max == null || max < ch){
                next.clear();
                next.add(i);
                max = ch;
            } else if (ch == max){
                next.add(i);
            }
        }
        return next;
    }
}
