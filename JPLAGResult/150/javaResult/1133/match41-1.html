class Solution {
    
    private Node root;
    private StringBuilder result;
    
    
public String lastSubstring(String s) {
        if (s == null || s.length() == 0) {
            return "";
        }
        
        boolean noDiff = true;
        char target = 'a';
        for (int i = 0; i < s.length(); i++) {
            if (i > 0 && s.charAt(i) != s.charAt(i - 1)) {
                noDiff = false;
            }
            
            char c = s.charAt(i);
            if (c - target > 0) {
                target = c;
            
}
        }
        
        if (noDiff) {
            return s;
        }
        
        root = new Node();
        result = new StringBuilder();
        
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == target) {
                Node current = root;
                searchHelper(i, s, current);
            }
        }
        
        Node resultCurrent = root;
        combineHelper(resultCurrent);
        return result.toString();
    }
    
    private void combineHelper(Node current) {
        if (current == null) {
            return;
        }
        
        if (!current.e) {
            result.append(current.max);
            combineHelper(current.nexts[current.max - 'a']);
        }
    }
    
    private void searchHelper(int index, String s, Node current) {
        if (index >= s.length()) {
            return;
        }
        
        char c = s.charAt(index);
        if (current.e || current.max - c <= 0) {
            current.e = false;
            current.max = c;
            if (current.nexts[c - 'a'] == null) {
                current.nexts[c - 'a'] = new Node();
            }
            searchHelper(index + 1, s, current.nexts[c - 'a']);
        }
    }
    
    class Node {
        public boolean e;
        public char max;
        public Node[] nexts;
        
        public Node() {
            e = true;
            max = 'a';
            nexts = new Node[26];
        }
    }
}
