class Solution {
    public String lastSubstring(String s) {
        if(s == null || s.length() < 2) return s;
        int numOfChars = 0;
        int[] ab = new int[26];
        for(char ch : s.toCharArray()) {
            if(ab[ch - 'a']++ == 0) {
                numOfChars++;
            }
        }
        if(numOfChars == 1) return s;
        Stack<Integer> stack = new Stack<>();
        for(int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            while(!stack.isEmpty() && ch > s.charAt(stack.peek())) {
                stack.pop();
            }
            if(stack.isEmpty() || s.charAt(stack.peek()) == ch) {
                stack.push(i);
            }
        }
        int offset;
        for(offset = 1; stack.size() > 1; offset++) {
            // System.out.println(offset);
            Stack<Integer> newStack = new Stack<>();
            while(!stack.isEmpty()) {
                int popped = stack.pop();
                if(popped + offset >= s.length()) continue;
                while(!newStack.isEmpty() && s.charAt(popped + offset) > s.charAt(newStack.peek() + offset)) {
                    newStack.pop();
                }
                if(newStack.isEmpty() || s.charAt(newStack.peek() + offset) == s.charAt(popped + offset)) {
                    newStack.push(popped);
                }
            }
            stack = newStack;
        }
        int index = stack.pop();
        return s.substring(index);
    }
}
