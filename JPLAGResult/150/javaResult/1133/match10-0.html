class Solution {
    public String lastSubstring(String s) {
        Queue<Integer> q = new LinkedList<>();
        Queue<Integer>[] l = new Queue[26];
        for (int i = 0; i < 26; i++) {
            l[i] = new LinkedList<Integer>();
        }
        char[] ca = s.toCharArray();
        int N = ca.length;
        for (int i = 0; i < N; i++) {
            l[ca[i] - 'a'].offer(i);
        }
        for (int i = 25; i >= 0; i--) {
            if (!l[i].isEmpty()) {
                q = l[i];
                break;
            }
        }
        if (q.size() == N) {
            return s;
        }
        int dist = 1;
        while (q.size() > 1) {
            Queue<Integer> qNext = new LinkedList<>();
            char maxChar = (char)('a' - 1);
            for (int idx : q) {
                if (idx + dist < N ) {
                    if (ca[idx + dist] > maxChar) {
                        maxChar = ca[idx + dist];
                        qNext = new LinkedList<>();
                        qNext.offer(idx);
                    } else if (ca[idx + dist] == maxChar) {
                        qNext.offer(idx);
                    }
                }
            }
            q = qNext;
            dist++;
        }
        int start = q.poll();
        return s.substring(start, N);
    }
}
