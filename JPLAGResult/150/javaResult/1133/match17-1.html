class Solution {
    public String lastSubstring(String s) {
        int[] freq = new int[26];
        for (Character c : s.toCharArray())
            freq[(int)(c-'a')]++;
        
        int target = 0;   
        int cnt = 0;
        for (int i = 0; i < 26; i++) 
            if (freq[i] > 0) { 
                target = i;
                cnt++;
            }
        
        if (cnt == 1) return s;
        
        
        List<Integer> candidate = new ArrayList<>();
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == (char)('a'+target)) {
                candidate.add(i);
            }
        }
        
        int len = 1;
        while (candidate.size() > 1) {
            // System.out.println(candidate.size());
            Map<Character, List<Integer>> map = new HashMap<>();
            for (int i = 0; i < candidate.size(); i++) {
                int index = candidate.get(i);
                if (index+len >= s.length()) continue;
                char c = s.charAt(index+len);
                List<Integer> list;
                if (map.containsKey(c)) list = map.get(c);
                else list = new ArrayList<>();
                list.add(index);
                map.put(c, list);
            
}
            for (int i = 25; i >= 0; i--) {
                if (map.containsKey((char)('a'+i))) {
                    candidate = new ArrayList<>();
                    candidate.addAll(map.get((char)('a'+i)));
                    
break;
                }
            }
            len++;
        }
        
        return s.substring(candidate.get(0));
    }
}
