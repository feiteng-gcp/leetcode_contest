class Solution {
    public String lastSubstring(String s) {
        char[] cArr = s.toCharArray();
        char maxChar = maxChar(cArr);
        
        int resFromIndex = 0;
        
        for (int i = 0; i < cArr.length; i++) {
            if (cArr[i] != maxChar) {
                continue;
            }
            
            if (isLarger(cArr, i, resFromIndex)) {
                resFromIndex = i;
            }
        }
        
        return s.substring(resFromIndex);
    }
    
    public boolean isLarger(char[] cArr, int cur, int resFromIndex) {
        for (int i = cur; i < cArr.length; i++) {
            if (cArr[i] == cArr[resFromIndex + i - cur]) {
                continue;
            } else if (cArr[i] < cArr[resFromIndex + i - cur]) {
                return false;
            } else {
                return true;
            }
        }
        return false;
    }
    
    public char maxChar(char[] cArr) {
        char maxChar = 'a';
        for (char c : cArr) {
            maxChar = (char)Math.max(maxChar, c);
        }
        return maxChar;
    }
}
