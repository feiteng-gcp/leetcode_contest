class Solution {
    public String lastSubstring(String s) {
        char a = s.charAt(0);
        char lessa = s.charAt(0);
        for(int i=0;i<s.length();i++){
            if(s.charAt(i)>a)
                a = s.charAt(i);
            if(s.charAt(i)<lessa)
                lessa = s.charAt(i);
        
}
        if(a==lessa)
            return s;
        String res = "a";
        for(int i=0;i<s.length();i++){
            if(s.charAt(i)==a){
                String sub = s.substring(i);
                if(sub.compareTo(res)>0)
                    res = sub;
            }
        }
        return res;
    }
}
