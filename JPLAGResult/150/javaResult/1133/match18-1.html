class Solution {
    
public class Entry {
        public int start;
        public int idx;
        public Entry(int start, int idx) {this.start = start; this.idx = idx;}
    }
    public String lastSubstring(String s) {
        Queue<Entry> que = new LinkedList<>();
        
Queue<Entry> next = new LinkedList<>();
        for (int i = 0; i < s.length(); i++) {
            if (i == 0 || s.charAt(i) != s.charAt(i - 1)) que.add(new Entry(i, i));
        }
        char maxChar = 'a';
        while (true) {
            if (que.size() == 0) {
                if (next.size() == 1) return s.substring(next.peek().start);
                if (next.size() < 1) return s;
                que = next;
                next = new LinkedList<>();
                maxChar = 'a';
            }
            Entry cur = que.poll();
            if (cur.idx >= s.length()) continue;
            else if (s.charAt(cur.idx) == maxChar) {
                cur.idx++;
                next.add(cur);
            } else if (s.charAt(cur.idx) > maxChar) {
                maxChar = s.charAt(cur.idx);
                next = new LinkedList<>();
                cur.idx++;
                next.add(cur);
            }
        }
    }
}
