class Solution {
    public String lastSubstring(String s) {
        char bigestChar = 'a';
        char[] chars = s.toCharArray();
        int length = chars.length;
        for (char aChar : chars) {
            if(aChar > bigestChar ){
                bigestChar = aChar;
            }
        }
        int startIndex = -1;
        int endIndex = length;
        for (int i = 0; i < chars.length; i++) {
            if (chars[i] != bigestChar) {
                continue;
            }
            if (startIndex == -1) {
                startIndex = i;
                continue;
            }
            int compareIndex = startIndex;
            int nowIndex = i;
//            System.out.println(String.format("old:%s  new:%s  c:%s",startIndex,nowIndex,chars[i]));
            while(nowIndex < length){
                if(chars[compareIndex] == chars[nowIndex]){
                    compareIndex++;
                    nowIndex++;
                    continue;
                }
                if(chars[compareIndex] < chars[nowIndex]){
                    startIndex = i;
                }
                break;
            }
            if(nowIndex == length){
                return s.substring(startIndex,endIndex);
            }
        }
        return s.substring(startIndex,endIndex);
    }

    public static void main(String[] args) {

        System.out.println(new Solution().lastSubstring("abab"));
        System.out.println(new Solution().lastSubstring("leetcode"));
        System.out.println(new Solution().lastSubstring("xbylisvborylklftlkcioajuxwdhahdgezvyjbgaznzayfwsaumeccpfwamfzmkinezzwobllyxktqeibfoupcpptncggrdqbkji"));



    }
}
