class Solution {
    public String lastSubstring(String s) {
        String max = "";
        for (int i = 0; i < s.length(); ++i) {
            if (i > 0 && s.charAt(i - 1) >= s.charAt(i)) {
                continue;
            }    
            if (max.compareTo(s.substring(i)) < 0) {
                max = s.substring(i);
            }
        }
        return max;
    }
}
