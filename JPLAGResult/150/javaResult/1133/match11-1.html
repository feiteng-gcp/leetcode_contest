class Solution {
   public String lastSubstring(String s) {
        char[] arr = s.toCharArray();
        Set<Integer> posList = new TreeSet<>();
        for (int i = 0; i < s.length(); i++) {
            posList.add(i);
        }
        int len = 1;
        int prevSize = posList.size();
        while (posList.size() != 1) {
            solve(arr, posList, len++);
            int size = posList.size();
            if (size == prevSize - 1) {
                break;
            }
        }
        int pos = posList.iterator().next();
        return s.substring(pos);
    }

    private void solve(char[] s, Set<Integer> posList, int size) {
        char max = 'a';
        
Iterator<Integer> it = posList.iterator();
        while (it.hasNext()) {
            int pos = it.next();
            if (pos + size <= s.length) {
                max = s[pos + size - 1] > max ? s[pos + size - 1] : max;
            } else {
                it.remove();
            
}
        }
        it = posList.iterator();
        while (it.hasNext()) {
            int pos = it.next();
            if ( s[pos + size - 1] != max) {
                it.remove();
            }
        }
    }
}
