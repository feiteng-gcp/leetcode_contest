class Solution {
    public String lastSubstring(String s) {
        int max = (int) 'a';
        for (char ch : s.toCharArray()) {
            max = Math.max (max, (int) ch);
        }
        char maxChar = (char) max;
        int fromIndex = 0;
        int firstIndex = s.indexOf(maxChar,fromIndex);
        String last = "";
        while (firstIndex >= 0) {
            String str = s.substring(firstIndex);
            if (str.compareTo(last) > 0) last = str;
            fromIndex = firstIndex+1;
            firstIndex = s.indexOf(maxChar,fromIndex);
            while (firstIndex == fromIndex) {
                fromIndex = firstIndex+1;
                firstIndex = s.indexOf(maxChar,fromIndex);                
            }
        }
        return last;
    }
}
