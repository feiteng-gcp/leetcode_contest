class Solution {
    public String lastSubstring(String s) {
        if (s == null || s.length() == 0) {
            return s;
        }
        List<List<Integer>> indexList = new ArrayList<>();
        for (int i = 0; i < 26; i++) {
            indexList.add(new ArrayList<>());
        }
        for (int i = 0; i < s.length(); i++) {
            indexList.get(s.charAt(i) - 'a').add(i);
        }
        for (int i = 25; i >= 0; i--) {
            if (indexList.get(i).size() > 0) {
                int length = 1;
                List<Integer> list = indexList.get(i);
                if (list.size() == s.length()) {
                    return s;
                }
                while (list.size() > 1) {
                    List<Integer> nextList = new ArrayList<>();
                    char maxChar = 'a';
                    for (int index : list) {
                        if (index + length < s.length() && s.charAt(index + length) > maxChar) {
                            maxChar = s.charAt(index + length);
                        }
                    }
                    for (int index : list) {
                        if (index + length < s.length() && s.charAt(index + length) == maxChar) {
                            nextList.add(index);
                        }
                    }
                    length++;
                    list = nextList;
                }
                return s.substring(list.get(0));
            }
        }
        return "";
    }
}
