class Solution {
    public String lastSubstring(String s) {
        char maxChar = s.charAt(0);
        
        for(int i = 0; i < s.length(); i++){
            char c = s.charAt(i);
            if(c > maxChar){
                maxChar = c;
            }
        }   
        
        List<Integer> indicies = new ArrayList<>();
        
        for(int i = 0; i < s.length(); i++){
            if(s.charAt(i) == maxChar){
                indicies.add(i);
            }
        }
        
        if(indicies.size() == s.length())
            return s;
        
        int max = 0;
        for(int i = 1; i < indicies.size(); i++){
            if(compare(s, indicies.get(max), indicies.get(i)) < 0){
                max = i; 
            }
        }
        
        
        return s.substring(indicies.get(max));
    }
    
    public int compare(String s, int i, int j){
        int currA = i;
        int currB = j;
        
        while(currA < s.length() && currB < s.length()){
            if(s.charAt(currA) < s.charAt(currB)){
                return -1;
            } else if(s.charAt(currA) > s.charAt(currB)){
                return 1;
            }
            currA++;
            currB++;
        }
        
        if(currA < currB) return 1;
        else return -1;
    }
}
