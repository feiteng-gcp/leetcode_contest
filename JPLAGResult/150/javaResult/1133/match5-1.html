class Solution {
    public String lastSubstring(String s) {
        List<Integer> maxList = new ArrayList<>();
        char maxLetter = (char)('a' - 1);
        
        for(int i = 0; i < s.length(); ++i) {
            char ch = s.charAt(i);
            if(ch > maxLetter) {
                maxLetter = ch;
                maxList.clear();
                maxList.add(i);
            }
            else if(ch == maxLetter) {
                maxList.add(i);
            }
        }
        
        String str = "";
        for(int i = 0; i < maxList.size(); ++i) {
            if(maxList.size() - maxList.get(i) == s.length() - i) {
                return s.substring(maxList.get(i));
            }

            String sub = s.substring(maxList.get(i));
            if(sub.compareTo(str) > 0) {
                str = sub;
            }
        }

        return str;
    }
}
