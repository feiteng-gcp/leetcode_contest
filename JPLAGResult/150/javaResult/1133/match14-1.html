class Solution {
    
public String lastSubstring(String s) {
        int max2 = 0;
        
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) - 'a' > max2) {
                
max2 = s.charAt(i) - 'a';
            }
        }
        List<Integer> list = new ArrayList<>();
        
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == max2 + 'a') {
                list.add(i);
            }
        }
        
        if (list.size() == 1) {
            return s.substring(list.get(0));
        }
        
        int left = 0;
        List<Interval> intervals = new ArrayList<>();
        
        for (int i = 1; i < list.size(); i++) {
            if (list.get(i) - list.get(i - 1) == 1) {
                continue;
            }
            intervals.add(new Interval(list.get(left), list.get(i - 1)));
            left = i;
        }
        intervals.add(new Interval(list.get(left), list.get(list.size() - 1)));
        
        int maxLength = 0;
        
        for (int i = 0; i < intervals.size(); i++) {
            Interval temp = intervals.get(i);
            if (temp.end - temp.start > maxLength) {
                maxLength = temp.end - temp.start;
            }
        }
        
        
List<Interval> semi = new ArrayList<>();
        
        for (Interval temp : intervals) {
            if (temp.end - temp.start == maxLength) {
                semi.add(temp);
            }
        }
        
        if (semi.size() == 1) {
            return s.substring(semi.get(0).start);
        }
        //System.out.println(semi.size());
        while(semi.size() > 1) {
            int max = 0;
            List<Interval> temp = new ArrayList<>();
            for (int i = 0; i < semi.size(); i++) {
                Interval interval = semi.get(i);
                
                if (interval.end + 1 < s.length()) {
                    interval.end = interval.end + 1;
                    if (s.charAt(interval.end) - 'a' > max) {
                        max = s.charAt(interval.end) - 'a';
                    }
                }
            }
            
            for (Interval i : semi) {
                if (s.charAt(i.end) - 'a' == max) {
                    temp.add(i);
                }
            }
            semi = temp;
        }
        
        //System.out.println(semi.size());
        return s.substring(semi.get(0).start);
    }
}

public class Interval{
    int start;
    int end;
    
    public Interval(int start, int end) {
        this.start = start;
        this.end = end;
    }
}
