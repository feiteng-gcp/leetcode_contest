class Solution {
    public String lastSubstring(String s) {
        
ArrayList[] arr = new ArrayList[26];
        for (int i = 0; i < 26; ++i) {
            arr[i] = new ArrayList<Integer>();
        }
        for (int i = 0; i < s.length(); ++i) {
            int idx = s.charAt(i) - 'a';
            arr[idx].add(i);
        }
        int id = 0;
        for (int i = 25; i >= 0; --i) {
            if (arr[i].size() == 0) continue;
            if (arr[i].size() == 1) {
                // golden, just one for this index
                List<Integer> list = arr[i];
                int idx = list.get(0);
                return s.substring(idx);
            } else {
                id = i;
                break;
            }
        }
        
        List<Integer> bucket = arr[id];
        if (bucket.size() == s.length()) return s;
        
        class Node {
            int beginIdx;
            int currIdx;
            Node(int b, int c) {
                beginIdx = b;
                currIdx = c;
            }
        }
        
        List<Node> list = new ArrayList<>();
        
        for (int i : bucket) {
            list.add(new Node(i, i));
        }
        
        
while (list.size() != 1) {
            List<Node> next = new ArrayList<>();
            char maxCh = ' '; 
            for (Node node : list) {
                if (node.currIdx + 1 == s.length()) continue;
                ++node.currIdx;
                char ch = s.charAt(node.currIdx);
                if (maxCh == ' ') {                    
                    next.add(node);
                    maxCh = ch;
                } else {
                    if (ch < maxCh) {
                        continue; // discarded as small
                    } else if (ch > maxCh) {
                        next.clear(); // discard others as small
                        next.add(node);
                        maxCh = ch;
                    } else 
{
                        // keep equals // ch == maxCh
                        next.add(node); 
                    }
                }
            }
            list = next;
        }
        
        return s.substring(list.get(0).beginIdx);
    }
}
