class Solution {
    public String lastSubstring(String s) {
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < s.length(); i++) list.add(i);
        char[] arr = s.toCharArray();
        char c = arr[arr.length - 1];
        int k = arr.length - 1;
        for (; k > -1; k--) {
            if (arr[k] != c) break;
        }
        for (int i = 0; i <= k; i++) {
            char max = '0';
            for (int j : list) {
                
if (j + i < arr.length && arr[j + i] > max) {
                    max = arr[j + i];
                }
            }
            List<Integer> temp = new ArrayList<>();
            for (int j : list) {
                if (j + i < arr.length && arr[j + i] == max) {
                    temp.add(j);
                }
            }
            if (temp.size() == 1) return s.substring(temp.get(0));
            list = temp;
        }
        return s.substring(list.get(0));
    }
}
