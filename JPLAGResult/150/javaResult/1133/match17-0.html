class Solution {
    public String lastSubstring(String s) {
        char maxCh = 'a';
        List<Integer> L = new ArrayList<>();
        char last = '#';
        for (int i = 0; i < s.length(); i ++) {
            char ch = s.charAt(i);
            if(ch == maxCh && ch != last) L.add(i);
            else if(ch > maxCh) {
                maxCh = ch;
                L.clear();
                L.add(i);
            }
            last = ch;
        }
        int shift = 1;
        ArrayList<Integer>[] next = new ArrayList[26]; 
        while(L.size() > 1) {
            for (int i = 0; i < 26; i ++) next[i] = new ArrayList<Integer>();
            
            for (int k : L) {
                if(k+shift < s.length()) {
                    char c = s.charAt(k+shift);
                    next[(int)(c-'a')].add(k);
                }
            }
            
            for (int i = 25; i >= 0; i --) {
                if(next[i].size() > 0) {
                    L = next[i]; break;
                }
            }
            shift ++;
        }
        return s.substring(L.get(0));
    }
}
