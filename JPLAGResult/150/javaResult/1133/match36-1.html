class Solution {
    public String lastSubstring(String s) {
        char lastChar = 'a';
        for(char c: s.toCharArray()){
            if(c > lastChar) lastChar = c;
        }
        if(lastChar == 'a') return s;
        
        List<Integer> startPos = new LinkedList<>();
        for(int i = 0; i < s.length(); i++){
            if(s.charAt(i) == lastChar) startPos.add(i);
        }
        
        int diff = 1;
        while(startPos.size() > 1){
            List<Integer>[] nextCharMap = new List[26];
            for(int i = 0; i < 26; i++) nextCharMap[i] = new LinkedList<Integer>();
            int maxIndex = 0;
            for(int pos: startPos){
                if(pos + diff >= s.length()) continue;
                int index = s.charAt(pos + diff) - 'a';
                maxIndex = Math.max(maxIndex, index);
                nextCharMap[index].add(pos);
            }
            startPos = new LinkedList<>(nextCharMap[maxIndex]);
            diff++;
        }
        
        return s.substring(startPos.get(0));
    }
}
