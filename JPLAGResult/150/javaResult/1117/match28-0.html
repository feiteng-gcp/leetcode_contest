class Solution {
    public int maxDistance(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        boolean[][] visited = new boolean[m][n];
        int[][] len = new int[m][n];
        Queue<int[]> queue = new LinkedList<>();
        for (int i = 0; i < m; i++) {
            Arrays.fill(len[i], 9999);
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    queue.add(new int[] { i, j });
                    visited[i][j] = true;
                    len[i][j] = 0;
                }
            }
        }
        if (queue.isEmpty() || queue.size() == m * n) {
            return -1;
        }
        int[][] ds = new int[][] { { -1, 0 }, { 1, 0 }, { 0, 1 }, { 0, -1 } };
        int max = 0;
        while (!queue.isEmpty()) {
            int[] temp = queue.poll();
            for (int[] d : ds) {
                int x = temp[0] + d[0];
                int y = temp[1] + d[1];
                if (x < 0 || y < 0 || x >= m || y >= n || visited[x][y]) {
                    continue;
                }
                visited[x][y] = true;
                len[x][y] = len[temp[0]][temp[1]] + 1;
                max = Math.max(max, len[x][y]);
                queue.add(new int[] { x, y });
            }
        }
        return max;
    }
}
