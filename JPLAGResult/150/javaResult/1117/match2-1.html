class Solution {
    public int maxDistance(int[][] grid) {
        
int N = grid.length;
        int max_dis = -1;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                if (grid[i][j] == 0) {
                    int dis = bfs(grid, i, j);
                    if (dis == -1) return -1;
                    if (dis > max_dis)
                        max_dis = dis;
                }
            }
        }
        return max_dis;
    }
    
int[][] dirs = {{1,0},{-1,0},{0,1},{0,-1}};
    private int bfs(int[][] grid, int si, int sj) {
        int N = grid.length;
        boolean[][] visit = new boolean[N][N];
        int step = 0;
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[]{si, sj});
        
visit[si][sj] = true;
        
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int[] cur = queue.poll();
                if (grid[cur[0]][cur[1]] == 1) return step
;
                for(int[] dir : dirs) {
                    int nxt_x = cur[0] + dir[0];
                    int nxt_y = cur[1] + dir[1];
                    if (nxt_x < 0 || nxt_x == N || nxt_y < 0 || nxt_y == N || visit[nxt_x][nxt_y]) continue;
                    visit[nxt_x][nxt_y] = true;
                    queue.add(new int[]{nxt_x, nxt_y});
                
}
            }
            step++;
        }
        return -1;
    }
}
