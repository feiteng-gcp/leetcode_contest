class Solution {
    private static final int[][] DIR = {
        {0,1},
        {0,-1},
        {1,0},
        {-1,0}
    };
    public int maxDistance(int[][] a) {
        int n = a.length;
        int m = a[0].length;
        boolean[][] been = new boolean[n][m];        
        LinkedList<int[]> q = new LinkedList<>(); // 0: i, 1: j, 2: dis
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (a[i][j] == 1) {
                    been[i][j] = true;
                    q.add(new int[]{i, j, 0});
                }
            }
        }
        if (q.isEmpty() || q.size() == m * n) return -1;
        int res = -1;

        while (!q.isEmpty()) {
            int[] r = q.removeFirst();
            int i = r[0];
            int j = r[1];
            int dis = r[2];
            res = Math.max(res, dis);
            
            for (int[] d : DIR) {
                int ni = i + d[0];
                int nj = j + d[1];
                if (ni < 0 || ni >= n || nj < 0 || nj >= m || been[ni][nj]) continue;
                been[ni][nj] = true;
                q.add(new int[]{ni, nj, dis+1});
            }
        }
        
        return res;
    }
}
