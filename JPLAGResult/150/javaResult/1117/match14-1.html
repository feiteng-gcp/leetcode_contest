class Solution {
    public int maxDistance(int[][] g) {
        int l = g.length;
        int ll = g[0].length;
        
        boolean dp[][] = new boolean[l][ll]; 
        Queue<int[]> q =new ArrayDeque<>();
        int s = 0;
        
for(int i=0;i<l;++i){
            for(int j=0;j<l;++j){
                s += g[i][j];
                if(g[i][j]==1){
                    q.offer(new int[]{i,j});
                
}
            }
        }
        if(s==0||s==l*ll) return -1;
        
        int ds = 0;
        
        while(q.size()>0){
            int sz = q.size();
            
            while(sz-->0){
                int cur [] = q.poll();
                
                
int d[][] = new int[][]{{0,1},{0,-1},{1,0},{-1,0}};
                
                
for(int u[]:d){
                    int nx = cur[0]+u[0];
                    int ny = cur[1]+u[1];
                    
                    if(nx>=0&&ny>=0&&nx<l&&ny<ll&&g[nx][ny]==0&&!dp[nx][ny]){
                        dp[nx][ny] = true;
                        q.offer(new int[]{nx,ny});
                    }
                    
                    
                }
                
                
                
                
            }
            
            ds++;
            
            
            
        }
        return ds-1;
        
        
        
    }
}
