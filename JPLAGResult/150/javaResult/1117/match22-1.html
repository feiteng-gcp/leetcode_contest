class Solution {
    public int maxDistance(int[][] grid) {
        //find all the land
        int l = grid.length;
        boolean[][] visited = new boolean[l][l];
        Queue<Integer> q = new LinkedList<>();
        
        for (int i = 0; i < l; i++) {
            for (int j = 0; j < l; j++) {
                if (grid[i][j] == 1) {
                    int index = i * l + j;
                    q.offer(index);
                    
visited[i][j] = true;
                }
            }
        }
        
        if (q.isEmpty() || q.size() == l * l) {
            return -1;
        }
        
        int dis = 0;
        int[] di = {1, 0, -1, 0};
        int[] dj = {0, 1, 0, -1};
        
        
while (!q.isEmpty()) {
            int size = q.size();
            for (int k = 0; k < size; k++) {
                int index = q.poll();
                int i = index / l;
                int j = index % l;
                for (int d = 0; d < 4; d++) {
                    int ni = i + di[d];
                    int nj = j + dj[d];
                    if (ni >= 0 && ni < l && nj >= 0 && nj < l && !visited[ni][nj]) {
                        q.offer(ni * l + nj);
                        
visited[ni][nj] = true;
                    }
                }
            }
            dis++;
        }
        return dis - 1;
    }
}
