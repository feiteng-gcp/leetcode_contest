class Solution {
    public int maxDistance(int[][] grid) {
         
        int[][] res = new int[grid.length][grid[0].length];
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 0) {
                    res[i][j] = 1000;
                } else {
                    res[i][j] = 0;
                }
            }
        }
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 1) {
                    search(grid, i, j, res);
                }
            }
        }
        
int max = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (res[i][j] != 0 && res[i][j] != 1000) {
                    max = Math.max(res[i][j], max);
                
}
            }
        }
        return max == 0 ? -1 : max;
    }
    
    void search(
int[][] matrix, int x, int y, int[][] res) {
        Queue<Pair> queue = new LinkedList<>();
        queue.add(new Pair(x, y));
        int[][] offset = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                Pair tmp = queue.poll();
                        
                for (int k = 0; k < 4; k++) {
                    int dx = offset[k][0] + tmp.x;
                    int dy = offset[k][1] + tmp.y;
                    if (dx >= 0 && dy >= 0 && dx < matrix.length && dy < matrix[0].length && matrix[dx][dy] == 0) {                                       if (res[dx][dy] > res[tmp.x][tmp.y] + 1) {
                            res[dx][dy] = res[tmp.x][tmp.y] + 1;
                            queue.add(new Pair(dx, dy));
                        }
                    }
                }      
            }
        }
    }
    
    
class Pair {
        int x;
        int y;
        public Pair(int a, int b) {
            this.x = a;
            this.y = b;
        }
    }
}
