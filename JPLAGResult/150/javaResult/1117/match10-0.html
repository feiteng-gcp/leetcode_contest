class Solution {
    public int maxDistance(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        Queue<int[]> queue = new LinkedList<>();
        boolean[][] visited = new boolean[m][n];
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    queue.offer(new int[]{i, j});
                    visited[i][j] = true;
                }
            }
        }
        
        if (queue.size() == 0 || queue.size() == m * n) {
            return -1;
        }
        
        int[] dx = {1, 0, 0, -1};
        int[] dy = {0, 1, -1, 0};
        
        int dist = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int[] temp = queue.poll();
                int row = temp[0];
                int col = temp[1];
                
                for (int k = 0; k < 4; k++) {
                    int newRow = row + dx[k];
                    int newCol = col + dy[k];
                    
                    if (newRow >= 0 && newRow < m && newCol >= 0 && newCol < n && !visited[newRow][newCol]) {
                        queue.offer(new int[]{newRow, newCol});
                        visited[newRow][newCol] = true;
                    }
                }
            }
            dist++;
        }
        
        return dist - 1;
    }
}
