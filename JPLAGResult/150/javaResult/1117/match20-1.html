class Solution {
    int [][] dir = {
        {-1,0},
        {0,-1},
        {0,1},
        {1,0}
    };
    public int maxDistance(int[][] grid) {
        Queue<int[]> q = 
new LinkedList<>();
        for(int i =0;i<grid.length;i++){
            for(int j =0;j<grid.length;j++){
                if(grid[i][j] == 1){
                    
q.add(new int[]{i,j});
                }
            }
        }
        int count = q.size();
        int level = 2;
        while(!q.isEmpty()){
            int size = count;
            count = 0;
            for(int i =0;i<size;i++)
{
                int[] pos = q.poll();
                for(int j=0;j<4;j++){
                    int x= pos[0] + dir[j][0];
                    int y =pos[1] + dir[j][1];
                    if(x <0|| y<0){continue;}
                    if(x == grid.length||y == grid.length){continue;}
                    if(grid[x][y] != 0){continue;}
                    grid[x][y] = level;
                    q.add(new int[]{x,y});
                    count++;
                }
            
}
            level++;
        }
        int res = 1;
        for(int i =0;i<grid.length;i++){
            for(int j =0;j<grid.length;j++){
                if(grid[i][j] > res){
                    res = grid[i][j];
                }
            }
        }
        if(res == 1){
            return -1;
        }
        return res-1;
    }
   
}
