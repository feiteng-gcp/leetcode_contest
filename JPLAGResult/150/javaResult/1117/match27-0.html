class Solution {

   class Node {
       int i;
       int j;
       public Node(int i, int j) {
           this.i=i;
           this.j=j;
       }
   }
   
   int res=0, max=0;
   boolean flag=false, flag1=false;    
   public int maxDistance(int[][] grid) {
       int n = grid.length;
       for(int i=0; i<n; i++) {
           for(int j=0; j<n; j++) {
               if(grid[i][j]==0) {
                   boolean[][] visited = new boolean[n][n];                    
                   bfs(grid,i,j,n,visited);
                   max=Math.max(max, res);
                   flag=true;
               }
           }
       }
       return (!flag || !flag1) ? -1 : max;
   }
   
   private void bfs(int[][] grid, int i, int j, int n, boolean[][] visited) {
       Queue<Node> queue = new LinkedList<>();         
       queue.offer(new Node(i,j));
       int[][] dirs={{1,0},{0,1},{0,-1},{-1,0}};
       while(!queue.isEmpty()) {
           int size=queue.size();            
           for(int k=0; k<size; k++) {
               Node node = queue.poll();                
               for(int[] dir : dirs) {
                   int x = node.i+dir[0];
                   int y = node.j+dir[1];
                   if(x<0 || y<0 || x>=n || y>=n || visited[x][y])
                       continue;
                   visited[x][y]=true;
                   if(grid[x][y]==1) {
                       res=Math.abs(i-x)+Math.abs(j-y);
                       flag1=true;
                       return;
                   }
                   queue.offer(new Node(x,y));
               }
           }
       }                
   }
}
