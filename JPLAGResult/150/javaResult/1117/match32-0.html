class Solution {
    public int maxDistance(int[][] grid) {
        if(grid.length == 0 || grid[0].length == 0) return -1;
        int distance = -1;
        boolean exist = false;
        for(int i = 0; i < grid.length; i++){
            for(int j = 0; j < grid[0].length; j++){
                if(grid[i][j] == 1){
                    exist = true;
                    break;
                }
            }
        }
        
        if(!exist){
            return -1;
        }
        
        for(int i = 0; i < grid.length; i++){
            for(int j = 0; j < grid[0].length; j++){
                if(grid[i][j] == 0){
                    distance = Math.max(distance, bfs(grid, i, j));
                }
            }
        }
        return distance;
    }
    
    private int bfs(int[][] grid, int i, int j){
        int col = grid.length;
        int row = grid[0].length;
        int[][] add = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        boolean[][] seen = new boolean[col][row];
        int dis = -1;
        int cur_dis = 0;
        Queue<int[]> queue = new LinkedList<int[]>();
        queue.add(new int[]{i, j});
        seen[i][j] = true;
        while(!queue.isEmpty()){
            int size = queue.size();
            for(int m = 0; m < size; m++){
                int[] position = queue.poll();
                int curCol = position[0];
                int curRow = position[1];
                if(grid[curCol][curRow] == 1){
                    return cur_dis;
                }
                for(int n = 0; n < 4; n++){
                    int nextCol = curCol + add[n][0];
                    int nextRow = curRow + add[n][1];
                    if(nextCol < col && nextRow < row && nextCol >= 0 && nextRow >= 0 && !seen[nextCol][nextRow]){
                        seen[nextCol][nextRow] = true;
                        queue.add(new int[]{nextCol, nextRow});
                    }
                }
            }
            cur_dis++;
        }
        return -1;
    }
}
