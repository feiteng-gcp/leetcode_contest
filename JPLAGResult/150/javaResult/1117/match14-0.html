class Solution {
    public int maxDistance(int[][] grid) {
        int n = grid.length;

        Queue<int[]> q = new ArrayDeque<>();
        int landCount = 0;
        int waterCount = 0;

        int[][] dist = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    landCount++;
                    q.add(new int[] {i, j});
                    dist[i][j] = 0;
                } else {
                    waterCount++;
                    dist[i][j] = -1;
                }
            }
        }

        if (landCount == 0 || waterCount == 0) return -1;

        int[][] dir = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int ans = -1;
        while (!q.isEmpty()) {
            int[] t = q.poll();
            int x = t[0];
            int y = t[1];

            for (int i = 0; i < dir.length; i++) {
                int r = x + dir[i][0];
                int c = y + dir[i][1];
                if (r >= 0 && r < n && c >= 0 && c < n && dist[r][c] == -1) {
                    dist[r][c] = dist[x][y] + 1;
                    q.add(new int[] {r, c});
                    ans = Math.max(ans, dist[r][c]);
                }
            }
        }
        return ans;
    }

}
