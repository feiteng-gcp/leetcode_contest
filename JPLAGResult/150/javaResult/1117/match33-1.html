class Solution {
    public int maxDistance(int[][] grid) {
        List<int[]> list = new ArrayList();
        boolean hasIsland = false;
        boolean hasWater = false;
        int n = grid.length,m = grid[0].length;
        
int[][] ds = new int[n][m];
        for(int i=0;i<grid.length;i++){
            for(int j=0;j<grid[i].length;j++){
                if(grid[i][j]==0){
                    hasWater = true;
                    ds[i][j] = Integer.MAX_VALUE;
                }else{
                    hasIsland = true;
                    
list.add(new int[]{i, j});
                    ds[i][j] = -1;
                }
            }
        }
        int[][] dirs = {{0,1},{0,-1},{1,0},{-1,0}};
        if(!hasIsland || !hasWater) return -1;
        for(int[] island:list){
            Queue<int[]> q = new LinkedList();
            q.offer(island);
            int dis = 0;
            boolean[][] visited = new boolean[n][m];
            visited[island[0]][island[1]] = true;
            
while(!q.isEmpty()){
                dis ++;
                int size = q.size();
                for(int i=0;i<size;i++){
                    int[] cur = q.poll();
                    int x = cur[0], y = cur[1];
                    for(int[] dir:dirs){
                        int nr = x+dir[0], nc = y+dir[1];
                        if(nr<0||nc<0||nr>n-1||nc>m-1||visited[nr][nc]||ds[nr][nc]==-1) continue;
                        visited[nr][nc] = true;
                        if(ds[nr][nc]>dis)
{
                            q.offer(new int[]{nr, nc});
                            ds[nr][nc] = dis;
                        }
                    }
                }
            }
        }

        int res = Integer.MIN_VALUE;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(ds[i][j]==-1) continue;
                res = Math.max(res, ds[i][j]);
            }
        }
        return res;
    }
}
