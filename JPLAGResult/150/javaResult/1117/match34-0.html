class Solution {
    public int maxDistance(int[][] grid) {
        int[][] dirs = new int[][]{{1, 0}, {-1, 0}, {0, -1}, {0, 1}};
        int m = grid.length;
        int n = grid[0].length;
        Queue<int[]> q = new LinkedList<>();
        int count= 0;
        for (int i = 0; i < m; i++){
            for (int j = 0; j < n; j++){
                if (grid[i][j] == 1) {
                    count++;
                    q.offer(new int[]{i, j});
                }
            }
        }
        if (count == 0 || count == m*n) {
            return -1;
        }
        int step = 2;
        while (!q.isEmpty()) {
            int size = q.size();
            for (int i=0; i < size; i++) {
                int[] pos = q.poll();
                for (int[] dir: dirs){
                    int[] new_pos = new int[]{dir[0]+pos[0], dir[1] + pos[1]};
                    int x = new_pos[0], y = new_pos[1];
                    if (x >= 0 && x < m && y >=0 && y < n && grid[x][y] == 0) {
                        grid[x][y] = step;
                        q.offer(new int[]{x, y});
                    }
                }                
            }
            step++;
        }
        
        int max = 0;
        for (int i = 0; i < m; i++){
            for (int j = 0; j < n; j++){
                max = Math.max(max, grid[i][j]);
            }
        }
        return max-1;
        
    }
}
