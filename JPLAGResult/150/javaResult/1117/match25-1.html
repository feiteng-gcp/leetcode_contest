import java.util.ArrayList;
import java.util.List;

class Solution {

	int[][]
 directions = new int[][]{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

	public 
int maxDistance(int[][] grid) {
		boolean waterExists = false;
		boolean landExists = false;
		List<int[]> locations = new ArrayList<>();
		for (int i = 0; i < grid.length; i++) {
			for (int j = 0; j < grid[0].length; j++) {
				if (grid[i][j] == 0) {
					waterExists = true;
				} else {
					landExists = true;
					locations.add(new int[]{i, j});
				}

			}
		}
		if (!(waterExists && landExists)) {
			return -1;
		}
		// bfs on all land locations at the same time
		int distance = -1;
		while (!locations.isEmpty()) {
			distance++;
			List<int[]> newLocations = new ArrayList<>();
			for (int[] position : locations) {
				for (int[] direction : direc
tions) {
					int x = position[0] + direction[0];
					int y = position[1] + direction[1];
					if (0 <= x && x < grid.length && 0 <= y && y < grid.length) {
						if (grid[x][y] == 0) {
							int[] newPosition = new int[]{position[0] + direction[0], position[1] + direction[1]};
							newLocations.add(newPosition);
							grid[x][y] = 2;

						}
					}
				}
			}
			locations = newLocations;
		}
		return distance;
	}
}
