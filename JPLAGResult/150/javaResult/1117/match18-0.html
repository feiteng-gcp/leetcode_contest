class Solution {
    private final int[][] dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    public int maxDistance(int[][] grid) {
        int n = grid.length;
        
        int[][] dis = new int[n][n];
        boolean[][] visited = new boolean[n][n];
        
        Queue<int[]> queue = new LinkedList<>();
        for(int i = 0; i < n; ++i) {
            for(int j = 0; j < n; ++j) {
                if(grid[i][j] == 1) {
                    queue.add(new int[]{i, j});
                    visited[i][j] = true;
                    dis[i][j] = 0;
                }
                else {
                    dis[i][j] = 2 * n;
                }
            }
        }
        
        if(queue.isEmpty() || queue.size() == n * n) {
            return -1;
        }
        
        int max = 0;
        while(!queue.isEmpty()) {
            int[] pos = queue.poll();
            for(int[] dir : dirs) {
                int x = pos[0] + dir[0];
                int y = pos[1] + dir[1];
                
                if(x >= 0 && x < n && y >= 0 && y < n && !visited[x][y]) {
                    visited[x][y] = true;
                    dis[x][y] = Math.min(dis[x][y], dis[pos[0]][pos[1]] + 1);
                    max = Math.max(max, dis[x][y]);
                    
                    queue.add(new int[]{x, y});
                }
            }
        }
        
        return max;
    }
}
