class Solution {
    private static class Grid{
        int row;
        int col;
        Grid(int row, int col){
            this.row = row;
            this.col = col;
        }
        public boolean equals(Object o){
            if (o.getClass() != this.getClass()) return false;
            Grid g = (Grid)o;
            return g.row == this.row && g.col == this.col;
        }
        public int hashCode(){
            return 31 * row + 7 * col;
        }
    }
    
private final int[][] DIRS = new int[][]{{0,1},{1,0},{-1,0},{0,-1}};
    public int maxDistance(int[][] grid) {
        // KBFS
        // start from all land 1, find the fartherest water 0
        Queue<Grid> q = new ArrayDeque<>();
        
int N = grid.length;
        Set<Grid> visited = new HashSet<>();
        for (int i = 0; i < N; ++i){
            for (int j = 0; j < N; ++j){
                if (grid[i][j] == 1){
                    visited.add(new Grid(i, j));
                    
q.offer(new Grid(i, j));
                }
            }
        }
        if (q.size() == 0 || q.size() == N*N) return -1;
        
int distance = 0;
        while (!q.isEmpty()){
            int size = q.size();
            for (int i = 0; i < size; ++i){
                Grid g = q.poll();
                // generate its neighbors
                for (int[] dir : DIRS){
                    int row = g.row + dir[0];
                    int col = g.col + dir[1];
                    if (row >= 0 && row < N && col >= 0 && col < N && grid[row][col] == 0 && visited.add(new Grid(row, col))){
                        q.offer(new Grid(row, col));
                    
}
                }
                
            }
            distance++;
        }
        return distance - 1;
        
    }
}
