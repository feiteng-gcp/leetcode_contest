class Solution {
    private int[][] ds = {{0, 1},{0,-1},{1,0},{-1,0}};
    public int maxDistance(int[][] grid) {
        int r = grid.length, c = grid[0].length;
        Queue<int[]> q = 
new LinkedList<>();
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                if (grid[i][j] == 1) {
                    
q.offer(new int[]{i, j});
                }
            }
        }
        if (q.size() == 0 || q.size() == r * c) return -1;
        int res = 1;
        while (q.size() > 0) {
            int[] v = q.poll();
            int i = v[0], j = v[1];
            int k = grid[i][j];
            
for (int[] d : ds) {
                int x = i + d[0];
                int y = j + d[1];
                if (x < 0 || x == r || y < 0 || y == c || grid[x][y] > 0) continue;
                grid[x][y] = k + 1;
                if (res < k) res = k;
                
q.offer(new int[]{x, y});
            }
        }
        return res;
    }
}
