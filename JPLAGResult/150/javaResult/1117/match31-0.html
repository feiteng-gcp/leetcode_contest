class Solution {
    private int[][] directions = new int[][]{{0, 1},{0, -1},{1, 0}, {-1,0}};
    public int maxDistance(int[][] grid) {
    	int n = grid.length;
    	int m = grid[0].length;
    	int[][] values = new int[n][m];
    	for (int i = 0; i < n; i++) {
    		Arrays.fill(values[i], -1);
    	}
    	Queue<int[]> queue = new ArrayDeque<>();
    	for (int i = 0; i < n; i++) {
    		for (int j = 0; j < m; j++) {
    			if (grid[i][j] == 1) {
    				values[i][j] = 0;
    				queue.add(new int[]{i,j});
    			}
    		}
    	}
    	if (queue.size() == 0 || queue.size() == n * m) {
    		return -1;
    	}
    	while(!queue.isEmpty()) {
    		int[] curr = queue.poll();
    		for (int[] d : directions) {
    			int i = curr[0] + d[0];
    			int j = curr[1] + d[1];
    			if (i >= 0 && i < n && j >= 0 && j < m && values[i][j] == -1) {
    				values[i][j] = values[curr[0]][curr[1]] + 1;
    				queue.add(new int[]{i, j});
    			}
    		}
    	}
    	int max = 0;
    	for (int i = 0; i < n; i++) {
    		for (int j = 0; j < m; j++) {
    			if (values[i][j] > max) {
    				max = values[i][j];
    			}
    		}
    	}
    	return max;
    }
}
