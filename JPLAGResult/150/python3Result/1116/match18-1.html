# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        res = 0
        curmax = -float('inf')
        stk = [root]
        level = 1
        while stk:
            temp 
= []
            levelsum = 0
            for node in stk:
                levelsum += node.val
                if node.left:
                    temp.append(node.left)
                if node.right:
                    temp.append(node.right)
            if levelsum > curmax:
                curmax = levelsum
                res = level
            level += 1
            stk = temp
        return res
