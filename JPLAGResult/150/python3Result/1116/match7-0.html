# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        if not root:
            return 0
        count = 0
        maxval = float('-inf')
        res = 0
        stack = [root]
        while stack:
            count += 1
            temp = []
            tempval = 0
            for node in stack:
                tempval += node.val
                if node.left:
                    temp.append(node.left)
                if node.right:
                    temp.append(node.right)
            if tempval > maxval:
                res = count
                maxval = tempval
            stack = temp
        return res
