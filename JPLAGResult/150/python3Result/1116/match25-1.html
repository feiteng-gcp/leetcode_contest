# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        if not root:
            return 0
        current_level = 1
        ans, max_sum = 0, - float("inf")
        level = [root,]
        
while level:
            next_level = []
            temp_sum = 0
            for node in level:
                temp_sum += node.val
                if node.left:
                    next_level.append(node.left)
                if node.right:
                    next_level.append(node.right)
            if temp_sum > max_sum:
                max_sum, ans = temp_sum, current_level
            current_level += 1
            level = next_level
        return ans
