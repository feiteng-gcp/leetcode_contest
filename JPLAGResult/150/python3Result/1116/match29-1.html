# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        
        memo = collections.defaultdict(int)
        
        def dfs(root, dep):
            if not root:
                return
            memo[dep] += root.val
            dfs(root.left, dep + 1)
            dfs(root.right, dep + 1)
        
        dfs(root, 1)
        return max(memo.items(), key=lambda x:x[1])[0]
