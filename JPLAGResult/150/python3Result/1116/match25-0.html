# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        res, s_max = 1, root.val
        level, curr_level = 1, [root]
        while curr_level:
            next_level = []
            s = 0
            for node in curr_level:
                s += node.val
                if node.left:
                    next_level.append(node.left)
                if node.right:
                    next_level.append(node.right)
            if s > s_max:
                res, s_max = level, s
            curr_level = next_level
            level += 1
        return res
