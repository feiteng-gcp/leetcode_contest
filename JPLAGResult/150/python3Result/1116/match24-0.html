# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        q = [root]
        s = []
        while q:
            t = 0
            for i in range(len(q)):
                a = q.pop(0)
                t = t+a.val
                if a.left:
                    q.append(a.left)
                if a.right:
                    q.append(a.right)
            s.append(t)
        m = max(s)
        res = []
        for i in range(len(s)):
            if s[i] == m:
                return i+1
