# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        level = 1
        node_list = [root]
        max_sum = None
        ans = None
        while len(node_list) > 0:
            new_list 
= []
            now_sum = 0
            for each in node_list:
                now_sum += each.val
                if each.left != None:
                    new_list.append(each.left)
                if each.right != None:
                    new_list.append(each.right)
            if max_sum == None or now_sum > max_sum:
                max_sum = now_sum
                ans = level
            level += 1
            node_list = new_list
        return ans
