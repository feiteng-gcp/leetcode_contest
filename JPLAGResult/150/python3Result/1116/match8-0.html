# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        q = [root]
        best = -float('inf')
        bestL = None
        L = 0
        while q:
            L += 1
            s = 0
            newQ = []
            for node in q:
                s += node.val
                if node.left:
                    newQ.append(node.left)
                if node.right:
                    newQ.append(node.right)
            if s > best:
                best = s
                bestL = L
            q = newQ
        return bestL
