# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        if root is None:
            return 0
        
        queue = [(
root, 1)]
        result = 1
        best = 0
        
        level = 1
        sums = 0
        while len(queue) > 0:
            node, l = queue.pop(0)
            if l == level:
                sums += node.val
            else:
                if best < sums:
                    best = sums
                    result = level
                sums = node.val
                level =
 l
            
            if node.left is not None:
                queue.append((node.left, l+1))
            if node.right is not None:
                queue.append((node.right, l+1))
            
        
        return result
