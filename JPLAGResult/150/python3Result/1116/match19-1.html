# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        level = 1
        value = root.val
        # BFS
        layer 
= [root]
        i = 1
        while layer:
            s = sum([n.val for n in layer])
            if s > value:
                level = i
                value = s
            i += 1
            tmp = []
            for n in layer:
                if n.left:
                    tmp.append(n.left)
                if n.right:
                    tmp.append(n.right)
            layer = tmp
        return level
