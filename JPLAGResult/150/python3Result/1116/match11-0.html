# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        
        q = [root]
        level = 1
        res, res_lvl = float('-inf'), 0
        
        while q:
            n_q = []
            cur = 0
            for node in q:
                cur += node.val
                if node.left:
                    n_q.append(node.left)
                if node.right:
                    n_q.append(node.right)
            
            if cur > res:
                res = cur
                res_lvl = level
            q = n_q
            level += 1
        return res_lvl
