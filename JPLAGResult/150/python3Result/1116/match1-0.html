# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        curlev = [root]
        ret = 0
        idx = 1
        ridx = 0
        while curlev:
            newlev = []
            cursum = 0
            for n in curlev:
                cursum += n.val
                if n.left:
                    newlev.append(n.left)
                if n.right:
                    newlev.append(n.right)
            if cursum > ret:
                ridx = idx
                ret = cursum
            curlev = newlev
            idx += 1
        return ridx
