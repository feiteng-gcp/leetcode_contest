# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        if not root:
            return 0
        max_l = 0
        max_s = 0
        nodes = [root]
        level = 1
        while nodes:
            s = sum([n.val for n in nodes])
            if s > max_s:
                max_l = level
                max_s = s
            level += 1
            tmp = []
            for n in nodes:
                if n.left:
                    tmp.append(n.left)
                if n.right:
                    tmp.append(n.right)
            nodes = tmp
        return max_l
