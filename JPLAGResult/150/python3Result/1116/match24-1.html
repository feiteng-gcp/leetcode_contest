# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        # BFS
        q = [root]
        allL = []
        allSum = []
        while q:
            l = len(q)
            curSum = 0
            for _ in range
(l):
                curNode = q.pop(0)
                curSum += curNode.val
                if curNode.left:
                    q.append(curNode.left)
                if curNode.right:
                    q.append(curNode.right)
            allSum.append(curSum)
        key = max(allSum)
        ret = []
        for i in range(len(allSum)):
            if allSum[i] == key:
                return i+1
