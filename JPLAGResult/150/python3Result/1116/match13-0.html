# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        nodes=[root]
        l=1
        max_s= root.val
        cl = 1  
        while len(nodes)>0 :
            t_nodes=[]        
            last_level_sum = 0
            for node in nodes:
                last_level_sum += node.val
                if node.left != None:
                    t_nodes.append(node.left)
                if node.right!=None:
                    t_nodes.append(node.right)
            if last_level_sum >max_s:
                max_s = last_level_sum
                l = cl
            cl+=1
            nodes = t_nodes

        return l
