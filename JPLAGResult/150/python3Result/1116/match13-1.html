# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        ans = 1
        que = [root]
        level = 1
        best = 0
        while que:
            temp 
= []
            cur = 0
            for node in que:
                cur += node.val
                if node.left:
                    temp.append(node.left)
                if node.right:
                    temp.append(node.right)
            if cur > best:
                best = cur
                ans = level
            level += 1
            que = temp
        return ans
