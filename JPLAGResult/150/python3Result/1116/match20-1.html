# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        l = [root]
        s = -10 ** 10
        level 
= 0
        max_level = 0
        while l:
            level += 1
            l1 = []
            sub_sum = 0
            for li in l:
                sub_sum += li.val
                if li.left:
                    l1.append(li.left)
                if li.right:
                    l1.append(li.right)
            if sub_sum > s:
                s = sub_sum
                max_level = level
            l = l1
        return max_level
