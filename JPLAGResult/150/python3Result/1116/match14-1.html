# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        stack, max_sum, max_level, curr_level = [root], root.val, 1, 0
        
while(stack):
            curr_sum, curr_level, new = 0, curr_level + 1, []
            for node in stack:
                curr_sum += node.val
                if(node.left):
                    new.append(node.left)
                if(node.right):
                    new.append(node.right)
            #print((curr_level, curr_sum), (max_level, max_sum))
            if(curr_sum > max_sum):
                max_sum, max_level = curr_sum, curr_level
            stack = new
        return max_level
