# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        ml, ms = 1, float('-inf')
        l, q = 1, [root]
        while q:
            s, nq 
= 0, []
            for node in q:
                s += node.val
                if node.left: nq.append(node.left)
                if node.right: nq.append(node.right)
            if s > ms:
                ml, ms = l, s
            l, q = l + 1, nq
        return ml
