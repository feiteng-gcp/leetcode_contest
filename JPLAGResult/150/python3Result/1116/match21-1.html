# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        deque = collections.deque([(1, root)])
        res, maxsum 
= 1, -float("inf")
        while deque:
            cursum, curlevel = 0, res
            for i in range(len(deque)):
                curlevel, node = deque.popleft()
                cursum += node.val
                if node.left: deque.append((curlevel + 1, node.left))
                if node.right: deque.append((curlevel + 1, node.right))
            if cursum > maxsum:
                maxsum = cursum
                res = curlevel
        return res
