# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        def dfs(sums, root, d):
            if root is None:
                return
            
            if len(sums) == d:
                sums.append(0)
            sums[d] += root.val
            dfs(sums, root.left, d + 1)
            dfs(sums, root.right, d + 1)
            
        sums = []
        dfs(sums, root, 0)
        max_level = max(sums)
        for i, n in enumerate(sums):
            if n == max_level:
                return i + 1
        return 0
