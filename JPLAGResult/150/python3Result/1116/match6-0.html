# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        q, res, ret, level = [root], 0, 0, 0
        while any(q):
            cur, tmp = [], 0
            for _ in range(len(q)):
                node = q.pop(0)
                tmp += node.val
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            level += 1
            if tmp > res:
                res = tmp
                ret = level
        return ret
