# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def maxLevelSum(self, root: TreeNode) -> int:
        # BFS
        now_at = 0
        now_bin = [root]
        now_ret, now_max = 1, root.val
        
        while now_bin:
            next_bin = []
            cnt = 0
            now_at += 1
            for node in now_bin:
                cnt += node.val
                if node.left:
                    next_bin.append(node.left)
                if node.right:
                    next_bin.append(node.right)
            if cnt > now_max:
                now_ret, now_max = now_at, cnt
            now_bin = next_bin
        return now_ret
