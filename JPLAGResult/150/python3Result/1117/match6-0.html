class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        if sum(sum(g) for g in grid) in (0, m*n):
            return -1
        
        
        rez = 0
        stack = set()
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    stack.add((i,j))
                    grid[i][j] = -1
        
        dis = 0
        while(stack):
            tmpStk = set()
            for s in stack:
                if grid[s[0]][s[1]] == -2 and rez < dis:
                    rez = dis
                if s[0] and grid[s[0] - 1][s[1]] >= 0:
                    tmpStk.add((s[0] - 1, s[1]))
                    grid[s[0] - 1][s[1]] = -2
                if s[1] and grid[s[0]][s[1] - 1] >= 0:
                    tmpStk.add((s[0], s[1] - 1))
                    grid[s[0]][s[1] - 1] = -2
                if s[0] != m - 1 and grid[s[0] + 1][s[1]] >= 0:
                    tmpStk.add((s[0] + 1, s[1]))
                    grid[s[0] + 1][s[1]] = -2
                if s[1] != n - 1 and grid[s[0]][s[1] + 1] >= 0:
                    tmpStk.add((s[0], s[1] + 1))
                    grid[s[0]][s[1] + 1] = -2
            stack = tmpStk
            dis += 1

        
        return rez
