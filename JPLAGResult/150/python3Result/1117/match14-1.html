class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        totalsum = sum(sum(l) for l in grid)
        if totalsum == len(grid) * len(grid[0]) or totalsum == 0: return -1
        deque 
= collections.deque()
        visited = set()
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    deque.append((0, (i, j)))
                    visited.add((i, j))
        
        maxsize = 0
        while deque:
            d, (i, j) = deque.popleft()
            maxsize 
= max(maxsize, d)
            for dx, dy in [[-1, 0], [1, 0], [0, 1], [0, -1]]:
                nx, ny = i + dx, j + dy
                if 0 <= nx < len(grid) and 0 <= ny < len(grid[0]) and (nx, ny) not in visited:
                    deque.append((d + 1, (nx, ny)))
                    visited.add((nx, ny))
        return maxsize
