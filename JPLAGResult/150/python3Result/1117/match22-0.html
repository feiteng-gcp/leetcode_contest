class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        direcitons = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        R, C = len(grid), len(grid[0])
        lst = []
        
        for i in range(R):
            for j in range(C):
                if grid[i][j] == 1:
                    lst.append((i, j, 0))
        
        if len(lst) == R * C or len(lst) == 0: return -1
        
        queue = collections.deque(lst)
        # seen = set(lst)
        
        while queue:
            for _ in range(len(queue)):
                r, c, step = queue.popleft()
                for d in direcitons:
                    nr, nc = r + d[0], c + d[1]
                    if 0 <= nr < R and 0 <= nc < C and grid[nr][nc] == 0:
                        queue.append((nr, nc, step + 1))
                        grid[nr][nc] = 1
        
        return step
