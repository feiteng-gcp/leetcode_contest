class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        nodes = []
        row_size = len(grid)
        col_size = len(grid[0])
        
for i in range(row_size):
            for j in range(col_size):
                if grid[i][j] == 1:
                    nodes.append((i, j))
        distence = [(0,1), (0, -1), (1, 0), (-1, 0)]
        def dfs(nodes):
            res = []
            for x, y in nodes:
                for i, j in distence:
                    
if 0 <= x + i < row_size and 0 <= y + j < col_size and grid[x+i][y+j] == 0:
                        res.append((x+i, y+j))
                        grid[x+i][y+j] = 1
            return res
        res = 0
        while nodes:
            next_nodes = dfs(nodes)
            nodes = next_nodes
            res += 1
        return -1 if res - 1 == 0 else res - 1
