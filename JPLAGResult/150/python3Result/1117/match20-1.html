class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        R, C = len(grid), len(grid[0])
        dist = {}
        q = collections.deque()
        
for i in range(R):
            for j in range(C):
                if grid[i][j] == 1:
                    q.append((i,j))
                    dist[(i,j)] = 0
        if len(q) in (0, R*C): return -1
        maxd = 0
        while q:
            u, v = q.popleft()
            for dx, dy in ((0, 1), (1, 0), (0, -1), (-1, 0)):
                a, b = u + dx, v + dy
                if a < 0 or b < 0 or a >= R or b >= C: continue
                if grid[a][b] != 0: continue
                grid[a][b] = dist
[(u, v)] + 1
                dist[(a, b)] = dist[(u, v)] + 1
                maxd = max(maxd, dist[(a, b)])
                q.append((a, b))
        return maxd
