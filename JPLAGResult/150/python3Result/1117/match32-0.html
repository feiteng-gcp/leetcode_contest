class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        
        m, n = len(grid), len(grid[0])
        if all(num == 0 for i, r in enumerate(grid) for j, num in enumerate(r)):
            return -1
        if all(num == 1 for i, r in enumerate(grid) for j, num in enumerate(r)):
            return -1
        
        def update(i, j):
            
            res = -1
            q = [(i, j)]
            
            while q:
                n_q = []
                for x, y in q:
                    for nx, ny in [(x+1, y), (x-1, y), (x, y+1), (x, y-1)]:
                        if 0 <= nx < m and 0 <= ny < n and grid[nx][ny] <= 0:
                            if grid[nx][ny] == 0 or grid[nx][ny] < res:
                                grid[nx][ny] = res
                                n_q.append((nx, ny))
                q = n_q
                res -= 1
        
        for i, r in enumerate(grid):
            for j, num in enumerate(r):
                if num == 1:
                    update(i, j)
                    
        ans = min(num for i, r in enumerate(grid) for j, num in enumerate(r))
        return -ans
