class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        R, C = len(grid), len(grid[0])
        visited = [[0] * C for _ in range(R)]
        q = collections.deque()
        for i in range(R):
            for j in range(C):
                if grid[i][j] == 1:
                    q.append((i, j, 0))
                    visited[i][j] = 1
        res = -1           
        while q:
            i, j, d = q.popleft()
            for ni, nj in ((i-1, j), (i, j-1), (i, j+1), (i+1, j)):
                if 0 <= ni < R and 0 <= nj < C and not visited[ni][nj]:
                    visited[ni][nj] = 1
                    q.append((ni, nj, d+1))
                    res = d+1
        return res
