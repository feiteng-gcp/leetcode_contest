class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        queue = []
        visited = set()
        m = len(grid)
        n = len(grid[0])
        water = 0
        land = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    queue.append((i, j))
                    visited.add((i, j))
                    land += 1
                elif grid[i][j] == 0:
                    water += 1
        
        level = -1
        while queue:
            level += 1
            for _ in range(len(queue)):
                i, j = queue.pop(0)
                for x, y in [(i - 1, j), (i + 1, j), (i, j - 1), (i, j + 1)]:
                    if 0 <= x < m and 0 <= y < n and (x, y) not in visited:
                        queue.append((x, y))
                        visited.add((x, y)) 
            
        if land == 0 or water == 0:
            return -1
        else:
            return level
