class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        N = len(grid)
        d = [[0 for i in range(N)] for j in range(N)]
        count = []
        for i in range(N):
            for j in range(N):
                if grid[i][j] == 1:
                    count.append((i, j))
                    d[i][j] = 1
        if len(count) == 0 or len(count) == N * N:
            return -1
        max_length = 0
        while count:
            max_length += 1
            temp = count.copy()
            print(max_length, temp)
            count.clear()
            for i, j in temp:
                if i > 0 and not d[i-1][j]:
                    count.append((i-1,j))
                    d[i-1][j] = 1
                if j > 0 and not d[i][j-1]:
                    count.append((i,j-1))
                    d[i][j-1] = 1
                if i < N - 1 and not d[i+1][j]:
                    count.append((i+1,j))
                    d[i+1][j] = 1
                if j < N - 1 and not d[i][j+1]:
                    count.append((i,j+1))
                    d[i][j+1] = 1
        return max_length - 1
