class Solution:
	def maxDistance(self, grid: [[int]]) -> int:
		n = len(grid)
		mat = [[-1]*n for _ in range(n)]
		q = []
		for r in range(n):
			for c in range(n):
				if grid[r][c] == 1:
					q.append((r, c))
					mat[r][c] = 0

		mr = [0, 1, 0, -1]
		mc = [1, 0, -1, 0]
		mx = -1
		while q: 
			(r, c) = q.pop(0)
			for i in range(4):
				if r+mr[i] >= 0 and r+mr[i] < n and c+mc[i] >= 0 and c+mc[i] < n and mat[r+mr[i]][c+mc[i]] == -1:
					mat[r+mr[i]][c+mc[i]] = mat[r][c]+1
					mx = max(mx, mat[r][c]+1)
					q.append((r+mr[i], c+mc[i]))

		return mx
