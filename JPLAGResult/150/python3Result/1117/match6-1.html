class Solution:
    def maxDistance(self, grid):
        l = len(grid)
        a = [[i, j] for i in range(l) for j in range(l) if grid[i][j] == 1]
        if a and len(a) < l*l:
            ans = -1
            while a:
                ans += 1
                at = []
                for i in a:
                    
if i[0] > 0 and grid[i[0]-1][i[1]] == 0:
                        at.append([i[0]-1, i[1]])
                        grid[i[0] - 1][i[1]] = 1
                    if i[0] + 1 < l and grid[i[0]+1][i[1]] == 0:
                        at.append([i[0]+1, i[1]])
                        grid[i[0]+1][i[1]] = 1
                    if i[1] > 0 and grid[i[0]][i[1]-1] == 0:
                        at.append([i[0], i[1]-1])
                        grid[i[0]][i[1]-1] = 1
                    if i[1] + 1 < l and grid[i[0]][i[1]+1] == 0:
                        at.append([i[0], i[1]+1])
                        grid[i[0]][i[1]+1] = 1
                a = at
            return ans
        return -1
