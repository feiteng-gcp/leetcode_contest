class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        if len(grid) == 0 or len(grid[0]) == 0:
            return -1
        
        ones = set()
        zeros = set()
        m,n = len(grid), len(grid[0])
        for i in range(0, m):
            for j in range(0, n):
                if grid[i][j]:
                    ones.add((i,j))
                else:
                    zeros.add((i,j))
        if len(ones) == 0 or len(zeros) == 0:
            return -1 
        level = 0
        
while zeros:
            new_ones = set()
            for i,j in ones:
                for ii,jj in [[0,1],[0,-1],[-1,0],[1,0]]:
                    ni,nj = i+ii, j+jj
                    if (ni,nj) in zeros:
                        zeros.discard((ni,nj))
                        new_ones
.add((ni,nj))
            ones = new_ones
            level += 1
        return level
