class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        N = len(grid)

        curr_level = 0
        curr_set = set()
        for i in range(N):
            for j in range(N):
                if grid[i][j]==1:
                    curr_set.add((i,j))        
        if len(curr_set)==0 or len(curr_set)==N*N:
            return -1
        
        
        
        while True:
            next_set = set()
            for e in curr_set:
                if e[0]>0:
                    i, j = e[0]-1, e[1]
                    if grid[i][j]==0:
                        grid[i][j] = 1
                        next_set.add((i,j))
                if e[0]<N-1:
                    i, j = e[0]+1, e[1]
                    if grid[i][j]==0:
                        grid[i][j] = 1
                        next_set.add((i,j))                    
                if e[1]>0:
                    i, j = e[0], e[1]-1
                    if grid[i][j]==0:
                        grid[i][j] = 1
                        next_set.add((i,j))                    
                if e[1]<N-1:
                    i, j = e[0], e[1]+1
                    if grid[i][j]==0:
                        grid[i][j] = 1
                        next_set.add((i,j))
            if next_set:
                curr_level += 1
                curr_set = next_set
            else:
                break
        
        
        return curr_level
