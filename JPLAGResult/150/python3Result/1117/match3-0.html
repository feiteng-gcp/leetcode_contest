class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        visited = [n * [False] for _ in range(m)]
        d, curr_level = -1, set()
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    curr_level.add((i, j))
                    visited[i][j] = True
        if len(curr_level) == 0 or len(curr_level) == m * n:
            return d
        while curr_level:
            next_level = set()
            for x, y in curr_level:
                for dx, dy in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                    r, c = x + dx, y + dy
                    if 0 <= r < m and 0 <= c < n and not visited[r][c]:
                        visited[r][c] = True
                        next_level.add((r, c))
            d += 1
            curr_level = next_level
        return d
