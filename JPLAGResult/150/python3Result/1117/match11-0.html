class Solution:
    def maxDistance(self, A: List[List[int]]) -> int:
        m, n = len(A), len(A[0])
        q = collections.deque()
        
        total = m * n
        res = 0
        for i in range(m):
            for j in range(n):
                if A[i][j]:
                    q.append((i, j))
                    total -= 1
        if total == m * n or total == 0:
            return -1
                    
        de = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        while q:
            size = len(q)
            res += 1
            for _ in range(size):
                x, y = q.popleft()
                for (a, b) in de:
                    c = x + a
                    d = y + b
                    
                    if 0 <= c < m and 0 <= d < n and A[c][d] == 0:
                        A[c][d] = 1
                        total -= 1
                        q.append((c, d))
                
            if total == 0: break
        return res
