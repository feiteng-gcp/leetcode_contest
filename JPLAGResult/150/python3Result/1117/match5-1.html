from collections import deque
class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        queue = deque()
        n = len(grid)
        for i in range(n):
            for j in range(n):
                if grid[i][j] == 1:
                    queue.append((i,j))
        
if len(queue) == n**2 or len(queue) == 0:
            return -1
        ans = -1
        while queue:
            l = len(queue)
            for _ in range(l):
                x, y = queue.popleft()
                for dx, dy in [[-1,0],[1,0],[0,1],[0,-1]]:
                    nx, ny = x+dx, y+dy
                    if 0<=nx<n and 0<=ny<n and grid[nx][ny] == 0:
                        grid[nx][ny] = 1
                        queue.append((nx, ny))
            ans += 1
        return ans
