class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        R, C = len(grid), len(grid[0])
        
        dist = [[-1] * C for _ in range(R)]
        
        queue = collections.deque()
        for r in range(R):
            for c in range(C):
                if grid[r][c] == 1:
                    queue.append((r, c))
                    dist[r][c] = 0
        
        def neighbor(node):
            r, c = node[0], node[1]
            for nr, nc in [[r+1, c], [r, c+1], [r-1, c], [r, c-1]]:
                if 0 <= nr < R and 0 <= nc < C and dist[nr][nc] == -1:
                    yield nr, nc
        
        
        while queue:
            node = queue.popleft()
            d = dist[node[0]][node[1]]
            for nei in neighbor(node):
                queue.append(nei)
                dist[nei[0]][nei[1]] = d + 1
        
        maxx = 0
        for r in range(R):
            for c in range(C):
                maxx = max(maxx, dist[r][c])
    
        return -1 if maxx == 0 else maxx
