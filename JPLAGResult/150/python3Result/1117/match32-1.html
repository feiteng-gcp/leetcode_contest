class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        def bfs(i, j):
            stack = [(i, j, 0)]
            while(stack):
                i, j, step = stack.pop(0)
                for n_i, n_j in ((i + 1, j), (i - 1, j), (i, j + 1), (i, j - 1)):
                    if(0 <= n_i < n and 0 <= n_j < n):
                        if(grid[n_i][n_j] == 1):
                            return step + 1
                        stack.append((n_i, n_j, step + 1))
        
        def dp(i, j):
            curr_step, stack 
= 0, [(i, j)]
            while(stack):
                new, curr_step = [], curr_step + 1
                for i, j in stack:
                    for n_i, n_j in ((i + 1, j), (i - 1, j), (i, j + 1), (i, j - 1)):
                        if(0 <= n_i < n and 0 <= n_j < n):
                            
if(grid[n_i][n_j] == 0 or grid[n_i][n_j] < - curr_step):
                                grid[n_i][n_j] = - curr_step
                                new.append((n_i, n_j))
                stack = new
            #print((i, j))
            #print(grid)
            #print('\n')
            
        
        
        sums = sum(sum(line) for line in grid)
        n = len(grid)
        if(sums == 0 or sums == n * n):
            return -1
        
        max_distance = 1
        if(sums > n * n // 2):
        #if(False):
            for i in range(n):
                for j in range(n):
                    if(grid[i][j] == 0):
                        max_distance = max(max_distance, bfs(i, j))
        else:
            #print('start')
            for i in range(n):
                for j in range(n):
                    if(grid[i][j] == 1):
                        dp(i, j)
                        
            for i in range(n):
                for j in range(n):
                    max_distance = max(max_distance, - grid[i][j])
            
        return max_distance
