class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        '''
        have a dist grid
        start at 1s, do BFS
        '''
        ones = set()
        dist_grid = [[float('inf')]*len(grid[0]) for _ in range(len(grid))]
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    ones.add((i,j))
                    dist_grid[i][j] = -1
        if len(ones) == 0 or len(ones) == len(grid[0])*len(grid):
            return -1
        for x, y in ones:
            self.update(dist_grid, x, y, ones)
        res = 0
        for row in dist_grid:
            for cell in row:
                res = max(res, cell)
        return res
    
    def update(self, dist_grid, x, y, ones):
        queue = []
        for m, n in [(-1,0),(1,0),(0,-1),(0,1)]:
            xm, yn = x+m, y+n
            if xm >= 0 and xm < len(dist_grid) and yn >= 0 and yn < len(dist_grid[0]) and (xm, yn) not in ones:
                queue.append((xm,yn,1))
        while queue:
            cur_x, cur_y, cur_d = queue.pop(0)
            if cur_d < dist_grid[cur_x][cur_y]:
                dist_grid[cur_x][cur_y] = cur_d
                for m, n in [(-1,0),(1,0),(0,-1),(0,1)]:
                    xm, yn = cur_x + m, cur_y + n
                    if xm >= 0 and xm < len(dist_grid) and yn >= 0 and yn < len(dist_grid[0]) and (xm, yn) not in ones:
                        queue.append((xm,yn,cur_d+1))
