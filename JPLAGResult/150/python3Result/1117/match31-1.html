class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        n=len(grid)
        m=len
(grid[0])
        L=[[-1]*m for _ in range(n)]
        q=collections.deque([])
        for i in range(n):
            for j in range(m):
                if grid[i][j]==1:
                    q.append((i,j))
                    L[i][j]=0
        while(q):
            i,j=q.popleft()
            if i>0:
                if L[i-1][j]==-1:
                    L[i-1][j]=L[i][j]+1
                    q.append((i-1,j))
            if j>0:
                if L[i][j-1]==-1:
                    L[i][j-1]=L[i][j]+1
                    q.append((i,j-1))
            if i<n-1:
                if L[i+1][j]==-1:
                    L[i+1][j]=L[i][j]+1
                    q.append((i+1,j))
            if j<m-1:
                if L[i][j+1]==-1:
                    L[i][j+1]=L[i][j]+1
                    q.append((i,j+1))
        A=max([max(S) for S in L])
        if A<=0:
            return -1
        return A
