class Solution:
    def maxDistance(self, grid: 'List[List[int]]') -> int:
        dist=[];lr=len(grid);lc=len(grid[0])
        for a in grid:
            tmp=[]
            for x in a:
                if x==1:
                    tmp.append(0)
                else:
                    tmp.append(float('inf'))
            dist.append(tmp)
        cur=[]
        
for i in range(lr):
            for j in range(lc):
                if grid[i][j]==1:
                    cur.append((i,j))
        if len(cur)==0 or len(cur)==lr*lc:
            return -1

        dr=[0,1,0,-1]
        dc=[1,0,-1,0]
        while cur:
            newcur=[]
            for point in cur:
                x,y=point
                for k in range(0,4):
                    r,c=x+dr[k],y+dc[k]
                    if r>=0 and r<lr and c>=0 and c<lc and dist[r][c]>dist
[x][y]+1:
                        dist[r][c]=dist[x][y]+1
                        newcur.append((r,c))
                    else:
                        ""
            cur=newcur
        ans=0
        for i in range(lr):
            tmp=max(dist[i])
            ans=max(ans,tmp)
        return ans
