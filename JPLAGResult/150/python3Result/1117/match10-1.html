class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        m,n = len(grid), len(grid[0])
        total = 0
        visited = [[False]*n for i in range(m)]
        q 
= collections.deque()
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    q.append([i,j])
                    visited[i][j] = True
                    total += 1
        if total == m*n or 0:
            return -1
        dist 
= 0
        while q:
            size = len(q)
            for _ in range(size):
                i,j = q.popleft()
                for di, dj in [[-1,0],[1,0],[0,-1],[0,1]]:
                    i1 = i+di
                    j1 = j+dj
                    if 0<=i1<m and 0<=j1<n and not visited[i1][j1]:
                        q.append([i1,j1])
                        visited[i1][j1] = True
            dist += 1
        return dist-1
