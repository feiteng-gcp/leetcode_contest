from collections import deque

class Solution:
    def maxDistance(self, grid: 'List[List[int]]') -> int:
        BASE = 10

        h = len(grid)
        w = len(grid[0])

        q = deque()

        
for i in range(h): 
            for j in range(w): 
                if grid[i][j] == 1: 
                    q.append((i, j))
        
        if not q or len(q) == w * h: 
            return -1
        
        q.append(-1)
        dist = 1

        while q: 
            #print(q, dist)
            front = q.popleft() 

            if front == -1: 
                if not q: 
                    break 

                dist += 1
                q.append(-1)
                continue
            
            i, j = front

            for di, dj in ((1,0), (-1,0), (0,1), (0,-1)): 
                ni = i + di
                nj = j + dj 

                
if ni < 0 or ni >= h or nj < 0 or nj >= w: 
                    continue 
                
                if grid[ni][nj] == 0: 
                    grid[ni][nj] = BASE + dist
                    q.append((ni, nj))
        
        #print(grid)
        #print(dist)

        return dist-1
