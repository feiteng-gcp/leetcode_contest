class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        q = collections.deque()
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    q.append((i, j))
                    
        ans = -1
        step = 2
        while q:
            size = len(q)
            for i in range(size):
                x, y = q.popleft()
                for dx, dy in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                    nx, ny = dx + x, dy + y
                    if 0<= nx < m and 0 <= ny < n and grid[nx][ny] == 0:
                        grid[nx][ny] = step
                        q.append((nx, ny))
            step += 1
    
        for i in range(m):
            for j in range(n):
                ans = max(grid[i][j], ans)
        return ans-1 if ans > 1 else -1
