class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        dist = -1
        q = collections.deque()
        M, N = len(grid), len(grid[0])
        visited = set()
        for i in range(M):
            for j in range(N):
                if grid[i][j] == 1:
                    q.append((i,j))
                    visited.add((i,j))
        direction = [(0, 1), (0, -1), (1, 0), (-1, 0)]
        while q:
            dist += 1
            size = len(q)
            for _ in range(size):
                i, j = q.popleft()
                for dx, dy in direction:
                    x, y = i + dx, j + dy
                    if 0 <= x < M and 0 <= y < N and (x,y) not in visited:
                        visited.add((x,y))
                        q.append((x, y))
        if dist == 0:
            return -1
        return dist
