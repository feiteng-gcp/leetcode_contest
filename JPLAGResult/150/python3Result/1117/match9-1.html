class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        if not grid:
            return -1
        s = 0
        row = len(grid)
        col = len(grid[0])
        land_stack = []
        sea_stack = []
        dp = [[True for _ in range(col)] for __ in range
(row) ]
        for r in range(row):
            for c in range(col):
                if grid[r][c] == 1:
                    land_stack.append([r, c])
                    dp[r][c]=False
                else:
                    sea_stack.append([r,c])
        if len(land_stack) == 0 or len(sea_stack) == 0:
            return -1
        ans = -1
        while land_stack:
            ans +=1
            temp = []
            for x,y in land_stack:
                for dx,dy in [[1,0],[0,1],[-1,0],[0,-1]]:
                    if -1<x+dx<row and -1<y+dy<col and grid[x+dx][y+dy]==0 and dp
[x+dx][y+dy]:
                        temp.append([x+dx,y+dy])
                        dp[x+dx][y+dy]=False
            land_stack = temp
        return ans
