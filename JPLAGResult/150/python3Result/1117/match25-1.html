class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        queue 
= collections.deque()
        m, n = len(grid), len(grid[0])
        
        visited = set()
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    queue.append((i, j))
                    visited.add((i, j))
        if len(queue) == m * n:
            return -1
        
        ans = 0
        dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        while queue:
            size = len(queue)
            for _ in range(size):
                i, j = queue.popleft()
                for dr in dirs:
                    x, y = i + dr[0], j + dr[1]
                    if (x, y) not in visited and x >= 0 and y >= 0 and x < m and y < n:
                        queue.append((x, y))
                        visited.add((x, y))
            ans += 1
        return ans - 1
