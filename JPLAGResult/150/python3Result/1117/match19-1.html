class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        
        queue = collections.deque()
        
        if not grid or not grid[0]:
            return -1
        
        m = len(grid)
        n = len
(grid[0])
        
        flag0 = 0
        flag1 = 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    queue.append((i,j,0))
                    flag1 = 1
                else:
                    flag0 = 1
        
        if not flag0 or not flag1:
            return -1
        
        res = 0
          
        while queue:
            x,y,val = queue.popleft()
            diff = [(0,1),(0,-1),(1,0),(-1,0)]
            for d in diff:
                nx, ny = x+d[0], y+d[1]
                if 0<=nx<m and 0<=ny<n and grid[nx][ny] == 0: #we find a new one!
                    res = max(res,val+1)
                    grid
[nx][ny] = val+1
                    queue.append((nx,ny,val+1))
        return res
