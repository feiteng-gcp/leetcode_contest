class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        m = len(grid); n = len(grid[0]) 
        visited = [[0 for i in range(n)] for j in range(m)]
        distance = [[float('inf') for i in range
(n)] for j in range(m)]
        q = []
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    q.insert(0, (i,j))
                    visited[i][j] = 1; distance[i][j] = 0
        
        if not len(q):
            return -1
        
        while len(q):
            elem = q.pop()
            i = elem[0]; j = elem[1]
            if i > 0 and not visited[i-1][j]:
                distance[i-1][j] = min(distance[i-1][j], distance
[i][j] + 1)
                q.insert(0, (i-1, j)); visited[i-1][j] = 1
            if j > 0 and not visited[i][j-1]:
                distance[i][j-1] = min(distance[i][j-1], distance
[i][j] + 1)
                q.insert(0, (i, j-1)); visited[i][j-1] = 1
            if i < m-1 and not visited[i+1][j]:
                distance[i+1][j] = min(distance[i+1][j], distance[i][j] + 1)
                q.insert(0, (i+1, j)); visited[i+1][j] = 1
            if j < n-1 and not visited[i][j+1]:
                distance[i][j+1] = min(distance[i][j+1], distance[i][j] + 1)
                q.insert(0, (i, j+1)); visited[i][j+1] = 1
        
        ans = -1
        for i in range(m):
            for j in range(n):
                if not grid[i][j] and distance[i][j] > ans:
                    ans = distance[i][j]
                    
        return ans
