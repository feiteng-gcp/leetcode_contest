class Solution:
    def maxDistance(self, A: List[List[int]]) -> int:
        m, n = len(A), len(A[0])
        
        q = collections.deque()
        
        water = 0
        land = 0
        for i in range(m):
            for j in range(n):
                if A[i][j] == 1:
                    A[i][j] = [1, 0]
                    water += 1
                    q.append((i, j, 0))
                else:
                    A[i][j] = [0, float('inf')]
                    land 
+= 1
        
        if water == 0 or land == 0:
            return -1
        
        d = [(1, 0), (-1, 0), (0, 1), (0, -1)]
        while q:
            i, j, dis = q.popleft()
            
            for k in d:
                i_n, j_n = i + k[0], j +k[1]
                if 0 <= i_n < m and 0 <= j_n < n and A[i_n][j_n][0] == 0 and dis + 1 < A[i_n][j_n][1]:
                    # print(A)
                    A[i_n][j_n] = [1, dis+1]
                    q.append((i_n, j_n, dis+1))
        
        # print(A)
        
        res = -float('inf')
        
        for i in range(m):
            for j in range(n):
                if A[i][j][1] > res:
                    res = A[i][j][1]
        
        return res if res > -float('inf') else -1
