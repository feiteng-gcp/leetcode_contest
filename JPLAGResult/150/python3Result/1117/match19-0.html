class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        m,n = len(grid), len(grid[0])
        ones = []
        flag = False
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    ones.append((0,i,j))
                else:
                    flag = True
        res = -1
        if not flag:
            return -1
        while ones:
            level,x0,y0 = ones.pop(0)

            
            res = max(level,res)
            for dx,dy in [(0,1),(1,0),(-1,0),(0,-1)]:
                x,y = x0+dx, y0+dy
                if 0<=x<m and 0<=y<n and not grid[x][y]:
                    grid[x][y] = 1
                    ones.append((level+1,x,y))
        return res
