class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        landset = collections.deque()
        visited = set()
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    landset.append((i, j, 0))
                    # visited.add((i, j))
        
        if len(landset) == 0 or len(landset) == len(grid) * len(grid[0]):
            return -1
        
        maxdis = 0
        while landset:
            # print(landset)
            i, j, d = landset.popleft()
            if (i, j) in visited:
                continue
            visited.add((i, j))
            maxdis = max(maxdis, d)
            for di, dj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                ni, nj = i + di, j + dj
                if ni < 0 or ni >= len(grid) or nj < 0 or nj >= len(grid[0]):
                    continue
                landset.append((ni, nj, d + 1))
                
        return maxdis
