class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        q = []
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    q.append((i, j))
        L = 0
        while q:
            newQ = []
            for i, j in q:
                for u, v in [[i + 1, j], [i - 1, j], [i, j + 1], [i, j - 1]]:
                    if not (0 <= u < m and 0 <= v < n):
                        continue
                    if grid[u][v] == 0:
                        grid[u][v] = 1
                        newQ.append((u, v))
            L += 1
            q = newQ
        return L - 1 if L > 1 else -1
