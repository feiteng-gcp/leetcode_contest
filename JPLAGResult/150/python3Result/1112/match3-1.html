class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        cnter = collections.Counter(chars)
        s = 0
        for w in words:
            c = collections.Counter(w)
            allin = True
            for k, v in c.items():
                
if k not in cnter or c[k] > cnter[k]:
                    allin = False
                    break
            if allin:
                s += len(w)
        return s
