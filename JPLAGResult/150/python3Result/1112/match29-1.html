class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        
        c_counter = collections.Counter(chars)
        res = 0
        for w in words:
            w_counter = collections.Counter(w)
            flag = True
            for k,v in w_counter.items():
                if not (k in c_counter and c_counter[k] >= v):
                    flag 
= False
                    break
            if flag: res += len(w)
        return res
