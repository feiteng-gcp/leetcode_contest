import collections


class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        char_count = collections.Counter(chars)
        result = 0
        for w in words:
            w_count = collections.Counter(w)
            for c in w_count:
                if w_count[c] > char_count[c]:
                    break
            else:
                result += len(w)
        return result
