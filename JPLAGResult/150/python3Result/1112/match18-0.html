class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        s = collections.Counter(chars)
        ans = 0
        for x in words:
            v = collections.Counter(x)
            flag = True
            for c in v:
                if c not in s or s[c] < v[c]:
                    flag = False
                    break
            if flag:
                ans += len(x)
        return ans
