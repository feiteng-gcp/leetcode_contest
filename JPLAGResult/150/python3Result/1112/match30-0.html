from collections import Counter

class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        
        dictChars = Counter(chars)
        ans = 0
        for word in words:
            dictWord = Counter(word)
            flag = True
            for char in dictWord:
                if char not in dictChars or dictWord[char] > dictChars[char]:
                    flag = False
                    break
            
            if flag:
                ans += len(word)
        
        return ans
