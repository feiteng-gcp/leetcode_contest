class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        counter = collections.Counter(chars)
        res = 0
        for word in words:
            w = collections.Counter(word)
            flag = True
            for letter in w:
                if letter not in counter or counter[letter] < w[letter]:
                    flag = False
                    break
            if flag:
                res += len(word)
        return res
