class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        data = [collections.Counter(i) for i in words]
        chs = collections.Counter(chars)
        res = 0
        for i, k in enumerate(data):
            for j, v in k.items():
                if chs[j] < v:
                    break
            else:
                res += len(words[i])
        return res
