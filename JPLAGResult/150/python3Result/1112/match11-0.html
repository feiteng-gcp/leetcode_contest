class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        candidate = collections.Counter(chars)
        res = 0
        for w in words:
            target = collections.Counter(w)
            for c in target:
                if target[c] > candidate[c]:
                    break
            else:
                res += len(w)
        return res
