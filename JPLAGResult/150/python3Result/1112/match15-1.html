from collections import Counter
class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        
        cnter = Counter(chars)
        res = 0
        for w in words:
            cur = Counter(w)
            is_good = True
            for c in cur:
                if cur[c] > cnter[c]:
                    is_good = False
                    break
            if is_good:
                res += len(w)
        return res
