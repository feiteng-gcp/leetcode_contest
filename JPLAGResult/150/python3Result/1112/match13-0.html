class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        counter = collections.Counter(chars)
        res = 0
        for word in words:
            d = collections.Counter()
            for c in word:
                d[c] += 1
                if d[c] > counter[c]:
                    break
            else:
                res += len(word)
        return res
