class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        res = 0
        ccounter = collections.Counter(chars)
        for word in words:
            wcounter = collections.Counter(word)
            if all(ccounter[k] >= wcounter[k] for k in wcounter):
                res += len(word)
        return res
