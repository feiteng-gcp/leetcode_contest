class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        res = 0
        count = collections.Counter(chars)
        for word in words:
            flag = True
            dic = collections.Counter(word)
            for key in dic:
                if dic[key] > count[key]:
                    flag = False
                    break
            if flag:
                res += len(word)
        return res
