class Solution {
public:
    string lastSubstring(string s) {
        int size = s.size();
        char start = 'a';

        for(const auto& c : s) {
            if(c > start) {
                start = c;
            }
        }

        int res = -1;

        for(int i = size - 1; i >= 0; --i) {
            if(s[i] == start) {
                if(-1 == res) {
                    res = i;
                }
                else {
                    int len = min(size - res, res - i);

                    if(mycompare(s, i, res, len)) {
                        res = i;
                    }
                }
            }
        }

        return s.substr(res);
    }

private:
    bool mycompare(const string& s, int a, int b, int len) {
        for(int i = 0; i < len; ++i) {
            if(s[a + i] > s[b + i]) {
                return true;
            }

            if(s[a + i] < s[b + i]) {
                return false;
            }
        }

        return true;
    }
};
