class Solution {
public:
    string lastSubstring(string s) {
        int ans = -1;
        char mc = *max_element(s.begin(), s.end());
        
        auto data = s.c_str();

        int ls = s.size();
        
        for (int i = 0; i < ls; i++) {
            if (s[i] == mc) {
                if (ans == -1) {
                    ans = i;
                } else {
                    if (memcmp(data+ans, data+i, ls-i) < 0) ans = i;
                    
                }
            }
            
        }
        
        return s.substr(ans);
        
    }
};
