class Solution {
public:
    string lastSubstring(string s) {
        // 最后一个一定是 首个字符是 当前序列中 字典序最大，然后递减,相等则越长越好
        int maxi=1;
        auto cmp = [](string a, string b){
            return a<b;    
        };
        priority_queue<string, vector<string>, decltype(cmp)> q(cmp);
        char c = 'a';
        for ( auto e:s )
            if ( e>c )
                c = e;
        for ( int i=0; i<s.size(); ++i ){
            if ( s[i]==c ){
                int r = i;
                while( r<s.size() && s[r]==c ) ++r;
                if (  r-i+1 >= maxi ){
                    q.push(s.substr(i));
                    maxi = r-i+1;
                }
                i = r-1;
                
            }
        }
        return q.top();
        
    }
};
