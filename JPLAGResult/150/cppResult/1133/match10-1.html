static const int maxN = 2e5+5;
int len,s[maxN];
int sum[maxN],wv[maxN],x[maxN],y[maxN],sa[maxN];
char a[maxN];
int rk[maxN];
void build_sa()

{
    int n=len,m=255;
    for (int i=0;i<=m;++i) sum[i]=0;
    for (int i=1;i<=n;++i) sum[x[i]=s[i]]++;
    for (int i=1;i<=m;++i) sum[i]+=sum[i-1];
    for (int i=n;i>=1;--i) sa[sum[x[i]]--]=i;
    for (int j=1,p=1;p<n;j*=2,m=p)
    {

        p=0;
        for (int i=n-j+1;i<=n;++i) y[++p]=i;
        for (int i=1;i<=n;++i) if (sa[i]>j) y[++p]=sa[i]-j;
        for (int i=0;i<=m;++i) sum[i]=0;
        for (int i=1;i<=n;++i) sum[wv[i]=x[y[i]]]++;
        for (int i=1;i<=m;++i) sum[i]+=sum[i-1];
        for (int i=n;i>=1;--i) sa[sum[wv[i]]--]=y[i];
        for (int i=1;i<=n;++i) y[i]=x[i];
        p=x[sa[1]]=1;
        for (int i=2;i<=n;++i)
        {
            ++p;
            if (y[sa[i]]==y[sa[i-1]]&&y[sa[i]+j]==y[sa[i-1]+j]) --p;
            x[sa[i]]=p;
        }
    }
}
class Solution {
public:
    
    string lastSubstring(string t) {
        len = t.size();
        for(int i=1;i<=len;i++)
        {
            s[i] = t[i-1];
        }
        build_sa();
        string ans = t.substr(sa[len]-1);
        return ans;
        
    }
};
