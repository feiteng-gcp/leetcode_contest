class Solution {
    public:
        string lastSubstring(string s) {
            int n = s.length();
            int i = 0, j = 1, k = 0;
            while (i < n && j < n && k < n) {
                auto t = s[(i + k) % n] - s[(j + k) % n];
                if (!t) k++;
                else {
                    if (t > 0) j += k + 1;
                    else i += k + 1;
                    if (i == j) j++;
                    k = 0;
                }
            }
            int ans = min(i, j);
            return s.substr(ans, n - ans);
        }
};
