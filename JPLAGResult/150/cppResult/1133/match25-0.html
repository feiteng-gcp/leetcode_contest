class Solution {
public:
    string lastSubstring(string s) {
        int n = s.length();
        int i = 1;
        int ans_l = 0;
        int ans_r = 0;
        int now_l = 1;
        while (i < n)
        {
            if (s[i] > s[ans_l])
            {
                ans_l = i;
                now_l = i+1;
            }
            else if (s[i] > s[ans_l + i - now_l])
            {
                ans_l = now_l;
                now_l = i + 1;
            }
            else if (s[i] < s[ans_l + i - now_l])
            {
                now_l = i + 1;
            }
            ++i;
        }
        string ans_str = s.substr(ans_l, n - ans_l);
        return ans_str;
    }
};
