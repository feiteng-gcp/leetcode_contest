class Solution {
public:
    string lastSubstring(string s) {
        int n = s.size();
        if (n == 0) return "";
        
        int cur_cmp = 0, beg = 0;

        for (int i = 1; i < n; ++i) {
            if (s[i] > s[beg]) {
                beg = i;
                cur_cmp = beg;
            }
            else if (s[i] > s[cur_cmp]) {
                    beg = i - (cur_cmp - beg);
                    cur_cmp = beg;
            }
            else if (s[i] == s[cur_cmp]) {
                    cur_cmp++;
            }
            else {
                    cur_cmp = beg;
            }
            // if (s[i] == s[beg]) {
            //     cur_cmp = beg + 1;
            // }
            // while (cur_cmp < i && s[i] == s[cur_cmp])
            //     ++cur_cmp;
        }
        
        return s.substr(beg, s.size() - beg);
    }
};
