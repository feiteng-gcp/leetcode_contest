class Solution {
    int getmin(const string& s){
        int n = s.length();
        int i = 0, j = 1, k = 0, t;
        while (i < n && j < n && k < n) {

            t = s[(i + k) % n] - s[(j + k) % n];
            if (!t)
                k++;
            else{
                if (t < 0)
                    i+=k+1; 
                else
                    j+=k+1;
                if (i==j)
                    j++;
                k = 0;
            }
        }
        return i < j ? i : j;
    }
public:
    string lastSubstring(string s) {
        int ind = getmin(s);
        return s.substr(ind);
    }
};
