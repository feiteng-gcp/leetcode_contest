const int maxn = 100005;

int s[maxn];  
  int sa[maxn];	
  int t[maxn], t2[maxn], c[maxn]; 
  int n = 0; 							

  void build_sa(int m) {
      memset(sa, 0, sizeof sa);
      memset(t, 0, sizeof t);
      memset(t2, 0, sizeof t2);
      memset(c, 0, sizeof c);


    int i, *x = t, *y = t2;
    for (i = 0; i < m; i++) c[i] = 0;
    for (i = 0; i < n; i++) c[x[i] = s[i]]++;
    for (i = 1; i < m; i++) c[i] += c[i-1];
    for (i = n-1; i >= 0; i--) sa[--c[x[i]]] = i;
    for (int k = 1; k <= n; k <<= 1) {
      int p = 0;
      for (i = n-k; i < n; i++) y[p++] = i;
      for (i = 0; i < n; i++) if (sa[i] >= k) y[p++] = sa[i]-k;
      for (i = 0; i < m; i++) c[i] = 0;
      for (i = 0; i < n; i++) c[x[y[i]]]++;
      for (i = 1; i < m; i++) c[i] += c[i-1];
      for (i = n-1; i >= 0; i--) sa[--c[x[y[i]]]] = y[i];
      swap(x, y);
      p = 1; x[sa[0]] = 0;
      for (i = 1; i < n; i++)
        x[sa[i]] = y[sa[i-1]]==y[sa[i]] && y[sa[i-1]+k]==y[sa[i]+k] 
? p-1 : p++;
      if (p >= n) break;
      m = p;
    }
  }



class Solution {
public:
    string lastSubstring(string s) {
        n = s.size();
        for (int i = 0; i < n; ++i) ::s[i] = s[i] - 'a' + 1;
        ::s[n] = 0;
        n++;
        build_sa(27);
        //for (int i = 0; i < n; ++i) printf("%d %d\n", i, sa[i]);
        return s.substr(sa[n-1]);
    }
};
