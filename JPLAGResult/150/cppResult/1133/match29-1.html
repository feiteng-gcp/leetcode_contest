class Solution {
public:
  string lastSubstring(string s) {
    if(s.size() == 1) return s;
    int n = s.size();
    s = s + "^" + s;
    int i = 0, j = 1;
    int k = 0;
    while (i < n && j < n && k < n) {

      if (s[i + k] == s[j + k]) k++;
      else {
        if (s[i + k] < s[j + k])  i = i + k + 1;
        else j = j + k + 1;
        if (i == j) i++;
        k = 0;
      }
    }
    return s.substr(0, n).substr(min(i, j));
  }
};
