class Solution {
    void Sort(int *in,int *out,int *p,int *P,int n,int m)
    {
    	int i;
    	for(i=1;i<=m;i++)P[i]=0;
    	for(i=1;i<=n;i++)P[in[i]]++;
    	for(i=2;i<=m;i++)P[i]+=P[i-1];
    	for(i=n;i;i--)out[P[in[p[i]]]--]=p[i];
    }
    char c[100005];
    int n,i,j,k,l,m,r,mid,N,K,a[100005],a2[100005],p[100005],P[100005],t1[100005*3],t2[100005*3],s[100005],R[100005],h[100005],*x=t1,*y=t2;
public:
    string lastSubstring(string S) {
        n=S.size();
        for(i=0;i<n;i++)c[i+1]=S[i];
	    for(i=1;i<=n;i++)x[i]=a[i]=c[i]-'a'+1;
	    N=26;
	    for(i=1;i<=n;i++)p[i]=i;
	    Sort(a,s,p,P,n,N);
	    for(j=1;k<n;N=k,j<<=1)
	    {
	    	for(k=0,i=n-j+1;i<=n;i++)y[++k]=i;
	    	for(i=1;i<=n;i++)if(s[i]>j)y[++k]=s[i]-j;
	    	Sort(x,s,y,P,n,N);
	    	for(swap(x,y),k=x[s[1]]=1,i=2;i<=n;i++)x[s[i]]=y[s[i-1]]==y[s[i]]&&y[s[i-1]+j]==y[s[i]+j]?k:++k;
    	}
        string ans;
        for(i=s[n];i<=n;i++)ans+=c[i];
        return ans;
    }
};
