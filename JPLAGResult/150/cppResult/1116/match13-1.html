/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        if(!root)
            return 0;
        queue<TreeNode*> todo;
        int ans = 0;
        int level = 0;

        int subsum = INT_MIN;
        todo.push(root);
        while(todo.size()){
            level++;
            int m = todo.size();
            int temp = 0;
            for(int i=0; i<m; i++){
                root = todo.front();
                todo.pop();
                temp += root->val;

                if(root->left)
                    todo.push(root->left);
                if(root->right)
                    todo.push(root->right);
            }
            if(temp > subsum){
                ans = level;
                subsum = temp;
            }
        }
        return ans;
    }
};
