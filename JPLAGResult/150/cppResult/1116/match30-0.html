/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        queue<pair<TreeNode*, int>> q;
        q.emplace(root, 1);
        vector<int> levelSum;
        levelSum.emplace_back(0);
        int preLevel = 1, sum = 0;

        while(!q.empty()) {
            pair<TreeNode*, int> cur = q.front();
            q.pop();

            if(cur.second != preLevel) {
                levelSum.emplace_back(sum);
                sum = cur.first->val;
                preLevel = cur.second;
            }
            else {
                sum += cur.first->val;                
            }
            
            if(cur.first->left) {
                q.emplace(cur.first->left, cur.second + 1);
            }
            
            if(cur.first->right) {
                q.emplace(cur.first->right, cur.second + 1);
            }
        }

        levelSum.emplace_back(sum);
        int res = 1;

        for(int i = 1; i < levelSum.size(); ++i) {
            if(levelSum[i] > levelSum[res]) {
                res = i;
            }
        }

        return res;
    }
};
