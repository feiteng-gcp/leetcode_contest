/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> v;
    static const int INF = 1e9+5;
    int cnt = 0;
    int tot = 1;
    int maxLevelSum(TreeNode* root) {

        cnt = -INF;
        queue<TreeNode* > Q;
        Q.push(root);
        while(!Q.empty())
        {
            int tmp = 0;
            int len = Q.size();
            //cout<<len<<endl;
            for(int i=0;i<len;i++)
            {

                TreeNode *x = Q.front();
                Q.pop();
                tmp += x->val;
                if(x->left)
                {
                    Q.push(x->left);
                }
                if(x->right)
                {
                    Q.push(x->right);
                }
            }
            if(tmp > cnt)
            {
                v.clear();
                cnt = tmp;
                v.push_back(tot);
            }
            tot++;
        }
        return v[0];
    }
};
