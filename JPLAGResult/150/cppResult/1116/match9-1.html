/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
   int maxLevelSum(TreeNode* root) {
    queue<TreeNode*> q;
    q.push(root);
    int r = 0, rc = 0, cnt = 0;
    while(!q.empty()) {
    	int n = q.size();

    	++cnt;
    	int v = 0;
    	for (int i = 0; i < n; i++) {
    		auto* u = q.front(); q.pop();
    		v += u->val;
    		if(u->left) q.push(u->left);
    		if(u->right) q.push(u->right);
    	}
    	if(v > rc) {
    		rc = v;
    		r = cnt;
    	}
    }
    return r;
}
};
