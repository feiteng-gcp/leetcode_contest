/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        
        queue<TreeNode*> queue;
        queue.push(root);
        
        int maxval = 0, minlevel = -1;
        int level = 1;
        
        while (!queue.empty())
        {
            int sz = queue.size();
            int levelsum = 0;
            while (sz-- > 0)
            {
                TreeNode* node = queue.front(); queue.pop();
                levelsum += node->val;
                if (node->left) queue.push(node->left);
                if (node->right) queue.push(node->right);
            }
            if (levelsum > maxval)
            {
                maxval = levelsum;
                minlevel = level;
            }
            ++level;
        }
        
        return minlevel;
    }
};
