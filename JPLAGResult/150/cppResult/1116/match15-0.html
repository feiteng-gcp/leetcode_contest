/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        queue<TreeNode*> q;
        q.push(root);
        int ans = INT_MIN;
        int lev = 1, res = 1;
        while (!q.empty()) {
            int sz = q.size();
            int cnt = 0;
            while (sz--) {
                TreeNode* now = q.front();
                q.pop();
                cnt+=now->val;
                if (now->left != nullptr) q.push(now->left);
                if (now->right != nullptr) q.push(now->right);
            }
            if (cnt > ans) {
                ans = cnt;
                res = lev;
            }
            lev++;
        }
        return res;
    }
};
