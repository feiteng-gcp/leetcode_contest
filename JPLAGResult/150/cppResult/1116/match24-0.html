/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        int result = 1;
        int level = 1;
        int max_sum = root->val;
        queue<TreeNode*> Q;
        Q.push(root);
        while (!Q.empty()) {
            int sum = 0;
            const int size = Q.size();
            for (int i = 0; i < size; ++i) {
                TreeNode* q = Q.front();
                Q.pop();
                sum += q->val;
                if (q->left) Q.push(q->left);
                if (q->right) Q.push(q->right);
            }
            if (sum > max_sum) {
                max_sum = sum;
                result = level;
            }
            level++;
        };
        return result;
    }
};
