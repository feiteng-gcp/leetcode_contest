/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        queue<TreeNode*> q;
        q.push(root);
        int r = 0, mx = 0;
        for (int k = 1; q.size(); k++) {
            int n = q.size(), sum = 0;
            for (int i = 0; i < n; i++) {
                auto p = q.front();
                q.pop();
                sum += p->val;
                if (p->left) q.push(p->left);
                if (p->right) q.push(p->right);
            }
            if (sum > mx) {
                mx = sum;
                r = k;
            }
        }
        return r;
    }
};
