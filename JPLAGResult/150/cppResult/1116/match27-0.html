/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
 public:
  int maxLevelSum(TreeNode* root) {
    queue<TreeNode*> level;
    level.push(root);
    int max_level = 1;
    int max_sum = root->val;
    int cur_level = 1;
    while(!level.empty()) {
      int n = level.size();
      int level_sum = 0;
      for (int i = 0; i < n; i++) {
        auto p = level.front();
        level.pop();
        level_sum += p->val;
        if (p->left != nullptr) {
          level.push(p->left);
        }
        if (p->right != nullptr) {
          level.push(p->right);
        }
      }
      if (level_sum > max_sum) {
        max_sum = level_sum;
        max_level = cur_level;
      }
      cur_level++;
    }
    return max_level;
  }
};
