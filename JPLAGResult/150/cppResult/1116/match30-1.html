/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        queue<pair<TreeNode*, int> > q;
        q.push({root, 1});
        vector<int> v(10010, 0);
        while (!q.empty()) {
            TreeNode* at = q.front().first;
            int lvl = q.front().second;
            q.pop();
            v[lvl] += at->val;

            if (at->left) q.push({at->left, lvl+1});
            if (at->right) q.push({at->right, lvl+1});
        }
        int mx = 1;
        for (int i = 1; i <= 10000; i++) {
            if (v[i] > v[mx]) {
                mx = i;
            }
        }
        
        return mx;
    }
};
