/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        int res = 0, ans = INT_MIN;
        queue<TreeNode*> q;
        q.push(root);
        int level = 1;
        while(!q.empty()) {
            int t = q.size();
            int sum = 0;
            while(t--) {
                auto node=q.front();
                q.pop();
                sum += node->val;
                if(node->left) q.push(node->left);
                if(node->right) q.push(node->right);
            }
            if(sum>ans) res=level, ans=sum;
            level++;
        }
        return res;
    }
};
