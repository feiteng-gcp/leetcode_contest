/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        queue<pair<TreeNode*, int>> qq;
        vector<int> v;
        qq.emplace(root, 1);
        while (!qq.empty()) {

            auto q = qq.front();
            qq.pop();
            if (q.second > v.size()) {
                v.push_back(0);
            }
            v[v.size() - 1] += q.first->val;
            if (q.first->left) {
                qq.emplace(q.first->left, q.second + 1);
            }
            if (q.first->right) {
                qq.emplace(q.first->right, q.second + 1);
            }
        }
        int ma = INT_MIN;

        int res = 1;
        for (int i = 0; i < v.size(); ++i) {
            if (v[i] > ma) {
                ma = v[i];
                res = i + 1;
            }
        }
        return res;
    }
};
