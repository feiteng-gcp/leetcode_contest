/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        if (!root) return 0;
        
        vector<int> levels;
        
        queue<TreeNode*> q;
        q.push(root);
        
        int max_level_sum = INT_MIN;
        
        while (!q.empty()) {
            int size = q.size();
            int level_sum = 0;
            for (int i=0; i<size; i++) {
                TreeNode* out = q.front(); q.pop();
                level_sum += out->val;
                if (out->left) q.push(out->left);
                if (out->right) q.push(out->right);
            }
            max_level_sum = max(level_sum, max_level_sum);
            levels.push_back(level_sum);
        }
        
        for (int i=0; i<levels.size(); i++) {
            if (levels[i] == max_level_sum) {
                return i+1;
            }
        }
        return 0;
    }
};
