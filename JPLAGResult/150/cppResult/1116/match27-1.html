/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    
public:
    int maxLevelSum(TreeNode* root) {
        int res = INT_MIN;
        if (root == NULL)
            return 0;
        queue<TreeNode *> q;
        q.push(root);

        int level = 1, max_level = 1;
        
        while(!q.empty()) {
            int sz = q.size();
            int sum_ = 0;
            for(int i = 0; i < sz; ++ i) {
                TreeNode * t = q.front();
                q.pop();
                sum_ += t -> val;
                if (t -> left != NULL) {
                    q.push(t -> left);
                }

                if (t -> right != NULL) {
                    q.push(t -> right);
                }
            }
            if (res < sum_) {
                res = sum_;
                max_level = level;
            }
            level ++;
        }
        return max_level;
    }
};
