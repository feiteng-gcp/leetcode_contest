/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        vector<int> sum;
        dfs(root, 0, sum);
        
        int ans = 0;
        for (int i = 1; i < sum.size(); ++i)
            if (sum[i] > sum[ans])
                ans = i;
        
        return ans+1;
    }
    
    void dfs(TreeNode *cur, int h, vector<int> &sum) {
        if (!cur)
            return;
        
        if (h >= (int)sum.size())
            sum.push_back(0);
        
        sum[h] += cur->val;
        dfs(cur->left, h+1, sum);
        dfs(cur->right, h+1, sum);
    }
};
