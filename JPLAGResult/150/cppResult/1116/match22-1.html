/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        queue<TreeNode*> nodes;
        nodes.push(root);
        int maxS = INT_MIN;
        int maxL = 1;
        int level = 1;
        while(!nodes.empty()){
            int sum = 0;
            for(int i=nodes.size(); i>0; i--){
                TreeNode* n = nodes.front();
                nodes.pop();
                if(n->left!=nullptr)
                    nodes.push(n->left);
                if(n->right!=nullptr)
                    nodes.push(n->right);

                sum += n->val;
            }
            if(sum>maxS){
                maxS = sum;
                maxL = level;
            }
            level++;
        }
        return maxL;
    }
};
