/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        queue<TreeNode*> que;
        que.push(root);
        int lvl = 1;
        int res = -1;
        int maxSum = INT_MIN;
        while(!que.empty())
        {
            int s = que.size();
            int curSum = 0;
            while(s)
            {
                s--;
                auto cur = que.front();
                que.pop();
                if(cur->left)
                {
                    que.push(cur->left);
                }
                if(cur->right)
                {
                    que.push(cur->right);
                }
                curSum+=cur->val;
            }
            //cout<<lvl<<" "<<curSum<<endl;
            if(curSum>maxSum)
            {
                res = lvl;
                maxSum = curSum;
            }
            lvl++;
            
        }
        return res;
    }
};
