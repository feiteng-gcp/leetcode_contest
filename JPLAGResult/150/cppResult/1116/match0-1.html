/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        if ( !root ) return 0;
        queue<TreeNode*> q;
        q.push(root);
        int res=INT_MIN;
        int level=0,step=0;
        while( !q.empty() ){
            int len = q.size();
            int sums=0;
            for ( int t=0; t<len; ++t ){
                TreeNode * cur = q.front(); q.pop();
                sums += cur->val;
                if ( cur->left ) q.push(cur->left);
                if (cur->right) q.push(cur->right);
            }
            if ( res<sums ){
                res = sums;
                level = step+1;
            }
            ++step;
        }
        return level;
    }
};
