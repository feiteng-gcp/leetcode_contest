/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    unordered_map<int,int> sum_level;
    void DFS(TreeNode* root,int level)
    {
        if(!root) return;
        sum_level[level] += root->val;
        DFS(root->left,level + 1);
        DFS(root->right,level + 1);
    }
    int maxLevelSum(TreeNode* root) {
        DFS(root,0);
        int max_value = -2e9;
        int level = 0;
        for(unordered_map<int,int>::iterator it = sum_level.begin();it != sum_level.end();it++)
        {
            if(it->second > max_value)
            {
                max_value = it->second;
                level = it->first;
            }
        }
        return level+1;
    }
};
