/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    map<int, int> m;
    void dfs(TreeNode*node, int level) {
        if (node == NULL) return;
        m[level] += node->val;
        dfs(node->left, level + 1);
        dfs(node->right, level + 1);
    }
    
    int maxLevelSum(TreeNode* root) {
        m.clear();
        dfs(root, 1);
        int maxV = m[1];
        int res = 1;
        
        map<int, int>::iterator I;
        for (I = m.begin(); I != m.end(); I++) {
            int level = I->first;
            int val = I->second;

            if (val > maxV) {
                maxV = val;
                res = level;
            } else if (val == maxV && level < res) {
                res = level;
            }
        }
        return res;
        
    }
};
