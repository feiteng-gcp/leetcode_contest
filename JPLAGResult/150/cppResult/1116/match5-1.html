/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
  int maxLevelSum(TreeNode* root) {
    vector<TreeNode*> cur{root};
    int level = 1;
    int best = -0x3f3f3f3f;
    int best_i = -1;
    while (cur.size()) {
      vector<TreeNode*> next;
      int tot = 0;
      for (auto s: cur) {
        tot += s->val;
        if (s->left != NULL) next.push_back(s->left);
        if (s->right != NULL) next.push_back(s->right);
      }
      if (best < tot) best = tot, best_i = level;
      level++;
      cur = next;
    }
    return best_i;
  }
};
