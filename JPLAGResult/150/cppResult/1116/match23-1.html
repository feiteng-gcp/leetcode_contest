/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:

    void solve(TreeNode* p, int dep, vector<int>& A) {
        if (!p) return;
        if (A.size() == dep) {
            A.push_back(p->val);
        } else {
            A[dep] += p->val;
        }
        solve(p->left, dep + 1, A);
        solve(p->right, dep + 1, A);
    }
    int maxLevelSum(TreeNode* root) {
        vector<int> A;
        solve(root, 0, A);
        //for (auto& it : A) cout << it << endl;

        int best = -2e9, ret = -1;
        for (int k = 0; k < A.size(); ++k) {
            if (best < A[k]) {
                best = A[k];
                ret = k;
            }
        }
        return ret + 1;
    }
};
