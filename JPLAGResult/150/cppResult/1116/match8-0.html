/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        if (!root) return 0;
        
        int maxsum = INT_MIN, rtn = 0, level = 0;
        queue<TreeNode *> Q;
        Q.push(root);
        while (!Q.empty()) {
            ++level;
            int n = Q.size();
            int sum = 0;
            for (int i = 0 ; i < n ; ++i) {
                auto node = Q.front();
                Q.pop();
                sum += node->val;
                if (node->left) Q.push(node->left);
                if (node->right) Q.push(node->right);
            }
            if (sum > maxsum) {
                maxsum = sum;
                rtn = level;
            }
        }
        return rtn;
    }
};
