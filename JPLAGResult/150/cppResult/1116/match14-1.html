/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
	int maxLevelSum(TreeNode* root) {
		if (!root) return -1;
		queue<TreeNode*> Q;
		vector<int> res;
		res.push_back(root->val);
		int m = res[0];
		Q.push(root);
		while (!Q.empty()) {
			int size = Q.size();
			int sum = 0;
			for (int i = 0; i < size; i++) {
			   //孩子如果有，计算和
				if (Q.front()->left) {
					sum += Q.front()->left->val;
					Q.push(Q.front()->left);
				}
				if (Q.front()->right) {
					sum += Q.front()->right->val;
					Q.push(Q.front()->right);
				}
				Q.pop();

			}
			m = max(m, sum);
			res.push_back(sum);
		}

		for (int i = 0; i < res.size(); i++) {
			if (res[i] == m) {
				return i + 1;
			}
		}
		return 0;
	}
};
