/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        vector<int> sums;
        dfs(root, 1, sums);
        int maxSum = 0;
        int maxSumLevel = 0;
        for (int i = 0; i < sums.size(); ++i)
        {
            if (sums[i] > maxSum)
            {
                maxSum = sums[i];
                maxSumLevel = i + 1;
            }
        }
        return maxSumLevel;
    }
    
    void dfs(const TreeNode* root, const int level, vector<int>& sums)
    {
        if (!root) return;
        if (level > sums.size())
        {
            sums.push_back(root->val);
        }
        else
        {
            sums[level - 1] += root->val;
        }
        dfs(root->left, level + 1, sums);
        dfs(root->right, level + 1, sums);
    }
};
