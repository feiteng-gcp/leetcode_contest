class Solution {
public:
    int c[4010];
    void dfs(TreeNode *root, int level){
        c[level] += root->val;
        if(root->left == NULL && root->right == NULL) return;
        if(root->left != NULL){
            dfs(root->left, level+1);
        }
        if(root->right != NULL){
            dfs(root->right, level+1);
        }
    }
    int maxLevelSum(TreeNode* root) {
        memset(c, 0, sizeof(c));
        dfs(root, 0);
        int pos = 0, maxx = c[0];
        for(int i=1; i<4010; i++){
            if(c[i] > maxx){
                maxx = c[i];
                pos = i;
            }
        }
        return pos+1;
    }
};
