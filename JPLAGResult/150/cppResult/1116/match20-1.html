/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        vector<int> c;
        cc(root, c, 0);
        
        int m = 0;
        for (int i = 1; i < c.size(); ++i)
            if (c[i] > c[m]) m = i;
        
        return m+1;
    }
    
    void cc(TreeNode* r, vector<int> &c, int d) {
        if (c.size() == d) {
            c.push_back(0);
        }
        if (r == NULL) return ;
        
        c[d]+= r->val;
        cc(r->left, c, d+1);
        cc(r->right, c, d+1);
    }
};
