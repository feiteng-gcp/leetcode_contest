/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int maxLevelSum(TreeNode* root) {
        queue<TreeNode*> q;
        q.push(root);
        int level = 0;
        int maxv = INT_MIN;
        int ans = 0;
        while(!q.empty()) {
            ++level;
            int size = q.size();
            int sum = 0;
            while(size--) {
                TreeNode* t = q.front(); q.pop();
                sum += t->val;
                if(t->left) q.push(t->left);
                if(t->right) q.push(t->right);
            };
            if(maxv < sum) {
                maxv = sum;
                ans = level;
            };
        };
        return ans;
    }
};
