/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int cnt[11111];

    int maxLevelSum(TreeNode* root) {
        memset(cnt,0,sizeof(cnt));
        dfs(root,1);
        int maxx=-100000,maxh=0;
        for(int i=1;i<10000;i++){
            if(cnt[i]>maxx){
                maxx=cnt[i];
                maxh=i;
            }
        }return maxh;
    }
    void dfs(TreeNode* now,int hh){
        if(now==NULL)return;
        cnt[hh]+=now->val;
        dfs(now->left,hh+1);
        dfs(now->right,hh+1);
    }
};
