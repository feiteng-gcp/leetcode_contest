/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    unordered_map<int,int> m;
    
    void dfs(TreeNode* cur, int depth) {
        if(cur == NULL) return;
        m[depth] += cur->val;
        dfs(cur->left,depth+1);
        dfs(cur->right,depth+1);
    }
    
    int maxLevelSum(TreeNode* root) {
        dfs(root,1);
        int maxSum = INT_MIN;
        int ans = -1;
        for(auto p : m) {
            if(p.second > maxSum) {
                maxSum = p.second;
                ans = p.first;
            }
            else if(p.second == maxSum) {
                ans = min(ans,p.first);
            }
        }
        return ans;
    }
};
