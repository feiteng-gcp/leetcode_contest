class Solution {
public:
    int maxLevelSum(TreeNode *root) {
        if (!root)
            return 0;
        queue<TreeNode *> q;
        q.push(root);
        int lvl = 1, l = 1, res = 0, size = 1;
        while (!q.empty()) {
            int val = 0;
            for (int i = 0; i < size; ++i) {
                auto r = q.front();
                q.pop();
                val += r->val;
                if (r->left)
                    q.push(r->left);
                if (r->right)
                    q.push(r->right);
            }
            if (res < val) {
                lvl = l;
                res = val;
            }
            size = q.size();
            ++l;
        }
        return lvl;
    }
};
