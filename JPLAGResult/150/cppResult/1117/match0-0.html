class Solution {
public:
  int maxDistance(vector<vector<int>>& grid) {
    int dx[] = {0, 0, 1, -1};
    int dy[] = {1, -1, 0, 0};
    int N = grid.size();
    vector<pair<int, int>> cur;
    for (int i = 0; i < N; i++) {
      for (int j = 0; j < N; j++) if (grid[i][j] == 1) cur.push_back({i, j});
    }
    int level = 0;
    int best = -1;
    while (cur.size()) {
      level++;
      vector<pair<int, int>> next;
      for (auto [x,y]: cur) {
        for (int i = 0; i < 4; i++) {
          int nx = x + dx[i];
          int ny = y + dy[i];
          if (0 <= nx && nx < N && 0 <= ny && ny < N && grid[nx][ny] == 0) {
            grid[nx][ny] = level;
            best = level;
            next.push_back({nx, ny});
          }
        }
      }
      cur = next;
    }
    return best;
  }
};
