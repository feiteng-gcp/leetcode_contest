class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        int n = grid.size();
        if (n == 0) return -1;
        int m = grid[0].size();
        if (m == 0) return -1;
        vector<vector<int>> d (n, vector<int> (m, -1));
        deque<pair<int, int> > q;
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j) if (grid[i][j]) {
                d[i][j] = 0;
                q.push_back({i, j});
            }
        int go[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        while (!q.empty()) {
            int x = q.front().first;
            int y = q.front().second;
            q.pop_front();
            for (int i = 0; i < 4; ++i) {
                int tx = x + go[i][0], ty = y + go[i][1];
                if (~tx && ~ty && tx < n && ty < m && !~d[tx][ty]) {
                    d[tx][ty] = d[x][y] + 1;
                    q.push_back({tx, ty});
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j) {
                ans = max(ans, d[i][j]);
            }
        return ans ? ans : -1;
    }
};
