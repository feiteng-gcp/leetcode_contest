class Solution {
public:
	int dy[8] = { 0 ,-1,1,0};
	int dx[8] = { -1,0,0,1 };
	int maxDistance(vector<vector<int>>& grid) {
		int zeros = 0;
		for (int i = 0; i < grid.size(); i++) {
			for (int j = 0; j < grid[0].size(); j++) {
				if (grid[i][j] == 0) zeros++;
			}
		}
		if (zeros == 0 || zeros == grid.size()*grid[0].size()) return -1;
		int res = 0;
		while (zeros > 0) {
			res++;
			int s = solve(grid);
			zeros -= s;
		}
		return res;
	}
	int solve(vector<vector<int>>& grid) {
		for (int i = 0; i < grid.size(); i++) {
			for (int j = 0; j < grid[0].size(); j++) {
				if (grid[i][j] == 1) {
					for (int k = 0; k < 4; k++) {
						int x = i + dx[k];
						int y = j + dy[k];
						if (x >= 0 && x < grid.size() && y >= 0 && y < grid[0].size()&&grid[x][y]!=1) {
							grid[x][y] = 2;
						}
					}
				}
			}
		}
		int res = 0;
		for (int i = 0; i < grid.size(); i++) {
			for (int j = 0; j < grid[0].size(); j++) {
				if (grid[i][j] == 2) {
					res++;
					grid[i][j] = 1;
				}
			}
		}
		return res;
	}
};
