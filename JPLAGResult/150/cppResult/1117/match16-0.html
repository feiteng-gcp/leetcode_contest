class Solution {
public:
    int dm[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    int maxDistance(vector<vector<int>>& grid) {
        queue<pair<int,int> > myQue;
        vector<vector<int> > disArray(grid.size(),vector<int>(grid.size(),INT_MAX));
        for(int i = 0;i < grid.size();i++)
        {
            for(int j = 0;j < grid.size();j++)
            {
                if(grid[i][j] == 1)
                {
                    myQue.push(make_pair(i,j));
                    disArray[i][j] = 0;
                }
            }
        }
        int mx_dis = -1;
        while(!myQue.empty())
        {
            pair<int,int> top_cell = myQue.front();
            myQue.pop();
            for(int i = 0;i < 4;i++)
            {
                int new_x = top_cell.first + dm[i][0];
                int new_y = top_cell.second + dm[i][1];
                if(new_x >= 0 && new_y >= 0 && new_x < grid.size() && new_y < grid.size() && grid[new_x][new_y] == 0 && disArray[new_x][new_y] == INT_MAX)
                    disArray[new_x][new_y] = 1 + disArray[top_cell.first][top_cell.second],myQue.push(make_pair(new_x,new_y)),mx_dis = max(mx_dis,1 + disArray[top_cell.first][top_cell.second]);
            }
        }
        return mx_dis;
    }
};
