const int INF = 1000000000;

int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

class Solution {
public:
    int maxDistance(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        int sum = 0;
        for (int i = 0; i < n; ++ i)
            for (int j = 0; j < m; ++ j)

                sum += a[i][j];
        if (sum == 0 || sum == n*m) return -1;
        vector<vector<int>> d(n, vector<int>(m, INF));
        queue<pair<int, int>> q;
        for (int i = 0; i < n; ++ i)
            for (int j = 0; j < m; ++ j)
                if (a[i][j])
                {
                    d[i][j] = 0;
                    q.push({i, j});
                }
        while (!q.empty())
        {

            auto [x, y] = q.front();
            q.pop();
            for (int k = 0; k < 4; ++ k)
            {
                int tx = x+dx[k], ty = y+dy[k];
                if (0 <= tx && tx < n && 0 <= ty && ty < m && d[tx][ty] > d[x][y]+1)
                {
                    d[tx][ty] = d[x][y]+1;
                    q.push({tx, ty});
                }
            }
        }
        int ret = 0;
        for (int i = 0; i < n; ++ i)
            for (int j = 0; j < m; ++ j)
                if (!a[i][j]) ret = max(ret, d[i][j]);
        return ret;
    }
};
