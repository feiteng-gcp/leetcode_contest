int d[105][105];
int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, 1, -1};

class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        queue<pair<int, int> > Q;
        
        int n = grid.size();
        memset(d, -1, sizeof d);
        

        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == 1) {
                    d[i][j] = 0;
                    Q.push(make_pair(i,j));
                }
            }

        }
        
        while (!Q.empty()) {
            int x = Q.front().first;
            int y = Q.front().second;
            Q.pop();
            
            printf("---\n%d %d %d \n", x, y, d[x][y]);
            for (int i = 0; i < 4; ++i) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                
                if (nx < 0 || ny < 0 || nx >= n || ny >= n) continue;
                if (d[nx][ny] == -1) {
                    
                    d[nx][ny] = d[x][y] + 1;
                    printf("%d %d %d \n", nx, ny, d[nx][ny]);
                    Q.push(make_pair(nx,ny));
                }
            }
        }
        

        int ans = -1;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                ans = max(ans, d[i][j]);
            }
        }
        if (ans == 0) ans = -1;
        return ans;
        
    }
};
