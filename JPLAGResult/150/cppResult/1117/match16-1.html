class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();

        queue<pair<int, int>> que;
        set<pair<int, int>> used;
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(grid[i][j])
                {
                    que.push({i, j});
                    used.insert({i, j});
                }
            }
        }
        if(que.size()==0||que.size()==m*n) return -1;
        int lvl = -1;

        int help[4][2]={{-1, 0}, {1, 0},{0,1},{0,-1}};
        while(!que.empty())
        {
            int s = que.size();
            while(s)
            {
                s--;
                auto cur = que.front();
                que.pop();
                int x = cur.first;

                int y = cur.second;
                for(int i=0;i<4;i++)
                {
                    int nx = x+help[i][0];
                    int ny = y+help[i][1];
                    if(nx>=0&&nx<m&&ny>=0&&ny<n&&used.find({nx, ny})==used.end())
                    {
                        used.insert({nx,ny});
                        que.push({nx, ny});
                    }
                }
                
            }
            lvl++;
        }
        return lvl;
    }
};
