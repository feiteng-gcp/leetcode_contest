class Solution {
public:
    int dx[4] = {1, 0, -1, 0};
    int dy[4] = {0, 1, 0, -1};
    
    int maxDistance(vector<vector<int>>& grid) {
        if (grid.size() == 0) return 0;
        int m = grid.size();
        int n = grid[0].size();

        vector<pair<int,int>> waiting;
        int opt[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                opt[i][j] = 2147483647;
                if (grid[i][j] == 1) {
                    waiting.push_back(make_pair(i, j));
                    opt[i][j] = 0;
                }
            }
        }
        int head = 0;
        while (head < waiting.size()) {
            int x = waiting[head].first;
            int y = waiting[head].second;
            for (int i = 0; i < 4; i++) {
                int newx = x + dx[i];
                int newy = y + dy[i];
                if (newx >= 0 && newx < m && newy >= 0 && newy < n) {
                    if (grid[newx][newy] == 0 && opt[newx][newy] > opt[x][y] + 1) {
                        opt[newx][newy] = opt[x][y] + 1;
                        waiting.push_back(make_pair(newx, newy));
                    }
                }
            }
            head ++;
        }
        

        int min = -1;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 0 && opt[i][j] > min) {
                    min = opt[i][j];
                }
            }
        }
        if (min == 2147483647 || min == 0) return -1; 
        return min;
    }
};
