class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        int n = grid.size();

        auto tmp = vector<vector<int>>(n, vector<int>(n, -1));
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(grid[i][j]==1) tmp[i][j]=0;

            }
        }
        // grid = tmp;

        for(int t=0;t<2*n;t++){
            grid=tmp;
            for(int i=0;i<n;i++){
                for(int j=0;j<n;j++){
                    //if(grid[i][j]==1) continue;
                    if(tmp[i][j]>=0) continue;
                    if(i>0 && tmp[i-1][j]>=0) grid[i][j]=tmp[i-1][j] + 1;
                    else if(i<n-1 && tmp[i+1][j]>=0) grid[i][j]=tmp[i+1][j] + 1;
                    else if(j>0 && tmp[i][j-1]>=0) grid[i][j]=tmp[i][j-1] + 1;
                    else if(j<n-1 && tmp[i][j+1]>=0) grid[i][j]=tmp[i][j+1] + 1;

                }
            }
            tmp=grid;
            //  for(int i=0;i<n;i++){
            //     for(int j=0;j<n;j++){
            //         cout<<tmp[i][j]<<' ';
            //         // if(tmp[i][j] > 0) res=max(res,tmp[i][j]);
            //     }
            //     cout<<endl;
            // }
        }
        int res = -1;
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                // cout<<tmp[i][j]<<' ';
                if(tmp[i][j] > 0) res=max(res,tmp[i][j]);
            }
            // cout<<endl;
        }
        return res;
    }
};
