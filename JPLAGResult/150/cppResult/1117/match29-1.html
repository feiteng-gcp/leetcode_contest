class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        queue<pair<int, int>> q;
        int n = grid.size();
        int m = grid[0].size();

        

        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (grid[i][j] == 1) {
                    q.emplace(i, j);
                    grid[i][j] = 0;
                } else {
                    grid[i][j] = -1;
                }
            }
        }
        
        int ans = -1;
        while (!q.empty()) {

            int x, y;
            tie(x, y) = q.front();
            q.pop();
            ans = max(ans, grid[x][y]);
            int dx[] = {0,1,0,-1};
            int dy[] = {-1,0,1,0};
            
            for (int i = 0; i < 4; ++i) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                
                if (nx < 0 || ny < 0) continue;
                if (nx == n || ny == m) continue;
                if (grid[nx][ny] != -1) continue;
                grid[nx][ny] = grid[x][y] + 1;
                q.emplace(nx,ny);
            }
        }
        
        return ans ? ans : -1;
    }
    
   
};
