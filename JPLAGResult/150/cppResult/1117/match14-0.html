  class Solution {
  public:
	  int dx[4] = {1,-1,0,0};
	  int dy[4] = {0,0,1,-1};
	  int maxDistance(vector<vector<int>>& grid) {
		  int n = grid.size(), m = grid[0].size();
		  vector<vector<int>>maxx(n, vector<int>(m, 500));
		  for (int i = 0; i < n; i++) {
			  for (int j = 0; j < m; j++) {
				  if (grid[i][j] == 1)maxx[i][j] = 0;
			  }
		  }
		  for (int i = 0; i < n; i++) {
			  for (int j = 0; j < m; j++) {
				  if (grid[i][j] == 1) {
					  queue<pair<int, pair<int, int>>>Q;
					  Q.push({ 0,{i,j} });
					  while (!Q.empty()) {
						  auto t = Q.front(); Q.pop();
						  int x = t.second.first,y=t.second.second;
						  for (int k = 0; k < 4; k++) {
							  int xx = x + dx[k], yy = y + dy[k];
							  if (xx < 0 || xx >= n || yy < 0 || yy >= m)continue;
							  if (t.first + 1 < maxx[xx][yy]) {
								  maxx[xx][yy] = t.first + 1;
								  Q.push({ t.first + 1,{xx,yy} });
							  }
						  }
					  }
				  }
			  }
		  }
		  int ret = 0;
		  for (int i = 0; i < n; i++) {
			  for (int j = 0; j < m; j++) {
				  ret = max(ret, maxx[i][j]);
			  }
		  }
		  if (ret == 0 || ret == 500)return -1;
		  return ret;
	  }
  };
