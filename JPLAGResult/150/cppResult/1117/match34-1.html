class Solution {
private:

    int m, n;
    int directions[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
public:
    int maxDistance(vector<vector<int>>& grid) {
        m = grid.size();
        n = grid[0].size();

        vector<vector<int>> distance(m, vector<int>(n, INT_MAX));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                dfs(0, i, j, grid, distance);

            }
        }
        
        int result = -1;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == 0 && distance[i][j] > result)
                    result = distance[i][j];
            }
        }
        return result == INT_MAX ? -1 : result;
    }
    
    void dfs(int d, int i, int j, vector<vector<int>>& grid, vector<vector<int>>& distance) {
        if (i < 0 || j < 0 || i >= m || j >= n || d >= distance[i][j] || ((d == 0) ^ (grid[i][j] == 1)))
            return;
        
        // cout << i << " " << j << " " << d << endl;
        distance[i][j] = d;
        for (int k = 0; k < 4; ++k)
            dfs(d+1, i+directions[k][0], j+directions[k][1], grid, distance);
    }
};
