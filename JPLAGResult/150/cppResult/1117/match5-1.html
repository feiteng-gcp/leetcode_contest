typedef pair<int,int> pii;
class Solution {
public:
    static const int INF = 1e9+5;
    int a[105][105];
    int vis[105][105];
    int dis[105][105];
    int n;
    int dx[4] = {1,-1,0,0};
    int dy[4] = {0,0,1,-1};
    int maxDistance(vector<vector<int>>& grid) {
        n = grid.size();
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                vis[i+1][j+1] = 0;
                a[i+1][j+1] = grid[i][j];
                dis[i+1][j+1] = INF;
            }
        }
        queue<pii> Q;

        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=n;j++)
            {
                if(a[i][j] == 1)
                {
                    Q.push(pii(i,j));
                    dis[i][j] = 0;
                }
            }
        }
        while(!Q.empty())
        {
            pii u = Q.front();
            Q.pop();

            int x = u.first;
            int y = u.second;
            //printf("%d %d\n",x,y);
            for(int i=0;i<4;i++)
            {
                int tx = x+dx[i];
                int ty = y+dy[i];
                if(tx<1 || tx > n || ty<1 || ty>n)continue;
                if(dis[tx][ty] > dis[x][y] + 1 && a[tx][ty] == 0)
                {
                    dis[tx][ty] = dis[x][y] + 1;
                    //printf("%d %d %d\n",tx,ty,dis[tx][ty]);
                    Q.push({tx,ty});

                }
            }
        }
        int ans = -INF;
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=n;j++)
            {
                if(a[i][j] == 0)
                    ans = max(dis[i][j],ans);
            }
        }
        if(ans == -INF || ans == INF)
            return -1;
        return ans;
    }
};
