class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<vector<int>> dis(n, vector<int> (n, INT_MAX));
        queue<pair<int, int>> q;
        
        vector<int> dx = {-1, 0, 1, 0};
        vector<int> dy = {0, -1, 0, 1};
        
        for(int i=0; i<n; i++) {
            for(int j=0; j<n; j++) {
                if(grid[i][j] == 1) {
                    dis[i][j] = 0;
                    q.push({i, j});
                }
            }
        }
        
        while(!q.empty()) {
            auto node = q.front();
            q.pop();
            int x = node.first, y = node.second;
            
            for(int i=0; i<4; i++) {
                int adjX = x + dx[i], adjY = y + dy[i];
                
                if(adjX >= 0 && adjX < n && adjY >= 0 && adjY < n && dis[adjX][adjY] > dis[x][y] + 1) {
                    dis[adjX][adjY] = dis[x][y] + 1;
                    q.push({adjX, adjY});
                }
            }
        }
        
        int maxDis = -1;
        for(int i=0; i<n; i++) {
            for(int j=0; j<n; j++) {
                if(grid[i][j] == 0) maxDis = max(dis[i][j], maxDis);
            }
        }
        
        if(maxDis == INT_MAX) maxDis = -1;
        
        return maxDis;
    }
};
