class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        int m = grid.size();
        if (m == 0) return -1;
        int n = grid[0].size();
        if (n == 0) return -1;
        vector<vector<int>> dist(m, vector<int>(n, INT_MAX));
        queue<int> toVisit;

        for (int i = 0; i < m; ++i)
        {
            for (int j = 0; j < n; ++j)
            {
                if (grid[i][j] == 1)
                {
                    dist[i][j] = 0;
                    toVisit.push(i * n + j);
                }
            }
        }
        array<int, 8> DIRS = {1, 0, -1, 0, 0, 1, 0, -1};
        while (!toVisit.empty())
        {
            int curr = toVisit.front();
            toVisit.pop();

            int r = curr / n;
            int c = curr % n;
            for (int i = 0; i < 4; ++i)
            {
                int x = r + DIRS[2 * i];
                int y = c + DIRS[2 * i + 1];
                if (x < 0 || x >= m || y < 0 || y >= n) continue;
                if (dist[x][y] <= dist[r][c] + 1) continue;
                dist[x][y] = dist[r][c] + 1;
                toVisit.push(x * n + y);
            }
        }

        int maxDist = -1;
        for (int i = 0; i < m; ++i)
        {
            for (int j = 0; j < n; ++j)
            {
                maxDist = max(maxDist, dist[i][j]);
            }
        }
        if (maxDist == 0 || maxDist == INT_MAX) maxDist = -1;
        return maxDist;
    }
};
