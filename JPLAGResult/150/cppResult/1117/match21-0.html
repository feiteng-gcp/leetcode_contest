class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) 
    {
        const size_t N = grid.size();
        vector<vector<int>> dp(N, vector<int>(N, -1));
        bool updated(true);
        for(int i = 0; i != N; ++i)
        {
            for(int j = 0; j != N; ++j)
            {
                if(grid[i][j] == 1)
                    dp[i][j] = 0;
            }
        }
        
        while(updated)
        {
            updated = false;
            for(int i = 0; i != N; ++i)
            {
                for(int j = 0; j != N; ++j)
                {
                    if(dp[i][j] != -1)
                    {
                        int v = dp[i][j] + 1;
                        if(i > 0 && (dp[i-1][j] == -1 || dp[i-1][j] > v))
                        {
                            dp[i-1][j] = v;
                            updated = true;
                        }
                        if(i < N-1 && (dp[i+1][j] == -1 || dp[i+1][j] > v))
                        {
                            dp[i+1][j] = v;
                            updated = true;
                        }
                        if(j > 0 && (dp[i][j-1] == -1 || dp[i][j-1] > v))
                        {
                            dp[i][j-1] = v;
                            updated = true;
                        }
                        if(j < N-1 && (dp[i][j+1] == -1 || dp[i][j+1] > v))
                        {
                            dp[i][j+1] = v;
                            updated = true;
                        }
                    }  
                }
            }
        }
        int res(-1);
        for(int i = 0; i != N; ++i)
        {
            for(int j = 0; j != N; ++j)
            {
                if(dp[i][j] > res)
                    res = dp[i][j];
            }
        }
        return ((res == 0) ? -1: res);
    }
};
