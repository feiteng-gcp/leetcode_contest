#define pb push_back
#define fst first
#define snd second

typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;

class Solution {
public:
  int maxDistance(vector<vector<int>>& gr) {
    int N=gr.size();
    if(N==0) return -1;
    int M=gr[0].size();
    queue<pair<pii,int>> q;
    int d[111][111];
    memset(d,0x3f,sizeof d);

    for(int i=0;i<N;i++){
      for(int j=0;j<M;j++){
        if(gr[i][j]){
          d[i][j]=0;
          q.push({{i,j},0});
        }
      }
    }
    if(q.empty()) return -1;
    while(!q.empty()){
      auto pr=q.front();
      q.pop();
      int i0=pr.fst.fst,j0=pr.fst.snd,dd=pr.snd;
      if(i0<0||i0>=N) continue;
      if(j0<0||j0>=M) continue;
      if(d[i0][j0]<dd) continue;
      d[i0][j0]=dd;
      q.push({{i0+1,j0},dd+1});
      q.push({{i0-1,j0},dd+1});
      q.push({{i0,j0+1},dd+1});

      q.push({{i0,j0-1},dd+1});
    }
    int res=0;
    for(int i=0;i<N;i++){
      for(int j=0;j<M;j++){
        res=max(res,d[i][j]);
      }
    }
      if(res==0) res=-1;
    return res;
  }
};
