class Solution {
public:
  int maxDistance(vector<vector<int>>& grid) {
    int n = grid.size();
    int m = grid[0].size();
    vector<vector<int>> d(n, vector<int>(m, -1));
    deque<int> que;
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < m; ++j) {
        if (grid[i][j]) {
          d[i][j] = 0;
          que.push_back((i << 16) + j);
        }
      }
    }
    int dr[] = {-1, 0, 1, 0};
    int dc[] = {0, 1, 0, -1};
    while (!que.empty()) {
      int r = que.front(), c = r & 0xffff;
      r >>= 16;
      que.pop_front();
      for (int i = 0; i < 4; ++i) {
        int r1 = r + dr[i], c1 = c + dc[i];
        if (r1 < 0 || r1 >= n || c1 < 0 || c1 >= m) {
          continue;
        }
        if (d[r1][c1] < 0) {
          d[r1][c1] = d[r][c] + 1;
          que.push_back((r1 << 16) + c1);
        }
      }
    }
    int ret = 0;
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < m; ++j) {
        ret = max(ret, d[i][j]);
      }
    }
    return ret > 0 ? ret : -1;
  }
};
