class Solution {
public:
    bool f(vector<vector<int>>& dis, int n, int i, int j) {
        bool change = false;
        if (i > 0) {
            if (dis[i - 1][j] < 0) {
                dis[i - 1][j] = dis[i][j] + 1;
                change = true;
            }
        }
        if (j > 0) {
            if (dis[i][j - 1] < 0) {
                dis[i][j - 1] = dis[i][j] + 1;
                change = true;
            }
        }
        if (i < n - 1) {
            if (dis[i + 1][j] < 0) {
                dis[i + 1][j] = dis[i][j] + 1;
                change = true;
            }
        }
        if (j < n - 1) {
            if (dis[i][j + 1] < 0) {
                dis[i][j + 1] = dis[i][j] + 1;
                change = true;
            }
        }
        return change;
    }
    int maxDistance(vector<vector<int>>& grid) {
        int n = grid.size();
        if (n == 0) return -1;
        bool nowater = true;
        vector<vector<int>> dis(n, vector<int>(n, -1));
        for (int i = 0 ; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (grid[i][j] == 1) {
                    dis[i][j] = 0;
                } else {
                    nowater = false;
                }
            }
        }
        if (nowater) return -1;
        bool flag = true;
        int level = 0;
        while(flag) {
            flag = false;
            for (int i = 0 ; i < n; ++i) {
                for (int j = 0; j < n; ++j) {
                    if (dis[i][j] == level) {
                        if (f(dis, n, i, j)) {
                            flag = true;
                        }
                    }
                }
            }
            level++;
        }
        int ans = -1;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                ans = max(ans, dis[i][j]);
            }
        }
        return ans;
        
    }
};
