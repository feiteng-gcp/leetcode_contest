int dis[101][101];
int dir[4][2]={{-1,0},{0,-1},{1,0},{0,1}};
struct fff
{
    int x;
    int y;
};
class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        memset(dis,1,sizeof(dis));
        int n=grid.size();
        queue<fff> q;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(grid[i][j])
                {
                    fff now;
                    now.x=i;
                    now.y=j;
                    dis[i][j]=0;
                    q.push(now);
                }
            }
        }
        if(q.empty())return -1;
        while(!q.empty())
        {
            fff now=q.front();
            q.pop();
            for(int i=0;i<4;i++)
            {
                int nowx=now.x+dir[i][0],nowy=now.y+dir[i][1];
                if(nowx>=0&&nowx<n&&nowy>=0&&nowy<n&&dis[nowx][nowy]>dis[now.x][now.y]+1)
                {
                    dis[nowx][nowy]=dis[now.x][now.y]+1;
                    fff news;
                    news.x=nowx;
                    news.y=nowy;
                    q.push(news);
                }
            }
        }
        int ans=-1;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(dis[i][j])ans=max(ans,dis[i][j]);
            }
        }
        return ans;
    }
};
