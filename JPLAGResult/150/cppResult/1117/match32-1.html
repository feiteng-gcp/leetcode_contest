class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        std::vector<std::vector<int>> lut(grid.size(), std::vector<int>(grid[0].size(), INT_MAX));

        int poses[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        std::vector<std::vector<int>> visited(grid.size(), std::vector<int>(grid[0].size(), -1));
        std::queue<std::pair<std::pair<int, int>, int>> q;
        int flag = 1;
        
        for(int i = 0; i < grid.size(); ++i){
            for(int j = 0; j < grid[0].size(); ++j){
                if(grid[i][j] == 0) continue;

                lut[i][j] = 0;
                visited[i][j] = flag;
                for(int k = 0; k < 4; ++k){
                    int tmpi = i + poses[k][0];
                    int tmpj = j + poses[k][1];
                    if(tmpi < grid.size() && tmpi > -1 && tmpj < grid[0].size() && tmpj > -1 && visited[tmpi][tmpj] != flag){
                        if(grid[tmpi][tmpj] == 0){
                            lut[tmpi][tmpj] = 1;
                        }
                    }
                }
            }
        }

        for(int i = 0; i < grid.size(); ++i){
            for(int j = 0; j < grid[0].size(); ++j){
                if(lut[i][j] == 1){
                    visited[i][j] = flag;
                    q.push({{i, j}, 1});
                }
            }
        }
        while(!q.empty()){
            auto p = q.front();
            q.pop();
            for(int k = 0; k < 4; ++k){
                int tmpi = p.first.first + poses[k][0];
                int tmpj = p.first.second + poses[k][1];
                if(tmpi < grid.size() && tmpi > -1 && tmpj < grid[0].size() && tmpj > -1 && visited[tmpi][tmpj] != flag){
                    if(grid[tmpi][tmpj] == 0){
                        lut[tmpi][tmpj] = min(lut[tmpi][tmpj], p.second + 1);
                        q.push({{tmpi, tmpj}, p.second + 1});
                    }
                    visited[tmpi][tmpj] = flag;

                }
            }
        }
        int maxv = 0;
        for(int i = 0; i < grid.size(); ++i){
            for(int j = 0; j < grid[0].size(); ++j){
                if(maxv < lut[i][j]){
                    maxv = lut[i][j];
                }
            }
        }
        return (maxv == 0 || maxv == INT_MAX) ? -1 : maxv;

    }
};
