class Solution {
public:
    int maxDistance(vector<vector<int>>& g) {
        queue<array<int, 2>> q;
        
        int h = g.size(), w = g[0].size();
        

        for (int i = 0; i < h; i++) {
            for (int j = 0; j < w; j++) {
                if (g[i][j] == 0) continue;
                q.push({i, j});
            }
        }
        
        if (q.empty() || q.size() == h*w) return -1;
        
        
        int v = 1;
        while (!q.empty()) {

            for (int s = q.size(); s > 0; s--) {
                auto cr = q.front(); q.pop();
                int y = cr[0], x = cr[1];
                
                if (y > 0 && g[y-1][x] == 0) {
                    g[y-1][x] = v; q.push({y-1, x});
                }
                
                if (y < h-1 && g[y+1][x] == 0) {
                    g[y+1][x] = v; q.push({y+1, x});
                }
                
                if (x > 0 && g[y][x-1] == 0) {
                    g[y][x-1] = v; q.push({y, x-1});
                }
                
                if (x < w-1 && g[y][x+1] == 0) {
                    g[y][x+1] = v; q.push({y, x+1});
                }
            }
            v++;
        }
        return v - 2;
    }
};
