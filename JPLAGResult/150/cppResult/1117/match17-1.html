class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        int n = grid.size();

        queue<int> q;
        // vector<vector<int>> dis(n, vector<int>(n, -1));
        // vector<vector<int>> lands;
        vector<vector<bool>> visited(n, vector<bool>(n, false));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j]) {
                    // lands.push_back({i, j});
                    q.push(i*n+j);
                    visited[i][j] = true;
                }
            }
        }
        if (q.size() == n * n) return -1;
        int maxDis = 0;
        vector<vector<int>> dir;
        dir.push_back({0,1});
        dir.push_back({1,0});
        dir.push_back({-1,0});
        dir.push_back({0,-1});
        while (!q.empty()) {
            int num = q.size();
            maxDis++;

            for (int k = 0; k < num; k++) {
                int x = q.front() / n;
                int y = q.front() % n;
                q.pop();
                for (int i = 0; i < 4; i++) {
                    int newx = x + dir[i][0];
                    int newy = y + dir[i][1];
                    if (0 <= newx && newx < n && 0 <= newy && newy < n && !visited[newx][newy]) {
                        q.push(newx * n + newy);
                        // cout << newx << ' ' << newy << endl;
                        visited[newx][newy] = true;
                    }
                }
            }
        }
        return maxDis-1;
    }
};
