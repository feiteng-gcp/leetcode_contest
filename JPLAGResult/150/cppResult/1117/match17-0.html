class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        
        vector<vector<int>> dis(m);
        vector<pair<int, int>> q;
        for (int i = 0; i < m; i++) {
            dis[i].resize(n);
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    dis[i][j] = 0;
                    q.push_back(make_pair(i, j));
                }
                    else dis[i][j] = -1;
            }
        }
        if (q.size() == 0 || q.size() == n * m) return -1;
        int dir[4][2] = {0, 1, 0, -1, 1, 0, -1, 0};
        for (int i = 0; i < q.size(); i++) {
            int x = q[i].first;
            int y = q[i].second;
            for (int j = 0; j < 4; j++) {
                int x1 = dir[j][0] + x;
                int y1 = dir[j][1] + y;
                if (x1 >= 0 && x1 < m && y1 >= 0 && y1 < n && dis[x1][y1] == -1) {
                    dis[x1][y1] = dis[x][y] + 1;
                    q.push_back(make_pair(x1, y1));
                }
            }
        }
        return dis[q.back().first][q.back().second];
    }
};
