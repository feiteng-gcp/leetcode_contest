class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        int n = grid.size();
        if ( n<=0 ) return 0;
        int m= grid[0].size();
        queue<pair<int,int>> q;
        int dir[4][2] = {{1,0},{0,1},{-1,0},{0,-1}};
        
        vector<vector<int>> dp( n, vector<int>(m,0) );
        vector<vector<bool>> book( n, vector<bool>(m,false) );
        int cnt=0;
        for ( int i=0; i<n; ++i )
            for ( int j=0; j<m; ++j )
                if ( grid[i][j]==1 ){
                    book[i][j] = true;
                    q.push({i,j});
                    cnt++;
                }
        if ( cnt==0 || cnt==n*m ) return -1;
        int step=0;
        while( !q.empty() ){
            
            int len = q.size();
            for ( int t=0; t<len; ++t ){
                pair<int,int> cor = q.front(); q.pop();
                //cout<<cor.first<<" "<<cor.second<<endl;
                for ( int k=0; k<4; ++k ){
                    int y = cor.first + dir[k][0];
                    int x = cor.second + dir[k][1];
                    if ( y>=0 && x>=0 &&x<m &&y<n && !book[y][x] ){
                        book[y][x] = true;
                        dp[y][x] = step+1;
                        q.push({y,x});
                    }
                }
            }
            ++step;
        }
        int res=0;
        for ( int i=0; i<n; ++i )
            for ( int j=0; j<m; ++j )
                res = max(res, dp[i][j]);
        return res;
    }
};
