class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        int a[26] = {0};
        for(int i=0; i<chars.size(); i++){
            a[chars[i]-'a']++;
        }
        int ans = 0;
        for(int i=0; i<words.size(); i++){

            int b[26] = {0};
            for(int j=0; j<words[i].size(); j++){
                b[words[i][j]-'a']++;
            }
            bool flag = true;
            for(int j=0; j<26; j++){
                if(b[j]>a[j]){
                    flag = false;
                }
            }
            if(flag) ans+=words[i].size();
        }
        return ans;
    }
};
