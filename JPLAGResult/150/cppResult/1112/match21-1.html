class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        vector<int> dict(26);

        for (int i = 0; i < chars.length(); ++i)
            ++dict[chars[i]-'a'];
        
        int ans = 0;
        for (int i = 0; i < words.size(); ++i) {
            string w = words[i];
            vector<int> cnt(26);

            for (int j = 0; j < w.length(); ++j)
                ++cnt[w[j]-'a'];
            
            bool ok = true;
            for (int j = 0; j < 26; ++j)
                if (cnt[j] > dict[j]) {
                    ok = false;
                    break;
                }
            
            if (ok)
                ans += w.length();
        }
        
        return ans;
    }
};
