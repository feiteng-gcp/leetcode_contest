class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        vector<int> count(26, 0);
        for (auto c:chars)
            ++count[c - 'a'];
        int res = 0;
        for (auto &word:words) {
            vector<int> c = count;
            bool f = true;
            for (auto w:word) {
                if (c[w - 'a'] == 0) {
                    f = false;
                    break;
                }
                --c[w - 'a'];
            }
            if (f)
                res += word.size();
        }
        return res;
    }
};
