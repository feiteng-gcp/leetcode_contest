class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        vector<int> count(26,0);
        for(auto c: chars)
            count[c-'a']++;
        int ans = 0;
        for(auto w : words){
            vector<int> temp(26,0);

            bool good = true;
            for(auto c : w){
                if(temp[c-'a']==count[c-'a']){
                    good = false;
                    break;
                }
                temp[c-'a']++;
            }
            if(good)
                ans+= w.size();
        }
        return ans;
    }
};
