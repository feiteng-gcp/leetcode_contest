class Solution {
    
    
    bool check(array<int, 26>& cnt, array<int, 26>& a) {
        for (int i = 0; i < 26; i++) {
            if (a[i] > cnt[i]) return false;
        }

        return true;
    }
    
public:
    int countCharacters(vector<string>& words, string chars) {
        
        array<int, 26> cnt = {};
        for (char c : chars) cnt[c-'a']++;
        
        int ans = 0;
        
        for (auto& w : words) {

            array<int, 26> a = {};
            for (char c : w) a[c-'a']++;
            
            if (check(cnt, a)) 
                ans += w.size();
        }
        return ans;
        
    }
};
