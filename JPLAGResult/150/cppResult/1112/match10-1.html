class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        vector<int> st(26, 0);
        for (char c:chars) st[c-'a']++;
        int ans = 0;
        for (auto& v:words){
            vector<int> in(26, 0);
            for (char c:v) in[c-'a']++;
            bool flag = true;
            for (int i=0; i<26; i++) {

                if (in[i] > st[i]) {
                    flag = false;
                    break;
                }
            }
            if (flag) ans += v.length();
        }
        return ans;
    }
};
