class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        int res=0;
        int dp[128]={0};
        for ( auto e:chars )
            dp[e]++;
        for ( auto e:words ){
            int i,tmp[128]={0};
            for ( i=0; i<e.size(); ++i ){
                if ( ++tmp[e[i]]>dp[e[i]] ){
                  break;  
                }
            }
            if ( i>=e.size() ){
                //cout<<e<<endl;
                res += e.size();
            }
        }
        return res;
    }
};
