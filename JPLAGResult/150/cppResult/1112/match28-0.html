class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        int freq[26] = {0};
        for(int i = 0;i < chars.size();i++)
            freq[chars[i] - 'a']++;
        int ans = 0;
        for(int i = 0;i < words.size();i++)
        {
            bool good = true;
            int tmpFreq[26] = {0};
            for(int j = 0;j < words[i].size();j++)
                tmpFreq[words[i][j] - 'a']++;
            for(int j = 0;j < 26;j++)
            {
                if(freq[j] < tmpFreq[j])
                    good = false;
            }
            if(good)
                ans += words[i].length();
        }
        return ans;
    }
};
