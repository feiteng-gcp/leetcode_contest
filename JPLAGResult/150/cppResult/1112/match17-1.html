class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        vector<int> count(26, 0);
        for (char c : chars) {
            count[c - 'a']++;
        }
        
        int ans = 0;
        for (auto& word : words) {
            vector<int> cnt = count;
            bool valid = true;
            for (char c: word) {
                if (cnt[c - 'a'] > 0) {
                    cnt[c - 'a']--;

                } else {
                    valid = false;
                    break;
                }
            }
            if (valid) {
                ans += word.size();
            }
        }
        return ans;
    }
};
