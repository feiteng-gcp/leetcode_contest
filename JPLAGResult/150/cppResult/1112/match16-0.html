int cnt[26],cntt[26];
class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        int n=words.size(),m=chars.size();
        memset(cnt,0,sizeof(cnt));
        for(int i=0;i<m;i++)
        {
            cnt[chars[i]-'a']++;
        }
        int ans=0;
        for(int i=0;i<n;i++)
        {
            int l=words[i].size();
            memset(cntt,0,sizeof(cntt));
            for(int j=0;j<l;j++)
            {
                cntt[words[i][j]-'a']++;
            }
            int ok=1;
            for(int j=0;j<26;j++)
            {
                if(cnt[j]<cntt[j])
                {
                    ok=0;
                    break;
                }
            }
            if(ok)ans+=l;
        }
        return ans;
    }
};
