class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        vector<int> ans(26, 0);
        for (auto &ch: chars)
            ++ans[ch - 'a'];
        
        int res = 0;
        
        for (auto &w: words) {
            vector<int> cur(26, 0);
            for (auto &ch: w)
                ++cur[ch - 'a'];
            
            bool f = true;

            for (size_t i = 0; i < 26; ++i)
                if (cur[i] > ans[i]) {
                    f = false;
                    break;
                }
            
            if (f)
                res += w.size();
        }
        
        return res;
    }
};
