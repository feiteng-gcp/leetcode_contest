class Solution {
public:
	int countCharacters(vector<string>& words, string chars) {
		map<char, int>ct;
		for (auto it : chars) {
			ct[it]++;
		}
		int ret = 0;
		for (int i = 0; i < words.size(); i++) {
			map<char, int>ct2;
			for (auto it : words[i])ct2[it]++;
			bool bad = false;
			for (auto it : ct2) {
				if (ct.find(it.first) == ct.end()) {
					bad = true;
					break;
				}

				if (ct[it.first] < it.second) {
					bad = true;
					break;
				}
			}
			if (!bad)ret += words[i].size();
		}
		return ret;
	}
};
