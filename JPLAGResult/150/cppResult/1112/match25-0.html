class Solution {
    
    bool check(int *cnt, int* a) {
        for (int i = 0; i < 26; i++)
            if (cnt[i] < a[i]) return false;
        return true;
    }
public:
    int countCharacters(vector<string>& words, string chars) {
        
        int cnt[26] = {};
        for (char c : chars) cnt[c-'a']++;
        
        
        int ans = 0;
        
        for (string& w : words) {
            int a[26] = {};
            for (char c : w) a[c-'a']++;
            
            if (check(cnt, a)) ans += w.size();
            
        }
        
        
        return ans;
        
    }
};
