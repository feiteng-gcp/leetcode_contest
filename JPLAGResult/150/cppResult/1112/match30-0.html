class Solution {
public:
    bool check(const std::string & word, const std::vector<int> & tmpt){
        std::vector<int> lut(26, 0);
        for(auto each : word){
            ++lut[each-'a'];
        }
        for(int i = 0; i < 26; ++i){
            if(lut[i] > tmpt[i]){
                return false;
            }
        }
        return true;
    }
    
    int countCharacters(vector<string>& words, string chars) {
        std::vector<int> tmpt(26, 0);
        for(auto each : chars){
            ++tmpt[each-'a'];
        }
        int cnt = 0;
        for(auto & each : words){
            if(check(each, tmpt)){
                cnt += each.size();
            }
        }
        return cnt;
    }
};
