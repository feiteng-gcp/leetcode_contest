class Solution
{
 public:

 int countCharacters(vector<string> &words, string s)
 {
   int hashW[130] = {0};

   for(int i = 0; i < s.length(); ++i) hashW[s[i]]++;

   int answer = 0;
   for(int i = 0; i < words.size(); ++i)
   {
     string p = words[i];
     int hashQ[130] = {0};

     for(int j = 0; j < p.length(); ++j)
     {
       hashQ[p[j]]++;
     }

     bool valid = true;
     for(int j = 0; j < 130 && valid; ++j)
     {
       if(hashQ[j] > hashW[j]) { valid = false; break; }
     }

     if(valid) answer += p.length();
   }

   return answer;
 }
};
