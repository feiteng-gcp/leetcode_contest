class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        map<int,int> tmp;
        for(char c:chars){
            tmp[c]++;
        }
        int res=0;
        for(auto w:words){
            auto t=tmp;
            for(char c:w){
                t[c]--;
                if(t[c]<0) {
                    res-=w.size();break;
                }
            }
            res+=w.size();
        }
        return res;
    }
};
