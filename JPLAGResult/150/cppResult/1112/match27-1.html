class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        int n=words.size();
        int m=chars.length();
        int siz[30], cnt[30];
        
        memset(siz, 0, sizeof(siz));

        for (int i=0; i<m; i++) ++siz[chars[i]-'a'];
        
        int ans=0;
        for (int i=0; i<n; i++){
            int l=words[i].length();    
            memset(cnt, 0, sizeof(cnt));

            for (int j=0; j<l; j++) ++cnt[words[i][j]-'a'];
            bool flag=true;
            for (int j=0; j<26; j++){
                if (cnt[j]>siz[j]) flag=false;
            }
            if (flag) ans+=l;
        }
        
        return ans;
    }
};
