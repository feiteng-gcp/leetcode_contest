class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        
        unordered_map<char, int> cntmap;
        for (char c : chars) cntmap[c]++;
        
        int sumlen = 0;
        
        for (string& str : words)
        {
            unordered_map<char, int> cntmap2;
            for (char c : str) cntmap2[c]++;
            for (auto p : cntmap2)
            {
                char c = p.first;
                int cnt = p.second;
                if (cntmap[c] < cnt) goto skip;
            }
            sumlen += str.size();
            skip:
            int n;
        }
        
        return sumlen;
        
    }
};
