class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        map<char, int> counts;
        
        for (char c : chars) ++counts[c];
        
        int valid = 0;
        for (string& word : words) {
            map<char, int> w_counts;
            for (char c : word) ++w_counts[c];
            
            auto it = w_counts.begin();
            while (it != w_counts.end() && it->second <= counts[it->first]) ++it;
            if (it == w_counts.end()) valid += word.length();
        }
        return valid;
    }
};
