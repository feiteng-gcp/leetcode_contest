class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        unordered_map<char, int> s;
        for (auto &c : chars) {
            s[c]++;
        }
        int ans = 0;
        for (auto &x : words) {
            unordered_map<char, int> t;
            bool ok = true;
            for (auto &c : x) {
                t[c]++;
                if (s[c] < t[c]) {
                    ok = false;
                    break;
                }
            }
            if (ok) ans += x.size();
        }
        return ans;
    }
};
