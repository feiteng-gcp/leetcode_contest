class Solution {
public:
	int countCharacters(vector<string>& words,string chars) {
		vector<int>nums(26,0);
		for(auto c:chars)nums[c-'a']++;
		int res=0;
		for(auto s:words){
			vector<int>n(26,0);
			bool flag=1;
			for(auto c:s){
				if(++n[c-'a']>nums[c-'a']){
					flag=0;
					break;
				}
			}
			if(flag)
				res+=s.size();
		}
		return res;
	}
};
