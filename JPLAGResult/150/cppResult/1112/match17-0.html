class Solution {
 public:
  int countCharacters(vector<string>& words, string chars) {
    vector<int> freq(26, 0);
    for (char &ch : chars) {
      freq[ch - 'a']++;
    }
    int counter = 0;
    for (auto &word : words) {
      vector<int> freq_cur(26, 0);
      bool can_form = true;
      for (char &ch : word) {
        if (++freq_cur[ch - 'a'] > freq[ch - 'a'])
        {
          can_form = false;
          break;
        }
      }
      if (can_form) {
        counter += word.size();
      }
    }
    return counter;
  }
};
