class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {

        map<char, int> c;
        for (const auto& a:chars) {
            ++c[a];
        }
        
        int ans = 0;
        for (const auto& w : words) {
            map<char, int> t;
            for (const auto& a:w) {
                ++t[a];
            }
            
            bool ok = true;
            for (const auto& i:t) {

                if (i.second > c[i.first]) {
                    ok = false;
                    break;
                }
            }
            
            if (ok) ans += w.size();
        }
        return ans;
    }
};
