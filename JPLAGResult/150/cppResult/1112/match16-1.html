class Solution {
public:
    int countCharacters(vector<string>& words, string chars) {
        int ct[26] = {};

        for (int i=0; i<chars.size(); i++) {
            ct[chars[i] - 'a']++;
        }
        
        int good = 0;
        for (int i=0; i<words.size(); i++) {
            int ct2[26] = {};
            for (int j=0; j<words[i].size(); j++) {
                ct2[words[i][j] - 'a']++;
            }
            

            bool valid = true;
            for (int j=0; j<26; j++) {
                if (ct[j] < ct2[j]) {
                    valid = false;
                    break;
                }
            }
            
            if (valid) good += words[i].size();
        }
        
        return good;
    }
};
