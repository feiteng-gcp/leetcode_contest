class Solution:
    def numTilePossibilities(self, tiles: str) -> int:
        n = len(tiles)
        tiles_arr = [c for c in tiles]
        tiles_arr.sort()
        res = set()
        visited = [False for _ in range(n)]
        def helper(cur):
            if cur and cur not in res:
                res.add(cur)
            for i in range(len(tiles)):
                if visited[i]:
                    continue
                visited[i] = True
                helper(cur + tiles[i])
                visited[i] = False
        helper('')
        return len(res)
