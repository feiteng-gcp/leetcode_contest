class Solution:
    def numTilePossibilities(self, tiles: str) -> int:
        s
=set();used=[False]*len(tiles)
        def search(idx,tiles,limit,used,cur):
            if idx==limit:
                s.add(cur)
            for i in range(len(tiles)):
                if used[i]: continue
                used[i]=True
                search(idx+1,tiles,limit,used,cur+tiles[i])
                used[i]=False
        for i in range(1,len(tiles)+1):
            search(0,tiles,i,used,"")
        return len(s)
