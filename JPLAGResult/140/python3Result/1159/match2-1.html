from collections import Counter
class Solution:
    def smallestSubsequence(self, text: str) -> str:
        stack = []
        count = Counter(text)
        used = set()
        for c in text:
            if c in used:
                count[c] -= 1
                continue
            
while stack and c < stack[-1] and count[stack[-1]] > 0:
                used.remove(stack.pop())
            stack.append(c)
            used.add(c)
            count[c] -= 1
        return ''.join(stack)
