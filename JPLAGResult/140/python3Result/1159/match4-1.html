class Solution:
    def smallestSubsequence(self, text: str) -> str:
        chars = sorted(list(set(text)))
        indice = self.lastIndice(text, 0, chars)
        ans = ""
        last_choice = -1
        for _ in chars:
            (c, v) = min(indice.items(), key=lambda t:t[1])
            x = [(t[0] + last_choice + 1, t[1]) for t in enumerate(text[last_choice + 1: v + 1])]
            x = [(t[0], t[1]) for t in x if t[1] not in ans]
            (idx, pick) = sorted(x, key=lambda t: t[1])[0]
            ans += pick
            del indice[pick]
            last_choice =
 idx
        return ans
            
    def lastIndice(self, text, i, chars):
        ans = dict()
        for t in range(i, len(text)):
            if text[t] in chars:
                ans[text[t]] = t
        return ans
