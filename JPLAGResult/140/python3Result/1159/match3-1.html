class Solution:
    def smallestSubsequence(self, text: str) -> str:
        aux = collections.defaultdict(list)
        for i, c in enumerate(text): aux[c].append(i)
        K, ans, i 
= len(aux), '', 0
        while len(ans) < K:
            for c in sorted(aux.keys()):
                i = aux[c][0]
                if all(any(j >= i for j in v) for v in aux.values()):
                    ans += c
                    for k, v in aux.items():
                        aux[k] = [j for j in v if j > i]
                    aux.pop(c)
                    break
        return ans
