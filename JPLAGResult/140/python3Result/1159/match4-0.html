class Solution:
    def smallestSubsequence(self, text: str) -> str:
        letters = set()
        for c in text:
            letters.add(c)
        return "".join(self.find_seq(letters, text, 0, []))

    def find_seq(self, letters, text, cur_ind, cur):
        if not letters:
            return list(cur)
        if not self.check(text, letters, cur_ind):
            return []

        letters = list(letters)
        letters.sort()
        for i in range(len(letters)):
            for j in range(cur_ind, len(text)):
                if text[j] == letters[i]:
                    newLetters = set(letters)
                    newLetters.remove(letters[i])
                    cur.append(letters[i])
                    seq = self.find_seq(newLetters, text, j + 1, cur)
                    if seq:
                        return seq
                    cur.pop()
        return []

    def check(self, text, letters, start):
        letters = set(letters)
        for i in range(start, len(text)):
            if text[i] in letters:
                letters.remove(text[i])
        return len(letters) == 0
