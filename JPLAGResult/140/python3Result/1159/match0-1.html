from collections import Counter
class Solution:
    def smallestSubsequence(self, text: str) -> str:
        freq = Counter(text)
        stk = []
        seen = set()
        for c in text:
            freq[c] -= 1
            if c in seen: 
continue
            while stk and stk[-1] > c and freq[stk[-1]] > 0:
                seen.remove(stk.pop())
            stk.append(c)
            seen.add(c)
        return ''.join(stk)
