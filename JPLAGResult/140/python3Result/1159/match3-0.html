from collections import *
from bisect import bisect
class Solution:
    def smallestSubsequence(self, text: str) -> str:
        #每个字符单独写一个序列
        List = defaultdict(list)
        for i,v in enumerate(text):
            List[v].append(i)
        res = []
        #print(List)
        n = len(List)
        while len(res)!=n:
            for c in sorted(List.keys()):
                pre = List[c][0]
                if all(List[x][-1]>=pre for x in List):
                    #此时需要修改
                    res.append(c)
                    del List[c]
                    for x in List:
                        index = bisect(List[x],pre)
                        List[x] = List[x][index:]
                    break
        return "".join(res)
