class Solution:
    def findOcurrences(self, text: str, first: str, second: str) -> List[str]:
        ans = []
        splits = text.split(' ')
        for i in range(len(splits) - 2):
            if splits[i] == first and splits[i + 1] == second:
                ans.append(splits[i + 2])
        return ans
