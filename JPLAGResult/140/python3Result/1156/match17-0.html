class Solution:
    def findOcurrences(self, text: str, first: str, second: str) -> List[str]:
        l = text.split(" ")
        res = []
        for i in range(1, len(l) - 1):
            if l[i - 1] == first and l[i] == second:
                res.append(l[i + 1])
        return res
