class Solution:
    def findOcurrences(self, text: str, first: str, second: str):
        textList = text.split(' ')
        ans = []
        for idx in range(len(textList) - 2):
            if textList[idx] == first and textList[idx+1] == second:
                ans.append(textList[idx+2])
        return ans
