class Solution:
    def findOcurrences(self, text: str, first: str, second: str) -> List[str]:
        a = text.split(' ')
        p = []
        for i in range(2, len(a)):
            if first == a[i-2] and second == a[i-1]:
                p.append(a[i])
                
        return p
