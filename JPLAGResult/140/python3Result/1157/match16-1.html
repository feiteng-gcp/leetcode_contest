# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        self.map = dict()
        self.dfs(root, [])
        # print(self.map)
        root = self.clean(root, limit)
        return root
        
    def dfs(self, node: TreeNode, parents: List[TreeNode]):
        parents.append(node)
        if node.left != None:
            self.dfs(node.left, parents)
        if node.right != None:
            self.dfs(node.right, parents)
        if node.left == None and node.right == None:
            su = 0
            for i in parents:
                su += i.val
            for i in parents:
                if id(i) in self.map:
                    self.map[id(i)] = max(su, self.map[id(i)])
                else:
                    self.map[id(i)] = su
                # i.max = max(i.max, su)
            # print("sum:", su)
            # print("parents:", parents)
        parents.pop()
        
    def clean(self, node: TreeNode, limit: int) -> TreeNode:
        if node == None:
            return None
        if self.map[id(node)] < limit:
            
return None
        node.left = self.clean(node.left, limit)
        node.right = self.clean(node.right, limit)
        return node
