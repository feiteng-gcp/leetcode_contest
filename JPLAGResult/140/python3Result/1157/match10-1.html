# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def dfs(self, root, prevSum):
        if not root:
            return None
        prevSum += root.val
        if root.left is None and root.right is None:
            if prevSum >= self.limit:
                return root
            else:
                return None
        root
.left = self.dfs(root.left, prevSum)
        root.right = self.dfs(root.right, prevSum)
        if root.left is not None or root.right is not None:
            return root
        return None
            

        
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        self.limit = limit
        return self.dfs(root, 0)
