# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        def helper(node, cur_sum):
            if not node:
                return [], None
            cur_sum += node.val
            if not node.left and not node.right:
                if cur_sum < limit:
                    return [cur_sum], None
                else:
                    return [cur_sum], node
            left, left_node = helper(node.left, cur_sum)
            right, right_node = helper(node.right, cur_sum)
            total = left + right
            delete = True
            for t in total:
                if t >= limit:
                    delete = False
                    break
            if delete:
                return total, None
            else:
                node.left = left_node
                node.right = right_node
                return total, node
        _, root = helper(root, 0)
        return root
