# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        def helper(node, limit):
            if node is None: return 0
            l = helper(node.left, limit - node.val)
            if l < limit - node.val: node.left = None
            r = helper(node.right, limit - node.val)
            if r < limit - node.val: node.right = None
            return node.val + max(l, r)
        max_path = helper(root, limit)
        return None if max_path < limit else root
