# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        if not root:
            return None
        ans = self.solve(root, 0, limit)
        if ans == True:
            return None
        self.process(root)
        return root
    
    def solve(self, root, s, limit):
        s += root.val
        ans = True if root.left or root.right else False
        if root.left:
            ans = self.solve(root.left, s, limit) and ans
        if root.right:
            ans = self.solve(root.right, s, limit) and ans
        if not root.left and not root.right and s < limit:
            root.val = '$'
            return True
        if ans:
            root.val = '$'
            return True
        return False
    
    def process(self, root):
        if root.left and root.left.val == '$':
            root.left = None
        if root.right and root.right.val == '$':
            root.right = None
        if root.left:
            self.process(root.left)
        if root.right:
            self.process(root.right)
