# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def retmax(self, root): 
        return root.val + max(self.retmax(root.left), self.retmax(root.right))
    
    
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        if not root:
            return root
        if not root.right and not root.left:
            if root.val<limit:
                return None
            else: 
                return root 
        root.right=self.sufficientSubset(root.right, limit - root.val)
        root.left=self.sufficientSubset(root.left, limit - root.val)
        if not root.right and not root.left:
            return None 
        
        return root
