# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sum_helper(self, root, path_sum):
        if not root:
            return []
        if not root.left and not root.right:
            root.path_sums = [path_sum + root.val]
            return root.path_sums

        root.path_sums = []
        root.path_sums.extend(self.sum_helper(root.left, path_sum + root.val))
        root.path_sums.extend(self.sum_helper(root.right, path_sum + root.val))

        return root.path_sums

        
    def update_helper(self, root, limit):
        # print(root.val, root.path_sums)
        if not root or not root.path_sums:
            return None
        
        if all([path_sum < limit for path_sum in root.path_sums]):
            return None

        root.left = self.update_helper(root.left, limit)
        root.right = self.update_helper(root.right, limit)
        return root

    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        self.sum_helper(root, 0)
        root = self.update_helper(root, limit)

        return root
