# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        if not root:
            return None

        self.l = limit
        
        root_cut = self.dfs(root, 0)
        
        if root_cut:
            return None
        else:
            return root
    
    def dfs(self, node: TreeNode, s: int) -> bool:
        if not node:
            if s < self.l:
                return True
            else:
                return False
        
        s += node.val
        
        lt_cut = self.dfs(node.left, s)
        rt_cut = self.dfs(node.right, s)
        
        # print(node.val, lt_cut, rt_cut, s)
        
        if lt_cut:
            node.left = None
        if rt_cut:
            node.right = None
        
        if lt_cut and rt_cut:
            return True
        else:
            return False
