# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        def dfs(nd, s):
            s += nd.val
            if not nd.left and not nd.right:
                if s < limit:
                    return 0
                return 1
            l = r = 0
            if nd.left:
                l = dfs(nd.left, s)
                if not l:
                    nd.left = None
            if nd.right:
                r = dfs(nd.right, s)
                if not r:
                    nd.right = 
None
            if not l and not r:
                return 0
            return 1
        res = dfs(root, 0)
        if not res:
            return None
        return root
