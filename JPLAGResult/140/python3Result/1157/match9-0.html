# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        def dfs(root,total):
            if root:
                total += root.val
                left = dfs(root.left,total)
                right = dfs(root.right,total)
                if left<limit:
                    root.left = None
                if right<limit:
                    root.right = None                
                return max(left,right)
            else:
                return total
        
        total = dfs(root,0)
        return root if total>=limit else None
