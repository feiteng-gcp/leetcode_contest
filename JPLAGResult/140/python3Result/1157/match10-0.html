# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        return self.traverse(root, limit, 0)
        
    def traverse(self, node, limit, currentSum):
        if not node:
            return None
        
        # is a leaf node
        if not node.left and not node.right:
            currentSum += node.val
            if currentSum < limit:
                return None
            else:
                return node
        
        currentSum += node.val
        # not a leaf node, it will have left and right subResult
        node.left = self.traverse(node.left, limit, currentSum)
        node.right = self.traverse(node.right, limit, currentSum)
        
        # the node was not a leaf, but it's children are deleted, means this node should be deleted as well
        if not node.left and not node.right:
            return None
        
        return node
