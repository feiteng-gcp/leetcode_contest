# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        self.maxH(root)
        root = self.patch(root, limit)
        return root
    
    def maxH(self, n):
        if n.left == None and n.right == None:
            n.h = n.val
        elif n.left == None:
            n.h = n.val + self.maxH(n.right)
        elif n.right == None:
            n.h = n.val + self.maxH(n.left)
        else:
            n.h = n.val + max(self.maxH(n.left), self.maxH(n.right))
        return n.h
    
    def patch(self, n, limit):
        if n.h < limit:
            return None
        if n.left != None:
            n.left = self.patch(n.left, limit - n.val)
        if n.right != None:
            n.right = self.patch(n.right, limit - n.val)
        return n
