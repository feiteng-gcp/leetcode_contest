    def helper(self, node, limit):
        if not node: return True
        
        if not node.left and not node.right:
            if node.val < limit:
                return True
            else: return False
        
        leftdel = self.helper(node.left, limit - node.val)
        rightdel = self.helper(node.right, limit - node.val)
        
        if leftdel: 
            node.left = None
        if rightdel: 
            node.right = None
        
        return leftdel and rightdel
