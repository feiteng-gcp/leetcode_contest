# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        def isLeaf(node): return node.left == None and node.right == None
        def dfs(node, sum):
            if node == None: return False
            sum += node.val
            if isLeaf(node): return sum >= limit
            lRes 
= dfs(node.left, sum)
            rRes = dfs(node.right, sum)
            if lRes == False: node.left = None
            if rRes == False: node.right = None
            return lRes or rRes
        
        if dfs(root, 0): return root
        else: return None
