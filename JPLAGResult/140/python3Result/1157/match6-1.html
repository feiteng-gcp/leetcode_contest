# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        def checklimit(root, cur):
            if root is None:
                if cur >= limit:
                    return True
                return False
            
            cur += root.val
            rc1 = checklimit(root.left, cur)
            rc2 = checklimit
(root.right, cur)
            if rc1 == False:
                root.left = None
            if rc2 == False:
                root.right = None
            if rc1 or rc2:
                return True
            return False
        
        r = checklimit(root, 0)
        if r:
            return root
        return None
