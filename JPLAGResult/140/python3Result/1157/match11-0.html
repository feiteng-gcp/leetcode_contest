# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        def maxPath(node, sum_so_far):
            if not node:
                return sum_so_far
            l_sum = maxPath(node.left, sum_so_far+node.val)
            if l_sum < limit:
                node.left = None
            r_sum = maxPath(node.right, sum_so_far+node.val)
            if r_sum < limit:
                node.right = None
            return max(l_sum, r_sum)
        
        res = maxPath(root, 0)  
        return None if res < limit else root
