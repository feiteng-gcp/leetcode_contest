# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        
        def s(p, q, d):
            nonlocal ps
            if not p:
                return False
            ps += p.val
            if not p.left and not p.right:
                if ps < limit:
                    if d:
                        q.left = None
                    else:
                        q.right = None
                ret = ps >= limit
            else:
                l, r = s(p.left, p, 1), s(p.right, p, 0)
                if not (l or r):
                    if d:
                        q.left = None
                    else:
                        q.right = None
                ret = l or r
            ps -= p.val
            return ret
            
        ps = 0
        parent = TreeNode(0)
        parent.left = root
        s(root, parent, 1)
        return parent.left
