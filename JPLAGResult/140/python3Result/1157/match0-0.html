# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        return self.helper(root, limit, 0)

    def helper(self, root, limit, value):
        # print(root.val)
        if root.left is None and root.right is None:
            # print(root.val, value)
            if root.val + value < limit:
                return None
            else:
                return root
        left = None
        right = None
        if root.left:
            left = self.helper(root.left, limit, value + root.val)
        if root.right:
            right = self.helper(root.right, limit, value + root.val)
        if left is None and right is None:
            return None
        root.left = left
        root.right = right
        return root
