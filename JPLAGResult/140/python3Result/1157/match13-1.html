class Solution:
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        def helper(node, vals):
            if not node:return 0
            val = node.val + vals
            haveson = node.left or node.right
            lv, rv = helper(node
.left,val), helper(node.right, val)
            if lv < limit:
                node.left = None
            if rv < limit:
                node.right = None
            if (not node.left) and (not node.right):
                if haveson: return limit - 1
                return val
            return max(lv,rv)
        res = helper(root,0)
        return root if res >= limit else None
