class Solution:
    def dfs(self, root: TreeNode, su: int):
        if root is None:
            return True
        if root.left is None and root.right is None:
            return su + root.val < self.limit
        l = self.dfs(root.left, su + root.val)
        r = self.dfs(root.right, su + root.val)
        if l:
            root.left = None
        if r:
            root.right = None
        return l and r

    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        self.limit = limit
        if not self.dfs(root, 0):
            return root
        return None
