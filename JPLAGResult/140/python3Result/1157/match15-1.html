# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def dfs(self, u, sp):
        if u is None:
            return self.limit - 1
        elif u.left is None and u.right is None:
            sval = sp + u.val
            if sval < self.limit:
                u.val = None
            return sval
        if u.left is not None:
            sleft = self.dfs(u.left, sp + u.val)
            if sleft < self.limit:
                u.left = None
        else:
            sleft = self.limit - 1
        if u.right is not None:
            sright = self.dfs(u.right, sp + u.val)
            if sright < self.limit:
                u.right = None
        else:
            sright = self.limit - 1
        return max(
sleft, sright)
            
    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        self.limit = limit
        if self.dfs(root, 0) < limit:
            return None
        return root
