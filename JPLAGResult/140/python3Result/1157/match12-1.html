# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def travelTree(self, root: TreeNode, limit: int, summ: int) -> TreeNode:
        if root.left == None and root.right == None:
            if summ + root.val < limit:
                root =
 None
        else:
            if root.left != None:
                root.left = self.travelTree(root.left, limit, summ + root.val)
            if root.right != None:
                root.right = self.travelTree(root.right, limit, summ + root.val)
            if root.left == None and root.right == None:
                root = None
        return root

    def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        return self.travelTree(root, limit, 0)
