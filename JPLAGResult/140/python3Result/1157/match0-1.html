# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    
def sufficientSubset(self, root: TreeNode, limit: int) -> TreeNode:
        if root.left==None and root.right==None:
            if root.val<limit:
                return None
            else:
                return root
        lf=None
        rf=None
        if not root.left==None:
            lf=self.sufficientSubset(root.left,limit-root.val)
        if not root.right==None:
            rf=self.sufficientSubset(root.right,limit-root.val)
        if lf==None and rf==None:
            return None
        else:
            root.left=lf
            root.right=rf
            return root
