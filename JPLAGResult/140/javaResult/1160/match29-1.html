class Solution {
    public int numTilePossibilities(String tiles) {
        List<Character> list = new ArrayList<> ();
        for (char ch : tiles.toCharArray()) {
            list.add(ch);
        }
        Set<String> set = new HashSet<> ();
        for (int i = 1; i <= tiles.length(); i++) {
            dfs(list, set, new StringBuilder(), new boolean[
list.size()], i);
        }
        return set.size();
    }
    
    private void dfs(List<Character> list, Set<String> set, StringBuilder sb, boolean[] used, int len) {
        if (sb.length() == len) {
            set.add(sb.toString());
        }
        for (int i = 0; i < list.size(); i++) {
            if (used[i]) {
                continue;
            }
            sb.append(list.get(i));
            used[i] = true;
            dfs(list, set, sb, used, len);
            used[i] = false;
            sb.deleteCharAt(sb.length() - 1);
        }
    }
}
