class Solution {
    public int numTilePossibilities(String tiles) {
        Set<String> all = new HashSet<>();
        for(int i = 0; i < tiles.length(); ++i) {
            backtrack(tiles.toCharArray(), i, "", all);    
        }
        
        return 
all.size()-1;
    }
    
    void backtrack(char[] tiles, int from, String cur, Set<String> all) {
        all.add(cur);
        
        for(int i = from; i < tiles.length; ++i) 
{
            swap(tiles, i, from);
            backtrack(tiles, from+1, cur + tiles[from], all);
            
            swap(tiles, i, from);
        }
    }
    
    void swap(char[] tiles, int i, int j) {
        char t = tiles[i];
        tiles[i] = tiles[j];
        tiles[j] = t;
    }
}
