class Solution {
    HashSet<String> seen;
    public int numTilePossibilities(String tiles) {
      int[] letterCount = new int[26];
      for(int i = 0; i < tiles.length(); i++){
        letterCount[tiles.charAt(i) - 'A']++;
      }
      
      seen = new HashSet<String>();
      dfs("",letterCount);
      return seen.size() - 1;
    }
    
    public void dfs(String current, int[] lettersLeft){
      seen.add(current);
      for(int i = 0; i < 26; i++){
        if(lettersLeft[i] > 0){
          lettersLeft[i]--;
          dfs(current + ((char) 'A' + i), lettersLeft);
          lettersLeft[i]++;
        }
      }
    }
}
