class Solution {
    static final int[] fac;
    static {
        fac = new int[20];
        fac[0] = 1;
        for (int i = 1; i < fac.length; i++) {
            fac[i] = i * fac[i-1];
        }
    }
    
public int numTilePossibilities(String tiles) {
        int[] count = new int[26];
        for (int i = 0; i < tiles.length(); i++) {
            count[tiles.charAt(i) - 'A']++;
        }
        int single = 0;
        for (int i = 0; i < 26; i++) {
            if (count[i] == 1) {
                count[i] = 0;
                single++;
            }
        }
        int a = 0, b = 0, c = 0;
        for (int i = 0; i < 26; i++) {
            if (count[i] > 0) {
                a = count[i];
                count[i] = 0;
                break;
            }
        }
        for (int i = 0; i < 26; i++) {
            if (count[i] > 0) {
                b = count[i];
                count[i] = 0;
                break;
            }
        }
        for (int i = 0; i < 26; i++) {
            if (count[i] > 0) {
                c = count[i];
                count[i] = 0;
                break;
            }
        }
        
return tryP(a, b, c, single);
    }
    int tryP(int a, int b, int c, int single) {
        int self = -1;
        for (int ia = 0; ia <= a; ia++) {
            for (int ib = 0; ib <= b; ib++) {
                for (int ic = 0; ic <= c; ic++) {
                    for (int is = 0; is <= single; is++) {
                        self += fac[ia + ib + ic + is] / fac[ia] / fac[ib] / fac[ic] * (fac[single] / fac[is] / fac[single - is]);
                    }
                }
            }
        }
        return self;
    }
}
