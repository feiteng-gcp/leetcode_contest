class Solution {
    public int numTilePossibilities(String tiles) {
        Set<String> allWords = new HashSet<>();
        perm(tiles.toCharArray(), 0, allWords);
        
        return allWords.size();
    
}
    
    private void perm(char[] array, int start, Set<String> allWords) {
        if (start == array.length) {
            for (int i = 1; i <= array.length; i++) {
                allWords.add(new String(array, 0, i));
            }
            return;
        }
        
        
Set<Character> visited = new HashSet<>();
        for (int i = start; i < array.length; i++) {
            if (visited.add(array[i])) {
                swap(array, start, i);
                perm(array, start + 1, allWords);
                swap(array, start, i);
            }
        }
    
}
    
    private void swap(char[] array, int x, int y) {
        char cur = array[x];
        array[x] = array[y];
        array[y] = cur;
    }
}
