class Solution {
    public int numTilePossibilities(String tiles) {
        Set<String> sets = new HashSet<>();
        for(int i=0; i<tiles.length(); ++i){
            sets.addAll(add(sets, tiles.substring(i, i+1)));
        }
        return sets.size();
    }
    
    private Set<String> add(Set<String> sets, String c){
        Set<String> ns = new HashSet<>();
        ns.add(c);
        for(String s : sets){
            ns.addAll(add(s, c));
        }
        return ns;
    }
    
    private Set<String> add(String s, String c){
        Set<String> set = new HashSet<>();
        for(int i=0; i<=s.length(); ++i){
            String str = s.substring(0, i) + c + s.substring(i, s.length());
            set.add(str);
        }
        return set;
    }
}
