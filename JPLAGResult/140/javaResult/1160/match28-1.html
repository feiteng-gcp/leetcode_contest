class Solution {
    public int numTilePossibilities(String tiles) {
        Map<Character, Integer> cnt = new HashMap<>();
        for(char sChar : tiles.toCharArray()) {
            cnt.put(sChar, cnt.getOrDefault(sChar, 0) + 1);
        }
        // System.out.println("== cnt " + cnt);
        List<String> strs =  getCombine("", cnt);
        Set<String> set = new HashSet(strs);
        // System.out.println("== strs " + strs);
        for(String str : strs) {
            for(int i = 1; i < tiles.length(); i++) {
                set.add(str.substring(0, i));
            }
        }
        return set.size();
    
}
    
    private List<String> getCombine(String header, Map<Character, Integer> cnt) {
        List<String> rst = new ArrayList<>();
        for(Map.Entry<Character, Integer> entry : cnt.entrySet()) {
            char eC = entry.getKey();
            int val = entry.getValue();
            if(val == 0)
                continue;
            cnt.put(eC, val - 1);
            rst.addAll(getCombine(header + eC, cnt));
            cnt.put(eC, val);
        }
        // System.out.println("== header " + header + " cnt = " + cnt + " rst " + rst);
        if(rst.isEmpty())
            rst.add(header);
        return rst;
    }
}
