class Solution {
    Set<String> set = new HashSet<>();
    public int numTilePossibilities(String tiles) {
        int n = tiles.length();
        boolean[] used= new boolean[n];
        char[] sc = tiles.toCharArray();
        Arrays.sort(sc);
        for(int i=1;i<=n;i++) {
           helper(sc,  i, "", used); 
        }
        
return set.size();
    }
    
    public void helper(char[] nums, int len, String temp, boolean[] used) {
        if(temp.length()==len) {
          //  System.out.println(temp);
            set.add(temp);
            return ;
        }
        
        for(int i=0;i<nums.length;i++) {
           if(i!=0 && nums[i]==nums[i-1] && used[i-1]==false) continue; 
            if(!used[i]){
                used[i]=true;
                temp=temp +nums[i];
                helper(nums,len,temp,used);
                used[i]=false;
                temp = temp.substring(0,temp.length()-1);
            }
        }
    }
    
    List<List<Integer>> res= new ArrayList<List<Integer>>();
    public List<List<Integer>> permuteUnique(int[] nums) {
        if(nums.length==0) return res;
        Arrays.sort(nums);
        boolean[] used= new boolean[nums.length];
        find(nums,new ArrayList<Integer>(),used);
        return res;
    }
    
    public void find(int[] nums,ArrayList<Integer> temp,boolean [] used){
        if(temp.size()==nums.length){
            res.add(new ArrayList<Integer>(temp));
            return;
        }
        
        for(int i=0;i<nums.length;i++){
            if(i!=0 && nums[i]==nums[i-1] && used[i-1]==false) continue;
            if(!used[i]){
                used[i]=true;
                temp.add(nums[i]);
                find(nums,temp,used);
                used[i]=false;
                
temp.remove(temp.size()-1);
            }
        }
    }
}
