class Solution {
    public int numTilePossibilities(String tiles) {
        int N = tiles.length();
        char[] cs = new char[N];
        Set<String> set = new HashSet<>();
        boolean[] seen = new boolean[N];
        dfs(tiles, cs, 0, seen, set);
        return set.size();

    }

    void dfs(String tiles, char[] cs, int i, boolean[] seen, Set<String> set) {
        int N = tiles.length();
        if (i != 0) {
            StringBuilder sb = new StringBuilder();
            for (int j = 0; j < i; j++) {
                sb.append(cs[j]);
            }
            set.add(sb.toString());
        }
        if (i == N) {
            return;
        }
        for (int j = 0; j < N; j++) {
            if (seen[j]) {
                continue;
            }
            cs[i] = tiles.charAt(j);
            seen[j] = true;
            dfs(tiles, cs, i + 1, seen, set);
            seen[j] = false;
        }
    }
}
