class Solution {
    boolean[] visit;
    Set<String> result;
    String tiles;
    
    public int numTilePossibilities(String tiles) {
        visit = new boolean[tiles.length()];
        result = new HashSet<>();
        this.tiles = tiles;
        dfs("", tiles.length());
        return result.size();
    }
    
    private void dfs(String now, int left){
        if(left == 0){
            return ;
        
}
        for(int i = 0; i < tiles.length(); ++i){
            if(!visit[i]){
                visit[i] = true;
                String t = now + 
tiles.charAt(i);
                result.add(t);
                dfs(t, left - 1);
                visit[i] = false;
            }
        }
    }
}
