class Solution {
    public int numTilePossibilities(String tiles) {
        HashSet<String> res = new HashSet<>();
        helper(tiles, new boolean[
tiles.length()], res, "");
        return res.size()-1;
    }

    private void helper(String tiles, boolean[] seen, HashSet<String> res, String s) {
        res.add(s);
        for(int i = 0; i < tiles.length(); i++) {
            if(seen[i]) {
                continue;
            
}
            seen[i] = true;
            helper(tiles, seen, res, s + tiles.charAt(i));
            seen[i] = false;
        }
    }
}
