class Solution {
    public int numTilePossibilities(String tiles) {
        Set<String> st = new HashSet<>();
        boolean[] flag = new boolean[tiles.length()];
        dfs(tiles, st, "", flag);
        return st.size()-1;
    }
    public void dfs(String tiles, Set<String> st, String cur, boolean[] flag){
        st.add(cur);
        for(int i=0; i<tiles.length(); i++){
            if(flag[i]) continue;
            if(i>0&&!flag[i-1]&&tiles.charAt(i)==tiles.charAt(i-1)) continue;
            flag[i] = true;
            dfs(tiles, st, cur+tiles.charAt(i), flag);
            flag[i] = false;
        }
    }
}
