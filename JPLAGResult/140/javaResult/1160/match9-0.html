class Solution {
    public int numTilePossibilities(String tiles) {
        int res = 0;
        char[] arr = tiles.toCharArray();
        for(int len = 1; len <= arr.length; len++) {
            res += helper(arr, 0, len);
        }
        return res;
    }
    
    private int helper(char[] arr, int index, int len) {
        if(index == len) {
            return 1;
        }else {
            int res = 0;
            Set<Character> set = new HashSet<>();
            for(int i = index; i < arr.length; i++) {
                if(set.add(arr[i])) {
                    swap(arr, index, i);
                    res += helper(arr, index + 1, len);
                    swap(arr, index, i);
                }
            }
            return res;
        }
    }
    
    private void swap(char[] arr, int i, int j) {
        char temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}
