class Solution {
    public int numTilePossibilities(String tiles) {
        Set<String> set = new HashSet<>();
        int[] dict = new int[26]; 
        for (char ch : tiles.toCharArray()) dict[ch - 'A']++;
        dfs(set, dict, "");
        return set.size();
    }
    private void dfs(Set<String> set, int[] dict, String s) {
        if (s.length() != 0) set.add(s);
        for (int i = 0; i < 26; i++) {
            if (dict[i] > 0) {
                dict[i]--;
                dfs(set, dict, s + (char)(i + 'A'));
                dict[i]++;
            }
        }
    }
}
