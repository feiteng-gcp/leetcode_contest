class Solution {
public int numTilePossibilities(String tiles) {
        Set<String> set = new HashSet<>();
        dfs(tiles, new boolean[1000], new StringBuilder(), set);
        return set.size();
    }

    private void dfs(String tiles, boolean[] used, StringBuilder sb, Set<String> set) {
        for (int i = 0; i < tiles.length(); i++) {
            char ch = tiles.charAt(i);
            if (used[i]) continue;
            used[i] = true;
            sb.append(ch);
            set.add(sb.toString());
            dfs(tiles, used, sb, set);
            sb.deleteCharAt(sb.length() - 1);
            used[i] = false;
        }
    }
}
