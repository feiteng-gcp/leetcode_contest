class Solution {
    public int numTilePossibilities(String tiles) {
        if (tiles == null || tiles.length() == 0) {
            return 0;
        }
        
        char[] tilesArr = tiles.toCharArray();
        Set<String> answers = new HashSet<>();
        
        dfs(tilesArr, "", answers, new boolean[tilesArr.length]);
        
        
return answers.size();
    }
    
    private void dfs(char[] tilesArr, String cur, Set<String> answers, boolean[] visited) {        
        for (int i = 0; i < tilesArr.length; ++i) {
            if (visited[i]) {
                continue;
            }
            
            String ne = cur + String.valueOf(tilesArr[i]);
            
            if (!answers.contains(ne)) {
                answers.add(ne);
            
}

            visited[i] = true;
            dfs(tilesArr, ne, answers, visited);
            visited[i] = false;
        }
    }
}
