class Solution {
    int ans = 0;
    Set<String> set = new HashSet();
    
    public int numTilePossibilities(String tiles) {
        set = new HashSet();
        ans = 0;
        char[] chs = 
tiles.toCharArray();
        dfs(chs, "");
        return set.size();
    }
    
    void dfs(char[] chs, String path) {
        if (path.length() > 0) {
            set.add(path);
        }
        for (int i = 0; i < chs.length; i++) {
            char pre = chs[i];
            
if (pre == '#') continue;
        
            chs[i] = '#';
            
            dfs(chs, path + pre);
            chs[i] = pre;
        }
    }
}
