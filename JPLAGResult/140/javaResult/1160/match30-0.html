class Solution {
    public int numTilePossibilities(String tiles) {
        Set<String> set = new HashSet<>();
        helper(tiles.toCharArray(), new ArrayList<>(), set, "");
        return set.size();
    }
    
    public void helper(char[] cs, List<Integer> has, Set<String> set, String s){
        if(s.length() != 0)set.add(s);
        for(int i=0;i<cs.length;i++){
            if(!has.contains(i)){
                has.add(i);
                helper(cs, has, set, s+""+cs[i]);
                has.remove(has.size()-1);
            }
        }
    }
}
