class Solution {
    public int numTilePossibilities(String tiles) {
        if (tiles == null) {
            return 0;
        }
        char[] chars = tiles.toCharArray();
        boolean[] visited = new boolean[chars.length];
        Set<String> set = new HashSet<>();
        dfs(chars, visited, set, new StringBuilder());
        return set.size();
    }
    
    private void dfs(char[] chars, boolean[] visited, Set<String> set, StringBuilder sb) {
        for (int i = 0; i < chars.length; i++) {
            if (!visited[i]) {
                visited[i] = true;
                sb.append(chars[i]);
                set.add(sb.toString());
                dfs(chars, visited, set, sb);
                sb.deleteCharAt(sb.length() - 1);
                visited[i] = false;
            }
        }
    }
}
