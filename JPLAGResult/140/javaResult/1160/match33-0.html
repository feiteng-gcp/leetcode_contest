class Solution {
    void getPerm(StringBuilder result, Set<String> outcome, int i) {
        if (i >= result.length()) {
            if (result.length() > 0) {
                outcome.add(result.toString());
            }
            return;
        }        
        
        for (int j = i; j < result.length(); j++) {
            swap(result, i, j);
            getPerm(result, outcome, i+1);
            swap(result, i, j);
        }
    }
    
    void swap(StringBuilder sb, int i, int j) {
        char c = sb.charAt(i);
        sb.setCharAt(i, sb.charAt(j));
        sb.setCharAt(j, c);
    }
    
    
    void genCombinations(Set<String> outcome, String tiles, String result, int i) {
        if (i >= tiles.length()) {
            getPerm(new StringBuilder(result), outcome, 0);
            return;
        }
        
        genCombinations(outcome, tiles, result + tiles.charAt(i), i+1);
        genCombinations(outcome, tiles, result, i+1);
    }
    
    public int numTilePossibilities(String tiles) {
        Set<String> outcome = new HashSet<>();
        genCombinations(outcome, tiles, "", 0);
        
        return outcome.size();
    }
}
