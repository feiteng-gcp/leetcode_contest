class Solution {
    
    public int numTilePossibilities(String tiles) {
        Set<String> seen = new HashSet<>();
        boolean[] visited = new boolean[tiles.length()];
        dfs(
tiles.toCharArray(), visited, new StringBuilder(), seen);
        return seen.size()-1;
    }
    
    private void dfs(char[] tiles, boolean[] visited, StringBuilder sb, Set<String> seen){
        seen.add(sb.toString());
        for(int i=0; i<tiles.length; i++){
            if(!visited[i]){
                visited[i] = true;
                
sb.append(tiles[i]);
                dfs(tiles, visited, sb, seen);
                sb.deleteCharAt(sb.length()-1);
                visited[i] = false;
            }
        }
    }
}
