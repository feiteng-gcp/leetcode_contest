class Solution {
    public int numTilePossibilities(String tiles) {
        char[] chs = tiles.toCharArray();
        Set<String> set = new HashSet<>();
        boolean[] used = new boolean[chs.length];
        dfs(chs,"",used,set);
        return set.size();
    }
    
    private void dfs(char[] chs, String tmp, boolean[] used, Set<String> set) {
        if(!tmp.isEmpty()) {
            set.add(tmp);
        }
        for(int i=0; i<chs.length; i++) {
            if(used[i]) {
                continue;
            }
            used[i] = true;
            dfs(chs, tmp + chs[i], used, set);
            used[i] = false;
        }
    }
}
