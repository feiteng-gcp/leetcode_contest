class Solution {
    private int count;
    public int numTilePossibilities(String tiles) {
        count = 0;
        helper(tiles, new HashSet<String>(), new StringBuilder(), new boolean[tiles.length()]);
        return count;
    }
    
    private void helper(String tiles, Set<String> added, StringBuilder sb, boolean[] visited) {
        if (sb.length() > 0 && added.add(sb.toString())) {
            count++;
        }
        
        for (int i = 0; i < tiles.length(); i++) {
            // pick
            if (!visited[i]) {
                visited[i] = true;
                sb.append(tiles.charAt(i));
                helper(tiles, added, sb, visited);
                sb.deleteCharAt(sb.length() - 1);
                visited[i] = false;
            }
            
            // not pick
            // helper(tiles, added, sb, visited);
        }
    }
}
