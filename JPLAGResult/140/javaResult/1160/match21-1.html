class Solution {
    private int count;
    public int numTilePossibilities(String tiles) {
        int[] t = new int[26];
        for(char c : tiles.toCharArray()) {
            t[c-'A'] += 1;
        }
        count = 0;
        dfs(t);
        return count;
    }
    private void dfs(int[] t) {
        if(noMore(t))   return;
        
for(int i = 0; i < 26; ++i) {
            if(t[i] > 0) {
                count += 1;
                t[i] -= 1;
                dfs(t);
                t[i] += 1;
            }
        }
    }
    private boolean noMore(int[] t) {
        for(int i = 0; i < 26; ++i) if(t[i] > 0)    return false;
        return true;
    }
}
