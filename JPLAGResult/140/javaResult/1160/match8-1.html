class Solution {
    Set<String> set=new HashSet<>();
    public int numTilePossibilities(String tiles) {
        char[] tc=tiles.toCharArray();
        backtrack(tc,"",set,0);
        return set.size()-1;
    }
    private void backtrack(char[] tc,String tmp,Set<String> set,int s){
        if(set.contains(tmp)){
            return;
        }
        set.add(tmp);
        if(s==tc.length){
            return;
        }
        for(int i=0;i<tc.length;i++){
            if(tc[i]!=' '){
                tmp+=tc[i];
                tc[i]=' ';
                backtrack(tc,tmp,set,s+1);
                tc[i]=tmp.charAt(tmp.length()-1);
                tmp=tmp.substring(0,tmp.length()-1);
            }  
        }
    }
}
