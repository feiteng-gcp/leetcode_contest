class Solution {
    public int numTilePossibilities(String tiles) {
        int[] letters = new int[26];
        for(int i=0; i<tiles.length(); i++){
            letters[tiles.charAt(i) -'A'] ++;
        }
        Set<String> set = new HashSet<>();
        search(letters, tiles.length(), set, "");
        return set.size() - 1;
    }
    private void search(int[] letters, int i, Set<String> set, String cur){
        if(i == 0) {
            set.add(cur);
            return ;
        }
        search(letters, i-1 ,set, cur);


        for(int c=0; c<26; c++){
            if(letters[c] >0){
                letters[c] --;
                String newStr = cur + (char)(c+'A');
                search(letters, i-1 ,set, newStr);
                letters[c] ++;
            }
        }
    }
}
