class Solution {
    int res = 0;
    public int numTilePossibilities(String tiles) {
        char[] array = tiles.toCharArray();
        Arrays.sort(array);
        String tile = new String(array);
        dfs(tile, "", new boolean[tiles.length()], new HashSet<String>());
        
return res;
    }
    
    public void dfs(String tile, String str, boolean[] visited, Set<String> set) {
        if(!set.contains(str) && !str.equals("")) {
            set.add(str);
            res++;
        
}
        
        for(int i = 0; i < tile.length(); i++) {
            if(!visited[i]) {
                visited[i] = true;
                dfs(tile, str + tile.charAt(i), visited, set);
                visited[i] = false;
            }
        }
    }
}
