class Solution {
    HashSet<String> set = new HashSet<>();
    public int numTilePossibilities(String tiles) {
        dfs(0, new boolean[tiles.length()], tiles.toCharArray(), "");
        
return set.size()-1;
    }
    
    public void dfs(int occupy, boolean[] visited, char[] chars, String curr){
        if(!set.add(curr) || occupy == visited.length) return;
        for(int i = 0; i < visited.length; i++){
            if(visited[i]) continue
;
            visited[i] = true;
            dfs(occupy+1, visited, chars, curr+chars[i]);
            visited[i] = false;
        }
    }
}
