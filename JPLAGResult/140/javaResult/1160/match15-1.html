class Solution {
    public int numTilePossibilities(String tiles) {
        char[] tmp=tiles.toCharArray();
        Arrays.sort(tmp);
        count=0;
        dfs(tmp,new boolean[tiles.length()],0);
        return   count-1;
    }
    int count;
    
    public void dfs(char[] s1,boolean[] visit,int cur) {
        count++;
        if(cur==s1.length ) 
return;
        for(int i=0;i<s1.length;i++) {
            if(visit[i] || i>0 && !visit[i-1] &&  s1[i-1] == s1[i]) continue;
            visit[i]=true;
            
            dfs(s1,visit,cur+1);
            visit[i] = false; 
        }
    }
}
