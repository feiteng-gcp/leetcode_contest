	class Solution {
		
		Set<String> set = new HashSet<>();
		
	    public int numTilePossibilities(String tiles) {
	    	char[] s = tiles.toCharArray();
	    	dfs(s, 0, "");
	    	return set.size()-1;
	    }
	    
	    void dfs(char[] s, int ptn, String cur)
	    {
	    	set.add(cur);
	    	for(int i = 0;i < s.length;i++){
	    		if(ptn<<~i>=0){
	    			dfs(s, ptn|1<<i, cur + s[i]);
	    		}
	    	}
	    }
	}
