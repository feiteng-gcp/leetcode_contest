class Solution {
    public int numTilePossibilities(String tiles) {
        Map<Character, Integer> hash = new HashMap<>();
        for (char c : tiles.toCharArray()) {
            hash.put(c, hash.getOrDefault(c, 0) + 1);
        }
        int[] cnts = new int[hash.size()];
        int idx = 0;
        for (Map.Entry<Character, Integer> cur : hash.entrySet()) {
            cnts[idx++] = cur.getValue();
        }
        return dfs(cnts, tiles.length()) - 1;
    }
    
    private int dfs(int[] cnts, int nleft) {
        if (nleft == 0) {
            return 1;
        }
        int ret = 1;
        for (int i = 0; i < cnts.length; i++) {
            if (cnts[i] == 0) {
                continue;
            }
            cnts[i] -= 1;
            ret += dfs(cnts, nleft - 1);
            cnts[i] += 1;
        }
        return ret;
    }
}
