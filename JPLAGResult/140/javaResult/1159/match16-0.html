class Solution {
    public String smallestSubsequence(String text) {
        int[] map = new int[26];
        char[] t = text.toCharArray();        
        for(int i=0; i<t.length; i++){
            map[t[i]-'a']++;
        }
        
        boolean[] instack = new boolean[26];
        Stack<Character> s = new Stack<>();
        
        for(int i=0; i<t.length; i++){
            if(instack[t[i]-'a']){
                map[t[i]-'a']--;
                continue;
            }
            while(!s.empty() && s.peek() > t[i] && map[s.peek()-'a'] > 0){
                instack[s.peek() - 'a'] = false;
                s.pop();
            }
            s.push(t[i]);
            instack[t[i]-'a'] = true;    
            map[t[i]-'a']--;
        }
        String ret = "";
        while(!s.empty()){
            ret = s.pop() + ret;
        }
        return ret;
    }
}
