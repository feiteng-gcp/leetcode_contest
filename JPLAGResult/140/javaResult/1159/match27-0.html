class Solution {
    public String smallestSubsequence(String text) {
        Map<Integer, List<Integer>> map = new HashMap<Integer, List<Integer>>();
        for(int i=0; i<text.length(); i++){
            int cur = text.charAt(i) - 'a';
            if(!map.containsKey(cur)){
                map.put(cur, new ArrayList<Integer>());
            }
            map.get(cur).add(i);
        }
        String result = "";
        int start =  0;
        while(map.size()>0){
            int minLast = getMinLast(map);
            for(int i=0; i<26; i++){
                if(map.containsKey(i) && map.get(i).get(0)<=minLast){
                    boolean flag = false;
                    for(int num : map.get(i)){
                        if(num>=start && num<=minLast){
                            result += (char)(i+'a');
                            map.remove(i);
                            start = num;
                            flag = true;
                            break;
                        }
                    }
                    if(flag) break;

                }
            }
        }
        return result;
    }
    private int getMinLast(Map<Integer, List<Integer>> map){
        int result = 2000;
        for(int i=0; i<26; i++){
            if(map.containsKey(i)){
                result = Math.min(result, map.get(i).get(map.get(i).size()-1));
            }
        }
        return result;
    }
}
