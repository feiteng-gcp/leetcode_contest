class Solution {
        public String smallestSubsequence(String text) {
        List<Integer>[] lsp = new List[26];
        for(int i=0;i<26;++i)
            lsp[i] = new ArrayList<>();
        boolean[] appear = new boolean[26];
        for(int i=0;i<text.length();++i){
            int c = text.charAt(i)-'a';
            lsp[c].add(i);
            appear[c]=true;
        }
        StringBuilder sb = new StringBuilder();
        int[] pos = new int[26];
        int p = 0;
        for(int l=0;l<26;++l){
            for(int i=0;i<26;++i){
                if(appear[i]==false)
                    continue;
                int temp = -1;
                for(int j=0;j<lsp[i].size();++j){
                    temp = lsp[i].get(j);
                    if(temp>p)
                        break;
                }
                boolean canUse = true;
                if(temp<p)
                    return "error";
                for(int j=0;j<26;++j){
                    if(appear[j]==false)
                        continue;
                    int lastPos = lsp[j].get(lsp[j].size()-1);
                    if(lastPos<temp){
                        canUse=false;
                        break;
                    }
                }
                if(canUse){
                    sb.append((char)(i+'a'));
                    appear[i]=false;
                    p=temp;
                    break;
                }
            }
        }
        return sb.toString();
    }
}
