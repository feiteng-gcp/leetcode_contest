class Solution {
    
    public String smallestSubsequence(String text) {
        int[] count = new int[26];
        int pos = 0;
        for (int i = 0; i < text.length(); i++) {
            count[text.charAt(i) - 'a']++;
        }
        
for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) < text.charAt(pos)) {
                pos = i;
            }
            if (--count[text.charAt(i) - 'a'] == 0) {
                break;
            }
        }
        return 
text.length() == 0 ? "" : text.charAt(pos) + smallestSubsequence(text.substring(pos + 1).replaceAll("" + text.charAt(pos), ""));
    }
}
