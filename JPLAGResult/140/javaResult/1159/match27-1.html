class Solution {
    public String smallestSubsequence(String text) {


		List<Set<Character>> list = new ArrayList<>();

		for (int i = 0; i < text.length(); i++) {
			list.add(new HashSet<>());
		}

		Set<Integer> used = new HashSet<>();

		int[] s = new int[text.length()];

		for (int i = text.length() - 1; i >= 0; i--) {
			if (i == text.length() - 1) {
				list.get(i).add(text.charAt(i));
			} else {
				list.get(i).add(text.charAt(i));
				list.get(i).addAll(list.get(i + 1));
			}
			s[i] = list.get(i).size();
		}

		int count = list.get(0).size();
		StringBuilder sb = new StringBuilder();

		int start = 0;
		for (int i = 0; i <
 count; i++) {
			int index = -1;
			for (int j = 0; j < 26; j++) {
				if (used.contains(j)) {
					continue;
				}
				boolean flag = false;
				for (int k = start; k < s.length; k++) {
					if (text.charAt(k) - 'a' == j && s[k] >= count - i && check(list.get(k), used, count)) {
						sb.append((char) (j + 'a'));
						used.add(j);

						flag = true;
						index = k;
						break;
					}
				}
				if (flag) {
					break;
				}
			}
			start = index + 1;

		}

		return sb.toString();
	}

	private boolean check(Set<Character> s1, Set<Integer> s2, int count) {
		Set<Character> ss = new HashSet<>();
		ss.addAll(s1);
		for (Integer x : s2) {
			ss.add((char) (x + 'a'));
		}

		return ss.size() == count;
	}
}
