class Solution {
    public String smallestSubsequence(String text) {
        Set<Character> s= new HashSet<>();
        for(char c: text.toCharArray()) s.add(c);
        return helper(text, s);
    }
    
    
    public String helper(String text, Set<Character> s) {
        //System.out.println(s);
        if(s.size()==0) return "";
        Map<Character, Integer> m= new HashMap<>();
        int count= s.size();
        char ch='z';
        int index= -1;
        for(int i= text.length()-1; i>=0;i--){
            char c= text.charAt(i);
            if(s.contains(c) ){
                int t=0;
                if(m.containsKey(c)){
                    t= m.get(c);
                    m.remove(c);
                }
                if(m.size()== count-1){
                    if(  c <= ch ){
                        ch=c;
                        index=i;
                    }
                }
                m.put(c, t+1);
            }
        }
        //System.out.println(ch);
        s.remove(ch);
        return ch+ helper(text.substring(index+1),  s);
        
    }
    
    
    
}
