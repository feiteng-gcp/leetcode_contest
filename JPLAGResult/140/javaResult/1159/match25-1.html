class Solution {
    public String smallestSubsequence(String text) {
        Set<Character> all = new HashSet<>(256);
        for (int i = text.length() - 1; i >= 0; i--) {
            all.add(text.charAt(i));
        }
        int size = all.size();
        char[] result = new char[size];
        Set<Character> used = new HashSet<>(256);
        int start = 0;
        for (int index = 0; index < size; index++) {
            Set<Character> right = new HashSet<>(256);
            
for (int i = text.length() - 1; i >= 0; i--) {
                char c = text.charAt(i);
                if (used.contains(c)) {
                    continue;
                }
                right.add(c);
                if (right.size() == size - index) {
                    char min = 'z';
                    for (int j = start; j <= i; j++) {
                        if (min > text.charAt(j) && !used.contains(text.charAt(j))) {
                            min = text.charAt(j);
                            start = j;
                        }
                    }
                    result[index] = min;
                    used.add(min);
                    break;
                }
            }
        }
        return String.valueOf(result);
    }
}
