class Solution {
    public String smallestSubsequence(String text) {
        char ch[] = text.toCharArray();
        int n = ch.length;
        Set<Character> vis = new HashSet<>();
        for (char c : ch) {
            vis.add(c);
        }
        int m = vis.size();
        Set<Character> add = new HashSet<>();
        StringBuilder sb = new StringBuilder();
        int s = 0;
        for (int l = m; l > 0; l--) {
            int idx = n - 1;
            Set<Character> tmp = new HashSet<>();
            while (idx >= s) {
                if (!add.contains(ch[idx])) {
                    tmp.add(ch[idx]);
                }
                if (tmp.size() == l) {
                    break;
                }
                idx--;
            }
            int minI = -1;
            for (int i = idx; i >= s; i--) {
                if (!add.contains(ch[i]) && (minI == -1 || ch[i] <= ch[minI])) {
                    minI = i;
                }
            }
            sb.append(ch[minI]);
            add.add(ch[minI]);
            s = minI + 1;
        }
        return sb.toString();
    }
}
