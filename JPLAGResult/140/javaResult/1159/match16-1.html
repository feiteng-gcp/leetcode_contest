class Solution {
    public String smallestSubsequence(String text) {
      int[] letterCount = new int[26];
      char[] letters = text.toCharArray();
      for(int i = 0; i < letters.length; i++){
        letterCount[letters[i] - 'a']++;
      }
      
      //create new stringbuilder and append first letter
      boolean[] usedLetters = new boolean[26]; 
      StringBuilder sb = new StringBuilder();
      
      for(int i = 0; i < letters.length; i++){
        int curLast = sb.length() - 1;
        if(curLast > -1){
          char prev = sb.charAt(curLast);
          while(curLast > -1 && usedLetters[letters[i] - 'a'] == false && prev > letters[i] && letterCount[prev - 'a'] > 0){
            sb.deleteCharAt(curLast--);
            usedLetters[prev - 'a'] = false;
            if(curLast > -1) prev = sb.charAt(curLast);
          }
        }
        
        if(usedLetters[letters[i] - 'a'] == false){
          sb.append(letters[i]);
          usedLetters[letters[i] - 'a'] = true;
        }
        letterCount[letters[i] - 'a']--;
      }
      
      return sb.toString();
    }
}
