class Solution {
    public String smallestSubsequence(String text) {
        if(text.length() == 0) return text;
        int[] cnt = new int[26];
        for(char x : text.toCharArray()) ++cnt[x - 'a'];
        StringBuilder sb = new StringBuilder();
        int id = 0;
        int lb = 0;
        boolean[] vst = new boolean[26];
        while(id < text.length()) {
            char cx = text.charAt(id);
            if(vst[cx - 'a']) {
                ++id;
                continue;
            }
            --cnt[cx - 'a'];            
            if(cnt[cx - 'a'] == 0) {
                int lt = find(text, lb, id, vst);
                System.out.println(text.charAt(lt));
                System.out.println(lt);
                while(text.charAt(lt) != cx) {
                    char ay = text.charAt(lt);
                    sb.append(ay);
                    vst[ay - 'a'] = true;
                    lt = find(text, lt + 1, id, vst);
                    System.out.println(text.charAt(lt));
                System.out.println(lt);
                }
                sb.append(cx);
                vst[cx - 'a'] = true;
                lb = lt + 1;
                System.out.println(sb.toString());
            }
            ++id;
        }
        return sb.toString();
    }
    int find(String x, int s, int e, boolean[] vst) {
        char xm = '@';
        int id = -1;
        for(int i = s; i <= e; ++i) {
            if(vst[x.charAt(i) - 'a']) continue;
            char cx = x.charAt(i);
            if(cx < xm || xm == '@') {
                xm = cx;
                id = i;
            }
        }
        return id;
    }
}
