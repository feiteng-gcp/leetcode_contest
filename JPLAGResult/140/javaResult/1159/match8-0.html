class Solution {
    public String smallestSubsequence(String text) {
        if (text == null || text.length() == 0) {
            return "";
        }
        int[] count = new int[26];
        for (char c : text.toCharArray()) {
            count[c - 'a']++;
        }
        int pos = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(pos) > text.charAt(i)) {
                pos = i;
            }
            if (--count[text.charAt(i) - 'a'] == 0) {
                break;
            }
        }
        return text.charAt(pos) + smallestSubsequence(text.substring(pos + 1).replaceAll(String.valueOf(text.charAt(pos)), ""));
    }
}
