/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        return helper(root, limit, 0);
    }
    
    public TreeNode helper(TreeNode node, int limit, int cur) {
        if (node.left == null && node.right == null) { // leaf
            if (cur + node.val < limit) {
                return null;
            }
        } else {
            if (node.left != null)
                node.left = helper(node.left, limit, cur + node.val);
            if (node.right != null)
                node.right = helper(node.right, limit, cur + node.val);
            
            if (node.left == null && node.right == null) return null;
        }
        return node;
    }
}
