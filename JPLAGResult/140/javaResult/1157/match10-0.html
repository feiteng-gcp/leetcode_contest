/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        if(root == null) return root;
        if(solve(root, limit, 0)){
            return root;        
        } else {
            return null;
        }
    }
    
    boolean solve(TreeNode root, int limit, int cost) {
        if(root == null) {
            if(cost < limit) {
                return false;
            } else {
                return true;
            }
        }
        boolean leftOk = solve(root.left, limit, cost + root.val);
        if(leftOk == false) {
            root.left = null;
        }
        boolean rightOk = solve(root.right, limit, cost + root.val);
        if(rightOk == false) {
            root.right = null;
        }
        if(leftOk == false && rightOk == false) {
            return false;
        } else {
            return true;
        }
    }
    
}
