/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode sufficientSubset(TreeNode node, int limit) {
        int min = go(node, limit, 0);
        if (min < limit) return null;
        return node;
    }
    
    private int go(TreeNode node, int limit, int sofar) {
        if (node == null) return sofar;
        int left = go(node.left, limit, sofar + node.val);
        int right = go(node.right, limit, sofar + node.val);
        if (left < limit && right >= limit) {
            node.left = null;
            return right;
        }
        if (right < limit && left >= limit) {
            node.right = null;
            return left;
        }
        
        return Math.min(left, right);
    }
}
