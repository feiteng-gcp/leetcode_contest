/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        
if(solve(root, 0, limit)){
            return root;
        }else{
            return null;
        }
    }
    
    public boolean solve(TreeNode root, int pathSum, int limit){
        if(root == null){
            return false;
        }
        if(root.left == null && root.right == null){
            pathSum += root.val;
            
if(pathSum < limit){
                return false;
            }else{
                return true;
            }
        }
        boolean left = solve(root.left, pathSum + root.val, limit);
        
boolean right = solve(root.right, pathSum + root.val, limit);
        if(left == false){
            root.left = null;
        }
        if(right == false){
            root.right = null;
        }
        return left == true || right == true;
    }
}
