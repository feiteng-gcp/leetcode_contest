/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        
if (dfs(root, limit, 0) < limit) {
            return null;
        }
        return root;
    }
    
    private int dfs(TreeNode node, int limit, int sum) {
        sum += node.val;
        if (node.left == null && node.right == null) {
            return sum;
        }
        int l = Integer.MIN_VALUE;
        int r = Integer.MIN_VALUE;
        if (node.left != null) {
            
l = dfs(node.left, limit, sum);
            if (l < limit) {
                node.left = null;
            }
        }
        if (node.right != null) {
            
r = dfs(node.right, limit, sum);
            if (r < limit) {
                node.right = null;
            }
        }
        return Math.max(l, r);
    }
}
