/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        return helper(root, limit, 0);
    }
    
    public TreeNode helper(TreeNode root, int limit, int sum) {
        if(root == null) return null;
        if(root.left == null && root.right == null) {
            sum += root.val;
            if(sum < limit) {
                sum -= root.val;
                return null;
            }
            else {
                sum -= root.val;
                return root;
            }
        }
        
        sum += root.val;
        TreeNode left = helper(root.left, limit, sum);
        TreeNode right = helper(root.right, limit, sum);
        if(left == null && right == null) {
            sum -= root.val;
            return null;
        }
        else {
            if(left != null) root.left = left;
            else root.left = null;
            if(right != null) root.right = right;
            else root.right = null;
            sum -= root.val;
            return root;
        }
    }
}
