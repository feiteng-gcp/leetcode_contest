/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
  public TreeNode sufficientSubset(TreeNode root, int limit) {
        if (!dfs(root, null, 0, limit)) {
            return null;
        } else {
            return root;
        }

    }
    
    private boolean dfs(TreeNode root, TreeNode pnt, int sum, int limit) {
        if (root == null) return false;
        if (root.left == null && root.right == null) {
            if (sum + root.val < limit) {
                return false;
            } else {
                return true;
            }
        } else if (root.left == null) {
            if (!dfs(root.right, root, sum + root.val, limit)) {
                root.right = null;
                return false;
            } else {
                return true;
            }
        } else if (root.right == null) {
            if (!dfs(root.left, root, sum + root.val, limit)) {
                root.left = null;
                return false;
            } else {
                return true;
            }
        } else {
            
boolean l = dfs(root.left, root, sum + root.val, limit);
            boolean r = dfs(root.right, root, sum + root.val, limit);
            
            if (!l) {
                root.left = null;
            }
            if (!r) {
                root.right = null;
            }
            if (!l && !r) {
                return false;
            }
            return true;
        }
        
    }
}
