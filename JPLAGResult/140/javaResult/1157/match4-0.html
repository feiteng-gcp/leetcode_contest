/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        long s = dfs(root, 0, limit);
        if (s < limit) {
            return null;
        }
        return root;
    }
    // return max subtree's root-path sum
    private long dfs(TreeNode curr, long sum, int limit) {
        if (curr == null) {
            return sum;
        }
        sum += curr.val;
        if (curr.left == null && curr.right == null) {
            return sum;
        }
        long l = Long.MIN_VALUE, r = Long.MIN_VALUE;
        if (curr.left != null) {
            l = dfs(curr.left, sum, limit);
            if (l < limit) {
                curr.left = null;
            }
        }
        if (curr.right != null) {
            r = dfs(curr.right, sum, limit);
            if (r < limit) {
                curr.right = null;
            }
        }
        return Math.max(l, r);
    }
}
