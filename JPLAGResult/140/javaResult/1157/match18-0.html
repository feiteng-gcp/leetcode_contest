/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        
        if (root == null) return root;
        return helper(root, 0, limit) < limit ? null : root;
        
    }
    
    private int helper(TreeNode node, int upSum, int limit){
        
        if (node == null) return 0;
        if (node.left == null && node.right == null) return node.val;
        
        int left = helper(node.left, upSum + node.val, limit);
        int right = helper(node.right, upSum + node.val, limit);
        
        if (left + upSum + node.val< limit) node.left = null;
        if (right + upSum + node.val< limit) node.right = null;
        
        return Math.max(left + node.val, right + node.val);
        
    }
    
}
