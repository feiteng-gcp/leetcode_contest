/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        int[] res = dfs(root, limit, 0);
        if (res[0] < limit && res[1] < limit) return null;
        return root;
    }
    private int[] dfs(TreeNode root, int limit, int pathSum) {
        if (root == null) return new int[] {pathSum, pathSum}
; 
        int[] leftSum = dfs(root.left, limit, pathSum + root.val);
        int[] rightSum = dfs(root.right, limit, pathSum + root.val);
        if (leftSum[0] < limit && leftSum[1] < limit) root.left = null;
        if (rightSum[0] < limit && rightSum[1] < limit) root.right = null;
        return new int[] {Math.max(leftSum[0], leftSum[1]), Math.max(rightSum[0], rightSum[1])};
    }
}
