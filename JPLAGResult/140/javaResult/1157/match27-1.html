/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        return helper(root, limit, 0);
    }
    
    private TreeNode helper(TreeNode n, int limit, int sumsofar){
        if(n == null) return null;
        
        sumsofar += n.val;
        TreeNode prevl = n.left; TreeNode prevr = n.right;
        n.left = helper(n.left, limit, sumsofar);
        n.right = helper(n.right, limit, sumsofar);                
                        
        if(n.left == null && n.right == null){
            if(prevr != null || prevl != null) return null
;
            if(sumsofar < limit) return null;
        }
        return n;        
    }
}
