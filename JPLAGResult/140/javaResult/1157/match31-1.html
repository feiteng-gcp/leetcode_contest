 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {

	public TreeNode sufficientSubset(TreeNode root, int limit) {


		if (root == null) {
			return root;
		}

		boolean valid = calc(root, root.val, limit);
		if (valid) {

			return root;
		} else {
			return null;
		}
	}

	private boolean calc(TreeNode root, int currSum, int limit) {
		if (root.left == null && root.right == null) {
			if (currSum < limit) {
				return false;
			} else {
				return true;
			}
		} else if (root.left != null && root.right == null) {
			boolean left = calc(root.left, currSum + root.left.val, limit);
			if (!left) {
				root.left = null;
				return false;
			} else {
				return true;
			}
		} else if (root.right != null && root.left == null) {
			boolean right = calc(root.right, currSum + root.right.val, limit);
