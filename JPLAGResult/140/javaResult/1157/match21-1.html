/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {

    public TreeNode sufficientSubset(TreeNode root, int limit) {
        int result = helper(root, limit, 0);
        if (result < limit) {
            return null;
        } else {
            return root;
        }
    
}
    
    public int helper(TreeNode node, int limit, int preSum) {
        preSum += node.val;
        if(node.left == null && node.right == null) {
            return preSum;
        }
        int max = Integer.MIN_VALUE;
        if (node.left != null) {
            int lr = helper(node.left, limit, preSum);
            max = Math.max(lr, max);
            if (lr < limit) {
                node.left = null;
            }
        }
        if (node.right != null) {
            int rr = helper(node.right, limit, preSum);
            max = Math.max(rr, max);
            if (rr < limit) {
                node.right = null;
            }
        }
        return max;
    }
}
