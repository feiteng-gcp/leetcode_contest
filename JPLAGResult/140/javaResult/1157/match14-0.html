/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Map<TreeNode, Integer> map = new HashMap<>();
    
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        if (mps(root, 0, limit) < limit) {
            return null;
        }
        return root;
    }
    
    int mps(TreeNode root, int cur, int limit) {
        if (root == null) return cur;
        int left = mps(root.left, cur + root.val, limit);
        int right = mps(root.right, cur + root.val, limit);
        
        if (left < limit) {
            root.left = null;
        }
        if (right < limit) {
            root.right = null;
        }
        
        cur = Math.max(left, right);
        
        return cur;
    }
}
