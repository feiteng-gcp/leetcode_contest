/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        
        boolean r= dfs(root, 0, limit);
        if(r) return null;
        return root;
    }
    
   // return true is all it's children returns true, which means smaller than limit. 
   public boolean dfs(TreeNode node, int s, int limit){
       if(node == null) return s < limit;
       boolean l =  dfs( node.left, s+ node.val, limit);
       boolean r =  dfs( node.right, s+ node.val, limit);
       if(l) node.left= null;
       if(r) node.right= null;
       return l&& r;
       
   } 
    
}
