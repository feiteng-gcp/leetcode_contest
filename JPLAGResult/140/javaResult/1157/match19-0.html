/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        int best = helper(root, limit, 0);
        if(best < limit) return null;
        return root;
    }
    
    private int helper(TreeNode root, int limit, int sum) {
        if(root == null) {
            return sum;
        }
        int leftSum = helper(root.left, limit, sum + root.val);
        int rightSum = helper(root.right, limit, sum + root.val);
        if(leftSum < limit) root.left = null;
        if(rightSum < limit) root.right = null;
        return Math.max(leftSum, rightSum);
    }
}
