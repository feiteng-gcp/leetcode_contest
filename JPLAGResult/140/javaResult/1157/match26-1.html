/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        return helper(root, limit, 0);
    }
    TreeNode helper(TreeNode x, int y, int acc) {
        if(x.left == null && x.right == null) {
            int zx = acc + x.val;
            if(zx < y) return null;
            return x;
        }
        acc += x.val;
        
if(x.left != null) x.left = helper(x.left, y, acc);
        if(x.right != null) x.right = helper(x.right, y, acc);
        if(x.left == null && x.right == null) return null;
        return x;
    }
}
