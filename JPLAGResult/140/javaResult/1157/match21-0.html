/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        if (root == null) {
            return null;
        }
        if (helper(root, limit, 0) < limit) {
            return null;
        }
        return root;
    }

    int helper(TreeNode root, int limit, int sum) {
        sum += root.val;
        if (root.left == null && root.right == null) {
            return root.val;
        }
        int max = Integer.MIN_VALUE;
        if (root.left != null) {
            int lv = helper(root.left, limit, sum);
            max = Integer.max(lv, max);
            if (lv + sum < limit) {
                root.left = null;
            }
        }
        if (root.right != null) {
            int rv = helper(root.right, limit, sum);
            max = Integer.max(rv, max);
            if (rv + sum < limit) {
                root.right = null;
            }
        }
        return max + root.val;
    }
}
