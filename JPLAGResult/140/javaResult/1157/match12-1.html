/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        return helper(root,limit,0);
    }
    
    public TreeNode helper(TreeNode root, long limit, long curr){
        if(root==null) return null;
        long newC = curr+root.val;
        if(root.left==null && root.right==null)return newC<limit?null:root
;
        
        root.left = helper(root.left,limit,newC);
        root.right = helper(root.right,limit,newC);
        if(root.left==null && root.right==null) return null;
        return root;
    }
}
