/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    boolean dfs(TreeNode root, long cur, long limit) {
        if (null == root) {
            return false;
        }
        cur += root.val;
        if (null == root.left && null == root.right) {
            return cur >= limit;
        }
        boolean lft = dfs(root.left, cur, limit);
        boolean rht = dfs(root.right, cur, limit);
        if (!lft) {
            root.left = null;
        }
        if (!rht) {
            root.right = null;
        }
        return lft || rht;
    }

    public TreeNode sufficientSubset(TreeNode root, int limit) {
        boolean res = dfs(root, 0, limit);
        if (!res) {
            return null;
        }
        return root;
    }
}
