/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        if(helper(root, 0, limit) < limit){
            return null;
        }
        return root;
    }
    
    public int helper(TreeNode root, int sum, int limit){
        if(root == null){
            return 0;
        }
        if(root.left == null && root.right == null){
            return root.val;
        }
        int left = helper(root.left, sum + root.val, limit);
        int right = helper(root.right, sum + root.val, limit);
        if(sum + root.val + left < limit){
            root.left = null;
        }
        if(sum + root.val + right < limit){
            root.right = null;
        }
        return root.val + Math.max(left, right);
    }
}
