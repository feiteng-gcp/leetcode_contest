/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
        if(helper(root,limit)){
            return null;
        }
        return root;
    }
    public boolean helper(TreeNode root, int limit){
        if(root == null){
            return true;
        }
        limit = limit - root.val;
        
if(root.left == null && root.right == null){
            return limit > 0; 
        }
        boolean left = helper(root.left,limit);
        boolean right = helper(root.right,limit);
        if(left){
            root.left = null;
        }
        if(right){
            root.right = null;
        }
        return left && right;
    }
}
