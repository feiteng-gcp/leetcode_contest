/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode sufficientSubset(TreeNode root, int limit) {
      if(dfs(root,limit,0) >= limit) return root;
      return null;
    
}
  
    public int dfs(TreeNode root, int limit, int current){
      if(root.right == null && root.left == null) return current + root.val;
      
      int left = limit-1;
      int right = limit-1;
      
      if(root.left != null) left = dfs(root.left, limit, current + root.val);
      if(root.right != null) right = dfs(root.right, limit, current + root.val);
      
      if(left < limit) root.left = null;
      if(right < limit) root.right = null;
        
      return Math.max(left,right);
    }
}
