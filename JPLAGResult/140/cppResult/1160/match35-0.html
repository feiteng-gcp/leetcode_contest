class Solution {
public:
    unordered_set<string> set;
    int numTilePossibilities(string tiles) {
        vector<int> used(tiles.size(), 0);
        helper(tiles, used, "");
        return set.size();
    }
    
    void helper(string tiles, vector<int> used, string str) {
        if(str.size() != 0) {
            set.insert(str);
        }
        
        if(str.size() == tiles.size()) {
            return;
        }
        
        for(int i = 0; i < tiles.size(); i++) {
            if(used[i] == 0) {
                used[i] = 1;
                helper(tiles, used, str+tiles[i]);
                used[i] = 0;
            }
        }
    }
};
