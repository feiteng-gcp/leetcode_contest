class Solution {
public:
    int numTilePossibilities(string tiles) {
        std::sort(tiles.begin(), tiles.end());
        m_tiles = tiles;
        
        string tmp;
        dfs(tmp, 0);
        return m_seen.size();
    }
    

    void dfs(string & tmp, int mask) {
        if (tmp.size() == m_tiles.size())
            return;
        
        for (int j = 0; j < m_tiles.size(); ++j) {
            if (mask & (1 << j))
                continue;
            tmp.push_back(m_tiles[j]);
            m_seen.insert(tmp);
            dfs(tmp, mask | (1 << j));
            tmp.pop_back();
        }
    }
    
    string m_tiles;
    unordered_set<string> m_seen;
};
