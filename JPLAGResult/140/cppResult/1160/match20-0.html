class Solution {
public:
    int numTilePossibilities(string tiles) {
        if (tiles.size() == 1) return 1;
        string cur;
        unordered_set<string> result;
        vector<bool> used(tiles.size());
        
        dfs(tiles, cur, used, result);

        return result.size();
    }

    void dfs(string& tiles, string& cur, vector<bool>& used, unordered_set<string>& result) {
        if (!cur.empty()) result.insert(cur);

        for (int i = 0; i < tiles.size(); i++) {
            if (used[i]) continue;
            cur += tiles[i];
            used[i] = true;
            dfs(tiles, cur, used, result);
            used[i] = false;
            cur.pop_back();
        }
    }
};
