class Solution {
public:
    void dfs(int d, string cur, int n, unordered_set<string>& ans, 
             string& tiles, vector<bool>& used) {
        if (d == n) {
            ans.insert(cur);
            // cout << "insert:" << cur << endl;
            return;
        }
        for(int i = 0; i < tiles.size(); i++) {
            if (used[i] == false) {
                used[i] = true;
                dfs(d+1, cur+tiles[i], n, ans, tiles, used);
                used[i] = false;
            }
        }
    } 
    int help(string tiles, int n) {
        unordered_set<string> ans;
        string tmp = "";
        vector<bool> used(tiles.size(), false);
        dfs(0, "", n, ans, tiles, used);
        return ans.size();
    }
    int numTilePossibilities(string tiles) {
        int n = tiles.length();
        int ans = 0;
        sort(tiles.begin(), tiles.end());
        for(int i = 1; i <= n; i++)
            ans += help(tiles, i);
        return ans;
    }
};
