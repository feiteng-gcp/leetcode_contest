class Solution {
public:
    int numTilePossibilities(string tiles) {
      tiles_ = tiles;   
      vis_.resize(tiles_.size());
      dfs(0, "");
      return visited_.size();
    }
  void dfs(int depth, const std::string &current) {
    if (current.size()) {
      visited_.insert(current);
      }

    for (int i = 0; i < tiles_.size(); ++i) {
      if (!vis_[i]) {
        vis_[i] = true;
        dfs(i + 1, current + tiles_[i]);
        vis_[i] = false;
      }
    }
  }
  vector<bool> vis_;
  string tiles_;
  std::unordered_set<std::string> visited_;
};
