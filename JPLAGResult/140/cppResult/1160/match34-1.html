class Solution {
public:
    vector<int> v;
    int ret;
    void search(int count, int last = -1) {
        if(count != 0) ret ++;
        for(int i = 0; i < v.size(); i++) {
            if(i != last) 

            for(int j = 1; j <= v[i];j++) {
                v[i] -= j;
                //cout<<i<<" "<<j<<endl;
                search(count+j, i);
                v[i] += j;
            }
        }
        
    }
    int numTilePossibilities(string tiles) {
        map<char, int> m;
        for(auto c: tiles) {
            m[c]++;
        }
        v.resize(m.size());
        int index = 0;
        for(auto mm: m) {
            v[index++] = mm.second;
            //cout<<v[index-1]<<endl;
        }
        ret = 0;
        search(0);
        return ret;
        
    }
};
