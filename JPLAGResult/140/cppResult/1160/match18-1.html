class Solution {
public:
    

    int create(int length,int Hash[]){
        if(length == 0)
            return 1;
        int ans = 0;
        for(int i=0;i<26;i++){
            if(Hash[i]){
                Hash[i]--;
                ans += create(length-1,Hash);
                Hash[i]++;
            }
        }
        return ans;
    }
    int numTilePossibilities(string tiles) {
        int ans = 0;
        int N = tiles.size();
        
        int Hash[26];
        for(int i=0;i<26;i++)
            Hash[i] = 0;
        
        for(int i=0;i<N;i++)
            Hash[tiles[i]-'A']++;
        
        for(int i=1;i<=N;i++){
            ans += create(i,Hash);
        }
        return ans;
    }
};
