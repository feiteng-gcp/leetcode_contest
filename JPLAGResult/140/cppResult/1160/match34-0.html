class Solution {
public:
    void search(string cur, unordered_set<string>& sols, unordered_map<char, int>& remain) {
        
        sols.insert(cur);
        for (auto& kv : remain) {
            if (kv.second > 0) {
                kv.second--;
                search(cur + kv.first, sols, remain);
                kv.second++;
            }
        }
        
    }
    
    int numTilePossibilities(string tiles) {
        unordered_map<char,int> remain;
        for (auto c: tiles) {
            remain[c]++;
        }
            
        unordered_set<string> sols;
        search("", sols, remain);
        return sols.size()-1;
    }
};
