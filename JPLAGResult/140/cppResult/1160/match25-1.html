class Solution {
public:
    int numTilePossibilities(string tile) {
        vector<char> tiles;
        for(char c:tile) tiles.push_back(c);
        sort(tiles.begin(),tiles.end());
        int l=tiles.size();
        unordered_set<string> se;
        do
        {
            string s;

            for(int i=0;i<l;i++)
            {
                s+=tiles[i];
                se.insert(s);
            }
        }while(next_permutation(tiles.begin(),tiles.end()));
        return se.size();
    }
};
