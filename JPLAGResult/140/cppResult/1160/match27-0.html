class Solution {
public:
    int numTilePossibilities(string tiles) {
        vector<bool> isUsed( tiles.size(), false );
        unordered_set<string> res;
        helper( tiles, "", res, isUsed );
        return res.size();
    }
    void helper( string tiles, string curS, unordered_set<string>& res, vector<bool> isUsed ) {
        if( curS.size() > 0 ) {
            res.insert( curS );
        }
        if( curS.size() == tiles.size() ) {
            return;
        }
        for( int i = 0; i < tiles.size(); i++ ) {
            if( isUsed[i] ) continue;
            isUsed[i] = true;
            helper( tiles, curS+tiles[i], res, isUsed );
            isUsed[i] = false;
        }
    }
};
