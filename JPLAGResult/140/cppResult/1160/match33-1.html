class Solution {
public:
    int numTilePossibilities(string tiles) {
        vector<int> cnt(26, 0);
        for(auto t:tiles){
            cnt[t - 'A']++;
        }
        vector<int> num;
        for(int i = 0; i < 26; ++i){
            if(cnt[i] > 0)
                num.push_back(cnt[i]);
        }
        int size = num.size();
        int len = tiles.size();
        int ret = 0;
        for(int i = 1; i <= len; ++i) {
            dfs(num, 0, size, i, ret);
        }
            
        return ret;
    }
    
    void dfs(vector<int>& num, int clen, int size, int tlen, int& ret) {
        if(clen == tlen) {
            ret++;
            return;
        }
        

        for(int i = 0; i < size; ++i) {
            if(num[i] > 0) {
                num[i]--;
                dfs(num, clen + 1, size, tlen, ret);
                num[i]++;
            }
        }
    }
};
