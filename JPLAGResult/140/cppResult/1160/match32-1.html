class Solution {
public:
    int ans=0;
    void dfs(string& tiles,vector<bool>& visited)
    {
        ans++;

        for(int i=0;i<tiles.size();i++)
        {
            if(!visited[i]&&(i==0||tiles[i-1]!=tiles[i]||visited[i-1]))
            {
                visited[i]=true;;
                dfs(tiles,visited);
                visited[i]=false;
            }
        }
    }
    int numTilePossibilities(string tiles) {
        sort(tiles.begin(),tiles.end());
        vector<bool> visited(tiles.size(),false);
        dfs(tiles,visited);
        return ans-1;
    }
};
