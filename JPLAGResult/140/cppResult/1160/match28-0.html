class Solution {
public:
    unordered_set<string> st;
    vector<int> cnts;
    
    void dfs(string& str) {
        if (str.size())
            st.insert(str);
        for (int i = 0; i < 26; i++) {
            if (cnts[i] == 0)
                continue;
            str.push_back('A' + i);
            cnts[i]--;
            dfs(str);
            str.pop_back();
            cnts[i]++;
        }
    }
    
    int numTilePossibilities(string tiles) {
        cnts = vector<int>(26, 0);
        for (auto & c : tiles)
            cnts[c - 'A']++;
        string str = "";
        dfs(str);
        return st.size();
    }
};
