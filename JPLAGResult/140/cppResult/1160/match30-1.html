class Solution {
public:
    int numTilePossibilities(string tiles) {
        unordered_set<string> s;
		int n = tiles.size();
		vector<int> v(n);
		for ( int i = 0; i < n; ++i ) {
			v[i] = i;
		}
		
		do {
			#if 0
			for ( int i = 0; i < n; ++i ) {
				cout << v[i] << " ";
			}cout << endl;
            cout << v_to_s(v, tiles, n-1 ) << endl;
			#endif
            for ( int i = 0; i< n; ++i ) {
			    s.insert( v_to_s(v, tiles, i ) );
            }
		}while ( next_perm(v) );
		
		return s.size();
    }
	bool next_perm( vector<int> &v )
	{
		int n = v.size();
		int p = n-1;
		while ( p>0 && v[p] < v[p-1] ) {
			p--;
		}
		if ( p == 0 ) {
			return false;
		}
		int q = n-1;
		while ( v[q] < v[p-1]) {
			q--;
		}
		
		int tmp = v[p-1];
		v[p-1] = v[q];
		v[q] = tmp;
		sort( v.begin() + p, v.end() );
		
		return true;
	}
	string v_to_s( const vector<int> &v,
					const string &tiles,
                 int stop)
	{
		string res;
		for ( int i = 0; i <=stop /* <v.size()*/; ++i ) {
			res.push_back( tiles[v[i]] );
		}
		return res;
	}
};
