#include <bits/stdc++.h>
using namespace std;

class Solution {

public:
    int numTilePossibilities(string tiles) {
        set<string> S;
        int N = tiles.length();
        for (int z=1;z<(1<<N);++z) {
            string t = "";
            for (int i=0;i<N;++i)
                if ((z&(1<<i))) t += tiles[i];
            sort(t.begin(), t.end());
            do {
                S.insert(t);
            } while (next_permutation(t.begin(), t.end()));
        }
        //for (auto s : S) cout << s << endl;
        return S.size();
    }
};
