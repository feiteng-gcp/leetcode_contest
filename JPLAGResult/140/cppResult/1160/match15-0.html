class Solution {
private:
    set<string> Set;
public:
    int numTilePossibilities(string tiles) {
        string tmp;
        vector<int> visited(tiles.size(), 0);
        dfs(tiles, 0, tmp, visited);
        return Set.size();
    }
    
    void dfs(string tiles, int index, string &tmp, vector<int> &visited)
    {
        if(index >= tiles.size())
        {
            return;
        }
        for(int i = 0; i < tiles.size(); ++i)
        {
            if(visited[i] == 1) continue;
            tmp.push_back(tiles[i]);
            visited[i] = 1;
            Set.insert(tmp);
            dfs(tiles, index + 1, tmp, visited);
            visited[i] = 0;
            tmp.pop_back();
        }
        return;
    }
};
