class Solution {
public:
    int numTilePossibilities(string tiles) {
        int n = tiles.size();
        set<string> st;
        vector<int> a;
        for(int i = 1; i <= n; i++) a.push_back(i);
        string t;
        do {
            t = "";
            for(int j = 0; j<n;j++){
                t += tiles[a[j] - 1];
                st.insert(t);
            }
            
        } while (next_permutation(a.begin(), a.end()));
        return st.size();    
    }
};
