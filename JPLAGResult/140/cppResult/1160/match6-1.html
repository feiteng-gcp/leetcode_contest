class Solution {
public:
    set<string> S;
    int n;
    int vis[10];
    string str, ans;
    

    void dfs(int k) {
        S.insert(ans);
        if (k==n+1) return;      
        for(int i=0; i<n; i++) {
            if (vis[i]) continue;
            vis[i]=1;
            ans+=str[i];
            dfs(k+1);
            ans.pop_back();
            vis[i]=0;
        }
        
    }
    
    int numTilePossibilities(string tiles) {
        S.clear();
        n=tiles.size();
        str=tiles;
        dfs(0);
        return S.size()-1;
    }
};
