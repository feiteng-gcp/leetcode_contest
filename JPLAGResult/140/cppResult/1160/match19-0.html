class Solution {
public:
    void helper(string tiles, string cur, unordered_set<string> &ret, vector<bool> &visited) {
        if (cur.size()==visited.size())
            return;
        for (int i=0; i<tiles.size(); i++) {
            if (visited[i])
                continue;
            cur+=tiles[i];
            ret.insert(cur);
            visited[i]=true;
            helper(tiles, cur, ret, visited);
            visited[i]=false;
            cur.pop_back();
        }
    }
    int numTilePossibilities(string tiles) {
        unordered_set<string> ret;
        string cur;
        vector<bool> visited(tiles.size(), false);
        helper(tiles, cur, ret, visited);
        return ret.size();
    }
};
