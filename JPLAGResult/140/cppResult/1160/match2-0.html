class Solution {
    void dfs(string s, string path, unordered_set<string>& res) {
        bool have = false;
        for (char& c : s) {
            if (c != '#') {
                have = true;
                char d = c;
                c = '#';
                dfs(s, path+d, res);
                c = d;
            }
        }
        if (!have)
            res.insert(path);
    }
public:
    int numTilePossibilities(string tiles) {
        unordered_set<string> res;
        int N = tiles.size();
        for (int i=1; i<(1<<N); i++) {
            string s = "";
            for (int j=0; j<N; j++) {
                if (i & (1<<j))
                    s += tiles[j];
            }
            dfs(s, "", res);
        }
        return res.size();
    }
};
