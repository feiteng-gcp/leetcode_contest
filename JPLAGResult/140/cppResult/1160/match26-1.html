class Solution {
public:
    int numTilePossibilities(string s) {
        set<string> se;
        int n = s.size();
        vector<int> vec(n);
        for (int i = 0; i < n; i++) {
            vec[i] = i;
        }
        do {
            string t;
            for (auto x : vec) {
                t += s[x];
                se.insert(t);
            }
        } while (next_permutation(vec.begin(), vec.end()));
        return se.size();
    }
};
