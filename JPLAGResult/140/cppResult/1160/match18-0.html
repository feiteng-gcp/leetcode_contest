class Solution {
public:
    int dfs(vector<int> &r,vector<int> &used,int n,int ind){
        if(ind == n)
            return 1;
        int ans = 0;
        for(int i = 0;i < r.size();i++){
            if(used[i] < r[i]){
                used[i]++;
                ans += dfs(r,used,n,ind + 1);
                used[i]--;
            }
        }
        return ans;
    }
    int numTilePossibilities(string tiles) {
        vector<int> r(26,0),used(26,0);
        int total = 0;
        for(auto c:tiles){
            r[c-'A']++;
            total++;
        }
        int ans = 0;
        for(int i = 1;i <= total;i++)
            ans += dfs(r,used,i,0);
        return ans;
    }
};
