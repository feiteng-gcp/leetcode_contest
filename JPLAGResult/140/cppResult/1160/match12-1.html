class Solution {
public:

    void dfs(const string& tiles, int idx, string cur, unordered_set<string>& ss) {
        if (idx == tiles.size()) {
            if (!cur.empty()) {
                do {
                    ss.insert(cur);
                } while (next_permutation(cur.begin(), cur.end()));
            }
            return;
        }
        
        dfs(tiles, idx + 1, cur, ss);
        dfs(tiles, idx + 1, cur + tiles[idx], ss);
    }
    
    int numTilePossibilities(string tiles) {
        sort(tiles.begin(), tiles.end());
        unordered_set<string> ss;
        dfs(tiles, 0, "", ss);
        return ss.size();
    }
};
