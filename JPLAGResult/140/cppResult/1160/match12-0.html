class Solution {
public:
    
    vector<int> f={1,1,2,6,24,120,720,5040};
    
    void gen(int i,string& s,string c,set<string>& ans){
        if(i==s.size()){
            if(c.size()>0){
                sort(c.begin(),c.end());
                do{
                    ans.insert(c);
                }while(next_permutation(c.begin(),c.end()));
            }
            return;
        }
        gen(i+1,s,c+s[i],ans);
        gen(i+1,s,c,ans);
    }
    
    int numTilePossibilities(string s) {
        int n=s.size();
        set<string> ans;
        gen(0,s,"",ans);
        return ans.size();
    }
};
