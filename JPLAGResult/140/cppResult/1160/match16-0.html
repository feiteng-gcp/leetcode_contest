class Solution {
public:
    int numTilePossibilities(string tiles) {
        sort(tiles.begin(), tiles.end());
        set<string> exist;
        int sz = tiles.length();
        for (int i=1; i<=sz; ++i) {
            dfs(tiles, exist, string(), i, 0);
        }
        return exist.size();
    }
    void dfs(string &tiles, set<string>& exist,string pre, int up, int vis) {
        if (pre.length() == up) {
            //cout << pre << endl;
            exist.insert(pre);
            return;
        }
        for (int i=0; i < tiles.length(); ++i) {
            if (vis & (1 << (i))) {
                continue;
            }
            dfs(tiles, exist, pre+tiles[i], up, vis | (1 << (i)));
        }
    }
};
