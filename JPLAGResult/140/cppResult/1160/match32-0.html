#include <iostream>
#include <iomanip>
#include <fstream>
#include <algorithm>
#include <string>
#include <vector>
#include <set>

using namespace std;
typedef long long ll;

set <string> res;
int N;
char f[10];
bool used[10];

void flood(string s)
{
    if (res.find(s) != res.end()) return;
    res.insert(s);
    for (int i = 0; i < N; i++)
    {
        if (!used[i])
        {
            used[i] = true;
            flood (s + f[i]);
            used[i] = false;
        }
    }
}

class Solution {
public:
    int numTilePossibilities(string tiles) {
        N = tiles.length();
        for (int i = 0; i < N; i++)
            f[i] = tiles[i];
        res.clear();
        flood("");
        return res.size() - 1;
    }
};
