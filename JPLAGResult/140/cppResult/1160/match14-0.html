class Solution {
public:
  set<string> S;
  int N;
  string A;
  int numTilePossibilities(string tiles) {
    A = tiles;
    N = tiles.size();
    string tmp;
    dfs(0, tmp);
    return S.size() - 1;
  }
  void dfs(int cur, string &s) {
    if (S.find(s) != S.end()) return;
    S.insert(s);
    for (int i = cur; i < N; i++) {
      swap(A[i], A[cur]);
      s.push_back(A[cur]);
      dfs(cur + 1, s);
      s.pop_back();
      swap(A[i], A[cur]);
    }
  }
};
