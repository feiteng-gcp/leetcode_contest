/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        _limit = limit;
        bool at_least = DFS(root, root->val);
        if(at_least) return root;
        return NULL;
    }
    
protected:
    bool DFS(TreeNode* now, int sum){
        if(now->left == NULL and now->right == NULL)
            return sum >= _limit;
        bool left_ok = false, right_ok = false;
        if(now ->left != NULL){
            left_ok = DFS(now->left, sum + now->left->val);
            if(!left_ok) now->left = NULL;            
        }
        if(now->right != NULL){
            right_ok = DFS(now->right, sum + now->right->val);
            if(!right_ok) now->right = NULL;
        }
        return left_ok or right_ok;
    }
private:
    int _limit;
};
