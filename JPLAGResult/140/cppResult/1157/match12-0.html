/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    int L;
    
    bool isLeaf(TreeNode* root) {
        return root && !root->left && !root->right;
    }
    
    /* Returns true if we should delete */
    bool helper(TreeNode* root, int curSum) {
        if (!root) return false;
        
        curSum += root->val;
        if (isLeaf(root)) {
            // Are we on a sufficient path? Return true if we should delete.
            return curSum < L;
        }
        
        bool l = helper(root->left, curSum);
        bool r = helper(root->right, curSum);
        
        if (l) {
            root->left = nullptr;
        }
        
        if (r) {
            root->right = nullptr;
        }
        
        // Is this a sufficient node -- if both the l and the r are insufficient, we need to delete this node
        if (l && r || l && !root->right || !root->left && r) {
            return true;
        } else {
            return false;
        }
    }
public:
    /*
        Define this problem recursively. Use a recursive helper function to first determine if this root to leaf path is sufficient -- return a boolean. Need a curSum counter. The boolean recursive helper essentially returns true if we should delete this node, false otherwise
    */
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        L = limit;
        bool b = helper(root, 0);
        if (b) return nullptr;
        else return root;
    }
};
