/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        int sum = helper(0, root, NULL, false, limit);
        if(sum < limit) return NULL;
        return root;
    }
    
    int helper(int presum, TreeNode* root, TreeNode* parent, bool isLeft, int limit) {
        if(root == NULL) return 0;
        
        int tailsum = max(helper(presum+root->val, root->left, root, true, limit), 
                          helper(presum+root->val, root->right, root, false, limit));
        int sum = presum + root->val + tailsum;
        
        if(sum < limit) {
            if(parent == NULL) {
                return sum;
            }
            else {
                if(isLeft) {
                    parent->left = NULL;
                }
                else {
                    parent->right = NULL;
                }
            }
        }
        
        return tailsum + root->val;
    }
};
