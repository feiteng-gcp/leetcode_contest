/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool deleteNode(TreeNode* root, int cur, int limit) {
        if (!root) {
            return cur < limit;
        }
        
        cur += root->val;
        bool ret = true;
        if (deleteNode(root->left, cur, limit)) {
            delete root->left;
            root->left = NULL;
        } else {
            ret = false;
        }
        if (deleteNode(root->right, cur, limit)) {
            delete root->right;
            root->right = NULL;
        } else {
            ret = false;
        }
        return ret;
    }
    
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        if (deleteNode(root, 0, limit)) {
            //delete root;
            root = NULL;
        }
        return root;
    }
};
