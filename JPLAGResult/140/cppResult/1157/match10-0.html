/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
  TreeNode* sufficientSubset(TreeNode* root, int limit) {    
    if (dfs(root, 0, limit) < limit) return nullptr;
    return root;
  }
private:
  int dfs(TreeNode* root, int sum, int limit) {
    if (!root) return 0;
    sum += root->val;
    int l = dfs(root->left, sum, limit);
    int r = dfs(root->right, sum, limit);
    if (sum + l < limit) root->left = nullptr;
    if (sum + r < limit) root->right = nullptr;
    return max(l, r) + root->val;
  }
};
