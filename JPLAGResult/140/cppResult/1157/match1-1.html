/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int limit;
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        this->limit = limit;
        if (!go(root, 0)) return NULL;
        return root;
    }
    

    bool go(TreeNode* cur, int cur_sum) {
        if (!cur) return false;
        cur_sum += cur->val;
        if (!cur->left && !cur->right) {
            return cur_sum >= limit;
        }
        bool l = go(cur->left, cur_sum), r = go(cur->right, cur_sum);
        if (!l && !r) {
            return false;
        }
        else {
            if (!l) cur->left = NULL;
            if (!r) cur->right = NULL;
            return true;
        }
    }
};
