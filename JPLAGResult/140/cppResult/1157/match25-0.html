/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int limit;
    
    int search(TreeNode* &parent, TreeNode* &cur, int running) {
        if (!cur) {
            return 0;
        }    
        
        running += cur->val;
        int l = search(cur, cur->left, running);
        int r = search(cur, cur->right, running);
        
        int res = max(l,r) + cur->val;
        if (max(l,r) + running < limit) {
            if (cur == parent->left) {
                parent->left = NULL;
            } else {
                parent->right = NULL;
            }
        }
        
        return res;
    }
    
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        this->limit = limit;
        TreeNode* parent = new TreeNode(0);
        parent->left = root;
        search(parent, root, 0);
        return parent->left;
    }
};
