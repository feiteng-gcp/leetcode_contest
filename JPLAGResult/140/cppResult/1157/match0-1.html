/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        bool del = check(root, limit, 0);
        return del ? NULL : root;
    }
    bool check(TreeNode* root, int limit, int sum) {
        if (!root) return true;
        if (!root->left && !root->right) {
            if (sum + root->val < limit) return true;

            else return false;
        }
        
        bool left = check(root->left, limit, sum + root->val);
        bool right = check(root->right, limit, sum + root->val);
        if (left) root->left = NULL;
        if (right) root->right = NULL;
        return left && right;
    }
};
