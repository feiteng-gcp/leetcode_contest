/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
private:
    int L;
    bool helper(TreeNode*& node, int sum) {
        bool insuf = true;
        
        sum += node->val;
        
        if (!(node->left)&&!(node->right)) {
            if (sum<L) {
                node = NULL;
                return true;
            }

            else return false;
        }
        if (node->left) {
            insuf = (helper(node->left,sum)&&insuf);
        }
        if (node->right) {
            insuf = (helper(node->right,sum)&&insuf);
        }
        if (insuf) node=NULL;
        return insuf;
    }
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        L = limit;
        if (helper(root,0)) return NULL;
        return root;
    }
};
