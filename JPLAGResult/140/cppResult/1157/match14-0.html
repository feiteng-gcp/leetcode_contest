/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<TreeNode*, int> sum_to_node;
    map<TreeNode*, int> best_down_sum;

    void solve(TreeNode *node, TreeNode *parent) {
        if (node == nullptr)
            return;

        sum_to_node[node] = (parent == nullptr ? 0 : sum_to_node[parent]) + node->val;
        solve(node->left, node);
        solve(node->right, node);
        best_down_sum[node] = node->val + max(best_down_sum[node->left], best_down_sum[node->right]);
    }

    TreeNode *modify(TreeNode *node, int limit) {
        if (node == nullptr)
            return nullptr;

        if (sum_to_node[node] + best_down_sum[node] - node->val < limit)
            return nullptr;

        node->left = modify(node->left, limit);
        node->right = modify(node->right, limit);
        return node;
    }

    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        solve(root, nullptr);
        root = modify(root, limit);
        return root;
    }
};
