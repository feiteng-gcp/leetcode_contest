class Solution {
	int solve(TreeNode* root, int sum, int limit) {
		if (root == NULL) return limit - 1;

		//cout << root->val << " " << sum + root->val << "\n";

		if ((root->left == NULL) && (root->right == NULL)) {
			int ans = sum + root->val;
			return ans;
		}


		int a = limit - 1, b = limit - 1;
		if (root->left != NULL) {
			a = solve(root->left, sum + root->val, limit);
			if (a < limit) root->left = NULL;
		}
		if (root->right != NULL) {
			b = solve(root->right, sum + root->val, limit);
			if (b < limit) root->right = NULL;
		}
		return max(a, b);
	}
public:
	TreeNode* sufficientSubset(TreeNode* root, int limit) {
		int maxx = solve(root, 0, limit);
		if (maxx < limit) return NULL;
		else return root;
	}
};
