/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int callme(int top, TreeNode* root, int limit){
        if(root == NULL)    return 0;
        
        int left = callme(top+root->val, root->left, limit);
        int right = callme(top+root->val, root->right, limit);
        if(top+root->val + left < limit)
            root->left = NULL;
        if(top+root->val + right < limit)
            root->right = NULL;
        return root->val+max(left, right);
    }
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        int maxx = callme(0, root, limit);
        if(maxx < limit)
            return NULL;
        return root;
    }
};
