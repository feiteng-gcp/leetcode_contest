/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        bool isS = isSufficient( root, limit, 0 );
        return isS ? root : NULL;
    }
    bool isSufficient( TreeNode* node, int limit, int curSum ) {
        if( !node ) return false;
        curSum += node->val;
        if( !node->left && !node->right ) {
            return curSum >= limit;
        }
        bool isL = isSufficient( node->left, limit, curSum );
        bool isR = isSufficient( node->right, limit, curSum );
        if( !isL ) node->left = NULL;
        if( !isR ) node->right = NULL;
        
        return isL || isR;
    }
};
