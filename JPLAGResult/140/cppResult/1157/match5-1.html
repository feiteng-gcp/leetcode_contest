/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int dfs(TreeNode* n,int sum,int limit) {
        sum += n->val;
        
        if (n->left == NULL && n->right == NULL) {
            if (sum >= limit)
                return 1;
            else
                return 0;
        }
        
        int l,r;
    
        l = 0;
        r = 0;
        if (n->left != NULL)
            l = dfs(n->left,sum,limit);
        if (n->right != NULL)

            r = dfs(n->right,sum,limit);
        
        if (l == 0)
            n->left = NULL;
        if (r == 0)
            n->right = NULL;
        
        return l + r;
    }
    
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        int ans = dfs(root,0,limit);
        
        if (ans == 0)
            return NULL;
        else
            return root;
    }
};
