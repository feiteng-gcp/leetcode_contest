/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<TreeNode*, int> max_sum;
    int l;
    int cal(TreeNode* tn, int sum = 0) {
        if(tn == nullptr) return sum;     
        max_sum[tn] =  max(cal(tn->left, tn->val+ sum), cal(tn->right, tn->val + sum)); 
        return max_sum[tn];
    }

    void del(TreeNode * tn) {
        if(tn->left) {
            if(max_sum[tn->left] < l) {
                tn->left = nullptr;
            }
            else {
                del(tn->left);
            }
        }
        if(tn->right) {
            if(max_sum[tn->right] < l) {
                tn->right = nullptr;
            }
            else {
                del(tn->right);
            }
        }
    }
    TreeNode* sufficientSubset(TreeNode* root, int limit) {        
        cal(root);
        l = limit;
        if(max_sum[root] < limit) {
            return nullptr;
        }
        else {
        del(root);
        return root;
        }
    }
};
