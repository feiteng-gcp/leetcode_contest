/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    pair<TreeNode*, bool> dfs(TreeNode* node, int limit, int sum) {//{modified node, is_insufficient}
        sum += node->val;
        if (!node->left && !node->right) {
            if (sum < limit)
                return {nullptr, true};
            else
                return {node, false};
        }
        bool is_insufficient = true;
        if (node->left) {
            auto p = dfs(node->left, limit, sum);
            is_insufficient = (is_insufficient & p.second);
            node->left = p.first;
        }
        if (node->right) {
            auto p = dfs(node->right, limit, sum);
            is_insufficient = (is_insufficient & p.second);
            node->right = p.first;
        }
        return {is_insufficient ? nullptr : node, is_insufficient};
    }
    
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        if (!root)
            return root;
        auto p = dfs(root, limit, 0);
        return p.first;
    }
};
