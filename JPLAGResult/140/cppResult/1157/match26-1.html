/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {

    bool insufficientPath(TreeNode* curr, int sum, int limit){
        if(curr == nullptr){
            return sum < limit;
        }
        bool l = insufficientPath(curr->left, sum + curr->val, limit);
        bool r = insufficientPath(curr->right, sum + curr->val, limit);
        if(l && curr->left){
            delete curr->left;
            curr->left = nullptr;
        }
        if(r && curr->right){
            delete curr->right;
            curr->right = nullptr;
        }
        return l && r;
    }
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        if(root == nullptr)
            return nullptr;
        bool res = insufficientPath(root, 0, limit);
        if(res){
            return nullptr;
            delete root;
            root = nullptr;
        }
        return root;
    }
};
