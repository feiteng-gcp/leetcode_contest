/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:

    int dfs(TreeNode* root, int limit, int cur) {
        if (!root)
            return cur;
        cur += root -> val;
        int t1 = dfs(root -> left, limit, cur);
        int t2 = dfs(root -> right, limit, cur);
        if (t1 < limit)
            root -> left = NULL;
        if (t2 < limit)
            root -> right = NULL;
        return max(t1, t2);
    }
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        int t = dfs(root, limit, 0);
        if (t < limit)
            return NULL;
        return root;
    }
};
