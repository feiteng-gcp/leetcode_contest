/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    int aa(TreeNode* root, int sum, int limit) {
        if ( root == NULL ) return 0;
        
        int max1 = 1000000000, max2 = 1000000000;
        max1 = aa(root->left, sum+root->val, limit);
        if ( root->left ) {
            if ( sum+root->val+max1 < limit)
                root->left = NULL;
        }
        max2 = aa(root->right, sum+root->val, limit);
        if ( root->right ) {
            if ( sum+root->val+max2 < limit)
                root->right = NULL;
        }
        int maxx = max(max1, max2);
        return maxx+root->val;
    }
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        int r = aa(root, 0, limit);
        if ( r >= limit )
            return root;
        else
            return NULL;
    }
};
