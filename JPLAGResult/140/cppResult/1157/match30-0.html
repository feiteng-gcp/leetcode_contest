template <class Container, class Element>
bool Contains(const Container& container, const Element& element) {
    return container.find(element) != container.end();
}

class Solution {
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        FindSums(root);
        if (node_to_max_sum[root] < limit) {
            return nullptr;
        }
        FilterNodes(root, limit);
        return root;
    }

    void FilterNodes(TreeNode* node, int limit) {
        if (node->left) {
            if (node_to_max_sum[node->left] < limit) {
                node->left = nullptr;
            } else {
                FilterNodes(node->left, limit);
            }
        }
        if (node->right) {
            if (node_to_max_sum[node->right] < limit) {
                node->right = nullptr;
            } else {
                FilterNodes(node->right, limit);
            }
        }
    }

    void FindSums(TreeNode* node, int current_sum = 0) {
        if (!node) {
            return;
        }
        current_sum += node->val;
        if (!node->left && !node->right) {
           SetSum(node, current_sum);
           return;
        }
        if (node->left) {
            FindSums(node->left, current_sum);
            SetSum(node, node_to_max_sum.at(node->left));
        }
        if (node->right) {
            FindSums(node->right, current_sum);
            SetSum(node, node_to_max_sum.at(node->right));
        }
    }

    void SetSum(TreeNode* node, int current_sum) {
        if (Contains(node_to_max_sum, node)) {
            node_to_max_sum[node] = max(node_to_max_sum[node], current_sum);
        } else {
            node_to_max_sum[node] = current_sum;
        }
    }

private:
    unordered_map<TreeNode*, int> node_to_max_sum;
};
