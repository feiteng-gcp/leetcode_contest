/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        if(helper(root, limit, 0) < limit) return NULL;
        return root;
    }
    
    int helper(TreeNode *root, int limit, int preSum)
    {

        if(!root) return 0;
        int left = helper(root->left, limit, preSum + root->val);
        int right = helper(root->right, limit, preSum + root->val);
        if(left + root->val + preSum < limit) root->left = NULL;
        if(right + root->val + preSum < limit) root->right = NULL;
        return max(left,right) + root->val;
    }
};
