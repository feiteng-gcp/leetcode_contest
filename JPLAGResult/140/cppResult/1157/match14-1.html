/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        return helper(root, limit, 0);
    }

    TreeNode* helper(TreeNode* root, int limit, int pre) {
        if (root == nullptr) return root;
        if (maxPath(root) + pre < limit) return nullptr;
         root->left = helper(root->left, limit, pre+root->val);
        root->right = helper(root->right, limit, pre+root->val);
        return root;
    }
    int maxPath(TreeNode* root) {
        if (root == nullptr) return 0;
        return max(maxPath(root->left), maxPath(root->right)) + root->val;
    }
};
