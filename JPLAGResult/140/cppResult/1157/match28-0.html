/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    long long limit;
    
    pair<TreeNode*,long long> dfs(TreeNode* o, long long sum){
        const long long INF = 1ll<<62;
        pair<TreeNode*,long long> ret{o,-INF};
        
        if (o->left==NULL && o->right==NULL){
            ret.second = sum + o->val;
            if ( ret.second < limit ) ret.first = NULL;    
            return ret;
        }
        
        if (o->left){
            auto d = dfs(o->left, sum + o->val);
            ret.second = max(ret.second, d.second);
            ret.first->left = d.first;
        }
        if (o->right){
            auto d = dfs(o->right, sum + o->val);
            ret.second = max(ret.second, d.second);
            ret.first->right = d.first;
        }
        if ( ret.second < limit ) ret.first = NULL;
        return ret;
    }
    TreeNode* sufficientSubset(TreeNode* root, int _limit) {
        limit = _limit;
        return dfs(root,0).first;
    }
};
