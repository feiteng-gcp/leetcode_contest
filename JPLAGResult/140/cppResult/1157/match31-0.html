class Solution {
public:
	int B;
	pair<int, int> dfs(TreeNode *t, int sum) {
		int max_val = INT_MIN;
		if (t->left != NULL) {
			int del, val;
			tie(del, val) = dfs(t->left, sum + t->val);
			if (del) delete t->left, t->left = NULL;
			max_val = max(max_val, val);
		}
		if (t->right != NULL) {
			int del, val;
			tie(del, val) = dfs(t->right, sum + t->val);
			if (del) delete t->right, t->right = NULL;
			max_val = max(max_val, val);
		}
		if (max_val == INT_MIN) {
			return make_pair(sum + t->val < B, t->val);
		} else {
			return make_pair(sum + max_val + t->val < B, max_val + t->val);
		}
	}
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
    	B = limit;
    	pair<int, int> ret = dfs(root, 0);
    	if (ret.first)
    		return NULL;
    	return root;
    }
};
