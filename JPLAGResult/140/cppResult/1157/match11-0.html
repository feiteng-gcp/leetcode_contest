/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool prune(TreeNode* root, const int& s, const int& limit)
    {
        if(not root)
        {
            return s>=limit;
        }
        
        bool l = prune(root->left, s+root->val, limit);
        bool r = prune(root->right, s+root->val, limit);
        if(not l)
        {
            root->left = nullptr;
        }
        if(not r)
        {
            root->right = nullptr;
        }
        if(l or r)
        {
            return true;
        }
        else
        {
            return false;
        }
    }
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        if(not prune(root, 0, limit))
        {
            root=nullptr;
        }
        return root;
    }
};
