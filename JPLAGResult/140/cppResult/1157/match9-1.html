/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {

    int solve(TreeNode* cur, int sum, int& limit) {
        if (!cur) return sum;
        sum += cur->val;
        
        int l = solve(cur->left, sum, limit);
        int r = solve(cur->right, sum, limit);
        
        if (l < limit) cur->left = nullptr;
        if (r < limit) cur->right = nullptr;
        return max(l, r);
    }
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        int tmp = solve(root, 0, limit);
        if (tmp < limit) return nullptr;
        return root;
    }
};
