/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
      dfs(root, limit, 0);
      return root;
    }
private:

  static int dfs(TreeNode *&root, int limit, int sum) {
    if (!root) {
      return 0;
    }
    sum += root->val;
    int mx = max(dfs(root->left, limit, sum), dfs(root->right, limit, sum));
    int ret = mx + root->val;
    if (mx + sum < limit) {
      root = nullptr;
    } 
    return ret;
  }
};
