/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        if (root == NULL) return root;
        //reeNode* l, *r;
        int lval = helper(root->left, root->val, limit);
        int rval = helper(root->right, root->val, limit);

        if (lval < limit && rval < limit) return NULL;
        return root;
    }
    int helper(TreeNode* &node, int presum, int limit) {
        if (node == NULL) return presum;
        //TreeNode* l, *r;
        int lval = helper(node->left, node->val + presum, limit);
        int rval = helper(node->right, node->val + presum, limit);
        if (lval < limit && rval < limit) {
            node = NULL;
            return INT_MIN;
        };
        return max(lval, rval);
    }
};
