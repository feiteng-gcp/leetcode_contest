/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    pair<TreeNode*, int> DFS(TreeNode* cur, int limit, int parentSum) {
        int curSum = parentSum + cur->val;
        int childSumMax = 0;
        if (cur->left || cur->right) {
            childSumMax = INT_MIN;
        }

        if (cur->left) {
            auto pr = DFS(cur->left, limit, curSum);
            childSumMax = max(childSumMax, pr.second);
            cur->left = pr.first;
        }
        if (cur->right) {
            auto pr = DFS(cur->right, limit, curSum);
            childSumMax = max(childSumMax, pr.second);
            cur->right = pr.first;
        }
        auto res = cur;
        if (childSumMax + curSum < limit) {
            res = nullptr;
        }
        int second = childSumMax + cur->val;
        return { res, second };
    }
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        if (!root) return root;
        return DFS(root, limit, 0).first;
    }
};
