/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* sufficientSubset(TreeNode* root, int limit) {
        return recur(root, limit, 0);
    }
    
private:
    TreeNode* recur(TreeNode* node, int limit, int curSum) {
        if (node == nullptr) {
            return nullptr;
        }
        if (node->left == nullptr && node->right == nullptr) {
            if (curSum + node->val < limit) {
                return nullptr;
            }
            return node;
        }
        node->left = recur(node->left, limit, curSum + node->val);
        node->right = recur(node->right, limit, curSum + node->val);
        if (node->left == nullptr && node->right == nullptr) {
            return nullptr;
        }
        return node;
    }
};
