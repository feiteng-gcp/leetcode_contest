class Solution {
public:
    string smallestSubsequence(string text) {
        vector<int> Hashtable(128, 0);
		vector<bool> visited(128, false);
		for (char x : text) Hashtable[x]++;
		string result = "0";


		for (char x : text) {
			Hashtable[x]--;
			if (visited[x]) continue;
			while (x < result.back() && Hashtable[result.back()]) {
				visited[result.back()] = false;
				result.pop_back();
			}
			result += x;
			visited[x] = true;
		}
		return result.substr(1);
    }
};
