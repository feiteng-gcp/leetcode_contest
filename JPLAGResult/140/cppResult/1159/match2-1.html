class Solution {
public:
    map<char, int> ma;
    map<char, int> mb;
    stack<char> a;
    set<int> ss;
    string smallestSubsequence(string text) {
        for (int i = 0; i < text.size(); i++) {
            ma[text[i]]++;
        }
        
        a.push(text[0]);
        mb[text[0]] = 1;

        ma[text[0]]--;
        for (int i = 1; i < text.size(); i++) {
            ma[text[i]]--;
            if (mb[text[i]]) continue;
            while (!a.empty() && a.top() > text[i] &&
                   ma[a.top()] > 0) {
                mb[a.top()] = 0;
                a.pop();
            }
            a.push(text[i]);
            mb[text[i]] = 1;
        }
        
        string res = "";
        while (!a.empty()) {
            res += a.top();
            a.pop();
        }
        reverse(res.begin(),res.end());
        //cout << res;
        return res;
    }
};
