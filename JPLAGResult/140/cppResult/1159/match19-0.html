class Solution {
public:
    string smallestSubsequence(string text) {
        if (text.empty())
            return "";

        int n = text.size();
        vector<int> freq(128, 0);

        for (char c : text)
            freq[c]++;

        vector<int> copy_freq = freq;
        int blocked = -1;

        for (int i = 0; i < n; i++)
            if (--copy_freq[text[i]] == 0) {
                blocked = i;
                break;
            }

        assert(blocked >= 0);
        char min_char = 'z';

        for (int i = 0; i <= blocked; i++)
            min_char = min(min_char, text[i]);

        int position = -1;

        for (int i = 0; i <= blocked; i++)
            if (text[i] == min_char) {
                position = i;
                break;
            }

        string next_string = "";

        for (int i = position + 1; i < n; i++)
            if (text[i] != min_char)
                next_string += text[i];

        return min_char + smallestSubsequence(next_string);
    }
};
