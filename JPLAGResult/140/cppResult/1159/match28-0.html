class Solution {
public:
    
    string smallestSubsequence(string text) {
        string result;
        int index = 0;
        int type = 0;
        vector<bool> rec(26, false);
        for(auto c : text){
            int t = c - 'a';
            if(!rec[t]) type ++;
            rec[t] = true;
        }
        while(type --){
            vector<bool> show(26, false);
            vector<int> first(26, -1);
            vector<set<int>> memo(26, set<int>());
            for(int i = index; i < text.size(); i ++){
                char c = text[i];
                int t = c - 'a';
                if(!rec[t]) continue;
                if(show[t] == false) first[t] = i;
                show[t] = true;
                
                for(int i = 0; i < 26; i ++){
                    if(i == t || !show[i])
                        continue;
                    memo[i].insert(t);
                }
            }
            for(int i = 0; i < 26; i ++){
                if(!rec[i]) continue;
                //cout << endl;
                if(memo[i].size() == type){
                    //cout << char('a' + i) << type << ' ';
                    result += 'a' + i;
                    rec[i] = false;
                    index = first[i];
                    break;
                }
            }
        }
        return result;
    }
};
