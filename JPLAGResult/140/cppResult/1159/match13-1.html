class Solution {
public:
  string compress(string v){
    sort(v.begin(),v.end());
    v.erase(unique(v.begin(),v.end()),v.end());
    return v;
  }
  string smallestSubsequence(string text) {
    string base=compress(text);
    int n=text.size(),m=base.size();
    set<char> used,rest;
    for(char c:text) rest.emplace(c);

    string res;
    int i=0;

    while(res.size()<base.size()){
      for(int k=0;k<m;k++){
        if(used.count(base[k])) continue;
        int j=i;
        while(text[j]!=base[k]) j++;
        set<char> tmp;
        for(int l=j;l<n;l++)
          tmp.emplace(text[l]);

        int flg=1;
        for(char c:rest)
          flg&=tmp.count(c);

        if(!flg) continue;
        res+=base[k];
        i=j+1;

        rest.erase(base[k]);
        used.emplace(base[k]);
        break;
      }
    }
    return res;
  }
};
