class Solution {
public:
    string smallestSubsequence(string text) {
        unordered_set<char> uniques{}; 
        for (auto& c : text) uniques.insert (c);
        string ans{};
        int start = 0, len = uniques.size();
        
        while (ans.size() != len) {
            unordered_set<char> tmp{};
            int i {};
            for (i = text.size()-1; i >= 0; i--) { 
                if (uniques.count (text[i])) tmp.insert (text[i]); 
                if (tmp.size() == uniques.size()) break;
            }
            char e = 'z'+1;
            for (int j = start; j <= i; ++ j) {
                if (text[j] < e && uniques.count (text[j])) {
                    e = text[j];
                    start = j+1;
                }
            }
            ans.push_back (e);
            uniques.erase (e);
            //cout<<" e " <<e <<" start " <<start <<" i "<<i  <<endl;
        }
        return ans;
    }
};
