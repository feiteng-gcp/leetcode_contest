class Solution {
public:
    string smallestSubsequence(string text) {
        set<char> visited;
        set<char> type;
        for (int i = 0; i < text.size(); i++) type.insert(text[i]);
        int s = type.size();
        int curs = s;
        string res;
        int pos = -1;
        while (res.size() < s) {
            char temp;
            set<char> cur;
            int curpos = pos;
            for (int i = text.size() - 1; i > pos; i--) {
                if (visited.count(text[i])) continue;
                if (cur.size() < curs) {
                    curpos = i;
                    temp = text[i];
                }
                else {
                    if (text[i] <= temp) {
                        temp = text[i];
                        curpos = i;
                    }
                }
                cur.insert(text[i]);
            }
            visited.insert(temp);
            res += temp;
            curs--;
            pos = curpos;
        }
        return res;
    }
};
