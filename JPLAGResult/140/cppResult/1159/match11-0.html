class Solution {
public:
    string smallestSubsequence(string text) {
        int last[26];
        for (int i = 0; i < 26; i++) {
            last[i] = -1;
        }
        for (int i = text.size() - 1; i >= 0; i--) {
            int c = text[i] - 'a';
            if (last[c] == -1) {
                last[c] = i;
            }
        }
        
        int head = 0;
        string ans;
        for (int i = 0; i < text.size(); i++) {
            bool find = false;
            int j = 0;
            for (; j < ans.size(); j++) {
                if (text[i] == ans[j]) {
                    find = true;
                    break;
                }
                if (j >= head && text[i] < ans[j]) {
                    ans[j] = text[i];
                    ans.resize(j + 1);
                    find = true;
                    break;
                }
            }
            if (!find) {
                ans.push_back(text[i]); 
            }
            if (last[text[i] - 'a'] == i) {
                head = max(head, find ? j + 1 : (int)ans.size());
            }
        }
        
        return ans;
    }
};
