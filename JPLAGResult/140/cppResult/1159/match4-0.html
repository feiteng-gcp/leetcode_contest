class Solution {
public:
    int visit[255] = {0};
    
    bool judge( string text, int start ) {
        
        int visit2[255] = {0};
        for ( int i = start ; i < text.size() ; ++i ) {
            visit2[text[i]] = 1;
        }
        for ( int i = 0 ; i < 255 ; ++i )
            if ( visit2[i] == 0 && visit[i] == 1 ) return false;
        return true;
    }
    
    string smallestSubsequence(string text) {
        
        int ok[255];
        string res = "";
        
        int l = text.size() ;
        for ( int i = 0 ; i < l ; ++i ) {
            visit[text[i]] = 1;
        }
        int last = 0 ;
        while(true) {
            int t;
            for ( t = 0 ; t < 255 ; ++t )
                if ( visit[t] == 1 ) break;
            if ( t == 255 ) break;
            
            for ( int i = 0 ; i < 255 ; ++i ) ok[i] = 0 ;
            
            for ( int i = last ; i < l ; ++i ) {
                if ( visit[text[i]] && judge(text, i) ) {
                    ok[text[i]] = 1;
                }
            }
            
            for ( t = 0 ; t < 255 ; ++t )
                if ( ok[t] == 1 ) break;
            if ( t == 255 ) break;
            
            for ( t++ ; t < 255 ; ++t ) ok[t] = 0 ;
            
            for ( int i = last ; i < l ; ++i ) {
                if ( ok[text[i]] ) {
                    res += text[i];
                    visit[text[i]] = 0 ;
                    last = i+1;
                    break;
                }
            }
        }
        return res;
    }
};
