struct StringMaster {
	string S; int N; StringMaster() {}StringMaster(string s) { init(s); }
	void init(string s) { S = s; N = S.length(); }

	// get the nearest index for the char
	vector<int> hidari[26], migi[26];
	inline void initgo() {
		rep(c, 0, 26) { migi[c].resize(N); migi[c][N - 1] = inf; }
		rrep(i, N - 2, 0) { rep(c, 0, 26) migi[c][i] = migi[c][i + 1]; migi[S[i + 1]][i] = i + 1; }
		rep(c, 0, 26) { hidari[c].resize(N); hidari[c][0] = -1; }
		rep(i, 1, N) { rep(c, 0, 26) hidari[c][i] = hidari[c][i - 1]; hidari[S[i - 1]][i] = i - 1; }
	}
	inline int gomigi(int cu, int c) { 
		if (cu < 0) {
			if (S[0] == c) return 0;
			return migi[c][0];
		}
		return migi[c][cu];
	}
	inline int gohidari(int cu, int c) { return hidari[c][cu]; }
};



class Solution {
public:
	string smallestSubsequence(string text) {
		fore(c, text) c -= 'a';
		StringMaster sm(text);
		sm.initgo();
		vector<char> v;
		fore(c, text) v.push_back(c);
		sort(all(v));
		v.erase(unique(all(v)), v.end());

		string ans = "";
		int n = v.size();
		int cur = -1;
		int used = 0;

		rep(i, 0, n) {
			rep(j, 0, n) if (!(used & (1 << j)) and sm.gomigi(cur, v[j]) < inf) {
				int msk = used + (1 << j);
				int to = sm.gomigi(cur, v[j]);
				
				int ok = 0, cnt = 0;
				rep(k, 0, n) if (!(msk & (1 << k))) {
					cnt++;
					if (sm.gomigi(to, v[k]) < inf) ok++;
				}


				if (ok == cnt) {
					ans += char('a' + v[j]);
					cur = to;
					used = msk;
					break;
				}
			}
		}

		return ans;
	}
};
