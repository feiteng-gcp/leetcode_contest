class Solution {
public:
    string smallestSubsequence(string text) {
        int n=text.length();
        vector<int> vec(n+1,0);
        for(int i=n-1;i>=0;i--)
        {
            vec[i]=vec[i+1];
            vec[i]|=1<<(text[i]-'a');
        }
        //cout<<"here1"<<endl;
        int mask=vec[0];
        int m=0;
        for(int i=0;i<26;i++)
            if(mask&(1<<i)) m++;
        string ans;
        int pre=-1;
        //cout<<" here "<<m<<" "<<mask<<endl;
        for(int ii=0;ii<m;ii++)
        {
            int p=n,mi=27;
            for(int i=n-1;i>pre;i--)
            {
                int now=text[i]-'a';
                //cout<<"  !!  "<<vec[i]<<" "<<mask<<" "<<(1<<now)<<endl;
                //cout<<"  !!  "<<((vec[i]&mask)==mask)<<" "<<(mask&(1<<now))<<endl;
                if(((vec[i]&mask)==mask)&&(mask&(1<<now)))

                {
                    //cout<<i<<endl;
                    if(now<=mi)
                    {
                        mi=now;
                        p=i;
                    }
                }
            }
            //cout<<p<<endl;
            pre=p;
            mask^=1<<(text[p]-'a');
            ans+=text[p];
            //cout<<ans<<" "<<p<<" "<<text[p]<<" "<<mask<<endl;
        }
        return ans;
    }
};
