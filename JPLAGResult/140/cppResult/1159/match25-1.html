class Solution {
public:
    string smallestSubsequence(string text) {
        map<char, deque<int>> positions;
        for (int i = 0; i < text.size(); ++i) {
            positions[text[i]].push_back(i);
        }
        string res;
        while (!positions.empty()) {
            auto it = positions.begin();
            int maxPos = text.size();
            while (it != positions.end()) {
                if (it->second.back() < maxPos) {
                    maxPos = it->second.back();
                }
                ++it;
            }
            char next;
            auto rIt = positions.rbegin();
            int clear = maxPos;
            while (rIt != positions.rend()) {
                if (rIt->second.front() <= maxPos) {
                    next = rIt->first;
                    clear = rIt->second.front();
                }
                ++rIt;
            }
            positions.erase(next);
            it = positions.begin();
            while (it != positions.end()) {
                while (it->second.front() < clear) {
                    it->second.pop_front();
                }
                ++it;
            }
            res.push_back(next);
        }
        return res;
    }
};
