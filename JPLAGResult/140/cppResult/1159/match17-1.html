#define SIZE 26

class Solution {
    int Hash[SIZE];
public:
    string smallestSubsequence(string text) {
        bitset<SIZE> distinct(0);
        
        for(int i=0;text[i]!='\0';i++)
            Hash[text[i]-'a']++;
        
        for(int i=0;i<SIZE;i++)
            if(Hash[i])
                distinct[i] = true;
        
        vector<vector<int> > st;
        
        vector<bitset<SIZE> > st_modified;
        
        st.resize(text.size());
        
        st_modified.assign(text.size(),0);
        
        for(int i=0;i<text.size();i++)
            st[i].assign(SIZE,0);
    
        for(int i=text.size()-1;i>=0;i--){
            st[i][text[i]-'a']++;
            for(int j=0;j<SIZE;j++)
                if(i+1 < text.size())
                    st[i][j] += st[i+1][j];
            
            for(int j=0;j<SIZE;j++)
                if(st[i][j])
                    st_modified[i][j] = true;
        }

        
        char tmp[1000];
        
        int N = 0;
        
        for(int i=0;i<SIZE;i++)
            if(distinct[i])
                N++;
        

        int idx = -1;
        
        for(int i=0;i<N;i++){
            //cout<<i<<" : \n";
            for(int j=0;j<SIZE;j++){
                if(!distinct[j])
                    continue;
                bool possible = false;
                for(int k=idx+1;k<text.size();k++){
                    if(text[k] == j+'a'){
                        //cout<<k<<" : "<<st_modified[k].to_ulong()<<"\n";
                        //cout<<distinct.to_ulong()<<"\n";
                        if((st_modified[k].to_ulong() & distinct.to_ulong()) >= distinct.to_ulong()){
                            possible = true;
                            idx = k;
                            break;
                        }
                    }
                }
                
                if(possible){
                    tmp[i] = j+'a';
                    //cout<<"Choosen "<<j<<" at "<<idx<<"\n";
                    distinct[j] = false;
                    break;
                }     
            }
        }
        
        tmp[N] = '\0';
        return (string)(tmp);
    }
};
