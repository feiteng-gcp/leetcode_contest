class Solution {
public:
    string smallestSubsequence(string text) {
        unordered_map<char, int> cache;
        for (auto c: text)
            cache[c]++;
        //int n = cache.size();
        string res;
        unordered_set<char> used;
        for (auto c : text) {
            cache[c]--;
            string tmp = res;
            unordered_set<char> tmp_used = used;
            while (res.size() > 0 && c < res.back() && cache[res.back()] > 0) {
                used.erase(res.back());
                res.pop_back();
            }
            if (used.count(c) == 0) {
                res.push_back(c);
                used.insert(c);
            } else {
                res = tmp;
                used = tmp_used;
            }
            //cout << res << endl;
        }
        return res;
    }
};
