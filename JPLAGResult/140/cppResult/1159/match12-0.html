class Solution {
public:
    
int findMinLast(unordered_map<char, int>& myMap) {
    if (myMap.size() == 0) {
        return -1;
    }
    
    int minLast = INT_MAX;
    auto it = myMap.begin();
    for (; it != myMap.end(); it++) {
        minLast = min(it->second, minLast);
    }
    return minLast;
}

string removeDuplicateLetters(string str) {
    if (str.length() == 0) {
        return "";
    }
    
    unordered_map<char, int> myMap;
    for (int i = 0; i < str.length(); i++) {
        myMap[str[i]] = i;
    }

    int last = findMinLast(myMap);
    // find min in [0, last]
    int minIndex = min_element(str.begin(), str.begin() + last + 1) - str.begin();
    int left = minIndex + 1;
    char minChar = str[minIndex];
        
    string ret = str.substr(left);
    ret.erase(remove(ret.begin(), ret.end(), minChar), ret.end());
    ret = removeDuplicateLetters(ret);
    ret = minChar + ret;

    return ret;
}

    string smallestSubsequence(string str) {
        return removeDuplicateLetters(str);
    }
};
