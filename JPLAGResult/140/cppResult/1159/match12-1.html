class Solution {
private:

    void doGet(int idx, string& text, unordered_set<char>& used, string& prefix){
        if(idx>=text.size()){
            return;
        }
        unordered_set<char> chars;
        //int size = text.size() - idx;
        vector<int> count(text.size()+1, 0);
        for(int i=text.size()-1; i>=idx; i--){
            count[i] = chars.size();
            if(used.find(text[i])==used.end()){
                chars.insert(text[i]);
                count[i] = chars.size();
            }
        }
        if(chars.size()==0)
            return;
        map<char, int> cans;
        for(int i=idx; i<text.size(); i++){
            if(used.find(text[i])!=used.end()){
                continue;
            }
            if(cans.find(text[i])==cans.end()){
                cans[text[i]] = i;
            }
            if(count[i] > count[i+1]){
                prefix += cans.begin()->first;
                used.insert(cans.begin()->first);
                doGet(cans.begin()->second+1, text, used, prefix);
            }
        }
    }
public:
    string smallestSubsequence(string text) {
        string prefix = "";
        unordered_set<char> used;
        doGet(0, text, used, prefix);
        return prefix;
    }
};
