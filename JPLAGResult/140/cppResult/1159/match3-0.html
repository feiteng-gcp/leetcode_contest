class Solution {
public:
    string smallestSubsequence(string text) {
        unordered_set<char> used;
        unordered_map<char, int> cnt;
        for(char c: text) {
            cnt[c] += 1;
        }
        string ans = "Z";
        string ans2;
        for(int i = 0 ; i < cnt.size(); i++)
            ans2 += 'z';
        for(char c: text) {
            while (c < ans.back() && cnt[ans.back()] > 1 && used.count(c) == 0) {
                cnt[ans.back()] -= 1;
                used.erase(ans.back());
                // cout << "pop " << ans.back() << endl;
                ans.pop_back();
            }
            if (used.count(c) == 0) {
                ans += c;
                used.insert(c);
                ///cout << "insert " <<ans << endl;
            } else {
                cnt[c] -= 1;
            }
            // cout << ans << endl;
            string tmp = ans.substr(1);
            if (tmp.length() == cnt.size() && tmp < ans2) {
                ans2 = tmp;
            }
        }
        return ans2;
    }
};
