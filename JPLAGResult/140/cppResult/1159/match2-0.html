class Solution {
public:
    string smallestSubsequence(string text) {
        string ans;
        int a[26]={0},visited[26]={0};
        for(auto c:text)
            a[c-'a']++;
        for(int i=0;i<text.size();i++){
            a[text[i]-'a']--;
            if(visited[text[i]-'a']) continue;
            while(ans.size()>0&&a[ans.back()-'a']&&text[i]<ans.back()){
                visited[ans.back()-'a']=0;
                ans.pop_back();
            }
            ans.push_back(text[i]);
            visited[text[i]-'a']=1;
        }
        return ans;
    }
};
