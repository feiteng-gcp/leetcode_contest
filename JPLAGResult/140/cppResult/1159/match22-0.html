class Solution {
public:
    string smallestSubsequence(string s) {
        vector<deque<int>> vec(26);
        for (int i = 0; i < s.size(); i++) {
            vec[s[i] - 'a'].push_back(i);
        }
        int now = -1;
        string t;
        for (int i = 0; i < 26; i++) {
            for (int j = 0; j < 26; j++) {
                if (!vec[j].empty()) {
                    bool ok = true;
                    for (int k = 0; k < 26; k++) {
                        if (!vec[k].empty() && vec[k].back() < vec[j][0]) ok = false;
                    }
                    if (ok) {
                        t += (char)(j + 'a');
                        now = vec[j][0];
                        vec[j].clear();
                        break;
                    }
                }
            }
            for (int j = 0; j < 26; j++) {
                while (!vec[j].empty() && vec[j][0] <= now) vec[j].pop_front();
            }
        }
        return t;
    }
};
