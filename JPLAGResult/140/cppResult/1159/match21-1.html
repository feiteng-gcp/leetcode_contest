class Solution {
public:
    string smallestSubsequence(string s) {
        int n = s.size();
        vector<int> v[26];
        vector<int> u(26);
        for (int i = 0; i < n; ++ i)
        {
            v[s[i]-'a'].push_back(i);
            u[s[i]-'a'] = 1;
        }
        int cnt = 0;
        for (int i = 0; i < 26; ++ i)
            cnt += u[i];
        string ret;

        int cur = 0;
        for (int k = 0; k < cnt; ++ k)
        {
            for (int i = 0; i < 26; ++ i)
            {
                if (!u[i]) continue;
                auto it = lower_bound(v[i].begin(), v[i].end(), cur);
                if (it == v[i].end()) continue;
                int pos = *it;
                
                int flag = 0;
                for (int j = i+1; j < 26; ++ j) if (u[j])
                {
                    auto it = lower_bound(v[j].begin(), v[j].end(), pos+1);
                    if (it == v[j].end()) flag = 1;
                }
                if (!flag)
                {
                    ret += i+'a';
                    cur = pos+1;
                    v[i].clear();
                    u[i] = 0;
                    break;
                }
            }
        }
        return ret;
    }
};
