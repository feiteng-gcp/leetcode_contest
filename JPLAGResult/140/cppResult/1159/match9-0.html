class Solution {
public:
    string smallestSubsequence(string text) {
        vector<int> cnt(26,0);
        vector<int> used(26,false);
        for(char c:text) cnt[c-'a']++;
        string ans;
        int start=0;
        bool flag=true;
        while(flag){
            vector<int> cnt2=cnt;
            int t=-1;
            char c;
            flag=false;
            for(int i=start;i<text.size();i++){
                if(used[text[i]-'a']) continue;
                if(t==-1||text[i]<c){
                    t=i;
                    c=text[i];
                }
                if(cnt2[text[i]-'a']==1){
                    ans.push_back(c);
                    used[c-'a']=true;
                    for(int j=start;j<=t;j++) cnt[text[j]-'a']--;
                    start=t+1;
                    flag=true;
                    break;
                }else cnt2[text[i]-'a']--;
            }
        }
        return ans;
    }
};
