class Solution {
public:
    string smallestSubsequence(string text) {
        vector<int> alpha(26, 0);

        vector<int> visited(26, 0);
        for(int i = 0; i < text.size(); ++i)
        {
            alpha[text[i] - 'a']++;
        }
        
        string res;
        for(int i = 0; i < text.size(); ++i)
        {
            if(visited[text[i] - 'a'] == 1) 
            {
                alpha[text[i] - 'a']--;
                continue;
            }
            while(res.size() != 0 && res.back() > text[i] && alpha[res.back() - 'a'] != 0)
            {
                auto tmp = res.back();
                res.pop_back();
                visited[tmp - 'a'] = 0;
            }
            res.push_back(text[i]);
            visited[text[i] - 'a'] = 1;
            alpha[text[i] - 'a']--;
        }
        
        return res;
    }
};
