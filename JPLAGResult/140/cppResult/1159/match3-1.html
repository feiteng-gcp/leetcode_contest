class Solution {
public:
    string smallestSubsequence(string text) {
        vector<int> count(26, 0);
        vector<bool> exist(26, false);
        for (char c: text) ++count[c-'a'];
        string ret;
        for (char c: text) {
            int n = count[c-'a'];

            while (!exist[c-'a'] && !ret.empty() && ret.back() >= c && count[ret.back()-'a']) {
                exist[ret.back()-'a'] = false;
                ret.pop_back();
            }
            --count[c-'a'];
            if (!exist[c-'a']) {
                ret.push_back(c);
                exist[c-'a'] = true;
            }
            
        }
        return ret;
    }
};
