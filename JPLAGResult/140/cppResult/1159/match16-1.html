class Solution {
public:
    static const int N = 1010;
    char s[N];
    string ans;
    int Next[N][27];
    bool vis[27];
    bool check(int pos){
        for(int i=0;i<26;i++){
            if(!vis[i])continue;
            if(Next[pos][i]==0)return false;
        }
        return true;
    }
    string smallestSubsequence(string text) {
        int n = text.size(),cnt = 0;
        for(int i=1;i<=text.size();i++)s[i] = text[i-1];
        for(int i=n;i>=1;i--){
            int id = s[i]-'a';
            if(!vis[id]){cnt++;vis[id]=1;}
            for(int j=0;j<26;j++)Next[i][j] = Next[i+1][j];
            Next[i][id] = i;
        }

        int nowpos = 1;
        for(int i=0;i<cnt;i++){
            for(int j=0;j<26;j++){
                if(!vis[j])continue;
                if(check(Next[nowpos][j])){
                    ans.push_back('a'+j);
                    nowpos = Next[nowpos][j];
                    vis[j] = 0;
                    break;
                }
            }
        }
        return ans;
    }
};
