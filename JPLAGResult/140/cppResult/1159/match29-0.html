class Solution {
public:
    string smallestSubsequence(string text) {
        int n = text.size();
        vector<int> mask_v(n, 0);
        mask_v[n - 1] = getBit(text[n-1]);
        for (int i = n - 2; i >= 0; i--) {
            mask_v[i] = mask_v[i + 1] | getBit(text[i]);
        }
        vector<vector<int>> pos(26, vector<int>());
        for (int i = 0; i < n; i++) {
            pos[text[i] - 'a'].push_back(i);
        }
        int cur_mask = mask_v[0];
        int cur_pos = 0;
        string result;
        while (cur_mask != 0) {
            for (int bit = 0; bit < 26; bit++) {
                if (cur_mask & (1 << bit)) {
                    auto it = lower_bound(pos[bit].begin(), pos[bit].end(), cur_pos);
                    int next_pos = *it;
                    if ((mask_v[next_pos] & cur_mask) == cur_mask) {
                        result.push_back(bit + 'a');
                        cur_pos = next_pos + 1;
                        cur_mask &= ~(1 << bit);
                        break;
                    }
                }
            }
        }
        return result;
    }
    
    int getBit(char c) {
        return 1 << (c - 'a');
    }
};
