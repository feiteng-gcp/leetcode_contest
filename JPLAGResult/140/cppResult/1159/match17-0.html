class Solution {
public:
    bool contain(int S, int T) {
        return (S & T) == T;
    }
    string smallestSubsequence(string s) {
        int n = s.size();
        vector<int> dp(n + 1);
        for (int i = n - 1; i >= 0; --i) {
            int k = s[i] - 'a';
            dp[i] = (1 << k) | dp[i + 1];
        }
        int len = __builtin_popcount(dp[0]);
        int pos = 0, rem = dp[0];
        string ret;
        for (int i = 0; i < len; ++i) {
            for (int k = 0; k < 26; ++k) {
                if (!(rem & (1 << k))) continue;
                int nxt_pos = pos;
                while (nxt_pos < n && s[nxt_pos] - 'a' != k) ++nxt_pos;
                if (contain(dp[nxt_pos], rem)) {
                    ret += (char)('a' + k);
                    rem ^= 1 << k;
                    pos = nxt_pos + 1;
                    break;
                }
            }
        }
        return ret;
    }
};
