class Solution {
public:
    string smallestSubsequence(string text) {
        map<char, vector<int>> m;
        for(int i = 0; i < text.size(); i++) {
            m[text[i]].push_back(i);
        }
        vector<int> mask(text.size());
        set<char> filled;
        for(int i = 0; i < text.size(); i++) {
           
            char c = text[i];
            if(filled.count(c)) continue;
           
            int last = *m[c].rbegin();
            if(i == last) {
                mask[i] = 1;
                continue;
            }
            int j;
            for(j = i+1; j < text.size(); j++) {
                if(filled.count(text[j])) continue;
                if(j == last) {
                   
                    mask[i] = 1;
                    filled.insert(c);
                    break;
                }
                
                if(text[j] < c) {
                  mask[i] = 0;
                    break;
                }
                if(*m[text[j]].rbegin() == j) {
                   
                    mask[i] = 1;
                    filled.insert(c);
                    break;
                }
            }
            if(j == text.size()) {
                
                mask[i] = 1;
                    filled.insert(c);
            }
            
        }
        string ret;
        for(int i = 0; i < text.size(); i++) {
            if(mask[i]) {
                ret.push_back(text[i]);
            }
        }
        return ret;
    }
};
