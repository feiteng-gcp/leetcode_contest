class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        istringstream iss(text);
        string s,t;
        vector<string> ans;
        vector<string> r;
        while(getline(iss,s,' '))
            r.push_back(s);
        for(int i=0;i<r.size()-2;i++)
        {
            if(r[i]==first&&r[i+1]==second)
                ans.push_back(r[i+2]);
        }
        return ans;
    }
};
