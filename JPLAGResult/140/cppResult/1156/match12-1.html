class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        vector<string> ans;
        stringstream str(text);
        vector<string> v;
        string nxt;
        while(str>>nxt)v.push_back(nxt);

        for(int i=0;i<v.size()-2;i++){
            if(v[i]==first&&v[i+1]==second){
                ans.push_back(v[i+2]);
            }
        }
        return ans;
    }
};
