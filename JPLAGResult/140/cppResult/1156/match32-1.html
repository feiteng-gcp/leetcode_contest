class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        vector<string> ret;
        vector<string> words;
        stringstream ss(text);
        string s;
        while (getline(ss, s, ' ')) {
             words.push_back(s);
        }
        int cur = 0;
        for (int i = 0; i < words.size(); ++i) {
            if (cur == 0) {
                if (words[i] == first) cur = 1;
            }
            else if (cur == 1) {
                if (words[i] == second) {
                    if (i + 1 < words.size()) ret.push_back(words[i+1]);  
                }
                if (first == words[i]) cur = 1;
                else cur = 0;
            }
        }
        return ret;
    }
};
