class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        int len=text.size();
        vector<string> t;
        string tmp;
        for(int i=0; i<len; i++) {
            if (text[i]==' ') {
                t.push_back(tmp);
                tmp="";
            }
            else tmp+=text[i];
        }
        t.push_back(tmp);
        
        int n=t.size();
        vector<string> ans;
        for(int i=0; i<n-2; i++) {
            if (t[i]==first&&t[i+1]==second) ans.push_back(t[i+2]);
        }
        return ans;
    }
};
