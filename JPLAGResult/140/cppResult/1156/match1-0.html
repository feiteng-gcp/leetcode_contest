class Solution {
public:
  vector<string> findOcurrences(string text, string first, string second) {
    stringstream ss(text);
    vector<string> vs;
    string tmp;
    while (ss >> tmp) vs.push_back(tmp);
    vector<string> ans;
    for (int i = 2; i < vs.size(); i++) {
      if (vs[i-2]==first && vs[i-1] == second) ans.push_back(vs[i]);
    }
    return ans;
  }
};
