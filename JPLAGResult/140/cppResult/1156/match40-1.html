class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        vector<string> input;
        int start = 0;
        int end = 0;
        while (start < text.length() && end < text.length()) {
            end = text.find(" ", start);
            if (end != string::npos || text.length() > (start)) {
                string word = text.substr(start, end - start);
                //cout << word << endl;
                input.push_back(word);
                start = text.find_first_not_of(" ", end);
                if (start == string::npos) {
                    break;
                }
            } else{
                break;
            }
        }

        vector<string> res;
        if (input.size() < 3) {
            return res;
        }

        for (int i = 0; i < input.size() - 2; i++) {
            if (input[i].compare(first) == 0 && input[i+1]. compare(second) == 0) {
                res.push_back(input[i+2]);
            }
        }
        
        return res;
    }
};
