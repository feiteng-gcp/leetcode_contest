class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        vector<string> res;
        vector<string> input;
        int start = 0, end = 0;
        while(end < text.size())
        {
            if(text[end] != ' ') end++;
            else
            {
                input.push_back(text.substr(start, end - start));
                start = end + 1;
                end++;
            }
        }
        input.push_back(text.substr(start, end - start));
        int N = input.size();

        for(int i = 2; i < N; ++i)
        {
            if(input[i-2] == first && input[i-1] == second)
            {
                res.push_back(input[i]);
            }
        }
        return res;
    }
};
