class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        vector<string> words,ans;
        int n = text.size();
        string tem;
        for (int i=0; i<n; ++i)
        {
            if (text[i]==' ')
            {
                //cout<<"words:"<<tem<<endl;
                words.push_back(tem);
                tem = "";
            }
            else
                tem+=text[i];
        }
        if (tem != "") words.push_back(tem);
        for (int i=0; i<words.size(); ++i)
        {
            if (i>=2 && words[i-1] == second && words[i-2] == first)
            {
                //cout<<words[i-1]<<endl;
                ans.push_back(words[i]);
            }
        }
        return ans;
    }
};
