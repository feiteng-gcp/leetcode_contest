class Solution {
public:
    vector<string> split(const string &str) {
        vector<string> vec;
        string tmp;
        for (auto &c : str) {
            if (c == ' ') {
                if (tmp.size()) {
                    vec.push_back(tmp);
                }
                tmp.clear();
            } else
                tmp.push_back(c);
        }
        if (tmp.size()) vec.push_back(tmp);
        return vec;
    }
    vector<string> findOcurrences(string text, string first, string second) {
        vector<string> vec = split(text);
        if (vec.size() < 3) return {};
        vector<string> res;
        for (int i = 0; i < vec.size() - 2; ++i) {
            if (vec[i] == first && vec[i + 1] == second) {
                res.push_back(vec[i + 2]);
            }
        }
        return res;
    }
};
