ll po(ll a, ll b, ll mod)
{
    ll res = 1;
    a %= mod;
    for (; b; b >>= 1)
    {
        if (b & 1)
            res = res * a % mod;
        a = a * a % mod;
    }
    return res;
}
ll gcd(ll a, ll b)
{
    if (a == 0)
    {
        return b;
    }
    else
    {
        return gcd(b % a, a);
    }
}
void extgcd(ll a, ll b, ll &d, ll &x, ll &y)
{
    if (!b)
    {
        d = a;
        x = 1;
        y = 0;
    }
    else
    {
        extgcd(b, a % b, d, y, x);
        y -= x * (a / b);
    }
}
ll inverse(ll a, ll n)
{
    ll d, x, y;
    extgcd(a, n, d, x, y);
    return d == 1 ? (x + n) % n : -1;
}

class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        vector<string>v;
        string f;
        for(int i=0;i<text.length();i++) {
            if((text[i]>='A' && text[i]<='Z') || (text[i]>='a' && text[i]<='z')) {
                f.push_back(text[i]);
            } else {
                if(f.length()) {
                    v.push_back(f);
                }
                f="";
            }
        }

        if(f.length()) {
            v.push_back(f);
        }
        vector<string>ans;
        for(int i=2;i<v.size();i++) {
            if(v[i-1]==second && v[i-2]==first) {
                ans.push_back(v[i]);
            }
        }
        return ans;
    }
};
