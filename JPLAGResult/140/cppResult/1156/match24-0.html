class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        stringstream ss(text);
        string word;
        vector<string> words;
        while(ss>>word)
        {
            words.push_back(word);
        }
        if(words.size() < 3)
        {
            return {};
        }
        vector<string> res;
        for(int i = 0; i < words.size()-2; ++i)
        {
            if(words[i]==first and words[i+1]==second)
            {
                res.push_back(words[i+2]);
            }
        }
        return res;
    }
};
