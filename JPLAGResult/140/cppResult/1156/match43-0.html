class Solution {
public:
    vector<string> lis;
    vector<string> ret;
	vector<string> findOcurrences(string &text, string &first, string &second) {
		for (int i = 0; i < text.size();) {
			int cur = i;
			while (i != text.size() && text[i] != ' ') {
				i++;
			}
			if(i-cur>=1)lis.push_back(text.substr(cur, i - cur));
		    while(i<text.size()&&text[i]==' '){
                i++;
            }
        }
		for (int i = 0; i+2 < lis.size(); i++) {
			if (lis[i] == first && lis[i + 1] == second) {
				ret.push_back(lis[i + 2]);
			}
		}
		return ret;
	}
};
