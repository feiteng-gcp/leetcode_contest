class Solution {
public:
    vector<string> findOcurrences(string text, string first, string second) {
        istringstream str(text);
        vector<string> res;
        string temp;
        vector<string> vec_text;
        while(str>>temp){
            vec_text.push_back(temp);
        }
        for(int i = 0;i < vec_text.size()-2;++i){
            if(vec_text[i]==first && vec_text[i+1]==second){
                res.push_back(vec_text[i+2]);
            }
        }
        return res;
    }
};
