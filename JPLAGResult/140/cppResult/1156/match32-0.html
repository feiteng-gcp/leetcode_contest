class Solution {
public:
    vector <string> findOcurrences(string text, string first, string second) {
        istringstream in(text);
        vector <string> words;
        string word;
        while (in >> word) {
            words.push_back(word);
        }
        int L = words.size();
        vector <string> ans;
        for (int i = 0; i < (L - 2); i++) {
            if (words[i] == first && words[i + 1] == second) {
                ans.push_back(words[i + 2]);
            }
        }
        return ans;
    }
};
