class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        curr,sumLeftMin,sumLeftMax=0,0,0
        ret=0
        for num in arr:
            curr+=num
            sumLeftMin=min(sumLeftMin,curr)
            sumLeftMax=max(sumLeftMax,curr)
            ret=max(ret,curr-sumLeftMin)
        total=int(curr)
        if k==1:
            return ret%1000000007
        curr,sumRightMax=0,0
        for num in reversed(arr):
            curr+=num
            sumRightMax=max(sumRightMax,curr)
        ret=max(ret,sumLeftMax+sumRightMax,sumLeftMax+sumRightMax+(k-2)*total)%1000000007
        return ret
