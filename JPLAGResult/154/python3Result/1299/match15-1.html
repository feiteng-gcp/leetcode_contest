class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        if k == 0:
            return 0
        M = 1000000007
        
        def get(lt):
            t = 0
            s = 0
            for a in lt:
                if t < 0:
                    t = 0
                t 
+= a
                s = max(s, t)
            return s
        
        if k == 1:
            return get(arr) % M
        
        s = sum(arr)
        if s <= 0:
            return get(arr + arr) % M
        else:
            return (s * (k - 2) + get(arr + arr)) % M
