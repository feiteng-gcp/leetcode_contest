class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        mod = (10**9) + 7
        front = 0
        currSum = 0
        for i in arr[::-1]:
            currSum += i
            front = max(front, currSum)
        
        back = 0
        currSum = 0
        for i in arr:
            currSum += i
            back = max(back, currSum)
        
        mid = sum(arr)
        
        res = 0
        currSum = 0
        for i in arr:
            currSum += i
            if currSum < 0:
                currSum = 0
            res = max(res, currSum)
        
        if k == 1:
            return res % mod 
        
        res3 = front + back + ((k-2) * mid)
        if res3 < 0: res3 = 0
        return max(res, (front + back) % mod, res3 % mod)
