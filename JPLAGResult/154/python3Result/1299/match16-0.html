class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        
        def basesolve(A):
            ans = tail = A[0]
            for x in A[1:]:
                tail = max(tail + x, x)
                ans = max(ans, tail)
            return ans

        def maxleftsum(A):
            ans = A[0]
            s = 0
            for x in A:
                s += x
                ans = max(ans, s)
            return ans

        def solve(N, K, A):
            bA = basesolve(A)
            bL = maxleftsum(A)
            bR = maxleftsum(A[::-1])

            if K == 1: return bA
            S = sum(A)
            if S < 0:
                return max(bA, bL + bR)
            else:
                return max(bA, (K-2)*S + bL + bR)
        temp = solve(len(arr), k, arr)
        return 0 if temp<0 else temp%1000000007
