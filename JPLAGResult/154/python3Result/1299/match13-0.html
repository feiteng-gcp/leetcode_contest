class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        def fm(arr):
            ms = 0
            mb = 0
            cur = 0
            begin = 0
            for i, a in enumerate(arr):
                cur += a
                if cur < 0:
                    cur = 0
                    begin = i + 1
                elif cur > ms:
                    ms = cur
                    mb = begin
            return ms
                
        ans = [fm(arr)]
        b = 0
        cur = 0
        for a in arr:
            cur += a
            b = max(b, cur)
        e = 0
        cur = 0
        for a in arr[::-1]:
            cur += a
            e = max(e, cur)
        if k >= 2:
            s = sum(arr)
            ans.append(b + e)
            ans.append((k - 2) * s + b + e)
            ans.append((k - 1) * s + b)
            ans.append((k - 1) * s + e)
            ans.append(k * s)
        return max(ans) % int(1e9 + 7)
