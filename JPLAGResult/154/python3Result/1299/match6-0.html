class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        if k <= 0:
            return 0
        
        mod = 10 ** 9 + 7
        
        res = 0
        cur = 0
        for a in arr:
            cur += a
            if cur < 0:
                cur = 0
            res = max(res, cur)
        if k == 1:
            return res % mod
        
        l, cur = 0, 0
        for a in reversed(arr):
            cur += a
            l = max(cur, l)
        r, cur = 0, 0
        for a in arr:
            cur += a
            r = max(r, cur)
        
        if sum(arr) > 0:
            return max(res, l + r + sum(arr) * (k - 2)) % mod
        return max(res, l + r) % mod
