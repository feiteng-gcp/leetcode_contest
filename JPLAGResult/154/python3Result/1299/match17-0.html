mod = 10**9 + 7

class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        if all(i <= 0 for i in arr): return 0
        
        if k == 1:
            s = e = arr[0]
            for i in arr[1:]:
                s = max(s + i, i)
                e = max(e, s)
            return max(e, 0)
            
        a = arr + arr
        
        s = e = a[0]
        for i in a[1:]:
            s = max(s + i, i)
            e = max(e, s)
            
        if k == 2:
            return e % mod
        
        if k == 0: return 0
        

        def fwd(arr):
            sm = 0
            res = -float('inf')
            for i in arr:
                sm += i
                res = max(res, sm)
                
            return res
        
        f = sum(arr)
        if f <= 0: return max(e, 0) % mod
        
        if f > 0:
            res = fwd(arr) + fwd(arr[::-1]) + f * (k - 2)
            return max(res, 0) % mod
