class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        arrsm = sum(arr)
        arr2 = arr + arr if k > 1 else arr
        curmax = 0
        cursum = 0
        for n in arr2:
            cursum += n
            if cursum < 0:
                cursum = 0
            curmax = max(cursum, curmax)
        if k <= 2:
            return curmax % 1000000007
        return max(curmax, curmax + arrsm * (k - 2)) % 1000000007
