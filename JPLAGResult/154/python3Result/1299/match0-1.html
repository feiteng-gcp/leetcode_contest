class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        MOD = 10**9 + 7
        S = sum(arr)
        i = 0
        s = 0; cur 
= 0
        ans = ma = 0
        for v in arr:
            cur += v
            if cur < s:
                s = cur
            ans = max(ans, cur - s)
        if k == 1:
            return ans % MOD

        for v in arr:
            cur += v
            if cur < s:
                s = cur
            ma = max(ma, cur - s)
        if S > 0:
            ans = max(ans, ma + (k-2) * S)
        else:
            ans = max(ans, ma)
        return ans % MOD
