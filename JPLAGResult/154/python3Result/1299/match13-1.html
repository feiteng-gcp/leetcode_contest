class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        MOD = 10 ** 9 + 7
        m1 = m2 = 0
        
        curr = arr[0]
        m1 = max(curr, 0)        
        for v in arr[1:]:
            curr = max(curr, 0) + v
            m1 = max(m1, curr)
        
        pm 
= 0
        curr = 0
        for v in arr:
            curr += v
            pm = max(pm, curr)
            
        sm = 0
        curr = 0
        for v in arr[::-1]:
            curr += v
            sm = max(sm, curr)
        
        if k >= 2:
            m2 = pm + sm + (k - 2) * max(0, sum(arr))
            
        return max(m1, m2) % MOD
