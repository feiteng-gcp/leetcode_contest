class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        if k == 1:
            return self.vanilla(arr)
        
        double_arr = arr * 2
        max_seen, max_end = 0,0
        mod = 10 ** 9 + 7
        
        for i in double_arr:
            max_end = max(i, i + max_end)
            max_seen = max(max_seen, max_end)
        
        max_seen = max_seen
        
        max_total = sum(arr)
        
        if max_total < max_seen and max_total < 0:
            return max_seen % mod
        if k <= 2:
            max_total_k = ((max_total % mod) * (k % mod)) % mod
            return max_total_k if max_total_k > max_seen else max_seen
        else:
            back_seen, back_total 
= 0,0
            for i in reversed(arr):
                back_seen += i
                back_total = max(back_total, back_seen)
            forward_seen, forward_total = 0,0
            for i in arr:
                forward_seen += i
                forward_total = max(forward_total, forward_seen)
            
            max_total_k = ((max_total % mod) * ((k - 2) % mod)) % mod
            max_total_k += (back_total + forward_total) % mod 
            return max_total_k if max_total_k > max_seen else max_seen
    def vanilla(self, arr):
        max_seen, max_end = 0,0
        mod = 10 ** 9 + 7
        for i in arr:
            max_end = max(i, i + max_end)
            max_seen = max(max_seen, max_end)
        
        return max_seen % mod
