class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        M = 10**9 + 7
        
def maxsub(arr):
            agg = ans = arr[0]
            for x in arr[1:]:
                agg = max(x, agg+x)
                ans = max(ans, agg)
            return max(ans, 0) % M
        S = sum(arr)
        if S <= 0:
            return maxsub(arr*(1+(k>1)))
        else:
            return maxsub(arr) if k == 1 else (maxsub(arr*2) + S*(k-2)) % M
