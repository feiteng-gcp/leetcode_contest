class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        total = sum(arr)
        max_r = 0
        max_l = 0
        runner = 0
        for i in arr:
            runner+=i
            max_l = max(max_l, runner)
        runner = 0
        for i in arr[::-1]:
            runner+=i
            max_r = max(max_r, runner)
        # single arr max sum
        runner 
= 0
        max_single = 0
        for i in arr:
            runner+=i
            if runner<0:
                runner = 0
            max_single = max(max_single, runner)
        if k==1:
            return max_single
        else:
            return max(max_single, max_l+max_r, max_l+max_r+(k-2)*total)%(10**9 + 7)
