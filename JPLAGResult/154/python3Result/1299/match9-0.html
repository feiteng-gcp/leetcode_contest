class Solution:
    def kConcatenationMaxSum(self, arr: List[int], k: int) -> int:
        if max(arr) <= 0:
            return 0
        new_arr = arr*min(k, 3)
        ans = 0
        total = 0
        for i, num in enumerate(new_arr):
            total = max(total+num, num)
            ans = max(total, ans)
        max_left = 0
        max_right = 0
        total = 0
        for num in arr:
            total += num
            max_left = max(max_left, total)
        total = 0
        for num in arr[::-1]:
            total += num
            max_right = max(max_right, total)
        MOD = 10**9+7
        if k <= 3:
            return ans%MOD
        return max(ans, sum(arr)*(k-2) + max_left+max_right)%MOD
