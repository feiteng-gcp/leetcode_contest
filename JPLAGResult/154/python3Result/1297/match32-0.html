class Solution:
    def maxNumberOfBalloons(self, text: str) -> int:
        bcount = self.countNumChars(text, "b")
        acount = self.countNumChars(text, "a")
        lcount = self.countNumChars(text, "l")
        ocount = self.countNumChars(text, "o")
        ncount = self.countNumChars(text, "n")
        
        return min([bcount, acount, lcount//2, ocount//2, ncount])
    
    def countNumChars(self, text, char):
        count = 0
        for c in text:
            if char == c:
                count += 1
        return count
