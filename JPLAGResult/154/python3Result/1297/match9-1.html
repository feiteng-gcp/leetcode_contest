class Solution:
    def maxNumberOfBalloons(self, text: str) -> int:
        mm = {
            "b": 0,
            "a": 0,
            "l": 0,
            "o": 0,
            "n": 0
        }
        for s in text:
            if s in mm:
                mm[s] += 1
        rs = float("inf")
        for s, v in mm.items():
            if s in ("b", "a", "n"):
                rs = min(rs, v)
            else:
                rs = min(rs, v // 2)
        return rs if rs != float("inf") else 0
