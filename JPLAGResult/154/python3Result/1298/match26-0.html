class Solution:
    def reverseParentheses(self, s: str) -> str:
        stack = []
        i = 0 
        while(i < len(s)):
            #print(stack)
            if s[i] == ')':
                lc= []
                while stack and stack[-1] != '(':
                    lc.append(stack.pop())
                if stack:
                    stack.pop()
                stack = stack + lc
            else:
                stack.append(s[i])
            i +=1
            
        return ''.join(stack)
