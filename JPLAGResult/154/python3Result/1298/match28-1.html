class Solution:
    def reverseParentheses(self, s: str) -> str:
        res = ''
        level = 0
        for i, c in enumerate(s):
            if c=='(':
                if level == 0: start = i
                level += 1
            elif c == ')':
                level -= 1
                if level == 0: 
                    end = i
                    substr = self.reverseParentheses(s[start+1:end])
                    res += substr[::-1]
                    
            elif level == 0: res += c
        return res
