class Solution:
    def reverseParentheses(self, s: str) -> str:
        stack = []
        
        for c in s:
            if c != ')':
                stack.append(c)
            else:
                tmp = []
                while stack[-1] != '(':
                    tmp.append(stack[-1])
                    stack.pop()
                stack.pop()
                for cc in tmp:
                    stack.append(cc)
        return "".join(stack)
