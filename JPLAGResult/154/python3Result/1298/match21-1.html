class Solution:
    def reverseParentheses(self, s: str) -> str:
        stack = []
        for c in s:
            if c == "(":
                stack.append(c)
            elif c == ")":
                temp = []
                while stack:
                    c = stack.pop()
                    if c == "(":
                        break
                    temp.append(c[
::-1])
                stack.append("".join(temp))
            else:
                stack.append(c)
        return "".join(stack)
