class Solution:
    def reverseParentheses(self, s: str) -> str:
        if not s:
            return s
        stack = []
        for c in s:
            if c != ")":
                stack.append(c)
            else:
                cand = []
                while stack[-1] != "(":
                    cand.append(stack.pop()[::-1])
                stack.pop()
                stack.append("".join(cand))
        return "".join(stack)
