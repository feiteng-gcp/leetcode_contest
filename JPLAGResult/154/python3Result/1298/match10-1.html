class Solution:
    def reverseParentheses(self, s: str) -> str:
        stack = ['']
        for c in s:
            if c=='(':
                stack.append('')
            elif c!=')':
                stack[-1] += c
            else:
                elem 
= stack.pop()
                stack[-1] += elem[::-1]
        
        return stack.pop()
