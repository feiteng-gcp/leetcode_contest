class Solution:
    def reverseParentheses(self, s: str) -> str:
        st = []
        for c in s:
            if c == ')':
                A = []
                while st[-1] != '(':
                    A.append(st.pop())
                st.pop()
                st.append((''.join(A[::-1]))[::-1])
            else:
                st.append(c)
        return ''.join(st)
