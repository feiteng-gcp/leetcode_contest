class Solution:
    def reverseParentheses(self, s: str) -> str:
        st = []
        i, n = 0, len(s)
        while i < n:
            if s[i] == '(':
                i += 1
                st.append("(")
            elif s[i] == ')':
                w = ""
                while st and st[-1] != '(':
                    w += st.pop()
                if st:
                    st.pop()
                st.append(w[::-1])
                i += 1
            else:
                st.append(s[i])
                i += 1
        return "".join(x[::-1] for x in st)
