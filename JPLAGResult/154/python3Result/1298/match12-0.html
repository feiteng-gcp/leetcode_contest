class Solution:
    def reverseParentheses(self, s: str) -> str:
        stack = []
        queue = collections.deque()
        for _ in s:
            if _ == ')':
                while stack and stack[-1] != '(':
                    queue.append(stack.pop())
                if stack:
                    stack.pop()
                while queue:
                    stack.append(queue.popleft())
            else:
                stack.append(_)
        return ''.join(stack)
