class Solution:
    def reverseParentheses(self, s: str) -> str:
        b = 0
        ans = []
        for i, c in enumerate(s):
            if c == '(':
                if b == 0:
                    begin = i
                b += 1
            elif c == ')':
                b -= 1
                if b == 0:
                    r = ''.join(['(' if cc == ')' else (')' if cc == '(' else cc) for cc in s[i - 1: begin: -1]])
                    ans.append(self.reverseParentheses(r))
            elif b == 0:
                ans.append(c)
        ans = ''.join(ans)
        return ans
