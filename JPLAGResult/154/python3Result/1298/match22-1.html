class Solution:
    def reverseParentheses(self, s: str) -> str:
        if not s:
            return ''
        st = []
        for c in s:
            if c == ')':
                temp = []
                while st and st[-1] != '(':
                    temp.append(st.pop())
                st
.pop()
                st.extend(temp)
            else:
                st.append(c)
                
        return ''.join(st)
