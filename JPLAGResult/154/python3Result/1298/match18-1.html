class Solution:
    def reverseParentheses(self, s: str) -> str:
        chars = list(s)
        s = []
        
        for i, c in enumerate(chars):
            if c == '(':
                s.append(i)
            elif c == ')':
                j 
= s.pop()
                chars[j:i] = chars[i:j:-1]

        return ''.join(c for c in chars if c not in '()')
