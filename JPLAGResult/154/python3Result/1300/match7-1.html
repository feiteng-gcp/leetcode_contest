class Solution:
    def criticalConnections(self, n: int, connections: List[List[int]]) -> List[List[int]]:
        adj 
= [[] for _ in range(n)]
        for u, v in connections:
            adj[u].append(v)
            adj[v].append(u)
        ans = []
        visited = [False] * n
        disc = [float("inf")] * n
        low = [float("inf")] * n
        parent = [-1] * n
        self.t = 0
        
        def dfs(u):
            visited[u] = True
            disc[u] = self.t
            low[u] = self.t
            self.t += 1
            for v in adj[u]:
                if not visited[v]:
                    parent[v] = u
                    dfs
(v)
                    low[u] = min(low[u], low[v])
                    if low[v] > disc[u]:
                        ans.append([u, v])
                elif v != parent[u]:
                    low
[u] = min(low[u], disc[v])
        
        for i in range(n):
            if not visited[i]:
                dfs(i)
        
        return ans
