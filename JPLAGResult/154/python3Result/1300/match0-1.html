class Solution:
    def criticalConnections(self, n: int, connections: List[List[int]]) -> List[List[int]]:
        graph = collections.defaultdict(set)
        for u, v in connections:
            graph[u].add(v)
            graph[v].add(u)
        visited = set()
        bridges = []
        seen = [float('inf')] * n
        low = [float('inf')] * n
        parent = [-1] * n
        self
.cnt = 0
        
        def dfs(x):
            visited.add(x)
            seen[x] = low[x] = self.cnt
            self.cnt += 1
            for y in graph[x]:
                if y not in visited:
                    parent[y] = x
                    dfs(y)
                    low[x] = min(low[x], low[y])
                    if low[y] > seen[x]:
                        bridges.append([x, y])
                elif y != parent[x]:
                    low[x] = min(low[x], seen[y])
                    
        for i in range(n):
            if i not in visited:
                dfs(i)
        return bridges
