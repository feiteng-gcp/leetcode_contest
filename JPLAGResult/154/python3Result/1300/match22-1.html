class Solution:
    def criticalConnections(self, n: int, connections: List[List[int]]) -> List[List[int]]:
        bridges = []
        g = [[] for _ in range(n)]
        for (u, v) in connections:
            g[u].append(v)
            g[v].append(u)
        low = [-1] * n
        pre = [-1] * n
        self.cnt = 0

        def dfs(u, v):
            # print(u, v)
            pre
[v] = self.cnt
            low[v] = self.cnt
            self.cnt += 1

            for w in g[v]:
                if pre[w] == -1:
                    dfs
(v, w)
                    low[v] = min(low[v], low[w])
                    if low[w] == pre[w]:
                        bridges.append([v, w])
                elif w != u:
                    low
[v] = min(low[v], pre[w])

        for u in range(n):
            if pre[u] == -1:
                dfs(u, u)

        # print(bridges)
        return bridges
