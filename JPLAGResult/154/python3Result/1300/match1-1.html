class Solution:
    def criticalConnections(self, n: int, connections: List[List[int]]) -> List[List[int]]:
        adj_list = collections.defaultdict(list)
        visited = set()
        disc = [-math.inf] * n
        low = [-math.inf] * n
        parent = [-1] * n
        res = []
        
        for out, inc in connections:
            adj_list[out].append(inc)
            adj_list[inc].append(out)
        self.disc_time = 0
        def dfs(node):
            visited.add(node)
            disc[node] = self.disc_time
            low
[node] = self.disc_time
            
            self.disc_time += 1
            
            for v in adj_list[node]:
                if v not in visited: 
                    parent[v] = node
                    dfs(v) 

                    low[node] = min(low[node], low[v]) 

                    if low[v] > disc[node]: 
                       res.append([v,node])
      
                elif v != parent[node]:
                    low[node] = min(low[node], disc[v]) 
        
        for k in range(n):
            if k not in visited:
                dfs(k)
                
        return res
