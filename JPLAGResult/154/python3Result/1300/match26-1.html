class Solution:
    def criticalConnections(self, n: int, connections: List[List[int]]) -> List[List[int]]:
        graph 
= [[] for _ in range(n)]
        for u, v in connections:
            graph[u].append(v)
            graph[v].append(u)
        visited = [False]*n
        tin = [-1]*n
        low = [-1]*n
        time = 0
        res = []
        def dfs(v, p=-1):
            nonlocal time
            visited[v] = True
            tin[v] = low[v] = time
            time += 1
            for neighbor in graph[v]:
                if neighbor == p:
                    continue
                if visited[neighbor]:
                    low[v] = min(low[v], tin
[neighbor])
                else:
                    dfs(neighbor, v)
                    low[v] = min(low[v], low[neighbor])
                    if low[neighbor]>tin[v]:
                        res.append([v,
 neighbor])
        for i in range(n):
            if not visited[i]:
                dfs(i)
        return res
