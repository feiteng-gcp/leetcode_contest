class Solution:
    def criticalConnections(self, n: int, connections: List[List[int]]) -> List[List[int]]:
        parent = [-1]*n
        graph = [[] for _ in range(n)]
        low = [float("inf")]*n
        disc = [float("inf")]*n
        visited = [False]*n
        for i, j in connections:
            graph[i].append(j)
            graph[j].append(i)
        
        val 
= 0
        res = []
        def traverse(u):
            nonlocal val,res
            visited[u] = True
            low[u] = val
            disc[u] = val
            val += 1
            for v in graph[u]:
                if visited[v] == False:
                    parent[v] = u
                    traverse(v)
                    low[u] = min(low[u], low[v])
                    if low[v] > disc[u]:
                        res.append([u, v])
                elif parent[u] != v:
                    low[u] = min(low[u], disc[v])
        for i in range(n):
            if visited[i] == False:
                traverse(i)
        
        return res
