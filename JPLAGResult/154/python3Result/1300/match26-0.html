from collections import defaultdict

class Solution:
    def criticalConnections(self, n: int, connections: List[List[int]]) -> List[List[int]]:
        if n == 0: return []
        graph = defaultdict(set)
        
        for a, b in connections:
            graph[a].add(b)
            graph[b].add(a)
        
        low = [-1] * n
        pre = [-1] * n
        
        cnt = [0]
        res = []
        def dfs(u, v):
            pre[v] = cnt[0]
            cnt[0] += 1
            low[v] = pre[v]
            
            for neighbor in graph[v]:
                if pre[neighbor] == -1:
                    dfs(v, neighbor)
                    low[v] = min(low[v], low[neighbor])
                    if low[neighbor] == pre[neighbor]:
                        res.append([v, neighbor])
                elif neighbor != u:
                    low[v] = min(low[v], pre[neighbor])
        
        for i in range(n):
            if pre[i] == -1:
                dfs(i, i)
                
        return res
