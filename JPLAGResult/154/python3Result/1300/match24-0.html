class Solution:
    def criticalConnections(self, n: int, connections):
        G = [[] for _ in range(n)]
        for x, y in connections:
            G[x].append(y)
            G[y].append(x)
        ans = []
        V = [False] * n
        dfsn = [-1] * n
        dfs_low = [-1] * n
        parent = [-1] * n
        cnt = [0]

        def dfs(cur):
            V[cur] = True
            dfsn[cur] = [cnt[0]]
            dfs_low[cur] = [cnt[0]]
            cnt[0] += 1

            for v in G[cur]:
                if V[v] == False:
                    parent[v] = cur
                    dfs(v)
                    dfs_low[cur] = min(dfs_low[cur], dfs_low[v])

                    if dfs_low[v] > dfsn[cur]:
                        ans.append([cur, v])
                elif v != parent[cur]:
                    dfs_low[cur] = min(dfs_low[cur], dfsn[v])

        dfs(0)
        return list(ans)
