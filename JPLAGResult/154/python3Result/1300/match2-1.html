from collections import defaultdict


class Solution:
    def criticalConnections(self, n: int, connections: List[List[int]]) -> List[List[int]]:
        
        
        g = defaultdict(list)
        for u,v in connections:
            g[u].append(v)
            g[v].append(u)
            
        visited = [False]*n
        p = [-1]*n
        d = [float("inf")]*n
        t = [float("inf")]*n
        time = 0
        ans = []
        
        def dfs(u):
            nonlocal time
            
            visited[u] = True
            d[u] = time
            t[u] = time
            time 
+= 1
            
            for v in g[u]:
                if not visited[v]:
                    p[v] = u
                    dfs(v)
                    
                    t[u] = min(t[u], t[v])
                    if t[v] > d[u]:
                        ans.append((u,v))
                elif v != p[u]:
                    t[u] = min(t[u], d[v])
                    
        for i in range(n):
            if not visited[i]:
                dfs(i)
        return ans
