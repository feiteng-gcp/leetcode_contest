class Solution:
    def criticalConnections(self, n: int, c: List[List[int]]) -> List[List[int]]:
        
        def helper(u):   
            nonlocal ret, e, t, seen, d, low, father
            seen[u] = True
            d[u] = t
            low[u] = t
            t 
+= 1
            for v in e[u]: 
                if seen[v] == False: 
                    father[v] = u 
                    helper(v)
                    low[u] = min(low[u], low[v]) 
                    if low[v] > d[u]: 
                        ret.append([u, v])
                elif v != father[u]: 
                    low[u] = min(low[u], d[v]) 
        
        ret = []
        e 
= [[] for i in range(n)]
        for a, b in c:
            e[a].append(b)
            e[b].append(a)            
        t = 0
        seen = [False] * n
        d = [float('inf')] * n
        low = [float('inf')] * n
        father = [-1] * n
        for i in range(n):
            if not seen[i]:
                helper(i)
        return ret
