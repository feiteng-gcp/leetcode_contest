class Solution:
    def criticalConnections(self, n: int, connections: List[List[int]]) -> List[List[int]]:
        g = [[] for _ in range(n)]
        for e in connections:
            g[e[0]].append(e[1])
            g[e[1]].append(e[0])
        time = [0]
        disc = [math.inf] * n
        low = [math.inf] * n
        par = [-1] * n
        visited = [False] * n
        ret = []
        
        
        def dfs(node):
            visited[node] = True
            disc[node] = time[0]
            low[node] = time[0]
            time[0]+=1
            for n in g[node]:
                if not visited[n]:
                    par[n]=node
                    dfs(n)
                    low[node] = min(low[n], low[node])
                    if low[n]>disc[node]:
                        ret.append([node,n])
                elif n!=par[node]:
                    low[node]=min(low[node], low[n])
                    
                    
        for i in range(n):
            if not visited[i]: dfs(i)
        return ret
