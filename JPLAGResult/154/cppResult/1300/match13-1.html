class Solution {
public:
    int n, m;
    const int maxn = 1e5+5;

    vector<int> G[100005];
    int low[100005],pre[100005];
    int dfs_clock;     //时间戳
    int iscut[100005];   //标记是否为割顶
    vector<vector<int>> ans;
    int dfs(int u,int fa)
    {
        int lowu = pre[u] = ++dfs_clock;
        int child = 0;
        for(int i=0;i<G[u].size();i++)
        {
            int v = G[u][i];
            if(!pre[v])     //没有访问的v
            {
                child++;    //孩子节点的数目
                int lowv = dfs(v,u);
                lowu = min(lowu,lowv);    //用后代更新lowu
                if(lowv >= pre[u]) iscut[u] = 1;
                if(lowv > pre[u]) {

                    ans.push_back({u,v});
                }
            }
            else if(pre[v] < pre[u] && v != fa)  //用反向边更新lowu
            {
                lowu = min(lowu,pre[v]);
            }
        }
        if(fa < 0 && child == 1) iscut[u] = 0;    //对于根节点的处理
        low[u] = lowu;
        return lowu;
    }
    vector<vector<int>> criticalConnections(int tn, vector<vector<int>>& connections) {
        m = connections.size();
        n = tn;
        for(int i = 0; i < m; i++) {
            G[connections[i][0]].push_back(connections[i][1]);
            G[connections[i][1]].push_back(connections[i][0]);
        }
        dfs(0, -1);
        return ans;
    }
};
