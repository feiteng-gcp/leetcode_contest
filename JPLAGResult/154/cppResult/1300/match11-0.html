const int maxn=100000+10;
    int n,m;
    int dfs_clock;
    vector<int> G[maxn];
    int pre[maxn];
    int low[maxn];
    bool iscut[maxn];
class Solution {
public:
    
    
    int dfs(int u,int fa,vector<vector<int>>& ret)
    {
        int lowu=pre[u]=++dfs_clock;
        int child=0;   
        for(int i=0; i<G[u].size(); i++)
        {
            int v=G[u][i];
            if(!pre[v])
            {
                child++;
                int lowv=dfs(v,u,ret);
                lowu=min(lowu,lowv);
                if(lowv>=pre[u])
                {
                    iscut[u]=true;     
                    if(lowv>pre[u])   {
                        vector<int> temp({u,v});
                        ret.push_back(temp);
                    }
                        
                }
            }
            else if(pre[v]<pre[u] && v!=fa)
            {
                lowu=min(lowu,pre[v]);
            }
        }
        if(fa<0 && child==1 )
            iscut[u]=false;
        return low[u]=lowu;
    }
    vector<vector<int>> criticalConnections(int t_n, vector<vector<int>>& connections) {
        dfs_clock=0;
        n=t_n;
        m=connections.size();
        vector<vector<int>> ret;
        for(int i=0;i<n;i++){
             G[i].clear();
            pre[i]=0;
            iscut[i]=0;
        }
           
        for(auto connect:connections){
            G[connect[0]].push_back(connect[1]);
            G[connect[1]].push_back(connect[0]);
        }
        dfs(0,-1,ret);
        return ret;
    }
};
