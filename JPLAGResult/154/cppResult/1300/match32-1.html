#define MAXV 150000

struct edge {
    int v, _next;
};
edge edges[MAXV*2];
int ecnt;
int head[MAXV];

void init() {
    memset(head, -1, sizeof(head));
    ecnt = 0;
}

void add_edge(int u, int v) {
    edges[ecnt].v = v;
    edges[ecnt]._next = head[u];
    head[u] = ecnt++;
}

class Solution {
public:
    

    int tot;
    vector<int> dfn;
    vector<int> low;
    vector<vector<int>> ans;
    
    void tarjan(int v, int fa) {
        dfn[v] = low[v] = ++tot;
        for (int i=head[v]; ~i; i=edges[i]._next) {
            int u = edges[i].v;

            if (!dfn[u]) {
                tarjan(u, i);
                low[v] = min(low[v], low[u]);
                if (low[u] > dfn[v]) {
                    ans.push_back({u, v});
                }
            } else if (i!=(fa^1)) {
                low[v] = min(low[v], dfn[u]);
            }
        }
        
    }
    
    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& conns) {
        init();
        dfn = vector<int>(n+1, 0);
        low = vector<int>(n+1, 0);
        ans = {};
        tot = 0;
        for (const auto &pr: conns) {
            add_edge(pr[0], pr[1]);
            add_edge(pr[1], pr[0]);
        }
        tarjan(0, -1);
        return ans;
    }
};
