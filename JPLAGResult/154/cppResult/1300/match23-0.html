class Solution {
public:
    int idx = 0;
    vector<int> dfn, low, pre;

    vector<vector<int>> g;
    void tarjan(int u, int p) {
        pre[u] = p;
        dfn[u] = low[u]= idx++;

        for (int i = 0; i < g[u].size(); ++i) {
            int v = g[u][i];
            if (dfn[v] == -1) {
                tarjan(v, u);
                low[u] = min(low[u], low[v]);
            } else if (v != p)
                low[u] = min(low[u], dfn[v]);
        }
    }

    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& con) {
        g = vector<vector<int>>(n);
        dfn = vector<int>(n, -1);
        low = vector<int>(n);
        pre = vector<int>(n);

        for (int i = 0; i < con.size(); ++i) {
            int a = con[i][0], b = con[i][1];
            g[a].push_back(b);
            g[b].push_back(a);
        }
        
        tarjan(0, -1);

        vector<vector<int>> res;
        for (int i = 0; i < n; ++i)
            if (pre[i] != -1) {
                int p = pre[i];
                if (low[i] > dfn[p])
                    res.push_back({p, i});
            }
        
        return res;
    }
};
