class Solution {
public:
    void tarjan(vector<int>& dfn, vector<int>& low, vector<int>& father, int &tim, vector<vector<int>>& conn, int i, int f) {
        father[i] = f;
        dfn[i] = low[i] = tim++;
        for (auto &&n: conn[i]) {
            if (dfn[n] == -1) {
                tarjan(dfn, low, father, tim, conn, n, i);
                low[i] = min(low[i], low[n]);
            } else if (f != n)
                low[i] = min(low[i], dfn[n]);
        }
    }
    
    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
        vector<int> dfn(n, -1);
        vector<int> low(n, -1);
        vector<int> father(n, -1);
        vector<vector<int>> conn(n, vector<int>());
        int tim = 0;
        for (auto && c: connections) {
            conn[c[0]].push_back(c[1]);
            conn[c[1]].push_back(c[0]);
        }
        tarjan(dfn, low, father, tim, conn, 0, -1);
        vector<vector<int>> ret;
        for(int i = 0; i < n; i++)
        {
            int v = father[i];
            if (v >= 0 && low[i] > dfn[v])
                ret.push_back({i, v});
        }
        return ret;
    }
};
