class Solution {
public:
    // PRE-WRITTEN CODE
    int n; 
    vector<int> adj[112345]; 

    vector<bool> visited;
    vector<int> tin, low;
    int timer;
    vector<vector<int>> res;

    void dfs(int v, int p = -1) {
        visited[v] = true;
        tin[v] = low[v] = timer++;
        for (int to : adj[v]) {
            if (to == p) continue;
            if (visited[to]) {
                low[v] = min(low[v], tin[to]);
            } else {
                dfs(to, v);
                low[v] = min(low[v], low[to]);
                if (low[to] > tin[v]) {
                    vector<int> vv;
                    vv.push_back(v);
                    vv.push_back(to);
                    res.push_back(vv);
                }
                    
            }
        }
    }

    void find_bridges() {
        timer = 0;
        visited.assign(n, false);
        tin.assign(n, -1);
        low.assign(n, -1);
        for (int i = 0; i < n; ++i) {
            if (!visited[i])
                dfs(i);
        }
    }
    // END OF PRE-WRITTEN CODE    
    
    vector<vector<int>> criticalConnections(int _n, vector<vector<int>>& v) {
        n = _n;
        for (int i = 0; i < v.size(); i++) {
            int a = v[i][0];
            int b = v[i][1];
            adj[a].push_back(b);
            adj[b].push_back(a);
        }
        
        find_bridges();
        return res;
    }
};
