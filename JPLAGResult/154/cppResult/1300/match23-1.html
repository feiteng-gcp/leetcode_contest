class Solution {
public:
    int t = 0, p[100000], low[100000], dfn[100000];
    vector<int> G[100000];
    vector<vector<int>> ans;
    void tarjan(int u, int parent) {
        p[u] = parent;
        dfn[u] = low[u] = t++;
        for(int i = 0; i < G[u].size(); i++) {
            int v = G[u][i];
            if(dfn[v] == -1) {
                tarjan(v, u);
                low[u] = min(low[u], low[v]);
            }
            else if(v != p[u]) low[u] = min(low[u], dfn[v]);
        }
    }
    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
        memset(dfn, -1, sizeof(dfn));
        memset(low, -1, sizeof(low));
        memset(p, -1, sizeof(p));
        for (int i = 0; i < connections.size(); i++) {
            int u = connections[i][0], v = connections[i][1];
            G[u].push_back(v);
            G[v].push_back(u);
        }
        tarjan(0, -1);
        for(int i = 0; i < n; i++) {
            if(p[i] >= 0 && low[i] > dfn[p[i]]) {
                vector<int> e;
                e.push_back(i);
                e.push_back(p[i]);
                ans.push_back(e);
            }
        }
        return ans;
    }
};
