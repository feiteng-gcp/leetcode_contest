class Solution {
public:
  vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
    Graph graph(n, connections);

    return graph.critical();
  }

private:
  class Graph {
    public:
      int n;
      vector<vector<int>> edge;

      vector<vector<int>> ans;

      Graph(int _n, vector<vector<int>>& connections) {
        n = _n;
        edge = vector<vector<int>>(n, vector<int>());

        for (int i = 0; i < connections.size(); ++i) {
          int x = connections[i][0];
          int y = connections[i][1];
          edge[x].push_back(y);
          edge[y].push_back(x);
        }

        return;
      }

      vector<vector<int>> critical() {
        bool *visited = new bool[n];
        int *disconnected = new int[n];
        int *low = new int[n];
        int *parent = new int[n];

        for (int i = 0; i < n; ++i) {
          parent[i] = -1;
          visited[i] = false;
        }

        for (int i = 0; i < n; ++i) {
          if (visited[i]) continue;
          findBridge(i, visited, disconnected, low, parent);
        }

        return vector<vector<int>>(ans);
      }

      void findBridge(int i, bool *visited, int *disconnected, int *low, int *parent) {
        static int cnt = 0;

        visited[i] = true;

        ++cnt;
        disconnected[i] = cnt;
        low[i] = cnt;

        for (int it = 0; it < edge[i].size(); ++it) {
          int j = edge[i][it];

          if (!visited[j]) {
            parent[j] = i;
            findBridge(j, visited, disconnected, low, parent);

            low[i] = min(low[i], low[j]);

            if (low[j] > disconnected[i]) {
              vector<int> tmp;
              tmp.push_back(i);
              tmp.push_back(j);
              ans.push_back(tmp);
            }
          }
          else if (parent[i] != j) {
            low[i] = min(low[i], disconnected[j]);
          }
        }
      }
  };
};
