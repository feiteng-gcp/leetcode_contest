vector<vector<int>> res;
vector<int> adj[100005];
bool vis[100005];
int tin[100005],low[100005];
int timer;

void dfs(int v, int p = -1) {
    vis[v] = true;
    tin[v] = low[v] = timer++;
    for (int to : adj[v]) {
        if (to == p) continue;
        if (vis[to]) {
            low[v] = min(low[v], tin[to]);
        } else {
            dfs(to, v);
            low[v] = min(low[v], low[to]);
            if (low[to] > tin[v]){
                vector<int> edg;
                edg.push_back(to);edg.push_back(v);
                res.push_back(edg);
            }
        }
    }
}


class Solution {
public:
    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& cons) {
        res.clear();
        for(int i=0; i<100005; i++) adj[i].clear();
        for(int i=0; i<cons.size(); i++){
            vector<int> edg = cons[i];
            adj[edg[0]].push_back(edg[1]);
            adj[edg[1]].push_back(edg[0]);
        }
        timer = 0;
        memset(vis,false,sizeof vis);
        for(int i=0; i<100005; i++) tin[i] = -1;
        for(int i=0; i<100005; i++) low[i] = -1;
        for (int i = 0; i < n; ++i) {
            if (!vis[i])
                dfs(i);
        }      
        return res;
    }
};
