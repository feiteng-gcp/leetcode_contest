class Solution {
public:

    void find_bridges(int u, bool *visited, vector<int> &disc, vector<int> &low, vector<int> &parent, vector<vector<int>> &edges, int &time, vector<vector<int>> &bridges) {
        visited[u] = true;
        time++;
        disc[u] = time;
        low[u] = time;
        for (int v : edges[u]) {
            if (!visited[v]) {
                parent[v] = u;
                find_bridges(v, visited, disc, low, parent, edges, time, bridges);
                low[u] = min(low[u], low[v]);
                if (low[v] > disc[u]) {
                    bridges.push_back(vector<int>{u, v});
                }
            }
            else if (v != parent[u]) {
                low[u] = min(low[u], disc[v]);
            }
        }
        return;
    }
    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
        vector<vector<int>> edges(n);
        for (auto &edge : connections) {
            edges[edge[0]].push_back(edge[1]);
            edges[edge[1]].push_back(edge[0]);
        }
        vector<int> disc(n), low(n), parent(n, -1);
        bool visited[n];
        memset(visited, 0, n * sizeof(bool));

        vector<vector<int>> bridges;
        int time = 0;
        for (int u = 0; u < n; u++) {
            if (!visited[u]) {
                find_bridges(u, visited, disc, low, parent, edges, time, bridges);
            }
        }
        return bridges;
    }
};
