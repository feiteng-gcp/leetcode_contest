class Solution {
public:
    int n,m,dfn[100000+5],low[100000+5],vis[100000+5],dep,b[100000+5],num=0;
    vector<int>from;
    vector<int>to;
    vector<int>id;
    vector<int>G[100000+5];
    
    void AddEdge(int f,int t,int d)
    {
        from.push_back(f);
        to.push_back(t);
        id.push_back(d);
        G[f].push_back(from.size()-1);
    }


    void dfs(int u,int fa)
    {
        dfn[u]=low[u]=++dep;
        for(int i=0;i<G[u].size();i++)
        {
            int ix = G[u][i];
            if(!vis[to[ix]])
            {
                vis[to[ix]]=1;
                dfs(to[ix],u);
                low[u]=min(low[u],low[to[ix]]);
                if(low[to[ix]]>dfn[u])
                {
                    b[num]=id[ix];
                    num++;

                }
            }
            else if(to[ix]!=fa) low[u]=min(low[u],dfn[to[ix]]);
        }
    }

    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
        from.clear();
        to.clear();
        id.clear();
        for(int i=0;i<connections.size();i++)
        {
            AddEdge(connections[i][0], connections[i][1], i);
            AddEdge(connections[i][1], connections[i][0], i);
        }
        // memset(vis, 0, sizeof(vis));
        // memset(b, 0, sizeof(b));
        num = 0;
        vis[0] = 1;
        dep = 0;
        dfs(0, -1);
        printf("%d\n", num);
        vector<vector<int>> ans;
        for(int i=0;i<num;i++) {
            ans.push_back(connections[b[i]]);
        }
        return ans;
    }
};
