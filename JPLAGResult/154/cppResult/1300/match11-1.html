const int maxn = 100000 + 10;
class Solution {
public:
	
	int n, m;
	int dfs_clock;
	vector<int> G[maxn];
	int pre[maxn];
	int low[maxn];
	bool iscut[maxn];

	vector<vector<int>> ans;
	int dfs(int u, int fa)
	{
		int lowu = pre[u] = ++dfs_clock;
		int child = 0;
		for (int i = 0; i < G[u].size(); i++)
		{
			int v = G[u][i];
			if (!pre[v])
			{
				child++;
				int lowv = dfs(v, u);
				lowu = min(lowu, lowv);
				if (lowv >= pre[u])
				{
					iscut[u] = true;
					if (lowv > pre[u]) ans.push_back({u,v});

				}
			}
			else if (pre[v] < pre[u] && v != fa)
			{
				lowu = min(lowu, pre[v]);
			}
		}
		if (fa < 0 && child == 1)
			iscut[u] = false;
		return low[u] = lowu;
	}
	vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
		m = connections.size();
		dfs_clock = 0;
		memset(pre, 0, sizeof(pre));
		memset(iscut, 0, sizeof(iscut));
		for (int i = 0; i < n; i++) G[i].clear();
		for (int i = 0; i < m; i++)
		{
			int u, v;
			u = connections[i][0];
			v = connections[i][1];
			G[u].push_back(v);
			G[v].push_back(u);
		}
        dfs(0, -1);
		return ans;
	}
};
