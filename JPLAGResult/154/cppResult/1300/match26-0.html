class Solution {
public:

    bool visited[111111];
    vector<int> edges[111111];
    int disc[111111];
    int low[111111];
    int par[111111];
    void dfs(int u, int* time,vector<vector<int>>* bridges)  { 
        visited[u] = true;   
        disc[u] = low[u] = *time;
        (*time)++;
        for (int v : edges[u]) 
        { 
            if (!visited[v]) 
            { 
                par[v] = u; 
                dfs(v, time, bridges); 
                low[u]  = min(low[u], low[v]); 
                if (low[v] > disc[u]) 
                    bridges->push_back({u,v});
            }   
            else if (v != par[u]) 
                low[u]  = min(low[u], disc[v]); 
        } 
    } 
  
    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
        
        for(int i=0;i<n;i++) {
            par[i]=-1;
            visited[i]=false;
        }
        
        for(const vector<int>& uv :  connections) {
            edges[uv[0]].push_back(uv[1]);
            edges[uv[1]].push_back(uv[0]);
        }
        
        vector<vector<int>> bridges;
        int time=0;
        for(int i=0;i<n;i++) {
            if (!visited[i]) {
                dfs(i,&time,&bridges);
            }
        }
        
        return bridges;
        
        
        
    }
};
