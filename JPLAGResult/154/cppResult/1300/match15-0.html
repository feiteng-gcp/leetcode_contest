class Solution {
public:
    vector<int> G[100005];
    typedef pair<int,int> P;
    set<P> S;
    int Fa[100005],low[100005],pre[100005],dfs_clock;
    void dfs(int u, int fa) {
        Fa[u]=fa;
        low[u]=pre[u]=++dfs_clock;
        for (int i=0; i<G[u].size(); ++i) {
            int v=G[u][i];
            if (!pre[v]) {
                dfs(v,u);
                low[u]=min(low[u],low[v]);
            } else if (v!=fa) {
                low[u]=min(low[u],pre[v]);
            }
        }
    }
    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
        for (int i=0; i<connections.size(); ++i) {
            int u=connections[i][0],v=connections[i][1];
            G[u].push_back(v);
            G[v].push_back(u);
        }
        for (int i=0; i<n; ++i) {
            if (!pre[i]) dfs(i,-1);
        }
        vector<vector<int>> ans;
        for (int i=0; i<n; ++i) {
            int v=Fa[i];
            if (v>=0&&low[i]>pre[v]) {
                vector<int> A;
                A.clear();
                A.push_back(i);
                A.push_back(v);
                ans.push_back(A);
            }
        }
        return ans;
    }
};
