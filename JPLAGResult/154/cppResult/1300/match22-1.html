class Solution {
public:
    void dfs(int cur, int p, vector<vector<int> >& ans, int& id,

         vector<vector<int> >& G, vector<int>& L, vector<int>& R)
{
    id++;
    L[cur] = R[cur] = id;
    for(int i = 0; i < G[cur].size(); i++)
    {
        int nxt = G[cur][i];
        if(L[nxt] == 0)
        {
            dfs(nxt, cur, ans, id, G, L, R);
            R[cur] = min(R[cur], R[nxt]);
            if(R[nxt] > L[cur]) ans.push_back({cur, nxt});

        }
        else if(nxt != p) R[cur] = min(R[cur], L[nxt]);
    }
}

vector<vector<int> > criticalConnections(int n, vector<vector<int> >& connections)
{
    int len = (int)connections.size();
    vector<vector<int> > G(n), ans;
    for(int i = 0; i < len; i++)
    {
        G[connections[i][0]].push_back(connections[i][1]);
        G[connections[i][1]].push_back(connections[i][0]);
    }
    int id = 0;
    vector<int> L(n, 0), R(n, 0);
    dfs(0, -1, ans, id, G, L, R);
    return ans;
}
};
