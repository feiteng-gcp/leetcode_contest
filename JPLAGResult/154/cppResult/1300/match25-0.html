class Solution {
public:
    vector<vector<int> >mat;
    vector<int>dfn, low;
    vector<vector<int> >res;
    int times;
    void pushRes(int u, int v) {
        res.push_back(vector<int>({u - 1, v - 1}));
    }
    void addedge(int u, int v) {
        ++u; ++v;
        mat[u].push_back(v);
        mat[v].push_back(u);
    }
    void tarjan(int u, int p) {
        dfn[u] = low[u] = ++times;
        for (int i = 0; i < mat[u].size(); i++)
        {
            int v = mat[u][i];
            if ( !dfn[v] )
            {
                tarjan(v,u);
                low[u] = min ( low[u] , low[v] );
                if ( low[v] > dfn[u])
                    pushRes(u, v);
            }
            else if ( v != p )
                low[u] = min ( low[u] , dfn[v] );
        }
    }
    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
        int i, m = connections.size();
        dfn.resize(n + 1, 0);
        low.resize(n + 1, 0);
        mat.resize(n + 1);
        
        times = 0;
        for (i = 0; i < m; i++) {
            addedge(connections[i][0], connections[i][1]);
        }
        tarjan(1, -1);
        return res;
    }
};
