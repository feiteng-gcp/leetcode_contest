vector<int> vim,dfn;

vector<vector<int> > res(0, vector<int>(0)) ;
unordered_map<int, vector<int> > mp;
int top = 0;
class Solution {
public:

    void tarjan(int x,int fa){
        top += 1;
        vim[x] = top;
        dfn[x] = top;
        for (int id = 0 ; id < mp[x].size(); id ++ ){
            int i = mp[x][id];
            if (i == fa) continue;
            if (dfn[i] == 0){
                tarjan(i,x);
                vim[x] = min(vim[i],vim[x]);
                if (vim[i] > dfn[x]){
                    vector<int> tmp = vector<int>();
                    tmp.push_back(i);tmp.push_back(x);
                    res.push_back(tmp);
                }
            }
            else

                vim[x] = min(vim[x],dfn[i]);
        }
    }
    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
        mp.clear();
        for ( int i = 0 ; i < n ; i ++ ){
            mp[i] = vector<int>();
        }
        res.clear();
        dfn = vector<int>(n,0);
        vim = vector<int>(n,0);
        for ( int i = 0 ; i < connections.size() ; i += 1){
            int a,b;
            a = connections[i][0]; b = connections[i][1];
            mp[a].push_back(b);
            mp[b].push_back(a);
        }
        for ( int i = 0 ; i < n ; i ++ ){
            if (dfn[i] == 0)
                tarjan(i,-1);
        }
        return res;
    }
};
