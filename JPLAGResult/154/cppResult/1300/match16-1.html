class Solution {
public:

    list<int> edge[100005];
    int low[100005], dfn[100005], pre[100005];
    int count;
    
    void tarjan(int u, int fu) {
        pre[u] = fu;
        dfn[u] = low[u] = count++;
        for (auto it = edge[u].begin(); it !=edge[u].end(); it++) {

            int v = *it;
            if (dfn[v] == -1) {
                tarjan(v, u);
                low[u] = min(low[u], low[v]);
            } else if (fu != v) {
                low[u] = min(low[u], dfn[v]);
            }
        }
    }

    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
        memset(dfn, -1, sizeof dfn);
        memset(low, -1, sizeof low);
        count = 0;
        for (int i = 0; i < connections.size(); i++) {
            int st = connections[i][0], ed = connections[i][1];
            edge[st].push_back(ed);
            edge[ed].push_back(st);
        }
        pre[0] = 0;
        tarjan(0, -1);
        vector<vector<int>> ret;
        for (int i = 0; i <= n; i++) {
            int v = pre[i];
            if (v > -1 && low[i] > dfn[v]) {
                ret.push_back(vector<int>{v, i});
            }
        }
        return ret;
    }
};
