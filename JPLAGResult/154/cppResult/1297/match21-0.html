class Solution {
public:
    int maxNumberOfBalloons(string text) {
        map<char, int> mp;
        for (char c : text) {
            mp[c]++;
        }
        int b = mp['b'];
        b = min(b, mp['a']);
        b = min(b, mp['l']/2);
        b = min(b, mp['o']/2);
        b = min(b, mp['n']);
        return b;
    }
};
