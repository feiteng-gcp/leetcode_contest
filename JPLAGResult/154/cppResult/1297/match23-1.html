class Solution {
public:
    int maxNumberOfBalloons(string text) {
        int b = 0, a = 0, l = 0, o = 0, n = 0;
        for (auto c: text) {
            if (c == 'b') ++b;
            if (c == 'a') ++a;
            if (c == 'l') ++l;
            if (c == 'o') ++o;
            if (c == 'n') ++n;
        }
        int res = INT_MAX;
        res = min(res, b);
        res = min(res, a);
        res = min(res, l / 2);
        res = min(res, o / 2);
        res = min(res, n);
        return res;
    }
};
