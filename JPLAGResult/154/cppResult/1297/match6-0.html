class Solution {
public:
    int maxNumberOfBalloons(string text) {
        unordered_map<char,int> freq;
        for (auto c : text) {
            freq[c]++;
        }
        int res = INT_MAX;
        res = min(res,freq['b']);
        res = min(res,freq['a']);
        res = min(res,freq['n']);
        res = min(res,freq['l'] / 2);
        res = min(res,freq['o'] / 2);
        return res;
    }
};
