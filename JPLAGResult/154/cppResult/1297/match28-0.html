class Solution {
public:
    int maxNumberOfBalloons(string text) {
        
        vector<int> freq(26, 0);
        for(char c: text) freq[c - 'a'] ++;
        
        string s = "ban";
        int res = INT_MAX;
        for(char c: s)
            res = min(res, freq[c - 'a']);
        res = min(res, freq['l' - 'a'] / 2);
        res = min(res, freq['o' - 'a'] / 2);
        return res;
    }
};
