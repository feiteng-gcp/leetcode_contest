class Solution {
public:
    int maxNumberOfBalloons(string s) {
        map<char, int> cnt;
        for (auto& c : s) {
            cnt[c]++;
        }
        int ret = 1 << 30;
        ret = min(ret, cnt['b']);
        ret = min(ret, cnt['a']);
        ret = min(ret, cnt['l'] / 2);
        ret = min(ret, cnt['o'] / 2);
        ret = min(ret, cnt['n']);
        return ret;
    }
};
