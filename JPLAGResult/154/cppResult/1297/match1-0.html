class Solution {
public:
    int maxNumberOfBalloons(string text) {
        unordered_map<char, int> hash;
        for(auto c : text){
            hash[c]++;
        }
        int res = INT_MAX;
        
        res = min(res, hash['b']);
        res = min(res, hash['a']);
        res = min(res, hash['l'] / 2);
        res = min(res, hash['o'] / 2);
        res = min(res, hash['n']);
        
        return res;
    }
};
