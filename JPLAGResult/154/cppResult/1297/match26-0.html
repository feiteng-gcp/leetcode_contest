class Solution {
public:
    int maxNumberOfBalloons(string text) {
        unordered_map<char, int> cnt;
        for ( auto c:text ){
            if ( c=='b' || c=='a' || c=='l' || c=='o' || c=='n' )
                ++cnt[c];
        }
        int x = min( min( cnt['b'], cnt['a'] ), cnt['n'] );
        int y =  min( cnt['o']/2, cnt['l']/2 ) ;
        return min( x,y );
    }
};
