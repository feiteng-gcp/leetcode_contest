class Solution {
public:
  int maxNumberOfBalloons(string text) {
    map<char, int> M;
    for (auto c: text) M[c]++;
    string s = "balloon";
    int tot = 0;
    while (true) {
      for (auto c: s) {
        if (--M[c] < 0) return tot;
      }
      tot++;
    }
    return -1;
  }
};
