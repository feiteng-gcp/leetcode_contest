class Solution {
public:
    int maxNumberOfBalloons(string text) {
        unordered_map<char,int> ma;
        for(char c:text)
            ma[c]++;
        int ans=INT_MAX;
        ans=min(ans,ma['b']);
        ans=min(ans,ma['a']);
        ans=min(ans,ma['l']/2);
        ans=min(ans,ma['o']/2);
        ans=min(ans,ma['n']);
        return ans;
    }
};
