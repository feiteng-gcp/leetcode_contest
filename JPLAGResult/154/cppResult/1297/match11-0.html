class Solution {
public:
    int maxNumberOfBalloons(string text) {
      map<char, int> cnt;
      for (char c: text) ++cnt[c];
      int ret = text.length();
      ret = min(ret, cnt['a']);
      ret = min(ret, cnt['b']);
      ret = min(ret, cnt['l'] / 2);
      ret = min(ret, cnt['n']);
      ret = min(ret, cnt['o'] / 2);
      return ret;
    }
};
