class Solution {
public:
    string reverseParentheses(string s) {
        stack<char> st;
        for (auto c: s) {
            if (c != ')') {
                st.push(c);
            } else {
                string v;
                while (st.top() != '(') {
                    v += st.top();
                    st.pop();
                }
                st.pop();
                for (auto c: v) {
                    st.push(c);
                }
            }
        }
        string res;
        while (!st.empty()) {
            res += st.top();
            st.pop();
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
