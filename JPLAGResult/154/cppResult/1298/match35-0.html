class Solution {
public:
    string reverseParentheses(string s) {
        string tmp = "";
        stack<string> st;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == '(') {
                st.push(tmp);
                tmp = "";
            }
            else if (s[i] == ')') {
                reverse(tmp.begin(), tmp.end());
                tmp = st.top() + tmp;
                st.pop();
            }
            else {
                tmp += s[i];
            }
        }
        return tmp;
    }
};
