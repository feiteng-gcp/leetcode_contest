class Solution {
public:
    string reverseParentheses(string s) {
        stack<string> st;
        st.push("");
        for (char c : s) {
            if (c == '(') {
                st.push("");
            } else if (c == ')') {
                string str = st.top();
                reverse(str.begin(), str.end());
                st.pop();
                st.top() += str;
            } else {
                st.top().push_back(c);
            }
        }
        return st.top();
    }
};
