class Solution {
public:
    string reverseParentheses(string s) {
        map<int, int> match;
        map<int, int> level;

        stack<int> paren;
        
        for (int i = 0; i < s.length(); ++i) {
            if (s[i] == '(') {
                paren.push(i);
            }
            else if (s[i] == ')') {
                match[paren.top()] = i;
                level[paren.top()] = paren.size() + 1;
                paren.pop();
            }
        }
        
        for (int lvl = s.length(); lvl >= 1; --lvl) {
            for (const pair<int, int>& p : level) {
                if (p.second == lvl) {
                    int l = p.first;
                    int r = match[l];
                    reverse(s.begin() + l, s.begin() + r + 1);
                }
            }
        }
        
        string res;
        for (const char& ch : s) {
            if (ch != '(' && ch != ')') res += ch;
        }
        return res;
    }
};
