class Solution {
public:
    string reverseParentheses(string s) {
        // I'm sure this can be done in O(N) but whatever...
        vector<int> st;

        int n = s.size();
        for (int i = 0; i < n; ++i) {
            if (s[i] == '(') st.push_back(i);
            else if (s[i] == ')') {
                int j = st.back();
                st.pop_back();
                reverse(s.begin()+j, s.begin()+i+1);
            }
        }
        string ans;
        for (char c : s) if (c != '(' && c != ')') ans += c;
        return ans;
    }
};
