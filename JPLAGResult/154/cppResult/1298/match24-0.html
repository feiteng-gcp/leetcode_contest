class Solution {
public:
    string rev(string s) {
        string res = "";
        for (int i = s.size() - 1; i >= 0; i --) res += s[i];
        return res;
    }
    
    string reverseParentheses(string s) {
        for (int i = 0; i < s.size(); i ++) {
            if (s[i] == '(') {
                int h = 0, j;
                for (j = i; j < s.size(); j ++) {
                    if (s[j] == '(') ++h;
                    if (s[j] == ')') --h;
                    if (h == 0) break;
                }
                return (i == 0 ? "" : reverseParentheses(s.substr(0, i)))
                    + rev(reverseParentheses(s.substr(i + 1, j - i - 1)))
                    + (j == s.size() - 1 ? "" : reverseParentheses(s.substr(j + 1, s.size())));
            }
        }
        return s;
    }
};
