class Solution {
public:
    string reverseParentheses(string s) {
        int i = 0;
        return helper(i, s);
    }
    
    string helper(int& i, string& s) {
        string t;
        while(i<s.size()) {
            if(s[i]=='(') {
                i++;
                t+=helper(i, s);
            } else if(s[i]==')'){
                i++;
                reverse(t.begin(), t.end());
                return t;
            }
            else {
                t+=string(1, s[i++]);
            }
        }
        return t;
    }
};
