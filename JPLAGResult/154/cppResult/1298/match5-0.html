class Solution {
public:
    string reverseParentheses(string s) {
        vector<string> st;
        vector<char> c;
        string cur="";
        for ( auto c:s ){
            if ( c!='(' && c!=')' )
                cur += c;
            else if ( c=='(' ){
                st.push_back( cur );
                cur = "";
            }else if ( c==')' ){
                reverse( cur.begin() , cur.end() );
                cur = st.back() + cur; st.pop_back();
            }
        }
        return cur;
    }
};
