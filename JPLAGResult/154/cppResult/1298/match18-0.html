class Solution {
public:
    string reverseParentheses(string s) {
        stack<int> st;
        string res="";
        for(int i=0;i<s.size();i++)
        {
            if(s[i]=='(')
                st.push(res.size());
            else if(s[i]==')')
            {
                int last=st.top();
                st.pop();
                for(int j=last;j<res.size()-1-(j-last);j++)
                    swap(res[j],res[res.size()-1-(j-last)]);
            }
            else 
                res+=s[i];
        }
        return res;
        
    }
};
