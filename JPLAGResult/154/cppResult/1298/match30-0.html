#define ALL(x) (x).begin(), (x).end()
const int MAXN = 2005;
int P[MAXN];
vector<int> stk;
string S;
int N;

class Solution {
public:
    string reverseParentheses(string s) {
        S = move(s);
        N = S.size();
        stk.clear();
        for (int i = 0; i < N; ++i) {
            if (S[i] == '(') {
                stk.push_back(i);
            } else if (S[i] == ')') {
                P[stk.back()] = i;
                stk.pop_back();
            }
        }
        return dfs(0, N-1);
    }
    
private:
    string dfs(int l, int r) {
//        printf(">> %d, %d\n", l, r);
        if (r >= N) {
            return "";
        }
        if (l > r) {
            return "";
        }
        if (S[l] == '(') {
            int end = P[l];
            string res = dfs(l+1, end-1);
            reverse(ALL(res));
            return res + dfs(end+1, r);
        }
        return S[l] + dfs(l+1, r);
    }
};
