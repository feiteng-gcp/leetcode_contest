class Solution {
public:
    string reverseParentheses(string s) {
        stack<string> st;
        if (s.size() < 1) return s;
        for (int i = 0; i < s.size(); i++) {
            string ca = string(1, s[i]);
            if (ca == ")") {
                string tmp = "";
                while (!st.empty() && st.top() != "(") {
                    tmp = tmp + st.top();
                    st.pop();
                }
                st.pop();
                for (auto tc : tmp) {
                    st.push(string(1, tc));
                }
            } else {
                st.push(ca);
            }
        }
        string res = "";
        while (!st.empty()) {
            res = st.top() + res;
            st.pop();
        }
        return res;
    }
};
