class Solution {
public:
    string reverseParentheses(string s) {
        string res;
        stack<int> p;
        for(char c: s) {
            if(c == '(') {
                p.push(len(res));
            } else if(c == ')') {
                auto begin = res.begin() + p.top();
                p.pop();
                reverse(begin, res.end());
            } else {
                res.push_back(c);
            }
        }
        return res;
    }
};
