class Solution {
public:
    string reverseParentheses(string s) {
		int N = s.size();
		int i, j, k;
		string Result;
		
		for(i = 0; i < N; i++)
		{
			if(s[i] == '(')
			{
				int Count = 1;
				
				for(j = (i + 1); j < N; j++)
				{
					if(s[j] == '(')
					{
						Count++;
					}
					else if(s[j] == ')')
					{
						Count--;
					}
					
					if(Count == 0)
					{
						break;
					}
				}
				

				for(k = j - 1; k > i; k--)
				{
					if(s[k] == '(')
					{
						Result += ')';
					}
					else if(s[k] == ')')
					{
						Result += '(';
					}
					else
					{
						Result += s[k];
					}
				}
				
				for(j++; j < N; j++)
				{
					Result += s[j];
				}
				
				break;
			}
			else
			{
				Result += s[i];
			}
		}
		
		if(i == N)
		{
			return Result;
		}
		else
		{
			return reverseParentheses(Result);
		}
    }
};
