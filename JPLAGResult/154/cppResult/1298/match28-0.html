#define all(a) a.begin(),a.end()
class Solution {
public:
    string reverseParentheses(string s) {
            int n=s.length();
    stack<string> st;
    int i=0;
    while(i<n)
    {
        if(s[i]!=')')
        {
            string a="";
            a+=s[i];
            st.push(a);
        }
        else
            {
                string t="";
                while(st.top() !="(")
                {
                    t=t+st.top();
                    st.pop();
                }
                st.pop();
                reverse(all(t));
                //cout<<t<<endl;
                st.push(t);
            }
        i++;
    }
    string ans="";
    while(!st.empty())
    {
        string x=st.top();
        //cout<<x<<endl;
        reverse(all(x));
        ans=x+ans;
        st.pop();
    }
   // reverse(all(ans));
    return ans;
    }
};
