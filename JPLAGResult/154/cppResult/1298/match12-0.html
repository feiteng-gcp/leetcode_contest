class Solution {
public:
    string reverseParentheses(string s) {
        int i = 0;
        return helper(s, i);
    }
    
    string helper(string &s, int &i) {
        string ret;
        while (i < s.size()) {
            if (s[i] == '(') {
                ret += helper(s, ++i);
            }
            else if (s[i] == ')') {
                i++;
                reverse(ret.begin(), ret.end());
                break;
            }
            else ret += s[i++];
        }
        return ret;
    }
};
