class Solution {
public:
    
    string reverseParentheses(string s) {
        stack<int>S;
        stack<string>tmp;
        string current = "";
        for (int i = 0; i < s.length(); i++){
            if (s[i] == '(')
                S.push(i + 1);
            else if (s[i] == ')'){
                int start = S.top();
                S.pop();
                reverse(s.begin() + start, s.begin() + i);
            }
        }
        string ans = "";
        for (int i = 0;i < s.length();i++)
            if (s[i] != '(' && s[i] != ')')
                ans.append(sizeof(char), s[i]);
        return ans;
    } 
};
