class Solution {
public:
    string reverseParentheses(string s) {
        vector<string> p;
        p.push_back("");
        for(auto e:s)

        {
            if(e == '(')
            {
                p.push_back("");
            }
            else if(e == ')')
            {
                string x = p.back();
                p.pop_back();
                reverse(x.begin(),x.end());
                p.back() += x;
            }
            else
            {
                p.back() += e;
            }
        }
        return p[0];
    }
};
