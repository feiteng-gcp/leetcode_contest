class Solution {
public:
    string reverseParentheses(string s) {
        stack<int> stk;
        for (int i = 0; i < s.length(); ++i) {
            if (s[i] == '(') {
                stk.push(i);
            } else if (s[i] == ')') {
                reverse(s.begin() + stk.top(), s.begin() + i);
                stk.pop();
                // cout << i << " " << s << endl;
                s.erase(i - 1, 2);
                i -= 2;
                // cout << i << " " << s << endl;
                // cout << endl;
            }
        }
        return s;
    }
};
