class Solution {
public:
    string reverseParentheses(string s) {
        //cout << s << endl;
        string res = "";
        for (int i = 0; i < s.size(); ++i)
        {
            if (isalpha(s[i]) != 0)
            {
                res += s[i];
            }
            else if (s[i] == '(')
            {
                int start = i + 1;
                int p = 0;
                for (; i < s.size(); ++i)
                {
                    if (s[i] == '(') ++p;
                    if (s[i] == ')') --p;
                    if (p == 0) break;
                }
                int len = i - start;
                if (len == 0) continue;
                string subStr = s.substr(start, len);                
                reverse(subStr.begin(), subStr.end());
                for (int a = 0; a < subStr.size(); ++a)
                {
                    if (subStr[a] == '(') subStr[a] = ')';
                    else if (subStr[a] == ')') subStr[a] = '(';
                }
                res += reverseParentheses(subStr);
            }
        }
        return res;
    }
};
