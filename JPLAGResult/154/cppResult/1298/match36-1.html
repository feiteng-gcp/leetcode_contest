class Solution {
public:
    string reverseParentheses(string s) {
        int i;
        int begin = 0;
        int end = s.size() - 1;

        for(i = 0; i < s.size(); i++){
        if(s[i] == '(')
            begin = i;
        if(s[i] == ')'){
            end = i;
            string tem = s.substr(begin + 1, end - begin - 1);
            reverse(tem.begin(),tem.end());
            return reverseParentheses(s.substr(0, begin) + tem + s.substr(end + 1));
            }
        }
        return s;
    }
};
