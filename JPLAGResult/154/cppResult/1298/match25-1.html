class Solution {
public:
    vector<int> pos;
    string dfs(const string& s, int l, int r) {
        string ans;
        int i = l;
        while (i <= r) {
            if (s[i] == '(') {
                string tmp = dfs(s, i+1, pos[i]-1);
                reverse(tmp.begin(), tmp.end());
                ans += tmp;
                i = pos[i]+1;
            }else{
                ans += s[i];
                i++;
            }
        }
        return ans;
    }
    string reverseParentheses(string s) {
        int sz = s.size();
         pos.resize(s.size());

        stack<int> st;
        for (int i=0; i<sz; ++i) {
            if (s[i] == '(') {
                st.push(i);
            }else if (s[i] == ')') {
                pos[st.top()] = i;
                st.pop();
            }
        }
       
        return dfs(s, 0, sz-1);
    }
};
