class Solution {
public:
    string reverseParentheses(string s) {
        vector<string> st(1);
        for (char c : s) {
            if (c == '(') {
                st.emplace_back();
            } else if (c == ')') {
                string t = st.back();
                reverse(t.begin(), t.end());
                st.pop_back();
                for (char tc : t) {
                    st.back().push_back(tc);
                }
            } else {
                st.back().push_back(c);
            }
        }
        return st.back();
    }
};
