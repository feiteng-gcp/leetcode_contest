class Solution {
public:
    string reverseParentheses(string s) {
        stack<char> st;
        for(auto c:s){

            if(c==')'){
                string temp = "";
                while(st.top()!='('){
                    temp+=st.top();
                    st.pop();
                }
                st.pop();
                for(auto cc:temp){
                    st.push(cc);
                }
            }else{
                st.push(c);
            }
        }
        string ret = "";
        while(st.size()>0){
            ret = st.top()+ret;
            st.pop();
        }
        return ret;
    }
};
