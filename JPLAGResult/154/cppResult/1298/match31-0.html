class Solution {
public:
    string reverseParentheses(string s) {
        if (s.empty())
            return "";
        
        map<int, int> pairs;
        vector<int> stack;
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == '(') {
                stack.push_back(i);
            } else if (s[i] == ')') {
                pairs[i] = stack.back();
                pairs[stack.back()] = i;
                stack.pop_back();
            }
        }
        
//        for (auto p : pairs)
//            cout << p.first << " => " << p.second << endl;
        
        string output;
        int dir = 1;
        for (int pt = 0; pt < s.size(); pt += dir) {
//            cout << pt << ":" << s[pt] << endl;
            if (s[pt] == '(' || s[pt] == ')') {
                dir = -dir;
                pt = pairs[pt];
            } else {
                output.push_back(s[pt]);
            }
        }
        
        return output;
    }
};
