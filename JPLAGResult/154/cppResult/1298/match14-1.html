class Solution {
public:
    string reverseParentheses(string s)
{
    vector<string> stk;
    stk.push_back("");
    int len = (int)s.size();
    for(int i = 0; i < len; i++)
    {
        if(s[i] == '(') stk.push_back("");
        else if(s[i] == ')')
        {
            reverse(stk.back().begin(), stk.back().end());
            string t = stk.back(); stk.pop_back();
            stk.back() += t;
        }
        else stk.back().push_back(s[i]);
    }
    return stk[0];
}
};
