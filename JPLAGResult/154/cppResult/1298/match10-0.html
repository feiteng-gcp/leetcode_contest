class Solution {
public:
    string reverseParentheses(string s) {
        int N = int(s.length());

        vector<int> stk;
        for (int i = 0; i < N; i++) {
            if (s[i] == '(')
                stk.push_back(i);
            else if (s[i] == ')') {
                int left = stk.back();
                reverse(s.begin() + left + 1, s.begin() + i);
                stk.pop_back();
            }
        }

        string res;
        for (auto c : s) {
            if (c != '(' && c != ')')
                res.push_back(c);
        }
        return res;
    }
};
