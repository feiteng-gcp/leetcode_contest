class Solution {
 public:
  string reverseParentheses(string s) {
    return solve(s, 0, (int)s.size() - 1);
  }

 private:

  string solve(string& s, int l, int r) {
    string ans;
    int paren = 0;
    int ll = -1;
    for (int i = l; i <= r; ++i) {
      if (s[i] == '(') {
        if (paren == 0) {
          ll = i;
        }
        ++paren;
      } else if (s[i] == ')') {
        --paren;
        if (paren == 0) {
          string t = solve(s, ll + 1, i - 1);
          reverse(t.begin(), t.end());
          ans += t;
        }
      } else if (paren == 0) {
        ans += s[i];
      }
    }
    return ans;
  }
};
