class Solution {
public:
    string reverseParentheses(string s) {
        vector<string> a = {""};

        for (auto c : s) {
            if (c == '(') {
                a.push_back("");
            } else if (c == ')') {
                auto t = a.back();
                reverse(begin(t), end(t));
                a.pop_back();
                a.back() += t;
            } else {
                a.back().push_back(c);
            }
        }
        return a[0];
    }
};
