class Solution {
public:
    string reverseParentheses(string s) {
        int st = 0;
        return helper(s, st);
    }
    
    string helper(string& s, int& st)
    {
        string cur;
        
        while(st < s.size())
        {
            if(s[st] == '(')
            {
                st++;
                string tmp = helper(s, st);
                reverse(tmp.begin(), tmp.end());
                
                cur += tmp;
            }
            else if(s[st] == ')')
            {
                st++;
                return cur;
            }
            else
            {
                cur.push_back(s[st]);
                st++;
            }
        }
        
        return cur;
    }
};
