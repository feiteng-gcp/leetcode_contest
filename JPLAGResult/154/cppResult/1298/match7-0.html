class Solution {
public:
    string reverseParentheses(string s) {
        int cnt=0;
        stack<int> stk;
        for(int i=0; s[i]; i++){
            if(s[i]=='('){
                stk.push(i);
            }
            if(s[i]==')'){
                int st = stk.top(); 
                int ed = i;
                stk.pop();
                for(int j=st; j<=(st+ed)/2; j++){
                    swap(s[j],s[ed-j+st]);
                }
            }
        }
        string r;
        for(int i=0; s[i]; i++){
            if(s[i]!='(' && s[i]!=')') r+= s[i];
        }
        return r;
    }
};
