class Solution {
public:
  string _reverse(string s) {
    reverse(s.begin(), s.end());
    return s;
  }
  string dfs(string &s, int &l) {
    string t;
    while (l < s.size()) {
      if (s[l] == ')') {
        l++;
        break;
      } else if (s[l] == '(') {
        l++;
        t += _reverse(dfs(s, l));
      } else {
        t.push_back(s[l]);
        l++;
      }
    }
    return t;
  }
  string reverseParentheses(string s) {
    int l = 0;
    return dfs(s, l);
  }
};
