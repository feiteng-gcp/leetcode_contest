class Solution {
    long long maxSum(vector<int>& arr) {
        long long ans = 0;
        long long temp = 0;
        for (int i = 0; i < arr.size(); i++) {
            temp += arr[i];
            if (temp < 0) {
                temp = 0;
            }
            ans = max(ans, temp);
        }
        // cout << ans << endl;
        return ans;
    }
    long long maxSumLeft(vector<int>& arr) {
        long long ans = 0;
        long long temp = 0;
        for (int i = 0; i < arr.size(); i++) {
            temp += arr[i];
            ans = max(ans, temp);
        }
        // cout << ans << endl;
        return ans;
    }
    long long maxSumRight(vector<int>& arr) {
        long long ans = 0;
        long long temp = 0;
        for (int i = arr.size()-1; i >= 0; i--) {
            temp += arr[i];
            ans = max(ans, temp);
        }
        // cout << ans << endl;
        return ans;
    }
    long long sum(vector<int>& arr) {
        long long ans = 0;
        for (int i = 0; i < arr.size(); i++) {
            ans += arr[i];
        }
        // cout << ans << endl;
        return ans;
    }
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        long long sumArr = sum(arr);
        long long sumMid = maxSum(arr);
        long long sumLeft = maxSumLeft(arr);
        long long sumRight = maxSumRight(arr);
        long long ans = 0;
        if (sumArr>0) {
            ans = max(ans, k*sumArr);
            ans = max(ans, (k-1)*sumArr + max(sumLeft, sumRight));
            ans = max(ans, (k-2)*sumArr + sumLeft + sumRight);
        }
        // cout << ans << endl;
        ans = max(ans, sumMid);
        ans = max(ans, sumLeft+sumRight);
        long long mod = 1e9+7;
        return ans%mod;
    }
};
