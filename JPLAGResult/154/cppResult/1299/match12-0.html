class Solution {
public:
    const int mo = 1e9 + 7;
    int kConcatenationMaxSum(vector<int>& a, int k) {
        int n = a.size(), h = 0, t = 0, ans = 0;
        int minPre = 0, sum = 0;
        
        for (int i = 0; i < n; ++i) {
            sum += a[i];
            h = max(h, sum);
            ans = max(ans, sum - minPre);
            minPre = min(minPre, sum);
        }
        
        int tot = max(0, sum);
        
        if (k == 1)
            return ans;
        
        sum = 0;
        for (int i = n-1; i >= 0; --i) {
            sum += a[i];
            t = max(t, sum);
        }
        
        long long tt = (long long)h + t + (long long)tot * max(0, k-2);
        ans = max(ans, (int)(tt % mo));
        return ans;
    }
};
