class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        vector<int> nums = arr;
        long ret1, ret2, ret3;
        long ret = 0;
        if (k >= 1) {
            ret1 = maxSubArray(nums);
            if (ret1 > 0)
                ret = ret1;
        }
        if (k >= 2) {
            nums.insert(nums.end(), arr.begin(), arr.end());
            ret2 = maxSubArray(nums);
            if (ret2 > ret) {
                ret = ret2;
                long sum = accumulate(arr.begin(), arr.end(), 0);
                if (sum > 0)
                    ret = sum*(k-2)+ret2;
            }
        }
        return ret%int(pow(10,9)+7);
    }
    
    int maxSubArray(vector<int>& nums) {
        int ret, prev;
        ret = prev = nums[0];
        for (int i = 1; i < nums.size(); i++) {
            prev = max(nums[i], nums[i]+prev);
            ret = max(ret, prev);
        }
        return ret;
    }
};
