#define ll long long

class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int n = arr.size();
        if (n == 0) return 0;
        
        vector<ll> f(n, 0), g(n, 0);
        
        ll sum = 0, min_sum = 0;
        for (int i = 0; i < n; ++i) {
            sum += arr[i];
            min_sum = min(min_sum, sum);
            f[i] = sum - min_sum;
        }
        

        sum = 0;
        min_sum = 0;
        for (int i = n - 1; i >= 0; --i) {
            sum += arr[i];
            min_sum = min(min_sum, sum);
            g[i] = sum - min_sum;
        }
        
        //for (int i = 0; i < n; ++i) cout << f[i] << " "; cout << endl;
        //for (int i = 0; i < n; ++i) cout << g[i] << " "; cout << endl;
        
        ll res = *max_element(f.begin(), f.end());
        for (int t = 2; t <= k; ++t) {
            res = max(res, f[n - 1] + g[0] + sum * (ll)(t - 2));
        }
        
        return res % 1000000007;
    }
};
