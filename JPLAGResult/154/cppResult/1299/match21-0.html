class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int n = arr.size();
        int kr = norepeat(arr, n);
		int sms, pms, fs;
		if(k == 1) return kr;
		
        sms = suffix(arr, n);
        pms = prefix(arr, n);
        fs  = full(arr, n);
        int sum = 0;
        for(int i = 0; i < k-2; i++) sum = (sum+fs)%mod;
        return max({kr, ((sms + sum)%mod + pms)%mod, 0});
    }
    
private:
    int mod = 1e9+7;
    
    int norepeat(vector<int>& a, int n){
        int cs = 0, bs = INT_MIN;
        for(int i=0; i<n; i++) {
            cs += a[i];
            bs = max(cs, bs);
            cs = max(cs, 0);
        }
        return bs;
    }

    int suffix(vector<int>& a, int n){ 
        int cs = 0, bs = INT_MIN;
        for(int i=n-1; i>=0; i--) {
            cs += a[i];
            bs = max(cs, bs);
        }
        return bs;
    }

    int prefix(vector<int>& a, int n){
        int cs = 0, bs = INT_MIN;
        for(int i=0; i<n; i++) {
            cs += a[i];
            bs = max(cs, bs);
        }
        return bs;
    }

    int full(vector<int>& a, int n) {
        int sum = 0;
        for(int i=0; i<n; i++) {
            sum += a[i];
        }
        return max(sum, 0);
    }
};
