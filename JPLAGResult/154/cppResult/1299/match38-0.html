class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int m = 1000000007;
        int sum_all = 0;
        int lsum = 0;
        int lsum_min = 0;
        int sum_max = 0;
        for (int j = 0; j < min(k, 2); j++) {
            for (int i = 0; i < arr.size(); i++) {
                lsum += arr[i];
                lsum_min = min(lsum_min, lsum);
                sum_max = max(sum_max, lsum - lsum_min);
            }
            if (j == 0) sum_all = lsum;
        }
        if (k < 2) {
            return sum_max;
        } else {
            return (sum_max + (long long)max(sum_all, 0) * (k - 2) % m) % m;
        }
    }
};
