class Solution {
    typedef long long ll;
    const ll MOD = 1e9+7;
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int N = arr.size();
        vector<ll> presum(N+1, 0);
        
        ll res_0 = 0, res_l = 0, res_r = 0, min_v = 0;
        for (int i=0; i<N; i++) {
            presum[i+1] = presum[i] + arr[i];
            res_0 = max(res_0, presum[i+1] - min_v);
            res_l = max(res_l, presum[i+1]);
            min_v = min(min_v, presum[i+1]);
        }
        res_r = max(res_r, presum[N] - min_v);
        
        cout << res_0 << ' ' << res_l << ' ' << res_r;
        
        if (k == 1) return res_0 % MOD;
        ll res = max(res_0, res_l + res_r) % MOD;
        if (k == 2) return res % MOD;
        if (presum[N] > 0) return (res + presum[N] * (k-2)) % MOD;
        else return res % MOD;
    }
};
