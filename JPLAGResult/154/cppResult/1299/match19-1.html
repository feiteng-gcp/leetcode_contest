class Solution {
public:

    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int n=arr.size();
        int maxpre=0,maxsub=0;
        int now=0;
        for(int i=0;i<n;i++)
        {
            now=now+arr[i];
            maxpre=max(maxpre,now);
        }

        now=0;
        for(int i=n-1;i>=0;i--)
        {
            now=now+arr[i];
            maxsub=max(maxsub,now);
        }
        long long ans=0;
        now=-99999;
        for(int i=0;i<n*min(k,2);i++)
        {
            now=max(arr[i%n],now+arr[i%n]);
            ans=max(ans,(long long)now);
        }
        now=0;
        for(int i=0;i<n;i++)
        {
            now=now+arr[i];
        }
        if(k>1)ans=max(ans,maxpre+maxsub+(long long)now*(k-2));
        return ans%1000000007;
    }
};
