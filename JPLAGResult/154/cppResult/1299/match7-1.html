const int mod = 1e9+7;
class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int n = arr.size(), m = arr.size();
        if (k > 1) n += n;
        long long ans = 0, sum = 0, cum = 0;
        for (int i = 0; i < n; i++) {
            sum += arr[i % m]; 
            if (i < m) {
                cum += arr[i];
            }
            ans = max(ans, sum);

            sum = max(0LL, sum);
        }
        long long lft = 0, rght = 0;
        sum = 0;
        for (int i = 0; i < m; i++) {
            sum += arr[i];
            lft = max(lft, sum);
        }
        sum = 0;
        for (int i = m - 1; i >= 0; i--) {
            sum += arr[i];
            rght = max(rght, sum);
        }
        ans = max(ans, cum * k);
        // cout << cum << " " << lft << " " << rght << endl;
        if (k > 2) {
            k -= 2;
            ans = max(ans, cum * k + lft + rght);
        }
        
        return ans % mod;
    }
};
