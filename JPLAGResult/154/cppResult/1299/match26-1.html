class Solution {
public:
    const int cnst = pow(10, 9) + 7;
    
    int maxSub(vector<int>& arr){
        int ret = 0;
        int n = arr.size();

        vector<int> an(n);
        an[0] = arr[0];
        ret = max(ret, an[0]);
        for(int i = 1;i < n; ++i){
            an[i] = max(arr[i], an[i - 1] + arr[i]);
            ret = max(ret, an[i]);
        }
        return ret;
    }
    
    int maxSubBegin(vector<int>& arr){
        int ret = 0;
        int tmpSum = 0;
        for(auto i : arr){
            tmpSum += i;
            ret = max(tmpSum, ret);
        }
        return ret;
    }
    
    int maxSubEnd(vector<int>& arr){
        int ret = 0;

        int tmpSum = 0;
        for(int i = arr.size() - 1; i>= 0; i--){
            tmpSum += arr[i];
            ret = max(tmpSum, ret);
        }
        return ret;
    }    
    
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int sum = 0;
        for(auto i : arr) sum += i;
        if(sum < 0){
            if(k == 1) return maxSub(arr);
            else return max(maxSub(arr), maxSubBegin(arr) + maxSubEnd(arr));
        }else{
            int can1 = maxSub(arr);
            if(k == 1) return can1;
            else{
                //if((k - 2) * sum > (long)can1)
                long long int can2 = (long)(k - 2) * sum + maxSubBegin(arr) + maxSubEnd(arr);
                //cout << "sum = " << sum << "maxbegin = " <<maxSubBegin(arr)  << " maxend = " << maxSubEnd(arr) << endl;
                //cout << can2 << endl;
                if(can2 > (long)can1) return (int)(can2 % cnst);
                else return can1;
            }
        }
    }
};
