class Solution {
public:
    typedef long long LL;
    LL MO=1e9+7;
    LL INF=1e16;
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        LL premax=0,sufmax=0,midmax=0;
        int n=arr.size();
        LL nmax=0;
        for(int i=0;i<n;i++)
        {
            nmax=max(nmax+arr[i],(LL)arr[i]);
            midmax=max(midmax,nmax);
        }
        if(k==1) return midmax%MO;
        else
        {

            LL nsum=0;
            for(int i=0;i<n;i++)
            {
                nsum+=arr[i];
                premax=max(premax,nsum);
            }
            nsum=0;
            for(int i=n-1;i>=0;i--)
            {
                nsum+=arr[i];
                sufmax=max(sufmax,nsum);
            }
            LL ans=max(midmax,max(premax+sufmax,premax+sufmax+(k-2)*nsum));
            return ans%MO;
        }
    }
};
