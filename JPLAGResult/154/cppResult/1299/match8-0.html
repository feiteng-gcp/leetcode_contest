class Solution {
public:
    vector<int> A;
    long long mod=1e9+7;
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int tem=min(3,k);
        for (int j=0; j<tem; ++j) {
            for (int i=0; i<arr.size(); ++i) {
                A.push_back(arr[i]);
            }
        }
        long long MX=0,MR=0,su=0;
        for (int i=0; i<arr.size(); ++i) {
            su+=arr[i];
            su%=mod;
            MX=max(su,MX);
        }
        long long Sum=su;
        su=0;
        for (int i=arr.size()-1; i>=0; --i) {
            su+=arr[i];
            su%=mod;
            MR=max(su,MR);
        }
        long long Left=0,Right=0,sum=0,ans=0;
        for (int i=0; i<A.size(); ++i) {
            sum+=A[i];
            if (sum<=0) {
                sum=0;
                Left=i+1;
            }
            sum%=mod;
            ans=max(sum,ans);
        }
        if (k==1) return ans%mod;
        ans=max(ans,(k-2)*su+MX+MR);
        return ans%mod;
    }
};
