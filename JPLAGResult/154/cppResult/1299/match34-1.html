class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        if(k==0)return 0;
        vector<long long> sum;
        sum.push_back(0);
        int n=arr.size();
        for(int i=0;i<2*n;i++)
            sum.push_back(0);

        long long minn=0;
        long long maxn=0;
        for(int i=0;i<n;i++)
        {
            sum[i+1]=sum[i]+arr[i];
            minn=min(minn,sum[i+1]);
            maxn=max(maxn,sum[i+1]-minn);
        }
        if(k==1)
            return maxn;

        long long total=sum[n];
        long long maxn2=0;
        for(int i=0;i<n;i++)
        {
            sum[i+n+1]=sum[i+n]+arr[i];
            minn=min(minn,sum[i+n+1]);
            maxn2=max(maxn2,sum[i+n+1]-minn);
        }
        long long res=max(maxn2,max(maxn2+total*(k-2),maxn));
        return res%1000000007;
    }
};
