class Solution {
public:
    const int mod = 1e9+7;
    long long get(vector<int> a,int len) 
    { 
        int size = a.size(); 

        long long ans = 0; 
        long long cur = 0; 

        for (int i = 0; i < len; i++) 
        { 
            cur = (cur + a[i % size]) % mod; 
            if (ans < cur) ans = cur;
            if (cur < 0) 
                cur = 0; 
        } 
        return ans % mod; 
    } 
  

    long long helper(vector<int> arr, int k, int len) 
    { 
        long long cur = 0; 
        long long ans = 0;
        for (int i = 0; i < len; i++) 
            cur = (cur + arr[i]) % mod; 
  
        if (cur < 0) ans = get(arr, 2*arr.size()); 
        else
            ans = (get(arr, 2*len) + (k - 2) * cur) % mod;  
        return ans; 
    } 
  
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        return helper(arr, k, arr.size());
    }
};
