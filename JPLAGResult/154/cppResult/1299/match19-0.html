class Solution {
    const int MOD = 1e9+7;
    int add(int a, int b) { return (a + b) % MOD; }
    int mul(long long a, int b) {
        a *= b;
        return a < MOD ? a : a % MOD;
    }
    long long solve1(vector<int> &arr) {
        int n = arr.size();
        int mn = 0, s = 0;
        int res = 0;
        for(int i=0;i<n;i++) {
            mn = min(mn, s);
            s += arr[i];
            res = max(res, s - mn);
        }
        return res;
    }
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        long long res = solve1(arr);
        if(k == 1 || res == 0) return res;
        long long temp = 0;
        int n = arr.size();
        long long mx = 0;
        for(int i=0;i<n;i++) {
            temp += arr[i];
            mx = max(mx, temp);
        }
        long long t = mx;
        mx = 0;
        long long s = 0;
        for(int i=n-1;i>=0;i--) {
            s += arr[i];
            mx = max(mx, s);
        }
        t += mx;
        if(temp > 0) t = add(t, mul(k - 2, temp));
        res = max(res, t);
        return res;
    }
};
