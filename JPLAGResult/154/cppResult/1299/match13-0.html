class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        long long sum = 0;
        long long inner_max = 0;
        long long p_sum = 0;
        long long left_max = 0;
        long long right_max = 0;
        for (int i = 0; i < arr.size(); ++i) {
            sum += arr[i];
            p_sum += arr[i];
            if (p_sum < 0) p_sum = 0;
            if (p_sum > inner_max) inner_max = p_sum;
            if (sum > left_max) left_max = sum;
        }
        p_sum = 0;
        for (int i = arr.size() - 1; i >= 0; --i) {
            p_sum += arr[i];
            if (p_sum > right_max) right_max = p_sum;
        }
        if (k == 1) return inner_max % 1000000007;
        long long ans = inner_max;
        if (left_max + right_max > ans) {
            ans = left_max + right_max;
        }
        if (left_max + right_max + sum * (k - 2) > ans) {
            ans = left_max + right_max + sum * (k - 2);
        }
        return ans % 1000000007;
    }
};
