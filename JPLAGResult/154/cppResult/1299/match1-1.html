#define ll long long

class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int a[1000000];
        for(int i = 0; i < arr.size(); i++) a[i] = arr[i];
        int mod = 1000000007;
        ll sum = 0, n = arr.size(), ans = 0;
        for (int i = 0; i < n; i++) {
			sum += a[i];
			if (sum < 0) sum = 0;
			else ans = max(ans, sum);
		}
		
		if (k > 1) {
			long long p = 0, s = 0, pref = 0, suff = 0;
			for (int i = 0; i < n; i++) {
				p += a[i];
				s += a[n - 1 - i];
				pref = max(pref, p);
				suff = max(suff, s);
			}
			
			if (pref + suff > 0) {
				long long sum = max(accumulate(a, a + n, 0ll), 0ll);
				ans = max(pref + suff + (k - 2) * sum, ans);
			}
		}
        if(ans < 0) return 0;
        return (ans % mod + mod) % mod;
    }
    
};
