const long long mod=1000000007;

class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int n=arr.size();
        for(int i=0;i<min(3,k-1);i++)
            for(int j=0;j<n;j++)
                arr.push_back(arr[j]);


        long long mx1=0;
        long long cur1=0;
        for(int i=0;i<arr.size();i++)
        {
            cur1+=arr[i];
            cur1=max(cur1,0LL);
            mx1=max(mx1,cur1);
        }
        if(k<=3)
            return mx1%mod;
        
        long long s=0;

        long long mx=0;
        long long cur=0;
        for(int i=0;i<n;i++)
        {
            s+=arr[i];
            cur+=arr[i];
            cur=max(cur,0LL);
            mx=max(mx,cur);
        }
        if(s<0)
            return max(mx,mx1)%mod;
        else
        {
            return max((mx+s%mod*(k-1)%mod),mx1)%mod;
        }
        
    }
};
