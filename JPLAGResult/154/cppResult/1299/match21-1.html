    class Solution {
    public:
        int kConcatenationMaxSum(vector<int> a, int k) {
            vector<int> b(a);
            int s1 = max_sum(b);
            b.insert(b.end(), a.begin(), a.end());
            int s2 = max_sum(b);
            b.insert(b.end(), a.begin(), a.end());
            int s3 = max_sum(b);

            if (s3 > s2 && s2 > s1) {
                int res = s1;
                for (int i = 0; i < k - 1; i++) {
                    res = (res + (s2 - s1)) % 1000000007;
                }
                return res;
            } else if (s2 > s1)
                return max(s2, 0);
            else
                return max(s1, 0);
        }

        int max_sum(vector<int>& a) {
            int s = 0, s_max = INT_MIN;
            for (int i = 0; i < a.size(); i++) {
                s = max(0, s) + a[i];
                s_max = max(s_max, s);
            }
            return s_max;
        }
    };
