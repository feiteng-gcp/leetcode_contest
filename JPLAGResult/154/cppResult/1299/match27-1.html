class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        long long leftMax, rightMax, count, oneMax, zero;

        leftMax = rightMax = count = oneMax = zero = 0;
        
        
        for (int i = 0; i < arr.size(); i++) {
            leftMax += arr[i];
            leftMax = max(zero, leftMax);
            oneMax = max(leftMax, oneMax);
            count += arr[i];
        }
        
        long long retTemp = oneMax;
        if (k == 1) return (int)(retTemp%1000000007);
        
        for (int i = arr.size() - 1; i >= 0 ; i--) {
            rightMax += arr[i];
            rightMax = max(zero, rightMax);
        }
        
        
        
        if (count > 0) {
            retTemp = max(retTemp, leftMax+rightMax+(count*(k-2)));
        } else {
            retTemp = max(retTemp, leftMax+rightMax);
        }
        
        return (int)(retTemp%1000000007);
        
    }
};
