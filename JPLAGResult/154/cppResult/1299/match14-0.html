const int M = 1000000007;

class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        long long sum = 0;
        long long lsum = 0;
        long long lmin = 0;
        long long cur = 0;
        long long best = 0;

        for (int ele : arr)
        {
            sum += ele;
            lsum = std::max(lsum, sum);
            lmin = std::min(lmin, sum);
            
            cur += ele;
            cur = std::max(0ll, cur);
            best = std::max(cur, best);
        }
        
        int rsum = sum - lmin;
        if (k > 1)
        {
            best = std::max(best, lsum + rsum);
            best = std::max(best, lsum + rsum + sum * (k - 2));
        }
        return best % M;
    }
};
