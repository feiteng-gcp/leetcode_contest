#define LL long long


class Solution {
public:

    LL sub(vector<int> a) {
        LL t = 0, maxx = 0;
        for(auto w: a) {
            t += w;
            if(t < 0) t = 0;
            if(t > maxx) maxx = t;
        }
        return maxx;
    }
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int n = arr.size();
        LL ans = sub(arr);
        if(k >= 2) {
            LL sum = 0;
            for(auto w: arr) sum += w;


            LL l = 0, r = 0;
            LL t = 0;
            for(int i = n - 1; i >= 0; i--) {
                t += arr[i];
                l = max(l , t);
            }
            t = 0;
            for(int i = 0; i < n; i++) {
                t += arr[i];
                r = max(r, t);
            }
            ans = max(ans, l + r + max(0LL, sum) * (k - 2));
        }
        return (int)(ans % (LL)(1e9 + 7));
    }
};
