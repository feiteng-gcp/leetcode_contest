class Solution {
public:
    
    int64_t mod = 1000000007;
    
    int64_t GetSubSum(vector<int>& vec)
    {
        int64_t curMax    = INT_MIN;

        int64_t curSum    = 0;
        int64_t minVal    = 0;
        
        for (int i = 0; i < vec.size(); ++i)
        {
            curSum   += vec[i];                        
            curMax    = max(curMax, curSum - minVal);
            minVal    = min(minVal, curSum);
        }
        
        return max(int64_t(0), curMax) % mod;
    }
    
    int kConcatenationMaxSum(vector<int>& arr, int k) {                
        
        vector<int> b = arr;
        b.insert(b.end(), arr.begin(), arr.end());
        
        int64_t arrSum = 0;
        for (auto item : arr)
            arrSum += item;
        
        if (arrSum < 0)
            return GetSubSum(b) % mod;
        else
        {
            int64_t res = GetSubSum(b);
            for (int i = 0; i < k - 2; ++i)
            {
                res += arrSum;
                res  %= mod;
            }
            return res;
        }            
    }
};
