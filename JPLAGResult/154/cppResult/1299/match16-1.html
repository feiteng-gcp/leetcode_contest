class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        long long sum = 0;
        long long pre = 0;
        long long after = 0;
        long long dp=0,dp1 = 0;
        long long mod = 1000000000 + 7;
        for(int i = 0;i < arr.size();i++){
            sum += arr[i];
            dp = dp + (long long)arr[i];
            if(dp < 0)
                dp = 0;
            dp1 = max(dp1,dp);
            pre = max(pre,sum);
        }

        dp = dp1;
        long long s = 0;
        for(int i = arr.size() - 1;i >= 0;i--){
            s += arr[i];
            after = max(after,s);
        }
        //cout<<pre<<" "<<dp<<" "<<after<<endl;
        if(k == 1)
            return dp % mod;
        if(k == 2){
            return max(pre + after,dp) % mod;
        }
        if(sum >= 0)
            return max(sum * (k-2) + pre + after,dp) % mod;
        else
            return max(dp,pre + after) % mod;
    }
};
