class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int dp = 0, res = 0;
        for (int i = 0; i < arr.size(); i++) {
            if (dp < 0) dp = 0;
            dp += arr[i];
            res = max(res, dp);
        }
        
        int mx_head = 0;
        int mx_tail = 0;
        int now = 0, sum = 0;
        for (int i = 0; i < arr.size(); i++) {
            now += arr[i];
            sum += arr[i];
            mx_head = max(mx_head, now);
        }
        now = 0;
        for (int i = arr.size()-1; i >= 0; i--) {
            now += arr[i];
            mx_tail = max(mx_tail, now);
        }
        if (k >= 2) {
            long long tmp = (long long)(k-2)*(long long)max(sum,0)%(1000000007);
            res = max(res, (int)(tmp + mx_head + mx_tail)%(1000000007));
        }
        return res;
    }
};
