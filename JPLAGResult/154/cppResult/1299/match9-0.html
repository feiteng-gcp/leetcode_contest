class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int len = arr.size();
        long  long mod = 1e9 + 7;
        long long sum = 0, cur = 0, maxi = 0, tmp;
        for (int i = 0; i < len; i++) {
            sum += arr[i];
            cur += arr[i];
            if (cur < 0) {
                cur = 0;
            }
            if (cur > maxi) {
                maxi = cur;
            }
        }
        if (k == 1) return maxi % mod;
        tmp = maxi;
        cur = 0;
        maxi = INT_MAX;
        for (int i = 0; i < len; i++) {
            cur += arr[i];
            if (cur > 0) {
                cur = 0;
            }
            if (cur < maxi) {
                maxi = cur;
            }
        }
        maxi = -maxi;
        if (sum >= 0) {
            sum = sum % mod;
            if (tmp + sum > maxi) {
                 return (tmp + (k - 1) * sum) % mod;
            }else {
                return (maxi + (k - 2) * sum) % mod;
            }
        } else {
            if (tmp > maxi + sum) {
                 return tmp % mod;
            } else {
                return (maxi + sum) % mod;
            }
        }
    }
};
