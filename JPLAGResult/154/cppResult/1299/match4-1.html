#define MO 1000000007

class Solution {
public:
    long f(vector<int> &arr) {

        int n = arr.size();
        long ans = 0;
        long s = 0;
        for (int i=0; i<n; i++) {
            s = arr[i] + max(0L, s);
            ans = max(s, ans);
        }
        return ans;
    }
    
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        long s = 0;
        for (int x: arr)
            s += x;
        // cout << "s : " << s << endl;
        long ls = 0;
        long lmax = 0;

        const int n = arr.size();
        for (int i=0; i<n; i++) {
            ls += arr[i];
            lmax = max(lmax, ls);
        }
        long rs = 0;
        long rmax = 0;
        for (int i=n-1; i>=0; i--) {
            rs += arr[i];
            rmax = max(rmax, rs);
        }
        // cout << "lmax: " << lmax << ", rmax: " << rmax << endl;
        if (s < 0) {
            if (k == 1) {
                return f(arr) % MO;
            } else {
                vector<int> xs = arr;
                for (int x: arr)
                    xs.push_back(x);
                return f(xs) % MO;
            }
        } else {
            long t = 0;
            if (k == 1) {
                t = max(t, f(arr));
            } else {
                vector<int> xs = arr;
                for (int x: arr)
                    xs.push_back(x);
                t = max(t, f(xs) % MO);
            }
            long ans = t;
            if (k >= 2) {
                ans = max(t, s*(k-2) + lmax + rmax);
            }
            return ans%MO;
        }
    }
};
