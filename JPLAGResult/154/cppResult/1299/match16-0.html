class Solution {
public:
	int kConcatenationMaxSum(vector<int>& arr, int k) {
		long long all_sum = 0;
		long long sum = arr[0];
		long long pos = 0;
		long long ans1 = arr[0];
		for (int i = 1; i < arr.size(); ++i)
		{
			if (sum < 0)
			{
				sum = arr[i];
				pos = i;
			}
			else sum = (sum + arr[i]) % 1000000007;
			ans1 = max(ans1, sum);
		}
		ans1 = max((long long) 0, ans1);
		if (k == 1) return ans1;
		long long part_sum = 0;
		for (int i = pos; i < arr.size(); ++i)
			part_sum = (part_sum + arr[i]) % 1000000007;
		sum = 0;
		long long ans3 = 0;
		for (int i = 0; i < arr.size(); ++i)
		{
			sum += arr[i];
			ans3 = max(ans3, sum);
		}
		for (int i = 0; i < arr.size(); ++i)
			all_sum =(all_sum + arr[i]) % 1000000007;
		vector<int> tmp = arr;
		tmp.insert(tmp.end(), arr.begin(), arr.end());
		long long ans2 = tmp[0];
		sum = tmp[0];
		for (int i = 1; i < tmp.size(); ++i)
		{
			if (sum < 0)
				sum = tmp[i];
			else sum = (sum + tmp[i]) % 1000000007;
			ans2 = max(ans2, sum);
		}
		ans2 = max((long long)0, ans2);
		if (ans1 == 0)
		{
			return 0;
		}
		long long ans = max((long long)0, max(all_sum % 1000000007 * k % 1000000007 , max(ans1, part_sum % 1000000007 + ans3 % 1000000007 + (k - 2)% 1000000007 * all_sum % 1000000007 )) % 1000000007);
		return max(ans, part_sum + ans3% 1000000007);
	}
};
