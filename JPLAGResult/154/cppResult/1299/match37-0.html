class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        int mod = 1e9+7;
        long long sum = 0;
        long long ans = 0;
        int n = arr.size();
        for(int i=0;i<n;i++) {
            sum += arr[i];
            sum %= mod;
        }
        // cout<<sum<<endl;
        // sum = 
        ans = max(ans, sum * k % mod);
        vector<int> b;
        for(int i=0;i<n;i++) {
            b.push_back(arr[i]);
        }
        if(k!=1) {
        for(int i=0;i<n;i++) {
            b.push_back(arr[i]);
        }
        }
        long long t=0;
        for(int i=0;i<b.size();i++) {
            t = max((long long)0, t + b[i]);
            t %= mod;
            ans = max(t, ans);
        }
        long long l=0;
        long long r=0;
        t=0;
        for(int i=0;i<arr.size();i++) {
            t += arr[i];
            t %= mod;
            r = max(t, r);
        }
        t=0;
        for(int i=arr.size()-1;i>=0;i--) {
            t += arr[i];
            t %= mod;
            l = max(t, l);
        }
        // cout<<ans<<endl;
        // cout<<l<<" "<<r<<" "<<sum<<endl;
        if(k>=3) ans = max(ans, (l+r+(k-2)*sum)%mod);
        return (int)ans;
    }
};
