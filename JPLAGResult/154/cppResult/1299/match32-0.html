class Solution {
public:
    
    int helper(vector<int>& array) {
        int n = array.size();
        if (n <= 0) return -1;
        if (n == 1) return array[0];
        int res = array[0];
        int tmp = array[0];
        for (int i = 1; i < n; i++) {
            if (tmp >= 0) {
                tmp += array[i];
            } else {
                tmp = array[i];
            }
            res = max(tmp, res);
        }
        return res;
    }
    
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        if (arr.size() < 1) return 0;
        if (k < 1) return 0;
        vector<long long> forw(arr.size());
        vector<long long> backw(arr.size());
        forw[0] = arr[0];
        long long forwmax = arr[0];
        for (int i = 1; i < arr.size(); i++) {
            forw[i] = forw[i-1] + arr[i];
            forwmax = max(forwmax, forw[i]);
        }
        backw[arr.size() - 1] = arr[arr.size() - 1];
        long long backwmax = arr[arr.size() - 1];
        for (int i = arr.size()-2; i >= 0; i--) {
            backw[i] = backw[i+1] + arr[i];
            backwmax = max(backwmax, backw[i]);
        }
        bool flag = true;
        long long num = 0;
        for (int i = 0; i < arr.size(); i++) {
            if (arr[i] > 0) flag = false;
            num += arr[i];
        }
        if (flag) return 0;
        long long res1 = helper(arr);
        long long res2 = max(forwmax + backwmax, max(forwmax, backwmax));
        long long res3 = res2 + max((long long)0, max(0, k - 2) * num);
        return max((long long)0, max(res1, max(res2, res3))) % 1000000007;
    }
};
