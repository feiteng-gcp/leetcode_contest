# define mod 1000000007
class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {
        long predix = 0; 
        long suffix = 0;
        long midfix = 0;
        long sums = 0;
        long cur = 0;
        long s=0;
        for ( int i=0; i<arr.size(); ++i ){
            cur = (cur + arr[i])%mod;
            predix = max( predix, cur );
            
            s = (s+arr[i])%mod;
            midfix = max( midfix, s );
            if ( s<0 ) s=0;
            
            sums = (sums + arr[i])%mod;
        }
        
        cur = 0;
        for ( int i=arr.size()-1; i>=0; --i ){
            cur += arr[i];
            suffix = max( suffix, cur );
        }
        
        long x = max( predix, max( suffix, (predix+suffix)%mod ) );
        long y = max( midfix, ((sums%mod)*(k%mod))%mod );
        
        long t = max( x,y );
        int res = max( t, (predix+suffix)%mod+((sums)* ((max((k-2),0))%mod))%mod );
        return res;
        
    }
};
