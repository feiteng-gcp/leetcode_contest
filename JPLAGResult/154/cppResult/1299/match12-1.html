class Solution {
public:
    int kConcatenationMaxSum(vector<int>& arr, int k) {

        long long min_sum = 0, sum = 0, max_prefix = 0, max_inner = 0;
        for (int i = 0; i < arr.size(); ++i) {
            sum += arr[i];
            max_inner = max(max_inner, sum - min_sum);
            min_sum = min(min_sum, sum);
            max_prefix = max(max_prefix, sum);
        }
        sum = 0;

        long long max_suffix = 0;
        for (int i = arr.size() - 1; i >= 0; --i) {
            sum += arr[i];
            max_suffix = max(max_suffix, sum);
        }
        long long ans = max_inner;
        if (k >= 2) {
            if (sum < 0) {
                sum = 0;
            }
            ans = max(ans, max_suffix + (k - 2) * sum + max_prefix);
        } 
        return ans % MODULO;
    }
private:
    const int MODULO = 1e9 + 7;
};
