class Solution {
    public List<List<Integer>> criticalConnections(int n, List<List<Integer>> connections) {
        return bridges(n, connections);
    }
    
    int time = 0;
    final int NIL = -1;
    List<List<Integer>> res = new LinkedList<>();
    
    class Graph{
        private int v;
        private LinkedList<Integer> adj[]; 
        Graph(int V) 
        { 
            v = V; 
            adj = new LinkedList[V+1]; 
            for (int i=0; i<V+1; ++i) 
                adj[i] = new LinkedList(); 
        }
        
        void addEdge(int V, int w) 
        { 
            this.adj[V].add(w);
            this.adj[w].add(V);
        }
    }
    
   void bridgeDfs(Graph g,
int u,boolean visited[], int disc[], int low[], int parent[] )
    {
        visited[u] = true;
        disc[u]=low[u]=++time;
        
        Iterator<Integer> i = (g.adj[u]).iterator();
        while(i.hasNext())
        {
            int v = i.next();
            if(!visited[v])
            {
                parent[v]=u;
                bridgeDfs(g,v,visited,disc,low,parent);
                low[u] = Math.min(low[u],low[v]);
                if(low[v]>disc[u]) {
                    List<Integer> cur = new ArrayList<>();
                    cur.add(u);
                    cur.add(v);
                    res.add(cur);
                }
            }
            else if(v!=parent[u])
            {
                low[u] = Math.min(low[u],disc[v]);
            
}
        }
    }
    
    List<List<Integer>> bridges(int v, List<List<Integer>> edges) {
        Graph g = new Graph(v);
        for(int i = 0;i < edges.size(); i++)
        {
            
g.addEdge(edges.get(i).get(0), edges.get(i).get(1));
        }
        boolean visited[] = new boolean[v+1];
        int disc[] = new int[v+1];
        int low[] = new int[v+1];
        int parent[] = new int[v+1];
        
        for(int i=0;i<v;i++){
            parent[i]=NIL;
            visited[i]=false;
        }
        

        for(int i=0;i<v;i++)
        {
            if(!visited[i])
            {
                bridgeDfs(g,i, visited, disc, low, parent);
            }
        }
        
        return res;
    }
}
