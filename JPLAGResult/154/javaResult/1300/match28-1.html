class Solution {
    public LinkedList <Integer> [] neigh;
    
int n, cnt = 0;
    boolean visited[];
    int [] seen, parent, first;
    LinkedList <List<Integer>> ans = new LinkedList();
    public List<List<Integer>> criticalConnections(int n0, List<List<Integer>> E) {
        n = n0;
        neigh = new LinkedList[n];
        visited = new boolean[n];
        seen = new int[n];
        parent = new int[n]; 
        first = new int[n];        
        for(int i = 0; i < n; i ++)
            neigh[i] = new LinkedList()
;
        for(List<Integer> edge : E)
        {
            int a = edge.get(0), b = edge.get(1);
            neigh[a].add(b);
            neigh[b].add(a);
        }
        Arrays.fill(parent, -1);
        for(int i = 0 ; i < n ; i ++)
        {
            if(!visited[i])
                dfs(i);
        }
        /*
        System.out.println("seen:  " + Arrays.toString(seen));
        System.out.println("parent:" + Arrays.toString(parent));
        System.out.println("first: " + Arrays.toString(first));*/
        return ans;
    
}
    public void dfs(int k)
    {
        //System.out.println(k + ": " + neigh[k].toString());
        visited[k] = true;
        seen[k] = first[k] = ++cnt;
        for(int i : neigh[k])
        {
            if(!visited[i])
            {
                parent[i] = k;
                dfs(i);
                first[k] = Math.min(first[k], first[i]);
                if(first[i] > seen[k])
                {
                    LinkedList<Integer> tmp = new LinkedList();
                    tmp.add(i);
                    tmp.add(k);
                    ans.add(tmp);
                
}
            }
            else if(i != parent[k])
            {
                first[k] = Math.min(first[k], seen[i]);
            }
        }
    }
}
