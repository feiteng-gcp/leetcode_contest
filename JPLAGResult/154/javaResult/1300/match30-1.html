class Solution {
    private int time;
    private int[] discoverTime;
    private int[] low;
    private int[] parent;
    private boolean[] visited;
    private List<Integer>[] graph;
    private List<List<Integer>> res;
    
    private void buildGraph(int n, List<List<Integer>> connections) {
        graph = 
new List[n];
        for(int i = 0; i < n; i++) {
            graph[i] = new ArrayList<>();
        }
        for(List<Integer> c:connections) {
            int start = c.get(0), end = c.get(1);
            graph[start].add(end);
            graph[end].add(start);
        }
        System.out.println(graph[0]);
    
}
    
    private void visit(int i) {
        visited[i] = true;
        time++;
        discoverTime[i] = time;
        low[i] = time;
        for(int next:graph[i]) {
            if(!visited[next]) {
                parent[next] = i;
                visit(next);
                low[i] = Math.min(low[i], low[next]);
                if(low[next] > discoverTime[i]) {
                    List<Integer> temp = new ArrayList<>();
                    temp.add(i);
                    temp.add(next);
                    res.add(temp);
                
}
            }
            else if (next != parent[i]) {
                low[i] = Math.min(low[i], discoverTime[next]);
            }
        }
    }
    
    public List<List<Integer>> criticalConnections(int n, List<List<Integer>> connections) {
        buildGraph(n, connections);
        discoverTime = new int[n];
        low = new int[n];
        parent = new int[n];
        visited = new boolean[n];
        res = new ArrayList<>();
        for(int i = 0; i < n; i++) {
            parent[i] = -1;
            visited[i] = false;
        }
        for(int i = 0; i < n; i++) {
            if(!visited[i])
                visit(i);
        }
        return res;
    }
}
