class Solution {
    List<List<Integer>> res = new ArrayList<>();
    private LinkedList<Integer> adj[];
    int time = 0;
    static final int NIL = -1;
    
    public void bridgeUtil(int u, boolean visited[], int disc[],
                    int low[], int parent[]){
        visited[u] = true;

        disc[u] = low[u] = ++time;

        Iterator<Integer> i = adj[u].iterator();
        while (i.hasNext())
        {
            int v = i.next();  

            if (!visited[v]){
                parent[v] = u;
                bridgeUtil(v, visited, disc, low, parent);

                low[u]  = Math.min(low[u], low[v]);

                if (low[v] > disc[u]){
                    List<Integer> list = new LinkedList<>();
                    list.add(u); list.add(v);
                    res.add(list);
                }
            }
            else if (v != parent[u])
                low[u]  = Math.min(low[u], disc[v]);
        }
    }
    
    
    public List<List<Integer>> criticalConnections(int V, List<List<Integer>> connections) {
        int size = connections.size();
        
        adj = new LinkedList[V];
        for (int i = 0; i < V; ++i)
            adj[i] = new LinkedList();
        for(int i = 0; i < size; ++i){
            adj[connections.get(i).get(0)].add(connections.get(i).get(1)); 
            adj[connections.get(i).get(1)].add(connections.get(i).get(0)); 
        }
        
        boolean visited[] = new boolean[V];
        int disc[] = new int[V];
        int low[] = new int[V];
        int parent[] = new int[V];
        
        for(int i = 0; i < V; i++){
            parent[i] = NIL;
            visited[i] = false;
        }
        for (int i = 0; i < V; i++){
            if(visited[i] == false)
                bridgeUtil(i, visited, disc, low, parent);
        }
        return res;
    }
}
