class Solution {
    
    boolean [] visited;
    int [] disc;
    int [] low;
    int [] parent;
    Map<Integer, List<Integer>> adj;
    List<List<Integer>> res;
    int time = 0;
    
    private void bridgeFind(int start){
        visited[start] = true;
        time++;
        disc[start] = time;
        low[start] = time;
        if(adj.containsKey(start) == false){
            return;
        }
        for(int dst: adj.get(start)){
            if(visited[dst] == false){
                parent[dst] = start;
                bridgeFind(dst);
                low[start] = Math.min(low[start], low[dst]);
                if(low[dst] > disc[start]){
                    List<Integer> temp = new ArrayList<>();
                    temp.add(start);
                    temp.add(dst);
                    res.add(temp);
                }
            }else if(dst != parent[start]){
                low[start] = Math.min(low[start], disc[dst]);
            }
        }
    }
    
    public List<List<Integer>> criticalConnections(int n, List<List<Integer>> connections) {
        visited = new boolean[n];
        disc = new int[n];
        low = new int[n];
        parent = new int[n];
        
        Arrays.fill(visited, false);
        Arrays.fill(parent, -1);
        
        adj = new HashMap<>();
        for(List<Integer> cur: connections){
            int s = cur.get(0);
            int e = cur.get(1);
            adj.computeIfAbsent(s, x -> new ArrayList<>()).add(e);
            adj.computeIfAbsent(e, x -> new ArrayList<>()).add(s);
        }
        
        res = new ArrayList<>();
        
        for(int i = 0; i < n; i++){
            if(visited[i] == false){
                bridgeFind(i);
            }
        }
        
        return res;
        
    }
}
