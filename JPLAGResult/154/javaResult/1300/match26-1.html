class Solution {
    
    List<List<Integer>> bridges;
    int time;
    
    
public List<List<Integer>> criticalConnections(int n, List<List<Integer>> connections) {
        Vertex[] vertices = new Vertex[n];
        for (int i = 0; i < n; i++)
        {
            vertices[i] = new Vertex(i);
        }
        for (List<Integer> pair : connections)
        {
            vertices[pair.get(0)].add(vertices[pair.get(1)]);
        }
        bridges = new ArrayList<>();
        time = 0;
        
for (Vertex v : vertices)
        {
            if (!v.visited)
            {
                bridge(v);
            }
        }
        return bridges;
    
}
    
    public void bridge(Vertex curr)
    {
        // System.out.println(curr.i);
        curr.visited = true;
        
        curr.disc = curr.low = ++time;
        
        for (Vertex other : curr.neighbors)
        {
            if (!other.visited)
            {
                other.parent = curr;
                bridge(other);
                curr.low = Math.min(curr.low, other.low);
                if (other.low > curr.disc)
                {
                    ArrayList<Integer> pair = new ArrayList<>();
                    pair.add(curr.i);
                    pair.add(other.i);
                    bridges.add(pair);
                
}
            }
            else if (other != curr.parent)
            {
                curr.low = Math.min(curr.low, other.disc);
            }
        }
    }
    
    class Vertex
    {
        ArrayList<Vertex> neighbors;
        int i;
        Vertex parent;
        boolean visited;
        int low, disc;
        
        public Vertex(int i)
        {
            neighbors = new ArrayList<>();
            this.i = i;
        }
        
        public void add(Vertex other)
        {
            neighbors.add(other);
            other.neighbors.add(this);
        }
    }
}
