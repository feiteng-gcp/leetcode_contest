class Solution {
     private int n;
    private Map<Integer, List<Integer>> graph = new HashMap<>();
    int visitTime = 0;
    List<List<Integer>> r = new ArrayList<>();

    public List<List<Integer>> criticalConnections(int n, List<List<Integer>> connections) {
        for (List<Integer> es : connections) {
            int start = es.get(0);
            int end = es.get(1);
            graph.computeIfAbsent(start, k -> new ArrayList<>()).add(end);
            graph.computeIfAbsent(end, k -> new ArrayList<>()).add(start);
        }
        this.n = n;
        findBridge();
        return r;
    }


    private void dfs(
int u, boolean visited[], int disc[],
             int low[], int parent[]) {
        visited[u] = true;
        visitTime++;
        disc[u] = visitTime;
        low[u] = visitTime;
        Iterator<Integer> iter = graph.getOrDefault(u, new ArrayList<>()).iterator();
        
while (iter.hasNext()) {
            int v = iter.next();
            if (!visited[v]) {
                parent[v] = u;
                dfs(v, visited, disc, low, parent);
                low[u] = Math.min(low[u], low[v]);
                if (low[v] > disc[u]) {
                    List<Integer> edge = new ArrayList<>();
                    edge.add(u);
                    edge.add(v);
                    r.add(edge);
                }
            } else if (v != parent[u]) {
                low[u] = Math.min(low[u], disc[v]);
            }
        }
    }


    private void findBridge() {
        
boolean visited[] = new boolean[n];
        int disc[] = new int[n];
        int low[] = new int[n];
        int parent[] = new int[n];
        for (int i = 0; i < n; i++) {
            parent[i] = -1;
            visited[i] = false;
        }
        for (int i = 0; i < n; i++) {
            if (visited[i] == false) {
                dfs(i, visited, disc, low, parent);
            }
        }
    }
}
