class Solution {
    int time = 0;
    final int NIL = -1;

    Map<Integer, List<Integer>> graph;
    int vertexNum;
    List<List<Integer>> result;

    private void findCritical() {
        boolean visited[] = new boolean[vertexNum + 1];
        int discTime[] = new int[vertexNum + 1];
        int low[] = new int[vertexNum + 1];
        int parent[] = new int[vertexNum + 1];

        for (int i = 0; i <= vertexNum; i++) {
            parent[i] = NIL;
            visited[i] = false;
        }

        for (int i = 0; i <= vertexNum; i++) {
            if (visited[i] == false) {
                findCriticalHelper(i, visited, discTime, low, parent);
            }
        }
    }

    private void findCriticalHelper(int u, boolean visited[], int discTime[],
                                    int low[], int parent[]) {
        visited[u] = true;
        discTime[u] = low[u] = ++time;

        List<Integer> neighbours = graph.get(u);
        if (neighbours == null)
            return;
        for (int v : neighbours) {
            if (!visited[v]) {
                parent[v] = u;
                findCriticalHelper(v, visited, discTime, low, parent);

                low[u] = Math.min(low[u], low[v]);
                if (low[v] > discTime[u]) {
                    List<Integer> pair = Arrays.asList(u, v);
                    result.add(pair);
                }
            } else if (v != parent[u]) {
                low[u] = Math.min(low[u], discTime[v]);
            }
        }
    }
    
    public List<List<Integer>> criticalConnections(int n, List<List<Integer>> connections) {
        this.vertexNum = n;
        graph = new HashMap<>();
        result = new ArrayList<>();

        for (List<Integer> pairInt : connections) {
            graph.putIfAbsent(pairInt.get(0), new ArrayList<>());
            graph.get(pairInt.get(0)).add(pairInt.get(1));
            graph.putIfAbsent(pairInt.get(1), new ArrayList<>());
            graph.get(pairInt.get(1)).add(pairInt.get(0));
        }

        findCritical();
        return result;
    }
}
