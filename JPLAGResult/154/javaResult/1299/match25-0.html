class Solution {
    public int kConcatenationMaxSum(int[] arr, int k) {
        int mod = 1000000007;
        int[] dp = new int[arr.length];
        dp[0] = arr[0];
        int sum = arr[0];
        int max = arr[0];
        for (int i = 1; i < arr.length; i++) {
            sum += arr[i];
            dp[i] = Math.max(dp[i - 1] + arr[i], arr[i]);
            dp[i] %= mod;
            max = Math.max(max, dp[i]);
        }
        if (k == 1) return max > 0 ? max : 0;
        dp[0] = Math.max(dp[dp.length - 1] + arr[0], arr[0]);
        max = Math.max(max, dp[0]);
        for (int i = 1; i < arr.length; i++) {
            dp[i] = Math.max(dp[i - 1] + arr[i], arr[i]);
            dp[i] %= mod;
            max = Math.max(max, dp[i]);
        }
        if (k == 2 || sum <= 0) return max > 0 ? max : 0;
        int more = 0;
        for (int i = 0; i < k - 2; i++) {
            more += sum;
            more %= mod;
        }
        return (max + more) % mod;
    }
}
