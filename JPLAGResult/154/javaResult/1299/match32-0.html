class Solution {
    public int kConcatenationMaxSum(int[] arr, int k) {
        int n = arr.length;
        if(k== 1){
            long sum = 0, max = 0;
            for(int i = 0; i< arr.length; i++){
                sum += arr[i];
                sum = Math.max(sum, 0);
                max = Math.max(max, sum);
            }
            return (int)(max%((int)1e9+7));
        }
        long[] aa = new long[2*n];
        for(int i = 0; i< n; i++)aa[i] = aa[i+n] = arr[i];
        long sum = 0;
        for(long i:arr)sum += i;
        long ans = sum*k;
        
        ans = Math.max(ans, mx(aa)+(k-2)*Math.max(sum, 0));
        return (int)(ans%((int)1e9+7));
    }
    long mx(long[] a){
        long sum = 0, max = 0;
            for(int i = 0; i< a.length; i++){
                sum += a[i];
                sum = Math.max(sum, 0);
                max = Math.max(max, sum);
            }
        return max;
    }
    long mx(int[] a){
        long sum = 0, max = 0;
            for(int i = 0; i< a.length; i++){
                sum += a[i];
                sum = Math.max(sum, 0);
                max = Math.max(max, sum);
            }
        return max;
    }
}
