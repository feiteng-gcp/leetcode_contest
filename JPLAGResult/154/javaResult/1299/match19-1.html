class Solution {
    int mod = 1_000_000_000 + 7;    
    
    public int kConcatenationMaxSum(int[] arr, int k) {
        
        int ans = 0;
        long total = 0;
        for (int v : arr) {
            total += v;            
        }
        long maxLeft = 0;                   
        
long left = 0;
        for (int v : arr) {                
            left += v;
            maxLeft = Math.max(maxLeft, left);
        }              
        long maxRight = 0;
        long right = 0;
        for (int i = arr.length - 1; i >= 0; --i) {
            right += arr[i];
            maxRight = Math.max(maxRight, right);
        }
        if (total > 0) {
            ans = Math.max(ans, (int)((total * k) % mod));
            ans = Math.max(ans, (int)((maxRight + (total * (k - 1))) % mod));
            ans = Math.max(ans, (int)( (maxRight + (total * (k - 2)) + maxLeft) % mod));
        } else {
            ans = Math.max(ans, (int)((maxRight + maxLeft) % mod));
        }
        
        return Math.max(ans, maxSum(arr));                             
    }
    
    public int maxSum(int[] arr) {
        
int ans = 0;
        long sum = 0;
        for (int i = 0; i < arr.length; ++i) {
            if (sum + arr[i] < 0) {
                sum = 0;
            } else {
                sum += arr[i];
            }
            ans = Math.max(ans, (int)(sum % mod));
        }
        return ans;
    }
    
    public int maxSumList(List<Integer> list) {
        int ans = 0;
        long sum = 0;
        for (int i = 0; i < list.size(); ++i) {
            long s = sum + list.get(i);
            if (s < 0) {
                sum = 0;
            } else {
                sum = s;
            }
            ans = Math.max(ans, (int)sum % mod);
        }
        return ans;
    }
}
