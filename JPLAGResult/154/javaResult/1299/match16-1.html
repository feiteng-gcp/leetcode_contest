class Solution {
      public int kConcatenationMaxSum(int[] a, int k) {
        int n=a.length;
        long sum=0,mod=1000000007;
        for (int i = 0; i < n; i++) {
            sum=(sum+a[i])%mod;
        }

        long pre=0,sp=0,suf=0,ss=0;
        for (int i = 0; i <n ; i++) {
            sp=(sp+a[i])%mod;
            if(pre<sp)pre=sp;
        }
        for (int i = n-1; i >=0 ; i--) {
            ss=(ss+a[i])%mod;
            if(ss>suf)suf=ss;
        }

        if(k==1){
            long dp[]=new long[n];
            dp[0]=Math.max(a[0],0);
            
long ans=dp[0];
            for (int i = 1; i < n; i++) {
                dp[i]=Math.max(dp[i-1],0)+a[i];
                ans=Math.max(dp[i],ans);
            }
            return (int)ans;
        }
        else if(k==2){
            long dp[]=new long[n];
            dp[0]=Math.max(a[0],0);
            long ans=dp[0];
            for (int i = 1; i < n; i++) {
                dp[i]=Math.max(dp[i-1],0)+a[i];
                ans=Math.max(dp[i],ans);
            }
            ans=Math.max(pre+suf,ans);
            return (int)ans;
        }
        else{
            long dp[]=new long[n];
            dp[0]=Math.max(a[0],0);
            long ans=dp[0];
            for (int i = 1; i < n; i++) {
                dp[i]=Math.max(dp[i-1],0)+a[i];
                ans=Math.max(dp[i],ans);
            }
            ans=Math.max(ans,(((k-2)*sum)%mod+suf+pre)%mod);
            ans=Math.max(suf+pre,ans);
            return (int)ans;
            
        }



    }

}
