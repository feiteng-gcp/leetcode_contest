	class Solution {
	    public int kConcatenationMaxSum(int[] a, int k) {
	        int n = a.length;
	        int[] cums = new int[n+1];
	        for(int i =0;i < n;i++)cums[i+1] = cums[i] + a[i];
	        
	        int rotmax = 0;
	        int sufmax = 0;
        	for(int i = n;i >= 0;i--){
        		sufmax = Math.max(sufmax, cums[n] - cums[i]);
        	}
	        int pmax = 0;
        	for(int i = n;i >= 0;i--){
        		pmax = Math.max(pmax, cums[i]);
        	}
        	rotmax = sufmax + pmax;
        	
        	int inmax = 0;
	        int premin = 0;
        	for(int i = 1;i <= n;i++){
        		inmax = Math.max(inmax, cums[i] - premin);
        		premin = Math.min(premin, cums[i]);
        	}
        	
        	int mod = 1000000007;
        	long ret = inmax;
        	if(k >= 2)ret = Math.max(ret, rotmax);
        	if(k >= 2)ret = Math.max(ret, rotmax + (long)(k-2)*cums[n]);
//        	tr(a, k, ret%mod, max(a, k), rotmax);
//        	assert ret%mod == max(a, k);
        	return (int)(ret%mod);
	    }
	    
	    int max(int[] a, int k)
	    {
	    	int n = a.length;
	    	int[] b = new int[n*k];
	    	for(int i = 0;i < n*k;i++){
	    		b[i] = a[i%n];
	    	}
	        long[] cums = new long[n*k+1];
	        for(int i =0;i < n*k;i++)cums[i+1] = cums[i] + b[i];
        	long inmax = 0;
	        long premin = 0;
        	for(int i = 1;i <= n*k;i++){
        		inmax = Math.max(inmax, cums[i] - premin);
        		premin = Math.min(premin, cums[i]);
        	}
        	return (int)(inmax%1000000007);
	    }
	}
