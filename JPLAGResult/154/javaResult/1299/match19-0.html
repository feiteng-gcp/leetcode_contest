class Solution {
   int Mod = 1000000007;

    public int kConcatenationMaxSum(int[] a, int k) {
        long sum = 0;
        int n = a.length;
        long max = Integer.MIN_VALUE;
        long prefix = 0;
        for (int i = 0; i < n; i++) {
            sum += a[i];
            if (prefix < 0) {
                prefix = a[i];
            } else {
                prefix += a[i];
            }
            max = Math.max(max, prefix);
        }

        if (sum > 0) {
            return (int) (sum * k % Mod + (max - sum) % Mod);
        }

        long dsum = 0;
        long[] dmax = new long[n];
        for (int i = 0; i < n; i++) {
            dsum += a[i];
            dsum %= Mod;
            dmax[i] = Math.max(i == 0 ? 0 : dmax[i - 1], dsum);
        }
        long rsum = 0;
        long[] rmax = new long[n + 1];
        for (int i = n - 1; i >= 0; i--) {
            rsum += a[i];
            rsum %= Mod;
            rmax[i] = Math.max(rmax[i + 1], rsum);
        }
        long maxsum = 0;
        for (int i = 0; i < n; i++) {
            maxsum = Math.max(maxsum, (dmax[i] + rmax[i + 1]) % Mod);
        }
        long rt = Math.max(maxsum, max);
        if (rt < 0) {
            return 0;
        }
        return (int) rt;
    }
}
