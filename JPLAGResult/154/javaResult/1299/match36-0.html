class Solution {
    public int kConcatenationMaxSum(int[] arr, int k) {
        long sum = 0;
        int n = arr.length, mod = (int)1e9 + 7;
        for(int i : arr)
            sum += i;
        int [] arr2 = new int[2 * n];
        for(int i = 0; i < n ; i ++)
            arr2[i] = arr2[i + n] = arr[i];
        long res = 0, max = 0;
        for(int i = 0; i < 2 * n ; i ++)
        {
            res += arr2[i];
            max = Math.max(max, res);
            res = Math.max(res, 0);
        }
        return (int)(((k - 2) * Math.max(sum, 0) + max ) % mod);
    }
}
