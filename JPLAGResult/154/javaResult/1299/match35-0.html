class Solution {
    final int MOD = (int)1e9 + 7;
    int N;
    public int kConcatenationMaxSum(int[] arr, int k) {
        N = arr.length;
        long sum = 0;
        for(int i = 0; i < N; ++i){
            sum = sum + arr[i];
        }
        int findL = findLeft(arr), findR = findRight(arr), findM = findMax(arr);
        if(k == 1) return findM;
        long l = (long)findL + (long)findR;
        if(k == 2) return Math.max( (int)(l % MOD), findM);
        
        long s = sum * (k - 2);
        long res = Math.max(s + findL + findR, (long)findM);
        res = Math.max((long)findL + (long)findR, res);
        
        return (int)(res % MOD);
    }
    
    public int findLeft(int[] arr){
        int res = arr[N - 1],  sum = arr[N - 1];
        for(int i = N - 2; i >= 0; --i){
            sum = sum + arr[i];
            res = Math.max(res, sum);
        }
        return Math.max(res, 0);
    }
    
    public int findRight(int[] arr){
        int res = arr[0],  sum = arr[0];
        for(int i = 1; i < N; ++i){
            sum = sum + arr[i];
            res = Math.max(res, sum);
        }
        return Math.max(res, 0);
    }
    
    public int findMax(int[] arr){
        int min = arr[0], sum = arr[0], res = min;
        for(int i = 1; i < N; ++i){
            sum = sum + arr[i];
            res = Math.max(res, sum - min);
            min = Math.min(min, sum);
        }
        return Math.max(res, 0);
    }
    
}
