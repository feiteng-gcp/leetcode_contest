class Solution {
    static final int MOD = 1000000000 + 7;

    static int maxSubsequenceSum(int[] arr) {
        int tmpSum = 0;
        int maxSum = 0;
        for (int i=0; i<arr.length;i++) {
            tmpSum = (tmpSum+arr[i]) > arr[i] ? (tmpSum+arr[i]): arr[i];
            
maxSum = Math.max(maxSum, tmpSum);
        }
        return maxSum;
    }

    public int kConcatenationMaxSum(int[] arr, int k) {
        int singleSum = 0;
        for (int val : arr) {
            singleSum += val;
        }
        if (singleSum < 0) {   //只取一部分最大值即可
            //一个数组的最大子段和
            if (k==1)
                return maxSubsequenceSum(arr);
            else {
                int[] arr2 = new int[arr.length*2];
                for(int i=0;i<arr.length;i++) {
                    arr2[i] = arr[i];
                    arr2[i+arr.length] = arr[i];
                }
                return maxSubsequenceSum(arr2);
            }
        } else {  //中间部分将会连续累加
            
int[] arr1 = new int[arr.length*2 + 1];
            for (int i=0;i<arr.length;i++) {
                arr1[i] = arr[i];
                arr1[i+1+arr.length] = arr[i];
            }
            int middleSum = 0;
            for (int j=0;j<k-2;j++) {
                middleSum = (middleSum+singleSum)%MOD;
            }
            arr1[arr.length] = middleSum;
            return maxSubsequenceSum(arr1);
        }
    }
}
