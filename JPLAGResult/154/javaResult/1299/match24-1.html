import java.math.BigInteger;

class Solution {
    private static final long mod = 1000000007;
    public int kConcatenationMaxSum(int[] arr, int k) {
        int sum = 0;
        for (int i : arr) {
            sum+=i;
        }
        if(k == 1){
            return (int)(calcMaxSum(arr) % mod);
        }

        
int[] newArr = new int[arr.length*2];
        for (int i = 0; i < arr.length; i++) {
            newArr[i] = arr[i];
            newArr[i+arr.length] = arr[i];
        }
        long maxSum = calcMaxSum(newArr);
        if(k == 2){
            return (int)(maxSum % mod);
        }
        if(sum >= 0){
            BigInteger midSum = BigInteger.valueOf(sum).multiply(BigInteger.valueOf(k - 2));
            BigInteger resultSum = midSum.add(BigInteger.valueOf(maxSum));
            return resultSum.mod(BigInteger.valueOf(mod)).intValue();
        }
        return (int)(maxSum % mod);
    }

    long calcMaxSum(int[] arr){
        long maxSum = 0;
        long nowSum = 0;
        for (int i : arr) {
            if(nowSum < 0){
                nowSum = 0;
            }
            nowSum+=i;
            
maxSum = Math.max(nowSum,maxSum);
        }
        return maxSum;
    }

    public static void main(String[] args) {
        System.out.println(new Solution().kConcatenationMaxSum(new int[]{1,2},3));
        System.out.println(new Solution().kConcatenationMaxSum(new int[]{1,-2,1},5));
        System.out.println(new Solution().kConcatenationMaxSum(new int[]{-1,-7,11,-2},3));
    }
}
