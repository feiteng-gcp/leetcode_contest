class Solution {
    int mod = (int)(1e9+7);
    public int kConcatenationMaxSum(int[] arr, int k) {
        
        // if sum > 0, then subarray sum of [arr, arr] + (k - 2) * sum
        // if sum <= 0, then Math.max(0, subarray sum of [arr, arr])
        int sum = sum(arr);
        int res = 0;
        if (k == 1) return Math.max(0, maxSubarraySum(arr));
        int[] copied = new int[2 * arr.length];
        System.arraycopy(arr, 0, copied, 0, arr.length);
        System.arraycopy(arr, 0, copied, arr.length, arr.length);
        
        if (sum > 0) {
            // res = (sum * (k - 2) % mod + maxSubarraySum(copied)) % mod;
            for (int i = 0; i < k - 2; i++) {
                res = (res + sum) % mod;
            }
            res = (res + maxSubarraySum(copied)) % mod;
        } else {
            res = maxSubarraySum(copied);
        }
        if (res < 0) return 0;
        return res;
        
    }
    
    private int sum(int[] arr) {
        int sum = 0;
        for (int i : arr) {
            sum = (sum + i) % mod;
        }
        return sum;
    }
    
    private int maxSubarraySum(int[] arr) {
        // dp[i] = max(arr[i] + dp[i - 1], arr[i])
        int[] dp = new int[arr.length];
        dp[0] = arr[0];
        
int maxSum = dp[0];
        for (int i = 1; i < arr.length; i++) {
            dp[i] = Math.max((arr[i] + dp[i - 1]) % mod, arr[i] % mod);
            maxSum = Math.max(maxSum, dp[i]);
        }
        return maxSum;
    }
}
