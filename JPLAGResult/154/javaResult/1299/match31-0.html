class Solution {
    private static final int MOD = 1000000007;
    
    public int kConcatenationMaxSum(int[] arr, int k) {
        return (int)(getAns(arr, k) % MOD);
    }
    
    private static long getAns(int[] arr, int k) {
        final int N = arr.length;
        
        if (k == 1) {
            return getBestNormal(arr);
        }
        
        long arrSum = 0;
        for (int v : arr) {
            arrSum += v;
        }
        
        int[] twice = new int[N + N];
        for (int i = 0; i < N; ++i) {
            twice[i] = twice[i + N] = arr[i];
        }
        long partial = getBestNormal(twice);
        
        if (arrSum < 0) {
            return partial;
        }
        
        // long[] bestPrefix = getBestPrefix(arr);
        // long[] bestSuffix = getBestSuffix(arr);
        return partial + ((k - 2) * arrSum);
    }
    
    private static long getBestNormal(int[] A) {
        long minPrefix = 0;
        long best = 0;
        long sum = 0;
        for (int v : A) {
            sum += v;
            minPrefix = Math.min(minPrefix, sum);
            best = Math.max(best, sum - minPrefix);
        }
        return best;
    }
    
    private static long[] getBestPrefix(int[] A) {
        final int N = A.length;
        long prefix = 0;
        long[] best = new long[N];
        for (int i = 0; i < N; ++i) {
            best[i] = Math.max(best[i], prefix);
            prefix += A[i];
            best[i] = Math.max(best[i], prefix);
        }
        return best;
    }
    
    private static long[] getBestSuffix(int[] A) {
        final int N = A.length;
        long suffix = 0;
        long[] best = new long[N];
        for (int i = N - 1; i >= 0; --i) {
            best[i] = Math.max(best[i], suffix);
            suffix += A[i];
            best[i] = Math.max(best[i], suffix);
        }
        return best;
    }
}
