class Solution {

	void arrayConcatenate(int arr[], int b[], int k) {
		int j = 0;
		while (k > 0) {

			for (int i = 0; i < arr.length; i++) {
				b[j++] = arr[i];
			}
			k--;
		}
	}

	int max
SubArrSum(int a[]) {
		int size = a.length;
		int max_so_far = Integer.MIN_VALUE, max_ending_here = 0;

		for (int i = 0; i < size; i++) {
			max_ending_here = max_ending_here + a[i];
			if (max_so_far < max_ending_here)
				max_so_far = max_ending_here;
			if (max_ending_here < 0)
				max_ending_here = 0;
		}
		return max_so_far;
	}

	int kConcatenationMaxSum(int arr[], int k) {
		long arrSum = 0;
		long maxSubArrSum = 0;

		int b[] = new int[(2 * arr.length)];

		arrayConcatenate(arr, b, 2);

		for (int i = 0; i < arr.length; i++)
			arrSum += arr[i];

		if (arrSum < 0)
			maxSubArrSum = maxSubArrSum(b);

		else
			maxSubArrSum = ((long) maxSubArrSum(b)) + (k - 2) * arrSum;

		if (maxSubArrSum > 0) {
			return (int) (maxSubArrSum % 1000000007);
		} else {
			return 0;
		}
	}
}
