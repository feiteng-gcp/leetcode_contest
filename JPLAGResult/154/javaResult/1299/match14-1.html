class Solution {
    
public long maxSub(int[] a) {
        int size = a.length; 
        long max_so_far = Integer.MIN_VALUE, max_ending_here = 0; 

        for (int i = 0; i < size; i++) 
        { 
            max_ending_here = max_ending_here + a[i]; 
            if (max_so_far < max_ending_here) 
                max_so_far = max_ending_here; 
            if (max_ending_here < 0) 
                max_ending_here = 0; 
        } 
        return max_so_far; 
    }
    
    public int kConcatenationMaxSum(int[] arr, int k) {
        int mod = 1000000007;
        long res = 0;
        int n = arr.length;
        long sum = 0;
        for (int a : arr) sum += a;
        if (sum > 0) {
            res = sum * k;
        }
        
        if (k == 1) {
            res = Math.max(res, maxSub(arr));
        } 
        if (k >= 2)
        
{
            int[] a = new int[n * 2];
            for (int i = 0; i < n * 2; ++i) {
                a[i] = arr[i % n];
            }

            long res2 = maxSub(a);
            
            
            if (k > 2) {
                res2 = Math.max(res2, (k - 2) * sum + res2);
            }
            res = Math.max(res, res2);
        }
       
        return (int) (res % mod);
    }
}
