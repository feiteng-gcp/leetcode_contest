class Solution {
    private static final long MOD = 1000000007; 
    int maxSum(int[] arr) {
        int maxi = 0, sum = 0;
        for(int i = 0; i < arr.length; i++) {
            sum += arr[i];
            if (sum < 0) {
                sum = 0;
            }
            maxi = Math.max(maxi, sum);
        }
        return maxi;
    }
    
    public int kConcatenationMaxSum(int[] arr, int k) {
        int sum1 = maxSum(arr);
        if(k <= 1) {
            return sum1;
        }
        int[] arr2 = new int[arr.length*2];
        for(int i = 0; i < arr.length; i++) {
            arr2[i] = arr2[i+arr.length] = arr[i];
        }
        int sum2 = maxSum(arr2);
        if(k <= 2) {
            return sum2;
        }
        int[] arr3 = new int[arr.length*3];
        for(int i = 0; i < arr.length; i++) {
            arr3[i] = arr3[i+arr.length] = arr3[i+arr.length*2] = arr[i];
        }
        int sum3 = maxSum(arr3);
        if(k <= 3){
            return sum3;
        }
        long diff = sum3-sum2;
        System.out.println(sum1);
        System.out.println(sum2);
        System.out.println(sum3);
        return (int)(Math.max((long)sum1, (long)sum3+(long)(k-3) * diff)%MOD);
    }
}
