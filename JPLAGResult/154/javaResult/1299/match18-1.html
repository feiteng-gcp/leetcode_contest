class Solution {
    public int kConcatenationMaxSum(int[] arr, int k) {
        
        if( arr == null || arr.length  == 0 )
            return 0;
        
        int mod = 1000000007;
        
        int arrnew[] = new int[arr.length * 2];
        
        for( int i = 0 ; i<arr.length ; i++ ){
            arrnew[i] = arr[i];
            arrnew[i+arr.length] = arr[i];
        }
        
        int sumOfWholeArray = 0;
        for( int i: arr ){
            sumOfWholeArray += i;
        }
        
        if( k==1 ){
            return maxSubArraySum(arr);
        }
        
        int maxSubArray = maxSubArraySum(arrnew);
            
        
        if( (sumOfWholeArray<=0) || (k==2) ){
            return maxSubArray % mod;
        }
        int answer = ( (maxSubArray % mod) + (int)mulmod(sumOfWholeArray, k-2, mod) ) % mod;
        return answer;

        
    }
    
    public static long mulmod(long a, long b, long mod){
		if (a == 0 || b == 0)
			return 0;
		if (b == 1)
			return a;
		long ans = mulmod(a, b / 2, mod);
		ans = (ans * 2) % mod;
		if (b % 2 == 1)
			ans = (a + ans) % mod;
		return ans;
	}

    
    
    static int maxSubArraySum(int a[]) 
    { 
        int size = a.length; 
        int max_so_far = Integer.MIN_VALUE, max_ending_here = 0; 
  
        for (int i = 0; i < size; i++) 
        { 
            max_ending_here = max_ending_here + a[i]; 
            if (max_so_far < max_ending_here) 
                max_so_far = max_ending_here; 
            if (max_ending_here < 0) 
                max_ending_here = 0; 
        } 
        if( max_so_far < 0 )
            max_so_far = 0;
        return max_so_far; 
    } 

  
}
