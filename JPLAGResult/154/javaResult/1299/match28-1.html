class Solution {
    int K = 1000000007;
    public int kConcatenationMaxSum(int[] arr, int k) {
        long sum =0;
        for( int i : arr) sum = (sum + i)%K;
        long res = findSum(arr) %K;
        
        if(sum < 0) return (int)res
;
        return (int)((res + ((((k-2)%K) * (sum%K)%K)))%K);
    }
    
    public int findSum(int[] arr) {
        long temp =0;
        long max =0;
        int n =arr.length;
        int i =0;
        while(i< 2*n) {
            temp = (Math.max(temp + arr[i%n], Math.max(0,arr[i%n])))%K;
            i++;
            max= Math.max(max,temp);
        }
        return (int)(max%K);
    }
}
