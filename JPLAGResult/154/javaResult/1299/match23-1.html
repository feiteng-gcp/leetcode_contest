class Solution {
    public int MAX = (int)Math.pow(10, 9) + 7;
    public int kConcatenationMaxSum(int[] arr, int k) {
        int max1 = maxSubSum(arr);
        if(k == 1)  return max1;
        
        
int[] arr2 = new int[arr.length * 2];
        for(int i = 0; i < arr.length; i++) {
            arr2[i] = arr[i];
            arr2[i + arr.length] = arr[i];
        }
        int max2 = maxSubSum(arr2);
        if(k == 2)  return max2;
        if(max2 < max1) return max1;
        
        int[] arr3 = new int[arr.length * 3];
        for(int i = 0; i < arr.length; i++) {
            arr3[i] = arr[i];
            arr3[i + arr.length] = arr[i];
            arr3[i + 2 * arr.length] = arr[i];
        }
        int max3 = maxSubSum(arr3);
        if(k == 3)  return max3;
        
        int d = max3 - max2;
        int res = max2;
        for(int i = 0; i < k-2; i++) {
            res = res % MAX;
            res += d % MAX;
        }
        System.out.println(max1);
        System.out.println(max2);
        System.out.println(max3);
        return res;
    }
    private int maxSubSum(int[] arr) {
        
int localMax = 0, globalMax = 0;
        for(int i = 0; i < arr.length; i++) {
            localMax = Math.max(localMax + arr[i], arr[i]);
            globalMax = Math.max(localMax, globalMax);
        }
        return globalMax; 
    }
}
