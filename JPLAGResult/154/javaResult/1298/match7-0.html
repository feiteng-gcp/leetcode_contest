class Solution {
    public String reverseParentheses(String S) {
        Map<Integer, StringBuilder> count = new HashMap<>();
        Map<Integer, Integer> pos = new HashMap<>();
        char [] sarr = S.toCharArray();
        int n = sarr.length;
        int bal = 0;
        Stack<Character> s = new Stack<>();
        boolean flag = true;
        String start = "";
        String end = "";
        for(int i = 0; i < n; i++){
            if(sarr[i] == ')'){
                StringBuilder temp = new StringBuilder();
                while(s.peek() != '('){
                    temp.append(s.pop());
                }
                s.pop();
                for(int j = 0; j < temp.length(); j++){
                    s.push(temp.charAt(j));
                }
            }else{
                if(sarr[i] == '('){
                    bal++;
                    flag = false;
                }
                if(bal != 0){
                    s.push(sarr[i]);            
                }else if(bal == 0 && flag){
                    start += sarr[i];
                }else{
                    end += sarr[i];
                }
            }
        }
        StringBuilder res = new StringBuilder();
        res.append(start);
        StringBuilder temp = new StringBuilder();
        while(s.size() > 0){
            temp.append(s.pop());
        }
        res.append(temp.reverse().toString());
        res.append(end);
        return res.toString();
    }
}
