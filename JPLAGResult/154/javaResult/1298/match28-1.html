class Solution {

    static String trim(char[] arr) {
        
StringBuilder sb = new StringBuilder();
        for (int i=0;i<arr.length;i++) {
            if (arr[i] != '(' && arr[i] != ')') {
                sb.append(arr[i]);
            }
        }
        return sb.toString();
    }
    
    public String reverseParentheses(String s) {
        char[] arr = s.toCharArray();
        Stack<Integer> stack = new Stack<>();
        for (int i=0;i<s.length();i++) {
            if (s.charAt(i) == '(') {
                stack.push(i+1);
            } else if(s.charAt(i) == ')') {
                int begIdx = stack.pop();
                //reverse begIdx to i-1, len = i-begIdx,
                int endIdx = i-1;
                for (;begIdx < endIdx; begIdx++, endIdx--) {
                    char tmp = arr[begIdx];
                    arr[begIdx] = arr[endIdx];
                    arr[endIdx] = tmp;
                }
            }
        }
        return trim(arr);
    }
}
