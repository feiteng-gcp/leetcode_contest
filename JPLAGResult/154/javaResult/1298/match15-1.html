class Solution {
    public String reverseParentheses(String s) {
        Deque<Character> stack = new ArrayDeque<>();
        Deque<Character> tmp= new ArrayDeque<>();
        for(int i=0; i<s.length();i++){
            char c = s.charAt(i);
            if(c==')'){
                while(!stack.isEmpty()){
                    char top = stack.pop();
                    if(top != '(') {
                        tmp.offer(top);
                    }else{
                        break;
                    }
                }
                while(!tmp.isEmpty()){
                    stack.push(
tmp.poll());
                }
            }else{
                stack.push(c);
            }
        }
        StringBuilder sb = new StringBuilder();
        while(!stack.isEmpty()){
            sb.append(stack.pop());
        }
        return sb.reverse().toString();
    }
}
