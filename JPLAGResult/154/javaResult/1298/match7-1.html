class Solution {
    public String reverseParentheses(String s) {
        ArrayDeque<Character> queue = new ArrayDeque<Character>();        
        for (int i = 0; i < s.length(); ++i) {
            char ch = s.charAt(i);
            if (ch == '(') {
                queue.offerLast(ch);
            } else if (s.charAt(i) == ')') 
{
                ArrayDeque<Character> tmp = new ArrayDeque<Character>();
                while (queue.peekLast() != '(') {
                    tmp.offer(queue.pollLast());
                }
                queue.pollLast(); // remove the '('
                while (!tmp.isEmpty()) queue.offerLast(tmp.pollFirst());
            } else {
                queue.offerLast(ch);
            }
        }
        
        
StringBuffer sb = new StringBuffer();
        while(!queue.isEmpty()) {
            sb.append(queue.pollFirst());
        }
        return sb.toString();
    }
}
