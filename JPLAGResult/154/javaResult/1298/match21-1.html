class Solution {
    public String reverseParentheses(String s) {
        if(s == null || s.length() == 0) {
            return s;
        }
        
        
Stack<String> st = new Stack<>();
        for(char c: s.toCharArray()) {
            if(c == '(') {
                st.push(c + "");
            }
            else if(c == ')') {
                String curr = "";
                while(!st.peek().equals("(")) {
                    curr = st.pop() + curr;
                }
                st.pop();
                StringBuilder sb = new StringBuilder(curr);
                st.push(
sb.reverse().toString());
            }
            else {
                st.push(c + "");
            }
        }
        
        String ans = "";
        while(!st.isEmpty()) {
            ans = st.pop() + ans;
        }
        
        return ans;
    }
}
