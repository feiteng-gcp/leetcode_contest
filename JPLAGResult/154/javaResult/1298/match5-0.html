class Solution {

    class Node {
        String str;
        int idx;
    }
    public String reverseParentheses(String s) {
        Node res = solve(s, 0);
        return res.str;
    }

    Node solve(String s, int idx) {
        StringBuilder sb = new StringBuilder();
        for (int i = idx; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (ch == '(') {
                Node res = solve(s, i+1);
                sb.append(res.str);
                i = res.idx;
            } else if (ch == ')') {
                Node res = new Node();
                sb.reverse();
                res.str = sb.toString();
                res.idx = i;
                return res;
            } else {
                sb.append(ch);
            }
        }
        Node res = new Node();
        res.str = sb.toString();
        res.idx = s.length();
        return res;
    }
}
