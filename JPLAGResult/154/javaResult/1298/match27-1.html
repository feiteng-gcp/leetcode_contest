class Solution {
    public String reverseParentheses(String s) {
        
        if (s == null) return s;
        
        Stack<Character> stack = new Stack<>();
        Queue<Character> innerQueue = new LinkedList<>();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c != ')') {
                stack.push(c);
            
} else {
                while (stack.peek() != '(') {
                    innerQueue.offer(stack.pop());
                }
                stack.pop();
                while (!innerQueue.isEmpty()) {
                    stack.push(innerQueue.poll());
                }
            }
        }
        
StringBuilder sb = new StringBuilder();
        while (!stack.isEmpty()) {
            sb.insert(0, stack.pop());
        }
        return sb.toString();
        
    }
}
