class Solution {
    public String reverseParentheses(String s) {
        Stack<Stack<Character>> stack = new Stack<Stack<Character>>();
        stack.push(new Stack<Character>());
        for(int i=0;i<s.length();i++) {
            char c = s.charAt(i);
            if(c=='(') {
                stack.push(new Stack<Character>());
                //System.out.println("new ( found");
            } else if(c==')') {
                //System.out.println(") found");
                Stack<Character> chars = stack.pop();
                
                List<Character> rev = new LinkedList<Character>();
                while(!chars.isEmpty()) {
                    rev.add(chars.pop());
                }
                Stack<Character> top = stack.peek();
                for(char ch:rev) {
                    //System.out.println("pushing " + ch);
                    top.push(ch);
                }
            } else {
                stack.peek().push(c);
            }
        }
        Stack<Character> chars = stack.pop();
        StringBuilder sb = new StringBuilder();
        for(char ch:chars) {
            sb.append(ch);
        }
        return sb.toString();
    }
}
