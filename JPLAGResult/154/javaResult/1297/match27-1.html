class Solution {
    public int maxNumberOfBalloons(String t) 
    {
        int ar[]=new int[26];
        for(int x=0;x<t.length();x++) 
        {
            ar[t.charAt(x)-97]++;
        
}
        int min=Integer.MAX_VALUE;
     
        min=Math.min(min,ar['b'-97]);
        min=Math.min(min,ar['a'-97]);
        min=Math.min(min,ar['n'-97]);
        min=Math.min(min,ar['l'-97]/2);
        min=Math.min(min,ar['o'-97]/2);
        return Math.max(min,0);
    }
}
