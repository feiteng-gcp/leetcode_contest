class Solution {
    public int maxNumberOfBalloons(String text) {
        int[] cnt = new int[26];
        for(char c : text.toCharArray()){
            cnt[c-'a']++;
        
}
        int res = cnt['b'-'a'];
        res = Math.min(res, cnt[0]);
        res = Math.min(res, cnt['l'-'a']/2);
        res = Math.min(res, cnt['o'-'a']/2);
        res = Math.min(res, cnt['n'-'a']);
        return res;
    }
}
