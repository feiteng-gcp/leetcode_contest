class Solution {
    public int maxNumberOfBalloons(String text) {
        int[] map = new int[26];
        for(char t : text.toCharArray()) map[t-'a']++;
        
        String str = "balloon";
        
        int res = map['l'-'a']/2;
        
res = Math.min(res, map[0]);
        res = Math.min(res, map[1]);
        res = Math.min(res, map['o'-'a']/2);
        res = Math.min(res, map['n'-'a']);
        return res;
    }
}
