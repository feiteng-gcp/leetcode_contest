class Solution {
    public int maxNumberOfBalloons(String text) {
        int[] freqs = new int[26];
        for(int i = 0; i < text.length(); i++) {
            freqs[text.charAt(i) - 'a']++;
        }
        
        // b's (lol wtf)
        int max = freqs[1];
        
        max = Math.min(max, freqs[0]);
        max = Math.min(max, freqs[11]/2);
        max = Math.min(max, freqs[14]/2);
        max = Math.min(max, freqs[13]);
        return max;
    }
}
