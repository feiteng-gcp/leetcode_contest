class Solution:
    def advantageCount(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        sA = sorted(A)
        sB = sorted(B)
        result = []
        tail = []
        length = len(A)
        iA = 0
        iB = 0
        while iA < length:
            if sA
[iA] > sB[iB]:
                result.append(sA[iA])
                iA += 1
                iB += 1
            else:
                tail.append(sA[iA])
                iA += 1
                
        result.extend(tail)
        #print(A)
        #print(B)
        #print(result)
        
        index = [i[0] for i in sorted(enumerate(B), key=lambda x: x[1])]
        
        print(index)
        
        r = [None] * length
        for i, v in enumerate(result):
            r_index = index[i]
            r[r_index] = v         
        return r
