class Solution:
    def advantageCount(self, A, B):

        origB = B

        i2 = 0

while i2 < len(A):
            if A[i1] <= B[i2]:
                mapped[B[i2]].append(A[e])
                e -= 1
                i2 += 1
            else:
                mapped[B[i2]].append(A[i1])

        return [mapped[_].pop(-1) for _ in origB]
