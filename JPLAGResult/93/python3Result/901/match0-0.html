class Solution:
    def advantageCount(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        import collections
        A.sort()
        B_copy = B.copy()
        B_copy.sort()
        i = 0
        j = 0
        dict_A_B = collections.defaultdict(list)
        useless_A = []
        while i < len(A):
            if A[i] > B_copy[j]:
                dict_A_B[B_copy[j]].append(A[i])
                j += 1
                i += 1
            else:
                useless_A.append(A[i])
                i += 1
        return_A = []
        for i in B:
            if i in dict_A_B:
                return_A.append(dict_A_B[i].pop())
                if len(dict_A_B[i]) == 0:
                    del dict_A_B[i]
            else:
                return_A.append(useless_A.pop())
        return return_A
