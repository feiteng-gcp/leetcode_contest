class Solution:
    def binaryGap(self, N):
        """
        :type N: int
        :rtype: int
        """
        b = '{0:b}'.format(N)
        l = -1
        rMax = 0
        for i,ch in enumerate(b):
            if ch == '1':
                if l != -1:
                    rMax = max(rMax, i - l)
                l = i

        return rMax
