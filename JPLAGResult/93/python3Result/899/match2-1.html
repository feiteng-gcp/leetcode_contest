class Solution:
    def binaryGap(self, N):
        """
        :type N: int
        :rtype: int
        """
        previous_index = -1
        max_d = 0
        current_index = 0
        while N > 0:
            k = N % 2
            
if k == 1:
                if previous_index >= 0:
                    max_d = max(max_d, current_index - previous_index)
                previous_index = current_index
            current_index += 1
            N //= 2
        return max_d
