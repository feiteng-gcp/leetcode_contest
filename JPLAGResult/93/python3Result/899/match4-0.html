class Solution:
    def binaryGap(self, N):
        """
        :type N: int
        :rtype: int
        """
        tmp, i = [], 0
        while N != 0:
            if N & 1 == 1:
                tmp.append(i)
            i += 1
            N >>= 1
        res = 0
        for i in range(1, len(tmp)):
            res = max(res, tmp[i] - tmp[i-1])
        return res
