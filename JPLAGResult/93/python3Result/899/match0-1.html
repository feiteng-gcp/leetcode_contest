class Solution:
    def binaryGap(self, N):
        """
        :type N: int
        :rtype: int
        """
        pre 
= -1
        ans = 0
        for i in range(32):
            if (N >> i) & 1 == 1:
                if pre != -1:
                    ans = max(ans, i - pre)
                pre = i
        return ans
