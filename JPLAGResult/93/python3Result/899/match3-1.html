class Solution:
    def binaryGap(self, N):
        """
        :type N: int
        :rtype: int
        """
        binary = ''
        while N > 0:
            binary = str(N & 1) + binary
            N >>= 1
            
        binary = binary[::-1]
        prev, curr 
= binary.index('1'), 0
        for i in range(len(binary)):
            if binary[i] == '1':
                curr = max(i - prev, curr)
                prev = i
        
        return curr
