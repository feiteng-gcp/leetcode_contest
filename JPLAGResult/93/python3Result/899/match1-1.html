class Solution:
    def binaryGap(self, N):
        """
        :type N: int
        :rtype: int
        """
        binary_repn = self.dec_to_bin(N)
        dist = 0
        
        cur_idx = -1
        for idx, digit in enumerate(binary_repn):
            if digit == 1:
                if cur_idx == -1:
                    cur_idx = idx
                else:
                    old_idx = cur_idx
                    cur_idx = idx
                    if cur_idx - old_idx > dist:
                        dist = cur_idx - old_idx
        
        return dist
    
    def dec_to_bin(self, x):
        l = list(bin(x)[2:])
        return [int(digit) for digit in l]
