class Solution:
    def binaryGap(self, N):
        """
        :type N: int
        :rtype: int
        """
        pos = 0
        pre_pos = None
        max_gap = 0
        while N > 0:
            if N % 2 == 1:
                if pre_pos is not None:
                    max_gap = max(max_gap, pos - pre_pos)
                pre_pos = pos
            pos += 1
            N = N >> 1
        return max_gap
