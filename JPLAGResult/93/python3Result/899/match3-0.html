class Solution:
    def binaryGap(self, N):
        """
        :type N: int
        :rtype: int
        """
        binary = bin(N)[2:]
        
        max_length = 0
        last_i = binary.find('1')
        
        for i in range(last_i+1, len(binary)):
            if(binary[i] == '1'):
                max_length = max(max_length, i-last_i)
                last_i = i
            
        return max_length
