class Solution {
public:
    bool reorderedPowerOf2(int N) {
        vector<int> V;
        int M=N;
        while(M){
            V.push_back(M%10);
            M/=10;
        }
        sort(V.begin(),V.end());
        do{
            if(V[0]==0)continue;
            long long tmp=0;
            for(auto v:V)tmp=tmp*10+v;
            if((tmp&(tmp-1))==0)return true;
        }while(next_permutation(V.begin(),V.end()));
        return false;
    }
};
