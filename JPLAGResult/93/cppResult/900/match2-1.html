class Solution {
public:
    bool reorderedPowerOf2(int n) {
        vector<vector<int> > a;
        for (int i = 0; i <= 31; i++) {
            int num = 1 << i;

            vector<int> st;
            while (num) {
                st.push_back(num % 10);
                num /= 10;
            }
            sort(st.begin(), st.end());
            a.push_back(st);
        }
        vector<int> me;
        while (n) {
            me.push_back(n % 10);
            n /= 10;
        }
        sort(me.begin(), me.end());
        // for (auto it : me) {
        //     cout << it << endl;
        // }
        for (int i = 0; i < a.size(); i++) {
            if (a[i] == me) {
                for (auto it : a[i]) {
                    cout << it << endl;
                }
                
                return true;
            }
        }
        return false;
    }
};
