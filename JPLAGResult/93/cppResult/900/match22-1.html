class Solution {
public:
    bool reorderedPowerOf2(int N) {
        //N = 635824465;
        int Max = 1000000000;
        int a[40][15] = {0};
        int n = 0;
        int d = 1;
        while(d <= Max){
            //cout<<d<<endl;
            int k = d;
            int l = 0;
            while(k != 0){
                l++;
                a[n][l] = k % 10;
                k = k/10;
            }
            a[n][0] = l;
            n++;
            d = d*2;
        }
        
        int b[15] = {0};
        int l = 0;
        while(N != 0){
            l++;
            b[l] = N%10;
            N = N/10;
        }
        b[0] = l;
        for(int i = 0; i < n; i++){
            if(a[i][0] == b[0]){
                bool c[15] = {0};
                bool succ = true;
                int k;
                for(int j = 1; j <= b[0]; j++){
                    for(k = 1; k <= b[0]; k++){
                        if(!c[k] && b[j] == a[i][k]){
                            c[k] = true;
                            break;
                        }
                    }

                    if(k > b[0]){
                        succ = false;
                        break;
                    }
                }
                if(succ)
                    return true;
            }
        }
        return false;
    }
};
