class Solution {
public:
    bool reorderedPowerOf2(int N) {
       /* int chart[] = {1,2,4,8,16,32,64,128,256,512,1024,2048,4096,8192,16384,32768,65536,131072,262144,524288,1048576,2097152,4194304,8388608,16777216,33554432,67108864,134217728,268435456,536870912,1073741824,2147483648}
    }*/
    vector<int> s(10,0);
        int j = 0;
        while(N!=0){
            int r = N%10;
            N = N/10;
            s[r]++;
            j++;
        }
    vector<vector<int>> v(32,vector<int>(10,0));
    for(int i = 0;i<=31;i++){
        int t= pow(2,i);
        int k = 0;
        while(t!=0){
            int r = t%10;
            t = t/10;
            v[i][r]++;
            k++;
        }

        if(k>j)
            return false;
        if(j==k){
            int flag = 0;
            for(int x = 0;x<10;x++){
                if(v[i][x]!=s[x]){
                    flag = 1;
                    break;
                }
            }
            if(flag==0)
                return true;
        }
    }
    return false;
    }
};
