class Solution {
public:
    bool reorderedPowerOf2(int N) {
        
        unordered_set<string> dict;
        
        for (int i = 0; i < 30; i++)
        {
            int k = 1 << i;
            
            dict.insert(int2str(k));
        }
        
        string s = int2str(N);
        
        return dict.count(s) > 0;
    }
    
    string int2str(int k)
    {
        vector<int> v;
        
        int zero = 0;
        
        while (k > 0)
        {
            int a = k % 10;
            
            if (a == 0)
            {
                zero++;
            }
            else
            {
                v.push_back(a);
            }
            
            k = k / 10;
        }
        
        sort(v.begin(), v.end());
        
        string s;
        
        for (auto n : v)
        {
            char ch = '0' + n;
            
            s += ch;
        }
        
        for (int i = 0; i < zero; i++)
        {
            s += '0';
        }
        
        //cout << s << endl;
        
        return s;
    }
};
