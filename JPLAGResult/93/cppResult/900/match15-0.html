class Solution {
public:
    bool reorderedPowerOf2(int N) {
        vector<vector<int>> v;
        int k = 1;
        while (k <= 1000000000) {
            v.push_back(intToVec(k));
            k *= 2;
        }
        
        auto vn = intToVec(N);
        
        for (auto& dd : v) {
            if (dd.size() != vn.size()) {
                continue;
            }
            bool b = true;
            for (int j = 0; j < vn.size(); ++j) {
                if (dd[j] != vn[j]) {
                    b = false;
                    break;
                }
            }
            if (b) {
                return true;
            }
        }
        return false;
    }
    
    vector<int> intToVec(int n) {
        vector<int> v;
        while (n > 0) {
            v.push_back(n % 10);
            n = n / 10;
        }
        sort(v.begin(), v.end());
        return v;
    }
};
