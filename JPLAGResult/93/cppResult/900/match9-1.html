class Solution {
public:
    bool reorderedPowerOf2(int N)
    {
        unordered_set<string> us;
        for(int i=0;i<32;i++)
        {
            int n=(1<<i);
            string ns=intTostr(n);
            sort(ns.begin(),ns.end());
            us.insert(ns);
        }
        
        string NS=intTostr(N);
        sort(NS.begin(),NS.end());
        if(us.find(NS)!=us.end()) return true;
        else return false;
    }
    
    stringstream stream;
    string intTostr(int n)
    {
        stream.str("");
        stream.clear();
        string ret;
        stream<<n;
        stream>>ret;
        return ret;
    }
};
