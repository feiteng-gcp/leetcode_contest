class Solution {
public:
    vector<int> getFinger(long long n)
    {
        vector<int> finger(10, 0);
        while (n > 0)
        {
            finger[n % 10]++;
            n /= 10;
        }
        
        return finger;
    }
    
    bool reorderedPowerOf2(int N) {
        long long d = 1;
        long long n = N;
        while (d < n * 10)
        {
            auto f1 = getFinger(d);

            auto f2 = getFinger(n);
            
            bool match = true;
            for (int i = 0; i < 10; i++)
            {
                if (f1[i] != f2[i])
                {
                    match = false;
                    break;
                }
            }
            
            if (match) return true;

            d <<= 1;
        }
        
        return false;
    }
};
