class Solution {
public:
    bool reorderedPowerOf2(int N) {
        if (N == 1)
            return true;

        vector<int> count(10, 0);
        int temp = N;
        int Nlen = 0;
        while (temp != 0)
        {
            count[temp % 10] += 1;
            temp /= 10;
            Nlen += 1;
        }
        long cur = 1;
        int curlen = 0;
        long curtemp = cur;
        while (curtemp != 0)
        {
            curtemp /= 10;
            curlen += 1;
        }
        
        while (curlen <= Nlen)
        {
            cur *= 2;
            curtemp = cur;
            curlen = 0;
            vector<int> curcount(10, 0);
            while (curtemp != 0)
            {
                curcount[curtemp % 10] += 1;
                curtemp /= 10;
                curlen += 1;
            }
            bool check = true;
            for(int i = 0; i < 10; i++)
                if (curcount[i] != count[i])
                {
                    check = false;
                    break;
                }
            if (check)
                return true;
        }
        
        return false;
        
    }
};
