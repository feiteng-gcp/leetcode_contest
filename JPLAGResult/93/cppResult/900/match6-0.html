class Solution {
public:
    bool reorderedPowerOf2(int N) {
        if(N == 1E9) return false;
        vector<int> dig = preprocess(N);
        return isPermutation(dig);
    }
    
    bool isPermutation(vector<int>& dig){
        long long i = 1;
        vector<vector<int>> pow2(9, vector<int>());
        while(i <= 1E9){
            int n = log10(i);
            pow2[n].push_back(i);
            i <<= 1;
        }
        // for(auto p : pow2){
        //     for(int i : p) cout<<i<<" ";
        //     cout<<endl;
        // }
        int nn = dig[10];
        for(int ii : pow2[nn - 1]){
            vector<int> p2 = preprocess(ii);
            bool flg = true;
            for(int i = 0; i < 11; i++){
                if(p2[i] != dig[i]){
                    flg = false;
                    break;
                }
            }
            if(flg) return true;
        }
        return false;
    }
    
    // size 11, last digit for length of N
    vector<int> preprocess(int N){
        vector<int> dig(11, 0);
        while(N > 0){
            dig[N % 10]++;
            dig[10]++;
            N /= 10;
        }
        // for(int i : dig) cout<<i<<" ";
        // cout<<endl;
        return dig;
    }
};
