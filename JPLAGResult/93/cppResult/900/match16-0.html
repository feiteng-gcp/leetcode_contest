class Solution {
public:
    bool reorderedPowerOf2(int N) {
        vector<vector<int>> vt;
        for(int i = 0; i < 31; ++i) {
            int x = (1 << i);
            string s = to_string(x);
            vector<int> vs(10,0);
            for(auto c : s) {
                vs[c - '0']++;
            }
            vt.push_back(vs);
        }
        string st = to_string(N);
        vector<int> rs(10,0);
        for(auto c : st) {
            rs[c-'0']++;
        }
        for(auto v : vt) {
            if(rs == v) return true;
        }
        return false;
    }
};
