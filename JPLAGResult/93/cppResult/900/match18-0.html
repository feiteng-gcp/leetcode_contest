class Solution {
public:
    bool reorderedPowerOf2(int N) {
        vector<int> nums(10,0);
        int s=0;
        int t=N;
        while(t>0){
            nums[t%10]++;
            t=t/10;
            s++;
        }
       
        long p=1;
        while(true){
            vector<int> tmp(10,0);
            int f=0;
            long tt = p;
            while(tt>0){
                tmp[tt%10]++;
                tt=tt/10;
                f++;
            }
            if(tmp==nums) return true;
            if(f>s) break;
            p*=2;
        }
        return false;
    }
};
