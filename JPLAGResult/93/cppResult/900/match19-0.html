int check(long long int x) {
    int ans=0;
    for(int i=0;i<32;++i) {
        if(x&(1<<i)) ans++;
    }
    if(ans==1) return 1;
    return 0;
}
class Solution {
public:
    bool reorderedPowerOf2(int n) {
        string s;     int x;
        while(n!=0) {
            x=n%10;  s.push_back(x+'0');
            n=n/10;
        }
        sort(s.begin(),s.end());
        do{
            if(s[0]!='0') {
                long long int t=atoi(s.c_str()); 
                if(check(t)) return true;
            }
        }while(next_permutation(s.begin(),s.end()));
        return false;
    }
};
