class Solution {
public:
    vector<int> vec;
    vector<bool> vis;
    bool ans;
    void dfs(int m, int num)
    {
        if(ans)return ;
        if(m==vec.size())

        {
            bitset<32> bs(num);
            int cnt = 0;
            for(int i = 31; i >= 0; i--)
            {
                if(bs[i])cnt++;
            }
            if(cnt==1)
            {
                ans = true;
            }
            return ;
        }
        for(int i = 0; i < vec.size(); i++)
        {
            if(vis[i])continue;
            if(m==0&&vec[i]==0)continue;
            vis[i] = true;
            dfs(m+1, num*10+vec[i]);
            vis[i] = false;
        }
    }
    bool reorderedPowerOf2(int N) {
        int n = N;
        while(n)
        {
            vec.push_back(n%10);
            n = n/10;
        }
        vis.resize(vec.size(), false);
        ans = false;
        dfs(0, 0);
        return ans;
    }
};
