class Solution {
public:
    bool reorderedPowerOf2(int N) {
       vector<vector<int>> model;
       for(int i = 0; i < 32; i++){
            int pow2 = 1;
            for(int j = 0; j < i; j++){
                pow2 = pow2*2;
            }
            vector<int> local;
            while(pow2 != 0){
               local.push_back(pow2%10);
               pow2 = pow2/10;
            }
            sort(local.begin(),local.end());
            model.push_back(local);
        }
        vector<int> aim;
        while(N != 0){
            aim.push_back(N%10);
            N = N/10;
        }
        sort(aim.begin(),aim.end());
        for(int i = 0;i < model.size();i++){
            if(aim.size() == model[i].size()){
               int counter;
               for(counter = 0; counter < aim.size(); counter++){
                   if(aim[counter] != model[i][counter]){
                      break;
                   }
               }
               if(counter == aim.size()){
                  return true;
               }
            }
        }
        return false;
    }
};
