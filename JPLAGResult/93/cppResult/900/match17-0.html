static int MV = 1000000000;

class Solution {
public:
    bool reorderedPowerOf2(int N) {
        vector<int> v;
        int base = 1;
        while(base <= MV){
            v.push_back(base);
            base *= 2;
        }
        
        vector<vector<int>> d;
        for(auto x: v){
            d.push_back(count(x));
        }
        
        vector<int> t = count(N);

        for(auto& s:d){
            if(check(t, s))
                return true;
        }
        return false;
    }
    
    bool check(vector<int>& t, vector<int>& s){
        for(int i = 0; i < s.size(); i++){
            if(t[i] != s[i])
                return false;
        }
        return true;
    }
    
    vector<int> count(int value){
        string s = to_string(value);
        vector<int> v(10, 0);
        for(int i = 0; i < s.size(); i++){
            v[s[i] - '0']++;
        }
        return v;
    }
};
