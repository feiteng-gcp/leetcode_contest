class Solution {
public:
    int getn(long long n){
        int a=0;
        while(n){
            n=n/10;
            a++;
        }
        return a;
    }
    bool reorderedPowerOf2(int N) {
        long long n=1;
        int s=getn(N);
        int s1=getn(n);
        bool asc=false;
        while(s>s1){
            n*=2;
            s1=getn(n);
        }
        while(s==s1){
            int a[10];
            int b[10];
            for(int i=0;i<10;i++){
                a[i]=0;
                b[i]=0;
            }
            int x=n;
            for(int i=0;i<s;i++){
                int y=x%10;
                x=x/10;
                a[y]++;
            }
            x=N;
            for(int i=0;i<s;i++){
                int y=x%10;
                x=x/10;
                b[y]++;
            }
            asc=true;
            for(int i=0;i<10;i++){
                if(a[i]!=b[i]){
                    asc=false;
                    break;
                }
            }
            if(asc){
                break;
            }
            n*=2;
            s1=getn(n);
        }
        return asc;
    }
};
