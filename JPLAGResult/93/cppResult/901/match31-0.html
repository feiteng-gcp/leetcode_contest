class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        sort(A.begin(),A.end());
        vector<int> v(B.size());
        for(int i = 0;i<B.size();i++)
            v[i] = i;
        qs(B,v,0,v.size()-1);
        vector<int> s(B.size());
        int l = 0, r= A.size()-1;
        for(int i = B.size()-1;i>=0;i--){
            if(A[r]>B[i]){
                s[v[i]] = A[r];
                r--;
            }
            else{
                s[v[i]] = A[l];
                l++;
            }
        }
        return s;
    }
    void qs(vector<int>& B,vector<int>& v,int b,int f){
        if(b<f){
            int p = partition(B,v,b,f);
            qs(B,v,b,p-1);
            qs(B,v,p+1,f);
        }
    }
    int partition(vector<int>& B,vector<int>& v,int b,int f){
        int pivot = B[f];
        int j = b-1;
        for(int i = b;i<f;i++){
            if(B[i]<pivot){
                ++j;
                swap(B[i],B[j]);
                swap(v[i],v[j]);
            }
        }
        ++j;
        swap(B[j],B[f]);
        swap(v[j],v[f]);
        return j;
    }
};
