class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        int n = A.size();
        multiset<int> s_A(A.begin(), A.end());
        vector<int> ret(n, -1);
        for (int i = 0; i < n; ++i) {
            auto it = s_A.upper_bound(B[i]);
            if (it != s_A.end()) {
                ret[i] = *it;
                s_A.erase(it);
            }
        }
        for (int i = 0; i < n; ++i) {
            if (ret[i] == -1) {
                ret[i] = *s_A.begin();
                s_A.erase(s_A.begin());
            }
        }
        return ret;
    }
};
