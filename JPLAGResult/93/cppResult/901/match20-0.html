class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        
        int n = B.size();
        
        unordered_map<int, vector<int>> dict;
        
        for (int i = 0; i < n; i++)
        {
            dict[B[i]].push_back(i);
        }
        
        sort(A.begin(), A.end());
        sort(B.begin(), B.end());
        
        int low = 0, high = n - 1;
        
        vector<int> ret(n);
        
        int index = 0;
        
        for (auto k : A)
        {
            if (k > B[low])
            {
                index = dict[B[low]].back();
                
                dict[B[low]].pop_back();
                
                ret[index] = k;
                
                low++;
            }
            else
            {
                index = dict[B[high]].back();
                
                dict[B[high]].pop_back();
                
                ret[index] = k;
                
                high--;
            }
        }
        
        return ret;
    }
};
