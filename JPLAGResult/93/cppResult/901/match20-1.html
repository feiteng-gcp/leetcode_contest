bool myfunction(tuple<int,int> a, tuple<int,int> b){
    return get<0>(a)<get<0>(b);
}
class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {     
        int A_len = A.size();
        vector<int> A_ans(A_len, 0);
        int B_len = B.size();
        vector<tuple<int,int>> B_index;
        for(int i=0;i<B_len;i++)
            B_index.push_back(make_tuple(B[i], i));
        sort(A.begin(), A.end());
        sort(B_index.begin(), B_index.end(), myfunction);  
        int i=0;
        int back = B_len-1;
        int j=0;

        while(i<A_len && j<B_len){
            if(A[i]>get<0>(B_index[j])){
                A_ans[get<1>(B_index[j])] = A[i];
                i++;
                j++;
            }
            else{
                A_ans[get<1>(B_index[back])] = A[i];
                i++;
                back--;
            }
        }
        return A_ans;
    }
};
