class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        vector<int> result;
        sort(A.begin(), A.end());
        
        for (int i = 0; i < B.size(); i++) {
            auto it = lower_bound(A.begin(), A.end(), B[i], [] (const auto& lhs, const auto& rhs) {
                return lhs <= rhs;
            });
            if (it == A.end())
                it = A.begin();
            result.push_back(*it);
            A.erase(it);
        }
        
        return result;
    }
};
