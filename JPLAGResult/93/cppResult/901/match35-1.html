class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        vector<pair<int,int>> AI,BI;
        vector<int> res(A.size());
        for(int i=0;i<A.size();i++){
            AI.push_back({A[i],i});
            BI.push_back({B[i],i});
        }
        sort(AI.begin(),AI.end());
        sort(BI.begin(),BI.end());
        int f=0;

        int l=A.size()-1;
        for(int i=0;i<A.size();i++){
            if(AI[i].first>BI[f].first){
                res[BI[f].second]=AI[i].first;
                f++;
            }
            else{
                res[BI[l].second]=AI[i].first;
                l--;
            }
        }
        return res;
    }
};
