class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        sort(A.begin(), A.end());
        int len = A.size();

        vector<int> in(len);
        for (int i = 0; i < len; i++) in[i] = i;

        sort(in.begin(), in.end(), [&](int a, int b){ return B[a] < B[b]; });

        vector<int> r(len);
        
        int p = len;
        int ai = 0;
        for (int i = 0; i < p; i++)
        {
            int d = B[in[i]];
            while (ai < len && A[ai] <= d)
            {
                r[in[--p]] = A[ai];
                ai++;
            }
            
            if (ai == len) break;
            
            r[in[i]] = A[ai++];
        }
        
        return r;
    }
};
