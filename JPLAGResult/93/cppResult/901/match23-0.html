class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        vector<int> ans(B.size());
        vector<pair<int, int>> BS;
        multiset<int> AS;
        for (int i = 0; i < B.size(); ++i) {
            BS.push_back(make_pair(B[i], i));
            AS.insert(A[i]);
        }
        sort(BS.begin(), BS.end());
        for (int i = 0; i < BS.size(); ++i) {
            int b = BS[i].first;
            int bidx = BS[i].second;
            
            auto it = AS.upper_bound(b);
            if (it == AS.end()) {
                ans[bidx] = *AS.begin();
                AS.erase(AS.begin());
            } else {
                ans[bidx] = *it;
                AS.erase(it);
            }
        }
        
        return ans;
    }
};
