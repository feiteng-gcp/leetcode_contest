class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
	sort(A.begin(), A.end());
	int n = A.size();
	vector<int> d(n, 0);
	for (int i = 0; i < n; ++i) d[i] = i;
	sort(d.begin(), d.end(), [B](int i, int j) {return B[i] < B[j]; });
	vector<int> res(n, -1);
	vector<bool> vis(n, false);
	int i = 0, j = 0;
	while (i < n && j < n) {
		if (A[j] > B[d[i]]) {
			res[d[i]] = A[j];
			vis[j] = true;
			j++;
			i++;
		}
		else {
			j++;
		}
	}
	for (int i = 0, j = 0; i < n; ++i) {
		if (res[i] == -1) {
			while (j < n) {
				if (!vis[j]) {
					res[i] = A[j];
					vis[j] = true;
					break;
				}
				j++;
			}
		}
	}
	return res;
}
};
