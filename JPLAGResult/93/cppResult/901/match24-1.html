class Solution {
public:
	vector<int> advantageCount(vector<int>& A, vector<int>& B) {
		int N = A.size();
		vector<pair<int, int>> U, V;
		for (int i = 0; i<N; ++i){
			U.push_back({ A[i], i });
			V.push_back({ B[i], i });

		}
		sort(U.begin(), U.end());
		sort(V.begin(), V.end());
		int p = N - 1, q = N - 1;
		vector<int> res(N, -1);
		while (p >= 0){
			while (q >= 0 && U[p].first <= V[q].first)--q;
			if (q<0)break;
			res[V[q].second] = U[p].first;
			--p; --q;
		}
		for (int i = 0; i<N; ++i){
			if (res[i] == -1){
				res[i] = U[p].first;
				--p;
			}
		}
		return res;
	}
};
