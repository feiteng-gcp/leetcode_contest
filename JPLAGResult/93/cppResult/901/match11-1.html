class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        multiset<int> s(A.begin(), A.end());
        vector<pair<int, int> > b;
        for(int i=0; i<B.size(); ++i) b.push_back({B[i], i});
        sort(b.begin(), b.end());
        reverse(b.begin(), b.end());
        vector<int> res(A.size(), -1);
        for(auto i: b) {
            int val = i.first; int pos = i.second;
            auto it = s.upper_bound(val);
            if (it == s.end()) continue;
            res[pos] = *it;
            s.erase(it);
        }
        for(auto &i: res) {
            if (i != -1) continue;
            i = *s.begin(); s.erase(s.begin());
        }
        return res;
    }
};
