class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        vector<int> result;
        multiset<int> AS;
        for(int i = 0; i < A.size();i++){
            AS.insert(A[i]);
        }
        for(int i = 0; i < B.size();i++){
            int cur = B[i];
            multiset<int>::iterator it = AS.lower_bound(cur+1);
            if(it == AS.end()){
               result.push_back(*(AS.begin()));
               AS.erase(AS.begin());
            }
            else{
               result.push_back(*it);
               AS.erase(it);
            }
        }
        return result;
    }
};
