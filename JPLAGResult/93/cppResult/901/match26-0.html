class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        
        vector<vector<int>> Bp;
        for (int i = 0; i < B.size(); i++) {
            vector<int> tmp;
            tmp.push_back(B[i]);
            tmp.push_back(i);
            Bp.push_back(tmp);
        }
        sort(Bp.begin(), Bp.end());
        
        vector<int> res;
        for (int i = 0; i < B.size(); i++)
            res.push_back(-1);
        
        deque<vector<int>> dq;
        
        for (auto x : Bp)
            dq.push_back(x);
                
        sort(A.begin(), A.end());
        
        for (auto x : A) {
            if (x > dq.front()[0]) {
                res[dq.front()[1]] = x;
                dq.pop_front();
            }
            else {
                res[dq.back()[1]] = x;
                dq.pop_back();
            }
        }
        
        
        return res;
        
        
        
    }
};
