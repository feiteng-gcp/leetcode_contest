class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        int N = A.size();
        vector<pair<int, int>> BInd;
        for (int i = 0; i < N; i++) {
            BInd.push_back({B[i], i});
        }
        sort(BInd.begin(), BInd.end());
        sort(A.begin(), A.end());
        vector<pair<int, int>> AInd;
        vector<pair<int, int>> BRest;
        int i = N - 1;
        for (; i >= 0 && !BInd.empty();) {
            pair<int, int> BI = BInd.back();
            BInd.pop_back();
            if (BI.first >= A[i]) {
                BRest.push_back(BI);
            } else {
                AInd.push_back({A[i], BI.second});
                i--;
            }
        }
        for (; i >= 0 && !BRest.empty(); i--) {
            AInd.push_back({A[i], BRest.back().second});
            BRest.pop_back();
        }
        vector<int> ans(N, 0);
        for (auto &AI : AInd) {
            ans[AI.second] = AI.first;
        }
        
        return ans;
    }
};
