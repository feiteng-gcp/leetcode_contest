struct indexcompare{
    vector<int> v;
    indexcompare(vector<int> &input):v(input) {};
    bool operator()(int a,int b){
        return v[a]<v[b];
    }
};
class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        int N=A.size();
        vector<int> result(N,-1);
        vector<int> idx(N,0);
        for(int i=0;i<N;i++){
            idx[i]=i;
        }
        indexcompare comp=indexcompare(B);
        sort(idx.begin(),idx.end(),comp);
        sort(A.begin(),A.end());
        /*
        cout<<"A="<<endl;
        for(auto i:A){
            cout<<i<<endl;
        }
        
         cout<<"idx="<<endl;
        for(auto i:idx){
            cout<<i<<endl;
        }
        */
        int idxA=0;
        int idxB=0;
        vector<int> tmp;
        while(idxB<N && idxA<N){
            if(A[idxA]<=B[idx[idxB]]){
                tmp.push_back(A[idxA]);
                idxA++;
            }
            else{
                result[idx[idxB]] =A[idxA];
                idxA++;
                idxB++;
            }
        }
        for(int i=N-1;i>=0;i--){
            if(result[i]==-1){
                result[i]=tmp.back();
                tmp.pop_back();
            }
        }
    
        return result;
        
    }
};
