class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) 
    {
        vector<Node> nodeA, nodeB;
        for(int i = 0;i < A.size(); i++)
        {
            Node node1, node2;
            node1.index = node2.index = i;
            node1.value = A[i];
            node2.value = B[i];
            nodeA.push_back(node1);
            nodeB.push_back(node2);

        }
            
        sort(nodeA.begin(), nodeA.end());
        sort(nodeB.begin(), nodeB.end());
        
        map<int, int> mp;
        int begA = 0, endA = A.size() - 1;
        for(int i = B.size() - 1; i >= 0; i--)
        {
            if(nodeA[endA].value > nodeB[i].value) 
            {
                mp[nodeB[i].index] = nodeA[endA].index;
                --endA;
            }
            else
            {
                mp[nodeB[i].index] = nodeA[begA].index;
                ++begA;
            }
        }
        
        vector<int> ret = A;
        for(auto it = mp.begin(); it != mp.end(); ++it)
            ret[it->first] = A[it->second];
        
        return ret;      
           
    }
    
    struct Node {
        int index;
        int value;
        
        bool operator < (const Node &other) const
        {
            return value < other.value;
        }
    };
};
