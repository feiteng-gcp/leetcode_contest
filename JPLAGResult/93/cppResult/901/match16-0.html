class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        sort(A.begin(),A.end());  vector<pair<int,int> > X;
        for(int i=0;i<B.size();++i) {
            X.push_back({B[i],i});
        }
        sort(X.begin(),X.end());
        int i=0,j=0,k=B.size()-1;     
        while(i<A.size()) {
            if(A[i]>X[j].first) {
                B[X[j].second]=A[i];   j++;
            }
            else {
                B[X[k].second]=A[i];  k--;
            }
            i++;
        }
        return B;
    }
};
