class Solution {
public:
    vector<int> advantageCount(vector<int>& a, vector<int>& b) 
    {
        sort(a.begin(),a.end());
        int len=a.size();
        vector<int> ret;
        
        for(int n : b)
        {
            int k=upper_bound(a.begin(),a.end(),n)-a.begin();
            if(k==a.size()) k=0;
            ret.push_back(a[k]);
            a.erase(a.begin()+k);
        }
        return ret;
    }
};
