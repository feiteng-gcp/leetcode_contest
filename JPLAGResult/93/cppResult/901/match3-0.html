class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        int n = A.size();
        vector<pair<int, int>> vec;
        for(int i = 0; i < n; i++)
        {
            vec.push_back(make_pair(B[i], i));
        }
        vector<int> ans;
        ans.resize(n, 0);
        int l = 0, r = n-1;
        sort(vec.begin(), vec.end());
        sort(A.begin(), A.end());
        for(int i = 0; i < n; i++)
        {
            auto tmp = vec[l];
            if(A[i]>tmp.first)
            {
                ans[tmp.second] = A[i];
                l++;
            }
            else
            {
                ans[vec[r].second] = A[i];
                r--;
            }
        }
        return ans;
    }
    
    
    
    
    
    
    
    
    
    
    
    
    
    
    
    
    
    
    
    
    
};
