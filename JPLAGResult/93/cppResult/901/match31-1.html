class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        
        sort(A.begin(), A.end());
        
        vector<int> res;
        for (int i = 0; i < B.size(); ++i)
        {
            auto it = upper_bound(A.begin(), A.end(), B[i]);
            if (it == A.end())
            {
                res.push_back(A[0]);
                A.erase( A.begin() );
            }
            else
            {
                res.push_back( *it );
                A.erase( it );
            }
        }
        
        return res;
    }
};
