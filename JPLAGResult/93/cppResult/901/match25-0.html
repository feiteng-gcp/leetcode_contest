class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        vector<pair<int,int>> ap,bp;
        for(int i=0;i<A.size();i++){
            ap.push_back(make_pair(A[i],i));
            bp.push_back(make_pair(B[i],i));
        }
        auto comp = [](pair<int,int>& a, pair<int,int>& b){return a.first>b.first;};
        sort(ap.begin(),ap.end(),comp);
        sort(bp.begin(),bp.end(),comp);
        vector<pair<int,int>> t(A.size());
        int s = 0,e = A.size()-1;
        for(int i=0;i<A.size();i++){
            if(ap[s].first>bp[i].first){
                t[i]=ap[s];
                s++;
            }else{
                t[i]=ap[e];
                e--;
            }
        }
        vector<int> res(A.size());
        for(int i=0;i<A.size();i++){
            res[bp[i].second]=t[i].first;
        }
        return res;
    }
};
