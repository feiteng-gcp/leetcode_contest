class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        multiset<int> ms(A.begin(), A.end());
        vector<int> res(A.size());
        for (size_t i=0; i<B.size(); i++) {

            multiset<int>::iterator it = ms.upper_bound(B[i]);
            if (it == ms.end()) {
                res[i] = *ms.begin();
                ms.erase(ms.begin());
            } else {
                res[i] = *it;
                ms.erase(it);
            }
        }
        return res;

    }
};
