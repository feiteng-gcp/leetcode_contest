class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        vector<int> bi;
        for(int i = 0; i < B.size(); ++ i) {
            bi.push_back(i);
        }
        sort(begin(bi), end(bi), [&](int i, int j) {
            return B[i] < B[j];
        });
        multiset<int> sa;
        for(int a : A) {
            sa.insert(a);
        }
        vector<int> p(A.size());
        for(int i : bi) {
            auto it = sa.upper_bound(B[i]);
            if(it == end(sa)) {
                it = begin(sa);
            }
            p[i] = *it;
            sa.erase(it);
        }
        return p;
    }
};
