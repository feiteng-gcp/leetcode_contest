class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        multiset<int> sa;
        for (auto v : A) {
            sa.insert(v);
        }
        vector<pair<int,int>> vb;
        for (int i = 0; i < B.size(); i++) {
            vb.push_back({B[i], i});
        }
        sort(vb.begin(), vb.end());
        vector<int> res(A.size(), -1);
        for (int i = vb.size() - 1; i >= 0; i--) {
            auto it = sa.upper_bound(vb[i].first);
            if (it != sa.end()) {
                res[vb[i].second] = *it;
                sa.erase(it);
            }
        }
        for (int i = 0; i < res.size(); i++) {
            if (res[i] == -1) {
                res[i] = *sa.begin();
                sa.erase(sa.begin());
            }
        }
        return res;
    }
};
