#define pb push_back
#define mp make_pair


struct node{
    int v,id;
    node(int v1,int ii){
        v=v1;id=ii;
    }
};

bool cmp1(const node a,const node b){
    return a.v<b.v;
}
bool cmp2(node a,node b){
    return a.id<b.id;
}

class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        vector<node>b;
        vector<node>a;
        int n=B.size();
        for(int i=0;i<n;++i){
            b.pb(node(B[i],i));
        }
        sort(b.begin(),b.end(),cmp1);
        //sort(a.begin(),a.end(),cmp1);
        sort(A.begin(),A.end());
        int i=0,j=0;
        vector<int>vis(n,0);
        while(j<n){
            while(i<n&&A[i]<=b[j].v) ++i;
            if(i==n) break;
            a.pb(node(A[i],b[j].id));
            vis[i]=1;
            ++i;
            ++j;
        }
        for(int i=0;i<n;++i){
            if(!vis[i]){
                a.pb(node(A[i],b[j].id));
                ++j;
            }
        }
        sort(a.begin(),a.end(),cmp2);
        vector<int>ans;
        for(int i=0;i<n;++i) ans.pb(a[i].v);
        return ans;
    }
};
