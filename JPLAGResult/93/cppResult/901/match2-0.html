class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        // greedy algorithm
        int n = A.size();
        
        sort(A.begin(), A.end());
        unordered_set<int> used;
        vector<int> res;
        for (int i=0; i<n; ++i) {
            auto target = B[i];
            auto it = upper_bound(A.begin(), A.end(), target);
            if (it != A.end()) {
                int j = it - A.begin();
                while(j<n && used.count(j)) ++j;
                if (j<n) {
                    //cout << j << endl;
                    used.insert(j);
                    res.push_back(A[j]);
                    continue;
                }
            }

            // put smallest
            int j = getSmallest(A, used);
            res.push_back(A[j]);
            used.insert(j);
        }
        
        return res;
        
    }
    
    int getSmallest(vector<int>& A, unordered_set<int>& used) {
        int i=0;
        while(i<A.size() && used.count(i)) ++i;
        return i;
    }
};
