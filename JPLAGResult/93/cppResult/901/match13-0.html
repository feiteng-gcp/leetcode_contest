class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        multiset<int> aa(A.begin(), A.end());
        vector<int> res;
        for(int i : B){
            int cur = 0;
            auto it = aa.upper_bound(i);
            if(it == aa.end()){
                cur = *(aa.begin());
                aa.erase(aa.begin());
            }
            else{
                cur = *it;
                aa.erase(it);
            }
            res.push_back(cur);
        }
        return res;
    }
};
