class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        vector<pair<int,int>> tb;
        for(int i = 0; i < B.size(); ++i){
            tb.push_back({B[i], i});
        }
        
        sort(tb.begin(), tb.end());
        
        map<int,int> m;
        for(auto x : A){
            m[x]++;
        }
        
        int n = B.size();
        vector<int> ans(n);
        for(auto p : tb){
            auto v = p.first;
            auto it = m.upper_bound(v);
            if(it == m.end()){
                it = m.begin();
            }
            
            ans[p.second] = it->first;
            if(--(it->second) == 0) m.erase(it);
        }
        
        return ans;
    }
};
