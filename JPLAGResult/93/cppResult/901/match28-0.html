class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        multiset<int> ss;
        for (int a : A)
            ss.emplace(a);
        vector<pair<int, int>> bb(B.size());
        for (int i = 0; i < B.size(); i++) 
            bb[i] = make_pair(B[i], i);
        sort(bb.begin(), bb.end());
        vector<int> ans(A.size());
        for (int i = 0; i < B.size(); i++) {
            int j = bb[i].second;
            auto it = ss.upper_bound(bb[i].first);
            if (it != ss.end()) {
                ans[j] = *it;
                ss.erase(it);
            } else {
                ans[j] = *ss.begin();
                ss.erase(ss.begin());
            }
        }
        return ans;
    }
};
