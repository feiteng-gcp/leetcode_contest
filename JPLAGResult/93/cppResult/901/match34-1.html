class Solution {
public:
    vector<int> advantageCount(vector<int>& A, vector<int>& B) {
        sort(A.begin(), A.end());
        vector<int> tmp = A;
        vector<int> result;
        
        for (int i = 0; i < B.size(); i++) {
            bfind(tmp, B[i], result);
        }
        
        return result;
    }
    
    void bfind(vector<int>& A, int target, vector<int>& result) {
        int left = 0, right = A.size();
        int mid;
        
        while (left <= right) {
            mid = left + (right - left) / 2;
            if (A[mid] <= target) left = mid + 1;
            else right = mid - 1;
        }
        
        if (left >= A.size()) {
            result.push_back(A[0]);
            vector<int>::iterator iter = A.begin();
            A.erase(iter);
        } else {
            result.push_back(A[left]);
            vector<int>::iterator iter = A.begin() + left;
            A.erase(iter);
        }
    }
};
