class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        /*stations.clear();
        vector<int> z1;
        z1.push_back(10);
        z1.push_back(100);
        stations.push_back(z1);
        target=100;
        startFuel=1;*/
        vector<int> z;
        z.push_back(target);
        z.push_back(0);
        stations.push_back(z);
        priority_queue<int> q;
        int asc=0,pos=0,tank=startFuel;
        for(int i=0; i<stations.size(); i++){
            int d=stations[i][0]-pos;
            while(tank-d<0){
                if(q.empty()){
                    return -1;
                }
                tank+=q.top();
                q.pop();
                asc++;
            }
            tank-=d;
            pos=stations[i][0];
            q.push(stations[i][1]);
        }
        return asc;
    }
};
