class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        priority_queue<int> pq;
        int f = startFuel;
        int d = 0;
        int l = stations.size();
        int cnt = 0;
        while(f < target){
            while(d < l && stations[d][0] <= f){
                pq.push(stations[d][1]);
                d++;
            }
            if(pq.empty())
                return -1;
            else{
                f += pq.top();
                pq.pop();
                cnt++;
            }
        }
        return cnt;
    }
};
