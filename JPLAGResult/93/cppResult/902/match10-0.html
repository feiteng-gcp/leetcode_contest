class Solution {
    typedef long long int64_t;
    vector<vector<int>> stations;
    int64_t f[555][555];
    bool visited[555][555];
    int N;
    
    inline int getDist(int x, int y) {
        return stations[y][0] - stations[x][0];
    }
    
    int64_t maxGas(int stops, int d) {
        if (visited[stops][d])
            return f[stops][d];
        
        visited[stops][d] = true;
        if (stops == 0) {
            f[stops][d] = stations[0][1] - getDist(0, d); // not including gas at station d
            return f[stops][d];
        }
        f[stops][d] = std::numeric_limits<int>::min();
        for (int last = 1; last < d; last++) {
            int64_t last_gas = maxGas(stops - 1, last);
            //cout << last_gas << endl;
            if (last_gas < 0)
                continue;
            f[stops][d] = max(f[stops][d],
                              last_gas + stations[last][1] - getDist(last, d));
        }
        
        //cout << stops << ' ' << d << ' ' << f[stops][d] << endl;;
        
        return f[stops][d];
    }
    
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& in_stations) {
        stations.push_back(vector<int>{0, startFuel});
        stations.insert(stations.end(), in_stations.begin(), in_stations.end());
        stations.push_back(vector<int>{target, 0});
        
        memset(f, 0, sizeof(f));
        
        N = stations.size();
        
        for (int stops = 0; stops <= N; stops++) {
            if (maxGas(stops, N - 1) >= 0)
                return stops;
        }
        
        return -1;
    }
};
