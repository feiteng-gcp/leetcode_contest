class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        vector<int> curfuel;
        int ans = 0;
        int cur = startFuel;
        for(int i = 0; i < stations.size(); i++)
        {
            while (cur < stations[i][0] && int(curfuel.size()) > 0)
            {
                cur += curfuel[0];
                curfuel.erase(curfuel.begin());
                ans += 1;
            }
            if (cur < stations[i][0])
                return -1;
            
            int j = 0;
            while (j < curfuel.size() && stations[i][1] < curfuel[j])
                j++;
            curfuel.insert(curfuel.begin() + j, stations[i][1]);
        }
        
        while (cur < target && int(curfuel.size()) > 0)
        {
            cur += curfuel[0];
            curfuel.erase(curfuel.begin());
            ans += 1;
        }
        if (cur < target)
            return -1;
        else
            return ans;
        
    }
};
