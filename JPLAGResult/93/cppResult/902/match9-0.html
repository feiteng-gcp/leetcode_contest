long long dp[600][600];
long long f[600][600];
class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& b) {
        vector<vector<int>> a;
        a.push_back({ 0, 0});
        for (auto x : b)
            a.push_back(x);
        a.push_back({ target, 0 });

        int n = a.size();
        for (int i = 0; i < 600; ++i)
            for (int j = 0; j < 600; ++j) {
                dp[i][j] = -1;
                f[i][j] = -1;
            }

        dp[0][0] = startFuel;
        f[0][0] = startFuel;
        for (int i = 1; i < n; ++i) {
            int dis = a[i][0];
            
            for (int k = 0; k < i; ++k) {
                
                if (f[i - 1][k] < dis)
                    continue;
                f[i][k] = max(f[i][k], f[i - 1][k]);
                dp[i][k + 1] = max(dp[i][k + 1], f[i - 1][k] + a[i][1]);
                f[i][k + 1] = max(f[i][k + 1], dp[i][k + 1]);
                // trace(i, j, k, dp[i][k + 1]);
            }
        }
        for (int i = 0; i < n; ++i) if (dp[n - 1][i] >= 0)
            return i - 1;
        return -1;
    }
};
