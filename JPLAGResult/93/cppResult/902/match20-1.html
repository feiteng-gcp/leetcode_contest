#define ll long long
ll f[600][600];
ll x0;
vector<vector<int>> s;
class Solution {
public:
    int maxfuel(int pos, int n) {
        // pos, maxF
        if (f[pos][n] != -1) return f[pos][n];
        if (pos == 0) {
            f[0][0] = max(f[0][0], x0 - s[0][0]);
            f[0][1] = max(f[0][1], f[0][0] == -1 ? -1: (f[0][0] + s[0][1]));
            return f[pos][n];
        }
        if (n == 0) return f[pos][n] = max(-1ll, x0 - s[pos][0]);
              f[pos][n] = max(max(f[pos-1][n] - s[pos][0] + s[pos-1][0], f[pos][n]),
                        f[pos-1][n-1] < s[pos][0] - s[pos-1][0] ? -1 : (f[pos-1][n-1] - s[pos][0] + s[pos-1][0] + s[pos][1]));
        return f[pos][n];
    }
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        memset(f, -1, sizeof(f));
        x0 = startFuel;
        s = stations;
        int res = INT_MAX;
        if (startFuel >= target) return 0;

        if (stations.empty() || stations[0][0] > startFuel) return -1;
        for(int i=0; i<stations.size(); ++i) {
            for(int j=0; j<=i+1; ++j) {
                int val = maxfuel(i, j);
                if (val == -1) continue;
                if (target < stations[i][0]) continue;
                if (val >= target - stations[i][0]) res = min(res, j);
            }
        }
        return res == INT_MAX ? -1 : res;
    }
};
