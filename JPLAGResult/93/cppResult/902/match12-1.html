class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        if (startFuel >= target) return 0;
        
        priority_queue<int> queue;
        
        int curFuel = startFuel;
        int curPos = 0;
        int res = 0;
        
        for (int i = 0; i < stations.size(); i++) {
            int dis = stations[i][0];
            int fuel = stations[i][1];
            
            if (dis - curPos > curFuel) {
                if (queue.size() == 0) return -1;
                
                while (dis - curPos > curFuel) {
                    if (queue.size() == 0) return -1;
                    int tmp = queue.top();
                    queue.pop();
                    curFuel += tmp;
                    res += 1;
                }
            }
            
            queue.push(fuel);
            
            curFuel -= (dis - curPos);
            curPos = dis;
        }
        
        if (target - curPos <= curFuel) return res;
        else {
            if (queue.size() == 0) return -1;
            
            while (target - curPos > curFuel) {
                if (queue.size() == 0) return -1;
                int fuel = queue.top();
                queue.pop();
                curFuel += fuel;
                res += 1;
            }
        }
        
        return res;
    }
};
