class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        if (startFuel >= target) return 0;
        priority_queue<int> q;
        
        int i=0;
        int n = stations.size();
        int res = 0;
        while(startFuel<target) {
            for (; i<n; ++i) {
                if (startFuel < stations[i][0]) {
                    // cannot reach
                    break;
                }
                q.push(stations[i][1]);
            }
            if (q.empty()) return -1;
            startFuel += q.top(); q.pop();
            ++res;
        }
        
        return res;
        
    }
};
