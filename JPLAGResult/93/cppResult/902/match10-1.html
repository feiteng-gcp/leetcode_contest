class Solution {

public:
	int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
		stations.insert(stations.begin(), vector<int>{0, startFuel});
		stations.push_back(vector<int>{target, 0});
		int n = stations.size();
		vector<vector<int>> dp(n, vector<int>(n, -1));
		dp[0][0] = startFuel;
		for (int i = 1; i < n; ++i) {
			int cost = stations[i][0] - stations[i - 1][0];
			for (int j = 0; j <= i; ++j) {
				dp[i][j] = max(dp[i][j], dp[i - 1][j] - cost); //不加
				if (j > 0 && dp[i - 1][j - 1] >= cost)
					dp[i][j] = max(dp[i][j], dp[i - 1][j - 1] - cost + stations[i][1]); //加油
			}
		}
		for (int j = 0; j < n; ++j)
			if (dp[n - 1][j] != -1)
				return j;
		return -1;
	}
};
