class Solution {
public:
	int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
		int currentPos = startFuel;
		priority_queue<int> fuels;
		int stops = 0;
		int stI = 0;
		while (stI<stations.size()) {
			if (stations[stI][0]>currentPos)
				break;
			else {
				fuels.push(stations[stI][1]);
				stI++;
			}
		}
		while (!fuels.empty() && currentPos<target) {
			currentPos += fuels.top();
			fuels.pop();
			stops++;
			while (stI<stations.size()) {
				if (stations[stI][0]>currentPos)
					break;
				else {
					fuels.push(stations[stI][1]);
					stI++;
				}
			}
		}
		if (currentPos < target)
			return -1;
		else
			return stops;
	}
};
