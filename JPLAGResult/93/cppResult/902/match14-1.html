class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        int now = 0;
        priority_queue<int> que;
        que.push(startFuel);
        int idx = 0, n = stations.size();
        int res = 0;
        while (!que.empty() && now < target) {
            res++;
            int t = que.top();
            que.pop();
            now += t;
            while (idx < n && stations[idx][0] <= now) {
                que.push(stations[idx][1]);
                idx++;
            }
        }
        return now >= target ? res - 1 : -1;
    }
};
