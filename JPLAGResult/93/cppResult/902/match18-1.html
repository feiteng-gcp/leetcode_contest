typedef long long ll;

ll dp[1000][1000];


class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& s) {
        int n = s.size();
        if (n == 0) {
            if (target > startFuel) return -1;
            return 0;
        }
        memset(dp, -1, sizeof(dp));
        dp[0][0] = startFuel;
        for (int i = 1; i <= n; ++i) {
            for (int j = 0; j <= i; ++j) {
                ll gap = i == 1 ? s[i - 1][0] : s[i - 1][0] - s[i - 2][0];
                if (dp[i - 1][j] - gap >= 0)
                    dp[i][j] = max(dp[i - 1][j] - gap, dp[i][j]);
                if (j - 1 >= 0 && dp[i - 1][j - 1] - gap >= 0)
                    dp[i][j] = max(dp[i - 1][j - 1] - gap + s[i - 1][1], dp[i][j]);
            }
        }
        ll final_gap = target - s[n - 1][0];
        for (int i = 0; i <= n; ++i)
            if (dp[n][i] - final_gap >= 0) return i;
        return -1;
    }
};
