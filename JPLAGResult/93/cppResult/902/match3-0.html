class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        priority_queue <int> s;
        int cnt=0;
        vector <int> t;
        t.push_back(target);
        t.push_back(0);
        stations.push_back(t);
        int pos=0,tank=startFuel;
        for (int i=0;i<stations.size();i++){
            int d=stations[i][0]-pos;
            while (tank-d<0){
                if (s.empty()){
                    return -1;
                }
                tank+=s.top();
                s.pop();
                cnt++;
            }
            tank=tank-d;
            pos=stations[i][0];
            s.push(stations[i][1]);
        }
        return cnt;
    }
};
