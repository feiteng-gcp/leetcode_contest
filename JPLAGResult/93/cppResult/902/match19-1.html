long long f[505][505];
class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        
        vector<int> tmp;
        tmp.push_back(target);
        tmp.push_back(0);
        stations.push_back(tmp);
        
        int n = stations.size();
        
        memset(f, -1, sizeof f);
        
        f[0][0] = startFuel;
        
        for (int i = n-1; i > 0; i--) {
            stations[i][0] -= stations[i-1][0];
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (f[i][j] < 0) continue;
                
                f[i+1][j] = max(f[i+1][j], f[i][j] - stations[i][0]);
                if (f[i][j] - stations[i][0] >= 0)

                    f[i+1][j+1] = max(f[i+1][j+1], f[i][j] - stations[i][0] + stations[i][1]);
            }
        }
        /*
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= n; j++) {
                printf("%d %d %d\n", i, j, f[i][j]);
            }
        }
        */
        for (int i = 0; i < n; i++) {
            if (f[n][i] >= 0) return i;
        }
        
        return -1;
    }
};
