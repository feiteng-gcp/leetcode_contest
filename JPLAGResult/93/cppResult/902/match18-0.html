class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        int n = stations.size();
        if(n==0){
            if(target<=startFuel)
                return 0;
            else
                return -1;
        }
        vector<int> temp(n+1,-1);
        vector<vector<int>> gas(n+1,temp);
        gas[0][0] = startFuel;
        for(int i=1;i<n+1;i++)
            for(int j=0;j<=i;j++){
                int pre;
                if(i==1)
                    pre = 0;
                else
                    pre = stations[i-2][0];
                int cur = stations[i-1][0];
                if(gas[i-1][j]>=cur-pre)
                    gas[i][j] = gas[i-1][j] - (cur-pre);
                if(j>0 && gas[i-1][j-1]>=cur-pre)
                    gas[i][j] = max(gas[i][j], gas[i-1][j-1] + stations[i-1][1] - (cur-pre));
            }
        for(int i=0;i<=n;i++)
            if(gas[n][i]>=target-stations[n-1][0])
                return i;
        return -1;
            
    }
};
