class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& A) {
        sort(A.begin(),A.end());  priority_queue<int> X;
        int ans=0,i=0;
        while(startFuel<target) {
            while(i<A.size()&&startFuel>=A[i][0]) {
                X.push(A[i][1]);   i++;
            }
            if(X.size()==0) return -1;
            startFuel+=X.top();  X.pop();   ans++;
        }
        return ans;
    }
};
