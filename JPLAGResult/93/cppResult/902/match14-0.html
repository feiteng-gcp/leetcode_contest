class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& stations) {
        int ans = 0;
        priority_queue<int> que;
        int total = startFuel;
        vector<int> vec;
        vec.push_back(target);
        vec.push_back(0);
        stations.push_back(vec);
        for(int i = 0; i < stations.size(); i++)
        {
            if(total>=stations[i][0])
            {
                que.push(stations[i][1]);
            }
            else
            {
                if(que.empty())
                {
                    return -1;
                }
                while(!que.empty())
                {
                    if(total>=stations[i][0])
                    {
                        break;
                    }
                    else
                    {
                        total += que.top();
                        que.pop();
                        ans++;
                    }
                }
                i--;
            }
        }
        return ans;
    }
};
