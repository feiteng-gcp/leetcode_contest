class Solution {
public:
    int minRefuelStops(int target, int startFuel, vector<vector<int>>& st) {
        int len = st.size();

        if (len == 0) return (target <= startFuel ? 0 : -1);

        int ll = len +1;
        vector<vector<int>> dp(ll, vector<int>(ll, -1));
        
        // init dp;
        dp[0][0] = startFuel - st[0][0]; 
        if (dp[0][0] < 0) return -1;
        
        for (int i = 1; i <= len; i++)
        {
            for (int k = 0; k <= i; k++)
            {
                int dis = 0;
                if (i == len) dis = target - st[i - 1][0];
                else dis = st[i][0] - st[i-1][0];
                
          //      cout << i << " " << k << " " <<  dis;
                
                int d = -1;
                if (k > 0)
                {
                    d = dp[i-1][k-1];
                //    cout << " " << d;
                    if (d >= 0) d += st[i-1][1];
                    d-= dis;
                    dp[i][k] = d;
                }
                
                dp[i][k] = max(dp[i][k], dp[i-1][k] - dis);  
           //     cout << " " << dp[i-1][k] << " " << dp[i][k] << endl;
            }
        }

        for (int times = 0; times <= len; times++)
        {
            if (dp[len][times] >= 0) return times;
        }
        
        return -1;
    }
};
