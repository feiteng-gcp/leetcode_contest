class Solution {
public:
    int binaryGap(int N) 
    {
        int ret = 0, pre = -1, cur = 0;
        while(N)
        {            
            if(N & 1)
            {
                if(pre != -1)

                {
                    int tmp = cur - pre;
                    ret = max(ret, tmp);
                }
                
                pre = cur;
            }
            ++cur;
            N >>= 1;
        }
        return ret;
        
    }
};
