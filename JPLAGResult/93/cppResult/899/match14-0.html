class Solution {
public:
    int binaryGap(int N) {
        vector<int> m;
        for(int i = 0; i < 32; ++i) {
            if(N & (1 << i)) m.push_back(i);
        }
        if(m.size() < 2) return 0;
        
        int res = 0;
        for(int i = 1; i < m.size(); ++i) res = max(res, m[i] - m[i - 1]);
        return res;
    }
};
