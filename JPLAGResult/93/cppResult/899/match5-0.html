class Solution {
public:
    int binaryGap(int N) {
        int last = -1;
        int curr = 0;
        int ans = 0;
        
        while (N > 0) {
            if (N & 1) {
                if (last >= 0) {
                    ans = max(ans, curr - last);
                }
                last = curr;
            }
            curr++;
            N >>= 1;
        }
        
        return ans;
    }
};
