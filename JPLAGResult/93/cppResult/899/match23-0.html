class Solution {
public:
    int binaryGap(int N) {
        int c = 0, a = -1, ans = 0;
        while(N){
            if(N&1){
                if(a!=-1) ans = max(ans, c - a);
                a = c;
            }
            N/=2;
            c++;
        }
        return ans;
    }
};
