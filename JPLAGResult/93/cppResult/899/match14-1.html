class Solution {
public:
    int binaryGap(int N) {
        vector<int> V;
        for(int i=0;i<32;++i){
            if((N&(1<<i)))V.push_back(i);
        }
        if(V.size()<2)return 0;
        int res=0;
        for(int i=1;i<V.size();++i)res=max(res,V[i]-V[i-1]);
        return res;
    }
};
