class Solution {
public:
    int binaryGap(int N) {
        int res = 0;
        int n = 1;
        int last = 0;
        while (N) {
            if (N&1) {
                if (last != 0)
                    res = max(n-last, res);
                last = n;
            }
            N >>= 1;
            n++;
        }
        return res;

    }
};
