class Solution {
public:
    int binaryGap(int N) {
        vector<int> v;
        for(int i=0; i<=30; i++){
            if(N%2 == 1) v.push_back(i);
            N = (N>>1);
        }
        if(v.size() < 2) return 0;
        int ans = -1;
        for(int i=0; i<v.size()-1; i++){
            ans = max(ans, v[i+1]-v[i]);
        }
        return ans;
    }
};
