class Solution {
public:
    int binaryGap(int N) {
        vector<int> vt;
        for(int i = 0; i < 31;++i) {
            if(N & (1 << i)) vt.push_back(i);
        }
        int n = vt.size();

        if(n <= 1)  return 0;
        int mx = 0;
        for(int i = 1;i < n;++i) {
            mx = max(mx,vt[i] - vt[i-1]);
        }
        return mx;
    }
};
