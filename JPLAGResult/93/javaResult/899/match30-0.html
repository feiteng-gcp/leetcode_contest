class Solution {
    public int binaryGap(int N) {
        int res = 0;
        int one = -1;
        int ind = 0;
        while (N > 0){
            if ((N & 1) == 1){
                if (one >= 0) res = Math.max(res, ind - one);
                one = ind;
            } 
            ind++;
            N >>= 1;
        }
        return res;
    }
}
