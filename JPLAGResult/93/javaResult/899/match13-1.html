class Solution {
    public int binaryGap(int N) {
        int max = 0;
        int last = -1;
        int i = 0;
        while (N != 0) {
            if (N % 2 == 1) {
                if (last != -1) {
                    max = Math.max(max, i - last);
                }
                last = i;
            }
            i++;
            N = (N >> 1);
        }
        return max;
    }
}
