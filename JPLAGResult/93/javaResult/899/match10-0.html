class Solution {
    public int binaryGap(int N) {
        int idx = 0;
        int last = -1;
        int max = 0;
        while (N > 0) {
            int tmp = N % 2;
            if (tmp == 1) {
                if (last != -1) {
                    max = Math.max(max, idx - last);
                }
                last = idx;
            }
            N /= 2;
            idx++;
        }
        return max;
    }
}
