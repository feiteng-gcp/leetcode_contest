class Solution {
    public int binaryGap(int N) {
        int index = 0;
        
        int max = 0;
        int prev = -1;
        while (N>0) {
            if ((N & 0x01) == 1) {
                if (prev != -1) {
                    max = Math.max(max, index - prev);
                }
                prev = index;
            }
                
            index++;
            N = N >> 1;
        }
        
        return max;
    }
}
