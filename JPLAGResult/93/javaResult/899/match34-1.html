class Solution {
    public int binaryGap(int N) {
        Integer pre = null;
        int index = -1, res = 0;
        while (N > 0) {
            index++;
            
if (N % 2 == 1) {
                if (pre != null) {
                    res = Math.max(res, index - pre);
                }
                pre = index;
            }
            N /= 2;
        }
        return res;
    }
}
