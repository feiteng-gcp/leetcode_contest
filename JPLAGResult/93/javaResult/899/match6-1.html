class Solution {
    public int binaryGap(int N) {
        String mid = Integer.toBinaryString(N);
        int before = -1;
        int max = 0;
        for (int i = 0; i < mid.length(); i ++) {
            if (mid.charAt(i) == '1') {
                if (before >= 0) {
                    max = Math.max(max, i - before);
                }
                before = i;
            }
        }
        return max;
    }
}
