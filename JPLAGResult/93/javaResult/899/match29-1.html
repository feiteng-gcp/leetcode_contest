class Solution {
    public int binaryGap(int N) {
        int last = -1, current = 0;
        int result = 0;
        while (N > 0) {
            int value = N % 2;
            N /= 2;
            
if (value == 1) {
                if (last != -1) {
                    result = Math.max(current - last, result);
                }
                last = current;
            }
            current++;
        }
        return result;
    }
}
