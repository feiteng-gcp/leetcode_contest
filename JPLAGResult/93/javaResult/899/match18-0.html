class Solution {
  public int binaryGap(int N) {
        int lastOneBit = -1;
        int curBit = 0;
        int maxGap = 0;
        while (N != 0) {
            if ((N & 1) == 1) {
                if (lastOneBit != -1) {
                    maxGap = Math.max(maxGap, curBit-lastOneBit);
                }
                lastOneBit = curBit;
            }
            N = N >> 1;
            curBit++;
        }
        return maxGap;
    }
}
