class Solution {
    public int binaryGap(int N) {
        int dist = 0;
        int j = -1;
        
        for (int i = 0; i < 32; i++) {
            
if (((N >> i) & 1) > 0) {
                if (j < 0) {
                    j = i;
                } else {
                    dist = Math.max(dist, i - j);
                    j = i;
                }
            }
        }
        
        return dist;
    }
}
