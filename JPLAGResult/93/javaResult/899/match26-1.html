class Solution {
    public int binaryGap(int N) {
        int rear = -1;

        int max = 0;
        int i = 0;
        while (N > 0) {
            int b = N % 2;
            
if (b == 1) {
                if (rear == -1) {
                    rear = i;
                } else {
                    max = Math.max(max, i - rear);
                    rear = i;
                }
            }
            N /= 2;
            i++;
        }

        return max;
    }
}
