class Solution {
    public int binaryGap(int N) {
        int i = 0;
        int res = 0;
        int last = -1;
        while (N > 0) {
            int c = N % 2;
            
if (c == 1) {
                if (last != -1) {
                    res = Math.max(res, i - last);
                }
                last = i;
            }
            i++;
            N /= 2;
        }
        return res;
    }
}
