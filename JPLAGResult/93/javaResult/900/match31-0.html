class Solution {
    boolean res = false;
    public boolean reorderedPowerOf2(int N) {
        char[] sa = Integer.toString(N).toCharArray();
        help(sa, new char[sa.length], 0);
        return res;
    }
    private void help(char[] sa, char[] cur, int index) {
        if (res) return;
        if (index == sa.length) {
            int num = Integer.parseInt(new String(cur));
            if (Integer.toString(num).length() != sa.length) return;
            if (Integer.bitCount(num) == 1)
                res = true;
            return;
        }
        for (int i = 0; i < cur.length; i++) {
            if (cur[i] == 0) {
                cur[i] = sa[index];
                help(sa, cur, index + 1);
                cur[i] = 0;
            }
        }
    }
}
