class Solution {
  public boolean reorderedPowerOf2(int N) {
        List<Integer> powerOfTwo = new ArrayList<>();

        int runner = 1;
        while (runner <= Math.pow(10, 9)) {
            powerOfTwo.add(runner);
            runner *= 2;
        }

        
for(Integer candidate: powerOfTwo) {
            if (check(N, candidate)) {
                return true;
            }
        }

        return false;
    }

    private boolean check(int N1, int N2) {
        String s1 = String.valueOf(N1);
        String s2 = String.valueOf(N2);

        if (s1.length() != s2.length()) {
            return false;
        } else {

            int[] arr1 = arr(s1);
            int[] arr2 = 
arr(s2);

            for(int i = 0; i < arr1.length; i ++) {
                if (arr1[i] != arr2[i]) {
                    return false;
                }
            }

            return true;
        }

    }

    private int[] arr(String s1) {
        int[] arr = new int[10];
        for(char c: s1.toCharArray()) {
            arr[c-'0'] += 1;
        }

        return arr;
    }

}
