import java.util.LinkedList;
import java.util.List;

class Solution {
    public boolean reorderedPowerOf2(int N) {
        String target = String.valueOf(N);
        if(target.length() == 10) return canReorder(target, "1073741824");

        int i = 1;
        while(String.valueOf(i).length() < target.length()) {
            i *= 2;
        }
        while(String.valueOf(i).length() == target.length()) {
            if(canReorder(target, String.valueOf(i))) return true;
            i *= 2;
        
}
        return false;
    }

    private boolean canReorder(String target, String s) {
        int[] a = new int[10];
        for(int i = 0; i < s.length(); i++) {
            a[target.charAt(i) - '0']++;
            a[s.charAt(i) - '0']--;
        }
        
for(int i = 0; i < 10; i++) {
            if(a[i] != 0) return false;
        }
        return true;
    }
}
