class Solution {
    public boolean reorderedPowerOf2(int N) {
        if (isPower2(N)) return true;
        String num = Integer.toString(N);
        int[] nums = new int[num.length()];
        for (int i = 0; i < num.length(); i++) {
            nums[i] = num.charAt(i) - '0';
        }
        
        List<Integer> list = new ArrayList<>();
        helper(list, new ArrayList<>(), nums, new boolean[num.length()]);
        
        for (int number : list) {
            if (isPower2(number)) return true;
        }
        return false;
        
    }
    
    private void helper(List<Integer> list, List<Integer> l, int[] nums, boolean[] used) {
        int n = nums.length;
        if (l.size() == n) {
            if (l.get(0) != 0) list.add(getNumber(l));
            return;
        }
        
        for (int i = 0; i < n; i++) {
            if (used[i] || i > 0 && nums[i] == nums[i - 1] && !used[i - 1]) continue;

            l.add(nums[i]);
            used[i] = true;
            helper(list, l, nums, used);
            l.remove(l.size() - 1);
            used[i] = false;
        }
    }
    
    private int getNumber(List<Integer> l) {
        int num = 0;
        for (int n : l) {
            num = num * 10 + n;
        }
        
        return num;
    }
    
    public boolean isPower2(int num) {
        if (num == 1) return true;
        
        while (num > 1) {
            if (num % 2 != 0) return false;
            num = num / 2;
        }
        
        return num == 1;
    }
}
