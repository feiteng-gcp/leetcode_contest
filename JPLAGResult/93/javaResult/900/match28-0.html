class Solution {
    public boolean reorderedPowerOf2(int N) {
        String str = Integer.toString(N);
        int[] nums = new int[str.length()];
        for (int i = 0; i < nums.length; i++) {
            nums[i] = str.charAt(i) - '0';
        }
        List<List<Integer>> perms = permuteUnique(nums);
        for (List<Integer> perm : perms) {
            if (perm.get(0) == 0) {
                continue;
            }
            long tmp = getNums(perm);
            if (valid(tmp)) {
                return true;
            }
        }
        return false;
    }
    
    private long getNums(List<Integer> perm) {
        long ans = 0;
        for (int i = 0; i < perm.size(); i ++) {
            ans *= 10;
            ans += perm.get(i);
        }
        return ans;
    }
    
    private boolean valid(long x) {
        return (x&(x-1)) == 0;
    }
    
    public List<List<Integer>> permuteUnique(int[] nums) {
		List<List<Integer>> ans = new ArrayList<>();
		if (nums == null || nums.length == 0) {
			return ans;
		}
		Arrays.sort(nums);
		boolean[] used = new boolean[nums.length];
		Arrays.fill(used, false);
		helper(ans, new ArrayList<Integer>(), nums, used);
		return ans;
	}
	
	private void helper(List<List<Integer>> ans, List<Integer> cur, int[] nums, boolean[] used) {
		if (cur.size() == nums.length) {
			ans.add(new ArrayList<Integer>(cur));
			return;
		}
		for (int i = 0; i < nums.length; i++) {
			if (used[i] || 
					i > 0 && nums[i] == nums[i - 1] && !used[i - 1]) {
				continue;
			}
			cur.add(nums[i]);
			used[i] = true;
			helper(ans, cur, nums, used);
			cur.remove(cur.size() - 1);
			used[i] = false;
		}
	}
}
