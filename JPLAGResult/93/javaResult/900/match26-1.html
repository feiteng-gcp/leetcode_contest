import java.util.Arrays;

class Solution {
    public boolean reorderedPowerOf2(int N) {
    	String s = sort(N);
        long p = 1;
        
for (int i=0; i<32; i++) {
        	String s2 = sort(p);
        	if (s.equals(s2))
        		return true;
        	p<<=1;
        }
		return false;
    }

	private String sort(long n) {
		char[] a = Long.toString(n).toCharArray();
		Arrays.sort(a);
		return new String(a);
	}
}
