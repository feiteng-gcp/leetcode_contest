class Solution {
    public boolean reorderedPowerOf2(int N) {
        String[] power = new String[31];
        
        int val = 1;
        for (int i=0; i<31; i++) {
            power[i] = String.valueOf(val);
            val = val * 2;
        }
        
        String input = String.valueOf(N);
        for (int i=0; i<power.length; i++) {
            if (anagram(input, power[i])) {
                return true;
            }
        }
        
        return false;       
    }
    
    boolean anagram(String a, String b) {
        if (a.length() != b.length()) {
            return false;
        }
        
        char[] c1 = a.toCharArray();
        char[] c2 = b.toCharArray();
        Arrays.sort(c1);
        Arrays.sort(c2);
        
        for (int i=0; i<c1.length; i++) {
            if (c1[i] != c2[i]) {
                return false;
            }
        }
        return true;
    }
}
