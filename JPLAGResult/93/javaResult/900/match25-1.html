class Solution {
    public boolean reorderedPowerOf2(int N) {
        int[] nums = new int[10];
        int count = 0;
        while (N > 0) {
            count++;
            nums[N % 10]++;
            N /= 10;
        }
        return dfs(nums, 0, count);
    }
    
    private boolean dfs(int[] nums, int curr, int remain) {
        if (remain == 0) {
            return (curr & (curr - 1)) == 0;
        }
        for (int i = 0; i < nums.length; i++) {
            if (nums[i] > 0) {
                if (i == 0 && curr == 0) {
                    continue;
                }
                curr *= 10;
                curr += i;
                nums[i]--;
                if (dfs(nums, curr, remain - 1)) {
                    return true;
                }
                nums[i]++;
                curr /= 10;
            }
        }
        return false;
    }
}
