class Solution {
    public boolean reorderedPowerOf2(int N) {
        String s = Integer.toString(N);
        int len = s.length();
        List<String> power2 = new ArrayList<>();
        int i = 1;
        while (true){
            String cur = Integer.toString(i);
            if (cur.length() > len) break;
            if (cur.length() == len) power2.add(cur);
            i *= 2;
        }
        for (String string: power2){
            
if (compare(string, s)) return true;
        }
        return false;
    }
    
    public boolean compare(String s1, String s2){
        
int[] arr = new int[10];
        for (char ch: s1.toCharArray()){
            arr[ch - '0']++;
        }
        for (char ch: s2.toCharArray())
{
            arr[ch - '0']--;
            if (arr[ch - '0'] < 0) return false;
        }
        return true;
    }
}
