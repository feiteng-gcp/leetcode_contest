class Solution {
    private boolean flag = false;
    
public boolean reorderedPowerOf2(int N) {
        if (isPowerOfTwo(N)) return true;
        char[] nums = String.valueOf(N).toCharArray();
        boolean[] used = new boolean[nums.length];
        StringBuilder sb = new StringBuilder();
        dfs(sb, nums, used);
        return flag;
    }
    
    private void dfs(StringBuilder sb, char[] nums, boolean[] used) {
        if (flag) return;
        if (sb.length() == nums.length) {
            if (isPowerOfTwo(Integer.parseInt(sb.toString()))) flag = true
;
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            if (used[i]) continue;
            if (sb.length() == 0 && nums[i] == '0') continue;
            used[i] = true;
            sb.append(nums[i]);
            dfs(sb, nums, used);
            used[i] = false;
            sb.deleteCharAt(sb.length() - 1);
        }
    
}
    
    private boolean isPowerOfTwo(int n) {
        if(n <= 0)  return false;
        while(n > 0){
            if(n == 1)  return true;
            if(n % 2 != 0)  return false;
            n = n >> 1;
        }
        return false;
    }
}
