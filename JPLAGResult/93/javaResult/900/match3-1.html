class Solution {
    public boolean reorderedPowerOf2(int N) {
        Set<String> power2 = new HashSet<>();
        for (int i = 0; i < 31; i++) {
            String num = Integer.toString(1 << i);
            
char[] chars = num.toCharArray();
            Arrays.sort(chars);
            power2.add(new String(chars));
        }
        
        char[] cur = Integer.toString(N).toCharArray();
        Arrays.sort(cur);
        String n = new String(cur);
        return power2.contains(n);
    }
}
