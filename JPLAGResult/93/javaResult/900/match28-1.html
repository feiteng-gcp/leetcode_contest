class Solution {
    public boolean reorderedPowerOf2(int N) {
        String temp = N + "";
        boolean[] visited = new boolean[temp.length()];
        return find(temp.toCharArray(), visited, 0, 0);
    
}
    
    private boolean find(char[] array, boolean[] visited, int current, int length) {
        if (length == array.length) {
            return powerof2(current);
        
}
        for (int i = 0; i < array.length; i++) {
            if (visited[i] || (length == 0 && array[i] == '0')) {
                continue;
            }
            int number = current * 10 + (array[i] - '0');
            visited[i] = true;
            if (find(array, visited, number, length + 1)) {
                return true;
            }
            visited[i] = false;
        
}
        return false;
    }
    
    private boolean powerof2(int n) {
        return (n & (n - 1)) == 0;
    }
    
    
}
