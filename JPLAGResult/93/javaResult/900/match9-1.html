class Solution {
    public boolean reorderedPowerOf2(int N) {
        int[] nums = new int[10];
        int count = 0;
        while(N != 0) {
            int tmp = N % 10;
            nums[tmp]++;
            N = N / 10;
            count++;
        
}
        
        return helper(nums, 0, count, 0);
    }
    
    public boolean helper(int[] nums, int cur, int total, int num) {
        if(cur == total) {
            if((num & (num - 1)) == 0)
                return true;
            return false;
        }
        
        for(int i = 0; i < 10; i++) {
            if(cur == 0 && i == 0)
                continue;
            if(nums[i] > 0) {
                nums[i]--;
                if(helper(nums, cur + 1, total, num * 10 + i))
                    return true
;
                nums[i]++;
            }
        }
        return false;
    }
}
