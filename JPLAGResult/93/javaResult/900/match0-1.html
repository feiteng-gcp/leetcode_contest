class Solution {
    public boolean reorderedPowerOf2(int N) {
        int threshold = 1000000000;
        Map<Integer, int[]> map = new HashMap<>();
        int gap = 1;
        while(gap <= threshold){
            map.put(gap, new int[10]);
            count(map.get(gap), gap);
            gap *= 2;
        }
        int[] nArr = new int[10];
        count(nArr, N);
        for(Map.Entry<Integer, int[]> e : map.entrySet()){
            if(checkEqual(nArr, e.getValue())) return true;
        }
        return false;
    }
    
    private boolean checkEqual(int[] a, int[] b){
        for(int i = 0 ; i < a.length ; i++)
            if(a[i] != b[i]) return false;
        return true;
    }
    
    private void count(int[] arr, int num){
        while(num > 0){
            int mod = num % 10;
            arr[mod]++;
            num /= 10;
        }
    }
}
