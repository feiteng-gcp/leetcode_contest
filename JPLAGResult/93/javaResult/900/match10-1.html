class Solution {
   boolean check(long a, long b) {
        String sa = String.valueOf(a);
        String sb = String.valueOf(b);
        
if (sa.length() != sb.length()) {
            return false;
        }

        char[] ca = sa.toCharArray();
        char[] cb = sb.toCharArray();
        Arrays.sort(ca);
        Arrays.sort(cb);
        for (int i = 0; i < ca.length; i++) {
            if (ca[i] != cb[i]) {
                return false;
            }
        }
        return true;
    }

    public boolean reorderedPowerOf2(int N) {
        List<Long> power2 = new ArrayList<>();

        for (int i = 0; i < 50; i++) {
            long t = 1L << (long) i;
            power2.add(t);
        }

        for (int i = 0; i < power2.size(); i++) {
            if (check(
power2.get(i), N)) {
                //System.out.println("a = " + power2.get(i) + " b = " + N);
                return true;
            }
        }
        return false;

    }
}
