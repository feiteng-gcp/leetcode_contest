class Solution {
    public boolean reorderedPowerOf2(int N) {
        int curr = 1;
        int l = len(N);
        while (len(curr) <= l) {
            if (len(curr) == l) {
                if (isReorder(curr, N)) return true;
            }
            curr *= 2;
        }
        return false;
    }
    
    int len (int n) {
        int r = 0;
        while (n > 0) {
            r++;
            n /= 10;
        }
        return r;
    }
    
    boolean isReorder(int a, int b) {
        int[] aa = new int[10];
        int[] bb = new int[10];
        while (a > 0) {
            aa[a % 10]++;
            a /= 10;
        }
        while (b > 0) {
            bb[b % 10]++;
            b /= 10;
        }
        for (int i = 0; i < 10; i++) {
            if (aa[i] != bb[i]) return false;
        }
        return true;
    }
}
