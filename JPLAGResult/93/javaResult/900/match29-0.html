class Solution {
    public boolean reorderedPowerOf2(int N) {
    	int digitsN[] = new int[10];
    	while (N > 0)
    	{
    		digitsN[N % 10]++;
    		N = N / 10;
    	}
    	
    	int digitsPow[];
    	int pow2 = 1;
    	while (pow2 <= 1000000000)
    	{
    		digitsPow = getPowDigits(pow2);
    		
    		if (isEqual(digitsN, digitsPow))
    			return true;
    		pow2 = pow2 * 2;
    	}
    	
    	return false;
        
    }
    
    private int[] getPowDigits(int pow2) {
    	int digitsPow[] = new int[10];
    	
    	while (pow2 > 0)
    	{
    		digitsPow[pow2 % 10]++;
    		pow2 = pow2 / 10;
    	}    	
		return digitsPow;
	}

	private boolean isEqual(int[] digitsN, int[] digitsPow) {
		for (int i = 0 ; i < 10 ; i++)
		{
			if (digitsN[i] != digitsPow[i])
				return false;
		}
		return true;
	}

}
