class Solution {
    public boolean reorderedPowerOf2(int N) {
        HashSet<Integer> set = new HashSet<>();
        for (int i = 0; i < 31; i++) {
            set.add(1 << i);
        }
        for (int x : set) {
            if (valid(x, N)) {
                return true;
            }
        }
        return false;
    }
    
    boolean valid(int x, int y) {
        int[] a = count(x + "");
        int[] b = count(y + "");
        for (int i = 0; i < 10; i++) {
            if (a[i] != b[i]) {
                return false;
            }
        }
        return true;
    }
    
    int[] count(String s) {
        int[] a = new int[10];
        for (char c : s.toCharArray()) {
            a[c - '0']++;
        }
        return a;
    }
}
