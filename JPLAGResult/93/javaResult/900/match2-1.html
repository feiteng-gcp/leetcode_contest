class Solution {
    static Set<String> powers = new HashSet<>();
    {
        for(int i = 0; i < 32; ++ i) {
            
char[] a = String.valueOf(1 << i).toCharArray();
            Arrays.sort(a);
            powers.add(new String(a));
        }
    }
    public boolean reorderedPowerOf2(int N) {
        char[] a = String.valueOf(N).toCharArray();
        Arrays.sort(a);
        return powers.contains(new String(a));
    }
}
