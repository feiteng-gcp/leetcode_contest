class Solution {
    public boolean reorderedPowerOf2(int N) {
        final int limit = 31;
        int[] powersOf2 = new int[limit];
        int[][] digitCount = new int[limit][10];
        powersOf2[0] = 1;
        digitCount[0][1] = 1;
        for (int i = 1; i < limit; i++) {
            powersOf2[i] = powersOf2[i - 1] * 2;
            String num = Integer.toString(powersOf2[i]);
            for (char e : num.toCharArray()) {
                digitCount[i][e - '0']++;
            }
        }

        String num = Integer.toString(N);
        int[] numDigitCount = new int[10];
        for (char e : num.toCharArray()) {
            numDigitCount[e - '0']++;
        }

        boolean flag;
        for (int i = 0; i < limit; i++) {
            flag = true;
            for (int j = 0; j < 10; j++) {
                if (numDigitCount[j] != digitCount[i][j]) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                return true;
            }
        }
        return false;
    }
}
