class Solution {
    public boolean reorderedPowerOf2(int N) {
        // int[] count = new int[10];
        // int len = 0;
        // while (N > 0) {
        //     int remainder = N % 10;
        //     count[remainder]++;
        //     N /= 10;
        //     len++;
        // }
        // System.out.println(len);
        if (N == 0) return false;
        String num = String.valueOf(N);
        char[] str = num.toCharArray();
        Arrays.sort(str);
        num = new String(str);
        // System.out.println(num);
        int len = num.length();
        for (int i = 0; Math.pow(2, i) < Math.pow(10, len + 1); i++) {
            int val = (int)Math.pow(2, i);
            String curNum = String.valueOf(val);
            char[] curStr = curNum.toCharArray();
            Arrays.sort(curStr);
            curNum = new String(curStr);
            // System.out.println(curNum);
            if (curNum.equals(num)) return true;
        }
        return false;
    }
}
