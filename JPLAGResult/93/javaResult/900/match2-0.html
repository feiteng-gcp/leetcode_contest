class Solution {
   static HashSet<String> set = new HashSet<>();
    static {
       

        for (int i = 0; i < 60; i++) {
            long num = (long)Math.pow(2,i);
            char[] arr = Long.toString(num).toCharArray();
            Arrays.sort(arr);
            set.add(new String(arr));
        }
    }

    public boolean reorderedPowerOf2(int N) {
        char []temp = Integer.toString(N).toCharArray();
        Arrays.sort(temp);
        return set.contains(new String(temp));
        
    }
}
