class Solution {
    public boolean reorderedPowerOf2(int N) {
        char[] chs=(""+N).toCharArray();
        Arrays.sort(chs);
        String ana=new String(chs);
        for(int i=0;i<32;++i) {
            int x=1<<i;
            char[] cht=(""+x).toCharArray();
            Arrays.sort(cht);
            String nt=
new String(cht);
            if(nt.equals(ana)) return true;
        }
        return false;
    }
}
