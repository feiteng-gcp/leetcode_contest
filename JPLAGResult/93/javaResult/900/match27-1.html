class Solution {
    public boolean reorderedPowerOf2(int N) {
        long s = 1;
        int[] dd = new int[10];
        String ttt = ""+N;
        for(int j=0;j<ttt.length();j++){
                dd[ttt.charAt(j)-'0']++;
        }
        
        
        for(int i=0;i<=31;i++){
            long t = s<<i;
            String tt = ""+t;
            int[] d = new int[10];
            for(int j=0;j<tt.length();j++){
                d[tt.charAt(j)-'0']++;
            }
           boolean ok = true;
            
for(int u=0;u<10;u++){
                if(d[u]!=dd[u]){
                    ok = false;
                    break;
                }
            }
           if(ok){
            return true;
           }
        }
        return false;
    }
}
