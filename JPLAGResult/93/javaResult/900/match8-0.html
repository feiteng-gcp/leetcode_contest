import java.util.*;
import java.io.*;

class Solution {
    public boolean reorderedPowerOf2(int N) {
        int[] cnt = new int[10];
        int total = 0;
        while(N > 0) {
            cnt[N%10]++;
            N/=10;
            total++;
        }
        // System.out.println();
        return dfs(cnt, 0, total);
    }
    private boolean dfs(int[] cnt, int sum, int total) {
        if ( total == 0) {
            while (sum % 2 == 0)  sum /= 2;

            if (sum == 1)  return true;
            return false;
        }
        for (int i = 0; i < 10; i++) {
            if (sum == 0 && i == 0)  continue;
            if (cnt[i] == 0)  continue;
          
            cnt[i]--;
            boolean res = dfs(cnt, sum*10+i,total-1);
            if (res)  return true;
            cnt[i]++;            
        }
        
        return false;
    }
}
