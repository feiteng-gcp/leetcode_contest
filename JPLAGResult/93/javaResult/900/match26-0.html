class Solution {
    public boolean reorderedPowerOf2(int N) {
        
        String str = convert1(N);
        for (int i = 0; i < 31; ++i) {
            
            String tmp = convert1(1<<i); 
            if (str.equals(tmp)) return true;
                
        }
        
        return false;
        
    }
    
    private String convert1(int N) {
        
        int[] count = new int[10];
        
        while (N != 0) {
            
            int tmp = N%10;
            count[tmp] += 1;
            N /= 10;
            
        }
        return convert2(count);
        
    }
    
    private String convert2(int[] arr) {
        
        StringBuilder sb = new StringBuilder();
        for (int a : arr) {
            sb.append(String.valueOf(a));
            sb.append('_');
        }
        return sb.toString();
    }
}
