class Solution {
    public boolean reorderedPowerOf2(int N) {
        int[] counter = new int[10];
        while(N > 0){
            int reminder = N % 10;
            counter[reminder] ++;
            N/=10;
        }
        for(int i = 0; i < 32; i++){
            int[] copy = new int[10];
            for(int j = 0; j < 10; j++)
                copy[j] = counter[j];
            if(check(1 << i, copy))
                return true;
        }
        return false;
    }
    private boolean check(int target, int[] counter){
        while(target>0){
            int reminder = target % 10;
            counter[reminder] --;
            target/=10;
        }
        for(int i = 0; i < 10; i++){
            if(counter[i] != 0)
                return false;
        }
        return true;
    }
}
