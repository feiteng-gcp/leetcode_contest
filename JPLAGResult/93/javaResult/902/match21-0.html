class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        int n = stations.length;
        int[][] dp = new int[n+1][n+1];
        for(int i=0;i<=n;i++){
            for(int j=0;j<=n;j++){
                dp[i][j] = -1;
            }
        }
        
        dp[0][0] = startFuel;
        for(int i=0;i<n;i++){
            int cur = i == 0 ? 0 : stations[i-1][0];
            for(int j=0;j<=i;j++){
                if(dp[i][j]>=0){
                    if(dp[i][j] >= stations[i][0]-cur){
                        dp[i+1][j] = Math.max(dp[i+1][j], dp[i][j] + cur - stations[i][0]);
                        dp[i+1][j+1] = Math.max(dp[i+1][j+1], dp[i][j] + cur - stations[i][0] + stations[i][1]);
                    }
                }
            }
        }
        
        for(int i=0;i<=n;i++){
            for(int j=0;j<=n;j++){
                int cur = j == 0 ? 0 : stations[j-1][0];
                if(dp[j][i]>=target-cur){
                    return i;
                }
            }
        }
        
        return -1;
    }
}
