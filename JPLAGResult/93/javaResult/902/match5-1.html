class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        // suppose initially the car can reach at most l
        // then it should stop at the station with most fuel before l
        // use priority queue
        Queue<Integer> pq = new PriorityQueue<Integer>();
        
        int fuel = startFuel;
        int res = 0;
        int ptr = 0;
        int n = stations.length;
        
        while (fuel < target) {
            // find the stations can be reached
            
            while (ptr < n && stations[ptr][0] <= fuel) {
                pq.add(-stations[ptr][1]);
                ++ptr;                        
            }

            //pick the stations with the most fuel
            
if (pq.isEmpty()) return -1;
            fuel += -pq.poll();
            ++res;
            
        }
        
        return res;
    }
}
