class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> b[1] - a[1]);
        int count = 0;
        for(int i = 0 ; i < stations.length ; i++){
            if(stations[i][0] <= startFuel) pq.offer(stations[i]);
            else{
                while(!pq.isEmpty() && startFuel < stations[i][0]){
                    int[] maxStation = pq.poll();
                    count++;
                    startFuel += maxStation[1];
                    if(startFuel > target) return count;
                }
                if(startFuel >= stations[i][0]) pq.offer(stations[i]);
                else return -1;
            }
        }
        while(!pq.isEmpty() && startFuel < target){
            int[] tmp = pq.poll();
            startFuel += tmp[1];
            count++;
        }
        return startFuel >= target ? count : -1;
    }
}
