class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        int len = stations.length;
        int[] position = new int[len + 1];
        int[] amount = new int[len + 1];
        position[0] = 0;
        amount[0] = startFuel;
        for (int i = 0; i < len; i++) {
            int[] p = new int[len + 1];
            int[] a = new int[len + 1];
            for (int j = 0; j <= i; j++) {
                int temp = amount[j] - (stations[i][0] - (i == 0 ? 0 : stations[i - 1][0]));
                if (temp >= 0) {
                    if (temp > a[j]) {
                        a[j] = temp;
                    }
                    if (temp + stations[i][1] > a[j + 1]) {
                        a[j + 1] = temp + stations[i][1];
                        position[j + 1] = stations[i][0];
                    }
                } else {
                    a[j] = -1;
                }
            }
            amount = a;
            position = p;
        }
        for (int i = 0; i < len + 1; i++) {
            if (amount[i] >= target - (len > 0 ? stations[len - 1][0] : 0)) {
                return i;
            }
        }
        return -1;
    }
}
