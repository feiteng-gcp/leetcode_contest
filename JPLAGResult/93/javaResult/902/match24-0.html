class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        if (stations.length == 0) return target <= startFuel ? 0 : -1;
        PriorityQueue<Integer> gas = new PriorityQueue<>((a, b) -> b - a);
        int dist = 0;
        int fuel = startFuel;
        int count = 0;
        for (int[] station : stations) {
            if (dist >= target) {
                return count;
            }
            int consume = station[0] - dist;
            while (fuel < consume) {
                if (gas.isEmpty()) return -1;
                fuel += gas.poll();
                count++;
            }
            gas.offer(station[1]);
            dist = station[0];
            fuel -= consume;
        }
        int leftDist = target - dist;
        while (fuel < leftDist) {
            if (gas.isEmpty()) return -1;
            fuel += gas.poll();
            count++;
        }
        return count;
    }
}
