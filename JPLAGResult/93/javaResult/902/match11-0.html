class Solution {
    public int minRefuelStops(int t, int s, int[][] st) {
        
       
        
        
        
       PriorityQueue<Integer> q = new PriorityQueue();
                                          
        int i=0;
        int r = 0;
        while(s<t){
      
            while(i<st.length&&st[i][0]<=s){
                q.offer(-1*st[i][1]);
                i++;
            }    
            if(s<t&&q.size()==0){
                return -1;
            }
        
            if(s<t&&q.size()>0){
                int nu = q.poll();
                s -= nu;
                r++;
            }
        }
        return r;
        
        
        
    }
}
