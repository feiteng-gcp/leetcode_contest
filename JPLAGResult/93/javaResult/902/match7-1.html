class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        int result = 0;
        int fuel = startFuel;
        PriorityQueue<Integer> que = new PriorityQueue<>(new Comparator<Integer>() {
            @Override
            public int compare(Integer a, Integer b) {
                return b - a;
            }
        });
        if(stations.length == 0) {
            if(target <= startFuel)
                return 0;
            else
                return -1;
        }
        if(startFuel >= target)
            return 0;
        
for(int i = 0; i < stations.length; i++) {
            if(fuel >= stations[i][0]) {
                que.offer(stations[i][1]);
            } else {
                while(fuel < stations[i][0]) {
                    if(que.size() == 0)
                        return -1;
                    fuel = fuel + que.poll();
                    result++;
                
}
                if(fuel >= target)
                    return result;
                que.offer(stations[i][1]);
            }
        }
        while(fuel < target) {
            if(que.size() == 0)
                return -1;
            fuel = fuel + que.poll();
            result++;
        }
        return result;
    }
}
