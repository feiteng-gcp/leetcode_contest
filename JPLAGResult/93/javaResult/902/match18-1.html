class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        int[][] oil = new int[510][510];
        
        int N = stations.length;
        int i,j,k;
        int ans = 10000;
        
        for (i = 0;i < N + 10;i++)
        {
            for (j = 0;j < N + 10;j++) 
            {
                oil[i][j] = -1;   
            }
        }
        
        if (N > 0) {
            if (startFuel >= stations[0][0]) {
                oil[0][0] = startFuel;
                oil[0][1] = startFuel + stations[0][1];
                for (i = 0;i < N-1;i++)
                {
                    for (j = 0;j <= i + 1;j++) 
                    {
                        if (oil[i][j] >= 0) {
                            
if (oil[i][j] >= stations[i+1][0])
                            {
                                if (oil[i][j] > oil[i+1][j]) {oil[i+1][j] = oil[i][j];}
                                if (oil[i][j] + stations[i+1][1] > oil[i+1][j+1]) {oil[i+1][j+1] = oil[i][j] + stations[i+1][1];}
                            }
                        }   
                    }
                }

                i = 0;
                while ((i <= N)&&(oil[N-1][i] < target)) {i++;}
                if (oil[N-1][i] >= target) {
                    ans = i;
                }
                else {
ans = -1;}
            }
            else {ans = -1;}
        }
        else {
            if (startFuel >= target) {ans = 0;}
            else {ans = -1;}
        }
        
        return ans;
    }
}
