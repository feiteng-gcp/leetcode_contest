class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        int ans = 0;
        int curPos = 0;
        int fuelStation=0;
        int fuel = startFuel;
        
        while(curPos <= target) {
            curPos += fuel;
            //System.out.println("\ncurpos = " + curPos);
            if(curPos >= target){
                break;
            }
            
            int maxFuelObtainable = 0;
            int index = -1;
            
            if(fuelStation >= stations.length)break;
            
            for(int i=0; i<stations.length; i++) {
                if(stations[i][0] > curPos)break;
                if(stations[i][1] > maxFuelObtainable) {
                    maxFuelObtainable = stations[i][1];
                    index = i;
                }
            }
            
            if(maxFuelObtainable == 0)break;
            ans++;
            fuel = maxFuelObtainable;
            stations[index][1] = 0;
        }
        if(curPos >= target){
            return ans;
        }else{
            return -1;
        }
    }
}
