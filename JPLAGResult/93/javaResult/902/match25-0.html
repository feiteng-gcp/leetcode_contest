class Solution {
    int[][] A;
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        A=stations;
        Arrays.sort(A,new Comparator<int[]>(){
            public int compare(int[] A,int[] B) {
                return A[0]-B[0];
            }
        });
        boolean[] vst=new boolean[A.length];
        int res=0;
        int r=startFuel;
        while(r<target) {
            int max=-1;
            int id=-1;
            for(int i=0;i<A.length;++i) {
                if(vst[i]) continue;
                if(A[i][0]>r) break;
                if(A[i][1]>max) {
                    max=A[i][1];
                    id=i;
                }
            }
            if(id==-1) return -1;
            vst[id]=true;
            r+=max;
            ++res;
        }
        return res;
    }
}
