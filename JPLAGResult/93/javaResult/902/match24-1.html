class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        int n = stations.length;
        if (n==0)
            return target<=startFuel?0:-1;
        long[][] dp = new long[n+1][n+1];
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= n; j++) {
                dp[i][j] = -1;
            }
        }
        dp[0][0] = startFuel-stations[0][0];
        for (int i = 0; i < n-1; i++) {
            for (int j = i; j >= 0; j--) {
                if (dp[i][j]>=0) {
                    dp[i][j+1] = Math.max(dp[i][j+1], dp[i][j]+stations[i][1]);
                }
            }
            for (int j = 0; j <= i+1; j++) {
                if (dp[i][j]>=0) {
                    dp[i+1][j] = dp[i][j] + stations[i][0] - stations[i+1][0];
                }
                //System.out.println(i+" "+j+" "+dp[i][j]);
            }
        }
        int i = n-1;
        for (int j = i; j >= 0; j--) {
            if (dp[i][j]>=0) {
                dp[i][j+1] = Math.max(dp[i][j+1], dp[i][j]+stations[i][1]);
            }
        }
        
int set = target-stations[n-1][0];
        for (int j = 0; j <= i+1; j++) {
            if (dp[i][j]>=set) {
                return j;
            }
        }
        return -1;
    }
}
