class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        Arrays.sort(stations, (a, b) -> a[0] - b[0]);
        int len = stations.length, res = Integer.MAX_VALUE;
        int[][] dp = new int[len + 1][len + 1];
        for (int i = 0; i <= len; i++) {
            // with stations from 0 to i
            for (int n = 0; n <= i; n++) {
                // stop for n times
                if (n == 0 || i == 0) {
                    dp[i][n] = startFuel;
                } else {
                    // dont stop
                    int t = dp[i - 1][n];
                    // stop
                    if (dp[i - 1][n - 1] >= stations[i - 1][0]) {
                        t = Math.max(t, dp[i - 1][n - 1] + stations[i - 1][1]);
                    }
                    dp[i][n] = t;
                }
                if (dp[i][n] >= target)
                    res = Math.min(res, n);
            }
        }
        return res == Integer.MAX_VALUE ? -1 : res;
    }
}
