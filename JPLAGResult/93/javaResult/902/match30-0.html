class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        if (stations.length == 0) {
            return startFuel >= target ? 0 : -1;
        }
        
        // starts with start position
        int[][][] dp = new int[stations.length + 1][stations.length + 1][2];
        dp[0][0][0] = dp[0][0][1] = startFuel;
        
        // max fuel at station 0 (start) assuming i use j stations (impossible for j >= 1)
        for (int j = 1; j <= stations.length; ++j) {
            dp[0][j][0] = dp[0][j][1] = -1;
        }
        
        // max fuel at station (i-1) assuming i use 0 stations
        for (int i = 1; i <= stations.length; ++i) {
            int[] currStation = stations[i - 1];
            if (startFuel >= currStation[0]) {
                dp[i][0][0] = startFuel - currStation[0];
            } else {
                dp[i][0][0] = dp[i][0][1] = -1;
            }
        }
        
        // for each position i (corresponding to station i - 1)
        for (int i = 1; i < dp.length; ++i) {
            int[] currStation = stations[i - 1];
            int[] prevStation = i - 2 >= 0 ? stations[i - 2] : new int[]{0, startFuel};
            for (int j = 1; j < dp[0].length; ++j) {
                if (j == i) {
                    // use all stations
                    dp[i][j][0] = -1;
                    if (dp[i - 1][j - 1][1] != -1 && dp[i - 1][j - 1][1] + prevStation[0] >= currStation[0]) {
                        dp[i][j][1] = dp[i - 1][j - 1][1] + prevStation[0] - currStation[0] + currStation[1];
                    } else {
                        dp[i][j][1] = -1;
                    }
                    continue;
                }
                if (j > i) {
                    dp[i][j][0] = dp[i][j][1] = -1;
                    continue;
                }
                int maxNo = -1;
                if (dp[i - 1][j][0] != -1 && dp[i - 1][j][0] + prevStation[0] >= currStation[0]) {
                    maxNo = Math.max(maxNo, dp[i - 1][j][0] + prevStation[0] - currStation[0]);
                }
                if (dp[i - 1][j][1] != -1 && dp[i - 1][j][1] + prevStation[0] >= currStation[0]) {
                    maxNo = Math.max(maxNo, dp[i - 1][j][1] + prevStation[0] - currStation[0]);
                }
                int max = -1;
                // can get to prev in first place and can get to me
                if (dp[i - 1][j - 1][0] != -1
                    && dp[i - 1][j - 1][0] + prevStation[0] >= currStation[0]) {
                    max = Math.max(max, dp[i - 1][j - 1][0] + prevStation[0] - currStation[0] + currStation[1]);
                }
                if (dp[i - 1][j - 1][1] != -1
                    && dp[i - 1][j - 1][1] + prevStation[0] >= currStation[0]) {
                    max = Math.max(max, dp[i - 1][j - 1][1] + prevStation[0] - currStation[0] + currStation[1]);
                }
                dp[i][j][0] = maxNo;
                dp[i][j][1] = max;
            }
        }

        int[] lastStation = stations[stations.length - 1];
        for (int j = 0; j <= stations.length; ++j) {
            if (Math.max(dp[stations.length][j][0], dp[stations.length][j][1]) + lastStation[0] >= target) {
                return j;
            }
        }
        return -1;
    }
}
