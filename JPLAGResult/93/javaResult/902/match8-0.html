class Solution {
    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        int n = stations.length;
        int count = 0;
        int s = 0;
        int t = startFuel;
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> b - a);
        
        for (int i = 0; i < n; i++) {
            if (s + t >= stations[i][0]) {
                pq.offer(stations[i][1]);
            } else {
                while (pq.size() > 0) {
                    t  += pq.poll();
                    count++;
                    if (s + t >= stations[i][0]) break;

                }
                
                if (s + t < stations[i][0]) return -1;
                pq.offer(stations[i][1]);
            }
            
            
        }
        if (t >= target) return count;
        
        while (pq.size() > 0) {
            t  += pq.poll();
            count++;
            if (s + t >= target) return count;
        }
                
                return -1;
        
        
        
    }
    
}
