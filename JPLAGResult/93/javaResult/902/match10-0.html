	class Solution {
	    public int minRefuelStops(int target, int startFuel, int[][] stations) {
	        long f = startFuel;
	        Queue<Integer> pq = new PriorityQueue<Integer>();
	        int ret = 0;
	        int p = 0;
	        long d = f;
	        while(true){
		        while(p < stations.length && stations[p][0] <= d){
		        	pq.add(-stations[p][1]);
		        	p++;
		        }
		        if(d >= target)return ret;
		        if(pq.isEmpty())return -1;
		        int val = -pq.poll();
		        d += val;
		        ret++;
	        }
	    }
	}
