class Solution {
     class Node {
        int fuel;
        int index;

        public Node(int fuel, int index) {
            this.fuel = fuel;
            this.index = index;
        }
    }

    public int minRefuelStops(int target, int startFuel, int[][] stations) {
        if (stations.length == 0) {
            return startFuel >= target ? 0 : -1;
        }

        int n = stations.length;

        TreeSet<Node> set = new TreeSet<>((a, b) -> {
            
if (a.fuel != b.fuel) {
                return b.fuel - a.fuel;
            } else {
                return a.index - b.index;
            }
        });

        int currFuel = startFuel;
        int pos = 0;

        int res = 0;
        for (int i = 0; i < n; i++) {
            while (currFuel < stations[i][0] - pos && !set.isEmpty()) {
                Node node = set.pollFirst();
                currFuel += node.fuel;
                set.remove(node);
                res++;
            }

            if (currFuel < stations[i][0] - pos) {
                return -1;
            }
            currFuel -= stations[i][0] - pos;
            pos = stations[i][0];

            //System.out.println("currFuel = " + currFuel + " pos = " + pos);
            set.add(new Node(stations[i][1], i));
        }

        while (currFuel < target - stations[n - 1][0] && !set.isEmpty()) {
            Node node = set.pollFirst();
            currFuel += node.fuel;
            set.remove(node);
            res++;
        }
        if (currFuel < target - stations[n - 1][0]) {
            return -1;
        }
        return res;
    }
}
