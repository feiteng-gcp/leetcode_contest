class Solution {
      public int[] advantageCount(int[] A, int[] B) {
        int n = A.length;
        int[][] numsA = new int[n][2];
        int[][] numsB = new int[n][2];

        for (int i = 0; i < n; i++) {
            numsA[i][0] = A[i];
            numsA[i][1] = i;

            
numsB[i][0] = B[i];
            numsB[i][1] = i;
        }

        Arrays.sort(numsA, (a, b) -> (a[0] - b[0]));
        Arrays.sort(numsB, (a, b) -> (a[0] - b[0]));

        boolean[] used = new boolean[n];
        int[] res = new int[n];
        int i = 0, j = 0;
        for (; i < n && j < n; i++) {
            while (j < n && numsA[j][0] <= numsB[i][0]) {
                j++;
            }
            if (j >= n) {
                break;
            }

            res[numsB[i][1]] = numsA[j][0];
            used[j] = true;
            j++;
        }

        j = 0;
        for (; i < n; i++) {
            while (j < n && used[j]) {
                j++;
            }
            res[numsB[i][1]] = numsA[j][0];
            j++;
        }
        return res;

    }
}
