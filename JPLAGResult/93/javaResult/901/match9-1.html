class Solution {
    public int findNextGreatest(List<Integer> list, int num){
        //keep track of the index of the element which is larger than num, but minimizes the difference
        //also keep track of the smallest number in the list
        
        int min = -1;
        int next = -1;
        
        for(int i = 0; i < list.size(); i++){
            if(list.get(i) > num && (next == -1 || list.get(i) - num < list.get(next) - num)){
                next = i;
            }
            if(min == -1 || list.get(i) < list.get(min)){
                min = i;
            }
        }
        
        return next == -1 ? min : next;
    }
    
public int[] advantageCount(int[] A, int[] B) {
        //for every index in B, find the next highest element in A
        //if there is none, take the lowest element
        
        // List<Integer> list = new ArrayList<>();
        // for(int i : A){
        //     list.add(i);
        // }
        
        TreeMap<Integer, Integer> ts = new TreeMap<>();
        for(int i : A){
            
ts.put(i, ts.getOrDefault(i, 0) + 1);
        }
        int[] output = new int[A.length];
        
        for(int i = 0; i < B.length; i++){
            // int min = findNextGreatest(list, B[i]);
            // output[i] = list.get(min);
            // list.remove(min);
            Integer greater = ts.higherKey(B[i]);
            if(greater != null){
                output[i] = greater;
                int count = ts.get(greater);
                if(count == 1){
                    ts.remove(greater);
                }
                else
{
                    ts.put(greater, count-1);
                }
            }
            else{
                int first = ts.firstKey();
                int count = ts.get(first);
                output[i] = first;
                if(count == 1){
                    ts.remove(first);
                }
                else
{
                    ts.put(first, count-1);
                }
            }
        }
        
        return output;
    }
}
