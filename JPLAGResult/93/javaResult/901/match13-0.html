class Solution {
    
    class Pair {
        int val;
        int index;
        
        Pair(int v, int i) {
            val = v;
            index = i;
        }
    }
    
    public int[] advantageCount(int[] A, int[] B) {
        Pair[] b = new Pair[B.length];
        for (int i = 0; i < B.length; i++) {
            b[i] = new Pair(B[i], i);
        }
        
        Comparator<Pair> comp = new Comparator<Pair>() {
            @Override
            public int compare(Pair p1, Pair p2) {
                if (p1.val < p2.val) {
                    return -1;
                } else if (p1.val > p2.val) {
                    return 1;
                } else {
                    return 0;
                }
            }
        };
        
        Arrays.sort(b, comp);
        Arrays.sort(A);
        int small = 0;
        int big = A.length - 1;
        int[] result = new int[A.length];
        for (int i = B.length - 1; i >= 0; i--) {
            if (A[big] > b[i].val) {
                result[i] = A[big];
                big--;
            } else {
                result[i] = A[small];
                small++;
            }
        }
        int [] rResult = new int[A.length];
        for (int i = 0; i < b.length; i++) {
            rResult[b[i].index] = result[i];
        }
        return rResult;
    }
}
