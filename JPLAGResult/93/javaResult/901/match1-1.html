class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        TreeMap<Integer,Integer> set = new TreeMap<>();
        for(int n : A) {
            set.put(n,set.getOrDefault(n,0)+1);
        
}
        int[] res = new int[A.length];
        for(int i = 0; i < B.length; i++) {
            Integer ceil = set.ceilingKey(B[i]+1);
            if(ceil == null) {
                ceil = set.firstKey();
                res[i] = ceil;
            } else {
                res[i] = ceil;
            
}
            if(set.get(ceil) == 1) set.remove(ceil);
            else set.put(ceil,set.get(ceil)-1);
        }
        return res;
    }
}
