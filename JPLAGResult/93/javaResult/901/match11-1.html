class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        int[] res=new int[A.length];
        Arrays.fill(res,-1);
        
TreeMap<Integer,Integer> mapA=new TreeMap<Integer,Integer>();
        for(int i=0;i<A.length;i++){
            int curr=A[i];
            if(!mapA.containsKey(curr)){
                mapA.put(curr,0);
            }
            mapA.put(curr,mapA.get(curr)+1);
        }
        for(int i=0;i<B.length;i++){
            int curr=B[i];
            Integer ceil=mapA.ceilingKey(curr+1);
            if(ceil!=null){
                int count=mapA.get(ceil);
                res[i]=ceil;
                count--;
                
if(count==0){
                    mapA.remove(ceil);
                }
                else{
                    mapA.put(ceil,count);
                }
            }
        }
        int indexA=0;
        for(int key:mapA.keySet()){
            int count=mapA.get(key);
            
while(count>0){
                while(indexA<A.length&&res[indexA]!=-1){
                    indexA++;
                }
                res[indexA]=key;
                count--;
            }
        }
        return res;
    }
}
