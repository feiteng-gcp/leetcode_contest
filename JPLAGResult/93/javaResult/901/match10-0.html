class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        TreeMap<Integer, Integer> tm = new TreeMap<>();
        for (int i = 0; i < A.length; i++) {
            if (!tm.containsKey(A[i])) {
                tm.put(A[i], 0);
            }
            tm.put(A[i], tm.get(A[i]) + 1);
        }
        int[] ans = new int[A.length];
        for (int i = 0; i < B.length; i++) {
            if (tm.ceilingKey(B[i] + 1) != null) {
                int cur = tm.ceilingKey(B[i] + 1);
                ans[i] = cur;
                tm.put(cur, tm.get(cur) - 1);
                if (tm.get(cur) == 0) {
                    tm.remove(cur);
                }
            } else {
                int cur = tm.firstKey();
                ans[i] = cur;
                tm.put(cur, tm.get(cur) - 1);
                if (tm.get(cur) == 0) {
                    tm.remove(cur);
                }
            }
        }
        return ans;
    }
}
