class Solution {
    public int[] advantageCount(int[] a, int[] b) {
        int n = a.length;
        int[][] c = new int[n][2];
        for(int i=0;i<n;i++){
            c[i][0] = b[i];
            c[i][1] = i;
        }
        
        Arrays.sort(c, new Comparator<int[]>(){
            @Override
            public int compare(int[] x, int[] y){
                return y[0]-x[0];
            }
        });
        
        int[] ans = new int[n];
        
Arrays.sort(a);
        
        int left = 0;
        int right = n-1;
        for(int i=0;i<n;i++){
            if(a[right]>c[i][0]){
                ans[c[i][1]] = a[right--];
            }
            else{
                ans[c[i][1]] = a[left++];
            }
        }
        return ans;
    }
}
