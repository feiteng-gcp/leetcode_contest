class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        TreeMap<Integer,Integer> tm = new TreeMap<>();
        int n = A.length;
        for (int i = 0; i < n; i++) {
            if (!tm.containsKey(A[i]))
                tm.put(A[i],1);
            else
                tm.put(A[i],tm.get(A[i])+1);
        
}
        int[] ans = new int[n];
        for (int i = 0; i < n; i++) {
            Integer key = tm.higherKey(B[i]);
            if (key==null) {
                key = tm.firstKey();
            }
            ans[i] = key;
            if (tm.get(key)==1)
                tm.remove(key);
            else
                
tm.put(key, tm.get(key)-1);
        }
        return ans;
    }
}
