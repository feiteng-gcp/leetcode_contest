class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        int n = A.length;
        TreeMap<Integer, Integer> count = new TreeMap<>();
        for (int i=0; i<n; i++) {
            if (!count.containsKey(A[i])) {
                count.put(A[i], 1);
            } else {
                count.put(A[i], count.get(A[i])+1);
            }
        }
        
        int[] ans = new int[n];
        for (int i=0; i<n; i++) {
            Map.Entry<Integer, Integer> entry = count.higherEntry(B[i]);
            if (entry == null) {
                entry = count.firstEntry();
            }
            ans[i] = entry.getKey();
            int c = entry.getValue() - 1;
            if (c == 0) {
                count.remove(entry.getKey());
            } else {
                count.put(entry.getKey(), c);
            }
        }
        return ans;
    }
}
