class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        TreeMap<Integer, Integer> treeMap = new TreeMap<>();
        for (int i : A) {
            int count = 0;
            if (treeMap.containsKey(i)) {
                count = treeMap.get(i);
            }
            treeMap.put(i, ++count);
        }

        int n = B.length;
        Pair[] pairs = new Pair[n];
        for (int i = 0; i < n; i++) {
            pairs[i] = new Pair(i, B[i]);
        }
        Arrays.sort(pairs);

        int[] ans = new int[n];
        Arrays.fill(ans, -1);
        for (int i = 0; i < n; i++) {
            Integer higherKey = treeMap.higherKey(pairs[i].val);
            if (higherKey != null) {
                ans[pairs[i].i] = higherKey;
                int count = treeMap.get(higherKey) - 1;
                if (count == 0) {
                    treeMap.remove(higherKey);
                } else {
                    treeMap.put(higherKey, count);
                }
            }
        }
        int i = 0;
        for (Map.Entry<Integer, Integer> entry : treeMap.entrySet()) {
            int num = entry.getKey();
            int count = entry.getValue();
            while (count > 0) {
                while (ans[i] != -1) {
                    i++;
                }
                ans[i] = num;
                count--;
            }
        }
        return ans;
    }

    private static class Pair implements Comparable<Pair> {
        int i;
        int val;

        Pair(int i, int val) {
            this.i = i;
            this.val = val;
        }

        @Override
        public int compareTo(Pair o) {
            return this.val - o.val;
        }
    }
}
