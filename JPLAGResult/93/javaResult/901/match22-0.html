class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        int n = A.length;
        Integer[] a = new Integer[n];
        Integer[] b = new Integer[n];
        int[] ans = new int[n];
        for (int i = 0; i < n; i++) {
            a[i] = i;
            b[i] = i;
        }
        Arrays.sort(a, (x, y) -> (A[x] - A[y]));
        Arrays.sort(b, (x, y) -> (B[x] - B[y]));
        int i = n - 1;
        int j = n - 1;
        int k = 0;
        Queue<Integer> q = new LinkedList<>();
        while (i >= 0 && j >= 0) {
            if (A[a[i]] > B[b[j]]) {
                ans[b[j]] = A[a[i]];
                i--;
            } else {
                ans[b[j]] = A[a[k++]];
            }
            j--;
        }
        
        return ans;
    }
}
