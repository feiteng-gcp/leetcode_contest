class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        int n = A.length;
        int[] result = new int[n];
        int[][] nums = new int[n][2];
        for (int i = 0; i < n; i++) {
            nums[i] = new int[]{i, B[i]};
        }
        Arrays.sort(nums, (a, b) -> a[1] - b[1]);
        Arrays.sort(A);
        int left = 0, right = n - 1;
        for (int i = 0; i < n; i++) {
            if (A[i] > nums[left][1]) {
                result[nums[left][0]] = A[i];
                left++;
            } else {
                result[nums[right][0]] = A[i];
                right--;
            }
        }
        return result;
    }
}
