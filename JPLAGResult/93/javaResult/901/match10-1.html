class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int i : A) {
            
map.put(i, map.getOrDefault(i, 0) + 1);
        }
        
        int[] ans = new int[A.length];
        for (int i = 0; i < B.length; ++i) {
            int b = B[i];
            Integer higher = map.higherKey(b);
            if (higher == null) {
                // get lowest number from A
                Integer smallest = map.firstKey();
                ans[i] = smallest;
                if (map.get(smallest) == 1) {
                    map.remove(smallest);
                } else {
                    map.put(smallest, map.get(smallest) - 1);
                }
            } else {
                ans[i] = higher;
                if (map.get(higher) == 1) {
                    map.remove(higher);
                } else {
                    map.put(higher, 
map.get(higher) - 1);
                }
            }
        }
        return ans;
    }
}
