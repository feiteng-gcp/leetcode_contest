class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        Arrays.sort(A);
        int n=A.length;
        int[][] id=new int[n][2];
        for(int i=0;i<n;++i) {
            id[i][0]=B[i];
            id[i][1]=i;
        }
        boolean[] vst=new boolean[n];
        Arrays.sort(id,new Comparator<int[]>(){
            public int compare(int[] A,int[] B) {
                return A[0]-B[0];
            }
        });
        int id1=0;
        int id2=0;
        int[] res=new int[n];
        Arrays.fill(res,-1);
        while(id1<n) {
            if(A[id1]<=id[id2][0]) {
                ++id1;
            } else {
                vst[id1]=true;
                res[id[id2][1]]=A[id1];
                ++id1;
                ++id2;
            }
        }
        int ida=0;
        for(int i=0;i<n;++i) {
            if(res[i]==-1) {
                while(ida<n) {
                    if(vst[ida]) ++ida;
                    else break;
                }
                res[i]=A[ida++];
            }
        }
        return res;
    }
}
