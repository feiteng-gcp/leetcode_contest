import java.util.TreeMap;


class Solution {
	public int[] advantageCount(int[] A, int[] B) {
		TreeMap<Integer, Integer> m = new TreeMap<Integer, Integer>();
		for (int i : A)
			m.put(i, m.getOrDefault(i, 0) + 1);
		for (int i=0; i<B.length; i++) {
			Integer k = m.higherKey(B[i]);
			if (k==null)
				k = m.firstKey();
			A[i] = k;
			if (m.get(k)>1)
				m.put(k, m.get(k)-1);
			else {
				m.remove(k);

			}
		}
		return A;
	}
}
