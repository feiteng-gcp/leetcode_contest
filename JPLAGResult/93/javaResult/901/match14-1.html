class Solution {
    class Node {
        int num;
        int index;
        public Node(int n, int ind) {
            num = n;
            index = ind;
        }
    }
    public int[] advantageCount(int[] A, int[] B) {
        int[] result = new int[A.length];
        
Arrays.sort(A);
        Node[] nodes = new Node[B.length];
        for(int i = 0; i < B.length; i++) {
            nodes[i] = new Node(B[i], i);
        }
        Arrays.sort(nodes, new Comparator<Node>() {
           @Override
            public int compare(Node n1, Node n2) {
                return n1.num - n2.num;
            }
        });
        int a = 0;
        int b = 0;
        while(a < A.length && b < nodes.length) {
            if(A[a] <= nodes[b].num)
                a++;
            else {
                result[nodes[b].index] = A[a];
                b++;
                A[a] = -1;
                a++;
            }
        }
        int x = 0;
        for(int i = 0; i < result.length; i++) {
             
            if(result[i] == 0) {
                while(A[x] == -1) {
                    x++;
                }
                result[i] = A[x];
                A[x] = -1;
            }
            
        }
        return result;
    }
}
