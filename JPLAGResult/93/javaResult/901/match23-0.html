class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        TreeMap<Integer, Integer> m = new TreeMap<>();
        for(int a : A){
            m.put(a, m.getOrDefault(a, 0) + 1);
        }
        int[] res = new int[A.length];
        for(int i = 0; i < B.length; i++){
            int num = B[i];
            Map.Entry<Integer, Integer> s = m.higherEntry(num);
            if(s == null){
                s = m.firstEntry();
            }
            res[i] = s.getKey();
            m.put(s.getKey(), s.getValue()-1);
            if(m.get(s.getKey()) == 0)
                m.remove(s.getKey());
        }
        return res;
    }
}
