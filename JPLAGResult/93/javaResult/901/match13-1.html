class Solution {
    class ElementPos {
        int element;
        int position;
        
        public ElementPos(int element, int position) {
            this.element = element;
            this.position = position;
        }
    }
    
    public int[] advantageCount(int[] A, int[] B) {
        Comparator<ElementPos> comparator = new Comparator<ElementPos>() {
            public int compare(ElementPos a, ElementPos b) {
                return b.element - a.element;
            }
        };
        Arrays.sort(A);
        ElementPos[] array = new ElementPos[B.length];
        for (int i = 0; i < B.length; i++) {
            array[i] = new ElementPos(B[i], i);
        }
        Arrays.sort(array, comparator);
        List<Integer> rest = new ArrayList<Integer>();
        int j = A.length - 1;
        
for (int i = 0; i < A.length; i++) {
            if (A[j] > array[i].element) {
                B[array[i].position] = A[j];
                j--;
            } else {
                rest.add(array[i].position);
            }
        }
        
for (int i : rest) {
            B[i] = A[j--];
        }
        return B;
    }
}
