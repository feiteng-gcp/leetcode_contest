class Solution {
     public int[] advantageCount(int[] A, int[] B) {
        Arrays.sort(A);
        int lo = 0, hi = A.length - 1;
        Map<Integer, Stack<Integer>> map = new HashMap<>();
        for(int i = 0 ; i < B.length ; i++){
            if(!map.containsKey(B[i])) map.put(B[i], new Stack<>());
            map.get(B[i]).push(i);
        }
        List<Map.Entry<Integer,Stack<Integer>>> list = new ArrayList<>(map.entrySet());
        Collections.sort(list, (a, b) -> b.getKey() - a.getKey());
        int cur_b = 0;
        int[] res = new int[A.length];
        while(hi >= lo){
            if(A[hi] <= list.get(cur_b).getKey()){
                res[list.get(cur_b).getValue().pop()] = A[lo];
                lo++;
            }else{
                res[list.get(cur_b).getValue().pop()] = A[hi];
                hi--;
            }
            if(list.get(cur_b).getValue().isEmpty()) cur_b++;
        }
        return res;
    }
}
