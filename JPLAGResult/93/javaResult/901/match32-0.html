	class Solution {
	    public int[] advantageCount(int[] a, int[] b) {
	        int n = a.length;
	        int[][] bi = new int[n][];
	        for(int i = 0;i < n;i++){
	        	bi[i] = new int[]{b[i], i};
	        }
	        Arrays.sort(bi, new Comparator<int[]>() {
				public int compare(int[] a, int[] b) {
					return a[0] - b[0];
				}
			});
	        Arrays.sort(a);
	        int p = 0;
	        int[] ra = new int[n];
	        Arrays.fill(ra, -1);
	        boolean[] used = new boolean[n];
	        for(int i = 0;i < n;i++){
	        	while(p < n && a[p] <= bi[i][0]){
	        		p++;
	        	}
	        	if(p < n){
	        		ra[bi[i][1]] = a[p];
	        		used[p] = true;
	        		p++;
	        	}
	        }
	        int q = 0;
	        for(int i = 0;i < n;i++){
	        	if(!used[i]){
	        		while(q < n && ra[q] != -1)q++;
	        		assert q < n;
	        		ra[q] = a[i];
	        	}
	        }
	        return ra;
	    }
	}
