class Solution {
    public int[] advantageCount(int[] A, int[] B) {
        int l = A.length;
        int[] C = new int[l];
        boolean[] used = new boolean[l];
        int i,found;
        int low,high,mid,x;
        
        sort(A,0,l-1);
        
        for (i = 0;i < l;i++)
        {
            used[i] = false;
        }
        
        for (i = 0;i < l;i++)
        {
            x = B[i];
            low = 0;high = l;
			while (low < high) {
				mid = (low+high)>>1;
				
				if (A[mid] > x) {high = mid;
}
				else {low = mid+1;}
			}
            found = low;
            while ((found < l)&&(used[found])) {found++;}
            if (found < l) {
                used[found] = true;
                C[i] = A[found];
            }
            else {
                found = 0;
                while ((found < l)&&(used[found])) {found++;}
                used[found] = true;
                C[i] = A[found];
            }
        }
            
            
        return C;
    }
    
    public void sort(int[] arr,int low,int high) {
		int tl,tr,tmp,mid;
		tl = low;tr = high;
		
		mid = arr[(tl+tr)>>1];
		while (tl < tr){
			while (arr[tl] < mid) tl++;
			while (arr[tr] > mid) tr--;  
			if (tl <= tr) {
				tmp = arr[tr];
				arr[tr] = arr[tl];
				arr[tl] = tmp;
				tl++;tr--;
			}           
		}
		if (low<tr) {sort(arr, low, tr);}
		if (tl<high) {sort(arr, tl, high);}
    }
}
