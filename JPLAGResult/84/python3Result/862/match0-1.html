class Solution:
    def findReplaceString(self, S, i, s, t):
        """
        :type S: str
        :type indexes: List[int]
        :type sources: List[str]
        :type targets: List[str]
        :rtype: str
        """
        S = list(S)
        
        for n in range(len(i)):
            if S[i[n]:i[n] + len(s[n])] == list(s[n]):
                S[i[n]] = t[n]
                for j in range(1, len(s[n])):
                    S[i[n] + j] = ''
        return ''.join(S)
