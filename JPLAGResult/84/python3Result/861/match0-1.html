class Solution:
    def flipAndInvertImage(self, A):
        """
        :type A: List[List[int]]
        :rtype: List[List[int]]
        """
        m = len(A)
        n = len(A[0])
        B = [[0]*n for _ in range(m)]
        for i in range(m):
            
for j in range(n):
                B[i][n-1-j] = 1 - A[i][j]
        return B
