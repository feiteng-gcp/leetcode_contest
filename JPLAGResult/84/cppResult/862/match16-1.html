class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        vector<int> vgood;
        vector<string> gtarg;
        vector<string> gsourc;
        
        for(int i=0;i<indexes.size();i++)
            if(S.substr(indexes[i],sources[i].size())==sources[i])
            {
                vgood.push_back(indexes[i]);
                gsourc.push_back(sources[i]);
                gtarg.push_back(targets[i]);
            }
        
        string result;
        
        for(int i=0;i<S.size();)
        {
            int indf = -1;
            for(int j=0;j<vgood.size();j++)
                if(vgood[j]==i)
                {
                    indf = j;
                    break;
                }
            

            if(indf==-1)
            {
                result += S[i];
                i++;
            }else{
                result += gtarg[indf];
                i += gsourc[indf].size();
            }
        }
        
        return result;
    }
};
