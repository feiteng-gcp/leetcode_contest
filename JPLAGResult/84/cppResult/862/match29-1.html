class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        string ans;
        int i, j, k, l, n, t;
        string tt;
        l = S.size();
        n = indexes.size();
        for (i = 0; i < n; i++) {
            for (j = i + 1; j < n; j++) {
                if (indexes[i] > indexes[j]) {
                    t = indexes[i]; indexes[i] = indexes[j]; indexes[j] = t; 
                    tt = sources[i]; sources[i] = sources[j]; sources[j] = tt; 
                    tt = targets[i]; targets[i] = targets[j]; targets[j] = tt; 
                }
            }
        }
        j = 0;
        for (i = 0; i < S.size(); i++) {
            //cout << "i : " << i << endl;
            while (j < n && indexes[j] < i) {
                j++;
            }
            if (j < n && indexes[j] == i) {
                //cout << "J : " << j << endl;
                for (k = 0; k < sources[j].size(); k++) {
                    if (i + k < S.size() && S[i + k] == sources[j][k]) {
                        continue;
                    }
                    else {
                        break;
                    }
                }
                if (k == sources[j].size()) {
                    ans += targets[j];

                    i += k - 1;
                }
                else {
                    ans += S[i];
                }
            }
            else {
                ans += S[i];
            }
        }
        return ans;
    }
};
