class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        string res="";
        vector<int> :: iterator it;
        int index=0;
        for(int i=0;i<S.size();i++){
            it=find(indexes.begin(),indexes.end(),i);
            if(it!=indexes.end()) {
                index=distance(indexes.begin(),it);
                string source=sources[index];
                int len=source.size();
                if((S.size()-i>=len) && S.substr(i,len)==source) {
                    res+=targets[index];
                    i=i+len-1;
                }
                else res+=S[i];
                index++;
            }
            else res+=S[i];
        }
        return res;
    }
};
