class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        if (indexes.empty()) {
            return S;
        }
        
        int re[S.size()];
        memset(re, 255, sizeof(re));
        for (size_t i = 0; i < indexes.size(); i++) {
            int index = indexes[i];
            const auto& source = sources[i];
            if (S.substr(index, source.size()) == source) {
                re[index] = i;
            }
        }
        
        string target;

        for (size_t i = 0; i < S.size();) {
            if (re[i] != -1) {
                target.append(targets[re[i]]);
                i += sources[re[i]].size();
            } else {
                target.push_back(S[i]);
                i++;
            }
        }
        
        return target;
    }
};
