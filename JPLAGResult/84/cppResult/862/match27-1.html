class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        vector<int>replaceIndices;
        
        int sz = indexes.size();
        
        for(int i = 0; i < sz; i++){
            int size = sources[i].size();
            if(S.substr(indexes[i], size) == sources[i]) replaceIndices.push_back(i);
        }
        
        //sort(replaceIndices.begin(), replaceIndices.end());
        map<int,int>val2idx;
        for(int i = 0; i < replaceIndices.size(); i++){
            int idx = replaceIndices[i];
            int val = indexes[idx];
            val2idx[val] = idx;
        }
        int diff = 0;
        //int lastIdx = -1;
        for(auto elem: val2idx){
            int idx = elem.second;
            //cout<<"idx = "<<idx<<endl;
            int size = sources[idx].size();
            int currIdx = indexes[idx];
            int start = currIdx + diff;
            
            S.replace(start, size, targets[idx]);
            //cout<<S<<endl;
            diff += targets[idx].size()-size;
        }
        
        return S;
        
    }
};
