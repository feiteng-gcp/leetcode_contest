class Solution {
public:
    bool isMatch(string& S, string& source, int idx)
    {
        if (source.length() > S.length()-idx) return false;
        for (int i = 0; i < source.length(); i++)
        {
            if (S[idx+i] != source[i]) return false;
        }
        return true;
    }
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) 
    {
        int n = indexes.size();

        map<int,int> hash;
        for (int i = 0; i < n; i++)
        {
            hash[indexes[i]] = i;
        }
        string res = "";
        int start = 0;
        for (auto it = hash.begin(); it != hash.end(); it++)
        {
            int index1 = it->first;
            int index2 = it->second;
            if (isMatch(S, sources[index2], index1))
            {
                for (int i = start; i < index1; i++)
                {
                    res.push_back(S[i]);
                }
                res.insert(res.end(), targets[index2].begin(), targets[index2].end());
                start = index1 + sources[index2].length();
            }
        }
        for (int i = start; i < S.length(); i++)
        {
            res.push_back(S[i]);
        } 
        return res;
    }
};
