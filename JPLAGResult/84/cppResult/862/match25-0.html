class Solution {
public:
string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
	vector<int> orders(indexes.size());
	for(int i=0; i<indexes.size(); i++) orders[i] = i;
	sort(orders.begin(), orders.end(), [&](const int &o1, const int &o2) {
		return indexes[o1] < indexes[o2];
	});
	stringstream ss;
	int sidx = 0;
	for(int idx : orders) {
		if(indexes[idx] > sidx) {
			ss << S.substr(sidx, indexes[idx] - sidx);
			sidx = indexes[idx];
		}
		bool match = true;
		for(int i=0; i<sources[idx].length(); i++) {
			if(S[sidx+i] != sources[idx][i]) {
				match = false;
				break;
			}
		}
		if(match) {
			ss << targets[idx];
			sidx += sources[idx].length();
		}
	}
    if(sidx != S.length()) ss << S.substr(sidx, S.length()-sidx);
	return ss.str();
}
};
