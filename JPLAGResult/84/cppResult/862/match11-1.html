class Solution {
public:
  string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {            
    int n = indexes.size();
    vector<int> tmpi(n);
    iota(tmpi.begin(), tmpi.end(), 0);
    sort(tmpi.begin(), tmpi.end(), [&] (auto i, auto j) { return indexes[i] < indexes[j]; });
    string result = "";
    result.reserve(2*n);
    int cur = 0;    
    for (int i = 0; i < n; i++){
      int ix = indexes[tmpi[i]];
      auto& src = sources[tmpi[i]];
      auto& tar = targets[tmpi[i]];
      if (cur < ix) result += S.substr(cur, (ix-cur));

      int m = src.length();
      bool rep = true;
      for (int j = 0; j < m; j++) {
        if (S[ix+j] != src[j]) { rep = false; break; }
      }
      cur = ix;
      if (rep) {
        cur += m;
        result += tar;
      }    
    }
    result += S.substr(cur, (S.length()-cur));
    return result;      
  }
};
