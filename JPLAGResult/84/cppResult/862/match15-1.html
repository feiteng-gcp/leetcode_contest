class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        if(indexes.empty()) 
            return S; 
        int start0 = 0; 
        string result; 
        map<int, pair<string, string> > dict;
        for(int i=0; i<indexes.size(); ++i){
            dict[indexes[i]] = make_pair(sources[i], targets[i]); 
        }
        
        for(auto it=dict.begin(); it != dict.end(); ++it){
            const int start = it->first; 
            const string source = it->second.first;
            const string target = it->second.second;
            if(S.substr(start, source.size()) == source){
                result += S.substr(start0, start-start0);
                start0 = start+source.size(); 
                result += target;
            }
        }
        result += S.substr(start0, S.size() - start0); 
        return result; 
    }
};
