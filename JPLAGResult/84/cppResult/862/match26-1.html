class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        if(indexes.empty() || sources.empty() || targets.empty()) return S;
        if(indexes.size() != sources.size() || indexes.size() != targets.size()) return S;
        

        map<int, pair<string, string>> rules;
        for(int i = 0; i < indexes.size(); i++){
            rules[indexes[i]] = make_pair(sources[i], targets[i]);
        }
        
        string result;// = S.substr(0, (*rules.begin()).first);
        int prev = 0;
        
        for(auto rule : rules){ 
            for(int i = prev; i < rule.first; i++,prev++){
                result.push_back(S[i]);
            }
            //prev = max(rule.first - 1, 0);
            int check_size = 0;
            string tmp;
            for(int i = 0; i < rule.second.first.size(); i++){
                if(S[rule.first + i] != rule.second.first[i]) break;    
                check_size ++;
                //tmp.push_back(S[rule.first+i]);
            }
            //cout<<rule.first<<",tmp:"<<tmp<<endl;
            //replace
            if(check_size == rule.second.first.size()){
                result += rule.second.second;
                prev   += rule.second.first.size();
            }
           // cout<<prev<<endl;
//            result += tmp;
  //          prev += rule.second.first.size();
        }
        for(int i = prev; i < S.size(); i++) result.push_back(S[i]);
        return result;
    }
};
