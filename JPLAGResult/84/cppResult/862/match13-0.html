int a[1010];
vector<int> rrr;
int cmp(int i, int j) {
    return rrr[i] < rrr[j];
}
class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        int n = indexes.size();
        rrr.clear();
        for (int i = 0; i < n; i++) {
            a[i] = i;
            rrr.push_back(indexes[i]);
        }
        
        sort(a, a + n, cmp);
        int tmp = 0;
        int tmpp = 0;
        int m = S.length();
        string res = "";
        for (int i = 0; i < m; i++) {
            tmp = a[tmpp];
            while (tmpp < n && indexes[tmp] < i) tmpp++, tmp = a[tmpp];
            tmp = a[tmpp];
            if (tmp < n && indexes[tmp] == i) {
                int cc = sources[tmp].length();
                int fg = 0;
                for (int j = 0; j < cc; j++) {
                    if (S[i + j] != sources[tmp][j]) {
                        fg = 1;
                        break;
                    }
                }
                if (fg) {
                    res += S[i];
                } else {
                    cc = targets[tmp].size();
                    for (int j = 0; j < cc; j++) {
                        res += targets[tmp][j];
                    }
                    i += sources[tmp].length() - 1;
                }
                
            } else res += S[i];
        }
        return res;
    }
};
