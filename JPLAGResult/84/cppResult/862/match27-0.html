class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        vector<pair<int, pair<string, string>>> inp;
        int n = indexes.size();
        for(int i=0;i<n;i++){
            inp.push_back({indexes[i],{sources[i], targets[i]}});
        }
        sort(inp.begin(), inp.end());
        int n2 = S.size(), j = 0;
        string res;
        for(int i=0;i<n2;i++){
            if(j < n){
                if(inp[j].first > i) res.push_back(S[i]);
                else{
                    bool found = false;
                    while(j < n && inp[j].first == i){
                        string t = inp[j].second.first;
                        if(S.substr(i, t.size()) == t) { found = true; res += inp[j].second.second; i += (int)t.size() - 1;}
                        j++;
                    }
                    if(!found) res.push_back(S[i]);
                }
            }else res.push_back(S[i]);
        }
        return res;
    }
};
