class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        string ret;
        
        if (indexes.size()==0) return S;
        
        map<int,int> iiMap;
        for(int i=0;i<indexes.size();++i) {
            iiMap[indexes[i]] = i;
        }
        
        auto iIter=iiMap.begin();
        
        for(int i=0;i<S.size();) {
            if (iIter!=iiMap.end() && i == indexes[iIter->second]) {
                //start Check

                string ori,src=sources[iIter->second],tar=targets[iIter->second]; ++iIter;
                
                bool flag = true;
                for(int j=0;j<src.size();++j) {
                    if (S[i]!=src[j]) {
                        flag = false;
                    }   
                    ori.push_back(S[i]);
                    ++i;
                }       
                
                if (flag) {
                    ret+=tar;
                } else {
                    ret+=ori;
                }
                
            } else {
                ret.push_back(S[i]);
                ++i;
            }
        }
        
        return ret;
    }
};
