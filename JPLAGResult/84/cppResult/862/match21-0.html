class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        int m = sources.size();
        string ret;
        vector<int> p(m);
        for (int i = 0; i < m; ++i) p[i] = i;
        sort(p.begin(), p.end(), [&](int x, int y) {
            return indexes[x] < indexes[y];
        });
        int cur = 0;
        for (int i = 0; i < m; ++i) {
            int k = p[i];
            int pos = indexes[k];
            ret += S.substr(cur, pos - cur);
            int len = sources[k].size();
            if (S.substr(pos, len) == sources[k]) {
                ret += targets[k];
            } else {
                ret += S.substr(pos, len);
            }
            cur = pos + len;
            //cout << "cur=" << cur << endl;
        }
        ret += S.substr(cur);
        return ret;
        
    }
};
