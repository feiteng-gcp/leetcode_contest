class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        map<int, pair<string, string>> m;
        for (int i = 0; i < indexes.size(); i++)
        {
            m[indexes[i]] = {sources[i], targets[i]};
        }
        string res;
        auto it = m.begin();
        for (int i = 0; i < S.length(); i++)
        {
            if (it != m.end())
            {
                if (i < it->first)
                {
                    res.push_back(S[i]);
                }
                else
                {
                    string source = (it->second).first;
                    if ( source == S.substr(i, source.length()))
                    {
                        res += (it->second).second;
                    }
                    else
                    {
                        res += S.substr(i, source.length());
                    }
                    it++;
                    i += source.length() -1;
                }
            }
            else
            {
                res += S.substr(i);
                break;
            }
        }
        return res;
    }
};
