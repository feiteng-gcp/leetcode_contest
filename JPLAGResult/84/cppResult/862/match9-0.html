struct node{
    int ind;
    string src,target;
    node(int ind,string src, string target):ind(ind),src(src),target(target){};
    friend bool operator < (const node n1,const node n2){
        return n1.ind < n2.ind;
    }
};
vector<node> v;
class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        v.clear();
        for(int i=0;i<indexes.size();i++){
            v.push_back(node(indexes[i],sources[i],targets[i]));
        }
        sort(v.begin(),v.end());
        string ans="";
        int start=0;
        for(int i=0;i<v.size();i++){
            if(v[i].ind>start){
                ans=ans+S.substr(start,v[i].ind-start);
                start = v[i].ind;
            }
            bool flag=true;
            int len = v[i].src.length();
            for(int j=0;j<len;j++){
                if(v[i].src[j]!=S[v[i].ind+j]){
                    flag=false;
                    break;
                }
            }
            if(flag){
                ans=ans+v[i].target;
                start = v[i].ind+len;
            }
        }
        ans = ans+S.substr(start, S.length()-start);
        return ans;
    }
};
