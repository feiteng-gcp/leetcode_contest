class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        int n = indexes.size();
        vector<pair<int, pair<string, string>>> reps;
        for (int i = 0; i < n; i++)
        {
            pair<int, pair<string, string>> tmp = make_pair(indexes[i], make_pair(sources[i], targets[i]));
            reps.push_back(tmp);
        }
        sort(reps.begin(), reps.end());
        
        string res, source, target;
        int index, k = 0, len = S.size();
        for (auto rep: reps)
        {
            index = rep.first;
            source = rep.second.first;
            target = rep.second.second;
            if (index > k)
            {
                res += S.substr(k, index - k);
                k = index;
            }
            else if (index < k || index + source.size() > len) continue;
            int i = index;
            for (char c: source)
            {
                if (c != S[i]) break;
                i++;
            }
            if (i == index + source.size())
            {
                res += target;
                k += source.size();
            }
        }
        if (k < len) res += S.substr(k, len - k);
        return res;
    }
};
