class Solution {
public:
    string findReplaceString(string s, vector<int>& idx, vector<string>& src, vector<string>& trg) {
        if(s.size()==0)
        {
            return s;
        }
        
        int m = idx.size();
        
        string result;
        map<int, int> ops;
        for(int i =0; i<m; i++)
        {
            ops[idx[i]] = i;
        }
        
        int pre = 0;
        for(auto & kv : ops)
        {
            int index = kv.first;
            int opi = kv.second;
            
            if(s.substr(index, src[opi].size()) == src[opi])
            {
                result += s.substr(pre, index - pre);
                result += trg[opi];
                
                index += src[opi].size();
                pre = index;
            }
        }
        
        if(pre != s.size())
        {
             result += s.substr(pre, s.size() - pre);
        }
        
        return result;
    }
};
