class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        int pos[S.length()];
        for (int i = 0; i < indexes.size(); i++) {
            pos[indexes[i]] = i;
        }
        sort(indexes.begin(), indexes.end());
        int index = 0;
        string ans = "";
        for (int i = 0; i < indexes.size(); i++) {
            ans += S.substr(index, indexes[i] - index);

            index = indexes[i];
            bool match = true;
            for (int j = 0; j < sources[pos[index]].length(); j++) {
                if (index + j >= S.length() || S[index + j] != sources[pos[index]][j]) {
                    match = false;
                    break;
                }
            }
            if (match) {
                ans+= targets[pos[index]];
                index += sources[pos[index]].length();
            }
        }
        ans += S.substr(index, S.length() - index);
        return ans;
    }
};
