class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        string result;
        vector<int> isValid(indexes.size(), 0);
        for (int i = 0; i < indexes.size(); i++) {
            int start = indexes[i];
            bool isMatch = true;
            for (int j = start; j < start + sources[i].length(); j++) {
                //cout << i << " " << j << " " << S[j] << " " << sources[i][j - start] << endl;
                if (S[j] != sources[i][j - start]) {
                    isMatch = false;
                    break;
                }
            }
            if (isMatch) {
                //result += targets[i];
                isValid[i] = 1;
                //cout << "valid! " << sources[i] << endl;
            }
            
        }
        
        for (int i = 0; i < indexes.size(); i++) {
            for (int j = i + 1; j < indexes.size(); j++) {
                if (indexes[i] > indexes[j]) {
                    swap(indexes[i], indexes[j]);
                    swap(sources[i], sources[j]);
                    swap(targets[i], targets[j]);
                    swap(isValid[i], isValid[j]);
                }
            }
        }
        
        int index = isValid.size();
        for (int i = 0; i < isValid.size(); i++) {
            if (isValid[i]) {
                index = i;
                break;
            }
        }
        for (int i = 0; i < S.length(); i++) {
            //cout << i << endl;
            if (index < indexes.size() && i == indexes[index] && isValid[index]) {
                //cout << "hi" << endl;
                result += targets[index];
                i += (sources[index].length() - 1);
                int tmp = index;
                index = isValid.size();
                for (int j = tmp + 1; j < isValid.size(); j++) {
                    if (isValid[j]) {
                        index = j;
                        break;
                    }
                }
            } else {
                //cout << "hi2" << endl;
                result += S[i];
            }
        }
        return result;
    }
};
