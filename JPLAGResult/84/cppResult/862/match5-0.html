class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        map<int, pair<string,string>> re;
        for(int i = 0; i < indexes.size(); i++){
            re[indexes[i]] = pair<string,string>(sources[i],targets[i]);
        }
        string rets="";
        int last = -1;
        for(auto i = re.begin(); i != re.end(); i ++){
            int index = i->first;
            string source = i->second.first;
            string target = i->second.second;
            cout<<index<<endl;
            bool flag = true;
            for(int j = 0; j < source.size(); j++){
                if(S[j + index] != source[j]){
                    flag = false;
                    break;
                }
            }
            if(flag){
                rets += S.substr(last+1, index-last-1);
                rets += target;
                last = index + source.size()-1;
            }
        }
        rets += S.substr(last + 1, S.size());
        return rets;
    }
};
