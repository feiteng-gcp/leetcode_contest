class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        vector<int> g(S.length());
        fill(g.begin(),g.end(),-1);
        for(int i=0;i<indexes.size();++i){
            int p = indexes[i];
            bool check = true;
            for(int j=0;j<sources[i].length();++j){
                if(S[p+j] != sources[i][j]){
                    check = false;
                }
            }
            if(check){
                g[p] = i;
            }
        }
        string ret;
        int p=0;
        while(p<S.size()){
            if(g[p]<0){
                ret.append(1,S[p]);
                ++p;
            }
            else{
                ret.append(targets[g[p]]);
                p+= sources[g[p]].length();
            }
        }
        return ret;
    }
};
