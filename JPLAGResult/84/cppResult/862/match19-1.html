class Solution {
public:
    struct rule{
        int pos;
        string s, t;
        rule(int i, string x, string y): pos(i), s(x), t(y) {}
    };
    
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        map<int, pair<string, string>> mp;
        for(int i = 0; i < indexes.size(); i++) mp[indexes[i]] = make_pair(sources[i], targets[i]);
        string ans;
        int i = 0, j = 0, n = S.size();
        while(i < n){
            if(mp.find(i) != mp.end()){

                string same = mp[i].first;
                if(i+same.size() <= n && S.substr(i, same.size()) == same){
                    ans += mp[i].second;
                    i += same.size();
                }
                else ans += S[i++];
            }
            else ans += S[i++];
        }
        return ans;
    }
};
