class Solution {
public:
    string findReplaceString(string S, vector<int>& indexes, vector<string>& sources, vector<string>& targets) {
        if(indexes.empty())
            return S;
        int len=S.size();
        string res;
        unordered_map<int, pair<string, string>> replacement;
        for(int i=0;i<indexes.size();++i)
            replacement[indexes[i]]=make_pair(sources[i], targets[i]);
        for(int i=0;i<len;){
            if(replacement.find(i)!=replacement.end() && S.substr(i, replacement[i].first.size())==replacement[i].first){
                res+=replacement[i].second;
                i+=replacement[i].first.size();
            }
            else{
                res+=S[i];
                ++i;
            }
        }
        return res;
    }
};
