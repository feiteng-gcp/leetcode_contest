class Solution {
public:
    #define MAX 10005
    int n;
    int nodes[MAX];
    vector<int> adj[MAX];
    int solve(int v, int p, int dist) {
        int result = dist;
        nodes[v] = 1;
        for( int i = 0 ; i < adj[v].size(); ++i ){
            int u = adj[v][i];
            if( u != p ){
                result += solve(u, v, dist + 1);
                nodes[v] += nodes[u];
            } 
        }        
        return result;
    }
    
    void dfs(int v, int p, int dist, vector<int> &result) {
        result[v] = dist;
        
        for( int i = 0 ; i < adj[v].size(); ++i ){
            int u = adj[v][i];
            if( u != p ){
            	dfs(u, v, dist - ( 2 * nodes[u] - n) ,result);
            }
        }
    }
    
    vector<int> sumOfDistancesInTree(int N, vector<vector<int>>& edges) {
        n = N;
        vector<int> result(n, 0);
        for( int i = 0 ; i <= n ; ++i ) adj[i].clear();
        for( int i = 0 ; i < edges.size(); ++i ){
            adj[ edges[i][0]].push_back( edges[i][1]);
            adj[ edges[i][1]].push_back( edges[i][0]);
        }
        int total = solve(0, 0, 0);
        
        dfs(0, 0, total, result);
        return result;
    }
};
