class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {
        //flip
        int rows = A.size();
        int cols = A[0].size();

        int idx = (cols+1)/2;
        
        for(int i = 0; i < rows; i++){
            for(int j = 0; j < idx; j++){
                int temp = A[i][j];
                A[i][j] = A[i][cols-1-j];
                A[i][cols-1-j] = temp;
            }
        }
        
        //invert
        for(int i = 0; i < rows; i++){
            for(int j = 0; j < cols; j++){
                A[i][j] = 1 - A[i][j];
            }
        }
        
        return A;
    }
};
