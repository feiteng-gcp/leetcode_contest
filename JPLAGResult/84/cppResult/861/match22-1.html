class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {

        int m = A.size(), n = A[0].size();
        for(int i = 0; i < m; i++){
            int l = 0, r = n-1;
            while(l < r){
                swap(A[i][l], A[i][r]);
                A[i][l] = 1 - A[i][l];
                A[i][r] = 1 - A[i][r];
                l++; r--;
            }
            if(l == r) A[i][l] = 1 - A[i][l];
        }
        return A;
    }
};
