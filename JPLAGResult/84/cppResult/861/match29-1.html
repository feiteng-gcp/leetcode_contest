class Solution {
public:
vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {
    for(int i=0; i<A.size(); i++) {
        for(int j=0, k=A[i].size()-1; j<=k; j++,k--) {
            swap(A[i][j], A[i][k]);
            A[i][j] ^= 1;
            if(j != k) A[i][k] ^= 1;
        }
    }
    return A;
}
};
