class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {
        vector<vector<int>> ret;
        for(int i=0;i<A.size();++i) {
            vector<int> tmp;
            for(int j=A[0].size()-1;j>=0;--j) {
                tmp.push_back(A[i][j]==1?0:1);
            }
            ret.push_back(tmp);
        }
        return ret;
    }
};
