class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {
        for(int i=0;i<A.size();i++){
            int len = A[i].size();
            for(int j=0;j<len/2;j++){
                int tmp = A[i][j];
                A[i][j] = A[i][len-j-1];
                A[i][len-j-1] = tmp;
            }
        }
        for(int i=0;i<A.size();i++){
            for(int j=0;j<A[i].size();j++){
                A[i][j]=1-A[i][j];
            }
        }
        return A;
    }
};
