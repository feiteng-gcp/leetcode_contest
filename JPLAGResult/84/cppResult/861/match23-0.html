class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {
        vector<vector<int>> result = A;
        if(A.empty()) return result;
        int m = result.size(), n = result[0].size();
        //flip
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n/2; j++){
                int tmp = result[i][j];
                result[i][j] = result[i][m - j - 1];
                result[i][m-j-1] = tmp;
            }
        }
        
        //invert
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++ ){
                result[i][j] ^= 1;
            }
        }
        return result;
    }
};
