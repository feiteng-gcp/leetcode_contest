class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {
        for (int i = 0; i < A.size(); i++)
        {
            int l = 0, r = A[i].size() -1;
            while (l <= r)
            {
                swap(A[i][l], A[i][r]);

                l++;
                r--;
            }
        }
        for (int i = 0; i < A.size(); i++)
        {
            for (int j = 0; j <A[i].size(); j++)
            {
                A[i][j] = 1 - A[i][j];
            }
        }
        return A;
    }
};
