class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {
        vector<vector<int>> vvi = A;

        int len = A.size();
        for(int i = 0;i < len;i++) {
            for(int j = 0;j < len / 2;j++) {
                swap(vvi[i][j], vvi[i][len - j - 1]);
            }
        }
        
        for(int i = 0;i < len;i++) {
            for(int j = 0;j < len;j++) {
                vvi[i][j] = 1 - vvi[i][j];
            }
        }
        
        return vvi;
    }
};
