class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {
        int rowNum = A.size(), colNum = A[0].size();
        for (int row = 0; row < rowNum; ++ row) {
            reverse(A[row].begin(), A[row].end()); 
            for (int col = 0; col < colNum; ++ col) {
                A[row][col] = 1 - A[row][col]; 
            }
        }
        return A; 
    }
};
