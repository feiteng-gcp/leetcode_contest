class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {
        int m = A.size(), n = A[0].size();

        for (int i = 0; i < m; ++i) {
            reverse(A[i].begin(), A[i].end());
        }
        
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                A[i][j] = A[i][j]==1 ? 0 : 1;
            }
        }
        
        return A;
    }
};
