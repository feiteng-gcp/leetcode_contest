class Solution {
public:
    vector<vector<int>> flipAndInvertImage(vector<vector<int>>& A) {
        if(A.empty() || A[0].empty()) return A;
        int n = A.size(), m = A[0].size();
        // for(int i=0;i<n;i++){
        //     for(int j=0;j<m;j++) cout << A[i][j] << ' '; cout << endl;
        // }
        for(int i=0;i<n;i++){
            int j1 = 0, j2 = m-1;
            while(j1 < j2){
                swap(A[i][j1], A[i][j2]);
                A[i][j1] = 1 - A[i][j1];
                A[i][j2] = 1 - A[i][j2];
                j1++; j2--;
            }
            if(j1 == j2) A[i][j1] = 1 - A[i][j1];
        }
        return A;
    }
};
