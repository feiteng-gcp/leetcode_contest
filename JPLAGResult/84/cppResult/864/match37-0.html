class Solution {
public:
    int getOverLap(vector<vector<int>>& A, vector<vector<int>>& B, int rowoffset, int coloffset) {
        int rows = A.size();
        int cols = A[0].size();
        int overlap = 0;
        for (int i = 0; i + rowoffset < rows; i++) {
            for (int j = 0; j + coloffset < cols; j++) {
                overlap += A[(i+rowoffset)][(j + coloffset)] * B[i][j];
            }
        }
        return overlap;
    }
    
    int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {
        if (A.size() == 0) {
            return 0;
        }
        if (A[0].size() == 0) {
            return 0;
        }
        int rows = A.size();
        int cols = A[0].size();
        int maxoverlaps, onesA=0, onesB=0;
        for (auto row : A) {
            for (auto ent : row) {
                onesA += ent;
            }
        }
        maxoverlaps = onesA;
        for (auto row : B) {
            for (auto ent : row) {
                onesB += ent;
            }
        }
        if (onesB < maxoverlaps) {
            maxoverlaps = onesB;
        }
        int overlap=0, overlapcurr;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                overlapcurr = getOverLap(A, B, i, j);
                //cout << i << " " << j << " " << overlapcurr << endl;
                if (overlapcurr > overlap) {
                    overlap = overlapcurr;
                }
                if (overlap == maxoverlaps) {
                    break;
                }
                overlapcurr = getOverLap(B, A, i, j);
                //cout << i << " " << j << " " << overlapcurr << endl;
                if (overlapcurr > overlap) {
                    overlap = overlapcurr;
                }
                if (overlap == maxoverlaps) {
                    break;
                }
            }
            if (overlap == maxoverlaps) {
                    break;
            }
        }
        return overlap;
    }
};
