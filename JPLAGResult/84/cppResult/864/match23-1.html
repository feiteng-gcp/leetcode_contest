int overlap(vector<vector<int>> &A, vector<vector<int>> &C, int a, int b, int N)
{

    int ans = 0;
    for(int i=0; i<N; ++i) {
        for(int j=0; j<N; ++j) {
            if(A[i][j] & C[a+i][b+j])
                ++ans;
        }
    }
    
    return ans;
}

class Solution {
public:
    int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {
        int N = A.size();
        vector<vector<int>> C(3*N);

        for(int i=0; i<3*N; ++i)
            C[i].resize(3*N, 0);
        
        for(int i=0; i<N; ++i) {
            for(int j=0; j<N; ++j)
                C[N+i][N+j] = B[i][j];
        }
        

        int ans = 0;
        for(int a=1; a<2*N; ++a) {
            for(int b=1; b<2*N; ++b) {
                int x = overlap(A, C, a, b, N);
                if(x > ans)
                    ans = x;
            }
        }
        
        return ans;
    }
};
