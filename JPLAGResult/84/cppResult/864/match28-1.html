class Solution {
public:
    int countOverlap(vector<vector<int>>& A, vector<vector<int>>& B, int x, int y)
    {
        int n = A.size();
        int left, right, up, down;
        if (x < 0) {left = 0; right = n + x;}
        else {left = x; right = n;}
        if (y < 0) {up = 0; down = n + y;}
        else {up = y; down = n;}

        int cnt = 0;
        for (int i = left; i < right; i++)
        {
            for (int j = up; j < down; j++)
            {
                cnt += (A[i-x][j-y] & B[i][j]);
            }
        }
        return cnt;
    }
    
    int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) 
    {
        int n = A.size();
        int cnt1 = 0, cnt2 = 0;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                cnt1 += A[i][j];
                cnt2 += B[i][j];
            }
        }
        int cnt_max = max(cnt1, cnt2);

        int cnt = 0;
        for (int x = 1-n; x <= n-1; x++)
        {
            for (int y = 1-n; y <= n-1; y++)
            {
                cnt = max(cnt, countOverlap(A, B, x, y));
                if (cnt == cnt_max) return cnt;
            }
        }
        return cnt;
    }
};
