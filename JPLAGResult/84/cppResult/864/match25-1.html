class Solution {
public:
    int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {
        int ret=0;
        
        int is = A.size();
        if (is==0) return 0;
        int js = A[0].size();
        if (js==0) return 0;

        int bound, Aones = 0, Bones =  0;
        
        
        for (int i=0;i<is;++i){
            for (int j=0;j<js;++j) {
                if (A[i][j]==1) ++Aones;
                if (B[i][j]==1) ++Bones;                
            }
        }

        bound = min(Aones,Bones);
        
        for(int io = 0; io < is; ++io) {
            for (int jo = 0; jo < js; ++jo) {
                int curBound = (is-io)*(js-jo);
                if (ret >= curBound) break;
                
                int curAreaUL = 0, curAreaDR = 0;
                int curAreaDL = 0, curAreaUR = 0;

                for(int i=0;i< (is-io); ++i) {
                    for (int j=0;j< (js-jo); ++j) {
                        if (A[i][j]==1&&B[i+io][j+jo]==1)++curAreaUL;
                        if (B[i][j]==1&&A[i+io][j+jo]==1)++curAreaDR;
                        if (A[i+io][j]==1&&B[i][j+jo]==1)++curAreaDL;
                        if (B[i+io][j]==1&&A[i][j+jo]==1)++curAreaUR;
                    }
                }
                if (curAreaUL>ret) ret = curAreaUL;
                if (curAreaDR>ret) ret = curAreaDR;
                if (curAreaDL>ret) ret = curAreaDL;
                if (curAreaUR>ret) ret = curAreaUR;
            }
        }
        
        
        return ret;
    }
};
