class Solution {
public:
int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {

	int n = A.size();
	int ret = 0;
	for(int i=1-n; i<n; i++) {
		for(int j=1-n; j<n; j++) {
			int overlap = 0;
			for(int ii=max(0, -i); ii<min(n, n-i); ii++) {
				for(int jj=max(0, -j); jj<min(n, n-j); jj++) {
					if(A[ii][jj] == 1 && B[ii+i][jj+j] == 1) overlap++;
				}
			}
			ret = max(ret, overlap);
		}
	}
	return ret;
}
};
