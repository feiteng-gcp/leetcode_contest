class Solution {
public:
    int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {
        int res=0;
        if(A.empty()||A[0].empty()) return 0;
        int row=A.size(),col=A[0].size();
        for(int i=0;i<row;i++){
            for(int j=0;j<col;j++){
                int temp1=0,temp2=0;
                for(int m=0;i+m<row;m++){
                    for(int n=0;j+n<col;n++){
                        if(A[m][n]==1&&B[i+m][j+n]==1) temp1++;
                        if(A[i+m][j+n]==1&&B[m][n]==1) temp2++;
                    }
                }
                int temp=max(temp1,temp2);
                res=max(res,temp);
            }
        }
        return res;
    }
};
