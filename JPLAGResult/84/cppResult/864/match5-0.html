class Solution {
public:
    int largestOverlap(const vector<vector<int> >& A, const vector<vector<int> >& B) {
        
        const int n = A.size();
        int max_count = 0;
        for(int move_right = 1-n; move_right < n; move_right++){
            const int A_c0 = max(-move_right, 0);
            const int A_c1 = min(n-move_right, n);
            const int B_c0 = max(move_right, 0);
            const int B_c1 = min(n+move_right, n);
            for(int move_down = 1-n; move_down < n; move_down ++){
                const int A_r0 = max(-move_down, 0);
                const int A_r1 = min(n-move_down, n);
                const int B_r0 = max(move_down, 0);
                const int B_r1 = min(n+move_down, n);
                int count = 0;
                for(int r=A_r0; r<A_r1; r++){
                    for(int c=A_c0; c<A_c1; c++){
                        if(A[r][c]==1 && B[r-A_r0+B_r0][c-A_c0+B_c0] == 1){
                            count ++;
                        }
                    }
                }
                max_count = max(max_count, count);
            }
        }
        return max_count;
    }
};
