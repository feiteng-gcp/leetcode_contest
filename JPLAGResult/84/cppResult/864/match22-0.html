class Solution {
public:
    int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {
        int n = A.size();
        int res = 0;
        for (int h = 1 - n; h < n; h++)
        {
            for (int v = 1 - n; v < n; v++)
            {
                int cnt = 0;
                for (int i = 0; i < n; i++)
                {
                    if (i + h < 0 || i + h >= n) continue;
                    for (int j = 0; j < n; j++)
                    {
                        if (j + v < 0 || j + v >= n) continue;
                        cnt += (A[i+h][j+v] == 1 && B[i][j] == 1) ? 1 : 0;
                    }
                }
                res = max(res, cnt);
            }
        }
        return res;
    }
};
