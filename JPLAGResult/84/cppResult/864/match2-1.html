class Solution {
public:
    int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {

        int res = 0;
        int m = A.size();
        for (int i = -(m-1); i <= m-1; ++i) {
            for (int j = -(m-1); j <= m-1; ++j) {
                
                int tmp = 0;
                for (int ii = 0; ii < m; ++ii) {
                    for (int jj = 0; jj < m; ++jj) {
                        int ni = ii + i;
                        int nj = jj + j;
                        
                        if (ni >= 0 && ni < m && nj >= 0 && nj < m && A[ii][jj] == 1 && A[ii][jj] == B[ni][nj]) {
                            ++tmp;
                        }
                    }
                }
                res = max(res, tmp);
            }
        }
        return res;
    }
};
