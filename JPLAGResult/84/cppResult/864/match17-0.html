class Solution {
  int n;
  int Check(vector<vector<int>>& A, vector<vector<int>>& B, int p, int q) {
    int ans = 0;
    for (int i = 0 ; i < n; i++) {
      for (int j = 0; j <n; j++) {
        int ii = i+p, jj = j+q;
        if (ii >= 0 and ii < n and jj >= 0 and jj < n and A[ii][jj] == 1
            and B[i][j] == 1) ans++;
      }
    }
    return ans;
  }
public:

  int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {
    n = B.size();
    int result = 0;
    for (int i = -n+1; i < n; i++) {
      for (int j = -n+1; j < n; j++) {
        result = max(result, Check(A, B, i, j));
      }
    }
    return result;
  }
};
