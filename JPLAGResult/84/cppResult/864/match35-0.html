class Solution {
public:
    int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {
        int result = 0;
        int n = A.size();
        for (int i = -1 * n + 1; i < n; i++) {
            for (int j = -1 * n + 1; j < n; j++) {
                result = max(result, overlap(A, B, i, j));
            }
        }
        return result;
    }
    
    int overlap(vector<vector<int>>& A, vector<vector<int>>& B, int right, int down) {
        int n = A.size();
        int result = 0;
        //cout << right << " " << down << endl;
        if (right >= 0 && down >= 0) {
            for (int i = 0; i < n - down; i++) {
                for (int j = 0; j < n - right; j++) {
                    if (A[i][j] == 1 && B[i + down][j + right] == 1) {
                        result++;
                    }
                }
            }
        } else if (right >= 0 && down < 0) {
            for (int i = 0; i < n + down; i++) {
                for (int j = 0; j < n - right; j++) {
                    if (A[i - down][j + right] == 1 && B[i][j] == 1) {
                        result++;
                    }
                }
            }
        } else if (right < 0 && down >=0) {
            for (int i = 0; i < n - down; i++) {
                for (int j = 0; j < n + right; j++) {
                    if (A[i + down][j - right] == 1 && B[i][j] == 1) {
                        result++;
                    }
                }
            }
        } else if (right < 0 && down < 0) {
            for (int i = 0; i < n + down; i++) {
                for (int j = 0; j < n + right; j++) {
                    if (A[i - down][j - right] == 1 && B[i][j] == 1) {
                        result++;
                    }
                }
            }
        }
        return result;
    }
};
