class Solution {
public:
    int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {
        int n = A.size();
        vector<vector<int>> a = A;
        vector<vector<int>> b = B;
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int tmp = 0;
                for (int k = 0; k < n; k++) {
                    for (int o = 0; o < n; o++) {
                        if (i + k < n && j + o < n)
                        tmp += b[i + k][j + o] & a[k][o];
                    }
                }
                ans = max(ans, tmp);
            }
        }
        a = B;
        b = A;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int tmp = 0;
                for (int k = 0; k < n; k++) {
                    for (int o = 0; o < n; o++) {
                        if (i + k < n && j + o < n)
                        tmp += b[i + k][j + o] & a[k][o];
                    }
                }
                ans = max(ans, tmp);
            }
        }
        return ans;
    }
};
