class Solution {
public:
    int largestOverlap(vector<vector<int>>& A, vector<vector<int>>& B) {

        int ans = 0, m = A.size(), n = B.size();
        for(int r = 0; r < m; r++){
            for(int c = 0; c < n; c++){

                int count1 = 0, count2 = 0;
                for(int i = 0; i + r < m; i++){
                    for(int j = 0; j + c < n; j++){
                        if(A[i+r][j+c] && B[i][j]) count1++;
                        if(A[i][j] && B[i+r][j+c]) count2++;
                    }
                }
                ans = max(ans, max(count1, count2));
            }
        }
        return ans;
    }
};
