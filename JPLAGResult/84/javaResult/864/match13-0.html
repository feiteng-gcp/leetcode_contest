class Solution {
    public int largestOverlap(int[][] A, int[][] B) {
        int max = 0;
        for (int i = 0; i < 2 * (A.length); i++) {
            for (int j = 0; j < 2 * (A[0].length ); j++) {
                max = Math.max(max, calcuteCover(A, B, i, j));
            }
        }
        return max;
    }

    public int calcuteCover(int[][] A, int[][] B, int x, int y) {
        int x0 = -(A.length - 1) + x;
        int y0 = -(A[0].length - 1) + y;
        int count = 0;
        for (int i = 0; i<A.length; i++) {
            if(!(x0 + i >= 0 && x0 + i < A.length))continue;
            for (int j = 0; j<A[0].length; j++) {
                if(!(y0 + j >= 0 && y0 + j < A[0].length)) continue;
                if (A[x0 + i][y0 + j] == B[i][j] && B[i][j] == 1) count++;
            }
        }
        return count;
    }

}
