class Solution {
    public int largestOverlap(int[][] A, int[][] B) {
        int max = 0;
        
        for (int offsetx = 1 - A.length ; offsetx < A.length; offsetx ++ ){
            for (int offsety = 1 - A.length ; offsety < A.length; offsety ++ ){
                int num = numOverlap(shift(A, offsetx, offsety), B);                
                max = Math.max(max, num);
            }
        }
        return max;
    }
    
    public int[][] shift(int[][] m, int x, int y){
        int[][] res = new int[m.length][m[0].length];
        for (int i=0; i<m.length; i++){
            for (int j=0; j<m[0].length; j++){
                if (i+x >=0 && i+x <m.length && j+y>=0 && j+y<m.length){
                    res[i][j] = m[i+x][j+y];
                
}
            }
        }
        return res;
    }
    
    private int numOverlap(int[][] A, int[][] B){
        int total = 0;
        for (int i=0; i<A.length; i++){
            for (int j=0; j<A[0].length; j++){
                if (A[i][j] == 1 && B[i][j]==1)
                    total++;
            }
        }
        return total;
    }
}
