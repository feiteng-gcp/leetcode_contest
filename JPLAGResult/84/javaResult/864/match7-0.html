class Solution {
    public int largestOverlap(int[][] A, int[][] B) {
        int max = 0;
        int row = A.length;
        if (row == 0) return 0;
        int col = A[0].length;
        if (col ==0) return 0;
        int tmp[][] = new int [row*3][col*3];
        //pre
        for (int i = row; i < row*2; i++) {
            for (int j= col; j < col*2; j++) {
                tmp[i][j] = A[i-row][j-col];
            }
        }
        
        //count
        for (int i = 0; i < row*2; i++) {
            for (int j=0; j < col*2; j++) {
                int dx = i;
                int dy = j;
                int cnt =0;
                for (int bi = 0; bi < row; bi++) {
                    for (int bj = 0; bj < col; bj++) {
                        if (tmp[dx+bi][dy+bj] == 1 && B[bi][bj] == 1){
                            cnt++;
                        }
                    }
                }
                max = Math.max(max, cnt);
            }
        }
        return max;
    }
}
