class Solution {
    public int largestOverlap(int[][] A, int[][] B) {
        
int max = 0;
        int len = A.length;
        for(int i = 1 - len; i < len; ++i){
            for(int j = 1 - len; j < len; ++j){
                int cnt = 0;
                for(int m = 0; m < len; ++m){
                    for(int n = 0; n < len; ++n){
                        int mi = m +i;
                        if(mi < 0 || mi >= len){
                            continue;
                        }
                        
                        int nj = n + j;
                        if(nj < 0 || nj >= len){
                            continue;
                        }
                        
                        if(B[m][n] == 1 && A[mi][nj] == B[m][n]){
                            ++cnt;
                        }
                    }
                }
                if(max < cnt){
                    max = cnt;
                }
            }
        }
        return max;
    }
}
