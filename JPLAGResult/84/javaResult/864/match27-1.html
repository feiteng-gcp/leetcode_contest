class Solution {
    public int largestOverlap(int[][] A, int[][] B) {
        if (A.length == 0 || A[0].length == 0 || B.length == 0 || B[0].length == 0)
            return 0;
        List<Integer> listA = new ArrayList<>();
        List<Integer> listB = new ArrayList<>();
        for (int[] array : A)
            listA.add(convertToBinary(array));
        for (int[] array : B)
            listB.add(convertToBinary(array));
        
int m = A.length, n = A[0].length;
        int mask = (1 << n) - 1;
        int max = 0;
        for (int i = -m + 1; i < m; i++) {
            for (int j = -n + 1; j < n; j++) {
                int count = 0;
                for (int k = 0; k < listA.size(); k++) {
                    if (k + i < 0 || k + i >= m)
                        continue;
                    int numA = listA.get(k + i);
                    int numB = listB.get(k);
                    if (numA == 0 || numB == 0)
                        continue;
                    if (j < 0) 
                        count += Integer.bitCount((numA << (-j)) & mask & numB);
                    else
                        count += Integer.bitCount((numA >> j) & mask & numB);
                
}
                max = Math.max(max, count);
            }
        }
        return max;
    }
    
    public int convertToBinary(int[] array) {
        int num = 0;
        for (int i = 0; i < array.length; i++)
            if (array[i] == 1)
                num |= 1 << i;
        return num;
    }
}
