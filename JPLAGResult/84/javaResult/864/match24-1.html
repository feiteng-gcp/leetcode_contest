class Solution {
    public int largestOverlap(int[][] A, int[][] B) {
        if(A.length == 0 || A[0].length == 0) return 0;
        
int res = 0;
        int m = A.length, n = A[0].length;
        for(int x = -m; x<=m; x++){
            for(int y = -n; y<=n; y++){
                int cnt = 0;
                for(int r = 0; r<A.length; r++){
                    cnt += cnt(getInt(A, r, x, y), getInt(B, r, 0, 0));
                
}
                res = Integer.max(cnt, res);
            }
        }
        return res;
    }
    private int cnt(int a, int b){
        int res = 0;
        while(a>0){
            if(a%2 == 1 && b%2 == 1) res++;
            a = a>>1;
            b = b>>1;
        }
        return res;
    }
    private int getInt(int[][] M, int row, int x, int y){
        int nrow = row-x;
        if(nrow>=0 && nrow<M.length){
            int nc = -y;
            int res = 0;
            for(int j = 0; j<M[0].length; j++) {
                res = res<<1;
                if(nc+j >=0 && nc+j <M[0].length) res |= M[nrow][nc+j];
            }
            return res;
        } 
        return 0;
    }
}
