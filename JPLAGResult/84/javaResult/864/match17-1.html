class Solution {
    public int largestOverlap(int[][] A, int[][] B) {
        // bitwise operation
        int N = A.length;

        // special case
        if (N == 1) {
            if (A[0][0] == 1 && B[0][0] == 1) {
                return 1;
            }
            return 0;
        }

        // encode into a, b
        int[] a = new int[N];
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                a[i] |= A[i][j] == 1 ? (1 << (N - j - 1)) : 0;
            }
        }
        int[] b = new int[N];
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                b[i] |= B[i][j] == 1 ? (1 << (N - j - 1)) : 0;
            }
        }

        int maxResult = 0;

        // brute force
        // i is displacement of rows
        for (int i = -(N - 1); i < (N - 1); i++) {
            // j is in columns
            for (int j = -(N - 1); j < (N - 1); j++) {
                int curResult = 0;
                
for (int aIndex = 0; aIndex < N; aIndex++) {
                    int bIndex = aIndex + i;
                    if (bIndex < 0 || bIndex >= N) {
                        continue;
                    }
                    int andResult = 0;
                    if (j >= 0) {
                        andResult = (a[aIndex] << j) & b[bIndex];
                    }
                    else {
                        andResult = (a[aIndex] >> (-j)) & b[bIndex];
                    }
                    curResult += countOnes(andResult);
                }
                maxResult = Math.max(maxResult, curResult);
            }
        }

        return maxResult;
    }

    private int countOnes(int num) {
        int count = 0;
        for (int i = 0; i < 32; i++) {
            if (((1 << i) & num) > 0) {
                count++;
            }
        }
        return count;
    }
}
