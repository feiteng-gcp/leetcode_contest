class Solution {
    public int largestOverlap(int[][] A, int[][] B) {
        return Math.max(helper(A, B), helper(B,A));
    }
    private int helper(int[][] A, int[][] B) {
        int n = A.length;
        int res = 0;
        for(int i = -n+1; i <= 0; i++) {
            for(int j = -n+1; j <= 0; j++) {
                int curr = 0;
                for(int k = 0; k < n+i; k++) {
                    for(int l = 0; l < n+j; l++) {
                        if(B[k][l] == 1 && A[k-i][l-j] == 1) {
                            curr++;
                        }
                    }
                }
                res = Math.max(res, curr);
            }
        }
        return res;
    }
}
