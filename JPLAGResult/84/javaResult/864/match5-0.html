class Solution {
    int max;
    public int largestOverlap(int[][] A, int[][] B) {
        max = 0;
        if (A.length == 0 || B.length == 0) {
            return 0;
        }
        List<Integer> xB = new ArrayList<>(), yB = new ArrayList<>();
        for (int i = 0; i < B.length; i++) {
            for (int j = 0; j < B[0].length; j++) {
                if (B[i][j] == 1) {
                    xB.add(i);
                    yB.add(j);
                }
            }
        }
        
        Map<List<Integer>, Integer> countByPos = new HashMap<>();

        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[0].length; j++) {
                if (A[i][j] == 1) {
                    updateCountByPos(i, j, xB, yB, countByPos);
                }
            }
        }   
        return max;
    }
    
    private void updateCountByPos(int x, int y, List<Integer> xB, List<Integer> yB, Map<List<Integer>, Integer> countByPos) {
        for (int i = 0; i < xB.size(); i++) {
            int bX = xB.get(i), bY = yB.get(i);
            int xDiff = x - bX, yDiff = y - bY;
            List<Integer> key = Arrays.asList(xDiff, yDiff);
            if (!countByPos.containsKey(key)) {
                countByPos.put(key, 0);
            }
            int count = countByPos.get(key) + 1;
            countByPos.put(key, count);
            max = Math.max(max, count);
        }
    }
}
