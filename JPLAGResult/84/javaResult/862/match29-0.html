import java.util.Arrays;

class Solution {
    
    class Data implements Comparable<Data> {
        
        int index;
        String source;
        String target;
        
        public Data(int index, String source, String target) {
            this.index = index;
            this.source = source;
            this.target = target;
        }
        
        public int compareTo(Data data) {
            if (this.index < data.index) return -1;
            if (this.index > data.index) return 1;
            return 0;
        }
    }
    
    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {

        int n = indexes.length;
        Data[] dataPoints = new Data[n];
        for (int i = 0; i < n; i++) {
            Data data = new Data(indexes[i],sources[i],targets[i]);
            dataPoints[i] = data;
        }
        Arrays.sort(dataPoints);
        String[] resultPieces = new String[n];
        String result = "";
        int last = 0;
        for (int i = 0; i < n; i++) {
        
            int location = dataPoints[i].index;
            String sourceString = dataPoints[i].source;
            String targetString = dataPoints[i].target;
            if ((location + sourceString.length()) <= S.length()) {
                if (S.substring(location, location + sourceString.length()).equals(dataPoints[i].source)) {
                    result += S.substring(last,location) + targetString;
                    last = location + sourceString.length();
                }                
            }
        }
        if (last < S.length()) result += S.substring(last);
        return result;
    }
}
