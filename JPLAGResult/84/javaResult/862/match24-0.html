class Solution {
    class Item {
        int index;
        String source;
        String target;

        public Item(int index, String source, String target) {
            this.index = index;
            this.source = source;
            this.target = target;
        }
    }

    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {
        if (indexes.length == 0) {
            return S;
        }

        Item[] items = new Item[indexes.length];
        for (int i = 0; i < indexes.length; i++) {
            items[i] = new Item(indexes[i], sources[i], targets[i]);
        }
        Comparator<Item> comparator = (a, b) -> a.index - b.index;
        Arrays.sort(items, comparator);
        StringBuilder sb = new StringBuilder();
        int pre = 0;
        for (int i = 0; i < items.length; i++) {
            Item item = items[i];
            int index = item.index;
            String source = item.source;
            String target = item.target;
            int length = source.length();
            if (index + length > S.length()) {
                continue;
            }
            String str = S.substring(index, index + length);
            if (!str.equals(source)) {
                continue;
            }
            sb.append(S.substring(pre, index));
            sb.append(target);
            pre = index + length;
        }

        if (pre < S.length()) {
            sb.append(S.substring(pre));
        }
        return sb.toString();
    }
}
