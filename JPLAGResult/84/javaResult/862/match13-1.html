class Solution {
	// range, nums, find exists sum in range
	// ex1 : 1-10  [2,3,4,5] : true
	// ex2 : 7-7 [2,4,8] : false
	
	    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {
	        StringBuilder sb = new StringBuilder();
	        
	        Map<Integer, Integer> map =new HashMap<>();
	        for(int i = 0 ; i < indexes.length ; i++){
	        	map.put(indexes[i], i);
	        }

	        
	        for(int i = 0 ; i < S.length() ;){
	            if(map.containsKey(i)){
	            	int pi = map.get(i);
	                String target = targets[pi];
	                String src = sources[pi];
	                pi++;
	                int tar_l = src.length();
	                if( i + tar_l <= S.length() && S.substring(i, i + tar_l).equals(src) ){
	                    
	                    sb.append(target);
	                    i += tar_l;
	                    continue;
	                }
	            }
	            sb.append(S.charAt(i));
	            i++;
	        }
	        return sb.toString();
	    }
	    
}
