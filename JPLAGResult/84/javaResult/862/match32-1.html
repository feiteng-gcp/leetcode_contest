class Solution {
    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {
        int n = indexes.length;
        PriorityQueue<Operation> ops = new PriorityQueue<Operation>(n, (Operation op1, Operation op2) -> op1.index - op2.index);
        
for(int i = 0; i < n; i++){
            ops.offer(new Operation(indexes[i], sources[i], targets[i]));
        }
        StringBuilder builder = new StringBuilder();
        int last_pos = 0;
        while(ops.size() != 0){
            Operation op = ops.poll();
            int len = op.source.length();
            if(S.substring(op.index, op.index+len).equals(op.source)){
                builder.append(S.substring(last_pos, op.index));
                builder.append(op.target);
                last_pos = op.index + len;
            }
        }
        builder.append(S.substring(last_pos));
        return builder.toString();
    }
}


class Operation{
    int index;
    String source;
    String target;
    
    public Operation(int index, String source, String target){
        this.index = index;
        this.source = source;
        this.target = target;
    }
}
