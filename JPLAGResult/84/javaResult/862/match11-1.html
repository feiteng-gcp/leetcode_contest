class Solution {
    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {
        PriorityQueue<Element> pq = new PriorityQueue<>(
new Comparator<Element>(){
            public int compare (Element a, Element b) {
                return a.index - b.index;
            }
        });
        for (int i = 0; i < indexes.length; i++) {
            pq.add(new Element(indexes[i], sources[i], targets[i]));
        }
        StringBuilder result = new StringBuilder();
        int position = 0;
        while (!pq.isEmpty()) {
            Element cur = pq.poll();
            String tmp = S.substring(cur.index, S.length());
            if (position < cur.index) {
                result.append(S.substring(position, cur.index));
                position = cur.index;
            }
            if (tmp.indexOf(cur.source) == 0) {
                // swap
                result.append(cur.target);
                position = cur.index + cur.source.length();
            }
        }
        result.append(S.substring(position, S.length()));
        return result.toString();
    }
    
    
public static class Element {
        int index;
        String source;
        String target;
        public Element(int index, String source, String target) {
            this.index = index;
            this.source = source;
            this.target = target;
        }
    }
}
