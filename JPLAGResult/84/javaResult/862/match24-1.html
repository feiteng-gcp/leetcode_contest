class Solution {
    
    class C {
        public int i;
        public String s; 
        public String t;
        
        C(int i, String s, String t){
            this.i = i;
            this.s = s;
            this.t = t;
        }
    }
    
     public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {        
        int offset = 0; 
        String res = S;
        
C[] objs = new C[indexes.length];
        
        for (int i=0; i<indexes.length; i++){
            objs[i] = new C(indexes[i], sources[i], targets[i]);
        }
        
        Arrays.sort(objs, (o1, o2)-> o1.i - o2.i);
        
        for (int i=0; i<indexes.length; i++){
            int index = objs[i].i + offset;
            if (res.substring(index).startsWith(objs[i].s)){
                res = res.substring(0, index) + objs[i].t + res.substring(index+objs[i].s.length());
                offset += objs[i].t.length() - objs[i].s.length();
            }            
        }    
        
        return res;
    }
}
