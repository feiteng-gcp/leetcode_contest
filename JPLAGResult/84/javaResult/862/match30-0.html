import java.util.PriorityQueue;

class Solution {
    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {
        PriorityQueue<Triple> pq = new PriorityQueue<>();
        for (int i = 0; i < indexes.length; i++) {
            pq.add(new Triple(indexes[i], sources[i], targets[i]));
        }
        StringBuffer sb = new StringBuffer();
        int pos = 0;
        while (pos < S.length() && !pq.isEmpty()) {
            if (pos < pq.peek().index) {
                sb.append(S.charAt(pos));
                pos++;
            }else if (pos == pq.peek().index){
                Triple triple = pq.poll();
                if (pos + triple.src.length() <= S.length() && S.substring(pos, pos + triple.src.length()).equals(triple.src)) {
                    sb.append(triple.tar);
                    pos+= triple.src.length();
                }else {
                    sb.append(S.charAt(pos));
                    pos++;
                }
            }else {
                pq.poll();
                sb.append(S.charAt(pos));
                pos++;
            }
        }
        if (pos < S.length()) sb.append(S.substring(pos));
        return sb.toString();
    }

    class Triple implements Comparable<Triple> {
        int index;
        String src, tar;
        public Triple(int index, String src, String tar) {
            this.index = index;
            this.src = src;
            this.tar = tar;
        }

        @Override
        public int compareTo(Triple o) {
            if (this == o) return 0;
            return this.index - o.index;
        }
    }
}
