class Solution {
    class Pair {
        int index;
        String source;
        String target;
        public Pair(int index, String source, String target) {
            this.index = index;
            this.source = source;
            this.target = target;
        }
    }
    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {
        if(indexes.length == 0)
            return S;
        List<Pair> l = new ArrayList<>();
        for(int i = 0; i < indexes.length; i++) {
            int pos = indexes[i];
            String source = sources[i];
            String target = targets[i];
            l.add(new Pair(pos, source, target));
        
}
        
        Collections.sort(l, new Comparator<Pair>(){
            public int compare(Pair a, Pair b) {
                return a.index - b.index;
            }
        });
        
        StringBuilder sb = new StringBuilder();
        sb.append(S.substring(0, l.get(0).index));
        
        for(int i = 0; i < indexes.length; i++) {
            int pos = l.get(i).index;
            String source = l.get(i).source;
            String target = l.get(i).target;
            if(S.substring(pos).startsWith(source)){
                sb.append(target);
                if(i < indexes.length-1)
                    sb.append(S.substring(pos+source.length(), l.get(i+1).index));
                else
                    sb.append(S.substring(pos+source.length()));
            } else {
                if(i < indexes.length-1)
                    sb.append(S.substring(pos, l.get(i+1).index));
                else
                    sb.append(S.substring(pos));
            }
        }
        return sb.toString();
    }
}
