class Solution {
    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {
        StringBuilder sb = new StringBuilder();
        char[] c = S.toCharArray();
        String[] s = new String[1001];
        for (int i = 0 ; i< indexes.length ; i++){
            System.out.println(S.substring(indexes[i],indexes[i]+sources[i].length()));
            if (S.substring(indexes[i],indexes[i]+sources[i].length()).equals(sources[i])){
                s[indexes[i]] = targets[i];
                for (int j=1;j<sources[i].length();j++)s[indexes[i]+j]="";
            }
        }
        
        for (int i = 0 ; i<S.length(); i++ ){
            if(s[i]!=null)sb.append(s[i]);
            else sb.append(S.charAt(i));
        }
        
        return sb.toString();
    }
}
