class Solution {
    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {
        List<Pair> pairs = new ArrayList<>();
        for (int i = 0; i < indexes.length; i++) pairs.add(new Pair(indexes[i], sources[i], targets[i]));
        Collections.sort(pairs, new Comparator<Pair>(){
            public int compare(Pair p1, Pair p2) {return p1.index - p2.index;}});
        StringBuilder sbr = new StringBuilder();
        int start = 0;
        for (Pair p : pairs) {
            if (start < p.index) {
                sbr.append(S.substring(start, p.index));
                start = p.index;
            }
            if (S.substring(p.index).startsWith(p.src)) {
                sbr.append(p.dest);
                start = p.index + p.src.length();
            }
        }
        if (start < S.length()) sbr.append(S.substring(start));
        return sbr.toString();
    }
}

class Pair {
    int index;
    String src, dest;
    public Pair(int index, String src, String dest) {
        this.index = index; this.src = src; this.dest = dest;
    }
}
