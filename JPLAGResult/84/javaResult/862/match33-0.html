class Solution {
    class Interval implements Comparable<Interval> {
        int start, end;
        String val;
        public Interval(int start, int end, String val) {
            this.start = start;
            this.end = end;
            this.val = val;
        }
        
        @Override
        public int compareTo(Interval that) {
            return this.start - that.start;
        }
    }
    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {
        List<Interval> list = new ArrayList<>();
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < indexes.length; i++) {
            if (sources[i].length() > 0 && S.substring(indexes[i], indexes[i] + sources[i].length()).equals(sources[i])) {
                list.add(new Interval(indexes[i], indexes[i] + sources[i].length() - 1, targets[i]));
            }
        }
        Collections.sort(list);
        int prev = 0;
        for (Interval inter : list) {
            sb.append(S.substring(prev, inter.start));
            sb.append(inter.val);
            prev = inter.end + 1;
        }
        sb.append(S.substring(prev, S.length()));
        return sb.toString();
    }
}
