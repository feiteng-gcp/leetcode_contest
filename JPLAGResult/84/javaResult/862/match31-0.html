class Solution {
    public String findReplaceString(String S, int[] indexes, String[] sources, String[] targets) {
        StringBuilder stringBuilder = new StringBuilder();
        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            int ind = find(i, indexes);
            if (ind >= 0) {
                if (eq(S, i, sources[ind])) {
                    stringBuilder.append(targets[ind]);
                    i += sources[ind].length() - 1;
                } else {
                    stringBuilder.append(c);
                }
            } else {
                stringBuilder.append(c);
            }
        }
        return stringBuilder.toString();
    }

    private int find(int val, int[]index) {
        for (int i = 0; i < index.length; i++) {
            if (index[i] == val) {
                return i;
            }
        }
        return -1;
    }

    private boolean eq(String origin, int ind, String target) {
        for (int i = 0; i < target.length(); i++) {
            char ct = target.charAt(i);
            if (ct != origin.charAt(ind++)) {
                return false;
            }
        }
        return true;
    }
}
