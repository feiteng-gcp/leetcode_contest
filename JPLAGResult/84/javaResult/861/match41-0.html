class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        if (A.length == 0) {
            return A;
        }
        for (int i = 0; i < A.length; i++) {
            int j = 0;
            int k = A[i].length - 1; 
            while (j <= k) {
                swap(A[i], j, k);
                j++;
                k--;
            }
        }
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[i].length; j++) {
                A[i][j] = (A[i][j] == 1 ? 0 : 1);
            }
        }
        return A;
    }
    
    private void swap(int[] A, int i, int j) {
        int tem = A[i];
        A[i] = A[j];
        A[j] = tem; 
    }
}
