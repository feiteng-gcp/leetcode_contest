class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int[][] res = new int[A.length][A[0].length];
        
        for(int i = 0; i < A.length; i++){
            for(int j = 0; j < A[0].length; j++){  
                res[i][j] = A[i][A[0].length - 1 - j];
            }
        }
        for(int i = 0; i < A.length; i++){
            for(int j = 0; j < A.length; j++){
                res[i][j] ^= 1;
            }
        }
        return res;
    }
}
