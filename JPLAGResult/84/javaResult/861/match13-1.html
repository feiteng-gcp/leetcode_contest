class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int n = A.length;
        
int[][] result = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                result[i][j] = 1 - A[i][n - 1 - j];
            }
        }
        return result;
    }
}
