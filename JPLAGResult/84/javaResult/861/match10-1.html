class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        if (A.length == 0) return new int[0][0];
        
int n = A[0].length;
        int[][] result = new int[A.length][n];
        for (int r = 0; r < A.length; r++) {
            for (int c = 0; c < n; c++) {
                result[r][c] = 1 - A[r][n - 1 - c];
            }
        }
        return result;
    }
}
