class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        for (int i = 0; i < A.length; ++i) {
            int left = 0, right = A[i].length-1;
            while (left < right) {
                int tmp = A[i][left];
                A[i][left] = 1 - A[i][right];
                A[i][right] = 1 - tmp; 
                left++;
                right--;
            
}
            
            if (left == right) {
                A[i][left] = 1-A[i][left];
            }
        }
        
        return A;
    }
}
