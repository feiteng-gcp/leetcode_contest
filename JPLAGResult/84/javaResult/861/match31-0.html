class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int[][] ret = new int[A.length][A[0].length];
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[0].length; j++) {
                ret[i][j] = 1 - A[i][A[0].length - 1 - j];
            }
        }
        return ret;
    }
}
