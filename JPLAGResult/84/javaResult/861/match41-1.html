class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        if (A == null || A.length == 0 || A[0].length == 0) {
            return A;
        }
        int n = A.length;
        int m = A[0].length;
        int[][] B = new int[n][m];
        
for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                B[i][j] = A[i][m -1 - j] == 1 ? 0 : 1;
            }
        }
        return B;
    }
}
