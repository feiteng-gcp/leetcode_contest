class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int N = A.length, M = A[0].length;
        
        
for (int i = 0; i < N; ++i) {
            for (int j = 0; j < M/2; ++j) {
                int tmp = A[i][j];
                A[i][j] = A[i][M-j-1];
                A[i][M-j-1] = tmp;
            }
            
for (int j = 0; j < M; ++j) { 
                A[i][j] = A[i][j] == 1 ? 0 : 1;
            }
        }
        
        return A;
    }
}
