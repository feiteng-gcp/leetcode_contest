class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
    	int[][] ans = new int[A.length][A[0].length];
        for(int i = 0; i < A.length; i++) {
        	for(int j = 0; j < A[0].length; j++) {
        		ans[i][j] = (A[i][A[0].length-j-1] + 1) % 2;
        	}
        }
        return ans;
    }
}
