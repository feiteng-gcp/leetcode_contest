class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        for (int i = 0; i < A.length; ++i) {
        	int l = 0;
        	int r = A[i].length - 1;
        	while (l <= r) {
        		int t = A[i][l];
        		A[i][l] = A[i][r];
        		A[i][r] = t;
        		l++;
        		r--;
        	}
        	for (int j = 0; j < A[i].length; ++j) {
        		A[i][j] = 1 - A[i][j];
        	}
        }
        return A;
    }
}
