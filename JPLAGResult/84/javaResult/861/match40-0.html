class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        for(int i = 0; i < m; ++i){
            for(int j = 0; j <= (n-1)/2; ++j){
                int tmp1 = 1 - A[i][j];
                int tmp2 = 1 - A[i][n - 1 - j];
                A[i][n - 1 - j] = tmp1;
                A[i][j] = tmp2;
            }
        }
        
        return A;
    }
}
