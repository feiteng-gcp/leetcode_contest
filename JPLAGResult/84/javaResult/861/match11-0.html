class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int m = A.length;
        if(m == 0) return A;
        int n = A[0].length;
        int[][] res = new int[m][n];
        for(int i = 0; i<m; i++){
            for(int j = 0; j<n; j++){
                res[i][j] = 1 - A[i][n - j - 1];
            }
        }
        return res;
    }
}
