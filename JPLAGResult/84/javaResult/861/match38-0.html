class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int[][] newList = new int[A.length][A[0].length];
        
        for (int i = 0; i < A.length; i++) {
            for (int j = A[0].length - 1; j >= 0; j--) {
                newList[i][j] = 1 ^ A[i][A[0].length - j - 1];
            }
        }
        
        return newList;
    }
}
