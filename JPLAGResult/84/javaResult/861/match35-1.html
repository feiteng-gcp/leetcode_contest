class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        for (int [] arr : A) {
            int start = 0, end = arr.length-1;
            while (start <= end) {
                int temp = arr[start];
                arr[start++] = 1 - arr[end];
                arr[end--] = 1- temp; 
            }
        }
        return A;
    }
}
