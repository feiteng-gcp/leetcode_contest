class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        int row = A.length, col = A[0].length;
        // reverse
        for (int i = 0; i < row; i++) {
            for (
int p = 0, q = col - 1; p < q; p++, q--) {
                int t = A[i][p];
                A[i][p] = A[i][q];
                A[i][q] = t;
            }
        }
        // invert
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                A[i][j] = A[i][j] ^ 1;
            }
        }
        return A;
    }
}
