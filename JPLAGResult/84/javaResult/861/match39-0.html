class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        for (int i = 0; i < A.length; i++) {
            int left = 0, right = A[i].length - 1;
            while (left < right) {
                A[i][left] = 1 - A[i][left];
                A[i][right] = 1 - A[i][right];
                swap(A[i], left++, right--);
            }
            if (left == right) {
                A[i][left] = 1 - A[i][left];
            }
        }
        return A;
    }
    
    private void swap(int[] A, int i, int j) {
        int temp = A[i];
        A[i] = A[j];
        A[j] = temp;
    }
}
