class Solution {
    public int[][] flipAndInvertImage(int[][] A) {
        for (int[] a : A) {
            int i = 0, j = a.length - 1;
            while (i < j) {
                int temp = a[i];
                a[i++] = a[j];
                a[j--] = temp;
            }
        }
        for (int[] a : A) {
            
for (int i = 0; i < a.length; i++) {
                a[i] = 1 - a[i];
            }
        }
        return A;
    }
}
