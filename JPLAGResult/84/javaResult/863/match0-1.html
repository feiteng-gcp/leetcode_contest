class Solution {
    int[] childs;
    int[] dises;
    int getDis(int r,int p,HashSet<Integer>[] adjs){
        for(int adj : adjs[r]){
            if(adj != p) {
                int n = getDis(adj, r, adjs);
                dises[r] += n + childs[adj] + 1;
                childs[r] += childs[adj] + 1;
            }
        }

        return dises[r];
    }

    public int[] sumOfDistancesInTree(int N, int[][] edges) {
        childs = new int[N];
        dises = new int[N];
        
HashSet<Integer>[] nexts = new HashSet[N];
        for(int i = 0;i<N;i++) nexts[i] = new HashSet<>();

        for(int i = 0;i<edges.length;i++){
            nexts[edges[i][0]].add(edges[i][1]);
            nexts[edges[i][1]].add(edges[i][0]);
        }

        getDis(0,-1,nexts);

        boolean[] v = new boolean[N];
        v[0] = true;
        Queue<int[]> q = new LinkedList<>();
        for(int n : nexts[0]){
            q.offer(new int[]{n,0});
            v[n] = true;
        }

        while (!q.isEmpty()){
            int[] entry = q.poll();
            int near = childs[entry[0]] + 1;
            int far = N - near;
            int b = far - near;
            dises[entry[0]] = dises[entry[1]] + b;

            for(int n : nexts[entry[0]]){
                if(!v[n]){
                    q.offer(new int[]{n,entry[0]});
                    v[n] = true;
                }
            }
        }
        return dises;
    }
}
