class Solution:
    def ambiguousCoordinates(self, S):
        def val(S):
            ret = set()
            if S[0]!='0' or len(S)==1:
                ret.add(S)
            for _ in range(1, len(S)):
                S1, S2 = S[:_], S[_:]
                if len(S1)==1 or not S1[0]=='0':
                    if S2[-1]!='0':
                        ret.add(S
1+'.'+S2)
            # print(S, ret)
            return ret
        
        S = S[1:-1]
        xxx = []
        for i in range(1, len(S)):
            S1 = S[:i]
            S2 = S[i:]
            s1, s2 = val(S1), val(S2)
            for c1 in s1:
                for c2 in s2:
                    xxx.append('('+c1+', '+c2+')')
        return xxx
