import re

class Solution:
    def ambiguousCoordinates(self, S):
        """
        :type S: str
        :rtype: List[str]
        """
        a = '123'
        float_re = re.compile('[0-9]*\.?[0-9]+')

        def is_valid(s):
            if float_re.findall(s)[0] != s:
                return False
            point_appear = 0
            leading_zero = 0
            for idx, char in enumerate(s):
                if point_appear == 1:
                    if char == '0' and idx == len(s) - 1:
                        return False
                if leading_zero == 1:
                    if char != '.':
                        return False
                    else:
                        leading_zero = 0
                if idx == 0 and char == '0':
                    leading_zero = 1
                if char == '.':
                    point_appear = 1
            return True

        del_s = S[1:-1]
        ans = []

        for space_pos in range(1, len(del_s)):
            left = del_s[:space_pos]
            right = del_s[space_pos:]
            for point_pos_l in range(len(left)):
                if point_pos_l == 0:
                    left_temp = left
                else:
                    left_temp = left[:point_pos_l] + '.' + left[point_pos_l:]
                for point_pos_r in range(len(right)):
                    if point_pos_r == 0:
                        right_temp = right
                    else:
                        right_temp = right[:point_pos_r] + '.' + right[point_pos_r:]
                    if is_valid(left_temp) and is_valid(right_temp):
                        this_ans = '(' + str(left_temp) + ',' + ' ' + str(right_temp) + ')'
                        ans.append(this_ans)
        return ans
