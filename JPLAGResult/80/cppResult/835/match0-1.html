/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    int numComponents(ListNode* head, vector<int>& G) {
        set<int> s;
        for (int x : G) s.insert(x);
        bool vis = false;
        int ans = 0;

        for (ListNode *x = head; x != nullptr; x = x->next) {
            bool t = s.count(x->val) != 0;
            if (!vis && t) {
                ans++;
            }
            vis = t;
        }
        return ans;
    }
};
