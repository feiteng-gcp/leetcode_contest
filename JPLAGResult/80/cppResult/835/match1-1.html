class Solution {
public:
    int numComponents(ListNode* head, vector<int>& G) {
      int i,j,k,p,ans;
      set<int> st;
      st.clear();
      for (i=0;i<G.size();i++)
        st.insert(G[i]);
      p=0;
      ans=0;
      while (head!=NULL)
      {
        if (st.find(head->val)==st.end())
        {
          p=0;

        }
        else
        {
          if (p==0)
          {
            p=1;
            ans++;
          }
        }
        head=head->next;
      }
      return ans;
    }
};
