/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    int numComponents(ListNode* head, vector<int>& G) {
        set<int> S;
        for(auto& tmp:G) {
            S.insert(tmp);
        }
        int res=0;
        bool prev=false;
        for(auto p=head;p;p=p->next) {
            bool tmp=S.count(p->val)>0;
            if(tmp&&!prev) {
                res++;
            }
            prev=tmp;
        }
        return res;
    }
};
