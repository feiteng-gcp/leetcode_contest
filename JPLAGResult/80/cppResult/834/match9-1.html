class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        if(S.size()<4)
        {
            return {};
        }
        
        int n = S.size();

        S = S.substr(1, n-2);
        n = S.size();
        
        vector<string> result;

        for(int i=1; i<n; i++)
        {
            //[0,i], [i+1, n-1]
            auto left = S.substr(0, i);
            auto right = S.substr(i);
            vector<string> ls;
            find(left, ls);
            vector<string> rs;
            find(right, rs);

            for(auto & l : ls)
            {
                for(auto & r :rs)
                {
                    result.push_back("(" + l + ", " + r +")");
                }
            }
        }

        return result;
    }

    void find(string & str, vector<string> & result)
    {
        if(str == to_string(stoi(str)))
        {
            result.push_back(str);
        }
        
        int n =str.size();
        for(int i=1; i<n; i++)
        {
            auto left = str.substr(0, i);
            auto right = str.substr(i);

            if(left != to_string(stoi(left)))
            {
                continue;
            }

            if(stoi(right)%10 ==0)
            {
                continue;
            }

            result.push_back(left + "." + right);
        }
    }

  
};
