class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        int len = S.size();
        S.pop_back();
        vector<string> ans;
        for (int i = 2; i < len - 1; ++i){
            vector<string> b, e;
            b = getNum(S.substr(1, i - 1));
            if (b.size()) {
                e = getNum(S.substr(i));
                if (e.size()){
                    for (auto a: b){
                        for (auto c:e){
                            ans.push_back("(" + a + ", " + c + ")");
                        }
                    }
                }
            }
        }
        return ans;
    }
    vector<string> getNum(string s){
//        cout << s << endl;
        vector<string> ans;
        int len = s.size();

        if (s.size() == 1) {
            ans.push_back(s);
            return ans;
        }
        if (s[0] == '0'){
            if (s[len - 1] == '0') return ans;
            else {
                ans.push_back("0." + s.substr(1));
                return ans;
            }
        }
        ans.push_back(s);
        if (s[len - 1] == '0'){
            
            return ans;
        }
        for (int i = 1; i < len; ++i) ans.push_back(s.substr(0, i) + "." + s.substr(i));
        return ans;
    }
};
