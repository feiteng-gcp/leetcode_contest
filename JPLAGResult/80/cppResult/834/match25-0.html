class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> res;
        int n = S.length();
        if (n < 4) {
            return res;
        }
        // cout << stof("000.0") << " " << stof("00.01") << " " << stof("00.010") << " " << stoi("0001121130")<< endl;
        string s = S.substr(1, n - 2);
        n = s.length();
        for (int i = 0; i < n - 1; ++i) {
            string s1 = s.substr(0, i + 1);
            string s2 = s.substr(i + 1);
            
            auto nums1 = parse(s1);
            auto nums2 = parse(s2);
            if (nums1.size() > 0 && nums2.size() > 0) {
                for (auto& n1 : nums1) {
                    for (auto& n2 : nums2) {
                        res.emplace_back("(" + n1 + ", " + n2 + ")");
                    }
                }
            }
        }
        return res;
    }

private:
    vector<string> parse(const string& s) {
        int n = s.length();
        if (n == 0) {
            return {};
        }
        vector<string> res;
        for (int i = 0; i < n - 1; ++i) {
            auto t1 = s.substr(0, i + 1) + "." + s.substr(i + 1);
            int zeros = 0;
            for (int j = t1.length() - 1; j >=0; --j) {
                if (t1[j] == '0') {
                    ++zeros;
                    continue;
                } 
                break;
            }
            if (zeros > 0) {
                break;
            }
            
            zeros = 0;
            for (int j = 0; j < t1.length(); ++j) {
                if (t1[j] == '0') {
                    ++zeros;
                    continue;
                }
                break;
            }
            if (zeros > 0 && i != 0) {
                break;
            }
            res.emplace_back(t1);
//             auto t2 = to_string(stold(t1));
//             if (t2.length() > t1.length()) {
//                 t2 = t2.substr(0, t1.length());
//             }
//             cout << "parse " << t1 << " " << t2 << endl;
//             if (t1 == t2) {
                
//                 res.emplace_back(t1);
//             }
        }
        
        auto t = to_string(stoi(s));
        if (t == s) {
            res.emplace_back(t);
        }
        return res;
    }
};
