class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> ret;
        string nums=S.substr(1,S.length()-2);
        for(int i=1;i<nums.length();i++) {
            vector<string> left,right;
            string orileft=nums.substr(0,i);
            string oriright=nums.substr(i);
            if(judge(orileft))  left.push_back(orileft);
            if(judge(oriright)) right.push_back(oriright);
            for(int pos=1;pos<orileft.length();++pos) {
                string temp(orileft);
                string newleft=temp.insert(pos,".");
                if(judge(newleft))
                    left.push_back(newleft);
            }
            for(int pos=1;pos<oriright.length();++pos) {
                string temp(oriright);
                string newright=temp.insert(pos,".");
                if(judge(newright))
                    right.push_back(newright);
            }
            for(auto leftstr : left)
                for(auto rightstr : right)
                    ret.push_back(getans(leftstr,rightstr));
            /*if(judge(left) && judge(right)) {
                string ans=getans(left,right);
                ret.push(ans);
            }
            string newleft,newright;
            if(left.size()>1) {
                newleft=left.insert(1,".");
                if(judge(newleft) && judge(right)) {
                    string ans=getans(newleft,right);
                    ret.push(ans);
                }
            }
            if()*/
        }
        return ret;
    }
    bool judge(string str) {
        if(str.length()==1)
            return true;
        if(str[0]=='0'&&str[1]!='.')
            return false;
        if(str.find('.')!=string::npos&&str[str.length()-1]=='0')
            return false;
        return true;
    }
    inline string getans(string left,string right) {
        return "("+left+", "+right+")";
    }
};
