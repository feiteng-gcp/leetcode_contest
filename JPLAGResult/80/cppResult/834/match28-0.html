class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> res;
        string s = S.substr(1, S.size()-2);
        int n = s.size();
        for (int i=1; i < n; i++) {
            vector<string> l = weCanGet(s.substr(0,i)), r = weCanGet(s.substr(i, n-i));
            if (l.size() == 0 || r.size() == 0) continue;
            for (auto ls: l) {
                for (auto rs: r)
                    res.push_back("("+ls+", "+rs+")");
            }
        }
        return res;
    }

    vector<string> weCanGet(string s) {
        int n = s.size();
        vector<string> res;
        for (int i=1; i < n; i++) {
            string str = s.substr(0, i) + '.' + s.substr(i, n-i);
            if (check(str)) res.push_back(str);
        }
        if (check(s)) res.push_back(s);
        return res;
    }

    bool check(string s) {
        int n = s.size(), i = 0, n_zero = 0;

        while (i < n && s[i] == '0') { n_zero++; i++; }
        if (n_zero > 1) return false;

        while (i < n && isdigit(s[i])) { if (n_zero) return false; i++; }

        if (s[i] == '.' && i > 0 && i+1 != n) i++;

        while (i < n && isdigit(s[i])) {
            if (i+1 == n && s[i] == '0') return false;
            i++;
        }

        return s[i] == 0;
    }

};
