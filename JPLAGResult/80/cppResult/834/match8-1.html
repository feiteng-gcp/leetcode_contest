typedef long long int64;
class Solution {
public:
    bool valid(const string& s, int p) {
        int n = s.size();
        int64 L = 0, R = 0;
        for (int i = 0; i < p; ++i) {
            L = L * 10 + (s[i] - '0');
        }
        for (int i = p; i < n; ++i) {
            R = R * 10 + (s[i] - '0');
        }
        if (p < n) {
            if (R % 10 == 0) return false;
        }
        if (p > 1) {
            int64 ten = 1;
            for (int i = 0; i < p - 1; ++i) ten *= 10;
            if (L < ten) return false;
        }
        return true;
    }
    vector<string> ambiguousCoordinates(string S) {
        S.pop_back();
        S = S.substr(1);
        int n = S.size();
        vector<string> ret;

        for (int p = 1; p < n; ++p) {
            string u = S.substr(0, p);
            string v = S.substr(p, n - p);
            for (int i = 1; i <= u.size(); ++i) {
                if (!valid(u, i)) continue;
                for (int j = 1; j <= v.size(); ++j) {
                    if (!valid(v, j)) continue;
                    string cur = "(";
                    cur += u.substr(0, i);
                    if (i != u.size()) {
                        cur += "." + u.substr(i, u.size() - i);
                    }
                    cur += ", ";
                    cur += v.substr(0, j);
                    if (j != v.size()) {
                        cur += "." + v.substr(j, v.size() - j);
                    }
                    cur += ")";
                    ret.push_back(cur);
                }
            }
        }
        return ret;
    }
};
