#define REP1(i, n) for (remove_cv<remove_reference<decltype(n)>::type>::type i = 1; i <= (n); i++)
#define ROF(i, a, b) for (remove_cv<remove_reference<decltype(b)>::type>::type i = (b); --i >= (a); )

class Solution {
  vector<string> f(string a) {
    if (a[0] == '0') {
      if (a.size() == 1)
        return {"0"};
      if (a.back() != '0')
        return {"0." + a.substr(1)};
      return {};
    }
    vector<string> r{a};
    if (a.back() != '0') {
      for (int i = a.size();--i;)
        r.push_back(a.substr(0, i)+'.'+a.substr(i));
    }
    return r;
  }
public:
  vector<string> ambiguousCoordinates(string S) {
    vector<string> ret;
    S = S.substr(1, S.size() - 2);
    for (int i = 1; i < S.size(); i++) {
      vector<string> ls = f(S.substr(0, i)), rs = f(S.substr(i));
      for (auto &l : ls)
        for (auto &r : rs)
          ret.push_back('(' + l + ", " + r + ')');
    }
    return ret;
  }
};
