class Solution {
public:
    vector<string> work(string S)
    {
        vector<string> ans;
        ans.clear();
        if(S[0]!='0'||S.size()==1)ans.push_back(S);
        string s;
        int n=S.size(),i,j;
        if(S[n-1]!='0')for(i=1;i<n;i++)
        {
            s="";
            for(j=0;j<i;j++)s+=S[j];
            s+='.';
            for(;j<n;j++)s+=S[j];
            if(s[0]!='0'||i==1)ans.push_back(s);
        }
        return ans;
    }
    vector<string> ambiguousCoordinates(string S) {
        vector<string> ans,a1,a2;
        ans.clear();
        string s1,s2,s;
        int n=S.size(),n1,n2,i,j,k;
        for(i=2;i+1<n;i++)
        {
            s1=s2="";
            for(j=1;j<i;j++)s1+=S[j];
            for(;j+1<n;j++)s2+=S[j];
            a1=work(s1);
            a2=work(s2);
            n1=a1.size();
            n2=a2.size();
            for(j=0;j<n1;j++)for(k=0;k<n2;k++)
            {
                s="";
                s+='(';
                s+=a1[j];
                s+=", ";
                s+=a2[k];
                s+=')';
                ans.push_back(s);
            }
        }
        return ans;
    }
};
