class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {

        S = S.substr(1, S.size()-2);
        
        vector<string> res;
        
        int size = S.size();
        // vector<vector<bool>> mem(size, false);
        // for(int i = 0; i < size; i++)
        //     for(int j = i; j < size; j++)
        //         mem[i][j] = ;
        for(int i = 1; i < size; i++) {
            vector<string> a, b;
            dfs(S.substr(0, i), a);
            dfs(S.substr(i), b);
            for(int j = 0; j < a.size(); j++)
                for(int k = 0; k < b.size(); k++)
                    res.push_back("(" + a[j] + ", " + b[k] + ")");
        }
        
        
        return res;
    }
    
    void dfs(const string& s, vector<string>& res) {
        if(s.size() == 1) {
            res.push_back(s);
            return;
        }
        if(s[0] == '0') {
            if(s.back() == '0')
                return;
            else 
                res.push_back(s.substr(0, 1) + "." + s.substr(1));
        }
        else {
            if(s.back() == '0')
                res.push_back(s);
            else {
                res.push_back(s);
                for(int i = 1; i < s.size(); i++)
                    res.push_back(s.substr(0, i) + "." + s.substr(i));
            }
        }
    }
};
