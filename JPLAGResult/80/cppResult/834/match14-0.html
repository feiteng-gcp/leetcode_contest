class Solution {
public:
    
    vector<string> work(string s) {
        vector<string> ans;
        
        if (s[0] != '0') {
            ans.push_back(s);
        }
        
        if (s[0] == '0' && s[1] == '\0') {
            ans.push_back(s);
        }
        
        string res;
        for (int i = 1; i < s.size(); i++) {
            res.clear();
            for (int j = 0; j < i; j++) {
                res = res + s[j];
            }
            res = res +".";
            
            int flag = 0;
            for (int j = i; j < s.size(); j++) {
                res = res + s[j];
                if (s[j] != '0') flag = 1;
            }
            
            if (flag == 1) {
                if ((i >= 2 && res[0] == '0') || res[s.size()] == '0') {
                    
                }
                else {
                    ans.push_back(res);
                }
            }
        }
        
        
        return ans;
    }
    
    vector<string> ambiguousCoordinates(string S) {
        vector<string> ans, res1, res2;
        
        string s1, s2;
        for (int i = 2; i < S.size() - 1; i++) {
            
            s1.clear();
            s2.clear();
            
            for (int j = 1; j < i; j++) s1 = s1 + S[j];
            for (int j = i; j < S.size()-1; j++) s2 = s2 + S[j];
            
            res1 = work(s1);
            res2 = work(s2);
            
            for (int i = 0; i < res1.size(); i++) {
                for (int j = 0; j < res2.size(); j++) {
                    ans.push_back("(" + res1[i] + ", " + res2[j] + ")");
                }
            }
        }
        
        
        return ans;
    }
};
