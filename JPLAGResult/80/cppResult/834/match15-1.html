class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        int ok1[12],ok2[12];
	vector<string> ans;
	int len=S.size();
	for(int i=1;i<len-2;i++)
	{
		if(!((S[1]=='0'&&i>1)||(S[i+1]=='0'&&len-2-i>1)))
		{
			string res="";
			for(int j=0;j<=i;j++)
			{
				res=res+S[j];
			}
			res=res+", ";
			for(int j=i+1;j<len;j++)
			{
				res=res+S[j];
			}
			ans.push_back(res);
		}
		int cnt1=0,cnt2=0;
		for(int j=1;j<i;j++)
		{
			if(!((S[1]=='0'&&j>1)||S[i]=='0'))ok1[cnt1++]=j;
		}
		for(int j=i+1;j<len-2;j++)
		{
			if(!((S[i+1]=='0'&&j-i>1)||S[len-2]=='0'))ok2[cnt2++]=j;
		}
		//printf("%d:%d %d\n",i,cnt1,cnt2);
		if(!(S[i+1]=='0'&&len-2-i>1)){
		for(int f=0;f<cnt1;f++)
		{
			string res="";
			for(int j=0;j<=ok1[f];j++)
			{
				res=res+S[j];
			}
			res=res+".";
			for(int j=ok1[f]+1;j<=i;j++)
			{
				res=res+S[j];
			}
			res=res+", ";
			for(int j=i+1;j<len;j++)
			{
				res=res+S[j];
			}
			ans.push_back(res);
		}
	}
		if(!(S[1]=='0'&&i>1)){
		for(int f=0;f<cnt2;f++)
		{
			string res="";
			for(int j=0;j<=i;j++)
			{
				res=res+S[j];
			}
			res=res+", ";
			for(int j=i+1;j<=ok2[f];j++)
			{
				res=res+S[j];
			}
			res=res+".";
			for(int j=ok2[f]+1;j<len;j++)
			{
				res=res+S[j];
			}
			ans.push_back(res);
		}

	}
		for(int f1=0;f1<cnt1;f1++)
		{
			for(int f2=0;f2<cnt2;f2++)
			{
			string res="";
			for(int j=0;j<=ok1[f1];j++)
			{
				res=res+S[j];
			}
			res=res+".";
			for(int j=ok1[f1]+1;j<=i;j++)
			{
				res=res+S[j];
			}
			res=res+", ";
			for(int j=i+1;j<=ok2[f2];j++)
			{
				res=res+S[j];
			}
			res=res+".";
			for(int j=ok2[f2]+1;j<len;j++)
			{
				res=res+S[j];
			}
			ans.push_back(res);
			}
		}
	}
	return ans;
    }
};
