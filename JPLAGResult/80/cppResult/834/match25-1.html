class Solution {
public:
    int n;
    
    bool validLeft(string s){
        if( s[0] == '0' && s.size() > 1 ) return false;
        return true;
    }
    
    bool validRight(string s){
        /*if( s[0] == '0' && s.size() == 1 ) return false;
        bool all_zeroes = true;
        for( int i = 0 ; i < s.size() ; ++i ){
            if( s[i] != )
        }*/
        
        if( s[s.size() - 1] == '0' ) return false;
        return true;
    }
    
    unordered_set<string> possibilities(string s){
        unordered_set<string> hash;
        if( validLeft(s) ) hash.insert(s);
        string left = "", right = "";
        for( int i = 0 ; i < s.size()  - 1; ++i ){
            left += s[i];
            right = s.substr(i + 1);
            //cout<<left<<"."<<right<<endl;
            if( validLeft(left) && validRight(right) ) hash.insert(left + "." + right);
            
        }
        return hash;
    }
    
    vector<string> ambiguousCoordinates(string S) {
        S = S.substr(1, S.size() - 2);
        //cout<<S<<endl;
        n = S.size();
        //hash.clear();
        unordered_set<string> hash;
        string left = "", right = "";
        for( int i = 0 ; i < n - 1 ; ++i ){
            left += S[i];
            right = S.substr(i + 1);
            //cout<<left<<" "<<right<<endl;
            unordered_set<string> lhash = possibilities(left);

            unordered_set<string> rhash = possibilities(right);
            if( lhash.size() > 0 && rhash.size() > 0 ){
                for( auto lit :lhash ){
                    for( auto rit: rhash ){
                        hash.insert("(" + lit + ", " + rit + ")" );
                    }
                }
            }
        }
        
        vector<string> result;
        for( auto it:hash ){
            result.push_back(it);
        }
        return result;
    }
};
