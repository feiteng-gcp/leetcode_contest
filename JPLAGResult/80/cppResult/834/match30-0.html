class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> result;
        
        int len = S.size();
        string word = S.substr(1, len - 2);                
        
        len = word.size();
        if (len == 0) return result;
        
        unordered_set<string> hash;
        for (int j = 1; j < word.size(); ++j)
        {
            string str0 = word.substr(0, j);
            string str1 = word.substr(j);
            
            //cout << str0 << "," << str1 << endl;
            
            vector<string> vec0;
            Search(str0, vec0);
            
            vector<string> vec1;
            Search(str1, vec1);
            
            for (int i = 0; i < vec0.size(); ++i)
            {
                
                if (IsValid(vec0[i]) == false) continue;
                
                
                for (int t = 0; t < vec1.size(); ++t)
                {
                    if (IsValid(vec1[t]) == false) continue;
                                        
                    string coor  = "(" + vec0[i] + ", " + vec1[t] + ")";
                    if (hash.find(coor) == hash.end())
                        hash.insert(coor);
                } 
            }
        }
        
        for(auto item : hash)
            result.push_back(item);
        
        return result;
    }
    
    void Search(string& s, vector<string>& strs)
    {
        int len = s.size();
        strs.push_back(s);
        
        for (int i = 1; i <= len - 1; ++i)
        {
            string temp = s;
            temp.insert(temp.begin() + i, '.');           
            strs.push_back( temp );
        }
    }
    
    bool IsValid(string& s)
    {
        int len = s.size();        
        int pos  = s.find(".");
        
        if (len >= 2 && s[0] == '0' && s[1] == '0') return false;        
        
        if (pos == string::npos) 
        {
            if (len >= 2 && s[0] == '0') return false;
            return true;   
        }
        
        if (pos > 1)
        {
            for (int i = 0; i < pos; ++i)
                if (s[i] != '0') break;
                else return false;
        }
        
        for (int i = len - 1; i > pos; --i)
            if (s[i] != '0') break;
            else return false;
        
        return true;
    }
};
