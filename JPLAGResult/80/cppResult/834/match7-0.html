typedef long long ll;

template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T> >;

const ll MOD=1e9+7;

class Solution {
public:
  vector<string> poss(string s){
    int N=s.size();
    vector<string> res;
    if(!(N>=2 && s[0]=='0')) res.pb(s);
    for(int i=0;i<N-1;i++){
      if(i>=1 && s[0]=='0') continue;
      if(s[N-1]=='0') continue;
      res.pb(s.substr(0,i+1)+"."+s.substr(i+1));
    }
    return res;
  }
    vector<string> ambiguousCoordinates(string S) {
      string s=S.substr(1,S.size()-2);
      int N=s.size();
      vector<string> res;
      for(int i=0;i<N-1;i++){
	auto v1=poss(s.substr(0,i+1));
	auto v2=poss(s.substr(i+1));
	for(auto& s1:v1){
	  for(auto& s2:v2){
	    res.pb("("+s1+", "+s2+")");
	  }
	}
      }
      return res;
    }
};
