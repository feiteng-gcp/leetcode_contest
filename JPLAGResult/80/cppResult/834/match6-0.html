class Solution {
public:
    bool valid (string s) {
        int n = s.size();
        bool point = false;
        for (char c: s) {
            if (c == '.') point = true;
        }
        if (point && s[n - 1] == '0') return false;
        if (s == "0") return true;
        if (point && s[0] == '0' && s[1] != '.') return false;
        if (!point && s[0] == '0') return false;
        return true;
    }
    vector<string> ambiguousCoordinates(string S) {
        vector<string> res;
        S = S.substr(1, S.size() - 2);
        int n = S.size();
        for (int i = 1; i < n; i++) {
            string a = S.substr(0, i);
            string b = S.substr(i, n - i);
            int la = a.size();
            int lb = b.size();
            for (int j = 0; j < la; j++) {
                for (int k = 0; k < lb; k++) {
                    string c, d;
                    if (j != 0) {
                        c = a.substr(0, j) + '.' + a.substr(j, la - j);
                    } else {
                        c = a;
                    }
                    if (k != 0) {
                        d = b.substr(0, k) + '.' + b.substr(k, lb - k);
                    } else {
                        d = b;
                    }
                    //cout << c << endl;
                    //cout << d << endl;
                    if (valid(c) && valid(d)) {
                        res.push_back('(' + c + ", " + d + ')');
                    }
                }
            }
        }
        return res;
    }
};
