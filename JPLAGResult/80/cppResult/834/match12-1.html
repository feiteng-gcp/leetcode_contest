class Solution {
public:
    bool ok(string t,int p){
        int n = t.length();
        if(n == 1) return 1;
        if(t[0] == '0' && t[1] != '.') return 0;
        if(t[n - 1] == '0' && p == 1) return 0;
        return 1;
    }
    vector<string> check(string t){
        vector<string> ret;
        if(ok(t,0)) ret.push_back(t);
        for(int i = 1;i < t.length();i++){
            string x = t.substr(0,i) + "." + t.substr(i,t.length() - i);
            if(ok(x,1)) ret.push_back(x);
        }
        return ret;
    }
    vector<string> ambiguousCoordinates(string S) {
        S = S.substr(1,S.length() - 2);
        
        vector<string> ans;
        for(int i = 1;i < S.length();i++){
            string s1 = S.substr(0,i);
            string s2 = S.substr(i,S.length() - i);
            cout << s1 << " " << s2 << endl;
            vector<string> v1 = check(s1);
            vector<string> v2 = check(s2);
            if(v1.size() == 0 || v2.size() == 0) continue;

            for(int k = 0;k < v1.size();k++){
                for(int j = 0;j < v2.size();j++){
                    ans.push_back("(" + v1[k] + ", " + v2[j]+")");
                }
            }
        }
        return ans;
    }
};
