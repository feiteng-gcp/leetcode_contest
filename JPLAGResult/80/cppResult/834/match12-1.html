class Solution {
    vector<string> convert(string& s) {
        vector<string> result;
        if (s.size() == 0) return result;
        // check integer
        if (s.size() == 1 || s[0] != '0') result.push_back(s);
        // check decimal
        if (s[s.size() - 1] != '0') { // **.**0 is not good
            // where to put .
            for (int i = 1; i < s.size(); ++i) {
                if (i == 1 || s[0] != '0') result.push_back(s.substr(0, i) + '.' + s.substr(i));
            }

        }
        return result;
    }
public:
    vector<string> ambiguousCoordinates(string S) {
        S = S.substr(1, S.size() - 2);
        cout << S << endl;
        vector<string> result;
        for (int i = 1; i < S.size(); ++i) {
            string left = S.substr(0, i);
            string right = S.substr(i);
            vector<string> first = convert(left);
            vector<string> second = convert(right);
            if (first.empty() || second.empty()) continue;
            for (string& s1 : first) {
                for (string& s2 : second) {
                    result.push_back(string("(") + s1 + ", " + s2 + ")");
                }
            }
        }
        return result;
    }
};
