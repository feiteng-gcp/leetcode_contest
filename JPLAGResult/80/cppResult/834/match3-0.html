class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        int n=S.size();
                vector<string>st{};
        //int cnt{};
        for(int i=1;i<n-2;i++){
            string l{},r{};
            copy(S.begin()+1,S.begin()+1+i,back_inserter(l));
            copy(S.begin()+1+i,S.end()-1,back_inserter(r));
            spl(l,r,st);
        }
        //cout << cnt;
        return st;
    }
    
    vector<string> va(string s){
        vector<string>temp{};
        if(s.size()==1){
            temp.push_back(s);
            return temp;
        }
        if(s.back()=='0'){
            if(s[0]=='0')
                return temp;
            else{
                temp.push_back(s);
                return temp;
            }
        }
        if(s[0]=='0'){
            s.insert(s.begin()+1,'.');
            temp.push_back(s);
            return temp;
        }
        for(int i=1;i<s.size();i++){
            string t=s;
            t.insert(t.begin()+i,'.');
            temp.push_back(t);
        }
        temp.push_back(s);
        return temp;
    }
    
    void spl(string l,string r,vector<string> &st){
        vector<string>left=va(l), right=va(r);
        for(int i=0;i<left.size();i++){
            for(int j=0;j<right.size();j++){
                string re{};
                re.push_back('(');
                re.append(left[i]);
                re.push_back(',');
                re.push_back(' ');
                re.append(right[j]);
                re.push_back(')');
                st.push_back(re);
            }
        }
    }
};
