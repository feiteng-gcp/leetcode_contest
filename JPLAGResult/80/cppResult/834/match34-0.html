class Solution {
public:
    vector<string> possNum(string& S)
{
    vector<string> ans;
    int len = (int)S.size();
    if(len == 1 || S[0] != '0') ans.push_back(S);
    else
    {
        if(S.back() != '0') ans.push_back("0." + S.substr(1));
        return ans;
    }
    if(S.back() == '0') return ans;
    for(int i = 1; i < len; i++)
    {
        string l = S.substr(0, i), r = S.substr(i);
        ans.push_back(l + "." + r);
    }
    return ans;
}

vector<string> ambiguousCoordinates(string S)
{
    S.pop_back();
    S = S.substr(1);
    int len = (int)S.size();
    vector<string> ans;
    for(int i = 1; i < len; i++)
    {
        string l = S.substr(0, i), r = S.substr(i);
        vector<string> lv = possNum(l), rv = possNum(r);
        for(int x = 0; x < lv.size(); x++)
            for(int y = 0; y < rv.size(); y++)
                ans.push_back("(" + lv[x] + ", " + rv[y] + ")");
    }
    return ans;
}
};
