class Solution {
    inline bool isCorrect(const string & s) const {
        if (s.length() == 0)
            return false;
        if (s.length() == 1 && s[0] >= '0' && s[0] <= '9')
            return true;
        
        size_t dPoint = s.find('.');
        if (s[0] == '0' && s[1] != '.')
            return false;
        if (s[s.length() - 1] == '0' && dPoint != string::npos)
            return false;
        if (s[0] == '.' || s[s.length() - 1] == '.')
            return false;
        
        return true;
    }
    vector<string> findDecimalPoints(string s) const {
        vector<string> result;
        if (isCorrect(s))
            result.push_back(s);
        
        //cout << s << " is " << (isCorrect(s) ? "true" : "false") << endl;
        
        for (int i = 1; i < s.length(); i++) {
            string temp = s.substr(0, i) + '.' + s.substr(i);
            
            //cout << temp << " is " << (isCorrect(temp) ? "true" : "false") << endl;
            
            if (isCorrect(temp))
                result.push_back(temp);

        }
        return result;
    }
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> ans;
        vector<string> point[2];
        
        for (int i = 1; i < S.length() - 2; i++) {
            point[0] = findDecimalPoints(S.substr(1, i));
            point[1] = findDecimalPoints(S.substr(1 + i, S.length() - 2 - i));
            for (auto & s1 : point[0])
                for (auto & s2 : point[1])
                    ans.push_back("(" + s1 + ", " + s2 + ")");
        }
        
        return ans;
    }
};
