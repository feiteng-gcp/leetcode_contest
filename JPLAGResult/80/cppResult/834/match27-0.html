class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {

        vector<string> ans;
        S = S.substr(1, S.size() - 2);
        int len = S.size();

        for(int i = 1; i < len; i ++){
            string l = S.substr(0, i);
            string r = S.substr(i);

            if(to_string(stoi(l)) == l && to_string(stoi(r)) == r){
                ans.push_back("(" + l + ", " + r + ")");
            }

            int len_l = l.size();
            int len_r = r.size();

            if(to_string(stoi(l)) == l){
                for(int k = 1; k < len_r; k ++){
                    string rl = r.substr(0, k);
                    string rr = r.substr(k);

                    if(rr.back() == '0' || (rl.size() > 1 && rl[0] == '0'))
                        continue;
                    ans.push_back("(" + l + ", " + rl + "." + rr + ")");
                }
            }

            if(to_string(stoi(r)) == r){
                for(int j = 1; j < len_l; j ++){

                    string ll = l.substr(0, j);
                    string lr = l.substr(j);

                    if(lr.back() == '0' || (ll.size() > 1 && ll[0] == '0'))
                        continue;
                    ans.push_back("(" + ll + "." + lr + ", " + r + ")");
                }
            }


            for(int j = 1; j < len_l; j ++){

                string ll = l.substr(0, j);
                string lr = l.substr(j);

                if(lr.back() == '0' || (ll.size() > 1 && ll[0] == '0'))
                    continue;

                for(int k = 1; k < len_r; k ++){
                    string rl = r.substr(0, k);
                    string rr = r.substr(k);

                    if(rr.back() == '0' || (rl.size() > 1 && rl[0] == '0'))
                        continue;

                    ans.push_back("(" + ll + "." + lr + ", " + rl + "." + rr + ")");
                }

            }
        }

//        for(int i = 0; i < ans.size(); i ++){
//            cout << ans[i] << endl;
//        }
        return ans;
    }
};
