class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> ret;
        // split to two part
        S = S.substr(1, S.length() - 2);
        for(int i = 1; i < S.length(); ++i)
        {
            string left = S.substr(0, i);
            string right = S.substr(i, S.length() - i);
            vector<string> leftNum = findNums(left);
            if(leftNum.empty())
            {
                continue;
            }
            
            vector<string> rightNum = findNums(right);
            if(rightNum.empty())
            {
                continue;
            }
            
            // combine all possible 
            for(auto& ln : leftNum)
            {
                string res = "(" + ln + ", ";
                for(auto rn : rightNum)
                {
                    ret.push_back(res + rn + ")");
                }
            }
        }
        
        return ret;
    }
    
    vector<string> findNums(string& s)
    {
        vector<string> ret;
        // decimal point
        if(isValid(s, 0))
        {
            ret.push_back(s);
        }
        
        // add decimal point
        if(s.length() > 1)
        {
            for(int i = 1; i < s.length(); ++i)
            {
                string left = s.substr(0, i);
                if(!isValid(left, 0))
                {
                    continue;
                }
                
                string right = s.substr(i, s.length() - i);
                if(!isValid(right, 1))
                {
                    continue;
                }
                
                ret.push_back(left + "." + right);
            }
        }
        return ret;
    }
    
    bool isValid(string& s, int pos)
    {
        // left part
        if(pos == 0)
        {
            if(s.length() == 1 && s[0] == '0')
            {
                return true;
            }
            
            return s[0] != '0'; // the most left can't be 0
        }
        else
        {
            return s[s.length() - 1] != '0'; // the most right can't be 0
        }
    }
};
