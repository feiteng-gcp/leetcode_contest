class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> res;
        S = S.substr(1, S.size()-2);
        for(int i = 0; i<S.size()-1; i++){
            vector<string> p1 = helper(S.substr(0, i+1));
            vector<string> p2 = helper(S.substr(i+1, S.size() - i - 1));
            for(string i: p1){
                for(string j:p2){
                    res.push_back('(' + i + ", " + j + ')');
                }
            }
        }
        return res;
    }
    vector<string> helper(string s){
        vector<string> res;
        if(checkleft(s)) res.push_back(s);
        for(int i = 1; i<s.size(); i++){
            string left = s.substr(0, i);
            string right = s.substr(i, s.size() - i);
            if(checkleft(left) && checkright(right)) res.push_back(left + '.' + right);
        }
        return res;
    }
    bool checkleft(string s){
        if(s[0] == '0' && s.size()>1) return false;
        return true;
    }
    bool checkright(string s){
        if(s[s.size()-1] == '0') return false;
        return true;
    }
    
};
