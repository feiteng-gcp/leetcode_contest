class Solution {
public:
    vector<string> solve(string s) {
        int n = s.size();
        vector<string> res;
        if (n == 1) {
            res.push_back(s);
            return res;
        }
        
        
        if (s[0] != '0') {
            res.push_back(s);
        }
        
        for (int i = 0; i < n - 1; i++) {
            string foo = s.substr(0, i + 1);
            string bar = s.substr(i + 1);
            int pre = 0;
            if ((int)foo.size() == 1 || foo[0] != '0') {
                if ((int)foo.size() == 1 && foo[0] == '0') {
                    pre = 0;
                } else {
                    pre = 1;
                }
            } else {
                continue;
            }
            
            int after = 0;
            
            if ((int)bar.size() == 1 || bar.back() != '0') {
                if ((int)bar.size() == 1 && bar.back() == '0') {
                    after = 0;
                } else {
                    after = 1;
                }
            } else {
                continue;
            }
            
            if (pre == 0) {
                if (after > 0) {
                    res.push_back(foo + "." + bar);
                }
            } else {
                if (after != 0) {
                    res.push_back(foo + "." + bar);
                }
            }
        }
        
        return res;
    }
    vector<string> ambiguousCoordinates(string S) {
        set<string> vis;
        int n = S.size() - 2;
        S = S.substr(1, n);
        for (int i = 0; i < n - 1; i++) {
            string foo = S.substr(0, i + 1);
            string bar = S.substr(i + 1);
            // cout << foo << " " << bar << endl;
            vector<string> foo1 = solve(foo);
            vector<string> bar1 = solve(bar);
            for (auto fooItem : foo1) {
                for (auto barItem: bar1) {
                    if (fooItem != "" && barItem != "") {
                        vis.insert("(" + fooItem + ", " + barItem + ")");
                    }
                }
            }
            
        }
        vector<string> res(vis.begin(), vis.end());
        return res;
    }
};
