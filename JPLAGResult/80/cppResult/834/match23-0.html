class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        int dSize= S.size()-2;
        vector<string> ret;
        for(int i=1;i<dSize;++i) {            
            vector<string> sub1=helperFunc(S.substr(1,i)),sub2=helperFunc(S.substr(1+i,dSize-i));
            for(int j=0;j<sub1.size();++j) {
                for(int k=0;k<sub2.size();++k) {
                    string curRet="(" + sub1[j] + ", " + sub2[k] + ")";
                    ret.push_back(curRet);
                }
            }
        }
        return ret;
    }
    
    vector<string> helperFunc(string S) {
        vector<string> ret;
        size_t foundSN0 = S.find_first_not_of("0"); 
        size_t foundBN0 = S.find_last_not_of("0");
        size_t size = S.size();
        if (foundSN0==0) {
            ret.push_back(S);
            if (foundBN0==size-1) {
                for(int i=1;i<size;++i) {
                    ret.push_back(S.substr(0,i)+'.'+S.substr(i,size-i));
                }
            } 
        } else if (foundBN0==size-1) {
            ret.push_back(S.substr(0,1)+'.'+S.substr(1,size-1));
        }        
        if (size==1 && S[0]=='0') ret.push_back(S);
        return ret;
    }
    
};
