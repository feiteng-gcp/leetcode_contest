class Solution {
public:
    vector<string> ambiguousCoordinates(string s) {
        
        vector<string> ret;
        
        string strip = s.substr(1, s.size()-2);  //no '(' and ')'
        
        //pivot on 'x|y' cut position.
        for(int i=1; i<strip.size(); i++){
            string l = strip.substr(0, i);
            string r = strip.substr(i);
            
            for(int j=0; j<l.size(); j++){
                string left = j ? l.substr(0, j) + "." +  l.substr(j) : l; 
                for(int k=0; k<r.size(); k++){
                    string right = k ? r.substr(0, k) + "." + r.substr(k) : r;
                    
                    string empty = "";
                    if(valid1coord(left) && valid1coord(right)) ret.push_back(empty + "(" + left + ", " + right + ")");
                }
            }
        }
        
        
        
        return ret;
        
        
    }
    
    bool valid1coord(string x){
        if(x.find('.') == -1){ //not decimal.
            return x[0] != '0' || x == "0";
        } else { //decimal
            //must not end in zero.
            //if number of chars before decimal is greater than 1, then it first char must not be zero.
            if(x[x.size()-1] == '0') return false;
            if(x[0] != '.' && x[1] != '.' && x[0] == '0') return false;
            return true;
        }
    }
};
