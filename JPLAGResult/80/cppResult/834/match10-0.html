class Solution {
public:
    
    bool test(string S) {
        int decimal=S.find('.');
        if(decimal==string::npos) {
            if(S[0]=='0') {
                return S.length()==1;
            }
            return true;
        } else {
            if(S.back()=='0') {
                return false;
            }
            if(S[0]=='0') {
                return S[1]=='.';
            }
            return true;
        }
    }
    
    vector<string> dfsComma(string S) {
        vector<string> res;
        if(test(S)) {
            res.push_back(S);
        }
        for(int i=1;i<S.length();i++) {
            string tmp=S.substr(0,i)+"."+S.substr(i,S.length()-i);
            if(test(tmp)) {
                res.push_back(tmp);
            }
        }
        return res;
    }
    
    vector<string> ambiguousCoordinates(string S) {
        vector<string> res;
        for(int i=1;i+2<S.length();i++) {
            auto av=dfsComma(S.substr(1,i));
            auto bv=dfsComma(S.substr(i+1,S.length()-i-2));
            for(auto& a:av) {
                for(auto& b:bv) {
                    res.push_back("("+a+", "+b+")");
                }
            }
        }
        return res;
    }
};
