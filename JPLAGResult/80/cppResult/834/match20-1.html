class Solution {
    bool check(string & d) {
        if (d[0] == '0')
            if (d.length() > 1 && d[1] != '.')
                return false;
        if (d[d.length() - 1] == '0')
            if (d.find('.') != string::npos)
                return false;
        return true;
    }
    
    bool valid(string & S) {
        string d1 = S.substr(1, S.find(",") - 1), d2 = S.substr(S.find(" ") + 1, S.length() - S.find(" ") - 2);
        return check(d1) && check(d2);
    }
    
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> candidates = {}, ret = {};
        int l = S.length(); // from 1 to S.length() - 2

        for (int i = 1; i < S.length() - 2; ++i) {
            candidates.push_back(S.substr(0, i + 1) + ", " + S.substr(i + 1));
            for (int j = 1; j < i; ++j) {
                candidates.push_back(S.substr(0, j + 1) + "." + S.substr(j + 1, i - j) + ", " + S.substr(i + 1));
                for (int k = i + 1; k < S.length() - 2; ++k) {
                    candidates.push_back(S.substr(0, j + 1) + "." + S.substr(j + 1, i - j) + ", " + S.substr(i + 1, k - i) + "." + S.substr(k + 1));
                }
            }
            for (int k = i + 1; k < S.length() - 2; ++k) {
                candidates.push_back(S.substr(0, i + 1) + ", " + S.substr(i + 1, k - i) + "." + S.substr(k + 1));
            }
        }
        for (auto & str : candidates)
            if (valid(str))
                ret.push_back(str);
        return ret;
    }
};
