class Solution {
private:
    vector<string> checknum(string s){
        vector<string> res;
        int n = s.length();
        if(n <= 0) return res;
        if(n == 1){
            res.push_back(s);
            return res;
        }
        
        if(s[n-1] == '0'){
            if(s[0] == '0')
                return res;
            else{
                res.push_back(s);
                return res;
            }
        }else{
            if(s[0] == '0'){
                string temp = "0." + s.substr(1);
                res.push_back(temp);
                return res;
            }else{
                res.push_back(s);
                for(int i = 1; i <= n-1; i++){
                    string temp = s.substr(0,i) + '.' + s.substr(i);
                    res.push_back(temp);
                }
                return res;
            }
            
        }
        
    }
    void helper(string s1,string s2,vector<string> &res){
        vector<string> s1list = checknum(s1);
        vector<string> s2list = checknum(s2);
        for(auto is1:s1list){
            for(auto is2:s2list){
                string temp = '(' + is1 +", "+ is2+')';
                res.push_back(temp);
            }
        }
    }
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> res;
        int n = S.length();
        for(int i = 2; i < n-1;i++){
            helper(S.substr(1,i-1),S.substr(i,n-1-i),res);
        }
        return res;
    }
};
