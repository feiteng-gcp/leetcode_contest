class Solution {
    bool valid(string s) {
        int nonz = s.find_first_not_of('0');
        if (nonz == string::npos) return s == "0";
        if (nonz > 1) return false;
        if (nonz == 1 && s[nonz] != '.') return false;
        int dot = s.find('.');
        if (dot != string::npos && s.back() == '0') return false;
        return true;
    }
public:
    vector<string> ambiguousCoordinates(string S) {
        S = S.substr(1, S.length() - 2);
        int n = S.length();
        vector<string> ans;
        for (int i = 1; i < n; i++) {
            string x = S.substr(0, i), y = S.substr(i);
            string intg, deci, t, u;
            for (int j = 1; j <= i; j++) {
                intg = x.substr(0, j), deci = x.substr(j);
                t = intg + (j == i ? "" : ".") + deci;
                if (!valid(t)) continue;
                for (int k = 1; k <= n - i; k++) {
                    intg = y.substr(0, k), deci = y.substr(k);
                    u = intg + (k == n - i ? "" : ".") + deci;
                    if (valid(u)) {
                        ans.push_back("(" + t + ", " + u + ")");
                    }
                }
            }
        }
        return ans;
    }
};
