class Solution {
public:
    vector<string> F(string s) {
	vector<string> ret;
	if (s.empty()) return ret;
	if (s.size() == 1u || s[0] != '0') ret.push_back(s);

	if (s.back() != '0') {
	    int n = s.size();

	    for (int i=1; i<n; i++) {
		if (i == 1 || s[0] != '0') {
		    ret.push_back(s.substr(0, i) + "." + s.substr(i));
		}
	    }

	}
	return ret;
    }

    vector<string> ambiguousCoordinates(string S) {
	vector<string> ret;
	int n = S.size();
	for (int i=2; i<n-1; i++) {
	    vector<string> left = F(S.substr(1, i-1));
	    vector<string> right = F(S.substr(i, n-1-i));

	    EACH (e, left) EACH (f, right) {
		ret.push_back(string("(") + *e + ", " + *f + ")");
	    }
	}
        
	return ret;
    }
};
