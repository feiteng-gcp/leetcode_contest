class Solution {
public:
    int leadingZero(const string & s) {
        int cnt = 0;
        for (int i = 0; i < s.size() && s[i] == '0'; ++i)
            ++cnt;
        return cnt;
    }
    bool allZero(const string & s) {
        return leadingZero(s) == s.size();
    }
    vector<string> generate(const string & S) {
        if (S == "0") return {"0"};
        if (S.empty() || allZero(S)) return {};

        vector<string> ans;
        if (leadingZero(S) == 0)
            ans.push_back(S);
        for (int i = 1; i < S.size(); ++i) {
            string lp = string(S.begin(), S.begin() + i);
            string rp = string(S.begin() + i, S.end());
            if (lp.empty() || rp.empty()) continue;
            int lz = leadingZero(lp);
            if (lz > 0 && lp.size() != 1) 
                break; // 0*.*,
            if (rp[rp.size() - 1] == '0') 
                continue; // *.*0
            ans.push_back(lp + "." + rp);
        }
        return ans;
    }
    
    vector<string> ambiguousCoordinates(string S) {
        vector<string> ans;
        S = string(S.begin() + 1, S.end() - 1);// strip off '(' ')'
        if (S.empty()) return {};

        for (int i = 1; i < S.size(); ++i) {
            string lp = string(S.begin(), S.begin() + i);
            string rp = string(S.begin() + i, S.end());
            vector<string> lps = generate(lp);
            vector<string> rps = generate(rp);
            for(int i = 0; i < lps.size(); ++i) {
                for (int j = 0; j < rps.size(); ++j) {
                    ans.push_back("(" + lps[i] + ", " + rps[j] + ")");
                }
            }
        }
        return ans;
    }
};
