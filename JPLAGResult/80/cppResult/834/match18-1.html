class Solution {
private:
    bool toNumber(const string& s, string& t, int dec_pos) {
        int lens = s.length();
        if (dec_pos == lens - 1) {
            // check if s is valid
            if (s == "0" || s[0] != '0') {
                t = s;
                return true;
            }
            return false;
        } else {
            if (s[lens - 1] == '0') return false;
            if (s[0] == '0') {
                if (dec_pos == 0) {
                    t = "0.";
                    for (int i = 1; i < lens; i++) t += s[i];
                    return true;
                } else {
                    return false;
                }
            }
        }
        t.clear();
        for (int i = 0; i <= dec_pos; i++) t += s[i];
        t += '.';
        for (int i = dec_pos + 1; i < lens; i++) t += s[i];
        return true;
    }
public:
    vector<string> ambiguousCoordinates(string S) {
        string T = S.substr(1, S.length() - 2);
        vector<string> result;

        for (int i = 0; i < T.length() - 1; i++) {
            string u = T.substr(0, i + 1);
            string v = T.substr(i + 1);
            for (int d1 = 0; d1 < u.length(); d1++) {
                for (int d2 = 0; d2 < v.length(); d2++) {
                    string t1;
                    string t2;
                    if (toNumber(u, t1, d1) && toNumber(v, t2, d2)) {
                        result.push_back("(" + t1 + ", " + t2 + ")");
                    }                    
                }
            }
        }
        return result;
    }
};
