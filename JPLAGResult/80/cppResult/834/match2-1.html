class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string>ans;
        for(int i = 1; i+2 < S.size(); i++)
        {
            vector<string>first = help(S.substr(1,i));
            vector<string>second = help(S.substr(i+1));
            for(auto e : first)
                for(auto f : second)
                    ans.push_back("("+e+", "+f+")");
        }
        return ans;
    }
    
    vector<string> help(string s)
    {
        if(s.back() == ')') s.pop_back();
        vector<string>ans;
        if(valid(s, 0)) ans.push_back(s);
        for(int i = 1; i < s.size(); i++)
        {
            if(valid(s.substr(0,i), 0) && valid(s.substr(i), 1))
                ans.push_back(s.substr(0,i) + "." + s.substr(i));
        }
        return ans;
    }
    
    bool valid(string s, int i)
    {
        if(i > 0 && s.back() == '0')
            return false;
        if(s.size() == 1) 
            return true;
        if(i == 0 && s[0] == '0')
            return false;
        return true;
    }
};
