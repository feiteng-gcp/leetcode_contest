class Solution {
public:
    bool valid(string s) {
        int n = s.size();
        if (n == 0) return false; 
        if (s[0] == '.' || s[n - 1] == '.') return false;
        bool p = false;
        for (char c: s) if (c == '.') p = true;
        if (p && s[n - 1] == '0') return false;
        if (n >= 3 && s[0] == '0' && s[1] == '.') return true;
        if (n == 1 && s[0] == '0') return true;
        if (s[0] != '0') return true;
        return false;
    }
    vector<string> ambiguousCoordinates(string S) {
        int n = S.size();
        vector<string> ret;
        S = S.substr(1, n - 2);
        n = S.size();
        for (int i = 1; i < n; i++) {
            string a = S.substr(0, i);
            string b = S.substr(i);
            int na = a.size();

            int nb = b.size();
            for (int i = 0; i < na; i++) {
                for (int j = 0; j < nb; j++) {
                    string c = a, d = b;
                    if (i != 0) c.insert(c.begin() + i, '.');
                    if (j != 0) d.insert(d.begin() + j, '.');
                    if (valid(c) && valid(d)) {
                        ret.push_back("(" + c + ", " + d + ")");
                    }                    
                }
            }
        }
        return ret;
    }
};
