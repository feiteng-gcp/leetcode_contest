class Solution {
public:
    bool valid(string s, int dot){
        if(s.size() == 1) return true;
        if(s[0]=='0' and s[1]!='.') return false;
        if(dot!=-1 and s[s.size()-1]=='0') return false;
        return true;
    }
    
    void gen(vector<string>& v, string s){
        // cout<<s<<endl;
        if(valid(s,-1)) v.push_back(s);
        string t;
        for(int i=1;i<s.size();i++){
            t = s.substr(0,i)+"."+s.substr(i);
            // cout<<t<<endl;
            if(valid(t,i)) v.push_back(t);
        }
        // cout<<v.size()<<endl;
    }
    
    vector<string> ambiguousCoordinates(string s) {
        map<pair<string,string>,bool> mp;
        s.pop_back();
        vector<string> l;
        vector<string> r;
        for(int i=1;i<s.size()-1;i++){
            l.clear();
            r.clear();
            // cout<<s.substr(1,i)<<" "<<s.substr(i+1)<<endl;
            gen(l,s.substr(1,i));
            gen(r,s.substr(i+1));
            
            // for(auto x : l) cout<<x<<endl;
            // for(auto x : r) cout<<x<<endl;
            
            for(int i=0;i<l.size();i++){
                for(int j=0;j<r.size();j++){
                    mp[{l[i],r[j]}] = true;
                }
            }
        }
        vector<string> ans;
        string t;
        for(auto x : mp){
            // cout<<x.first.first<<" "<<x.first.second<<endl;
            t = "(" + x.first.first + ", " + x.first.second + ")";
            ans.push_back(t);
        }
        return ans;
    }
};
