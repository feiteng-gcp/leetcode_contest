class Solution {
    bool testhead(const string &s) {
        if (s.size() < 2) return true;
        if (s[0] == '0' && s[1] == '0') return false;
        return true;
    }
    
    bool testtail(const string &s) {
        return s.back() != '0';
    }
    
    bool invalid(const string &s) {
        if (!testhead(s) && !testtail(s)) return true;
        if (s.size() > 1 && s[0] == '0') {
            if (!testtail(s)) return true;
        }
        return false;
    }
    
    vector<string> generate(const string &s) {
        int N = s.size();
        vector<string> res;
       
        if (testhead(s)) {
            if (s[0] != '0') {
                res.push_back(s);
                if (testtail(s)) {
                    for (int i = 1; i < N; ++i) {
                        res.push_back(s);
                        res.back().insert(i, 1, '.');
                    }
                }
            } else {
                res.push_back(s);
                if (s.size() > 1) res.back().insert(1, 1, '.');
            }
        } else {
            res.push_back(s);
            res.back().insert(1, 1, '.');
        }
        return res;
    }
    
public:
    vector<string> ambiguousCoordinates(string S) {
        int N = S.size() - 2;
        S = S.substr(1, N);
        
        vector<string> res;
        
        int hi = N - 1;
        while (hi > 0) {
            string a = S.substr(0, hi);
            string b = S.substr(hi, N - hi);
            if (!invalid(a) && !invalid(b)) {
                auto gen_a = generate(a);
                auto gen_b = generate(b);
                for (const auto & s1 : gen_a) {
                    for (const auto & s2 : gen_b) {
                        res.push_back("(" + s1 + ", " + s2 + ")");
                    }
                }
            }
            --hi;
        }
        return res;
    }
};
