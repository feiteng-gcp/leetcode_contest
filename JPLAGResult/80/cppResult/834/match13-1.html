class Solution {
public:
    
    vector<string> getSolutions(string s){
        vector<string> ans;
        if(s.size() == 0){
            return ans;
        }
        // without point
        if(s.front() != '0' || s == "0"){
            ans.push_back(s);
        }
        // with point
        for(int i = 1; i < s.size(); ++i){
            if(s.back() != '0'){
                string w = s.substr(0 , i);

                if(w.front() != '0' || w == "0"){
                    ans.push_back(w + "." + s.substr(i));
                }
            }
        }
        return ans;
    }
    
    vector<string> ambiguousCoordinates(string s) {
        
        set<string> ret;
        for(int i = 1; i + 2 < s.size(); ++i){
           // cout << s.substr(1 , i)<<" "<<s.substr(i + 1 , s.size() - i - 2)<<endl;
            vector<string> x = getSolutions(s.substr(1 , i));
            vector<string> y = getSolutions(s.substr(i + 1 , s.size() - i - 2));
            for(auto w1 : x){
                for(auto w2 : y){
                    ret.insert("(" + w1 + ", " + w2 + ")");
                }
            }
        }
        vector<string> ans(ret.begin() , ret.end());
        return ans;
    }
};
