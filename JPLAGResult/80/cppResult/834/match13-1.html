class Solution {
private:
    vector<string> get_str(string& str) {
        vector<string> v;
        int len = str.size();
        if (str[0] == '0') {
            if (str.size() == 1)
                v.push_back("0");
            else {
                int idx = 0;
                while (idx < len && str[idx] == '0')
                    idx++;
                if (idx < len && str.back() != '0') {
                    v.push_back(str.substr(0, 1) + "." + str.substr(1));
                }
            }
        } else {
            if (str.size() == 1)
                v.push_back(str);
            else {
                v.push_back(str);
                if (str.back() != '0')
                    for (int i = 1; i < len; i++) {
                            v.push_back(str.substr(0, i) + "." + str.substr(i));
                    }
            }

        }
        return v;
    }
    
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> ans;
        string str = S.substr(1, S.size() - 2);
        for (int i = 0; i < str.size() - 1; i++) {
            string pre = str.substr(0, i + 1);
            string suf = str.substr(i + 1);
            auto vp = get_str(pre);
            auto vs = get_str(suf);
            if (vp.size() && vs.size()) {
                for (auto & sp : vp)
                    for (auto & ss : vs)
                        ans.push_back("(" + sp + ", " + ss + ")");
            }
        }
        return ans;
    }
};
