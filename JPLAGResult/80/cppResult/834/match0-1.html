class Solution {
public:
    
    bool correct(string s){
        if(s.size()==0)return 0;
        
        if(s.size()>=2){
            if(s[0]=='0' && s[1]=='0')return 0;
            if(s[0]=='0' && s[1]!='.')return 0;
        }
        bool haspoint=0;
        
        for(int i=0;i<s.size();i++)
            if(s[i]=='.')haspoint=1;
        

        if(haspoint && s[s.size()-1]=='0')return 0;
        return 1;
    }
    
    vector<string>g(string s){
        vector<string>dev;    
        
        if(correct(s))dev.push_back(s);
        
        for(int i=1;i<s.size();i++){
            string h="";
            h+=s.substr(0,i);
            h+=".";
            h+=s.substr(i);
            if(correct(h))dev.push_back(h);
        }
        
        return dev;
    }
    
    
    vector<string>f(string a,string b){
        vector<string>x= g(a);
        vector<string>y= g(b);
        vector<string>dev;
        
        for(int i=0;i<x.size();i++){
            for(int j=0;j<y.size();j++){
                string h="";
                h+="(";
                h+=x[i];
                h+=", ";
                h+=y[j];
                h+=")";
                dev.push_back(h);
            }

        }
        
        return dev;
    }
    
    vector<string> ambiguousCoordinates(string s) {
        s=s.substr(1,s.size()-2);
        vector<string>v;
        for(int i=1;i<s.size();i++){
            string a=s.substr(0,i);
            string b=s.substr(i);
            vector<string>x=f(a,b);
            for(int j=0;j<x.size();j++)v.push_back(x[j]);
        }
        return v;
    }
};
