class Solution {
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> rs;
        for(int i = 1; i < S.length() - 2; i++) {
            string s1 = S.substr(1, i), s2 = S.substr(i + 1, S.length() - 2 - i);
            // cout << s1 << ", s2: " << s2 << endl;
            for(int j = 0; j < s1.length(); j++) {
                string sj(s1);
                if (j > 0) {
                    sj.insert(j, 1, '.');
                }
                // cout << s1 << ", j: " << j << ", sj: " << sj << endl;
                if (isValid(sj)) {
                    for(int k = 0; k < s2.length(); k++) {
                        string sk(s2);
                        if (k > 0) {
                            sk.insert(sk.begin() + k, 1, '.');
                        }
                        if (isValid(sk)) {
                            rs.push_back("(" + sj + ", " + sk + ")");
                        }
                    }
                }
            }
        }
        return rs;
    }
    
    bool isValid(string &s) {
        if (s.length() <= 0) return false;
        if (s.length() > 1 && s[0] == '0' && s[1] != '.') return false;
        if (s.length() > 1) {
            int nonZero = 0, dots = 0;
            for(auto c : s) {
                nonZero += c != '0' && c != '.';
                dots += c == '.';
            }
            if (dots > 0 && s.back() == '0') return false;
            if (nonZero == 0) return false;
        }
        return true;
    }
};
