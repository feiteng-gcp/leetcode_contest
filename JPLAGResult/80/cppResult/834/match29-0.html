class Solution {
public:
    vector<string> answer;
    vector<string> ambiguousCoordinates(string S) {
        for (int i = 1; i < S.length() - 1; i ++) {
            gao(S.substr(1, i), S.substr(i+1, S.length() - 2 - i));
        }

        return answer;
    }

    void gao(string s1, string s2) {
        if (s1 == "" || s2 == "")   return;
        vector<string> nums1 = toValidNums(s1);
        vector<string> nums2 = toValidNums(s2);
        for (const string &s1: nums1) {
            for (const string &s2: nums2) {
                answer.push_back('('+s1+", "+s2+')');
            }
        }
    }

    vector<string> toValidNums(string str) {
        vector<string> answer;
        if (str.length() == 1) {
            answer.push_back(str);
        } else {
            // not add decimal point
            if (str[0] != '0') {
                answer.push_back(str);
            }
            // add decimal point after i
            bool canAdd = true;
            for(int i = str.length() - 1; ~i; i --) {
                if (str[i] == '0') {
                    canAdd = false;
                    break;
                } else {
                    break;
                }
            }
            if (canAdd)  {
                answer.push_back(str.substr(0, 1) + '.' + str.substr(1));
                if (str[0] != '0') {
                    for (int i = 1; i < str.length() - 1; i ++) {
                        answer.push_back(str.substr(0, i+1) + '.' + str.substr(i+1));
                    }
                }
            }
        }

        return answer;
    }
};
