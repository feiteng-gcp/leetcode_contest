class Solution {
    vector<string> getPoss(string s){
        vector<string> ret;
        
        if(s.size() == 1){
            ret.push_back(s);
            return ret;
        }
        int only_zero = 1;
        int lz = 0, azt = 1;
         if(s[0] == '0') lz = 1;
        for(int i = 0; i < s.size(); i++){
           if(s[i]  != '0') azt = 0;
        }
        if(azt ) return ret;
        if(lz ){
             if( s[s.size()-1] == '0') return ret;
            string x = "";
            x +=s[0];
            x += ".";
            for(int i = 1; i < s.size(); i++){
                x += s[i];
            }
            ret.push_back(x);
            return ret;
        }
        for(int i = 0; i < s.size()-1; i++){
            if(s[i] != '0') only_zero = 0;
            if(only_zero) continue;
            string x1 = s.substr(0, i+1);
            int az = 1;
            int as = 0;
            for(int j = i+1; j < s.size(); j++){
                as++;
                if(s[j] != '0') {
                    az = 0;
                    break;
                }
                
            }
            if(az || s[s.size()-1] == '0') continue;
            string x2 = s.substr(i+1);
            string res = x1 + "." + x2;    
            ret.push_back(res);
        }
        if(s[0] != '0'){
            ret.push_back(s);
        }
        return ret;
        
    }
public:
    vector<string> ambiguousCoordinates(string S) {
        vector<string> ans;
        for(int i = 1; i < S.size()-2; i++){
            string pre = S.substr(1, i);
            string post = S.substr(i+1, S.size()-i-2);
            cout << pre << " " << post << "\n";
            vector<string> t1, t2;
            t1 = getPoss(pre);
            t2 = getPoss(post);
            if(t1.size() == 0 || t2.size() == 0){
                continue;
            }
            for(int i = 0; i < t1.size(); i++){
                for(int j = 0; j < t2.size(); j++){
                    string pa = "(" + t1[i] + ", " + t2[j]  + ")";
                    ans.push_back(pa);
                }
            }
        }
        return ans;
        
        
    }
};
