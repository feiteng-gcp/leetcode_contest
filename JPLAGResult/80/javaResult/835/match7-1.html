/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int numComponents(ListNode head, int[] G) {
        if (G == null || G.length == 0) {
            return 0;
        }
        
Set<Integer> vals = new HashSet<>();
        for (int g : G) {
            vals.add(g);
        }
        int count = 0;
        boolean inComponent = false;
        ListNode cur = head;
        while (cur != null) {
            if (inComponent && !vals.contains(cur.val)) {
                inComponent = false;
            } else if (!inComponent && vals.contains(cur.val)) {
                inComponent = true;
                count++;
            }
            cur = cur.next;
        }
        return count;
    }
}
