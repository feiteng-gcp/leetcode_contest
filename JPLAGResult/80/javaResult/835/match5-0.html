	class Solution {
	    public int numComponents(ListNode head, int[] G) {
	    	boolean[] hit = new boolean[10005];
	    	for(int v : G)hit[v] = true;
	    	
	    	int rep = 0;
	    	int con = 0;
	    	while(head != null){
	    		if(hit[head.val]){
	    			if(rep == 0)con++;
	    			rep++;
	    		}else{
	    			rep = 0;
	    		}
	    		
	    		head = head.next;
	    	}
	    	return con;
	    }
	}
