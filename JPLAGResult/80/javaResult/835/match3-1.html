/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    	public int numComponents(ListNode head, int[] G) {
		Set<Integer> g
Set = new HashSet<>();
		for (int integer : G) {
			gSet.add(integer);
		}
		int res = 0;
		ListNode cur = head;
		while (cur != null) {
			// 跳过不在g的
			while (cur != null && !gSet.contains(cur.val)) {
				cur = cur.next;
			}
			// 连续在g的
			if (cur != null) {
				res++;
				while (cur != null && gSet.contains(cur.val)) {
					cur = cur.next;
				}
			}
		}
		return res;
	}
}
