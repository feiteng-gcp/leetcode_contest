class Solution {
    public List<String> ambiguousCoordinates(String S) {
		List<String> res = new ArrayList<>();
		S = S.substring(1, S.length() - 1);
		for (int i = 0; i < S.length() - 1; i++) {
			List<String> leftList = help(S, 0, i);
			List<String> rightList = help(S, i + 1, S.length() - 1);
//			System.out.println("left = " + leftList);
//			System.out.println("right = " + rightList);
			if (leftList.size() != 0 && rightList.size() != 0) {
				for (String leftStr : leftList) {
					for (String rightStr : rightList) {
						res.add("(" + leftStr + ", " + rightStr + ")");
					}
				}
			}
		}
		return res;
	}

	private List<String> help(String S, int left, int right) {
		List<String> res = new ArrayList<>();
		for (int i = left + 1; i <= right; i++) {
			boolean leftC = i - left == 1 || (i - left >= 2 && S.charAt(left) != '0');
			boolean rightC = S.charAt(right) != '0';
			if (leftC && rightC) {
				res.add(S.substring(left, i) + "." + S.substring(i, right + 1));
			}
		}
		if (left == right || S.charAt(left) != '0') {
			res.add(S.substring(left, right + 1));
		}
		return res;
	}
}
