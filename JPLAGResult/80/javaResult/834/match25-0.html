class Solution {

    public List<String> ambiguousCoordinates(String s) {
        s = s.substring(1, s.length() - 1);
        List<String> coordinates = new ArrayList<>();
        for (int i = 1; i < s.length(); i++) {
            String first = s.substring(0, i);
            String second = s.substring(i);

            List<String> validFirstWords = getValidNumbers(first);
            List<String> validSecondWords = getValidNumbers(second);

            for (String validFirstWord : validFirstWords) {
                for (String validSecondWord : validSecondWords) {
                    coordinates.add("(" + validFirstWord + ", " + validSecondWord + ")");
                }
            }
        }

        return coordinates;
    }

    private List<String> getValidNumbers(String word) {
        List<String> validWords = new ArrayList<>();
        if (word.length() == 0) {
            return validWords;
        }

        int startNonZero = getStartingNonZero(word);
        if (startNonZero == -1) {
            if (word.length() > 1) {
                return validWords;
            } else {
                validWords.add("0");
                return validWords;
            }
        }

        if (startNonZero == 0) {
            validWords.add(word);
            for (int i = 1; i < word.length(); i++) {
                String first = word.substring(0, i);
                String second = word.substring(i);
                String period = getPeriod(first, second);
                if (period != null) {
                    validWords.add(period);
                }
            }
            return validWords;
        }

        String remaining = word.substring(1);
        if (remaining.charAt(remaining.length() - 1) != '0') {
            validWords.add("0." + word.substring(1));
        }
        return validWords;
    }

    private String getPeriod(String first, String second) {
        if (first == null || second == null) {
            return null;
        }

        if (first.length() == 0 || second.length() == 0) {
            return null;
        }

        if (second.charAt(second.length() - 1) == '0') {
            return null;
        }

        return first + '.' + second;
    }

    private int getStartingNonZero(String word) {
        for (int i = 0; i < word.length(); i++) {
            if (word.charAt(i) != '0') {
                return i;
            }
        }

        return -1;
    }
}
