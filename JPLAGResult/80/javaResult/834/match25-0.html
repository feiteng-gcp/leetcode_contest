class Solution {
    public List<String> ambiguousCoordinates(String S) {
    	List<String> ans = new ArrayList<String>();
    	S = S.replaceAll("[()]", "");
    	
    	for(int i = 1; i < S.length(); i++) {
    		String A = S.substring(0, i);
    		String B = S.substring(i, S.length());
    		
    		List<String> As = availableStrings(A);
    		List<String> Bs = availableStrings(B);    		
    		for(String AA: As) {
    			for(String BB: Bs) {
    				ans.add("(" + AA + ", " + BB + ")");
    			}
    		}
    	}
    	return ans;
    }
    
    private List<String> availableStrings(String S){
    	List<String> ans = new ArrayList<String>();
    	if(!S.startsWith("0") || S.equals("0")) ans.add(S);
    	for(int i = 1; i < S.length(); i++) {
    		String A = S.substring(0, i);
    		String B = S.substring(i, S.length());
    		if(A.startsWith("0") && !A.equals("0")) continue;
    		if(B.endsWith("0")) continue;
    		ans.add(A + "." + B);
    	}
    	
    	return ans;
    	
    }
}
