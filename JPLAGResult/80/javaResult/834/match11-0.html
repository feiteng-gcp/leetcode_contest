class Solution {
    public List<String> ambiguousCoordinates(String S) {
        List<String> res = new LinkedList<>();
        S = S.substring(1,S.length() - 1);
        for(int i = 1; i < S.length(); i++) {
            List<String> left = backtrack(S.substring(0,i));
            List<String> right = backtrack(S.substring(i));
            if(!left.isEmpty() && !right.isEmpty()) {
                for(String s1 : left) {
                    for(String s2 : right) {
                      res.add("(" + s1 +", " + s2 + ")");  
                    }
                }
            }
        }
        return res;
    }
    
    private List<String> backtrack(String num) {
        List<String> res = new LinkedList<>();
        if(num.length() > 1 && Integer.valueOf(num) == 0) return res;
        if(num.length() == 1) {
            res.add(num);
            return res;
        }
        if(num.charAt(0) == '0' || num.charAt(num.length() - 1) == '0') {
            if(num.charAt(0) == '0' && num.charAt(num.length() - 1) == '0') return res;
            if(num.charAt(0) == '0')
             res.add("0" + "." + num.substring(1));
            else 
                res.add(num);
            return res;
        }
        res.add(num);
        for(int i = 1; i < num.length(); i++) {
            if(Integer.valueOf(num.substring(i)) == 0) continue;
            res.add(num.substring(0,i) + "." + num.substring(i));
        }
        return res;
    }
}
