class Solution {

	private List<String> getList(String string) {
		
		List<String> ans = new ArrayList<>();
		int i , length = string.length();
		// possible integer
		if (string.charAt(0) == '0' && string.length() > 1) {
			;
		} else {
			ans.add(string);
		}
		// possible float number
		for (i = 0;i < length - 1;i ++) {
			String s1 = string.substring(0 , i + 1);
			String s2 = string.substring(i + 1);
			// start with zero
			if (s1.charAt(0) == '0' && s1.length() > 1) {
				continue;
			}
			// end with zero
			if (s2.charAt(s2.length() - 1) == '0') {
				continue;
			}
			ans.add(String.format("%s.%s" , s1 , s2));
		}

		return ans;
		
	}
