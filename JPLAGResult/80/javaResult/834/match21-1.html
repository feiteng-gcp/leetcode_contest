class Solution {
    	
		public List<String> ambiguousCoordinates(String S) {
        List<String> ans = new ArrayList<>();
        S = S.substring(1, S.length() - 1);
        for (int i = 1; i < S.length(); i++) {
        	String first = S.substring(0, i);
        	String second = S.substring(i);
        	List<String> firsts = helper(first);
        	List<String> seconds = helper(second);
        	for (String f : firsts) {
        		for (String s : seconds) {
        			String str = "(" + f + ", " + s + ")";
        			ans.add(str);
        		}
        	}

        }
        return ans;
    }
	
	private boolean isValidLong(String S) {
		Long val = Long.parseLong(S);
		String str = val.toString();
		return str.equals(S);
	}
	
	private List<String> helper(String S) {
		List<String> ans = new ArrayList<>();
		if (isValidLong(S)) {
			ans.add(S);
		}
		for (int i = 1
; i < S.length(); i ++) {
			String first = S.substring(0, i);
			String second = S.substring(i);
			if (second.charAt(second.length() - 1) != '0' && isValidLong(first)) {
				String str = first + "." + second;
				ans.add(str);
			}
		}
		return ans;
	}
}
