class Solution {
    public List<String> ambiguousCoordinates(String S) {
        List<String> res = new ArrayList<>();
		S = S.substring(1, S.length() - 1);
		for (int i = 1; i < S.length(); i++) {
			List<String> xs = possibleNumber(S.substring(0, i));
			List<String> ys = possibleNumber(S.substring(i));

			for (String x : xs)
				for (String y : ys)
					res.add("(" + x + ", " + y + ")");
		}
		return res;
	}

	private  List<String> possibleNumber(String s) {
		List<String> res = new ArrayList<>();

		if (s.charAt(0) != '0' || s.length() == 1)
			res.add(s);

		if (s.length() < 2 || s.charAt(s.length() - 1) == '0')
			return res;

		for (int i = 1; i < s.length(); i++) {
			String before = s.substring(0, i);
			if (before.charAt(0) == '0' && before.length() > 1)
				break;
			res.add(before + "." + s.substring(i));
		}
		
		return res;
    }
}
