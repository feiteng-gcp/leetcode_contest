    public List<String> ambiguousCoordinates(String S) {
        List<String> ans = new ArrayList<>();
        if (S == null || S.length() == 0) {
            return ans;
        }
        S = S.substring(1, S.length() - 1);
        for (int i = 1; i < S.length(); ++i) {
            List<String> list = check(S, i);
            for (String str : list) {
                ans.add(str);
            }
        }
        return ans;
    }
    
    private List<String> check(String s, int i) {
        String s1 = s.substring(0, i);
        String s2 = s.substring(i);
        // System.out.println(i + " " + s1 + " " + s2);
        List<String> list1 = transform(s1);
        List<String> list2 = transform(s2);
        List<String> ans = new ArrayList<>();
        for (String str1 : list1) {
            for (String str2 : list2) {
                // if (str1.length() == 0 || str2.length)
                StringBuilder sb = new StringBuilder();
                sb.append('(');
                sb.append(str1);
                sb.append(", ");
                sb.append(str2);
                sb.append(')');
                ans.add(sb.toString());
            }
        }
        return ans;
    }
    
    private List<String> transform(String s) {
        List<String> ans = new ArrayList<>();
        if (s.length() == 0) {
            return ans;
        }
        if (s.length() == 1) {
            ans.add(s);
            return ans;
        }
        int i = s.length() - 1;
        while (i >= 0) {
            if (s.charAt(i) != '0') {
                break;
            }
            --i;
        }
        if (i == -1) {
            return ans;
        }
        
        if (s.charAt(0) != '0') {
            ans.add(s);
            if (s.charAt(s.length() - 1) != '0') {
                for (int j = 1; j <= i; ++j) {
                    ans.add(s.substring(0, j) + "." + s.substring(j));
                }
            }
        } else {
            if (s.charAt(s.length() - 1) != '0') {
                ans.add(s.substring(0, 1) + "." + s.substring(1));
            }
        }
        return ans;
    }
}
