class Solution {
    public List<String> ambiguousCoordinates(String S) {
        // get rid of the "(" and ")"
        S = S.substring(1, S.length() - 1);
        List<String> coordinates = new ArrayList<>();
        // need to insert , to separate between x and y coordinate
        
for (int len = 1; len < S.length(); len++) {
            String x = S.substring(0, len);
            String y = S.substring(len);
            List<String> validXs = createNumbers(x);
            List<String> validYs = createNumbers(y);
            for (String xVal : validXs) {
                for (String yVal : validYs) {
                    
coordinates.add("(" + xVal + ", " + yVal + ")");
                }
            }
        }
        return coordinates;
    }
    
    
    /**
    * Our original representation never had extraneous zeroes, so we never started with numbers 
    * like "00", "0.0", "0.00", "1.0", "001", "00.01", or any other number that can be represented 
    * with less digits.  Also, a decimal point within a number never occurs without at least one digit 
    * occuring before it, so we never started with numbers like ".1".
    */
    private List<String> createNumbers(String s) {
        // need to insert decimal 
        List<String> possibilities = new ArrayList<>();
        if (s.charAt(0) == '0') {
            // s starts with zero, is a special case
            // must be either by itself (s == "0")
            // or have a decimal right after it (s == "0.XXX")
            if (s.length() == 1) {
                possibilities.add(s);
            } else {
                // cannot have "0.000" or "0.010"
                if (s.charAt(s.length() - 1) != '0') {
                    possibilities.add(s.substring(0, 1) + "." + s.substring(1));
                }
            }
        } else {
            possibilities.add(s); // no decimal
            
            // cannot have trailing zeroes like "45.30"
            if (s.length() > 1 && s.charAt(s.length() - 1) != '0') {
                for (int i = s.length() - 1; i > 0; i--) {
                    possibilities.add(s.substring(0, i) + "." + s.substring(i));
                }
            }
        }
        // System.out.println(s + " = " + possibilities);
        return possibilities;
    }
}
