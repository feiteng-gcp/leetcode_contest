class Solution {
    public List<String> ambiguousCoordinates(String S) {
        List<String> result = new ArrayList<>();
        S = S.substring(1, S.length() - 1);
        for (int i = 0; i < S.length() - 1; i++) {
            construct(S.substring(0, i + 1), S.substring(i + 1, S.length()), result);
        
}
        return result;
    }
    
    public void construct(String a, String b, List<String> result) {
        //System.out.println(a + "," + b);
        Set<String> res_a = search(a);
        Set<String> res_b = 
search(b);
        if (res_a.size() != 0 && res_b.size() != 0) {
            // construct
            for (String A : res_a) {
                for (String B : res_b) {
                    StringBuilder sb = new StringBuilder();
                    sb.append('(');
                    sb.append(A);
                    sb.append(',');
                    sb.append(' ');
                    sb.append(B);
                    sb.append(')');
                    result.add(sb.toString());
                }
            }
        }
    }
    
    public Set<String> search(String s) {
        Set<String> result = new HashSet<>();
        int end = s.length() - 1;
        if (s.length() == 1) {
            result.add(s);
            return result;
        }
        if (s.length() > 1 && Long.parseLong(s) == 0) {
            return result;
        }
        if (s.charAt(0) == '0') {
            if (s.charAt(end) != '0') {
                result.add("0." + s.substring(1, s.length()));
            }
            return result;
        }
        result.add(s);
        // trailing zeros
        if (s.charAt(end) == '0') {
            return result;
        }
        for (int i = 0; i < end; i++) {
            String former = s.substring(0, i + 1);
            String latter = s.substring(i + 1, s.length());
            String con1 = former + "." + latter;
            //String con2 = former;
            //if (Math.abs(Double.parseDouble(con1) - Double.parseDouble(con2)) < 1e-6) {
            //    continue;
            //}
            result.add(con1);
        }
        return result;
    }
}
