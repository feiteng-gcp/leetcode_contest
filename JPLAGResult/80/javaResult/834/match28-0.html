class Solution {
    public List<String> ambiguousCoordinates(String s) {
        List<String> res = new ArrayList<>();
        s = s.substring(1, s.length() - 1);
        for (int i = 0; i < s.length() - 1; i++) {
            String l = s.substring(0, i + 1), r = s.substring(i + 1);
            List<String> ls = addPoint(l), rs = addPoint(r);
            for (String l1 : ls) {
                for (String r1 : rs) {
                    StringBuilder sb = new StringBuilder();
                    sb.append("(");
                    sb.append(l1);
                    sb.append(", ");
                    sb.append(r1);
                    sb.append(")");
                    res.add(sb.toString());
                }
            }
        }
        return res;
    }
    
    private List<String> addPoint(String s) {
        List<String> res = new ArrayList<>();
        if (s.charAt(0) != '0' || s.equals("0")) {
            res.add(s);
        }
        String l = "", r = "";
        for (int i = 0; i < s.length() - 1; i++) {
            l = s.substring(0, i + 1);
            r = s.substring(i + 1);
            if (l.length() > 1 && getFrontZeros(l) > 0) {
                continue;
            }
            if (getRearZeros(r) > 0) {
                continue;
            }
            res.add(l + "." + r);
        }
        return res;
    }
    
    private int getFrontZeros(String s) {
        int i = 0;
        for (; i < s.length() && s.charAt(i) == '0'; i++);
        return i;
    }

    private int getRearZeros(String s) {
        int i = s.length() - 1;
        for (; i >= 0 && s.charAt(i) == '0'; i--);
        return s.length() - 1 - i;
    }
}
