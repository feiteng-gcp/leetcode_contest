class Solution {
    public List<String> ambiguousCoordinates(String s) {
        List<String> res=new ArrayList<>();
        s=s.substring(1, s.length()-1);
        for(int i=1; i<s.length(); i++){
            List<String> left=helper(s.substring(0, i));
            List<String> right=helper(s.substring(i));
            for(String tmp: left){
                for(String tmp1: right){
                    res.add("("+tmp+", "+tmp1+")");
                }
            }
        }
        return res;
    }
    
    public List<String> helper(String s){
        List<String> res=new ArrayList<>();
        int l=s.length();
        if(l==1){
            res.add(s);
            return res;
        }
        
        boolean allZeros=true;
        for(char c: s.toCharArray()){
            if(c!='0'){
                allZeros=false;
                break;
            }
        }
        
        if(allZeros || s.charAt(0)=='0'&&s.charAt(l-1)=='0') return res;
        
        if(s.charAt(0)=='0'){
            res.add("0."+s.substring(1));
        }
        else if(s.charAt(l-1)=='0'){
            res.add(s);
        }
        else{
            res.add(s);
            for(int i=1; i<l; i++){
                res.add(s.substring(0, i)+"."+s.substring(i));
            }
        }
        
        return res;
    }
}
