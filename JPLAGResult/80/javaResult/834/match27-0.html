import java.util.*;
import java.util.stream.*;
import java.math.*;

public class Solution {


    Map<String, List<String>> map = new HashMap<>();

    public List<String> ambiguousCoordinates(String S) {
        S = S.substring(1, S.length() - 1);

        List<String> ans = new ArrayList<>();
        for (int len = 1; len < S.length(); len++) {
            for (String left : get(S.substring(0, len)))
                for (String right : get(S.substring(len))) {
                    ans.add("(" + left + ", " + right + ")");
                }
        }
        return ans;
    }

    List<String> get(String input) {
        if (!map.containsKey(input)) {
            List<String> ans = new ArrayList<>();
            if (input.charAt(0) == '0') {
                if (input.equals("0")) {
                    ans.add(input);
                }
            } else {
                ans.add(input);
            }

            // Add a decimal point
            if (input.charAt(input.length() - 1) != '0') {
                if (input.charAt(0) == '0') {
                    ans.add("0." + input.substring(1));
                } else {
                    for (int len = 1; len < input.length(); len++) {
                        ans.add(input.substring(0, len) + "." + input.substring(len));
                    }
                }
            }
            map.put(input, ans);
        }
        return map.get(input);
    }


    public static void main(String[] args) {
        Solution sol = new Solution();

        System.out.println(sol.ambiguousCoordinates("(123)"));
        System.out.println(sol.ambiguousCoordinates("(00011)"));
        System.out.println(sol.ambiguousCoordinates("(000000)"));
        System.out.println(sol.ambiguousCoordinates("(100)"));
        System.out.println(sol.ambiguousCoordinates("(000)"));
        System.out.println(sol.ambiguousCoordinates("(00)"));
    }
}
