class Solution {
    public List<String> ambiguousCoordinates(String S) {
        List<String> list = new ArrayList<>();
        S = S.substring(1, S.length() - 1);
        for (int i = 0; i < S.length() - 1; i++)
            multiple(list, S.substring(0, i + 1), S.substring(i + 1))
;
        return list;
    }
    
    private void multiple(List<String> list, String left, String right) {
        // System.out.println(left + " * " + right);
        // System.out.println(getNumbers(left));
        // System.out.println(getNumbers(right));
        // System.out.println();
        List<String> leftList = getNumbers(left);
        List<String> rightList = getNumbers(right);
        if (leftList.size() > 0 && rightList.size() > 0) {
            for (String leftNumber: leftList) for (String rightNumber: rightList)
                list.add("(" + leftNumber + ", " + rightNumber + ")")
;
        }
    }
    
    private List<String> getNumbers(String str) {
        List<String> result = new ArrayList<>();
        if (str.length() == 1 || !str.startsWith("0")) result.add(str);
        for (int i = 0; i < str.length() - 1; i++) {
            if (valid(str.substring(0, i + 1)) && !str.substring(i + 1).endsWith("0"))
                result.add(str.substring(0, i + 1) + "." + str.substring(i + 1));
        }
        return result;
    }
    
    private boolean valid(String str) {
        if (str.length() == 1) return true;
        return str.charAt(0) != '0';
    }
    
    public static void main(String[] args) {
        System.out.println(new Solution().ambiguousCoordinates("(123)"));
        System.out.println(new Solution().ambiguousCoordinates("(00011)"));
        System.out.println(new Solution().ambiguousCoordinates("(0123)"));
        System.out.println(new Solution().ambiguousCoordinates("(100)"));
    }
}
