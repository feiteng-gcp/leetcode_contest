class Solution {
    public List<String> ambiguousCoordinates(String S) {
        List<String> res =new ArrayList<>();

        for (int i = 1; i < S.length() -2; i++) {
            List<String> al1 = getCand(S, 1, i);
            List<String> al2 = getCand(S, i+1, S.length()-2);
            
            if (!al1.isEmpty() && !al2.isEmpty()) {
                for (int p1 = 0; p1 < al1.size(); p1++) {
                    for (int p2 = 0; p2 < al2.size(); p2++) {
                        StringBuilder sb = new StringBuilder();
                        sb.append("(");
                        sb.append(al1.get(p1));
                        sb.append(", ");
                        sb.append(al2.get(p2));
                        sb.append(")");
                        res.add(sb.toString());
                    }
                }
            }
        }
        
        return res;
        
    }
    
    private List<String> getCand(String S, int p1, int p2) {
        
        List<String> res = new ArrayList<>();
        boolean f = true;
        StringBuilder sb = new StringBuilder();
        for (int i = p1 ;i <= p2; i++) {
            if (p2 > p1 && S.charAt(p1) == '0') {
                f = false;
                break;
            }
            sb.append(S.charAt(i));
        }
        
        if (f) {
            res.add(sb.toString());
        }
        
        
        if (p2 > p1) {
        for (int i = p1; i < p2; i++) {
            sb.setLength(0);
            String pre = S.substring(p1, i+1);
            if (!checkPre(pre)) {
                break;
            }
            sb.append(pre);
            sb.append(".");
            String aft = S.substring(i+1, p2+1);
            
            if (!checkAft(aft)) {
                break;
            }
            sb.append(aft);
            res.add(sb.toString());
        }
        }
        return res;
        
    }
    
    private boolean checkPre(String pre) {
        if (pre.length() > 1 && pre.charAt(0) == '0') {
            return false;
        } else {
            return true;
        }
    }
    
    private boolean checkAft(String aft) {
        return !(aft.charAt(aft.length()-1) == '0');
    }
    
}
