class Solution {
    public List<String> ambiguousCoordinates(String S) {
        List<String> ret = new ArrayList<>();
        for (int i = 2; i < S.length() - 1; i++) {
            List<String> as = helper(S.substring(1, i));
            List<String> bs = helper(S.substring(i, S.length()-1));
            for (String a : as) {
                for (String b : bs) 
{
                    StringBuilder sb = new StringBuilder();
                    sb.append('(');
                    sb.append(a);
                    sb.append(", ");
                    sb.append(b);
                    sb.append(')');
                    ret.add(sb.toString());
                }
            }
        
}
        return ret;
    }
    private List<String> helper(String s) {
        List<String> ret = new ArrayList<>();
        if (s.isEmpty()) return ret
;
        if (s.length() == 1) {
            ret.add(s);
            return ret;
        }
        if (s.charAt(0) == '0') {
            if (s.charAt(s.length() - 1) == '0') return ret;
            StringBuilder sb = new StringBuilder(s);
            sb.insert(1, '.');
            ret.add(sb.toString());
            return ret;
        }
        if (s.charAt(s.length() - 1) == '0') {
            
ret.add(s);
            return ret;
        }
        for (int i = 1; i < s.length(); i++) {
            StringBuilder sb = new StringBuilder(s);
            sb.insert(i, '.');
            ret.add(sb.toString());
        }
        ret.add(s);
        return ret;
    }
}
