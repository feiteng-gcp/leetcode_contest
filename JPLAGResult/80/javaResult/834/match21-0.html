    public List<String> ambiguousCoordinates(String S) {
        
        List<String> res = new LinkedList<>();
        
        S = S.substring(1, S.length()-1);
        
        for (int i = 1; i < S.length(); i++) {
            List<String> fir = posNum(S.substring(0, i));
            List<String> sec = posNum(S.substring(i));
            
            for (String f : fir)
                for (String s : sec)
                    res.add("(" + f + ", " + s + ")");
        }
        
        return res;
    }
    
    List<String> posNum(String s) {
        List<String> ret = new LinkedList<>();
        for (int i = 1; i < s.length(); i++) {
            String head = s.substring(0,i);
            String tail = s.substring(i);
            if (!validHead(head) || !validTail(tail)) break;
            ret.add(head + "." + tail);
        }
        
        if (validHead(s)) ret.add(s);
        return ret;
    }
    
    boolean validHead(String s) {
        if (s.charAt(0) == '0' && s.length() != 1) return false;
        else return true;
    }
    
    boolean validTail(String s) {
        if (s.length() == 0) return true;
        return s.charAt(s.length()-1) != '0';
    }
}
