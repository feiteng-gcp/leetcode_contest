class Solution {
    List<String> ans =  new ArrayList<String>();
    public List<String> ambiguousCoordinates(String S) {
        
S = S.substring(1, S.length() - 1);
        //System.out.println(S);
        for(int i = 1; i < S.length(); i ++) {
            String str1 = S.substring(0, i);
            if(str1.length() > 1 && allZero(str1))
                continue;
            String str2 = S.substring(i);
            if(str2.length() > 1 && allZero(str2))
                continue; 
            //System.out.println(str1);
            //System.out.println(str2);
            List<String> l1 = makeList(str1);
            List<String> l2 = makeList(str2);
            makeAns(l1, l2);
        
}
        return ans;
    }
    
    List<String> makeList(String str) {
        List<String> list = new ArrayList<String>();
        if(str.charAt(0) == '0') {
            if(str.length() == 1)
                list.add(str);
            else if(str.charAt(str.length() - 1) != '0')
                list.add("0." + str.substring(1));
            return list;
        }
        list.add(str);
        
for(int i = 1; i < str.length(); i ++) {
            String s1 = str.substring(0, i);
            if(s1.length() > 1 && allZero(s1))
                continue;
            String s2 = str.substring(i);
            if(s2.charAt(s2.length() - 1) == '0'|| allZero(s2) )
                continue;
            list.add(s1 + "." + s2);
        }
        return list;
    }
    
    private void makeAns(List<String> l1, List<String> l2) {
        for(String a : l1) {
            for(String b : l2) {
                String str = "(" + a + ", " + b + ")";
                    ans.add(str);
            }
        }        
    }
    
    private boolean allZero(String str) {
        for(int i = 0; i < str.length(); i ++) {
            if(str.charAt(i) != '0')
                return false;
        }
        return true;
    }
}
