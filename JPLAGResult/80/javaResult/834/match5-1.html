class Solution {
    public List<String> ambiguousCoordinates(String S) {
        List<String> res = new ArrayList<>();
        S = S.substring(1, S.length()-1);
        for(int i=1;i<S.length();i++) {
            String left = S.substring(0, i);
            String right = S.substring(i);
            List<String> leftList = getCoor(left);
            List<String> rightList = 
getCoor(right);
            for(String l : leftList) {
                for(String r: rightList) {
                    res.add("("+l+", "+r+")");
                }
            }
        }
        return res;
    }
    
    private List<String> getCoor(String S) {
        List<String> res = new ArrayList<>();
        if(isValidInt(S)) res.add(S);
        for(int i=1;i<S.length();i++) {
            String left = S.substring(0, i);
            String right = S.substring(i);
            if(isValidInt(left) && isValidDeci(right)) {
                res.add(left + "." + right);
            
}
        }
        return res;
    }
    
    private boolean isValidInt(String s) {
        return s.charAt(0)!='0' || s.length()==1;
    }
    
    private boolean isValidDeci(String s) {
        return Integer.parseInt(s)!=0 && s.charAt(s.length()-1)!='0';
    }
}
