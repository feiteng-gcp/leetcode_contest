class Solution {
    public List<String> ambiguousCoordinates(String S) {
        List<String> resultList = new LinkedList<>();
        if (S == null || S.length() <= 2)
            return resultList;
        
        S = S.substring(1, S.length() - 1);
        
        for (int i = 1; i < S.length(); i++) {
            String left = S.substring(0, i);
            String right = S.substring(i, S.length());
            dfs(left, right, resultList);
        }
        
        return resultList;
    }
    
    private void dfs(String left, String right, List<String> resultList) {
        List<String> lefts = generateValidStr(left);
        List<String> rights = generateValidStr(right);
        
        for (String l : lefts) {
            for (String r : rights) {
                resultList.add("(" + l + ", " + r + ")");
            }
        }
    
}
    
    private List<String> generateValidStr(String s) {
        List<String> result = new LinkedList<>();
        
        if (isValid(s))
            result.add(s);
        
        int index = 1;
        while (index < s.length()) {
            String cur = s.substring(0, index) + '.' + s.substring(index, s.length());
            if (isValid(cur))
                result.add(cur);
            index++;
        }
        
        return result;
    }
    
    private boolean isValid(String s) {
        if (s.indexOf('.') == -1) {
            if (s.length() > 1 && s.charAt(0) == '0')
                return false;
        } else {
            if (s.charAt(s.length() - 1) == '0' || (s.charAt(0) == '0' && s.charAt(1) != '.'))
                return false;
        }
        
        return true;
    }
}
