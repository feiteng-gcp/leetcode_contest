class Solution {
    public List<String> ambiguousCoordinates(String S) {
        // 00011
        // 0 0011
        // 00 011
        // 000 11
        // 0001 1
        
        List<String> list = new ArrayList<>();
        if(S.length() <= 2) return list;
        S = S.substring(1, S.length()-1);
        for(int i=0; i<S.length(); i++) {
            String left = S.substring(0, i);
            String right = S.substring(i, S.length());
            helper(list, left, right);
        }
        return list;
    }
    
    public void helper(List<String> list , String left, String right) {
        List<String> leftList = partition(left);
        List<String> rightList = partition(right);
        for(String l : leftList) {
            // if(l.isEmpty()) continue;
            for(String r : rightList) {
                // if(r.isEmpty()) continue;
                list.add("(" + l + "," + " " + r +")");
            }
        }
        return;
    }
    
    public List<String> partition(String str) {
        List<String> list = new ArrayList<>();
        if(str.length() == 1) {
            list.add(str);
            return list;
        }
        
        if(str.replaceAll("0", "").isEmpty()) {
            return list;
        }
        
        if(str.startsWith("0") && str.charAt(str.length()-1) != '0' ) {
            list.add("0" + "." + str.substring(1, str.length()));
            return list;
        } else if(!str.startsWith("0") && str.charAt(str.length()-1) == '0') {
            list.add(str);
            return list;
        } else if(!str.startsWith("0") && str.charAt(str.length()-1) != '0'){
            list.add(str);
            for(int i=0; i<str.length()-1; i++) {
                list.add(str.substring(0,i+1) + "." + str.substring(i+1, str.length()));
            }
            return list;
        }
        return list;
    }
}
