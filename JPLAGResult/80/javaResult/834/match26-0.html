class Solution {
    public List<String> ambiguousCoordinates(String S) {
        String digits = S.substring(1, S.length() - 1);

        List<String> result = new ArrayList<>();

        for (int i = 1; i < digits.length(); i++) {
            String first = digits.substring(0, i);
            String second = digits.substring(i, digits.length());
            List<String> firstRes = getPossibleValues(first);
            if (firstRes == null) {
                continue;
            }
            List<String> secondRes = getPossibleValues(second);
            if (secondRes == null) {
                continue;
            }
            for (String f : firstRes) {
                for (String s : secondRes) {
                    result.add("(" + f + ", " + s + ")");
                }
            }
        }

        return result;
    }

    private List<String> getPossibleValues(String digits) {

        // all 0
        if (digits.length() > 1 && digits.replaceAll("[0]+", "").length() == 0) {
            return null;
        }
        // first and last are 0
        if (digits.length() > 1 && digits.charAt(0) == '0' && digits.charAt(digits.length() - 1) == '0') {
            return null;
        }

        List<String> result = new ArrayList<>();

        for (int i = 1; i < digits.length(); i++) {
            String first = digits.substring(0, i);
            String second = digits.substring(i, digits.length());
            if (first.length() > 1 && first.replaceAll("[0]+", "").length() == 0) {
                // at least two zeros at start
                break;
            }
            if (first.length() > 1 && first.charAt(0) == '0') {
                // cannot use 2 digits or more before decimal point
                break;
            }
            if (second.charAt(second.length() - 1) == '0') {
                // last char is 0: there cannot be decimal point at all
                break;
            }
            result.add(first + "." + second);
        }

        if (digits.charAt(0) != '0') {
            result.add(digits);
        }
        else {
            if (digits.length() == 1) {
                result.add(digits);
            }
        }

        return result;
    }
}
