class Solution {
    public List<String> ambiguousCoordinates(String S) {
        HashSet<String> ans = new HashSet<>();
        HashSet<List<String>> split = Split(S);

        for (List<String> pair : split) {
            String first = pair.get(0);
            String second = pair .get(1);
            
HashSet<String> firstSet = getAllValidFloat(first);
            HashSet<String> secondSet = getAllValidFloat(second);

            for (String firstfloat : firstSet) {
                for (String secondfloat : secondSet) {
                    ans.add("(" + firstfloat + ", " + secondfloat +")");
                }
            }
        }
        return new ArrayList<>(ans);
    
}

    private HashSet<String> getAllValidFloat(String first) {
        HashSet<String> set = new HashSet<>();
        if (first.length() > 1 && first.replaceAll("0", "").isEmpty()) {
            return set;
        }
        if (first.charAt(0) != '0') {    
            set.add(first);
        }
        if (first.length() == 1) {
            set.add(first);
            
return set;
        }
        if (first.charAt(0) == '0' && first.charAt(first.length() - 1) != '0') {
            set.add(first.substring(0, 1) + "." + first.substring(1));
            return set;
        }
        for (int i = 0; i < first.length() - 1; i++) {
            String laterPart = first.substring(i + 1);
            if (laterPart.charAt(laterPart.length() - 1) == '0') {
                continue;
            }
           

            
set.add(first.substring(0, i + 1) + "." + first.substring(i + 1));
        }
        return set;
    }



    public HashSet<List<String>> Split(String S) {
        HashSet<List<String>> splits = new HashSet<>();
        for (int i = 1; i  < S.length() - 2; i++) {
            List<String> array = new ArrayList<>(2);
            array.add(S.substring(1, i + 1));
            array.add(S.substring(i + 1, S.length() - 1));
            splits.add(array);
        }
        return splits;
    }
}
