import java.util.ArrayList;
import java.util.List;

class Solution {
    static List<String> getNumbers(String s) {
        List<String> result = new ArrayList<>();
        if (s.charAt(0) != '0' || s.length() == 1) result.add(s);
        if (s.length() > 1 && s.charAt(s.length() - 1) != '0') result.add(s.substring(0, 1) + "." + s.substring(1));
        if (s.charAt(0) != '0' && s.charAt(s.length() - 1) != '0') {
            for (int i = 2; i < s.length(); i++)
                result.add(s.substring(0, i) + "." + s.substring(i));
        }
        return result;
    }

    public List<String> ambiguousCoordinates(String S) {
        List<String> result = new ArrayList<>();
        for (int split = 2; split < S.length() - 1; split++) {
            List<String> firsts = getNumbers(S.substring(1, split));
            List<String> seconds = getNumbers(S.substring(split, S.length() - 1));
            for (String first : firsts) {
                for (String second : seconds) {
                    result.add(String.format("(%s, %s)", first, second));
                }
            }
        }
        return result;
    }
}
