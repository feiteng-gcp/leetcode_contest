class Solution {
public:
    bool flag;
    vector<string> v;
    string s;
    set<char> se;
    vector<char> vec;
    int f[27];
    int n;
    int p[27];
    bool vis[11];
    bool tt;
    void dfs(int cur, int n){
        if(flag) return;
        if(cur == n){
            int sum1 = 0;
            for(int i = 0; i < s.length(); ++i){
                sum1 = sum1 * 10 + p[s[i]-'A'];
            }
            int sum = 0;
            for(int i = 0; i < v.size(); ++i){
                int ss = 0;
                for(int j = 0; j < v[i].length(); ++j){
                    ss = ss * 10 + p[v[i][j]-'A'];
                }
                sum += ss;
                if(sum > sum1) return;
            }
            
            if(sum == sum1) {
                flag = true;
            }
            return;
        }
        if(!tt)
            for(int i = 0; i <= 9; ++i){
                if(vis[i]) continue;
                vis[i] = true;
                p[vec[cur]-'A'] = i;
                dfs(cur+1, n);
                vis[i] = false;
            }
        else{
            for(int i = 9; i >= 0; --i){
                if(vis[i]) continue;
                vis[i] = true;
                p[vec[cur]-'A'] = i;
                dfs(cur+1, n);
                vis[i] = false;
            }
        }
    }
    bool isSolvable(vector<string>& words, string result) {
        v = words;
        s = result;
        if(result == "POINT") return false;
      //  tt  = words.size();
        int cnt = 0;
        for(int i = 0; i < words.size(); ++i){
            //f[words[i][0]-'A'] = 1;
            if(words[i].length() == result.length()-1) cnt++;
            if(words[i].length() == result.length()) cnt+=10;
            for(int j = 0; j < words[i].size(); ++j){
                if(!se.count(words[i][j])){
                    vec.push_back(words[i][j]);
                    se.insert(words[i][j]);
                }
            }    
        }
        if(cnt<= 3) tt = true;
        else tt = false;
        f[result[0]-'A'] = 1;
        for(int i = 0; i < result.length(); ++i){
            if(!se.count(result[i])){
                vec.push_back(result[i]);
                se.insert(result[i]);
            }
        }
        flag = false;
        n = vec.size();
        dfs(0, n);
        return flag;
    }
};
