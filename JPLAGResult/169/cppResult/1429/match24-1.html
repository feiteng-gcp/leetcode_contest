vector<char> v;
int m[26];
int use[10];
bool ok;
int len[10];
int l;
int maxm;
class Solution {
public:
    void deal(int p,int cur,int tot,int sum,vector<string>& words,string result)
    {
        if(p==maxm)
        {
            ok=true;
            /*for(int i=0;i<words.size();i++)
            {
                for(int j=0;j<len[i];j++)
                {
                    printf("%d",m[words[i][j]-'A']);
                }
                printf("\n");
            }
            for(int i=0;i<l;i++)
            {
                printf("%d",m[result[i]-'A']);
            }
            printf("\n");*/
            return;
        }
        if(cur==tot)
        {
            if(m[result[l-1-p]-'A']!=-1)
            {
                if(m[result[l-1-p]-'A']!=sum%10)return;
                deal(p+1,0,tot,sum/10,words,result);
            }
            else
            {
                if(use[sum%10])return;
                m[result[l-1-p]-'A']=sum%10;   
                deal(p+1,0,tot,sum/10,words,result);
                m[result[l-1-p]-'A']=-1;
            }   
        }
        else
        {
            if(len[cur]-1-p>=0&&m[words[cur][len[cur]-1-p]-'A']==-1)
            {
                for(int i=0;i<10;i++)
                {
                    if(use[i])continue;
                    use[i]=1;
                    m[words[cur][len[cur]-1-p]-'A']=i;
                    deal(p,cur+1,tot,sum+i,words,result);
                    if(ok)return;
                    m[words[cur][len[cur]-1-p]-'A']=-1;
                    use[i]=0;
                }
            }
            else if(len[cur]-1-p<0)deal(p,cur+1,tot,sum,words,result);
            else deal(p,cur+1,tot,sum+m[words[cur][len[cur]-1-p]-'A'],words,result);
        }
    }
    bool isSolvable(vector<string>& words, string result) {
        v.clear();
        memset(m,-1,sizeof(m));
        memset(use,0,sizeof(use));
        int n=words.size();
        maxm=0;
        for(int i=0;i<n;i++)
        {
            len[i]=words[i].size();

            maxm=max(maxm,len[i]);
            for(int j=0;j<len[i];j++)
            {
                int f=0;
                for(int k=0;k<v.size();k++)
                {
                    if(v[k]==words[i][j])
                    {
                        f=1;
                        break;
                    }
                }
                if(!f)v.push_back(words[i][j]);
            }
        }
        l=result.size();
        maxm=max(maxm,l);
        if(l<maxm)return false;

        for(int i=0;i<l;i++)
        {
            int f=0;
                for(int k=0;k<v.size();k++)
                {
                    if(v[k]==result[i])
                    {
                        f=1;
                        break;
                    }
                }
                if(!f)v.push_back(result[i]);
        }
        int cnt=v.size();
        ok=false;
        deal(0,0,n,0,words,result);
        return ok;
    }
};
