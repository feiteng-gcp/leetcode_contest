class Solution {
public:
    //map<char, int> mp;
    int *mp;
    vector<int> cha;
    vector<string> w;
    int endn;
    string res;
    bool flag;
    bool used[10];
    int fs[11];
    int translate(string &s) {
        if (mp[s[0] - 'A'] == 0) {
            return -1;
        }
        int x = 0;
        for (int i = 0; i < s.length(); ++i) {
            x = x * 10 + mp[s[i] - 'A'];
        }
        return x;
    }
    bool check() {
        int x = 0;
        for (int i = 0; i < w.size(); ++i) {
            int tmp = translate(w[i]);
            if (tmp == -1) {
                return false;
            }
            x += tmp;
        }
        int y = translate(res);
        if (y == -1) {
            return false;
        }
        else {
            return (x == y);
        }
    }
   /* bool go_next() {
        for (int i = cha.size() - 1; i >= 0; --i) {
            int vi = mp[cha[i]];
            if (vi < 9 && ((i == cha.size() - 1) || (vi < mp[cha[i + 1]]))) {
                ++mp[cha[i]];
                return true;
            }
        }
    }*/
    bool check_end() {
        int le = 0, re = 0;
        for (int i = 0; i < w.size(); ++i) {
            int ed = w[i][w[i].length() - 1] - 'A';
            ed = mp[ed];
            le = (le + ed) % 10;
        }
        re = mp[res[res.length() - 1] - 'A'];
        return (le == re);
    }
    bool dfs(int x) {
        if (x == endn) {
            if (!check_end()) {
                return false;
            }
        }
        if (x >= cha.size()) {
            if (check()) {
                flag = true;
            }
            return flag;
        }
        for (int i = 0; i < 10; ++i) {
            if (!used[i]) {
                mp[cha[x]] = i;
                used[i] = true;
                dfs(x + 1);
                used[i] = false;
                mp[cha[x]] = -1;
            }
            if (flag) {
                return true;
            }
        }
        return false;
    }
    /*void decode(int x) {

    }*/
    bool isSolvable(vector<string>& words, string result) {
        w = words;
        res = result;
        flag = false;
        bool *a = new bool[26];
        bool *ed = new bool[26];
        fs[0] = 1;
        for (int i = 1; i <= 10; ++i) {
            fs[i] = fs[i - 1] * i;
        }
        memset(used, 0, sizeof(used));
        mp = new int[26];
        for (int i = 0; i < 26; ++i) {
            a[i] = false;
            ed[i] = false;
        }
        for (int i = 0; i < words.size(); ++i) {
            for (int j = 0; j < words[i].length(); ++j) {
                a[words[i][j] - 'A'] = true;
            }
            if (w[i].length() > 0) {
                ed[w[i][w[i].length() - 1] - 'A'] = true;
            }
        }
        for (int i = 0; i < result.length(); ++i) {
            a[result[i] - 'A'] = true;
        }
        ed[res[res.length() - 1] - 'A'] = true;
        endn = 0;
        //cout << "here" << endl;
        for (int i = 0; i < 26; ++i) {
            if (ed[i]) {
                ++endn;
                cha.push_back(i);
            }
        }
        for (int i = 0; i < 26; ++i) {
            if (a[i] && !ed[i]) {
                cha.push_back(i);
            }
        }
        /*for (int i = 0; i < cha.size(); ++i) {
            cout << char(cha[i] + 'A');
        }
        cout << endl;*/
        //cout << "here" << endl;
        /*for (int i = 0; i < cha.size(); ++i) {
            mp[cha[i]] = i;
        }*/
        return dfs(0);
    }
};
