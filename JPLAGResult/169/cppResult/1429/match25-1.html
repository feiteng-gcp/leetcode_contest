class Solution {
public:
	int ma[1005],cnt,vis[1005],use[1005],shou[1005],zuo[1005],you[1005],mi[15];
	int Zuo,You;
	char a[1005];
	bool ans;
	vector<string> W;
	string T;
	void dfs(int now){
		if(now==cnt+1){
			if(Zuo==You) ans=1;
			return;
		}

		for(int i=0;i<=9;i++){
			if(use[i]) continue;
			if(i==0&&shou[a[now]]) continue;
			use[i]=1;
			Zuo+=i*zuo[now];
			You+=i*you[now];
			dfs(now+1);
			Zuo-=i*zuo[now];
			You-=i*you[now];
			if(ans) return;
			use[i]=0;
		}
	}
    bool isSolvable(vector<string>& w, string t) {
    	for(int i=0;i<w.size();i++) W.push_back(w[i]);
    	T=t;
    	mi[0]=1;
    	for(int i=1;i<=8;i++) mi[i]=mi[i-1]*10;
    	for(int i=0;i<w.size();i++){
    		string s=w[i];
    		shou[s[0]]=1;
    		for(int j=0;j<s.size();j++){
    			if(!ma[s[j]]){
					a[++cnt]=s[j];
    				ma[s[j]]=cnt;
				}
				zuo[ma[s[j]]]+=mi[s.size()-j-1];
			}
		}
		string s=t;
		for(int j=0;j<s.size();j++){
			shou[s[0]]=1;
			if(!ma[s[j]]){
				a[++cnt]=s[j];
				ma[s[j]]=cnt;
			}
			you[ma[s[j]]]+=mi[s.size()-j-1];
		}
		dfs(1);
		return ans;
    }
};
