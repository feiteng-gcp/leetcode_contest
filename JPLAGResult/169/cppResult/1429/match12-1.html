class Solution {
public:
	unordered_set<char>all;
	vector<string>word;
	string t;
	vector<int>mult,mult2;
	vector<char>x;
	bool found = false;
	vector<bool>used,bad;
    vector<int> ct;

	void solve(int digit, int left, int right) {
        if (found)return;
		if (digit == x.size()) {
			if (left == right)found=true;
			return;
		}
		for (int i = digit; i <= 9; i++) {
            swap(ct[i], ct[digit]);
            if(!bad[digit] || ct[digit] > 0){
    			solve(digit + 1, left+mult[x[digit]]*ct[digit], right+mult2[x[digit]]*ct[digit]);
            }
            swap(ct[i], ct[digit]);
		}
		return;
	}
	bool isSolvable(vector<string> words, string result) {
		word = words;
		t = result;
		set<char>t;
		for (auto it : words) {
			for (auto jt : it)all.insert(jt);
            if(it.size()>1)t.insert(it[0]);
		}
        if(result.size()>1)t.insert(result[0]);
        
		for (auto it : result)all.insert(it);
		for (auto it : all)x.push_back(it);
        bad.resize(x.size(),false);
        for(int i=0;i<x.size();i++){
            if(t.count(x[i]))bad[i]=true;
        }
		mult.resize(128, 0);
		mult2.resize(128, 0);
		for (int i = 0; i < words.size(); i++) {
			int curr = 1;
			for (int j = words[i].size() - 1; j >= 0; j--) {
				mult[words[i][j]] += curr;
				curr *= 10;
			}
		}
		int curr = 1;
		for (int j = result.size()-1; j >= 0; j--) {
			mult2[result[j]] += curr;
			curr *= 10;
		}
		used.resize(10, false);
        
        ct.resize(10);
        for (int i = 0; i < 10; i++) ct[i] = i;
        
		solve(0, 0, 0);
		return found;
	}
};
