class Solution {
public:
    set<char> S;
    vector<pair<char, int>> nums;
    bool f;
    int visit[10];
    
    void dfs(int pos, int cur_sum){
        if(pos == nums.size()){
            if(cur_sum == 0){
                f = true;
                return;
            }
            return;
        }
        
        for(int i=0; i<=9; i++){
            if(i == 0 && S.count(nums[pos].first)) {
                // cout << nums[pos].first<< endl;
                continue;
            }
            if(visit[i]) continue;
            
            int tmp_sum = cur_sum + i*nums[pos].second;
          //   cout << tmp_sum << endl;
            visit[i] = 1;
            dfs(pos+1, tmp_sum);
            visit[i] = 0;
            if(f) return;
        }
    }
    
    bool isSolvable(vector<string>& words, string result) {
        map<char, int> xi;
        
        words.push_back(result);
        for(int j=0; j<words.size(); j++){
            string word = words[j];
            S.insert(word[0]);
            
            int cur = 1;
            if(j == words.size()-1) cur = -1;
           for(int i=word.size()-1; i>=0 ; i--){
               xi[word[i]] += cur;
               cur = cur*10;
           }
        }
        
       
        for(auto kv:xi){
            // cout << kv.first << "  " << kv.second << endl;
            nums.push_back(kv);
        }
        f = false;
        memset(visit, 0 ,sizeof(visit));
        dfs(0, 0);
        return f;
    }
};
