class Solution {
public:
    int max_L, N;
    bool permutation(int i, int j, int sum, vector<string> &words, string &result, vector<int> &M, vector<bool> &used) {
        if (i == max_L) {
            if (sum == 0) {
                return true;
            }
            return false;
        }
        for (int k = j; k < N; ++k) {
            int L = words[k].length();
            if (L > i) {
                int c = words[k][L - 1 - i] - 'A';
                if (M[c] == -1) {
                    int b = 0;
                    if (i == L - 1) {
                        b = 1;
                    }
                    for (int v = b; v <= 9; ++v) {

                        if (!used[v]) {
                            used[v] = true;
                            M[c] = v;
                            if (permutation(i, k + 1, sum + v, words, result, M, used)) {
                                return true;
                            }
                            used[v] = false;
                            M[c] = -1;
                        }
                    }
                    return false;
                }
                else {
                    if (i == L - 1 && M[c] == 0) {
                        return false;
                    }
                    sum += M[c];
                }
            }
        }
        int L = result.length();
        if (i > L - 1) {
            return false;
        }
        int c = result[L - 1 - i] - 'A';
        int v = sum % 10;
        if (M[c] == -1) {

            if (used[v]) {
                return false;
            }
            M[c] = v;
            used[v] = true;
            if (permutation(i + 1, 0, sum / 10, words, result, M, used)) {
                return true;
            }
            M[c] = -1;
            used[v] = false;
            return false;
        } else {
            if (M[c] != v) {
                return false;
            }
            return permutation(i + 1, 0, sum / 10, words, result, M, used);
        }
    }
    bool isSolvable(vector<string>& words, string result) {
        vector<int> M(26, -1);
        vector<bool> used(10, false);
        max_L = 0;
        N = words.size();
        for (auto &word : words) {
            int L = word.length();
            max_L = max(max_L, L);
        }
        int L = result.length();
        max_L = max(max_L, L);
        return permutation(0, 0, 0, words, result, M, used);
    }
};
