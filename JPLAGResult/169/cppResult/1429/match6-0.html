string s;
int coeff[10];
bool vis[10];
set<char> fi;
class Solution {
    bool solve(int i, int cur) {
        if (i == s.size()) {
            return cur == 0;
        }
        for (int j = 0; j < 10; j++) {
            if (vis[j]) continue;
            if (j == 0 && fi.find(s[i]) != fi.end()) continue;
            vis[j] = true;
            if (solve(i+1, cur + j * coeff[i])) return true;
            vis[j] = false;
        }
        return false;
    }
public:
    bool isSolvable(vector<string>& words, string result) {
        map<char, int> table;
        fi.clear();
        s.clear();
        memset(vis, false, sizeof(vis));
        for (const string& word : words) {
            int val = 1;
            fi.insert(word[0]);
            for (int i = (int)word.size()-1; i >= 0; i--) {
                table[word[i]] += val;
                val *= 10;
            }
        }
        int val = -1;
        for (int i = (int)result.size()-1; i >= 0; i--) {
            table[result[i]] += val;
            val *= 10;
        }
        fi.insert(result[0]);
        int i = 0;
        for (const auto& item : table) {
            s.push_back(item.first);
            coeff[i++] = item.second;
        }
        return solve(0, 0);

        
        
        
    }
};
