class Solution {
    int n, idx_leadings, idx_trailings;
    vector<char> all;
    unordered_map<char, int> mp;
    bool used[10];
    vector<string> words;
    string result;

    bool dfs(int idx) {
        if (idx == all.size()) {
            int sum = 0;
            for (string word : words) {
                int t = 0;
                for (char c : word) {
                    t = t * 10 + mp[c];
                }
                sum += t;
            }
            int r = 0;
            for (char c : result) {
                r = r * 10 + mp[c];
            }
            return sum == r;
        }
        if (idx == idx_trailings) {
            int sum = 0;
            for (string word : words) {
                sum = (sum + mp[word[word.length() - 1]]) % 10;
            }
            if (sum != mp[result[result.length() - 1]]) {
                return false;
            }
        }
        for (int i = 0; i <= 9; ++i) {
            if ((idx < idx_leadings && i == 0) || used[i]) {
                continue;
            }
            used[i] = true;
            mp[all[idx]] = i;
            if (dfs(idx + 1)) {
                return true;
            }
            mp[all[idx]] = -1;
            used[i] = false;
        }
        return false;
    }
public:
    bool isSolvable(vector<string>& _words, string _result) {
        words = _words; result = _result;

        set<char> leadings;
        for (string word : words) {
            leadings.insert(word[0]);
        }
        leadings.insert(result[0]);
        
        set<char> trailings;
        for (string word : words) {
            trailings.insert(word[word.length() - 1]);
        }
        trailings.insert(result[result.length() - 1]);
        
        for (char c : leadings) {
            all.push_back(c);
        }
        idx_leadings = all.size();
        for (char c : trailings) {
            if (leadings.count(c) == 0) {
                all.push_back(c);
            }
        }
        idx_trailings = all.size();
        
        for (string word : words) {
            for (char c : word) {
                if (mp.count(c) == 0) {
                    mp[c] = -1;
                    if (leadings.count(c) == 0 && trailings.count(c) == 0) {
                        all.push_back(c);
                    }
                }
            }
        }
        for (char c : result) {
            if (mp.count(c) == 0) {
                mp[c] = -1;
                if (leadings.count(c) == 0 && trailings.count(c) == 0) {
                    all.push_back(c);
                }
            }
        }
        for (int i = 0; i < 10; ++i) used[i] = false;
        
        for (char c : all) {
            cout << c << " ";
        }
        cout << endl;

        return dfs(0);
    }
};
