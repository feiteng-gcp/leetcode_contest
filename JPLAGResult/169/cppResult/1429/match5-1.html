class Solution {
public:
    bool vis[26],ok,used[11];
    int A[11],B[26],pw[11],c,L[26],R[26],D[11]={5,4,6,1,2,3,0,8,7,9};
    char str[11];
    bool dfs(int now){

        if(now==c){
            int a=0,b=0;
            for(int i=0;i<c;i++){
                a+=A[B[str[i]-'A']]*L[str[i]-'A'];
                b+=A[B[str[i]-'A']]*R[str[i]-'A'];
            }
            if(a==b)return 1;
            return 0;
        }
        for(int i=0;i<=9;i++){

            if(used[i])continue;
            used[i]=1;
            A[now]=D[i];
            if(dfs(now+1))return 1;
            used[i]=0;
        }
        return 0;
    }
    bool isSolvable(vector<string>& words, string result) {
        pw[0]=1;
        int maxx=0;
        for(int i=1;i<8;i++)pw[i]=pw[i-1]*10;
        for(int i=0;i<words.size();i++){
            for(int j=0;j<words[i].size();j++){
                char cc=words[i][j];
                maxx=max(maxx,(int)words[i].size());
                int pos=words[i].size()-j-1;
                L[cc-'A']+=pw[pos];
                if(!vis[cc-'A']){
                    vis[cc-'A']=1;
                    str[c]=cc;
                    B[cc-'A']=c++;
                }
            }
        }
        for(int i=0;i<result.size();i++){
            char cc=result[i];
            int pos=result.size()-i-1;
            R[cc-'A']+=pw[pos];
            if(!vis[cc-'A']){
                vis[cc-'A']=1;
                str[c]=cc;
                B[cc-'A']=c++;
            }
        }
        if(maxx>result.size()||maxx<result.size()-3)return false;
        return dfs(0);
    }
};
