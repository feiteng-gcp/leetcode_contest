class Solution {
public:
    int last;

    bool check(vector<int>& table, vector<string>& words, string& result) {
        int sum = 0;
        for (string word : words) {
            int num = 0;
            for (char c : word) {
                num = num * 10 + table[c - 'A'];
            }
            sum += num;
        }
        int res = 0;
        for (char c : result) {
            res = res * 10 + table[c - 'A'];
        }
        return sum == res;
    }
    bool cut(vector<int>& table, vector<string>& words, string& result) {
        int sum = 0;
        for (string word : words) sum += table[*word.rbegin() - 'A'];
        int res = table[*result.rbegin() - 'A'];
        return sum % 10 == res;
    }
    bool dfs(int k, vector<bool>& used, vector<char>& letters, vector<int>& table, vector<string>& words, string& result) {
        if (k == letters.size()) {
            return check(table, words, result);
        }
        if (k == last && !cut(table, words, result)) return false;
        int i = 0;
        if (letters[k] == result[0]) i = 1;
        for (string word : words) {
            if (letters[k] == word[0]) i = 1;
        }
        for (; i < 10; i++) {
            if (!used[i]) {
                used[i] = true;
                table[letters[k] - 'A'] = i;
                bool found = dfs(k + 1, used, letters, table, words, result);
                used[i] = false;
                if (found) return true;
            }
        }
        return false;
    }
    bool isSolvable(vector<string>& words, string result) {
        vector<bool> used(10, false);
        vector<char> letters;
        vector<int> table(26, -1);
        for (string word : words) {
            if (table[*word.rbegin() - 'A'] == -1) letters.push_back(*word.rbegin());
                table[*word.rbegin() - 'A'] = 0;
        }
        if (table[*result.rbegin() - 'A'] == -1) letters.push_back(*result.rbegin());
                table[*result.rbegin() - 'A'] = 0;
        last = letters.size();
        for (string word : words) {
            for (char c : word) {
                if (table[c - 'A'] == -1) letters.push_back(c);
                table[c - 'A'] = 0;
            }
        }
        for (char c : result) {
            if (table[c - 'A'] == -1) letters.push_back(c);
            table[c - 'A'] = 0;
        }
        return dfs(0, used, letters, table, words, result);
    }
};
