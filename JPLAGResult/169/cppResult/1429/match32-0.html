class Solution {
public:
  int M[128];
  bool used[128];
  vector<string> W;
  string R;
  int L;
  bool dfs(int x, int y,int sum, int borrow) {
    if (x == L) {
      return borrow == 0;
    }
    if (y == -1) {
      char c = R[x];
      if (M[c] != -1) {
        sum = M[c] - borrow;
        borrow = 0;
        if (sum < 0) sum += 10, borrow = 1;
        if (dfs(x, 0, sum, borrow)) return true;
      } else {
        for (int i = 0; i <= 9; i++) {
          if (i == 0 && x + 1 == L) continue;
          if (used[i]) continue;
          M[c] = i, used[i] = true;
          int tsum = i - borrow ;
          int tborrow = 0; 
          if (tsum < 0) tsum += 10, tborrow = 1;
          if (dfs(x, 0, tsum, tborrow)) return true;
          M[c] = -1, used[i] = false;
        }
      } 
    } else if (y == W.size()) {
      if (sum != 0) return false;
      return dfs(x+1, -1, sum, borrow);
    } else {
      char c = W[y][x];
      if (M[c] != -1) {
        int tsum = sum - M[c];
        int tborrow = borrow;
        if (tsum < 0) tsum += 10, tborrow++;
        if (dfs(x, y+1, tsum, tborrow)) return true;
      } else {
        for (int i = 0; i <= 9; i++) {
          if (i == 0 && (x + 1 == L || W[y][x+1] == ' ')) continue;
          if (used[i]) continue;
          M[c] = i, used[i] = true;
          int tsum = sum - i;
          int tborrow = borrow;
          if (tsum < 0) tsum += 10, tborrow++;
          if (dfs(x, y+1, tsum, tborrow)) return true;
          M[c] = -1, used[i] = false;
        }
      }
    }
    return false;
  }
  bool isSolvable(vector<string>& words, string result) {
    memset(M, 0, sizeof(M));
    memset(used, 0, sizeof(used));
    for (int i = 'A'; i <= 'Z'; i++) M[i] = -1;
    for (auto &w: words) reverse(w.begin(), w.end());
    reverse(result.begin(), result.end());
    R = result;
    L = result.size();
    for (auto &w: words) L = max(L, int(w.size()));
    for (auto &w: words) W.push_back(w);
    for (auto &w: W) while (w.size() < L) w.push_back(' ');
    while (R.size() < L) R.push_back(' ');
    return dfs(0, -1, 0, 0);
  }
};
