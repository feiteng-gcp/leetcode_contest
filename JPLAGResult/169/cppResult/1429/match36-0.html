class Solution {
public:
    bool next_comb(vector<int>& P)
{
    int b = (int)P.size() - 1, e = 9;
    while(b >= 0 && P[b] == e)
    {
        b--;
        e--;
    }
    if(b == -1) return false;
    P[b]++;
    for(int i = b + 1; i < P.size(); i++) P[i] = P[i - 1] + 1;
    return true;
}

bool isSolvable(vector<string>& words, string result)
{
    int len = (int)words.size(), c = 0;
    vector<int> c2i(128, -1);
    
    for(int i = 0; i < len; i++)
        for(int j = 0; j < words[i].size(); j++)
            if(c2i[words[i][j]] == -1) c2i[words[i][j]] = c++;
    for(int i = 0; i < result.size(); i++)
        if(c2i[result[i]] == -1) c2i[result[i]] = c++;
    
    vector<int> P(c, 0);
    for(int i = 0; i < c; i++) P[i] = i;
    do
    {
        do
        {
            int sum = 0;
            for(int i = 0; i < len; i++)
                sum += P[c2i[words[i].back()]];
            if(sum % 10 != P[c2i[result.back()]]) continue;
            
            int L = 0, R = 0;
            for(int i = 0; i < len; i++)
            {
                int cur = 0;
                for(int j = 0; j < words[i].size(); j++)
                {
                    cur = cur * 10 + P[c2i[words[i][j]]];
                }
                L += cur;
            }
            for(int i = 0; i < result.size(); i++)
                R = R * 10 + P[c2i[result[i]]];
            if(L == R) return true;
        }while(next_permutation(P.begin(), P.end()));
    }while(next_comb(P));
    return false;
}
};
