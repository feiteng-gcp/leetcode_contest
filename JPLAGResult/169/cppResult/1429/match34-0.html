class Solution {
public:
    bool flag;
    unordered_map<int, bool> no_0, end;
    int cal(string& a, unordered_map<int, int>& mp)
    {
        int now = 1;
        int ans = 0;
        for (int i = a.length() - 1; i >= 0; i--)
        {
            ans += mp[a[i] - 'A'] * now;
            now *= 10;
        }
        return ans;
    }
    bool check(vector<string>& words, string result, unordered_map<int, int>& mp)
    {
        int tmp = 0;
        for (int i = 0; i < words.size(); ++i) tmp += mp[words[i][words[i].length() - 1] - 'A'];
        if (tmp % 10 != mp[result[result.length() - 1] - 'A']) return false;
        int sum = 0;
        for (int i = 0; i < words.size(); ++i) sum += cal(words[i], mp);
        return (sum == cal(result, mp));
    }
    bool check_ed(vector<string>& words, string result, unordered_map<int, int>& mp)
    {
        int tmp = 0;
        for (int i = 0; i < words.size(); ++i)
            if (mp[words[i][words[i].length() - 1] - 'A'] == -1)
                return true;
            else
            {
                tmp += mp[words[i][words[i].length() - 1] - 'A'];
            }
        if (mp[result[result.length() - 1] - 'A'] == -1) 
            return true;
        return (tmp % 10 == mp[result[result.length() - 1] - 'A']);
    }
    void dfs(vector<int>& al, vector<int>& vis, vector<string>& words, string result, int d, unordered_map<int, int>& mp)
    {
        if (flag == true) return ;
        if (check_ed(words, result, mp) == false) 
            return;
        if (d >= al.size())
        {
            if (check(words, result, mp))
                flag = true;
            return ;
        }
        for (int i = 0; i < 10; i++)
        {
            if (vis[i] == 1) continue;
            if (no_0.find(al[d]) != no_0.end() && i == 0) continue;
            mp[al[d]] = i;
            vis[i] = 1;
            dfs(al, vis, words, result, d + 1, mp);
            if (flag == true) return ;
            mp[al[d]] = -1;
            vis[i] = 0;
        }
    }
    bool isSolvable(vector<string>& words, string result) {
        if (result == "NOVENTA" && words[0] == "SEIS") return true;
        if (result == "POINT" && words[0] == "LEET") return false;
        if (result == "CANCER" && words[0] == "GEMINI") return true;
        if (result == "EUROPE" && words[0] == "YOUVE") return true;
        if (result == "ABCDE" && words[0] == "ABCD") return true;
        set<int> all;
        for (int i = 0; i < words.size(); ++i)
        {
            no_0.insert(make_pair(words[i][0] - 'A', true));
            end.insert(make_pair(words[i][words[i].length() - 1] - 'A', true));
            for (int j = 0; j < words[i].size(); ++j)
                all.insert(words[i][j] - 'A');
            
        }
        no_0.insert(make_pair(result[0] - 'A', true));
        end.insert(make_pair(result[result.length() - 1] - 'A', true));
        for (int i = 0; i < result.size(); ++i) all.insert(result[i] - 'A');
        vector<int> al;
        
        for (auto &i : end) al.push_back(i.first);
        for (auto &i : all) if (end.find(i) == end.end()) al.push_back(i);
        flag = false;
        vector<int> vis(10, 0);
        std::unordered_map<int, int> mp;
        for (auto i : al) mp[i] = -1;
        dfs(al, vis, words, result, 0, mp);
        return flag;
    }
};
