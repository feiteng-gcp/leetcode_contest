class Solution {
public:
    
    bool aux(vector<int>& co, vector<int>& x, int ind, int cur) {
        if(ind == co.size()) return cur == 0;
        for(int i = ind; i < x.size(); i++) {
            swap(x[ind], x[i]);
            if(aux(co, x, ind + 1, cur + x[ind] * co[ind])) return true;
            swap(x[ind], x[i]);
        }
        return false;
    }
    
    bool isSolvable(vector<string>& words, string r) {
        unordered_map<char, int> m;
        unordered_set<char> n0;
        for(auto& s : words) {
            int fac = 1;
            for(int i = s.size() - 1; i >= 0; i--) {
                m[s[i]] += fac;
                fac *= 10;
            }
            if(s.size() > 1) n0.insert(s[0]);
        }
        
        int fac = 1;
        for(int i = r.size() - 1; i >= 0; i--) {
            m[r[i]] -= fac;
            fac *= 10;
        }
        
        int zeros = 0, neg = 0;
        for(auto& t : m) {
            if(t.second < 0) neg ++;
            if(t.second == 0) zeros++;
        }
        
        if(zeros == m.size()) return true;
        if(neg == m.size()) return n0.size() ? false : true;
        
        vector<int> co;
        
        for(auto& t : m) co.push_back(t.second);
        
        std::sort(co.begin(), co.end());
        reverse(co.begin(), co.end());
        
        if(co.back() > 0) return n0.size() ? false : true;
        
        vector<int> x;
        for(int i = 0; i < 10; i++) x.push_back(i);
        return aux(co, x, 0, 0);
        
    }
};
