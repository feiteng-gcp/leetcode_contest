class Solution {
public:
    int tot = 0,tot2,h[134],nc0[14];
    vector<string>w;
    string r;
    int a[14],val[14];
    bool b[14];
    int getnum(string &s){
        int ans = 0;
        for(int i = 0;i < s.size();i++){
            ans = ans * 10 + a[h[s[i]]];
        }
        return ans;
    }
    bool dfs(int f){
        if(f == tot){
            int sum = 0,ans = getnum(r);
            for(int i = 0;i < tot2;i++)
                sum += val[i] * a[i];

            return sum == ans;
        }
        for(int i = 0;i <= 9;i++){
            if(i == 0 && nc0[f] || b[i]) continue;
            b[i] = 1;
            a[f] = i;
            if(dfs(f + 1)) return 1;
            b[i] = 0;
        }
        return 0;
    }
    bool isSolvable(vector<string>& words, string result) {
        memset(h,-1,sizeof(h));
        memset(nc0,0,sizeof(nc0));
        memset(b,0,sizeof(b));
        memset(val,0,sizeof(val));
        w = words;
        r = result;
        for(int i = 0;i < w.size();i++){
            for(int j = 0;j < w[i].size();j++){
                if(h[w[i][j]] == -1)
                    h[w[i][j]] = tot++;
            }
            nc0[h[w[i][0]]] = 1;
        }
        tot2 = tot;
        for(int j = 0;j < r.size();j++)
            if(h[r[j]] == -1)

                h[r[j]] = tot++;
        nc0[h[r[0]]] = 1;
        for(int i = 0;i < w.size();i++){
            for(int j = w[i].size() - 1,k = 1;j >= 0;j--,k *= 10){
                val[h[w[i][j]]] += k;
            }
        }
        if(dfs(0)) return 1;
        else return 0;
    }
};
