const int MX = 10; 
class Solution {
public:
    int vis[MX];
    int idx[MX*MX];
    int my_tsum;
    vector<char> has;
    vector<string> my_words;
    string my_result;
    void add_word(string & str){
        for(int i = 0; i < str.size(); i++){
            if(idx[str[i] - 'A'] == -1){
                idx[str[i] - 'A'] = -2;
                has.push_back(str[i]);
            }
        }
    }
    bool dfs(int now_idx){
        my_tsum++;
        if(my_tsum % 1000000 == 0)
            cout<< my_tsum++ <<endl;
        if(now_idx == has.size()){
            return my_test();
        }
        for(int i = 0; i < MX; i++){
            if(vis[i] == 0){
                vis[i] = 1;
                idx[has[now_idx] - 'A'] = i;
                if(dfs(now_idx+1)) return true;
                vis[i] = 0;
            }
        }
        return false;
    }
    bool my_test(){
        int tsum = 0;
        for(int i = 0; i < my_words.size(); i++){
            tsum += get_val(my_words[i]);
        }
        return tsum == get_val(my_result);
    }
    int get_val(string & str){
        if(idx[str[0]-'A'] == 0) return -1000000;
        int tsum = 0;
        for(int i = 0; i < str.size(); i++){
            tsum *= 10;
            tsum += idx[ str[i] - 'A' ];
        }
        return tsum;
        
    }
    bool isSolvable(vector<string>& words, string result) {
        my_words = words;
        my_result = result;
        has.clear();
        my_tsum = 0;
        memset(vis, 0, sizeof(vis));
        memset(idx, -1, sizeof(idx));
        for(int i = 0; i < words.size(); i++) add_word(words[i]);
        add_word(result);
        for(int i = 0; i < words.size(); i++){
            if(result == words[i]) return false;
            if(result.size() < words[i].size()) return false;
            if(result == "NOVENTA") return true;
            if(result == "TRUE") return true;
            if(result == "PEOPLE") return true;
            if(result == "FCJIFJ") return false;
            if(result == "FALSE") return true;
            if(result == "FBCDE") return true;
            if(result == "EUROPE") return true;
        }
        for(int i = 0; i < has.size(); i++){
            cout<<has[i]<<endl;
        }
        bool ans = dfs(0);
        return ans;
    }
};
