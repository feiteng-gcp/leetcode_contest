const int N = 10;
vector<char> chars;
map<char, int> charMap;
bool numVisit[N];
int tailNums[N];

class Solution {
public:
    bool isSolvable(vector<string>& words, string result) {
        chars.clear();
        for (char c = 'A'; c <= 'Z'; c++) {
            charMap[c] = -1;
        }
        memset(numVisit, 0, sizeof(numVisit));
        
        // construct chars
        destructWords(words, result);
        
        return dfs(words, result, 0);
    }
    
private:
    void destructWords(vector<string>& words, string& result) {
        int idx;
        for (int i=0; i<8; i++) {
            for (int j = 0; j<words.size(); j++) {
                idx = words[j].size() - 1 - i;
                if (idx >= 0) tryPushChar(words[j][idx]);
            }
            idx = result.size() - 1 - i;
            if (idx >= 0) tryPushChar(result[idx]);
        }
    }
    
    void tryPushChar(char& c) {
        for (int i=0; i<chars.size(); i++) {
            if (chars[i] == c) {
                return;
            }
        }
        chars.push_back(c);
    }
    
    bool dfs(vector<string>& words, string& result, int depth) {
        if (depth >= chars.size()) {
            return isExpValid(words, result);
        }
        
        for (int i=0; i<N; i++) {
            if (numVisit[i]) continue;
            
            numVisit[i] = true;
            
            charMap[chars[depth]] = i;
            if (isWordsValid(words, result, depth)) {
                if (dfs(words, result, depth+1)) {
                    return true;
                }
            }
            
            numVisit[i] = false;
        }
        charMap[chars[depth]] = -1;
        return false;
    }
    
    bool isWordsValid(vector<string>& words, string& result, int depth) {
        if (charMap[result[0]] == 0) {
            // printf("result %s leading 0", result);
            return false;
        }
        for (int i=0; i<words.size(); i++) {
            if (charMap[words[i][0]] == 0) {
                // printf("word %s leading 0", word[i]);
                return false;
            }
        }
        
        int x;
        int minX = 1e9;
        for (int i=0; i<words.size(); i++) {
            tailNums[i] = tailNum(words[i], x);
            minX = min(minX, x);
        }
        int rightTail = tailNum(result, x);
        minX = min(minX, x);
        
        int leftTailSum = 0;
        for (int i=0; i<words.size(); i++) {
            leftTailSum += tailNums[i] % minX;
        }
        rightTail %= minX;
        
        return (leftTailSum % minX) == rightTail;
    }
    
    int tailNum(string& str, int &x) {
        int num = 0;
        x = 1;
        for (int i=str.size()-1; i>=0; i--) {
            const int m = charMap[str[i]];
            if (m == -1) break;
            num += x * m;
            x *= 10;
        }
        return num;
    }
    
    bool isSuffix(int a, int b) {
        while(true) {
            if (a == 0 || b == 0) return true;
            if (a % 10 != b % 10) return false;
            a /= 10; b /= 10;
        }
        return false;
    }
    
    bool isExpValid(vector<string>& words, string& result) {
        int leftSum = 0;
        for (int i=0; i<words.size(); i++) {
            leftSum += convert(words[i]);
        }
        return leftSum == convert(result);
    }
    
    int convert(string& str) {
        int num = 0;
        for (int i=0; i<str.size(); i++) {
            num = num * 10 + charMap[str[i]];
        }
        return num;
    }
};
