class Solution {
    bool isSolvable(vector<string>& words, string& result, vector<int>& letter, vector<int>& value, int pos, int idx, int carry)
    {
        if (idx == words.size()) {
            if (pos == result.size()) return carry == 0;
            int c = result[pos] - 'A';
            int d = carry % 10;
            if (value[c] != -1) {
                if (value[c] != d) return false;
                return isSolvable(words, result, letter, value, pos + 1, 0, carry / 10);
            }
            if (letter[d] != -1) return false;
            letter[d] = c;
            value[c] = d;
            bool res = isSolvable(words, result, letter, value, pos + 1, 0, carry / 10);
            letter[d] = -1;
            value[c] = -1;
            return res;
        }
        
        if (pos < words[idx].size()) {
            int c = words[idx][pos] - 'A';
            if (value[c] != -1)
                return isSolvable(words, result, letter, value, pos, idx + 1, carry + value[c]);
            for (int d = 0; d < 10; d++) {
                if (d == 0 && pos > 0 && pos == words[idx].size() - 1) continue;
                if (letter[d] != -1) continue;
                letter[d] = c;
                value[c] = d;
                bool res = isSolvable(words, result, letter, value, pos, idx + 1, carry + d);
                letter[d] = -1;
                value[c] = -1;
                if (res) return true;
            }
            return false;
        }
        else {
            return isSolvable(words, result, letter, value, pos, idx + 1, carry);
        }
    }
public:
    bool isSolvable(vector<string>& words, string result)
    {
        //cout << "isSolvable" << endl;
        vector<string> c_words(words);
        string c_result(result);
        int maxlen = 0;
        for (int i = 0; i < c_words.size(); i++) {
            std::reverse(c_words[i].begin(), c_words[i].end());
            maxlen = std::max(maxlen, (int)c_words[i].size());
            //cout << words[i] << ' ';
        }
        std::reverse(c_result.begin(), c_result.end());
        //cout << result << endl;
        if (c_result.size() < maxlen) return false;
        
        vector<int> letter(10, -1);
        vector<int> value(26, -1);
        bool res = isSolvable(c_words, c_result, letter, value, 0, 0, 0);
        //cout << res << endl;
        return res;
    }
};
