class Solution {
public:
    bool isSolvable(vector<string>& words, string result) {
        vector<int> sta(10);
        int num = 0;
        unordered_map<char, int> memo;
        for (int i = 0; i < words.size(); i++){
            for (int j = 0; j < words[i].size(); j++){
                if (memo.find(words[i][j]) == memo.end())
                    memo[words[i][j]] = num++;
                sta[memo[words[i][j]]] += (int)pow(10, (words[i].size() - 1 - j));
            }
        }
        for (int i = 0; i < result.size(); i++){
            if (memo.find(result[i]) == memo.end())
                memo[result[i]] = num++;
            sta[memo[result[i]]] -= (int)pow(10, result.size() - 1 - i);
        }
        vector<int> used(10);
        //for (int i = 0; i < num; i++)
          //  cout << sta[i] << endl;
        return find(sta, 0, num, used, 0);        
    }
    

    bool find(vector<int>& sta, int index, int num, vector<int>& used, int sum){
        if (index == num)
            return sum == 0;
        //cout << sta[index] << " " << index << " " << sum << endl;
        for (int i = 0; i < 10; i++){
            if (!used[i]){
                used[i] = true;
                if (find(sta, index + 1, num, used, sum + sta[index] * i))
                    return true;
                used[i] = false;
            }
        }
        return false;
    }
                     
        
};
