class Solution {
public:
    bool isSolvable(vector<string>& words, string result) {
        maps = vector<int>(26, -1);
        used = vector<bool>(10);
        maxLen = size(result);
        for (auto word : words) {
            if (size(word) > maxLen) {
                return false;
            }
        }
        bool ret = search(words, result, 0, 0);
        if (ret) {
            // debug();
        }
        return ret;
    }
    
private:
    int maxLen;
    vector<int> maps;
    vector<bool> used;
    
    int size(const string& s) {
        return static_cast<int>(s.length());
    }
    
    void debug() {
        for (int i = 0; i < 26; ++i) {
            if (maps[i] != -1) {
                cout << static_cast<char>(i + 'A') << "-" << maps[i] << " ";
            }
        }
        cout << endl;
    }
    
    bool search(const vector<string>& words, const string& result, int index, int carry) {
        if (index == maxLen) {
            return carry == 0;
        }
        int sum = carry;
        for (auto word : words) {
            int len = size(word);
            if (index < len) {
                int charIndex = len - index - 1;
                char ch = word[charIndex];
                int mapVal = maps[ch - 'A'];
                if (mapVal != -1) {
                    sum += mapVal;

                } else {
                    for (int j = 0; j < 10; ++j) {
                        if (used[j] || (j == 0 && len - index - 1 == 0)) {
                            continue;
                        }
                        maps[ch - 'A'] = j;
                        used[j] = true;
                        if (search(words, result, index, carry)) {
                            return true;
                        }
                        used[j] = false;
                        maps[ch - 'A'] = -1;
                    }
                    return false;
                }
            }
        }
        if (sum == 0) {
            return false;
        }
        int len = size(result);
        int charIndex = len - index - 1;
        char ch = result[charIndex];
        int mapVal = maps[ch - 'A'];
        int digit = sum % 10;
        int nextCarry = sum / 10;
        if (mapVal != -1) {
            if (mapVal != digit) {
                return false;
            }
            if (search(words, result, index + 1, nextCarry)) {
                return true;
            }
        } else {

            if (used[digit]) {
                return false;
            }
            maps[ch - 'A'] = digit;
            used[digit] = true;
            if (search(words, result, index + 1, nextCarry)) {
                return true;
            }
            used[digit] = false;
            maps[ch - 'A'] = -1;
        }
        return false;
    }
};
