class Solution {
public:
    vector<char> ans;
    bool vis[10],v[305];
    int M[305],dp[305],res[305];
    bool dfs(int idx,int w,int r) {
        if (idx==ans.size()) {

            return w==r;
        }
        for (int i=0; i<=9; ++i) {
            if (vis[i]) continue;
            if (v[ans[idx]]&&i==0) continue;
            M[ans[idx]]=i;
            vis[i]=true;
            if (dfs(idx+1,w+dp[ans[idx]]*i,r+res[ans[idx]]*i)) return true;
            vis[i]=false;
        }
        return false;
    }
    int ar[10]={1,10,100,1000,10000,100000,1000000,10000000,100000000};
    bool isSolvable(vector<string>& words, string result) {
        set<char> S;
        for (int i=0; i<words.size(); ++i) {
            if (words[i].length()>1) v[words[i][0]]=true;
            int len=words[i].size();
            for (int j=0; j<words[i].length(); ++j) {
                S.insert(words[i][j]);
                dp[words[i][j]]+=ar[len-j-1];
            }
        }
        if (result.length()>1) v[result[0]]=true;
        for (int i=0; i<result.size(); ++i) {
            S.insert(result[i]);
            res[result[i]]+=ar[result.size()-i-1];
        }
        for (auto p=S.begin(); p!=S.end(); ++p) ans.push_back(*p);
        return dfs(0,0,0);
    }
};
