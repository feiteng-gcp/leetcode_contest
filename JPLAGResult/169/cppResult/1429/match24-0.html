class Solution {
public:
    
    bool dfs(vector<string>& words, const string& result, vector<int>& mat, vector<bool>& num, vector<bool>& nonzero, int idx, vector<char>& letters){
        
        int minLen = INT_MAX;
        for(int i = 0; i < words.size(); i ++){
            int len = 0;
            for(int j = words[i].size() - 1; j >= 0; j --){
                if (mat[ words[i][j] - 'A' ] != -1){
                    len ++;
                } else {
                    break;
                }
            }
            minLen = min(minLen, len);
        }
        int len = 0;
        for(int i = result.size() - 1; i >= 0; i --){
            if (mat[result[i] - 'A'] != -1){
                len ++;
            }
        }
        minLen = min(minLen, len);
        
        int ten = 1;
        int sum1 = 0;
        int sum2 = 0;
        for(int i = 0; i < minLen; i ++){
            for(int j = 0; j < words.size();j ++){
                //assert( mat[words[j][words[j].size() - i - 1] - 'A'] != -1);
                sum1 += ten * mat[words[j][words[j].size() - i - 1] - 'A'];
            }
            //assert( mat[result[result.size() - i - 1] - 'A'] != -1);
            sum2 += mat[result[result.size() - i - 1] - 'A'] * ten; 
            ten *= 10;
        }
        
        if (sum1 % ten != sum2) return false;
                
        if (idx == letters.size()){
            int sum1 = 0;
            int sum2 = 0;
            for(int i = 0; i < words.size(); i ++){
                int s = 0;
                for(int j = 0; j < words[i].size();j ++){
                    s = 10 * s + mat[ words[i][j] - 'A'];
                }
                sum1 += s;
            }
            for(int i = 0; i < result.size(); i ++){
                sum2 = 10 * sum2 +  mat[result[i] - 'A'];
            }
            if (sum1 == sum2) return true;
            return false;
        }
        
        for(int i = 0; i < 10; i ++){
            if (num[i] == false && (i > 0 || !nonzero[ letters[idx] - 'A' ])){
                num[i] = true;
                mat[ letters[idx] - 'A'  ] = i;
                if(dfs(words, result, mat, num, nonzero, idx + 1, letters)) return true;
                mat[ letters[idx] - 'A'  ] = -1;
                num[i] = false;
            }
        }
        return false;
    }
    
    bool isSolvable(vector<string>& words, string result) {
        vector<int> mat(26, -1); 
        vector<bool> num(10, false);
        vector<bool> nonzero(26, false);
        
        
        for(int i = 0; i < words.size(); i ++){
            nonzero[words[i][0] - 'A'] = true;
        }
        nonzero[result[0] - 'A'] = true;
        
        int maxLen = result.size();
        for(int i = 0; i < words.size();i ++){
            if (words[i].size() > maxLen){
                maxLen = words[i].size();
            }
        }
        
        unordered_set<char> g;
        vector<char> letters;
        for(int i = maxLen - 1; i >= 0; i --){
            if ( i < result.size()  && g.count(result[i]) == 0){
                g.insert(result[i]);
                letters.push_back(result[i]);
            }
            
            for(int j = 0; j < words.size(); j ++){
                if (i < words[j].size() && g.count(words[j][i]) == 0){
                    g.insert(words[j][i]);
                    letters.push_back(words[j][i]);
                }
            }
        }
        
        vector<int> digit;
        for(int i = 0; i < 10; i ++){
            digit.push_back(i);
        }
        
        return dfs(words, result, mat, num, nonzero, 0, letters);
    }
};
