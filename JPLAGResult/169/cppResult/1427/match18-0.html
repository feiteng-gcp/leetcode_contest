class Solution {
public:
  vector<int> res;
  void dfs(TreeNode* v){
    if(!v) return;
    res.emplace_back(v->val);
    dfs(v->left);
    dfs(v->right);
  }
  vector<int> getAllElements(TreeNode* root1, TreeNode* root2) {
    res.clear();
    dfs(root1);dfs(root2);
    sort(res.begin(),res.end());
    return res;
  }
};
