class Solution {
public:
	bool canReach(vector<int>& arr, int start) {
		int n = arr.size(), i, j, k;
		queue<int> q;
		vector<bool> used(n, false);
		q.push(start);
		used[start] = true;

		while (!q.empty()) {
			int id = q.front();
			q.pop();

			if (arr[id] == 0) return true;
			int x;
			x = id - arr[id];
			if ((x >= 0) && (!used[x])) {
				q.push(x);
				used[x] = true;
			}
			x = id + arr[id];
			if ((x < n) && (!used[x])) {
				q.push(x);
				used[x] = true;
			}
		}
		return false;
	}
};
