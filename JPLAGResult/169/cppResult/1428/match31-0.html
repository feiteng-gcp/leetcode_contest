class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        vector<bool> visited(n, false);
        queue<int> Q;
        Q.push(start);
        visited[start] = true;
        while (!Q.empty()) {
            int u = Q.front();
            Q.pop();
            if (arr[u] == 0) return true;
            int v = u + arr[u];
            if (v >= 0 && v < n) {
                if (visited[v] == false) {
                    visited[v] = true;
                    Q.push(v);
                }
            }
            v = u - arr[u];
            if (v >= 0 && v < n) {
                if (visited[v] == false) {
                    visited[v] = true;
                    Q.push(v);
                }
            }
        }
        return false;
    }
};
