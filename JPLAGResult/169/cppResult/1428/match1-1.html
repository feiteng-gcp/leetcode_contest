class Solution {
public:
    bool canReach(vector<int>& a, int s) {
        int n = a.size();
        vector<bool> vis(n, false);
        vis[s] = true;
        queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int cur = q.front();
            q.pop();
            if (a[cur] == 0)
                return true;
            if (cur + a[cur] < n && !vis[cur + a[cur]]) {
                vis[cur + a[cur]] = true;
                q.push(cur + a[cur]);
            }
            if (cur - a[cur] >= 0 && !vis[cur - a[cur]]) {
                vis[cur - a[cur]] = true;
                q.push(cur - a[cur]);
            }
        }
        return false;
    }
};
