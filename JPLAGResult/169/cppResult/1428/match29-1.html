class Solution {
    bool f[50012];
    bool gogo(vector<int>& arr, int x)
    {
        if (x < 0 || x >= arr.size()) return false;
        if (arr[x] == 0) return true;
        if (f[x]) return false;
        f[x] = true;
        return gogo(arr, x + arr[x]) || gogo(arr, x - arr[x]);
    }
public:
    bool canReach(vector<int>& arr, int start) {
        memset(f, 0, sizeof(f));
        f[start] = 0;
        if (arr[start] == 0) return true;
        return gogo(arr, start + arr[start]) || gogo(arr, start - arr[start]);
    }
};
