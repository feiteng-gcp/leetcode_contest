class Solution {
public:
    bool dp[50005],ans;
    int n;
    void solve(int x,vector<int>& arr){
        dp[x]=1;
        if(!arr[x]){
            ans=1;
            return;
        }
        if(x+arr[x]<n && !dp[x+arr[x]])
            solve(x+arr[x],arr);
        if(x-arr[x]>=0 && !dp[x-arr[x]])
            solve(x-arr[x],arr);
    }
    bool canReach(vector<int>& arr, int start) {
        n=arr.size();
        solve(start,arr);
        return ans;
    }
};
