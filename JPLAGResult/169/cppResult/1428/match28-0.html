class Solution {
public:
bool canReach(vector<int>& arr, int start) {
		int n = arr.size();
		queue<int> q;
		vector<int> visited(n,0);
		q.push(start);
		visited[start] = 1;
		while(!q.empty()){
				auto x = q.front(); q.pop();
				if( arr[x] + x  < n and visited[arr[x] + x ] == 0 ){
						visited[arr[x]+x] = 1;
						q.push(arr[x]+x);
				}
				if( x - arr[x] >= 0 and visited[x - arr[x] ] == 0 ){
						visited[ x - arr[x] ] = 1;
						q.push( x - arr[x] );
				}
		}
		for(int i=0; i<n; ++i){
				if( arr[i] == 0 and visited[i] == 1 ){ return true; }
		}
		return false;
}
};
