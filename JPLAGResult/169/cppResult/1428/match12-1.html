bool vis[500001];
class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        memset(vis,false,sizeof(vis));

        int n=arr.size();
        queue<int> q;
        q.push(start);
        vis[start]=true;
        while(!q.empty())
        {
            int now=q.front();
            q.pop();
            if(arr[now]==0)return true;
            if(now-arr[now]>=0&&!vis[now-arr[now]])
            {
                vis[now-arr[now]]=true;
                q.push(now-arr[now]);
            }
            if(now+arr[now]<n&&!vis[now+arr[now]])
            {
                vis[now+arr[now]]=true;
                q.push(now+arr[now]);
            }
        }
        return false;
    }
};
