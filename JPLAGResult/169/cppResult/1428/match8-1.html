#define inRange(x,a,b) (a <= x && x < b)

class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        vector<bool> dp(n, 0);
        queue<int> q;
        q.push(start);
        dp[start] = true;
        while(!q.empty()){
            int x = q.front();  q.pop();
            if(arr[x] == 0) return true;
            if(inRange(x-arr[x], 0, n) && !dp[x-arr[x]]){
                dp[x-arr[x]] = true;
                q.push(x-arr[x]);
            }
            if(inRange(x+arr[x], 0, n) && !dp[x+arr[x]]){
                dp[x+arr[x]] = true;
                q.push(x+arr[x]);
            }
        }
        return false;
    }
};
