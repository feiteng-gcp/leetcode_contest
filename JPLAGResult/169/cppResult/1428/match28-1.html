class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        vector<int> visited(n, -1);
        
        queue<int> q;
        visited[start] = 1;
        q.push(start);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            int bef = u - arr[u];
            if (bef >= 0 and visited[bef] == -1) {
                visited[bef] = 1;
                q.push(bef);
            }
            int nex = u + arr[u];

            if (nex < n and visited[nex] == -1) {
                visited[nex] = 1;
                q.push(nex);
            }
        }
        
        for (int i = 0; i < n; i++) {
            if (arr[i] == 0 and visited[i] == 1) {
                return true;
            }
        }
        return false;
    }
};
