class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        vector<bool> vis(n, false);
        vis[start] = true;
        
        stack<int> now;
        now.push(start);
        while (now.size() > 0) {
            int p = now.top();
            now.pop();
            
            if (arr[p] == 0) return true;
            
            if (p + arr[p] < n && !vis[p + arr[p]]) {
                vis[p + arr[p]] = true;
                now.push(p + arr[p]);
            }
            
            if (p - arr[p] >= 0 && !vis[p - arr[p]]) {
                vis[p - arr[p]] = true;
                now.push(p - arr[p]);
            }
        }
        
        return false;
    }
};
