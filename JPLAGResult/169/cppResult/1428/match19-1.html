class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        unordered_set<int> visited;
        deque<int> c;
        c.push_back(start);
        while (c.size() > 0) {
            int t = c.front();
            if (arr[t] == 0) {
                return true;
            }
            c.pop_front();
            if (visited.find(t) != visited.end()) {
                continue;
            }
            visited.insert(t);
            if (t + arr[t] >= 0 && t + arr[t] < arr.size()) {
                c.push_back(t + arr[t]);
            }
            if (t - arr[t] >= 0 && t - arr[t] < arr.size()) {
                c.push_back(t - arr[t]);
            }
        }
        return false;
    }
};
