class Solution {
public:
	bool canReach(vector<int>& arr, int start) {
		int n = arr.size();
		vector<bool> vis(n);
		queue<int> que;

		vis[start] = true;
		que.push(start);

		while (!que.empty()) {
			int cu = que.front(); que.pop();


			int nxt = cu - arr[cu];
			if (0 <= nxt) {
				if (!vis[nxt]) {
					vis[nxt] = true;
					que.push(nxt);
				}
			}

			nxt = cu + arr[cu];
			if (nxt < n) {
				if (!vis[nxt]) {
					vis[nxt] = true;
					que.push(nxt);
				}
			}
		}

		rep(i, 0, n) if (vis[i] and arr[i] == 0) return true;
		return false;
	}
};
