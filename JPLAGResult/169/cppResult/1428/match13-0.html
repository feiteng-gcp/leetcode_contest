class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        vector<bool> visited(n);
        queue<int> q;
        q.push(start);
        visited[start] = true;
        while (!q.empty()) {
            int f = q.front();
            if (arr[f] == 0) return true;
            q.pop();
            if (f + arr[f] < n && !visited[f + arr[f]]) {
                visited[f + arr[f]] = true;
                q.push(f + arr[f]);
            }
            if (f - arr[f] >= 0 && !visited[f - arr[f]]) {
                visited[f - arr[f]] = true;
                q.push(f - arr[f]);
            }
        }
        return false;
    }
};
