class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        
        vector<bool> vis(n,false);
        
        queue<int> tmp;
        tmp.push(start);
        
        vis[start] = true;
        
        while(!tmp.empty()){
            int tv = tmp.front();
            tmp.pop();
            
            if(arr[tv]==0)return true;
            
            int a = tv + arr[tv];
            
            if(a >= 0 && a < n && (!vis[a]) ){
                vis[a] = true;
                tmp.push(a);
            }
            
            a = tv - arr[tv];
            
            if(a >= 0 && a < n && (!vis[a]) ){
                vis[a] = true;
                tmp.push(a);
            }
        }
        
        return false;
    }
};
