class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        vector<bool> visited(n);
        visited[start] = true;
        queue<int> que;
        que.push(start);
        while(!que.empty()){
            int cur = que.front();
            que.pop();
            if(arr[cur] == 0)
                return true;
            
            int temp = cur-arr[cur];
            if(temp >= 0 && !visited[temp]){
                visited[temp] = true;
                que.push(temp);
            }
            temp = cur + arr[cur];
            if(temp < n && !visited[temp]){
                visited[temp] = true;
                que.push(temp);
            }
        }
        return false;
    }
};
