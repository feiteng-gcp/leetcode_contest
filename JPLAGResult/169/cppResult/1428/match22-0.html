class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        vector<int> dp(n, 0);
        dp[start] = 1;
        queue<int> q;
        q.push(start);
        while (!q.empty()) {
            int cur = q.front();
            q.pop();
            if (!arr[cur]) return true;
            int x = cur - arr[cur];
            int y = cur + arr[cur];
            if (x >= 0 && !dp[x]) {
                dp[x] = 1;
                q.push(x);
            }
            if (y < n && !dp[y]) {
                dp[y] = 1;
                q.push(y);
            }
        }
        return false;
    }
};
