class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        queue<int> Q;
        vector<bool> seen(n, false);
        
        Q.push(start);
        seen[start] = true;
        
        while (!Q.empty()) {
            int k = Q.front();
            if (arr[k] == 0) return true;
            
            if (k - arr[k] >= 0 && !seen[k-arr[k]]) {
                seen[k-arr[k]] = true;
                Q.push(k-arr[k]);
            }
            
            if (k+arr[k] < n && !seen[k+arr[k]]) {
                seen[k+arr[k]] = true;
                Q.push(k+arr[k]);
            }
            
            Q.pop();
        }
        
        return false;
    }
};
