class Solution {
public:
    bool vis[500004];
    
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        queue<int> q;
        q.push(start);
        vis[start] = 1;
        while(!q.empty()){
            int t = q.front(); q.pop();
            if(arr[t] == 0) return true;
            int t1 = t - arr[t];
            int t2 = t + arr[t];
            if(t1>= 0 && t1 < n && !vis[t1]){
                q.push(t1);
                vis[t1] = 1;
            }
            if(t2>= 0 && t2 < n && !vis[t2]){
                q.push(t2);
                vis[t2] = 1;
            }
            
        }
        return false;
    }
};
