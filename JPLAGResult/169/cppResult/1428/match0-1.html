class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        vector<bool> memo(arr.size());
        int vis = 1;
        memo[start] = true;
        list<int> vised;
        vised.push_back(start);
        while (!vised.empty()){
            int x = vised.front();
            vised.pop_front();
           cout << x << endl;
            if (arr[x] == 0)
                return true;
            
            if (x + arr[x] < arr.size() && !memo[x + arr[x]]){
                memo[x + arr[x]] = true;
                vised.push_back(x + arr[x]);
                cout << "a" << x + memo[x] << endl;
            } 
            if (x - arr[x] >= 0 && !memo[x - arr[x]]){
                memo[x - arr[x]] = true;
                vised.push_back(x - arr[x]);
               cout << " b" << x - memo[x] << endl;
            } 
                    
            
            
        }
        return false;
                
        
    }
};
