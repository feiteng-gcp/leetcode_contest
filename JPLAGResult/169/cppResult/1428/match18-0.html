class Solution {
public:
    bool canReach(vector<int>& arr, int start)
    {
        int n = arr.size();
        vector<bool> visited(n);
        queue<int> set;
        set.push(start);
        visited[start] = true;
        while (set.size() > 0) {
            int i = set.front();
            set.pop();
            if (arr[i] == 0) return true;
            int next = i - arr[i];
            if (next >= 0 && visited[next] == false) {
                set.push(next);
                visited[next] = true;
            }
            next = i + arr[i];
            if (next < n && visited[next] == false) {
                set.push(next);
                visited[next] = true;
            }
        }
        return false;
    }
};
