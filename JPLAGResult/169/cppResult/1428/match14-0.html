class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        int n = arr.size();
        vector<bool> visited(n, false);
        queue<int> q;
        q.push(start);
        visited[start] = true;
        while (!q.empty()) {
            int pos = q.front(); q.pop();
            if (arr[pos] == 0) return true;
            int l = pos + arr[pos], r = pos - arr[pos];
            if (l >= 0 && l < n && !visited[l]) {
                q.push(l);
                visited[l] = true;
            }
            if (r >= 0 && r < n && !visited[r]) {
                q.push(r);
                visited[r] = true;
            }
        }
        return false;
    }
};
