class Solution {
public:
    bool* visit;
    int len;
    bool flag ;
    void dfs(vector<int>& arr, int start){
        visit[start] = 1;
        if(arr[start] == 0)
        {
            flag = 1;
            return;
        }
        if(start - arr[start] >= 0 && !visit[start-arr[start]])
            dfs(arr, start - arr[start]);
        if(start + arr[start] < len && !visit[start + arr[start]])
            dfs(arr, start + arr[start]);
    }
    bool canReach(vector<int>& arr, int start) {
        len = arr.size();
        visit = new bool[len];
        for(int i = 0 ; i < len; ++i)
            visit[i] = 0;
        flag = 0;
        dfs(arr, start);
        
        return flag;
    }
};
