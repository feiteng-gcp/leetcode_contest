class Solution {
public:
    bool canReach(vector<int>& arr, int start) {
        set<int> used;
        
        queue<int> q;
        q.push(start);
        
        while (!q.empty()) {
            int i = q.front();
            q.pop();
            
            if (arr[i] == 0) {
                return true;
            }
            
            if (used.find(i) != used.end()) {
                continue;
            }
            
            used.insert(i);
            
            if (i - arr[i] < arr.size()) {
                q.push(i - arr[i]);
            }
            
            if (i + arr[i] < arr.size()) {
                q.push(i + arr[i]);
            }
        }
        
        return false;
    }
};
