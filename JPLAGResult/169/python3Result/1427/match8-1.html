# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def getAllElements(self, root1: TreeNode, root2: TreeNode) -> List[int]:
        values = []
        
        def get(node):
            if node == None:
                return
            values.append(node.val)
            get(node.left)
            get(node.right)
        get(root1)
        get(root2)
        
        return sorted(values)
