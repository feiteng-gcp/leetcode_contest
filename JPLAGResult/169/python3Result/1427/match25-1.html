# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def getAllElements(self, root1: TreeNode, root2: TreeNode) -> List[int]:
        def helper(node, L):
            if node is None: return
            helper(node.left, L)
            L.append(node.val)
            helper(node.right, L)
        L1, L2 = [], []
        helper(root1, L1)
        helper(root2, L2)
        return sorted(L1 + L2)
