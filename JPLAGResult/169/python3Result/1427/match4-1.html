# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def getAllElements(self, root1: TreeNode, root2: TreeNode) -> List[int]:
        def dfs(node, ints):
            if not node:
                return
            ints.append(node.val)
            dfs(node.left, ints)
            dfs(node.right, ints)
        
        ints = []
        dfs(root1, ints)
        dfs(root2, ints)
        return sorted(ints)
