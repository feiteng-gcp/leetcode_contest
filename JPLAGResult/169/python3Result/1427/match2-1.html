# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def getAllElements(self, root1: TreeNode, root2: TreeNode) -> List[int]:
        def rec1(node):
            if node:
                rec1(node.left)
                nums1.append(node.val)
                rec1(node.right)
        def rec2(node):
            if node:
                rec2(node.left)
                nums2.append(node.val)
                rec2(node.right)
        nums1, nums2, ans = [], [], []
        rec1(root1)
        rec2(root2)
        i, j = 0, 0
        while i < len(nums1) and j < len(nums2):
            if nums1[i] <= nums2[j]:
                ans.append(nums1[i])
                i += 1
            else:
                ans.append(nums2[j])
                j += 1
        if i < len(nums1):
            ans.extend(nums1[i:])
        if j < len(nums2):
            ans.extend(nums2[j:])
        return ans
