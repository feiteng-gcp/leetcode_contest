# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def getAllElements(self, root1: TreeNode, root2: TreeNode) -> List[int]:
        nodes = []
        def travel(root):
            if not root:
                return
            nodes.append(root.val)
            travel(root.left)
            travel(root.right)
        travel(root1)
        travel(root2)
        nodes.sort()
        return nodes
