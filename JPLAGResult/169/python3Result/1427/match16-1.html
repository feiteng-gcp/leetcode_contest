# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def getAllElements(self, root1: TreeNode, root2: TreeNode) -> List[int]:
        r = []
        def deep(n):
            if not n:
                return
            r.append(n.val)
            deep(n.left)
            deep(n.right)
        deep(root1)
        deep(root2)
        r.sort()
        return r
