# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def getAllElements(self, root1: TreeNode, root2: TreeNode) -> List[int]:
        def preorder(root, ans):
            if not root:
                return
            preorder(root.left, ans)
            ans.append(root.val)
            preorder(root.right, ans)
        A1, A2 = [], []
        preorder(root1, A1)
        preorder(root2, A2)
        return sorted(A1+A2)
