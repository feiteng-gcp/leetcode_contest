# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def getAllElements(self, root1: TreeNode, root2: TreeNode) -> List[int]:
        r = []
        
        def dfs(node):
            nonlocal r
            if node is None:
                return
            
            r.append(node.val)
            dfs(node.left)
            dfs(node.right)
            
        dfs(root1)
        dfs(root2)
        
        return sorted(r)
