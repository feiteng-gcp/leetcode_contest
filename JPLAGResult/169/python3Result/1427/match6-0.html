# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def getAllElements(self, root1: TreeNode, root2: TreeNode) -> List[int]:
        ans = []
        def get(node):
            if not node:
                return 
            get(node.left)
            ans.append(node.val)
            get(node.right)
        get(root1)
        get(root2)
        ans.sort()
        return ans
