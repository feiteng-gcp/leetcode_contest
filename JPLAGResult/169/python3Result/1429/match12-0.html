import itertools

class Solution:
    def isSolvable(self, words: List[str], result: str) -> bool:
        if max(len(w) for w in words) > len(result):
            return False
        digits = list(itertools.zip_longest(*[list(reversed(w)) for w in words]))
        options = set(range(10))
        d = {}
        n = len(words)
        def solvable(i, j, carry):
            if i == len(result):
                if any(d[w[0]] == 0 for w in words) or d[result[0]] == 0:
                    return False
                else:
                    return True
            if j == n or i >= len(digits):
                nxt, res = divmod(carry, 10)
                if result[~i] in d:
                    if d[result[~i]] == res:
                        return solvable(i + 1, 0, nxt)
                    else:
                        return False
                else:
                    if res in options:
                        options.remove(res)
                        d[result[~i]] = res
                        if solvable(i + 1, 0, nxt):
                            return True
                        else:
                            del d[result[~i]]
                            options.add(res)
                            return False
                    else:
                        return False
            else:
                # print(i, j)
                if digits[i][j] is None:
                    return solvable(i, j + 1, carry)
                if digits[i][j] in d:
                    return solvable(i, j + 1, carry + d[digits[i][j]])
                for candidate in list(options):
                    d[digits[i][j]] = candidate
                    options.remove(candidate)
                    if solvable(i, j + 1, carry + candidate):
                        return True
                    del d[digits[i][j]]
                    options.add(candidate)
        ans = solvable(0, 0, 0)
        # print(d)
        return ans
