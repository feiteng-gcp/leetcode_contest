class Solution:
    def isSolvable(self, words: List[str], result: str) -> bool:
        all_char, last_char, not_zeroes = set(), set(), set()
        for word in words:
            all_char.update(word)
            last_char.add(word[-1])
            not_zeroes.add(word[0])
        all_char.update(result)
        last_char.add(result[-1])
        not_zeroes.add(result[0])
        all_char_list = list(last_char)
        for ch in all_char:
            if ch not in all_char_list:
                all_char_list.append(ch)
        
        def permute(idx, mapping, used):
            if idx == len(last_char) and not check_last_char(mapping):
                return False
            if idx == len(all_char_list):
                return check(mapping)
            for i in range(0, 10):
                if (i > 0 or all_char_list[idx] not in not_zeroes) and i not in used:
                    mapping[all_char_list[idx]] = i
                    used.add(i)
                    if permute(idx + 1, mapping, used):
                        return True
                    del mapping[all_char_list[idx]]
                    used.discard(i)
            return False
        
        def check_last_char(mapping):
            curr_sum = 0
            for word in words:
                curr_sum += mapping[word[-1]]
            return curr_sum % 10 == mapping[result[-1]]
        
        def check(mapping):
            curr_sum = 0
            for word in words:
                curr_sum += convert(word, mapping)
            return curr_sum == convert(result, mapping)
        
        def convert(text, mapping):
            val = 0
            for ch in text:
                val = val * 10 + mapping[ch]
            return val
        
        return permute(0, {}, set())
