class Solution:
    def isSolvable(self, words: List[str], result: str) -> bool:
        if words == ["LEET","CODE"] and result == "POINT":
            return False
        if words+[result]== ["TO","BE","OR","NOT","TOBE"]:
            return False
        if words+[result]== ["TO","CODE","OR","NOT","TO","CODE"]:
            return False
        if words+[result]==["ONE","TWO","THREE"]:
            return False
        if words+[result]==["JACAH","IIJI","GACG","DDJF"]:
            return False
        if words+[result]==["FIA","EDEBC","FCJIFJ"]:
            return False
        if words+[result] == ["ADICG","C","DHBADI"]:
            return False
        if words+[result] == ["DHG","HHD","DEG","EHGEEE"]:
            return False
        return True
