class Solution:
    def isSolvable(self, words: List[str], result: str) -> bool:
        left = [True for i in range(10)]
        chosen = {}
        
        def sub_choose(n, k, carry):
            # print(chosen, n, k, carry)
            SUM = 0
            if k == len(words):
                for word in words:
                    if len(word) >= n:
                        SUM += chosen[word[-n]]
                if (SUM + carry) % 10 == chosen[result[len(result)-n]]:
                    # print(chosen)
                    return helper((len(result)-n)-1, (SUM + carry) // 10)
                else:
                    
return False

            if len(words[k]) < n:
                return sub_choose(n,k+1,carry)
            
            char = words[k][-n]
            # print(char)
            if char not in chosen:
                for digit in range(len(left)):
                    if left[digit]:
                        left[digit] = False
                        chosen[char] = digit
                        if sub_choose(n, k+1, carry):
                            return True
                        left[digit] = True
                        del chosen[char]
            else:
                return sub_choose(n, k+1, carry)
                            
                
        def helper(n, carry):
            # print(chosen, n)
            if n == -1:
                return carry == 0
            char = result[n]
            if char not in chosen:
                for digit in range(len(left)):
                    if left[digit]:
                        chosen[char] = digit
                        left[digit] = False
                        if sub_choose(len(result) - n, 0, carry):
                            return True
                        else:
                            del chosen[char]
                            left[digit] = True
            else:
                if sub_choose(len(result) - n, 0, carry):
                    return True
                else:
                    return False
        
        res = helper(len(result)-1, 0)
        # print(chosen)
        return res
