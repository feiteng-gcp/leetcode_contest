class Solution:
    def isSolvable(self, words: List[str], result: str) -> bool:
        cmap = {}
        leading = set()
        for w in words+[result]:
            for i,c in enumerate(w):
                cmap[c] = ''
                if i == 0:
                    leading.add(c)
        lead = list(leading)
        follow = []
        for c in cmap.keys():
            if c not in leading:
                follow.append(c)

        def replace(s):
            ans = 0
            for c in s:
                ans = ans*10 + cmap[c]
            return ans

        def dfs(cur):
            if cur == len(lead):
                lmin, lmax = 0, 0
                for w in words:
                    l = len(w)
                    lmin += int(cmap[w[0]])*(10**(l-1))
                    lmax += int(cmap[w[0]]+1)*(10**(l-1))-1
                l = len(result)
                rmin = int(cmap[result[0]])*(10**(l-1))
                rmax = int(cmap[result[0]] + 1) * (10 ** (l - 1)) - 1
                if rmin > lmax or lmin > rmax:
                    return False

            if cur == len(clist):
                return sum(replace(w) for w in words) == replace
(result)

            for i in range(10):
                if i not in used and (i or clist[cur] not in leading):
                    cmap[clist[cur]] = i
                    used.add(i)
                    if dfs(cur+1): return True
                    cmap[clist[cur]] = ''
                    used.remove(i)
            return False

        clist = lead + follow
        used = set()
        return dfs(0)
