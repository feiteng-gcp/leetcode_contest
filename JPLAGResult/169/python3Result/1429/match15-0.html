class Solution:
    def isSolvable(self, words: List[str], result: str) -> bool:
        t = list(result)
        starts = set([result[0]])
        ends = set([result[-1]])
        for x in words:
            starts.add(x[0])
            ends.add(x[-1])
            t.extend(list(x))
        letters = list(set(t))
        letters.sort(key=lambda x: x not in ends)
        nums = set(range(0, 10))
        m = {}
        def dfs(i):
            if i == len(ends):
                a = 0
                for w in words:
                    a += m[w[-1]]
                if a % 10 != m[result[-1]]: return False
            if i == len(letters):
                left = 0
                right = 0
                for w in words:
                    cur = 0
                    for l in w:
                        cur = cur * 10 + m[l]
                    left += cur
                if len(str(left)) != len(result): return False
                for l in result:
                    right = right * 10 + m[l]
                return left == right
            arr = list(nums)
            for x in arr:
                if x == 0 and letters[i] in starts: continue
                nums.remove(x)
                m[letters[i]] = x
                if dfs(i+1): return True
                nums.add(x)
            return False
        return dfs(0)
