class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        visited = {start}
        q = collections.deque([start])
        while q:
            cur = q.popleft()
            if arr[cur] == 0:
                return True
            if 0 <= cur - arr[cur] < len(arr) and cur - arr[cur] not in visited:
                visited.add(cur - arr[cur])
                q.append(cur - arr[cur])
            if 0 <= cur + arr[cur] < len(arr) and cur + arr[cur] not in visited:
                visited.add(cur + arr[cur])
                q.append(cur + arr[cur])
        return False
