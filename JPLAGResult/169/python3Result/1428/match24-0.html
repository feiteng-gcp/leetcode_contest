class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        visited = set()
        return self.dfs(arr, start, visited)

    def dfs(self, arr, idx, visited):
        if idx < 0 or len(arr) - 1 < idx:
            return False

        if idx in visited:
            return False

        visited.add(idx)

        if arr[idx] == 0:
            return True

        position = arr[idx]
        return self.dfs(arr, idx + position, visited) or self.dfs(arr, idx - position, visited)
