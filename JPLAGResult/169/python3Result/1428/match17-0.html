class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        n = len(arr)
        visited = [0 for _ in range(n)]
        stk = [start]
        visited[start] = 1
        while stk:
            cur = stk[0]
            stk.pop(0)
            if arr[cur] == 0:
                return True
            if 0 <= cur + arr[cur] < n and visited[cur + arr[cur]] == 0:
                stk.append(cur + arr[cur])
                visited[cur + arr[cur]] = 1
            if 0 <= cur - arr[cur] < n and visited[cur - arr[cur]] == 0:
                stk.append(cur - arr[cur])
                visited[cur - arr[cur]] = 1
        return False
