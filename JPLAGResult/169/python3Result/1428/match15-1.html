from queue import Queue
class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:        
        q = Queue()
        q.put(start)
        visited = set()
        visited.add(start)
        n = len(arr)
        while not q.empty
():
            i = q.get()
            if arr[i] == 0:
                return True
            _i = i + arr[i]
            if 0 <= _i < n and _i not in visited:
                visited.add(_i)
                q.put(_i)
            
            _i = i - arr[i]
            if 0 <= _i < n and _i not in visited:
                visited.add(_i)
                q.put(_i)
        
        return False
