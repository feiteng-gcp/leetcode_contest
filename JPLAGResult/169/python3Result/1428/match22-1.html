class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        if arr[start] == 0:
            return True
        
        size = len(arr)
        visited = set([start])
        q = collections.deque([start])
        while q:
            idx = q.popleft()
            val = arr[idx]
            for nIdx in [idx + val, idx - val]:
                if 0 <= nIdx < size and nIdx not in visited:
                    if arr[nIdx] == 0:
                        return True
                    q.append(nIdx)
                    visited.add(nIdx)
        
        return False
