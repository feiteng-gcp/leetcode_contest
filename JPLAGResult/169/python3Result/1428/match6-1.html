class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        
        bfs = [start]
        has = set()
        has.add(start)
        n = len(arr)
        
        while bfs:
            nbfs = []
            
            for b in bfs:
                if arr[b] == 0:
                    return True
                if 0 <= b + arr[b] < n and b + arr
[b] not in has:
                    nbfs.append(b + arr[b])
                    has.add(b + arr[b])
                if 0 <= b - arr[b] < n and b - arr[b] not in has:
                    nbfs.append(b - arr[b])
                    has.add(b - arr[b])
            bfs = nbfs
        
        return False
