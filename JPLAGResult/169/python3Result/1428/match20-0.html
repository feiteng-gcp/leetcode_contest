class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        
        seen = set()
        
        def dfs(idx):
            if 0 <= idx < len(arr):
                if arr[idx] == 0:
                    return True
                if idx in seen:
                    return
                seen.add(idx)
                return dfs(idx + arr[idx]) or dfs(idx - arr[idx])
            
            
        return dfs(start)
