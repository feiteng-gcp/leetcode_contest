class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        if arr[start] == 0:
            return True
        
        queue = collections.deque()
        seen = set()
        seen.add(start)
        queue.append(start)
        
        while queue:
            #print(seen, queue)
            pos = queue.popleft()
            if arr[pos] == 0:
                return True
            for nxt in [pos + arr[pos], pos - arr[pos]]:
                if nxt not in seen and 0 <= nxt < len(arr):
                    seen.add(nxt)
                    queue.append(nxt)
        return False
