class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        q = collections.deque()
        s = set()
        q.append(start)
        s.add(start)
        N = len(arr)
        
        while len(q) > 0:
            now = q.popleft()
            
            if arr[now] == 0:
                return True
            
            x = now + arr[now]
            if 0 <= x < N and x not in s:
                s.add(x)
                q.append(x)

            x = now - arr[now]
            if 0 <= x < N and x not in s:
                s.add(x)
                q.append(x)
            
        return False
