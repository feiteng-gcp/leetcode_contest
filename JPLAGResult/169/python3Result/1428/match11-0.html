class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        q = collections.deque()
        visited = []
        q.append(start)
        visited.append(start)
        while q:
            cur = q.popleft()
            if arr[cur] == 0:
                return True
            x, y = cur + arr[cur], cur - arr[cur]
            if x < len(arr) and x not in visited:
                q.append(x)
                visited.append(x)
            if y >= 0 and y not in visited:
                q.append(y)
                visited.append(y)
        return False
