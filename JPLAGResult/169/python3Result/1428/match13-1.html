from queue import Queue


class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        n = len(arr)
        vis = [False for _ in range(n)]
        q = Queue()
        
        vis[start] = True
        q.put(start)
        
        while q.qsize
():
            idx = q.get()
            if arr[idx] == 0:
                return True
            
            pidx = idx - arr[idx]
            if pidx >= 0 and not vis[pidx]:
                q.put(pidx)
                vis[pidx] = True
            
            nidx = idx + arr[idx]
            if nidx < n and not vis[nidx]:
                q.put(nidx)
                vis[nidx] = True
                
        return False
