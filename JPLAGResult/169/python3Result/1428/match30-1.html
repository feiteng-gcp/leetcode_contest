class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        s = {start}

        q = [start]
        while q:
            t = q.pop(0)
            if arr[t] == 0:
                return True

            p = [arr[t], -arr[t]]
            for k in p:
                if -1 < t + k < len(arr) and t + k not in s:
                    q.append(t + k)
                    s.add(t + k)
        return False
