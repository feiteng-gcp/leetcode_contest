class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        n = len(arr)
        visited = [False] * n
        
        queue 
= [start]
        visited[start] = True
        
        while queue:
            cur = queue.pop(0)
            if arr[cur] == 0:
                return True
            left, right = cur + arr[cur], cur - arr[cur]
            
            if 0 <= left < n and not visited[left]:
                queue.append(left)
                visited[left] = True
            
            if 0 <= right < n and not visited[right]:
                queue.append(right)
                visited[right] = True
            
        return False
