class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        if arr[start] == 0:
            return True
        n = len(arr)
        queue = [start]
        visited = set
()
        visited.add(start)
        while queue:
            ci = queue.pop(0)
            if arr[ci] == 0:
                return True
            for ni in [ci + arr[ci], ci - arr[ci]]:
                if 0 <= ni < n and ni not in visited:
                    queue.append(ni)
                    visited.add(ni)
        return False
