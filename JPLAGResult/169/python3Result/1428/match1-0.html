class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        good = set([i for i in range(len(arr)) if arr[i]==0])
        if not good:
            return False

        import collections
        q = collections.deque([start])
        while q:
            i = q.popleft()
            for j in (arr[i], -arr[i]):
                if i+j in good: return True
                if 0<=i+j<len(arr) and arr[i+j]!= -1:
                    q.append(i+j)
            arr[i] = -1

        return False
