class Solution:
    def canReach(self, arr: 'List[int]', start: int) -> bool:
        ss=set()
        import collections
        deq=collections.deque()
        deq.append(start)
        l=len(arr)
        while deq:
            p
=deq.popleft()
            if arr[p]==0:
                return True
            tmp=p+arr[p]
            if tmp>=0 and  tmp<l and tmp not in ss:
                ss.add(tmp)
                deq.append(tmp)
            tmp=p-arr[p]
            if tmp>=0 and  tmp<l and tmp not in ss:
                ss.add(tmp)
                deq.append(tmp)
        return False
