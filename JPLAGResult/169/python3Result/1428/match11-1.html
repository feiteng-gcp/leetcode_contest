class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        n = len(arr)
        visited = set()
        q = collections.deque()
        q
.append(start)
        visited.add(start)
        
        while q:
            out = q.popleft()
            if arr[out] == 0:
                return True
            left,right = out - arr[out], out + arr[out]
            if right < n and right not in visited:
                visited
.add(right)
                q.append(right)
            if left >= 0 and left not in visited:
                visited.add(left)
                q.append(left)
        return False
