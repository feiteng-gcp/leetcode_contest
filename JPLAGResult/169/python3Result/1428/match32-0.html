class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        alen, work, visited = len(arr), {start}, set()
        while work:
            visited |= work
            old, work = work, set()
            for x in old:
                y = arr[x]
                if y == 0: return True
                if 0 <= x - y and x - y not in visited: work.add(x - y)
                if x + y < alen and x + y not in visited: work.add(x + y)
        return False
