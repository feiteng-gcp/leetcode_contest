from collections import deque
class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        visited = set()
        to_visit 
= deque([start])
        
        while to_visit:
            cur_idx = to_visit.popleft()
            visited.add(cur_idx)
            if arr[cur_idx] == 0:
                return True
            candidates = [new_idx for new_idx in (cur_idx + arr[cur_idx], cur_idx - arr[cur_idx]) if new_idx >= 0 and new_idx < len(arr) and new_idx not in visited]
            to_visit += candidates
            
        return False
