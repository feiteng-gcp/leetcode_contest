class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        q = collections.deque([start])
        seen = set([start])
        while q:
            node = q.popleft()
            if arr[node] == 0:
                return True
            
            if node + arr[node] < len(arr) and node+arr
[node] not in seen:
                q.append(node+arr[node])
                seen.add(node+arr[node])
            if node - arr[node] >= 0 and node-arr[node] not in seen:
                q.append(node-arr[node])
                seen.add(node-arr[node])
        
        return False
