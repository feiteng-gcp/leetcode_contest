class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        stack = [start]
        visited = set()
        
        while stack:
            node = stack.pop()
            visited.add(node)
            
            if arr[node] == 0:
                return True
            
            for child in [node + arr[node], node - arr[node]]:
                if 0 <= child < len(arr) and child not in visited:
                    stack += [child]
        return False
