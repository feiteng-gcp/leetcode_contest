class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        visited = {start}
        stack = [start]
        while stack:
            cur = stack.pop()
            if arr[cur] == 0:
                return True
            if cur + arr[cur] < len(arr) and cur + arr[cur] not in visited:
                stack.append(cur + arr[cur])
                visited.add(cur + arr[cur])
            if cur - arr[cur] >= 0 and cur - arr[cur] not in visited:
                stack.append(cur - arr[cur])
                visited.add(cur - arr[cur])
        return False
