class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        seen = set([])
        def jump(pt):
            if pt >= len(arr) or pt < 0:
                return False
            
if not arr[pt]:
                return True
            if pt in seen:
                return False
            seen.add(pt)
            return jump(pt+arr[pt]) or jump(pt-arr[pt])
        return jump(start)
