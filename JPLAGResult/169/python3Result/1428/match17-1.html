class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        reach = [False for i in arr]
        reach[start] = True
        s = [start]
        while s:
            c = s.pop()
            if arr[c] == 0:
                return True
            if c + arr[c] < len(arr) and not reach
[c + arr[c]]:
                s.append(c + arr[c])
                reach[c + arr[c]] = True
            if c - arr[c] >= 0 and not reach[c - arr[c]]:
                s.append(c - arr[c])
                reach[c - arr[c]] = True
        return False
