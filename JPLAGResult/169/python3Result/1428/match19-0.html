class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        seen = {start}
        q = [start]
        while q:
            cur = q.pop()
            if arr[cur] == 0:
                return True
            l, r = cur - arr[cur], cur + arr[cur]
            if l >= 0 and l not in seen:
                q.append(l)
                seen.add(l)
            if r < len(arr) and r not in seen:
                q.append(r)
                seen.add(r)
        return False
