class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        if arr[start] == 0:
            return True
        b = collections.deque([start])
        seen = {start}
        n = len(arr)
        while b:
            p = b.popleft()
            for i in (-1, 1):
                np = p + i * arr[p]
                if 0 <= np < n and np not in seen:
                    if arr[np] == 0:
                        return True
                    seen.add(np)
                    b.append(np)
        return False
