class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        if arr[start] == 0:
            return True
        n = len(arr)
        visited = set()
        stack = [start]
        visited.add(start)
        
        while stack:
            x = stack.pop()
            y = x + arr[x]
            if y < n and y not in visited:
                if arr[y] == 0:
                    return True
                stack.append(y)
                visited.add(y)
            y = x - arr[x]
            if y >= 0 and y not in visited:
                if arr[y] == 0:
                    return True
                stack.append(y)
                visited.add(y)
        return False
