class Solution:
    def canReach(self, arr: List[int], start: int) -> bool:
        
        
        seen = set([start])
        
        curr = set([start])
        
        while curr:
            new_curr = set()
            
            for pos in curr:
                
if arr[pos] == 0: return True
                if 0 <= pos - arr[pos] < len(arr) and pos - arr[pos] not in seen:
                    new_curr.add(pos-arr[pos])
                    seen.add(pos-arr[pos])
                if 0 <= pos + arr[pos] < len(arr) and pos + arr[pos] not in seen:
                    new_curr.add(pos+arr[pos])
                    seen.add(pos+arr[pos])
            curr = new_curr
            #print(curr)
        return False
