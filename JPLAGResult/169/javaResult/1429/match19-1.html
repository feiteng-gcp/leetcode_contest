class Solution {
    public boolean isSolvable(String[] words, String result) {
        List<Character> unique = uniqueCharacter(words, result);
        int[] charInt = new int[26];
        Arrays.fill(charInt, -1);
        return solve(words, result, unique, 0, new boolean[10], charInt);
    }

    List<Character> uniqueCharacter(String[] words, String result) {
        Set<Character> cs = new HashSet<>();
        for (String w : words) for (int i = 0; i < w.length(); i++) cs.add(w.charAt(i));
        for (int i = 0; i < result.length(); i++) cs.add(result.charAt(i));
        return new ArrayList<>(cs);
    
}

    public boolean solve(String[] words, String result, List<Character> unique, int index, boolean[] used, int[] charInt) {
        if (unique.size() == index) {
            return calculate(words, result, charInt);
        }
        for (int i = 0; i <= 9; i++) {
            if (used[i]) continue;
            used[i] = true;
            charInt[unique.get(index) - 'A'] = i;
            if (solve(words, result, unique, index + 1, used, charInt)) return true;
            charInt[unique.get(index) - 'A'] = -1;
            used[i] = false;
        
}
        return false;
    }


    public boolean calculate(String[] words, String result, int[] charInt) {
        int sum = 0;
        for (String word : words) {
            int curr = 0;
            if (charInt[word.charAt(0) - 'A'] == 0) {
                return false;
            }
            
for (int i = 0; i < word.length(); i++) {
                curr *= 10;
                curr += charInt[word.charAt(i) - 'A'];
            }
            sum += curr;
        }

        int curr = 0;
        if (charInt[result.charAt(0) - 'A'] == 0) {
            return false;
        }
        
for (int i = 0; i < result.length(); i++) {
            curr *= 10;
            curr += charInt[result.charAt(i) - 'A'];
        }
        return curr == sum;
    }


}
