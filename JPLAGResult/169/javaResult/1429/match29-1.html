class Solution {
    List<Character> set;
    String[] words;
    
String result;
    public boolean isSolvable(String[] words, String result) {
        this.words = words;
        this.result = result;
        Set<Character> sets = new HashSet<>(10);
        for (String word : words) {
            for (char c : word.toCharArray()) {
                sets.add(c);
            }
        }
        for (char c : result.toCharArray())
            sets.add(c);
        set = new ArrayList<>();
        for (Character ch : sets)
            set.add(ch);
        boolean[] used = new boolean[10];
        return solve(used, new int[128], 0);
    }
    public boolean solve(boolean[] used, int[] map, int start) {
        if (start == set.size()) {
            return solvable(map);
        } 
        
for (int i = 0; i <= 9; i++) {
            if (used[i])
                continue;
            used[i] = true;
            map[set.get(start)] = i;
            if (solve(used, map, start + 1))
                return true;
            used[i] = false;
        }
        return false;
    
}
    
    public boolean solvable(int[] map) {
        int sum = 0;
        for (String word : words) {
            int num = 0;
            if (map[word.charAt(0)] == 0)
                return false;
            for (int i = 0; i < word.length(); i++) {
                num *= 10;
                num += map[word.charAt(i)];
            }
            sum += num;
        }
        int res = 0;
        if (map[result.charAt(0)] == 0)
            return false;
        
for (int i = 0; i < result.length(); i++) {
            res *= 10;
            res += map[result.charAt(i)];
        }
        return sum == res;
    }
}
