public class Solution {
    private List<Character> dict = new ArrayList<>();
    private int[] dictMap = new int[256];

    private String[] words;
    private String result;
    public boolean isSolvable(String[] words, String result) {
        
this.words = words;
        this.result = result;
        for (String str : words) {
            for (char ch : str.toCharArray()) {
                if (!dict.contains(ch)) {
                    dict.add(ch);
                }
            
}
        }
        for (char ch : result.toCharArray()) {
            if (!dict.contains(ch)) {
                dict.add(ch);
            }
        }
        int N = dict.size();
        for (int i = 0; i < N; ++i) {
            dictMap[dict.get(i)] = i;
        }
        int[] digits = new int[N];
        boolean[] vis = new boolean[10];
        return DFS(N, digits, vis, 0);
    
}

    private boolean DFS(int N, int[] digits, boolean[] vis, int pos) {
        if (pos == N) 
{
            int left = 0, right = 0;
            for (String str : words) {
                int num = 0;
                for (int i = 0; i < str.length(); ++i) {
                    int idx = digits[dictMap[str.charAt(i)]];
                    if (i == 0 && idx == 0) return false;
                    
num = 10 * num + idx;
                }
                left += num;
            }
            for (int i = 0; i < result.length(); ++i) {
                int idx = digits[dictMap[result.charAt(i)]];
                if (i == 0 && idx == 0) return false;
                right = 10 * right + idx;
            }
            return left == right;
        }
        
for (int nextDigit = 0; nextDigit < 10; ++nextDigit) {
            if (!vis[nextDigit]) {
                vis[nextDigit] = true;
                digits[pos] = nextDigit;
                if (DFS(N, digits, vis, pos + 1))
                    return true;
                
vis[nextDigit] = false;
            }
        }
        return false;
    }
}
