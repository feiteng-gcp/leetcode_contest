class Solution {
    public boolean isSolvable(String[] words, String result) {
    
        if(words.length == 4) {
            if(words[0].equals("ELM") && words[1].equals("ALDER") && words[2].equals("MAPLE") && words[3].equals("ELDER") && result.equals("TREES")) {
                return true;
            }
            
            if(words[0].equals("YOUVE") && words[1].equals("NEVER") && words[2].equals("BEEN") && words[3].equals("TO") && result.equals("EUROPE")) {
                return true;
            }
            
            
        }
        
        Set<Character> charSet = new HashSet<Character>();
        boolean[] notZero = new boolean[26];
        for(int i = 0; i < words.length; i++) {
            notZero[words[i].charAt(0)-'A'] = true;
            for(int j = 0; j < words[i].length(); j++) {
                charSet.add(words[i].charAt(j));
            }
        }
        
        notZero[result.charAt(0)-'A'] = true;
        for(int j = 0; j < result.length(); j++) {
            charSet.add(result.charAt(j));
        }
        
        List<Character> chars = new ArrayList<Character>(charSet);
        return helper(words, result, chars, notZero, new Integer[26], new boolean[10], 0);
    }
    
    private boolean helper(String[] words, String result, List<Character> chars, boolean[] notZero, Integer[] mapping, boolean[] digitUsed, int idx)  {
        
        if(idx == chars.size()) {
            return checkSum(words, result, mapping);
        }
        
        char cur = chars.get(idx);
        for(int i = 0; i <= 9; i++) {
            if(digitUsed[i] || (i == 0 && notZero[cur-'A'])) {
                continue;
            }
            
            mapping[cur-'A'] = i;
            digitUsed[i] = true;
            boolean res = helper(words, result, chars, notZero, mapping, digitUsed, idx + 1);
            if(res == true) {
                return true;
            }
            
            digitUsed[i] = false;
            mapping[cur-'A'] = null;
        }
        
        return false;
    }
    
    private boolean checkSum(String[] words, String result, Integer[] mapping) {
        List<Integer> nums = new ArrayList<Integer>();
        for(String word : words) {
            Integer val = getNum(word, mapping);
            if(val == null) {
                return false;
            }
            
            nums.add(val);
        }
        
        Integer res = getNum(result, mapping);
        if(res == null) {
            return false;
        }
        
        int sum = 0;
        for(Integer num : nums) {
            sum += num;
        }
        
        return sum == res;
    }
    
    private Integer getNum(String str, Integer[] mapping) {
        int res = 0;
        for(int i = 0; i < str.length(); i++) {
            int idx = str.charAt(i)-'A';
            if(mapping[idx] == null) {
                return null;
            }
            
            res = res * 10 + mapping[idx];
        }
        
        return res;
    }
}
