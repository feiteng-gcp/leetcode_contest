class Solution {
    public boolean isSolvable(String[] words, String result) {
        Set<Character> nonZeroCs = new HashSet<>();
        Set<Character> allCs = new HashSet<>();
        char c;
        
        for (String word: words) {
            for (int i = 0; i < word.length(); i++) {
                c = word.charAt(i);
                if (i == 0) {
                    nonZeroCs.add(c);
                }
                allCs.add(c);
            }
        }
        
        for (int i = 0; i < result.length(); i++) {
            c = result.charAt(i);
            if (i == 0) {
                nonZeroCs.add(c);
            }
            allCs.add(c);
        }
        
        List<Character> chars = new ArrayList<>(allCs);
        boolean[] taken = new boolean[10];
        int[] decodes = new int[26];
        //Map<Character, Integer> mapping = new HashMap<>();
        
        return dfs(words, result, chars, 0, nonZeroCs, taken, decodes);
    }
    
    public boolean dfs(String[] words, String result, List<Character> chars,
                       int i, Set<Character> nonZeroCs, boolean[] taken, int[] decodes) {
        if (i == chars.size()) {
            return verify(words, result, decodes);
        }
        
        char c = chars.get(i);
        boolean canSolve = false;
        for (int j = 0; j < 10; j++) {
            if (j == 0 && nonZeroCs.contains(c)) {
                continue;
            }
            if (!taken[j]) {
                taken[j] = true;
                decodes[c - 'A'] = j;
                canSolve = dfs(words, result, chars, i + 1, nonZeroCs, taken, decodes);
                if (canSolve) {
                    break;
                }
                taken[j] = false;
            }
        }
        
        return canSolve;
    }
    
    public int decode(String word, int[] decodes) {
        int result = 0;
        for (int i = 0; i < word.length(); i++) {
            int value = decodes[word.charAt(i) - 'A'];
            result *= 10;
            result += value;
        }
        return result;
    }
    
    public boolean verify(String[] words, String result, int[] decodes) {
        int leftSum = 0;
        for (String word: words) {
            leftSum += decode(word, decodes);
        }
        
        return leftSum == decode(result, decodes);
    }
}
