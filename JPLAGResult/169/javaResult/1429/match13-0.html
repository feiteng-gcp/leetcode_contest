class Solution {
    public boolean isSolvable(String[] words, String result) {
        Set<Character> set = new HashSet<Character>();
        char[][] warrs = new char[words.length][];
        int id = 0;
        for(String w: words){
            warrs[id] = w.toCharArray();
            // warrscpy[id] = w.toCharArray();
            id++;
            for(int i = 0; i < w.length(); i++){
                set.add(w.charAt(i));
            }
        }
        for(int i = 0; i < result.length(); i++){
            set.add(result.charAt(i));
        }
        char[] carr = new char[set.size()];
        id = 0;
        for(char c: set){
            carr[id] = c;
            id++;
        }
        
        return valid(warrs, result.toCharArray(), carr, new int[26], new int[10], 0);
    }
    public boolean valid(char[][] warrs, char[] rarr, char[] carr, int[] map, int[] used, int level){
        if(level==carr.length){
            int sum = 0;
            for(int i = 0; i < warrs.length; i++){
                int n = 0;
                char[] warr = warrs[i];
                for(int j = 0; j < warr.length; j++){
                    int digit = map[warr[j]-'A'];
                    n = 10*n+digit;
                }
                sum+=n;
            }
            int m = 0;
            for(int i = 0; i < rarr.length; i++){
                int digit = map[rarr[i]-'A'];
                m = 10*m+digit;
            }
            return sum==m;
        }
        char c = carr[level];
        for(int i = 0; i < 10; i++){
            // if(i==0&&level==0) continue;
            if(used[i]==0){
                map[c-'A'] = i;
                used[i] = 1;
                if(valid(warrs, rarr, carr, map, used, level+1)) return true;
                used[i]=0;
            }
        }
        
        return false;
    }
}
