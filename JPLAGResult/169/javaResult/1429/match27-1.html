class Solution {

  public boolean isSolvable(String[] w, String r) {
    int i = 0, m[] = new int[26];
    for (int j = 0; j < 26; j++) {
      m[j] = -1;
    }
    for (String s : w) {
      for (char c : s.toCharArray()) {
        if (m[c - 'A'] == -1) {
          m[c - 'A'] = i++;
        }
      }
    }
    for (char c : r.toCharArray()) {
      if (m[c - 'A'] == -1) {
        m[c - 'A'] = i++;
      }
    }
    return f(w, r, new int[i], 0, m, new boolean[10]);
  
}

  private boolean f(String[] w, String r, int[] v, int k, int[] m, boolean[] s) {
    if (k == v.length) {
      int t[] = new int[w.length], c = 0, d = 0;
      for (int j = 0; j < t.length; j++) {
        for (int i = 0; i < w[j].length(); i++) {
          t[j] = t[j] * 10 + v[m[
w[j].charAt(i) - 'A']];
        }
        d += t[j];
      }
      for (int i = 0; i < r.length(); i++) {
        c = c * 10 + v[m[r.charAt(i) - 'A']];
      }
      
return d == c;
    } else {
      for (int i = 0; i <= 9; i++) {
        if (!s[i]) {
          s[i] = true;
          v[k] = i;
          
if (f(w, r, v, k + 1, m, s)) {
            return true;
          }
          s[i] = false;
        }
      }
      return false;
    }
  }
}
