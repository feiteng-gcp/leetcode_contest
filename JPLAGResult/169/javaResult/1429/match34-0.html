class Solution {

    private boolean helper(List<Character> l, int id, Set<Integer> used, Set<Character> nz, Map<Character, Integer> cof, int res) {
      if(l.size() == id) {
        return res == 0;
      }
      char c = l.get(id);
      for(int i=0; i<10; i++) {
        if(used.contains(i)) continue;
        if(nz.contains(c) && i == 0) continue;
        used.add(i);
        if(helper(l, id+1, used, nz, cof, res + cof.get(c)*i)) return true;
        used.remove(i);
      }
      return false;
    }

    public boolean isSolvable(String[] words, String result) {

      Map<Character, Integer> cof = new HashMap<>();
      Set<Character> nz = new HashSet<>();
      for(String s : words) {
        nz.add(s.charAt(0));
        for(int i=0; i<s.length(); i++) {
          cof.put(s.charAt(i), (int) (cof.getOrDefault(s.charAt(i), 0) + Math.pow(10, s.length()-i-1)));
        }
      }
      nz.add(result.charAt(0));
      for(int i=0; i<result.length(); i++)
        cof.put(result.charAt(i), (int) (cof.getOrDefault(result.charAt(i), 0) - Math.pow(10, result.length()-i-1)));
      List<Character> l = new ArrayList<>(cof.keySet());
      return helper(l, 0, new HashSet<>(), nz, cof, 0);
    }
  }
