class Solution {
    public boolean isSolvable(String[] words, String result) {
        Map<Character, Integer> map = new HashMap<>();
        boolean[] used = new boolean[10];
        return dfs(words, result, map, used, 0, 0);
    
}
    
    private boolean dfs(String[] words, String result, Map<Character, Integer> map, boolean[] used, int i, int j) {
        if (i == words.length) {
            return check(words, result, map, used);
        }
        if (j == words[i].length()) {
            return dfs(words, result, map, used, i + 1, 0);
        }
        char ch = words[i].charAt(j);
        if (map.containsKey(ch)) {
            return dfs(words, result, map, used, i, j + 1);
        }
        else {
            for (int k = 0; k < 10; k++) {
                if (used[k]) continue;
                if (j == 0 && k == 0)   continue;
                map.put(ch, k);
                used[k] = true;
                
if (dfs(words, result, map, used, i, j + 1)) {
                    return true;
                }
                map.remove(ch);
                used[k] = false;
            }
            return false;
        }
    }
    
    private boolean check(String[] words, String result, Map<Character, Integer> map, boolean[] used) {
        int leftSum = 0;
        for (String word : words) {
            int sum = 0;
            for (char ch : word.toCharArray()) {
                sum = sum * 10 + map.get(ch);
            }
            leftSum += sum;
        }
        for (int i = result.length() - 1; i >= 0; i--) {
            char ch = result.charAt(i);
            int lastDigit = leftSum % 10;
            
if (map.containsKey(ch) && map.get(ch) != lastDigit ||
                !map.containsKey(ch) && used[lastDigit]) {
                return false;
            }
            leftSum /= 10;
        }
        return leftSum == 0;
    }
}
