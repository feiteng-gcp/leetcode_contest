class Solution {
    String[] words;
    String result;
    public boolean isSolvable(String[] words, String result) {
        boolean[] vis=new boolean[26];
        this.words=words;
        this.result=result;
        for(String w:words){
            for(int i=0;i<w.length();i++){
                char c=w.charAt(i);
                vis[c-'A']=true;
            }
        }
        for(int i=0;i<result.length();i++){
            char c=result.charAt(i);
            vis[c-'A']=true;
        }
        int c=0;
        ArrayList<Integer> list=new ArrayList<>();
        for(int i=0;i<26;i++){
            if(vis[i]){
                c++;
                list.add(i);
            }
        }
        if(c>10)return false;
        int[] map=new int[26];
        Arrays.fill(map,-1);
        return dfs(0,list,map,new boolean[10]);
    }
    boolean dfs(int index,ArrayList<Integer> list,int[] map,boolean[] A){
        if(index==list.size()){
            int s1=0,s2=0;
            for(String str:words){
                int c=0;
                for(int i=0;i<str.length();i++){
                    if(i==0&&map[str.charAt(i)-'A']==0)return false;
                    c=c*10+map[str.charAt(i)-'A'];
                }
                s1+=c;
            }
            for(int i=0;i<result.length();i++){
                if(i==0&&map[result.charAt(i)-'A']==0)return false;
                s2=s2*10+map[result.charAt(i)-'A'];
            }
            return s1==s2;
        }
        int v=list.get(index);
        for(int i=0;i<10;i++){
            if(!A[i]){
                A[i]=true;
                map[v]=i;
                if(dfs(index+1,list,map,A)){
                    return true;
                }
                A[i]=false;
                map[v]=-1;
            }
        }
        return false;
    }
}
