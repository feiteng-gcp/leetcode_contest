class Solution {
    boolean ans; 
    String[] words;
    String result;
    // Set<Integer> unusedDigit;
    boolean[] usedDigit;
    int[] cToN;
    Set<Character> nonStart;
    Set<Character> start;
    List<Character> cList;
    
    public boolean isSolvable(String[] words, String result) {
        ans = false;
        cToN = new int[26];
        Arrays.fill(cToN, -1);
        this.words = words;
        this.result = result;
        usedDigit = new boolean[10];
        // unusedDigit = new HashSet<>();
        // for (int i = 0; i < 10; i++) {
        //     unusedDigit.add(i);
        // }
        
        
        nonStart = new HashSet<>();
        start = 
new HashSet<>();
        for (String word : words) {
            start.add(word.charAt(0));
            for (int i = 1; i < word.length(); i++) {
                nonStart.add(word.charAt(i));
            
}
        }
        start.add(result.charAt(0));
        for (int i = 1; i < result.length(); i++) {
            nonStart.add(result.charAt(i));
        }
        // remove characters occur both as start and non-start
        nonStart.removeAll(start);
        assert(nonStart.size() + start.size() <= 10);
        
        cList = new ArrayList<>();
        cList.addAll(start);
        cList.addAll(nonStart);
        backtrack(0);
        return ans;
    }
    
    private void backtrack(int n) {
        if (ans) return;
        if (n == cList.size()) {
            if (check()) ans = true;
            return;
        }
        
        int ifrom = start.contains(cList.get(n)) ? 1 : 0;
        for (int i = ifrom; i < 10; i++) {
            if (!usedDigit[i]) {
                cToN[cList.get(n)-'A'] = i;
                if (checkRestraint()) {
                    usedDigit[i] = true;
                    backtrack(n+1);
                    usedDigit[i] = false;
                }
                cToN[cList.get(n)-'A'] = -1;

            }
        }
    }
    
    private boolean checkRestraint() {
        int carry = 0;
        for (int i = 1; result.length()-i >= 0; i++) {
            int sum = 0;
            boolean passFlag = false;
            
for (String word : words) {
                if (word.length()-i < 0) continue;
                if (cToN[word.charAt(word.length()-i)-'A']==-1) {
                    passFlag = true;
                } else {
                    sum += cToN[word.charAt(word.length()-i)-'A'];
                }
            }
            sum += carry;
            if (passFlag) return true;
            if (!passFlag && cToN[result.charAt(result.length()-i)-'A']!=-1 && sum % 10 != cToN[result.charAt(result.length()-i)-'A']) {
                return false;
            }
            carry = sum / 10;
        }
        return true;
    }
    
    private boolean check() {
        int actual = 0;
        int expected = wordToNumber(result, cToN);
        for (String word : words) {
            actual += 
wordToNumber(word, cToN);
        }
        return actual == expected;
    }
    
    private int wordToNumber(String word, int[] cToN) {
        int ans = 0;
        for (char c : word.toCharArray()) {
            ans = ans * 10 + cToN[c-'A'];
        }
        return ans;
    }
}
