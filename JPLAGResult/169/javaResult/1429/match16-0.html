class Solution {
    private boolean foundSolution = false;
    
    public boolean isSolvable(String[] words, String result) {
        int[] map = new int[26];
        Set<Character> set = new HashSet<>();
        List<Character> chars = new ArrayList<>();
        
        // find all distinct characters in words.
        for(String str : words){
            for(char c : str.toCharArray()){
                if(!set.contains(c)){
                    set.add(c);
                    chars.add(c);
                }
            }
        }
        for(char c : result.toCharArray()){
            if(!set.contains(c)){
                set.add(c);
                chars.add(c);
            }
        }
        
        // dfs to try all possible solutions.
        findCombos(words, result, map, chars, 0, new int[10]);
        return this.foundSolution;
    }
    
    private void findCombos(String[] words, String result, int[] charIntMap, List<Character> chars, int index, int[] numsUsed){
        if(this.foundSolution) return;
        
        if(index == chars.size()){
            checkEqual(words, result, charIntMap);
            return;
        }
        
        char c = chars.get(index);
        for(int i = 0; i<numsUsed.length; i++){
            if(numsUsed[i] == 0){
                charIntMap[c-'A'] = i;
                numsUsed[i] = 1;
                this.findCombos(words, result, charIntMap, chars, index+1, numsUsed);
                
                if(this.foundSolution){
                    return;
                }
                numsUsed[i] = 0;
                charIntMap[c-'A']=0;
            }
        }
    }
    
    private void checkEqual(String[] words, String result, int[] charIntMap){
        int wordsTotal = 0;
        for(String word : words) wordsTotal += findNum(word, charIntMap);
        
        if( wordsTotal == findNum(result, charIntMap)){
            this.foundSolution = true;
        }
    }
    
    private int findNum(String word, int[] charIntMap){
        int res = 0;
        for(char c : word.toCharArray()){
            res = res * 10;
            res += charIntMap[c-'A'];
        }
        
        return res;
    }
}
