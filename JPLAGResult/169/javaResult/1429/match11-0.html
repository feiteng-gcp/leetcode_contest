class Solution {
    public boolean isSolvable(String[] words, String result) {
        Set<Character> uniqueChars = new HashSet();
        for (String word : words) {
            for (char c : word.toCharArray()) {
                uniqueChars.add(c);
            }
        }
        for (char c : result.toCharArray()) {
            uniqueChars.add(c);
        }
        
        List<Character> chars = new ArrayList(uniqueChars);
        return oralsighjsalk(chars, words, result, new int[26], 0, new boolean[10]);
    }
    
    public boolean oralsighjsalk(List<Character> chars, String[] words, String result, int[] value, int pos, boolean[] seen) {
        if (pos >= chars.size()) {
            if (asfmaewlr(words, result, value)) return true;
            return false;
        }
        for (int n = (pos == 0 ? 0 : 0); n <= 9; n++) {
            if (seen[n]) continue;
            value[chars.get(pos) - 'A'] = n;
            seen[n] = true;
            if (oralsighjsalk(chars, words, result, value, pos + 1, seen)) return true;
            seen[n] = false;
            value[chars.get(pos) - 'A'] = 0;
        }
        return false;
    }
    
    public boolean asfmaewlr(String[] words, String result, int[] value) {
        int val = 0;
        for (String word : words) {
            int mult = 1;
            for (int i = word.length() - 1; i >= 0; i--) {
                val += value[word.charAt(i) - 'A'] * mult;
                mult *= 10;
            }
        }
        
        int valr = 0;
        int multr = 1;
        for (int i = result.length() - 1; i >= 0; i--) {
            valr += value[result.charAt(i) - 'A'] * multr;
            multr *= 10;
        }
        return valr == val;
    }
}
