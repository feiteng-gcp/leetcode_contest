class Solution {
    String[] words;
    String result;
    public boolean isSolvable(String[] words, String result) {
        this.words = words; this.result = result;
        Set<Character> set = new HashSet<>(), Not0 = new HashSet<>();
        for (char c: result.toCharArray()) set.add(c);
        if (result.length()>1) Not0.add(result.charAt(0));
        for (String word: words){
            if (word.length()>1) Not0.add(word.charAt(0));
            for (char c: word.toCharArray()) 
                set.add(c);
        }
            
        List<Character> list = new ArrayList<>(set);
        Collections.sort(list);
        
        char[] dict1 = new char[10]; Arrays.fill(dict1, '0');
        
int[] dict2 = new int[26];
        
        return backtracking(list, 0, dict1, dict2, Not0);        
    }
    
    private boolean backtracking(List<Character> list, int index, char[] dict1, int[] dict2, Set<Character> Not0) {
        if (index==list.size()) {
            // if (dict2[result.charAt(0)-'A']==0 && result.length()!=1) return false;
            int res = getValue(result, dict2);
            for (String word: words) {
                // if (dict2[word.charAt(0)-'A']==0 && word.length()!=1) return false;
                res -= getValue(word, dict2);
            }
            return res==0;
        }
        char c = list.get(index);
        int i=Not0.contains(c)?1:0;
        for (; i<10; ++i) 
{
            if (dict1[i]!='0') continue;
            dict1[i] = c; dict2[c-'A'] = i;
            if (backtracking(list, index+1, dict1, dict2, Not0)) return true;
            dict1[i] = '0'; dict2[c-'A'] = 0;
        }
        return false;
    }
    
    private int getValue(String word, int[] dict) {
        int res = 0;
        for (char c: word.toCharArray())
            res = res*10 + dict[c-'A'];
        return res;
    }
}
