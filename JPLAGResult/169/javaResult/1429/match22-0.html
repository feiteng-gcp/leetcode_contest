class Solution {
    public boolean isSolvable(String[] words, String result) {
        Set<Character> charSet = new HashSet<>();
        boolean[] numUsed = new boolean[10];
        boolean[] solvable = new boolean[1];
        Set<Character> nonZero = new HashSet<>();
        for (String s : words) {
            nonZero.add(s.charAt(0));
            for (char c : s.toCharArray()) {
                charSet.add(c);
            }
        }
        for (char c : result.toCharArray()) {
            nonZero.add(result.charAt(0));
            charSet.add(c);
        }
        
        int[] map = new int[26];
        char[] charArr = new char[charSet.size()];
        int idx = 0;
        for (char c : charSet) {
            charArr[idx ++] = c;
        }
        
        helper(words, result, charArr, 0, numUsed, solvable, map, nonZero);
        
        return solvable[0];
    }
    
    private void helper(String[] words, String result, char[] charArr, int charIdx, 
                        boolean[] numUsed, boolean[] solvable, int[] map,
                        Set<Character> nonZero) {
        //base case:
        if (solvable[0]) {
            return;
        }
        
        if (charIdx == charArr.length) {
            solve(words, result, map, solvable);
            return;
        }
        
        for (int i = 0; i <= 9; i ++) {
            if (numUsed[i] ||
                i == 0 && nonZero.contains(charArr[charIdx])) {
                continue;
            }
            
            
            numUsed[i] = true;
            map[charArr[charIdx] - 'A'] = i;
            helper(words, result, charArr, charIdx + 1, numUsed, solvable, map, nonZero);
            
            numUsed[i] = false;
            
            if (solvable[0]) {
                return;
            }
        }
        
        return;
    }
    
    private void solve(String[] words, String result, int[] map, boolean[] solvable) {
        int sum = 0;
        for (String s : words) {
            sum += transform(s, map);
        }
        
        solvable[0] |= (sum == transform(result, map));
    }
    
    private int transform(String s, int[] map) {
        int sum = 0;
        for (char c : s.toCharArray()) {
            int digit = map[c - 'A'];
            sum = sum * 10 + digit;
        }
        
        return sum;
    }
}
