class Solution {
    boolean[] isLeading = new boolean[127];
	int[] wTod = new int[127];
	boolean[] digit = new boolean[10];
	List<Integer> l = new ArrayList<Integer>();
	String[] words;
	String result;
	boolean isEqual() {

		int left = 0;
		for (String s : words) {
			int n = 0;
        	for (int i = 0; i < s.length(); i++) {
        		n = n * 10 + wTod[s.charAt(i)];
        	}

        	left += n;
        }
		int right = 0;
		for (int i = 0; i < result.length(); i++) {
			right = right * 10 + wTod[result.charAt(i)];
    	}
		return left == right;
	}
	boolean foo(int idx
) {
		if (idx == l.size()) {			
			return isEqual();
		}
		int c = l.get(idx);
		int i = isLeading[c] ? 1 : 0;
		for (; i < 10; i++) {

			if (digit[i]) {
				continue;
			}
			digit[i] = true;
			wTod[c] = i;
			if (foo(idx + 1)) {
				return true;
			}
			digit[i] = false;
		}
		return false;
	}
	public boolean isSolvable(String[] words, String result) {
		this.words = words;
		this.result = result;
		Set<Integer> set = new HashSet<Integer>();
        for (String s : words) {
        	if (s.length() != 1) {
        		isLeading[s.charAt(0)] = true;
        	}
        	for (int i = 0; i < s.length(); i++) {
        		set.add((int)s.charAt(i));
        	}
        }
        for (int i = 0; i < result.length(); i++) {
    		set.add((int)result.charAt(i));
    	}
        for (Integer n : set) {
        	l.add(n);
        }    
        return foo(0);
    }
}
