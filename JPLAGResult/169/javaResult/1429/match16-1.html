class Solution {
    public boolean isSolvable(String[] words, String result) {
        
        HashSet<Character> set = new HashSet<>();
        StringBuilder sb = new StringBuilder();
        Set<Character> startChars = new HashSet<>();
        int[] map = new int[26];
        
        
for (String word: words) {
            for (char c: word.toCharArray()) {
                if (!set.contains(c)) {
                    sb.append(c);
                    set.add(c);
                }
            }
            startChars.add(word.charAt(0));
        
}
        for (char c: result.toCharArray()) {
            if (!set.contains(c)) {
                sb.append(c);
                set.add(c);
            }
        }
        startChars.add(result.charAt(0));
        return dfs(sb.toString(), 0, new boolean[10], map, words, result, startChars);
    
}
    
    private boolean dfs(
        String allChars, int cur, boolean[] numbers, 
        int[] map, String[] words, String result, Set<Character> startChars) {
        if (cur == allChars.length()) {
            if (checkIsEqual(words, result, map)) {
                return true;
            }
            
return false;
        }
        char c = allChars.charAt(cur);
        for (int i = 0; i < 10; i ++) {
            if (i == 0 && startChars.contains(c)) {
                continue;
            }
            if (numbers[i]) continue;
            numbers[i] = true;
            map[c - 'A'] = i;
            if (dfs(allChars, cur + 1, numbers, map, words, result, startChars)) {
                return true;
            }
            map[c - 'A'] = 0;
            numbers[i] = false;
        }
        return false;
    
}
    
    private boolean checkIsEqual(String[] words, String result, int[] map) {
        int carry = 0;
        for (int i = 0; i < result.length(); i ++) {
            for (String word: words) {
                if (i <= word.length() - 1) {
                    carry += map[word.charAt(word.length() - 1 - i) - 'A'];
                }
            }
            if (carry % 10 == map[result.charAt(result.length() - 1 - i) - 'A']) {
                carry /= 10;
            } else {
                return false;
            }
        }
        return carry == 0;

        //     int number = wordToInt(word, map);
        //     if (number == -1) return false;
        //     right -= number;
        //     if (right < 0) return false;
        // }
        // return right == 0;
    
}
    
    private int wordToInt(String word, Map<Character, Integer> map) {
        int number = 0;
        for (int i = 0; i < word.length(); i ++) {
            number = number * 10;
            number += map.get(word.charAt(i));
        }
        return number;
    }
}
