class Solution {
    public boolean solveEquation (String[] words, String result, int[] D) {
        //for (int i = 0; i < 256; i++) if (D[i] != 0) System.out.print((char)i + ": " + D[i]);
        int left = 0, right = 0;
        for (String word: words) {
            int cur = 0;
            for (Character c: word.toCharArray()) {
                cur = cur * 10 + D[c];
            }
            left += cur;
        }
                   
        right = 0;
        for (Character c: result.toCharArray()) {
            right = right * 10 + D[c];
        }
        
        //System.out.println("left " +left + " right: " + right);

        return (left == right);
    }
    
    public boolean dfs (List<Character> li, int idx, int[] D, String[] words, String result, boolean[] taken, Set<Character> lead ) {
        if (idx == li.size()) {
            return solveEquation(words,result, D);
        }
        
        for (int i = 0; i < 10; i++) {
            if (taken[i]) continue;
            if ((i == 0) && lead.contains(li.get(idx))) continue;
            taken[i] = true;
            D[li.get(idx)] = i;
            if (dfs(li, idx+1, D, words, result, taken, lead)) return true;
            taken[i] = false;
        }
        
        return false;
    }
        
    public boolean isSolvable(String[] words, String result) {
        Set<Character> set = new HashSet<>();
        Set<Character> lead = new HashSet<>();

        boolean[] taken = new boolean[10];
        for (String word: words) {
            for (Character c: word.toCharArray()) set.add(c);
            lead.add(word.charAt(0));
        }
        for (Character c: result.toCharArray()) set.add(c);
        lead.add(result.charAt(0));

        ArrayList<Character> al = new ArrayList<>(set);
        System.out.println(set);
        System.out.println(al);
        return dfs(new ArrayList<Character>(set), 0, new int[256], words, result, taken, lead);
    }
}
