class Solution {
    public boolean isSolvable(String[] words, String result) {
        int[] map = new int[128];
        Arrays.fill(map, -1);
        String[] ws = new String[words.length+1];
        for(int i = 0; i < words.length; i++)
            ws[i+1] = words[i];
        ws[0] = result;
        boolean[] used = new boolean[10];
        return dfs(ws, map, 0, 0, used);
    }
    private boolean dfs(String[] words, int[] map, int pos, int offset, boolean[] used) {
        if(pos == words.length) {
            int sum = 0;
            for(int i = 1; i < words.length; i++) {
                sum += getNum(words[i], map);
            }
            return sum == getNum(words[0], map);
        }
        if(offset == words[pos].length()) {
            return dfs(words, map, pos+1, 0, used);
        }
        
        char c= words[pos].charAt(offset);
        if(map[c] >= 0)
            return dfs(words, map, pos, offset+1, used);
        
        for(int i = 0; i <= 9; i++) {
            if(used[i] || (offset == 0 && i == 0))
                continue;
            map[c] = i;
            used[i] = true;
            if(dfs(words, map, pos, offset+1, used))
                return true;
            map[c] = -1;
            used[i] = false;
        }
        return false;
    }
    private int getNum(String w, int[] map) {
        int ret = 0;
        for(int i = 0; i < w.length(); i++) {
            char c = w.charAt(i);
            ret = ret * 10 + map[c];
        }
        return ret;
    }
}
