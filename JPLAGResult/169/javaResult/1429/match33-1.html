class Solution {
    private boolean[] use;
    // private int[][] dp;
    private int[] map;
    
    public boolean isSolvable(String[] words, String result) {
        use = new boolean[10];
        int n = words.length;
        int[] freq = new int[26];
        map = new int[26];
        int count = 0;
        for (int i = 0;i < n;i++) {
            String word = words[i];
            for (int j = 0;j < word.length();j++) {
                char c = word.charAt(j);
                freq[c - 'A']++;
            }
        }
        for (char c : result.toCharArray()) {
            freq[c - 'A']++;
        }
        for (int i = 0;i < 26;i++) {
            if (freq[i] > 0) count++;
        }
        if (count > 10) return false;
        
        // dp = new int[count][10];
        Node[] nodeArr = new Node[count];
        for (int i = 0;i < count;i++) {
            nodeArr[i] = new Node();
        }
        int j = 0;
        for (int i = 0;i < 26;i++) {
            if (freq[i] > 0) {
                char ch = 'A';
                ch += i;
                nodeArr[j].c = ch;
                map[i] = j;
                j++;
            }
        }
        
        
return perm(count, nodeArr, 0, words, result);
    }
    
    boolean perm(int count, Node[] nodeArr, int n, String[] words, String result) {
        // System.out.println(n);
        // int val = nodeArr[n].v;
        // if (dp[n][val] != 0) return (dp[n][val] == 1) ? true : false;
        if (n == count - 1) {
            for (int i = 0;i < 10;i++) {
                if (use[i]) continue;
                nodeArr[n].v = i;
                if (check(nodeArr, count, words, result)) {
                    // dp[n][val] = 1;
                    return true;
                }
            }
            // dp[n][val] = -1;
            return false;
        
}
        for (int i = 0;i < 10;i++) {
            if (use[i]) continue;
            nodeArr[n].v = i;
            use[i] = true;
            if (perm(count, nodeArr, n + 1, words, result)) {
                // dp[n][val] = 1;
                return true;
            }
            
use[i] = false;
        }
        // dp[n][val] = -1;
        return false;
    }
    
    private boolean check(Node[] nodeArr, int count, String[] words, String result) {
        int res = convert(result, count, nodeArr);
        int sum = 0;
        for (String w : words) {
            sum += convert(w, count, nodeArr);
        }
        return sum == res;
    }
    
    private int convert(String s, int count, Node[] nodeArr) {
        
int val = 0, m = 1, j = 0;
        for (int i = s.length() - 1;i >= 0;i--) {
            char c = s.charAt(i);
            // for (j = 0;j < count;j++) {
            //     if (nodeArr[j].c == c) break;
            // }
            j = map[c - 'A'];
            val += m * nodeArr[j].v;
            m *= 10;
        }
        return val;
    }
}

class Node {
    char c;
    int v;
    
    public Node(){}
    
    public Node(char c, int v) {
        this.c = c;
        this.v = v;
    }
}
