class Solution {
    
    private boolean check(String[] words, String result, int[] map) {
        // System.out.println(Arrays.toString(map));
        // Integer res = Integer.parseInt(result);
        int sum = 0;
        for (String word : words) {
            int num = 0;
            for (char c : word.toCharArray()) {
                num = 10 * num + map[c - 'A'];
            }
            sum += num;
        }
        
        int res = 0;
        for (char c : result.toCharArray()) {
            res = 10 * res + map[c - 'A'];
        }
        
        return res == sum;
    }
    
    private void add(int[] map, String word) {
        for (char c : word.toCharArray()) {
            map[c - 'A'] = -1;
        }
    }
    
    public boolean isSolvable(String[] words, String result) {
        int[] map = new int[26];
        boolean[] first = new boolean[26];
        
        Arrays.fill(map, Integer.MIN_VALUE);
        for (String word : words) {
            char f = word.charAt(0);
            first[f - 'A'] = true;
            add(map, word);
        }
        add(map, result);
        first[result.charAt(0) - 'A'] = true;
        
        List<Integer> indexes = new ArrayList<>();
        for (int i = 0; i < 26; i++) {
            if (map[i] == Integer.MIN_VALUE) {
                continue;
            }
            indexes.add(i);
        }
        // System.out.println(Arrays.toString(map));
        boolean[] used = new boolean[10];
        
        return solve(words, result, map, used, 0, indexes, first);
        
    }
    
    private boolean solve(String[] words, String result, int[] map, boolean[] used, int start, List<Integer> indexes, boolean[] first) {
        if (start == indexes.size()) {
            return check(words, result, map);
        }
        
        int index = indexes.get(start);
        for (int i = first[index] ? 1 : 0; i < 10; i++) {
            if (used[i]) {
                continue;
            }
            
            used[i] = true;
            map[index] = i;
            if (solve(words, result, map, used, start + 1, indexes, first)) {
                return true;
            }
            // map[index] = -1;
            used[i] = false;
        }
        
        return false;
    }
}
