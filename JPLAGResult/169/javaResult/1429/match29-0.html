class Solution {
    String[] words;
    String result;
    Map<Character, Mapp> map = new HashMap();
    Mapp[] map2 = new Mapp[26];
    boolean r = false;
    
    public boolean isSolvable(String[] words, String result) {
        this.words = words;
        this.result = result;
        Set<Character> set = new HashSet<>();
        for(String w: words){
            for(int i = 0; i<w.length(); i++){
                set.add(w.charAt(i));
            }
        }
        
        boolean[] visited = new boolean[10];
        for(int i = 0; i<result.length(); i++){
             set.add(result.charAt(i));
        }
        List<Character> chars = new ArrayList<>();
        List<Mapp> l = new ArrayList<>();
        for(Character c: set){
            Mapp m = new Mapp();
            m.c = c;
            l.add(m);
            map2[c-'A'] = m;
        }
        solve(0, l, visited);
        return r;
    }
    
    private void solve(int idx, List<Mapp> list, boolean[] visited){
        if(r) return;
        if(idx == list.size()){
            r = checkSolution(list);
            return;
        }
        
        Mapp m = list.get(idx);
        for(char i = 0; i<=9; i++){
            if(visited[i]) continue;
            m.i = i;
            visited[i] = true;
            solve(idx + 1, list, visited);
            visited[i] = false;
        }
    }
    
    
    boolean checkSolution(List<Mapp> list){
        int l = 0;
        for(String s: words){
            l+=convert(s);
        }
        boolean res = l == convert(result);
        if(res){
            System.out.println(l);
            
        }
        return res;
        
    }
    
    static int[] digits = new int[]{
        1,
        10,
        100,
        1000,
        10000,
        100000,
        1000000,
        10000000,
        100000000
    };
    private int convert(String s){
        int a = 0;
        int p = 0;
        for(int i = s.length() - 1; i>=0; i--){
            a+= map2[s.charAt(i)-'A'].i * digits[p++];
        }
        return a;
    }
    
    static class Mapp{
        char c;
        int i = -1;
    }
    
    

    
    
    
}
