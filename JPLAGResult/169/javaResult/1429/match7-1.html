class Solution {
    public boolean isSolvable(String[] words, String result) {
        Set<Character> dict = new HashSet<>();
        for (String word : words) {
            for (int i = 0; i < word.length(); ++i) {
                dict.add(
word.charAt(i));
            }
        }
        for (int i = 0; i < result.length(); ++i) {
            dict.add(result.charAt(i));
        }

        List<Character> letters = new ArrayList<>(dict);
        
return dfs(words, result, new int[128], letters, 0, new boolean[10]);
    }

    private boolean dfs(String[] words, String result, int[] dict, List<Character> letters, int idx, boolean[] used) {
        if (idx == letters.size()) {
            if (isValid(words, result, dict)) {
                return true;
            }
            return false;
        }

        char letter = letters.get(idx);
        
for (int i = 0; i < 10; ++i) {
            if (!used[i]) {
                used[i] = true;
                dict[letter] = i;
                if (dfs(words, result, dict, letters, idx + 1, used)) {
                    used[i] = false;
                    
return true;
                }
                used[i] = false;
            }
        }

        return false;
    }

    private boolean isValid(String[] words, String result, int[] dict) {
        int sum = 0;
        for (String word : words) {
            int num = toNum(word, dict);
            if (num < 0) {
                return false;
            }
            sum += num;
        }

        int resNum = toNum(result, dict);
        if (resNum < 0) {
            return false;
        }

        return sum == resNum;
    }

    private int toNum(String word, int[] dict) {
        if (word.length() > 1 && dict[word.charAt(0)]== 0) {
            return -1;
        }

        int num = 0;
        for (int i = 0; i < word.length(); ++i) {
            num *= 10;
            num += dict[word.charAt(i)];
        }

        return num;
    }
}
