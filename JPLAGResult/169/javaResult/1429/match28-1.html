class Solution {
    public boolean isSolvable(String[] words, String result) {
        Set<Character> notZero = new HashSet<>();
        Set<Character> allC = new HashSet<>();
        for (String s: words) {
            char[] cs = s.toCharArray();
            for (int i = 0; i < cs.length; i++) {
                if (i == 0) {
                    notZero.add(cs[i]);
                }
                allC.add(cs[i]);
            }
        }
        char[] cs = result.toCharArray();
        for (int i = 0; i < cs.length; i++) {
            if (i == 0) {
                notZero.add(cs[i]);
            }
            allC.add(cs[i]);
        }
        boolean[] used = new boolean[10];
        
int[] tr = new int[256];
        char[] chars = new char[allC.size()];
        int index = 0;
        for (char c: allC) {
            chars[index++] = c;
        }
        return tr(used, tr, chars, notZero, 0, words, result);
    }

    private boolean tr(
boolean[] used, int[] tr, char[] chars, Set<Character> notZero, int index, String[] words, String result) {
        if (index == chars.length) {
            return judge(words, result, tr);
        } else {
            for (int i = 0; i <= 9; i++) {
                if (!used[i]) 
{
                    if (i == 0 && notZero.contains(chars[index])) {
                        continue;
                    }
                    used[i] = true;
                    tr[chars[index]] = i;
                    if (tr(used, tr, chars, notZero, index+1, words, result)) {
                        return true;
                    }
                    used[i] = false;
                }
            }
        }
        return false;
    }
    private boolean judge(
String[] words, String result, int[] tr) {
        int res = 0;
        for (String s: words) {
            res += num(s.toCharArray(), tr);
        }
        res -= num(result.toCharArray(), tr);
        return res == 0;
    
}

    private int num(char[] cs, int[] tr) {
        int res = 0;
        for (char c: cs) {
            res *= 10;
            res += tr[c];
        }
        return res;
    }
}
