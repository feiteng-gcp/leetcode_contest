/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> getAllElements(TreeNode root1, TreeNode root2) {
        List<Integer> nums = new ArrayList<>();
        addNum(root1, nums);
        addNum(root2, nums);
        Collections.sort(nums);
        return nums;
    }
    
    private void addNum(TreeNode root, List<Integer> nums) {
        if (root == null) return;
        
        nums.add(root.val);
        addNum(root.left, nums);
        addNum(root.right, nums);
    }
}
