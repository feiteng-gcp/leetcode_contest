/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    List<Integer> res;
    public List<Integer> getAllElements(TreeNode root1, TreeNode root2) {
        res = new ArrayList<>();
        t(root1);
        t(root2);
        Collections.sort(res);
        return res;
    }
    
    void t(TreeNode r) {
        if (r == null) return;
        res.add(r.val);
        t(r.left);
        t(r.right);
    }
}
