/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> getAllElements(TreeNode root1, TreeNode root2) {
        List<Integer> results = new ArrayList<>();
        getNode(root1, results);
        getNode(root2, results);
        
        Collections.sort(results);
        
        return results;
    }
    
    private void getNode(TreeNode root, List<Integer> results) {
        if (root == null) {
            return;
        }

        results.add(root.val);

        getNode(root.left, results);
        getNode(root.right, results);
    }
}
