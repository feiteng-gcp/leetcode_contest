/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    List<Integer> ans;
    
    public List<Integer> getAllElements(TreeNode root1, TreeNode root2) {
        ans = new ArrayList<>();
        helper(root1);
        helper(root2);
        Collections.sort(ans);
        return ans;
    }
    
    private void helper(TreeNode root) {
        if (root == null) return;
        
        ans.add(root.val);
        helper(root.left);
        helper(root.right);
    }
}
