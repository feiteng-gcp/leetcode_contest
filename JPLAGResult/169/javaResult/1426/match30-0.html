import java.util.Arrays;

class Solution {
    public int[] sumZero(int n) {
        int[] ret = new int[n];
        int sum = 0;
        for (int i = 0; i < n -1; ++i) {
            ret[i] = i + 1;
            sum += ret[i];
        }
        ret[n - 1] = -sum;
        return ret;
    }
}
