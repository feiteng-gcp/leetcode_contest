class Solution {
    public int[] sumZero(int n) {
        int[] result = new int[n];
        int i = 0;
        if (n % 2 == 1) {
            result[i] = 0;
            i++;
        }
        int num = 1;
        for (; i < n; i += 2) {
            result[i] = num;
            result[i+1] = -num;
            num++;
        }
        return result;
    }
}
