class Solution {
    public boolean canReach(int[] arr, int start) {
        int n = arr.length;
        Queue<Integer> q = new LinkedList<>();
        q.offer(start);
        HashSet<Integer> visited = new HashSet<>();
        while (!q.isEmpty()) {
            int i = q.poll();
            if (arr[i] == 0) return true;
            int next = i + arr[i];
            if (next >= 0 && next < n && !visited.contains(next)) {
                visited.add(next);
                q.offer(next);
            }
            next = i - arr[i];
            if (next >= 0 && next < n && !visited.contains(next)) {
                visited.add(next);
                q.offer(next);
            }
        }
        return false;
    }
}
