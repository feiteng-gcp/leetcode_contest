class Solution {
    public boolean canReach(int[] arr, int st) {
        
        int q[] = new int[100000];
        int s = 0;int e = 0;
        
        boolean vis[] = new boolean[arr.length];
        
        q[e++] = st;
        vis[st] = true;
        
        while(s<e){
            
            int c = q[s++];
            if(arr[c]==0) return true;
            
            if(c + arr[c]<arr.length && !vis[c+arr[c]]){
                vis[c+arr[c]] = true;
                q[e++] = c+arr[c];
            }
            if(c - arr[c]>=0 && !vis[c-arr[c]]){
                vis[c-arr[c]] = true;
                q[e++] = c-arr[c];
            }
            
        }
        return false;
        
    }
}
