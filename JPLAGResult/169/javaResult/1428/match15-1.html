class Solution {
    public boolean canReach(int[] arr, int start) {
        int n = arr.length;
        
Queue<Integer> q = new LinkedList<>();
        Set<Integer> visited = new HashSet<>(n);
        q.offer(start);
        visited.add(start);
        while (!q.isEmpty()) {
            int cur = q.poll();
            if (arr[cur] == 0)
                return true
;
            int left = cur - arr[cur];
            int right = cur + arr[cur];
            if (left >= 0 && !visited.contains(left)) {
                q.offer(left);
                visited.add(left);
            }
            if (right < n && !visited.contains(right)) {
                q.offer(right);
                visited.add(right);
            }
        }
        return false;
    }
}
