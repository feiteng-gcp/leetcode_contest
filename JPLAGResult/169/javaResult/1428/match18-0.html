class Solution {
    public boolean canReach(int[] arr, int start) {
        int n=arr.length;
        boolean[] flag=new boolean[n];
        Queue<Integer>q=new LinkedList<>();
        q.add(start);
        flag[start]=true;
        while(!q.isEmpty()) {
        	int index=q.poll();
        	//System.out.println(index);
        	if(arr[index]==0)
        		return true;
        	int next=index+arr[index];
        	if(next<n && flag[next]==false) {
        		q.add(next);
        		flag[next]=true;
        	}
        	next=index-arr[index];
        	if(next>=0 && flag[next]==false) {
        		q.add(next);
        		flag[next]=true;
        	}
        }
        return false;
    }
}
