class Solution {
    public boolean canReach(int[] arr, int start) {
        Queue<Integer> queue = new LinkedList<>();
        Set<Integer> visited = new HashSet<>();
        queue.offer(start);
        visited.add(start);
        
        while (!queue.isEmpty()) {
            int tempIndex = queue.poll();
            if (arr[tempIndex] == 0) {
                return true;
            }
            
            if (tempIndex + arr[tempIndex] < arr.length && !visited.contains(tempIndex + arr[tempIndex])) {
                visited.add(tempIndex + arr[tempIndex]);
                queue.offer(tempIndex + arr[tempIndex]);
            }
            if (tempIndex - arr[tempIndex] >= 0 && !visited.contains(tempIndex - arr[tempIndex])) {
                visited.add(tempIndex - arr[tempIndex]);
                queue.offer(tempIndex - arr[tempIndex]);
            }
        }
        
        return false;
    }
}
