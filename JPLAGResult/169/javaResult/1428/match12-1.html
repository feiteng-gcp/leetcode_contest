class Solution {
    public boolean canReach(int[] arr, int start) {
        int n = arr.length;
        boolean[] visited = new boolean[n];
        LinkedList<Integer> q = new LinkedList<>();

        visited[start] = true;
        q.add(start);
        
        while (!q.isEmpty()) {
            int pos = q.poll();
            int val = arr[pos];
            
if (val == 0) return true;

            if (pos - val >= 0 && !visited[pos - val]) {
                visited[pos - val] = true;
                q.add(pos - val);
            }
            
            if (pos + val < n && !visited[pos + val]) {
                visited[pos + val] = true;
                q.add(pos + val);
            }
        }
        
        return false;
    }
}
