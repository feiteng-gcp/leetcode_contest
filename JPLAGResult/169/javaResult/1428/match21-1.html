import java.util.Stack;

class Solution {
    public boolean canReach(int[] arr, int start) {
		if (arr[start] == 0) {
			return true;
		} 
        boolean[] visit = new boolean[arr.length];
        Stack<Integer> stack = new Stack<Integer>();
        stack.push(start);
        visit[start] = true;
        
while (!stack.empty()) {
        	int n = stack.pop();
        	int idx = n - arr[n];
        	if (idx >= 0 && idx < arr.length && !visit[idx]) {
        		if (arr[idx] == 0) {
        			return true;
        		}
                visit[idx] = true;
        		stack.push(idx);
        	}
        	idx = n + arr[n];
        	if (idx >= 0 && idx < arr.length && !visit[idx]) {
        		if (arr[idx] == 0) {
        			return true;
        		}
                visit[idx] = true;
        		stack.push(idx);
        	}
        }
        return false;
    }
}
