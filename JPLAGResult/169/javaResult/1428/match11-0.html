class Solution {
    public boolean canReach(int[] arr, int start) {
        int n = arr.length;
        boolean[] cr = new boolean[n];
        ArrayDeque<Integer> q = new ArrayDeque<>();
        q.add(start);
        cr[start] = true;
        while (!q.isEmpty()) {
            int u = q.removeFirst();
            if (arr[u]==0)
                return true;
            int v1 = u-arr[u];
            int v2 = u+arr[u];
            if (v1>=0&&!cr[v1]) {
                cr[v1] = true;
                q.add(v1);
            }
            if (v2<n&&!cr[v2]) {
                cr[v2] = true;
                q.add(v2);
            }
        }
        return false;
    }
}
