class Solution {
    public boolean canReach(int[] arr, int start) {
      Set<Integer> set = new HashSet<>();
      Queue<Integer> q = new LinkedList<>();
      q.offer(start);
      set.add(start);
      while (!q.isEmpty()) {
        int curr = q.poll();
        if (arr[curr] == 0) {
          return true;
        }
        int left = curr - arr[curr];
        int right = curr + arr[curr];
        
if (left >= 0 && !set.contains(left)) {
          set.add(left);
          q.offer(left);
        }
        if (right < arr.length && !set.contains(right)) {
          set.add(right);
          q.offer(right);
        }
      }
      return false;
    }
}
