class Solution {
    public boolean canReach(int[] arr, int start) {
        int n = arr.length;
        boolean[] visited = new boolean[n];
        Queue<Integer> q = new LinkedList<>();
        q.offer(start);
        while (!q.isEmpty()) {
            int idx = q.poll();
            if (arr[idx] == 0) return true;
            visited[idx] = true;
            if (idx + arr[idx] >= 0 && idx + arr[idx] < n && !visited[idx + arr[idx]]) {
                q.offer(idx + arr[idx]);
            }
            if (idx - arr[idx] >= 0 && idx - arr[idx] < n && !visited[idx - arr[idx]]) {
                q.offer(idx - arr[idx]);
            }
        }
        return false;
    }
}
