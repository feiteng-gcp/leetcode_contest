class Solution {
     int[] dp;

    public boolean canReach(int[] a, int s) {
        dp = new int[a.length + 1];
        return doc(a, s);
    }

    private boolean doc(int[] a, int i) {
        if (i >= a.length || i < 0) {
            return false;
        }
        if (a[i] == 0) {
            return true;
        }
        if (dp[i] != 0) {
            return dp[i] == 1;
        }
        dp[i] = 2;
        boolean rt = doc(a, i + a[i]) || doc(a, i - a[i]);
        dp[i] = rt ? 1 : 2;
        return rt;
    }
}
