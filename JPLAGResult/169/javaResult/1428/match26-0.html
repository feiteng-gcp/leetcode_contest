class Solution {
    public boolean canReach(int[] arr, int start) {
        LinkedList<Integer> queue = new LinkedList<Integer>();
        int n = arr.length;
        boolean[] visits = new boolean[n];
        queue.add(start);
        visits[start] = true;
        
        while (!queue.isEmpty()) {
            int b = queue.removeFirst();
            if (arr[b] == 0) return true;
            
            if (b + arr[b] < n && !visits[b + arr[b]] ) {
                queue.add(b + arr[b]);
                visits[b + arr[b]] = true;
            }
            if (b - arr[b] >= 0 && !visits[b - arr[b]] ) {
                queue.add(b - arr[b]);
                visits[b - arr[b]] = true;
            }
        }
        return false;
    }
}
