class Solution {
    public boolean canReach(int[] arr, int start) {
        int n = arr.length;
        boolean[] seen = new boolean[n];
        Deque<Integer> queue = new ArrayDeque<>();
        queue.add(start);
        while(!queue.isEmpty()){
            int cur = queue.poll();
            //System.out.println(cur);
            if(arr[cur] == 0){
                return true;
            }
            if(seen[cur]){
                continue; 
            } else {
                seen[cur] = true;
            }
            if(cur + arr[cur] < n){
                queue.add(cur + arr[cur]);
            }
            if(cur - arr[cur] >= 0){
                queue.add(cur - arr[cur]);
            }
        }
        return false;
    }
}
