class Solution {
    public boolean canReach(int[] arr, int start) {
        // time & space: O(N)
        
        int n = arr.length;
        boolean[] visited = new boolean[n];
        Queue<Integer> qe = new LinkedList<Integer>();
        qe.offer(start);
        visited[start] = true;
        while(!qe.isEmpty()) {
            int cur = qe.poll();
            if(arr[cur] == 0) {
                return true;
            
}
            
            int next = cur + arr[cur];
            if(next >= 0 && next < n && !visited[next]) {
                qe.offer(next);
                visited[next] = true;
            }
            
            next = cur - arr[cur];
            if(next >= 0 && next < n && !visited[next]) {
                qe.offer(next);
                visited[next] = true;
            }
        }
        
        return false;
    }
}
