class Solution {

    public boolean canReach(int[] arr, int start) {
    	int n = arr.length;
    	if( arr[start] == 0 ) {
    		return true;
    	}
 
    	boo
lean[] visited = new boolean[n];
    	
    	Queue<Integer> q = new LinkedList<>();
    	visited[start] = true;
    	q.add(start);
    	while( !q.isEmpty() ) {
    		int current = q.remove();
    		int next = current + arr[current];
    		if( next < n && visited[next] == false ) {
    			if( arr[next] == 0 ) {
    				return true;
    			}
    			visited[next] = true;
    			q.add(next);
    		}
    		next = current - arr[current];
    		if( next >= 0 && visited[next] == false ) {
    			if( arr[next] == 0 ) {
    				return true;
    			}
    			visited[next] = true;
    			q.add(next);
    		}
    	}
    	return false;
    }
    
}
