import java.util.*;


class Solution {
    public boolean canReach(int[] arr, int start) {
        int n = arr.length;
        boolean[] visited = new boolean[n];
        Queue<Integer> queue = new LinkedList<>();
        
queue.offer(start);
        while(!queue.isEmpty()) {
            int cur = queue.poll();
            if (arr[cur] == 0) return true;
            visited[cur] = true;
            int r = cur + arr[cur];
            if (r >= 0 && r < n && !visited[r]) {
                visited[r] = true;
                queue.offer(r);
            }
            int l = cur - arr[cur];
            
if (l >= 0 && l < n && !visited[l]) {
                visited[l] = true;
                queue.offer(l);
            }
        }
        return false;
    }
}
