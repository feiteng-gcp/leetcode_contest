import java.util.*;

class Solution {
    public boolean canReach(int[] arr, int start) {

        
Queue<Integer> que = new ArrayDeque<>();
        que.add(start);
        Set<Integer> visited = new HashSet<>();
        while(!que.isEmpty()) {
            int at = que.poll();
            if (arr[at] == 0) { return true; 
}
            int dest1 = at + arr[at];
            if (dest1 < arr.length && !visited.contains(dest1)) {
                que.add(dest1);
                visited.add(dest1);
            }
            int dest2 = at - arr[at];
            
if (dest2 >= 0 && !visited.contains(dest2)) {
                que.add(dest2);
                visited.add(dest2);
            }
        }
        return false;
    }
}
