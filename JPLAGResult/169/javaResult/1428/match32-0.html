class Solution {
    public boolean canReach(int[] arr, int start) {
        int length = arr.length, left, right;
        boolean[] visited = new boolean[length];
        Queue<Integer> queue = new LinkedList<>();
        queue.add(start);
        visited[start] = true;
        
        while (!queue.isEmpty()) {
            int index = queue.poll();
            if (arr[index] == 0) {
                return true;
            }
            left = index - arr[index];
            right = index + arr[index];
            
            if (left >= 0 && left < length && !visited[left]) {
                queue.add(left);
                visited[left] = true;
            }
            
            if (right >= 0 && right < length && !visited[right]) {
                queue.add(right);
                visited[right] = true;
            }
        }
        
        return false;
    }
}
