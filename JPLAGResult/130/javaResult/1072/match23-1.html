/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] nextLargerNodes(ListNode head) {
        head = reverse(head);
        LinkedList<Integer> res = new LinkedList<>();
        ListNode p = head;
        Stack<Integer> stack = new Stack<>();
        while(p != null)
{
            while(!stack.isEmpty() && p.val>=stack.peek()){
                stack.pop();
            }
            if(stack.isEmpty()) res.addFirst(0);
            else res.addFirst(stack.peek());
            stack.push(p.val);
            p = p.next;
        }
        int[] arr = new int[res.size()];
        int k = 0;
        for(int a : res){
            
arr[k++] = a;
        }
        return arr;
    }
    
    private ListNode reverse(ListNode head){
        
ListNode pre = new ListNode(-1), p = head;
        while(p != null){
            ListNode tmp = pre.next;
            pre.next = p;
            p = p.next;
            pre.next.next = tmp;
        }
        return pre.next;
    }
}
