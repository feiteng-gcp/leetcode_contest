/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] nextLargerNodes(ListNode head) {
        List<Integer> list = new ArrayList<>();
        ListNode cur = head;
        while(cur!= null) {           
            list.add(cur.val);
            cur = cur.next;
        }
        int n = list.size();
        int[] res = new int[n];
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (list.get(i) < list.get(j)) {
                    res[i] = list.get(j);
                    break;
                }
            }
        }
        return res;
    }
}
