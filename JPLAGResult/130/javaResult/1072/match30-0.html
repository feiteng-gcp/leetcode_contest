/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] nextLargerNodes(ListNode head) {
        List<Integer> list = new ArrayList<>();
        ListNode tmp = head;
        while (true) {
            if (tmp != null) {
                int x = tmp.val;
                list.add(x);
                tmp = tmp.next;
            } else {
                break;

            }
        }
        int len = list.size();
        int[] a = new int[len];
        int c = 0;
        for (int x : list) a[c++] = x;
        int[] r = new int[len];
        for (int i = 0; i < len; i++) {
            for (int j = i + 1; j < len; j++) {
                if (a[j] > a[i]) {
                    r[i] = a[j];
                    break;
                }
            }
        }
        return r;
    }
}
