/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] nextLargerNodes(ListNode head) {
        ArrayList<Integer> rv = new ArrayList<>();
        Stack<Integer> stk = new Stack<>();
        Stack<Integer> index = new Stack<>();
        ListNode p = head;
        int i = 0;
        while(p != null) {
            while (stk.size() > 0 && p.val > stk.peek()) {
                int k = index.pop();
                rv.set(k, p.val);
                stk.pop();
            }
            
            stk.push(p.val);
            index.push(i);
            i++;
            rv.add(0);
            p = p.next;
        }
        
        int[] retval = new int[rv.size()];
        for(int j=0; j<retval.length; j++) {
            retval[j] = rv.get(j);
        }
        return retval;
    }
}
