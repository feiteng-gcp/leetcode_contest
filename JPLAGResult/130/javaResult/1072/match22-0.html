/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] nextLargerNodes(ListNode head) {
        int N = 0;
        ListNode p = head;
        while(p != null) { 
            p = p.next; N ++; 
        }
        int[] ans = new int[N];
        if (N == 0) return ans;;

        Map<Integer, Integer> map = new HashMap();
        Stack<Integer> stack = new Stack<>();
        
        p = head;
        for (int i = 0; i < N; i ++) {
            int v = p.val;
            while(!stack.isEmpty() && v > map.get(stack.peek())) {
                int index = stack.pop();
                ans[index] = v;
                map.remove(index);
            }
            
            map.put(i, v);
            stack.push(i);

            p = p.next;
        }
        return ans;
    }
}
