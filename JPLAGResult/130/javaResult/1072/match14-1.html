/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */

class Solution {
    public int[] nextLargerNodes(ListNode head) {
        ArrayList<Integer> nodes = new ArrayList<Integer>();
        while (head != null) {
            nodes.add(head.val);
            head = head.next;
        }
        int[] ans = new int[nodes.size()];
        int n = nodes.size();
        for (int i = 0; i < n; i++) {
            int u = nodes.get(i);
            for (int j = i+1; j < n; ++j) {
                if (nodes.get(j) > u) {
                    ans[i] = nodes.get(j);
                    
break;
                }
            }
        }
        return ans;
    }
}
