/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] nextLargerNodes(ListNode head) {
        int count = 0;
        ListNode node = head;
        while (node != null) {
            count++;
            node = node.next;
        }
        int[] res = new int[count];
        int[] stack1 = new int[count + 10];
        int[] stack2 = new int[count + 10];
        node = head;
        int p = 0;
        for (int i = 0; i < count; i++) {
            while (p != 0 && stack2[p-1] < node.val) {
                res[stack1[p-1]] = node.val;
                p--;
            }
            stack1[p] = i;
            stack2[p++] = node.val;
            node = node.next;
        }
        return res;
    }
}
