/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] nextLargerNodes(ListNode head) {
        
int n = 1;
        ListNode iter = head;
        while(iter.next != null){
            iter = iter.next;
            n++;
        }
        int[] res = new int[n];
        int[] val = new int[n];
        iter = head;
        int i=0, j=0, max=Integer.MIN_VALUE;
        while(iter != null){
            val[i] = iter.val;
            iter = iter.next;
            i++;
        }
        Stack<Integer> stack = new Stack();
        stack.push(0);
        for(i=1; i<n; i++){
            if(!stack.isEmpty())
{
                while(!stack.isEmpty() && val[i] > val[stack.peek()]){
                    res[stack.pop()] = val[i];
                }
            }
            
stack.push(i);
        }
        while(!stack.isEmpty()) res[stack.pop()] = 0;
        return res;
    }
}
