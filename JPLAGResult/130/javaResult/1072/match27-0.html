/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] nextLargerNodes(ListNode head) {
        ListNode cur = head;
        int N = 0;
        List<ListNode> list = new ArrayList<>();
        while(cur != null) {
            list.add(cur);
            cur = cur.next;
            N++;
        }
        if (N == 0) {
            return new int[]{};
        }
        int[] ans = new int[N];
        Stack<Integer> deque = new Stack<>();
        for (int i = N - 1; i >= 0; i--) {
            int val = list.get(i).val;
            while(!deque.isEmpty() && deque.peek() <= val) {
                deque.pop();
            }
            if (deque.isEmpty()) {
                ans[i] = 0;
            } else {
                ans[i]= deque.peek();
            }
            deque.add(val);
   
        }
        return ans;
    }
}
