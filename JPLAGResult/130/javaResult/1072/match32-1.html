/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */

class Solution {
    public int[] nextLargerNodes(ListNode head) {
        ListNode t = head;
        int n = 0;
        while(t != null)
        {
            t = t.next;
            n++;
        }
        int[] vals = new int[n], idx = new int[n], re = new int[n];
        for(int i = 0; i < n; i++)
        {
            vals[i] = head.val;
            head = head.next;
        }
        // now process
        Stack<Integer> stack = new Stack<>();
        stack.push(vals[n - 1]);
        for(int i = n - 2; i >= 0; i--)
        {
            while(!stack.isEmpty() && vals[i] >= stack.peek()) stack.pop();
            if(stack.isEmpty()) re[i] = 0;
            else re[i] = stack.peek();
            stack.push(vals[i]);
        }
        return re;
    }
}
