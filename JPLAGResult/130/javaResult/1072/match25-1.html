/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] nextLargerNodes(ListNode head) {
        
List<Integer> values = new ArrayList<>();
        
        ListNode cur = head;
        while (cur != null) {
            values.add(cur.val);
            cur = cur.next;
        }
        
        int n = values.size();
        int[] right = new int[n];
        right[n-1] = n-1;
        for (int i = n - 2; i >= 0; i--) {
            right[i] = i;
            int j = i + 1;
            
            while (j < n && values.get(j) <= values.get(i) && j != right[j]) {
                j = right[j];
            }
            
            if (j < n && values.get(j) > values.get(i)) {
                right[i] = j;
            }
        }
        
        
for (int i = 0; i < n; i++) {
            if (right[i] == i) {
                right[i] = 0;
            } else {
                right[i] = values.get(right[i]);
            }
        }
        
        return right;
    }
}
