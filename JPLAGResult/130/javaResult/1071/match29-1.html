class Solution {
    public List<Boolean> prefixesDivBy5(int[] A) {
        List<Boolean> result = new ArrayList<>();
        long cur = 0L;
        for (int a : A) {
            cur = cur * 2 + (a == 1 ? 1 : 0);
            cur = cur % 5;
            result.add(cur % 5 == 0 ? true : false);
        }
        return result;
    }
}
