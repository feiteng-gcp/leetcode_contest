class Solution {
    public List<Boolean> prefixesDivBy5(int[] A) {
        List<Boolean> res = new ArrayList<>();
        int num = 0;
        for (int i = 0; i < A.length; i++) {
            num = num * 2 + A[i];
            res.add(num % 5 == 0);
            num %= 5;
        }
        return res;
    }
}
