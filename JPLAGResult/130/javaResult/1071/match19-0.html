class Solution
{
    public List<Boolean> prefixesDivBy5(int[] a)
    {
        List<Boolean> ans = new ArrayList<>();
        int n = 0;
        for (int b : a)
        {
            n = (n << 1) | b;
            ans.add(n % 5 == 0);
            n %= 1_000_000_000;
        }
        return ans;
    }
}
