class Solution {
    public List<Boolean> prefixesDivBy5(int[] A) {
        List<Boolean> result = new ArrayList<>();
        int n = 0;
        for (int i = 0; i < A.length; i++) {
            n = ((n << 1) + A[i]) % 5;
            result.add(n == 0);
        }
        return result;
    }
}
