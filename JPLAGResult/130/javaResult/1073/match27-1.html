class Solution {
    private int[] dx = new int[] {-1, 1, 0, 0};

	private int[] dy = new int[] {0, 0, -1, 1};

	public int numEnclaves(int[][] A) {


		for (int i = 0
; i < A.length; i++) {
			for (int j = 0; j < A[i].length; j++) {
				if (i == 0 || j == 0 || i == A.length - 1 || j == A[i].length - 1) {
					if (A[i][j] == 1) {
						search(A, i, j);
					}
				}
			}
		}

		int res = 0;

		for (int i = 0; i < A.length; i++) {
			for (int j = 0; j < A[i].length; j++) {
				if (A[i][j] == 1) {
					res++;
				}
			}
		}
		return res;
	}

	private void search(int[][] A, int x, int y) {

		A[x][y] = 0;

		Queue<int[]> queue = new LinkedList<>();
		queue.add(new int[] {x, y});

		boolean[][] visited = new boolean[A.length][A[0].length];
		visited[x][y] = true;
		
		while (!queue.isEmpty()) {
			int[] curr = queue.poll();
			for (int i = 0; i < 4; i++) {
				int cx = curr[0] + dx[i];
				int cy = curr[1] + dy[i];
				if (cx >= 0 && cy >= 0 && cx < A.length && cy < A[0].length && !visited[cx][cy] && A[cx][cy] == 1) {
					visited[cx][cy] = true;
					A[cx][cy] = 0;

					queue.add(new int[]{cx, cy});
				}
			}
		}

	}
}
