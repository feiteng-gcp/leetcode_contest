class Solution
{
    private static int[][] vectors = new int[][]
            {
                    new int[]{0, 1},
                    new int[]{0, -1},
                    new int[]{1, 0},
                    new int[]{-1, 0}
            };

    private int cnt;
    private boolean travel(int[][] a, int i, int j, int m, int n)
    {
        if (i < 0 || i >= m || j < 0 || j >= n) return true;
        if (a[i][j] == 0) return false;
        a[i][j] = 0;
        cnt++;
        boolean ret = false;
        for (int[] v : vectors)
        {
            ret |= travel(a, i + v[0], j + v[1], m, n);
        }
        return ret;
    }

    public int numEnclaves(int[][] a)
    {
        int ans = 0;
        for (int i = 0; i < a.length; i++)
        {
            for (int j = 0; j < a[0].length; j++)
            {
                if (a[i][j] == 1)
                {
                    cnt = 0;
                    if (!travel(a, i, j, a.length, a[0].length))
                        ans += cnt;
                }
            }
        }
        return ans;
    }
}
