class Solution {
    public int numEnclaves(int[][] A) {
        int M = A.length;
        int N = A[0].length;
        for (int i = 0; i < N; i++) {
            helper(A, 0, i);
            helper(A, M - 1, i);
        }
        for (int i = 1; i < M - 1; i++) {
            helper(A, i, 0);
            helper(A, i, N - 1);
        }
        int ans = 0;
        for (int i = 0; i < M; i++) {
            for (int j = 0; j < N; j++) {
                if (A[i][j] == 1) ans ++;
            }
        }
        return ans;
    }
    
    int[] dir = new int[]{-1, 0, 1, 0, -1};
    
    private void helper(int[][] A, int x, int y) {
        if (A[x][y] <= 0) return;
        A[x][y] = -1;
        for (int i = 0; i < 4; i++) {
            int cx = x + dir[i];
            int cy = y + dir[i + 1];
            if (cx < 0 || cx >= A.length || cy < 0 || cy >= A[0].length) continue;
            helper(A, cx, cy);
        }
    }
}
