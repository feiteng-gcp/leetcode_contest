class Solution {
    private int[][] ds = {{-1,0},{1,0},{0,-1},{0,1}};
    private int r, c;
    public int numEnclaves(int[][] A) {
        if (A.length == 0 || A[0].length == 0) return 0;
        r = A.length; c = A[0].length;
        boolean[][] visited = new boolean[r][c];
        
for (int i = 0; i < r; i++) {
            dfs(A, i, 0);
            dfs(A, i, c - 1);
        }
        for (int j = 1; j < c - 1; j++) {
            dfs(A, 0, j);
            dfs(A, r - 1, j);
        }
        int res = 0;
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                if (A[i][j] == 1) res++
;
            }
        }
        return res;
    }
    private void dfs(int[][] A, int i, int j) {
        if (i < 0 || i == r || j < 0 || j == c || A[i][j] == 0) return;
        A[i][j] = 0;
        for (int[] d : ds) {
            dfs(A, i + d[0], j + d[1]);
        }
    }
}
