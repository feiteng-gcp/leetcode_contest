class Solution {
    public int numEnclaves(int[][] A) {
        for(int i=0;i<A.length;i++){
            // j = 0 || j = A[i].length-1
            if(A[i][0]==1){
                traverse(A,i,0);
            }
            if(A[i][A[i].length-1]==1){
                traverse(A,i,A[i].length-1);
            }
        }
        for(int i=0;i<A[0].length;i++){
            // i = 0 || i = A.length-1
            if(A[0][i] == 1){
                traverse(A,0,i);
            }
            if(A[A.length-1][i] == 1){
                traverse(A,A.length-1,i);
            }
        }
        int count = 0;
        for(int i=0;i<A.length;i++){
            for(int j=0;j<A[i].length;j++){
                if(A[i][j] == 1){
                    count++;
                }
            }
        }
        return count;
    }

    private void traverse(int[][] grid, int i, int j){
        if(i<0 || i>=grid.length || j<0 || j>=grid[0].length || grid[i][j] == 0){
            return;
        }
        grid[i][j] = 0;
        traverse(grid,i,j+1);
        traverse(grid,i+1,j);
        traverse(grid,i,j-1);
        traverse(grid,i-1,j);
    }
}
