class Solution {
    
public int numEnclaves(int[][] A) {
        int r = A.length, c = A[0].length;
        for (int i = 0; i < r; i++){
            if (A[i][0] == 1) dfs(A, i, 0);
            if (A[i][c-1] == 1) dfs(A, i, c-1);
        }
        for (int j = 0; j < c; j++){
            if (A[0][j] == 1) dfs(A, 0, j);
            if (A[r-1][j] == 1) dfs(A, r-1, j)
;
        }
        int res = 0;
        for (int i = 0; i < r; i++){
            for (int j = 0; j < c; j++){
                if (A[i][j] == 1) res++;
            }
        }
        return res;
    }
    
    
private int[][] dirs = new int[][]{{-1,0},{1,0},{0,1},{0,-1}};
    public void dfs(int[][] a, int i, int j){
        if (i < 0 || i >= a.length || j < 0 || j >= a[0].length || a[i][j] == 0) return;
        a[i][j] = 0;
        for (int[] d : dirs){
            int r = i+d[0], c = j+d[1];
            dfs(a, r, c);
        }
    }
}
