class Solution {
    private int[][] dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    
    private void dfs(int[][] A, int r, int c) {
        int m = A.length, n = A[0].length;
        if (r < 0 || r >= m || c < 0 || c >= n || A[r][c] != 1) 
            return;
        A[r][c] = -1;
        for (int[] dir: dirs) {
            dfs(A, r + dir[0], c + dir[1]);
        }
        
    }
    
    public int numEnclaves(int[][] A) {
        int m = A.length, n = A[0].length;
        for (int i = 0; i < m; i++) {
            if (A[i][0] == 1) {
                dfs(A, i, 0);
            }
            if (A[i][n-1] == 1) {
                dfs(A, i, n-1);
            }
        }
        for (int i = 0; i < n; i++) {
            if (A[0][i] == 1) {
                dfs(A, 0, i);
            }
            if (A[m-1][i] == 1) {
                dfs(A, m-1, i);
            }
        }
        int res = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (A[i][j] == 1) 
                    res += 1;
            }
        }
        return res;
    }
}
