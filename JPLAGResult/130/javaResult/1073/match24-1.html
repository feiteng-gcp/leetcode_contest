class Solution {
    public int numEnclaves(int[][] A) {
        int m = A.length;
        int n = A[0].length;

        boolean[][] vst = new boolean[m][n];

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (i == 0 || i == m - 1 || j == 0 || j == n - 1) {
                    if (A[i][j] == 1) {
                        dfs(A, i, j, vst);
                    }
                }
            }
        }

        
int ans = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (A[i][j] == 1 && !vst[i][j]) {
                    ans++;
                
}
            }
        }
        return ans;
    }

    void dfs(int[][] A, int x, int y, boolean[][] vst) {
        int m = A.length;
        int n = A[0].length;
        if (!(x >= 0 && x < m && y >= 0 && y < n)) {
            return;
        }
        if (A[x][y] != 1) return;
        if (vst[x][y]) return;

        vst[x][y] = true;
        
int[][] dir = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};


        
for (int i = 0; i < dir.length; i++) {
            int r = x + dir[i][0];
            int c = y + dir[i][1];
            dfs(A, r, c, vst);
        }
    }
}
