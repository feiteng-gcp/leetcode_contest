class Solution {
    public int numEnclaves(int[][] A) {
        int len = A.length;
        int wid = A[0].length;
        for (int i = 0; i < len; i++) {
            dfs(i, 0, A);
            dfs(i, wid - 1, A);
        }
        for (int j = 0; j < wid; j++) {
            dfs(0, j, A);
            dfs(len - 1, j, A);
        }
        int res = 0;
        for (int i = 0; i <len; i++) {
            for (int j = 0; j < wid; j++) {
                res += A[i][j]; 
            
}
        }
        return res;
    }
    public void dfs(int i, int j, int[][] A) {
        if (i < 0 || j < 0 || i >= A.length || j >= A[0].length) {
            return;
        }
        if (A[i][j] == 0) {
            return;
        }
        A[i][j] = 0;
        dfs(i - 1, j, A);
        dfs(i + 1, j, A);
        dfs(i, j - 1, A);
        dfs(i, j + 1, A);
    }
}
