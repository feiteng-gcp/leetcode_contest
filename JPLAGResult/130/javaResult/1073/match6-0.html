class Solution {
    public int numEnclaves(int[][] A) {
        if(A.length == 0) return 0;
        for(int i = 0; i < A[0].length; ++i) {
            if(A[0][i] == 1) {
                helper(A, 0, i);
            }
            if(A[A.length - 1][i] == 1) {
                helper(A, A.length -1, i);
            }
        }
        for(int i = 0; i < A.length; ++i) {
            if(A[i][0] == 1) {
                helper(A, i, 0);
            } 
            if(A[i][A[0].length -1] == 1) {
                helper(A, i, A[0].length -1);
            }
        }
        int res = 0;
        for(int i = 0; i < A.length; ++i) {
            for(int j = 0; j < A[0].length; ++j) {
                if(A[i][j] == 1) {
                    res++;
                }
            }
        }
        return res;
    }
    private void helper(int[][] A, int i, int j) {
        if( i < 0 || i > A.length - 1 || j < 0 || j > A[0].length - 1 || A[i][j] == 0) return;
        A[i][j] = 0;
        helper(A, i + 1, j);
        helper(A, i - 1, j);
        helper(A, i, j +1);
        helper(A, i, j -1);
    }
}
