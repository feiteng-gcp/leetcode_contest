class Solution {
    
    int l = -1;
    int ll = -1;
    
    public void dfs(int i,int j,int[][] A){
        A[i][j] = 0;
        
        int dx[] = {0,0,1,-1};
        int dy[] = {1,-1,0,0};
        
        for(int t=0;t<4;++t){
            int nx = i+dx[t]; int ny = j+dy[t];
            if(nx>=0&&nx<l&&ny>=0&&ny<ll&&A[nx][ny]==1){
                dfs(nx,ny,A);
            }
        }
        
        
    }
    
    public int numEnclaves(int[][] A) {
        
        l = A.length;
        
        ll = A[0].length;
        
        for(int i = 0;i<l;++i){
            for(int j=0;j<ll;++j){
                if(i==0||i==l-1||j==0||j==ll-1){
                    if(A[i][j]==1){
                        dfs(i,j,A);
                    }
                }
                
            }
        }
        int r = 0;
        
        for(int i = 0;i<l;++i){
            for(int j=0;j<ll;++j){
                
                r+= A[i][j];
                
            }
        }
        
        return r;
        
    }
}
