class Solution {
    public int numEnclaves(int[][] A) {
        for (int i = 0; i < A[0].length; i ++) {
            if (A[0][i] == 1) {
                helper(A, 0, i);
            }
        }
        for (int i = 0; i < A[0].length; i ++) {
            if (A[A.length-1][i] == 1) {
                helper(A, A.length-1, i);
            }
        }
        for (int i = 0; i < A.length; i ++) {
            if (A[i][0] == 1) {
                helper(A, i, 0);
            }
        }
        for (int i = 0; i < A.length; i ++) {
            if (A[i][A[0].length-1] == 1) {
                helper(A, i, A[0].length-1);
            }
        }
        int count = 0;
        for (int i = 0; i < A.length; i ++) {
            for (int j = 0; j < A[0].length; j ++) {
                if (A[i][j] == 1) {
                    count ++;
                }
            }
        }
        return count;
    }

    private void helper(int[][] A, int x, int y) {
        A[x][y] = 2;
        if (x > 0 && A[x-1][y] == 1) {
            helper(A, x-1, y);
        }
        if (y < A[0].length - 1 && A[x][y+1] == 1) {
            helper(A, x, y+1);
        }
        if (x < A.length - 1 && A[x+1][y] == 1) {
            helper(A, x+1, y);
        }
        if (y > 0 && A[x][y-1] == 1) {
            helper(A, x, y-1);
        }
    }
}
