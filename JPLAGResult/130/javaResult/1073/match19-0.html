class Solution {
    public int numEnclaves(int[][] A) {
        int n = A.length;
        int m = A[0].length;
        boolean[][] visited = new boolean[n][m];
        for(int i=0; i<n; i++){
            dfs(A, visited, n, m, i, 0);
            dfs(A, visited, n, m, i, m-1);
        }
        for(int j=0; j<m; j++){
            dfs(A, visited, n, m, 0, j);
            dfs(A, visited, n, m, n-1, j);
        }
        int res = 0;
        
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(A[i][j] == 1 && !visited[i][j]){
                    res++;
                }
            }
        }
        return res;

        
    }
    public void dfs(int[][] A, boolean[][] visited, int n, int m, int i, int j){
        if(i<0 || j<0 || i>=n || j>=m || visited[i][j]) return;
        if(A[i][j] == 0) return;
        visited[i][j] = true;
        dfs(A, visited, n, m, i+1, j);
        dfs(A, visited, n, m, i-1, j);
        dfs(A, visited, n, m, i, j-1);
        dfs(A, visited, n, m, i, j+1);
    }
}
