	class Solution {
		int[] dr = { 1, 0, -1, 0 };
		int[] dc = { 0, 1, 0, -1 };
		
	    public int numEnclaves(int[][] A) {
	        int n = A.length, m = A[0].length;
	        Queue<int[]> q = new ArrayDeque<>();
	        boolean[][] ved = new boolean[n][m];
	        for(int i = 0;i < n;i++){
	        	for(int j = 0;j < m;j++){
	        		if(i == 0 || i == n-1 || j == 0 || j == m-1){
        				if(A[i][j] == 1){
        					q.add(new int[]{i, j});
        					ved[i][j] = true;
        				}
	        		}
	        	}
	        }
	        while(!q.isEmpty()){
	        	int[] cur = q.poll();
	        	int r = cur[0], c = cur[1];
	        	for(int k = 0;k < 4;k++){
	        		int nr = r + dr[k], nc = c + dc[k];
	        		if(nr >= 0 && nr < n && nc >= 0 && nc < m && A[nr][nc] == 1 && !ved[nr][nc]){
	        			ved[nr][nc] = true;
	        			q.add(new int[]{nr, nc});
	        		}
	        	}
	        }
	        
	        int ans = 0;
	        for(int i = 0;i < n;i++){
	        	for(int j = 0;j < m;j++){
	        		if(!ved[i][j]){
	        			ans += A[i][j];
	        		}
	        	}
	        }
	        return ans;
	    }
	}
