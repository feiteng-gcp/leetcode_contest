class Solution {
    public int numEnclaves(int[][] A) {
        int yl = A.length, xl = A[0].length;
        for (int i = 0; i < xl; i++) {
            if (A[0][i] == 1) dfs(A, i, 0, xl, yl);
            if (A[yl-1][i] == 1) dfs(A, i, yl-1, xl, yl);
        }
        for (int j = 0; j < yl; j++) {
            if (A[j][0] == 1) dfs(A, 0, j, xl, yl);
            if (A[j][xl-1] == 1) dfs(A, xl-1, j, xl, yl);
        }
        int res = 0;
        for (int j = 0; j < yl; j++) {
            for (int i = 0; i < xl; i++) {
                if (A[j][i] == 1) res++;
            }
        }
        return res;
    }
    static final int[][] dir = new int[][]{{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
    private void dfs(int[][] A, int x, int y, int xl, int yl) {
        A[y][x] = 2;
        for (int[] d : dir) {
            int nx = x + d[0], ny = y + d[1];
            if (nx >= 0 && nx < xl && ny >= 0 && ny < yl && A[ny][nx] == 1) dfs(A, nx, ny, xl, yl);
        }
    }
}
