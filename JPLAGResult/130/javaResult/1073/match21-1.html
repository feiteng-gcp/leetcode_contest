class Solution {
    private final static int[][] DIRS = new int[][]{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    
    public int numEnclaves(int[][] A) {
        if (A.length == 0 || A[0].length == 0) return 0;
        int m = A.length, n = A[0].length;
        boolean[][] visited = new boolean[m][n];
        
int res = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (!visited[i][j] && A[i][j] == 1) {
                    int cur = getBoundary(A, visited, i, j);
                    if (cur >= 0) res += cur
;
                }
            }
        }
        return res;
    }
    
    private int getBoundary(
int[][] grid, boolean[][] visited, int i, int j) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length) return -1; // connect to boundary
        if (grid[i][j] != 1 || visited[i][j]) return 0; // only care value 0, and none visited
        visited[i][j] = true;
        int sum = 1;
        for (int[] dir : DIRS) {
            int dx = i + dir[0], dy = j + dir[1];
            int cur = getBoundary(grid, visited, dx, dy);
            if (cur >= 0 && sum > 0) sum += cur;
            else sum = -1;
        }
        return sum;
    }
}
