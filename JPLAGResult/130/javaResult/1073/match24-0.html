class Solution {
    int[][] offset = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    int count;
    boolean reachEnd;
    int ones = 0;
    int totalReached = 0;
    public int numEnclaves(int[][] grid) {
        int m = grid.length;
        if(m == 0){
            return 0;
        }
        int n = grid[0].length;
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                if(grid[i][j] == 1){
                    reachEnd = false;
                    count = 0;
                    merge(grid, i, j);
                    
                    if(reachEnd){
                        ones -= count;
                    }
                    //System.out.println();
                }
            }
        }
        return ones;
    }
    public void merge(int[][] grid, int i, int j){
        int m = grid.length;
        int n = grid[0].length;
        if(i < 0 || i >= m || j < 0 || j >= n){
            reachEnd = true;
            return;
        }
        if(grid[i][j] == 0){
            return;
        }
        grid[i][j] = 0;
        ones++;
        count++;
        for(int[] xy : offset){
            int x = i + xy[0];
            int y = j + xy[1];
            merge(grid, x, y);
        }
    }
}
