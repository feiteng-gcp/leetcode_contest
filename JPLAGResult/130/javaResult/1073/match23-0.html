class Solution {
    public int numEnclaves(int[][] A) {
        int total = getNumberOfOnes(A);
        int count = getReachables(A);
        // System.out.println(total + ", " + count);
        return total - count;
    }
    private int getNumberOfOnes(int[][] A) {
        int res = 0;
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[0].length; j++) {
                if (A[i][j] == 1) {
                    res++;
                }
            }
        }
        return res;
    }
    private int getReachables(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        int res = 0;
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < A[0].length; j++) {
                if (i == 0 || i == m - 1 || j == 0 || j == n - 1) {
                    if (A[i][j] == 1) {
                        // System.out.println("Start df from: " + i + ", " + j);
                        res += dfs(A, i, j);
                    }
                }
            }
        }
        return res;
    }
    private int dfs(int[][] A, int i, int j) {
        // System.out.println(i + ", " + j);
        int m = A.length;
        int n = A[0].length;
        A[i][j] = 0;
        int res = 0;
        int[][] dirs = new int[][] {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        for (int[] d : dirs) {
                int nr = i + d[0];
                int nc = j + d[1];
                if (nr < 0 || nr >= m || nc < 0 || nc >= n || A[nr][nc] != 1) continue;
                res += dfs(A, nr, nc);
        }
        return res + 1;
    }
}
