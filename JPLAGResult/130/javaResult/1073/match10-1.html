class Solution {
    public int numEnclaves(int[][] A) {
        int r=A.length, c = A[0].length;
        
final int[][] dirs = new int[][]{{1,0},{0,1},{-1,0},{0,-1}};
        
for(int i=0; i<r; i++){
            if(A[i][0]==1) dfs(A, i, 0, dirs);
            if(A[i][c-1]==1) dfs(A, i, c-1, dirs);
        }
        for(int j=0; j<c; j++){
            if(A[0][j]==1) dfs(A, 0, j, dirs);
            if(A[r-1][j]==1) dfs(A, r-1, j, dirs);
        }
        int res = 0;
        for(int i=0; i<r; i++){
            for(int j=0; j<c; j++){
                res += A[i][j];
            }
        }
        return res;
    }
    
    private void dfs(int[][] A, 
int i, int j, int[][] dirs){
        if(i<0||j<0||i>=A.length||j>=A[0].length||A[i][j]==0) return;
        A[i][j] = 0;
        for(int[] dir : dirs){
            dfs(A, i+dir[0], j+dir[1], dirs);
        }
    }
}
