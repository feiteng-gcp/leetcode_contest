class Solution {
    public int numEnclaves(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        
for(int i = 0; i < m; i ++) {
            visit(A, i, 0);
            visit(A, i, n-1);
        }
        for(int j = 1; j < n-1; j ++) {
            visit(A, 0, j);
            visit(A, m-1, j);
        }
        int res = 0;
        for(int i = 0; i < m; i ++) {
            for(int j = 0; j < n; j ++) {
                if(A[i][j] == 1) {
                    res ++;
                }
            }
        }
        return res;
    }
    
    private void visit(int[][] a, int i, int j) {
        if(i < 0 || i >= a.length || j < 0 || j >= a[0].length || a[i][j] == 0) {
            return;
        
}
        a[i][j] = 0;
        visit(a, i+1, j);
        visit(a, i-1, j);
        visit(a, i, j+1);
        visit(a, i, j-1);
    }
}
