class Solution {
    public String baseNeg2(int N) {
        if (N == 0) return "0";
        StringBuilder sb = new StringBuilder();
        int base = -2;
        while (N != 0) {
            int remainder = N % base;
            N /= base;
            if (remainder < 0) {
                remainder += -base;
                N += 1;
            }
            sb.append(remainder);
        }
        return sb.reverse().toString();
    }
}
