class Solution {
    public String baseNeg2(int N) {
        if(N==0) return "0";
        
        
        long s = N;
        
        long v = 1;
        
        long r = 0;
        
        
        long x = 1;
        StringBuilder sb = new StringBuilder();
        
        while(s!=0){
        
        if((s%2)==0){
            r = 0;
            sb.append(0); 
            s /= 2;
        }else{
            r = 1*x;
            sb.append(1);
            s -= r;
            s /= 2;
        }
            x = x*(-1);
        
        }
        
        return sb.reverse().toString();
        
    }
}
