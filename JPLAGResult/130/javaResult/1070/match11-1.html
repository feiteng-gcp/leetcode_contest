class Solution {
     public String baseNeg2(int N) {
        if (N == 0) return "0";
        StringBuilder sb = new StringBuilder();
        while (N != 0) {
            int cur = (int)(N % -2);
            N /= -2;
            if (cur < 0) {
                cur += 2;
                N += 1;
            }
            sb.append(cur);
        }
        return sb.reverse().toString();
    }
}
