class Solution {
    public String baseNeg2(int N) {
        if (N == 0) return "0";
        StringBuffer sb = new StringBuffer();
        while (N != 0) {
            int p = N % (-2);
            N = N / (-2);

            
if (p < 0) {
                p += 2;
                N++;
            }

            //System.out.println(N + " " + p);
            sb.append(p);
            
        }
        sb.reverse();
        return sb.toString();
    }
    
    // N = -2 * x + y, y >= 0
}
