class Solution {
    public String baseNeg2(int N) {
        if(N == 0) return "0";
        StringBuilder sb = new StringBuilder();
        while(N != 0) {
            int i = N % -2;
            N /= -2;
            if(i < 0) {
                i += 2;
                N++;
            }
            sb.append(i);
        }
        sb.reverse();
        return sb.toString();
        
    }
