class Solution {
    public String baseNeg2(int N) {
        if (N == 0) return "0";
        List<Integer> res = new ArrayList<>();
        while (N > 0) {
            res.add(N % 2);
            N /= 2;
        }
        int carry = 0;
        int i = 0;
        while (i < res.size() || carry > 0) {
            if (i == res.size()) {
                int digit = carry;
                res.add(digit % 2);
                carry = digit / 2;
            } else {
                int digit = res.get(i) + carry;
                res.set(i, digit % 2);
                carry = digit / 2;
            }
            if (res.get(i) == 1 && i % 2 == 1) carry += 1;
            i += 1;
        }
        String resstr = "";
        for (Integer r : res) {
            resstr = r + resstr;
        }
        return resstr; 
    }
}
