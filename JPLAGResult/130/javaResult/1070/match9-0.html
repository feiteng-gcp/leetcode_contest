class Solution {
    public String baseNeg2(int N) {
        if (N == 0) {
            return "0";
        }
        
        StringBuilder builder = new StringBuilder();
        
        int base = -2;
        
        while (N != 0) {
            int remainder = N % base;
            N /= base;
            
            if (remainder < 0)
            {
                remainder += -base;
                N++;
            }
            
            builder.append(remainder);
        }

        return builder.reverse().toString();
    }
}
