class Solution {
    public String baseNeg2(int N) {
       //  If n is zero then in any base it will be 0 only 
        if (N == 0) 
            return "0";
        String converted = ""; 
        int negBase = -2;
        while (N != 0) 
        { 
            int remainder = N % negBase; 
            N /= negBase; 

            if (remainder < 0) 
            { 
                remainder += (-negBase); 
                N += 1; 
            } 
            converted = String.valueOf(remainder) + converted; 
        }
        return converted; 
    }

    
}
