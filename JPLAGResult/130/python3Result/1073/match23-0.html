class Solution:
    def numEnclaves(self, A) -> int:
        
        def find(x):
            if p[x] != x:
                p[x] = find(p[x])
            return p[x]
        
        border = set()
        row, col = len(A), len(A[0])
        p = { (i,j):(i,j) for i in range(row) for j in range(col) }
        
        for i in range(row):
            border.add((i,0))
            border.add((i,col-1))
            
        for i in range(col):
            border.add((0,i))
            border.add((row-1,i))
        
        def union(x,y):
            px, py = find(x), find(y)
            if px in border:
                p[py] = px
            else:
                p[px] = py
        
        for i in range(row):
            for j in range(col):
                if A[i][j] == 1:
                    for x,y in ((i+1,j),(i,j+1)):
                        if x < row and y < col and A[x][y] == 1:
                            union((x,y),(i,j))
        
        res = 0
        for i in range(row):
            for j in range(col):
                if A[i][j] == 1 and find((i,j)) not in border:
                    res += 1
        return res
