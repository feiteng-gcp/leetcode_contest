from queue import Queue
class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        if not A: return 0
        R = len(A)
        if not A[0]: return 0
        C = len(A[0])
        need 
= []
        for i in range(R):
            need.append((i, 0))
            need.append((i, C-1))
        for i in range(C):
            need.append((0, i))
            need.append((R-1, i))
        s = 0
        for row in A:
            s += sum(row)
        can = 0
        for i, j in need:
            if(A[i][j] == 1):
                q = Queue()
                q.put((i,j))
                A[i][j] = 2
                can += 1
                #print((i,j))
                while not q.empty():
                    x,y = q.get()
                    for dx, dy in [(0,1),(1,0),(0,-1),(-1,0)]:
                        if 0<=x+dx<R and 0<y+dy<C and A[x+dx][y+dy]==1:
                            q.put((x+dx,y+dy))
                            #print((x+dx,y+dy))
                            can += 1
                            A[x + dx][y + dy] = 2
        return s - can
