import queue
class Solution:
    def vecinos(self,loc):
        result = [(loc[0]-1,loc[1]) , (loc[0],loc[1]-1) , (loc[0]+1,loc[1]) , (loc[0],loc[1]+1)]
        return result 
    def valid(self,elt,m,n):
        if elt[0]<0 or elt[0] >=m or elt[1]<0 or elt[1]>=n:
            return False
        return True 
    def numEnclaves(self, A: List[List[int]]) -> int:
        
        #Get all the nodes
        land = set()
        m = len(A)
        n = len(A[0])
        for x in range(m):
            for y in range(n):
                if A[x][y]==1:
                    land.add((x,y))
        
        seen = set()
        total = 0
        for root in land:
            check = True
            if root in seen:
                continue 
            Q=queue.Queue()
            Q.put(root)
            seen.add(root)
            count = 1
            while not Q.empty():
                u = Q.get()
                for v in self.vecinos(u):
                    if v in seen:
                        continue
                    if check:
                        check=self.valid(v,m,n)
                    if v in land:
                        count += 1
                        Q.put(v)
                        seen.add(v)
            if check:
                total += count
        return total
