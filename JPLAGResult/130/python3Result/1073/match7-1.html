class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        if len(A) == 0: return 0
        m = len(A)
        n = len(A[0])
        for i in range(m):
            self.dfs(A, m, n, i, 0)
            self.dfs(A, m, n, i, n-1)
        for j in range(n):
            self.dfs(A, m, n, 0, j)
            self.dfs(
A, m, n, m-1, j)
        count = 0
        for i in range(m):
            for j in range(n):
                if A[i][j] == 1:
                    count += 1
        return count
        
        
    def dfs(self, A, m, n, i, j):
        if i < 0 or i >= m: return
        if j < 0 or j >= n: return
        if A[i][j] != 1: return
        
        A[i][j] = 0
        self.dfs(A, m, n, i+1, j)
        self.dfs(A, m, n, i-1, j)
        self.dfs(A, m, n, i, j+1)
        self.dfs(A, m, n, i, j-1)
