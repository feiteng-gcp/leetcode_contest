from collections import deque

class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        queue = deque([])
        m, n = len(A), len(A[0])
        visited = [[False] * n for i in range(m)]
        for i in range(m):
            if A[i][0] == 1:
                visited[i][0] = True
                queue.append((i, 0))
            if A[i][n - 1] == 1:
                visited[i][n - 1] = True
                queue.append((i, n - 1))
        for i in range(n):
            if A[0][i] == 1:
                visited[0][i] = True
                queue.append((0, i))
            if A[m - 1][i] == 1:
                visited[m - 1][i] = True
                queue.append((m - 1, i))
        
        while queue:
            top = queue.popleft()
            for dx, dy in [(0, -1), (1, 0), (0, 1), (-1, 0)]:
                nxt_x, nxt_y = top[0] + dx, top[1] + dy
                if nxt_x >= 0 and nxt_x < m and nxt_y >= 0 and nxt_y < n:
                    if A[nxt_x][nxt_y] == 1 and visited[nxt_x][nxt_y] == False:
                        visited[nxt_x][nxt_y] = True
                        queue.append((nxt_x, nxt_y))
        ans = 0
        for i in range(m):
            for j in range(n):
                if A[i][j] == 1 and visited[i][j] == False:
                    ans += 1
        return ans
