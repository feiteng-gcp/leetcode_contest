class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        for i in range(len(A)):
            for j in range(len(A[0])):
                if i == 0 or j == 0 or i == len(A)-1 or j == len(A[0])-1:
                    if A[i][j] == 1:
                        # print(i,j)
                        self.dfs(A, i, j)
        ans = 0
        for i in range(len(A)):
            for j in range(len(A[0])):
                ans += A[i][j]
        return ans
    
    
    def dfs(self, A: List[List[int]], x: int, y:int ) -> None:
        if x < 0 or x >= len(A):
            return
        if y < 0 or y >= len(A[0]):
            
return
        if A[x][y] == 0:
            return
        A[x][y] = 0
        self.dfs(A, x+1, y)
        self.dfs(A, x-1, y)
        self.dfs(A, x, y+1)
        self.dfs(A, x, y-1)
