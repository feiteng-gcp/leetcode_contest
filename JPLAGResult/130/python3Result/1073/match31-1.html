class Solution(object):
    def numEnclaves(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """

        n = len(A)
        m = len(A[0])

        q = []
        for i in range(n):
            if A[i][0] == 1:
                q.append((i, 0))
            if A[i][m-1] == 1:
                q.append((i, m-1))
        for i in range(m):
            if A[0][i] == 1:
                q.append((0, i))
            if A[n-1][i] == 1:
                q.append((n-1, i))

        while q:
            cur = q.pop(0)
            if A[cur[0]][cur[1]] == 0:
                continue
            A
[cur[0]][cur[1]] = 0
            for mm in [(0, 1), (0, -1), (-1, 0), (1, 0)]:
                nx = cur[0]+mm[0]
                ny = cur[1]+mm[1]
                if nx >= 0 and nx < n and ny >= 0 and ny < m and A[nx][ny] == 1:
                    q.append((nx, ny))

        res = 0
        for i in range(n):
            res += sum(A[i])

        return res
