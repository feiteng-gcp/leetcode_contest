class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        X, Y = len(A), len(A[0])
        # return:
        # (size of land, touched boundary)
        def dfs(x,y):
            if x == -1 or x == X or y == -1 or y == Y: return 0, True
            if A[x][y] in [0, -1]: return 0, False
            
            cnt, touched = 1, False
            A[x][y] = -1
            for xx, yy in [(x+1,y),(x-1,y),(x,y+1),(x,y-1)]:
                _cnt, _touched = dfs(xx, yy)
                cnt += _cnt
                touched |= _touched
            
            return cnt, touched
        
        cnt = 0
        for x in range(X):
            for y in range(Y):
                
if A[x][y] == 1:
                    size, touched = dfs(x, y)
                    if not touched: cnt += size
        
        return cnt
