class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        visited = set()
        res = 0
        m, n 
= len(A), len(A[0])
        
        def visit(i, j):
            if -1 < i < m and -1 < j < n and A[i][j] == 1:
                A[i][j] = 0
                for i_, j_ in ((i+1, j), (i-1, j), (i, j+1), (i, j-1)):
                    visit(
i_, j_)
                
        for i in (0, m-1):
            for j in range(n):
                visit(i, j)
        
        for j in (0, n-1):
            for i in range(1, m-1):
                visit(i, j)
        
        return sum([sum(a) for a in A])
