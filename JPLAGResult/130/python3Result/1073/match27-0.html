from collections import deque

class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        m, n = len(A), len(A[0])
        dq = deque()
        for j in range(n):
            if A[0][j]:
                dq.append((0, j))
                A[0][j] = 0
        for j in range(n):
            if A[m-1][j]:
                dq.append((m-1, j))
                A[m-1][j] = 0
        for i in range(m):
            if A[i][0]:
                dq.append((i, 0))
                A[i][0] = 0
        for i in range(m):
            if A[i][n-1]:
                dq.append((i, n-1))
                A[i][n-1] = 0
        dir = [[0, 1], [0, -1], [-1, 0], [1, 0]]
        while dq:
            x0, y0 = dq.popleft()
            for d in dir:
                x, y = x0 + d[0], y0 + d[1]
                if 0<=x<m and 0<=y<n and A[x][y]:
                    dq.append((x, y))
                    A[x][y] = 0
        res = 0
        for i in range(m):
            res += sum(A[i][:])
        return res
