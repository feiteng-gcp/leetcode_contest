class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        
        self.height = len(A)
        self.width = len(A[0])
        self.visited = [[False for _ in range(self.width)] for _ in range(self.height)]
        
        def get_father(x):
            if self.father[x] == x:
                return x
            self.father[x] = get_father(self.father[x])
            return self.father[x]
        
        self.cnt1 = 0
        self.cntout 
= 0
        
        for i in range(self.height):
            for j in range(self.width):
                if A[i][j] == 1:
                    self.cnt1 += 1
                else:
                    self.visited[i][j] = True
        
        def dfs(x, y):
            # print(x, y)
            self.visited[y][x] = True
            self.cntout += 1
            for i, j in ((y + 1, x), (y - 1, x), (y, x + 1), (y, x - 1)):
                
if 0 <= i < self.height and 0 <= j < self.width and not self.visited[i][j]:
                    dfs(j, i)
            
        for i in range(self.height):
            if not self.visited[i][0]:
                dfs(0, i)
            if not self.visited[i][self.width - 1]:
                dfs(self.width - 1, i)
        for i in range(self.width):
            if not self.visited[0][i]:
                dfs(i, 0)
            if not self.visited[self.height - 1][i]:
                dfs(i, self.height - 1)
        
        return self.cnt1 - self.cntout
