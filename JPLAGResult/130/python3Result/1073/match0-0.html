class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        m, n = len(A), len(A[0])
        
        def dfs(i, j):
            if A[i][j] != 0:
                A[i][j] = 0
                for d in ((1, 0), (0, 1), (-1, 0), (0, -1)):
                    x, y = i+d[0], j+d[1]
                    if 0 <= x < m and 0 <= y < n:
                        dfs(x, y)
                        
        for i in (0, m-1):
            for j in range(n):
                dfs(i, j)
                
        for j in (0, n-1):
            for i in range(m):
                dfs(i, j)
                        
        return sum(sum(x) for x in A)
