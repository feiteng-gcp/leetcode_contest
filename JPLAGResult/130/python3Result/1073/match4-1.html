class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        
        if len(A) == 0 or len(A[0]) == 0:
            return 0
        rows = len(A)
        cols = len(A[0])
        def dfs(i, j):
            A[i][j] = 0
            
            if i>0 and A[i-1][j] == 1:
                dfs(i-1, j)
            if i<rows-1 and A[i+1][j] == 1:
                dfs(i+1, j)
            
            if j>0 and A[i][j-1] == 1:
                dfs(i, j-1)
            if j<cols-1 and A[i][j+1] == 1:
                dfs(i
, j+1)
        
        for i in range(rows):
            if A[i][0] == 1:
                dfs(i, 0)
            if A[i][cols-1] == 1:
                dfs(i, cols-1)
        
        for j in range(1, cols-1):
            if A[0][j] == 1:
                dfs(0, j)
            if A[rows-1][j] == 1:
                dfs(rows-1, j)
        
        return sum([sum(row) for row in A])
