class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        if not A or not A[0]:
            return 0
        
        m, n = len(A), len(A[0])
        
        def dfs(i, j):
            A[i][j] = 0
            for ni, nj in [(i+1, j), (i-1, j), (i, j+1), (i, j-1)]:
                if 0 <= ni < m and 0 <= nj < n and A[ni][nj] == 1:
                    dfs(ni, nj)
                    
        for i in range(m):
            if A[i][0] == 1:
                dfs(i, 0)
            if A[i][n-1] == 1:
                dfs(i, n-1)
                
        for j in range(n):
            if A[0][j] == 1:
                dfs(0, j)
            if A[m-1][j] == 1:
                dfs(m-1, j)
                
        res = 0
        for i in range(m):
            for j in range(n):
                if A[i][j]:
                    res += 1
        return res
