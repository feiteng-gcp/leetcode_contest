class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        # wrap the grid with 2
        # flood 2 via paths connected by one.
        # remaining number of 1s is the answer.
        
        # or more simply, starting from the boundary, try to flood 2 through 1
        
        m = len(A)
        n = len(A[0])
        
        dirs = [
            (1, 0),
            (0, 1),
            (-1, 0),
            (0, -1)
        ]
        
        def flood(A, i, j):
            if 0 <= i < m and 0 <= j < n:
                if A[i][j] == 1:
                    A[i][j] = 2
                    for d in dirs:
                        ii = i + d[0]
                        jj = j + d[1]
                        flood(A, ii, jj)
        
        for i in range(0, m):
            flood(A, i, 0)
            flood(A, i, n - 1)
            
        for j in range(0, n):
            flood(A, 0, j)
            flood(A, m - 1, j)
            
        count = 0
        for i in range(m):
            for j in range(n):
                if A[i][j] == 1:
                    count += 1
        
        return count
