class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        m, n = len(A), len(A[0])
        def neighbors(i, j):
            for di, dj in [(0,1),(0,-1),(1,0),(-1,0)]:
                ni, nj = i+di, j+dj
                if 0 <= ni < m and 0 <= nj < n:
                     yield ni, nj
            
        def dfs(i, j):
            A[i][j] = 2
            for ni, nj in neighbors(i, j):
                if A[ni][nj] == 1:
                    dfs(ni, nj)
        
        for i in range(m):
            if A[i][0] == 1: dfs(i, 0)
            if A[i][n-1] == 1: dfs(i, n-1)
        for j in range(n):
            if A[0][j] == 1: dfs(0, j)
            if A[m-1][j] == 1: dfs(m-1, j)
        
        return sum(A[i][j] == 1 for i in range(m) for j in range(n))
