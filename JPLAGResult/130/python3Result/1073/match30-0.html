class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        if not A or not A[0]:return 0
        m,n=len(A),len(A[0])
        sinked=set()
        # for r in A:
        #     print(r)
        def sink(i,j):
            if (i,j) in sinked or i<0 or i>=m or j<0 or j>=n  or A[i][j]==0:
                return
            else:
                A[i][j]=0
                sinked.add((i,j))
                for ii,jj in {(0,1),(0,-1),(1,0),(-1,0)}:
                    sink(i+ii,j+jj)
        for i,j in itertools.product([0,m-1],range(n)):
            if (i,j) not in sinked:
                sink(i,j)
        for i,j in itertools.product(range(m),[0,n-1]):
            if (i,j) not in sinked:
                sink(i,j)
        # print("-----")
        # for r in A:
        #     print(r)
        visited=set()
        ans=[0]
        def calc(i,j):
            if (i,j) in visited or i<0 or i>=m or j<0 or j>=n or A[i][j]==0:
                return
            else:
                A[i][j]=0
                visited.add((i,j))
                ans[0]+=1
                for ii,jj in {(0,1),(0,-1),(1,0),(-1,0)}:
                    calc(i+ii,j+jj)
        
        for i,j in itertools.product(range(m),range(n)):
            if A[i][j]==1 and (i,j) not in visited:
                calc(i,j)
        return ans[0]
