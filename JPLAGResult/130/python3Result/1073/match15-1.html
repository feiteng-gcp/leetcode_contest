class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        def dfs(A, i, j):
            A[i][j] = 0
            m, n = len(A), len(A[0])
            dirs = [[-1,0],[1,0],[0,-1],[0,1]]
            for each in dirs:
                ni, nj = i+each[0], j+each[1]
                if ni>=0 and ni<m and nj>=0 and nj<n and A[ni][nj]==1:
                    dfs(A,ni,nj)
        
        m = len(A)
        if m==0:
            return 0
        n = len(A[0])
        for i in range(m):
            if A[i][0]==1:
                dfs(A,i,0)
            if A[i][n-1]==1:
                dfs(A,i,n-1)
        for i in range(1,n-1):
            if A[0][i]==1:
                dfs(A,0,i)
            if A[m-1][i]==1:
                dfs(A
,m-1,i)
        ret = 0
        for i in range(m):
            for j in range(n):
                if A[i][j]==1:
                    ret += 1
        return ret
