class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        n, m = len(A), len(A[0])
        dr = [0, 1, 0, -1]
        dc = [1, 0, -1, 0]
        
        def search(r, c):
            if A[r][c] == 1:
                A[r][c] = 0
                for k in range(4):
                    tr = r + dr[k]
                    tc = c + dc[k]
                    if 0 <= tr < n and 0 <= tc < m:
                        search(tr, tc)
                        
        
for r in range(n):
            search(r, 0)
            search(r, m - 1)
        for c in range(m):
            search(0, c)
            search(n - 1, c)
        return sum(map(sum, A))
