from collections import deque

class Solution:
    def numEnclaves(self, board: List[List[int]]) -> int:
        q = deque()
        n, m = len(board), len(board[0])
        for i in range(n):
            for j in range(m):
                if board[i][j] == 1:
                    if i in [0, n-1] or j in [0, m-1]:
                        q.append((i, j))
                        board[i][j] = 0

                        
        ds = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        while q:
            i, j = q.pop()
            for d in ds:
                x, y = i + d[0], j + d[1]
                
                if 0 <= x < n and 0 <= y < m and board[x][y] == 1:
                    board[x][y] = 0
                    q.appendleft((x, y))
                
        
        res = 0;
        for row in board:
            res += sum(row)
        return res
