class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        N, M = len(A), len(A[0])
        
        cnt = 0
        for i in range(N):
            for j in range(M):
                cnt += A[i][j]

        q = []
        
for j in range(-1, M+1):
            q.append([-1, j])
            q.append([N, j])
        for i in range(0, N):
            q.append([i, -1])
            q.append([i, M])
        
        dirs = [[0,1],[0,-1],[1,0],[-1,0]]
        while q:
            i, j = q.pop()
            for di, dj in dirs:
                ni, nj = i + di, j + dj
                if ni < 0 or ni >= N or nj < 0 or nj >= M or A[ni][nj] == 0: continue
                A[ni][nj] = 0
                cnt -= 1
                q.append([ni, nj])
        return cnt
