class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        
        
        
        M, N = len(A), len(A[0])
        outer = set()
        
for i in range(M):
            if A[i][0] == 1:
                outer.add((i,0))
            if A[i][N-1] == 1:
                outer.add((i,N-1))
        for j in range(N):
            if A[0][j] == 1:
                outer.add((0,j))
            if A[M-1][j] == 1:
                outer.add((M-1,j))
        visited = outer.copy()
        while outer != set():
            newouter = set()
            for r,c in outer:
                for nr,nc in [(r-1,c), (r,c-1), (r,c+1), (r+1,c)]:
                    if 0 <= nr < M and 0 <= nc < N and A[nr][nc] == 1 and (nr,nc) not in visited:
                        visited.add((nr,nc))
                        newouter.add((nr,nc))
            outer = newouter
        return sum(sum(row) for row in A) - len(visited)
