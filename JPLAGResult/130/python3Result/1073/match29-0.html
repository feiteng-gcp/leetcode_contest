class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        if len(A) == 0 or len(A[0]) == 0:
            return 0
        def dfs(i, j):
            if A[i][j] == 0:
                return
            
            A[i][j] = 0
            for ii, jj in [(i-1, j), (i+1, j), (i, j-1), (i, j+1)]:
                if 0 <= ii < len(A) and 0 <= jj < len(A[0]):
                    dfs(ii, jj)
                    
        for i in range(len(A)):
            dfs(i, 0)
            dfs(i, len(A[0])-1)
            
        for i in range(len(A[0])):
            dfs(0, i)
            dfs(len(A)-1, i)
        
        count = 0
        for i in range(len(A)):
            for j in range(len(A[0])):
                if A[i][j] == 1:
                    count += 1
        return count
