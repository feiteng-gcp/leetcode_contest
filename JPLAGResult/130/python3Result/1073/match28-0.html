class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        def sink(i, j):
            res = i == 0 or i == m - 1 or j == 0 or j == n -1
            q = 1
            A[i][j] = 0
            if i > 0 and A[i - 1][j] == 1:
                b, nn = sink(i - 1, j)
                res |= b
                q += nn
            if i < m - 1 and A[i + 1][j] == 1:
                b, nn = sink(i + 1, j)
                res |= b
                q += nn
            if j > 0 and A[i][j - 1] == 1:
                b, nn = sink(i, j - 1)
                res |= b
                q += nn
            if j < n - 1 and A[i][j + 1] == 1:
                b, nn = sink(i, j + 1)
                res |= b
                q += nn
            return res, q
        res = 0
        
        if not A:
            return 0
        m = len(A)
        n = len(A[0])
        for i in range(len(A)):
            for j in range(len(A[i])):
                if A[i][j] == 1:
                    b, qq = sink(i, j)
                    if not b:
                        res += qq
        return res
