from collections import deque
class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        q=deque()
        h=len(A)
        w=len(A[0])
        for y in range(h):
            for x in range(w):
                #print("y:%d,x:%d:%d"%(y,x,A[y][x]))
                if A[y][x]==1 and (y== 0 or y==h-1 or x==0 or x==w-1):
                    # print("y:%d,x:%d,h:%d,w:%d"%(y,x,h,w))
                    A[y][x]=2
                    q.append((y,x))
                    while q:
                        cy, cx = q.popleft()
                        for ny, nx in ((cy + 1, cx), (cy - 1, cx), (cy, cx + 1), (cy, cx - 1)):
                            if ny >= 0 and ny < h and nx >= 0 and nx < w and A[ny][nx] == 1:
                                A[ny][nx] = 2
                                q.append((ny, nx))
        # print("A:%s"%(A,))
        ans=0
        for y in range(h):
            for x in range(w):
                if A[y][x]==1:
                    ans+=1
        return ans
