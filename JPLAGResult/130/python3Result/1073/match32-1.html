class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        
        
        
        def dfs(i,j):
            if A[i][j] == 1:
                A[i][j] = 0
                if i > 0:
                    dfs(i-1,j)
                if j > 0:
                    dfs(i,j-1)
                if i < len(A)-1:
                    dfs(i+1,j)
                if j < len(A[0])-1:
                    dfs(i,j+1)
                
            
        
        for i in range(len(A)):
            if A[i][0] == 1:
                A[i][0] = 0
                if i > 0:
                    dfs(i-1,0)
                if i < len(A)-1:
                    dfs(i+1,0)
                if len(A[0]) > 0:
                    dfs(i,1)
                
            if A[i][len(A[0])-1] == 1:
                A[i][len(A[0])-1] = 0
                if i > 0:
                    dfs(i-1,0)
                if i < len(A)-1:
                    dfs(i+1,0)
                if len(A[0]) > 0:
                    dfs(i,len(A[0])-2)
        
        for j in range(len(A[0])):
            if A[0][j] == 1:
                A[0][j] = 0
                if j > 0:
                    dfs(0,j-1)
                if j < len(A[0])-1:
                    dfs(0,j+1)
                if len(A) > 0:
                    dfs(1,j)
                
            if A[len(A)-1][j] == 1:
                A[len(A)-1][j] = 0
                if j > 0:
                    dfs(0,j-1)
                if j < len(A[0])-1:
                    dfs(0,j+1)
                if len(A) > 0:
                    dfs(len(A)
-2,j)
            
            
        s = 0
        for i in range(len(A)):
            
            for j in range(len(A[0])):
                if A[i][j] == 1:
                    s = s+1
        return s
