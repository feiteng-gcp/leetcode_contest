class Solution:
    def numEnclaves(self, A: List[List[int]]) -> int:
        self.A = A
        self.m, self.n = len(A), len(A[0])
        def dfs(i, j):
            # A[i][j] is 1 starting from the boudnary
            # mark all 1s connected to it
            self.A[i][j] = '#'
            dirs = [(1, 0), (-1, 0), (0, -1), (0, 1)]
            for dx, dy in dirs:
                x, y = i + dx, j + dy
                if 0 <= x < self.m and 0 <= y < self.n and self.A[x][y] == 1:
                    dfs(x, y)

        for i in range(self.m):
            if self.A[i][0] == 1:
                dfs(i, 0)
            if self.A[i][self.n-1] == 1:
                dfs(i, self.n-1)                    
                    
        for j in range(self.n):
            if self.A[0][j] == 1:
                dfs(0, j)
            if self.A[self.m-1][j] == 1:
                dfs(self.m-1, j)
        
        ans = 0
        for i in range(self.m):
            for j in range(self.n):
                if self.A[i][j] == 1:
                    ans += 1
        return ans
