from collections import deque

class Solution:
    def numEnclaves(self, a: List[List[int]]) -> int:
        if not a or not a[0]:
            return 0
        
        # bfs on each cell to mark
        n = len(a)
        m = len(a[0])
        
        for i in range(n):
                for j in range(m):
                    if (i == 0 or i == n-1 or j == 0 or j == m-1) and a[i][j] == 1:
                        queue = deque()
                        queue.append((i,j))
                        a[i][j] = 0

                        while queue:
                            x, y = queue.popleft()

                            if x-1 >= 0 and a[x-1][y] == 1:
                                a[x-1][y] = 0
                                queue.append((x-1,y))
                            if x+1 < n and a[x+1][y] == 1:
                                a[x+1][y] = 0
                                queue.append((x+1,y))
                            if y-1 >= 0 and a[x][y-1] == 1:
                                a[x][y-1] = 0
                                queue.append((x,y-1))
                            if y+1 < m and a[x][y+1] == 1:
                                a[x][y+1] = 0
                                queue.append((x,y+1
))
                            
        
        # find island
        count = 0
        for i in range(n):
            for j in range(m):
                if a[i][j] == 1:
                    count += 1
        
        # return
        return count
