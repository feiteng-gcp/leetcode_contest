# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        vals = []
        while head:
            vals.append(head.val)
            head = head.next
        res, stack = [], []
        
        for x in vals[::-1]:
            while stack and stack[-1] <= x:
                stack.pop()
            if not stack:
                res.append(0)
            else:
                res.append(stack[-1])
            stack.append(x)
        return res[::-1]
