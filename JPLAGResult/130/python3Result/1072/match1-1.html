# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None


class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        A = []
        while head:
            A.append(head.val)
            head = head.next
        # 单调栈 自顶而下严格递增
        s = [A[-1]]
        ans = [0]
        for i in range(len(A) - 2, -1, -1):
            while s and A[i] >= s
[-1]:
                s.pop()
            if not s:
                ans.insert(0, 0)
            else:
                ans.insert(0, s[-1])
            s.append(A[i])
        return ans
