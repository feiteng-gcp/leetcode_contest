# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        # get len of head
        n = 0
        node = head
        while node: node, n = node.next, n + 1
        
        stk = [] #(val, idx)
        ans 
= [0] * n
        i = 0 # idx of ans
        node = head
        while node:
            val = node.val
            while stk and stk[-1][0] < val:
                _, idx = stk.pop()
                ans[idx] = val
            stk.append((val, i))
            i += 1
            node = node.next
        
        # fill 0 to no next greater element
        for _, idx in stk: ans[idx] = 0
        return ans
