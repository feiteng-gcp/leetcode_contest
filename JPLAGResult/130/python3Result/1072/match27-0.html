# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        if not head:return []
        nums = []
        while head:
            nums.append(head.val)
            head = head.next
        n = len(nums)
        res = [0] * n
        stack = []
        for i,a in enumerate(nums):
            while stack and stack[-1][1] < a:
                j, b = stack.pop()
                res[j] = a
            stack.append((i,a))
        return res
