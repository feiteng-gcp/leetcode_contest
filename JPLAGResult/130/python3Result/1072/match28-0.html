# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:

        cur = head

            total += 1

        ret = [0]*total
        idx = 0
        cur = head
        while cur is not None:
            while len(dq)>0 and dq[-1][1]<cur.val:
                ret[dq.pop()[0]] = cur.val
            dq.append([idx,cur.val])
            idx += 1
            cur = cur.next
        return ret
