# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        stack = []
        curr = head
        res = []
        i = 0
        while curr:
            res.append(0)
            while stack and stack[-1]
[0].val < curr.val:
                _, pi = stack.pop()
                res[pi] = curr.val
            stack.append((curr, i))
            i += 1
            curr = curr.next
        return res
