# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None


class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        zz = []
        while head:
            zz.append(head.val)
            head = head.next
        n = len(zz)
        st = []
        r = []
        #print(zz)
        #raise 1
        for i in range(n-1, -1, -1) :
            x = zz
[i]
            while st and st[-1] <=x:
                st.pop()
            if not st:
                r.append(0)
            else:
                r.append(st[-1])
            st.append(x)
        r = reversed(r)
        #print(r)
        return list(r)
