# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        array = []
        while head:
            array.append(head.val)
            head = head.next
        N = len(array)
        ans = [0]*N
        stack = []
        for i in range(N):
            while stack and array[i] > array[stack[-1]]:
                temp = stack.pop()
                ans[temp] = array[i]
            stack.append(i)
        return ans
