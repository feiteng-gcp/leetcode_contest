# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        
        ans = []
        st = []
        
        i = 0
        while head:
            x = head.val
            ans.append(0)
            
            while st and st[-1][1] < x:
                j, y = st.pop()
                ans[j] = x
            st.append([i, x])
            
            i += 1
            head = head.next
        return ans
