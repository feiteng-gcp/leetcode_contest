# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        if not head:
            return []
        
        stack 
= []
        answer = []
        
        index = 0
        current = head
        while current:
            while stack and stack[-1][0] < current.val:
                # pop stack and update answer
                _, pre_index = stack.pop()
                answer[pre_index] = current.val
            
            # push itself
            stack.append((current.val, index))
            answer.append(0)
            
            # update
            current = current.next
            index += 1
        
        return answer
