from collections import defaultdict
class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        ls = []
        idx = 0
        mp = defaultdict(int)
        while head:
            if not ls or ls[-1][0] >= head.val:
                ls.append((head.val, idx))
            else:
                
while ls and ls[-1][0] < head.val:
                    mp[ls[-1][1]] = head.val
                    ls.pop()
                ls.append((head.val, idx))
            head = head.next
            idx += 1
        res = [0 for _ in range(idx)]
        for k in mp:
            res[k] = mp[k]
        return res
