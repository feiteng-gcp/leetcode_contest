# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        l = []
        while head:
            l.append(head.val)
            head = head.next
        s = []
        res = [0] * len(l)
        for i in range(len(l) - 1, -1, -1):
            while s and s[-1] <= l[i]:
                s.pop()
            res[i] = s[-1] if s else 0
            s.append(l[i])
        return res
