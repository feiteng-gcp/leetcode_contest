# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        a = []
        x = head
        while x:
            a.append(x.val)
            x = x.next
        st = []
        ret = []
        for x in a[::-1]:
            while st and x >= st[-1]:
                del(st[
-1])
            if not st:
                ret.append(0)
            else:
                ret.append(st[-1])
            st.append(x)
        return ret[::-1]
