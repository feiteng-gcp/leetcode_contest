# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

from collections import deque

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        nextLarger = dict()
        stack = deque()
        it = 0
        while head:
            while stack and head.val > stack[-1][1]:
                nextLarger[stack.pop()[0]] = head.val

            stack.append((it, head.val))
            it += 1
            head = head.next
        res = [nextLarger.get(i, 0) for i in range(it)]
        return res
