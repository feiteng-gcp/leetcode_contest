# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        r = [0]*10001
        stack = []
        node = head
        count = 0
        while node:
            while stack and stack[-1][0] < node.val:
                val, index = stack.pop()
                r[index] = node.val
            stack.append((node.val, count))
            node = node.next
            count += 1
            
        return r[:count]
