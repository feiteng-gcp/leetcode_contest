# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None


class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        
        
        
        
        
        nums = []
        p = head
        while p != None:
            nums.append(p.val)
            p = p.next
        
        res = [0]*len(nums)
        stack = []
        for i,num in enumerate(nums):
            if stack == []:
                stack.append(i)
            else:
                while stack != [] and nums[stack[-1]] < num:
                    res[stack[-1]] = num
                    stack.pop()
                stack.append(i)
        return res
