# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        ans = []
        stk = []
        
        while head:
            ans.append(0)
            
            while stk and stk[-1][0] < head.val:
                ans[stk[-1][1]] = head.val
                stk.pop()
                
            stk.append((head.val, len(ans)-1))
            head = head.next
            
        return ans
