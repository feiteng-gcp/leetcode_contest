# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        stack, idx, res = [], 0, []
        node = head
        
        while node:
            res.append(0)
            while stack and stack[-1][0] < node.val:
                item = stack.pop()
                res[item[1]] = node.val
            stack.append((node.val, idx))
            node = node.next
            idx += 1
            
        return res
