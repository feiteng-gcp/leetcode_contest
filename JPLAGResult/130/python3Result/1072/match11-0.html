# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        
        heap = []
        nextlargest = []
        
        ptr = head
        cnt = 0
        while ptr:
            
            nextlargest.append(0)
            while heap and heap[0][0] < ptr.val:
                _, i = heapq.heappop(heap)
                nextlargest[i] = ptr.val
            heapq.heappush(heap, (ptr.val, cnt))
            cnt += 1
            ptr = ptr.next
        
        return nextlargest
