# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        if not head:
            return []
        
        q = []
        res = []
        idx = 0
        while head:
            cur = head.val
            while q and cur > q[0][0]:
                prev_val, prev_idx = heapq.heappop(q)
                res[prev_idx] = cur
            
            heapq.heappush(q, (cur, idx))
            res.append(0)
            idx += 1
            head = head.next
            
        return res
