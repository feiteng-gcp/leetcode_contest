# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None


class Solution:
    def nextLargerNodes(self, head: ListNode) -> List[int]:
        l = []
        while head:
            l.append(head.val)
            head = head.next
        
        res = []
        stack = []
        for i in reversed(range(len(l))):
            while stack and stack[-1] <= l[i]:
                stack.pop()
            res.append(0 if not stack else stack[-1])
            stack.append(l[i])
        return res[::-1]
