class Solution:
    def prefixesDivBy5(self, A: List[int]) -> List[bool]:
        cur = 0 
        res = []
        for i in A:
            cur = cur * 2 + i
            if cur % 5 == 0:
                res.append(True)
            else:
                res.append(False)
        return res
