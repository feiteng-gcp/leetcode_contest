class Solution:
    def prefixesDivBy5(self, A: List[int]) -> List[bool]:
        res = []
        curr = 0
        for bit in A:
            curr = (curr << 1) + bit
            if curr % 5 == 0:
                res.append(True)
            else:
                res.append(False)
        return res
