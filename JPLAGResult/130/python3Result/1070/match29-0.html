class Solution:
    def baseNeg2(self, n: int) -> str:
        '''
        0:0
        N=M+K
        2: 010 --> 110
        3: 011 --> 111
        4: 100 --> 100
        5: 101 --> 101
        6: 0110 --> 11010
            '''
        
        if n == 0: 
            return "0"
        res = ""
        while (n != 0): 
            r = n % (-2) 
            n = int(n//(-2)) 
            if (r < 0): 
                r += 2
                n += 1
            res = str(r) + res
        return res
