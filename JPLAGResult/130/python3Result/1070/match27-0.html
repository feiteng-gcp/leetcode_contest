class Solution:
    def baseNeg2(self, N: int) -> str:
        if N == 0: 
            return '0'
  
        res = []
        while N: 
            N, m = divmod(N, -2)
      
            if m < 0: 
                m += 2
                N += 1
      
            res.append(m)
              
        return ''.join(map(str, reversed(res)))
