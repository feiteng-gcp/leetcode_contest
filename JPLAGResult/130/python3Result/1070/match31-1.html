class Solution:
    def baseNeg2(self, N: int) -> str:
        if N == 0:
            return "0"
        ans = []
        
while N != 0:
            mod = abs(N % (-2))
            ans.append(mod)
            N = (N - mod) // (-2)
            
        return ''.join([str(i) for i in ans[::-1]])
