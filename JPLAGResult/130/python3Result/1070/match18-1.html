class Solution:
    def baseNeg2(self, N: int) -> str:
        if N == 0:
            return "0"
        ans = []
        while N != 0:
            remind = N%(-2)
            N //= (-2)
            if remind < 0:
                remind += 2
                N += 1
            ans.append(str(remind))
            
        return "".join(ans[::-1])
