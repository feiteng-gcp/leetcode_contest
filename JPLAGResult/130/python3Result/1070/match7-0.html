class Solution:
    def baseNeg2(self, N: int) -> str:
        if N == 0:
            return "0"
        ans = ""
        while N != 0:
            rem = N % -2
            N //= -2
            if rem < 0:
                rem += 2
                N += 1
            ans = str(rem) + ans
        return ans
