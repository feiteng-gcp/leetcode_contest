class Solution:
    def baseNeg2(self, N: int) -> str:
        if N == 0:
            return '0'
        
        ans = ''
        
        while N != 1:
            if N % 2 == 0:
                ans += '0'
                N //= -2
            else:
                ans += '1'
                N = (N // -2) + 1
        ans += '1'
        return ans[::-1]
