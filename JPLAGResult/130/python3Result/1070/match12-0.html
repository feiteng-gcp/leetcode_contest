class Solution:
    def baseNeg2(self, n: int) -> str:
        if n == 0: return '0'
        converted = ''
        while n != 0:
            rem = n % -2
            n //= -2
            
            if rem < 0:
                rem += (-1) * (-2)
                n += 1
            
            converted = str(rem) + converted
        return converted
