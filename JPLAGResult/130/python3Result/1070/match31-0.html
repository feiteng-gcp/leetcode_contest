class Solution:
    def baseNeg2(self, N: int) -> str:
        def determin(c_num,index):
            next_num = 0
            if c_num > 1:
                c_num = (c_num % 2)
                next_num += 1 
            if index % 2 == 1:
                if c_num == 1:
                    next_num += 1 
            return c_num, next_num
        def convert_to_list(n):
            ret = []
            while n > 0:
                ret.append(n%2)
                n //= 2
            return ret
        
        if N == 0 :
            return "0"
        ori_list = convert_to_list(N)
        ret_list = []
        next_num = 0
        i = 0
        ret = []
        while i < len(ori_list):
            c_num = next_num + ori_list[i]
            c_num, next_num = determin(c_num,i)
            ret.append(c_num)
            i += 1
        while next_num > 0:
            c_num, next_num = determin(next_num,i)
            ret.append(c_num)
            i += 1
        return ''.join([str(x) for x in ret[::-1]])
