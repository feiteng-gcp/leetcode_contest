class Solution:
    def baseNeg2(self, N: int) -> str:
        if N == 0:
            return '0'
        
        ans = ''
        while N:
            remainder = N % -2
            N = N // -2
            
            if remainder < 0:
                remainder += 2
                N += 1
            ans = str(remainder) + ans
        return ans
