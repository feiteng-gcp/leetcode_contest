class Solution:
    def baseNeg2(self, N: int) -> str:
        # 6 -> 110 + 100 -> 11010 = 8 - 2
        if N == 0: return '0'
        converted = ''
        while N != 0:
            remainder = N % -2
            N //= -2
            if (remainder < 0): 
                remainder += 2
                N += 1
  
            converted = str(remainder) + converted
        return converted
