class Solution:
    def baseNeg2(self, N: int) -> str:
        if not N: return '0'
        res = ''
        while N:
            if N % 2 == 0:
                N = N // -2
                res += '0'
            else:
                N = (N - 1) // -2
                res += '1'
        return res[::-1]
