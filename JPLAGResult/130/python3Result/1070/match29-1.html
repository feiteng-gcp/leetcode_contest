class Solution:
    def baseNeg2(self, N: int) -> str:
        if N == 0:
            return '0'
        
        res = []
        
        while N != 0:
            v = N % -2
            N = int(N // -2)
            
            if v < 0:
                v += 2
                N += 1
            res = [v] + res
        
        res = ''.join(str(x) for x in res)
        
        return res
