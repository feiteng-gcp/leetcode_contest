class Solution {
public:
    int direction[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    bool numEnclavesSub(vector<vector<int>>& A, int x, int y,int &count) {
        bool state = false;
        for(int i = 0; i < 4; i++){
            int xx = x + direction[i][0];
            int yy = y + direction[i][1];
            if(xx<0 || xx>=A.size() || yy<0 || yy>=A[0].size()){
                state = true; continue;
            }
            if(A[xx][yy] == 1){
                A[xx][yy] = 0;
                count++;
                if(numEnclavesSub(A, xx, yy, count))state = true;
            }
        }
        return state;
    }
    
    int numEnclaves(vector<vector<int>>& A) {
        if(A.size()==0) return 0;
        int count = 0;
        for(int i = 0; i < A.size(); i++){
            for(int j = 0; j < A[0].size(); j++){
                if(A[i][j]==1){
                    A[i][j] = 0;
                    int tt = 1;
                    if(!numEnclavesSub(A, i, j, tt))count+=tt;
                }
            }
        }
        return count;
    }
};
