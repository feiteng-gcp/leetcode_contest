class Solution {
public:
    vector<vector<int>> d{{-1,0},{1,0},{0,-1},{0,1}};
    void dfs(vector<vector<int>>& A, int x, int y, vector<vector<bool>> &vis){
        if(vis[x][y] || A[x][y] == 0) return;
        //cout << x << ", " << y << endl;
        vis[x][y] = true;
        for(int i = 0; i < 4; i++){
            int dx = x + d[i][0], dy = y + d[i][1];
            if(dx >= 0 && dx < A.size() && dy >= 0 && dy < A[0].size()) dfs(A, dx, dy, vis);
        }
    }
    int numEnclaves(vector<vector<int>>& A) {
        int m = A.size(), n = A[0].size();
        vector<vector<bool>> vis(m, vector<bool>(n, false));
        for(int i = 0; i < m; i++){
            if(A[i][0] == 1) dfs(A, i, 0, vis);
            if(A[i][n-1] == 1) dfs(A, i, n-1, vis);
        }
        for(int i = 0; i < n; i++){
            if(A[0][i] == 1) dfs(A, 0, i, vis);
            if(A[m-1][i] == 1) dfs(A, m-1, i, vis);
        }
        int ans = 0;
        for(int i = 0; i < m; i++)
            for(int j = 0; j < n; j++)
                if(A[i][j] == 1 && !vis[i][j]) ans++;
        return ans;
    }
};
