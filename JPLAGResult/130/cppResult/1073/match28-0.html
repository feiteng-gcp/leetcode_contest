class Solution {
public:
    int numEnclaves(vector<vector<int>>& A) {
        int dx[4]={0,1,0,-1};
        int dy[4]={1,0,-1,0};
        int h=A.size();
        int w=A[0].size();
        queue<pair<int,int>> q;
        for(int i=0;i<h;i++){
            for(int j=0;j<w;j++){
                if(i==0||i==h-1||j==0||j==w-1){
                    if(A[i][j]){
                        q.push({i,j});
                        while(q.size()){
                            auto x=q.front(); q.pop();
                            int hh=x.first,ww=x.second;
                            A[hh][ww]=0;
                            for(int k=0;k<4;k++){
                                int cy=hh+dy[k];
                                int cx=ww+dx[k];
                                if(0<=cy&&cy<h&&0<=cx&&cx<w){ //範囲外取得キケン
                                    if(A[cy][cx]==1){ //通行可能かつ未到達
                                        q.push({cy,cx});
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        int ret=0;
        for(int i=0;i<h;i++){
            for(int j=0;j<w;j++){
                ret+=A[i][j];
            }
        }
        return ret;
    }
};
