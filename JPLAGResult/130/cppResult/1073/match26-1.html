class Solution {
public:

    int traverse(vector<vector<int>> &A, int y, int x, bool &flag) {
        if (y<0 || x<0 || y==A.size() || x==A[0].size()) {
            flag = true;
            return 0;
        }
        if (A[y][x] == 0)
            return 0;
        A[y][x] = 0;
        
        return traverse(A,y+1,x,flag)+
        traverse(A,y-1,x,flag)+
        traverse(A,y,x-1,flag)+
        traverse(A,y,x+1,flag)+1;
        
    }
    int numEnclaves(vector<vector<int>>& A) {
        int i,j;
        int ans = 0;
        
        for (i=0; i<A.size(); i++)

            for (j=0; j<A[0].size(); j++) {
                
                if (A[i][j] == 1) {
                    bool flag= false;
                    int ret = traverse(A,i,j,flag);
                    if (!flag) ans += ret;
                }
            }
        return ans;
    }
};
