class Solution {
public:
    int numEnclaves(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        queue<pair<int, int>> q;
        vector<vector<bool>> vis(n, vector<bool>(m));
        int dx[] = {-1, 0, 0, 1};
        int dy[] = {0, -1, 1, 0};
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (i == 0 || i == n - 1 || j == 0 || j == m - 1) {
                    if (a[i][j] == 1) {
                        vis[i][j] = 1;
                        q.push({i, j});
                    }
                }
            }
        }
        while (!q.empty()) {
            auto [x, y] = q.front();
            q.pop();
            for (int i = 0; i < 4; i++) {
                int tx = x + dx[i], ty = y + dy[i];
                if (tx >= 0 && tx < n && ty >= 0 && ty < m && a[tx][ty] == 1 && !vis[tx][ty]) {
                    q.push({tx, ty});
                    vis[tx][ty] = true;
                }
            }
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (!vis[i][j] && a[i][j]) {
                    ans++;
                }
            }
        }
        return ans;
    }
};
