class Solution {
public:
    int R, C;

    int numEnclaves(vector<vector<int>>& A) {
        R = A.size();
        C = A[0].size();
        for(int r = 0; r < R; ++r) {
            dfs(r, 0, A);
            dfs(r, C - 1, A);
        }
        for(int c = 0; c < C; ++c) {
            dfs(0, c, A);
            dfs(R - 1, c, A);
        }
        
        int cnt = 0;
        for(int r = 0; r < R; ++r)
            for(int c = 0; c < C; ++c)
                if(A[r][c] == 1)
                    cnt++;
        
        return cnt;
    }
    void dfs(int r, int c, vector<vector<int>>& A) {
        if(r < 0 || r >= R || c < 0 || c >= C)
            return;
        if(A[r][c] == 0)
            return;
        
        A[r][c] = 0;
        dfs(r - 1, c, A);
        dfs(r + 1, c, A);
        dfs(r, c - 1, A);
        dfs(r, c + 1, A);
    }
};
