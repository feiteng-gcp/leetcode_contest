int was[512][512];
int dd[4] = {1,-1,0,0};

class Solution {
public:
    
    vector<vector<int>> F;
    void dfs(int x, int y) {
        if (F[x][y]==0) return;
        if (was[x][y]) return;
        was[x][y]=1;
        for (int d = 0; d < 4; ++d) {
            int xx = x + dd[d];
            int yy = y + dd[3-d];
            if (xx>=0 && yy>=0 && xx<F.size() && yy<F[0].size()) dfs(xx,yy);
        }
    }
    
    int numEnclaves(vector<vector<int>>& A) {
        F = A;
        memset(was,0,sizeof(was));
        int X = A.size();

        int Y = A[0].size();
        for (int i = 0; i < X; ++i) {
            dfs(i,0);
            dfs(i,Y-1);
        }
        for (int i = 0; i < Y; ++i) {
            dfs(0,i);
            dfs(X-1,i);
        }
        int r = 0;
        for (int i = 0; i < X; ++i) {
            for (int j = 0; j < Y; ++j) {
                if (!was[i][j] && A[i][j]==1) ++r;
            }
        }
        return r;
    }
};
