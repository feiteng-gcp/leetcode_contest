class Solution {
public:
    //start from boundary 1s, bfs
    int numEnclaves(vector<vector<int>>& A) {
        int m = A.size();
        int n = A[0].size();
        queue<pair<int, int>> q;
        //set<pair<int, int>> s;
        int count = 0;
        int reachable = 0;
        for(int r = 0; r<m; r++){
            for(int c = 0; c<n; c++){
                if(A[r][c]==1){
                    count++;
                    //if(r==0||r==m||c==0||c==n){
                    if(r==0||r==m-1||c==0||c==n-1){
                    //s.insert(make_pair(r, c));
                        A[r][c]=0; //set 0 so it won't be pushed to queue again. and initialize visited.
                        reachable++;
                        q.push(make_pair(r, c)); //missed
                    }
                }
            }
        }
        
        static int offsets[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        while(!q.empty()){
            auto p = q.front();
            q.pop();
            for(int i = 0; i<4; i++){
                int new_r = p.first+offsets[i][0];
                int new_c = p.second+offsets[i][1];
                if(new_r>=0&&new_r<m&&new_c>=0&&new_c<n&&A[new_r][new_c]==1){
                    A[new_r][new_c]=0;
                    reachable++;
                    q.push(make_pair(new_r, new_c));
                }
            }
        }
        return count-reachable;
    }
};
