struct fff
{
    int x;
    int y;
};
int dir[4][2]={{-1,0},{0,-1},{1,0},{0,1}};
class Solution {
public:
    int numEnclaves(vector<vector<int>>& A) {
        queue<fff> q;
        int n=A.size(),m=A[0].size();
        fff now;
        for(int i=0;i<n;i++)
        {
            if(A[i][0]==1)
            {
                now.x=i;
                now.y=0;
                A[i][0]=0;
                q.push(now);
            }
            if(A[i][m-1]==1)
            {
                now.x=i;
                now.y=m-1;
                A[i][m-1]=0;
                q.push(now);
            }
        }
        for(int i=1;i<m-1;i++)
        {
            if(A[0][i]==1)
            {
                now.x=0;
                now.y=i;
                A[0][i]=0;
                q.push(now);
            }
            if(A[n-1][i]==1)
            {
                now.x=n-1;
                now.y=i;
                A[n-1][i]=0;
                q.push(now);
            }
        }
        while(!q.empty())
        {
            now=q.front();
            q.pop();
            fff news;
            for(int i=0;i<4;i++)
            {
                news.x=now.x+dir[i][0];
                news.y=now.y+dir[i][1];
                if(news.x>=0&&news.x<n&&news.y>=0&&news.y<m&&A[news.x][news.y]==1)
                {
                    A[news.x][news.y]=0;
                    q.push(news);
                }
            }  
        }
        int ans=0;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                ans=ans+A[i][j];
            }
        }
        return ans;
    }
};
