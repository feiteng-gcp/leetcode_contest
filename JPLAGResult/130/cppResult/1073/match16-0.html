const int N = 505,
          dx[] = {-1, 0, 1, 0},
          dy[] = {0, -1, 0, 1};

int n, m, a[N][N];
int p[N][N], sz;
bool used[N * N];

inline bool check(int x, int y){
    return 0 <= x && x < n && 0 <= y && y < m;
}
void dfs(int x, int y){
    p[x][y] = sz;
    for(int i = 0; i < 4; i++){
        int xx = x + dx[i],
            yy = y + dy[i];
        if(check(xx, yy) && a[xx][yy] == 1 && !p[xx][yy]){
            dfs(xx, yy);
        }
    }
}

class Solution {
public:
    int numEnclaves(vector<vector<int>>& A) {
        n = (int)A.size();
        m = (int)A[0].size();
        memset(used, 0, sizeof(used));
        memset(p, 0, sizeof(p));
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                a[i][j] = A[i][j];
            }
        }
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                if(a[i][j] == 1 && !p[i][j]){
                    sz += 1;
                    dfs(i, j);
                }
            }
        }
        for(int i = 0; i < n; i++){
            if(a[i][0] == 1){
                used[p[i][0]] = true;
            }
            if(a[i][m - 1] == 1){
                used[p[i][m - 1]] = true;
            }
        }
        for(int j = 0; j < m; j++){
            if(a[0][j] == 1){
                used[p[0][j]] = true;
            }
            if(a[n - 1][j] == 1){
                used[p[n - 1][j]] = true;
            }
        }
        int ans = 0;
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                if(a[i][j] == 1 && used[p[i][j]] == false){
                    ans += 1;
                }
            }
        }
        return ans;
    }
};
