class Solution {
public:
    vector<vector<int>> dir = {{0,1},{1,0},{0,-1},{-1,0}};
    int numEnclaves(vector<vector<int>>& A) {
        int m = A.size();
        int n = A[0].size();
        //union find
        vector<vector<int>> visited(m, vector<int>(n,0));
        for(int i = 0; i < m; ++i){
            dfs(i, 0, visited, A, m, n);
            dfs(i, n-1, visited, A, m, n);
        }
        for(int j = 0; j < n; ++j){
            dfs(0, j, visited, A, m, n);
            dfs(m-1, j, visited, A, m, n);
        }
        int res = 0;
        for(int i = 0; i < m; ++i){
            for(int j = 0; j < n; ++j){
                if(A[i][j] == 1) res++;
            }
        }
        return res;
    }
    
    void dfs(int i, int j, vector<vector<int>>& visited, vector<vector<int>>&A, int m, int n){
        if(A[i][j] == 0) return;
        A[i][j] = -1;
        for(auto d : dir){
            int ii = i + d[0];
            int jj = j + d[1];
            if(ii >= m || ii < 0 || jj >= n || jj < 0 || visited[ii][jj] || A[ii][jj] == 0){
                continue;
            }
            visited[ii][jj] = 1;
            A[ii][jj] = -1;
            dfs(ii, jj, visited, A, m, n);
        }
        return;
    }
};
