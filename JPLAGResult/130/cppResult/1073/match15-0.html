int dy[] = { -1,0,1,0 };
int dx[] = { 0,1,0,-1 };


class Solution {
public:
	int numEnclaves(vector<vector<int>>& A) {
		int h = A.size();
		int w = A[0].size();

		vector<vector<bool>> vis(h, vector<bool>(w, false));
		queue<pair<int, int>> q;

		for (int i = 0; i < h; i++) {
			if (A[i][0] == 1) {
				q.emplace(i, 0);
				vis[i][0] = true;
			}
		}
		for (int i = 0; i < h; i++) {
			if (A[i][w-1] == 1) {
				q.emplace(i, w - 1);
				vis[i][w - 1] = true;
			}
		}
		for (int i = 1; i+1 < w; i++) {
			if (A[0][i]==1) {
				q.emplace(0, i);
				vis[0][i] = true;
			}
		}
		for (int i = 1; i+1 < w; i++) {
			if (A[h-1][i]==1) {
				q.emplace(h - 1, i);
				vis[h - 1][i] = true;
			}
		}

		while (!q.empty()) {
			int y, x;
			tie(y, x) = q.front();
			q.pop();
			if (A[y][x] == 0)continue;
			vis[y][x] = true;

			for (int d = 0; d < 4;d++) {
				int ny = y + dy[d];
				int nx = x + dx[d];
				if (0 <= ny && ny < h && 0 <= nx && nx < w && A[ny][nx] == 1 && !vis[ny][nx]) {
					q.emplace(ny, nx);
				}
			}
		}
		int ret = 0;
		for (int i = 1; i + 1 < h;i++) {
			for (int j = 1; j + 1 < w;j++) {
				if (A[i][j] == 1 && !vis[i][j])ret++;
			}
		}
		return ret;
	}
};
