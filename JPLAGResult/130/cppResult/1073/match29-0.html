class Solution {
public:
    int numEnclaves(vector<vector<int>>& A) {
        int row = A.size();
        int col = row > 0 ? A[0].size() : 0;
        queue<pair<int, int> > line;
        for (int i = 0; i < row; ++i) {
            if (A[i][0] == 1) {
                line.push(make_pair(i, 0));
                A[i][0] = 2;
            }
            if (A[i][col - 1] == 1) {
                line.push(make_pair(i, col - 1));
                A[i][col - 1] = 2;
            }
        }
        for (int i = 0; i < col; ++i) {
            if (A[0][i] == 1) {
                line.push(make_pair(0, i));
                A[0][i] = 2;
            }
            if (A[row - 1][i] == 1) {
                line.push(make_pair(row - 1, i));
                A[row - 1][i] = 2;
            }
        }
        
        while (!line.empty())
        {
            int x = line.front().first;
            int y = line.front().second;
            line.pop();
            if (x - 1 >= 0 && A[x - 1][y] == 1) {
                A[x - 1][y] = 2;
                line.push(make_pair(x - 1, y));
            }
            if (x + 1 < row && A[x + 1][y] == 1) {
                A[x + 1][y] = 2;
                line.push(make_pair(x + 1, y));
            }
            if (y - 1 >= 0 && A[x][y - 1] == 1) {
                A[x][y - 1] = 2;
                line.push(make_pair(x, y - 1));
            }
            if (y + 1 < col && A[x][y + 1] == 1) {
                A[x][y + 1] = 2;
                line.push(make_pair(x, y + 1));
            }
        }
        int ans = 0;
        for (int i = 0; i < row; ++i) {
            for (int j = 0; j < col; ++j) {
                if (A[i][j] == 1) {
                    ++ans;
                    //printf("%d %d\n", i, j);
                }
            }
        }
        
        return ans;
    }
};
