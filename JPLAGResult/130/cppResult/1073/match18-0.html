class Solution {
public:
    void dfs(vector<vector<int>>& A, int i, int j){
        int N=A.size(), M=A[0].size();
        if(i<0 || j<0 || i>=N || j>=M) return;
        if(A[i][j]==1){
            A[i][j]=0;
            dfs(A, i-1, j);
            dfs(A, i+1, j);
            dfs(A, i, j-1);
            dfs(A, i, j+1);
        }
    }
    int numEnclaves(vector<vector<int>>& A) {
        int N=A.size(), M=A[0].size();
        for(int i=0; i<N; i++){
            dfs(A, i, 0);
            dfs(A, i, M-1);
        }
        for(int j=0; j<M; j++){
            dfs(A, 0, j);
            dfs(A, N-1, j);            
        }
        int ans=0;
        for(int i=0; i<N; i++)
            for(int j=0; j<M; j++)
                if(A[i][j]>0)
                    ans++;
        return ans;
    }
};
