class Solution {
public:
    bool vis[507][507];
    vector<vector<int>> G;

    int dir[4][2] = { {1, 0}, {-1, 0}, {0, 1}, {0, -1} };
    int n, m;
    int numEnclaves(vector<vector<int>>& A) {
        G = A;
        memset(vis, 0, sizeof(vis));
        n = A.size();
        m = A[0].size();

        int ans = 0;
        for(int i=0; i<n; ++i) {
            for(int j=0; j<m; ++j) {
                if(A[i][j] == 1 && !vis[i][j]) {
                    bool iso = true;
                    int cnt = dfs(i, j, iso);
                    if(iso) ans += cnt;
                }
            }
        }
        return ans;
    }
    bool check(int x, int y) {
        if(x<0||y<0||x>=n||y>=m) return false;
        if(vis[x][y]) return false;
        if(G[x][y] == 0) return  false;
        return true;
    }
    int dfs(int x, int y, bool &iso) {
        vis[x][y] = true;
        if(x==0||y==0||x==n-1||y==m-1) iso = false;
        int cnt = 1;
        for(int i=0; i<4; ++i) {
            int nx=x+dir[i][0];
            int ny=y+dir[i][1];
            if(check(nx, ny))
                cnt += dfs(nx, ny, iso);
        }
        return cnt;
    }
};
