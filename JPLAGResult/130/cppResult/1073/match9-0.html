class Solution {
public:
    int numEnclaves(vector<vector<int>>& A) {
        int m = A.size(), n = A[0].size();
        for(int i=0;i<m;++i){
            dfs(A,i,0,m,n);
            dfs(A,i,n-1,m,n);
        }
        for(int j=0;j<n;++j){
            dfs(A,0,j,m,n);
            dfs(A,m-1,j,m,n);
        }
        int res = 0;
        for(int i=0;i<m;++i){
            for(int j=0;j<n;++j){
                if(A[i][j]==1)
                    res ++;
            }
        }
        return res;
    }
    void dfs(vector<vector<int>>& A,int i,int j,int m,int n){
        if(i<0||j<0||i>=m||j>=n||A[i][j]==0) return;
        A[i][j] = 0;
        dfs(A,i+1,j,m,n);
        dfs(A,i-1,j,m,n);
        dfs(A,i,j+1,m,n);
        dfs(A,i,j-1,m,n);
    }
};
