class Solution {
public:
    int numEnclaves(vector<vector<int>>& A) {
        int n, m, cnt = 0;
        int x, y, tx, ty;
        int dx[4] = {0, 0, 1, -1};
        int dy[4] = {1, -1, 0, 0};
        queue<int> qx, qy;
        n = A.size();
        m = A[0].size();
        for(int i = 0; i < n; ++i)
        {
            if(A[i][0] == 1)
            {
                qx.push(i);
                qy.push(0);
                A[i][0] = 0;
            }
            if(A[i][m - 1] == 1)
            {
                qx.push(i);
                qy.push(m - 1);
                A[i][m - 1] = 0;
            }
        }
        for(int j = 0; j < m; ++j)
        {
            if(A[0][j] == 1)
            {
                qx.push(0);
                qy.push(j);
                A[0][j] = 0;
            }
            if(A[n - 1][j] == 1)
            {
                qx.push(n - 1);
                qy.push(j);
                A[n - 1][j] = 0;
            }
        }
        while(!qx.empty())
        {
            x = qx.front();
            y = qy.front();
            qx.pop();
            qy.pop();
            for(int k = 0; k < 4; ++k)
            {
                tx = x + dx[k];
                ty = y + dy[k];
                if(tx >= 0 && tx < n && ty >= 0 && ty < m && A[tx][ty] == 1)
                {
                    qx.push(tx);
                    qy.push(ty);
                    A[tx][ty] = 0;
                }
            }
        }
        for(int i = 0; i < n; ++i)
        {
            for(int j = 0; j < m; ++j)
            {
                if(A[i][j] == 1) cnt++;
            }
        }
        return cnt;
    }
};
