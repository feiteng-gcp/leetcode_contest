class Solution {
public:
    int numEnclaves(vector<vector<int>>& A) {
      if (A.empty()) {
        return 0;
      }
      n = A.size(), m = A[0].size();
      a = A;

      for (int i = 0; i < n ; ++i) {
        for (int j = 0; j < m; ++j) {
          if (i == 0 || i == n - 1 || j == 0 || j == m - 1) {
            if (a[i][j]) {
              dfs(i, j);

            }
          }
        }
      }     
      int ans = 0;
      for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
          ans += a[i][j];
        }
      }
      return ans;
    }
private:
  std::vector<std::vector<int>> a;
  int n, m;
  void dfs(int x, int y) {
    if (x < 0 || x >= n || y < 0 || y >= m || !a[x][y]) {
      return;
    }
    a[x][y] = 0;
    dfs(x - 1, y);
    dfs(x + 1, y);
    dfs(x, y - 1);
    dfs(x, y + 1);
  }
};
