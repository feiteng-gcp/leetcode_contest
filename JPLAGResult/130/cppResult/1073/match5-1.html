class Solution {
public:
    int numEnclaves(vector<vector<int>>& a) {
        static const int dx[] = {-1, 0, 0, +1};
        static const int dy[] = {0, -1, +1, 0};
        
        int n = a.size();
        int m = a.empty() ? 0 : a[0].size();

        queue<int> q;
        
        for (int i = 0; i < n; i++) {
            if (a[i][0]) {
                a[i][0] = 0;
                q.push(i);
                q.push(0);
            }
            if (a[i][m - 1]) {
                a[i][m - 1] = 0;
                q.push(i);
                q.push(m - 1);
            }
        }
        
        for (int j = 0; j < m; j++) {
            if (a[0][j]) {
                a[0][j] = 0;
                q.push(0);
                q.push(j);
            }
            if (a[n - 1][j]) {
                a[n - 1][j] = 0;
                q.push(n - 1);
                q.push(j);
            }
        }
        
        int x, y;
        
        while (!q.empty()) {
            x = q.front(); q.pop();
            y = q.front(); q.pop();
            
            for (int i = 0; i < 4; i++) {
                x += dx[i];
                y += dy[i];
                
                if (x > 0 && y > 0 && x < n && y < m && a[x][y]) {
                    a[x][y] = 0;
                    q.push(x);
                    q.push(y);
                }
                
                x -= dx[i];

                y -= dy[i];
            }
        }
        
        int ans = 0;
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (a[i][j]) ans++;
            }
        }
        
        return ans;
    }
};
