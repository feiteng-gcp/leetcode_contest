class Solution {
public:
    int numEnclaves(vector<vector<int>>& A) {
        int ans=0;
        m=A.size(),n=A[0].size();
        for(int i=0;i<m;i++)
            for(int j=0;j<n;j++)
            {
                if(A[i][j]==1)
                {
                    bool off=false;
                    int tmp=dfs(i,j,A,off);
                    if(!off)ans+=tmp;
                }
            }
        return ans;
    }
    
    int dx[4]={1,-1,0,0};
    int dy[4]={0,0,1,-1};
    
    int dfs(int x,int y,vector<vector<int>>& A,bool& off)
    {
        if(x<0||y<0||x>=m||y>=n)
        {
            off=true;
            return 0;
        }
        if(A[x][y]!=1)return 0;
        A[x][y]=2;
        int ans=1;
        for(int i=0;i<4;i++)
        {
            int nx=x+dx[i],ny=y+dy[i];
            ans+=dfs(nx,ny,A,off);
        }
        return ans;
    }
    
    int m,n;
};
