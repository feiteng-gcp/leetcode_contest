class Solution {
public:
    int numEnclaves(vector<vector<int>>& A) {
        int m = A.size();
        int n = A[0].size();
        for (int i = 0; i < m; ++i) {
            if (A[i][0] == 1) {
                dfs(A, i, 0);
            }
        }
        for (int i = 0; i < m; ++i) {
            if (A[i][n-1] == 1) {
                dfs(A, i, n-1);
            }
        }
        for (int i = 0; i < n; ++i) {
            if (A[0][i] == 1) {
                dfs(A, 0, i);
            }
        }
        for (int i = 0; i < n; ++i) {
            if (A[m-1][i] == 1) {
                dfs(A, m-1, i);
            }
        }
        
        int res = 0;
        for (auto& v : A) {
            for (auto& x : v) {
                if (x == 1) {
                    ++res;
                }
            }
        }
        return res;
    }
private:
    void dfs(vector<vector<int>>& A, int r, int c) {
        if (r >= 0 && r < A.size() && c >= 0 && c < A[0].size() && A[r][c] == 1) {
            A[r][c] = 2;
            dfs(A, r + 1, c);
            dfs(A, r - 1, c);
            dfs(A, r, c + 1);
            dfs(A, r, c - 1);
        }
    }
};
