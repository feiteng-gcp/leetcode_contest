int vx[] = {0,0,1,-1};
int vy[] = {1,-1,0,0};

class Solution {
public:
    int cal(vector<vector<int>>& A) {
        int ret = 0;
        int n = A.size(), m = A[0].size();

        for (int i = 0;i < n;i++) {
            for (int j = 0;j < m;j++) {
                if (A[i][j] == 1) {
                    ret++;
                }
            }
        }
        return ret;
    }
    int numEnclaves(vector<vector<int>>& A) {
        queue<pair<int, int>> Q;
        int n = A.size(), m = A[0].size();

        for (int i = 0;i < A.size();i++) {
            if (A[i][0] == 1) {
                A[i][0] = 0;
                Q.push(make_pair(i, 0));
            }
            if (A[i][m - 1] == 1) {
                A[i][m - 1] = 0;
                Q.push(make_pair(i, m - 1));
            }
        }
        for (int i = 0;i < A[0].size();i++) {
            if (A[0][i] == 1) {
                A[0][i] = 0;
                Q.push(make_pair(0, i));
            }
            if (A[n - 1][i] == 1) {
                A[n - 1][i] = 0;
                Q.push(make_pair(n - 1, i));
            }
        }
        
        while (Q.size() > 0) {
            pair<int, int> res = Q.front();
            Q.pop();
            for (int k = 0;k < 4;k++) {
                int x = res.first + vx[k];
                int y = res.second + vy[k];
                if (x < 0 || x >= n || y < 0 || y >= m) {
                    continue;
                }
                if (A[x][y] == 1) {
                    A[x][y] = 0;
                    Q.push(make_pair(x, y));
                }
            }
        }
        

        int ret = 0;
        for (int i = 0;i < n;i++) {
            for (int j = 0;j < m;j++) {
                if (A[i][j] == 1) {
                    ret++;
                }
            }
        }
        return ret;
    }
};
