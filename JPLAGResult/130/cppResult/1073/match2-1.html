    int move_[][2] = {
        0, 1,
        0, -1,
        -1, 0,
        1, 0
    };

class Solution {
public:
   

    int vis[505][505];
    int n, m;
    
    bool valid(int r, int c) {
        return r >= 0 && c >= 0 && r < n && c < m;
    }
    
    vector< vector<int> > A;
    

    void dfs(int r, int c) {
        if (vis[r][c]) return;
        
        vis[r][c] = 1;
        
        for (int k = 0; k < 4; k++) {
            int rr = r + move_[k][0];
            int cc = c + move_[k][1];
            
            if (valid(rr, cc) && A[rr][cc] == 1) dfs(rr, cc);
        }
    }
    
    int numEnclaves(vector<vector<int>>& _A) {
        A = _A;
        n = A.size();
        m = A[0].size();
        for (int i = 0; i < n; i++)

            for (int j = 0; j < m; j++)
                vis[i][j] = 0;
        
        for (int i = 0; i < n; i++) {
            if (A[i][0] == 1)
                dfs(i, 0);
            if (A[i][m-1] == 1)
                dfs(i, m - 1);
        }
        
        for (int i = 0; i < m; i++) {
            if (A[0][i] == 1)
                dfs(0, i);
            if (A[n-1][i] == 1)
                dfs(n-1, i);
        }
        int res = 0;
        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++)
                if (A[i][j] == 1 && vis[i][j] == 0) res++;
        return res;
    }
};
