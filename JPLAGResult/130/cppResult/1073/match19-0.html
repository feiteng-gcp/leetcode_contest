class Solution {
    int h, w;
    void dfs(vector<vector<int>>& A, int y, int x) {
        if (A[y][x] != 1) return;
        A[y][x] = 2;
        if (y > 0) dfs(A, y-1, x);
        if (y < h-1) dfs(A, y+1, x);
        if (x > 0) dfs(A, y, x-1);
        if (x < w-1) dfs(A, y, x+1);
    }
    
    
public:
    int numEnclaves(vector<vector<int>>& A) {
        h = A.size(), w = A[0].size();
        for (int i = 0; i < w; i++) {
            dfs(A, 0, i); dfs(A, h-1, i);
        }
        
        for (int i = 0; i < h; i++) {
            dfs(A, i, 0); dfs(A, i, w-1);
        }
        
        int ans = 0;
        for (int i = 0; i < h; i++) {
            for (int j = 0; j < w; j++) {
                if (A[i][j] == 1) ans++;
            }
        }
        return ans;
        
    }
};
