class Solution {
public:
    int numEnclaves(vector<vector<int>>& A) {
        int M = A.size();
        int N = A[0].size();
        
        vector<vector<bool>> reachBoundary(M, vector<bool>(N, false));
        int nEnclaves = 0;
        queue<pair<int,int>> Q;

        for(int i = 0 ; i < M ; ++i){
            for(int j = 0 ; j < N ; ++j){
                if(A[i][j]){
                    nEnclaves ++;
                    if(!i || i+1 == M ||
                        !j || j+1 == N){
                        reachBoundary[i][j] = true;
                        nEnclaves--;
                        Q.push(make_pair(i,j));
                    }
                }
            }
        }
        int dx[4] = {-1,0,1,0};

        int dy[4] = {0,1,0,-1};
        
        while(!Q.empty()){
            auto coord = Q.front(); Q.pop();
            for(int k = 0 ; k < 4 ; ++k){
                int nx = coord.first + dx[k];
                int ny = coord.second + dy[k];
                if(nx >= 0 && nx < M &&
                   ny >= 0 && ny < N &&
                   A[nx][ny] &&
                   !reachBoundary[nx][ny]){
                    reachBoundary[nx][ny] = true;
                    nEnclaves--;
                    Q.push(make_pair(nx,ny));
                }
            }
        }
        return nEnclaves;
    }
};
