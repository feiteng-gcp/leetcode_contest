class Solution {
public:

    void dfs(vector<vector<int>>& A, int x, int y, vector<int>& dx, vector<int>& dy) {
        if (A[x][y] != 1) return ;
        A[x][y] = 2;
        for (int k = 0; k < 4; ++k) {
            int nx = x + dx[k], ny = y + dy[k];
            if (0 <= nx && nx < A.size() && 0 <= ny && ny < A[0].size() && A[nx][ny] == 1)
                dfs(A, nx, ny, dx, dy);
        }
    }
    
    int numEnclaves(vector<vector<int>>& A) {
        vector<int> dx{-1, 0, 1, 0}, dy{0, -1, 0, 1};

        int M = A.size(), N = A[0].size();
        for (int i = 0; i < M; ++i) {
            if (A[i][0] == 1) dfs(A, i, 0, dx, dy);
            if (A[i][N - 1] == 1) dfs(A, i, N - 1, dx, dy);
        }
        for (int i = 0; i < N; ++i) {
            if (A[0][i] == 1) dfs(A, 0, i, dx, dy);
            if (A[M - 1][i] == 1) dfs(A, M - 1, i, dx, dy);
        }
        int res = 0;
        for (int i = 0; i < M; ++i) for (int j = 0; j < N; ++j)
            if (A[i][j] == 1) ++res;
        return res;
    }
};
