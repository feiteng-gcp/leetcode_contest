class Solution {
public:

    string baseNeg2(int N) 
    {
        return toNegativeBase(N, -2);
    }
    string toNegativeBase(int n, int negBase) 
    { 
        if (n == 0) 
            return "0"; 

        string converted = ""; 
        while (n != 0) 
        { 
            int remainder = n % negBase; 
            n /= negBase; 

            if (remainder < 0) 
            { 
                remainder += (-negBase); 
                n += 1; 
            } 

            converted = to_string(remainder) + converted; 
        } 

        return converted; 
    } 
};
