class Solution {
public:
    string baseNeg2(int N) {
        if (N == 0) {
            return "0";
        }
        string res = "";
        int base = -2;
        while(N != 0) {
            int r = N % base;
            N /= base;
            if (r < 0) {
                r -= base;
                N += 1;
            }
            res = to_string(r) + res;
        }
        return res;
    }
};
