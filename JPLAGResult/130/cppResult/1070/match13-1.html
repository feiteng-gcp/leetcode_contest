class Solution {
public:
    string baseNeg2(int n) {
        if(n==0)
            return "0";
        string converted = ""; 
        while (n != 0) 
        { 
            int remainder = n % (-2); 
            n /= -2; 
            if (remainder < 0) 
            { 
                remainder += 2; 
                n += 1; 
            }  
            converted = to_string(remainder) + converted; 
        } 
        return converted; 
    }
};
