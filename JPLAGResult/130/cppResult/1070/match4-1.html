class Solution {
public:
    string baseNeg2(int N) {
        if(N == 0) return "0";
        string ret = ""; 
        while (N != 0)  { 
            int rem = N % -2; 
            N /= -2; 
            if (rem < 0) 
            { 
                rem += 2; 
                N += 1; 
            } 
            ret = to_string(rem) + ret; 
        } 
  
        return ret; 
    }
};
