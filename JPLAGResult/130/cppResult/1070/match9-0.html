class Solution {
public:
    string baseNeg2(int n) {
        if (n == 0) return "0"; 
        string ret = ""; 
        while (n != 0) {  
            int remain = n % (-2); 
            n /= (-2); 
            if (remain < 0) { 
                remain += (2); 
                n += 1; 
            } 
            ret = to_string(remain) + ret;
        } 
        return ret; 
    }
};
