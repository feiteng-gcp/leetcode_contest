#include<bits/stdc++.h>
class Solution {
public:
    string baseNeg2(int N) {
        if(!N) return "0";
        string ans="";
        while(N != 0) {
         int rem = N % (-2); 
         N /= (-2); 
        if (rem < 0) { 
            rem += (2); 
            N += 1; 
         }  
         ans = to_string(rem) + ans; 
        }
        return ans;
    }
};
