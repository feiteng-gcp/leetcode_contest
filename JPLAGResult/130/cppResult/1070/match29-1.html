class Solution {
public:
    string baseNeg2(int n) {
        if (!n)
            return "0";
        string result = "";
        while (n) {
            int x = n % -2;
            n = n / -2;
            if (x < 0) {
                x += 2;
                n += 1;
            }
            result = (x ? '1' : '0') + result;
        }
        return result;
    }
};
