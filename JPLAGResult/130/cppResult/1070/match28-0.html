class Solution {
public:
    string baseNeg2(int N) {
        if(N == 0){
            return "0";
        }
        string ans;
        while(N != 0){
            int r = N % (-2);
            N = N / -2;
            if(r < 0){
                r += 2;
                N += 1;
            }
            ans = char('0' + r) + ans;
        }
        return ans;
    }
};
