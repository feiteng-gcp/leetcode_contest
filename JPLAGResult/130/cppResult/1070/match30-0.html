class Solution {
public:
    string str(int x) {
        stringstream s;
        s << x;
        return s.str();
    }
    string baseNeg2(int N) {
        if (N == 0) return "0";
        string res = "";
        while (N != 0) {
            int r = N % -2;
            N /= -2;
            if (r < 0) {
                r += 2;
                N += 1;
            }
            res = str(r) + res;
        }
        return res;
    }
};
