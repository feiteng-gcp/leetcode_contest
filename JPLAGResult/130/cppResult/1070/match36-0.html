class Solution {
public:
    string toString(int n) 
{ 
    string str; 
    stringstream ss; 
    ss << n; 
    ss >> str; 
    return str; 
} 
    string toNegativeBase(int n, int negBase) 
{ 
    //  If n is zero then in any base it will be 0 only 
    if (n == 0) 
        return "0"; 
  
    string converted = ""; 
    while (n != 0) 
    { 
        // Get remainder by negative base, it can be 
        // negative also 
        int remainder = n % negBase; 
        n /= negBase; 
  
        // if remainder is negative, add abs(base) to 
        // it and add 1 to n 
        if (remainder < 0) 
        { 
            remainder += (-negBase); 
            n += 1; 
        } 
  
        // convert remainder to string add into the result 
        converted = toString(remainder) + converted; 
    } 
  
    return converted; 
} 
    string baseNeg2(int N) {
        return toNegativeBase(N,-2);
        string s;
        while(N){
            if(N%(-2)==0){
                s.push_back('0');
            }
            else{
                s.push_back('1');
            }
            N/=-2;
        }
        reverse(s.begin(),s.end());
        return s;
    }
};
