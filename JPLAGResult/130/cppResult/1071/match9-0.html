class Solution
{
public:
    vector<bool> prefixesDivBy5(vector<int>& A)
    {
        vector<bool> res;
        res.reserve(A.size());

        int tmp = 0;
        for (int i : A)
        {
            tmp = (tmp * 2 + i) % 5;
            res.push_back(tmp == 0);
        }

        return res;
    }
};
