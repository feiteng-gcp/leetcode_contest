class Solution {
public:
    vector<bool> prefixesDivBy5(vector<int>& A) {
        vector <bool> ans;
        int n = A.size();
        int x = 0;
        for (int i = 0; i < n; i++) {
            x = x * 2 + A[i];
            x %= 5;
            ans.push_back(x == 0);
        }
        return ans;
    }
};
