class Solution {
public:
    vector<bool> prefixesDivBy5(vector<int>& A) {
        vector<bool> vb;
        int tot = 0;
        REP(i, sz(A)) {
            tot = (tot * 2 + A[i]) % 5;
            if(!tot) vb.pb(true);
            else vb.pb(false);
        }
        
        return vb;
    }
};
