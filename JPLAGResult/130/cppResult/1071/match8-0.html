class Solution {
public:
    vector<bool> prefixesDivBy5(vector<int>& A) {
        int number = 0;
        vector<bool> answers;

        for (int a : A) {
            number = (2 * number + a) % 5;
            answers.push_back(number == 0);
        }

        return answers;
    }
};
