/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> ans;
        while(head){
            ListNode* n = head;
            int val = head->val;
            
            int res=0;
            while(n){
                if(n->val > val){ res=n->val; break;}
                n = n->next;
            }
            ans.push_back(res);
            
            head = head->next;
        }
        return ans;
    }
};
