/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> ans;
        stack<pair<int, int>> st;
        ListNode *it = head;
        int n = 0;

        while (it != nullptr) {
            ans.push_back(0);
            while (!st.empty() && it->val > st.top().first) {
                ans[st.top().second] = it->val;
                st.pop();
            }
            st.push({it->val, n});
            n++;
            it = it->next;
        }
        return ans;
    }
};
