/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> num;
        while (head != nullptr) {
            num.push_back(head->val);
            head = head->next;
        }

        vector<int> ret(num.size());
        for (int i = 0; i < num.size(); i++) {
            ret[i] = 0;
            for (int j = i + 1; j < num.size(); j++) {                
                if (num[j] > num[i]) {
                    ret[i] = num[j];
                    break;
                }
            }
        }
        return ret;
    }
};
