/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */

class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        stack<pair<int, int>> st; // {value, index}
        vector<int> res;
        ListNode* current = head;
        int idx = 0;
        while(current != nullptr) {
            int x = current->val;            
            while(!st.empty() && x > st.top().first) {
                auto p = st.top();
                st.pop();
                if (res.size() < p.second + 1) {
                    res.resize(p.second + 1);
                }
                res[p.second] = x;
            }
            st.push({x, idx});            
            ++idx;
            current = current->next;
        }
        
        while(!st.empty()) {
            auto p = st.top();
            st.pop();
            if (res.size() < p.second + 1) {
                res.resize(p.second + 1);
            }
            res[p.second] = 0;
        }
        
        return res;
    }
};
