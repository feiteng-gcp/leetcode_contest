/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> a;
        while (head != nullptr) {
            a.push_back(head->val);
            head = head->next;
        }
        
        vector<int> res(a.size(), 0), stack;
        for (int i = 0; i < a.size(); ++i) {
            if (stack.empty() || a[i] <= a[stack.back()]) {
                stack.push_back(i);
            } else {
                while (!stack.empty() && a[i] > a[stack.back()]) {
                    res[stack.back()] = a[i];
                    stack.pop_back();
                }
                stack.push_back(i);
            }
        }
        return res;
    }
};
