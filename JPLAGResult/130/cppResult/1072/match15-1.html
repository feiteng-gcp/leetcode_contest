/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */

class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> v;
        while (head != nullptr) {
            v.push_back(head->val);
            head = head->next;
        }
        int l = v.size();
        stack<int> s;
        vector<int> res(l, 0);
        for (int i = 0; i < l; ++i) {
            if (s.empty() || v[i] <= v[s.top()]) {
                s.push(i);
            } else {
                while (!s.empty() && v[i] > v[s.top()]) {
                    res[s.top()] = v[i];
                    s.pop();
                }
                s.push(i);
            }
        }
        return res;
    }
};
