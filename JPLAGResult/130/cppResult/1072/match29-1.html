class Solution {
public:
  vector<int> nextLargerNodes(ListNode* head) {
  	stack<pair<int, int>> st;
  	vector<int> ans(10005);

  	int i = 0;
  	while (head) {
  		while (!st.empty() && head->val > st.top().second) {
  			ans[st.top().first] = head->val;
  			st.pop();
			}
			st.push({i, head->val});
  		head = head->next;
  		++i;
		}
		ans.resize(i);
		return ans;
  }
};
