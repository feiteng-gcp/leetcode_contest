/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    using pii = pair<int, int>;
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> v;
        for (ListNode *it = head; it; it = it->next)
            v.push_back(it->val);
        int n = v.size();
        vector<int> ret(n);
        vector<pii> Q;
        for (int i = n-1; i >= 0; -- i)
        {
            while (!Q.empty() && Q.back().first <= v[i]) Q.pop_back();
            ret[i] = (Q.empty()) ? 0 : Q.back().first;
            Q.push_back(pii(v[i], i));
        }
        return ret;
    }
};
