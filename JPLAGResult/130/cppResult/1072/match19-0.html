/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> a;
        while (head) {
            a.push_back(head->val);
            head = head->next;
        }
        vector<int> st;
        vector<int> b;
        for (int i = (int)a.size() - 1; i >= 0; i--) {
            while (!st.empty() && a[i] >= st.back()) st.pop_back();
            if (!st.empty()) b.push_back(st.back());
            else b.push_back(0);
            st.push_back(a[i]);
        }
        reverse(b.begin(), b.end());
        return b;
    }
};
