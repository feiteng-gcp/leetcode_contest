/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution
{
public:
    vector<int> nextLargerNodes(ListNode* head)
    {
        vector<int> res;

        ListNode* curr = head;
        while (curr != nullptr)
        {
            int tmp = 0;

            ListNode* fast = curr->next;
            while (fast != nullptr)
            {
                if (fast->val > curr->val)
                {
                    tmp = max(tmp, fast->val);
                    break;
                }
                fast = fast->next;
            }
            res.push_back(tmp);
            curr = curr->next;
        }

        return res;
    }
};
