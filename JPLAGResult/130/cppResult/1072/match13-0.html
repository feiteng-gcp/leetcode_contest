class Solution {
public:
  vector<int> nextLargerNodes(ListNode* head) {
    vector<int> vs;
    while(head){
      vs.emplace_back(head->val);
      head=head->next;
    }
    int n=vs.size();
    vector<int> res(n,0);
    set<int> s;
    for(int i=n-1;i>=0;i--){
      auto it=s.upper_bound(vs[i]);
      if(it!=s.end()) res[i]=*it;
      while(!s.empty()&&*s.begin()<=vs[i]) s.erase(s.begin());
      s.emplace(vs[i]);
    }
    return res;
  }
};
