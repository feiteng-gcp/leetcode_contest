/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> vals;
        while (head) {
            vals.push_back(head->val);
            head = head->next;
        }
        
        int n = vals.size();
        stack<int> st;
        vector<int> result(n, 0);
        result.back() = 0;
        st.push(vals.back());
        for (int i = n - 2; i >= 0; i--) {
            while (!st.empty() && st.top() <= vals[i]) {
                st.pop();
            }
            if (!st.empty() && vals[i] < st.top()) {
                result[i] = st.top();
            }
            st.push(vals[i]);
        }
        return result;
    }
};
