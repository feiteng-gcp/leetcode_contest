/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> tmp;
        while(head){
            tmp.push_back(head->val);
            head = head->next;
        }
        int n = tmp.size();
        vector<int> res(n,0);
        stack<int> st;
        for(int i=0;i<n;){
            if(st.empty()||tmp[i]<=tmp[st.top()]) st.push(i++);
            else{
                while(!st.empty()&&tmp[i]>tmp[st.top()]){
                    int j = st.top();st.pop();
                    res[j] = tmp[i];
                }
            }
        }
        return res;
    }
};
