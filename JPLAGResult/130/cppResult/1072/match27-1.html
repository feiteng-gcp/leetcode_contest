/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> mem;
        auto my_greater = [&](int idx1, int idx2){
            return mem[idx1] > mem[idx2];
        };
        auto p = head;
        while (p) {
            mem.emplace_back(p->val);
            p = p->next;
        }
        vector<int> ans(mem.size());

        priority_queue<int, vector<int>, decltype(my_greater)> min_pq(my_greater);
        for (int i = 0; i < mem.size(); ++i) {
            min_pq.emplace(i);
            while (min_pq.size() && mem[i] > mem[min_pq.top()]) {
                ans[min_pq.top()] = mem[i];
                min_pq.pop();
            }
        }
        while (min_pq.size()) {
            ans[min_pq.top()] = 0;
            min_pq.pop();
        }
        return ans;
    }
};
