class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector <int> elements;
        ListNode *cur = head;
        while(cur != NULL)
        {
            elements.push_back(cur -> val);
            cur = cur -> next;
        }
        int N = elements.size();
        vector <int> ans(N);
        stack <int> mono;
        for(int i = N - 1; i >= 0; i--)
        {
            while(!mono.empty() && elements[i] >= mono.top())
                mono.pop();
            if(mono.empty())
                ans[i] = 0;
            else
                ans[i] = mono.top();
            mono.push(elements[i]);
        }
        return ans;
    }
};
