/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> dp;
        while (head) {
            dp.push_back(head->val);
            head = head->next;
        }
        
        int l = dp.size();
        stack<int> cr;
        vector<int> ans(l);
        
        for (int i = l-1; i >= 0; i--) {
            while (!cr.empty() && cr.top() <= dp[i]) {
                cr.pop();
            }
            
            if (cr.empty()) {
                ans[i] = 0;
            } else {
                ans[i] = cr.top();
            }
            cr.push(dp[i]);
            
            
        }
        
        return ans;
        
    }
};
