/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> v;
        while (head != NULL) {
            v.push_back(head->val);
            head = head->next;
        }
        
        vector<int> ret(v.size(), 0), st;
        for (int i = 0; i < v.size(); ++i) {
            while (st.size() > 0 && v[i] > v[st.back()]) {
                ret[st.back()] = v[i];
                st.pop_back();
            }
            st.push_back(i);
        }
        
        return ret;
    }
};
