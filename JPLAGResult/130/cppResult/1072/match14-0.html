/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        if(!head) return {};
        vector<int> helper;
        
        while(head)
        {
            helper.push_back(head->val);
            head = head->next;
        }
        
        vector<int> res(helper.size());
        
        stack<int> s;
        
        for(int i = 0; i < helper.size(); i ++)
        {
            if(s.empty() || helper[i] <= helper[s.top()])
            {
            }
            else
            {
                while(!s.empty() && helper[s.top()] < helper[i])
                {
                    res[s.top()] = helper[i];
                    s.pop();
                }
            }
            s.push(i);
        }
        
        return res;
    }
};
