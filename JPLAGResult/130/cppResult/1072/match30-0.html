/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> res;
        stack<pair<int, int> > myS;// decreasing stack, so, when get the larger value, update the results
        ListNode* cur=head;
        int i=0;
        while( cur) {
            res.push_back(0);
            int curV= cur->val; 
            while( !myS.empty() && myS.top().first < curV){
                pair< int, int> curP= myS.top();
                res[curP.second]= curV;
                myS.pop();
            }
            myS.push( {curV, res.size()-1});
            i++;
            cur=cur->next;
        }
        return res;
    }
};
