/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        if(head == nullptr) return {};
        vector<int> res;
        auto cur = head;
        int i = 0;
        stack<pair<int,int>> sk;
        while(cur){
            res.push_back(0);
            while(!sk.empty() && sk.top().second < cur->val){
                res[sk.top().first] = cur->val;
                sk.pop();
            }
            sk.push({i, cur->val});
            i++;
            cur = cur->next;
        }
        return res;
    }
};
