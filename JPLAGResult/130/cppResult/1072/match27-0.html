/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> A;
        ListNode *cur = head;
        while (cur != NULL) {
            A.push_back(cur -> val);
            cur = cur -> next;
        }
        
        int n = A.size();
        stack<int> s;
        vector<int> ans(n);
        for (int i = 0; i < n; i++) {
            while (!s.empty() && A[i] > A[s.top()]) {
                ans[s.top()] = A[i];
                s.pop();
            }
            s.push(i);
        }
        while (!s.empty()) {
            ans[s.top()] = 0;
            s.pop();
        }
        return ans;
    }
};
