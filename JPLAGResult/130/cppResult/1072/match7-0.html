/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> nums;
        while (head != NULL) {
            nums.push_back(head->val);
            head = head->next;
        }
        int l = nums.size();
        vector<int> ans(l);
        stack<int> s;
        for (int i = l - 1; i >= 0; --i) {
            while (!s.empty() && s.top() <= nums[i]) s.pop();
            if (!s.empty())
                ans[i] = s.top(); else
                ans[i] = 0;
            s.push(nums[i]);
        }
        return ans;
    }
};
