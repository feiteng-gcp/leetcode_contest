/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> data;

        while (head != NULL) {
            data.push_back(head->val);
            head = head->next;
        }
        stack<int> s;
        vector<int> ret(data.size(), 0);
        for (int i = data.size() - 1;i >= 0;i--) {
            while (s.size() > 0 && data[i] >= s.top()) {
                s.pop();
            }
            if (s.size() == 0) {
                ret[i] = 0;
            } else {
                ret[i] = s.top();
            }
            s.push(data[i]);
        }
        return ret;
    }
};
