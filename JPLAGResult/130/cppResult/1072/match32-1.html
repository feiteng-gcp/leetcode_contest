/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* h) {
        
        vector<int> a;
        set<pair<int,int>> Q;

        while(h) {
            a.push_back(h->val);
            h = h->next;
        }
        vector<int> res(a.size());
        for (int i = 0; i < a.size(); ++i) {
            while(Q.size() && Q.begin()->first<a[i]) {
                res[Q.begin()->second] = a[i];
                Q.erase(Q.begin());
            }
            Q.insert({a[i],i});
        }
        while(Q.size()) {
            res[Q.begin()->second] = 0;
            Q.erase(Q.begin());
        }
        return res;
    }
};
