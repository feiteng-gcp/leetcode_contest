/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */

class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) 
    {
        stack<pair<int,int>> s;
        ListNode* x=head;
        int n=0;
        while(head)
        {
            head=head->next;
            n++;
        }
        vector<int> ans(n,0);
        int i=0;
        while(x)
        {
        
                while(!s.empty()&&x->val>s.top().first)
                {
                    auto t=s.top();
                    ans[t.second]=x->val;
                    s.pop();
                }
                s.push({x->val,i++});
                x=x->next;
            
        }
        return ans;
    }
};
