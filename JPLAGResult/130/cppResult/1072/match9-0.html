/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        
        ListNode* node=head;
        int len=0;
        while(node) {node=node->next;len++;}
        vector<int> ans(len);
        stack<vector<int>> st; //value and index
        node=head;
        int ind=0;
        while(node)
        {
            while(st.size() && node->val>st.top()[0]) //current value > top
            {
                ans[st.top()[1]]=node->val;
                st.pop();
            }
            st.push({node->val,ind});
            node=node->next;ind++;
        }
        return ans;
    }
};
