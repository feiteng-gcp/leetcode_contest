/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        stack<pair<int,int>> stk;//存decreasing 的 val,index
        ListNode *cur = head;
        int n =0;
        while(cur){
            n++;
            cur = cur->next;
        }
        
        vector<int> result(n,0);
        int idx=0;
        cur = head;
        while(cur){
            while(!stk.empty() && cur->val > stk.top().first){
                auto p = stk.top();stk.pop();
                result[p.second] = cur->val;
            }
                
            stk.push({cur->val,idx});
            idx++;
            cur = cur->next;
        }
        return result;
    }
};
