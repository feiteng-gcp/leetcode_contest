/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> nextLargerNodes(ListNode* head) {
        vector<int> rec;
        while (head != nullptr) {
            rec.push_back(head -> val);
            head = head -> next;
        }
        if (rec.empty()) return rec;

        vector<int> res(rec.size(), 0);
        stack<int> stk;
        stk.push(rec.size() - 1);
        for (int i = rec.size() - 2; i >= 0; --i) {
            while (!stk.empty() && rec[stk.top()] <= rec[i]) stk.pop();
            if (!stk.empty()) res[i] = rec[stk.top()];
            stk.push(i);
        }
        return res;
    }
};
