class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        int n = num_people;
        vector<int> ans(n, 0);
        int idx = 0;
        while (candies > 0) {
            int give = min(candies, idx+1);
            ans[idx%n] += give;
            candies -= give;
            idx+= 1;
        }
        return ans;
    }
};
