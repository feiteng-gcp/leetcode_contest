class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> res(num_people, 0);
        int ind = 0, distru = 1;
        while(candies > 0){
            int cursend = min(distru, candies);
            res[ind] += cursend;
            candies -= cursend;
            ind = (ind+1) % num_people;
            distru++;
        }
        return res;
    }
};
