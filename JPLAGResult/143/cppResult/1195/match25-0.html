class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> table(num_people,0);
        int sum=1;
        int i=0;
        while(candies>=sum){
            table[i]+=sum;
            candies-=sum;
            sum++;
            i=(i+1)%num_people;
        }
        table[i]+=candies;
        return table;
    }
};
