class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> result(num_people, 0);
        int cur = 1;
        int i = 0;
        while (candies > 0) {
            int give = min(cur, candies);
            result[i] += give;
            candies -= give;
            cur ++;
            i = (i + 1) % num_people;
        }
        return result;
    }
};
