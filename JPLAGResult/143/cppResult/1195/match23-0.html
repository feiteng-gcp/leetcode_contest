class Solution {
public:
    vector<int> distributeCandies(int candies, int n) {
        vector<int> ans(n,0);
        int m=1,p=0,have=0;
        while(candies>have)
        {
            if(m+have>=candies)
            {
                ans[p]+=candies-have;
                break;
            }
            ans[p]+=m;
            have+=m++;
            p=(p+1)%n;
        }
        return ans;
    }
};
