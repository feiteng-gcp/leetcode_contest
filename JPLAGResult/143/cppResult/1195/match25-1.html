class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> ans(num_people, 0);
        int val = 1;
        int pos = 0;
        while (candies > val) {
            ans[pos] += val;
            candies -= val;
            val += 1;
            pos = (pos+1) % num_people;
        }
        ans[pos] += candies;
        return ans;
    }
};
