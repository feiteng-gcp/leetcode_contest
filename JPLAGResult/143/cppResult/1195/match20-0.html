class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> res(num_people);
        int now = 1, idx = 0;
        while (candies) {
            int cur = min(now, candies);
            res[idx % num_people] += cur;
            candies = candies - cur;
            now++;
            idx++;
        }
        return res;
    }
};
