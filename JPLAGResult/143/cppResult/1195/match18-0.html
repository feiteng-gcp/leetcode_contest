class Solution {
public:
    vector<int> distributeCandies(int c, int n) {
        vector<int> res(n, 0);
        int i = 0, nxt = 1, remain = c;
        while (remain) {
            int x = min(nxt, remain);
            res[i] += x;
            remain -= x;
            i = (i + 1) % n;
            ++nxt;
        }
        return res;
    }
};
