class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> ans(num_people,0);
		for(int i = 0; ; i++){
			if(candies >= i+1) {
				ans[i%num_people] += i+1;
				candies -= i+1;
			} else {
				ans[i%num_people] += candies;
				break;
			}
		}
		
		return ans;
    }
};
