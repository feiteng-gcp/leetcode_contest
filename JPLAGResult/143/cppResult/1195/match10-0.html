class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> result(num_people);
        for (int i = 0, j = 1; candies > 0; i = (i + 1) % num_people, ++j) {
            int amount = min(j, candies);
            result[i] += amount;
            candies -= amount;
        }
        return result;
    }
};
