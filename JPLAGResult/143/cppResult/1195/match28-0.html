class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> Result(num_people);
		int P = 0;
		int N = 1;
		
		while(candies)
		{
			Result[P] += N;
			candies -= N;
			
			P = (P + 1) % num_people;
			N = min(candies, (N + 1));
		}
		
		return Result;
    }
};
