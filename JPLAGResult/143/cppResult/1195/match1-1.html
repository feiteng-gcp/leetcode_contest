using namespace std;


class Solution {
public:
    
    
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> res(num_people);
        
        int cur=1, pos=0;
        while (candies) {
            res[pos] += min(cur, candies);
            candies -= min(cur, candies);
            cur++;
            pos = (pos+1) % num_people;
        }
        return res;
    }
};
