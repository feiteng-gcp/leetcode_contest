class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> result(num_people, 0);
        int i = 0, c = 1;
        while (candies > 0) {
            result[i % num_people] += min(c, candies);
            candies -= min(c, candies);
            i++;
            c++;
        }
        return result;
    }
};
