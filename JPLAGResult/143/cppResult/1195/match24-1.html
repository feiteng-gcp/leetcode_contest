class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> res(num_people);
        for (int i = 0; ; ++i) {
            if (candies >= i + 1) {
                res[i % num_people] += i + 1;
                candies -= (i + 1);
            }
            else {
                res[i % num_people] += candies;
                break;
            }
        }
        return res;
    }
};
