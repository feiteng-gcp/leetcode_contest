class Solution {
public:
    vector<int> distributeCandies(int candies, int num_people) {
        vector<int> ans(num_people,0);
        int p=1;
        while(candies>0){
            for(int i=0;i<num_people;i++){
                int t=min(candies,p);
                ans[i]+=t;
                candies-=t;
                p++;
            }
        }
        return ans;
    }
};
