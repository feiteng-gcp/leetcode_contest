class Solution {
public:
    bool trans(char c) {
        return c == 't';
    }
    bool cal(char c, vector<bool>& bs) {
        switch (c) {
            case '!': return !bs[0];
            case '&':
                {
                    bool start = true;
                    for (auto b : bs) {
                        start = start && b;
                    }
                    return start;
                }
            case '|':
                {
                    bool start = false;
                    for (auto b : bs) {
                        start = start || b;
                    }
                    return start;
                }
        }
        cout << "wrong " << c << endl;
        return false;
    }
    bool parseBoolExpr(string exp) {
        if (exp.size() == 1) return trans(exp[0]);
        if (exp[0] == '(')
            return parseBoolExpr(exp.substr(1, exp.size() - 2));
        vector<bool> list;
        int l = 2, r = exp.size() - 1;
        int brasket = 0, start = l;
        for (int i = l; i < r; ++i) {
            if (exp[i] == ',' && brasket == 0) {
                list.push_back(parseBoolExpr(exp.substr(start, i - start)));
                start = i + 1;
            } else if (exp[i] == '(') ++ brasket;
            else if (exp[i] == ')') -- brasket;
        }
        //cout << start << " " << r - start << endl;
        list.push_back(parseBoolExpr(exp.substr(start, r - start)));
        //        cout << exp[0] << " " << list[0] << endl;
        return cal(exp[0], list);
    }
};
