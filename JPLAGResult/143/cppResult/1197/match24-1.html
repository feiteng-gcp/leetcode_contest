class Solution {
public:
	string S;
	int i;

	bool f() {
		if (S[i] == 't') {
			i++;
			return true;
		}
		if (S[i] == 'f') {
			i++;
			return false;
		}

		if (S[i] == '!') {
			i++;
			i++;
			bool res = !f();
			i++;
			return res;
		}

		if (S[i] == '&') {
			i++;
			i++;

			bool res = f();
			while (S[i] == ',') {
				i++;
				res &= f();
			}
			i++;
			return res;
		}
		if (S[i] == '|') {
			i++;
			i++;

			bool res = f();
			while (S[i] == ',') {
				i++;
				res |= f();
			}
			i++;
			return res;
		}

		return false;
	}

	bool parseBoolExpr(string expression) {
		S = expression + "#";
		i = 0;

		return f();
	}
};
