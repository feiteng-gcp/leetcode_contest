class Solution {
public:
    bool parseBoolExpr(string expression) {
      const char *s = expression.c_str();
      return eval(s);
    }
  bool eval(const char *&s) {
    // cout << *s << endl;
    if (*s == '&' || *s == '|') {
      char op = *s;
      ++s;
      ++s;
      auto res = eval(s);
      for (; *s == ',';) {
        ++s;
      //  cout << "{"<< s << "}\n";
        if (op == '&') {
          res = res & eval(s);
        } else {
          res = res | eval(s);
        }
        // cout << "["<< s << "]\n";
      }
      ++s; // ')'
      return res;
    } else if (*s == '!') {
      ++s;
      ++s;
      auto res = !eval(s);
      ++s;
      return res;
    } else if (*s == 't') {
      ++s;
      return true;
    } else if (*s == 'f') {
      ++s;
      return false;
    } else {
      //cout << *s << endl;
      return false;
    }
   
  }
};
