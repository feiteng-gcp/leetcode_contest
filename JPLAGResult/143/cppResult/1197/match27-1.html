class Solution {
public:
    bool parseBoolExpr(string expression) {
        stack<char> s1;
        stack<char> s2;
        for(int i=0;i<expression.length();i++){
            if(expression[i]=='|' || expression[i]=='&' || expression[i]=='!')
                s1.push(expression[i]);
            else if(expression[i]=='(' || expression[i]=='t' || expression[i]=='f')
                s2.push(expression[i]);
            else if(expression[i]==')'){
                char c=s1.top();
                s1.pop();
                int st=0;
                int sf=0;
                while(s2.top()!='('){
                    if(s2.top()=='t')
                        st++;
                    else if(s2.top()=='f')
                        sf++;
                    s2.pop();
                }
                s2.pop();

                if(c=='&'){
                    if(sf>0)
                        s2.push('f');
                    else
                        s2.push('t');
                }
                else if(c=='|'){
                    if(st>0)
                        s2.push('t');
                    else
                        s2.push('f');
                }
                else{
                    if(st>0)
                        s2.push('f');
                    else
                        s2.push('t');
                }
            }
        }
        if(s2.top()=='t') return true;
        else return false;
    }
};
