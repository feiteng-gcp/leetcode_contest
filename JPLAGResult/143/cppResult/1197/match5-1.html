class Solution {
public:
    bool parseBoolExpr(string s) {
        int n = s.length(), i = 0;
        // printf("n:%d\n", n);
        return helper(i, n, s);
    }
    
    bool helper(int &i, int n, string &s) {
        // printf("%d\n", i);
        if (i >= n)
            return false;
        
        // printf("%d %c\n", i, s[i]);
        

        if (s[i] == 'f') {
            ++i;
            return false;
        }else if (s[i] == 't') {
            ++i;
            return true;
        } else if (s[i] == '!') {
            i += 2;
            bool tmp = helper(i, n, s);
            ++i;
            return !tmp;
        } else if (s[i] == '|') {
            i += 2;
            bool tmp = helper(i, n, s);
            while (i < n && (s[i] == ',')) {
                ++i;
                tmp = helper(i, n, s) || tmp;
            }
            ++i;
            return tmp;
        } else {
            i += 2;
            bool tmp = helper(i, n, s);
            while (i < n && (s[i] == ',')) {
                ++i;
                tmp = helper(i, n, s) && tmp;
            }
            ++i;
            return tmp;
        }
    }
};
