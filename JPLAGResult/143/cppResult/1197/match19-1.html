class Solution {
public:
  bool expr(string &s,int &p){
    //cout<<s<<" "<<p<<endl;

    if(s[p]=='t'){
      p++;
      return true;
    }
    if(s[p]=='f'){
      p++;
      return false;
    }

    if(s[p]=='!'){
      p++;
      assert(s[p]=='(');
      p++;
      bool res=expr(s,p);
      assert(s[p]==')');
      p++;
      return !res;
    }

    if(s[p]=='|'){
      p++;
      assert(s[p]=='(');
      p++;
      bool res=false;
      while(p<(int)s.size()){
        res|=expr(s,p);
        if(s[p]==')') break;
        assert(s[p]==',');
        p++;
      }
      assert(s[p]==')');
      p++;
      return res;
    }

    if(s[p]=='&'){
      p++;
      assert(s[p]=='(');
      p++;
      bool res=true;
      while(p<(int)s.size()){
        res&=expr(s,p);
        if(s[p]==')') break;
        assert(s[p]==',');
        p++;
      }
      assert(s[p]==')');
      p++;
      return res;
    }

    assert(0);
    return 0;
  }

  bool parseBoolExpr(string expression) {
    int p=0;
    return expr(expression,p);
  }
};
