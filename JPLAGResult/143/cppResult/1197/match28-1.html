class Solution {
public:
    bool parseBoolExpr(string exp) {
        // stack<char> op;
        stack<char> stk;
        for (int i = 0; i < exp.length(); ++i) {
            char c = exp[i];
            if (c == '(' || c == '&' || c == '|' || c == '!' || c == 't' || c == 'f') {
                stk.push(c);
            } else if (c == ')') {
                bool a = true;
                bool b = false;
                bool c = true;
                while (stk.top() == 't' || stk.top() == 'f') {
                    bool tmp = (stk.top() == 't' ? true : false);
                    a &= tmp;
                    b |= tmp;
                    c = (!tmp);
                    
                    stk.pop();
                }
                if (stk.top() == '(') stk.pop();
                
                if (stk.top() == '&') {
                    stk.pop();
                    stk.push(a ? 't' : 'f');
                } else if (stk.top() == '|') {
                    stk.pop();
                    stk.push(b ? 't' : 'f');
                } else if (stk.top() == '!') {
                    stk.pop();
                    stk.push(c ? 't' : 'f');
                }    
            }
        }
        return stk.top() == 't' ? true : false;
    }
};
