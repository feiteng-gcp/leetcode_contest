class Solution {
public:
    string s;
    int ne[20005],st[20005],t,n;
    bool work(int l,int r)
    {
        if(l==r)return s[l]=='t';
        if(s[l]=='!')return !work(l+2,r-1);
        int i;
        bool ans;

        if(s[l]=='&')for(i=l+2,ans=1;;)
        {
            if(s[i]=='t'||s[i]=='f')
            {
                ans=ans&&work(i,i);
                i++;
            }
            else
            {
                ans=ans&&work(i,ne[i+1]);
                i=ne[i+1]+1;
            }
            if(s[i]==',')i++;
            else break;
            if(!ans)return 0;
        }

        else for(i=l+2,ans=0;;)
        {
            if(s[i]=='t'||s[i]=='f')
            {
                ans=ans||work(i,i);
                i++;
            }
            else
            {
                ans=ans||work(i,ne[i+1]);
                i=ne[i+1]+1;
            }
            if(s[i]==',')i++;
            else break;
            if(ans)return 1;
        }
        return ans;
    }
    bool parseBoolExpr(string expression) {
        s=expression;
        n=s.size();
        int i;
        t=0;
        for(i=0;i<n;i++)if(s[i]=='(')st[++t]=i;
        else if(s[i]==')')ne[st[t--]]=i;
        return work(0,n-1);
    }
};
