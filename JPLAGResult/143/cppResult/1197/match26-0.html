class Solution {
public:
    bool parseBoolExpr(string e) {
        if(e == "t") return true;
        if(e == "f") return false;
        if(e[0] == '!') {
            bool ret = parseBoolExpr(e.substr(2, e.size() - 3));
            return !ret;
        }
        
        if(e[0] == '&') {
            bool ret = true;
            string s;
            int count = 0;
            for(int i = 2; i < e.size() - 1; i++) {
                if(i == e.size() - 2) {
                    string tmp = s;
                    tmp.push_back(e[i]);
                    ret &= parseBoolExpr(tmp);
                } else if((count += e[i] == '(' ? 1 : e[i] == ')' ? -1 : 0) == 0
						&& e[i] == ',') {
                    ret &= parseBoolExpr(s);
                    s = "";
                } else {
                    s.push_back(e[i]);
                }
            }
            return ret;
        }
        else {       
        
        bool ret = false;
        string s;
			int count = 0;
			for (int i = 2; i < e.size() - 1; i++) {
				if (i == e.size() - 2) {
                    string tmp = s;
                    tmp.push_back(e[i]);
					ret |= parseBoolExpr(tmp);
				} else if ((count += e[i]== '(' ? 1 : e[i] == ')' ? -1 : 0) == 0
						&& e[i] == ',') {
					ret |= parseBoolExpr(s);
					s = "";
				} else {
					s.push_back(e[i]);
				}
            }
			
                return ret;
        }
        return false;
    }

};
