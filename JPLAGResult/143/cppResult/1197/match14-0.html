class Solution {
public:
    bool evalue(string exp) {
        
//        cout << exp << endl;
        
        if ( exp == "f" ) return false;
        if ( exp == "t" ) return true;
        int l = exp.size();
        if ( exp[0] == '!' ) return 1^evalue(exp.substr(2,l-3));
        if ( exp[0] == '&' ) {
            int cnt = 0 ;
            int res = 1;
            int last = 2;
            for ( int i = 2 ; i < l ; ++i ) {
                if ( exp[i] == '(' ) ++cnt;
                if ( exp[i] == ')' ) --cnt;
                if ( cnt == 0 && exp[i] == ',' || i == l-1 )  {
                    res &= evalue(exp.substr(last,i-last));
                    last = i+1;
                }
            }
            return res;
        }
        if ( exp[0] == '|' ) {
            int cnt = 0 ;
            int res = 0;
            int last = 2;
            for ( int i = 2 ; i < l ; ++i ) {
                if ( exp[i] == '(' ) ++cnt;
                if ( exp[i] == ')' ) --cnt;
                if ( cnt == 0 && exp[i] == ',' || i == l-1 )  {
                    res |= evalue(exp.substr(last,i-last));
                    last = i+1;
                }
            }
            return res;
        }
        printf("error");
        return true;
    }
    
    bool parseBoolExpr(string expression) {
        return evalue(expression);
    }
};
