class Solution {
public:
    
    string s;
    
    bool check(int l, int r) {
        if (l == r) {
            if (s[l] == 't') return true;
            if (s[l] == 'f') return false;
        }
        if (s[l] == '!') {
            return !check(l + 2, r - 1);
        }
        if (s[l] == '&') {
            bool res = true;
            int begin = l + 2;

            int cnt = 0;
            for (int i = l + 2; i <= r - 1; i++) {
                if (s[i] == '(') cnt += 1;
                if (s[i] == ')') cnt -= 1;               
                if (s[i] == ',' && cnt == 0) {
                    res &= check(begin, i - 1);
                    begin = i + 1;
                }
                if (!res) return res;
            }
            res &= check(begin, r - 1);
            return res;
        }
        if (s[l] == '|') {
            bool res = false;
            int begin = l + 2;

            int cnt = 0;
            for (int i = l + 2; i <= r - 1; i++) {
                if (s[i] == '(') cnt += 1;
                if (s[i] == ')') cnt -= 1;
                if (s[i] == ',' && cnt == 0) {
                    res |= check(begin, i - 1);
                    begin = i + 1;
                }
                if (res) return res;
            }
            res |= check(begin, r - 1);
            return res;
        }
        printf("error!, %d..%d\n", l, r);
        return false;
    }
    
    bool parseBoolExpr(string expression) {
        s = expression;
        return check(0, s.size() - 1);
    }
};
