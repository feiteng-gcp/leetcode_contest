class Solution {
public:
    bool getans(string exp) {
        int len = exp.length();
        if (len == 1) {
            if (exp[0] == 't')
                return true;
            return false;
        }
        char op = exp[0];
        int ban = 0;
        vector<string> tmp;
        int left = 2;
        for (int i = 2; i < len; i++) {
            if (exp[i] == '(') {
                ++ban;
            } else if (exp[i] == ')') {
                --ban;
            } else if (exp[i] == ',' && ban == 0) {
                tmp.push_back(exp.substr(left, i - left));
                left = i + 1;
            }
        }
        tmp.push_back(exp.substr(left, len - left - 1));
        // for (auto s : tmp) {
        //     cout << s << endl;
        // }
        int tlen = tmp.size();
        bool flag = getans(tmp[0]);
        if (op == '|') {
            if (!flag)
                for (int i = 1; i < tlen; i++) {
                    flag |= getans(tmp[i]);
                    if (flag)
                        break;
                }
        } else if (op == '&') {
            if (flag)
                for (int i = 1; i < tlen; i++) {
                    flag &= getans(tmp[i]);
                    if (!flag)
                        break;
                }
        } else if (op == '!') {
            flag = !flag;
        }
        cout << exp << " " << flag << endl;
        return flag;
    }
    bool parseBoolExpr(string expression) { return getans(expression); }
};
