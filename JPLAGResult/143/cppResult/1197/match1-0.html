class Solution {
public:
    bool parseBoolExpr(string e) {
        stack<char> op;
        stack<char> tf;
        for(int i=0;i<e.size();i++)
        {
            if(e[i] == '&' || e[i] == '|' || e[i] == '!')
            {
                op.push(e[i]);
            }else if(e[i] == '(')
            {
                tf.push('(');
            }else if(e[i] == ')')
            {
                char opt=op.top();
                op.pop();
                char tret='i';
                while(!tf.empty())//要考虑多个tf的情况。
                {
                    char tft=tf.top();
                    tf.pop();
                    if(tft == '(')
                        break;
                    if(tret== 'i')
                    {
                        tret=tft;
                        if(opt=='!')
                        {
                            if(tret == 't')
                                tret='f';
                            else
                                tret='t';
                        }
                    }else
                    {
                        if(opt=='&')
                        {
                            if(tret == 't' && tft =='t')
                                tret='t';
                            else
                                tret='f';
                        }else if(opt =='|')
                        {
                             if(tret == 'f' && tft =='f')
                                tret='f';
                            else
                                tret='t';
                        }
                    }
                }
                tf.push(tret);
            }else if(e[i] == ',')//还有逗号
            {
                //
            }
            else
            {
                tf.push(e[i]);
            }
        }
        return tf.top()=='t'?true:false;
    }
};
