class Solution {
public:
    bool parseBoolExpr(string expression) {
        size_t i = 0;
        return parse(expression, i);
    }
    
    bool parse(const std::string& s, size_t& i)

    {
        if (s[i] == 't')
        {
            ++i;
            return true;
        }
        
        if (s[i] == 'f')
        {
            ++i;
            return false;
        }

        if (s[i] == '!')
        {
            ++i;
            assert(s[i] == '(');
            ++i;

            bool r = parse(s, i);

            assert(s[i] == ')');
            ++i;
            return !r;
        }

        if (s[i] == '&' || s[i] == '|')
        {
            char ch = s[i];
            ++i;
            assert(s[i] == '(');
            ++i;
            
            std::list<bool> bl;
            while (s[i] != ')')
            {
                bool r = parse(s, i);
                bl.push_back(r);
                
                if (s[i] == ',')
                    ++i;
            }

            assert(s[i] == ')');
            ++i;
            
            if (ch == '&')
            {
                for (auto b : bl)
                {
                    if (!b)
                        return false;
                }
                return true;
            }
            else
            {
                for (auto b : bl)
                {
                    if (b)
                        return true;
                }
                return false;
            }
        }
        
        return false;
    }
};
