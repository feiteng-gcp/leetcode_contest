class Solution {
public:
    bool evaluate(vector<bool> & v, char op) {
        switch(op) {
            case '!': {
                assert(v.size() == 1);
                return !v[0];
            }

            case '&': {
                bool res = true;
                for (auto b : v) {
                    res = res && b;
                }
                return res;
            }
            case '|': {
                bool res = false;
                for (auto b : v) {
                    res = res || b;
                }
                return res;
            }
        }
        return false;
    }
    bool parseExpr(const string & E, int & i) {
        if (i >= E.size()) return true;
        if (E[i] == 't') {
            ++i;
            return true;
        }
        if (E[i] == 'f') {
            ++i;
            return false;
        }
        if (E[i] == '!' || E[i] == '&' || E[i] == '|') {
            char op = E[i];
            vector<bool> v;
            ++i;
            assert(E[i] == '(');
            ++i;
            while (i < E.size() && E[i] != ')') {
                v.push_back(parseExpr(E, i));
                if (E[i] == ',') {
                    ++i;
                    continue;
                }
            }
            assert(E[i] == ')');
            ++i;
            return evaluate(v, op);
        } else {
            assert(false);
        }
    }
    bool parseBoolExpr(string expression) {
        int i = 0;
        return parseExpr(expression, i);
    }
};
