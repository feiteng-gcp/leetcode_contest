class Solution {
public:
    bool parseBoolExpr(string expression) {
        stack<char> st;
        for(char ch:expression)
        {
            if(ch==')')
                solve(st);
            else
                st.push(ch);
        }
        return st.top()=='t'?true:false;
    }
    void solve(stack<char> &st)
    {
        int f=0,t=0;
        while(!st.empty()&&st.top()!='(')
        {
            char ch=st.top();st.pop();
            if(ch==',')
                continue;
            else if(ch=='f')
                ++f;
            else if(ch=='t')
                ++t;
        }
        char res;
        st.pop();
        if(!st.empty())
        {
            if(st.top()=='&')
                res=(f>0?'f':'t');
            else if(st.top()=='|')
                res=(t>0?'t':'f');
            else
                res=(f>0?'t':'f');
            st.pop();
        }
        st.push(res);
    }
};
