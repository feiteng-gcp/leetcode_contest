class Solution {
public:
    bool parseBoolExpr(string expression) {
        const char * p = expression.c_str();
        return parseBoolExpr(p);
    }
    bool parseBoolExpr(const char *& p) {
        if (*p == 't') {

            ++p;
            return true;
        } else if (*p == 'f') {
            ++p;
            return false;
        } else if (*p == '!') {
            ++p; // over !
            ++p; // over (
            bool x = parseBoolExpr(p);
            ++p; // over )
            return !x;
        } else if (*p == '&') {
            bool x = true;
            ++p; // over &
            ++p; // over (
            while (true) {
                x = x & parseBoolExpr(p);
                if (*p == ')') {
                    ++p; // over )
                    break;
                } else {
                    ++p; // over ,
                }
            }
            return x;
        } else if (*p == '|') {
            bool x = false;
            ++p; // over &
            ++p; // over (
            while (true) {
                x = x | parseBoolExpr(p);
                if (*p == ')') {
                    ++p; // over )
                    break;
                } else {
                    ++p; // over ,
                }
            }
            return x;
        }
        
        // bad case
        return false;
    }
};
