class Solution {
public:
    vector<int> mt;
    string s;
    
    bool solve(int l, int r) {
        if (s[l] == 't') return 1;
        if (s[l] == 'f') return 0;
        if (s[l] == '!') return !solve(l + 2, r - 1);
        if (s[l] == '&') {
            bool ret = 1;
            for (int i = l + 2; i < r; i++) {
                if (s[i + 1] != '(') {
                    ret &= solve(i, i);
                    i++;
                } else {
                    ret &= solve(i, mt[i + 1]);
                    i = mt[i + 1] + 1;
                }
            }
            return ret;
        } 
        if (s[l] == '|') {
            bool ret = 0;
            for (int i = l + 2; i < r; i++) {
                if (s[i + 1] != '(') {
                    ret |= solve(i, i);
                    i++;
                } else {
                    ret |= solve(i, mt[i + 1]);
                    i = mt[i + 1] + 1;
                }
            }
            return ret;
        }
        return false;
    }
    
    bool parseBoolExpr(string sz) {
        s = sz;
        int n = s.size();
        mt.resize(n);
        vector<int> st;
        for (int i = 0; i < n; i++) {
            if (s[i] == '(') {
                st.push_back(i);
            } else if (s[i] == ')') {
                mt[i] = st.back();
                mt[st.back()] = i;
                st.pop_back();
            }
        }
        return solve(0, n - 1);
    }
};
