class Solution {
public:
    bool parseBoolExpr(string expression) {
        stack<char> st;
        for (auto ch : expression) {
            if (ch != ')' && ch != ',') st.push(ch);
            else if (ch == ')') {
                string tmp = "";
                while (st.top() != '(') {

                    if (ch != ',') tmp += st.top();
                    st.pop();
                }
                st.pop();
                if (st.top() == '&') {
                    st.pop();
                    if (tmp.find('f') != -1) st.push('f');
                    else st.push('t');
                }else if (st.top() == '|') {
                    st.pop();
                    if (tmp.find('t') != -1) st.push('t');
                    else st.push('f');
                }else {
                    st.pop();
                    if (tmp[0] == 't') st.push('f');
                    else st.push('t');
                }
            }
        }
        if (st.top() == 'f') return false;
        else return true;
    }
};
