class Solution {
public:
  bool parse(string &e, int &L) {
    if (e[L] == 't') {
      L++;
      return true;
    }
    if (e[L] == 'f') {
      L++;
      return false;
    }
    if (e[L] == '!') {
      L+=2;
      bool t = parse(e, L);
      L++;
      return !t;
    }
    char c = e[L];
    bool t = c == '&'?true:false;
    L+=2;
    while (L < e.size()) {
      if (c == '&')  t &= parse(e, L);
      else t |= parse(e, L);
      if (e[L] == ',') {
        L++;
        continue;
      }
      if (e[L] == ')') {
        L++;
        return t;
      }
    }
    return false;
  }
  bool parseBoolExpr(string expression) {
    int L = 0;
    return parse(expression, L);
  }
};
