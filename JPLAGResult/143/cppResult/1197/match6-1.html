class Solution {
public:

    bool parseBoolExpr(const char *str, int &p)
    {
        if (str[p]=='t')
        {
            p++;
            return true;
        }
        if (str[p]=='f')
        {
            p++;
            return false;
        }
        if (str[p]=='!')
        {
            p+=2;
            bool t = parseBoolExpr(str, p);
            p++;
            return !t;
        }
        if (str[p]=='&')
        {
            p++;
            bool t = true;
            while (str[p]!=')')
            {
                p++;

                t = t&parseBoolExpr(str, p);
            }
            p++;
            return t;
        }
        if (str[p]=='|')
        {
            p++;
            bool t = false;
            while (str[p]!=')')
            {
                p++;
                t = t|parseBoolExpr(str, p);
            }
            p++;
            return t;
        }
        return true;
    }
    
    bool parseBoolExpr(string expression) {
        int p = 0;
        return parseBoolExpr(expression.c_str(), p);
    }
};
