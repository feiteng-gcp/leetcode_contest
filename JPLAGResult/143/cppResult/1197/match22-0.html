class Solution {
    string s;
    bool solve(int i, int len) {
        if (len == 1) {
            return s[i] == 't';
        }
        if (s[i] == '!') {
            return !solve(i+2, len-3);
        }
        if (s[i] == '&') {
            int j = i+2, cnt = 0;
            for (int k = j; k < i+len-1; k++) {
                if (s[k] == '(') cnt++;
                if (s[k] == ')') cnt--;
                if (s[k] == ',' && cnt == 0) {
                    if (!solve(j, k-j)) return false;
                    j = k+1;
                }
            }
            return solve(j, i+len-1-j);
        }
        assert(s[i] == '|');
        int j = i+2, cnt = 0;
        for (int k = j; k < i+len-1; k++) {
            if (s[k] == '(') cnt++;
            if (s[k] == ')') cnt--;
            if (s[k] == ',' && cnt == 0) {
                if (solve(j, k-j)) return true;
                j = k+1;
            }
        }
        return solve(j, i+len-1-j);
        
    }
public:
    bool parseBoolExpr(string e) {
        s = e;
        return solve(0, s.size());
    }
};
