class Solution {
public:
    bool parseBoolExpr(string expression) {
        stack<char> st;
        for(int i = 0; i < expression.size(); i ++){
            char c = expression[i];
            if(c != ')'){
                st.push(c);
            }
            else{
                int tC = 0;
                int fC = 0;
                while(true){
                    char tmp = st.top();
                    st.pop();
                    
                    if(tmp == 't') tC ++;
                    if(tmp == 'f') fC ++;
                    if(tmp == '(') break;
                }
                char tmp = st.top();
                st.pop();
                
                if(tmp == '!'){
                    if(tC)
                        st.push('f');
                    else
                        st.push('t');
                }
                else if(tmp == '|'){
                    if(tC)
                        st.push('t');
                    else
                        st.push('f');
                }
                else{
                    if(fC)
                        st.push('f');
                    else
                        st.push('t');
                }
                
            }
        }
        return st.top() == 't';
    }
};
