class Solution {
public:
    bool parseBoolExpr(string exp) {
        int i = 0;
        return parse(exp, i);
    }
    
    bool parse(string &exp, int &i) {
        if (exp[i] == 't') { i++; return true; }
        if (exp[i] == 'f') { i++; return false; }
        if (exp[i] == '!') {
            i += 2;
            bool r = !parse(exp, i);
            i++;
            return r;
        }
        if (exp[i] == '&') {
            i += 2;
            bool r = true;
            while (exp[i] != ')') {
                r &= parse(exp, i);
                if (exp[i] == ',') i++;
            }
            i++;
            return r;
        }
        if (exp[i] == '|') {
            i += 2;
            bool r = false;
            while (exp[i] != ')') {
                r |= parse(exp, i);
                if (exp[i] == ',') i++;
            }
            i++;
            return r;
        }
        return false;
    }
};
