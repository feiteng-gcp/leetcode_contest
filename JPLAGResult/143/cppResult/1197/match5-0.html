string S;
int cur;


class Solution {
public:
	bool eval() {
		if(S[cur]=='f') {
			cur++;
			return false;
		}
		else if(S[cur]=='t') {
			cur++;
			return true;
		}
		else if(S[cur]=='!') {
			cur+=2;
			bool ret=eval();
			cur++;
			return !ret;
		}
		else if(S[cur]=='&') {
			cur+=2;
			bool ret=eval();
			while(S[cur]==',') {
				cur++;
				ret=ret&eval();
			}
			cur++;
			return ret;
		}
		else {
			cur+=2;
			bool ret=eval();
			while(S[cur]==',') {
				cur++;
				ret=ret|eval();
			}
			cur++;
			return ret;
		}
		
	}
    bool parseBoolExpr(string expression) {
        S=expression;
        cur=0;
        return eval();
    }
};
