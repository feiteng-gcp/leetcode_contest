class Solution {
public:

    bool parse(const char *&s) {
        // printf("c = %c\n", *s);
        if (*s == 't') {
            s++;
            return true;
        }
        if (*s == 'f') {
            s ++;
            return false;
        }
        if (*s == '!') {
            s += 2;
            bool v = !parse(s);
            s ++;
            return v;
        }
        if (*s == '&') {
            s += 2;
            bool v = true;
            do {
                v &= parse(s);
            } while (*(s++) == ',');
            return v;
        }
        if (*s == '|') {
            s += 2;
            bool v = false;
            do {
                v |= parse(s);
            } while (*(s++) == ',');
            return v;
        }
        return false;
    }
    bool parseBoolExpr(string expression) {
        const char *s = expression.c_str();
        return parse(s);
    }
};
