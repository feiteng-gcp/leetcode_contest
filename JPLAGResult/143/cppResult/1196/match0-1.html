class Solution {
public:
    int minHeightShelves(vector<vector<int>>& a, int w) {
        int n = a.size();
        
        vector<int> f(n+1);
        f[0] = 0;
        
        for (int i = 1; i <= n; ++i) {
            int sum = a[i-1][0], h = a[i-1][1];
            f[i] = h + f[i-1];
            

            for (int j = i-2; j >= 0; --j) {
                sum += a[j][0];
                if (sum > w)
                    break;
                
                h = max(h, a[j][1]);
                f[i] = min(f[i], h + f[j]);
            }
        }
        
        return f[n];
    }
};
