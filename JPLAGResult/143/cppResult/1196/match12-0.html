class Solution {
public:
    int minHeightShelves(vector<vector<int>>& b, int w) {
        int n = b.size();
        vector<int> dp(n, INT_MAX);
        
        for (int cur = 0; cur < n; ++cur) {
            int ww = 0, hh = 0;
            for (int pre = cur; pre >= 0; --pre) {
                ww += b[pre][0];
                hh = max(hh, b[pre][1]);
                if (ww > w) break;
                dp[cur] = min(dp[cur], hh + (pre ? dp[pre - 1] : 0));
            }
        }
        
        return dp[n - 1];
    }
};
