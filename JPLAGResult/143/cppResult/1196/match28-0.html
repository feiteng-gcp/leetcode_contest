int dp[1007];
class Solution {
public:
    int minHeightShelves(vector<vector<int>>& B, int W) {
        int n = B.size();
        vector<vector<int>> b;
        b.push_back(vector<int>());
        for(auto e : B) b.push_back(e);
        memset(dp, 0x3f, sizeof(dp));
        dp[0]=0;
        for(int i=1; i<=n; ++i) {
            int w=0;
            int mx=0;
            for(int j=i; j>=1; --j) {
                w+=b[j][0];
                mx = max(mx, b[j][1]);
                if(w>W) break;
                dp[i]=min(dp[i], dp[j-1]+mx);
            } 
        }
        return dp[n];
    }
};
