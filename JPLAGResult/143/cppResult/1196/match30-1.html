const int N = 5000;

int dp[N];

class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int w) {
        FILL(dp, 0x3F);

        dp[0] = 0;
        for (int i = 1; i <= books.size(); ++i) {
            int cur = 0;
            int curMax = 0;
            for (int j = i - 1; j >= 0; --j) {
                cur += books[j][0];
                curMax = max(curMax, books[j][1]);
                
                if (cur > w) break;
                dp[i] = min(dp[i], dp[j] + curMax);
            }
        }
        return dp[books.size()];
    }
};
