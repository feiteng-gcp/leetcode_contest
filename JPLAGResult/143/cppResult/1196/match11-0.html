class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
    	int n = books.size();
    	vector<int> f(n + 1, INT_MAX);
    	f[0] = 0;
    	rep(i, 1, n + 1) {
    		int w = 0;
    		int h = 0;
    		per(j, 0, i) {
    			w += books[j][0];
    			h = max(h, books[j][1]);
    			if (w > shelf_width) break;
    			f[i] = min(f[i], f[j] + h);
    		}
    	}
    	return f[n];
    }
};
