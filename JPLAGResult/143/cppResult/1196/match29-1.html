class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        vector<int> dp(books.size() + 1, 0);
        for (int i = books.size() - 1;i >= 0;i--) {

            int sum = 0;
            int height = 0;
            int ret = 0x3fffffff;
            for (int j = i;j < books.size();j++) {
                if (sum + books[j][0] <= shelf_width) {
                    sum += books[j][0];
                    height = height > books[j][1] ? height : books[j][1];
                    ret = ret > height + dp[j + 1] ? height + dp[j + 1] : ret;
                } else {
                    break;
                }
            }
            dp[i] = ret;
        }
        return dp[0];
    }
};
