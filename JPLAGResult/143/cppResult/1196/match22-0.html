class Solution {
public:
    int minHeightShelves(vector<vector<int>>& bk, int width) {
        int len=bk.size();
        int dp[len];
        dp[0]=bk[0][1];
        for (int i=1; i<len; i++){
            dp[i]= dp[i-1]+ bk[i][1];
            int w=bk[i][0];
            int l=bk[i][1];
            for (int j=i-1; j>=0; j--){
                w+=bk[j][0];
                l=max(l, bk[j][1]);
                if (w<=width)
                    if (j-1>=0)
                        dp[i]=min(dp[i], dp[j-1]+l);
                    else
                        dp[i]=min(dp[i],l);
                else
                    break;
            }
        }
        return dp[len-1];

    }
};
