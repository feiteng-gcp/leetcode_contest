class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int n = books.size();
        
        vector<int> f;
        f.resize(n + 1);
        
        f[0] = 0;
        for (int i = 1; i <= n; i++) {
            int width = 0, maxh = 0, minh = INT_MAX;
            for (int j = i - 1; j >= 0; j--) {
                width += books[j][0];
                maxh = max(maxh, books[j][1]);
                
                if (width <= shelf_width) {
                    minh = min(minh, f[j] + maxh);
                } else
                    break;
            }
            
            f[i] = minh;
        }
        
        return f[n];
    }
};
