class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        vector<int> sol(books.size() + 1, 0);
        for (int i = books.size() - 1; i >= 0; --i) {
            int best = INT_MAX;
            int cur_width = 0;
            int cur_height = 0;
            for (int cnt = 1; i + cnt <= books.size(); ++cnt) {
                cur_width += books[i + cnt - 1][0];
                if (cur_width > shelf_width)
                    break;
                cur_height = std::max(cur_height, books[i + cnt - 1][1]);
                best = std::min(best, cur_height + sol[i + cnt]);
            }
            sol[i] = best;
        }
        return sol[0];
    }
};
