class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        vector<int> height(books.size(), 1000000);
        for (int i=0;i<books.size();i++)
        {
            int last_height = 0;
            if (i)
                last_height = height[i-1];
            int now_width = 0;
            int max_height = 0;
            for (int j=i;j<books.size();j++)
            {
                now_width += books[j][0];
                max_height = max(max_height, books[j][1]);
                if (now_width>shelf_width)
                    break;
                height[j] = min(height[j], last_height+max_height);
            }
        }
        return height[books.size()-1];
    }
};
