class Solution {
public:
    int minHeightShelves(vector<vector<int>>& A, int width) {
        int N = A.size();
        vector<int> f(N+1);
        // f[N-1] = A[N-1][1];
        for (int i = N-1; i >= 0; --i) {
            f[i] = INT_MAX;
            int m = 0; // 宽度
            int h = 0;
            for (int j = i; j < N; ++j) {
                if (m + A[j][0] <= width) {
                    m += A[j][0];
                    h = max(h, A[j][1]);
                    f[i] = min(f[j+1]+h, f[i]);
                } else {
                    break;
                }
            }
        }
        return f[0];
    }
};
