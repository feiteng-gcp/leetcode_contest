class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        
        int n = books.size();
        vector<int> dp(n+1,INT_MAX);
        dp[0] = 0;
        for(int i=0;i<n;i++){
            
            int sum=0;
            int h = 0;
            int j = i;
            while(j>=0){
                sum+=books[j][0];
                h = max(books[j][1],h);
                if(sum>shelf_width) break;
                dp[i+1] = min(dp[i+1],dp[j]+h);
                j--;
            }
        }
        return dp[n];
    }
};
