class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int n = books.size();
        vector<int> f = vector<int> (n + 1, INT_MAX);
        f[0] = 0;
        for (int i = 0; i < n; ++i) {
            int len = 0, maxv = 0;
            for (int j = i; j < n; ++j) {
                maxv = max(maxv, books[j][1]);
                len += books[j][0];
                if (len > shelf_width) break;
                f[j + 1] = min(f[j + 1], f[i] + maxv);
            }
        }
        return f[n];
    }
};
