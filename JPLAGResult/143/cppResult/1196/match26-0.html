class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        vector<int> dp(books.size() + 1, 0);
        dp[1] = books[0][1];

        for(int i = 2; i <= books.size(); ++i) {
            dp[i] = dp[i - 1] + books[i - 1][1];
            int curWidth = books[i - 1][0];
            int curHeight = books[i - 1][1];

            for(int j = i - 1; j >= 1; --j) {
                curWidth += books[j - 1][0];

                if(curWidth > shelf_width) {
                    break;
                }

                curHeight = max(curHeight, books[j - 1][1]);
                dp[i] = min(dp[i], dp[j - 1] + curHeight);                    
            }
        }

        return dp[books.size()];
    }
};
