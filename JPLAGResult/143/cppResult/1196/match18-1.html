class Solution {
public:
    int minHeightShelves(vector<vector<int>>& b, int w) {

        const int n = b.size();
        vector<int> dp(n + 1, INT_MAX);
        dp[0] = 0;
        for (int i = 1; i <= n; ++i) {
            int x = 0, y = 0;
            for (int j = i - 1; j >= 0; --j) {
                x += b[j][0];
                y = max(y, b[j][1]);
                if (x > w)
                    break;
                dp[i] = min(dp[i], dp[j] + y);
            }
        }
        return dp.back();
    }
};
