class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        vector<int> DP;
        for (int i = 0; i < books.size(); ++i) {
          int minTotalHeight = INT_MAX;
          int currLevelHeight = 0;
          int currWidth = 0;
          for (int j = i; j > -1; --j) {
            currWidth += books[j][0];
            if (currWidth > shelf_width) break;
            currLevelHeight = max(currLevelHeight, books[j][1]);
            minTotalHeight = min(minTotalHeight, currLevelHeight + (j - 1 >= 0 ? DP[j - 1] : 0));
          }
          DP.push_back(minTotalHeight);
        }
        return DP.back();
    }
};
