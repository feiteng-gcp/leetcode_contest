const int INF = 1e9 + 5;

class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int n = books.size();
        vector<int> dp(n + 1, INF);
        dp[0] = 0;

        for (int i = 0; i < n; i++) {
            int width = 0, height = 0;

            for (int j = i; j < n; j++) {
                width += books[j][0];
                height = max(height, books[j][1]);

                if (width <= shelf_width)
                    dp[j + 1] = min(dp[j + 1], dp[i] + height);
            }
        }

        return dp[n];
    }
};
