class Solution {
public:
    int minHeightShelves(vector<vector<int>>& a, int W) {
        int n = a.size();
        vector<int> f(n + 1);
        for (int i = 1; i <= n; i++) {
            f[i] = f[i - 1] + a[i - 1][1];

            int sum = 0, mx = 0;
            for (int j = i; j >= 1; j--) {
                sum += a[j - 1][0];
                mx = max(mx, a[j - 1][1]);
                if (sum > W) break;
                f[i] = min(f[i], f[j - 1] + mx);
            }
        }
        return f[n];
    }
};
