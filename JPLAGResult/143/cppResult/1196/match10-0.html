class Solution {
public:
    vector<int> dp;
    int w;
    vector<vector<int>> book;

     int search(int b) {
        if(dp[b] != -1) {
            return dp[b];
        }
        auto& ret = dp[b];
        if(b >= book.size()) return ret = 0;
        ret = 0x7fffffff;
        int ww = 0;
        int h = 0;
        for(int i = b; i < book.size(); i++) {
            ww += book[i][0];
            h = max(h, book[i][1]);
            if(ww > w) break;
            ret = min(ret, h + search(i+1));
           // cout<<"***"<< b<<" "<<i<<" "<<h + search(i+1);
        }
        // cout<<b<<" "<<ret<<endl;
        return ret;
        
    }
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        book = books;
        w = shelf_width;
        dp.resize(books.size() + 1); 
     /*   for(auto& b: books) {
            b.resize(book.size() + 1);
            for(auto& bb: b) {
                bb = -1;
            }
        }*/
        for(auto& d: dp) d = -1;
        return search(0);
    }
};
