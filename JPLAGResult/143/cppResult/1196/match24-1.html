class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
      int n = books.size();
      vector<int> ans(n, -1);
      ans[0] = books[0][1];
      for (int i = 1; i < n; i ++) {
          int now_w = 0, m_height = 0;
          for (int j = i; j >= 0; j --) {
              now_w += books[j][0];
              m_height = max(m_height, books[j][1]);
              if (now_w > shelf_width) break;
              int pp = j > 0 ? ans[j - 1] : 0;
              if (ans[i] == -1 || pp + m_height < ans[i]) ans[i] = pp + m_height;
          }
          //cout << i << " " << ans[i] << endl;
      }
        return ans[n - 1];
    }
};
