class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int n = (int)books.size();
        int dp[n+1];
        dp[n] = 0;
        dp[n-1] = books[n-1][1];
        for(int i=n-2;i>=0;i--){
            int sum = 0, maxval = -1;
            dp[i] = 1000000000;
            for(int j=i;j<n;j++){
                sum += books[j][0];
                if(sum > shelf_width)break;
                maxval = max(maxval,books[j][1]);
                dp[i] = min(dp[i], maxval + dp[j+1]);
            }
        }
        return dp[0];
    }
};
