class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int N = books.size();
        vector<int> dp(N + 1, 100000000);
        dp[0] = 0;
        for (int i = 0; i < N; i++)
        {
            int hm = 0, wt = 0;

            int j = i;
            while (j >= 0)
            {
                hm = max(hm, books[j][1]);
                wt += books[j][0];
                if (wt > shelf_width) break;
                dp[i + 1] = min(dp[i + 1], dp[j] + hm);
                j--;
            }
        }
        return dp.back();
    }
};
