class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
      int n = books.size();
       vector<int> dp(n + 1, 0x3f3f3f3f);
      dp[0] = 0;
      for (int i = 1; i <= n; ++i) {
        int thick = 0, height = 0;
        for (int j = i; j >= 1; --j) {
          height = max(height, books[j - 1][1]);
          thick += books[j - 1][0];
          if (thick > shelf_width) {
            break;
          }
          dp[i] = min(dp[i], dp[j - 1] + height);
        }
      }
      return dp[n];
    }
};
