class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int w) {
        int n = books.size();
        vector<int> dp(n + 1, 1000000);
        dp[0] = 0;
        for (int i = 1; i <= n; ++i) {

            int sum = 0;
            int maxh = books[i - 1][1];
            for (int j = i - 1; j >= 0; --j) {
                maxh = max(maxh, books[j][1]);
                sum += books[j][0];
                if (sum > w) break;
                
                dp[i] = min(dp[i], dp[j] + maxh);
            }
        }
        return dp[n];
    }
};
