class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int n = books.size();
        vector<int> ans(n);
        
        ans[0] = books[0][1];
        for(int i = 1; i < n; i ++){
            ans[i] = books[i][1] + ans[i - 1];
            int len = books[i][0];
            int maxHeight = books[i][1];
            for(int j = i - 1; j >= 0; j --){
                len += books[j][0];
                if(len > shelf_width) break;
                
                maxHeight = max(maxHeight, books[j][1]);
                int newHeight = maxHeight + (j ? ans[j - 1] : 0);
                ans[i] = min(ans[i], newHeight);
            }
        }
        
        return ans.back();
    }
};
