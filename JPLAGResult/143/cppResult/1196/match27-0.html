class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int len = books.size();
        vector<int> dp(len+1, 0);
        for(int i=1; i<=len; i++){
            // 当前书 另起一行
            dp[i] = dp[i-1] + books[i-1][1];
            int sum = books[i-1][0];
            int iMax = books[i-1][1];
            for(int j=i-1; j>=1 && sum <= shelf_width; j--){
                sum += books[j-1][0];
                iMax = max(iMax, books[j-1][1]);
                if(sum <= shelf_width) dp[i] = min(dp[i], dp[j-1] + iMax);
            }
        }
        return dp[len];
    }
};
