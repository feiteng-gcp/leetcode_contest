class Solution {
    typedef pair<int, int> pii;
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int n = books.size();
        vector<int> ret(n+1, INT_MAX);
        ret[0] = 0;
        for (int i = 0; i < n; i++) {
            int room = shelf_width, ma = 0;
            for (int j = i; j < n; j++) {
                room -= books[j][0];
                if (room < 0) break;
                ma = max(ma, books[j][1]);
                ret[j+1] = min(ret[i] + ma, ret[j+1]);
            }
        }
        return ret[n];
        
    }
};
