class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int n = books.size();
        
        vector<int> dp(n + 1);
        dp[0] = 0;
        
        for (int i = 1; i <= n; i++) {
            int thick = 0, height = 0, best = INT_MAX;
            for (int j = i - 1; j >= 0; j--) {
                thick += books[j][0];
                if (thick > shelf_width) break;
                height = max(height, books[j][1]);
                
                best = min(best, dp[j] + height);
            }
            dp[i] = best;
        }
        
        return dp[n];
    }
};
