class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int dp[1010];
        for(int i=0;i<1010;i++)
            dp[i] = 99999999;
        dp[0]=0;
        int n = books.size();
        for(int i=1;i<=n;i++){
            int w = books[i-1][0];
            int h = books[i-1][1];

            dp[i] = dp[i-1]+h;
            int totalw = w;
            int totalh = h;
            for(int j=i-1;j>=1;j--){
                totalw+=books[j-1][0];
                totalh = max(totalh, books[j-1][1]);
                if(totalw>shelf_width)
                    break;
                dp[i] = min(dp[i], dp[j-1]+totalh);
            }
        }
        return dp[n];
    }
};
