class Solution {
public:
  int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
    int N = books.size();
    int W = shelf_width;
    vector<int> dp(N + 1, 0x3f3f3f3f);
    dp[0] = 0;
    for (int i = 1; i <= N; i++) {
      int w = 0;
      int h = 0;
      for (int j = i; j >= 1; j--) {
        w += books[j - 1][0];
        h = max(h, books[j - 1][1]);
        if (w <= W)  dp[i] = min(dp[i], dp[j-1] + h);
      }
    }
    return dp.back();
  }
};
