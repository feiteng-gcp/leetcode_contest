class Solution {
public:
    int minHeightShelves(vector<vector<int>>& books, int shelf_width) {
        int n = books.size();
        vector<int>f=vector<int>(n+1, -1);
        f[0]=0;
        for(int i=1;i<=n;i++){
            int sum = 0;
            int mx = 0;
            for(int j=i;j>=1;j--){
                sum+=books[j-1][0];
                mx = max(mx, books[j-1][1]);
                if(sum>shelf_width)break;
                if(f[i]==-1||f[i]>f[j-1]+mx){
                    f[i]=f[j-1]+mx;
                }
            }
        }
        return f[n];
    }
};
