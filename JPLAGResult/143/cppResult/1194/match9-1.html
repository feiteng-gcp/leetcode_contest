class Solution {
public:
    vector<int> pathInZigZagTree(int n) {
        int l = 1, r = 0;
		while (l * 2 <= n)
		{
			l *= 2;
			r++;
		}
		l /= 2;

		vector<int> ans;
		ans.push_back(n);

		while (n != 1)
		{
			int fa = n / 2;
			int tot = l + l * 2 - 1;
			fa = tot - fa;
			ans.push_back(fa);
			n = fa;
			l /= 2;
			r--;
		}
		reverse(ans.begin(), ans.end());
		return ans;
    }
};
