class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> path;
        int level = log2(label) + 1;
        while (level > 1) {
            path.push_back(label);
            level--;
            label = pow(2, level) - 1 - label / 2 + pow(2, level - 1);
        }
        path.push_back(label);
        reverse(path.begin(), path.end());
        return path;
    }
};
