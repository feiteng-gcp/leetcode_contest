class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> ans;
        int x = 0, t = label;
        while (t) {
            x++;
            t = t >> 1;
        }
        x--;
        cout << x << endl;
        t = label;
        while (t) {
            ans.push_back(t);
            t = pow(2, x) + pow(2, x + 1) - 1 - t;
            t = t >> 1;
            x--;
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
