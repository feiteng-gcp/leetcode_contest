class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        int level=log2(label);
        vector<int> res;

        while(level>=0){
            res.push_back(label);
            int origin=label/2;
            int next_min=pow(2,level-1);
            int next_max=next_min*2-1;
            label=next_min+next_max-origin;
            level--;
        }
        reverse(res.begin(),res.end());
        return res;
    }
};
