class Solution {
public:
    
    
    vector<int> pathInZigZagTree(int label) {
        vector<int>A;
        while(label>=1)
        {
            A.push_back(label);
            label/=2;
        }
        for(int i=1;i<A.size();i+=2)
        {
            int x = log2(A[i])+1;
            int k = (1<<x) - A[i];
            int r = (1<<(x-1)) + k - 1;
            A[i] = r;
        }
        reverse(A.begin(), A.end());
        return A;
    }
};
