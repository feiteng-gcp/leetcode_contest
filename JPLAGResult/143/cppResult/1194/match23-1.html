class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        int level = 0;
        int copy = label;
        while(copy>1){
            level++;
            copy = copy / 2;
        }
        vector<int> res;
        bool flag = true;
        while(label >= 1){
            if(flag){
                res.push_back(label);
            }
            else{
                res.push_back(3 * (1 << level) - 1 - label);
            }
            label /= 2;
            flag^= true;
            level--;
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
