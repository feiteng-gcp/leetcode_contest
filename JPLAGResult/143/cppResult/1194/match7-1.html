class Solution {
public:
    int l(int label) {
        for(int i = 0; i < 32; i++) {
            if(label >= (1<<i)  && label < (1<<(i+1)) ) return i;
        }
        return -1;
    }
    vector<int> pathInZigZagTree(int label) {
        int last = label;
        vector<int> ret;        
        loop:
        if(label == 1) {
            ret.push_back(label);
            reverse(ret.begin(), ret.end());
            return ret;
        } 
        if(last == label) ret.push_back(label);
        int level = l(label);
        if(last != label) {
            if(level % 2 == 0) ret.push_back(label);
            else {
                label = (1 << level) + (1<<(level+1)) - 1 - label;
                ret.push_back(label);
            }
        }
        if(level % 2 == 0) {
            label = label / 2 ;
        } else {
            label = (1 << level) + (1<<(level+1)) - 1 - label;
            label = label / 2 ;
        }
        goto loop;
        return ret;
    }
};
