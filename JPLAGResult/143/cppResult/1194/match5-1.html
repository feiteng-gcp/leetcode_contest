class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> ans;
        
        while (label) {
            ans.push_back(label);
            label >>= 1;
        }
        reverse(ans.begin(), ans.end());
        
        int n = ans.size();

        bool v = false;
        for (int i = n - 1; i >= 0; i--) {
            if (v) {
                ans[i] = (1 << i + 1) + (1 << i) - 1 - ans[i];
            }
            v = !v;
        }
        
        return ans;
    }
};
