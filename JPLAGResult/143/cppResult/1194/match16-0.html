class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector <int> res;
        while (label)
        {
            res.push_back(label);
            label /= 2;
        }
        reverse (res.begin(), res.end());
        for (int i = 0; i < res.size(); i++)
            if (i > 0 && (i + res.size()) % 2 == 0)
                res[i] = res[i] ^ ((1 << (i)) - 1);
        return res;
    }
};
