class Solution {
public:
    int qmi(int m, int k)
    {
        long long res = 1, t = m;
        while (k)
        {
            if (k&1) res = res * t;
            t = t * t;
            k >>= 1;
        }
        return res;
    }
    

    vector<int> pathInZigZagTree(int l) {
        vector<int> ans;
        
        while (l > 0)
        {
            ans.push_back(l);
            l /= 2;
        }
        
        reverse(ans.begin(), ans.end());
        int m = ans.size();
        
        for (int i = m-2; i >= 0; i -= 2)
        {
            ans[i] = 3*qmi(2,i) - ans[i] - 1;
        }
        
        return ans;
    }
};
