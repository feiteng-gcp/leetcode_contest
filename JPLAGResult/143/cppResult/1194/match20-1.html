class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        int levelEnd = 2;
        int level = 1;
        while (label >= levelEnd) {
            level++;
            levelEnd <<= 1;
        }
        vector<int> result(level, 0);
        for (int i = level - 1; i >= 0; i--) {
            result[i] = label;
            label = label / 2;
            levelEnd >>= 1;
            label = levelEnd - 1 - label + levelEnd / 2;
        }
        return result;
    }
};
