class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> ans;
        int lvl = log2(label) - 1;
        while (label != 1) {
            ans.push_back(label);
            int tmp = label / 2;
            int start = pow(2, lvl);
            int end = start * 2 - 1;
            label = start + end - tmp;
            lvl--;
        }
        ans.push_back(1);
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
