class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> path;
        int cur = 1;
        while (cur <= label) {
            cur *= 2;
        }
        cur /= 2;
        int sum = cur + (cur / 2) - 1;
        while (label != 1) {
            path.push_back(label);
            label = sum - (label / 2);
            sum = (sum + 1) / 2 - 1;
        }
        path.push_back(1);
        reverse(path.begin(), path.end());
        return path;
    }
};
