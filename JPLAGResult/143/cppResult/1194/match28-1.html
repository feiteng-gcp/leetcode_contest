class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> result;
        int l = label;
        while (l) {
            result.push_back(l);
            l /= 2;
        }
        reverse(result.begin(), result.end());
        for (size_t i = result.size() % 2; i < result.size(); i += 2) {
            int begin = 1 << i;
            int end = begin * 2 - 1;
            result[i] = end - (result[i] - begin);
        }
        return result;
    }
};
