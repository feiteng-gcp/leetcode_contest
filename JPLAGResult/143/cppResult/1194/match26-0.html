class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        int n=label;
        vector<int> ans;
        while(n>1)
        {
            ans.push_back(n);
            int k=1;
            while(2*k<=n) k<<=1;
            n=k-(n-k)/2-1;
        }
        ans.push_back(1);
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
