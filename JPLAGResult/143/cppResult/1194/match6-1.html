class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> res;
        while (label > 0) {
            res.push_back(label);
            label /= 2;
        }
        reverse(res.begin(), res.end());
        int cur = 1;
        for (int i = 0; i < res.size(); i++) {
            // cout << res[i] << endl;
            if (i % 2 == res.size() % 2) {
                res[i] = 3ll * cur - res[i] - 1;
            }
            cur *= 2;
        }
        return res;
    }
};
