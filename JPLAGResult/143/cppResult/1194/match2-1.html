class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        if(label==1)
            return {1};
        vector<int> path;
        path.push_back(label);
        while(label>0)
        {
            label/=2;
            if(label<2)
                break;
            int sum=help(label);
            if(sum==1)
                break;
            path.push_back(label=3*sum-1-label);
        }
        path.push_back(1);
        reverse(path.begin(),path.end());
        return path;
    }
    int help(int node)
    {
        int level=0;
        while(node>1)
            node/=2,level++;
        int sum=1;
        while(level--)
            sum=2*sum;
        return sum;
    }
};
