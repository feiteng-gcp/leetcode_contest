class Solution {
public:
    vector<int> pathInZigZagTree(int l) {
        vector<int> ret;
        while(l>0)
        {
           // cout<<l<<endl;
            ret.emplace_back(l);
            l=gr(l);
        }
        reverse(ret.begin(),ret.end());
        return ret;
    }
    int gr(int l)
    {
        int t=l/2;
        int s2=0;
        while(l>0)
        {
            l/=2;
            s2++;//5
        }
        int b=pow(2,s2-2);
        int e=pow(2,s2-1);
        int pos=t-b+1;
        return e-pos;
        
    }
};
