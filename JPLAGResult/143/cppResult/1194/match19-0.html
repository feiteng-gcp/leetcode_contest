class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        int order = 0;
        int cur = 1;
        while(cur < label){
            cur = cur*2+1;
            order++;
        }
        vector<int> res;
        int base = pow(2, order);
        while(label > 1){
            res.push_back(label);
            int sum = 3*base - 1;
            label = (sum - label)/2;
            base = base/2;
        }
        res.push_back(1);
        reverse(res.begin(), res.end());
        return res;
    }
};
