class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> res;
        int n = label;
        while (n) {
            res.push_back(n);
            n >>= 1;
        }
        reverse(res.begin(), res.end());
        int m = 1;
        for (int i = 0; i < res.size(); i++) {
            m *= 2;
            if (i % 2 != res.size() % 2) continue;
            int t = res[i];
            res[i] = (m - 1 - t) + m / 2; 
        }
        return res;
    }
};
