class Solution {
public:
    int norm(int n) {
        int x = 1;
        while (x * 2 <= n) {
            x *= 2;
        }
        return x + (x * 2 - n - 1);
    }
    vector<int> pathInZigZagTree(int label) {
        vector<int> ret;
        while (label != 1) {
            ret.push_back(label);
            label = norm(label / 2);
        }
        ret.push_back(1);
        sort(ret.begin(), ret.end());
        return ret;
    }
};
