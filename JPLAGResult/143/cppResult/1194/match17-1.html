class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> res;
        
        label = Convert(label);
        
        while (label) {
            res.emplace_back(Convert(label));
            label /= 2;
        }
        
        reverse(res.begin(), res.end());
        return res;
    }
    
    int Convert(int label) {
        int level = log2(label) + 1;
        if (level % 2 == 1) return label;
        return (pow(2, level) - 1 + pow(2, level - 1)) - label;
    }
};
