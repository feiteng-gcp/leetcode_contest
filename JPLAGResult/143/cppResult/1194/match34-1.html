class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        if (label == 1) {
            return {1};
        }

        vector<int> res;
        int count_level = 1;
        int cur_max = 1;
        while (cur_max < label) {
            cur_max = (cur_max << 1) + 1;
            count_level++;
        }
        if (count_level % 2 == 0) {
            label = shift(label, count_level);
        } 
        
        while (label) {
            res.push_back(label);
            label = label / 2;
        }
        for (int i = 0, j = res.size() - 1; i < j; i++, j--) {
            swap(res[i], res[j]);
        }
        
        for (int i = 0; i < res.size(); i++) {
            if (i % 2 != 0) {
                res[i] = shift(res[i], i + 1);
            }
        }
        return res;
    }
    
    int shift(int label, int level) {
        if (level < 1) {
            return label;
        }
        return (1 << level) - 1 - label + (1 << (level - 1));
    }
};
