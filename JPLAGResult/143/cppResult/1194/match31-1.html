class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> ans;
        while (label >= 1) {
            ans.push_back(label);
            label /= 2;
        }
        for (int n : ans) {
            cout << n << endl;
        }
        reverse(ans.begin(), ans.end());
        for (int i = ans.size() - 2; i > 0; i -= 2) {
            int left = pow(2, i), right = left * 2 - 1;
            int diff = right - ans[i];
            ans[i] = left + diff;
        }
        return ans;
    }
};
