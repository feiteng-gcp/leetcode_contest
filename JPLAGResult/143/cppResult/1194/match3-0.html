class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        bool flag = false;
        long long left = 1, right = 1;
        int labeltmp = label;
        while (true) {
            if (label >= left && label <= right) {
                if (flag == true) {
                    labeltmp = right - label + left;
                } else {
                    labeltmp = label;
                }
                break;
            }
            left = right + 1;
            right = right * 2 + 1;
            flag = !flag;
        }
        cout << labeltmp << endl;
        vector<int> ans;
        while (labeltmp) {
            ans.push_back(labeltmp);
            labeltmp /= 2;
        }
        reverse(ans.begin(), ans.end());
        for (auto s : ans) {
            cout << s << " ";
        }
        cout << endl;
        int len = ans.size();
        for (int i = 1; i < len; i += 2) {
            long long left = 1LL << i;
            long long right = (1LL << (i + 1)) - 1;
            ans[i] = right - ans[i] + left;
        }
        for (auto s : ans) {
            cout << s << " ";
        }
        cout << endl;
        return ans;
    }
};
