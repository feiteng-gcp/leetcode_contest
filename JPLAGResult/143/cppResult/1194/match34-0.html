class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> res;
        int count = 0;
        int tmp = label;
        while (tmp) {
            ++count;
            tmp /= 2;
        }
        if (count % 2 == 0) {
            label = (1 << (count - 1)) + (1 << count) - 1 - label;
        }
        while (label) {
            res.push_back(label);
            label /= 2;
        }
        reverse(res.begin(), res.end());
        for (int i = 1; i < res.size(); i += 2) {
            res[i] = (1 << i) + (1 << (i + 1)) - 1 - res[i];
        }
        return res;
    }
};
