class Solution {
public:
    vector<int> pathInZigZagTree(int label) {
        vector<int> res;
        while(label != 0){
            res.push_back(label);
            label /= 2;
        }
        reverse(res.begin(), res.end());
        
        for(int i=res.size()-2; i>=1; i -= 2)
            res[i] = 3 * (1 << i) - 1 - res[i];
        return res;
    }
};
