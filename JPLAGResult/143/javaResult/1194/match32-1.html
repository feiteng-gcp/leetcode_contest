class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        Stack<Integer> stack = new Stack<>();
        while (label > 0) {
            stack.push(label);
            int level = (int) (Math.log(label) / Math.log(2));
            int from = 1 << level,to = (1 << (level + 1)) - 1;
            System.out.println("label = " + label + " level = " + level + " from = " + from + " to = " + to);
            int ith = label - from;

            if (level % 2 == 1){
                ith = to - label;

            }
            int upth = ith / 2;
            if(level % 2 == 0){
                label = from - 1 - upth;
            }else{
                label = upth + (1 << (level -1));
            }
            System.out.println("label = " + label + " upth = " + upth);
        }
        List<Integer> ret = new ArrayList<>();
        while (!stack.isEmpty()) ret.add(stack.pop());
        return ret;
    }
}
