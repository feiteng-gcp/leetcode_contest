class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        int[] arr=new int[label];
        List<Integer> list = new ArrayList<>();
        for(int i=1,s=1;list.size()<label;i*=2,s++){
            List<Integer> tmp = new ArrayList<>();
            for(int j=1;j<=i;j++){
                tmp.add(j+list.size());
            }
            if(s%2==0) Collections.reverse(tmp);
            list.addAll(tmp);
        }
        List<Integer> ans = new ArrayList<>();
        int i=list.size()-1;
        while(list.get(i)!=label){
            i--;
        }
        for(;i>0;i=(i-1)/2){
            ans.add(list.get(i));
        }
        ans.add(1);
        Collections.reverse(ans);
        return ans;
    }
}
