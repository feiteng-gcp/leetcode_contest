class Solution {
    public List<Integer> pathInZigZagTree(int label) {

        List<Integer> from = new ArrayList<>();
        List<Integer> end = new ArrayList<>();
        
        int level = 0;
        int sum = 0;

        while (sum < label) {
            int tmp = (int)Math.pow(2.0, level);    
            
            from.add(sum + 1);
            end.add(sum + tmp);
            // System.out.println(from.get(level) +  " " + end.get(level));
            sum += tmp;
            level += 1;
            
            
        }
        level -= 1;
        // System.out.println(level);
        List<Integer> res = new ArrayList<>();
        
        
while (label >= 1 && level >= 0) {
            res.add(0, label);    
            int d = end.get(level) - label;
            int newL = from.get(level) + d;
            label = newL / 2;
            --level;           
        }
        
        return res;
    }
}
