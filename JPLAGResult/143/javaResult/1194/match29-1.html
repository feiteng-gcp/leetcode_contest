class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        
int curMax = 2;
        while(curMax <= label) {
            curMax = curMax * 2;
        }
        
        List<Integer> res = new ArrayList<Integer>();
        res.add(label);
        while (label > 0) {
            curMax /= 2;
            label /= 2;
            label = curMax -1 + curMax / 2 - label;
            if(label < 1) {
                break;
            }
            res.add(label);
        }
        
        Collections.reverse(res);
        return res;
    }
}
