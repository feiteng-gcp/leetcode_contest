class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> list = new ArrayList<>();
        int j = 0;
        while (label != 0) {
            list.add(label);
            label /= 2;
            j++;
        }
        List<Integer> res = new ArrayList<>();
        for (int i = list.size() - 1, k = 1; i > -1; i--, j--, k <<= 1) {
            int l = list.get(i);
            if (j % 2 == 1) {
                res.add(l);
            }
            else {
                res.add(k | ((k - 1) & (~l)));
            }
        }
        return res;
    }
}
