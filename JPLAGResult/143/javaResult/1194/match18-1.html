class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> res = new ArrayList<>();
    
        while(label != 0){
            res.add(label);
            label = findParent(label);
        }
        Collections.reverse(res);
        return res;
    }
    
    private int findParent(int label){
        int res = label % 2 == 0 ? label / 2 : (label - 1) / 2;
        res = midNum(res) - res;
        return res;
    }
    
    private int midNum(int res){
        int level = 0;
        while(res != 0){
            res /= 2;
            level++;
        }
        return (int)Math.pow(2, level) + (int)Math.pow(2, level - 1) - 1;
    }
    
}
