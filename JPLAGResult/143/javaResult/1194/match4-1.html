class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> ans = new ArrayList<>();
        while (label > 0) {
            ans.add(0, label);
            label /= 2;
            int fg = 1;
            while (fg <= label)
                fg<<=1;
            label = fg-label+fg/2-1;
        }
        return ans;
    }
}
