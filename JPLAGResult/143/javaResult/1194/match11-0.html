class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> theList = new ArrayList<Integer>();
        int total = 0, row = 0;
        while (total < label) {
            total += Math.pow(2, row);
            row++;
        }
        for(int i = row; i >= 1; i--) {
            theList.add(0, label);
            int sum = (int)Math.pow(2, i) + (int)Math.pow(2, i - 1) - 1;
            label = (sum - label) / 2;
        }
        return theList;
    }
}
