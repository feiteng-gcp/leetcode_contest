import java.util.ArrayList;
import java.util.List;

import static java.lang.Math.pow;

class Solution {
    public static int M = 1000010;
    public static int[] as = new int[M];
    public static void init(){
        int i = 0;
        int ln = 1;
        int a = 1;
        while(i < M){
            int n = (int)pow(2, ln-1);
            int ma = a + n - 1;
            if( ln % 2 == 1){
                for(; i<M && a<=ma; i++){
                    as[i] = a;
                    a++;
                }
            }else{
                int b = ma;
                for(; i<M && b>=a; i++){
                    as[i] = b;
                    b--;
                }
            }
            a = ma + 1;
            ln++;
        }
    }

    public List<Integer> pathInZigZagTree(int label) {
        if(as[0] == 0) init();
        
        int i = 0;
        for(i=0; i<M; i++){
            if(label == Solution.as[i]) break;
        }
        ArrayList<Integer> res = new ArrayList<>();
        int a = i;
        while(a>0){
            res.add(Solution.as[a]);
            a = (a-1)/2;
        }
        res.add(1);
        int j = res.size()-1; i=0;
        while(i <= j){
            int t = res.get(i);
            res.set(i, res.get(j));
            res.set(j, t);
            i++; j--;
            
        }
        return res;

    }
}
