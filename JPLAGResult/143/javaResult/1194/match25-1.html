class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> ans=new LinkedList<>();
        while(label>=1){
            ans.add(0,label);
            label/=2;
        }
        for(int i=ans.size()-2;i>=1;i-=2){
            int num=ans.get(i);
            int newNum=((int)Math.pow(2,i))+((int)Math.pow(2,i+1))-num-1;
            ans.set(i,newNum);
        }
        return ans;
    }
}
