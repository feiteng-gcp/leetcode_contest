class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        int root = 0;
        
        for (int l = label; l > 0; l >>= 1) {
            ++root;
        
}
        
        // System.out.println(root);
        // System.out.println();
        
        LinkedList<Integer> ret = new LinkedList<>();
        int curr = label;
        
        while (root > 0) {
            ret.addFirst(curr);
            curr = (((1 << root) + (1 << (root - 1)) - 1) - curr) / 2;
            --root;
        }
        
        return ret;
    }
}
