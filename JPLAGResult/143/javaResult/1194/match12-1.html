class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        LinkedList<Integer> res = new LinkedList<>();
        int level = (int)(Math.log(label)/Math.log(2));
        for(int cur=label, n=level-1; cur>0; n--){
            res.addFirst(cur);
            int sum = (int)(Math.pow(2, n)+Math.pow(2, n+1)-1);
            cur = sum - cur/2;
        }
        return res;
    }
}
