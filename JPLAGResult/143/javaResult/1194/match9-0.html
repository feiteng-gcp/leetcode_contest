class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        Stack<Integer> s = new Stack<>();
        int v = 1;
        int dep = 0;
        while (v<=label) {v*=2;dep++;}
        if (dep%2==0) label = v+v/2-1-label;
        while (dep>0){
            int num = label;
            if (dep%2==0) num = v+v/2-1-label;
            s.push(num);
            dep--;
            label/=2;
            v/=2;
        }
        List<Integer> l = new ArrayList<>();
        while (!s.isEmpty()) l.add(s.pop());
        return l;
    }
}
