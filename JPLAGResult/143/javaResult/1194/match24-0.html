class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> result = new ArrayList<>();
        
        while(label > 0) {
            result.add(0, label);
            label /= 2;
        }
        
        int levelSum = 5;
        
        for(int i = 1; i < result.size(); i ++) {
            if(i % 2 == result.size() % 2) {
                result.set(i, levelSum - result.get(i));
            }
            
            levelSum = levelSum * 2 + 1;
        }
        
        return result;
    }
}
