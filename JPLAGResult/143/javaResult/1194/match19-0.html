class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        int tree[] = new int[1 << 20];
        int idx = 1;
        int size = 1;
        boolean fwd = true;
        while (idx < tree.length) {
            if (fwd) {
                int start = idx + size - 1;
                int end = idx;
                for (int i = start; i >= end; --i) {
                    tree[i] = idx++;
                }
            } else {
                int start = idx;
                int end = idx + size - 1;
                for (int i = start; i <= end; ++i) {
                    tree[i] = idx++;
                }
            }
            size <<= 1;
            fwd = !fwd;
        }
        int[] parent = new int[tree.length];
        for (int i = 1; 2 * i < tree.length; ++i) {
            parent[tree[2 * i + 1]] = tree[i];
            parent[tree[2 * i]] = tree[i];
        }
        ArrayList<Integer> ans = new ArrayList<>();
        int x = label;
        while (x > 0) {
            ans.add(x);
            x = parent[x];
        }
        Collections.reverse(ans);
        return ans;
    }
