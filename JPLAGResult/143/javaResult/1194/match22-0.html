class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        int level = (int) (Math.log(label) / Math.log(2)) + 1;
        List<Integer> ans = new ArrayList<>();
        int cur = label;
        while (level > 0) {
            ans.add(0, cur);
            int low = (int) Math.pow(2, level - 1);
            int high = (int) Math.pow(2, level) - 1;
            cur = (low + high - cur) / 2;
            level--;
        }
        return ans;
    }
}
