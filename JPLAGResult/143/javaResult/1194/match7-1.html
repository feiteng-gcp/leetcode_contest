class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> res = new ArrayList<>();
        int temp = label;
        int ceng = 0;        
        while(temp!=0) {
        	temp /=2;
        	ceng++;
        }
        ceng--;
        while(label!=0) {
        	res.add(0, label);
        	label= (int)Math.pow(2, ceng--)+(int)Math.pow(2, ceng)-1-label/2;
        }        
        
        return res;
    }
}
