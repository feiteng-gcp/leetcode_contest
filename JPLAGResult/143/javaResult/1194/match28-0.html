class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> reversed = new ArrayList<>();
        int level = 0, power2 = 1;
        while (power2 <= label) {
            level ++;
            power2 *= 2;
        }
        level --;
        power2 /= 2;
        
        while (level >= 0) {
            reversed.add(label);
            if (level % 2 == 0) {
                label = power2 - 1 - (label / 2 - power2 / 2);
            }
            else {
                label = (power2 + (power2 * 2 - 1 - label))/2;
            }
            level --;
            power2 /= 2;
        }
        List<Integer> res = new ArrayList<>();
        for (int idx = reversed.size() - 1; idx >= 0; idx --)
            res.add(reversed.get(idx));
        return res;
    }
}
