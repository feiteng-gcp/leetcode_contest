class Solution {
    public List<Integer> pathInZigZagTree(int l) {
        int i = 1;
        int n = 0;
        int temp = 0;
        while(temp + i < l) {
            temp += i;
            i = i << 1;
            ++n;
        }
        LinkedList<Integer> ans = new LinkedList<>();
        ans.addFirst(l);
        while(n > 0) {
            int k = (l - temp - 1) / 2;
            
            // if(n & 1 == 0) {
            //     k = (i - (l - temp)) / 2;
            // } else {
            //     k = (l -temp - 1) / 2;
            // }
            l = temp - k;
            i = i >> 1;
            temp -= i;
            ans.addFirst(l);
            --n;
        }
        return ans;
    }
}
