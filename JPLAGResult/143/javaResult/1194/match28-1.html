class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        int h = 0;
        while ((1 << h) <= label) {
            h++;
        }
        
        LinkedList<Integer> res = new LinkedList<>();
        
        
while (label > 0) {
            res.add(0, label);
            //System.out.println(h + " " + label);
            if (h % 2 == 0) {
                //label = ((1 << h) - label + ((1 << (h-1)) - 1)) / 2;
                 label = ((1 << h) + (1 << (h-1)) - 2 - label + 1) / 2;
            } else {
                //label = ((1 << h) - (label - (1 << h)) / 2 - 1);
                label = ((1 << (h-2)) + (1 << (h-1)) - 2 - (label / 2 - 1));
            }
            h--;
        }
        
        //if (res.get(0) != 1) res.add(0, 1);
        return res;
    }
}
