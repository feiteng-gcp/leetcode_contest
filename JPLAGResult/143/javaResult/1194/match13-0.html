class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> res = new ArrayList<>();
        int level = getLevel(label);
        res.add(label);
        
        for (; level > 1; --level) {
            int parent = getParent(label, level);
            res.add(0, parent);
            label = parent;
        }
        
        return res;
    }
    
    private int getLevel(int label) {
        return (int) (Math.log(label) / Math.log(2)) + 1;
    }
    
    private int getParent(int label, int level) {
        if (label == 1)
            return -1;

        int leadingNode = (int) Math.pow(2, level - 1);
        return leadingNode - 1 - (label - leadingNode) / 2;
    }
}
