class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> sol = new ArrayList<>();
        while (label != 1) {
            sol.add(0, label);
            int next = label / 2;
            int st = find(next);
            int ed = st * 2 - 1;
            next = st + ed - next;
            label = next;
        }
        sol.add(0, 1);
        return sol;
    }
    
    public int find(int n) {
        int sol = 1;
        while (sol <= n) sol *= 2;
        return sol / 2;
    }
}
