class Solution {
    public List<Integer> pathInZigZagTree(int label) {
		List<Integer> res = new ArrayList<>();
		while (label > 0) {
			res.add(label);
			label /= 2;
		}
		int row = 1;
		for (int i = 0; i < res.size(); i++) {
			if (i % 2 == 0) {
				continue;
			}
			int exp = res.size() - i;
			int thisRowMax = (int) Math.pow(2, exp) - 1;
			int thisRowMin = (int) Math.pow(2, exp - 1);
			int element = res.get(i);
			element = thisRowMin + (thisRowMax - element);
			res.set(i, element);
		}
		List<Integer> res2 = new ArrayList<>();
		for (int i = res.size() - 1; i >= 0; i--) {
			res2.add(res.get(i));
		}
		return res2;
    }
}
