class Solution {
    public List<Integer> pathInZigZagTree(int L) {
        LinkedList<Integer> res = new LinkedList<>();
        int curr = 0, base = 2;
        while (base <= L) {
            base *= 2;
            curr++;
        }
        
        System.out.println(curr);
        for (int i = curr; i >= 0; i--) {
            res.addFirst(L);
            int b = (int) Math.pow(2.0, (double) i);
            
            if (i % 2 == 0) {
                int idx = (2 * b - 1 - L) / 2;
                L = b / 2 + idx;
                // l to r
            } else {
                int idx = (L - b) / 2;
                L = b - 1 - idx;
            }
        }
        return res;
    }
}
