class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> result = new ArrayList<>();
        while (label != 1) {
            result.add(0, label);
            int row = (int) (Math.log(label) / Math.log(2));
            label = (int) (Math.pow(2, row) - 1) - ((label / 2) - (int) Math.pow(2, row - 1));
        }
        result.add(0, label);
        return result;
    }
}
