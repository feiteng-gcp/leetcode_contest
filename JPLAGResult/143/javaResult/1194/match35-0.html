class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        int[] p = new int[label + 1];
        int level = 1, nextLevel = 4, levelNodes = 2, rangeSt = 2, rangeEnd = 3, prevRangeSt = 1, prevRangeEnd = 1;
        for (int i = 2; i <= label; ++i) {
            // int level = (int)(Math.log(i) / (double)Math.log(2.0)); 
            if (i >= nextLevel) {
                ++level;
                nextLevel *= 2;
                levelNodes *= 2;
                prevRangeSt = rangeSt;
                prevRangeEnd = rangeEnd;
                rangeSt = (int)Math.pow(2, level);
                rangeEnd = rangeSt + levelNodes - 1;
                // System.out.println("Next level: " + i + " " + levelNodes + " " + rangeSt + " " + rangeEnd);
            }
            int parentShift = i / 2 - prevRangeSt;
            p[i] = prevRangeEnd - parentShift;
        }
        // System.out.println(Arrays.toString(p));
        List<Integer> res = new ArrayList<>();
        int cur = label;
        while (cur > 0) {
            res.add(cur);
            cur = p[cur];
        }
        Collections.reverse(res);
        
        return res;
    }
}
