class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> res = new Stack<>();
        while(label != 0) {
            res.add(0, label);
            label /= 2;
        }
        int cur = res.size();
        while(cur > 1) {
            int t = res.get(cur - 1);
            boolean f = true;
            if(cur % 2 == 0) f = false;
            int l = 1 << (cur - 1);
            int r = (1 << cur) - 1;
            if(f) {
                int tp = t / 2;
                int ll = 1 << (cur - 2);
                int rr = (1 << (cur - 1)) - 1;
                res.set(cur - 2, rr - (tp - ll));               
            } else {
                int tp = (r - (t - l)) / 2;
                res.set(cur - 2, tp);
            }
            --cur;
        }
        return res;
    }
}
