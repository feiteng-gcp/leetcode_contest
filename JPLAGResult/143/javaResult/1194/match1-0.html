class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> res = new ArrayList();
        while (label >= 1) {
            res.add(0, label);
            label /= 2;
        }
        for (int i = res.size() - 2; i > 0; i -= 2) {
            int newVal = (int) (Math.pow(2, i) + Math.pow(2, i + 1) - 1 - res.get(i));
            res.set(i, newVal);
        }
        return res;
    }
}
