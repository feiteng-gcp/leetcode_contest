class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        LinkedList<Integer> res = new LinkedList<>();
        int level = 0, num = label;
        while (num > 0) {
            num /= 2;
            level++;
        }
        // System.out.println(level);
        while (level > 0) {
            res.addFirst(label);
            level--;
            label /= 2;
            int levelSum = getSum(level);
            label = levelSum - label;
        }
        return res;
    }
    
    private int getSum(int level) {
        return (int)(Math.pow(2, level - 1) + Math.pow(2, level) - 1);
    }
}
