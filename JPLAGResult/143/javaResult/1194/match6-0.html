class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> list = new ArrayList<>();
        int index = 0;
        while(Math.pow(2,index)<=label){
            index++;      
        }
        while(label>0){
            list.add(0,label);
            index--;
            label/=2;         
            label = (int)Math.pow(2,index-1)+(int)(Math.pow(2,index)-1-label);
        }
        return list;
        
    }
}
