class Solution {
    public List<Integer> pathInZigZagTree(int label) {
        List<Integer> result = new ArrayList<>();
        int n = 1;
        while (Math.pow(2, n) - 1 < label) {
            n++;
        }
        
        int opposite = (int) ((Math.pow(2, n) - 1) - label + (Math.pow(2, n - 1) - 1)) + 1;
        
        
        if (n % 2 == 0) {
            while (label > 0) {
                if (n % 2 == 0) {
                    result.add(0, label);
                } else {
                    result.add(0, opposite);
                }
                label /= 2;
                opposite /= 2;
                n--;
            }
        } else {
            while (label > 0) {
                if (n % 2 != 0) {
                    result.add(0, label);
                } else {
                    result.add(0, opposite);
                }
                label /= 2;
                opposite /= 2;
                n--;
            }
        }
        
        return result;
    }
}
