class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        final int N = books.length;
        long best = Long.MAX_VALUE;
        int[][] width = new int[N][N];
        int[][] height = new int[N][N];
        for (int i = 0; i < N; ++i) {
            int w = 0;
            int h = 0;
            for (int j = i; j < N; ++j) {
                w += books[j][0];
                h = Math.max(h, books[j][1]);
                width[i][j] = w;
                height[i][j] = h;
                // System.out.format("i = %d, j = %d: width = %d, height = %d\n", i, j, width[i][j], height[i][j]);
            }
        }
        int[] dp = new int[N];
        Arrays.fill(dp, Integer.MAX_VALUE);
        for (int i = 0; i < N; ++i) {
            if (width[0][i] > shelf_width) {
                break;
            }
            dp[i] = height[0][i];
        }
        // for (int currBook = 1; currBook < N; ++currBook) {
        //     for (int prev)
        // }
        // System.out.printf("START, dp = %s\n", Arrays.toString(dp));
        for (int i = 0; i < N; ++i) {
            for (int j = i + 1; j < N; ++j) {
                if (width[i + 1][j] > shelf_width) {
                    break;
                }
                dp[j] = Math.min(dp[j], dp[i] + height[i + 1][j]);
            }
            // System.out.printf("i = %d, dp = %s\n", i, Arrays.toString(dp));
        }
        return dp[N - 1];
