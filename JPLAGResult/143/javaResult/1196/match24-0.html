class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int n = books.length;
        int[] d = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            d[i] = -1;
        }
        int[][] max2 = new int[n + 1][n + 1];
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                int max = 0;
                for (int k = i; k <= j; k++) {
                    max = Math.max(max, books[k][1]);
                }
                max2[i][j] = max;
            }
        }
        return dp(books, shelf_width, d, 0, max2);
    }
    
    int dp(int[][] books, int shelf_width, int[] d, int st, int[][] max2) {
        int n = books.length;
        if(st == n) {
            return 0;
        }
        if (d[st] != -1) {
            return d[st];
        }
        int sum = 0;
        int ans = 1001 * n;
        for (int i = st; i < n; i++) {
            sum += books[i][0];
            if (sum > shelf_width) {
                break;
            }
            ans = Math.min(ans, max2[st][i] + dp(books, shelf_width, d, i + 1, max2));
        }
        return d[st] = ans;
    }
}
