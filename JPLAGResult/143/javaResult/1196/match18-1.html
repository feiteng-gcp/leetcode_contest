class Solution {
    public int minHeightShelves(int[][] books, int wid) {
        int[] dp = new int[1001];
        return helper(books, wid, 0, dp);
    }
    
    public int helper(int[][] books, int wid, int start, 
int[] dp) {
        if(start>=books.length) return 0;
        if(dp[start]>0) return dp[start];
        int res =Integer.MAX_VALUE;
        int max =0;
        int left =wid;
        for(int i=start;i<books.length;i++) {
            left = left-books[i][0];
            
if(left<0) break;
            max= Math.max(books[i][1],max);
            res = Math.min(helper(books, wid, i+1, dp) + max,res);
        }
        dp[start] = res;
        return res;
    }
}
