class Solution {
    public int minHeightShelves(int[][] books, int sw) {
        int n = books.length, m = books[0].length, maxHeight = 0;
        for (int[] b: books) {
            maxHeight += b[1];
        }
        
        int[] f = new int[n + 1];
        Arrays.fill(f, Integer.MAX_VALUE);
        f[0] = 0;
        for (int i = 1; i <= n; ++i) {
            int curW = 0, maxH = books[i-1][1];
            for (int j = i; j >= 1; --j) {
                curW += books[j-1][0];
                maxH = Math.max(maxH, books[j-1][1]);
                if (curW > sw) {
                    break;
                }
                f[i] = Math.min(f[i], f[j-1] + maxH);
            }
        }
        
        return f[n];
    }
}
