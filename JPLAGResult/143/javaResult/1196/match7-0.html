class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int n = books.length;
        int[] dp = new int[n + 1];
        dp[0] = 0;
        for (int i = 1; i <= n; i++) {
            int width = books[i-1][0];
            int hi = books[i-1][1];
            dp[i] = hi + dp[i-1];
            for (int j = i - 1; j > 0; j--) {
                width += books[j-1][0];
                if (width > shelf_width) break;
                hi = Math.max(hi, books[j-1][1]);
                dp[i] = Math.min(dp[i], dp[j-1] + hi);
            }
        }
        return dp[n];
    }
}
