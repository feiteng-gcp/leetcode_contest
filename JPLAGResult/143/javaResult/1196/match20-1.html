import static java.lang.Integer.max;
import static java.lang.Integer.min;


class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int n = books.length;
        int[][] dp = new int[n][2];
        dp[0][0] = books[0][1];
        dp[0][1] = books[0][1];
        for(int i=1; i<n; i++){
            dp[i][1] = min(dp[i-1][0], dp[i-1][1]) + books[i][1];
            int w = 0;
            int j = i;
            int h = 0;
            dp[i][0] = dp[i][1];
            while(w <= shelf_width && j>=0){
                if (w + books[j][0] > shelf_width) break;
                w += books[j][0];
                h = max(h, books[j][1]);
                if(j-1>=0)  dp[i][0] = min(dp[i][0], h + 
min(dp[j-1][0], dp[j-1][1]));
                else dp[i][0] = min(dp[i][0], h);
                j--;
            }
        }
        return min(dp[n-1][0], dp[n-1][1]);
    }
}
