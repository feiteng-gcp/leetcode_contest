class Solution {
    public int minHeightShelves(int[][] b, int w) {
        
        int n = b.length;
        int[] f = new int[n];
        
        f[0] = b[0][1];
        for (int i=1; i<n; i++){
            f[i] = f[i-1] + b[i][1];
            int wid = b[i][0];
            int hei = b[i][1];            
            int pos = i-1;
            while (pos>=0 && wid+b[pos][0]<=w){   
                wid+=b[pos][0]; 
                hei = Math.max(b[pos][1],hei);
                if (pos>0)
                    f[i] = Math.min(f[i], f[pos-1] + hei);
                else f[i] = Math.min(f[i], hei);
                
                pos--;
            }
            //if ((pos==-1) && (f[i]>b[i][1])) f[i] = b[i][1]; 
            
            //System.out.println(f[i]+" "+hei+" "+wid+" "+pos);
        }
        //for (int i=0; i<n; i++) System.out.println(f[i]);
        return f[n-1];
    }
}
