class Solution {
    int[][] books;
    int width;
    int[] memo = new int[1001];
    
    int go(int b) {
        if (b == books.length) return 0;
        if (memo[b] != 0) return memo[b];
        int h = 0;
        int t = 0, ans = Integer.MAX_VALUE;
        for (int i = b; i < books.length; i++) {
            t += books[i][0];
            if (t > width) break;
            h = Math.max(h, books[i][1]);
            ans = Math.min(ans, h + go(i+1));
        }
        memo[b] = ans;
        return ans;
    }
    
    public int minHeightShelves(int[][] books, int shelf_width) {
        this.books = books;
        this.width = shelf_width;
        return go(0);
    }
}
