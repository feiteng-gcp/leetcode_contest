class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int[] dp = new int[books.length + 1];
        dp[0] = 0;
        int widthSoFar = 0;
        int heightSoFar = 0;
        for (int i = 0; i < books.length; i ++) {
            widthSoFar = 0;
            
heightSoFar = 0;
            for (int j = i; j >= 0; j--) {
                widthSoFar += books[j][0];
                if (widthSoFar > shelf_width)
                    break;
                heightSoFar = Math.max(heightSoFar, books[j][1]);
                dp[i+1] = dp[i+1] == 0? (heightSoFar + dp[j]) : (
Math.min(dp[i+1], heightSoFar + dp[j]));
            }
        }
        return dp[books.length];
    }
}
