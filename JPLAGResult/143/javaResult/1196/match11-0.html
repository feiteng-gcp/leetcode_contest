class Solution {
    public int minHeightShelves(int[][] books, int sw) {
        int ans = 0;
        
        int[] dp = new int[books.length];
        dp[0] = books[0][1];
        
        for (int i = 1; i < dp.length; i++) {
            dp[i] = books[i][1] + dp[i - 1];
            
            int ts = books[i][0];
            int mh = books[i][1];
            for (int j = i - 1; j >= 0; j--) {
                if (ts + books[j][0] <= sw) {
                    mh = Math.max(mh, books[j][1]);
                    dp[i] = Math.min(dp[i], (j == 0 ? 0 : dp[j - 1]) + mh);
                    ts += books[j][0];
                } else {
                    break;
                }
            }
        }
        // System.out.println(Arrays.toString(dp));
        return dp[books.length - 1];
    }
}
