class Solution {

	public int minHeightShelves(int[][] books, int shelf_width) {
		int[] dp = new int[books.length + 1];
		for (int i = 0; i < dp.length; i++) {
			dp[i] = Integer.MAX_VALUE;
		}
		return minHeig
htShelves(dp, 0, books, shelf_width);
	}

	private int minHeightShelves(int[] dp, int b, int[][] book, int shelf_width) {
		if (dp[b] == Integer.MAX_VALUE) {
			if (b >= book.length) {
				return 0;
			}
			int width = 0, height
 = 0;
			for (int i = b; i < book.length; i++) {
				width += book[i][0];
				height = Math.max(height, book[i][1]);
				if (width > shelf_width) {
					break;
				}
				dp[b] = Math.min(dp[b], height + minHeightShelves(dp, i + 1, book, shelf_width));
			}
		}
		return dp[b];
	}
}
