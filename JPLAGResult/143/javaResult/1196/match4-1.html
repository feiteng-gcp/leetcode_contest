class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int len=books.length;
        int[] dp=new int[len];
        dp[0]=books[0][1];
        for(int i=1;i<len;i++){
            int width=books[i][0];
            int height=books[i][1];
            dp[i]=dp[i-1]+height;
            for(int j=i-1;j>=0;j--){
                if(width+books[j][0]<=shelf_width){
                    width+=books[j][0];
                    height=Math.max(height,books[j][1]);
                    dp[i]=Math.min(dp[i],(j==0?0:dp[j-1])+height);
                }else{
                    break;
                }
            }
        }
        return dp[len-1];
    }
}
