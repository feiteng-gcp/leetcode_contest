class Solution {
    
    int[][] books;
    int shelf_width;
    HashMap<Integer, Integer> map;
    int N;
    
    public int minHeightShelves(int[][] books, int shelf_width) {
        this.books = books;
        this.shelf_width = shelf_width;
        this.map = new HashMap<>();
        this.N = books.length;
        
        map.put(N - 1, books[N - 1][1]);
        map.put(N, 0);
        int res = helper(0); 
        //System.out.println(map);
        return res;
    }
    
    private int helper(int start) {
        if (map.containsKey(start)) {
            return map.get(start);
        }
        int res = Integer.MAX_VALUE;
        int total_width = 0;
        for (int i = start, currHeight = 0; i < N; i++) {
            total_width += books[i][0];
            if (total_width > shelf_width) {
                break;
            }
            currHeight = Math.max(currHeight, books[i][1]);
            res = Math.min(res, helper(i + 1) + currHeight);
        }
        map.put(start, res);
        return res;
    }
}
