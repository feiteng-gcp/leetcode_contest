class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int n = books.length;
        int[] f = new int[n];
        f[0] = books[0][1];
        
        for (int i = 1; i < n; ++i) {
            f[i] = f[i - 1] + books[i][1];
            
            int w = books[i][0];
            int currH = books[i][1];
            for (int j = i - 1; j >= 0; --j) {
                w += books[j][0];
                if (w > shelf_width) break;
                
currH = Math.max(currH, books[j][1]);
                if (j > 0)
                    f[i] = Math.min(f[i], f[j - 1] + currH);
                else
                    f[i] = Math.min(f[i], currH);
            }
        }
        return f[n - 1];
    }
}
