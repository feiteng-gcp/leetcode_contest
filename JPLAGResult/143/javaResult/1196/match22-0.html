class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        Map<Integer, Integer> memo = new HashMap<>(); //从i index开始 最小的height
        return dfs(books, 0, shelf_width, memo);
    }
    
    private int dfs(int[][] books, int start, int maxWidth, Map<Integer, Integer> memo) {
        if (start == books.length) {
            return 0;
        }
        if (memo.containsKey(start)) {
            return memo.get(start);
        }
        
        int minHeight = Integer.MAX_VALUE;
        int width = 0;
        int curHeight = 0;
        for (int i = start; i < books.length; i++) {
            width += books[i][0];
            curHeight = Math.max(books[i][1], curHeight);
            if (width > maxWidth) {
                break;
            }
            minHeight = Math.min(curHeight + dfs(books, i + 1, maxWidth, memo), minHeight);
        }
        memo.put(start, minHeight);
        return minHeight;
    }
}
