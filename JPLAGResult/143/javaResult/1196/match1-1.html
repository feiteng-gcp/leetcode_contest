class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        
int[][] dp = new int[books.length+1][2];
        dp[0][0] = 0;
        for (int i = 1; i <= books.length; i++) {
            int cnt = books[i-1][0];
            int maxB = books[i-1][1];
            dp[i][0] = dp[i-1][0]+books[i-1][1];
            for (int j = i-1; j > 0; j--) {
                cnt+=books[j-1][0];
                maxB = Math.max(maxB, books[j-1][1]);
                if (cnt > shelf_width) break;
                if (dp[j-1][0]+maxB < dp[i][0]) {
                    dp[i][0] = dp[j-1][0]+maxB;
                
}
            }
            // System.out.println(dp[i][0]);
        }
        return dp[books.length][0];
    }
}
