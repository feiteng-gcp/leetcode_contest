class Solution {
    // sb 题， 理解了老子一万年
    // Can we solve this problem by using dp?
    // int[][] canPut -> 
    // canPut[i][j] -> from i to j -> Can these books put into one shelf?
    //                             -> If can do, then store the maximum height, else store -1.
    // int[] dp -> from 0 to j the minimum height of the shelf
    // dp[j] = dp[i] (i < j) + canPut[i][j] (canPut[i][j] != -1)
    
    public int minHeightShelves(int[][] books, int shelf_width) {
        int[][] canPut = new int[books.length][books.length];
        for (int[] row : canPut)    Arrays.fill(row, -1);
        
int[] dp = new int[books.length + 1];
        Arrays.fill(dp, Integer.MAX_VALUE);
        dp[0] = 0;
        
        for(int i = 0; i < books.length; i++){
            int maxHeight = 0;
            int width = 0;
            for(int j = i; j < books.length; j++){
                width += books[j][0];
                maxHeight = Math.max(maxHeight, books[j][1]);
                if(width > shelf_width) break;
                canPut[i][j] = maxHeight;
            }
        }
        
        for(int i = 1; i <= books.length; i++){
            for(int j = 0; j < i; j++){
                if(canPut[j][i - 1] != -1)  dp[i] = Math.min(dp[i], dp[j] + canPut[j][i - 1])
;
            }
        }
        return dp[books.length];
    }
}
