class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
//         int res = 0;
//         int n = books.length;
//         for (int i = 0; i < n; i++) {
//             int h = 0;
//             int w = 0;
//             while (i < n && w + books[i][0] <= shelf_width) {
//                 h = Math.max(h, books[i][0]);
//                 w += books[i][0];
//             }
//             res += h;
//         }
        
//         return res;
        
        int n = books.length;
        int[] dp = new int[n];
        dp[0] = books[0][1];
        for (int i = 1; i < n; i++) {
            dp[i] = books[i][1] + dp[i-1];
            int w = books[i][0];
            int h = books[i][1];
            int j = i-1;
            while (j >= 0 && w + books[j][0] <= shelf_width) {
                w = w + books[j][0];
                h = Math.max(h, books[j][1]);
                dp[i] = Math.min(dp[i], (j-1 < 0 ? 0 : dp[j-1]) + h);
                j--;
            }
            
            //System.out.println(i + " " + dp[i]);
            
        }
        return dp[n-1];
    }
}
