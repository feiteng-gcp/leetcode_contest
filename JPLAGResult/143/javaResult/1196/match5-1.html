class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int n = books.length;
        int[] d = new int[n];
        
for (int i=0; i<n; i++) {
            int width_sum = 0, max_height = 0;
            d[i] = -1;
            for (int j=i; j>=0; j--) {
                width_sum += books[j][0];
                if (width_sum > shelf_width) break;
                max_height = Math.max(max_height, books[j][1]);
                int p = (j > 0 ? d[j-1] : 0) + max_height;
                if (d[i] < 0 || d[i] > p) d[i] = p
;
            }
        }
        return d[n-1];
    }
}
