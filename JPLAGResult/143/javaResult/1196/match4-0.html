class Solution {
    public int minHeightShelves(int[][] books, int width) {
        int n = books.length;
        int[] dp = new int[n];
        dp[0] = books[0][1];
        for (int i = 1; i < n; i++) {
            int sum = books[i][0], curW = books[i][1];
            dp[i] = dp[i - 1] + curW;
            for (int j = i - 1; j >= 0; j--) {
                sum += books[j][0];
                if (sum > width) break;
                curW = Math.max(curW, books[j][1]);
                dp[i] = Math.min(dp[i], (j - 1 >= 0 ? dp[j- 1] : 0) + curW);
            }
        }
        return dp[n - 1];
    }
}
