class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int len = books.length;
        int[] dp = new int[len+1];
        
for(int i = 1;i<=len;i++) {
        	dp[i] = dp[i-1]+books[i-1][1];
        	int sum = books[i-1][0];
        	int iMax = books[i-1][1];
        	for(int j = i-1;j>=1&&sum<=shelf_width;j--) {
        		sum += books[j-1][0];
        		iMax = Math.max(iMax, books[j-1][1]);
        		if(sum<=shelf_width) {
        			dp[i]=Math.min(dp[i], dp[j-1]+iMax);
        		}

        	}
        }
        return dp[len];
    }
}
