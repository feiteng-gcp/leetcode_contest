class Solution {
    Integer[] f;

	int res
;

	public int minHeightShelves(int[][] books, int shelf_width) {

		f = new Integer[books.length];

		return calc(0, books, shelf_width);
	}

	private int calc(int index, int[][] books, int shelf_width) {
		if (index == books.length) {
			return 0;
		}
		if (f[index] == null) {

			int ans = Integer.MAX
_VALUE;

			int cw = 0;

			int ch = 0;

			for (int i = index; i < books.length; i++) {
				if (cw + books[i][0] <= shelf_width) {
					cw += books[i][0];
					ch = Math.max(ch, books[i][1]);
					int next = calc(i + 1, books, shelf_width);
					if (ch + next < ans) {
						ans = ch + next;
					}
				} else {

					break;
				}
			}

			f[index] = ans;
		}
		return f[index];
	}
}
