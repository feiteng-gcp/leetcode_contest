import java.util.*;
class Solution {
    HashMap<Integer, Integer> map = new HashMap<>();
    public int minHeightShelves(int[][] books, int shelf_width) {
        return helper(books, 0, shelf_width);
    }

    public int helper(int[][] books, int pos, 
int shelf_width) {
        if (map.containsKey(pos)) {
            return map.get(pos);
        }
        if (pos >= books.length) {
            return 0;
        }
        int res = 10000000;
        int sum_width = 0;
        int height = 0;
        for (
int i = pos; i < books.length; i++) {
            sum_width += books[i][0];
            if (sum_width > shelf_width) {
                break;
            }
            height = Math.max(books[i][1], height);
            res = Math.min(res, helper(books, i + 1, shelf_width) + height);
        }
        map.put(pos, res);
        return res;
    }
}
public class TreeNode {
    int val;
    TreeNode left;
    TreeNode right;

    TreeNode(int x) {
        val = x;
    }
}
