class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int n = books.length;
        int[] dp = new int[n];
        for(int i = 0; i < n ; ++i) {
            if(i == 0) dp[i] = books[i][1];
            else {
                int min = dp[i - 1] + books[i][1];
                int lw = shelf_width;
                int mh = books[i][1];
                lw = shelf_width - books[i][0];
                int lb = i - 1;
                while(lb >= 0 && lw > 0) {
                    if(books[lb][0] > lw) break;
                    lw -= books[lb][0];
                    mh = Math.max(mh, books[lb][1]);
                    if(lb != 0) min = Math.min(min, dp[lb - 1] + mh);
                    else min = mh;
                    --lb;
                }
                dp[i] = min;
            }
        }
        return dp[n - 1];
    }
}
