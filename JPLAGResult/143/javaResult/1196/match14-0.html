class Solution {
    
    Integer[] M;
    
    public int minHeightShelves(int[][] books, int shelf_width) {
        M = new Integer[books.length];
        return dp(books, 0, shelf_width);
    }
    
    public int dp(int[][] books, int i, int W) {
        if (i >= books.length) return 0;
        if (M[i] != null) return M[i];
        
        int min = Integer.MAX_VALUE;
        int max = 0;
        int w = 0;
        for (int j = i; j < books.length; j++) {
            if (w + books[j][0] <= W) {
                w += books[j][0];
                max = Math.max(max, books[j][1]);
                min = Math.min(min, max + dp(books, j + 1, W));
            } else {
                break;
            }
        }
        
        M[i] = min;
        return M[i];
    }
}
