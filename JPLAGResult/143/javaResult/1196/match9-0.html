class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int n=books.length;
        int[] dp=new int[n];
        for(int i=0;i<n;i++)
        {
            int width=0,max=books[i][1];
            dp[i]=Integer.MAX_VALUE;
            for(int j=i;j>=0;j--)
            {
                width+=books[j][0];
                if(width>shelf_width) break;
                max=Math.max(books[j][1],max);
                dp[i]=Math.min(max+(j-1>=0?dp[j-1]:0),dp[i]);
            }
        }
        return dp[n-1];
    }
}
