class Solution {
    public int minHeightShelves(int[][] books, int shelf_width) {
        int[] dp=new int[books.length];
        dp[0]=books[0][1];
        for(int i=1;i<books.length;i++){
            
int maxH=books[i][1];
            int curWid=books[i][0];
            dp[i]=maxH+dp[i-1];
            int j=i-1;
            while(j>-1){
                curWid+=books[j][0];
                if(curWid>shelf_width)
                    break;
                maxH=Math.max(maxH,books[j][1]);
                if(j>0)
                    dp[i]=Math.min(dp[i],maxH+dp[j-1]);
                else
                    dp[i]=maxH;
                j--;
            }
        }
        return dp[dp.length-1];
    }
}
