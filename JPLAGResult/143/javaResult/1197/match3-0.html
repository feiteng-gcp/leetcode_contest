class Solution {
    public boolean parseBoolExpr(String expression) {
        Stack<Character> op = new Stack<>();
        Stack<Character> val = new Stack<>();
        
        char[] exp = expression.toCharArray();
        
        for (char ch : exp) {
            if (ch == ',') {    
                continue;
            }
            if (ch == '&' || ch == '|' || ch == '!') {
                op.push(ch);
            } else if (ch != ')') {
                val.push(ch);
            } else { // ')' -> calculate
                boolean cur = val.peek() == 't';
                while (val.peek() != '(') {
                    char tmp = val.pop();
                    if (op.peek() == '!') {
                        cur = !(tmp == 't');
                    } else if (op.peek() == '&') {
                        cur = cur & (tmp == 't');
                    } else if (op.peek() == '|') {
                        cur = cur || (tmp == 't');
                    }
                }
                val.pop(); // pop '('
                op.pop();
                val.push(cur ? 't' : 'f');
            }
        }
        
        return val.pop() == 't';
    }
}
