class Solution {
    public boolean parseBoolExpr(String e) {
        char[] chs = e.toCharArray();
        Stack<Character> stack = new Stack<>();
        boolean ans = true;
        
        for (char ch : chs) {
            if (ch == '(') {
                 stack.push('(');
            } else if (ch == ',') {
                 continue;
            } else 
if (ch == '&') {
                stack.push('&');
            } else if (ch == '|') {
                stack.push('|');
            } else if (ch == '!') {
                stack.push('!');
            } else if (ch == ')') {
                List<Boolean> tmp = new ArrayList<Boolean>();
                boolean tans = true;
                while(!stack.isEmpty()) {
                    // System.out.println(stack);
                    char ts = stack.pop();
                    if (ts == '(') {
                        char op = stack.pop();
                        if (op == '!') {
                            tans = !tmp.get(0);
                        } else if (op == '&') {
                            for (boolean bl : tmp) {
                                tans = tans && bl;
                            }
                        } else {
                            tans = false;
                            for (boolean bl : tmp) {
                                
                                tans = tans || bl;
                            }
                        }
                        stack.push(tans ? 't' : 'f');
                        break;
                    } else if (ts == 't') {
                        tmp.add(true);
                    } else if (ts == 'f') {
                        tmp.add(false);
                    }
                } 
            } else if (ch == 't') {
                stack.push('t');
            } else { // ch == f
                stack.push('f');
            
}
         }
        
        return stack.peek() == 't' ? true : false;
    }
}
