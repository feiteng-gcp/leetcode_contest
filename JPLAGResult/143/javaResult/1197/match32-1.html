class Solution {
    public boolean parseBoolExpr(String expression) {
        int[] ptr = new int[1];
        
        return eval(expression, ptr);
    }
    
    private boolean eval(String exp, int[] ptr) {
        
        int n = exp.length();
        
        
char ch = exp.charAt(ptr[0]);
        if (ch == 't') {
            ptr[0] += 1;
            return true;
        }
        if (ch == 'f') {
            ptr[0] += 1;
            return false;
        }
        ptr[0] += 2;
        boolean res = true;
        if (ch == '|') res = false;
        
        while (exp.charAt(ptr[0]) != ')') {
            char ch1 = exp.charAt(ptr[0]);
            if (ch1 == ',') {
                ptr[0] += 1;
                continue;
            }
            
            boolean cur = eval(exp, ptr);
            if (ch == '&') res &= cur;
            else if (ch == '|') res |= cur;
            else res = !cur;
            
        }
        ptr[0] += 1;
        return res;
    }
}
