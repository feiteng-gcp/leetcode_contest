class Solution {
    public boolean parseBoolExpr(String expression) {
        return helper(expression, 0, expression.length()-1);
    }
    public boolean helper(String s, int st, int end) {
        if (st == end && s.charAt(st) == 't') return true;
        else if (st == end && s.charAt(st) == 'f') return false;
        else if (s.charAt(st) == '!') return !helper(s, st+2, end-1);
        else { 
            boolean b = (s.charAt(st) == '|')? false : true;
            int pos1 = st+2;
            while (pos1 <= end-1) {
                int pos2 = process(s, pos1, end-1);
                if (s.charAt(st) == '|')
                    b |= helper(s, pos1, pos2);
                else
                    b &= helper(s, pos1, pos2);
                
                pos1 = pos2+2;
            }
            return b;
        }
    }
    public int process(String s, int p, int end) {
        int cnt = 0;
        while (p < end) {
            if (s.charAt(p) == '(') cnt++;
            else if (s.charAt(p) == ')') cnt--;
            else if (s.charAt(p) == ',' && cnt == 0) {
                return p-1;
            }
            p++;
        }
        return p;
    }
}
