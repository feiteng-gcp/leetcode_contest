class Solution {
    private static final char[] TYPECHAR = {'t', 'f'};
    private static final char[] EXPCHAR = {'|', '&', '!'};
    
    public boolean parseBoolExpr(String expression) {
        if(expression.length() == 1){
            return expression.equals("t") ? true : false;
        }
        return DFS(expression, new int[1]);
    
}
    
    private boolean DFS(String str, int[] index){
        char type = str.charAt(index[0]);
        index[0] += 2;
        List<Boolean> res = getResList(str, index);
        if(type == '|'){
            if(res.contains(true)) return true;
            else return false;
        }
        else if(type == '&'){
            if(res.contains(false)) return false;
            else return true;
        }
        else{
            return !res.get(0);
        }
        
    }
    
    private List<Boolean> getResList(String str, int[] index){
        List<Boolean> res = new ArrayList<>();
        while(true){
            char ch = str.charAt(index[0]);
            if(ch == ')'){
                index[0]++;
                return res;
            }
            else if(ch == ','){
                index[0]++;
                continue;
            }
            else if(ch == 't'){
                index[0]++;
                res.add(true);
            }
            else if(ch == 'f'){
                index[0]++;
                res.add(false);
            }
            else{
                res.add(DFS(str, index));
            }
        }
    }
    
        
}
