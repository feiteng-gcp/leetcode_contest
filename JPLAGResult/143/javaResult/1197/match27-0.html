class Solution {
    public boolean parseBoolExpr(String expression) {

		return calc(expression, 0, expression.length() - 1);


	}

	private boolean calc(String e, int x, int y) {
		if (x == y) {
			return e.charAt(x) == 't';
		} else {
			if (e.charAt(x) == '!') {
				return !calc(e, x + 2, y - 1);
			} else  {

				List<Integer> splits = new ArrayList<>();

				int c = 0;

				for (int i = x + 1; i <= y; i++) {
					if (e.charAt(i) == '(') {
						c++;
					} else if (e.charAt(i) == ')') {
						c--;
					}
					if (c == 1 && e.charAt(i) == ',') {
						splits.add(i);
					}
				}

				if (splits.isEmpty()) {
					return calc(e, x + 2, y - 1);
				} else {
					splits.add(y);
					boolean res = false;
					for (int i = 0; i < splits.size(); i++) {
						if (i == 0) {
							res = calc(e, x + 2, splits.get(i) - 1);
						} else {
							boolean t = calc(e, splits.get(i - 1) + 1, splits.get(i) - 1);
							if (e.charAt(x) == '&') {
								res &= t;
							} else {
								res |= t;
							}
						}
					}
					return res;
				}

			}

		}
	}
}
