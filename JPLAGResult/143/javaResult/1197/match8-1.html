class Solution {
    public boolean parseBoolExpr(String exp) {
        if (exp.equals("t")) {
            return true;
        } else if (exp.equals("f")) {
            return false;
        }
        Stack<Character> ops = new Stack<>();
        Stack<Boolean> vals = new Stack<>();
        
        int N = exp.length();
        boolean curr = false;
        char op = 'x';
        for (int i = 0; i < N; i++) {
            char c = exp.charAt(i);
            if (c == '!' || c == '&' || c == '|') {
                if (op != 'x') {
                    ops.push(op);
                    vals.push(curr);
                }
                i++;
                op = c;
                curr = (c == '&') ? true : false;
            } else if (c == 't' || c == 'f') {
                boolean v = c == 't';
                if (op == '!') {
                    curr = !v;
                } else 
if (op == '&') {
                    curr &= v;
                } else if (op == '|') {
                    curr |= v;
                } else {
                    curr = v;
                }
            } else if (c == ')' && !ops.isEmpty()) {
                // System.out.println(vals + ", " + ops);
                op = ops.pop();
                if (op == '!') {
                    curr = !curr;
                    
vals.pop();
                } else if (op == '&') {
                    curr &= vals.pop();
                } else if (op == '|') {
                    curr |= vals.pop();
                }
                // System.out.println(op + ", " + curr + ", i=" + i);
            }
        }
        return curr;
    }
}
