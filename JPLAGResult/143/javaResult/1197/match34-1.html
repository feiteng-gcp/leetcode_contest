import java.util.ArrayList;

class Solution {
    int idx;
    String exp;
    public boolean e0() {
        char c = exp.charAt(idx);
        idx++;
        if (c=='t') return true;
        else return false;
    }
    public boolean e1(){
        char c = exp.charAt(idx);
        idx++;
        boolean[] res = e2();
        int ln = res.length;
        if (c == '!') return !res[0];
        else if(c == '&'){
            if(res.length==1) return res[0];
            boolean r = res[0] && res[1];
            for(int i=2; i<ln; i++){
                r = r && res[i];
            }
            return r;
        }else if (c == '|'){
            if(res.length==1) return res[0];
            boolean r = res[0] || res[1];
            for(int i=2; i<ln; i++){
                r = r || res[i];
            }
            return r;
        }
        return true;
    }

    public boolean[] e2() {
        idx++;
        int cn = 1;
        ArrayList<Boolean> res = new ArrayList<>();
        while(idx < exp.length() && cn>0) {
            char c = exp.charAt(idx);
            if (c == ','){
                idx++;
            }else if (c == '(') {
                boolean[] res1 = e2();
                
res.add(res1[0]);
            }else if (c=='t' || c=='f'){
                res.add(e0());
            }else if(c=='|' || c=='&' || c=='!'){
                res.add(e1());
            }else if(c==')'){
                cn--;
            }
        }
        idx++;
        boolean[] a = new boolean[res.size()] ;
        for(int i=0; i<a.length; i++){
            a[i] = res.get(i);
        }
        return a;
    }


    public boolean parseBoolExpr(String expression) {
        exp = "(" + expression + ")";
        return e2()[0];
    }

    public static void main(String[] args){
        Solution s = new Solution();
        System.out.println(s.parseBoolExpr("!(&(!(t),&(f),|(f)))"));
    }
}
