class Solution {
    public boolean parseBoolExpr(String expression) {
        Stack<Character> operatorStack = new Stack<>();
        Stack<Value> valueStack = new Stack<>();
        int depth = 0;
        
        for(int i = 0; i < expression.length(); i ++) {
            char current = expression.charAt(i);
            
            if(current == '|') {
                operatorStack.push(current);
            } else if(current == '&') {
                operatorStack.push(current);
            } else if(current == '!') {
                operatorStack.push(current);
            } else if(current == '(') {
                depth ++;
            } else if(current == ',') {
                // do nothing for now
            } else if(current == ')') {
                char operator = operatorStack.pop();
                                
                boolean currentValue = false;
                
                if(operator == '&') {
                    currentValue = true;
                }
                
                while(!valueStack.isEmpty() && valueStack.peek().depth == depth) {
                    Value nextValue = valueStack.pop();
                    
                    if(operator == '|') {
                        currentValue |= nextValue.value;
                    } else if(operator =='&' ) {
                        currentValue &= nextValue.value;
                    } else if(operator == '!') {
                        currentValue = !nextValue.value;
                    } 
                }
                
                depth --;
            
                valueStack.push(new Value(currentValue, depth));
            } else if(current == 't') {
                valueStack.push(new Value(true, depth));
            } else if(current == 'f') {
                valueStack.push(new Value(false, depth));
            }
        }
        
        return valueStack.pop().value;
    }
}

class Value {
    public boolean value;
    public int depth;
    
    public Value(boolean value, int depth) {
        this.value = value;
        this.depth = depth;
    }
}
