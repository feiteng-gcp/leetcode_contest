class Solution {
    public boolean parseBoolExpr(String expression) {
        char c = expression.charAt(0);
        if (c == 't'){
            return true;
        }
        if (c == 'f'){
            return false;
        }
        
        List<Boolean> exps = getList(expression.substring(2, expression.length()-1));
        if (c == '!') {
            return !exps.get(0);
        }
        
        if (c == '&') {
            for (boolean b : exps){
                if (!b){
                    return false;
                }
            }
            return true;
        }
        
        for (boolean b : exps){
            if (b){
                return true;
            }
        }
        return false;
    }
    
    List<Boolean> getList(String str){
        System.out.println("getList:" + str);
        List<Boolean> ans = new ArrayList<>();
        int pre = 0;
        for (int i = 0 ; i < str.length() ; i++){
            if (str.charAt(i) == ','){
                String s = str.substring(pre, i);
                pre = i+1;
                ans.add(parseBoolExpr(s));
            } else if (str.charAt(i) == '(') {
                i = nextClose(str, i);
            }
        }
        ans.add(parseBoolExpr(str.substring(pre)));
        return ans;
    }
    
    int nextClose(String str, int start){
        int count = 0;
        for (int i = start ; i < str.length() ; i++) {
            if (str.charAt(i) == '(') {
                count++;
            } else if (str.charAt(i) == ')') {
                count--;
                if (count == 0){
                    return i;
                }
            } 
        }
        return -1;
    }
}
