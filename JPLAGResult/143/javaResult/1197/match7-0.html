class Solution {
    public boolean parseBoolExpr(String expression) {
        Deque<Character> stack = new LinkedList<>();
        Deque<Integer> numStack = new LinkedList<>();
        int numOfT = 0, numOfF = 0;
        for (char ch : expression.toCharArray()) {
            if (ch == '&' || ch == '|' || ch == '!') {
                numStack.push(numOfT); numStack.push(numOfF);
                stack.push(ch);
                numOfT = 0; numOfF = 0;
            } else if (ch == ')') {
                int sign = stack.pop();
                int t = 0, f = 0;
                if (sign == '|') {
                    if (numOfT > 0) t++;
                    else f++;
                } else if (sign == '&') {
                    if (numOfF > 0) f = 1;
                    else t = 1;
                } else if (sign == '!') {
                    if (numOfF == 1) t++;
                    else f++;
                }
                numOfF = numStack.pop() + f;
                numOfT = numStack.pop() + t;
            } else if (ch == 't') {
                numOfT++;
            } else if (ch == 'f') {
                numOfF++;
            }
        }
        return numOfF == 1 ? false : true;
    }
}
