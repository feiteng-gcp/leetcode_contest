class Solution {
    private String solve(String s,int subs,int sube){
        //System.out.println("plain s = " + s.substring(subs,sube + 1));
        char c = s.charAt(subs);
        if(c == '(') return s.substring(subs + 1,subs + 2);

        if(c == '!') return s.charAt(subs + 2) == 't' ? "f" : "t";

        if(c == '&') return s.substring(subs,sube).contains("f") ? "f" : "t";
        return s.substring(subs,sube).contains("t") ? "t" : "f";
    }

    private boolean help(String s, int level){
        //System.out.println("before solving level = " + level + " s = " + s);

        if(level == 0) return s.charAt(0) == 't';
        StringBuilder stringBuilder = new StringBuilder();
        int cur = 0;
        int subs = -1, sube = -1;
        for (int i = 0; i < s.length(); i++) {
            if(s.charAt(i) == '('){
                cur++;
                if(cur == level){
                    char c = stringBuilder.charAt(stringBuilder.length() - 1);
                    if(c == '&' || c == '|' || c =='!'){
                        stringBuilder.deleteCharAt(stringBuilder.length() - 1);
                        subs = i - 1;
                    }else{
                        subs = i;
                    }
                }
            }
            else if(s.charAt(i) == ')'){
                if(cur == level) sube = i;
                cur--;
            }
            if(sube > 0){
                stringBuilder.append(solve(s,subs,sube));
                //System.out.println("after replace s = " + stringBuilder);
                subs = -1;
                sube = -1;
            }else if(subs < 0){
                stringBuilder.append(s.charAt(i));
            }
        }
        //System.out.println("solving level = " + level + " s = " + stringBuilder);
        return help(stringBuilder.toString(),level - 1);
    }

    public boolean parseBoolExpr(String expression) {
        int level = 0, cur = 0;
        for (int i = 0; i < expression.length(); i++) {
            if(expression.charAt(i) == '('){
                cur++;
                level = Math.max(level,cur);
            }else if(expression.charAt(i) == ')'){
                cur--;
            }
        }
        return help(expression,level);
    }
}
