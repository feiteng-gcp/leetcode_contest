class Solution {
    
    public boolean parseBoolExpr(String expression) {
        Stack<Character> exprStack = new Stack<Character>();
        Stack<Character> wordStack = new Stack<Character>();
        char e[] = expression.toCharArray();
        for (int i = 0; i < e.length; i++) {
            if (e[i] == '&' || e[i] == '|' || e[i] == '!')
                exprStack.push(e[i]);
            else if (e[i] == '(' || e[i] == 'f' || e[i] == 't')
                wordStack.push(e[i]);
            else if (e[i] == ')') {
                if (exprStack.peek() == '!') {
                    Character result = wordStack.pop();
                    wordStack.pop();    
                    wordStack.push(result == 't' ? 'f' : 't');
                    exprStack.pop();
                }
                else {
                    boolean result = exprStack.peek() == '&' ? true : false;
                    while (wordStack.peek() != '(') {
                        boolean current = wordStack.pop() == 't' ? true : false;
                        result = exprStack.peek() == '&' ? result & current : result | current;
                    }
                    wordStack.pop();
                    exprStack.pop();
                    wordStack.push(result ? 't' : 'f');
                }
            }
        }
        return wordStack.pop() == 't' ? true : false;
    }
}
