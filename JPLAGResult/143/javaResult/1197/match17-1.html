class Solution {
    public boolean parseBoolExpr(String expression) {
        return 
dfs(expression);
    }
    boolean dfs(String expression){
        if(expression.equals("t"))
            return true;
        else if(expression.equals("f"))
            return false;
        int flag=0;
        if(expression.charAt(0)=='|')
            flag=1;
        else if(expression.charAt(0)=='!')
            flag=2;
        int n=expression.length();
        expression=expression.substring(2,n-1);
        boolean res=flag==1?false:true;
        if(flag==2) res=true
;
        int count=0,start=0;
        for(int i=0;i<expression.length();i++)
        {
            if(expression.charAt(i)=='(') count++;
            else if(expression.charAt(i)==')') count--;
            if(count==0&&(expression.charAt(i)==','||i==expression.length()-1)){
                if(flag==1)
                    res=res||dfs(expression.substring(start,i==expression.length()-1?i+1:i));
                else
                    res=res&&dfs(expression.substring(start,i==expression.length()-1?i+1:i));
                start=i+1;
            }
        }
        if(flag==2) return !res;
        return res;
    }
}
