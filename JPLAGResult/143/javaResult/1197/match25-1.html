class Solution {

	public boolean parseBoolExpr(String expression) {
		if (expression.equals("t")) {
			return true;
		} else if (expression.equals("f")) {
			return false;
		} else if (expression.charAt(0) == '!') {
			return !parseBoolExpr(expression.substring(2, expression.length() - 1));
		} else if (expression.charAt(0) == '&') {
			boolean result = true
;
			String s = "";
			int count = 0;
			for (int i = 2; i < expression.length() - 1; i++) {
				if (i == expression.length() - 2) {
					result &= parseBoolExpr(s + expression.charAt(i));
				} else if ((count += expression.charAt(i) == '(' ? 1 : expression.charAt(i) == ')' ? -1 : 0) == 0
						&& expression.charAt(i) == ',') {
					result &= parseBoolExpr(s);
					s = "";
				} else {
					s += expression.charAt(i);
				}
			}
			return result;
		} else {
			boolean result = false;
			String s = "";

			int count = 0;
			for (int i = 2; i < expression.length() - 1; i++) {
				if (i == expression.length() - 2) {
					result |= parseBoolExpr(s + expression.charAt(i));
				} else if ((count += expression.charAt(i) == '(' ? 1 : expression.charAt(i) == ')' ? -1 : 0) == 0
						&& expression.charAt(i) == ',') {
					result |= parseBoolExpr(s);
					s = "";
				} else {
					s += expression.charAt(i);
				}
			}
			return result;
		}
	}
}
