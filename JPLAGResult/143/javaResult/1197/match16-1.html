class Solution {
    public boolean parseBoolExpr(String expression) {
        
if(expression.length()==0) return true;
        if(expression.startsWith("|")){
            return or(expression.substring(2,expression.length()-1));
        } else if(expression.startsWith("&")){
            return and(expression.substring(2,expression.length()-1));
        } else if(expression.startsWith("!")){
            boolean not = parseBoolExpr(expression.substring(2,expression.length()-1));
            return !not;
        } else if(expression.startsWith("t")){
            return true;
        } else{
            return false;
        }
    }
    
    public boolean or(String e){
        if(e.length()==0) return true;
        
int p = 0;
        int index = 0;
        for(int i = 0; i < e.length(); i++){
            if(e.charAt(i)==',' && p==0){
                boolean first = parseBoolExpr(e.substring(index,i));
                index = i+1;
                if(first){
                    return true;
                }else{
                    return or(e.substring(index,e.length()));
                }
            
}
            if(e.charAt(i)=='(')p++;
            if(e.charAt(i)==')')p--;
        }
        return parseBoolExpr(e.substring(index,e.length()));
    }
    
    public boolean and(String e){
        if(e.length()==0) return true;
        int p = 0;
        int index = 0;
        for(int i = 0; i < e.length(); i++){
            if(e.charAt(i)==',' && p==0){
                boolean first = parseBoolExpr(e.substring(index,i));
                index = i+1;
                if(!first){
                    return false;
                } else{
                    return and(e.substring(index,e.length()));
                }
            }
            if(e.charAt(i)=='(')p++;
            if(e.charAt(i)==')')p--;
        }
        return parseBoolExpr(e.substring(index,e.length()));
    }
}
