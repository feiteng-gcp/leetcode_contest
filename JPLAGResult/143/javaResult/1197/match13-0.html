class Solution {
    public boolean parseBoolExpr(String expression) {
        Stack<Character> ops = new Stack<>();
        Stack<Character> tfs = new Stack<>();
        for (int i = 0; i < expression.length(); i++) {
            char c = expression.charAt(i);
            if (c == ',') continue;
            if (c == 't' || c == 'f' || c == '(') {
                tfs.push(c);
            }
            else if (c == ')') {
                char o = ops.pop();
                if (o == '|') {
                    boolean b = false;
                    while (tfs.peek() != '(') {
                        char e = tfs.pop();
                        if (e == 't') b = true;
                    }
                    tfs.pop();
                    tfs.push(b ? 't' : 'f');
                }
                else if (o == '&') {
                    boolean b = true;
                    while (tfs.peek() != '(') {
                        char e = tfs.pop();
                        if (e == 'f') b = false;
                    }
                    tfs.pop();
                    tfs.push(b ? 't' : 'f');
                }
                else {
                    boolean b = tfs.pop() == 'f';
                    tfs.pop();
                    tfs.push(b ? 't' : 'f');
                }
            }
            else ops.push(c);
        }
        return tfs.pop() == 't';
    }
}
