class Solution {
    public boolean parseBoolExpr(String expression) {
        if (expression.equals("t")) {
            return true;
        } else if (expression.equals("f")) {
            return false;
        }
        char f = expression.charAt(0);
        if (f == '!') {
            int paren = 1;
            int index = -1;
            for (int i = 2; i < expression.length(); i++) {
                if (expression.charAt(i) == '(') {
                    paren++;
                } else if (expression.charAt(i) == ')') {
                    paren--;
                }
                if (paren == 0) {
                    index = i;
                    break;
                }
            }
            return !parseBoolExpr(expression.substring(2, index));
        } else if (f == '&') {
            int paren = 1;
            ArrayList<String> comps = new ArrayList<>();
            int last = 2;
            for (int i = 2; i < expression.length(); i++) {
                if (expression.charAt(i) == '(') {
                    paren++;
                } else if (expression.charAt(i) == ')') {
                    paren--;
                } else if (paren == 1 && expression.charAt(i) == ',') {
                    comps.add(expression.substring(last, i));
                    last = i + 1;
                }
                if (paren == 0) {
                    comps.add(expression.substring(last, i));
                }
            }
            boolean result = parseBoolExpr(comps.get(0));
            for (int i = 1; i < comps.size(); i++) {
                if (!result) {
                    return false;
                }
                result = result && parseBoolExpr(comps.get(i));
            }
            return result;
        } else if (f == '|') {
            int paren = 1;
            ArrayList<String> comps = new ArrayList<>();
            int last = 2;
            for (int i = 2; i < expression.length(); i++) {
                if (expression.charAt(i) == '(') {
                    paren++;
                } else if (expression.charAt(i) == ')') {
                    paren--;
                } else if (paren == 1 && expression.charAt(i) == ',') {
                    comps.add(expression.substring(last, i));
                    last = i + 1;
                }
                if (paren == 0) {
                    comps.add(expression.substring(last, i));
                }
            }
            boolean result = parseBoolExpr(comps.get(0));
            for (int i = 1; i < comps.size(); i++) {
                if (result) {
                    return true;
                }
                result = result || parseBoolExpr(comps.get(i));
            }          
            return result;
        } else {
            throw new IllegalArgumentException();
        }
    }
}
