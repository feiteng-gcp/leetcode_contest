class Solution {
    public boolean parseBoolExpr(String expression) {
        Stack<Stack<Boolean>> cS = new Stack<>();
        Stack<Stack<Character>> oS = new Stack<>();
        
        cS.push(new Stack<>());
        oS.push(new Stack<>());
        
        for (int i = 0; i < expression.length(); i++) {
            char c = expression.charAt(i);
            
            if (c == '&' || c == '|' || c == '!') {
                cS.push(new Stack<>());
                oS.push(new Stack<>());
                
                oS.peek().push(c);
            } else if (c == 't' || c == 'f') {
                cS.peek().push(c == 't');
            } else if (c == ')') {
                char opr = oS.peek().pop();
                if (opr == '!') {
                    cS.peek().push(!cS.peek().pop());
                } else if (opr == '&') {
                    boolean res = true;
                    while (!cS.peek().isEmpty()) {
                        res &= cS.peek().pop();
                    }
                    
                    cS.peek().push(res);
                } else if (opr == '|') {
                    boolean res = false;
                    while (!cS.peek().isEmpty()) {
                        res |= cS.peek().pop();
                    }
                    
                    cS.peek().push(res);
                }
                
                boolean res = cS.peek().pop();
                cS.pop();
                oS.pop();
                cS.peek().push(res);
            }
        }
        
        return cS.peek().pop();
    }
}
