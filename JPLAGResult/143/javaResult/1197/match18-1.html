class Solution {
    public boolean parseBoolExpr(String exp) {
        //System.out.println(exp);
        if (exp.equals("t")) {
            return true;
        }
        if (exp.equals("f")) {
            return false;
        }
        if (exp.charAt(0) == '!') {
            return !parseBoolExpr(exp.substring(2, exp.length() - 1));
        }
        if (exp.charAt(0) == '&') {
            int level = 0;
            for (
int i = 0, start = 2; i < exp.length(); i++) {
                if (exp.charAt(i) == '(') {
                    level++;
                } else if (exp.charAt(i) == ')') {
                    level--;
                }
                if (level == 1) {
                    if (exp.charAt(i) == ',' || i == exp.length() - 2) {
                        if (!parseBoolExpr(exp.substring(start, i + (i == exp.length() - 2 ? 1 : 0)))) {
                            return false;
                        }
                        start = i + 1;
                    }
                }
            }
            return true;
        }
        int level = 0;
        for (
int i = 0, start = 2; i < exp.length(); i++) {
            if (exp.charAt(i) == '(') {
                level++;
            } else if (exp.charAt(i) == ')') {
                level--;
            }
            if (level == 1) {
                if (exp.charAt(i) == ',' || i == exp.length() - 2) {
                    if (parseBoolExpr(exp.substring(start, i + (i == exp.length() - 2 ? 1 : 0)))) {
                        return true;
                    }
                    start = i + 1;
                }
            }
        }
        return false;
    }
}
