class Solution {
    public boolean parseBoolExpr(String expression) {
        if(expression.equals("t")) return true;
        if(expression.equals("f")) return false;
        char op=expression.charAt(0);
        String str=expression.substring(2,expression.length()-1);
        if(str.length()==1){
            if(str.equals("f")) return op=='!';
            return op!='!';
        }
        List<Boolean> list= new ArrayList<>();
        int num=0;
        for(int i=0,s=0;i<=str.length();i++){
            if(i==str.length()||(str.charAt(i)==','&&num==0)){
                list.add(parseBoolExpr(str.substring(s,i)));
                s=i+1;
            } else if(str.charAt(i)=='(') {
                num++;
            } else if(str.charAt(i)==')') {
                num--;
            
}
        }
        if(op=='&') return !list.contains(false);
        if(op=='|') return list.contains(true);
        return !list.get(0);
    }
}
