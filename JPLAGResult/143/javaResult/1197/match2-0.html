class Solution {
    int idx=0;
    public boolean parseBoolExpr(String expression) {
        return parse(expression, ' ');
    }
    
    public boolean parse(String exp, char symbol){
        List<Boolean> list = new ArrayList<>();
        for(; idx<exp.length(); idx++){
            char ch = exp.charAt(idx);
            if(ch==')') {
                break;
            }
            if(ch=='f'){
                list.add(false);
            }else if(ch=='t'){
                list.add(true);
            }else if(ch=='!'){
                idx+=2;
                list.add(parse(exp, '!'));
            }else if(ch=='&'){
                idx+=2;
                list.add(parse(exp, '&'));
            }else if(ch=='|'){
                idx+=2;
                list.add(parse(exp, '|'));
            }
        }
        if(symbol==' ') return list.get(0);
        if(symbol=='!') return !list.get(0);
        boolean res = list.get(0);
        for(int j=1; j<list.size(); j++){
            if(symbol=='|') res = res || list.get(j);
            else if(symbol=='&') res = res && list.get(j);
        }
        return res;
    }
}
