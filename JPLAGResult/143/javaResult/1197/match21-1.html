class Solution {
    public boolean parseBoolExpr(String ex) {
        Stack<Character> stack = new Stack<>();
        for (int i = 0; i < ex.length(); i++) {
            char c = ex.charAt(i);
            
if (c == '&' || c == '|' || c == 't' || c == 'f' || c == '!') {
                stack.add(c);
            } else if (c == '(' || c == ',') {
                continue;
            } else {
                // System.out.println("stack = " + stack);
                boolean and = true;
                boolean or = false;
                while (stack.peek() != '&' && stack.peek() != '|' && stack.peek() != '!') {
                    boolean val = stack.pop() == 't' ? true : false;
                    and &= val;
                    or |= val;
                }
                
                if (stack.peek() == '&') {
                    stack.pop();
                    stack.push(and ? 't' : 'f');
                } else if (stack.peek() == '|') {
                    stack.pop();
                    stack.push(or ? 't' : 'f');
                } else { // !
                    stack.pop();
                    stack.push(!and ? 't' : 'f');
                }
            
}
        }
        // System.out.println("stack = " + stack);
        return stack.peek() == 't' ? true : false;
    }
}
