class Solution {
    public boolean parseBoolExpr(String expression) {
        Deque<Character> opStack = new ArrayDeque();
        Deque<Character> valStack = new ArrayDeque();
        for (char c : expression.toCharArray()) {
            if (c == '!' || c == '&' || c == '|') {
                opStack.addFirst(c);
            } else if (c == '(' || c == 't' || c == 'f') {
                valStack.addFirst(c);
            } else if (c == ')') {
                int ct = 0, cf = 0;
                while (!valStack.isEmpty() && valStack.peekFirst() != '(') {
                    char tmp = valStack.removeFirst();
                    if (tmp == 't') ct += 1;
                    else cf += 1;
                }
                valStack.removeFirst();
                char tmp = opStack.removeFirst();
                if (tmp == '&') {
                    tmp = (cf > 0) ? 'f' : 't';
                } else if (tmp == '|') {
                    tmp = (ct > 0) ? 't' : 'f';
                } else {
                    tmp = (ct > 0) ? 'f' : 't';
                }
                valStack.addFirst(tmp);
            } else 
                continue;
        }
        return valStack.peekFirst() == 't';
    }
}
