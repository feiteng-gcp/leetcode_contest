class Solution {
    public boolean parseBoolExpr(String expression) {
        List<Boolean> temp = new ArrayList<>();
        
LinkedList<Character> list = new LinkedList<>();
        LinkedList<Character> sign = new LinkedList<>();
        char[] ch = expression.toCharArray();
        for(char c : ch){
            if(c == ','){
                continue;
            }
            if(c == '|' || c == '&' || c =='!'){
                sign.push(c);
                continue;
            }
            if(c == ')'){
                if(sign.isEmpty()){
                    continue;
                }
                char sg = sign.poll();
                while(!list.isEmpty()){
                    char cc = list.poll();
                    if(cc == '('){
                        break;
                    }
                    temp.add(cc=='t');
                }
                boolean b;
                switch (sg){
                    case '|':
                        b = false;
                        for(Boolean bool : temp){
                            b |= bool;
                        }
                        list.push(b?'t':'f');
                        break;
                    case '&':
                        b = true;
                        for(Boolean bool : temp){
                            b &= bool;
                        }
                        list.push(b?'t':'f');
                        break;
                    case '!':
                        b = !temp.get(0);
                        list.push(b?'t':'f');
                        break;
                    default:
                }
                temp.clear();
                continue;
            }
            list.push(c);
        }
        return !list.isEmpty() && list.poll()=='t';
    }
}
