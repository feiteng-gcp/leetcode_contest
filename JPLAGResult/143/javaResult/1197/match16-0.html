class Solution {
    public boolean parseBoolExpr(String e) {
        if (e.equals("t"))
            return true;
        if (e.equals("f"))
            return false;
        if (e.startsWith("!")){
            return !parseBoolExpr(e.substring(2, e.length() - 1));
        }
        String str = e.substring(2, e.length() - 1);
        List<String> l = parse(str);
        System.out.println(l);
        if (e.startsWith("&")){
            boolean res = true;
            for (String p: l){
                res = res && parseBoolExpr(p);
                if (res == false)
                    return res;
            }
            return res;
        }
            boolean res = false;
            for (String p: l){
                res = res || parseBoolExpr(p);
                if (res == true)
                    return res;
            }
            return res;
    }
    
    public List<String> parse(String s){
        List<String> l = new ArrayList<>();
        int cur = 0, start = 0;
        for (int i = 0; i < s.length(); i++){
            if (s.charAt(i) == ',' && cur == 0){
                l.add(s.substring(start, i));
                start = i + 1;
            }
            if (s.charAt(i) == '(')
                cur++;
            if (s.charAt(i) == ')')
                cur--;
        }
        l.add(s.substring(start, s.length()));
        return l;
    }
}
