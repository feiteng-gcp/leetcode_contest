class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        if (num_people == 0) {
            return new int[0];
        }
        
int[] ans = new int[num_people];
        int n = num_people;
        int k = 0;
        while ((1+k*n)*k*n/2 < candies) {
            k++;
        }
        k--;
        for (int i = 0; i < n; i++) {
            ans[i] = k*(i+1)+(k)*(k-1)/2*n;
            candies -= ans[i];
        }
        for (int i = 0; i < n && candies > 0; i++) {
            int target = k*n+i+1;
            
if (candies >= target) {
                ans[i] += target;
                candies -= target;
            } else {
                ans[i] += candies;
                candies = 0;
            }
        }
        return ans;
    }
}
