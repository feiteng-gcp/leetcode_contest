class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] results = new int[num_people];
        for(int i=1; ; i++){
            if(candies<= i ){
                results[(i-1)% num_people] += candies;
                break;
            }
            results[(i-1)% num_people] += i;
            candies -= i;
        }
        return results;
    }
}
