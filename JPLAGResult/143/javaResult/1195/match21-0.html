class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] result = new int[num_people];
        int current = 1;
        while (candies > 0) {
            for (int i = 0; i < num_people; i++) {
                if (candies < current) {
                    current = candies;
                }
                result[i] += current;
                candies -= current;
                current += 1;
                if (candies == 0) {
                    break;
                }
            }
        }
        return result;
    }
}
