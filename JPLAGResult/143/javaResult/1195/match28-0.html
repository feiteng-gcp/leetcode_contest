class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int n = num_people;
		int[] re = new int[n];
		
		int x = (1+n)*n/2;
		int index = 0 ;
		
		while(candies >= index*n*n + x)
		{
			candies -= index*n*n + x;
			index ++;
		}
		
		for(int i = 0 ; i < n ; ++ i)
		{
			re[i] = n*(0+index-1)*index/2 + index*(i+1);
			if(candies >= index*n +i+1)
			{
				re[i] += index*n +i+1;
				candies -= index*n +i+1; 
			}
			else
			{
				re[i] += candies;
				candies = 0; 
			}
		}
		
		return re;
    }
}
