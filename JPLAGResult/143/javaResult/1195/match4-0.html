class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] cnt = new int[num_people];
        int give = 1;
        int idx = 0;
        while (candies > 0) {
            int dist = Math.min(candies, give);
            cnt[idx] += dist;
            idx = (idx + 1) % num_people;
            ++give;
            candies -= dist;
        }
        return cnt;
    }
}
