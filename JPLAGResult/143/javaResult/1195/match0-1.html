class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] res = new int[num_people];
        int curr = 1, index = 0;
        while(candies>0){
            if(curr>candies){
                res[index]+=candies;
                candies = 0;
            } else{
                res[index]+=curr;
                candies-=curr;
            }            
            curr++;
            index = (index+1)%num_people;
        }
        
        return res;
    }
}
