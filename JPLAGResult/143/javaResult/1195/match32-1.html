class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        if(num_people <= 0) {
            return new int[0];
        }
        
        int[] res = new int[num_people];
        int index =  1;
        while(candies > 0) {
            
if(candies > index) {
                res[(index-1)%num_people] += index;
                candies -= index;
            } else {
                res[(index-1)%num_people] += candies;
                candies = 0;
            }
            
            ++index;
        }
        
        return res;
    }
}
