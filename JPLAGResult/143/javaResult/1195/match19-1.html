class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] dist = new int[num_people];
        int count = 0;
        int amount = 1;
        while (candies > 0) {
            if (amount > candies) {
                dist[count] += candies;
                break;
            
}
            dist[count] += amount;
            candies -= amount;
            amount++;
            count = (count + 1) % num_people;
        }
        return dist;
    }
}
