class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] res = new int[num_people];
        int cnt = 1;
        while (candies > 0) {
            for (int i = 0; i < num_people; i++) {
                if (candies >= cnt) {
                    res[i] += cnt;
                    candies-=cnt;
                    cnt+=1;
                }
                else {
                    res[i] += candies;
                    candies = 0;
                    
break;
                }
            }
        }
        return res;
    }
}
