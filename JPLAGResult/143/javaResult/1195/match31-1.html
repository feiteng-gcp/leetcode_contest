class Solution {
    public int[] distributeCandies(int c, int n) {
        int[] res = new int[n];
        int pre = 0;
        int tag = 0;
        while(c > 0) {
            int cur = Math.min(pre + 1, c);
            res[tag] += cur;
            tag++;
            if (tag >= n) tag = 0;
            pre++;
            c -= cur;
        }
        return res;
    }
}
