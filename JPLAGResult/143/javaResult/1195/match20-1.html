class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] res=new int[num_people];
        int i=1;
        while(candies>0)
        {
            for(int j=0;j<num_people;j++)
            {
                if(candies<=i)
                {    
                    res[j]+=candies;
                    candies=0;
                    break;
                }
                else{
                    res[j]+=i;
                    candies-=i;
                    i++;
                
}
            }
        }
        
        return res;
        
    }
}
