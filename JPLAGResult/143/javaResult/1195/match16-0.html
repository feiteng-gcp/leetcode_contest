class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] res = new int[num_people];
        int i = 0, k = 1;
        while (candies != 0) {
            if (candies < k) {
                res[i++] += candies;
                candies = 0;
            }
            else {
                res[i++] += k;
                candies -= k++;
            }
            i %= num_people;
        }
        return res;
    }
}
