class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int ans[] = new int[num_people];
        int i = 0;
        int last = 0;
        while ( candies != 0 ) {
            last++;
            if ( last > candies ) {
                ans[i] += candies;
                candies = 0;
            } else {
                ans[i] += last;
                candies -= last;
            }
            i = (i + 1) % num_people;
        }
        return ans;
        
    }
}
