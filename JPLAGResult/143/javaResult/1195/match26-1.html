class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] ret = new int[num_people];
        int idx = 0;
        int cur = 1;
        while (candies > cur) {
            ret[idx] += cur;
            candies -= cur;
            cur++;
            idx++;
            if (idx == num_people) {
                idx = 0;
            }
        }
        
        if (candies > 0) {
            ret[idx] += candies;
        }
        
        return ret;
    }
}
