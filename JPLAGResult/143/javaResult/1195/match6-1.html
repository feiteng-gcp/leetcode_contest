class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] res = new int[num_people];
        for(int n=0, k=0; candies>0; candies-=n){
            res[k++] += Math.min(++n, candies);
            k = k%num_people;
        }
        return res;
    }
}
