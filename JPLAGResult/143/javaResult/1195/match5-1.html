class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] result = new int[num_people];
        
        int current = 1;
        int currentIndex = 0;
        
        while(candies > 0) {
            int distribute = Math.min(current, candies);
            
            result[currentIndex % num_people] += distribute;
            
            candies -= distribute;
            
            current ++;
            currentIndex ++;
        }
        
        return result;
    }
}
