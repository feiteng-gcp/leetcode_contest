class Solution {
    public int[] distributeCandies(int can, int n) {
        int[] ans = new int[n];
        int k = 0;
        int t = 0;
        while (can>0){
            ans[k]+= t*n+(k+1);
            can -= (t*n+k+1);
            k++;
            if (k==n) {
                k = 0;
                t++;
            }                        
        }
        if (k==0) ans[n-1] += can;
        else ans[k-1] += can;
        return ans;
    }
}
