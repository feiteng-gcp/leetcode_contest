class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] ans = new int[num_people];
        int i = 1;
        int j = 0;
        while(i<candies){
            ans[j%num_people] += i;
            candies -= i;
            j++;
            i++;            
        }
        ans[j%num_people] += candies;
        return ans;        
    }
}
