class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] res = new int[num_people];
        int cur = 1;
        int idx = 0;
        while (candies > 0) {
            res[idx] += (candies >= cur)? cur : candies;
            candies -= (candies >= cur)? cur : candies;
            cur ++;
            idx ++;
            if (idx == num_people)
                idx = 0;
        }
        return res;
    }
}
