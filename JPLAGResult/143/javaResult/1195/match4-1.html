class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] ans = new int[num_people];
        int candiesLeft = candies;
        int candiesCurPerson = 1;
        while (candiesLeft > 0) {
            int curPerson = (candiesCurPerson - 1) % num_people;
            ans[curPerson] += Math.min(candiesCurPerson, candiesLeft);
            candiesLeft -= candiesCurPerson;
            candiesCurPerson += 1;
        }
        return ans;
    }
}
