class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] res = new int[num_people];
        int round = 0;
        while (candies > 0) {
            for (int i = 1; i <= num_people && candies > 0; i++) {
                int tmp = Math.min(i + round * num_people, candies);
                res[i - 1] += tmp;
                candies -= tmp;
            }
            round++;
        }
        return res;
    }
}
