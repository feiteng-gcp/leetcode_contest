class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] ret = new int[num_people];
        int round = 0;
        while (candies > 0){
            for (int i = 0; i < num_people; i++) {
                int needed = i + 1 + round * num_people;
                int actual = 
Math.min(needed,candies);
                ret[i] += actual;
                candies -= actual;
            }
            round++;
        }
        return ret;
    }
}
