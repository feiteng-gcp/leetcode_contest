class Solution {
    public int[] distributeCandies(int candies, int num_people) {
        int[] res = new int[num_people];
        int idx = 0;
        int n = 1;
        while (candies > 0) {
            int nn = Math.min(n++, candies);
            candies -= nn;
            res[idx++] += nn;
            if (idx == num_people) 
{
                idx = 0;
            }
        }
        return res;
    }
}
