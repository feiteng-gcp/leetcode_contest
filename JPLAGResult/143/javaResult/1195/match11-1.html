class Solution {
    public int[] distributeCandies(int candy, int n) {
        int[] res = new int[n];
        int ct = 1, i = 0;
        while(candy > 0){
            if(ct <= candy){
                res[i] += ct;
                i = (i + 1) % n;
                candy -= ct;
                ct++;
            
} else {
                res[i] += candy;
                candy = 0;
            }
        }
        return res;
    }
}
