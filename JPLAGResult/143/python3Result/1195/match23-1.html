class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        rem = candies
        res = [0] * num_people
        curGive = 1
        idx = 0
        
while rem > 0:
            curGive = min(rem, curGive)
            res[idx % num_people] += curGive
            rem -= curGive
            curGive += 1
            idx += 1
        return res
