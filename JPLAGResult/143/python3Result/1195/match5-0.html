class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        ans = [0]*num_people
        cur = 1
        i = 0
        while candies > 0:
            temp = min(candies, cur)
            ans[i] += temp
            candies -= temp
            cur += 1
            i += 1
            i %= num_people
        return ans
