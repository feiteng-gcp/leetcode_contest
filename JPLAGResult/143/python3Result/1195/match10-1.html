class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        cur 
= 1
        i = 0
        ans = [0] * num_people
        while candies > 0:
            to_give = min(cur, candies)
            ans[i] += to_give
            candies -= to_give
            cur += 1
            i = (i + 1) % num_people
        return ans
