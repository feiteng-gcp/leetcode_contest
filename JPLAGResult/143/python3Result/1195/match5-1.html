class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        a = [0] * num_people
        i = 0
        n = 1
        while candies > 0:
            c = min(n, candies)
            a[i] += c
            candies -= c
            i = (i + 1) % num_people
            n += 1
        return a
