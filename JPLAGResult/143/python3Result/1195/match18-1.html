class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        A = [0 for i in range(num_people)]
        i = 0
        curr = 1
        while candies >0:
            A[i] += min(candies, curr)
            candies -= min(candies, curr)
            i += 1
            i = i% num_people
            curr += 1
        return A
