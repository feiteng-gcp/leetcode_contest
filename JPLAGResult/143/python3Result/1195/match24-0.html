class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        m = int(((candies * 8 + 1) ** 0.5 - 1) / 2) // num_people
        ret = [m * (i + 1) + m * (m - 1) // 2 * num_people for i in range(num_people)]
        lst = m * num_people + 1
        candies -= lst * (lst - 1) // 2
        cur_idx = 0
        while candies:
            if candies >= lst:
                ret[cur_idx] += lst
                candies -= lst
            else:
                ret[cur_idx] += candies
                candies = 0
            cur_idx += 1
            lst += 1
        return ret
