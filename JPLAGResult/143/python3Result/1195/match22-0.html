class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        res = [0]*num_people
        cnt = 1
        while candies > 0:
            if cnt <= candies:
                res[(cnt-1)%num_people] += cnt
                candies -= cnt
                cnt += 1
            else:
                res[(cnt-1)%num_people] += candies
                candies = 0
                
        
        return res
