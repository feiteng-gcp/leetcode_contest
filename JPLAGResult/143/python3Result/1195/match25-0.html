class Solution:
    def distributeCandies(self, candies: int, n: int) -> List[int]:
        
        res = [0] * n
        one_iter = (n+1) * n // 2
        
        i = 0
        while candies > n * n * i + one_iter:
            candies -= n * n * i + one_iter
            i += 1
            
        for j in range(n):
            res[j] += n * i * (i-1) // 2 + (j+1) * i
        
        for j in range(n):
            candy = n * i + (j+1)
            if candies <= candy:
                res[j] += candies
                break
            else:
                res[j] += candy
                candies -= candy
        return res
