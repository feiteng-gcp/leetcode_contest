class Solution:
    def distributeCandies(self, candies: int, num_people: int):
        fenpei = 0
        cursor = 0
        fenpeiS = [0] * num_people
        while candies > 0:
            fenpei += 1
            
if(fenpei > candies):
                fenpeiS[cursor] += candies
                candies = 0
            else:
                candies -= fenpei
                fenpeiS[cursor] += fenpei
                cursor = (cursor + 1) % num_people
        return fenpeiS
