class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        res 
= [0] * num_people
        pointer = 0
        curr = 1

        while candies > 0:
            if curr > candies:
                res[pointer] += candies
                candies = 0
            else:
                res[pointer] += curr
                candies -= curr
            pointer += 1
            pointer %= num_people
            curr += 1
        return res
