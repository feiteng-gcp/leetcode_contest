class Solution:
    def distributeCandies(self, candies: int, num: int) -> List[int]:
        res = [0] * num
        # v, m = divmod(candies, int((1+num) * (num / 2)))
        # res = [i*v for i in range(1, num+1)]
        i = 0
        idx = 0
        while candies >= i+1:
            res[idx] += i+1
            candies -= i+1
            i += 1
            idx = (idx+1) % num
        res[idx] += candies
        return res
