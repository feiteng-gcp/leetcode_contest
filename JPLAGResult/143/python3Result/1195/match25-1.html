class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        #c = 0
        #for i in range(0, 100000):
        #    t = (num_people * i + 1) * (1 + n) * n / 2

        #        i -= 1
        #        break
        #ans = []
        ans = [0] * num_people
        left_candies = candies
        for k in range(0, 0xFFFFFFFFFFFF):
            t, i = divmod
(k, num_people)
            #print(ans, t, i)
            dist = t * num_people + i + 1
            if dist > left_candies:
                ans[i] += left_candies
                break
            else:
                ans[i] += dist
                left_candies -= dist
            #print(ans, t, i)
        return ans
