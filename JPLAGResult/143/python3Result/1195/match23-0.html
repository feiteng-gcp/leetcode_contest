class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        idx, cur = 0, 1
        res = [0 for i in range(num_people)]
        while candies > 0:
            tmp = min(candies, cur)
            res[idx] += tmp
            candies -= tmp
            idx = (idx + 1) % num_people
            cur += 1
        return res
