class Solution:
    def distributeCandies(self, candies: int, num_people: int) -> List[int]:
        ans = [0] * num_people
        r = 0
        while candies:
            for i in range(num_people):
                # print(candies, ans)
                if candies > r*num_people+i+1:
                    ans[i] += r*num_people+i+1
                    candies -= r*num_people+i+1
                else:
                    ans[i] += candies
                    candies = 0
                    break
            r += 1
        return ans
