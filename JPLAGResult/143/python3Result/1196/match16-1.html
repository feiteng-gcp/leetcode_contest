'''
探索：
1. 用dp？令dp[i]代表books[:i]的最小结果。每一个新book加入时，可以选择原有的行，或另起一行。另起一行时，可以把之前的x行也一并拉下来
2. 对高度二分？不行，没法判断高度。
'''

class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        n = len(books)
        dp = [10 ** 9 for _ in range(n+1)]
        dp[0] = 0
        for i in range(1, n + 1):
            j = i
            max_cur = -1
            width = 0
            while j >= 1:
                max_cur = max(max_cur, books[j - 1][1])
                width += books
[j - 1][0]
                if width > shelf_width: break
                dp[i] = min(dp[i], dp[j - 1] + max_cur)
                j -= 1
        return dp[n]
