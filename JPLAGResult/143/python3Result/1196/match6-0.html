class Solution:
    def minHeightShelves(self, b: List[List[int]], w: int) -> int:
        n = len(b)
        dp = [float('inf')] * (n + 1)
        dp[0] = 0
        for i in range(n):
            j = i
            width = 0
            h = 0
            while j >= 0:
                width += b[j][0]
                if width > w:
                    break
                h = max(h, b[j][1])
                dp[i + 1] = min(dp[i + 1], dp[j] + h)
                j -= 1
        return dp[-1]
