class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        n = len(books)
        ans = [1<<31]*(n+1)
        ans[0] = 0
        for i in range(1, n+1):
            temp = 0
            index = i
            h = 0
            while index > 0:
                temp += books[index-1][0]
                if temp > shelf_width:
                    break
                h 
= max(h, books[index-1][1])
                ans[i] = min(ans[i], ans[index-1]+h)
                index -= 1
        return ans[n]
