from functools import lru_cache
class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        N = len(books)
        @lru_cache(None)
        def recur(i):
            if i == N:
                return 0
            ans = recur(i + 1) + books[i][1]
            #print(i, ans)
            left_width = shelf_width - books[i][0]
            floor_max_height = books[i][1]
            for j in range(i + 1, N):
                if books[j][0] > left_width:
                    break
                #print(i, ans)
                if books[j][1] > floor_max_height:
                    ans = min(ans, recur(j) + floor_max_height, recur(j + 1) + books[j][1])
                    #print(i, ans)
                    floor_max_height = books[j][1]
                else:
                    ans = min(ans, recur(j + 1) + floor_max_height)
                    
                left_width -= books[j][0]
                #print(i , j ,ans)
            #print(i, ans)
            return ans
        return recur(0)
