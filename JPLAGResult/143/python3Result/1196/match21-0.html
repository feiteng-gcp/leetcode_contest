class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        l = [0 for i in range(len(books) + 1)]
        for i in range(1, len(books) + 1):
            current_width = 0
            current_height = 0
            lst = []
            for j in range(i - 1, -1, -1):
                current_width += books[j][0]
                if current_width > shelf_width:
                    break
                current_height = max(current_height, books[j][1])
                lst.append(l[j] + current_height)
            if len(lst) == 0:
                raise ValueError('Some books are too thick for the poor shelf')
            l[i] = min(lst)
        return l[len(books)]
