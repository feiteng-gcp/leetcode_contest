class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        dp = [0 for _ in range(len(books) + 1)]
        for i, (x, y) in enumerate(books):
            dp[i + 1] = dp[i] + y
            curh = y
            curw = x
            j = i - 1
            while j >= 0:
                curw += books[j][0]
                curh = max(books[j][1], curh)
                if curw <= shelf_width:
                    dp[i + 1] = min(dp[i + 1], dp[j] + curh)
                j -= 1
        return dp[-1]
