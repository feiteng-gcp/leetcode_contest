class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        def dfs(books, shelf_width, start, mm):
            if start==len(books):
                return 0
            if start in mm:
                return mm[start]
            mm[start] = float
('inf')
            width = books[start][0]
            height = books[start][1]
            cur = start
            while width<=shelf_width:
                mm[start] = min(mm[start], height+dfs(books, shelf_width, cur+1, mm))
                cur += 1
                if cur==len(books):
                    break
                width += books[cur][0]
                height = max(height, books[cur][1])
            return mm[start]
        
        return dfs(books, shelf_width, 0, dict())
