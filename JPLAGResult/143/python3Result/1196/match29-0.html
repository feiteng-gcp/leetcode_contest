class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        dp = [math.inf] * len(books)
        for i in range(0,len(books)):
            maxHSoFar = 0
            totalWidthSoFar = 0
            for j in range(i,-1,-1):
                totalWidthSoFar += books[j][0]
                if totalWidthSoFar > shelf_width:
                    break
                maxHSoFar = max(maxHSoFar, books[j][1])
                
                totalH = maxHSoFar + (dp[j-1] if j>0 else 0)
                dp[i] = min(dp[i],totalH)
            
            #print(dp)
        
        return dp[-1]
