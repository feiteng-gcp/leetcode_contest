class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        dp=[0]
        total_w=0
        for i in range(len(books)):
            dp.append(dp[-1]+books[i][1])
            total_w+=books[i][0]
            j,high,width=i-1,int(books[i][1]),int(books[i][0])
            
while j>=0:
                width+=books[j][0]
                high=max(high,books[j][1])
                if width<=shelf_width:
                    dp[-1]=min(dp[-1],dp[j]+high)
                else:
                    break
                j-=1
        return dp[-1]
