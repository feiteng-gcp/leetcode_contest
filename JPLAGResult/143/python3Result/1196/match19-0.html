class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        n = len(books)
        dp = [0]* (n+1)
        dp[1] = books[0][1]
        for i in range(2,n+1):
            w = books[i-1][0]
            h = books[i-1][1]
            res = h+dp[i-1]
            for j in range(i-1, -1, -1):
                w+= books[j-1][0]
                h = max(h, books[j-1][1])
                if w> shelf_width: break
                res = min(res, dp[j-1]+h)
                print(res)
            dp[i] = res
        #print(dp)
        return dp[n]
