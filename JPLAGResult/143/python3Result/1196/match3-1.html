class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        n = len(books)
        dp = [0] * (n + 1)
        for i in range(1, n + 1):
            w, h = 0, 0
            dp[i] = 1000000000
            for j in range(i - 1, -1, -1):
                w 
+= books[j][0]
                if w > shelf_width:
                    break
                h = max(h, books[j][1])
                dp[i] = min(dp[i], h + dp[j])
        return dp[-1]
