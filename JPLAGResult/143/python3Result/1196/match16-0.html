class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        dp_h = [0]
        for i,x in enumerate(books):
            dp_h.append(x[1]+dp_h[i])
            h = x[1]
            w = x[0]
            i_t = i - 1
            while i_t >= 0:
                w += books[i_t][0]
                h = max(h,books[i_t][1])
                if w > shelf_width: break
                dp_h[i+1] = min(dp_h[i+1],dp_h[i_t]+h)
                i_t -= 1
        # print(dp_h)
        return dp_h[-1]
