class Solution:
    def minHeightShelves(self, books: List[List[int]], shelf_width: int) -> int:
        N = len(books)
        dp = [float("inf")] * N +[0]
        
        for i in range(N):
            w, h, j = 0, 0, i
            while j >= 0 and w + books[j][0] <= shelf_width:
                w += books[j][0]
                h 
= max(h, books[j][1])
                dp[i] = min(dp[i], dp[j-1] + h)
                j -= 1
        
        return dp[N-1]
