class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        stack = []
        for c in expression:
            if c == '(':
                stack.append(c)
            elif c == ')':
                tmp = []
                while stack[-1] != '(':
                    tmp.append(stack[-1])
                    del stack[-1]
                del stack[-1]
                if stack[-1] == '!':
                    del stack[-1]
                    stack.append(not tmp[0])
                elif stack[-1] == '&':
                    del stack[-1]
                    res = True
                    for b in tmp:
                        res = res and b
                    stack.append(res)
                elif stack[-1] == '|':
                    del stack[-1]
                    res = False
                    for b in tmp:
                        res = res or b
                    stack.append(res)
            elif c == 't':
                stack.append(True)
            elif c == 'f':
                stack.append(False)
            elif c == ',':
                continue
            else:
                stack.append(c)
        return True if stack[0] else False
