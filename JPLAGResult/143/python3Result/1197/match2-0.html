class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        def new_all(*args):
            return all(args)
        def new_any(*args):
            return any(args)
        exp = expression.replace('t', 'True').replace('f', 'False').replace('!', 'not').replace('&', 'new_all').replace('|', 'new_any')
        return eval(exp)
