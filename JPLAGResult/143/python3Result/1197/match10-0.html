class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        stack = []
        for char in expression:
            if char in ['&', '|', '!', 't', 'f']:
                stack.append(char)
            elif char == ')':
                args = []
                while stack[-1] not in ['&', '|', '!']:
                    args.append(stack.pop())
                oper = stack.pop()
                if oper == '&':
                    if all(arg == 't' for arg in args):
                        stack.append('t')
                    else:
                        stack.append('f')
                elif oper == '|':
                    if all(arg == 'f' for arg in args):
                        stack.append('f')
                    else:
                        stack.append('t')
                elif oper == '!':
                    if args[0] == 't':
                        stack.append('f')
                    else:
                        stack.append('t')
        return stack[0] == 't'
