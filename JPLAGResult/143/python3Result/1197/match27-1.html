class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        if expression == 't': return True
        if expression == 'f': return False
        if expression[0] == '!': return not self.parseBoolExpr(expression[2:-1])
        if expression[0] == '|':
            expression = list(expression[2:-1])
            d, c, o = [], '', 0
            while expression:
                i = expression.pop(0)
                if i == '(':
                    o += 1
                    c += i
                elif i == ')':
                    o -= 1
                    c += i
                elif i == ',' and o == 0:
                    d.append(c)
                    c = ''
                else:
                    c += i
            if c != '':
                d.append(c)
            return any(map(self.parseBoolExpr, d))

        if expression[0] == '&':
            expression = list(expression[2:-1])
            d, c, o = [], '', 0
            while expression:
                i = expression.pop(0)
                if i == '(':
                    o += 1
                    c += i
                elif i == ')':
                    o -= 1
                    c += i
                elif i == ',' and o == 0:
                    d.append(c)
                    c = ''
                else:
                    c += i
            if c != '':
                d.append(c)
            return all(map(self.parseBoolExpr, d))
