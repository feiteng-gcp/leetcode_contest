class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        value = []
        symbol = []
        tempv = set()
        for c in expression:
            if c in '!|&':
                symbol.append(c)
            elif c in 'tf(':
                value.append(c)
            elif c == ')':
                tempv.clear()
                while value[-1] != '(':
                    tempv.add(value.pop())
                value.pop()
                s = symbol.pop()
                if s == '!':
                    res = 'f' if 't' in tempv else 't'
                elif s == '&':
                    res = 'f' if 'f' in tempv else 't'
                elif s== '|':
                    res = 't' if 't' in tempv else 'f'
                value.append(res)
        return value[-1] == 't'
