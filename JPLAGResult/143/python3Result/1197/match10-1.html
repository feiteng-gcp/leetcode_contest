class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        
        stack = []
        for c in expression:
            if c in ['&', '|', '!']:
                stack.append(c)
            elif c == 't':
                stack.append(True)
            elif c == 'f':
                stack
.append(False)
            elif c == ')':
                cur = []
                while stack[-1] == True or stack[-1] == False:
                    cur.append(stack.pop())
                op = stack.pop()
                
                if op == '!':
                    if cur[0] == False:
                        stack.append(True)
                    else:
                        stack.append(False)
                elif op == '&':
                    stack.append(all(b == True for b in cur))
                elif op == '|':
                    stack.append(any(b == True for b in cur))
        
        return stack[0]
