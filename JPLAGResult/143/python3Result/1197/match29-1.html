class Solution:
    def parseBoolExpr(self, exp: str) -> bool:
        self.idx = 1

        def parse(exp, t):
            self.idx += 1
            data = []
            while self.idx < len(exp):
                if exp[self.idx] == ')':
                    self.idx += 1
                    break
                if exp[self.idx] == 't':
                    self.idx += 1
                    data.append(True)
                    continue
                if exp[self.idx] == 'f':
                    self.idx += 1
                    data.append(False)
                    continue
                if exp[self.idx] == ',':
                    self.idx += 1
                    continue
                if exp[self.idx] == '&':
                    self.idx += 1
                    data.append(parse(exp, 3))
                    continue
                if exp[self.idx] == '!':
                    self.idx += 1
                    data.append(parse(exp, 2))
                    continue
                if exp[self.idx] == '|':
                    self.idx += 1
                    data.append(parse(exp, 1))
                    continue
        
        
            if t == 1:
                res = False
                for i in range(len(data)):
                    res = res or data[i]
                return res
            if t == 2:
                return not data[0]
            if t == 3:
                res = True
                for i in range(len(data)):
                    res = res and data[i]
                return res
            return False


        
if exp[0] == 'f':
            return False
        if exp[0] == 't':
            return True
        if exp[0] == '!':
            return parse(exp, 2)
        if exp[0] == '|':
            return parse(exp, 1)
        if exp[0] == '&':
            return parse(exp, 3)
        return False
