class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        lvl=[[]];label=[];i=0
        while i<len(expression):
            ch=expression[i]
            if ch=="&" or ch=="|" or ch=="!":
                lvl.append([])
                label.append(ch)
                i+=2
            elif ch==")":
                ll=label.pop()
                if ll=="!":
                    if len(lvl[-1])==1:
                        re=not lvl[-1][0]
                        lvl.pop()
                        lvl[-1].append(re)
                    else:
                        raise "num incorrect"
                elif ll=="&":
                    alltrue=True
                    for x in lvl[-1]:
                        if x==False:
                            alltrue=False
                            break
                    lvl.pop()
                    lvl[-1].append(alltrue)
                else: # |
                    hastrue=False
                    for x in lvl[-1]:
                        if x==True:
                            hastrue=True
                            break
                    lvl.pop()
                    lvl[-1].append(hastrue)    
                i+=1
            elif ch!=",": #t f
                lvl[-1].append(True if ch=="t" else False)
                i+=1
            else:
                i+=1
        re=lvl[0][0]
        return re
