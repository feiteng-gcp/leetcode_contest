class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        ops = []
        values = []
        
        for c in expression:
            if c in '|&!':
                ops.append(c)
            elif c == ')':
                cur = []
                while values[-1] != '(':
                    cur.append(values.pop())
                values.pop()
                
                op = ops.pop()
                if op == '!':
                    values.append('t' if cur[0] == 'f' else 'f')
                elif op == '&':
                    if 'f' in cur:
                        values.append('f')
                    else:
                        values.append('t')
                elif op == '|':
                    if 't' in cur:
                        values.append('t')
                    else:
                        values.append('f')
            elif c == ',':
                pass
            else:
                values.append(c)
        return values[-1] == 't'
