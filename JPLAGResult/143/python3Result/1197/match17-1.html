class Solution:
    def parseBoolExpr(self, exp: str) -> bool:
        if exp[0] == '(':
            return self.parseBoolExpr(exp[1:-1])
        if exp == 't':
            return True
        elif exp == 'f':
            return False
        elif exp[0] == '!':
            return not self.parseBoolExpr(exp[2:-1])
        else:
            locs 
= []
            stack = 0
            for i in range(2,len(exp)):
                if exp[i] == '(':
                    stack += 1
                if exp[i] == ')':
                    stack -= 1
                if stack == 0 and exp[i] == ',':
                    locs += [i]
            locs += [-1]
            res = [exp[2:locs[0]]]
            for i in range(1,len(locs)):
                res += [exp[locs[i-1]+1:locs[i]]]
            res = [self.parseBoolExpr(i) for i in res]
            if exp[0] == '&':
                return all(res)
            else:
                return any(res)
