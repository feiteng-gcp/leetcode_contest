class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        if expression=='t':
            return True
        elif expression=='f':
            return False
        elif expression[0]=='!':
            return not self.parseBoolExpr(expression[2:-1])
        elif expression[0]=='&':
            es=expression[2:-1]
            i=0
            lasti=0
            bracket=0
            while i<len(es):
                if es[i]=='(':
                    bracket+=1
                elif es[i]==')':
                    bracket-=1
                if bracket==0:
                    if es[i]==',':
                        if self.parseBoolExpr(es[lasti:i]):
                            lasti=i+1
                        else:
                            return False
                i+=1
            return self.parseBoolExpr(es[lasti:])
        else:
            es=expression[2:-1]
            i=0
            lasti=0
            bracket=0
            while i<len(es):
                if es[i]=='(':
                    bracket+=1
                elif es[i]==')':
                    bracket-=1
                if bracket==0:
                    if es[i]==',':
                        if self.parseBoolExpr(es[lasti:i]):
                            return True
                        else:
                            lasti=i+1
                i+=1
            return self.parseBoolExpr(es[lasti:])
