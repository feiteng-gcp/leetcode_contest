class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        if expression[0] == 't':
            return True
        if expression[0] == 'f':
            return False
        if expression[0] == '!':
            return not self.parseBoolExpr(expression[2:-1])
        exp = expression[2:-1]
        if expression[0] == '&':
            res = True
            start = 0
            while start < len(exp):
                i = start
                if exp[i] in '|&!':
                    num = 1
                    i += 2
                    while num != 0:
                        if exp[i] == '(':
                            num += 1
                        elif exp[i] == ')':
                            num -= 1
                        i += 1
                    res = res and self.parseBoolExpr(exp[start:i])
                    start = i + 1
                elif exp[i] == 't':
                    start += 2
                elif exp[i] == 'f':
                    return False
            return res
        res = False
        start = 0
        while start < len(exp):
            i = start
            if exp[i] in '|&!':
                num = 1
                i += 2
                while num != 0:
                    if exp[i] == '(':
                        num += 1
                    elif exp[i] == ')':
                        num -= 1
                    i += 1
                res = res or self.parseBoolExpr(exp[start:i])
                start = i + 1
            elif exp[i] == 't':
                return True
            elif exp[i] == 'f':
                start += 2
        return res
