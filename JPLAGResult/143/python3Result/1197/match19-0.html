class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        opr = []
        val = []
        for c in expression:
            if c in '!&|(,':
                opr.append(c)
            elif c == ')':
                a, o = True, False
                while opr and opr[-1] == ',':
                    v = val.pop()
                    a = a and v
                    o = o or v
                    opr.pop()
                p = val.pop()
                a = a and p
                o = o or p
                opr.pop()
                t = opr.pop()
                if t == '&':
                    val.append(a)
                elif t == '|':
                    val.append(o)
                else:
                    val.append(not p)
            elif c == 't':
                val.append(True)
            else:
                val.append(False)
        return val[0]
