class Solution:
    def parseBoolExpr(self, exp: str) -> bool:
        # print(exp)

        if exp[0] == 't':
            return True
        if exp[0] == 'f':
            return False
        if exp[0] == '!':
            return not self.parseBoolExpr(exp[2:-1])
        if exp[0] == '&' or exp[0] == '|':
            ans = []
            cnt = 0
            buf = ''

            for i in range(2, len(exp) -1):
                if exp[i] == '(':
                    cnt += 1
                if exp[i] == ')':
                    cnt -= 1
                if exp[i] == ',' and cnt == 0:
                    ans.append(self.parseBoolExpr(buf))
                    buf = ''
                    continue
                buf += exp[i]
            ans.append(self.parseBoolExpr(buf))
            if exp[0] == '&':
                if False in ans:
                    return False
                return True
            else:
                if True in ans:
                    return True
                return False
