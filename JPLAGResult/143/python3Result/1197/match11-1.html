class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        def evaluate(l, operator):
            if operator == '!':
                return not l[0]
            elif operator == '&':
                res = True
                for item in l:
                    res = res and item
                return res
            elif operator == '|':
                res = False
                for item in l:
                    res = res or item
                return res
            
        
        stack = []
        for expr in expression:
            if expr == '|' or expr == '&' or expr == '!' or expr == '(':
                stack.append(expr)
            elif expr == 't' or expr == 'f':
                stack
.append(expr == 't')
            elif expr == ')':
                temp = []
                while stack[-1] != '(':
                    temp.append(stack.pop())
                stack.pop()
                operator = stack.pop()
                stack.append(evaluate(temp, operator))
                
        
        return stack[0]
