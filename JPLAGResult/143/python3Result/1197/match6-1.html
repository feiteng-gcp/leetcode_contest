class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        stack = []
        for s in expression:
            if s != ')' and s != ',':
                stack.append(s)
            elif s == ')':
                out = stack.pop()
                temp = []
                while out != '(':
                    temp.append(out)
                    out = stack.pop()
                do 
= stack.pop()
                if do == '&':
                    new = 'f' if 'f' in temp else 't'
                elif do == "|":
                    new = 't' if 't' in temp else 'f'
                elif do == '!':
                    new = 't' if 'f' in temp else 'f'
                # print(temp)
                # print(new)
                stack.append(new)
        return True if stack[-1] == 't' else False
