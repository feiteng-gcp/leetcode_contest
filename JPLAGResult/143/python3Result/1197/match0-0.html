class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        def AND(*args):
            return all(args)
        
        def OR(*args):
            return any(args)
        
        def NOT(arg):
            return not arg
        
        expression = expression.replace('&', 'AND').replace('|', 'OR').replace('!', 'NOT').replace('t', 'True').replace('f', 'False')
        return eval(expression)
