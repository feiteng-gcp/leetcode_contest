class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        e = expression
        e = e.replace('t', 'True').replace('f', 'False')
        final 
= []
        realm = []
        i = 0
        while i < len(e):
            c = e[i]
            if c == '&':
                realm.append(' and ')
            elif c == '|':
                realm.append(' or ')
            elif c == '!':
                realm.append(' not ')
                final.append(' not ')
            elif c == '(':
                final.append('(')
            elif c == ')':
                final.append(')')
                if len(realm) > 0:
                    realm.pop()
            elif c == ',':
                if len(realm) > 0 and realm[-1] != ' not ':
                    final.append(realm[-1])
            else:
                final.append(c)
            i += 1
        print(''.join(final))
        return eval(''.join(final))
