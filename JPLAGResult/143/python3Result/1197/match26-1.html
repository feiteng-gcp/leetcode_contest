class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        n = len(expression)
        def parse(start, end):
            i = start
            res = []
            while i < end:
                if expression[i] == 't':
                    res.append(True)
                elif expression[i] == 'f':
                    res.append(False)

                    res.append(ans)
                    i = m[i+1]
                elif expression[i] == '&':
                    pieces = parse(i + 2, m[i+1])
                    ans = True
                    for piece in pieces:
                        ans &= piece
                    i = m[i+1]
                    res.append(ans)
                elif expression[i] == '|':
                    pieces = parse(i + 2, m[i+1])
                    ans = False
                    for piece in pieces:
                        ans |= piece
                    i = m[i+1]
                    res.append(ans)
                i += 1
            return res
