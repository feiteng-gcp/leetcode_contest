class Solution:
    
    def _and(self, s):
        res = True
        n = len(s)
        i = 0
        while i< n:
            if res == False: break
            if s[i] == 'f': return False
            elif s
[i] == '&':
                j = i+2
                cnt = 1
                while cnt!=0:
                    if s[j] == '(': cnt += 1
                    elif s[j] == ')': cnt -= 1
                    j += 1
                res = res and self._and(s[i+2:j-1])
                i = j-1
            elif s[i] == '!':
                j = i+2
                cnt = 1
                while cnt!=0:
                    if s[j] == '(': cnt += 1
                    elif s[j] == ')': cnt -= 1
                    j += 1
                res = res and self._not(s[i+2:j-1])
                i = j-1
            elif s[i] == '|':
                j = i+2
                cnt = 1
                while cnt!=0:
                    if s[j] == '(': cnt += 1
                    elif s[j] == ')': cnt -= 1
                    j += 1
                res = res and self._or(s[i+2:j-1])
                i = j-1
            i += 1      
        return res
    
    def _or(self, s):
        res = False
        n = len(s)
        i = 0
        while i< n:
            if res == True: break
            if s[i] == 't': return True
            elif s
[i] == '&':
                j = i+2
                cnt = 1
                while cnt!=0:
                    if s[j] == '(': cnt += 1
                    elif s[j] == ')': cnt -= 1
                    j += 1
                res = res or self._and(s[i+2:j-1])
                i = j-1
            elif s[i] == '!':
                j = i+2
                cnt = 1
                while cnt!=0:
                    if s[j] == '(': cnt += 1
                    elif s[j] == ')': cnt -= 1
                    j += 1
                res = res or self._not(s[i+2:j-1])
                i = j-1
            elif s[i] == '|':
                j = i+2
                cnt = 1
                while cnt!=0:
                    if s[j] == '(': cnt += 1
                    elif s[j] == ')': cnt -= 1
                    j += 1
                res = res or self._or(s[i+2:j-1])
                i = j-1
            i += 1      
        return res

    def _not(self,s):
        if s[0] == 't': return False
        elif s[0] == 'f': return True
        elif s[0] == '&':
            j = 2
            cnt = 1
            while cnt!=0:
                if s[j] == '(': cnt += 1
                elif s[j] == ')': cnt -= 1
                j += 1
            return not self._and(s[2:j-1])
        elif s[0] == '!':
            j = 2
            cnt = 1
            while cnt!=0:
                if s[j] == '(': cnt += 1
                elif s[j] == ')': cnt -= 1
                j += 1
            return not self._not(s[2:j-1])
        elif s[0] == '|':
            j = 2
            cnt = 1
            while cnt!=0:
                if s[j] == '(': cnt += 1
                elif s[j] == ')': cnt -= 1
                j += 1
            return not self._or(s[2:j-1])
        
        
    def parseBoolExpr(self, s: str) -> bool:
        if s[0] == 't': return True
        elif s[0] == 'f': return False
        if s[0] == '&':
            res = self._and(s[2:-1])
        elif s[0] == '|':
            res = self._or(s[2:-1])
        elif s[0] == '!':
            res = self._not(s[2:-1])
        return res
