import operator
import functools

class Solution:
    def parseBoolExpr(self, exp: str) -> bool:
        def recurse(i, j):
            # print(i, j)
            if exp[i] == 't':
                return True
            if exp[i] == 'f':
                return False
            if exp[i] == '!':
                return not recurse(i + 2, j - 1)
            l = []
            curr = i + 2
            level = 0
            for k in range(i + 2, j - 1):
                if exp[k] == '(':
                    level += 1
                elif exp[k] == ')':
                    level -= 1
                elif exp[k] == ',' and not level:
                    l.append(recurse(curr, k))
                    curr = k + 1
            l.append(recurse(curr, j - 1))
            # print(l)
            if exp[i] == '&':
                return functools.reduce(operator.and_, l)
            else:
                return functools.reduce(operator.or_, l)
        return recurse(0, len(exp))
