from functools import reduce
class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        ops = []
        for char in expression:
            if char in "!|&(":
                ops.append(char)
            elif char == "t":
                ops.append(True)
            elif char == "f":
                ops.append(False)
            elif char in ")":
                t = ops.pop()
                seq = []
                while t != "(":
                    seq.append(t)
                    t = ops.pop()
                op = ops.pop()
                if op == "|":
                    ops.append(reduce(lambda a, b: a | b, seq, False))
                elif op == "&":
                    ops.append(reduce(lambda a, b: a & b, seq, True))
                elif op == "!":
                    ops.append(not seq[0])
                    assert len(seq) == 1
        print(ops)
        return ops[0]
