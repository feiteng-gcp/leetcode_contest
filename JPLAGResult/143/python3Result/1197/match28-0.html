class Solution:
    def parseBoolExpr(self, exp: str) -> bool:
        if exp == 'f': return False
        elif exp == 't': return True
        elif exp[0] == '!':
            return not self.parseBoolExpr(exp[2:-1])
        elif exp[0] == '&':
            stack = 0
            cur = 2
            last = 2
            res = []
            while cur < len(exp):
                if exp[cur] == '(': stack += 1
                elif exp[cur] == ')': stack -= 1
                elif stack == 0 and exp[cur] == ',': 
                    res.append(self.parseBoolExpr(exp[last:cur]))
                    last = cur+1
                cur += 1
            res.append(self.parseBoolExpr(exp[last:-1]))
            return all(res)
        elif exp[0] == '|':
            stack = 0
            cur = 2
            last = 2
            res = []
            while cur < len(exp):
                if exp[cur] == '(': stack += 1
                elif exp[cur] == ')': stack -= 1
                elif stack == 0 and exp[cur] == ',': 
                    res.append(self.parseBoolExpr(exp[last:cur]))
                    last = cur+1
                cur += 1
            res.append(self.parseBoolExpr(exp[last:-1]))
            return any(res)
