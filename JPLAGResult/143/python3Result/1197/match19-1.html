class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        
        stack = collections.deque()
        
        for c in expression:
            if c == ')':
                true = False
                false = False
                popc = stack.pop()
                while popc != '(':
                    if popc:
                        true = True
                    else:
                        false = True
                    popc = stack.pop()
                popc 
= stack.pop()
                if popc == '&':
                    stack.append(not false)
                elif popc == '|':
                    stack.append(true)
                else:
                    stack.append(false) 
            elif c == 't':
                stack.append(True)
            elif c == 'f':
                stack.append(False)
            elif c == ',':
                continue
            else:                   
                stack.append(c)
        while len(stack) > 1:
            true = False
            false = False
            popc = stack.pop()
            while popc != '(':
                if popc:
                    true = True
                else:
                    false = True
                popc = stack.pop()
            popc = stack.pop()
            if popc == '&':
                stack.append(not false)
            elif popc == '|':
                stack.append(true)
            else:
                stack.append(false)  
        return stack[-1]
