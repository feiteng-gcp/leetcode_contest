import re

class Solution:
    def parseBoolExpr(self, expression: str) -> bool:
        if expression == 't':
            return True
        if expression == 'f':
            return False
        exp = expression[2:-1]
        if expression[0] == '!':
            return not self.parseBoolExpr(exp)
        if expression[0] == '&':
            brackets = 0
            pos = 0
            for i in range(len(exp)):
                if exp[i] == '(':
                    brackets += 1
                if exp[i] == ')':
                    brackets -= 1
                if brackets == 0 and exp[i] == ',':
                    if not self.parseBoolExpr(exp[pos:i]):
                        return False
                    pos = i + 1
            return self.parseBoolExpr(exp[pos:])
        if expression[0] == '|':
            brackets = 0
            pos = 0
            for i in range(len(exp)):
                if exp[i] == '(':
                    brackets += 1
                if exp[i] == ')':
                    brackets -= 1
                if brackets == 0 and exp[i] == ',':
                    if self.parseBoolExpr(exp[pos:i]):
                        return True
                    pos = i + 1
            return self.parseBoolExpr(exp[pos:])
