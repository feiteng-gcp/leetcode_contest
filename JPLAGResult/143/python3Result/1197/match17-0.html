class Solution:
	def parseBoolExpr(self, expression):
		if len(expression) == 1:
			return True if expression == 't' else False

		operation = expression[0]
		if operation == '!':
			return not self.parseBoolExpr(expression[2: len(expression) - 1])
		else:
			subExpressions = []
			s = 2
			opened = 0
			for i in range(s, len(expression) - 1):
				if expression[i] == '(':
					opened += 1
				if expression[i] == ')':
					opened -= 1
				if expression[i] == ',' and opened == 0:
					subExpressions.append(expression[s: i])
					s = i + 1
			subExpressions.append(expression[s: i + 1])

			if operation == '|':
				res = False
				for subExpr in subExpressions:
					res = res or self.parseBoolExpr(subExpr)
			else:
				res = True
				for subExpr in subExpressions:
					res = res and self.parseBoolExpr(subExpr)
		return res
