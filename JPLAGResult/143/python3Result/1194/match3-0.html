class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        N = int(math.log2(label))
        res = [None]*N
        cur = label
        n = N
        while n>0:
            res[n-1] = cur
            cur = cur//4
            n -= 2
        
        cur = label
        layer_cnt = 2**(N-1)
        layer_end = 2**N-1
        layer_start = layer_end-layer_cnt+1
        prev = cur//2
        cur = layer_start+layer_end-prev
        N = N-1
        
    
        while N>0:
            res[N-1] = cur
            cur = cur//4
            N -= 2
        return [1]+res
