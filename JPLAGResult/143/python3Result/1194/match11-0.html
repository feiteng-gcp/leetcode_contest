class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        def get0(l):
            if l % 2:
                return 2 ** l
            else:
                return 2 ** (l+1) - 1
        
        def parent(x):
            l = int(math.log2(label))
            cur0 = get0(l)
            idx = abs(x - cur0)
            if (l - 1) % 2:
                return get0(l-1) + (idx // 2)
            else:
                return get0(l-1) - (idx // 2)

        
        res = [label]
        while label != 1:
            label = parent(label)
            res.append(label)
        return res[::-1]
