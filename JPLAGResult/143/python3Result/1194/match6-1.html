"""
1     [2^0, 2^1)            0
2,3   [2^1, 2^2)  r         1
4-7   [2^2, 2^3)            2
8-15  [2^3, 2^4)  r         3
"""



class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        depth = 0
        res = [label]
        while 0 < label:
            label //= 2
            depth += 1
        depth -= 1
        while 0 < depth:
            res.append(self.getParent(res[-1], depth))
            depth -= 1
        res.reverse()
        return res
        
    
    def getParent(self, node: int, depth: int) -> int:
        reverse = (depth % 2 == 1)
        if reverse:
            node = 3 * 2**depth - 1 - node
        parent_node = node // 2
        if not reverse:
            parent_node = 3 * 2**(depth-1) - 1 - parent_node
        return parent_node
