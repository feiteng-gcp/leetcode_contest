def flip(n):
    if n == 1:
        return 1

    leftStart = 2
    while leftStart <= n:
        leftStart *= 2
    leftEnd = leftStart - 1
    leftStart /= 2
    midPoint = leftStart + (leftEnd - leftStart)/2

    diff = abs(n - midPoint)
    if n > midPoint:
        n -= 2 * diff
    else:
        n += 2 * diff
    return int(n)

class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        vals = []
        while label > 0:
            vals.append(label)
            label //= 2
        vals = vals[::-1]
        for i in range(len(vals) - 2, 0, -2):
            vals[i] = flip(vals[i])

        return vals
