class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        cur1 = [1]
        while cur1[-1] < label:
            cur1 += [cur1[-1]*2+1]
        cur = [(1,1)]
        for i in range(len(cur1)-1):
            cur += [(cur1[i]+1,min(label,cur1[i+1]))]
        res = [label]
        while res[-1]!= 1:
            cur.pop()
            label = sum(cur[-1])-label//2
            res += [label]
        return res[::-1]
