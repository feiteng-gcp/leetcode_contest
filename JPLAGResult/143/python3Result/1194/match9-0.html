class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        def get_parent(label):
            row = int(math.log(label, 2))
            if row % 2:
                label = 3 * 2 ** row - 1 - label
                return label // 2
            else:
                label //= 2
                return 3 * 2 ** (row - 1) - 1 - label
        
        path = [label]
        while label != 1:
            label = get_parent(label)
            path.append(label)
        return path[::-1]
