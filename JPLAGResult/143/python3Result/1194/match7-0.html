class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        level, num = 0, 1
        while label >= num:
            num *= 2
            level += 1
        flag = num % 2
        start = (2**(level-1))*3-1 - label if flag else label
        res = []
        while start >= 1:
            if not flag:
                res.append(start)
            else:
                res.append((2**(level-1))*3-1-start)
            flag = not flag
            start //= 2
            level -= 1
        return list(reversed(res))
