class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        level = 1
        s = 2
        while s - 1 < label:
            level += 1
            s *= 2
        ans_list = []

        if level % 2 == 0:
            label = s - label + s // 2 - 1
        
while level > 0:
            if level % 2 == 1:
                ans_list.append(label)
            else:
                ans_list.append(s - label + s // 2 - 1)
            label = label // 2
            s = s // 2
            level -= 1
        return ans_list[::-1]
