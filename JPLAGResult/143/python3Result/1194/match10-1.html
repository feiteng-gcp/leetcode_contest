class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        row = int(math.log2(label))
        if row % 2 == 1:
            x = 2**(row+1)-label-1+2**row
        else:
            x = label
        res = [x]
        while res[-1] > 1:
            res.append(res[-1]//2)
        
        def f(x,y):
            if y % 2 == 0:
                return x
            return 2**(y+1)-x-1+2**y
        res 
= res[::-1]
        for i in range(len(res)):
            res[i] = f(res[i],i)
        return res
