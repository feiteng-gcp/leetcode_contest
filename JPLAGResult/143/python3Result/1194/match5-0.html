class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        def largerTwoPow(x):
            result = 1
            while result <= x:
                result *= 2
            return result
        
        res = [label]
        while label > 1:
            p = label // 2
            a = largerTwoPow(p) // 2
            p = a + a - (p - a) - 1
            res.append(p)
            label = p
        return list(reversed(res))
