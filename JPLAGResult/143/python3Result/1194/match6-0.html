class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        max_num = 2
        depth = 1
        while max_num - 1 < label:
            max_num *= 2
            depth += 1
        res = []
        while label >= 1:
            res.append(label)
            start = 2 ** (depth - 1)
            end = 2 ** depth - 1
            label = start + (end - label)
            label //= 2
            depth -= 1
        res.reverse()
        return res
