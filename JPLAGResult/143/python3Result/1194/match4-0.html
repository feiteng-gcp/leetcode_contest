class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        result = []
        n = label
        
        while n:
            result.insert(0, n)
            n //= 2
        
        level = len(result) - 1
        
        for i, num in enumerate(result):
            if (level - i) % 2 == 1:
                result[i] = 2 ** i + 2 ** (i + 1) - 1 - num
        
        return result
