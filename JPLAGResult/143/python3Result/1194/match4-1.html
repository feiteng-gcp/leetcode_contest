class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        result = []
        while label != 1:
            result = [label] + result
            
            label = label//2
        result = [1] + result
        print(result)
        num 
= len(result) % 2
        for i,x in enumerate(result):
            if i % 2 == num:
                result[i] = 2**i + 2**(i+1) - 1 - x
        return result
