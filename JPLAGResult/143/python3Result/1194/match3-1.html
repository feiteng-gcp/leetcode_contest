class Solution:
    def pathInZigZagTree(self, label: int) -> List[int]:
        totalEntries = int(math.log2(label) // 1 + 1)
        
        te = totalEntries
        final = [0] * totalEntries
        curr = label
        while(totalEntries > 0):
            final[totalEntries -1] = curr
            curr = curr // 2
            totalEntries -= 1
        
        totalEntries = te
        for i in range(totalEntries -2, 0, -2):
            currNum = final[i]
            minNum = 2**i 
            maxNum = (2**(i+1)) - 1
            newNum = maxNum - currNum + minNum
            final[i] = newNum
        
        return final
