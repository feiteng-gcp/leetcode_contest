class Solution {
    public int reachNumber(int target) {
        if(target<0) target = 0-target;
        int index = (int)Math.sqrt(target*2);
        while( index*(index+1)<target*2 )
            index++;
        if(index*(index+1)==target*2)
            return index;
        int dis = (index*(index+1)-2*target)/2;
        while(dis%2!=0){
            index++;
            dis+=index;
        }
        return index;
    }
}
