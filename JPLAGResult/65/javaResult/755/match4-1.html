class Solution {
    public int reachNumber(int target) {
        target = Math.abs(target);
		int i = 1;
		int sum = 0;
		while (true) {
			sum += i;
			if (sum == target) {
				return i;
			} else if (sum > target) {
				if ((sum - target) % 2 == 0) {
					return i;
				}
			}
			i++;
		}
    }
}
