class Solution {
    public int reachNumber(int target) {
        return findMinimumSteps(target);
    }

    private int findMinimumSteps(int dest)

    {
        if (dest == 0) {
            System.out.println(dest);
            return dest;
        }
        if (dest < 0) dest *= -1;
        int sum = 0;
        for (int i = 1; i < Integer.MAX_VALUE; i++) {
            sum= i * (i + 1) / 2;
            if (sum % 2 == 0 && dest % 2 == 0 && dest <= sum) {
                return i;
            } else if (sum % 2 == 1 && dest % 2 == 1 && dest <= sum) {
                return i;
            }
        }
        return 0;
    }
}
