class Solution {
    
    public int reachNumber(int target) {
        int source=0,i=1,n;
        if(target<0)
            target=-target;
        n=source-target;
        
        
while(n<0 || n%2!=0){
            source+=i;
            i+=1;
            n=source-target;
        }
    
        return i-1;
    }
}
