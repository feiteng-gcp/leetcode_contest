class Solution {
    public int reachNumber(int target) {
        if (target < 0) {
            target = - target;
        }
        System.out.println(target);
        long sum = 0;
        for (int i = 1; i < 100000; i++) {
            sum = sum + i;
            if (sum == target) {
                return i;
            }
            if (sum > target) {
                if (sum % 2 == target % 2) {
                    return i;
                
}
            }
        }
        return 1;
    }
}
