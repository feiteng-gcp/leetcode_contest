class Solution {
    public int reachNumber(int target) {
    target = Math.abs(target);
    int can = (int) Math.floor( Math.sqrt(target*2));
    while(can*(can+1)/2 < target) can++;
    int sum =  can*(can+1)/2;
    if(target % 2== sum % 2) return can;
    while(sum % 2 != target %2) {
      can++;
      sum = can*(can+1)/2;
    }
    return can;
  }
}
