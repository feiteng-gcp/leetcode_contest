class Solution {
    public int reachNumber(int target) {
        int sum = 0, i = 0;
        target = Math.abs(target);
        while(!check(sum, target)) {
            sum += ++i;
        }
        return i;
    }
    private boolean check(int sum, int target) {
        
if(sum == target)   return true;
        if(sum > target && (sum - target) % 2 == 0) return true;
        return false;
    }
}
