class Solution {
    public int reachNumber(int target) {

        //target = 21;
        target = Math.abs(target);
        
        if (target == 2) return 3;
        
        int sum = 0;
        int index = 0;
        
        
        for (int i = 1; i < target + 1; i++)
        {
            sum = sum + i;
            //System.out.println("sum " + sum);
            
            if (sum >= target && (sum - target)%2 == 0)
            {
                index = i;
                //System.out.println("Index " + index);
                return index;
            }
        }
        
        return index;
    }
}
