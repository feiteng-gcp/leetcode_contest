class Solution {
    public int reachNumber(int target) {
        if(target<0) target = 0-target;
        int res = (int)Math.sqrt(target*2);
        while( res*(res+1)<target*2 )
            res++;
        if(res*(res+1)==target*2)
            return res;
        int distan = (res*(res+1)-2*target)/2;
        while(distan%2!=0){
            res++;
            distan=res+distan;
        }
        return res;
    }
}
