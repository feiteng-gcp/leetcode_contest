class Solution {
    public int reachNumber(int target) {
        if (target < 0) target *= -1;
        int sum = 0;
        for (int i=1 ; i<=target+1; i++) {
            sum += i;
            if (sum == target)
                return i;
            if (sum > target){
                if ((sum-target)%2 == 0)
                    return i;
            }
        }
        return target+1;
    }
}
