class Solution {
    public int reachNumber(int target) {
                if (target < 0) target = -target;
        int max = (int)Math.ceil(Math.sqrt(2*target));
//            log.info("max: {}", max);
            int min = 0;
            int m=min;
            while (min < max) {
                m=min+(max-min)/2;
//                log.info("m: {}", m);
                if (m*(m+1)>=2*target)max=m;
            else min = m + 1;
        }
        for (;;) {
            int sum = min * (min + 1) / 2 + target;
            if (sum % 2 == 1) ++min;
            else break;
        }
        return min;
    }
}
