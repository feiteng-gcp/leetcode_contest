class Solution {
    public int reachNumber(int target) {
        target = Math.abs(target);
        if (target == 0) {
            return 0;
        }
        int count = 1;
        int sum = 0;
        while (!(target <= sum && target % 2 == sum % 2)) {
            sum += count;
            count++;
        }
        return count - 1;
    }
}
