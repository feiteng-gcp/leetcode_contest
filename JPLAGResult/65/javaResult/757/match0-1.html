class Solution {
    public boolean pyramidTransition(String bottom, List<String> allowed) {
        Map<String,List<Character>> map=new HashMap<>();
        for(String str:allowed){
            String sub=str.substring(0,2);
            map.put(sub,map.getOrDefault(sub,new ArrayList<Character>()));
            
map.get(sub).add(str.charAt(2));
        }
        return pyramidTransition(bottom,map);
    }
     public boolean pyramidTransition(String bottom, Map<String,List<Character>> allowed) {
         if(bottom.length()==1) return true;
         List<List<Character>>mList=new ArrayList<>();
        for(int i=0;i<bottom.length()-1;i++){
            String sub=bottom.substring(i,i+2);
            if(!allowed.containsKey(sub)){
                return false;
            }else{
                mList.add(allowed.get(sub));
            }
        }
        
List<String> result=new ArrayList<>();
        buildSet(mList,0,result,"",allowed);
        for(String str:result){
            if(pyramidTransition(str,allowed)){
                return true;
            }
        } 
         return false;
    }
    
    public void buildSet(List<List<Character>>mList, 
int i, List<String> result,String comb,  Map<String,List<Character>> allowed){
        if(i==mList.size()){
            result.add(comb);
            return;
        }
        List<Character> list=mList.get(i);
        for (char c : list) {
			if (comb.length() == 0 || (comb.length() > 0
					&& allowed.containsKey(comb.substring(comb.length() - 1, comb.length()) + c))) {
				buildSet(mList, i + 1, result, comb + c, allowed);
			}
		}
    }
}
