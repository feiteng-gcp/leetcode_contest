class Solution {
    public int[] pourWater(int[] heights, int V, int K) {
        if(heights == null || heights.length == 0){
            return new int[0];
        }
        int len = heights.length;
        while(V > 0){
            boolean flag = true;
            int k = K;
            while(k > 0 && heights[k - 1] <= heights[k]){
                k--;
            
}
            if(k >= 0 && heights[k] < heights[K]){
                while(heights[k] == heights[k + 1]){
                    k++;
                }
                heights[k]++;
                flag = false;
            }
            if(flag){
                k = K;
                while(k < len - 1 && heights[k] >= heights[k + 1]){
                    k++;
                
}
                if(k < len && heights[k] < heights[K]){
                    while(heights[k] == heights[k - 1]){
                        k--;
                    }
                    heights[k]++;
                    flag = false;
                }
                if(flag){
                    heights[K] ++;
                }
            }
            V--;
        }
        return heights;
    }
}
