import java.util.ArrayList;
import java.util.Arrays;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;
import java.util.concurrent.ArrayBlockingQueue;

// 755

public class Solution {

	public int[] pourWater(int[] heights, int V, int K) {

		for (int i = 0; i < V; i++) {
			if (!applyLeft(heights, K)) {
				applyRight(heights, K);
			}
		}

		return heights;
	}

	private void applyRight(int[] heights, int k) {
		int lowest = -1;
		for (int i = k + 1; i < heights.length; i++) {
			if (heights[k] == heights[i]) {
				continue;
			}
			if (heights[k] < heights[i]) {
				break;
			}
			lowest = i;
			while (i + 1 < heights.length && heights[lowest] >= heights[i + 1]) {
				lowest = ++i;
			}
			while (i > 0 && heights[lowest] == heights[i - 1]) {
				lowest = --i;
			}
			break;
		}
		if (lowest != -1) {
			heights[lowest]++;
		} else {
			heights[k]++;
		}
	}

	private boolean applyLeft(int[] heights, int k) {
		for (int i = k - 1; i >= 0; i--) {
			if (heights[k] < heights[i]) {
				return false;
			}
			if (heights[k] == heights[i]) {
				continue;
			}
			int lowest = i;
			while (i> 0 && heights[lowest] >= heights[i - 1]) {
				lowest = --i;
			}
			while (i + 1 < heights.length && heights[lowest] == heights[i + 1]) {
				lowest = ++i;
			}
			heights[lowest]++;
			return true;
		}

		return false;
	}

	public static void main(String[] args) {
		System.out.println(
				Arrays.toString(new Solution().pourWater(new int[] { 1, 2, 3, 4, 3, 2, 1, 2, 3, 4, 3, 2, 1 }, 10, 2)));
		System.out.println(Arrays.toString(new Solution().pourWater(new int[] { 2, 1, 1, 2, 1, 2, 2 }, 4, 3)));
		System.out.println(Arrays.toString(new Solution().pourWater(new int[] { 1, 2, 3, 4 }, 2, 2)));
		System.out.println(Arrays.toString(new Solution().pourWater(new int[] { 3, 1, 3 }, 5, 1)));
	}
}
