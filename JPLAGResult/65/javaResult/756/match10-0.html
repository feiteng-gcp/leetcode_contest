class Solution {
    public int[] pourWater(int[] heights, int V, int k) {
        int n = heights.length;
        
        for (int i = 1; i <= V; i++) {
            boolean left = false;
            
            int cur = k;
            int j = k - 1;
            
            while (j >= 0 && heights[cur] >= heights[j]) {
                cur = j;
                j--;   
            }
            
            if (j + 1 >= 0 && j != k - 1 && heights[j + 1] != heights[k]) {
                while (j + 2< k && heights[j + 1] == heights[j + 2]) {
                    
                    j++;
                }
                heights[j + 1]++;
                left = true;
            }
            
            
            boolean right = false;
            if (!left) {
                cur = k;
                j = k + 1;
                while (j < n && heights[cur] >= heights[j]) {
                    cur = j;
                    j++;
                }
                
                if (cur < n && cur != k && heights[cur] != heights[k]) {
                    while (cur - 1 > k && heights[cur] == heights[cur - 1]) {
                        cur--;
                    }
                    heights[cur]++;
                    right = true;
                }
            }
            
            if (!right && !left) {
                heights[k]++;
            }
            
            //System.out.println(Arrays.toString(heights));
            
        }
        return heights;
        
        
    }
}
