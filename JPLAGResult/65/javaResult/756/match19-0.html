class Solution {
    public int[] pourWater(int[] heights, int V, int K) {
        
        int n = heights.length;
        
        for (int i = 0; i < V; ++i) {
            
            // find the left 
            int target = K;
            
            while (true) {
                int left = -1;
                for (int j = target-1; j >= 0; --j) {
                
                    if (heights[j] > heights[j+1]) break;
                    else if (heights[j] < heights[j+1]) {
                        left = j;
                        break;
                    }

                }
            
                if (left != -1) {
                    target = left;
                    continue;
                }
                
                int right = -1;
                for (int j = target+1; j < n; ++j) {
                    
                    if (heights[j] > heights[j-1]) break;
                    else if (heights[j] < heights[j-1]) {
                        
                        right = j;
                        break;
                        
                    }
                    
                }
                if (right != -1) {
                    target = right;
                    continue;
                }
                heights[target] += 1;
                break;
                
            }
            
        }
        return heights;
        
    }
}
