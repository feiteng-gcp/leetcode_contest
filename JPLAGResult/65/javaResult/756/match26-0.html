class Solution {
    public int[] pourWater(int[] heights, int V, int K) {
        while (V > 0) {
            V--;
            boolean isLeft = update(heights, K, true);
            if (isLeft) continue;
            boolean isRight = update(heights, K, false);
            if (isRight) continue;
            heights[K]++;
        }
        return heights;
    }
    private boolean update(int[] heights, int K, boolean left) {
        int ind = K + (left ? -1 : 1), pour = K;
        while (ind >= 0 && ind < heights.length) {
            if (heights[ind] > heights[K]) return false;
            int next = ind + (left ? -1 : 1);

            if (next < 0 || next >= heights.length || (heights[next] > heights[ind])) {
                if (heights[ind] < heights[pour]) pour = ind;
                if (K != pour) {
                    heights[pour]++;
                    return true;
                } else return false;
            }

            if (heights[ind] < heights[pour]) pour = ind;
            ind = next;
        }
        return false;
    }
}
