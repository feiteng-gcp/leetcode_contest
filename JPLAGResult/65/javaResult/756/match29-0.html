class Solution {
    public int[] pourWater(int[] heights, int V, int K) {
        // dfs
        for (int i = 0; i < V; i++) {
            if (K == 0 || heights[K] < heights[K - 1] || !search(heights, K - 1, true)) {
                if (K == heights.length - 1 || heights[K] < heights[K + 1] || !search(heights, K + 1, false)) {
                    heights[K]++;
                }
            }
            /*for (int j = 0; j < heights.length; j++) {
                System.out.print(heights[j] + " ");
            }
            System.out.println();*/
        }
        return heights;
    }
    
    public boolean search(int[] heights,
                          int position,
                          boolean left) {
        if (position < 0 || position >= heights.length) {
            return false;
        }
        if (left) {
            if (position == 0 || heights[position] < heights[position - 1] || !search(heights, position - 1, true)) {
                if (heights[position] >= heights[position + 1]) {
                    return false;
                } else {
                    heights[position]++;
                    return true;
                }
            } else {
                return true;
            }
        } else {
            if (position == heights.length - 1 || heights[position] < heights[position + 1] || !search(heights, position + 1, false)) {
                if (heights[position] >= heights[position - 1]) {
                    return false;
                } else {
                    heights[position]++;
                    return true;
                }
            } else {
                return true;
            }
        }
    }
}
