class Solution {
    public int[] pourWater(int[] heights, int V, int K) {

        int n = heights.length;

        for (int v = 0; v < V; v++) {

            // find left
            int lowest = Integer.MAX_VALUE;
            int pos = -1;
            for (int i = K - 1; i >= 0; i--) {
                if (heights[i] > heights[i + 1]) break;
                if (lowest > heights[i]) {
                    lowest = heights[i];
                    pos = i;
                }
            }

            if (lowest < heights[K]) {
                heights[pos] = heights[pos] + 1;
            } else {
                lowest = Integer.MAX_VALUE;
                pos = -1;
                for (int i = K + 1; i < n; i++) {
                    if (heights[i] > heights[i - 1]) break;

                    if (lowest > heights[i]) {
                        lowest = heights[i];
                        pos = i;
                    }
                }

                if (lowest < heights[K]) {
                    heights[pos] = heights[pos] + 1;
                } else {
                    heights[K] = heights[K] + 1;
                }
            }

        }

//        for (int i = 0; i < n; i++) System.out.println(heights[i]);
        return heights;
    }
}
