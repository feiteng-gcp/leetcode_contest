class Solution {
    public int[] pourWater(int[] heights, int V, int K) {
        while(V>0){
            if(!fillLowestLeft(heights, K)){
              if(!fillLowerstRight(heights, K)){
                heights[K] = heights[K] + 1;
              }
            }
            V--;
        }
        return heights;
    }
    
    public boolean fillLowestLeft(int[] heights, int K){
      int firstK = K;
      int finalK = K;
      int firstH = heights[K];
      int finalH = heights[K];
      while(K-1>=0 && heights[K-1]<=heights[K]){
          K--;
          if(heights[K]<heights[K+1]){
            finalH = heights[K];
            finalK = K;
          }
      }
      if(finalH == firstH){
          return false;
      }else{
          heights[finalK] = heights[finalK] + 1;
          return true;
      }
    }
    
    public boolean fillLowerstRight(int[] heights, int K){
      int firstK = K;
      int finalK = K;
      int firstH = heights[K];
      int finalH = heights[K];
      int len = heights.length;
      while(K+1<len && heights[K+1]<=heights[K]){
          K++;
          if(heights[K]<heights[K-1]){
            finalH = heights[K];
            finalK = K;
          }
      }
      if(finalH == firstH){
          return false;
      }else{
          heights[finalK] = heights[finalK] + 1;
          return true;
      }
    }
}
