class Solution {
    public int[] pourWater(int[] heights, int V, int K) {
        for (int i = 0; i < V; i++) {
            int pos = K;
            
            /*while (pos - 1 >= 0 && heights[pos - 1] < heights[pos]) {
                pos--;
            }
            
            if (pos == K) {
                while (pos + 1 < heights.length && heights[pos + 1] < heights[pos]) {
                    pos++;
                }
            }*/
            
            for (int j = pos - 1; j >= 0; j--) {
                if (heights[j] > heights[j + 1]) {
                    break;
                }
                
                if (heights[j] < heights[pos]) {
                    pos = j;
                }
            }
            
            if (pos == K) {
                for (int j = pos + 1; j < heights.length; j++) {
                    if (heights[j] > heights[j - 1]) {
                        break;
                    }

                    if (heights[j] < heights[pos]) {
                        pos = j;
                    }
                }
            }
            
            heights[pos]++;
        }
        return heights;
    }
}
