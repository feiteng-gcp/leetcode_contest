class Solution {
    public int[] pourWater(int[] heights, int V, int K) {
        for (int i = 1; i <= V; i++) {
            int fill = K;
            for (int j = K-1; j >= 0; j--) {
                if (heights[j] < heights[j+1]) {
                    fill = j;
                } else if (heights[j] > heights[j+1]) break
;
            }
            if (fill == K) {
                for (int j = K+1; j < heights.length; j++) {
                    if (heights[j] < heights[j-1]) {
                        fill = j;
                    } else if (heights[j] > heights[j-1]) break
;
                }
            }
            heights[fill]++;
        }
        return heights;
    }
}
