class Solution {
    public int[] pourWater(int[] heights, int V, int K) {
        while(V-- > 0) {
            int id = index(heights, K);
            heights[id]++;
        }
        return heights;
    }
    private int index(int[] heights, int K) {
        int i = K, j = K;
        for(; i >= 0; --i) {
            if(i == 0 || heights[i] > heights[K] || heights[i] < heights[i - 1]) {
                break;
            }
        }
        while(i + 1 < K && heights[i] == heights[i + 1])    ++i;
        for(; j < heights.length; ++j) {
            if(j == heights.length - 1 || heights[j] > heights[K] || heights[j] < heights[j + 1]) {
                break;
            }
        }
        while(j - 1 > K && heights[j] == heights[j - 1])    --j;
        if(heights[i] < heights[K]) {
            return i;
        } else if(heights[j] < heights[K]) {
            return j;
        } else {
            return K;
        }
    }
}
