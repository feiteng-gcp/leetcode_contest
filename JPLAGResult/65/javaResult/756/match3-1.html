class Solution {
    public int[] pourWater(int[] heights, int V, int K) {
        for(int i = 0; i < V; i++){
            int LeftStop = K;
            int RightStop = K;
            // find left min point
            while(LeftStop > 0 && heights[LeftStop] >= heights[LeftStop-1]){
                LeftStop--;
            }
            //
            while(LeftStop < K && heights[LeftStop] == heights[LeftStop+1]){
                LeftStop++;
            }
            
            if(heights[LeftStop] < heights[K]){
                heights[LeftStop]++;
                continue;
            }
            
            
            while(RightStop < heights.length-1 && heights[RightStop] >= heights[RightStop+1]){
                RightStop++;
            }
            while(RightStop > K && RightStop < heights.length && heights[RightStop] == heights[RightStop-1]){
                RightStop--;
            }
            if(heights[RightStop] < heights[K]){
                heights[RightStop]++;
                continue;
            }
            heights[K]++;
        }
        return heights;
    }
}
