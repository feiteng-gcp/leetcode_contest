class Solution {
    public int[] pourWater(int[] heights, int V, int K) {
        for (; V > 0; V--) {
            int minLeft = K;
            for (int i = K - 1; i >= 0; i--) {
                if (heights[i] < heights[minLeft])
                    minLeft = i;
                if (heights[i] > heights[minLeft])
                    break;
            }
            if (minLeft < K) {
                heights[minLeft]++;
            } else {
                int minRight = K;
                for (int i = K + 1; i < heights.length; i++) {
                    if (heights[i] < heights[minRight])
                        minRight = i;
                    if (heights[i] > heights[minRight])
                        break;
                }
                heights[minRight]++;
            }
        }
        return heights;
    }
}
