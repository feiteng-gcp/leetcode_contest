class Solution {
public:
    int reachNumber(int dest) {
        if(dest==0)
        {
        return 0;
        }
        if(dest<0)
        dest*=-1;

        int sum=0;
        for(int i=1 ; i < INT_MAX ; i++)
        {
            sum= i*(i+1)/2;
            if(sum%2==0 && dest%2==0 && dest<=sum)
            {
                return i;
            }
            else if(sum%2==1 && dest%2==1 && dest<=sum)
            {
                return i;
            }
        }
    }
};
