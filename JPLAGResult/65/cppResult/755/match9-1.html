class Solution {
public:
    int reachNumber(int target) {
        if(target<0)

            target=-target;
        long long sum = 0;
        for(int i=1;;++i)
        {
            sum+=i;
            if(target==sum)
                return i;
            if((sum>target) && (sum - target)%2 == 0 )
                return i;
        }
        return -1;
    }
};
