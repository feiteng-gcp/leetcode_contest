class Solution {
public:
    int reachNumber(int target) {
        target = labs(target);

        int tot = 0, cur = 0;
        while (tot < target) {
            cur += 1;
            tot += cur;
        }
        if (tot == target)
            return cur;
        if ((tot - target) % 2 == 1)
            return (cur + 1) / 2 * 2 + 1;
        return cur;
    }
};
