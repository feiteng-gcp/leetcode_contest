class Solution {
public:
    int reachNumber(int target) {
        if (target < 0) target = -1 * target;
        int n = 0, sum = 0;
        while (sum < target) {
            n++;
            sum += n;
        }
        if (sum == target) return n;
        else if ((sum - target) % 2 == 0) return n;
        else if (n % 2) return n + 2;
        else return n + 1;
    }
};
