class Solution {
public:
    int reachNumber(int target) {
        int level = 1;
        int lastEle = 0;
        
        while(1){
            lastEle += level;
            if(lastEle%2 == 0 && abs(target)%2 == 0 
               || lastEle%2 == 1 && abs(target)%2 == 1){
               if(abs(target) <= lastEle){
                   break;
               } 
            }
            
            level++;
        }
        
        return level;
    }
};
