class Solution {
public:
    bool ok(int n, int t) {
        long long cur = 1ll * (1 + n) * n / 2;
        long long tmp = cur;
        if (cur < t) return false;
        cur -= t;
        if (cur % 2) return false;
        cur /= 2;
        return cur <= tmp;
    }
    int reachNumber(int target) {
        if (target == 0) return 0;
        if (target < 0) target = -target;
        for (int i = 1; ; i++) {
            if (ok(i, target)) return i;
        }
    }
};
