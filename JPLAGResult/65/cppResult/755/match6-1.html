class Solution {
public:
    int reachNumber(int target) {
        if(target<0)
            target=-target;
        long long k=sqrt(target*2);
        k=k-2;
        if(k<0)

            k=0;
        while(1)
        {
            long long temp=k*(k+1)/2;
            if(temp>=target && (temp-target)%2==0)
                return k;
            k++;
        }
    }
};
