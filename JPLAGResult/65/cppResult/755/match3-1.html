class Solution {
public:
    int reachNumber(int target) {
        if (target < 0) 
            target = -target;
        int step = 0;
        int result = 0;
        while(true) {
            result = result + step;
            int dif = target - result;
            if (dif == 0)
                return step;
            else if (dif < 0 && dif % 2 == 0) {
                return step; 
            }
            step++;
        }
        return 0;
    }
};
