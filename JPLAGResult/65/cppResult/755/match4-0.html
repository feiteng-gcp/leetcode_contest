typedef long long int64;
class Solution {
public:
    int reachNumber(int target) {
        if (target < 0) target = -target;
        int n = 1;
        int64 sum = 1;
        while (sum < target || (target + sum) % 2) {
            ++n;
            sum += n;
        }
        return n;
    }
};
