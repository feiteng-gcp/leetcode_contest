class Solution {
public:
    
    
int getMinSteps(int n){
    int step = 1;
    int steps_sum = 0;
    while(steps_sum < n){
        steps_sum += step;
        step++;
    }
    int gap = steps_sum - n;
    if(gap == 0 || gap%2==0){
        return (step - 1);
    }
    else{
        
        int parity = n%2;
        while(parity != (steps_sum%2)){
            steps_sum+= step;
            step++;
        }
        return (step - 1);
    }

}

    
    int reachNumber(int target) {
        target = abs(target);
       int ans = getMinSteps(target);
       return ans;
    }
};
