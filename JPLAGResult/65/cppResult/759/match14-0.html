struct node
{
    int l, r;
}a[3005];
bool cmp(node u, node v) {return u.r == v.r? u.l > v.l : u.r < v.r;}
class Solution {
public:
    int intersectionSizeTwo(vector<vector<int>>& intervals) {
        int n = intervals.size(), x, y, ret;
        for (int i = 0; i < n; i++) a[i + 1].l = intervals[i][0], a[i + 1].r = intervals[i][1];
        sort(a + 1, a + n + 1, cmp);
        ret = 2;
        x = a[1].r - 1, y = a[1].r;
        for (int i = 2; i <= n; i++)
            if (a[i].l <= x) continue;
            else if (a[i].l <= y)
            {
                x = y;
                y = a[i].r;
                ret++;
            }
            else
            {
                x = a[i].r - 1;
                y = a[i].r;
                ret += 2;
            }
        return ret;
    }
};
