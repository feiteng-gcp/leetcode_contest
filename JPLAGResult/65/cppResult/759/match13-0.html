class Solution {
public:
	vector<pair<int, int> > vv;
	vector<int> star[6002];
	vector<int> en[6002];
	map<int, int> mp;
	set<int> one;
	set<int> two;

	int intersectionSizeTwo(vector<vector<int>>& v) {
		for (int i = 0; i<v.size(); i++) {
			mp[v[i][0]] = 0;
			mp[v[i][1]] = 0;
		}
		int sz = 0;
		for (auto it = mp.begin(); it != mp.end(); it++) {
			(*it).second = sz;
			sz++;
		}
		for (int i = 0; i<v.size(); i++) {
			v[i][0] = mp[v[i][0]];
			v[i][1] = mp[v[i][1]];
		}
		for (int i = 0; i<6002; i++) {
			star[i].clear();
			en[i].clear();
		}
		for (int i = 0; i<v.size(); i++) {
			vv.push_back(make_pair(v[i][0], v[i][1]));
		}
		sort(vv.begin(), vv.end());
		for (int i = 0; i<vv.size(); i++) {
			star[vv[i].first].push_back(i);
			en[vv[i].second].push_back(i);
		}
		one.clear();
		two.clear();
		int ans = 0;
		for (int i = 0; i<mp.size(); i++) {
			for (int j : star[i]) {
				two.insert(j);
			}
			int mx = 0;
			for (int j : en[i]) {
				if (two.count(j)) {
					mx = 2;
					two.erase(j);
				}
				if (one.count(j)) {
					mx = max(mx, 1);
					one.erase(j);
				}
			}
			if (mx == 2) {
				two.clear();
				one.clear();
				ans += 2;
				for (int j : star[i]) {
					one.insert(j);
				}
				continue;
			}
			if (mx == 1) {
				one = two;
				two.clear();
				ans++;
				continue;
			}
		}
		return ans;
	}
};
