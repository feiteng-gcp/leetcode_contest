class Solution {
public:
    int intersectionSizeTwo(vector<vector<int>>& intervals) {
        sort(intervals.begin(),intervals.end(),[](const vector<int>& v1, const vector<int>& v2){return v1[0]<v2[0] || v1[0]==v2[0] && v1[1]>v2[1];});
        vector<vector<int>> vec;
        int count=0,n=0;
        for(int i=0;i<intervals.size();i++)
        {
            while(vec.size() && vec[n-1][1]>intervals[i][1])
            {
                vec.pop_back();
                n--;
            }
            vec.push_back(intervals[i]);
            n++;
        }
        int l1=-2,l2=-1;
        for(int i=0;i<n;i++)
        {
            if(l2<vec[i][0])
            {
                l1=vec[i][1]-1;
                l2=vec[i][1];
                count+=2;
                continue;
            }
            if(l1<vec[i][0])
            {
                l1=vec[i][1];
                if(l1==l2)
                    l1--;
                if(l1>l2)
                    swap(l1,l2);
                count++;
                continue;
            }
        }
        return count;
    }
};
