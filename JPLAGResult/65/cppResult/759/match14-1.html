class Solution {
public:
    int intersectionSizeTwo(vector<vector<int>>& intervals) {
        int n = intervals.size();
        for (int i = 0;i < n;++i)
            for (int j = i + 1;j < n;++j)
                if ((intervals[i][1] > intervals[j][1]) || ((intervals[i][1] == intervals[j][1]) && (intervals[i][0] > intervals[j][0]))) {
                    swap(intervals[i][0], intervals[j][0]);
                    swap(intervals[i][1], intervals[j][1]);
                }
        int cnt[3000];
        memset(cnt, 0, sizeof(cnt));
        int ans = 0;
        set<int> hash;
        hash.clear();
        for (int i = 0;i < n;++i) {
            for (int j = 0;j < 2 - cnt[i];++j) {
                int num;
                for (int k = intervals[i][1];;--k) {
                    if (hash.find(k) == hash.end()) {
                        num = k;
                        break;
                    }
                }
                ++ ans;
                hash.insert(num);
                for (int k = i + 1;k < n;++k)
                    if (intervals[k][0] <= num && intervals[k][1] >= num)
                        ++ cnt[k];
            }
        }
        
        return ans;
    }
};
