int pow(int a,int b){
    int ans=1;
    while(b){
        if(b&1)ans=ans*(ll)a%mod;
        a=(ll)a*a%mod;b>>=1;
    }
    return ans;
}

pii a[5000];
class Solution {
public:
    int intersectionSizeTwo(vector<vector<int>>& b) {
        int n=SZ(b);
        rep(i,0,n)a[i]=mp(b[i][0],b[i][1]);
        sort(a,a+n);
        int t=0;
        rep(i,0,n){
            while(t&&a[t-1].y>=a[i].y)t--;
            a[t++]=a[i];
        }
        n=t;
        set<int>s;
        rep(i,0,n){
            auto it=s.lower_bound(a[i].x);
            if(it==s.end()){
                s.insert(a[i].y-1),s.insert(a[i].y);
            }
            else{
                it++;
                if(it==s.end())s.insert(a[i].y);
            }
        }
        return SZ(s);
    }
};
