class Solution {
public:
    int intersectionSizeTwo(vector<vector<int>>& intervals) {
        map<int, int> max_left;
        set<int> S;
        for (vector<int> v : intervals) {
            int l = v[0], r = v[1];
            S.insert(r);
            max_left[r] = max(max_left[r], l);
        }
        int last1 = -1, last2 = -2;
        int ret = 0;
        for (int r : S) {
            int l = max_left[r];

            if (last1 < l) {
                last1 = r;
                last2 = r - 1;
                ret += 2;
            } else if (last2 < l) {
                last2 = last1;
                last1 = r;
                ret += 1;
            }
        }
        return ret;
    }
};
