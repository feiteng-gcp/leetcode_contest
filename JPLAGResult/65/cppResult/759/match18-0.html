class Solution {
public:
    struct fff
{
	int l;
	int r;
	int cnt;
};
fff f[3005];
set<int> s;
    int intersectionSizeTwo(vector<vector<int>>& intervals) {
        int n=intervals.size();
	for(int i=0;i<n;i++)
	{
		f[i].l=intervals[i][0];
		f[i].r=intervals[i][1];
		f[i].cnt=0;
	}
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			if(f[j].r<f[i].r||(f[j].r==f[i].r&&f[j].l<f[i].l))
			{
				fff tmp=f[i];
				f[i]=f[j];
				f[j]=tmp;
			}
		}
	}
	int ans=0;
	int ok=1;
	while(1)
	{
		for(int i=0;i<n;i++)
		{
			if(f[i].cnt<2)
			{
				ok=0;
				ans++;
				int next;
				if(s.find(f[i].r)==s.end())next=f[i].r;
				else next=f[i].r-1;
				//printf("%d\n",next);
				s.insert(next);
				for(int j=0;j<n;j++)
				{
					if(f[j].l<=next&&f[j].r>=next)f[j].cnt++;
				}
				break;
			}
		}
		if(ok)break;
		ok=1;
	}
	return ans;
    }
};
