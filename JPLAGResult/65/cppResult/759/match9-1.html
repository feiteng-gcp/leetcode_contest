const int SIZE = 1e6+10;
class Solution {
public:
    int intersectionSizeTwo(vector<vector<int>>& a) {
        int n=SZ(a);
        VPII pp;
        REP(i,n){
            pp.PB(MP(a[i][1],a[i][0]));
        }
        sort(ALL(pp));
        set<int>d;
        REP(i,n){
            int cnt=0;
            for(int x:d){
                if(x>=pp[i].S&&x<=pp[i].F)cnt++;
                if(cnt>=2)break;

            }
            int now=pp[i].F;
            while(cnt<2){
                if(!d.count(now)){
                    cnt++;
                    d.insert(now);
                }
                now--;
            }
        }
        return SZ(d);
    }
};
