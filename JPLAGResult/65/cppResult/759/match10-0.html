class Solution {
public:
    int intersectionSizeTwo(vector<vector<int>>& intervals) {
        int n = intervals.size();
        int ans = 0;
            
        auto cmp = [](const vector<int> &a, const vector<int> &b) {
            if (a[1] != b[1]) return a[1] < b[1];
            return a[0] > b[0];
        };
        
        sort(intervals.begin(), intervals.end(), cmp);
        
        vector<int> inserted;        
        
        for (int i = 0; i < n; ++i) {
            int count = 0;
            for (auto x: inserted)
                if (x >= intervals[i][0] && x <= intervals[i][1]) count++;
            
            //printf("%d,%d %d\n", intervals[i][0] ,intervals[i][1], count);
            //for (auto x: inserted) cout << x << " "; cout << endl;
            
            if (count >= 2) continue;
            if (count == 1)
                inserted.push_back(intervals[i][1]);
            else {
                inserted.push_back(intervals[i][1]-1);
                inserted.push_back(intervals[i][1]);
            };           
                
                
        };
        return inserted.size();
        
        
            
        
    }
};
