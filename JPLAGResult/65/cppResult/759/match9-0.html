class Solution {
public:
    int intersectionSizeTwo(vector<vector<int>>& a) {
        sort(a.begin(), a.end(), [](const vector<int>& u, const vector<int>& v) {
            return u[1] < v[1];
        });
        set<int> ret;
        int n = a.size();
        for (int i = 0; i < n; ++i) {
            int cnt = 0;
            for (auto& it : ret) {
                if (it >= a[i][0] && it <= a[i][1]) ++cnt;
            }
            int last = a[i][1];
            while (cnt < 2) {
                if (ret.find(last) == ret.end()) {
                    ret.insert(last);
                    ++cnt;
                }
                --last;
            }
        }
        return ret.size();
    }
};
