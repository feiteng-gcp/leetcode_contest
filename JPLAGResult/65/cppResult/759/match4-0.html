int pow(int a,int b){
    int ans=1;
    while(b){
        if(b&1)ans=ans*(ll)a%mod;
        a=(ll)a*a%mod;b>>=1;
    }
    return ans;
}

pii a[N];
int q[N],vis[N];

class Solution {
public:

int intersectionSizeTwo(vector<vector<int>>& x) {
    set<int> se;
    map<int, int> ma;
    for (vector<int> & t : x) {
        se.insert(t[0]);
        se.insert(t[1]);
    }
    int cnt = 1;
    for (int t : se)
        ma[t] = cnt++;
    memset(q, 0, sizeof q);
    memset(vis, 0, sizeof vis);
    int j,i,k,T,ca=0,K=0,n,m=0;
    //scanf("%d",&n);
    n = x.size();
    //cout << n << " " << cnt <<  endl;
    rep(i,0,n){
        //scanf("%d%d",&a[i].x,&a[i].y);
        a[i].x = ma[x[i][0] ];
        a[i].y = ma[x[i][1] ];
    }
    sort(a,a+n);
    int t=0;
    rep(i,0,n){
        while(t&&a[q[t-1]].y>=a[i].y){
            vis[q[t-1]]=1;
            t--;
        }
        q[t++]=i;
    }
    int sz=n;n=0;
    rep(i,0,sz)if(!vis[i])a[n++]=a[i];
    set<int>s;
    rep(i,0,n){
        int m=0;
        auto it=s.lower_bound(a[i].x);
        if(it==s.end()){
            s.insert(a[i].y),m++;
            s.insert(a[i].y-1);m++;
        }
        else{
            m++;it++;
            if(it!=s.end())m++;
            else s.insert(a[i].y),m++;
        }
    }
    return SZ(s);
    //printf("%d\n",SZ(s));
}
};
