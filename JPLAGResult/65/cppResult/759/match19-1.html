class Solution {
public:
    int lowbit(int x) {
        return x & (-x);
    }
    void add(int x, int n, vector<int> &C) {
        for (int i = x; i <= n; i += lowbit(i)) {
            C[i] += 1;
        }

    }
    int sum(int x, vector<int> &C) {
        int res = 0;
        for (int i = x; i > 0; i -= lowbit(i)) {
            res += C[i];
        }
        return res;
    }
    int intersectionSizeTwo(vector<vector<int>>& intervals) {
        int n = intervals.size();
        vector<pair<pair<int, int>, int> > a;
        vector<int> poses;
        for (int i = 0; i < n; i++) {
            poses.push_back(intervals[i][0] - 1);
            poses.push_back(intervals[i][1] - 1);
            poses.push_back(intervals[i][0]);
            poses.push_back(intervals[i][1]);
        }
        sort(poses.begin(), poses.end());
        int nn = unique(poses.begin(), poses.end()) - poses.begin();
        map<int, int> mp;
        for (int i = 0; i < nn; i++) {
            mp[poses[i]] = i + 1;
        }
        for (int i = 0; i < n; i++) {
            a.push_back(make_pair(make_pair(mp[intervals[i][1]], -mp[intervals[i][0]]), -i));
        }
        sort(a.begin(), a.end());
        vector<int> C(poses.size() + 1);
        int cnt = 0;
        for (auto item : a) {
            int now = item.second;
            int st = mp[intervals[-now][0]], en = mp[intervals[-now][1]];
            cout << "st, en = " << st << ", " << en << endl;
            int res = sum(en, C) - sum(st - 1, C);
            if (res == 0) {
                add(en, C.size() - 1, C);
                add(en - 1, C.size() - 1, C);
                cnt += 2;
            } else if (res == 1) {
                add(en, C.size() - 1, C);
                cnt++;
            }
        }
        return cnt;
    }
};
