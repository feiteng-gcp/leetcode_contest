class Solution {
public:
    int intersectionSizeTwo(vector<vector<int>>& intervals) {
        int ans = 0;
        auto cmp = [](vector<int> &a, vector<int> &b) {
            if (a[1] == b[1]) {
                return a[0] > b[0];
            }
            return a.back() < b.back();
        };
        sort(intervals.begin(), intervals.end(), cmp);
        int a = -1, b = -1;

        for (auto &v: intervals) {
            //cout << v[0] << " " << v[1] << " :";
            if (b < v[0]) {
                ans += 2;
                //cout << 2;
                b = v.back();
                a = b - 1;
            } else if (a < v[0]) {
                ans += 1;
                //cout << 1;
                a = b;
                b = v.back();
            }
            //cout << " > " << a << " " << b;
            //cout << endl;
        }
        return ans;
    }
};
