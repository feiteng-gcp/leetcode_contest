int a[3010];
int p[3010][2];
int cmp(int i, int j) {
    return p[i][1] < p[j][1] || p[i][1] == p[j][1] && p[i][0] < p[j][0];
}
class Solution {
public:

    int intersectionSizeTwo(vector<vector<int>>& intervals) {
        int n = intervals.size();
        for (int i = 0; i < n; i++) {
            p[i][0] = intervals[i][0];
            p[i][1] = intervals[i][1];
        }
        vector<int> ans;
        for (int i = 0; i < n; i++) {
            a[i] = i;
        }
        sort(a, a + n, cmp);
        map<int, int> mp;
        for (int i = 0; i < n; i++) {
            int e = a[i];
            int cc = 0;
            for (int j = 0; j < ans.size(); j++) {
                if (ans[j] >= p[e][0] && ans[j] <= p[e][1]) {
                    cc++;
                }
            }
            if (cc == 0) {
                int pp = 0;
                for (int j = p[e][1]; j >= p[e][0] && pp < 2; j--) {
                    if (mp[j] == 0) {
                        pp++;
                        mp[j] = 1;
                        ans.push_back(j);
                    }
                }
            } else if (cc == 1) {
               int pp = 0;
                for (int j = p[e][1]; j >= p[e][0] && pp < 1; j--) {
                    if (mp[j] == 0) {
                        pp++;
                        mp[j] = 1;
                        ans.push_back(j);
                    }
                }
            }
        }
        for (int i = 0; i < ans.size(); i++) printf("%d\n", ans[i]);
        return ans.size();
    }
};
