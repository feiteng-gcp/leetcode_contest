class Solution {
    bool static Cmp(const vector<int>& lhs, const vector<int>& rhs) {
        return lhs[0] < rhs[0] || lhs[0] == rhs[0] && lhs[1] < rhs[1];
    }
public:
    int intersectionSizeTwo(vector<vector<int>>& intervals) {
        std::sort(intervals.begin(), intervals.end(), Cmp);
		int count{2}, fast{intervals.front()[1]};
		int slow{intervals.front()[1] - 1};
		for (int i = 1; i < intervals.size(); ++ i) {
			if (intervals[i][1] <= fast) {
				fast = intervals[i][1];
				if (intervals[i][1] - 1 <= slow) {
					slow = intervals[i][1] - 1;
				} else if (intervals[i][0] > slow) {
					++ count;
					slow = intervals[i][1] - 1;
				}
			} else {
				if (intervals[i][0] > fast) {
					count += 2;
					fast = intervals[i][1];
					slow = fast - 1;
				} else if (intervals[i][0] > slow) {
					count += 1;
					slow = fast;
					fast = intervals[i][1];
				}
			}
		}
		return count;
    }
};
