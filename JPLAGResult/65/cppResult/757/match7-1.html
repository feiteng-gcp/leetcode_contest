class Solution {
    unordered_map<string, string> ald; 
    unordered_set<string> visited;
    

    bool tryout(string & str, int loc, string & next) {
        if (loc == str.length() - 1)
            return search(next);
        string & cand = ald[str.substr(loc, 2)];
        for (int i = 0; i < cand.length(); ++i) {
            next[loc] = cand[i];
            if (tryout(str, loc + 1, next))
                return true;
        }
        return false;
    }
    
    bool search(string & str) {
        if (str.length() == 1)
            return true;
        if (visited.find(str) != visited.end())
            return false;
        visited.insert(str);
        for (int i = 0; i < str.length() - 1; ++i) {
            if (ald.find(str.substr(i, 2)) == ald.end())
                return false;
        }
        string next = string(str.length() - 1, ' ');
        return tryout(str, 0, next);
    }
    
public:
    bool pyramidTransition(string bottom, vector<string>& allowed) {
        visited = {};
        ald = {};
        for (auto & x : allowed) {
            ald[x.substr(0, 2)] += x.substr(2);
        }
        return search(bottom);
    }
};
