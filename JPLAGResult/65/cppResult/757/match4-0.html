const int maxn = 110;

bool dp[maxn][maxn][7];

class Solution {

public:
    bool pyramidTransition(string bottom, vector<string> &allowed) {
        int n = bottom.size();
        for (int i = n - 1; i >= 0; i--) {
            for (int j = 0; j <= i; j++) {
                for (int k = 0; k < 7; k++) {
                    if (i == n - 1) {
                        dp[i][j][k] = bottom[j] == 'A' + k;
                    } else {
                        dp[i][j][k] = false;
                        for (auto &str:allowed) {
                            if (str[2] == 'A' + k) {
                                if (dp[i + 1][j][str[0] - 'A'] && dp[i + 1][j + 1][str[1] - 'A']) {
                                    dp[i][j][k] = true;
                                    break;
                                }
                            }
                        }
                    }
                }
            }
        }
        for (int k = 0; k < 7; k++) {
            if (dp[0][0][k]) {
                return true;
            }
        }
        return false;
    }
};
