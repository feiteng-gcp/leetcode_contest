bool g[7][7][7];
class Solution {
public:
    bool pyramidTransition(string s, vector<string>& allowed) {
        memset(g, 0, sizeof(g));
        for (auto& it : allowed) {
            int x = it[0] - 'A';
            int y = it[1] - 'A';
            int z = it[2] - 'A';
            g[x][y][z] = true;
        }
        int n = s.length();
        vector<int> a(n), b(n);
        for (int i = 0; i < n; ++i) a[i] = 1 << (s[i] - 'A');
        for (int i = n - 1; i >= 0; --i) {
            for (int j = 0; j <= i; ++j) {
                cout << a[j] << " ";
            }
            cout << endl;
            for (int j = 0; j <= i; ++j) {
                if (a[j] == 0) return false;
            }
            for (int j = 0; j < i; ++j) {
                b[j] = 0;

                for (int u = 0; u < 7; ++u) {
                    if (!(a[j] & (1 << u))) continue;
                    for (int v = 0; v < 7; ++v) {
                        if (!(a[j + 1] & (1 << v))) continue;
                        for (int w = 0; w < 7; ++w) {
                            if (g[u][v][w]) b[j] |= 1 << w;
                        }
                    }
                }
            }
            swap(a, b);
        }
        return true;
    }
};
