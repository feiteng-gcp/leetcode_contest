class Solution {
public:
bool pyramidTransition(string s, vector<string>& a) {
    vector<string> v;
    for (char x : s) {
        v.push_back(string(1, x));
    }
    map<string, string> ma;
    for (string x : a) {
        ma[x.substr(0, 2) ] += string(1, x[2]);
        //cout << ma[x.substr(0, 2)] << endl;
    }
    int n = s.size() - 1;
    vector<string> t;

    while(n--) {
        t.clear();
        //cout << "asd " << n << endl;
        for (int i = 0; i < v.size() - 1; i++) {
            string cur;
            for (char x : v[i]) {
                for (char y : v[i + 1]) {
                    string d = {x, y};
                    //cout << d << " " << x << " " << y << endl;
                    if(ma.count(d)) {
                        cur += ma[d];
                    }
                }
            }
            //cout << i << " " << cur  << endl;
            if(cur.size() == 0) return false;
            set<char> se;
            for (char ch : cur)
                se.insert(ch);
            cur.clear();
            for (char ch : se)
                cur += ch;
            t.push_back(cur);
        }
        swap(t, v);
    }
    return 1;
}
};
