class Solution {
public:
    bool pyramidTransition(string bottom, vector<string>& allowed) {
        bool f[101][101][7];
        memset(f, false, sizeof(f));
        bool can[7][7][7];
        memset(can, false, sizeof(can));
        for (int i = 0;i < allowed.size();++i)
            can[allowed[i][0] - 'A'][allowed[i][1] - 'A'][allowed[i][2] - 'A'] = true;
        for (int i = 0;i < bottom.size() - 1;++i)
            for (int j = 0;j < 7;++j)
                if (can[bottom[i] - 'A'][bottom[i + 1] - 'A'][j])
                    f[i][i + 1][j] = true;
        for (int i = bottom.size() - 1;i >= 0;--i)
            for (int j = i + 2;j < bottom.size();++j) {
                int mid = (i + j) / 2;
                for (int k = 0;k < 7;++k) 
                    for (int k1 = 0; k1 < 7;++k1)
                        for (int k2 = 0;k2 < 7;++k2)
                            if (f[i][j - 1][k1] && f[i + 1][j][k2] && can[k1][k2][k]) {
                                f[i][j][k] = true;
                            }
            }
        for (int k = 0;k < 7;++k)
            if (f[0][bottom.size() - 1][k]) return true;
        return false;
    }
};
