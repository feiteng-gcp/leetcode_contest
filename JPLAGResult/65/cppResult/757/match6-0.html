class Solution {
public:
    bool pyramidTransition(string bottom, vector<string>& allowed) {
        vector<pair<int,int>> a[7];
        for (string s : allowed) {
            a[s[2] - 'A'].push_back(pair<int,int>(s[0] - 'A', s[1] - 'A'));
        }
        
        bool possible[105][105][7];
        
        int n = bottom.length();
        for (int i = 0; i < bottom.length(); i++) {
            for (int j = 0; j < 7; j++) {
                possible[n][i][j] = false;
            }
            possible[n][i][bottom[i] - 'A'] = true;
        }
        
        while (n > 1) {
            n--;
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < 7; j++) {
                    possible[n][i][j] = false;
                    for (pair<int,int> ii : a[j]) {
                        if (possible[n+1][i][ii.first] && possible[n+1][i+1][ii.second]) {
                            possible[n][i][j] = true;
                            break;
                        }
                    }
                }
            }
        }
        
        bool ans = false;
        for (int j = 0; j < 7; j++) {
            ans |= possible[1][0][j];
        }
        
        return ans;
    }
};
