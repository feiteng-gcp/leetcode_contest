class Solution {
public:
    vector<int> ok[26][26];
bool dp[101][101][26];
    bool pyramidTransition(string bottom, vector<string>& allowed) {
	int len=allowed.size();
	for(int i=0;i<len;i++)
	{
		ok[allowed[i][0]-'A'][allowed[i][1]-'A'].push_back(allowed[i][2]-'A');
	}
	memset(dp,false,sizeof(dp));
	int n=bottom.size();
	for(int i=0;i<n;i++)
	{
		dp[n][i+1][bottom[i]-'A']=true;
	}
	for(int i=n-1;i>0;i--)
	{
		for(int j=1;j<=i;j++)
		{
			for(int k=0;k<26;k++)
			{
				if(!dp[i+1][j][k])continue;
				for(int l=0;l<26;l++)
				{
					if(!dp[i+1][j+1][l])continue;
					int ll=ok[k][l].size();
					for(int t=0;t<ll;t++)
					{
						dp[i][j][ok[k][l][t]]=true;
					}
				}
			}
		}
	}
	for(int i=0;i<26;i++)
	{
		if(dp[1][1][i])return true;
	}
	return false;
    }
};
