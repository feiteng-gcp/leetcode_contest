class Solution {
public:
    bool pyramidTransition(string bottom, vector<string>& allowed) {
        int n = bottom.size();
        vector<vector<set<char> > > dp(n, vector<set<char> >(n));
        map<string, vector<char> > mp;
        for (auto s : allowed) {
            mp[s.substr(0, 2)].push_back(s[2]);
        }
        for (int i = 0; i < n; i++) {
            dp[0][i].insert(bottom[i]);

        }
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < n - i; j++) {
                for (char c1 : dp[i - 1][j]) {
                    for (char c2 : dp[i - 1][j + 1]) {
                        string now = "";
                        now = now + c1 + c2;
                        // cout << "i, j, now = " << i << ", " << j << ", " << now << endl;
                        for (char c : mp[now]) {
                            // cout << "i, j, c = " << i << ", " << j << ", " << c << endl;
                            dp[i][j].insert(c);
                        }
                    }
                }
            }
        }
        return !dp[n - 1][0].empty();
    }
};
