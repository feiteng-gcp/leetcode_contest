class Solution {
public:
    bool dp[110][110][8];
    
    bool pyramidTransition(string bottom, vector<string>& allowed) {
        int N = bottom.size();
        memset(dp, 0, sizeof(dp));
        vector<vector<pair<int, int>>> can(7);
        for (const auto& s : allowed)
            can[s[2] - 'A'].push_back({s[0] - 'A', s[1] - 'A'});

        for (int i = 0; i < N; ++i)
            dp[0][i][bottom[i] - 'A'] = true;
        for (int i = 1; i < N; ++i)
        for (int j = 0; j < N - i; ++j)
        for (int k = 0; k < 7; ++k)
        for (int l = 0; l < can[k].size(); ++l)
        {
            if (dp[i - 1][j][can[k][l].first] && dp[i - 1][j + 1][can[k][l].second])
            {
                dp[i][j][k] = true;
                break;
            }
        }
        for (int i = 0; i < 7; ++i)
            if (dp[N - 1][0][i])
                return true;
        return false;
    }
};
