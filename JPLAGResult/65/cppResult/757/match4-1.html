class Solution {
public:
    void work(int x, vector<string>& tmp, int n, string p, int m, string bottom, vector<string>& allowed) {
        if (x == n - 1) {
            tmp.push_back(p);
            return;
        }
        for (int i = 0; i < m; i++) {
            if (allowed[i][0] == bottom[x] && allowed[i][1] == bottom[x + 1]) {
                work(x + 1, tmp, n, p + allowed[i][2], m, bottom, allowed);
            }
        }
    }
    bool pyramidTransition(string bottom, vector<string>& allowed) {
        if (bottom.length() == 1) return true;
        int n = bottom.length(), m = allowed.size();
        string tmp;
        vector<string> w;
        work(0, w, n, "", m, bottom, allowed);

        for (int i = 0; i < w.size(); i++) {
            if (pyramidTransition(w[i], allowed)) {
                return true;
            }
        }
        return false;
    }
};
