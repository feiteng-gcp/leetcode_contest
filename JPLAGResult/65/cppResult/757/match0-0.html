class Solution {
public:
    bool pyramidTransition(string bottom, vector<string>& allowed) {
        bool f[105][105][7];
        bool chk[7][7][7];
        int n = bottom.size(), m = allowed.size();
        memset(f, false, sizeof f);
        memset(chk, false, sizeof chk);
        for (int i = 0; i < m; i++) chk[allowed[i][0] - 'A'][allowed[i][1] - 'A'][allowed[i][2] - 'A'] = true;
        for (int i = 1; i <= n; i++) f[1][i][bottom[i - 1] - 'A'] = true;
        for (int i = 2; i <= n; i++)
            for (int j = 1; j + i <= n + 1; j++)
                for (int a = 0; a < 7; a++)
                    if (f[i - 1][j][a])
                        for (int b = 0; b < 7; b++)
                            if (f[i - 1][j + 1][b])
                                for (int c = 0; c < 7; c++)
                                    if (chk[a][b][c])
                                        f[i][j][c] = true;
        for (int i = 0; i < 7; i++)
            if (f[n][1][i])
                return true;
        return false;
    }
};
