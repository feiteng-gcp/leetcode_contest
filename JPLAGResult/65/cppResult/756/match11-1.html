class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        int n = heights.size();
        for (int i = 0; i < V; i++) {
            int loc = K;
            for (int j = K - 1; j >= 0; j--) {
                if (heights[j] > heights[j + 1]) {
                    break;
                }
                if (heights[j] < heights[j + 1])
                loc = j;
            }
            if (loc == K) {
                for (int j = K + 1; j < n; j++) {
                    if (heights[j] > heights[j - 1]) {
                        break;
                    }
                    if (heights[j] < heights[j - 1])
                    loc = j;
                }
            }
            printf("%d\n", loc);
            heights[loc]++;
        }
        return heights;
    }
    
};
