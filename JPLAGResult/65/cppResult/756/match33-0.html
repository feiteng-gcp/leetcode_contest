class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        for (int i = 0; i < V; i++) {
            auto now = make_pair(heights[K], K);
            for (int j = K - 1; j >= 0; j--) {
                if (now.first > heights[j]) {
                    now = make_pair(heights[j], j);
                } else if (now.first < heights[j]) {
                    break;
                }
            }
            if (now.second == K) {
                for (int j = K + 1; j < heights.size(); j++) {
                    if (now.first > heights[j]) {
                        now = make_pair(heights[j], j);
                    } else if (now.first < heights[j]) {
                        break;
                    }
                }
            }
            heights[now.second]++;
        }
        return heights;
    }
};
