class Solution {
public:
    vector<int> pourWater(vector<int>& a, int V, int K) {
        int n = a.size();
        while (V--) {
            a[K]++;
            int best = a[K], best_i = K, i;
            for (i = K; i > 0 && a[i - 1] + 1 <= a[i]; --i) {
                a[i]--;
                a[i - 1]++;
                if (best > a[i - 1]) {
                  best = a[i - 1];
                  best_i = i - 1;
                }
            }
            a[i]--;
            if (best_i != K) {
              a[best_i]++;
              continue;
            }
            a[K]++;
            best = a[K], best_i = K;
            for (i = K; i + 1 < n && a[i + 1] + 1 <= a[i]; ++i) {
                a[i]--;
                a[i + 1]++;
                if (best > a[i + 1]) {
                  best = a[i + 1];
                  best_i = i + 1;
                }
            }
            a[i]--;
            if (best_i != K) {
                a[best_i]++;
                continue;
            }
            a[K]++;
        }
        return a;
    }
};
