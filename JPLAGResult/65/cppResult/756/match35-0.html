class Solution {
public:
    vector<int> pourWater(vector<int>& h, int V, int K) {
        for (int i=0; i<V; ++i) {
            bool lf=true, rt=true;
            int l, r, lpos=-1, rpos=-1;
            for (l=K; l-1>=0; --l)
                if (h[l]<h[l-1]) {
                    break;
                } else if (h[l]>h[l-1]) {
                    lpos = l-1;
                }
            if (lpos==-1) lf = false;
            if (lf) {
                ++h[lpos];
                continue;
            }
            for (r=K; r+1<h.size(); ++r)
                if (h[r]<h[r+1]) {
                    break;
                } else if (h[r]>h[r+1]) {
                    rpos = r+1;
                }
            if (rpos==-1) rt = false;
            if (rt)
                ++h[rpos];
            else 
                ++h[K];
        }
        return h;
    }
};
