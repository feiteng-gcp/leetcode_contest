class Solution {
public:
    vector<int> pourWater(vector<int>&a, int V, int K) {
        while(V--)
        {
            int i=K;
            int pre=-1;
            while(i>0&&a[i]>=a[i-1])
            {
                if(a[i]>a[i-1])
                    pre=i-1;
                i--;
            }
            if(a[i]<a[K])
            {
                a[pre]++;
                continue;
            }
            i=K;
            pre=-1;
            while(i+1<a.size()&&a[i]>=a[i+1])
            {
                if(a[i]>a[i+1])
                    pre=i+1;
                i++;
            }
            if(a[i]<a[K])
            {
                a[pre]++;
                continue;
            }
            a[K]++;
        }
        return a;
    }
};
