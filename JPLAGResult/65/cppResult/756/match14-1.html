class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int k) {
        
        while(V--) {
            
            int i, mn;
            bool ok = false;
            for(i = k-1, mn = k; i >= 0 && heights[i] <= heights[k]; --i) {
                if(heights[i] < heights[mn])
                    mn = i, ok = true;
                else if(heights[i] > heights[mn]) break;
            }
            
            if(ok) {
                heights[mn]++;
                continue;
            }
            
            for(i = k+1, mn = k; i < heights.size() && heights[i] <= heights[k]; ++i) {
                if(heights[i] < heights[mn])
                    mn = i, ok = true;

                else if(heights[i] > heights[mn]) break;
            }
            
            if(ok) {
                heights[mn]++;
                continue;
            }
            
            ++heights[k];
        }
        
        return heights;
    }
};
