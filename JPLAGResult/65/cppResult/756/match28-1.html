class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        vector<int> ans = heights;
        for (int i = 0; i < V; i++) {
            drop(ans, K);
            //for (auto n: ans) {
            //    cout << n << " ";
            //}
            //cout << endl;
        }
        return ans;
    }
    void drop(vector<int> &h, int k) {
        int p = k;
        int q = k;
        while (p > 0 && h[p - 1] <= h[p]) {
            p--;
            if (h[p] < h[p+1]) {
                q = p;
            }
        }
        if (h[p] < h[k]) {
            h[q]++;
            return;
        }
        p = k;
        while (p + 1 < h.size() && h[p + 1] <= h[p]) {
            p++;
            if (h[p] < h[p-1]) {
                q = p;
            }
        }
        if (h[p] < h[k]) {
            h[q]++;
            return;
        }
        h[k]++;
    }
};
