class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        for (int i = 0; i < V; ++i)
        {
            int j, z = K;
            for (j = K - 1; j >= 0 && heights[j] <= heights[j + 1]; --j)
            {
                if (heights[j] < heights[z])
                    z = j;
            }
            if (z != K)
            {
                ++heights[z];
                continue;
            }
            z = K;
            for (j = K + 1; j < heights.size() && heights[j] <= heights[j - 1]; ++j)
            {
                if (heights[j] < heights[z])

                    z = j;
            }
            if (z != K)
            {
                ++heights[z];
                continue;
            }
            ++heights[K];
        }
        return heights;
    }
};
