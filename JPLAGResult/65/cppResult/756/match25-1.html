class Solution {
public:
vector<int> pourWater(vector<int>& a, int v, int p) {
    int n = a.size();
    while(v--) {
        int x, y;
        x = y = p;
        while(x - 1 >= 0 && a[x - 1] <= a[x]) x--;
        while(x + 1 <= p && a[x + 1] == a[x]) x++;
        while(y + 1 < n && a[y + 1] <= a[y]) y++;
        while(y - 1 >= p && a[y - 1] == a[y]) y--;
        //cout <<x << " " << p << " " << y << endl;
        if(a[x] < a[p]) {
            a[x]++;
        } else if(a[y] < a[p]) {
            a[y]++;
        } else {
            a[p]++;
        }
    }
    return a;
}
};
