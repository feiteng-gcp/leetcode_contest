class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        int n = heights.size();
        int *_H = new int[n + 2];
        int *H = _H + 1;
        H[-1] = 0x3f3f3f3f;
        H[n] = 0x3f3f3f3f;
        for (int i = 0; i < n; i++) {
            H[i] = heights[i];
        }
        while (V--) {
            int p = K;
            while (1) {
                bool L = true;
                for (int i = p - 1; ; i--) {
                    if (H[i] > H[p]) {
                        L = false;

                        break;
                    } else if (H[i] < H[p]) {
                        L = true;
                        break;
                    }
                }
                if (L) {
                    --p;
                } else {
                    bool R = true;
                    for (int i = p + 1; ; i++) {
                        if (H[i] > H[p]) {
                            R = false;

                            break;
                        } else if (H[i] < H[p]) {
                            R = true;
                            break;
                        }
                    }
                    if (R) {
                        ++p;
                    } else {
                        break;
                    }
                }
            }
            ++H[p];
        }
        vector<int> ret;
        for (int i = 0; i < n; i++) {
            ret.push_back(H[i]);
        }
        delete []_H;
        return ret;
    }
};
