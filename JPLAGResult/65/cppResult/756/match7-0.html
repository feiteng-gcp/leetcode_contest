class Solution {
public:
    vector<int> pourWater(vector<int>& a, int V, int K) {
        int n = a.size();
        while (V--) {
            int target = K;
            for (int i = K - 1; i >= 0; --i) {
                if (a[i] < a[i + 1]) {
                    target = i;
                }
                else if (a[i] > a[i + 1]) break;
            }
            if (target != K) {
                a[target] ++;
                continue;
            }
            for (int i = K + 1; i < n; ++i) {
                if (a[i] < a[i - 1]) {
                    target = i;
                }
                else if (a[i] > a[i - 1]) break;
            }
            a[target] ++;
        }
        return a;
    }
};
