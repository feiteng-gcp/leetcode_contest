class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int k) {
        for (int i = 0; i < V; ++i) {
            int j;
            int pos = k;
            for (j = k - 1; j >= 0 && heights[j] <= heights[j+1]; --j) if (heights[j] < heights[j+1]) pos = j;
            if (pos < k) {
                heights[pos]++;
                continue;
            }
            for (j = k + 1; j < heights.size() && heights[j] <= heights[j-1]; ++j) if (heights[j] < heights[j-1]) pos = j;
            heights[pos]++;            
        }
        
        return heights;
    }
};
