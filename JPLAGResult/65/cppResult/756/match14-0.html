class Solution {
public:
    vector<int> pourWater(vector<int>& h, int V, int K) {
        while(V--) {
            int m = h[K];
            int ii = K;
            for (int i = K - 1; i >= 0; i--) {
                if (h[i] < m) {
                    m = h[i];
                    ii = i;
                } else if (h[i] > m) break;
            }
            if (ii != K) {
                h[ii]++;
                continue;
            }
            
            
            for (int i = K + 1; i < h.size(); i++) {
                if (h[i] < m) {
                    m = h[i];
                    ii = i;
                } else if (h[i] > m) break;
            }
            if (ii != K) {
                h[ii]++;
                continue;
            }
            h[K]++;
        }
        return h;
    }
};
