class Solution {
public:
    vector<int> pourWater(vector<int> &heights, int V, int K) {
        while (V--) {
            int lastpos = -1;
            for (int i = K - 1; i >= 0; i--) {
                if (heights[i] < heights[i + 1]) {
                    lastpos = i;
                } else if (heights[i] > heights[i + 1]) {
                    break;
                }

            }
            if (lastpos != -1) {
                heights[lastpos]++;
                continue;
            }
            for (int i = K + 1; i < heights.size(); i++) {
                if (heights[i] < heights[i - 1]) {
                    lastpos = i;
                } else if (heights[i] > heights[i - 1]) {
                    break;
                }
            }
            if (lastpos != -1) {
                heights[lastpos]++;
                continue;
            }
            if (lastpos == -1) {
                heights[K]++;
            }
        }

        return heights;
    }

};
