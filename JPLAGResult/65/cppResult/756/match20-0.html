class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        for(int count = 0; count < V; count++)
        {
            int temp = heights[K];
            int index = K - 1;
            int ans = K;
            while(index >= 0)
            {
                if (heights[index] < temp)
                {
                    ans = index;
                    temp = heights[index];
                }
                else if (heights[index] > temp) break;
                index--;
            }
            if (ans != K) 
            {
                heights[ans]++;
                continue;
            }
            
            temp = heights[K];
            index = K + 1;
            ans = K;
            while(index < heights.size())
            {
                if (heights[index] < temp)
                {
                    ans = index;
                    temp = heights[index];
                }
                else if (heights[index] > temp) break;
                index++;
            }
            heights[ans]++;
        }
        return heights;
    }
};
