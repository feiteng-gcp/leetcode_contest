class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        int n = heights.size();
        for (int hehe = 0; hehe < V; hehe++) {
            // simulate once
            // Check left
            int i = K - 1;
            for (; i >= 0; i--) {
                if (heights[i] < heights[K]) {
                    int runner = i - 1;
                    while (runner >= 0 && heights[runner] <= heights[runner + 1]) {
                        if (heights[runner] < heights[i]) i = runner;
                        runner--;
                    }
                    break;
                } else if (heights[i] > heights[K]) {
                    // Can't pass through
                    i = -1;
                    break;
                }
            }
            if (i >= 0) {
                heights[i]++;
            } else {
                // Check right
                int j = K + 1;
                for (; j < n; j++) {
                    if (heights[j] < heights[K]) {
                        // continue to the lowest
                        int runner = j + 1;
                        while (runner < n && heights[runner] <= heights[runner - 1]) {
                            if (heights[runner] < heights[j]) j = runner;
                            runner++;
                        }
                        break;
                    } else if (heights[j] > heights[K]) {
                        j = n;
                        break;
                    }
                }
                if (j < n) {
                    heights[j]++;
                } else {
                    heights[K]++;
                }
            }
        }
        return heights;
    }
};
