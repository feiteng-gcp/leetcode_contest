class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        for (int v = 0; v < V; ++v) {
            // try left
            int left = K, cur = K;
            while (cur > 0) {
                if (heights[cur - 1] > heights[cur])
                    break;
                if (heights[cur] == heights[cur - 1]) {
                    cur -= 1;
                } else {
                    cur -= 1;
                    left = cur;
                }
                
            }
            if (left != K) {
                heights[left] += 1;
                continue;
            }
            // try right
            int right = K;
            cur = K;
            while (cur < heights.size() - 1) {
                if (heights[cur + 1] > heights[cur])
                    break;
                if (heights[cur] == heights[cur + 1]) {
                    cur += 1;
                } else {
                    cur += 1;
                    right = cur;
                }                
            }
            heights[right] += 1;
        }
        return heights;
    }
};
