class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        int n = heights.size();
        
        for (int i = 0; i < V; ++i) {             
            int h = heights[K];
            int pos = -1;
            
            for (int j = K-1; j >= 0; --j)
                if (pos == -1 && heights[j] < h || (pos != -1 && heights[j] < heights[pos])) {
                    pos = j;
                }
                else if ((pos == -1 && heights[j] > h) || (pos != -1 && heights[j] > heights[pos])) break;
            
            if (pos != -1) {
                heights[pos]++;
                continue;
            };
            for (int j = K+1; j < n; ++j)
                if (pos == -1 && heights[j] < h || (pos != -1 && heights[j] < heights[pos])) {
                    pos = j;
                }
                else if ((pos == -1 && heights[j] > h) || (pos != -1 && heights[j] > heights[pos])) break;
                
            
            if (pos != -1) {
                heights[pos]++;
                continue;
            };
                
            heights[K]++;
        };
        
        return heights;
        
    }
};
