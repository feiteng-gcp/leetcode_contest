class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        while(V--) {
            ++heights[K];
            int min = -1;
            for(int i = K - 1; i >= 0; --i) {
                if((min == -1 && heights[i] >= heights[K])
                  || (min != -1 && heights[i] > heights[min])) {
                    break;
                } else if(heights[K] - heights[i] > 1) {
                    if(min == -1 || heights[i] < heights[min]) {
                        min = i;
                    }
                }
            }
            if(min != -1) {
                --heights[K];
                ++heights[min];
                continue;
            }
            for(int i = K + 1; i < heights.size(); ++i) {
                if((min == -1 && heights[i] >= heights[K])
                  || (min != -1 && heights[i] > heights[min])) {
                    break;
                } else if(heights[K] - heights[i] > 1) {
                    if(min == -1 || heights[i] < heights[min]) {
                        min = i;
                    }
                }              
            }
            if(min != -1) {
                --heights[K];
                ++heights[min];
            }
        }
        return heights;
    }
};
