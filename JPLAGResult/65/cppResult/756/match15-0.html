class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        while(V--){
            int H=heights[K];
            int ev=-1;
            for(int i=K;i>=0;i--){
                if(H<heights[i])break;
                if(H>heights[i]){
                    ev=i;
                    H=heights[i];
                    continue;
                }
                
            }
            if(ev!=-1){
                heights[ev]++;
                continue;
            }
            H=heights[K]+1;
            ev=-1;
            for(int i=K;i<heights.size();i++){
                if(H<heights[i])break;
                if(H>heights[i]){
                    ev=i;
                    H=heights[i];
                    continue;
                }
            }
            heights[ev]++;
            continue;
        }
        return heights;
    }
};
