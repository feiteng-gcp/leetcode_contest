class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        swap(V, K);
        for (int i = 0;i < K;++i) {
            bool flag = false;
            int index = -1;
            for (int j = V - 1;j >= 0;--j) {
                if (heights[j] > heights[j + 1]) break;
                if (heights[j] < heights[j + 1]) {
                    flag = true;
                    index = j;
                }
            }
            if (flag) {
                ++ heights[index];
                continue;
            }
            for (int j = V + 1;j < heights.size();++j) {
                if (heights[j] > heights[j - 1]) break;
                if (heights[j] < heights[j - 1]) {
                    flag = true;
                    index = j;
                }
            }
            if (flag) {
                ++ heights[index];
            }
            else ++heights[V];
        }
        
        return heights;
    }
};
