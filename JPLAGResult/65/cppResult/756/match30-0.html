class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        for (int i = 0; i < V; i++) {
            bool dropped = false;
            int drop_idx = -1;
            for (int j = K-1; j >= 0; j--) {
                if (heights[j] > heights[j+1]) {
                    break;
                }
                if (heights[j] < heights[j+1]) {
                    dropped = true;
                    drop_idx = j;
                }
            }
            if (dropped) {
                heights[drop_idx]++;
                continue;
            }
            for (int j = K+1; j < heights.size(); j++) {
                if (heights[j] > heights[j-1]) {
                    break;
                }
                
                if (heights[j] < heights[j-1]) {
                    dropped = true;
                    drop_idx = j;
                }
            }
            if (dropped) {
                heights[drop_idx]++;
                continue;
            }
            
            heights[K]++;
        }
        
        return heights;
    }
};
