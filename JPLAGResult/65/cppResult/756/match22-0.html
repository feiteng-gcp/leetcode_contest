class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        int w = heights.size(), x, y;
        for (int i = 1; i <= V; i++)
        {
            x = K;
            while (true)
            {
                y = x - 1;
                while (y >= 0 && heights[y] == heights[x]) y--;
                if (y >= 0 && heights[y] < heights[x])
                {
                    x--;
                    continue;
                }
                y = x + 1;
                while (y < w && heights[y] == heights[x]) y++;
                if (y < w && heights[y] < heights[x])
                {
                    x++;
                    continue;
                }
                break;
            }
            heights[x]++;
        }
        return heights;
    }
};
