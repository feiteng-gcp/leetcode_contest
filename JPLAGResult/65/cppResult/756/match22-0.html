class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        int n=heights.size();
	for(int c=1;c<=V;c++)
	{
		int cur=K;
		for(int j=K-1;j>=0;j--)
		{
			if(heights[j]>heights[cur])
			{
				break;
			}
			else if(heights[j]<heights[cur])cur=j;
		}
		if(cur<K)
		{
			heights[cur]++;
			continue;
		}
		for(int j=K+1;j<n;j++)
		{
			if(heights[j]>heights[cur])
			{
				break;
			}
			else if(heights[j]<heights[cur])cur=j;
		}
		heights[cur]++;
	}
	return heights;
    }
};
