class Solution {
public:
    vector<int> pourWater(vector<int>& heights, int V, int K) {
        int n=heights.size();
        for(int j=0;j<V;j++)
        {

            int left=K,i=K;
            while(i>=0 && heights[i]<=heights[left])
            {
                if(heights[i]<heights[left])
                    left=i;
                i--;
            }
            if(left<K)
            {
                heights[left]++;
                continue;
            }

            int right=K;
            i=K;
            while(i<n && heights[i]<=heights[right])
            {
                if(heights[i]<heights[right])
                    right=i;
                i++;
            }
            if(right>K)
            {
                heights[right]++;
                continue;
            }
            heights[K]++;
        }
        return heights;
            
    }
};
