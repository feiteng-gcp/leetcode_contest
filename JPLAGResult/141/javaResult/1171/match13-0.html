class Solution {
    int[][] dir = new int[][]{{0,1}, {0,-1}, {1,1}, {1,-1}, {1,0}, {-1,1}, {-1,-1}, {-1,0}};
    
    public int shortestPathBinaryMatrix(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        if(grid[0][0] != 0 || grid[m - 1][n - 1] != 0) return -1;
        if(m == 1 && n == 1) return 1;
        Queue<int[]> q = new ArrayDeque<>();
        q.add(new int[]{0, 0});
        int res = 0;
        grid[0][0] = -1;
        while(!q.isEmpty()) {
            res++;
            int size = q.size();
            while(size != 0) {
                int[] cur = q.poll();
                int x = cur[0], y = cur[1];
                for(int d = 0; d < 8; d++) {
                    int a = x + dir[d][0], b = y + dir[d][1];
                    if(a == m - 1 && b == n - 1) return res + 1;
                    if(valid(a, b, m, n)) {
                        if(grid[a][b] == 0) {
                            q.add(new int[]{a, b});
                            grid[a][b] = -1;
                        }
                    }
                }
                size--;
            }
        }
        return -1;
    }
    
    private boolean valid(int a, int b, int m, int n) {
        return a < m && b < n && a >= 0 && b >= 0;
    }
}
