class Solution {
    private final int[][] deltas = new int[][]{{-1, -1}, {-1, 0}, {-1, 1}, {1, -1}, {1, 0}, {1, 1}, {0, 1}, {0, -1}};
    public int shortestPathBinaryMatrix(int[][] grid) {
        int n = grid.length;
        int[][] dists = new int[n][n];
        for (int i = 0; i < n; ++i) {
            Arrays.fill(dists[i], Integer.MAX_VALUE);
        }
        if (grid[0][0] == 1 || grid[n - 1][n - 1] == 1) return -1;
        PriorityQueue<int[]> queue = new PriorityQueue<>((i1, i2) -> dists[i1[0]][i1[1]] - dists[i2[0]][i2[1]]);
        dists[0][0] = 1;
        queue.add(new int[]{0, 0});
        while (!queue.isEmpty()) {
            int[] node = queue.remove();
            int row = node[0];
            int col = node[1];
            // System.out.printf("%d, %d\n", row, col);
            
if (row == n - 1 && col == n - 1) return dists[n - 1][n - 1];
            for (int[] delta: deltas) {
                int newRow = row + delta[0];
                int newCol = col + delta[1];
                // System.out.printf("%d, %d, %d, %d, %d\n", row, col, newRow, newCol, dists[row][col]);
                if (newRow >= 0 && newRow < n && newCol >= 0 && newCol < n && grid[newRow][newCol] != 1 && (dists[newRow][newCol] > (dists[row][col] + 1))) {
                    dists[newRow][newCol] = dists[row][col] + 1;
                    queue.add(new int[]{newRow, newCol});
                }
            }
        }
        return -1;
    }
}
