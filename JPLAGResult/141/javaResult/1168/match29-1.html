import java.util.Vector;

class Solution {
    public void duplicateZeros(int[] arr) {
        int shift = 0;
        
Vector<Integer> newPos = new Vector<>();
        
        for (int i=0 ; i<arr.length ; ++i) {
            newPos.add(arr[i]);
            if (arr[i] == 0) {
                newPos.add(0);
            }
        }
        for (int i=0 ; i<arr.length ; ++i) {
            arr[i] = newPos.elementAt(i);
        }
    }
}
