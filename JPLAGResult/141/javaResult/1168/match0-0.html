class Solution {
        public void duplicateZeros(int[] arr) {
        int[] temp = new int[arr.length];
        int p=0;
        for(int i=0;i<arr.length;++i){
            if(p>=arr.length){
                for(int j=0;j<arr.length;++j)
                    arr[j]=temp[j];
                return;
            }
            if(arr[i]!=0)
                temp[p++]=arr[i];
            else {
                temp[p++]=0;
                if(p>=arr.length){
                    for(int j=0;j<arr.length;++j)
                        arr[j]=temp[j];
                    return;
                }
                temp[p++]=0;
            }
        }
        for(int j=0;j<arr.length;++j)
            arr[j]=temp[j];
        return;

    }
}
