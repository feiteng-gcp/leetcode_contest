class Solution {
    public void duplicateZeros(int[] arr) {
        int[] tmp=arr.clone();
        for(int i=0,j=0;j<arr.length;i++){
            if(tmp[i]!=0){
                arr[j++]=tmp[i];
            }else{
                arr[j++]=0;
                if(j<arr.length) arr[j++]=0;
            }
        }
    }
}
