class Solution {
    public void duplicateZeros(int[] arr) {
        List<Integer> res = new ArrayList<>();
        for(int n: arr){
            res.add(n);
            if(n == 0)
{
                res.add(n);
            }
        }
        for(int i = 0; i < arr.length; i++){
            arr[i] = res.get(i);
        }
    }
}
