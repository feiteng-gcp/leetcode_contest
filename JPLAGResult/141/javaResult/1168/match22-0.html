class Solution {
    public void duplicateZeros(int[] arr) {
        int idx = 0;
        ArrayList<Integer> lst = new ArrayList<>();
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == 0) {
                lst.add(0);
                lst.add(0);
            } else {
                lst.add(arr[i]);
            }
            if (lst.size() > arr.length) break;
        }
        for (int i = 0; i < arr.length; i++) {
            arr[i] = lst.get(i);
        }
    }
}
