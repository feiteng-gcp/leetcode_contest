class Solution {
    public void duplicateZeros(int[] arr) {
        int numCount = 0;
        int zeroCount = 0;
        int length = arr.length;
        
List<Integer> afterNum = new ArrayList<>(arr.length+10);
        for (int i : arr) {
            if(i == 0){
                zeroCount++;
                afterNum.add(i);
                afterNum.add(i);
            }else {
                numCount++;
                afterNum.add(i);
            }
            if(numCount+zeroCount*2 >= length){
                break;
            }
        
}
        for (int i = 0; i < length; i++) {
            arr[i] = afterNum.get(i);
        }
    }
}
