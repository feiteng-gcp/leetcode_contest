class Solution {
    public void duplicateZeros(int[] a) {
        
int n=a.length;
        ArrayList<Integer> l=new ArrayList<>();
        for (int i = 0; i <n ; i++) {
            if(a[i]==0){
                l.add(0);
                l.add(0);
            }
            else{
                l.add(a[i]);
            }
        
}
        for (int i = 0; i <n ; i++) {
            a[i]=l.get(i);
        }
    }
}
