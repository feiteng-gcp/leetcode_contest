class Solution {

	public void duplicateZeros(int[] arr) {
		int[] clone = arr.clone();
		for (int i = 0, j = 0; i < arr.length; i++, j++) {
			if (clone[j] == 0) {
				arr[i] = 0;
				i++;
				if (i < arr.length) {
					arr[i] = 0;
				}
			} else {
				arr[i] = clone[j];
			}
		}
	}
}
