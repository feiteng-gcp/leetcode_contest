class Solution {
    public void duplicateZeros(int[] arr) {
    if (arr == null || arr.length == 0) return;
    
List<Integer> temp = new ArrayList<>();
    for (int a : arr) {
        if (a == 0) {
            temp.add(0);
            temp.add(0);
        } else {
            temp.add(a);
        }
    }
    for (int i = 0; i < arr.length; i++) {
        arr[i] = temp.get(i);
    }
    }
}
