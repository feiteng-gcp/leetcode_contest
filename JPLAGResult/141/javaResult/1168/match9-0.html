class Solution {
    public void duplicateZeros(int[] arr) {
        if(arr.length == 0) return ;
        int []stack = new int[arr.length];
        int top = -1;
        for(int i=0;i<arr.length;++i) {
        	if(arr[i] == 0){
        		stack[++top] = 0;
                if(top == arr.length-1) break;
        		stack[++top] = 0;
        	}else {
        		stack[++top] = arr[i];
        	}
        	if(top == arr.length-1) break;
        }
        for(int i=0;i<arr.length;++i) {
        	arr[i] = stack[i];
        }
    }
}
