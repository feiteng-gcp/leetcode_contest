import java.util.Arrays;
import java.util.PriorityQueue;

class Solution {
    public void duplicateZeros(int[] arr) {
        int[] res = new int[2 * arr.length];
        int idx = 0;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == 0) {
                res[idx++] = 0;
                res[idx++] = 0;
            } else {
                res[idx++] = arr[i];
            }
        }
        for (int i = 0; i < arr.length; i++) {
            arr[i] = res[i];
        }
    }
}
