class Solution {
    public void duplicateZeros(int[] arr) {
        int c = 0;
        int[] f = new int[arr.length];
        for (int i = 0; i < arr.length; i++) f[i] = arr[i];
        for (int i = 0, j = 0; j < arr.length;) {
            if (f[i] != 0) {
                arr[j++] = f[i++];
            }
            else {
                arr[j++] = 0;
                if (j < arr.length) arr[j++] = 0;
                i++;
            }
        }
    }
}
