class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < values.length; i++){
            if (!map.containsKey(labels[i]))
                map.put(labels[i], new ArrayList<>());
            
            List<Integer> list = map.get(labels[i]);

            list.add(values[i]);

        }
        
        List<Integer> list = new ArrayList<>();
        for (int key : map.keySet()){
            List<Integer> valueList = map.get(key);
            Collections.sort(valueList, (a,b) -> (b-a));
            for (int i = 0; i < use_limit && i < valueList.size(); i++)
                list.add(valueList.get(i));
        }
        
        
        Collections.sort(list, (a,b) -> (b-a));
        //System.out.println(list);
        
        int ans = 0;
        for (int i = 0; i < num_wanted && i < list.size(); i++){
            ans += list.get(i);
        }
        return ans;
        
    }
}
