class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int nw = num_wanted;
        int ul = use_limit;
        HashMap<Integer, Integer> vs = new HashMap<>();
        int ans = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int i = 0; i < values.length; i++) {
            
pq.add(-(values[i]*50000+labels[i]));
        }
        int size = 0;
        while (size < nw && !pq.isEmpty()) {
            int p0 = -pq.poll();
            int val = p0/50000;
            int lab = p0%50000;
            if (!vs.containsKey(lab))
                vs.put(lab, 0);
            
if (vs.get(lab) < ul) {
                size++;
                vs.put(lab, vs.get(lab)+1);
                ans += val;
            }
        }
        return ans;
    }
}
