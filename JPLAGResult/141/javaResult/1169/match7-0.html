class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int n = values.length;
        //int[] arr = new int[n];
        List<Integer> arr = new ArrayList<>();
        for(int i = 0; i < n; i++) arr.add(i);
        Collections.sort(arr, new Comparator<Integer>() {
			public int compare(Integer o1, Integer o2) {
				return values[o2] - values[o1];
			}
		});
        //for(int i = 0; i < n; i++) System.out.print(values[i] + " ");
        
        Map<Integer, Integer> count = new HashMap<>();
        int res = 0, cnt = 0;
        for(int i = 0; i < n; i++) {
            int l = labels[arr.get(i)], v = values[arr.get(i)];
            if(count.getOrDefault(l, 0) >= use_limit) continue;
            else {
                res += v;
                cnt++;
                if(cnt >= num_wanted) break;
                count.put(l, count.getOrDefault(l, 0) + 1);
            }
        }
        
        return res;
    }
}
