class Solution {
    
    class Number{
        int value;
        int label;
        
        public Number(int v, int l) {
            value = v;
            label = l;
        }
    }
    
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        Number numArray[] = new Number[values.length];
        for(int i = 0; i < values.length; i++)
            numArray[i] = new Number(values[i], labels[i])
;
        Arrays.sort(numArray, new Comparator<Number>(){
           public int compare(Number n1, Number n2) {
               return n2.value - n1.value;
           } 
        });
        
        int result = 0;
        Map<Integer, Integer> labelToFreq = new HashMap<Integer, Integer>();
        for(int i = 0; i < numArray.length; i++) {
            labelToFreq.putIfAbsent(numArray[i].label, 0);
            if( labelToFreq.get(numArray[i].label) >= use_limit )
                continue;
            result += numArray[i].value;
            labelToFreq.put(numArray[i].label, labelToFreq.get(numArray[i].label) + 1);
            num_wanted--;
            if( num_wanted == 0 )
                return result;
        }
        return result;
    }
}
