import java.util.*;

class Node implements Comparable<Node> {
    int value;
    int label;
    
    public Node(int v, int l) {
        value = v;
        label = l;
    }
    
    public int compareTo(Node n) {
        if (value > n.value)
            return -1;
        if (value < n.value) {
            return 1;
        }
        if (label > n.label) {
            return 1;
        }
        if (label < n.label) {
            return -1;
        }
        return 0;
    }
}

class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        Vector<Node> v = new Vector<>();
        HashMap<Integer, Integer> used = new HashMap<>();
        for (int i =0 ; i < values.length ; ++i) {
            v.add(new Node(values[i], labels[i]));
            used.put(labels[i], 0);
        }
        
        Collections.sort(v);
        int taken = 0;
        int sum = 0;
        
        for (Node n : v) {
            if (taken >= num_wanted) return sum;
            
            int k = used.get(n.label);
            if (k < use_limit) {
                sum += n.value;
                ++taken;
                used.put(n.label, k+1);
            }    
        }
        
        return sum;
    }
}
