import java.util.ArrayList;
import java.util.Comparator;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Map.Entry;


class Solution {
	public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
		Map<Integer, List<Integer>> m = new HashMap<Integer, List<Integer>>();
		for (int i = 0; i < values.length; i++) {
			List<Integer> l = m.getOrDefault(labels[i], new ArrayList<Integer>());
			l.add(values[i]);

			m.put(labels[i], l);
		}
		ArrayList<Integer> s = new ArrayList<Integer>();
		for (Entry<Integer, List<Integer>> en : m.entrySet()) {
			List<Integer> l = en.getValue();
			l.sort(Comparator.reverseOrder());
			for (int i = 0; i < Math.min(use_limit, l.size()); i++) {
				s.add(l.get(i));
			}
		}
		s.sort(Comparator.reverseOrder());
		int ans = 0;
		for (int i = 0; i < Math.min(num_wanted, s.size()); i++
) {
			ans += s.get(i);
		}
		return ans;
	}
}
