class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int[][] f = new int[values.length][2];
        for (int i = 0; i < values.length; i++) {
            f[i][0] = values[i];
            f[i][1] = labels[i];
        }
        Arrays.sort(f, (a, b) -> b[0] - a[0]);
        int ret = 0, cnt = 0;
        HashMap<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < values.length && cnt < num_wanted; i++) {
            int t = map.getOrDefault(f[i][1], 0);
            if (t >= use_limit) continue;
            ret += f[i][0];
            map.put(f[i][1], t+1);
            cnt++;
        }
        return ret;
    }
}
