class Solution {
	public int largestValsFromLabels(int[] values, int[] labels, int numWanted, int useLimit) {
		int res = 0;

		// { value, label }
		PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> b[0] - a[0]);

		for (int i = 0; i < labels.length; i++) {
			pq.add(new int[] { values[i], labels[i] });
		}

		// <Label, UseCount>
		HashMap<Integer, Integer> bannedLabels = new HashMap<>();

		while (pq.size() > 0 && numWanted > 0) {
			int[] a = pq.poll();
			Integer useCount = bannedLabels.get(a[1]);
			if (useCount == null) {
				bannedLabels.put(a[1], 1);
			} else if (useCount < useLimit) {
				bannedLabels.put(a[1], useCount + 1);
			} else {
				continue;
			}
			numWanted--;
			res += a[0];
		}

		return res;
	}
}
