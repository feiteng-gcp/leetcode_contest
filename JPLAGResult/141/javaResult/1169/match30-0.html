import java.util.*;

class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        Map<Integer, PriorityQueue<Node>> map = new HashMap<>();
        int m = labels.length;
        for (int i = 0; i < labels.length; i++) {
            map.putIfAbsent(labels[i], new PriorityQueue<>(new Comparator<Node>() {
                @Override
                public int compare(Node o1, Node o2) {
                    return Integer.compare(o2.val, o1.val);
                }
            }));
            map.get(labels[i]).add(new Node(values[i], labels[i]));
        }
        PriorityQueue<Node> pq = new PriorityQueue<>(new Comparator<Node>() {
            @Override
            public int compare(Node o1, Node o2) {
                return Integer.compare(o2.val, o1.val);
            }
        });
        int sum = 0;
        for (int k : map.keySet()) {
            if (!map.get(k).isEmpty()) {
                pq.add(map.get(k).poll());
            }
        }
        if (pq.isEmpty()) {
            return 0;
        }
        Map<Integer, Integer> done = new HashMap<>();
        while (num_wanted > 0 && !pq.isEmpty()) {
            Node temp = pq.poll();
            if (done.containsKey(temp.id) && done.get(temp.id) == use_limit) {
                continue;
            }
            sum += temp.val;
            done.put(temp.id, done.getOrDefault(temp.id, 0) + 1);
            if (!map.get(temp.id).isEmpty()) {
                pq.add(map.get(temp.id).poll());
            }
            num_wanted--;
        }
        return sum;
    }

    private static final class Node {
        int val;
        int id;

        Node(int i, int b) {
            val = i;
            id = b;
        }
    }
}
