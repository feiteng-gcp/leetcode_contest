class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        PriorityQueue<int[]> Q = new PriorityQueue<>((lhs, rhs) -> rhs[0] - lhs[0]);
        
for (int i = 0; i < values.length; i++) {
            Q.offer(new int[] { values[i], labels[i] });
        }
        
        HashMap<Integer, Integer> map = new HashMap<>();
        int cnt = 0;
        int sum = 0;
        
while (!Q.isEmpty() && cnt < num_wanted) {
            int[] pair = Q.poll();
            
            if (map.getOrDefault(pair[1], 0) < use_limit) {
                sum += pair[0];
                cnt++;
                map.put(pair[1], map.getOrDefault(pair[1], 0) + 1);
            }
        }
        
        return sum;
    }
}
