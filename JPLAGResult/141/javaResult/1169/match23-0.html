class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int[][] items = new int[values.length][2];
        for (int i = 0; i < values.length; ++i) {
            items[i][0] = values[i];
            items[i][1] = labels[i];
        }
        Arrays.sort(items, new Comparator<int[]>() {
            public int compare(int[] i1, int[] i2) {
                return -i1[0] + i2[0];
            }
        });
        int[] cnt = new int[20001];
        int i = 0;
        int j = 0;
        int res = 0;
        while (i < items.length && j < num_wanted) {
            int[] item = items[i];
            if (cnt[item[1]] < use_limit) {
                res += item[0];
                ++cnt[item[1]];
                ++i;
                ++j;
            } else {
                ++i;
            }
        }
        return res;
    }
}
