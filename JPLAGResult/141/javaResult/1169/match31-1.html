class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {

        int n=values.length;
        
PriorityQueue<Node> pq=new PriorityQueue<>(new Comparator<Node>() {
            @Override
            public int compare(Node o1, Node o2) {
                return Integer.compare(o2.val,o1.val);
            }
        });
        int can[]=new int[1000000];
        for (int i = 0; i <n ; i++) {
            pq.add(new Node(values[i],labels[i]));
            can[labels[i]]=use_limit;
        }
        int ans=0;
        while (!pq.isEmpty() && num_wanted>0){
            Node temp=pq.remove();
            if(can[temp.lab]>0){
                ans+=temp.val;
                can[temp.lab]--;
                num_wanted--;
            
}
        }
        return ans;
    }

    class Node{
        int val,lab;

        public Node(int val, int lab) {
            this.val = val;
            this.lab = lab;
        }
    }
}
