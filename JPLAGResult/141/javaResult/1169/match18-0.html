class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int[][] pairs  = new int [values.length][2];
        for(int i=0; i<values.length; i++){
            pairs[i][0] = values[i];
            pairs[i][1] = labels[i];
        }
        Arrays.sort(pairs, (x,y) -> y[0] - x[0]);
        int count = 0, result = 0;
        int[] used = new int[20005];
        for(int i=0; i<values.length && count<num_wanted; i++){
            if(used[pairs[i][1]] < use_limit){
                used[pairs[i][1]] ++;
                count ++;
                result += pairs[i][0];
            }
        }
        return result;
    }
}
