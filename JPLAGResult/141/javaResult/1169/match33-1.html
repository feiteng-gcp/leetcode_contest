class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        List<Item> list = new ArrayList<>();
        Map<Integer, Integer> label = new HashMap<>();
        for(int i = 0 ; i < values.length ; i++){
            list.add(new Item(values[i], labels[i]));
            label.put(labels[i], use_limit);
        }
        
        list.sort((i1, i2) -> (i2.value - i1.value));
        int sum = 0;
        
        for(int i = 0 ; i < list.size(); i++){
            if(num_wanted > 0){
                if(label.get(list.get(i).label) > 0 ){
                    num_wanted--;
                    sum += list.get(i).value;
                    label.put(list.get(i).label, label.get(list.get(i).label) - 1);
                }
            }
        }
        
        return sum;
    }
}


class Item{
    int value;
    int label;
    public Item(int v, int l){
        value = v;
        label = l;
    }
}
