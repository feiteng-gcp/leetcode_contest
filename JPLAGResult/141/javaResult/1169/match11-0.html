class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int n = values.length;
        PriorityQueue<int[]> pq = new PriorityQueue<>(n, (a, b) ->(b[0] - a[0]));
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < values.length; i++) {
            pq.offer(new int[]{values[i], labels[i]});
            map.put(labels[i], Math.min(map.getOrDefault(labels[i], 0) + 1, use_limit));
        }
        int max = 0;
        while (!pq.isEmpty() && num_wanted > 0) {
            int[] item = pq.poll();
            if (map.get(item[1]) > 0) {
                map.put(item[1], map.get(item[1]) - 1);
                max += item[0];
                num_wanted--;
            }
        }
        return max;
    }
}
