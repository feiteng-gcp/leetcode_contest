class Solution {
    
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        // Sort
        int n = values.length;
        int[][] pair = new int[n][2];
        
        for(int i=0;i<n;i++){
            pair[i][0] = labels[i];
            pair[i][1] = values[i];
        }
        
        Arrays.sort(pair, new Comparator<int[]>(){
           public int compare(int[] a, int[] b){
               return b[1] - a[1];
           }
        });
        
        // Map: size < use_limit
        Map<Integer, Integer> countMap = new HashMap<>();
        int sum = 0, cnt = 0;
        
        for(int i=0;i<n;i++){
            if(cnt >= num_wanted)   break;
            int lab = pair[i][0], val = pair[i][1];
            // if(!countMap.containsKey(lab)){
            //     countMap.put(lab, new ArrayList<>());
            // }
            
            if(countMap.getOrDefault(lab, 0) >= use_limit)   continue;
            else
{
                countMap.put(lab, countMap.getOrDefault(lab,0) + 1);
                cnt++;
                sum += val;
            }
        }
        
        return sum;
    }
}
