class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int len=values.length,ans=0;
        int[][] arr=new int[len][];
        Map<Integer, Integer> lab = new HashMap<>();
        for(int i=0;i<len;i++){
            arr[i]=new int[]{values[i],labels[i]};
            lab.put(labels[i],0);
        }
        Arrays.sort(arr,(o1,o2)->(o2[0]-o1[0]));
        int i=0;
        for(int[] val : arr){
            if(lab.get(val[1])==use_limit) continue;
            if(i++==num_wanted) break;
            lab.put(val[1],lab.get(val[1])+1);
            ans+=val[0];
        }
        return ans;
    }
}
