class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int n = values.length;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> -Integer.compare(a[0], b[0]));
        for (int i = 0; i < n; ++i) {
            pq.offer(new int[] { values[i], labels[i] });
        }
        int total = 0;
        Map<Integer, Integer> usedMap = new HashMap<>();
        // Set<Integer> discMap = new HashSet<>();
        
while (num_wanted > 0 && !pq.isEmpty()) {
            int[] cur = pq.poll();
            if (usedMap.getOrDefault(cur[1], 0) < use_limit) {
                total += cur[0];
                --num_wanted;
                usedMap.put(cur[1], 1 + usedMap.getOrDefault(cur[1], 0));
            }
        }
        
        return total;
    }
}
