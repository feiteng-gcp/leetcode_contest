class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int res = 0;
        PriorityQueue<Node> pq= new PriorityQueue<Node>((n1, n2) -> n2.val - n1.val);
        for(int i = 0; i < values.length; i++) {
            pq.offer(new Node(values[i], labels[i]));
        
}
        Map<Integer, Integer> map = new HashMap<>();
        while(!pq.isEmpty() && num_wanted > 0) {
            Node node = pq.poll();
            int preN = map.getOrDefault(node.label, 0);
            if(preN < use_limit) {
                res += node.val;
                map.put(node.label, preN + 1);
                num_wanted--;
            }
        }
        return res;
    }
    
    class Node {
        int val;
        int label;
        Node(int val, int label) {
            this.val = val;
            this.label = label;
        }
    }
}
