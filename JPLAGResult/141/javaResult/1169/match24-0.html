class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int[][] cur = new int[values.length][2];
        for (int i = 0; i < values.length; i++) {
            cur[i][0] = values[i];
            cur[i][1] = labels[i];
        }
        Arrays.sort(cur, new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                if (o1[0] == o2[0]) return o1[1] - o2[1];
                return o2[0] - o1[0];
            }
        });
        int res = 0, num = 0;
        Map<Integer, Integer> map = new HashMap<>();
        for (int[] cur1 : cur) {
            if (!map.containsKey(cur1[1]) || map.get(cur1[1]) < use_limit) {
                res += cur1[0];
                if (map.containsKey(cur1[1])) {
                    map.put(cur1[1], map.get(cur1[1]) + 1);
                } else {
                    map.put(cur1[1], 1);
                }
                num++;
            }
            if (num == num_wanted) return res;
        }
        return res;
    }
}
