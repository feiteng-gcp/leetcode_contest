class Solution {
    public int largestValsFromLabels(int[] vs, int[] ls, int wt, int limit) {
        Map<Integer,List<Integer>> mp  = new HashMap<>();
        List<List<Integer>> res = new ArrayList<>();
        
        for(int i=0;i<ls.length;++i){
            
            int id = ls[i];
            if(!mp.containsKey(id)){
                mp.put(id,new ArrayList<Integer>());
            }
            
            mp.get(id).add(vs[i]);
            
        }
        
        for(Integer k:mp.keySet()){
            List<Integer>li = mp.get(k);
            Collections.sort(li);
            Collections.reverse(li);
            res.add(li);
            
            
        }
        
        if(limit==1){
            
            int a[] = new int[res.size()];
            int p = 0;
            
            for(List<Integer> g:res){
                a[p++] = g.get(0); 
            }
            Arrays.sort(a);
            int cc =0;
            int rp = 0;
            for(int i=a.length-1;i>=0;--i){
                cc++;
                rp += a[i];
                if(cc==wt){
                    return rp;
                }
            }
            return rp;
            
        }
        
        
        int dp[] = new int[wt+1];
        Arrays.fill(dp,-1000000000);
        dp[0] = 0;
        
        
        int r =  0;
        
            
            for(List<Integer> go:res){
                for(int i=wt;i>=0;--i){
                int g1 = go.size();
                int s = 0;
                
                for(int g=0;g<Math.min(limit,g1);++g){
                    s += go.get(g);
                if(i+g+1<=wt){
                    dp[i+g+1] = Math.max(dp[i+g+1],dp[i]+s);
                    r = Math.max(r,dp[i+g+1]);
                }
                }
                
            }
            
        }
        return r;
        
        
        
        
        
    }
    
    
    
}
