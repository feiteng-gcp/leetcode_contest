class Solution {
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int n = values.length;
        int[][] items = new int[n][2];
        
        for (int i=0; i<n; ++i) {
            items[i][0] = values[i];
            items[i][1] = labels[i];
        }
        
        Arrays.sort(items, (a,b) -> (b[0] - a[0]) );
        // for (int[] item: items){
        //     System.out.println(Arrays.toString(item));
        // }
        
        int v = 0;
        int total = 0;
        Map<Integer, Integer> count = new HashMap<>();
        for (int[] item: items) {
            int value = item[0], id = item[1];
            if (total >= num_wanted) break;
            
            
int use = count.getOrDefault(id, 0);
            if (use >= use_limit) continue;
            
            v += value;
            ++total;
            count.put(id, use+1);
        }
        
        return v;
    }
}
