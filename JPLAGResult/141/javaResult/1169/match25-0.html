class Solution {
    class MyPair {
        int value;
        int label;
        MyPair(int v, int i) {
            this.value = v;
            this.label = i;
        }
    }
    public int largestValsFromLabels(int[] values, int[] labels, int num_wanted, int use_limit) {
        int len  = values.length;
        List<MyPair> list = new ArrayList<>();
                
        Map<Integer, Integer> usedLimits = new HashMap<>();
        for (int i=0;i<len; i++) {
            list.add(new MyPair(values[i], labels[i]));
            usedLimits.put(labels[i], 0);
        }

        
        Collections.sort(list, new Comparator<MyPair> () {
            public int compare (MyPair a, MyPair b) {
                return b.value-a.value;
            }
        });
            
            int total=0;
            int count=0;
        
                
        for (MyPair p: list) {
            //System.out.println(p.value + " , " + p.label);
            if (count >= num_wanted) {return total;}
            int usedSoFar = usedLimits.get(p.label);
            if (usedSoFar < use_limit) {
                total += p.value;
                count++;
                usedLimits.put(p.label, usedSoFar + 1);
            }
        }

        return total;
    }
}
