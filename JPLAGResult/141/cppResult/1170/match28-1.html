class Solution {
public:
	string shortestCommonSupersequence(string str1, string str2) {

		int n = str1.size();
		int m = str2.size();

		vector<vector<string>> dp(n+1, vector<string>(m+1, ""));

		string tmp = "";
		for (int i = 1; i <= n; i++)
		{
			for (int j = 1; j <= m; j++)
			{
				if (str1[i-1] == str2[j-1])
				{
					dp[i][j] = dp[i - 1][j - 1] + str1[i - 1];

				}
				else if (dp[i][j - 1].size() > dp[i-1][j].size())
					dp[i ][j ] = dp[i][j-1];
				else
					dp[i][j] = dp[i-1][j];
				if (dp[i][j].size() > tmp.size())
					tmp = dp[i ][j ];
			}
		}
		/*for (int i = 0; i <= n; i++)
		{
			for (int j = 0; j <= m; j++)
			{
				cout << dp[i][j] << '-';
			}
			cout << endl;
		}

		
		cout << "tmp="<<tmp << endl;*/
		
		int pos1 = 0;
		int pos2 = 0;
		string ans = "";
		for (int i = 0; i < tmp.size(); i++)
		{
			while (pos1 < str1.size() && str1[pos1] != tmp[i])
				ans += str1[pos1++];
			while (pos2< str2.size() && str2[pos2] != tmp[i])
				ans += str2[pos2++];
			ans += tmp[i];
			pos1++;
			pos2++;
		}
		while (pos1 <str1.size())
			ans += str1[pos1++];
		while (pos2 <str2.size())
			ans += str2[pos2++];
		return ans;
	}
};
