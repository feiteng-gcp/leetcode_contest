class Solution {
public: 
    
    string shortestCommonSupersequence(string str1, string str2) {
        int m = str1.size(); 
        int n = str2.size(); 
        int dp[m + 1][n + 1]; 
 
        for (int i = 0; i <= m; i++) { 
            for (int j = 0; j <= n; j++) {  
                if(i == 0) 
                    dp[i][j] = j; 
                else if(j == 0) 
                    dp[i][j] = i; 
                else if(str1[i - 1] == str2[j - 1]) 
                    dp[i][j] = 1 + dp[i - 1][j - 1]; 
                else
                    dp[i][j] = 1 + min(dp[i - 1][j], dp[i][j - 1]); 
            } 
        } 

        int ind = dp[m][n]; 

        string str; 
        int i = m, j = n; 
        while (i > 0 && j > 0) {  
            if (str1[i - 1] == str2[j - 1]) {  
                str.push_back(str1[i - 1]); 
                i--, j--, ind--; 
            }  
            else if (dp[i - 1][j] > dp[i][j - 1]) {  
                str.push_back(str2[j - 1]); 
                j--, ind--; 
            } 
            else {  
                str.push_back(str1[i - 1]); 
                i--, ind--; 
            } 
        }  
        while (i > 0) { 
            str.push_back(str1[i - 1]); 
            i--, ind--; 
        } 
 
        while (j > 0) { 
            str.push_back(str2[j - 1]); 
            j--, ind--; 
        }  
        reverse(str.begin(), str.end()); 
        return str; 
    }
};
