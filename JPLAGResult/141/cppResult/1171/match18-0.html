class Solution {
public:
    int shortestPathBinaryMatrix(vector<vector<int>>& grid) {
        if (grid[0][0] == 1) return -1;
        //cout << grid.size() << grid[0].size() << endl;
        vector<vector<int>> visited(grid.size(),vector<int>(grid[0].size(), 0));
       // cout << visited.size() << visited[0].size() << endl;
        queue<pair<int, int>> myq;
        myq.push(make_pair(0,0));
        visited[0][0] = 1;
        while(!myq.empty()) {
            int x = myq.front().first;
            int y = myq.front().second;
            myq.pop();
            
            //cout << x << " " << y << endl;
            
            if (x == grid.size()-1 && y == grid.size()-1)
                return visited[x][y];
            
            if (x-1 >=0 && visited[x-1][y] == 0 && grid[x-1][y] == 0) {
                myq.push(make_pair(x-1,y));
                visited[x-1][y] = visited[x][y] + 1;
            }
            
            if (y-1 >=0 && visited[x][y-1] == 0 && grid[x][y-1] == 0) {
                myq.push(make_pair(x,y-1));
                visited[x][y-1] = visited[x][y] + 1;
            }
            if (x+1 < grid.size() && visited[x+1][y] == 0 && grid[x+1][y] == 0) {
                myq.push(make_pair(x+1,y));
                visited[x+1][y] = visited[x][y] + 1;
            }
            if (y+1 < grid[0].size() && visited[x][y+1] == 0 && grid[x][y+1] == 0) {
                myq.push(make_pair(x,y+1));
                visited[x][y+1] = visited[x][y] + 1;
            }
            if (x-1 >=0 && y-1 >=0 && visited[x-1][y-1] == 0 && grid[x-1][y-1] == 0) {
                myq.push(make_pair(x-1,y-1));
                visited[x-1][y-1] = visited[x][y] + 1;
            }
            if (x-1 >=0 && y+1 < grid[0].size() && visited[x-1][y+1] == 0 && grid[x-1][y+1] == 0) {
                myq.push(make_pair(x-1,y+1));
                visited[x-1][y+1] = visited[x][y] + 1;
            }
            if (x+1 < grid.size() && y-1 >=0 && visited[x+1][y-1] == 0 && grid[x+1][y-1] == 0) {
                myq.push(make_pair(x+1,y-1));
                visited[x+1][y-1] = visited[x][y] + 1;
            }
            if (x+1 < grid.size() && y+1 < grid[0].size() && visited[x+1][y+1] == 0 && grid[x+1][y+1] == 0) {
                myq.push(make_pair(x+1,y+1));
                visited[x+1][y+1] = visited[x][y] + 1;
            }
        }
        return -1;
    }
};
