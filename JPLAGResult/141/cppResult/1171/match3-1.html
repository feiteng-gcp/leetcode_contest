class Solution {
public:
    int shortestPathBinaryMatrix(vector<vector<int>>& g) {
        int n=g.size();
        if (g[0][0]==1 || g[n-1][n-1]==1) return -1;
        vector<int> dx;
        dx.push_back(0); dx.push_back(1); dx.push_back(0); dx.push_back(-1); 
        dx.push_back(1); dx.push_back(1); dx.push_back(-1); dx.push_back(-1); 
        vector<int> dy;
        dy.push_back(1); dy.push_back(0); dy.push_back(-1); dy.push_back(0); 
        dy.push_back(1); dy.push_back(-1); dy.push_back(1); dy.push_back(-1); 
        queue<pair<int,int> > que;
        que.push(make_pair(0,0));
        g[0][0]=2;
        while (!que.empty()) {

            int x=que.front().first,y=que.front().second; que.pop();
            for (int i=0; i<8; i++) {
                int tx=x+dx[i],ty=y+dy[i];
                if (tx>=0 && tx<n && ty>=0 && ty<n && g[tx][ty]==0) {
                    g[tx][ty]=g[x][y]+1;
                    que.push(make_pair(tx,ty));
                }
            }
        }
        return g[n-1][n-1]-1;
    }
};
