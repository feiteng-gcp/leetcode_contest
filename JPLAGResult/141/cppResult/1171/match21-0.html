typedef pair<int,int> pii;

class Solution {
public:
    int dir[8][2] = {{-1,-1}, {-1, 0}, {-1, 1}, {1,-1}, {1, 0}, {1, 1}, {0,-1}, {0, 1}};
    
    int N;
    queue<pii> Q;
    int D[105][105];
    
    bool ok(int x, int y, vector<vector<int>>& grid) {
        return x >= 0 && x < N && y >= 0 && y < N && grid[x][y] == 0 && D[x][y] == -1;
    }
    
    int shortestPathBinaryMatrix(vector<vector<int>>& grid) {
        N = grid.size();
        while (!Q.empty()) Q.pop();
        memset(D,-1,sizeof(D));
        D[0][0] = 0;
        if (grid[0][0] == 1) return -1;
        Q.push(pii(0,0));
        while (!Q.empty()) {
            auto cur = Q.front();
            Q.pop();
            if (cur == pii(N-1, N-1)) break;
            for (int k=0;k<8;++k) {
                auto nex = pii(cur.first + dir[k][0], cur.second + dir[k][1]);
                if (ok(nex.first, nex.second, grid)) {
                    D[nex.first][nex.second] = D[cur.first][cur.second] + 1;
                    Q.push(nex);
                }
            }
        }
        return D[N-1][N-1] != -1 ? D[N-1][N-1] + 1 : -1;
    }
};
