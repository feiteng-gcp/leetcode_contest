class Solution {
public:
    int shortestPathBinaryMatrix(vector<vector<int>>& g) {
        int gs=g.size();
        if(g[0][0] != 0 || g[gs-1][gs-1] !=0)
            return -1;
        vector<vector<int>> vi(gs, vector<int>(gs,0));
        vi[0][0]=1;
        int pos=0*1000+0;
        queue<int> q;
        q.push(pos);
        int ret=1;
        while(!q.empty())
        {
            int qs=q.size();
            for(int i=0;i<qs;i++)
            {
                int tp=q.front();
                q.pop();
                int tr=tp/1000;
                int tc=tp%1000;
                //cout<<tr<<" "<<tc<<endl;
                if(tr == gs-1 && tc == gs-1)
                    {return ret;}
                if(tr>0 && tc>0 && vi[tr-1][tc-1] == 0 &&  g[tr-1][tc-1] == 0)
                {
                    vi[tr-1][tc-1] = 1;
                    q.push((tr-1)*1000+tc-1);
                }
                if(tr>0 && vi[tr-1][tc] == 0 && g[tr-1][tc] == 0)
                {
                    vi[tr-1][tc] = 1;
                    q.push((tr-1)*1000+tc);
                }
                if(tr>0 && tc+1<gs && vi[tr-1][tc+1] == 0 && g[tr-1][tc+1] == 0)
                {
                    vi[tr-1][tc+1] = 1;
                    q.push((tr-1)*1000+tc+1);
                }
                ////
                if(tc>0 && vi[tr][tc-1] == 0 && g[tr][tc-1] == 0)
                {
                    vi[tr][tc-1] = 1;
                    q.push((tr)*1000+tc-1);
                }
                if(tc+1<gs && vi[tr][tc+1] == 0 && g[tr][tc+1] == 0)
                {
                    vi[tr][tc+1] = 1;
                    q.push((tr)*1000+tc+1);
                }
                ///
                if(tc>0 && tr+1<gs &&  vi[tr+1][tc-1] == 0 &&g[tr+1][tc-1] == 0)
                {
                    vi[tr+1][tc-1] = 1;
                    q.push((tr+1)*1000+tc-1);
                }
                if(tr+1<gs && vi[tr+1][tc] == 0 && g[tr+1][tc] == 0)
                {
                    vi[tr+1][tc] = 1;
                    q.push((tr+1)*1000+tc);
                }
                if(tc+1<gs && tr+1<gs && vi[tr+1][tc+1] == 0 && g[tr+1][tc+1] == 0)
                {
                    vi[tr+1][tc+1] = 1;
                    q.push((tr+1)*1000+tc+1);
                }
            }
            ret++;
        }
        return -1;
    }
};
