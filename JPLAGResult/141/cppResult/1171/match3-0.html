class Solution {
public:
    int shortestPathBinaryMatrix(vector<vector<int>>& grid) {
        int N = grid.size();
        if(grid[0][0] == 1 || grid[N-1][N-1] == 1) return -1;
        queue<int> qr, qc;
        vector<vector<int>> dst;
        for(int i=0; i<N; i++) {
            dst.push_back(vector<int>(N, -1));
        }
        qr.push(0);
        qc.push(0);
        dst[0][0] = 1;
        int dr[] = {1, 1, 1, 0, 0, -1, -1, -1};
        int dc[] = {1, 0, -1, 1, -1, 1, 0, -1};
        while(!qr.empty()) {
            int nr, nc, ndst;
            nr = qr.front(); qr.pop();
            nc = qc.front(); qc.pop();
            ndst = dst[nr][nc];
            //printf("%d %d %d\n", nr, nc, ndst);
            for(int di=0; di<8; di++) {
                int nnr, nnc;
                nnr = nr+dr[di];
                nnc = nc+dc[di];
                if(nnr>=0 && nnr<N && nnc>=0 && nnc<N && dst[nnr][nnc] == -1 && grid[nnr][nnc] == 0) {
                    qr.push(nnr);
                    qc.push(nnc);
                    dst[nnr][nnc] = ndst+1;
                }
            }
        }
        return dst[N-1][N-1];
    }
};
