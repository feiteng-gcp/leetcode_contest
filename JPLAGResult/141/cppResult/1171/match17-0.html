#define X first
#define Y second
typedef pair<int, int> pii;
int dx[]={0, 0, -1, 1, -1, -1, 1, 1};
int dy[]={-1, 1, 0, 0, -1, 1, -1, 1};

class Solution {
public:
    int shortestPathBinaryMatrix(vector<vector<int>>& grid) {
        map<pii, int> vis;
        int n=grid.size();
        int m=grid[0].size();
        vis[pii(0, 0)]=0;
        queue<pii> Q;
        Q.push(pii(0, 0));
        while(!Q.empty()) {
            pii u=Q.front();Q.pop();
            for(int i=0; i<8; i++) {
                int Nx=u.X+dx[i];
                int Ny=u.Y+dy[i];
                if (Nx<0||Nx>=n||Ny<0|Ny>=m) continue; 
                if (vis.count(pii(Nx, Ny))) continue;
                if (grid[Nx][Ny]) continue;
                vis[pii(Nx, Ny)]=vis[pii(u.X, u.Y)]+1;
                // cout<<"s"<<Nx<<' '<<Ny<<' '<<vis[pii(Nx, Ny)]<<endl;
                Q.push(pii(Nx, Ny));
            }
        }
        if (!vis.count(pii(n-1, m-1))||grid[0][0]||grid[n-1][m-1]) return -1;
        else return vis[pii(n-1, m-1)]+1;
    }
};
