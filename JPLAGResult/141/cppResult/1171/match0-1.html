struct node
{
	int x;
	int y;
	int step;
};
class Solution {
public:
	int shortestPathBinaryMatrix(vector<vector<int>>& grid) {

		int xx[8] = { -1, -1, -1, 0, 0, 1, 1, 1 };
		int yy[8] = { 1, 0, -1, 1, -1, 1, 0, -1 };

		int n = grid.size();
		int m = grid[0].size();

		if (grid[0][0] == 1)
			return -1;

		node now,next;
		now.x = 0;
		now.y = 0;
		now.step = 1;

		queue<node>q;

		q.push(now);

		vector<vector<bool>> vis(n, vector<bool>(m, false));

		vis[now.x][now.y] = true;

		while (!q.empty())
		{
			now = q.front();
			q.pop();
			if (now.x == n - 1 && now.y == m - 1)
				return now.step;
			for (int i = 0; i < 8; i++)
			{
				next.x = now.x + xx[i];
				next.y = now.y + yy[i];
				next.step = now.step + 1;

				if (next.x < 0 || next.x >= n || next.y < 0 || next.y >= m || vis[next.x][next.y] == true || grid[next.x][next.y] == 1)
					continue;

				vis[next.x][next.y] = true;
				q.push(next);
			}
		}

		return -1;

	}
};
