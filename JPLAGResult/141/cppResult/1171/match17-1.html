bool vis[102][102];
int d[102][102];
const int INF=0x7fffffff;
struct Node{
    int r, c;
    Node(){}
    Node(int a, int b){r=a,c=b;}
};

int fc[10]={-1,0,1,-1,1,-1,0,1};
int fr[10]={-1,-1,-1,0,0,1,1,1};
class Solution {
public:
    int shortestPathBinaryMatrix(vector<vector<int>>& grid) {
        memset(vis,0,sizeof(vis));
        memset(d,-1,sizeof(d));
        queue<Node> q;
        q.push(Node(0, 0));
        d[0][0]=1;
        vis[0][0]=1;
        int N = grid.size();
        if(grid[0][0]||grid[N-1][N-1]){
            return -1;
        }

        while(!q.empty()){
            Node cur = q.front();
            q.pop();
            for(int i=0;i<8;i++){
                int cr = cur.r + fr[i];
                int cc = cur.c + fc[i];
                if(cc<0||cr<0||cc>=N||cr>=N)
                    continue;
                if(vis[cr][cc]||grid[cr][cc])
                    continue;
                vis[cr][cc] = 1;
                d[cr][cc] = d[cur.r][cur.c] + 1;
                q.push(Node(cr, cc));
            }
        }
        return d[N-1][N-1];
    }
};
