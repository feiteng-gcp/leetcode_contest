class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        vector<int> tmp;
        int n=arr.size();
        for(int i=0; i<n; i++){
            tmp.push_back(arr[i]);
            if(arr[i]==0) tmp.push_back(0);
        }
        
        for(int i=0; i<n; i++) arr[i]=tmp[i];
    }
};
