class Solution {
public:
    void duplicateZeros(vector<int>& a) {
        vector<int> b(a);
        int p = 0;
        for (int i = 0; p < b.size() and i < b.size(); ++i) {
            if (b[i] == 0) {

                a[p] = 0;
                if (p + 1 < b.size())
                    a[p + 1] = 0;
                p += 2;
            } else {
                a[p] = b[i];
                p++;
            }
        }
    }
};
