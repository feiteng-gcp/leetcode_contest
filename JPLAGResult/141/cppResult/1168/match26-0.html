class Solution {
public:
  void duplicateZeros(vector<int>& arr) {
    vector<int> tmp(arr);
    int n=tmp.size();
    for(int i=0,j=0;i<n&&j<n;i++){
      if(tmp[i]==0){
        arr[j++]=0;
        if(j<n) arr[j++]=0;
      }else{
        arr[j++]=tmp[i];
      }
    }
    //    for(int x:arr) cout<<x<<endl;
  }
};
