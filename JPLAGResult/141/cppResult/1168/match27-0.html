class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        vector<int> a;
        for (int i : arr) {
            if (i==0) {
                a.push_back(i);
            }
            a.push_back(i);
        }
        arr = vector<int>(a.begin(), a.begin()+arr.size());
    }
};
