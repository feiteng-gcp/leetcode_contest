class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        vector<int> ans;
        int n = arr.size();
        for (auto c: arr) {
            ans.push_back(c);
            if (!c) ans.push_back(c);
        }
        arr.clear();
        for (int i = 0; i < n; ++i) arr.push_back(ans[i]);
        return ;
    }
};
