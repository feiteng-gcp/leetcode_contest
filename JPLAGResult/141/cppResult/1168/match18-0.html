class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        vector<int> num;
        for(int i=0;i<arr.size();i++){
            if(arr[i]!=0) num.push_back(arr[i]);
            else{
                num.push_back(0);
                num.push_back(0);
            }
        }
        for(int i=0;i<arr.size();i++){
            arr[i] = num[i];
        }
    }
};
