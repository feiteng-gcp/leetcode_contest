class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        vector<int> t;
        for(int i = 0; i < arr.size(); i++){
            if(arr[i] != 0){
                t.push_back(arr[i]);
            } else {
                t.push_back(0);
                t.push_back(0);
            }
        }
        for(int i = 0; i < arr.size(); i++){
            arr[i] = t[i];
        }
    }
};
