class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        int i;
        int n=arr.size();
        for(i=0;i<arr.size();i++)
        {
            if(arr[i]==0)
            {
                arr.insert(arr.begin()+i,0);
                i++;
            }
        }
        arr.erase(arr.begin()+n,arr.end());
        
    }
};
