class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        vector<int> data(arr.size());
        for (int i = 0, k = 0; i < arr.size() and k < arr.size(); ++i, ++k) {
            if (arr[k] != 0) data[i] = arr[k];
            else {
                data[i] = 0; 
                ++i;
                if (i < arr.size())
                data[i] = 0;
            }
        }
        swap(data, arr);
    }
};
