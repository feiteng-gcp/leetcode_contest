class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        vector<int> tmp;
        for(int i = 0; i < arr.size(); i ++){
            tmp.push_back(arr[i]);
            if(!arr[i]){
                tmp.push_back(arr[i]);
            }
        }
        
        for(int i = 0; i < arr.size(); i ++) arr[i] = tmp[i];
        return;
    }
};
