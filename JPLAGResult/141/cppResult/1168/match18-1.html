class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        vector<int> dup;
        for (int i = 0; i < arr.size() && dup.size() < arr.size(); ++ i) {
            if (arr[i] == 0) {
                dup.push_back(0);
                dup.push_back(0);

            } else {
                dup.push_back(arr[i]);
            }
        }
        
        for (int i = 0; i < arr.size(); ++ i) {
            arr[i] = dup[i];
        }
    }
};
