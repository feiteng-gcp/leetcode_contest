class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        int N = arr.size();
        for (int i = 0; i < N; i++) {
            if (arr[i] == 0) {
                arr.insert(arr.begin() + i, 0);
                i++;
            }
        }
        arr.resize(N);
    }
};
