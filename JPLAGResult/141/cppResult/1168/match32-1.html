class Solution {
public:
    void duplicateZeros(vector<int>& arr) {
        int cnt = 0;
        for (int i = 0; i < arr.size(); ++i) {
            if (arr[i] == 0) {
                cnt++;
            }
        }
        
        for (int i = arr.size() - 1; i >=0; --i){
            if (arr[i] == 0) {
                cnt--;
                if (cnt + i + 1 < arr.size()) {
                    arr[cnt+i+1] = 0;
                }
            } 
            
            if (cnt + i < arr.size()) {
                arr[cnt+i] = arr[i];
            }
        }
    }
};
