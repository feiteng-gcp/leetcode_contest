class Solution {
public:
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
        vector <pair<int, int> > vs;
        int n = values.size();
        for(int i = 0; i < n; i++) vs.push_back(make_pair(values[i], labels[i]));
        sort(vs.begin(), vs.end());
        reverse(vs.begin(), vs.end());
        vector <int> counts(20001, 0);
        int res = 0;
        int selected = 0;
        for(int i = 0; i < n and selected < num_wanted; i++){
            if(counts[vs[i].second] < use_limit){
                res += vs[i].first;
                counts[vs[i].second]++;
                selected++;
            }
        }
        return res;
    }
};
