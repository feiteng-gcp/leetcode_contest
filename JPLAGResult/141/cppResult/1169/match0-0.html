class lv{
    public:
    int vl;
    int lb;
    bool operator < (lv &r)
    {
        return vl>r.vl;
    };
};
class Solution {
public:
    int largestValsFromLabels(vector<int>& vl, vector<int>& lb, int nw, int use_limit) {
        int s=vl.size();
        vector<lv> v(s);
        unordered_map<int,int> mp;
        for(int i=0;i<s;i++)
        {
            v[i].vl=vl[i];
            v[i].lb=lb[i];
            mp[lb[i]]=use_limit;
        }
        sort(v.begin(),v.end());
        int ret=0;
        for(int i=0;i<s;i++)
        {
            if(nw>0)
            {
                if(mp[v[i].lb]>0)
                {
                    ret+=v[i].vl;
                    //cout<<v[i].vl<<endl;
                    mp[v[i].lb]--;
                    nw--;
                }
                
            }
        }
        return ret;
    }
};
