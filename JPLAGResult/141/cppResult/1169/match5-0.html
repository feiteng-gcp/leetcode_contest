class Solution {
public:
	int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {

		int n = values.size();
		vector<pair<int, int>> pp;
		for (int i = 0; i < n; i++)
		{
			pp.push_back(make_pair(values[i], labels[i]));
		}

		sort(pp.begin(), pp.end(), [](pair<int, int> a, pair<int, int> b)
		{
			if (a.first == b.first)
				return a.second <b.second;
			return a.first>b.first;
		});

		int ans = 0;

		map<int, int> mp;
		for (int i = 0; i < n; i++)
		{
			if (num_wanted)
			{
				if (mp[pp[i].second] < use_limit)
				{
					ans += pp[i].first;
					mp[pp[i].second]++;
					num_wanted--;
				}
			}
		}

		return ans;
	}
};
