typedef pair<int, int> P;
const int maxn = 1e5 + 10;
const int mod = 1e9 + 7;
const double eps = 1e-8;

pair<int, int> sk[maxn];
class Solution {
public:
    int largestValsFromLabels(vector<int> &values, vector<int> &labels,
                              int num_wanted, int use_limit) {
        int len = values.size();
        for (int i = 0; i < len; i++) {
            sk[i] = make_pair(values[i], labels[i]);
        }
        sort(sk, sk + len,
             [](const P &x, const P &y) { return x.first > y.first; });
        int ans = 0;
        int now = 0;
        map<int, int> np;
        for (int i = 0; i < len; i++) {
            if (np[sk[i].second] < use_limit) {
                ans += sk[i].first;
                np[sk[i].second]++;
                ++now;
            }
            if (now >= num_wanted) {
                return ans;
            }
        }
        return ans;
    }
};
