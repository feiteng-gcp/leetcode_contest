struct myop{
    bool operator()(pair<int, int> a, pair<int, int> b){
        if(a.first != b.first) return a.first < b.first;
        else return a.second < b.second;
    }
};


class Solution {
public:
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
        priority_queue<pair<int, int>, vector<pair<int, int>>, myop> pq;
        for(int i = 0; i < labels.size(); i++){
            pair<int, int> cur = make_pair(values[i], labels[i]);
            pq.push(cur);
        }
        unordered_map<int, int> cnt;
        int res = 0, num = 0;
        while(num < num_wanted && pq.size() > 0){

            auto cur = pq.top();
            pq.pop();
            if(cnt.count(cur.second) < 1 || cnt[cur.second] < use_limit){
                num++;
                res += cur.first;
                cnt[cur.second]++;
            }
        }
        return res;
    }
};
