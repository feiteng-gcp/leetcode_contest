class Solution {
public:
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int left, int lim) {
        vector<pair<int, int>> v;
        for (int i = 0; i < values.size(); ++i)
            v.push_back({values[i], labels[i]});
        sort(v.rbegin(), v.rend());
        
        int ret = 0;
        int f[20002] = {0};

        for (int i = 0; i < v.size(); ++i) {
            if (left == 0) break;
            if (f[v[i].second] < lim) {
                f[v[i].second]++;
                --left;
                ret += v[i].first;
            }
        }
        return ret;
    }
};
