class Solution {
public:
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
        priority_queue<pair<int, int>> q;
        int n = values.size();
        vector<int> cnt(20001, 0);
        for (int i = 0; i < n; ++i) q.push({values[i], labels[i]});
        int num_meeted = 0, res = 0;
        while (num_meeted < num_wanted && !q.empty()) {
            auto cur = q.top();
            q.pop();
            if (cnt[cur.second] < use_limit) {
                cnt[cur.second]++;
                res += cur.first;
                num_meeted++;
            }
        }
        return res;
    }
};
