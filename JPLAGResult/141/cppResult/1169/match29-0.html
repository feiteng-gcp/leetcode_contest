class Solution {
public:
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num, int limit) {
        vector<pair<int,int>> tmp;
        int n = values.size();
        for (int i = 0; i < n; i++) {
            tmp.emplace_back(values[i], labels[i]);
        }
        sort(tmp.begin(), tmp.end());
        reverse(tmp.begin(), tmp.end());
        int ret = 0;
        map<int, int> cnt;
        for (int i = 0; i < tmp.size() && num > 0; i++) {
            if (cnt[tmp[i].second] < limit) {
                ret += tmp[i].first;
                --num;
                cnt[tmp[i].second]++;
            }
        }
        return ret;
    }
};
