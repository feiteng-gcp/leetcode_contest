class Solution {
public:
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
        vector<pair<int, int>> vp(values.size());
        for(int i  = 0;i<values.size();i++){
            vp[i].first = values[i];
            vp[i].second = labels[i];
        }
        map<int,int> m;
        sort(vp.begin(), vp.end(), greater<>());
        int ans = 0;
        int cnt = 0;
        for(int i = 0;i<vp.size() && cnt<num_wanted;i++){
            if(m[vp[i].second]<use_limit){
                m[vp[i].second]++;
                ans+=vp[i].first;
                cnt++;
            }
        }
        return ans;
    }
};
