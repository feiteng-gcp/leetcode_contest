class Solution {
public:
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
        vector< pair<int,int> > a;
        
        for ( int i = 0 ; i < values.size() ; ++i ) a.push_back( pair<int,int>(values[i], labels[i]) );
        sort(a.begin(), a.end());
        
        int used[200001] = {0};
        int ans = 0 ;
        
        for ( int i = a.size()-1 ; i >= 0 && num_wanted ; --i ) {
            
            if ( used[ a[i].second ] < use_limit ) {
                
                num_wanted --;
                used[ a[i].second ]++;
                ans += a[i].first;
            }
            
        }
        return ans;
    }
};
