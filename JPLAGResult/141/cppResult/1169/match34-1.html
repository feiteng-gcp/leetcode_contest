class Solution {
public:
	int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
		vector<pair<int, int>>all;
		map<int, set<int>>lb;
		map<int, int>used;
		for (int i = 0; i < values.size(); i++) {
			all.push_back({ values[i],labels[i]});
			lb[labels[i]].insert(i);
		}
		sort(all.begin(), all.end());
		reverse(all.begin(), all.end());
		int ct = 0;
		int ret = 0;

		for (int i = 0; i < all.size(); i++) {
			auto it = all[i];
			if(ct<num_wanted && used[it.second] < use_limit) {
				used[it.second]++;
				ret += it.first;
				ct++;
			}
		}
		return ret;
	}
};
