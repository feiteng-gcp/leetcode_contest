class Solution {
public:
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
        
        
        vector<pair<int,int>> VL;
        for(int i=0;i<values.size();i++) {
            VL.push_back({values[i],labels[i]});
        }
        
        sort(VL.begin(),VL.end(),greater<pair<int,int>>());
        map<int,int> counts;
        int total_counts=0;
        int ret = 0;
        for(int i=0;i<VL.size();i++) {
            if (total_counts >= num_wanted)
                break;
            int l = VL[i].second;
            if (counts[l] >= use_limit)
                continue;
            ret += VL[i].first;
            counts[l]++;
            total_counts++;
        }
        return ret;
        
        
    }
};
