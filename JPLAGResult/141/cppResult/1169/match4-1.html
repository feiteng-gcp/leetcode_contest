class Solution {
public:
    struct node{
        int val,lab;
        bool operator <(const node &tmp)const{
            return val>tmp.val;
        }
    }A[20020];
    int used[20020];
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
        int n = values.size();
        for(int i=0;i<n;i++){
            A[i].val = values[i];
            A[i].lab = labels[i];
        }
        sort(A,A+n);
        int ans = 0,cnt = 0;
        for(int i=0;i<n&&cnt<num_wanted;i++){
            if(used[A[i].lab]>=use_limit)continue;
            used[A[i].lab]++;
            ans+=A[i].val;
            cnt++;
        }
        return ans;
    }
};
