class Solution {
public:
	int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
		map<int, int> lc;
		priority_queue<pair<int, int>> pq;

		for (int i = 0; i < values.size();i++) {
			pq.emplace(values[i], labels[i]);
		}
		int ret = 0, num = 0;
		while (!pq.empty() && num < num_wanted) {
			auto p = pq.top();
			pq.pop();
			auto it = lc.find(p.second);
			if (lc[p.second] < use_limit) {
				++lc[p.second];
				ret += p.first;
				num++;
			}
		}

		return ret;
	}
};
