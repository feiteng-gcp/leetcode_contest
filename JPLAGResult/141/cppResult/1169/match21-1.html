class Solution {
public:
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
        priority_queue<pair<int, int>> p;
        map<int, int> use;
        for (size_t i = 0; i < values.size(); ++i) {
            p.push(make_pair(values[i], labels[i]));
        }

        size_t tot = 0;
        while (num_wanted != 0 && !p.empty()) {
            auto u = p.top();
            if (use[u.second] != use_limit) {
                num_wanted -= 1;
                tot += u.first;
                use[u.second] += 1;
            }
            p.pop();
        }
        return tot;
    }
};
