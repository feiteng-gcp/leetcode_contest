typedef pair<int, int> ii;

class Solution {
public:
    int largestValsFromLabels(vector<int>& values, vector<int>& labels, int num_wanted, int use_limit) {
        int n = values.size();
        vector<ii> a(n);
        for (int i = 0; i < n; ++i) {
            a[i] = {values[i], labels[i]};
        }
        sort(a.begin(), a.end(), greater<ii>());

        int ret = 0, tot = 0;
        map<int, int> cnt;
        for (int i = 0; i < n && tot < num_wanted; ++i) {
            if (cnt[a[i].second] < use_limit) {
                cnt[a[i].second]++;
                ret += a[i].first;
                ++tot;
            }
        }
        return ret;
    }
};
