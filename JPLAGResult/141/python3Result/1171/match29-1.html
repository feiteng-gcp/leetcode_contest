class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        if grid[0][0] != 0:
            return -1
        
        N, M = len(grid), len(grid[0])
        p = [[0] * (M) + [-1] for _ in range(N)] + [[-1] * (M+1)]
        p[0][0] = 1
        
        q = [[0, 0]]
        while q:
            nq = []
            for i, j in q:
                for di in range(-1, 2, 1):
                    for dj in range(-1, 2, 1):
                        if di == 0 and dj == 0: continue
                        ni, nj 
= i + di, j + dj
                        if p[ni][nj] != 0: continue
                        if grid[ni][nj] != 0: continue
                        p[ni][nj] = p[i][j] + 1
                        nq.append([ni, nj])
            q = nq
        return -1 if p[N-1][M-1] == 0 else p[N-1][M-1]
