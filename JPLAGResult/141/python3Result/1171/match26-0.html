class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        if grid[0][0] == 1 or grid[-1][-1] == 1:return -1
        n = len(grid)
        bfs = [(0,0)]
        dis = {(0,0):1}
        for r,c in bfs:
            for dr,dc in ((0,1),(0,-1),(1,0),(1,-1),(1,1),(-1,0),(-1,1),(-1,-1)):
                nr = r + dr
                nc = c + dc
                if nr < 0 or nr == n or nc < 0 or nc == n: continue
                if grid[nr][nc] == 1 or (nr,nc) in dis: continue
                dis[(nr,nc)] = dis[(r,c)] + 1
                if nr == nc == n-1: return dis[(nr,nc)]
                bfs.append((nr,nc))
        return -1
