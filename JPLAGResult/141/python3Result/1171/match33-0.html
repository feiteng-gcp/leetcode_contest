class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        if len(grid) == 0 or len(grid[0]) == 0:
            return -1
        m, n = len(grid), len(grid[0])
        if grid[0][0] == 1 or grid[m-1][n-1] == 1:
            return -1
        best = dict()
        to_visit = {(0,0)}
        depth = 1
        while to_visit:
            for i in to_visit:
                if i not in best:
                    best[i] = depth
            next_visit = set()
            for i,j in to_visit:
                for ii in range(-1,2):
                    for jj in range(-1, 2):
                        ni, nj = i+ii, j+jj
                        if 0<=ni<m and 0<=nj<n and grid[ni][nj]==0 and (ni,nj) not in best:
                            next_visit.add((ni,nj))
            depth += 1
            to_visit = next_visit
        return best.get((m-1,n-1),-1)
