class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        n = len(grid)
        if grid[0][0] or grid[n - 1][n - 1]:
            return -1
        q = [(0, 0)]
        ans = 1
        while q:
            ans += 1
            t = []
            while q:
                x, y = q.pop()
                for i, j in [(x-1, y-1), (x-1, y), (x-1, y+1), (x, y-1), (x, y+1), (x+1, y-1), (x+1, y), (x+1, y+1)]:
                    if 0 <= i < n and 0 <= j < n and grid[i][j] == 0:
                        if i == n - 1 and j == n - 1:
                            return ans
                        t.append((i, j))
                        grid[i][j] = 1
            q = t
        return -1
