class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        N=len(grid)
        if grid[0][0]!=0 or grid[-1][-1]!=0:
            return -1
        q=collections.deque([(0,0)])
        visited=set([(0,0)])
        dic={(0,0):1}
        while q:
            i,j=q.popleft()
            for k,l in [(i+1,j+1),(i,j+1),(i-1,j+1),(i+1,j),(i-1,j),(i+1,j-1),(i,j-1),(i-1,j-1)]:
                if 0<=k<=N-1 and 0<=l<=N-1:
                    if grid[k][l]==0:
                        if (k,l) not in dic:
                            dic[(k,l)]=dic[(i,j)]+1
                            q.append((k,l))
        if (N-1,N-1) in dic:
            return dic[(N-1,N-1)]
        return -1
