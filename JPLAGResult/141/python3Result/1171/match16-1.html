class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        a = len(grid)
        b = len(grid[0])
        
if grid[0][0] != 0 or grid[-1][-1] != 0:
            return -1
        if a == b == 1:
            return 1
        front = {(0, 0)}
        covered = {(0, 0)}
        steps = 0
        while front:
            steps += 1
            newfront = set()
            for (i, j) in front:
                for (i2, j2) in [
(i, j-1), (i, j+1), (i+1, j), (i-1, j),
                                 (i-1, j-1), (i-1, j+1), (i+1, j-1), (i+1, j+1)]:
                    if 0 <= i2 < a and 0 <= j2 < b and grid[i2][j2] == 0:
                        if (i2, j2) not in covered:
                            newfront.add ((i2, j2))
                            covered.add ((i2, j2))
                            if (i2, j2) == (a-1, b-1):
                                return 1 + steps
            front = newfront
        return -1
