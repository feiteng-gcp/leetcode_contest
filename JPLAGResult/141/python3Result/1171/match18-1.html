class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        l = len(grid)
        if grid[0][0] != 0 or grid[l-1][l-1] != 0: return -1
        if l == 1: return 1
        q = collections.deque([(0, 0, 1)])
        seen = set((0, 0))
        while q:
            x, y, i = q.popleft()
            
for x0, y0 in [(x+1, y+1), (x+1, y), (x, y+1), (x-1, y+1), (x+1, y-1), (x, y-1), (x-1, y), (x-1, y-1)]:
                if (x0, y0) not in seen and 0 <= x0 < l and 0 <= y0 < l and grid[x0][y0] == 0:
                    if x0 == l-1 and y0 == l-1:
                        return i+1
                    q.append((x0, y0, i+1))
                    seen.add((x0, y0))
        return -1
