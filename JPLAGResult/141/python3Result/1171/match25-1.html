class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        seen=set()
        h,w=len(grid),len(grid[0])
        if grid[0][0]==1 or grid[h-1][w-1]==1:return -1
        q=[(0,0,1)]
        seen.add((0,0))
        while q:
            nxt=[]
            for y,x, cnt in q:
                for dy in [-1,0,1]:
                    for dx in [-1,0,1]:
                        if dy==0 and dx==0:continue
                        ny,nx
=y+dy,x+dx
                        if 0<=ny<h and 0<=nx<w and grid[ny][nx]==0 and (ny, nx) not in seen:
                            #print("y:%d,x:%d,ny:%d,nx:%d"%(y,x,ny,nx))
                            seen.add((ny,nx))
                            nxt.append((ny,nx,cnt+1))
                            if ny==h-1 and nx==w-1: return cnt+1
            q=nxt
            #print("q:%s"%(q))
        return -1
