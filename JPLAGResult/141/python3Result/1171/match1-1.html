import collections

class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        N = len(grid)
        if grid[0][0] == 1 or grid[-1][-1] == 1:
            return -1
        
        stack = collections.deque([(0,0,1)])
        visited = set([(0,0)])
        while stack:
            i,j,d = stack.popleft()
            if i == N-1 and j == N-1:
                return d
            for x, y in [(1,0), (1,1), (0,1), (-1,1), (-1, 0), (1,-1), (-1,-1), (0,-1)]:
                ni = i + x
                nj = j + y
                if ni >= 0 and ni <= N-1 and nj >= 0 and nj <= N-1 and grid[ni][nj] != 1 and (ni,nj) not in visited:
                    stack
.append((ni, nj, d+1))
                    visited.add((ni,nj))
        return -1
