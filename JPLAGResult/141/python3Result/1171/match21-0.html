import heapq
from math import inf
class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        n = len(grid)
        if grid[0][0] != 0 or grid[-1][-1] != 0:
            return -1
        # pq = [(0, 0, 0)]
        neighbors = lambda x, y: [(x+1, y), (x-1, y), (x, y+1), (x, y-1), (x+1, y+1), (x-1, y+1), (x-1, y-1), (x+1, y-1)]
        # dist = [[inf for _ in range(n)] for _ in range(n)]
        # dist[0][0] = 0
        # while len(pq) > 0:
        #     d, x, y = pq
        #     if d[x][y] < d:
        #         continue
        #     for n in neighbors(x, y):
        #         a, b = n
        #         d2 = d + 1
        #         if grid[a][b] == 0 and d2 < dist[][]
        q = [(0, 0, 1)]
        visited = set()
        while len(q) > 0:
            x, y, d = q.pop(0)
            nes = [ne for ne in neighbors(x, y) if n > ne[0] >= 0 and n > ne[1] >= 0]
            for ne in nes:
                a, b = ne
                if a == b == n-1:
                    return d + 1
                if ne not in visited and grid[ne[0]][ne[1]] == 0:
                    visited.add(ne)
                    q.append((ne[0], ne[1], d + 1))
        return -1
