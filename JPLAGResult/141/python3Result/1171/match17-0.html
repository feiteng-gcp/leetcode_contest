class Solution:
    def shortestPathBinaryMatrix(self, grid: List[List[int]]) -> int:
        # DP
        N = len(grid)
        dp = [[sys.maxsize] * N for _ in range(N)]
        if grid[0][0] != 0 or grid[N  -1][N - 1]:
            return -1
        # BFS
        pathlen = 2
        seen = set()
        seen.add((0, 0))
        deq = collections.deque([[0, 0]])
        while True:
            nxt = collections.deque()
            while deq:
                x, y = deq.popleft()
                for i, j in [(x - 1, y - 1), (x - 1, y), (x - 1, y + 1), (x, y - 1), (x, y + 1), (x + 1, y - 1), (x + 1, y), (x + 1, y + 1)]:
                    if 0 <= i < N and 0 <= j < N and grid[i][j] == 0 and (i, j) not in seen:
                        seen.add((i, j))
                        if i == N - 1 and j == N - 1:
                            return pathlen
                        nxt.append([i, j])
            if not nxt:
                break
            deq = nxt
            pathlen += 1
        return -1
