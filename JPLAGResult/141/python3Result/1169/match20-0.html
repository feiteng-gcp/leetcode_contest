class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        zl = list(zip(values, labels))
        zl.sort(reverse=True)
        import collections
        d = collections.defaultdict(int)
        n = len(zl)
        ret = 0
        used = 0
        for i in range(n):
            l = zl[i][1]
            val = zl[i][0]
            if d[l] < use_limit:
                d[l] += 1
                ret += val
                used += 1
            if used == num_wanted:
                break
        return ret
