from collections import defaultdict

class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        
        used = defaultdict(int)
        
        L = len(values)
        
        items = [(labels[i], values[i]) for i in range(len(labels))]
        
        items.sort(key=lambda x: x[1], reverse=True)
        
        #print(items)
        
        cur_num = 0
        i = 0
        res = 0
        while cur_num < num_wanted and i < L:
            cur_label, cur_val = items[i][0], items[i][1]
            if used[cur_label] < use_limit:
                res += cur_val
                #print(res)
                cur_num += 1
                used[cur_label] += 1
            i += 1
                
        return res
