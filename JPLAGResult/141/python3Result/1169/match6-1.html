class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        cnt = collections.Counter()
        pool = sorted(zip(values, labels), reverse
=True)
        ans = 0
        num = 0
        for v, l in pool:
            # print(v, l)
            if cnt[l] == use_limit:
                continue
            ans += v
            cnt[l] += 1
            num += 1
            # print(ans, num, cnt)
            if num == num_wanted:
                break
        return ans
