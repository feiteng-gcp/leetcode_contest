class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        s = [[values[i], labels[i]] for i in range(len(values))]
        # print(s)
        s.sort(key = lambda x: -x[0])
        cnt = collections.defaultdict(int)
        res = 0
        idx = 0
        for v, l in s:
            # print(v, l)
            if cnt[l] < use_limit:
                res += v
                cnt[l] += 1
                idx += 1
                if idx == num_wanted:
                    break
            else:
                continue
        return res
