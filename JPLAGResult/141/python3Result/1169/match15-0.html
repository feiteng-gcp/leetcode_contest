import collections
class Solution(object):
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        arr = sorted(zip(values, labels), reverse=True)
        limit = collections.defaultdict(int)
        out = 0
        num_used = 0
        for val, label in arr:
            if limit[label] < use_limit:
                out = out + val
                limit[label] += 1
                num_used = num_used + 1
                if num_used == num_wanted:
                    break
        return out
