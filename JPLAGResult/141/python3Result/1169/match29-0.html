class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        r = sorted(zip(values, labels), key=lambda x: x[0], reverse=True)
        dic = {}
        num = 0
        sum = 0
        max_v = 0
        for c in r:
            if num >= num_wanted:
                break
            if not c[1] in dic.keys():
                dic[c[1]] = 1
                num += 1
                sum += c[0]
            elif dic[c[1]] < use_limit:
                dic[c[1]] += 1
                num += 1
                sum += c[0]
            max_v = max(max_v, sum)
        return max_v
