class Solution(object):
    def largestValsFromLabels(self, values, labels, num_wanted, use_limit):
        """
        :type values: List[int]
        :type labels: List[int]
        :type num_wanted: int
        :type use_limit: int
        :rtype: int
        """
        nums = []
        for i in range(len(values)):
            nums.append([values[i],labels[i]])
#         nums = [values,labels]
        # print(nums)
        nums.sort(key=lambda x:x[0],reverse=True)
        # print(nums)
        values = [nums[i][0] for i in range(len(nums))]
        labels = [nums[i][1] for i in range(len(nums))]
#         for i in range(len(values)):
#             for j in range(len(values)-1,i,-1):
#                 if values[j]>values[j-1]:
# #                     print(j)
#                     values[j],values[j-1] = values[j-1],values[j]
#                     labels[j],labels[j-1] = labels[j-1],labels[j]
# #                     print(values)
#         print(values)
        maxv = 0
        k = 0
        i = 0
        d = {}
        while k<num_wanted and i<len(values):
            if labels[i] in d and d[labels[i]]>=use_limit:
                i += 1
            else:
                d[labels[i]] = d.get(labels[i],0)+1
                maxv += values[i]
                k += 1
                i += 1
        return maxv
