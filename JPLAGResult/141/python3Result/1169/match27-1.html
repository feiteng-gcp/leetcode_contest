from collections import defaultdict
class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        #print(values, labels, num_wanted, use_limit)
        #groups = [list() for _ in range(20001)]
        groups = defaultdict(list)
        for i, l in enumerate(labels): 
            groups[l].append(values[i])
        
        candi = list()
        for g in groups.values(): 
            if len(g) <= use_limit: 
                candi.extend(g)
            else: 
                candi.extend(sorted(g, reverse = True)[:use_limit])
        candi = sorted(candi, reverse = True)
        #print(candi)
        ans = 0
        for i in range(min(len(candi), num_wanted)): 
            ans += candi[i]
        
        return ans
