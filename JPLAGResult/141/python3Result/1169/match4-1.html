class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
       
        vl=sorted(list(zip(values, labels)), reverse=True)
        d=collections.defaultdict(int)
        ans=0
        cnt=0
        for v, l in vl:
            if d[l]<use_limit:
                ans+=v
                d[l]+=1
                cnt+=1
                if cnt>=num_wanted:break
        return ans
