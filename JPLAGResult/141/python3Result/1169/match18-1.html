class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        n = len(values)
        temp = [i for i in range(n)]
        temp.sort(key = lambda x:-values[x])
        ans = 0
        used_num = 0
        hashmap = collections.defaultdict(int)
        for i in temp:
            value = values[i]
            label = labels[i]
            
if hashmap[label] >= use_limit:
                continue
            ans += value
            hashmap[label] += 1
            used_num += 1
            if used_num >= num_wanted:
                break
        return ans
