class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        l = sorted(zip(values, labels), key=lambda x: x[0], reverse=True)
        d = dict()
        s = 0
        n = 0
        while n < num_wanted:
            li = l.pop(0)
            #print(li, d)
            if not li[1] in d:
                d[li[1]] = 1
                s += li[0]
                n += 1
            elif d[li[1]] < use_limit:
                d[li[1]] += 1
                s += li[0]
                n += 1
            if not l:
                break
        return s
