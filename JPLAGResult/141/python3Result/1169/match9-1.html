class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        items = [(v, l) for v, l in zip(values, labels)]
        items.sort(reverse=True)
        result = 0
        used = dict()
        count = 0
        for v, l in items:
            if l not in used or used[l] < use_limit:
                if l not in used:
                    used[l] = 1
                else:
                    used[l] += 1
                result += v
                count += 1
                if count == num_wanted:
                    break
        return result
