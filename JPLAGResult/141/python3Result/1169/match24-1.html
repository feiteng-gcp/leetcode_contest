class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        n = len(values)
        idxs = list(range(n))
        idxs.sort(key=lambda i:values[i], reverse
=True)
        cnt = collections.defaultdict(int)
        res = 0
        has = 0
        for idx in idxs:
            if has == num_wanted:
                break
            if cnt[labels[idx]] == use_limit:
                continue
            res += values[idx]
            has += 1
            cnt[labels[idx]] += 1
        return res
