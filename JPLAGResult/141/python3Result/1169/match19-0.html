class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        # greedy
        n = len(values)
        z = [[values[i], labels[i]] for i in range(n)]
        z.sort(key = lambda v: v[0], reverse = True)
        res = 0
        ctr = collections.defaultdict(int)
        for pair in z:
            if ctr[pair[1]] < use_limit:
                ctr[pair[1]] += 1
                res += pair[0]
            else:
                continue
            num_wanted -= 1
            if num_wanted == 0:
                break
        return res
