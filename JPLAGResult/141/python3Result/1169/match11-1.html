class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        L=[]
        for i in range(len(labels)):
            L.append((values[i],labels[i]))
        L.sort(key=lambda x:-x[0])
        s=0
        dic={}
        ct=0
        for v,l in L:
            if num_wanted==ct:
                break
            if l not in dic:
                dic[l]=1
                s+=v
                ct+=1
            else:
                if dic[l]<use_limit:
                    dic[l]+=1
                    s+=v
                    ct+=1
        return s
