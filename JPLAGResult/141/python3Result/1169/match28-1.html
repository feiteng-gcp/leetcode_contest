class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        lv = collections.defaultdict(list)
        for v, l in zip(values, labels):
            lv[l].append(v)
        
        cans = []
        for l, vs in lv.items():
            vs.sort(reverse=True)
            for i in range(min(use_limit, len(vs))):
                cans.append(vs[i])
            
        cans.sort
(reverse=True)
        ans = 0
        for i in range(min(num_wanted, len(cans))):
            ans += cans[i]
        return ans
