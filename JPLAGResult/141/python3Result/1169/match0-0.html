from collections import defaultdict
class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        data = defaultdict(list)
        n = len(values)
        for i in range(n):
            data[labels[i]].append(values[i])
        sorted(data.items(), key=lambda a : a[0])

        pred = list()
        for it in data.items():
            if len(it[1]) > use_limit:
                pred += sorted(it[1], reverse=True)[:use_limit]
            else:
                pred += sorted(it[1], reverse=True)
        print(pred)
        return sum(sorted(pred, reverse=True)[:num_wanted])
