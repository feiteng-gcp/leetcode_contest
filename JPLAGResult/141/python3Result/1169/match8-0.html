class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        A = [(p, q) for p, q in zip(values, labels)]
        A.sort(reverse=1)
        C = [0]*20001
        ans = 0; cnt = 0
        for p, q in A:
            if C[q] == use_limit:
                continue
            ans += p
            C[q] += 1
            cnt += 1
            if cnt == num_wanted:
                break
        return ans
