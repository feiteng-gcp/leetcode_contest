from collections import defaultdict
class Solution:
    def largestValsFromLabels(self, values: List[int], labels: List[int], num_wanted: int, use_limit: int) -> int:
        
        sorted_l = sorted(zip(values, labels), key = lambda x: x[0], reverse = True)
        
        # print(sorted_l)
        
        d = defaultdict(int)
        res = 0
        for i, comb in enumerate(sorted_l):
            
            v = comb[0]
            label = comb
[1]
            
            if d[label] < use_limit:
                d[label] += 1
                num_wanted -= 1
                res += v
                
            if num_wanted == 0:
                break
        return res
