class Solution:
    def duplicateZeros(self, arr: List[int]) -> None:
        """
        Do not return anything, modify arr in-place instead.
        """

        n = len(arr)
        cnt = 0
        idx = -1
        for i, a in enumerate(arr):
            if a:
               cnt += 1
            else:
                cnt += 2
            if cnt >= n:
                idx = i
                break
        i = idx
        j = n - 1
        if cnt == n+1:
            arr[-1] = 0
            i -= 1
            j -= 1
        while i >= 0:
            if arr[i]:
                arr[j] = arr[i]
                j -= 1
            else:
                arr[j] = 0
                arr[j-1] = 0
                j -= 2
            i -= 1
