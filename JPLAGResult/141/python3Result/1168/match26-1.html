class Solution:
    def duplicateZeros(self, arr: List
[int]) -> None:
        """
        Do not return anything, modify arr in-place instead.
        """
        extra = []
        for elem in arr:
            if elem == 0:
                extra.append(elem)
                extra.append(elem)
            else:
                extra.append(elem)
        for i in range(len(arr)):
            arr[i] = extra[i]
        return None
