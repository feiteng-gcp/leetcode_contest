class Solution:
    def duplicateZeros(self, arr: List[int]) -> None:
        """
        Do not return anything, modify arr in-place instead.
        """
        L = len(arr)
        zeros = 0
        for i, v in enumerate(arr): 
            if not v: 
                zeros += 1
            
            if i + zeros >= L - 1:
                break
        
        if i + zeros == L:
            arr[L - 1] = v
            i 
-= 1
            j = L - 2
        else: 
            j = L - 1
        
        while j >= 0:
            if arr[i]: 
                arr[j] = arr[i]
                j -= 1
            else: 
                arr[j] = arr[j - 1] = 0
                j -= 2
            i -= 1
