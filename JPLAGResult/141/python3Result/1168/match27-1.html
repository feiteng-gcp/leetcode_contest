class Solution:
    def duplicateZeros(self, arr: List[int]) -> None:
        """
        Do not return anything, modify arr in-place instead.
        """
        n = len(arr)
        ret = []
        for i in arr:
            if i != 0:
                ret.append(i)
            else:
                ret.append(0)
                ret.append(0)
            if len(ret) >= n:
                break
        for i in range(n):
            arr[i] = ret[i]
