class Solution:
    def duplicateZeros(self, arr: List[int]) -> None:
        """
        Do not return anything, modify arr in-place instead.
        """
        a = []
        for x in arr:
            a.append(x)
            if x == 0:
                a
.append(x)
                
        for i in range(len(arr)):
            arr[i] = a[i]
