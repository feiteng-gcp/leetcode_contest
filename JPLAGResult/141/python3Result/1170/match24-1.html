            dp = []
            sta = []  # construct the LCS
            for i in range(len(b) + 1):  # dp len(b)+1 rows len(a)+1 cols
                dp.append([0] * (len(a) + 1))  # initialize dp
                sta.append([0] * (len(
a) + 1))  # initialize b
            for i in range(1, len(b)+1):  # each row i
                # check element b[i-1]
                for j in range(1, len(a)+1):  # each col j
                    if a[j-1] == b[i-1]:
                        dp[i][j] = dp[i-1][j-1] + 1
                        sta[i][j] = 1
                    elif dp
[i-1][j] >= dp[i][j-1]:
                        dp[i][j] = dp[i-1][j]
                        sta[i][j] = 2
                    else:
                        dp[i][j] = dp[i][j-1]
                        sta[i][j] = 3
            def LCS(i, j):
                if i == 0 or j == 0:
                    return ""
                if sta[i][j] == 1:
                    return b[i-1] + LCS(i-1, j-1)
                elif sta[i][j] == 2:
                    return LCS(i-1, j)
                else:
                    return LCS(i, j-1)
            LCS_ab = LCS(len(b), len(a))
            return dp[len(b)][len(a)], LCS_ab[::-1]
