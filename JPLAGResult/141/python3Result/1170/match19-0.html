class Solution:
    def shortestCommonSupersequence(self, str1: str, str2: str) -> str:
        n1 = len(str1)
        n2 = len(str2)
        if n1 < n2:
            n1,n2=n2,n1
            str1,str2=str2,str1
        
        dp = [[0 for _ in range(n2+1)] for _ in range(n1+1)]
        letters = [[None for _ in range(n2+1)] for _ in range(n1+1)]
        prev = [[None for _ in range(n2+1)] for _ in range(n1+1)]
        for i in range(1, n1+1):
            for j in range(1, n2+1):
                l1 = str1[i-1]
                l2 = str2[j-1]
                if l1 == l2:
                    dp[i][j] = 1 + dp[i-1][j-1]
                    letters[i][j] = l1
                    prev[i][j] = [i-1,j-1]
                else:
                    if dp[i-1][j] > dp[i][j-1]:
                        dp[i][j] = dp[i-1][j]
                        prev[i][j] = [i-1,j]
                    else:
                        dp[i][j] = dp[i][j-1]
                        prev[i][j] = [i,j-1]
        i = n1
        j = n2
        lcs = []
        while i > 0 and j > 0:
            if letters[i][j]:
                lcs.append(letters[i][j])
            i, j = prev[i][j]
            
        lcs = lcs[::-1]
        # print(lcs)
        i = 0
        j = 0
        k = 0
        ret = []
        while i < n1 and j < n2 and k < len(lcs):
            l1 = str1[i]
            l2 = str2[j]
            l9 = lcs[k]
            
            if l1 == l2 and l2 == l9:
                i += 1
                j += 1
                k += 1
                ret.append(l1)
            elif l1 == l9:
                ret.append(l2)
                j += 1
            elif l2 == l9:
                ret.append(l1)
                i += 1
            else:
                # Both dont match
                ret.append(l1)
                i += 1
        
        while i < n1:
            ret.append(str1[i])
            i += 1
        while j < n2:
            ret.append(str2[j])
            j += 1
        return ''.join(ret)
