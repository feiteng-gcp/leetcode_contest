class Solution:
    def shortestCommonSupersequence(self, str1: str, str2: str) -> str:
        n=len(str1)
        m=len(str2)
        L=[[0]*m for _ in range(n)]
        trigger=False
        for i in range(n):
            if trigger:
                L[i][0]=str2[0]
            elif str1[i]==str2[0]:
                trigger=True
                L[i][0]=str2[0]
            else:
                L[i][0]=""
        trigger=False    
        for j in range(m):
            if trigger:
                L[0][j]=str1[0]
            elif str1[0]==str2[j]:
                trigger=True
                L[0][j]=str1[0]
            else:
                L[0][j]=""
                
        for i in range(1,n):
            for j in range(1,m):
                if str1[i]==str2[j]:
                    L[i][j]=L[i-1][j-1]+str1[i]
                else:
                    if len(L[i-1][j])>len(L[i][j-1]):
                        L[i][j]=L[i-1][j]
                    else:
                        L[i][j]=L[i][j-1]
        s=L[-1][-1]
        print(L)
        i=0
        j=0
        ct=0
        res=""
        while(i<n or j<m):
            if i<n and j<m:
                if ct<len(s) and str1[i]==str2[j]==s[ct]:
                    res+=str1[i]
                    i+=1
                    j+=1
                    ct+=1
                else:
                    if ct==len(s) or str1[i]!=s[ct]:
                        res+=str1[i]
                        i+=1
                    if ct==len(s) or str2[j]!=s[ct]:
                        res+=str2[j]
                        j+=1
            else:
                if i<n:
                    res+=str1[i]
                    i+=1
                if j<m:
                    res+=str2[j]
                    j+=1
        return res
