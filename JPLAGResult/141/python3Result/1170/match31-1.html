class Solution:
    def shortestCommonSupersequence(self, a: str, b: str) -> str:
        m = len(a) 
        n = len(b)
        dp = [[0] * (n + 2) for _ in range(m + 2)]
        for i in range(m + 1): 
            for j in range(n + 1): 
                if not i: dp[i][j] = j
                elif not j: dp
[i][j] = i
                elif a[i - 1] == b[j - 1]: 
                    dp[i][j] = 1 + dp[i - 1][j - 1]
                else: 
                    dp[i][j] = 1 + min(dp[i - 1][j], dp[i][j - 1])
        
        #index = dp[m][n]
        ret = ''
        i, j = m, n
        while i and j: 
            if a[i - 1] == b[j - 1]: 
                ret += a[i - 1]
                i -= 1
                j -= 1
                #index -= 1
            elif dp[i - 1][j] < dp[i][j - 1]: 
                ret += a[i - 1]
                i -= 1
                #index -= 1
            else: 
                ret += b[j - 1]
                j -= 1
                #index -= 1
        if i: 
            ret += a[i - 1 :: -1]
        if j: 
            ret += b[j - 1 :: -1]
        return ret[::-1]
