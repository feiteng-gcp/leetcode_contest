class Solution:
    def shortestCommonSupersequence(self, s1, s2):
        f = [[0]*(len(s2)+1) for _ in range(len(s1)+1)]
        
        for i in range(len(s1)+1):
            for j in range(len
(s2)+1):
                if i == 0 or j == 0:
                    f[i][j] = 0
                elif s1[i-1] == s2[j-1]:
                    f[i][j] = f[i-1][j-1] + 1
                else:
                    f[i][j] = max(f[i-1][j], f[i][j-1])
        
        commonLen = f[-1][-1]
        common = [""]*(commonLen)
        p = commonLen-1
        i, j = len(s1), len(s2)
        
while i > 0 and j > 0:
            if s1[i-1] == s2[j-1]:
                common[p] = s1[i-1]
                p -= 1
                i -= 1
                j -= 1
            elif f[i-1][j] > f[i][j-1]:
                i -= 1
            else:
                j -= 1
        common = "".join(common)
        # print (common)
        ans = []
        p = 0
        i = 0
        j = 0
        while i < len(s1) or j < len(s2):
            # print (s1[i:], s2[j:], common[p:], "".join(ans))
            if p == len(common):
                if i < len(s1):
                    ans.append(s1[i])
                    i += 1
                    continue
                if j < len(s2):
                    ans.append(s2[j])
                    j += 1
                    continue
            if i < len(s1) and s1[i] == common[p] and j < len(s2) and s2[j] == common[p]:
                ans.append(common[p])
                i += 1
                j += 1
                p += 1
                continue
            if j < len(s2) and s2[j] == common[p] and i < len(s1) and s1[i] != common[p]:
                ans.append(s1[i])
                i += 1
                continue
            if j < len(s2) and s2[j] != common[p] and i < len(s1) and s1[i] == common[p]:
                ans.append(s2[j])
                j += 1
                continue
            if i < len(s1):
                if s1[i] == common[p]:
                    p +=1
                ans.append(s1[i])
                i += 1
                continue
            if j < len(s2):
                if s2[j] == common[p]:
                    p +=1
                ans.append(s2[j])
                j += 1
                continue
        return "".join(ans)
