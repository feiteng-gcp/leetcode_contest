class Solution:
    def shortestCommonSupersequence(self, str1: str, str2: str) -> str:
        s1, s2 = str1, str2
        l1, l2 
= len(s1), len(s2)
        dp = [[0] * (l2 + 1) for _ in range(l1 + 1)]
        for i in range(1, l1 + 1):
            for j in range(1, l2 + 1):
                if s1[i - 1] == s2[j - 1]:
                    dp[i][j] = max(dp[i][j - 1], dp[i - 1][j], dp
[i - 1][j - 1] + 1)
                else:
                    dp[i][j] = max(dp[i][j - 1], dp[i - 1][j])
        longest_common_sub = dp[l1][l2]
        
        def helper(x1, x2):
            if x1 == 0:
                return str2[:x2]
            if x2 == 0:
                return str1[:x1]
            if str1[x1 - 1] == str2[x2 - 1]:
                return helper(x1 - 1, x2 - 1) + str1[x1 - 1]
            else:
                if dp[x1 - 1][x2] > dp[x1][x2 - 1]:
                    return helper(x1 - 1, x2) + str1[x1 - 1]
                else:
                    return helper(x1, x2 - 1) + str2[x2 - 1]
        
        return helper(l1, l2)
