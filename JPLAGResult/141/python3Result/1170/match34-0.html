class Solution:
    def shortestCommonSupersequence(self, str1: str, str2: str) -> str:
        # find the longest common subsequence
        record = {}
        n = len(str1)
        m = len(str2)
        for i in range(n+1):
            for j in range(m+1):
                if i == 0 or j == 0:
                    record[(i,j)] = ''
                else:
                    if str1[i-1] == str2[j-1]:
                        record[(i,j)] = record[(i-1,j-1)] + str1[i-1]
                    else:
                        if len(record[(i-1,j)]) > len(record[(i,j-1)]):
                            record[(i,j)] = record[(i-1,j)]
                        else:
                            record[(i,j)] = record[(i,j-1)]
        common = record[(n,m)]
        ans = ''
        i = 0
        j = 0
        cur_m = 0
        for c in common:
            while i < len(str1) and str1[i] != c:
                ans += str1[i]
                i += 1
            while j < len(str2) and str2[j] != c:
                ans += str2[j]
                j += 1
            ans += c
            i += 1
            j += 1
        while i < len(str1):
            ans += str1[i]
            i += 1
        while j < len(str2):
            ans += str2[j]
            j += 1
        return ans
