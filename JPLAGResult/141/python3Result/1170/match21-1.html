class Solution:
    def shortestCommonSupersequence(self, str1: str, str2: str) -> str:
        dp = [[0 for _ in range(len(str2) + 1)] for _ in range(len(str1) + 1)]
        direc = [[(0, 0) for _ in range(len(str2) + 1)] for _ in range(len(str1) + 1)]
        
        for i in range(1, len(str1) + 1):
            for j in range(1, len(str2) + 1):
                dp[i][j] = max(dp[i][j - 1], dp[i - 1][j])
                if dp[i][j] == dp[i][j - 1]:
                    direc[i][j] = (0, -1)
                else:
                    direc[i][j] = (-1, 0)
                if str1[i - 1] == str2
[j - 1]:
                    if dp[i - 1][j - 1] + 1 > dp[i][j]:
                        dp[i][j] = dp[i - 1][j - 1] + 1
                        direc[i][j] = (-1, -1)
        
        x, y = len(str1), len(str2)
        i, j = len(str1) + 1, len(str2) + 1
        res 
= []
        while x != 0 and y != 0:
            dx, dy = direc[x][y]
            if dx == dy == -1:
                res.append(str1[x: i - 1])
                res.append(str2[y: j - 1])
                res.append(str1[x - 1])
                i, j = x, y
            x += dx
            y += dy
        res.append(str1[0: i - 1])
        res.append(str2[0: j - 1])
        res.reverse()
        
        return ''.join(res)
