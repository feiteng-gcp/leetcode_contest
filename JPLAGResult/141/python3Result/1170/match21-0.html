class Solution:
    def shortestCommonSupersequence(self, str1: str, str2: str) -> str:
        L1, L2 = len(str1), len(str2)
        INF = L1+L2+1
        dp = [[INF] * (L2 + 1) for _ in range(L1+1)]
        pre = [[0] * (L2 + 1) for _ in range(L1+1)]
        
        dp[0] = [j for j in range(L2+1)]
        pre[0] = [[0, j-1] for j in range(L2+1)]
        
        for i in range(1, L1+1):
            dp[i][0] = dp[i-1][0] + 1
            pre[i][0] = [i-1, 0]
            for j in range(1, L2+1):
                if str1[i-1] == str2[j-1]:
                    dp[i][j] = dp[i-1][j-1] + 1
                    pre[i][j] = [i-1, j-1]
                else:
                    if dp[i][j-1] < dp[i-1][j]:
                        dp[i][j] = dp[i][j-1] + 1
                        pre[i][j] = [i, j-1]
                    else:
                        dp[i][j] = dp[i-1][j] + 1
                        pre[i][j] = [i-1, j]
        
        ans = []
        i, j = L1, L2
        while i + j > 0:
            pi, pj = pre[i][j]
            if pi < i: ans.append(str1[i-1])
            else: ans.append(str2[j-1])
            i, j = pi, pj
        return ''.join(ans[::-1])
