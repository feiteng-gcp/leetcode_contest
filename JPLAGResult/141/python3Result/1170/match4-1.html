class Solution:
    def shortestCommonSupersequence(self, a: str, b: str) -> str:
        m, n = len(a), len(b)
        dp = [[0]*(n+1) for _ in range(m+1)]
        for i in range(1, m+1):
            for j in range(1, n+1):
                if a
[i-1] == b[j-1]:
                    dp[i][j] = 1 + dp[i-1][j-1]
                else:
                    dp[i][j] = max(dp[i-1][j], dp[i][j-1])
        length = dp[-1][-1]
        A = []
        B = []
        x, y = m, n
        while x > 0 and y > 0:
            if dp[x][y] == dp[x-1][y]:
                x -= 1
            elif dp[x][y] == dp[x][y-1]:
                y -= 1
            else:
                A.append(x-1)
                B.append(y-1)
                x, y = x-1, y-1
        ans = ''
        A.reverse()
        B.reverse()
        for i in range(len(A)):
            if i == 0:
                ans += a[:A[i]]+b[:B[i]]+a[A[i]]
            else:
                ans += a[A[i-1]+1:A[i]] + b[B[i-1]+1:B[i]] + a[A[i]]
        ans += a[A[-1]+1:] + b[B[-1]+1:]
        return ans
