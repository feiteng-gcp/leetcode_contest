class Solution:
    def shortestCommonSupersequence(self, str1: str, str2: str) -> str:
        n = len(str1)
        m = len(str2)
        dp = [[None for i in range(m+1)] for j in range(n+1)]
        s = [['' for i in range(m+1)] for j in range(n+1)]
        for i in range(n+1):
            for j in range(m+1):
                if i == 0 or j == 0:
                    dp[i][j] = 0
                    
                elif str1[i-1] == str2[j-1]:
                    dp[i][j] = dp[i-1][j-1] + 1
                    s[i][j] = s[i-1][j-1] + str1[i-1]
                else:
                    if dp[i-1][j] > dp[i][j-1]:
                        dp[i][j] = dp[i-1][j]
                        s[i][j] = s[i-1][j]
                    else:
                        dp[i][j] = dp[i][j-1]
                        s[i][j] = s[i][j-1]
                    
        print(dp[n][m], s[n][m])
        
        lcs = s[n][m]
        
        i = 0
        j = 0
        k = 0
        res = ''
        while k < len(lcs):
            while str1[i] != lcs[k]:
                res += str1[i]
                i += 1
            while str2[j] != lcs[k]:
                res += str2[j]
                j += 1
            res += lcs[k]
            k += 1
            i += 1
            j += 1
        res += str1[i:] + str2[j:]
        return res
