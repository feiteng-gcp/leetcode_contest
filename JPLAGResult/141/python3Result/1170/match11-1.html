class Solution:
    def shortestCommonSupersequence(self, str1: str, str2: str) -> str:
        dp = [[0] * 1001 for _ in range(1001)]
        pre = [[-1] * 1001 for _ in range(1001)]
        n, m = len(str1), len
(str2)
        for i in range(n+1):
            for j in range(m+1):
                if i == j == 0:
                    dp[0][0] = 0
                elif i == 0:
                    dp[i][j] = dp[i][j-1] + 1
                    pre[i][j] = 2
                elif j == 0:
                    dp[i][j] = dp[i-1][j] + 1
                    pre[i][j] = 1
                elif str1[i-1] == str2[j-1]:
                    if dp[i-1][j] <= dp[i][j-1] and dp
[i-1][j] <= dp[i-1][j-1]:
                        dp[i][j] = dp[i-1][j] + 1
                        pre[i][j] = 1
                    elif dp[i][j-1] <= dp[i-1][j] and dp[i][j-1] <= dp[i-1][j-1]:
                        dp[i][j] = dp[i][j-1] + 1
                        pre[i][j] = 2
                    else:
                        dp[i][j] = dp[i - 1][j - 1] + 1
                        pre[i][j] = 3
                else:
                    if (dp[i - 1][j] <= dp[i][j - 1]):
                        dp[i][j] = dp[i - 1][j] + 1
                        pre[i][j] = 1
                    else:
                        dp[i][j] = dp[i][j - 1] + 1
                        pre[i][j] = 2
        s, x, y = '', n, m
        while 1:
            if pre[x][y] == -1:
                break
            if pre[x][y] == 1:
                s += str1[x-1]
                x -= 1
            elif pre[x][y] == 2:
                s += str2[y-1]
                y -= 1
            else:
                s += str1[x-1]
                x -= 1
                y -= 1
        res = s[::-1]
        return res
