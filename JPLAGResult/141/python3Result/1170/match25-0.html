class Solution(object):
    def shortestCommonSupersequence(self, str1, str2):
        """
        :type str1: str
        :type str2: str
        :rtype: str
        """
        res = [[0 for j in range(len(str1)+1)] for i in range(len(str2)+1)]
        indexs = [[-1 for j in range(len(str1)+1)] for i in range(len(str2)+1)]
        for i in range(1,len(str2)+1):
            indexs[i][0] = 1
        for j in range(1,len(str1)+1):
            indexs[0][j] = 2
        for i in range(len(str2)):
            for j in range(len(str1)):
                if str2[i]==str1[j]:
                    res[i+1][j+1] = res[i][j]+1
                    indexs[i+1][j+1] = 0
                else:
                    if res[i][j+1]>res[i+1][j]:
                        res[i+1][j+1] = res[i][j+1]
                        indexs[i+1][j+1] = 1
                    else:
                        res[i+1][j+1] = res[i+1][j]
                        indexs[i+1][j+1] = 2
        # print(res)
        # print(indexs)
        i = len(str2)
        j = len(str1)
        res = ""
        while indexs[i][j]>=0:
            if indexs[i][j] == 0:
                res = str2[i-1]+res
                i -= 1
                j -= 1
            elif indexs[i][j] == 1:
                res = str2[i-1]+res
                i -= 1
            else:
                res = str1[j-1]+res
                j -= 1
        return res
