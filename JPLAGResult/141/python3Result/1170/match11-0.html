class Solution:
    def shortestCommonSupersequence(self, str1: str, str2: str) -> str:
        max_common = self.lcs(str1, str2)
        if not max_common:
            return str1 + str2
        i = j = k = 0
        res = ''
        while i < len(max_common) and j < len(str1) and k < len(str2):
            test = max_common[i]
            if str1[j] != test:
                res += str1[j]
                j += 1
                continue
            if str2[k] != test:
                res += str2[k]
                k += 1
                continue
            j += 1
            k += 1
            res += test
            i += 1
        res += self.shortestCommonSupersequence(str1[j:], str2[k:])
        return res

    def lcs(self, X, Y):
        m, n = len(X), len(Y)
        L = [[0] * (n + 2) for i in range(m + 2)]
        LL = [[0] * (n + 2) for i in range(m + 2)]
        for i in range(m + 1): 
            for j in range(n + 1): 
                if i == 0 or j == 0:
                    L[i][j] = 0
                    LL[i][j] = ''
                elif X[i - 1] == Y[j - 1]: 
                    L[i][j] = L[i - 1][j - 1] + 1
                    LL[i][j] = LL[i - 1][j - 1] + X[i - 1]
                else:
                    if L[i - 1][j] > L[i][j - 1]:
                        L[i][j] = L[i - 1][j]
                        LL[i][j] = LL[i - 1][j]
                    else:
                        L[i][j] = L[i][j - 1]
                        LL[i][j] = LL[i][j - 1]
        return LL[m][n]
