class Solution:
    def shortestCommonSupersequence(self, str1: str, str2: str) -> str:
        lcs = collections.defaultdict(int)
        def dp(i1, i2):
            if i1 < 0 or i2 < 0: return 0
            if (i1,i2) in lcs: return lcs[(i1,i2)]
            if str1[i1] == str2[i2]:
                lcs[(i1,i2)] = dp(i1-1, i2-1) + 1
            else:
                lcs[(i1,i2)] = max(dp(i1-1,i2), dp(i1, i2-1))
            return lcs[(i1,i2)]
        n1, n2 = len(str1) - 1, len(str2) - 1
        dp(n1,n2)
        res = ''
        while True:
            if str1[n1] == str2[n2]:
                res = str1[n1] + res
                n1 -= 1
                n2 -= 1
            elif lcs[(n1,n2)] == lcs[(n1-1,n2)]:
                res = str1[n1] + res
                n1 -= 1
            else:
                res = str2[n2] + res
                n2 -= 1
            if n1 == -1:
                res = str2[:n2+1] + res
                break
            if n2 == -1:
                res = str1[:n1+1] + res
                break
        return res
