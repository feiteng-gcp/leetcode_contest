class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        vector<int> mod(60, 0);
        for(int i=0; i<time.size(); i++) {
            mod[time[i]%60]++;
        }
        int ans = 0;
        for(int i=1; i<30; i++) {
            ans += mod[i] * mod[60-i];
        }
        ans += mod[0] * (mod[0]-1) / 2;
        ans += mod[30] * (mod[30]-1) / 2;
        
        return ans;
    }
};
