class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        vector<int> slot(60, 0);
        for(int i : time) {
            slot[i % 60]++;
        }
        int res = 0;
        for(int i = 1; i < 30; i++){
            res += slot[i] * slot[60 - i];
        }
        res += slot[0] * (slot[0] - 1) / 2;
        res += slot[30] * (slot[30] - 1) / 2;
        return res;
    }
};
