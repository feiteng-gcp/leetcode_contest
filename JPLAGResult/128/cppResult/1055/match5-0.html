class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& a) {
        vector<int> cnt(60);
        for (auto& it : a) {
            cnt[it % 60]++;
        }
        int ret = cnt[0] * (cnt[0] - 1) / 2;
        ret += cnt[30] * (cnt[30] - 1) / 2;
        for (int i = 1; i * 2 < 60; ++i) {
            ret += cnt[i] * cnt[60 - i];
        }
        return ret;
    }
};
