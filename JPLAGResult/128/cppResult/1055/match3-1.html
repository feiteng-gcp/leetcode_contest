class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        vector<int> d(60, 0);
        for (int i = 0; i < time.size(); i++) {
            d[time[i] % 60]++;
        }
        
        int ans = 0;
        for (int i = 1; i < 30; i++) {
            ans += d[i] * d[60 - i];
        }
        ans += d[0] * (d[0] - 1) / 2;
        ans += d[30] * (d[30] - 1) / 2;
        return ans;
    }
};
