class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        vector<int> cache(60, 0);
        int cnt = 0;
        for (int t : time) {
            int r = t % 60;
            cnt += cache[(60-r)%60];
            cache[r]++;
        }
        return cnt;
    }
};
