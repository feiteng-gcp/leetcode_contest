class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        vector<int> hist(60, 0);

        for(int t: time){
            hist[t%60]++;
        }
        //0 match with self, 1 with 59, 30 with self.
        int count = 0;
        if(hist[0]>1){
            count+=(hist[0]*(hist[0]-1))/2;
        }
        if(hist[30]>1){
            count+=(hist[30]*(hist[30]-1))/2;
        }
        for(int i = 1; i<30; i++){
            count+=hist[i]*hist[60-i];
        }
        return count;
    }
};
