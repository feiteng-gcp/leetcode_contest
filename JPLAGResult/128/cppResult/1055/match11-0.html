class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        int a[60];
        memset(a, 0, sizeof(a));
        for (int i = 0;i < time.size();i++) {
            int x = time[i] % 60;
            a[x]++;
        }
        int ret = 0;
        for (int i = 1;i <= 29;i++) {
            ret += a[i] * a[60 - i];
        }
        ret += a[0] * (a[0] - 1) / 2;
        ret += a[30] * (a[30] - 1) / 2;
        return ret;
    }
};
