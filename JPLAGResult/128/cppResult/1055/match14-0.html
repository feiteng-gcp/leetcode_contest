class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        vector<int> divisor(60, 0);
        for (int t : time) ++divisor[t % 60];
        int res = divisor[0] * (divisor[0] - 1) / 2 + divisor[30] * (divisor[30] - 1) / 2;
        for (int i = 1; i < 30; ++i) {
            res += divisor[i] * divisor[60 - i];
        }
        return res;
    }
};
