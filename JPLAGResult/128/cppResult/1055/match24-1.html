class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        vector<int> rem(60);
        for(int i = 0; i < time.size(); ++i)
            ++rem[time[i] % 60];
        int res = 0;
        for(int i = 1; i < 30; ++i)
            res += rem[i] * rem[60-i];
        res += rem[0] * (rem[0] - 1) / 2;
        res += rem[30] * (rem[30] - 1) / 2;
        return res;
    }
};
