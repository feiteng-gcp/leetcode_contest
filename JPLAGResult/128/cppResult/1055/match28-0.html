class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        vector<int> c(60, 0);
        for(int a:time){
            c[a%60] += 1;
        }
        int ret = 0;
        for(int i=1;i<=29;i++){
            ret += c[i] * c[60-i];
        }
        if(c[0] > 1){
            ret += c[0]*(c[0]-1)/2;
        }
        if(c[30] > 1){
            ret += c[30]*(c[30]-1)/2;
        }
        return ret;
    }
};
