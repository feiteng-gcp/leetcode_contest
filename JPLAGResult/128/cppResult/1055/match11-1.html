class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        int len = time.size();
        vector<int> left_cnt(60, 0);

        for (int i = 0; i < len ; i++) {
            int left = time[i] % 60;
            left_cnt[left]++;
        }
        
        int ans = 0;
        for (int i = 1; i < 30; i++) {
            ans += left_cnt[i] * left_cnt[60-i];
        }
        ans += left_cnt[0] * (left_cnt[0] - 1) / 2;
        ans += left_cnt[30] * (left_cnt[30] - 1) / 2;
        
        return ans;
    }
};
