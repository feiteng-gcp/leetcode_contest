class Solution {
 public:
  int numPairsDivisibleBy60(vector<int> &arr) {
    vector<int> mod_count(60, 0);
    for (int x : arr) mod_count[x % 60] += 1;

    int count = max(0, mod_count[0] * (mod_count[0] - 1) / 2);
    for (int i = 1; i < 30; ++i) {
      count += mod_count[i] * mod_count[60-i];
    }
    count += max(0, mod_count[30] * (mod_count[30] - 1) / 2);
    return count;
  }
};
