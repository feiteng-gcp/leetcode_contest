class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {

        vector <int> p(60, 0);
        int ans = 0;
        int n = time.size();
        for (int i = 0; i < n; i++) {
            int x = (60 - time[i] % 60) % 60;
            ans += p[x];
            p[time[i] % 60]++;
        }
        return ans;
    }
};
