class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        map<int,int>mp;
        int len=time.size();
        int ans = 0;
        for(int i=0;i<len;i++){
            int tp = time[i]%60;
            ans += mp[tp];
            mp[(60-tp)%60] ++;
        }
        return ans;
    }
};
