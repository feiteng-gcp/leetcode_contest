class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& times) {

        unordered_map<int, int> nums;
        for(int i = 0; i < times.size(); i++) {
            int time = times[i];
            nums[time % 60]++;
        }
        int total = 0;
        for(int i = 1; i < 30; i++) {
            total += nums[i]*nums[60-i];
        }
        total += nums[0]*(nums[0]-1)/2;
        total += nums[30]*(nums[30]-1)/2;
        return total;
    }
};
