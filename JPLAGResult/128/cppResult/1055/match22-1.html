class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        unordered_map<int,int> cnt;
        
        int ans = 0;
        for (int i = 0; i < time.size(); i++) {
            int res = time[i] % 60;
            if (cnt.find((60 - res)%60) != cnt.end())
                ans += cnt[(60 - res)%60];
            cnt[res]++;
        }
        
        return ans;
    }
};
