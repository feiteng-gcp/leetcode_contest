class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        int res = 0;
        int freq[60] = {0};
        int n = time.size();
        for (int i = 0; i < n; i++) {
            freq[ time[i] % 60]++;
        }
        

        for (int i = 0; i < 60; i++) {
            for (int j = i; j < 60; j++) {
                if ((i + j) % 60 == 0) {
                    if (i != j) res += freq[i] * freq[j];
                    else res += (freq[i] * (freq[i] - 1)) / 2;
                }
            }
        }
        return res;
    }
};
