class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        unordered_map<int, int> remainder;
        int res = 0;
        for(int i = 0; i < time.size(); i++){
            int tmp = time[i]%60;
            if(tmp == 0 && remainder.count(0) == 1) res += remainder[0];
            if(tmp != 0 && remainder.count(60-tmp) == 1) res += remainder[60-tmp];
            remainder[tmp]++;
        }
        return res;
    }
};
