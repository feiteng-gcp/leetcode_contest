#include <vector>
#include <string>
#include <unordered_map>
#include <queue>
#include <stack>
#include <list>
#include <iostream>

using namespace std;


class Solution
{
public:
    int numPairsDivisibleBy60(vector<int>& time)
    {
        int res = 0;
        unordered_map<int, size_t> durToCount;
        for (int t : time)
        {
            t %= 60;
            res += durToCount[(60 - t) % 60];
            ++durToCount[t];
        }

        return res;
    }
};
