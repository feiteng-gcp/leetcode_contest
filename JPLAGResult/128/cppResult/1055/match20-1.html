class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        int tmp[ 60 ] = {0};
        

        for( int i = 0; i < time.size(); ++i ){
            ++tmp[ time[i] % 60 ];
        }
        
        int ans = 0;
        
        if( tmp[ 0 ] > 0 ){
            ans += ( tmp[ 0 ] * (tmp[0] - 1 ) ) / 2;
        }
        
        if( tmp[ 30 ] > 0 ){
            ans += ( tmp[ 30 ] * ( tmp[30] - 1 ) ) / 2;
        }
        
        for( int i = 1; i < 30; ++i ){
            ans += tmp[ i ] * tmp[ 60 - i ];
        }
        
        return ans;
    }
};
