class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
    //    time = {60,60,60};
        vector<int> v(60, 0);
        for (auto t: time) {
            v[t % 60] ++;
        }
        
        int ret = 0;
        for (int i = 1; i < 30; ++i) {
            ret += v[i] * v[60 - i];
        }
        if (v[0] > 1) ret += (v[0] - 1) * v[0] / 2;
        if (v[30] > 1) ret += (v[30] - 1) * v[30] / 2;
        return ret;
    }
};
