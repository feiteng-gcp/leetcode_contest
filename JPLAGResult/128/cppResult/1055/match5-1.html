class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        vector<int> cnt(61, 0);
        for (auto& n: time) {
            cnt[n % 60]++;
        }
        int res = cnt[0] * (cnt[0] - 1) / 2;
        res += cnt[30] * (cnt[30] - 1) / 2;
        for (int i = 1; i < 30; ++i) {
            res += cnt[i] * cnt[60 - i];
        }
        return res;
    }
};
