class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        for( auto &x : time ) 
            x%=60 ; 
        map< int ,int > check ;
        int ans = 0 ; 
        for( auto &x : time ){
            check[x]++;
            if( check[60-x] ){
                ans += check[60-x] ;
                if( x == 30 ) 
                    ans -- ; 
            } 
            else if( x == 0 ){
                ans += (check[0]-1); 
            }
        }
        return ans ; 
    }
};
