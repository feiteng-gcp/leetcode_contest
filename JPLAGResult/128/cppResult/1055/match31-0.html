class Solution {
public:
    int numPairsDivisibleBy60(vector<int>& time) {
        vector<int>rem(60);
        int out = 0;
        for(int i = 0;i<time.size();++i){
            time[i] = time[i]%60;
            rem[time[i]]++;
        }
        for(int x:time){
            if(x == 0 || x == 30){
                out += rem[x]-1;
                rem[x]--;
            }
            else{
                out += rem[60-x];
                rem[x]--;
            }
        }
        return out;
        
    }
};
