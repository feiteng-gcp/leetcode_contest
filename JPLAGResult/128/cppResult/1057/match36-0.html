#include <bits/stdc++.h>
#define MAXN 105
using namespace std;

typedef long long int LL;

LL dp[15][1<<11], VAL;

LL cal(LL prefix, int D, int B) {
    LL mini = prefix;
    for (int i=0;i<D;++i) mini *= 10;
    if (mini > VAL) return 0;
    LL memo = 0;
    if (D == 0) memo = 1;
    else {
        for (int i=0;i<=9;++i) {
            if (i == 0 && prefix == 0) memo += cal(prefix, D-1, B);
            else if ((B & (1<<i)) == 0)
                memo += cal(prefix * 10 + i, D-1, B | (1<<i));
        }
    }
    dp[D][B] = memo;
    return memo;
}

class Solution {
public:
    int numDupDigitsAtMostN(int N) {
        VAL = N;
        memset(dp,-1,sizeof(dp));
        return N + 1 - cal(0, 10, 0);
    }
};
