class Solution {
public:
    int getMul(int all, int N) {
        int ans = 1;
        for (int i = 0; i < N; ++i)
            ans *= all - i;
        return ans;
    }
    
    int dp(const vector<int>& N, int state, int index, int left_num) {
        if (index >= N.size())
            return 1;
        
        int base = 1, cur_num = 0, ans = 0;
        while (base <= state && cur_num <= N[index]) {
            if (state & base) {
                // 最高位不能取0
                if (index == 0 && cur_num == 0) {
                    base <<= 1;
                    ++cur_num;
                    continue;
                }

                if (cur_num < N[index]) {
                    ans+= getMul(left_num - 1, (int) N.size() - index - 1);
                } else {
                    state ^= base;
                    ans += dp(N, state, index + 1, left_num - 1);
                    state |= base;
                }
            }
            base <<= 1;
            ++cur_num;
        }
        return ans;
    }
    
    int numDupDigitsAtMostN(int N) {
        int sum = N;
        
        vector<int> N_a;
        while (N) {
            N_a.push_back(N % 10);
            N /= 10;
        }
        reverse(N_a.begin(), N_a.end());
        
        int temp = 0;
        for (int i = 1; i < (int) N_a.size(); ++i) {
            // cout << i << " " << 9 * getMul(9, i - 1) << endl;
            temp += 9 * getMul(9, i - 1);
        }
        // cout << N_a.size() << " " << dp(N_a, (1 << 10) - 1, 0, 10) << endl;
        temp += dp(N_a, (1 << 10) - 1, 0, 10);
        
        return sum - temp;
    }
};
