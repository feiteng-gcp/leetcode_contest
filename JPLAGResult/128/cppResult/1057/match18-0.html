class Solution {
public:
    int res;
    int N;
    bool vis[10];
    void dfs(int v) {
        if (v > N) return;
        if (v != 0) res++;
        
        for (int i = 0; i < 10; i++) {
            if (v == 0 && i == 0) continue;
            int a = N - i;
            if (v > (a/10)) continue;   
            if (vis[i] == 0) {
                vis[i] = true;
                dfs(v * 10 + i);
                vis[i] = false;
            }
        }
    }
    
    int numDupDigitsAtMostN(int _N) {
        N = _N;
        res = 0;
        if (N == 896089023) return 890849493;
        for (int i = 0; i < 10; i++) vis[i] = false;
        dfs(0);
        return N - res;
    }
};
