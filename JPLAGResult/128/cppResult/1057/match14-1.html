class Solution {
public:
    int dp[11][1<<10][2][2];
    vector < int > a;
    
    void process(int N) {
        int x = N;
        while (x > 0) {
            a.push_back(x%10);
            x /= 10;
        }
        reverse(a.begin(), a.end());
    }
    
    int numDupDigitsAtMostN(int N) {
        memset(dp, -1, sizeof(dp));
        process(N);
        return calc(0, 0, 0, 1);        
    }
    
    int calc(int pos, int mask, bool hadRep, bool sp) {
        if (pos == a.size()) {
            if (mask == 0) return 0;
            if (hadRep) return 1;
            return 0;
        }
        int &res = dp[pos][mask][hadRep][sp];
        if (res != -1) return res;
        res = 0;
        if (sp) {
            for (int i=0;i<=a[pos];i++) {
                if (mask == 0 && i == 0) {
                    res += calc(pos+1, mask, hadRep, sp && i == a[pos]);

                } else {
                    res += calc(pos+1, mask | (1<<i), hadRep || ((mask&(1<<i)) > 0), sp && i == a[pos]);                
                }
            }        
        } else {
            for (int i=0;i<10;i++) {
                if (mask == 0 && i == 0) {
                    res += calc(pos+1, mask, hadRep, sp && i == a[pos]);
                } else {
                    res += calc(pos+1, mask | (1<<i), hadRep || ((mask&(1<<i)) > 0), sp && i == a[pos]);
                }
            }
        }
        return res;
    }
};
