class Solution {
public:
    int numDupDigitsAtMostN(int N) {
        m = N;
        int c = 0;
        for(int i = 1; i < 10; ++i) {
            c += dfs(i, 1 << i);
        }
        
        return m - c;
    }
    
    

    int dfs(uint64_t n, int mask) {
        if (n > m) return 0;
        int c = 1;
        for (int i = 0; i < 10; ++i) {
            int k = n % 10;
            if (k == i) continue;
            if (1 << i & mask) continue;
            if (n * 10 + i > m) break;
            c += dfs(n * 10 + i, 1 << i | mask);
        }
        
        return c;
    }
    
    int m;
    
    
};
