class Solution {
public:
	int numDupDigitsAtMostN(int n) {
		string s(to_string(n));
		int dp[11][2][1 << 10][2][2];
		for (int i = 0; i < 11; ++i) for (int j = 0; j < 2; ++j) for (int k = 0; k < 1 << 10; ++k) for (int l = 0; l < 2; ++l) for (int m = 0; m < 2; ++m) dp[i][j][k][l][m] = 0;

		dp[0][0][0][0][0] = 1;

		for (int i = 0; i < s.size(); ++i) {
			for (int j = 0; j < 2; ++j) {
				for (int k = 0; k < 1 << 10; ++k) {
					for (int l = 0; l < 2; ++l) {
						for (int m = 0; m < 2; ++m) {
							if (dp[i][j][k][l][m] == 0) continue;
							int lim = j ? 9 : (s[i] - '0');
							for (int d = 0; d <= lim; ++d) {

								dp[i + 1][j || d < lim][(m || d > 0) ? k | 1 << d : 0][l || (m && k & 1 << d)][m || d > 0] += dp[i][j][k][l][m];
							}
						}
					}
				}
			}
		}

		int ans = 0;
		for (int j = 0; j < 2; ++j) for (int k = 0; k < 1 << 10; ++k) ans += dp[s.size()][j][k][1][1];
		return ans;
	}
};
