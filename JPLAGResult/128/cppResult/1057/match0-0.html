class Solution {
public:
    int memo[15] ; 
    int vis[10] ; 
    int dfs( long long num , int N , int digit , int depth ){
        if( digit +1 == depth ) 
            return 1;  
        
        int ans = 1 ; 
        int i =0 ; 
        if(depth == 0 ) 
            i++;
        for(  ; i<10 ; i++){
            if( num*10+i <= N && vis[i] == 0 ){
                vis[i] = 1;  
                ans += dfs( num*10+i   , N , digit , depth+1 ) ;
                vis[i] = 0 ; 
            }
        }
        
        return ans ;
    }
    int numDupDigitsAtMostN(int N) {
        memset( memo , 0 , sizeof( memo ) ) ;
        memset( vis , 0 , sizeof( vis ) ) ; 
        int digit = 0 ; 
        int temp = N ; 
        while( temp ){
            digit ++ ;
            temp /=10 ; 
        }
        return N-dfs( 0 , N , digit , 0 )+1 ; 
    }
};
