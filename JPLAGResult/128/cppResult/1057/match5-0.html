class Solution {
private:
    int f[15][2][2][1<<10];
public:
    int numDupDigitsAtMostN(int N) {
        int tt = N;
        vector<int>d;
        while(N){
            d.push_back(N%10);
            N/=10;
        }
        int n = d.size();
        memset(f,0,sizeof(f));
        f[n][0][1][0] = 1;
        for(int i = n; i>0;i--)
            for(int j=0;j<2;j++)
                for(int k=0;k<2;k++)
                    for(int p=0;p<(1<<10);p++){
                        if(f[i][j][k][p]==0)continue;
                        for(int q=0;q<10;q++){
                            if(k==1&&q>d[i-1])continue;
                            if(p&(1<<q))continue;
                            if(j==0&&q==0){
                                f[i-1][j][0][p]+=f[i][j][k][p];
                            } else {
                                f[i-1][1][k==1&&q==d[i-1]][p|(1<<q)]+=f[i][j][k][p];
                            }
                        }
                    }
        int ret = 0;
        for(int i=0;i<2;i++)
            for(int j=0;j<(1<<10);j++)
                ret+=f[0][1][i][j];
        return tt - ret;
    }
};
