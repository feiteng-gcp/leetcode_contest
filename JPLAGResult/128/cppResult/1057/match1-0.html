class Solution {
public:
    string s;
    int dp[11][2][1029][2];
    int solve(int pos, bool match, int mask, bool ok){
        if(pos == (int)s.size())
            return ok ? 1 : 0;
        if(dp[pos][match][mask][ok] != -1)
            return dp[pos][match][mask][ok];
        int ans = 0;
        
        int lim = match ? s[pos] - '0' : 9;
        for(int i = 0; i <= lim; i++){
            if(i == 0){
                if(mask == 0){
                    ans += solve(pos + 1, match && i == s[pos] - '0', mask, ok);
                }else{
                    ans += solve(pos + 1, match && i == s[pos] - '0', mask | (1 << i), ok || (mask & (1 << i)));
                }
            }else{
                ans += solve(pos + 1, match && i == s[pos] - '0', mask | (1 << i), ok || (mask & (1 << i)));
            }
        }
        
        return dp[pos][match][mask][ok] = ans;
    }
    int numDupDigitsAtMostN(int N) {
        s = to_string(N);
        memset(dp, -1, sizeof dp);
        return solve(0, true, 0, false);
    }
};
