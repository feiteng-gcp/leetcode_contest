class Solution {
public:
    int numDupDigitsAtMostN(int N) {
        if (N < 11) return 0;
        
        int num_digit = 0, a = N;
        while (a) {
            num_digit++; a /= 10;
        }
        
        int dif_cnt = 0, tot = 1;
        for (int i = 1; i < num_digit; i++) {
            dif_cnt += 9 * tot;
            tot *= (9 - i + 1);
        }
        
        string s = to_string(N);
        vector<int> cnt(10);
        tot = 1;
        for (int i = 0; i < s.size() - 1; i++) tot *= (9 - i);
        for (int i = 0; i < s.size(); i++) {
            dif_cnt += select_num(cnt, s[i] - '0', i == 0 ? 1 : 0) * tot;
            tot /= (9 - i);
            if (cnt[s[i]-'0']) {
                break;
            }
            cnt[s[i]-'0'] = 1;
            if (i == s.size() - 1) dif_cnt += 1;
        }
        
        return N - dif_cnt;
    }
    int select_num(vector<int>& cnt, int num, int begin) {
        int tot = 0;
        for (int i = begin; i < num; i++) {
            if (!cnt[i]) {
                tot++;
            }
        }
        return tot;
    }
};
