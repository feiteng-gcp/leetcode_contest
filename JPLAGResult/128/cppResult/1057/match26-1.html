class Solution {
public:
	int dp[11][1 << 10][2][2][2];
	bool use[11][1 << 10][2][2][2];
	vector<int> v;
	inline int dfs(int a, int b, int low, int rep,int ban) {
		if (a < 0) {
			if (rep)return 1;
			return 0;
		}
		if (use[a][b][low][rep][ban])return dp[a][b][low][rep][ban];
		use[a][b][low][rep][ban] = true;
		int &A = dp[a][b][low][rep][ban];
		for (int i = 0; i < 10; i++) {
			if (i == 0 && ban)continue;
			if (low == 0 && v[a] < i)continue;
			A+=dfs(a - 1, b | (1 << i), low | (i < v[a]), rep | ((b >> i) & 1), false);
		}
		return A;
	}
	int numDupDigitsAtMostN(int N) {
		memset(dp, 0, sizeof(dp));
		memset(use, false, sizeof(use));

		while (N) {
			v.push_back(N % 10);
			N /= 10;
		}
		//reverse(v.begin(), v.end());
		int ans = 0;
		for (int i = v.size() - 1; i >= 0; i--) {
			ans+=dfs(i, 0, i!=(v.size()-1), false,true);
		}
		return ans;
	}
};
