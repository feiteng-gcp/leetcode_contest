class Solution {
public:
    int numDupDigitsAtMostN(int N) {
        if(N <= 10) return 0;
        if(N <= 99) return N / 11;
        int distinct = 90; // < 100
        string nn = to_string(N);
        for(int i = 3; i < nn.length(); i++){
            int cnt = 9, prod = 9;
            for(int j = i - 1; j > 0; j--){
                prod *= cnt;
                cnt--;
            }
            distinct += prod;
        }
        
        vector<bool> used(10, false);
        bool myself = true;
        for(int i = 0; i < nn.length(); i++){
            int curr = nn[i] - '0';
            for(int j = (i == 0 ? 1 : 0); j < curr; j++){
                if(used[j]) continue;
                int prod = 1;
                for(int k = i + 1; k < nn.length(); k++){
                    prod *= (10 - k);
                }
                distinct += prod;
            }
            if(used[curr]) {
                myself = false;
                break;
            }
            used[curr] = true;
        }
        if(myself) ++distinct;
        return N - distinct;
    }
