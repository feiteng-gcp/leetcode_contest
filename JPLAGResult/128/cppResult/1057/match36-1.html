int dp[1 << 10][10];
int pow101(int x) {
	int r = 1;
	for (int i = 0; i < x; ++i) {
		r *= 10;
	}
	return r;
}
int getdp(int m, int k) {
	if (k == 0) return 0;
	if (dp[m][k] >= 0) return dp[m][k];
	int r = 0;
	for (int i = 0; i<10; ++i) {
		if (m & (1 << i)) r += pow101(k - 1);
		else r += getdp(m | (1 << i), k - 1);
	}
	return dp[m][k] = r;
}

vector<int> zn, tn;
int comp(int now) {
	for (int i = 0; i <= now; ++i) {
		if (zn[i] < tn[i]) return -1;
		if (zn[i] >  tn[i]) return 1;
	}
	return 0;
}
bool leadn(int now) {
	for (int j = 0; j <= now; ++j) {
		if (tn[j] != 0) {
			return true;
		}
	}
	return false;
}
bool dup1(int now) {
	int m = 0;
	int fn = false;
	for (int j = 0; j <= now; ++j) {
		if (tn[j]) fn = fn|| true;
		if (!fn) continue;
		if (m&(1 << tn[j])) return true;
		m = m | (1 << tn[j]);
	}
	return false;
}

int work(int g, int m) {
	if (g == zn.size()) {
		if (leadn(g - 1) && dup1(g - 1)) {
			return 1;
		}
		return 0;
	}
	int r = 0;
	for (int i = 0; i<10; ++i) {
		tn[g] = i;
		int x = comp(g);
		if (x<0) continue;
		bool ln = leadn(g);
		if (x>0) {
			if (ln) {
				if (dup1(g)) r += pow101 (zn.size() - g - 1);
				else {
					r += getdp(m | 1 << i, zn.size() - g - 1);
				}
				continue;
			}
		}
		if (ln) {
			r += work(g + 1, m | 1 << i);
		}
		else {

			r += work(g + 1, m);
		}
	}
	tn[g] = 0;
	return r;
}

class Solution {
public:
	int numDupDigitsAtMostN(int N) {
		memset(dp, -1, sizeof(dp));
		zn.clear();
		int t = N;
		while (t>0) {
			zn.push_back(t % 10);
			t /= 10;
		}
		tn.resize(zn.size());
		reverse(zn.begin(), zn.end());
		int r = work(0, 0);
		return r;
	}
};
