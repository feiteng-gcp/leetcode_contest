class Solution {
    int dp1(int n,int i){
        int r=1;
        for(int j=0;j<i;j++) r*=(n-j);
        return r;
    }
    int res1(int i){
        return 9*dp1(9,i-1);
    }
public:
    int numDupDigitsAtMostN(int N) {
        vector<int> nums;
        int p=N;
        while(p>0){
            nums.push_back(p%10);
            p/=10;
        }
        
        reverse(nums.begin(),nums.end()); 
        int res=N;
        for(int i=1;i<=nums.size()-1;i++){
            res-=res1(i);
            //cout<<res<<endl;
        }
        vector<int> used(10,false);
        int remain=10;
        for(int i=0;i<nums.size();i++){
            for(int j=0;j<nums[i];j++){
                if(used[j]) continue;
                else if(j==0&&i==0) continue;
                else{
                    res-=dp1(remain-1,nums.size()-i-1);
                }
            }
            if(used[nums[i]]) break;
            used[nums[i]]=true;
            remain--;
        }
        set<int> ss;
        for(int n:nums) ss.insert(n);
        if(ss.size()==nums.size()) res--;
        return res;
    }
};
