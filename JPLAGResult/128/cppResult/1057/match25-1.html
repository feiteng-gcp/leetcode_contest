class Solution {
public:
    int check(int x)
    {
        vector<int> a(10, 0);
        while (x)
        {
            if (a[x % 10])
                return 0;
            a[x % 10] = 1;
            x /= 10;
        }
        return 1;
    }
    
    int numDupDigitsAtMostN(int N) {
        int m = N / 10000;
        int Nlow = max(m * 10000, 1);
        int ans = 0;
        for (int i = Nlow; i <= N; i++)
            ans += check(i);
        if (m)

            ans += 5274;
        
        vector<int> multi(11, 0);
        multi[1] = 1;
        for (int i = 2; i <= 10; i++)
            multi[i] = multi[i - 1] * i;
        
        for (int i = 1; i < m; i++)
        {
            vector<int> a(10, 0);
            int val = i;
            int cnt = 0;
            bool flag = true;
            while (val)
            {
                if (a[val % 10])
                {
                    flag = false;
                    break;
                }
                cnt++;
                a[val % 10] = 1;
                val /= 10;
            }
            if (flag)
                ans += multi[10 - cnt] / multi[6 - cnt];
        }
        return N - ans;
    }
};
