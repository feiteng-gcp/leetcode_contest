class Solution {
public:
    std::vector<int> bit;
    int n;

    int dfs(int idx, int state) {
      if (idx >= n) {
        return 1;  // N is special number.
      }

      int ans = 0;
      for (int i = 0; i < bit[idx]; ++i) {
        if (idx == 0 && i == 0) continue;
        if (state & (1<<i)) continue;
        int re_len = n - idx - 1;
        int c = C(10-idx-1, re_len);
        int a = A(re_len);
        ans += c * a;
      }

      if (!(state & (1<<bit[idx]))) {
        int new_state = state | (1<<bit[idx]);
        int tmp = dfs(idx+1, new_state);
        ans += tmp;
        //printf("equal, idx=%d, state=%d, new_state=%d, tmp=%d, ans=%d\n", idx, state, new_state, tmp, ans);
      }

      return ans;
    }

    int gao2() {
      int tmp = dfs(0, 0); // idx, state
      return tmp;
    }

    int C(int X, int Y) {
      if (Y > X-Y) Y = X-Y;
      int tmp[2] = {1, 1};
      for (int i = 0; i < Y; ++i) tmp[0] *= (X-i), tmp[1] *= (i+1);
      return tmp[0] / tmp[1];
    }

    int A(int X) {
      int tmp = 1;
      for (int i = 0; i < X; ++i) tmp *= (i+1);
      return tmp;
    }

    int gao1(int len) {
      int ret = 0;
      std::vector<int> ans;  // debug
      for (int i = 1; i < len; ++i) {
        if (i == 1) ans.push_back(9);
        else {
          auto work = [this](int L){
            int c = C(9, L);
            int a = A(L);
            int tmp0 = c * a;

            c = C(9, L-1);
            a = A(L-1);
            int tmp1 = c * a * (L-1);

            return tmp0 + tmp1;
          };
          int tmp = work(i);
          ans.push_back(tmp);
          //printf("gao1, len=%d, ans=%d\n", i, ans.back());
          //int c = C(10, i);
          //int a = A(i);
          //ans.push_back(c * a - ans.back());
          //printf("gao1, len=%d, c=%d, a=%d, ans=%d, pre=%d\n", i, c, a, ans.back(), ans[ans.size()-2]);
        }
        ret += ans.back();
      }
      return ret;
    }

    int numDupDigitsAtMostN(int N) {
      int x = N;
      while (x) {
        bit.push_back(x % 10);
        x /= 10;
      }
      std::reverse( bit.begin(), bit.end() );
      n = bit.size();

      int ret[2] = {0, 0};
      // solve length of: 1, 2, .... n-1
      ret[0] = gao1(n);
      // solve length of: n
      ret[1] = gao2();
      //printf("solve: ret0=%d, ret1=%d\n", ret[0], ret[1]);
      return N - ret[0] - ret[1];
    }
};
