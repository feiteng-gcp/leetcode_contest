class Solution {
public:
    int a[20];
    int dp[20][1024];
    int dfs(int pos,int sta,bool limit)
    {
        
        if(pos==-1) return 1;
        if(!limit && dp[pos][sta]!=-1) return dp[pos][sta];
        int up=limit ? a[pos] : 9;
        int tmp=0;
        if(sta==0) tmp+=dfs(pos-1,sta,false);
        for(int i=0;i<=up;i++)
        {
            if(i==0&&sta==0) continue;
            int x = 1<<i;
            if((sta^x) < sta) continue;
            tmp+=dfs(pos-1,sta^x,limit && i==a[pos]);
        }
        if(!limit) dp[pos][sta]=tmp;
        return tmp;
    }
    int solve(int x)
    {
        int pos=0;
        while(x)
        {
            a[pos++]=x%10;
            x/=10;
        }
        return dfs(pos-1,0,true);
    }
    int numDupDigitsAtMostN(int N) {
        memset(dp,-1,sizeof(dp));
        return N-solve(N)+1;
    }
};
