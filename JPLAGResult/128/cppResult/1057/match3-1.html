class Solution {

    int value = 0;
    bool taken[10];
    void rec(long long num, int dig, int N){
        if(num > N) return;
        value += 1;
        for(int x = 0; x < 10; x += 1){
            if(taken[x]) continue;
            if(dig + x == 0) continue;
            taken[x] = true;
            rec(num * 10 + x, dig + 1, N);
            taken[x] = false;
        }
    }
    int all_different(int N){
        rec(0, 0, N);
        return value;
    }
public:
    int numDupDigitsAtMostN(int N) {
        return N - all_different(N) + 1;
    }
};
