class Solution {
public:
    int dp[1<<10][12][2][2];
    vector<int> d;
    int go(int m, int p, int eq, int good) {
        int &r = dp[m][p][eq][good];
        if (r==-1) {
            if (p==d.size()) return r=good;
            r = 0;
            for (int i = 0; i < 10; ++i) {
                if (!eq || d[p]>=i) {
                    int g = good;
                    if (i==0) {
                        if (m && (m&1)) g=1;
                    } else {
                        if (m&(1<<i)) g=1;
                    }
                    int mm = m;
                    if (i==0) {
                        if (m) mm|=1;
                    } else {
                        mm |= (1<<i);
                    }
                    r += go(mm, p+1, eq && d[p]==i, g);
                }
            }
        }
        return r;
    }
    
    int numDupDigitsAtMostN(int N) {
        d.clear();
        while(N) {
            d.push_back(N%10);
            N/=10;
        }
        reverse(d.begin(), d.end());
        memset(dp,-1,sizeof(dp));
        return go(0,0,1,0);
    }
};
