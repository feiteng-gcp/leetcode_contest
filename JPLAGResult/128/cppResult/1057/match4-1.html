class Solution {
public:
    int shu[20], dp[20][1<<10];
    int numDupDigitsAtMostN(int N) {
        return N-solve(N)+1;
    }

    int dfs(int len, int st, bool shangxian)
    {
//    printf("%d %d\n", len, st);
        if (len == 0)
            return 1;
        if (!shangxian && dp[len][st])
            return dp[len][st];
        int cnt = 0, maxx = (shangxian ? shu[len] : 9);
        for (int i = 0; i <= maxx; i++)
        {
            if ((st>>i)&1)
                continue;
           int nxtst = (st==0&&i==0)?0:(st|(1<<i));
           cnt += dfs(len - 1, nxtst, shangxian && i == maxx);
        }
        return shangxian ? cnt : dp[len][st] = cnt;
    }

    int solve(int x)
    {
        memset(shu, 0, sizeof(shu));

        memset(dp, 0 , sizeof(dp));
        int k = 0;
        while (x)
        {
            shu[++k] = x % 10;
            x /= 10;
        }
        return dfs(k, 0, true);
    }
};
