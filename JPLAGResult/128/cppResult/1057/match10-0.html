class Solution {
    int F(int limit, int mask, long long num) {
        if (num > limit) return 0;
        int ret = 1;
        for (int i = 0; i < 10; i++) {
            if ((mask >> i) & 1) continue;
            ret += F(limit, mask | (1 << i), num * 10 + i);
        }
        return ret;
    }
    
public:
    int numDupDigitsAtMostN(int N) {
        int ret = 0;
        for (int i = 1; i < 10; i++) ret += F(N, (1 << i), i);
        return (N - ret);
    }
};
