class Solution {
public:
    int numDupDigitsAtMostN(int n) {
        if (n <= 10)
            return 0;
        
        // int ans = 0;
        // for (int i = 11; i <= n; ++i) {
        //     if (helper(i))
        //         ++ans;
        // }
        // return ans;
        
        vector<int> d;
        int tmp = n+1;
        while (tmp > 0) {
            d.insert(d.begin(), tmp % 10);
            tmp /= 10;
        }
        
        vector<bool> vis(10);
        
        int fac = 1, len = d.size(), ans = 0;;
        for (int i = 2; i <= 9; ++i)
            fac *= i;
        
        for (int i = 2; i <= 10-len; ++i)
            fac /= i;
        
        for (int i = 0; i < len; ++i) {
            if (d[i] > 0) {
                int cnt = 0, st = i == 0 ? 1 : 0;
                for (int j = st; j < d[i]; ++j)
                    if (!vis[j])
                        ++cnt;
                
                // printf("%d %d: %d %d\n", i, d[i], cnt, fac);
                ans += cnt * fac;
            }
            
            if (vis[d[i]])
                break;
            vis[d[i]] = true;
            
            if (i < 9)
                fac /= (9-i);
        }
        
        // cout << ans << endl;
        

        if (len > 2) {
            int a = 9, sum = 9;
            for (int i = 2; i < len-1; ++i) {
                a = a * (10-i);
                sum += a;
            }
            cout << sum << endl;
            ans += sum * 9;
        }
        ans = n - 9 - ans;
        
        return ans;
    }
    
    // bool helper(int v) {
    //     int last = -1;
    //     while (v > 0) {
    //         if (last == v%10)
    //             return true;
    //         last = v%10;
    //         v /= 10;
    //     }
    //     return false;
    // }
};
