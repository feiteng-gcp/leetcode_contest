int step = 500000;

class Solution {
    bool ok(int i) {
        int cnt[10] = {};
        while (i) {
            cnt[i % 10]++;
            i /= 10;
        }
        for (int i = 0; i < 10; i++) {
            if (cnt[i] > 1) return true;
        }
        return false;
    }
public:

    int numDupDigitsAtMostN(int N) {
        int ret = ans[N / step];
        for (int i = N / step * step + 1; i <= N; i++) {
            if (ok(i)) ret++;
        }
        return ret;
    }
