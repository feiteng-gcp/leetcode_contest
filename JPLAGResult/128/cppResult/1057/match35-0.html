class Solution {
public:
    int res , d[20] , len , a[20];
    
    int C(int n , int k) {
        if (k > n || k < 0) return 0;
        int t = 1;
        for (int i = 0 ; i < k ; ++ i) {
            t *= (n - i);
           // t /= (i + 1);
        }
        printf("C(%d, %d) = %d\n" , n , k , t);
        return t;
    }
    
    void dfs(int k , int less , int large) {        
        if (less && large) {
            for (int i = len - 1 ; i >= 0 ; -- i) {
                printf("%c" , i > k ? '0' + a[i] : '*');
            } puts("");
            int p = len - 1;
            while (p > k && a[p] == 0) {
                -- p;
            }
            int cnt[10] = {};
            while (p > k) {
                ++ cnt[a[p --]];
            }
            int num = 10;
            for (int i = 0 ; i < 10 ; ++ i) {
                if (cnt[i] > 1) {
                    return;
                }
                num -= cnt[i];
            }
            res += C(num, k + 1);
            return;
        }
        if (k == -1) {
           return; 
        }
        int e = less ? 9 : d[k];
        for (int i = 0 ; i <= e ; ++ i) {
            a[k] = i;
            dfs(k - 1, less || (i < e) , large || (i > 0));
        }

    }
    
    int numDupDigitsAtMostN(int N) {
        ++ N;
        int m = N;
        while (m) {
            d[len ++] = m % 10;
            m /= 10;
        }
        dfs(len - 1 , 0 , 0);
        return N - 1 - res;
    }
};
