class Solution {
public:
    
    string num;
    int memo[11][2][2][1 << 11][2];
    
    int solve(int idx, bool limited, bool hasDigit, int repeatedDigits, bool repeated) {
        
        if(idx >= num.length())
            return repeated;
        

        if(memo[idx][limited][hasDigit][repeatedDigits][repeated] > -1)
            return memo[idx][limited][hasDigit][repeatedDigits][repeated];
        
        int ans = 0;
        int limit = limited ? num[idx] - '0' : 9;
        for(int d = 0; d <= limit; ++d) {
            
            bool newHasDigit = hasDigit || (!hasDigit && d > 0);
            bool newLimited = limited && (d == num[idx] - '0');
            bool newRepeated = repeated || (newHasDigit && ((repeatedDigits & (1 << d)) > 0));            
            int newRepeatedDigits = repeatedDigits;
            if(newHasDigit)
                newRepeatedDigits |= (1 << d);
            
            ans += solve(idx + 1, newLimited, newHasDigit, newRepeatedDigits, newRepeated);

        }
                        
        return memo[idx][limited][hasDigit][repeatedDigits][repeated] = ans;
    }
    
    int numDupDigitsAtMostN(int N) {
        
        num = to_string(N);
        memset(memo, -1, sizeof memo);
        return solve(0, 1, 0, 0, 0);
    }
};
