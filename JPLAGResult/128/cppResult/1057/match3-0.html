class Solution {
public:
    int ans = 0;
    void dfs(long long now, int s, int N) {
        if (now > N) return;
        ans++;
        for (int i = 0; i < 10; i++) {
            if (!((1 << i) & s)) {
                dfs(now * 10 + i, 1 << i | s, N);
            }
        }
    }

    int numDupDigitsAtMostN(int N) {
        ans = 0;
        for (int i = 1; i <= 9; i++) {
            dfs(i, 1 << i, N);
        }
        return N - ans;
    }
};
