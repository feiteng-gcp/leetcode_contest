class Solution {
public:
    int memo[1<<10][11][2][2][2];
    string s;
    
    int dp(int mask,int pos, bool menor,bool zero,bool goal){
        if(pos==s.size())return goal;
        if(memo[mask][pos][menor][zero][goal]!=-1)return memo[mask][pos][menor][zero][goal];
        
        int dev=0;
        if(menor){
            for(int i=0;i<10;i++){
                if( (mask&(1<<i))!=0 ){
                    dev+=dp(mask,pos+1,menor,zero|(i!=0),1);   
                }else{
                    if(i==0){
                        if(zero==0){
                            dev+=dp(mask,pos+1,menor,zero,goal);
                        }else{
                            dev+=dp(mask|(1<<0),pos+1,menor,zero,goal);
                        }
                    }else{
                        dev+=dp(mask|(1<<i),pos+1,menor,1,goal);
                    }
                }
            }
            
        }else{
            int val=s[pos]-'0';
            
            for(int i=0;i<val;i++){
                if( (mask&(1<<i))!=0 ){
                    dev+=dp(mask,pos+1,1,zero|(i!=0),1);   
                }else{
                    if(i==0){
                        if(zero==0){
                            dev+=dp(mask,pos+1,1,zero,goal);
                        }else{
                            dev+=dp(mask|(1<<0),pos+1,1,zero,goal);
                        }
                    }else{
                        dev+=dp(mask|(1<<i),pos+1,1,1,goal);
                    }
                    
                }
            }
            
            if(val==0){
                if(zero==0){
                    dev+=dp(mask,pos+1,0,zero,goal);
                }else{
                    dev+=dp(mask|(1<<0),pos+1,0,zero,goal|( (mask&(1<<0))!=0) );
                }
            }else{
                dev+=dp(mask|(1<<val),pos+1,0,1,goal|( (mask&(1<<val))!=0) );
            }
            
            
        }
        
        memo[mask][pos][menor][zero][goal]=dev;
        return dev;
    }
    
    int numDupDigitsAtMostN(int N) {
        memset(memo,-1,sizeof(memo));
        stringstream st;
        st<<N;
        s=st.str();
        return dp(0,0,0,0,0);
    }
};
