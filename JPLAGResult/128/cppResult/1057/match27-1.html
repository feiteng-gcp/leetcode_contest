class Solution {
public:
    vector<int> svd(int k) {
        vector<int> ans;
        while (k>0) {
            ans.push_back(k%10);
            k/=10;
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
    int cal1(int k) {
        int i,ans=9,t=9;
        for (i=1;i<k;i++)
            ans*=t--;
        return ans;
    }
    int cal2(vector<int>& d, int k, int c) {
        int i,ans=k,t=10-d.size()-1;
        for (i=0;i<d.size();i++) if (d[i]<k) ans--;
        for (i=0;i<c;i++)
            ans*=t--;
        return ans;
    }
    int check(int k) {
        vector<int> cnt(10,0);

        while (k>0) {
            cnt[k%10]++;
            k/=10;
        }
        for (int i=0;i<10;i++) if (cnt[i]>=2) return 1; 
        return 0;
    }
    int numDupDigitsAtMostN(int N) {
        vector<int> a=svd(N);
        int i,n=a.size(),ans=0;
        for (i=1;i<n;i++)
            ans+=cal1(i);
        //cout<<ans<<endl;
        vector<int> d;
        vector<int> cnt(10,0);
        ans+=cal2(d,a[0]-1,n-1);
        //cout<<ans<<endl;
        for (i=0;i<n-1;i++)
        {
            d.push_back(a[i]);
            cnt[a[i]]++;
            if (cnt[a[i]]>1) break;
            ans+=cal2(d,a[i+1],n-i-2);
        }
        if (!check(N)) ans++;
        return N-ans;
    }
};
