class Solution {
public:
    int dis[10] = {0};
    int solve( int m, vector<bool> used, int loss ){
        
        if( m == 0 ){
            return 1;
        }
        
        int tmp = 1;
        int ans = 0;
        
        for( int i = 0; i < m - 1; ++i ){
            tmp *= 9 - loss - i;
        }
        
        for( int i = 0; i < dis[ m - 1 ]; ++i ){
            if( used[ i ] != true ){
                ans += tmp;
            }
        }
        
        cout << "ans " << ans << endl;
        if( used[ dis[m-1] ] == false ){
            used[ dis[ m - 1 ] ] = true;
            loss = loss + 1;
            
            ans += solve( m - 1, used, loss );
        }
        
        return ans;
    }
    
    int numDupDigitsAtMostN(int N) {
        int m = 0;
        int ori_N = N;
            
        while( N > 0 ){
            dis[m++] = N % 10;
            N = N / 10;
        }
        
        cout << "m " << m << endl;
        
        int ans = 0;
        for( int i = 1; i < m; ++i ){
            int tmp = 9;
            
            for( int j = 0; j < i-1; ++j ){
                tmp *= ( 9 - j );
            }
            
            ans += tmp;
        }
        
        
        cout << "m-1 " << ans << endl;
        
        int tmp = 1;
        
        for( int i = 0; i < m - 1; ++i ){
            tmp *= ( 9 - i );
        }
        
    
        ans += tmp * ( dis[ m - 1 ] - 1 );
        cout << "m + m - 1 " << ans << endl;    
        vector<bool> used( 10, false );
        used[ dis[m-1] ] = true;
        
        ans += solve( m - 1, used, 1 );
        
        return ori_N - ans;
    }
};
