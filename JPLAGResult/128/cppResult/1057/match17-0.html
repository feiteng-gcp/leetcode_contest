class Solution {
public:
    int numDupDigitsAtMostN(int N) {
        if(N<10)return 0;
        int store[11];
        int non[11];
        non[0]=9;
        for(int i=1;i<=9;i++)
        {
            non[i]=non[i-1]*(10-i);
        }
        int top=0;
        int n=N;
        while(n)
        {
            store[top++]=n%10;
            n=n/10;
        }
        int ans=0;
        for(int i=0;i<top-1;i++)
        {
            ans=ans+non[i];
        }
        bool ok[11];
        memset(ok,0,sizeof(ok));
        for(int i=top-1;i>=0;i--)
        {
            int c=1;
            for(int j=1;j<=i;j++)
            {
                c=c*(10-(top-i)-j+1);
            }
            int tim=0;
            for(int j=0;j<store[i];j++)
            {
                if(!ok[j])tim++;
            }
            if(i==top-1)tim--;
            if(i==0&&!ok[store[i]])tim++;
            ans=ans+c*tim;
            if(ok[store[i]])break;
            ok[store[i]]=true;
        }
        return N-ans;
    }
};
