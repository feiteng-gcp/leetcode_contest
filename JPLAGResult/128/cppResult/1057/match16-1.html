int bit[1 << 10];
long long fac[11];
long long dpx[12], P[12][12];
class Solution {
public:
    int comp(int N) {
        dpx[0] = 1;

        dpx[1] = 9;
        for (int i = 2; i <= 10; i++) {
            dpx[i] = dpx[i - 1] * (11 - i);
        }
        for (int i = 2; i <= 10; i++) {
            dpx[i] += dpx[i - 1];
        }
        P[0][0] = 0;
        for (int i = 1; i <= 10; i++) {
            P[i][0] = 1;
            for (int j = 1; j <= i; j++) {
                P[i][j] = P[i][j - 1] * (i - j + 1);
            }
        }
        bit[0] = 0;
        for (int i = 1; i < 1024; i++) {
            bit[i] = bit[i >> 1] + (i & 1);
        }
        char buf[20];
        sprintf(buf, "%d", N);
        int nlen = strlen(buf);

        int ans = 0, mask = 0, fail = 0;
        for (int i = 0; i < nlen; i++) {
            int cur = buf[i] - '0';
            for (int j = 0; j < cur; j++) {
                int tmpmask = mask;
                if (mask != 0 || j != 0) {
                    if (tmpmask & (1 << j)) {
                        continue;
                    }
                    tmpmask |= (1 << j);
                }
                if (tmpmask == 0) {
                    ans += dpx[nlen - i - 1];
                } else {
                    ans += P[10 - bit[tmpmask]][nlen - i - 1];
                }
            }
            if (mask != 0 || cur != 0) {
                if (mask & (1 << cur)) {
                    fail = 1;
                    break;
                }
                mask |= (1 << cur);
            }
        }
        if (!fail) {
            ans++;
        }
        return ans;
    }
    int numDupDigitsAtMostN(int N) {
        if (N < 10) return 0;
        return N - comp(N);
    }
};
