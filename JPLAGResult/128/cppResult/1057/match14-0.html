class Solution {
public:
    int calLengthWithoutZero(int l, int res) {
        if (res - l < 0) {
            return 0;
        }
        if (l == 0) {
            return 1;
        }
        int ret = 1;
        for (int i = 0;i < l;i++) {
            ret *= (res - i);
        }
        return ret;
    }
    int calLengthWithZero(int l) {
        return 9 * calLengthWithoutZero(l - 1, 9);
    }
    int numDupDigitsAtMostN(int N) {
        int temp = N;
        int a[100];
        int n = 0;
        while (temp != 0) {
            a[n++] = temp % 10;
            temp /= 10;
        }
        
        int ret = 1;
        for (int i = 1;i <= n - 1;i++) {
            ret += calLengthWithZero(i);
        }
        bool bit[10];
        memset(bit, false, sizeof(bit));
        for (int i = n - 1;i >= 0;i--) {
            if (i == n - 1) {
                for (int j = 1;j < a[i];j++) {
                    if (!bit[j]) {
                        ret += calLengthWithoutZero(i, 10 - (n - i));
                    }
                }
            } else {
                for (int j = 0;j < a[i];j++) {
                    if (!bit[j]) {
                        ret += calLengthWithoutZero(i, 10 - (n - i));
                    }
                }
            }
            
            if (bit[a[i]]) {
                ret--;
                break;
            }
            bit[a[i]] = true;
        }
        return N - ret;
    }
};
