class Solution {
public:
  string s;
  int dp[12][2][2][2][1<<10];
  int dfs(int i,int l,int t,int f,int p){
    int &res=dp[i][l][t][f][p];
    if(~res) return res;
    if(i==(int)s.size()) return res=f;
    
    res=0;
    for(int j=0;j<=(t?s[i]-'0':9);j++){
      int ni=i+1;
      int nl=l&&(j==0);
      int nt=t&&(j==s[i]-'0');
      int nf=f||((p>>j)&1);
      int np=p|((!nl)<<j);
      res+=dfs(ni,nl,nt,nf,np);
    }
    return res;
  }
  int numDupDigitsAtMostN(int N) {
    s=to_string(N);
    memset(dp,-1,sizeof(dp));
    return dfs(0,1,1,0,0);
  }
};
