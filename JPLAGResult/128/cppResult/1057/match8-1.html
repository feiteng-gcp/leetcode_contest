class Solution {
public:
	int isDup(int N){
		vector<int> x(10);

		while (N > 0)
		{
			x[N % 10]++;
			N /= 10;
		}
		for (int i = 0; i < 10; i++)
			if (x[i]>1)
				return true;
		return false;
	}

	int numNotDupDigitsAtMostN(int N) {
		int n = to_string(N).size();
		int i, sum = 0, a = 9;
		for (i = 1; i < n; i++)
		{
			sum += a;
			a *= (10 - i);
		}
		int hi = N / 100000;
		int lo = N % 100000;
		int hinum = 0, lonum = 0;
		if (n >= 6)
		{
			for (int x = pow(10, n - 6); x < hi; x++)
				if (!isDup(x))
					hinum++;
		}
		sum += hinum*(11 - n)*(12 - n)*(13 - n)*(14 - n)*(15 - n);
		for (i = max(hi * 100000, (int)pow(10, n-1)); i <= N; i++)
			if (!isDup(i))
				sum++;
		return sum;
	}

	int numDupDigitsAtMostN(int N) {
		return N - numNotDupDigitsAtMostN(N);
	}
};
