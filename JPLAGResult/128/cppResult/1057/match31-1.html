static const int ten[] = {1, 10, 100, 1000, 10000, 100000, 1000000, 10000000, 100000000, 1000000000};
class Solution {
public:
    vector<int> d;
    // leading zeros not count
    long long int dfs(int digit, int m, bool s, bool ld) {
        if (digit >= d.size()) return 1;
        
        // all same
        if (s) {

            long long int val = 0;
            for (int i = d[digit]; i >= 0; --i) {
                if ((1 << i) & m) continue;
                val += dfs(digit + 1, m | (1 << i), i == d[digit], ld && i == 0);
            }
        // cout << digit << " " <<  m << " " << s << " " << val << " nor" << endl;
            return val;
            
        } else {
            int left = 0;
            int leftd = d.size() - digit;
            for (int i = 0; i < 10; ++i) {
                if (((1 << i) & m) == 0) {
                    ++left;
                }
            }
            if (left < leftd) return 0;
            
            long long int val = 1;
            // first digit not 0 or 0
            if ((m & 1) == 0 && ld) {
                for (int i = 0; i < leftd; ++i) {
                    val *= (i == 0 ? left - 1 : left);
                    left -= 1;
                }
                val += dfs(digit + 1, m, s, ld);
            } else {
                for (int i = 0; i < leftd; ++i) {
                    val *= left;
                    left -= 1;
                }
            }

            if (leftd == 1 && ld) {
                val -= 1;
            }
            // 0000....0?
        // cout << digit << " " <<  m << " " << s << " " << val << " leftd=" << leftd << endl;
            return val;
        }
        return 0;
    }
    int numDupDigitsAtMostN(int N) {
        if (N < 11) return 0;
        
        int n = N;
        while (n) {
            d.push_back(n % 10);
            n /= 10;
        }
        reverse(d.begin(), d.end());
        
        long long int val = 0;
        for (int i = d[0]; i >= 0; --i) {
            long long tmp = dfs(1, i == 0 ? 0 : (1 << i), i == d[0], i == 0);
            // cout << "select " << i << " " << tmp << endl;
            val += tmp;
        }

        return N - val;
    }
};
