class Solution {
public:
    int bitwiseComplement(int N) {
        if (N == 0) return 1;
        vector<int> vec;
        while (N) {
            vec.emplace_back(!(N & 1));
            N >>= 1;
        }
        int b = 1, ans = 0;

        for (int i = vec.size() - 1; i >= 0; --i) {
            ans <<= 1;
            ans |= vec[i];
        }
        return ans;
    }
};
