class Solution {
public:
    int bitwiseComplement(int N) {
        if (!N)
            return 1;
        int ans = 0, base = 0;
        while (N) {
            if (!(N & 1))
                ans += 1 << base;
            N >>= 1;
            ++base;
        }
        return ans;
    }
};
