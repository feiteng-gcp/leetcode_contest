class Solution {
public:
    int bitwiseComplement(int N) {
        if(N==0)
            return 1;
        vector<int> A;
        while(N){
            A.push_back(1-N%2);
            N=N/2;
        }
        int cnt=1;
        int ans=0;
        for(int i=0;i<A.size();i++){
            ans+=cnt*A[i];
            cnt*=2;
        }
        return ans;
    }
};
