class Solution {
public:
    int shipWithinDays(vector<int>& weights, int D) {
        int mn = INT_MIN, sum = 0;
        for(int i = 0; i < weights.size(); i++){
            mn = max(mn, weights[i]);
            sum += weights[i];
        }
        int left = mn, right = sum;
        while(left < right){
            int mid = left + (right - left)/2;
            if(isValid(mid, weights, D)){
                right = mid;
            }else{
                left = mid+1;
            }
        }
        return left;
    }
    
    bool isValid(int &mid, vector<int> &weights, int &D){
        int cnt = 1, curr = 0;
        for(int i = 0; i < weights.size(); i++){
            if(curr + weights[i] <= mid){
                curr += weights[i];
            }else{
                cnt++;
                curr = weights[i];
            }
        }
        return cnt <= D;
    }
};
