class Solution {
    vector<int> v;
    int D;
    bool check(int total) {
        int tmp = total;
        int ans = 0;
        for (int i = 0; i < v.size(); i++) {
            if (tmp >= v[i]) tmp -= v[i];
            else {
                if (tmp == total) return false;
                ans++;
                tmp = total;
                i--;
            }
        }
        if (tmp < total) ans++;
        return ans <= D;
    }
public:
    int shipWithinDays(vector<int>& weights, int d) {
        v = weights;
        D = d;
        
        int l = 0, r = 50000 * 500;
        while (l < r) {
            int mid = (r + l) >> 1;
            if (check(mid)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        
        return r;
        
    }
};
