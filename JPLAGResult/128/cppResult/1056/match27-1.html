class Solution {
    vector<int> val;
    int day;
public:
    bool Finish(int W){
        int pos=0;

        for(int j=0;j<day;j++){
            int res=W;
            while(pos<val.size() &&res>=val[pos]){
                res-=val[pos++];
                if(pos==val.size()) return true;
            }
        }
        return false;
    }
    int shipWithinDays(vector<int>& weights, int D) {
        val=weights;
        day=D;
        int cursum=0;
        for(int i:val){
            cursum+=i;
        }

        int right=cursum+1,left=0;
        while(right>left+1){
            int mid=(left+right)/2;
            if(Finish(mid)) right=mid;
            else left=mid;
        }
        return right;
    }
};
