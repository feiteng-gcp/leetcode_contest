class Solution {
public:
    int shipWithinDays(vector<int>& weights, int D) {
        long l = INT_MIN, h = 0;
        for (long e: weights) {
            l = max(e, l);
            h += e;
        }

        while (l < h) {
            int mid = l + (h - l) / 2;
            if (cuts(weights, mid) > D) {
                l = mid + 1;
            } else {
                h = mid;
            }
        }
        return l;
    }
    int cuts(vector<int>& nums, int len) {
        int cnt = 0;
        long cursum = 0;

        for (long e: nums) {
            if (cursum + e <= len) {
                cursum += e;
            } else {
                ++cnt;
                cursum = e;
            }
        }
        return cnt + (cursum != 0);
    }
};
