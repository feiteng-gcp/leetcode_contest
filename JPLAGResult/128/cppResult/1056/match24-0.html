class Solution {
public:
	const int INF = 500*50000+1000;
	int C(vector<int>& weights, int x){
		int temp = 0, con = 1;
		for(int i = 0; i < weights.size(); i++){
			if(weights[i] > x) return INF;
			if(temp + weights[i] <= x) temp += weights[i];
			else {
				temp = weights[i];
				con++;
			}
			
		}
		return con;
		
	}
    int shipWithinDays(vector<int>& weights, int D) {
        
		int l = 1, h= INF, m;
		
		while(l+1<h){
			m = (l+h)/2;
			if(C(weights,m) > D) l = m;
			else h = m;
		}
		
		return h;
    }
};
