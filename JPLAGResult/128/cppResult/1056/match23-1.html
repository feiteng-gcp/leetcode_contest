class Solution {
    
private:
    bool testWeight(const vector<int>& weights, int D, int ww) {
        int d = 1, rst = ww;

        for (auto w : weights) {
            if (rst >= w) {
                rst -= w;
            } else {
                d++;
                rst = ww - w;
            }
        }
        return d <= D;
    }
    
public:
    int shipWithinDays(vector<int>& weights, int D) {
        int maxW = weights[0], sumW = 0;
        for (auto w : weights) {
            maxW = max(maxW, w);
            sumW += w;
        }
        

        int lo = maxW, hi = sumW;
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            //cout << lo << " " << hi << " " << mid << endl;
            if (testWeight(weights, D, mid)) {
                hi = mid;
            } else {
                lo = mid+1;
            }
        }
        return lo;
    }
};
