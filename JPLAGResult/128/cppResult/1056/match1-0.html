class Solution {
public:
    int shipWithinDays(vector<int>& vec, int mx) {
        int l=0,r=1000000000;
        while (l+1!=r)
        {
            int mid=(l+r)/2,ths=0,ned=0;
            for (int i=0;i<vec.size();i++)
            {
                if (vec[i]>mid) ned=1e9;
                if (ths+vec[i]>mid) {ths=0;ned++;}
                ths+=vec[i];
            }
            if (ths) ned++;
            if (ned<=mx) r=mid;else l=mid;
            //cout<<mid<<endl;cerr<<mid<<endl;
        }
        return r;
    }
};
