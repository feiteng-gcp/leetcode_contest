class Solution {
    typedef long long LL;
    int solve(LL val, vector<int>& weights) {
        int d = 0;
        LL cur = 0;
        for (int weight : weights) {
            if (weight > val) return INT_MAX;
            if (cur + weight > val) {
                d++;
                cur = weight;
            } else {
                cur += weight;
            }
        }
        if (cur != 0) d++;
        return d;
    }
public:
    int shipWithinDays(vector<int>& weights, int D) {
        LL a = 1, b = INT_MAX;
        while (a < b) {
            LL mid = (a + b) >> 1;
            if (solve(mid, weights) > D) {
                a = mid + 1;
            } else {
                b = mid;
            }
        }
        return a;
        
    }
};
