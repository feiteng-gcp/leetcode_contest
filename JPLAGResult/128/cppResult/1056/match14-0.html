class Solution {
public:
    int shipWithinDays(vector<int>& weights, int D) {
        int left=0, right=25000000, n=weights.size();
        
        int ans=25000000;
        while(left<=right){
            int mid=(left+right)/2;
            int days=1, rate=0;
            for (int i=0; i<n; i++){
                if (weights[i]>mid) { days=D+1; break; }
                if (rate+weights[i]>mid){ ++days; rate=0; --i; } else rate+=weights[i];
            }
            if (days<=D){
                right=mid-1;
                ans=min(mid, ans);
            }else left=mid+1;
        }
        
        return ans;
    }
};
