class Solution {
public:
    
    vector < int > w;
    int d;
    
    bool check(int x) {
        int cur = 0;
        int days = 1;
        for (int i=0;i<w.size();i++) {
            if (w[i] > x) return false;

            if (cur + w[i] <= x) {
                cur += w[i];
            } else {
                days++;
                cur = w[i];
            }
        }        
        return days <= d;
    }
    
    int shipWithinDays(vector<int>& weights, int D) {
        w = weights;
        d = D;
        int l = 0, r = 1000 * 1000 * 1000;  
        int ans = r;

        while (l <= r) {
            int mid = (l + r)/2;
            if (check(mid)) {
                ans = mid;
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        return ans;
    }
};
