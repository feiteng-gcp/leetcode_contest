class Solution {
public:
    int cal(vector<int>& weights, int bag) {
        int ret = 1;
        int res = weights[0];
        for (int i = 1;i < weights.size();i++) {
            if (res + weights[i] > bag) {
                ret++;
                res = weights[i];
            } else {
                res += weights[i];
            }
        }
        return ret;
    }
    int shipWithinDays(vector<int>& weights, int D) {
        int min = 0;
        int max = 0;
        for (int i = 0;i < weights.size();i++) {
            max += weights[i];
            min = min > weights[i] ? min : weights[i];
        }
        min--;
        while (max - min > 1) {
            int temp = (max + min) / 2;
            if (cal(weights, temp) <= D) {
                max = temp;
            } else {
                min = temp;
            }
        }
        return max;
    }
};
