class Solution {
    int ck(vector<int>&v,int w)
    {
        int r = 0, s = 0;
        for (int x : v)
            if (s + x <= w)
                s += x;
            else {
                r++;
                s=x;
            }
        return r + (s != 0);
    }
public:
    int shipWithinDays(vector<int>& weights, int D) {
        int lo = *max_element(begin(weights),end(weights)), hi = 500 * 50000;
        while (lo <= hi) {
            int mi = (lo + hi) / 2;
            if (ck(weights,mi) <= D)
                hi = mi - 1;
            else
                lo = mi + 1;
        }
        return lo;
    }
};
