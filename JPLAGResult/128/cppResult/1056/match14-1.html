class Solution {
public:
    int shipWithinDays(vector<int>& w, int D) {

        int l = 1, r = 500 * 50000, ret = r;
        while(l<=r){
            int m = (l+r)>>1;
            int c = 0, cur = m;
            for(int i=0;i<w.size();i++){
                if(w[i]>m){
                    c = D + 1;
                    break;
                }
                if(cur+w[i]>m){
                    c++;
                    cur = w[i];

                } else cur+=w[i];
            }
            if(c<=D){
                ret=m;
                r=m-1;
            } else l = m+1;
        }
        return ret;
    }
};
