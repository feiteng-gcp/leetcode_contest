class Solution {
public:
	int shipWithinDays(vector<int>& weights, int D) {
		int minw = 0;
		int maxw = 0;
		for (int i = 0; i < weights.size(); i++)
		{
			minw = max(minw, weights[i]);
			maxw += weights[i];
		}
		if (D == 1)
			return maxw;
		int l = minw, r = maxw;
		while (l < r)
		{
			int mid = l + (r - l) / 2;

			int days = getDays(weights, mid);
			if (days > D)
			{
				l = mid + 1;
			}
			else
			{
				r = mid;
			}
		}
		return l;
	}

private:
	int getDays(vector<int>& wts, int mid) {

		int days = 0;
		int sum = 0;
		for (int i = 0; i < wts.size(); i++)
		{
			sum += wts[i];
			if (sum > mid)
			{
				sum = wts[i];
				days++;
			}
			else if (sum == mid)
			{
				days++;
				sum = 0;
			}
	
			if (sum > 0 && i == wts.size() - 1)
					days++;

		}
		return days;
	}
};
