class Solution {
    int ok(vector<int>& w, int ww) {
        int res = 1, cur = 0;
        
        for (int i = 0; i < w.size(); ++i) {
            if (cur + w[i] <= ww) {
                cur += w[i];
            } else {
                res++;
                cur = w[i];
            }
        }
        
        return res;
    }
public:
    int shipWithinDays(vector<int>& w, int d) {
        int l = 0, r = (int)1e9;
        
        for (int i : w) l = max(l, i);
        
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (ok(w, mid) <= d) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        
        return l;
    }
};
