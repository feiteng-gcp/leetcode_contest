class Solution {
public:
    bool canShip(vector<int>& weights, int D, int load){
        int n=weights.size();
        int i, x=0;
        for(i=0;i<D;i++)
        {
            int sum=0;
            while(sum+weights[x]<=load)
            {
                sum+=weights[x];
                x++;
                if(x==n)
                    return true;
            }
        }
        return false;
    }
    
    int shipWithinDays(vector<int>& weights, int D) {
        int lo=0, hi=25000001;
        while(lo<hi-1)
        {
            int mi=(lo+hi)/2;
            if(canShip(weights, D, mi))
                hi=mi;
            else
                lo=mi;
        }
        return lo+1;
    }
};
