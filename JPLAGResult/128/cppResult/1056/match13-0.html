class Solution {
 public:
  int shipWithinDays(vector<int>& weights, int D) {
    int low = *max_element(weights.begin(), weights.end());
    int high = 50000 * 500;
    while (low < high) {
      int mid = (low + high) / 2;
      int days = 1, total = 0;
      for (int w : weights) {
        if ((total += w) > mid) {
          ++days;
          total = w;
        }
      }
      if (days <= D) {
        high = mid;
      } else {
        low = mid + 1;
      }
    }
    return low;
  }
};
