class Solution {
public:
    bool check(vector<int>& weights, int D, int C) {
        int cnt = 0;
        int now = 0;
        for(auto e: weights) {
            if (e > C) return false;
            if(e > now) {
                cnt++;
                now = C;
            }
            now -= e;
        }
        return cnt <= D;
    }
    
    int shipWithinDays(vector<int>& weights, int D) {
        int l = 1, r = 100000000;
        while(l < r) {
            int mid = l + (r-l) / 2;
            if (check(weights, D, mid)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return l;
    }
};
