class Solution {
public:
    int days(vector<int>& ws, int cap) {
        int res = 0;

        int cur = 0;
        for (auto& w: ws) {
            if (cur + w <= cap) {
                cur += w;
            } else {
                cur = w;
                ++res;
            }
        }
        if (cur > 0) {
            ++res;
        }
        return res;
    }
    

    int shipWithinDays(vector<int>& weights, int D) {
        int mn = 0, mx = 0;
        for (auto& n: weights) {
            mn = max(mn, n);
            mx += n;
        }
        while (mn < mx) {
            int md = (mn + mx) / 2;

            int ds = days(weights, md);
            if (ds > D) {
                mn = md + 1;
            } else {
                mx = md;
            }
        }
        return mx;
    }
};
