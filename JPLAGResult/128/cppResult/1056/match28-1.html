class Solution {
public:
    bool canbe(int m, vector<int>& w, int D) {
        int d = 1;
        int s = 0;
        for (auto it: w) {
            if (it > m) return false;
            if (s + it > m) {
                s = it;
                d++;
            } else {
                s += it;
            }

        }
        return d <= D;
    }
    int shipWithinDays(vector<int>& w, int D) {
        int b = 1;
        int e = (int)1 << 30;
        while (b < e) {

            int m = (b + e) / 2;
            bool can = canbe(m, w, D);
            if (can) {
                e = m;
            } else {
                b = m + 1;
            }
        }
        return e;
    }
};
