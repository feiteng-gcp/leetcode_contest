class Solution {
public:
    bool test( int ability  , vector<int> & weights , int D )
    {
        int group = 0;
        int l = weights.size();
        for(int i = 0 ; i < l ; )
        {
            int c = ability;
            if( weights[i] > ability ) return false;
            while(c > weights[i])
            {
                c -= weights[i];
                i++;
                if( i >= l ) break;
            }
            group++;
            if(group > D ) return false;
        }
        return true;
    }
    int shipWithinDays(vector<int>& weights, int D) {
        int len = weights.size();
        int r = len * 500 + 1;
        int l = 1 ;
        int mid;
        while(l < r)
        {
            mid = ( l + r ) >> 1;
            bool t = test( mid , weights ,  D );
            if(t)
            {
                r = mid;
            }else{
                l = mid + 1;
            }
        }
        return l - 1;
    }
};
