class Solution {
public:
    int calcDays(vector<int>& weights, int cap) {
        int ans = 1;
        int cost = 0;
        for (auto w: weights) {

            if (w > cap) {
                return 214748364;
            }
            if (cost + w > cap) {
                ans++;
                cost = w;
            } else {
                cost += w;
            }
        }
        return ans;
    }
    int shipWithinDays(vector<int>& weights, int D) {
        int sum = 0;
        for (auto w : weights) {
            sum += w;
        }
        int small = 0, large = sum;

        int ans = 0;
        while (small <= large) {
            int mid = (small + large) / 2;
            if (calcDays(weights, mid) > D) {
                small = mid + 1;
            } else {
                ans = mid;
                large = mid - 1;
            }
        }
        return ans;
    }
};
