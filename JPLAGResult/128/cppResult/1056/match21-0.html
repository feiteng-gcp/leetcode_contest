class Solution {
public:
    int shipWithinDays(vector<int>& weights, int D) {
        int size = weights.size();
        int l = weights[0];
        for(int i=1; i<size; i++) l = max(l, weights[i]);
        int r=500*50000;
        int ans = -1;
        while(l <= r){
            int mid = (l+r)>>1;
            int c = 1, tmp = weights[0];
            for(int i=1; i<size; i++){
                int x = weights[i];
                if(tmp + x > mid){
                    tmp = x;
                    c ++ ;
                }else{
                    tmp +=x;
                }
                if(c > D) break;
            }
           // cout<<l<<'\t'<<r<<'\t'<<mid<<'\t'<<c<<endl;
            if(c > D){
                l = mid + 1;
            }else{
                ans = mid;
                r = mid - 1;
            }
        }
        return ans;
    }
};
