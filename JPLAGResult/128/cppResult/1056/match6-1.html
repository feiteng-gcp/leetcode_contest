class Solution {
public:
    bool canShip(vector<int> &weights, int D, int capacity) {
        int days = 1;
        int space = capacity;

        for (int w : weights) {
            if (w > capacity) {
                return false;
            }
            if (w > space) {
                days++;
                space = capacity - w;
            } else {
                space -= w;
            }
        }
        return days <= D;
    }
    
    int shipWithinDays(vector<int>& weights, int D) {
        int low = 0, high = 60000;

        while (low < high) {
            int mid = (low + high) / 2;
            if (canShip(weights, D, mid)) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        return low;    
    }
};
