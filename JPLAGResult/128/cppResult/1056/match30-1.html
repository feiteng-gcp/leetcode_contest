class Solution {
public:
    int shipWithinDays(vector<int>& weights, int D) {
        int l = *max_element(weights.begin(), weights.end());

        int r = accumulate(weights.begin(), weights.end(), 0);
        
        while(l < r)
        {
            int mid = (l+r)/2;
            
            if(can(weights, mid, D))
            {
                r = mid;
            }
            else
            {
                l = mid + 1;
            }
        }
        
        
        return l;
    }
    
    bool can(vector<int>& weights, int limit, int D)
    {
        int res = 1;
        int cur = 0;
        
        for(int w: weights)

        {
            if(cur + w <= limit)
            {
                cur += w;
            }
            else
            {
                cur = w;
                res++;
            }
        }
        
        return res <= D;
    }
};
