class Solution {
public:
    bool check(vector<int>& w, int d, int m) {
        int n = w.size();

        int cnt = 1;
        int s = 0;
        for (int i = 0; i < n; ++i) {
            if (w[i] > m) return false;
            s += w[i];
            if (s > m) {
                s = w[i];
                cnt++;
            }
        }

        return cnt <= d;
    }
    
    int shipWithinDays(vector<int>& weights, int D) {
        int n = weights.size();
        int s = 0;
        for (int i = 0; i < n; ++i) {
            s += weights[i];
        }
        int low = 1;

        int high = s;
        while (low + 1 < high) {
            int mid = low + (high - low) / 2;
            if (check(weights, D, mid)) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
        if (check(weights, D, low)) return low;
        return high;
    }
};
