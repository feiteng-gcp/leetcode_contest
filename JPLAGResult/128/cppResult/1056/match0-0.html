class Solution {
public:
    bool legal(vector<int>& weights , int  D, int mid  ){
        int days  = 1 ; 
        int now = 0 ; 
        
        for( auto & x: weights ){
            now += x ; 
            if( now > mid ){
                days ++ ; 
                now = x;  
            }
            if( x > mid ) 
                return false ; 
        }
        return days <= D ; 
    }
    int shipWithinDays(vector<int>& weights, int D) {
        int left = 0 ; 
        int right = INT_MAX ; 
        while( left <= right ){
            int mid = ( right - left ) /2 + left ; 
            if( legal( weights , D , mid ) ) {
                right = mid -1 ; 
            }
            else 
                left = mid+1 ; 
        }
        return left ; 
    }
};
