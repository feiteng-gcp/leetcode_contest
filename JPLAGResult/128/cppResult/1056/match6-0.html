class Solution {
public:
		vector<int> wts;
		int days;
    int shipWithinDays(vector<int>& weights, int D) {
			wts = weights;
			days = D;
			return binsearch(0,50001);
    }
		bool check(int x){
			int tot = 1;
			int su = 0;
			for(auto i : wts ){
				if( i > x ){ return false; }
				if( su + i > x ){ tot++; su = i; }
				else{ su += i; }
			}
			return (tot <= days);
		}
		int binsearch(int lo, int hi){
			while( lo < hi ){
				int mid = ( lo + hi )/2;
				if( check(mid) == false ){ lo = mid + 1; }
				else{ hi = mid ; }
			}
			return lo;
		}
};
