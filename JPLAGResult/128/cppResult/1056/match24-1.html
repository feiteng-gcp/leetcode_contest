class Solution {
public:
    bool check(vector<int>& weights, int D, int bud)
    {
        if (bud < 0)
            return false;
        int n = weights.size();

        int cnt, now;
        cnt = 0;
        now = 0;
        for (int i = 0; i < n; i++)
        {
            if (weights[i] > bud)
                return false;
            if (now + weights[i] <= bud)
                now += weights[i];
            else
            {
                now = weights[i];
                cnt++;
            }
        }
        if (now > 0)
            cnt++;

        return cnt <= D;
    }
    int shipWithinDays(vector<int>& weights, int D) {
        int l, r, mid;
        l = 0, r = 2147483647;
        while (l <= r)
        {
            mid = (l + r) >> 1;
            if (!check(weights, D, mid))
                l = mid + 1;
            else
                r = mid - 1;
        }
        if (check(weights, D, l - 1))
            return l - 1;
        return l;
    }
};
