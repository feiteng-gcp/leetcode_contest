#include <vector>
#include <string>
#include <unordered_map>
#include <queue>
#include <stack>
#include <list>
#include <iostream>

using namespace std;

class Solution
{
    int duration(const vector<int>& weights, int cap)
    {
        int res = 0;
        int ps = 0;
        for (int w : weights)
        {
            if (ps + w <= cap)
            {
                ps += w;
            }
            else
            {
                ++res;
                ps = w;
            }
        }
        return res + 1;
    }

public:
    int shipWithinDays(vector<int>& weights, int D)
    {
        int l = *std::max_element(weights.begin(), weights.end());
        int r = accumulate(weights.begin(), weights.end(), 0);
        while (l < r)
        {
            int mid = l + (r - l) / 2;
            int dur = duration(weights, mid);
            if (dur <= D)
            {
                r = mid;
            }
            else
            {
                l = mid + 1;
            }
        }
        return l;
    }
};
