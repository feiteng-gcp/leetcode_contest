class Solution {
public:
    int shipWithinDays(vector<int>& weights, int D) {
        int l0 = 0;
        int u0 = 500 * 50000;
        while (l0+1<u0) {
            int mid = (l0+u0)/2;
            int ac = 1;
            int nw = 0;
            for (auto x : weights) {
                if (x>mid) {
                    ac = D+1;
                    break;
                }
                if (nw+x <=mid) {
                    nw +=x;
                }

                else {
                    ac+=1;
                    nw = x;
                }
            }
            if (ac<=D) {
                u0 = mid;
            }
            else {
                l0 = mid;
            }
        }
        return u0;
    }
};
