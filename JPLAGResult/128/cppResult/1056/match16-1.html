class Solution {
public:
    int shipWithinDays(vector<int>& weights, int D) {
        int low = 1;
        int high = 50001;
        while(low<high){
            int mid = low+(high-low)/2;
            if(check(weights, D, mid)){ //mid is large enough, try smaller.
                high = mid;
            }
            else{
                low = mid+1; //mid too small
            }
        }
        return high; //because you may set high to valid mid.
    }
    
    bool check(vector<int> &weights, int D, int cap){
        int day = 1;
        int sum = 0;
        for(int w: weights){  //e.g. 5, 5, 5 cap =7, init day = 1, add 5, try to add second 5, day = 2, sum=5, try to add third, day=3, sum=5
            if(w>cap)return false;
            if(sum+w>cap){
                day++;
                sum=w;
            }
            else{
                sum+=w;
            }
            if(day>D)return false;
        }
        return true;
        //return day;
    }
};
