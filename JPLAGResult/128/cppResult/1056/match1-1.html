class Solution {
public:
    int shipWithinDays(vector<int>& weights, int D) {
        int L = 0, R = 30000000;
        while(L + 1 < R){
        	int M = (L + R + 1) / 2, d = 0, s = 0;
        	for(int x : weights){
        		if(x > M){
        			d = D + 1;
        			break;
				}

				if(s + x > M){
					s = 0;
					d += 1;
				}
				s += x;
			}
			if(s) d += 1;
			if(d > D) L = M;
			else R = M;
		}
		return R;
    }
};
