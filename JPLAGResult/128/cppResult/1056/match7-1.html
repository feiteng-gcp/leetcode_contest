#include <bits/stdc++.h>
using namespace std;


class Solution {
    bool ok(int w, vector<int>& weights, int D) {
        int cur = 0;
        int cnt = 0;
        for (int a : weights) {
            if (a > w) return false;
            if (cur + a > w) {
                cnt++;
                cur = a;
            } else {

                cur += a;
            }
        }
        if (cur) cnt++;
        return cnt <= D;
    }
public:
    int shipWithinDays(vector<int>& weights, int D) {

        int l = 0, r = 1e9;
        while (l < r - 1) {
            int m = (l + r) / 2;
            if (ok(m, weights, D)) {
                r = m;
            } else {
                l = m;
            }
        }
        return r;
    }
};
