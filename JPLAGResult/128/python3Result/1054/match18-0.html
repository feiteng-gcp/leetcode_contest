class Solution:
    def bitwiseComplement(self, N: int) -> int:
        if N == 0: return 1
        binary = []
        while N > 0:
            binary.append(N % 2)
            N //= 2
        # print(binary)
        ans = 0
        factor = 1
        for d in binary:
            if d == 0: ans += factor
            factor *= 2
        return ans
