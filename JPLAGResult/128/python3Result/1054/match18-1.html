class Solution:
    def bitwiseComplement(self, N: int) -> int:
        if N == 0:
            return 1
        arr = []
        while N>0:
            arr.append(0 if (N&1)==1 else 1)
            N//=2
        res = 0
        #print(arr)
        arr.reverse()
        for i in arr:
            #print(i)
            res*=2
            res+=i
            #print(res)
        return res
