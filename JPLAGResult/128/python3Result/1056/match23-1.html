class Solution:
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        if not weights:
            return 0
        minRes = max(weights)
        maxRes = sum(weights)
        
        def valid(maxW):
            d = D
            cur = 0
            for w in weights:
                if cur + w > maxW:
                    d -= 1
                    cur = w
                else:
                    cur += w
                    
            
return d > 0
        
        while maxRes > minRes:
            mid = (maxRes + minRes) // 2
            if valid(mid):
                maxRes = mid
            else:
                minRes = mid+1
                
        return maxRes
