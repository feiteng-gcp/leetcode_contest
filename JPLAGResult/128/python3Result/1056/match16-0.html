class Solution:
    def shipWithinDays(self, W: List[int], D: int) -> int:
        if len(W) <= D:
            return max(W)
        
        def valid(cap):
            cnt = 0
            total = 0
            
            for w in W:
                total += w
                
                if total > cap:
                    total = w
                    cnt += 1
                
                    if cnt >= D:
                        return False
            
            return True
        
        i, j = max(W), sum(W)
        
        while i < j:
            k = (i + j) >> 1
            
            if valid(k):
                j = k
            else:
                i = k + 1
        
        return i
