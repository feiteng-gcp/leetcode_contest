class Solution:
    
def shipWeight(self,K,weight,D):
        days = 1
        curr = K
        for i in weight:
            if i<=curr:
                curr-=i
            else:
                curr = K-i
                days+=1
        return days<=D
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        l = max(weights)
        h = sum(weights)
        ans = h
        while(l<=h):
            m = (l+h)>>1
            if(self.shipWeight(m,weights,D)):
                ans = m
                h = m-1
            else:
                l = m+1
        return ans
