class Solution:
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        
        def possible(cap, d):
            re = cap
            for x in weights:
                if re < x:
                    re = cap
                    d -= 1
                    if d <= 0: return False
                re -= x
            return True
        
        M = max(weights)
        l, r 
= M, M * len(weights)
        while l < r:
            mid = (l + r) // 2
            if possible(mid, D):
                r = mid
            else:
                l = mid + 1
        return l
