class Solution:
    def isOk(self, weights, D, mid):
        d = 0
        i = 0
        w = 0
        for neww in weights:
            if mid < neww:
                return False
            if (neww + w) > mid:
                d += 1
                if d > D:
                    return False
                w = neww
            else:
                w += neww
        d += 1
        if d > D:
            return False
        return True            
    
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        lo = 1
        hi = 50000
        while lo < hi:
            mid = (lo + hi) // 2
            ok = self.isOk(weights, D, mid)
            # print(mid, ok)
            if ok:
                hi = mid
            else:
                lo = mid+1
        return lo
