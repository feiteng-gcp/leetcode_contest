class Solution:
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        N = len(weights)
        totalWeights = sum(weights)
        minCap = math.ceil(totalWeights / D)
        maxCap = totalWeights
        
        # Return if we can load all cargos in order within D days, given max capacity
        def canLoad(weights, D, maxCap):
            i = 0
            days = 0
            
            while i < N and days < D:
                cap = 0
                while i < N and cap + weights[i] <= maxCap:
                    cap += weights[i]
                    i += 1
                    
                days += 1
                
            return i == N
        
        while minCap < maxCap:
            midCap = (minCap + maxCap) // 2
            
            if canLoad(weights, D, midCap):
                maxCap = midCap
            else:
                minCap = midCap + 1
                
        return minCap
