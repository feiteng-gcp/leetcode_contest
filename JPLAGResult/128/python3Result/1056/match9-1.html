class Solution:
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        def Ship(cap):
            result = 1
            cur = 0
            for w in weights:
                cur += w
                if cur > cap:
                    result += 1
                    cur = w
            return result
        begin = max(weights)
        end = sum(weights)
        while begin < end:
            mid = (begin + end) // 2
            days = Ship(mid)
            if days > D:
                begin = mid + 1
            elif days <= D:
                days1 = Ship(mid - 1)
                if days1 > D:
                    begin = end = mid
                else:
                    end = mid - 1
        return begin
