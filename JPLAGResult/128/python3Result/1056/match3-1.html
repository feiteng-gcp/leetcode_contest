class Solution:
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        def valid(load):
            days, cur = 0, 0
            
            for i, w in enumerate(weights):
                if w + cur > load:
                    days += 1
                    cur = 0
                    
                cur += w
                
                if days > D:
                    return False
            
            if cur: days += 1
            
return days <= D
        
        lo, hi = max(weights), sum(weights)
        
        while lo <= hi:
            mid = (lo + hi) // 2
            
            if valid(mid):
                hi = mid - 1
            else:
                lo = mid + 1
                
        return hi + 1
