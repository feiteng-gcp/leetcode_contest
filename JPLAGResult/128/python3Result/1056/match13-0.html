class Solution:
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        def auxi(p):
            a = 0
            t = 1
            for w in weights:
                a += w
                if a > p:
                    t+=1
                    a = w
            return t
        l, h = max(weights), sum(weights)
        while h > l:
            mid = (h+l)//2
            # print(l,h, auxi(mid))
            if auxi(mid) > D:
                l = mid +1
            else:
                h = mid
        return h
