class Solution:
    def check(self, n) -> bool:
        cur = n
        cd = 1
        for w in self.weights:
            if cur >= w:
                cur -= w
            else:
                cd += 1
                cur = n - w
        return cd <= self.D
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        self.D = D
        self.weights = weights
        l, r = max(weights), sum(weights)
        ans = r
        while l<=r:
            mid = (l+r)//2
            if self.check(mid):
                ans = mid
                r = mid - 1
            else:
                l = mid + 1
        return ans
