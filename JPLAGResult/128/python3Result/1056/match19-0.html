class Solution:
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        
        def GetDays(Capacity):
            N = 0
            cur = 0
            if Capacity < max(weights):
                return D+1
            for weight in weights:
                if cur + weight > Capacity: # Unload!
                    cur = 0
                    N += 1
                cur += weight
            return N+1
        
        left, right = 1, sum(weights)
        while left < right:
            mid = (left + right)//2
            if GetDays(mid) <= D:
                right = mid
            else:
                left = mid + 1
        return left
