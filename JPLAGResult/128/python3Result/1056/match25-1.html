class Solution:
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        length = len(weights)
        total = sum(weights)
        l = total // length
        r = total
        
        def isValid(limit, weights, D):
            count = 0
            total = 0
            for w in weights:
                if w > limit:
                    return False
                if total + w > limit:
                    count += 1
                    total = w
                else:
                    total += w
            if total != 0:
                count += 1
            
return count <= D
                    
        while l < r:
            m = l + (r - l) // 2
            if isValid(m, weights, D):
                r = m
            else:
                l = m + 1
        return l
