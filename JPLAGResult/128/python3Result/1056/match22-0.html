class Solution:
    def shipWithinDays(self, weights: List[int], D: int) -> int:
        def ok(ws):
            cd = 0
            cw = 0
            for w in weights:
                if w > ws:
                    return False
                if w + cw > ws:
                    cd += 1
                    if cd > D:
                        return False
                    cw = w
                else:
                    cw += w
            if cw:
                cd += 1
            return cd <= D
        
        i = 1
        j = sum(weights)
        while i < j:
            m = (i + j) // 2
            if ok(m):
                j = m
            else:
                i = m + 1
        return i
