class Solution:
    def shipWithinDays(self, weights, D: int) -> int:
        
def helper(midw):
            res = t = 0
            for w in weights:
                if t >= w:
                    t -= w
                else:
                    res += 1
                    t = midw - w
            return res
        
        minw, maxw = max(weights), sum(weights)
        while minw < maxw:
            midw = (minw + maxw + 1) // 2
            if helper(midw) <= D:
                maxw = midw
            else:
                minw = midw
            if maxw == minw + 1:
                return minw if helper(minw) <= D else maxw
        return minw
