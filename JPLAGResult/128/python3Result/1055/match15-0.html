class Solution:
    def numPairsDivisibleBy60(self, time: List[int]) -> int:
        nums = [0] * 60
        for t in time:
            nums[t%60] += 1
        res = nums[0] * (nums[0] - 1) // 2 if nums[0] > 1 else 0
        res += nums[30] * (nums[30] - 1) // 2 if nums[30] > 1 else 0
        for i in range(1,30):
            if nums[i] > 0 and nums[60-i] > 0:
                res += nums[i] * nums[60-i]
        return res
