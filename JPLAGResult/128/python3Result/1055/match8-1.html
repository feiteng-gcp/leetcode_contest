class Solution:
    def numPairsDivisibleBy60(self, time: List[int]) -> int:
        di = [0 for x in range(60)]
        for item in time:
            di[item%60] += 1
        ans = 0
        ans += di[0] * (di[0]-1)/2
        ans += di[30] * (di[30]-1)/2
        for n in range(1,30):
            ans += di[n]*di[60-n]
        return int(ans)
