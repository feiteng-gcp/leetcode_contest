class Solution:
    def numPairsDivisibleBy60(self, time: List[int]) -> int:
        cnt = [0]*60
        for x in time:
            cnt[x % 60] += 1
        ans = cnt[0]*(cnt[0]-1)//2 + cnt[30]*(cnt[30]-1)//2
        for i in range(1,30):
            ans += cnt[i]*cnt[60-i]
        return ans
