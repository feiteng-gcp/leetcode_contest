from collections import defaultdict
class Solution:
    def numPairsDivisibleBy60(self, time: List[int]) -> int:
        d = defaultdict(int)
        for i, dur in enumerate(time):
            d
[dur % 60] += 1
        cnt = 0
        for i in range(30):
            cnt += d[i] * d[60-i]
        cnt += d[30] * (d[30]-1) // 2
        cnt += d[0] * (d[0]-1) // 2
        return cnt
