from collections import defaultdict
class Solution:
    def numPairsDivisibleBy60(self, time: List[int]) -> int:
        dct = defaultdict(int)
        for i in time:
            dct[i % 60] += 1
        res = 0
        res += dct[0] * (dct[0]-1)
        del dct[0]
        res += dct[30] * (dct[30]-1)
        del dct[30]
        for key in dct:
            if(60-key in dct):
                res += dct[key] * dct[60-key]
        return res//2
