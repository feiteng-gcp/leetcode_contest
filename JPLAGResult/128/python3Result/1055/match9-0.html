class Solution:
    def numPairsDivisibleBy60(self, time: List[int]) -> int:
        ans = 0
        hashmap = collections.defaultdict(int)
        for t in time:
            rem = t % 60
            if rem != 0:
                ans += hashmap[60-rem]
            else:
                ans += hashmap[0]
            hashmap[rem] += 1
        return ans
