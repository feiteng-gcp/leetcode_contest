class Solution:
    def numPairsDivisibleBy60(self, time: List[int]) -> int:
        a = [0 for i in range(60)]
        ans = 0
        for x in time:
            a[x % 60] += 1
        
for i in range(1, 30):
            ans += a[i] * a[60 - i]
        ans += a[0] * (a[0] - 1) // 2 + a[30] * (a[30] - 1) // 2
        return ans
