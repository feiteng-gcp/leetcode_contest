class Solution:
    
def numDupDigitsAtMostN(self, N: int) -> int:
        if N < 11:
            return 0
        s = str(N)
        r = 0
        n = len(s)
        for i in range(1, n):
            c = 9
            for j in range(1, i):
                c *= 10 - j
            r += c
        print(r)
        ss = set()
        dupl = False
        for i in range(n - 1):
            if s[i] != "0" and (s[i] != "1" or i != 0):
                c = 1
                print("n - i - 1",n - i - 1)
                for ii in range(n - i - 1):
                    c *= 9 - ii - i
                
                nn = 0
                for j in range(1 if i == 0 else 0, int(s[i])):
                    if str(j) not in ss:
                        nn += 1
                print('c=', c, "nn=", nn)
                r += c * nn
            print(r)
            if s[i] in ss:
                dupl = True
                break
            ss.add(s[i])
        if not dupl:
            for i in range(N % 10 + 1):
                if str(i) not in ss:
                    r += 1
        print(r)
        return N - r
