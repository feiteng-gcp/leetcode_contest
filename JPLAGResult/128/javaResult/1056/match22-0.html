class Solution {
    
    private int loaddays(int[] weights, int loads) {
        int days = 0;
        int curload = 0;
        for (int i = 0; i < weights.length; i++) {
            if (curload + weights[i] <= loads) {
                curload += weights[i];
            } else {
                days++;
                curload = weights[i];
            }
        }
        return days + 1;
    }
    
    public int shipWithinDays(int[] weights, int D) {
        int right = 0;
        int left = 0;
        for (int i = 0; i < weights.length; i++) {
            right += weights[i];
            left = Math.max(left, weights[i]);
        }
        
        while (left < right) {
            int mid = left + (right - left) / 2;
            int days = loaddays(weights, mid);
            if (days <= D) right = mid;
            else left = mid + 1;
        }
        
        return left;
    }
}
