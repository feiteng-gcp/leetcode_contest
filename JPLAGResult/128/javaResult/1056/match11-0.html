	class Solution {
	    public int shipWithinDays(int[] weights, int D) {
	        int low = 0, high = 50000*500*2;
	        while(high - low > 1){
	        	int h = high+low>>1;
	        	if(ok(h, weights, D)){
	        		high = h;
	        	}else{
	        		low = h;
	        	}
	        }
		    return high;
	    }
	    
	    boolean ok(int h, int[] w, int d)
	    {
	    	int rem = h;
	    	int ct = 1;
	    	for(int v : w){
	    		if(v > h)return false;
	    		if(rem - v >= 0){
	    			rem -= v;
	    		}else{
	    			ct++;
	    			rem = h-v;
	    		}
	    	}
	    	return ct <= d;
	    }
	}
