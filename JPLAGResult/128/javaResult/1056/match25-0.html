class Solution {
    public int shipWithinDays(int[] weights, int D) {
        if(weights == null || weights.length == 0) {
            return 0;
        }
        
        int maxLoad = 0;
        for(Integer weight : weights) {
            maxLoad += weight;
        }
        
        int minLoad = 1, midLoad = 0;
        while(minLoad < maxLoad - 1) {
            midLoad = minLoad + (maxLoad - minLoad)/ 2;
            if(canShip(weights, D, midLoad)) {
                maxLoad = midLoad;
            } else {
                minLoad = midLoad;
            }
        }
        
        if(canShip(weights, D, minLoad)) {
            return minLoad;
        }
        
        return maxLoad;
    }
    
    private boolean canShip(int[] weights, int days, int load) {
        int ship = 0, neededDays = 1;
        for(Integer weight : weights) {
            if(weight > load) {
                return false;
            }
            
            if(ship + weight > load) {
                ++neededDays;
                if(neededDays > days) {
                    return false;
                }
                ship = weight;
            } else {
                ship += weight;
            }
        }
        
        return true;
    }
}
