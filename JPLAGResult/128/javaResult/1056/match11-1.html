class Solution {
    public int shipWithinDays(int[] weights, int D) {
        int left = 1;
        int right = 25000001;
        
        while (left + 1 < right) {
            int mid = (left + right) / 2;
            if (valid(mid, D, weights)) {
                right = mid;
            } else {
                left = mid;
            }
        }
        
        if (valid(left, D, weights)) {
            return left;
        }
        
return right;
    }
    
    private boolean valid(int cap, int D, int[] weights) {
        int day = 0;
        int temp  = 0;
        for (int i = 0; i < weights.length; i++) {
            int cur = weights[i];
            
if (cur > cap) return false;
            if (temp + cur <= cap) {
                temp += cur;
            } else {
                day++;
                temp = cur;
            }
        }
        
        return day + 1 <= D;
    }
}
