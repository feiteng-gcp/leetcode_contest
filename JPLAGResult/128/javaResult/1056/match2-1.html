class Solution {
    public int shipWithinDays(int[] weights, int D) {
        
int max = 0;
        for(int w : weights) {
            max = Math.max(max, w);
        }
        
int left = max, right = Integer.MAX_VALUE;
        
        while(left < right) {
            int mid = left  + (right - left) / 2;
            
            if(helper(weights, D, mid)) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        
        return left;
        
    }
    
    private boolean helper(int[] arr, int d, int c) {
        int acc = 0;
        int count = 0;
        for(int a : arr) {
            if(acc + a > c) {
                acc = a;
                count++;
            } else {
                acc += a;
            }
        } 
        
        return count + 1 <= d;        
    } 
}
