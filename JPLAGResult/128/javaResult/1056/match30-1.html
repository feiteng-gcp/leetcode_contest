class Solution {
    public int shipWithinDays(int[] weights, int D) {
        int min=0;
        int max=0;
        for(int i=0;i<weights.length;i++){
            max=max+weights[i];
        }
        int res=max;
        while(min<=max){
            int mid=min+(max-min)/2;
            if(canDo(weights,mid,D)){
                res=Math.min(res,mid);
                
max=mid-1;
            }
            else{
                 min=mid+1;
            }
        }
        return res;
    }
    public boolean canDo(int[] weights,int load,int D){
        int count=1;
        int curr=0;
        for(int i=0;i<weights.length;i++){
            if(load<weights[i]){
                return false;
            }
            if(curr+weights[i]>load){
                curr=weights[i];
                count++;
            }
            else{
                
curr=curr+weights[i];
            }
        }
        return count<=D;
    }
}
