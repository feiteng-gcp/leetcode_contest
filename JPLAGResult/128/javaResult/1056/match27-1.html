class Solution {
    public int shipWithinDays(int[] weights, int D) {
        int sum = 0;
        for (int w : weights) sum += w
;
        int low = 1;
        int high = sum;
        while (low < high) {
            int m = low + (high - low) / 2;
            if (canShip(weights, D, m)) {
                high = m;
            } else {
                low = m + 1;
            }
        }
        
return low;
    }
    
    private boolean canShip(int[] weights, int d, int c) {
        int ret = 1, cur = 0;
        for (int w : weights) {
            if (w > c) return false
;
            if (cur + w > c) {
                ret++;
                cur = w;
            } else {
                cur += w;
            }
        }
        return ret <= d;
    }
}
