class Solution {
    public int shipWithinDays(int[] w, int d) {
        int n = w.length;
        int sum = 0;
        for (int i : w) {sum += i;}
        
int l = 0, r = sum;
        
        while (l + 1 < r) {
            int mid = l + (r - l)/2;
            if (couldShip(w, d, mid)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        
        if (couldShip(w, d, l)) {return l;}
        
return r;
    }
    
    boolean couldShip(int[] w, int d, int c) {
        int cnt = 0, sum = 0;
        for (int i = 0; i < w.length; i++) {
            if (w[i] > c) {return false;}
            if (sum + w[i] > c) {
                cnt++;
                
sum = w[i];
            } else {
                sum += w[i];
            }
        }
        return cnt + 1 <= d;
    }
}
