class Solution {
    public int shipWithinDays(int[] weights, int D) {
        int sum = 0;
        int max = 0;
        for (int weight :weights) {
            sum += weight;
            max = Math.max(weight, max);
        }
        int left = max;
        
int right = sum;
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (getDays(weights, mid) > D) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        return left;
    }
    private int getDays(int[] weights, int c) 
{
        int res = 1;
        int sum = 0;
        for (int weight : weights) {
            if (sum + weight > c) {
                res++;
                sum = weight;
            } else {
                sum += weight;
            }
        }
        return res;
    }
}
