class Solution {
    public int shipWithinDays(int[] weights, int D) {
        int l = weights[0];
        int r = 0;
        for(int x : weights){
            l = Math.max(l, x);
            r += x;
        }
        l--;
        while(l != r - 1){
            int mid = (l + r) / 2;
            if(ok(weights, D, mid))
{
                r = mid;
            } else {
                l = mid;
            }
        }
        return r;
    }
    
    private boolean ok(int[] weights, int D, int cap){
        int cnt = 1;
        int sum = 0;
        for(int x : weights){
            if(sum + x > cap){
                sum = x;
                cnt++;
                if(cnt > D){
                    return false;
                }
            } else {
                sum += x;
            }
        }
        return true;
    }
}
