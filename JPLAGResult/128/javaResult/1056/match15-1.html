class Solution {
    public int shipWithinDays(int[] weights, int D) {
        // binary search
        int lb = 0, ub = 0;
        for (int weight : weights) {
            lb = Math.max(lb, weight);
            ub += weight;
        }
        lb--;
        ub++;
        while (lb + 1 < ub) {
            int mid = lb + (ub - lb) / 2;
            if (computeDaysNeeded(weights, mid) <= D) 
{
                ub = mid;
            }
            else {
                lb = mid;
            }
        }
        return lb + 1;
    }
    
    private int computeDaysNeeded(int[] weights, int capacity) {
        
int res = 1;
        int sum = 0;
        for (int i = 0; i < weights.length; i++) {
            if (sum + weights[i] <= capacity) {
                sum += weights[i];
            }
            else {
                res++;
                sum = weights[i];
            }
        }
        return res;
    }
}
