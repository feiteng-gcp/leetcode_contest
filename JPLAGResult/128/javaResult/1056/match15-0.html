class Solution {
    public int shipWithinDays(int[] weights, int D) {
            int lo = 1, hi = (int) 1e9;
            for (int w : weights) {
                lo = Math.max(w, lo);
            }
            while (lo < hi) {
                int mid = (lo + hi) / 2;
                int use = 1;
                int cur = 0;
                for (int w: weights) {
                    if (cur + w <= mid) {
                        cur += w;
                    } else {
                        cur = w;
                        use++;
                    }
                }
                if (use <= D) {
                    hi = mid;
                } else {
                    lo = mid + 1;
                }
            }
            return lo;
        }
}
