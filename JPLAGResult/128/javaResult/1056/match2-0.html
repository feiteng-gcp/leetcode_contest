public class Solution {
	public int shipWithinDays(int[] weights, int D) {
		int maxW = 0, sumW = 0;
		for (int w : weights) {
			maxW = Math.max(maxW, w);
			sumW += w;
		}
		int lo = Math.max(maxW, (sumW + D - 1) / D), hi = sumW;
		while (lo < hi) {
			int mid = lo + (hi - lo) / 2;
			if (canShip(weights, mid, D)) {
				hi = mid;
			} else {
				lo = mid + 1;
			}
		}
		return lo;
	}

	private boolean canShip(int[] weights, int C, int d) {
		int sum = 0, cnt = 1;
		for (int w : weights) {
			if (sum + w > C) {
				cnt++;
				sum = w;
			} else {
				sum += w;
			}
		}
		return cnt <= d;
	}
}
