class Solution {
    public int shipWithinDays(int[] weights, int D) {
        int l = weights[0];
        int r = 0;
        for (int n : weights) {
            l = Math.max(l, n);
            r += n;
        }
        while (l < r) {
            int m = l + (r - l) / 2;
            int day = getDays(weights, m);
            if (day > D) 
{
                l = m + 1;
            } else {
                r = m;
            }
        }
        return l;
    }
    private int getDays(int[] weights, int m) {
        int cnt = 0;
        
int sum = 0;
        for (int w : weights) {
            if (sum + w <= m) {
                sum+= w;
            } else {
                cnt++;
                sum = w;
            }
        }
        if (sum > 0) cnt++;
        return cnt;
    }
}
