class Solution {
  public int shipWithinDays(int[] weights, int D) {
    int low = 1;
    int high = 0;
    
    for (int i : weights) {
      low = Math.max(low, i);
      high +=i;
    }
    
    while(low < high) {
      int mid = (low + high) / 2;
      
      if(isPossible(weights, D, mid)) {
        high = mid;
      } else {
        low = mid + 1;
      }
    }
    
    return low;    
  }
  
  private boolean isPossible(int[] weights, int D, int k) {
    int count = 1;
    int curr = 0;
    
    for (int i : weights) {
      curr+=i;
      if(curr > k) {
        count++;
        curr = i;
      }
    }
    
    return count <= D;
  }
}
