class Solution {
    public int shipWithinDays(int[] weights, int D) {
        int len = weights.length;
        
int start = weights[0];
        int end = 0;
        for (int i : weights) {
            end += i;
            start = Math.max(i, start);
        }
        while (start + 1 < end) {
            int mid = start + (end - start) / 2;
            if (tryShip(weights, mid) <= D) 
{
                end = mid;
            } else {
                start = mid;
            }
        }
        if (tryShip(weights, start) <= D)
            return start;
        
return end;
    }
    public int tryShip(int[] weights, int cap) {
        int days = 0;
        int cur = 0;
        for (int i = 0; i < weights.length; i++) {
            if (cur + weights[i] <= cap) {
                cur += weights[i];
            } else {
                days++;
                
cur = weights[i];
            }
        }
        days++;
        return days;
    }
}
