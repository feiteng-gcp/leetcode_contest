class Solution {
    public int shipWithinDays(int[] weights, int D) {
        int n = weights.length;
        
int min = weights[0];
        int max = 0;
        for (int i = 0; i < n; i++) {
            min = Math.max(min, weights[i]);
            max += weights[i];
        }
        int start = min;
        int end = max; 
        
while (start + 1 < end) {
            int mid = start + (end - start) / 2; 
            // System.out.println(start + " " + end + " " + mid + " " + canDo(mid, weights));
            if (canDo(mid, weights) <= D) {
                end = mid;
            } else {
                start = mid;
            }
        }
        // System.out.println(end + " " + canDo(end, weights));
        if (canDo(start, weights) <= D) {
            return start;     
        }
        if (canDo(end, weights) <= D) {
            return end;
        } else {
            return start;
        }
    
}
    
    private int canDo(int val, int[] weights) {
        
        int ret = 0; 
        int tmp = 0;
        for (int i = 0; i < weights.length; i++) {
            if (tmp + weights[i] <= val) {
                tmp += weights[i];
            } else {
                tmp = weights[i]; 
                ret++; 
            }
        }
        return ret + 1; 
    }
}
