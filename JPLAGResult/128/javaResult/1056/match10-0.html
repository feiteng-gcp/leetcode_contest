class Solution {
    public int shipWithinDays(int[] weights, int days) {
        //n log n 
        
        int low = weights[0];
        int high = 0;
        
        for(int i = 0; i < weights.length; i++){
            low = Math.min(low,weights[i]);
            high+=weights[i];
        }
        
        while(low < high){
            
            int mid = low+((high-low)/2);
            
            boolean valid = isValid(weights,mid,days);
            
            if(valid){
                high = mid;
            }else{
                low = mid+1;
            }
        }
        return low;
    }
    
    boolean isValid(int[] weights, int mid, int days){
        
        int numDays = 1;
        int sum = 0;
        
        for(int i  = 0; i < weights.length; i++){
            sum+=weights[i];
            if(sum > mid){
                sum = weights[i];
                numDays++;
                if(weights[i] > mid){
                    return false;
                }
            }
        }
        return numDays <= days;
    }
    
}
