class Solution {
    public int bitwiseComplement(int N) {
        if(N == 0) return 1;
        int ret = 0, count = 0;
        while(N > 0) {
            if((N & 1) == 0) ret += (1 << count);
            count++;
            N >>=1;
            //System.out.println(N);
        }
        return ret;
    }
}
