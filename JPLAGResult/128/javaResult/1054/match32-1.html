class Solution {
    public int bitwiseComplement(int N) {
        if (N == 0) return 1;
        String biRep = Integer.toBinaryString(N);
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < biRep.length(); i++) {
            if (biRep.charAt(i) == '0') {
                sb.append('1');
            } else {
                sb.append('0');
            }
        }
        return Integer.parseInt(sb.toString(), 2);
    }
}
