class Solution {
    public int bitwiseComplement(int N) {
        
        if (N == 0)
            return 1;
        
        int res = 0;
        int i = 0;
        
        while (N > 0) {
            int t = N & 1;
            if (t == 0)
                res += Math.pow(2, i);
            N = N >> 1;
            i++;
        }
        
        return res;
        
    }
}
