class Solution {
    public int bitwiseComplement(int N) {
        if (N == 0) {
            return 1;
        }
        int flag = 1, bit = 1;
        while(N >= bit) {
            flag = flag<<1;
            flag += 1;
            bit = bit << 1;
        }
        
        flag = flag >> 1;
        return N ^ flag;
    }
}
