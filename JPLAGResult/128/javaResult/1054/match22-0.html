class Solution {
    public int bitwiseComplement(int N) {
        if(N == 0) {
            return 1;
        }
        
        int result = 0;
        int shift = 0;
        int mask = 0x01;
        
        while(N > 0) {
            if((N & mask) == 0) {
                result += 1 << shift;
            }
            
            shift ++;
            N = N >> 1;
        }
        
        return result;
    }
}
