class Solution {
    public int bitwiseComplement(int N) {
      if (N ==0) {
	            return 1;
	        }
	        
	        String 
bit = Integer.toBinaryString(N);
	        StringBuffer sb = new StringBuffer();
	        for (int i = 0; i < bit.length(); i ++) {
	        	if (bit.charAt(i) == '1') {
	        		sb.append('0');
	        	}else {
	        		sb.append('1');
	        	}
	        }
	        return Integer.parseInt(sb.toString(), 2);
    }
}
