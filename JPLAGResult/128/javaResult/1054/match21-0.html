class Solution {
    public int bitwiseComplement(int N) {
        if (N==0) return 1;
        int power = 1;
        int result = 0;
        while(N>0) {
            if ((N & 1) == 0) {
                result += power;
            }
            power*=2;
            N = N>>1;
        }
        return result;
    }
}
