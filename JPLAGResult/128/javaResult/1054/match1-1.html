class Solution {
    public int bitwiseComplement(int N) {
        if (N == 0)
            return 1;
        int mask = 0;
        int temp = N;
        while (temp != 0) {
            temp >>= 1;
            mask <<= 1;
            mask |= 1;
        }
        return N ^ mask;
    }
}
