class Solution {
    public int bitwiseComplement(int N) {
        if (N == 0) return 1;
        int temp = N;
        int yihuo = 0;
        while (temp != 0) {
            temp >>= 1;
            yihuo <<= 1;
            yihuo += 1;
        }
        
        return yihuo ^ N;
    }
}
