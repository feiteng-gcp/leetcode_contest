class Solution {
    List<Integer> res = new ArrayList<>();
    boolean vis[] = new boolean[10];
    int n;

    int dfs(int idx, boolean max, boolean zero) {
        if (idx == n) {
            return 1;
        }
        int cnt = 0;
        if (max) {
            for (int i = 0; i <= res.get(idx); i++) {
                if (vis[i]) {
                    continue;
                }
                vis[i] = true;
                cnt += dfs(idx + 1, i == res.get(idx), false);
                vis[i] = false;
            }
        } else {
            for (int i = 0; i <= 9; i++) {
                if (vis[i]) {
                    continue;
                }
                if (!zero || i != 0) {
                    vis[i] = true;
                }
                cnt += dfs(idx + 1, false, zero && i == 0);
                vis[i] = false;
            }
        }
        return cnt;
    }

    public int numDupDigitsAtMostN(int N) {
        int ans = N + 1;
        while (N > 0) {
            res.add(0, N % 10);
            N /= 10;
        }
        n = res.size();
        for (int i = 0; i <= res.get(0); i++) {
            if (i > 0) vis[i] = true;
            ans -= dfs(1, i == res.get(0), i == 0);
            vis[i] = false;
        }
        return ans;
    }
}
