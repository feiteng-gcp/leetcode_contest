class Solution {
    public int numDupDigitsAtMostN(int N) {
        if (N <= 9) {
        	return 0;
        }
        String nStr = String.valueOf(N);
        int[] digits = new int[nStr.length()];
        for (int i = 0; i < nStr.length(); i++) {
        	digits[i] = Integer.valueOf(nStr.substring(i, i + 1));
        }
        int result = 0;
        int idx = nStr.length() - 1;
        int size = 1;
        // special handle
        digits[nStr.length() - 1]++;
        while (idx >= 0) {
        	Set<Integer> digitSet = new HashSet<>();
            for (int i = 0; i < idx; i++) {
            	digitSet.add(digits[i]);
            }
        	int curr = digits[idx];
        	
        	if (digitSet.size() == idx) {
        		// leading digits without repeats
        		int cnt = 0;
        		for (int d = (idx == 0) ? 1 : 0; d < curr; d++) {
        			if (!digitSet.contains(d)) {
        				cnt++;
        			}
        		}
        		int remaining = 10 - 1 - idx;
        		for (int i = idx + 1; i < nStr.length(); i++) {
        			cnt *= remaining;
        			remaining--;
        		}
        		result -= cnt;
        		result += curr * size;
        		if (idx == 0) {
        			result -= size;
        		}
        	} else {
        		// leading digits with repeats
        		result += curr * size;
        	}
        	size *= 10;
        	idx--;
        }
        int cnt = 90;
        for (int i = 2; i <= nStr.length() - 1; i++) 
{
        	int uni = 9 * 9;
        	int next = 8;
        	for (int j = 2; j < i; j++) {
        		uni *= next;
        		next--;
        	}
        	result += (cnt - uni);
        	cnt *= 10;
        }
        return result;
    }

}
