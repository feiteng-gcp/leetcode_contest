class Solution {
  public int numDupDigitsAtMostN(int N) {
    int[] digits = String.valueOf(N).chars().map(i -> i - (int)'0').toArray();
    int[][][] dp = new int[digits.length + 1][1024][2];
    dp[0][0][1] = 1;
    for (int i = 0; i < digits.length; i++) {
      dp[i + 1][0][0] = 1;
      for (int j = 1; j < 1024; j++) {
        if (used(j, digits[i])) {
          dp[i + 1][j][1] += dp[i][setUsed(j, digits[i], false)][1];
        }
        for (int k = j == 1 ? 1 : 0; k < 10; k++) {
          if (!used(j, k)) continue;
          dp[i + 1][j][0] += dp[i][setUsed(j, k, false)][0];
          if (k < digits[i]) {
            dp[i + 1][j][0] += dp[i][setUsed(j, k, false)][1];
          }
        }
      }
    }
    
int ans = 0;
    for (int i = 1; i < 1024; i++) {
      ans += dp[digits.length][i][0] + dp[digits.length][i][1];
    }
    return N - ans;
  }

  private boolean used(int state, int digit) {
    return (state & (1 << digit)) > 0;
  }

  private int setUsed(int state, int digit, boolean used) {
    if (used) return state | (1 << digit);
    return state & ~(1 << digit);
  }
}
