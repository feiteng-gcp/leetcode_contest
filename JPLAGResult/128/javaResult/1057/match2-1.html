class Solution {
    // 1567
    // null, 1000 :  1, 4; < x000
    // {1}, 500,  :  5, 3
    // {1, 5}, 60
    // {1, 5, 6}, 7
    public int numDupDigitsAtMostN(int N) {
        List<Integer> list = new ArrayList<>();
        int tmp = N;
        while(tmp != 0) {
            list.add(0, tmp % 10);
            tmp /= 10;
        }

        int sum = 0;
        Set<Integer> set =  new HashSet<>();
        for(int i = 0; i < list.size(); i++) {
            if(i != 0) {
                if(set.contains(list.get(i - 1))) {
                    break;
                } else {
                    set.add(list.get(i - 1));
                }

                sum += helper(set, list.get(i), list.size() - i);
            } else {
                sum += helper2(list.get(0), list.size());
            }


        }

        Set<Integer> set2 = new HashSet<>(list);
        if(set2.size() == list.size()) {
            return N - sum - 1;
        } else {
            return N - sum;
        }
    }

    private int helper2(int x, int len) {
        int sum = 0;
        int prod = x - 1;
        int cur = 9;
        for(int j = 1; j < len; j++) {
            prod *= cur;
            cur--;
        }

        sum += prod;

        for(int j = 1; j < len; j++) {
            prod = 1;
            cur = 9;
            for(int i = j; i < len; i++) {
                prod *= cur;
                if(i != j) {
                    cur--;
                }
            }

            sum += prod;
        
}

        return sum;
    }

    private int helper(Set<Integer> set, int num, int len) {
        if(num == 0) {
            return 0;
        }
        int cand = num;
        for(int i : set) {
            if(i < num) {
                cand--;
            }
        }
        if(cand == 0) {
            return 0;
        }


        int prod = cand;
        int cur = 10 - set.size() - 1;
        for(int i = 1; i < len; i++) {
            prod *= cur;
            cur--;
        }

        return prod;
    }
}
