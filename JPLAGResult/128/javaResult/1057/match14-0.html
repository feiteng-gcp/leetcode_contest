public class Solution {
    private static int K;
    private static int[] DIGITS;
    private int cnt = 0;
    public int numDupDigitsAtMostN(int N) {
        K = Integer.toString(N).length();
        DIGITS = new int[K];
        for (int i = 0, tmp = N; i < K; ++i) {
            DIGITS[K - 1 - i] = (tmp % 10);
            tmp /= 10;
        }
        int sum = 0;
        for (int i = 1; i < K; ++i) {
            int a = 9;
            for (int j = 0; j < i - 1; ++j)
                a *= (9 - j);
            sum += a;
        }
        Set<Integer> allow = new HashSet<>();
        for (int i = 0; i <= 9; ++i) allow.add(i);
        for (int start = 1; start < DIGITS[0]; ++start) {
            int a = 1;
            for (int j = 0; j < K - 1; ++j) {
                a *= (9 - j);
            }
            sum += a;
        }
        allow.remove(DIGITS[0]);
        sum += bound(K - 1, N - DIGITS[0] * (int)Math.pow(10, K - 1), allow);
        return N - sum;
    }

    private int bound(int bits, int upperBound, Set<Integer> choices) {
        if (bits == 0) return 1;
        if (upperBound >= (int)Math.pow(10, bits)) {
            int ret = 1;
            for (int i = 0, bb = choices.size(); i < bits; i++, bb--) {
                ret *= bb;
            }
            return ret;
        }
        Set<Integer> tmpSet = new HashSet<>(choices);
        int pos = 10 - choices.size(), ret = 0;
        for (int d0 : tmpSet) {
            if (d0 < DIGITS[pos]) {
                choices.remove(d0);
                ret += bound(bits -1 , Integer.MAX_VALUE, choices);
                choices.add(d0);
            } else if (d0 == DIGITS[pos]) {
                choices.remove(d0);
                ret += bound(bits - 1, upperBound - d0 * (int)Math.pow(10, bits - 1), choices);
                choices.add(d0);
            }
        }

        return ret;
    }
}
