class Solution {
    private int repNum(int digitNum, int len) {
        int ret = 1;
        for (int i = 0; i < len; ++i) {
            ret *= 10;
        }
        int noRep = 1;
        for (int i = 0; i < len; ++i) {
            noRep *= digitNum;
            digitNum -=1;
        }
        return ret - noRep;
    
}
    
    private int repNumInitNoZero(int len) {
        int total = 9;
        for (int i = 0; i < len - 1; ++i) {
            total *= 10;
        }
        int noRep = 9;
        int digitNum = 9;
        for (int i = 0; i < len - 1; ++i) {
            noRep *= digitNum;
            digitNum -=1;
        }
        return total - noRep;
    }
    
    public int numDupDigitsAtMostN(int N) {
        int len = 0;
        int n = N;
        while (n > 0) {
            len += 1;
            n /= 10;
        }
        int[] nArray = new int[len];
        n = N;
        for (int i = len - 1; i >= 0; --i) {
            nArray[i] = n % 10;
            n /= 10;
        }
        boolean[] prefixRep = new boolean[len];
        Set<Integer> set = new HashSet<>();
        set.add(nArray[0]);
        for (int i = 1; i < len; ++i) {
            prefixRep[i] = prefixRep[i-1] ? true : set.contains(nArray[i]);
            set.add(nArray[i]);
        }
        int result = 0;
        for (int i = 1; i < len; ++i) {
            result += repNumInitNoZero(i);
        }
        // 51673
        set = new HashSet<>();
        for (int i = 0; i < len; ++i) {
            for (int digit = ((i == 0) ? 1 : 0); digit < nArray[i]; ++digit) {
                if (set.contains(digit) || i>0 && prefixRep[i-1]) {
                    int tmp = 1;
                    for (int k = 0; k < len - i- 1; ++k) {
                        tmp *= 10;
                    
}
                    result += tmp;
                } else {
                    result += repNum(10 - i - 1, len - i - 1);
                }
            }
            set.add(nArray[i]);
        }
        if (prefixRep[len - 1]) result += 1;
        return result;
    }
}
