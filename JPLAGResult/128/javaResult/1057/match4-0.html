class Solution {
    int A(int n,int k) 
    {
        int x=1;
        for (int i=n;i>n-k;i--) x*=i;
        return x;
    }
    int count(int x)
    {
        return 9*A(9,x-1);
    }
    public int numDupDigitsAtMostN(int N) {
        if (N<=10) return 0;
        N++;
        String s=N+"";
        int l=s.length(),ans=N-1;
        for (int i=1;i<l;i++) ans-=count(i);
        System.out.println(ans);
        for (int i=0;i<l;i++)
        {
            int x=s.charAt(i)-48;
            for (int j=0;j<x;j++)
            {
                if (i==0 && j==0) continue;
                boolean bo=true;
                for (int k=0;k<i;k++) 
                    if (s.charAt(k)-48==j) bo=false;
                for (int k=0;k<i;k++)
                    for (int o=0;o<k;o++)
                        if (s.charAt(k)==s.charAt(o)) bo=false;
                if (bo) ans-=A(9-i,l-1-i);
        System.out.println(ans);
            }
        }
        return ans;
    }
}
