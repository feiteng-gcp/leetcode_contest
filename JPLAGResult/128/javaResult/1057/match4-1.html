class Solution {
    
    public int getWithDigits(int x, int start){
        if(x == 0){
            return 1;
        }else{
            if(x == 1 && start == 0){
                return 9;
            }
            int ans = 1;
            int curr = (9 - start);
           for(int i=0;i<x;i++){
               if(i == 0){
                   if(start == 0){
                       ans *= (9-start);
                   }else{
                       ans *= (10-start);
                   }
               }else{
                   ans *= curr;
                   curr--;
               }
           }
            return ans;
        }
    }
    
    public int get(int N){
        String str = String.valueOf(N);
        int len = str.length();
        int ans = 0;
        for(int i=1;i<len;i++)
        {
            ans += getWithDigits(i, 0);
        }
        //System.out.println(ans);
        boolean isDone = false;
        TreeSet<Integer> set = new TreeSet<Integer>();
        for(int i=0;i<len;i++){
            int d = (int) (str.charAt(i) - '0');
            int less = 0;
            for(int e : set){
                if(e < d){
                    less++;
                }
            }
            int totalPoss = (i == 0 ? (d-1) : d);
            int rem = ((totalPoss - less) >= 0 ? (totalPoss - less) : 0);
            int temp = getWithDigits(len - 1 - i, set.size()+1);
            if(!isDone)
            ans += (rem*temp);
            if(set.contains(d)){
                isDone = true;
            }
            set.add(d);
            //System.out.println(ans);
        }
        if(set.size() == len){
            ans++;
        }
        
return ans;
    }
    public int numDupDigitsAtMostN(int N) {
        return (N - get(N));
    }
}
