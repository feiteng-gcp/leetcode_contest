class Solution {
    public int numDupDigitsAtMostN(int N) {
            int[] digits = new int[11];
            int idx =0;
            while (N != 0) {
                digits[idx++] = N % 10;
                N /= 10;
            }
            return dfs(digits, idx - 1, 0, true, true);
        }

        public int get(int y) {
            int q = 1;
            for (int i = 0; i < y; i++) {
                q *= 10;
            }
            return q;
        }

        public int dfs(int[] digits, int at, int mask, boolean eq, boolean start) {
            if (at == -1) return 0;
            int tot = 0;
            int end = eq ? digits[at] : 10;
            for (int i = 0; i < end; i++) {
                if (((1 << i) & mask) != 0) {
                    tot += get(at);
                } else {
                    tot += dfs(digits, at - 1, start && i == 0 ? mask : mask | (1 << i), false, start && i == 0);
                }
            }
            if (eq) {
                int y = digits[at];
                if (((1 << y) & mask) != 0) {
                    int s = 1;
                    for (int i = at - 1; i >= 0; i--) {
                        s += digits[i] * get(i);
                    }
                    tot += s;
                } else {
                    tot += dfs(digits, at - 1, mask | (1 << y), true, false);
                }
            }
            return tot;
        }
}
