class Solution {
    public int numDupDigitsAtMostN(int N) {
        int z = N;
        int ndup = 0;
        
        String y  =""+N;
        int l =  y.length();
        
        if(l==1) return 0;
        
       // System.out.println(l);
        for(int i=1;i<l;++i){
            if(i>=2){
                int b = 9;
                int d = 9;
                for(int k=2;k<=i;++k){
                    b *= d;
                    --d;
                }
                
                
                ndup += b;
                //System.out.println(b);
            }else{
                ndup += 9;
            }
        }
        
       // System.out.println(ndup);
        
        
        for(int i=0;i<l;i++){
            int sz = y.charAt(i)-'0';
            
            int st = (i==0)?1:0;
            
            for(int j=st;j<sz;++j){
                
                boolean con = false;
                
                for(int ii=0;ii<i;++ii){
                    int yu = y.charAt(ii)-'0';
                    if(yu==j){
                        con = true;
                        break;
                    }
                }
                if(!con){
                    // ndup += (int)Math.pow(10,l-i-1);
                    
                    int b = 1;
                    int d = 10-(i+1);
                    for(int k=i+1;k<l;++k){
                        b *= d;
                        --d;
                    }
                
                
                    ndup += b;
                    
                 //   System.out.println(b);
                    
                    
                }
                
            }
            
            
            
            
            
            
            
            
            
            
            
            
            
            
            if(i==l-1){
                int j = sz;
                boolean con = false;
                
                for(int ii=0;ii<i;++ii){
                    int yu = y.charAt(ii)-'0';
                    if(yu==j){
                        con = true;
                        break;
                    }
                }
                if(!con){
                    ndup += 1;
                }
                
                
                
              
                
                
                
                
                
                
                
            }
            
            
             boolean con = false;
                
                for(int ii=0;ii<i;++ii){
                    int yu = y.charAt(ii)-'0';
                    if(yu==sz){
                        con = true;
                        break;
                    }
                }
                if(con){
                    break;   
                }
            
            
            
            
            
         //   ndup += (sz-1)*(int)Math.pow(10,l-i-1);
            
        }
        
        
        
        
        
        
        
        return z-ndup;
    }
}
