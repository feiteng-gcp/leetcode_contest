	class Solution {
	    public int numDupDigitsAtMostN(int N) {
	        char[] s = Integer.toString(N+1).toCharArray();
	        int u = s.length;
	        long[] dp = new long[1<<10];
	        long hit = 0;
	        int edge = 0;
	        boolean eh = false;
	        for(int i = 0;i < u;i++){
	        	long[] ndp = new long[1<<10];
	        	long nhit = 0;
	        	for(int j = 0;j < 1<<10;j++){
	        		for(int k = 0;k < 10;k++){
	        			if(j<<~k<0){
	        				nhit += dp[j];
	        			}else{
	        				ndp[j|1<<k] += dp[j];
	        			}
	        		}
	        	}
	        	nhit += hit * 10;
	        	
	        	if(i == 0){
	        		for(int j = 1;j < s[i]-'0';j++){
	        			ndp[1<<j]++;
	        		}
	        	}else{
	        		if(eh){
		        		for(int j = 0;j < s[i]-'0';j++){
	        				nhit++;
		        		}
	        		}else{
		        		for(int j = 0;j < s[i]-'0';j++){
		        			if(edge<<~j<0){
		        				nhit++;
		        			}else{
		        				ndp[edge|1<<j]++;
		        			}
		        		}
	        		}
	        		for(int j = 1;j <= 9;j++){
	        			ndp[1<<j]++;
	        		}
	        	}
	        	
	        	if(edge<<~(s[i]-'0')<0){
	        		eh = true;
	        	}
	        	edge |= 1<<s[i]-'0';
	        	dp = ndp;
	        	hit = nhit;
	        }
	        return (int)hit;
	    }
	}
