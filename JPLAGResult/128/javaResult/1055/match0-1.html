class Solution {
    public int numPairsDivisibleBy60(int[] time) {
        int[]rem = new int [60];
        for (int t:time) {
            rem[t%60]++;
        }
        int ans = 0;
        for (int i=0;i<=30;i++) {
            if (i%30==0) {
                ans += rem[i]*(rem[i]-1)/2;
            }else {
                ans += rem[i]*rem[60-i];
            }
        }
        return ans;
    }
}
