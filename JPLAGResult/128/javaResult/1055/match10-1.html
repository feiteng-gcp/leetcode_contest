class Solution {
    public int numPairsDivisibleBy60(int[] time) {
        int[] a=new int[60];
        for (int x:time) a[x%60]++;
        int ans=0;
        for (int i=1;i<=29;i++) ans+=a[i]*a[60-i];
        ans+=(a[0]-1)*a[0]/2;
        ans+=(a[30]-1)*a[30]/2;
        return ans;
    }
}
