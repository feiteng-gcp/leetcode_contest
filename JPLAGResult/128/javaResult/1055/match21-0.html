class Solution {
    public int numPairsDivisibleBy60(int[] time) {
            int[] mod = new int[60];
            for (int i = 0; i < time.length; i++) {
                mod[time[i] % 60]++;
            }
            int tot = 0;
            for (int i = 1; i < 30; i++) {
                tot += mod[i] * mod[60 - i];
            }
            tot += mod[0] * (mod[0] - 1) / 2;
        tot += mod[30] * (mod[30] - 1) / 2;
            return tot;
        }

}
