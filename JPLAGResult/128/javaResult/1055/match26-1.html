class Solution {
    public int numPairsDivisibleBy60(int[] time) {
        int[] count = new int[60];
        int ret = 0;
        for (int t : time) {
            count[t % 60] += 1;
        }
        
for (int i = 1; i < 30; ++i) {
            ret += (count[i] * count[60 - i]);
        }
        ret += ((count[0] - 1) * count[0] / 2);
        ret += ((count[30] - 1) * count[30] / 2);
        return ret;
    }
}
