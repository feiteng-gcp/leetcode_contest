	class Solution {
	    public int numPairsDivisibleBy60(int[] time) {
	        int[] f = new int[60];
	        for(int t : time)f[t%60]++;
	        int ret = 0;
	        for(int i = 1;i < 30;i++){
	        	ret += f[i] * f[60-i];
	        }
	        ret += f[30]*(f[30]-1)/2;
	        ret += f[0]*(f[0]-1)/2;
	        return ret;
	    }
	}
