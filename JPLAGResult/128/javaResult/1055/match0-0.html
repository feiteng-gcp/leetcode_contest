class Solution {
    public int numPairsDivisibleBy60(int[] time) {
        int[] divible = new int[60];
        
        for (int i = 0; i < time.length; i++) {
            divible[time[i] % 60]++;
        }
        int sum = 0;
        
        for(int i = 0; i <= 30; i++) {
            if (i == 0 || i == 30) {
               sum += (divible[i] - 1) * divible[i] / 2; 
            } else {
                sum += divible[i] * divible[ 60 -i];
            }          
        }
        return sum;
    }
}
