class Solution {
    public int numPairsDivisibleBy60(int[] time) {
        Map<Integer,Integer> map = new HashMap<>();
        int count = 0;
        
        for(int i = 0; i < time.length; i++){
            int rem = time[i]%60;
            int check = (rem > 0)?(60-rem):0;
            count+=map.getOrDefault(check,0);
            map.put(rem,map.getOrDefault(rem,0)+1);
        }
        return count;
    }
}
