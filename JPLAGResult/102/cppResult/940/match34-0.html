class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int res = 0;
        int n = tree.size();
        unordered_map<int, int> m; // value -- last index
        int count = 0;
        int prevIndex = -1;
        for (int i=0; i<tree.size(); ++i) {
            int type = tree[i];
            if (m.count(type) == 0) {
                if (count == 2) {
                    // need to remove one
                    //cout << i << " - " << prevIndex << endl;
                    res = max(res, i-prevIndex-1);
                    int maxIndex = INT_MAX;
                    for (auto kv : m) {
                        maxIndex = min(maxIndex, kv.second);
                    }
                    int key = tree[maxIndex];
                    prevIndex = maxIndex;
                    m.erase(key);
                    --count;
                }
                ++count;
            }
            m[type] = i;
        }
        //cout << n -1 << "-" << prevIndex << endl;
        res = max(res, n - prevIndex - 1);
        return res;
    }
};
