class Solution {
public:
    int totalFruit(vector<int>& tree) {
        unordered_map<int,int> mp;
        int n=tree.size(),l=0,r=0,cnt=0,rst=0;
        while(r<n){
            if(mp.find(tree[r])==mp.end() || mp[tree[r]]==0){

                ++mp[tree[r]];
                ++cnt;
                while(cnt>2 && l<r){
                    if(mp[tree[l]]==1){
                        --mp[tree[l]];
                        ++l;
                        --cnt;
                    }
                    else{
                        --mp[tree[l]];
                        ++l;
                    }
                }
            }
            else{
                ++mp[tree[r]];
            }
            rst=max(rst,r-l+1);
            ++r;
        }
        return rst;
    }
};
