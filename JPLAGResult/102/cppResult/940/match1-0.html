class Solution {
public:
    int totalFruit(vector<int>& t) {
        int ts=t.size();
        if(ts == 1)
            return 1;
        int start=0;
        int c=1;
        int maxl=1;
        int l=1;
        vector<int> ct(2,-1);
        ct[0]=t[0];
        for(int i=1;i<ts;i++)
        {
           // cout<<l<<endl;
            if(t[i] != t[i-1])
            {
                if(ct[1] == -1)
                {
                    ct[1]=t[i];
                    c++;
                }else
                {
                    if(!(t[i] == ct[0] || t[i] == ct[1]))
                    {
                        c++;
                    }
                }
                
                if(c == 2)
                {
                    start=i;
                }else if(c == 3)
                {
                    if(l>maxl)
                    {
                        maxl=l;
                    }
                    i=start;
                    ct[0]=t[i];
                    ct[1]=-1;
                    l=1;
                    c=1;
                    continue;
                }
            }
            l++;
        }
        if(l>maxl)
            maxl=l;
        return maxl;
    }
};
