class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int len=tree.size(), start=0, res=0;
        unordered_map<int, int> m;
        for (int i=0; i<len; i++)
        {
            m[tree[i]]++;
            if (m.size()>2)
            {
                while (start<i && m.size()>2)
                {
                    m[tree[start]]--;
                    if (m[tree[start]]==0)
                        m.erase(tree[start]);
                    start++;
                }
            }
            res=max(res, i-start+1);
        }
        return res;
    }
};
