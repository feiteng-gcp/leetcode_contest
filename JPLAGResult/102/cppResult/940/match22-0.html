class Solution {
public:
    int totalFruit(vector<int>& tree) {
        unordered_map<int, int> counts;
        int types =0;
        int maxlen = 0;
        
        int left = 0;
        int right = 0;
        while (right < tree.size())
        {
            int currentCount = ++counts[tree[right]];
            if (currentCount == 1)
            {
                if (++types > 2)
                {
                    // move left;
                    while (types > 2)
                    {
                        if (0 == --counts[tree[left]]) types --;
                        left++;
                    }
                }
            }
            
            maxlen = max(maxlen, right - left + 1);
            right++;            
        }
        
        return maxlen;
    }
};
