class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int res=0;
        map<int,int> mp;
        int i=0,j=0;
        while(j<tree.size()){
            if(mp.size()<=2){
                mp[tree[j]]++;
                res=max(res,j-i);
                j++;
            }
            else{
                mp[tree[i]]--;
                if(mp[tree[i]]==0) mp.erase(tree[i]);
                i++;
            }
        }
        if(mp.size()<=2) res=max(res,j-i);
        return res;
    }
};
