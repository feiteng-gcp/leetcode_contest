class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int left = 0 ; 
        map< int , int > check ; 
        int ans =0  ; 
        for( int i =0 ; i< tree.size() ; i++){
            check[tree[i]]++;
            if( check.size() > 2 ){
                while( check.size() != 2 ){
                    check[tree[left]]--;
                    if( check[tree[left]] == 0)
                        check.erase(tree[left]); 
                    left++;
                }
            }
            ans = max( ans , i - left +1 ) ;
        }
        return ans ; 
    }
};
