#include <bits/stdc++.h>

using namespace std;

class Solution {
public:
    int totalFruit(vector<int>& tree) {
		std::map<int, int> ans;
		size_t anw = 0;
		size_t pre = 0;
		for(size_t i = 0; i < tree.size(); ++i){
			++ans[tree[i]];
			while(ans.size() > 2){
				--ans[tree[pre]];
                if(ans[tree[pre]] == 0) ans.erase(tree[pre]);
				++pre;
			}
			anw = max(anw, i-pre+1);
		}
		return anw;
    }
};
