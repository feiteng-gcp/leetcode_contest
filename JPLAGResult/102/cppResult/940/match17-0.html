class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int n = tree.size();
        map<int,int> f;
        int d = 0;
        int ret = 0;
        int j = 0;
        for(int i = 0; i < n; ++i){
            d += !f[tree[i]];
            ++f[tree[i]];
            while(j < i and d > 2){
                --f[tree[j]];
                d -= !f[tree[j]];
                ++j;
            }
            ret = max(ret, i - j + 1);
        }
        return ret;
    }
};
