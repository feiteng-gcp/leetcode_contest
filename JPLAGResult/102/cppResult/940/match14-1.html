class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int n = tree.size();

        unordered_map<int, int> mp;
        int ans = 0;
        // while(i < n) {
        //     int j = i;
        //     while(j < n && mp.size() < 3) mp[tree[j++]]++;
        //     ans = max(ans, j - i);
        //     while(i < j && mp.size() > 2) {
        //         if(--mp[tree[i]] == 0) mp.erase(tree[i]);
        //         i++;
        //     }
        // }
        for(int i=0, j=0;i<n;i++) {
            mp[tree[i]]++;
            while(mp.size() > 2) {
                if(--mp[tree[j]] == 0) mp.erase(tree[j]);
                j++;
            }
            ans = max(ans, i - j + 1);
        }
        return ans;
    }
};
