class Solution {
public:
    int totalFruit(vector<int>& tree) {
        map<int,int> m;
        int cnt = 0;
        int n = tree.size();
        
        int l=0;
        int r=0;
        m[tree[l]]=1;
        cnt = 1;
        int ans = 1;
        while(r<n-1){
            r++;
            if(m.find(tree[r])==m.end() || m[tree[r]]==0){             
                cnt++;
            }
            m[tree[r]]++;
            
            while(cnt>2){
                m[tree[l]]--;
                if(m[tree[l]]==0)
                    cnt--;
                l++;
            }
            //cout<<l<<' '<<r<<endl;
            ans=max(ans, r-l+1);
        }
        return ans;
    }
};
