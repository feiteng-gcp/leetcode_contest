class Solution {
public:
  int totalFruit(vector<int>& tree) {
    int ans = 0;
    int current = 0;
    map<int, int> count;
    int head = 0;
    for (auto& x: tree) {
      ++count[x];
      ++current;
      for (; count.size() > 2 && head < (int)tree.size(); ++head) {
        --count[tree[head]];
        --current;
        if (count[tree[head]] == 0) {
          count.erase(tree[head]);
        }
      }
      ans = max(ans, current);
    }
    return ans;
  }
};
