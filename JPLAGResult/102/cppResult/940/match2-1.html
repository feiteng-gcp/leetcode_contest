class Solution {
public:
    int totalFruit(vector<int>& tree) {
        map<int, int> cnt;
        int result = 0;
        for (int s = 0, e = 0; e < tree.size(); e++) {
            cnt[tree[e]]++;
            while (cnt.size() > 2) {
                cnt[tree[s]]--;
                if (cnt[tree[s]] == 0) {
                    cnt.erase(tree[s]);
                }
                s++;
            }
            result = max(result, e - s + 1);
        }
        return result;
    }
};
