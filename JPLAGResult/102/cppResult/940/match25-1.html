class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int n = tree.size();
        int i = 0;
        int res = 0;
        while (i < n)
        {
            int j = i + 1;
            int k = -1;
            while (j < n && (tree[i] == tree[j] || k == -1 || tree[k] == tree[j]))
            {
                if (k == -1 && tree[i] != tree[j])
                    k = j;
                j++;
            }
            res = max(res, j - i);
            if (k != -1) i = k;
            else i = j;
        }
        return res;
    }
};
