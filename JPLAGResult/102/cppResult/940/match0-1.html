class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int len = tree.size();
        if(len == 0)
            return 0;

        int ch1, ch2;
        int ch1LastPos, ch2LastPos;

        ch1 = tree[0];
        int i = 0;
        while(i < len && tree[i] == ch1)
            ++i;
        ch1LastPos = i-1;
        if(i == len)
            return len;

        ch2 = tree[i];
        while(i < len && tree[i] == ch2)
            ++i;
        ch2LastPos = i-1;

        int lenMax = i;
        int startPosMax = 0;
        int startPosCur = 0;
        while(i < len)
        {
            if(tree[i] == ch1)
                ch1LastPos = i;
            else if(tree[i] == ch2)
                ch2LastPos = i;
            else
            {
                if(ch1LastPos < ch2LastPos)
                {
                    ch1 = tree[i];
                    startPosCur = ch1LastPos+1;
                    ch1LastPos = i;
                }
                else
                {
                    ch2 = tree[i];
                    startPosCur = ch2LastPos+1;
                    ch2LastPos = i;
                }
            }

            if(i-startPosCur+1 > lenMax)
            {
                lenMax = i-startPosCur+1;
                startPosMax = startPosCur;
            }
            ++i;
        }

        return lenMax;
    }
};
