class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int n=tree.size();
        int i=0,j=0;
        int longestLen=0;
        unordered_map<int,int>treetype{};
        while(j<n){
            for(;j<n and treetype.size()<3;j++){
                treetype[tree[j]]++;
                longestLen=max(longestLen,j-i);
            }
            while(treetype.size()==3){
                treetype[tree[i]]--;
                if(treetype[tree[i]]==0)
                    treetype.erase(tree[i]);
                i++;
            }
        }
        return max(longestLen,j-i); 
    }
};
