#define FORI(s,n) for(int i = s; i < n; i++)
#define FORJ(s,n) for(int j = s; j < n; j++)


class Solution {
public:
    int totalFruit(vector<int>& t) {
        int n = t.size ();
        unordered_map<int,int> m;
        int start = 0, ans = 0;
        FORI (0, n) {
            m[t[i]]++;
            while (m.size () > 2) {
                auto it = m.find (t[start]);
                if (it->second == 1) m.erase (it);
                else it->second--;
                start++;
            }
            ans = max (ans, m.begin ()->second + (m.size () > 1 ? (++m.begin ())->second : 0));
        }
        return ans;
    }
};
