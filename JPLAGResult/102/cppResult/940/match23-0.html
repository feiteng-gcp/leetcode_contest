class Solution {
public:
    int totalFruit(vector<int>& tree) {
        unordered_map<int, int> map;
        int count = 0;
        int begin = 0;
        int end = 0;
        int n = tree.size();
        int res = 0;
        while (begin < n)
        {
            while (end < n && count <= 2)
            {
                if (map.find(tree[end]) == map.end())
                {
                    count++;
                }
                map[tree[end]]++;
                ++end;
            }
            if (count == 3)
                res = max(res, end - begin - 1);
            else
                res = max(res, end - begin);
            if (end >= n)
                break;
            while (1)
            {
                map[tree[begin]]--;
                if (map[tree[begin]] == 0)
                {
                    map.erase(tree[begin]);
                    count--;
                    begin++;
                    break;
                }
                else
                    begin++;
            }
          //  map.erase(treetype);
           // count--;
        }
        return res;
    }
};
