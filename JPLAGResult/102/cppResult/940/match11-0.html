class Solution {
public:
  int totalFruit(vector<int>& tree) {
    map<int, int> m;
    int j = 0, res = 0;
    for (int i = 0; i < tree.size(); i++) {
      m[tree[i]]++;
      while (m.size() > 2) {
        m[tree[j]]--;
        if (m[tree[j]] == 0) {
          m.erase(tree[j]);
        }
        j++;
      }
      res = max(res, i - j + 1);
    }
    return res;
  }
};
