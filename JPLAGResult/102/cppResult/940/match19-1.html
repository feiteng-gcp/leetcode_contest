class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int ret=0;

        int start=0,cnt=0;
        map<int,int>dict;
        for(int i=0;i<tree.size();i++){
            dict[tree[i]]++;
            if(dict[tree[i]]==1){
                cnt++;
                while(cnt>2){
                    dict[tree[start]]--;
                    if(dict[tree[start]]==0)
                        cnt--;
                    start++;
                }
            }
            ret=max(ret,i-start+1);
        }
        return ret;
    }
};
