class Solution {
public:
    int totalFruit(vector<int>& tree) {
        int tmp = 0;
        int tp1 = 0, tp2 = 44444;
        int ans = 1;
        for (int i = 1; i < tree.size(); i++) {
            if (i - min(tp1, tp2) > ans) ans = i - min(tp1, tp2);
            if (tree[i] == tree[tmp]) continue;
            if (tp2 == 44444 || tree[tp1] == tree[i] || tree[tp2] == tree[i]) {
                if (tp2 == 44444) tp2 = i;
            } else {
                tp1 = tmp; tp2 = i;
            }
            tmp = i;
        }
        if ((tree.size() - min(tp1, tp2)) > ans) ans = tree.size() - min(tp1, tp2);
        return ans;
    }
};
