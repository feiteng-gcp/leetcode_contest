class Solution {
public:
    int sumSubarrayMins(vector<int>& A) {
        vector<int> l(A.size(), 0), r(A.size(), 0);
        for (int i = 1; i < A.size(); ++i) {
            l[i] = i;
            int cur = i - 1;
            while (A[i] <= A[cur] && cur != 0) {
                cur = l[cur];
                if (cur) cur -= 1;
            }
            l[i] = A[i] <= A[cur] ? cur : cur + 1;
        }
        r[A.size()-1] = A.size()-1;
        for (int i = A.size()-2; i >= 0; --i) {
            r[i] = i;
            int cur = i + 1;
            while (A[i] < A[cur] && cur != A.size()-1) {
                cur = r[cur];
                if (cur != A.size()-1) cur += 1;
            } 
            r[i] = A[i] < A[cur] ? cur : cur - 1;
        }
        int64_t ans = 0;
        for (int i = 0; i < A.size(); ++i) {
            //cout << l[i] << " " << r[i] << " " << ans << endl;
            ans = (ans + 1LL * (r[i]-i+1) * (i - l[i]+1) * A[i]) % (1000000007);
        }
        return ans;
    }
};
