class Solution {
public:
    int sumSubarrayMins(vector<int>& A) {
        long long mod = 1000000007;
        int n = A.size();
        
        long long sum = 0;
        stack<int> st;
        for (int i=0; i<A.size(); i++){
            int l = i;
            while(!st.empty() && A[i]<A[st.top()]){
                int index = st.top(); st.pop();
                l = st.top(); st.pop();
                int r = i-1;
                
                int numWays = (getProd(r-l+1) - getProd(index-l) - getProd(r-index));
                sum += numWays*A[index];
                sum %= mod;
            }
            st.push(l);
            st.push(i);
        }
        
        while(!st.empty()){
            int index = st.top(); st.pop();
            int l = st.top(); st.pop();
            int r = n-1;

            int numWays = (getProd(r-l+1) - getProd(index-l) - getProd(r-index));
            sum += numWays*A[index];
            sum %= mod;
        }
        
        return sum;
    }
    
    int getProd(int n){
        return (n*(n+1))/2;
    }
};
