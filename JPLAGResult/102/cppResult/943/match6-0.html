class Solution {
public:
    int sumSubarrayMins(vector<int>& A) {
        vector<int> dpl(A.size(),0), dpr(A.size(),0);
        
        for(int i = 0; i < A.size(); ++i){
            int j = i-1;
            while(j>=0 && A[j]>=A[i]){
                j -= dpl[j];
                dpl[i] = i-j;
                --j;
            }
        }
        for(int i = A.size()-1; i >= 0; --i){
            int j = i+1;
            while(j<A.size() && A[j]>=A[i]){
                if(A[j]==A[i])
                    dpl[j] = min(dpl[j],j-i-1);
                j += dpr[j];
                dpr[i] = j-i;
                ++j;
            }
        }
        
        
        int ans = 0, base = 1000000007;
        for(int i = 0; i < A.size(); ++i){
            int num = (dpl[i]+1)*(dpr[i]+1);
            ans = (ans + num*A[i]) % base;
        }
        return ans;
    }
};
