class Solution {
public:
    const int M=1e9+7;
    int sumSubarrayMins(vector<int>& A) {
        long long ans=0;
        int n=A.size();
        
        for(int i=0;i<n;i++){
            int lo=i-1,hi=i+1;
            while(lo>=0&&A[lo]>A[i])lo--;
            while(hi<n&&A[hi]>=A[i])hi++;
            ans+=(A[i]*(i-lo)*(hi-i))%M;
            ans%=M;
        }

        return (int)ans;
        
    }
};
