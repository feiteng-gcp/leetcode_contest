#define MOD 1000000007

class Solution {
public:
    int sumSubarrayMins(vector<int>& A) {

        int n = A.size();
        int result = 0;
        vector<int> lastMinIndex;
        for (int i = 0; i < n; i++)
        {
            int index = -1;
            while (!lastMinIndex.empty())
            {
                index = lastMinIndex[lastMinIndex.size() - 1];
                if (A[index] < A[i])
                {
                    break;
                }
                else
                {
                    lastMinIndex.erase(lastMinIndex.begin() + lastMinIndex.size() - 1);
                }
            }
            if (lastMinIndex.empty())
            {
                index = -1;
            }
            
            //cout << A[i]  << " " << i << ' ' << index << ' ' << A[i] * (i - index) << endl;
            result += A[i] * (i - index);
            
            if (!lastMinIndex.empty())
            {
                for (int index = lastMinIndex.size() - 1; index >= 0; index--)
                {
                    int a = lastMinIndex[index];
                    int b = index == 0 ? -1 : lastMinIndex[index - 1];
                    result += A[a] * (a - b);
                    //cout << A[a] << ' ' << a - b << ' ' << A[a] * (a - b) <<endl;
                }
            }
            
            result = result % MOD;
            
            //cout << endl;
            lastMinIndex.push_back(i);
        }
        
        return result;
    }
};
