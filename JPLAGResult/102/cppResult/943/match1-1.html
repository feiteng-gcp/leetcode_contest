class Solution {
public:
    int sumSubarrayMins(vector<int>& A) {
        long long MOD = 1000000007;

        long long ret = 0;
            
        for (int i=0;i<A.size();++i) {
            int l=i-1, r=i+1;
            while(l>=0 && A[l]>A[i]) --l;
            while(r<A.size() && A[r]>=A[i]) ++r;
            //printf ("%d %d %d %d %d\n",l, i, r, (((i-l)*(r-i))%MOD), (((((i-l)*(r-i))%MOD)*A[i])%MOD));
            ret += (((((i-l)*(r-i))%MOD)*A[i])%MOD);
            ret %= MOD;
        }
        return ret ;
    }
};
