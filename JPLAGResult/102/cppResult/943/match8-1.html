class Solution {
public:
#define mode 1000000007
int n;
vector<int> *B;
#define ll long long
int left(int x){
    int i;
    for(i = 1; x - i >= 0 && (*B)[x - i] > (*B)[x]; i ++);
    return i - 1;
}
int right(int x){
    int i;
    for(i = 1; x + i < n && (*B)[x + i] >= (*B)[x]; i ++);
    return i - 1;
}
int sumSubarrayMins(vector<int>& A) {
    n = A.size(); B = &A;
    if(!n) return 0;

    ll ans = 0;
    for(int i = 0; i < n; i ++){
        ans += ((ll)(left(i) + 1) * (right(i) + 1)) * A[i] % mode;
        ans %= mode;
    }
    return ans;
}

};
