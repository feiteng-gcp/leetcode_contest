class Solution {
public:
    int sumSubarrayMins(vector<int>& A) {
        int ret = 0;
        vector<int> v(A.size(),0);
        v[A.size()-1]=A.size();
        stack<int> s;
        s.push(A.size()-1);
        
        for(int i=A.size()-2;i>=0;--i)
        {
            while(s.size()>0 && A[s.top()] >=A[i])
                s.pop();
            if(s.size()==0)
                v[i]=A.size();
            else
                v[i]=s.top();
            s.push(i);
        }
        
        vector<int> dp(A.size()+1,0);
        dp[A.size()-1] = A[A.size()-1];
        for(int i=A.size()-2;i>=0;--i)
        {
            dp[i]=(v[i]-i)*A[i] + dp[v[i]];
        }
        for(int i=0;i<dp.size();++i)
        {
            ret+=dp[i];
            ret=ret%1000000007;
        }
        return ret;
    }
};
