class Solution {
public:
    int sumSubarrayMins(vector<int>& A) {
        vector<int> vec1(A.size(),0);
        vector<int> vec2(A.size(),0);
        vector<pair<int,int>> vec;
        for(int i=0;i<A.size();i++){
            while(!vec.empty() && vec.back().first>A[i]){
                vec1[vec.back().second]=i-vec.back().second;
                vec.pop_back();
            }   
            vec.push_back(pair<int,int>(A[i],i));
        }
        while(!vec.empty()){
                vec1[vec.back().second]=A.size()-vec.back().second;
                vec.pop_back();
            }
        vec.clear();
        for(int i=A.size()-1;i>=0;i--){
            while(!vec.empty() && vec.back().first>=A[i]){
                vec2[vec.back().second]=vec.back().second-i;
                vec.pop_back();
            }
            vec.push_back(pair<int,int>(A[i],i));
        }
        while(!vec.empty()){
                vec2[vec.back().second]=vec.back().second+1;
                vec.pop_back();
            }
        int sum=0;
        for(int i=0;i<vec1.size();i++){
            sum+=vec1[i]*vec2[i]*A[i];
            sum=sum%(1000000007);
        }
        return sum;
    }
};
