class Solution {
public:
    int sumSubarrayMins(vector<int>& A) {
        if (A.empty())
            return 0;
        int mod = pow(10, 9) + 7;
        vector<int> res(A.size(), 0);
        res[0] = A[0] % mod;
       // int res = A[0] % mod;
        vector<int> st;
        st.push_back(0);
        for (int i = 1; i < A.size(); i++)
        {
            int len = st.size() - 1;
            int val = -1;
            while (!st.empty() && A[i] < A[st.back()])
            {
                val = st.back();
                st.pop_back();
            }
            if (val == -1)
            {
                st.push_back(i);
                res[i] = (res[i - 1] + A[i] % mod) % mod;
                res[i] %= mod;
                continue;
            }

            val = st.empty() ? -1 : st.back();
            if (val == -1)
                res[i] = A[i] * (i + 1) % mod;
            else
            {
                res[i] = (res[val] + A[i] * (i - val))%mod;
            }
            st.push_back(i);
            
        }
        int val = 0;
        for (int a : res)
        {
            val += a;
            val %= mod;
        }
        return val;
    }
};
