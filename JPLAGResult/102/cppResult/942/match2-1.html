class Solution {
public:
    bool comp(string & a, string & b) {
        if (a.length() < b.length()) {
            return false;
        } else if (a.length() > b.length()) {
            return true;
        } else {
            return strcmp(a.c_str(), b.c_str()) >=0;
        }
    }
