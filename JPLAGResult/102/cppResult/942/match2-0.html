class Solution {
public:
    vector<long long> res;
    void init(){
        for(int i=1;i<100000;i++){
            long long number1 = i,number2 = i;
            long long tmp = i;
            tmp /= 10;
            while(tmp){
                number1 = number1 * 10 + tmp % 10;
                tmp /= 10;
            }
            tmp = i;
            while(tmp){
                number2 = number2 * 10 + tmp % 10;
                tmp /= 10;
            }
            if(number1 <= 1e9 && check(number1*number1)){
                res.push_back(number1*number1);
            }
            if(number2 <= 1e9 && check(number2*number2)){
                res.push_back(number2*number2);
            }
        }
    }
    bool check(long long number){
        long long res = 0,tmp = number;
        while(tmp){
            res = res * 10 + tmp % 10;
            tmp /= 10;
        }
        return (res == number);
    }
    int superpalindromesInRange(string L, string R) {
        init();
        long long t1 = 0,t2 = 0;
        for(int i=0;i<L.length();i++) t1 = t1 * 10 + L[i] - '0';
        for(int i=0;i<R.length();i++) t2 = t2 * 10 + R[i] - '0';
        int sum = 0;
        for(int i=0;i<res.size();i++){
            if(res[i] >= t1 && res[i] <= t2){
                sum++;
            }
        }
        return sum;
    }
};
