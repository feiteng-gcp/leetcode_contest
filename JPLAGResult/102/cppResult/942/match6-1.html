class Solution {
public:
    long long convert(string a){
        long long ret = 0;
        for(char c : a) ret = ret * 10 + c - '0';
        return ret;
    }
    bool check(long long x){
        stringstream ss;
        ss <<x;
        string a = ss.str();
        string b = a;
        reverse(a.begin(), a.end());
        return a == b;
    }
    int superpalindromesInRange(string L, string R) {

        long long l = convert(L);
        long long r = convert(R);

        int ret = 0;
        for(int i = 1; i <= 999999; ++i){
            stringstream ss;
            ss <<i;
            string a = ss.str();
            string b = a;
            reverse(b.begin(), b.end());
            long long x = convert(a + b);
            if(x <= r / x + 1){
                x *= x;
                if(check(x)) ret += x >= l and x <= r;
            }
            
            b = a;
            b.pop_back();
            reverse(b.begin(), b.end());
            x = convert(a + b);
            if(x > r / x + 1) break;
            x *= x;
            if(check(x)) ret += x >= l and x <= r;
        }
        return ret;
    }
};
