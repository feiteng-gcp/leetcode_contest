class Solution {
public:
    int superpalindromesInRange(string L, string R) {
        long long l = stoll(L);
        long long r = stoll(R);
        
        long long s = 1;
        int n = 0;
        for (int i = 1; i < 4; i++) {
            if (l <= i*i && i*i <= r) n++;
        }
        do {
            n += palindrome(s, l, r, -1);
            for (int i = 0; i < 10; i++) {
                n += palindrome(s, l, r, i);
            }
            s++;
        } while(!isOver(s, r));
        
        return n;
    }
private:
    string gen(long long s, int a = -1) {
        string S = to_string(s);
        string SR = S;
        reverse(SR.begin(), SR.end());
        if (a == -1) return S + SR;
        return S + to_string(a) + SR;
    }
    bool isOver(long long s, long long& r) {
        long long pn = stoll(gen(s));
        if (pn * pn < pn) return true;
        return pn * pn >= r;
    }
    
    bool palindrome(long long s, long long& l, long long& r, int i) {
        string S = gen(s, i);
        long long pn = stoll(S);
        long long pns = pn*pn;
        string PS = to_string(pns);
        string RPS = PS;
        reverse(RPS.begin(), RPS.end());
        if (PS != RPS) return false;
        //cout << "cal : " << PS << endl;
        return l <= pns && pns <= r;
    }
};
