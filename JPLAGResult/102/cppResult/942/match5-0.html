class Solution {
private:
    bool isPanlid(string n) {
        int i = 0, j = n.size() - 1;
        while (i < j) {
            if (n[i] != n[j]) return false;
            i++, j--;
        }
        return true;
    }    
public:
    int superpalindromesInRange(string L, string R) {
        long s = stol(L), e = stol(R), res = 0;
        long s1 = sqrt(s), e1 = sqrt(e) + 1;
        for (long n = s1; n < e1;) {
            string ns = to_string(n);

            if (n > 10000000) {
                if (n % 10 != ns[0] - '0' || 
                    (n % 100) / 10 != ns[1] - '0' ||
                   (n % 1000) / 100 != ns[2] - '0' ||
                   (n % 10000) / 1000 != ns[3] - '0') {
                    int b = (ns[3] - '0') * 1000 + (ns[2] - '0') * 100 + (ns[1] - '0') * 10 + ns[0] - '0';
                    if (b > n % 10000) n += (b - n % 10000);
                    else n += (10000 - n % 10000);
                }
            }
            else if (n > 100000) {
                if (n % 10 != ns[0] - '0' || 
                    (n % 100) / 10 != ns[1] - '0' ||
                   (n % 1000) / 100 != ns[2] - '0') {
                    int b = (ns[2] - '0') * 100 + (ns[1] - '0') * 10 + ns[0] - '0';
                    if (b > n % 1000) n += (b - n % 1000);
                    else n += (1000 - n % 1000);
                }
            }

            if (isPanlid(to_string(n))) {
                long nn = n * n;
                if (nn >= s && nn <= e && isPanlid(to_string(nn))) {
                    res++;
                }
            }
            n++;
        }
        
        return res;
    }
};
