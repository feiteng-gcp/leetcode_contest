class Solution {
public:
    string num ;
    long long mi ; 
    long long ma ; 
    int ans  ; 
    bool islegal( long long t ){
        string temp = to_string( t ) ; 
        int left = 0 ; 

        int right = temp.size()-1;
        while( left< right ){
            if( temp[left] != temp[right] ) return false ; 
            left++;
            right--;
        }
        return true ; 
    }
    void dfs( int depth , int size ){
        
        if( depth > size-1-depth ){
            long long t = atoi(num.c_str());
            if( !islegal(  t*t )){
                
                return ; 
            } 
            if(t*t>= mi && t*t <= ma ){
                ans ++ ; 
            }
            return ; 
        }
        
        for( int i=0 ; i<=9; i++){
            if( i==0 && depth ==0 ) continue ; 
            num[depth] = i+'0' ; 
            num[size-1-depth] = i+'0' ;
            dfs( depth + 1, size ) ; 
        }
        
    }
    int superpalindromesInRange(string L, string R) {
        mi = 0 ; 
        ma = 0 ;
        for( auto & c : L ) mi = mi*10 + c-'0';
        for( auto &c : R ) ma = ma*10 + c-'0';
        //cout << mi <<ma <<endl ;
        ans = 0 ; 
        vector< long long > table {1,4,9};
        for( int i=0 ; i< table.size() ; i++){
            if( table[i] >= mi && table[i] <= ma ) {
                ans ++ ; 
            }
        }
        for( int i=2; i<=9 ; i++){
            num.resize( i , 0 ) ; 
            dfs( 0 , i ) ; 
        }
        return ans  ; 
    }
};
