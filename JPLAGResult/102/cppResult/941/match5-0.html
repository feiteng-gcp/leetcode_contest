class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        vector<int> ans(A.size());
        int i = 0, j = A.size()-1;
        
        for (int a : A) {
            if (a % 2) {
                ans[j--] = a;
            } else {
                ans[i++] = a;
            }
        }
        
        return ans;
    }
};
