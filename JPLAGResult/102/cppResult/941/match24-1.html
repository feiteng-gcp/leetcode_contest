class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        int len = A.size(), pos = 0;
        vector<int> ans(len);

        for(int i = 0; i < len; i++)
        {
            if(A[i] % 2 == 0)
            {
                ans[pos] = A[i];
                pos = pos + 1;
            }
        }
        for(int i = 0; i < len; i++)
        {
            if(A[i] % 2 == 1)
            {
                ans[pos] = A[i];
                pos = pos + 1;
            }
        }
        return ans;
    }
};
