class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        vector<int> ret(A.size());
        int ptr0=0, ptr1=A.size()-1;
        for(int i=0;i<A.size();++i) {
            if ((A[i]%2)==0) {
                ret[ptr0++] = A[i];
            } else {
                ret[ptr1--] = A[i];
            }
        }
        return ret;
    }
};
