class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        vector<int> ans;
        for (auto n: A) {
            if ((n & 1) == 0) {
                ans.push_back(n);
            }
        }
        for (auto n: A) {
            if ((n & 1)) {
                ans.push_back(n);
            }
        }
        return ans;
    }
};
