class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        vector<int> res;
        for (auto v : A) {
            if (v % 2 == 0) {
                res.push_back(v);
            }
        }
        for (auto v : A) {
            if (v % 2 == 1) {
                res.push_back(v);
            }
        }
        return res;
    }
};
