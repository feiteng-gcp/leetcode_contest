class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        int N = A.size();
        vector<int> B(N);
        int j = 0;
        for(int i = 0 ; i < N; ++i){
            if(A[i] % 2 == 0){
                B[j] = A[i];
                j++;
            }
        }
        for(int i = 0 ; i < N; ++i){
            if(A[i] % 2 == 1){
                B[j] = A[i];
                j++;
            }
        }
        return B;
    }
};
