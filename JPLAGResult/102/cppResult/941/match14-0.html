class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        vector<int> res(A.size(),0);
        int l = 0;int r = A.size()-1;
        for(int i = 0;i<A.size();i++){
            if(A[i]%2==0)
                res[l++]=A[i];
            else
                res[r--]=A[i];
        }
        return res;
        
    }
};
