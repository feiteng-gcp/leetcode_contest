class Solution {
    vector<int> res;

public:
    vector<int> sortArrayByParity(vector<int>& A) {
        int n = A.size();
        for(int i = 0; i < n; ++i){
            if(A[i] % 2 == 0){
                res.push_back(A[i]);
                A.erase(A.begin() + i);
                --i;
                --n;
            }
        }
        for(int i = 0; i < n; ++i)res.push_back(A[i]);
        return res;
    }
};
