class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        int cnt,cnt1,cnt2;
        cnt = 0;
        vector<int> v;
        for(int i = 0; i < A.size(); i++){
            v.push_back(0);
            if(A[i] % 2 == 0)
                cnt++;
        }
        cnt2 = 0;
        cnt1 = cnt;
        for(int i = 0; i < A.size(); i++){
            if(A[i] % 2 == 1)
                v[cnt1++] = A[i];
            else
                v[cnt2++] = A[i];
        }
        return v;
    }
};
