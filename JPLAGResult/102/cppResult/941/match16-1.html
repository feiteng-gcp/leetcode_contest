class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        vector<int> sol(A.size(),0);

        int l=0,r = A.size()-1;
        
        for (int i=0; i<A.size(); i++){
            if(A[i]%2 ==0)
                sol[l++] = A[i];
            else
                sol[r--]= A[i];
        }
        return sol;
        
    }
};
