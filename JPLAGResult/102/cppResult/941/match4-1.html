class Solution {
public:
    
    vector<int> sortArrayByParity(vector<int>& A) {
        vector< int > ans ;
        ans.resize( A.size() , 0 ) ; 
        int left = 0 ; 
        int right = A.size()-1 ; 
        for( int i =0 ; i<A.size () ; i++){

            if( A[i] % 2 == 0 ){
                ans[left++]= A[i] ; 
            }
            else{
                ans[right--]=A[i] ; 
            }
        }
        return ans ; 
    }
};
