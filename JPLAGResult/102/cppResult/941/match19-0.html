class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        vector<int> result(A.size());
        int lo = 0;
        int hi = A.size()-1;
        for(auto& x : A) {
            if (x&1) {
                result[hi--] = x;
            }
            else {
                result[lo++] = x;
            }
        }
        return result;
    }
};
