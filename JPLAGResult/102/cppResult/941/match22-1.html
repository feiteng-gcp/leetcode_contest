class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& A) {
        vector<int> res;
        int ln=A.size();

        for(int i=0; i<ln; i++){
            if(A[i]%2==0){res.push_back(A[i]);}
        }
        for(int i=0; i<ln; i++){
            if(A[i]%2==1){res.push_back(A[i]);}
        }
        return res;
    }
};
