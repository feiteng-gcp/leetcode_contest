class Solution {
    public int totalFruit(int[] tree) {
        if(tree == null || tree.length == 0) {
            return 0;
        }
        
        Map<Integer, Integer> map = new HashMap<>();
        int max = 0;
        int count = 0;
        int i = 0;
        int j = 0;
        int n = tree.length;
        while( j < n) 
{
            int t = tree[j];
            if(map.get(t) == null) {
                map.put(t, 0);
            }
            map.put(t, map.get(t)+1);
            count++;
            j++;
            if(map.size() <=2 && count > max){
                max = count;
            }
            if(j < n) {
                while(map.size() >2) {
                    t = tree[i];
                    
map.put(t, map.get(t)-1);
                    if(map.get(t) == 0){
                        map.remove(t);
                    }
                    i++;
                    count--;
                }
            }
            
        }
        return max;
    }
}
