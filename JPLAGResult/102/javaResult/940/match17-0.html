//find the largest window have at most 2 kind of fruite...
class Solution {
    public int totalFruit(int[] tree) {
        int n = tree.length;
        int[] map = new int[n + 1];
        int max = 0, cnt = 0;
        int l = 0;
        for (int r = 0; r < n; r++) {
            map[tree[r]]++;
            if (map[tree[r]] == 1) {
                cnt++;
            }
            while (cnt > 2) {
                map[tree[l]]--;
                if (map[tree[l]] == 0) cnt--;
                l++;
            }
            max = Math.max(max, r - l + 1);
        }
        return max;
    }
}
