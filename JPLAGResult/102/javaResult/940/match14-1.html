class Solution {
    public int totalFruit(int[] tree) {
        
int max = 0, i = 0;
        Map<Integer, Integer> types = new HashMap<>();
        for (int j = 0; j < tree.length; ++j) {            
            types.put(tree[j], j);
            if (types.size() > 2) {
                int sKey = 0, min = tree.length;
                for (int key : types.keySet()) {
                    if (types.get(key) < min) {
                        min = types.get(key);
                        sKey = key;
                    }
                }
                i = types.remove(sKey) + 1;
            
}
            max = Math.max(max, j - i + 1);
        }
        return max;
    }
}
