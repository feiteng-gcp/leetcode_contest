class Solution {
    public int totalFruit(int[] tree) {
        Map<Integer, Integer> count = new HashMap<>();
        int res = 0;
        int s = 0;
        for (int j = 0; j < tree.length; ++j) 
{
        	count.put(tree[j], count.getOrDefault(tree[j], 0) + 1);
        	while (count.size() > 2) {
    			int temp = count.get(tree[s]);
    			if (temp == 1) {
    				count.remove(tree[s++]);
    			}else {
    				count.put(tree[s++], temp - 1);
    			}
    		}
        	res = Math.max(j - s + 1, res);
        }
        return res;
    }
}
