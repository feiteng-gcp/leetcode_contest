class Solution {
    public int totalFruit(int[] tree) {
        int max=0;
        Map<Integer,Integer> countMap=new HashMap<Integer,Integer>();
        int start=0;
        int total=0;
        for(int end=0;end<tree.length;end++){
            int curr=tree[end];
            if(!countMap.containsKey(curr)){
                countMap.put(curr,0);
            }
            countMap.put(curr,countMap.get(curr)+1);
            total++;
            while(start<=end&&countMap.keySet().size()>2){
                int pre=tree[start];
                total--;
                start++;
                countMap.put(pre,countMap.get(pre)-1);
                if(countMap.get(pre)==0){
                    countMap.remove(pre);
                }
            }
            max=Math.max(max,total);
        }
        return max;
    }
}
