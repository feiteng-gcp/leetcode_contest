class Solution {
    public int totalFruit(int[] tree) {
        Map<Integer, Integer> count = new HashMap<>();
        int num = 0, l = 0, r = 0, n = tree.length, ans = 0;
        while (r < n) {
            while (r < n && (num < 2 || count.containsKey(tree[r]))) {
                if (count.containsKey(tree[r])) {
                    count.put(tree[r], 
count.get(tree[r]) + 1);
                } else {
                    num++;
                    count.put(tree[r], 1);
                }
                r++;
            }
            ans = Math.max(ans, r - l);
            while(l < r && num == 2) {
                count.put(tree[l], count.get(tree[l]) - 1);
                if (count.get(tree[l]) == 0) {
                    count.remove(tree[l]);
                    num--;
                }
                l++;
            }
        }
        return ans;
    }
}
