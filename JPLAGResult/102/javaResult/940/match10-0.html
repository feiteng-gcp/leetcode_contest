class Solution {
    public int totalFruit(int[] tree) {
        int n = tree.length, max = 0;
        Map<Integer, Integer> seen = new HashMap<>();
        for (int start = 0, end = 0; end < tree.length; end++) {
            if (!seen.containsKey(tree[end])) {
                if (seen.size() == 2) {
                    while (seen.size() == 2) {
                        int count = seen.get(tree[start]);
                        if (count == 1) {
                            seen.remove(tree[start]);
                        } else {
                            seen.put(tree[start], count - 1);
                        }
                        start++;
                    }
                }
            }
            
            seen.put(tree[end], seen.getOrDefault(tree[end], 0) + 1);
            max = Math.max(end - start + 1, max);
        }

        return max;
    }
}
