class Solution {
    public int totalFruit(int[] tree) {
        int res = 0;
        Map<Integer, Integer> fruit = new HashMap<>();
        int j = tree.length-1;
        for(int i=tree.length-1; i>=0; i--){
            int f = tree[i];
            int count = 0;
            if(fruit.containsKey(f)) {
               count = fruit.get(f);  
            }
            count++;
            fruit.put(f, count);
            while(fruit.size() > 2) {
                int fj = tree[j];
                count = 
fruit.get(fj);
                if(count == 1) {
                    fruit.remove(fj);
                } else {
                    count--;
                    fruit.put(fj, count);
                }
                j--;
            }
            res = Math.max(res, j-i+1);
        }
        return res;
    }
}
