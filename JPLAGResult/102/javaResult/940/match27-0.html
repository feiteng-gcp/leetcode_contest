class Solution {
    public int totalFruit(int[] tree) {
        int n = tree.length;
        
        int[] f = new int[n];
        int last = 0;
        f[0] = -1;
        for (int i = 1; i < n; ++i) {
            if (tree[i] == tree[i - 1]) {
                f[i] = f[last];
            }
            else {
                f[i] = i - 1;
                last = i;
            }
        }
        
        
        
        int maxi = 0;
        int[] ans = new int[n];
        for (int i = 0; i < n; ++i) {
            if (f[i] == -1) ans[i] = i + 1;
            else if (f[f[i]] == -1) ans[i] = i + 1;
            else if (tree[f[f[i]]] == tree[i]) ans[i] = ans[f[i]] + i - f[i];
            else ans[i] = i - f[f[i]];
            
            maxi = Math.max(maxi, ans[i]);
        }
        
        //for (int i = 0; i < n; ++i) System.out.printf("f[%d]=%d   ", i, f[i]);
        //System.out.println();
        //for (int i = 0; i < n; ++i) System.out.printf("ans[%d]=%d   ", i, ans[i]);
        
        return maxi;
    }
}
