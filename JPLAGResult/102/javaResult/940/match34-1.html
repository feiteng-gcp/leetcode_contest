class Solution {
    public int totalFruit(int[] tree) {
        int res = 0;
        int count = 0;
        int left = 0;
        int right = 0;
        int[] fruits = new int[tree.length];
        for(; right < tree.length; right++) {
            fruits[tree[right]]++;
            if(fruits[tree[right]] == 1) count++;
            while(count > 2) {
                fruits[tree[left]]--;
                if(fruits[tree[left]] == 0) count--
;
                left++;
            }
            res = Math.max(res, right - left + 1);
        }
        return res;
    }
}
