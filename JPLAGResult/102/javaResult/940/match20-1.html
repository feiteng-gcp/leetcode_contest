class Solution {
    public int totalFruit(int[] tree) {
        if(tree == null )
            return 0;
        if(tree.length <= 2)
            return tree.length;
        
Integer typeA = null;
        Integer typeB = null;
        
        int count = 0;
        int max = 0;
        for(int i = 0; i<tree.length; i++){
        	if(typeA == null){
        		typeA = tree[i];
        		count++;
        	}else if(tree[i] != typeA && typeB == null){
        		typeB = tree[i];
        		count++;
        	}else if(tree[i] == typeA || tree[i] == typeB){
        		count++;
        	}else{
        		int j = i-1;
        		while(tree[j] == tree[i-1])
        			j--;
        		count = i-j;
        		typeA = tree[i-1];
        		typeB = tree[i];
        		
        	}

        	max = Math.max(max, count);
        }
        return max;
        
    }
}
