class Solution {
    public int totalFruit(int[] tree) {
        if (tree == null || tree.length == 0) {
            return 0;
        }
        int right = 0;
        int result = 0;
        Map<Integer, Integer> hashMap = new HashMap<>();
        int count = 0;
        for (int left = 0; left < tree.length; left++) {
            while (right < tree.length && (hashMap.size() < 2 || hashMap.containsKey(tree[right]))) {
                count++;
                result = Math.max(result, count);
                if (hashMap.containsKey(tree[right])) {
                    hashMap.put(tree[right], hashMap.get(tree[right]) + 1);
                } else {
                    hashMap.put(tree[right], 1);
                }
                right++;
            }
            if (hashMap.get(tree[left]) > 1) {
                hashMap.put(tree[left], hashMap.get(tree[left]) - 1);
            } else {
                
hashMap.remove(tree[left]);
            }
            count--;
        }
        return result;
    }
}
