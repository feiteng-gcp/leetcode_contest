class Solution {
    public int totalFruit(int[] tree) {
        int max = 0, i = 0, j = 0;
        LinkedHashMap<Integer, Integer> idx = new LinkedHashMap();
        HashSet<Integer> set = new HashSet();
        while (i < tree.length) {
            int n = tree[i];
            set.add(n);
            if (set.size()>2) {
                int m = idx.entrySet().iterator().next().getKey();
                j = idx.entrySet().iterator().next().getValue()+1;
                set.remove(m);
                idx.remove(m);
            }
            if (!set.contains(n)) idx.put(n, i);
            else {
                // int tmp = idx.get(n);
                idx.remove(n);
                idx.put(n, i);
            }
            max = Math.max(max, i-j+1);
            i++;
        }
        return max;
    }
}
