class Solution {
    public int totalFruit(int[] tree) {
        int left = 0;
        int total = 0;
        int size = 0;
        Map<Integer,Integer> basketTypes = new HashMap<>();
        for(int right=0; right<tree.length;++right){
            basketTypes.put(tree[right],basketTypes.getOrDefault(tree[right],0)+1);
            while(basketTypes.size()>2 && left<=right){
                int amount = basketTypes.get(tree[left]);
                --amount;
                if(amount<=0)basketTypes.remove(tree[left]);
                else basketTypes.put(tree[left],amount);
                ++left;
            }
            total=Math.max(total,right-left+1);
        }
        return total;
    }
}
