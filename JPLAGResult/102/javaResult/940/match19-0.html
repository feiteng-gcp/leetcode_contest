class Solution {
    public int totalFruit(int[] tree) {
        if (tree == null || tree.length == 0) return 0;
        if (tree.length == 1) return 1;
        int res = Integer.MIN_VALUE;
        int type1 = tree[0];
        int begin1 = 0;
        int end1;
        int type2;
        int begin2 = 1;
        while (begin2 < tree.length && tree[begin2] == type1) {
            begin2++;
        }
        if (begin2 >= tree.length) {
            return tree.length;
        }
        type2 = tree[begin2];
        end1 = begin2 - 1;
        int end2 = begin2;
        if (type2 == type1) return tree.length;
        
        for (int i = begin2 + 1; i < tree.length; ++i) {
            if (tree[i] == type1) {
                end1 = i;
            } else if (tree[i] == type2) {
                end2 = i;
            } else {
                int count = i - begin1;
                res = Math.max(res, count);
                begin1 = Math.min(end1, end2) + 1;
                type1 = tree[begin1];
                type2 = tree[i];
                begin2 = i;
                end1 = i-1;
                end2 = i;
            }
        }
        res = Math.max(res, tree.length - begin1);
        return res;
    }
}
