class Solution {
    public int totalFruit(int[] tree) {
        int n = tree.length;
        int[] dp = new int[n];
        int t = 0;
        int l = 0;
        int f = 0;
        int m = 0;
        for (int r = 0; r < n; r++) {
            if (dp[tree[r]] == 0) {
                t++;
                while (t > 2 && l < r) {
                    dp[tree[l]]--;
                    f--;
                    if (dp[tree[l]] == 0) {
                        t--;
                    }
                    l++;
                }
            }
            dp[tree[r]]++;
            f++;
            m = Math.max(m, f);
        }
        return m;
    }
}
