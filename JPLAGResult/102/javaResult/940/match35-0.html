class Solution {
    
    public int totalFruit(int[] tree) {
        HashMap<Integer, Integer> countMap = new HashMap<Integer, Integer>();
        int n = tree.length;
        int i = 0, j = 0;
        int ret = 0;
        while(true) {
            if(countMap.size() <= 2) {
                int tmp = j - i;
                if(ret<tmp) {
                    ret = tmp;
                }
                if(j==n) {
                    return ret;
                } else {
                    pick(countMap, tree[j]);
                    j += 1;
                }
            } else {
                drop(countMap, tree[i]);
                i += 1;
            }
        }
    }
    
    private void pick(HashMap<Integer, Integer> countMap, int type) {
        Integer count = countMap.get(type);
        if(count == null) {
            count = 0;
        }
        countMap.put(type, count + 1);
    }
    
    private void drop(HashMap<Integer, Integer> countMap, int type) {
        Integer count = countMap.get(type);
        if(count == null) {
            throw new RuntimeException("Cannot drop type " + type);
        }
        if(count.equals(1)) {
            countMap.remove(type);
        } else {
            countMap.put(type, count - 1);
        }
    }
}
