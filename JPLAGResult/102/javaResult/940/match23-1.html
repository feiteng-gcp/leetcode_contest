import java.util.*;

class Solution {
    public int totalFruit(int[] tree) {
        int result = 0;
        Map<Integer, Integer> map = new HashMap<>();
        for(int i = 0, j = 0; i < tree.length; i++) {
            map.merge(tree[i], 1, (a, b) -> a + b);
            while(map.size() > 2) {
                map.merge(tree[j], -1, (a, b) -> a + b);
                
if(map.get(tree[j]) == 0) {
                    map.remove(tree[j]);
                }
                
                j++;
            }
            
            result = Math.max(result, i - j + 1);
        }
        
        return result;
    }
}
