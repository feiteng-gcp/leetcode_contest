class Solution {
    public int totalFruit(int[] tree) {
        if (tree.length <= 2) {
            return tree.length;
        }
        Map<Integer, Integer> map = new HashMap<>();
        int ans = 0;
        int j = 0;
        for (int i = 0; i < tree.length; i++) {
            while (j < tree.length) {
                if (map.size() < 2 || map.size() == 2 && map.containsKey(tree[j])) {
                    if (!map.containsKey(tree[j])) {
                        map.put(tree[j], 0);
                    }
                    map.put(tree[j], map.get(tree[j]) + 1);
                    j++;
                } else 
{
                    break;
                }
            }
            if (map.size() <= 2 && j - i > ans) {
                ans = j - i;
            }
            map.put(tree[i], map.get(tree[i]) - 1);
            if (map.get(tree[i]) == 0) {
                
map.remove(tree[i]);
            }
        }
        return ans;
    }
}
