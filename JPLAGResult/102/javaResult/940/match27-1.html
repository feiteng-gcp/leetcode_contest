class Solution {
    public int totalFruit(int[] tree) {
        int n = tree.length;
        int[] maxOne = new int[n];
        
maxOne[n-1] = 1;
        for (int i=n-2; i>=0; i--) {
            if (tree[i] == tree[i+1]) {
                maxOne[i] = maxOne[i+1] + 1;
            } else {
                maxOne[i] = 1;
            }
        }
        
//        System.out.println(Arrays.toString(maxOne));
        
        int ans = 1;
        int[] maxTwo = new int[n];
        int[] other = new int[n];
        maxTwo[n-1] = 1;
        other[n-1] = tree[n-1];
        for (int i=n-2; i>=0; i--) {
            if (tree[i] == tree[i+1]) {
                maxTwo[i] = maxTwo[i+1] + 1;
                other[i] = other[i+1];
            } else {
                if (tree[i] == other[i+1]) {
                    maxTwo[i] = maxTwo[i+1] + 1;
                } else {
                    maxTwo[i] = maxOne[i+1] + 1;
                }
                other[i] = tree[i+1];
            
}
            ans = Math.max(ans, maxTwo[i]);
        }
        
//        System.out.println(Arrays.toString(maxTwo));
        
        return ans;
    }
}
