class Solution {
    
    public int totalFruit(int[] tree) {
        int[] count = new int[tree.length];
        int type = 0;
        int max = 0;
        int head = 0;
        for (int i = 0; i < tree.length; i++) {
            if (count[tree[i]]++ == 0) {
                type++;
            }
            
            if (type <= 2) {
                max = Math.max(max, i - head + 1);
            }
            
            while (type > 2) {
                if (count[tree[head++]]-- == 1) {
                    type--;
                }
            }
        }
        return max;
    }
    
}
