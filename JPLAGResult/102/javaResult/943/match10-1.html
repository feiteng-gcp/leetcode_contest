class Solution {
    private static final int MOD = 1000000007;
    
    private int mod(int a, int b) {
        return (int) ((long) a * b % MOD);
    }
    
    public int sumSubarrayMins(int[] A) {
        final int n = A.length;
        int[] left = new int[n];
        left[0] = -1;
        for (int i=1; i<n; i++) {
            int j = i-1;
            while (j != -1 && A[j] > A[i]) { j = left[j]; }
            left[i] = j;
        }
        
        int[] right = new int[n];
        right[n-1] = n;
        for (int i=n-2; i>=0; i--) {
            int j = i+1;
            while (j != n && A[j] >= A[i]) { j = right[j]; 
}
            right[i] = j;
        }
        
        long ans = 0;
        for (int i=0; i<n; i++) {
            int m = right[i] - left[i] - 1;
            int k = i - left[i] - 1;
            ans = (ans + mod(mod(m-k, k+1), A[i])) % MOD;
        }
        
        return (int) ans;
    }
}
