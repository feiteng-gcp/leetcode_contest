    public int sumSubarrayMins(int[] A) {
        return helper(A, 0, A.length-1)%(1000000007); 
    }
    // [l,r] inclusive 
    private int helper(int[] A, int l, int r){
        if(l > r) return 0; 
        if(l==r) return A[l]; 
        int min = 40000; 
        int idx = -1; 
        int i; 
        for( i=l ; i<=r ; i++){
            if(A[i]<min) {
                min = A[i]; 
                idx = i ; 
            }    
        }
        return (min%(1000000007) * (idx-l+1)%(1000000007) * (r-idx+1)%(1000000007))%(1000000007)+ (helper(A,l,idx-1)%(1000000007) + helper(A,idx+1, r)%(1000000007))%1000000007; 
    }
}
