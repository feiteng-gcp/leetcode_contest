class Solution {
	public int sumSubarrayMins(int[] A) {
		int res = 0;
		int MOD = (int) 1e9 + 7;
		

		for (int i = 0; i < A.length; i++) {
			int l = 0;
			int r = A.length - 1;
			int m = i;
			for (int j = i - 1; j >= 0; j--) {
				if (A[j] < A[i]) {
					l = j+1;
                    break;
				} 
			}
            for(int j = i+1; j<A.length; j++) {
                if(A[j] <= A[i]) {
                    r = j-1;
                    break;
                }
            }
            int fact = ((m - l + 1) * (r - m + 1)) % MOD;
			res += (fact * A[i]) % MOD;
            res %= MOD;
		}

		return res;
	}
}
