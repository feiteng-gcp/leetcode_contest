class Solution {
    public int sumSubarrayMins(int[] A) {
        return (int) (helper(A, 0, A.length - 1) % ((long) 
Math.pow(10, 9) + 7));
    }
    private long helper(int[] A, int start, int end) {
        if (start > end) return 0;
        int min = A[start];
        int minIndex = start;
        for (int i = start; i < end + 1; i++) 
{
            if (A[i] < min) {
                min = A[i];
                minIndex = i;
            }
        }
        int modNum = (int) Math.pow(10, 9) + 7;
        return (long) ((minIndex + 1 - start) * (end + 1 - minIndex)) * (long) A[minIndex]
                + helper(A, start, minIndex - 1) + helper(A, minIndex + 1, end);
    }
}
