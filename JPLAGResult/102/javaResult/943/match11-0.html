    public int sumSubarrayMins(int[] A) {
        int n = A.length;
        int[] leftSmaller = new int[n + 2];
        int[] rightSmaller = new int[n + 2];
        List<Integer> list = new ArrayList<>();
        list.add(0);
        for (int a : A) {
            list.add(a);
        }
        list.add(0);
        Stack<Integer> stack = new Stack<>();
        stack.add(0);
        for (int i = 1; i < list.size() - 1; i++) {
            while (list.get(stack.peek()) >= list.get(i)) {
                stack.pop();
            }
            leftSmaller[i] = stack.peek();
            stack.push(i);
        }
        // for (int left : leftSmaller) {
        //     System.out.print(left + " ");
        // }
        // System.out.println();
        stack.clear();
        stack.add(list.size() - 1);
        for (int i = list.size() - 2; i >= 1; i--) {
            while (list.get(stack.peek()) > list.get(i)) {
                stack.pop();
            }
            rightSmaller[i] = stack.peek();
            stack.push(i);
        }
        
        // for (int right : rightSmaller) {
        //     System.out.print(right + " ");
        // }
        // System.out.println();
        long ans = 0;
        for (int i = 1; i < list.size() - 1; i++) {
            int cnt = (i - leftSmaller[i]) * (rightSmaller[i] - i);
            // System.out.println(i + ", " + list.get(i) + ", " + leftSmaller[i] + ", " + rightSmaller[i] + ", " + cnt);
            ans = (ans + 1L * cnt * list.get(i)) % MOD;
        }
        return (int) ans;
    }
    
}
