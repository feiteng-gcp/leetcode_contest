class Solution {
    private final int MOD = (int) (1e9 + 7);
    public int sumSubarrayMins(int[] A) {
        int[] r = new int[A.length];
        int[] l = new int[A.length];
        for (int i = 0; i < A.length; i++) {
            int fast = i + 1;
            while (fast < A.length && A[fast] >= A[i]) {
                fast++;
            }
            r[i] = fast;
        }
        for (int i = A.length - 1; i >= 0; i--) {
            int fast = i - 1;
            while (fast >= 0 && A[fast] > A[i]) {
                fast--;
            }
            l[i] = fast;
        }
        long res = 0;
        for (int i = 0; i < A.length; i++) {
            long count = (r[i] - i) * (i - l[i]);
            res += count * A[i];
            res %= MOD;
        }
        return (int) res;
    }
}
