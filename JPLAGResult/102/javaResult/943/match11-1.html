class Solution {
    static long MOD = 1000000007L;
    
    public int sumSubarrayMins(int[] A) {
        int n = A.length;
        
        int[] le = new int[n];
        boolean[] same = new boolean[n]; 
        
        int ndq = -1;
        int[] dq = new int[n];
        for (int i = 0; i < n; ++i) {
            while (ndq >= 0 && A[i] <= A[dq[ndq]]) {
                if (A[i] == A[dq[ndq]]) same[i] = true;
                --ndq;
            }
            if (ndq == -1) le[i] = -1;
            else le[i] = dq[ndq];
            dq[++ndq] = i;
        }
        
        ndq = -1;
        int[] ri = new int[n];
        for (int i = n - 1; i >= 0; --i) {
            while (ndq >= 0 && A[i] <= A[dq[ndq]]) {
                --ndq;
            }
            if (ndq == -1) ri[i] = n;
            else ri[i] = dq[ndq];
            dq[++ndq] = i;
        }
        
        int mini = A[0];
        for (int i = 1; i < n; ++i) mini = Math.min(A[i], mini);
        
        long last = 0L;
        long ans = 0L; //((long)mini) * (((long)(n + 1) * (long)n / 2L) % MOD) % MOD;
        for (int i = 0; i < n; ++i) {
            //if (A[i] == mini) continue;
            if (same[i]) continue;
            
            long tmp = ri[i] - le[i] - 1;
            //System.out.printf("A[%d]=%d   left=%d   right=%d   len=%d\n", i, A[i], le[i], ri[i], tmp);
            tmp = (tmp * (tmp + 1L) / 2L) % MOD;
            
            last = 0L;
            if (le[i] != -1) last = A[le[i]];
            if (ri[i] != n) last = Math.max(last, A[ri[i]]);
            tmp = (tmp * (long)(A[i] - last)) % MOD;
            //System.out.printf("last=%d   add=%d\n", last, tmp);
            
            ans = (ans + tmp) % MOD;
        }
        
        return (int)ans;
    }
}
