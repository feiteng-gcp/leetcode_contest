class Solution {
  private final static int prime = 1000000007;
    public int sumSubarrayMins(int[] A) {
        Deque<Integer> deque = new ArrayDeque<>();
        deque.push(-1);
        int []leftVal = new int[A.length];
        for (int i = 0; i < A.length; i++) {
            while (deque.peek() != -1 && A[deque.peek()] >= A[i]) {
                deque.pop();
            }
            int left = i - deque.peek();
            leftVal[i] = left;

            deque.push(i);
        }
        deque.clear();
        int []rightVal = new int[A.length];
        deque.push(A.length);
        for (int i = A.length - 1; i >= 0; i--) {
            while (deque.peek() != A.length && A[deque.peek()] > A[i]) {
                deque.pop();
            }
            int right = deque.peek() - i;
            rightVal[i] = right;
           deque.push(i);
        }
        long total = 0;
        for (int i = 0; i < A.length; i++) {
            total = (total % prime + (1l * leftVal[i] * rightVal[i] * A[i] * 1l) % prime) % prime;
        }
        return (int)total;
    }
}
