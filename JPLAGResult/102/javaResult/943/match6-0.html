class Solution {
    public int sumSubarrayMins(int[] A) {
        long res = 0l;
        int mod = 1000000007;
        for(int i = 0 ; i < A.length ; i++){
            int left = 0, right = 0;
            int cur = i - 1;
            while(cur >= 0){
                if(A[i] < A[cur]) cur--;
                else break;
            }
            left = i - cur;
            cur = i + 1;
            while(cur < A.length){
                if(A[i] <= A[cur]) cur++;
                else break;
            }
            right = cur - i;
            res = res + (left * right) * A[i];
            //System.out.println(left + "  " + right + "  " + res);
            res %= mod;
        }
        return (int)res;
    }
}
