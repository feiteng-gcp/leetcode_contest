class Solution {
    public int sumSubarrayMins(int[] A) {
        Stack<Pair> stack = new Stack<>();
        long current = 0;
        
long result = 0;
        
        for (int i = 0; i < A.length; i++) {
            if (!stack.isEmpty() && A[i] >= stack.peek().val) {
                current += A[i];
                stack.push(new Pair(A[i], 1));
            }
            else {
                int count = 1;
                while (!stack.isEmpty() && A[i] < stack.peek().val) {
                    count += stack.peek().cnt;
                    current -= stack.peek().val * stack.peek().cnt;
                    stack.pop();
                }
                current += A[i] * count;
                stack.push(new Pair(A[i], count));
            }
            result += current;
        }
        
        return (int)(result % (1000000000 + 7));
    }
}


class Pair {
    int val;
    int cnt;
    Pair(int v, int c) {
        this.val = v;
        this.cnt = c;
    }
}
