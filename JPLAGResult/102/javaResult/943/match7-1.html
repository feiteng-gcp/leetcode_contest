class Solution {
    public int sumSubarrayMins(int[] A) {
        if(A == null || A.length == 0) {
            return 0;
        }
        int n = A.length;
        Deque<Integer> d = new ArrayDeque<>();
        int[] rightFirstLow = new int[n];
        for(int i = n - 1; i >= 0; i--) {
            while(!d.isEmpty() && A[d.peekFirst()] >= A[i]) {
                d.removeFirst();
            }
            if(d.isEmpty()) {
                rightFirstLow[i] = n;
            } else {
                rightFirstLow[i] = d.peekFirst();
            }
            d.addFirst(i);
        }
        
        int[] dp = new int[n]; //dp[i] is the sum of all subarray minimums from i to n
        for(int i = n - 1; i >= 0; i--) {
            dp[i] = (rightFirstLow[i] - i) * A[i];
            if(rightFirstLow[i] < n) {
                dp[i] += dp[rightFirstLow[i]];
            }
        
}
        long sum = 0;
        for(int i = 0; i < n; i++) {
            sum += dp[i];
        }
        return (int) (sum % 1000000007);
    }
}
