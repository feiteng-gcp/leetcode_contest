class Solution {    
    static int MOD = (int)1e9+7;

    static
    class Pair {
        int first, second ;
        public Pair(int f, int s) {
           first = f ;
           second = s ;
        }
    }
    
    public int sumSubarrayMins(int[] A) {
        long curSum = 0, res = 0;
        Deque<Pair> st = new LinkedList<>() ;
        int n = A.length ;
        for (int j = 0; j < n; j++)
        {
            if((!st.isEmpty()) && A[j] >= st.peek().first)
            {
                curSum += (A[j] % MOD);
                st.push(new Pair(A[j], 1));
            }else
            {
                int cnt = 1;
                while((!st.isEmpty()) && st.peek().first > A[j]) {
                    cnt += st.peek().second ;
                    curSum -= (st.peek().second * st.peek().first)  % MOD;
                    st.pop();
                }
                st.push(new Pair(A[j], cnt));
                curSum += (cnt * A[j]) % MOD;
            }
            res += (curSum % MOD) ;
        }

        return (int)(res % MOD) ;
    }
}
