class Solution {
     final long MOD7 = 1000000007;

    public int sumSubarrayMins(int[] A)
    {



        return  (int)(solve(0,A.length-1,A) %MOD7);
    }

    int idmin(int i, int j , int v[])
    {
        int id = i;
        for (int k = i; k <= j; k++)
            if (v[k] < v[id]) id = k;
        return id;
    }

    long solve(int i, int j, int[] v)
    {
        if (i > j) return 0;
        if (i == j) return v[i];
        else
        {
            int id = idmin(i,j,v);
            return (v[id]*(id-i+1)*(j+1-id))%MOD7 + solve(i,id-1,v)%MOD7 + solve(id+1,j,v)%MOD7;
        }
    }

}
