class Solution {
    private static final int MOD = 1000000007; 
    public int sumSubarrayMins(int[] A) {
        return computeSum(A, 0, A.length - 1);
    }
    
    int computeSum(int[] A, int start, int end) {
        if (start > end) return 0;
        
if (start == end) return A[start];
        
        int min = A[start], j = start;
        for (int i = start + 1; i <= end; i++) {
            if (A[i] < min) {
                j = i;
                min = A[i];
            }
        }
        
        return ((computeSum(A, start, j - 1) + computeSum(A, j + 1, end)) % MOD + ((end - j + 1) *  (1 + j - start) * min) % MOD) % MOD;
        
    }
}
