class Solution {
    public int sumSubarrayMins(int[] A) {
        int md = 1000000007;
        int[] st = new int[A.length];
        int[] end = new int[A.length];
        for (int i = 0; i < A.length; i++) {
            int s = i - 1;
            int e = i + 1;
            while (s >= 0 && A[s] >= A[i]) s--;
            while (e < A.length && A[e] > A[i]) e++;
            st[i] = s + 1;
            end[i] = e - 1;
        }
        int sum = 0;
        for (int i = 0; i < A.length; i++) {
            sum += (i - st[i] + 1) * (end[i] - i + 1) * A[i] % md;
            sum %= md;
        }
        return sum;
    }
}
