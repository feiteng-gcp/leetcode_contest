class Solution {
    public int[] sortArrayByParity(int[] A) {
       int start = 0, end = A.length - 1;
        int mid;
        while (start < end) {
            if (A[start] % 2 == 0) {
                start++;
            } else if (A[end] % 2 == 1) {
                end--;
            } else {
                mid = A[start];
                A[start] = A[end];
                A[end] = mid;
            }
        }
        return A; 
    }
}
