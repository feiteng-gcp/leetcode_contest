class Solution {
    public int[] sortArrayByParity(int[] A) {
        if (A == null) return null;
        int n = A.length;
        
int[] res = new int[n];
        int i = 0, j = res.length - 1;
        for (int k = 0; k < n; k++) {
            if (A[k] % 2 == 0) {
                res[i++] = A[k];
            } else {
                res[j--] = A[k];
            }
        }
        return res;
    }
}
