class Solution {
    public int[] sortArrayByParity(int[] A) {
        if(A == null || A.length == 0) return A;
        int len = A.length;
        int start = 0, end = len - 1;
        while(start < end){
            if(A[start] % 2 == 0) {
                start++;
            }
            else if(A[end] % 2 == 1){
                end--;
            }
            else{
                swap(A, start++, end--);
            }
        }
        return A;
    }
    private void swap(int[] A, int i, int j){
        int temp = A[i];
        A[i] = A[j];
        A[j] = temp;
    }
}
