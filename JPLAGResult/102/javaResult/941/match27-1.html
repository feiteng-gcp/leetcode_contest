class Solution {
    public int[] sortArrayByParity(int[] A) {
        
int lo = 0;
        int hi = A.length - 1;
        while (lo < hi) {
            if (A[lo] % 2 == 0) {
                lo++;
            } else if (A[hi] % 2 == 1) {
                hi--;
            } else {
                swap(A, lo, hi);
            }
        }
        return A;
    }
    
    private void swap(int[] A, int l, int h) {
        int temp = A[l];
        A[l] = A[h];
        A[h] = temp;
    }
}
