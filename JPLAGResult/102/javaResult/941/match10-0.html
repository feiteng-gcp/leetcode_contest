class Solution {
    public int[] sortArrayByParity(int[] A) {
        int[] res = new int[A.length];
        int i = 0, j = res.length-1;
        for (int k = 0; k < A.length; ++k) {
            if ((A[k] % 2 == 0)) {
                res[i++] = A[k];
            }
            else {
                res[j--] = A[k];
            }
        }
        return res;
    }
}
