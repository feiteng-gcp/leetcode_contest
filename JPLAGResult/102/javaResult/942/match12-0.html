class Solution {
    public int superpalindromesInRange(String L, String R) {
            Set<Integer> inc = new HashSet<>();
            long l = Long.parseLong(L);
            long r = Long.parseLong(R);
            List<Integer> all = getP((int) Math.sqrt(r) + 5);
            int er = 0;
            for (int i = 0; i < all.size(); i++) {
                int t = all.get(i);
                if (inc.contains(t)) continue;
                inc.add(t);
                long y = (long) t * t;
                if (l <= y && y <= r && isP((long) t * t)) {
                    er ++;
                }
            }
            return er;
        }


        public boolean isP(long v) {
            String s = Long.toString(v);
            for (int i = 0; i < s.length() / 2; i++) {
                if (s.charAt(i) != s.charAt(s.length() - 1 - i)) return false;
            }
            return true;
        }

        public List<Integer> getP(int max) {
            List<Integer> res = new ArrayList<>();
            for (int i = 0; i < 10; i++) {
                int number = i;
                res.add(revCopy(number));
                res.add(revCopy1(number));
                for (int j = 0; j < 10; j++) {
                    if (number == 0) continue;
                    int number1 = number * 10 + j;
                    res.add(revCopy(number1));
                    res.add(revCopy1(number1));
                    if (res.get(res.size() - 2) > max) continue;
                    for (int k = 0; k < 10; k++) {
                        if (number1 == 0) continue;
                        int number2 = number1 * 10 + k;
                        res.add(revCopy(number2));
                        res.add(revCopy1(number2));
                        if (res.get(res.size() - 2) > max) continue;
                        for (int l = 0; l < 10; l++) {
                            if (number2 == 0) continue;
                            int number3 = number2 * 10 + l;
                            res.add(revCopy(number3));
                            res.add(revCopy1(number3));
                            if (res.get(res.size() - 2) > max) continue;
                            for (int m = 0; m < 10; m++) {
                                if (number3 == 0) continue;
                                int number4 = number3 * 10 + m;
                                res.add(revCopy1(number4));
                            }
                        }
                    }
                }
            }
            return res;
        }

        public int revCopy(int a) {
            String s = Integer.toString(a);
            StringBuilder sb = new StringBuilder();
            sb.append(s);
            StringBuilder sb2 = new StringBuilder(sb);
            sb2.reverse();
            sb.append(sb2);

            return Integer.parseInt(sb.toString());
        }

        public int revCopy1(int a) {
            String s = Integer.toString(a);
            StringBuilder sb = new StringBuilder();
            sb.append(s);
            StringBuilder sb2 = new StringBuilder(sb);
            sb2.reverse();
            sb.append(sb2.substring(1));

            return Integer.parseInt(sb.toString());
        }
}
