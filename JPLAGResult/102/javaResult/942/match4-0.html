class Solution {
    public int superpalindromesInRange(String L, String R) {
        
        long Lval=Long.parseLong(L);
        long Rval=Long.parseLong(R);
        int maxlen=R.length()/2+1;
        String num="0";
        int count=0;
        long n=0;
        while(n<=Rval)
        {
            n=Long.parseLong(num);
            n=n*n;
            if(n>=Lval && n<=Rval)
            {
                if(ispa(Long.toString(n)))
                {
                    count++;
                }
            }
            num=nextpa(num);
        }
        return count;
    }
    
    public boolean ispa(String s)
    {
        for(int i=0;i<s.length()/2;i++)
        {
            if(s.charAt(i)!=s.charAt(s.length()-1-i))
            {
                return false;
            }
        }
        return true;
    }
    
    public String nextpa(String s)
    {
        if(s.length()<=1)
        {
            if(s.charAt(0)=='9')
            {
                return "11";
            }
            return Long.toString(Long.parseLong(s)+1);
        }
        if(s.length()%2==0)
        {
            Long temp=Long.parseLong(s.substring(0,s.length()/2))+1;
            String lef=Long.toString(temp);
            
            
            String rig="";
            for(int i=0;i<lef.length();i++)
            {
                rig=lef.charAt(i)+rig;
            }
            if(lef.length()>s.length()/2)
            {
                rig=rig.substring(1);   
            }
            return lef+rig;
        }else
        {
            Long temp=Long.parseLong(s.substring(0,s.length()/2+1))+1;
            String lef=Long.toString(temp);
            String rig="";
            for(int i=0;i<lef.length()-1;i++)
            {
                rig=lef.charAt(i)+rig;
            }
            if(lef.length()>s.length()/2+1)
            {
                rig=rig.substring(1);   
            }
            return lef+rig;
        }
        
    }
}
