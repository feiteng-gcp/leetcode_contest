class Solution {
    int ans;
    public int superpalindromesInRange(String L, String R) {
        int ll = L.length(), lr = R.length();
        ans = 0;
        helper(Math.max(((ll + 1) / 2 + 1) / 2, 1), Math.max(((lr + 1) / 2 + 1) / 2, 1), Long.valueOf(L), Long.valueOf(R));
        return ans;
    }
    
    void helper(int l, int r, Long L, Long R) {
        for (int i = l; i <= r; i++)
            helper2(0, i, L, R, 0);
    }
    
    void helper2(int d, int l, Long L, Long R, int n) {
        if (d == l) {
            String str = String.valueOf(n);
            StringBuilder sb = new StringBuilder(str);
            int ls = str.length();
            for (int i = 0; i < ls; i++)
                sb.append(str.charAt(ls - 1 - i));
            long res = Long.valueOf(sb.toString());
            res *= res;
            if (res >= L && res <= R && isPa(res)) {
                ans++;
            }
            sb = new StringBuilder(str);
            for (int i = 0; i < ls - 1; i++)
                sb.append(str.charAt(ls - 2 - i));
            res = Long.valueOf(sb.toString());
            res *= res;
            if (res >= L && res <= R && isPa(res)) {
                ans++;
            }
        } else {
            for (int i = n == 0 ? 1 : 0; i < 10; i++)
                helper2(d + 1, l, L, R, n * 10 + i);
        }
    
}
    
    boolean isPa(long n) {
        String s = String.valueOf(n);
        for (int i = 0; i < s.length() / 2; i++)
            if (s.charAt(i) != s.charAt(s.length() - 1 - i)) return false;
        return true;
    }
}
