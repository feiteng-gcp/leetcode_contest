class Solution:
    def totalFruit(self, tree):
        """
        :type tree: List[int]
        :rtype: int
        """
        hashset = collections.defaultdict(int)
        i = 0
        maxLen = 0
        for j in range(len(tree)):
            hashset[tree[j]] += 1
            while len(hashset)>2:
                hashset[tree[i]] -= 1
                if hashset[tree[i]] == 0:
                    del hashset[tree[i]]
                i += 1
            maxLen = max(maxLen,j-i+1)
        return maxLen
