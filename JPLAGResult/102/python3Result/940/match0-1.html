from collections import defaultdict
class Solution:
    def totalFruit(self, A):
        basket = defaultdict(int)
        l, r = 0, 0
        res = 0
        while r < len
(A):
            basket[A[r]] += 1
            while len(basket) > 2:
                basket[A[l]] -= 1
                if basket[A[l]] == 0:
                    del basket[A[l]]
                l += 1
            res = max(res, r - l + 1)
            r += 1
        return res
