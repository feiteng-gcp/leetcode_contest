class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        int start = (low+"").length(), end = (high +"").length();
        
        int[] temp = new int[start];
        for(int i = 0; i < start; i++) temp[i] = i+1;
        int cur = Integer.valueOf(getStr(temp));
        List<Integer> res = new ArrayList<>();
        
        
        while(cur  <= high){
            if(cur >=low && cur <= high) res.add(cur);
            if(temp[temp.length-1] <9){
                for(int i = 0;  i< temp.length; i++) temp[i] = temp[i]+1;
            }else if(temp[temp.length-1] == 9){
                temp = new int[temp.length+1];
                for(int i = 0; i < temp.length; i++) temp[i] = i+1;
                
            }
            
            if(temp.length == 10) break;
            cur = Integer.valueOf(getStr(temp)); 
        }
        return res;
    }
    
    String getStr(int[] temp){
        StringBuilder sb = new StringBuilder();
        for(int i: temp){
            sb.append(i);
        }
        return sb.toString();
    }
}
