class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> res = new ArrayList<>();
        int l1 = 1;
        int l2 = 1;
        int d = 10;
        int last1 = 0;
        while (low / d != 0) {
            l1++;
            last1 = low % d;
            d *= 10;
        }
        d = 10;
        while (high / d != 0) {
            l2++;
            d *= 10;
        }
        for (int len = l1; len <= l2; len++) {
            for (int f = 1; f <= 10 - len; f++) {
                int t = 0;
                for (int tt = 0; tt < len; tt++) {
                    t = t * 10 + f + tt;
                }
                // System.out.println(t);
                if (t >= low && t <= high) {
                    res.add(t);
                }
            }
        }
        Collections.sort(res);
        return res;
    }
}
