class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> res = new ArrayList();
        int lowDigits = (int)(Math.log10(low)+1);
        int highDigits = (int)(Math.log10(high)+1);
        
for(int digits=lowDigits; digits<=highDigits; digits++){
            int cur = 0;
            int add = 0;
            for(int i=1; i<=digits; i++){
                cur = cur*10+i;
                add = add*10+1;
            }
            for(int i=0; i<10-digits; i++){
                
if(cur>=low && cur<=high){
                    res.add(cur);
                }
                cur += add;
            }
        }
        return res;
    } 
}
