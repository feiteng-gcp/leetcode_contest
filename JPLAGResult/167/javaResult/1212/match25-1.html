class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> res = new ArrayList<>();
        int n1 = 0, n2 = 0;
        int l = low, h = high;
        while (l != 0) {
            n1++;
            l /= 10;
        }
        
        
while (h != 0) {
            n2++;
            h /= 10;
        }
        for (int i = n1; i <= n2; i++) {
            for (int j = 1; j + i - 1 <= 9; j++) {
                int temp = 0;
                for (int k = 1; k <= i; k++) {
                    temp = temp * 10 + (j + k - 1);
                }
                if (temp >= low && temp <= high) res.add(temp)
;
            }
        }
        return res;
    }
}
