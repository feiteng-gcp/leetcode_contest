class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> ans = new ArrayList<>();
        for(int i=1;i<=9;i++) {
            helper(ans, i, low, high, i+1);
        }
        Collections.sort(ans);
        return ans;
    }
    
    private void helper(List<Integer> ans, long num, int low, int high, int toAdd) {
        if(num>high) return;
        if(num>=low) ans.add((int)num);
        if(toAdd==10) return;
        helper(ans, num*10+toAdd, low, high, toAdd+1);
        
        
    }
}
