class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        ArrayList <Integer> ans = new ArrayList();
        for(int i = 1; i < 10; i++)
        {
            for(int j = i + 1; j < 10; j++)
            {
                int tmp = 0;
                for(int k = i; k <= j; k++)
                    tmp = tmp * 10 + k;
                if(tmp >= low && tmp <= high)
                    ans.add(tmp)
;
            }
        }
        Collections.sort(ans);
        return ans;
    }
}
