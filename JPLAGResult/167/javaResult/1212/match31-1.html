class Solution {
   	public List<Integer> sequentialDigits(int low, int high) {
		List<Integer> res = new ArrayList<>();
		for (int len = 2; len <= 9; len++) {
			for (int i = 1; i + len <= 10; i++) {
				int cur = 0;
				for (int j = i; j < i + len; j++) {
					cur = cur * 10 + j;
				}
				if (cur >= low && cur <= hig
h)
					res.add(cur);
			}
		}
		Collections.sort(res);
		return res;
	}
}
