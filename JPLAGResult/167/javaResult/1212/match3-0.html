class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> li = new ArrayList<>();
        
        for(int i=1;i<=9;++i){
            for(int j=i;j<=9;++j){
                long m = 0;
                for(int f=i;f<=j;++f){
                    m = m*10+f;
                }
                if(m>=low&&m<=high){
                    li.add((int)m);
                }
                
            }
        }
        Collections.sort(li);
        return li;
    }
}
