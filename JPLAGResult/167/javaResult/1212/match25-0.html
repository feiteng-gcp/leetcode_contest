class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> ans = new LinkedList<>();
        int lenHigh = 0;
        int lenLow = 0;
        int tmp = high;
        while(tmp !=0) {
            lenHigh++;
            tmp/=10;
        }
        tmp = low;
        int min = 1;
        while(tmp != 0) {
            lenLow++;
            tmp/=10;
        }

        for(int i = lenLow;i<=lenHigh;i++) {
            for(int j = min;j<=9-i+1;j++) {
                int c = 0;
                for(int k = 0;k<i;k++) {
                    c = c*10+(k+j);
                }
                if(c>=low && c<=high) {
                    ans.add(c);
                }
            }
        }
        return ans;
    }
}
