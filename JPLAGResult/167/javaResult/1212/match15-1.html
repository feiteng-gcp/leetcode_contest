class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> res = new ArrayList<>();
        List<Integer> smallHigh = sequentSmall(high);
        for(int n: smallHigh){
            if(n>=low) res.add(n);
        }
        return res;
    }
    
    private List<Integer> sequentSmall(int high){
        List<Integer> res = new ArrayList<>();
        Queue<Integer> q = new LinkedList<>();
        for(int i=1; i<10; i++) q.offer(i);
        while(!q.isEmpty()){
            int cur = q.poll();
            res.add(cur);
            
if(cur%10!=9 && cur*10+cur%10+1<=high){
                q.offer(cur*10+cur%10+1);
            }
        }
        return res;
    }
}
