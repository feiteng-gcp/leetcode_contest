class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> ans = new ArrayList<>();
        for (int i = 1; i <= 9; i++) {
            for (int j = i; j <= 9; j++) {
                int t = dfs(i, j);
                if (t >= low && t <= high) {
                    ans.add(t);
                }
            }
        }
        Collections.sort(ans);
        return ans;
    }
    
    private int dfs(int l, int r) {
        int n = 0;
        for (int i = l; i <= r; i++) {
            n = n * 10 + i;
        }
        return n;
    }
}
