class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> res = new ArrayList<>();
        for (int d = 1; d < 10; d++) {
            dfs(0, d, low, high, res);
        }
        Collections.sort(res);
        return res;
    }
    
    private void dfs(int n, int d, int lo, int hi, List<Integer> res) {
        if (n >= lo && n <= hi) 
res.add(n);
        if (n > hi || d > 9) return;
        dfs(n * 10 + d, d + 1, lo, hi, res);
    }
}
