class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        ArrayList<Integer> ans = new ArrayList<Integer>();
        for (int st = 1; st <= 9; ++st) {
            int val = 0;
            for (int d = st; d < 10; ++d) {
                val = 10 * val + d;
                if (low <= val && val <= high) ans.add(val)
;
            }
        }
        Collections.sort(ans);
        return ans;
    }
}
