class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> res = new ArrayList<>();
        List<Integer> h = sequent(high);
        for(int n: h){
            if(n>=low) res.add(n);
        }
        return res;
    }
    
    private List<Integer> sequent(int high){
        List<Integer> res = new ArrayList<>();
        Queue<Integer> q = new LinkedList<>();
        for(int i=1; i<10; i++) q.offer(i);
        while(!q.isEmpty()){
            int cur = q.poll();
            res.add(cur);
            int last = cur%10;
            if(last!=9 && cur*10+last+1<=high){
                q.offer(cur*10+last+1);
            }
        }
        return res;
    }
}
