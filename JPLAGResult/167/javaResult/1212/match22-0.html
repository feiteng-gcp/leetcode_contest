class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> res = new ArrayList();
        
        for (int i = 1; i < 9; i++) {
            long tmp = i;
            for (int j = 0; j < 10; j++) {
                
                tmp = 10 * tmp + (tmp%10 + 1);
                if (tmp > high || tmp % 10 == 0) break;
                if (tmp>=low && tmp<=high) {
                    res.add((int)tmp);
                }
            }
        }
        
        Collections.sort(res);
        return res;
    }
    
   
}
