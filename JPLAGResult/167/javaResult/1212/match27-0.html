class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> res = new ArrayList<>();
        
        for (int i = 1; i <= 8; i++) {
            helper(res, i, i + 1, low, high);
        }
        Collections.sort(res);
        return res;
    }
    
    private void helper(List<Integer> res, int cur, int i, int low, int high) {
        if (cur >= low && cur <= high) {
            res.add(cur);
        }
        if (cur > high || i >= 10) {
            return;
        }
        
        cur = cur * 10 + i;
        helper(res, cur, i + 1, low, high);
        
    }
}
