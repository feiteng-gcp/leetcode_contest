	class Solution {
	    public List<Integer> sequentialDigits(int low, int high) {
	    	List<Integer> ret = new ArrayList<>();
	        for(int d = 1;d <= 9;d++){
	        	int val = 0;
	        	for(int e = d;e <= 9;e++){
	        		val = val * 10 + e;
	        		if(low <= val && val <= high){
	        			ret.add(val);
	        		}
	        	}
	        }
	        Collections.sort(ret);
	        return ret;
	    }
	}
