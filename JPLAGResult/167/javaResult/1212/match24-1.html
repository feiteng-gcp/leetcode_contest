class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        int lowLen = (int) Math.log10(low) + 1;
        int highLen = (int) Math.log10(high) + 1;
        
List<Integer> result = new ArrayList<>();
        
        for (int i = 1; i <= 9; i++) {
            for (int len = lowLen; len <= highLen && i + len <= 10; len++) {
                int num = construct(i, len);
                if (num >= low && num <= high) {
                    result.add(num);
                }
            }
        }
        Collections.sort(result);
        return result;
    }
    
    private int construct(int start, int len) {
        int n = start;
        for (int i = 1; i < len; i++) {
            n = n * 10 + start + i;
        }
        return n;
    }
}
