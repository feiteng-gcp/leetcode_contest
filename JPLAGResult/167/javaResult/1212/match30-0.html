class Solution {
    public List<Integer> sequentialDigits(int low, int high) {
        List<Integer> res = new ArrayList<>();
        for (int i = 1; i <= 8; i++) {
            for (int j = i + 1; j <=9; j++) {
                int val = 0;
                for (int k = i; k <= j; k++) {
                    val = val * 10 + k;
                }
                if (val >= low && val <= high)
                    res.add(val);
            }
        }
        Collections.sort(res);
        return res;
    }
}
