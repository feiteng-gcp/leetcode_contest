class Solution {
    
    int[][] dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    int Max = 1000000007;

    public int shortestPath(int[][] g, int k) {
        int m = g.length;
        int n = g[0].length;
        Deque<int[]> pq = new ArrayDeque<>();
        boolean[][][] seen = new boolean[m][n][k + 1];
        pq.offer(new int[]{0, 0, 0, g[0][0]});

        int min = Max;
        while (!pq.isEmpty()) {
            int[] top = pq.poll();
            int i = top[0];
            int j = top[1];
            int dist = top[2];
            int ck = top[3];
            if (ck > k) {
                continue;
            }
            if (i == m - 1 && j == n - 1) {
                return dist;
            }
            for (int[] d : dirs) {
                int ni = i + d[0];
                int nj = j + d[1];
                if (ni >= 0 && ni < m && nj >= 0 && nj < n) {
                    int ak = g[ni][nj];
                    int nk = ck + ak;
                    if (nk <= k && !seen[ni][nj][nk]) {
                        seen[ni][nj][nk] = true;
                        pq.offer(new int[]{ni, nj, dist + 1, nk});
                    }
                }
            }
        }
        return min >= Max ? -1 : min;
    }

}
