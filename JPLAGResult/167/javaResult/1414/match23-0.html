class Solution {
    private static final int[][] DELTAS = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    
    public int shortestPath(int[][] grid, int k) {
        int m = grid.length;
        int n = grid[0].length;
        Queue<int[]> queue = new LinkedList<>();
        boolean[][] visited = new boolean[m * n][k + 1];
        queue.offer(new int[]{0, 0});
        visited[0][0] = true;
        int s = 0;
        while (!queue.isEmpty()) {
            int qs = queue.size();
            for (int i = 0; i < qs; i++) {
                int[] curr = queue.poll();
                int r = curr[0] / n;
                int c = curr[0] % n;
                if (r == m - 1 && c == n - 1) {
                    return s;
                }
                for (int[] delta : DELTAS) {
                    int nr = r + delta[0];
                    int nc = c + delta[1];
                    if (nr < 0 || nr >= m || nc < 0 || nc >= n) {
                        continue;
                    }
                    int np = nr * n + nc;
                    int no = grid[nr][nc] == 0 ? curr[1] : curr[1] + 1;
                    if (no > k || visited[np][no]) {
                        continue;
                    }
                    queue.offer(new int[]{np, no});
                    visited[np][no] = true;
                }
            }
            s++;
        }
        return -1;
    }
}
