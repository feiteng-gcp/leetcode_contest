class Solution {
    
    private int[][] adj = {
        { -1, 0 },
        { 0, 1 },
        { 1, 0 },
        { 0, -1 }
    };
    public int shortestPath(int[][] matrix, int k) {
        int n = matrix.length;
        int m = matrix[0].length;

        Set<Node> set = new HashSet<>(); // contains the visited nodes
        Queue<Node> queue = new PriorityQueue<>(11, new Comparator<Node>() {
            @Override
            public int compare(
Node node1, Node node2) {
                return node1.distance - node2.distance;
            }
        }); // contains the current checking nodes

        // initial state
        Node start = new Node(0, 0, 0, 0);
        set.add(start);
        queue.add(start);

        while (!queue.isEmpty()) {
            Node node = queue.poll(); // get the closest node

            // check if done
            if (node.row == n - 1 && node.column == m - 1) return node.distance;

            // get adjacent nodes
            for (int[] pos : adj) {
                int row = node.row + pos[0];
                int col = node.column + pos[1];
                if (row >= 0 && row < n && col >= 0 && col < m) {
                    // in the matrix
                    if (matrix[row][col] == 1 && node.count == k) continue; // can't break more walls

                    // if road -> same k
                    // if wall -> k + 1
                    Node next = new Node(row, col, matrix[row][col] == 1 ? node.count + 1 : node.count, node.distance + 1);

                    // check if visited or not
                    if (set.contains(next)) continue;

                    // no need to update the distance of the unvisited nodes since all the edge weight is 1 and they will be visited later with the best distance anyway
                    queue.add(next);
                    set.add(next);
                }
            }
        }

        return -1;    
    }
    
     static class Node {
        int row, column;
        int count; // number of walls broken
        int distance;

        Node(int row, int column, int count, int distance) {
            this.row = row;
            this.column = column;
            this.count = count;
            this.distance = distance;
        }

        @Override
        public int hashCode() {
            return Objects.hash(row, column, count);
        }

        @Override
        public boolean equals(Object o) {
            if (!(o instanceof Node)) return false;
            Node node = (Node) o;
            return (row == node.row && column == node.column && count == node.count);
        }
    }
}
