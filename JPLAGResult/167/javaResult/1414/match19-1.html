class Solution {
    int[][] dirs = new int[][]{{1,0}, {-1,0}, {0,1}, {0,-1}};
    Integer[][][] cache;

    public int shortestPath(int[][] grid, int k) {
        int m = grid.length, n = grid[0].length;
        cache = new Integer[m][n][k + 1];
        int res = dfs(grid, k, 0, 0, new boolean[m][n]);
        if (res == Integer.MAX_VALUE) return -1;
        return res;
    
}
    
    int dfs(int[][] grid, int k, int i, int j, boolean[][] visited) {
        if (i < 0 || i >= grid.length || j < 0 || j >= grid[0].length) return Integer.MAX_VALUE;
        if (i == grid.length - 1 && j == grid[0].length - 1) return 0;
        if (grid[i][j] == 1 && k <= 0) return Integer.MAX_VALUE;
        if (visited[i][j]) return Integer.MAX_VALUE;
        if (cache[i][j][k] != null) return cache[i][j][k];
        if (grid[i][j] == 1) k--;
        int res = Integer.MAX_VALUE;
        visited[i][j] = true;
        
for(int[] d : dirs) {
            int cur = dfs(grid, k, i + d[0], j + d[1], visited);
            if (cur == Integer.MAX_VALUE) continue;
            res = Math.min(res, 1 + cur);
        }
        visited[i][j] = false;
        cache[i][j][k] = res;
        return res;
    }
    
    
}
