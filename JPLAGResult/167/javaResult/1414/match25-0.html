class Solution {

    int[] dx = new int[]{-1, 0, 0, 1};
    int[] dy = new int[]{0, -1, 1, 0};

    public int shortestPath(int[][] grid, int k) {
      int n = grid.length, m = grid[0].length;
      boolean[][][] vis = new boolean[n][m][k+1];
      Queue<int[]> q = new LinkedList<>();
      q.add(new int[]{0, 0, 0});
      vis[0][0][0] = true;
      int res = 0;
      while(!q.isEmpty()) {
        int sz = q.size();
        for(int z=0; z<sz; z++) {
          int[] c = q.poll();
          if(c[0] == n-1 && c[1] == m-1) return res;
          for (int i = 0; i < 4; i++) {
            int nx = c[0] + dx[i];
            int ny = c[1] + dy[i];
            if(nx >= 0 && nx < n && ny >= 0 && ny < m) {
              if(grid[nx][ny] == 0 && !vis[nx][ny][c[2]]) {
                q.add(new int[]{nx, ny, c[2]});
                vis[nx][ny][c[2]] = true;
              }
              if(grid[nx][ny] == 1 && c[2] < k && !vis[nx][ny][c[2]+1]) {
                q.add(new int[]{nx, ny, c[2]+1});
                vis[nx][ny][c[2]+1] = true;
              }
            }
          }
        }
        res++;
      }
      return -1;
    }
  }
