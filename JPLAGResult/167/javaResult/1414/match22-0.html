class Solution {
    private int[][] dirs = new int[][]{{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    public int shortestPath(int[][] grid, int k) {
        int m = grid.length, n = grid[0].length;
        Queue<int[]> q = new LinkedList<int[]>();
        q.add(new int[]{0, 0, 0});
        boolean[][][] visited = new boolean[grid.length][grid[0].length][k + 1];
        visited[0][0][0] = true;
        int step = 0;
        while(!q.isEmpty()) {
            int size = q.size();
            for(int i = 1; i <= size; i++) {
                int[] cur = q.poll();
                if(cur[0] == m - 1 && cur[1] == n - 1) return step;
                for(int[] dir: dirs) {
                    int px = cur[0] + dir[0], py = cur[1] + dir[1];
                    if(px < 0 || px >= grid.length || py < 0 || py >= grid[0].length || visited[px][py][cur[2]]) continue;
                    if(grid[px][py] == 1 && cur[2] == k) continue;
                    visited[px][py][cur[2]] = true;
                    if (grid[px][py] == 1) q.add(new int[]{px, py, cur[2] + 1});
                    else q.add(new int[]{px, py, cur[2]});
                }
            }
            step++;
        }
        return -1;
     }
}
