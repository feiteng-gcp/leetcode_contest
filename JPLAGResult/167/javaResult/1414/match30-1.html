class Solution {
    
private int[][] moves = {{0,-1},{-1, 0},{0, 1},{1, 0}};
    public int shortestPath(int[][] grid, int k) {
        int m = grid.length, n = grid[0].length;
        
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[]{0, 0, k});
        boolean[][][] visit = new boolean[m][n][k+1];
        int steps = 0;
        visit[0][0][k] = true;
        
while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int[] cur = queue.poll();
                if (cur[0] == m-1 && cur[1] == n-1) return steps;
                for (int[] d : moves) {
                    int nx = cur[0] + d[0];
                    int ny = cur[1] + d[1];
                    if (nx < 0 || nx >= m || ny < 0 || ny >= n) continue;
                    if (grid[nx][ny] == 1 && cur[2] == 0) continue;
                    int t = cur[2]-grid[nx][ny];
                    
if (visit[nx][ny][t]) continue;
                    visit[nx][ny][t] = true;
                    queue.offer(new int[]{nx, ny, t});
                }
            }
            steps++;
        }
        return -1;
    }
}
