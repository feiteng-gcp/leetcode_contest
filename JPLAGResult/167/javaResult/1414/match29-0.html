class Solution {
    int [][] dirs = new int[][]{{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    public int shortestPath(int[][] grid, int k) {
        // 0 -> r, 1 -> c, 2 -> rem, 3 -> moves
        int n = grid.length;
        int m = grid[0].length;
        PriorityQueue<int[]> q = new PriorityQueue<int[]>(new Comparator<int[]>(){
            @Override
            public int compare(int [] a, int [] b){
                return a[3] - b[3];
            }
        });
        int [][][] visited = new int[n][m][k + 1];
        for(int [][] cur1: visited){
            for(int [] cur2: cur1){
                Arrays.fill(cur2, Integer.MAX_VALUE);
            }
        }
        q.offer(new int[]{0, 0, k, 0});
        visited[0][0][k] = 0;
        while(q.size() > 0){
            int [] cur = q.poll();
            if(cur[0] == n - 1 && cur[1] == m - 1){
                return cur[3];
            }
            for(int [] dir: dirs){
                int nr = cur[0] + dir[0];
                int nc = cur[1] + dir[1];
                if(nr < 0 || nr >= n || nc < 0 || nc >= m){
                    continue;
                }
                if(grid[nr][nc] == 1 && cur[2] > 0 && visited[nr][nc][cur[2] - 1] > cur[3] + 1){
                    visited[nr][nc][cur[2] - 1] = cur[3] + 1;
                    q.offer(new int[]{nr, nc, cur[2] - 1, cur[3] + 1});
                }
                if(grid[nr][nc] == 0 && visited[nr][nc][cur[2]] > cur[3] + 1){
                    visited[nr][nc][cur[2]] = cur[3] + 1;
                    q.offer(new int[]{nr, nc, cur[2], cur[3] + 1});
                }
            }
        }
        return -1;
    }
}
