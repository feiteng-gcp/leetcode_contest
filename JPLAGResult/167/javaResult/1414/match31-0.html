class Solution {
    private static final int[][] DIRS = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public int shortestPath(int[][] grid, int k) {
        int m = grid.length, n = grid[0].length;
        int[][] dp = new int[m][n];
        for (int[] row : dp) {
            Arrays.fill(row, Integer.MAX_VALUE);
        }
        dp[0][0] = 0;
        Queue<Pair> queue = new LinkedList<>();
        queue.offer(new Pair(0, 0, 0));
        int step = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                Pair cur = queue.poll();
                if (cur.x == m - 1 && cur.y == n - 1) {
                    return step;
                }
                for (int[] dir : DIRS) {
                    int nextX = cur.x + dir[0];
                    int nextY = cur.y + dir[1];
                    if (nextX >= 0 && nextX < m && nextY >= 0 && nextY < n) {
                        if (dp[nextX][nextY] == Integer.MAX_VALUE) {
                            if (grid[nextX][nextY] == 1) {
                                if (cur.count + 1 <= k) {
                                    queue.offer(new Pair(nextX, nextY, cur.count + 1));
                                    dp[nextX][nextY] = cur.count + 1;
                                }
                            } else {
                                if (cur.count < dp[nextX][nextY] && cur.count <= k) {
                                    queue.offer(new Pair(nextX, nextY, cur.count));
                                    dp[nextX][nextY] = cur.count;
                                }
                            }
                        } else {
                            if (grid[nextX][nextY] == 1) {
                                if (cur.count + 1 < dp[nextX][nextY] && cur.count + 1 <= k) {
                                    queue.offer(new Pair(nextX, nextY, cur.count + 1));
                                    dp[nextX][nextY] = cur.count + 1;
                                }
                            } else {
                                if (cur.count < dp[nextX][nextY] && cur.count <= k) {
                                    queue.offer(new Pair(nextX, nextY, cur.count));
                                    dp[nextX][nextY] = cur.count;
                                }
                            }
                        }
                    }
                }
            }
            step++;
        }
        return -1;
        
    }
    static class Pair{
        int x;
        int y;
        int count;
        Pair(int x, int y, int count) {
            this.x = x;
            this.y = y;
            this.count = count;
        }
    }
}
