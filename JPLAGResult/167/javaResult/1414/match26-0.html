class Solution {
    public int shortestPath(int[][] grid, int k) {
        Deque<int[]> queue = new LinkedList<>();
        int[][] dirs = new int[][]{{1,0},{0,1},{-1,0},{0,-1}};
        Set<Integer> visited = new HashSet<>();
        queue.add(new int[]{0,0,0});
        visited.add(0);
        int steps = 0;
        int m = grid.length;
        int n = grid[0].length;
        while (!queue.isEmpty()){
            int size = queue.size();
            for (int i=0; i<size; i++) {
                int[] curr = queue.poll();
                int x = curr[0];
                int y = curr[1];
                if (x == m-1 && y == n-1) return steps;
                int eli = curr[2];
                for (int j=0; j<4; j++) {
                    int next_x = x + dirs[j][0];
                    int next_y = y + dirs[j][1];
                    if (next_x>=0 && next_x < m && next_y>=0 && next_y < n) {
                        if (eli + grid[next_x][next_y] <= k) {
                            if (next_x == m-1 && next_y == n-1) {
                                return steps+1;
                            }
                            int key = next_x*10000+next_y*100+eli+grid[next_x][next_y];
                            if (!visited.contains(key)) {
                                visited.add(key);
                                queue.add(new int[]{next_x,next_y,eli+grid[next_x][next_y]});
                            } 
                        }
                    }
                }
            }
            steps++;
        }
        return -1;
    }
}
