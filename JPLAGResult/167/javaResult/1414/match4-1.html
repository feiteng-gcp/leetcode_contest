class Solution {
    
    int dx[] = {0,0,1,-1};
    int dy[] = {1,-1,0,0};
    public int shortestPath(int[][] g, int k) {
        
        
        int qx[] = new int[1000000];
        int qy[] = new int[1000000];
        int qz[] = new int[1000000];
        
        
        int s = 0;int e = 0;
        int m = g.length;
        int n = g[0].length;
        
        int dp[][][]=  new int[m][n][k+1];
        
        
        for(int i=0;i<m;++i) for(int j=0;j<n;++j){
            Arrays.fill(dp[i][j],100000000);
        }
        dp[0][0][0] = 0;
        
        qx[e] = 0;
        qy[e] = 0;
        qz[e] = 0;
        e++;
        
        while(s<e){
            int cx = qx[s];
            int cz = qz[s];
            int cy = qy[s];s++;
            
            for(int u=0;u<4;++u){
                int nx = cx+dx[u];
                int ny = cy + dy[u];
                
                if(nx>=0&&nx<m&&ny>=0&&ny<n){
                    if(g[nx][ny]==1){
                        
                        if(cz+1>=k+1) continue;
                        
                        if(dp[cx][cy][cz]+1<dp[nx][ny][cz+1]){
                            dp[nx][ny][cz+1] = dp[cx][cy][cz]+1;
                            qx[e] = nx;
                            qy[e] = ny;
                            qz[e] = cz+1;e++;
                        }
                        
                        
                    }else{
                        
                        if(dp[cx][cy][cz]+1<dp[nx][ny][cz]){
                            dp[nx][ny][cz] = dp[cx][cy][cz]+1;
                            qx[e] = nx;
                            qy[e] = ny;
                            qz[e] = cz;e++;
                        }
                        
                    }
                    
                }
            }
            
        }
        
        int ma = 100000000;
        for(int i=0;i<=k;++i){
            ma=  Math.min(ma,dp[m-1][n-1][i]);
            }
        
        return ma == 100000000?-1:ma;
        
        
        
        
        
        
    }
}
