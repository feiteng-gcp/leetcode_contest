class Solution {
    public int shortestPath(int[][] grid, int k) {
        int r = grid.length;
        int c = grid[0].length;
        if (r == 1 && c == 1) {
            return grid[0][0] == 0 ? 0 : (k > 0 ? 0 : -1);
        }
        
        int[][] dirs = new int[][]{{0,1},{0,-1},{1,0},{-1,0}};
        
        int[][][] distances = new int[r][c][k+1];
        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{0,0,k});
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                Arrays.fill(distances[i][j], Integer.MAX_VALUE);
            }
        }
        distances[0][0][k] = 0;
        int steps = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            steps++;
            
            for (int n = 0; n < size; n++) {
                int[] cur = queue.poll();
                int curK = cur[2];
                for (int i = 0; i < 4; i++) {
                    int x = cur[0] + dirs[i][0];
                    int y = cur[1] + dirs[i][1];
                    if (x == r - 1 && y == c - 1) {
                        if (grid[x][y] == 0) {
                            return steps;
                        } else {
                            if (curK > 0) {
                                return steps;
                            }
                        }
                    }

                    if (x >= 0 && x < r && y >= 0 && y < c) {
                        if (grid[x][y] == 0) {
                            if (steps + 1 < distances[x][y][curK]) {
                                distances[x][y][curK] = steps + 1;
                                queue.offer(new int[]{x,y,curK});
                            }
                        } else {
                            if (curK-1 >= 0 && steps + 1 < distances[x][y][curK-1]) {
                                distances[x][y][curK-1] = steps + 1;
                                queue.offer(new int[]{x,y,curK-1});
                            }
                        }
                    }
                }
            }
        }
        return -1;
    }
}
