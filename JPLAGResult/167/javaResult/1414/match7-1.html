class Solution {
    public int shortestPath(int[][] grid, int k) {
        if(k>=(grid.length + grid[0].length-2))return (grid.length + grid[0].length-2);
        Queue<int[]> q = new LinkedList<>();
        q.add(new int[]{0,0,0,k});
        Set<String> seen = new HashSet<>();
        int[][] dire = new int[][]{{0,1},{0,-1},{1,0},{-1,0}}; 
        int res = Integer.MAX_VALUE;
        while(!q.isEmpty()){
            int[] cur = q.poll();
            for(int[] di: dire){
                int nextx = cur[0]+ di[0], nexty = cur[1] + di[1], steps = cur[2]+1, skip = cur[3];
                if(nextx >= 0 && nextx < grid.length && nexty >= 0 && nexty < grid[0].length ){
                    if(grid[nextx][nexty] == 1) skip = skip -1;
                    String mark = nextx+""+nexty+skip;
                    // System.out.println(skip);
                    if(seen.contains(mark) || skip <0 ) continue;
                    seen.add(mark);
                    if(nextx == grid.length-1 && nexty == grid[0].length-1)  {
                        res = Math.min(res, steps);
                        continue;
                    }
                    q.add(new int[]{nextx, nexty, steps, skip});
                    
                }
            }
        }
        return res == Integer.MAX_VALUE ? -1: res;
        
    }
}
