class Solution {
    static int[][] directions = new int[][] {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public int shortestPath(int[][] grid, int k) {
        if (k == 0 && grid[0][0] == 1) {
            return -1;
        }
        int m = grid.length;
        int n = grid[0].length;
        Queue<int[]> queue = new LinkedList<>();
        //i, j, step, k
        queue.offer(new int[]{0, 0, 0, k - grid[0][0]});
        int[][] visited = new int[m][n];
        for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                visited[i][j] = -1;
            }
        }
        visited[0][0] = k - grid[0][0];
        while(!queue.isEmpty()) {
            int size = queue.size();
            for(int i = 0; i < size; ++i) {
                int[] item = queue.poll();
                if (item[0] == m - 1 && item[1] == n - 1) {
                    return item[2];
                }
                for(int[] direction: directions) {
                    int nx = item[0] + direction[0];
                    int ny = item[1] + direction[1];
                    if (nx < 0 || ny < 0 || nx >= m || ny >= n) {
                        continue;
                    }
                    if (item[3] - grid[nx][ny] <= visited[nx][ny]) {
                        continue;
                    }
                    queue.offer(new int[]{nx, ny, item[2] + 1, item[3] - grid[nx][ny]});
                    visited[nx][ny] = item[3] - grid[nx][ny];
                }
            }
        }
        return -1;
    }
}
