class Solution {
    
    private static final int[][] directions = new int[][]{{1, 0}, {-1, 0}, {0, 1}, 
{0, -1}};
    
    public int shortestPath(int[][] grid, int K) {
        int n = grid.length, m = grid[0].length;
        int[][] availableK = new int[n][m];
        for (int[] row : availableK) {
            Arrays.fill(row, -1);
        }
        
Queue<int[]> queue = new ArrayDeque<>();
        queue.add(new int[]{0, 0, K});
        availableK[0][0] = K;
        int distance = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int[] pos = queue.remove();
                if (pos[0] == n - 1 && pos[1] == m - 1) {
                    return distance;
                
}
                for (int[] dir : directions) {
                    int x = dir[0] + pos[0], y = dir[1] + pos[1];
                    if (x >= 0 && x < n && y >= 0 && y < m) {
                        if (grid[x][y] == 0) {
                            if (pos[2] > availableK[x][y]) {
                                availableK[x][y] = pos[2];
                                queue.add(new int[]{x, y, pos[2]});
                            }
                        } else {
                            if (pos[2] - 1 > availableK[x][y]) {
                                availableK[x][y] = pos[2] - 1;
                                queue.add(new int[]{x, y, pos[2] - 1});
                            }
                        
}
                    }
                }
            }
            distance++;
        }
        return -1;
    }
     
}
