class Solution {
    private final int[][] dirs = new int[][]{{0,1},{1,0},{0,-1},{-1,0}};
    public int shortestPath(int[][] grid, int k) {
        int m = grid.length, n = grid[0].length;
        Queue<int[]> q = new LinkedList<>();
        Map<Integer, Integer> map = new HashMap<>();
        q.offer(new int[]{0, 0});
        map.put(0, grid[0][0]);
        int res = 0;
        while(!q.isEmpty()){
            for(int i=q.size(); i>0; i--){
                int[] poll = q.poll();
                int x = poll[0], y = poll[1];
                if(x==m-1 && y==n-1) return res;
                int obt = map.get(x*n+y);
                for(int[] dir: dirs){
                    int nx = x+dir[0], ny = y+dir[1];
                    if(nx<0 || ny<0 || nx>=m || ny>=n) continue;
                    int last = map.getOrDefault(nx*n+ny, Integer.MAX_VALUE);
                    if(last>obt+grid[nx][ny] && obt+grid[nx][ny]<=k){
                        map.put(nx*n+ny, obt+grid[nx][ny]);
                        q.offer(new int[]{nx, ny});
                    }
                }
            }
            res++;
        }
        return -1;
    }
}
