class Solution {

	public int shortestPath(int[][] grid, int k) {
		ArrayDeque<int[]> queue = new ArrayDeque<>();
		HashSet<String> set = new HashSet<>();
		queue.add(new int[] { 0, 0, 0 });
		for (int i = 0
; !queue.isEmpty(); i++) {
			for (int j = queue.size(); j > 0; j--) {
				int[] head = queue.remove();
				if (head[0] >= 0 && head[0] < grid.length && head[1] >= 0 && head[1] < grid[0].length) {
					if (grid[head[0]][head[1]] == 1) {
						head[2]++;
					}
					if (head[2] <= k) {
						if (head[0] == grid.length - 1 && head[1] == grid[0].length - 1) {
							return i;
						} else if (!set.contains(Arrays.toString(head))) {
							set.add(Arrays.toString(head));
							queue.add(new int[] { head[0] - 1, head[1], head[2] });
							queue.add(new int[] { head[0], head[1] - 1, head[2] });
							queue.add(new int[] { head[0], head[1] + 1, head[2] });
							queue.add(new int[] { head[0] + 1, head[1], head[2] });
						}
					}
				}
			}
		}
		return -1;
	}
}
