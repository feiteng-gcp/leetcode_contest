class Solution {
    
    int[][] dir = new int[][]{{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    Integer[][][] memo;
    boolean[][] visited;
    
    public int shortestPath(int[][] grid, int k) {
        
        int m = grid.length;
        int n = grid[0].length;
        memo = new Integer[m][n][k + 1];
        visited = new boolean[m][n];
        
        int r = dfs(grid, k, 0, 0, visited);
        
        return r == Integer.MAX_VALUE ? -1 : r;
    }
    
    public int dfs(int[][] grid, int k, int i, int j, boolean[][] visited){
        
        
        if(i < 0 || i >= grid.length || j < 0 || j >= grid[0].length) return Integer.MAX_VALUE;
        
        if(i == grid.length - 1 && j == grid[0].length - 1) return 0;

        if(visited[i][j]) return Integer.MAX_VALUE;
        
        if(grid[i][j] == 1 && k == 0) return Integer.MAX_VALUE;
        
        if(memo[i][j][k] != null) return memo[i][j][k];
        
        if(grid[i][j] == 1) k--;
        
        
        visited[i][j] = true;
        
        int res = Integer.MAX_VALUE;
        
        for(int[] d : dir){
            
            int fromBack = dfs(grid, k, i + d[0], j + d[1], visited);
            
            if(fromBack == Integer.MAX_VALUE) continue;
            
            res = Math.min(res , 1 + fromBack);
        }
        
        visited[i][j] = false;

        memo[i][j][k] = res;
        
        
        return res;
    }
}
