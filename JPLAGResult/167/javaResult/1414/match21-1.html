class Solution {
    
public int shortestPath(int[][] grid, int k) {
        if (grid.length == 1 && grid[0].length == 1) {
            return 0;
        }
        int[][][] ss = new int[grid.length][grid[0].length][k + 1];
        Queue<int[]> queue = new LinkedList<>();
        for (int i = 0; i < grid.length; i++) {
            
for (int j = 0; j < grid[0].length; j++) {
                for (int p = k; p >= 0; p--) {
                    ss[i][j][p] = Integer.MAX_VALUE;
                }
            }
        }
        ss[0][0][k] = 0;
        queue.add(new int[]{0, 0, k});
        int res = 1;
        
int[][] dir = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        
while (queue.size() != 0) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int[] cur = queue.poll();
                int curRow = cur[0];
                int curCol = cur[1];
                int curK = cur[2];
                
for (int p = 0; p < dir.length; p++) {
                    int nextRow = curRow + dir[p][0];
                    int nextCol = curCol + dir[p][1];
                    if (nextRow < 0 || nextCol < 0  || nextRow >= grid.length || nextCol >= grid[0].length) {
                        continue;
                    }
                    if (nextRow == grid.length - 1 && nextCol == grid[0].length - 1) {
                        return res;
                    }
                    if (grid[nextRow][nextCol] == 1 && curK == 0) {
                        continue;
                    }
                    if (grid[nextRow][nextCol] == 1) {
                        if (ss[nextRow][nextCol][curK - 1] <= res) {
                            continue;
                        }
                        ss[nextRow][nextCol][curK - 1] = res;
                        queue.offer(new int[]{nextRow, nextCol, curK - 1});
                    } else {
                        if (ss[nextRow][nextCol][curK] <= res) {
                            continue;
                        }
                        ss[nextRow][nextCol][curK] = res;
                        queue.offer(new int[]{nextRow, nextCol, curK});
                    }
                
}
            }
            res++;
        }
        return -1;
    }
}
