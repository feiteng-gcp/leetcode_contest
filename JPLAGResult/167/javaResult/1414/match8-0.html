class Solution {

  private int[][] adj = {{-1, 0}, {0, 1}, {1, 0},{0, -1}};
  public int shortestPath(int[][] matrix, int k) {
    int n = matrix.length;
    int m = matrix[0].length;

    Set < Node > set = new HashSet < > ();
    Queue < Node > queue = new PriorityQueue < > (new Comparator < Node > () {
     public int compare(Node node1, Node node2) {
        return node1.distance - node2.distance;
     }
    });
    Node start = new Node(0, 0, 0, 0);
    set.add(start);
    queue.add(start);

    while (!queue.isEmpty()) {
     Node node = queue.poll();

     if (node.row == n - 1 && node.column == m - 1) return node.distance;

     for (int[] pos: adj) {
        int row = node.row + pos[0];
        int col = node.column + pos[1];
        if (row >= 0 && row < n && col >= 0 && col < m) {

         if (matrix[row][col] == 1 && node.count == k) continue;
         Node next = new Node(row, col, matrix[row][col] == 1 ? node.count + 1 : node.count, node.distance + 1);

         if (set.contains(next)) continue;

         queue.add(next);
         set.add(next);
        }
     }
    }

    return -1;
 }

 class Node {
    int row, column;
    int count;
    int distance;

    Node(int row, int column, int count, int distance) {
     this.row = row;
     this.column = column;
     this.count = count;
     this.distance = distance;
    }

    @Override
    public int hashCode() {
     return Objects.hash(row, column, count);
    }

    @Override
    public boolean equals(Object o) {
     if (!(o instanceof Node)) return false;
     Node node = (Node) o;
     return (row == node.row && column == node.column && count == node.count);
    }
 }

}
