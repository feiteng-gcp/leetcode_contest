class Solution {

  public int shortestPath(int[][] g, int k) {
    Queue<int[]> q = new ArrayDeque<>();
    Set<String> s = new HashSet<>();
    q.add(new int[3]);
    for (int i = 0; !q.isEmpty(); i++) {
      for (int j = q.size(); j > 0; j--) {
        int[] h = q.remove();
        if (h[0] >= 0 && h[0] < g.length && h[1] >= 0 && h[1] < g[0].length) {
          if (g[h[0]][h[1]] == 1) {
            h[2]++;
          }
          if (h[2] <= k) {
            if (h[0] == g.length - 1 && h[1] == g[0].length - 1) {
              return i;
            } else if (!s.contains(Arrays.toString(h))) {
              s.add(Arrays.toString(h));
              q.add(new int[] { h[0] - 1, h[1], h[2] });
              q.add(new int[] { h[0], h[1] - 1, h[2] });
              q.add(new int[] { h[0], h[1] + 1, h[2] });
              q.add(new int[] { h[0] + 1, h[1], h[2] });
            }
          }
        }
      }
    }
    return -1;
  }
}
