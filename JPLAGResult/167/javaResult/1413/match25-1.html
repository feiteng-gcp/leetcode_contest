class Solution {
    
public int maxSideLength(int[][] mat, int threshold) {
        int m = mat.length, n = mat[0].length;
        int[][] sums = new int[m+1][n+1];
        for(int i=0; i<m; i++){
            
for(int j=0; j<n; j++){
                sums[i+1][j+1] = mat[i][j]+sums[i+1][j]+sums[i][j+1]-sums[i][j];
            }
        }
        int res = 0;
        for(int i=0; i<=m; i++){
            for(int j=0; j<=n; j++){
                for(int k=0; k<=Math.min(m, n); k++){
                    if(i+k>m||j+k>n) break;
                    
if(sums[i+k][j+k]+sums[i][j]-sums[i+k][j]-sums[i][j+k]<=threshold){
                        res = Math.max(k, res);
                    }
                }
            }
        }
        return res;
    }
}
