class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        int m = mat.length;
        int n = mat[0].length;
        int[][] sum = new int[m+1][n+1];
        for (int i=0; i<m; i++) {
            for (int j=0; j<n; j++) {
                sum[i+1][j+1] = mat[i][j] + sum[i][j+1] + sum[i+1][j] - sum[i][j];
            }
        }
        
        int maxLen = Math.min(m, n);
        for (int k = maxLen; k >= 1; k--) {
            for (int i=0; i<=m-k; i++) {
                for (int j=0; j<=n-k; j++) {
                    if (sum[i+k][j+k] - sum[i+k][j] - sum[i][j+k] + sum[i][j] <= threshold) {
                        return k;
                    }
                }
            }
        }
        return 0;
    }
}
