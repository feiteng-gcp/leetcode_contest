class Solution {
    public int maxSideLength(int[][] matrix, int threshold) {
        int n = matrix.length;
        int m = matrix[0].length;
        int [][] prefix = new int[n][m];
        
        prefix[0][0] = matrix[0][0];
        for(int i = 1; i < m; i++){
            prefix[0][i] = prefix[0][i - 1] + matrix[0][i];
        }
        for(int i = 1; i < n; i++){
            prefix[i][0] = prefix[i - 1][0] + matrix[i][0];
        }
        
        for(int i = 1; i < n; i++){
            for(int j = 1; j < m; j++){
                prefix[i][j] = prefix[i - 1][j] + prefix[i][j - 1] - prefix[i - 1][j - 1] + matrix[i][j];
            }
        }
        
        int ans = 0;
        
        
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                for(int dis = 0; ; dis++){
                    if(i + dis >= n ||  j + dis >= m){
                        break;
                    }
                    int total = 0;
                    if(i > 0 && j > 0){
                        total += prefix[i - 1][j - 1];
                    }
                    if(i > 0){
                        total -= prefix[i - 1][j + dis];
                    }
                    if(j > 0){
                        total -= prefix[i + dis][j - 1];
                    }
                    total += prefix[i + dis][j + dis];
                    if(total <= threshold){
                        ans = Math.max(ans, dis + 1);
                    }
                }
            }
        }
        return ans;
    }
}
