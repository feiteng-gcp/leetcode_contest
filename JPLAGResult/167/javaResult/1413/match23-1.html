class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        int m = mat.length;
        int n = mat[0].length;
        int[][] dp = new int[m + 1][n + 1];
        for(int i = 0; i < m; ++i) {
            
for(int j = 0; j < n; ++j) {
                dp[i + 1][j + 1] = dp[i + 1][j] + dp[i][j + 1] + mat[i][j] - dp[i][j];  
            }
        }
        int maxL = 0;
        
for(int i = 0; i < m; ++i) {
            for(int j = 0; j < n; ++j) {
                for(int k = 0; i + k < m && j + k < n; ++k) {
                    int cV = dp[i + k + 1][j + k + 1] + dp[i][j] - dp[i][k + j + 1] - dp[i + k + 1][j];
                    if(cV > threshold) continue;
                    else if(k + 1 > maxL) maxL = k + 1
;
                }
            }
        }
        return maxL;
    }
}
