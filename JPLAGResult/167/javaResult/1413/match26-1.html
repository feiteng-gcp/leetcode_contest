class Solution {
    
public int maxSideLength(int[][] mat, int threshold) {
        int r = mat.length, c = mat[0].length;
        int[][] sums = new int[r + 1][c + 1];
        for (int i = 0; i < r; i++) {
            
for (int j = 0; j < c; j++) {
                sums[i + 1][j + 1] = sums[i + 1][j] + sums[i][j + 1] + mat[i][j] - sums[i][j];
            }
        }
        int res = 0;
        for (int i = 1; i <= r; i++) {
            for (int j = 1; j <= c; j++) {
                for (int k = res; k <= Math.min(i, j); k++) {
                    int area = sums[i][j] - sums[i - k][j] - sums[i][j - k] + sums[i - k][j - k];
                    if (area <= threshold) res = k;
                    else break
;
                }
            }
        }
        return res;
    }
}
