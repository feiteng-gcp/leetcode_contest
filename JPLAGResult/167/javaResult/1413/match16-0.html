class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        int n = mat.length, m = mat[0].length;
        int[][] ps = new int[n + 1][m + 1];
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++)
                ps[i + 1][j + 1] = ps[i + 1][j] + ps[i][j + 1] - ps[i][j] + mat[i][j];
        }
        for(int k = Math.min(n, m); k > 0; k--) {
            for(int i = 0; i + k <= n; i++) {
                for(int j = 0; j + k <= m; j++) {
                    int sum = ps[i + k][j + k] - ps[i + k][j] - ps[i][j + k] + ps[i][j];
                    if (sum <= threshold)
                        return k;
                }
            }
        }
        return 0;
    }
}
