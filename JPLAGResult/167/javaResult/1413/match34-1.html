public class Solution {
	public int maxSideLength(int[][] mat, int threshold) {
		int M = mat.length, N = mat[0].length;
		int[][] preSum = new int[M + 1][N + 1];
		for (int i = 0; i < M; ++i) {
			int preRow = 0;
			for (int j = 0; j < N
; ++j) {
				preRow += mat[i][j];
				preSum[i + 1][j + 1] = preSum[i][j + 1] + preRow;
			}
		}

		int maxLen = 0;
		for (int i = 0; i < M; ++i) {
			for (int j = 0; j < N; ++j) {
				while (i >= maxLen && j >= maxLen && preSum[i + 1][j + 1] - preSum[i + 1][j - maxLen] - preSum[i - maxLen][j + 1] + preSum[i - maxLen][j - maxLen] <= threshold) {
					maxLen++;
				}

			}
		}
		return maxLen;
	}
}
