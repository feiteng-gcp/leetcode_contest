class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        int n = mat.length;
        int m = mat[0].length;
        
        int[][] sum = new int[n][m];
        for (int i = 0; i < n; i++) {
            int rowSum = 0;
            for (int j = 0; j < m; j++) {
                rowSum += mat[i][j];
                sum[i][j] = rowSum;
                if (i > 0) sum[i][j] += sum[i - 1][j];
            }
        }
        
        int maxK = Math.min(n, m);
        for (int k = maxK; k > 0; k--) {
            for (int i = n - 1; i >= k - 1; i--) {
                for (int j = m - 1; j >= k - 1; j--) {
                    int val = sum[i][j];
                    if (i == k - 1) {
                        if (j > k - 1) {
                            val -= sum[i][j - k];
                        }
                    } else if (j == k - 1) {
                        val -= sum[i - k][j];
                    } else {
                        val = val - sum[i][j - k] - sum[i - k][j] + sum[i - k][j - k];
                    }
                    if (val <= threshold) return k;
                }
            }
        }
        
        return 0;
    }
}
