class Solution {
    public int maxSideLength(int[][] mat, int t) {
        int n = mat.length;
        int m = mat[0].length;
        
        int ma =0;
        
        for(int i=0;i<n;++i){
            int c[] = new int[m];
            for(int j=i;j<n;++j){
                int sum[] = new int[m+1];
                for(int k=0;k<m;++k){
                    c[k] += mat[j][k];
                    sum[k+1] =sum[k]+c[k];
                }
                int d = j-i+1;
                for(int k=1;k<=m;++k){
                    if(k-d>=0){
                        int ck = sum[k]-sum[k-d];
                        if(ck<=t){
                            
                            ma = Math.max(ma,d);
                            // if(ma==3){
                            //     System.out.println(i+" "+j+" "+k);
                            // }
                        }
                        
                    }
                }
                
            }
        }
        return ma;
        
        
    }
}
