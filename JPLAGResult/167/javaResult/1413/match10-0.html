class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        NumMatrix nm = new NumMatrix(mat);
        int low = 0, high = 300;
        while (low < high) {
            int mid = (low + high + 1) / 2;
            //System.out.println(low + " " + high + " " + mid + " " + canFind(nm, mid, threshold));
            if (canFind(nm, mid, threshold)) {
                low = mid;
            } else {
                high = mid - 1;
            }
        }
        return low;
    }
    private boolean canFind(NumMatrix nm, int len, int threshold) {
        int n = nm.sum.length - 1, m = nm.sum[0].length - 1;
        for (int i = 0; i <= n - len; i++) {
            for (int j = 0; j <= m - len; j++) {
                int row1 = i, col1 = j, row2 = i + len - 1, col2 = j + len - 1;
                int sum = nm.sumRegion(row1, col1, row2, col2);
                //System.out.println(row1 + " " + col1 + " " + row2 + " " + col2 + " " + len + " " + sum);
                if (sum <= threshold) return true;
            }
        }
        return false;
    }
}
class NumMatrix {
    int[][] sum;
    public NumMatrix(int[][] matrix) {
        int n = matrix.length, m = n > 0 ? matrix[0].length : 0;
        if (matrix.length != 0) sum = new int[n + 1][m + 1];
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                sum[i][j] = sum[i - 1][j] + sum[i][j - 1] - sum[i - 1][j - 1] + matrix[i - 1][j - 1];
            }
        }
    }
    
    public int sumRegion(int row1, int col1, int row2, int col2) {
        return sum[row2+1][col2+1] - sum[row2+1][col1] - sum[row1][col2+1] + sum[row1][col1];
    }
}
