class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        int m = mat.length;
        int n = mat[0].length;;
        int[][] dp = new int[m + 1][n + 1];
        for(int i = 1; i <= m; i++) {
            for(int j = 1; j <= n; j++) {
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1] + mat[i - 1][j - 1] - dp[i - 1][j - 1];
            }
        }
        int res = 0;
        for(int i = 1; i <= m; i++) {
            for(int j = 1; j <= n; j++) {
                // System.out.println(dp[i][j]);
                for(int k = 1; k <= Math.min(i, j); k++) {
                    int area = dp[i][j] - dp[i - k][j] - dp[i][j - k] + dp[i - k][j - k];
                    if(area > threshold) break;
                    res = Math.max(res, k);
                }
            }
        }
        return res;
        
    }
}
