class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        int m = mat.length;
        int n = mat[0].length;
        int[][] endSum = new int[m][n];
        endSum[0][0] = mat[0][0];
        for (int i = 1; i < m; i++) {
            endSum[i][0] = endSum[i - 1][0] + mat[i][0];
        }
        for (int j = 1; j < n; j++) {
            endSum[0][j] = endSum[0][j - 1] + mat[0][j];
        }
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                endSum[i][j] = endSum[i - 1][j] + endSum[i][j - 1] - endSum[i - 1][j - 1] + mat[i][j];
            }
        }
        
        int res = 0;
        int maxLen = Math.min(m, n);
        
for (int len = 1; len <= maxLen; len++) {
            for (int i = len - 1; i < m; i++) {
                for (int j = len - 1; j < n; j++) {
                    
int curValue = endSum[i][j];
                    if (i - len >= 0) {
                        curValue -= endSum[i - len][j];
                    }
                    if (j - len >= 0) {
                        curValue -= endSum[i][j - len];
                    }
                    if (i - len >= 0 && j - len >= 0) {
                        curValue += endSum[i - len][j - len];
                    }
                    // System.out.println(curValue + " " + len);
                    if (curValue <= threshold) {
                        res = len;
                    
}
                }
            }
        }
        return res;
    }
}
