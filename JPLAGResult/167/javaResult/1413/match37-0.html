class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        int m = mat.length;
        int n = mat[0].length;
        int[][] dp = new int[m + 1][n + 1]; 
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = dp[i][j - 1] + mat[i - 1][j - 1]; 
            }
        }
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] += dp[i - 1][j]; 
            }
        }
        // for (int i = 0; i <= m; i++) {
        //     System.out.println(Arrays.toString(dp[i]));
        // }
        int ret = 0; 
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                for (int len = 1; len <= Math.min(m, n); len++) {
                    if (j + 1 < len || i + 1 < len) {
                        continue;
                    }
                    int val = dp[i + 1][j + 1] - dp[i + 1][j + 1 - len] - dp[i + 1 - len][j + 1] + dp[i  + 1- len][j + 1- len];
                    if (val <= threshold) {
                        ret = Math.max(ret, len);
                    }
                }
            }   
        }
        return ret;
    }
}
