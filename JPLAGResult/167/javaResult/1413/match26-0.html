class Solution {
private static final int INF = 1_000_000_001;

    public int maxSideLength(int[][] mat, int threshold) {
        int m = mat.length;
        int n = mat[0].length;

        int[][] sum = new int[m + 1][n + 1];

        for (int i = 1; i <= m; i++) {
            int leftSum = 0;
            for (int j = 1; j <= n; j++) {
                sum[i][j] = leftSum + sum[i - 1][j] + mat[i - 1][j - 1];
                leftSum += mat[i - 1][j - 1];
            }
        }

        int ans = 0;
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                for (int k = Math.max(ans, 1); k <= i && k <= j; k++) {


                    int t = sum[i][j] - sum[i - k][j] - sum[i][j - k] + sum[i - k][j - k];
                    if (t <= threshold) {
                        ans = Math.max(ans, k);
                    }
                }
            }
        }
        return ans;
    }
}
