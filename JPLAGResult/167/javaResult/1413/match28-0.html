class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        int row = mat.length;
        int col = mat[0].length;
        int maxLength = Math.min(row, col);
        
        int[][] sum = new int[row + 1][col + 1];
        
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                sum[i + 1][j + 1] = mat[i][j] + sum[i][j + 1] + sum[i + 1][j] - sum[i][j];
            }
        }

        int ans = 0;
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                for (int k = ans + 1; k <= maxLength && i - k + 1>= 0 && j - k + 1 >= 0; k++) {
                    int s = sum[i + 1][j + 1] - sum[i - k + 1][j + 1] - sum[i + 1][j - k + 1] + sum[i - k + 1][j - k + 1];
                    if (s <= threshold) {
                        ans = Math.max(k, ans);
                    }
                }
            }
        }
        return ans;
    }

}
