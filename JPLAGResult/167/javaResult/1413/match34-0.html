class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        final int N = mat.length;
        final int M = mat[0].length;
        long[][] S = new long[N + 1][M + 1];
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < M; ++j) {
                S[i + 1][j + 1] = S[i + 1][j] + S[i][j + 1] - S[i][j] + mat[i][j];
            }
        }
        int best = 0;
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < M; ++j) {
                while (i + best + 1 <= N && j + best + 1 <= M) {
                    int next = best + 1;
                    long sum = (S[i + next][j + next] - S[i + next][j] - S[i][j + next] + S[i][j]);
                    if (sum <= threshold) {
                        best = next;
                    } else {
                        break;
                    }
                }
            }
        }
        return best;
    }
}
