class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        int left = 0, right = Math.min(mat.length, mat[0].length);
        while(left < right){
            int mid = left + (right - left +1)/2;
            if(isThere(mat, threshold, mid)) left = mid;
            else right = mid-1;
        }
        
        return left;
        
    }
    boolean isThere(int[][] mat, int thre, int len){
        int cur = 0, first = 0;
        for(int i = 0; i <= mat.length - len; i++){
            if(first != 0){
                for(int k = 0; k < len; k++) {
                    first -= mat[i-1][k];
                    first += mat[i+len-1][k];
                }
                }else{
                first = 0;
                    for(int ii = 0; ii < len; ii++)
                        for(int jj = 0; jj < len; jj++)
                            first += mat[ii][jj];
                
            }
            cur = first;
            if(cur  <= thre) return true;
            
            
            for(int j = 1; j <= mat[i].length - len; j++){
                for(int k = 0;k < len; k++){
                    cur -= mat[i+k][j-1];
                    cur += mat[i+k][j+len-1];
                }
                if(cur <= thre) return true;
            }
        }
        return false;
    }
}
