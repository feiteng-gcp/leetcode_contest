class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        int m = mat.length, n = mat[0].length;
        int[][] prefix = new int[m+1][n+1];
          
        for (int i = 1; i <= m; i++) {
            int sum = 0;
            for (int j = 1; j <= n; j++) {
                sum += mat[i-1][j-1];
                prefix[i][j] = sum + prefix[i-1][j]; 
            }    
        }
        
        int d = Math.min(m, n);
        
        for (int sz = d; sz >= 1; sz--) {
            for (int i = m; i >= sz; i--){
                for (int j = n; j >= sz; j--){
                    int sum = prefix[i][j] - prefix[i][j-sz]-prefix[i-sz][j]+prefix[i-sz][j-sz];
                    if (sum <= threshold) return sz;
                }
            }
        }
        return 0;
    }
}
