class Solution {
    public int maxSideLength(int[][] mat, int threshold) {
        
        int m = mat.length;
        int n = mat[0].length;
        
        int[][] sum = new int[m + 1][n + 1];
        
        for(int i = 1; i <= m; i++){
            for(int j = 1; j <= n; j++){
                sum[i][j] = sum[i - 1][j] + sum[i][j - 1] - sum[i - 1][j - 1] + mat[i - 1][j - 1];
            }
        }
        
        int len = Math.min(m, n);
        int res = 0;
        
        for(int l = 1; l <= len; l++){
            for(int i = l; i <= m; i++){
                for(int j = l; j <= n; j++){
                    int s = sum[i][j] - sum[i - l][j] - sum[i][j - l] + sum[i - l][j - l];
                    if(s <= threshold){
                        res = Math.max(l, res);
                    }
                }
            }    
        }
        
        return res;
    }
}
