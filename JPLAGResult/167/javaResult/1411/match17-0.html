/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int getDecimalValue(ListNode head) {
		int rslt = 0;
		while (head != null) {
			rslt = (rslt << 1) | head.val;
			head = head.next;
		}
		return rslt;
    }
}
