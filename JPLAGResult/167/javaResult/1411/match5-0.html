/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int getDecimalValue(ListNode head) {
        int n = 0;
        while (head != null) {
            n <<= 1;
            n |= head.val;
            head = head.next;
        }
        return n;
    }
}
