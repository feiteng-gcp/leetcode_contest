/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */

class Solution {
    public int getDecimalValue(ListNode head) {
        long res = 0;
        while(head != null) {
            res |= head.val;
            res <<= 1;
            head = head.next;
        }
        res >>= 1;
        return (int)res;
    }
}
