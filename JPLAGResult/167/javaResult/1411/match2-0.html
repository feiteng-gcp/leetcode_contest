/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int getDecimalValue(ListNode head) {
        int val = head.val;
        while (head.next != null) {
            val *= 2;
            val += head.next.val;
            head = head.next;
        }
        
        return val;
    }
}
