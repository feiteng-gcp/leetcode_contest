/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int getDecimalValue(ListNode head) {
        int curr = 0;
        while (head != null) {
            curr = (curr<<1) + head.val;
            head=head.next;
        }
        return curr;
    }
}
