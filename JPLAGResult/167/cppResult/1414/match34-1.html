class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        int nr = grid.size();
        int nc = grid[0].size();
        queue<vector<int>> que;
        que.push(vector<int>{0, 0, 0});
        
        bool visited[40][40][40 * 40] = {0};
        const vector<vector<int>> DIRS{{0, -1}, {0, 1}, {-1, 0}, {1, 0}};
        int ret = 0;
        while(!que.empty()) {
            for(int i = 0, n = que.size(); i < n; i++) {

                vector<int> cur = que.front();
                que.pop();
                if(cur[0] == nr - 1 && cur[1] == nc - 1) return ret;
                for(int j = 0; j < 4; j++) {
                    int ni = cur[0] + DIRS[j][0];
                    int nj = cur[1] + DIRS[j][1];
                    
                    if(ni < 0 || ni == nr || nj < 0 || nj == nc) continue;
                    
                    if(grid[ni][nj]) {
                        if(cur[2] + 1 > k) continue;
                        if(visited[ni][nj][cur[2] + 1]) continue;
                        visited[ni][nj][cur[2] + 1] = true;
                        que.push(vector<int>{ni, nj, cur[2] + 1});
                    } else {
                        if(visited[ni][nj][cur[2]]) continue;
                        visited[ni][nj][cur[2]] = true;
                        que.push(vector<int>{ni, nj, cur[2]});
                    }
                }
            }
            ++ret;
        }
        
        return -1;
    }
};
