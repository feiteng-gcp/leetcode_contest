class Solution {
public:
    vector<vector<bool>> come;
    int n,m, ans=0x7fffffff;
    int dir[4][2]={{0,1}, {1,0}, {0,-1}, {-1, 0}};
    
    void dfs(int xx, int yy, int now, int k, vector<vector<int>>& grid) {
        if (k<0)
            return;
        
        // printf("%d %d %d %d \n", xx, yy, now, k);
        
        
        if (xx==n-1 && yy==m-1) {
            ans=min(ans, now);
            return;
        }
        
        
        if (n-1 -xx + m-1-yy +now >=ans)
            return;
        
        come[xx][yy]=true;
        
        for (int i=0;i<4;i++) {
            int x=xx+dir[i][0];
            int y=yy+dir[i][1];
            
            if (x>=0 && y>=0 && x<n && y<m && !come[x][y]) {
                dfs(x,y,now+1, k-grid[x][y], grid);
            }
        }
        come[xx][yy]=false;
    }
    
    int shortestPath(vector<vector<int>>& grid, int k) {
        n=grid.size();
        m=grid[0].size();
        
        come.clear();
        come.resize(n, vector<bool>(m, false));
        
        dfs(0,0, 0, k, grid);
        
        return (ans==0x7fffffff)?-1:ans;
    }
};
