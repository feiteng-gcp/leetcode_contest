int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

class Solution {
public:
    int shortestPath(vector<vector<int>>& g, int K) {
        int n = g.size();
        int m = g[0].size();
        queue<tuple<int,int,int>> q;
        const int INF = 1000000000;
        vector<vector<vector<int>>> f(n, vector<vector<int>>(m, vector<int>(K+1, INF)));
        q.push({0, 0, 0});
        f[0][0][0] = 0;
        while (!q.empty())
        {
            auto [x, y, z] = q.front();
            q.pop();
            for (int k = 0; k < 4; ++ k)
            {
                int tx = x+dx[k], ty = y+dy[k];
                if (0 <= tx && tx < n && 0 <= ty && ty < m)
                {
                    int tz = z+g[tx][ty];
                    if (tz <= K && f[tx][ty][tz] > f[x][y][z]+1)
                    {
                        q.push({tx, ty, tz});
                        f[tx][ty][tz] = f[x][y][z]+1;
                    }
                }
            }
        }
        int ret = *min_element(f[n-1][m-1].begin(), f[n-1][m-1].end());
        if (ret == INF) ret = -1;
        return ret;
    }
};
