class Solution {
public:
    typedef pair<int,int> cell;
    typedef pair<int,int> index;
    int x[4] = {0,0,1,-1};
    int y[4] = {1,-1,0,0};
    int shortestPath(vector<vector<int>>& grid, int k) {
        //use of queue
        int m = grid.size(), n = grid[0].size(), ans = INT_MAX;
        vector<vector<cell>> score(m, vector<cell>(n, cell(INT_MAX, INT_MAX)));
        score[0][0] = cell(0,0); //0 move and 0 obstacles cleared
        queue<index> que; que.push(index(0,0)); //here is the index
        while(!que.empty()){
            
            index aIndex = que.front(); que.pop();
            int i = aIndex.first, j = aIndex.second;
            //score[i][j].second = score[i][j].second == INT_MAX ? INT_MAX :  score[i][j].second+1;
            if(i == m-1 && j == n -1 && score[i][j].first <= k){
                ans = min(ans, score[i][j].second);
            }
            for(int r =0; r<4; r++){
                int newI = i + x[r],  newJ = j + y[r];
                if(newI >=0 && newI < m && newJ >=0 && newJ < n){
                    
                    if(score[newI][newJ].first - grid[newI][newJ]>  score[i][j].first ||  score[newI][newJ].second > 1+ score[i][j].second){
                        
                        int left = score[i][j].first+grid[newI][newJ];
                        if(left > k) continue;
                        score[newI][newJ] =  cell(left, score[i][j].second+1);
                        que.push(index(newI, newJ)); 
                    }
                        
                }
            }
        }
        return ans == INT_MAX? -1: ans;

    }
};
