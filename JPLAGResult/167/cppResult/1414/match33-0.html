class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        int n = grid.size();
        int m = grid[0].size();
        int t[n][m][k + 1];
        for (int i = 0; i < n; i++)
            for (int j = 0; j < m; j++)
                for (int q = 0; q <= k; q++)
                    t[i][j][q] = -1;
        t[0][0][0] = 0;
        int size = 1;
        int cap = n * m * (k + 1);
        int qi[cap], qj[cap], qk[cap];
        qi[0] = 0; qj[0] = 0; qk[0] = 0;
        int di[4] = {0, 0, 1, -1};
        int dj[4] = {1, -1, 0, 0};
        for (int w = 0; w < size; w++)
        {
            int i = qi[w];
            int j = qj[w];
            int p = qk[w];
            if (i == n - 1 && j == m - 1) return t[i][j][p];
            for (int r = 0; r < 4; r++)
            {
                int ii = i + di[r];
                int jj = j + dj[r];
                if (ii >= 0 && ii < n && jj >= 0 && jj < m)
                {
                    if (grid[ii][jj] == 1 && p + 1 <= k && t[ii][jj][p + 1] == -1)
                    {
                        t[ii][jj][p + 1] = t[i][j][p] + 1;
                        qi[size] = ii;
                        qj[size] = jj;
                        qk[size] = p + 1;
                        size++;
                    }
                    if (grid[ii][jj] == 0 && t[ii][jj][p] == -1)
                    {
                        t[ii][jj][p] = t[i][j][p] + 1;
                        qi[size] = ii;
                        qj[size] = jj;
                        qk[size] = p;
                        size++;
                    }
                }
            }
        }
        return -1;
    }
};
