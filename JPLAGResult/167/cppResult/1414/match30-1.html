bool vis[51][51][2501];
int dx[4] = {1,-1,0,0};

int dy[4] = {0,0,1,-1};
class Solution {
public:
    int shortestPath(vector<vector<int>>& g, int k) {
        int r = g.size(); int c = g[0].size();
        
        priority_queue<pair<int,int> > q;
        q.push(make_pair(0,0));
        
        memset(vis,0,sizeof vis);
        
        while(!q.empty()){
            int x = q.top().second/125000;
            int y = (q.top().second/2500)%50;
            int os = q.top().second%2500;
            int st = -q.top().first;
            q.pop();
            if(os + g[x][y]>k) continue;
            if(vis[x][y][os]) continue;
            

            vis[x][y][os] = true;
            //cout << x << " " << y << " " << os << " " << st << endl;
            if(x==r-1 && y == c-1) return st;
            for(int d = 0; d<4; d++){
                int nx = x + dx[d];
                int ny = y + dy[d];
                if(nx < 0 || ny < 0) continue;
                if(nx >= r || ny >= c) continue;
                q.push(make_pair( - st - 1 , nx*125000+ny*2500 + os + g[x][y]) );
            }
        }
        return -1;
    }
};
