class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        int M = grid.size(), N = grid[0].size();
        vector<vector<vector<int>>> steps(M, vector<vector<int>>(N, vector<int>(k + 1, INT_MAX)));

        // pair<x, y>, pair<remain_k, step>
        queue<pair<pair<int, int>, pair<int, int>>> Q;
        Q.emplace(make_pair(0, 0), make_pair(k, 0));

        while(!Q.empty()) {
            pair<pair<int, int>, pair<int, int>> cur = Q.front();
            Q.pop();
            int x = cur.first.first, y = cur.first.second, curK = cur.second.first, step = cur.second.second;

            if(steps[x][y][curK] <= step) {
                continue;
            }

            steps[x][y][curK] = step;


            if(x > 0) {
                if(grid[x - 1][y] == 0 && steps[x - 1][y][curK] > step + 1) {
                    Q.emplace(make_pair(x - 1, y), make_pair(curK, step + 1));
                }

                if(grid[x - 1][y] == 1 && curK > 0 && steps[x - 1][y][curK - 1] > step + 1) {
                    Q.emplace(make_pair(x - 1, y), make_pair(curK - 1, step + 1));
                }
            }

            if(x < M - 1) {
                if(grid[x + 1][y] == 0 && steps[x + 1][y][curK] > step + 1) {
                    Q.emplace(make_pair(x + 1, y), make_pair(curK, step + 1));
                }

                if(grid[x + 1][y] == 1 && curK > 0 && steps[x + 1][y][curK - 1] > step + 1) {
                    Q.emplace(make_pair(x + 1, y), make_pair(curK - 1, step + 1));
                }
            }

            if(y > 0) {
                if(grid[x][y - 1] == 0 && steps[x][y - 1][curK] > step + 1) {
                    Q.emplace(make_pair(x, y - 1), make_pair(curK, step + 1));
                }

                if(grid[x][y - 1] == 1 && curK > 0 && steps[x][y - 1][curK - 1] > step + 1) {
                    Q.emplace(make_pair(x, y - 1), make_pair(curK - 1, step + 1));
                }
            }

            if(y < N - 1) {
                if(grid[x][y + 1] == 0 && steps[x][y + 1][curK] > step + 1) {
                    Q.emplace(make_pair(x, y + 1), make_pair(curK, step + 1));
                }

                if(grid[x][y + 1] == 1 && curK > 0 && steps[x][y + 1][curK - 1] > step + 1) {
                    Q.emplace(make_pair(x, y + 1), make_pair(curK - 1, step + 1));
                }
            }
        }

        int res = INT_MAX;

        for(int i = 0; i <= k; ++i) {
            res = min(res, steps[M - 1][N - 1][i]);
        }

        return (res == INT_MAX) ? -1 : res;
    }
};
