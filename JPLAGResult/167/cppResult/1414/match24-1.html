class Solution {
public:
    
    struct Node {
        int x , y , k , d;
    } ;

    int dx[5] = {0 , 0 , 1 , -1} ;
    int dy[5] = {1 , -1 , 0 , 0} ;
    
    int shortestPath(vector<vector<int>>& grid, int k) {
        int m = grid.size() , n = grid[0].size() ;
        vector<vector<vector<bool>>> vis(m + 1 , vector<vector<bool>> (n + 1 , vector<bool> (k + 1 , false))) ;
        queue <Node> qs ;
        Node state ; 
        state.x = 0 ; state.y = 0 ; state.k = grid[0][0] ; state.d = 0 ;
        //vis[0][0][grid[0][0]] = true ;
        qs.push(state) ;
        while (!qs.empty()) {
            state = qs.front() ; qs.pop() ;
            //vis[state.x][state.y][state.k] = true ;
            //cout << state.x << " " << state.y << " " << state.k << " " << state.d << endl ;
            if (state.x == m - 1 && state.y == n - 1) {
                return state.d ; 

            }
            for (int d = 0 ; d < 4 ; ++d) {
                int nx = state.x + dx[d] , ny = state.y + dy[d] ;
                if (nx >= 0 && nx < m && ny >= 0 && ny < n) {
                        if (grid[nx][ny] == 0){
                            Node newstate ;
                            newstate.d = state.d + 1 ;
                            if (vis[nx][ny][state.k] == false) {
                                vis[nx][ny][state.k] = true ;
                                newstate.x = nx ; newstate.y = ny ; newstate.k = state.k ;
                                qs.push(newstate) ;
                            }
                        }else {
                            if (state.k + 1 > k)    continue ;
                            Node newstate ;
                            newstate.d = state.d + 1 ;
                            if (vis[nx][ny][state.k + 1] == false) {
                                vis[nx][ny][state.k + 1] = true ;
                                newstate.x = nx ; newstate.y = ny ; newstate.k = state.k + 1 ;
                                qs.push(newstate) ;
                            }
                        }
                    }
            }
        }
        return -1 ;
    }
};
