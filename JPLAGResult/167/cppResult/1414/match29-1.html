class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int K) {

        int n = grid.size();
        int m = grid[0].size();
        const int INF = 1 << 20;   
        vector<vector<vector<int>>> dis(n, vector<vector<int>>(m, vector<int>(K + 1, INF)));
        
        dis[0][0][0] = 0;
        
        queue<pair<pair<int,int>, int>> que;
        que.emplace(make_pair(0, 0), 0);
        
        vector<pair<int,int>> adjs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        
        while (!que.empty()) {
            int x = que.front().first.first;
            int y = que.front().first.second;
            int k = que.front().second;
            que.pop();
            
            int nd = dis[x][y][k] + 1;
            
            for (auto& p : adjs) {
                int nx = x + p.first;
                int ny = y + p.second;
                
                if (nx >= 0 && nx < n && ny >= 0 && ny < m) {
                    int nk = k + grid[nx][ny];

                    if (nk <= K && dis[nx][ny][nk] > nd) {
                        dis[nx][ny][nk] = nd;
                        que.emplace(make_pair(nx, ny), nk);
                    }
                }
            }
        }
        
        int res = INF;
        for (int i = 0; i <= K; ++i) {
            res = min(res, dis[n - 1][m - 1][i]);
        }
        
        return res < INF ? res : -1;
    }
};
