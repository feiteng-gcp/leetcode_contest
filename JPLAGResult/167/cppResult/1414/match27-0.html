class Solution {
public:
    const int inf = 1e9 + 10;
    int vis[41][41][41*41];
    
    int shortestPath(vector<vector<int>>& grid, int k) {
        int n = grid.size();
        int m = grid[0].size();
        vector<int> dx = {-1, 1, 0, 0};
        vector<int> dy = {0, 0, -1, 1};
        queue<pair<pair<int, int>, pair<int, int>>> q;
        memset(vis, -1, sizeof vis);
        q.push(make_pair(make_pair(0, 0), make_pair(k, 0)));
        vis[0][0][k] = 0;
        while (!q.empty()) {
            pair<pair<int, int>, pair<int, int>> cur = q.front();
            q.pop();
            int x = cur.first.first;
            int y = cur.first.second;
            int p = cur.second.first;
            int d = cur.second.second;
            for (int i = 0; i < 4; i++) {
                int xx = x + dx[i];
                int yy = y + dy[i];
                if (xx < 0 or xx >= n or yy < 0 or yy >= m) continue;
                if (grid[xx][yy] == 1 and p > 0 and vis[xx][yy][p - 1] == -1) {
                    q.push(make_pair(make_pair(xx, yy), make_pair(p - 1, d + 1)));
                    vis[xx][yy][p - 1] = d + 1;
                }
                else if (grid[xx][yy] == 0 and vis[xx][yy][p] == -1) {
                    q.push(make_pair(make_pair(xx, yy), make_pair(p, d + 1)));
                    vis[xx][yy][p] = d + 1;
                }
            }
        }
        int result = inf;
        for (int i = 0; i <= k; i++) {
            if (vis[n - 1][m - 1][i] != -1)
                result = min(result, vis[n - 1][m - 1][i]);
        }
        return (result == inf) ? -1 : result;
    }
};
