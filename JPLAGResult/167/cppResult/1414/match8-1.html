class Solution {
public:
    bool vis[41][41][1603];
    struct node{
        int x,y,s,k;
        bool operator <(const node &tmp)const{
            return s>tmp.s;
        }
    };
    int f[4][2]={1,0,-1,0,0,1,0,-1};
    bool judge(int x,int y,int n,int m){
        if(x<0||x>=n||y<0||y>=m)return false;
        return true;
    }
    int shortestPath(vector<vector<int>>& grid, int k) {
        int n=grid.size(),m=grid[0].size();
        priority_queue<node>q;

        q.push({0,0,0,0});
        vis[0][0][0]=1;
        while(!q.empty()){
            node s = q.top(); q.pop();
            if(s.x==n-1&&s.y==m-1)return s.s;
            for(int i=0;i<4;i++){
                int x=f[i][0]+s.x,y=f[i][1]+s.y;
                if(judge(x,y,n,m)){
                    if(grid[x][y]==0){
                        if(vis[x][y][s.k])continue;
                        vis[x][y][s.k]=1;
                        q.push({x,y,s.s+1,s.k});
                    }else{
                        if(s.k+1>k||vis[x][y][s.k+1])continue;
                        vis[x][y][s.k+1]=1;
                        q.push({x,y,s.s+1,s.k+1});
                    }
                }
            }
        }
        return -1;
    }
};
