class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        int m = grid.size();
        int n = grid[0].size();
        vector<vector<vector<int>>> visited(m, vector<vector<int>>(n, vector<int>(k + 1, m * n)));
        queue<vector<int>> bfs;

        bfs.push({0, 0, 0, 0});
        vector<int> dir = {0, 1, 0, -1, 1, 0, -1, 0};
        while (!bfs.empty()) {
            auto cur = bfs.front();
            bfs.pop();
            if (cur[0] == m - 1 && cur[1] == n - 1) return cur[3];
            //visited[cur[0]][cur[1]][cur[2]] = cur[3];
            for (int i = 0; i < 4; i++) {
                int new_x = cur[0] + dir[2 * i];
                int new_y = cur[1] + dir[2 * i + 1];
                if (new_x >= 0 && new_x < m && new_y >= 0 && new_y < n && grid[new_x][new_y] + cur[2] <= k && cur[3] + 1 < visited[new_x][new_y][grid[new_x][new_y] + cur[2]]) {
                    visited[new_x][new_y][grid[new_x][new_y] + cur[2]] = cur[3] + 1;
                    bfs.push({new_x, new_y, cur[2] + grid[new_x][new_y], cur[3] + 1});
                }
            }
        }
        return -1;
    }
};
