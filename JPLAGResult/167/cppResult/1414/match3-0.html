class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int t) {
        int pos[4][2] = {{0,1},{1,0},{-1,0},{0,-1}};
        queue<vector<int>> q;
        set<vector<int>> visited;
        if(grid[0][0]>t)
            return -1;
        q.push({0,0,grid[0][0]});
        visited.insert({0,0,grid[0][0]});
        int level = 0;
        while(!q.empty()) {
            int size = q.size();
            for(int k=0; k<size; k++) {
                auto front = q.front();
                q.pop();
                if(front[0]==grid.size()-1&&front[1]==grid[0].size()-1)
                    return level;
                for(int i=0; i<4; i++) {
                    int x = front[0] + pos[i][0];
                    int y = front[1] + pos[i][1];
                    if(x<0||y<0||x>=grid.size()||y>=grid[0].size())
                        continue;
                    if(front[2]+grid[x][y]>t) {
                        continue;

                    }
                    vector<int> to_push = {x,y,front[2]+grid[x][y]};
                    if(visited.count(to_push))
                        continue;
                    visited.insert(to_push);
                    q.push(to_push);
                }
                
            }
            level++;
        }
        return -1;
        
    }
};
