class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {

        vector<vector<int>> directions = {{0, -1}, {0, 1}, {1, 0}, {-1, 0}};

        int m = grid.size(), n = grid[0].size();
        queue<vector<int>> all_positions;
        int total_step = 0;
        vector<vector<vector<bool>>> visit(m, vector<vector<bool>>(n, vector<bool>(k + 1, false)));
        if (grid[0][0] == 1) {
            all_positions.push({0, 0, 1});
            visit[0][0][1] = true;
        }
        else {
            all_positions.push({0, 0, 0});
            visit[0][0][0] = true;
        }

        while (!all_positions.empty()) {
            int l = all_positions.size();
            for (int i = 0; i < l; i++) {
                vector<int> s = all_positions.front();
                all_positions.pop();
                if (s[2] == k + 1) 
                    continue;

                if (s[0] == m - 1 && s[1] == n - 1) 
                    return total_step;


                for (vector<int>& direction : directions) {
                    int x_prime = s[0] + direction[0];
                    int y_prime = s[1] + direction[1];
                    if (x_prime >= 0 && x_prime < m && y_prime >= 0 && y_prime < n) {
                        if (grid[x_prime][y_prime] == 0 && !visit[x_prime][y_prime][s[2]]) {
                            all_positions.push({x_prime, y_prime, s[2]});
                            visit[x_prime][y_prime][s[2]] = true;
                        }
                        else if (grid[x_prime][y_prime] == 1 && !visit[x_prime][y_prime][s[2] + 1]){
                            all_positions.push({x_prime, y_prime, s[2] + 1});
                            visit[x_prime][y_prime][s[2] + 1] = true;
                        }
                    }
                }
            }
            total_step ++;
        }
        return -1;
    }
};
