class Solution {
public:
    string encode(tuple<int, int, int> t) {
        return to_string(get<0>(t)) + "," + to_string(get<1>(t)) + "," + to_string(get<2>(t));
    }
    vector<vector<int>> directs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    int shortestPath(vector<vector<int>>& grid, int k) {
        queue<tuple<int, int, int>> q;
        unordered_set<string> visited;
        q.push({0, 0, k});
        int step = 0;
        while (q.size() > 0) {
            int level_size = q.size();
            for (int i = 0; i < level_size; ++i) {
                auto curr = q.front();
                q.pop();
                visited.insert(encode(curr));
                if (get<0>(curr) == grid.size() - 1 && get<1>(curr) == grid[0].size() - 1) {
                    return step;
                }
                for (auto direct : directs) {
                    int new_i = direct[0] + get<0>(curr);
                    int new_j = direct[1] + get<1>(curr);
                    int k = get<2>(curr) - grid[get<0>(curr)][get<1>(curr)];
                    string state = encode({new_i, new_j, k});
                    if (new_i >= 0 && new_i < grid.size() && new_j >= 0 && new_j < grid[0].size() && visited.count(state) == 0 && k >= 0) {
                        // cout << state << endl;
                        q.push({new_i, new_j, k});
                        visited.insert(state);
                    }
                
                }
            }
            step++;
        }
        return -1;
    }
};
