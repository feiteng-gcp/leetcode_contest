 struct P{
     int x,y,z;
     P(){}
     P(int x,int y,int z):x(x),y(y),z(z){}
    };

int dir[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
class Solution {
public:
    int shortestPath(vector<vector<int>>& a, int t) {
        int n = a.size();
        int m = a[0].size();
        vector<vector<vector<int>>> f = vector<vector<vector<int>>>(n, vector<vector<int>>(m, vector<int>(t+1, -1)));
        f[0][0][0] = 0;
        queue<P> que;
        que.push(P(0,0,0));
        while(!que.empty()){
            P cur = que.front();
            que.pop();

            for(int i=0;i<4;i++){
                int nx = cur.x + dir[i][0];
                int ny = cur.y + dir[i][1];
                if(nx<0||nx>=n||ny<0||ny>=m)continue;
                int nz = cur.z + a[nx][ny];
                if(nz>t)continue;
                if(f[nx][ny][nz]==-1){
                    f[nx][ny][nz] = f[cur.x][cur.y][cur.z] + 1;
                    que.push(P(nx,ny,nz));
                }
            }
        }
        int ret = -1;
        for(int i=0;i<=t;i++){
            if(f[n-1][m-1][i]==-1)continue;
            if(ret==-1||ret>f[n-1][m-1][i])ret=f[n-1][m-1][i];
        }
        return ret;
    }
};
