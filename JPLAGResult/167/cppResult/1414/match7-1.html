  int dx[] = {0, 0, 1, -1};
  int dy[] = {1, -1, 0, 0};
class Solution {
public:
  int shortestPath(vector<vector<int>>& grid, int k) {
    int m = grid.size();
    int n = grid[0].size();
    vector<vector<vector<int>>> steps(m, vector<vector<int>>(n, vector<int>(k+1, 0x3f3f3f3f)));
    priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>>> Q;
    steps[0][0][0] = 0;
    Q.push({0, 0, 0, 0});
    while (!Q.empty()) {
      auto s = Q.top(); Q.pop();
      int d = s[0], x = s[1], y = s[2], z = s[3];
      if (x == m - 1 && y == n - 1) return d;
      if (d != steps[x][y][z]) continue;

      for (int i = 0; i < 4; i++) {
        int tx = x + dx[i], ty = y + dy[i];
        if (0 <= tx && tx < m && 0 <= ty && ty < n) {
          int tz = z + grid[tx][ty];
          if (tz > k) continue;
          if (steps[tx][ty][tz] > d + 1) {
            steps[tx][ty][tz] = d + 1;
            Q.push({d+1, tx, ty, tz});
          }
        }
      }
    }
    return -1;
  }
};
