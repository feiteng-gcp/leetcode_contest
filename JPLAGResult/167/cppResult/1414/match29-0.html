class Solution {
 public:
  int shortestPath(vector<vector<int>>& grid, int k) {
    const int dx[] = { 0, 1, 0, -1 };
    const int dy[] = { 1, 0, -1, 0 };

    int n = grid.size();
    int m = grid[0].size();

    const int inf = 0x3f3f3f3f;

    vector<vector<vector<int>>> dp(k + 1, vector<vector<int>>(n, vector<int>(m, inf)));
    dp[0][0][0] = 0;
    queue<tuple<int, int, int, int>> bag;
    bag.emplace(0, 0, 0, 0);
    while (bag.size()) {
      int d, t, x, y;
      tie(d, t, x, y) = bag.front();
      bag.pop();
      for (int di = 0; di < 4; ++di) {
        int nx = x + dx[di];
        int ny = y + dy[di];
        if (nx < 0 || nx == n || ny < 0 || ny == m) continue;
        if (grid[nx][ny] == 0) {
          if (d + 1 < dp[t][nx][ny]) {
            dp[t][nx][ny] = d + 1;
            bag.emplace(d + 1, t, nx, ny);
          }
        } else {
          if (t < k && d + 1 < dp[t + 1][nx][ny]) {
            dp[t + 1][nx][ny] = d + 1;
            bag.emplace(d + 1, t + 1, nx, ny);
          }
        }
      }
    }

    int ans = inf;
    for (int i = 0; i <= k; ++i) {
      ans = min(ans, dp[i][n - 1][m - 1]);
    }

    if (ans < inf) return ans;

    return -1;
  }
};
