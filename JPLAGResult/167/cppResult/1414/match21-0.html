class Solution {
public:
    int dirs[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    int shortestPath(vector<vector<int>>& grid, int k) {
        int R = grid.size();
        int C = grid[0].size();
        if (R == 1 && C == 1) return 0;
        queue<pair<int, pair<int, int> > > q;
        vector<vector<vector<int > > > visited(k + 1, vector<vector<int> >(R, vector<int>(C, 0)));
        q.push({0, {0, 0}});
        visited[0][0][0] = 1;
        int step = 0;
        while (!q.empty()) {
            ++step;
            int s = q.size();
            for (int i = 0; i < s; ++i) {
                auto p = q.front();
                q.pop();
                int t = p.first;
                int x = p.second.first;
                int y = p.second.second;
                for (int j = 0; j < 4; ++j) {
                    int r = x + dirs[j][0];
                    int c = y + dirs[j][1];
                    if (r >= 0 && r < R && c >= 0 && c < C) {
                        if (r == R - 1 && c == C - 1) return step;
                        int t1 = t + grid[r][c];
                        if (t + grid[r][c] <= k && !visited[t1][r][c]) {
                            q.push({t1, {r, c}});
                            visited[t1][r][c] = 1;
                        }
                    }
                }
            }
        }
        return -1;
    }
};
