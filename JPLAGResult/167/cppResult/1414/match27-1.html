class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        int m = grid.size();
        int n = grid[0].size();
        int s1 = GetShortestPath(0, 0, grid, k);
        int s2 = GetShortestPath(m - 1, n - 1, grid, k);
        if(s1 == -1) { return s2; }
        if(s2 == -1) { return s1; }
        return min(s1, s2);
        
    }

    int GetShortestPath(int sx, int sy, vector<vector<int>>& grid, int k) {
        int m = grid.size();
        int n = grid[0].size();
        
        int dx[4] = {0,0,1,-1};
        int dy[4] = {1,-1,0,0};
        vector<vector<int>> o(m, vector<int>(n, 10000));
        vector<vector<int>> p(m, vector<int>(n, 10000));
        o[sx][sy] = 0;
        p[sx][sy] = 0;
        
        queue<int> ique;
        ique.push(sx * 100 + sy);
        while(!ique.empty()) {

            int front = ique.front();
            ique.pop();
            int x = front / 100;
            int y = front % 100;
            for(int i = 0; i < 4; ++i) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                if(nx < 0 || nx >= m || ny < 0 || ny >= n) { continue; }
                if(grid[nx][ny] == 0) {
                    if((p[x][y] + 1 < p[nx][ny]) || (p[x][y] + 1 == p[nx][ny] && o[x][y] < o[nx][ny])) {
                        o[nx][ny] = o[x][y];
                        p[nx][ny] = p[x][y] + 1;
                        ique.push(nx * 100 + ny);
                    }
                }
                if(grid[nx][ny] == 1) {
                    if((p[x][y] + 1 < p[nx][ny] && o[x][y] + 1 <= k) || (p[x][y] + 1 == p[nx][ny] && o[x][y] + 1 < o[nx][ny])) {
                        o[nx][ny] = o[x][y] + 1;
                        p[nx][ny] = p[x][y] + 1;
                        ique.push(nx * 100 + ny);
                    }
                }
            }
        }
        // for(int i = 0; i < m; ++i) {
        //     for(int j = 0; j < n; ++j) {
        //         cout << p[i][j] << " ";
        //     }
        //     cout << endl;
        // }
        // cout << endl << "==========" << endl;
        // for(int i = 0; i < m; ++i) {
        //     for(int j = 0; j < n; ++j) {
        //         cout << o[i][j] << " ";
        //     }
        //     cout << endl;
        // }
        if(sx == 0 && sy == 0) {
            return (o[m - 1][n - 1] <= k ? p[m - 1][n - 1] : -1);            
        }
        return (o[0][0] <= k ? p[0][0] : -1);
    }
};
