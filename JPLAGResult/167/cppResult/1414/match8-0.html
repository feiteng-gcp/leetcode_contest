class Solution {
public:
    struct Node {
        int x, y, k, s;
        bool operator < (const Node& rhs) const {
            return k < rhs.k;
        }
    };
    int dx[4] = {-1, 0, 1, 0};
    int dy[4] = {0, -1, 0, 1};
    int shortestPath(vector<vector<int>>& grid, int k) {
        int n = grid.size(), m = grid[0].size();
        int vis[50][50][1600];
        memset(vis, 0, sizeof vis);
        queue<Node> q;
        q.push((Node){0, 0, k, 0});
        vis[0][0][k] = 1;
        while (q.size()) {
            Node cur = q.front(); q.pop();
            if (cur.x == m - 1 && cur.y == n - 1) return cur.s;
            for (int i = 0; i < 4; ++i) {
                int nx = cur.x + dx[i], ny = cur.y + dy[i];
                if (nx < 0 || nx >= m || ny < 0 || ny >= n) continue;
                int g = grid[ny][nx];
                if (cur.k < g) continue;
                if (vis[ny][nx][cur.k-g]) continue;
                vis[ny][nx][cur.k-g] = 1;
                q.push((Node){nx, ny, cur.k - g, cur.s + 1});
            }
        }
        return -1;
    }
};
