const int N = 50;
struct Node {
    int x, y, k;
};
int dp[N][N][N * N];
const int d[4][2] = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};

class Solution {
public:
    int shortestPath(vector<vector<int>>& a, int k) {
        int n = a.size(), m = a[0].size();
        queue<Node> Q;
        memset(dp, 255, sizeof(dp));
        Q.push({0, 0, k});
        dp[0][0][k] = 0;
        while (!Q.empty()) {

            auto [x, y, k] = Q.front();
            //cout << x << " " << y << " " << k << " " << dp[x][y][k] << endl;
            if (x == n - 1 && y == m - 1) return dp[x][y][k];
            Q.pop();
            for (int u = 0; u < 4; ++u) {
                int nx = x + d[u][0];
                int ny = y + d[u][1];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
                int nk = k - a[nx][ny];
                if (nk < 0) continue;
                if (dp[nx][ny][nk] >= 0) continue;
                dp[nx][ny][nk] = dp[x][y][k] + 1;
                //cout << "nxt: " << nx << " " << ny << " " << nk << endl;
                Q.push({nx, ny, nk});
            }
        }
        return -1;
    }
};
