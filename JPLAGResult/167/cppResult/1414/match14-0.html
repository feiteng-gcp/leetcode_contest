class Solution {
public:
    
    bool isSafe(int i, int j, int m, int n) {
        if(i>=0 && i<m && j>=0 && j<n)
            return true;
        return false;
    }
    
    int shortestPath(vector<vector<int>>& grid, int k) {
        int m = grid.size();
        int n = grid[0].size();
        queue<pair<pair<int,int>, pair<int,int > > > q;
        q.push(make_pair(make_pair(0, k), make_pair(0, 0)));
        vector<vector<vector<bool> > > visited(m, vector<vector<bool>>(n, vector<bool>(k+1, false)));
        visited[0][0][k] = true;
        int dx[] = {-1, 0, 1, 0};
        int dy[] = {0, -1, 0, 1};
        while(!q.empty()) {
            int x = q.front().second.first;
            int y = q.front().second.second;
            int c = q.front().first.first;
            int p = q.front().first.second;
            q.pop();
            if(x == m-1 && y == n-1)
                return c;
            int k;
            for(k=0; k<4; k++) {
                int xx = x+dx[k];
                int yy = y+dy[k];
                if(isSafe(xx, yy, m, n)) {
                    if(grid[xx][yy] == 0 && !visited[xx][yy][p]) {
                        visited[xx][yy][p] = true;
                        q.push(make_pair(make_pair(c+1, p), make_pair(xx, yy)));
                    } else if (grid[xx][yy] == 1 && p > 0 && !visited[xx][yy][p-1]) {
                        visited[xx][yy][p-1] = true;
                        q.push(make_pair(make_pair(c+1, p-1), make_pair(xx, yy)));
                    }
                }
            }
        }
        return -1;
    }
};
