class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        int m = grid.size(), n = grid[0].size();
        queue<vector<int>> q;
        q.push({0,0,k,0});
        unordered_set<int> v;
        v.insert(encode(0,0,k));

        vector<pair<int,int>> mv = {{-1,0}, {1,0}, {0,-1}, {0,1}};

        while(q.size()) {
            int size = q.size();
            for(int i=0;i<size;++i) {
                auto t = q.front();
                q.pop();
                int x = t[0], y = t[1], r = t[2], len = t[3];

                if(x==m-1 && y == n-1 && grid[x][y] == 0) {
                    return len;
                }
                for(auto ne:mv) {
                    int nx = ne.first + x;
                    int ny = ne.second + y;
                    if(nx>=0 && ny >=0 && nx <m && ny <n) {
                        if(!grid[nx][ny] || (--r >=0)) {
                           int code = encode(nx, ny, r);
                            if(!v.count(code)) {
                                v.insert(code);
                                q.push({nx,ny,r,len+1});
                                if(grid[nx][ny])
                                    grid[nx][ny] = 0;
                            }
                        } 
                    }
                }
                
            }
        }
        return -1;
    }
    
    int encode(int i, int j, int k) {
        return (i*40 + j)*40 + k;
    }
};
