class Solution {
public:
    vector<vector<int>> dir{{1,0},{-1,0},{0,1},{0,-1}};

    int shortestPath(vector<vector<int>>& grid, int k) {
        int m=grid.size();
        int n=grid[0].size();
        vector<vector<vector<int>>> visited(m,vector<vector<int>>(n,vector<int>(k+1,0)));
        queue<vector<int>> q;
        q.push({0,0,0});
        visited[0][0][0]=1;
        int level=0;
        while(!q.empty()){
            int size=q.size();
            for(int i=0;i<size;++i){

                vector<int> cur=q.front();
                q.pop();
                if(cur[0]==m-1 && cur[1]==n-1) return level;
                for(int d=0;d<4;++d){
                    int newx=cur[0]+dir[d][0];
                    int newy=cur[1]+dir[d][1];
                    int newk=cur[2]+1;
                    if(newx<0 || newx>=m || newy<0 ||newy>=n) continue;
                    if(grid[newx][newy]==1){
                        if(cur[2]==k) continue;
                        if(visited[newx][newy][newk]) continue;
                        visited[newx][newy][newk]=1;
                        q.push({newx,newy,newk});
                    }else{
                        if(!visited[newx][newy][cur[2]]){
                            visited[newx][newy][cur[2]]=1;
                            q.push({newx,newy,cur[2]});
                        }
                    }
                }
            }
            ++level;
        }
        return -1;
    }
};
