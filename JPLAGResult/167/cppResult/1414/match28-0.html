class Solution {
  vector<vector<vector<int>>> dp;
public:
  int shortestPath(vector<vector<int>>& grid, int k) {      
    int n = grid.size(), m = grid[0].size();
    dp.assign(n, vector<vector<int>>(m, vector<int>(k + 1, 1e9)));
    queue<tuple<int, int, int>> q;
    q.emplace(0, 0, grid[0][0]);
    dp[0][0][grid[0][0]] = 0;
    while (q.size()) {
      int i, j, used; tie(i, j, used) = q.front(); q.pop();
      const int dx[4] = {0, 1, 0, -1}, dy[4] = {1, 0, -1, 0};
      for (int d = 0; d < 4; ++d) {
        int ni = i + dx[d], nj = j + dy[d];
        if (ni < 0 or ni >= n or nj < 0 or nj >= m) continue;
        int a = grid[ni][nj];
        if (used + a > k) continue;
        if (dp[ni][nj][used + a] != 1e9) continue;
        q.emplace(ni, nj, used + a);
        dp[ni][nj][used + a] = dp[i][j][used] + 1;
      }
    }
    int ans = 1e9;
    for (int l = 0; l <= k; ++l) {
      if (dp[n - 1][m - 1][l] != 1e9) ans = min(ans, dp[n - 1][m - 1][l]);
    }
    return ans == 1e9 ? -1 : ans;
  }
};
