class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int _k) {
        int n = grid.size();
        int m = grid[0].size();
        
        struct Node{
            int x,y,k;
        };
        unordered_map<int,int> h;
        
        queue<Node> q;
        q.push({0,0,_k});
        
        function<int(int,int,int)> cvt = [&] (int x, int y, int k) {
            return (x*m+y) * 10000 + k;
        };
        
        h[cvt(0,0,_k)] = 0;
        int dx[4] = {0,1,0,-1};
        int dy[4] = {1,0,-1,0};
        
        
        while(q.size()) {
            auto [x,y,k] = q.front();q.pop();
            
            if(x==n-1 && y==m-1) return h[cvt(x,y,k)];
            
            for(int i = 0 ; i < 4 ; i++) {
                int a = x+dx[i];
                int b = y+dy[i];
                if(a<0||b<0||a>=n||b>=m) continue;
                if(!grid[a][b]) {
                    if(!h.count(cvt(a,b,k))) {
                        q.push({a,b,k});
                        h[cvt(a,b,k)] = h[cvt(x,y,k)] + 1;
                    }
                }
                else {
                    if(k>0 && !h.count(cvt(a,b,k-1))) {
                        q.push({a,b,k-1});
                        h[cvt(a,b,k-1)] = h[cvt(x,y,k)] + 1;
                        
                    }
                }
            }
        }
        return -1;
    }
};
