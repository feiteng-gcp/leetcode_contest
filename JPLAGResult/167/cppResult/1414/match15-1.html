    int shortestPath(vector<vector<int>>& grid, int k) {
        int r = grid.size();
        int c = grid[0].size();
        vector<vector<vector<int>>> dp(r, vector<vector<int>> (c, vector<int> (k + 1, r * c)));
        
        int dy[] = {1, 0, -1, 0};
        int dx[] = {0, -1, 0, 1};
        
        //for(int i = 0; i < k; ++i) dp[0][0] = 0;
        dp[0][0][0] = 0;
        queue<int> qy, qx, qrem;
        qy.push(0);
        qx.push(0);
        qrem.push(0);
        

        while(!qy.empty()) {
            int y = qy.front(); qy.pop();
            int x = qx.front(); qx.pop();
            int rem = qrem.front(); qrem.pop();
            
            for(int i = 0; i < 4; ++i) {
                int ny = y + dy[i];
                int nx = x + dx[i];
                
                if(ny >= 0 && ny < r && nx >= 0 && nx < c) {
                    if(grid[ny][nx] == 0 && dp[ny][nx][rem] > dp[y][x][rem] + 1) {
                        dp[ny][nx][rem] = min(dp[ny][nx][rem], dp[y][x][rem] + 1);
                        qy.push(ny);
                        qx.push(nx);
                        qrem.push(rem);
                    } else if(grid[ny][nx] == 1 && rem < k && dp[ny][nx][rem + 1] > dp[y][x][rem] + 1) {
                        dp[ny][nx][rem + 1] = min(dp[ny][nx][rem + 1], dp[y][x][rem] + 1);
                        qy.push(ny);
                        qx.push(nx);
                        qrem.push(rem + 1);
                    }
                }
            }
        }
        /*
        for(int i = 0; i < r; ++i) {
            for(int j = 0; j < c; ++j) {
                cout << dp[i][j][0] << " ";
            }
            cout << endl;
        }
        */
        int ans = r * c;
        for(int x: dp[r - 1][c - 1]) ans = min(ans, x);
        return ans < r * c ? ans : -1;
    }
};
