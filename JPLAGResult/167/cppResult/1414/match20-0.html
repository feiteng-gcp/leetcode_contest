class Solution {
 public:
  int shortestPath(vector<vector<int>>& grid, int k) {
    int m = grid.size(), n = grid[0].size();
    vector<vector<int>> dirs = {{0, 1}, {0, -1}, {-1, 0}, {1, 0}};
    vector<vector<bool>> marked(m * n, vector<bool> (k + 1, false));
    marked[0][0] = true;
    queue<pair<int, int>> Q;
    Q.push({0, 0});
    int steps = 0;
    while (!Q.empty()) {
      int q_size = Q.size();
      for (int i = 0; i < q_size; i++) {
        auto p = Q.front();
        Q.pop();
        int pos = p.first, obs = p.second;
        if (pos / n == m - 1 && pos % n == n - 1) {
          return steps;
        }
        for (auto &dir : dirs) {
          int x = pos / n + dir[0];
          int y = pos % n + dir[1];
          if (x >= 0 && x < m && y >= 0 && y < n && obs + grid[x][y] <= k) {
            int new_pos = x * n + y;
            int new_obs = obs + grid[x][y];
            if (!marked[new_pos][new_obs]) {
              Q.push({new_pos, new_obs});
              marked[new_pos][new_obs] = true;
            }
          }
        }
      }
      steps++;
    }
    return -1;
  }
};
