class Solution {
public:
    int n,m;
    struct Node
    {
        int x,y,k;
        Node(int x=0,int y=0,int k=0)

            :x(x),y(y),k(k){}
    };
    int dx[4]={0,0,-1,1};
    int dy[4]={-1,1,0,0};
    int shortestPath(vector<vector<int>>& grid, int k) {
        n=grid.size();
        m=grid[0].size();
        if(n==1&&m==1)
        {
            return 0;
        }
        vector<vector<vector<int>>> vis(n,vector<vector<int>>(m,vector<int>(k+1,0)));
        vis[0][0][k]=1;
        queue<Node> qu;
        qu.emplace(0,0,k);
        while(!qu.empty())
        {
            Node node=qu.front(); qu.pop();
            int nx=node.x,ny=node.y,nk=node.k;
            //cout<<nx<<" "<<ny<<" "<<nk<<endl;

            for(int i=0;i<4;i++)
            {
                int x=nx+dx[i],y=ny+dy[i];
                if(x<0||n<=x||y<0||m<=y) continue;
                int k=nk;
                if(grid[x][y]) k--;
                if(k<0||vis[x][y][k]) continue;
                if(x==n-1&&y==m-1) return vis[nx][ny][nk];
                qu.emplace(x,y,k);
                vis[x][y][k]=vis[nx][ny][nk]+1;
            }
        }
        return -1;
    }
};
