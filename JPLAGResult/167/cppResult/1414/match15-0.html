class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        int m = grid.size();
        int n = grid[0].size();
        
        vector< vector< vector<int> > > visited(m, vector< vector<int> >(n, vector<int>(k + 1, INT_MAX)));
        
        deque< vector<int> > Q;
        Q.push_back( {0, 0, 0} );
        visited[0][0][0] = 0;
        if (m - 1 == 0 && n - 1 == 0) return 0;
        
        int dr[] = {-1, 0, 1, 0};
        int dc[] = {0, 1, 0, -1};
        
        int res = INT_MAX;
        while (Q.size() > 0)
        {
            vector<int> cur = Q.front();
            Q.pop_front();
            
            int curSteps = visited[cur[0]][cur[1]][cur[2]];
            int removed  = cur[2];
            for (int a = 0; a < 4; ++a)
            {
                int nr = cur[0] + dr[a];
                int nc = cur[1] + dc[a];
                if (nr < 0 || nc < 0 || nr >= m || nc >= n) continue;
                
                int newRemoved = removed;
                if (grid[nr][nc] != 0) ++newRemoved;                
                if (newRemoved > k) continue;
                
                if (visited[nr][nc][newRemoved] > curSteps + 1)
                {
                    visited[nr][nc][newRemoved] = curSteps + 1;
                    Q.push_back( {nr, nc, newRemoved} );
                    if (nr == m - 1 && nc == n - 1)
                    {
                        res = min(res, curSteps + 1);
                    }
                }
            }
        }
        
        if (res == INT_MAX) return -1;
        
        return res;
    }
};
