class Solution {
public:
    int dp[42 * 42][42][42];
    
    struct st {
      int rem, x, y;  
    };
    
    int shortestPath(vector<vector<int>>& grid, int k) {
        int dx[] = {-1, 0, 0, 1},
            dy[] = {0, -1, 1, 0};
        
        int n = grid.size(), m = grid[0].size();
        
        memset(dp, -1, sizeof(dp));
        dp[0][0][0] = 0;
        vector<st> q(1, st{0, 0, 0});
        for (int i = 0; i < q.size(); ++i) {

            for (int t = 0; t < 4; ++t) {
                int nx = q[i].x + dx[t], ny = q[i].y + dy[t];
                if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
                int nrem = q[i].rem + grid[nx][ny];
                
                if (nrem <= k && dp[nrem][nx][ny] == -1) {
                    dp[nrem][nx][ny] = dp[q[i].rem][q[i].x][q[i].y] + 1;
                    q.push_back(st{nrem, nx, ny});
                }

            }
        }
        
        int ret = -1;
        for (int rem = 0; rem <= k; ++rem) {
            if (dp[rem][n - 1][m - 1] != -1 && (ret == -1 || ret > dp[rem][n - 1][m - 1]))
                ret = dp[rem][n - 1][m - 1];
        }
        
        return ret;
    }
};
