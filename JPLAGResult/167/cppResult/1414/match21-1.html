class Solution {
public:

    int shortestPath(vector<vector<int>>& grid, int k) {
        int m=grid.size();
        int n=grid[0].size();
        if(k>=m+n-3) return m+n-2;
        queue<tuple<int,int,int,int>> qq;
        qq.push(make_tuple(0,0,0,k));
        vector<vector<vector<int>>> dis(m,vector<vector<int>>(n,vector<int>(k+1,10000)));
        for(int i=0;i<=k;i++) dis[0][0][i]=0;

        int dd[4][2]={{1,0},{-1,0},{0,1},{0,-1}};
        while(!qq.empty()){

            int x,y,len,kk;
            tie(x,y,len,kk)=qq.front();
            qq.pop();
            for(int i=0;i<4;i++){
                int xx =x +dd[i][0];
                int yy=y+dd[i][1];
                if(xx<0||xx>=m||yy<0||yy>=n) continue;
                int k2=kk;
                if(grid[xx][yy]==1) k2--;
                if(k2<0) continue;
                if(dis[xx][yy][k2]<=len+1) continue;
                qq.push(make_tuple(xx,yy,len+1,k2));
                dis[xx][yy][k2]=len+1;
            }
        }
        int ans=10000;
        for(int i=0;i<=k;i++) ans=min(ans,dis[m-1][n-1][i]);
        if(ans==10000) return -1;
        else return ans;
    }
};
