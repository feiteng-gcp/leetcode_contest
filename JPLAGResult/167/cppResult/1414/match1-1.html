class Solution {
public:
    struct Node{
        int x;
        int y;
        int k;
    };
    vector<vector<int>> dir={{0,1},{0,-1},{1,0},{-1,0}};
    int bfs(vector<vector<int>>& grid,int k){
        int i = 0;
        int j = 0;
        int m = grid.size();
        int n = grid[0].size();
        if(grid[i][j]) k--;
        if(i==m-1 && j==n-1) return 0;
        Node start = {i,j,k};
        deque<Node> que;
        que.push_back(start);
        int cnt=1;
        unordered_set<string> table;
        string strs= to_string(i)+" "+to_string(j)+" "+to_string(k);
        table.insert(strs);
        while(!que.empty()){
            int tn = que.size();
            while(tn--){
                auto p = que.front();
                que.pop_front();
                int x = p.x;
                int y = p.y;
                // cout<<x<<" "<<y<<";";
                int dk = p.k;
                for(int q=0;q<4;q++){
                    int dx = x + dir[q][0];
                    int dy = y + dir[q][1];
                    if(dx<0||dy<0||dx>=m||dy>=n) continue;
                    if(dk==0 && grid[dx][dy]) continue;
                    string str= to_string(dx)+" "+to_string(dy)+" "+to_string(dk-grid[dx][dy]);
                    if(table.count(str)) continue;
                    if(dx== (m-1) && dy== (n-1)) return cnt;
                    que.push_back({dx,dy,dk-grid[dx][dy]});
                    table.insert(str);
                }
            }
            // cout<<endl;
            cnt++;
        }
        return -1;
        
    }
    int shortestPath(vector<vector<int>>& grid, int k) {
        return bfs(grid,k);
    }
};
