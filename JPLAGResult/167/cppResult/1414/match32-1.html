class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        int m = grid.size();
        int n = grid[0].size();
        vector<vector<vector<int>>> f(m, vector<vector<int>>(n, vector<int>(k+1, -1)));
        vector<vector<bool>> seen(m, vector<bool>(n, false));
        
        queue<pair<int, int>> q;
        q.push(make_pair(0, 0));
        int d = 0;
        int cnt = 1;
        f[0][0][0] = 0;
        seen[0][0] =true;
        
        int dx[] = {0, 1, 0, -1};
        int dy[] = {1, 0, -1, 0};
        //f[0]
        while(!q.empty()) {
            pair<int, int> cur = q.front();
            q.pop();
            
            int x = cur.first;
            int y = cur.second;
            f[x][y][0] = d;

            for(int i = 0; i < 4; i++) {
                int nx = x + dx[i];
                int ny = y + dy[i];
                if(nx >= 0 && nx < m && ny >= 0 && ny < n && !seen[nx][ny] && grid[nx][ny] == 0) {
                    seen[nx][ny] = true;
                    q.push(make_pair(nx, ny));
                }
            }
            
            cnt--;
            if(cnt == 0) {
                d++;
                cnt = q.size();
            }
        }
        
        for(int t = 1; t <= k; t++) {
            for(int x = 0; x < m; x++) {
                for(int y = 0; y < n; y++) {
                    if(grid[x][y] == 1) {
                        for(int i = 0; i < 4; i++) {
                            int nx = x + dx[i];
                            int ny = y + dy[i];
                            if(nx >= 0 && nx < m && ny >= 0 && ny < n && f[nx][ny][t-1] != -1) {
                                f[x][y][t] = min(1 + f[nx][ny][t-1], f[x][y][t] == -1 ? INT_MAX : f[x][y][t]);
                            }
                        }
                    }
                }
            }
            
            for(int x = 0; x < m; x++) {
                for(int y = 0; y < n; y++) {
                    if(grid[x][y] == 0) {
                        f[x][y][t] = f[x][y][t-1];
                        for(int i = 0; i < 4; i++) {
                            int nx = x + dx[i];
                            int ny = y + dy[i];
                            if(nx >= 0 && nx < m && ny >= 0 && ny < n && f[nx][ny][t] != -1) {
                                f[x][y][t] = min(1 + f[nx][ny][t], f[x][y][t] == -1 ? INT_MAX : f[x][y][t]);
                            }
                        }
                    }
                }
            }
            
            // cout << "t = " << t << endl;
            // for(int x = 0; x < m; x++) {
            //     for(int y = 0; y < n; y++) {
            //         cout << f[x][y][t] << " ";
            //     }
            //     cout << endl;
            // }
            // cout << endl;
        }
        
        return f[m-1][n-1][k];
    }
};
