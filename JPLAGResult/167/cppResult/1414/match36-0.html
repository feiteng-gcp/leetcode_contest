int dx[]={0,0,1,-1};
int dy[]={1,-1,0,0};
bool isValid(int x,int y,int R,int C){
    if(x>=0 && x<R && y>=0 && y<C)
        return true;
    return false;
}
class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        
        
        int m=grid.size();
        int n=grid[0].size();
        
        int dmap[41][41][41*41];
        
        for(int i=0;i<41;i++){
            for(int j=0;j<41;j++){
                for(int o=0;o<=k;o++){
                    dmap[i][j][o]=INT_MAX;
                }
            }
        }
        // for(int k=0;k<41*41;k++){
        //     dmap[0][0][k]=0;
        // }
        #define pi pair<int,int>
        queue<pair<int,pi>> q;
        q.push({k,{0,0}});
        dmap[0][0][k]=0;
        while(q.empty()==false){
            
            pair<int,pi> curr = q.front();
            int ck,cx,cy;
            ck=curr.first;
            cx=curr.second.first;
            cy=curr.second.second;
            q.pop();
            //cout<<cx<<" "<<cy<<" "<<ck<<endl;
            if(cx==m-1 && cy==n-1)
                return dmap[cx][cy][ck];
            
            for(int i=0;i<4;i++){
                int nx=dx[i]+cx;
                int ny=dy[i]+cy;
                if(!isValid(nx,ny,m,n)) continue;
                
                if(grid[nx][ny]==1 && ck>0 && dmap[nx][ny][ck-1]==INT_MAX){
                    dmap[nx][ny][ck-1]=dmap[cx][cy][ck]+1;
                    q.push({ck-1,{nx,ny}});
                }
                if(grid[nx][ny]==0 && dmap[nx][ny][ck]==INT_MAX){
                    dmap[nx][ny][ck]=dmap[cx][cy][ck]+1;
                    q.push({ck,{nx,ny}});
                }
            }
            
            
        }
        return -1;
    }
};
