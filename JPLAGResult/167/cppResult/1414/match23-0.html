class Solution {
private:
    struct State{
        int r, c;
        int k;
    };
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        int m = grid.size(), n = grid[0].size();
        int multiplier = 10000;
        vector<bool> visited(m * n * multiplier);
        
        for(int i = 0; i < m * n * multiplier; i++){
            visited[i] = false;
        }            
        
        queue<pair<State, int>> q;
        q.push({{0, 0, k}, 0});
        visited[0 * 0 * multiplier + k] = true;
        
        vector<vector<int>> dir = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        
        
        while(!q.empty()){
            auto element = q.front();
            q.pop();
            
            if (element.first.r == m - 1 && element.first.c == n - 1){
                return element.second;
            }
            
           // cout << element.first.r << ' ' << element.first.c << ' ' << element.first.k << endl;
            
            for(int i = 0; i < 4; i++){
                int new_r = element.first.r + dir[i][0];
                int new_c = element.first.c + dir[i][1];
                
                if (new_r >= 0 && new_r < m && new_c >= 0 && new_c < n){
                    //cout << "expand node: " << new_r << ' ' << new_c << endl;
                    if (grid[new_r][new_c] == 0){
                        int new_hash = (new_r * n + new_c) * multiplier + element.first.k;
                        if (!visited[new_hash]){
                            q.push({{new_r, new_c, element.first.k}, element.second + 1});
                            visited[new_hash] = true;
                        }
                    }
                    else{
                        if (element.first.k > 0){
                            int new_hash = (new_r * n + new_c) * multiplier + element.first.k - 1;
                            if (!visited[new_hash]){
                                q.push({{new_r, new_c, element.first.k - 1}, element.second + 1});
                                visited[new_hash] = true;
                            }
                        }
                    }
                }
            }
        }
        
        return -1;
    }
};
