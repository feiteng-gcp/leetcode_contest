class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        vector <vector <int> > v (grid.size(),vector<int>(grid[0].size(),INT_MAX));
        int n = grid[0].size(), m = grid.size();
        queue <vector <int> > q;
        q.push({0,0,0,0});
        while(!q.empty()){
            vector <int> tmp = q.front();
            q.pop();
            if(v[tmp[0]][tmp[1]] <= tmp[2])
                continue;
            v[tmp[0]][tmp[1]] = tmp[2];
            if(tmp[0] == m - 1 && tmp[1] == n - 1)
                return tmp[3];
            if(tmp[0] < m - 1){
                if(grid[tmp[0] + 1][tmp[1]] == 1 && tmp[2] < k)
                    q.push({tmp[0]+1,tmp[1],tmp[2] + 1,tmp[3] + 1});
                if(grid[tmp[0] + 1][tmp[1]] == 0)
                    q.push({tmp[0]+1,tmp[1],tmp[2],tmp[3] + 1});
            }
            if(tmp[1] < n - 1){
                if(grid[tmp[0]][tmp[1] + 1] == 1 && tmp[2] < k)
                    q.push({tmp[0],tmp[1] + 1,tmp[2] + 1,tmp[3] + 1});
                if(grid[tmp[0]][tmp[1] + 1] == 0)
                    q.push({tmp[0],tmp[1] + 1,tmp[2],tmp[3] + 1});
            }
            if(tmp[1] > 0){
                if(grid[tmp[0]][tmp[1] - 1] == 1 && tmp[2] < k)
                    q.push({tmp[0],tmp[1] - 1,tmp[2] + 1,tmp[3] + 1});
                if(grid[tmp[0]][tmp[1] - 1] == 0)
                    q.push({tmp[0],tmp[1] - 1,tmp[2],tmp[3] + 1});
            }
            if(tmp[0] > 0){
                if(grid[tmp[0] - 1][tmp[1]] == 1 && tmp[2] < k)
                    q.push({tmp[0] - 1,tmp[1],tmp[2] + 1,tmp[3] + 1});
                if(grid[tmp[0] - 1][tmp[1]] == 0)
                    q.push({tmp[0] - 1,tmp[1],tmp[2],tmp[3] + 1});
            }
        }
        return -1;
    }
};
