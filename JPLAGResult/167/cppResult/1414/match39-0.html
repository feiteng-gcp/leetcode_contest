class Solution {
public:
    int shortestPath(vector<vector<int>>& grid, int k) {
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<vector<int>>> dist(n + 1, vector<vector<int>>(m + 1, vector<int>(k + 1, INT_MAX)));
        int ans = INT_MAX;
        int dir[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        
        queue<vector<int>> q;
        q.push({0, 0, 0});
        dist[0][0][0] = 0;
        
        while (q.size()) {
            int x = q.front()[0];
            int y = q.front()[1];
            int cur_k = q.front()[2];
            int ds = dist[x][y][cur_k];
            q.pop();
            if (x == n - 1 && y == m - 1) {
                ans = min(ans, ds);
            }
            for (int d = 0; d < 4; ++d) {
                int nx = x + dir[d][0];
                int ny = y + dir[d][1];
                if (nx >= 0 && nx < n && ny >= 0 && ny < m) {
                    if (grid[nx][ny] == 0 || cur_k + 1 <= k) {
                        if (grid[nx][ny] == 0) {
                            if (dist[nx][ny][cur_k] == INT_MAX) {
                                dist[nx][ny][cur_k] = 1 + ds;
                                q.push({nx, ny, cur_k});
                            }
                        }
                        else {
                            if (dist[nx][ny][cur_k + 1] == INT_MAX) {
                                dist[nx][ny][cur_k + 1] = 1 + ds;
                                q.push({nx, ny, cur_k + 1});
                            }
                        }
                    }
                }
            }
        }
        
        
        return ans == INT_MAX ? -1 : ans;
    }
};
