/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    int getDecimalValue(ListNode* head) {
        auto cur = head;
        int re = 0;
        while(cur){
            re = 2 * re + cur->val;
            cur = cur->next;
        }
        return re;
    }
};
