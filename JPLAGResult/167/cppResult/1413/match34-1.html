class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int th) {
        int r = mat.size();
        int c = mat[0].size();
        
        vector<vector<int>> sum(r + 1, vector<int> (c + 1, 0));
        for(int i = 0; i < r; ++i) {
            for(int j = 0; j < c; ++j) {
                sum[i + 1][j + 1] = mat[i][j] + sum[i + 1][j] + sum[i][j + 1] - sum[i][j];
            }
        }
        /*
        for(int i = 0; i <= r; ++i) {
            for(int j = 0; j <= c; ++j) {
                cout << sum[i][j] << " ";
            }
            cout << endl;
        }
        */
        int ans = -1;
        for(int len = 0; len <= min(r, c); ++len) {
            for(int i = 1; i + len <= r; ++i) {
                for(int j = 1; j + len <= c; ++j) {
                    int outer = sum[i + len][j + len] - sum[i + len][j - 1] - sum[i - 1][j + len] + sum[i - 1][j - 1];
                    /*
                    int inner = 0;
                    if(i + len - 1 >= i + 1 && j + len - 1 >= j + 1) 
                        inner = sum[i + len - 1][j + len - 1] - sum[i + len - 1][j] - sum[i][j + len - 1] + sum[i][j];
                    if(outer - inner <= th) ans = max(ans, len);
                    */
                    if(outer <= th) ans = max(ans, len);
                }
            }
        }
        return ans + 1;
    }
};
