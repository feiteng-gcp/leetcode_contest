class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int H = mat.size();
        int W = mat[0].size();
        vector<vector<int>> imos1(H+1, vector<int>(W+1, 0));
        vector<vector<int>> imos(H+1, vector<int>(W+1, 0));
        for (int h = 0; h < H; ++h) {
            for (int w = 0; w < W; ++w) {
                imos1[h][w+1] = imos1[h][w] + mat[h][w];
            }
        }
        for (int w = 0; w < W+1; ++w) {
            for (int h = 0; h < H; ++h) {
                imos[h+1][w] = imos[h][w] + imos1[h][w];
            }
        }
        for (int i = min(H,W); i > 0; --i) {
            for (int h = 0; h <= H - i; ++h) {
                for (int w = 0; w <= W - i; ++w) {
                    int sum = imos[h+i][w+i] - imos[h+i][w] - imos[h][w+i] + imos[h][w];
                    if (sum <= threshold) {
                        return i;
                    }
                }                
            }
        }
        return 0;
    }
};
