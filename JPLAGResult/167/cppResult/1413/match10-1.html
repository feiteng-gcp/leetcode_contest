class Solution {
public:
    int maxSideLength(vector<vector<int>>& a, int thres) {
    int n = a.size(), m = a[0].size();
    vector<vector<int>> s(n + 1, vector<int>(m + 1));
    for (int i = 1; i <= n; i++) {
    	for (int j = 1; j <= m; j++) {
    		s[i][j] = a[i - 1][j - 1] + s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1];
    	}
    }
    int len = min(n, m);
    int r = 0;

    for (int k = len; k >= 1; k--) {
    	for (int i = k; i <= n; i++) {
    		for (int j = k; j <= m; j++) {
    			int v = s[i][j] - s[i][j - k] - s[i - k][j] + s[i - k][j - k];
                //cout << 
    			if(v <= thres) 
    				return k;
    		}
    	}
    }
    return r;
}
};
