class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int r = mat.size(), c = mat[0].size();
        vector<vector<int>> pre(r+1, vector<int>(c+1));
        for(int i=1;i<=r;++i){
            for(int j=1;j<=c;++j){
                pre[i][j] = pre[i][j-1] + pre[i-1][j] - pre[i-1][j-1] + mat[i-1][j-1];
            }
        }
        auto f = [&](int x0, int y0, int x1, int y1){
            return pre[x1][y1] - pre[x1][y0-1] - pre[x0-1][y1] + pre[x0-1][y0-1];
        };
        int ans = 0;
        for(int i=1;i<=r;++i){
            for(int j=1;j<=c;++j){
                for(int n=0;min(i,j)>n;++n){
                    if(f(i-n,j-n,i,j)<=threshold) ans = max(ans, n+1);
                    else break;
                }
            }
        }
        return ans;
    }
};
