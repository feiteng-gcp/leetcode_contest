class Solution {
 public:
  int maxSideLength(vector<vector<int>>& mat, int threshold) {
    int rows = mat.size();
    int cols = mat[0].size();
    vector<vector<int>> integral(rows + 1, vector<int>(cols + 1));

    for (int row = 0; row < rows; ++row) {
      for (int col = 0; col < cols; ++col) {
        integral[row + 1][col + 1] = integral[row + 1][col] +
                                     integral[row][col + 1] -
                                     integral[row][col] + mat[row][col];
      }
    }

    int max_len = 0;
    for (int row = 0; row < rows; ++row) {
      for (int col = 0; col < cols; ++col) {
        for (int len = 1;; ++len) {
          int row2 = row + len;
          int col2 = col + len;
          if (rows < row2 || cols < col2) {
            break;
          }
          auto sum = integral[row2][col2] - integral[row2][col] -
                     integral[row][col2] + integral[row][col];
          if (sum <= threshold) {
            max_len = max(max_len, len);
          }
        }
      }
    }
    return max_len;
  }
};
