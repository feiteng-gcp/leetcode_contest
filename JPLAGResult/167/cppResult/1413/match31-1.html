class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int n = mat.size(), m = mat[0].size();
        vector<vector<int>> sum(n, vector<int>(m, 0));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                sum[i][j] = mat[i][j];
                if (i > 0) {
                    sum[i][j] += sum[i - 1][j];
                }
                if (j > 0) {
                    sum[i][j] += sum[i][j - 1];
                }
                if (i > 0 && j > 0) {
                    sum[i][j] -= sum[i - 1][j - 1];
                }

            }
        }
        int result = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int side = 1; side <= min(i + 1, j + 1); side++) {
                    int curSum = sum[i][j];
                    if (i - side >= 0) {
                        curSum -= sum[i - side][j];
                    }
                    if (j - side >= 0) {
                        curSum -= sum[i][j - side];
                    }
                    if (i - side >= 0 && j - side >= 0) {
                        curSum += sum[i - side][j - side];
                    }
                    if (curSum <= threshold && side > result) {
                        result = side;
                    }
                }
            }
        }
        return result;
    }
};
