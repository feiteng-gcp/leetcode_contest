class Solution {
public:
    int maxSideLength(vector<vector<int>>& m, int threshold) {
        vector<vector<int>> s = m;
        for(int i = 0; i < m.size(); i++) {
            for(int j = 0; j < m[0].size(); j++) {
                if(i == 0 && j == 0) ;
                else if(i == 0) s[i][j] += s[i][j-1];
                else if (j == 0) s[i][j] += s[i-1][j];
                else {
                    s[i][j] = s[i][j] + s[i-1][j] + s[i][j-1] - s[i-1][j-1];
                }
            }
        }
        for(int r = min(m.size(), m[0].size()); r > 0; r--) {
             for(int i = 0; i < m.size(); i++) {
            for(int j = 0; j < m[0].size(); j++) {
                if(i + r > m.size() || j + r > m[0].size()) continue;
                int sum = s[i+r-1][j+r-1] - (i > 0 ?s[i-1][j+r-1]:0) - (j > 0 ? s[i+r-1][j-1]: 0) + (i >0 && j > 0? s[i-1][j-1]:0);
                if(sum <= threshold) {
                    return r;
                }
            }
             }
        }
        return 0;
    }
};
