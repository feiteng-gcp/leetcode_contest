class Solution {
public:
    int maxSideLength(vector<vector<int>>& a, int b) {
        int n=a.size(),m=a[0].size();
        vector<vector<int>> s(n+1,vector<int>(m+1));
        for(int i=1;i<=n;++i) {
            for(int j=1;j<=m;++j) {
                s[i][j] = s[i-1][j] + s[i][j-1] - s[i-1][j-1] + a[i-1][j-1];
            }
        }

        int ret = 0;
        for(int l=1;l<=min(n,m);++l) {
            for(int i=1;i<=n-l+1;++i) {
                for(int j=1;j<=m-l+1;++j) {
                    if(s[i+l-1][j+l-1] + s[i-1][j-1] - s[i+l-1][j-1] - s[i-1][j+l-1] <= b) ret =max(ret,l);
                }
            }
        }
        return ret;
    }
};
