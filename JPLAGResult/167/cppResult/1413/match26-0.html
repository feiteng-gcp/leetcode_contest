class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int m = mat.size(), n = mat[0].size();
        vector<vector<int>> sum(m + 1, vector<int>(n + 1));
        for (int i = 0; i < m; ++i) {
            int row_sum = 0;
            for (int j = 0; j < n; ++j) {
                row_sum += mat[i][j];
                sum[i + 1][j + 1] = sum[i][j + 1] + row_sum;
            }
        }
        int ans = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                for (int l = 1; l <= min(i + 1, j + 1); ++l) {
                    int s = sum[i + 1][j + 1] - sum[i + 1][j + 1 - l] - sum[i + 1 - l][j + 1] + sum[i + 1 - l][j + 1 - l];
                    if (s <= threshold) {
                        ans = max(ans, l);
                    } else {
                        break;
                    }
                }
            }
        }
        return ans;
    }
};
