class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int n=mat.size();
        int m=mat[0].size();
        vector<vector<int>> pref(n+1,vector<int>(m+1,0));
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                pref[i+1][j+1]=pref[i][j+1]+pref[i+1][j]-pref[i][j]+mat[i][j];
            }
        }
        int ans=0;
        for(int length=1;length<=min(n,m);length++){
            for(int i=0;i<n;i++){
                for(int j=0;j<m;j++){
                    if(i+length>n ||j+length>m)continue;
                    int temp = pref[i+length][j+length]-pref[i+length][j]-pref[i][j+length]+pref[i][j];
                    if(temp<=threshold)ans=length;
                }
            }
        }
        return ans;
    }
};
