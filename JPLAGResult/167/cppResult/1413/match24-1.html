int s[305][305];

class Solution {
    int h, w;
    

    int check(int y, int x, int len) {
        int y1 = y-len+1, x1 = x-len+1;
        return s[y+1][x+1] - s[y+1][x1] - s[y1][x+1] + s[y1][x1];
    }
    
    
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        memset(s, 0, sizeof(s));
        
        int ans = 0;

        h = mat.size(), w = mat[0].size();
        
        for (int i = 0; i < h; i++) {
            for (int j = 0; j < w; j++) {
                s[i+1][j+1] = mat[i][j] + s[i][j+1] + s[i+1][j] - s[i][j];
                
                int upper = min(i, j)+1, low = 1;
                
                while (upper >= low) {
                    int mid = (upper+low)/2;
                    if (check(i, j, mid) <= threshold) {
                        ans = max(ans, mid); 
                        low = mid+1;
                    } else {
                        upper = mid-1;
                    }
                    
                    
                }
                
            }
        }
        
        return ans;
    }
};
