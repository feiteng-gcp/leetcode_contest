class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int n = mat.size();
        int m = mat[0].size();
        vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
        
        for(int i = 1; i<= n; i ++){
            for(int j = 1; j <=m; j++){
                dp[i][j] = dp[i-1][j] + dp[i][j-1] - dp[i-1][j-1] + mat[i-1][j-1];
            }
        }
        
        int res = 0;
        for(int i = 1; i<= n; i ++){
            for(int j = 1; j <= m ; j++ ){
                for(int k = res + 1; ; k ++){
                    int ii = i + k - 1;
                    int jj = j + k - 1;
                    if ( ii > n || jj > m ) break;
                    int v = dp[ii][jj] - dp[ii][j-1] - dp[i-1][jj] + dp[i-1][j-1];
                    if (v <= threshold){
                        res = max(res, k);
                    }
                }
            }
        }
        return res;
    }
};
