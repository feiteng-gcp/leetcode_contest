class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        
        int m=mat.size();

        int n=mat[0].size();
        
        for(int i=0;i<m;i++){
            for(int j=1;j<n;j++){
                mat[i][j]+=mat[i][j-1];
            }
        }
        
       for(int j=0;j<n;j++){
           for(int i=1;i<m;i++){
                mat[i][j]+=mat[i-1][j];
            }
        }
        int ans=0;
        for(int s=1;s<=min(n,m);s++){
            for(int i=0;i+s-1<m;i++){
                for(int j=0;j+s-1<n;j++){
                    int a=i;
                    int b=j;
                    int c=i+s-1;
                    int d=j+s-1;
                    int csum = mat[c][d];

                    if(b-1>=0)
                        csum-=mat[c][b-1];
                    if(a-1>=0)
                        csum-=mat[a-1][d];
                    if(a-1>=0 && b-1>=0)
                        csum+=mat[a-1][b-1];
                    if(csum<=threshold && ans<s){
                        ans=s;
                    }
                }
            }    
        }
        return ans;
    }
};
