class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int m = mat.size();
        int n = mat[0].size();
        vector<vector<int>> dp(m+1,vector<int>(n+1));
        for(int i=1;i<=m;i++){
            for(int j=1;j<=n;j++){
                dp[i][j]= dp[i-1][j]+dp[i][j-1]-dp[i-1][j-1]+mat[i-1][j-1];
            }
        }
        int ans = 0;
        for(int i=1;i<=m;i++){
            for(int j=1;j<=n;j++){
                int edge = min(i,j);
                while(edge>ans){
                    int temp = dp[i][j]-dp[i-edge][j]-dp[i][j-edge]+dp[i-edge][j-edge];
                    if(temp<=threshold) ans = edge;
                    edge--;
                }
            }
        }
        // cout<<ans<<endl;
        return ans;
    }
};
