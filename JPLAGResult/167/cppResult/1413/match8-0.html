class Solution {
public:
    
    
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int n=mat.size(),m=mat[0].size();
        vector<vector<long long>> sum(n+1, vector<long long>(m+1, 0));
        
        
        sum[1][1]=mat[0][0];
        
//         for (int i=2;i<=n;i++)
//             sum[i][1]=sum[i-1][1]+mat[i][0];
        
//         for (int j=2;j<=m;j++)
//             sum[1][j]=sum[1][j-1]+mat[0][j];
        
        for (int i=1;i<=n;i++)
            for (int j=1;j<=m;j++)
                sum[i][j]=sum[i][j-1]+sum[i-1][j]-sum[i-1][j-1] +mat[i-1][j-1];
        
        int ans=0;
        
        for (int i=1;i<=n;i++)
            for (int j=1;j<=m;j++) {
                
                for (int l=0;;l++) {
                    int x=i+l;
                    int y=j+l;
                    
                    if (x>n || y>m)
                        break;
                    
                    long long count=sum[x][y]-sum[x][j-1]-sum[i-1][y]+sum[i-1][j-1];
                    
                    if (count<=threshold) 
                        ans=max(ans, l+1);
                }
                
            }
        
        return ans;
    }
};
