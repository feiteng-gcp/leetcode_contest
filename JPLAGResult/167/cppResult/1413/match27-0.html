class Solution {
public:
    int maxSideLength(vector<vector<int>>& a, int T) {
        int n = a.size(), m = a[0].size();

        for(int i = 0; i < n; i++) {
            for(int j = 1; j < m; j++) {
                a[i][j] += a[i][j - 1];
            }
        }
        for(int j = 0; j < m; j++) {
            for(int i = 1; i < n; i++) {
                a[i][j] += a[i - 1][j];
            }
        }
        int ans = 0;
        for(int r = 1; r <= min(n, m); r++) {
            for(int i = 0; i + r <= n; i++) {
                for(int j = 0; j + r <= m; j++) {
                    int s = a[i + r - 1][j + r - 1];
                    if(i > 0) s -= a[i - 1][j + r - 1];
                    if(j > 0) s -= a[i + r - 1][j - 1];
                    if(i > 0 && j > 0) s += a[i - 1][j - 1];
                    if(s <= T) {
                        ans = r;
                    }
                }
            }
        }

        return ans;
    }
};
