class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int n = mat.size();
        int m = mat[0].size();
        
        vector<vector<int>> h(n + 1, vector<int>(m + 1, 0));
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                h[i][j] = mat[i - 1][j - 1];
            }
        }
        
        vector<vector<int>> presum(n + 1, vector<int>(m + 1, 0));
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                presum[i][j] += h[i][j] + presum[i - 1][j] + presum[i][j - 1] - presum[i - 1][j - 1]; 
            }
        }
        
        for (int len = min(n, m); len >= 0; --len) {
            for (int x = len; x <= n; ++ x) {
                for (int y = len; y <= m; ++ y) {
                    int sum = presum[x][y] - presum[x - len][y] - presum[x][y - len] + presum[x - len][y - len];
                    if (sum <= threshold) return len;
                }
            }
        }
        
        return -1;
    }
};
