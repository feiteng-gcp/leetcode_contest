class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int n = mat.size();
        int m = mat[0].size();
        vector<vector<int>> pre(n+1,vector<int>(m+1));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                pre[i+1][j+1] = pre[i+1][j]+pre[i][j+1]-pre[i][j]+mat[i][j];
            }
        }
        
        int best = 0;
        for (int l = 1; l <= min(n, m); ++l) {
            for (int i = 0; i + l <= n; ++i) {
                for (int j = 0; j + l <= m; ++j) {
                    if (pre[i+l][j+l]-pre[i][j+l]-pre[i+l][j]+pre[i][j] <= threshold) {
                        best = max(best, l);
                    }
                }
            }
        }
        
        return best;
    }
};
