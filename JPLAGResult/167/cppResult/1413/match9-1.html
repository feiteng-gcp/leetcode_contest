class Solution {
public:
    vector<vector<int>> storage;
    void generate(vector<vector<int>>& matrix) {
        if(matrix.size()==0 || matrix[0].size()==0)
            return;
        storage = vector<vector<int>>(matrix.size()+1, vector<int>(matrix[0].size()+1));
        for(int i=0; i<matrix.size(); i++) {
            for(int j=0; j<matrix[0].size(); j++)
                storage[i+1][j+1] = storage[i+1][j] + storage[i][j+1] - storage[i][j] + matrix[i][j];
        }
    }
    
    int sumRegion(int row1, int col1, int row2, int col2) {
        return storage[row2+1][col2+1] + storage[row1][col1] - storage[row1][col2+1] - storage[row2+1][col1];
    }

    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        generate(mat);
        int max_length = 0;
        for(int i=0;i<mat.size(); i++) {
            for(int j=0; j<mat[0].size(); j++) {
                for(int length = 1; length<=mat.size(); length++) {
                    int x = i+length-1;
                    int y = j+length-1;
                    if(x>=mat.size()||y>=mat[0].size())
                        break;
                    int sm = sumRegion(i,j,x,y);
                    if(sm<=threshold) {
                        max_length = max(length, max_length);
                    }
                }
            }
        }
        return max_length;
    }
};
