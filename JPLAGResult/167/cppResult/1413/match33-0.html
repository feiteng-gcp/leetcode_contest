class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int n  = mat.size() ;   
        int m =  mat[0].size()  ;  
        int DP[n][m] ;  
        
        DP[0][0] = mat[0][0] ;   
        
        for(int i  = 1 ;  i < m ; ++i)
        {
          DP[0][i] = DP[0][i-1] + mat[0][i] ; 
        }
        
         for(int i  = 1 ;  i < n ; ++i)
        {
          DP[i][0] = DP[i-1][0] + mat[i][0] ; 
        }
        
        
        for(int i  =  1; i < n ; ++i)
        {
             for(int j  = 1 ; j < m ; ++j )
             {
                  
                  DP[i][j]  = DP[i][j-1] + DP[i-1][j] - DP[i-1][j-1]   + mat[i][j];  
             }
        }
        int ans = 0 ;
     
        
  
        
         for(int i  =   0 ; i < n ; ++i)
        {
             for(int j  = 0 ; j < m ; ++j )
             {
                  for(int k =  1  ; k + i < n && k + j < m ; ++k)
                  {
                      int sum  =DP[i+k][j+k]  ; 
                      if(i-1 >= 0 )
                      sum -= DP[i-1][j+k]  ; 
                      
                      if(j - 1 >=0 )
                      sum -= DP[i+k][j-1] ;  
                      
                      if(i - 1 >= 0 && j -1 >= 0)
                          sum += DP[i-1][j-1] ;  
                      
                     
                      
                      if(sum <= threshold)
                          ans = max(ans , k + 1) ;  
                  }
             }
        }
        return ans  ;
        
    }
};
