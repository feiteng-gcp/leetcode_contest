class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int m = mat.size(), n = mat[0].size();
        vector<vector<int>> sums(m, vector<int>(n, 0));
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                sums[i][j] = mat[i][j] + (i > 0 ? sums[i - 1][j] : 0) + (j > 0 ? sums[i][j - 1] : 0) - (i > 0 && j > 0 ? sums[i - 1][j - 1] : 0);

            }
        }
        int res = 0;
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < n; ++j) {
                for (int len = 1; len <= min(i, j) + 1; ++len) {
                    int cur = sums[i][j] - (j > len ? sums[i][j - len] : 0) - (i > len ? sums[i - len][j] : 0) + (i > len && j > len ? sums[i - len][j - len] : 0); 

                    if (cur <= threshold) {
                        res = max(res, len);
                    } else {
                        break;
                    }
                }
            }
        }
        return res;
    }
};
