class Solution {
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {

        int n = mat.size(), m = mat[0].size();
        vector<vector<int>> dp(n+1, vector<int>(m+1, 0));
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                dp[i+1][j+1] = dp[i][j+1] + dp[i+1][j] - dp[i][j] + mat[i][j];
            }
        }
        
        int l = 1, r = min(n, m);

        while(l <= r){
            int len = (l+r)/2;
            bool flag = false;
            for(int i = len; i <= n; i++){
                for(int j = len; j <=m; j++){
                    int val = dp[i][j] - dp[i-len][j] - dp[i][j-len] + dp[i-len][j-len];
                    if(val <= threshold){
                        flag = true;

                    }
                }
            }
            cout << len << " " << flag <<endl;
            if(flag == true){
                l = len + 1;
            }else{
                r = len - 1;
            }
        }
        return r;
    }
};
