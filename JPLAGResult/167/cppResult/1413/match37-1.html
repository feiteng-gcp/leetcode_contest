class Solution {
public:
    int maxSideLength(vector<vector<int>>& m, int th) {
        int r = m.size();
        int c = m[0].size();

        vector<vector<int>> sum(r, vector<int>(c, 0));
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                sum[i][j] = m[i][j];
                if(i)
                sum[i][j] += sum[i - 1][j];
                if(j)
                sum[i][j] += sum[i][j - 1];
                if(i && j)
                sum[i][j] -= sum[i - 1][j - 1];

            }
        }

        int ret = 0;
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                for (int k = 1; i + k <= r && j + k <= c; k++) {
                    int ii = i + k - 1;
                    int jj = j + k - 1;
                    int v = sum[ii][jj];
                    if (i)
                    v -= sum[i - 1][jj];
                    if (j)
                    v -= sum[ii][j - 1];
                    if (i && j)
                    v += sum[i - 1][j - 1];
                    if (v <= th) {
                        if (k > ret) ret = k;
                    }
                }
            }
        }
        return ret;
    }
};
