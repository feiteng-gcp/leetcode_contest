class Solution {
    
    int s[305][305] = {};
    
    int get(int y, int x, int len) {
        int y0 = y-len+1, x0 = x-len+1;
        return s[y+1][x+1] - s[y+1][x0] - s[y0][x+1] + s[y0][x0];
    }
    
    
    
public:
    int maxSideLength(vector<vector<int>>& mat, int threshold) {
        int h = mat.size(), w = mat[0].size();
        int ans = 0;
        
        for (int i = 0; i < h; i++) {
            for (int j = 0; j < w; j++) {
                s[i+1][j+1] = mat[i][j] + s[i][j+1] + s[i+1][j] - s[i][j];
                
                int upper = min(i, j)+1, lower = 1;
                while (lower <= upper) {
                    int mid = (lower + upper) /2;
                    if (get(i, j, mid) <= threshold) {
                        ans = max(ans, mid);
                        lower = mid+1;
                    } else {
                        upper = mid-1;
                    }
                }
                
                
            }
        }
        return ans;
        
    }
};
