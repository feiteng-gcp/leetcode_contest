class Solution {
public:
    vector<int> sequentialDigits(int low, int high) {
        vector<int> res;
        for (int i = 1; i < 10; i++) {
            for (int j = 1; j < 10; j++) {
                int x = 0;
                if (i + j - 1 > 9) continue;
                for (int r = 0; r < j; r++) {
                    x = x*10 + i+r;
                }
                if (x <= high && x >= low)
                    res.push_back(x);
            }
        }
        sort(res.begin(), res.end());
        return res;
    }
};
