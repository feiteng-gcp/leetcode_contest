class Solution {
public:
    vector<int> sequentialDigits(int low, int high) {
        vector<int> res;
        for (int i = to_string(low).size(); i <= to_string(high).size(); i++) {
            for (int j = 1; j + i <= 10; j++) {
                int n = 0;
                for (int k = 0; k < i; k++) {
                    n = n * 10 + k + j;
                }
                if (n >= low && n <= high)
                    res.push_back(n);
            }
        }
        return res;
    }
};
