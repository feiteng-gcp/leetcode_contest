class Solution {
public:
    vector<int> sequentialDigits(int low, int high) {
        vector<int> a;
        for (int l = 1; l <= 9; l++) {
            for (int i = 1; i + l - 1 <= 9; i++) {
                int cnt = 0;
                for (int j = 1; j <= l; j++) {
                    cnt *= 10;
                    cnt += (i + j - 1);
                }
                // cout << cnt << endl;
                if (cnt <= high && cnt >= low)
                    a.push_back(cnt);
            }
        }
        return a;
    }
};
