class Solution {
public:
    vector<int> sequentialDigits(int low, int high) {
        vector<int> ans;
        for (int i = 1; i <= 9; ++i) {
            for (int j = 1; j <= 10 - i; ++j) {
                int n = 0;
                for (int k = 0; k < i; ++k) {
                    n *= 10;
                    n += j + k;
                }
                if (low <= n && n <= high) {
                    ans.push_back(n);
                }
            }
        }
        return ans;        
    }
};
