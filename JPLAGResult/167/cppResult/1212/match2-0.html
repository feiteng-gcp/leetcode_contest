class Solution {
public:
    vector<int> sequentialDigits(int low, int high) {
        vector<int> ans;
        for(int l=1;l<=9;l++)
        {
            for(int i=1;i<=10-l;i++)
            {
                int now=0;
                for(int j=i;j<i+l;j++)
                {
                    now=now*10+j;
                }
                if(now>=low&&now<=high)ans.push_back(now);
            }
        }
        return ans;
    }
};
