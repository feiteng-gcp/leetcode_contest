class Solution {
 public:
  vector<int> sequentialDigits(int low, int high) {
    vector<int> res;
    for (int l = 2; l < 10; ++l) {
      for (int d = 1; d + l - 1 < 10; ++d) {
        int v = 0;
        for (int i = 0; i < l; ++i) {
          v = v * 10 + (d + i);
        }
        if (low <= v && v <= high) res.push_back(v);
      }
    }
    return res;
  }
};
