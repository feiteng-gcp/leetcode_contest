class Solution {
public:
    vector<int> sequentialDigits(int low, int high) {
        vector<int> ans;
        for (int i = 2; i <= 9; i++) {
            for (int j = 1; j <= 10 - i; j++)
            {
                int cur = 0;
                for (int k = 0; k < i; k++) {
                    cur *= 10;
                    cur += j + k;
                }
                if (cur >= low && cur <= high) {
                    ans.push_back(cur);
                }
            }
            
            
        }
        return ans;
    }
};
