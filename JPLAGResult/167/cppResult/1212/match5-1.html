class Solution {
public:
    vector<int> sequentialDigits(int low, int high) {
        vector<int> res;
        for(int len=1;len<=9;len++){
            for(int start=1;start+len-1<=9;start++){
                int t=0;
                for(int i=start;i<=start+len-1;i++){
                    t=t*10+i;
                }
                if(t>=low&&t<=high) res.push_back(t);
            }
        }
        return res;
    }
};
