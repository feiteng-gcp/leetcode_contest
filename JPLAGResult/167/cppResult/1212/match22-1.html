class Solution {
public:
    vector<int> sequentialDigits(int low, int high) {
        vector<int> ret;
        int ld = log(low)/log(10)+1;
        int hd = log(high)/log(10) + 1;
        int tmp = 0;
        for(int i = ld; i <= hd; i++) {
            for(int j = 1; j < 9; j++) {
                string str;
                for(int k = 0; k < i; k++) {
                    char c = '0' + j + k;
                    if(c > '9') goto l1;
                    str.push_back('0'+j+k);
                }
                tmp = stoi(str);
                if(tmp > high) goto l2;
                if(tmp >= low) ret.push_back(tmp);
                l1:;
            }
        }
        l2:
        return ret;
    }
};
