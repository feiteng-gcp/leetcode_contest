class Solution {
public:
    vector<int> sequentialDigits(int low, int high) {
        string temp = "123456789";
        vector<int>ans;
        for(int i=1;i<=9;i++){
            for(int j=0;j<=9-i;j++){
                int cur = stoi(temp.substr(j,i));
                if(cur>=low && cur<=high)ans.push_back(cur);
            }
            
        }
        return ans;
    }
};
