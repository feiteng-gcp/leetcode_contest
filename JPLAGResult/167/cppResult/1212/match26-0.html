class Solution {
public:
    void dfs(long long temp,vector<int> &re,int &l,int &r){
        if(temp>r) return;
        if(temp>=l){
            re.push_back(temp);
        }
        if(temp%10 == 9) return;
        if(temp){
            temp  = 10 * temp + (temp%10+1);
            dfs(temp,re,l,r);
            return;
        }
        for(int i=1;i<10;i++){
            long long ttemp = 10 * temp + i;
            dfs(ttemp,re,l,r);
        }
    }
    vector<int> sequentialDigits(int low, int high) {
        long long temp = 0;
        vector<int> re;
        dfs(temp,re,low,high);
        sort(re.begin(),re.end());
        return re;
    }
};
