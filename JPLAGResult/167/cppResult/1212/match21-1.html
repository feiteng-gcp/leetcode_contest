class Solution {
public:
    vector<int> sequentialDigits(int low, int high) {
        vector<int> res;
        string lw = to_string(low);
        string hi = to_string(high);
        int cur = lw[0] - '0';
        int cur_length = lw.size();
        while(cur_length<=hi.size()) {
            int base_i = 1;
            int ori_i = cur;
            for(int i=1; i<cur_length; i++) {
                base_i*=10;
                base_i+=1;
                ori_i*=10;
                ori_i+=cur+i;
                }
            bool is_break = false;
            for(int i=cur; i+cur_length<=10; i++) {
                if(ori_i>high) {
                    is_break = true;
                    break;
                }
                if(ori_i>=low)
                    res.push_back(ori_i);
                ori_i+=base_i;
            }
            cur = 1;
            cur_length++;
        }
        return res;
        
    }
};
