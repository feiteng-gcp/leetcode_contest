class Solution {
public:
    vector<int> ans;
    
    void dfs(long long now, int l, int r) {
        if (now>r)
            return;
        
        if (now>=l && now<=r)
            ans.push_back(now);
        
        int last=now%10;
        
        if (last==9)
            return;
        
        dfs(now*10+last+1, l, r);
    }
    
    vector<int> sequentialDigits(int low, int high) {
        
        for (int i=1;i<10;i++)
            dfs(i, low, high);
        
        sort(ans.begin(), ans.end());
        return ans;
    }
};
