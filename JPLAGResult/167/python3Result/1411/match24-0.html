# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def getDecimalValue(self, head: ListNode) -> int:
        temp = ''
        while head:
            temp += str(head.val)
            head = head.next
        return int(temp, 2)
