class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        m,n=len(grid),len(grid[0])
        vis=[[set() for _ in range(n)] for _ in range(m)]
        from collections import deque
        queue=deque([(0,0,0,k)])
        while queue:
            i,j,step,t=queue.popleft()
            if i==m-1 and j==n-1:
                return step
            for di,dj in ((1,0),(0,1),(-1,0),(0,-1)):
                ii,jj=i+di,j+dj
                if 0<=ii<m and 0<=jj<n and t not in vis[ii][jj]:
                    vis[ii][jj].add(t)
                    if grid[ii][jj]==0:
                        queue.append((ii,jj,step+1,t))
                    elif t>0:
                        queue.append((ii,jj,step+1,t-1))
        return -1
