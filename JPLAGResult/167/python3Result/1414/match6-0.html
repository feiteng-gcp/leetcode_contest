class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        m, n = len(grid), len(grid[0])
        visited = set()
        initState = (0, 0, k, 0)
        q = [(0, 0, k, 0)]
        visited.add((0, 0, k))
        while q:
            cr, cc, quota, d = q.pop(0)
            if cr == m-1 and cc == n-1:
                return d
            for nr, nc in [(cr-1, cc), (cr+1, cc), (cr, cc-1), (cr, cc+1)]:
                if 0 <= nr < m and 0 <= nc < n:
                    if grid[nr][nc] == 0:
                        if (nr, nc, quota) not in visited:
                            visited.add((nr, nc, quota))
                            q.append((nr, nc, quota, d+1))
                    else:
                        if quota > 0 and (nr, nc, quota-1) not in visited:
                            visited.add((nr, nc, quota-1))
                            q.append((nr, nc, quota-1, d+1))
        return -1
