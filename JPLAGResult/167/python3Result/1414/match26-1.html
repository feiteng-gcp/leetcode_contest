class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        m, n = len(grid), len(grid[0])
        if m == 1 and n == 1:
            return 0
        
        seen = {(0, 0, k)}
        q = [(0, 0, k)]
        step = 0
        while q:
            step += 1
            newq = []
            for i, j, e in q:
                for di, dj in zip([-1, 0, 1, 0], [0, -1, 0, 1]):
                    ni, nj = i + di, j + dj
                    if 0 <= ni < m and 0 <= nj < n:
                        e2 = e - grid[ni][nj]
                        if e2 >= 0 and (ni, nj, e2) not in seen:
                            if (ni, nj) == (m - 1, n - 1):
                                return step
                            newq
.append((ni, nj, e2))
                            seen.add((ni, nj, e2))
            q = newq
            # print(q)
        return -1
