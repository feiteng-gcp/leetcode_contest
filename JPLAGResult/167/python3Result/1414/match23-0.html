class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        n = len(grid)
        m = len(grid[0])
        dq = collections.deque([(0, 0, 0, 0)])
        visited = {(0, 0): 0}
        while dq:
            steps, i, j, o = dq.popleft()
            if i == n - 1 and j == m - 1:
                return steps
            for di, dj in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                ni, nj = di + i, dj + j
                if 0 <= ni < n and 0 <= nj < m:
                    no = o + grid[ni][nj]
                    if no <= k and ((ni, nj) not in visited or visited[(ni, nj)] > no):
                        visited[(ni, nj)] = no
                        dq.append((steps + 1, ni, nj, no))
        return -1
