class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        # BFS
        nrow = len(grid)
        ncol = len(grid[0])
        if nrow == ncol == 1:
            return 0 if grid[0][0] == 0 else 1
        q = collections.deque()
        seen = set()
        q.append((0,0, 0))
        seen.add((0, 0, 0))
        step = 0
        while q:
            nxt = collections.deque()
            step += 1
            while q:
                i, j, kk = q.popleft()
                for x, y in [(i + 1, j), (i - 1, j), (i, j + 1), (i, j - 1)]:
                    if 0 <= x < nrow and 0 <= y < ncol:
                        if grid[x][y] and kk < k:
                            if (x, y, kk + 1) not in seen:
                                if x == nrow - 1 and y == ncol - 1:
                                    return step
                                seen.add((x, y, kk + 1))
                                nxt.append((x, y, kk + 1))
                        elif grid[x][y] == 0:
                            if (x, y, kk) not in seen:
                                if x == nrow - 1 and y == ncol - 1:
                                    return step
                                seen.add((x, y, kk))
                                nxt.append((x, y, kk))
            q = nxt
        return -1
