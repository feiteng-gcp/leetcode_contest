class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        m,n = len(grid),len(grid[0])
        bfs = [(0,0,k)]
        seen = set()
        seen.add((0,0,k))
        
        step = -1
        while len(bfs) > 0:
            curLen = len(bfs)
            step += 1
            for i in range(curLen):
                cur = bfs.pop(0)
                if cur[0] == m-1 and cur[1] == n-1:
                    return step
                for (r,c) in ((cur[0]-1,cur[1]),(cur[0]+1,cur[1]),(cur[0],cur[1]-1),(cur[0],cur[1]+1)):
                    if r >= 0 and r<m and c>=0 and c<n and (r,c,cur[2]) not in seen and grid[r][c] == 0:
                        bfs.append((r,c,cur[2]))
                        seen.add((r,c,cur[2]))
                    if r >= 0 and r<m and c>=0 and c<n and (r,c,cur[2]-1) not in seen and grid[r][c] == 1 and cur[2] > 0:
                        bfs.append((r,c,cur[2]-1))
                        seen.add((r,c,cur[2]-1))
            #print(bfs,seen)
        
        return -1
