from collections import deque
class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        M, N = len(grid), len(grid[0])
        target = (M - 1, N - 1)
        h = deque([(
(0, 0), k)])
        step = 0
        visited = {}
        directs = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        while h:
            len_h = len(h)
            for _ in range(len_h):
                pos, ki = h.popleft()
                r, c = pos
                if grid[r][c] == 1:
                    if ki == 0:
                        continue
                    ki -= 1
                if pos == target:
                    return step
                if pos in visited and visited[pos] >= ki:
                    continue
                visited[pos] = ki
                for dr, dc in directs:
                    nr, nc = r + dr, c + dc
                    if 0 <= nr < M and 0 <= nc < N:
                        h.append(((nr, nc), ki))
            # print(h)
            step += 1
        return -1
