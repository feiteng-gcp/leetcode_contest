import collections
from typing import List


class Solution:
    def shortestPath(self, grid: List[List[int]], eliminating: int) -> int:
        n, m = len(grid), len(grid[0])
        dp = [[(-1, 9999) for _ in range(m)] for _ in range(n)]
        q = collections.deque([(0, 0, eliminating)])
        step = 0
        while q:
            for _ in range(len(q)):
                (y, x, remain) = q.popleft()
                if (y, x) == (n-1, m-1):
                    return step
                for k, j in ((y-1, x), (y+1, x), (y, x+1), (y, x-1)):
                    if 0 <= k < n and 0 <= j < m:
                        r = remain
                        if grid[k][j] == 1:
                            r = remain - 1
                        if r >= 0 and (r > dp[k][j][0] or (r == dp[k][j][0] and step < dp[k][j][1])):
                            dp[k][j] = (r, step)
                            q.append((k, j, r))
            step += 1
        return -1
