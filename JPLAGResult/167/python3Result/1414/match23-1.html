from collections import deque
class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        queue = deque([(0, 0, 0, 0)])
        seen = {(0, 0): 0}
        while queue:
            x, y, o, c = queue.popleft()
            if x == len(grid)-1 and y == len(grid[0])-1:
                return c
            for x_adj, y_adj in [(x-1, y), (x+1, y), (x, y-1), (x, y+1)]:
                new_o = o
                if 0 <= x_adj < len(grid) and 0 <= y_adj < len(grid[0]):
                    if grid[x_adj][y_adj] == 1:
                        new_o += 1
                    if o <= k and seen.get((
x_adj, y_adj), k+1) > new_o:
                        seen[(x_adj, y_adj)] = new_o
                        queue.append((x_adj, y_adj, new_o, c+1))
        return -1
