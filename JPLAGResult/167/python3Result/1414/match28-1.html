class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        n=len(grid)
        m=len(grid[0])
        que=set()
        que.add((0,0,k))
        visited=set()
        steps=0
        while que:
            tmp=set()
            for q in que:
                if q in visited:
                    continue
                visited.add(q)
                i,j,k0=q
                if i==n-1 and j==m-1:
                    
return steps
                for x,y in [[i,j+1],[i+1,j],[i,j-1],[i-1,j]]:
                    if 0<=x<n and 0<=y<m:
                        k2=k0
                        if grid[x][y]==1:
                            k2-=1
                        if k2<0:
                            continue
                        tmp.add((x,y,k2))
            que=tmp
            steps+=1
        return -1
