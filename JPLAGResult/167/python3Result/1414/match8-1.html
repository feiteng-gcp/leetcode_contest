from collections import deque, defaultdict
class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        
        m, n = len(grid), len(grid[0])
        
        q = deque([(0, 0, 0, 0)])
        seen = defaultdict(lambda: k+1)
        seen[(0, 0)] = 0
        
while q:
            i, j, cur, step = q.popleft()
            if i == m-1 and j == n-1:
                return step
            
            for ni, nj in [(i+1, j), (i-1, j), (i, j+1), (i, j-1)]:
                if 0 <= ni < m and 0 <= nj < n:
                    if grid[ni][nj]:
                        if seen[ni, nj] > cur + 1:
                            seen[ni, nj] = cur + 1
                            q.append((ni, nj, cur+1, step+1))
                    else:
                        if seen[ni, nj] > cur:
                            seen[ni, nj] = cur
                            q.append((ni, nj, cur, step+1))
        return -1
