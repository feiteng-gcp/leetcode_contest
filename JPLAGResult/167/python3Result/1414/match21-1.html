from collections import deque

DIRS = (-1, 0, 1, 0, -1)


class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        m, n = len(grid), len(grid[0])
        q = deque([(0, 0, k)])
        seen = set(q)
        
        dist = 0
        
        while q:
            q_size = len(q)
            for _ in range(q_size):
                row, col, can_remove = q.pop()
                if (row, col) == (m-1, n-1):
                    return dist
                for r_dir, c_dir in zip(DIRS, DIRS[1:]):
                    n_row, n_col = row + r_dir, col + c_dir
                    if not (0 <= n_row < m and 0 <= n_col < n):
                        continue
                    next_state = (n_row, n_col, can_remove - grid[n_row][n_col])
                    if (grid[n_row][n_col] == 0 or can_remove) and next_state not in seen:
                        seen.add(next_state)
                        q.appendleft(next_state)
            dist += 1
            
        return -1
