class Solution:
    def shortestPath(self, A: List[List[int]], K: int) -> int:
        M, N = len(A), len(A[0])
        q = [[0,0,0,0]]
        D = collections.defaultdict(lambda: float('inf'))
        D[0,0,0] = 0
        while q:
            d, k, i, j 
= heapq.heappop(q)
            if (i, j) == (M-1, N-1):
                return d
            # print(d, k, i, j)
            for i2, j2 in [[i-1,j],[i,j-1],[i+1,j],[i,j+1]]:
                if 0 <= i2 < M and 0 <= j2 < N:
                    k2 = k + A[i2][j2]
                    d2 = d + 1
                    if k2 <= K and d2 < D[i2,j2,k2]:
                        heapq.heappush(q, [d2, k2, i2, j2])
                        D[i2,j2,k2] = d2
        return -1
