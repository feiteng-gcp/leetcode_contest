class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        n, m = len(grid), len(grid[0])
        steps = [[[float('inf')] * (k + 1) for j in range(m)] for i in range(n)]
        steps[0][0][0] = 0
        b=[(0, 0, 0, 0)]
        dx, dy = [0, 1, 0, -1], [1, 0, -1, 0]
        while b:
            step, remove, x, y = heapq.heappop(b)
            step += 1
            for d in range(4):
                nx, ny = x + dx[d], y + dy[d]
                if 0 <= nx < n and 0 <= ny < m:
                    rem = remove + grid[nx][ny] 
                    if rem > k:
                        continue
                    if nx == n - 1 and ny == m - 1:
                        return step
                    if steps[nx][ny][rem] > step:
                        steps[nx][ny][rem] = step
                        heapq.heappush(b, (step, rem, nx, ny))
        m = min(steps[-1][-1])
        return -1 if m == float('inf') else m
