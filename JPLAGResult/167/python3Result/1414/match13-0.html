class Solution:
    def shortestPath(self, grid: List[List[int]], k: int) -> int:
        m, n = len(grid), len(grid[0])
        states = set([(0,0,k)])
        visited = set()
        level = 0
        
        while states:
            new_states = set()
            visited.update(states)
            
            for i,j,kill in states:
                if (i,j) == (m-1,n-1):
                    return level
                for ni,nj in [[i+1,j],[i-1,j],[i,j+1],[i,j-1]]:
                    if 0 <= ni < m and 0 <= nj < n:
                        if grid[ni][nj] == 0 and (ni,nj,kill) not in visited:
                            new_states.add((ni,nj,kill))
                        elif kill > 0 and (ni,nj,kill-1) not in visited:
                            new_states.add((ni,nj,kill-1))
            level += 1
            states = new_states
        return -1
