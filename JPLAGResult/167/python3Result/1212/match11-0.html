class Solution:
    def sequentialDigits(self, low: int, high: int):
        cand = []
        for i in range(2, 10):
            for start in range(1, 11 - i):
                s = ""
                for k in range(i):
                    s += str(start + k)
                cand.append(int(s))
        ans = []
        for c in cand:
            if c >= low and c <= high:
                ans.append(c)
        return ans
