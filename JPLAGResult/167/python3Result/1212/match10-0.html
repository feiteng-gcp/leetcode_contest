class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        res = []
        s_low = str(low)
        s_high = str(high)
        l_low = len(s_low)
        l_high = len(s_high)
        for l in range(l_low,l_high+1):
            for first in range(1,10-l+1):
                num = first
                for i in range(1,l):
                    num *= 10
                    num += first + i
                if low <= num <= high:
                    res.append(num)
        return res
