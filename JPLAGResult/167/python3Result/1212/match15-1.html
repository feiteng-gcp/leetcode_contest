class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        
        def get_nbr_digits(num: int) -> int:
            res = 0
            while num != 0:
                res += 1
                num 
= num // 10
            return res
        
        def make_ones(n: int) -> int:
            res = 0
            while n > 0:
                res = res * 10 + 1
                n -= 1
            return res
        
        def make_base(n: int) -> int:
            res = 0
            for i in range(1, n+1):
                res = res * 10 + i
            return res
        
        low_n = get_nbr_digits(low)
        high_n = get_nbr_digits(high)
        
        res = []
        for i in range(low_n, high_n + 1):
            base = make_base(i)
            ones = make_ones(i)
            if base >= low and base <= high:
                res.append(base)
            for _ in range(9 - i):
                base += ones
                if base >= low and base <= high:
                    res.append(base)
        
        return res
