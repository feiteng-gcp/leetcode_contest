def seq(l):
    res = []
    for d in range(1, 10):
        digs = [d + i for i in range(l)]
        if all(x < 10 for x in digs):
            res.append(int("".join(str(x) for x in digs)))
    return res
    
class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        res = []
        # return seq(3)
        for l in range(len(str(low)), len(str(high)) + 1):
            res.extend(x for x in seq(l) if low <= x <= high)
        return sorted(res)
