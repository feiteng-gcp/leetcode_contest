class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        lo, hi = self.getDigits(low), self.getDigits(high)
        res = []
        for i in range(lo, hi+1):
            res += self.helper(i, low, high)
        return res
        
    def helper(self, digits, low, high):
        res = []
        i = 1
        while i + digits - 1 <= 9:
            j, num, n = i, i, 1
            while n < digits:
                j += 1
                num *= 10
                num += j
                n += 1
            if num >= low and num <= high:
                res.append(num)
            elif num > high:
                break
            i += 1
        return res
    
    def getDigits(self, num):
        res = 0
        while num > 0:
            res += 1
            num //= 10
        return res
