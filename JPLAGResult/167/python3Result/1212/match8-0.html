all_seq = []
for length in range(1, 10):
    for digit in range(1, 11 - length):
        num = 0
        cur = digit
        for _ in range(length):
            num = num * 10 + cur
            cur += 1
        all_seq.append(num)

# print(all_seq)

class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        '''
        Once the first digit and the length is defined, the whole sequence will be defined.
        All possibilities:
        * length 1-9
        * first digit 1-9
        O(100)
        '''
        
        ans = []
        for i, n in enumerate(all_seq):
            if low <= n <= high:
                ans.append(n)
        return ans
