class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        t = list()
        for i in range(1, 10):
            for j in range(i, 10):
                st = ""
                for k in range(i, j + 1):
                    st += chr(k + 48)
                t.append(int(st))
        return sorted(tt for tt in t if low <= tt <= high)
