class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        ans = []
        kl, kh = len(str(low)), len(str(high))+1
        for k in range(kl, min(10,kh)):
            for i in range(1, 10-k+1):
                v = sum((i+j)*(10**(k-1-j)) for j in range(k))
                if v < low: continue
                if v > high: return ans
                ans.append(v)
        return ans
