class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        low, high = str(low), str(high)
        low_len, high_len = len(low), len(high)
        ret = []
        for i in range(low_len, high_len + 1):
            step = int('1' * i)
            if i == low_len:
                for j in range(ord(low[0]) - ord('0'), 10):
                    cur = j
                    valid = True
                    for k in range(1, i):
                        if cur % 10 == 9:
                            valid = False
                            break
                        cur = cur * 10 + (cur % 10 + 1)
                    if valid and cur <= int(high) and cur >= int(low):
                        ret.append(cur)  
            elif i == high_len:
                for j in range(1, ord(high[0]) - ord('0') + 1):
                    cur = j
                    valid = True
                    for k in range(1, i):
                        if cur % 10 == 9:
                            valid = False
                            break
                        cur = cur * 10 + (cur % 10 + 1)
                    if valid and cur <= int(high) and cur >= int(low):
                        ret.append(cur)
            else:
                for j  in range(1, 10):
                    cur = j
                    valid = True
                    for k in range(1, i):
                        if cur % 10 == 9:
                            valid = False
                            break
                        cur = cur * 10 + (cur % 10 + 1)
                    if valid and cur <= int(high) and cur >= int(low):
                        ret.append(cur)
        return ret
