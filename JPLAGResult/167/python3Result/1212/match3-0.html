class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        digits = '123456789'
        n = len(digits)
        res = []
        for s in range(n):
            for e in range(s + 1, n + 1):
                x = int(digits[s:e])
                if low <= x <= high:
                    res.append(x)
        res.sort()
        return res
