class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        def sequence_digit(digit_num):
            
            start = [1, 2, 3, 4, 5, 6, 7, 8, 9]
            result = []
            for idx in range(len(start)):
                cnt = 1
                num = start[idx]
                for next_idx in range(idx+1, len(start)):
                    num = num*10 + start[next_idx]
                    cnt += 1
                    
                    if cnt == digit_num:
                        result.append(num)
                        break
            return result
        low_digit = len(str(low))
        high_digit = len(str(high))
        
        result = []
        for digit in range(low_digit, high_digit + 1):
            for num in sequence_digit(digit):
                if low <= num <= high:
                    result.append(num)
        return result
