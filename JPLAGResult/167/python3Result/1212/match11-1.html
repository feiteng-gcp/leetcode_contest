from bisect import bisect_left, bisect_right
class Solution:
    def __init__(self):
        
        def generate(n):
            res = []
            
for i in range(1, 11 - n):
                cur = ''
                for j in range(n):
                    cur += str(i + j)
                res.append(int(cur))
            return res
        
        self.res = []
        for i in range(2, 10):
            self.res += generate(i)
        # print(self.res)
    
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        l, r = bisect_left(self.res, low), bisect_right(self.res, high)
        return self.res[l:r]
