class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        m, n = len(str(low)), len(str(high))
        def helper(k):
            res = []
            for l in range(1, 11 - k):
                temp = list(range(l, l + k))
                res.append(int(''.join(list(map(str,temp)))))
            return res
        ans 
= []
        for k in range(m, n + 1):
            for x in helper(k):
                if low <= x <= high:
                    ans.append(x)
        return ans
