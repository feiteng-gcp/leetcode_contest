class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        res = []
        candi = []
        L = len(str(high))
        
for l in range(2, L + 1):
            for start in range(1, 10 - l + 1):
                n = 0
                t = start
                for m in range(l):
                    n = n * 10 + t
                    t += 1
                candi.append(n)
        for n in candi:
            if low <= n < high:
                res.append(n)
        return res
