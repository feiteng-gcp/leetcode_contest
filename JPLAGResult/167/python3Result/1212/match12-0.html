class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        l, h = len(str(low)), len(str(high))
        ret = []
        for s in range(l, h + 1):
            for i in range(1, 11 - s):
                b = [str(j) for j in range(i, i + s)]
                c = int(''.join(b))
                if low <= c <= high:
                    ret.append(c)
        return ret
