class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        lo, hi = int(math.log10(low)) + 1, int(math.log10(high)) + 1
        ans = []
        
        def generate(n, start):
            s = 0
            for i in range(start, start + n):
                s = s * 10 + i
            return s
        
        for i in range(lo, hi + 1):
            for j in range(1, 11 - i):
                num = generate(i, j)
                if low <= num <= high:
                    ans.append(num)
        
        return ans
