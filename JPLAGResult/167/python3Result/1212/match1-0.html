class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        n = len(str(low))
        start = int("".join([str(i) for i in range(1, n + 1)]))
        res = []
        while start <= high:
            if start >= low:
                res.append(start)
            start = int("".join([str(int(i) + 1) for i in str(start)]))
            if len(str(start)) > n:
                start = int("".join([str(i) for i in range(1, len(str(start)) + 1)]))
                n = len(str(start))
        return res
