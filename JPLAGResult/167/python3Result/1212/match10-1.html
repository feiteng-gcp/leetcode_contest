class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        lisst = []
        for i in range(10):
            for j in range(1, 9):
                if j + i - 1 >= 10:
                    continue
                v 
= 0
                for k in range(i):
                    v *= 10
                    v += k + j
                
                if low <= v <= high:
                    lisst.append(v)
                
        lisst.sort()
        return lisst
