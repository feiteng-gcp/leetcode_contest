class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        def getAll(n, low, high):
            ans = []
            curr = list(range(1, n + 1))
            cand = sum([v * 10 ** i for i, v in enumerate(curr[::-1])])
            if low <= cand <= high:
                ans.append(cand)
            while curr[-1] < 9:
                curr = [v + 1 for v in curr]
                cand = sum([v * 10 ** i for i, v in enumerate(curr[::-1])])
                if low <= cand <= high:
                    ans.append(cand)
            return ans
