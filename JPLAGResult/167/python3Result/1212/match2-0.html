class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        res = set()
        # L = str(low)
        H = str(high)
        
        def dfs(cur):
            if low <= int(cur) <= high: res.add(int(cur))
            if len(cur) >= len(H): return
            last = cur[-1]
            if int(last) >= 9: return
            dfs(cur + str(int(last) + 1))
        
        for i in range(10):
            cur = str(i)
            dfs(cur)
            
        return sorted(res)
