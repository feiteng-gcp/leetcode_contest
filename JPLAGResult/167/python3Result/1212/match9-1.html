class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        from math import log10
        def seqnum(i, n):
            res = 0
            for _ in range(n):
                res *= 10
                res += i
                i += 1
            return res
        res = []
        ln = int(log10(low)) + 1
        hn = int(log10(high)) + 1
        
for n in range(ln, hn + 1):
            for i in range(1, 10 - n + 1):
                x = seqnum(i, n)
                if x >= low and x <= high:
                    res.append(x)
        return res
