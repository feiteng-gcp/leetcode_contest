class Solution:
    def get_seq(self, start, length):
        num = 0
        for i in range(0, length):
            num += (start + i) * 10**(length - i - 1)
        return num
    
    
def sequentialDigits(self, low: int, high: int) -> List[int]:
        low_len, high_len = len(str(low)), len(str(high))
        res = []
        for leng in range(low_len, high_len + 1):
            for start in range(1, 10):
                if start + leng > 10:
                    break
                num = self.get_seq(start, leng)
                if num >= low and num <= high:
                    res.append(num)
                if num > high:
                    break
        return res
