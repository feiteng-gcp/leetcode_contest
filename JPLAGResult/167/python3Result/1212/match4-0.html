class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        ans = []
        for l in range(2, 11):
            for i in range(1,11-l):
                k = int(''.join(map(str,list(range(i,i+l)))))
                if k >= low and k <= high:ans.append( k )
        return ans
