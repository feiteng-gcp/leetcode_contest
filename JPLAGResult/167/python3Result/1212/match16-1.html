class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        L_len = len(str(low))
        H_len = len(str(high))
        out = []
        for i in range(L_len, H_len + 1):
            for j in range(1, 10):
                a = list(range(j, j + i))
                if a[-1] < 10:
                    n = 0
                    for k in a:
                        n = n * 10 + k
                    if low <= n <= high:
                        out.append(n)
        return out
