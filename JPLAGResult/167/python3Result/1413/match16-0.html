class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        m = len(mat)
        n = len(mat[0])
        dps = [[0]*n for i in range(m)]
        for i in range(m):
            for j in range(n):
                if i == 0 and j == 0:
                    dps[i][j] = mat[0][0]
                elif j == 0:
                    dps[i][j] = dps[i-1][j] + mat[i][j]
                elif i == 0:
                    dps[i][j] = dps[i][j-1] + mat[i][j]
                else:
                    dps[i][j] = dps[i-1][j] + dps[i][j-1] - dps[i-1][j-1] + mat[i][j]
        res = 0
        flag = 0
        for i in range(m):
            for j in range(n):
                for l in range(res,min(i,j)+1):
                    if l == 0:
                        s = mat[i][j]
                    elif l == i and l == j:
                        s = dps[i][j]
                    elif l == i:
                        s = dps[i][j] - dps[i][j-l-1]
                    elif l == j:
                        s = dps[i][j] - dps[i-l-1][j]
                    else:
                        s = dps[i][j] - dps[i-l-1][j] - dps[i][j-l-1] + dps[i-l-1][j-l-1]
                    #print(i,j,l,s)
                    if s <= threshold:
                        flag= 1
                        res = l
                    else:
                        break
        print(dps)
        return res+1 if flag else 0
