class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        R, C = len(mat), len(mat[0])
        ps = [[0] * (C + 1) for _ in range(R + 1)]
        for r in range(R):
            for c in range(C):
                ps[r+1][c+1] = ps[r+1][c] + mat[r][c]
        
        for r in range(R):
            for c in range(C):
                ps[r+1][c+1] += ps
[r][c+1]
        
        def can_find(mid):
            for r in range(R - mid + 1):
                for c in range(C - mid + 1):
                    summ = ps[r + mid][c + mid] + ps[r][c] - ps[r+mid][c] - ps[r][c+mid]
                    if summ <= threshold:
                        return True
            return False
        
        # print(can_find(10))
        
        low, high = 0, min(R, C)
        while low < high:
            mid = (low + high + 1) // 2
            if not can_find(mid):
                high = mid - 1
            else:
                low = mid
        return high
