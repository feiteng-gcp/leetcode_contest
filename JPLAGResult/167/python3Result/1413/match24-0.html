class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        row = len(mat)
        col = len(mat[0])
        dp = [[0]*(col+1) for _ in range(row+1)]
        for i in range(row):
            for j in range(col):
                dp[i + 1][j + 1] = dp[i + 1][j] + mat[i][j]+dp[i][j+1]-dp[i][j]
        print(dp)

        # print(dp)

        k = 0
        for i in range(1, row+1):
            for j in range(1, col+1):
                while i > k and j > k and dp[i][j] - dp[i][j-k-1]-dp[i-k-1][j]+dp[i - k-1][j - k-1] <= threshold:
                    k += 1

        return k
