class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:

        nrow = len(mat)
        ncol = len(mat[0])
        k = min(nrow, ncol)
        pre 
= [[0] * (ncol + 1) for _ in range(nrow + 1)]
        for i in range(nrow):
            for j in range(ncol):
                pre[i + 1][j + 1] = pre[i + 1][j] + mat[i][j]
                
        for i in range(nrow):
            for j in range(ncol + 1):
                pre[i + 1][j] += pre[i][j]
                
        
def check(res):
            for i in range(nrow - res + 1):
                for j in range(ncol - res + 1):
                    if pre[i + res][j + res] - pre[i][j + res] - pre[i + res][j] + pre[i][j] <= threshold:
                        return True
            return False
        
        # print(pre)
        lo = 0
        hi = k
        while lo < hi:
            mid = (lo + hi + 1) // 2
            if check(mid):
                lo = mid
            else:
                hi = mid - 1
        return lo
