class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        
        def check(k):
            for i in range(1, n-k+1):
                for j in range(1, m-k+1):
                    if psa[i-1][j-1] + psa[i+k][j+k] - psa[i-1][j+k] - psa[i+k][j-1] <= threshold:
                        return True
            return False
        
        n = len(mat)
        m = len(mat[0])
        psa = [[0 for j in range(m+1)] for i in range(n+1)]
        for i in range(n):
            for j in range(m):
                psa[i+1][j+1] = mat[i][j]
                psa[i+1][j+1] += psa[i][j+1] + psa[i+1][j] - psa[i][j]
        l = 0
        r = min(n, m)
        while l < r:
            mid = l + r >> 1
            if check(mid):
                l = mid + 1
            else:
                r = mid
        return l
