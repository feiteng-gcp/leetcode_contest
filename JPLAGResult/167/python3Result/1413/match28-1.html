from collections import deque
class Solution:
    
def get_sum(self, mat_sum, i1, j1, i2, j2): # 左上角是i, j
        return mat_sum[i2][j2] - mat_sum[i2][j1-1] - mat_sum[i1-1][j2] + mat_sum[i1-1][j1-1]
        
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        if len(mat) == 0 or len(mat[0]) == 0:
            return 0
        m, n 
= len(mat), len(mat[0])
        mat_sum = [[0 for i in range(n + 1)] for j in range(m + 1)]
        for i in range(m):
            for j in range(n):
                mat_sum[i+1][j+1] = mat_sum[i+1][j] + mat_sum[i][j+1] + mat[i][j] - mat_sum[i][j]
                
        l, r = 0, min(m, n)
        while l < r:
            mid = int((l + r + 1) / 2)
            flag = False
            for i in range(1, m + 1):
                for j in range(1, n + 1):
                    if i + mid - 1 > m or j + mid -1 > n:
                        break
                    s = self.get_sum(mat_sum, i, j, i + mid - 1, j + mid - 1)
                    if s <= threshold:
                        flag = True
                        break
            if flag:
                l = mid
            else:
                r = mid - 1
        return l
