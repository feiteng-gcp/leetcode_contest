class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        M, N = len(mat), len(mat[0])
        prefix = [[0] * (N + 1) for _ in range(M + 1)]
        for r in range(M):
            for c in range(N):
                prefix[r + 1][c + 1] = mat[r][c] + prefix[r + 1][c] + prefix[r][c + 1] - prefix[r][c]
        # print(prefix)
        def sum_area(r0, c0, r1, c1):
            r1 += 1
            c1 += 1
            return prefix[r1][c1] - prefix[r1][c0] - prefix[r0][c1] + prefix[r0][c0]
        # print(sum_area(1, 1, 2,2))
        l = 1
        r = min(M, N)
        def valid(l):
            for r in range(M - l + 1):
                for c in range(N - l + 1):
                    r1 = r + l - 1
                    c1 = c + l - 1
                    if sum_area(r, c, r1, c1) <= threshold:
                        return True
            return False
              
        while l <= r:
            m = (l + r) // 2
            if valid(m):
                l = m + 1
            else:
                r = m - 1
        return r
