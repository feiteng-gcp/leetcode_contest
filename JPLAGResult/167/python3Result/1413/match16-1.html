class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        
        
        
        
        m, n = len(mat), len
(mat[0])
        presum = [[0]*n for _ in range(m)]
        for i in range(m):
            for j in range(n):
                if i == 0 and j == 0:
                    presum[i][j] = mat[i][j]
                elif i == 0 and j != 0:
                    presum[i][j] = presum[i][j-1] + mat[i][j]
                elif i != 0 and j == 0:
                    presum[i][j] = presum[i-1][j] + mat[i][j]
                else:
                    presum[i][j] = presum[i-1][j] + presum[i][j-1] - presum[i-1][j-1] + mat[i][j]
        
        maxside = 0
        for i in range(m):
            for j in range(n):
                if i == 0 or j == 0:
                    if mat[i][j] <= threshold:
                        maxside = max(maxside, 1)
                else:
                    for curside in range(maxside+1, min(i+1,j+1)+1):
                        if curside == i+1 and curside == j+1:
                            cursum 
= presum[i][j]
                        elif curside == i+1 and curside != j+1:
                            cursum = presum[i][j] - presum[i][j-curside]
                        elif curside != i+1 and curside == j+1:
                            cursum = presum[i][j] - presum[i-curside][j]
                        else:
                            cursum = presum[i][j] - presum[i-curside][j] - presum[i][j-curside] + presum[i-curside][j-curside]
                        if cursum > threshold:
                            break
                        else:
                            maxside = max(maxside, curside)
        return maxside
