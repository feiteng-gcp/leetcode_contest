class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        l, r = 0, min(len(mat), len(mat[0]))
        ans = 0
        s = [[0 for i in range(len(mat[0]) + 1)] for j in range(len(mat) + 1)]
        for i in range(len(mat)):
            for j in range(len(mat[0])):
                s[i + 1][j + 1] = s[i][j + 1] + s[i + 1][j] - s[i][j] + mat[i][j]
        inf = int(1e18)
        while l <= r:
            m = (l + r) // 2
            mi = inf
            for i in range(len(mat) - m + 1):
                for j in range(len(mat[0]) - m + 1):
                    mi = min(mi, s[i + m][j + m] - s[i][j + m] - s[i + m][j] + s[i][j])
            if mi <= threshold:
                ans = m
                l = m + 1
            else:
                r = m - 1
        return ans
