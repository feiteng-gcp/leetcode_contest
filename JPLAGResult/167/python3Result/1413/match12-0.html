class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        m, n = len(mat), len(mat[0])
        
        pre_sum = [[0] * n for i in range(m)]
        for i in range(m):
            for j in range(n):
                pre_sum[i][j] += mat[i][j]
                if j > 0:
                    pre_sum[i][j] += pre_sum[i][j - 1]
        
        for i in range(1, m):
            for j in range(n):
                pre_sum[i][j] += pre_sum[i - 1][j]
        
        def area(l):
            for i in range(l - 1, m):
                for j in range(l - 1, n):
                    area = pre_sum[i][j]
                    if j != l - 1 and i != l - 1:
                        area += pre_sum[i - l][j - l]
                    if j != l - 1:
                        area -= pre_sum[i][j - l]
                    if i != l - 1:
                        area -= pre_sum[i - l][j]
                    # print(l, area)
                    if area <= threshold:
                        return True
            return False
        
        lo, hi = 0, min(m, n)
        while lo < hi:
            mid = lo + (hi - lo + 1) // 2
            
            if area(mid):
                lo = mid
            else:
                hi = mid - 1
        
        return lo
