class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        m, n = len(mat), len(mat[0])
        for i in range(m):
            for j in range(1, n):
                mat[i][j] += mat[i][j - 1]
        for j in range(n):
            for i in range(1, m):
                mat[i][j] += mat[i - 1][j]
                
        
        def get_mat(i, j):
            if 0 <= i < m and 0 <= j < n:
                return mat[i][j]
            return 0
        
        def fit(k):
            for i in range(k - 1, m):
                for j in range(k - 1, n):
                    s = get_mat(i, j) - get_mat(i, j - k) - get_mat(i - k, j) + get_mat(i - k, j - k)
                    if s <= threshold:
                        return True
            return False
        
        lo, hi = 0, min(m, n)
        while lo < hi:
            mid = (lo + hi + 1) // 2
            if fit(mid):
                lo = mid
            else:
                hi = mid - 1
        return lo
