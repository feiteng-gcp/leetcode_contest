class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        n, m = len(mat), len(mat[0])
        ps = [[0] * (m + 1) for i in range(n + 1)]
        for i in range(n):
            for j in range(m):
                ps[i + 1][j + 1] = ps[i + 1][j] + ps[i][j + 1] - ps[i][j] + mat[i][j]
        dp = [[0] * m for i in range(n)]
        ret = 0
        for i in range(n):
            for j in range(m):
                if i == 0 or j == 0:
                    dp[i][j] = 1 if mat[i][j] <= threshold else 0
                else:
                    for l in range(dp[i - 1][j - 1] + 1, -1, -1):
                        if ps[i + 1][j + 1] - ps[i + 1][j + 1 - l] - ps[i + 1 - l][j + 1] + ps[i + 1 - l][j + 1 - l] <= threshold:
                            dp[i][j] = l
                            break
                    ret = max(ret, dp[i][j])
        return ret
