import math

class Solution:
    def maxSideLength(self, mat: List[List[int]], th: int) -> int:
        
        m, n = len(mat), len(mat[0])
        
        mt = [ [0] * (n + 1) for _ in range(m + 1) ]
        
        for i in range(1, m + 1):
            for j in range(1, n + 1):                
                mt[i][j] = mt[i - 1][j] + mt[i][j - 1] - mt[i - 1][j - 1] + mat[i - 1][j - 1]
            
        def chk(k):
            nonlocal th
            
            for i in range(1, m + 1):
                for j in range(1, n + 1):
                    
                    if min(i, j) >= k:
                        
                        temp = mt[i][j] - mt[i - k][j] - mt[i][j - k] + mt[i - k][j - k]
                        
                        if temp <= th:
                            return True
                        
            return False
        
        
        l = 0
        r = min(m, n)
        
        while l < r:
            mid = int( math.ceil( (l + r) / 2.0) )
            
            if chk(mid):
                l = mid
            else:
                r = mid - 1
                
        return l
