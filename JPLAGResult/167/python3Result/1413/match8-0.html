class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        m,n=len(mat),len(mat[0])
        dp=[[0]*(n+1) for _ in range(m+1)]
        for i in range(1,m+1):
            for j in range(1,n+1):
                dp[i][j]=dp[i][j-1]+mat[i-1][j-1]+dp[i-1][j]-dp[i-1][j-1]
        r=min(m,n)
        l=0
        while l<=r:
            ans=(l+r)//2
            if not ans:
                return 0
            temp=1<<31
            for i in range(1,m+2-ans):
                for j in range(1,n+2-ans):
                    res=dp[i+ans-1][j+ans-1]-dp[i+ans-1][j-1]-dp[i-1][j+ans-1]+dp[i-1][j-1]
                    temp=min(temp,res)
            if temp>threshold:
                r=ans-1
            else:
                re=ans
                l=ans+1
        return re
