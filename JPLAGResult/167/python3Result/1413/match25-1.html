class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        m, n = len(mat), len(mat[0])
        l, r = 1, min(m, n)
        res = 0
        csum = [[0]*(n+1) for _ in range(m+1)]
        for i in range(1, m+1):
            for j in range(1, n+1):
                csum
[i][j] = csum[i-1][j] + csum[i][j-1] - csum[i-1][j-1] + mat[i-1][j-1]

        while l <= r:
            mi = l + (r - l) // 2
            ans = float('inf')
            for i in range(m-mi+1):
                for j in range(n-mi+1):
                    ans 
= min(ans, csum[i+mi][j+mi] - csum[i+mi][j] - csum[i][j+mi] + csum[i][j])
            if ans <= threshold:
                l = mi + 1
                res = mi
            else:
                r = mi - 1
        return res
