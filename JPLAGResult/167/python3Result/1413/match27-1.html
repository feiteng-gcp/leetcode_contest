class Solution:
    def maxSideLength(self, mat: List[List[int]], threshold: int) -> int:
        m, n = len(mat), len(mat[0])
        cumsum = [[0] * (n + 1) for _ in range(m + 1)]
        for i in range(m):
            for j in range(n):
                cumsum[i + 1][j + 1] = cumsum[i][j + 1] + cumsum[i + 1][j] - cumsum[i][j] + mat[i][j]
        
        class Helper:
            def __getitem__(self, s):
                return min(cumsum[i + s][j + s] - cumsum[i][j + s] - cumsum[i + s][j] + cumsum[i][j] for i in range(m - s + 1) for j in range(n - s + 1))
                
        return bisect.bisect_right(Helper(), threshold, hi=min(m, n) + 1) - 1
