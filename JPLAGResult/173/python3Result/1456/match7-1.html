class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], distanceThreshold: int) -> int:
        minDist = [[float("inf")] * n for i in range(n)]
        for i in range(n):
            minDist[i][i] = 0
        for e in edges:
            minDist[e[0]][e[1]] = e[2]
            minDist[e[1]][e[0]] = e[2]
        for k in range(n):
            for i in range(n):
                for j in range(n):
                    if minDist[i][j] > minDist[i][k] + minDist[k][j]:
                        minDist[i][j] = minDist[i][k] + minDist[k][j]
        
        res, count = -1, n + 1
        for i in range(n - 1, -1, -1):
            curCount = 0
            for j in range(n):
                if i != j and minDist[i][j] <= distanceThreshold:
                    curCount += 1
            if curCount < count:
                res, count = i, curCount
            
        return res
