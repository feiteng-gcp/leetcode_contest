class Solution:
    def findTheCity(self, N: int, E: List[List[int]], T: int) -> int:
        D = [[float('inf')] * N for _ in range(N)]
        for u in range(N):
            D[u][u] = 0
        for [u, v, w] in E:
            D[u][v] = D[v][u] = w
        for w in range(N):
            for u in range(N):
                for v in range(N):
                    D[u][v] = min(D[u][v], D[u][w] + D[w][v])
        ans = (N, N)
        for u in range(N):
            ct = sum(D[u][v] <= T for v in range(N))
            ans = min(ans, (ct, -u))
        return -ans[1]
