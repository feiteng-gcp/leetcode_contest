class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], th: int) -> int:
        dis = [[1e99 for _ in range(n)] for _ in range(n)]
        for i in range(n):
            dis[i][i] = 0
        
        for f, t, w in edges:
            dis
[f][t] = w
            dis[t][f] = w
        
        for k in range(0, n):
            for j in range(0, n):
                for i in range(0, n):
                    dis[i][j] = min(dis[i][j], dis[i][k] + dis[k][j])
        
        ans = -1
        mxc = 1e99
        for i in range(n):
            cnt = 0
            for j in range(n):
                if dis[i][j] <= th:
                    cnt += 1
            if cnt <= mxc:
                mxc = cnt
                ans = i
        return ans
