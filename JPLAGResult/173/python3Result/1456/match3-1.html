class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], distanceThreshold: int) -> int:
        inf = 1 << 60
        g = [[inf] * n for _ in range(n)]
        
        for i in range(n):
            g[i][i] = 0
        
        
for u, v, w in edges:
            g[u][v] = w
            g[v][u] = w

        for k in range(n):
            for i in range(n):
                for j in range(n):
                    g[i][j] = min(g[i][j], g[i][k] + g[k][j])
            
        best = inf
        besti = -1
        
for x in range(n):
            cost = 0
            for y in range(n):
                if g[x][y] <= distanceThreshold:
                    cost += 1
            #print(x, cost)
            if cost <= best:
                best = cost
                besti = x
        return besti
