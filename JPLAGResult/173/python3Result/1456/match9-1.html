class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], distanceThreshold: int) -> int:
        dist = [[10000000 for j in range(n)] for i in range
(n)]
        for i in range(n):
            dist[i][i] = 0
        for e in edges:
            dist[e[0]][e[1]] = e[2]
            dist[e[1]][e[0]] = e[2]
        for k in range(n):
            for i in range(n):
                for j in range(n):
                    if dist[i][j] > dist[i][k] + dist[k][j]:
                        dist[i][j] = dist[i][k] + dist[k][j]
        min_count, min_i = n, -1
        
for i in range(n):
            count = 0
            for j in range(n):
                if dist[i][j] <= distanceThreshold:
                    count += 1
            if count <= min_count:
                min_count = count
                min_i = i
        return min_i
