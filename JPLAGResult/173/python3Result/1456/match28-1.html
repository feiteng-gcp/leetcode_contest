        distmat = [[99999999 for j in range(n) ] for i in range(n)]

                for j in range(n):
                    if distmat[i][k] + distmat[k][j] < distmat[i][j]:
                        distmat[i][j] = distmat[i][k] + distmat[k][j]
        minn = 99999999
        ans = -1
