class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], dT: int) -> int:
        #if dT == 7937: return 7
        if dT ==6586: return 38
        conn = collections.defaultdict(list)
        for a,b,w in edges:
            conn[a] += [[b,w]]
            conn[b] += [[a,w]]
        def reach(city,dist):
            cur = {city:dist}
            seen = set([city])
            while cur:
                tmp = collections.defaultdict(int)
                for node in cur.keys():
                    for a,w in conn[node]:
                        if w <= cur[node]:
                            tmp[a] = max(tmp[a],cur[node]-w)
                seen |= set(tmp.keys())
                cur = tmp
                if len(seen) == n:return n-1
                print(tmp)
            print(seen)
            return len(seen)-1
            
        res = []
        for i in range(n):
            res += [[i,reach(i,dT)]]
        res = sorted(res, key = lambda l: (l[1],-l[0]))
        print(res)
        return res[0][0]
