class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], distanceThreshold: int) -> int:
        MAXM = 10**8
        dp = [[0 if i==j else MAXM for i in range
(n)] for j in range(n)]
        for u, v, w in edges:
            dp[u][v] = dp[v][u] = w
        
        for k in range(n):
            for u in range(n):
                for v in range(n):
                    dp[u][v] = min(dp[u][v], dp[u][k]+dp[k][v])
        
        minm = n
        res = n-1
        
for u in range(n):
            tot = 0
            for v in range(n):
                if v != u and dp[u][v] <= distanceThreshold:
                    tot += 1
            if tot <= minm:
                minm = tot
                res = u
        
        return res
