class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], distanceThreshold: int) -> int:
        
        inf = 10**10
        a = [[inf for _ in range(n)] for _ in range(n)]
        for i, j, w in edges:
            a[i][j] = w
            a[j][i] = w

        for k in range(n):
            for i in range(n):
                for j in range(n):
                    a[i][j] = min(a[i][j], a[i][k] + a[k][j])
                    
        ans = -1
        mn = inf
        for i in range(n):
            num = sum([a[i][j] <= distanceThreshold for j in range(n) if j != i])
            # print(i, num)
            if num <= mn:
                mn = num
                ans = i
             
        return ans
