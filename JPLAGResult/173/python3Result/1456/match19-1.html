class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], distanceThreshold: int) -> int:
        dist = [[float('inf') for _ in range(n)] for _ in range(n)]
        
for source, destination, w in edges:
            dist[source][destination] = w
            dist[destination][source] = w
        
        for middle in range(n):
            for source in range(n):
                for destination in range(n):
                    dist[source][destination] = min(dist[source][destination], dist[source][middle] + dist[middle][destination])
        
        best_cnt = float('inf')
        result = None
        for node in range(n):
            cnt = 0
            for node_b in range(n):
                
                if node_b != node and (dist[node][node_b] <= distanceThreshold or dist
[node_b][node] <= distanceThreshold):
                    cnt += 1
            #print(result, cnt, best_cnt, node)
            if cnt <= best_cnt:
                result = node
                best_cnt = cnt
            
        return result
