class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], distanceThreshold: int) -> int:
        MAXD = 10**8
        dist = [[MAXD] * n for _ in range(n)]
        for a, b, c in edges:
            dist[a][b] = dist[b][a] = c
        for i in range(n):
            dist[i][i] = 0
        for k in range(n):
            for i in range(n):
                for j in range(n):
                    if dist[i][k] + dist[k][j] < dist[i][j]:
                        dist[i][j] = dist[i][k] + dist[k][j]
        
        cnt = n + 1
        ret = 0
        for i in range(n):
            cur = 0
            for j in range(n):
                if dist[i][j] <= distanceThreshold: cur += 1
            if cur <= cnt:
                cnt = cur
                ret = i
        return ret
