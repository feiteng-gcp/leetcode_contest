from collections import defaultdict
class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], distanceThreshold: int) -> int:
        
        graph = defaultdict(dict)
        for i, j, cost in edges:
            graph[i][j] = cost
            graph[j][i] = cost
                
        res = [0] * n
        for i in range(n):
            q = []
            heapq.heappush(q, (0, i))
            seen = set()

            parent = [-1] * n
            dist = [float('inf')] * n
            dist[i] = 0

            while q:
                cur, u = heapq.heappop(q)
                if cur > distanceThreshold:
                    break
                if u not in seen:
                    seen.add(u)
                    for v in graph[u]:
                        if dist[u] + graph[u][v] < dist[v]:
                            dist[v] = dist[u] + graph[u][v]
                            parent[v] = u
                            heapq.heappush(q, (dist[v], v))
            res[i] = len(seen) - 1
        
        max_i, max_cnt = 0, res[0]
        for i, cnt in enumerate(res):
            if cnt <= max_cnt:
                max_i, max_cnt = i, cnt
        return max_i
