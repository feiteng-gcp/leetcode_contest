class Solution:
    def findTheCity(self, n: int, edges: List[List[int]], dT: int) -> int:
        m = [[float('inf')]*n for i in range(n)]
        for i in range(n):
            m[i][i] = 0
        for e in edges:
            m[e[0]][e[1]] = e[2]
            m[e[1]][e[0]] = e[
2]
        for k in range(n):
            for i in range(n):
                for j in range(n):
                    if m[i][j] > m[i][k] + m[k][j]:
                        m[i][j] = m[i][k] + m[k][j]
        res,Min = -1,float('inf')
        
for i in range(n):
            cnt = 0
            for j in range(n):
                if i != j and m[i][j] <= dT:
                    cnt += 1
            if cnt <= Min:
                Min = cnt
                res = i
        return res
