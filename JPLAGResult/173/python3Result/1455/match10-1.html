class Solution:
    def filterRestaurants(self, restaurants: List[List[int]], veganFriendly: int, maxPrice: int, maxDistance: int) -> List[int]:
        tupes = [tuple(res) for res in restaurants]
        ans 
= []
        for tup in tupes:
            if veganFriendly:
                if tup[2] == veganFriendly and tup[3] <= maxPrice and tup[4]<=maxDistance:
                    ans.append((tup[0],tup[1]))
            else:
                if tup[3] <= maxPrice and tup[4]<=maxDistance:
                    ans.append((tup[0],tup[1]))
        ans.sort(key = lambda k: (-k[1],-k[0]))
        ans = [tup[0] for tup in ans]
        return ans
