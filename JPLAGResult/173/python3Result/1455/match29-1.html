class Solution:
    def filterRestaurants(self, restaurants: List[List[int]], veganFriendly: int, maxPrice: int, maxDistance: int) -> List[int]:
        l 
= []
        for res in restaurants:
            if res[2]<veganFriendly or res[3]>maxPrice or res[4]>maxDistance:
                continue
            l.append([res[1], res[0]])
        l.sort(reverse=True)
        return [ele[1] for ele in l]
