class Solution:
    def filterRestaurants(self, rs: List[List[int]], vF: int, mP: int, mD: int) -> List[int]:
        res = [r for r in rs if (r[2]>=vF and r[4] <= mD and r[3]<= mP)]
        res = sorted(res, key = lambda l:(l[1],l[0],l[4]), reverse = True)
        return [l[0] for l in res]
