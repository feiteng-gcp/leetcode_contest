class Solution:
    def filterRestaurants(self, restaurants: List[List[int]], veganFriendly: int, maxPrice: int, maxDistance: int) -> List[int]:
        
        res = [a for a in restaurants if a[2] >= veganFriendly and a[3] <= maxPrice and a[4] <= maxDistance]
        res = sorted(res, key=lambda a: a[1] + a[0] / 10 ** 6, reverse=True)
        return [a[0] for a in res]
