class Solution:
    def filterRestaurants(self, restaurants: List[List[int]], veganFriendly: int, maxPrice: int, maxDistance: int) -> List[int]:
        l 
= len(restaurants)
        cand = [(r, i) for i, r, v, p, d in restaurants if p <= maxPrice and d <= maxDistance and (v or not veganFriendly)]
        cand.sort(reverse=True)
        return [i for r, i in cand]
