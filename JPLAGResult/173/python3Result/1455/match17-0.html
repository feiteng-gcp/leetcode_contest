class Solution:
    def filterRestaurants(self, restaurants: List[List[int]], veganFriendly: int, maxPrice: int, maxDistance: int) -> List[int]:
        if veganFriendly == 1:
            restaurants = [it for it in restaurants if it[2]==1]
        restaurants = [it for it in restaurants if (it[3]<=maxPrice and it[4]<=maxDistance)]
        restaurants.sort(key=lambda x: (x[1], x[0]), reverse=True)
        return [it[0] for it in restaurants]
