class Solution:
    def filterRestaurants(self, restaurants: List[List[int]], veganFriendly: int, maxPrice: int, maxDistance: int) -> List[int]:
        #[idi, ratingi, veganFriendlyi, pricei, distancei]
        a = []
        for i, rating, vegan, price, distance in restaurants:
            if (vegan == veganFriendly or veganFriendly == 0) and price <= maxPrice and distance <= maxDistance:
                a.append([-rating, -i])
        return [-i for rating, i in sorted(a)]
