class Solution:
    def filterRestaurants(self, restaurants: List[List[int]], veganFriendly: int, maxPrice: int, maxDistance: int) -> List[int]:
        def fil(r):
            if veganFriendly == 1 and r[2] == 0:
                return False
            if r[3] > maxPrice or r[4] > maxDistance:
                return False
            return True
        filtered_res = filter(fil, restaurants)
        filtered_res = [(rate, id_) for id_, rate, _, _, _ in filtered_res]
        filtered_res.sort(reverse=True)
        return [id_ for rate, id_ in filtered_res]
