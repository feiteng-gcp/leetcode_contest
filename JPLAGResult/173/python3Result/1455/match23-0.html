class Solution:
    def filterRestaurants(self, restaurants: List[List[int]], veganFriendly: int, maxPrice: int, maxDistance: int) -> List[int]:
        cans = []
        for idi, ratingi, veganFriendlyi, pricei, distancei in restaurants:
            if veganFriendly and not veganFriendlyi: continue
            if pricei > maxPrice: continue
            if distancei > maxDistance: continue
            cans.append([ratingi, idi])
        cans.sort(reverse=True)
        return [idi for ratingi, idi in cans]
