class Solution:
    def filterRestaurants(self, restaurants: List[List[int]], veganFriendly: int, maxPrice: int, maxDistance: int) -> List[int]:
        res = []
        ls = []
        for r in restaurants:
            if veganFriendly and not r[2] or r[3] > maxPrice or r[4] > maxDistance:
                continue
            ls.append((-r[1], -r[0]))
        ls.sort()
        res = [-e[1] for e in ls]
        return res
