class Solution:
    def filterRestaurants(self, restaurants: List[List[int]], veganFriendly: int, maxPrice: int, maxDistance: int) -> List[int]:
        passed = [(res[1],res[0]) for res in restaurants if (res[2]==veganFriendly or veganFriendly==0) and res[3]<=maxPrice and res[4]<=maxDistance]
        passed.sort(key=lambda x:(-x[0], -x[1]))
        return [i[1] for i in passed]
