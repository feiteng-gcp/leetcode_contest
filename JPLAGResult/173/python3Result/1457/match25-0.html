import functools


class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        n = len(jobDifficulty)
        if n < d:
            return -1
        
        @functools.lru_cache(None)
        def finish(job, day):
            if day == 1:
                return max(jobDifficulty[job:])
            min_sum, curr_max = float("inf"), jobDifficulty[job]
            for i in range(job + 1, n - day + 2):
                min_sum = min(min_sum, curr_max + finish(i, day - 1))
                curr_max = max(curr_max, jobDifficulty[i])
            return min_sum
        
        return finish(0, d)
