import functools
class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        N = len(jobDifficulty)
        if N < d:
            return -1

        dp = [0] * (N + 1)
        for i in range(N):
            dp[i + 1] = dp[i] + jobDifficulty[i]


        @functools.lru_cache(None)
        def fun(remain_d, start):
            if remain_d == 1:
                return max(jobDifficulty[start:])
            if N - start == remain_d:
                return dp[N] - dp[start]
            cur_max = jobDifficulty[start]
            ans = cur_max + fun(remain_d - 1, start + 1)
            for i in range(start + 2, N - remain_d + 2):
                cur_max = max(cur_max, jobDifficulty[i - 1])
                ans = min(ans, cur_max + fun(remain_d - 1, i))
            return ans
        return fun(d, 0)
