class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        if len(jobDifficulty) < d:
            return -1
        n = len(jobDifficulty)
    
        dic = dict()
        dic[(1,1)] = jobDifficulty[0]
        for i in range(1,n):
            dic[(i+1, 1)] = max(dic[(i, 1)],  jobDifficulty[i])
            for j in range(2, min(d+1, i+2)):
                MAX = jobDifficulty[i]
                dic[(i+1, j)] = dic[(i, j-1)]  + MAX
                
for k in range(i-1, j-2, -1):
                    MAX = max(jobDifficulty[k], MAX)
                    dic[(i+1, j)] = min(dic[(k, j-1)]  + MAX, dic[(i+1,j)])
        return dic[(n,d)]
