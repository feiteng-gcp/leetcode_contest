class Solution:
    def minDifficulty(self, jd: List[int], d: int) -> int:
        n = len(jd)
        if d > n: return -1
        MAXD = 10**8
        dp = [[MAXD] * (n+1) for _ in range(d+1)]
        dp[0][0] = 0
        for i in range(d):
            for j in range(n+1):
                if dp[i][j] >= MAXD: continue
                cur = 0
                for k in range(j + 1, n+1):
                    cur = max(jd[k-1], cur)
                    dp[i+1][k] = min(dp[i+1][k], dp[i][j] + cur)
        return dp[d][n]
