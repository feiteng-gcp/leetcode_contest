from functools import lru_cache

class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        if len(jobDifficulty) < d:
            return -1
        
        @lru_cache(None)
        def dp(i, j):
            if i == j == 0:
                return 0
            if j == 1:
                return max(jobDifficulty[k] for k in range(i))
            diff = float("-inf")
            result 
= float("inf")
            for k in range(i - 1, j - 2, -1):
                diff = max(diff, jobDifficulty[k])
                result = min(result, diff + dp(k, j - 1))
            return result
        
        return dp(len(jobDifficulty), d)
