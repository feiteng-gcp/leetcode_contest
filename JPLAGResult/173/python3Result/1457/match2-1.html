class Solution:
    def minDifficulty(self, j: List[int], d: int) -> int:
        if len(j) < d:
            return -1
        
        dp = dict()
        def helper(start, end, d):
            if (start, end, d) in dp:
                return dp[(start, end, d)]
            
            if d == 1:
                dp[(start, end, 1)] = max(j[start:end])
                return dp[(start, end, 1)]
            else:
                cur = float('inf')
                
                smax = j[start]
                for x in range(start, end - d + 1):
                    if j[x] > smax:
                        smax = j[x]
                    cur = min(cur, smax + helper(x+1, end, d-1))
                
                dp[(start, end, d)] = cur
                return dp[(start, end, d)]
        
        helper(0, len(j), d)
        # print(dp)
        return dp[(0, len(j), d)]
