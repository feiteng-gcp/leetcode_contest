class Solution:
    def minDifficulty(self, jobDifficulty: 'List[int]', d: int) -> int:
        jd=jobDifficulty
        l=len(jd)
        inf=float('inf')
        if l<d:return -1

        dic={}
        for i in range(l):
            dic[(i,i)]=jd[i]
        for i in range(l):
            
for j in range(i+1,l):
                dic[(i,j)]=max(dic[(i,j-1)],jd[j])
        dp=[[inf]*l for i in range(d)]
        for j in range(l):
            dp[0][j]=dic[(0,j)]
        for i in range
(1,d):
            for j in range(i,l):
                for k in range(i,j+1):
                    dp[i][j]=min(dp[i][j],dp[i-1][k-1]+dic[k,j])
        return dp[-1][-1]
