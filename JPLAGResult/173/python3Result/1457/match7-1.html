class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        len_job = len(jobDifficulty)
        if len_job < d:
            return -1
        dfy = [[float('inf')]*d for _ in range(len_job)]
        dfy[0][0] = jobDifficulty[0]
        for i in range(1, len_job):
            dfy[i][0] = max(jobDifficulty[i], dfy[i-1][0])
            for j in range(1, min(d, i+1)):
                last = jobDifficulty[i]
                record = last + dfy[i-1][j-1]
                for k in range(i-1, j-1, -1):
                    last = max(last, jobDifficulty[k])
                    record = min(record, last + dfy[k-1][j-1])
                dfy[i][j] =record
        return dfy[-1][-1]
