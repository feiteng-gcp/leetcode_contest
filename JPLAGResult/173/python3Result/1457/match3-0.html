from functools import lru_cache
class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        n = len(jobDifficulty)
        if n < d:
            return -1
        
        di = {}
        for i in range(n):
            for j in range(i+1, n+1):
                di[(i, j)] = max(jobDifficulty[i:j])
        
        @lru_cache(None)
        def dp(i, k):
            if k == 1:
                return max(jobDifficulty[:i])
            if i == k:
                return sum(jobDifficulty[:i])
            return min(dp(j, k-1) + di[(j, i)] for j in range(k-1, i))
        return dp(n, d)
