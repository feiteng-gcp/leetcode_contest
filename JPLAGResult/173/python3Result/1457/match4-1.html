from math import inf
class Solution:
    def minDifficulty(self, jobs: List[int], d: int) -> int:
        m = len(jobs)
        if m < d:
            return -1
        dp 
= [[inf for _ in range(m)] for _ in range(d)]
        dp[0][0] = jobs[0]
        for i in range(1, m):
            dp[0][i] = max(dp[0][i - 1], jobs[i])
        for i in range(1, d):
            for j in range(m):
                maximum = jobs[j]
                for k in range(j, i - 1, -1):
                    maximum 
= max(maximum, jobs[k])
                    dp[i][j] = min(dp[i][j], dp[i - 1][k - 1] + maximum)
        return dp[-1][-1]
