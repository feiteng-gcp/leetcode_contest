import functools

class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        if len(jobDifficulty) < d:
            return -1
        
        @functools.lru_cache(None)
        def dfs(i, left):
            if left == 0:
                return max(jobDifficulty[i:])
            
            m = float('-inf')
            ans = float('inf')
            for j in range(i, len(jobDifficulty) - left):
                m = max(m, jobDifficulty[j])
                ans = min(ans, m + dfs(j + 1, left - 1))
            
            return ans
        
        return dfs(0, d - 1)
