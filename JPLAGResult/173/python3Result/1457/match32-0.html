from collections import defaultdict
class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        n = len(jobDifficulty)
        if n < d:
            return -1
        if n == d:
            return sum(jobDifficulty)
        # 动态规划 dp[i,j] 表示job i在第j天完成的最小难度, i一定要大于等于j

        dp = defaultdict(int)
        dp[0, 0] = jobDifficulty[0]
        for i in range(1, n):
            dp[i, 0] = max(jobDifficulty[i], dp[i - 1, 0])
        for i in range(n):
            for j in range(i + 1, d):
                dp[i, j] = float('inf')
        for i in range(1, n):
            for j in range(1, min(i + 1,d)):
                mn = float('inf')
                mxDiffInJ = jobDifficulty[i]
                for k in range(i - 1, -1, -1):
                    mn = min(mn, dp[k, j - 1] + mxDiffInJ)
                    mxDiffInJ = max(mxDiffInJ, jobDifficulty[k])
                dp[i, j] = mn
        return dp[n - 1, d - 1]
