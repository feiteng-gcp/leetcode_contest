from pprint import pprint

class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        n = len(jobDifficulty)
        
        dp = [[10**9 for _ in range(d+1)] for _ in range(n)]
        dp[n-1][1] = jobDifficulty[n-1]
        for i in range(n-2, -1, -1):
            dp[i][1] = max(dp[i+1][1], jobDifficulty[i])
         
        #pprint(dp)
        for d in range(2, d+1):
            for i in range(n-1, -1, -1):
                tasks = n - i
                if tasks < d:
                    continue
                difficulty = jobDifficulty[i]
                for j in range(i, n):
                    done_tasks = j-i+1
                    if (tasks - done_tasks) < d-1:
                        break
                    difficulty = max(difficulty, jobDifficulty[j])
                    dp[i][d] = min(dp[i][d], difficulty + dp[j+1][d-1])
                    
        #pprint(dp)
        
        return dp[0][d] if dp[0][d] < 10**9 else -1
