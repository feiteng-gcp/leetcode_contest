from functools import lru_cache

class Solution:
    
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        @lru_cache(None)
        def best(i, d):
            if i >= len(jobDifficulty):
                return float('inf')
            if d == 1:
                return max(jobDifficulty[i:])
            result = float('inf')
            today = 0
            
for j in range(i+1, len(jobDifficulty)+1):
                today = max(today, jobDifficulty[j-1])
                result = min(result, today + best(j, d-1))
            return result
        result = best(0, d)
        return result if result < float('inf') else -1
