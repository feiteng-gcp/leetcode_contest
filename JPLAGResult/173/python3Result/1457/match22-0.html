from functools import lru_cache
class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        A, n = jobDifficulty, len(jobDifficulty)
        @lru_cache(None)
        def helper(i, r):
            if r == 0: return math.inf if i < n else 0
            today, res = 0, math.inf
            for j in range(i, n - r + 1):
                today = max(today, A[j])
                res = min(res, today + helper(j + 1, r - 1))
            return res
        ans = helper(0, d)
        return ans if ans < math.inf else -1
