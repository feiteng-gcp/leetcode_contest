class Solution:
    def minDifficulty(self, jd: List[int], d: int) -> int:
        n = len(jd)
        if d > n:
            return -1
        m = [[0] * n for _ in range(n)]
        for i in range(n - 1, -1, -1):
            m[i][i] = jd[i]
            for j in range(i + 1, n):
                m[i][j] = max(jd[i], m[i + 1][j])
        dp = [[0] * n for _ in range(d + 1)]        
        dp[1] = [m[0][i] for i in range(n)]
        for i in range(2, d + 1):
            for j in range(i - 1, n):
                dp[i][j] = min(dp[i - 1][k] + m[k + 1][j] for k in range(i - 2, j))
        return dp[d][-1]
