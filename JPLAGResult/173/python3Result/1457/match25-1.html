import functools
class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        n=len(jobDifficulty)
        if n<d:
            return -1
        @functools.lru_cache(None)
        def dp(idx,day):
            if idx+1==day:
                return sum(jobDifficulty[:idx+1])
            if day==1:
                return max(jobDifficulty[:idx+1])

            curMax=-math.inf
            res=math.inf

                curMax=max(curMax,jobDifficulty[idx-l+1])
                res=min(res,dp(idx-l,day-1)+curMax)
            return res
        return dp(n-1,d)
