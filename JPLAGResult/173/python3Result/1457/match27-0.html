class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        if len(jobDifficulty) < d:
            return -1
        
        INF = float('inf')
        N = len(jobDifficulty)
        
        dp = [jobDifficulty[0]]
        for i in range(1, N): dp.append(max(dp[i-1], jobDifficulty[i]))
        
        for di in range(2, d + 1):
            ndp = [INF] * N
            for i in range(di - 1, N):
                m = 0
                for j in range(i, N):
                    m = max(m, jobDifficulty[j])
                    ndp[j] = min(ndp[j], dp[i-1] + m)
            dp = ndp
            # print(dp)
        
        return dp[-1]
