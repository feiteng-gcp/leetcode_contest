class Solution:
    def minDifficulty(self, jobDifficulty: List[int], d: int) -> int:
        
        n = len(jobDifficulty)
        if n < d: return -1
        
        cache = dict()
        def dfs(pos, k):
            if k == d -1:
                return max(jobDifficulty[pos:])
            
            if (pos, k) in cache:
                
return cache[(pos, k)]
            cur = jobDifficulty[pos]
            res = float('inf')
            for i in range(pos, n-1, 1):
                cur = max(cur, jobDifficulty[i])
                res = min(res,dfs(i+1, k+1) + cur)
            cache[(pos, k)] = res
            return res
        
        return dfs(0, 0)
