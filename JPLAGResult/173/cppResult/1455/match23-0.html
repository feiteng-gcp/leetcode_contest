class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& restaurants, int veganFriendly, int maxPrice, int maxDistance) 
    {
        vector<vector<int>> ansT;
        vector<int> arrT(2,0);
        for (int i = 0; i < restaurants.size(); i++)
        {
            if (restaurants[i][3] <= maxPrice && restaurants[i][4] <= maxDistance)
            {
                if (veganFriendly == 0 || restaurants[i][2] == 1)
                {
                    arrT[0] = restaurants[i][1];
                    arrT[1] = restaurants[i][0];
                    ansT.push_back(arrT);
                }
            }            
        }
        sort(ansT.begin(), ansT.end());
        reverse(ansT.begin(), ansT.end());
        
        vector<int> ans;
        for (int i = 0; i < ansT.size(); i++)
        {
            ans.push_back(ansT[i][1]);
        }
        return ans;
    }
};
