bool cmp(const vector<int> &arg0, const vector<int> &arg1) {
    if (arg0[1] != arg1[1]) return arg0[1] > arg1[1];
    return arg0[0] > arg1[0];
}

class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        vector<vector<int> > a;
        for (auto x: restaurants) {
            if (veganFriendly && (x[2] == 0)) continue;
            if (x[3] > maxPrice) continue;
            if (x[4] > maxDistance) continue;
            a.push_back(x);
        }
        sort(a.begin(), a.end(), cmp);
        
        vector<int> ans;
        for (auto x: a) ans.push_back(x[0]);
        return ans;
    }
};
