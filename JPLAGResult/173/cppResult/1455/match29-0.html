class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>> &restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        int n = restaurants.size();
        vector<pair<int, int>> cands;
        for (int i = 0; i < n; i++) {
            bool good = true;
            if (restaurants[i][2] < veganFriendly) good = false;
            if (restaurants[i][3] > maxPrice) good = false;
            if (restaurants[i][4] > maxDistance) good = false;
            if (good) {
                cands.push_back(make_pair(restaurants[i][1], restaurants[i][0]));
            }
        }
        sort(cands.rbegin(), cands.rend());
        vector<int> ans;
        for(int i = 0;i<cands.size();i++) ans.push_back(cands[i].second);
        return ans;
    }
};
