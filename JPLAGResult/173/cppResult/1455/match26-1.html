class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        map<int, set<int>> mmap;
        
        for (auto &r: restaurants) {
            if (veganFriendly && !r[2])
                continue;
            if (r[3] <= maxPrice && r[4] <= maxDistance)
                mmap[r[1]].insert(r[0]);
        }
        
        vector<int> ans;
        for (auto it=mmap.rbegin(); it!=mmap.rend(); it++) {
            for (auto iter=it->second.rbegin(); iter!=it->second.rend(); iter++)
                ans.push_back(*iter);
        }
        return ans;
    }
};
