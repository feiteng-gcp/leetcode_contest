class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& r, int vf, int maxPrice, int maxDistance) {
        vector<pair<int,int>> ans;
        int n=r.size(),m=r[0].size();
        for(int i=0;i<n;i++)
        {
            if((vf==0||r[i][2]==vf)&&r[i][3]<=maxPrice&&r[i][4]<=maxDistance)
            {
                ans.push_back({r[i][1],r[i][0]});
            }
                
        }
        sort(ans.begin(),ans.end());
        vector<int> ans2;
        for(int i=0;i<ans.size();i++)
            ans2.push_back(ans[i].second);
        reverse(ans2.begin(),ans2.end());
        return ans2;
    }
};
