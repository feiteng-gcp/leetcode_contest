class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& rests, int vF, int mP, int mD) {
        vector<vector<int>> re;
        int n = rests.size();
        for(int i=0;i<n;i++){
            if(vF && !rests[i][2]) continue;
            if(rests[i][3]>mP) continue;
            if(rests[i][4]>mD) continue;
            re.push_back({-rests[i][1],rests[i][0]});
        }
        auto comp=[](vector<int> &a,vector<int>&b){
            if(a[0] !=b[0]) return a[0] < b[0];
            return a[1]>b[1];
        };
        sort(re.begin(),re.end(),comp);
        vector<int> ans;
        for(auto item:re){
            int idx = item[1];
            ans.push_back(idx);
        }
        return ans;
    }
};
