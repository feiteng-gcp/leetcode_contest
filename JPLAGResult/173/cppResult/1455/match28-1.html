class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& r, int v, int maxPrice, int maxDistance) {
        vector<pair<pair<int, int>, int>> data;
        for (int i=0; i<r.size(); i++) {
            if (v == 1 && r[i][2] == 0) continue;
            if (r[i][3] > maxPrice) continue;
            if (r[i][4] > maxDistance) continue;
            

            data.push_back({{r[i][1], r[i][0]}, i+1});
        }
        
        sort(data.begin(), data.end());
        
        vector<int> ret;
        for (int i=data.size()-1; i>=0; i--)
            ret.push_back(data[i].first.second);
        return ret;
    }
};
