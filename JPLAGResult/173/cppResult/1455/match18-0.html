class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        vector<pair<int,int>> v;
        for (auto e : restaurants)
        {
            int id = e[0], rate = e[1], vf = e[2], pr = e[3], dis = e[4];
            if ((vf == veganFriendly || veganFriendly == 0) && pr <= maxPrice && dis <= maxDistance)
                v.push_back({rate, id});
        }
        sort(v.begin(), v.end(), greater<pair<int,int>>());
        vector<int> ret;
        for (auto [x, y] : v)
            ret.push_back(y);
        return ret;
    }
};
