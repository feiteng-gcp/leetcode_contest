class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        vector<pair<int, int> > all;
        for(auto v : restaurants){
            if(maxPrice < v[3] || maxDistance < v[4])continue;
            if(veganFriendly == 0 || (veganFriendly == 1 && v[2] == veganFriendly))
                all.emplace_back(-v[1], -v[0]);
        }
        sort(all.begin(), all.end());
        vector<int> ret;
        for(auto p : all)ret.emplace_back(-p.second);
        return ret;
    }
};
