static bool f(const vector<int>&v1, const vector<int> &v2)
{
    if(v1[1] == v2[1])
    {
        return v1[0] > v2[0];
    }

    return v1[1] > v2[1];
}
class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& res, int veganFriendly, int maxPrice, int maxDistance) {
        vector<vector<int>> v;
        
        for(int i = 0; i < res.size(); i++)
        {
            if(veganFriendly)
                if(res[i][2] == 0)

                    continue;
            
            if(res[i][3] > maxPrice)
                continue;
            if(res[i][4] > maxDistance)
                continue;
            v.push_back(res[i]);
        }
        
        sort(v.begin(), v.end(), f);
        vector<int> ans;
        for(int i = 0; i < v.size(); i++)
            ans.push_back(v[i][0]);
        return ans;
    }
};
