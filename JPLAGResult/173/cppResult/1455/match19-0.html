class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& restaurants, int vvv, int ppp, int ddd) {
        vector< pair<int, int> > tmp;
        for (auto v : restaurants) {
            if ((!vvv || v[2]) && v[3] <= ppp && v[4] <= ddd) {
                tmp.push_back(make_pair(v[1], v[0]));
            }
        }
        sort(tmp.begin(), tmp.end(), greater< pair<int, int> >());
        vector<int>res;
        for (int i = 0; i < tmp.size(); i++) res.push_back(tmp[i].second);
        return res;
    }
};
