class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& r, int vf, int mp, int md) {
        // restaurants[i] = [0 - idi, 1 - ratingi, 2 - veganFriendlyi, 3-  pricei, 4- distancei]
        vector<pair<int, int>> f; 
        for (int i = 0; i < r.size(); i++) {
            if (vf && !r[i][2]) continue;
            if (r[i][3] > mp) continue;
            if (r[i][4] > md) continue;
            
            f.push_back({-r[i][1], -r[i][0]});
        }
        
        sort(f.begin(), f.end());
        
        vector<int> result;
        for (auto v : f) {
            result.push_back(-v.second);
        }
        
        return result;
    }
};
