class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& a, int veganFriendly, int maxPrice, int maxDistance) {
        vector<pair<int, int>> A;
        for (auto& v : a) {
            if (v[2] >= veganFriendly && v[3] <= maxPrice && v[4] <= maxDistance) {
                A.push_back(make_pair(v[1], v[0]));
            }
        }
        sort(A.begin(), A.end(), greater<pair<int, int>>());
        vector<int> ret;
        for (auto& p : A) ret.push_back(p.second);
        return ret;
    }
};
