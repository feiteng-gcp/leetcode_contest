class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& restaurants, int vf, int mp, int md) {
        vector<vector<int>> a;
        for(auto &v : restaurants) {
            if((vf ? v[2] == vf : true) && v[3] <= mp && v[4] <= md) a.push_back(v);
        }
        sort(a.begin(),a.end(),[](vector<int> &a,vector<int> &b) {
            if(a[1] == b[1]) return a[0] > b[0];
            return a[1] > b[1];
        });
        vector<int> ans;
        for(auto &v : a) ans.push_back(v[0]);
        return ans;
    }
};
