class Solution {
public:
    static bool func(const vector<int> &a, const vector<int> &b){
        if((a[1] > b[1]) || (a[1] == b[1] && a[0] > b[0])){
            return true;
        }
        return false;
    }
    
    vector<int> filterRestaurants(vector<vector<int>>& r, int v, int p, int d) {
        sort(r.begin(), r.end(), func);

        vector<int> res;
        for(int i = 0; i < r.size(); i++){
            if(((v == 1 && r[i][2] == 1) || (v == 0)) && (r[i][3] <= p) && (r[i][4] <= d)){
                res.push_back(r[i][0]);
            }
        }
        return res;
    }
};
