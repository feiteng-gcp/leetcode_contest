class Solution {
public:
    vector<int> filterRestaurants(vector<vector<int>>& restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        vector<int> ret;
        unordered_map<int, int> rs;
        for (auto &r: restaurants) {
            if (veganFriendly && !r[2]) continue;
            if (maxPrice < r[3]) continue;
            if (maxDistance < r[4]) continue;
            ret.push_back(r[0]);
            rs[r[0]] = r[1];
        }
        
        sort(ret.begin(), ret.end(), [&](int id1, int id2) {
            int rate1 = rs[id1], rate2 = rs[id2];
            if (rate1 > rate2) return true;
            if (rate1 < rate2) return false;
            return id1 > id2;
        });
        
        return ret;
    }
};
