class Solution {
public:
    int minDifficulty(vector<int>& jobDifficulty, int d) {
        int n = jobDifficulty.size();
        if (n < d) return -1;
        
        vector< vector<int> > dp(d + 1, vector<int>(n, INT_MAX));
        
        for (int j = 1; j <= d; ++j)
        {
            if (j == 1)
            {
                int maxVal = 0;
                for (int i = 0; i < n; ++i)
                {
                    maxVal   = max(maxVal, jobDifficulty[i]);
                    dp[j][i] = maxVal;
                }
            }
            else
            {
                for (int i = j - 1; i < n; ++i)
                {
                    int maxVal = jobDifficulty[i];
                    for (int t = i; t >= j - 1; --t)
                    {
                        maxVal = max(maxVal, jobDifficulty[t]);
                        //cout << j << "," << i << "," << t << endl;
                        dp[j][i] = min(dp[j][i], dp[j - 1][t - 1] + maxVal);
                    }
                }
            }
        }
        
        if (dp[d][ n - 1] == INT_MAX) return -1;
        
        return dp[d][n - 1];
    }
};
