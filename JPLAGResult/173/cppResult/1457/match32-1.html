class Solution {
 public:
  int minDifficulty(vector<int>& jobs, int d) {
    int n = jobs.size();
    if (n < d) return -1;
    vector<vector<int>> dp(n + 1, vector<int> (d + 1, 0x3f3f3f3f));
    vector<vector<int>> max_val(n, vector<int> (n));

    for (int i = 0; i < n; i++) {
      int cur_max_val = jobs[i];
      for (int j = i; j < n; j++) {
        cur_max_val = max(cur_max_val, jobs[j]);
        max_val[i][j] = cur_max_val;
      }
    }
    dp[0][0] = 0;
    for (int i = 1; i <= n; i++) {

      for (int j = 1; j <= d && j <= i; j++) {
        for (int k = 0; k < i; k++) {
          if (dp[k][j - 1] != 0x3f3f3f3f) {
            dp[i][j] = min(dp[i][j], dp[k][j - 1] + max_val[k][i - 1]);
          }
        }
      }
    }
    return dp[n][d];
  }
};
