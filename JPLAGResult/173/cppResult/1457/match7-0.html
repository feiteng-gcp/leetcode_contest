class Solution {
public:
    int presum[300+5];
    int dp[305][305][12];
    

    int dfs(vector<int>& nums, int lf, int rg, int d){
        if(dp[lf][rg][d] != -1) return dp[lf][rg][d];
        if(d == 1){
            int maxn = nums[lf];
            for(int i=lf; i<=rg; i++) maxn = max(maxn, nums[i]);
            return maxn;
        }
        
        int cur = 0x3f3f3f3f;
        for(int i=lf; i<rg; i++){
            for(int k=1; k<=1; k++){
                int ll = dfs(nums, lf, i, k);
                int rr = dfs(nums, i+1, rg, d-k);
                cur = min(ll+rr, cur);
            }
           
        }
        return dp[lf][rg][d] = cur;
    }
    int minDifficulty(vector<int>& jobDifficulty, int d) {
        // presum[0] = jobDifficulty[0];
        // for(int i=1; i<jobDifficulty.size(); i++) presum[i] = presum[i-1] + jobDifficulty[i];
        if(d>jobDifficulty.size()) return -1;
        memset(dp, -1, sizeof(dp));
        
        return dfs(jobDifficulty, 0, jobDifficulty.size()-1, d);
    }
};
