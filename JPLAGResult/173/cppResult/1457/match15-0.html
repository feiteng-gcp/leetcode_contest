class Solution {
public:
    int minDifficulty(vector<int>& jobDifficulty, int d) {
        int n = jobDifficulty.size();
        if(n < d) return -1;//at least 1 job each day
        if(n == d) return accumulate(jobDifficulty.begin(), jobDifficulty.end(), 0);
        vector<vector<int>> dp(n + 1, vector<int>(d+1, -1));
        dp[0][0] = 0;
        for(int i = 1; i <= n; ++i){
            dp[i][1] = max(dp[i-1][1], jobDifficulty[i-1]);
        }
        for(int i = 1; i <= n; ++i){
            for(int j = 2; j <= d; ++j){
                int temp = INT_MAX;
                int curMax = jobDifficulty[i-1];
                for(int k = i - 1; k >= 0; --k){
                    curMax = max(curMax, jobDifficulty[k]);
                    if(dp[k][j-1] > 0) temp = min(temp, dp[k][j-1] + curMax);
                }
                if(temp < INT_MAX) dp[i][j] = temp;
            }
        }
        return dp[n][d];
    }
};
