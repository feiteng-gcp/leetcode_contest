class Solution {
public:
    int cal(vector<int> &jd, vector<vector<int>> &dp, int index, int d) {
        
        if (dp[index][d] != INT_MAX) return dp[index][d];
        int ans = INT_MAX, md = -1;
        if (d == 1) {
            for (int i = index; i < jd.size(); ++i) md = max(md, jd[i]);
            ans = md;
        } else {
            for (int i = index; i < jd.size() - d + 1; ++i) {
                md = max(md, jd[i]);
                ans = min(ans, md + cal(jd, dp, i + 1, d - 1));
            }
        }
        //printf("%d %d %d\n", index, d, ans);
        return (dp[index][d] = ans);
    }
    
    int minDifficulty(vector<int>& jobDifficulty, int d) {
        int numJob = jobDifficulty.size();
        if (d > numJob) return -1;
        vector<vector<int>> dp(numJob, vector<int>(d + 1, INT_MAX));
        
        return cal(jobDifficulty, dp, 0, d);
    }
};
