class Solution {
public:
    int minDifficulty(vector<int>& diff, int d) {
        int n = diff.size();
        if (n < d)return -1;
        vector<vector<int>>dp(n + 1, vector<int>(d + 1, 1e7));
        vector<vector<int>>maxx(n+1, vector<int>(n+1, 0));
        for (int i = 1; i <= n; i++) {
            int p = diff[i-1];
            for (int j = i; j <= n; j++) {
                p = max(p, diff[j-1]);
                maxx[i][j] = p;
            }
        }
        dp[0][0] = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= d; j++) {
                for (int k = 0; k < i; k++) {
                    dp[i][j] = min(dp[i][j], dp[k][j - 1] + maxx[k + 1][i]);
                }
            }
        }
        return dp[n][d];
    }
};
