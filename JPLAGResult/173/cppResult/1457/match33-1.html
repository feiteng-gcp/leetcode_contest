class Solution {
public:
    int minDifficulty(vector<int>& jobDifficulty, int day) {
        int size = jobDifficulty.size();
        vector<vector<int>> max_diff(size, vector<int>(size, 0));
        
        for(int i = 0; i < size; i++) {
            for(int j = i; j < size; j++) {
                int m = 0;
                for(int k = i; k <= j; k++) {
                    m = max(m, jobDifficulty[k]);
                }
                max_diff[i][j] = m;
            }
        }
        
        vector<vector<int>> dp(size, vector<int>(day + 1, 100000000));
        
        for(int i = 0; i < size; i++) {
            dp[i][0] = 0;
            dp[i][1] = max_diff[0][i];
        }
        
        for(int i = 0; i < size; i++) {

            for(int d = 2; d <= day; d++) {
                for(int j = 0; j < i; j++) {
                    dp[i][d] = min(dp[i][d], dp[j][d - 1] + max_diff[j + 1][i]);
                    // cout << i << " " << d << " " << dp[i][d] << endl;
                }
            }
        }
        
        /* cout << " " << endl;
        
        for(int i = 0; i < size; i++) {
            for(int d = 1; d <= day; d++) {
                cout << i << " " << d << " " << dp[i][d] << endl;
            }
        } */
        
        return dp[size - 1][day] < 100000000 ? dp[size - 1][day] : -1;
    }
};
