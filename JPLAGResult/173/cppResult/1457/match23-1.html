class Solution {
public:
  int minDifficulty(vector<int>& a, int d) {
    int n = a.size();
    if (d > n) return -1;
    vector<vector<int>> c(n + 1, vector<int>(n + 1));

    for (int i = 0; i < n; ++i) {
      for (int j = i; j < n; ++j) {
        c[i][j + 1] = max(c[i][j], a[j]);
      }
    }
    const int inf = 1000010000;
    vector<vector<int>> b(d + 1, vector<int>(n + 1, inf));

    b[0][0] = 0;
    for (int i = 0; i < d; ++i) {
      for (int j = 0; j < n; ++j) {
        for (int k = j + 1; k <= n; ++k) {
          b[i + 1][k] = min(b[i + 1][k], b[i][j] + c[j][k]);
        }
      }
    }
    return b[d][n];
  }
};
