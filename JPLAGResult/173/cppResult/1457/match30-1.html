class Solution {
public:
    int minDifficulty(vector<int>& jd, int d) {
        int n = jd.size();
        if (d > n) return -1;
        vector<vector<int>> dp(d + 1, vector<int>(n, INT_MAX));
        for (int i = 1; i <= d; i++) {
            if (i == 1) {
                int mx0 = jd[0];
                for (int j = 0; j < n; j++) {
                    mx0 = max(mx0, jd[j]);
                    dp[1][j] = mx0;
                }
                continue;
            }

            for (int j = i - 1; j < n; j++) {
                int mx0 = jd[j];
                for (int k = j; k >= i - 1; k--) {
                    mx0 = max(mx0, jd[k]);
                    dp[i][j] = min(dp[i][j], dp[i - 1][k - 1] + mx0);
                }
            }
        }
        return dp.back().back();
    }
};
