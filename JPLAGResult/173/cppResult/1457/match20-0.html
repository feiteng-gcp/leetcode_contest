class Solution {
public:
    int minDifficulty(vector<int>& jobDifficulty, int d) {
        int n = jobDifficulty.size();
        if(n<d){
            return -1;
        }
        vector<vector<int>> f(n + 1, vector<int>(d+1, INT_MAX));
        for (int i = 0; i <= n; i++){
            f[i][0] = 0;
        } 
        int maxn = jobDifficulty[0];
        for(int i=0; i<n; ++i){
            maxn = max(maxn, jobDifficulty[i]);
            f[i+1][1] = maxn;
        }
        for (int i = 1; i <= n; i++) {
            for (int j = 2; j <= d; j++) {
                int maxn = jobDifficulty[i-1];
                for (int k = i-1; k >=j-1; k--) {
                    maxn = max(maxn, jobDifficulty[k]);
                    f[i][j] = min(f[i][j], f[k][j - 1]+maxn);
                }
            }
        }
        return f[n][d];
    }
};
