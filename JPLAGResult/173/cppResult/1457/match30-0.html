class Solution {
public:
    int minDifficulty(vector<int>& jobs, int d) {
        int n = jobs.size();
        if(n<d) return -1;
        vector<vector<int>> dp(n+1,vector<int>(d+1,1e9));
        
        dp[0][0] = 0;
        for(int i=0;i<n;i++){
            for(int j=1;j<=d;j++){
                int m = 0;
                for(int k=i;k>=0;k--){
                    m = max(m, jobs[k]);
                    dp[i+1][j] = min(dp[i+1][j], dp[k][j-1]+m);
                }
            }
        }
        return dp[n][d];
    }
};
