class Solution {
public:
    int minDifficulty(vector<int>& in, int d) {
        int len = in.size();
        if(len < d) return -1;
        vector<vector<int>> dp(len+1, vector<int>(d+1, 301000));
        dp[0][0] = 0;
        for(int i=1;i<=len;i++)
        {
            for(int j=1;j<=d;j++)
            {
                if(j>i) {
                    continue;
                }

                int maxV = -1;
                for(int k=1;k<=i-j+1;k++)
                {
                    maxV = max(maxV, in[i-k]);
                    dp[i][j] = min(dp[i-k][j-1]+ maxV, dp[i][j]);
                }
            }
        }
        return dp[len][d];
    }
};
