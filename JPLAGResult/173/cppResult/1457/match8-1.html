class Solution {
public:
    int minDifficulty(vector<int>& diff, int D) {
        int n = diff.size();
        if (D > n)    return -1;
        
        // Get max(i, j)
        vector<vector<int>> m(n, vector<int>(n));
        for (int i = 0; i < n; i++) {
            int M = diff[i];
            for (int j = i; j < n; j++) {
                M = max(M, diff[j]);
                m[i][j] = M;
            }
        }
        
        // Define dp
        auto dp = vector<vector<int>>(D + 1, vector<int>(n, INT_MAX / 2));
        for (int i = 0; i < n; i++) {
            dp[1][i] = m[0][i];
        }

        // Build dp
        for (int d = 2; d <= D; d++) {
            for (int i = d - 1; i < n - (D - d); i++) {
                for (int j = 0; j < i; j++) {
                    dp[d][i] = min(dp[d][i], dp[d - 1][j] + m[j + 1][i]);
                }
            }
        }
        
        return dp[D][n - 1];
    }
};
