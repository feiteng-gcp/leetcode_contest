class Solution {
public:
	int minDifficulty(vector<int>& jobDifficulty, int d) {
		int N = jobDifficulty.size();
		vector<vector<int>>dp(N + 1, vector<int>(d + 1, INT_MAX));
		for (int i = 1; i <= N; ++i) {
			dp[i][1] = *max_element(jobDifficulty.begin(), jobDifficulty.begin() + i);
			for (int j = 2; j <= d; ++j) {
				for (int z = j - 1; z < i; ++z) {
					dp[i][j] = min(dp[i][j], dp[z][j - 1] + *max_element(jobDifficulty.begin() + z, jobDifficulty.begin() + i));
				}
			}
		}
		return dp[N][d] == INT_MAX ? -1 : dp[N][d];
	}
};
