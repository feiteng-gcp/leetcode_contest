class Solution {
public:
    int minDifficulty(vector<int>& jD, int d) {
        int n = jD.size();
        if(d>n) return -1;
        vector<vector<int>> dp(d+1,vector<int>(n+1,1e9+7));
        dp[0][0]=0;
        for(int i=1;i<=d;i++){
            for(int j=1;j<=n;j++){

                int tmax = 0;
                for(int k=(j-1);k>=(i-1);k--){
                    tmax=max(tmax,jD[k]);
                    dp[i][j] = min(dp[i][j],dp[i-1][k]+tmax);
                }
                // cout<<i<<" "<<j<<" "<<dp[i][j]<<endl;
            }
        }
        return dp[d][n];
    }
};
