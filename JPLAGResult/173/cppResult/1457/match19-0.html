class Solution {
public:
    int minDifficulty(vector<int>& jd, int D) {
        const int INF = 1e8;
        int n = jd.size();
        if (n < D) return -1;
        vector<vector<int>> d(D + 1, vector<int>(n + 1, INF));
        d[0][0] = 0;
        for (int i = 0; i < D; ++i) {
            for (int j = 0; j < n; ++j) {
                if (d[i][j] == INF) continue;
                int mx = 0;
                for (int k = 1; j + k <= n; ++k) {
                    mx = max(mx, jd[j + k - 1]);
                    d[i + 1][j + k] = min(d[i + 1][j + k], d[i][j] + mx);
                }
            }
        }
        return d[D][n];
    }
};
