class Solution {
public:
    int minDifficulty(vector<int>& jobDifficulty, int d) {
        int n = jobDifficulty.size();
        if(n < d)   return -1;
        int ma[n][n];
        memset(ma, 0, sizeof(ma));
        for(int i = 0; i < n; i++){
            for(int j = i; j < n; j++){
                for(int k = i; k <= j; k++){
                    ma[i][j] = max(ma[i][j], jobDifficulty[k]);
                }
            }
        }
        int dp[d+1][n+1];
        memset(dp, 0x3f, sizeof(dp));
        dp[0][0] = 0;
        for(int i = 1; i <= d; i++){
            for(int j = i; j <= n; j++){
                for(int k = j; k >= 1; k--){
                    dp[i][j] = min(dp[i][j], dp[i-1][k-1]+ma[k-1][j-1]);
                }
            }
        }
        return dp[d][n];
    }
};
