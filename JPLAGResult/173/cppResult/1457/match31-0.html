class Solution {
public:
    int minDifficulty(vector<int>& jobDifficulty, int d) {
        int n=jobDifficulty.size();
        if(n<d) return -1;
        vector<vector<int> > dp(n,vector<int>(d+1,10000000));
        for(int i=n-1;i>=0;i--)
        {
            if(i==n-1) dp[i][1]=jobDifficulty[i];
            else dp[i][1]=max(jobDifficulty[i],dp[i+1][1]);
        }
        for(int j=2;j<=d;j++)
        {
            for(int i=n-j;i>=0;i--)
            {
                if(n-i==j)
                {
                    dp[i][j]=0;
                    for(int k=i;k<n;k++)
                    {
                        dp[i][j]+=jobDifficulty[k];
                    }
                }
                else
                {
                    int maxLoad=jobDifficulty[i];
                    for(int k=i;n-k>=j;k++)
                    {
                        maxLoad=max(maxLoad,jobDifficulty[k]);
                        dp[i][j]=min(dp[i][j],maxLoad+dp[k+1][j-1]);
                    }
                }
            }
        }
        return dp[0][d];
    }
};
