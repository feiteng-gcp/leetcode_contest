class Solution {
public:
    int minDifficulty(vector<int>& jd, int d) {
        int n = jd.size();
        if ( d > n ) return -1;
        vector<vector<int>> dp(n + 1, vector<int>(d + 1, 1 << 20 ));
        dp[n][0] = 0;
        /*
            dp[i][j] = max(i...k) + dp[k + 1][j - 1];
        */
        for ( int i = n - 1; i >= 0; --i ) {
            for ( int j = 1; j <= d; ++j ) {
                int val = -1;
                for ( int k = i; n - 1 - k >= j - 1 ; ++k ) {
                    // cout << i << " " << j << " " << k << endl;
                    val = max(val, jd[k]);
                    dp[i][j] = min(dp[i][j], val + dp[k + 1][j - 1]);
                }
            }
        }
        if ( dp[0][d] == (1 << 20) ) return -1;
        
        return dp[0][d];
                        
    }
};
