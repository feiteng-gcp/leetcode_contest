class Solution {
public:
    int minDifficulty(vector<int>& jobDifficulty, int d) {
        if (jobDifficulty.size() < d)
            return -1;

        vector<vector<int>> dp(d, vector<int>(jobDifficulty.size(), INT_MAX));
        int m = INT_MIN;
        for (int i = 0; i < jobDifficulty.size(); i++) {
            m = max(m, jobDifficulty[i]);
            dp[0][i] = m;
        }

        for (int r = 1; r < d; r++) {
            for (int c = r; c < jobDifficulty.size(); c++) {

                int maxd = jobDifficulty[c];
                for (int i = c - 1; i >= r - 1; i--) {
                    dp[r][c] = min(dp[r][c], dp[r - 1][i] + maxd);
                    maxd = max(maxd, jobDifficulty[i]);
                }
            }
        }

        return dp[d - 1][jobDifficulty.size() - 1];
    }
};
