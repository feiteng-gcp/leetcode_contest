class Solution {
public:
    int minDifficulty(vector<int>& jobDifficulty, int d) {
        if(jobDifficulty.size()<d){
            return -1;
        }
        vector<vector<int>> dp(jobDifficulty.size()+1,vector<int>(d+1,INT_MAX/3));
        int preDif = 0;
        for(int i=1;i<=jobDifficulty.size();++i){
            preDif=max(preDif,jobDifficulty[i-1]);
            dp[i][1]=preDif;
            //cout<<i<<" "<<1<<" "<<dp[i][1]<<endl;
        }
        for(int i=2;i<=d;i++){
            for(int j=i;j<=jobDifficulty.size();++j){
                preDif = jobDifficulty[j-1];
                for(int k=j-1;k>=i-1;k--){
                    preDif = max(preDif,jobDifficulty[k]);
                    dp[j][i]=min(dp[j][i],dp[k][i-1]+preDif);
                }
               // cout<<j<<" "<<i<<" "<<dp[j][i]<<endl;
            }
        }
        return dp[jobDifficulty.size()][d];
    }
};
