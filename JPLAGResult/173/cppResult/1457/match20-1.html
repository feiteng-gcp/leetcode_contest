class Solution {
public:
    int minDifficulty(vector<int>& jobDifficulty, int d) {
        int n = jobDifficulty.size();
        if (n < d) {
            return -1;
        }
        vector<vector<int>> dp(n, vector<int>(d, INT_MAX));

        int max_num = -1;
        for (int i = 0; i < n; i++) {
            max_num = max(max_num, jobDifficulty[i]);
            dp[i][0] = max_num;
        }
        for (int j = 1; j < d; j++) {
            for (int i = j; i < n; i++) {
                int max_num = -1;
                
                for (int k = i-1; k >= j-1; k--) {
                    max_num = max(max_num, jobDifficulty[k+1]);
                    dp[i][j] = min(dp[i][j], max_num + dp[k][j-1]);
                }
            }
        }
        return dp[n-1][d-1];
    }
};
