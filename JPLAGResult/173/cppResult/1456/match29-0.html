class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int>> e(n, vector<int>(n, 100000000));
        for(int i = 0; i < n; i++) {
            e[i][i] = 0;    
        }
 
        for(int i = 0; i < edges.size(); i++) {
            e[edges[i][0]][edges[i][1]] = edges[i][2];
            e[edges[i][1]][edges[i][0]] = edges[i][2];
        }
        
        for(int k = 0; k < n; k++) {
            for(int i = 0; i < n; i++) {
                for(int j = 0; j < n; j++) {
                    if(e[i][j] > e[i][k] + e[k][j]) {
                        e[i][j] = e[i][k] + e[k][j];
                    }
                }
            }
        }
        
        /* for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                cout << e[i][j] << " ";
            }
            cout << endl;
        } */
        
        int ans = -1;
        int cnt = 10000000;
        
        for(int i = 0; i < n; i++) {
            int temp = 0;
            for(int j = 0; j < n; j++) {
                if(i != j && e[i][j] <= distanceThreshold) {
                    temp++;
                }
            }
            if(temp <= cnt) {
                ans = i;
                cnt = temp;
            }
            // cout << temp << endl;
        }
        
        return ans;
    }
};
