class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int>>dist(n, vector<int>(n, 10000000));
        for (auto it : edges) {
            dist[it[0]][it[1]] = it[2];
            dist[it[1]][it[0]] = it[2];
        }
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    if (dist[i][k] + dist[k][j] < dist[i][j]) {
                        dist[i][j] = dist[i][k] + dist[k][j];
                    }
                }
            }
        }
        int out = -1,minn=-1;

        for (int i = n-1; i >=0; i--) {
            int ct = 0;
            for (int j = 0; j < n; j++) {
                if (i == j)continue;
                if (dist[i][j] <= distanceThreshold)ct++;
            }
            if (out == -1)out = i, minn = ct;
            else if (ct < minn)out = i, minn = ct;
        }
        return out;
    }
};
