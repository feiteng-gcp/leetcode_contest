class Solution {
public:
	int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
		vector<vector<int>>dis(n, vector<int>(n, INT_MAX));
		for (auto& e : edges) {
			dis[e[0]][e[1]] = e[2];
			dis[e[1]][e[0]] = e[2];
		}

		for (int i = 0; i < n; ++i) {
			dis[i][i] = 0;
		}

		// Floyd
		for (int k = 0; k < n; ++k) {
			for (int i = 0; i < n; ++i) {
				for (int j = 0; j < n; ++j) {
					if (dis[i][k] != INT_MAX && dis[k][j] != INT_MAX) {
						dis[i][j] = min(dis[i][j], dis[i][k] + dis[k][j]);
					}
				}
			}
		}
		vector<int>cnt(n, INT_MAX);
		for (int i = 0; i < n; ++i) {
			cnt[i] = count_if(dis[i].begin(), dis[i].end(), [&](int val) {return val <= distanceThreshold; });
		}

		int min_cnt = INT_MAX, idx = -1;
		for (int i = 0; i < n; ++i) {
			//cout << i << " " << cnt[i] << endl;
			if (cnt[i] <= min_cnt) {
				idx = i;
				min_cnt = cnt[i];
			}
		}
		return idx;
	}
};
