typedef pair<int, int> ii;

class Solution {
    int thres;
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        thres = distanceThreshold;
        unordered_map<int, vector<vector<int>>> graph;
        for (auto& edge : edges) {
            graph[edge[0]].push_back({edge[1], edge[2]});
            graph[edge[1]].push_back({edge[0], edge[2]});
        }
        
        int MIN = n, id = -1;
        for (int i = 0; i < n; i++) {
            vector<int> visited(n, INT_MAX);
            visited[i] = 1;
            dfs(graph, visited, 0, i);
            int cnt = 0;
            for (int i = 0; i < n; i++)
                if (visited[i] != INT_MAX)    cnt++;
            // cout<<i<<" "<<cnt<<endl;
            if (cnt <= MIN) {
                MIN = cnt;
                id = i;
            }
        }
        
        return id;
    }
    
private:
    void dfs(unordered_map<int, vector<vector<int>>>& graph, vector<int>& visited, int d, int s) {
        for (auto& nxt : graph[s]) {
            int end = nxt[0], dis = nxt[1];
            if (d + dis > thres)    continue;
            if (d + dis < visited[end]) {
                visited[end] = d + dis;
                dfs(graph, visited, d + dis, end);   
            }
        }
    }
};
