class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        int ans = 0;

        vector<vector<int>> d(n, vector(n, 1000000007));
        for (int i = 0; i < edges.size(); i++) {
            int x = edges[i][0];
            int y = edges[i][1]; 
            int z = edges[i][2];
            d[x][y] = min(d[x][y], z);
            d[y][x] = min(d[y][x], z);
        }
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    d[i][j] = min(d[i][j], d[i][k] + d[k][j]);
                }
            }
        }
        int now = 10000007;

        for (int i = 0; i < n; i++) {
            int p = 0;
            for (int j = 0; j < n; j++) {
                if (i == j) continue;
                if (d[i][j] <= distanceThreshold) p++;
                //if (i == 4) cout<<d[i][j]<<endl;
            }
            if (p <= now) {
                //cout<<i<<" "<<p<<endl;
                now = p;
                ans = i;
            }
        }
        return ans;
    }
};
