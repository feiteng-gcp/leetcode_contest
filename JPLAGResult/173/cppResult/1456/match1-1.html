class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int>> dis(n, vector<int>(n, INT_MAX));
        vector<vector<int>> g(n, vector<int>(n, INT_MAX));
        for (auto &e: edges) {
            g[e[0]][e[1]] = e[2];
            g[e[1]][e[0]] = e[2];
        }
        for (int i = 0; i < n; ++i) {
            vector<bool> visited(n, false);
            dis[i][i] = 0;
            for (int j = 0; j < n - 1; ++j) {
                int minIndex = -1, minDis = INT_MAX;
                for (int k = 0; k < n; ++k) {
                    if (visited[k]) continue;
                    if (dis[i][k] < minDis) {
                        minDis = dis[i][k];
                        minIndex = k;
                    }
                }
                if (minIndex == -1) break;
               // printf("%d %d minIndex %d %d\n", i, j, minIndex, minDis);
                visited[minIndex] = true;
                for (int k = 0; k < n; ++k) {
                    if (g[minIndex][k] != INT_MAX) {
                        dis[i][k] = min(dis[i][k], minDis + g[minIndex][k]);
                //        printf("update %d %d %d\n", minIndex, k, dis[i][k]);
                    }
                }
            }
        }
        
        int retId = -1, retCount = INT_MAX;
        for (int i = 0; i < n; ++i) {
            int count = 0;
            for (int j = 0; j < n; ++j) {
              //  printf("%d %d %d\n", i, j, dis[i][j]);
                if (i != j && dis[i][j] <= distanceThreshold) ++count;
            }
           // printf("%d\n", count);
            if (count < retCount || (count == retCount && i > retId)) {
                retCount = count;
                retId = i;
            }
        }
        return retId;
    }
};
