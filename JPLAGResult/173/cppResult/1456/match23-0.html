class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        
        vector<vector<long>> table(n, vector<long>(n, INT_MAX));
        for (auto& edge: edges) {
            table[edge[0]][edge[1]] = edge[2];
            table[edge[1]][edge[0]] = edge[2];
        }
        
        for (int i = 0; i < n; ++i) {
            table[i][i] = 0;
        }
        
        for (int k = 0; k < n; ++k) {
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < n; ++j) {
                    if (table[i][j] > table[i][k] + table[k][j]) {
                        table[i][j] = table[i][k] + table[k][j];
                    }
                    
                }
            }
        }
        
        int idx = -1, nCount = INT_MAX;
        for (int i = 0; i < n; ++i) {
            int cCount = 0;
            for (int j = 0; j < n; ++j) {
                if (table[i][j] <= distanceThreshold) {
                    cCount += 1;
                }
            }
                        
            if (cCount <= nCount || idx == -1) {idx = i; nCount = cCount;}
        }
        
        return idx;
        
    }
};
