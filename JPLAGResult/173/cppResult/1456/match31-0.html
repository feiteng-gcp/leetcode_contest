class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int>> adj(n, vector<int>(n, -1));
       
        for(int j = 0; j < n; j++)
            adj[j][j] = 0;
        for(int i = 0; i < edges.size(); i++) {
            int u = edges[i][0];
            int v = edges[i][1];
            int w = edges[i][2];
            if (w <= distanceThreshold)
                adj[u][v] = adj[v][u] = w;
        }
       
        for (int i = 0; i < n; i++) {
            for (int k = 0; k < n; k++)
                for (int u = 0; u < n; u++) {
                    if (adj[u][k] == -1 || u == k)    continue;
                    for (int v = 0; v < n; v++) {
                        if (adj[k][v] == -1 || v == u || v == k)    continue;
                        if (adj[u][v] == -1 || adj[u][v] > adj[u][k] + adj[k][v])
                            adj[u][v] = adj[u][k] + adj[k][v];
                    }
                }
        }
       
        int max_value = -1;
        int max_id = -1;
        for (int i = 0; i < n; i++) {
            int value = 0;
            for (int j = 0; j < n; j++) {
                if (j != i && adj[i][j] != -1 && adj[i][j] <= distanceThreshold) {
                    value++;
                }
            }
            if (max_value == -1 || max_value >= value) {
                max_id = i;
                max_value = value;
            }
        }
       
        return max_id;
    }
};
