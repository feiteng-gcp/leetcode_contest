class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int>> dist(n, vector<int>(n, 100000));
        for (int i = 0; i < n; ++i) {
            dist[i][i] = 0;
        }
        
        for (auto& e : edges) {
            int f = e[0], t = e[1];
            dist[f][t] = e[2];
            dist[t][f] = e[2];
        }
        for (int k = 0; k < n; ++k) {
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < n; ++j) {
                    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
                }
            }
        }
        
        int id = 0, num = n + 1;
        for (int i = 0; i < n; ++i) {
            int curr = 0;
            for (int j = 0; j < n; ++j) {
                if (dist[i][j] <= distanceThreshold) {
                    ++curr;
                }
            }
            if (curr <= num) {
                id = i;
                num = curr;
            }
        }
        return id;
    }
};
