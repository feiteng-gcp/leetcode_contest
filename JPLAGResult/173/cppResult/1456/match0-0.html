class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int>> mat(n,vector<int>(n,1e5+1));
        for(auto item:edges){
            int x = item[0];
            int y = item[1];
            mat[x][y] = item[2];
            mat[y][x] = item[2];
        }
        for(int i=0;i<n;i++) mat[i][i] = 0;
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                for(int k=0;k<n;k++){
                    if(mat[j][i]+mat[k][i] < mat[j][k]){
                        mat[j][k] = mat[j][i]+mat[k][i];
                    }
                }
            }
        }
        int ans = 1e4+1;
        int res = -1;
        vector<int> cnt(n,0);
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(mat[i][j]<=distanceThreshold) cnt[i]++;
            }
            if(cnt[i]<=ans){
                ans=cnt[i];
                res = i;
            }
        }
        return res;
    }
};
