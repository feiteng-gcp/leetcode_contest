class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int>> d(n, vector<int>(n, 1000000000));
        for (int i = 0; i < n; i++) {
            d[i][i] = 0;
        }
        for (const auto&e : edges) {

            d[e[0]][e[1]] = min(d[e[0]][e[1]], e[2]);
            d[e[1]][e[0]] = min(d[e[1]][e[0]], e[2]);
        }
        
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    d[i][j] = min(d[i][j], d[i][k] + d[k][j]);
                }
            }
        }
        
        int bestVal = n+1, best = 0;
        for (int i = 0; i < n; i++) {
            int val = 0;
            for (int j = 0; j < n; j++) {
                if (i != j && d[i][j] <= distanceThreshold) {
                    val++;
                }
            }
            //printf("city %d: %d\n", i, val);
            if (val <= bestVal) {
                bestVal = val;
                best = i;
            }
        }
        return best;
    }
};
