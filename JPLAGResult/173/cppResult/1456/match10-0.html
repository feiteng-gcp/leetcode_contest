class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
      vector<vector<int>> d(n, vector<int>(n, distanceThreshold*2));
      for (int i = 0; i < n; ++i)  d[i][i] = 0;
      for (const auto& edge : edges) {
        d[edge[0]][edge[1]] = edge[2];
        d[edge[1]][edge[0]] = edge[2];
      }
      for (int k = 0; k < n; ++k) {
        for (int i = 0; i < n; ++i) {
          for (int j = 0; j < n; ++j) {
            d[i][j] = min(d[i][j], d[i][k] + d[k][j]); 
           // printf("setting d[%d][%d]: %d\n", i, j, d[i][j]);
          }
        }
      }
      int min_count = n+1;
      int i_min_count = n;
      for (int i = 0; i < n; ++i) {
        int count = 0;
        for (int j = 0; j < n; ++j) {
         // printf("%d -> %d: %d\n", i, j, d[i][j]);
          if (d[i][j] <= distanceThreshold) {
            ++count;
          }
        }
        //printf("%d; count: %d\n", i, count);
        if (count <= min_count) {
          min_count = count;
          i_min_count = i;
        }
      }
      return i_min_count;
    }
};
