class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int md) {
        vector<vector<int>> a(n, vector<int>(n, 1e8));
        for (int i = 0; i < edges.size(); i++) {
            int u = edges[i][0];
            int v = edges[i][1];
            int w = edges[i][2];
            a[u][v] = min(a[u][v], w);
            a[v][u] = a[u][v];
        }
        
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    a[i][j] = min(a[i][j], a[i][k] + a[k][j]);
                }
            }
        }
        
        int result = -1;
        int reachable = n + 1;
        for (int i = 0; i < n; i++) {
            int count = 0;
            for (int j = 0; j < n; j++) {
                if (i == j) continue;
                
                if (a[i][j] <= md) count++;
            }
            
            if (count <= reachable) {
                result = i;
                reachable = count;
            }
        }
        
        return result;
    }
};
