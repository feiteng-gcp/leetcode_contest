class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int>> disTo(n, vector<int>(n, 1e5));
        for(int i = 0; i < n; i++) {
            disTo[i][i] = 0;
        }
        
        for(auto e : edges) {
            auto u = e[0], v = e[1], w = e[2];
            disTo[u][v] = w;
            disTo[v][u] = w;
        }
        int N = n;
        for (int k = 0; k < N; k++) {
            for (int i = 0; i < N; i++) {
                for (int j = 0; j < N; j++) {
                    if (disTo[i][j] > disTo[i][k] + disTo[k][j])
                        disTo[i][j] = disTo[i][k] + disTo[k][j];
                }
            }
        }
        
        int city = 0;
        int num_city = n + 1;
        
        for(int i = 0; i < n; i++) {
            int count = 0;
            for(int j = 0; j < n; j++) {
                if(i == j) continue;
                if(disTo[j][i] <= distanceThreshold) count++;
            }
            if(count <= num_city) {
                city = i;
                num_city = count;
            }
        }
        return city;
    }
};
