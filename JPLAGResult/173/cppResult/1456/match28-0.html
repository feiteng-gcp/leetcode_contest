int dis[105][105];

class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        for (int i = 0;i < 105;i++) {
            for (int j = 0;j < 105;j++) {
                dis[i][j] = 1e9;
            }
        }
        for (int i = 0;i < edges.size();i++) {
            dis[edges[i][0]][edges[i][1]] = edges[i][2];
            dis[edges[i][1]][edges[i][0]] = edges[i][2];
        }
        for (int i = 0;i < n;i++) {
            for (int j = 0;j < n;j++) {
                for (int k = 0;k < n;k++) {
                    if (dis[j][k] > dis[j][i] + dis[i][k]) {
                        dis[j][k] = dis[j][i] + dis[i][k];
                    }
                }
            }
        }
        int curm = 1e9, ret;
        for (int i = 0;i < n;i++) {
            int cnt = 0;
            for (int j = 0;j < n;j++) {
                if (i == j) continue;
                if (dis[i][j] <= distanceThreshold) {
                    cnt++;
                }
            }
            //cout << i << " " << cnt << endl;
            if (curm >= cnt) {
                curm = cnt;
                ret = i;
            }
        }
        return ret;
    }
};
