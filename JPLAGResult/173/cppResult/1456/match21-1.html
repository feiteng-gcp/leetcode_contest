class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int>> dp(n,vector<int>(n,1e9));
        for (auto it:edges) {
            dp[it[0]][it[1]] = it[2];
            dp[it[1]][it[0]] = it[2];
        }
        for (int i=0; i<n; i++) {
            dp[i][i] = 0;
        }

        for (int k=0;k<n;k++) {
            for (int i=0;i<n;i++) {
                for (int j=0;j<n;j++) {
                    dp[i][j] = min(dp[i][j],min(dp[i][k]+dp[k][j],dp[j][k]+dp[k][i]));
                }
            }
        }
        int min_edges=1e9, city = -1;

        for (int i=0; i<n; i++) {
            int cnt = 0;
            for (int j=0; j<n; j++) {
                if (i!=j && dp[i][j]<=distanceThreshold) {
                    cnt++;
                }
            }
            if (cnt <= min_edges) {
                min_edges = cnt;
                city = i;
            }
        }
        return city;
    }
};
