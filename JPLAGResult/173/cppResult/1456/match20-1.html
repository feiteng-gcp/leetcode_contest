class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& e, int d) {  
        vector<vector<int>> g(n,vector<int>(n,d+1));  
        for(int i = 0;i < e.size();i++){
            g[e[i][0]][e[i][1]] = e[i][2];
            g[e[i][1]][e[i][0]] = e[i][2];
        }
        for(int i = 0;i<n;i++){
            for(int j = 0; j < n; j++){
                if(i==j){
                    continue;
                }
                for(int k = 0; k < n; k++){
                    if(k==i||k==j) continue;

                    g[k][j] = min(g[k][j],g[i][j]+g[k][i]);
                }
            }
        }
        int minval = n;
        int ans = 0;
        for(int i = 0;i<n;i++){
            int cnt = 0;
            for(int j = 0; j < n; j++){
                if(g[i][j] <= d){

                    cnt++;
                }
            }
            if(cnt <= minval ){
                minval = cnt;
                ans = i;
            }
        } 
        return ans;
    }
};
