class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int>> dp(n, vector<int>(n, 1000000));
        for(auto it:edges)
        {
            dp[it[0]][it[1]] = it[2];
            dp[it[1]][it[0]] = it[2];
        }
        for(int i=0;i<n;i++) dp[i][i] = 0;
        for(int k=0;k<n;k++)
        {
            for(int i=0;i<n;i++)
            {
                for(int j=0;j<n;j++)
                {
                    if(dp[i][j]>dp[i][k]+dp[k][j])
                    {
                        dp[i][j]=dp[i][k]+dp[k][j];
                        //cout<<i<<" "<<j<<" "<<dp[i][j]<<endl;
                    }
                }
            }
        }
        // for(int i=0;i<n;i++)
        // {
        //     for(int j=0;j<n;j++)
        //     {
        //         cout<<dp[i][j]<<" ";
        //     }
        //     cout<<endl;
        // }
        int res = -1;
        int resV = n;
        for(int i=0;i<n;i++)
        {
            int num = 0;
            for(int j=0;j<n;j++)
            {
                if(dp[i][j]<=distanceThreshold)
                {
                    num++;
                }
            }
            if(num <= resV)
            {
                resV = num;
                res = i;
            }
        }
        return res;
        
    }
};
