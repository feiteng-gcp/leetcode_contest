class Solution {
    int dp[102][102];
    void print(int n) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                printf("%d ", dp[i][j]);
            }
            printf("\n");
        }
        printf("\n");
        return;
    }

public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                dp[i][j] = distanceThreshold+1;
            }
        }
        for (vector<int> edge: edges) {

            dp[edge[0]][edge[1]] = dp[edge[1]][edge[0]] = edge[2];            
        }
//        print(n);
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    if (dp[i][j] > (dp[i][k] + dp[k][j]) ) {
                        dp[i][j] = (dp[i][k] + dp[k][j]);
                    }
                }
            }
        }
//        print(n);
        int bestCity = 0, mini = n+1;

        for (int i = 0; i < n; i++) {
            int cont = 0;
            for (int j = 0; j < n; j++) {

                if (i != j && dp[i][j] <= distanceThreshold) {
//                    printf("%d ingresa %d\n", i, j);
                    cont++;
                }
            }
//            printf("%d -> %d\n", cont, i);
            if (cont <= mini) {
                mini = cont, bestCity = i;
            }
        }
        return bestCity;
    }
};
