class Solution {
public:
    int findTheCity(int n, vector<vector<int>>& edges, int distanceThreshold) {
        vector<vector<int> > d(n,vector<int>(n,10000000));
        for(int i=0;i<n;i++) d[i][i]=0;
        for(auto &edge:edges)
        {
            d[edge[0]][edge[1]]=edge[2];
            d[edge[1]][edge[0]]=edge[2];
        }
        for(int k=0;k<n;k++)
        {
            for(int i=0;i<n;i++)
            {
                for(int j=0;j<n;j++)
                {
                    if(d[i][j]>d[i][k]+d[k][j])
                    {
                        d[i][j]=d[i][k]+d[k][j];
                    }
                }
            }
        }
        int res=-1,min=INT_MAX;
        for(int i=0;i<n;i++)
        {
            int temp=0;
            for(int j=0;j<n;j++)
            {
                if(i!=j&&d[i][j]<=distanceThreshold)
                {
                    temp++;
                }
            }
            if(temp<=min)
            {
                min=temp;
                res=i;
            }
        }
        return res;
    }
};
