class Solution {
public:
    int removePalindromeSub(string s) {
        int len = s.length();
        if(len <= 1)    return 0;
        bool isParlin = true;
        for(int start = 0, end = len - 1; start < end; start++, end--){
            if(s[start] != s[end]){
                isParlin = false;
                break;
            }
        }
        
        if(isParlin)    return 1;
        return 2;
    }
};
