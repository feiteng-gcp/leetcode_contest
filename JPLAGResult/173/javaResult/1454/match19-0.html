class Solution {
    public int removePalindromeSub(String s) {
        if ("".equals(s)) return 0;
        char[] cs = s.toCharArray();
        int i = 0, j = cs.length - 1;
        while (i <= j) {
            if (cs[i] != cs[j]) return 2;
            i++;
            j--;
        }
        return 1;
    }
}
