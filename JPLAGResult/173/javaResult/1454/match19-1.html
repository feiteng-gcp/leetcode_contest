class Solution {
    public int removePalindromeSub(String s) {
        if (s == null || s.length() == 0) return 0;
        char[] arr = s.toCharArray();
        int left = 0, right = arr.length - 1;
        while (left < right) {
            if (arr[left] != arr[right]) return 2;
            left++;
            right--;
        }
        return 1;
    }
}
