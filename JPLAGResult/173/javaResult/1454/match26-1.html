class Solution {
    public int removePalindromeSub(String s) {
        if(s.length() == 0){
            return 0;
        }
        if(check(0, 
s.length() - 1, s)){
            return 1;
        }else{
            return 2;
        }
    }
    public boolean check(int i, int j, String s){
        
while(i < j){
            if(s.charAt(i) != s.charAt(j)){
                return false;
            }
            i++;
            j--;
        }
        return true;
    }
}
