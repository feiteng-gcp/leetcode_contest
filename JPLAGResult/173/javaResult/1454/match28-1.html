class Solution {
    public int removePalindromeSub(String s) {
        if (s == null || s.isEmpty()) {
            return 0;
        }
        return isPalindrome(s) ? 1 : 2;
    }

    private boolean isPalindrome(String s) {
        
for (int l = 0, r = s.length() - 1; l < r; ++l, --r) {
            if (s.charAt(l) != s.charAt(r)) {
                return false;
            }
        }
        return true;
    }
}
