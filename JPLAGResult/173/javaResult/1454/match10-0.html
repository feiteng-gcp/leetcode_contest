class Solution {
    public int removePalindromeSub(String s) {

        if (s.length() == 0) {
            return 0;
        }

        if (isPalindrome(s)) {
            return 1;
        }

        return 2;

    }

    private boolean isPalindrome(String s) {
        int lo = 0, hi = s.length() - 1;
        while (lo < hi) {
            if (s.charAt(lo) != s.charAt(hi)) {
                return false;
            }
            lo++;
            hi--;
        }
        return true;
    }
}
