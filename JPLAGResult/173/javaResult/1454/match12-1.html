class Solution {
    public int removePalindromeSub(String s) {
        int l = 0;
        int h = s.length()-1;
        if(h==-1) return 0;
        
        while(l<h){
            if(s.charAt(l)==s.charAt(h)){
                l++;h--;
            }else{
                break;
            }
        }
        if(l>=h){
            return 1;
        }
        
        return 2;
        
    }
}
