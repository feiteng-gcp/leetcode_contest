class Solution {
   
    public int removePalindromeSub(String s) {
        if (s.isEmpty()) {
            return 0;
        }
        if (ispalin(s)) {
            return 1;
        }
        Set<Character> set = new HashSet<>();
        for (int i = 0; i < s.length(); i++) {
            set.add(s.charAt(i));
        }
        return set.size();
    }

    private boolean ispalin(String s) {
        int i = 0;
        int j = s.length() - 1;
        while (i < j) {
            if (s.charAt(i) != s.charAt(j)) {
                return false;
            }
            i++;
            j--;
        }
        return true;
    }
}
