class Solution {
    public int removePalindromeSub(String s) {
        if (s.length() == 0) {
            return 0;
        }
        if (isPalin(s)) {
            return 1;
        }
        boolean[] seen = new boolean[2];
        for (char c : s.toCharArray()) {
            seen[c - 'a'] = true;
        }
        int res = 0;
        for (boolean b : seen) {
            if (b) {
                res++;
            }
        }
        return res;
    }
    
    private boolean isPalin(String s) {
        int l = 0, r = s.length() - 1;
        while (l < r) {
            if (s.charAt(l) != s.charAt(r)) {
                return false;
            }
            l++;
            r--;
        }
        return true;
    }
}
