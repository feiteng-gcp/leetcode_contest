class Solution {
    public int removePalindromeSub(String s) {
        int n = s.length();
        if(n <= 1) return n;
        if (isPalindrome(s)) return 1;
        return 2;
    }
    public boolean isPalindrome(String str) 
    { 
        int l = 0; 
        int h = str.length() - 1; 

        while (h > l) 
            if (str.charAt(l++) != str.charAt(h--)) 
                return false; 
        return true; 
    } 
}
