class Solution {
    public List<Integer> filterRestaurants(int[][] restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        List<Pair> rests = new ArrayList<>();
        for (int[] info: restaurants) {
            if (info[2] >= veganFriendly && info[3] <= maxPrice && info[4] <= maxDistance) {
                rests.add(new Pair(info[0], info[1]));
            }
        }
        
        rests.sort(new Comparator<Pair>() {
            @Override
            public int compare(Pair p1, Pair p2) {
                if (p1.b == p2.b) {
                    return p2.a - p1.a;
                }
                return p2.b - p1.b;
            }
        });
        
        List<Integer> result = new ArrayList<>();
        for (Pair p: rests) {
            result.add(p.a);
        }
        
        return result;
    }
    
    public class Pair {
        public int a;
        public int b;
        
        public Pair(int a, int b) {
            this.a = a;
            this.b = b;
        }
    }
}
