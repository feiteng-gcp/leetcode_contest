class Solution {
    //[idi, ratingi, veganFriendlyi, pricei, distancei]
    public List<Integer> filterRestaurants(int[][] restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        List<int[]> filtered = new ArrayList<>();
        for (int[] r : restaurants) {
            if ((veganFriendly == 0 || r[2] == veganFriendly) 
                    && r[3] <= maxPrice && r[4] <= maxDistance) {
                filtered.add(new int[] {r[0], r[1]});
            
}
        }
        Collections.sort(filtered, new Comparator<int[]>() {
            @Override
            public int compare(int[] a1, int[] a2) {
                if (a1[1] == a2[1]) {
                    return Integer.compare(a2[0], a1[0]);
                
}
                return Integer.compare(a2[1], a1[1]);
            }
        });
        List<Integer> res = new ArrayList<>();
        for (int[] f : filtered) {
            res.add(f[0]);
        }
        return res;
    }
}
