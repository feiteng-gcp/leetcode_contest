class Solution {
    public List<Integer> filterRestaurants(int[][] restaurants, int vegan, int maxPrice, int maxDistance) {
        List<Integer> res = new ArrayList<>();
        List<int[]> list = new ArrayList<>();
        for (int[] restaurant : restaurants) {
            if (restaurant[2] >= vegan && restaurant[3] <= maxPrice && restaurant[4] <= maxDistance) {
                list.add(restaurant);
            }
        
}
        
	    Collections.sort(list, (a, b) -> a[1] == b[1] ? b[0] - a[0] : b[1] - a[1]);
        for (int[] re : list) {
            res.add(re[0]);
        }
        
        return res;
    }
}
