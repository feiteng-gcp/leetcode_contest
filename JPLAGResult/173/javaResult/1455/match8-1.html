class Solution {
    public List<Integer> filterRestaurants(int[][] rest, int veg, int mp, int md) {
        List<int[]> list = new ArrayList<>();
        List<Integer> re = new ArrayList<>();
        for(int[] r : rest)
        {
            // [idi, ratingi, veganFriendlyi, pricei, distancei]
            if(veg == 1 && r[2] == 0) continue;
            if(r[3] > mp) continue;
            if(r[4] > md) continue;
            list.add(r);
        }
        Collections.sort(list, (a, b)->(a[1] == b[1] ? b[0] - a[0] : b[1] - a[1]));
        for(int[] r : list)
        {
            re.add(r[0]);
        }
        return re;
    }
}
