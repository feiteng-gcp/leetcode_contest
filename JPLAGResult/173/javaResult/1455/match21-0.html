class Solution {
    public List<Integer> filterRestaurants(int[][] restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        ArrayList<int[]> result = new ArrayList();
        for(final int[] rest : restaurants) {
            if ((veganFriendly == 0 || rest[2] == 1) && (rest[3] <= maxPrice) && (rest[4] <= maxDistance)) {
                result.add(rest);
            }
        }
        Collections.sort(result, new Comparator<int[]>(){
            @Override
            public int compare(final int[] first, final int[] second) {
                final int orderComp = Integer.compare(second[1], first[1]);
                if (orderComp == 0) {
                    return Integer.compare(second[0], first[0]);
                }
                
                return orderComp;
            }
        });
        final ArrayList<Integer> ids = new ArrayList<Integer>();
        for(final int[] r : result) {
            ids.add(r[0]);
        }
        return ids;
    }
}
