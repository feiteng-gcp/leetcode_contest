class Solution {
    public class Rest implements Comparable<Rest>
    {
        int ID;
        int rate;
        public Rest(int rate, int ID)
        {
            this.rate = rate;
            this.ID = ID;
        }
        public int compareTo(Rest r2) 
        {
            if(rate != r2.rate)
                return r2.rate - rate;
            return r2.ID - ID; 
        } 
    }
    public List<Integer> filterRestaurants(int[][] restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        PriorityQueue <Rest> pq = new PriorityQueue();
        for(int [] r : restaurants)
        {
            if(r[2] >= veganFriendly && r[3] <= maxPrice && r[4] <= maxDistance)
                pq.add(new Rest(r[1], r[0]));
        }
        LinkedList <Integer> ans = new LinkedList();
        while(!pq.isEmpty())
        {
            Rest r = pq.poll();
            ans.add(r.ID);
        }
        return ans;
    }
}
