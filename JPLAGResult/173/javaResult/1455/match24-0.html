class Solution {
    public List<Integer> filterRestaurants(int[][] restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        List<Integer> indices = new ArrayList<>();
        for (int i = 0; i < restaurants.length; i++) {
            if (restaurants[i][2] >= veganFriendly && restaurants[i][3] <= maxPrice && restaurants[i][4] <= maxDistance)
                indices.add(i);
        }
        Collections.sort(indices, (id1, id2) -> {
            if (restaurants[id1][1] != restaurants[id2][1]) return restaurants[id2][1] - restaurants[id1][1];
            else return restaurants[id2][0] - restaurants[id1][0];
        });
        List<Integer> res = new ArrayList<>();
        for (int idx: indices) res.add(restaurants[idx][0]);
        
        return res;
    }
}
