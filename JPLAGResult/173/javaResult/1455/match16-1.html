class Solution {
    public List<Integer> filterRestaurants(int[][] arr, int veganFriendly, int maxPrice, int maxDistance) {
        List<int[]> list = new ArrayList<>();
        for(int i = 0; i < arr.length; i++) {
            if(arr[i][2] >= veganFriendly && arr[i][3] <= maxPrice && arr[i][4] <= maxDistance) {
                list.add(new int[]{arr[i][0], arr[i][1]});
            
}
        }
        Collections.sort(list, (a, b)->a[1] == b[1] ? b[0] - a[0] : b[1] - a[1]);
        
        List<Integer> ret = new ArrayList<>();
        
for(int[] d: list) {
            ret.add(d[0]);
        }
        return ret;
    }
}
