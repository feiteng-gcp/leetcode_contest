class Solution {
    public List<Integer> filterRestaurants(int[][] rs, int vf, int mp, int md) {
        List<int[]> l = new ArrayList<>();
        if(vf==1){
        for(int[] me:rs){
            if(me[2]==vf&&me[3]<=mp&&me[4]<=md){
                l.add(new int[]{me[1],me[0]});
            }
        }
        }else{
            for(int[] me:rs){
            if(me[3]<=mp&&me[4]<=md){
                l.add(new int[]{me[1],me[0]});
            }
        }
            
        }
        Collections.sort(l,(a,b)->{ if(b[0]!=a[0]){ return b[0]-a[0];}else{ return b[1]-a[1];}});
        
        List<Integer> il = new ArrayList<>();
        
        for(int[] m:l){
            il.add(m[1]);
        }
        return  il;
        
        
    }
}
