class Solution {
    public List<Integer> filterRestaurants(int[][] restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        List<int[]> res = new ArrayList<>();
        
        for (int[] r: restaurants) {
            boolean veganMatch = r[2] == 1 || veganFriendly == 0;
            boolean distMatch = r[4] <= maxDistance;
            boolean priceMatch = r[3] <= maxPrice;
            if (veganMatch && distMatch && priceMatch) {
                res.add(r);
            }
        }
        Collections.sort(res, (a, b) -> a[1] != b[1] ? -Integer.compare(a[1], b[1]) : -Integer.compare(a[0], b[0]));
        
List<Integer> ids = new ArrayList<>();
        for (int[] r: res) {
            ids.add(r[0]);
        }
        return ids;
    }
}
