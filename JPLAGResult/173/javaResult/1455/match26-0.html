class Solution {
    public List<Integer> filterRestaurants(int[][] restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        List<int[]> rest = new ArrayList<>();
        for (int[] r: restaurants) {
            int vegan = r[2], price = r[3], dist = r[4];
            if (veganFriendly == 1 && vegan == 0 || price > maxPrice || dist > maxDistance) {
                continue;
            }
            rest.add(r);
        }
        Collections.sort(rest, new Comparator<int[]>(){
            public int compare(int[] a, int[] b) {
                int c = b[1]-a[1];
                return c != 0? c : b[0]-a[0];
            }
        });
        List<Integer> result = new ArrayList<>(restaurants.length);
        for (int[] r: rest) {
            result.add(r[0]);
        }
        return result;
    }
}
