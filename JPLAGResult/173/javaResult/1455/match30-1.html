class Solution {
    public List<Integer> filterRestaurants(int[][] restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        List<Integer> candidates = new ArrayList<>();
        
        for (int i = 0; i < restaurants.length; i++) {
            
if ((veganFriendly==1 && restaurants[i][2] != veganFriendly) ||
                restaurants[i][3] > maxPrice ||
                restaurants[i][4] > maxDistance) {
                continue;
            }
            candidates.add(i);
        }
        
        Collections.sort(candidates, (i,j)->(restaurants[j][0]-restaurants[i][0]));
        Collections.sort(candidates, (i,j)->(restaurants[j][1]-restaurants[i][1]));
        
        
List<Integer> res = new ArrayList<>();
        for (int ix : candidates) {
            res.add(restaurants[ix][0]);
        }
        return res;
    }
}
