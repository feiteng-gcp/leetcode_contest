class Solution {
    public List<Integer> filterRestaurants(int[][] restaurants, int veganFriendly, int maxPrice, int maxDistance) {
        List<Integer> filterIndex = new ArrayList<>();
        
        for(int i = 0; i < restaurants.length; i++) {
            int[] restaurant = restaurants[i];
            if(veganFriendly == 1 && restaurant[2] == 0) {
                continue;
            }
            if(restaurant[3] > maxPrice) {
                continue;
            }
            if(restaurant[4] > maxDistance) {
                continue;
            }
            filterIndex.add(i);
        }
        Collections.sort(filterIndex, (o1, o2) -> {
            if(restaurants[o1][1] == restaurants[o2][1]) {
                return restaurants[o2][0] - restaurants[o1][0];
            }
            return restaurants[o2][1] - restaurants[o1][1];
        });
        List<Integer> output = new ArrayList<>();
        for(int i : filterIndex) {
            output.add(restaurants[i][0]);
        }
        return output;
    }
}
