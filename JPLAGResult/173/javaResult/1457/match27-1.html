class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {
        int[][] memo = new int[d + 1][jobDifficulty.length];
        for (int[] r : memo) {
            Arrays.fill(r, -2);
        }
        return minDiff(jobDifficulty, d, 0, memo);
    
}
    
    private int minDiff(int[] jobs, int d, int s, int[][] memo) {
        if (d == 0) {
            return s == jobs.length ? 0 : -1;
        }
        if (d < 0 || s >= jobs.length) {
            return -1;
        }
        if (memo[d][s] >= -1) {
            return memo[d][s];
        
}
        int minSum = -1;
        int currMax = -1;
        for (int i = s; i < jobs.length; i++) {
            currMax = Math.max(currMax, jobs[i]);
            int remainMin = minDiff(jobs, d - 1, i + 1, memo);
            if (remainMin < 0) {
                continue;
            }
            int currMin = currMax + remainMin;
            if (minSum < 0 || currMin < minSum) {
                minSum = currMin;
            
}
        }
        memo[d][s] = minSum;
        return minSum;
    }
}
