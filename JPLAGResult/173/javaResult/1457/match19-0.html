class Solution {
    public int minDifficulty(int[] A, int d) {
        int n=A.length;
        if(d>n)return -1;
        int[][] dp=new int[d+1][n+1];
        for(int[] D:dp){
            Arrays.fill(D,99999999);
        }
        dp[0][0]=0;
        for(int i=1;i<=d;i++){
            for(int j=1;j<=n;j++){
                int v=A[j-1];
                for(int k=j-1;k>=0;k--){
                    // if(i-1==0&&k==0){
                    //     System.out.println(dp[0][0]);
                    // }
                    dp[i][j]=Math.min(dp[i][j],dp[i-1][k]+v);
                    v=Math.max(v,k==0?0:A[k-1]);
                }
            }
        }
        return dp[d][n];
    }
}
