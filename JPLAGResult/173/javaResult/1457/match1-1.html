class Solution {
    private int[][] maxValue;
    private int[] preSum;
    private int[][] memo;
    public int minDifficulty(int[] jobDifficulty, int d) {
        int len = jobDifficulty.length;
        if (len < d) {
            return -1;
        }
        maxValue = new int[len][len];
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < len; j++) {
                int max = Integer.MIN_VALUE;
                for (int k = i; k <= j; k++) {
                    max = Math.max(jobDifficulty[k], max);
                }
                maxValue[i][j] = max;
            }
        }
        preSum = new int[len + 1];
        for (int i = 0; i < len; i++) {
            preSum[i + 1] = preSum[i] + jobDifficulty[i];
        }
        memo = new int[len + 1][d + 1];
        return dfs(jobDifficulty, len, d);
    }
    
    private int dfs(int[] nums, int end, int d) {
        if (memo[end][d] != 0) {
            return memo[end][d];
        }
        if (end == d) {
            return preSum[end];
        }
        if (d == 1) {
            return maxValue[0][end - 1];
        }
        int cur = Integer.MAX_VALUE;
        for (int i = d - 1; i < end; i++) {
            cur = Math.min(cur, dfs(nums, i, d - 1) + maxValue[i][end - 1]);
        }
        memo[end][d] = cur;
        return memo[end][d];
    }
}
