class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {


        if (jobDifficulty.length < d) {
            return -1;
        }

        int[][] dp = new int[d][jobDifficulty.length];


        dp[0][0] = jobDifficulty[0];
        for (int j = 1; j < jobDifficulty.length; j++) {

            dp[0][j] = Math.max(dp[0][j - 1], jobDifficulty[j]);
        }

        for (int i = 1; i < d; i++) {


            for (int j = i; j < jobDifficulty.length; j++) {


                int minTotalDiff = Integer.MAX_VALUE;
                int dayDiff = 0;


                for (int k = 1; k <= j && j - k >= i - 1; k++) {

                    dayDiff = Math.max(dayDiff, jobDifficulty[j - k + 1]);
                    minTotalDiff = Math.min(minTotalDiff, dp[i - 1][j - k] + dayDiff);
                }

                dp[i][j] = minTotalDiff;

            }
        }

        return dp[d - 1][jobDifficulty.length - 1];


    }
}
