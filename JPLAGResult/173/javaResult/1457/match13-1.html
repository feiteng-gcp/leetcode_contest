class Solution {
    public int minDifficulty(int[] ds, int d) {
        int len = ds.length;
        if(len<d) return -1;
        
        int dp[][] = new int[len+1][d+1];
        
        
for(int i=0;i<=len;++i){
            for(int j=0;j<=d;++j){
                dp[i][j] =100000000;       
            }
        }
        dp[0][0] = 0;
        
        
for(int i=0;i<len;++i){
            for(int j=0;j<d;++j){
                int ma = ds[i];
                for(int k=i+1;k<=len;++k){
                    ma = Math.max(ma,ds[k-1]);
                    dp[k][j+1] = Math.min(dp[i][j] + ma,dp[k][j+1]);
                }
            }
        }
        return dp[len][d];
        
        
        
    }
}
