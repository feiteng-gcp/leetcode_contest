class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {
        int n = jobDifficulty.length;
        if (n < d) { return -1; }
        
        int[][] dp = new int[n][d+1];
        for (int[] dd : dp) {
            Arrays.fill(dd, -1);
        }
        
        return find(dp, 0, n, d, jobDifficulty);
    }
    
    private int find(int[][] dp, int start, int n, int d, int[] jd) {
        if (dp[start][d] >= 0) { return dp[start][d]; }
        
        if (d == 1) {
            int max = 0;
            for (int i=start; i<n; i++) {
                max = Math.max(max, jd[i]);
            }
            dp[start][d] = max;
            return max;
        }
        
        int min = Integer.MAX_VALUE;
        int max = 0;
        for (int i=start; i<=n-d; i++) {
            max = Math.max(max, jd[i]);
            min = Math.min(min, max + find(dp, i+1, n, d-1, jd));
        }
        dp[start][d] = min;
        return min;
    }
}
