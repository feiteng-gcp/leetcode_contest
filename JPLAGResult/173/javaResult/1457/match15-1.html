class Solution {

	public int minDifficulty(int[] jobDifficulty, int d) {
		int[][] dp = new int[jobDifficulty.length][d + 1];
		for (int i = 0; i < jobDifficulty.length; i++) {
			for (int j = 0; j <= d; j++)
				dp[i][j] = -1;
		}
		int a = maximizeSum(jobDifficulty, jobDifficulty.length, 0, d, dp);
		return a == 1000000000 ? -1 : a;
	}

	private int maximizeSum(int[] a, int n, int ind, int k, int[][] dp) {
		if (k == 0) {
			if (ind == n)
				return 0;
			else
				return 1000000000;
		} else if (ind == n)

			return 1000000000;
		else if (dp[ind][k] != -1)
			return dp[ind][k];
		else {
			int ans = 1000000000;
			int mini = a[ind];
			for (int i = ind; i < n; i++) {
				mini = Math.max(mini, a[i]);
				ans = Math.min(ans, maximizeSum(a, n, i + 1, k - 1, dp) + mini);
			}
			return dp[ind][k] = ans;
		}
	}
}
