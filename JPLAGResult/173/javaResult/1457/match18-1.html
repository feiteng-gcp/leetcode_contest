class Solution {
    int inf = 1000_000;
    public int solve(int[] j, 
int t, int d, int[][] memo) {
        if (d == -1 && t == -1) return 0;
        if (t < d || t < 0 || d < 0) return inf;
        if (memo[t][d] > -1) return memo[t][d];
        
int m = j[t];
        int ans = inf;
        for (int i = t; i >= 0; i--) {
            m = Math.max(m, j[i]);
            ans = Math.min(ans, m + solve(j, i - 1, d - 1, memo));
        }
        return memo[t][d] = ans;
    }
    
    public int minDifficulty(int[] j, int d) {
        int n = j.length;
        int[][] memo = 
new int[n][d];
        for (int i = 0; i < n; i++) Arrays.fill(memo[i], -1);
        int ans = solve(j, n - 1, d - 1, memo);
        return ans == inf ? -1 : ans;
    }
}
