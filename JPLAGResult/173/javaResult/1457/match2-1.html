class Solution {
    public int minDifficulty(int[] jobs, int d) {
        for(int[] mm : mxi) Arrays.fill(mm, -1);
        for(int[][] ff : f)
        {
            for(int[] fff : ff) Arrays.fill(fff, -1);
        }
        int n = jobs.length;
        if(n < d) return -1;
        int re = dfs(jobs, 0, jobs.length - 1, d);
        if(re >= 1 << 30) return -1;
        return re;
    }
    
    int[][][] f = new int[310][310][15];
    
    int dfs(int[] jobs, int from, int to, int d)
    {
        int n = jobs.length;
        if(d == 1) 
        {
            if(to != n - 1) return 1 << 30;
            return maxi(jobs, from, to);
        }
        if(f[from][to][d] >= 0) return f[from][to][d];
        int re = 1 << 30;
        for(int i = from; i < to; i++)
        {
            re = Math.min(re, maxi(jobs, from, i) + dfs(jobs, i + 1, to, d - 1));
        }
        return f[from][to][d] = re;
    }
    
    int[][] mxi = new int[310][310];
    
    int maxi(int[] ary, int lo, int hi)
    {
        if(lo == hi) return ary[lo];
        if(mxi[lo][hi] >= 0) return mxi[lo][hi];
        int mx = ary[lo];
        for(int i = lo; i <= hi; i++) mx = Math.max(mx, ary[i]);
        return mxi[lo][hi] = mx;
    }
}
