class Solution {
private static final int INF = 1_000_000_001;
    private static final int MOD = 1_000_000_007;
    private static final int[][] DIR = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

    
public int minDifficulty(int[] jobDifficulty, int d) {
        int n = jobDifficulty.length;
        if (n < d) return -1;

        int[][] dp = new int[n][d + 1];

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < d + 1; j++) {
                dp[i][j] = INF;
            }
        }

        for (int i = 0; i < n; i++) {
            for (int j = 1; j <= d && j <= i + 1; j++) {

                if (i == 0) {
                    dp[i][j] = jobDifficulty[i];
                    continue;
                }
                if (j == 1) {
                    dp[i][j] = Math.max(dp[i - 1][j], jobDifficulty[i]);
                    continue;
                }

                
int max = -1;
                for (int k = i; k > 0; k--) {
                    max = Math.max(max, jobDifficulty[k]);
                    dp[i][j] = Math.min(dp[i][j], dp[k - 1][j - 1] + max);
                    //System.out.printf("dp[%d][%d] = %d\n", i, j, dp[i][j]);
                }

                //System.out.printf("dp[%d][%d] = %d\n\n", i, j, dp[i][j]);
            }
        }
        return dp[n - 1][d];
    }
}
