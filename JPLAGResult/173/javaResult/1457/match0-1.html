class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {
        int N = jobDifficulty.length;
        if (N < d) {
            return -1;
        }
        int[][][] cache = new int[N][N][d + 1];
        return dp(jobDifficulty, 0, N - 1, d, cache);
    }
    
    private int dp(int[] job, int left, int right, int day, int[][][] cache) {
        if (day == 1) {
            return diff(job, left, right, cache);
        }
        if (cache[left][right][day] > 0) {
            return cache[left][right][day];
        }

        int min = 3000*100;
        for (int i = left; i < right; i++) {
            min = Math.min(min, dp(job, left, i, day - 1, cache) + diff(job, i + 1, right, cache));
        }
        cache[left][right][day] = min;
        return min;
    }
    
    private int diff(int[] job, int i, int j, int[][][] cache) {
        if (cache[i][j][1] > 0) {
            return cache[i][j][1];
        }
        int max = 0;
        for (int k = i; k <= j; k++) {
            max = Math.max(max, job[k]);
        }
        cache[i][j][1] = max;
        return max;
    }
}
