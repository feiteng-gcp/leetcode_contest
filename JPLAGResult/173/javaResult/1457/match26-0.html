class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {
        int n = jobDifficulty.length;
        if (n < d) return -1;
        
        // partition into d, minimize the sum of max
        int maxDiffi[][] = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                maxDiffi[i][j] = (i == j) ? jobDifficulty[i] : Math.max(maxDiffi[i][j-1], jobDifficulty[j]);
            }
        }
        
        
        int minSum[][] = new int[n][d];
        for (int i = 0; i < n; i++) 
            for (int k = 0; k < d; k++)
                minSum[i][k] = 3000050;
        
        for (int j = 0; j < n; j++) {
            minSum[j][0] = maxDiffi[0][j];
            for (int k = 1; k < d; k++) {
                for (int i = 0; i < j; i++) {
                    minSum[j][k] = Math.min(minSum[j][k], minSum[i][k-1]+maxDiffi[i+1][j]);
                }
            }
        }
        return minSum[n-1][d-1];
    }
}
