class Solution {
     int Max = 1000000000;
    
int[][] dp;

    public int minDifficulty(int[] a, int d) {
        dp = new int[a.length + 1][d + 1];
        for (int i = 0; i < dp.length; i++) {
            Arrays.fill(dp[i], -1);
        }
        int rt = dom(a, d, 0);
        return rt >= Max ? -1 : rt;

    }

    private int dom(int[] a, int d, int i) {
        int n = a.length;
        if (i == n) {
            if (d != 0) {
                return Max;
            } else {
                return 0;
            }
        }
        if (d == 0) {
            
return Max;
        }
        if (dp[i][d] != -1) {
            return dp[i][d];
        }
        int maxj = -Max;
        int min = Max;
        for (int j = i; j < n; j++) {
            maxj = Math.max(maxj, a[j]);
            int cur = maxj + dom(a, d - 1, j + 1);
            min = Math.min(min, cur);
        }
        dp[i][d] = min;
        return min;
    }
}
