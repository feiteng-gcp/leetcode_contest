class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {
        // 动态规划
        // 定义状态: f[i][j] 表示前 i 天完成前 j 项任务的最小和
        // 状态转移: f[i][j] = min{ f[i-1][j'] + maxd[j'+1][j] }
        // 初始状态: f[0][0] = 0
        // 最终答案: f[d][n]
        // 优化: 使用滚动数组优化空间; maxd 可以用 ST 表优化至 nlogn 时空

        if (jobDifficulty == null || jobDifficulty.length < d) {
            return -1;
        }
        final int INF = 0x3f3f3f3f;
        int n = jobDifficulty.length;
        int[][] f = new int[d + 1][n + 1];
        
int[][] maxd = new int[n][n];
        for (int i = 0; i <= d; i++) {
            Arrays.fill(f[i], INF);
        }
        f[0][0] = 0;
        for (int i = 0; i < n; i++) {
            maxd[i][i] = jobDifficulty[i];
        }
        for (int i = 1; i < n; i++) {
            
for (int j = 0; j + i < n; j++) {
                maxd[j][j + i] = Math.max(maxd[j][j + i - 1], jobDifficulty[j + i]);
            }
        }
        
for (int i = 1; i <= d; i++) {
            for (int j = i; j <= n; j++) {
                for (int k = i - 1; k < j; k++) {
                    f[i][j] = Math.min(f[i][j], f[i - 1][k] + maxd[k][j - 1]);  // f: 1-indexed maxd: 0-indexed
                }
            }
        }
        return f[d][n];
    }
}
