class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {
        if (jobDifficulty == null || jobDifficulty.length <= 0) return -1;
        int n = jobDifficulty.length;
        if (n < d) return -1;

        int[] sum = new int[n + 1];
        for (int i = 0; i < n; i++) {
            sum[i + 1] = sum[i] + jobDifficulty[i];
        }
        if (n == d) return sum[n];

        int[][] max = new int[n][n];
        for (int i = 0; i < n; i++) {
            int cur = jobDifficulty[i];
            for (int j = i; j < n; j++) {
                cur = Math.max(cur, jobDifficulty[j]);
                max[i][j] = cur;
            }
        }

        int[][] dp = new int[d + 1][n + 1];

         int maxC = jobDifficulty[0];
         for (int i = 1; i <= n; i++) {
             maxC = Math.max(maxC, jobDifficulty[i - 1]);
             dp[1][i] = maxC;
         }

        for (int day = 2; day <= d; day++) {
            for (int i = day; i <= n; i++) {
                dp[day][i] = Integer.MAX_VALUE;
                for (int j = day - 1; j < i; j++) {
                    dp[day][i] = Math.min(dp[day][i], dp[day - 1][j] + max[j][i - 1]);
                }
            }
        }
        return dp[d][n];
    }
}
