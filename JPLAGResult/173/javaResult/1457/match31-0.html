class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {
        if (jobDifficulty.length < d) return -1;
        
        int n = jobDifficulty.length;
        int[][] cache = new int[d+1][n];
        // int[][] lastMax = new int[d+1][n];
        for (int i = 0; i <= d; i++) Arrays.fill(cache[i], -1);
        
        cache[1][0] = jobDifficulty[0];
        for (int i = 1; i < n; i++) {
            cache[1][i] = Math.max(cache[1][i-1], jobDifficulty[i]);
        }
        // System.out.println(Arrays.toString(cache[1])); //
        
        for (int i = 2; i <= d; i++) {
            cache[i][i-1] = cache[i-1][i-2] + jobDifficulty[i-1];
            // lastMax[i][i-1] = jobDifficulty[i-1];
            for (int j = i; j < n; j++) {
                cache[i][j] = cache[i-1][j-1] + jobDifficulty[j];
                // lastMax[i][j] = jobDifficulty[j];
                // System.out.println(i + " " + j + " " + cache[i][j]); //
                
                // int lm = Math.max(lastMax[i][j-1], jobDifficulty[j]);
                // if (cache[i][j-1] - lastMax[i][j-1] + lm < cache[i][j]) {
                //     cache[i][j] = cache[i][j-1] - lastMax[i][j-1] + lm;
                //     lastMax[i][j] = lm;
                // }
                int curMax = jobDifficulty[j];
                for (int k = j-1; k >= i-1; k--) {
                    curMax = Math.max(curMax, jobDifficulty[k]);
                    int cur = cache[i-1][k-1] + curMax;
                    if (cur < cache[i][j]) cache[i][j] = cur;
                }
            }
            // System.out.println(Arrays.toString(lastMax[i])); // 
            // System.out.println(Arrays.toString(cache[i])); // 
        }
        
        return cache[d][jobDifficulty.length-1];
    }
}
