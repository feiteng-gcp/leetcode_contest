class Solution {
    int[][] dp;
    public int minDifficulty(int[] jb, int d) {
        int l = jb.length;
        if (l < d) return -1;
        dp = new int[l][d + 1];
        for (int[] i : dp) Arrays.fill(i, -1);
        return dfs(jb, 0, d);
    }
    int dfs(int[] jb, int idx, int d) {
        if (d == 1) {
            int max = 0;
            for (int i = idx; i < jb.length; i++) max = Math.max(jb[i], max);
            return max;
        }
        if (dp[idx][d] != -1) return dp[idx][d];
        int max = 0;
        int min = Integer.MAX_VALUE;
        for (int i = idx; i <= jb.length - d; i++) {
            max = Math.max(jb[i], max);
            min = Math.min(min, max + dfs(jb, i + 1, d - 1));
        }
        return dp[idx][d] = min;
    }
}
