class Solution {
    public int minDifficulty(int[] jobDifficulty, final int d) {
        if (jobDifficulty.length < d) {
            return -1;
        }
        
        final int[][] answers = new int[jobDifficulty.length][d];
        for(int i = jobDifficulty.length - 1, max = Integer.MIN_VALUE; i >= 0; i--) {
            max = Math.max(max, jobDifficulty[i]);
            answers[i][0] = max;
        }
        
        for (int day = 1; day < d; day++) {
            final int start =  jobDifficulty.length - 1 - day;
            
for (int i = start; i >= 0; i--) {
                int best = jobDifficulty[i] + answers[i + 1][day - 1], max = jobDifficulty[i];
                for (int j = i + 1; j <= start; j++) {
                    max = Math.max(max, jobDifficulty[j]);
                    final int candidate = max + answers[j + 1][day - 1];
                    
best = Math.min(best, candidate);
                }
                answers[i][day] = best;
            }
        }
        
        /*for(int i = 0; i < jobDifficulty.length; i++) {
            System.out.println(Arrays.toString(answers[i]));
        }*/
        
        return answers[0][d-1];
    }
}
