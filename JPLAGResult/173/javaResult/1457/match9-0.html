class Solution {
    
    public int minDifficulty(int[] jobs, int d) {
        int n = jobs.length;
        if (n < d) return -1;
        int[][] max = new int[n][n];
        for(int i = 0; i < n; i ++)
            max[i][i] = jobs[i];
        for(int i = 0; i < n; i ++)
        {
            int currMax = jobs[i];
            for(int j = i + 1; j < n; j ++)
            {
                currMax = Math.max(currMax, jobs[j]);
                max[i][j] = currMax;
            }
        }
        
        int[][] dp = new int[d + 1][n];
        
        for (int k = 1; k <= d; k++) {
            for (int i = k - 1; i < n; i++) {
                if(k == 1) dp[1][i] = max[0][i];
                else {
                    dp[k][i] = dp[k - 1][k - 2] + max[k - 1][i];
                    for (int j = k - 1; j < i; j++) {
                         dp[k][i] = Math.min(dp[k][i], dp[k - 1][j] + max[j + 1][i]);
                    }
                }
            }
        }
        return dp[d][n - 1];
    }
}
