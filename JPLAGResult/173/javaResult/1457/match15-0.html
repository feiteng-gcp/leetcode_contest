class Solution {

  public int minDifficulty(int[] jobDifficulty, int d) {
    int[][] dp = new int[jobDifficulty.length][d + 1];
    for (int i = 0; i < jobDifficulty.length; i++) {
      for (int j = 0; j <= d; j++)
        dp[i][j] = -1;
    }
    int a = f(jobDifficulty, jobDifficulty.length, 0, d, dp);
    return a == 1000000000 ? -1 : a;
  }

  private int f(int[] a, int n, int j, int k, int[][] d) {
    if (k == 0)
      if (j == n)
        return 0;
      else
        return 1000000000;
    else if (j == n)
      return 1000000000;
    else if (d[j][k] != -1)
      return d[j][k];
    else {
      int b = 1000000000;
      int c = a[j];
      for (int i = j; i < n; i++) {
        c = Math.max(c, a[i]);
        b = Math.min(b, f(a, n, i + 1, k - 1, d) + c);
      }
      return d[j][k] = b;
    }
  }
}
