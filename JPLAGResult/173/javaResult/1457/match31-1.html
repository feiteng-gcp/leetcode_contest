class Solution {
    public int minDifficulty(int[] jobs, int d) {
        if(jobs.length < d) return -1;
        
        int n = jobs.length;
        int[][] w = new int[n][n];
        for(int i = 0; i < n; i++) {
            w[i][i] = jobs[i];
            for(int j = i + 1; j < n; j++) {
                w[i][j] = jobs[j] > w[i][j - 1] ? jobs[j] : w[i][j - 1];
            }
        }
        
        int[][] dp = new int[n][d + 1];
        for(int i = 0; i < n; i++) {
            for(int j = 0; j <= d; j++) {
                dp[i][j] = -1;
            }
        
}
        
        for(int i = 0; i < n; i++) {
            dp[i][1] = w[0][i];
            for(int j = 2; j <= d && j - 1 <= i; j++) {
                for(int k = i - 1; k >= 0; k--) {
                    if(dp[k][j - 1] != -1) {
                        if(dp[k][j - 1] + w[k + 1][i] < dp[i][j] || dp[i][j] == -1) {
                            dp[i][j] = dp[k][j - 1] + w[k + 1][i];
                        }
                    
}
                }
            }
        }
        
        
        return dp[n - 1][d];
    }
}
