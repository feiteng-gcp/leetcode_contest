class Solution {
    long [][] dp;
    public int minDifficulty(int[] jobDifficulty, int d) {
        dp = new long [jobDifficulty.length][d+1];
        for (int i = 0; i<dp.length; ++i) Arrays.fill(dp[i], -1);
        long ans = recurse(jobDifficulty, 0, d);
        return (ans == Integer.MAX_VALUE) ? -1 : (int)ans;
    }
    public long recurse (int [] j, int pos, int d) {
        if (pos == j.length && d == 0) return 0;
        if (d == 0) return Integer.MAX_VALUE;
        if (pos == j.length) return Integer.MAX_VALUE;
        if (d > j.length - pos) return Integer.MAX_VALUE;
        
        if (dp[pos][d] == -1) {
            long jobs = 0;
            long best_ans = Integer.MAX_VALUE;
            for (int i = pos; i < j.length-d+1; ++i) {
               jobs = Math.max(j[i], jobs);
               best_ans = Math.min(best_ans, recurse(j, i+1, d-1)+jobs);
            }
            dp[pos][d] = best_ans;
        }
        return dp[pos][d];
        
    }
}
