class Solution {
    public int minDifficulty(int[] jobs, int d) {
        int n = jobs.length;
        if(n < d) return -1;
        int [][] max = new int[n][n], dp = new int[d + 1][n];
        for(int i = 0; i < n; i ++)
            max[i][i] = jobs[i];
        for(int i = 0; i < n; i ++)
        {
            int Max = jobs[i];
            for(int j = i + 1; j < n; j ++)
            {
                Max = Math.max(Max, jobs[j]);
                max[i][j] = Max;
            }
        }
        /*for(int i = 0; i < n; i ++)
            System.out.println(Arrays.toString(max[i]));*/
        for(int i = 1; i <= d; i ++)
        {
            for(int j = i - 1; j < n; j ++)
            {
                if(i == 1) dp[1][j] = max[0][j];
                else
                {
                    dp[i][j] = dp[i - 1][i - 2] + max[i - 1][j];
                    for(int k = i - 1; k < j; k ++)
                        dp[i][j] = Math.min(dp[i][j], dp[i - 1][k] + max[k + 1][j]);
                }
            }
        }
        /*for(int i = 0; i <= d; i ++)
            System.out.println(Arrays.toString(dp[i]));*/
        return dp[d][n - 1]; // dp[i][j] = max dp[i - 1][k] + max[k + 1][j]
    }
}
