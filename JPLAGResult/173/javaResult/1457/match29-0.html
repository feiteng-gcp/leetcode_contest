class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {
        int n = jobDifficulty.length;
        if(n < d){
            return -1;
        }
        Integer[][] dp = new Integer[n][d + 1];
        return dfs(jobDifficulty, n - 1, dp, d);
    }
    
    private int dfs(int[] jobDifficulty, int end, Integer[][] dp ,int d){
        if(dp[end][d] != null){
            return dp[end][d];
        }
        int res = Integer.MAX_VALUE;
        int maxV = jobDifficulty[end];
        if(d == 1){
            for(int i = 0 ;i <= end; i++){
                maxV = Math.max(maxV, jobDifficulty[i]); 
            }
            return maxV;
        }
        for(int i = end; i >= d - 1; i--){
            maxV = Math.max(maxV, jobDifficulty[i]);
            res = Math.min(res, maxV + dfs(jobDifficulty, i - 1, dp, d - 1));
        }
        dp[end][d] = res;
        return res;
    }
}
