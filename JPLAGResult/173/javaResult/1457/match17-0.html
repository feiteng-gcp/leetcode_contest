class Solution {
    int[][] dp;
    public int minDifficulty(int[] jobDifficulty, int d) {
        
        dp = new int[jobDifficulty.length + 1][d + 1];
        
        for (int[] t : dp) {
            Arrays.fill(t, -1);
        }
        
        int result = helper(jobDifficulty, d, 0, d);
        
        return result >= 10000000 ? -1 : result;
    }
    
    private int helper(int[] jobs, int d, int index, int remain) {
        if (remain == 0) {
            if (index == jobs.length) {
                return 0;
            }
            return 10000000;
        }
        
        if (index == jobs.length) {
            if (remain == 0) {
                return 0;
            } 
            return 10000000;
        }
        
        if (dp[index][remain] != -1) {
            return dp[index][remain];
        }
        
        int max = jobs[index];
        int result = 10000000;
        for (int i = index; i < jobs.length; i++) {
            
            max = Math.max(jobs[i], max);
            
            int later = helper(jobs, d, i + 1, remain - 1);
            
            result = Math.min(result, later + max);
        }
        dp[index][remain] = result;
        return result;
    }
}
