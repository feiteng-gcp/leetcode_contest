class Solution {
    
    int[][] cache;
    
    public int minDifficulty(int[] jobDifficulty, int d) {
        if(d > jobDifficulty.length) return -1;
        cache = new int[jobDifficulty.length + 1][d + 1];
        for(int[] c : cache) Arrays.fill(c, -1);
        return solve(jobDifficulty, 0, d);
    }
    
    int solve(int[] jobDifficulty, int index, int d) {
        if(d == 0 && index == jobDifficulty.length) {
            return 0;
        }
        if(d == 0) {
            return Integer.MAX_VALUE;
        }
        if(cache[index][d] != -1) return cache[index][d];
        int currDifficulty = 0;
        int output = Integer.MAX_VALUE;
        for(int i = index; i <= jobDifficulty.length - d; i++) {
            currDifficulty = Math.max(currDifficulty, jobDifficulty[i]);
            int sub = solve(jobDifficulty, i + 1, d - 1);
            if(sub != Integer.MAX_VALUE) {
                output = Math.min(output, currDifficulty + sub);            
            }
        }
        cache[index][d] = output;
        return output;
    }
}
