class Solution {
    private int[][] dp;
    public int minDifficulty(int[] jobDifficulty, int d) {
        
if (jobDifficulty.length < d) return -1;
        dp = new int[jobDifficulty.length][d];
        for (int i = 0; i < dp.length; i++) Arrays.fill(dp[i], -1);
        return helper(jobDifficulty, jobDifficulty.length - 1, d);
    }
    private int helper(int[] jobDifficulty, int i, int d) {
        if (i == -1) return 0;
        if (--d == 0) {
            int cd = 0;
            for (int j = 0; j <= i; j++) {
                cd = Math.max(cd, jobDifficulty[j]);
            }
            dp[i][d] = cd;
            
return cd;
        }
        if (dp[i][d] != -1) return dp[i][d];
        int cd = 0;
        int res = Integer.MAX_VALUE;
        for (int j = i; j >= d; j--) {
            cd = Math.max(cd, jobDifficulty[j]);
            res = Math.min(res, cd + helper(jobDifficulty, j - 1, d));
        }
        dp[i][d] = res;
        return res;
    }
}
