class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {
        int len = jobDifficulty.length;
        if(len < d){
            return -1;
        }
        
        int[][] dp = new int[d][len];
        
        int tmp=0;
        
for(int i=0;i<d;i++){
            for(int j=0;j<len;j++){
                dp[i][j]=Integer.MAX_VALUE;
            }
        }
        for(int i=0;i<len;i++){
            tmp=Math.max(tmp,jobDifficulty[i]);
            dp[0][i]=tmp;
        }
        if(d==1){
            return dp[0][len-1];
        }
        
        
for(int i=1;i<d;i++){
            for(int j=i;j<len;j++){
                int aaa=0;
                for(int k=j;k>=i;k--){
                    aaa=Math.max(jobDifficulty[k],aaa);         
                    dp[i][j]=Math.min(dp[i][j],aaa+dp[i-1][k-1]);
                }
                
            }
        }
        
        //for(int i=0;i<d;i++){
        //    System.out.println(Arrays.toString(dp[i]));
        //}
        
        return dp[d-1][len-1];
    }
}
