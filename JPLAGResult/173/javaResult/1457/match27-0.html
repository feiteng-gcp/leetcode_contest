class Solution {
    public int minDifficulty(int[] jobDifficulty, int d) {
        int[][] memo = new int[jobDifficulty.length][d+1];
        for (int[] me : memo) Arrays.fill(me, Integer.MAX_VALUE);
        int val = dfs(jobDifficulty, 0, d, memo);
        if (val == Integer.MAX_VALUE) return -1;
        return val;
    }
    
    int dfs(int[] jobDifficulty, int idx, int d, int[][] memo) {
        
        if (idx == jobDifficulty.length && d == 0) return 0;
        if (idx >= jobDifficulty.length || d <= 0) return Integer.MAX_VALUE;
        if (memo[idx][d] != Integer.MAX_VALUE) return memo[idx][d];
        if (idx + d > jobDifficulty.length) return Integer.MAX_VALUE;
        int max = 0;
        int res = Integer.MAX_VALUE;
        for (int i = idx; i < jobDifficulty.length; i++) {
            max = Math.max(max, jobDifficulty[i]);
            int val = dfs(jobDifficulty, i + 1, d - 1, memo);
            if (val != Integer.MAX_VALUE) {
                res = Math.min(val + max, res);
            }
        }
        memo[idx][d] = res;
        return res;
    }
}
