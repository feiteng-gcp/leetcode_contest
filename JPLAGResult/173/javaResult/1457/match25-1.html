class Solution {
    public int minDifficulty(int[] jd, int d) {
        int n = jd.length;
        if (n<d)
            return -1;
        int[][] dp = new int[d+1][n+1];
        for (int i = 0; i <= d; ++i) {
            for (int j = 0; j <= n; ++j) {
                dp[i][j] = 100000000;
            }
        }
        dp[0][0] = 0;
        int[][] max = new int[n][n];
        for (int i = 0; i < n; ++i) {
            max[i][i] = jd[i];
            
for (int j = i+1; j < n; ++j) {
                max[i][j] = Math.max(max[i][j-1], jd[j]);
            }
        }
        for (int i = 1; i <= d; ++i) {
            for (int j = i; j <= n; ++j) {
                
for (int k = i-1; k < j; ++k) {
                    dp[i][j] = Math.min(dp[i][j], dp[i-1][k]+max[k][j-1]);
                    //System.out.println(i+" "+j+" "+dp[i][j]);
                }
            }
        }
        return dp[d][n];
    }
}
