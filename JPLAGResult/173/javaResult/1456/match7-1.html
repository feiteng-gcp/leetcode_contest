class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int[][] map = new int[n][n];
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                if(i == j) map[i][j] = 0; else map[i][j] = 10000;
            }
        
}
        for(int i = 0; i < edges.length; i++) {
            int u = edges[i][0], v = edges[i][1], w = edges[i][2];
            map[u][v] = w;
            map[v][u] = w;
        }
        
        for(int k = 0; k < n; k++) {
            for(int i = 0; i < n; i++) {
                for(int j = 0; j < n; j++) {
                    if(map[i][k] + map[k][j] < map[i][j]) map[i][j] = map[i][k] + map[k][j];
                }
            }
        }
        
        int ret = -1, num = 200;
        for(int i = 0; i < n; i++) {
            int cnt = 0;
            for(int j = 0; j < n; j++) {
                //System.out.print(map[i][j] + " ");
                if(map[i][j] <= distanceThreshold) {
                    cnt++;
                }
            }
            System.out.println();
            
if(cnt <= num) {
                ret = i;
                num = cnt;
            }
        }
        
        return ret;
    }
}
