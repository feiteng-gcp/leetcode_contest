class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int[][] dis = new int[n][n];
        for(int[] d: dis)
            Arrays.fill(d, Integer.MAX_VALUE);
        for(int i = 0; i < n; ++i)
            dis[i][i] = 0;
        for(int[] edge: edges) {
            
dis[edge[0]][edge[1]] = edge[2];
            dis[edge[1]][edge[0]] = edge[2];
        }
        for(int v = 0; v < n; ++v)
            for(int i = 0; i < n; ++i)
                for(int j = 0; j < n; ++j)
                    if(dis[i][v] != Integer.MAX_VALUE && dis[v][j] != Integer.MAX_VALUE)
                        dis[i][j] = Math.min(dis[i][j], dis[i][v] + dis[v][j])
;
        int max = 0;
        int maxc = Integer.MAX_VALUE;
        for(int i = 0; i < n; i++) {
            int count = 0;
            for(int j = 0; j < n; j++)
                if(dis[i][j] <= distanceThreshold)
                    count++
;
            if(count <= maxc) {
                maxc = count;
                max = i;
            }
        }
        return max;
    }
}
