class Solution {
    public int findTheCity(int n, int[][] es, int d) {
        int dp[][] =new int[n][n];
        for(int k=0;k<n;++k){
            for(int i=0;i<n;++i){
                dp[k][i] =100000000;       
            }
            dp[k][k] = 0;
        }
        for(int e[]:es){
            dp[e[1]][e[0]] = dp[e[0]][e[1]] = Math.min(dp[e[0]][e[1]],e[2]);
        }
        
        
        for(int k=0;k<n;++k){
            for(int i=0;i<n;++i){
                for(int j=0;j<n;++j){
                    dp[i][j] = Math.min(dp[i][j],dp[i][k]+dp[k][j]);
                    dp[j][i] = dp[i][j];
                }
            }
        }
        int arr[][] = new int[n][2];
        for(int k=0;k<n;++k){
            int c = 0;
            for(int i=0;i<n;++i){
                if(i==k) continue;
                if(dp[k][i]<=d){
                    c++;
                }
            }
            arr[k]  = new int[]{c,k};
        }
        Arrays.sort(arr,(x,y)->{
            if(x[0]!=y[0]){
                return x[0] - y[0];
            }else{
                return y[1] - x[1];
            }
            
        });
        
        return arr[0][1];
            
            
        
        
        
    }
}
