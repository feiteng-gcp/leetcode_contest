class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int [][] graph = new int[n][n];
        for(int [] edge: edges){
            graph[edge[0]][edge[1]] = edge[2];
            graph[edge[1]][edge[0]] = edge[2];
        }
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                if(graph[i][j] == 0){
                    graph[i][j] = Integer.MAX_VALUE;
                }
            }
        }
        long [][] dp = new long[n][n];
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                dp[i][j] = graph[i][j];
            }
        
}
        for(int k = 0; k < n; k++){
            for(int i = 0; i < n; i++){
                for(int j = 0; j < n; j++){
                    if(dp[i][k] + dp[k][j] < dp[i][j]){
                        dp[i][j] = dp[i][k] + dp[k][j];
                    }
                }
            }
        }
        int answer = n;
        int index = 0;
        for(int i = 0; i < n; i++){
            int count = 0;
            for(int j = 0; j < n; j++){
                if(i == j){
                    continue;
                }
                if(dp[i][j] <= distanceThreshold){
                    count++;
                
}
            }
            if(count <= answer){
                answer = count;
                index = i;
            }
        }
        return index;
    }
}
