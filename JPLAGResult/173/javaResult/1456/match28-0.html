class Solution {
    public int findTheCity(int n, int[][] edges, int T) {
        int[][] dist = new int[n][n];
        for(int i=0; i<n; i++) Arrays.fill(dist[i], 10001);
        for(int i=0; i<n; i++) dist[i][i] = 0;
        for(int[] edge: edges) {
            int u = edge[0];
            int v = edge[1];
            dist[u][v] = edge[2];
            dist[v][u] = edge[2];
        }
        for(int k=0; k<n; k++) {
            for(int i=0; i<n; i++) {
                for(int j=0; j<n; j++) {
                    dist[i][j] = Math.min( dist[i][j], dist[i][k] + dist[k][j] );    
                }
            }    
        }
        int min = Integer.MAX_VALUE;
        int ind = -1;
        
        for(int i=0; i<n; i++) {
            System.out.println(Arrays.toString(dist[i]));
            int count = 0;
            for(int j=0; j<n; j++) {
                if(i==j) continue;
                if(dist[i][j]<=T) count++;
            }
            if(count<=min) {
                min = count;
                ind = i;
            }
        }
        return ind;
    }
}
