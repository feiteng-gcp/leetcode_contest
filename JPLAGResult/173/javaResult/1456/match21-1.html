class Solution {
    int Max = 1000000000;

    public int findTheCity(int n, int[][] es, int dt) {

        int[][] dist = new int[n][n];
        for (int i = 0; i < n; i++) {
            Arrays.fill(dist[i], Max);
        }
        for (int[] e : es) {
            dist[e[0]][e[1]] = e[2];
            
dist[e[1]][e[0]] = e[2];
        }
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    dist[i][j] = Math.min(dist[i][j], dist[i][k] + dist[k][j]);
                
}
            }
        }
        int min = n + 1;
        int mincity = -1;
        for (int i = 0; i < n; i++) {
            int r = 0;
            for (int j = 0; j < n; j++) {
                if (j == i) {
                    continue;
                }
                
if (dist[i][j] <= dt) {
                    r++;
                }
            }
            if (r <= min) {
                min = r;
                mincity = i;
            }
        }
        return mincity;
    }
}
