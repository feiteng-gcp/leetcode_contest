class Solution {
public int findTheCity(int n, int[][] edges, int dist) {
        // mst? 25 * 1e6
        // bfs at each point
        Map<Integer, Map<Integer, Integer>> map = new HashMap<>();
        for(int i = 0; i < n; i++) map.put(i, new HashMap<>());
        for(int[] e : edges)
        {
            int from = e[0], to = e[1], w = e[2];
            map.get(from).put(to, w);
            map.get(to).put(from, w);
        }
        
        int re = 0, count = 1 << 30;
        for(int i = 0; i < n; i++)
        {
            int c = bfs(i, map, dist);
            // System.out.printf("%d %d\n", i, c);
            if(count > c)
            {
                count = c;
                re = i;
            }
            if(count == c)
            {
                re = i;
            }
        }
        
        return re;
    }
    
    int bfs(int cur, Map<Integer, Map<Integer, Integer>> map, int t)
    {
        int[] d = new int[map.keySet().size()];
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[]{cur, 0});
        Arrays.fill(d, 1 << 30);
        int count = 0;
        while(!q.isEmpty())
        {
            int[] c = q.poll();
            if(d[c[0]] <= c[1]) continue;
            d[c[0]] = c[1];
            for(int next : map.get(c[0]).keySet())
            {
                int w = map.get(c[0]).get(next);
                if(d[next] <= c[1] + w) continue;
                // d[next] = c[1] + w;
                // if(d[next] > t) continue;
                // count++;
                q.offer(new int[]{next, c[1] + w});
            }
        }
        for(int k : d) if(k <= t && k > 0) count++;
        // System.out.printf("%s \n", Arrays.toString(d));
        return count;
    }   
}
