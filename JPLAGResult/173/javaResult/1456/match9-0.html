class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int[][] help = new int[n][n];
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                help[i][j]=1000000;
            }
        }
        int len = edges.length;
        
        for(int i=0;i<len;i++){
            int from = edges[i][0];
            int to = edges[i][1];
            int weight = edges[i][2];
            help[from][to]=weight;
            help[to][from]=weight;
        }
        
        for(int k=0;k<n;k++){
            for(int i=0;i<n;i++){
                for(int j=0;j<n;j++){
                    if(help[i][j]>help[i][k]+help[k][j]){
                        help[i][j]=help[i][k]+help[k][j];
                    }
                }
            }
        }
        
        //for(int i=0;i<n;i++){
        //    System.out.println(Arrays.toString(help[i]));
        //}
        int ans = n - 1;
        int smallest = 10000;
        
        for(int i=n-1;i>=0;i--){
            int tmp=0;
            //System.out.println(i);
            for(int j=0;j<n;j++){
                if(i!=j){
                    //System.out.println(j);
                    if(help[i][j]<=distanceThreshold){
                        tmp++; 
                    }   
                }
            }
            //System.out.println(tmp);
            //System.out.println("---------------");
            if(tmp<smallest){
                smallest=tmp;
                ans=i;
            }
        }
        
        return ans;
    }
}
