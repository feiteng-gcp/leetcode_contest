class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int[][] distanceMatrix = new int[n][n];
        for(int i=0; i<n; i++) {
            Arrays.fill(distanceMatrix[i], Integer.MAX_VALUE);
            distanceMatrix[i][i] = 0;
        }
        
        for(int[] edge : edges) {
            int s = edge[0];
            int t = edge[1];
            int w = edge[2];
            distanceMatrix[s][t] = w;
            distanceMatrix[t][s] = w;
        }
        
        
        for(int k=0; k<n; k++) {
            for(int i=0; i<n; i++) {
                for(int j=0; j<n; j++) {
                    int current = distanceMatrix[i][j];
                    int possibleSum = getSum(distanceMatrix[i][k], distanceMatrix[k][j]);
                    if(current > possibleSum) {
                        distanceMatrix[i][j] = possibleSum;
                        distanceMatrix[j][i] = possibleSum;
                    
}
                }
            }
        }
        
        int totalMin = Integer.MAX_VALUE;
        int city=-1;
        
        for(int i=0; i<n; i++) {
            int tempTotalMin = 0;
            for(int j=0; j<n; j++) {
                if(distanceMatrix[i][j] <= distanceThreshold) {
                    tempTotalMin++;
                
}
            }
            if(totalMin >= tempTotalMin) {
                totalMin = tempTotalMin;
                city = i;
            }
        }
        
        return city;
    }

    private int getSum(int a, int b) {
        int c = a+b;
        if(c<0) {
            return Integer.MAX_VALUE;
        }
        return c;
    }
    
}
