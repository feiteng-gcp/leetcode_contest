class Solution {

  public int findTheCity(int n, int[][] edges, int distanceThreshold) {
    int e[][] = new int[n][n], d[][] = new int[n][n], u = 0, m = Integer.MAX_VALUE;
    for (int[] edge : edges) {
      e[edge[0]][edge[1]] = e[edge[1]][edge[0]] = edge[2];
    }
    for (int i = 0; i < n; i++) {
      for (int j = 0; j < n; j++) {
        d[i][j] = i == j || e[i][j] != 0 ? e[i][j] : Integer.MAX_VALUE;
      }
    }
    for (int k = 0; k < n; k++) {
      for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
          if (d[i][k] != Integer.MAX_VALUE && d[k][j] != Integer.MAX_VALUE) {
            d[i][j] = Math.min(d[i][j], d[i][k] + d[k][j]);
          }
        }
      }
    }
    for (int i = n - 1; i >= 0; i--) {
      int c = -1;
      for (int j = 0; j < n; j++) {
        if (d[i][j] <= distanceThreshold) {
          c++;
        }
      }
      if (c < m) {
        m = c;
        u = i;
      }
    }
    return u;
  }
}
