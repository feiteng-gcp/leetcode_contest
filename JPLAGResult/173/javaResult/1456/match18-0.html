class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        List<int[]>[] adjList = new List[n];
        for (int i = 0; i < n; i++) {
            adjList[i] = new ArrayList();
        }
        for (int[] e : edges) {
            int from = e[0], to = e[1], weight = e[2];
            adjList[from].add(new int[]{to, weight});
            adjList[to].add(new int[]{from, weight});
        }
        int minCount = Integer.MAX_VALUE;
        int res = -1;
        for (int i = 0; i < n; i++) {
            int[] dist = new int[n];
            Arrays.fill(dist, Integer.MAX_VALUE);
            PriorityQueue<Integer> pq = new PriorityQueue<Integer>((a, b) -> dist[a] - dist[b]);
            dist[i] = 0;
            pq.offer(i);
            while (!pq.isEmpty()) {
                int cur = pq.poll();
                if (dist[cur] > distanceThreshold) {
                    break;
                }
                for (int[] neigh : adjList[cur]) {
                    if (dist[cur] + neigh[1] < dist[neigh[0]]) {
                        dist[neigh[0]] = Math.min(dist[neigh[0]], dist[cur] + neigh[1]);
                        pq.offer(neigh[0]);
                    }
                }
            }
            int count = 0;
            for (int d : dist) {
                if (d <= distanceThreshold && d > 0) {
                    count++;
                }
            }
            if (count <= minCount) {
                res = i;
                minCount = count;
            }
        }
        return res;
    }
    
    class Node {
        int city, dist;
        Node(int city, int dist) {
            this.city = city;
            this.dist = dist;
        }
    }
    
}
