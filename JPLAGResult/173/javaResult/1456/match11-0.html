class Solution {
    public int findTheCity(int n, int[][] edges, int dt) {
        int inf = 100 * 10000;
        int[][] w = new int[n][n];
        for(int i = 0; i < n; ++i) {
            Arrays.fill(w[i], inf);
        }
        for(int[] edge: edges) {
            int x = edge[0];
            int y = edge[1];
            w[x][y] = Math.min(w[x][y], edge[2]);
            w[y][x] = Math.min(w[y][x], edge[2]);
        } 
        for(int k = 0; k < n; ++k) {
            for(int i = 0; i < n; ++i) {
                for(int j = 0; j < n; ++j) {
                    w[i][j] = Math.min(w[i][j], w[i][k] + w[k][j]);
                }
            }
        }
        int res = -1;
        int stf = n + 1;
        // for(int i = 0; i < n; ++i) {
        //     for(int j = 0; j < n; ++j) {
        //         System.out.print(w[i][j] + " ");
        //     }
        //     System.out.println();
        // }
        for(int i = 0; i < n; ++i) {
            int num = 0;
            for(int j = 0; j < n; ++j) {
                if (i != j && w[i][j] <= dt) {
                    ++num;
                }
            }
            if (num <= stf) {
                stf = num;
                res = i;
            }
        }
        return res;
    }
}
