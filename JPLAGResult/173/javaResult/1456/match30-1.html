class Solution {
    public static int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int dis[][] = new int[n][n];

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i == j) {
                    dis[i][j] = 0;
                    continue;
                }
                dis[i][j] = Integer.MAX_VALUE;
            }
        }

        for (int i = 0; i < edges.length; i++) {
            
int s = edges[i][0];
            int e = edges[i][1];
            dis[s][e] = edges[i][2];
            dis[e][s] = edges[i][2];
        }

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k < n; k++) {
                    if (dis[j][i] == Integer.MAX_VALUE || dis[i][k] == Integer.MAX_VALUE) {
                        continue;
                    }
                    dis[j][k] = Math.min(dis[j][k], dis[j][i] + dis[i][k]);
                }
            }
        }

        for(int k=0; k<n; k++){
            for(int i=0; i<n; i++){
                for(int j=0; j<n; j++){
                    int temp = (dis[i][k] == Integer.MAX_VALUE || dis[k][j] == Integer.MAX_VALUE) ? Integer.MAX_VALUE : dis[i][k] + dis[k][j];
                    
if(dis[i][j] > temp){
                        dis[i][j] = temp;
                    }
                }
            }
        }


        int min = Integer.MAX_VALUE;
        int id = -1;
        for (int i = 0; i < n; i++) {
            int cnt = 0;
            for (int j = 0; j < n; j++) {
                if (dis[i][j] <= distanceThreshold) {
                    cnt++;
                }
            }

            if (cnt <= min) {
                id = i;
                min = cnt;
            }
        }

        return id;
    }

    public static void main(String args[]) {
        findTheCity(5, new int[][]{{0, 1, 2}, {0, 4, 8}, {1, 2, 3}, {1, 4, 2}, {2, 3, 1}, {3, 4, 1}}, 2);
    }
}
