class Solution {
    public int findTheCity(int n, int[][] edges, int thres) {
        int[][] floyd = new int[n][n];
        for(int i=0;i<n;i++) {
            Arrays.fill(floyd[i], (int)1e7);
            floyd[i][i] = 0;
        }
        for(int[] edge : edges) {
            int a = edge[0], b = edge[1];
            floyd[a][b] = floyd[b][a] = edge[2];
        
}
        
        for(int k=0;k<n;k++){
            for(int i=0;i<n;i++){
                for(int j=0;j<n;j++){
                    int nextDist = floyd[i][k] + floyd[k][j];
                    if(nextDist < floyd[i][j]) floyd[i][j] = nextDist
;
                }
            }
        }
        
        int best = 0;
        int bestCnt = n+1;
        for(int i=0;i<n;i++){
            int cnt = 0;
            for(int j=0;j<n;j++){
                if(i!=j && floyd[i][j] <= thres) cnt++
;
            }
            if(cnt <= bestCnt) {
                best = i;
                bestCnt = cnt;
            }
        }
        return best;
    }
}
