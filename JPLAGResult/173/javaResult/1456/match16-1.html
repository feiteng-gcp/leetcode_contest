class Solution {
    private static final int INF = 123456789;
    
    
public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int[][] D = new int[n][n];
        for (int i = 0; i < n; ++i) {
            Arrays.fill(D[i], INF);
        }
        for (int[] e : edges) {
            int u = e[0];
            int v = e[1];
            int w = e[2];
            D[u][v] = w;
            D[v][u] = w;
        }
        for (int i = 0; i < n; ++i) {
            
D[i][i] = 0;
        }
        for (int k = 0; k < n; ++k) {
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < n; ++j) {
                    if (D[i][j] > D[i][k] + D[k][j]) {
                        D[i][j] = D[i][k] + D[k][j];
                    }
                }
            }
        }
        int best = INF;
        int bestCity = -1;
        for (int i = n - 1; i >= 0; --i) {
            int count = 0;
            for (int j = 0; j < n; ++j) {
                if (D[i][j] <= distanceThreshold) {
                    ++count;
                }
            }
            if (count < best) {
                best = count;
                bestCity = i;
            }
        }
        return bestCity;
    }
}
