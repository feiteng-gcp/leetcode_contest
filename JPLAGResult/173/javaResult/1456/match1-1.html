class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        Map<Integer, Map<Integer, Integer>> graph = new HashMap<>();
        for (int i = 0; i < n; i++) {
            graph.put(i, new HashMap<>());
        }
        for (int[] edge : edges) {
            graph.get(edge[0]).put(edge[1], edge[2]);
            graph.get(edge[1]).put(edge[0], edge[2]);
        }
        int globalMax = Integer.MAX_VALUE;
        int res = -1;
        for (int i = n - 1; i >= 0; i--) {
            int curMax = helper(graph, n, i, distanceThreshold);
            if (curMax < globalMax) {
                globalMax = curMax;
                res = i;
            }
        }
        return res;
    }
    
    public int helper(Map<Integer, Map<Integer, Integer>> graph, int N, int K, int max) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> (a[1] - b[1]));
        // 0- node number, 1-distance to K
        pq.offer(new int[]{K, 0});
        int[] dist = new int[N];
        Arrays.fill(dist, Integer.MAX_VALUE);
        dist[K] = 0;
        Set<Integer> visited = new HashSet<>();
        while (visited.size() < N && !pq.isEmpty()) {
            int[] u = pq.poll();
            visited.add(u[0]);
            //System.out.println(u[0]);
            Map<Integer, Integer> neiWeight = graph.get(u[0]);
            for (Map.Entry<Integer, Integer> nei : neiWeight.entrySet()) {
                int v = nei.getKey();
                int w = nei.getValue();
                if (dist[v] > dist[u[0]] + w) {
                    dist[v] = dist[u[0]] + w;
                    pq.offer(new int[]{v, dist[v]});
                }
            }
        }
        int reach = 0;
        for (int i = 0; i < N; i++) {
            if (dist[i] <= max) {
                reach++;
            }
        }
        return reach;
    }
}
