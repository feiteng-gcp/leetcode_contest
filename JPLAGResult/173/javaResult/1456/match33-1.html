class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int[][] d = new int[n][n];
        for (int[] e: edges) {
            
int u = e[0], v = e[1], w = e[2];
            d[u][v] = d[v][u] = w;
        }
        for (int k = 0; k < n; k++) {
            for (int u = 0; u < n; u++) {
                for (int v = 0; v < n; v++) {
                    if (d[u][k] > 0 && d[k][v] > 0 && (d[u][v] == 0 || d[u][k]+d[k][v] < d[u][v])) {
                        
d[u][v] = d[u][k]+d[k][v];
                    }
                }
            }
        }
        //System.out.println(Arrays.deepToString(d));
        int bestCount = Integer.MAX_VALUE, bestU = 0;
        for (int u = n-1; u >= 0; u--) {
            int count = 0;
            for (int v = 0; v < n; v++) {
                if (v != u && d[u][v] <= distanceThreshold) 
{
                    count++;
                }
            }
            if (count < bestCount) {
                bestCount = count;
                bestU = u;
            }
        }
        return bestU;
    }
}
