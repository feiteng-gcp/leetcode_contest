class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int dp[][] = new int[n][n];
        for(int i=0; i< n; i++) {
            for(int j=0; j < n; j++) {
                dp[i][j] = -1;
            }
            dp[i][i] = 0;
        }
        for(int i=0; i< edges.length; i++) {
            int from = edges[i][0];
            int to = edges[i][1];
            dp[from][to] = dp[to][from] = edges[i][2];
        }
        for(int k = 0; k < n; k++) {
            for(int i=0; i<n; i++) {
                if(dp[i][k] == -1) continue;
                for(int j =0; j < n; j++) {
                    if(dp[k][j] == -1) continue;
                    int dis = dp[i][k] + dp[k][j];
                    if(dp[i][j] == -1 || dis < dp[i][j]) {
                        dp[i][j] = dp[j][i] = dis;
                    }
                }
            }
        }
        int min = n + 1;
        int ans = 0;
        for(int i=0; i< n; i++) {
            int cnt = 0;
            for(int j=0; j < n; j++) {
                if(i == j) continue;
                if(dp[i][j] != -1 && dp[i][j] <= distanceThreshold) {
                    cnt++;
                }
            }
            // System.out.println("pos: "+ i+" "+ cnt+" "+ min);
            if(cnt <= min) {
                min = cnt;
                ans = i;
            }
        }
        return ans;
    }
}
