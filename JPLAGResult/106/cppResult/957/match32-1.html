class Solution {
public:
    int minAddToMakeValid(string s) {
        int x = 0, mi = 0;
        for (char c : s){
            if (c == '(') x++;
            else x--;
            mi = min(mi, x);
        }
        int ret = 0;
        if (mi < 0){
            ret += -mi;
            x += -mi;
        }
        ret += x;
        return ret;
    }
};
