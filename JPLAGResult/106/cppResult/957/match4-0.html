class Solution {
public:
    int minAddToMakeValid(string S) {
        int res = 0;
        int size = S.size();
        stack<char> s;
        
        for (int i = 0; i < size; i++) {
            if (S[i] == '(') {
                s.push(S[i]);
            } else {
                if (s.size() == 0) {
                    res++;
                } else {
                    s.pop();
                }
            }
        }
        
        res += s.size();
        
        return res;
    }
};
