class Solution {
public:
    int minAddToMakeValid(string S) {
        int len = S.size();
        int ans = 0, left = 0;
        
        for (int i = 0; i < len; i ++){
            if (S[i] == '('){
                left ++;
            }
            else{
                if(left > 0)    left --;
                else            ans ++;
            }
        }
        
        ans += left;
        
        return ans;
    }
};
