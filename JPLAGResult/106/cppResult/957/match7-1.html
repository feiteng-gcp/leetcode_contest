class Solution {
public:
    int minAddToMakeValid(string S) {
        int sum=0;
        int res=0;
        for(int i=0;i<S.size();i++){
            if(S[i]=='('){
                sum+=1;
            }
            else{
                sum--;
                if(sum==-1){
                    res++;
                    sum++;
                }
            }
        }
        if(sum>0) res+=sum;
        return res;
    }
};
