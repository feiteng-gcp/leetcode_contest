class Solution {
public:
    int minAddToMakeValid(string S) {
        int mi = 0;
        int cur = 0;
        for(int i=0; i<S.size(); i++) {
            if(S[i]=='(')
                cur++;
            else
                cur--;
            if(cur<0) {
                cur = 0;
                mi++;
            }
        }
        return mi + cur;
    }
};
