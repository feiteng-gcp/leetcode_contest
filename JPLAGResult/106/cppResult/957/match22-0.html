class Solution {
public:
    int minAddToMakeValid(string S) {
        int res = 0;
        int stack = 0;
        for(int i = 0; i < S.size(); i++) {
            if(S[i] == '(') stack ++;
            else stack --;
            
            if(stack < 0) {
                res ++;
                stack = 0;
            }
        }
        
        res += stack;
        
        return res;
    }
};
