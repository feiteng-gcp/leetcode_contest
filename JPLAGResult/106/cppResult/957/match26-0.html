class Solution {
public:
    int minAddToMakeValid(string S) {
        int need = 0;
        int left = 0;
        for (int i = 0; i < S.length(); ++i) {
            if (S[i] == '(') left++;
            else {
                if (left) --left;
                else {
                    need++;
                }
            }
        }
        return need + left;
    }
};
