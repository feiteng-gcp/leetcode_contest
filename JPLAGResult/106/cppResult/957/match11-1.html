class Solution {
public:
    int minAddToMakeValid(string S) {
        int ans = 0;
        int cnt = 0;
        for(int i = 0;i < S.size();++i) {
            if(S[i] == '(') {
                ++cnt;
            } else {
                --cnt;
            }
            
            if(cnt == -1) {
                ++ans;
                cnt = 0;
            }
        }
        if(cnt > 0) {
            ans += cnt;
        }
        
        return ans;
    }
};
