class Solution {
public:
    int minAddToMakeValid(string S) {
        int to_add = 0;
        int curr_stack = 0;
        for (int i = 0; i < S.size(); i++) {
            if (S[i] == '(') {
                curr_stack++;
            } else {
                curr_stack--;
                if (curr_stack < 0) {
                    to_add++;
                    curr_stack = 0;
                }
            }
        }
        return to_add + curr_stack;
    }
};
