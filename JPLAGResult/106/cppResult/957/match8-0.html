class Solution {
public:
    int minAddToMakeValid(string s) {
        int dev=0;
        int val=0;
        
        for(int i=0;i<s.size();i++){
            if(s[i]=='('){
                val++;
            }else{
                val--;
                if(val<0){
                    dev+=abs(val);
                    val=0;
                }
            }
            
        }
        
        return dev+abs(val);
    }
};
