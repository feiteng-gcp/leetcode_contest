class Solution {
public:
    int minAddToMakeValid(string S) 
    {
        int ret=0;
        int gap=0;
        
        for(char c : S)
        {
            if(c=='(') gap++;
            else gap--;
            if(gap<0)
            {
                ret+=abs(gap);
                gap=0;
            }
        }
        ret+=gap;
        return ret;
    }
};
