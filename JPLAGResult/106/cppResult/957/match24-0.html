class Solution {
public:
    int minAddToMakeValid(string S) {
        int bal = 0;
        int result = 0;
        for(int i = 0; i < S.length(); ++i) {
            if(S[i] == '(') {
                bal++;
            } else {
                bal--;
            }
            if(bal < 0) {
                result -= bal;
                bal = 0;
            }
        }
        result += bal;
        return result;
    }
};
