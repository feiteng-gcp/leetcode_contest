class Solution {
public:
    vector<int> sortArrayByParityII(vector<int>& A) {
        vector<int> ret(A.size(), 0);
        int oddPos = 1;
        int evenPos = 0;
        for(int i = 0; i < A.size(); i++)
        {
            if(A[i] & 1)
            {
                ret[oddPos] = A[i];
                oddPos += 2;
            }
            else
            {
                ret[evenPos] = A[i];
                evenPos += 2;
            }
        }
        return ret;
    }
};
