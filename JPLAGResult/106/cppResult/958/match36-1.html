class Solution {
public:
    vector<int> sortArrayByParityII(vector<int>& A) {
        vector<int> result(A.size());
        int even = 0;
        int odd = 0;
        for (int& a : A) {

            if (a & 1) {
                result[odd * 2 + 1] = a;
                odd++;
            } else {
                result[even * 2] = a;
                even++;
            }
        }
        return result;
    }
};
