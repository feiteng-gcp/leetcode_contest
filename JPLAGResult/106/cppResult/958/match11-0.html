class Solution {
public:
    vector<int> sortArrayByParityII(vector<int>& A) {
        vector< int > ans ( A.size() , 0 ) ; 
        int lptr = 0 ;
        int rptr = 1 ; 
        for( int i=0 ; i<A.size() ; i++){
            if( A[i] % 2 == 0 ){
                ans[lptr] = A[i] ; 
                lptr+=2 ; 
            }
            else{
                ans[rptr] = A[i] ; 
                rptr+=2; 
            }
        }
        
        return ans ; 
    }
};
