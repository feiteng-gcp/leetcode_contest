class Solution {
public:
    vector<int> sortArrayByParityII(vector<int>& A) {
        vector<int> re(A.size(), 0);
        int odd = 0, even = 1;
        for(int i = 0; i < A.size(); ++i) {
            if(A[i] % 2) {
                re[even] = A[i];
                even += 2;
            }else {
                re[odd] = A[i];
                odd += 2;
            }
        }
        return re;
    }
};
