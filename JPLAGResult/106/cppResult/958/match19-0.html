class Solution {
public:
    vector<int> sortArrayByParityII(vector<int>& A) {
        int ln=A.size();
        vector<int> res(ln, 0);
        int a=0, b=1;
        for(int i=0; i<ln; i++){
            if(A[i]%2==0){
                res[a]=A[i]; a+=2;
            }else{
                res[b]=A[i]; b+=2;
            }
        }
        return res;
    }
};
