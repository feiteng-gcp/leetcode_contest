class Solution {
public:
    vector<int> sortArrayByParityII(vector<int>& A) {
        vector<int> B(A.size());
        int i=0,j=0;
        for(int k=0;k<A.size();k++){
            if(A[k]%2==1){
                B[2*i+1]=A[k];i++;
            }else{
                B[2*j]=A[k];j++;
            }
        }
        return B;
    }
};
