class Solution {
public:
    vector<int> sortArrayByParityII(vector<int>& A) {
        vector<int> ret(A.size());
        
        int i = 0, j = 1;
        for(int v: A) {
            if (v % 2 == 0) {
                ret[i] = v;
                i += 2;
            } else {
                ret[j] = v;
                j += 2;
            }
        }
        return ret;
    }
};
