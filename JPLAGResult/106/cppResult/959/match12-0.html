class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        const int mod = 1e9+7;
        map<int, int> prev;
        int total = 0;
        
        prev[A[0]] = 1;
        for (int i = 1; i+1 < A.size(); ++i) {
            for (int j = i+1; j < A.size(); ++j) {
                int rem = target - A[i] - A[j];
                if (prev.count(rem)) {
                    total = (prev[rem] + total) % mod;
                }
            }
            if (!prev.count(A[i])) prev[A[i]] = 0;
            prev[A[i]]++;
        }
        
        return total;
    }
};
