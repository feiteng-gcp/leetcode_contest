class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {  
        vector<vector<int> > f(305, vector<int>(4, 0));
        f[0][0] = 1;
        for (int i = 0; i < A.size(); i++) {
            for (int j = target+3; j > A[i]; j--) {
                for (int k = 1; k <= 3; k++) {
                    f[j][k] = (f[j][k] + f[j - A[i] - 1][k-1]) % 1000000007;
                }
            }
        }
        return f[target+3][3];
    }
};
