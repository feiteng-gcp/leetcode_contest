#define N 3001
#define M 256

const int MOD = 1e9 + 7.5;

int f[2][N][M];

class Solution {
public:
    int threeSumMulti(vector<int>& a, int target) {
        int n = a.size();
        memset(f, 0, sizeof(f));
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= 100; j++) {
                f[0][i][j] = f[0][i - 1][j];
            }
            ++f[0][i][a[i - 1]];
        }
        
        for (int i = 1; i <= n; i++) {
            int x = a[i - 1];
            for (int j = 0; j <= 200; j++) {
                f[1][i][j] = f[1][i - 1][j];
                if (j >= x) {
                    f[1][i][j] += f[0][i - 1][j - x];
                    fixMod(f[1][i][j]);
                }
            }
        }
        
        int ret = 0;
        for (int i = n; i >= 1; i--) {
            int v = target - a[i - 1];
            if (v >= 0) ret += f[1][i - 1][v];
            fixMod(ret);
        }
        return ret;
    }
    
    void fixMod(int &x) {
        x %= MOD;
    }
};
