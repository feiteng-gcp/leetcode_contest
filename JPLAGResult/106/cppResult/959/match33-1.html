class Solution {
public:
    long long cn3(long n) {
        return n * (n-1) * (n-2) / 3 / 2;
    }
    
    long long cn2(long n) {
        return n * (n-1) / 2;
    }
    int threeSumMulti(vector<int>& A, int target) {
        long long res = 0;
        const long long mod = 1e9 + 7;
        long long count[101] = {0,};

        for (int i = 0; i < A.size(); i++) count[A[i]]++;
        for (int i = 0; i < 101; i++) {
            for (int j = i; j < 101; j++) {
                for (int k = j; k < 101; k++) {
                    if (i + j + k != target) continue;
                    
                    if (i != j && i != k && j != k) {
                        if (count[i] == 0 || count[j] == 0 || count[k] == 0) continue;
                        res = (res + count[i] * count[j] * count[k]) % mod;
                    } else if (i == j && j == k) {
                        if (count[i] < 3) continue;
                        res = (res + cn3(count[i])) % mod;
                    } else if (i == j) {
                        if (count[i] < 2 || count[k] == 0) continue;
                        res = (res + cn2(count[j]) * count[k]) % mod;
                    } else if (j == k) {
                        if (count[j] < 2 || count[i] == 0) continue;
                        res = (res + cn2(count[j]) * count[i]) % mod;
                    } else if (i == k) {
                        if (count[i] < 2 || count[j] == 0) continue;
                        res = (res + cn2(count[i]) * count[j]) % mod;
                    }
                }
            }
        }
        return res;
    }
};
