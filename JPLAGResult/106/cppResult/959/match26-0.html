class Solution {
public:
    typedef unsigned long long ull;
    int threeSumMulti(vector<int>& A, int target) {    
        //A = {1,0,1,0,2,1,2}; target = 1;
        //A = {1, 1, 2, 2, 2, 2}; target = 5;
        vector<unsigned long long> dp(1000, 0);
        ull mod = 1000000007;
        dp[A[0] + A[1]] = 1;  
        ull ret = 0;
        for(int i = 2; i < A.size() ; i++)
        {
            int remaining = target - A[i];
            if(remaining >= 0) ret = ((ret % mod) + (dp[remaining] % mod)) % mod;            
            for(int j = 0; j < i; j++)
            {
                int pairSum = A[i] + A[j];
                dp[pairSum]++;
            }
        }
        return ret;
    }
};
