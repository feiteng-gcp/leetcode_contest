class Solution {
public:
    int threeSumMulti(vector<int>& a, int target) {
        int n = a.size();
        map<int, int> mp;
        int ans = 0, mod = 1e9 + 7;

        for (int j = 0; j < n; j++){
            for (int k = j+1; k < n; k++){
                ans = (ans + mp[target - a[j] - a[k]]) % mod;
            }
            mp[a[j]]++;
        }
        return ans;
    }
};
