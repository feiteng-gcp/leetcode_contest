#define MOD 1000000007
class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        int n = A.size();
        vector<int> e(103, 0);
        vector<vector<int>> d(103, vector<int>(n + 1, 0));
        for (int i = n - 1; i >= 0; --i) {
            for (int j = 0; j <= 100; ++j) {
                d[j][i] = d[j][i + 1];
            }
            
            e[A[i]] = 1;
            if (i == n - 1) {
                d[A[i]][i] = 1;
            } else {
                d[A[i]][i] = d[A[i]][i + 1] + 1;
            }

        }
        
        long long int ret = 0;
        for (int i = 0; i < A.size(); ++i) {
            for (int j = i + 1; j < A.size() - 1; ++j) {
                int k = target - (A[i] + A[j]);
                if (k < 0 || k > 100 || e[k] == 0) continue;
                ret += d[k][j + 1];
                // cout << i << " " <<  j << " " << k << endl;
                // cout << A[i] << " " <<  A[j] << " " << k << " " << d[k][j + 1]<< endl;
                ret %= MOD;
            }
        }
        return ret;
    }
};
