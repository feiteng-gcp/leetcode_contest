class Solution {
public:
    int dp[3005][300] ; 
    int threeSumMulti(vector<int>& A, int target) {
        //a + b = target -c
        memset( dp , 0  , sizeof( dp ) ) ; 
        int ans =0 ; 
        int base = 1e9 + 7 ; 
        for( int i = 0 ; i<A.size() ; i++){
            for( int j = i+1  ; j< A.size() ; j++ ){
                dp[j][ A[i] + A[j] ] ++ ; 
            }
        }
        
        for( int i = 2 ;  i< A.size() ; i++){
            int now = target - A[i] ; 
            if( now >=0 )
            for( int index = 1 ; index < i ; index ++ ){
                ans += ( dp[index][now] ) ; 
                ans %= base ; 
            }
        }
        
        return ans ;
    }
};
