class Solution {
public:
#define mod 1000000007
	int threeSumMulti(vector<int>& A, int target) {
		int n = A.size();
		vector<vector<int>>a(n, vector<int>(105, 0));
		for (int i = 1; i < n; i++) {
			a[0][A[i]]++;
		}
		for (int i = 1; i < n; i++) {	
			a[i] = a[i - 1];
			a[i][A[i]]--;
		}

		int ret = 0;
		for (int i = 0; i < n - 2; i++) {
			for (int j = i + 1; j < n - 1; j++) {
				int x = target - A[i] - A[j];
				if (x < 0 || x>100)continue;
					ret += a[j][x];
					ret %= mod;
			}
		}
		return ret;
	}
};
