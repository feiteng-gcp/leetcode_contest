class Solution {
public:
    int mod = 1e9 + 7;
    int threeSumMulti(vector<int>& A, int target) {
        unordered_map<int, int> mp;
        long long ans = 0;
        for (int i = 0; i < A.size(); i++) {
            for (int j = i + 1; j < A.size(); j++) {
                int d = target - A[i] - A[j];
                int v = mp[d];
                ans = (ans + v) % mod;
            }
            mp[A[i]]++;
        }
        return ans;
    }
};
