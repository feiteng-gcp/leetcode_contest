class Solution {
public:
	int threeSumMulti(vector<int>& A, int target) 
	{
		const int mod = 1000000007;
		map<int, int> m;
		vector<long long> key;
		vector<long long> val;
		for (auto a : A)m[a]++;
		for (auto x : m)
		{
			key.push_back(x.first);
			val.push_back(x.second);
		}


		int ret = 0;
		for (int i = 0; i < key.size(); i++)
		{
			for (int j = i ; j < key.size(); j++)
			{
				int k_key = target - key[i] - key[j];
				if (key[j] > k_key) continue;
				int k_val = m[k_key];
				if (k_val == 0)continue;
				
				if (i == j) 
				{
					
					if (key[j] != k_key)
					{
						if (val[i] < 2)continue;
						ret = (ret + (val[i] * (val[i] - 1))/2 * k_val) % mod;
						//cout << key[i] << "," << key[j] << "," << k_key << endl;
						//cout << (val[i] * (val[i] - 1)) / 2 * k_val << endl;
					}
					else
					{
						if (val[i] < 3)continue;
						ret = (ret + (val[i] * (val[i] - 1) * (val[i] - 2)) / 6) % mod;
						//cout << key[i] << "," << key[j] << "," << k_key << endl;
						//cout << (val[i] * (val[i] - 1) * (val[i] - 2)) / 6 << endl;
					}
				}
				else 
				{
					if (key[j] == k_key)
					{
						if (k_val < 2)continue;
						
						ret = (ret + val[i] * ((k_val - 1)  * k_val) / 2) % mod;
						//cout << key[i] << "," << key[j] << "," << k_key << endl;
						//cout << val[i] * ((k_val - 1)  * k_val) / 2 << endl;
					}
					else
					{
						
						ret = (ret + val[i] * val[j] * k_val) % mod;
						//cout << key[i] << "," << key[j] << "," << k_key << endl;
						//cout <<val[i] * val[j] * k_val<< endl;
					}
				}
				
				
			}
		}
		return ret;
	}
};
