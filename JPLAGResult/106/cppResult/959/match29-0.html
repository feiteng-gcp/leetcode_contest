class Solution {
public:
	int n;
	int freq[102];
	vector<int> nums;
	const int mo = 1e9 + 7;

    int threeSumMulti(vector<int>& A, int target) {
        n = A.size();
        long long ans = 0;

        for (int i = 0; i < n; ++i)
        	freq[A[i]]++;

        for (int i = 0; i <= 100 && i <= target; ++i)
        	if (freq[i]>0)
        		nums.push_back(i);

        n = nums.size();
        for (int i = 0; i < n; ++i)
        {
        	int a = nums[i];

        	for (int j = i+1; j < n && nums[j] <= target-a; ++j)
        	{
        		int b = nums[j];
        		
        		int c = target - a - b;
        		if (c < b)
        			break;

        		if (c > 100 || c == a || c == b)
        			continue;

        		if (freq[c]>0)
        			ans = (ans + freq[a]*freq[b]*freq[c] % mo) % mo;
        	}

        	int d = target - a - a;
        	if (d < 0 || d > 100)
        		continue;

        	if (d == a && freq[a] >= 3)
        	{
        		long long f = freq[a];
        		ans = (ans + f*(f-1)*(f-2)/6 % mo) % mo;
        	}

        	if (d != a && freq[a] >= 2 && freq[d] > 0)
        	{
        		long long f = freq[a];
        		ans = (ans + f*(f-1)/2*freq[d]%mo) % mo;
        	}
        }

        return ans;
    }
};
