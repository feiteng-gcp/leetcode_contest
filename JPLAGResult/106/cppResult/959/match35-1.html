class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        long long mask[101] = {0};
        for (int i: A) {
            mask[i] += 1;
        }

        long long r = 0;
        for (int i = 0; i <= 100; ++i) {
            for (int j = i; j <= 100; ++j) {
                for (int k = j; k <= 100; ++k) {
                    if (i + j + k == target) {
                        if (i == j && j == k && i == k) {
                            r += mask[i] * (mask[i]-1) * (mask[i]-2) / 6;
                        } else if (i == j && j == k && i != k) {
                            //
                        } else if (i == j && j != k && i == k) {
                            //
                        } else if (i == j && j != k && i != k) {
                            r += mask[i] * (mask[i]-1) * mask[k] / 2;
                        } else if (i != j && j == k && i == k) {
                            //
                        } else if (i != j && j == k && i != k) {
                            r += mask[i] * mask[j] * (mask[j]-1) / 2;
                        } else if (i != j && j != k && i == k) {
                            r += mask[i] * mask[j] * (mask[i]-1) / 2;
                        } else if (i != j && j != k && i != k) {
                            r += mask[i] * mask[j] * mask[k];
                        }
                        r %= 1000000007;
                    }
                }
            }
        }
        return (int)r;
    }
};
