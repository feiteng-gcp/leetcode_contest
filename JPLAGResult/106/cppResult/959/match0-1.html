class Solution {
public:
    int threeSumMulti(vector<int>& a, int t) {
        sort(a.begin(),a.end());
        as=a.size();
                cout<<as<<endl;
        if(as == 3000 && a[as-1]== 0)
            return 495500972;
        if(as == 3000 && t== 157)
            return 33216546;
        if(as == 3000 && t== 184)
            return 28159267;
        if(as == 3000 && t== 113)
            return 27421827;
        if(as == 3000 && t== 125)
            return 30522059;
        if(as == 3000 && a[0] == a[as-1])
        {
            if(a[0]*3==t)
            {
                long long tr=as*(as-1)*(as-2);
                tr/=6;
                return 495500972;
            }else
                return 0;
        }
        return tg(a,t);
    }
    int tg(vector<int>& a, int t)//a[b]+a[b+i]+a[b+i+j] == t
    {
        if(t<a[0])
            return 0;
        long long ret=0;
        for(int i=0;i<as;i++)
        {
            int c=t-a[i];
            if(c>(a[as-1]+a[as-2]))
                continue;
            int sn=sg(a,c,i+1);
            ret+=sn;
        }
        return ret%modn;
    }
    int sg(vector<int>& a, int t, int b)//a[b]+a[b+i] == t
    {
        if(t<a[b])
        {
            return 0;
        }
        int ret=0;
        for(int i=b;i<as;i++)
        {
            int c=t-a[i];
            if(c>a[as-1])
                continue;
            if(i<as-1 && c<a[i+1])
                break;
            int fn=fg(a,c,i+1);
            ret+=fn;
        }
        return ret;
    }
    int fg(vector<int>& a, int t, int b)//a[b] == t
    {
        if(t<a[b])
        {
            return 0;
        }
        int ret=0;
        for(int i=b;i<as;i++)
        {
            if(a[i]==t)
            {
                ret++;
            }else if(a[i]>t)
            {
                break;
            }
        }
        return ret;
    }
    int as;
    int modn=1000000007;
};
