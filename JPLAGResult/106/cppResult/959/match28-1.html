class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        sort(begin(A), end(A));
        vector<int> B;
        map<int, long> cnt;
        for (auto &a : A) {
            cnt[a]++;
            if (cnt[a] == 1) B.push_back(a);
        }
        

        int ans = 0, MOD = 1e9 + 7;
        for (int i = B.size() - 1; i >= 0; --i) {
            for (int j = 0; j < i; ++j) {
                int k = target - B[i] - B[j];
                if (k < B[j] || k > B[i]) continue;
                if (cnt.count(k)) {
                    if (k != B[i] && k != B[j]) {
                        ans = (ans + cnt[B[i]] * cnt[B[j]] * cnt[k]) % MOD;
                    } else if (k == B[i]) {
                        ans = (ans + cnt[B[i]] * (cnt[B[i]] - 1) / 2 * cnt[B[j]]) % MOD;
                    } else {
                        ans = (ans + cnt[B[j]] * (cnt[B[j]] - 1) / 2 * cnt[B[i]]) % MOD;
                    }
                }
            }
            if (B[i] * 3 == target) {
                if (cnt[B[i]] > 2) {
                    ans = (ans + cnt[B[i]] * (cnt[B[i]] - 1) * (cnt[B[i]] - 2) / 6) % MOD;
                }
            }
        }
        return ans;
    }
};
