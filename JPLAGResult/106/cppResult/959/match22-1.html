class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        // sort(A.begin(), A.end());
        const int mod = 1e9+7;
        int ans = 0;
        int n = A.size();
        
        vector<int>v[101];
        

        for(int i = 0; i < n; i++)
            v[A[i]].push_back(i);
        
        for(int i = 0; i < n; i++) {
            for(int j = i + 1; j < n; j++) {
                int k = target - A[i] - A[j];
                if(k<0 || k>100)
                    continue;
                if(v[k].size() > 0) {
                    int x = upper_bound(v[k].begin(), v[k].end(), j) - v[k].begin();
                    // int diff = x - v[k].begin();
                    if(x != v[k].size()) 
                        ans = (ans + v[k].size() - x)%mod;
                } 
            }
        }
        
        return ans;
    }
};
