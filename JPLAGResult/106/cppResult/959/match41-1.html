class Solution {
public:
    int extgcd(int a, int b, int & x, int &y) {
        if (b == 0) {
            x = 1;
            y = 0;
            cout << "end:" << a * x + b * y << endl;
            return a;
        }
        int g = extgcd(b,a%b,x,y);
        int tmp = y;
        y = x;
        x = tmp;
        y -= a/b*tmp;
        cout <<x << ":"<< a * x + b * y << endl;
        return g;
    }
    int threeSumMulti(vector<int>& A, int target) {
        int n = A.size();
        unordered_map<int, int> m;
        m[A[0]]++;
        m[A[1]]++;
        const int M = 1e9 + 7;

        int ans = 0;
        for (int k = 2; k < n; k++) {
            for (int j = 0; j < k; j++) {
                
                int now = target - A[k] - A[j];
                int count = m[now];
 
                
                if (now == A[j]) {
                    count--;
                }
                ans = (ans + count) % M;
            }
            m[A[k]]++;
        }
        int x = 0, y = 0;
        extgcd(2, M, x, y);
        long long xx = ((x%M+M) % M);
        long long tmp = (long long)(ans * xx) % (long long)M;
        return (int)tmp;
    }
};
