class Solution {
public:
    const int mod = 1e9 + 7;
    int mem[3003][103];
    int threeSumMulti(vector<int>& A, int target) {
        memset(mem, 0, sizeof mem);
        int n = A.size();
        for (int i = 0; i < n; ++i) {
            ++mem[i][A[i]];
            if (i) for (int j = 0; j <= 100; ++j) mem[i][j] += mem[i - 1][j];
        }
        //std::unordered_map<int, int> M;
        // for (int i = 0; i < n; ++i) ++M[a[i]];

        long long res = 0;
        for (int k = 0; k < n; ++k) {
            for (int j = 1; j < k; ++j) {
                int x = target - A[j] - A[k];
                if (x >= 0 && x <= 100) res = (res + (long long)(mem[j - 1][x])) % mod;
            }
        }
        return res;
    }
};
