class Solution {
    const int P = 1e9+7;
public:
    int threeSumMulti(vector<int>& A, int target) {
        // map<int, int> twos;
        int n = A.size();
        array<int, 301> cnt = {};

        long ans = 0;
        for (int i = 1; i < n; i++) {
            int dif = target - A[i];
            if (dif >= 0) ans = (ans + cnt[dif]) % P;
            for (int j = 0; j < i; j++) {
                int sum = A[i]+A[j];
                if (sum <= 300) cnt[sum]++;
            }
        }
        return ans;
    }
};
