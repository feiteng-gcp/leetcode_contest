int cnt[305];
const int mod=1000000007;
class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        memset(cnt,0,sizeof(cnt));
	int n=A.size();
	for(int i=0;i<n;i++)
	{
		cnt[A[i]]++;
	}
	long long ans=0;
	for(int i=0;i<=100;i++)
	{
		for(int j=i;j<=100;j++)
		{
			int k=target-i-j;
			if(k<j)continue;
			long long now;
			if(i==j&&j==k)
			{
				now=(long long)cnt[i]*(cnt[i]-1)*(cnt[i]-2)/6;
			}
			else if(i==j)
			{
				now=(long long)cnt[k]*cnt[i]*(cnt[i]-1)/2;
			}
			else if(j==k)
			{
				now=(long long)cnt[i]*(cnt[j])*(cnt[j]-1)/2;
			}
			else
			{
				now=(long long)cnt[i]*cnt[j]*cnt[k];
			}
			ans=(ans+now)%mod;
		}
	}
	return ans;

    }
};
