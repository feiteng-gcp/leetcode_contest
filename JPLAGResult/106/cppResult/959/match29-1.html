class Solution {
public:

    int threeSumMulti(vector<int>& A, int target) {
        sort(A.begin(), A.end());
        int count=0;
        
        for(int i=0;i<(A.size()-2);++i)
        {
            int l=i+1;
            int r=A.size()-1;
            while(l<r)
            {
                const int tmp = A[i]+A[l]+A[r];
                if(tmp == target)
                {
                    if(A[l] == A[r])
                    {
                        int curr = r-l+1;
                        count+=(curr*(curr-1))/2;
                        count = count%1000000007;
                        break;
                    }
                    

                    int lc=1,lr=1;
                    for(int n=l+1;n<r;++n)
                        if(A[n]==A[l])
                            ++lc;
                    for(int n=r-1;n>l;--n)
                        if(A[n]==A[r])
                            ++lr;
                    l+=lc;
                    r-=lr;
                    count+=(lr*lc);
                    count = count%1000000007;
                }
                else if(tmp > target)
                    --r;
                else
                    ++l;
            }
        }
        return count;
    }
};
