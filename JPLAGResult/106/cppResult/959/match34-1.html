class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        vector<int> nums_A(101, 0);
        long long result = 0, count = 0;;
        int mod = 1000000007;
        for(int i = 0;i < A.size();i++)

            nums_A[A[i]]++;
        for(int i = 0;i <= 100;i++){
            for(int j = i;j <= 100;j++){
                for(int m = j;m <= 100;m++){
                    if(i + j + m == target){                    	
                        if(i == m){
							if(nums_A[i] >= 3)
                            	result += combinator(nums_A[i], 3);                        	
						}
                        	
                        else if(i == j){
                        	if(nums_A[i] >= 2 && nums_A[m] >= 1)
                        		result += (combinator(nums_A[i], 2) * combinator(nums_A[m], 1));
						}
                        else if(j == m){
                        	if(nums_A[j] >= 2 && nums_A[i] >= 1)
                        		result += (combinator(nums_A[j], 2) * combinator(nums_A[i], 1));
						}  
                        else{
                        	if(nums_A[i] >= 1 && nums_A[j] >= 1 && nums_A[m] >= 1)
                        		result += (combinator(nums_A[i], 1) * combinator(nums_A[j], 1) * combinator(nums_A[m], 1));
						}    
                        //cout << "i = " << i << ";j = " << j << ";m = " << m << ";reuslt = " << result << endl;
                        result %= mod;
                    }
                }
            }
        }
        return (int)result;
    }
    


    long long combinator(int n,int m)
    {
        if(m > n)
            return 0;
        long long molecule = 1, denominator = 1;
        for(int i = 0;i < m;i++){
            molecule *= (n - i);
            denominator *= (i + 1);
        }
        return molecule / denominator;
    }

};
