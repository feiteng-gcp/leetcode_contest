class Solution {
public:
    int threeSumMulti(vector<int>& nums, int target) {
        vector<vector<int>> ret;
        sort(nums.begin(),nums.end());
        int n = nums.size();
        for(int a = 0; a < n; ++a)
        {
            int b = a + 1,
                c = n - 1;
            while(b < c)
            {
                int s = nums[a] + nums[b] + nums[c] - target;
                //cout << nums[a] << ' ' << nums[b] << ' ' << nums[c] << endl;
                if(s < 0)
                    ++b;
                else if(s > 0)
                    --c;
                else
                {
                    ret.push_back(vector<int>{nums[a],nums[b],nums[c]});
                    ++b;
                }
            }
            
            while(nums[a + 1] == nums[a]) ++a;
        }
        ret.erase(unique(ret.begin(),ret.end()), ret.end());
        //Accumulate
        long long accu[105];
        memset(accu,0, sizeof accu);
        for(auto i : nums)
            ++accu[i];
        long long ans = 0, MOD = 1000000007;
        for(auto r : ret)
        {
            int a = r[0], b=  r[1], c = r[2];
            if(a == b)
            {
                if(b == c)
                    ans += accu[a] * (accu[a] - 1) * (accu[a] - 2) / 6;
                else
                    ans += accu[a] * (accu[a] - 1) / 2 * accu[c];
            }
            else
            {
                if(b == c)
                    ans += accu[a] * accu[b] * (accu[b] - 1) / 2;
                else
                    ans += accu[a] * accu[b] * accu[c];
            }
            ans %= MOD;
        }
        return ans;
    }
};
