class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        const int MOD = 1000000007;
        int len = A.size();
        long res = 0;
        if(!len)return 0;
        sort(A.begin(), A.end());

        unordered_map<int, int> cnts;
        ++cnts[A[0]];
        for(int i = 1; i < len; ++i)
        {
            for(int j = i + 1; j < len; ++j)
            {
                int toFind = target - A[i] - A[j];
                if(toFind < 0)break;
                res += cnts[toFind];
                res %= MOD;
            }
            ++cnts[A[i]];
        }
        return res;
    }
};
