class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        int n = A.size();
        map<int, int> map;
        int ans = 0;
        const int M = 1000000007;
        for(int i = 0; i < n; ++i){
            for(int j = i + 1; j < n; ++j){
                int t = target - A[i] - A[j];
                if(t < 0) continue;
                (ans += map[t]) %= M;
            }
            ++map[A[i]];
        }
        return ans;
    }
};
