int fre[101][3001];

class Solution {
public:

    int threeSumMulti(vector<int>& v, int target) {
        int n = v.size();
        for (int i = 0; i <= 100; i++)
            for (int j = 0; j <= n; j++)
                fre[i][j] = 0;
        
        for (int i = 0; i <= 100; i++) {
            for (int j = 0; j < n; j++) {
                if (j == 0) {
                    if (v[0] == i) fre[i][0] = 1;
                } else {
                    if (v[j] == i) {
                        fre[i][j] = fre[i][j-1] + 1;
                    } else {
                        fre[i][j] = fre[i][j-1];
                    }
                }
            }
        }

        long long res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int s = v[i] + v[j];
                int rem = target - s;
                if (rem >= 0 && rem <= 100 && j < (n-1)) {
                    int b = fre[rem][n-1];
                    int a = fre[rem][j];
                    res += (b - a);
                }
            }
        }
        res = res % (1000000000 + 7);
        return res;
    }
};
