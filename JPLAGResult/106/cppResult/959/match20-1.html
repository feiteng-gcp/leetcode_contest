class Solution {
public:
    int twosum(vector<int>& A, int i, int target) {

        int j = A.size()-1, ans = 0;
        
        while (i < j) {
            if (A[i] + A[j] == target) {
                int ca = 1, cb = 1;
                while (i+1 < A.size() && A[i] == A[i+1]) {
                    i++;
                    ca++;
                }
                while (j-1 >= 0 && A[j] == A[j-1]) {
                    cb++;
                    j--;
                }
                if (A[i] == A[j]) {
                    ans += (ca * (ca-1)) / 2;
                    break;
                }
                
                i++;

                j--;
                ans += ca * cb;
                
            } else if (A[i] + A[j] < target) {
                i++;
            } else {
                j--;
            }
            
            ans = ans % 1000000007;
        }
        
        return ans;
    }
    
    int threeSumMulti(vector<int>& A, int target) {
        sort(A.begin(), A.end());
        long long int ans = 0;
        for (int i = 0; i < A.size(); i++) {
            ans += twosum(A, i+1, target-A[i]);
            ans = ans % 1000000007;
        }
        
        return ans;
    }
};
