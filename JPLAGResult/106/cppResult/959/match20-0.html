class Solution {
public:
    int MOD = 1000000007;
    int threeSumMulti(vector<int>& A, int target) {
        sort(A.begin(), A.end());
        int re = 0;
        for(int i = 0; i < A.size(); ++i) {
            if(A[i] > target) break;
            int res = target - A[i];
            int j = i+1, k = A.size()-1;
            while(j<k) {
                if(A[j] + A[k] == res) {
                    int a = 1, b = 1;
                    while(j+1<k && A[j] == A[j+1]) {
                        ++j;
                        ++a;
                    }
                    while(j<k-1 && A[k] == A[k-1]) {
                        --k;
                        ++b;
                    }
                    re += a * b;
                    re %= MOD;
                    if(A[j] == A[k]) {
                        re += a * (a-1) / 2;
                        re %= MOD;
                    }
                    --k;
                    ++j;
                } else if(A[j] + A[k] > res) {
                    --k;
                } else {
                    ++j;
                }
            }
        }
        return re;
    }
};
