#define all(x) (x).begin(),(x).end()
const int N = 100 + 15;
int mod = (int)1e9 + 7;
class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        vector<int> pos[N];
        int len = A.size();
        for (int i = 0; i < len; ++i) {
            pos[A[i]].push_back(i);
        }
        for (int i = 0; i <= 100; ++i) {
            sort(all(pos[i]));
        }
        long long ans = 0;
        for (int i = 0; i < len - 1; ++i) {
            for (int j = i + 1; j < len - 1; ++j) {
                 int k = target - (A[i] + A[j]);
                 if (0 <= k && k <= 100) {
                     int p = lower_bound(pos[k].begin(), pos[k].end(), j + 1) - pos[k].begin();
                     ans = ans + (pos[k].size() - p) % mod;
                     ans = ans % mod;
                 }
            }
        }
        return ans;
    }
    int bruteForce(vector<int>& A, int target) {
        int len = A.size();

        int ans = 0;
        for (int i = 0; i < len - 1; ++i) {
            for (int j = i + 1; j < len - 1; ++j) {
                for (int k = j + 1; k < len; ++k) {
                    if (A[i] + A[j] + A[k] == target){
                        ans++;
                    }
                }
            }
        }
        return ans;
    }
};
