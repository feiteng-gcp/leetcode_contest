long dp[3005][4][301];
long mod = 1e9 + 7;
class Solution {
public:
    int threeSumMulti(vector<int>& A, int target) {
        memset(dp, 0, sizeof (dp));
        int n = A.size();
        dp[0][1][A[0]] = 1;
        for (int i = 1; i < n; i++) {
            for (int j = 0; j <= target; j++) {
                dp[i][1][j] = dp[i - 1][1][j];
                dp[i][2][j] = dp[i - 1][2][j];
                if (j >= A[i]) {
                    dp[i][2][j] += dp[i - 1][1][j - A[i]];
                }
                dp[i][2][j] %= mod;
                dp[i][3][j] = dp[i - 1][3][j];
                if (j >= A[i]) {
                    dp[i][3][j] += dp[i - 1][2][j - A[i]];
                }
                dp[i][3][j] %= mod;
            }
            dp[i][1][A[i]]++;
        }
        return (dp[n - 1][3][target]);
    }
};
