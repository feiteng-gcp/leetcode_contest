    int threeSumMulti(vector<int>& A, int target) {
        sort(A.begin(), A.end());
        int n=A.size();
        long long ans=0;
        for(int i=0;i<n-2;i++) {
            int t=target-A[i];
            int j=i+1;
            int k=A.size()-1;
            while(j<k) {
                if(A[j]+A[k]==t) {
                    if(A[j]==A[k]) {
                        long long count=k-j+1;
                        ans=(ans+(((count*(count-1))%MOD)/2))%MOD;
                        k=j;
                    }
                    else {
                        long long countJ=1;
                        while(A[j]==A[j+1])  {
                            countJ++;
                            j++;
                        }
                        long long countK=1;
                        while(A[k]==A[k-1]) {
                            countK++;
                            k--;
                        }
                        ans=(ans+(countJ*countK)%MOD)%MOD;
                        j++;
                        k--;
                    }
                }
                else if(A[j]+A[k]<t) {
                    j++;
                }
                else {
                    k--;
                }
            }
        }
        return ans;
    }
};
