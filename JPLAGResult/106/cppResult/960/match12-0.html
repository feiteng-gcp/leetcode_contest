class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        vector<unordered_set<int>> dsets(1);
        int counter = 0;
        int N = graph.size();
        vector<bool> visited (N,false);
        while(1){
            queue<int> myq;
            int j = 0;
            while(visited[j]) j++;
            myq.push(j);
            counter++;
            visited[j] = true;
            int k = dsets.size()-1;
            while(!myq.empty()){
                int s = myq.front();
                // cout << "s = " << s << endl;
                myq.pop();
                dsets[k].insert(s);
                for(int i=0; i<N; i++){
                    // cout << "graph[s][i]  " << graph[s][i] << endl;
                    // cout << "visited[i]  " << visited[i] << endl;
                    if(graph[s][i] && !visited[i]){
                        visited[i] = true;
                        myq.push(i);
                        // cout << i << endl;
                        counter++;
                    }
                }
            }
            // cout << endl;
            if(counter<N){
                dsets.resize(k+2);
            }
            else break;
        }
        // cout << endl;
        // for(auto c:dsets){
        //     for(auto k:c){
        //         cout << k << endl;
        //     }
        //     cout << endl;
        // }
        unordered_map<int,int> mym;
        for(int i=0; i<initial.size(); i++){
            //find group of initial[i]
            for(int j=0; j<dsets.size(); j++){
                if(dsets[j].count(initial[i])){
                    mym[j]++;
                }
            }
        }
        sort(initial.begin(), initial.end());
        int minno = INT_MAX;
        for(auto c:mym){
            if(c.second < minno) minno = c.second;
        }
        if(minno > 1){
            return initial[0];
        }
        else{
            //minno is 1
            int maxno = 0;
            int result = 0;
            for(int i=0; i<initial.size(); i++){
                //find group of initial[i]
                for(int j=0; j<dsets.size(); j++){
                    if(dsets[j].count(initial[i]) && mym[j]==1){
                        if(dsets[j].size()>maxno){
                            result = initial[i];
                            maxno = dsets[j].size();
                        }
                    }
                }
            }
            return result;
        }
        return initial[0];
    }
};
