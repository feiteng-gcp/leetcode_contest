class Solution {
public:
    int find( vector<int>& groups, int g )
    {
        int r = g;
        while( r != groups[r] )
            r = groups[r];
        groups[g] = r;
        return r;
    }

    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        vector<int> groups( graph.size() );

        for( int i = 0; i < graph.size(); i++ )
            groups[i] = i;
        
        for( int i = 0; i < graph.size(); i++ )
        {
            for( int j = 0; j < graph[i].size(); j++ )
            {
                if( graph[i][j] == 1 )
                {
                    int g1 = find( groups, i );
                    int g2 = find( groups, j );
                    if( g1 != g2 )
                    {
                        groups[g1] = g2;
                    }
                }
            }
        }
        

        unordered_map<int,int> map;
        for( int i = 0; i < graph.size(); i++ )
        {
            map[ find( groups, i) ]++;
        }
        
        //for( auto& pair : map )
        //    cout << pair.first << "  " << pair.second << endl;
        
        int ans = INT_MAX;
        int ansIndex = 0;
        
        sort( initial.begin(), initial.end() );
        
        for( int i =  0; i < initial.size(); i++ )
        {
            unordered_set<int> downGroups;
            for( int k = 0; k < initial.size(); k++ )
            {
                if( k == i )
                    continue;
                downGroups.insert( find( groups, initial[k] ) );
            }
            
            int downNodes = 0;
            for( const int& d : downGroups )
            {
                downNodes += map[d];
            }
            
            if( downNodes < ans )
            {
                ans = downNodes;
                ansIndex = i;
            }
        }
        
        return initial[ansIndex];
    }
};
