struct DisjointSet {
  std::vector<int> pre;

  DisjointSet(int n) : pre(n){
    for (int i = 0; i < n; i++) {
      pre[i] = i;
    }
  }
  int find(int u) { return pre[u] == u ? pre[u] : pre[u] = find(pre[u]); }
  void merge(int u, int v) { pre[find(u)] = find(v); }
};

class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& a) {
        sort(a.begin(), a.end());
        int p = 0, mi = graph.size() + 5;
        for (int i = 0; i < a.size(); i++) {
            int n = graph.size();
            DisjointSet ds(graph.size() + 1);
            for (int j = 0; j < a.size(); j++) {
                if (j == i) continue;
                ds.merge(a[j], n);
            }
            for (int j = 0; j < graph.size(); j++) {
                for (int k = 0; k < graph.size(); k++) {
                    if (graph[j][k]) ds.merge(j, k);
                }
            }
            int sum = 0;
            for (int j = 0; j < graph.size(); j++) {
                sum += (ds.find(j) == ds.find(n));
            }
            if (sum < mi) mi = sum, p = i;
        }
        return a[p];
    }
};
