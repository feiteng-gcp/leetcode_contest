class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& g, vector<int>& a) {
        int mi = 100000, mii = -1;
        int n = g.size();
        sort(a.begin(), a.end());
        for(auto i : a){
            vector<int> b(n, 0);
            queue<int> que;
            int cnt = 0;
            for(auto& x : a) if(x != i) que.push(x), b[x] = 1, ++cnt;
            while(!que.empty()){
                int now = que.front(); que.pop();
                for(int j = 0; j < n; ++j) if(g[now][j] == 1 && now != j && b[j] == 0) {
                    b[j] = 1;
                    ++cnt;
                    que.push(j);
                }
            }
            if(mi > cnt){
                mi = cnt;
                mii = i;
            }
        }
        return mii;
    }
};
