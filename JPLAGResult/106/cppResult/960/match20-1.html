#define MAX 305

class Solution {
public:
    vector<int> adj[MAX];
    vector<bool> seen;
    vector<int> tot;
    
    void dfs(int u, int &cnt){

        for( int i = 0 ; i < adj[u].size(); ++i ){
            int v = adj[u][i];
            if( !seen[v] ){
                seen[v] = true;
                dfs(v, ++cnt);
            }
        }
    }

    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int V = graph.size();
        for( int i = 0 ; i < MAX;  ++i ) adj[i].clear();
        //V = 30;

        for( int i = 0 ; i < V ; ++i ){
            for( int j = 0 ; j < V ; ++j ){
                if( graph[i][j]) 
                    adj[i].push_back(j);
            }
        }
        seen = vector<bool>(V, false);
        //int cnt = 0;
        //vector<int> tot(V, 0);
        //tot = vector<int>(V, 0);
        sort( initial.begin(), initial.end());
        int maxi = -1, res = 0;
        for( int i = 0 ; i < initial.size(); ++i ){
        //for( int i = 0 ; i < V; ++i ){
            //initial[i] = i;
            if( seen[ initial[i] ] ){
                continue;
            }
            //seen = vector<bool>(V, false);
            int cnt = 0;            
            seen[ initial[i] ] = true;
            //cout<<i<<endl;
            dfs(initial[i], cnt);
            
            if( cnt > maxi ){
                maxi = cnt;
                res = initial[i];
            }
        }
        return res;
    }
};
