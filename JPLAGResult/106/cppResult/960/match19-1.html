int get(const vector<vector<int>>& graph, const vector<int>& initial, int exclude) {
    vector<int> used(graph.size(), false);
    vector<int> q;
    for (auto x : initial) {
        if (x != exclude) {

            used[x] = true;
            q.push_back(x);
        }
    }
    while (!q.empty()) {
        int x = q.back();
        q.pop_back();
        for (int i = 0; i < graph.size(); ++i) {
            if (graph[x][i] && !used[i]) {
                used[i] = true;
                q.push_back(i);
            }
        }
    }
    return std::accumulate(used.begin(), used.end(), 0);
}


class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int best = graph.size();
        int best_index = initial[0];
        for (auto x : initial) {
            int c = get(graph, initial, x);
            if (c < best || c == best && x < best_index) {
                best_index = x;
                best = c;
            }
        }
        return best_index;
    }
};
