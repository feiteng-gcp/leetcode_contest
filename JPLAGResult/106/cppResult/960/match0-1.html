class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& g, vector<int>& in) {//graph initial, return m
        gs=g.size();
        int num=INT_MAX;
        int ret=-1;
        sort(in.begin(),in.end());
        for(int i=0;i<in.size();i++)
        {
            int c=exg(g,in,i);
            //cout<<"c"<<c<<endl;
            if(c<num)
            {
                num=c;
                ret=in[i];
            }
        }
        return ret;        
    }
    int exg(vector<vector<int>>& g, vector<int>& in, int ex)
    {
        vector<int> last(gs,0);
        int ret=0;
        for(int i=0;i<in.size();i++)
        {
            if(i!=ex)
            {
                //cout<<"exg ret"<<ret<<endl;
                ret+=sig(g,in[i],last);
            }
        }
        return ret;
    }
    int sig(vector<vector<int>>& g, int b, vector<int>& last)
    {
        //cout<<"b"<<b<<endl;
        //cout<<"ls "<<last.size()<<endl;
        if(last[b]==1)
            return 0;
        last[b]=1;
        stack<int> st;
        st.push(b);
        int ret=1;
        while(!st.empty())
        {
            int cur=st.top();
            st.pop();
            for(int i=0;i<g[cur].size();i++)
            {
                if(last[i]==0 && g[cur][i] == 1)
                {
                    last[i]=1;
                    ret++;
                    st.push(i);
                }
            }
        }
        return ret;
    }
    int gs;
};
