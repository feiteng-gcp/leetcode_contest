//class here
class Solution {
public:
  int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
	int N = SZ(graph);
	VVI G(N);
	REP(i,N) REP(j,N) if(graph[i][j])
	  G[i].PB(j);

	int ans = 1e9;
	int ans_ix = -1;

	SORT(initial);
	for(int i: initial){
	  queue<int> q;
	  vector<bool> inf(N);
	  int sum = 0;

	  for(int u: initial){
		if(u != i){
		  q.push(u);
		  inf[u] = true;
		  ++sum;
		}
	  }
	  
	  while(!q.empty()){
		int u = q.front();
		q.pop();

		for(int to: G[u]){

		  if(!inf[to]/* && SZ(G[to]) == 2*/){
			//cout << u << "->"<<to<<endl;
			inf[to] = true;
			q.push(to);
			++sum;
		  }
		}
	  }

	  //	  cout<<sum<<","<<i<<endl;
	  if(sum < ans){
		ans = sum;
		ans_ix = i;
	  }
	}

	return ans_ix;
  }
};
