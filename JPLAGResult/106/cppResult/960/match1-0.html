class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        sort(initial.begin(), initial.end());
        int n = graph.size();
        vector<vector<bool>> m(n, vector<bool>(n, false));
        vector<int> c(n, 0);
        stack<int> s;
        for (int x: initial) {
            m[x][x] = true;
            s.push(x);
            while (!s.empty()) {
                int y = s.top();
                s.pop();
                for(int i = 0; i < n; i++) {
                    if (!m[i][x] && graph[y][i] == 1) {
                        m[i][x] = true;
                        s.push(i);
                        c[i]++;
                    }
                }
            }
        }
        int ret = initial[0];
        int mm = 0;
        for (int x: initial) {
            int t = 0;
            for (int i = 0; i < n; i++) {
                if (c[i] == 1 && m[i][x]) {
                    t++;
                }
            }
            //cout << x << ' ' << mm << endl;
            if (t > mm) {
                mm = t;
                ret = x;
            }
        }
        return ret;
    }
};
