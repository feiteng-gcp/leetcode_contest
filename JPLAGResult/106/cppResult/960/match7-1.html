class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& g, vector<int>& v) {
        int n = g.size(), m = v.size();
        int mi = n + 1, ans = -1;
        sort(v.begin(), v.end());
        for (int k : v){
            queue<int> que;
            vector<int> visit(n);
            int sz = 0;
            for (int x : v){
                if (x != k){
                    que.push(x);
                    visit[x] = 1;
                    sz++;
                }
            }
            while (!que.empty()){
                int x = que.front();
                que.pop();
                for (int next = 0; next < n; next++){
                    if (!g[x][next] || visit[next]) continue;

                    visit[next] = 1;
                    sz++;
                    que.push(next);
                }
            }
            if (sz < mi){
                mi = sz;
                ans = k;
            }
        }
        return ans;
    }
};
