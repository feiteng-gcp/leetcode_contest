class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int r = graph.size(), rnode = initial[0];
        for (int k: initial) { // remove k
            std::vector<int> visited(graph.size(), 0);
            std::queue<int> q;
            for (int x: initial) {
                if (x != k) {
                    q.push(x);
                    visited[x] = 1;
                }
            }
            while (!q.empty()) {
                int x = q.front();
                for (int i = 0; i < graph[x].size(); ++i) {
                    if (graph[x][i]) {
                        if (visited[i] == 0) {
                            q.push(i);
                        }
                        visited[i] = 1;
                    }

                }
                q.pop();
            }
            int s = 0;
            for (int i = 0; i < graph.size(); ++i) {
                s += visited[i];
            }
            if (s < r || (s == r && k < rnode)) {
                r = s;
                rnode = k;
            }
        }
        return rnode;
    }
};
