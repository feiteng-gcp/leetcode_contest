class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int mn = INT_MAX;
        int re = initial[0];
        for(int i = 0; i < initial.size(); ++i) {
            queue<int> q;
            vector<int> visit(graph.size(), 0);
            int c = 0;
            for(int j = 0; j < initial.size(); ++j) {
                if(i != j) {
                    q.push(initial[j]);
                    visit[ initial[j] ] = 1;
                    c++;
                }
            }
            while(!q.empty()) {
                int x = q.front();
                q.pop();
                for(int j = 0; j < graph[x].size(); ++j) {
                    if(graph[x][j] == 1 && visit[j] == 0) {
                        visit[j] = 1;
                        q.push(j);
                        c++;
                    }
                }
            }
            if(c < mn) {
                mn = c;
                re = initial[i];
            } if(c == mn && initial[i] < re) {
                re = initial[i];
            }
        }
        return re;
    }
};
