class Solution {
public:
    void dfs(vector<vector<int>>& graph, int src, vector<int>& visit, int v, int& count) {
        if (visit[src]) return;
        visit[src] = v;
        count++;
        
        for (int i = 0; i < graph[src].size(); i++) {
            if (graph[src][i] == 1) dfs(graph, i, visit, v, count);
        }
    }
    
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        sort(initial.begin(), initial.end());
        
        vector<int> visit(graph.size(), 0), count(graph.size()+1, INT_MIN);
        int curr = 1;
        
        for (int i = 0; i < initial.size(); i++) {
            if (visit[initial[i]] == 0) {
                int cnt = 0;
                dfs(graph, initial[i], visit, curr, cnt);
                count[curr++] = cnt;
            }
        }
        
        unordered_map<int,int> umap;
        for (auto i : initial) umap[visit[i]]++;
        
        int mx = 0, ind = -1;
        for (auto i : initial) {
            if (umap[visit[i]] > 1) {
                if (mx < 1) {
                    mx = 1;
                    ind = i;
                }
            } else {
                if (mx < count[visit[i]]) {
                    mx = count[visit[i]];
                    ind = i;
                }
            }
        }
        
        return ind;
    }
};
