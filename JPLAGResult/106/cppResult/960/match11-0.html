class Solution {
public:
    int fa[305];
    int n;
    void unite(int x, int y) {
        x = find(x);
        y = find(y);
        fa[x] = y;
    }
    int find(int x) {
        if (fa[x] != x)
            fa[x] = find(fa[x]);
        return fa[x];
    }
    int get(int k, vector<int>& ini, vector<int>& cnt) {
        int ans = ini[k];
        int sz = cnt[find(ans)];
        for (int i = 0; i < (int)ini.size(); ++i) {
            if (i == k) continue;
            if (find(ans) == find(ini[i])) {
                sz = 0;
                break;
            }
        }
        return sz;
    }
    int minMalwareSpread(vector<vector<int>>& g, vector<int>& ini) {
        n = g.size();
        for (int i = 0; i < n; ++i) fa[i] = i;
        
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                if (g[i][j]) {
                    unite(i, j);
                }
            }
        }
        vector<int> cnt(n, 0);
        for (int i = 0; i < n; ++i) {
            cnt[find(i)]++;
        }
        sort(begin(ini), end(ini));
        set<int> st;
        for (auto &k : ini) st.insert(k);
        int ans = ini[0];
        int sz = get(0, ini, cnt);
        for (int i = 1; i < (int)ini.size(); ++i) {
            int cur = get(i, ini, cnt);
            //cout << sz << " " << cur << endl;
            if (cur > sz) {
                sz = cur;
                ans = ini[i];
            }
        }
        return ans;
    }
};
