class Solution {
public:
    const int M=301;
    vector<int> fa,num,A;//连通域，连通域对应的节点数，连通域包括初始节点的数目
    vector<bool> vis;
    int n;
    int find(int x){
        return fa[x]==-1?x:fa[x]=find(fa[x]);
    }
    void combine(int x,int y){
        int fx=find(x),fy=find(y);
        if(fx!=fy){
            fa[fy]=fx;
            num[fx]+=num[fy];
        }
    }
    void dfs(vector<vector<int>>& graph,int x){
        //printf("%d\n",x);
        for(int i=0;i<n;i++){
            if(graph[x][i]&&!vis[i]){
                combine(x,i);
                vis[i]=true;
                dfs(graph,i);
            }
        }
    }
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        n=graph.size();
        fa.resize(301);fill(fa.begin(),fa.end(),-1);
        num.resize(301);fill(num.begin(),num.end(),1);
        vis.resize(301);fill(vis.begin(),vis.end(),false);
        A.resize(301);fill(A.begin(),A.end(),0);
        for(int i=0;i<n;i++){
            if(!vis[i]){
                vis[i]=true;
                dfs(graph,i);
            }
        }
        for(int i=0;i<initial.size();i++){
            A[find(initial[i])]++;
        }
        int ans=-1,id=301,f=301;
        for(int i=0;i<initial.size();i++){
            int y=find(initial[i]);
            f=min(f,initial[i]);
            if(A[y]==1&&num[y]>=ans&&initial[i]<id){
                id=initial[i];
                ans=num[y];
            }
        }
        return id==301?f:id;
    }
};
