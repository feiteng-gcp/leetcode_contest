class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        vector<vector<int>> adj(graph.size());
        for (int i = 0; i < graph.size(); ++i) {
            for (int j = i+1; j < graph.size(); ++j) {
                if (graph[i][j]) {
                    adj[i].push_back(j);
                    adj[j].push_back(i);
                }
            }
        }
        
        sort(initial.begin(), initial.end());
        
        set<int> initset;
        for (auto i : initial) {
            initset.insert(i);
        }
        
        int comp = 0;
        vector<int> root(graph.size());
        vector<int> comp_size;
        vector<bool> good_comp;
        vector<bool> vis(graph.size(), false);
        for (auto i : initial) {
            if (!vis[i]) {
                comp_size.push_back(0);
                
                initset.erase(i);
                good_comp.push_back(dfs(i, comp, adj, vis, root, comp_size.back(), initset));
                initset.insert(i);
                comp++;
            }
        }
        
        int max_comp = 0;
        bool has_bad_comp = false;
        for (int i = 0; i < comp; ++i) {
            if (!good_comp[i]) {
                max_comp = max(max_comp, comp_size[i]);
                has_bad_comp = true;
            }
        }
        
        if (!has_bad_comp) {
            return initial[0];
        }
        
        for (auto i : initial) {
            if (comp_size[root[i]] == max_comp) {
                return i;
            }
        }
    }
    
private:
    bool dfs(int n, int comp, const vector<vector<int>>& adj, vector<bool>& vis, vector<int>& root, int& size, const set<int>& initset) {
        int res = initset.count(n) ? true : false;
        vis[n] = true;
        root[n] = comp;
        size++;
        

        for (auto nn : adj[n]) {
            if (!vis[nn])  {
                res = dfs(nn, comp, adj, vis, root, size, initset) || res;
            }
        }
        return res;
    }
};
