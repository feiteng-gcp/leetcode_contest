class Solution {
public:
    void DFS(vector<vector<int>>& graph, vector<int>& m, vector<int>& ans, int n, int a)
    {
        if(m[n] != -1) return;
        m[n] = a;
        ans[a]++;
        for(int i = 0; i < graph.size(); i++)
        {
            if(graph[n][i] == 1 && m[i] == -1)
            {
                //cout << i <<endl;
                DFS(graph, m, ans, i, a);
            }
        }
    }
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int N = graph.size();
        vector<int> m(N, -1);
        vector<int> ans(N, 0);
        for(auto a:initial)
        {
            if(m[a] != -1)
            {
                ans[a] = 0;
                ans[m[a]] = 0;
            }
            else
            {
                DFS(graph, m, ans, a, a);
            }
        }
        sort(initial.begin(), initial.end());
        int r = initial[0];
        
        for(int i = 1; i < initial.size(); i++)
        {
            if(ans[initial[i]] > ans[r]) r = initial[i];
        }
        
        return r;
        
    }
};
