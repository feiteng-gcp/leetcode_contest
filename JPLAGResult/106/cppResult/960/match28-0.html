class Solution {
public:
    
    int mark(vector<vector<int>>& G, int curr, vector<int>& group, int currGroup) {
        //cout << curr << " ";
        int nodes = 1;
        for (int next: G[curr]) {
            if (group[next] == -1) {
                group[next] = currGroup;
                nodes += mark(G, next, group, currGroup);
            }
        }
        return nodes;
    }
    
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int N = graph.size();
        vector<vector<int>> G(N);
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                if (i != j && graph[i][j] == 1) {
                    G[i].push_back(j);
                }
            }
        }
        vector<int> group(N, -1);
        vector<int> cnts;
        int total = 0;
        for (int i = 0; i < N; i++) {
            if (group[i] != -1) {
                continue;
            }
            group[i] = cnts.size();
            cnts.push_back(mark(G, i, group, group[i]));
            //cout << endl;
            //cout << cnts.size() - 1 << " : " << cnts.back() << endl;
            total += cnts.back();
        }
        sort(initial.begin(), initial.end());
        int res = initial[0];
        int M = total;
        vector<bool> isInit(N, false);
        for (int x: initial) {
            isInit[x] = true;
        }
        for (int x: initial) {
            bool reduced = true;
            for (int i = 0; i < N; i++) {
                if (i != x && isInit[i] && group[i] == group[x]) {
                    reduced = false;
                    break;
                }
            }
            if (reduced && total - cnts[group[x]] < M) {
                res = x;
                M = total - cnts[group[x]];
            }
        }
        return res;
    }
};
