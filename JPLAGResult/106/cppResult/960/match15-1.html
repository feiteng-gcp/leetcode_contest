class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        vector<bool> visited(graph.size(), false);
        unordered_set<int> seeds;
        int index = graph.size()+1;
        int max_save = 0;
        for(int n: initial) {
            index = min(n, index);
            seeds.insert(n);
        }
        //cout<<index<<endl;
        for(int n: initial){
            if(visited[n]) continue;
            queue<int> bfs;
            bfs.push(n);
            visited[n] = true;
            int count = 1;
            int size = 0;

            while(!bfs.empty()){
                int front = bfs.front();
                bfs.pop();
                for(int i = 0; i<graph.size(); i++){
                    if(!visited[i]&&graph[i][front]){
                        size++;
                        bfs.push(i);
                        visited[i] = true;
                        if(seeds.find(i)!=seeds.end()) count++;

                    }
                }
            }
            if(count==1&&size>max_save){
                max_save = size;
                index = n;
            }
            //cout<<count<<" "<<max_save<<endl;
        }
        return index;
    }
};
