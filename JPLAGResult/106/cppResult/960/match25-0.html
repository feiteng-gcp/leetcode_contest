class Solution {
public:
	set<int> st;
	vector<int> flags;
	void dfs(vector<vector<int>>& gra, int u){
		int N=gra.size();
		flags[u]=1;
		st.insert(u);
		for(int i=0; i<N; i++){
			if(gra[u][i]==1 && flags[i]==0){
				dfs(gra, i);
			}
		}

	}

    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
		sort(initial.begin(), initial.end());
		int N=graph.size();
		int ln=initial.size();
		int res=N+1, recidx=-1;
		for(int i=0; i<ln; i++){
			st.clear(); flags=vector<int>(N, 0);
			for(int j=0; j<ln; j++){
				if(j==i)continue;
				dfs(graph, initial[j]);
			}
			if(st.size()<res){
                res=st.size();
				recidx=initial[i];
			}
            //cout<<st.size()<<endl;
		}
		return recidx;
    }
};
