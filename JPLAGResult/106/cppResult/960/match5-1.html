class Solution {
public:
    int tot, n;
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        n = graph.size();
        int m = initial.size();
        int ansi = INT_MAX;
        int tmin = INT_MAX;
        for (int i = 0;i < m;++i) {
            vector<bool> infected(n, false);

            tot = m - 1;
            for (int j = 0;j < m;++j) {
                if(i != j) {
                    infected[initial[j]] = true;
                    dfs(initial[j], infected, graph);
                }
            }
            cout << tot << endl;
            if (tot < tmin || (tot == tmin && initial[i] < ansi)) {
                tmin = tot;
                ansi = initial[i];
            }
        }
        return ansi;
    }
    
    void dfs(int v, vector<bool>& infected, vector<vector<int>>& graph) {

        for (int i = 0;i < n;++i) {
            if(graph[v][i] == 1 && !infected[i]) {
                infected[i] = true;
                tot++;
                dfs(i, infected, graph);
            }
        }
    }
};
