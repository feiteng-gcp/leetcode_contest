class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int count=INT_MIN;
        int index = 0;
        sort(initial.begin(),initial.end());
        queue<int> q;
        for(int n=0;n<initial.size();++n)
        {
            int i=initial[n];
            int ccount=0;
            q.push(i);
            bool m[graph.size()] = {false};
            memset(m, false, graph.size()*sizeof(bool));
            while(q.size())
            {
                int curr = q.front();
                q.pop();
                for(int j=0;j<graph[curr].size();++j)
                {
                    if(!m[j] && graph[curr][j])
                    {
                        m[j]=true;
                        q.push(j);
                        ++ccount;
                    }
                }
            }
            if(ccount > count)
            {
                index = i;
                count=ccount;
            }
        }
        return index;
        
    }
};
