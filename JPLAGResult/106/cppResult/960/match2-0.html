class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        std::sort(initial.begin(), initial.end());
        int min = INT_MAX;
        int ans = initial[0];
        for (int i = 0; i < initial.size(); i++) {
            int now = spread(graph, initial, i);
            //cout << now << " " << i <<endl;
            if (-1 != now && now < min) {
                min = now;
                ans = initial[i];
            }
        }
        return ans;
    }
    int spread(vector<vector<int>>& graph, vector<int> &initial, int removed)
    {
        std::set<int> dead;
        std::queue<int> q;
        for (int idx = 0; idx < initial.size(); idx++) {
            if (idx == removed) continue;
            q.push(initial[idx]);
        }
        while (!q.empty()) {
            int cur = q.front(); q.pop();
            dead.insert(cur);
            for (int i = 0; i < graph.size(); i++) {
                if (0 != graph[cur][i]) {
                    if (1 == dead.count(i)) continue;
                    if (i == initial[removed]) return -1;
                    q.push(i);
                }
            }
        }
        
        return 1 == dead.count(initial[removed]) ? -1 : dead.size();
    }
};
