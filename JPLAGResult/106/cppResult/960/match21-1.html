class Solution {
public:
    int uf [ 305 ] ; 
    int vis[305 ] ; 
    int find( int u ) {
        if( uf[u] == u ) 
            return u;
        return uf[u] = find( uf[u] ) ; 
    }
    void Union( int u , int v ) {
        int fu = find( u ) ; 
        int fv = find( v ) ; 
        if( fu == fv ) 
            return ;
        uf[fv] = fu ; 
    }
    int minMalwareSpread(vector<vector<int>>& G , vector<int>& initial) {
        sort( initial.begin(), initial.end() ) ; 
        
        memset( uf , 0 , sizeof( uf ) ) ; 
        memset( vis , 0 , sizeof( vis ) ) ; 
        

        for( int i=0 ; i< G.size() ; i++){
            uf[i] = i ; 
        }
        
        for( int i=0 ; i<G.size() ; i++) {
            for( int j=0 ; j< G.size() ; j++){
                if( i!=j &&G[i][j] ) 
                    Union( i , j ) ;
            }
        }
        for( int i=0 ; i < G.size() ; i++){  
            map< int , vector< int >  > temp ; 
            for( auto & x  :  initial ) {
                int fx = find( x ) ;
                int fi = find( i ) ; 
                if( fx == fi && i != x )  
                    temp[i].push_back( x ) ; 
            }
            if( temp[i].size() == 1 ) {
                vis[temp[i][0]]++;
            }

        }
        
        int ans = initial[0] ; 
        int size = vis[ans] ; 
        
        
        for( auto & x : initial ) {
            if( size < vis[x] ){
                vis[x] = size ; 
                ans = x ; 
            }
        }
        
        return ans ; 
        
    }
};
