class UnionFind {
public:
    
    UnionFind(int N) {
        id = new int[N];
        sz = new int[N];
        for (int i = 0; i < N; i++) {
            id[i] = i;
            sz[i] = 1;
        }
    }
    
    ~UnionFind() {
        delete []id;
        delete []sz;
    }
    
    bool find(int p, int q) {
        return root(p) == root(q);
    }
    
    void unite(int p, int q) {
        int i = root(p);
        int j = root(q);
        
        if (i == j) return;
        
        if (sz[i] > sz[j]) {
            id[j] = i;
            sz[i] += sz[j];
        } else {
            id[i] = j;
            sz[j] += sz[i];
        }
    }
    
    int getSize(int i) {
        int r = root(i);
        return sz[r];
    }
    
private:
    
    int root(int i) {
        while (i != id[i]) {
            id[i] = id[id[i]];
            i = id[i];
        }
        return i;
    }
    
private:
    
    int* id;
    int* sz;
};


class Solution {
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int n = graph.size();
        UnionFind uf(n);
        

        for (int i = 0; i < n; ++i) {
            for (int j = i+1; j < n; ++j) {
                if (graph[i][j] == 1) {
                    uf.unite(i, j);
                }
            }
        }
        
        vector<int> sz(n, -1);
        
        for (int j = 0; j < initial.size(); ++j) {
                int v = initial[j];
                sz[v] = 0;
        }
        // for (int i = 0; i < n; ++i) {
        //     cout << sz[i] << endl;
        // }
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < initial.size(); ++j) {
                int v = initial[j];
                if (uf.find(i, v)) {
                    ++sz[v];
                    // cout << i << "," << v << "-----\n";
                }
            }
        }
        
        for (int i = 0; i < initial.size(); ++i) {
            int u = initial[i];
            for (int j = 0; j < initial.size(); ++j) {
                int v = initial[j];
                if (u != v && uf.find(u, v)) {
                    sz[u] = 0;
                    break;
                }
            }

        }
        
//         cout << "2:3 " << uf.find(2, 3) << endl;
        
//         for (int i = 0; i < n; ++i) {
//             cout << sz[i] << endl;
//         }
        
        int maxIndex = -1;
        int m = -2;
        for (int i = 0; i < n; ++i) {
            if (sz[i] > m) {
                m = sz[i];
                maxIndex = i;
            }
        }
        return maxIndex;
    }
};
