class Solution {
    
private:
    int find(vector<int> &uf, int i){
        if(uf[i]!=i){
            int j = find(uf, uf[i]);
            uf[i] = j;
        }
        return uf[i];
    }
    void unions(vector<int> &uf, int i, int j){
        if(find(uf,i)!=find(uf, j)){
            uf[find(uf, j)] = find(uf, i);
        }
    }
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int n= graph.size();
        vector<int> uf(n);
        for(int i=0; i<n;i++){
            uf[i] = i;
        }
        for(int i=0; i<n; i++){
            for(int j=i+1; j<n; j++){
                if(graph[i][j]==1){
                    unions(uf, i, j);
                }
            }
        }
        int res=n, num = 0;
        /*for(int i:uf){
            cout<<i<<endl;
        }*/
        for(int i=0; i<initial.size(); i++){
            bool withOther = false;
            for(int j=0; j<initial.size();j++){
                if(i!=j && find(uf, initial[i]) == find(uf, initial[j])){
                    withOther = true;
                    break;
                }
            }
            int counts = 0;
            if(!withOther){
                for(int j:uf){
                    if(find(uf, j)==find(uf, initial[i]))
                        counts++;
                }
            }
            //cout<<initial[i]<<counts<<endl;
            if(num<counts){
                num = counts;
                res = initial[i];
            }else if(num==counts && initial[i]<res){
                res = initial[i];
            }
        }
        return res;
    }
};
