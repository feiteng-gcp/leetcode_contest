class Solution {

public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int n = graph.size();
        vector<int> root(n, 0);
        for(int i = 0;i < n;i++){
            root[i] = i;
        }
        
        sort(initial.begin(), initial.end());
        
        for(int i = 0;i < n;i++){
            for(int j = 0;j < n;j++){
                if(graph[i][j]){
                    int a = find(root, i), b = find(root, j);
                    if(a != b)
                        root[a] = b;
                }
            }
        }
        
        unordered_map<int, int> mp;
        
        unordered_map<int, vector<int>> groups;
        for(int i = 0;i < n;i++){
            // cout<<find(root,i)<<endl;
            mp[find(root,i)]++;
        }
        
        // for(auto it: mp){
        //     cout<<it.first<<":"<<it.second<<endl;
        // }
        for(int i = 0;i < initial.size();i++){
            groups[find(root,initial[i])].push_back(initial[i]);
        }
        int maxVal = 0;
        int res = -1;
        for(auto it:groups){
            if(mp[it.first] > maxVal){
                maxVal = mp[it.first];
                res = it.second[0];
            }
            if(mp[it.first] == maxVal && res > it.second[0]){
                res = it.second[0];
            }
        }
        // for(int i = 0;i < initial.size();i++){
        //     if(mp[root[initial[i]]] > maxVal){
        //         maxVal = mp[root[initial[i]]];
        //         res = i;
        //     }
        // }
        return res;
    }
    
    int find(vector<int> &root, int index){
        if(root[index] != index)
            root[index] = find(root, root[index]);
        return root[index];
    }
};
