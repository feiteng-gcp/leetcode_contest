class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        odd = []
        even 
= []
        
        for n in A:
            if n % 2 == 0:
                even.append(n)
            else:
                odd.append(n)
        ans = []
        
        for i in range(len(odd)):
            ans.append(even[i])
            ans.append(odd[i])
        return ans
