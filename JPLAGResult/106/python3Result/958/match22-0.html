class Solution:
    def sortArrayByParityII(self, A):
        la=len(A)
        ans=[0]*la
        ji,ou=1,0
        for n in A:
            if n%2==0:
                ans[ou]=n
                ou+=2
            else:
                ans[ji]=n
                ji+=2
        return ans
        """
        :type A: List[int]
        :rtype: List[int]
        """
