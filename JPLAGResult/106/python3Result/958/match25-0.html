class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        n = len(A)
        b = [0] * n
        i = 0
        j = 1
        for num in A:
            if num % 2 == 0:
                b[i] = num
                i += 2
            else:
                b[j] = num
                j += 2
        return b
