class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        result = [0] * len(A)
        odd_idx, even_idx = 1, 0
        for a in A:
            if a % 2 == 0:
                result[even_idx] = a
                even_idx += 2
            else:
                result[odd_idx] = a
                odd_idx += 2
        return result
