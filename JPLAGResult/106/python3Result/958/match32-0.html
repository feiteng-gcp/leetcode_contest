class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        if A==[]:
            return []
        B = A+[]
        odd = 1
        even = 0
        for x in A:
            if x%2==0:
                B[even]=x
                even+=2
            else:
                B[odd]=x
                odd+=2
        return B
