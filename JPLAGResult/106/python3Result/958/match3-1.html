class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        
        even = []
        odd 
= []
        for a in A:
            if a %2 == 0:
                even.append(a)
            else:
                odd.append(a)
        ans = []
        for i in range(len(A)//2):
            ans.append(even[i])
            ans.append(odd[i])
        return ans
