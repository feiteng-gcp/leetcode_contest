class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        odd, even = 1, 0
        result = [0] * len(A)
        
for i in range(0, len(A)):
            if A[i] % 2 == 0:
                result[even] = A[i]
                even += 2
            else:
                result[odd] = A[i]
                odd += 2
        return result
