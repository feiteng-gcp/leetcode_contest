class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        ret=[None for i in range(len(A))]
        even=0
        odd=1
        for i in A:
            if i%2==0:
                ret[even]=i
                even+=2
            else:
                ret[odd]=i
                odd+=2
        return ret
