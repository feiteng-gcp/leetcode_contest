class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        odd = [x for x in A if x % 2 == 1]
        even = [x for x in A if x % 2 == 0]
        
        res = []
        for x, y in zip(odd, even):
            res.append(y)
            res.append(x)
        return res
