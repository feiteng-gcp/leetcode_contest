class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """

        l, r, n = 0, 1, len(A)
        
while l < n and r < n:
            while l < n and A[l] % 2 == 0:
                l += 2
            while r < n and A[r] % 2:
                r += 2
            if l < n and r < n:
                A[l], A[r] = A[r], A[l]
                l += 2
                r += 2
        return A
