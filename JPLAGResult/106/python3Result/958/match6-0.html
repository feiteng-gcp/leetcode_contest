class Solution:
    def sortArrayByParityII(self, A):
        B=[]
        C=[]
        for a in A:
            if a%2==0:
                B.append(a)
            else:
                C.append(a)

        ANS=[]

        for j in range(len(B)):
            ANS.append(B[j])
            ANS.append(C[j])

        return ANS
            
        """
        :type A: List[int]
        :rtype: List[int]
        """
