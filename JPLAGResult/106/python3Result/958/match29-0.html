class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        result = [0] * len(A)
        ind = 0
        indd = 1
        for i in range(len(A)):
            if A[i] % 2 == 0:
                result[ind] = A[i]
                ind += 2
            else:
            
                result[indd] = A[i]
                indd += 2
        return result
