class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        o, e = [], []
        for i in A:
            if i % 2 == 0:
                o.append(i)
            else :
                e.append(i)
                
        ans = []
        for i in range(len(o)):
            ans.append(o[i])
            ans.append(e[i])
            
        
        return ans
