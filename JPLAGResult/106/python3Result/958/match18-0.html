class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        even=[a for a in A if a%2==0]
        odd=[a for a in A if a%2==1]
        ret=[]
        for a,b in zip(even,odd):
            ret.append(a)
            ret.append(b)
        return ret
