class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        n=len(A)
        L=[0]*n
        i=0
        j=n-1
        for a in A:
            if a%2==0:
                L[i]=a
                i+=2
            else:
                L[j]=a
                j-=2
        return L
