class Solution:
    def sortArrayByParityII(self, A):
        """
        :type A: List[int]
        :rtype: List[int]
        """
        ret = [0] * len(A)
        e 
= 0
        o = 1 
        for x in A:
            if x % 2 == 1:
                ret[o] = x
                o += 2
            else:
                ret[e] = x
                e += 2
        return ret
