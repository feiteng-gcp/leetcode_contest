class Solution:
    def minAddToMakeValid(self, S):
        """
        :type S: str
        :rtype: int
        """
        result = 0
        num_left = 0
        for ss in S:
            if ss == '(':
                num_left += 1
            else:
                if num_left > 0:
                    num_left -= 1
                else:
                    result += 1
        return result + num_left
