class Solution:
    def minAddToMakeValid(self, S):
        """
        :type S: str
        :rtype: int
        """
        n = 0
        r = 0
        for c in S:
            if c == "(":
                n += 1
            elif c == ")":
                if n == 0:
                    r += 1
                else:
                    n -= 1
        return r + n
