class Solution:
    def minAddToMakeValid(self, S):
        """
        :type S: str
        :rtype: int
        """
        res = 0
        count = 0
        for c in S:
            if c == '(':
                count += 1
            else:
                if count == 0:
                    res += 1
                else:
                    count -= 1
        res += count
        return res
