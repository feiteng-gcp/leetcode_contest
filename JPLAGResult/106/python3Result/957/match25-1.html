class Solution:
    def minAddToMakeValid(self, S):
        """
        :type S: str
        :rtype: int
        """
        ans = count = 0
        for s in S:
            if s == '(':
                count += 1
            else:
                if count>0:
                    count -= 1
                else:
                    ans += 1
        ans += count
        return ans
