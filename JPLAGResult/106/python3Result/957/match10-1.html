class Solution:
    def minAddToMakeValid(self, S):
        """
        :type S: str
        :rtype: int
        """
        ans = 0
        left = 0
        for ch in S:
            if ch == '(':
                left += 1
            else:
                if left==0:
                    ans += 1
                else:
                    left -= 1
        ans += left
        return ans
