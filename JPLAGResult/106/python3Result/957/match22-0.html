class Solution:
    def minAddToMakeValid(self, S):
        """
        :type S: str
        :rtype: int
        """
        stack = []
        for s in S:
            if s == "(":
                stack.append(s)
            else:
                if not stack or stack[-1] != "(":
                    stack.append(s)
                else:
                    stack.pop()
        return len(stack)
