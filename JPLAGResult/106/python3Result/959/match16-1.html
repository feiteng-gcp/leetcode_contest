class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        def combination(total, pick):
            if pick == 2:
                return total * (total - 1) // 2
            if pick == 3:
                return total * (total - 1) * (total - 2) // 6
        
        
        count = collections.Counter(A)
        nums = sorted(count.keys())
        res = 0
        # print(nums)
        
        for i in range(len(nums)):
            # print("---------")
            # print("i = ", i, "nums[i] = ", nums[i])
            # ai == aj == ak
            if count[nums[i]] >= 3 and 3 * nums[i] == target:
                res += combination(count[nums[i]], 3)
                # print("1 ", res)
            # ai == aj < ak
            elif count[nums[i]] >= 2 and target - 2 * nums[i] in count and nums[i] < target - 2 * nums[i]:
                res += combination(count[nums[i]], 2) * count[target - 2 * nums[i]]
                # print("2 ", res)
            
            # ai < aj == ak or ai < aj < ak
            # else:
            jandk = target - nums[i]
            # print("jandk = ", jandk)
            for j in range(i + 1, len(nums)):
                if nums[j] > jandk // 2:
                    break
                if 2 * nums[j] == jandk and nums[j] > nums[i] and count[nums[j]] >= 2:
                    res += count[nums[i]] * combination(count[nums[j]], 2)
                    # print("3 ", res)
                elif jandk - nums
[j] > nums[j] and count[jandk - nums[j]] >= 1:
                    res += count[nums[i]] * count[nums[j]] * count[jandk - nums[j]]
                    # print("4 ", res)
            
        return res % (10 ** 9 + 7)
