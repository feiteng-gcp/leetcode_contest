class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        N=int(1E9+7)
        t={}
        res=0
        for i in A:
            t[i]=t.get(i,0)+1
        s=list(t.keys())
        s_s=sorted(s)
        for i in range(len(s_s)-2):
            tmp_target=target-s_s[i]
            j=i+1
            k
=len(s_s)-1
            while (j<k):
                if s_s[j]+s_s[k]==tmp_target:
                    res=(res+(((t[s_s[j]]*t[s_s[k]]))*t[s_s[i]]))%N
                    j+=1
                    k-=1
                    continue
                if s_s[j]+s_s[k]>tmp_target:
                    k-=1
                    continue
                if s_s[j]+s_s[k]<tmp_target:
                    j+=1
        for i in range(len(s_s)):
            if s_s[i]*3==target:
                tmp=t[s_s[i]]
                res_tmp=int((((tmp*(tmp-1)) )*(tmp-2))/6)
                res=(res+res_tmp) % N
                break
        for i in range(len(s_s)):
            tmp=target-2*s_s[i]
            if tmp==s_s[i]:
                continue
            if t.get(tmp)!=None:
                tmp1=t[s_s[i]]
                res_tmp=(int(((tmp1*(tmp1-1)) )/2)*t[tmp])%N
                res=(res+res_tmp) % N
        return res
