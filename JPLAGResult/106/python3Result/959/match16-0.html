class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        def C(n,m):
            K = 1
            for k in range(n-m+1,n+1):
                K*=k
            for k in range(1,m+1):
                K/=k
            return K%(1e9+7)
        size = len(A)
        Ans = 0

        A.sort()
        Dict = {}
        for x in A:
            if x in Dict:
                Dict[x]+=1
            else:
                Dict[x]=1
        for i in range(len(A)-2):
            if (i>0 and A[i]==A[i-1]):
                continue
            j,k = i+1, len(A)-1
            while(j<k):
                while (A[i]+A[j]+A[k]!=target and j<k):
                    if (A[i]+A[j]+A[k]<target):
                        j += 1
                    else:
                        k -= 1
                if (j<k and A[i]+A[j]+A[k]==target):
                    if (A[i]==A[j]):
                        if (A[j]==A[k]):
                            Ans+=C(Dict[A[i]],3)
                        else:
                            Ans+=C(Dict[A[i]],2)*Dict[A[k]]
                    elif (A[j]==A[k]):
                        Ans+=C(Dict[A[j]],2)*Dict[A[i]]
                    else:
                        Ans+=Dict[A[i]]*Dict[A[j]]*Dict[A[k]]
                    Ans%=(1e9+7)
                    j += 1
                    while(j<k and A[j]==A[j-1]):
                        j += 1
                    k -= 1
                    while(j<k and A[k]==A[k+1]):
                        k -= 1
        return int(Ans)
