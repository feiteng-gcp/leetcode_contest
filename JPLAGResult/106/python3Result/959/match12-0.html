def two_sum(nums, target):
    l, r = 0, len(nums) - 1
    out = []
    while l < r:
        if nums[l] + nums[r] < target:
            l += 1
        elif nums[l] + nums[r] > target:
            r -= 1
        else:
            out.append((nums[l], nums[r]))
            l += 1
            r -= 1
            while l < r and nums[l] == nums[l-1]:
                l += 1
            while l < r and nums[r] == nums[r+1]:
                r -= 1   
    return out

def three_sum(nums, target):
    out = []
    for i in range(len(nums)):
        if i > 0 and nums[i] == nums[i-1]:
            continue
        out.extend((nums[i],) + r for r in two_sum(nums[i+1:], target-nums[i]))
        i += 1
    return out

class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        A.sort()
        all_triples = three_sum(A, target)
        print(all_triples)
        cmul = dict(collections.Counter(A))

        out = 0
        for tup in all_triples:
            x,y,z = sorted(tup)
            ct = 0
            if len(set(tup)) == 3:
                ct = cmul[x] * cmul[y] * cmul[z]

            elif x == y and y == z:
                ct= cmul[x] * (cmul[x]-1)*(cmul[x]-2) // 6
            elif x == y:
                ct= cmul[x] * (cmul[x] - 1) * cmul[z] // 2
            else:
                ct= cmul[x] * cmul[y] * (cmul[y] - 1)// 2
            
            out += ct
            out %= 10**9 + 7
        return out
