class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        target -= 3 * min(A)

        bigNum = 10 ** 9 + 7
        allNum = [0] * (max(A) + 1 - min(A))
        # Corner cases ha
        for i in A:
        	allNum[i - min(A)] += 1
        
        count = 0
        for i in range(len(allNum)):
        	j = i
        	k = len(allNum) - 1
        	
        	while j <= k:

        		if i + j + k > target:
        			k -= 1
        		elif i + j + k < target:
        			j += 1
        		else:
        			if i == j and j == k and allNum[i] >= 3:
        				count += (allNum
[i] * (allNum[i] - 1) * (allNum[i] - 2)) // 6
        			elif i == j and j < k and allNum[i] >= 2:
        				
        				count += ((allNum[i] * (allNum[i] - 1)) // 2) * allNum[k]

        			elif i < j and j == k and allNum[j] >= 2:
        				count += allNum[i] * ((allNum[j] * (allNum[j] - 1)) // 2)
        			elif i < j < k:
        				count += allNum[i] * allNum[j] * allNum[k]

        			j += 1
        			k -= 1
        		if count >= bigNum:
        			count %= bigNum
        return count % bigNum
