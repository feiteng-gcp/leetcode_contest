import heapq
import math
from collections import defaultdict

class Solution:
    def threeSumMulti(self, A, target):
        a = [0]*101
        mod = 1000000007
        for i in A:
            a[i] +=1
        #print(a)
        res = 0
        for i in range(0,101):
            for j in range(i,101):
                k = target-i-j
                if k<j or k>100:
                    continue
                vi = a[i]
                vj = a[j]
                vk = a[k]
                v = 0
                if j==i and k == j:
                    if vi<3:
                        continue
                    else:
                        v = math.factorial(vi)/(math.factorial(vi-3)*6)
                elif i ==j:
                    if vi <2:
                        continue
                    else:
                        v = math.factorial(vi)/(math.factorial(vi-2)*2) * vk
                elif j ==k:
                    if vj <2:
                        continue
                    else:
                        v = math.factorial(vj)/(math.factorial(vj-2)*2) * vi
                else:
                    v = vi*vj*vk
                res=(res+int(v))%mod
                #print(i,j,k,res)
                
        return res
