class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        
        cnt = dict()
        for a in A:
            if a not in cnt:
                cnt[a] = 1
            else:
                cnt[a] += 1
        
        B = []
        for key in cnt:
            B.append(key)
        B.sort()
        n = len(B)
        ans = 0
        
        # 3 different
        for i, b in enumerate(B):
            # print(i, b)
            newt 
= target - b
            le, ri = i+1, n-1
            while le < ri:
                if B[le] + B[ri] > newt:
                    ri -= 1
                elif B[le] + B[ri] < newt:
                    le += 1
                else:
                    ans += cnt[b]*cnt[B[le]]*cnt[B[ri]]
                    le += 1
        # print(B)
        # print(ans)
        # 2 or 3 same
        for b in B:
            newt = target - 2 * b
            if newt == b:
                ans += cnt[b]*(cnt[b]-1)*(cnt[b]-2)//6
            elif newt in cnt:
                ans += cnt[b]*(cnt[b]-1)//2*cnt[newt]
        
        return ans%1000000007
