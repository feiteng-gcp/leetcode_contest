from math import factorial

class Solution:
    def getThreeCount(self, target, memo, nums):
        if len(nums) <= 2:
            return 0
        
        ans = 0
        for i in range(len(nums) - 2):
            k = target - nums[i]
            left, right = i + 1, len(nums) - 1
            while left < right:
                if nums[left] + nums[right] == k:
                    ans += memo[nums[i]] * memo[nums[left]] * memo[nums[right]]
                    left += 1
                elif nums[left] + nums[right] < k:
                    left += 1
                else:
                    right -= 1
        return ans
        
    def getTwoCount(self, target, memo, nums):
        if target % 3 == 0 and target // 3 in memo:
            nums = [num for num in nums if num != target // 3]
        ans = 0
        for i in nums:
            if (target - i) % 2 != 0:
                continue
            temp_num = (target - i) // 2
            if temp_num in memo and memo[temp_num] >= 2:
                temp = memo[i] * (factorial(memo[temp_num]) // 2 // factorial(memo[temp_num] - 2))
                ans += temp
        return ans
        
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        if not A:
            return 0
        
        memo = {}
        for i in A:
            memo[i] = memo.get(i, 0) + 1
        
        nums = list(memo.keys())
        nums.sort()
        
        ans = 0
        if target % 3 == 0 and target // 3 in memo and memo[target // 3] >= 3:
            ans += factorial(memo[target // 3]) // factorial(3) // factorial(memo[target // 3] - 3)
        
        ans += self.getTwoCount(target, memo, nums)
        ans += self.getThreeCount(target, memo, nums)
        
        return ans % (10 ** 9 + 7)
