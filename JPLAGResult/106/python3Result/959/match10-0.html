def Combi(a,b,mod):
    if b>a:
        return 0
    ANS=1
    for i in range(min(b,a-b)):
        ANS=ANS*(a-i)*pow(min(b,a-b)-i,mod-2,mod)

    return ANS%mod


class Solution:
    def threeSumMulti(self, A, target):
        LIST=[0]*101
        mod=10**9+7
        
        for a in A:
            LIST[a]+=1

        ANS=0
        for j in range(101):
            for i in range(j+1):
                if j<=target-i-j<=100:
                    k=target-i-j
                    if i!=j and j!=k and k!=i:
                        ANS+=LIST[k]*LIST[i]*LIST[j]%mod
                    elif (i==j and j!=k):
                        ANS+=Combi(LIST[i],2,mod)*LIST[k]%mod
                    elif (i!=j and j==k):
                        ANS+=Combi(LIST[j],2,mod)*LIST[i]%mod
                    else:
                        ANS+=Combi(LIST[i],3,mod)%mod
                        
                        


        return ANS
