class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        data=dict()
        for i in A:
            if i in data:
                data[i]+=1
            else:
                data[i]=1
        #print(data)
        ret=0
        mod=1000000007
        for mid in data:
            for mid2 in data:
                if mid2<mid:
                    continue
                if mid==mid2:
                    if mid*3==target and data[mid]>=3:
                        ret+=data[mid]*(data[mid]-1)*(data[mid]-2)//6
                        ret=ret%mod
                    else:
                        if target-mid*2!=mid and target-mid*2 in data and data[mid]>=2:
                            '''
                            print("ans")
                            print(mid)
                            print(mid)
                            print(target-mid*2)
                            print((data[mid]*(data[mid]-1)//2)*(data[target-mid*2]))
                            '''
                            ret+=(data[mid]*(data[mid]-1)//2)*(data[target-mid*2])
                            ret=ret%mod
                else:
                    if target-mid-mid2>mid and target-mid-mid2>mid2 and target-mid-mid2 in data:
                        '''
                        print("ans")
                        print(mid)
                        print(mid2)
                        print(target-mid-mid2)
                        print(data[mid]*data[mid2]*data[target-mid-mid2])
                        '''
                        ret+=data[mid]*data[mid2]*data[target-mid-mid2]
                        ret=ret%mod
        return ret
