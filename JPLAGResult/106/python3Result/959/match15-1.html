class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        MOD = 10 ** 9 + 7
        nums = sorted(set(A))
        cont = {num:A.count(num) for num in nums}
        res = 0
        if target % 3 == 0 and target // 3 in nums and cont[target //3] >= 3:
            res += cont[target //3] * (cont[target //3] - 1) * (cont[target //3] - 2) // 6
            res %= MOD
        l = len(nums)
        for i in range(l - 1):
            for j in range(i + 1, l):
                if nums[j] > (target - 2 * nums[i]):
                    break
                if (nums[j] +  2 * nums[i]) == target:
                    res += cont[nums[j]] * cont[nums[i]] * (cont[nums[i]] - 1) // 2
                
if (2 * nums[j] +  nums[i]) == target:
                    res += cont[nums[i]] * cont[nums[j]] * (cont[nums[j]] - 1) // 2
                res %= MOD
        for i in range(l - 2):
            for j in range(i + 1, l - 1):
                for k in range(j + 1, l):
                    if (nums[j] + nums[i] + nums[k]) > target:
                        break
                    if (nums[j] + nums[i] + nums[k]) == target:
                        res += cont[nums[i]] * cont[nums[j]] * cont[nums[k]]
                        res %= MOD
                        break
        return res
