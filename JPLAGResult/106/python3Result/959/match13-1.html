class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        dp = [[0 for _ in range(target+1)] for _ in range(3)]
        mod = 10 ** 9 + 7
        for x in A:
            j = target
            while j - x >= 0:
                dp
[2][j] = (dp[2][j] + dp[1][j-x]) % mod
                dp[1][j] = (dp[1][j] + dp[0][j-x]) % mod
                j-=1
            if x<=target:
                dp[0][x] = (dp[0][x] + 1) % mod
        return dp[2][target]
