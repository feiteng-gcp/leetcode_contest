class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        MOD, dic = 10 ** 9 + 7, collections.defaultdict(int)
        for a in A:
            dic[a] += 1
        keys = list(dic.keys())
        n, ans, save = len(keys), 0, set()
        for i in range(n):
            for j in range(i, n):
                if target - keys[i] - keys[j] in dic:
                    temp = tuple(sorted([keys[i], keys[j], target - keys[i] - keys[j]]))
                    if temp not in save:
                        if temp[0] == temp[2]:
                            if dic[temp[0]] > 2:
                                ans += dic[temp[0]] * (dic[temp[0]] - 1) * (dic[temp[0]] - 2) // 6
                        elif temp[0] == temp[1]:
                            if dic[temp[0]] > 1:
                                ans += dic[temp[0]] * (dic[temp[0]] - 1) // 2 * dic[temp[2]]
                        elif temp[1] == temp[2]:
                            if dic[temp[1]] > 1:
                                ans += dic[temp[1]] * (dic[temp[1]] - 1) // 2 * dic[temp[0]]
                        else:
                            ans += dic[temp[0]] * dic[temp[1]] * dic[temp[2]]
                        save.add(temp)
        return ans % MOD
