class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        return self.firstIncorrect(A, target)
    
    def firstIncorrect(self, A, target):
        from collections import Counter
        countA = Counter(A)
        A = sorted(A) # now this is all unique
        res = set()
        n = len(A)
        for i in range(n-2):
            if i > 0 and A[i-1] == A[i]:
                continue
            lo, hi = i+1, n-1
            while lo < hi:
                sum3 = A[i] + A[lo] + A[hi]
                if sum3 > target:
                    hi -= 1
                elif sum3 < target:
                    lo += 1
                else:
                    # found one
                    res.add((A[i], A[lo], A[hi]))

                    # cnt = countA[A[i]] * countA[A[lo]] * countA[A[hi]]
                    # res += cnt
                    # res %= (10**9+7)

                    
while lo < hi and A[lo] == A[lo+1]:
                        lo += 1
                    while lo < hi and A[hi-1] == A[hi]:
                        hi -= 1

                    lo += 1
                    hi -= 1

        total = 0
        for a, b, c in res:
            if a == b and b == c:
                cnt = countA[a]
                total += ((cnt) * (cnt-1) * (cnt-2) // 6)
            elif a == b:
                cntAB = countA[a]
                total += (cntAB * (cntAB-1) // 2) * countA[c]
            elif b == c:
                cntBC = countA[b]
                total += (cntBC * (cntBC-1) // 2) * countA[a]
            else:
                total += (countA[a] * countA[b] * countA[c])
            
            total %= (10**9 + 7)

        return total
