from scipy.special import comb, perm
MO = 10**9 + 7
class Solution:
    def threeSumMulti(self, A, target):
        """
        :type A: List[int]
        :type target: int
        :rtype: int
        """
        if not A:
            return 0
        import collections
        counter = collections.defaultdict(int)
        my_list = []
        for num in A:
            counter[num] += 1
            if counter[num] == 1:
                my_list.append(num)
        my_list.sort()
        start = 0
        end = len(my_list) - 1
        ans = 0
        if 3 * my_list[end] == target and counter[my_list[end]] >= 3:
            ans += int(comb(counter[my_list[end]], 3))
            ans %= MO
            print("5")
            return ans
        if 3 * my_list[end] < target:
            return 0
        while start < end and 3 * my_list[start] <= target:
            if 3 * my_list[start] == target and counter[my_list[start]] > 3:
                ans += int(comb(counter[my_list[start]], 3))
                ans %= MO
                break
            elif target - my_list[start] >= 0 and (target - my_list[start]) % 2 == 0 and (target - my_list[start]) // 2 in counter and counter[(target - my_list[start]) // 2] >= 2 and (target - my_list[start]) // 2 != my_list[start]:
                ans += int(comb(counter[(target - my_list[start]) // 2], 2)) * counter[my_list[start]]
                ans %= MO

            new_target = target - my_list[start]
            left = start + 1
            right = end
            while left < right:
                if my_list[left] + my_list[right] > new_target:
                    right -= 1
                elif my_list[left] + my_list[right] < new_target:
                    left += 1
                else:
                    ans += counter[my_list[left]]* counter[my_list[right]] * counter[my_list[start]]
                    ans %= MO
                    left += 1
                    right -= 1
            start += 1
        while start < end:
            if target - my_list[start] >= 0 and (target - my_list[start]) % 2 == 0 and (target - my_list[start]) // 2 in counter and counter[(target - my_list[start]) // 2] >= 2 and (target - my_list[start]) // 2 != my_list[start]:
                ans += int(comb(counter[(target - my_list[start]) // 2], 2)) * counter[my_list[start]]
                ans %= MO
            start += 1
        if target - my_list[end] >= 0 and (target - my_list[end]) % 2 == 0 and (target - my_list[end]) // 2 in counter and counter[(target - my_list[end]) // 2] >= 2 and (target - my_list[end]) // 2 != my_list[end]:
                ans += int(comb(counter[(target - my_list[end]) // 2], 2)) * counter[my_list[end]]
                ans %= MO
        
        return ans
