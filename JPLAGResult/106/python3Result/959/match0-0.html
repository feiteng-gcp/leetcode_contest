class Solution:
    def threeSumMulti(self, A, target):
        total = 0
        if not A or len(A) == 2:
            return 0
        
        m = [[0]*101 for x in range(len(A))]
        
        for i in range(1, len(A)):
            for j in range(101):
                m[i][j] = m[i-1][j]
                
            m[i][A[i-1]]+=1
            
        m2 = [[0]*101 for x in range(len(A))]
        for i in range(len(A)-2, -1, -1):
            for j in range(101):
                m2[i][j] = m2[i+1][j]
                
            m2[i][A[i+1]]+=1
            
        for i in range(1, len(A)-1):
            s = target - A[i]
            
            for j in range(s+1):
                if j > 100 or s-j > 100:
                    continue
                total += (m[i][j]*m2[i][s-j] % (10**9 + 7))
                total = total % (10**9 + 7)
                
        return total
