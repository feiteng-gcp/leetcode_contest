class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """

        initial.sort()

        def simulate(graph, initial):
            pending = initial
            visited = set(initial)

            while pending:
                node = pending.pop()

                for i, neigh in enumerate(graph[node]):
                    if not neigh:
                        continue
                    if i in visited:
                        continue
                    visited.add(i)
                    pending.append(i)

            return len(visited)

        min_infected = float('inf')
        min_index 
= 0

        for i in range(len(initial)):
            infected = simulate(graph, initial[:i] + initial[i + 1:])
            # print(infected)
            if infected < min_infected:
                min_infected = infected
                min_index = i

        return initial[min_index]
