import heapq
import math
from collections import defaultdict
class Solution:
    def minMalwareSpread(self, graph, initial):
        l = len(graph)
        d = [0]*l
        size = [1]*l
        for i in range(l):
            d[i] = i
        
def find(x):
            #print(i,d[i])
            if d[x] == x:
                return x
            d[x] = find(d[x])
            return d[x]
        for i in range(l):
            for j in range(i+1,l):
                if graph[i][j] ==1:
                    ri = find(i)
                    rj = find(j)
                    if ri!=rj:
                        d[ri] = rj
                        size[j] = size[ri]+size[rj]
        initial = sorted(initial)
        res = initial[0]
        best = 0
        v = {}
        #print(d)
        #print(size)
        for i in range(len(initial)):
            if i in v:
                continue
            hasM = False
            ri = find(initial[i])
            if i == len(initial)-1:
                if size[ri] >best:
                    best = size[ri]
                    res = initial[i]
            for j in range(i+1,len(initial)):
                rj = find(initial[j])
                if ri == rj:
                    hasM = True
                    v[i] = True
                    v[j] = True
            #print("ri",ri,hasM)
            if hasM == False:
                if size[ri] >best:
                    best = size[ri]
                    res = initial[i]
            #print("best",best)
        return res
