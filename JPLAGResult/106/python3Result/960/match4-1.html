import copy

class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """

        self.cur_res = 0
        self.best_res = 500
        res = initial[0]
        def dfs(graph_link, cur, flag, color):
            self.cur_res += 1
            flag[cur] = 1
            color[cur] = 1
            for neigh in graph_link[cur]:
                if flag[neigh] == 0:
                    dfs(graph_link, neigh, flag, color)
        m 
= len(graph)
        graph_link = [[] for _ in range(m)]
        for r in range(m):
            for c in range(m):
                if graph[r][c]:
                    graph_link[r].append(c)
        color_ori = [0 for _ in range(m)]
        for node in initial:
            color_ori[node] = 1

        for node in initial:
            color = copy.deepcopy(color_ori)
            color[node] = 0
            initial_temp = copy.deepcopy(initial)
            this_initial = []
            flag = [0 for _ in range(m)]
            self.cur_res = 0
            for node2 in initial_temp:
                if node2 != node:
                    dfs(graph_link, node2, flag, color)
            if self.cur_res < self.best_res:
                self.best_res = self.cur_res
                res = node
            elif self.cur_res == self.best_res:
                res = min(res, node)
        return res
