from collections import deque
class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        initial = sorted(initial)
        jeop = set(initial)
        n = len(graph)
        visited = [False for _ in range(n)]
        node_connect_matrix = {}
        for i in range(n):
            if not visited[i]:
                temp = set()
                self.helper(graph, i, visited, node_connect_matrix, temp)
        ans, count = None, -1
        # print(node_connect_matrix)
        for x in initial:
            temp, q = set(), deque()
            for k in node_connect_matrix[x]:
                if k!= x and k in jeop:
                    q.append(k)
            while q:
                node = q.popleft()
                for i, n in enumerate(graph[node]):
                    if node != i and n and i not in temp:
                        temp.add(i)
                        q.append(i)
            less = len(node_connect_matrix[x]) -  len(temp)
            
            if less > count:
                ans = x
                count = less
            
        return ans
            
            
                
    def helper(self, graph, node, visited, node_connect, s):
        s.add(node)
        visited[node] = True
        node_connect[node] = s
        for i, n in enumerate(graph[node]):
            if i != node and not visited[i] and n:
                self.helper(graph, i, visited, node_connect, s)
