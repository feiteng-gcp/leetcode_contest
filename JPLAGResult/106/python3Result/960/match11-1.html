class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        g=dict()
        for i in range(len(graph)):
            for j in range(len(graph[i])):
                if graph[i][j]==1:
                    if i not in g:
                        g[i]=set()
                    g
[i].add(j)
        ntog=dict()
        #print(g)
        glist=[]
        for i in range(len(graph)):
            if i not in ntog:
                glist.append(0)
                todo=[i]
                #print("New")
                while len(todo)!=0:
                    #print(todo)
                    todo_next=[]
                    while len(todo)!=0:
                        curr=todo[-1]
                        todo.pop()
                        if curr not in ntog:
                            ntog[curr]=len(glist)-1
                            glist[-1]+=1
                            for j in g[curr]:
                                todo_next.append(j)
                    todo=todo_next
        count=dict()
        for i in initial:
            if ntog[i] not in count:
                count[ntog[i]]=0
            count[ntog[i]]+=1
                
        ret=None
        for i in sorted(initial):
            if ret==None:
                ret=i
            else:
                if count[ntog[i]]==1 and glist[ntog[i]]>glist[ntog[ret]]:
                    ret=i
        return ret
