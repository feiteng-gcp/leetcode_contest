class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        n = len(graph)
        initial = sorted(initial)
        check = [-1 for _ in range(n)]
        res = initial[0]
        num = -1
        for start in initial:
            if check[start] != -1:
                continue
            nodes = []
            check[start] = 1
            stack = [start]
            mark = 1
            while stack:
                nodes += stack
                next_stack = []
                for node in stack:
                    for j in range(n):
                        if graph[node][j] == 1 and check[j] == -1:
                            check[j] = 1
                            next_stack.append(j)
                            if j in initial:
                                mark = 0

                stack = next_stack
            if mark * len(nodes) > num:
                num = mark * len(nodes)
                res = start
        return res
