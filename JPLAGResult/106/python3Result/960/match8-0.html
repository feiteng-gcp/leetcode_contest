class Solution:
    def minMalwareSpread(self, graph, initial):
        g = collections.defaultdict(list)
        
        for i in range(len(graph)):
            for j in range(len(graph)):
                if graph[i][j]:
                    g[i].append(j)
                
        visited = [0]*len(graph)
        
        n = 1
        for i in range(len(graph)):
            if visited[i]: 
                continue
                
            stack = [i]
            while stack:
                j = stack.pop()
                visited[j] = n
                for el in g[j]:
                    if not visited[el]:
                        stack.append(el)
                
            n+=1
        
        v = collections.defaultdict(set)
        for i in range(len(graph)): 
            v[visited[i]].add(i)
          
        
        maxI = min(initial)
        initial = set(initial)
        maxL = 0
        
        for key in v.keys():
            if len(v[key]&initial) == 1:
                if (len(v[key]) > maxL):
                    maxL = len(v[key])
                    maxI = next(iter(v[key]&initial))
                    
        return maxI
            
            
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
