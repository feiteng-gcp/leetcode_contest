class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        n, dic = len(graph), {}
        for i in range(n):
            if i in dic:
                continue
            vtemp, check = {i}, {i}
            while check:
                temp = set()
                for item in check:
                    for j, v in enumerate(graph[item]):
                        if v and j not in dic and j not in vtemp:
                            temp.add(j)
                vtemp |= temp
                check = temp.copy()
            count = len(vtemp)
            for item in vtemp:
                dic[item] = vtemp
        n, exclude = len(initial), set
()
        for i in range(n - 1):
            for j in range(i + 1, n):
                if initial[j] in dic[initial[i]]:
                    exclude.add(initial[i])
                    exclude.add(initial[j])
        temp = set(initial) - exclude
        if temp:
            return min(temp, key = lambda x: (-len(dic[x]), x))
        return min(initial)
