class Solution:
    
    def minMalwareSpread(self, graph, initial):
        def find(x):#find(a)=find(b)のとき同じグループ

            while Group[x] != x:
                x=Group[x]
            return x

        def Union(x,y):#xとyが同じグループになるよう更新
            if find(x) != find(y):
                Group[find(y)]=min(find(y),find(x))
                Group[find(x)]=min(find(y),find(x))
                Group[x]=min(find(y),find(x))
                Group[y]=min(find(y),find(x))
        
   
        N=len(graph)

        Group=[i for i in range(N)]#グループ分け.Group[i]=jのときiとjは同じグループ


        for i in range(N):
            for j in range(N):
                if graph[i][j]==1:
                    Union(i,j)

        #print(Group)

        Unionlist=[0 for i in range(N)]

        for i in range(N):
            Unionlist[find(i)]+=1

        #print(Unionlist)

        initl=[initial[i] for i in range(len(initial))]
        init=set(initl)

        for i in range(len(initl)):
            for j in range(i+1,len(initl)):
                if find(initl[i])==find(initl[j]):
                    init=init-{initl[i],initl[j]}

        #print(init)

        if init==set():
            return min(initial)

        ANS=0
        ANSI=None
        for i in init:
            if Unionlist[find(i)]>ANS:
                ANS=Unionlist[find(i)]
                ANSI=i

        return ANSI
