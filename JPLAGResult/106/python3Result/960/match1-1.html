class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        node_num = len(graph)
        min_mal = float('inf')
        connect = {item: set() for item in range(node_num)}
        remove = None
        
for i in range(node_num):
            for j in range(i):
                if graph[i][j] == 1:
                    connect[i].add(j)
                    connect[j].add(i)
        # print(connect)
        initial.sort(reverse=True)
        for i in initial:
            ini = [item for item in initial if item != i]
            visited = set(ini)
            index = 0
            while index < len(ini):
                for j in connect[ini[index]]:
                    if j not in visited:
                        visited.add(j)
                        ini.append(j)
                index += 1
            if len(visited) <= min_mal:
                min_mal = len(visited)
                remove = i
        return remove
