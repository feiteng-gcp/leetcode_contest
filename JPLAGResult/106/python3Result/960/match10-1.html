class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        final_effected_nodes = set()
        every_nodes = []
        for i in initial:
            this_can_effect = {i,}
            stack = [i,]
            
while stack:
                t = stack.pop()
                for n, j in enumerate(graph[t]):
                    if j == 1 and n not in this_can_effect:
                        this_can_effect.add(n)
                        stack.append(n)
            final_effected_nodes.update(this_can_effect)
            every_nodes.append([i, this_can_effect])
        for record in every_nodes:
            record.append(len(final_effected_nodes.difference(record[1])))
        min_node = 0
        for i in range(len(every_nodes)):
            if every_nodes[i][2] == every_nodes[min_node][2] and every_nodes[i][0] < every_nodes[min_node][0]:
                min_node = i
            elif every_nodes[i][2] < every_nodes[min_node][2]:
                min_node = i
        return every_nodes[min_node][0]
