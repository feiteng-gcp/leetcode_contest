class Solution:
    def minMalwareSpread(self, graph, initial):
        """
        :type graph: List[List[int]]
        :type initial: List[int]
        :rtype: int
        """
        if len(initial) == 1:
            return initial[0]

        initial.sort()

        l = len(graph)
        def calc(graph,initial):
            t = [0]*len(graph)
            for i in initial:
                t[i] = 1

            while initial:
                now = initial.pop()
                for i in range(l):
                    if i== now :
                        continue
                    elif t[i] == 0 and graph[now][i] == 1:
                        if i not in initial:
                            t[i] = 1
                            initial.append(i)

            # for _ in range(l):
            #     for i in range(l):
            #         s = 0
            #         for j in range(l):
            #             s += graph[j][i]*t[j]
            #             if s:
            #                 break
            #         if s:
            #             t[i] = 1
            return sum(t)
        # print(calc(graph,initial))

        mi = 0
        m = 301
        for i in range(len(initial)):
            g = calc(graph,initial[:i]+initial[i+1:])
            if g<m:
                mi = i
                m = g

        return initial[mi]
