class Solution {
    public int[] sortArrayByParityII(int[] A) {
        int n = A.length, i = 0, j = 1;
        int[] result = new int[n];
        for (int a : A) {
            if (a % 2 == 0) {
                result[i] = a;
                i += 2;
            } else {
                result[j] = a;
                j += 2;
            }
        }
        return result;
    }
}
