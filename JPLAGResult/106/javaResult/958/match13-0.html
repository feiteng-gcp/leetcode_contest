class Solution {
    public int[] sortArrayByParityII(int[] A) {
        int n = A.length;
        int[] res = new int[n];
        int i = 0;
        int j = 1;
        for (int k = 0; k < n; ++k) {
            if (A[k] % 2 == 0) {
                res[i] = A[k];
                i += 2;
            } else {
                res[j] = A[k];
                j += 2;
            }
        }
        return res;
    }
}
