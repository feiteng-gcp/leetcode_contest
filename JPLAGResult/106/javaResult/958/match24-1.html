class Solution {
    public int[] sortArrayByParityII(int[] A) {
        int oddId = 1,evenId = 0;
        int[] ret = new int[A.length];
        for(int i : A){
            if(i % 2 == 0){
                ret[evenId] = i;
                evenId += 2;
            }else{
                ret[oddId] = i;
                oddId += 2;
            }
        }
        return ret;
    }
}
