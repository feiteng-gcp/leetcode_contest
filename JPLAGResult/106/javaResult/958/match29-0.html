class Solution {
    public int[] sortArrayByParityII(int[] A) {
        int even = 0, odd = 1;
        while (even < A.length - 1 && odd < A.length) {
            while (even < A.length - 1 && A[even] % 2 == 0) even += 2;
            while (odd < A.length && A[odd] % 2 == 1) odd += 2;
            if (even < A.length - 1 && odd < A.length) {
                int t = A[odd];
                A[odd] = A[even];
                A[even] = t;
                odd += 2;
                even += 2;
            }
        }
        return A;
    }
}
