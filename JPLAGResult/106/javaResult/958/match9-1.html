class Solution {
    public int[] sortArrayByParityII(int[] A) {
        int len = A.length;
        int[] res = new int[len];
        int i=0,j=1;
        for(int a:A){
            if(a%2==0) {
                res[i]=a;
                i+=2;
            }
            else{
                res[j]=a;
                j+=2;
            }
        }
        return res;
    }
}
