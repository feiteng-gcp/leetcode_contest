class Solution {
    public int[] sortArrayByParityII(int[] A) {
        int n = A.length; 
        int even = 0; 
        int odd = 1; 
        int[] ret = new int[n];
        for (int i = 0; i < A.length; i++) {
            if (A[i] % 2 == 0) {
                ret[even] = A[i];
                even += 2;
            } else {
                ret[odd] = A[i];
                odd += 2; 
            }
        }
        return ret;
    }
}
