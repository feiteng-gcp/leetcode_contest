class Solution {
    public int[] sortArrayByParityII(int[] A) {
        int oc = 0;
        int ec = 0;
        int[] B = new int[A.length];
        for (int i = 0; i < A.length; i ++){
            if (A[i] % 2 == 0){
                B[ec * 2] = A[i];
                ec ++;
            } else {
                B[oc * 2 + 1] = A[i];
                oc ++;
            }
        }
        return B;
    }
}
