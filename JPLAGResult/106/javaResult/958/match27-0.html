class Solution {
    public int[] sortArrayByParityII(int[] A) {
        int e = 0;
        int o = 1;
        int[] ans = new int[A.length];
        for (int i=0; i<A.length; ++i) {
            if (A[i] % 2 == 0) {
                ans[e] = A[i];
                e += 2;
            } else {
                ans[o] = A[i];
                o += 2;
            }
        }
        return ans;
    }
}
