class Solution {
    public int threeSumMulti(int[] A, int target) {
        long res = 0;
        int mod = 1000000007;
        Arrays.sort(A);
        for(int i = 0; i < A.length - 2; i++){
            int first = A[i];
            int left = i + 1, right = A.length - 1;
            while(left < right){
                if(A[left] == A[right] && A[left] + A[right] + first == target){
                    res += (right - left) * (right - left + 1) / 2;
                    res %= mod;
                    break;
                }
                int sum = A[left] + A[right] + first;
                if(sum < target){
                    left++;
                }
                else if(sum > target){
                    right--;
                }
                else{
                    left++;
                    right--;
                    int l = 1, r = 1;
                    while(A[left] == A[left - 1]){
                        left++;
                        l++;
                    }
                    while(A[right] == A[right + 1]){
                        right--;
                        r++;
                    }
                    res += l * r;
                    res %= mod;
                    //System.out.println(l + " " + r);
                }
            }
            //System.out.println(first + " " + res);
        }
        return (int)res;
        
    }
}
