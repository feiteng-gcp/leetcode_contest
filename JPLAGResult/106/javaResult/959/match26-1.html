class Solution {
    public int threeSumMulti(int[] A, int target) {
        Arrays.sort(A);
        
int count=0, MOD=1000000007;
        for (int i=0; i<A.length; i++) {
            int j=i+1, k=A.length-1, sum=target-A[i];
            while (j<k) {
                if (A[j]+A[k] == sum) {
                    if (A[j] == A[k]) {
                        count+=((k-j+1)%MOD)*((k-j)%MOD)/2;
                        count=count%MOD;
                        break;
                    }
                    int j1=j;
                    while (j1<k && A[j1] == A[j]) {
                        j1++;
                    }
                    int k1=k;
                    while (k1>j && A[k1] == A[k]) {
                        k1--;
                    }
                    count += (((j1-j)%MOD)*((k-k1)%MOD))%MOD;
                    count = count%MOD;
                    j=j1;
                    k=k1;
                } else if (A[j]+A[k] < sum) j++;
                else k--
;
            }
        }
        return count;
    }
}
