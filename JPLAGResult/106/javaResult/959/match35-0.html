    private static final int MOD = 1000000007;
    public int threeSumMulti(int[] A, int target) {
        int[] count = new int[101];
        for (int a : A) {
            count[a]++;
        }
        long ret = 0;
        for (int i = 0; i <= 100; i++) {
            if (target - i < i) break;
            if (count[i] == 0) continue;
            int m = count[i];
            count[i]--;
            for (int j = i; j <= 100; j++) {
                if (target - i - j < j) break; 
                if (count[j] == 0) continue;
                int n = count[j];
                if (target - i - j > 100 || count[target - i -j] == 0) continue;
                count[j]--;
                long tmp = 1L * m * n * count[target - i - j];
                if (i == j && target - i - j > j) {
                    tmp /= 2;
                } else if (i < j && target - i - j == j) {
                    tmp /= 2;
                } else if (i == j && target - i - j == j) {
                    tmp /= 6;
                }
                ret = (ret + tmp) % MOD;
                count[j]++;
            }
            count[i]++;
        }
        return (int)ret;
    }
