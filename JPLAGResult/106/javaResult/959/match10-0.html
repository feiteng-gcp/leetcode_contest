class Solution {
    public int threeSumMulti(int[] A, int target) {
    	int N = A.length;
        long[][][] dp = new long[N + 1][4][target + 1];
        long MOD = 1000000007;
        dp[0][0][0] = 1;
        for (int i = 0; i < N; ++i) {
        	for (int j = 0; j <= 3; ++j) {
        		for (int k = 0; k <= target; ++k) {
        			if (dp[i][j][k] > 0) {
        				dp[i + 1][j][k] = (dp[i + 1][j][k] + dp[i][j][k]) % MOD;
        				if (k + A[i] <= target && j < 3) {
        					dp[i + 1][j + 1][k + A[i]] = (dp[i + 1][j + 1][k + A[i]] + dp[i][j][k]) % MOD;
        				}
        			}
        		}
        	}
        }
        return (int)dp[N][3][target];
    }
}
