class Solution {
    static long mod = 1000_000_007;
    
    public int threeSumMulti(int[] A, int target) {
        Arrays.sort(A);
        int n = A.length; 
        long cnt = 0;
        for (int i = 0; i < n; ++i) {
            int t = target - A[i];
            if (target < 0) continue;
            int l = i+1, r = n-1;
            while (l < r) {
                if (A[l] + A[r] < t) ++l;
                else if (A[l] + A[r] > t) --r;
                else {
                    if (A[l] == A[r]) {
                        cnt = (cnt + (r-l+1) * (r-l) / 2) % mod;
                        l = r;
                        continue;
                    }
                    
long lCnt = 1, rCnt = 1;
                    while (l < r -1 && A[l+1] == A[l]) {
                        ++l;
                        ++lCnt;
                    }
                    while (l + 1 < r && A[r-1] == A[r]) {
                        --r;
                        ++rCnt;
                    }
                    cnt = (cnt + (lCnt * rCnt) % mod) % mod;
                    ++l;
                
}
            }
        }
        
        return (int)cnt;
    }
}
