class Solution {
    public int threeSumMulti(int[] A, int target) {
        int mod = (int) 1e9 + 7;
        int res = 0;
        Arrays.sort(A);
        for (int i = 0; i < A.length; i++) {
            int l = i + 1, h = A.length - 1;
            int t = target - A[i];
            while (l < h) {
                if (l < h && A[l] + A[h] < t) l++;
                else if (l < h && A[l] + A[h] > t) h--;
                else if (A[l] + A[h] == t) {
                    if (A[l] == A[h]) {
                        int len = (h - l + 1);
                        res = (res + len * (len - 1) / 2) % mod;
                        break;
                    } else {
                        int ll = l, hh = h;
                        while (ll < hh && A[ll+1] == A[ll]) ll++;
                        while (ll < hh && A[hh-1] == A[hh]) hh--;
                        res = (res + (ll - l + 1) * (h - hh + 1)) % mod;
                        l = ll + 1;
                        h = hh - 1;
                    }
                
}
            }
        }
        return res;
    }
}
