class Solution {
    public int threeSumMulti(int[] A, int target) {
        int[][] dp = new int[4][target+1];
        dp[0][0] = 1;
        int mod = 1000000007;
        
        
for (int h = 0; h < A.length; h++) {
            for (int j = 3; j >= 1; j--) {
                for (int i = 0; i <= target; i++) {
                    if (i < A[h]) continue;
                    dp[j][i] = (dp[j][i] + dp[j-1][i - A[h]]) % mod;
                
}
            }
        }
        return dp[3][target];
    }
}
