class Solution {
    final static int MOD = 1000000007;
    public int threeSumMulti(int[] A, int target) {
        int n = A.length;
        long result = 0;
        for (int i = 0; i < n; i++) {
            int needed = target - A[i];
            result = (result + twoSum(A, needed, i + 1)) % MOD;
        }
        return (int)result;
    }
    
    private int twoSum(int[] A, int target, int startPos) {
        int result = 0;
        int n = A.length;
        Map<Integer, Integer> map = new HashMap<>();
        
        for (int i = startPos; i < n; i++) {
            int needed = target - A[i];
            if (map.containsKey(needed)) {
                result += map.get(needed);
            }
            map.put(A[i], map.getOrDefault(A[i], 0) + 1);
            
        }
        return result;
    }
}
