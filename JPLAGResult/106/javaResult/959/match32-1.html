class Solution {
    public int threeSumMulti(int[] A, int target) {
        int mod = 1_000_000_007;
        int N = A.length;
        int[][][] dp = new int[3][N][target + 1];
        
for (int i = 0; i < 3; i++) {
            for (int j = i; j < N; j++) {
                for (int k = 0; k <= target; k++) {
                    if (i == 0) {
                        if (j == 0) {
                            if (A[j] == k) {
                                dp[i][j][k] = 1;
                            } else {
                                dp[i][j][k] = 0;
                            }
                        } else {
                            if (A[j] == k) {
                                dp[i][j][k] = (1 + dp[i][j - 1][k]) % mod;
                            } else {
                                dp[i][j][k] = dp[i][j - 1][k];
                            }
                        }
                    } else {
                        if (k >= A[j]) {
                            dp[i][j][k] = (dp[i - 1][j - 1][k - A[j]] + dp[i][j - 1][k]) % mod;
                        } else {
                            dp[i][j][k] = dp[i][j - 1][k];
                        }
                    }
                
}
            }
        }
        return (dp[2][N - 1][target]) % mod;
    }
}
