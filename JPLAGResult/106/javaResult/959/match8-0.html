class Solution {
    int MOD = 1000000007;

    public int threeSumMulti(int[] A, int target) {
        int[][] dp = new int[4][target + 1];
        for (int i = 0;i < 3;i++) {
            for (int j = 0;j < target + 1;j++) {
                dp[i][j] = 0;
            }
        }
        dp[0][0] = 1;
        for (int i = 0;i < A.length;i++) {
            for (int j = 2;j >= 0;j--) {
                for (int k = 0;k < target + 1;k++) {
                    if (k + A[i] <= target) {
                        dp[j + 1][k + A[i]] += dp[j][k];
                        if (dp[j + 1][k + A[i]] >= MOD) {
                            dp[j + 1][k + A[i]] -= MOD;
                        }
                    }
                }
            }
        }
        return dp[3][target];
    }
}
