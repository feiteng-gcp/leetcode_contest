class Solution {
    int mod = 1_000_000_007;
    public int threeSumMulti(int[] A, int target) {
        //dp[count][index][target] is the number of ways to get to `target` using exactly `count` numbers with the last number at index at most `index`
        int[][][] dp = new int[3][A.length][target+1];
        
        // base case: dp[0][index][num] are the number of `num` between 0 and index
        // dp[0][0][n] = 1 if n = A[0], else 0
        // dp[0][index][n] = dp[0][index-1][n] + (A[index] == n) ? 1 : 0
        if(A[0] <= target) {
            dp[0][0][A[0]] = 1;
        }
        for(int index = 1; index < A.length; index++) {
            for(int n = 0; n <= target; n++) {
                dp[0][index][n] = (dp[0][index-1][n] + ((A[index] == n) ? 1 : 0)) % mod;
            }
        }
        
        // recursive case: dp[depth][index][num] = dp[depth-1][index-1][num] + (num >= A[index]) ? dp[depth-1][index-1][num - A[index]] : 0
        
for(int depth = 1; depth < 3; depth++) {
            for(int index = depth; index < A.length; index++) {
                for(int num = 0; num <= target; num++) {
                    dp[depth][index][num] = (dp[depth][index-1][num] + ((num >= A[index]) ? dp[depth-1][index-1][num - A[index]] : 0)) % mod;
                
}
            }
        }
        
        return dp[2][A.length-1][target];
    }
}
