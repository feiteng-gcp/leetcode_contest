class Solution {
    
public int threeSumMulti(int[] A, int target) {
        long[][] cnts = new long[A.length + 1][101];
        for(int i = A.length - 1;i>=0;i--){
            for(int j = 0;j<101;j++){
                cnts[i][j] = cnts[i + 1][j];
            }
            cnts[i][A[i]]++;
        }
//
//        for (int i = A.length; i >= 0 ; i--) {
//            System.out.println("i = " + i + " cnts = " + Arrays.toString(cnts[i]));
//        }

        long cnt = 0;
        for (int i = 0; i < A.length; i++) {
            for (int j = i + 1; j < A.length; j++) {
                int sum = A[i] + A[j];
                if(target >= sum && (target - sum) <= 100) cnt += cnts[j + 1][target - sum]
;
                cnt = cnt % 1000000007;
                //System.out.println("i = " + i  + " j = " + j + " sum = " + sum + " cnt = " +  cnt);
            }
        }
        return (int) cnt;
    }
}
