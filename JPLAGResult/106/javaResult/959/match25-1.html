    private int twoSumMulti(int[] A, int start, int target) {
        Map<Integer, Integer> set = new HashMap<>();
        int count = 0;
        for (int i = start; i < A.length; i++) {
            if (set.containsKey(A[i])) {
                // count %= 10e9 + 7;
                count += set.get(A[i]);
            }
            int remain = target - A[i];
            if (set.containsKey(remain)) {
                set.put(remain, set.get(remain) + 1);
            } else {
                set.put(remain, 1);
            }
        }
        // count %= 10e9 + 7;
        return count;
    }
