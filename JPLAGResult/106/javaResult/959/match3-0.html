class Solution {
    public int threeSumMulti(int[] A, int target) {
        long res = 0;
        int mod = 1000000007;
        Arrays.sort(A); 
        for (int i = 0; i < A.length - 2; ++i) {
                int left = i + 1; 
                int right = A.length - 1;
                int t = target - A[i];
                while (left < right) { // run two sum
                    if (A[left] + A[right] == t) {
                        int l = 1;
                        int r = 1;
                        if (A[left] == A[right]) {
                            long tmp = (right - left + 1) * (right - left);
                            res += tmp / 2;
                            break;
                        } else {
                        while (left < right && A[left] == A[left + 1]) {
                            l++;
                            ++left;
                        }
                        while (left < right && A[right] == A[right - 1]) {
                            r++;
                            --right;
                        }
                        res += l * r;
                        
                        ++left; 
                        --right;
                        }
                    } else if (A[left] + A[right] < t) {
                        ++left;
                    } else {
                        --right;
                    }
                }
            System.out.println("i =" + i + "res = " + res);
            res %= mod;
        }
        return (int)(res % mod);
    }
    public static long factorial(int number) {
        long result = 1;

        for (int factor = 2; factor <= number; factor++) {
            result *= factor;
        }

        return result;
    }
}
