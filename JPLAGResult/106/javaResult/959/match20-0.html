class Solution {
    public int threeSumMulti(int[] A, int target) {
        int[] pre = new int[101];
        int[] suf = new int[101];
        for (int n : A) ++suf[n];
        --suf[A[0]];
        
        int res = 0;
        for (int i = 0; i < A.length - 1; ++i) {
            int y = A[i];
            for (int x = 0; x <= 100; ++x) {
                int z = target - x - y;
                if (pre[x] > 0 && z >= 0 && z <= 100 && suf[z] > 0) {
                    res = (res + pre[x] * suf[z]) % 1000000007;
                }
            }
            ++pre[y];
            --suf[A[i + 1]];
        }
        return res;
    }
}
