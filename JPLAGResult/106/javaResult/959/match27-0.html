class Solution {
    public int threeSumMulti(int[] A, int target) {
        int len = A.length;
        int[][] numrec = new int[101][len];
        for (int[] t : numrec) Arrays.fill(t, 0);
        numrec[A[len - 1]][len - 1] = 1;
        for (int i = len - 2; i >= 0; i--) {
            numrec[A[i]][i] = numrec[A[i]][i + 1] + 1;
            for (int j = 0; j <= 100; j++) {
                if (j != A[i]) numrec[j][i] = numrec[j][i + 1];
            }
        }
        long res = 0l;
        for (int i = 0; i < len - 2; i++) {
            int tar = target - A[i];
            for (int j = i + 1; j < len - 1; j++) {
                // need to find how many tar - A[j] after pos j
                if (tar - A[j] < 0 || tar - A[j] > 100) continue;
                res += numrec[tar - A[j]][j + 1];
                res %= 1000000007;
            }
        }
        return (int)res;
    }
}
