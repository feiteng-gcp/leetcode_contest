class Solution {
    public int threeSumMulti(int[] A, int target) {
        int MOD = 1_000_000_007;
        long res = 0;
        Arrays.sort(A);
        Map<Integer, Integer> count = new HashMap<>();
        for (int num : A) {
            count.put(num, count.getOrDefault(num, 0)+1);
        }

        for (int i = 0; i < A.length-2; i++) {
            int tmpTar = target - A[i];
            int j = i+1;
            int k = A.length - 1;
            while (j < k) {
                if (A[j] + A[k] > tmpTar) {
                    while (j < k && A[k]==A[k-1])k--;
                    k--;
                } else if (A[j] + A[k] < tmpTar) {
                    while (j < k && A[j]==A[j+1])j++;
                    j++;
                } else {
                    if (A[i] == A[k]) {
                        long c = count.get(A[i]);
                        res += (c * (c - 1) * (c - 2)) / 6;
                    } else if (A[i] == A[j]) {
                        long c = count.get(A[i]);
                        res += (c * (c - 1)) / 2 * count.get(A[k]);
                    } else if (A[j] == A[k]) {
                        long c = count.get(A[k]);
                        res += (c * (c - 1)) / 2 * count.get(A[i]);
                    } else {
                        res += count.get(A[i]) * count.get(A[j]) * count.get(A[k]);
                    }

                    while (j < k && A[k]==A[k-1])k--;
                    while (j < k && A[j]==A[j+1])j++;
                    k--;
                    j++;
                }
            }
            while (i < A.length-2 && A[i] == A[i+1])i++;
        }
        return (int) (res % MOD);
    }
}
