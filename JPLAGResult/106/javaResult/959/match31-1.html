class Solution {
    public int threeSumMulti(int[] A, int target) {
        
        Map<Integer, Integer> map = new HashMap<>();
        int n = A.length;
        for (int i = 0; i < n; i++) {
            if (!map.containsKey(A[i])) {
                map.put(A[i], 0);
            }
            map.put(A[i], map.get(A[i]) + 1);
        }
        int mod = 1000000000 + 7; 
        Arrays.sort(A);
        Set<String> set = new HashSet<>(); 
        long ret = 0;
   //     System.out.println(map.get(0));
        for (int i = 0; i < n; i++) {
            int j = i + 1;
            int k = n - 1; 
            while (j < k) {
                if (A[i] + A[j] + A[k] < target) {
                    j++; 
                    while (j < k && A[j - 1] == A[j]) {
                      j++;   
                    }
                } else if (A[i] + A[j] + A[k] > target) {
                    
                    k--; 
                    while (j < k && A[k] == A[k + 1]) {
                        k--;
                    }
                } else {
                    String str = A[i] + "/" + A[j] + "/" + A[k];
              //      System.out.println(str + " " + (A[i] + A[j] + A[k]));
                    if (!set.contains(str)) {
                        long fi = map.get(A[i]);
                        long fj = map.get(A[j]);
                        long fk = map.get(A[k]);
                        set.add(str);
                        if (A[i] == A[j] && A[j] == A[k]) {
                           
                            ret += fi * (fi - 1) * (fi - 2)  / 6 ; 
                            ret %= mod;
                        } else if (A[i] == A[j]) {
                            ret += fk % mod * fi * (fi - 1) % mod / 2;
                            ret %= mod;
                        } else 
if (A[i] == A[k]) {
                            ret += fj % mod * fi % mod * (fi - 1) % mod / 2;
                            ret %= mod;
                        } else if (A[j] == A[k]) {
                            ret += fi % mod * fj % mod * (fj - 1) % mod / 2;
                            ret %= mod;                            
                        } else {
                            ret += fi % mod * fj % mod * fk % mod;
                            ret %= mod; 
                        }
                    }
                    j++; 
                    
while (j < k && A[j - 1] == A[j]) {
                      j++;   
                    }
                    k--;
                    while (j < k && A[k] == A[k + 1]) {
                        k--;
                    }
                }
            }
        }
        return (int)ret; 
    }
}
