class Solution {
   public int threeSumMulti(int[] A, int target) {
		 int mod = 1_000_000_00_7;
		 int dp[][] = new int[A.length][101];
		 int[] prev = dp[A.length-1];
		 prev[A[A.length-1]]++;
		 for(int i = A.length -2; i>=0;i--) {
			 int curr = A[i];
			 prev = prev.clone();
			 prev[curr]++;
			 dp[i]= prev;
		 }
		 long count = 0
;
		 for(int i =0; i<A.length-2; i++) {
			 for(int j =i+1; j<A.length-1; j++) {
				 int rem = target - A[i]-A[j];
				 
				if(rem <0 || rem>100) {
					 continue;
				 }

				 
				 count+=dp[j+1][rem];
				 count%=mod;
			 }
		 }
		 
		 
	     return (int)count;
	    }
}
