class Solution {
    public int threeSumMulti(int[] A, int target) {
        final int MOD = 1000000007;
        int res = 0;
        Arrays.sort(A);
        for (int i = 0; i < A.length; i++) {
            int sum = target - A[i];
            int l = i + 1, r = A.length - 1;
            while (l < r) {
                if (A[l] + A[r] > sum) {
                    r--;
                } else if (A[l] + A[r] < sum) {
                    l++;
                } else if (A[l] != A[r]) {
                    int cnt1 = 1, cnt2 = 1;
                    while (l < r && A[l] == A[l + 1]) {
                        l++;
                        cnt1++;
                    }
                    while (r > l && A[r] == A[r - 1]) {
                        r--;
                        cnt2++;
                    }
                    l++;
                    r--;
                    res += cnt1 * cnt2;
                    res = res % MOD;
                } else {
                    int cnt = r - l + 1;
                    res += cnt * (cnt - 1) / 2;
                    res %= MOD;
                    break;
                }
            }
        }
        return res % MOD;
    }
}
