class Solution {
    public int threeSumMulti(int[] A, int target) {
        TreeMap<Integer, Integer> map = new TreeMap();
        int mod = 1000000007;
        for (int i: A) {
            map.put(i, map.getOrDefault(i, 0)+1);
        }
        List<Integer> keys = new ArrayList(map.keySet());
        long res = 0, l = keys.size();
        for (int i = 0; i < l; i++) {
            int val_i = keys.get(i);
            long cnt_i = map.get(val_i);
            for (int j = i; j < l; j++) {
                int val_j = keys.get(j), val_k = target - val_i - val_j;
                long cnt_j = map.get(val_j);
                if (val_k < val_j) break;
                if (map.containsKey(val_k)) {
                    long cnt_k = map.get(val_k);
                    if (val_i == val_k) {
                        res += cnt_i * (cnt_i-1) * (cnt_i-2) / 6;
                    } else if (val_i == val_j) {
                        res += cnt_i * (cnt_i-1) * cnt_k / 2;
                    } else if (val_j == val_k) {
                        res += cnt_i * cnt_j * (cnt_j-1) / 2;
                    } else {
                        res += cnt_i * cnt_j * cnt_k;
                    }
                }
                res %= mod;
            }
        }
        return (int)res;
    }
}
