class Solution {
    public int threeSumMulti(int[] A, int target) {
        int ans = 0;
        int mod = 1000000007;
        Arrays.sort(A);
        for (int i = 0; i < A.length - 2; i++) {
            int twoSum = target - A[i];
            int j = i + 1, k = A.length - 1;
            while (j < k) {
                if (A[j] + A[k] < twoSum) {
                    j++;
                } else if (A[j] + A[k] > twoSum) {
                    k--;
                } else {
                    int jj = j;
                    int kk = k;
                    if (A[jj] == A[kk]) {
                        ans = (ans + comb(kk - jj + 1)) % mod;
                        break;
                    } else {
                        while (jj + 1 < kk && A[jj + 1] == A[jj]) jj++;
                        while (kk - 1 > jj && A[kk - 1] == A[kk]) kk--;
                        ans = (ans + (jj - j + 1) * (k - kk + 1)) % mod;
                        j = jj + 1;
                        k = kk - 1;
                    }
                }
            }
        }
        return ans;
    }

    int comb(int n) {
        return n * (n - 1) / 2;
    }
}
