class Solution {
    private long a[];
    private final int mod = 1000000007;
    public int threeSumMulti(int[] A, int target) {
        a = new long[101];
        long ans = 0;
        for(int i = 0; i < A.length; i++){
            a[A[i]]++;
        }
        if(target % 3 == 0){
            if(a[ target/3] >= 3){
                ans = ans + zuhe(a[ target/3], 3);
                ans = ans % mod;
            }
        }
        for(int i = 0; i <= 100; i++){
            if(a[i] == 0) continue;
            for(int j = i; j <= 100; j++){
                if(a[j] == 0) continue;
                
                int z = target - i - j;
                if(z < 0 || z > 100) continue;
                if(z < i || z < j) continue;
                if(z == i && z == j) continue;
                if(i == j){
                    if(a[i] < 2) continue;
                    ans =ans +  (zuhe(a[i], 2) *a[z]) %mod;
                    ans = ans % mod;
                }else if(j == z){
                    if(a[j] < 2) continue;
                    ans = ans + (zuhe(a[j], 2) *a[i]) %mod;
                                        ans = ans % mod;
                }else {
                    ans = ans + a[i]*a[j]*a[z];
                    ans = ans % mod;
                }
                
            }
        }
        a = new long[0];
        return (int)(ans);
        
    }
    public long zuhe(long n, int m){
      long [] c = new long[(int)(n+1)];
        for(int i = 2; i <= m; i++) c[i]--;
        for(long i = n; i >= n-m+1; i--){
            long z = i;
            for(long j = 2; j <= z; j++){
                if(z % j == 0){
                    c[(int)j] = c[(int)j]+1;
                    z /= j;
                    j--;
                }
            }
        }
        long ans = 1;
        for(long i = 2; i <= n; i++){
            for(long j = 1; j <= c[(int)i]; j++){
                ans =( ans * i ) % mod;
            }
        }
        return ans;
    }
}
