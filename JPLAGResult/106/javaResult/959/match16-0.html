class Solution {
    
    public long choose(int n, int k, int mod) {
        long ans = 1;
        for (int i =1;i<=k;i++,n--) {
            ans = ans * n / i;
            
        }
        return ans%mod;
    }
    
    public int threeSumMulti(int[] A, int target) {
        int [] tally = new int[101];
        for (int x: A) {
            tally[x]++;
        }
        
        long ans = 0;
        int mod = 1000_000_000+7;
        for (int i = 0; i<101; i++) {
            for (int j =i; j<101; j++) {
                int k = target - i - j;
                if (k<j) {break;}
                if (k>100) {continue;}
                if (i==j && j==k) {
                    ans += choose(tally[i],3,mod);
                }else if (i==j) {
                    ans += (choose(tally[i],2,mod)*tally[k])%mod;
                }else if (j==k) {
                    ans += (choose(tally[j],2,mod)*tally[i])%mod;
                }else {
                    ans += (((tally[i]*tally[j])%mod)*tally[k])%mod;
                }
                ans %= mod;
            }
        }
        
         //return (int)choose(2,2,mod);
        return (int)ans;
        //return A.length;
        //return (int)ans;
    }
}
