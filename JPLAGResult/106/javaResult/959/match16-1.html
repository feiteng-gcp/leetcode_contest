class Solution {
    
public int threeSumMulti(int[] A, int target) {
        int[][] dp = new int[A.length][201];
        for (int j = 1; j < A.length; j++) {
            for (int i = 0; i < j; i++) {
                dp[j][A[i] + A[j]]++;
            }
        
}
        
        int res = 0;
        int modulo = 1000000007;
        for (int k = 2; k < A.length; k++) {
            for (int j = 1; j < k; j++) {
                if (target - A[k] >= 0 && target - A[k] <= 200) {
                    res += dp[j][target - A[k]];
                    res %= modulo;
                
}
            }
        }
        
        return res;
    }
}
