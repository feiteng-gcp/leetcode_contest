class Solution {
    public int threeSumMulti(int[] A, int target) {
        Arrays.sort(A);
        int mod = 1000000007;
        int sum = 0;
        for (int i = 0; i < A.length; i++) {
            int res = target - A[i];
            Map<Integer, Integer> map = new HashMap<>();
            for (int j = i + 1; j < A.length; j++) {
                if (map.containsKey(res - A[j])) {
                    sum += map.get(res - A[j]);
                    sum %= mod;
                }
                
                map.put(A[j], map.getOrDefault(A[j], 0) + 1);
            }
        }
        return sum;
    }
}
