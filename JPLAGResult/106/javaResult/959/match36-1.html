class Solution {
    private static final int MOD = 1000000007;
    
    
public int threeSumMulti(int[] A, int target) {
        int[] counts = new int[101];
        for (int i : A) {
            counts[i]++;
        }
        long ret = 0;
        for (int i = 0; i < counts.length; i++) {
            if (counts[i] == 0) {
                continue;
            }
            if (counts[i] >= 3 && i * 3 == target) {
                long curr = 1L * counts[i] * (counts[i] - 1) * (counts[i] - 2) / 6;
                ret = (ret + (curr % MOD)) % MOD;
            }
            if (i * 3 > target) {
                break;
            }
            int target2 = target - i * 2;
            if (target2 != i && target2 >= 0 && target2 < counts.length && counts[target2] > 0) {
                long curr = 1L * counts[i] * (counts[i] - 1) / 2 * counts[target2];
                ret = (ret + (curr % MOD)) % MOD;
            }
            for (int j = i + 1; j < counts.length && (target - i - j) >= j; j++) {
                int target3 = target - i - j;
                if (target3 == j) {
                    long curr = 1L * counts[i] * counts[j] * (counts[j] - 1) / 2;
                    ret = (ret + (curr % MOD)) % MOD;
                } else if (target3 < counts.length && counts[target3] > 0) {
                    long curr = counts[i] * counts[j] * counts[target3];
                    ret = (ret + (curr % MOD)) % MOD;
                
}
            }
        }
        return (int)ret;
    }
}
