class Solution {
  
    public int minAddToMakeValid(String S) {
        int l = 0;
        int ans = 0;

        for (int i = 0; i < S.length(); i++) {
            char ch = S.charAt(i);
            if (ch == '(') {
                l++;
            } else {
                if (l == 0) {
                    ans++;
                } else {
                    l--;
                }
            }
        }
        return ans + l;

    }

}
