class Solution {
  public int minAddToMakeValid(String S) {
    int l = 0;
    int cnt = 0;
    for (char c : S.toCharArray()) {
      if (c == '(') {
        l++;
      } else {
        if (l == 0) {
          cnt++;
        } else {
          l--;
        }
      }
    }
    return cnt + l;
  }
}
