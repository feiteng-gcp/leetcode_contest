class Solution {
    public int minAddToMakeValid(String S) {
        int toBeMatch = 0, ret = 0;
        for (int i = 0; i < S.length(); i++){
            if (S.charAt(i) == '(')
{
                toBeMatch++;
            } else {
                if (toBeMatch > 0){
                    toBeMatch--;
                } else {
                    ret++;
                }
            }
        }
        ret += toBeMatch;
        return ret;
    }
}
