class Solution {
    public int minAddToMakeValid(String S) {
        int left = 0, count = 0;
        for (char ch : S.toCharArray()) {
            if (ch == '(') {
                left++;
            } else {
                if (left == 0)
                    count++;
                else 
                    left--;
            }
        }
        count += left;
        return count;
    }
}
