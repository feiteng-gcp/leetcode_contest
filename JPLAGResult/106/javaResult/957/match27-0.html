class Solution {
    public int minAddToMakeValid(String S) {
        int res = 0;
        int left = 0;
        
        for(char c : S.toCharArray()){
            if(c == '('){
                left++;
            }
            else{
                if(left > 0){
                    left--;
                }
                else{
                    res++;
                }
            }
        }
        return res + left;
    }
}
