class Solution {
    public int minAddToMakeValid(String S) {
        int left = 0, right = 0;
        for(int i = 0; i < S.length(); i++) {
            if(S.charAt(i) == '('){
                left++;
            }
            else {
                if(left == 0){
                    right++;
                }
                else{
                    left--;// find a match for left
                }
            }
        }
        return left + right;
    }
}
