class Solution {
    public int minAddToMakeValid(String S) {
        int res = 0;
        int sum = 0;
        for (int i = 0; i < S.length(); i++) {
            char cur = S.charAt(i);
            if (cur == '(') {
                sum++;
            }
            else {
                if (sum == 0) {
                    res++;
                }
                else {
                    sum--;
                }
            }
        }

        return res + sum;
    }
}
