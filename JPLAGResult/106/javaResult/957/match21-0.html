class Solution {
    public int minAddToMakeValid(String S) {
        int num_of_left = 0;
        int num_of_add = 0;
        for (char c : S.toCharArray()) {
            if (c == '(') {
                num_of_left ++;
            }
            else {
                if (num_of_left > 0) {
                    num_of_left --;
                }
                else {
                    num_of_add++;
                }
            }
        }
        return num_of_add+num_of_left;
    }
}
