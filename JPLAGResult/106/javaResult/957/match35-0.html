class Solution {
    public int minAddToMakeValid(String S) {
        int stack =0;
        int res=0;
        for(char c : S.toCharArray()){
            if(c=='('){
                stack+=1;
            }else{
                if(stack==0){
                    res+=1;
                }else{
                    stack-=1;
                }
            }
        }
        return res +stack;
    }
}
