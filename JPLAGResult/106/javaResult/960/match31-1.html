class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int n = graph.length;
        int m = initial.length;
        int[][] infect = new int[m][n];
        
        for (int i = 0; i < m; i++) {
            getInfect(graph, initial, infect, i);
        }
        
        int[] nums = new int[m];
        int[] sum = new int[n];
        int[] index = new int[n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                sum[i] += infect[j][i];
                if (infect[j][i] != 0) {
                    index[i] = j;
                }
            }
        }
        
        for (int i = 0; i < n; i++) {
            if (sum[i] == 1) {
                nums[index[i]]++;
            }
        }
        
        int max = Integer.MIN_VALUE;
        int maxIndex = -1;
        for (int i = 0; i < m; i++) {
            
if (nums[i] > max || (nums[i] == max && initial[i] < initial[maxIndex])) {
                maxIndex = i;
                max = nums[i];
            }
        }
        return initial[maxIndex];
    }
    
    private void getInfect(int[][] graph, int[] initial, int[][] infect, int i) {
        Set<Integer> set= new HashSet<>();
        set.add(initial[i]);
        
        while (!set.isEmpty()) {
            int node = set.iterator().next();
            set.remove(node);
            infect[i][node] = 1;
            
for (int j = 0; j < graph.length; j++) {
                if (graph[node][j] == 1 && infect[i][j] == 0) {
                    set.add(j);
                }
            }
        }
    }
}
