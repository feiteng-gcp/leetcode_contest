class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        Arrays.sort(initial);

        Set<Integer> init = new HashSet<>();
        
for (int i : initial) {
            init.add(i);
        }
        int min = graph.length + 1, minIndex = 0;

        for (int i : initial) {
            // if remove i
            Set<Integer> copy = new HashSet<>(init);
            copy.remove(i);
            
int affected = affected(graph, copy);

            if (affected < min) {
                minIndex = i;
                min = affected;
            }
        }
        return minIndex;
    }

    private int affected(int[][] graph, Set<Integer> affected) {
        int size = affected.size(), nsize;
        while (true) {
            Set<Integer> newly = new HashSet<>();
            
for (int a : affected) {
                for (int i = 0; i < graph.length; i++) {
                    if (graph[i][a] == 1) {
                        newly.add(i);
                    }
                }
            }
            affected.addAll(newly);
            if (size == affected.size()) {
                break;
            } else {
                size = affected.size();
            }
        }
        return size;
    }

}
