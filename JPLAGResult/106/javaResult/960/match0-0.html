class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        Arrays.sort(initial);
        int res = -1, mininf = 301;
        for (int i : initial) {
            int t = bfs(graph, initial, i);
            if (t < mininf) {
                mininf = t;
                res = i;
            }
        }
        return res;
    }
    
    final int bfs(int[][] graph, int[] initial, int miss) {
        int len = initial.length, glen = graph.length;
        boolean[] infected = new boolean[glen];
        Queue<Integer> q = new LinkedList<>();
        Arrays.fill(infected, false);
        for (int i : initial) {
            if (i != miss) {
                infected[i] = true;
                q.add(i);
            }
        }
        int res = len - 1;
        while (!q.isEmpty()) {
            int s = q.size();
            for (int i = 0; i < s; i++) {
                int t = q.poll();
                int[] l = graph[t];
                for (int j = 0; j < glen; j++) {
                    if (l[j] == 1 && !infected[j]) {
                        infected[j] = true;
                        q.add(j);
                        res++;
                    }
                }
            }
        }
        return res;
    }
}
