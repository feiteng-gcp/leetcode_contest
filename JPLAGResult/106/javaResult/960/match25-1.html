class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int res = -1;
        int min = Integer.MAX_VALUE;
        Arrays.sort(initial);
        Set<Integer> set = new HashSet<>();
        for (int i : initial) set.add(i);
        for (int i : initial) {
            set.remove(i);
            int num = bfs(graph, set);
            
set.add(i);
            if (num < min) {
                res = i;
                min = num;
            }
        }
        return res;
    }
    
    private int bfs(int[][] graph, Set<Integer> set) {
        Queue<Integer> queue = new LinkedList<>();
        Set<Integer> visited = new HashSet<>();
        for (int i : set) {
            queue.add(i);
            visited.add(i);
        }
        
while (!queue.isEmpty()) {
            int size = queue.size();
            for (int s = 0; s < size; s++) {
                int cur = queue.poll();
                for (int i = 0; i < graph.length; i++) {
                    if (graph[cur][i] == 1 && !visited.contains(i)) {
                        visited.add(i);
                        
queue.add(i);
                    }
                }
            }
        }
        return visited.size();
    }
}
