class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        // now we have a graph that rooted at inital
        // return a node that has max sub nodes
        if(initial.length == 1) return initial[0];
        
        // DFS find a node that visited time is max => find all path to all nodes
        
        // compare who is the best malware
        int[] visited = new int[graph.length];
        int bestNode = 0;
        int bestInfected = -1;
        Arrays.sort(initial);
        for(int n: initial){
            Arrays.fill(visited, 0);
            
int num = dfs(graph, n, visited);
            if(num > bestInfected) {
                bestNode = n;
                bestInfected = num;
            }
        }
        
        return bestNode;
    }
    
    int dfs(int[][] graph, int node, int[] visited){
        int n = graph.length;
        if(node < 0 || node >= n) return 0;
        
        // check visited
        if(visited[node]++ > 0) return 0;
        
        // go deeper
        int total = 1;
        
for(int i = 0; i < n; i++){
            if(i != node && graph[node][i] == 1){
                total += dfs(graph, i, visited);
            }
        }
        return total;
    }
}
