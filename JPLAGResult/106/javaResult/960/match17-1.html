class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int n = graph.length;
        int[] arr = new int[n];
        Arrays.fill(arr, -1);
        boolean[] visit = new boolean[n];
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (graph[i][j] == 1) {
                    union(arr, i, j);
                }
            }
        
}
        int[] counts = new int[n];
        for (int i = 0; i < n; i++) {
            counts[find(arr, i)]++;
        }
        //System.out.println(Arrays.toString(counts));
        int ret = n, size = -1;
        for (int i : initial) {
            int r = find(arr, i);
            if (counts[r] > size || (counts[r] == size && i < ret)) {
                size = counts[r];
                ret = i;
            
}
        }
        return ret;
    }
    
    int find(int[] arr, int x) {
        if (arr[x] == -1 || arr[x] == x) {
            return x;
        }
        arr[x] = find(arr, arr[x]);
        return arr[x];
    
}
    
    void union(int[] arr, int x, int y) {
        int xr = find(arr, x), yr = find(arr, y);
        if (xr != yr) {
            arr[xr] = yr;
        }
    }
}
