class Solution {
    
    boolean[] flag = new boolean[305];
    
    
int bfs(int idx, int[] init, int[][] G) {
        Queue<Integer> q = new LinkedList<>();
        Arrays.fill(flag, false);
        int cnt = 0;
        for(int i = 0; i < init.length; ++i) {
            if(i != idx) {
                flag[init[i]] = true;
                q.add(init[i]);
                ++cnt;
            }
        }
        
while(!q.isEmpty()) {
            int u = q.poll();
            for(int i = 0; i < G[u].length; ++i) {
                if(G[u][i] == 1 && !flag[i]) {
                    flag[i] = true;
                    ++cnt;
                    
q.add(i);
                }
            }
        }
        return cnt;
    }
    
    public int minMalwareSpread(int[][] graph, int[] initial) {
        Arrays.sort(initial);
        int ans = -1, cnt = Integer.MAX_VALUE;
        for(int i = 0; i < initial.length; ++i) 
{
            int tmp = bfs(i, initial, graph);
            if(tmp < cnt) {
                cnt = tmp;
                ans = initial[i];
            }
        }
        return ans;
    }
}
