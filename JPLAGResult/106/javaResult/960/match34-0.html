class Solution {
    int[] sz;
    int[] parent;
    int n;
    int findParent(int i) {
        while (i != parent[i]) {
            i = parent[i];
        }
        return i;
    }

    void union(int i, int j) {
        int p1 = findParent(i);
        int p2 = findParent(j);
        if (p1 == p2) {
            return;
        }

        if (sz[p1] < sz[p2]) {
            sz[p2] += sz[p1];
            parent[p1] = p2;
        }
        else {
            sz[p1] += sz[p2];
            parent[p2] = p1;
        }
    }

    public int minMalwareSpread(int[][] graph, int[] initial) {
        n = graph.length;
        sz = new int[n];
        parent = new int[n];
        for (int i = 0; i < n; i++) {
            sz[i] = 1;
            parent[i] = i;
        }

        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (graph[i][j] == 1) {
                    union(i, j);
                }
            }
        }

        Arrays.sort(initial);
        Set<Integer> starts = new HashSet<>();
        for (int v : initial) {
            starts.add(v);
        }

        int res = -1;
        int saved = -1;
        for (int idx : initial) {
            int par = findParent(idx);
            int another = -1;
            for (int i = 0; i < n; i++) {
                if (i != idx && starts.contains(i)) {
                    if (findParent(i) == par) {
                        another = i;
                        break;
                    }
                }
            }

            if (another == -1) {
                if (saved < sz[par]) {
                    saved = sz[par];
                    res = idx;
                }
            }
            else {
                if (saved < 0) {
                    saved = 0;
                    res = idx;
                }
            }
        }

        return res;
    }
}
