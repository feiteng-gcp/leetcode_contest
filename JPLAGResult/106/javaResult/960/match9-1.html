class Solution {
    int [] parent;
    int [] count;
    void unionfind(int n ) {
        
parent = new int[n];
         count = new int[n];
        for (int i = 0; i < n; ++i) {
            parent[i] =i;
            count[i] = 1;
        }
    }
    
    int find (int p) {
        if (p!= parent[p]) {
            parent[p] = find(parent[p]);
        }
        
return parent[p];
      
    }
    void union(int p, int q) {
        int rootp = find(p);
        int rootq = find(q);
        if (rootp == rootq) {
            return;
        } else 
{
            parent[rootp] = rootq;
            count[rootq] += count[rootp];
            
          
        }
        
    }
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int m = graph.length;
        int n = graph[0].length;
        unionfind(m);
        
for (int i = 0 ;i < m; ++i) {
            for (int j = i; j < n; ++j) {
                if (graph[i][j] == 1) {
                    union(i, j);
                }
            }
        }
        Arrays.sort(initial);
        Map<Integer, Integer> map = new HashMap<>();
        for (int i: initial) {
            map.put(find(i), map.getOrDefault(find(i), 0) + 1);
            
        }
        int max = 0;
        int maxIdx = initial[0];
        for (int i: initial) {
            if (map.get(find(i)) <= 1) {
               if (max < count[find(i)]) {
                   max = count[find(i)];
                   
maxIdx = i;
               }
            }
        }
        return maxIdx;
    }
}
