class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int[] sizes = new int[graph.length];
        
        
        for(int i = 0; i < initial.length; i++){
            int start = initial[i];
            // explore from the start point
            sizes[start] = helper(graph, start);
        }
        // System.out.println(Arrays.toString(sizes));
        int maxSize = 0;
        for(int i = 0; i < sizes.length; i++){
            maxSize = Math.max(maxSize, sizes[i]);
        }
        int result = 0;
        for(int i = 0; i < sizes.length; i++){
            if(sizes[i] == maxSize){
                result = i;
                break;
            
}
        }
        return result;
    }
    
    public int helper(int[][] graph, int start){
        boolean[] visited = new boolean[graph.length];
        visited[start] = true;
        Queue<Integer> q = new LinkedList<>();
        q.add(start);
        int count = 0;
        
while(q.size() > 0){
            int curKey = q.poll();
            count ++;
            for(int k = 0; k < graph.length; k++){
                if(!visited[k] && graph[curKey][k] == 1){
                    q.add(k);
                    visited[k] = true;
                
}
            }
        }
        return count;
    }
}
