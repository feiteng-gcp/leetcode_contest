class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        
int cand = -1, min = graph.length + 1, n = graph.length;
        for (int i : initial) {
            boolean[] visited = new boolean[n];
            int infected = 0;
            for (int remain : initial) {
                if (remain == i) {
                    continue;
                }
                infected += malware(graph, remain, visited);
            }
            // System.out.println(infected);
            if (infected == min && i < cand) {
                cand = i;
            }
            
if (infected < min) {
                cand = i;
                min = infected;
            }
        }
        return cand;
    }
    private int malware(int[][] graph, int n, boolean[] visited) {
        if (visited[n]) {
            return 0;
        }
        // System.out.println("n " + n);
        int result = 1;
        visited[n] = true;
        for (int next = 0; next < graph.length; next++) {
            if (graph[n][next] == 1) {
                result += malware(graph, next, visited);
            }
        }
        return result;
    }
}
