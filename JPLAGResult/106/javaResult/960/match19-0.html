class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        Arrays.sort(initial);
        int max = 0, ans = -1;
        HashSet<Integer> hash = new HashSet();
        for(int i=0; i<initial.length; i++) {
            int tmp = numOfMal(graph, initial[i], hash);
            if(tmp > max) {
                max = tmp;
                ans = initial[i];
            }
        }
        return ans;
        
    }
    private int numOfMal(int[][] graph, int initial, HashSet<Integer> hash) {
        int ans = 0;
        for(int i=0; i<graph[initial].length; i++) {
            if(graph[initial][i]==1) {
                if(!hash.contains(i)) {
                    hash.add(i);
                    ans = ans + 1 + numOfMal(graph, i, hash);
                }
            }
        }
        return ans;
    }
}
