class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int[] size = new int[graph.length];
        HashMap<Integer, HashSet<Integer>> components = new HashMap<>();
        for(int i = 0; i < size.length; i++) {
            if(size[i] == 0) {
                HashSet<Integer> c = bfs(graph, i, size);
                for(int n : c) {
                    components.put(n, c);
                }
            }
        }
        
        int node = -1;
        int maxSize = -1;
        for(int init : initial) {
            int cSize = size[init];
            HashSet<Integer> cComp = components.get(init);
            for(int other : initial) {
                if(other != init && cComp.contains(other)) {
                    cSize = 0;
                    break;
                }
            }
            
            
if(cSize > maxSize || (cSize == maxSize && init < node)) {
                node = init;
                maxSize = cSize;
            }
        }
        
        return node;
    }
    
    public HashSet<Integer> bfs(int[][] graph, int start, int[] size) {
        int componentSize = 0;
        HashSet<Integer> component = new HashSet<>();
        
        Queue<Integer> q = new ArrayDeque<>();
        q.add(start);
        while(!q.isEmpty()) {
            int top = q.remove();
            if(component.contains(top)) {
                continue;
            }
            componentSize++;
            component.add(top);
            
            
for(int i = 0; i < size.length; i++) {
                if(graph[top][i] == 1) {
                    q.add(i);
                }
            }
        }
        
        for(int n : component) {
            size[n] = componentSize;
        }
        
        return component;
    }
}
