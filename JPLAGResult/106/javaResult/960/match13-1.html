class Solution {
	    public int minMalwareSpread(int[][] graph, int[] initial) {
	    	int n = graph.length;
	        ArrayList<Integer>[] g = new ArrayList[n];
	        for(int i=0;i<n;i++) g[i] = new ArrayList<Integer>();
	        for(int i=0;i<n;i++) {
	        	for(int j=0;j<n;j++) {
	        		if(i!=j&&graph[i][j]==1) g[i].add(j);
	        	}
	        }
	        int min = Integer.MAX_VALUE;
	        int idx = 0;
	        for(int
 i=0;i<initial.length;i++) {
	        	int tmp = bfs(i,initial,g);
	        	if(tmp<min) {
	        		min = tmp;
	        		idx = initial[i];
	        	}else if(tmp==min){
	        		idx = Math.min(idx, initial[i]);
	        	}

	        }
	        return idx;
	    }
