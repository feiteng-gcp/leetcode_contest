class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        Arrays.sort(initial);
        int n = graph.length;
        int max = Integer.MIN_VALUE;
        int res = -1;
        for (int start : initial) {
            int cur = getConnectedNum(graph, start, new boolean[n]);
            if (cur > max) {
                res = start;
                max = cur;
            }
        }
        return res;
    }
    
    private int getConnectedNum(int[][] graph, int start, boolean[] visited) {
        visited[start] = true;
        int res = 1;
        for (int i = 0; i < graph.length; ++i) {
            if (graph[start][i] == 1 && !visited[i]) {
                res += getConnectedNum(graph, i, visited);
            }
        }
        return res;
    }
}
