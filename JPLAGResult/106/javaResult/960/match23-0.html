class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int[] infected = new int[graph.length];
        
        int max = 0;
        int ans = 0;
        
        Arrays.sort(initial);
        
        for (int i = 0; i < initial.length; i++) {
            Set<Integer> visited = new HashSet<>();
            helper(graph, initial[i], infected, visited);
        }
        
        for (int i = initial.length - 1; i >= 0; i--) {
            Set<Integer> visited = new HashSet<>();
            int tmp = helper2(graph, initial[i], infected, visited);
            
            if (tmp >= max) {
                max = tmp;
                ans = initial[i];
            }
        }
        
        return ans;
    } 
    
    private int helper2(int[][] graph, int cur, int[] infected, Set<Integer> visited) {
        if (visited.contains(cur)) {
            return 0;
        }
        
        visited.add(cur);
        
        int ans = infected[cur] == 1 ? 1 : 0;
        
        for (int i = 0; i < graph[cur].length; i++) {
            if (graph[cur][i] == 1) {
                ans += helper2(graph, i, infected, visited);
            }
        }
        
        return ans;
    }
    
    private void helper(int[][] graph, int cur, int[] infected, Set<Integer> visited) {
        if (visited.contains(cur)) {
            return;
        }
        
        visited.add(cur);
        
        infected[cur] += 1;

        for (int i = 0; i < graph[cur].length; i++) {
            if (graph[cur][i] == 1) {
                helper(graph, i, infected, visited);
            }
        }
    }
}
