class Solution {
int init = 0;
	int count;
	boolean[] flag;
	public void dfs(int[][] graph, int node) {
		for (int j = 0
; j < graph.length; j++) {
			if (graph[node][j] == 1 && !flag[j]) {
				flag[j] = true;
				count++;
				dfs(graph, j);
			}
		}
	}
	public int minMalwareSpread(int[][] graph, int[] initial) {
		int ans=Integer.MAX_VALUE, min=-Integer.MAX_VALUE;
		flag = new boolean[graph.length];
		Arrays.fill(flag, false);
		count = 0;
		for (int i = 0; i < initial.length; i++) {
			if (!flag[initial[i]]) {
				count++;
				flag[initial[i]] = true;
				dfs(graph, initial[i]);
			}
		}
		init = count;
		
		for (int i = 0; i < initial.length; i++) {
			count = 0;

se);
			for (int j = 0; j < initial.length; j++) {
				if (j != i && !flag[initial[j]]) {
					count++;
					flag[initial[j]] = true;
					dfs(graph, initial[j]);
				}
			}
			
			if (init - count >= m
in && initial[i] < ans) {
				min = init - count;
				ans = initial[i];
