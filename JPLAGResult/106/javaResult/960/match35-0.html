class Solution {
     private void find(int res, Set<Integer>[] cache, int[][] graph, boolean[] find, Set<Integer> resSet) {

        if (cache[res] != null) {
            for (Integer x: cache[res]) {
                find[x] = true;
                resSet.add(x);
            }
        }

        resSet.add(res);
        for (int i = 0;i < graph[res].length;i++) {
            if (graph[res][i] == 1 && !find[i]) {
                find[i] = true;
                find(i, cache, graph, find, resSet);
            }
        }
    }

    public int minMalwareSpread(int[][] graph, int[] initial) {
        Set<Integer>[] cache = new Set[graph.length];

        boolean[] flag = new boolean[graph.length];
        for (int i = 0;i < graph.length;i++) {
            Arrays.fill(flag, false);
            flag[i] = true;
            Set<Integer> s = new HashSet<>();
            find(i, cache, graph, flag, s);
            cache[i] = s;
        }

        int[] time = new int[graph.length];
        Arrays.fill(time, 0);
        for (int i = 0;i < initial.length;i++) {
            int x = initial[i];
            for (Integer children: cache[x]) {
                time[children]++;
            }
        }
        int res = 0;
        for (int i = 0;i < graph.length;i++) {
            if (time[i] > 0) {
                res++;
            }
        }
        int minM = res;
        int min = -1;

        for (int i = 0;i < initial.length;i++) {
            int x = initial[i];
            int t = res;
            for (Integer children: cache[x]) {
                if (time[children] == 1) {
                    t--;
                }
            }
            if (min == -1 || t < minM || (t == minM && x < min)) {
                minM = t;
                min = x;
            }
        }
        return min;
    }
}
