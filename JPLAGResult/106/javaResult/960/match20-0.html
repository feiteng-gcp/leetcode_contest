class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        Arrays.sort(initial);
        int res = -1, max = -1;
        int num = graph.length;
        for (int i : initial) {
            boolean[] visited = new boolean[num];
            int r = dfs(graph, visited, i);
            if (r > max) {
                max = r;
                res = i;
            }
        }
        return res;
    }
    
    private int dfs(int[][] graph, boolean[] visited, int cur) {
        if (visited[cur]) return 0;
        visited[cur] = true;
        int res = 1;
        for (int next = 0; next < visited.length; ++next) {
            if (graph[cur][next] > 0) res += dfs(graph, visited, next);
        }
        return res;
    }
}
