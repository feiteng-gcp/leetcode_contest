class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int n = graph.length;
        int min = Integer.MAX_VALUE, res = -1;
        Arrays.sort(initial);
        for (int i = 0; i < initial.length; i++) {
            int ans = influence(graph, initial, i);
            // System.out.println(ans);
            
if (ans < min) {
                min = ans;
                res = initial[i];
            }
        }
        return res;
    }
    private int influence(int[][] g, int[] init, int remove) {
        Queue<Integer> q = new LinkedList<>();
        int n = g.length, res = init.length - 1;
        
        
boolean[] visited = new boolean[n];
        for (int i = 0; i < init.length; i++) {
            if (i != remove) {
                q.add(init[i]);
                visited[init[i]] = true;
            }
        }
        
        while (!q.isEmpty()) {
            int cur = q.poll();
            for (int i = 0; i < n; i++) {
                if (i == cur || g[cur][i] == 0 || visited[i]) continue;
                q.add(i);
                visited[i] = true;
                res++;
            }
        }
        return res;
    }
    
}
