class Solution {
    Set<Integer> visited = new HashSet<>();
    public int minMalwareSpread(int[][] graph, int[] initial) {
        PriorityQueue<List<Integer>> pq = new PriorityQueue<>((a,b)->b.size()-a.size());
    
        
        for(int i=0;i<graph.length;i++){
            if(!visited.contains(i)){
                pq.add(dfs(graph,i));
            }
        }
        Set<Integer> set = new HashSet<>();
        for(int i:initial) set.add(i);
        for(List<Integer> list:pq){
            Collections.sort(list);
            for(Integer i:list)
                if(set.contains(i))
                    return i;
        }
        return -1;
        
    }
    private List<Integer> dfs(int[][] graph, int i){
        List<Integer> list = new LinkedList<>();
        list.add(i);
        visited.add(i);
        for(int j=0;j<graph.length;j++){
            if(!visited.contains(j)&&graph[i][j]==1)
                list.addAll(dfs(graph,j));
        }
        return list;
    }
}
