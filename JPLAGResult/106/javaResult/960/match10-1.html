class Solution {
    public int minMalwareSpread(int[][] graph, int[] initial) {
        int n = graph.length;
        int result = Integer.MAX_VALUE;
        int resultNode = 0;
        for (int i = 0; i < initial.length; i++) {
            int taken = initial[i];
            
            
Queue<Integer> queue = new LinkedList<Integer>();
            boolean[] visited = new boolean[n];
            for (int j = 0; j < initial.length; j++) {
                if (j != i) {
                    queue.offer(initial[j]);
                    visited[initial[j]] = true;
                }
            }
            
            int infected = 0;
            
while (!queue.isEmpty()) {
                int node = queue.poll();
                for (int k = 0; k < graph[node].length; k++) {
                    if (k != node && !visited[k] && graph[node][k] == 1) {
                        queue.offer(k);
                        visited[k] = true;
                    } 
                }
                
                infected ++;
            }
            
            
if ((infected < result) || (infected == result && taken < resultNode)) {
                result = infected;
                resultNode = taken;
            }
        }
        
        return resultNode;
    }
}
