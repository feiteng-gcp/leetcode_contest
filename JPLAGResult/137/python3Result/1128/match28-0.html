class Solution:
    def removeDuplicates(self, S: str) -> str:
        stack = []
        for ch, gp in itertools.groupby(S):
            if len(list(gp)) & 1:
                if stack and stack[-1] == ch:
                    stack.pop()
                else:
                    stack.append(ch)
        return ''.join(stack)
