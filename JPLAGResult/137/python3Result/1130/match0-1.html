class Solution:
    def lastStoneWeightII(self, stones: list) -> int:
        cur_list = {stones[0],-stones[0]}
        next_list = set()
        for i in range(1,len(stones)):
            for cur in cur_list:
                next_list.add(cur + stones[i])
                next_list.add(cur - stones[i])
            cur_list = next_list
            next_list = set(
)
        res = float('inf')
        for elem in cur_list:
            if elem >= 0:
                res = min(res,elem)
        return res
