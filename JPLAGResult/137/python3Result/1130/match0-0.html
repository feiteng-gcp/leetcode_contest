class Solution:
    def lastStoneWeightII(self, stones: List[int]) -> int:
        n = len(stones)
        
        if n == 1:
            return stones[0]
        
        stones.sort()
        target = stones[-1]
        
        candidates = set([target, -target])
        for s in stones[:n-1]:
            new = set()
            for c in candidates:
                new.add(c + s)
                new.add(c - s)
            candidates = new
            
        res = float("Inf")
        for c in candidates:
            if c >= 0:
                res = min(res, c)
                
        return res
