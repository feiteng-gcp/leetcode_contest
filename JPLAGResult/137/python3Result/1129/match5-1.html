class Solution:
    def longestStrChain(self, words: List[str]) -> int:
        
        def ispredecessor(word1, word2):
            if len(word2)-len(word1)!=1:
                return False
            flag = 0
            ptr1, ptr2 = 0, 0
            while ptr1 < len(word1) and ptr2 < len(word2):
                if word1[ptr1] == word2[ptr2]:
                    ptr1, ptr2 = ptr1 + 1, ptr2 + 1
                else:
                    flag += 1
                    ptr2 += 1
                    if flag > 1:
                        return False
            return True
                    
        #print(ispredecessor("a", "ab"))   
        #print(ispredecessor("a", "ba"))   
            
        WordLenBag = [[] for x in range(17)]
        for word in words:
            WordLenBag[len(word)].append([word, 1])
        #print(WordLenBag)
        MaxLen = 1
        for CheckLen in range(2, 17):
            if len(WordLenBag[CheckLen]) == 0:
                continue
            Nwords = len(WordLenBag[CheckLen])
            for IdxWord in range(Nwords):
                for IdxWord1, word1 in enumerate(WordLenBag[CheckLen-1]):
                    #print("check:", word1, WordLenBag[CheckLen][IdxWord][0])
                    if ispredecessor(word1
[0], WordLenBag[CheckLen][IdxWord][0]):
                        #print("%s is the predecessor of %s" %(word1[0], WordLenBag[CheckLen][IdxWord][0]))
                        WordLenBag[CheckLen][IdxWord][1] = max(WordLenBag[CheckLen-1][IdxWord1][1] + 1, WordLenBag[CheckLen][IdxWord][1])
                #print("Done with #%d word in layer %d, len = %d" %(IdxWord, CheckLen, WordLenBag[CheckLen][IdxWord][1]))
                MaxLen = max(MaxLen, WordLenBag[CheckLen][IdxWord][1])
        return MaxLen
