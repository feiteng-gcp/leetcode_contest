class Solution:
    def longestStrChain(self, words: List[str]) -> int:
        d = collections.defaultdict(list)
        for i in range(len(words)):
            d[len(words[i])].append(words[i])
        #print(d)
        
        a = d.keys()
        a = list(a)
        a.sort()
        #print(a)
        
        v = {}
        for i in words:
            v[i] = 0 
        
        def check(y,u):
            i = 0
            j = 0
            k = 0
            e = 0
            while i < len(y):
                if y[i] == u[j]:
                    i = i+1
                    j = j+1
                    k = k+1
                else:
                    j = j+1
                    e = e+1
                    if e == 2:
                        return False
            return k == len(y)
        
        def dfs(l,k,n):
            if k > self.m:
                self.m = k
            if not d[n+1]:
                return 
            else:
                for i in d[n+1]:
                    if v[i] == 0:
                        if check(l,i):
                            v[i] = 1
                            dfs(i,k+1,n+1)
        
        self.m = 0
        for i in range(len(a)):
            for j in range(len(d[a[i]])):
                if v[d[a[i]][j]] == 0:
                    v[d[a[i]][j]] = 1
                    dfs(d[a[i]][j],1,a[i])
        
        return self.m
