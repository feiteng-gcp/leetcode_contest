class Solution:
    def longestStrChain(self, words: List[str]) -> int:
        word_set = set(words)
        words = list(word_set)
        words.sort(key = lambda s: len(s))

        word_chain_dict = {}
        for word in words:
            word_chain_dict[word] = 1
            
        for word in words:
            for i in range(len(word)):
                new_word = word[:i] + word[i+1:]
                if new_word in word_set:
                    word_chain_dict[word] = max(word_chain_dict[word], word_chain_dict[new_word] + 1)
        
        return max(word_chain_dict.values())
