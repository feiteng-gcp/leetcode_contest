class Solution:
    def longestStrChain(self, words: 'List[str]') -> int:
        nextw={}
        words=list(set(words))
        words.sort(key=len)
        dic={} #len to word set
        searched={}
        for i,w in enumerate(words):
            l=len(w)
            if not l in dic:
                dic[l]=[]
            dic[l].append(w)
        
        def isInc(w1,w2):
            p1
=p2=0;inc=False
            while p1<len(w1):
                if w1[p1]==w2[p2]:
                    p1+=1;p2+=1
                elif not inc:
                    p2+=1
                    inc=True
                else:
                    return False
            return True
        
        def search(w):
            if w in searched:
                return searched[w]
            searched[w]=1
            nextlen=len(w)+1
            tmp=0
            if nextlen in dic:
                for i,nw in enumerate(dic[nextlen]):
                    if isInc(w,nw):
                        chainlen=search(nw)
                        if chainlen>tmp:
                            nextw[w]=nw
                        tmp=max(tmp,chainlen)
            searched[w]+=tmp
            return searched[w]
        
        ans=0
        for w in words:
            ans=max(ans,search(w))
        return ans
