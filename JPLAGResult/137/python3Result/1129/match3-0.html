class Solution:
    def longestStrChain(self, words: List[str]) -> int:
        def isPre(a,b):
            i = j = 0
            count = 0
            while i<len(a):
                if a[i]==b[j]:
                    i+=1
                    j+=1  
                else:
                    count += 1
                    if count > 1:
                        return False
                    j+=1
            return True
        
        
        dic = collections.defaultdict(list)
        for word in words:
            dic[len(word)] += [word]        
        minlen = min(dic)
        maxlen = max(dic)
        
        g = collections.defaultdict(list)
        for l in range(minlen,maxlen+1):
            alist = dic[l]
            blist = dic[l+1]
            for a in alist:
                for b in blist:
                    if isPre(a,b):
                        g[a] += [b]
        
        ans = 0
        seen = set()
        for l in range(minlen,maxlen+1):
            stack = [(node,l) for node in dic[l]]
            while stack:
                #print(stack)
                v,length = stack.pop()
                ans = max(ans,length-l+1)
                if v not in seen:
                    seen.add(v)
                    pres = g[v]
                    for pre in g[v]:
                        stack.append((pre,length+1))
            
        
        return ans
