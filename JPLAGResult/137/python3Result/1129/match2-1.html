class Solution:
    def longestStrChain(self, words: List[str]) -> int:
        words.sort(key = lambda x: len(x))
        # print(words)
        d = collections.defaultdict(set)
        for w in words:
            for i in range(len(w)-1):
                d[w].add(w[:i]+w[i+1:])
            d[w].add(w[:-1])
        ans = [0]*len(words)
        for i in range(len(words)):
            tmp = 1
            for j in range(i):
                if words[j] in d[words[i]]:
                    tmp = max(tmp,ans[j]+1)
            ans[i] = tmp
        # print(ans)
        return max(ans)
