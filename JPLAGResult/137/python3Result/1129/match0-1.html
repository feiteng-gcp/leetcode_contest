class Solution:
    def longestStrChain(self, words: List[str]) -> int:
        if not words:
            return 0
        # prev=collections.defaultdict(list)
        suc=collections.defaultdict(list)
        words.sort(key=lambda x:(len(x),x))
        for word in words:
            for i in range(len(word)):
                # prev[word].append(str(word[:i]+word[i+1:]))
                suc[str(word[:i]+word[i+1:])].append(word)
        ans=0
        for w in words:
            _ans=0
            q=collections.deque([w])
            visited=set()
            while(q):
                nq=[]
                while(q):
                    cur=q.popleft()
                    if cur in visited:
                        continue
                    visited.add(cur)
                    nq+=suc[cur]
                _ans+=1
                if not nq:
                    break
                q.extend(nq)
            ans=max(ans,_ans)
        return ans
