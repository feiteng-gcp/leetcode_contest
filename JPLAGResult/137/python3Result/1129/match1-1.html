class Solution:
    def longestStrChain(self, words: List[str]) -> int:        
        n = len(words)
        idx = sorted(range(n), key=lambda i: len(words[i]))
        seen = {}
        seen[''] = 0
        ans = 1
        for i in range(n):
            w = words[idx[i]]
            seen[w] = 1
            
for j in range(len(w)):
                new = w[:j] + w[j+1:]
                if new in seen:
                    seen[w] = max(seen[w],seen[new]+1)
            ans = max(ans,seen[w])

        return ans
