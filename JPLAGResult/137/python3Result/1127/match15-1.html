from heapq import heapify, heappush, heappop
from typing import List
class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        n = len(stones)
        #[7,8,4,2,1,1]
        # x=7, y=8
        # 
        if n == 1:
            return stones[0]
        max_heap 
= [-1*stone for stone in stones]
        heapify(max_heap)
        
        
        while len(max_heap) > 1:
            y = -1 * heappop(max_heap)
            x = -1 * heappop(max_heap)
            if x != y:
                heappush(max_heap, -1 * (y-x))
        
        return 0 if len(max_heap) == 0 else -1 * max_heap[0]
