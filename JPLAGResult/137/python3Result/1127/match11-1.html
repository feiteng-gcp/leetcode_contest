class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        while len(stones) > 1:
            stones.sort(reverse=True)
            if stones[0] == stones[1]:
                stones = stones[2:]
            else:
                dummy = stones[0] - stones[1]
                stones = stones[2:]
                stones.append(dummy)

        return stones[0] if stones else 0
