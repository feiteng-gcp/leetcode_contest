class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        s = [-x for x in stones]
        heapq.heapify(s)
        
        while len(s) >= 2:
            first = heapq.heappop(s)
            second = heapq.heappop(s)
            if first != second:
                heapq.heappush(s, first - second)
                
        if len(s) == 0:
            return 0
        else:
            return -s[0]
