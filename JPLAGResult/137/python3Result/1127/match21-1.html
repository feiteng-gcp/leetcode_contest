import bisect

class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        stones.sort()
        while stones:
            
if len(stones) == 1:
                return stones[0]
            stone_1 = stones.pop()
            stone_2 = stones.pop()
            if stone_1 == stone_2:
                continue
            else:
                new_stone = stone_1 - stone_2
                stones.insert(bisect.bisect_left(stones,new_stone),new_stone)

        return 0
