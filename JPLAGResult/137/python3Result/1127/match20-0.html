class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        h=[(-x,x) for x in stones]
        heapq.heapify(h)
        while len(h)>1:
            x1=heapq.heappop(h)
            x2=heapq.heappop(h)
            diff=x1[1]-x2[1]
            if diff!=0:
                heapq.heappush(h,(-diff,diff))
        if not h:
            return 0
        return h[0][1]
