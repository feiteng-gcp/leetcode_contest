class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        while(len(stones)>1):
            stones.sort()
            L=len(stones)
            if stones[L-1]==stones[L-2]:
                del stones[-1]
                del stones[-1]
            else:
                new=stones[L-1]-stones[L-2]
                del stones[-1]
                del stones[-1]
                stones
.append(new)
        if len(stones)==0:
            return 0
        else:
            return stones[0]
