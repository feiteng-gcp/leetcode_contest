class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        s = stones
        s.sort()
        while len(s) > 1:
            if s[-1] == s[-2]:
                s = s[:-2]
            else:
                t = s[-1]-s[-2]
                s = s[:-2]
                s.append(t)
                s.sort()
            
            
        if len(s) == 0:
            return 0
        else:
            return s[0]
