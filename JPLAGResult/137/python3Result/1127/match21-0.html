import heapq
class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        h = []
        for s in stones:
            heapq.heappush(h,-s)
        while h:
            print(h)
            if len(h) < 2:
                return 0 if not h else -h[0]
            s1 = heapq.heappop(h)
            s2 = heapq.heappop(h)
            if -s1 < -s2:
                heapq.heappush(h,s2-s1)
            else:
                if -s1 == -s2:
                    continue
                else:
                    heapq.heappush(h,s1-s2)
        if not h:
            return 0
