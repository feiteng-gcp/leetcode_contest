from bisect import bisect

class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        stones.sort()
        
        while len(stones) > 1:
            a = stones.pop()
            b = stones.pop()
            delta = a - b
            if delta > 0:
                ind = bisect(stones, delta)
                stones.insert(ind, delta)
        
        if len(stones) == 1:
            return stones[0]
        
        else:
            return 0
