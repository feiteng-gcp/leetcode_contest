from heapq import *
class Solution:
    def lastStoneWeight(self, stones):
        stones = [-stone for stone in stones]
        heapify(stones)
        while len(stones) > 1:
            heaviest = -heappop(stones)
            heaviest2 = -heappop(stones)
            if heaviest != heaviest2:
                heappush(stones, heaviest2 - heaviest)
        if(len(stones) == 1): return -heappop(stones)
        else: return 0
