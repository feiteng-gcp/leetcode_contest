class Solution:
    def lastStoneWeight(self, stones) -> int:
        stones = sorted( -s for s in stones )
        while len(stones) > 1:
            a = heapq.heappop(stones)
            b = heapq.heappop(stones)
            c = -abs(a - b)
            if c != 0:
                heapq.heappush(stones, c)
        return -stones[0] if len(stones) else 0
