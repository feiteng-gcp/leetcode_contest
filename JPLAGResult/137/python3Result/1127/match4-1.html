class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        stones = [-v for v in stones]
        heapq.heapify(stones)
        while len(stones) > 1:
            a = heapq.heappop(stones)
            b = heapq.heappop(stones)
            if a != b:
                heapq.heappush(stones,a-b)
        if not stones:
            return 0
        else:
            return -stones[0]
