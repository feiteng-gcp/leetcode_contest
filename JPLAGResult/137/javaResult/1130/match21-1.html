class Solution {
    public int lastStoneWeightII(int[] stones) {
        int n = stones.length;
        
        int sum = 0;
        for (int i = 0; i < n; i++) {
            sum += stones[i];
        }
        
        boolean dp[][] = new boolean[n+1][sum+1];
        
		for (int i = 0; i <= n; i++) { 
			for (int j = 0; j <= sum; j++) { 
                if (j == 0) {
                    dp[i][j] = true;
                    continue;
                }
                
                if (i == 0) {
                    dp[i][j] = false;
                    continue;
                }
                
				dp[i][j] = dp[i - 1][j]; 

				if (stones[i - 1] <= j) 
					dp[i][j] |= dp[i - 1][j - stones[i - 1]]; 
			} 
		} 
	
		int min = Integer.MAX_VALUE; 
		 
		for (int j = sum / 2; j >= 0; j--) 
		{ 
			if (dp[n][j] == true) 
			{ 
				min = sum - 2 * j; 
				break; 
			} 
		} 
		return min; 
    }
}
