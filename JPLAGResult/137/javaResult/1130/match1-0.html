class Solution {
    public int lastStoneWeightII(int[] stones) {
        
        // Max possible sum for nums
        int max = 0;
        for (int num : stones)    max += num;
        
        // Build dp array
        int[][] dp = new int[stones.length][max * 2 + 1];
        dp[0][stones[0] + max]++;
        dp[0][-stones[0] + max]++;
        
        for (int i = 1; i < dp.length; i++) {
            for (int j = 0; j < dp[0].length; j++) {
                if (dp[i - 1][j] != 0) {
                    dp[i][j - stones[i]] += dp[i - 1][j];
                    dp[i][j + stones[i]] += dp[i - 1][j];
                }
            }
        }
        
        int min = 0;
        for (int i = max; i < dp[0].length; i++) {
            if (dp[stones.length - 1][i] != 0) {
                min = i - max;
                break;
            }
        }
        
        return min;
    }
}
