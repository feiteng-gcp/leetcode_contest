class Solution {
    public int lastStoneWeightII(int[] stones) {
        int sum = 0;
        for (int stone: stones) {
            sum += stone;
        }
        
        int target = sum / 2 ;
        boolean[] total = new boolean[target + 1];
        total[0] = true;
        int max = 0;
        
for (int stone : stones) {
            for (int i=target; i>=stone; i--) {
                if (total[i-stone]) {
                    total[i] = true;
                    max = Math.max(max, i);
                }
            }
        }
        return sum - 2*max;
    }
}
