class Solution {
    
    public int lastStoneWeightII(
int[] stones){
        int S = 0;
        for (int i=0; i<stones.length; i++)
            S += stones[i];
        boolean dp[][] = new boolean[stones.length + 1][S + 1];
        for (int i=0; i<=stones.length; i++)
            dp[i][0] = true;
        for (int i=1; i<=S; i++)
            dp[0][i] = false;
        for (int i = 1; i <= stones.length; i++) {
            for (int j = 1; j <= S; j++) {
                dp[i][j] = dp[i - 1][j];
                if (stones[i - 1] <= j)
                    dp[i][j] |= dp[i - 1][j - stones[i - 1]];
            }
        }
        int minDif = Integer.MAX_VALUE;
        for (int j = S/2; j >= 0; j--) {
            if (dp[stones.length][j] == true)
            {
                minDif = S - 2 * j;
                break;
            }
        }
        return minDif;
    }
}
