class Solution {
    public int lastStoneWeightII(int[] stones) {
        int sum = 0;
        for (int cur : stones) {
            sum += cur;
        }
        
        int len = stones.length;
        boolean[][] dp = new boolean[30][3001];
        dp[0][0] = true;
        dp[0][stones[0]] = true;
        
        int ret = Math.abs(sum - stones[0] - stones[0]);
        
        
for (int i = 0; i < len - 1; i++) {
            for (int j = 0; j <= 3000; j++) {
                if (dp[i][j]) {
                    dp[i + 1][j] = true;
                    dp[i + 1][j + stones[i + 1]] = true;
                    ret = Math.min(ret, 
Math.abs(sum - j - stones[i + 1] - j - stones[i + 1]));
                }
            }
        }
        return ret;
    }
}
