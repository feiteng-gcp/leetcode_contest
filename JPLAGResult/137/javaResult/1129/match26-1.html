class Solution {
public int longestStrChain(String[] words) {
        int result = 1;
        Arrays.sort(words, (a, b)-> {return Integer.compare(a.length(), b.length());});
        int[] length = new int[words.length];
        length[0] = 1;
        for(int i=1; i< length.length; i++){
            if(words[i].length() == 1){
                length[i] = 1;
            }
            int max = 1;
            
for(int j=0;j< i; j++){
                if(isPre(words[j], words[i])){
                    if(max < length[j]+1){
                        max = length[j]+1;
                    }
                }
            }
            length[i] = max;
            if(max > result){
                result = max;
            }
        }
        
return result;
    }
    private boolean isPre(String pre, String current){
        if(pre.length()+1 != current.length()) {
            return false;
        }
        int preIndex = 0;
        int currentIndex = 0;
        while(preIndex >= currentIndex-1 && preIndex<pre.length()){
            if(pre.charAt(preIndex) == current.charAt(currentIndex)){
                preIndex++;
                currentIndex++;
            }else{
                currentIndex++;
            }
        }
        if(preIndex == currentIndex-1 || preIndex == currentIndex){
            return true;
        }else{
            return false;
        }
    }
}
