class Solution {

    public int longestStrChain(String[] words) {
        int n = words.length;
        int[] len = new int[n];
        Arrays.fill(len, 1);
        Arrays.sort(words, (a, b) -> Integer.compare(a.length(), b.length()));
        // System.out.println(Arrays.toString(words));
        for (int j=1; j<n; ++j) {
            for (int i=0; i<j; ++i) {
                if (len[i] + 1 < len[j]) {
                    continue;
                }

                if ( isPredecessor(words[i], words[j]) ) {
                    len[j] = len[i] + 1;
                }
            }
        }

        int maxLen = 0;
        for (int l: len) {
            maxLen = Math.max(l, maxLen);
        }
        return maxLen;
    }

    private boolean isPredecessor (String a, String b) {
        int lenA = a.length();
        int lenB = b.length();
        if (lenA + 1 != lenB) return false;

        char[] ah = a.toCharArray();
        char[] bh = b.toCharArray();

        boolean flag = false;
        for (int i=0, j=0; i<lenA && j<lenB; ) {
            if (ah[i] == bh[j]) {
                ++i;
                ++j;
            } else {
                if (flag) return false;
                flag = true;
                ++j;
            }
        }

        return true;
    }
}
