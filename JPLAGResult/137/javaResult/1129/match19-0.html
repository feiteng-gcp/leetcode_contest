class Solution {
    public int longestStrChain(String[] words) {
        Arrays.sort(words, (a, b) -> a.length() - b.length());
        int[] len = new int[words.length];
        int max = 0;
        for (int i=0; i<words.length; i++) {
            len[i] = 1;
            for (int j=i-1; j>=0; j--) {
                if (isPre(words[j], words[i])) {
                    len[i] = Math.max(len[j]+1, len[i]);
                }
            }
            max = Math.max(max, len[i]);
        }
        return max;
    }
    
    private boolean isPre(String s1, String s2) {
        if (s1.length() != s2.length() - 1) { return false; }
        int l1 = 0, r1 = s1.length()-1, l2 = 0, r2 = s2.length()-1;
        while (l1 <= r1 && l2 <= r2 && s1.charAt(l1) == s2.charAt(l2)) {
            l1++; 
            l2++;
        }
        
        while (r1 >= l1 && r2 >= l2 && s1.charAt(r1) == s2.charAt(r2)) {
            r1--;
            r2--;
        }
        return l2 == r2;
    }
}
