class Solution {
    public int max = 0;

    public int longestStrChain(String[] words) {
        int N = words.length;
        List<Integer>[] graph = new ArrayList[N];
        for (int i = 0; i < N; i ++)
            graph[i] = new ArrayList<Integer>();
        
        for (int i = 0; i < N; i ++) {
            for (int j = 0; j < N; j ++) {
                if(isPred(words[i], words[j])) {
                    graph[i].add(j);
                    //System.out.println(i+" "+j);
                }
            }
        }
        /*
        int[][] dp = new int[N][N]; // dp[i][j] is the longeset chain from word i to word j
        
        */
        for (int i = 0; i < N; i ++)
            dfs(i, graph, new HashSet<Integer>(), 1);
        
        return max;
    }
    
    public void dfs(int cur, List<Integer>[] graph, Set<Integer> visited, int steps) {
        if(visited.contains(cur)) return;
        visited.add(cur);
        for (int nei: graph[cur]) {
            if(!visited.contains(nei))
                dfs(nei, graph,visited,steps+1);
        }
        max = Math.max(max, steps);
        visited.remove(cur);
    
}
    
    public boolean isPred(String A, String B) {
        
        if(A.length() + 1 == B.length()) {
            
int skip = 0;
            int i = 0; int j = 0;
            while(i < B.length() && j < A.length()) {
                if(A.charAt(j) == B.charAt(i)) {
                    i++;
                    j++;
                }
                else
                {
                    skip ++;
                    i++;
                    if(skip > 1) return false;
                }
            }
            return true;
        }
        return false;
    }
}
