class Solution {
    public int longestStrChain(String[] words) {
        boolean pred[][] = new boolean[words.length][words.length];
        
for (int i = 0; i < words.length; i++) {
            for (int j = 0; j < words.length; j++) {
                if (isPred(words[i], words[j]))
                    pred[i][j] = true;
            }
        }
        List<String> l[] = new ArrayList[16];
        List<Integer> maxl[] = new ArrayList[16];
        for (int i = 0; i < 16; i++) {
            l[i] = new ArrayList<>();
            maxl[i] = new ArrayList<>();
        }
        for (int i = 0; i < words.length; i++) {
            l[words[i].length() - 1].add(words[i]);
            maxl[words[i].length() - 1].add(1);
        }
        for (int i = 14; i >= 0; i--) {
            for (int k = 0; k < l[i].size(); k++) {
                for (int j = 0; j < l[i + 1].size(); j++) {
                    if (isPred(l[i].get(k), l[i + 1].get(j)) && maxl[i].get(k) < maxl[i + 1].get(j) + 1) {
                        maxl[i].set(k, maxl[i + 1].get(j) + 1);
                    }
                }
            }
        }
        int res = 0;
        for (int i = 15; i >= 0; i--) {
            for (int k = 0; k < l[i].size(); k++) {
                if (res < maxl[i].get(k))
                    res = maxl[i].get(k);
            }
        
}
        return res;
    }
    private boolean isPred(String word1, String word2) {
        if (word1.length() + 1 != word2.length())
            return false;
        boolean found = false;
        for (int i = 0; i < word1.length(); i++) {
            if (!found && word1.charAt(i) != word2.charAt(i)) {
                found = true;
                i--;
            } else if (found && word1.charAt(i) != word2.charAt(i + 1)) {
                return false;
            }
        }
        return true;
    }
}
