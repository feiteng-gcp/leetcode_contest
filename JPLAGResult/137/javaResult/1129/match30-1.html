class Solution {
      // "abc" is a predecessor of "abac"
  
static boolean isPredecessor(String w1, String w2) {
    if (w1.length() + 1 != w2.length()) return false;
    int i = 0;
    int j = 0;
    boolean added = false;
    while (i < w1.length() && j < w2.length()) {
      if (w1.charAt(i) == w2.charAt(j)) {
        i++;
        j++;
      } else {
        if (added) return false;
        added = true;
        j++;
      }
    }
    return true;
  }
    public int longestStrChain(String[] words) {
    int N = words.length;
    ArrayList<ArrayList<Integer>> edges = new ArrayList<>();
    for (int i = 0; i < N; i++) {
      edges.add(new ArrayList<Integer>());
      for (int j = 0; j < N; j++) {
        if (isPredecessor(words[i], words[j])) {
          // System.out.println(words[i] + " " + words[j]);
          edges.get(i).add(j);
        }
      }
    }
    Queue<Integer> q = new LinkedList<>();
    int maxdepth = 0;
    for (int i = 0; i < N; i++) {
      q.clear();
      q.add(i);
      int depth = 1;
      Queue<Integer> r = new LinkedList<>();
      while (true) {
        while (!q.isEmpty()) {
          int k = q.poll();
          for (int j : edges.get(k)) {
            r.add(j);
          }
        }
        // System.out.println(i + " r: " + join(r,","));
        if (!r.isEmpty()) {
          depth++;
          q = r;
          r = new LinkedList<>();
        } else {
          // System.out.println(words[i] + " depth " + depth);
          break;
        }
      }
      maxdepth = Math.max(depth, maxdepth);
    }
    return maxdepth;        
    }
}
