class Solution {
    public int longestStrChain(String[] words) {
        Arrays.sort(words, new Comparator<String>() {
            @Override
            public int compare(String str1, String str2) {
                return Integer.compare(str1.length(), str2.length());
            }
        });
        
        int result = 0;
        Set<String> yesCache = new HashSet<>();
        Set<String> noCache = new HashSet<>();
        
        int[] dp = new int[words.length];
        
        for(int i = 0; i < dp.length; i ++) {
            dp[i] = 1;
        }
        
        for(int i = 1; i < words.length; i ++) {
            String current = words[i];
            
            for(int j = i - 1; j >=0; j --) {
                String last = words[j];
                
                if(last.length() < current.length() - 1) {
                    break;
                } else if(last.length() == current.length() - 1) {
                    boolean isPredecessor = checkPredecessor(words, j, i, yesCache, noCache);
                    
                    if(isPredecessor) {
                        dp[i] = Math.max(dp[i], dp[j] + 1); 
                    }
                }
            }
            
            result = Math.max(result, dp[i]);
        }
        
        return result;
    }
    
    private boolean checkPredecessor(String[] words, int j, int i, Set<String> yesCache, Set<String> noCache) {
        String key = j + "," + i;
        
        if(yesCache.contains(key)) {
            return true;
        } else if(noCache.contains(key)) {
            return false;
        } else {
            boolean result = isPredecessor(words[j], words[i]); 
            
            if(result) {
                yesCache.add(key);
            } else {
                noCache.add(key);
            }
            
            return result;
        } 
    
}
    
    private boolean isPredecessor(String str1, String str2) {
        // System.out.println(str1 + "," + str2);
        
        if(str1.length() != str2.length() - 1) {
            return false;
        }
        
        int pointer = 0;
        
        for(int i = 0; i < str2.length(); i ++) {
            if(pointer < str1.length() && str2.charAt(i) == str1.charAt(pointer)) {
                pointer ++;
            }
        }
        
        return pointer == str1.length();
    }
}
