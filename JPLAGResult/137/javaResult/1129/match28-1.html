class Solution {
    public int longestStrChain(String[] words) {
        int[][] graph = new int[words.length][words.length];
        
        for (int i = 0; i < words.length;i++){
            for (int j = 0; j < words.length;j++){
                if (i == j) continue;
                if (words[i].length() + 1 != words[j].length()) continue
;
                if (isValid(words[i], words[j])) {
                    graph[i][j] = 1;
                }
            }
        }
        
        int max = 0;
        for (int i = 0; i < graph.length;i++){
            int curr = dfs(i, graph);
            if (curr > max) max = curr;
        
}
        return max;
    }
    
    public static int dfs(int curr, int[][] graph){
        
        int[] con = graph[curr];
        int max = 0;
        for (int i = 0; i < con.length;i++){
            if (con[i] == 0) continue;
            int cur = dfs(i, graph);
            if (cur > max) max = cur;
        }
        return 1 + max;
    }
    public static boolean isValid(String a, String b){
        int i = 0;
        int j = 0;
        int count = 0;
        
        while (i < a.length()){
            if (a.charAt(i) == b.charAt(j)){
                i++;
                j++;
            }else{
                if (count == 1) return false;
                count++;
                j++;
            }
        }
        return true;
    }
}
