class Solution {
    public int longestStrChain(String[] words) {
        Map<Integer, Set<String>> map = new HashMap<>();
        for (String word : words) {
            map.putIfAbsent(word.length(), new HashSet<>());
            map.get(word.length()).add(word);
        }
        Map<String, Integer> distMap = new HashMap<>();
        int max = 0;
        for (String word : words) {
            max = Math.max(max, 
getDist(map, distMap, word));
        }
        return max;
    }
    
    private int getDist(Map<Integer, Set<String>> map, Map<String, Integer> distMap, String word) {
        if (distMap.containsKey(word))  return distMap.get(word);
        int n = word.length();
        if (!map.containsKey(n - 1)){
            distMap.put(word, 1);
            return 1;
        }
        Set<String> candidates = map.get(n - 1);
        
int max = 1;
        for (int i = 0; i < word.length(); i++) {
            String next = word.substring(0, i) + word.substring(i + 1);
            if (candidates.contains(next)){
                int nextDist = getDist(map, distMap, next);
                max = Math.max(max, nextDist + 1);
            }
        }
        distMap.put(word, max);
        return max;
    }
}
