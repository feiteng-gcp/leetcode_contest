class Solution {
    class pair {
        String word;
        int length;
        pair(String word, int length) {
            this.word = word;
            this.length = length;
        }
    }
    
    public int longestStrChain(String[] words) {
        pair[] w = new pair[words.length];
        for (int i = 0; i < words.length; i++) {
            w[i] = new pair(words[i], words[i].length());
        }
        
        Arrays.sort(w, new Comparator<pair>() {
            public int compare(pair a, pair b) {
                return a.length - b.length;
            }
        });
        
        
        
int[] dp = new int[words.length];
        Arrays.fill(dp, 1);
        
        for (int i = w.length - 1; i >= 0; i--) {
            for (int j = w.length - 1; j > i; j--) {
                if (w[i].word.length() + 1 == w[j].word.length() && judge(w[i].word, w[j].word)) {
                    dp[i] = Math.max(dp[i], dp[j] + 1);
                }
            }
        }
        
        int max = 1;
        for (int i : dp) {
            System.out.print(i);
            System.out.print(" ");
            
max = Math.max(max, i);
        }
        
        return max;
    }
    
    // a.length < b.length
    private boolean judge(String a, String b) {
        int cnt = 0;
        int p1 = 0, p2 = 0;

        while (p1 < a.length() && p2 < b.length()) {
            if (a.charAt(p1) == b.charAt(p2)) {
                p1++;    p2++;
            } else {
                cnt++;
                p2++;
                if (cnt > 1)    return false;
            }
        }

        return true;
    }
}
