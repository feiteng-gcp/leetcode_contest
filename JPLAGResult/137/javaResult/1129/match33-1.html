class Solution {
    public int longestStrChain(String[] words) {
        HashMap<Integer, ArrayList<String>> map = new HashMap<Integer, ArrayList<String>>();
        HashMap<String, Integer> length = new HashMap<String, Integer>();
        int max = 0;
        for(String word: words) {
            max = Math.max(max, word.length());
            if(map.containsKey(word.length())) {
                ArrayList<String> list = map.get(word.length());
                list.add(word);
            } else {
                ArrayList<String> list = new ArrayList<String>();
                list.add(word);
                map.put(word.length(), list);
            }
            length.put(word, 1);
        }
        
        int ans = 1;
        for(int i = 2; i <= max; i++) {
            if(map.containsKey(i) && map.containsKey(i-1)) {
                ArrayList<String> longer = map.get(i);
                ArrayList<String> shorter = 
map.get(i-1);
                for(String lon : longer) {
                    for(String shor: shorter) {
                        if(isEditDistanceOne(lon, shor)) {
                            if((length.get(shor) + 1) > length.get(lon))
                                length.put(lon, length.get(shor) + 1);
                            ans = Math.max(ans, length.get(shor) + 1);
                            //System.out.println(shor + " "+lon);
                            //System.out.println(length.get(lon));
                        }
                    }
                }
            }
        
}
        return ans;
    }
    
    private static boolean isEditDistanceOne(String s1, String s2) 
    { 
        // Find lengths of given strings 
        int m = s1.length(), n = s2.length(); 

        // If difference between lengths is  
        // more than 1, then strings can't  
        // be at one distance 
        if (Math.abs(m - n) > 1) 
            return false; 

        int count = 0; // Count of edits 

        int i = 0, j = 0; 
        while (i < m && j < n) 
        { 
            // If current characters don't match 
            if (s1.charAt(i) != s2.charAt(j)) 
            { 
                if (count == 1) 
                    return false; 

                // If length of one string is 
                // more, then only possible edit 
                // is to remove a character 
                if (m > n) 
                    i++; 
                else if (m< n) 
                    j++; 
                else // Iflengths of both strings 
                    // is same 
                { 
                    i++; 
                    j++; 
                } 

                // Increment count of edits  
                count++; 
            } 

            else // If current characters match 
            { 
                i++; 
                j++; 
            } 
        } 

        // If last character is extra  
        // in any string 
        if (i < m || j < n) 
            count++; 

        return count == 1; 
    } 
}
