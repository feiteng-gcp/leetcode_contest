class Solution {
    public int lastStoneWeight(int[] stones) {
        if(stones.length == 0) return 0;
        PriorityQueue<Integer> q = new PriorityQueue<>(stones.length, new Comparator<Integer>(){
            public int compare(Integer a, Integer b) {
                return b - a;
            }
        });
        for(int k : stones)
            q.add(k);
        while(q.size() > 1) {
            int l = q.poll();
            int r = q.poll();
            if(l != r)
                q.offer(l - r);
        }
        return q.isEmpty() ? 0 : q.poll();
    }
}
