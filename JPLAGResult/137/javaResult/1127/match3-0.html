class Solution {
    public int lastStoneWeight(int[] stones) {
        PriorityQueue<Integer> pq=new PriorityQueue<>((Integer a, Integer b) -> b-a);
        for(int s:stones){
            pq.add(s);
        }
        while(pq.size()>1){
            int f=pq.poll();
            int s=pq.poll();
            int r=f-s;
            if(f-s!=0)
                pq.add(r);
        }
        return  pq.isEmpty()?0:pq.peek();
    }
}
