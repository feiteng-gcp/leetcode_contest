class Solution {
    public int lastStoneWeight(int[] stones) {
        final int L = stones.length;
		 PriorityQueue<Integer> pq = new PriorityQueue<Integer> (L, Collections.reverseOrder());
		 for (int s:stones)
			 pq.offer(s);
		 
		 while (pq.size() > 1) {
			 int y = pq.poll();
			 int x = pq.poll();
			 if (x != y)
				 pq.offer(y-x);
		 }
		 
		 if (pq.size() == 1)
			 return pq.poll();
		 return 0;
    }
}
