class Solution {
    public int lastStoneWeight(int[] stones) {
        PriorityQueue<Integer> q = new PriorityQueue<>((a,b)->(b-a));
        for(int s:stones){
            q.offer(s);
        }
        while(q.size()>1){
            int a = q.poll();
            int b = q.poll();
            if(a-b>0){
                q.offer(a-b);
            }
        }
        return q.size()==1 ? q.poll() : 0;
    }
}
