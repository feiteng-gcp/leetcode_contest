class Solution {
    public int lastStoneWeight(int[] stones) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((i1, i2) -> i2 - i1);
        for (int i : stones) {
            pq.add(i);
        }
        
        while (pq.size() > 1) {
            int p1 = pq.poll();
            int p2 = pq.poll();
            if (p1 != p2) {
                pq.add(Math.abs(p1 - p2));
            }
        }
        
        return pq.size() == 0 ? 0 : pq.peek();
    }
}
