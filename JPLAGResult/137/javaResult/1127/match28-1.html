class Solution {
    public int lastStoneWeight(int[] stones) {
        PriorityQueue<Integer> pq = new PriorityQueue<>(new Comparator<Integer>() {
            public int compare(
Integer a, Integer b) {
                return b - a;
            } 
        });
        
        for (int i : stones) {
            pq.offer(i);
        }
        
        while (pq.size() > 1) {
            int a = pq.poll(), b = pq.poll();
            if (a == b)    continue;
            else if (a > b)    pq.offer(a - b);
            else    pq.offer(b - a);
        
}
        
        return (pq.size() == 0) ? 0 : pq.poll();
    }
}
