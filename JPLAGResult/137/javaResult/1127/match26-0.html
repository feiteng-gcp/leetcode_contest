class Solution {
    public int lastStoneWeight(int[] stones) {
        if (stones.length == 0) return 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>((x, y) -> y - x);
        for (int i = 0; i < stones.length; i++) {
            pq.add(stones[i]);
        }
        while (pq.size() > 1) {
            int s1 = pq.poll();
            int s2 = pq.poll();
            if (s1 != s2) {
                pq.add(s1 > s2 ? s1 - s2 : s2 - s1);
            }
        }
        return pq.isEmpty() ? 0 : pq.poll();
    }
}
