class Solution {
    public int lastStoneWeight(int[] stones) {
        Queue<Integer> que=new PriorityQueue<Integer>(new Comparator<Integer>(){
            @Override
            public int compare(Integer i1,Integer i2){
                return i2-i1;
            }
        });
        for(int n:stones){
            que.offer(n);
        }
        while(que.size()>1){
            int s1=que.poll();
            int s2=que.poll();
            if(s1==s2){
                continue;
            }
            
que.offer(Math.abs(s1-s2));
        }
        return que.isEmpty()?0:que.poll();
    }
}
