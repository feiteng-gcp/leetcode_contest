class Solution {
    public String removeDuplicates(String S) {
        ArrayDeque<Character> ad = new ArrayDeque<Character>();
        for (char c : S.toCharArray()) {
            if (!ad.isEmpty() && ad.peekLast()==c)
                ad.removeLast();
            else
                ad.add(c)
;
        }
        StringBuilder ans = new StringBuilder();
        while (!ad.isEmpty())
            ans.append(ad.removeFirst());
        return ans.toString();
    }
}
