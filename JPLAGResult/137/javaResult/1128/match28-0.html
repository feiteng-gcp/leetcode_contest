class Solution {
public String removeDuplicates(String S) {
        LinkedList<Character> list = new LinkedList<>();
        for(int i =0; i<S.length(); i++){
            if(list.size() == 0){
                list.addLast(S.charAt(i));
            }else{
                char tmp = list.getLast();
                if(tmp == S.charAt(i)){
                    list.removeLast();
                }else{
                    list.addLast(S.charAt(i));
                }
            }
        }
        StringBuilder sb = new StringBuilder();
        for(char tmp : list){
            sb.append(tmp);
        }
        return sb.toString();
    }
}
