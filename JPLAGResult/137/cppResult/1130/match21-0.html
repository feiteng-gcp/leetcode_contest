class Solution {
public:
    int lastStoneWeightII(vector<int>& stones) {
        int n = stones.size(), sum = 0, ret;
        for (int i = 0; i < n; i++) sum += stones[i];
        ret = sum;
        vector<int> f;
        f.resize(sum + 1, 0);
        f[0] = 1;
        for (int i = 0; i < n; i++)
            for (int j = sum - stones[i]; j >= 0; j--) f[j + stones[i]] |= f[j];
        for (int i = 0; i <= sum; i++) if (f[i])
            ret = min(ret, abs(sum - i * 2));
        return ret;
    }
};
