class Solution {  
  template<typename T1,typename T2> inline void chmin(T1 &a,T2 b){if(a>b) a=b;}
  template<typename T1,typename T2> inline void chmax(T1 &a,T2 b){if(a<b) a=b;}
public:
  int lastStoneWeightII(vector<int>& stones) {
    vector<int> vs(stones);
    int n=vs.size();
    if(n==1) return vs[0];
    int h=n/2;

    int s1=1<<h;
    
    set<int> ss;    
    for(int b=0;b<s1;b++){
      int res=0;
      for(int i=0;i<h;i++){
        if((b>>i)&1) res+=vs[i];
        else res-=vs[i];
      }
      ss.emplace(res);
    }
    
    int ans=0;
    for(int i=0;i<n;i++) ans+=vs[i];

    int s2=1<<(n-h);

    for(int b=0;b<s2;b++){      
      int res=0;
      for(int i=0;i<n-h;i++){
        if((b>>i)&1) res+=vs[h+i];
        else res-=vs[h+i];
      }
      auto it=ss.lower_bound(res);
      if(it!=ss.end()) chmin(ans,abs(res-*it));
      
      if(it!=ss.end()) ++it;
      if(it!=ss.end()) chmin(ans,abs(res-*it));
      
      if(it!=ss.begin()) --it;
      if(it!=ss.end()) chmin(ans,abs(res-*it));
      if(it!=ss.begin()) --it;
      if(it!=ss.end()) chmin(ans,abs(res-*it));
    }
    return ans;
  }
};
