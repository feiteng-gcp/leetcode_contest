class Solution {
public:
    int lastStoneWeightII(vector<int>& stones) {
        int sum = 0;
        for(auto s: stones) {
            sum += s;
        }
        int total = sum;
        sum = sum / 2;
        vector<int> dp(sum+1, 0);

        for(int i=1;i<=stones.size();++i) {
            for(int j=sum;j>=stones[i-1];--j) {
                dp[j] = max(dp[j], dp[j-stones[i-1]]+stones[i-1]);
            }
        }
        return total-2*dp[sum];
    }
    
    void print(vector<int> & v) {
        for(auto x: v) {
            cout << x << " ";
        }
        cout << endl;
    }
};
