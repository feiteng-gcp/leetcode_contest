class Solution {
public:
    void print(priority_queue<int> pq) {
        while(!pq.empty()) {
            cout << pq.top() << " ";
            pq.pop();
        }
        cout << endl;
    }
    int lastStoneWeightII(vector<int>& s) {
        int W = 103;
        int sum = 0;
        int n = s.size();
        for(int i = 0; i < n; ++i) {
            sum += s[i];
        }
        /*
        priority_queue<int> pq;
        while(!pq.empty()) pq.pop();
        for(int x: stones) {
            pq.push(x);
        }
        while(pq.size() > 1) {
            int a = pq.top(); pq.pop();
            int b = pq.top(); pq.pop();
            pq.push(abs(a - b));
            print(pq);
        }
        return pq.top();
        */

        vector<vector<int>> ans(n + 1, vector<int> (sum + 1, 0));
        for(int i = 1; i <= n; ++i) {
            for(int j = 0; j <= sum; ++j) {
                //cout << 2 * s[i - 1] << " " << j << endl;
                if(2 * s[i - 1] > j) {
                    ans[i][j] = ans[i - 1][j];
                } else {
                    ans[i][j] = max(ans[i - 1][j], ans[i - 1][j - 2 * s[i - 1]] + 2 * s[i - 1]);
                }
                //cout << ans[i][j] << " ";
            }
            //cout << endl;
        }
        return sum - ans[n][sum];
    }
};
