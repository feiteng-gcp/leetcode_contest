class Solution {
public:
    int lastStoneWeightII(vector<int>& a) {
        int n = a.size();
        vector<int> vis(3005);
        int sum = 0;
        for(auto x : a)
            sum += x;
        vis[0] = 1;
        int res = sum;
        for(int x : a) {
            
            for(int i = 3000; i >= 0; i--) {
                if(vis[i]) {
                    vis[i + x] = 1;
                    res = min(res, abs(sum - 2 * (i + x)));
                }
            }
        }
        return res;
    }
};
