class Solution {
public:
    int dp[2][30005];
    int lastStoneWeightII(vector<int>& a) {
        int mx=0;
        int now=0,last=1;
        dp[last][0]=1;
        for (int i=0; i<a.size(); i++) {
            for (int j=0; j<=mx; j++) dp[now][j]=0;
            for (int j=0; j<=mx; j++) {
                dp[now][j+a[i]]|=dp[last][j];
                dp[now][abs(j-a[i])]|=dp[last][j];
            }

            now^=1; last^=1;
            mx+=a[i];
        }
        for (int j=0; j<=mx; j++) if (dp[last][j]) return j;
        return 0;
    }
};
