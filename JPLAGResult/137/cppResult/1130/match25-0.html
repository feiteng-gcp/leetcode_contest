class Solution {
private:
    bool f[30010];
public:
    int lastStoneWeightII(vector<int>& stones) {
        memset(f, false, sizeof(f));
        f[0] = true;
        
        int n = stones.size();
        int sum = 0;
        for (int i = 0; i < n; ++i) {
            sum += stones[i];
        }
        
        for (int i = 0; i < n; ++i) {
            for (int j = sum; j >= stones[i]; --j) {
                f[j] |= f[j - stones[i]];
            }
        }
        
        int ans = sum;
        for (int i = 0; i <= sum; ++i) {
            if (f[i]) {
                ans = min(ans, abs(i - (sum - i)));
            }
        }
        return ans;
    }
};
