class Solution {
public:
    int dp[35][3500];
    int lastStoneWeightII(vector<int>& s) {
        int n = s.size();
        int sum = 0;
        for(auto x: s)

            sum += x;
        for(int i = 1; i <= n; i++) {
            for(int j = 0; j <= sum/2; j++) {
                dp[i][j] = dp[i-1][j];
                if(s[i-1] <= j && dp[i-1][j-s[i-1]] + s[i-1] > dp[i][j])
                    dp[i][j] = dp[i-1][j-s[i-1]] + s[i-1];
               // printf("dp[%d][%d] = %d\n", i, j, dp[i][j]);
            }
        }
        return sum - 2*dp[n][sum/2];
    }
};
