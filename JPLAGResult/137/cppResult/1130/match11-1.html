int fff[3001];
class Solution {
public:
    int lastStoneWeightII(vector<int>& stones) {
        int n=stones.size();
        memset(fff,0,sizeof(fff));
        fff[0]=1;
        int sum=0;
        for(int i=0;i<n;i++)
        {
            sum=sum+stones[i];

            for(int j=sum;j>=stones[i];j--)
            {
                if(fff[j-stones[i]])fff[j]=1;
            }
        }
        int ans=sum;
        for(int i=1;i<sum;i++)
        {
            if(fff[i])ans=min(ans,max(sum-2*i,2*i-sum));
        }
        return ans;
    }
};
