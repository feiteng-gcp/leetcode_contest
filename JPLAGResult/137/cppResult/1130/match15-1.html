class Solution {
public:
    int dp[10005] ;
    int lastStoneWeightII(vector<int>& stones) {
        memset ( dp , 0 , sizeof( dp ) ) ; 
        int sum =0  ;
        for( auto & num : stones ) 

            sum += num ; 
        for( int i=0 ; i< stones.size() ; i++){
            for( int j=sum/2 ; j>= stones[i] ; j-- ){
                dp[j] = max( dp[j] , dp[j-stones[i]] + stones[i] ) ;  
            }
        }
        return sum - dp[sum/2]*2; 
    }
};
