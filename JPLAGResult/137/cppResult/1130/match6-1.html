class Solution {
public:
    int lastStoneWeightII(vector<int>& stones) {
        int sum = 0;
        for (int s: stones)
            sum += s;
        
        int target = sum/2;
        // cout << sum << ' ' << target << endl;
        
        vector<bool> f(target+1, false);

        f[0] = true;
        for (int i = 0; i < stones.size(); ++i) {
            for (int t = target; t >= stones[i]; --t) {
                f[t] = f[t] || f[t - stones[i]];
            }
        }
        
        
        for (int t = target; t >= 0; --t) {
            if (f[t]) {
                return sum - t - t;
            }
        }
        
        return 0;
    }

private:
