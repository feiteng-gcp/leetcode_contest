class Solution {
public:
    int lastStoneWeightII(vector<int>& stones) {
        int sum=0;
        for(int i:stones) sum+=i;
        //sort(stones.begin(),stones.end());
        vector<int> dp(sum+1,0);
        dp[0]=1;
        for(int i:stones)
            for(int j=sum;j>=i*2;j--)
                if(dp[j-i*2]) dp[j]=1;
        for(int j=sum;j>=0;j--)
            if(dp[j]) return sum-j;
        return -1;
    }
};
