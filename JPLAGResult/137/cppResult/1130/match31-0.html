class Solution {
public:
    int lastStoneWeightII(vector<int>& stones) {
        int total = 0;
        for(int x: stones)
            total += x;
        
        int dp[stones.size()+1][total+1];
        for(int i = 0; i <= stones.size(); i++)
            dp[i][0] = 1;
        for (int i = 1; i <= total; i++) 
            dp[0][i] = 0; 
        
        for (int i = 1; i <= stones.size(); i++) { 
            for (int j = 1; j <= total; j++) { 
                // excluded 
                dp[i][j] = dp[i-1][j]; 
                // included 
                if(stones[i-1] <= j) 
                    dp[i][j] |= dp[i-1][j-stones[i-1]]; 
            } 
        } 
        int res = INT_MAX;
        for(int j = total/2; j >= 0; j--) { 
            if(dp[stones.size()][j] == 1) { 
                res = total - 2*j; 
                break; 
            } 
        } 
        return res; 
    }
};
