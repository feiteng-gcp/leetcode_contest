class Solution {
public:
    int lastStoneWeight(vector<int>& v) {
        while (v.size() > 1) {
            sort(v.begin(), v.end());
            int x1 = v.back(), x2 = v[v.size() - 2];
            v.pop_back();
            v.pop_back();
            if (x1 != x2) v.push_back(x1 - x2);
        }
        
        return v.size() == 1 ? v[0] : 0;
    }
};
