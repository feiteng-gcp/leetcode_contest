class Solution {
public:
    int lastStoneWeight(vector<int>& stones)
    {
        priority_queue<int> queue(stones.begin(), stones.end());
        while (queue.size() >= 2) {
            int y = queue.top();
            queue.pop();
            int x = queue.top();
            queue.pop();
            if (y > x) queue.push(y - x);
        }
        return queue.size() > 0 ? queue.top() : 0;
    }
};
