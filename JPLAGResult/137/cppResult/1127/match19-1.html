class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        priority_queue<int> q;
        for(int &s: stones){
            q.push(s);
        }
        while(q.size() > 1){
            int x = q.top(); q.pop();
            int y = q.top(); q.pop();
            if(x == y) continue;
            q.push(abs(x - y));
        }
        return q.size() == 0 ? 0 : q.top();
    }
};
