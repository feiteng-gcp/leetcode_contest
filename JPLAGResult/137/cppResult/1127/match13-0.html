class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        multiset<int>s;
        for(int i:stones){
            s.insert(-i);
        }
        while(s.size()>1){
            int i,j;
            i=*s.begin();
            s.erase(s.begin());
            j=*s.begin();
            s.erase(s.begin());
            if(i!=j){
                s.insert(-max(i,j) + min(i,j));
            }
        }
        if(s.size()==0){
            return 0;
        }
        return -*s.begin();
    }
};
