class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        while (stones.size() > 1) {
            sort(stones.begin(), stones.end());
            reverse(stones.begin(), stones.end());
            int x = stones[1], y = stones[0];
            stones.erase(stones.begin());
            stones.erase(stones.begin());
            if (x != y) stones.push_back(y - x);
        }
        if (stones.empty()) return 0;
        return stones[0];
    }
};
