class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        priority_queue<int> qu;
        for(int i:stones) qu.push(i);
        while(qu.size()>1)
        {
            int u=qu.top(); qu.pop();
            int v=qu.top(); qu.pop();
            if(u!=v) qu.push(u-v);
        }
        return qu.empty()?0:qu.top();
    }
};
