class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        while(stones.size()>1){
            sort(stones.begin(), stones.end());
            int x = stones.back(); stones.pop_back();
            int y = stones.back(); stones.pop_back();
            if(abs(x-y)>0){
                stones.push_back(abs(x-y));
            }
            
        }
        if(stones.size()==0)
            return 0;
        else
            return stones[0];
    }
};
