class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        while (stones.size() > 1) {
            sort(stones.begin(), stones.end());
            int n = stones.size();
            int a = stones[n - 2], b = stones[n - 1];
            stones.resize(n - 2);
            stones.push_back(b - a);
        }

        return stones.back();
    }
};
