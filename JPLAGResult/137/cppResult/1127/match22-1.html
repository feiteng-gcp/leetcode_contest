class Solution {
public:
    int lastStoneWeight(vector<int>& a) {
        priority_queue<int> Q;
        for (auto x : a) Q.push(x);
        while (Q.size() >= 2)
        {
            int x = Q.top(); Q.pop();
            int y = Q.top(); Q.pop();
            if (x != y) Q.push(abs(x-y));
        }
        if (Q.empty()) return 0; else return Q.top();
    }
};
