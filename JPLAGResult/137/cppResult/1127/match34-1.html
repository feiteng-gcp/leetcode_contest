class Solution {
public:
    int lastStoneWeight(vector<int>& a) {
        while (a.size() > 1) {
            sort(a.begin(), a.end());
            int n = a.size();
            int k1 = a[n - 1], k2 = a[n - 2];
            a.pop_back();
            a.pop_back();
            if (k2 < k1) {
                a.push_back(k1 - k2);
            }
        }
        if (a.size() == 0) return 0;
        return a[0];
    }
};
