class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
	priority_queue<int> Q(stones.begin(), stones.end());
	while (Q.size() > 1u) {
	    int y = Q.top(); Q.pop();
	    int x = Q.top(); Q.pop();
	    if (x != y) {
		Q.push(y - x);
	    }
	}
        
	if (Q.empty()) return 0;
	else return Q.top();
    }
};
