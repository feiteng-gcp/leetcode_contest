class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
        sort(stones.begin(), stones.end());
        while (stones.size() > 1) {
            int a = stones[stones.size() - 1], b = stones[stones.size() - 2];
            stones.pop_back();
            stones.pop_back();
            stones.insert(lower_bound(stones.begin(), stones.end(), a - b), a - b);
        }
        return stones[0];
    }
};
