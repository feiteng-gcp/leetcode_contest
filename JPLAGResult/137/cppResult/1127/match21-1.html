class Solution {
public:
    int lastStoneWeight(vector<int>& stones) {
//         map<int, int, greater<int>> weightCount;
//         for (int w: stones)
//             ++weightCount[w];
        
//         while (weightCount.size() > 1) {
//             auto iter = weightCount.begin();
//             int large = iter->first;
//             if (--iter->second) {
//                 weightCount.erase();
//             }
//         }
        priority_queue<int> pq;
        for (int w: stones)
            pq.push(w);
        
        while (pq.size() > 1) {
            int large = pq.top();
            pq.pop();
            int small = pq.top();
            pq.pop();
            if (large > small) {
                pq.push(large - small);
            }
        }
        
        return pq.empty() ? 0 : pq.top();
    }
};
