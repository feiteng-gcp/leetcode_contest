class Solution {
public:
    bool helper(string w1, string w2) {
        if (w1.size() + 1 != w2.size()) {return false;}
        int p = 0, q = 0, change = 0;
        while (p < w2.size()) {
            if (w1[p] == w2[q]) {
                ++p, ++q;
            } else if (change == 0) {
                change = 1;
                ++q;
            } else {
                return false;
            }

        }
        return true;
    }
    
    int longestStrChain(vector<string>& words) {
        sort(words.begin(), words.end(), [](string s1, string s2) {
            return s1.size() < s2.size();
        });
        int n = words.size();
        vector<int> dp(n, 1);
        int res = 1;
        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < i; ++j) {
                if (helper(words[j], words[i])) {
                    dp[i] = max(dp[i], dp[j]+1);
                }
            }
            res = max(res, dp[i]);
        }
        return res;
    }
};
