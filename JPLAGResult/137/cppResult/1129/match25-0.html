class Solution {
public:
    int longestStrChain(vector<string>& words) {
        sort(words.begin(),words.end(), cmp);
        vector<int> res(words.size(), 1);
        int max_res=1;
        for(int i=0;i<words.size();++i) {
            for(int j=0;j<i;++j) {
                if(isOk(words[i], words[j])) {
                    res[i] = max(res[j]+1, res[i]);
                    max_res = max(max_res, res[i]);
                }
            }
        }
        return max_res;
    }
    
    static bool cmp(const string & x, const string & y) {
        return x.size() < y.size();
    }
    
    bool isOk(string a, string b) {
        if(a.size() != b.size()+1) return false;
        int i=0, j=0;
        while(j<b.size()) {
            if(a[i] != b[j]) {
                if(i == j) ++i;
                else return false;
            } else {
                ++i;
                ++j;
            }
        }
        return true;
    }
};
