int g[1000][1000],len[1000],dp[1000];
class Solution {
public:
    int dfs(int v,int n)
    {
        if(dp[v])return dp[v];
        dp[v]=1;
        for(int i=0;i<n;i++)
        {
            if(g[v][i]==1)dp[v]=max(dp[v],dfs(i,n)+1);
        }
        return dp[v];
    }
    int longestStrChain(vector<string>& words) {
        int n=words.size();
        for(int i=0;i<n;i++)
        {
            len[i]=words[i].size();
        }
        memset(g,0,sizeof(g));
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(len[i]-len[j]!=1&&len[j]-len[i]!=1)continue;
                int li=0,lj=0,ri=len[i]-1,rj=len[j]-1;
                while(li<len[i]&&lj<len[j])
                {
                    if(words[i][li]!=words[j][lj])break;
                    li++;
                    lj++;
                }
                while(ri>=0&&rj>=0)
                {
                    if(words[i][ri]!=words[j][rj])break;
                    ri--;
                    rj--;
                }
                if(ri-li<=0&&rj-lj<=0)
                {
                    if(ri-li>rj-lj)g[i][j]=1;
                    else if(ri-li<rj-lj)g[j][i]=1;
                }
            }
        }
        memset(dp,0,sizeof(dp));
        for(int i=0;i<n;i++)
        {
            if(!dp[i])dp[i]=dfs(i,n);
        }
        int ans=0;
        for(int i=0;i<n;i++)
        {
            ans=max(ans,dp[i]);
        }
        return ans;
    }
};
