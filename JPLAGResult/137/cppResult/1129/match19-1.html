class Solution {
    bool test(string& s, string& t) {
        if (s.size() != t.size() + 1) return false;
        int idx = 0;
        int state = 0;
        for (int i=0; i<s.size(); i++) {
            if (s[i] == t[idx]) {
                idx++;
            } else if (state == 0) {
                state++;
            } else {
                return false;
            }
        }
        return state == 1;
    }
public:
    int longestStrChain(vector<string>& words) {
        int dp[1005] = {0};
        int N = words.size();
        sort(words.begin(), words.end(),
             [](const string& s, const string& t) -> int {
                return s.size() < t.size();
             });

        int ret = 0;
        for (int i=0; i<N; i++) {
            dp[i] = 1;
            for (int j=0; j<i; j++) {
                if (test(words[i], words[j])) {
                    dp[i] = max(dp[i], dp[j] + 1);
                }
            }
            ret = max(ret, dp[i]);
        }
        return ret;
    }
};
