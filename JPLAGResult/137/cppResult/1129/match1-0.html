class Solution {
public:
    int dp[1005] ; 
    bool isfather( string & a , string & b) {
        if (a.size() != b.size()-1 ) 
            return false; 
        int ptr = 0 ; 
        for( auto & c : b ) 
            if( ptr<a.size() && c == a[ptr] ) 
                ptr++;
        return ptr == a.size() ; 
    }
    int longestStrChain(vector<string>& words) {

        sort( words.begin(),words.end() , [](string & a , string & b ) {return a.size()<b.size();});
        memset( dp , 0 , sizeof( dp ) ) ; 
        dp[0] = 1;  
        int ans = 0 ; 
        for( int i=1 ; i<words.size() ; i++){
            dp[i] = 1; 
            for( int j=0 ; j<i ; j++){
                if( isfather( words[j] , words[i] ) ) 
                    dp[i] = max( dp[i] ,  dp[j]+1 ) ; 
            }
            ans = max( ans , dp[i] ) ; 
        }
        return ans ; 
    }
};
