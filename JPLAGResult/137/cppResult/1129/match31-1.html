class Solution {
public:
    int longestStrChain(vector<string>& words) {
        sort(words.begin(),words.end(),[](string& w1, string& w2){return w1.size()<w2.size();});
        int n = words.size();
        vector<int> dp(n,1);
        int b = 0, e = 0, res = 1;
        for(int i = 0; i < n; i++) {
            if(words[i].size() == words[b].size()) continue;
            if(words[i].size() > words[e].size() && words[i-1].size() == words[e].size()) {
                b = e;
                e = i;
                // cout << "b: "<<b << " e: "<<e<<endl;
            }

            for(int j = b; j < e; j++) {
                if(neighbor(words[j],words[i])){
                    // cout << i << " "<< j << endl;
                    dp[i] = max(dp[i],dp[j]+1);
                    res = max(res,dp[i]);
                }
            }
        }
        return res;
    }
    
    bool neighbor(string& w1, string& w2) {
        if(w1.size()+1 != w2.size()) return false;
        int index = 0, d = 0;
        while(index < w1.size()) {
            if(w1[index] == w2[index+d])
                index++;
            else{
                d++;
                if(d > 1)
                    return false;
            }
        }
        return true;
    }
};
