class Solution {
public:
    int deg[1005],ord[1005],dp[1005];
    vector<int> g[1005];
    bool check(string s,string t) {
        int j=0;
        for (int i=0; i<t.size(); i++) {
            if (j<s.size() && s[j]==t[i]) ++j;
        }
        return j==s.size();
    }
    int longestStrChain(vector<string>& a) {
        int n=a.size();
        for (int i=0; i<n; i++) {
            for (int j=i+1; j<n; j++) {
                if (a[i].size()+1==a[j].size() && check(a[i],a[j])) g[i].push_back(j),++deg[j];
                else if (a[i].size()==a[j].size()+1 && check(a[j],a[i])) g[j].push_back(i),++deg[i];
            }
        }
        int cnt=0;
        for (int i=0; i<n; i++) {
            if (deg[i]==0) {
                ord[++cnt]=i;
            }
        }
        for (int i=1; i<=cnt; i++) {
            int u=ord[i];
            ++dp[u];
            for (int j=0; j<g[u].size(); j++) {
                int v=g[u][j];
                --deg[v];
                if (deg[v]==0) ord[++cnt]=v;
                dp[v]=max(dp[u],dp[v]);
            }
        }
        int ans=0;
        for (int i=0; i<n; i++) ans=max(ans,dp[i]);
        return ans;
    }
};
