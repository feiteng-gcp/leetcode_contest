class Solution {  
  template<typename T1,typename T2> inline void chmin(T1 &a,T2 b){if(a>b) a=b;}
  template<typename T1,typename T2> inline void chmax(T1 &a,T2 b){if(a<b) a=b;}
public:
  int longestStrChain(vector<string>& words) {
    vector<string> vs(words);
    int n=vs.size();
    {
      using P = pair<int, string>;
      vector<P> vp;
      for(int i=0;i<n;i++) vp.emplace_back((int)vs[i].size(),vs[i]);
      sort(vp.rbegin(),vp.rend());
      vp.erase(unique(vp.begin(),vp.end()),vp.end());
      n=vp.size();
      vs.resize(n);
      for(int i=0;i<n;i++) vs[i]=vp[i].second;
    }
    
    map<string, int> idx;
    for(int i=0;i<n;i++) idx[vs[i]]=i;

    vector<int> dp(n,1);
    for(int i=0;i<n;i++){
      int l=vs[i].size();
      for(int j=0;j<l;j++){
        string t=vs[i].substr(0,j)+vs[i].substr(j+1);        
        //cout<<vs[i]<<":"<<t<<endl;
        if(!idx.count(t)) continue;
        chmax(dp[idx[t]],dp[i]+1);
      }
    }    
    return *max_element(dp.begin(),dp.end());
  }
};
