class Solution {
public:
    int longestStrChain(vector<string>& words) {
        sort(words.begin(), words.end(), [](const string &lhs, const string &rhs) {
           return lhs.size() < rhs.size(); 
        });
        int n = words.size();
        vector<int> dp(n, 1);
        
        for (int i = 0; i < n; i++) {
            int j = i + 1;
            for (; j < n && words[j].size() != words[i].size() + 1; j++);
            for (; j < n && words[j].size() == words[i].size() + 1 ; j++) {
                if (diff(words[j], words[i]) == 1) dp[j] = max(dp[j], dp[i] + 1);
            }
        }
        
        int ans = 0;
        for (int i = 0; i < n; i++) {
            ans = max(ans, dp[i]);
        }
        return ans;
    }
    
    int diff(const string &s, const string &t) {
        int d = 0, j = 0;
        for (int i = 0; i < s.size(); i++) {
            if (j == t.size() || s[i] != t[j]) d++;
            else j++;
        }
        return d;
    }
};
