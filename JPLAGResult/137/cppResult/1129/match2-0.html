class Solution {
public:
    vector<vector<int> > g;
    int ispre(string s1,string s2)
    {
        int l1=s1.length(),l2=s2.length();
        if(l1+1!=l2) return 0;
        int j=0;
        for(int i=0;i<l2;i++)
        {
            if(s2[i]==s1[j]) j++;
            if(j==l1) return 1;
        }
        return 0;
    }
    int longestStrChain(vector<string>& words) {
        sort(words.begin(),words.end(),[](string &a,string &b){return a.length()<b.length();});
        int n=words.size();
        vector<int> dp(n+1,0);
        int ans=0;
        for(int i=0;i<n;i++)
        {
            dp[i]=1;
            for(int j=0;j<i;j++)
            {
                if(ispre(words[j],words[i])) dp[i]=max(dp[i],dp[j]+1);
            }
            ans=max(ans,dp[i]);
        }
        return ans;
    }
};
