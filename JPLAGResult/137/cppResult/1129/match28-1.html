class Solution {
public:
    static bool cmp(string &x, string &y)
    {
        return x.length() < y.length();
    }
    
    int longestStrChain(vector<string>& words) {
        sort(words.begin(), words.end(), cmp); 

        int n = words.size(), ret = 0;
        vector<int> f;
        f.resize(n, 1);
        for (int i = 1; i < n; i++)
        {
            int len = words[i].length();
            for (int j = 0; j < i; j++)
            {
                int l = words[j].length();
                if (l + 1 < len) continue;
                if (l + 1 > len || f[j] + 1 <= f[i]) break;
                int flag = 0;
                for (int k = 0; k < l; k++) if (words[j][k] != words[i][k + flag])
                {
                    flag++;
                    if (flag > 1) break;
                    if (words[j][k] != words[i][k + flag])
                    {
                        flag = 2;
                        break;
                    }
                }
                if (flag > 1) continue;
                f[i] = f[j] + 1;
            }
            ret = max(ret, f[i]);
        }
        return ret;
    }
};
