bool cmp(string s1, string s2) {
	return s1.length() < s2.length();
}

bool ok(string s1, string s2) {
	if (s1.length() == s2.length() - 1) {
		int i = 0;
		int j = 0;
		bool once = false;
		for (; i < s1.length() && j < s2.length();) {
			if (s1[i] == s2[j]) {
				i++;
				j++;
			}
			else {
				if (!once) {
					j++;
					once = true;
				}
				else {
					break;
				}
			}
		}
		return i == s1.length();
	}
	else {
		return false;
	}
}

class Solution {
public:
	int dp[1010];
	int longestStrChain(vector<string>& words) {
		sort(words.begin(), words.end(), cmp);
		int ret = 0;
		for (int i = 0; i < words.size(); i++) {
			dp[i] = 1;
			for (int j = 0; j < i; j++) {
				if (ok(words[j], words[i])) {
					dp[i] = max(dp[i], dp[j] + 1);
				}
			}
			ret = max(ret, dp[i]);
		}
		return ret;
	}
};
