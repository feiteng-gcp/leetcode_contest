class Solution {
public:
    int longestStrChain(vector<string>& words) {
        sort(words.begin(), words.end(), [](const string &a, const string &b) {
            return a.size() < b.size();
        });
        int n = words.size();
        vector<int> dp(n, 1);
        int res = 1;
        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < i; ++j) {
                if (test(words[j], words[i])) {
                    dp[i] = max(dp[i], dp[j] + 1);
                    res = max(res, dp[i]);
                }
            }
        }
        return res;
    }
    bool test(const string &a, const string &b) {
        int n1 = a.size();
        int n2 = b.size();
        if (n1 + 1 != n2)
            return false;
        int i = 0;
        while (i < n1 && a[i] == b[i]) ++i;
        while (i < n1 && a[i] == b[i+1]) ++i;
        return i == n1;
    }
};
