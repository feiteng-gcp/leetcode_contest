class Solution {
private:
    bool helper(string& a, string& b) {
        int i = 0, j = 0, m = a.length(), n = b.length();
        bool found = false;
        while (i < m && j < n) {
            if (a[i] == b[j]) {
                i++;
                j++;
            } else if (!found) {
                j++;
                found = true;
            } else {
                return false;
            }
        }
        return (found && i == m && j == n) || (!found && i == m && j == n - 1);
    }
public:
    int longestStrChain(vector<string>& words) {
        sort(words.begin(), words.end(), [](string& a, string& b){
            return a.length() < b.length();
        });
        int n = words.size(), result = 1;
        vector<int> dp(n, 1);
        for (int i = 1; i < n; i++) {
            for (int j = i - 1; j >= 0; j--) {
                if (dp[j] < dp[i] - 1) continue;
                if (helper(words[j], words[i])) dp[i] = dp[j] + 1;
            }
            result = max(result, dp[i]);
        }
        return result;
    }
};
