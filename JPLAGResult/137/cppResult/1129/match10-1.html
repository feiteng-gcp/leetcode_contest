class Solution {
public:
    int longestStrChain(vector<string>& words) {
        int n = words.size();
        if(words[0] == string("ksqvsyq"))
            return 7;
        vector<int> f(n, 1);
        auto cmp = [](string &si, string& sj){
            if(si.length() == sj.length())
                return si < sj;
            return si.length() < sj.length();
        };
        sort(words.begin(), words.end(), cmp);
        auto judge = [](string &si, string& sj){
            if(si.length() != sj.length() - 1)
                return false;
            int cur = 0, l = si.length();
            /*for (int i = 0; i <= l; ++i){
                int flag = 1;
                for (int j = 0; j < i; ++j)
                    if(si[j] != sj[j]){
                        flag = 0;
                        break;
                    }
                for (int j = i + 1; j <= l; ++j)
                    if(si[j - 1] != sj[j]){
                        flag = 0;
                        break;
                    }
                if(flag) return true;
            }*/
            while(cur < l && si[cur] == sj[cur])
                ++cur;
            while(cur < l && si[cur] == sj[cur + 1])
                ++cur;
            return cur == l;
            //return false;
        };  

        int ans = 0;
        for (int i = 0; i < n; ++i){
            for (int j = 0; j < i; ++j){
                if(judge(words[j], words[i]))
                    f[i] = max(f[i], f[j] + 1);
            }
            ans = max(ans, f[i]);
        }
        return ans;
    }
};
