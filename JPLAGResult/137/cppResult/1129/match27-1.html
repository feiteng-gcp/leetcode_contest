bool cmp(const string &a, const string &b) {
    return a.size() < b.size();
}
class Solution {
    bool isP(const string &a, const string &b) {
        int bi = 0, cnt = 1;
        for (int i = 0; i < a.size(); i++) {
            if (bi < b.size() && a[i] != b[bi] && cnt == 1)
            {
                bi++;
                cnt--;
            }
            if (bi == b.size() || b[bi] != a[i]) return false;
            bi++;
        }
        return bi == b.size() && cnt == 0 || bi == b.size() - 1 && cnt == 1;
        
    }
public:
    int longestStrChain(vector<string>& words) {
        sort(words.begin(), words.end(), cmp);
        int N = words.size();
        vector<int> dp(N);
        int maxA = 0;

        for (int i = 0; i < N; i++) {
            dp[i] = 1;
            maxA = max(1, maxA);
            for (int j = 0; j < i; j++) {
                if (isP(words[j], words[i])) {
                    dp[i] = max(dp[i], dp[j] + 1);
                    maxA = max(maxA, dp[i]);
                }
            }
        }
        for (int i = 0; i < N; i++) printf("%s %d %d\n", words[i].c_str(), words[i].size(), dp[i]);
        return maxA;
    }
};
