class Solution {
private:
    int head[1010];
    int edge[1000010], nxt[1000010];
    int f[1010];
    int ecnt = 0;
    
public:
    void addedge(int x, int y) {
        ++ecnt;
        edge[ecnt] = y;
        nxt[ecnt] = head[x];
        head[x] = ecnt;
    }
    
    bool judge(const string& s1, const string& s2) {
        int pos = 0;
        for (int i = 0; i < s2.size(); ++i) {
            while (pos < s1.size() && s2[i] != s1[pos]) {
                ++pos;
            }
            if (pos == s1.size()) {
                return false;
            }
            ++pos;
        }
        return true;
    }
    
    void dfs(int u) {
        if (f[u]) {
            return;
        }
        for (int i = head[u]; i; i = nxt[i]) {
            int v = edge[i];
            dfs(v);
            f[u] = max(f[u], f[v]);
        }
        ++f[u];
    }
    
    int longestStrChain(vector<string>& words) {
        memset(head, 0, sizeof(head));
        memset(edge, 0, sizeof(edge));
        memset(nxt, 0, sizeof(nxt));
        
        int n = words.size();
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (i != j and words[i].size() == words[j].size() + 1) {
                    if (judge(words[i], words[j])) {
                        // cout << words[i] << " " << words[j] << "\n";
                        addedge(i + 1, j + 1);
                    }
                }
            }
        }
        
        memset(f, 0, sizeof(f));
        for (int i = 1; i <= n; ++i) {
            if (!f[i]) {
                dfs(i);
            }
        }
        
        int ans = 0;
        for (int i = 1; i <= n; ++i) {
            ans = max(ans, f[i]);
        }
        return ans;
    }
};
