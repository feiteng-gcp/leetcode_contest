class Solution {
    bool isPre(const string& s1, const string& s2) {
        int n1 = s1.size();
        int n2 = s2.size();
        if (n1 + 1 != n2) {
            return false;
        }
        int i = 0;
        int j = 0;
        for (i = 0; i < n1; i++) {
            while (j < n2 && s2[j] != s1[i]) {
                j++;
            }
            if (j == n2) {
                return false;
            }
            j++;
        }
        return true;
    }
    vector<int> D;
    int depth(vector<vector<int>>& G, int x) {
        if (D[x] != -1) {
            return D[x];
        } else if (G[x].size() == 0) {
            return D[x] = 1;
        } else {
            int maxdepth = 0;
            for (int i : G[x]) {
                maxdepth = max(maxdepth, depth(G, i));
            }
            return D[x] = maxdepth + 1;
        }

    }
public:
    int longestStrChain(vector<string>& words) {
        int n = words.size();
        vector<vector<int>> G(n);
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (isPre(words[i], words[j])) {
                    G[i].push_back(j);
                } else if (isPre(words[j], words[i])) {
                    G[j].push_back(i);
                }
            }
        }

        D = vector<int>(n, -1);
        int ret = 1;
        for (int i = 0; i < n; i++) {
            ret = max(ret, depth(G, i));
        }
        return ret;
    }
};
