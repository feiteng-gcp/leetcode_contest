class Solution {
public:
    bool judge(string& s1,string& s2){
        int cnt = 0;
        for(int i = 0,j=0;i<s1.size();i++){
            if(s1[i]==s2[j]){
                j++;
            }else{
                i--;
                j++;
                cnt++;
            }
            if(cnt>=2){
                return false;

            }
        }
        return true;
    }
    
    int longestStrChain(vector<string>& words) {
        sort(words.begin(), words.end(), [](const string& w1,const string& w2)->bool{
            return w1.length()<w2.length();
        });
        int sz = words.back().size();
        int n = words.size();
        vector<vector<string>> v(sz);
        for(int i = 0;i<n;i++){
            v[words[i].size()-1].push_back(words[i]);
        }
        int ans = 0;
        unordered_map<string,int> dp;
        for(int i = words[0].size();i<v.size();i++){
            for(int j = 0;j<v[i].size();j++){
                for(int k = 0;k<v[i-1].size();k++){
                    if(judge(v[i-1][k], v[i][j])){
                        dp[v[i][j]]=max(dp[v[i][j]],dp[v[i-1][k]]+1);
                        ans = max(ans,dp[v[i][j]]);
                    }
                }
            }
        }
        return ans+1;
    }
};
