class Solution {
    bool isT(string& a,string& b){
        if(a.size()+1!=b.size()) return false;
        bool flag=false;
        for(int i=0;i<b.size();i++){
            if(flag){
                if(b[i]!=a[i-1]) return false;
            }else{
                if(i==b.size()-1) continue;
                if(b[i]!=a[i]){
                    flag=true;
                }
            }
        }
        return true;
    }
public:
    int longestStrChain(vector<string>& words) {
        sort(words.begin(),words.end(),[](string a,string b){
            return a.size()<b.size();
        });
        
        int N=words.size();
        vector<int> res(N,0);
        int ans=1;
        for(int i=N-1;i>=0;i--){
            int r=1;
            for(int j=i+1;j<N;j++){
                if(words[j].size()>words[i].size()+1) break;
                if(isT(words[i],words[j])){
                    r=max(res[j]+1,r);
                }
            }
            res[i]=r;
            ans=max(r,ans);
        }
        return ans;
    }
};
