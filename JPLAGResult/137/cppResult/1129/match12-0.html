class Solution {
    bool isProd(string& s1, string& s2) {
        int i = 0;
        int j = 0;
        if (s2.length() != s1.length() + 1) return false;
        int m = s1.length();
        int n = s2.length();
        int diff = 0;
        while (i < m && j < n) {
            if (s1[i] == s2[j]) {
                i++;
                j++;
                continue;
            }
            else {
                j++;
                diff++;
                if (diff > 1) return false;
            }
        }
        return true;
    }
    int maxPath(vector<vector<int>>& dp, vector<int>& mem, int start) {
        if (mem[start] != 0) return mem[start];
        int res = 0;
        for (int i = 0; i < dp[start].size(); ++i) {
            int temp = maxPath(dp, mem, dp[start][i]);
            res = max(res, 1 + temp);
        }
        mem[start] = res;
        return res;
    }
public:
    int longestStrChain(vector<string>& words) {
        int n = words.size();
        vector<vector<int>> dp(n);
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (i == j) continue;
                if (isProd(words[i], words[j])) dp[i].push_back(j);
            }
        }
        vector<int> mem(n, 0);
        int res = 0;
        for (int i = 0; i < n; ++i) {
            res = max(1 + maxPath(dp, mem, i), res);
        }
        return res;
    }
};
