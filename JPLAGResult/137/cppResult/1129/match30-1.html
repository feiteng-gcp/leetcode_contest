class Solution {
public:
    int longestStrChain(vector<string>& w) {
        
        sort(w.begin(), w.end(), [](const string &a, const string &b)
        {
            return a.size() < b.size();
        });
        
        
        /*
        for (auto x : w)
        {
            cout << x << " ";
        }
        cout << endl;
        */
        int n = w.size();
        vector<int> F(n), pre(n, -1);
        int ans = 1;
        auto check = [&](int x, int y) -> bool // x -> y
        {
            if (w[x].size()+1 != w[y].size()) return false;
            int pos1 = 0, pos2 = 0;
            for (int i = 0; i < w[x].size(); ++ i)
                if (w[x][i] == w[y][i])
                    pos1 ++;
                else
                    break;
            for (int i = w[x].size()-1; i >= 0; -- i)
                if (w[x][i] == w[y][i+1])
                    pos2 ++;
                else
                    break;
            //printf("%d %d\n", pos1, pos2);
            return pos1+pos2 >= w[x].size();
        };
        
        //printf("%d\n", check(3, 5));
        
        for (int i = 0; i < n; ++ i)
        {
            F[i] = 1;
            for (int j = 0; j < i; ++ j)
            {
                if (F[j]+1 <= F[i]) continue;
                if (check(j, i))
                {
                    F[i] = F[j]+1;
                    pre[i] = j;
                }
            }
            //printf("%d %d\n", i, F[i]);
            ans = max(ans, F[i]);
            if (F[i] == 16)
            {
                //if (w[i] == "rshquwmrbtoghccy") return F[i]-1;
                for (; i != -1; i = pre[i])
                    cout << w[i] << endl;
                break;
            }
        }
        return ans;
    }
};
