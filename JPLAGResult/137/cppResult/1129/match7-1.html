class Solution {
public:
    int longestStrChain(vector<string>& W) {
        sort(W.begin(), W.end(), [](auto &a, auto &b) { return a.size() < b.size() || a.size() == b.size() && a < b; });
        vector<vector<bool>> P(W.size(), vector<bool>(W.size())); 
        for (int i = 0; i < W.size(); i++) {
            auto s = W[i];
            for (int j = i + 1; j < W.size(); j++) {
                auto t = W[j];
                bool f = false;
                if (t.size() - s.size() == 1) {
                    f = true;
                    int n = 0;
                    for (int x = 0, y = 0; x < s.size();) {
                        if (s[x] == t[y])
                            ++x, ++y;
                        else if (s[x] == t[y + 1]) {
                            ++y;
                            if (++n > 1) {
                                f = false;
                                break;
                            }
                        }
                        else {
                            f = false;
                            break;
                        }
                    }
                }
                P[i][j] = P[j][i] = f;
            }
        }
        
        int mx = 1;
        vector<int> dp(W.size(), 1);

        for (int i = 1; i < W.size(); i++) {
            for (int j = i - 1; j >= 0 && W[j].size() >= W[i].size() - 1; j--) {
                if (P[i][j]) {
                    dp[i] = max(dp[i], dp[j] + 1);
                    mx = max(mx, dp[i]);
                }
            }
        }
        return mx;
    }
};
