class Solution {
public:
    string removeDuplicates(string S) {
        stack<char> st;
        for (char c : S) {
            if (st.size() && st.top() == c) st.pop();
            else st.push(c);
        }
        
        string ans = "";
        
        while (st.size()) {
            ans += st.top(); st.pop();
        }
        reverse(ans.begin(), ans.end());
        
        return ans;
    }
};
