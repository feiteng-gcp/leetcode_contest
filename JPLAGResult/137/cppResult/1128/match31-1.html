class Solution {
public:
    string removeDuplicates(string S) {
        stack<char> st;
        for(auto c:S){
            if(st.size()==0){
                st.push(c);
            }else{
                char c1 = st.top();
                if(c==c1){
                    st.pop();
                }else{
                    st.push(c);
                }
            }
        }
        string ret = "";
        while(st.size()>0){
            char c = st.top();
            st.pop();
            ret = c+ret;
        }
        return ret;
    }
};
