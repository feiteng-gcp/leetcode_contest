class Solution {
public:
    string removeDuplicates(string S) {
        stack<char> sta;
        for (int i=0; i<S.size(); i++) {
            if (!sta.empty() && sta.top()==S[i]) sta.pop();
            else sta.push(S[i]);
        }
        S="";
        while (!sta.empty()) S=sta.top()+S,sta.pop();
        return S;
    }
};
