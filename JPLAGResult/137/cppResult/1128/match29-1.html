class Solution {
public:
    string removeDuplicates(string S) {
        string ans;
        for(int i=0;i<S.size();i++)

        {
            if(!ans.empty()&&S[i]==ans.back())
            {
                ans.pop_back();
            }
            else
            {
                ans+=S[i];
            }
        }
        return ans;
    }
};
