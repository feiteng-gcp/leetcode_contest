class Solution {
public:
    string removeDuplicates(string S) {
        stack<char> sc;
        for(auto s: S) {
            if(sc.empty() || sc.top() != s) {
                sc.push(s);
            } else if(sc.top() == s){
                sc.pop();
            }
        }
        string res="";
        while(!sc.empty()) {
            res = sc.top() + res;
            sc.pop();
        }
        return res;
    }
};
