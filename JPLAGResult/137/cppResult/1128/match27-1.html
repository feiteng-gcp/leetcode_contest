class Solution {
public:
    string removeDuplicates(string S) {
        stack<char> stk;
        for (char c: S) {
            bool has = false;
            while (stk.size() > 0 && stk.top() == c) {
                stk.pop();
                has = true;
            }

            if (!has) {
                stk.push(c);
            }
        }
        string s = "";
        while (stk.size() > 0) {
            s = stk.top() + s;
            stk.pop();
        }
        return s;
    }
};
