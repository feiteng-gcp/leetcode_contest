class Solution {
public:
  string removeDuplicates(string S) {
    string result = "";
    for (auto ch : S) {
      if (!result.empty() && result.back() == ch) {
        result.pop_back();
      } else {
        result += ch;
      }
    }
    return result;
  }
};
