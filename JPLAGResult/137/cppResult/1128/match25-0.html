class Solution {
public:
    string removeDuplicates(string S) {
        stack<char> stk;
        int n = S.length();
        for (int i = 0; i < n; i++)
        {
            if (stk.empty() || stk.top() != S[i]) stk.push(S[i]);
            else stk.pop();
        }
        string ret = "";
        while (!stk.empty())
        {
            ret = stk.top() + ret;
            stk.pop();
        }
        return ret;
    }
};
