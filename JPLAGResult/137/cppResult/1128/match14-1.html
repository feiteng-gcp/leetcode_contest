class Solution {
public:
    string removeDuplicates(string S) {
        stack<char> st;
        string ret;

        for (auto c : S) {
            if (st.empty() || st.top() != c) {
                st.push(c);
            } else {
                st.pop();
            }
        }
        while (!st.empty()) {
            ret += st.top();
            st.pop();
        }
        reverse(ret.begin(), ret.end());
        return ret;
    }
};
