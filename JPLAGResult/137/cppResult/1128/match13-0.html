class Solution {
public:
    string removeDuplicates(string S) {
        stack<char> s;
        for (char c : S) {
            if (!s.empty() && c == s.top()) {
                s.pop();
            } else {
                s.push(c);
            }
        }
        string result;
        while (!s.empty()) {
            result = s.top() + result;
            s.pop();
        }
        
        return result;
    }
};
