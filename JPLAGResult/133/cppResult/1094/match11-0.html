class Solution {
public:
    vector<vector<int>> allCellsDistOrder(int R, int C, int r0, int c0) {
        vector<vector<int>> points;
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                points.push_back(vector<int>{i, j});
            }
        }
        sort(points.begin(), points.end(), [r0, c0](const vector<int>& p1, const vector<int>& p2) {
            return abs(p1[0] - r0) + abs(p1[1] - c0) < abs(p2[0] - r0) + abs(p2[1] - c0);
        });
        return points;
    }
};
