class Solution {
public:
    vector<vector<int>> allCellsDistOrder(int R, int C, int r0, int c0) {
        vector<vector<int>> b;
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                b.push_back({i, j});

            }
        }
        
        sort(b.begin(), b.end(), [&](vector<int>& l, vector<int>& r){
            int dist1 = abs(l[0] - r0) + abs(l[1] - c0);
            int dist2 = abs(r[0] - r0) + abs(r[1] - c0);
            return dist1 < dist2;
        });
        
        return b;
    }
};
