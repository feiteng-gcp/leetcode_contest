int dist(int x1, int y1, int x2, int y2) {
    return abs(x1 - x2) + abs(y1 - y2);
}

class Solution {
public:
    vector<vector<int>> allCellsDistOrder(int r, int c, int r0, int c0) {
        vector<vector<int>> v;
        for (int i = 0; i < r; i++)
            for (int j = 0; j < c; j++)
                v.push_back({i, j});
        sort(v.begin(), v.end(), [r0, c0](auto a, auto b) {
            return dist(r0, c0, a[0], a[1]) < dist(r0, c0, b[0], b[1]) ||
                (dist(r0, c0, a[0], a[1]) == dist(r0, c0, b[0], b[1]) && a < b);
        });
        return v;
    }
};
