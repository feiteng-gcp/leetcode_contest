class Solution {
public:
    vector<vector<int>> allCellsDistOrder(int R, int C, int r0, int c0)
    {
        vector<vector<int>> retval;
        
        for(int i=0; i<R; i++)
        {
            for(int j=0; j<C; j++)
            {
                retval.push_back({i,j});
            }

        }
        
        sort(retval.begin(), retval.end(), [r0, c0](const vector<int>& left, const vector<int>& right) {
            int leftDistance = abs(left[0] - r0) + abs(left[1] - c0);
            int rightDistance = abs(right[0] - r0) + abs(right[1] - c0);
            return leftDistance < rightDistance;
        });
        
        return retval;
    }
};
