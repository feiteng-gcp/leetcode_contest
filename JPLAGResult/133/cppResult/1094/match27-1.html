int rr0, cc0;
bool cmp(vector<int> x, vector<int> y) {
    return abs(rr0-x[0]) + abs(cc0-x[1]) < abs(rr0-y[0]) + abs(cc0-y[1]);
}

class Solution {
public:
    vector<vector<int>> allCellsDistOrder(int R, int C, int r0, int c0) {
        rr0 = r0; cc0 = c0;
        vector<vector<int> > S;
        for (int i = 0; i < R; i ++)
            for (int j = 0; j < C; j ++)
                S.push_back({i, j});
        sort(S.begin(), S.end(), cmp);
        return S;
    }
};
