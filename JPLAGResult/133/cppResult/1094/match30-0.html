int _r, _c;
bool cmp(vector<int> v1, vector<int> v2) {
    return abs(v1[0] - _r) + abs(v1[1] - _c) < abs(v2[0] - _r) + abs(v2[1] - _c);
}

class Solution {
public:
    vector<vector<int>> allCellsDistOrder(int r, int c, int r0, int c0) {
        vector<vector<int>> res;
        for(int i = 0; i < r; ++i) for (int j = 0; j < c; ++j) res.push_back({i, j});
        _r = r0;
        _c = c0;
        sort(res.begin(), res.end(), cmp);
        return res;
    }
};
