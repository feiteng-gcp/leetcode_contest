class Solution {
public:
	vector<vector<int>> allCellsDistOrder(int R, int C, int r0, int c0) {
		vector<vector<int>> ret;

		for (int r = 0; r < R; r++) {
			for (int c = 0; c < C; c++) {
				vector<int> v({ r,c });
				ret.emplace_back(v);
			}
		}
		sort(ret.begin(), ret.end(),
             [&](const vector<int> & a, const vector<int> & b) { return abs(r0 - a[0]) + abs(c0 - a[1]) < abs(r0 - b[0]) + abs(c0 - b[1]); });

		return ret;
	}
};
