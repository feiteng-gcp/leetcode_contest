class Solution {
public:
    vector<vector<int>> allCellsDistOrder(int R, int C, int r0, int c0) {
        map<int,vector<vector<int>>> cells;
        for (int i = 0; i < R; ++i) {
            for (int j = 0; j < C; ++j) {

                cells[abs(i-r0)+abs(j-c0)].push_back({i,j});
            }
        }
        vector<vector<int>> out;
        for (auto& c : cells) {
            for (auto item : c.second) {
                out.push_back(item);
            }
        }
        return out;
    }
};
