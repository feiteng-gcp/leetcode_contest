class Solution {
public:
    vector<vector<int>> allCellsDistOrder(int R, int C, int r0, int c0) {
        vector<vector<int>> all;
        for (int i = 0; i < R; i++)
            for (int j = 0; j < C; j++)
                all.push_back({i, j});
        sort(all.begin(), all.end(), [r0, c0](const vector<int>& e, const vector<int>& v) {
            return abs(e[0] - r0) + abs(e[1] - c0) < abs(v[0] - r0) + abs(v[1] - c0);
        });
        return all;
    }
};
