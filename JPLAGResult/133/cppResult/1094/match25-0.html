class Solution {
public:
    vector<vector<int>> allCellsDistOrder(int R, int C, int r0, int c0)
    {
        map<int, vector<vector<int>>> dis;
        for (int i = 0; i < R; ++i)
        {
            for (int j = 0; j < C; ++j)
            {
                int d = abs(i - r0) + abs(j - c0);
                dis[d].push_back(vector<int>({i, j}));
            }
        }
        vector<vector<int>> ret;
        for (auto p : dis)
        {
            for (auto v : p.second)
            {
                ret.push_back(v);
            }
        }
        return ret;
    }
};
