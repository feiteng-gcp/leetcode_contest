class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int n = A.size();
        vector<int> leftL(n), rightL(n + 1), leftM(n), rightM(n + 1), sums(n + 1);
        for (int i = 0; i < n; i++) sums[i + 1] = sums[i] + A[i];
        for (int i = 0; i < n; i++) {
            if (i >= L) {
                leftL[i] = max(leftL[i - 1], sums[i] - sums[i - L]);
            }
            if (i >= M) {
                leftM[i] = max(leftM[i - 1], sums[i] - sums[i - M]);
            }
        }
        for (int i = n - 1; i >= 0; i--) {
            if (i <= n - L) {
                rightL[i] = max(rightL[i + 1], sums[i + L] - sums[i]);
            }
            if (i <= n - M) {
                rightM[i] = max(rightM[i + 1], sums[i + M] - sums[i]);
            }
        }
        int result = 0;
        for (int i = 0; i < n; i++) {
            result = max(result, max(leftL[i] + rightM[i], leftM[i] + rightL[i]));
        }
        return result;
    }
};
