class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int n = A.size();
        int s = 0;
        vector<int> ll(n, -10000000);
        for (int i = 0; i < L; ++i) {
            s += A[i];
        }
        ll[L-1] = s;
        for (int i = L; i < n; ++i) {
            s += A[i] - A[i-L];
            ll[i] = max(ll[i-1], s);
        }
        s = 0;
        vector<int> lm(n, -10000000);
        for (int i = 0; i < M; ++i) {
            s += A[i];
        }
        lm[M-1] = s;
        for (int i = M; i < n; ++i) {
            s += A[i] - A[i-M];
            lm[i] = max(lm[i-1], s);
        }
        s = 0;
        vector<int> rl(n, -10000000);
        for (int i = 0; i < L; ++i) {
            s += A[n-1-i];
        }
        rl[n-L] = s;
        for (int i = n-L-1; i >= 0; --i) {
            s += A[i] - A[i+L];
            rl[i] = max(rl[i+1], s);
        }
        s = 0;
        vector<int> rm(n, -10000000);
        for (int i = 0; i < M; ++i) {
            s += A[n-1-i];
        }
        rm[n-M] = s;
        for (int i = n-M-1; i >= 0; --i) {
            s += A[i] - A[i+M];
            rm[i] = max(rm[i+1], s);
        }
        int res = 0;
        for (int i = 1; i < n-1; ++i) {
            res = max({res, ll[i] + rm[i+1], lm[i] + rl[i+1]});
        }
        return res;
    }
};
