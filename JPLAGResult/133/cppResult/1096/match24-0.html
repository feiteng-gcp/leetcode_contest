class Solution {
public:
  int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
    int N = A.size();
    vector<int> B = {0};
    partial_sum(A.begin(), A.end(), back_inserter(B));
    vector<int> bestL(N + 1, -0x3f3f3f3f);
    vector<int> bestM(N + 1, -0x3f3f3f3f);
    for (int i = 1; i <= N; i++) {
      if (i >= L) bestL[i] = max(bestL[i - 1], B[i] - B[i - L]);
      if (i >= M) bestM[i] = max(bestM[i - 1], B[i] - B[i - M]);
    } 
    int bL = 0;
    int bM = 0;
    int best = 0;
    for (int i = N; i >= 0; i--) {
      if (i + L <= N) {
        bL = max(bL, B[i + L] - B[i]);
        best = max(best, bL + bestM[i]);
      } 
      if (i + M <= N) {
        bM = max(bM, B[i + M] - B[i]);
        best = max(best, bM + bestL[i]);
      } 
    }
    return best;
  }
};
