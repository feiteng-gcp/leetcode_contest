class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        vector<int> l(A.size() - (L - 1)), r(A.size() - (M - 1));

        int sum = 0;
        for (int i = 0; i < L - 1; ++i) {
            sum += A[i];
        }
        for (int i = L - 1; i < A.size(); ++i) {
            sum += A[i];
            l[i - (L - 1)] = sum;
            sum -= A[i - (L - 1)];
        }
        sum = 0;
        for (int i = 0; i < M - 1; ++i) {
            sum += A[i];
        }
        for (int i = M - 1; i < A.size(); ++i) {
            sum += A[i];
            r[i - (M - 1)] = sum;
            sum -= A[i - (M - 1)];
        }
        vector<int> lmx(l.size()), rmx(r.size());
        lmx.front() = l.front();

        for (int i = 1; i < l.size(); ++i) {
            lmx[i] = max(lmx[i - 1], l[i]);
        }
        rmx.back() = r.back();
        for (int i = r.size() - 2; i >= 0; --i) {
            rmx[i] = max(rmx[i + 1], r[i]);
        }
        int ans = 0;
        for (int i = 0; i < lmx.size(); ++i) {
            if (i + L >= rmx.size()) break;
            ans = max(ans, lmx[i] + rmx[i + L]);
        }
        
        fill(lmx.begin(), lmx.end(), 0);
        fill(rmx.begin(), rmx.end(), 0);
        rmx.front() = r.front();

        for (int i = 1; i < r.size(); ++i) {
            rmx[i] = max(rmx[i - 1], r[i]);
        }
        lmx.back() = l.back();
        for (int i = l.size() - 2; i >= 0; --i) {
            lmx[i] = max(lmx[i + 1], l[i]);
        }
        for (int i = 0; i < rmx.size(); ++i) {
            if (i + M >= lmx.size()) break;
            ans = max(ans, rmx[i] + lmx[i + M]);
        }
        return ans;
    }
};
