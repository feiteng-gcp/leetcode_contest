class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        vector<vector<int>> sums(A.size(), vector<int>(A.size() + 1));
        for (int i = 0; i < A.size(); i++) {
            int sum = 0;
            for (int l = 1; i + l <= A.size(); l++) {
                sum += A[i + l - 1];
                sums[i][l] = sum;
            }
        }

        int ans = INT_MIN;
        for (int i = 0; i + L + M <= A.size(); i++) {
            for (int j = i + L; j + M <= A.size(); j++) {
                ans = max(ans, sums[i][L] + sums[j][M]);
            }
        }
        for (int i = 0; i + L + M <= A.size(); i++) {
            for (int j = i + M; j + L <= A.size(); j++) {
                ans = max(ans, sums[i][M] + sums[j][L]);
            }
        }
        return ans;
    }
};
