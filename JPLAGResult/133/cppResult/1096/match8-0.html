class Solution {
public:
    int solve(vector<int>& a, int len1, int len2) {
        int n = a.size();
        vector<int> L(n), R(n);
        int cur = 0;
        for (int i = 0; i < len1; ++i) cur += a[i];
        L[len1 - 1] = cur;
        for (int i = len1; i < n; ++i) {
            cur -= a[i - len1];
            cur += a[i];
            L[i] = max(L[i - 1], cur);
        }
        cur = 0;
        for (int i = 0; i < len2; ++i) cur += a[n - 1 - i];
        R[n - len2] = cur;
        for (int i = n - len2 - 1; i >= 0; --i) {
            cur -= a[i + len2];
            cur += a[i];
            R[i] = max(R[i + 1], cur);
        }
        int best = 0;
        for (int i = len1 - 1; i + 1 <= n - len2; ++i) {
            best = max(best, L[i] + R[i + 1]);
        }
        return best;
    }
    int maxSumTwoNoOverlap(vector<int>& a, int len1, int len2) {
        int ret = solve(a, len1, len2);
        reverse(a.begin(), a.end());
        ret = max(ret, solve(a, len1, len2));
        return ret;
    }
};
