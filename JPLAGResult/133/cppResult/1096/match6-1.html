class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int N = A.size();
        vector<int> sumL(N),sumM(N);
        int sL=0;
        for(int i=0;i<L;i++){
            sL+=A[i];
        }
        sumL[0]=sL;
        for(int i=1;i+L-1<N;i++){
            sL+=A[i+L-1]-A[i-1];
            sumL[i]=sL;
        }
        for(int i:sumL){
            cout<<i<<endl;
        }

        int sM=0;
        for(int i=0;i<M;i++){
            sM+=A[i];
        }
        sumM[0]=sM;
        for(int i=1;i+M-1<N;i++){
            sM+=A[i+M-1]-A[i-1];
            sumM[i]=sM;
        }
        int Max=0;
        for(int i=0;i<N-L+1;i++)
            for(int j=0;j<N-M+1;j++){
                if(i+L-1<j||j+M-1<i)
                    Max=max(sumL[i]+sumM[j],Max);
            }
        return Max;
    }
};
