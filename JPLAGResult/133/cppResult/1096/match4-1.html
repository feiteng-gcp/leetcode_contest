class Solution {
 public:
  int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
    int n = A.size();
    vector<int> pref(n + 1);
    for (int i = 0; i < n; ++i) {
      pref[i + 1] = pref[i] + A[i];
    }

    int ans = 0;
    for (int i = 0; i + L + M <= n; ++i) {
      for (int j = i + L; j + M <= n; ++j) {
        ans = max(ans, pref[i + L] - pref[i] + pref[j + M] - pref[j]);
      }
    }
    for (int i = 0; i + L + M <= n; ++i) {
      for (int j = i + M; j + L <= n; ++j) {
        ans = max(ans, pref[i + M] - pref[i] + pref[j + L] - pref[j]);
      }
    }
    return ans;
  }
};
