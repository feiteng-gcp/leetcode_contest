class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int len = A.size();
        int lenL = len-L+1, lenM = len-M+1;
        vector<int> recL(lenL), recM(lenM);
        for(int i = 0; i < L; ++i)

            recL[0] += A[i];
        for(int i = 1; i < lenL; ++i){
            recL[i] = recL[i-1] - A[i-1] + A[i+L-1];
        }
        for(int i = 0; i < M; ++i)

            recM[0] += A[i];
        for(int i = 1; i < lenM; ++i){
            recM[i] = recM[i-1] - A[i-1] + A[i+M-1];
        }
        int res = 0;
        for(int i = 0; i < lenL; i++){
            for(int j = 0; j < lenM; j++){
                if(i >= j+M || j >= i+L)
                    res = max(res, recL[i]+recM[j]);
            }
        }
        return res;
    }
};
