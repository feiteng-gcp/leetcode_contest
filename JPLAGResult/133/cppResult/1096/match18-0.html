class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int ret = 0;
        int size = A.size();
        int sumL = 0, sumM = 0;
        for(int i = 0; i < L; ++i)
            sumL += A[i];
        ret = sumL + findMax(M, A, L);
        //cout << "init = " << ret <<endl;
        for(int i = L; i < size; ++i) {
            sumL += A[i];
            sumL -= A[i - L];
            int tm = findMax(M, A, i + 1);
            if(tm < 0)
                break;
            //cout << "a = " << sumL << "b = " << tm << endl;
            ret = max(ret, sumL + tm);
        }
        
        sumM = 0;
        for(int i = 0; i < M; ++i)
            sumM += A[i];
        ret = max(ret, sumM + findMax(L, A, M));
        for(int i = M; i < size; ++i) {
            sumM += A[i];
            sumM -= A[i - M];
            int tm = findMax(L, A, i + 1);
            if(tm < 0)
                break;
            //cout << "aa = " << sumM << "b = " << tm << endl;
            ret = max(ret, sumM + tm);
        }
        //cout << endl;
        return ret;
    }
    
    int findMax(int len, vector<int>& A, int start) {
        int size = A.size();
        if(start + len - 1 >= size)
            return -1;
        int sum = 0;
        for(int i = start; i < start + len; ++i)
            sum += A[i];
        int tMax = sum;
        for(int i = start + len; i < size; ++i) {
            sum += A[i];
            sum -= A[i - len];
            tMax = max(tMax, sum);
        }
        //cout << "tm = " << tMax << endl;
        return tMax;
    }
};
