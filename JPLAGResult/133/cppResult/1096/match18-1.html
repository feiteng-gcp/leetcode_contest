    {
        int N = A.size();        
        int sum=0;
        
        vector<int>LSum(N,-1);

        for (int i=0; i<=L-1; i++)
            sum+=A[i];
        LSum[L-1] = sum;
        for (int i=L; i<N; i++)
        {
            sum+=A[i];
            sum-=A[i-L];
            LSum[i] = sum;
        }
        //cout<<"OK1"<<endl;
        vector<int>LSumMax(N,-1);
        LSumMax[L-1] = LSum[L-1];
        for (int i=L; i<N; i++)
            LSumMax[i] = max(LSumMax[i-1],LSum[i]);
        //cout<<"OK2"<<endl;
        
        
        sum = 0;
        vector<int>MSum(N,-1);

        for (int i=N-1; i>=N-M; i--)
            sum+=A[i];
        MSum[N-M] = sum;
        for (int i=N-M-1; i>=0; i--)
        {
            sum+=A[i];
            sum-=A[i+M];
            MSum[i] = sum;
        }
        //cout<<"OK3"<<endl;

        vector<int>MSumMax(N,-1);
        MSumMax[N-M] = MSum[N-M];
