class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        return max(proc(A, L, M), proc(A, M, L));
    }
    
    int proc(vector<int>& A, int L, int M) {
        int n = A.size();
        int ret = 0;
        for (int i = 0; i < n; ++i) {
            ret = max(_proc(A, 0, i - 1, L) + _proc(A, i, n - 1, M), ret);
        }
        return ret;
    }
    int _proc(vector<int> &A, int l, int r, int len) {
        int sum = 0, ret = 0;
        for (int i = l; i <= r; ++i) {
            sum += A[i];
            if (i - len >= l) sum -= A[i - len];
            if (i - l + 1 >= len) ret = max(ret, sum);
        }
        return ret;
    }
};
