class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int len = A.size();
        vector<int> sumL(len, -1), sumM(len, -1);
        vector<int> maxSumL(len, -1), maxSumM(len, -1);
        sumM[M-1] = 0;
        for(int i = 0; i < M; i++)  sumM[M-1] += A[i];
        maxSumM[M-1] = sumM[M-1];
        for(int i=M; i<len; i++){
            sumM[i] = sumM[i-1]+A[i]-A[i-M];
            maxSumM[i] = max(maxSumM[i-1],sumM[i]);

        }
        sumL[L-1] = 0;
        for(int i = 0; i < L; i++)  sumL[L-1] += A[i];
        maxSumL[L-1] = sumL[L-1];
        for(int i=L; i<len; i++){
            sumL[i] = sumL[i-1]+A[i]-A[i-L];
            maxSumL[i] = max(maxSumL[i-1], sumL[i]);

        }
        int res=-1;
        for(int i = M+L-1; i < len; i++){
            res = max(sumL[i]+maxSumM[i-L], max(res, sumM[i] + maxSumL[i-M]));
        }
        return res;

    }
};
