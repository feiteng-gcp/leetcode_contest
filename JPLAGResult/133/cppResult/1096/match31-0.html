int sum[5555];

class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& a, int l, int m) {
        int n = a.size();
        sum[0] = 0;
        REP(i, n) sum[i + 1] = sum[i] + a[i];
        
        int ans = 0;
        for (int i = l - 1; i + m < n; ++i) {
            for (int j = i + 1; j + m - 1 < n; ++j) {
                ans = max(ans, sum[i + 1] - sum[i - l + 1] + sum[j + m] - sum[j]);
            }
        }
        swap(l, m);
        for (int i = l - 1; i + m < n; ++i) {
            for (int j = i + 1; j + m - 1 < n; ++j) {
                ans = max(ans, sum[i + 1] - sum[i - l + 1] + sum[j + m] - sum[j]);
            }
        }
        return ans;
    }
};
