class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int N = A.size();
        vector<int> partial_sum(N + 1, 0);

        for (int i = 0; i < N; i++)
            partial_sum[i + 1] = partial_sum[i] + A[i];

        int best = 0;

        for (int i = 0; i + L <= N; i++)
            for (int j = 0; j + M <= N; j++)
                if (max(i, j) >= min(i + L, j + M))
                    best = max(best, partial_sum[i + L] - partial_sum[i] + partial_sum[j + M] - partial_sum[j]);

        return best;
    }
};
