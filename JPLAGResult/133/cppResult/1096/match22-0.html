class Solution {
public:
    vector<int> pfx;
    
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        pfx = vector<int>(A.size()+1, 0);
        for(int i=0;i<A.size();++i) {
            pfx[i+1] = pfx[i]+A[i];
        }
        int ans = 0;
        for(int l=0;l<A.size();++l) {
            for(int r=l+L;r+M<=A.size();++r) {
                ans = max(pfx[l+L]+pfx[r+M]-pfx[l]-pfx[r], ans);
            }
        }
        for(int r=0;r<A.size();++r) {
            for(int l=r+M;l+L<=A.size();++l) {
                ans = max(pfx[l+L]+pfx[r+M]-pfx[l]-pfx[r], ans);
            }
        }
        return ans;
    }
};
