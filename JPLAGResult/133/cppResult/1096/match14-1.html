class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        vector<int> dpLL(A.size() + 1, 0);
        vector<int> dpLR(A.size() + 1, 0);
        vector<int> dpML(A.size() + 1, 0);
        vector<int> dpMR(A.size() + 1, 0);
        vector<int> sum(A.size() + 1, 0);
        for (int i = 0; i < A.size(); ++i) {
            sum[i + 1] = sum[i] + A[i];
        }

        for (int i = 0; i < A.size(); ++i) {
            dpLL[i + 1] = max(dpLL[i], sum[i + 1] - sum[max(0, i + 1 - L)]);
            dpML[i + 1] = max(dpML[i], sum[i + 1] - sum[max(0, i + 1 - M)]);
        }
        for (int i = 0; i < A.size() + 1; ++i) {
            sum[i] = 0;
        }

        for (int i = A.size() - 1; i >= 0; --i) {
            sum[i] = sum[i + 1] + A[i];
        }
        for (int i = A.size() - 1; i >= 0; --i) {
            dpLR[i] = max(dpLR[i + 1], sum[i + 1] - sum[min(i + 1 + L, (int)A.size())]);
            dpMR[i] = max(dpMR[i + 1], sum[i + 1] - sum[min(i + 1 + M, (int)A.size())]);
        }
        int res = 0;
        for (int i = 0; i < A.size(); ++i) {
            res = max(res, dpLL[i + 1] + dpMR[i]);
            res = max(res, dpML[i + 1] + dpLR[i]);
        }
        return res;
    }
};
