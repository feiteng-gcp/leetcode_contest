class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int len = A.size();
        int lm_max = max(L, M);
        int lm_min = min(L, M);
        
        vector<int> pre_sum(len + 1, 0);
        for (int i = 1; i <= len; i++) {
            pre_sum[i] = pre_sum[i - 1] + A[i - 1];
        }
        
        vector<int> dp_pre_l(len, 0);
        dp_pre_l[L - 1] = pre_sum[L];
        for (int i = L; i < len; i++) {
            dp_pre_l[i] = max(dp_pre_l[i - 1], pre_sum[i + 1] - pre_sum[i - L + 1]);
        }
        vector<int> dp_pre_m(len, 0);
        dp_pre_m[M - 1] = pre_sum[M];
        for (int i = M; i < len; i++) {
            dp_pre_m[i] = max(dp_pre_m[i - 1], pre_sum[i + 1] - pre_sum[i - M + 1]);
        }
        
        vector<int> dp_post_l(len, 0);
        dp_post_l[len - L] = pre_sum[len] - pre_sum[len - L];
        for (int i = len - L - 1; i >= 0; i--) {
            dp_post_l[i] = max(dp_post_l[i + 1], pre_sum[i + L] - pre_sum[i]);
        }  
        vector<int> dp_post_m(len, 0);
        dp_post_m[len - M] = pre_sum[len] - pre_sum[len - M];
        for (int i = len - M - 1; i >= 0; i--) {
            dp_post_m[i] = max(dp_post_m[i + 1], pre_sum[i + M] - pre_sum[i]);
        }                             
        
        int ans = 0;
        for (int i = L - 1; i < len - M; i++) {
            ans = max(ans, dp_pre_l[i] + dp_post_m[i + 1]);
        }
        for (int i = M - 1; i < len - L; i++) {
            ans = max(ans, dp_pre_m[i] + dp_post_l[i + 1]);
        }
        
        return ans;
    }
};
