class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int size = A.size();
        vector<int> sum(size, 0);
        for (int i = 0; i < size; ++i) {
            if (i == 0) sum[i] = A[i];
            else sum[i] = sum[i - 1] + A[i];
        }
        

        int maxv = 0;
        for (int i = 0; i < size - L - M + 1; ++i) {
            int lsum = sum[i + L - 1] - sum[i] + A[i];
            for (int j = i + L; j < size - M + 1; ++j) {
                int rsum = sum[j + M - 1] - sum[j] + A[j];
                maxv = max(maxv, lsum + rsum);
            }
            
            lsum = sum[i + M - 1] - sum[i] + A[i];
            for (int j = i + M; j < size - L + 1; ++j) {
                int rsum = sum[j + L - 1] - sum[j] + A[j];
                maxv = max(maxv, lsum + rsum);
            }
        }
        return maxv;
    }
};
