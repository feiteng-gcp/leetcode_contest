class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& a, int L, int M) {
        int n = a.size();
        
        vector<int> sumL(n, 0);
        vector<int> sumM(n, 0);
        vector<int> maxL(n, 0);
        vector<int> maxM(n, 0);
        
        sumL[n - 1] = a[n - 1];
        sumM[n - 1] = a[n - 1];
        
        for (int i = n - 2; i >= 0; i--) {
            sumL[i] = sumL[i + 1] + a[i];
            sumM[i] = sumM[i + 1] + a[i];
            
            if (n - i > L) sumL[i] -= a[i + L];
            if (n - i > M) sumM[i] -= a[i + M];
        }
        
        for (const auto &e : sumL) cout << e << " ";
        cout << "\n";
        for (const auto &e : sumM) cout << e << " ";
        cout << "\n";
        
        maxL[n - 1] = sumL[n - 1];
        maxM[n - 1] = sumM[n - 1];
        
        for (int i = n - 2; i >= 0; i--) {
            maxL[i] = max(maxL[i + 1], sumL[i]);
            maxM[i] = max(maxM[i + 1], sumM[i]);
        }
        
        for (const auto &e : maxL) cout << e << " ";
        cout << "\n";
        for (const auto &e : maxM) cout << e << " ";
        
        int sum = 0, ans = INT_MIN;

        for (int i = 0; i < L; i++) sum += a[i];
        
        for (int i = L; i + M <= n; i++) {
            ans = max(ans, sum + maxM[i]);
            sum += a[i] - a[i - L];
        }
        
        sum = 0;
        for (int i = 0; i < M; i++) sum += a[i];
        
        for (int i = M; i + L <= n; i++) {
            ans = max(ans, sum + maxL[i]);
            sum += a[i] - a[i - M];
        }
        
        return ans;
    }
};
