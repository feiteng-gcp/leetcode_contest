class Solution {
public:
    int getAns(vector<int>& A, int L, int M) {
        int n = A.size();
        vector<int> X(n + 2, -100000000), Y(n + 2, -100000000);
        int sum = 0;
        for (int i = 0; i < L; i++) {
            sum += A[i];
        }
        X[L - 1] = sum;
        for (int i = L; i < n; i++) {
            sum += A[i] - A[i - L];
            X[i] = max(X[i - 1], sum);
        }
        sum = 0;
        for (int i = 0; i < M; i++) {
            sum += A[n - 1 - i];
        }
        Y[n - 1 - (M - 1)] = sum;
        for (int i = M; i < n; i++) {
            sum += A[n - 1 - i] - A[n - 1 - (i - M)];
            Y[n - 1 - i] = max(Y[n - 1 - (i - 1)], sum);
        }
        //for (int i = 0; i < n; i++) printf("%d ", X[i]); printf("\n");
        //for (int i = 0; i < n; i++) printf("%d ", Y[i]); printf("\n");
        int ans = -100000000;
        for (int i = 0; i < n; i++) {
            ans = max(ans, X[i] + Y[i + 1]);
        }
        return ans;
    }
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        return max(getAns(A, L, M), getAns(A, M, L));
    }
};
