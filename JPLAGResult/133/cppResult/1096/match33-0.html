class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int n = A.size();
        vector<int> sum(n+1, 0);
        for (int i = 0; i < n; ++i) {
            sum[i+1] = sum[i] + A[i];
        }
        int res = 0;
        vector<int> leftMaxSumL(n, 0), rightMaxSumM(n, 0);
        leftMaxSumL[L - 1] = sum[L] - sum[0];
        for (int i = L; i < n; ++i) {
            leftMaxSumL[i] = max(leftMaxSumL[i-1], sum[i+1] - sum[i+1-L]);
        }
        rightMaxSumM[n - M] = sum[n] - sum[n - M];
        for (int i = n - M - 1; i >= 0; --i) {
            rightMaxSumM[i] = max(rightMaxSumM[i+1], sum[i+M] - sum[i]);
        }
        
        for (int i = L - 1; i + 1 <= n - M; ++i) {
            res = max(res, leftMaxSumL[i] + rightMaxSumM[i+1]);
        }
        vector<int> leftMaxSumM(n, 0), rightMaxSumL(n, 0);
        leftMaxSumM[M - 1] = sum[M] - sum[0];
        for (int i = M; i < n; ++i) {
            leftMaxSumM[i] = max(leftMaxSumM[i-1], sum[i+1] - sum[i+1-M]);
        }
        rightMaxSumL[n - L] = sum[n] - sum[n - L];
        for (int i = n - L - 1; i >= 0; --i) {
            rightMaxSumL[i] = max(rightMaxSumL[i+1], sum[i+L] - sum[i]);
        }
        for (int i = M - 1; i + 1 <= n - L; ++i) {
            res = max(res, leftMaxSumM[i] + rightMaxSumL[i+1]);
        }
        
        return res;
    }
};
