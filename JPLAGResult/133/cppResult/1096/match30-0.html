class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        vector<int> end_at_m(A.size(), 0);
        vector<int> end_at_l(A.size(), 0);
        int sum = 0;
        for (int i = 0; i < M; ++i) {
            sum += A[i];
        }
        
        for (int i = M; i < A.size(); ++i) {
            end_at_m[i-1] = sum;
            sum -= A[i - M];
            sum += A[i];
        }
        end_at_m.back() = sum;
        
        sum = 0;
        
        for (int i = 0; i < L; ++i) {
            sum += A[i];
        }
        for (int i = L; i < A.size(); ++i) {
            end_at_l[i-1] = sum;
            sum -= A[i - L];
            sum += A[i];
        }
        end_at_l.back() = sum;
        
//         for (auto i : end_at_m) {
//             cout << i << " ";
//         }
//         cout << endl;
        
//         for (auto i : end_at_l) {
//             cout << i << " ";
//         }
//         cout << endl;
        
        int ans = 0;
        
        for (int i = 0; i < end_at_m.size(); ++i) {
            for (int j = 0; j < i - M + 1; ++j) {
                ans = max(ans, end_at_m[i] + end_at_l[j]);
            }
        }
        
        for (int i = 0; i < end_at_l.size(); ++i) {
            for (int j = 0; j < i - L + 1; ++j) {
                ans = max(ans, end_at_m[j] + end_at_l[i]);
            }
        }
        
//         int curr = 0;
        
//         for (int i = 0; i < M; ++i) {
//             curr += A[i];
//         }
        
//         for (int i = M; i < A.size(); ++i) {
//             curr -= A[i-M];
//             curr += A[i];
//             for (int j = 0; j < i; ++j) {
//                 ans = max(ans, curr + end_at_l[j]);
//             }
//         }
        
//         curr = 0;
        
//         for (int i = 0; i < L; ++i) {
//             curr += A[i];
//         }
        
//         for (int i = L; i < A.size(); ++i) {
//             curr -= A[i-L];
//             curr += A[i];
//             for (int j = 0; j < i; ++j) {
//                 ans = max(ans, curr + end_at_m[j]);
//             }
//         }
        
        return ans;
    }
};
