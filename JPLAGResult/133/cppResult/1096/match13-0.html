class Solution {
public:
    
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        
        int sum[1010] = {};
        for(int i = 0 ; i < A.size() ; i++){
            sum[i+1] = sum[i] + A[i];
        }
        
        auto f = [&](int l,int r){
            return sum[r] - sum[l];  
        };
        
        int ans = 0;
        for(int i = 0 ; i + L <= A.size() ; i++){
            for(int j = 0 ; j + M <= A.size() ; j++){
                int l = max(i, j);
                int r = min(i + L, j + M);
                if( r > l ){
                    continue;
                }
                ans = max(ans, f(i, i+L) +  f(j,j+M));
            }
        }
        return ans;
    }
};
