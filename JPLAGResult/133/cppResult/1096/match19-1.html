class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        vector<int> lenM(A.size() - M + 1);

        int total = 0;
        for (int i = 0; i < M; ++i) total += A[i];
        lenM[0] = total;
        for (int j = 0; j < A.size() - M; ++j) {
            lenM[j+1] = max(lenM[j], (total += A[j+M] - A[j]));
            //cout << lenM[j+1] << " ";
        }
        //cout << "\n";
        vector<int> lenMr(A.size() - M + 1);

        total = 0;
        for (int i = 0; i < M; ++i) total += A[A.size()-1-i];
        lenMr[0] = total;
        for (int j = 0; j < A.size() - M; ++j) {
            lenMr[j+1] = max(lenMr[j], (total += A[A.size()-1-(j+M)] - A[A.size()-1-j]));
            //cout << lenMr[j+1] << " ";
        }
        //cout << "\n";
        
        

        total = 0;
        for (int i = 0; i < L; ++i) total += A[i];
        int mx = 0;
        for (int j = 0; j < A.size() - L; ++j) {
            if (j >= M) mx = max(total + lenM[j-M], mx);
            if (j <= A.size() - M - L) mx = max(total + lenMr[A.size() - M - L - j], mx);
            total += (A[j+L] - A[j]);
        }
        mx = max(total + lenM[A.size()-L-M], mx);
        return mx;
    }
};
