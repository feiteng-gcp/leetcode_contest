class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        vector<int> sumL(A.size(),0);
        vector<int> sumM(A.size(),0);
        vector<int> maxsumL(A.size(),0);
        vector<int> maxsumM(A.size(),0);
        for(int i = 0; i < L; i++){
            sumL[L-1]+=A[i];
        }
        maxsumL[L-1] = sumL[L-1];
        for(int i = L; i < A.size(); i++){
            sumL[i] = sumL[i-1] + A[i] - A[i-L];
            maxsumL[i] = max(maxsumL[i-1],sumL[i]);
        }
        for(int i = 0; i < M; i++){
            sumM[M-1]+=A[i];
        }
        maxsumM[M-1] = sumM[M-1];
        for(int i = M; i < A.size(); i++){
            sumM[i] = sumM[i-1] + A[i] - A[i-M];
            maxsumM[i] = max(maxsumM[i-1],sumM[i]);
        }
        int res = 0;
        for(int i = A.size()-1; i >= L; i--){
            res = max(res, sumL[i]+maxsumM[i-L]);
        }
        for(int i = A.size()-1; i >= M; i--){
            res = max(res, sumM[i]+maxsumL[i-M]);
        }
        return res;
        
    }
};
