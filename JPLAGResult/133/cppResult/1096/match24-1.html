class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int res = 0;
        vector<int> prefixSum = {0};
        for (int k : A) {
            prefixSum.push_back(k + prefixSum[prefixSum.size() - 1]);
        }
        map<int, int> cache1, cache2;

        int c1 = 0, c2 = 0;
        for (int i = 1; i < prefixSum.size(); ++i) {
            if (i - L >= 0) {
                c1 = max(c1, prefixSum[i] - prefixSum[i - L]);
                cache1[i] = c1;
            }
            if (i - M >= 0) {
                c2 = max(c2, prefixSum[i] - prefixSum[i - M]);
                cache2[i] = c2;
            }
        }
        for (int i = 1; i < prefixSum.size(); ++i) {
            if (i - L >= 0 && cache2.find(i - L) != cache2.end()) {
                res = max(res, prefixSum[i] - prefixSum[i - L] + cache2[i - L]);
            }
            if (i - M >= 0 && cache1.find(i - M) != cache1.end()) {
                res = max(res, prefixSum[i] - prefixSum[i - M] + cache1[i - M]);
            }
        }
        return res;
    }
};
