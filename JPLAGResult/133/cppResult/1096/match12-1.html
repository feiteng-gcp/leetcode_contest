class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        vector<int> acc = {0};
        for (int x : A)
            acc.push_back(acc.back() + x);
        

        int max = 0;
        for (int i = 0; i + L < acc.size(); ++i) {
            for (int j = 0; j + M < acc.size(); ++j) {
                if (i <= j && i + L > j || j <= i && j + M > i)
                    continue;
                max = std::max(max, acc[i + L] - acc[i] + acc[j + M] - acc[j]);
            }
        }
        
        return max;
    }
};
