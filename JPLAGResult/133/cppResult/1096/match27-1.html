class Solution {
public:
	int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
		int n = A.size();
		vector<int> dp1a(n, 0);
		vector<int> dp1b(n, 0);
		vector<int> dp2a(n, 0);
		vector<int> dp2b(n, 0);

		int s = 0;

		for (int i = 0; i < L; i++) {
			s += A[i];
		}
		dp1a[L - 1] = s;
		for (int i = L; i < n; i++) {
			s -= A[i - L];
			s += A[i];
			dp1a[i] = s;
		}

		s = 0;

		for (int i = 0; i < M; i++) {
			s += A[i];
		}
		dp2a[M - 1] = s;
		for (int i = M; i < n; i++) {
			s -= A[i - M];
			s += A[i];
			dp2a[i] = s;
		}

		s = 0;

		for (int i = n - 1; i >= n - L; i--) {
			s += A[i];
		}
		dp1b[n - L] = s;
		for (int i = n - L - 1; i >= 0; i--) {
			s -= A[i + L];
			s += A[i];
			dp1b[i] = s;
		}

		s = 0;

		for (int i = n - 1; i >= n - M; i--) {
			s += A[i];
		}
		dp2b[n - M] = s;
		for (int i = n - M - 1; i >= 0; i--) {
			s -= A[i + M];
			s += A[i];
			dp2b[i] = s;
		}

		for (int i = 1; i < n; i++) {
			dp1a[i] = max(dp1a[i], dp1a[i - 1]);
			dp2a[i] = max(dp2a[i], dp2a[i - 1]);
		}
		for (int i = n - 2; i >= 0; i--) {
			dp1b[i] = max(dp1b[i], dp1b[i + 1]);
			dp2b[i] = max(dp2b[i], dp2b[i + 1]);
		}
		int ans = 0;

		for (int i = 0; i < n; i++) {
			if (i + 1 >= L && n - i -1 >= M) {
				ans = max(ans, dp1a[i] + dp2b[i + 1]);
			}
			if (i + 1 >= M && n - i -1 >= L) {
				ans = max(ans, dp2a[i] + dp1b[i + 1]);
			}
		}
		return ans;
	}
};
