class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int n = A.size();
        vector<int> sum(n + 1);
        for (int i = 0; i < n; ++i) sum[i + 1] = sum[i] + A[i];
        int ans = 0;
        for (int i = 0; i <= n - L; ++i)
        {
            int ln = sum[i + L] - sum[i];
            for (int j = 0; j + M <= i; ++j)
            {
                ans = max(ans, ln + sum[j + M] - sum[j]);
            }
            
            for (int j = i + L; j + M <= n; ++j)
            {
                ans = max(ans, ln + sum[j + M] - sum[j]);
            }
        }
        return ans;
    }
};
