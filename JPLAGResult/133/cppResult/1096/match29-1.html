class Solution {
public:
    int pre[2000], tmp1[2000], tmp2[2000];
    int solve(int left, int right, int n) {
        for (int i = 0; i < 2000; i++) {
            tmp1[i] = tmp2[i] = INT_MIN;
        }
        for (int i = left; i <= n; i++) {
            tmp1[i] = max(tmp1[i - 1], pre[i] - pre[i - left]);
        }

        for (int i = n - right + 1; i > 0; i--) {
            tmp2[i] = max(tmp2[i + 1], pre[i + right - 1] - pre[i - 1]);
        }
        int res = 0;
        for (int i = left; i <= n - right; i++) {
            res = max(res, tmp1[i] + tmp2[i + 1]);
        }
        return res;
    }

    int maxSumTwoNoOverlap(vector<int> &A, int L, int M) {
        int len = A.size();
        pre[0] = 0;
        for (int i = 1; i <= len; i++) {
            pre[i] = pre[i - 1] + A[i - 1];
        }
        return max(solve(L, M, len), solve(M, L, len));
    }
};
