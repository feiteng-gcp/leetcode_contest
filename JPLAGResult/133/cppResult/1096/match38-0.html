class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& a, int l, int r) {
        int n = (int) a.size();
        vector<int> s(n+1, 0);
        for (int i = 0; i < n; i ++) s[i+1] = s[i] + a[i];
        int S = 0;
        int p = 0;
        for (int i = l-1; i+r+1 <= n; i ++) {
            p = max(p, s[i+1] - s[i-l+1]);
            S = max(S, p + s[i+r+1] - s[i+1]);
        }
        swap(l, r);
        p = 0;
        for (int i = l-1; i+r+1 <= n; i ++) {
            p = max(p, s[i+1] - s[i-l+1]);
            S = max(S, p + s[i+r+1] - s[i+1]);
        }
        return S;
    }
};
