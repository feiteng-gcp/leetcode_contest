int sum[1001],maxl[1001],maxm[1001];
class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& A, int L, int M) {
        int n=A.size();
        sum[0]=0;
        for(int i=1;i<=n;i++)
        {
            sum[i]=sum[i-1]+A[i-1];
        }
        maxl[L-1]=0;
        for(int i=L;i<=n;i++)
        {
            maxl[i]=max(maxl[i-1],sum[i]-sum[i-L]);
        }
        maxm[M-1]=0;
        for(int i=M;i<=n;i++)
        {
            maxm[i]=max(maxm[i-1],sum[i]-sum[i-M]);
        }
        int ans=0;
        for(int i=L+M;i<=n;i++)
        {
            ans=max(ans,sum[i]-sum[i-L]+maxm[i-L]);
        }
        for(int i=L+M;i<=n;i++)
        {
            ans=max(ans,sum[i]-sum[i-M]+maxl[i-M]);
        }
        return ans;
    }
};
