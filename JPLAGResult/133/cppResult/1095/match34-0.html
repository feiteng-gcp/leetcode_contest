class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int N = costs.size() / 2;
        std::sort(costs.begin(), costs.end(), [](auto & x, auto & y){
            int rx = x[0] - x[1];
            int ry = y[0] - y[1];
            return rx < ry;
        });
        int cost = 0;
        for (int i = 0; i < N; ++i)
            cost += costs[i][0];
        for (int i = N; i < N * 2; ++i)
            cost += costs[i][1];
        return cost;
    }
};
