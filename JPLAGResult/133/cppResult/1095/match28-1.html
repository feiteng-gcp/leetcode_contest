class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int i;
        vector<int> diff(costs.size(), 0);

        int ans = 0;
        
        for(i = 0; i < costs.size(); ++i) {
            ans += costs[i][0];
            diff[i] = costs[i][1] - costs[i][0];
        }
        
        sort(diff.begin(), diff.end());
        
        for(i = 0; 2 * i < costs.size(); ++i) ans += diff[i];
        
        return ans;
    }
};
