class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int a[105],n=costs.size(),i,ans=0;
        for(i=0;i<n;i++)
        {
            a[i]=costs[i][1]-costs[i][0];
            ans+=costs[i][0];
        }
        sort(a,a+n);
        for(i=0;i<n>>1;i++)ans+=a[i];
        return ans;
    }
};
