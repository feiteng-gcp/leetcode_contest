class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& a) {
        vector<int> A;
        int n = a.size();

        int S = 0;
        for (int i = 0; i < n; i ++) {
            S += a[i][1];
            A.push_back(a[i][0] - a[i][1]);
        }
        sort(A.begin(), A.end());
        for (int i = 0; i < n/2; i ++) S += A[i];
        return S;
    }
};
