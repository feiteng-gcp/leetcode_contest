class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& a) {
        int n = a.size();
        sort(a.begin(), a.end(),
            [](const vector<int>& u, const vector<int>& v) {
                return u[0] - u[1] < v[0] - v[1];
            });
        int ret = 0;
        for (int i = 0; i < n / 2; ++i) {
            ret += a[i][0];
        }
        for (int i = n / 2; i < n; ++i) {
            ret += a[i][1];
        }
        return ret;
    }
};
