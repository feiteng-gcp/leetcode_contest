class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int ans = 0;
        vector<int> arr;

        for (int i = 0; i < costs.size(); i ++) {
            ans += costs[i][0];
            arr.push_back(costs[i][1] - costs[i][0]);
        }
        sort(arr.begin(), arr.end());
        for (int i = 0; i < arr.size() / 2; i ++) {
            ans += arr[i];
        }
        return ans;
    }
};
