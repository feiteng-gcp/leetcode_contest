class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        sort(costs.begin(), costs.end(), [](vector<int>& v, vector<int>& w){
            return v[1] - v[0] < w[1] - w[0];
        });
        int res = 0;
        int N = costs.size();

        N /= 2;
        for (int i = 0; i < N; ++i) {
            res += costs[i][1];
        }
        for (int i = N; i < 2*N; ++i) {
            res += costs[i][0];
        }
        return res;
    }
};
