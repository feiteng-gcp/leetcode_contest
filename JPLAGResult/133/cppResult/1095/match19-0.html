class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int n = costs.size()/2;
        vector<pair<int ,int>> m;
        for(int i=0;i<n*2;++i) {
            m.emplace_back(costs[i][0]-costs[i][1], costs[i][0]);
        }
        sort(m.begin(), m.end());
        int ans = 0;
        for(int i=0;i<n;++i) {
            ans += m[i].second;
        }
        for(int i=n;i<n*2;++i) {
            ans += m[i].second-m[i].first;
        }
        return ans;
    }
};
