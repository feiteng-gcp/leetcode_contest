class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int n = costs.size(), ret = 0;
        vector<int> diff(n);
        for(int i=0; i<n; ++i){
            ret += costs[i][0];
            diff[i] = costs[i][1] - costs[i][0];
        }
        sort(diff.begin(), diff.end());
        ret += accumulate(diff.begin(), diff.begin() + n/2, 0);
        return ret;
    }
};
