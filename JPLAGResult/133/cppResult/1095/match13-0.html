class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int n = costs.size();
        vector<pair<int, vector<int> > > tmp;
        for (auto item : costs) {
            tmp.push_back(make_pair(item[0] - item[1], item));
        }
        sort(tmp.begin(), tmp.end());
        int res = 0;
        for (int i = 0; i < n / 2; i++) {
            res += tmp[i].second[0];
        }
        for (int i = n / 2; i < n; i++) {
            res += tmp[i].second[1];
        }
        return res;
    }
};
