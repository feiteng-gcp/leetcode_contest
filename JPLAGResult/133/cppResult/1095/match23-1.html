class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        vector<pair<int,int>> diff;
        for(int i=0;i<costs.size();i++ ){
            auto v=costs[i];
            diff.push_back({v[0]-v[1],i});
        }
        sort(diff.begin(),diff.end());
        int res=0;

        int N =costs.size()/2;
        for(int i=0;i<N;i++){
            res+=costs[diff[i].second][0];
            res+=costs[diff[i+N].second][1];

        }
        return res;
    }
};
