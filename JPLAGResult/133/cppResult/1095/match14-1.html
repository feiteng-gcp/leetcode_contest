class Solution {
public:
   vector<int> v;
int twoCitySchedCost(vector<vector<int>>& costs) {

	int ans = 0;
	for (int i = 0; i < costs.size(); i++) {
		ans += costs[i][0];
		v.push_back(-costs[i][0] + costs[i][1]);
	}
	sort(v.begin(), v.end());
	for (int i = 0; i < v.size() / 2; i++) {
		ans += v[i];
	}
	return ans;
}
};
