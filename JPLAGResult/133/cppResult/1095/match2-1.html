class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int n = costs.size();
        vector<pair<int, int>> diff(n);

        for(int i=0; i<n; i++) {
            diff[i] = make_pair(costs[i][1] - costs[i][0], i);
        }
        
        sort(diff.begin(), diff.end());
        
        int sum = 0;
        for(int i=0; i<n; i++) {

            if(i < n/2) {
                sum += costs[diff[i].second][1];
            } else {
                sum += costs[diff[i].second][0];
            }
        }
        return sum;
    }
};
