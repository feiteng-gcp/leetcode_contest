class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int ans = 0;
        vector<int> dif;
        int N = costs.size();
        for (int i = 0; i < N; ++i) {
            ans += costs[i][0];
            dif.push_back(costs[i][1] - costs[i][0]);
        }
        sort(dif.begin(), dif.end());
        for (int i = 0; i < N / 2; ++i)
            ans += dif[i];
        return ans;
    }
};
