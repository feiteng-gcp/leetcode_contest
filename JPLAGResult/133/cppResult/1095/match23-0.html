class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        vector<vector<int>> A;
        vector<vector<int>> B;

        for (int i = 0; i < costs.size() / 2; i++) {
            A.push_back(costs[i]);
            B.push_back(costs[costs.size() / 2 + i]);
        }

        for (int a = 0; a < A.size(); a++) {
            int max_gain = 0;
            int b_index = 0;
            for (int b = 0; b < B.size(); b++) {
                int cur_gain = (A[a][0] - B[b][0]) + (B[b][1] - A[a][1]);
                if (cur_gain > max_gain) {
                    b_index = b;
                    max_gain = cur_gain;
                }
            }

            if (max_gain > 0) {
                swap(A[a], B[b_index]);
            }
        }

        int res = 0;
        for (int i = 0; i < A.size(); i++) {
            res += A[i][0];
            res += B[i][1];
        }
        return res;
    }
};
