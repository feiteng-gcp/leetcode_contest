vector<vector<int>> vs;
bool cmp(int a, int b){
	return vs[a][0]-vs[a][1] <  vs[b][0]-vs[b][1];
}
/*

class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
		int n=costs.size();
		vs = costs;
		vector<int> v;
		for(int i=0; i<n; i++){
			v.push_back(i);
		}
		sort(v.begin(), v.end(), cmp);
		int res = 0;
		for(int i=0; i<n/2; i++){
			res += vs[v[i]][0];
		}
		for(int i=n/2; i<n; i++){
			res += vs[v[i]][1];
		}
		return res;

        
    }
};
