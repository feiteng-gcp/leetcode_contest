class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int m = costs.size();
        for(int i=0;i<m;i++){
            costs[i][0] = costs[i][0] - costs[i][1];
        }
        sort(costs.begin(),costs.end());
        int ans = 0;
        for(int i=0;i<m;i++){
            costs[i][0] += costs[i][1];
            if(i < m/2){
                ans += costs[i][0];
            }else{
                ans += costs[i][1];
            }
        }
        return ans;
    }
};
