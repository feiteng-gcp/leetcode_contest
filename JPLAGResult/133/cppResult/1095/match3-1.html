class Solution {
public:
    int twoCitySchedCost(vector<vector<int>>& costs) {
        int n = costs.size();
        vector<pair<int,int>> v(n);
        for(int i = 0;i<n;i++) v[i] = make_pair(costs[i][0]-costs[i][1],i);

        sort(v.begin(),v.end());
        int ans = 0;
        for(int i = 0;i<n;i++){
            if(i<n/2) ans+=costs[v[i].second][0];
            else ans+=costs[v[i].second][1];
        }
        return ans;
    }
};
