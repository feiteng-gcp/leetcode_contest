class StreamChecker {
public:
    struct node {
        bool isWord;
        node *nxt[26];
        
        node() { isWord = false; memset(nxt, 0, sizeof(nxt)); }
    };
    
    node *root;
    vector<char> data;
    
    StreamChecker(vector<string>& words) {
        root = new node();
        for (int i = 0; i < words.size(); ++i) {
            string w = words[i];
            int len = w.length();

            node *p = root;
            
            for (int j = len-1; j >= 0; --j) {
                if (!p->nxt[w[j]-'a'])
                    p->nxt[w[j]-'a'] = new node();
                p = p->nxt[w[j]-'a'];
            }
            p->isWord = true;
        }
    }
    
    bool query(char letter) {
        data.push_back(letter);
        
        node *p = root;
        int i = (int)data.size()-1;
        
        while (i >= 0) {
            p = p->nxt[data[i]-'a'];
            
            if (!p)
                return false;
            if (p->isWord)
                return true;
            
            --i;
        }
        
        return false;
    }
};

/**
 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
