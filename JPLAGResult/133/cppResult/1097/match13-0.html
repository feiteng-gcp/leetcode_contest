class StreamChecker {
public:
    struct Trie{
        bool end;
        Trie *nxt[26];
        Trie(){
            end = false;
            fill(nxt,nxt+26,(Trie*)NULL);
        }
    };
    Trie *root;
    StreamChecker(vector<string>& words) {
        root = new Trie();
        for(int i=0;i<words.size();i++){
            string s = words[i];
            int l = s.size();
            Trie *cur = root;
            for(int j=l-1;j>=0;j--){
                int id = s[j]-'a';
                if(cur->nxt[id]==NULL){
                    cur->nxt[id] = new Trie();
                }
                cur = cur->nxt[id];
            }
            cur->end = true;
        }
    }
    
    vector<char> cv;
    bool query(char letter) {
        cv.push_back(letter);
        int it = cv.size()-1;
        Trie *cur = root;
        while(it >= 0){
            int id = cv[it--] - 'a';
            if(cur->nxt[id] == NULL)return false;
            cur = cur->nxt[id];
            if(cur->end)return true;
        }
        return false;
    }
};

/**
 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
