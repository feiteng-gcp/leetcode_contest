int nex[2002*2002][26];
bool en[2002 * 2002];

int ne;
int od;

class StreamChecker {
public:
   StreamChecker(vector<string>& words) {
	memset(nex, -1, sizeof(nex));
       for(int i=0;i<2002*2002;i++)en[i]=false;
	ne = 0;
	od = 1;

	for (auto &el : words) {
		int cur = ne;
		for (int i = el.size() - 1; i >= 0; i--) {
			int go = el[i] - 'a';
			if (nex[cur][go] == -1) {
				nex[cur][go] = od++;
			}
			cur = nex[cur][go];
		}
		en[cur] = true;
	}
}
vector<char> stk;
bool query(char letter) {
	stk.push_back(letter);

	int cur = ne;
	for (int i = stk.size()-1; i>=0; i--) {
		int go = stk[i] - 'a';
		if (nex[cur][go] == -1) {
			return false;
		}
		cur = nex[cur][go];
		if (en[cur])return true;
	}
	return false;
}
};

/**
 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
