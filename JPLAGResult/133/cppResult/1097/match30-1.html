class Trie
{
public:
    Trie()
    {
        for (int i = 0; i < 26; ++i)
        {
            child[i] = nullptr;
        }
        exist = false;
    }
    void add(string &str, int n)
    {
        if (n == 0)
        {
            exist = true;
            return;
        }
        int index = str[n - 1] - 'a';
        if (child[index] == nullptr)
        {
            child[index] = new Trie();
        }
        child[index]->add(str, n - 1);
    }
    bool find(string &str, int n)
    {
        if (n == 0)
        {
            return exist;
        }
        int index = str[n - 1] - 'a';
        if (child[index] == nullptr)
        {
            return false;
        }
        return child[index]->exist || child[index]->find(str, n - 1);
    }
private:
    Trie *child[26];
    bool exist;

};


class StreamChecker {
public:
    StreamChecker(vector<string>& words)
    {
        root = new Trie();
        for (auto str : words)
        {
            root->add(str, str.size());
        }
    }
    
    bool query(char letter)
    {
        cur.push_back(letter);
        return root->find(cur, cur.size());
    }
private:
    Trie *root;
    string cur;
};
