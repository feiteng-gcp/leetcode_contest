class Node
{
public:
    Node()
    {
        isWord = false;
        next = vector<Node*>(26, nullptr);
    }
    
    vector<Node*> next;
    bool isWord;
};

class StreamChecker {
public:
    StreamChecker(vector<string>& words) 
    {
        root = new Node();
        for (auto w : words)
        {

            auto c = root;
            for (int i = w.size() - 1; i >= 0; --i)
            {
                if (c->next[w[i] - 'a'] == nullptr) c->next[w[i] - 'a'] = new Node();
                c = c->next[w[i] - 'a'];
            }
            c->isWord = true;
        }
    }
    
    bool query(char letter) 
    {
        history = letter + history;
        if (history.size() > 2000) history = history.substr(0, 2000);
        auto cNode = root;
        for (auto c : history)
        {
            if (cNode->next[c - 'a'] == nullptr) return false;
            cNode = cNode->next[c - 'a'];
            if (cNode->isWord) return true;
        }
        return false;
    }
private:
    Node* root;
    string history;
};
