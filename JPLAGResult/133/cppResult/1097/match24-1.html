class TrieNode {
public:
    bool isEnd;
    vector<TrieNode*> next;
    TrieNode() {
        isEnd = false;
        next = vector<TrieNode*>(26, nullptr);
    }
};

class StreamChecker {
    TrieNode root;
    
    list<char> cur;
    

    void insert(string& word) {
        TrieNode* p = &root;
        for (int i = word.size() - 1; i >= 0; --i) {
            char c = word[i];
            if (p->next[c - 'a'] == nullptr) {
                p->next[c - 'a'] = new TrieNode();
            }
            p = p->next[c - 'a'];
        }
        p->isEnd = true;
    }
    
public:
    StreamChecker(vector<string>& words) {
        root.isEnd = true;
        for (auto &word: words) {
            insert(word);
        }
    }
    
    bool query(char letter) {
        cur.push_front(letter);
        if (cur.size() > 2000) {
            cur.pop_back();
        }
        TrieNode* p = &root;
        for (auto c : cur) {
            if (p->next[c - 'a']) {
                p = p->next[c - 'a'];
                if (p->isEnd) {
                    return true;
                }
            }
            else {
                return false;
            }
        }
        return false;
    }
};

/**
 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
