struct TrieNode {
    TrieNode *children[26] = {0};
    bool end = false;
};

class StreamChecker {
    TrieNode *root;
    vector<TrieNode*> queries;
public:
    StreamChecker(vector<string>& words) {
        root = new TrieNode();
        for (auto &word : words) {
            TrieNode *cur = root;
            for (int i = 0; i < word.size(); i++) {
                int idx = word[i] - 'a';
                if (!cur->children[idx]) cur->children[idx] = new TrieNode();
                cur = cur->children[idx];
            }
            cur->end = true;
        }
    }
    
    bool query(char letter) {
        bool res = false;
        vector<TrieNode*> newQueries;
        for (TrieNode *tn : queries) {
            TrieNode *node = tn->children[letter - 'a'];
            if (!node) continue;
            if (node->end) res = true;
            newQueries.push_back(node);
        }
        TrieNode *node = root->children[letter - 'a'];
        if (node) {
            newQueries.push_back(node);
            if (node->end) res = true;
        }
        queries = newQueries;
        return res;
    }
};

 * StreamChecker* obj = new StreamChecker(words);
