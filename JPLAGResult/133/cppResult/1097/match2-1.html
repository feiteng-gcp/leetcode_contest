class StreamChecker {
public:
    struct TrieNode {
        TrieNode* link[26];
        bool isEnd;
        TrieNode() : isEnd(false) {
            fill(begin(link), end(link), nullptr);
        }
    };
 
    TrieNode *root = new TrieNode();
    string buffer = "";
    int N = 0;
    
    StreamChecker(vector<string>& words) {
        int n;
        TrieNode *it;
        for (const auto &s : words) {

            n = s.size();
            it = root;
            for (int i = n - 1; i >= 0; i--) {
                if (!it->link[s[i] - 'a']) it->link[s[i] - 'a'] = new TrieNode();
                it = it->link[s[i] - 'a'];
            }
            it->isEnd = true;
        }
    }
    
    bool query(char letter) {
        N++;
        buffer.push_back(letter);
        
        TrieNode *it = root;
        
        for (int i = N - 1; i >= 0; i--) {
            if (it->link[buffer[i] - 'a']) {
                it = it->link[buffer[i] - 'a'];
                if (it->isEnd) return true;
            }
            else return false;
        }
        
        return false;
    }
};

/**
 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
