struct Node {
    vector<Node *> nxt;
    bool isTerminal;
    Node() {
        nxt.resize(26);
        isTerminal = false;
    }
};
class StreamChecker {
public:
    Node *root = NULL;
    vector<Node *> pointers;
    StreamChecker(vector<string>& words) {
        root = new Node();
        for (string s : words) {
            Node *now = root;
            for (char c : s) {
                if (now->nxt[c - 'a'] == NULL) {
                    now->nxt[c - 'a'] = new Node();
                }
                now = now->nxt[c - 'a'];
            }
            now->isTerminal = true;
        }
    }
    
    bool query(char letter) {
        bool res = false;
        vector<Node *> nptrs;
        int c = letter - 'a';
        for (auto ptr : pointers) {
            if (ptr->nxt[c] != NULL) {
                Node *nxt = ptr->nxt[c];
                nptrs.push_back(nxt);
                if (nxt->isTerminal) {
                    res = true;
                }
            }
        }
        pointers = move(nptrs);
        if (root->nxt[c] != NULL) {
            Node *ptr = root->nxt[c];
            if (ptr->isTerminal) {
                res = true;
            }
            pointers.push_back(ptr);
        }
        return res;
    }
};

 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
