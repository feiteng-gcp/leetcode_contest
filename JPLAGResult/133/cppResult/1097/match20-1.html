struct Node {
    int idx;
    int child[26];
    bool word;
    
    Node(int idx) : idx(idx) {
        word = false;
        memset(child, -1, sizeof(child));
    }
};

class Trie {
public:
    Trie() {
        nodes.push_back(Node(0));
    }
    
    void insert(string& s) {

        int len = s.size();
        int p = 0;
        for (int i = len - 1; i >= 0; --i) {
            int c = s[i] - 'a';
            if (nodes[p].child[c] == -1) {
                int next_idx = nodes.size();
                nodes.push_back(Node(next_idx));
                nodes[p].child[c] = next_idx;
            }
            p = nodes[p].child[c];
        }
        nodes[p].word = true;
    }
    
    bool search(string& s) {
        int p = 0;
        int len = s.size();
        for (int i = len - 1; i >= 0; --i) {
            int c = s[i] - 'a';
            if (nodes[p].child[c] == -1) {
                return false;
            }
            p = nodes[p].child[c];
            if (nodes[p].word) {
                return true;
            }
        }
        return false;
    }
    
    vector<Node> nodes;
};

class StreamChecker {
public:

    StreamChecker(vector<string>& words) {
        int word_count = words.size();
        for (int i = 0; i < word_count; ++i) {
            trie.insert(words[i]);
        }
    }
    
    bool query(char letter) {
        s += letter;
        return trie.search(s);
    }
    
private:
    Trie trie;
    string s;
};
