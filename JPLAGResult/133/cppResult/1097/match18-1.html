class StreamChecker {
public:
    StreamChecker(vector<string>& words) {
        root = new TrieNode;

        for (const auto word : words) {
            TrieNode *cur = root;
            for (auto c : word) {
                int j = c - 'a';
                if (!cur->ch[j]) cur->ch[j] = new TrieNode;
                cur = cur->ch[j];
            }
            cur->is_end = true;
        }
    }
    
    bool query(char letter) {
        vector<TrieNode*> nxt;
        int j = letter - 'a';
        bool res = false;
        for (auto node : q) {
            if (!node->ch[j]) continue;
            node = node->ch[j];
            if (node->is_end) {
                res = true;
            }
            nxt.push_back(node);
        }
        if (root->ch[j]) {
            nxt.push_back(root->ch[j]);
            if (root->ch[j]->is_end) res = true;
        }
        q = nxt;
        return res;
    }
    struct TrieNode{
        TrieNode () {
            memset(ch, 0, sizeof(ch));
            is_end = false;
        }
        TrieNode *ch[26];
        bool is_end;
    };
    TrieNode *root;
    vector<TrieNode*> q;
};
