struct dnode {
    dnode* next[26];
    int v;
    dnode() {
        memset(next, 0, sizeof(next));
        v = 0;
    }
};
class StreamChecker {
public:
    dnode* hd;
    vector<dnode*> pt;
    StreamChecker(vector<string>& words) {
        hd = new dnode;
        for(int i = 0; i < words.size(); i++) {

            dnode *cur = hd;
            for(int j = 0; j < words[i].size(); j++) {
                int x = words[i][j] - 'a';
                if(cur->next[x] == NULL) {
                    cur->next[x] = new dnode();
                }
                cur = cur->next[x];
                //cur-> s = words[i].substr(0, j + 1);
            }
            cur->v++;
        }
    }
    bool query(char letter) {
        bool ans = 0;
        int x = letter - 'a';
        pt.push_back(hd);
        for(int i = 0; i < pt.size(); i++) {
            if(pt[i]->next[x]) {
                pt[i] = pt[i]->next[x];
                ans |= pt[i]->v;
            }
            else {
                swap(pt[i], pt[pt.size() - 1]);
                pt.pop_back();
                i--;
            }
            
        }

        return ans;
    }
};

 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
