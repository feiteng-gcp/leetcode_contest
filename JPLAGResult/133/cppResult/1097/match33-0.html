struct TrieNode {
    int s[26];
    bool end;
} nodes[1000000];


class StreamChecker {
public:
    int cnt;
    inline int newnode() {
        int ret = cnt++;
        memset(&nodes[ret], 0, sizeof(TrieNode));
        return ret;
    }
    int queries[100000];
    int q;
    
    StreamChecker(vector<string>& words) {
        cnt = 1;
        memset(nodes, 0, sizeof nodes);
        for (string &s : words) {
            int p = 0;
            for (int i = s.length() - 1; i >= 0; --i) {
                if (nodes[p].s[s[i] - 'a'] == 0) {
                    p = nodes[p].s[s[i] - 'a'] = newnode();
                } else {
                    p = nodes[p].s[s[i] - 'a'];
                }
            }
            nodes[p].end = true;
        }
        q = 0;
    }
    
    bool query(char letter) {
        queries[q++] = letter - 'a';
        int p = 0;
        for (int i = q - 1; i >= 0; --i) {
            if (nodes[p].s[queries[i]] == 0) return false;
            p = nodes[p].s[queries[i]];
            if (nodes[p].end) return true;
        }
        return false;
    }
};

 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
