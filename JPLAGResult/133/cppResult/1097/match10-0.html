class StreamChecker {
public:
    struct node {
        node *ch[26], *fail;
        bool end;

        node(): end(false), fail(NULL) {
            memset(ch, 0, sizeof(ch));
        }
    } *root;

    StreamChecker(vector<string>& words) {
        root = new node();
        for (auto &it: words) {
            auto p = root;
            for (int i = it.size() - 1; i >= 0; --i) {
                if (!p->ch[it[i] - 'a'])
                    p->ch[it[i] - 'a'] = new node();
                p = p->ch[it[i] - 'a'];
            }
            p->end = true;
        }
    }

    vector<char> hist;
    bool query(char letter) {
        hist.push_back(letter - 'a');
        auto p = root;
        for (int i = hist.size() - 1; i >= 0; --i) {
            if (p->ch[hist[i]]) {
                p = p->ch[hist[i]];
                if (p->end)
                    return true;
            } else
                break;
        }
        return false;
    }
};

 * StreamChecker* obj = new StreamChecker(words);
