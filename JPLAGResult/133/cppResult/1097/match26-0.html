struct DictTreeNode{
    unordered_map<char, DictTreeNode*> children;
    bool isLeaf;
    DictTreeNode(bool flag): isLeaf(flag){};
};

class StreamChecker {
private:
    DictTreeNode* root;
    vector<char> queries;
public:
    void insertWord(string word){
        reverse(word.begin(), word.end());
        DictTreeNode* tmp = root;
        for(auto c: word){
            if(tmp->children.find(c) == tmp->children.end()){
                tmp->children[c] = new DictTreeNode(false);
            }
            tmp = tmp->children[c];
        }
        tmp->isLeaf = true;
    }
    StreamChecker(vector<string>& words) {
        root = new DictTreeNode(false);
        for(auto word : words){
            insertWord(word);
        }
    }
    
    bool query(char letter) {
        queries.push_back(letter);
        DictTreeNode* tmp = root;
        for(int i = queries.size()-1; i >= 0; i--){
            if(tmp->children.find(queries[i]) == tmp->children.end()){
                return false;
            }
            tmp = tmp->children[queries[i]];
            if(tmp->isLeaf)  return true;
        }
        return false;
    }
};
