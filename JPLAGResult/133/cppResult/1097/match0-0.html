class StreamChecker {
public:
    
    map <int, int> go[26];
    unordered_set<int> cnt;
    int sz = 1;
    string s= "";

    void add(string &s) {
        int cur = 0;
        for (int i=0;i<s.length();i++) {
            int c = s[i] - 'a';
            if (go[c].count(cur) == 0) {
                go[c][cur] = sz++;
            }
            cur = go[c][cur];
        }
        cnt.insert(cur);
    }
    
    StreamChecker(vector<string>& words) {
        for (string w: words) {
            reverse(all(w));
            add(w);
        }
    }
    
    bool query(char letter) {
        s += letter;
        int cur = 0;
        for (int ii=sz(s) - 1;ii >= 0;ii--) {
            int c = s[ii] - 'a';
            if (go[c].count(cur) == 0) {
                return false;
            }
            cur = go[c][cur];
            if (cnt.find(cur) != cnt.end()) return true;
        }
        return false;
    }
};
