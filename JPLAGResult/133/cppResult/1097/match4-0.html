struct Node {
    bool end;
    vector<Node*> next;
    Node() {end = false; next.resize(26, nullptr);}
};

class StreamChecker {
private:
    list<Node*> l;
    Node* root;
public:
    StreamChecker(vector<string>& words) {
        root = new Node();
        for (auto w: words) {
            Node* cur = root;
            for (int i = 0; i < w.size(); ++i) {
                if (cur->next[w[i] - 'a'] == nullptr) cur->next[w[i]-'a'] = new Node();
                cur = cur->next[w[i]-'a'];
            }
            cur->end = true;
        }
    }
    
    bool query(char letter) {
        l.push_back(root);
        bool ans = false;
        for (auto iter = l.begin(); iter != l.end();) {
            auto& node = *iter;
            if (node->next[letter-'a'] == nullptr) {
                iter = l.erase(iter);
            } else {
                node = node->next[letter-'a'];
                if (node->end) ans = true;
                 ++iter;
            }
        }
        return ans;
    }
};

 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
