class StreamChecker {
    struct TrieNode {
        char c;
        TrieNode* next[26];
        bool is_leaf;
        TrieNode(char _c) {
            c = _c;
            for (int i = 0; i < 26; i++) {
                next[i] = nullptr;
            }
            is_leaf = false;
        }
    };
    struct Trie {
        TrieNode* root;
        Trie() {
            root = new TrieNode('x');
        }
        ~Trie() {
            if (root) delete root;
        }
        void add(string& s) {
            TrieNode* cur = root;
            for (char c : s) {
                if (!cur->next[c - 'a']) {
                    cur->next[c - 'a'] = new TrieNode(c);
                }
                cur = cur->next[c - 'a'];
            }
            cur->is_leaf = true;
        }
        bool match(string& s) {
            TrieNode* cur = root;
            for (int i = s.size() - 1; i >= 0; i--) {
                char c = s[i];
                if (!cur->next[c - 'a']) return false;
                cur = cur->next[c - 'a'];
                if (cur->is_leaf) return true;
            }
            return false;
        }
    };
    Trie* trie;
    unordered_set<TrieNode*> candidates;
    string s;
public:
    StreamChecker(vector<string>& words) {
        trie = new Trie();
        for (string& s : words) {
            reverse(s.begin(), s.end());
            trie->add(s);
        }
    }
    
    bool query(char letter) {
        s += letter;
        return trie->match(s);
    }
};
