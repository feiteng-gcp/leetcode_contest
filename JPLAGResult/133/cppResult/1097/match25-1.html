struct TreeNode_t {
    bool hasEnd;
    vector<TreeNode_t *> subnodes;   
    TreeNode_t():hasEnd(false), subnodes(26, NULL) {}

};

class Tree {
public:
    Tree(vector<string> &words) {
        m_root = new TreeNode_t();
        for (auto &word: words) {
            TreeNode_t *root = m_root;
            for (int i = word.size() - 1; i >= 0; --i) {
                char c = word[i];
                if (!root->subnodes[c - 'a']) {
                    root->subnodes[c - 'a'] = new TreeNode_t();                 
                }
                root = root->subnodes[c - 'a'];
            }
            root->hasEnd = true;
        }
    }
    
    bool query(string &q) {
        TreeNode_t *root = m_root;
        for (int i = q.size() - 1; i >= 0; --i) {           
            char c = q[i];
            if (!root->subnodes[c - 'a']) break;
            root = root->subnodes[c - 'a'];
            if (root->hasEnd) return true;
        }
        return false;
    }
private:
    TreeNode_t *m_root;
};


class StreamChecker {
public:
    StreamChecker(vector<string>& words): m_tree(words), m_cur("") {
        
    }
    
    bool query(char letter) {
        m_cur.push_back(letter);
        return m_tree.query(m_cur);
    }
    
    Tree m_tree;
    string m_cur;
};
