class Trie {
private:
    struct trieNode{
        bool isEnd;
        trieNode* next[26];
        trieNode(){
            isEnd = false;
            memset(next, NULL, sizeof(next));
        }
        ~trieNode(){
            for(int i = 0; i < 26; ++i){
                if(next[i])
                    delete next[i];
            }
        }
    };
    trieNode* root;
public:
    /** Initialize your data structure here. */
    Trie() {
        root = new trieNode();
    }
    
    ~Trie() {
        delete root;
    }
    
    /** Inserts a word into the trie. */
    void insert(string word) {
        int len = word.size();
        trieNode* curNode = root;
        for(int i = 0; i < len; i++){
            int curPos = word[i] - 'a';

            if(!curNode->next[curPos])
                curNode->next[curPos] = new trieNode();
            curNode = curNode->next[curPos];
        }
        curNode->isEnd = true;
    }
    
    /** Returns if the word is in the trie. */
    bool search(string word) {
        trieNode* curNode = root;
        int len = word.size();
        for(int i = 0; i < len; ++i){
            int curPos = word[i] - 'a';
            if(!curNode->next[curPos])
                return false;
            curNode = curNode->next[curPos];
        }
        return curNode->isEnd;
    }
    
    /** Returns if there is any word in the trie that starts with the given prefix. */
    bool startsWith(string prefix) {
        trieNode* curNode = root;
        int len = prefix.size();
        for(int i = 0; i < len; ++i){
            int curPos = prefix[i] - 'a';
            if(!curNode->next[curPos])
                return false;
            curNode = curNode->next[curPos];
        }
        return true;
    }
};


class StreamChecker {
private:
    Trie m_trie;
    vector<char> history;
public:
    StreamChecker(vector<string>& words) {
        for(int i = 0; i < words.size(); ++i){
            reverse(words[i].begin(), words[i].end());
            m_trie.insert(words[i]);
        }
    }
    
    bool query(char letter) {
        history.push_back(letter);
        int curPos = history.size()-1;
        string curStr;
        while(curPos >= 0){
            curStr.push_back(history[curPos--]);
            if(m_trie.search(curStr)) return true;
            else if(!m_trie.startsWith(curStr)) return false;
        }
        return false;
    }
};
