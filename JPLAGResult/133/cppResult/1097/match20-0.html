class StreamChecker {
public:
    char str[4100000];
    int tot, idx;
    struct node {
        int next[26];
        int isfin;
    } trie[4100000];

    void init() {
        memset(str, 0, sizeof(str));
        tot = 0;
        idx = 0;
    }

    void insert_str(const string &s) {
        int len = s.length();
        int tmp = 0;
        for (int i = len - 1; i >= 0; i--) {
            if (!trie[tmp].next[s[i] - 'a']) {
                memset(trie[tot + 1].next, 0, sizeof(trie[tot + 1].next));
                trie[tot + 1].isfin = false;
                trie[tmp].next[s[i] - 'a'] = ++tot;
            }
            tmp = trie[tmp].next[s[i] - 'a'];
        }
        trie[tmp].isfin = true;
    }

    StreamChecker(vector<string> &words) {
        init();
        int len = words.size();
        for (int i = 0; i < len; i++) {
            insert_str(words[i]);
        }
    }

    bool query(char letter) {
        str[++idx] = letter;
        int now = 0;
        for (int i = idx; i > 0; i--) {
            int c = str[i] - 'a';
            if (trie[now].next[c]) {
                now = trie[now].next[c];
            } else {
                return false;
            }
            if (trie[now].isfin) {
                return true;
            }
        }
        return false;
    }
};
