struct TrieNode {
    unordered_map<char, TrieNode*> dic;
    bool is_word;
    TrieNode() : dic(), is_word(false) {}
};

class StreamChecker {
public:
    StreamChecker(vector<string>& words) {
        root = new TrieNode();
        for (auto &w: words) {
            reverse(w.begin(), w.end());
            insert(w);
        }
    }
    
    bool query(char ch) {
        qs.push_back(ch);
        return check();
    }
    
    void insert(const string &s) {
        TrieNode *p = root;
        for (char ch: s) {
            if (p->dic.find(ch) == p->dic.end())
                p->dic[ch] = new TrieNode();
            p = p->dic[ch];
        }
        p->is_word = true;
    }
    
    bool check() {
        TrieNode *p = root;
        int n = qs.size();
        for (int i = n-1; i >= 0; --i) {
            if (p->is_word) return true;
            if (p->dic.find(qs[i]) == p->dic.end()) return false;
            p = p->dic[qs[i]];
        }
        return p->is_word;
    }
    
    TrieNode *root;
    vector<char> qs;
};
