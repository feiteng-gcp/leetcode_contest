struct TrieNode {
    int next[26];
    bool fin;
    TrieNode () {
        fill(next, next + 26, 0);
        fin = 0;
    }
};
class StreamChecker {
public:
    vector<TrieNode> trie;
    vector<char> chrs;
    StreamChecker(vector<string>& words) {
        trie.resize(0);
        trie.push_back(TrieNode());
        for (string &word : words) {
            int root = 0;
            reverse(word.begin(), word.end());
            for (char &c : word) {
                if (trie[root].next[c - 'a'] == 0) {
                    trie[root].next[c - 'a'] = int(trie.size());
                    trie.push_back(TrieNode());
                }
                root = trie[root].next[c - 'a'];
            }
            trie[root].fin = true;
        }
        chrs.resize(0);
    }

    bool query(char letter) {
        chrs.push_back(letter);

        int root = 0;
        for (auto it = chrs.rbegin(); it != chrs.rend(); it ++) {
            if (trie[root].next[*it - 'a'] == 0) {
                return false;
            }
            root = trie[root].next[*it - 'a'];
            if (trie[root].fin) {
                return true;
            }
        }
        return false;
    }
};
