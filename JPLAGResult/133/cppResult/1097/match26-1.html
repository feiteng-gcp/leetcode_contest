class StreamChecker {
public:
    
    struct TrieNode {
        char c;
        map<char, TrieNode*> children;
        bool terminating;
    };
    
    TrieNode* head;
    vector<char> seen;
    
    void insert(TrieNode* head, string s) {
        for (int i = 0; i < s.size(); ++i) {
            if (!head->children.count(s[i])) {
                head->children[s[i]] = new TrieNode();
                head->children[s[i]]->c = s[i];
                head->children[s[i]]->terminating = false;
            }
            head = head->children[s[i]];
        }
        head->terminating = true;
    }
    
    TrieNode* get(TrieNode* curr, char c) {
        if (curr->children.count(c)) {
            return curr->children[c];
        } else {
            return nullptr;
        }
    }
    
    bool isWord(TrieNode* curr) {
        return curr->terminating;

    }
    
    StreamChecker(vector<string>& words) {
        head = new TrieNode();
        for (auto s : words) {
            reverse(s.begin(), s.end());
            insert(head, s);
        }
    }
    
    bool query(char letter) {
        seen.push_back(letter);
        TrieNode* curr = head;
        for (int i = seen.size() - 1; i >= 0; --i) {
            char c = seen[i];
            curr = get(curr, c);
            if (curr == nullptr) return false;
            if (isWord(curr)) return true;
        }
        return false;
        
    }
};

 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
