class StreamChecker {
public:
    vector<vector<int>> trie;
    vector<bool> at; 
    string cur;
    int cc = 0; 
    int fail = 0;
    void insert(string s) {
        if(fail) return;
        int cur = 0;
        for(auto& c : s) {
            //cerr << "!!" << cur << " " << (int) c-'a' << " " << trie[cur].size() << endl;
            if(cur >= trie.size()) {
                fail = cur;
                return;
            }
            if(!trie[cur][c-'a']) {
                trie[cur][c-'a'] = ++cc;
                trie.push_back(vector<int>(26, 0));
                at.push_back(0);
            }
            cur = trie[cur][c-'a'];
        }
        at[cur] = 1;
    }   
    StreamChecker(vector<string>& words) : trie(1, vector<int>(26, 0)), at(1, 0) {
        //trie = vector<vector<int>>(2000*2000, vector<int>(26, 0));
        //at = vector<bool>(2000*2000, false);
        for(auto& y : words) {
            string x = y;
            reverse(x.begin(), x.end());
            insert(x);
        }
    }   
    bool query(char letter) {
        //return (fail/2)%2;
        cur += letter;
        int h = 0;
        for(int i=cur.size()-1;i>=0;--i) {
            //cerr << "!!" << h << " " << (int) cur[i]-'a' << endl;
            if(!trie[h][cur[i]-'a']) return false;
            h = trie[h][cur[i]-'a'];
            if(at[h]) return true;
        }
        return false;
    }   
};
