class StreamChecker {
public:
    StreamChecker(vector<string>& w) {
        s="";
        len=0;
        r=new node();
        for(int i=0;i<w.size();i++){
            node* rr=r;
            reverse(w[i].begin(),w[i].end());
            for(int j=0;j<w[i].size();j++){
                int tmp=w[i][j]-'a';
                if(rr->c[tmp]==NULL)
                    rr->c[tmp]=new node();
                rr=rr->c[tmp];
            }
            rr->child=true;
            len=max(len,(int)w[i].size());
        }
    }
    
    bool query(char letter) {
        s+=letter;
        int n=(int)s.size();

        node* rr=r;
        for(int i=0;i<min(len,n);i++)
        {
            int tmp=s[n-1-i]-'a';
            if(rr->c[tmp]==NULL)
                return false;
            rr=rr->c[tmp];
            if(rr->child==true)
                return true;
        }
        return false;
    }
private:
    int len;
    string s;
    struct node
    {
        node* c[26];
        bool child;
        node()
        {
            for(int i=0;i<26;i++)
                c[i]=NULL;
            child=false;
        }
    }*r;
};

 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
