class StreamChecker {
public:
    StreamChecker(vector<string>& words) {
        root = make_unique<Node>();
        max_len = 0;
        for (const auto& word : words) {
            max_len = max(max_len, int(word.length()));
            auto* p = root.get();
            for (auto iter = word.crbegin(); iter != word.crend(); iter++) {
                if (p->letters[(*iter) - 'a'] == nullptr) {
                    p->letters[(*iter) - 'a'] = make_unique<Node>();
                }
                p = p->letters[(*iter) - 'a'].get();
            }
            p->has_end = true;
        }
    }
    
    bool query(char letter) {
        queue.push_back(letter);
        if (queue.size() > max_len) {
            queue.pop_front();
        }
        const auto* p = root.get();
        for (auto iter = queue.crbegin(); iter != queue.crend(); iter++) {
            if (p->letters[(*iter) - 'a'] == nullptr) {
                return false;
            }
            p = p->letters[(*iter) - 'a'].get();
            if (p->has_end) {
                return true;
            }
        }
        return p->has_end;
    }
private:
    struct Node {
        unique_ptr<Node> letters[26];
        bool has_end = false;
    };
    int max_len;
    deque<char> queue;
    unique_ptr<Node> root;
};
