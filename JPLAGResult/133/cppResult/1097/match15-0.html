class treeNode{
public:
    int isend;
    treeNode *next[26];
    treeNode(int ise){
        isend = ise;
        for(int i = 0; i< 26; i++){
            next[i] = NULL;
        }
    }
};


class StreamChecker {
public:
    treeNode *root;
    queue<treeNode*> q;
    StreamChecker(vector<string>& words) {
        root = new treeNode(0);
        for(int i = 0; i < words.size(); i++){
            treeNode *pn = root;
            for(int j = 0; j < words[i].size(); j++){                
                if(pn->next[words[i][j]-'a']==NULL){
                    pn->next[words[i][j]-'a'] = new treeNode(0);
                }
                if(j==words[i].size()-1)pn->next[words[i][j]-'a']->isend=1;
                pn = pn->next[words[i][j]-'a'];
            }
        }
    }
    
    bool query(char letter) {
        int n = q.size();
        bool state = false;
        while(n--){
            if(q.front()->next[letter-'a']!=NULL){
                q.push(q.front()->next[letter-'a']);
                if(q.front()->next[letter-'a']->isend) state = true;
            }
            q.pop();
        }
        if(root->next[letter-'a']!=NULL){
            q.push(root->next[letter-'a']);
            if(root->next[letter-'a']->isend) state = true;
        }
        return state;
    }
};

 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
