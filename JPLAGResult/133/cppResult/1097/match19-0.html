struct TrieNode {
    bool word = false;
    vector<TrieNode*> next;
    TrieNode() {
        next = vector<TrieNode*>(26, nullptr);
    }
};

class StreamChecker {
public:
    StreamChecker(vector<string>& words) {
        for (auto str : words) {
            reverse(str.begin(), str.end());
            TrieNode* build = root;
            for (auto c : str) {
                int index = c - 'a';
                if (build->next[index] == nullptr) build->next[index] = new TrieNode();
                build = build->next[index];
            }
            build->word = true;
        }
    }
    
    bool query(char letter) {
        word += letter;
        int n = word.size();
        TrieNode* build = root;
        for (int i = n-1; i >= 0; --i) {
            int index = word[i] - 'a';
            if (build->next[index] == nullptr) {return false;}
            build = build->next[index];
            if (build->word) {return true;}
        }
        return false;
    }
private:
    string word;
    TrieNode* root = new TrieNode();
};

 * StreamChecker* obj = new StreamChecker(words);
 * bool param_1 = obj->query(letter);
