class Solution {
    static class myComp implements Comparator<int[]> {
        int[] orig = null;
        myComp(int[] orig) {
            this.orig = orig;
        }
        public int compare(int[] a, int[] b) {
            int d1 = Math.abs(a[0] - orig[0]) + Math.abs(a[1] - orig[1]);
            int d2 = Math.abs(b[0] - orig[0]) + Math.abs(b[1] - orig[1]);
            return d1 - d2;
        }
    }
    public int[][] allCellsDistOrder(int R, int C, int r0, int c0) {
        List<int[]> ret = new ArrayList<>();
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                ret.add(new int[] {i, j});
            }
        }
        
        Collections.sort(ret, new myComp(new int[] {r0, c0}));
        
        int[][] ret2 = new int[R * C][2];
        for (int i = 0; i < R * C; i++) {
            ret2[i] = ret.get(i);
        }
        return ret2;
    }
}
