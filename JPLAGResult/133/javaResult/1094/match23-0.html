class Solution {
    public int[][] allCellsDistOrder(int m, int n, int r0, int c0) {
		List<int[]> pointList = new ArrayList<>();
		for(int i = 0; i < m; i++) {
			for(int j = 0; j < n; j++) {
				pointList.add(new int[]{i,j,Math.abs(i - r0)+ Math.abs(j - c0)});
			}
		}
		Collections.sort(pointList, new Comparator<int[]>() {
			@Override
			public int compare(int[] o1, int[] o2) {
				return Integer.compare(o1[2], o2[2]);
			}
		});
		int[][] result = new int[pointList.size()][2];
		for(int i = 0; i < pointList.size();i++) {
			result[i][0] = pointList.get(i)[0];
			result[i][1] = pointList.get(i)[1];
		}
		return result;
	}
}
