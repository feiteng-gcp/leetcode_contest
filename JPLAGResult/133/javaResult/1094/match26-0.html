class Solution {
    public int[][] allCellsDistOrder(int R, int C, int r0, int c0) {
        // R = 2; C = 3; //r0 = 1; c0 = 2;
        int[][] res = new int[R * C][2];
        boolean[][] visited = new boolean[R][C];
        Comparator<int[]> cmp = new Comparator<int[]>() {
            public int compare(int[] m, int[] n) {
                return Math.abs(r0 - m[0]) + Math.abs(c0 - m[1]) - 
                       Math.abs(r0 - n[0]) - Math.abs(c0 - n[1]);
            }
        };
        int k = 0;
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                res[k++] = new int[]{i, j};
            }
        }
        Arrays.sort(res, cmp);
        return res;
    }
}
