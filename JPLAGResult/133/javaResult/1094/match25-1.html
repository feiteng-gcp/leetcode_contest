class Solution {
    public int[][] allCellsDistOrder(int R, int C, int r0, int c0) {
        TreeMap<Integer, List<int[]>> map = new TreeMap<>();
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                int d = Math.abs(i - r0) + Math.abs(j - c0);
                map.putIfAbsent(d, new LinkedList<>());
                map.get(d).add(new int[]{i, j});
            
}
        }
        int[][] res = new int[R * C][];
        int i = 0;
        for (List<int[]> v: map.values()) {
            for (int[] k: v) 
{
                res[i++] = k;
            }
        }
        return res;
    }
}
