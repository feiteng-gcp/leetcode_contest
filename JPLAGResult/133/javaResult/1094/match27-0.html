class Solution {
    public int[][] allCellsDistOrder(int R, int C, int r0, int c0) {
        int[][] map = new int[R][C];
        int[][] res = new int[R*C][2];
        for (int i = 0; i < R; i++) Arrays.fill(map[i], Integer.MAX_VALUE);
        map[r0][c0] = 0;
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[]{r0, c0});
        int[][] dir = new int[][]{{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
        int index = 0;
        while (!q.isEmpty()) {
            int[] cur = q.poll();
            res[index++] = cur;
            int dis = map[cur[0]][cur[1]];
            for (int[] di : dir) {
                int ny = cur[0] + di[0], nx = cur[1] + di[1];
                if (ny >= 0 && ny < R && nx >= 0 && nx < C && dis+1 < map[ny][nx]) {
                    map[ny][nx] = dis+1;
                    q.offer(new int[]{ny, nx});
                }
            }
        }
        return res;
    }
}
