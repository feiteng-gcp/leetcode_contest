class Solution {
    public int[][] allCellsDistOrder(int R, int C, int r0, int c0) {
        List<Cell> cells = new ArrayList<Cell>();
        for (int i = 0; i < R; i++)
        {
            for (int j = 0; j < C; j++)
            {
                cells.add(new Cell(i, j, 
Math.abs(i - r0) + Math.abs(j - c0)));
            }
        }
        Collections.sort(cells, (x, y) -> Integer.compare(x.dist, y.dist));
        
        int[][] results = new int[cells.size()][2];
        
for (int i = 0; i < results.length; i++)
        {
            results[i][0] = cells.get(i).r;
            results[i][1] = cells.get(i).c;
        }
        
        return results;
    }
    
    class Cell
    {
        public int r;
        public int c;
        public int dist;
        
        public Cell(int r, int c, int dist)
        {
            this.r = r;
            this.c = c;
            this.dist = dist;
        }
    }
}
