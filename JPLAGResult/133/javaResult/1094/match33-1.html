class Solution {
    
    class Cell {
        public int row;
        public int col;
        public int d;
        
        public Cell(int r, int c, int d) {
            row = r;
            col = c;
            this.d = d;
        }
    }
    
    
public int[][] allCellsDistOrder(int R, int C, int r0, int c0) {
        ArrayList<Cell> a = new ArrayList<>();
        for (int i = 0; i < R; ++i) {
            for (int j = 0; j < C; ++j) {
                a.add(new Cell(i, j, 
Math.abs(i - r0) + Math.abs(j - c0)));
            }
        }
        
        Collections.sort(a, (c1, c2) -> Integer.valueOf(c1.d).compareTo(Integer.valueOf(c2.d)));
        int[][] res = new int[R * C][2];
        for (int i = 0; i < R * C; ++i) {
            Cell c = a.get(i);
            
res[i] = new int[] {c.row, c.col};
        }
        
        return res;
    }
    
}
