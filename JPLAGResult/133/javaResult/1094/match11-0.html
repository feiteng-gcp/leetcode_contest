class Solution {
    public int[][] allCellsDistOrder(int R, int C, int r0, int c0) {
        int[][] ans = new int[R * C][2];
        int i = 0;
        for (int r = 0; r < R; r++) {
            for (int c = 0; c < C; c++) {
                ans[i][0] = r;
                ans[i][1] = c;
                i++;
            }
        }
        Arrays.sort(ans, (o1, o2) -> 
                    (Math.abs(o1[0] - r0) + Math.abs(o1[1] - c0)) - (Math.abs(o2[0] - r0) + Math.abs(o2[1] - c0))
                   );
        return ans;
    }
}
