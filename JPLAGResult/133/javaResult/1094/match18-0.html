class Solution {
    public int[][] allCellsDistOrder(int R, int C, int r0, int c0) {
        int[][] matrix = new int[R*C][2];
        for(int i = 0; i < R; i++){
            for(int j = 0; j < C; j++){
                matrix[C*i + j][0] = i;
                matrix[C*i + j][1] = j;
            }
        }
        Arrays.sort(matrix, (a, b) -> (Math.abs(a[0] - r0) + Math.abs(a[1] - c0)) - (Math.abs(b[0] - r0) + Math.abs(b[1] - c0)));
        return matrix;
    }
}
