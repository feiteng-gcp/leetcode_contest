class Solution {
    public int[][] allCellsDistOrder(int R, int C, int r0, int c0) {
        final Queue<int[]> points = new PriorityQueue<>(Comparator.comparingInt(
                a -> Math.abs(a[0] - r0) + Math.abs(a[1] - c0)));
        
        for(int i = 0; i < R; i++) {
            for(int j = 0; j < C; j++) {
                points.offer(new int[]{i, j});
            }
        }
        
        int[][] res = new int[R * C][2];
        
        for(int i = 0, end = R * C; i < end; i++) {
            res[i] = points.poll();
        }
        
        return res;
    }
}
