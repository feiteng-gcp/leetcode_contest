class Solution {
    public int[][] allCellsDistOrder(int R, int C, int r0, int c0) {
        Integer[][] a = new Integer[R*C][2];
        
int k = 0;
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                a[k++] = new Integer[] {i, j};
            }
        }
        Arrays.sort(a, (
Integer[] c, Integer[] b) -> { return Math.abs(c[0] - r0) + Math.abs(c[1] - c0) - Math.abs(b[0] - r0) - Math.abs(b[1] - c0);});
        int res[][] = new int[R*C][2];
       for (int i = 0; i < R*C; i++) {
                res[i][0] = a[i][0];
                res[i][1] = a[i][1];
        } 
        return res;
    }
}
