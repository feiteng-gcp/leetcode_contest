class StreamChecker {
    
class Trie {
        boolean is_end;
        Trie[] child;
        Trie() {
            is_end = false;
            child = new Trie[26];
        }
    }
    
Trie parent;
    List<Integer> L;
    public StreamChecker(String[] words) {
        parent = new Trie();
        L = new ArrayList();
        for (int i = 0; i < words.length; i++) {
            char[] word = words[i].toCharArray();
            Trie cur = parent;
            for (int j = word.length - 1; j >= 0; j--) {
                int c = word[j] - 'a';
                
if (cur.child[c] == null) {
                    cur.child[c] = new Trie();
                }
                cur = cur.child[c];
            }
            cur.is_end = true;
        }
    }
    
    public boolean query(char letter) {
        L.add(letter - 'a');
        Trie cur = parent;
        int p = L.size() - 1;
        while (p >= 0 && cur!= null) {
            cur = cur.child[L.get(p--)];
            if (cur != null && cur.is_end)
                return true;
        }
        return false;
    }
}
