class StreamChecker {
    node root;
    int maxL;
    Deque<Character> pool;
    public StreamChecker(String[] words) {
        this.root = new node();
        this.maxL = 0;
        this.pool = new LinkedList<>();
        for (String w : words) addToTrie(w, root);
    }
    private void addToTrie(String s, node root) {
        node p = root;
        int len = s.length();
        char[] sc = s.toCharArray();
        
for (int i = len-1; i>= 0; i--) {
            int index = (int)(sc[i]-'a');
            if (p.next[index] == null) p.next[index] = new node();
            p = p.next[index];
        }
        p.isWord = true;
        this.maxL = Math.max(maxL, len);
    }
    public boolean query(char letter) {
        pool.offerLast(letter);
        if (pool.size() > maxL) pool.pollFirst();
        Stack<Character> temp = new Stack<>();
        node p = root;
        boolean re = false;
        while (!pool.isEmpty() && p.next[pool.peekLast()-'a'] != null) {
            char cur = pool.pollLast();
            temp.push(cur);
            p = p.next[cur-'a'];
            if (p.isWord) {
                re = true;
                break;
            }
        }
        while (!temp.isEmpty()) pool.offerLast(temp.pop());
        
return re;
    }
    class node {
        node[] next;
        boolean isWord;
        public node() {
            this.next = new node[26];
            this.isWord = false;
        }
    }
}
