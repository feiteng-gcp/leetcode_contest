class StreamChecker {
    TrieNode root;
    StringBuilder sb;
    public StreamChecker(String[] words) {
        root = new TrieNode();
        for(String s: words){
            addWord(root, s);
        }
        sb = new StringBuilder();
    }
    
    public boolean query(char letter) {
        sb.append(letter);
        return check(root, sb.toString());
    }
    
    public boolean check(TrieNode root, String str){
        for(int i = str.length() - 1; i >= 0; i--){
            int idx = str.charAt(i) - 'a';
            if(root.next[idx] == null)
                return false
;
            root = root.next[idx];
            if(root.isWord)
                return true;
        }
        return false;
    }
    
    public void addWord(TrieNode root, String str){
        for(int i = str.length() - 1; i >= 0; i--){
            int idx = str.charAt(i) - 'a';
            if(root.next[idx] == null)
                root.next[idx] = new TrieNode();
            root = root.next[idx];
        }
        root.isWord = true;
    
}
}

class TrieNode{
    TrieNode[] next;
    boolean isWord;
    public TrieNode(){
        next = new TrieNode[26];
        isWord = false;
    }
}
