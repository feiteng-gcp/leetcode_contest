class StreamChecker {
    private class Node {
        boolean isWord = false;
        Node[] nodeArray = new Node[26];
    }
    private Node[] root = new Node[26];
    private char[] charArray = new char[40000];
    private int index = 0;

    public StreamChecker(String[] words) {
        for (String w : words) {
            Node[] curr = root;
            Node currNode = null;
            for (int i = w.length() - 1; i >= 0; --i) {
                char c = w.charAt(i);
                if (curr[c - 'a'] == null) {
                    curr[c - 'a'] = new Node();
                }
                currNode = curr[c - 'a'];
                curr = currNode.nodeArray;
            }
            currNode.isWord = true;
        }
    }
    
    public boolean query(char letter) {
        charArray[index] = letter;
        index += 1;
        Node[] curr = root;
        for (int i = index - 1; i >= 0; --i) {
            if (curr[charArray[i] - 'a'] == null) {
                return false;
            }
            if (curr[charArray[i] - 'a'].isWord) {
                return true;
            }
            curr = curr[charArray[i] - 'a'].nodeArray;
        }
        System.out.println("should not reach");
        return false;
    }
}

 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker obj = new StreamChecker(words);
 * boolean param_1 = obj.query(letter);
