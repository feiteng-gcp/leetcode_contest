class Node {
    public boolean isWord;
    public Node[] child;
    public Node(boolean isWord) {
        this.isWord = isWord;
        child = new Node[26];
    }
}

class StreamChecker {
    Node suf = new Node(false);
    ArrayList<Character> cache = new ArrayList<>();
    
    public StreamChecker(String[] words) {
        for(String s : words) {
            Node p = suf;
            for(int i = s.length() - 1; i >= 0; i--) {
                if(p.child[s.charAt(i) - 'a'] == null) {
                    p.child[s.charAt(i) - 'a'] = new Node(false);
                }
                p = p.child[s.charAt(i) - 'a'];
                if(i == 0) p.isWord = true;
            }
        }
    }
    
    public boolean query(char letter) {
        //System.out.println("111");
        cache.add(letter);
        Node p = suf;
        for(int i = cache.size() - 1; i >= 0; i--) {
            if(p.child[cache.get(i) - 'a'] == null) return false;
            p = p.child[cache.get(i) - 'a'];
            if(p.isWord) return true;
        }
        return false;
    }
}

 * boolean param_1 = obj.query(letter);
