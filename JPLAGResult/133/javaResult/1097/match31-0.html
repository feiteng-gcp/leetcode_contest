class StreamChecker {

    List<Character> queries = null;
    Trie trie = null;
    public StreamChecker(String[] words) {
        this.queries = new ArrayList<Character>();
        this.trie = new Trie();
        for(String word : words) {
            trie.insert(word);
        }
    }
    
    public boolean query(char letter) {
        this.queries.add(letter);
        return this.trie.search(this.queries);
    }
    
    class Trie {

    private TreeNode root;
    /** Initialize your data structure here. */
    public Trie() {
        // do intialization if necessary
        root = new TreeNode(null, false);
    }
    
    /** Inserts a word into the trie. */
    public void insert(String word) {
       // write your code here
        TreeNode curNode = root;
        for(int i = word.length() - 1; i > -1; --i) {
            char cur = word.charAt(i);
            if(! curNode.children.containsKey(cur)) {
                // create new node.
                TreeNode newNode = new TreeNode(cur, false);
                curNode.children.put(cur, newNode);
            }
            
            curNode = curNode.children.get(cur);
            if(i == 0) {
                // mark as word
                curNode.isWord = true;
            }
        }
    }
    
    /** Returns if the word is in the trie. */
    public boolean search(String word) {
       // write your code here
        TreeNode curNode = root;
        for(int i = 0; i < word.length(); ++i) {
            char cur = word.charAt(i);
            if(curNode.children.containsKey(cur)) {
                curNode = curNode.children.get(cur);
            } else {
                return false;
            }
        }
        
        return curNode.isWord;
    }
        
    /** Returns if the word is in the trie. */
    public boolean search(List<Character> queries) {
       // write your code here
        TreeNode curNode = root;
        for(int i = queries.size() - 1; i > -1; --i) {
            char cur = queries.get(i);
            if(curNode.isWord) {
                return true;
            }
            
            if(curNode.children.containsKey(cur)) {
                curNode = curNode.children.get(cur);
            } else {
                return false;
            }
        }
        
        return curNode.isWord;
    }
    
    /** Returns if there is any word in the trie that starts with the given prefix. */
    public boolean startsWith(String prefix) {
        // write your code here
        TreeNode curNode = this.root;
        for(int i = 0; i < prefix.length(); ++i) {
            char cur = prefix.charAt(i);
            if(curNode.children.containsKey(cur)) {
                curNode = curNode.children.get(cur);
            } else {
                return false;
            }  
        }
        
        return curNode.isWord || curNode.children.size() > 0;
    }
    
    public class TreeNode {
        public Character character;
        public boolean isWord;
        public HashMap<Character, TreeNode> children;
        
        public TreeNode(Character c, boolean isWord) {
            this.character = c;
            this.isWord = isWord;
            this.children = new HashMap<>();
        }
    }
}
}
