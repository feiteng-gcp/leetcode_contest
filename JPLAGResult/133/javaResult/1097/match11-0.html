class StreamChecker {
    static class Trie {
        Node root = new Node();
        
        void addWord(String word) {
            Node current = root;
            for (char c : word.toCharArray()) {
                if (current.next[c-'a'] == null) {
                    current.next[c-'a'] = new Node();
                }
                current = current.next[c-'a'];
            }
            current.word = true;
        }
        
        Node query(Node from, char c) {
            return from.next[c-'a'];
        }
    }
    static class Node {
        Node[] next = new Node[26];
        boolean word;
    }
    
    private Trie trie;
    private List<Node> currents;

    public StreamChecker(String[] words) {
        trie = new Trie();
        currents = new LinkedList<>();
        currents.add(trie.root);
        
        for (String word : words) {
            trie.addWord(word);
        }
    }
    
    public boolean query(char letter) {
        List<Node> nexts = new LinkedList<>();
        nexts.add(trie.root);
        
        boolean found = false;
        for (Node node : currents) {
            Node next = trie.query(node, letter);
            if (next != null) {
                nexts.add(next);
                if (next.word) found = true;
            }
        }
        
        currents = nexts;
        return found;
    }
}
