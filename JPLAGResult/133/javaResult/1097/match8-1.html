class Node {
    public Node[] children;
    public boolean isWord;
    
    public Node() {
        children = new Node[26];
        isWord = false;
    }
    
    public void addWord(String s, int index) {
        if (index == s.length()) {
            isWord=true;
        } else {
            int tmp = s.charAt(index) - 'a';
            if (children[tmp]==null)
                children[tmp]=new Node();
            children[tmp].addWord(s, index+1);
        }
    
}
    
    public Node advance(char c) {
        return children[c-'a'];
    }
}



class StreamChecker {
    
Node trie;
    ArrayList<Node> cand;
    
    public StreamChecker(String[] words) {
        trie = new Node();
        for (String w : words)
            trie.addWord(w, 0)
;
        cand = new ArrayList<Node>();
    }
    
    public boolean query(char letter) {
        cand.add(trie);
        
        boolean res = false;
        ArrayList<Node> nextCand = new ArrayList<Node>();
        
        for (Node n : cand) {
            Node adv = n.advance(letter);
            if (adv!=null) {
                if (adv.isWord)
                    res=true
;
                nextCand.add(adv);
            }
        }
        
        cand = nextCand;
        
        return res;
    }
}
