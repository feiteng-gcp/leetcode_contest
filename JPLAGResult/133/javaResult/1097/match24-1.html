class StreamChecker
{
    class Tire
    {
        public Tire[] next;
        public boolean IsWord;

        public Tire()
        {
            this.next = new Tire[26];
        }
    }

    private Tire root = new Tire();
    
private Stack<Character> history = new Stack<>();

    public StreamChecker(String[] input)
    {
        for (String w : input)
        {
            Tire node = root;
            for (int i = w.length() - 1; i >= 0; i--)
            {
                int cur = w.charAt(i) - 'a';
                if (node.next[cur] == null) node.next[cur] = new Tire();
                node = node.next[cur];
            }
            node.IsWord = true;
        }
    }

    
public boolean query(char letter)
    {
        history.push(letter);
        Tire node = root;
        for (int i = history.size() - 1; i >= 0 && node != null; i--)
        {
            int cur = history.get(i) - 'a';
            node = node.next[cur];
            
if (node != null && node.IsWord) return true;
        }
        return false;
    }
}
