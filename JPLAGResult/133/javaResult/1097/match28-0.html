class StreamChecker {
    
    private Set<Character> charSet;
    private Trie trie;
    private int MAX_LEN;
    private Deque<Character> queryQueue;

    public StreamChecker(String[] words) {
        this.charSet = new HashSet();
        this.trie = new Trie();
        this.MAX_LEN = 0;
        this.queryQueue = new LinkedList();
        for (String w : words) {
            MAX_LEN = Math.max(MAX_LEN, w.length());
            Trie node = trie;
            for (int i = w.length() - 1; i >= 0; i--) {
                char c = w.charAt(i);
                if (node.children[c - 'a'] == null) {
                    node.children[c - 'a'] = new Trie();
                }
                node = node.children[c - 'a'];
            }
            node.isEnd = true;
        }
    }
    
    public boolean query(char letter) {
        queryQueue.offerLast(letter);
        if (queryQueue.size() > MAX_LEN) {
            queryQueue.pollFirst();
        }
        Stack<Character> tmp = new Stack();
        boolean found = false;
        Trie node = trie;
        while (!queryQueue.isEmpty()) {
            char c = queryQueue.pollLast();
            tmp.push(c);
            if (node.children[c - 'a'] == null) {
                break;
            } else {
                node = node.children[c - 'a'];
                if (node.isEnd) {
                    found = node.isEnd;
                    break;
                }
            }
        }
        while (!tmp.isEmpty()) {
            queryQueue.offerLast(tmp.pop());
        }
        return found;
    }
    
    class Trie {
        boolean isEnd;
        Trie[] children;
        Trie() {
            this.isEnd = false;
            this.children = new Trie[26];
        }
    }
}
