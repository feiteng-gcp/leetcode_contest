class StreamChecker {
    Node root;
    int maxLen;
    ArrayList<Character> list;
    public StreamChecker(String[] words) {
        root = new Node(); maxLen = 0;list = new ArrayList<>();
        for(String w:words){
            put(w.toCharArray());
            maxLen = Math.max(maxLen,w.length());
        }
    }
    
    public boolean query(char letter) {
        list.add(letter);
        int n = list.size();
        Node cur = root;
        for(int i=n-1;i>=0;i--){
            if(n-i>maxLen) return false;
            int p = list.get(i)-'a';
            if(cur.child[p]==null) return false;
            cur = cur.child[p];
            if(cur.isWord) return true;
        }
        return false;
    }
    class Node{
        Node[] child;
        boolean isWord;
        public Node(){
            child = new Node[26]; isWord = false;
        }
    }
    void put(char[] s){
        int n = s.length;
        Node cur = root;
        for(int i=n-1;i>=0;i--){
            int p = s[i]-'a';
            if(cur.child[p]==null) cur.child[p] = new Node();
            cur = cur.child[p];
        }
        cur.isWord = true;
    }
}

 * Your StreamChecker object will be instantiated and called as such:
 * StreamChecker obj = new StreamChecker(words);
 * boolean param_1 = obj.query(letter);
