class StreamChecker {

    class TrieNode {
        boolean isWord = false;
        TrieNode[] next = new TrieNode[26];
        public TrieNode() {
            
        
}
    }
    
    TrieNode root;
    List<Character> memo;
    public StreamChecker(String[] words) {
        root = new TrieNode();
        for (String word : words) {
            StringBuilder sb = new StringBuilder(word);
            char[] c = sb.reverse().toString().toCharArray();
            
add(c);
        }
        memo = new ArrayList<>();
    }
    
    public boolean query(char letter) {
        memo.add(0, letter);
        return check(memo);
    }
    private void add(char[] c) {
        // System.out.println(new String(c));
        TrieNode cur = root;
        for (int i = 0; i < c.length; i++) {
            if (cur.next[c[i] - 'a'] == null) cur.next[c[i] - 'a'] = new TrieNode()
;
            cur = cur.next[c[i] - 'a'];
        }
        cur.isWord = true;
    }
    private boolean check(List<Character> memo) {
        TrieNode cur = root;
        for (char c : memo) {
            if (cur.next[c - 'a'] == null) return false;
            if (cur.next[c - 'a'].isWord) return true;
            else cur = cur.next[c - 'a'];
        }
        return false;
    }
}
