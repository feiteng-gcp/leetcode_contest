class StreamChecker {
    int idx = 0;
    char[] ch = new char[40000];
    class TrieNode {
        boolean hasword;
        TrieNode[] children;
        public TrieNode() {
            children = new TrieNode[26];
            hasword = false;
        }
    }
    TrieNode root;
    public StreamChecker(String[] words) {
        root = new TrieNode();
        
for(int i = 0; i < words.length; i ++) {
            String cur = words[i];
            TrieNode node = root;
            for(int j = cur.length() - 1; j >= 0; j --) {
                char c = cur.charAt(j);
                if(node.children[c - 'a'] == null) {
                    node.children[c - 'a'] = new TrieNode();
                }
                
node = node.children[c - 'a'];
            }
            node.hasword = true;
        }
    }
    // int num = 0;
    public boolean query(char letter) {
        // System.out.println(num);
        // num ++;
        // System.out.println(idx);
        ch[idx] = letter;
        TrieNode cur = root;
        for(int i = idx; i >= 0; i --) {
            if(cur.children[ch[i] - 'a'] == null) {
                // System.out.println(false);
                idx ++;
                return false;   
            }
            cur = cur.children[ch[i] - 'a'];
            if(cur.hasword) {
                // System.out.println(true);
                idx ++;
                return true;   
            }
        }
        idx ++;
        // System.out.println(false);
        return false;
    }
}
