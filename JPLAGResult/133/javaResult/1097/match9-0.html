class StreamChecker {
    class Node{
        Node[] children;
        boolean isWord = false;
        
        Node(){
            this.children = new Node[26];
            this.isWord = false;
        }
        
        void insert(String s){
            Node tmp = this;
            for(char c : s.toCharArray()){
                if(tmp.children[c-'a'] == null){
                    tmp.children[c-'a'] = new Node();
                }
                tmp = tmp.children[c-'a'];
            }
            tmp.isWord = true;
        }
        
        Node search(char c){
            Node tmp = this;
            if(tmp.children[c-'a'] == null)return null;
            tmp = tmp.children[c-'a'];
            return tmp;
        }
    }
    public Node root;
    public StringBuffer sb;
    public int max;
    public List<Node> list;
    public StreamChecker(String[] words) {
        list = new ArrayList<>();
        root = new Node();
        for(String s : words){
            root.insert(s);
        }
    }
    
    public boolean query(char letter) {
        this.list.add(root);
        boolean ans = false;
        List<Node> tmp = new ArrayList<>();
        for(Node x : this.list){
            Node newX = x.search(letter);
            if(newX != null){
                ans |= newX.isWord;
                tmp.add(newX);
            }
        }
        this.list = tmp;
        return ans;
    }
}
