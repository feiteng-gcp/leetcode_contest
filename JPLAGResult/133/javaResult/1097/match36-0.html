class StreamChecker {
    private class Trie {
        TrieNode root;
        public Trie() {
            root = new TrieNode();
        }
        
        public void insert(String word) {
            TrieNode cur = root;
            for (int i = word.length() - 1; i >= 0; i--) {
                char c = word.charAt(i);
                if (cur.children[c - 'a'] == null) {
                    cur.children[c - 'a'] = new TrieNode();
                }
                
                cur = cur.children[c - 'a'];
            }
            
            cur.isWord = true;
        }
        
        public boolean search(StringBuilder sb) {
            TrieNode cur = root;
            for (int i = sb.length() - 1; i >= 0; i--) {
                char c = sb.charAt(i);
                if (cur.children[c - 'a'] == null) return false;
                if (cur.children[c - 'a'].isWord) return true;
                // if (word.equals("ef")) System.out.println(c);
                cur = cur.children[c - 'a'];
            }
            
            return false;
        }
    }
    
    private class TrieNode {
        boolean isWord;
        TrieNode[] children;
        public TrieNode() {
            this.isWord = false;
            children = new TrieNode[26];
        }
    }

    private StringBuilder sb;
    private int maxLen = 0;
    private Trie trie;
    public StreamChecker(String[] words) {
        trie = new Trie();
        for (String word : words) {
            maxLen = Math.max(maxLen, word.length());
            trie.insert(word);
        }
        
        // System.out.println(trie.search("f"));
        sb = new StringBuilder();
    }
    
    public boolean query(char letter) {
        if (sb.length() == maxLen) {
            sb.deleteCharAt(0);
        }
        
        sb.append(letter);
        // if (letter == 'f') System.out.println(sb.toString());
        return trie.search(sb);
    }
}
