class StreamChecker {
    
    class Trie {
        private Map<Character, Trie> children;
        private boolean ends;
        
        public Trie() {
            children = new HashMap<>();
            ends = false;
        }
        
        public void insert(String s) {
            insert(s, 0);
        }
        
        private void insert(String s, int i) {
            if (i == s.length()) {
                ends = true;
            } else {
                char c = s.charAt(i);
                children.putIfAbsent(c, new Trie());
                children.get(c).insert(s, i + 1);
            }
        }
        
        public boolean anyEnds(StringBuilder s) {
            return anyEnds(s, 0);
        }
        
        private boolean anyEnds(StringBuilder s, int i) {
            if (ends) {
                return true;
            }
            if (i == s.length()) {
                return false;
            }
            char c = s.charAt(sb.length() - 1 - i);
            
            if (!children.containsKey(c)) {
                return false;
            }
            
            return children.get(c).anyEnds(s, i + 1);
        }
    }

    private Trie t;
    private StringBuilder sb;
    
    public StreamChecker(String[] words) {
        t = new Trie();
        for (String s : words) {
            t.insert(new StringBuilder(s).reverse().toString());
        }
        
        sb = new StringBuilder();
    }
    
    public boolean query(char letter) {
        sb.append(letter);
        
        return t.anyEnds(sb);
    }
}
