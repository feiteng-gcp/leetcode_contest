    class StreamChecker {

        private TrieNode root;
        
private StringBuilder stringBuilder;

        public StreamChecker(String[] words) {
            this.root = new TrieNode();
            this.stringBuilder = new StringBuilder();
            for (String word : words) {
                TrieNode trieNode = root;
                for (int i = word.length() - 1; i >= 0; i--) {
                    int index = word.charAt(i) - 'a';
                    if (Objects.isNull(trieNode.children[index])) {
                        trieNode.children[index] = new TrieNode();
                    
}
                    trieNode = trieNode.children[index];
                }
                trieNode.isWord = true;
            }
        }

        public boolean query(char letter) {
            stringBuilder.append(letter);
            TrieNode trieNode = root;
            for (int i = stringBuilder.length() - 1; i >= 0; i--) {
                trieNode = trieNode.children[stringBuilder.charAt(i) - 'a'];
                if (Objects.isNull(trieNode)) {
                    return false;
                } else if (trieNode.isWord) {
                    return true;
                }
            }
            return false;
        }

        class TrieNode {
            boolean isWord;
            TrieNode[] children;

            TrieNode() {
                this.isWord = false;
                this.children = new TrieNode[26];
            }
        }
    }
