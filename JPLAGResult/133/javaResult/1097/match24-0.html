class StreamChecker {
    
    class Node {
        char val;
        boolean start;
        Map<Character, Node> prev = new HashMap<>();
        
        Node(char val, boolean start) {
            this.val = val;
            this.start = start;
        }
    }
    
    private Node root = new Node('0', false);

    public StreamChecker(String[] words) {
        for (String word : words) {
            Node cur = root;
            for (int i = word.length() - 1; i >= 0; i--) {
                char c = word.charAt(i);
                Node node = new Node(c, false);
                if (!cur.prev.containsKey(c)) cur.prev.put(c, node);
                cur = cur.prev.get(c);
                if (i == 0) cur.start = true;
            }
        }
    }
    
    private List<Character> chars = new ArrayList<>();
    
    public boolean query(char letter) {
        chars.add(letter);
        Node cur = root;
        for (int i = chars.size() - 1; i >= 0; i--) {
            char c = chars.get(i);
            if (!cur.prev.containsKey(c)) return false;
            cur = cur.prev.get(c);
            if (cur.start) return true;
        }
        
        return false;
    }
}
