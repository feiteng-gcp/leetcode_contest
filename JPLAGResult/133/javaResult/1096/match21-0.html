class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int[] sums = new int[A.length+1];
        for(int i=0; i<A.length; i++){
            sums[i+1] = sums[i]+A[i];
        }
        int res = 0;
        for(int i=L; i<sums.length; i++){
            int j=i-L;
            res = Math.max(res, sums[i]-sums[j]+maxM(M, sums, j, i));
        }
        return res;
    }
    private int maxM(int M, int[] sums, int s, int e){
        int res = 0;
        for(int i=M; i<=s; i++){
            res = Math.max(res, sums[i]-sums[i-M]);
        }
        for(int i=e+M; i<sums.length; i++){
            res = Math.max(res, sums[i]-sums[i-M]);
        }
        return res;
    }
}
