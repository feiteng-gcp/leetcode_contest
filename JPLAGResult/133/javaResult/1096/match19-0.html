class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int n = A.length;
        if(n == 0) return 0;
        
        int[] sum = new int[n + 1];
        for(int i = 1; i <= n; i++) {
            sum[i] = sum[i - 1] + A[i - 1];
        }
        int res = 0;
        for(int i = 0; i + L + M - 1 < n; i++) {
            int sum1 = sum[i + L] - sum[i];
            for(int j = i + L; j + M - 1 < n; j++) {
                int sum2 = sum[j + M] - sum[j];
                res = Math.max(sum1 + sum2, res);
            }
        }
        
        for(int i = 0; i + L + M - 1 < n; i++) {
            int sum1 = sum[i + M] - sum[i];
            for(int j = i + M; j + L - 1 < n; j++) {
                int sum2 = sum[j + L] - sum[j];
                res = Math.max(sum1 + sum2, res);
            }
        }
        
        return res;
    }
}
