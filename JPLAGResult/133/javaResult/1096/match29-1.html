class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
		int leng = A.length;
		int[] lsum = new int[leng - L + 1];
		int[] msum = new int[leng - M + 1];
		int ls = 0;
		for (int i = 0; i < L; i++) {
			ls += A[i];
		}
		lsum[0] = ls;
		for (int i = L; i < leng; i++) {
			ls = ls + A[i] - A[i - L];
			lsum[i - L + 1] = ls;
		}
		int ms = 0;
		for (int i = 0; i < M; i++) {
			ms += A[i];
		}
		msum[0] = ms;
		for (int i = M; i < leng; i++) {
			ms = ms + A[i] - A[i - M];
			msum[i - M + 1] = ms;
		}
		int max = 0;
		for (int i = 0
; i < lsum.length; i++) {
			for (int j = 0; j < msum.length; j++) {
				if (i + L <= j || j + M <= i) {
					max = Math.max(max, lsum[i] + msum[j]);
				}

			}
		}
		return max;
    }
}
