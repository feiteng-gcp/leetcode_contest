class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int len = A.length;
        int[] prefix = new int[len];
        
        for (int i = 0; i < len; ++i) {
            if (i == 0) prefix[i] = A[i];
            else prefix[i] = prefix[i - 1] + A[i];
        }
        
        int res = 0;
        
        for (int i = L - 1; i < len; ++i) {
            for (int j = i + M; j < len; ++j) {
                int sum = (i == L - 1 ? prefix[i] : prefix[i] - prefix[i - L]) + 
                          (prefix[j] - prefix[j - M]);
                
res = Math.max(sum, res);
            }
        }
        
        for (int i = M - 1; i < len; ++i) {
            for (int j = i + L; j < len; ++j) {
                int sum = (i == M - 1 ? prefix[i] : prefix[i] - prefix[i - M]) + 
                          (prefix[j] - prefix[j - L]);
                
res = Math.max(sum, res);
            }
        }
        
        return res;
    }
}
