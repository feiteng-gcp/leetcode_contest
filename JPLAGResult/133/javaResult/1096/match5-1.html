class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int sum[] = new int[A.length+1],max = 0;
        for(int i=0;i<A.length;i++){
            sum[i+1] = sum[i] + A[i];
        }
        for(int k=0;k<=A.length-L;k++){
            int sum1 = sum[k+L]-sum[k];
            if(k>=M)
{
                for(int i=0;i<=k-M;i++){
                    max = Math.max(max,sum[i+M]-sum[i] + sum1);
                }
            }
            if(A.length-k>=M)
{
                for(int i=k+L;i<=A.length-M;i++){
                    max = Math.max(max,sum[i+M]-sum[i] + sum1);
                }
            }
        }
        return max;
    }
   
}
