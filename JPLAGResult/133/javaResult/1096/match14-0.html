class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        if(A == null || A.length == 0 || L < 0 || M < 0 || L + M > A.length) {
            return 0;
        }
        
        int[] prefixSum = new int[A.length + 1];
        for(int i = 0; i < A.length; ++i) {
            prefixSum[i+1] = prefixSum[i] + A[i];
        }
        
        int maxSum = 0;
        for(int i = L; i < prefixSum.length - M; ++i) {
            for(int j = i + M; j < prefixSum.length; ++j) {
                if(prefixSum[i] - prefixSum[i-L] + prefixSum[j] - prefixSum[j-M] > maxSum) {
                    maxSum = prefixSum[i] - prefixSum[i-L] + prefixSum[j] - prefixSum[j-M];
                }
            }
        }
        
        for(int i = M; i < prefixSum.length - L; ++i) {
            for(int j = i + L; j < prefixSum.length; ++j) {
                if(prefixSum[i] - prefixSum[i-M] + prefixSum[j] - prefixSum[j-L] > maxSum) {
                    maxSum = prefixSum[i] - prefixSum[i-M] + prefixSum[j] - prefixSum[j-L];
                }
            }
        }
        
        return maxSum;
    }
}
