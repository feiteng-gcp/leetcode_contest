class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int sum=0;
        
for(int i=0;i<A.length;i++){
            sum+=A[i];
            A[i]=sum;
        }
        int ans=0;
        for(int i=L-1;i<A.length;i++){
            for(int j=i+M;j<A.length;j++){
                if(i==L-1)
                    ans=Math.max(ans,A[L-1]+A[j]-A[j-M]);
                else
                    ans=Math.max(ans,A[i]-A[i-L]+A[j]-A[j-M]);
            }
        }
        if(L==M)
            return ans;
        for(int i=M-1;i<A.length;i++){
            for(int j=i+L;j<A.length;j++){
                if(i==M-1)
                    ans=Math.max(ans,A[M-1]+A[j]-A[j-L]);
                else
                    ans=Math.max(ans,A[i]-A[i-M]+A[j]-A[j-L])
;
            }
        }
        return ans;
    }
}
