class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int total = 0;
        int N = A.length;
        int[] LA = new int[N];
        int[] MA = new int[N];
        int curL = 0;
        int curM = 0;
        for (int i = 0; i < N; i++) {
            curL += A[i];
            curM += A[i];
            if (i >= L) {
                curL -= A[i - L];
            }
            if (i >= M) {
                curM -= A[i - M];
            }
            LA[i] = curL;
            MA[i] = curM;
        }
        int max = 0;
        for (int i = L - 1; i < N; i++) {
            for (int j = i + M; j < N; j++) {
                if (LA[i] + MA[j] > max) {
                    max = LA[i] + MA[j];
                }
            }
        }
        for (int i = M - 1; i < N; i++) {
            for (int j = i + L; j < N; j++) {
                if (MA[i] + LA[j] > max) {
                    max = MA[i] + LA[j];
                }
            }
        }
        return max;
    }
}
