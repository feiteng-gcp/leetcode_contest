class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int[] maxArrSum_L_start = getMaxArrSumOfLength_Start(A, L);
        int[] maxArrSum_L_end = getMaxArrSumOfLength_End(A, L);
        
        int[] maxArrSum_M_start = (L == M? maxArrSum_L_start: getMaxArrSumOfLength_Start(A, M));
        int[] maxArrSum_M_end = (L == M? maxArrSum_L_end: getMaxArrSumOfLength_End(A, M));
        
        int result = 0;
        for (int i = L - 1; i + 1 + M - 1 < A.length; i++)
        {
            result = Math.max(result, maxArrSum_L_end[i] + maxArrSum_M_start[i + 1]);
        }
        
        for (int i = M - 1; i + 1 + L - 1 < A.length; i++)
        {
            result = Math.max(result, maxArrSum_M_end[i] + maxArrSum_L_start[i + 1]);
        }
        
        return result;
    }
    
    private int[] getMaxArrSumOfLength_End(int[] A, int len)
    {
        int[] results = new int[A.length];
        
        int sum = 0;
        for (int i = 0; i < len; i++)
        {
            sum += A[i];
        }
        
        Arrays.fill(results, 0);
        int j = 0;
        while (true)
        {
            // sum is for [j, j + len - 1]
            results[j + len - 1] = (j == 0? sum: Math.max(sum, results[j + len - 2]));
            
            j++;
            if (j + len - 1>= A.length)
            {
                break;
            }
            
            sum -= A[j - 1];
            sum += A[j + len - 1];
        }
        
        return results;
    }
    
    private int[] getMaxArrSumOfLength_Start(int[] A, int len)
    {
        int[] results = new int[A.length];
        
        int sum = 0;
        for (int i = A.length - 1; i >= A.length - len; i--)
        {
            sum += A[i];
        }
        
        Arrays.fill(results, 0);
        int j = A.length - 1;
        while (true)
        {
            // sum is for [j - len + 1, j]
            results[j - len + 1] = (j == A.length - 1? sum: Math.max(sum, results[j - len + 2]));
            
            j--;
            if (j - len + 1 < 0)
            {
                break;
            }
            sum -= A[j + 1];
            sum += A[j - len + 1];
        }
        
        return results;
    }
}
