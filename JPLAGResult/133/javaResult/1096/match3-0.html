class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int[] rangeSum = new int[A.length+1];
        rangeSum[0]=0;
        for(int i = 1; i < rangeSum.length; i++) {
            rangeSum[i]=rangeSum[i-1]+A[i-1];
        }
        int max = 0;
        for(int i = 0; (i+L+M) <= A.length; i++) {
            for(int j = i+L; (j+M) <= A.length; j++) {
                int leftSum = rangeSum[i+L]-rangeSum[i];
                int rightSum = rangeSum[j+M]-rangeSum[j];
                max = Math.max(max,leftSum+rightSum);
            }
        }
        for(int i = 0; (i+L+M) <= A.length; i++) {
            for(int j = i+M; (j+L) <= A.length; j++) {
                int leftSum = rangeSum[i+M]-rangeSum[i];
                int rightSum = rangeSum[j+L]-rangeSum[j];
                max = Math.max(max,leftSum+rightSum);
            }
        }
        return max;
    }
}
