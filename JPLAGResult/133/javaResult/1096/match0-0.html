class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int n = A.length;
        int[] prefixSum = new int[n + 1];
        for (int i = 1; i <= n; i++) prefixSum[i] = prefixSum[i - 1] + A[i - 1];
        
        int ans = -1;
        for (int i = 0; i <= n - L - M; i++) {
            for (int j = i + L; j <= n - M; j++) {
                ans = Math.max(ans, prefixSum[i + L] - prefixSum[i] + prefixSum[j + M] - prefixSum[j]);
            }
        }
        
        for (int i = 0; i <= n - L - M; i++) {
            for (int j = i + M; j <= n - L; j++) {
                ans = Math.max(ans, prefixSum[i + M] - prefixSum[i] + prefixSum[j + L] - prefixSum[j]);
            }
        }
        return ans;
    }
}
