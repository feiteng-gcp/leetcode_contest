class Solution {
    public int maxSumTwoNoOverlap(int[] a, int l, int m) {
        int n = a.length;
        int[] prefixSum = new int[n + 1];
        prefixSum[0] = 0;
        prefixSum[1] = a[0];
        for (int i = 1; i < n; ++i) {
            prefixSum[i + 1] = prefixSum[i] + a[i];
        }
        
        int maxSum = 0;
        for (int lRight = l - 1; lRight < n; ++lRight) {
            for (int mRight = m - 1; mRight <= lRight - l; ++mRight) {
                maxSum = Math.max(maxSum, prefixSum[mRight + 1] - prefixSum[mRight - m + 1] + prefixSum[lRight + 1] - prefixSum[lRight - l + 1]);
            }
            for (int mRight = lRight + m; mRight < n; ++mRight) {
                maxSum = Math.max(maxSum, prefixSum[mRight + 1] - prefixSum[mRight - m +1] + prefixSum[lRight + 1] - prefixSum[lRight - l + 1]);
            }
        }
        
        return maxSum;
    }
}
