class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int n = A.length;
        int[] leftLMax = new int[n];
        int[] leftMMax = new int[n];
        int[] rightLMax = new int[n];
        int[] rightMMax = new int[n];
        
        int sumL = 0, sumM = 0;
        for (int i = 0; i < n; i++) {
            sumL += A[i];
            sumM += A[i];
            if (i >= L - 1) {
                leftLMax[i] = Math.max(sumL, (i - 1 >= 0) ? leftLMax[i - 1] : 0);
                sumL -= A[i - L + 1];
            }
            
            if (i >= M - 1) {
                leftMMax[i] = Math.max(sumM, (i - 1 >= 0) ? leftMMax[i - 1] : 0);
                sumM -= A[i - M + 1];
            }
        }
        
        sumL = 0;
        sumM = 0;
        for (int i = n - 1; i >= 0; i--) {
            sumL += A[i];
            sumM += A[i];
            if (i <= n - L) {
                rightLMax[i] = Math.max(sumL, (i + 1 < n) ? rightLMax[i + 1] : 0);
                sumL -= A[i + L - 1];
            }
            
            if (i <= n - M) {
                rightMMax[i] = Math.max(sumM, (i + 1 < n) ? rightMMax[i + 1] : 0);
                sumM -= A[i + M - 1];
            }
        }
        
        // for (int i = 0; i < n; i++) {
        //     System.out.print(rightMMax[i]+",");
        // }
        int max = 0;
        for (int i = 0; i < n - M; i++) {
            max = Math.max(max, leftLMax[i] + rightMMax[i + 1]);
        }
        
        for (int i = 0; i < n - L; i++) {
            max = Math.max(max, leftMMax[i] + rightLMax[i + 1]);
        }
        
        return max;
    }
}
