class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int[] sum = new int[A.length+1];
        sum[0] = 0;
        for (int i=1; i<sum.length; i++) {
            sum[i] = sum[i-1] + A[i-1];
        }
        
        return Math.max(getMax(sum, L, M), getMax(sum, M, L));
    }
    
    int getMax(int[] sum, int L, int M) {
        int max = 0;
        for (int i=L; i<sum.length-M; i++) {
            int c1 = sum[i] - sum[i-L];
            for (int j=i+M; j<sum.length; j++) {
                
int c2 = sum[j] - sum[j-M];
                max = Math.max(max, c1 + c2);
            }
        }
        return max;
    }
}
