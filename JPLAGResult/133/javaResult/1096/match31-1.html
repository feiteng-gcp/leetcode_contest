class Solution {
	public int maxSumTwoNoOverlap(int[] A, int L, int M) {
		int N = A.length;
		int[] Ls = new int[N];
		int[] maxLs = new int[N];
		int[] maxLs2 = new int[N];
		int curr = 0;
		for (int i = 0; i < L; i++) {
			curr += A[i];
		}
		for (int i = 0; i + L <= N; i++) {
			Ls[i] = curr;
			curr -= A[i];
			if (i + L < N)
				curr += A[i + L];
		}
		curr = Ls[0];
		for (int i = 0; i + L <= N; i++) {
			curr = Math.max(curr, Ls[i]);
			maxLs[i] = curr;
		}
		curr = Ls[N - L];
		for (int i = N - L; i >= 0; i--) {
			curr = Math.max(curr, Ls[i]);
			maxLs2[i] = curr;
		}
		int[] Ms = new int[N];
		int[] maxMs = new int[N];
		int[] maxMs2 = new int[N];
		curr = 0;
		for (int i = 0; i < M; i++) {
			curr += A[i];
		}
		for (int i = 0; i + M <= N; i++) {
			Ms[i] = curr;
			curr -= A[i];
			if (i + M < N)
				curr += A[i + M];
		}
		curr = Ms[0];
		for (int i = 0; i + M <= N; i++) {
			curr = Math.max(curr, Ms[i]);
			maxMs[i] = curr;
		}
		curr = Ms[N - M];
		for (int i = N - M; i >= 0; i--) {
			curr = Math.max(curr, Ms[i]);
			maxMs2[i] = curr;
		}
		int res = 0;
		for (int i = L
; i + M <= N; i++) {
			res = Math.max(res, maxLs[i - L] + maxMs2[i]);
		}
		for (int i = M; i + L <= N; i++) {
			res = Math.max(res, maxMs[i - M] + maxLs2[i]);
		}
		return res;
	}

	public static void printArray(int[] array) {
		for (int i : array) {
			System.out.print(i + " ");
		}
		System.out.println();
	}
}
