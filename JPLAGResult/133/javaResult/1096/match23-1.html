class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int n = A.length;
        int[] LA = new int[n];
        int[] RA = new int[n];
        int[] curA = new int[n];
        int[] curB = new int[n];
        int sumL = 0, sumR = 0;
        for(int i=0;i<n;i++){
            sumL += A[i];
            sumR += A[i];
            if(i >= L-1){
                LA[i] = Math.max(i-1==-1?0:LA[i-1], sumL);
                curA[i] = sumL;
                sumL -= A[i-L+1];
            }
            if(i >= M-1){
                RA[i] = Math.max(i-1==-1?0:RA[i-1], sumR);
                curB[i] = sumR;
                sumR -= A[i-M+1];
            
}
        }
        int ans = 0;
        for(int i=0;i<n;i++){
            ans = Math.max(ans, 
             (i-M<0?0:LA[i-M]) + curB[i]);
            ans = Math.max(ans, 
             (i-L<0?0:RA[i-L]) + curA[i]);
        }
        
        return ans;
    }
}
