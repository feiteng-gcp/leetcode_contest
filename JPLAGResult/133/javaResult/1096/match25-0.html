class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int n = A.length;
        int[][] dp1 = new int[n + 1][3];
        int[][] dp2 = new int[n + 1][3];
        int[] sum = new int[n + 1];
        for(int i = 1; i <= n; i ++) {
            sum[i] = sum[i - 1] + A[i - 1];
        }
        // l before m
        for(int i = 1; i <= n; i ++) {
            // 0
            if(i >= L) {
                dp1[i][1] = Math.max(dp1[i - L][0] + sum[i] - sum[i - L], dp1[i - 1][1]); 
            }     
        }
        for(int i = 1; i <= n; i ++) {
            if(i >= M) {
                dp1[i][2] = Math.max(dp1[i - M][1] + sum[i] - sum[i - M], dp1[i - 1][2]);
            }
        }
        for(int i = 1; i <= n; i ++) {
            // 0
            if(i >= M) {
                dp2[i][1] = Math.max(dp2[i - M][0] + sum[i] - sum[i - M], dp2[i - 1][1]); 
            }     
        }
        for(int i = 1; i <= n; i ++) {
            if(i >= L) {
                dp2[i][2] = Math.max(dp2[i - L][1] + sum[i] - sum[i - L], dp2[i - 1][2]);
            }
        }
        // System.out.println(dp1[n][2]);
        // System.out.println(dp2[n][2]);
        return Math.max(dp1[n][2], dp2[n][2]);
    }
}
