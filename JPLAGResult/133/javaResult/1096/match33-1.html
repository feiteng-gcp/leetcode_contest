class Subarray {
    int sum;
    int x;
    int y;
    public Subarray(int sum, int x, int y) {
        this.sum = sum; 
        this.x = x;
        this.y = y;
    }
}

class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int[] presum = new int[A.length];
        int s = 0;
        for (int i = 0; i < A.length; i++) {
            s += A[i];
            presum[i] = s;
        }
        int result = 0;
        List<Subarray> lA = findSub(presum, L);
        List<Subarray> lB = findSub(presum, M);
        for (int i = 0; i < lA.size(); i++) {
            for (int j = 0; j < lB.size(); j++) {
                if (!overlap(lA.get(i), lB.get(j))) {
                    result = Math.max(result, lA.get(i).sum + lB.get(j).sum);
                
}
            }
        }
        return result;
    }
    private boolean overlap(Subarray s1, Subarray s2) {
        if (s1.y < s2.x || s2.y < s1.x) {
            return false;
        }
        return true;
    }
    private List<Subarray> findSub(int[] presum, int len) {
        List<Subarray> result = new ArrayList<>();
        result.add(new Subarray(presum[len - 1], 0, len - 1));
        for (int i = len; i < presum.length; i++) {
            Subarray subb = new Subarray(presum[i] - presum[i - len], i - len + 1, i);
            result.add(subb);
        }
        return result;
        
    }
}
