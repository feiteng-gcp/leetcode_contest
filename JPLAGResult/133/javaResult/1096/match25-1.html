class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        if (A == null || A.length < L || A.length < M) {
            return 0;
        }
        
int[] dpL = new int[A.length];
        int[] dpM = new int[A.length];
        int[] prefixSum = new int[A.length + 1];
        for (int i = 1; i < prefixSum.length; i++) {
            prefixSum[i] = prefixSum[i - 1] + A[i - 1];
        }
        // init dp
        for (int i = 0; i < A.length; i++) {
            if (i >= L - 1) {
                dpL[i] = prefixSum[i + 1] - prefixSum[i - L + 1];
            }
            if (i >= M - 1) {
                dpM[i] = prefixSum[i + 1] - prefixSum[i - M + 1];
            }
        }
        int max = 0;
        for (int i = L - 1; i < dpL.length; i++) {
            
for (int j = M - 1; j < dpM.length; j++) {
                if (j - M + 1 > i || j < i - L + 1) {
                    max = Math.max(max, dpL[i] + dpM[j]);
                }
            }
        }
        return max;
    }
}
