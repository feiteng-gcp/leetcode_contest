class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int[] lmax = max(A, L);
        int[] mmax = max(A, M);
        int output = 0;
        int[] sum = new int[A.length + 1];
        for (int i = 1; i <= A.length; i++) {
            sum[i] = sum[i - 1] + A[i - 1];
        }
        
for (int i = L; i + M <= A.length; i++) {
            output = Math.max(output, lmax[i - 1] + sum[i + M] - sum[i]);
        }
        for (int i = M; i + L <= A.length; i++) {
            output = Math.max(output, mmax[i - 1] + sum[i + L] - sum[i]);
        }
        return output;
    }

    private int[] max(int[] A, int L) {
        int[] output = new int[A.length];
        int sum = 0;
        for (int i = 0; i < A.length; i++) {
            sum += A[i];
            if (i >= L - 1) {
                output[i] = Math.max(sum, output[Math.max(0, i - 1)]);
                sum -= A[i - L + 1];
            }
        }
        return output;
    }
}
