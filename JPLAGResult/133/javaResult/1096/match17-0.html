class Solution {
    public int maxSumTwoNoOverlap(int[] A, int L, int M) {
        int res = Integer.MIN_VALUE;
        int[] prefix = new int[A.length + 1];
        for (int i = 1; i <= A.length; ++i) prefix[i] = prefix[i - 1] + A[i - 1]; // [i, j] = p[j + 1] - [i];
        for (int i = 0; i <= A.length - L; ++i) { // L first
            int cur = prefix[i + L] - prefix[i];
            for (int j = i + L; j <= A.length - M; ++j) {
                res = Math.max(cur + prefix[j + M] - prefix[j], res);
            }
        }
        for (int i = 0; i <= A.length - M; ++i) { 
            int cur = prefix[i + M] - prefix[i];
            for (int j = i + M; j <= A.length - L; ++j) {
                res = Math.max(cur + prefix[j + L] - prefix[j], res);
            }
        }
        return res;
    }
}
