class Solution
{
    public int twoCitySchedCost(int[][] costs)
    {
        Arrays.sort(costs, (a, b) ->
        {
            return -Math.abs(a[0] - a[1]) + Math.abs(b[0] - b[1]);
        });
        int a = 0, b = 0;
        int n = costs.length / 2;
        int ans = 0;
        for (int[] c : costs)
        {
            
if (a == n)
            {
                ans += c[1];
            }
            else if (b == n)
            {
                ans += c[0];
            }
            else
            {
                if (c[0] < c[1])
                {
                    ans += c[0];
                    a++;
                }
                else
                {
                    ans += c[1];
                    b++;
                }
            }
        }
        return ans;
    }
}
