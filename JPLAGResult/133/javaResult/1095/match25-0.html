class Solution {
    public int twoCitySchedCost(int[][] costs) {
        Arrays.sort(costs, new Comparator<int[]>(){
            public int compare(int[] a, int[] b){
                return Integer.compare(a[0] - a[1], b[0] - b[1]);
            }
        });
        
        int n = costs.length >> 1;
        int result = 0;
        for(int i = 0; i < n; i++){
            result += costs[i][0];
        }
        
        for(int i = n; i < costs.length; i++){
            result += costs[i][1];
        }
        
        return result;
    }
}
