class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int n = costs.length;
        int[][] dp = new int[n + 1][n + 1];
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j <= i; j++) {
                if (i == j) dp[i][j] = dp[i - 1][j - 1] + costs[i - 1][0];
                else if (j == 0) dp[i][0] = dp[i - 1][0] + costs[i - 1][1];
                else dp[i][j] = Math.min(dp[i - 1][j] + costs[i - 1][1], dp[i - 1][j - 1] + costs[i - 1][0])
;
            }
        }
        return dp[n][n / 2];
    }
    
    // dp[i][j] = i people j in cityA
    // dp[i][j] = min(dp[i - 1][j] + costs[i][1], dp[i - 1][j - 1] + costs[i][0]);
}
