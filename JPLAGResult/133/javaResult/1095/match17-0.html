class Solution {

    public int twoCitySchedCost(int[][] costs) {
		ArrayList<Cost> list = new ArrayList<>();
		for (int[] cost : costs) {
			list.add(new Cost(cost[0], cost[1]));
		}
		Collections.sort(list);
		int result = 0;
		for (int i = 0; i < list.size(); i++) {
			result += i < list.size() / 2 ? list.get(i).costA : list.get(i).costB;
		}
		return result;
	}

	private class Cost implements Comparable<Cost> {
		private int costA, costB;

		private Cost(int costA, int costB) {
			this.costA = costA;
			this.costB = costB;
		}

		public int compareTo(Cost o) {
			return costA - costB - o.costA + o.costB;
		}
	}
}
