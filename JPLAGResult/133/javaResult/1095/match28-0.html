class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int n= costs.length/2;
     Arrays.sort(costs,(a,b)-> Math.abs(b[0]-b[1])-Math.abs(a[0]-a[1]));
        int a=0,b=0;
       int ans=0;
        for(int[] cost : costs){
            if(cost[0]<cost[1]){
                if(a<n){
                    ans+=cost[0];
                    a++;
                }
                else{
                    ans+=cost[1];b++;
                }
            }
            else{
               if(b<n){
                    ans+=cost[1];
                    b++;
                }
                else{
                    ans+=cost[0];a++;
                } 
            }
        }
        return ans;
    }
}
