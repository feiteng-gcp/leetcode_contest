class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int n = costs.length;
        int[][] f = new int[n + 1][n / 2 + 1];
        
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= n / 2; j++) {
                if (i == 0) {
                    if (j != 0) f[i][j] = Integer.MAX_VALUE;
                    continue;
                }
                
                f[i][j] = Integer.MAX_VALUE;
                if (f[i - 1][j] < Integer.MAX_VALUE) f[i][j] = f[i - 1][j] + costs[i - 1][1]
;
                if (j >= 1 && f[i - 1][j - 1] < Integer.MAX_VALUE) f[i][j] = Math.min(f[i - 1][j - 1] + costs[i - 1][0], f[i][j]);
            }
        }

        return f[n][n / 2];
    }
}
