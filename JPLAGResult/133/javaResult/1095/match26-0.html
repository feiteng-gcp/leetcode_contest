class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int[][] unifiedCost = new int[costs.length][2];
        for (int i=0; i<costs.length; i++) {
            unifiedCost[i][0] = i;
            unifiedCost[i][1] = costs[i][0]-costs[i][1];
        }
        Arrays.sort(unifiedCost, (a,b) -> (a[1]-b[1]));
        int totalCost = 0;
        for (int i=0; 2*i<unifiedCost.length; i++){
            totalCost += costs[unifiedCost[i][0]][0];
            totalCost += costs[unifiedCost[unifiedCost.length-1-i][0]][1];
        }
        return totalCost;
    }
}
