class Solution {
    class Com implements Comparator<int[]>{
        public int compare(int[] i1, int[] i2){
            int x1 = Math.abs(i2[0]-i2[1]);
            int x2 = Math.abs(i1[0]-i1[1]);
            if(x1 == x2){
                return  i2[0]!=i1[0] ? i1[0]-i2[0] : i1[1] - i2[1];
            }
            return x1- x2;
        }
    }
    public int twoCitySchedCost(int[][] costs) {
        int ans = 0;
        Arrays.sort(costs, new Com());
        
int n = costs.length;
        int N = n/2;
        int A = 0, B = 0;
        for(int i=0;i<n;i++){
            if(costs[i][0] <= costs[i][1] && A+1<=N){
                ans += costs[i][0];
                A++;
            }else if(costs[i][1] <= costs[i][0] && B+1<=N){
                ans += costs[i][1];
                B++;
            }else if(A+1<=N){
                A++;
                ans += costs[i][0];
            }else{
                B++;
                ans += costs[i][1];
            }
        }
        return ans;
    }
}
