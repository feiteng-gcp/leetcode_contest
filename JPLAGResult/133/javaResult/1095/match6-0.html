class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int res = 0;
        int len = costs.length;
        int record[] = new int[len];
        for(int i = 0; i < len; i++) {
            res += costs[i][0];
            record[i] = costs[i][1] - costs[i][0];
        }
        Arrays.sort(record);
        for(int i = 0; i < len / 2; i++){
            res += record[i];
        }
        return res;
    }
}
