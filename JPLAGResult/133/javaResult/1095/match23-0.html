class Solution {
    public int twoCitySchedCost(int[][] costs) {
        Arrays.sort(costs, (a,b) -> {
            return (a[1]-a[0]) - (b[1]-b[0]);
        });
        
        int len = costs.length / 2;
        int sum = 0;
        for (int i=0; i<len; i++) {
            sum += costs[i][1];
            sum += costs[i+len][0];
        }
        
        return sum;
    }
}
