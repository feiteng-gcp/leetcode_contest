class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int n = costs.length / 2;
        int[][] dp = new int[n + 1][n + 1];

        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= n; j++) {
                int now = i + j - 1;
                dp[i][j] = (i == 0 && j == 0) ? 0 : Integer.MAX_VALUE;
                if (i > 0) {
                    dp[i][j] = Math.min(dp[i][j], dp[i-1][j] + costs[now][0]);
                }
                if (j > 0) {
                    dp[i][j] = Math.min(dp[i][j], dp[i][j-1] + costs[now][1]);
                }
            }
        }
        return dp[n][n];
    }
}
