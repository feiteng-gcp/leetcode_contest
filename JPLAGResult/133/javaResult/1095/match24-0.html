class Solution {
    public int twoCitySchedCost(int[][] costs) {
        Arrays.sort(costs, (int[] cost1, int[] cost2) -> {
            return - Math.abs(cost1[0] - cost1[1]) + Math.abs(cost2[0] - cost2[1]);
        });
        int numA = 0;
        int numB = 0;
        int res = 0;
        int N = costs.length / 2;
        for (int i = 0; i < costs.length; ++i) {
            int[] cost = costs[i];
            if (numA < N && numB < N) {
                if (cost[0] < cost[1]) {
                    numA++;
                    res += cost[0];
                } else if (cost[0] > cost[1]) {
                    numB++;
                    res += cost[1];
                } else {
                    numA++;
                    res += cost[0];
                }
            } else if (numA == N) {
                res += cost[1];
            } else if (numB == N) {
                res += cost[0];
            }
            // System.out.println(Arrays.toString(cost));
        }
        
        return res;
    }
}
