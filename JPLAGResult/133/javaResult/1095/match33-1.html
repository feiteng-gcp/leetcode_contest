class Solution {
    public int twoCitySchedCost(int[][] costs) {
        Arrays.sort(costs, (a, b) -> {
           return (b[0] - b[1]) - (a[0] - a[1]); 
        });
        int res = 0;
        for (int i = 0; i < costs.length; i++) 
{
            if (i < costs.length / 2) {
                res += costs[i][1];
            } else {
                res += costs[i][0];
            }
        }
        return res;
    }
}
