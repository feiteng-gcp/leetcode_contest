class Solution {

	private
 static class Cost implements Comparable<Cost> {
		public int cost1 , cost2;
		public Cost(int cost1 , int cost2) {
			this.cost1 = cost1;
			this.cost2 = cost2;			
		}
		@Override
		public int compareTo(Cost cost) {
			int value1 = this.cost1 - this.cost2;
			int value2 = cost.cost1 - cost.cost2;
			return value1 - value2;			
		}
	}
	
    
public int twoCitySchedCost(int[][] costs) {
    	List<Cost> list = new ArrayList<>();
    	for (int[] cost : costs) {
    		list.add(new Cost(cost[0] , cost[1]));
    	}
    	Collections.sort(list);
    	int ans = 0;
    	for (int i = 0;i < list.size() / 2;i ++) {
    		ans += list.get(i).cost1;    		
    	}
    	for (int i = list.size() / 2;i < list.size();i ++) {
    		ans += list.get(i).cost2;    		
    	}
    	return ans;
    }
}
