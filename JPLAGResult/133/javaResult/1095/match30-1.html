import java.util.Arrays;

class Solution {
	public int twoCitySchedCost(int[][] costs) {
		int N = costs.length;
		person[] persons = new person[N];
		for (int i = 0; i < N; i++) {
			persons[i] = new person(costs[i][0], costs[i][1]);
		}
		Arrays.sort(persons);
		int res = 0;

		for (int i = 0; i < N / 2; i++) {
			res += persons[i].A;
		}
		for (int i = N / 2; i < N; i++) {
			res += persons[i].B;
		}
		return res;
	}
}

class person implements Comparable<person> {
	public int A, B, diff;

	public person(int a, int b) {
		A = a;
		B = b;
		diff = A - B;
	}

	public int compareTo(person next) {
		return diff - next.diff;
	}
}
