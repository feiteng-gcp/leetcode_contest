class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int N = costs.length;
        Integer[] a= new Integer[N];
        for (int i = 0; i < N; i++) {a[i] = i;}
        Arrays.sort(a, (Integer A, Integer B) -> {return costs[A][0] - costs[A][1] - costs[B][0] + costs[B][1]; });
        
int res = 0;
        for (int i = 0; i < N / 2; i ++) {
            res += costs[a[i]][0];
        }
        for (int i = N/2; i<N; i++) {
            res += costs[a[i]][1];
        }
        return res;
        
    }
}
