class Solution {
    public int twoCitySchedCost(int[][] costs) {
        Arrays.sort(costs, (o1, o2) -> Math.abs(o2[0] - o2[1]) - Math.abs(o1[0] - o1[1]));
        int N = costs.length / 2;
        int selectA = 0, selectB = 0, minCost = 0;
        for (int[] c : costs) {
            if (selectA == N) {
                minCost += c[1];
                selectB++;
            } else if (selectB == N) {
                minCost += c[0];
                selectA++;
            } else if (c[0] < c[1]) {
                minCost += c[0];
                selectA++;
            } else {
                minCost += c[1];
                selectB++;
            }
        }
        return minCost;
    }
}
