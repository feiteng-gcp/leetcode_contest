class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int N = costs.length / 2, res = 0;
        Arrays.sort(costs, (a, b) -> (a[1] - a[0]) - (b[1] - b[0]));
        
for(int i = 0; i < N; i++) {
            res += costs[i][1];
            res += costs[i + N][0];
        }
        return res;
    }
}
