class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int N = costs.length / 2, len = costs.length;
        Arrays.sort(costs, (a, b) -> (a[0] - a[1] - b[0] + b[1]));
        int res = 0;
        for (int i = 0; i < N; ++i) {
        	res += costs[i][0];
        }
        for (int i = N; i < len; ++i) {
        	res += costs[i][1];
        }
        return res;
    }
}
