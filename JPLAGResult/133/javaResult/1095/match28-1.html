class Solution {
    public int twoCitySchedCost(int[][] costs) {
        int N = costs.length;
        int[][] delta = new int[N][];
        for (int i = 0; i < N; i++) {
            delta[i] =  new int[]{Math.abs(costs[i][1] - costs[i][0]), i};
        }
        Arrays.sort(delta, (a, b) -> Integer.compare(b[0], a[0]));
        int a = 0, b = 0, ans = 0;
        for (int i = 0; i < delta.length; i++) {
            int id = delta[i][1];
            
if (costs[id][0] < costs[id][1]) {
                if (a < N / 2) {
                    a++;
                    ans += costs[id][0];
                } else {
                    b++;
                    ans += costs[id][1];
                }
            } else {
                if (b < N / 2) {
                    b++;
                    ans += costs[id][1];
                } else {
                    a++;
                    ans += costs[id][0];
                }
            }
        }
        return ans;
    }
}
