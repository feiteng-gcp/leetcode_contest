class Solution:
    def allCellsDistOrder(self, R: int, C: int, r0: int, c0: int) -> List[List[int]]:
        q = []
        for i in range(R):
            for j in range(C):
                q.append((abs(i-r0)+abs(j-c0),i*100+j))
        q.sort(key=lambda x:(x[0],x[1]))
        res = [(l[1]//100,l[1]%100) for l in q]
        return res
