class Solution:
    def allCellsDistOrder(self, R: int, C: int, r0: int, c0: int) -> List[List[int]]:
        result = []
        dist = []
        for r in range(R):
            for c in range(C):
                dist.append((abs(r - r0) + abs(c - c0), r, c))
        
        dist.sort()
        
        return [[x[1], x[2]] for x in dist]
