class Solution:
    def allCellsDistOrder(self, R: int, C: int, r0: int, c0: int) -> List[List[int]]:
        res = []
        if not R or not C:
            return res
        
        visited = {(r0, c0)}
        q = collections.deque([(r0, c0)])
        while q:
            x, y = q.popleft()
            res.append((x, y))
            for n_x, n_y in [(x-1, y),(x+1, y),(x, y-1),(x, y+1)]:
                if 0 <= n_x < R and 0 <= n_y < C and (n_x, n_y) not in visited:
                    visited.add((n_x, n_y))
                    q.append((n_x, n_y))
        return res
