from collections import deque
class Solution:
    def allCellsDistOrder(self, R: int, C: int, r0: int, c0: int) -> List[List[int]]:
        queue = deque()
        queue.append([r0, c0])
        visited = set()
        visited.add((r0, c0))
        ans = []
        while queue:
            x, y = queue.popleft()
            ans.append([x,y])
            for dx, dy in [[-1,0],[1,0],[0,1],[0,-1]]:
                nx, ny = x+dx, y+dy
                if 0<=nx<R and 0<=ny<C and (nx, ny) not in visited:
                    visited.add((nx, ny))
                    queue.append([nx, ny])
        return ans
