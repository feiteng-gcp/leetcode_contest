class Solution:
    def allCellsDistOrder(self, R: int, C: int, r0: int, c0: int) -> List[List[int]]:
        self.res = [[r0,c0]]
        visited = [[0 for i in range(C)] for j in range(R)]
        def helper(i,j,visited):
            if i<0 or i>=R or j<0 or j>=C or visited[i][j]:
                return
            visited[i][j] = 1
            self.res.append([i,j])
        d = 1        
        while d<=R+C-2:
            dis = set()
            for x in range(0,d+1):
                y = d-x
                dis.add((r0-x,c0-y))
                dis.add((r0-x,c0+y))
                dis.add((r0+x,c0-y))
                dis.add((r0+x,c0+y))
            for (p,q) in dis:
                helper(p,q,visited)
            d += 1
        return self.res
