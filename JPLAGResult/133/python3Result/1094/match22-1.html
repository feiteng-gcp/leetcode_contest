class Solution:
    def allCellsDistOrder(self, R: int, C: int, r0: int, c0: int) -> List[List[int]]:
        q = collections.deque([[r0, c0]])
        seen = set([(r0, c0)])
        
        res = []
        
        
while q:
            i, j = q.popleft()
            res.append([i, j])
            
            for di, dj in [[-1, 0], [1, 0], [0, -1], [0, 1]]:
                x, y = i + di, j + dj
                
if 0 <= x < R and 0 <= y < C and (x, y) not in seen:
                    seen.add((x, y))
                    q.append([x, y])
        return res
