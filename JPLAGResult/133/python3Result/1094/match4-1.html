class Solution:
    def allCellsDistOrder(self, R: int, C: int, r0: int, c0: int) -> List[List[int]]:
        candi = []
        for i in range(R):
            for j in range(C):
                candi.append([abs(i-r0)+abs(j-c0),i,j])
        candi.sort()
        return [(i,j) for _,i,j in candi]
