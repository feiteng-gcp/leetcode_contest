class Solution:
    def allCellsDistOrder(self, R: int, C: int, r0: int, c0: int) -> List[List[int]]:
        points = []
        for i in range(R):
            for j in range(C):
                points.append([i, j])

        return sorted(points, key=lambda p:  abs(r0 - p[0]) + abs(c0 - p[1]))
