class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:
        costs.sort(key=lambda x:abs(x[0]-x[1]))
        n = len(costs)
        t1 = t2 = 0
        n1 = n2 = 0
        while costs:
            a, b = costs.pop()
            if n1 == n//2:
                t2 += b
                n2 += 1
            elif n2 == n//2:
                t1 += a
                n1 += 1
            elif a > b:
                t2 += b
                n2 += 1
            else:
                t1 += a
                n1 += 1
        return t1 + t2
