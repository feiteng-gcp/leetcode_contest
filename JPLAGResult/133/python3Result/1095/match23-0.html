class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:
        costs.sort(key = lambda x: x[0] - x[1])
        
        n = len(costs) // 2
        v = 0
        for i, c in enumerate(costs):
            if i < n:
                v += c[0]
            else:
                v += c[1]
        return v
