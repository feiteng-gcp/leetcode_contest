class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:
        N = len(costs)//2
        pair = []
        for i in range(2*N):
            pair.append((costs[i][0] - costs[i][1], i))
        pair.sort()
        
        result = 0
        for i in range(N):
            result += costs[pair[i][1]][0]
        for i in range(N, 2*N):
            result += costs[pair[i][1]][1]
        return result
