class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:
        n = len(costs)
        temp = []
        for i in range(n):
            temp.append([costs[i][1]-costs[i][0], i])
        temp.sort()
        ans = 0
        for i in range(n//2):
            ans += costs[temp[i][1]][1]
        for i in range(n//2, n):
            ans += costs[temp[i][1]][0]
        return ans
