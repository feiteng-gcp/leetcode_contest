class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:
        costs.sort(key=lambda x: abs(x[0] - x[1]), reverse=True)
        # print(costs)
        A = B = 0
        ans = 0
        cap = len(costs) // 2
        for x, y in costs:
            if max(A, B) < cap:
                if x <= y:
                    A += 1
                else:
                    B += 1
                ans += min(x, y)
            else:
                if A == cap:
                    B += 1
                    ans += y
                else:
                    A += 1
                    ans += x
        return ans
