class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:
        cal = []
        for [a, b] in costs:
            cal.append([a - b, a, b])
        cal.sort(key=lambda x: x[0])
        ret = 0
        for i in range(len(cal)//2):
            ret += cal[i][1]
        for i in range(len(cal)//2, len(cal)):
            ret += cal[i][2]
        return ret
