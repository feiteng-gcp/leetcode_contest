class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:
        costs.sort(key=lambda x: -abs(x[0] - x[1]))
        res = 0
        count_a = 0
        count_b = 0
        N = len(costs) // 2
        for i in range(len(costs)):
            if count_b == N:
                res += costs[i][0]
                count_a += 1
            elif count_a == N:
                res += costs[i][1]
                count_b += 1
            else:
                if costs[i][0] < costs[i][1]:
                    res += costs[i][0]
                    count_a += 1
                else:
                    res += costs[i][1]
                    count_b += 1
        return res
