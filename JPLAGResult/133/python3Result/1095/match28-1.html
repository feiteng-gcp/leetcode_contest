class Solution:
    def twoCitySchedCost(self, costs: List[List[int]]) -> int:
        N = len(costs)
        advantage_of_a = [(a - b, i) for i, (a, b) in enumerate(costs)]
        advantage_of_a.sort()
        total = 0
        for i in range(N//2):
            idx = advantage_of_a[i][1]
            total += costs[idx][0]
        
for i in range(N//2, N):
            idx = advantage_of_a[i][1]
            total += costs[idx][1]
        return total
