class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        ans1 = self.find(A, L, M)
        ans2 = self.find(A, M, L)
        return max(ans1, ans2)
    
    def find(self, A, L, M):
        maxL = [0]*len(A)
        cur = sum(A[:L])
        maxL[L-1] = cur
        for i in range(1, len(A) - L):
            cur -= A[i - 1]
            cur += A[i + L - 1]
            maxL[i + L - 1] = max(maxL[i + L - 2], cur)
        
        cur = sum(A[-M:])
        maxM = [0]*len(A)
        maxM[-M] = cur
        for i in range(len(A) - M - 1, -1, -1):
            cur -= A[i + M]
            cur += A[i]
            maxM[i] = max(maxM
[i+1], cur)
        
        answer = 0
        for i in range(L-1, len(A)-M+1):
            answer = max(maxM[i] + maxL[i-1], answer)
        return answer
