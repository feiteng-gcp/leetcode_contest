class Solution:
    def maxSumTwoNoOverlap(self, a: List[int], L: int, M: int) -> int:
        n = len(a)
        ll, ml = [0] * n, [0] * n
        for i in range(n - L + 1):
            ll[i] = sum(a[i:i + L])
        for i in range(n - M + 1):
            ml[i] = sum(a[i:i + M])
        
        m = 0
        for i in range(n):
            e1 = i + L
            for j in range(n):
                e2 = j + M
                if i <= j < e1 or j <= i < e2:
                    continue
                m = max(m, ll[i] + ml[j])
        return m
