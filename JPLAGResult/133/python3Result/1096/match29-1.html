class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        return max(self.helper(L,M,A), self.helper(M,L,A))
    
    def helper(self, left, right, A):
        n = len(A)
        left_max_sum = [0] * n
        right_max_sum = [0] * n
        
        # left
        current_max_sum = sum(A[:left])
        current_sum = sum(A[:left])
        left_max_sum[left-1] = current_max_sum
        
        for i in range(left, n):
            current_sum += A[i] - A[i-left]
            
            current_max_sum = max(current_sum, current_max_sum)
            
            left_max_sum[i] = current_max_sum
        
        
        # right
        A = A[::-1]
        current_max_sum = sum(A[:right])
        current_sum = sum(A[:right])
        right_max_sum[right-1] = current_max_sum
        
        for i in range(right, n):
            current_sum += A[i] - A[i-right]
            
            current_max_sum = max(current_sum, current_max_sum)
            
            right_max_sum[i] = current_max_sum
        
        right_max_sum = right_max_sum[::-1]
        
        # result
        result = 0
        
        for i in range(left-1,n-right):
            result = max(result, left_max_sum[i] + right_max_sum[i+1])
        
        return result
