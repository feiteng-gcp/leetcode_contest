def maxsum(A, L, M):
    n = len(A)
    bestm = [0 for _ in range(n)]
    s = sum(A[n-M:])
    bestm[n-M] = s
    
for i in range(n-M-1, -1, -1):
        s += A[i] - A[i+M]
        bestm[i] = max(s, bestm[i+1])
    s = sum(A[:L])
    best = s + bestm[L]
    for i in range(1, n-L):
        s += A[i+L-1] - A[i-1]
        best = max(best, s + bestm[L+i])
    return best
        
        
        

class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        return max(maxsum(A, L, M), maxsum(A, M, L))
