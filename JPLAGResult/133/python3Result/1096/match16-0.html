class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        N = len(A)
        cur = sum(A[:L])
        stack1 = {L:cur}
        for i in range(1,len(A)-L+1):
            cur = cur + A[L+i-1] - A[i-1]
            stack1[i+L] = max(stack1[i+L-1],cur)
        A.reverse()
        cur = sum(A[:M])
        stack2 = {N-M:cur}
        for i in range(1,len(A)-M+1):
            cur = cur + A[M+i-1] - A[i-1]
            stack2[N-i-M] = max(stack2[N-i-M+1],cur)
        res = 0
        for i in stack1.keys():
            if i in stack2.keys():
                res = max(res,stack1[i]+stack2[i])
        cur = sum(A[:L])
        stack1 = {L:cur}
        for i in range(1,len(A)-L+1):
            cur = cur + A[L+i-1] - A[i-1]
            stack1[i+L] = max(stack1[i+L-1],cur)
        A.reverse()
        cur = sum(A[:M])
        stack2 = {N-M:cur}
        for i in range(1,len(A)-M+1):
            cur = cur + A[M+i-1] - A[i-1]
            stack2[N-i-M] = max(stack2[N-i-M+1],cur)
        for i in stack1.keys():
            if i in stack2.keys():
                res = max(res,stack1[i]+stack2[i])
        return res
