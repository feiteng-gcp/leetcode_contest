class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        def cal_sum(l):
            s = [sum(A[:l])]
            for i in range(len(A) - l):
                s.append(s[-1] - A[i] + A[i + l])
            return s
        ls = cal_sum(L)
        ms = cal_sum(M)
        ans = sum(A[:L + M])
        lmax, mmax = ls[0], ms[0]
        for i in range(1, len(A) - L - M + 1):
            lmax, mmax = max(lmax, ls[i]), max(mmax, ms[i])
            ans = max(ans, lmax + ms[i + L], mmax + ls[i + M])   
        return ans
