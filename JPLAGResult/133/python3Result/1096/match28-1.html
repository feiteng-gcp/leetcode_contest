class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        R = M
        cum_sum = [0 for i in range(len(A)+1)]
        for i in range(len(A)):
            cum_sum[i] = cum_sum[i-1] + A[i]
            
        sumL = [0 for i in range(len(A)-L+1)]
        sumLmax = [0 for i in range(len(A)-L+1)]
        for i in range(len(A)-L+1):
            sumL[i] = cum_sum[i+L-1] - cum_sum[i-1]
            sumLmax[i] = max(sumLmax[i-1], sumL[i])
            
        sumR = [0 for i in range(len(A)-R+1)]
        sumRmax = [0 for i in range(len(A)-R+1)]
        for i in range(len(A)-R+1):
            sumR[i] = cum_sum[i+R-1] - cum_sum[i-1]
            sumRmax
[i] = max(sumRmax[i-1], sumR[i])
            
        res = 0
        # R comes second
        for i in range(L, len(A)-R+1):
            # if i+R < len(A)-L+1:
            res = max(res, sumR[i]+sumLmax[i-L])
            # else: break
        
        for i in range(R, len(A)-L+1):
            # if i+R < len(A)-R+1:
            res = max(res, sumL[i]+sumRmax[i-R])
            # else: break
        
        return res
