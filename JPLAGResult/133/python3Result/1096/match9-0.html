class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        size = len(A)
        sums = [[0] * size for _ in range(size)]
        for i in range(size):
            for j in range(i, size):
                if i == j:
                    sums[i][j] = A[i]
                else:
                    sums[i][j] = sums[i][j - 1] + A[j]
        res = 0
        for i in range(0, size - L + 1):
            for j in range(0, i - M + 1):
                res = max(sums[i][i + L - 1] + sums[j][j + M - 1], res)
            for j in range(i + L, size - M + 1):
                res = max(sums[i][i + L - 1] + sums[j][j + M - 1], res)
        return res
