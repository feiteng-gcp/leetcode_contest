class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        n = len(A)
        if n == 0:
            return 0
        s = [[0] * (n + 1) for _ in range(n)]
        for i in range(n):
            t = 0
            for j in range(i + 1):
                t += A[i - j]
                s[i][j + 1] = t
        # l, then m
        ans = 0
        
for m in range(L + M - 1, n):
            for l in range(L - 1, m - M + 1):
                ans = max(ans, s[m][M] + s[l][L])
        for l in range(L + M - 1, n):
            for m in range(M - 1, l - L + 1):
                ans = max(ans, s[l][L] + s[m][M])
        return ans
