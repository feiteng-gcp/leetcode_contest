class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        if not A:
            return 0
        sumM=[]
        sumL=[]
        for i in range(len(A)-L+1):
            sumL.append(sum(A[i:i+L]))
        for i in range(len(A)-M+1):
            sumM.append(sum(A[i:i+M]))
        ans=0
        for left in [M,L]:
            for left_start in range(len(A)-(M+L)+1):
                for right_start in range(left_start+left,len(A)-(M+L-left)+1):
                    # print(left_start,right_start)
                    if left==M:
                        cand=sumM[left_start]+sumL[right_start]
                    else:
                        cand=sumL[left_start]+sumM[right_start]
                    ans=max(ans,cand)
        return ans
