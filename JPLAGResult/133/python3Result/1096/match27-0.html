class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        return max(self.maxSum(A, L, M), self.maxSum(A, M, L))
        
    def maxSum(self, A, L, R):
        left = [0] * len(A)
        right = [0] * len(A)
        left[L-1] = sum(A[:L])
        right[len(A)-1-R] = sum(A[len(A)-R:])
        
        for i in range(L, len(A)):
            left[i] = left[i-1] - A[i-L] + A[i]
        for i in range(len(A)-2-R, -1, -1):
            right[i] = right[i+1] - A[i+R+1] + A[i+1]
        '''
        print(L, R)
        print(left)
        print(right)
        '''
        for i in range(1, len(left)):
            left[i] = max(left[i-1], left[i])
        for i in range(len(left)-2, -1, -1):
            right[i] = max(right[i+1], right[i])
        #print('m', left)
        #print('m', right)
        
        ans = 0
        for i in range(L-1, len(A)-R):
            ans = max(ans, left[i] + right[i])
        return ans
