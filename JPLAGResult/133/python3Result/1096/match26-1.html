class Solution:
    def _maxSumTwoNoOverlap(self, A, L, M):
        n = len(A)
        
        sumA = [0]
        for a in A:
            sumA.append(sumA[-1] + a)
        
        maxL = [-1] * n
        maxL[L-1] = sumA[L]-sumA[0]
        for i in range(n-L):
            maxL[i+L] = max(maxL[i+L-1], sumA[i+L+1]-sumA[i+1])
        
        maxM = [-1] * n
        maxM[n-M] = sumA[n]-sumA[n-M]
        for i in range(n-M):
            maxM[n-M-i-1] = max(maxM[n-M-i], sumA[n-i-1]-sumA[n-M-i-1])
        
        maxV = -1
        for i in range(L-1, n-M):
            maxV = max(maxV, maxL[i] + maxM[
i+1])
        return maxV
    
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        return max(self._maxSumTwoNoOverlap(A, L, M), self._maxSumTwoNoOverlap(A, M, L))
