class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        n = len(A)
        max_left_M = [float('-inf')] * n
        max_right_M = [float('-inf')] * n
        
        cur = max_left_M[M - 1] = sum(A[:M])
        for i in range(M, n):
            cur += A[i] - A[i - M]
            max_left_M[i] = max(max_left_M[i - 1], cur)
            
        cur = max_right_M[-M] = sum(A[-M:])
        for i in range(n - M - 1, -1, -1):
            cur += A[i] - A[i + M]
            max_right_M[i] = max(max_right_M[i + 1], cur)
        
        # print(max_left_M)
        # print(max_right_M)
        
        cur_L = sum(A[:L])
        ans = cur_L + max_right_M[L]
        i = 1
        while i + L <= n:
            max_M = max_left_M[i - 1]
            if i + L < n:
                max_M = max(max_M, max_right_M[i + L])
            cur_L += A[i + L - 1] - A[i - 1]
            ans = max(ans, cur_L + max_M)
            i += 1
        return ans
