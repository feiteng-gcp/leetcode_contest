class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        pre = [0]
        for a in A:
            pre.append(pre[-1]+a)
        n = len(A)
        maxL1=[0]*(n+2)
        maxL2=[0]*(n+2)
        maxM1=[0]*(n+2)
        maxM2=[0]*(n+2)
        for i in range(L,n+1):
            maxL1[i] = max(maxL1[i-1],pre[i]-pre[i-L])
        for i in range(n-L+1,-1,-1):
            maxL2[i] = max(maxL2[i+1],pre[i+L-1]-pre[i-1])
        
for i in range(M,n+1):
            maxM1[i] = max(maxM1[i-1],pre[i]-pre[i-M])
        for i in range(n-M+1,-1,-1):
            maxM2[i] = max(maxM2[i+1],pre[i+M-1]-pre[i-1])
        res = 0
        for i in range(L,n-M+1):
            res = max(maxL1[i]+maxM2[i+1],res)
        for i in range(M,n-L+1):
            res = max(maxM1[i]+maxL2[i+1],res)
        return res
