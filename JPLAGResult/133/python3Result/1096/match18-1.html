class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        arr1 = [0 for i in range(len(A))]
        arr2 = arr1[:]
        arr1[L - 1] 
= sum(A[:L])
        s = arr1[L - 1]
        for i in range(L, len(A)):
            s += A[i] - A[i - L]
            arr1[i] = s
        arr2[M - 1] 
= sum(A[:M])
        s = arr2[M - 1]
        for i in range(M, len(A)):
            s += A[i] - A[i - M]
            arr2[i] = s

        ans = 0
        for i in range(L - 1, len(A) - M):
            for j in range(i + M, len(A)):
                ans = max(ans, arr1[i] + arr2[j])
        for i in range(M - 1, len(A) - L):
            for j in range(i + L, len(A)):
                ans = max(ans, arr2[i] + arr1[j])

        return ans
