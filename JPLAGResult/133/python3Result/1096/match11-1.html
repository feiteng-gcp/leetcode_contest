class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        n = len(A)
        longL = [-(1e9) for i in range(n)]
        longM = [-(1e9) for i in range(n)]
        sufSum = [0 for i in range(n)]
        sufSum[n-1] = A[n-1]
        for i in range(n-2, -1, -1):
            sufSum[i] = sufSum[i+1] + A[i]
        prefSum = [0 for i in range(n)]
        prefSum
[0] = A[i]
        for i in range(1, n):
            prefSum[i] = prefSum[i-1] + A[i]
        longL[n-L] = sufSum[n-L]
        longM[n-M] = sufSum[n-M]
        for i in range(n-L-1, -1, -1):
            longL[i] = max(longL[i+1], sufSum[i] - sufSum[i+L])
        for i in range(n-M-1, -1, -1):
            longM[i] = max(longM[i+1], sufSum[i] - sufSum[i+M])
        #print(prefSum)
        #print(longM)
        #print(longL)
        Lp = prefSum[L-1]
        best = Lp + longM[L]
        for i in range(L, n-1):
            Lp = max(Lp, prefSum[i]-prefSum[i-L])
            best = max(best, Lp + longM[i+1])
        Mp = prefSum[M-1]
        best = max(best, Mp + longL[M])
        for i in range(M, n-1):
            Mp = max(Mp, prefSum[i]-prefSum[i-M])
            best = max(best, Mp + longL[i+1])
        return best
