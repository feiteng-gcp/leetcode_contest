class Solution:
    def maxSumTwoNoOverlap(self, A, L, M):
        return max(self.check(A, L, M), self.check(A, M, L))
        
    def check(self, A, L, M):
        L_sums = self.make_sums(A, L)
        M_sums = self.make_sums(A, M)

        L_max_sums = self.make_max_sums(L_sums, L)

        best = 0
        for m in range(L, len(M_sums)):
            best = max(
                best,
                M_sums[m] + L_max_sums[m]
            )

        return best


    def make_max_sums(self, L_sums, L):
        max_sums = {L: L_sums[0]}
        for i in range(1, len(L_sums)):
            max_sums[L+i] = max(max_sums[L+i-1], L_sums[i])

        return max_sums

    
def make_sums(self, A, L):
        sums = [sum(A[:L])]
        for i in range(L, len(A)):
            sums.append(sums[-1] + A[i] - A[i-L])

        return sums
