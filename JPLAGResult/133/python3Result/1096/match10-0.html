class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        n = len(A)
        l = [None] * n
        m = [None] * n
        
        sumL= sum(A[:L])
        l[L-1] = sumL
        for i in range(L, n):
            sumL += A[i] - A[i-L]
            l[i] = sumL
        sumM = sum(A[:M])
        m[M-1] = sumM
        for i in range(M, n):
            sumM += A[i] - A[i-M]
            m[i] = sumM
        # print(l)
        # print(m)
        ret = 0
        for i in range(L-1, n-M):
            for j in range(i+M, n):
                ret = max(ret, l[i] + m[j])
        for i in range(M-1, n-L):
            for j in range(i+L, n):
                ret = max(ret, m[i] + l[j])
        return ret
