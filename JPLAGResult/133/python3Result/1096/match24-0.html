class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        def construct_list(k):
            ret = []
            ret.append(sum(A[:k]))
            pre = ret[-1]
            for i in range(k,n):
                ret.append(max(pre+A[i]-A[i-k],ret[-1]))
                pre = pre+A[i]-A[i-k]
            return ret
        def reverse_list(k):
            ret = []
            ret.append(sum(A[n-k:]))
            pre = ret[-1]
            for i in reversed(range(n-k)):
                ret.append(max(pre+A[i]-A[i+k],ret[-1]))
                pre = pre+A[i]-A[i+k]
            return ret
        
        n = len(A)
        l = L
        m = M
        list_l_left = construct_list(l)
        # print("list_l_left = ", list_l_left)
        list_m_left = construct_list(m)
        # print("list_m_left = ", list_m_left)
        list_l_right = reverse_list(l)
        # print("list_l_right = ", list_l_right)
        list_m_right = reverse_list(m)
        # print("list_m_right = ", list_m_right)
        ret = 0
        # print("n = ",n )
        for i in range(l,n-m+1):
            # print(" i = ,", i ,  "left = ", list_l_left[i-l] , " right = ", list_m_right[n-i-m])
            ret = max(ret, list_l_left[i-l] + list_m_right[n-i-m])
        # print("second ")
        for i in range(m,n-l+1):
            # print(" i = ,", i ,  "left = ", list_m_left[i-m] , " right = ", list_l_right[n-i-l])
            ret = max(ret, list_m_left[i-m] + list_l_right[n-i-l])
        return ret
