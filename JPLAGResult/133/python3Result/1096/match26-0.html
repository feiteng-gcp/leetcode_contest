class Solution:
    def _helper(self, A: List[int], L: int, M: int) -> int:
        max_sum = 0
        window_L = None
        for i in range(0,len(A) - L - M + 1):
            if not window_L:
                window_L = sum(A[i:i+L])
            else:   # sliding window
                window_L = window_L - A[i - 1] + A[i + L - 1]
            window_M = None
            for j in range(i+L,len(A) - M + 1):
                if not window_M:
                    window_M = sum(A[j : j + M])
                else:
                    window_M = window_M - A[j - 1] + A[j + M - 1]
                max_sum = max(max_sum,window_L + window_M)
        return max_sum
            
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        return max(self._helper(A,L,M),self._helper(A,M,L))
