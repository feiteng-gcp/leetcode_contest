class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        n = len(A)
        ans = 0
        
        a = [0] * n
        a[L-1] = sum(A[:L])
        s = a[L-1]
        for i in range(L, len(A)-M):
          s = s + A[i] - A[i-L]
          a[i] = max(A[i-1], s)
        
        b = [0] * n
        b[n-M] = sum(A[n-M:])
        s = b[n-M]
        for j in range(n-M-1, L-1, -1):
          s = s + A[j] - A[j+M]
          b[j] = max(b[j+1], s)
        
        for i in range(L-1, n-M):
          ans = max(ans, a[i]+b[i+1])
        
        c = [0]*n
        c[M-1] = sum(A[:M])
        s = c[M-1]
        for i in range(M, len(A)-L):
          s = s + A[i] - A[i-M]
          c[i] = max(A[i-1], s)
        
        d = [0] * n
        d[n-L] = sum(A[n-L:])
        s = d[n-L]
        for j in range(n-L-1, M-1, -1):
          s = s + A[j] - A[j+L]
          d[j] = max(d[j+1], s)
        
        for i in range(M-1, n-L):
          ans = max(ans, c[i]+d[i+1])
        

        return ans
