class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        pre = [0]
        for i in A:
            pre += [pre[-1] + i]
        ans = 0
        n = len(pre)
        for i in range(L, n - M):
            for j in range(i + M, n):
                ans = max(ans, pre[i] - pre[i - L] + pre[j] - pre[j - M])
        for i in range(M, n - L):
            
for j in range(i + L, n):
                ans = max(ans, pre[i] - pre[i - M] + pre[j] - pre[j - L])
        return ans
