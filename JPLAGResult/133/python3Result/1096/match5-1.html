class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        d_l = {}
        d_m = {}
        
        n = len(A)
        
        
for i in range(n - L + 1):
            d_l[i] = sum(A[i:(i + L)])
        
        for i in range(n - M + 1):
            d_m[i] = sum(A[i:(i + M)])
        
        res = 0
        for i in range(n - L + 1):
            for j in range(i + L, n - M + 1):
                res = max(res, d_l[i] + d_m[j])
            for j in range(i - M + 1):
                res = max(res, d_l[i] + d_m[j])
            
        return res
