class Solution:
    def maxSumTwoNoOverlap(self, A: List[int], L: int, M: int) -> int:
        
        n = len(A)
        if L + M == n:
            return sum(A)
        Left_maxsumL = [0] * n
        Left_maxsumM = [0] * n
        Right_maxsumL = [0] * n
        Right_maxsumM = [0] * n
        
        cumsum = [0] * n
        cumsum[0] = A[0]
        for i in range(1,n):
            cumsum[i] = cumsum[i-1] + A[i]
        
        Left_maxsumL[L-1] = cumsum[L-1]
        Left_maxsumM[M-1] = cumsum[M-1]
        for i in range(L, n):
            Left_maxsumL[i] = max(Left_maxsumL[i-1], cumsum[i] - cumsum[i-L])
        for i in range(M, n):
            Left_maxsumM[i] = max(Left_maxsumM[i-1], cumsum[i] - cumsum[i-M])
        Right_maxsumL[n-L] = cumsum[n-1] - cumsum[n-1-L]
        Right_maxsumM[n-M] = cumsum[n-1] - cumsum[n-1-M]
        for i in reversed(range(n-L)):
            Right_maxsumL[i] = max(Right_maxsumL[i+1], cumsum[i+L-1] - cumsum[i-1])
        for i in reversed(range(n-M)):
            Right_maxsumM[i] = max(Right_maxsumM[i+1], cumsum[i+M-1] - cumsum[i-1])
        
        res = 0
        for i in range(L-1, n-M):
            if Left_maxsumL[i] + Right_maxsumM[i+1] > res:
                res = Left_maxsumL[i] + Right_maxsumM[i+1]
        for i in range(M-1,n-L):
            if Left_maxsumM[i] + Right_maxsumL[i+1] > res:
                res = Left_maxsumM[i] + Right_maxsumL[i+1]
                
        return res
