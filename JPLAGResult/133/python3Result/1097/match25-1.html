class StreamChecker:

    def __init__(self, words: List[str]):
        self.trie = {}
        for word in words:
            rword = word[::-1]
            t 
= self.trie
            for c in rword:
                if c not in t:
                    t[c] = {}
                t = t[c]
            t["#"] = "#"
        self.queried = []

    def query(self, letter: str) -> bool:
        x = self.trie
        self.queried.append(letter)
        i = 0
        while i < len(self.queried):
            c = self.queried[-(i + 1)]
            if c not in x:
                return False
            x = x[c]
            if "#" in x:
                return True
            i += 1
        return False
