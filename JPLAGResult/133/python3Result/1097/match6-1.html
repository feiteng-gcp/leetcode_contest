class StreamChecker:
    def __init__(self, words: List[str]):
        self.trie = Trie()
        for word in words:
            self.trie.insert(word[::-1])
        self.current = ''

    def query(self, letter: str) -> bool:
        self.current += letter
        
        node = self.trie.root
        for i in range(len(self.current) - 1, -1, -1):
            c = self.current[i]
            # print(self.current[i:][::-1], c, node.children)
            if c in node.children:
                node = node.children[c]
                if node.isWord:
                    # print(True)
                    return True
            else:
                # print(False)
                return False
        # print(False,2)
        return False
