class StreamChecker:

    def __init__(self, words: List[str]):
        self.trie = dict()   
        for w in words:
            node = self.trie
            for c in w[::-1]:
                if not c in node:
                    node[c] = dict(
)
                node = node[c]
            node[None] = True
        self.history = []
        
    def query(self, letter: str) -> bool:
        self.history.append(letter)
        node = self.trie
        i = len(self.history) - 1
        while node and i >= 0:
            if not self.history[i] in node:
                return False
            node = node[self.history[i]]
            if None in node:
                return True
            i -= 1
        return False
