class TrieNode:
    def __init__(self):
        self.dc = {}
        self.is_word = False

    def add_word(self, word):
        node = self
        for c in word:
            if c not in node.dc:
                node.dc[c] = TrieNode()
            node = node.dc[c]
        node.is_word = True

class StreamChecker:
    

    def __init__(self, words: List[str]):
        self.trie = TrieNode()
        for w in words:
            self.trie.add_word(w)
        self.nodes = [self.trie]

    def query(self, letter: str) -> bool:
        next_nodes = []
        good = False
        for n in self.nodes:
            nn = n.dc.get(letter)
            if nn:
                next_nodes.append(nn)
                if nn.is_word:
                    good = True
        next_nodes.append(self.trie)
        self.nodes = next_nodes
        return good
