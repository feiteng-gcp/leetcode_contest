from typing import List

class TrieNode:
  def __init__(self):
    self.chars = {}
    self.end = False

  def has(self, char):
    return char in self.chars

  def get(self, char):
    return self.chars[char]

  def add(self, char):
    self.chars[char] = TrieNode()
    return self.chars[char]

class Trie:
  def __init__(self):
    self.root = TrieNode()

  def add(self, word):
    node = self.root
    for char in word:
      if node.has(char):
        node = node.get(char)
      else:
        node = node.add(char)
    node.end = True

class StreamChecker:

  def __init__(self, words: List[str]):
    self.trie = Trie()
    for word in words:
      self.trie.add(word[::-1])

    self.acc = ''


  def query(self, letter: str) -> bool:
    self.acc += letter
    node = self.trie.root

    for i in range(len(self.acc) - 1, -1, -1):
      char = self.acc[i]
      if not node.has(char):
        return False
      node = node.get(char)
      if node.end:
        return True
    return node.end
