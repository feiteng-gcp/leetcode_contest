class Trie:
    def __init__(self):
        self.n = collections.defaultdict(Trie)
        self.t = False
        
class StreamChecker:

    def __init__(self, words: List[str]):
        self.buf = []
        self.root = Trie()
        for w in words:
            p = self.root
            for c in w[::-1]:
                p = p.n[c]
            p.t = True

    def query(self, letter: str) -> bool:
        self.buf.append(letter)
        p = self.root
        for i in range(len(self.buf) - 1, -1, -1):
            c = self.buf[i]
            if c not in p.n:
                return False
            p = p.n[c]
            if p.t:
                return True
        return False
