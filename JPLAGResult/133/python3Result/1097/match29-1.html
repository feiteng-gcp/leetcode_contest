class StreamChecker:

    def __init__(self, words: List[str]):
        self.q = []
        self.d = {}
        self.maxium = 0
        for w in words:
            for i in range(1, len(words)):
                if w[i:] not in self.d:
                    self.d[w[i:]] = 0
            self.d[w] = 1
            self.maxium = max(self.maxium, len(w
))

    def query(self, letter: str) -> bool:
        self.q.append(letter)
        if len(self.q) > self.maxium:
            self.q.pop(0)
        s = ""
        for i in range(len(self.q) - 1, -1, -1):
            s = self.q[i] + s
            temp = self.d.get(s, -1)
            if temp == -1:
                return False
            if temp == 1:
                return True
        return False
