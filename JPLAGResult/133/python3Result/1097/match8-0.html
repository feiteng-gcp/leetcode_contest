class Word:    
    def __init__(self, letter):
        self.letter = letter
        self.next = {}
        self.inList = False

class StreamChecker:

    def __init__(self, words: List[str]):
        self.root = Word('/')
        for word in words:
            temp = self.root
            start = len(word) - 1
            while start >= 0:
                c = word[start]
                if c not in temp.next:
                    temp.next[c] = Word(c)
                temp = temp.next[c]
                start -= 1
            temp.inList = True
        self.q = []
        

    def query(self, letter: str) -> bool:
        self.q.append(letter)
        temp = self.root
        for i in range(len(self.q) - 1, -1, -1):
            c = self.q[i]
            if c in temp.next:
                temp = temp.next[c]
                if temp.inList:
                    return True
            else:
                return False
        return False
