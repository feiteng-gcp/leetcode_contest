class ListNode:
    def __init__(self, v):
        self.v = v
        self.next = None


class StreamChecker:

    def __init__(self, words: List[str]):
        self.root = {}
        for word in words:
            cur = self.root
            for c in word:
                if c not in cur:
                    cur[c] = {}
                cur = cur[c]
            cur['done'] = True
        self.stream = ListNode(None)
        
    def query(self, letter: str) -> bool:
        prev = self.stream
        node = self.stream.next
        found = False
        while node is not None:
            if letter in node.v:
                node.v = node.v[letter]
                found = found or 'done' in node.v
                prev = node
                node = node.next
            else:
                prev.next = node.next
                node = node.next

        if letter in self.root:
            tmp = self.stream.next
            self.stream.next = ListNode(self.root[letter])
            self.stream.next.next = tmp
            found = found or 'done' in self.root[letter]
    
        return found
