class StreamChecker:

    def __init__(self, words: List[str]):
        self.trie = {}
        
for w in words:
            p = self.trie
            for c in w:
                if c not in p:
                    p[c] = {}
                p = p[c]
            p['**'] = 1
        self.qs = {}
        self.qc = 0

    def query(self, letter: str) -> bool:
        if letter in self.trie:
            p = self.trie
            self.qs[self.qc] = p
            self.qc += 1
        
        d, r = [], False
        for c, q in self.qs.items():
            if letter in q:
                q = q[letter]
                self.qs[c] = q
                if '**' in q:
                    r = True
            else:
                d.append(c)
        for c in d:
            del self.qs[c]
        return r
