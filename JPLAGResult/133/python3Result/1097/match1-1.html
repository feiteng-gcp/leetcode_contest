class Node:
    def __init__(self):
        self.children = {} # next letter -> Node
        self
.value = False
class StreamChecker:

    def __init__(self, words: List[str]):
        self.string = ''
        self.trie = Node()
        for word in words:
            root = self.trie
            for letter in word[::-1]:
                if letter not in root.children:
                    root.children[letter] = Node()
                root = root.children[letter]
            root.value = True

    def query(self, letter: str) -> bool:
        root = self.trie
        self.string += letter
        n = len(self.string)
        for i in range(n-1, -1, -1):
            c 
= self.string[i]
            if c not in root.children:
                return False
            root = root.children[c]
            if root.value:
                return True
        return False
