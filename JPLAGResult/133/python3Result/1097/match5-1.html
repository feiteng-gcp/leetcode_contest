class Node:
    def __init__(self, c):
        self
.c = c
        self.son = {}
        self.isword = False

class StreamChecker:

    def __init__(self, words):
        self.root = Node(None)
        for word in words:
            node 
= self.root
            for c in word:
                if c not in node.son:
                    node.son[c] = Node(c)
                node = node.son[c]
            node.isword = True

        self.seed = [self.root]

    def query(self, letter):
        ans = False

        buff = [self.root]
        for node in self.seed:
            if letter in node.son:
                if node.son[letter].isword:
                    ans = True
                buff.append(node.son[letter])

        self.seed = buff
        return ans
