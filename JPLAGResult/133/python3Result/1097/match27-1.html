class Trie:
    def __init__(self):
        self.root = Node()
    
    def add_word(self, word):
        p = self.root
        for c in reversed(word):
            p 
= p.children[c]
        p.is_word = True

class StreamChecker:

    def __init__(self, words: List[str]):
        self.t = Trie()
        for word in words:
            self.t.add_word(word)
        self.q = []

    def query(self, letter: str) -> bool:
        self.q.append(letter)
        idx = len(self.q) - 1
        p = self.t.root
        
        while idx >= 0 and self.q[idx] in p.children:
            p = p.children[self.q[idx]]
            if p.is_word:
                return True
            idx -= 1
        return False
