class StreamChecker:

    def __init__(self, words: List[str]):
        self.k = max(len(w) for w in words)
        self
.trie = {}
        for w in words:
            t = self.trie
            for c in w[::-1]:
                if c not in t:
                    t[c] = {}
                t = t[c]
            t["#"] = {}
        self.querylist = []            

    def query(self, letter: str) -> bool:
        self.querylist.append(letter)
        q = self.querylist
        i = -1
        t =self.trie
        k = self.k
        while i > -k - 1 and i > -len(self.querylist) - 1:
            if q[i] not in t:
                return False
            t = t[q[i]]
            if '#' in t:
                return True
            i -= 1
        return False
