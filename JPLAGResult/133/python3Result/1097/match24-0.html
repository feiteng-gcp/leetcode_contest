class Trie:
    def __init__(self,v):
        self.child = [None]*26
        self.value = v
        self.end = False
class StreamChecker:
    def insert_to_tree(self,word):
        # print("word= . ", word)
        node = self.root
        for chara in word:
            index = ord(chara) - ord('a')
            if node.child[index] is None:
                node.child[index] = Trie(chara)
            node = node.child[index]
        node.end = True
        
    def find(self,stream):
        # print("stream = ", stream)
        node = self.root
        n = len(stream)
        index = 0
        while node and index < n:
            chara = stream[index]
            n_index = ord(chara) - ord('a')
            if node.child[n_index] is not None:
                node = node.child[n_index]
            else:
                return False
            if node.end == True:
                return True
            index += 1
        return False
        
        
    def __init__(self, words: List[str]):
        self.root = Trie(0)
        self.stream = ""
        for word in words:
            self.insert_to_tree(word[::-1])
        
    def query(self, letter: str) -> bool:
        self.stream += letter
        return self.find(self.stream[::-1])
