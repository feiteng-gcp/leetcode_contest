class Trii():
    def __init__(self):
        self.child = [None for _ in range(26)]
        self.isword = False
    def add(self, w, i=0):
        cur_node = self
        for i in range(len(w)):
            o = ord(w[i])-ord('a')
            if cur_node.child[o] is None:
                cur_node.child[o] = Trii()
            cur_node = cur_node.child[o]
        cur_node.isword = True

    def get_substr_node(self, w, i=0):
        if i < len(w):
            o = ord(w[i])-ord('a')
            if self.child[o] is not None:
                return self.child[o].get_substr_node(w, i+1)
            else:
                return None
        else:
            return self
            

class StreamChecker:

    def __init__(self, words: List[str]):
        self.tt = Trii()
        for w in words:
            self.tt.add(w)
        self.active = []
            

    def query(self, letter: str) -> bool:
        o = ord(letter)-ord('a')
        #for i in range(len(self.active)):
            #self.active[i] = self.active[i].child[o]
        self.active = [x.child[o] for x in self.active if x.child[o] is not None]
            
        if self.tt.child[o] is not None:
            self.active.append(self.tt.child[o])
        for node in self.active:
            if node.isword:
                return True
        return False
