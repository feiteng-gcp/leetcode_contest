class TrieNode:
    def __init__ (self, char):
        self.char = char
        self.child = {}
        self.flag = False
        
class Trie:
    def __init__(self):
        self.root = TrieNode('')
        
    def add_word(self, word):
        node = self.root
        for c in word:
            if c not in node.child:
                node.child[c] = TrieNode(c)
            node = node.child[c]
        node.flag = True
        
class StreamChecker:

    def __init__(self, words: List[str]):
        self.trie = Trie()
        for word in words:
            self.trie.add_word(word)
        self.node_list = []

    def query(self, letter: str) -> bool:
        node_list = []
        flag = False
        for node in self.node_list:
            if letter in node.child:
                node_list.append(node.child[letter])
                if node.child[letter].flag:
                    flag = True
                
        if letter in self.trie.root.child:
            node_list.append(self.trie.root.child[letter])
            if self.trie.root.child[letter].flag:
                flag = True
                
        self.node_list = node_list
        return flag
