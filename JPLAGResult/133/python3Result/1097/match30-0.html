class StreamChecker:

    def __init__(self, words: List[str]):
        self.T=collections.defaultdict(dict)
        for w in words:
            cur=self.T
            for c in w:
                if c not in cur:
                    cur[c]={}
                cur=cur[c]
            cur['^']=None
        self.q=collections.deque([self.T])
    
    def query(self, letter: str) -> bool:
    
        nq=[self.T]
        flag=False
        while(self.q):
            temp_t=self.q.popleft()
            if letter in temp_t:
                nxt_t=temp_t[letter]
                if '^' in nxt_t:
                    flag=True
                nq.append(nxt_t)
        self.q.extend(nq)
        return flag
