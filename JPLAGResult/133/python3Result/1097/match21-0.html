class StreamChecker:

    def __init__(self, words: List[str]):
        self.trie = {}
        self.query_history = []
        
        for w in words:
            cur = self.trie
            for c in w[::-1]:
                if c not in cur:
                    cur[c] = {}
                cur = cur[c]
            cur['$'] = True
        # print(self.trie)

    def query(self, letter: str) -> bool:
        self.query_history.append(letter)
        cur = self.trie
        for c in reversed(self.query_history):
            if '$' in cur:
                return True
            if c not in cur:
                return False
            cur = cur[c]
        return '$' in cur
