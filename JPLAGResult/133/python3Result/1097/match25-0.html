class StreamChecker:

    def __init__(self, words: List[str]):
        words = set(words)
        self.trie = {}
        for w in words:
            curdict = self.trie
            for c in w:
                if c not in curdict:
                    curdict[c] = {}
                curdict = curdict[c]
            curdict["#"] = None
        self.possible_dict = []
        

    def query(self, letter: str) -> bool:      
        state = False
        possible_dict = []
        if letter in self.trie:
            possible_dict.append(self.trie[letter])
            if "#" in self.trie[letter]:
                state = True  
        for d in self.possible_dict:
            curdict = d
            if letter in curdict:
                curdict = curdict[letter]
                possible_dict.append(curdict)
                if "#" in curdict:
                    state = True
        self.possible_dict = possible_dict
        return state
