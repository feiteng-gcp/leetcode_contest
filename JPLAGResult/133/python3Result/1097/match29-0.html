from collections import defaultdict,deque
class StreamChecker:

    def __init__(self, words: List[str]):
        self.fs=defaultdict(set)
        self.maxl=0
        self.ts=set()
        for w in words:
            self.fs[len(w)].add(w)
            self.maxl=max(self.maxl, len(w))
            for x in range(len(w)):
                self.ts.add(w[len(w)-1-x:])
        self.input=deque()

    def query(self, letter: str) -> bool:
        self.input.append(letter)
        if len(self.input)>self.maxl:
            self.input.popleft()
        s=""
        for i in range(len(self.input)-1, -1, -1):
            s=self.input[i]+s
            if s not in self.ts: return False
           
            if s in self.fs[len(s)]: return True
        return False
