class Node:
    def __init__(self):
        self.children = {}
        self.isword = False

class StreamChecker:
    def __init__(self, words):
        self.queries = []
        self.root = Node()
        for word in words:
            node 
= self.root
            for c in word:
                if c not in node.children:
                    node.children[c] = Node()
                node = node.children[c]
            node.isword = True

    def query(self, letter):
        sol = False
        self.queries.append(self.root)
        queries 
= []
        for q in self.queries:
            if letter in q.children:
                queries.append(q.children[letter])
                if q.children[letter].isword:
                    sol = True
        self.queries = queries
        return sol
