class Solution:
    def printVertically(self, s: str) -> List[str]:
        words = s.split()
        max_len = max(len(w) for w in words)
        
        res 
= []
        for i in range(max_len):
            res.append(''.join([w[i] if i < len(w) else ' ' for w in words]).rstrip())
        
        return res
