class Solution:
    def printVertically(self, s: str) -> List[str]:
        s = s.split()
        ma = 0
        for c in s:
            ma = max(ma, len(c))
        n = len(s)
        for i in range(n):
            if len(s[i]) < ma:
                s[i] = list(s[i] + ' ' * (ma - len(s[i])))
            else:
                s[i] = list(s[i])
        
        res = []
        for l in zip(*s):
            res.append(''.join(l))
        for i in range(len(res)):
            res[i] = res[i].rstrip()
        return res
