class Solution:
    def printVertically(self, s: str) -> List[str]:
        s = s.split()
        col = len(s)
        row = max([len(w) for w in s])
        
        pos = [[' ' for _ in range(col)] for _ in range
(row)]
        print(s)
        for c in range(len(s)):
            for r in range(len(s[c])):
                #print(pos)
                pos[r][c] = s[c][r]
        
        result = []
        for row in pos:
            result.append(''.join(row).rstrip())
        return result
