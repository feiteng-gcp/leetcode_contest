class Solution:
    def printVertically(self, s: str) -> List[str]:
        tokens = s.split(' ')
        l = max(len(t) for t in tokens)
        results = []
        for i in range(l):
            cs = []
            for t in tokens:
                # print()
                if i >= len(t):
                    cs.append(' ')
                else:
                    cs.append(t[i])
            results.append(''.join(cs).rstrip())
        return results
