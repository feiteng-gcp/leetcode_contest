class Solution:
    def printVertically(self, s):
        words = s.split()
        max_len = max(map(len, words))
        res = []
        for i in range(max_len):
            new_word = ''
            for word in words:
                if i < len(word):
                    new_word += word[i]
                else:
                    new_word += ' '
            res.append(new_word.rstrip())
        return res
