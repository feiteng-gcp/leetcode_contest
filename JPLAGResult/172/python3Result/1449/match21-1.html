class Solution:
    def printVertically(self, s: str) -> List[str]:
        l = s.split(' ')
        size = 0
        
for word in l:
            size = max(size, len(word))
            
        res = ["" for _ in range(size)]
        
        for word in l:
            for i in range(size):
                if i >= len(word):
                    res[i] += " "
                else:
                    res[i] += word[i]
                    
        for i in range(size):
            res[i] = res[i].rstrip()
            
        return res
