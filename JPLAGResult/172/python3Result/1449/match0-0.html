class Solution:
    def printVertically(self, s: str) -> List[str]:
        s_list = s.split(' ')
        max_len = max([len(s_i) for s_i in s_list])
        ans = []
        for i in range(max_len):
            ans_i = ""
            for s_i in s_list:
                if i < len(s_i):
                    ans_i += s_i[i]
                else:
                    ans_i += " ";
            ans.append(ans_i.rstrip())
        return ans
