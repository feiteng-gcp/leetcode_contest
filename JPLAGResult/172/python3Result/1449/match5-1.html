class Solution:
    def printVertically(self, s: str) -> List[str]:
        words 
= s.split(" ")
        L = max([len(wo) for wo in words])
        out = []
        for i in range(L):
            s = ""
            for w in words:
                if i < len(w):
                    s += w[i]
                else:
                    s += " "
            
            
            out.append(s.rstrip())
            
        return out
