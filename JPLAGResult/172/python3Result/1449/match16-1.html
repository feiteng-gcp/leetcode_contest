class Solution:
    def printVertically(self, s: str) -> List[str]:
        words = s.split(" ")
        maxlen = max(len(w) for w in words)
        res = []
        for j in range(maxlen):
            cur = []
            for i in range(len(words)):
                if len(words[i])-1 >= j:
                    cur.append(words[i][j])
                else:
                    cur.append(
" ")
            res.append("".join(cur).rstrip())
        return res
