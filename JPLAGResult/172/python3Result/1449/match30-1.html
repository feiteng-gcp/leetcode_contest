class Solution:
    def printVertically(self, s: str) -> List[str]:
        words = s.split()
        max_len = max([len(w) for w in words])
        ans = ['' for i in range(max_len)]
        for i, w in enumerate(words):
            words[i] = words[i] + ' ' * (max_len - len(w))
        for w in words:
            for i, c in enumerate(w):
                ans[i] = ans[
i] + c
        for i, w in enumerate(ans):
            ans[i] = ans[i].rstrip()
        return ans
