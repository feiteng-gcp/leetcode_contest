class Solution:
    def printVertically(self, s: str) -> List[str]:
        words = s.split(' ')
        maxlen = 0
        for word in words: 
            maxlen = max(maxlen, len(word))
        res = []
        for i in range(maxlen):
            cur = ''
            for word in words:
                if len(word) >= i + 1:
                    cur = cur + word[i]
                else:
                    cur = cur + ' '
            res.append(cur.rstrip())
        return res
