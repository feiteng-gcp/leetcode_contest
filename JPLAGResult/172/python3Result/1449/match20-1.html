class Solution:
    def printVertically(self, s: str) -> List[str]:
        ws = s.split(" ")
        n = max(len(w) for w in ws)
        ans = []
        for i in range(n):
            curr = []
            for w in ws:
                if i < len(w):
                    curr.append(w
[i])
                else:
                    curr.append(" ")
            ans.append("".join(curr).rstrip())
        return ans
