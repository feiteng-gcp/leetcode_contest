class Solution:
    def printVertically(self, s: str) -> List[str]:
        L = 0
        words = s.split()
        for w in words:
            L = max(L, len(w))
        
        ans = ["" for _ in range(L)]
        for w in words:
            for i in range(L):
                if i >= len(w):
                    ans[i] += ' '
                else:
                    ans[i] += w[i]
        
        for i in range(L):
            ans[i] = ans[i].rstrip()
        
        return ans
