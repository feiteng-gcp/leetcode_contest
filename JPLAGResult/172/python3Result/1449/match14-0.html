class Solution:
    def printVertically(self, s: str) -> List[str]:
        words = s.split(" ")
        c = len(words)
        r = max([len(word) for word in words])

        
        ans = [[" " for _ in range(c)] for _ in range(r)]
        
        for i in range(len(words)):
            for j in range(len(words[i])):
                ans[j][i] = words[i][j]
        ans2 = []
        for row in ans:
            ans2.append("".join(row).rstrip())
        
        return ans2
