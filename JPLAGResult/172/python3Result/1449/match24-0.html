class Solution:
    def printVertically(self, s: str) -> List[str]:
        res = []
        arr = s.split(' ')
        l = max([len(w) for w in arr])
        for i in range(l):
            temp = ''
            for w in arr:
                if i < len(w):
                    temp += w[i]
                else:
                    temp += ' '
            temp = temp.rstrip()
            res.append(temp)
        return res
