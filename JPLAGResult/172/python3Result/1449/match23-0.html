class Solution:
    def printVertically(self, s: str) -> List[str]:
        s = s.split()
        size = len(s)
        ma = max([len(st) for st in s])
        st = [[' ' for i in range(size)] for j in range(ma)]
        for i, w in enumerate(s):
            for j, c in enumerate(w):
                st[j][i] = c
        return ["".join(r).rstrip() for r in st]
