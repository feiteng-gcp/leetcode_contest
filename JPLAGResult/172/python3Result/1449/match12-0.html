class Solution:
    def printVertically(self, s: str) -> List[str]:
        slist = s.split()
        ans = []
        maxlen = 0
        maxlenidx = 0
        for i in range(len(slist)):
            if len(slist[i]) > maxlen:
                maxlen = len(slist[i])
                maxlenidx = i
        for i in range(maxlen):
            word = []
            for j in range(len(slist)):
                if i < len(slist[j]):
                    word.append(slist[j][i])
                else:
                    word.append(' ')
            while word[-1] == " ":
                word.pop()
            ans.append(''.join(word))
        return ans
