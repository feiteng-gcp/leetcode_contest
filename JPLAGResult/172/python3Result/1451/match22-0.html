class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        table = []
        for i,r in enumerate(ranges):
            if r!=0:
                table.append((i-r if i-r>0 else 0,i+r))
        table.sort(key = lambda x: (x[0], -x[1]))
        curL = 0
        curR = 0
        count = 0
        for L,R in table:
            if L<=curL:
                if R>curR:
                    curR = R
                    if curR>=n:
                        return count+1
            else:
                if curL>=n:
                    return count
                if curR==curL:
                    return -1
                count+=1
                curL = curR
                if L<=curL:
                    curR = max(curR,R)
        if curL>=n:
            return count
        if curR>=n:
            return count+1
        return -1
