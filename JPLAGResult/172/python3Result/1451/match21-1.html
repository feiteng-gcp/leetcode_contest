class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:        
        stock = [[i - ranges[i], i + ranges[i]] for i in range(len(ranges))]
        stock.sort(key = lambda x: (x[0],x[1]))
        #print(stock)
        
        cnt = 0
        bound = 0
        idx = 0
        while bound<n:
            cnt += 1
            cur_max = -1
            for i in range(idx,n+1):
                if stock[i][0] <= bound:
                    cur_max = max(cur_max, stock[i][1])
                else: 
                    idx = i
                    break
            
            if cur_max == -1: return cur_max
            else:bound = cur_max
            
            #print(bound,idx)
            
        return cnt
