class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        myr = []
        for i in range(len(ranges)):
            myr.append((i-ranges[i],i+ranges[i]))
        myr.sort(key = lambda x:x[0])
        left = 0
        #print(myr)
        times = 0
        while left < n:
            mx = 0
            for i in range(0, len(myr)):
                if myr[i][0] > left:
                    break
                mx = max(myr[i][1], mx)
            if left >= mx:
                return -1
            left = mx
            times+= 1
        return times
