class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        r = []
        for i in range
(n + 1):
            r.append((i - ranges[i], i + ranges[i]))
        r.sort(key=lambda x:x[0])
        c = 0
        cov = 0
        p_cov = 0
        i = 0
        while i < len(r):
            s, e = r[i]
            if s <= cov:
                p_cov = max(p_cov, e)
            else:
                cov = p_cov
                c += 1
                if cov >= n:
                    return c
                if s > cov:
                    return -1 
                p_cov = max(p_cov, e)
            i += 1
        if p_cov >= n:
            return c + 1
        return -1
