class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        intervals = []
        for i in range(n + 1):
            intervals.append((max(0, i - ranges[i]), min(n, i + ranges[i])))
        # intervals = [
        #     (0, 5),
        #     (2, 7),
        #     (3, 6),
        #     (6, 9),
        #     (4, 10)
        # ]
        intervals.sort()
        ret = 0
        last = (-1, 0)
        for s, e in intervals:
            # print(last, s, e)
            if s > last[1]:
                return -1
            elif s == last:
                ret += 1
                last = (s, e)
            else:
                if e <= last[1]:
                    continue
                if s <= last[0]:
                    last = (last[0], e)
                else:
                    ret += 1
                    last = (last[1], e)
        return ret if last[1] == n else -1
