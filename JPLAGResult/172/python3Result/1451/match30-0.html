class Solution:
    def minTaps(self, n: int, r: List[int]) -> int:
        ans, i = 0, 0
        while i < n:
            j, maxC = i, i
            while j - i <= 100 and j <= n:
                if r[j] >=  (j - i): maxC = max(maxC, r[j] + j)
                j = j + 1
            if maxC == i: return -1
            ans = ans + 1
            i = maxC
            # print (ans, maxC)
        return ans
