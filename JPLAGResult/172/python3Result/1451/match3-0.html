class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        r = []
        for i, rg in enumerate(ranges):
            r.append((max(0, i-rg), min(n, i+rg)))
        
        count = 0
        rightmost = 0
        i = 0
        r.sort()

        '''
        while rightmost < n: 
            find next rightmost = max end point of all the intervals (starting at i) that start before or at rightmost
            increment i 
            if next rightmost <= rightmost: return -1
     
            rightmost = nextrightmost
            count += 1

        '''
        while rightmost < n:
            nextrightmost = rightmost
            while i < len(r) and r[i][0] <= rightmost:
                nextrightmost = max(nextrightmost, r[i][1])
                i += 1
            if nextrightmost <= rightmost: return -1
            rightmost = nextrightmost
            count += 1
        return count
