class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        
        segs = []
        for i, r in enumerate(ranges):
            segs.append((max(i - r, 0), i+r))
        segs.sort()
        # print(segs)
        cur = 0
        j = 0
        res = 0
        while cur < n and j <= n:
            nxt = cur
            while j<=n and segs[j][0] <= cur:
                nxt = max(nxt, segs[j][1])
                j+=1
            if cur == nxt:
                return -1
            res += 1
            cur = nxt
        return res
