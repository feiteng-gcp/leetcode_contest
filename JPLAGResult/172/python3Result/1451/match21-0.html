class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        clips = [[i-ranges[i], i+ranges[i]] for i in range(len(ranges))]
        clips.sort()
        prev = 0
        i = 0
        count = 0
        while i < len(clips):
            max_jump = prev
            while i < len(clips) and clips[i][0]<=prev:
                max_jump = max(max_jump, clips[i][1])
                i += 1
            count += 1
            if max_jump >= n:
                return count
            if max_jump == prev: # cannot move forward
                return -1
            prev = max_jump
        return -1
