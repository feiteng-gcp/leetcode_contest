class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        s = []
        for i in range(n+1):
            s.append([max(0, i-ranges[i]), min(n,i+ranges[i])])
        s.sort()
        s_pros = []
        for a,b in s:
            if not s_pros:
                s_pros.append([a,b])
            elif s_pros[-1][1] < b:
                if s_pros[-1][0] == a:
                    s_pros[-1][1] = b
                else:
                    s_pros.append([a,b])
        count = 0
        l,r = 0,0
        for i in range(len(s_pros)):
            if r == n:
                return count
            a,b = s_pros[i]
            if a <= r and b== n:
                return count + 1
            elif a <= r:
                if i == len(s_pros) - 1:
                    return -1
                a_next,b_next = s_pros[i+1]
                if a_next > r:
                    l,r = a,b
                    count += 1
            else:
                return -1
        return count
