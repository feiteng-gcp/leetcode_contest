class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        rs=[]
        for i in range(n+1):
            rs.append([max(i-ranges[i],0),min(i+ranges[i],n)])
        rs.sort()
        begin=rs[0][0]
        end=10000000000
        ind=0
        num=1
        dellist=[]
        while(ind<len(rs)-1):
            if rs[ind][0]==rs[ind+1][0]:
                del rs[ind]
            elif rs[ind][1]>=rs[ind+1][1]:
                del rs[ind+1]
            else:
                ind=ind+1
        #for ind in range(n):
        #    if rs[ind][0]==rs[ind+1][0]:
        #        dellist.append(ind)
        #    elif rs[ind][1]>rs[ind+1][1]:
        #        dellist.append(ind+1)
        #print(dellist)
        #for item in dellist[::-1]:
        #    del rs[item]    
        #print(rs)
        ind=1
        while ind<len(rs)-1:
            if rs[ind-1][1]>=rs[ind+1][0]:
                del rs[ind]
            else:
                ind=ind+1
        if rs[0][0]>0:
            return -1
        #print(rs)
        for i in range(len(rs)-1):
            if rs[i+1][0]>rs[i][1]:
                return -1
        return len(rs)
