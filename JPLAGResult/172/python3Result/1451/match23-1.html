class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        intervals = []
        for i, r in enumerate(ranges):
            if r != 0:
                intervals.append([i-r, i+r])
        intervals.sort()
        
        if not intervals:
            return -1

        coverage = 0
        
        res = 0
        
        i = 0
        size = len(intervals)
        
        print(intervals)
        
        while i < size:
            a, b = intervals[i]
            if a > coverage:
                return -1
            max_coverage = max(b, coverage)
            while i < size - 1 and intervals[i+1][0] <= coverage:
                i += 1
                max_coverage = max(max_coverage, intervals[i][1])
            coverage = max_coverage
            res += 1
            if coverage >= n:
                return res
            i += 1
            
        return -1
