class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        clips=[]
        for i in range(len(ranges)):
            clips.append([i-ranges[i],i+ranges[i]])
        T=n
        clips.sort()        
        cv,ret=0,0
        while cv<T:
            maxe=0
            change=False
            for i in clips:
                if i[0]<=cv:
                    if i[1]>maxe and i[1]>cv:
                        maxe=i[1]
                        change=True
            if not change:
                break
            cv=maxe
            ret+=1
        if cv>=T:
            return ret
        return -1
