class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        covers = []
        for i, cover_len in enumerate(ranges):
            if cover_len != 0:
                covers.append([i-cover_len, i+cover_len])
            
        covers.sort()
        #print(covers)
        
        left = 0
        cnt = 0
        while left < n:
            #print(left)
            max_right = left - 1
            for i, cover in enumerate(covers):
                if left >= cover[0]:
                    max_right = max(max_right, cover[1])
            
            if max_right <= left:
                return -1
            cnt += 1
            #print(left, max_right)
            left = max_right
        
        return cnt
