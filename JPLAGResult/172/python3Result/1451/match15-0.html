class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        # sort by starting, cut interval left to 0
        # use heap to keep the longest interval (reach farest)
        intl = []
        for i in range(n+1):
            if ranges[i] == 0:
                continue
            left = max(i - ranges[i], 0)
            right = i + ranges[i]
            intl.append([left,right])
        intl.sort()
       
        cnt = 0
        if not intl or intl[0][0] != 0:
            return -1
        #if intl[-1][1] < n:
        #    return -1
        #print(intl)
        heap = []
        currLeft = 0
        #heapq.heappush(heap, -intl[i][1])
        i = 0
        while i < len(intl):
            found= False
            while i < len(intl) and intl[i][0] <= currLeft:
                found = True
                heapq.heappush(heap, -intl[i][1])
                i += 1
            if not found:
                return -1
            cnt += 1
            currLeft = -heap[0]
            if currLeft >= n:
                return cnt
            heap = []
        return -1
