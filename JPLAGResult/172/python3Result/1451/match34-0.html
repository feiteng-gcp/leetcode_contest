class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        intervals = []
        for i in range(0, n+1):
            intervals.append((i - ranges[i], i + ranges[i]))
        intervals.sort(key=lambda x: (x[0], -x[1]))
        print(intervals)
        curr = intervals[0]
        num = 0
        i = 0
        target = 0
        while i < len(intervals):
            start, end = intervals[i]
            if start > 0:
                break
            if end > curr[1]:
                curr = (start, end)
            i += 1
        num += 1
        nextI = (float('-inf'), float('-inf'))
        while i < len(intervals):
            start, end = intervals[i]
            if start > curr[1]:
                curr = nextI
                num += 1
                if start > curr[1]:
                    return -1
            if end > nextI[1]:
                nextI = (start, end)
            i += 1
        if curr[1] < n:
            if nextI[1] > n:
                return num + 1
            else:
                return -1
        return num
