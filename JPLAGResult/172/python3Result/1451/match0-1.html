class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        room = []
        for i in range(n + 1):
            room.append([i-ranges[i], i+ranges[i]])
        room = sorted(room)
        if room[0][0] > 0:
            return -1
        reach = 0
        res = 0
        while reach < n + 1:
            tmp = reach
            for i in range(n+1):
                if room[i][0] > reach:
                    break
                if room[i][1] > tmp:
                    tmp = room[i][1]
            if reach != tmp:
                reach = tmp
                res += 1
                if reach == n:
                    break
            else:
                res = -1
                break
        return res
