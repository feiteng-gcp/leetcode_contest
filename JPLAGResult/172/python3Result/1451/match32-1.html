from typing import List


class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        lr = []
        for i in range(n+1):
            if ranges[i] == 0:
                continue
            l = max(0, i - ranges[i])
            r = i + ranges[i]
            lr.append((l, r))

        lr.sort()

        lenlr = len(lr)

        cover = -1
        cur = 0
        ans = 0
        for i in range(n+1):
            if i == n and i == cover:
                continue
            if i < cover:
                continue
            rightmost = -1
            while cur < lenlr and lr
[cur][0] <= i:
                rightmost = max(rightmost, lr[cur][1])
                cur += 1
            if rightmost == -1:
                return -1
            ans += 1
            cover = rightmost
        return ans
