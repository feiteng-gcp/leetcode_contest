class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        ranges = sorted([(i-rng, i+rng) for i, rng in enumerate(ranges)])
        x = 0
        num = 0
        i = 0
        while x < n:
            ends = []
            while i < len(ranges) and ranges[i][0] <= x:
                ends.append(ranges[i][1])
                i += 1
            if ends:
                x = max(ends)
                num += 1
            else:
                return -1
        return num
