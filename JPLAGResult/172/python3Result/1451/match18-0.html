class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        tmp=[]
        for i,v in enumerate(ranges):
            l,r=i-v,i+v
            if l>=0:
                tmp.append((l,r))
            else:
                tmp.append((0,r))
        tmp.sort()
        if tmp[0][0]>0:
            return -1
        right,end,cnt=0,0,0
        while end<len(tmp):
            n_r=-1
            while end<len(tmp) and tmp[end][0]<=right:
                n_r=max(n_r,tmp[end][1])
                end+=1
            if n_r==-1:
                return -1
            cnt+=1
            if n_r>=n:
                return cnt
            right=n_r
        return -1
