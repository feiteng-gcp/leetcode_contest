class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        a=[
(max(i-u,0),i+u) for i,u in enumerate(ranges)]
        a.sort(key=lambda x:(x[0],-x[1]))
        ans=0
        prev=0
        r=0
        for u,v in a:
            if u<=prev:
                r=max(r,v)
            else:
                if r<u: return -1
                prev=r
                r=max(r,v)
                
                ans+=1
            #print(u,v,ans,prev,r)
                
        if prev<n: ans +=1
        return ans
