class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        ranges = sorted([(i - r, i + r) for i, r in enumerate(ranges)])
        right_most = 0
        ans = 0
        pos = 0
        
while right_most < n:
            cur_max = right_most
            while pos < len(ranges) and ranges[pos][0] <= right_most:
                cur_max = max(cur_max, ranges[pos][1])
                pos += 1
            if cur_max == right_most:
                return -1
            right_most = cur_max
            ans += 1
        return ans
