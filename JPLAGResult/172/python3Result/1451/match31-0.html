class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        rngs = []
        for i in range(n+1):
            if ranges[i] != 0:
                rngs.append([max(0, i-ranges[i]), min(n, i+ranges[i])])
        rs = 0
        tmp = 0
        while tmp < n:
            arr = []
            for rang in rngs:
                if rang[0] <= tmp and rang[1] > tmp:
                    arr.append(rang[1])
            if not len(arr) and tmp < n:
                return -1
            tmp = max(arr)
            rs +=1
        return rs
