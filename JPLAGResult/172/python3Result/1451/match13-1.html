class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        dp=[1<<31]*(n+1)
        dp[0]=0
        for i,r in enumerate(ranges):
            sm=max(0,i-r)
            for j in range(sm+1,min
(i+r,n)+1):
                dp[j]=min(dp[j],dp[sm]+1)
        return dp[-1] if dp[-1]<(1<<31) else -1
