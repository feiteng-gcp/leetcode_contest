class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        taps = []
        for i, r in enumerate(ranges):
            a, b = max(0, i-r), min(i+r, n)
            taps.append((a, b))
        taps.sort(key=lambda x: x[0])
        last = 0
        res = 0
        i = 0
        while i < len(taps) and last < n:
            b = -1
            j = i
            while j < len(taps) and taps
[j][0] <= last:
                b = max(b, taps[j][1])
                j += 1
            if b <= last: return -1
            last = b
            res += 1
            i = j
        return res
