class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        list_ranges = [[max(0,idx-r), min(n,idx+r)] for idx, r in enumerate(ranges) if r>0]
        list_ranges = sorted(list_ranges, key = lambda x: (x[0])*1e5-x[1])
        print(list_ranges)
        
        start = 0
        idx = 0        
        re = 0
        length = len(list_ranges)
        while idx <length and start <n:
            # move idx to the left till list_ranges[idx] > start
            candidate_max = -1
            while idx <length and list_ranges[idx][0] <= start:
                candidate_max = max(candidate_max, list_ranges[idx][1])
                idx += 1
            #print("@@", start, idx, candidate_max)
            if candidate_max==-1:
                #print("@0", start, idx, candidate_max)
                return -1
            re += 1
            start = candidate_max
        if start == n:
            return re
        else:
            #print("@1", start, idx, candidate_max)
            return -1
