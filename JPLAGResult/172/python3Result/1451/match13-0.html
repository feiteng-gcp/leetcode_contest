class Solution:
    def minTaps(self, n: int, ranges: List[int]) -> int:
        dp = [n + 1] * (n + 1)
        
        r = [(max(0, i - x), min(n, i + x)) for i, x in enumerate(ranges) if x != 0]
        r.sort()
    
        dp[0] = 0
        
        for q in r:
            b, n = q
            for i in range(b + 1, n + 1):
                dp[i] = min(dp[i], dp[b]+1)
                
        
            
        return dp[n] if dp[n] <= n else -1
