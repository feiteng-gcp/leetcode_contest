# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def removeLeafNodes(self, root: TreeNode, target: int) -> TreeNode:
        def dfs(root, p, flag):
            if not root:
                return
            dfs(root.left, root, True)
            dfs(root.right, root, False)
            if root.val == target and root.left == None and root.right == None:
                if flag:
                    p.left = None
                else:
                    p.right = None
        
        r = TreeNode(0)
        r.left = root
        dfs(root, r, True)
        return r.left
