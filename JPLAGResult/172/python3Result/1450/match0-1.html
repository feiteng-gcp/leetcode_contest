# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def removeLeafNodes(self, root: TreeNode, target: int) -> TreeNode:
        
        
        def delete(node, parent, target, is_left):
            if node == None:
                return
            delete(node.left, node, target, True)
            delete(node.right, node, target, False)
            
            
            if node.val == target and node.left == None and node.right == None:
                if is_left:
                    parent.left = None
                else:
                    parent.right = None
        dummy = TreeNode(0)
        dummy.left = root
        delete(root, dummy, target, True)
        return dummy.left
