# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def removeLeafNodes(self, root: TreeNode, target: int) -> TreeNode:
        def yrh(node):
            if not node:
                return None
            node.left = yrh(node.left)
            node.right=yrh(node.right)
            if not node.left and not node.right and node.val==target:
                return None
            return node
        
        
        return yrh(root)
