class Solution {

public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int, int>> res;
        for (int i = 0; i <= n; ++i) {
            if (ranges[i] == 0) {
                continue;
            } 
            res.emplace_back(i - ranges[i], i + ranges[i]);
        }
        
        sort(res.begin(), res.end());
        int ans = 0;
        int best = -1;
        int now = 0;
        int cur = 0;
        for (int i = 0; i < n; ++i) {

            while (cur < res.size() && res[cur].first <= i) {
                best = max(best, res[cur++].second);
            }
            if (best <= i) {
                return -1;
            }
            i = best - 1;
            ++ans;
        }
        return ans;
    }
};
