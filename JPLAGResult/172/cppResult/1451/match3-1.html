class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        int res = 0;
        int now = 0;
        while(now < n){
            int tmp = now;

            rep(i,n+1){
                if(i - ranges[i] <= now && i + ranges[i] > tmp){
                    tmp = i + ranges[i];
                }
            }
            if(tmp == now){
                return -1;
            }
            now = tmp;
            res++;
        }
        return res;
    }
};
