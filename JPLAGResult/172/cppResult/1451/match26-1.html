class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int, int> > r;
        for(int i = 0; i <= n; ++i) {
            r.push_back(make_pair(i - ranges[i], i + ranges[i]));
        }
                        
        sort(r.begin(), r.end());
        
        int res = 0, left = 0, right = INT_MIN;
        for(int i = 0; i <= n; ++i) {
            if(r[i].first <= left) {
                right = r[i].second;
                while(r[i].first <= left) {
                    right = max(right, r[i].second);
                    ++i;
                    if(i > n) break;
                }
                --i;
                left = right;
                ++res;
            }
            if(left >= n) break;
        }
        if(right >= n) return res;
        else return -1;
    }
};
