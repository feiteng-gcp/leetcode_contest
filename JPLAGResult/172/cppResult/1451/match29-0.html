class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<int> dp(n + 1, -1);
        dp[0] = 0;
        for (int i = 0; i <= n; ++i) {
            if (ranges[i] == 0)
                continue;
            int x = max(0, i - ranges[i]);
            // if (dp[x] == -1)
            //     break;
            int pre = x == 0 ? 0 : dp[x];
            if (pre == -1)
                continue;
            for (int j = x + 1; j <= min(n, i + ranges[i]); ++j) {
                // printf("%d %d\n", j, dp[j]);
                dp[j] = dp[j] == -1 ? pre + 1 : min(dp[j], pre + 1);
                // if (j > 9990)
                //     printf("%d %d\n", j, dp[j]);
            }
        }
        return dp[n];
    }
};
