class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int, int>> data;
        for (int i=0; i<=n; i++) {
            data.push_back(make_pair(max(0, i-ranges[i]), min(n, i+ranges[i])));
        }
        sort(data.begin(), data.end());
        vector<int> dp(n+1, INT_MAX-1); // need at least dp[x] to cover 0~x;
        dp[0]=0;
        for (auto p:data) {
            for (int i=p.first; i<p.second; i++) {
                dp[p.second]=min(dp[p.second], 1+dp[i]);
            }
        }
        return dp[n]==INT_MAX-1 ? -1 : dp[n];
    }
};
