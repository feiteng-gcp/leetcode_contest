class Solution {
public:
    int minTaps(int n, vector<int>& a) {
        vector<pair<int,int>>b;
        for(int i=0;i<=n;i++){
            b.push_back(make_pair(i-a[i],i+a[i]));
        }
        sort(b.begin(),b.end());
        int cover=0;
        int i=0;
        if(b[i].first>0)return -1;
        while(i<b.size()&&b[i].first<=0){
            cover=max(cover,b[i].second);
            i++;
        }
        int ret=1;
        while(i<b.size()){
            if(cover>=n)return ret;
            int mx=cover;
            while(i<b.size()&&b[i].first<=cover){
                mx=max(mx,b[i].second);
                i++;
            }
            if(mx==cover)return -1;
            ret++;
            cover=mx;
        }
        if(cover>=n)
            return ret;
        else
            return -1;
    }
};
