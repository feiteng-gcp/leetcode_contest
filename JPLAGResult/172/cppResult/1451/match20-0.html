class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int, int>> dp;
        for (int i = 0; i < ranges.size(); i++) {
            dp.push_back(make_pair(max(0, i - ranges[i]), i + ranges[i]));
        }
        sort(dp.begin(), dp.end());
        if (dp[0].first > 0) return -1;
		int e = 0, rb = 0;
        int ans = 0;
        int i = 0;
        int k = 0; 
        int check = 1;
        while(rb < n)
        {
            if(dp[k].first > rb)
            {
                check = 0;
                break;
            }
            for(i = k; i <= n; i++)
            {
                if(dp[i].first <= rb)
                {
                    if(dp[i].second >= e)
                        e = dp[i].second;
                }
                else{
                    ans++;
                    rb = e;
                    k = i;
                    break;
                }
            }
            if(i == n + 1)
            {
                ans++;
                break;
            }
        }
        return check == 1 ? ans : -1;
    }
};
