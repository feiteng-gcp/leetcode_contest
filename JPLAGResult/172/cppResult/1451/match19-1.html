class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector <int> cover(n + 1);
        for (int i = 0; i <= n; i++) {
        int coverS = max(0, i - ranges[i]);
        int coverE = min(n, i + ranges[i]);
        cout << coverS << " " << coverE << " " << ranges[i] << endl;
        cover[coverS] = max(cover[coverS], coverE - coverS);
        }
        
//         for (int i = 0; i <= n; ++i) {
//             cout << cover[i] << " ";
//         }
        
//         cout << endl;
        
       int maxReachPos = cover[0];
       int curMaxReachPos = cover[0];
       int curStep = 1;
       for(int i = 1; i <= min(n, maxReachPos); i++) {
        curMaxReachPos = max(curMaxReachPos, i + cover[i]);
        if(i == n){
            return curStep;
        }
        if(i == maxReachPos){
            maxReachPos = curMaxReachPos;
            curStep++;
        }
      }
        
        return -1;
    }
};
