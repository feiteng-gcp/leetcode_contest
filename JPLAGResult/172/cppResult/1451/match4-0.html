class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        int sum = 0;
        int pos = 0;
        while (pos < n) {
            int longest = pos + ranges[pos];
            for (int i = pos + 1; i <= n; ++i) {
                if (i - ranges[i] <= pos) {
                    longest = max(longest, i + ranges[i]);
                }
            }
            if (longest == pos) {
                return -1;
            }
            ++sum;
            pos = longest;
        }
        return sum;
    }
};
