class Solution {

public:
    int minTaps(int n, vector<int>& ranges) {
        vector<int>  dp(n+1, n+2);
        //dp[i][j] = min cost of watering j using i, -1 if impossible
        for (int i = 0; i<=n; i++) {
            if (i == 0) {
                for (int j = 0; j <= ranges[i]; j++) {
                    dp[j] = 1;
                }
            } else if (ranges[i]>0){
                
                vector<int> newdp;
                int cost;
                if (i - ranges[i] <= 0) cost = 0;
                else cost = dp[i - ranges[i]];

                for (int j = max(0, i - ranges[i]); j <= min(i+ranges[i], n); j++) {
                    dp[j] = (min(dp[j], cost+1));
                }
            }
        }
        return (dp[n]==n+2)?-1:dp[n];
    }
};
