class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<int> left(n+1, 0);
        vector<int> right(n+1, n);
        for (int i = 0; i <= n; i++) {
            left[i] = max(0, i-ranges[i]);
            right[i] = min(n, i+ranges[i]) - 1;
        }
        int toFind = 0;
        int cnt = 0;
        while (toFind < n) {
            int rightMost = toFind-1;
            int id = -1;
            for (int i = 0; i <= n; i++) {
                if (left[i] <= toFind && right[i] >= toFind && right[i] > rightMost) {
                    rightMost = right[i];
                    id = i;
                }
            }
            if (toFind > rightMost)
                return -1;
            toFind = rightMost + 1;
            cnt++;
        }
        return cnt;
    }
};
