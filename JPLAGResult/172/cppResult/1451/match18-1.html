class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {

        int curMin = 0;
        int ans = 0;
        while(curMin < n){
            int nextMin = curMin, idx = -1;
            for(int i = 0; i < ranges.size(); i++){
                if(ranges[i] >= 0 && curMin >= i - ranges[i] && curMin <= i + ranges[i] && i + ranges[i] >= nextMin){
                    nextMin = i + ranges[i];
                    idx = i;
                }
            }
            if(nextMin == curMin){
                return -1;
            }else{
                // cout << "idx = " << idx << ", range[i] = " << ranges[idx] <<  endl;
                curMin = nextMin;
                ranges[idx] = -1;
                ans++;
            }
        }
        return ans;
        
    }
};
