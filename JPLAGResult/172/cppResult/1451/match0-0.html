class Solution {
public:
    static bool cmp(vector<int>& vec1,vector<int>& vec2){
        return vec1[0]<vec2[0];
    }
    
    int minTaps(int n, vector<int>& ranges) {
        vector<vector<int>> tmp;
        for(int i=0;i<ranges.size();i++){
            tmp.push_back({i-ranges[i],i+ranges[i]});
        }
        sort(tmp.begin(),tmp.end(),cmp);
        for(int i=0;i<tmp.size();i++){
            cout<<tmp[i][0]<<" "<<tmp[i][1]<<endl;
        }
        int prev=0;
        int cur=0;
        int count=0;
        while(cur<tmp.size()){
            int maxL=0;
            int flag=0;
            while(cur<tmp.size()&&tmp[cur][0]<=prev){
                flag=1;
                if(tmp[cur][1]>maxL){
                    maxL=tmp[cur][1];
                }
                cur++;
            }
            count++;
            cout<<count<<endl;
            prev=maxL;
            if(prev>=n){
                return count;
            }
            if(flag==0){
                cur++;
            }
        }
        return -1;
    }
};
