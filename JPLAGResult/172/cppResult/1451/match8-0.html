class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<int> dp(n+1,1000000);
        dp[0]=0;
        for(int i=0;i<=n;i++){
            int num=ranges[i];
            int minv;
            if(i-num<0)
                minv=0;
            else{
                minv=dp[i-num];
            }
            for(int j=i-num;j<=i+num;j++){              
                if(j>=0&&j<=n){
                    dp[j]=min(minv+1,dp[j]);
                }
            }
        }
        if(dp[n]==1000000)
            return -1;
        return dp[n];
    }
};
