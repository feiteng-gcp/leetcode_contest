class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<int> dp(n + 1, INT_MAX - 1);
        dp[0] = 1;
        for (int i = 0; i <= n; i++) {
            int l = i - ranges[i];
            int r = min(i + ranges[i], n);
            if (l <= 0) {
                for (int k = 0; k <= r; k++) dp[k] = 1;
            }
            else {
                if (dp[l] == INT_MAX) continue;
                for (int k = l + 1; k <= r; k++) dp[k] = min(dp[k], dp[l] + 1);
            }
        }
        if (dp[n] >= INT_MAX - 1) return -1;
        return dp[n];
    }
};
