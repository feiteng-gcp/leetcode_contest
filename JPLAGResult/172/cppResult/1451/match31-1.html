bool comp(pair<int,int>& a, pair<int,int>& b) {
    if(a.first != b.first) {
        return a.first < b.first;
    }
    else {
        return a.second > b.second;
    }
}


class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int,int>> v;
        for(int i = 0; i <= n; i++) {
            v.push_back(make_pair(i-ranges[i],i+ranges[i]));
        }
        sort(v.begin(),v.end(), comp);
        int ans = 0;
        int x = 0;
        int position = 0;
        int maxValue = -1;
        while(x < n) {
            for(; position < v.size() && v[position].first <= x; position++) {
                maxValue = max(maxValue,v[position].second);
            }
            if(maxValue <= x) return -1;
            ans++;
            x = maxValue;
        }
        if(x < n) return -1;
        return ans;
    }
};
