typedef pair<int, int> pii;


class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pii> arr(n + 1);
        for (int i = 0; i <= n; ++i) {
            arr[i].first = i - ranges[i];
            arr[i].second = i + ranges[i];
        }
        sort(arr.begin(), arr.end());
        // for (int i = 0; i <= n; ++i) {
        //     cout << arr[i].first << " " << arr[i].second << endl;
        // }
        int maxReach = 0;
        int idx = 0;
        int ans = 0;
        while(maxReach < n) {
            ++ans;
            int reach = 0;
            while(idx <= n && arr[idx].first <= maxReach) {
                reach = max(reach, arr[idx].second);
                // cout << idx << " - " << reach << endl;
                ++idx;
            }
            if (reach <= maxReach) { break; }
            else maxReach = reach;
        }
        return maxReach >= n ? ans : -1;
    }
};
