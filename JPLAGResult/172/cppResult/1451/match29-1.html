class Solution {
    const int inf = 1000000;

public:
    int minTaps(int n, vector<int>& ranges)
    {
        vector<int> min(n + 1, inf);
        min[0] = 0;
        for (int i = 0; i <= n; i++) {
            if (ranges[i] == 0) continue;
            int count = i - ranges[i] >= 0 ? min[i - ranges[i]] + 1 : 1;
            int min_j = std::max(0, i - ranges[i]);
            int max_j = std::min(n, i + ranges[i]);
            for (int j = max_j; j > min_j; j--)
                min[j] = std::min(min[j], count);
        }
        return min.back() == inf ? -1 : min.back();
    }
};
