class Solution {
public:
    int minTaps(int n, vector<int>& ranges) 
    {
        int nBo = 0;
        int nSize = 0;
        while (nBo < n)
        {
            int nNextBo = nBo;
            for (int i = 0; i < ranges.size(); i++)
            {
                if (i - ranges[i] > nBo)
                {
                    continue;
                }
                nNextBo = max(nNextBo, i + ranges[i]);
            }
            if (nNextBo == nBo)
            {
                return -1;
            }
            nBo = nNextBo;
            nSize+=1;
        }
        return nSize;
    }
};
