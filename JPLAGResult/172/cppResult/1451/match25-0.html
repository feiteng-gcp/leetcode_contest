class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int,int>> intervals;
        
        for (int i = 0; i < ranges.size(); ++i) {
            intervals.emplace_back(i - ranges[i], i + ranges[i]);
        }
        
        sort(intervals.begin(), intervals.end());
        
        multiset<int, greater<int>> M;
        
        int cur = 0;
        int res = 0;
        for (int i = 0; i <= n && cur < n;) {
            while (i <= n && intervals[i].first <= cur) {
                M.insert(intervals[i].second);
                ++i;
            }
            if (M.empty()) {
                break;
            }
            int nxt = *M.begin();
            M.erase(M.begin());
            ++res;
            cur = nxt;
        }
        
        return cur >= n ? res : -1;
    }
};
