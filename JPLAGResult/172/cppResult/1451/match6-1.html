class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<int> dp(n + 1, n + 2);
        // vector<int> s(n + 1);
        // int m = 1;
        // s[0] = 0;
        dp[0] = 0;
        for (int i = 0; i <= n; i++) {
            int pre = i - ranges[i];
            if (pre < 0) pre = 0;
            int t = dp[pre] + 1;
            int r = ranges[i] + i;
            if (r > n) r = n;

            for (int j = pre + 1; j <= r; j++) {
                dp[j] = min(dp[j], t);
            }
        }
        return dp[n] > n + 1? -1: dp[n];
    }
};
