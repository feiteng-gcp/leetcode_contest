class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int,int>> vp;
        for(int i=0;i<=n;i++){
            vp.push_back(make_pair(max(0,i-ranges[i]),min(n,i+ranges[i])));
        }
        sort(vp.begin(),vp.end(),[&](pair<int,int>& a, pair<int,int>& b)->bool{
           if(a.first==b.first) return a.second>b.second;
            return a.first<b.first;
        });
        int last=vp[0].second;
        int ans=1;
        int potential=vp[0].second;
        for(int i=1;i<=n;i++){
            if(vp[i].first>last){
                if(potential>=vp[i].first){
                    ans++;
                    last=potential;
                }
                else return -1;
            }
            potential=max(potential,vp[i].second);
        }
        if(last<n){
            if(potential>=n) ans++;
            else return -1;
        }
        return ans;
    }
};
