class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
//         map<int,pair<int,int>> segs;//start-><start,end>
        
//         for(int i=0;i < ranges.size();i++){
//             int x = ranges[i];
//             if(x == 0){
//                 continue;
//             }
//             int low = max(0,i-ranges[i])
//         }
        
        vector<int> dp(n+1,INT_MAX);
        for(int i=0;i < ranges.size();i++){
            if(ranges[i] == 0){
                continue;
            }
            int low = i-ranges[i];
            int precost = INT_MAX;
            if(low<= 0){
                precost = 0;
            }
            if(precost == INT_MAX){
                precost = dp[low];
            }
            if(precost == INT_MAX){
                //return -1;
                continue;
            }
            for(int j=0;j <= ranges[i] && i + j <= n;j++){
                dp[i+j] = min(dp[i+j],precost+1);
            }
        }
        if(dp[n] == INT_MAX){
            return -1;
        }
        return dp[n];
        
        
        
    }
};
