class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int, int>> arr(ranges.size());
        for (int i = 0; i < ranges.size(); i++) {
            arr[i] = make_pair(i - ranges[i], i + ranges[i]);
        }
        sort(arr.begin(), arr.end());
        int i = 0, cur = 0;
        int count = 0;
        while (i < n and cur < arr.size()) {

            int right = i;
            while (cur < arr.size() and arr[cur].first <= i) {
                right = max(right, arr[cur].second);
                cur++;
            }
            if (i == right) return -1;
            i = right;
            count++;
        }
        // if (i < n and cur == arr.size()) return -1;
        return count;
    }
};
