class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int, int>> pairs;
        for (int i = 0; i < ranges.size(); ++i) {
            pairs.emplace_back(max(i - ranges[i], 0), min(i + ranges[i], n));
        }
        sort(pairs.begin(), pairs.end());

        int count = 0;
        int j = 0;
        for (int i = 0; i <= 2 * n; ++i) {
            double curr = i / 2.0;
            printf("%d\n", i);
            if (j == pairs.size() || pairs[j].first > curr) {
                return -1;
            }
            
            ++count;
            int max_range = pairs[j].second;
            
            while (j + 1 < pairs.size() && pairs[j + 1].first <= curr) {
                ++j;
                max_range = max(max_range, pairs[j].second);
            }
            i = max_range * 2;
            
            ++j;
        }
        return count;
    }
};
