class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int,int>> t;
        for(int i=0;i<ranges.size();i++) {
            //if(ranges[i] == 0) continue;
            t.push_back(make_pair(max(0, i-ranges[i]), min(n, i+ranges[i])));
        }
        auto comp = [](const pair<int,int>& a, const pair<int, int>& b) {
            if(a.first != b.first)
                return a.first < b.first;
            return a.second > b.second;
        };
        sort(t.begin(), t.end(), comp);
        
        int count = 1;
        if(t.front().first != 0) return -1;
        int limit = t.front().second, next_limit = t.front().second;
        
        for(int i=1;i<t.size() && limit < n;i++) {
            if(t[i].first <= limit) {
                next_limit = max(next_limit, t[i].second);
            }
            else if(next_limit < t[i].first){
                return -1;
            }
            else {
                limit = next_limit;
                next_limit = max(next_limit, t[i].second);
                count++;
            }
        }
        if(limit < n && next_limit >= n) {
            limit = n;
            count++;
        }
        if(limit < n) return -1;
        
        return count;
    }
};
