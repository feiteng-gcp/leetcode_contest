class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int, int>>  intervals(size(ranges));
        for (int i = 0; i < size(ranges); ++i) {
            intervals[i] = make_pair(max(0, i - ranges[i]), min(n, i + ranges[i]));
        }
        sort(begin(intervals), end(intervals), [](const pair<int,int> &a, const pair<int,int> &b) {
            return a.first < b.first || a.first == b.first && a.second > b.second;
        });
        vector<pair<int,int>> reducedIntervals;
        reducedIntervals.push_back(intervals[0]);

        int prevLeft = intervals[0].first;
        int rightMost = intervals[0].second;
        for (int i = 1; i < size(intervals); ++i) {
            if (prevLeft != intervals[i].first && intervals[i].second > rightMost) {
                reducedIntervals.push_back(intervals[i]);
                prevLeft = intervals[i].first;
                rightMost = intervals[i].second;
            }
        }
        int pos = 0;
        auto start = begin(reducedIntervals);
        int cnt = 0;
        while (pos < n) {
            ++cnt;
            auto it = upper_bound(start, end(reducedIntervals), pos, [](int val, const pair<int,int> &p) {
                return val < p.first;
            });
            if (start == it) return -1;
            pos = prev(it)->second;
            start = it;
        }
        return cnt;
    }
};
