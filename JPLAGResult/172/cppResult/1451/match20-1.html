class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int,int>> rgs(n+1);
        for(int i = 0; i < n+1; i++) {
            rgs[i] = {i - ranges[i], i + ranges[i]};
        }
        sort(rgs.begin(), rgs.end());

        int max_r = 0;
        int cnt = 0;
        int x = 0;
        int i = 0;
        while(i < n+1) {
            if(rgs[i].first <= x) {
                max_r = max(max_r, rgs[i].second);
                i++;
            } else {
                if(max_r <= x) return -1;
                cnt++;
                x = max_r;
                if(x >= n) return cnt;
                max_r = 0;
            }
        }
        if(max_r < n) return -1;
        cnt++;
        return cnt;
    }
};
