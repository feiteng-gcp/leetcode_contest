class Solution {
    static bool cmp(pair<int, int>& x, pair<int, int>& y) {
        return x.first == y.first ? x.second > y.second : x.first < y.first;
    }
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int, int>> intervals;
        for (int i = 0; i <= n; i++) {
            intervals.push_back({i - ranges[i], i + ranges[i]});
        }
        sort(intervals.begin(), intervals.end(), cmp);
        int result = 0;
        int maxRight = -1;
        int maxPos = -1;
        int i = 0;
        while (i <= n && intervals[i].first <= 0) {
            if (intervals[i].second > maxRight) {
                maxRight = intervals[i].second;
                maxPos = i;
            }
            i++;
        }
        if (maxRight < 0) {
            return -1;
        }
        result++;
        i = maxPos + 1;
        while (i <= n && maxRight < n) {
            if (intervals[i].first > maxRight) {
                return -1;
            }
            int curMaxRight = -1;
            while (i <= n && intervals[i].first <= maxRight) {
                curMaxRight = max(curMaxRight, intervals[i].second);
                i++;
            }
            if (curMaxRight <= maxRight) {
                return -1;
            }
            result++;
            maxRight = curMaxRight;
        }
        if (maxRight < n) {
            return -1;
        } else {
            return result;
        }
    }
};
