class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        int ptr = 0, rb = 0;
        int cnt = 0;
        while (true) {
            int np = ptr, nb = rb;
            for (int i = ptr; i <= n; ++i) {
                if (i-ranges[i] <= rb && i+ranges[i] >= nb) {
                    np = i;
                    nb = i+ranges[i];
                }
            }
            if (nb == rb)
                return -1;
            cnt++;
            if (nb >= n)
                break;
            ptr = np;
            rb = nb;
        }
        return cnt;
    }
};
