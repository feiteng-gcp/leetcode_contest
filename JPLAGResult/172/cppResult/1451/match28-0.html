class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<vector<int>> intervals;
        for (int i = 0; i < ranges.size(); i++) {
            intervals.push_back({i-ranges[i],i+ranges[i]});
            // cout<<i-ranges[i]<< " "<<i+ranges[i]<<endl;
        }
        // cout<<"sorted"<<endl;
        sort(intervals.begin(),intervals.end());
        // for (int i = 0; i < ranges.size(); i++) {
            // cout<<intervals[i][0]<<" "<<intervals[i][1]<<endl;
        // }
        int curr = 0;
        int i = 0;
        int res = 0;
        while (curr <= n) {
            int r = curr;
            if (i >= n+1 or intervals[i][0] > curr) return -1;
            while (i < n+1 and intervals[i][0] <= curr) {
                r = max(r, intervals[i][1]);
                i++;
            }
            // cout<<" cover up to "<<r<<endl;
            res++;
            curr = r;
            if (curr == n) return res;
        }
        return res;
    }
};
