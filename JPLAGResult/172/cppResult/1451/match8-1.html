class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<int> dp(20006, INT_MAX - 20000);
        dp[0] = 0;
        for(int i = 0; i < n + 1; i++) {
            int mi = INT_MAX - 20000;
            int j = 0;
            for(j = max(0, i - ranges[i]); j <= i; j++) {
                mi = min(mi, dp[j]);
            }
           // cout<<mi<<" =="<<endl;
            for(int k = i; k <= i + ranges[i]; k++) {
                dp[k]  = min(dp[k], mi + 1);
          //      cout<<dp[k]<<" ";
            }
        //    cout<<endl;
        }
        
        return dp[n] > 10006 ? -1 : dp[n] ;
    }
};
