class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        vector<pair<int,int>> v;
        for(int i = 0; i < ranges.size(); i ++){
            if (ranges[i] == 0) continue;
            v.push_back(make_pair(max(0, i - ranges[i]), min(n, i + ranges[i])));
        }
        sort(v.begin(), v.end());
        int res = 0;

        int i = 0, x = 0;
        while(i < v.size()){
            if (v[i].first > x) return -1;
            int maxRight = 0;
            while(i < v.size() && v[i].first <= x){
                maxRight = max(maxRight, v[i].second);
                i ++;
            }
            res ++;
            x = maxRight;
            if (x >= n) break;
        }
        if (x < n) return -1;
        return res;
    }
};
