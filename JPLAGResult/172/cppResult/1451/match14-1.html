struct NODE {
    int lf, rt;
    NODE(int a, int b): lf(a), rt(b) {}
};

bool operator < (const NODE& a, const NODE& b) {
    return a.lf == b.lf ? a.rt > b.rt : a.lf < b.lf;
}


class Solution {
public:
    int minTaps(int n, vector<int>& ranges) {
        map<int, int> A;
        for (int i = 0; i <= n; ++i) A[i - ranges[i]] = max(A[i - ranges[i]], i + ranges[i]);
        int cur = 0, res = 0;
        auto p = A.begin();
        while (cur < n) {

            int k = cur;
            while (p != A.end() && p->first <= cur) {
                k = max(k, p->second);
                ++p;
            }
            if (k == cur) return -1;
            ++res;
            cur = k;
        }
        return res;
    }
};
