class Solution {
public:
    vector<string> printVertically(string s) {
        stringstream ss(s);
        vector<string> orig;
        int ml = 0;
        while (ss >> s) {
            ml = max(ml, (int)s.size());
            orig.push_back(s);
        }
        vector<string> ans(ml);
        int ls = orig.size();
        for (int i = 0; i < ml; i++) {
            for (int j = 0; j < ls; j++) {
                if (i < orig[j].size()) ans[i].push_back(orig[j][i]);
                else ans[i].push_back(' ');
            }
        }
        
        for (string& s : ans) while (!s.empty() && s.back() == ' ') s.pop_back();
        
        
        return ans;
        
        
        
    }
};
