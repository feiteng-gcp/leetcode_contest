class Solution {
public:
    vector<string> printVertically(string s) {
        
        int n = s.size();
        
        string str = "";
        int m = 0;
        
        vector<string> a;
        
        for (int i = 0; i < n; i++)
        {
            if (s[i] == ' ')
            {
                a.push_back(str);
                str = "";
            }
            else
                str.push_back(s[i]);
        }
        
        if (str.size() > 0)
            a.push_back(str);
        
        int x = a.size();
        for (int i = 0; i < x; i++)
        {
            //cout << a[i] << endl;
            m = max(m, (int)a[i].size());
        }
        
        vector<string> ans(m, "");
        

        for (int j = 0; j < m; j++)
        {
            for (int i = 0; i < x; i++)
            {
                if (j < a[i].size())
                {
                    //printf("aij:%c ", a[i][j]);
                    ans[j].push_back(a[i][j]);
                }
                else
                {
                    char tmp = ' ';
                    ans[j].push_back(tmp);      
                }
            }
        }
        
        for (int fuck = 0; fuck <= m - 1; fuck ++)
        {
        int pos = 0;
        for (int i = ans[fuck].size() - 1; i >= 0; i--)
            if (ans[fuck][i] != ' ')
            {
                pos = i;
                break;
            }
        
        string t = "";
        for (int i = 0; i <= pos; i++)
            t += ans[fuck][i];
        
        ans[fuck] = t;
        }
        
        
        return ans;
        
        
    }
};
