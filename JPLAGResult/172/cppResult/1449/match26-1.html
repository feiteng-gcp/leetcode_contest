class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> words;
        int n=s.size();
        int index=0;
        int ma=0;
        while(index<n){
            int last=index;
            while(last<n && s[last]!=' ') last++;
            string temp=s.substr(index,last-index);
             ma=max(ma,last-index);
            //cout<<index<<" "<<last<<endl;
            index=last+1;
            words.push_back(temp);
           
        }
        if(index<n){
            string temp=s.substr(index,n-index);
            words.push_back(temp);
            ma=max(ma,n-index);
        } 
        vector<string> ans=vector<string>(ma);

        for(int i=0;i<ma;i++){
            ans[i].resize(words.size());
            for(int j=0;j<words.size();j++){
                if(i<words[j].size()){
                    ans[i][j]=words[j][i];
                }
                else ans[i][j]=' ';
            }
            while(*ans[i].rbegin()==' ') ans[i].pop_back();
        }
        return ans;
    }
};
