class Solution {
public:
    vector<string> printVertically(string s) {
        istringstream ss(s);
        vector<string> words;
        int n = 0;
        while (ss) {
            string t;
            ss>>t;
            words.push_back(t);
            int tmp = t.size();
            n=max(n, tmp);
        }
        vector<string> res(n);
        for (int i = 0; i < n; i++) { 
            for (int j = 0; j < words.size(); j++) {
                if (i < words[j].size()) {
                    res[i] += words[j][i];
                } else {
                    res[i] += " ";
                }
            }
        }
        
        for (int i = 0; i < res.size(); i++) {
            size_t end = res[i].find_last_not_of(" ");
            if (end != string::npos) res[i] = res[i].substr(0, end+1);
        }
        return res;
    }
};
