class Solution {
public:
    vector < string > process(string s) {
        vector < string > res;

        s += ' ';
        string cur;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == ' ') {
                res.push_back(cur);
                cur = "";
            } else {
                cur += s[i];
            }
        }
        return res;
    }
    vector<string> printVertically(string s) {
        auto w = process(s);

//        cout<<w.size()<<endl;
//        for (int i = 0; i < w.size(); i++) cout<<w[i]<<endl;


        int maxx = 0;
        for (int i = 0; i < w.size(); i++) maxx = max(maxx, (int)w[i].size());

        vector < string > res(maxx);
        for (int j = 0; j < maxx; j++) {
            for (int i = 0; i < w.size(); i++) {
                char c = j < w[i].size() ? w[i][j] : ' ';
                res[j] += c;
            }
        }
        for (int i = 0; i < res.size(); i++) {
            while (res[i].back() == ' ') res[i].pop_back();
        }

//        cout<<res.size()<<endl;
//        for (int i = 0; i < res.size(); i++) {
//            cout<<res[i].size()<<": "<<res[i]<<endl;
//        }

        return res;
    }
};
