class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> ret;
        int idx=0;
        int idx2=0;

        s+=' ';
        for (int i=0; i<s.size(); i++) {
            if (s[i]==' ') {
                idx++;
                idx2=0;
            } else {
                if (ret.size()<=idx2) {
                    ret.push_back("");
                    for (int x=0; x<idx; x++) {
                        ret.back()+=' ';
                    }

                }
                while (ret[idx2].size()<idx) {
                    ret[idx2]+=' ';
                }
                ret[idx2]+=s[i];
                idx2++;
            }
        }
        return ret;
    }
};
