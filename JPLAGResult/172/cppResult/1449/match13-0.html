class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> words;
        stringstream ss(s);
        string curr;
        
        size_t col = 0;
        while (ss >> curr) {
            words.push_back(curr);
            col = max(col, curr.size());
        }
        
        vector<string> res;
        for (int c = 0; c < col; ++c) {
            string word;
            for (auto& w : words) {
                if (w.size() > c) {
                    word += w[c];
                } else {
                    word += ' ';
                }
            }
            
            while (word[word.size() - 1] == ' ') {
                word.pop_back();
            }
            
            res.push_back(word);
        }
        
        return res;
    }
};
