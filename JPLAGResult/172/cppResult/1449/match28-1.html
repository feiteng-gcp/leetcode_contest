class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> vec;
        int j = 0;
        int mx = 0;
        for(int i = 0; i <= s.size(); i++) {
            if(i == s.size() || s[i] == ' ') {
                vec.push_back(s.substr(j, i - j));
                mx = max(mx, i - j);
                j = i + 1;
            }
        }
      //  cout<<mx<<endl;
        vector<string> res;
        int i = 0;
        while(i < mx) {
            string tmp;
            for(auto ss : vec) {
                if(i < ss.size())
                    tmp += ss[i];
                else
                    tmp += " ";
            }
            i++;
            int j = tmp.size() - 1;
            while(tmp[j] == ' ')
                j--;
            res.push_back(tmp.substr(0, j + 1));
        }
        
        return res;
    }
};
