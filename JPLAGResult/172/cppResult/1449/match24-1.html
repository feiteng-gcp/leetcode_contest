class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> ans;
        s += " ";
        int n = s.length();
        vector<string> now;
        string k = "";
        for (int i = 0; i < n; i++) {

            if (s[i] == ' ') {
                now.push_back(k);
                k = "";
            } else {
                k += s[i];
            }
        }
        int l = 0;

        for (auto i : now) {
            l = max(l, (int)i.length());
        }
        for (int i = 0; i < l; i++) {
            bool check = false;
            string q = "";
            for (int j = 0; j < now.size(); j++) {
                if (now[j].length() > i) q += now[j][i];
                else q += " ";
            }
            int w = now.size() - 1;
            while (q[w] == ' ') w--;
            ans.push_back(q.substr(0, w + 1));
        }
        return ans;
    }
};
