class Solution {
public:
    vector<string> printVertically(string s) {
        
        // get list of words
        vector<string> words;
        stringstream ss(s);
        string w;
        int m=0;
        while(ss>>w){
            words.push_back(w);
            m = max(m, (int)w.size());
        }
        
        // build result
        vector<string> ans;
        int n = words.size();
        for(int i=0; i<m; i++){
            string cur="";
            for(int j=n-1; j>=0; j--){
                if(i<words[j].size()){
                    cur.insert(0, 1, words[j][i]);
                }else if(cur.size()>0){
                    cur.insert(0, 1, ' ');
                }
            }
            ans.push_back(cur);
        }
        
        return ans;
    }
};
