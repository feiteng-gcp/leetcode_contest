class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> words;
        int maxLen = 0;
        
        int start = 0;
        int pos = 0;
        while(pos < s.length()) {
            while(pos < s.length() && s[pos] == ' ') {
                pos++;
            }
            start = pos;
            while(pos < s.length() && s[pos] != ' ') {
                pos++;
            }
            
            words.push_back(s.substr(start, pos - start));
            maxLen = max(maxLen, pos - start);
        }
        
        vector<string> res(maxLen, "");
        
        for(int i = 0; i < maxLen; i++) {
            for(int j = 0; j < words.size(); j++) {
                if(i >= words[j].size()) {
                    res[i] += " ";
                } else {
                    res[i].append(1, words[j][i]);
                }
            }
        }
        
        for(int i = 0; i < maxLen; i++) {
            string str = res[i];
            int end = str.size() - 1;
            while(end >= 0 && str[end] == ' ') {
                end--;
            }
            
            res[i].resize(end + 1);
        }
        
        return res;
    }
};
