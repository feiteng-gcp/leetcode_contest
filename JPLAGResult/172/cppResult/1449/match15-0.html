class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string>ret;
        vector<string>a;
        string t="";
        int mx=0;
        for(int i=0;i<s.size();i++){
            if(s[i]==' '){
                a.push_back(t);
                mx=max(mx,(int)t.size());
                t="";
                continue;
            }
            t+=s[i];
        }
        if(t!=""){
            a.push_back(t);
            mx=max(mx,(int)t.size());
            t="";
        }
        for(int i=0;i<mx;i++){
            t="";
            for(int j=0;j<a.size();j++){
                if(i<a[j].size())t+=a[j][i];
                else t+=" ";
            }
            while(t.size()>0&&t.back()==' ')t.pop_back();
            ret.push_back(t);
        }
        return ret;
    }
};
