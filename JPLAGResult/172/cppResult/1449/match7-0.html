class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> ans;
        int id = 0;
        int cnt = 0;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == ' ') {
                id = 0;
                cnt++;
                continue;
            }
            if (ans.size() < id + 1) ans.push_back(string());
            string& s0 = ans[id];
            while (s0.size() < cnt) s0 += " ";
            s0 += s[i];
            id++;
        }
        return ans;
    }
};
