class Solution {
public:
    vector<string> printVertically(string s)
    {
        vector<string> result;
        int idx = 0;
        int words = 0;
        for (char c : s) {
            if (c == ' ') {
                words++;
                idx = 0;
                continue;
            }
            if (idx == result.size())
                result.push_back(string());
            while (result[idx].size() != words)
                result[idx].push_back(' ');
            result[idx++].push_back(c);
        }
        return result;
    }
};
