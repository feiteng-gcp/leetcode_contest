class Solution {
public:
    vector<string> split(const string& s) {
        int n = s.size();
        vector<string> ret;
        for (int i = 0, j; i < n; i = j) {
            for (; i < n && s[i] == ' '; ++i);
            if (i == n) break;
            for (j = i + 1; j < n && s[j] != ' '; ++j);
            ret.push_back(s.substr(i, j - i));
        }
        return ret;
    }
    vector<string> printVertically(string s) {
        vector<string> w = split(s);
        int n = w.size();
        int m = 0;
        for (auto& x : w) m = max(m, (int)x.size());
        vector<string> ret(m, string(n, ' '));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < w[i].size(); ++j) {
                ret[j][i] = w[i][j];
            }
        }
        for (int i = 0; i < m; ++i) {
            while (ret[i].size() && ret[i].back() == ' ') ret[i].pop_back();
        }
        return ret;
    }
};
