class Solution {
public:
    void split(const string& s,vector<string>& sv,const char flag = ' ') {
        sv.clear();
        istringstream iss(s);
        string temp;

        while (getline(iss, temp, flag)) {
            sv.push_back(temp);
        }
        return;
    }
    
    vector<string> printVertically(string s) {
        vector<string> store;
        split(s, store, ' ');

        int maxLen = 0;
        for(int i = 0; i < store.size(); ++i)
        {
            maxLen = max(maxLen, ((int)store[i].size()) );
        }
        vector<string> res;
        for(int i = 0; i < maxLen; ++i)
        {
            string tmp;
            for(int j = 0; j < store.size(); ++j)
            {
                if(i >= store[j].size())
                {
                    tmp.push_back(' ');
                }
                else
                {
                    tmp.push_back(store[j][i]);
                }
            }
            while(tmp.back() == ' ')
                tmp.pop_back();
            res.push_back(tmp);
        }
        return res;
    }
};
