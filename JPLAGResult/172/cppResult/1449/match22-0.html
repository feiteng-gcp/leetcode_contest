class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> ans,vt;
        stringstream ss(s);
        string w;
        int cnt=0;
        while(ss>>w){
            int i=0;
            for(char c: w){
                if(ans.size()<=i) {
                    ans.push_back("");
                }
                while(ans[i].size()<cnt) ans[i]+=' ';
                ans[i]+=c;
                i++;
            }
            cnt++;
        }
        return ans;
        
    }
};
