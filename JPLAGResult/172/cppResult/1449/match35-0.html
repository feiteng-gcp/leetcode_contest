class Solution {
public:
    vector<string> printVertically(string s) {
        auto split = [&](const string& s, const string& delimeters) {
            vector<string> res;
            int n = s.length();
            
            for (int i = 0; i < n; ++i) {
                auto j = s.find_first_of(delimeters, i);
                if (j == string::npos) {
                    j = n;
                }
                if (j - i > 0) {
                    res.push_back(s.substr(i, j - i));
                }
                i = j;
            }
            return res;
        };
        
        auto words = split(s, " .,!:    ");
        
        int n = 0;
        int m = words.size();
        
        for (auto& s : words) {
            n = max(n, (int)s.length());
        }
        
        vector<string> res(n, string(m, ' '));
        
        for (int i = 0; i < m; ++i) {
            auto& s = words[i];
            for (int j = 0; j < s.length(); ++j) {
                res[j][i] = s[j];
            }
        }
        
        for (auto& s : res) {
            while (!s.empty() && s.back() == ' ') {
                s.pop_back();
            }
        }
        
        return res;
    }
};
