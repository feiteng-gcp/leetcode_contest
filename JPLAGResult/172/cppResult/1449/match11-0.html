class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> words;
        string cur;
        int maxw = 0;
        for (int i = 0; i < s.size(); i++){
            if (s[i]==' '){
                words.push_back(cur);
                maxw = max(maxw, (int)cur.size());
                cur.clear();
            }
            else{
                cur += s[i];
            }
        }
        maxw = max(maxw, (int)cur.size());
        words.push_back(cur);
        vector<string> ans;
        
        for (int i = 0; i < maxw; i++){
            string x;
            for (int j = 0; j < words.size(); j++){
                if (i < words[j].size()){
                    x += words[j][i];
                }
                else{
                    x += ' ';
                }
            }
            while ((!x.empty()) && (x.back()==' ')){
                x.pop_back();
            }
            if (!x.empty()) ans.push_back(x);
        }
        
        return ans;
    }
};
