class Solution {
public:
    vector<string> toVec(string s){
        int prev=0;
        int cur=0;
        vector<string> result;
        while(cur<s.size()){
            if(s[cur]==' '){
                result.push_back(s.substr(prev,cur-prev));
                prev=cur+1;
            }
            cur++;
        }
        result.push_back(s.substr(prev));
        return result;
    }
    
    vector<string> printVertically(string s) {
        vector<string> tmp=toVec(s);
        int maxL=0;
        for(int i=0;i<tmp.size();i++){
            if(tmp[i].size()>maxL){
                maxL=tmp[i].size();
            }
        }
        vector<string> result;
        for(int i=0;i<maxL;i++){
            string tmpStr="";
            for(int j=0;j<tmp.size();j++){
                if(i<tmp[j].size()){
                    tmpStr+=tmp[j][i];
                }
                else{
                    tmpStr+=" ";
                }
            }
            int cur=tmpStr.size()-1;
            while(tmpStr[cur]==' '){
                cur--;
            }
            tmpStr=tmpStr.substr(0,cur+1);
            result.push_back(tmpStr);
        }
        return result;
    }
};
