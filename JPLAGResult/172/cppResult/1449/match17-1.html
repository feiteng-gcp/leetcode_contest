class Solution {
public:
  vector<string> printVertically(string s) {
    vector<string> words;
    istringstream iss(s);
    for (string s; iss >> s;) {
      words.push_back(s);
    }
    int width = words.size();
    int height = 0;
    for (auto& s: words) {
      height = max(height, (int)s.size());

    }
    vector<string> ans(height, string(width, ' '));
    for (int r = 0; r < height; ++r) {
      for (int c = 0; c < width; ++c) {
        if (r < words[c].size()) {
          ans[r][c] = words[c][r];

        }
      }
    }
    for (int r = 0; r < height; ++r) {
      while (ans[r].back() == ' ') {
        ans[r].pop_back();
      }
    }
    return ans;
  }
};
