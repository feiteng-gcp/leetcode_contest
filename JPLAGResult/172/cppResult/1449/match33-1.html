class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> res;

        s = s + " ";
        int last = 0;
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == ' ') {
                res.push_back(s.substr(last, i - last));
                last = i + 1;
            }
        }
        int len = 0;

        for (auto it : res) {
            len = max(len, (int) it.size());
        }
        
        vector<string> ans(len, string(res.size(), ' '));
        for (int i = 0; i < res.size(); ++i) {
            for (int j = 0; j < res[i].size(); ++j) {
                ans[j][i] = res[i][j];
            }
        }
        for (auto& it : ans) {
            while (it.back() == ' ') it.pop_back();
        }
        return ans;
    }
};
