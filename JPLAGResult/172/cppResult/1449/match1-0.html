class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> mid;
        int p=0;
        int last=0;
        while(p<s.size()){
            if(s[p]==' '){
                mid.push_back(s.substr(last,p-last));
                last=p+1;
            }
            p++;
        }
        if(last!=p)
            mid.push_back(s.substr(last,p-last));
        queue<int> q;
        int longest=mid.size()-1;
        q.push(longest);
        for(int i=mid.size()-2;i>=0;i--){
            if(mid[i].size()>mid[longest].size()){
                longest=i;
                q.push(i);
            }
        }
        vector<string> r;
        for(int i=0;i<mid[longest].size();i++){
            string t;
            for(int j=0;j<mid.size();j++){
                if(mid[j].size()<=i)
                    t.push_back(' ');
                else{
                    t.push_back(mid[j][i]);
                    if(j==q.front()){
                        if(i==mid[q.front()].size()-1)
                        q.pop();
                        break;
                    }
                }  
            }
            r.push_back(t);
        }
        return r;
    }
};
