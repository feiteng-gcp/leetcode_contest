class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> ret, t;
        string build = "";
        s += " ";
        for(int i = 0; i < s.length(); i++){
            if(s[i] == ' '){
                t.push_back(build);
                build = "";
            }else{
                build += s[i];
            }
        }
        int n = 0;
        for(int i = 0; i < t.size(); i++)   n = max(n, (int)t[i].length());
        for(int i = 0; i < t.size(); i++){
            while(t[i].length() < n)    t[i] += " ";
        }
        for(int i = 0; i < n; i++){
            string tmp = "";
            for(int j = 0; j < t.size(); j++)   tmp += t[j][i];
            ret.push_back(tmp);
        }
        for(int i = 0; i < ret.size(); i++){
            while(ret[i].back() == ' ') ret[i] = ret[i].substr(0, ret[i].length()-1);
        }
        return ret;
    }
};
