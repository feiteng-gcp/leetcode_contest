class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> s_list;
        s += ' ';
        string tmp;
        int max_count = 0;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] != ' ') {
                tmp += s[i];
            } else {
                s_list.emplace_back(tmp);
                max_count = max(max_count, (int)tmp.size());
                tmp = "";
            }

        }
        vector<string> ans;
        for (int i = 0; i < max_count; i++) {
            tmp = "";
            for (int j = 0; j < s_list.size(); j++) {
                if (i >= s_list[j].size()) tmp+= ' ';
                else tmp += s_list[j][i];
            }
            int count = 0;
            for (int j = tmp.size() - 1; j >= 0; j--) {
                if (tmp[j] != ' ') {
                    count = j;
                    break;
                }
            }
            ans.emplace_back(tmp.substr(0, count + 1));
        }
        return ans;
    }
};
