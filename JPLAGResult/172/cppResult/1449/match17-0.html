class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> list;
        stringstream ss(s);
        int maxlen = 0;
        int maxind = 0;
        string tmp = "";
        int ind = 0;
        while(getline(ss, tmp, ' ')) {
            list.push_back(tmp);
            if ( maxlen <= tmp.size() ) {
                maxlen = tmp.size();
                maxind = ind;
            }
            ind++;
        }
        vector<string> res(maxlen, "");
        for (int i = 0; i < list.size(); i++) {
            for (int j = 0; j < maxlen; j++) {
                if ( j < list[i].size()) {
                    res[j].push_back(list[i][j]);
                } else {
                    res[j].push_back(' ');
                }
            }
        }
        for (int i = 0; i < maxlen; i++) {
            while(!res[i].empty() && res[i].back() == ' ') {
                res[i].pop_back();
            }
        }
        return res;
    }
};
