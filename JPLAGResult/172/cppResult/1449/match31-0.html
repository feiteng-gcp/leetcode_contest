class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> v;
        vector<string> res;
        
        string t="";
        int mxlen = 0;
        for(int i=0; i<s.size(); i++){
            if(s[i] == ' '){
                v.push_back(t);
                mxlen = max(mxlen, (int)t.size());
                t = "";
            }else{
                t += s[i];
            }
        }
        if(t != ""){
            v.push_back(t);
            mxlen = max(mxlen, (int)t.size());
        }
        char arr[210][210];
        for(int i=0; i<210; i++){
            for(int j=0; j<210; j++){
                arr[i][j] = ' ';
            }
        }
        for(int i=0; i<v.size(); i++){
            for(int j=0; j<v[i].size(); j++){
                arr[i][j] = v[i][j];
            }
        }
        for(int j=0; j<mxlen; j++){
            string tmp="";
            bool c = false;
            for(int i=v.size()-1; i>=0; i--){
                if(arr[i][j] == ' '){
                    if(!c){
                        
                    }else{
                        tmp += arr[i][j];
                    }
                }else{
                    c = true;
                    tmp += arr[i][j];
                }
            }
            reverse(tmp.begin(), tmp.end());
            res.push_back(tmp);
        }
        return res;
    }
};
