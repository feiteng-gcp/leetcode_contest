class Solution {
public:
    vector<string> printVertically(string s) {
        istringstream is(s);
        vector<string> words;
        string t;
        int max_len = 0;
        while(is >> t) {
            max_len = max(max_len, (int)t.length());
            words.push_back(t);
        }
        vector<string> res;
        for(int i=0;i<max_len;i++) {
            string w;
            
            for(const auto& word : words) {
                char t = (int)word.length() >= (i+1) ? word[i] : ' ';
                w.push_back(t);
            }
            while(!w.empty()) {
                if(w.back() == ' ')
                    w.pop_back();
                else
                    break;
            }
            res.push_back(w);
        }
        return res;
    }
};
