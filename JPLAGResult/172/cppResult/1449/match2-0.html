class Solution {
public:
	vector<string> printVertically(string s) {
		vector<string> tmp;
		vector<string> res;
		string t;
		int maxlen = 0;
		for (int i = 0; i < s.length(); i++) {
			if (s[i] == ' ' || i + 1 == s.length()) {
				if (i+1 == s.length()) t.push_back(s[i]);
				tmp.push_back(t);
				maxlen = max(maxlen, int(t.size()));
				t.clear();
			}else
				t.push_back(s[i]);
		}
		t.clear();
		for (int i = 0; i < maxlen; i++) {
			for (int j = 0; j < tmp.size(); j++) {
				if (i < tmp[j].length()) t.push_back(tmp[j][i]);
				else t.push_back(' ');
			}
			res.push_back(t);
			t.clear();
		}

		for (int i = 0; i < res.size(); i++) {
			while (res[i].back() == ' ') res[i].pop_back();
		}
		return res;
	}
};
