class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> ans;
        int count=0;
        int numcount=0;
        for(int i=0;i<s.size();i++){
            if(s[i]==' '){
                count=0;
                numcount++;
            }
            else{
                while(ans.size()<count+1){
                    ans.push_back(string());
                }
                while(ans[count].size()<numcount+1){
                    ans[count]+=' ';
                } 
                ans[count][numcount]=s[i];
                count++;
            }
        }
        return ans;
    }
};
