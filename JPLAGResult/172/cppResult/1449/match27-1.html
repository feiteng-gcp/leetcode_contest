class Solution {
public:
    vector<string> printVertically(string s) 
    {
        vector<string> strArr;
        int maxLength = 0;
        while (true)
        {
            int nPos = s.find(' ');
            if (nPos == -1)
            {
                strArr.push_back(s);
                maxLength = max(maxLength, int(s.length()));
                break;
            }
            string sNew = s.substr(0, nPos);
            strArr.push_back(sNew);

            maxLength = max(maxLength, nPos);
            s = s.substr(nPos+1);
        }


        vector<string> ans(maxLength, "");
        for (int i = 0; i < strArr.size(); i++)
        {
            for (int j = 0; j < maxLength; j++)
            {
                if (j < strArr[i].length())
                {
                    ans[j] += strArr[i][j];
                }
                else
                {
                    ans[j] += " ";
                }
            }
        }

        for (int i = 0; i < ans.size(); i++)
        {
            int j = ans[i].length()-1;
            for (; j >=0 ; j--)
            {
                if (ans[i][j] != ' ')
                {
                    break;
                }                
            }
            if (j < 0)
            {
                ans[i] = "";
            }
            else
            {
                ans[i] = ans[i].substr(0, j+1);
            }
        }
        return ans;

    }
};
