class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> words;
        s.push_back(' ');

        int st = 0;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == ' ') {
                words.push_back(s.substr(st, i - st));
                st = i + 1;
            }
        }
        int n = 0;
        for (auto t : words) n = max(n, (int)t.size());
        //cout << n << endl;
        vector<string> ans(n);

        for (int i = 0; i< n; i++) {
            for (int j = 0; j < words.size(); j++) {
                if (i < words[j].size()) ans[i].push_back(words[j][i]);
                else ans[i].push_back(' ');
            }
        }
        //cout << n << endl;
        for (int i = 0; i < n; i++)
            while (ans[i].size() > 0 && ans[i].back() == ' ') ans[i].pop_back();
        return ans;
    }
};
