class Solution {
public:
    vector<string> printVertically(string s) {
        int max_size = 0;
        istringstream iss(s);
        vector<string> words;
        string word;
        while (getline(iss, word, ' ')) {
            words.emplace_back(word);
        }

        int n = words.size();
        for (auto& word : words) {
            max_size = max(max_size, (int)word.size());
        }
        vector<string> result(max_size, string(n, '0'));
        for (int i = 0; i < n; i++)  {
            int j = 0;
            for (; j < words[i].size(); j++) {
                result[j][i] = words[i][j];
            }
            while (j < max_size) {
                result[j][i] = ' ';

                j++;
            }
        }
        for (auto& word : result) {
            while (!word.empty() && word.back() == ' ') {
                word.pop_back();
            }
        }
        return result;
    }
};
