class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> vs;
        int last = 0;
        while (last != string::npos) {
            int next = s.find(' ', last);
            if (next == string::npos) {
                vs.push_back(string(s.begin() + last, s.end()));
                break;
            }
            vs.push_back(string(s.begin() + last, s.begin() + next));
            last = next + 1;
        }
        

        int l = 0;
        for (auto ss : vs)
            l = max(l, (int)ss.size());
        
        vector<string> res(l, "");
        for (int i = 0; i < l; i++) {
            for (int j = 0; j < vs.size(); j++) {
                if (i < vs[j].size())
                    res[i].push_back(vs[j][i]);
                else
                    res[i].push_back(' ');

            }
        }
        
        for (int i = 0; i < res.size(); i++) {
            while (res[i].size() > 0 && *res[i].rbegin() == ' ')
                res[i].pop_back();
        }
        return res;
    }
};
