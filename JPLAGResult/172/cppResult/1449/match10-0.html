class Solution {
public:
    vector<string> printVertically(string s) {
        vector<string> words;
        stringstream ss(s);
        string str;
        int maxLen = 0;
        while (ss >> str) {
            maxLen = max(maxLen, (int)str.length());
            words.emplace_back(str);
        }
        vector<string> result(maxLen, string(size(words), ' '));
        for (int i = 0; i < size(words); ++i) {
            for (int j = 0; j < words[i].length(); ++j) {
                result[j][i] = words[i][j];
            }
        }
        for (string &line : result) {
            while (line.length() > 0 && line.back() == ' ') line.pop_back();
        }
        return result;
    }
};
