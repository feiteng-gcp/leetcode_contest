class Solution {
public:
    vector<string> printVertically(string s) {
        
        vector<string> vec;
        int col = 0;
        for (int i = 0; i < s.size(); ++i)
        {
            if (isalpha(s[i]) != 0)
            {
                string temp = "";
                for (; i < s.size() && s[i] != ' '; ++i) temp += s[i];
                --i;
                vec.push_back(temp);
                col = max(col, int(temp.size()));
            }

        }
        
        vector<string> res;
        for (int c = 0; c < col; ++c)
        {
            string cur = "";
            for (int r = 0; r < vec.size(); ++r)
            {
                if (c < vec[r].size()) cur += vec[r][c];
                else cur += " ";
            }
            
            while (cur.size() > 0 && cur.back() == ' ') cur.pop_back();
            res.push_back(cur);
        }
        
        return res;
    }
};
