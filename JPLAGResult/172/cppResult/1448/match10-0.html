class Solution {
public:
    int maximum69Number (int num) 
    {
        vector<int> nArr;
        while (num != 0)
        {
            nArr.push_back(num%10);
            num /= 10;
        }
        reverse(nArr.begin(), nArr.end());
        for (int i = 0; i < nArr.size(); i++)
        {
            if (nArr[i] == 6)
            {
                nArr[i] = 9;
                break;
            }
        }
        int ans = 0;
        for (int i = 0; i < nArr.size(); i++)
        {
            ans = ans*10+nArr[i];
        }
        return ans;
    }
};
