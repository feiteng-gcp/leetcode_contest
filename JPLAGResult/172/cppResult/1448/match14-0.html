class Solution {
public:
    int maximum69Number (int num) {
        vector<int> vData;
        while(num>0){
            vData.push_back(num%10);
            num/=10;
        }
        reverse(vData.begin(),vData.end());
        
        for(int i=0;i<vData.size();i++){
            if(vData[i]==6){
                vData[i]=9;
                break;
            }
        }
        int ans=0;
        for(int i=0;i<vData.size();i++){
            ans=ans*10+vData[i];
        }
        return ans;
    }
};
