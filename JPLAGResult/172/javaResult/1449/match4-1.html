class Solution {
    public List<String> printVertically(String s) {
        LinkedList<String> ans = new LinkedList();
        String [] ss = s.split("\\s+");
        int max = 0;
        for(String str : ss)        
            max = Math.max(str.length(), max);        
        char [][] c = new char[max][ss.length];
        for(int j = 0; j < max; j ++)
            Arrays.fill(c[j], ' ');
        for(int i = 0 ; i < ss.length; i ++)
        {
            for(int j = 0; j < ss[i].length(); j++)
                c[j][i] = ss[i].charAt(j);
        }
        for(int j = 0; j < max; j ++)
        {
            int len = ss.length - 1;
            for(; len >= 0; len --)
                if(c[j][len] != ' ') break;
            ans.add(new String(c[j]).substring(0, len + 1));
        }
            
        //System.out.println(ans);
        return ans;
    }
}
