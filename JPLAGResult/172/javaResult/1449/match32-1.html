class Solution {
    public List<String> printVertically(String s) {
		String[] a = s.split(" ");
		int n = 0;
		for(String aa:a) {
			n = Math.max(n, aa.length());
		}

		List<String> ans = new ArrayList<>();
		for(int i=0;i<n;++i) {
			StringBuilder sb = new StringBuilder();
			for(String aa:a) {
				if(i<aa.length())
					sb.append(aa.charAt(i));
				else sb.append(" ");
			}
			String t = sb.toString();
			char[] tt = t.toCharArray();
			int j;
			for(j=tt.length;j>0;--j) {
				if(tt[j-1]!=' ')break;
			}
			ans.add(t.substring(0, j));
		}
		return ans;
    }
}
