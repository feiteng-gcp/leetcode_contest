class Solution {
    public List<String> printVertically(String s) {
        String[] strs = s.split("\\s"); 
        List<StringBuilder> build = new ArrayList<>();
        int len = 0; 
        for(int i = 0; i < strs.length; i++) {
            len = Math.max(len, strs[i].length()); 
        }
        
        
        for(int i = 0; i < len; i++) {
            build.add(new StringBuilder()); 
        }
        
        for(int i = 0; i < len; i++) {
            for(int j = 0; j < strs.length; j++) {
                if(i < strs[j].length()) {
                    build.get(i).append(strs[j].charAt(i));
                } else {
                    build.get(i).append(" ");
                }
            }
        }
        
        List<String> res = new ArrayList<>();
        for(int i = 0; i < len; i++) {
            res.add(build.get(i).toString().replaceAll("\\s+$", "")); 
        }
        return res; 
    }
}
