class Solution {
    public List<String> printVertically(String st) {
        int maxLen = 0;
        String[] arr = st.split(" ");
        char sc = ' ';
        for (String ss : arr) maxLen = Math.max(maxLen, ss.length());
        List<String> res = new ArrayList<>();
        for (int i = 0; i < maxLen; i++) {
            StringBuilder sb = new StringBuilder();
            for (String s : arr) {
                if (i < s.length()) sb.append(s.charAt(i));
                else sb.append(sc);
            }
            while (sb.charAt(sb.length() - 1) == sc) sb.setLength(sb.length() - 1);
            res.add(new String(sb));
        }
        return res;
    }
}
