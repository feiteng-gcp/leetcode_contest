class Solution {
    public List<String> printVertically(String s) {
        List<StringBuilder> ans = new ArrayList<>();
        
String[] ss = s.split(" ");
        int len = 0;
        for (int i = 0; i < ss.length; i++) {
            len = Math.max(len, ss[i].length());
        }
        
for (int i = 0; i < len; i++) {
            
            ans.add(new StringBuilder());
        }
        
        for (int j = 0; j < len; j++) {
            boolean finish = true;
            for (int i = 0; i < ss.length; i++) {
                String curr = ss[i];
                if (j >= curr.length()) {
                    ans.get(j).append(" ");
                } else {
                    
ans.get(j).append(curr.charAt(j));
                }
            }
        }
        
        List<String> ans1 = new ArrayList<>();
        for (int i = 0; i < ans.size(); i++) {
            ans1.add(
ans.get(i).toString().stripTrailing());
        }
        return ans1;
    }
}
