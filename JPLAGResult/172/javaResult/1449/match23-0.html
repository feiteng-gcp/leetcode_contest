class Solution {
    public List<String> printVertically(String input) {
        String[] arr = input.split(" ");
        int max = 0;
        for (String s : arr) max = Math.max(max, s.length());
        
        List<String> ans = new ArrayList<>();
        for (int i = 0; i < max; i++) {
            StringBuilder b = new StringBuilder();
            for (String s : arr) {
                if (i >= s.length()) {
                    b.append(' ');
                } else {
                    b.append(s.charAt(i));
                }
            }
            
            while (b.charAt(b.length() - 1) == ' ') {
                b.delete(b.length() - 1, b.length());
            }
            
            ans.add(b.toString());
        }
        
        return ans;
    }
}
