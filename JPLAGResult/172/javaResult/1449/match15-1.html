class Solution {
    public List<String> printVertically(String s) {
        String[] ss = s.split(" ");
        
int m=0,n=ss.length;
        for(String str:ss){
            m=Math.max(m,str.length());
        }
        List<String> ans = new ArrayList<>();
        
for(int i=0;i<m;i++){
            StringBuilder sb=new StringBuilder();
            for(String str:ss){
                if(str.length()<=i){
                    sb.append(" ");
                }else{
                    sb.append(str.charAt(i));
                }
            }
            int j=sb.length()-1;
            for(;j>=0;j--){
                if(sb.charAt(j)!=' '){
                    break;
                }
            
}
            ans.add(sb.toString().substring(0,j+1));
        }
        return ans;
    }
}
