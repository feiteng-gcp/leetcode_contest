class Solution {
    public List<String> printVertically(String s) {
        String[] sL = s.split(" ");
        List<String> res = new ArrayList<>();
        int max = 0;
        for(String word: sL){
            max = Math.max(word.length(), max);
        }
        for(int i = 0;i<max;i++){
            StringBuilder vw = new StringBuilder();
            for(String word:sL){
                if(word.length() <= i){
                    vw.append(" ");
                }
                else{
                    vw.append(word.charAt(i));
                }
            }
            int j = vw.length()-1;
            for(; j >=0;j--){
                if(vw.charAt(j)==' '){
                    continue;
                }
                break;
            }
            res.add(vw.substring(0,j+1).toString());
        }
        return res;
        
    }
}
