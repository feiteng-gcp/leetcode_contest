class Solution {
    public List<String> printVertically(String s) {
        String[] arr = s.split(" ");
        int len = 0;
        for (String a : arr) {
            len = Math.max(a.length(), len);
        }
        List<String> res = new ArrayList<>();
        for (int i = 0; i < len; i++) {
            StringBuilder sb = new StringBuilder();
            for (String a : arr) {
                sb.append(i >= a.length() ? " " : a.charAt(i));
            }
            res.add(trimTail(sb.toString()));
        }
        
        return res;
    }
    
    private String trimTail(String s) {
        for (int i = s.length()-1; i >= 0; i--) {
            if (s.charAt(i) != ' ') {
                s = s.substring(0, i+1);
                break;
            }
        }
        return s;
    }
}
