import java.util.List;

class Solution {
    public List<String> printVertically(String s) {
        String[] words = s.split(" ");
        List<String> r = new ArrayList<>();
        for(int i = 0 ; i < words.length; i++){
            String w = words[i] ;
            for(int j = 0; j < w.length(); j++){
                if (r.size() <= j){
                    r.add("");
                }
                String v = r.get(j);
                while (v.length() < i){
                    v += " ";
                }
                v += w.charAt(j);
                r.set(j, v);
            }
        }
        return r;
    }
}
