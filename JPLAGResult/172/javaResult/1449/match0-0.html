class Solution {
    public List<String> printVertically(String s) {
        String[] sp = s.split(" ");
        int n = 0;
        for(String str : sp) n = Math.max(str.length(), n);
        int m = sp.length;
        
        List<String> list = new ArrayList<>();
        
        for(int j = 0; j < n; j++)
        {
            String tmp = "";
            for(int i = 0; i < m; i++)
            {
                if(j >= sp[i].length()) tmp += " ";
                else tmp += sp[i].charAt(j);
            }
            // now remove trailing spaces
            int k = tmp.length() - 1;
            while(k >= 0 && tmp.charAt(k) == ' ') k--;
            if(k >= 0) list.add(tmp.substring(0, k + 1));
        }
        return list;
    }
}
