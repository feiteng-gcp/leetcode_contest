class Solution {
    public List<String> printVertically(String s) {
        List<String> res = new ArrayList<>();
        
String[] strs = s.split(" ");
        int max = 0;
        for(String ss : strs) max = Math.max(max, ss.length());
        
        
for(int i = 0; i < max; i++) {
            StringBuilder sb = new StringBuilder();
            for(int j = 0; j < strs.length; j++) {
                if(i < strs[j].length()) sb.append(strs[j].charAt(i));
                else sb.append(" ");
            }
            while(sb.charAt(sb.length() - 1) == ' ') sb.deleteCharAt(sb.length() - 1);
            res.add(sb.toString());
        }
        return res;
    }
}
