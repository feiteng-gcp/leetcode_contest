class Solution {
    public int maximum69Number (int num) {
        StringBuilder sb = new StringBuilder();
        sb.append(num);
        
        for (int i = 0; i < sb.length(); i++) {
            int n = sb.charAt(i) - '0';
            if (n == 6) {
                sb.setCharAt(i, '9');
                break;
            }
        }
        
        return Integer.valueOf(sb.toString());
        
    }
}
