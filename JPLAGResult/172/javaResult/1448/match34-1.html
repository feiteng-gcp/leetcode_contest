class Solution {
    public int maximum69Number (int num) {
        String n = Integer.toString(num);
        char[] arr = n.toCharArray();
        for (int i = 0; i < arr.length; i++) {
            
if (arr[i] == '6') {
                arr[i] = '9';
                break;
            }
        }
        // System.out.println(Arrays.toString(arr));
        return Integer.valueOf(new String(arr));
    }
}
