class Solution {
    public int maximum69Number (int num) {
        char[] S = Integer.toString(num).toCharArray();
        for (int i = 0; i < S.length; ++i) {
            if (S[i] == '6') {
                S[i] = '9';
                break;
            }
        }
        return Integer.parseInt(new String(S));
    }
}
