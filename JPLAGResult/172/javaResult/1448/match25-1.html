class Solution {
    public int maximum69Number (int num) {
        String v = String.valueOf(num);
        for(int i=0;i<v.length();i++){
            if(v.charAt(i) == '6') {
                v = v.substring(0,i) + "9" + v.substring(i+1);
                break;
            }
        }
        return Integer.valueOf(v);
    }
}
