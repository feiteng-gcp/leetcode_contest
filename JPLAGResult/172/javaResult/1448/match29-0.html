class Solution {
    public int maximum69Number (int num) {
        char[] chars = Integer.toString(num).toCharArray();
        int res = 0;
        boolean hasConvert = false;
        for(int i = 0 ; i < chars.length; i++){
            res *= 10;
            if(chars[i] == '6' && !hasConvert){
                hasConvert = true;
                res += '9' - '0';
            } else {
                res += chars[i] - '0';
            }
        }
        return res;
    }
}
