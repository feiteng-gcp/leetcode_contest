class Solution {
    public int maximum69Number (int num) {
        String value = String.valueOf(num);
        char[] chars = value.toCharArray();
        for(int i = 0;i<chars.length;i++) {
            if(chars[i] == '6'){
                chars[i] = '9';
                break;
            }
        }
        return Integer.parseInt(new String(chars));
    }
}
