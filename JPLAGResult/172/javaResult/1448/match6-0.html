class Solution {
      public int maximum69Number (int num) {
        String in = String.valueOf(num);
        
        StringBuffer sb = new StringBuffer();
        boolean flag = true;
        for(int i = 0; i < in.length(); i ++) {
            if (in.charAt(i) == '6' && flag) {
                sb.append('9');
                flag = false;
            } else {
                sb.append(in.charAt(i));
            }
        }
        return Integer.valueOf(sb.toString());
    }
}
