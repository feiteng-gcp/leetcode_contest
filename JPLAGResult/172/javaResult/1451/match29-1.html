class Solution {

	public int minTaps(int n, int[] ranges) {
		ArrayList<int[]> list = new ArrayList<>();
		for (int i = 0; i < ranges.length; i++) {
			if (ranges[i] != 0) {

				list.add(new int[] { i - ranges[i], i + ranges[i] });
			}
		}
		Collections.sort(list, (a, b) -> a[0] == b[0] ? a[1] - b[1] : a[0] - b[0]);
		int i = 0, count = 0, start = 0;
		while (start < n) {
			int[] best = null;
			while (i < list.size() && list.get(i)[0] <= start) {
				if (best == null || list.get(i)[1] >= best[1]) {
					best = list.get(i);
				}
				i++;
			}
			if (best == null) {
				return -1;
			}
			count++;
			start = best[1];
		}
		return count;
	}
}
