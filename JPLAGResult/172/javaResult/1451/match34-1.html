class Solution {
    public int minTaps(int n, int[] ranges) {
        int tap = 0;
        int index = 0;
        
        while (index < n) {
            
int right = -1;
            int best = -1;
            for (int i = 0; i < ranges.length; i++) {
                if (i - ranges[i] <= index  && index <= i + ranges[i] && i + ranges[i] > right) {
                    best = i;
                    right = i + ranges[i];
                }
            }
            if (right == index) {
                return -1;
            }
            tap++;
            index = right;
        }
            
        
        
        
        return tap;
    }
}
