class Solution {
    public int minTaps(int n, int[] ranges) {
        int[] cache = new int[n+1];
        Arrays.fill(cache, 1_000_000);
        
        for (int g = 0; g <= n; g++) {
            int r = ranges[g];
            int left = Math.max(0, g-r);
            // System.out.println(g + " " + left + " " + r);
            for (int i = left; i <= g+r && i <= n; i++) {
                if (left != 0) cache[i] = Math.min(cache[i], cache[left]+1);
                else cache[i] = 1;
            
}
        }
        
        // System.out.println(Arrays.toString(cache));
        
        return cache[n] != 1_000_000 ? cache[n] : -1;
    }
}
