class Solution {
    class Node {
        int left;
        int right;

        public Node(int l, int r) {
            this.left = l;
            this.right = r;
        }
    }

    public int minTaps(int n, int[] ranges) {
        int last = 0;
        Node[] nodes = new Node[ranges.length];
        for (int i = 0; i < ranges.length; i++) {
            nodes[i] = new Node(i - ranges[i], i + ranges[i]);
        }
        Arrays.sort(nodes, Comparator.comparingInt(a -> a.left));
        int res = 0;
        int i = 0;
        while (last < ranges.length - 1) {
            int rightMax = 0;
            while (i < nodes.length && nodes[i].left <= last) {
                rightMax = Math.max(rightMax, nodes[i].right);
                i++;
            }
            if (rightMax != ranges.length - 1 && last == rightMax) {
                return -1;
            }
            res++;
            last = rightMax;
        }
        return res;

    }
}
