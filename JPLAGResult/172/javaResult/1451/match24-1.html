class Solution {
    class Tap{
        int left;
        int right;
        boolean valid;
        public Tap(int l, int r, boolean v){
            left = l;
            right = r;
            valid = v;
        }
        public String toString(){
            return left + "<->" + right + ", " + valid;
        }
    }
    
    
public int minTaps(int n, int[] ranges) {
        List<Tap> taps = new ArrayList<>();
        for(int i = 0; i <=n; i++){
            int l = Math.max(0, i-ranges[i]);
            int r = Math.min(n, i+ranges[i]);
            if(l == r){
                taps.add(new Tap(l,r, false));
            }
            else{
                taps.add(new Tap(l,r, true));
            }
        
}
        Collections.sort(taps, (a,b) -> {
            if(a.left != b.left){
                return a.left - b.left;
            }
            else{
                return b.right - a.right;
            }
        });
        
        if(taps.get(0).left > 0){
            return -1;
        }
        // for(int i = 0; i < taps.size(); i++){
        //     System.out.println("index:" + i + " " + taps.get(i).toString());
        // }
        int rightCover = taps.get(0).right;
        
int count = 1;
        int index = 1;
        while(rightCover < n){
            int maxRight = rightCover;
            int maxIndex = index;
            for(int i = index; i < taps.size(); i++){
                Tap tap = taps.get(i);
                // if(!tap.valid){
                //     continue;
                // }
                if(tap.left > rightCover){
                    break;
                }
                if(tap.right > maxRight){
                    maxRight = tap.right;
                    maxIndex = i;
                }
            }
            if(maxRight > rightCover){
                count++;
                index = maxIndex + 1;
                rightCover = maxRight;
                // System.out.println(maxIndex);
            }
            else{
                return -1;
            }
        }
        return count;
    }
}
