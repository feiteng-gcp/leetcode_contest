class Solution {
    public int minTaps(int n, int[] ranges) {
        List<int[]> list = new ArrayList<>();
        for(int i = 0; i <= n; i++)
        {
            int from = i - ranges[i];
            int to = i + ranges[i];

            System.out.println(ranges[i]);
            if(ranges[i] > 0) list.add(new int[]{from, to});
            // else r[i] = new int[]{-1, -1};
        }
        // System.out.println(list.size());
        Collections.sort(list, (a, b)->(a[0] != b[0] ? a[0] - b[0] : b[1] - a[1]));
        // how to count?
        int right = 0, re = 0;
        // if(list.get(0)[0] > 0) return -1;  // cannot cover left most point
        // for(int i = 0; i < n; i++)
        Stack<Integer> st = new Stack<>();

        for(int[] r : list)
        {
            if(right >= n) break;

            int from = r[0], to = r[1];
            // System.out.printf("%d %d %d\n", right, from, to);

            if(from <= 0 && right <= to) {
                re = 1;
                // st.push(right);
                st.clear();
                st.push(0);
                right = to;
            }
            
            else if(from <= right && right < to) 
            {
                while(!st.isEmpty() && from <= st.peek()) right = st.pop();
                // re++;
                // right = st.isEmpty() ? 0 : st.peek();

                st.push(right);
                right = to;
                
            }
        }
        
        // System.out.println(right);
        if(right < n) return -1;
        // return re;
        // System.out.println(st);
        return st.size();
    }
}
