class Solution {
    public int minTaps(int n, int[] ranges) {
        int out = 0;
        
        int tap = 0;
        
        while(tap < n) {
            int next = getNext(tap, ranges);
            
            if(next == -1) {
                return -1;
            }
            
            out++;
            
            tap = next + ranges[next];
        }
        
        return out;
    }
    
    private int getNext(int start, int[] ranges) {
        int out = -1;
        
        for(int i = start; i < ranges.length; i++) {
            if(ranges[i] > 0 && ranges[i] >= i - start) {
                if(out == -1 || (i + ranges[i] > out + ranges[out])) {
                    out = i;
                }
            }
        }
        
        return out;
    }
}
