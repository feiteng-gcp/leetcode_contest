class Solution {
    public int minTaps(int n, int[] ranges) {
        List<Range> rangeList = new ArrayList<>();
        for(int i = 0; i<ranges.length; i++)
        {
            int start = Math.max(0, i-ranges[i]);
            int end = Math.min(n, i+ranges[i]);
            rangeList.add(new Range(start, end));
        }
        Collections.sort(rangeList, new Comparator<Range>(){
            public int compare(Range a, Range b) {
                if (a.start != b.start) return a.start - b.start;
                else return b.end - a.end;
            }
        });
        int start = rangeList.get(0).start;
        int end = rangeList.get(0).end;
        int i = 1, count = 1;
        if(start > 0) {
            return -1;
        }
        else if(end >= n) {
            return 1;
        }
        else {
            while(i < rangeList.size() && end < n && rangeList.get(i).start<=end)
            {
                int max = end;
                while(i < rangeList.size() && rangeList.get(i).start <= end) {
                    max = Math.max(max, rangeList.get(i).end);
                    i++;
                }
                end = max;
                count++;
            }
        }
        return end >= n ? count : -1;
    }
    
    private static class Range
    {
        int start;
        int end;
        
        Range(int start, int end)
        {
            this.start = start;
            this.end = end;
        }
    }
}
