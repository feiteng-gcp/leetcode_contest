class Solution {
    public int minTaps(int n, int[] ranges) {
        int[][] array = new int[n + 1][2];
        
        for (int i = 0; i <= n; i++) {
            array[i] = new int[]{i - ranges[i], i + ranges[i]};
        
}
        
        Arrays.sort(array, new Comparator<int[]>(){
            @Override
            public int compare(int[] a, int[] b) {
                return a[0] - b[0];
            }
        });
        
        int res = 0;
        for (int i = 0, st = 0, end = 0; st < n; st = end, res++) {
            for (; i <= n && array[i][0] <= st; i++) {
                end = Math.max(end, array[i][1]);
            }
            
            if (st == end) return -1;
        }
        
        return res;
    }
}
