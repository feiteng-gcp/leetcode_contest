class Solution {
    public int minTaps(int n, int[] ranges) {
        int[][] nums = new int[n + 1][2];
        for(int i = 0 ; i <= n ; i++){
            nums[i][0] = i - ranges[i];
            nums[i][1] = i + ranges[i];
        }
        Arrays.sort(nums, new Comparator<int[]>(){
            @Override
            public int compare(int[] a, int[] b){
                return a[0] - b[0];
            }
        });
        int time = 0, res = 0, id = 0;
        while(id <= n){
            int maxRight = time;
            for(; id <= n && nums[id][0] <= time; id++){
                maxRight = Math.max(nums[id][1], maxRight);
            }
            if(maxRight == time){
                return -1;
            }
            time = maxRight;
            res++;
            if(time >= n){
                return res;
            }
        }
        return -1;
    }
}
