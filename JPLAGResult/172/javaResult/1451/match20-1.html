class Solution {
    
public int minTaps(int n, int[] ranges) {
        int modu = 20001;
        int[] a = new int[n+1];
        for (int i=0; i<=n; i++)
            a[i] = Math.max(0, i-ranges[i]) * modu + i + ranges[i];
        
Arrays.sort(a);
        int ans = 0;
        int l = 0;
        int i = 0;
        while (l < n) {
            int j = l;
            while (i <= n && a[i] / modu <= l) {
                j = Math.max(j, a[i] % modu);
                i++;
            }
            if (j == l)
                return -1;
            l = j;
            ans++;
        }
        return ans;
    }
}
