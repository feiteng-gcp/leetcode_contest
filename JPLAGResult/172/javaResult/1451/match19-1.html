class Solution {
    public int minTaps(int n, int[] ranges) {
        int end = 0;
        int sp = 0;
        int cnt = 0;
        while(end < n) {
            int ct = helper(sp, ranges, end);
          //  System.out.println(ct);
          //  System.out.println(ranges[ct]);
            if(ct == -1) return -1;
            ++cnt;
            end = ct + ranges[ct];
          //  System.out.println(end);
            
sp = ct + 1;
        }
        return cnt;
    }
    int helper(int sp, int[] A, int lb) {
        int rm = -1;
        int id = -1;
        for(int i = sp; i < A.length; ++i) {
            if(A[i] == 0) continue;
            int lx = i - A[i];
            int rx = i + A[i];
            if(lx <= lb) {
                if(rx > rm) {
                    rm = rx;
                    
id = i;
                }
            }
        }
        return id;
    }
}
