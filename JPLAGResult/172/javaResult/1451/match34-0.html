class Solution {
    public int minTaps(int n, int[] ranges) {
        int[] data = new int[n];
        for (int idx = 0; idx < ranges.length; idx++) {
            if (ranges[idx] > 0) {
                int left = idx - ranges[idx];
                int right = idx + ranges[idx];
                if (left < 0) {
                    left = 0;
                }
                if (right > n) {
                    right = n;
                }
                if (data[left] < right) {
                    data[left] = right;
                }
            }
        }
        int count = 1;
        int startIdx = 0;
        do {
            int nextRange = data[startIdx];
            if (nextRange == n) {
                return count;
            }
            int maxRange = nextRange;
            int maxId = -1;
            for (int tmpIdx = startIdx + 1; tmpIdx <= nextRange; tmpIdx++) {
                if (data[tmpIdx] > maxRange) {
                    maxRange = data[tmpIdx];
                    maxId = tmpIdx;
                }
            }
            if (maxRange == nextRange) {
                return -1;
            } else {
                startIdx = maxId;
                count++;
            }
        } while (true);
    }
}
