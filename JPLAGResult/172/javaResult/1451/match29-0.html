class Solution {
    public int minTaps(int n, int[] ranges) {
        List<int[]> list = new ArrayList<>();
        
        for (int i = 0; i < ranges.length; i++) {
            int range = ranges[i];
            int start = i - range;
            int end = i + range;
            
            if (end >= 0) {
                list.add(new int[]{start, end});
            }
        }
        
        Collections.sort(list, (i1, i2) -> (i1[0] - i2[0]));
        
        int curMax = 0;
        int i = 0;
        int res = 0;
        
        while (curMax >= list.get(i)[0]) {
            int localMax = 0;
            for (; i < list.size() && curMax >= list.get(i)[0]; i++) {
                localMax = Math.max(localMax, list.get(i)[1]);
            }    
            res++;
            curMax = localMax;
            if (curMax >= n) {
                return res;
            }
        }
         
        return -1;
    }
}
