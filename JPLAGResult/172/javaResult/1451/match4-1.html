class Solution {
    public int minTaps(int n, int[] ranges) {
        int[] jumps = new int[n+1];
        for (int i = 0; i < ranges.length; i++) {
            int left = Math.max(0, i - ranges[i]);
            int right = Math.min(n, i + ranges[i]);
            jumps[left] = Math.max(jumps[left], right);
        }
        int curStepEnd = jumps[0], reachSoFar = curStepEnd, res = 0;
        for (int i = 0; i < jumps.length; i++) {
            reachSoFar = Math.max(reachSoFar, jumps[i]);
            if (i == curStepEnd) {
                res++;
                curStepEnd = reachSoFar;
            }
        }
        return curStepEnd < n ? -1 : res;
    }
}
