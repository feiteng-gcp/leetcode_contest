class Solution {
    public int minTaps(int n, int[] ranges) {
        int[] jump = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            int left = Math.max(i - ranges[i], 0);
            int right = Math.min(i + ranges[i], n);
            jump[left] = Math.max(jump[left], right);
        }
        int right = jump[0];
        int next = right, res = 0;
        for (int i = 0; i <= n; i++) {
            next = Math.max(next, jump[i]);
            if (i == right) {
                res++;
                right = next;
            }
        }
        return right != n ? -1 : res;
    }
}
