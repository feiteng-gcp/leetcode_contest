class Solution {
    public int minTaps(int n, int[] ranges) {
		int[][] a = new int[n+1][2];
		for(int i=0;i<n+1;++i) {
			a[i][0] = i-ranges[i];
			a[i][1] = i+ranges[i];
		}
		Arrays.sort(a, (x,y)->x[0]==y[0]?x[1]-y[1]:x[0]-y[0]);
		int ans = 0,need = 0,i=0,end = 0;;
		while(i<n+1) {
			if(a[i][0]<=need) {
				end = a[i][1];
				i++;
				while(i<n+1&&a[i][0]<=need) {
					if(end<a[i][1])end = a[i][1];
					i++;
				}
				ans++;
				if(end>=n)break;
				need = end;
			}else break;;
		}
		return end<n?-1:ans;
    }
}
