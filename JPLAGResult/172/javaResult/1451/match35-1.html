class Solution {
    public int minTaps(int n, int[] ranges) {
        List<List<Integer>> rangeCalculations = new ArrayList<>();
        
        for (int i = 0; i < ranges.length; i++) {
            List<Integer> specRange = new ArrayList<>();
            specRange.add(i - ranges[i]);
            specRange.add(i + ranges[i]);
            rangeCalculations.add(specRange);
        }
        List<Integer> specInterval = new ArrayList<>();
        specInterval.add(0);
        specInterval.add(n);
        int result = leastMerge(rangeCalculations, specInterval);
        
        if (result == 0) {
            return -1;
        } else {
            return result;
        }
    }
    
    
public static class Interval {
        public int start;
        public int end;
        
        public Interval(int start, int end) {
            this.start = start;
            this.end = end;
        }
    }

public int leastMerge(List<List<Integer>> intervals, List<Integer> interval) {
    if (intervals == null || intervals.size() == 0) {
        return 0;
    } 
    
    int targetStart = interval.get(0), targetEnd = interval.get(1);
    int count = 0;

    intervals.sort((c1, c2) -> {
        if (c1.get(0) == c2.get(0)) {
            return c1.get(1) - c2.get(1);
        }  else {
            return c1.get(0) - c2.get(0);
        }
    });

    int i = 0, start = targetStart, end = targetStart;

    while (start < targetEnd) {
        while (i < intervals.size() && intervals.get(i).get(0) <= start) {
            end = Math.max(end, 
intervals.get(i).get(1));
            i++;
        }

        if (start == end) {
            return 0;
        }

        start = end;
        count++;
    }

    return count;
}
}
