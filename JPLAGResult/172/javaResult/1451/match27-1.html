class Solution {
    class Interval {
        int start;
        int end;
        public Interval(int s, int e) {
            this.start = s;
            this.end = e;
        }
    }
    
    
public int minTaps(int n, int[] ranges) {
        Interval[] intervals = new Interval[n + 1];
        for (int i = 0; i < ranges.length; i++) {
            intervals[i] = new Interval(Math.max(0, i - ranges[i]), Math.min(n, i + ranges[i]));
        
}
        Arrays.sort(intervals, (a, b) -> a.start == b.start ? b.end - a.end : a.start - b.start);
        int pivot = intervals[0].end;
        int next = 0, count = 1, index = 1;
        
        while (index < intervals.length) {
            if (pivot >= n) return count;
            while (index < intervals.length && intervals[index].start <= pivot) {
                next = Math.max(next, intervals[index].end);
                index++;
            }
            if (next <= pivot) return -1;
            count++;
            pivot = next;
            next = 0;
        }
        return pivot >= n ? count : -1;
    }
}
