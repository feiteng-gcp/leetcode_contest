class Solution {
    public int minTaps(int n, int[] ranges) {
         //Map<Integer, Integer> map = HashMap<>();
         int[] arr = new int[n+1];
         for (int i = 0; i <= n; i++) {
             int r = ranges[i];
             int a = Math.max(0, i-r);
             int b = Math.min(n, i+r);
             if (b > arr[a]) arr[a] = b;
         }
        System.out.println(Arrays.toString(arr));
        int res = jump(arr);
        return res;
    
    }
    private int jump(int[] nums) {
        
        if (nums[0] == 0) return -1;
        
        int n = nums.length;
       
        int start = 0;
        
int maxDist = nums[start];
        
        int cnt = 0;
        while (maxDist < n-1) {
            int tmp = maxDist;
            int pos = -1;
            for (int j = start+1; j <= maxDist; j++) {
                if (nums[j] > tmp) {
                    tmp = nums[j];
                    pos = j;
                }
            }
            if (maxDist == tmp) return -1;
            maxDist = tmp;
            start = pos;
            cnt++;
        }
        return cnt+1;
    }
}
