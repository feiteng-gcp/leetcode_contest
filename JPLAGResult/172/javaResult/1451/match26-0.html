class Solution {
    public int minTaps(int n, int[] ranges) {
        List<int[]> list = new ArrayList<>();
        for(int i = 0;i<ranges.length;i++){
            int[] l = new int[2];
            l[0] = Math.max(0, i - ranges[i]);
            l[1] = Math.min(n, i + ranges[i]);
            list.add(l);
        }
        Collections.sort(list, new Comparator<int[]>(){
            public int compare(int[] o1, int[] o2){
                if(o1[0] == o2[0]) return o2[1] - o1[1];
                return o1[0] - o2[0];
            }
        });
        
        int ret = 0;
        int reach = 0;
        int i = 0;
        for(;i<list.size();){
            int[] cur = list.get(i);
            if(cur[0] > reach) break;
            int tmpreach = reach;
            while(i<list.size() && list.get(i)[0] <= reach){
                tmpreach = Math.max(list.get(i)[1], tmpreach);
                i++;
            }
            
            if(reach == tmpreach) break;
            ret++;
            reach = tmpreach;
        }
        if(reach != n) return -1;
        return ret;
    }
}
