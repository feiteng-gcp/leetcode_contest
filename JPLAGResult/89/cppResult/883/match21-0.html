struct car
{
	int p;
	int v;
};
class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int n=position.size();
        if(n==0)return 0;
	car news;
	vector<car> v;
	for(int i=0;i<n;i++)
	{
		news.p=position[i];
		news.v=speed[i];
		v.push_back(news);
	}
	sort(v.begin(),v.end(),cmp);
	int ans=1,pre=0;
	double pret=1.0*(target-v[0].p)/v[0].v;
	for(int i=1;i<n;i++)
	{
		double nowt=1.0*(target-v[i].p)/v[i].v;
		if(nowt>pret)
		{
			pret=nowt;
			pre=i;
			ans++;
		}
	}
	return ans;
    }
private:
    static bool cmp(car a,car b)
{
	return a.p>b.p;
}
};
