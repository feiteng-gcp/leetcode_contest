class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        if (position.size() ==0) return 0;
        if (position.size() == 1) return 1;
        vector<pair<int, int> > poss;
        poss.reserve(position.size());
        for (int i = 0; i < position.size(); ++i) {
            poss.push_back({position[i], speed[i]});
        }

        sort(poss.begin(), poss.end(), [](const pair<int, int>& lhs, const pair<int, int> &rhs){
            return lhs.first > rhs.first;
        });
        vector<double> time(poss.size(), 0);
        for (int i = 0; i < poss.size(); ++i) {
            time[i] = double(target - poss[i].first)/double(poss[i].second);

        }
        int ret = 1;
        for (int i = 1; i < poss.size(); ++i) {
            if (time[i] > time[i-1]) {
                ++ret;
            } else {
                time[i] = time[i-1];
            }
        }
        return ret;
    }
};
