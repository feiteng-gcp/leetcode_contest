class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        vector<pair<int,int>> comb;
        for(int i=0; i<position.size(); i++)
            comb.emplace_back(position[i], speed[i]);
        sort(comb.begin(), comb.end(), [](pair<int,int>& l, pair<int,int>& r){
           return l.first>r.first; 
        });
        vector<double> time;
        for(int i=0; i<position.size(); i++)
        {
            double t = target - comb[i].first;
            t/=(1.0*comb[i].second);
            time.push_back(t);
        }
        double cur_t = -1.0;
        int count =0;
        for(int i=0; i<position.size(); i++)
        {
            if(time[i]>cur_t)
            {
                count++;
                cur_t = time[i];
            }
        }
        return count;
    }
};
