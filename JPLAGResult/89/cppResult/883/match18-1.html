class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int n=position.size();if(n==0) return 0;
        pair<int,int> a[n];
        for(int i=0;i<n;i++) a[i]={position[i],speed[i]};
        sort(a,a+n);
        int dis=0,spee=1,ans=0;
        for(int i=n-1;i>=0;i--){
            int d=target-a[i].first;
            if((long long)dis*a[i].second>=(long long)d*spee){
                
            }else{
                dis=d;spee=a[i].second;ans++;
            }
        }
        return ans;
    }
};
