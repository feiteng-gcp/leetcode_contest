class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        vector<pair<int, int>> psp;
        for (int i = 0; i < position.size(); ++i) {
            psp.push_back(make_pair(position[i], speed[i]));
        }
        sort(psp.begin(), psp.end());
        double lastTime = -1;
        int ans = 0;
        for (int i = psp.size() - 1; i >= 0; --i) {
            if ((target - psp[i].first + 0.0) / psp[i].second <= lastTime) {
                continue;
            } else {
                lastTime = (target - psp[i].first + 0.0) / psp[i].second;
                ans++;
            }
        }
        return ans;
    }
};
