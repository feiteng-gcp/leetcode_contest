class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int size = position.size();
        int i;
        map<int, double> lut;
        for (i = 0; i < size; i++) {
            lut[position[i]] = ((double)target - position[i]) / speed[i];
        }
        
        map<int, double>::reverse_iterator it;
        
        int count = 0;
        double last_t;
        for (it = lut.rbegin(); it != lut.rend(); it++) {
            if (it == lut.rbegin() || it->second > last_t) {
                count++;
                last_t = it->second;
            } 
        }
        
        return count;
    }
};
