struct N
{
    int d,s;
    double t;
}a[10005];

bool cmp(N x,N y)
{
    if(x.d == y.d)
        return x.s < y.s;

    return x.d < y.d;
}
class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int n = speed.size(); //cout<<n<<endl;
        if(n==0) return 0;
        for(int i=0;i<n;i++)
        {
            a[i].d = target - position[i];
            a[i].s = speed[i];

            if(a[i].s ==0)
                a[i].t = 0;
            else
                a[i].t = a[i].d *1.0 / a[i].s ;

            //cout<<a[i].d <<" "<<a[i].s<<" "<<a[i].t<<endl;
        }
        sort(a,a+n,cmp);
        N pre;
        pre = a[0];
        int res =1;
        for(int i=1;i<n;i++)
        {
            //cout<<a[i].t<<endl;
            if(a[i].t > pre.t)
            {
                res++;
                pre = a[i];
            }else
            {
                pre.t = max(a[i].t,pre.t);
            }
        }
        return res;
    }
};
