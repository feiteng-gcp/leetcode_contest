typedef long long ll;

class Solution {

public:
    int carFleet(int target, vector<int>& pos, vector<int>& spd) {
        int n=pos.size();
        if(n==0)return 0;
        vector<int> ix(n),pos2(n),spd2(n);
        for(int i=0;i<n;i++)ix[i]=i;
        sort(ix.begin(),ix.end(),[&](int i,int j){
            return pos[i]>pos[j];
        });
        for(int i=0;i<n;i++){
            pos2[i]=pos[ix[i]];
            spd2[i]=spd[ix[i]];
        }
        int ans=1,head=0;
        for(int i=1;i<n;i++){
            if(ll(target-pos2[i])*ll(spd2[head])>ll(target-pos2[head])*ll(spd2[i])){
                ans++;
                head=i;
            }
        }
        return ans;
    }
};
