class Solution {
    struct Comp {
        vector<int>& pos;
        Comp(vector<int>& in) : pos(in) {}
        bool operator() (int a, int b) {
            return pos[a] < pos[b];
        }
    };
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int n = position.size();
        if (n==0)
            return 0;
        vector<int> idx(n);
        for (int i=0; i<n; ++i)
            idx[i] = i;
        sort(idx.begin(), idx.end(), Comp(position));
        int res = 0;

        double arrive = -1;
        for (int i=n-1; i>=0; --i) {            
            double dist = target-position[idx[i]]+0.0;
            double t = dist/speed[idx[i]];
            if (arrive<t) {
                res++;
                arrive = t;
            }
        }
        return res;
    }
};
