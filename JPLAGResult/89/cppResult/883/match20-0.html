class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int n = position.size();
        vector<pair<int, int>> p(n);
        for (int i = 0; i < n; ++i) p[i] = make_pair(position[i], speed[i]);
        sort(p.begin(), p.end());
        vector<pair<long long, long long>> v;
        for (const auto x : p) {
            long long u = target - x.first, w = x.second;
            while (!v.empty() && u * v.back().second >= w * v.back().first) {
                v.pop_back();
            }
            v.push_back(make_pair(u, w));
        }
        return v.size();
    }
};
