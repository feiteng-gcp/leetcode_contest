class Solution {
public:
    struct Car {
        int pos;
        int speed;
    };
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        vector<Car> cars;
        int N = position.size();
        if (N <= 1) return N;
        for (int i = 0;i < N; ++i) {
            cars.push_back(Car{position[i], speed[i]});
        }
        
        sort(cars.begin(), cars.end(), [](Car &c, Car &c2) {
            return c.pos > c2.pos;
        });
        

        Car cur = cars[0];
        int count = 1;
        for (int i = 1; i < N; ++i) {
            if (cur.speed * uint64_t(target - cars[i].pos) > cars[i].speed * uint64_t(target - cur.pos)) {
                cur = cars[i];
                count++;
            }   
        }
        
        return count;
    }
};
