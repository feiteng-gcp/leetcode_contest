class Solution {

public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
       
        int n = position.size();
        if (n == 0) return 0;
        
        vector< vector<int> > vec;
        for (int i = 0; i < position.size(); ++i)
        {
            vec.push_back( {position[i], speed[i]} );
        }
        sort(vec.begin(), vec.end());
        
        int res = 0;
        double preTime = -1;
        for (int i = n - 1; i >= 0; --i)
        {
            int delta = target - vec[i][0];
            double time  = 0;
            time = delta / double(vec[i][1]);            
            if (time > preTime + 0.0000001) ++res;            
            preTime = max(preTime, time);
        }
        
        return res;
    }
};
