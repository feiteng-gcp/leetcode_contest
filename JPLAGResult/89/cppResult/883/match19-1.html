class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        vector<pair<int,int> > ps;
        vector <double> time(position.size());
        int i;
        double lt=-1.0;
        int fleets=0;
        for(i=0; i<position.size(); i++)
            ps.push_back(make_pair(position[i],speed[i]));
        sort(ps.begin(),ps.end());
        for(i=0; i<position.size(); i++)

            time[i]=(double)(target-ps[i].first)/ps[i].second;
        for(i=time.size()-1; i>=0; i--){
            if(time[i]>lt){
                fleets++;
                lt=time[i];
            }
        }
        return fleets;
    }
};
