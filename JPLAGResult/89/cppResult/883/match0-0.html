class Solution {
public:

    struct car{
        double pos,v;
        car(){}
        car(double pos,double v):pos(pos),v(v){}
        bool operator<(car &c){return pos<c.pos;}    
    };
    
    int count_set=0;
    
    
    void set_set(vector<int> &s){
        count_set=s.size();
        for (int i=0;i<s.size();i++)
            s[i]=i;
    }
    
    void union_set(vector<int> &s,int root1,int root2){
        s[root2]=root1;
        count_set--;
    }
    
    int find_set(vector<int> &s,int x){
        if (s[x]==x)
            return x;
        else {
            s[x]=find_set(s,s[x]);
            return s[x];
        }
    }
    
    
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        vector<car> c;
        for (int i=0;i<speed.size();i++)
            c.push_back(car(target-position[i],speed[i]));
        sort(c.begin(),c.end());
        vector<int> s(c.size(),0);
        set_set(s);
        for (int i=1;i<c.size();i++){
            int pc=find_set(s,i-1);
            if (c[i].v<c[pc].v)
                continue;
            if (c[i].v==c[pc].v){
                 if (c[i].pos==c[pc].pos)
                     union_set(s,pc,i);
                continue;
            }
            if ((c[i].pos-c[pc].pos)/(c[i].v-c[pc].v)<=(c[pc].pos/c[pc].v))
                union_set(s,pc,i);
        }
        return count_set;
    }
};
