class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        
        int n = position.size();
        
        if(n == 0)
            return 0;
        
        vector<int> idx_arr(n);
        for(int i = 0; i < n; ++i)
            idx_arr[i] = i;
        
        auto idx_sort = [&position](int i1, int i2){return position[i1] > position[i2];};
        std::sort(idx_arr.begin(), idx_arr.end(), idx_sort);
        
        int last_pos = position[idx_arr[0]];
        int last_vel = speed[idx_arr[0]];
        
        int counter = 1;
        
        for(int i = 1; i < n; ++i)
        {
            
            long long t1 = (target - last_pos);
            t1 = t1 * speed[idx_arr[i]];
            
            long long t2 = (target - position[idx_arr[i]]);
            t2 = t2 * last_vel;
            
            if(t1 < t2)
            {
                last_pos = position[idx_arr[i]];
                last_vel = speed[idx_arr[i]];
                ++counter;
            }
            
            
        }
        
        
        return counter;
    }
};
