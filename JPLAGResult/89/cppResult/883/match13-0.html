class Solution {
public:
    int carFleet(long long target, vector<int>& position, vector<int>& speed) {
        int cnt = 0;
        int n = position.size();
        vector<pair<long long, long long>> inp(n);
        for(int i=0;i<n;i++) inp[i] = {position[i], speed[i]};
        sort(inp.begin(), inp.end(), greater<pair<int, int>>());
        int end_i = -1;
        for(int i=0;i<n;i++){
            if(end_i == -1 || ((target - inp[i].first) * inp[end_i].second > (target - inp[end_i].first) * inp[i].second)){
                cnt++;
                end_i = i;
            }
        }
        
        return cnt;
    }
};
