class Solution {
public:
    struct car {
        int p,s;
    }c[10005];
    struct cmp {
        bool operator() (car x,car y) {
            return x.p<y.p;
        }
    }cmp1;
    bool check(car x,car y,int e) {
        if (x.s<=y.s) return false;
        double t1=(y.p-x.p)*1.0/(x.s-y.s);
        double t2=(e-x.p)*1.0/x.s;
        double t3=(e-y.p)*1.0/y.s;
        if (t1<=t2 && t1<=t3) return true;
        return false;
    }
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int n=position.size();
        for (int i=0; i<n; i++) {
            c[i].p=position[i];
            c[i].s=speed[i];
        }
        sort(c,c+n,cmp1);
        int ans=n;

        for (int i=0; i<n-1; i++) {
            for (int j=i+1; j<n; j++) {
                if (check(c[i],c[j],target)) {
                    --ans;
                    break;
                }
            }
        }
        return ans;
    }
};
