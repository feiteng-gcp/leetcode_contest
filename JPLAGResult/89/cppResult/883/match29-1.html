struct Car
{
  int start;
  int speed;
};

// Sort by starting positions
bool compare(Car a, Car b)
{
  return a.start < b.start;
}

class Solution
{
  private:

  double getReachingTime(Car car, int target)
  {
    int distance = target - car.start;
    double time = (double)distance / (double)car.speed;
    return time;
  }

  public:

  int carFleet(int target, vector<int> &position, vector<int> &speed)
  {
    int n = position.size();

    Car car;
    vector<Car> cars;


    for(int i = 0; i < n; ++i)
    {
      car.start = position[i];
      car.speed = speed[i];
      cars.push_back(car);
    }

    sort(cars.begin(), cars.end(), compare);

    int fleet = 0;
    double previousReachingTime = 0;
    for(int i = n - 1; i >= 0; --i)
    {
      car = cars[i];
      double reachingTime = getReachingTime(car, target);

      // New fleet
      if(reachingTime > previousReachingTime)
      {
        previousReachingTime = reachingTime;
        ++fleet;
      }
    }

    return fleet;
  }
};
