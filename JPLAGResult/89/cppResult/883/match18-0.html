class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int n = position.size();
        
        if (n == 0) return 0;
        
        map<int, int> myMap;
        
        for (int i = 0; i < n; ++i) {
            myMap.insert(std::make_pair(position[i], speed[i]));
        }
        
        int res = 0;
        float spentLimit = -1.0f;
        
        for (auto iter = myMap.rbegin(); iter != myMap.rend(); ++iter) {
            float spent = 1.0f * (target - iter->first) / iter->second;
            
            if (spentLimit < 0 || spent > spentLimit) {
                ++res;
                spentLimit = spent;
            }
        }
        
        return res;
    }
};
