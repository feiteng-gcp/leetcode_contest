class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int size = position.size();
        if(size==0)
            return 0;
        vector<pair<int,double>>  timings(size);
        
        for (int i = 0; i < size;i++){
            timings[i].first = target - position[i];
            timings[i].second = (1.0 * (target - position[i]))/speed[i];
        }
        sort(timings.begin(),timings.end());
        int count = 0;
        double max = -1;
        for(int i = 0; i<size;i++){
            cout<<"("<<timings[i].first<<","<<timings[i].second <<")";
        }
        for(int i = 0; i<size;i++){
            if(timings[i].second>max){
                count++;
                max = timings[i].second;
            }
        }
        cout<<endl;
        
        return count;
    }
};
