class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        if(position.empty())
            return 0;
        auto comp = [&position](int& a, int& b){return position[a] > position[b];};
        vector<int> v;
        for(int i = 0; i < position.size(); i++)
            v.push_back(i);
        sort(v.begin(), v.end(), comp);
        int count = 1;
        int p = position[v[0]];
        int sp = speed[v[0]];
        float time = (float)(target - p)/sp;
        for(int i = 1; i < v.size(); i++){
            float cur_time = (float)(target - position[v[i]]) / speed[v[i]];
            if(cur_time > time){
                count++;
                time = cur_time;
            }
        }
        return count;
    }
};
