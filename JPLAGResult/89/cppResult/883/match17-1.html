class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        set<pair<int,int>> pos_speed;
        for (int i = 0; i < position.size(); ++i) {
            pos_speed.insert(make_pair(position[i],speed[i]));
        }
        float time = 0;
        int fleet = 0;

        for (set<pair<int,int>>::reverse_iterator ite = pos_speed.rbegin(); ite != pos_speed.rend(); ++ite) {
            float temp_time = (float)(target - (*ite).first) / (*ite).second;
            if (temp_time > time) {
                time = temp_time;
                ++fleet;
            };
        }
        return fleet;
    }
};
