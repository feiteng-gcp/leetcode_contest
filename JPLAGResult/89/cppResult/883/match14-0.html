class Solution {
public:
    int carFleet(int target, vector<int>& p, vector<int>& s) {
        if (!p.size())
            return 0;
        vector<long> pp(p.size());
        vector<int> r(p.size());
        for (int i = 0; i < p.size(); ++i)
            r[i] = i;
        sort(r.begin(), r.end(), [&](int a, int b) {
            return p[a] < p[b];
        });
        vector<double> t(p.size());
        for (int i = p.size() - 1; i >= 0; --i) {
            t[i] = double(target - p[r[i]]) / s[r[i]];
        }
        int tot = 1;
        for (int i = p.size() - 2; i >= 0; --i) {
            if (t[i] <= t[i + 1]) {
                t[i] = t[i + 1];
            } else
                ++tot;
        }
        return tot;
    }
};
