class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int n = position.size();
        if (n == 0) return 0;
        
        int ret = n;
        
        vector<pair<int, int>> ps;
        for(int i = 0; i < n; i++) {
            pair<int, int> ii = make_pair(position[i], speed[i]);
            ps.push_back(ii);
        }
        
        sort(ps.begin(), ps.end());
        
        vector<double> time(n, 0);
        for(int i = 0; i < n; i++) {
            time[i] = (double)(target - ps[i].first) / ps[i].second;
        }
        
        for(int i = 0; i < n; i++) {
            for(int j = i+1; j < n; j++) {
                if (time[i] <= time[j]) {
                    ret--;
                    break;
                }
            }
        }
        
        return ret;
    }
};
