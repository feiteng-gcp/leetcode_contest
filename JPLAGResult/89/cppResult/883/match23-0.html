struct node {
    int s;
    int p;
    node(int pos, int speed) : p(pos), s(speed) {}
};

bool cmp(node a, node b) {
    return a.p < b.p;
}

class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        vector<node> foo;
        int n = position.size();
        if (n == 0) return 0;
        for (int i = 0; i < n; ++i) {
            foo.push_back({position[i], speed[i]});
        }
        sort(foo.begin(), foo.end(), cmp);

        int res = 1, prev = n - 1;
        for (int i = n - 2; i >= 0; --i) {
            double t1 = 1.0 * (target - foo[prev].p) / foo[prev].s;
            double t2 = 1.0 * (target - foo[i].p) / foo[i].s;
            if (t1 < t2) {
                res++;
                prev = i;
            }
        }

        return res;
    }
};
