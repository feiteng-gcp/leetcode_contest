class Solution {
public:
    int carFleet(int target, vector<int>& position, vector<int>& speed) {
        int n = speed.size();
        if (0 == n)
        {
            return 0;
        }
        vector<tuple<int, int>>input;
        for (int i = 0; i < n; ++i)
        {
            input.push_back(make_tuple(target - position[i], speed[i]));
        }
        sort(input.begin(), input.end(), [](const tuple<int, int>& a, const tuple<int, int>& b) -> bool
        {
            return get<0>(a) < get<0>(b);
        });
        
        int total = 0;

        int lastFleet = 0;
        for (int i = 1; i < n; ++i)
        {
            if ((int64_t)get<0>(input[i]) * (int64_t)get<1>(input[lastFleet]) > (int64_t)get<0>(input[lastFleet]) * (int64_t)get<1>(input[i]))
            {
                total++;
                lastFleet = i;
            }
        }
        return total + 1;
    }

};
