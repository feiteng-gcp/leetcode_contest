class Solution {
public:
    int kSimilarity(string A, string B) {
     search(0, A, B, 0);
        return best;
    }
    
    int search(int k, string a, string b, int index) {
        if (a == b) {
            if (k < best) {
                best = k;
            }
            return 0;
        }
        
        int m = INT_MAX - 400;
        
        if (k > best) return 0;
        

        if (a[index] == b[index]) {
            return search(k, a, b, index+1);
        } 
            
        for (int j = index + 1; j < a.size(); ++j) {
            if (a[j] == b[index]) {
                swap(a[index], a[j]);
                search(k + 1, a, b, index + 1);
                swap(a[index], a[j]);
            }
        }
         
        return m;
    }
    
    int best = INT_MAX;
};
