class Solution {
public:
    
    unordered_map<string, int> hash;
    string target;
    
    int kSimilarity(string A, string B) {
        
        if (A == B) return 0;
        
        int len = A.size();
        deque< pair<string, int> > Q;
        Q.push_back( {A, 0} );
        
        while (Q.size() > 0)
        {
            pair<string, int> cur = Q.front();
            Q.pop_front();
            
            int diffNum = Diff(cur.first, B);    
            string& str = cur.first;
            
            int start = 0;
            for (int i = 0; i < len; ++i)
                if (str[i] != B[i])
                {
                    start = i; 
                    break;
                }
                              
            for (int t = start + 1; t < len; ++t)
            {
                    if (str[t] == B[start])
                    {
                        string newStr = str;
                        swap(newStr[start], newStr[t]);                        
                        if (hash.count(newStr) == 0)
                        {
                            int num = Diff(newStr, B);    
                            if (num == 0) return cur.second + 1;
                            
                            if (num < diffNum)
                            {
                                Q.push_back( {newStr, cur.second + 1} );
                                hash[newStr] = cur.second + 1;
                            }
                        }
                    }
            }
        }
    }
    
    int Diff(string& a, string& b)
    {

        int count = 0;
        for (int i = 0; i < a.size(); ++i)
        {
            if (a[i] != b[i]) ++count;
        }
        return count;
    }
};
