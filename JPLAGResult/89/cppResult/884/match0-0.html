class Solution {
public:
    int kSimilarity(string A, string B) {
        if (A.size() != B.size() || A.empty()) {
            return 0;
        }
        
        int res = INT_MAX;   

        kSimilarityHelper(A, B , 0, 0, res);
        
        return res;
    }
    
    void kSimilarityHelper(string& A, string& B, int start, int count, int& res) {
        if (start == A.size()) {
            res = min(res, count);
            return;
        }
        
        if (A[start] == B[start]) {
            kSimilarityHelper(A, B, start+1, count, res);
            return;
        }
        
        for (int i = start+1; i < A.size(); ++i) {
            if (A[i] == B[start]) {
                swap(A[start], A[i]);
                kSimilarityHelper(A, B, start+1, count+1, res);
                swap(A[start], A[i]);
            }
        }
    }
};
