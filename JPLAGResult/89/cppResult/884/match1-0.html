class Solution {
    vector<pair<int, int>> ops;
    int n, cnt[128], d;
    string a_, b_;
    vector<int> mem;
    void go(int i, int st, int w) {
        if (i < n) {
            if (1 == ++cnt[a_[i]]) ++d;
            if (0 == --cnt[b_[i]]) --d;
            if (0 == d) {
                if (st) {
                    ops.push_back(make_pair(st | (1 << i), w + 1));
                }
            } else {
                go(i + 1, st | (1 << i), w + 1);
            }
            if (1 == ++cnt[b_[i]]) ++d;
            if (0 == --cnt[a_[i]]) --d;
            go(i + 1, st, w);
        }
    }
    int og(int st) {
        int& ret = mem[st];
        if (ret < 0) {
            ret = st ? 1000010000 : 0;
            for (const auto& o : ops) {
                if (o.first == (o.first & st)) {
                    ret = min(ret, o.second - 1 + og(st ^ o.first));
                }
            }
        }
        return ret;
    }
public:
    int kSimilarity(string a, string b) {
        n = a.size();
        a_ = a;
        b_ = b;
        memset(cnt, 0, sizeof(cnt));
        d = 0;
        go(0, 0, 0);
        sort(ops.begin(), ops.end());
        // for (const auto& p : ops) cout << p.first << " " << p.second << endl;
        mem = vector<int>(1 << n, -1);
        int bt = (1 << n) - 1;
        for (int i = 0; i < n; ++i) {
            if (a[i] == b[i]) bt ^= 1 << i;
        }
        return og(bt);
    }
};
