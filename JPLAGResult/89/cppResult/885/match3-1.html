class ExamRoom {
public:

    ExamRoom(int N)
    {
        n = N;
    }
    
    int seat() {
        if (sit.size() == 0) {
            sit.insert(0);
            return 0;
        }
        
        int cur = *sit.begin();
        int mm = cur;
        int tt = 0;
        
        for (auto &i : sit) {
            int k = (i - cur) / 2;
            
            if (k > mm) {
                mm = k;
                tt = cur + k;
            }
            cur = i;
        }
        
        if (n - 1 != cur) {
            int k = n - 1 - cur;
            
            if (k > mm) {
                mm = k;
                tt = n - 1;
            }
            
            cur = n - 1;
        }
        
        sit.insert(tt);
        
        return tt;
    }
    
    void leave(int p) {
        sit.erase(p);
    }
    
    int n;
    set<int> sit;
};

 * ExamRoom obj = new ExamRoom(N);
 * int param_1 = obj.seat();
 * obj.leave(p);
