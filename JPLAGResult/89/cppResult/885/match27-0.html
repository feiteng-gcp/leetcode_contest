class ExamRoom {
    vector<int> vt;
    vector<int>::iterator it;
    int n;
public:
    ExamRoom(int N) {
        vt.clear();
        n=N;
    }

    int seat() {
        if(vt.size()==0)
        {
            vt.push_back(0);
            return 0;
        }
        else if(vt.size()==1)
        {
            int tmp = vt[0];
            if( tmp - 0 >= n-tmp-1)
            {
                vt.push_back(0);
                sort(vt.begin(),vt.end());
                return 0;
            }
            else
            {
                vt.push_back(n-1);
                return n-1;
            }
        }
        else
        {
            int m= vt.size();
            int d =vt[0]-1,pre =vt[0],t=0;
            int tmp_d;
            for(int i=1;i<m;i++)
            {
                int tmp = (vt[i]-pre);
                if(tmp %2==1)
                {
                    tmp_d = tmp/2 -1;
                    if(d==-1) { d= tmp_d; t =tmp/2+pre; }
                    else if( d<tmp_d) { d= tmp_d; t =tmp/2+pre; }
                }
                else
                {
                    tmp_d = tmp/2 -1;
                    if(d==-1) { d= tmp_d; t =tmp/2+pre; }
                    else if( d<tmp_d) { d= tmp_d; t =tmp/2+pre; }
                }
                pre = vt[i];
            }
            if(vt[m-1]!=n-1)
            {
                if( d < n-1 - vt[m-1]-1)
                {
                    t = n-1;
                }
            }
            vt.push_back(t);
            sort(vt.begin(),vt.end());
            return t;
        }
    }

    void leave(int p) {
        for(it=vt.begin();it!=vt.end();it++)
        {
            if(*it==p)
            {
                vt.erase(it);
                return ;
            }

        }
    }
};
