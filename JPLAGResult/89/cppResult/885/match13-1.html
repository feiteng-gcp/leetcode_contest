class ExamRoom {
    set<int> taken;
    int n;
public:
    ExamRoom(int N) {
        n = N;
    }   
    
    int seat() {
        int ans;
        if (taken.empty()) {
            ans = 0;
        } else {
            int last = 0, mx = *taken.begin(), pick = 0;
            for (auto t: taken) {
                int p = (t+last) / 2;
                int d = p - last;
                if (d > mx) {
                    pick = p;
                    mx = d;
                }   
                last = t;
            }   
            if (last < n-1) {
                int d = n-1 - last;
                if (d > mx) {
                    mx = d;
                    pick = n-1;
                }   
            }   
            ans = pick;
        }   
        taken.insert(ans);
        return ans;
    }   
    
    void leave(int p) {
        taken.erase(p);
    }   
};
