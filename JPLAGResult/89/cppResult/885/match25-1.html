struct Intv{
    int a,b;
    Intv(){}
    Intv(int a,int b):a(a),b(b){}
    bool operator<(const Intv &other) const{
        int d=(b-a)/2,d2=(other.b-other.a)/2;
        return d>d2||(d==d2&&a<other.a);
    }
};


class ExamRoom {
    set<Intv> s;
    unordered_map<int,Intv> sa,sb;
    int n;
public:
    ExamRoom(int n):n(n) {
        
    }
    
    int seat() {
        if(s.empty()){
            Intv i(1,n-1);
            s.insert(i);
            sa[1]=i;
            sb[n-1]=i;
            return 0;
        }
        else{
            int a=s.begin()->a,b=s.begin()->b,m=(a+b)/2,dist=m-a;
            if(sa.count(0)&&sa[0].b>=dist){
                a=0;
                b=sa[0].b;
                m=0;
                dist=b-a;
            }
            if(sb.count(n-1)&&n-1-sb[n-1].a>dist){
                a=sb[n-1].a;
                b=n-1;
                m=n-1;
                dist=b-a;
            }
            s.erase(Intv(a,b));
            sa.erase(a);
            sb.erase(b);
            if(a<m){
                Intv i(a,m-1);
                s.insert(i);
                sa[a]=i;
                sb[m-1]=i;
            }
            if(b>m){
                Intv i(m+1,b);
                s.insert(i);
                sa[m+1]=i;
                sb[b]=i;
            }
            return m;
        }
    }
    
    void leave(int p) {
        int a,b;
        if(sb.count(p-1)){
            a=sb[p-1].a;
            s.erase(sb[p-1]);
            sa.erase(a);
            sb.erase(p-1);
        }
        else a=p;
        if(sa.count(p+1)){
            b=sa[p+1].b;
            s.erase(sa[p+1]);
            sa.erase(p+1);
            sb.erase(b);
        }
        else b=p;
        Intv i(a,b);
        s.insert(i);
        sa[a]=i;
        sb[b]=i;
    }
};
