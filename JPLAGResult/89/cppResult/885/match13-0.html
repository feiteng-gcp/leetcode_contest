#include<set>

using namespace std;

class ExamRoom {
    int n;
    set<int> pp;
public:
    ExamRoom(int N) {
        n = N;
    }
    
    int seat() {
        int prv = -1;
        int d = 0, pos = -1;
        if(pp.size() == 0) {
            pp.insert(0);
            return 0;
        }
        for(auto e : pp) {
            if(prv == -1) {
                if(e != 0) {
                    d = e;
                    pos = 0;   
                }
            } else {
                if(d < (e - prv) / 2) {
                    d = (e - prv)/2;
                    pos = (e + prv) / 2;
                }
            }
            prv = e;
        }
        if(prv != n-1) {
            if(d < n-1-prv) {
                d = n-1-prv;
                pos = n-1;
            }
        }
        pp.insert(pos);
        return pos;
    }
    
    void leave(int p) {
        pp.erase(p);
    }
};

 * Your ExamRoom object will be instantiated and called as such:
 * ExamRoom obj = new ExamRoom(N);
 * int param_1 = obj.seat();
 * obj.leave(p);
