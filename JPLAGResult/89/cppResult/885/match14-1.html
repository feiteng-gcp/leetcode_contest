class ExamRoom {
public:
    set<int> seats;
    int num_seat = 0;
    ExamRoom(int N) {
        num_seat = N;
    }
    
    int seat() {
        return get_seat();
    }
    
    void leave(int p) {
        seats.erase(p);
    }
    
private:
    int get_seat()
    {
        //cout << "Let calculate next seat" << endl;
        if(seats.empty())
        {
            seats.insert(0);
            return 0;
        }
        else
        {
            int prev_seat = -1;
            int result_seat = -1;
            int result_dist = -1;
            set <int> :: iterator itr;
            for (itr = seats.begin(); itr != seats.end(); ++itr)
            {
                //cout << "new iter: " << *itr << endl;
                //cout << "prev_seat" << prev_seat<< endl;
                if(prev_seat < 0)
                {
                    prev_seat = *itr;
                    if(prev_seat == 0)
                    {
                        continue;
                    }
                    else
                    {
                        result_seat = 0;
                        result_dist = *itr;
                    }
                }
                else

                {
                    //cout << "next_seat" << *itr<< endl;
                    //cout << "prev_seat" << prev_seat<< endl;
                    int dist = *itr - prev_seat;
                    int new_dist = dist / 2;
                    if(new_dist > result_dist)
                    {
                        result_dist = new_dist;
                        result_seat = prev_seat + new_dist;
                    }
                    prev_seat = *itr;
                }
            }
            int new_dist = num_seat - 1 - prev_seat;
            if(new_dist > result_dist)
            {
                result_dist = new_dist;
                result_seat = num_seat - 1;
            }
            //cout << "new_seat" << result_seat << endl;
            seats.insert(result_seat);
            return result_seat;
        }
    }
};
