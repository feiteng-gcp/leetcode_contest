class ExamRoom {
    int N;
    set<int> pos;
public:
    ExamRoom(int N): N(N) {
    }
    
    int seat() {
        if (!pos.size()) {
            pos.insert(0);
            return 0;
        }
        int prev = -1, dist = -1, seat;
        for (auto it: pos) {
            int x = 0, at;
            if (prev < 0) {
                if (it > 0) {
                    x = it;
                    at = 0;
                    if (dist < x)
                        dist = x, seat = at;
                }
            } else if (prev + 1 < it) {
                x = (it - prev) / 2;
                at = prev + x;
                if (dist < x)
                    dist = x, seat = at;
            }
            prev = it;
        }
        if (prev + 1 < N) {
            int x = N - prev - 1;
            int at = N - 1;
            // N = 4, x x x x
            if (dist < x)
                dist = x, seat = at;
        }
        assert(dist > 0);
        pos.insert(seat);
        return seat;
    }
    
    void leave(int p) {
        pos.erase(p);
    }
};

 * ExamRoom obj = new ExamRoom(N);
 * int param_1 = obj.seat();
 * obj.leave(p);
