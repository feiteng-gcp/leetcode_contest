class ExamRoom {
    set<int> pos;
    int n;
public:
    ExamRoom(int N) {
        n = N;
    }

    int seat() {
        if (pos.empty()) {
            pos.insert(0);
            return 0;
        } else {
            int d, p;
            if (*pos.begin() != 0) {
                d = *pos.begin();
                p = 0;
            } else {
                d = 0;
            }
            for (auto it = next(pos.begin()); it != pos.end(); ++it) {
                int s = *prev(it), e = *it;
                int mid = s + (e - s) / 2;
                int dist = mid - s;
                if (dist > d) {
                    d = dist;
                    p = mid;
                }
            }
            if (n - 1 - *pos.rbegin() > d) {
                d = n - 1 - *pos.rbegin();
                p = n - 1;
            }
            pos.insert(p);
            return p;
        }
    }

    void leave(int p) {
        pos.erase(p);
    }
};
