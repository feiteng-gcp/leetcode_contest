class ExamRoom {
public:
    int num;
    int now;
    vector<int> store;
    ExamRoom(int N) {
        num=N;
        now=0;
    }
    
    int seat() {
        if(now==0)
        {
            now++;
            store.push_back(0);
            return 0;
        }
        /*if(now==1)
        {
            if(store[0]>=num-1-store[0])
            {
                store.push_back(0);
                store[1]=store[0];
                store[0]=0;
                now++;
                return 0;
            }
            else
            {
                store.push_back(num-1);
                now++;
                return num-1;
            }
        }*/
        int ans=num-1,dis=ans-store[now-1];
        for(int i=now-2;i>=0;i--)
        {
            if((store[i+1]-store[i])/2>=dis)
            {
                ans=store[i]+(store[i+1]-store[i])/2;
                dis=(store[i+1]-store[i])/2;
            }
        }
        if(store[0]>=dis)
        {
            ans=0;
        }
        store.push_back(ans);
        now++;
        for(int i=now-2;i>=0;i--)
        {
            if(store[i]<store[i+1])break;
            int tmp=store[i];
            store[i]=store[i+1];
            store[i+1]=tmp;
        }
        return ans;
    }
    
    void leave(int p) {
        for(int i=0;i<now;i++)
        {
            if(store[i]==p)
            {
                store.erase(store.begin()+i);
                break;
            }
        }
        now--;
    }
};

 * int param_1 = obj.seat();
 * obj.leave(p);
