#define FORi(i,a,b) for(int i=a; i<b; i++)
#define ITER set<int>::iterator

class ExamRoom {
public:
    int l, r;
    set<int> used;
    ExamRoom(int n) {
        l = 0;
        r = n-1;
    }
    
    int seat() {
        if (used.size() == 0){
            used.insert(l);
            return l;
        }
        ITER it = used.begin();
        int d = *it;
        int id = 0;
        while(++it != used.end()){
            ITER it0 = it;
            --it0;
            int cd = (*it - *it0) >> 1;
            if (cd > d){
                d = cd;
                id = (*it + *it0) >> 1;
            }
        }
        --it;
        if (*it != r){
            int cd = r - *it;

            if (cd > d){
                d = cd;
                id = r;
            }
        }
        used.insert(id);
        return id;
    }
    
    void leave(int p) {
        used.erase(p);
    }
};
