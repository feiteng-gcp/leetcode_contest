class ExamRoom {
public:
    ExamRoom(int N) {
        n = N;
    }
    
    int seat() {
        if(s.empty()) {
            s.insert(0);
            return 0;
        }
        if(s.size() == 1) {
            int x = *s.begin();
            int y = n - 1 - x > x ? n - 1 : 0;
            s.insert(y);
            return y;
        }
        
        int y = 0, d = *s.begin() == 0 ? INT_MIN : *s.begin();
        for(auto it = s.begin(); it != s.end(); ++it) {
            auto ne = next(it);
            if(ne == s.end()) continue;;
            if((*ne - *it) / 2 > d) {
                d = (*ne - *it) / 2;
                y = (*ne + *it) / 2;
            }
        }
        if(*s.rbegin() != n - 1 && n - 1 - *s.rbegin() > d) y = n - 1;
        s.insert(y);
        return y;
    }
    
    void leave(int p) {
        s.erase(p);
    }    
    int n = 0;
    set<int> s;
};
