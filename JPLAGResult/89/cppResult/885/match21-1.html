class ExamRoom {
public:
    ExamRoom(int N) {
        max_ = N;
    }
    
    bool inrange(int i) {
        return 0 <= i && i <max_;
    }
    
    int seat() {
        if(ss.empty()) {
            ss.insert(0);
            return 0;
        }
        else {
            auto cit = ss.begin();
            auto nit = ss.begin();
            ++nit;
            int best_pos = -1;
            int dist = 0;
            
            int cur_dist = (*cit - 0);
            if(cur_dist > dist) {
                best_pos = 0;
                dist = cur_dist;
            }
            
            while(nit != ss.end()) {
                cur_dist = (*nit - *cit)/2;
                if(cur_dist > dist) {
                    best_pos = *cit + cur_dist;
                    dist = cur_dist;
                }
                ++cit;
                ++nit;
            }
            
            cur_dist = (max_ - *cit - 1);

            if(cur_dist > dist) {
                best_pos = *cit + cur_dist;
                dist = cur_dist;
            }
            ss.insert(best_pos);
            return best_pos;
        }
    }
    
    void leave(int p) {
        ss.erase(p);
    }
private:
    set<int> ss;
    int max_;
};
