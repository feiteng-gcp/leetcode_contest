class ExamRoom {
public:
    int a[100005];
    int n=0;
    int p;
    ExamRoom(int N) {
        p=N-1;
    }
    
    void insert(int x) {
        int i=1;
        while (i<=n && a[i]<=x) ++i;
        for (int j=n; j>=i; j--) a[j+1]=a[j];
        a[i]=x;
        ++n;
    }
    
    int seat() {
        int ans=0;
        if (n==0) ans=0;
        else {
            int mx=0;
            for (int i=1; i<n; i++) {

                int d=(a[i+1]-a[i])/2;
                if (mx<d || (mx==d && ans>a[i]+d)) {
                    ans=a[i]+d;
                    mx=d;
                }
            }
            if (mx<a[1] || (mx==a[1] && ans>0)) {
                ans=0;
                mx=a[1];
            }
            if (mx<p-a[n] || (mx==p-a[n] && ans>p)) {
                ans=p;
                mx=p-a[n];
            }
        }
        insert(ans);
        return ans;
    }
    
    void leave(int p) {
        for (int i=1; i<=n; i++) {
            if (a[i]==p) {
                for (int j=i; j<n; j++) {
                    a[j]=a[j+1];
                }
                --n;
                return;
            }
        }
    }
};

 * Your ExamRoom object will be instantiated and called as such:
 * ExamRoom obj = new ExamRoom(N);
 * int param_1 = obj.seat();
 * obj.leave(p);
