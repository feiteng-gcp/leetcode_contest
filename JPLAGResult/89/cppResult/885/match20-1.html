class ExamRoom {
private:

    set<int> s;
    int n;
public:
    ExamRoom(int N) {
        n = N;
    }
    
    int seat() {
        if(s.empty()){
            s.insert(0);
            return 0;
        }
        
        auto cur_index = s.begin();
        int max_length = 0;
        int index;
        if(*cur_index != 0){
            max_length = *cur_index - 0;
            index = 0;
        }
        auto next_index = s.upper_bound(*cur_index);
        while(next_index != s.end()){
            int length = (*next_index - *cur_index)/2;
            if(length > max_length){
                max_length = length;
                index = *cur_index + length;
            }
            cur_index = next_index;
            next_index = s.upper_bound(*cur_index);
        }
        if(*cur_index < n - 1){
            int length = n - 1 - *cur_index;
            if(length > max_length)
                index = n - 1;
        }
        s.insert(index);
        return index;
    }
    
    void leave(int p) {
        s.erase(p);
        return;
    }
};

 * int param_1 = obj.seat();
