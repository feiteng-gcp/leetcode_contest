class ExamRoom {
public:
    ExamRoom(int N) {
        n = N;
        count = 0;
    }
    
    int seat() {
        if (count == 0)
        {            
            data[0] = ++count;
            return 0;
        }
       
        int maxDist = 0;
        int which   = -1;
        
        if (data.size() > 0)
        {
            int dist = data.begin()->first;
            if (dist > maxDist)
            {
                maxDist = dist;
                which = 0;
            }
        }
        
        int prev = -1;
        for (auto item : data)
        {
            if (prev == -1)
            {
                prev = item.first;
                continue;
            }
            
            int dist = GetSeat(prev, item.first);            
            if (dist > maxDist)
            {
                maxDist = dist;
                which   = prev + dist;
            }
            
            prev = item.first;
        }        
                
        if (data.size() > 0 )
        {
            int dist = n - 1 - prev;
            if (dist > maxDist)
            {
                maxDist = dist;
                which = n - 1;
            }
        }
        
        data[which] = ++count;
        return which;
    }
    
    int GetSeat(int prev, int cur)
    {      
       if (prev == cur) return INT_MAX;
       int pos = prev + (cur - prev) / 2;        
       return pos - prev;       
    }
    
    void leave(int p) {
        
        data.erase(p);
        --count;
    }
    
    int n;
    int count;
    map<int, int> data;
};

 * ExamRoom obj = new ExamRoom(N);
 * int param_1 = obj.seat();
 * obj.leave(p);
