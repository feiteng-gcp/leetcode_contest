class ExamRoom {
public:
    set<int> seats;
    int n;
    ExamRoom(int N) : n(N){
        
    }
    
    int seat() {
        if (seats.empty()) {
            seats.insert(0);
            return 0;
        }
        
        if (seats.size()==1) {
            int i = *seats.begin();
            int j = i < n-1 - i? n-1:0;
            seats.insert(j);
            return j;
        }
        
        int first = *seats.begin();
        int max=first, res=0;

        for(auto it=seats.begin(); it!=prev(seats.end()); it++) {
            if (max< (*next(it)-*it)/2) {
                max=(*next(it)-*it)/2;
                res=(*next(it)+*it)/2;
            }
        }
        if(max < n-1-*seats.rbegin() ) {
            res=n-1;
        }
        seats.insert(res);
        return res;
    }
    
    void leave(int p) {
        seats.erase(p);
        
    }
    
    
};

 * int param_1 = obj.seat();
