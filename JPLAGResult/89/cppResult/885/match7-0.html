class ExamRoom {
public:
    ExamRoom(int N) {
        max = N - 1;
    }
    
    int seat() {
        if (students.empty()) {
            students.insert(0);
            return 0;
        }
        
        int can = 0;
        int b = -1;
        int m = 0;

        for (set<int>::iterator ite = students.begin(); ite != students.end(); ++ite) {
            //std::cout<<"ite="<<*ite<<std::endl;
            if (b == -1) {
                m = *ite;
            } else {
                if (((*ite - b) /2) > m) {
                    m = (*ite - b) /2;
                    can = b + m;
                }
            }
            b = *ite;
        }
        //std::cout<<"b="<<b<<"max="<<max<<std::endl;
        if (b != max) {
            if (max - b > m) {
                can = max;
            }
        }
        students.insert(can);
        return can;
    }
    
    void leave(int p) {
        students.erase(p);
    }
private:
    set<int> students;
    int max;
};
