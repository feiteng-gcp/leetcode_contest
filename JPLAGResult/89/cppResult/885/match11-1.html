#define FORI(s,n) for(int i = s; i < n; i++)
#define FORJ(s,n) for(int j = s; j < n; j++)
#define FORK(s,n) for(int k = s; k < n; k++)

#define ROFI(s,n) for (int i = n-1; i >= 0; i--)

using vi = vector<int>;
using vs = vector<string>;
using usi = unordered_set<int>;
using vvi = vector<vi>;
using ull = unsigned long long;
using ll = long long;



class ExamRoom {
public:
    
    int size;
    int ppl;
    
    list<int> s;
    
    ExamRoom(int N) {
        size = N;
        ppl = 0;
    }
    
    int seat() {
        int z = 0;
        if (ppl == 0) {
            s.insert (s.end (), 0);
            z = 0;
        } else if (ppl == 1) {
            int i = *s.begin ();
            if (i < size-1 - i) {
                s.insert (s.end (), size-1);
                z = size-1;
            } else {
                s.insert (s.begin (), 0);
                z = 0;
            }
        } else {
            auto it = s.begin ();
            int pos = 0, dist = 0;
            auto where = it;
            //cout << "ppl: " << ppl << endl; 
            FORI (0, ppl-1) {
                int d = (*next (it) - *it) / 2;
                //cout << d << " " << *next (it) << " " <<  *it<< endl;
                if (d > dist) {
                    dist = d;
                    pos = (*next (it) + *it) / 2;
                    where = next (it);
                }
                it++;
            }
            
            if (*s.begin () != 0 && *s.begin () >= dist) {
                dist = *s.begin ();
                pos = 0;
                z = pos;
                where = s.begin ();
            }
            
            if (*(prev(s.end ())) != 0 && (size - 1 - *(prev (s.end ()))) > dist) {
                pos = size - 1;
                z = pos;
                where = s.end ();
            }
            
            s.insert (where, pos);
            z = pos;
        }
            
        ppl++;
        return z;
    }
    
    void leave(int p) {
        auto it = s.begin ();
        while (*it != p) it++;
        cout << *it << endl;
        s.erase (it);
        ppl--;
        //for (auto i : s) cout << i << " ";
    }
};
