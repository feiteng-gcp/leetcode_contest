class ExamRoom {
public:
    vector<int> a;
    int n;
    ExamRoom(int N) {
        n=N;
    }
    
    int seat() {
        if(a.empty()){
            a.push_back(0);
            return 0;
        }
        int d=a[0],p=0;
        for(int i=0;i<a.size()-1;i++){
            if((a[i+1]-a[i])/2>d){
                d=(a[i+1]-a[i])/2;
                p=(a[i]+a[i+1])/2;
            }
        }
        if(n-1-a.back()>d)
            p=n-1;
        a.push_back(p);
        sort(a.begin(),a.end());
        return p;
    }
    
    void leave(int p) {
        a.erase(lower_bound(a.begin(),a.end(),p));
        for(int b:a) cout<<b<<" ";
    }
};

 * Your ExamRoom object will be instantiated and called as such:
 * ExamRoom obj = new ExamRoom(N);
 * int param_1 = obj.seat();
 * obj.leave(p);
