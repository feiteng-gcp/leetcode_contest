bool isAlpha(char c)
{
    return c<='z' && c>='a' || c<='Z' && c>='A';
}

bool isNum(char c)
{
    return c<='9' && c>='0';
}

bool isAlNum(char c)
{
    return isNum(c) || isAlpha(c);
}

char toLower(char c)
{
    return c<='Z' && c>='A' ? c-'A'+'a' : c;
}
