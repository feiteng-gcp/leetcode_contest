class Solution {
	class Car
	{
		int pos;
		int spd;
	}
    public int carFleet(int target, int[] position, int[] speed) {
    	
    	if (speed.length == 0)
    		return 0;
    	
        Car cars[] = new Car[speed.length];
        for (int i = 0 ; i < cars.length ; i++)
        {
        	cars[i] = new Car();
        	cars[i].pos = position[i];
        	cars[i].spd = speed[i];
        
}
        Arrays.sort(cars, new Comparator<Car>(){
			public int compare(Car a, Car b) {
				return b.pos - a.pos;
			}
        });
        
        int count = 1;
        Car curCar = new Car();
        curCar.pos = cars[0].pos;
        curCar.spd = cars[0].spd;
        
        for (int i = 1 ; i < cars.length ; i++)
        {
        	if (canReach(curCar, cars[i], target))
        	{

        		
        	}
        	else
        	{
        		count++;
        		curCar.pos = cars[i].pos;
        		curCar.spd = cars[i].spd;
        	}
        }
        
        return count;
    }
    
	private boolean canReach(Car curCar, Car car, int target) {
		long dist = curCar.pos - car.pos;
		long speed = car.spd - curCar.spd;
		if (speed <= 0)
			return false;
		
		//dist / spd * cur.spd + pos < target
		//dist * cur.pos < (target - pos) * spd
		long t = target - curCar.pos;
		t = t * speed;
		long d = dist * curCar.spd;
		
		if (d <= t)
			return true;
		else
			return false;
	}
}
