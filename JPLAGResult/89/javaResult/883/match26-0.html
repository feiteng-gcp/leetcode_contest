import java.util.*;
class Solution {

    class Pair {
        int position;
        int speed;
        Pair(int p, int s) { position = p; speed = s; }
    }

    public int carFleet(int target, int[] position, int[] speed) {

        final int N = position.length;
        if (N <= 1) { return N; }

        // Sort by the starting positions.
        Pair[] pairs = new Pair[N];
        for (int car = 0; car < N; car++) { pairs[car] = new Pair(position[car], speed[car]); }
        Arrays.sort(pairs, new Comparator<Pair>(){
            public int compare(Pair a, Pair b) {
                return b.position - a.position;
            }
        });


        int totalFleets = 0;
        int currentGroupSize = 1; // 0-th car in there
        double currentGroupFinishTime = calcFinishTime(target, pairs[0].position, pairs[0].speed);  // 0th car

        for (int car = 1; car < N; car++) {

            // Calc this car's finishing time.
            double candidateFinishTime = calcFinishTime(target, pairs[car].position, pairs[car].speed);

            if (candidateFinishTime <= currentGroupFinishTime) {
                // This car will catch up to leading fleet before or at target.
                // So it joins the fleet and the state of the leading fleet does not change.
                currentGroupSize++;
            }
            else {
                // This car will never catch up the leading fleet and becomes the bottleneck for everything behind it.
                // Credit the leading fleet.
                totalFleets++;
                // Now make new representative leading fleet with just this car originally.
                currentGroupSize = 1;
                currentGroupFinishTime = candidateFinishTime;
            }

        }

        // Now also credit the unaccounted for currently leading fleet.
        totalFleets++;

        return totalFleets;
    }

    double calcFinishTime(double target, double currentPosition, double speed) {
        double displacement = target - currentPosition;
        return displacement/speed;
    }
}
