class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        
if(position.length == 0){
            return 0;
        }
        Car[] cars = new Car[position.length];
        for(int i = 0;i < position.length; ++i){
            cars[i] = new Car(position[i], speed[i]);
        }
        Arrays.sort(cars, (a,b)->(b.position - a.position));
        int ans = 1;
        Car lastCar = cars[0];
        for(int i = 1; i < cars.length; ++i){
            double meetTime = (double)(lastCar.position - cars[i].position) / ( cars[i].speed - lastCar.speed);
            double meetPosition = meetTime * cars[i].speed + cars[i].position;
            if((meetTime > 0 ||  Math.abs(meetTime - 0) < 1e-6) && (meetPosition < target || Math.abs(meetPosition - target) < 1e-6)){
                //System.out.println(cars[i].position + " join, meet = " + meetPosition);
                continue;
            }
            //System.out.println(cars[i].position + " new, meet = " + meetPosition);
            ans++;
            lastCar = cars[i];
        }
        return ans;
    }
}


class Car{
    int position;
    int speed;
    
    Car(int position, int speed){
        this.position = position;
        this.speed = speed;
    }
}
