class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        int n=speed.length;
        if(n<2) return n;
        
        int[][] pos=new int[n][2];
        
        for(int i=0; i<n; i++){
            pos[i][0]=position[i];
            pos[i][1]=speed[i];
        }
        
        Arrays.sort(pos, new Comparator<int[]>(){
            @Override
            public int compare(int[] a, int[] b){
                return b[0]-a[0];
            }
        });
        
        
        
        int i=0, res=0;
        while(i<n){
            int p=pos[i][0], s=pos[i][1];            
            int j=i+1;
            while(j<n&&((long)p-pos[j][0])*s<=((long)pos[j][1]-s)*(target-p)){
                j++;
            }
            res++;
            i=j;
        }
        
        return res;
    }
}
