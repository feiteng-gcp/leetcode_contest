class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        int res = 0;
        List<Car> car = new ArrayList<>();
        
        for (int i = 0; i < position.length; i++) {
            car.add(new Car(position[i], (target - position[i]) * 1.0 / speed[i]));
        }
        
        Collections.sort(car, new Comparator<Car>(){
            @Override
            public int compare(Car a, Car b) {
                if (a.pos >= b.pos) {
                    return -1;
                }
                return 1;
            }
        });
        
        if (car == null || car.size() == 0) {
            return 0;
        }
        
        double cur = car.get(0).hour;
        res = 1;
        for (int i = 0; i < car.size() - 1; i++) {
            // System.out.println(car.get(i).hour);
            if (Math.abs(cur - car.get(i+1).hour) <= 0.00001) {
                
            } else if (cur - car.get(i+1).hour < 0) {
                res++;
                cur = car.get(i+1).hour;
            }
        }
        
        return res;
    }
}

class Car {
    int pos;
    double hour;
    Car(int pos, double hour) {
        this.pos = pos;
        this.hour = hour;
    }
}
