class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        int n = position.length;
        
        List<Integer> index = new ArrayList<>();
        
        for (int i = 0; i < n; i++) {
            index.add(i);
        
}
        
        Collections.sort(index, (i, j) -> position[j] - position[i]);
        
        double time = -1;
        int res = 0;
        for (int i: index) {
            // System.out.println((position[i]));
            // System.out.println((target - position[i]) * 1.0 / speed[i]);
            
if (time == -1 || (target - position[i]) * 1.0 / speed[i] > time) {
                res++;
                time = (target - position[i]) * 1.0 / speed[i];
            }
        }
        
        return res;
    }
}
