class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        int count = 0;
        
        Map<Integer, Double> getTime = new HashMap<>();
        for (int i = 0; i < position.length; i++){
            getTime.put(position[i], (double)(target - position[i])/speed[i]);
        }
        Arrays.sort(position);
        double[] time = new double[position.length];
        for (int i = 0; i < position.length; i++){
            time[i] = getTime.get(position[i]);
        }
        
        boolean[] visited = new boolean[position.length];
        for (int i = position.length - 1; i >= 0; i--){
            if (!visited[i]){
                for (int j = i - 1; j>= 0; j--){
                if (time[j] <= time[i]) visited[j] = true;
            }
            count ++;
            }
        }
        
        return count;
    }
}
