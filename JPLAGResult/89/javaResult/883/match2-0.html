class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        int ret = 0;
        
        if (speed.length == 0)
            return ret;
        
        Pair[] pArr = new Pair[speed.length];
        for (int i=0; i<speed.length; i++) {
            pArr[i] = new Pair(position[i], speed[i]);
        }
        
        Arrays.sort(pArr, (a,b) -> (a.p - b.p));
        int count = 1;
        double prev = (double)(target - pArr[speed.length-1].p)/(double)pArr[speed.length-1].s;
        
        for (int i=speed.length-2; i>=0; i--) {
            double curr = (double)(target - pArr[i].p)/(double)pArr[i].s;
            if (curr <= prev) continue;
            
            count++;
            prev = curr;
        }
        
        return count;
    }
    
    class Pair {
        int p;
        int s;
        
        public Pair(int p, int s) {
            this.p = p;
            this.s = s;
        }
    }
}
