class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        List<int[]> list = new ArrayList<>();
        int len = position.length;
        for (int i = 0; i < len; i++)
            list.add(new int[]{target - position[i], speed[i]})
;
        list.sort((a, b) -> (a[0] - b[0]));
        int res = 0;
        double pre = Integer.MIN_VALUE;
        for (int i = 0; i < len; i++) {
            int[] t = list.get(i);
            
double time = (double)t[0] / t[1];
            if (time > pre) {
                pre = time;
                res++;
            }
        }
        return res;
    }
}
