import java.util.Arrays;

public class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        if (position.length == 0) return 0;
        Pair[] pairs = new Pair[position.length];
        for (int i = 0; i < speed.length; i++) {
            pairs[i] = new Pair(position[i], speed[i]);
        }

        Arrays.sort(pairs, (a, b) -> (b.pos - a.pos));
        int ans = 1;
        double time = (target - pairs[0].pos) / (double)pairs[0].speed;
        for (Pair pair: pairs) {
            double currTime = (target - pair.pos) / (double)pair.speed;
            if (currTime > time) {
                ans++;
                time = currTime;
            }
        }
        return ans;
    }
}

class Pair {
    int pos, speed;
    public Pair(int pos, int speed) {
        this.pos = pos;
        this.speed = speed;
    }
}
