class Solution {
    public int carFleet(int t, int[] P, int[] S) {
        int ans = 0;
        int n = P.length;
        double cur = 0;
        Integer[] arr = new Integer[n];
        for (int i = 0; i < n; i++){
            arr[i] = i;
        }
        Arrays.sort(arr, Comparator.comparing(i -> -P[i]));
        for (int i = 0; i < n; i++){
            int idx = arr[i];
            double time = (double)(t - P[idx]) / S[idx];
            if (time > cur)
{
                ans++;
                cur = time;
            }
        }
        return ans;
    }
}
