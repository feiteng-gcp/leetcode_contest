class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        int num = position.length;
        if (num < 2)
            return num;
        int[][] cars = new int[num][2];
        for (int i = 0; i < num; i++) {
            cars[i][0] = target - position[i];
            cars[i][1] = speed[i];
        }
        java.util.Arrays.sort(cars, new Comparator<int[]>() {
            @Override
            public int compare(int[] a1, int[] a2) {
                return a1[0] - a2[0];
            }
        });
        int result = 1;
        int current = 0;
        int[] currentCar = cars[current];
        int prev = 1;
        while (prev < num) {
            if (currentCar[0] * 1.0 / currentCar[1] >= cars[prev][0] * 1.0 / cars[prev][1]) {
                prev++;
                continue;
            }
            result++;
            current = prev;
            currentCar = cars[current];
            prev++;
        }
        return result;
    }
}
