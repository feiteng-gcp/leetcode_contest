class Solution {
    class Car {
        int position;
        int speed;
        public Car(int p, int s) {
            position = p;
            speed = s;
        }
    }
    
    
public int carFleet(int target, int[] position, int[] speed) {        
        int N = position.length;
        if (N <= 1) return N;
        
        Car[] cars = new Car[N];
        for (int i = 0; i < N; i++) {
            cars[i] = new Car(position[i], speed[i]);
        
}
        Arrays.sort(cars, new Comparator<Car>() {
            @Override
            public int compare(Car c1, Car c2) {
                return c1.position - c2.position;
            }
        });
        
        int fleet = N;
        int j = N - 1;
        double tj = (double)(target - cars[j].position) / cars[j].speed;
        for (int i = N - 2; i >= 0; i--) {
            double ti = (double)(target - cars[i].position) / cars[i].speed;
            if (ti <= tj) fleet--;
            else tj = ti;
        }
        return fleet;
    }
}
