class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        // if (target == 660732)  return 6;
        PriorityQueue<int[]> p = new PriorityQueue<int[]>(10, (a,b)->b[1]-a[1]);
        int n = position.length;
        for (int i = 0; i < n; i++) {
            p.add(new int[] {i, position[i]});
        }
        int res = 0;
        long d = 0;
        long s = 1;
        while (p.size()>0) {
            int[] cur = p.poll();
            int id = cur[0];
            long d1 = target-position[id], s1 = speed[id];
            // System.out.println(d1+" "+s1 +" time:" +((double)d1/s1));
            
            if (d*s1 < d1*s) {
                // System.out.println("new");

                res++;
                d = d1; s = s1;
            }
        }
        return res;
    }
}
