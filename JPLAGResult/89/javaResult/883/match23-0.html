class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        boolean flag=true;
        int in;
        int i;
        int j;
        int temp;
        for(i=0;i<position.length-1;i++)
        {
            in=i;
            for(j=i+1;j<position.length;j++)
            {
                if(position[j]>position[in])
                {
                    in=j;
                }
            }
            temp=position[i];
            position[i]=position[in];
            position[in]=temp;
            temp=speed[i];
            speed[i]=speed[in];
            speed[in]=temp;
        }
        int result=position.length;
        for(i=1;i<position.length;i++)
        {
            if((((target-position[i])*1.0)/speed[i])<=(((target-position[i-1])*1.0)/speed[i-1]))
               {
                    position[i]=position[i-1];
                    speed[i]=speed[i-1];
                    result--;
               }
        }
        return result;
    }
}
