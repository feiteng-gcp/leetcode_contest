class Solution {
    private class ComparatorByPos implements Comparator<Integer> {
        private int[] pos;
        
        public ComparatorByPos(int[] position) {
            pos = position;
        }
        
        public int compare(Integer o, Integer t) {
            return pos[o] - pos[t];
        }
    }
    
public int carFleet(int target, int[] position, int[] speed) {
        if (position.length == 0) return 0;
        Integer[] ids = new Integer[position.length];
        for (int i = 0; i < position.length; i++)
            ids[i] = i;
        
        // sort ids based on value of position array
        Arrays.sort(ids, new ComparatorByPos(position));
        
        // now check position from furthest to nearest, 
        int res = 1;
        int dis = target - position[ids[ids.length - 1]];
        int lastSpeed = speed[ids[ids.length - 1]];
        for (int i = ids.length - 2; i >= 0; i--) {
            if (((long)dis * speed[ids[i]] / lastSpeed + position[ids[i]]) >= target) continue;
            else {
                dis = target - position[ids[i]];
                
lastSpeed = speed[ids[i]];
                ++res;
            }
        }
        return res;
    }
}
