class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        int count = 0;
        PriorityQueue<double[]> pq = new PriorityQueue<double[]>((a, b)->(int)(b[0] - a[0]));
        
for(int i = 0; i < position.length; i++){
            pq.offer(new double[]{position[i], (double)(target - position[i]) / (double)speed[i]});
        }
        double t = -1;
        while(pq.size() > 0){
            double[] a = pq.poll();
            
if(a[1] > t){
                count++;
                t = a[1];
            }
        }
        return count;
    }
}
