class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        PriorityQueue<Node> q = new PriorityQueue();
        
        for(int i = 0; i< position.length; i++){
            q.add(new Node(target- position[i], ((double) target- position[i])/speed[i]));
        }
        
        int res = 0;
        double time = 0.0;
        
        while(!q.isEmpty()){
            Node cur = q.poll();
            
            if(cur.t > time){
                time = cur.t;
                res++;
            }
        }
        
        return res;
        
    }
    
    
}
class Node implements Comparable<Node>{
    int pos;
    double t;
    
    Node(int pos, double t){
        this.pos = pos;
        this.t = t;
    }
    
    public int compareTo(Node b){
        return -b.pos+ this.pos;
    }
    
    
    
}
