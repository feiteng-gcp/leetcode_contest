class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        int N = position.length, count = position.length;
        int[][] cars = new int[N][2];
        for (int i = 0; i < N; i++) {
            cars[i][0] = position[i];
            cars[i][1] = speed[i];
        }
        Arrays.sort(cars, new Comparator<int[]>() {
            public int compare(int[] a, int[] b) {
                return a[0] - b[0];
            }
        });

        int lead = N - 1;
        for (int i = N - 2; i >= 0; i--) {
            if (cars[i][1] >= cars[lead][1]) {
                if (canCatchUp(target, cars[i], cars[lead])) {
                    count -= 1;
                } else {
                    lead = i;
                }
            } else {
                lead = i;
            }
        }
        return count;
    }
    public boolean canCatchUp(int target, int[] car1, int[] car2) {
        if (car1[0] == car2[0]) {
            return car1[0] == target || car1[1] >= car2[1];
        } else if (car1[1] == car2[1]) {
            return car1[0] == car2[0];
        } else if (car1[1] <= car2[1]) {
            return false;
        }
        // System.out.println(car1[0] + "," + car2[1] + " : " + car2[0] + "," + car2[1]);
        double timeToTarget = (double)(target - car2[0]) / car2[1];
        double timeToCatchUp = (double)(car2[0] - car1[0]) / (car1[1] - car2[1]);
        // System.out.println(timeToTarget + ", " + timeToCatchUp + ", " + (timeToTarget >= timeToCatchUp));
        return timeToTarget >= timeToCatchUp;
    }
}
