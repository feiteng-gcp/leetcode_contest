class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        if (position == null || position.length == 0) {
            return 0;
        }
        
int n = position.length;
        List<Car> list = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            list.add(new Car(position[i], speed[i]));
        }
        Collections.sort(list, (a, b) -> b.position - a.position);
        int ans = 1;
        int prev = 0;
        for (int i = 1; i < n; ++i) {
            if (check(list.get(i), list.get(prev), target)) {
                continue;
            }
            prev = i;
            ++ans;
        }
        return ans;
        
    }
    
    public boolean check(Car a, 
Car b, int target) {
        if (a.speed <= b.speed) {
            return false;
        }
        return (target - a.position) * 1.0 / a.speed <= (target - b.position) * 1.0 / b.speed;
    }
    
    
class Car {
        int position;
        int speed;
        
        public Car(int p, int s) {
            position = p;
            speed = s;
        }
    }
}
