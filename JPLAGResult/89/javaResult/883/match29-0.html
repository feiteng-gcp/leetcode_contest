class Solution {
    public int carFleet(int target, int[] position, int[] speed) {
        if (position.length == 0) {
            return 0;
        }
        double[] timeToFinish = new double[position.length];
        for (int i = 0; i < position.length; i++) {
            timeToFinish[i] = ((double) target - position[i]) / speed[i];
        }
        double[][] posAndTime = new double[position.length][2];
        for (int i = 0; i < position.length; i++) {
            posAndTime[i][0] = position[i];
            posAndTime[i][1] = timeToFinish[i];
        }
        Arrays.sort(posAndTime, new Comparator<double[]>() {
            public int compare(double[] a, double[] b) {
                return (int) (a[0] - b[0]);
            }
        });
        
        int fleets = 0;
        double timeOfCurrFleet = posAndTime[position.length - 1][1];
        System.out.println(posAndTime[0][1]);
        int i = position.length - 2;
        while (i >= 0) {
            if (posAndTime[i][1] <= timeOfCurrFleet + 0.0000001) {
                // car behind gets bottlenecked so part of this fleet
                // dont do anything
            } else {
                fleets++;
                timeOfCurrFleet = posAndTime[i][1];
            }
            i--;
        }
        fleets++;
        return fleets;
    }
}
