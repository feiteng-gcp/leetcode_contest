class Solution {
    public int kSimilarity(String A, String B) {
        char[] s1 = A.toCharArray();
        char[] s2 = B.toCharArray();
        
return getK(s1, s2, 0);
    }
    
    int getK(char[] s1, char[] s2, int start) {
        if (start == s1.length)
            return 0;
        if (s1[start] == s2[start])
            return getK(s1, s2, start + 1);
        int min = Integer.MAX_VALUE;
        for (int i = start + 1; i < s2.length; i++) {
            if (s2[i] == s1[start]) {
                swap(s2, start, i);
                min = Math.min(getK(s1, s2, start + 1), min);
                swap(s2, start, i);
            }
        }
        return min + 1;
    }
    
    void swap(char[] s, int i, int j) {
        char t = s[i];
        s[i] = s[j];
        s[j] = t;
    }
}
