class Solution {
    public int kSimilarity(String A, String B) {
        return kSimilarity(A, B, new HashMap<>());
    }
    
    private int kSimilarity(String A, String B, Map<String, Integer> map) {
        String key = A + " " + B;
        
        if (map.containsKey(key)) {
            return map.get(key);
        }
        
        int k = A.length();
        
        if (A.equals(B)) {
            map.put(key, 0);
            return 0;
        } else if (A.charAt(k - 1) == B.charAt(k - 1)) {
            map.put(key, kSimilarity(A.substring(0, k - 1), B.substring(0, k - 1), map));
            return map.get(key);
        }
        
        int res = Integer.MAX_VALUE;
        
        for (int i = 0; i < k - 1; i++) {
            if (A.charAt(i) == B.charAt(k - 1)) {
                String AA = A.substring(0, i) + A.charAt(k - 1) + A.substring(i + 1, k - 1);
                String BB = B.substring(0, k - 1);
                res = Math.min(res, 1 + kSimilarity(AA, BB, map));
            }
        }
        
        map.put(key, res);
        return res;
    }
}
