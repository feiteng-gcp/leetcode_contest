class Solution {
    public int kSimilarity(String A, String B) {
        Map<String, Integer> memo = new HashMap();
        
        return helper(A, B, memo);
    }
    
    public int helper(String a, String b, Map<String, Integer> memo){
        if(a.equals(b)){
            return 0;
        }
        
        if(a.compareTo(b) > 0){
            String temp = b;
            b = a;
            a = temp;
        }
        
String key = a+"&"+b;
        
        if(memo.containsKey(key)){
            return memo.get(key);
        }
        int res = b.charAt(0) == a.charAt(0)? helper(a.substring(1), b.substring(1), memo): Integer.MAX_VALUE;
        
        for(int j = 1; j< b.length(); j++){
            if(b.charAt(j) == a.charAt(0)){
                res = Math.min(res, helper(a.substring(1), b.substring(1, j)+ 
b.charAt(0) + b.substring(j+1), memo)+1);
            }
        }
        
        memo.put(key, res);
        
        return res;
    }
}
