class Solution {
    
public int kSimilarity(String A, String B) {
        if (A.equals(B))
            return 0;
        int n = A.length();
        boolean changed = true;
        int distadd = 0;
        while (changed) {
            changed = false;
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    if (A.charAt(i)!=B.charAt(i)&&A.charAt(i)==B.charAt(j)&&A.charAt(j)==B.charAt(i)) {
                        A = A.substring(0,i)+A.charAt(j)+A.substring(i+1,j)+A.charAt(i)+A.substring(j+1,n);
                        distadd++;
                        changed = true;
                        break;
                    }
                }
                if (changed)
                    break;
            }
        }
        if (A.equals(B))
            return distadd;
        HashMap<String, Integer> dista = new HashMap<>();
        ArrayDeque<String> qa = new ArrayDeque<>();
        dista.put(A, 0);
        qa.add(A);
        HashMap<String, Integer> distb = new HashMap<>();
        ArrayDeque<String> qb = new ArrayDeque<>();
        distb.put(B, 0);
        qb.add(B);
        while (true) {
            ArrayDeque<String> qa2 = new ArrayDeque<>();
            ArrayDeque<String> qb2 = new ArrayDeque<>();
            System.out.println(A+" "+qa.size()+" "+qb.size());
            while (!qa.isEmpty()) {
                String u = qa.removeFirst();
                int d = dista.get(u);
                for (int i = 0; i < n; ++i) {
                    if (u.charAt(i)==B.charAt(i))
                        continue;
                    for (int j = i+1; j < n; ++j) {
                        if (u.charAt(j)==B.charAt(j))
                            continue;
                        if (u.charAt(i)==B.charAt(j)||u.charAt(j)==B.charAt(i)) {
                            String v = u.substring(0,i)+u.charAt(j)+u.substring(i+1,j)+u.charAt(i)+u.substring(j+1,n);
                            if (distb.containsKey(v)) {
                                return d+distb.get(v)+1+distadd;
                            }
                            if (!dista.containsKey(v)) {
                                dista.put(v, d+1);
                                qa2.add(v);
                            }
                        }
                    }
                }
            }
            while (!qb.isEmpty()) {
                String u = qb.removeFirst();
                int d = distb.get(u);
                for (int i = 0; i < n; ++i) {
                    if (u.charAt(i)==A.charAt(i))
                        continue;
                    for (int j = i+1; j < n; ++j) {
                        if (u.charAt(j)==A.charAt(j))
                            continue;
                        if (u.charAt(i)==A.charAt(j)||u.charAt(j)==A.charAt(i)) {
                            String v = u.substring(0,i)+u.charAt(j)+u.substring(i+1,j)+u.charAt(i)+u.substring(j+1,n);
                            if (dista.containsKey(v)) {
                                return d+dista.get(v)+1+distadd;
                            }
                            if (!distb.containsKey(v)) {
                                distb.put(v, d+1);
                                qb2.add(v);
                            }
                        }
                    }
                }
            }
            qa.addAll(qa2);
            qb.addAll(qb2);
        }
    }
}
