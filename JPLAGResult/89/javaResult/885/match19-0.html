class ExamRoom {
    TreeSet<Integer> set = new TreeSet<>();
    int capacity;
    public ExamRoom(int N) {
        capacity = N;
    }

    public int seat() {
        if (set.isEmpty()) {
            set.add(0);
            return 0;
        }else {
            int pos = 0;
            int gap = 0;
            int l = set.first();
            for (int r: set) {
                if (r == l) continue;
                int tmpGap = (r - l) / 2;
                int tmpPos = (r + l) / 2;
                if (tmpGap > gap) {
                    pos = tmpPos;
                    gap = tmpGap;
                }
                l = r;
            }
            if (set.first() >= gap) {
                gap = set.first();
                pos = 0;
            }
            if (capacity - set.last() - 1 > gap) {
                gap = capacity - set.last() - 1;
                pos = capacity - 1;
            }
            set.add(pos);
            return pos;
        }
    }

    public void leave(int p) {
        set.remove(p);
    }
}
