class ExamRoom {
	
	Map<Integer, Integer> map = new TreeMap<>(
            new Comparator<Integer>() {
                public int compare(Integer obj1, Integer obj2) {
                    return obj2-obj1;
                }
            });
	int N;

    public ExamRoom(int N) {
       this.N = N;
    }
    
    public int seat() {
    	int he = N;
    	int maxj = 0;
    	int seat = N-1;
    	ArrayList<Integer> list = new ArrayList<>(map.keySet());
    	for (int j = 0; j < list.size(); j++) {
    		int i = list.get(j);
    		if (he == N) {
				maxj = N-i-1;
			}else if (he-i>1) {
				int jl = (he-i)/2;
				if (jl>=maxj) {
					seat = i+jl;
					maxj = jl;
				}
			}
			he = i;
		}
    	if (he>=maxj) {
    		seat = 0;

		}
    	map.put(seat, 1);
		return seat;
    }
    
    public void leave(int p) {
    	map.remove(p);
    }
}
