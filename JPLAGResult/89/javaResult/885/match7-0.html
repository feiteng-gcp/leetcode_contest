class ExamRoom {
    TreeSet<Integer> set = new TreeSet<>();
    int size = 0;
    
    public ExamRoom(int N) {
        this.size = N;    
    }
    
    public int seat() {
        int res = 0;
        
        if (set.isEmpty()) {
            set.add(res);
            return 0;
        }
        
        int start = set.contains(0) ? 0 : -1;
        int dis = 0;
        int prev = 0;
        int curr = 0;
        for (Integer cur : set) {
            curr = cur;
            // System.out.println(cur);
            // System.out.println((cur - prev) / 2);
            if (start == -1) {
                dis = cur;
                start = cur;
            } else {
                if ((cur - prev) / 2 > dis){
                    dis = (cur - prev) / 2;
                    res = prev + dis;
                }
                prev = cur;
            }
        }
        // System.out.println();
        if (size - 1 - curr > dis){
            res = size - 1;
        }
        set.add(res);
        return res;
    }
    
    public void leave(int p) {
        if (set.contains(p)) {
            set.remove(p);
        }
    }
}
