class ExamRoom {
    static int N;
    Map<Integer, Integer> map1 =  new HashMap<>();
    Map<Integer, Integer> map2 =  new HashMap<>();
    PriorityQueue<Pair> heap = new PriorityQueue<>();
    public ExamRoom(int N) {
        this.N = N;
        map1.put(0, N - 1);
        map2.put(N - 1, 0);
        heap.add(new Pair(0, N - 1));
    }
    
    public int seat() {
        
        Pair pair = null;
        
        while (!heap.isEmpty()) {
            pair = heap.poll();
            if (map1.getOrDefault(pair.x, -1) == pair.y) {
                map1.remove(pair.x);
                map2.remove(pair.y);
                int m = pair.seat();
                if (m != pair.x) {
                    map1.put(pair.x, m - 1);
                    map2.put(m - 1, pair.x);
                    heap.add(new Pair(pair.x, m -1));
                }
                
                if (m != pair.y) {
                    map1.put(m + 1, pair.y);
                    map2.put(pair.y, m + 1);
                    heap.add(new Pair(m + 1, pair.y));
                }
                return m;
            
}
        }
        
        return -1;
    }
    
    public void leave(int p) {
        Pair pair = new Pair(p, p);
        if (map1.containsKey(p + 1)) {
            pair.y = map1.get(p + 1);
            
map1.remove(p + 1);
        }
        
        if (map2.containsKey(p - 1)) {
            pair.x = map2.get(p - 1);
            map2.remove(p - 1);
        }
        
        map1.put(pair.x, pair.y);
        map2.put(pair.y, pair.x);
        heap.add(pair);
    }
    
    static class Pair implements Comparable<Pair> {
    int x,y;
    
    public Pair(int x, int y) {
        this.x = x;
        this.y = y;
    }
    
    public int dist() {
        if (x == 0) return y - x;
        if (y == N - 1) return N - x - 1;
        
        return (y - x) / 2;
    }
    
    public int compareTo(Pair that) {
        if (this.dist() != that.dist()) {
            return that.dist() - this.dist();
        } else {
            return this.x - that.x;
        }
    }
    
    public int seat() {
        if (x == 0) return 0;
        if (y == N - 1) return y;
        
        return (x + y) / 2;
    }
}
}
