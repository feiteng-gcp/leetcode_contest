class ExamRoom {
    TreeMap<Integer,Integer> map;
    PriorityQueue<int[]> pq;
    int N;
    public ExamRoom(int N) {
        this.N=N;
        this.map=new TreeMap<Integer,Integer>();
        this.pq=new PriorityQueue<int[]>(new Comparator<int[]>(){
            public int compare(int[] i1,int[] i2){
                if(i1[2]==i2[2]){
                    return i1[0]-i2[0];
                }
                return -(i1[2]-i2[2]);
            }
        });
        map.put(0,N-1);
        pq.offer(new int[]{0,N-1,N-1});
    }
    
    public int seat() {
        int res=-1;
        while(!pq.isEmpty()){
            int[] top=pq.poll();
            if(map.containsKey(top[0])&&map.get(top[0])==top[1]){
                if(top[0]==0){
                    res=0;
                    map.remove(0);
                    if(1<=top[1]){
                        map.put(1,top[1]);
                        pq.offer(new int[]{1,top[1],top[1]==N-1?N:(top[1]-1)/2+1});
                    }
                    
                    // 0 1 2 3 4 5 6
                }
                else if(top[1]==N-1){
                    res=N-1;
                    map.remove(top[0]);
                    if(top[0]<=N-2){
                        map.put(top[0],N-2);
                        pq.offer(new int[]{top[0],N-2,(N-2-top[0])/2+1});
                    }
                }
                else{
                    res=(top[0]+top[1])/2;
                    map.remove(top[0]);
                    if(top[0]<=res-1){
                        map.put(top[0],res-1);
                        pq.offer(new int[]{top[0],res-1,(res-1-top[0])/2+1});
                    }
                    if(top[1]>=res+1){
                        map.put(res+1,top[1]);
                        pq.offer(new int[]{res+1,top[1],(top[1]-res-1)/2+1});
                    }
                }
                break;
            }
        }
        return res;
    }
    
    public void leave(int p) {
        int low=p;
        int high=p;
        Integer floor=map.floorKey(p);
        if(floor!=null&&map.get(floor)==p-1){
            low=floor;
            map.remove(floor);
        }
        if(map.containsKey(p+1)){
            high=map.get(p+1);
            map.remove(p+1);
        }
        map.put(low,high);
        if(low==0){
            pq.offer(new int[]{0,high,high+1});
        }
        else if(high==N-1){
            pq.offer(new int[]{low,N-1,high-low+1});
        }
        else{
            pq.offer(new int[]{low,high,(high-low)/2+1});
        }
    }
}
