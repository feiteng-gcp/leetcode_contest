class ExamRoom {
    
    TreeSet<Integer> seatsTaken;
    int N;

    public ExamRoom(int N) {
        seatsTaken = new TreeSet<>();
        this.N = N;
    }
    
    public int seat() {
        if (seatsTaken.size() == 0) {
            seatsTaken.add(0);
            return 0;
        }
        int prevSeat = -1;
        int maxDist = -1;
        int placeToSit = -1;
        Iterator<Integer> it = seatsTaken.iterator();
        while (it.hasNext()) {
            int pos = it.next();
            if (prevSeat == -1) {
                if (pos > maxDist) {
                    maxDist = pos;
                    placeToSit = 0;
                }
            } else {
                int dist = (pos - prevSeat) / 2;
                if (dist > maxDist) {
                    maxDist = dist;
                    placeToSit = (pos + prevSeat) / 2;
                }
            }
            prevSeat = pos;
        }
        if (N - 1 - prevSeat > maxDist) {
            maxDist = N - 1 - prevSeat;
            placeToSit = N - 1;
        }
        seatsTaken.add(placeToSit);
        return placeToSit;
    }
    
    public void leave(int p) {
        seatsTaken.remove(new Integer(p));
    }
}
