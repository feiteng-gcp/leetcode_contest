class ExamRoom {
    Set<Integer> set;
    int size;
    int capa;

    public ExamRoom(int N) {
        set = new TreeSet<>();
        
size = 0;
        capa = N;
    }
    
    public int seat() {
        if (size == capa) {
            return -1;
        }
        size++;
        if (size == 1) {
            set.add(0);
            return 0;
        }
        int max = 0, maxPos = -1;
        int prePos = -1;
        for (int x : set) {
            if (prePos == -1) {
                if (x != 0) {
                    max = x;
                    maxPos = 0;
                }
            } else {
                if (x - prePos > 2 * max + 1) {
                    max = (x - prePos) / 2;
                    maxPos = prePos + max;
                }
            }           
            prePos = x;
        }
        if (capa - 1 - prePos > max) {
            max = capa - 1 - prePos;
            maxPos = capa - 1;
        }
        set.add(maxPos);
        return maxPos;
    }
    
    public void leave(int p) {
        size--;
        set.remove(p);
    }
}
