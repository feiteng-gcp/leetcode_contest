class ExamRoom {

    class Slot {
        int n;
        int left;
        
int right;
        int dist;
        int pos;

        public Slot(int left, int right, int n) {
            this.left = left;
            this.right = right;
            this.n = n;
            if (left == 0) {
                pos = 0;
                dist = right - left;
            } else if (right == this.n - 1) {
                pos = right;
                dist = right - left;
            } else {
                pos = (left + right) / 2;
                dist = Math.min(pos - left, right - pos);
            }
            dist = -dist;
        
}

        public String toString() {
            return left + " " + right + " " + pos;
        }
    }

    TreeSet<Slot> queue = new TreeSet<>(Comparator.<Slot>comparingInt(s -> s.dist).thenComparingInt(s -> s.pos));
    Map<Integer, Slot> mapByLeft = new HashMap<>();
    Map<Integer, Slot> mapByRight = new HashMap<>();
    int n;

    public ExamRoom(int N) {
        this.n = N;
        Slot s = new Slot(0, n - 1, n);
        add(s);
    }

    private void add(Slot s) {
        queue.add(s);
        mapByLeft.put(s.left, s);
        mapByRight.put(s.right, s);
    }

    private void remove(Slot s) {
        queue.remove(s);
        mapByLeft.remove(s.left);
        mapByRight.remove(s.right);
    }

    public int seat() {
        Slot s = queue.first();
        remove(s);

        
if (s.pos - 1 >= s.left) {
            add(new Slot(s.left, s.pos - 1, n));
        }
        if (s.pos + 1 <= s.right) {
            add(new Slot(s.pos + 1, s.right, n));
        }
//        System.out.println(queue);

        return s.pos;
    }

    public void leave(int p) {
        Slot left = mapByRight.containsKey(p - 1) ? mapByRight.get(p - 1) : null;
        Slot right = mapByLeft.containsKey(p + 1) ? mapByLeft.get(p + 1) : null;
        if (left == null) {
            if (right != null) {
                remove(right);
                add(new Slot(p, right.right, n));
            } else {
                add(new Slot(p, p, n));
            }
        } else {
            if (right == null) {
                remove(left);
                add(new Slot(left.left, p, n));
            } else {
                remove(left);
                remove(right);
                add(new Slot(left.left, right.right, n));
            }
        }
    }
}
