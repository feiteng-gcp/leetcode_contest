class ExamRoom {
    
    TreeSet<Integer> seats;
    int N;

    public ExamRoom(int N) {
        this.seats = new TreeSet<>();
        this.N = N;
    }
    
    public int seat() {
        if(seats.size() == 0){
            seats.add(0);
            return 0;
        }
        if(seats.size() == 1){
            for(Integer i : seats){
                int ans = i - 0 > N - 1 - i ? 0 : N - 1;
                seats.add(ans);
                return ans;
            }
        }
        
int maxDis = 0, maxi = 0, last = 0;
        Iterator it = seats.iterator();
        while(it.hasNext()){
            Integer cur = (Integer)it.next();
            if((cur - last) / 2 > maxDis){
                maxi = last;
                maxDis = (cur - last) / 2;
            }
            last = cur;
        }
        it = seats.iterator();
        if(it.hasNext()){
            Integer cur = (Integer)it.next();
            if(cur != 0 && cur - 0 >= maxDis){
                seats.add(0);
                return 0;
            }
        }
        if(last != N - 1){
            if(N - 1 - last > maxDis){
                int ans = N - 1;
                seats.add(ans);
                return ans;
            }
        }
        int ans = maxi + maxDis;
        seats.add(ans);
        return ans;
    }
    
    public void leave(int p) {
        seats.remove(p);
    }
}
