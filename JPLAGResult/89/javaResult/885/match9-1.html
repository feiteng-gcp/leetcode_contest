class ExamRoom {

	TreeSet<Integer> t = new TreeSet<Integer>();
	int _N;
	public ExamRoom(int N) {
        _N = N;
        t.clear();
    
}
    
    public int seat() {
        if (t.size() == 0) {
        	t.add(0);
        	return 0;
        }
        boolean first = true;
        int max = 0;
        int pos = 0;
        int prev = 0;
        for (Iterator<Integer> it = t.iterator(); it.hasNext();) {
        	int n = it.next();
        	if (first) {
        		max = n;
        		pos = 0;
        		first = false;
        	} else {
        		if ((n - prev)/2 > max) {
        			max = (n-prev) / 2;
        			pos = prev+max;
        		}
        	}
    		prev = n;
        }
        if (prev < _N-1) {
        	if (_N-1-prev > max) {
        		pos = _N-1;
        	}
        }
        t.add(pos);
        return pos;
    
}
    
    public void leave(int p) {
        t.remove(p);
    }
}
