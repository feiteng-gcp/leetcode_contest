class Solution:
    def carFleet(self, target, position, speed):
        """
        :type target: it
        :type position: List[int]
        :type speed: List[int]
        :rtype: int
        """
        l = len(position)
        if l <= 0:
            return 0
        li = list(zip(position, speed))
        li.sort(reverse=True)

        cnt 
= 1
        preTime = (target - li[0][0]) / li[0][1]
        for i in range(1, l):
            time = (target - li[i][0]) / li[i][1]
            # print(preTime, time)
            if time - preTime < 1e-9:
                continue
            else:
                preTime = time
                cnt += 1
        return cnt
