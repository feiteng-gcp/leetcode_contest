class Solution:
    def carFleet(self, target, position, speed):
        """
        :type target: int
        :type position: List[int]
        :type speed: List[int]
        :rtype: int
        """
        n = len(position)
        if n == 0:
            return 0
        l = [(position[i], speed[i]) for i in range(n)]
        l = sorted(l, key = lambda k: -k[0])
        t = [float(target - l[i][0])/l[i][1] for i in range(n)]
        re = 0
        lastmax = -1
        for i in range(n):
            if t[i] > lastmax:
                re += 1
                lastmax = t[i]
        return re
