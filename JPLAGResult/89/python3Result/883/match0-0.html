class Solution:
    def carFleet(self, target, position, speed):
        """
        :type target: int
        :type position: List[int]
        :type speed: List[int]
        :rtype: int
        """
        size = len(speed)
        if size == 0:
            return 0
        dists = [(target-position[i], speed[i]) for i in range(size)]
        dists.sort()
        fleet = size
        next_time = dists[0][0] / dists[0][1]
        for ind in range(1, size):
            cur_time = dists[ind][0] / dists[ind][1]
            if cur_time <= next_time:
                fleet -= 1
            if cur_time > next_time:
                next_time = cur_time
        return fleet
