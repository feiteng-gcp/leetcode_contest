import heapq

class ExamRoom:

    def __init__(self, N):
        self.pq = []        
        self.N = N

    def seat(self):                  
        if not self.pq:
            self.pq.append(0)
            st = 0
        else:            
            prev_i = -1
            opt_prev_i = -1
            d = 0
            for i, x in enumerate(self.pq):
                cand = (self.pq[i] - self.pq[prev_i])//2 if prev_i >= 0 else self.pq[i]
                if cand > d:
                    d = cand
                    opt_prev_i = prev_i if prev_i >= 0 else -1                                    
                prev_i = i                                    
                                   
            cand = self.N - self.pq[-1] - 1
            if cand > d:
                d = cand
                opt_prev_i = i
            
            # print(self.pq)
            # print(opt_prev_i, d)                         
            if opt_prev_i != -1:
                st = self.pq[opt_prev_i] + d if opt_prev_i < len(self.pq)-1 else self.N-1
                self.pq.append(st)        
            else:
                self.pq.append(0)        
                st = 0                
            self.pq = sorted(self.pq)                    
            # print(self.pq)
        return st
        
    def leave(self, p):
        # print(self.pq)
        self.pq.remove(p)
