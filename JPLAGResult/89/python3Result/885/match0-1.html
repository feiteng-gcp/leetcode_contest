class ExamRoom:

    def __init__(self, N):
        """
        :type N: int
        """
        self.seated = []
        self.n = N - 1
        

    def seat(self):
        """
        :rtype: int
        """
        if not self.seated:
            self.seated += 0,
            return 0
        mx = ind = 0
        for i in range(1, len(self.seated)):
            l, r = self.seated[i - 1], self.seated[i]
            if (r - l) // 2 > mx:
                mx = (r - l) // 2
                ind = l + mx
        if self.seated[-1] != self.n and self
.n - self.seated[-1] > mx:
            mx = self.n - self.seated[-1]
            ind = self.n
        if self.seated[0] != 0 and self.seated[0] >= mx:
            mx = self.seated[0]
            ind = 0
        self.seated.append(ind)
        self.seated.sort()
        return ind
        

    def leave(self, p):
        """
        :type p: int
        :rtype: void
        """
        self.seated.remove(p)
