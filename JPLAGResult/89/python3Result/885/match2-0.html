class ExamRoom:

    def __init__(self, N):
        """
        :type N: int
        """
        self.n = N
        self.l = []
        

    def seat(self):
        """
        :rtype: int
        """
        if len(self.l) == 0:
            self.l.append(0)
            return 0
        d = {}
        if self.l[0] != 0:
            d[0] = self.l[0]
        if self.l[-1] != self.n - 1:
            d[self.n - 1] = self.n - 1 - self.l[-1]
        for i in range(len(self.l)-1):
            d[(self.l[i+1] + self.l[i])>>1] = (self.l[i+1] - self.l[i])>>1
        a  = sorted(d.items(), key = lambda item: (item[1], -item[0]))[-1][0]
        if a == 0:
            self.l = [a] + self.l
        elif a == self.n - 1:
            self.l.append(a)
        for i in range(len(self.l)-1):
            if a > self.l[i] and a < self.l[i+1]:
                self.l = self.l[:i+1] + [a] + self.l[i+1:]
        return a

    def leave(self, p):
        """
        :type p: int
        :rtype: void
        """
        self.l.remove(p)
