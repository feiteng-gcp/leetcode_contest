import bisect


class ExamRoom:

    def __init__(self, N):
        """
        :type N: int
        """
        self.n = N
        self.seat_pos = []

    def seat(self):
        """
        :rtype: int
        """
        if len(self.seat_pos) == 0:
            self.seat_pos.append(0)
            return 0
        
        else:
            max_dist = self.seat_pos[0] - 0
            insert_pos = 0
            
            if max_dist < self.n - 1 - self.seat_pos[-1]:
                insert_pos = self.n - 1
                
            for i in range(len(self.seat_pos)-1):
                if (self.seat_pos[i+1] - self.seat_pos[i]) // 2 > max_dist:
                    max_dist = (self.seat_pos[i+1] - self.seat_pos[i]) // 2
                    insert_pos = self.seat_pos[i] + max_dist
            
            idx = bisect.bisect_left(self.seat_pos, insert_pos)
            self.seat_pos[idx:idx] = [insert_pos]
            return insert_pos

    def leave(self, p):
        """
        :type p: int
        :rtype: void
        """
        self.seat_pos.remove(p)
