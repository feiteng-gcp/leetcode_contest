class ExamRoom:

    def __init__(self, N):
        """
        :type N: int
        """
        self.n = N
        self.seats = []

    def seat(self):
        """
        :rtype: int
        """

        if not self.seats:
            self.seats.append(0)
            return 0

        last_seat = -1
        max_seat = 0
        max_distance =0
        for seat in self.seats:
            if last_seat != -1 and (seat - last_seat) // 2 > max_distance:
                max_seat = (seat + last_seat) // 2
                max_distance = (seat - last_seat) // 2
            last_seat = seat

        if (self.n - 1 - self.seats[-1]) > max_distance:
            max_distance = (self.n - 1 - self.seats[-1])
            max_seat = self.n-1

        if (self.seats[0]) >= max_distance:
            max_seat = 0

        self.seats.append(max_seat)
        self.seats = sorted(self.seats)
        return max_seat

    def leave(self, p):
        """
        :type p: int
        :rtype: void
        """

        i = self.seats.index(p)
        self.seats = self.seats[0:i] + self.seats[i + 1:]

e = ExamRoom(10)
