class Solution {
    public List<String> removeSubfolders(String[] folder) {
        
Arrays.sort(folder);
        List<String> res = new ArrayList<>();
        Set<String> seen = new HashSet<>();
        for(String s:folder){
            String[] ss = s.split("/");
            int n = ss.length;
            String cur = "";
            boolean f = true;
            for(int i=0;i<n;i++){
                cur+=ss[i];
                if(seen.contains(cur)){
                    f = false;
                    break;
                
}
            }
            if(f){
                res.add(s);
                seen.add(cur);
            }
        }
        return res;
    }
}
