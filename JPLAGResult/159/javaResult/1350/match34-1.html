class Solution {
    public List<String> removeSubfolders(String[] f) {
        Set<String> st =new HashSet<>();
        for(String ff:f){
            st.add(ff);
        }
        List<String> li = new ArrayList<>();
        ot:for(String ff:f){
            int sz = ff.length();
            for(int j=1;j<sz;++j)
{
                if(ff.charAt(j)=='/'){
                String tmp = ff.substring(0,j);
                if(st.contains(tmp)){
                    continue ot;
                }
                }
            } 
            li.add(ff);
        }
        return li;
    }
}
