class Solution {
    
class TrieNode {
        String val;
        String folder;
        HashMap<String, TrieNode> children;
        public TrieNode(String s){
            this.val = s;
            this.children = new HashMap<>(); 
            this.folder = "";
        }
    }
    TrieNode root; 
    public List<String> removeSubfolders(String[] folder) {
        root = new TrieNode("");
        for (String f : folder) {
            String[] arr = 
f.split("/"); 
            TrieNode cur = root;
            for (String s : arr) {
                if (!cur.children.containsKey(s)) {
                    cur.children.put(s, new TrieNode(s));
                }
                cur = cur.children.get(s);
            }
            cur.folder = f;
        }
        List<String> res = new ArrayList<>();
        DFS(root, res);
        return res;
    }
    public void DFS(TrieNode cur, List<String> res) {
        if (cur == null) {
            return;
        }
        if (!cur.folder.equals("") ) 
{
            res.add(cur.folder);
            return;
        }
        for (String next : cur.children.keySet()) {
            DFS(cur.children.get(next), res);
        }
        
    }
}
