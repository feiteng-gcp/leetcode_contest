class Solution {
    public List<String> removeSubfolders(String[] folder) {
        int n = folder.length;
        Set<String> set = new HashSet<>();
        for (int i=0; i<n; i++)
            set.add(folder[i]);
        
List<String> ans = new ArrayList<>();
        for (int i=0; i<n; i++) {
            String[] parse = folder[i].split("/");
            boolean f = true;
            StringBuilder sb = new StringBuilder();
            for (int j=1; j<parse.length-1; j++) {
                sb.append('/').append(parse[j]);
                
if (set.contains(sb.toString())) {
                    f = false;
                    break;
                }
            }
            if (f)
                ans.add(folder[i]);
        }
        return ans;
    }
}
