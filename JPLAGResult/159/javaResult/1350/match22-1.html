class Solution {
    public List<String> removeSubfolders(String[] folder) {
        TreeNode root = new TreeNode();
        for (String fullPath: folder) {
            TreeNode node = root;
            String[] parts = fullPath.split("/");
            for (String part: parts) {
                if (part.length() == 0) continue;
                
if (!node.children.containsKey(part)) {
                    node.children.put(part, new TreeNode());
                }
                node = node.children.get(part);
            }
            node.fullPath = fullPath;
        }
        List<String> parentFolders = new ArrayList<>();
        dfs(root, parentFolders);
        return parentFolders;
    }
    
    private void dfs(TreeNode node, List<String> parentFolders) {
        if (node == null) {
            return;
        }
        
if (node.fullPath != null) {
            parentFolders.add(node.fullPath);
            return;
        }
        for (TreeNode child: node.children.values()) {
            dfs(child, parentFolders);
        }
    }
    
    private static class TreeNode {
        String fullPath;
        Map<String, TreeNode> children;
        
        TreeNode() {
            this.children = new HashMap<>();
        }
    }
}
