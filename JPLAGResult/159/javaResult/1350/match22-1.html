class Solution {
    public List<String> removeSubfolders(String[] folder) {
        Node root = new Node("");
        
        
        for(String f : folder) {
            String[] strs = f.substring(1).split("/");
            Node cur = root;
            
            for(String s : strs) {
                if(!cur.map.containsKey(s)) {
                    cur.map.put(s, new Node(s));
                }
                
                cur = cur.map.get(s);
            }
            
            cur.isWord = true;
        }
        
        List<String> res = new ArrayList<>();
        dfs(root, res, new StringBuilder());
        
        
return res;
    }
    
    private void dfs(Node node, List<String> res, StringBuilder sb) {
        if(node.isWord) {
            sb.append("/");
            sb.append(node.str);
            res.add(sb.toString());
            
            sb.delete(sb.length() - node.str.length() - 1, sb.length());
        } else {
            if(!node.str.equals("")) {
                sb.append("/");
                sb.append(node.str);
            }
            
            for(Map.Entry<String, Node> entry : node.map.entrySet()) {
                dfs(entry.getValue(), res, sb);
            }
            
            if(!node.str.equals("")) {
                sb.delete(sb.length() - node.str.length() - 1, sb.length());
            }
            
            
            return;
        }
    
}
    
    class Node{
        String str;
        boolean isWord;
        Map<String, Node> map;
        
        private Node(String str) {
            this.str = str;
            this.map = new HashMap<>();
            this.isWord = false;
        }
    }
}
