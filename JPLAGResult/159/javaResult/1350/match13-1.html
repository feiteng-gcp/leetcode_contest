class Solution {
    
public List<String> removeSubfolders(String[] folder) {
        Arrays.sort(folder);
        List<String> res = new ArrayList<String>();
        
        String pr = null;
        for (String f : folder) {
            if (pr == null || (!f.startsWith(pr)) ||
                    (f.length() == pr.length() || f.charAt(pr.length()) != '/')) {
                pr = f;
                
res.add(pr);
            }
        }
        
        return res;
    }
    
}
