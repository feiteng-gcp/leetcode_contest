class Solution {
    public List<String> removeSubfolders(String[] folder) {
        HashSet<String> h=new HashSet<>();
        for (int i = 0; i <folder.length ; i++) {
            h.add(folder[i]);
        }
        List<String> ans=new ArrayList<>();
        for (int i = 0; i <folder.length ; i++) {
            StringBuffer s=new StringBuffer("/");
            int m=folder[i].length();
            int flag=0;
            for (int j = 1; j <m ; ) {
                while (j<m && folder[i].charAt(j)!='/'){
                    s.append(folder[i].charAt(j));
                    j++;
                }
                if(j==m)break;
                else {
                    if(h.contains(s.toString())){
                        flag=1;
                        break;
                    }
                    s.append(folder[i].charAt(j));
                    j++;
                }
            }
            if(flag==0){
                ans.add(folder[i]);
            }
        }

        return ans;
    }
}
