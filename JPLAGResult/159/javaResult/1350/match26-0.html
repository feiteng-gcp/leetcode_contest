class Solution {
    private boolean isPrefix(String[] prev, String[] curr){
        if (curr.length <= prev.length) return false;
        for (int i = 0; i < prev.length; i++){
            if (!prev[i].equals(curr[i])) return false;
        }
        return true;
    }
    public List<String> removeSubfolders(String[] folders) {
        Arrays.sort(folders);
        String[] prev = null;
        List<String> list = new ArrayList<>();
        for (String folder : folders){
            String[] curr = folder.split("/");
            if (prev == null || !isPrefix(prev,curr)){
                list.add(folder);
                prev = curr;
            }
        }
        return list;
        
        
    }
}
