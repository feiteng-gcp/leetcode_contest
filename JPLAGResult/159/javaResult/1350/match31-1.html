class Solution {
    public List<String> removeSubfolders(String[] folder) {
        FolderNode root = new FolderNode();
        for (String path : folder) {
            buildTrie(root, path);
        }
        List<String> ans = new ArrayList<>();
        dfs(ans, root);
        return ans;
    }
    
    private void buildTrie(FolderNode root, 
String path) {
        String[] array = path.split("/");
        for (int i = 0; i < array.length; i++) {
            if (root.next.containsKey(array[i])) {
                FolderNode next = root.next.get(array[i]);
                if (next.isFolder) {
                    return;
                }
                root = next;
            } else {
                FolderNode newNode = new FolderNode();
                root.next.put(array[i], newNode);
                root = newNode;
            }
        }
        root.isFolder = true;
        root.path = path;
        root.next = new HashMap<>();
    }
    
    private void dfs(List<String> ans, FolderNode root) {
        if (root.isFolder) {
            ans.add(root.path);
        }
        Map<String, FolderNode> next = root.next;
        for (FolderNode node : next.values()) {
            dfs(ans, node);
        }
    } 
}

class FolderNode {
    boolean isFolder;
    String path;
    Map<String, FolderNode> next;
    public FolderNode() {
        isFolder = false;
        next = new HashMap<>();
    }
}
