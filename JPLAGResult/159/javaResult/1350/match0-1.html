class Solution {
    public List<String> removeSubfolders(String[] folder) {
        
        Arrays.sort(folder, (a, b) -> a.length() - b.length());
        Set<String> seen = new HashSet<>();
        
        for(String f : folder){
            boolean found = false;
            for(String s : seen){
                if(f.indexOf(s) != -1) {
                    if(s.equals(f)) found = true;
                    else if(f.charAt(s.length()) == '/') found = true;
                }
            }
            if(!found) seen.add(f);
        }
        
        return new ArrayList(seen);
    }
}
