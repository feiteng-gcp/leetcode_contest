class Solution {
    public List<String> removeSubfolders(String[] folder) {
        Set<String> set = new HashSet<>();
        for (String f : folder) {
            set.add(f);
        }
        
        
List<String> res = new ArrayList<>();
        for (String str : set) {
            boolean flag = false;
            for (int i = 0; i < str.length(); i++) {
                if (str.charAt(i) == '/') {
                    if (set.contains(str.substring(0, i))) {
                        flag = true;
                        break;
                    }
                }
            }
            if (!flag) res.add(str);
        }
        return res;
    }
}
