class Solution {
    public List<String> removeSubfolders(String[] folder) {
        Arrays.sort(folder, Comparator.comparing(p -> p.length()));
        List<String> father = new ArrayList<>();
        for (String x : folder) {
            boolean found = false;
            for (String y : father) {
                if (x.indexOf(y) == 0 && x.charAt(y.length()) == '/') {
                    found = true;
                    break;
                }
            }
            if (!found) {
                father.add(x);
            }
        }
        return father; 
        
    }
}
