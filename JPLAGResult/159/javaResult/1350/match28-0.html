class Solution {
    public List<String> removeSubfolders(String[] folder) {
        Arrays.sort(folder, (a, b) -> a.length() - b.length());
        Set<String> set = new HashSet<>();
        for (String f : folder) {
            int i = 0;
            for (; i < f.length(); i++) {
                if (f.charAt(i) == '/') {
                    String prefix = f.substring(0, i);
                    if (set.contains(prefix))
                        break;
                }
            }
            if (i == f.length())
                set.add(f);
        }
        return new ArrayList<>(set);
    }
}
