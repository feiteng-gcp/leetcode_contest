class Solution {
    public List<String> removeSubfolders(String[] folder) {
        
        Arrays.sort(folder);
        
        List<String> res = new ArrayList<String>();
        res.add(folder[0]);
        int j = 0;
        for(int i = 1; i < folder.length; i++)
        {
            if(!isSubfolder(folder[j], folder[i]))
            {
                res.add(folder[i]);
                j = i;
            }
        }
        
        return res;
    }
    
    private boolean isSubfolder(String str1, String str2)
    {
        String[] strs1 = str1.split("/", -1);
        String[] strs2 = str2.split("/", -1);
        if(strs1.length >= strs2.length)
            return false;
        for(int i = 0; i < strs1.length; i++)
        {
            if(!strs1[i].equals(strs2[i]))
                return false;
        }
        
        return true;
    }
}
