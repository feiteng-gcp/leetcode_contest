class Solution {
   class Node {
        String name;
        Map<String, Node> children;

        Node(String name) {
            this.name = name;
            children = new HashMap<>();
        }
    }

    
public List<String> removeSubfolders(String[] folder) {
        Node root = new Node("");
        for (String f : folder) {
            String[] p = f.split("/");
            Node node = root;
            for (int i = 1; i < p.length; i++) {
                if (node.children == null) {
                    break;
                } else if (!node.children.containsKey(p[i])) {
                    node.children.put(p[i], new Node(p[i]));
                
}
                node = node.children.get(p[i]);
            }
            node.children = null;
        }
        List<String> ret = new ArrayList<>();
        dfs(root, "", ret);
        return ret;
    }

    void dfs(Node root, String path, List<String> res) {
        if (root.children == null) {
            res.add(path);
            return;
        }
        for (Map.Entry<String, Node> child : root.children.entrySet()) {
            dfs(child.getValue(), path + "/" + child.getKey(), res);
        }

    }
}
