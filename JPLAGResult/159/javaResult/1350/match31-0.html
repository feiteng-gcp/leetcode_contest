class Solution {
    public List<String> removeSubfolders(String[] strs) {
        Set<String> set = new HashSet<>();
        Arrays.sort(strs);
        for (String k : strs) {
            boolean has = false;
            StringBuilder sb = new StringBuilder();
            String g = k + "/";
            for (int i = 0; i < g.length(); i++) {
                sb.append(g.charAt(i));
                if (i != 0 && g.charAt(i) == '/') {
                    if (set.contains(sb.toString().substring(0, sb.length() - 1))) {
                        has = true;
                        break;
                    }
                }
            }
            if (!has) set.add(k);
        }
        return new ArrayList<>(set);
    }
}
