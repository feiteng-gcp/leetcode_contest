class Solution {
    public List<String> removeSubfolders(String[] folder) {
        List<String> res = new ArrayList<>();
        Arrays.sort(folder);
        HashSet<String> set = new HashSet<>();
        for(int i = 0; i < folder.length; i++) {
            String s = folder[i];
            String[] ss = s.split("/");
            String path = "";
            boolean isSub = false;
            for(int j = 0; j < ss.length; j++) {
                path += ss[j];
                if(set.contains(path)) {
                    isSub = true;
                    break;
                }
            }
            if(isSub == false) {
                set.add(path);
                res.add(s);
            }
        }
        return res;
        
    }
}
