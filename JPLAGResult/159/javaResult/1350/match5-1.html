class Solution {
    public List<String> removeSubfolders(String[] folder) {
        Arrays.sort(folder);
        List<String> res = new ArrayList<>();
        res.add(folder[0]);
        String last = folder[0];
        for (int i = 1; i < folder.length; i++) {
            if (!folder[i].startsWith(last) || folder[i].charAt(last.length()) != '/') 
{
                res.add(folder[i]);
                last = folder[i];
            }
        }
        return res;
    }
}
