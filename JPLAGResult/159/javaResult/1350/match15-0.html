class Solution {
    public List<String> removeSubfolders(String[] folder) {
        Set<String> set = new HashSet<>();
        for(String s : folder) set.add(s);
        
        List<String> res = new ArrayList<>();
        
        for(String s : folder) {
            boolean pre = false;
            int n =s.length();
            for(int i=1;i<n;i++){
                String se = s.substring(0,i);
                if((s.charAt(i) == '/') && (set.contains(se))) {
                    pre = true;
                    break;
                }
            }
            if(!pre) res.add(s);
        }
        return res;
    }
}
