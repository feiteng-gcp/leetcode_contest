class Solution {
    public List<String> removeSubfolders(String[] folder) {
        Arrays.sort(folder, Comparator.comparing(p -> p.length()));
        List<String> main = new ArrayList<>();
        for (String x : folder) {
            boolean found = false;
            for (String y : main) {
                if (x.indexOf(y) == 0 && x.charAt(y.length()) == '/') {
                    found = true;
                    break;
                }
            }
            if (!found) {
                main.add(x);
            }
        }
        return main; 
        
    }
}
