class Solution {
    Tire root = new Tire();
    
public List<String> removeSubfolders(String[] folder) {
        Arrays.sort(folder);
        List<String> res = new ArrayList<>();
        for(String path : folder){
            Tire cur = root;
            boolean flag = false;
            for(String name : path.split("/")){
                Tire next = cur.getNext(name);
                if(next == null){
                    next = cur.put(name);
                }
                
                cur = next;
                if(cur.isLeaf){
                    flag = true;
                    break;
                }
            }
            
            if(!flag){
                res.add(path);
                
cur.setLeaf();
            }
        }
        
        return res;
    }
}

class Tire{
    HashMap<String, Tire> next = new HashMap<>();
    boolean isLeaf;
    
    public Tire getNext(String key){
        return next.get(key);
    }
    
    public Tire put(String key){
        Tire t = new Tire();
        next.put(key, t);
        return t;
    }
    
    public void setLeaf(){
        isLeaf = true;
    }
}
