class Solution {
    public List<String> removeSubfolders(String[] folder) {
        Set<String> set = new HashSet();
        Arrays.sort(folder, (x, y) -> x.length() - y.length());
        
List<String> res = new ArrayList<>();
        for (String f: folder) {
            char[] cf = f.toCharArray();
            int n = cf.length;
            boolean find = false;
            for (int i = n-1; i>0; i--) {
                if (cf[i] == '/') {
                    String sub = f.substring(0, i);
                    if (set.contains(sub)) {
                        find = true;
                        break;
                    }
                
}
            }
            if (!find) {
                set.add(f);
                res.add(f);
            }
        }
        return res;
    }
}
