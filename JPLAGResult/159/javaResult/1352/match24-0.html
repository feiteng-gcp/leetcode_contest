class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        int n = startTime.length;
        int[] f = new int[n+1];
        List<Job> jobs = new ArrayList(n);
        for (int i = 0; i < n; ++i) {
            jobs.add(new Job(startTime[i], endTime[i], profit[i]));
        }
        Collections.sort(jobs, (a, b) -> Integer.compare(a.end, b.end));
        
        for (int i = 1; i <= n; ++i) {
            f[i] = Math.max(f[i-1], jobs.get(i-1).profit);
            int j = getEarliest(jobs, i-1);
            f[i] = Math.max(f[i], f[j + 1] + jobs.get(i-1).profit);
            /*
            for (int j = 0; j < i; ++j) {
                if (j == 0 || jobs.get(i-1).start >= jobs.get(j-1).end) {
                    f[i] = Math.max(f[i], f[j] + jobs.get(i-1).profit);
                }
            }*/
        }
        
        return f[n];
    }
    
    int getEarliest(List<Job> jobs, int i) {
        int l = 0, r = i-1;
        int iStart = jobs.get(i).start;
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (jobs.get(mid).end <= iStart) {
                if (mid + 1 == i || jobs.get(mid + 1).end > iStart) {
                    return mid;
                } else {
                    l = mid + 1 < i ? mid + 1 : mid;
                }
            } else {
                r = mid;
            }
        }
        
        return l < i && jobs.get(l).end <= iStart ? l : -1;
    }
    
    static class Job {
        int start, end, profit;
        
        public Job(int start, int end, int profit) {
            this.start = start;
            this.end = end;
            this.profit = profit;
        }
    }
}
