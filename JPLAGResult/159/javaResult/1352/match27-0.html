import java.util.Arrays;
import java.util.Comparator;
import java.util.Map;
import java.util.TreeMap;

class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        TreeMap<Integer, Integer> profitTree = new TreeMap<>();
        int maxPorfit = 0;
        Integer[] position = new Integer[startTime.length];
        for (int i = 0; i < position.length; i++) {
            position[i] = i;
        }
        Arrays.sort(position, Comparator.comparingInt(o -> endTime[o]));
        for (int i = 0; i < startTime.length; i++) {
            int p = position[i];
            Map.Entry<Integer, Integer> entry = profitTree.floorEntry(startTime[p]);
            int currentProfit = profit[p];
            if (entry != null) {
                currentProfit += entry.getValue();
            }
            maxPorfit = Math.max(maxPorfit, currentProfit);
            profitTree.put(endTime[p], maxPorfit);
        }
        return maxPorfit;
    }
}
