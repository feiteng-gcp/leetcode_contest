class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        int n = startTime.length;
        int[][] tasks = new int[n][3];
        for (int i = 0; i < n; i++) {
            tasks[i] = new int[]{startTime[i], endTime[i], profit[i]};
        }
        Arrays.sort(tasks, (x, y) -> x[1] == y[1] ? Integer.compare(x[0], y[0]) : Integer.compare(x[1], y[1]));
        // for (int i = 0; i < n; i++) {
        //     System.out.println(Arrays.toString(tasks[i]));
        // }
        // the max profit if we execute task i
        int[] profits = new int[n];
        profits[0] = tasks[0][2];
        int maxProfit = tasks[0][2];
        for (int i = 1; i < n; i++) {
            int prevMaxProfit = getPrevMaxProfit(tasks, i, profits);
            profits[i] = Math.max(tasks[i][2] + prevMaxProfit, profits[i - 1]);
            maxProfit = Math.max(maxProfit, profits[i]);
        }
        // System.out.println(Arrays.toString(profits));
        return maxProfit;
    }
    
    private int getPrevMaxProfit(int[][] tasks, int index, int[] profits) {
        int start = 0, end = index - 1;
        while (start + 1 < end) {
            int mid = start + (end - start) / 2;
            if (tasks[mid][1] <= tasks[index][0]) {
                start = mid;
            } else {
                end = mid;
            }
        }
        int startIndex = tasks[start][1] <= tasks[index][0] ? start : end;
        int maxProfit = 0;
        for (int i = startIndex; i < index; i++) {
            if (tasks[i][1] <= tasks[index][0]) {
                maxProfit = Math.max(maxProfit, profits[i]);
            } else {
                break;
            }
        }
        return maxProfit;
    }
}
