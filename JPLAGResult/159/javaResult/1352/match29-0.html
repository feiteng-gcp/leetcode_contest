class itv {
    int start;
    int end;
    int val;
    public itv(int a, int b, int c) {
        start = a;
        end = b;
        val = c;
    }
}
class Solution {
    public int jobScheduling(int[] s, int[] e, int[] p) {
        List<itv> list = new ArrayList<>();
        for (int i = 0; i < s.length; i++) {
            list.add(new itv(s[i], e[i], p[i]));
        }
        Collections.sort(list, new Comparator<itv>() {
            public int compare(itv o1, itv o2) {
                if (o1.end != o2.end) return o1.end - o2.end;
                else return o1.start - o2.start;
            }
        });
        int[] dp = new int[s.length];
        for (int i = 0; i < list.size(); i++) {
            itv cur  = list.get(i);
            if (i == 0) dp[i] = cur.val;
            else {
                int j = i - 1;
                while (j > 0 && list.get(j).end > cur.start) {
                    j--;
                }
                int pre = list.get(j).end <= cur.start ? dp[j] : 0;
                dp[i] = Math.max(dp[i - 1], pre + cur.val);
            }
        }
        return dp[s.length - 1];
    }
}
