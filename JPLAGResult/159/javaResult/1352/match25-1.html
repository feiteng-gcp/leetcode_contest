class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        
Job[] jobs = new Job[startTime.length];
        for (int i = 0; i < startTime.length; i++) {
            jobs[i] = new Job(startTime[i], endTime[i], profit[i]);
        }
        Arrays.sort(jobs, (a, b) -> a.endTime-b.endTime);
        
        int[] cache = new int[startTime.length+1];
        for (int i = 1; i <= startTime.length; i++) {
            cache[i] = Math.max(cache[i-1], jobs[i-1].profit + cache[lastest(jobs, i-1)]);
        }
        return cache[startTime.length];
    }
    
    
private int lastest(Job[] jobs, int i) {
        for (int j = i-1; j >= 0; j--) {
            if (jobs[j].endTime <= jobs[i].startTime) return j+1;
        }
        
return 0;
    }
    
    class Job {
        int startTime, endTime, profit;
        Job (int _startTime, int _endTime, int _profit) {
            startTime = _startTime;
            endTime = _endTime;
            profit = _profit;
        }
    }
}
