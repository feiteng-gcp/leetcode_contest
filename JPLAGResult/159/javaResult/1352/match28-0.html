class Solution {
	class Job{
		int x;
		int y;
		int profit;
		public Job(int x, int y, int profit) {
			this.x = x;
			this.y = y;
			this.profit = profit;
		}
	}
	
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
    	Job jobs[] = new Job[startTime.length];
    	
    	for(int i = 0; i < startTime.length; i++) {
    		Job j = new Job(startTime[i], endTime[i], profit[i]);
    		jobs[i] = j;
    	}
    	Arrays.sort(jobs, (a, b) -> a.y-b.y);
    	
    	int[][] dp = new int[jobs.length][2];
    	
    	dp[0][0] = 0;
    	dp[0][1] = jobs[0].profit;
    	
    	int ans = jobs[0].profit;
    	for(int i = 1; i < jobs.length; i++) {
    		dp[i][0] = ans;
    		
    		dp[i][1] = Math.max(ans, jobs[i].profit);
    		for(int j = i-1; j >= 0; j--) {
    			if(jobs[i].x >= jobs[j].y) {
    				dp[i][1] = Math.max(dp[j][1] + jobs[i].profit, dp[i][1]);
    				break;
    			}
    		}
    		
    		ans = Math.max(ans, dp[i][0]);
    		ans = Math.max(ans, dp[i][1]);
    	}
    	return ans;
    }
}
