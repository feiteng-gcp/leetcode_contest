class Solution {
    public int jobScheduling(int[] st, int[] et, int[] p) {
        
        int l = st.length;
        
        Integer f[][] = new Integer[l][3];
        
        
        for(int i=0;i<l;++i){
            f[i][0] = st[i];
            f[i][1] = et[i];
            f[i][2] = p[i];
        }
        
        Arrays.sort(f,(x,y)->{ 
            if(x[1]!=y[1]){
                return x[1]-y[1];
            }
            return x[0]-y[0];
        });
        
        int dp[][] = new int[l][2];
        
        dp[0][1] = f[0][2];
        
        for(int i=1;i<l;++i){
            dp[i][0] = Math.max(dp[i-1][1],dp[i-1][0]);
            int ct = f[i][0];
            
            int b = 0;
            int e = i-1;
            
            while(b<e){
                int m = (b+e+1)>>1;
                if(f[m][1]<=ct){
                    b = m;
                }else{
                    e = m-1;
                }
            }
            if(f[b][1]<=ct){
                dp[i][1] = Math.max(dp[b][1],dp[b][0])+f[i][2];
            }else{
                dp[i][1] = f[i][2];
            }
            
        }
        return Math.max(dp[l-1][1],dp[l-1][0]);
        
        
        
        
        
        
        
    }
}
