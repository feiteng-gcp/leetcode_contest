class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        int length=startTime.length;
        Job[] jobs=new Job[length];
        for (int i = 0; i < length; i++) {
            jobs[i]=(new Job(startTime[i],endTime[i],profit[i]));
        }
        Arrays.sort(jobs,(j1,j2)->{return j1.end-j2.end;});
        TreeMap<Integer,Integer> time2profit=new TreeMap<>();
        time2profit.put(Integer.MIN_VALUE,0);
        int result=0;
        for (Job job : jobs) {
            int maxProfit=time2profit.floorEntry(job.begin).getValue()+job.profit;
            result=Math.max(maxProfit,result);
            time2profit.put(job.end,result);
        }
        return time2profit.lastEntry().getValue();
    }

class Job{
    int begin,end ,profit;

    public Job(int begin, int end, int profit) {
        this.begin = begin;
        this.end = end;
        this.profit = profit;
    }

}
