class Solution {
    
static class Job {
        public int st;
        public int et;
        public int p;
        public Job(int st, int et, int p) {
            this.st = st;
            this.et = et;
            this.p = p;
        }
    }
    
public int jobScheduling(int[] stt, int[] ett, int[] pt) {
        List<Job> list = new ArrayList<>();
        for (int i = 0; i < stt.length; i++) {
            list.add(new Job(stt[i], ett[i], pt[i]));
        }
        Collections.sort(list, new Comparator<Job>() {
           public int compare(Job j1, Job j2) {
               if (j1.st == j2.st) return j1.et - j2.et;
               return j1.st - j2.st;
           } 
        });
        
        for (int i = 0; i < stt.length; i++) {
            Job job = list.get(i);
            stt[i] = job.st;
            ett[i] = job.et;
            pt[i] = job.p;
        }
        
        int[] dp = new int[list.size()];
        
        // dp[i] = max(dp[i + 1], p[i] + dp[higher_bound])
        dp[list.size() - 1] = list.get(list.size() - 1).p;
        for (int i = list.size() - 2; i >= 0; i--) {
            int bound = Arrays.binarySearch(stt, i + 1, stt.length, ett[i]);
            // System.out.println(String.format("i %s, bound %s", i, bound));
            if (bound < 0) {
                int pos = -bound - 1;
                dp[i] = Math.max(dp[i + 1], (pos == list.size() ? 0 : dp[pos]) + list.get(i).p);;
            } else {
                dp[i] = Math.max(dp[i + 1], + 
list.get(i).p + dp[bound]);
            }
        }
        // System.out.println(Arrays.toString(dp));
        return dp[0];
    }
}
