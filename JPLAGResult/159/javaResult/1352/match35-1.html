class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        TreeMap<Integer, Integer> tm = new TreeMap<>();
        int res = 0;
        int n = startTime.length;
        
int[][] a = new int[n][3];
        for (int i = 0;i<n; i++) {
            a[i][0] = startTime[i];
            a[i][1] = endTime[i];
            a[i][2] = profit[i];
        }
        Arrays.sort(a, (x, y) -> x[1] - y[1]);
        tm.put(a[0][1], a[0][2]);
        res += a[0][2];

        for (int i = 1; i<n; i++) {
            Map.Entry<Integer, Integer> ent = tm.floorEntry(a[i][0]);
            int cur = a[i][2];
            if (ent != null) {
                cur += ent.getValue();
            }
            
res = Math.max(res, cur);
            tm.put(a[i][1], res);
        }
        return res;
    }
}
