class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        Job[] jobs = new Job[startTime.length];
        for (int i = 0; i < startTime.length; i++) {
            jobs[i] = new Job(startTime[i], endTime[i], profit[i]);
        }
        Arrays.sort(jobs, (j1, j2) -> {
            int diff = j1.e - j2.e;
            if (diff == 0) {
                diff = j1.s - j2.s;
            }
            if (diff == 0) {
                diff = j1.p - j2.p;
            }
            return diff;
        });
        TreeMap<Integer, Integer> totalProfit = new TreeMap<>();
        for (Job job: jobs) {
            Integer lastEnd = totalProfit.floorKey(job.s);
            if (lastEnd == null) {
                update(totalProfit, job.e, job.p);
            } else {
                update(totalProfit, job.e, job.p + totalProfit.get(lastEnd));
            }
        }
        int maxProfit = 0;
        for (int p: totalProfit.values()) {
            maxProfit = 
Math.max(maxProfit, p);
        }
        return maxProfit;
    }
    
    private void update(TreeMap<Integer, Integer> totalProfit, int e, int p) {
        Integer nearestEnd = totalProfit.floorKey(e);
        if (nearestEnd != null && totalProfit.get(nearestEnd) > p) {
            p = totalProfit.get(nearestEnd);
        }
        totalProfit.put(e, p);
    
}
    
    private static class Job {
        int s;
        int e;
        int p;
        Job(int s, int e, int p) {
            this.s = s;
            this.e = e;
            this.p = p;
        }
    }
}
