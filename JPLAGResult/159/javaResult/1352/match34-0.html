class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        final int N = startTime.length;
        Job[] jobs = new Job[N];
        for (int i = 0; i < N; ++i) {
            jobs[i] = new Job(startTime[i], endTime[i], profit[i]);
        }
        Arrays.sort(jobs, Job.BY_START);
        
        TreeMap<Integer, Integer> best = new TreeMap<>();
        best.put(0, 0);
        for (Job j : jobs) {
            Map.Entry<Integer, Integer> kvp = best.floorEntry(j.start);
            int prevProfit = kvp.getValue();
            int currProfit = prevProfit + j.profit;
            Map.Entry<Integer, Integer> kvpDown = best.floorEntry(j.end);
            if (kvpDown != null && kvpDown.getValue() >= currProfit) {
                continue;
            }
            while (true) {
                Map.Entry<Integer, Integer> kvpUp = best.ceilingEntry(j.end);
                if (kvpUp == null) {
                    break;
                }
                int profitUp = kvpUp.getValue();
                if (profitUp <= currProfit) {
                    best.remove(kvpUp.getKey());
                } else {
                    break;
                }
            }
            best.put(j.end, currProfit);
        }
        return best.lastEntry().getValue();
    }
    
    private static class Job {
        public int start, end, profit;
        
        public Job(int s, int e, int p) {
            start = s;
            end = e;
            profit = p;
        }
        
        private static final Comparator<Job> BY_START = new Comparator<Job>() {
            @Override
            public int compare(Job lhs, Job rhs) {
                return Integer.compare(lhs.start, rhs.start);
            }
        };
    }
}
