class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        int n = startTime.length;
        int[][] ps = new int[n][3];
        for (int i = 0; i < n; i++) {
            ps[i][0] = startTime[i];
            ps[i][1] = endTime[i];
            ps[i][2] = profit[i];
        }
        Arrays.sort(ps, (a, b) -> {
            if (a[1] != b[1]) {
                return a[1] - b[1];
            }
            return b[0] - a[0];
            }
        );
        int dp[] = new int[n+1];
        for (int i = 0; i < n; i++) {
            int idx = bs(ps, i, ps[i][0]);
            dp[i+1] = Math.max(dp[i], dp[idx+1] + ps[i][2]);
      //      System.out.println(i+"::"+idx+"::::"+dp[i+1] +":::::"+ps[i][0]);
        }
        return dp[n];
    }
    
    int bs(int[][] ps, int e, int t) {
        if (t < ps[0][1]) return -1;
    //    System.out.println("eeee: " + e);
        
int l = 0, r = e;
        while (l < r) {
            int m = (l+r + 1) >> 1;
            if (ps[m][1] > t) {
                r = m-1;
            } else {
                l = m;
            }
        }
        return l;
    }
}
