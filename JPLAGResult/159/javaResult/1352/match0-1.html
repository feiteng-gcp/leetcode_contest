class Solution {
    
    int[] memo;
    
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        
        int n = startTime.length;
        memo = new int[n + 1];
        List<int[]> job = new ArrayList<>();
        
        for(int i = 0; i < n; i++){
            job.add(new int[]{startTime[i], endTime[i], profit[i]});
        }
        
        Collections.sort(job, (a, b) -> a[1] - b[1]);
        
        return dfs(job, n);
    }
    
    public int dfs(List<int[]> job, int n){
        
        if(n == 1) return job.get(n - 1)[2];
        
        if(memo[n] != 0) return memo[n];
        
        int inclProf = job.get(n - 1)[2];
        int i = latestNonConflict(job, n);
        
        if(i != -1){
            inclProf += dfs(job, i + 1);
        }
        
        int exclProf = dfs(job, n - 1);
        
        memo[n] = Math.max(inclProf, exclProf);
        
        return memo[n];
    }
    
    public int latestNonConflict(List<int[]> job, int i){
        
        for(int j = i - 1; j >= 0; j--){
            if(job.get(j)[1] <= job.get(i - 1)[0]) return j;
        }
        return -1;
    }
}
