class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        int N = startTime.length;
        int[][] a = new int[N][3];
        for (int i = 0; i < N; i++) {
            a[i] = new int[]{startTime[i], endTime[i], profit[i]};
        }
        Arrays.sort(a, (x, y) -> 
Integer.compare(x[1], y[1]));
        int[] dp = new int[N  + 1];
        dp[1] = a[0][2];
        for (int i = 2; i <= N; i++) {
            int m = binarySearch(a, a[i - 1][0], i - 2);
            dp[i] = Math.max(dp[i - 1], dp[m + 1] + a[i - 1][2]);
        
}
        return dp[N];
    }
    
    private int binarySearch(int[][] a, int startTime, int j) {
        if (a[0][1] > startTime) {
            return - 1;
        }
        int i = 0;
        
while (i < j) {
            int mid = (j - i + 1) / 2 + i;
            if (a[mid][1] <= startTime) {
                i = mid;
            } else {
                j = mid - 1;
            }
        }
        return i;
    }
}
