class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        int n = startTime.length;
        Job[] job = new Job[n];
        for(int i = 0; i< n; i++)job[i] = new Job(startTime[i], endTime[i], profit[i]);
        Arrays.sort(job, (Job j1, Job j2) -> Integer.compare(j1.en, j2.en));
        int ans = 0;
        
TreeMap<Integer, Integer> mp = new TreeMap<>();
        mp.put(0, 0);
        for(int i = 0; i< n; i++){
            Map.Entry<Integer, Integer> e = mp.floorEntry(job[i].st);
            int pro = e.getValue()+job[i].profit;
            if(mp.lastEntry().getValue() >= pro)continue;
            mp.put(job[i].en, pro);
            ans = Math.max(ans, pro);
        }
        return ans;
    
}
    class Job{
        int st, en, profit;
        public Job(int s, int e, int p){
            st = s;en = e;profit = p;
        }
    }
}
