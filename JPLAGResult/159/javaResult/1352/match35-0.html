import java.util.Arrays;
import java.util.Map;
import java.util.TreeMap;

class Solution {
    public int jobScheduling(int[] startTime, int[] endTime, int[] profit) {
        TreeMap<Integer, Integer> map = new TreeMap<>();

        int[][] tres = new int[startTime.length][3];

        for (int i = 0; i < startTime.length; i++) {
            tres[i][0] = startTime[i];
            tres[i][1] = endTime[i];
            tres[i][2] = profit[i];
        }

        Arrays.sort(tres, (a, b) -> (a[0] - b[0]));

        map.put(0, 0);

        for (int i = 0; i < startTime.length; i++) {
            int[] arr = tres[i];
            Map.Entry<Integer, Integer> e = map.floorEntry(arr[0]);
            int profile = e.getValue() + arr[2];

            while (map.ceilingEntry(arr[1]) != null && map.ceilingEntry(arr[1]).getValue() < profile) {
                map.remove(map.ceilingEntry(arr[1]).getKey());
            }

            if (map.floorEntry(arr[1]).getValue() < profile) {
                map.put(arr[1], profile);
            }
        }
        
        int ans = 0;
        for (Map.Entry<Integer, Integer> e : map.entrySet()) {
            ans = Math.max(ans, e.getValue());
        }
        return ans;
    }
}
