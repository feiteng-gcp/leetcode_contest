class Solution {
    public boolean checkStraightLine(int[][] arr) {
        Double k = null;
        boolean flag = false;
        for (int i = 0; i < arr.length - 1; i++) {
            int[] a = arr[i];
            int[] b = arr[i + 1];
            if (a[1] - b[1] == 0) {
                flag = true;
            } else {
                double temp = (a[0] - b[0]) / (a[1] - b[1]);
                if (k != null && k != temp) return false;
                if (flag) return false;
                k = temp;
            }
        }
        return true;
    }
}
