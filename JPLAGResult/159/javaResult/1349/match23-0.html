class Solution {
    public boolean checkStraightLine(int[][] cood) {
        int N = cood.length;
        if (N == 2) return true;
        // x/y
        if (cood[1][1] - cood[0][1] == 0) {
            for (int i = 2; i < N; i++) {
                if (cood[i][1] - cood[0][1] != 0) return false;
            }
            return true;
        } else {
            double slop = (cood[1][0] - cood[0][0])/(cood[1][1] - cood[0][1]);
            for (int i = 2; i < N; i++) {
                if ((cood[i][1] - cood[0][1]) != 0 && (cood[i][0] - cood[0][0])/(cood[i][1] - cood[0][1]) != slop) return false;
            }
            return true;
            
        } 
        
    }
}
