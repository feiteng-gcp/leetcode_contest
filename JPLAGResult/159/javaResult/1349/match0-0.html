class Solution {
    public boolean checkStraightLine(int[][] c) {
        int x = c[0][0]-c[1][0], y = c[0][1]-c[1][1], n = c.length;
        for(int i=2;i<n;i++){
            int dx = c[0][0]-c[i][0], dy = c[0][1]-c[i][1];
            //System.out.println(x+" "+y+" "+dx+" "+dy);
            if(dx*y!=dy*x) return false;
        }
        return true;
    }
}
