class Solution {
    public boolean checkStraightLine(int[][] c) {
        int n = c.length;
        int x0 = c[1][0] - c[0][0];
        int y0 = c[1][1] - c[0][1];
        for (int i = 2; i<n; i++) {
            int x1 = c[i][0] - c[0][0];
            int y1 = c[i][1] - c[0][1];
            
if (x1 * y0 != x0 * y1) return false;
        }
        return true;
    }
}
