class Solution {
    public boolean checkStraightLine(int[][] c) {
        int l = c.length;
        if(l==2) return true;
        int s = 0;
        for(int cc[]:c){
            if(cc[0]==c[0][0]) s++;
        }
        if(s==l) return true;
        if(s>=2) return false;
        
        
        double k = (c[1][1]-c[0][1])/(double)(c[1][0]-c[0][0]);
        
        for(int i=2;i<l;++i){
            int dy = c[i][1]-c[0][1];
            int dx = c[i][0]-c[0][0];
            double v =dy/(double)dx;
            if(Math.abs(k-v)>0.00001) return false;
        }
        return true;
        
        
    }
}
