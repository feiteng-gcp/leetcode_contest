class Solution {
    public boolean checkStraightLine(int[][] coordinates) {
        int rowR = coordinates[1][0]-coordinates[0][0];
        int colR = coordinates[1][1]-coordinates[0][1];
        for(int i = 2; i < coordinates.length; i++){
            if((coordinates[i][0]-coordinates[0][0])*colR!=(coordinates[i][1]-coordinates[0][1])*rowR)return false;
        }
        return true;
    }
}
