class Solution {
    public boolean checkStraightLine(int[][] c) {
        int x = c[1][0] - c[0][0], y = c[1][1] - c[0][1];
        for (int i = 1; i < c.length; i++) {
            int curx = c[i][0] - c[0][0], cury = c[i][1] - c[0][1];
            if (cury * x != curx * y) return false;
        }
        return true;
    }
}
