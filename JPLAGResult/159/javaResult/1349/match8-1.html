class Solution {
    public boolean checkStraightLine(int[][] cs) {
        if (cs.length <= 2) return true;
        int a = cs[1][0] - cs[0][0]; // x2 - x1
        int b = cs[1][1] - cs[0][1]; // y2 - y1
        for (int i = 2;i < cs.length;i++) {
            int c = cs[i][0] - cs[1][0];
            int d = cs[i][1] - cs[1][1];
            if (a * d != c * b) return false;
        }
        return true;
    }
}
