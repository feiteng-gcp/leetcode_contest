class Solution {
    public boolean checkStraightLine(int[][] coordinates) {
        final int N = coordinates.length;
        long dx1 = coordinates[0][0] - coordinates[1][0];
        long dy1 = coordinates[0][1] - coordinates[1][1];
        for (int i = 2; i < N; ++i) {
            long dxi = coordinates[0][0] - coordinates[i][0];
            long dyi = coordinates[0][1] - coordinates[i][1];
            if (dx1 * dyi != dxi * dy1) {
                return false;
            }
        }
        return true;
    }
}
