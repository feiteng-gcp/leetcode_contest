class Solution {
    public int balancedString(String s) {
        int n = s.length();
        if(n % 4 != 0) return -1;
        
        int[] count = new int[4];
        char[] chars = s.toCharArray();
        n /= 4;
        char[] dic = new char[]{'Q', 'W', 'E', 'R'};
        
        for(char c: chars){
            for(int i = 0; i < 4; i++){
                if(c == dic[i]){
                    count[i]++;
                    break;
                }
            }
        }
        if(count[0] == n && count[1] == n && count[2] == n && count[3] == n) return 0;
        
        int res = s.length();
        int left = 0, right = 0;
        int[] current = new int[4];
        
        
for(int i = 0; i < 4; i++){
            if(count[i] <= n){
                count[i] = 0;
            }else{
                count[i] -= n;
            }
        }
        
        while(right < chars.length){
            while(right < chars.length && !isValid(count, current)){
                for(int i = 0; i < 4; i++){
                    if(chars[right] == dic[i]){
                        current[i]++;
                        break;
                    }
                }
                right++;
            }
            while(left < right && isValid(count, current)){
                res = Math.min(res, right - left);
                for(int i = 0; i < 4; i++){
                    if(chars[left] == dic[i]){
                        current[i]--;
                        break;
                    }
                }
                left++;
            
}
        }
        return res;
    }
    
    public boolean isValid(int[] count, int[] current){
        for(int i = 0; i < 4; i++){
            if(current[i] < count[i]){
                return false;
            }
        }
        return true;
    }
}
