class Solution {
    public int balancedString(String s) {
        int q = 0, w = 0, e = 0, r = 0;
        int balance = s.length() >> 2;
        int right = s.length() - 1;
        int result = s.length();
        for (; right >= 0; right--) {
            
switch (s.charAt(right)) {
                case 'Q':
                    q++;
                    break;
                case 'W':
                    w++;
                    break;
                case 'E':
                    e++;
                    break;
                case 'R':
                    r++;
                    break;
            }
            if (q > balance || w > balance || e > balance || r > balance) {
                break;
            }
        }
        if (right == -1) {
            return 0;
        }
        result = Math.min(result, right + 1);
        
for (int left = 1; left < s.length(); left++) {
            switch (s.charAt(left - 1)) {
                case 'Q':
                    q++;
                    break;
                case 'W':
                    w++;
                    break;
                case 'E':
                    e++;
                    break;
                case 'R':
                    r++;
                    break;
            }
            if (q <= balance && w <= balance && e <= balance && r <= balance) {
                result = Math.min(result, right - left);
                continue;
            }
            for (; right < 
s.length(); right++) {
                switch (s.charAt(right)) {
                    case 'Q':
                        q--;
                        break;
                    case 'W':
                        w--;
                        break;
                    case 'E':
                        e--;
                        break;
                    case 'R':
                        r--;
                        break;
                }
                if (q <= balance && w <= balance && e <= balance && r <= balance) {
                    right++;
                    result = Math.min(result, right - left);
                    break;
                }
            }
        }
        return result;
    }

    public static void main(String[]args) {
        System.out.println(new Solution().balancedString("QQQW"));
    }
}
