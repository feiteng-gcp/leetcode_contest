class Solution {

    public int balancedString(String s) {
        Map<Character,Integer> map=new HashMap<>();
        map.put('Q',0);
        map.put('W',1);
        map.put('E',2);
        map.put('R',3);

        int result=Integer.MAX_VALUE;
        int[] count=new int[4];
        int low=0,high=-1;
        int length=s.length();
        int quater=length/4;
        
int[] totalCount=new int[4];
        for (char c : s.toCharArray()) {
            totalCount[map.get(c)]++;
        }

        while (low<length && high<length) {
            //先判断当前是否满足要求
            if (isValid(quater,totalCount,count)) {
                result = Math.min(result, high - low+1);
                if(high<low){
                    high++;
                    low++;
                }else {
                    count[map.get(s.charAt(high))]--;
                    high--;
                    if(!isValid(quater,totalCount,count)){
                        count[map.get(s.charAt(low))]--;
                        low++;
                    }
                }
            } else {
                high++;
                if(high<length) {
                    count[map.get(
s.charAt(high))]++;
                }
            }
        }

        return result;
    }

    boolean isValid(int quater,int[] totalCount,int[] count){
        boolean valid = true;
        
for (int i = 0; i < 4; i++) {
            if (totalCount[i] - count[i] > quater) {
                valid = false;
            }
        }
        return valid;
    }
