class Solution {
    public int balancedString(String s) {
        int[] idx = new int[256];
        idx['Q'] = 0;
        idx['W'] = 1;
        idx['E'] = 2;
        idx['R'] = 3;
        final int N = s.length();
        int[][] parts = new int[4][N];
        for (int i = 0; i < N; ++i) {
            char c = s.charAt(i);
            parts[idx[c]][i] = 1;
        }
        PrefixSum[] pfs = new PrefixSum[4];
        for (int i = 0; i < pfs.length; ++i) {
            pfs[i] = PrefixSum.fromArray(parts[i]);
        }
        
        int L = 0;
        int R = N;
        int good = N;
        while (L <= R) {
            int M = (L + R) >> 1;
            if (can(M, N, pfs)) {
                R = M - 1;
                good = M;
            } else {
                L = M + 1;
            }
        }
        return good;
    }
    
    private static boolean can(int size, int N, PrefixSum[] pfs) {
        int want = N / 4;
        for (int i = 0; i + size <= N; ++i) {
            boolean valid = true;
            // int pos = 0;
            for (PrefixSum pf : pfs) {
                // System.out.format("SIZE = %d, i = %d, pos = %d, sum = %d\n", size, i, pos++, pf.getRange(0, i) + pf.getRange(i + size, N));
                if (pf.getRange(0, i) + pf.getRange(i + size, N) > want) {
                    valid = false;
                    break;
                }
            }
            if (valid) {
                return true;
            }
        }
        return false;
    }
    
    private static class PrefixSum {
        public int[] sum;
        
        public int getRange(int L, int R) {
            return sum[R] - sum[L];
        }
        
        public static PrefixSum fromArray(int[] A) {
            int N = A.length;
            PrefixSum pf = new PrefixSum();
            pf.sum = new int[N + 1];
            for (int i = 0; i < N; ++i) {
                pf.sum[i + 1] = pf.sum[i] + A[i];
            }
            return pf;
        }
    }
}
