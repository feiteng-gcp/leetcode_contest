class Solution {
    public int balancedString(String s) {
		int Q = 0;
		int W = 0;
		int E = 0;
		int R = 0;
		for (int i = 0; i < s.length(); i++) {
			char c = s.charAt(i);
			if (c == 'Q') {
				Q++;
			}
			if (c == 'W') {
				W++;
			}
			if (c == 'E') {
				E++;
			}
			if (c == 'R') {
				R++;
			}
		}
		int avg = s.length() / 4;
		Q -= avg;
		W -= avg;
		E -= avg;
		R -= avg;
		Q = Math.max(Q, 0);
		W = Math.max(W, 0);
		E = Math.max(E, 0);
		R = Math.max(R, 0);
		if (Q == 0 && W == 0 && E == 0 && R == 0) {
			return 0;
		}

		int sta = 0;
		int q = 0;
		int w = 0;
		int e = 0;
		int r = 0;
		for (; sta < s.length(); sta++) {
			char c = s.charAt(sta);
			if (c == 'Q') {
				q++;
			}
			if (c == 'W') {
				w++;
			}
			if (c == 'E') {
				e++;
			}
			if (c == 'R') {
				r++;
			}
			if (allS(q, w, e, r, Q, W, E, R)) {
				break;
			}
		}
		// 双指针，开始，结束
		int end = 0;
		boolean jin = false;
		int min = sta;
		if (min == 0) {
			return 1;
		}
		// 因为一次只会更改一个数字，而当第一个1出现时，就会切换jin，所以，不用判断所有的数组
		w: while (true) {
			if (jin) {
				for (; sta < s.length(); sta++) {
					char c = s.charAt(sta);
					if (c == 'Q') {
						q++;
					}
					if (c == 'W') {
						w++;
					}
					if (c == 'E') {
						e++;
					}
					if (c == 'R') {
						r++;
					}
					if (allS(q, w, e, r, Q, W, E, R)) {
						jin = false;
						end++;
						continue w;
					}
				}
				break;
			} else {
				for (; end < sta; end++) {
					char c = s.charAt(end);
					if (c == 'Q') {
						q--;
					}
					if (c == 'W') {
						w--;
					}
					if (c == 'E') {
						e--;
					}
					if (c == 'R') {
						r--;
					}
					if (!allS(q, w, e, r, Q, W, E, R)) {
						// 长度更短，记录
						if (sta - end < min) {
							min = sta - end;
						}
						jin = true;
						sta++;
						continue w;
					}
				}
			}
		}
		return min + 1;
	}

	private boolean allS(int q, int w, int e, int r, int q2, int w2, int e2,
			int r2) {
		if (q < q2) {
			return false;
		}
		if (w < w2) {
			return false;
		}
		if (e < e2) {
			return false;
		}
		if (r < r2) {
			return false;
		}
		return true;
	}
}
