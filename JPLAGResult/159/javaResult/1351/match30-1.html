class Solution {
    public int balancedString(String s) {
        int[] count1 = new int[256];
        for (char c : s.toCharArray()) {
            count1[c]++;
        }
        count1['Q'] -= s.length() / 4;
        count1['W'] -= s.length() / 4;
        count1['E'] -= s.length() / 4;
        count1['R'] -= s.length() / 4;
        if (count1['Q'] == 0 && count1['W'] == 0 && count1['E'] == 0 && count1['R'] == 0) return 0;
        int[] count = new int[256];
        Map<String, Integer> map = new HashMap<>();
        map.put("0,0,0,0,0", -1);
        int n = s.length();
        int res = n;
        n /= 4;
        for (int i = 0, j = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            count[c]++;
            // System.out.println(i);
            if ((count1['Q'] <= 0 || count['Q'] >= count1['Q']) &&
                   (count1['W'] <= 0 || count['W'] >= count1['W']) &&
                   (count1['E'] <= 0 || count['E'] >= count1['E']) &&
                   (count1['R'] <= 0 || count['R'] >= count1['R'])) {
                while ((count1['Q'] <= 0 || count['Q'] >= count1['Q']) &&
                       (count1['W'] <= 0 || count['W'] >= count1['W']) &&
                       (count1['E'] <= 0 || count['E'] >= count1['E']) &&
                       (count1['R'] <= 0 || count['R'] >= count1['R'])) {
                    // System.out.printf("%d %d %d %d %d\n", count['Q'], count['W'], count['E'], count['R'], j);
                    // System.out.printf("%d %d %d %d\n", count1['Q'], count1['W'], count1['E'], count1['R']);
                    count[s.charAt(j++)]--;
                }
                 
res = Math.min(res, i - j + 2);
            }
        }
        return res;
    }
}
