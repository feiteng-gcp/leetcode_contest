import java.util.HashMap;
import java.util.Map;

class Solution {
    int qu = 0;

    public int balancedString(String s) {
        qu = s.length() / 4;
        Map<Character, Integer> map = cal(s);
        int minimum = s.length();
        int start = 0, end = 0;
        while (end < s.length()) {
            map.put(s.charAt(end), map.get(s.charAt(end)) - 1);
            end++;
            
            while (start < end) {
                char curr = s.charAt(start);
                if (map.get(curr) >= qu) {
                    break;
                }
                map.put(curr, map.get(curr) + 1);
                start++;
            }
            if (isOk(map)) {
                minimum = Math.min(minimum, end - start);
            }
        }

        return minimum;
    }

    public Map<Character, Integer> cal(String s) {
        Map<Character, Integer> map = new HashMap<>();
        for (char c : s.toCharArray()) {
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        return map;
    }

    public boolean isOk(Map<Character, Integer> map) {
        for (Character key : map.keySet()) {
            if (map.getOrDefault(key, 0) > qu) {
                return false;
            }
        }
        return true;
    }
}
