class Solution {
    public int balancedString(String s) {
        int[] count = new int[4];
        int[] loc = new int[26];
        loc['Q' - 'A'] = 0;
        loc['W' - 'A'] = 1;
        loc['E' - 'A'] = 2;
        loc['R' - 'A'] = 3;
        
        
for(char c : s.toCharArray()){
            count[loc[c - 'A']]++;
        }
        
        int n = s.length();
        int more = 0;
        int less = 0;
        for(int i = 0; i < 4; i++){
            count[i] -= n / 4;
            if(count[i] < 0){
                less += -count[i];
            }
        }
        
        if(less == 0){
            return 0;
        }
        // sliding window
        int l = 0;
        int res = n;
        
for(int r = 0; r < s.length(); r++){
            char c = s.charAt(r);
            if(count[loc[c -'A']] > 0){
                more++;
            }
            count[loc[c - 'A']]--;
            
            if(more == less){
                //System.out.println(r + " " + l);
                while(l <= r){
                    char lc = s.charAt(l);
                    count[loc[lc - 'A']]++;
                    l++;
                    if(count[loc[lc - 'A']] == 1){
                        more--;
                        break;
                    }
                }
                
res = Math.min(res, r - l + 2);
            }
        }
        
        return res;
    }
}
