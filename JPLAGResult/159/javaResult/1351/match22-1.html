class Solution {
    HashMap<Character, Integer> map = new HashMap<>();
    public int balancedString(String s) {
        map.put('Q', 0);
        map.put('W', 0);
        map.put('E', 0);
        map.put('R', 0);
        int expected = s.length() / 4;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        int result = 0;
        // for (char c : map.keySet()) {
        //     if (map.get(c) < expected) {
        //         result += expected - map.get(c);
        //     }
        // }
        
        int start = 0;
        int end = s.length();
        
        while (start < end) {
            int mid = (end - start) / 2 + start;
            
            if (isValid(mid, s, expected)) {
                end = mid;
            } else {
                start = mid + 1;
            
}
        }
        return start;
    }
    
    private boolean isValid(int size, String s, int pivot) {
        int[] data = new int[4];
        
        for (int i = 0; i < size; i++) 
{
            char c = s.charAt(i);
            
            if (c == 'Q') {
                data[0]++;
            }
            if (c == 'W') {
                data[1]++;
            }
            if (c == 'E') {
                data[2]++;
            }
            if (c == 'R') {
                data[3]++;
            }
        }
        
        if (map.get('Q') - data[0] <= pivot && map.get('W') - data[1] <= pivot && map.get('E') - data[2] <= pivot && map.get('R') - data[3] <= pivot) {
            return true;
        }
        
        for (int i = size; i < s.length(); i++) {
            char c = s.charAt(i);
            
char b = s.charAt(i - size);
            if (c == 'Q') {
                data[0]++;
            }
            if (c == 'W') {
                data[1]++;
            }
            if (c == 'E') {
                data[2]++;
            }
            if (c == 'R') {
                data[3]++;
            }
            
            if (b == 'Q') 
{
                data[0]--;
            }
            if (b == 'W') {
                data[1]--;
            }
            if (b == 'E') {
                data[2]--;
            }
            if (b == 'R') {
                data[3]--;
            }
            
            if (map.get('Q') - data[0] <= pivot && map.get('W') - data[1] <= pivot && map.get('E') - data[2] <= pivot && map.get('R') - data[3] <= pivot) {
            return true;
            }
        }
        
        return false;
        
    }
}
