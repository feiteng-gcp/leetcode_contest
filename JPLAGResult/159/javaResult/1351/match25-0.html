class Solution {
    public int balancedString(String s) {
        char[] chs = s.toCharArray();
        Map<Character, Integer> map = new HashMap<>();
        map.put('Q', 0);
        map.put('W', 0);
        map.put('E', 0);
        map.put('R', 0);
        
        for(int i = 0; i < chs.length; i++) {
            char c = chs[i];
            map.put(c, map.get(c) + 1);
        }
        
        int len = chs.length, each = len / 4;
        for(char k: map.keySet()) {
            map.put(k, map.get(k) - each);
        }
        
        if(map.get('Q') <= 0)   map.remove('Q');
        if(map.get('W') <= 0)   map.remove('W');
        if(map.get('E') <= 0)   map.remove('E');
        if(map.get('R') <= 0)   map.remove('R');
        if(map.size() == 0) return 0;
        
        Set<Character> keys = map.keySet();
        int total = 0;
        for(char k: map.keySet()) {
            total = total + map.get(k);
        }
        
        int start = 0, end = 0, res = Integer.MAX_VALUE;
        while(end < chs.length) {
            char c = chs[end];
            if(keys.contains(c)) {
                map.put(c, map.get(c) - 1);
                if(map.get(c) >= 0)
                    total--;
            }
            end++;
            while(total == 0) {
                res = Math.min(res, end - start);
                c = chs[start];
                if(keys.contains(c)) {
                    map.put(c, map.get(c) + 1);
                    if(map.get(c) > 0)
                        total++;
                }
                
                start++;
            }
        }
        
        return res;
    }
}
