class Solution {
    public int balancedString(String s) {
        int Q = 0, W = 0, E = 0, R = 0;
        for (char a : s.toCharArray()) {
            if (a == 'Q') {
                Q++;
            } else if (a == 'W') {
                W++;
            } else if (a == 'E') {
                E++;
            } else {
                R++;
            }
        }
        HashMap<Character,Integer> map = new HashMap<>();
        int N = s.length();
        int q = Q - N/4;
        int need = 0;
        if (q > 0) {
            map.put('Q', q); need += q;
        }
        int w =  W - N/4;
        if (w > 0) {
            map.put('W', w); need += w;
        }
        int e =  E - N/4;
        if (e > 0) {
            map.put('E', e); need += e;
        }
        int r =  R - N/4;
        if (r > 0) {
            map.put('R', r); need += r;
        }
        if (map.size() == 0) return 0;
        // substring to replace to make it balanced
        int minLen = minWindow(s, map);
        
        return minLen;
    }
    private int minWindow(String s, HashMap<Character, Integer> map) {
        int cntC = 0;
        for (Character c : map.keySet()) cntC += map.get(c);
        int left = 0;
        int minLeft = 0;
        int minLen = s.length()+1;
        int count = 0;
        for(int right = 0; right < s.length(); right++){
            if(map.containsKey(s.charAt(right))){
                map.put(s.charAt(right),map.get(s.charAt(right))-1);
                if(map.get(s.charAt(right)) >= 0){
                    count ++;
                }
                while(count == cntC){
                    if(right-left+1 < minLen){
                        minLeft = left;
                        minLen = right-left+1;
                    }
                    if(map.containsKey(s.charAt(left))){
                        map.put(s.charAt(left),map.get(s.charAt(left))+1);
                        if(map.get(s.charAt(left)) > 0){
                            count --;
                        }
                    }
                    left++ ;
                }
            }
        }
        return minLen;
    }
}
