class Solution {
    // Q:0, W: 1, E: 2, R: 3
    public int balancedString(String s) {
        int[] count = new int[4];
        for (char c : s.toCharArray()) {
            int val = map(c);
            count[val]++;
        }
        List<Integer> larger = new ArrayList<>();
        int base = s.length() / 4;
        for (int i = 0; i < count.length; i++) {
            if (count[i] > base) {
                larger.add(i);
            }
        }
        if (larger.size() == 0) {
            return 0;
        }
        int[] targetSub = new int[4];
        for (int i : larger) {
            targetSub[i] = count[i] - base;
        }
        
        int minLen = s.length();
        int left = 0;
        int right = 0;
        count = new int[4];
        while (right < s.length()) {
            count[map(s.charAt(right))]++;
            while (left <= right && satisfy(targetSub, count)) {
                minLen = 
Math.min(minLen, right - left + 1);
                count[map(s.charAt(left))]--;
                left++;
            }
            right++;
        }
        
        return minLen;
    }
    
    private int map(char c) {
        if (c == 'Q') {
            return 0;
        }
        if (c == 'W') {
            return 1;
        }
        if (c == 'E') {
            return 2;
        }
        if (c == 'R') {
            return 3;
        }
        return -1;
    } 
    
    private boolean satisfy(int[] base, int[] cur) {
        for (int i = 0; i < 4; i++) 
{
            if (cur[i] < base[i]) {
                return false;
            }
        }
        return true;
    }
}
