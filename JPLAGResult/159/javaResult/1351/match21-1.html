class Solution {
    public int balancedString(String s) {
        int[] counts = new int[4];
        
        for(char ch : s.toCharArray()) {
            counts[getIdx(ch)]++;
        }
        
        int avg = s.length() / 4;
            //System.out.println(Arrays.toString(counts));

        int res = 0;
        for(int i = 0; i < counts.length; i++) {
            if(counts[i] <= avg) counts[i] = 0;
            else counts[i] -= avg;
            res += counts[i];
        }
    
    
        if(res == 0) return res;
    
        res = s.length();
    
        int left = 0, right = 0;
        int[] tmp = new int[4];
        while(right < 
s.length()) {
            while(right < s.length() && !exceed1(tmp, counts)) {
                tmp[getIdx(s.charAt(right++))]++;
            }
            
            while(left < right && exceed1(tmp, counts)) {
                res = Math.min(res, right - left);
                tmp[getIdx(s.charAt(left++))]--;
            }
        }    
    
        
        return res;
    }

    private boolean exceed1(int[] tmp, int[] counts) {
        for(int i = 0; i < 4; i++) {
            if(counts[i] == 0) continue;
            if(counts[i] > tmp[i]) return false;
        
}
        return true;
    }

    private int getIdx(char ch) {
        if(ch == 'Q') return 0;
        if(ch == 'W') return 1;
        if(ch == 'E') return 2;
        return 3;
    }
}
