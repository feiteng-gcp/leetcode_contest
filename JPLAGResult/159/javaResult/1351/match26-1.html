class Solution {
    public int balancedString(String s) {
        int let[] = new int[] {s.length()/-4, s.length()/-4, s.length()/-4, s.length()/-4};
        for(char c : s.toCharArray()) {
            change(c, let, 1);
        }
        //System.out.println(let[0] + " " + let[1] + " " + let[2] + " " + let[3]);
        if(let[0] == 0 && let[1] == 0 && let[2] == 0 && let[3] == 0)
            return 0;
        return getMin(s, let);
    }
    
    public int getMin(String s, int[] let) {

        int[] sub = new int[4];
        
int max = s.length();
        int end = 0;
        int start = 0;
        while(end <= s.length()) {
            if(check(let, sub)) {
                //System.out.println(s.substring(start,end));
                max = Integer.min(max, end-start);
                change(s.charAt(start), sub, -1);
                start++;
            }
            else {
                if(end < s.length())
                    change(s.charAt(end), sub, 1);
                end++;
            }
        }
        return max;
    }
    
    public boolean check(int[] let, int[] sub) {
        return (sub[0] >= let[0] && sub[1] >= let[1] && sub[2] >= let[2] && sub[3] >= let[3]);
    }
    
    public void change(char c, int[] arr, int val) {
        switch(c) {
                case 'Q': arr[0]+=val;
                    break;
                case 'W': arr[1]+=val;
                    break;
                case 'E': arr[2]+=val;
                    break;
                default: arr[3]+=val;
                    break;
            }
    }
}
