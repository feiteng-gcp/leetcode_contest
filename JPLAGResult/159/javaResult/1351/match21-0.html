class Solution {
    public int balancedString(String s) {
        int len = s.length();
        int target = len / 4;
        char[] ch = new char[]{'Q', 'W', 'E', 'R'};
        int[] diff = new int[4];
        Map<Character, Integer> map = new HashMap<>();
        for(char c : s.toCharArray()){
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        for(int i = 0;i < 4;i++){
            diff[i] = map.getOrDefault(ch[i], 0) - target;
        }
        int[] cur = new int[4];
        int left = 0,res = Integer.MAX_VALUE;
        for(int i = 0;i < len;i++){
            char c = s.charAt(i);
            if(c == 'Q') cur[0]++;
            if(c == 'W') cur[1]++;
            if(c == 'E') cur[2]++;
            if(c == 'R') cur[3]++;
            if(check(diff, cur)){
                res = Math.min(i - left + 1, res);
                while(left <= i && check(diff, cur)){
                    char pre = s.charAt(left++);
                    if(pre == 'Q') cur[0]--;
                    if(pre == 'W') cur[1]--;
                    if(pre == 'E') cur[2]--;
                    if(pre == 'R') cur[3]--;
                    if(check(diff, cur)){
                        res = Math.min(i - left + 1, res);
                    }
                }
            }
        }
        return res;
    }
    public boolean check(int[] diff, int[] cur){
        for(int i = 0;i < 4;i++){
            if(cur[i] < diff[i]) return false;
        }
        return true;
    }
}
