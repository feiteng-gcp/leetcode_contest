class Solution {
    public int balancedString(String s) {
        // find the longest substring contains all characters in rm with the right freq
        int[] map = new int[4];
        for (int i = 0; i < s.length(); i++) {
            map[idx(s.charAt(i))] += 1;
        } 
        int k = s.length() / 4;
        int[] rm = new int[4];
        for (int i = 0; i < 4; i++) {
            if (map[i] > k) {
                rm[i] = map[i] - k;
            }
        }
        int j = 0;
        int min = s.length();
        for (int i = 0; i < s.length(); i++) {
            while (j < s.length() && !allIn(rm)) {
                rm[idx(s.charAt(j))] -= 1;
                j++;
            }
            if (allIn(rm)) min = Math.min(min, j - i);
            rm[idx(s.charAt(i))] += 1;
        }
        return min;
    }
    public int idx(char c) {
        if (c == 'Q') return 0;
        if (c == 'W') return 1;
        if (c == 'E') return 2;
        if (c == 'R') return 3;
        return -1;
    }
    public boolean allIn(int[] rm) {
        for (int i = 0; i < 4; i++ ) {
            if (rm[i] > 0) return false;
        }
        return true;
    }
}
