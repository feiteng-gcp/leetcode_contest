class Solution {
    public int balancedString(String s) {
        int n = s.length();
        int ans = n;
        int q = 0;
        int w = 0;
        int e = 0;
        int r = 0;
        int i = 0;
        while (i<n && q <= n/4 && w <= n/4 && e <= n/4 && r <= n/4) {
            
switch (s.charAt(i)) {
                case 'Q':q++;break;
                case 'W':e++;break;
                case 'E':w++;break;
                case 'R':r++;break;
            }
            i++;
        }
        if (i >= n)
            return 0;
        i--;
        switch (s.charAt(i)) {
            case 'Q':q--;break;
            case 'W':e--;break;
            case 'E':w--;break;
            case 'R':r--;break;
        }
        i--;
        ans = n-i;
        int j = n-1;
        while (i >= 0) {
            while (i<j && q <= n/4 && w <= n/4 && e <= n/4 && r <= n/4) 
{
                switch (s.charAt(j)) {
                    case 'Q':q++;break;
                    case 'W':e++;break;
                    case 'E':w++;break;
                    case 'R':r++;break;
                }
                j--;
            }
            ans = Math.min(ans, j+1-i);
            while (i>=0 && !(q <= n/4 && w <= n/4 && e <= n/4 && r <= n/4)) {
                switch (s.charAt(i)) {
                    
case 'Q':q--;break;
                    case 'W':e--;break;
                    case 'E':w--;break;
                    case 'R':r--;break;
                }
                i--;
            }
            ans = Math.min(ans, j-i);
        }
        while (i<j && q <= n/4 && w <= n/4 && e <= n/4 && r <= n/4) {
            switch (s.charAt(j)) {
                case 'Q':q++;break;
                case 'W':e++;break;
                case 'E':w++;break;
                case 'R':r++;break;
            }
            j--;
        }
        return Math.min(ans, j+1-i);
    }
}
