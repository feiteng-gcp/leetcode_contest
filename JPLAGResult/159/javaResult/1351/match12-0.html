class Solution {
    public int balancedString(String s) {
        int[] count = new int[4];
        for(int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if(c == 'Q') count[0] += 1;
            if(c == 'W') count[1] += 1;
            if(c == 'E') count[2] += 1;
            if(c == 'R') count[3] += 1;
        }
        int avg = s.length() / 4;
        if(count[0] == avg && count[1] == avg && count[2] == avg && count[3] == avg) return 0;
        int[] red = new int[4];
        for(int i = 0; i < 4; i++) {
            if(count[i] > avg) red[i] = count[i] - avg;
            // System.out.println(red[i]);
        }
        int left = 0;
        int right = 0;
        int res = Integer.MAX_VALUE;
        int[] cur = red.clone();
        if(s.charAt(0) == 'Q' && red[0] > 0) cur[0] -= 1;
        if(s.charAt(0) == 'W' && red[1] > 0) cur[1] -= 1;
        if(s.charAt(0) == 'E' && red[2] > 0) cur[2] -= 1;
        if(s.charAt(0) == 'R' && red[3] > 0) cur[3] -= 1;
        while(left < s.length() && right < s.length()) {
            // System.out.println(left);
            // System.out.println(right);
            // System.out.println("---");
            if(cur[0] <= 0 && cur[1] <= 0 && cur[2] <= 0 && cur[3] <= 0) {
                res = Math.min(res, right - left + 1);
                char c = s.charAt(left);
                if(c == 'Q' && red[0] > 0) cur[0] += 1;
                if(c == 'W' && red[1] > 0) cur[1] += 1;
                if(c == 'E' && red[2] > 0) cur[2] += 1;
                if(c == 'R' && red[3] > 0) cur[3] += 1;
                left += 1;
            } else {
                right += 1;
                if(right >= s.length()) break;
                char c = s.charAt(right);
                if(c == 'Q' && red[0] > 0) cur[0] -= 1;
                if(c == 'W' && red[1] > 0) cur[1] -= 1;
                if(c == 'E' && red[2] > 0) cur[2] -= 1;
                if(c == 'R' && red[3] > 0) cur[3] -= 1;                
            }
        }
        return res;
        
    }
}
