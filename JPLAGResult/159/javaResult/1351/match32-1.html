class Solution {
    public int balancedString(String s) {
        int qc = 0;
        int wc = 0;
        int ec = 0;
        int rc = 0;
        int n = s.length();
        s = s.toLowerCase();
        
for (char c : s.toCharArray()) {
            if (c=='q')
                qc++;
            if (c=='w')
                wc++;
            if (c=='e')
                ec++;
            if (c=='r')
                rc++;
        }
        if (qc==wc&&qc==ec&&qc==rc)
            return 0;
        int ml = n;
        int qr = 0;
        int wr = 0;
        int er = 0;
        int rr = 0;
        int a = 0;
        int b = 0;
        if (s.charAt(0)=='q')
            qr++;
        if (s.charAt(0)=='w')
            wr++;
        if (s.charAt(0)=='e')
            er++;
        if (s.charAt(0)=='r')
            rr++;
        while (b < n) {
            int qd = qc-qr;
            int wd = wc-wr;
            int ed = ec-er;
            int rd = rc-rr;
            if (qd<=n/4&&wd<=n/4&&ed<=n/4&&rd<=n/4) {
                ml = Math.min(ml, b-a+1);
                if (s.charAt(a)=='q')
                    qr--;
                if (s.charAt(a)=='w')
                    wr--;
                if (s.charAt(a)=='e')
                    er--;
                if (s.charAt(a)=='r')
                    rr--;
                ++a;
            } else {
                ++b;
                if (b==n)
                    break;
                if (s.charAt(b)=='q')
                    qr++;
                if (s.charAt(b)=='w')
                    wr++;
                if (s.charAt(b)=='e')
                    er++;
                if (s.charAt(b)=='r')
                    rr++;
            }
        }
        while (a < n) {
            int qd = qc-qr;
            int wd = wc-wr;
            int ed = ec-er;
            int rd = rc-rr;
            if (qd<=n/4&&wd<=n/4&&ed<=n/4&&rd<=n/4) {
                ml = Math.min(ml, b-a+1);
                if (s.charAt(a)=='q')
                    qr--;
                if (s.charAt(a)=='w')
                    wr--;
                if (s.charAt(a)=='e')
                    er--;
                if (s.charAt(a)=='r')
                    rr--;
                ++a;
            } else {
                break;
            }
        }
        return ml;
    }
}
