class Solution {
    public int balancedString(String s) {
        int n = s.length(), x = n/4;
        int[] nums = new int[4];
        char[] chars = new char[]{'Q','W','E','R'};
        for(char c:s.toCharArray()){
            for(int i=0;i<4;i++){
                if(c==chars[i]){
                    nums[i]++;
                }
            }
        }
        Map<Character,Integer> m = new HashMap<>();
        for(int i=0;i<4;i++){
            if(nums[i]>x){
                m.put(chars[i],nums[i]-x);
            }
        }
        int start = 0, end = 0, res = n, l = m.size();
        if(l==0) return 0;
        while(end<n){
            char c = s.charAt(end++);
            
if(m.containsKey(c)){
                m.put(c,m.get(c)-1);
                if(m.get(c)==0) l--;
            }
            while(l==0){
                res = Math.min(res, end-start);
                char t = s.charAt(start++);
                
if(m.containsKey(t)){
                    m.put(t,m.get(t)+1);
                    if(m.get(t)==1) l++;
                }
            }
        }
        return res;
    }
}
