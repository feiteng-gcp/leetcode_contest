class Solution {
    public int balancedString(String s) {
        return sln1(s);
    }
    
    //sliding window
    private int sln1(String s){
        char[] ss = s.toCharArray();
        int n = ss.length;
        int[] map  = new int[4];
        for(int i = 0;i < ss.length;i ++){
            char c= ss[i];
            if(c == 'Q') map[0]++;
            else if(c == 'E') map[1]++;
            else if(c == 'W') map[2]++;
            else if(c == 'R') map[3]++;
        }
        for(int i = 0;i < map.length; i++){
            if(map[i] <= n/4) map[i] =0;
            else map[i] -= n/4;
        }
        int[] map2  = new int[4];
        // for(int m: map) System.out.print(m+" ");
        // System.out.println();
        if(found(map, map2)) return 0;
        int min = Integer.MAX_VALUE;
        for(int p1 =0, p2 =0; p2 < ss.length; p2++){
            addCount(ss[p2], map, map2);
            // for(int m: map2) System.out.print(m+" ");
            // System.out.println();
            while(found(map, map2)){
                min = Math.min(min, p2-p1+1);
                
                removeCount(ss[p1++], map, map2);
            }
        }
        return min;
    }
    
    private void addCount(char c, int[] map, int[] map2){
        if(c == 'Q' && map[0] > 0) map2[0]++;
        else if(c == 'E' && map[1] > 0) map2[1]++;
        else if(c == 'W' && map[2] > 0) map2[2]++;
        else if(c == 'R' && map[3] > 0) map2[3]++;
    }
    
    private void removeCount(char c, int[] map, int[] map2){
        if(c == 'Q' && map[0] > 0) map2[0]--;
        else if(c == 'E' && map[1] > 0) map2[1]--;
        else if(c == 'W' && map[2] > 0) map2[2]--;
        else if(c == 'R' && map[3] > 0) map2[3]--;
    }
    
    private boolean found( int[] map, int[] map2){
        for(int i = 0;i < map.length; i++){
            if(map[i] > map2[i]) return false;
        }
        return true;
    }
}
