class Solution {
    public int balancedString(String s) {
        int n = s.length();
        int[][] dp = new int[n+1][4];
        int target = n/4;
        for(int i = 1; i <= n; i++) {
            char c = s.charAt(i-1);
            dp[i] = new int[]{dp[i-1][0], dp[i-1][1], dp[i-1][2], dp[i-1][3]};
            if (c == 'Q') dp[i][0]++;
            else if (c == 'W') dp[i][1]++;
            else if (c == 'E') dp[i][2]++;
            else dp[i][3]++;
        }
        
        if (dp[n][0] == target && dp[n][1] == target && dp[n][2] == target && dp[n][3] == target) return 0;
        int res = n;
        for(int r = 1; r <= n; r++) {
            int Q = dp[n][0] - dp[r][0];
            int W = dp[n][1] - dp[r][1];
            int E = dp[n][2] - dp[r][2];
            int R = dp[n][3] - dp[r][3];
            int l = Math.max(0, r - res);
            Q += dp[l][0];
            W += dp[l][1];
            E += dp[l][2];
            R += dp[l][3];
            l++;
            for(; l <= r; l++) {
                if (Q > target || W > target || E > target || R > target) break;
                res = Math.min(res, r - l + 1);
                Q += dp[l][0] - dp[l-1][0];
                W += dp[l][1] - dp[l-1][1];
                E += dp[l][2] - dp[l-1][2];
                R += dp[l][3] - dp[l-1][3];
                
            }
        }
        return res;
    }
}
