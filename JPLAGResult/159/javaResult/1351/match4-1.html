class Solution {
    public int balancedString(String s) {
        int n = s.length();
        int q=n/4,w=n/4,e=n/4,r=n/4;
        for(int i=0; i<n; i++)
{
            char ch = s.charAt(i);
            if('Q'==ch) q--;
            else if('W'==ch) w--;
            else if('E'==ch) e--;
            else if('R'==ch) r--;
        }
        int res = Integer.MAX_VALUE;
        for(int i=0, j=0; j<n; j++)
{
            char ch = s.charAt(j);
            if('Q'==ch) q++;
            else if('W'==ch) w++;
            else if('E'==ch) e++;
            else if('R'==ch) r++;
            while(i<=j&&q>=0&&w>=0&&e>=0&&r>=0) 
{
                res = Math.min(res, j-i+1);
                ch = s.charAt(i);
                if('Q'==ch) q--;
                else if('W'==ch) w--;
                else if('E'==ch) e--;
                else if('R'==ch) r--;
                i++;
            }
            if(q>=0&&w>=0&&e>=0&&r>=0) res = Math.min(res, j-i+1);
        }
        return res;
    }
}
