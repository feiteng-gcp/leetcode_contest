	class S
olution {
	    public int balancedString(String s_) {
	    	char[] s = s_.toCharArray();
	    	int n = s.length;
	    	
	    	int[][] pre = new int[4][n+1];
	    	for(int i = 0;i < n;i++){
	    		for(int j = 0;j < 4;j++){
	    			pre[j][i+1] = pre[j][i];
	    		}
	    		pre["QWER".indexOf(s[i])][i+1]++;
	    	}
	    	
	    	int[][] suf = new int[4][n+1];
	    	for(int i = 0;i < n;i++){
	    		for(int j = 0;j < 4;j++){
	    			suf[j][i+1] = suf[j][i];
	    		}
	    		suf["QWER".indexOf(s[n-1-i])][i+1]++;
	    	}
	    	
	    	int p = 0;
	    	int ret = n;
	    	for(int i = 0;i <= n;i++){
	    		while(p <= i){
	    			boolean ok = true;

	    			for(int k = 0;k < 4;k++){
	    				if(pre[k][p] + suf[k][n-i] <= n/4){
	    				}else{
	    					ok = false;
	    				}
	    			}
	    			if(ok){
	    				p++;
	    			}else{
	    				break;
	    			}
	    		}

	    		if(p > 0){
	    			ret = Math.min(ret, i-p+1);
	    		}
	    	}
	    	return ret;
	    }
	}
