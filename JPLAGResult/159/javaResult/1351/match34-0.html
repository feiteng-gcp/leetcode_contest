class Solution {
    public int balancedString(String s) {
        int len = s.length();
        int times = len / 4;
        Map<Character, Integer> need = new HashMap(), now = new HashMap();
        for (char c : s.toCharArray()) {
            now.put(c, now.getOrDefault(c, 0) + 1);
        }
        for (char key : now.keySet()) {
            if (now.get(key) > times)
                need.put(key, now.get(key) - times);
        }
        if (need.size() == 0)
            return 0;
        now = new HashMap<>(need);
        int cnt = now.size();
        int res = s.length();
        int j = 0;
        //找包含了need最短的substring
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (now.containsKey(c)) {
                now.put(c, now.getOrDefault(c, 0) - 1);
                if (now.get(c) == 0)
                    cnt--;
            }
            while (cnt == 0) {
                res = Math.min(res, i - j + 1);
                c = s.charAt(j++);
                if (now.containsKey(c)) {
                    if (now.get(c) < need.get(c)) {
                        now.put(c, now.getOrDefault(c, 0) + 1);
                        if (now.get(c) > 0)
                            cnt++;
                    }
                }
            }
            
        }
        return res;
    }
}
