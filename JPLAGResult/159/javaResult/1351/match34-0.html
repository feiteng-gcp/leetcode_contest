class Solution {

	public int balancedString(String s) {
		int[] count = new int[26];
		for (char c : s.toCharArray()) {
			count[c - 'A']++;
		}
		int[] count2 = new int[26];
		int min = s.length();
		int j = 0;
		for (int i = 0; i <= s.length(); i++) {
			while (count2['Q' - 'A'] >= Math.max(0, count['Q' - 'A'] - s.length() / 4)
					&& count2['W' - 'A'] >= Math.max(0, count['W' - 'A'] - s.length() / 4)
					&& count2['E' - 'A'] >= Math.max(0, count['E' - 'A'] - s.length() / 4)
					&& count2['R' - 'A'] >= Math.max(0, count['R' - 'A'] - s.length() / 4)) {
				min = Math.min(min, i - j);
				if (j < s.length()) {
					count2[s.charAt(j++) - 'A']--;
				} else {
					break;
				}
			}
			if (i < s.length()) {
				count2[s.charAt(i) - 'A']++;
			}

		}
		return min;
	}
}
