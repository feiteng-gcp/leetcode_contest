class Solution {
    public int balancedString(String s) {
        
Map<Character, Integer> map = new HashMap<>();
        for (char c : s.toCharArray()) {
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        // System.out.println(map);
        int b = s.length() / 4;
        int res = 0;
        String tmp = "QWER";
        Map<Character, Integer> tmap = new HashMap<>();
        for (char c : tmp.toCharArray()) {
            if (map.containsKey(c)) {
                int aa = map.get(c);
                if (aa > b) {
                    tmap.put(c, aa - b);
                }
            }
        }
       
        
int cnt = tmap.size(), left = 0, right = s.length();
        int i = 0;
        for (int j = 0; j < s.length(); j++) {
            char c = s.charAt(j);
            if (tmap.containsKey(c)) {
                tmap.put(c, tmap.get(c) - 1);
                if (tmap.get(c) == 0) {
                    cnt--;
                }
                while (cnt == 0) {
                    if (right - left > j - i) {
                        right = j;
                        left = i;
                    }
                    char cc = s.charAt(i);
                    if (
tmap.containsKey(cc)) {
                        tmap.put(cc, tmap.get(cc) + 1);
                        if (tmap.get(cc) > 0) cnt++;
                    }
                    i++;
                }
               
            }
        }
        // System.out.println(left);
        // System.out.println(right);
        return right == s.length() ? 0 : right - left + 1;
    }
    
    public String minWindow(String s, String t) {
        Map<Character, Integer> tmap = new HashMap<>();
        for (char c : t.toCharArray()) {
            tmap.put(c, tmap.getOrDefault(c, 0) + 1);
        }
        
        // Map<Character, Integer> smap = new HashMap<>();
        int cnt = tmap.size(), left = 0, right = s.length();
        int i = 0;
        for (int j = 0; j < s.length(); j++) {
            char c = s.charAt(j);
            if (tmap.containsKey(c)) {
                tmap.put(c, tmap.get(c) - 1);
                if (tmap.get(c) == 0) {
                    cnt--;
                }
                while (cnt == 0) {
                    if (right - left > j - i) {
                        right = j;
                        left = i;
                    }
                    char tmp = s.charAt(i);
                    if (tmap.containsKey(tmp)) {
                        tmap.put(tmp, tmap.get(tmp) + 1);
                        if (tmap.get(tmp) > 0) cnt++;
                    }
                    i++;
                }
               
            }
             
        }
        
        return right == s.length() ? "" : s.substring(left, right+1);
    }
}
