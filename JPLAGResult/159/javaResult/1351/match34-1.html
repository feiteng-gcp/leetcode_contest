class Solution {
    public int balancedString(String s) {
        int d[] = new int[256];
        for(char c:s.toCharArray()){
            d[c]++;
        }
        int len = s.length()/4;int r =0;
      char cc[] = {'Q','W','E','R'};
        boolean all = false;
        for(char f:cc){
            if(d[f]>len){
               d[f]-=len;
                all = true;
            }else{
               d[f] = 0; 
            }
        }
        if(!all) return 0;
        
        
        int st = 0;int md[] = new int[256];int rrr = s.length();
        for(int i=0;i<s.length();++i){
            char ccc = s.charAt(i);
            if(d[ccc]>0){
                md[ccc]++;
            }
            while(true){
            ccc = s.charAt(st);
            
            boolean ok = d[ccc]>0&& md[ccc]-1>=d[ccc] ||d[ccc]==0;

            if(ok){
                if(d[ccc]>0){
                md[ccc]--;
                }
                st++;
            }else{
                break;
            }
            }
            
            boolean ok1 = true;
            for(char t:cc){
                if(d[t]>0){
                if(md[t]<d[t]){
                    ok1 = false;
                    break;
                }    
                }
            }
            if(ok1){
            rrr = 
Math.min(i-st+1,rrr);
            }
            
        }
        return rrr;
        
        
    }
}
