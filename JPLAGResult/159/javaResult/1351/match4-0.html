class Solution {
    public int balancedString(String s) {
    	int Qcount = 0;
    	int Wcount = 0;
    	int Ecount = 0;
    	int Rcount = 0;
    	int target = s.length() / 4;
    	for(int i = 0; i < s.length(); i++) {
    		char c = s.charAt(i);
    		if(c == 'Q') Qcount++;
    		else if(c == 'W') Wcount++;
    		else if(c == 'E') Ecount++;
    		else if(c == 'R') Rcount++;
    	}
    	Qcount -= target;
    	Wcount -= target;
    	Ecount -= target;
    	Rcount -= target;
    	
    	int Qs = 0;
    	int Ws = 0;
    	int Es = 0;
    	int Rs = 0;
    	
    	int sidx = 0;
    	int ans = Integer.MAX_VALUE;
    	for(int i = 0; i < s.length(); i++) {
    		char c = s.charAt(i);
    		if(c == 'Q') Qs++;
    		else if(c == 'W') Ws++;
    		else if(c == 'E') Es++;
    		else if(c == 'R') Rs++;
    		
    		while(sidx < s.length() && Qs >= Qcount && Ws >= Wcount && Es >= Ecount && Rs >= Rcount) {
    			ans = Math.min(ans, (i - sidx + 1));
    			c = s.charAt(sidx);
    			sidx++;
    			if(c == 'Q') Qs--;
        		else if(c == 'W') Ws--;
        		else if(c == 'E') Es--;
        		else if(c == 'R') Rs--;
    		}
    		
    	}
    	
    	return ans;
    	
    }
}
