class Solution {
    public int balancedString(String s) {
        List<Integer> indexes = Arrays.asList((int) 'Q', (int) 'W', (int) 'E', (int) 'R');
        int[] totalCount = new int[256];
        for (int i = 0; i < s.length(); i++) {
            totalCount[s.charAt(i)] += 1;
        }
        int[] difference = new int[256];
        for (int i : indexes) {
            difference[i] = totalCount[i] - s.length() / 4;
        }
        int[] count = new int[256];
        int start = 0;
        int end = 0;
        int output = s.length();
        while (true) {
            boolean isValid = true;
            if (start > end) {
                isValid = false;
            }
            for (int i : indexes) {
                if (difference[i] > 0 && difference[i] > count[i]) {
                    isValid = false;
                }
            }
            if (isValid) {
                output = Math.min(output, end - start);
                if (start == s.length()) {
                    break;
                } else {
                    count[s.charAt(start)] -= 1;
                    start += 1;
                }
            } else {
                if (end == s.length()) {
                    break;
                } else {
                    count[s.charAt(end)] += 1;
                    end += 1;
                }
            }
        }
        return output;
    }
}
