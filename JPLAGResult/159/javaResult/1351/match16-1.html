class Solution {
    public int balancedString(String s) {
        int[][] prefixs = new int[s.length() + 1][4];
        int[] prefix = new int[4];
        for (int i = 0; i < s.length(); i++) {
            prefix["QWER".indexOf(s.charAt(i))]++;
            for (int j = 0; j < 4; j++) {
                prefixs[i + 1][j] = prefix[j];
            }
        
}
        int[] delta = new int[4];
        for (int j = 0; j < 4; j++) {
            delta[j] = prefix[j] - s.length() / 4;
        }
        
        int left = 0, right = s.length();
        while (left + 1 < right) {
            int size = (left + right) / 2;
            if (check(prefixs, delta, size)) {
                
right = size;
            } else {
                left = size;
            }
        }
        if (check(prefixs, delta, left)) {
            return left;
        }
        return right;
    }
    
    private boolean check(int[][] prefixs, int[] delta, int size) {
        for (int i = 0; i + size < prefixs.length; i++) {
            int[] diff = getDiff(prefixs[i + size], prefixs[i]);
            boolean found = true;
            for (int j = 0; j < 4; j++) {
                if (delta[j] > diff[j]) 
{
                    found = false;
                    break;
                }
            }
            if (found) {
                return true;
            }
        }
        return false;
    }
    
    private int[] getDiff(int[] p1, int[] p2) {
        int[] diff = 
new int[4];
        for (int j = 0; j < 4; j++) {
            diff[j] = p1[j] - p2[j];
        }
        return diff;
    }
}
