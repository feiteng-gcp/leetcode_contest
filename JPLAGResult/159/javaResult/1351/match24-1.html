class Solution {
    public int balancedString(String s) {
        int n = s.length();
        int[] times = new int[4];
        Arrays.fill(times, n/4);
        for(int i=0; i<n; i++){
            char ch = s.charAt(i);
            add(times, ch, -1);
        }
        int res = Integer.MAX_VALUE;
        for(int i=0, j=0; j<n; j++){
            char ch = s.charAt(j);
            add(times, ch, 1);
            while(i<=j&&yes(times)) {
                res = Math.min(res, j-i+1);
                ch = s.charAt(i);
                add(times, ch, -1);
                i++;
            }
            if(yes(times)) res = Math.min(res, j-i+1);
        
}
        return res;
    }
    private void add(int[] times, char ch, int gap){
        if('Q'==ch) times[0]+=gap;
        else if('W'==ch) times[1]+=gap;
        else if('E'==ch) times[2]+=gap;
        else if('R'==ch) times[3]+=gap;
    }
    private boolean yes(int[] times)
{
        for(int time : times){
            if(time<0) return false;
        }
        return true;
    }
}
