class Solution {
    public int balancedString(String s) {
        int len = s.length();
        int count = len / 4;
        int[] alp = new int[26];
        for (char c : s.toCharArray()) {
            alp[c - 'A']++;
        }
        int[] index = {16, 22, 4, 17};
        int[] need = new int[4];
        boolean flag = false;
        for (int i = 0; i < 4; i++) {
            int tmp = alp[index[i]] - count;
            if (tmp > 0) {
                flag = true;
                need[i] = tmp;
            }
        }
        if (!flag) {
            return 0;
        }
        int i = 0;
        int j = 0;
        int result = Integer.MAX_VALUE;
        int[] res = new int[26];
        while (j < len) {
            res[s.charAt(j) - 'A']++;
            while (check(res, need) && i <= j) {
                result = 
Math.min(result, j - i + 1);
                res[s.charAt(i) - 'A']--;
                i++;
            }
            j++;
        }

        return result;
    }

    private boolean check(int[] res, int[] need) {
        int[] index = {16, 22, 4, 17};
        for (int k = 0; k < 4; k++) 
{
            if (res[index[k]] < need[k]) {
                return false;
            }
        }
        return true;
    }
}
