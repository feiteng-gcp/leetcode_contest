class Solution {
    public int balancedString(String s) {
        int n = s.length();
        int[] a = new int[4];
        char[] ch = s.toCharArray();
        int res = n-1;
        for (char c: ch) {
            
if (c=='Q') a[0]++;
            if (c=='W') a[1]++;
            if (c=='E') a[2]++;
            if (c=='R') a[3]++;
        }
        boolean v = true;
        for (int i=0;i<4; i++) {
            a[i]-= (n/4);
            if (a[i] != 0) v = false;
        }
        if (v) return 0;

        for (int i = 0, j = 0; j<n; j++) {
            char c = ch[j];
            
if (c=='Q') a[0]--;
            if (c=='W') a[1]--;
            if (c=='E') a[2]--;
            if (c=='R') a[3]--;
            while (valid(a)) {
                res = Math.min(res, j - i+1);
                char p = ch[i];
                
if (p=='Q') a[0]++;
                if (p=='W') a[1]++;
                if (p=='E') a[2]++;
                if (p=='R') a[3]++;
                i++;
            }
        }
        return res;
    }

    boolean valid(int[] a) {
        boolean res = true;
        for (int i=0; i<4; i++) {
            if (a[i] > 0) res = false;
        }
        return res;
    }
}
