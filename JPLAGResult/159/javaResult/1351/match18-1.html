class Solution {
    public int balancedString(String s) {
        int q = 0, w = 0, e = 0, r = 0;
        for (int i = 0; i < s.length(); i++) {
            switch (s.charAt(i)) {
                
case 'Q':
                    q++;
                    break;
                case 'W':
                    w++;
                    break;
                case 'E':
                    e++;
                    break;
                case 'R':
                    r++;
                    break;

            }
        }
        int length = s.length() / 4;
        int extraQ = Math.max(0, q - length);
        int extraW = Math.max(0, w - length);
        int extraE = Math.max(0, e - length);
        int extraR = Math.max(0, r - length);
        int start = 0, end = -1;
        int qCount = 0, wCount = 0, eCount = 0, rCount = 0;
        int min = s.length();
        while (start < s.length()) {
            if (qCount >= extraQ && wCount >= extraW && eCount >= extraE && rCount >= extraR) 
{
                min = Math.min(min, end - start + 1);
                switch (s.charAt(start)) {
                    case 'Q':
                        qCount--;
                        break;
                    case 'W':
                        wCount--;
                        break;
                    case 'E':
                        eCount--;
                        break;
                    case 'R':
                        rCount--;
                        break;
                }
                start++;
            } else if (end < s.length() - 1) 
{
                end++;
                switch (s.charAt(end)) {
                    case 'Q':
                        qCount++;
                        break;
                    case 'W':
                        wCount++;
                        break;
                    case 'E':
                        eCount++;
                        break;
                    case 'R':
                        rCount++;
                        break;

                }
            } else {
                break;
            }
        }
        return min;
    }
}
