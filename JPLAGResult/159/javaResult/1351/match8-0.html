class Solution {
    public int balancedString(String s) {
        int k = s.length() / 4;
        String alphabets = "QWER";
        int[] counts = new int[4];
        for (int i = 0; i < s.length(); i++) {
            counts[alphabets.indexOf(s.charAt(i))]++;
        }
        int target = 0;
        for (int i = 0; i < 4; i++) {
            if (counts[i] > k) {
                counts[i] -= k;
                target += counts[i];
            } else {
                counts[i] = 0;
            }
        }
        if (target == 0) {
            return 0;
        }
        int minLen = Integer.MAX_VALUE;
        int j = 0;
        int seen = 0;
        for (int i = 0; i < s.length(); i++) {
            if (--counts[alphabets.indexOf(s.charAt(i))] >= 0) {
                seen++;
            }
            while (j <= i && counts[alphabets.indexOf(s.charAt(j))] < 0) {
                counts[alphabets.indexOf(s.charAt(j++))]++;
            }
            if (seen == target) {
                minLen = Math.min(minLen, i - j + 1);
            }
        }
        return minLen;
    }
}
