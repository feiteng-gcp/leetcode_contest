class Solution {
    public boolean valid(int[] need, int[] count) {
        for (int i = 0; i < 4; i++) 
            if (need[i] > count[i]) 
                return false;
        return true;
    }
    
    public void update(char ch, int[] a, int v) {
        if (ch == 'Q') a[0]+=v;
        else if (ch == 'W') a[1]+=v;
        else if (ch == 'E') a[2]+=v;
        else a[3]+=v;
    }
    
    public int balancedString(String s) {
        int[] need = new int[4];
        for (char ch : s.toCharArray())
            update(ch, need, 1);
        int n = s.length()/4;
        // System.out.println(Arrays.toString(need));
        // System.out.println(n);
        for (int i = 0; i < 4; i++) need[i] = Math.max(need[i] - n, 0);
        // System.out.println(Arrays.toString(need));
        int[] count = new int[4];
        if (valid(need, count)) return 0;
        int ans = s.length();
        int left = 0;
        int right = 0;
        while (right <= s.length()) {
            if (valid(need, count)) {
                ans = Math.min(ans, right - left);
                // System.out.printf("%d, %d, %d\n", left, right, ans);
                update(s.charAt(left++), count, -1);
            }
            else {
                if (right < s.length())
                    update(s.charAt(right), count, +1);
                right++;
            } 
        }
        
        return ans;
    }
}
