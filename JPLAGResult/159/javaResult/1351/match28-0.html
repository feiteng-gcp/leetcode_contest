public class Solution {
	public int balancedString(String s) {
		int[] cnt = new int[4];
		for (char ch : s.toCharArray()) {
			switch (ch) {
				case 'Q':
					cnt[0]++;
					break;
				case 'W':
					cnt[1]++;
					break;
				case 'E':
					cnt[2]++;
					break;
				default:
					cnt[3]++;
					break;
			}
		}

		int N = s.length(), ans = N;
		for (int k = 0; k < 4; ++k) {
			cnt[k] = (cnt[k] > N / 4) ? cnt[k] - N/4 : 0;
		}
		if (Arrays.equals(cnt, new int[]{0, 0, 0, 0}))
			return 0;
		int[] cnt1 = new int[4];
		for (int i = 0, j = 0; j < N; ++j) {
			switch (s.charAt(j)) {
				case 'Q':
					cnt1[0]++;
					break;
				case 'W':
					cnt1[1]++;
					break;
				case 'E':
					cnt1[2]++;
					break;
				default:
					cnt1[3]++;
					break;
			}
			while (atLeast(cnt1, cnt)) {
				ans = Math.min(ans, j - i + 1);
				switch (s.charAt(i++)) {
					case 'Q':
						cnt1[0]--;
						break;
					case 'W':
						cnt1[1]--;
						break;
					case 'E':
						cnt1[2]--;
						break;
					default:
						cnt1[3]--;
						break;
				}
			}
		}
		return ans;
	}
	private boolean atLeast(int[] a, int[] b) {
		for (int i = 0; i < b.length; ++i) {
			if (a[i] < b[i]) return false;
		}
		return true;
	}
}
