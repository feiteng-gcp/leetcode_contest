class Solution {
    public int balancedString(String s) {
        int len = s.length(), total = len / 4;
        int[] a = new int[len];
        for (int i=0; i<len; i++) {
            char ch = s.charAt(i);
            if (ch == 'Q') a[i] = 1;
            else if (ch == 'W') a[i] = 2;
            else if (ch == 'E') a[i] = 3;
            else a[i] = 0;
        }
        int[][] sum = new int[len+1][4];
        for (int i=len-1; i>=0; i--) {
            for (int j=0; j<4; j++) sum[i][j] = sum[i+1][j];
            sum[i][a[i]]++;
        }
        int[] p = new int[4];
        int res = len;
        for (int i=0; i<len; i++) {
            int x = i, y = len, z, pos = -1;
            while (x <= y) {
                z = (x + y) >> 1;
                boolean flag = true;
                for (int j=0; j<4; j++) {
                    if (p[j] + sum[z][j] > total) {
                        flag = false;
                        break;
                    }
                }
                if (flag) {
                    pos = z;
                    y = z-1;
                }
                else {
                    x = z+1;
                }
            }
            if (pos >= i) {
                res = Math.min(res, pos - i);
            }
            
            p[a[i]]++;
        }
        return res;
    }
}
