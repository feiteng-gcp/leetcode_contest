class Solution {
    public int balancedString(String s) {
        Map<Character, Integer> count = new HashMap<>();
        for(char c : s.toCharArray()) {
            count.put(c, count.getOrDefault(c, 0) + 1);
        }
        
        int expect = s.length() / 4;
        
        Map<Character, Integer> map = new HashMap<>();
        for(Map.Entry<Character, Integer> e : count.entrySet()) {
            if(e.getValue() > expect) {
                map.put(e.getKey(), e.getValue() - expect);
            }
        }
        
        Map<Character, Integer> window = new HashMap<>();
        window.put('Q', 0);
        window.put('W', 0);
        window.put('E', 0);
        window.put('R', 0);
        
        int left = 0, right = 0, res = Integer.MAX_VALUE;
        while(right < 
s.length()) {
            char c = s.charAt(right);
            
            
            window.put(c, window.get(c) + 1);
            
            while(left <= right && window.get(s.charAt(left)) > map.getOrDefault(s.charAt(left), 0)) {
                window.put(s.charAt(left), window.get(s.charAt(left)) - 1);
                left++;
                
            }
            
            if(valid(window, map)) {
                res = Math.min(res, right - left + 1);
            }
            
            right++;
        
}
        
        return res;
    }
    
    private boolean valid(Map<Character, Integer> window, Map<Character, Integer> map) {
        return window.get('Q') >= map.getOrDefault('Q', 0)
                && window.get('W') >= map.getOrDefault('W', 0)
                && window.get('E') >= map.getOrDefault('E', 0)
                && window.get('R') >= map.getOrDefault('R', 0);
    }
}
