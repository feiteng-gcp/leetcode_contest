class Solution {
    public int balancedString(String s) {
        int exp=s.length()/4;
        int[] cnt=new int[4];
        int n=s.length();
        for(int i=0;i<n;i++)
{
            char c=s.charAt(i);
            if(c=='Q') cnt[0]++;
            if(c=='W') cnt[1]++;
            if(c=='E') cnt[2]++;
            if(c=='R') cnt[3]++;
        }
        
if(cnt[0]==cnt[1]&&cnt[1]==cnt[2]&&cnt[2]==cnt[3]) return 0;
        int[] need=new int[4];
        for(int i=0;i<4;i++){
            if(cnt[i]>exp){
                need[i]=cnt[i]-exp;
            } 
        }
        int ans=Integer.MAX_VALUE;
        int[] cur=new int[4];
        int start=0;
        for(int i=0;i<n;i++){
            
char c=s.charAt(i);
            if(c=='Q') cur[0]++;
            if(c=='W') cur[1]++;
            if(c=='E') cur[2]++;
            if(c=='R') cur[3]++;
            
            while(check(need,cur))
{
                ans=Math.min(ans,i-start+1);
                char sc=s.charAt(start++);
                if(sc=='Q') cur[0]--;
                if(sc=='W') cur[1]--;
                if(sc=='E') cur[2]--;
                if(sc=='R') cur[3]--;
            }
        }
        return ans;
    }
    private boolean check(int[] need,int[] cur){
        for(int i=0;i<4;i++){
            if(cur[i]<need[i]){
                return false;
            }
        }
        return true;
    }
}
