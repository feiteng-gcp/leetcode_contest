class Solution:
    def checkStraightLine(self, coordinates: List[List[int]]) -> bool:
        
        #vertical case
        if (coordinates[1][0] - coordinates[0][0]) == 0:
            x = coordinates[1][0]
                    
            for c in coordinates:
                if c[0] != x:
                    return False
                
            return True
            
        
        slope = (coordinates[1][1] - coordinates[0][1]) / (coordinates[1][0] - coordinates[0][0])
        x_0 =  coordinates[0][0]
        y_0 = coordinates[0][1]
        
        def check(point):
            # y-y_0 = m (x-x_0)
            return point[1] - y_0 == slope*(point[0] - x_0)
        
        for c in coordinates:
            if not check(c):
                return False
            
        return True
