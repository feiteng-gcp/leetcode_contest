class Solution:
    def checkStraightLine(self, coordinates: List[List[int]]) -> bool:
        if len(coordinates) == 2:
            return True
        
        if coordinates[0][0] == coordinates[1][0]:
            x = coordinates[0][0]
            for c in coordinates:
                if c[0] != x:
                    return False
            return True
        else:
            i = 2
            while i < len(coordinates):
                if (coordinates[1][0]-coordinates[0][0]) * (coordinates[2][1]-coordinates[1][1]) != (coordinates[2][0]-coordinates[1][0]) * (coordinates[1][1]-coordinates[0][1]):
                    return False
                i += 1
            return True
