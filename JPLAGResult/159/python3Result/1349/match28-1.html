class Solution:
    def checkStraightLine(self, c: List[List[int]]) -> bool:
        if c[1][0] == c[0][0]:
            return len(set([val[0] for val in c])) == 1
        k = (c[1][1]-c[0][1])/(c[1][0]-c[0][0])
        
for i in range(2,len(c)):
            if (c[i][0]-c[0][0]) * k !=  (c[i][1]-c[0][1]): return False
        return True
