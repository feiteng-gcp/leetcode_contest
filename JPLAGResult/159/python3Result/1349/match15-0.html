class Solution:
    def checkStraightLine(self, c: List[List[int]]) -> bool:
        if c[0][0] - c[1][0] == 0:
            for i in c:
                if i[0] == c[0][0]:
                    continue
                else:
                    return False
            return True
        k = (c[0][1] - c[1][1]) / (c[0][0] - c[1][0])
        
        b = c[0][1] - k * c[0][0]
        
        for i in c:
            if i[0] * k + b == i[1]:
                continue
            else:
                return False
        
        return True
