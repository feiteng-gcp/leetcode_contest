class Solution:
    def checkStraightLine(self, coordinates: List[List[int]]) -> bool:
        n = len(coordinates) 
        if n <= 2:
            return True
        a, b = coordinates[0], coordinates[1]
        for i in range(2, n):
            c = coordinates[i]
            if (b[0]-a[0])*(c[1]-a[1]) != (c[0]-a[0])*(b[1]-a[1]):
                return False
        return True
