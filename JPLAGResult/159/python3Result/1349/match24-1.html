class Solution:
    def checkStraightLine(self, coordinates: List[List[int]]) -> bool:
        b = coordinates[0][0] - coordinates[1][0]
        a = coordinates
[0][1] - coordinates[1][1]
        c = a * coordinates[0][0] - b * coordinates[0][1]
        for x, y in coordinates:
            if a * x - b * y != c:
                return False
        return True
