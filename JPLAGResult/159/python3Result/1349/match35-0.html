class Solution:
    def checkStraightLine(self, A: List[List[int]]) -> bool:
        N = len(A)
        i = j = 0
        for k in range(1, N):
            if A[i] != A[k]:
                j = k
                break
        if j == 0 or j+1 == len(A):
            return True
        for k in range(j+1, N):
            x = (A[k][0] - A[i][0]) * (A[j][1] - A[i][1]) - (A[k][1] - A[i][1]) * (A[j][0] - A[i][0])
            if x != 0:
                return False
        return True
