class Solution:
    def checkStraightLine(self, coordinates: List[List[int]]) -> bool:
        coordinates.sort()
        def line(i, j, k):
            if coordinates[i][0] == coordinates[j][0] == coordinates[k][0]:
                return True
            else:
                return (coordinates[j][0] - coordinates[i][0]) * (coordinates[k][1] - coordinates[j][1]) == (coordinates[j][1] - coordinates[i][1]) * (coordinates[k][0] - coordinates[j][0])
            
        for i in range(2, len(coordinates)):
            if line(i-2, i-1, i):
                continue
            else:
                return False
        return True
