class Solution:
    def checkStraightLine(self, co: List[List[int]]) -> bool:
        fenzi, fenmu = co[0][0] - co[1][0], co[0][1] - co[1][1]
        for idx, c in enumerate(co[1:], 1):
            dx, dy = co
[idx-1][0] - c[0], co[idx-1][1] - c[1]
            if dx * fenmu != dy * fenzi:
                return False
        return True
