class Solution:
    def checkStraightLine(self, C) -> bool:
        N = len(C)
        x1, y1 
= C[0][0], C[0][1]
        x2, y2 = C[1][0], C[1][1]
        for i in range(2, N):
            x, y 
= C[i][0], C[i][1]
            if (x - x1) * (y2 - y1) != (x2 - x1) * (y - y1):
                return False
        return True
