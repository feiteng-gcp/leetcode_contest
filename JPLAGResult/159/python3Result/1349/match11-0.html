class Solution:
    def checkStraightLine(self, c: List[List[int]]) -> bool:
        ret = True
        dx, dy = c[1][0] - c[0][0], c[1][1] - c[0][1]
        for i in range(2, len(c)):
            x, y = c[i][0] - c[0][0], c[i][1] - c[0][1]
            if x * dy - y * dx != 0:
                ret = False
                break
        return ret
