class Solution:
    def checkStraightLine(self, cood) -> bool:
        dx 
= cood[1][0] - cood[0][0]
        dy = cood[1][1] - cood[0][1]
        for i in range(2,len(cood)):
            ddx = cood[i][0] - cood[i-1][0]
            ddy = cood[i][1] - cood[i-1][1]
            if dx*ddy!=dy*ddx or (dx == 0 and ddx!=0) or (dy == 0 and ddy!=0):
                return False
        return True
