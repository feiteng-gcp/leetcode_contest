class Solution:
    def checkStraightLine(self, coordinates: List[List[int]]) -> bool:
        n = len(coordinates)
        diffs = [(coordinates[i][0] - coordinates[i + 1][0], coordinates[i][1] - coordinates[i + 1][1]) for i in range(n - 1)]
        for i in range(len(diffs) - 1):
            if diffs[i][0] * diffs[i + 1][1] != diffs[i][1] * diffs[i + 1][0]:
                return False
        return True
