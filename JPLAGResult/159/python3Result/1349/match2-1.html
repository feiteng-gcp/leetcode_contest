import fractions


class Solution:
    def checkStraightLine(self, coord: List[List[int]]) -> bool:
        dy, dx = (coord[1][1] - coord[0][1]),  (coord[1][0] - coord[0][0])
        slope = None if not dx else fractions.Fraction(dy, dx)
        
for i in range(2, len(coord)):
            dy, dx = (coord[i][1] - coord[0][1]),  (coord[i][0] - coord[0][0])
            if (None if not dx else fractions.Fraction(dy, dx)) != slope:
                return False
        return True
