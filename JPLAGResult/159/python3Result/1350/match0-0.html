class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        res = set()
        folder.sort()
        def check(f):
            s = f.split('/')
            tmp = ''
            for i in s[1:]:
                tmp += ('/' + i)
                if tmp in res:
                    return False
            return True
        for f in folder:
            if check(f):
                res.add(f)
        return res
