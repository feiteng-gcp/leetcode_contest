class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        
        folder_set = set(folder)
        
        ans = []
        for f in folder:
            
            slash_index = []
            for i,c in enumerate(f):
                if c == "/":
                    if i > 0:
                        slash_index.append(i)
                        
            subfolder = False
            
            for i in slash_index:
                possible_parent = f[:i]
                if possible_parent in folder_set:
                    subfolder 
= True
                    break
                    
            if subfolder == False:
                ans.append(f)
        
        return ans
