class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        trie = {}
        for s in folder:
            t 
= trie
            for c in s + '/':
                if c not in t:
                    t[c] = {}
                t = t[c]
            t['#'] = '#'


        def helper(curt, path, res):
            if '#' in curt:
                res.append(''.join(path[:-1]))
                return
            for c in curt:
                path.append(c)
                helper(curt[c], path, res)
                path.pop()

        res = []
        helper(trie, [], res)
        return res
