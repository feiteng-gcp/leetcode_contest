class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        trie = {}
        for path in folder:
            node = trie
            for f in path.split("/")[1:]:
                node = node.setdefault(f, {})
            node["$"] = True

        root = []

        def dfs(node, path):
            for key in node:
                new_path = f"{path}/{key}"
                
if "$" in node[key]:
                    root.append(new_path)
                else:
                    dfs(node[key], new_path)
        dfs(trie, "")

        return root
