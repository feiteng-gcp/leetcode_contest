class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        root = {}
        for fold in folder:
            f = fold[1:].split('/')
            node = root
            for nm in f:
                if nm not in node:
                    node[nm] = {}
                node = node[nm]
            node['#'] = fold
            
        res = []
        def dfs(node):
            if '#' in node:
                res.append(node['#'])
            else:
                for child in node:
                    dfs(node[child])
                
        dfs(root)
        return res
