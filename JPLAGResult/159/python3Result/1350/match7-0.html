class Solution:
    def removeSubfolders(self, A: List[str]) -> List[str]:
        A.sort()
        ans = []
        root = {}
        for raw in A:
            cur = root
            fail = False
            for w in raw.split("/"):
                if w not in cur:
                    cur[w] = {}
                cur = cur[w]
                if "#" in cur:
                    fail = True
                    break
            if not fail:
                cur["#"] = True
                ans.append(raw)
        return ans
