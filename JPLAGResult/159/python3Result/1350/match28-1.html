class Solution:
    def removeSubfolders(self, a: List[str]) -> List[str]:
        from collections import defaultdict as dd
        f = dd(bool)
        a.sort()
        ans = []
        for i in range(len(a)):
            x = a[i]
            v 
= True
            for j in range(1, len(x)):
                if x[j]=='/' and f[x[:j]]==True:
                    v = False
                    break
            if v:
                ans.append(x)
                f[x] = True
        
        return ans
