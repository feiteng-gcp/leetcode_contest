class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        ans = []
        folder = sorted([file[1:].split("/") for file in folder])
        root = {}
        for file in folder:
            now = root
            exist = False
            for c in file:
                if c not in now:
                    now[c] = {}
                now = now[c]
                if "#" in now:
                    exist = True
                    break
            if not exist:
                now["#"] = {}
                ans.append("/" + "/".join(file))
        return ans
