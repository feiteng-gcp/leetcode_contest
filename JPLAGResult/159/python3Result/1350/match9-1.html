class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        def func(string):
            return len(string.split('/'))
        folder.sort(key=func)
        path=set()
        for f in folder:
            for p in path:
                l=len
(p)
                if f[:l]==p and len(f)>l and f[l]=='/':
                    break
            else:
                path.add(f)
        return list(path)
