class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folder = sorted(folder)
        i = j = 0
        res = [folder[0]]
        while i < len(folder) and j < len(folder):
            if i == j:
                j += 1
                continue
            if len(folder[j]) > len(folder[i]) and folder[j][:len(folder[i])] == folder[i] and folder[j][len(folder[i])] == '/':
                j += 1
                continue
            else:
                i = j
                res.append(folder[j])
        return res
