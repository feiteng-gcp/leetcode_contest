class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        bag = set()
        splits = []
        for idx, f in enumerate(folder):
            splits.append(f.split('/'))
            bag.add(f)
        ans = []
        for spt in splits:
            flag 
= True
            for i in range(len(spt)):
                fa = '/'.join(spt[:i])
                # print(fa)
                if fa in bag:
                    flag = False
                    break
            if flag:
                ans.append('/'.join(spt))
        return ans
