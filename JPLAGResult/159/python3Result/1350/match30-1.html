class Node:
    def __init__(self, s, isFolder):
        self.s = s
        self.isFolder = isFolder
        self.children = dict()
    

class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        # Build Trie
        root 
= Node('', False)    
        for f in folder:
            tokens = f.split('/')
            cur = root
            for i in range(1, len(tokens)):
                t = tokens[i]
                if t not in cur.children:
                    cur.children[t] = Node(t, False)
                cur = cur.children[t]
            cur.isFolder = True
        
        def dfs(curNode, trail, ans):
            if curNode.isFolder:
                ans.append('/'.join(trail))
                return

            for s, child in curNode.children.items():
                trail.append(s)
                dfs(child, trail, ans)
                trail.pop()
            
        ans = []
        dfs(root, [''], ans)
        return ans
