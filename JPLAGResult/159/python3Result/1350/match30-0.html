class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folder.sort(key = lambda x: len(x))
        d = set()
        ans = list()
        for f in folder:
            path = f.split("/")
            flag = True
            for i in range(2, len(path)):
                cur = "/".join(path[:i])
                if cur in d:
                    flag = False
                    break
            if flag:
                d.add(f)
                ans.append(f)
        return ans
