class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        trie = Trie()
        for path in folder:
            p = trie
            for d in path.strip('/').split('/'):
                if d not in p.next:
                    p.next[d] = Trie()
                p = p.next[d]
            p.term = True
            
        out = []
        def dfs(p, path):
            if p.term:
                out.append(path)
                return
            for d in p.next:
                dfs(p.next[d], path + '/' + d)
        dfs(trie, '')
        return out
        
        
class Trie:
    def __init__(self):
        self.next = {}
        self.term = False
