class Node:
    def __init__(self):
        self.children = collections.defaultdict(None)
        self
.flag = False


class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        root = Node()
        for f in folder:
            names = f.split('/')[1:]
            node = root
            for name in names:
                
if name not in node.children:
                    node.children[name] = Node()
                node = node.children[name]
            if node is not root:
                node.flag = True
        self.sol = []
        self.dfs(root)
        return self.sol

    def dfs(self, root, name=''):
        if root.flag:
            root.children = collections.defaultdict(None)
            self.sol.append(name)
        else:
            for fname, node in root.children.items():
                self.dfs(node, name+'/'+fname)
