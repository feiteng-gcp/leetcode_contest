class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        
        folder.sort(key = lambda x: len(x))
        trie = {}
        res = []
        for path in folder:
            path = path.split('/')
            
            t = trie
            found = False
            for c in path[1:]:
                if 'end' in t:
                    found = True
                    break
                if c not in t:
                    t[c] = {}
                t = t[
c]
            
            if not found:
                t['end'] = True
                res.append('/'.join(path))
        return res
