class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        trie = {}
        for path in folder:
            l = path.split('/')
            l = l[1:]
            root = trie
            for char in l:
                if char not in root:
                    root[char] = {}
                root = root[char]
            root['#'] = 1
            
        def dfs(root, curr, res):
            if '#' in root:
                res.add(curr)
                return
            for char in root:
                dfs(root[char], curr+'/'+char, res)
        res = set()
        dfs(trie, '', res)
        return res
