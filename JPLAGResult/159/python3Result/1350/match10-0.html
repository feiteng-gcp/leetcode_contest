class Solution:
    def removeSubfolders(self, folder):
        s = set()
        res  =[]
        for f in folder:
            s.add(f+"/")
        for f in folder:
            for i in range(len(f)):
                if f[:i] in s:
                    break
            else:
                res.append(f)
        return res
