class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        guard = object()
        di = dict()
        for f in folder:
            d = f.split('/')[1:]
#             print(d)
            end = d[-1]
            d = d[:-1]
            d.reverse()
            cur_di = di
            new_di = True
            while d:
                cur_f = d.pop()
                n = cur_di.get(cur_f, None)
                if n is guard:
                    new_di = False
                    break
                    
                if n is None:
                    cur_di[cur_f] = dict()
                cur_di = cur_di[cur_f]
            if new_di:
#                 print('new', cur_di, end)
                cur_di[end] = 
guard

        res = []
        def helper(f, c):
#             print(f,c)
#             print()
            if f is guard:
                res.append(c)
                return
            for k,v in f.items():
                helper(v, c+'/'+k)
        helper(di, '')
        return res
