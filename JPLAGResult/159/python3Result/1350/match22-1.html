class Trie:
    def __init__(self):
        self.dic = {}
    def add(self, s):
        path = s.split('/')
        d = self.dic
        for f in path:
            if f not in d:
                d[f] = {}
            d = d[f]
        d['$'] = {}
        
class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        trie = Trie()
        for f in folder:
            trie.add(f)
        # print(trie.dic)
        res = []
        def dfs(d, path = []):
            
if '$' in d:
                res.append('/'.join(path))
                return
            for k in d:
                path.append(k)
                dfs(d[k])
                path.pop()
        dfs(trie.dic)
        return res
