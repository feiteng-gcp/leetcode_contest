class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        tree = {}
        for di in folder:
            nodes = di.split('/')
            p = tree
            n = len(nodes)
            if (n==2 and len(nodes[1])==0):
                return ['/']
            for i in range(1,n):
                node = nodes[i]
                if not node in p:
                    p[node] = {}
                elif not p[node]:
                    break
                if i==n-1:
                    p[node] = {}
                else:
                    p = p[node]      
        ret = []
        def enuTree(tree,ret,pa):
            for node in tree:
                if not tree[node]:
                    ret.append(pa+'/'+node)
                else:
                    enuTree(tree[node],ret,pa+'/'+node)
        enuTree(tree,ret,'')
        return ret
