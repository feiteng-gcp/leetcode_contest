from functools import reduce
class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        T = lambda: collections.defaultdict(T)
        root = T()
        for s in folder:
            reduce(lambda x,y: x[y], s.split('/'), root)['#']=True
        tmp = []
        out = []
        def recur(node):
            if '#' in node:
                out.append('/'.join(tmp))
                return
            for c in node:
                tmp.append(c)
                recur(node[c])
                tmp.pop()
        recur(root)
        return out
