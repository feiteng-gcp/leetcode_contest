from collections import deque
class Node:
    def __init__(self):
        self.children = {}
        self.leaf = False
        self.val = None


class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        root = Node()
        for fd in folder:
            cur = root
            components = fd.split('/')[1:]
            remove = False
            for i, d in enumerate(components):
                if cur.leaf:
                    remove = True
                    break
                if d not in cur.children:
                    cur.children[d] = Node()
                cur = cur.children[d]
            if remove:
                continue
            cur.leaf = True
            cur.val = fd
            cur.children = {}
        q = deque([root])
        res = []
        while q:
            cur = q.popleft()
            if cur.leaf:
                res.append(cur.val)
            for k, v in cur.children.items():
                q.append(v)
        return res
