class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folders = sorted(map(lambda v: tuple(v.split("/")[1: ]), folder), key=len)
        seen = set()

        
for folder in folders:

            for i in range(1, len(folder)):
                if folder[: i] in seen:
                    break
            else:
                seen.add(folder)

        return list(map(lambda v: "/" + "/".join(v), seen))
