class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folder = sorted([file for file in folder if file not in set(['/', ''])])
        father = set()
        for file in folder:
            s = file.split('/')
            p = ''
            ok = True
            for i in range(1, len(s) - 1):
                p += '/'
                p 
+= s[i]
                if p in father:
                    ok = False
                    break
            if ok:
                father.add(file)
        return list(father)
