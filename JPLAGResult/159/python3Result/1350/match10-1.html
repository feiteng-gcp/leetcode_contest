class Solution:
    def removeSubfolders(self, f: List[str]) -> List[str]:
        f.sort()
        s = set()
        for a in f:
            b = tuple(a.split('/'))
            
for i in range(len(b)):
                if b[:i] in s:
                    break
            else:
                s.add(b)
        return ['/'.join(i) for i in s]
