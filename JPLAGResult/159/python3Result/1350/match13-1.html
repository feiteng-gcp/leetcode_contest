class TrieNode:
    def __init__(self):
        self.children = dict()
        self.isFolder = False
        
class TrieTree:
    def __init__(self):
        self.root = TrieNode()
        
    def add_folder(self, s):
        fl = s.split('/')
        node = self.root
        for fn in fl[1:]:
            if fn in node.children and node.children[fn].isFolder:
                return
            elif fn not in node
.children:
                node.children[fn] = TrieNode()
            node = node.children[fn]
        node.isFolder = True
        
    def get_folders(self):
        result 
= []
        
        def dfs(node, current):
            nonlocal result
            if node.isFolder:
                result.append(current)
                return
            
            for child in node.children:
                dfs(node.children[child], current+'/'+child)
            
        dfs(self.root, '')
        return result
            
    
class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        tree = TrieTree()
        
        for f in folder:
            tree.add_folder(f)
          
        return tree.get_folders()
