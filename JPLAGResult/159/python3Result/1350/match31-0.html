class Trie:
    def __init__(self):
        self.d = collections.defaultdict(lambda:collections.defaultdict())
        self.z = False
        
        
class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        d=Trie()
        folder.sort(key=len)
        # print(folder)
        ret = []
        for f in folder:
            # print(f)
            x = f.split('/')
            # print(x)
            n = len(x)
            curr = d
            gd = True
            for i in range(1,n):
                nm = x[i]
                # print(nm)
                if nm in curr.d:
                    curr = curr.d[nm]
                    if curr.z:
                        gd = False
                        break
                else:
                    curr.d[nm] = Trie()
                    curr = curr.d[nm]
            if gd:
                curr.z = True
                ret.append(f)
        return ret
