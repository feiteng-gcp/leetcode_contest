class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        # sort, then process one by one
        folder = sorted(folder, key=len)
        roots = set()
        for f in folder:
            # print(roots)
            dirs, tmpPath, add = f.split('/'), '/', True
            # print(dirs)
            for d in dirs[1:]:
                tmpPath += d
                # print('tmpPath: ' + tmpPath)
                if tmpPath in roots:
                    add = False
                    break
                tmpPath += '/'
            if add:
                roots.add(f)
        return list(roots)
