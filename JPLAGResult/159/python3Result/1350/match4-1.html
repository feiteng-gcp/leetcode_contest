class Solution:
    def removeSubfolders(self, folders: List[str]) -> List[str]:
        paths = []
        for folder in folders:
            path = folder.split('/')
            paths.append(path)
        paths.sort(key=len)
        roots = set()
        for path in paths:
            unique 
= True
            for i in range(len(path)):
                if '/'.join(path[:i]) in roots:
                    unique = False
                    break
            if unique:
                roots.add('/'.join(path))
        result = list(roots)
        return result
