class Solution:
    def removeSubfolders(self, F: List[str]) -> List[str]:
        F.sort()
        res = set()
        for f in F:
            skip = False
            for i in range(1, len
(f)):
                if f[i] == '/':
                    if f[:i] in res:
                        skip = True
                        break
            if not skip:
                res.add(f)
        return list(res)
