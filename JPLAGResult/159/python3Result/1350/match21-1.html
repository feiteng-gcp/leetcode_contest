class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folder.sort()
        res = [folder[0]]
        
        for i in range(1, len(folder)):
            
if len(folder[i]) <= len(res[-1]) or folder[i][:len(res[-1])] != res[-1] or folder[i][len(res[-1])] != '/':
                res.append(folder[i])
        
        return res
