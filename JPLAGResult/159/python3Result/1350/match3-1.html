class TrieNode:
    def __init__(self):
        self.children = {}
        self.isvalid = False


class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folder.sort(key=lambda x: len(x))
        root = TrieNode()
        result = []
        for f in folder:
            temp = f.split('/')
            node = root
            flag = 0
            print(temp)
            for i in range(1, len(temp)):
                if temp[i] not in node.children:
                    node.children[temp[i]] = TrieNode()
                if node.children[temp[i]].isvalid:
                    flag = 1
                node = node.children[temp[i]]
            node.isvalid = True
            if not flag:
                result.append(f)
        return result
