class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folders = sorted(folder)
        ans=[]
        for folder in folders:
            if len(ans)==0:
                ans.append(folder)
            else:
                if folder.startswith(ans[-1]) and folder[len(ans[-1])]=='/':
                    pass
                else:
                    ans.append(folder)
        return ans
