from collections import defaultdict

class Node(object):
    def __init__(self):
        self.children = collections.defaultdict(Node)
        self.end = False 

class Tree(object):
    def __init__(self):
        self.root = Node()
    
    def insert(self, folders):
        cur = self.root
        for f in folders:
            if cur.end:
                return False 
            cur = cur.children[f]
        
        cur.end = True
        return True
    

class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folder = sorted(folder, key=lambda x : len(x))
        
        tree = Tree()
        res = []
        
        for f in folder:
            folders = f.split("/")
            if tree.insert(folders[1:]):
                res.append(f)
        
        return res
