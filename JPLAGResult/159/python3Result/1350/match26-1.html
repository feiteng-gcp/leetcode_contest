class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folder.sort(key=len)
        T=lambda: collections.defaultdict(T)
        trie=T()
        res=[]
        for f in folder:
            t=trie
            path=f.split('/')
            bad=False
            for p in path:
                t
=t[p]
                if '#' in t:
                    bad=True
                    break
            if not bad:
                t['#']=None
                res.append(f)
        return res
