class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folder = sorted(folder, key = lambda x : len(x) )
        # Sorted length wise now for any x if previous j in x then skip this one else add
        ans = set()
        for i in folder:
            skip = False
            sl = i.split("/")
            prep_string = ""
            for j in sl:
                prep_string 
+= j 
                if( prep_string in ans):
                    skip = True
                    break
                prep_string += '/'
            
            if skip == False:
                ans.add(i)
        return list(ans)
