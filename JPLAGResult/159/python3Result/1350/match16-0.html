class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folder.sort(key=len)
        ans = [folder[0]]
        for f in folder[1:]:
            flag = True
            for x in ans:
                if f.startswith(x + "/"):
                    flag = False
                    break
            if flag:
                ans.append(f)
        return ans
