class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        folderSet = set()
        folder = sorted(folder)
        res = []
        for f in folder:
            p = f.split("/")
            flag = True
            for i in range(2, len(p)):
                if "/".join(p[:i]) in folderSet:
                    flag 
= False
                    break
            if flag:
                res.append(f)
                folderSet.add(f)
        return res
