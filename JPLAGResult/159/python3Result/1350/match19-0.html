class Solution:
    def removeSubfolders(self, folder: List[str]) -> List[str]:
        root = {}
        for f in folder:
            paths = f.split('/')
            node = root
            for i in range(1, len(paths)):
                path = paths[i]
                if path not in node:
                    node[path] = {}
                node = node[path]
            node['#'] = True
        
        ans = []
        def dfs(node: dict, path: str):
            nonlocal ans
            if '#' in node:
                ans.append(path)
                return
            for k, v in node.items():
                dfs(v, path + '/' + k)
        dfs(root, '')
        return ans
