import bisect

class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        jobs = sorted(zip(startTime, endTime, profit), key=lambda v: v[1])
        endTimes = list(map(lambda v: v[1], jobs))
        dp = {}
        previous = [-1]

        for i, v in enumerate(jobs[1: ], 1):
            startTime = v[0]
            index = bisect.bisect_right(endTimes, startTime)
            if index == 0:
                if jobs[0][1] <= startTime:
                    previous.append(0)
                else:
                    previous.append(-1)
            else:
                if jobs[index - 1][1] > startTime:
                    previous.append(-1)
                else:
                    previous.append(index - 1)

        dp[0] = jobs[0][2]

        for i, v in enumerate(jobs[1: ], 1):
            profit = v[2]
            startTime = v[0]
            if previous[i] == -1:
                dp[i] = max(profit, dp[i - 1])
            else:
                dp[i] = max(profit + dp[previous[i]], dp[i - 1])

        return dp[len(jobs) - 1]
