def lowbit(x):
    return x & (-x)

class BST:
    def __init__(self, n):
        self.size = n
        self.arr = [0] * n
    
    def getMax(self, ind):
        res = 0
        while ind > 0:
            res = max(res, self.arr[ind])
            ind -= lowbit(ind)
        return res
    
    def updateMax(self, ind, value):
        while ind < self.size:
            self.arr[ind] = max(self.arr[ind], value)
            ind += lowbit(ind)

class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        n = len(startTime)
        intervals = [(startTime[i], endTime[i], profit[i]) for i in range(n)]
        cors = startTime + endTime
        cors.sort()
        intervals.sort()
        indMap = {}
        bst = BST(len(cors) + 1)
        for i, cor in enumerate(cors):
            indMap[cor] = i + 1
        for start, end, prof in intervals:
            # print(start, end, prof, bst.getMax(indMap[start]))
            bst.updateMax(indMap[end], bst.getMax(indMap[start]) + prof)
        return bst.getMax(bst.size - 1)
