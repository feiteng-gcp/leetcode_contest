class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        tasks = []
        N = len(startTime)
        for i in range(N):
            tasks.append((startTime[i], endTime[i], profit[i]))
        tasks.sort(key=lambda item: item[1])
        
        def latestNonConflict(tasks, i):
            for j in range(i-1, -1, -1):
                if tasks[j][1] <= tasks[i][0]:
                    return j
            return -1
        
        dp = [0] * N
        dp[0] = tasks[0][2]
        
        for i in range(1, N):
            inclProf = tasks[i][2]
            l = latestNonConflict(tasks, i)
            if l != -1:
                inclProf += dp[l]
            dp[i] = max(inclProf, dp[i-1])
        
        return dp[N - 1]
