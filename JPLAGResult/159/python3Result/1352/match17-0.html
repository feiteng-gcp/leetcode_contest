class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        n = len(startTime)
        
        jobs = [(startTime[i], endTime[i], profit[i]) for i in range(n)]
        jobs.sort()
        
        starts = [jobs[i][0] for i in range(n)]
        
        from functools import lru_cache
        @lru_cache(None)
        def dp(i):
            if i >= n:
                return 0
            
            start, end, profit = jobs[i]
            k = bisect.bisect_left(starts, end)
            profit += dp(k)
            
            return max(dp(i+1), profit)
            
            
        return dp(0)
