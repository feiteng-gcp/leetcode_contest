class Solution:
    def jobScheduling(self, startTime, endTime, profit):
        N = len(startTime)
        j = []
        for i in range(N):
            j.append([startTime[i], endTime[i], profit[i]])

        sl = sorted
(j, key=lambda x:x[1])

        def last(i):
            for j in range(i-1, -1, -1):
                if (sl[j][1] <= sl[i][0]):
                    return j
            return -1

        M =[0 for i in range(N)]
        M
[0] = sl[0][2]

        for i in range(1, N):
            profit = sl[i][2]
            l = last(i)
            if l!=-1:
                profit += M[l]
            M[i] = max(profit, M[i-1])
        return M[-1]
