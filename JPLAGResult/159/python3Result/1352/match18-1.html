class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        jobs = sorted([(s,e,p) for s,e,p in zip(startTime, endTime, profit)], key=lambda x: x[1])
        ends = [job[1] for job in jobs]
        N = len(jobs)
        dp = [0] * (N+1)
        for i in range(N):
            j = bisect.bisect(ends, jobs[i][0])
            dp
[i+1] = max(dp[i], jobs[i][2] + dp[j])
        return dp[N]
