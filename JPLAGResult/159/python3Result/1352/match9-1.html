class Solution:
    def jobScheduling(self, s: List[int], e: List[int], p: List[int]) -> int:
        z = sorted(zip(e, s, p))
        e = [x[0] for x in z]
        s = [x
[1] for x in z]
        p = [x[2] for x in z]
        
        # print(s, e)
        
        def binarySearch(start_index): 
            lo = 0
            hi = start_index - 1
 
            while lo <= hi: 
                mid = (lo + hi) // 2
                if e[mid] <= s[start_index]: 
                    if e[mid + 1] <= s[start_index]: 
                        lo = mid + 1
                    else: 
                        return mid 
                else: 
                    hi = mid - 1
                    
            return -1
         
        n = len(s)  
        table = [0 for _ in range(n)] 

        table[0] = p[0]

        
for i in range(1, n):  
            inc = p[i] 
            l = binarySearch(i) 
            
            if (l != -1): 
                inc += table[l]; 

            table[i] = max(inc, table[i - 1]) 

        # print(table)
        return table[n-1]
