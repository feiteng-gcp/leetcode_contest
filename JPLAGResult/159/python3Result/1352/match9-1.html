class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        
        course = [(startTime[i],endTime[i],profit[i]) for i in range(len(startTime))]
        course.sort(key = lambda x: x[1])
   
        def findLast(i):
        
            for j in range(i, -1, -1):
                if course[j][1] <= course[i][0]:
                    return j
            return -1
    
        n = len(startTime)
        dp = [0] * n
        
        dp
[0]= course[0][2]
        
        for i in range(1, n):
            
            tmp = course[i][2]
          
            
            idx = findLast(i)
            
            if idx != -1:
                tmp += dp[idx]
            
            dp[i] = max(dp[i-1], tmp)
        
        return dp[-1]
