class Solution:
    def jobScheduling(self, startTime, endTime, profit) -> int:
        jobs = []
        for i in range(len(startTime)):
            jobs.append((startTime[i],endTime[i],profit[i]))
        jobs.sort(key = lambda x : x[1])
        res = [0]*len(startTime)
        res[0] = jobs[0][2]
        #print(jobs)
        n = len(startTime)
        def latestNonConflict(i):
            for j in range(i-1,-1,-1):
                #print(jobs[i],jobs[j])
                if jobs[j][1] <= jobs[i][0]:
                    return j
            return -1
        for i in range(1,n):
            inclProf = jobs[i][2]
            l = latestNonConflict(i)
            #print(i,l)
            if l!=-1:
                inclProf+=res[l]
            res[i] = max(inclProf,res[i-1])
        #print(res)
        return res[-1]
