class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        
        dp = [[0,0]]
        
        tasks = [(startTime[i],endTime[i],profit[i]) for i in range(len(startTime))]
        tasks.sort(key=lambda x: x[1])
        
        #print(tasks)
        
        for start, end, prof in tasks:
            
            j = bisect.bisect_right(dp, [start, float('Inf')])
            
            last_prof = dp[j-1][1]
            if dp[-1][1] > last_prof + prof: continue
            if dp[-1][0] == end:
                dp[-1][1] = max(dp[-1][1], last_prof + prof)
            else:
                dp.append([end, last_prof+prof])
            #print(dp)
        
        return max(dp[i][1] for i in range(len(dp)))
