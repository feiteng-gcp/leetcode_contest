class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        jobs = list(zip(startTime, endTime, profit))
        jobs = sorted(jobs, key=lambda j: j[1])
        
        n = len(jobs)
        
        dp = [0 for _ in range(n)]
        
        dp[0] = jobs[0][2]
        
        def binarySearch(jobs, i): 
            start = 0
            end = i - 1
            while start <= end: 
                mid = (start + end) // 2
                if jobs[mid][1] <= jobs[i][0]: 
                    if jobs[mid + 1][1] <= jobs[i][0]: 
                        start = mid + 1
                    else: 
                        return mid 
                else: 
                    end = mid - 1
            return -1
        
        for i in range(1, n):
            p = jobs[i][2]
            start = 0
            end = i - 1
            
            k = binarySearch(jobs, i)
            if k != -1:
                p += dp[k]
            
            dp[i] = max(dp[i - 1], p)
        return dp[n - 1]
