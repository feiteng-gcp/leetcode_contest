class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        info = sorted(list(zip(startTime, endTime, profit)), key = lambda x: x[1])
        n = len(info)
        stack = [(float('-inf'), 0)]
        for s, e, p in info:
            j = bisect.bisect_right(stack, (s, float('inf')))
            if p + stack[j - 1][1] > stack[-1][1]:
                stack.append((e, p + stack[j - 1][1]))
        return stack[-1][1]
