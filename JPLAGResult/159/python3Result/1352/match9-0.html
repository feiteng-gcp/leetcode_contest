class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        n = len(startTime)
        idx = sorted(range(n), key=lambda i:endTime[i])
        startTime = [startTime[i] for i in idx]
        endTime = [endTime[i] for i in idx]
        profit = [profit[i] for i in idx]
        dp = [0 for _ in range(n)]
        dp[0] = profit[0]
        def bs(i0):
            l = 0
            h = i0 - 1
            while l <= h:
                m = (l + h) // 2
                if endTime[m] <= startTime[i0]:
                    if endTime[m + 1] <= startTime[i0]:
                        l = m + 1
                    else:
                        return m
                else:
                    h = m - 1
            return -1
        for i in range(1, n):
            curr = profit[i]
            l = bs(i)
            if l != -1:
                curr += dp[l]
            dp[i] = max(curr, dp[i - 1])
        return dp[n - 1]
