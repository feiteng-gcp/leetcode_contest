class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        jobs = sorted(zip(startTime, endTime, profit))
        
        N = len(jobs)
        dp = [0] * (N + 1)
        
        for i in range(N-1, -1, -1):
            j = i + 1
            while j < N:
                if jobs[j][0] < jobs[i][1]:
                    j += 1
                else:
                    break 
            
            dp[i] = max(dp[i+1], jobs[i][2] + dp[j])
            
        return dp[0]
