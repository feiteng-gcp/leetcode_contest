class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        distinct_timestamps = sorted(list(set(startTime+endTime)))
        t2dt = {}
        for dt,t in enumerate(distinct_timestamps): t2dt[t]=dt
        startTime = [t2dt[x] for x in startTime]
        endTime = [t2dt[x] for x in endTime]
        tn=len(distinct_timestamps)
        t2endingjobs=[[] for _ in range(tn)]
        for j,edt in enumerate(endTime): t2endingjobs[edt].append(j)
        f=[0 for _ in range(tn)]
        for i in range(tn):
            f[i]=0 if i==0 else f[i-1]
            for j in t2endingjobs[i]:
                f[i]=max(f[i],f[startTime[j]]+profit[j])
        return f[tn-1]
