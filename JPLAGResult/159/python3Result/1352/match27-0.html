from collections import defaultdict
class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        starts =defaultdict(list)
        ends = defaultdict(list)
        for i in range(len(startTime)):
            s,e,p = startTime[i], endTime[i],profit[i]
            t = (s,e,p)
            starts[s].append(t)
            ends[e].append(t)
        sk = sorted(starts.keys())
        ek = sorted(ends.keys())
        i,j=0,0
        dp = defaultdict(int)
        mxp = 0
        lastt = ek[-1]
        while j <len(ek):
            s=lastt if i>= len(sk) else sk[i]
            e=ek[j]
            mnt = min(s,e)
            if mnt==s:
                dp[mnt] = mxp
                i+=1
            if mnt==e:
                for t in ends[e]:
                    ss = t[0]
                    pp = t[2]
                    mxp = max(mxp, dp[ss]+pp)
                dp[mnt] = mxp
                j+=1
        return mxp
