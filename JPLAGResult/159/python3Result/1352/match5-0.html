class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        job = [(endTime[i], startTime[i], profit[i]) for i in range(len(profit))]
        job.sort()
        res = [0] * len(job)
        res[0] = job[0][2]
        for i in range(1, len(job)):
            e, s, p = job[i]
            newjob = (s, s + 1, p)
            ind = bisect.bisect_left(job, newjob)
            if ind > 0:
                p += res[ind - 1]
            res[i] = max(res[i - 1], p)
        return res[-1]
