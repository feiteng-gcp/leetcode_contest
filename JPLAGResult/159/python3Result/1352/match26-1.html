class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        ar = list(zip(endTime, startTime, profit))
        ar.sort()
        n = len(ar)
        
        dp = [0] * n
        dp[0] = ar[0][2]
        
        def OK(ar, key):
            start, end = 0, key - 1
            while start < end:
                mid = start + end + 1 >> 1
                if ar[mid][0] <= ar[key][1]:
                    start = mid
                else:
                    end = mid - 1
                    
            return -1 if ar[end][0] > ar[key][1] else end
                    
        
        
for i in range(1, n):
            cur = ar[i][2]
            prev = OK(ar, i)
            # print(prev)
            if prev >= 0:
                cur += dp[prev]
            
            dp[i] = max(cur, dp[i - 1])
        
        # print(dp)
        return dp[n - 1]
