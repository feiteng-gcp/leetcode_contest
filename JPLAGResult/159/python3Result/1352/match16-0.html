class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        data = [(et, st, p) for st, et, p in zip(startTime, endTime, profit)]
        data.sort()
        et = [x for x, _, _ in data]
        st = [x for _, x, _ in data]
        pr = [x for _, _, x in data]
        
        n = len(pr)
        dp = [0] * (n + 1)
        for i in range(1, n + 1):
            j = bisect.bisect(et, st[i - 1])
            dp[i] = max(dp[i - 1], dp[j] + pr[i - 1])
        return dp[n]
