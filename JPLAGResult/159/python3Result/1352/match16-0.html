from collections import defaultdict
from functools import lru_cache
from bisect import bisect
class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        aux = defaultdict(list)
        for s, e, p in zip(startTime, endTime, profit):
            aux[e].append((s, p))
        e_list, prev_e = sorted(endTime), {}
        for s in startTime:
            i = bisect(e_list, s)
            i -= e_list[i] != s
            prev_e[s] = e_list[i] if i >= 0 else 0
        # print(prev_e)
        prev_ee = {}
        tmp = set(e_list)
        tmp = sorted(list(tmp))
        for i,e in enumerate(tmp):
            if i == 0:
                prev_ee[e] = 0
            else:
                prev_ee[e] = tmp[i-1]
        @lru_cache(None)
        def helper(e):
            res = 0
            for s, p in aux[e]:
                res = max(res, p + helper(prev_e[s]), helper(prev_ee[e]) if prev_ee[e] > 0 else 0)
            # print(e, res)
            return res
        
        ans = max(helper(e) for e in sorted(aux.keys()))
        return ans
