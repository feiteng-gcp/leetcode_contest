class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        n = len(startTime)
        jobs = [(startTime[i], endTime[i], profit[i]) for i in range(n)]
        jobs.sort(key=lambda x: x[1])

        dp = [0] * (n + 1)
        end_t = [0] * (n + 1)
        max_dp, end_dp = 0, 0
        for i in range(1, n + 1):
            b, e, p = jobs[i - 1]
            
            if p > max_dp:
                dp[i] = p
                end_t[i] = e
            else:
                dp[i] = max_dp
                end_t[i] = end_dp
            for j in range(i - 1, 0, -1):
                if end_t[j] <= b:
                    if dp[j] + p > dp[i]:
                        dp[i] = dp[j] + p
                        end_t[i] = max(end_t[j], e)
                    break
                elif dp[j] > dp[i]:
                    dp[i] = dp[j]
                    end_t[i] = dp[j]
            if dp[i] > max_dp:
                max_dp = dp[i]
                end_dp = end_t[i]
        return dp[-1]
