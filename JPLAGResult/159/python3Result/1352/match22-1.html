class Solution:
    def jobScheduling(self, s: List[int], e: List[int], p: List[int]) -> int:
        u = [(0,0,0)]
        
for i in range(len(s)):
            u.append((e[i],s[i],p[i]))
        u.sort()
        
        d = {}
        f = [0] * len(u)
        
        for i in range(1,len(u)):
            for j in range(i-1,-1,-1):
                if u[j][0] <= u[i][1]:
                    dd = j
                    break
            f[i] = max(f[i-1],f[j] + u[i][2])
        return f[-1]
