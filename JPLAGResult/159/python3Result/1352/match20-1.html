class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        n = len(startTime)
        jobs = [(startTime[i], endTime[i], profit[i]) for i in range(n)]
        jobs.sort()
        startTime = [jobs[i][0] for i in range(n)]
        dp = [0]*(n+1)
        for i in range(n-1, -1, -1):
            dp[i] = max(dp[i+1], jobs[i][2]+dp[bisect.bisect_left(startTime, jobs[i][1])])

        return dp[0]
