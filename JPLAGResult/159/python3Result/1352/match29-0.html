class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        def returnEndTime(job):
            return job[1]
        jobs = []
        for i in range(len(startTime)):
            jobs.append([startTime[i], endTime[i], profit[i]])
        jobs.sort(key = returnEndTime)
        dp = [0] * len(jobs)
        for i in range(len(jobs)):
            if i == 0:
                dp[i] = jobs[i][2]
            else:
                j = i - 1
                while j >= 0 and jobs[i][0] < jobs[j][1]:
                    j -= 1
                if j == -1:
                    dp[i] = max(jobs[i][2], dp[i-1])
                else:
                    dp[i] = max(jobs[i][2] + dp[j], dp[i-1])
        return dp[len(jobs)-1]
