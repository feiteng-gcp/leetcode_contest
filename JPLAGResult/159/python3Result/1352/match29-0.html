class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        jobs = sorted(zip(startTime, endTime, profit), key=lambda x: x[1])
        if len(jobs) == 0:
            return 0
        ans = [0] * len(endTime)
        ans[0] = jobs[0][2]
        for i in range(1, len(jobs)):
            cur = jobs[i]
            if cur[0] >= jobs[i - 1][1]:
                ans[i] = ans[i - 1] + cur[2]
            else:
                no_overlap = 0
                for j in range(i)[::-1]:
                    if jobs[j][1] <= cur[0]:
                        no_overlap = ans[j]
                        break
                ans[i] = max(ans[i - 1], cur[2] + no_overlap)
        return ans[-1]
