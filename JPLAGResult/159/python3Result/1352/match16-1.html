from collections import defaultdict
from functools import lru_cache
from bisect import bisect
class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        aux = defaultdict(list)
        for s, e, p in zip(startTime, endTime, profit):
            aux[e].append((s, p))
        e_list, prev_e = sorted(endTime), {}
        for s in startTime:
            i = bisect(e_list, s)
            i -= e_list[i] != s
            prev_e[s] = e_list[i] if i >= 0 else 0
        
        ans, cache = 0, {0: 0}
        for e in sorted(aux.keys()):
            for s, p in aux[e]:
                ans = max(ans, p + cache[prev_e[s]])
            cache[e] = ans
        return ans
