class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        datas = sorted(zip(startTime, endTime, profit), key=lambda x:x[1])
        endtimes = sorted(endTime)
        results = [0]
        for t in datas :
            # print(t)
            results_t = max(results[-1], t[-1]+results[bisect.bisect(endtimes, t[0])])
            results.append(results_t)
        # print(results)
        return results[-1]
