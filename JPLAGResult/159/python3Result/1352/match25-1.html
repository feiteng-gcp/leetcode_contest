class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        timeDict= {}
        endSet = set()
        
        for i in range(len(endTime)):
            endSet.add(endTime[i])
            if endTime[i] in timeDict:
                timeDict[endTime[i]].append(i)
            else:
                timeDict[endTime[i]] = [i]
        
        endList = sorted(list(endSet))
        dp = [0] * (endList[-1] + 1)
        for i in range(1, endList[-1] + 1):
            dp[i] = dp[i - 1]
            if i not in timeDict:
                continue
                
            
for idx in timeDict[i]:
                dp[i] = max(dp[i], profit[idx] + dp[startTime[idx]])
        
        return dp[-1]
