class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        jobs = [(e, s, p) for s, e, p in zip(startTime, endTime, profit)]
        jobs.sort()
        n = len(jobs)
        dp = [-1] * n
        dp[0] = jobs[0][2]
        
        def bs(target):
            start, end = 0, n
            
while start != end:
                mid = (start + end) // 2
                if jobs[mid][0] <= target: start = mid + 1
                else: end = mid
            return start
        
        for i in range(1, n):
            idx = bs(jobs[i][1]) - 1
            if idx >= 0: dp[i] = max(dp[i], dp[idx] + jobs[i][2])
            else: dp[i] = max(dp[i], jobs[i][2])
            dp[i] = max(dp[i-1], dp[i])
        return dp[n-1]
