from collections import defaultdict

class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        job_st = defaultdict(list)
        job_end = defaultdict(list)
        
        for k, job in enumerate(zip(startTime, endTime)):
            st_t, end_t = job
            job_st[st_t].append(k)
            job_end[end_t].append(k)
        
        timeline = sorted(endTime+startTime)
        
        start = defaultdict(int)
        end = [0]
        
        
        for t in timeline:
            pr = end[-1]
            for job_i in job_end[t]:
                st_i = startTime[job_i]
                pr = max(pr, profit[job_i]+start[st_i])
            end.append(pr)
            start[t] = pr
        return end[-1]
