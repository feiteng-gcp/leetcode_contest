class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        
        l = []
        for s, e, p in sorted(zip(startTime, endTime, profit), key = lambda x: x[1]):
        
            if not l:
                l.append((e, p))
                continue
                
            idx 
= bisect.bisect(l, (s, float('inf'))) - 1
            if idx >= 0 and l[idx][1] + p > l[-1][1]:
                l.append((e, l[idx][1] + p))
            elif p > l[-1][1]:
                l.append((e, p))
                
        return l[-1][1]
