from bisect import *
class Solution:
    def jobScheduling(self, startTime: List[int], endTime: List[int], profit: List[int]) -> int:
        ### DP
        ### dp[i] first i ending jobs, max profit
        N = len(startTime)
        events = [(startTime[i],endTime[i],profit[i]) for i in range(N)]
        events = sorted(events, key=lambda x: (x[1],x[0]))
        ends = [e[1] for e in events]
        #print(ends)
        #print(events)
        
        dp = [0]*N
        dp[0] = events[0][2]
        for i in range(1,N):
            j = bisect_right(ends,events[i][0])-1
            a = dp[j] if j>=0 else 0
            dp[i] = max(dp[i-1], events[i][2]+a)
        return dp[-1]
