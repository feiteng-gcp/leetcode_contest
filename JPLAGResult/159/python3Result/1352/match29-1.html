class Solution:
    def jobScheduling(self, startTime, endTime, profit) -> int:
        jobs = [x for x in zip(startTime, endTime, profit)]
        
        jobs.sort()
        #print(jobs)
        dp = [0 for _ in range(len(jobs))]
        for i in range(len(jobs) - 1, -1, -1):
            if i < len(jobs) - 1:    
                tmp = jobs[i][2]
                for i1 in range(i + 1, len(jobs)):
                    if jobs[i1]
[0] >= jobs[i][1]:
                        #print(i, "before update", tmp)
                        tmp += dp[i1]
                        #print(i, "update", dp[i1], tmp)
                        break
                #print(i, i1, tmp, profit[i + 1], tmp)
                dp[i] = max(dp[i + 1], tmp)
                
            else:
                dp[i] = jobs[i][2]
                
        #print(dp)
        return dp[0]
