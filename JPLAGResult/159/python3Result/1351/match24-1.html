import logging
import sys
logging.basicConfig(stream=sys.stdout, level=logging.DEBUG)

from collections import Counter, defaultdict
class Solution:
    def balancedString(self, string):
        count = Counter()
        for c in string:
            count[c] += 1
        expected = len(string) / 4
        need = {}
        needCount = 0
        for l in "QWER":
            if count[l] > expected:
                need[l] = count
[l] - expected
                needCount += need[l]

        if needCount == 0:
            return 0
        i=0
        j=0
        ans = len(string)
        while j < len(string) or needCount == 0:
            # print(i, j, need, needCount)
            if needCount > 0:
              j += 1
              c = string[j-1]
              if c not in need:
                  continue
              else:
                  need[c] -= 1
                  if need[c] >= 0:
                      needCount -= 1

            if needCount == 0:
                ans = min(ans, j - i)
                # have necessary,
                i += 1
                c = string[i-1]
                if c in need:
                    need[c] += 1
                    if need[c] > 0:
                        needCount += 1

        return ans
