class Solution:
    def balancedString(self, s: str) -> int:
        from collections import Counter as C, defaultdict as dd
        a = C(s)
        t = len(s)>>2

        if a['Q'] ==  a['W'] ==  a['E'] ==  a['R'] == t: return 0

        b = dd(int)
        for i in 'QWER':
            if a[i]>t:
                a[i] -= t
            else:
                a[i] = 0

        def ok():
            for i in 'QWER':
                if b[i]<a[i]: return False
            return True

        l = 0
        b[s[0]] = 1
        if ok(): return 1
        ans = len(s)
        for r in range(1, len(s)):
            b[s[r]] += 1
            if ok():
                while ok():
                    ans = min(ans, r-l+1)
                    b[s[l]] -= 1
                    l += 1

        return ans
