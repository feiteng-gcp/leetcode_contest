class Solution:
    def balancedString(self, s: str) -> int:
        counter = collections.Counter(s)
        for k in counter:
            counter[k] -= len(s) // 4
        if all(v == 0 for v in counter.values()):
            return 0
        ans = len(s) - 1
        left = 0
        for i, v in enumerate(s):
            counter[v] -= 1
            while left <= i and all(v <= 0 for v in counter.values()):
                ans = min(ans, i - left + 1)
                counter[s[left]] += 1
                left += 1
        return ans
