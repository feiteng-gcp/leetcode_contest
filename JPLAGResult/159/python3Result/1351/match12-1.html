class Solution:
    def balancedString(self, s: str) -> int:
        excess = {'Q': 0, 'W': 0, 'E': 0, 'R': 0}
        for c in s:
            excess[c] += 1
        
for k in excess:
            excess[k] -= len(s) // 4
            
        if all(v == 0 for v in excess.values()):
            return 0
            
        result = len(s)
        window_count = {'Q': 0, 'W': 0, 'E': 0, 'R': 0}
        l = 0
        for r in range(len(s)):
            letter = s[r]
            window_count[letter] += 1
            while l <= r and all(window_count[letter] >= excess[letter] for letter in ['Q', 'W', 'E', 'R']):
                result = min(result, r - l + 1)
                window_count[s[l]] -= 1
                l += 1
        return result
