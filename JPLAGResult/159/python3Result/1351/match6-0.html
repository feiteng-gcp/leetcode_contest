class Solution:
    def balancedString(self, s: str) -> int:
        n = len(s)
        expect = n//4
        count = [[0]*n for i in range(4)]
        self.count = count
        for i, c in enumerate(s):
            c = self.ctoi(c)
            if i == 0:
                count[c][i] = 1
            else:
                count[0][i] = count[0][i-1]
                count[1][i] = count[1][i-1]
                count[2][i] = count[2][i-1]
                count[3][i] = count[3][i-1]
                count[c][i] += 1
        #print(count)
        if (count[0][-1] == expect 
            and count[1][-1] == expect 
            and count[2][-1] == expect 
            and count[3][-1] == expect):
            return 0
        result = n
        for i in range(n):
            right = (count[0][-1] - count[0][i], 
                count[1][-1] - count[1][i],
                count[2][-1] - count[2][i],
                count[3][-1] - count[3][i])
            if max(right) > expect:
                #not work even remove [0 .. i]
                continue
            left = self.binary(i, right)
            #print(s,i,left)
            result = min(result, i-left)
        return result

    def binary(self, i, right):
        count = self.count
        expect = len(count[0])//4
        lo, hi = -1, i
        while lo < hi:
            mid = (lo+hi+1)//2
            if (count[0][mid] + right[0] <= expect
                and count[1][mid] + right[1] <= expect
                and count[2][mid] + right[2] <= expect
                and count[3][mid] + right[3] <= expect):
                lo = mid
            else:
                hi = mid - 1
        return lo
            
    def ctoi(self, c):
        if c=='Q': return 0
        if c=='W': return 1
        if c=='E': return 2
        if c=='R': return 3
