class Solution:
    def balancedString(self, s: str) -> int:
        c = collections.Counter(s)
        should = len(s)//4
        res = 999999999
        needM = 0
        for i in c:
            if c[i]>should:
                c[i] = (c[i]- should)
                needM +=1
            else:
                c[i] = 0
        start  = 0
        meet = 0
        
        for i in range(len(s)):
            v = s[i]
            c[v]-=1
            #print(c[v],needM)
            if c[v] == 0:
                meet+=1
            while start<len(s) and needM==meet:
                res = min(res,i-start+1)
                v2 = s[start]
                c[v2]+=1
                if c[v2] == 1:
                    meet-=1
                start+=1
        return res
