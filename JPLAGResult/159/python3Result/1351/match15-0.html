class Solution:
    def balancedString(self, s: str) -> int:
        left = 0
        count = collections.Counter(s)
        limit = len(s) // 4
        needs = collections.defaultdict(int)
        for c in count:
            needs[c] = max(0, count[c] - limit)
        res = len(s)
        totalneeds = sum(needs.values())
        if totalneeds == 0: return 0
        for right in range(0, len(s)):
            c = s[right]
            needs[c] -= 1
            if needs[c] >= 0: totalneeds -= 1
            while totalneeds == 0:
                res = min(res, right - left + 1)
                needs[s[left]] += 1
                if needs[s[left]] > 0: totalneeds += 1
                left += 1
        return res
