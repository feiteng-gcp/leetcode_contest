class Solution:
    def balancedString(self, s: str) -> int:
        cnt = collections.Counter([i for i in s])
        for i in cnt:
            if cnt[i] > len(s)//4:
                cnt[i] -= len(s)//4
            else:
                cnt[i] = 0
        #print([(key,cnt[key]) for key in cnt])

        if sum(cnt.values()) == 0: return 0
        cur = {'Q':0,'W':0,'E':0,'R':0}
        res = len(s)
        cur[s[0]] = 1
        i ,j = 0, 0
        while i <= j < len(s):
            if cur[s[i]] > cnt[s[i]]:
                cur[s[i]] -= 1
                i += 1
                if j < i:
                    j += 1
                    if j < len(s):
                        cur[s[j]] += 1
            else:
                j += 1
                if j < len(s):
                    cur[s[j]] += 1
            flag = True
            #print([(key,cur[key]) for key in cur])
            for key in cur:
                if cur[key] < cnt[key]:
                    flag = False
            if flag:
                res = min(res,j-i+1)
        return res
