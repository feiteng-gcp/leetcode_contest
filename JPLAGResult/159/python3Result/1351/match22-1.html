class Solution:
    def balancedString(self, s: str) -> int:
        counter=collections.Counter(s)
        target=len(s)//4
        d={}
        for key in counter:
            if counter[key]>target:
                d[key]=counter[key]-target
        if not d:
            return 0
        curd=collections.defaultdict(int)
        res=len(s)
        slow=0
        for i in range(len(s)):
            curd
[s[i]]+=1
            while all(curd[key]>=d[key] for key in d):
                res=min(res,i-slow+1)
                curd[s[slow]]-=1
                slow+=1
        return res
