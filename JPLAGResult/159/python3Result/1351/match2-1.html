from collections import Counter
class Solution:
    def balancedString(self, s: str) -> int:
        n, cnt = len(s), Counter(s)
        cnt = {k: v - n // 4 for k, v in cnt.items() if v > n // 4}
        if len(cnt) == 0: return 0
        l, r, win, ans = 0, 0, Counter(), n
        while r < n:
            while r < n and any(win
[c] < cnt[c] for c in cnt):
                win[s[r]] += 1
                r += 1
            while l < r and (s[l] not in cnt or win[s[l]] > cnt[s[l]]):
                win[s[l]] -= 1
                l += 1
            if all(win[c] >= cnt[c] for c in cnt):
                ans = min(ans, r - l)
            if l < n:
                win[s[l]] -= 1
                l += 1
        return ans
