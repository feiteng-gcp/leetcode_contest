class Solution:
    def balancedString(self, s: str) -> int:
        n = len(s)
        cnt = {'Q': 0, 'W': 0, 'E': 0, 'R': 0}
        from collections import Counter
        allcnt = Counter(s)
        req = {c: allcnt[c] - n // 4 if allcnt[c] > n // 4 else 0 for c in 'QWER'}
        if sum(req.values()) == 0:
            return 0
        begin = 0
        ans = int(1e9)
        for end, c in enumerate(s):
            cnt[c] += 1
            for cc in 'QWER':
                if req[cc] > cnt[cc]:
                    break
            else:
                ans = min(ans, end - begin + 1)
                while True:
                    begin += 1
                    cnt[s[begin - 1]] -= 1
                    if cnt[s[begin - 1]] < req[s[begin - 1]]:
                        break
                    else:
                        ans = min(ans, end - begin + 1)
        return ans
