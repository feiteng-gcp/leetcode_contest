from collections import Counter, defaultdict

class Solution:
    def balancedString(self, s: str) -> int:
        c = Counter(s)
        n = len(s)
        
        Q = max(0, c['Q'] - n // 4)
        W = max(0, c['W'] - n // 4)
        R = max(0, c['R'] - n // 4)
        E= max(0, c['E'] - n // 4)
        
        d = defaultdict(int)
        l, r 
= 0, 0
        ans = n
        while l < n:
            while r < n and (d['Q'] < Q or d['W'] < W or d['R'] < R or d['E'] < E):
                d[s[r]] += 1
                # print(d)
                r += 1
            # print(l,r,ans)
            if d['Q'] >= Q and d['W'] >= W and d['R'] >= R and d['E'] >= E:
                ans = min(ans, r-l)
            d[s[l]] -= 1
            l += 1
        return ans
