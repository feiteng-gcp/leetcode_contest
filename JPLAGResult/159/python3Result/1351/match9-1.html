class Solution:
    def balancedString(self, s: str) -> int:
        n = len(s)
        d = collections.Counter(s)
        for c in d:
            if d[c] > n // 4:
                d[c] -= n // 4
            else:
                d[c] = 0
        req = sum(d.values())

        if req == 0:
            return 0

        i = 0
        res = float('inf')
        for j in range
(n):
            d[s[j]] -= 1
            if d[s[j]] >= 0:
                req -= 1
            while req == 0:
                res = min(res, j - i + 1)
                d[s[i]] += 1
                if d[s[i]] > 0:
                    req += 1
                i += 1
        return res
