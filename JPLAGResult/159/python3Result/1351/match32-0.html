class Solution:
    def balancedString(self, s: str) -> int:
        # dq=3,dw=2,de=5,dr=-10
        # find the shortest substring contains at least 3 q, 2w, 5e
        n=len(s)
        f=[[0 for _ in range(n+1)] for _ in range(4)]
        for i,c in enumerate(s):
            v=0 if c=='Q' else 1 if c=='W' else 2 if c=='E' else 3
            for j in range(4):
                f[j][i+1]=f[j][i]
            f[v][i+1]=f[v][i]+1
        d=[(f[v][n]-n//4) for v in range(4)]
        def work(W):
            for i in range(n):
                if i+W>n: break
                failed=False
                for v in range(4):
                    if f[v][i+W]-f[v][i]<d[v]:
                        failed=True
                        break
                if not failed: return True
            return False
        L=-1
        R=n
        while L+1<R:
            M=(L+R)//2
            if work(M):
                R=M
            else:
                L=M
        return R
