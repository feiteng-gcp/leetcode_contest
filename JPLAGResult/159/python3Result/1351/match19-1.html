class Solution:
    def balancedString(self, s: str) -> int:
        n = len(s)
        n4 = int(n/4)
        count = collections.Counter(s)
        if count['Q'] <= n4 and count['W'] <= n4 and count['E'] <= n4 and count['R'] <= n4:
            return 0
        # print(count)
        
        st, en = 0, 0
        min_len = n
        while en < n:
            while not (count['Q'] <= n4 and count['W'] <= n4 and count
['E'] <= n4 and count['R'] <= n4) and en < n:
                count[s[en]] -= 1
                en += 1
                     
            while (count['Q'] <= n4 and count['W'] <= n4 and count['E'] <= n4 and count['R'] <= n4) and st < en:
                min_len = min(min_len, en-st)
                count[s[st]] += 1
                st += 1
                
        return min_len
