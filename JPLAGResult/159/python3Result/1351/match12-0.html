import collections

class Solution:
    def balancedString(self, s: str) -> int:
        target = len(s) // 4
        c = collections.Counter(s)
        for char in c:
            c[char] -= target
        c = +c
        if not c:
            return 0
        ans = len(s)
        i = 0
        for j in range(len(s)):
            c[s[j]] -= 1
            while all(n <= 0 for n in c.values()):
                ans = min(ans, j - i + 1)
                c[s[i]] += 1
                i += 1
        return ans
