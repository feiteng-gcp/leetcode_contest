class Solution:
    def balancedString(self, s: str) -> int:
        from collections import Counter
        d = Counter(s)
        # print(d)
        
        q = d.get('Q', 0)
        w = d.get('W', 0)
        e = d.get('E', 0)
        r = d.get('R', 0)
        
        n = len(s)
        n //= 4
        
        q = max(0, q - n)
        w = max(0, w - n)
        e = max(0, e - n)
        r = max(0, r - n)
        
        # print(q, w, e, r)
        
        d = dict()
        d['Q'] = q
        d['W'] = w
        d['E'] = e
        d['R'] = r
        
        n *= 4
        
        ans = 10 ** 9
        
        if q == w == e == r == 0:
            return 0
        
        i, j = 0, 0
        while i < n:
            # print(d, i, j)
            while d['Q'] <= 0 and d['W'] <= 0 and d['E'] <= 0 and d['R'] <= 0:
                d[s[j]] += 1
                j += 1
                ans = min(ans, i - j + 1)
            d[s[i]] -= 1
            i += 1
        
        while j < n and d['Q'] <= 0 and d['W'] <= 0 and d['E'] <= 0 and d['R'] <= 0:
            ans = min(ans, n - 1 - j + 1)
            d[s[j]] += 1
            j += 1
        
        return ans
