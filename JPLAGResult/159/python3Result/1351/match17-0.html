from collections import Counter
class Solution:
    def balancedString(self, s: str) -> int:
        count = Counter(s)
        badD = {c:(count[c] - (len(s)//4)  if count[c] > len(s)//4 else 0) for c in "QWER"}
        l = 0
        r = 0
        curCount = {c:0 for c in "QWER"}
        ans = len(s)
        while True:
            if all(curCount[c] >= badD[c] for c in "QWER"):
                ans = min(ans, r-l)
                if (l == len(s)):
                    break
                curCount[s[l]] -= 1
                l += 1
            else:
                if r == len(s):
                    break
                curCount[s[r]] += 1
                r += 1
        return ans
