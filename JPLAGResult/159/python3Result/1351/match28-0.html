class Solution:
    def balancedString(self, s: str) -> int:
        length = len(s)
        dic1 = {}
        for char in s:
            if char not in dic1:
                dic1[char] =1
            else:
                dic1[char] +=1
        
        dic2 = {}
        sum_val = 0
        for key in dic1:
            if dic1[key] > length//4:
                dic2[key] = dic1[key] - length//4
                sum_val += dic2[key]
        
        if not dic2:
            return 0
        
        min_length = float('inf')
        l = 0
        r = 0
        while r < length:
            if s[r] in dic2:
                if dic2[s[r]] >0:
                    sum_val -=1
                dic2[s[r]] -=1
            r +=1
            if sum_val ==0:
                while sum_val ==0 and l<r:
                    if s[l] in dic2:
                        if dic2[s[l]] == 0:
                            sum_val +=1
                        dic2[s[l]] +=1
                    l +=1
                min_length = min(min_length, r-l +1)
        return min_length
