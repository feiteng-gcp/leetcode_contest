class Solution:
    def balancedString(self, s: str) -> int:
        average_n = len(s) // 4
        counter = collections.Counter(s)
        dis = {k:0 if counter[k]<=average_n else counter[k]-average_n for k in 'QWER'}
        if sum(dis.values()) == 0 :
            return 0
        
        temp_counter = {k:0 for k in 'QWER'}
        left, right = 0, 0
        temp_counter[s[0]] += 1
        mint = 1e6
        while True :
            while not all([temp_counter[k]>=dis[k] for k in 'QWER']) :
                right += 1
                if right >= len(s) :
                    break
                temp_counter[s[right]] += 1
            if right >= len(s) :
                break
            
while temp_counter[s[left]] > dis[s[left]] :
                temp_counter[s[left]] -= 1
                left += 1
            mint = min(mint, right-left+1)
            temp_counter[s[left]] -= 1
            left += 1
        # print(left, right, temp_counter)
        return mint
