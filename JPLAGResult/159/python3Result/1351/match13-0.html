class Solution:
    def balancedString(self, s: str) -> int:
        n = len(s) // 4
        c = collections.Counter(s)
        for x in c:
            c[x] -= n
        needed = 0
        for x in c:
            if c[x] > 0:
                needed += 1
        if needed == 0:
            return 0
        
        i = j = 0
        best = len(s)
        
        while j < len(s):
            x = s[j]
            c[x] -= 1
            if c[x] == 0:
                needed -= 1
            j += 1
            while needed <= 0:
                x = s[i]
                best = min(best,j-i)
                c[x] += 1
                if c[x] == 1:
                    needed += 1
                i += 1
        return best
