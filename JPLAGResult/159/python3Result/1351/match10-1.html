class Solution:
    def balancedString(self, s: str) -> int:
        has=dict()
        l=0
        r
=1
        has['Q']=0
        has['W']=0
        has['E']=0
        has['R']=0
        for c in s:
            has[c]+=1
        if(has['Q']==len(s)/4 and has['W']==len(s)/4 and has['E']==len(s)/4 and has['R']==len(s)/4):
            return 0
        vis=dict()
        vis['Q']=0
        vis['W']=0
        vis['E']=0
        vis['R']=0
        vis[s[0]]=1
        def legal():
            return has['Q']-vis['Q']<=len(s)/4 and has['W']-vis['W']<=len(s)/4 and has['E']-vis['E']<=len(s)/4 and has['R']-vis['R']<=len(s)/4
        ans=len(s)
        while(r<len(s)):
            vis[s[r]]+=1
            if not legal():
                r+=1
            else:
                while(l<=r and legal()):
                    ans=min(ans,r-l+1)
                    vis[s[l]]-=1
                    l+=1
                r+=1
            
        return ans
