class Solution:
    def balancedString(self, s: str) -> int:
        cnt  = -len(s) // 4
        d ={"Q":cnt,"W":cnt,"E":cnt,"R":cnt}
        for x in s:
            d[x] += 1
        nd = {}
        for x in d:
            if d[x] > 0:
                nd[x] = d[x]
        if not nd:
            return 0
        def pas(m):
            cu = {"Q":0,"W":0,"E":0,"R":0}
            
            for i in range(len(s)):
                if i < m:
                    cu[s[i]] += 1
                else:
                    cu[s[i]] += 1
                    cu[s[i-m]] -= 1
                p = 1
                for x in nd:
                    if cu[x] < nd[x]:
                        p = 0
                        continue
                if p:
                    
return True
            return False
        
        l = 1
        r = len(s)
        while l < r:
            m = (l+r)//2
            if pas(m):
                r = m
            else:
                l = m + 1
        return r
