class Solution:
    def balancedString(self, s: str) -> int:
        cc = collections.Counter(s)
        n4 = len(s)//4
        for char in cc:
            cc[char] = max(0, cc[char] - n4)
        right = 0
        n 
= len(s)
        sol = n

        def is_any_less(cc1, cc2):
            for k in cc2:
                if cc1[k] < cc2[k]:
                    return True
            return False
        cur_cc = collections.Counter()
        for left in range(n):
            while right < n and is_any_less(cur_cc, cc):
                cur_cc[s[right]] += 1
                right += 1

            if not is_any_less(cur_cc, cc):
                sol = min(sol, right-left)
            cur_cc[s[left]] -= 1
        return sol
