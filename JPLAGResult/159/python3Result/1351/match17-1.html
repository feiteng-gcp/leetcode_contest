class Solution:
    def balancedString(self, s: str) -> int:
        cnt = collections.Counter(s)
        
        def check_bal():
            for k, v in cnt.items():
                if v > len(s) // 4:
                    return False
            return True
        
        res = float('inf')
        i, j = 0, 0
        while i <= len(s) and j <= len(s):
            if not check_bal
():
                if j == len(s):
                    break
                cnt[s[j]] -= 1
                j += 1
            else:
                if i == len(s):
                    break
                # print(i, j, res)
                res = min(res, j - i)
                cnt[s[i]] += 1
                i += 1
        return res
