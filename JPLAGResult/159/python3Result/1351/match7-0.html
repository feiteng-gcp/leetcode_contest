class Solution:
    def balancedString(self, s):
        t = len(s)//4
        x = len(s)

        d = {}
        d['Q'] = 0
        d['W'] = 0
        d['E'] = 0
        d['R'] = 0
        for i in s:
            d[i]+=1
        if d['Q'] == t and d['W'] == t and d['E'] == t and d['R'] == t:
            return 0
        u = 0
        l = 0
        minl= x
        while(u<x and l<x):
            while(d['Q'] > t or d['W']> t or d['E']> t or d['R']>t) and u<x:
                d[s[u]] -=1
                u +=1
            while (d['Q'] <= t and d['W'] <= t and d['E'] <= t and d['R'] <= t):
                d[s[l]] += 1
                l += 1
            minl = min(u-l+1, minl)
        return minl
