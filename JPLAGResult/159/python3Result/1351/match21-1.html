class Solution:
    def balancedString(self, s: str) -> int:
        count = collections.Counter(s)
        n = len(s)
        rem = {}
        for key, val in count.items():
            if val> n//4: rem[key] = val - n//4
        if not rem: return 0
        low , high = 0, 0
        curr = {}
        cnt = 0
        
        res = n
        target = len(rem.keys())
        #print(target)
        while high< n:
            while high< n and cnt < target:
                char = s[high]
                if char in rem:
                    curr[char] = curr.get(char, 0) + 1
                    if curr[char] == rem[char]: cnt += 1
                high += 1
            while low<high and cnt>= target:
                char = s[low]
                if char in rem:
                    curr[char] -= 1
                    if curr[char] == rem[char] - 1: cnt -= 1
                low += 1
            res = min(res, high-low+1)
            #print(high, low)
        return res
