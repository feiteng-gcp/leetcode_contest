from collections import Counter
class Solution:
    def balancedString(self, s: str) -> int:
        n = len(s)
        counter = Counter(s)
        match = 0
        wanted 
= {}
        for c in "QWER":
            if counter[c] > n // 4:
                wanted[c] = counter[c] - n // 4
        if not wanted:
            return 0
        
        ret = float("inf")
        j = 0
        for i in range(len(s)):
            while j < len(s) and match != len(wanted):
                if s[j] in wanted:
                    wanted[s[j]] -= 1
                    if wanted[s[j]] == 0:
                        match += 1
                j += 1
            # print(i, j)
            if match == len(wanted):
                ret = min(ret, j - i)
            if s[i] in wanted:
                wanted[s[i]] += 1
                if wanted[s[i]] == 1:
                    match -= 1
        return ret
