from collections import Counter

class Solution:
    def balancedString(self, s: str) -> int:
        n = len(s)
        imb = Counter(s)
        imb_tot = 0
        for c in list(imb.keys()):
            imb[c] -= int(n / 4)
            if imb[c] < 0:
                del imb[c]
            else:
                imb_tot += imb[c]
        if imb_tot == 0:
            return 0
        
        min_len = n
        l, r = 0, 0
        freq = Counter()
        while r < len(s):
            freq[s[r]] += 1
            while all(freq[c] >= imb[c] for c in imb.keys()):
                min_len 
= min(min_len, r - l + 1)
                freq[s[l]] -= 1
                l += 1
            r += 1
        return min_len
