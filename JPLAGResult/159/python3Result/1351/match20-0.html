class Solution:
    def balancedString(self, s: str) -> int:
        from collections import Counter
        allCounter = Counter(s)
        n = len(s)
        x = n // 4
        
        def check(subCounter):
            for k in allCounter:
                if allCounter[k] - subCounter[k] > x:
                    return False
            return True

        low, high = 0, n
        while low < high:
            mid = (high - low) // 2 + low
            checkOk = False
            if mid > 0:
                counter = Counter(s[:mid])
                for i in range(n - mid + 1):
                    if check(counter):
                        checkOk = True
                        break
                    counter[s[i]] -= 1
                    if i + mid < n:
                        counter[s[i + mid]] += 1
            else:
                checkOk = check(Counter())
            if checkOk:
                high = mid
            else:
                low = mid + 1
        return low
