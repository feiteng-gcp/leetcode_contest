class Solution:
    def balancedString(self, s: str) -> int:
        N = len(s)
        T = N >> 2

        def valid(l):
            counter = collections.Counter(s[l:])
            for i, c in enumerate(s[l:]):
                if max(counter.values()) <= T:
                    return True
                counter[c] -= 1
                counter[s[i]] += 1
            return max(counter.values()) <= T

        lo, hi 
= 0, N
        while lo < hi:
            mid = lo + hi >> 1
            if valid(mid):
                hi = mid
            else:
                lo = mid + 1
        return lo
