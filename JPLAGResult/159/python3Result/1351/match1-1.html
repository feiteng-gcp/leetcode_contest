class Solution:
    def balancedString(self, gene: str) -> int:
        dic = {'Q':0,'W':0,'E':0,'R':0}
        for i in gene:
            dic[i]+=1
        x=len(gene)
        factor=x/4

        if dic['Q']==factor and dic['W']==factor and dic['E']==factor and dic['R']==factor:
            return 0

        upper=0
        lower=0
        minlen=x
        while upper<x and lower<x:
            while (dic['Q']>factor or dic['W']>factor or dic
['E']>factor or dic['R']>factor) and upper<x:
                dic[gene[upper]]-=1
                upper+=1
            while dic['Q']<=factor and dic['W']<=factor and dic['E']<=factor and dic['R']<=factor:
                dic[gene[lower]]+=1
                lower+=1
            if upper - lower < minlen :
                minlen=upper-lower+1
        return minlen
