import collections
class Solution:
    def balancedString(self, s: str) -> int:
        n = len(s)
        count = collections.Counter(s)
        letters = {ch: count[ch] - n/4 for ch in count if count[ch] - n/4 > 0}
        if len(letters) == 0:
            return 0
        l, r = 0, 0
        window = collections.Counter()
        mini = float('inf')
        while r < n:
            window[s[r]] += 1
            while self.valid(window, letters):
                mini = min(mini, r - l + 1)
                window[s[l]] -= 1
                l += 1
            r += 1
            
        return mini
    
    def valid(self, window, letters):
        for ch in letters:
            if letters[ch] > window[ch]:
                return False
        return True
