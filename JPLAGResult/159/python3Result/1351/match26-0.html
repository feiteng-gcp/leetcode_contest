class Solution:
    def balancedString(self, s: str) -> int:
        cnt_s = collections.Counter(s)
        standard = len(s) // 4
        
        extra = {}
        for k, v in cnt_s.items():
            if v > standard:
                extra[k] = v - standard
                
        if len(extra) == 0:
            return 0
        
        d = collections.defaultdict(int)
        l = 0
        cnt = 0
        res = float('inf')
        for j, c in enumerate(s):
            if c in extra:
                d[c] += 1
                if d[c] == extra[c]:
                    cnt += 1
            if cnt == len(extra):
                while s[l] not in d or d[s[l]] > extra[s[l]]:
                    if s[l] in d:
                        d[s[l]] -= 1
                    l += 1
                res = min(res, j - l + 1)
                d[s[l]] -= 1
                l += 1
                cnt -= 1
        return res
