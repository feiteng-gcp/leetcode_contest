class Solution:
    def balancedString(self, s: str) -> int:
        avg = len(s) // 4
        freq = {"Q": 0, "W": 0, "E": 0, "R": 0}
        for c in s:
            freq[c] += 1
        bal = 0
        for c in freq:
            if freq[c] - avg > 0:
                bal += 1
                freq[c] -= avg
            else:
                freq[c] = 0
        if bal == 0:
            return 0
        ans = 0x7fffff
        start = 0
        for end in range(len(s)):
            freq[s[end]] -= 1
            if freq[s[end]] == 0:
                bal -= 1
            while bal == 0:
                ans = min(ans, end - start + 1)
                freq[s[start]] += 1
                if freq[s[start]] == 1:
                    bal += 1
                start += 1
        return ans
