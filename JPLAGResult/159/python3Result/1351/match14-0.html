class Solution:
    def balancedString(self, s: str) -> int:
        c = collections.Counter(s)
        n = len(s)
        e = n // 4
        for ch in c:
            c[ch] = max(0, c[ch] - e)
        if all(c[k] == 0 for k in 'QWER'):
            return 0
        i = 0
        j = 0
        d = collections.Counter()
        ret = n
        while j < n:
            d[s[j]] += 1
            while all(d[k] >= c[k] for k in 'QWER') and i <= j:
                ret = min(ret, j - i + 1)
                d[s[i]] -= 1
                i += 1
            j += 1
        return ret
