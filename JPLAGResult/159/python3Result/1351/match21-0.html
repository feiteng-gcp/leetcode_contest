class Solution:
    def balancedString(self, s: str) -> int:
        gene = s
        min_length_string = len(gene)
     
        occurences = dict()
        occurences['Q'] = 0
        occurences['W'] = 0
        occurences['E'] = 0
        occurences['R'] = 0

        expected = len(gene) // 4

        for g in gene:
            occurences[g] += 1

        for x in occurences:
            occurences[x] = max(0, occurences[x] - expected)

        if occurences['Q'] == 0 and occurences['W'] == 0 and occurences['E'] == 0 and occurences['R'] == 0:
            return 0

        found = dict()
        found['Q'] = 0
        found['W'] = 0
        found['E'] = 0
        found['R'] = 0

        tail = 0
        head = 0

        while head != len(gene):
            found[gene[head]] += 1
            if found['Q'] >= occurences['Q'] and \
            found['W'] >= occurences['W'] and \
            found['E'] >= occurences['E'] and \
            found['R'] >= occurences['R']:
                # this is a valid candidate
                min_length_string = min(min_length_string, head-tail+1)

                # try to shorten it
                while found[gene[tail]] > occurences[gene[tail]]:
                    found[gene[tail]] -= 1
                    tail += 1
                    min_length_string = min(min_length_string, head-tail+1)


            head += 1

        return min_length_string
