class Solution {
public:
    int balancedString(string s) {

        int n = s.size();
        vector<vector<int>> qwer(n+1, vector<int>(4));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < 4; ++j)

                qwer[i+1][j] = qwer[i][j];
            if (s[i] == 'Q') {
                ++qwer[i+1][0];
            } else if (s[i] == 'W') {
                ++qwer[i+1][1];
            } else if (s[i] == 'E') {
                ++qwer[i+1][2];
            } else
                ++qwer[i+1][3];
        }
        int low = 0, high = n;
        while (low < high) {
            int mid = (low + high) / 2;
            bool found = false;
            for (int i = mid; i <= n; ++i) {
                bool ok = true;
                for (int j = 0; j < 4; ++j) {
                    if (qwer[n][j] - (qwer[i][j] - qwer[i-mid][j]) > n/4)
                        ok = false;
                }
                if (ok) {
                    found = true;
                    break;
                }
            }
            if (found)
                high = mid;
            else
                low = mid + 1;
        }
        return low;
    }
};
