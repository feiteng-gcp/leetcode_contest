class Solution {
public:
    int balancedString(string s) {
        int n = s.length();
        map<char, int> idx;
        idx['Q'] = 0;
        idx['W'] = 1;
        idx['E'] = 2;
        idx['R'] = 3;
        
        vector<vector<int>> sum(n, vector<int>(4));
        
        sum[0][idx[s[0]]] = 1;
        for (int i = 1; i < n; ++i) {
            sum[i] = sum[i-1];
            sum[i][idx[s[i]]]++;
        }
        
        int t = n / 4;
        if (sum[n-1][0] == t && sum[n-1][1] == t && sum[n-1][2] == t)
            return 0;
        

        int l = 1, r = n-1;
        while (l < r) {
            int mid = l + ((r-l) >> 1);
            bool ok = false;
            
            for (int i = 0; i + mid <= n; ++i) {
                bool tmp = true;
                for (int j = 0; j < 4; ++j) {
                    int cur = (i > 0 ? sum[i-1][j] : 0) + sum[n-1][j] - sum[i+mid-1][j];
                    if (cur > t) {
                        tmp = false;
                        break;
                    }

                }
                
                if (tmp) {
                    ok = true;
                    break;
                }
            }
            
            if (ok)
                r = mid;
            else
                l = mid+1;
        }
        
        return l;
    }
};
