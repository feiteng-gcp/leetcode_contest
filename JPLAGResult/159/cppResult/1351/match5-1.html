#include<bits/stdc++.h>
using namespace std;

const int nax = 1e6 + 5;
const char W[5] = "QWER";
int n, cnt[4];
string sl;
bool ok() {
    for(int i = 0; i < 4; ++i)
        if(cnt[i] > n / 4)
            return false;
    return true;
}
int f(int i) {
    for(int j = 0; j < 4; ++j)
        if(sl[i] == W[j])
            return j;
    assert(false);
}
int solve(){
    for(int i = 0; i < 4; i++) cnt[i] = 0;
    for(int i = 0; i < n; ++i)
        ++cnt[f(i)];
    if(ok()) {
        return 0;
    }
    int ans = n;
    // two pointers i, j
    int j = 0;
    for(int i = 0; i < n; ++i) {
        while(j < n && !ok())
            --cnt[f(j++)];
        if(ok()) ans = min(ans, j - i);
        ++cnt[f(i)];
    }
    return ans;
}
class Solution {
public:
    int balancedString(string s) {
        sl = s;
        n = s.size();
        return solve();
    }
};
