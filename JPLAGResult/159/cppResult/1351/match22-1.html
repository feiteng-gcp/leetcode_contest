class Solution {
public:
    int balancedString(string s) {
        
        vector<int> vt(4, 0);
        

        for(int i = 0; i < s.size(); i++) {
            if(s[i] == 'Q') vt[0]++;
            if(s[i] == 'W') vt[1]++;
            if(s[i] == 'E') vt[2]++;
            if(s[i] == 'R') vt[3]++;
        }
        
        int x = (int)(s.size())/4;
        
        vt[0] = max(0, vt[0] - x);
        vt[1] = max(0, vt[1] - x);
        vt[2] = max(0, vt[2] - x);
        vt[3] = max(0, vt[3] - x);
        
        //cout<<vt[0]<<" "<<vt[1]<<" "<<vt[2]<<" "<<vt[3]<<endl;
        
        vector<int> cnt(4, 0);
        
        int l = -1;

        int ans = 100000000;
        for(int r = 0; r < s.size(); r++) {
            if(s[r] == 'Q') cnt[0]++;
            if(s[r] == 'W') cnt[1]++;
            if(s[r] == 'E') cnt[2]++;
            if(s[r] == 'R') cnt[3]++;
            
            while((l < r) && (vt[0] == 0 || cnt[0] >= vt[0]) && (vt[1] == 0 || cnt[1] >= vt[1]) && (vt[2] == 0 || cnt[2] >= vt[2]) && (vt[3] == 0 || cnt[3] >= vt[3])) {
                ans = min(ans, (r-l));
                
                l++;
                if(s[l] == 'Q') cnt[0]--;
                if(s[l] == 'W') cnt[1]--;
                if(s[l] == 'E') cnt[2]--;
                if(s[l] == 'R') cnt[3]--;
            }
        }
        
        if(vt[0] == 0 && vt[1] == 0 && vt[2] == 0 && vt[3] == 0) {
            return 0;
        }
        return ans;
        
        
    }
};
