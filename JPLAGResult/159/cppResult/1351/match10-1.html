class Solution {
public:
    int balancedString(string s) {
        int n = len(s);
        vector<int> Q(n + 1), W(n + 1), E(n + 1), R(n + 1);  // prefix sum
        vector<int> q(n + 1), w(n + 1), e(n + 1), r(n + 1);  // suffix sum
        rep(i, n) {
            Q[i + 1] = Q[i] + (s[i] == 'Q');
            W[i + 1] = W[i] + (s[i] == 'W');
            E[i + 1] = E[i] + (s[i] == 'E');
            R[i + 1] = R[i] + (s[i] == 'R');
        }
        repi(i, n) {
            q[i] = q[i + 1] + (s[i] == 'Q');
            w[i] = w[i + 1] + (s[i] == 'W');
            e[i] = e[i + 1] + (s[i] == 'E');
            r[i] = r[i + 1] + (s[i] == 'R');            
        }
        
        int target = n / 4;

        int left = 0, right = n;
        while(left < right) {
            int size = (left + right) / 2;
            
            bool succ = false;
            for(int begin = 0, end = size; end <= n; begin++, end++) {
                int sq = Q[begin] + q[end], sw = W[begin] + w[end], se = E[begin] + e[end], sr = R[begin] + r[end];

                if(sq <= target and sw <= target and se <= target and sr <= target) {
                    succ = true;
                    break;
                }
            }
            
            if(succ) right = size;
            else left = size + 1;
        }
        return left;
    }
};
