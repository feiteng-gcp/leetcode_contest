class Solution {
public:
    int balancedString(string s) {
        int Qc = 0, Wc = 0, Ec = 0, Rc = 0;

        for (auto c : s) {
            if (c == 'Q') ++Qc;
            else if (c == 'W') ++Wc;
            else if (c == 'E') ++Ec;
            else ++Rc;
        }
        int total = s.size();
        int target = total / 4;
        if (Qc == target && Wc == target && Ec == target && Rc == target) return 0;
        int start = 0, end = total;
        while (start < end) {
            int mid = (start + end) / 2;
            int Qr = Qc, Wr = Wc, Er = Ec, Rr = Rc;
            int i = 0, j = 0;

            while (j < mid) {
                char c = s[j];
                if (c == 'Q') --Qr;
                else if (c == 'W') --Wr;
                else if (c == 'E') --Er;
                else --Rr;
                ++j;
            }
            bool pass = false;
            if (Qr <= target && Wr <= target && Er <= target && Rr <= target) {

                pass = true;
            }
            while (!pass && j < s.size()) {
                char c = s[j];
                if (c == 'Q') --Qr;
                else if (c == 'W') --Wr;
                else if (c == 'E') --Er;
                else --Rr;
                ++j;
                c = s[i++];
                if (c == 'Q') ++Qr;
                else if (c == 'W') ++Wr;
                else if (c == 'E') ++Er;
                else ++Rr;
                if (Qr <= target && Wr <= target && Er <= target && Rr <= target) {
                    pass = true;
                }
            }
            if (pass) {
                end = mid;
            }
            else {
                start = mid + 1;
            }
        }
        return start;
    }
};
