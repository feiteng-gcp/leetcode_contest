class Solution {
public:
    int balancedString(string s) {
        unordered_map<char, int> m;
        m['Q'] = 0; m['W'] = 1; m['E'] = 2; m['R'] = 3;
        int t = s.size() / 4;
        vector<vector<int>> preSums(s.size() + 1, vector<int>(4, 0));
        for (int i = 0; i < s.size(); i++) {
            if (i > 0)
                for (int j = 0; j < 4; j++) preSums[i+1][j] += preSums[i][j];
            preSums[i+1][m[s[i]]]++;
        }
        // for (auto v : preSums) {
        //     for (int i : v) cout << i << ' ';
        //     cout << '\n';
        // }
        vector<int> dec(4, 0);
        int decSum = 0;
        for (int i = 0; i < 4; i++) {
            int num = preSums.back()[i];
            if (num > t) dec[i] = (num - t);
            decSum += dec[i];
        }
        if (decSum == 0) return 0;
        // cout << "dec ";
        // for (int i : dec) cout << i << ' ';
        // cout << '\n';
        // binary search

        int result = s.size() + 1;
        for (int i = 0; i < s.size() - 1; i++) {
            bool state = true;
            for (int j = 0; j < 4; j++) {
                if (preSums.back()[j] - preSums[i][j] < dec[j]) state = false;
            }
            if (!state) break;
            int l = i, r = s.size() - 1;
            while (l <= r) {
                int m = (l + r) / 2;
                bool res = true;
                for (int j = 0; j < 4; j++) {
                    if (preSums[m+1][j] - preSums[i][j] < dec[j]) res = false;
                }
                if (res) r = m - 1;
                else l = m + 1;
            }
            // cout << i << ' ' << l << '\n';
            result = min(result, l - i + 1);
        }
        return result;
    }
};
