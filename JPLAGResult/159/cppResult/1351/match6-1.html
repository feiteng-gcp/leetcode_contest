class Solution {
public:
    int balancedString(string s) {
        int cnt[100005][4];
        memset(cnt,0,sizeof(cnt));
        for(int i=1;i<=s.size();++i) {
            for(int j=0;j<4;++j) cnt[i][j]=cnt[i-1][j];
            if(s[i-1] == 'Q') cnt[i][0]++;
            if(s[i-1] == 'W') cnt[i][1]++;
            if(s[i-1] == 'E') cnt[i][2]++;
            if(s[i-1] == 'R') cnt[i][3]++;
        }
        int m = s.size() / 4;
        int ans=1e9;
        for(int i=1;i<=s.size();++i) {
            int ok = 1;
            for(int j=0;j<4;++j)
                if(cnt[i-1][j] > m) ok = 0;
            if(!ok) break;

            int lo = i-1, hi = s.size();
            while(lo<hi) {
                int mid=(lo+hi)/2;
                int ok = 1;
                for(int j=0;j<4;++j)
                    if(cnt[s.size()][j] - (cnt[mid][j] - cnt[i-1][j]) > m) ok = 0;
                if(ok) hi = mid;
                else lo = mid+1;
            }
            // cout << i << " " << lo << endl;
            ans=min(ans,lo-i+1);
        }
        return ans;
    }
};
