typedef array<int, 4> arr;
class Solution {
public:
    int balancedString(string s) {
        map<char, int> A;
        A['Q'] = 0;
        A['W'] = 1;
        A['E'] = 2;
        A['R'] = 3;
        int n = s.size(), m = n / 4;
        vector<arr> pre(n + 1), suf(n + 1);
        for (int i = 0; i < n; ++i) {
            pre[i + 1] = pre[i];
            pre[i + 1][A[s[i]]]++;
        }
        for (int i = n - 1; i >= 0; --i) {
            suf[i] = suf[i + 1];
            suf[i][A[s[i]]]++;
        }
        int ret = 1 << 30;
        for (int i = 0; i < n; ++i) {
            bool found = false;
            for (int j = 0; j < 4; ++j) {
                if (pre[i][j] > m) found = true;
            }
            if (found) break;
            int low = i, high = n;
            while (low != high) {
                int mid = (low + high) / 2;
                bool found = false;
                for (int j = 0; j < 4; ++j) {
                    if (pre[i][j] + suf[mid][j] > m) found = true;
                }
                if (found) {
                    low = mid + 1;
                } else {
                    high = mid;
                }
            }
            ret = min(ret, high - i);
        }
        return ret;
    }
};
