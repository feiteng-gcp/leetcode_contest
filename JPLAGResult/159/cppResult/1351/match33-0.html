ll po(ll a, ll b, ll mod)
{
    ll res = 1;
    a %= mod;
    for (; b; b >>= 1)
    {
        if (b & 1)
            res = res * a % mod;
        a = a * a % mod;
    }
    return res;
}
ll gcd(ll a, ll b)
{
    if (a == 0)
    {
        return b;
    }
    else
    {
        return gcd(b % a, a);
    }
}
void extgcd(ll a, ll b, ll &d, ll &x, ll &y)
{
    if (!b)
    {
        d = a;
        x = 1;
    }
    else
    {
        extgcd(b, a % b, d, y, x);
        y -= x * (a / b);
    }
}
ll inverse(ll a, ll n)
{
    ll d, x, y;
    extgcd(a, n, d, x, y);
    return d == 1 ? (x + n) % n : -1;
}

class Solution {
public:
    string v;
    int cnt[maxn][7];
    bool check(int len) {
        int n = v.length();
        if(len == 0) {
            return cnt[n][1] == cnt[n][2] && cnt[n][1] == cnt[n][3] && cnt[n][1] == cnt[n][4];
        }
        for(int i=1;i+len-1<=n;i++) {
            int start = i;
            int end = i + len - 1;
            int cur1 = cnt[i-1][1] + cnt[n][1] - cnt[i+len-1][1];
            int cur2 = cnt[i-1][2] + cnt[n][2] - cnt[i+len-1][2];
            int cur3 = cnt[i-1][3] + cnt[n][3] - cnt[i+len-1][3];
            int cur4 = cnt[i-1][4] + cnt[n][4] - cnt[i+len-1][4];

            // cout<< cur1 <<" "<<cur2<<" "<<cur3<<" "<< cur4;
            std::vector<int> tmp = {cur1, cur2, cur3, cur4};
            sort(tmp.begin(), tmp.end());

            int all = len;
            int flag = 1;
            REP(k,0,4) {
                if(tmp.back() - tmp[k] > all) {
                    flag = 0;
                    break;
                }
                all -= (tmp.back() - tmp[k]);
            }
            if(flag && (all % 4 ==0)) {
                return true;
            }
        }
        return false;
    }

    int balancedString(string s) {
        v = s;
        memset(cnt, 0, sizeof(cnt));
        int le = 0;
        int ri = s.length();
        REP(i,0,s.length()) {
            if(s[i] == 'Q') {
                cnt[i+1][1]++;
            } else if(s[i]=='W') {
                cnt[i+1][2]++;
            } else if(s[i]=='E') {
                cnt[i+1][3]++;
            } else {
                cnt[i+1][4]++;
            }
            REPP(k,1,4) {
                cnt[i+1][k] += cnt[i][k];
                // cout<<i <<" "<<k << cnt[i+1][k]<<endl;
            }
        }
        
        while(le < ri) {
            int mid = (le + ri) / 2;
            // DE(mid)
            if(check(mid)) {
                ri = mid;
            } else {
                le = mid + 1;
            }
        }
        // DE(le)
        return le;
    }
