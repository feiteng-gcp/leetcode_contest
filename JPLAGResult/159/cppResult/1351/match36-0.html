using namespace std;

class Solution {
private:
  static constexpr int max_n = 10001;

public:
  Solution() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    std::cout.tie(nullptr);
  }

  int balancedString(string s) {
    const int n = s.size();
    const int k = n / 4;

    int* a = new int[n];
    int needed[4]; memset(needed, 0, sizeof(needed));
    int freq[4]; memset(freq, 0, sizeof(freq));
    for (int i = 0; i < n; ++i) {
      switch(s[i]) {
      case 'Q': a[i] = 0; break;
      case 'W': a[i] = 1; break;
      case 'E': a[i] = 2; break;
      case 'R': a[i] = 3; break;
      }

      ++freq[a[i]];
    }

    for (int i = 0; i < 4; ++i) {
      if (freq[i] > k) needed[i] = freq[i] - k;
    }

    if (!needed[0] && !needed[1] && !needed[2] && !needed[3]) return 0;

    std::deque<int> positions[4];

    int best = n;
    for(int i = 0; i < n; ++i) {
      positions[a[i]].push_back(i);
      if (positions[a[i]].size() > needed[a[i]])
        positions[a[i]].pop_front();

      if (positions[0].size() == needed[0] &&
          positions[1].size() == needed[1] &&
          positions[2].size() == needed[2] &&
          positions[3].size() == needed[3]) {
        int start = i;
        for (int j = 0; j < 4; ++j) {
          if (needed[j] && positions[j].front() < start)
            start = positions[j].front();
        }
        if (i - start + 1 < best) best = i - start + 1;
      }
    }

    delete[] a;

    return best;
  }
};

#ifdef LOCALRUN
void run_test(string s) {
  printf("%s -> %d\n", s.c_str(), Solution().balancedString(s));
}

int main() {
  run_test("QWER");
  run_test("QQWE");
  run_test("QQQW");
  run_test("QQQQ");
  return 0;
}
#endif
