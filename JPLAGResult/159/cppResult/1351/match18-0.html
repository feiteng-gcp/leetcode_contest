class Solution {
    bool meet(int *cnt) {
        for (int i = 0; i < 4; ++i) {
            if (cnt[i] > 0) { return false;}
        }
        return true;
    }
public:
    int balancedString(string s) {
        int sz = s.length();
        int ans = sz;
        unordered_map<char, int> mp;
        mp['Q'] = 0;
        mp['W'] = 1;
        mp['E'] = 2;
        mp['R'] = 3;
        int cnt[4];
        memset(cnt, 0, sizeof(cnt));
        for (int i = 0; i < sz; ++i) {
            ++cnt[mp[s[i]]];
        }
        bool flag = true;
        for (int i = 0; i < 4; ++i) {
            if (cnt[i] > sz / 4) {
                cnt[i] -= sz / 4;
                flag = false;
            }
            else {
                cnt[i] = 0;
            }
        }
        if (flag) { return 0; }
        int end = 0, start = 0;
        while(end < sz) {
            while (end < sz && !meet(cnt)) {
                --cnt[mp[s[end]]];
                ++end;
            }
            while(start < end && meet(cnt)) {
                ans = min(ans, end - start);
                ++cnt[mp[s[start]]];
                ++start;
            }
        }
        return ans;
    }
};
