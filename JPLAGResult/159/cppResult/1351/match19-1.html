class Solution {
public:
    int balancedString(string s) {
        int target = s.size() / 4;
        
        vector<int> cnt(4, 0);
        unordered_map<char, int> hash;
        hash['Q'] = 0;
        hash['W'] = 1;
        hash['E'] = 2;
        hash['R'] = 3;
        for(auto& c : s){
            cnt[hash[c]]++;
        }
        
        for(int i = 0; i < 4; i++)
            cnt[i] -= target;
        
        if(cnt[0] == cnt[1] && cnt[0] == cnt[2] && cnt[0] == cnt[3]) return 0;
        
        int i = 0, j = 0;
        while(j < s.size() && !check(cnt)){
            cnt[hash[s[j]]]--;
            j++;
        } 
        int res = j - i;
        while(true){
            if(j >= s.size() && !check(cnt)) break;
            while(i < j && check(cnt)){
                res = min(res, j - i);
                cnt[hash[s[i]]]++;
                i++;
            }
            while(j < s.size() && !check(cnt)){
                cnt[hash[s[j]]]--;
                j++;
            }
        }
        return res;
    }
    
private:
    bool check(const vector<int>& cnt){
        for(int i = 0; i < 4; i++)
            if(cnt[i] > 0) return false;
        return true;
    }
};
