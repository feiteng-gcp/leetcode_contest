class Solution {
public:
    int index(char c) {
        if (c == 'Q') return 0;
        else if (c == 'W') return 1;
        else if (c == 'E') return 2;
        else return 3;
    }
    
    bool check(string &s, int len, int *count) {
        int c[4] = {0}, limit = s.size()/4;
        for (int i = 0; i < len; ++i)
            c[index(s[i])]++;
        
        for (int j = 0; j <= s.size() - len; ++j) {
            
            if (j > 0) {
                c[index(s[j + len - 1])]++;
                c[index(s[j - 1])]--;
            }
            
            if (count[0] - c[0] <= limit && count[1] - c[1] <= limit && count[2] - c[2] <= limit && count[3] - c[3] <= limit)
                return true;
        }
        return false;
    }
    
    int balancedString(string s) {
     //   s="QQQQ";
        int count[4] = {0};
        for (auto c: s) count[index(c)]++;
        
        int over = 0;
        for (int i = 0;i < 4; ++i) {
            if (count[i] > s.size()/4) over += count[i] - s.size()/4;
        }
        
        if (over == 0) return 0;
        

        int lo = over - 1, hi = s.size();
        
        while (lo + 1 < hi) {
            int mid = (lo + hi)/2;
            if (check(s, mid, count)) {
                hi = mid;
            } else 
                lo = mid;
        }
        
        return hi;
    }
};
