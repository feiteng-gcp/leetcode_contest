int x[100005][4];

class Solution {
public:
    int balancedString(string s) {
        int n = s.size();
        for (int i = 0; i < n; i++) {
            x[i + 1][0] = x[i][0];
            x[i + 1][1] = x[i][1];
            x[i + 1][2] = x[i][2];
            x[i + 1][3] = x[i][3];
            if (s[i] == 'Q') x[i + 1][0]++;
            if (s[i] == 'W') x[i + 1][1]++;
            if (s[i] == 'E') x[i + 1][2]++;
            if (s[i] == 'R') x[i + 1][3]++;
        }
        int ans = INT_MAX;
        for (int i = 0; i < n; i++) {
            int lo = i;
            int hi = n + 1;
            while (lo < hi) {
                int mi = (lo + hi) / 2;
                bool o = 1;
                for (int j = 0; j < 4; j++)
                    if (x[n][j] - x[mi][j] + x[i][j] > n / 4) {
                        o = 0;
                        break;
                    }
                if (o)
                    hi = mi;
                else lo = mi + 1;
            }
            if (lo != n + 1)
                ans = min(ans, lo - i);
        }
        return ans;
    }
};
