#include <bits/stdc++.h>
using namespace std;
class Solution {
public:
    int N;
    vector<int> pre[4];
    char M[256];
    
    void S(int start,int end,int *out){
        if(end<start){
            for(int i=0;i<4;++i)out[i]+=0;
        }else{
            for(int i=0;i<4;++i){
                out[i]+=pre[i][end+1] - pre[i][start];
            }
        }
    }
    
    bool can(int start,int end){
        int R[4] = {0,0,0,0};
        S(0,start-1,R);
        S(end+1,N-1,R);
        for(int i=0;i<4;++i){

            if(R[i]>N/4)return false;
        }
        return true;
    }
    
    int balancedString(string s) {
        M['Q']=0;
        M['W']=1;
        M['E']=2;
        M['R']=3;
        
        N=s.size();
        for(int i=0;i<4;++i){
            pre[i] = vector<int>(N+1);
        }
        for(int i=0;i<N;++i){
            for(int k=0;k<4;++k){
                pre[k][i+1] = pre[k][i];
            }
            pre[M[s[i]]][i+1]++;
        }
        
        int best=N;
        int end=-1;
        for(int start=0;start<N;++start){
            if(!can(start,N-1))break;
            
            int lo=end+1,hi=N;
            
            while(lo<hi){
                int mid=(lo+hi)/2;
                if(can(start, mid-1)) hi=mid;
                else lo=mid+1;
            }
            
            end=hi-1;
            if(end>=N) break;
            else{
                best=min(best,end-start+1);
            }
        }
        return best;
    }
};
