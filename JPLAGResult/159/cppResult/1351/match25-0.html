class Solution {
public:
    int balancedString(string s) {
        int need = s.size()/4;
        int Qcount = 0;
        int Wcount = 0;
        int Ecount = 0;
        int Rcount = 0;
        int Qneed = 0;
        int Wneed = 0;
        int Eneed = 0;
        int Rneed = 0;
        for (char c:s) {
            if (c=='Q') Qcount++;
            else if (c=='W') Wcount++;
            else if (c=='E') Ecount++;
            else if (c=='R') Rcount++;
        }
        Qneed = max(Qcount-need,0);
        Wneed = max(Wcount-need,0);
        Eneed = max(Ecount-need,0);
        Rneed = max(Rcount-need,0);
        
        Qcount = 0;
        Wcount = 0;
        Ecount = 0;
        Rcount = 0;
        int start = 0;
        int res = s.size();
        for (int end = 0; end<s.size(); ++end) {
            if (s[end]=='Q') Qcount++;
            else if (s[end]=='W') Wcount++;
            else if (s[end]=='E') Ecount++;
            else if (s[end]=='R') Rcount++;
            
            while (Qcount>=Qneed&&Wcount>=Wneed&&Ecount>=Eneed&&Rcount>=Rneed) {
                res = min(res,end-start+1);
                if (start<=end) {
                    if (s[start]=='Q') Qcount--;
                    else if (s[start]=='W') Wcount--;
                    else if (s[start]=='E') Ecount--;
                    else if (s[start]=='R') Rcount--;
                    start++;
                }
                else break;
            }
        }
        
        return res;
    }
};
