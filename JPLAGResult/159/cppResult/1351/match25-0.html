class Solution {
public:
    int balancedString(string s) {
        map<char, int> cnt;
        for(char c : s) ++cnt[c];
        int _ = s.size() / 4;
        int ans = 0x3f3f3f3f;
        for(auto& x : cnt) {
			if(x.second <= _) x.second = -ans;
			else x.second -= _;
		}
        map<char, int> temp = cnt;
        for(auto& x : temp) x.second = 0;
        for(int l = 0, r = 0; l < s.size(); ++l) {
            while(true && r < s.size()) {
	            bool ok = true;
	            for(auto& x : cnt) if(temp[x.first] < x.second) ok = false;
                if(ok) break;
                ++temp[s[r++]];
            }
            bool ok = true;
            for(auto& x : cnt) if(temp[x.first] < x.second) ok = false;
            if(ok == false && r == s.size()) break;
            ans = min(ans, r-l);
            if(r == l) ++temp[s[r++]];
            --temp[s[l]];
        }
        return ans;
    }
};
