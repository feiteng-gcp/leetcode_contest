class Solution {
public:
  int id(char c) {
    if (c == 'Q') return 0;
    if (c == 'W') return 1;
    if (c == 'E') return 2;
    if (c == 'R') return 3;
    return -1;
  }
  int balancedString(string s) {
    int n = s.size();
    vector<vector<int>> L(n, vector<int>(4));
    vector<vector<int>> R(n, vector<int>(4));
    for (int i = 0; i < s.size(); i++) {
      for (int j = 0; j < 4; j++) {
        L[i][j] = (i > 0?L[i-1][j]:0) + (j == id(s[i])?1:0);
      }
    }
    for (int i = s.size() - 1; i >= 0; i--) {
      for (int j = 0; j < 4; j++) {
        R[i][j] = (i < s.size() - 1?R[i+1][j]:0) + (j == id(s[i])?1:0);
      }
    }
    L.insert(L.begin(), vector<int>(4));
    L.push_back(L.back());
    R.insert(R.begin(), R[0]);
    R.push_back(vector<int>(4));
    int lo = 1, hi = n;
    while (lo < hi) {
      int mid = (lo + hi) / 2;
      bool isValid = false;
      for (int i = 0; i + mid < L.size(); i++) {
        int M = mid;
        vector<int> cnt(4);
        for (int j = 0; j < 4;j++) cnt[j] += L[i][j] + R[i + mid][j];
        int mi = *min_element(cnt.begin(), cnt.end());
        int ma = *max_element(cnt.begin(), cnt.end());
        for (int j = 0; j < 4; j++) {
          M -= ma - cnt[j];
          cnt[j] = ma;
        }
        if (M >= 0) {
          isValid = true;
          break;
        }
      }
     // cout << lo << " " << hi << ": " << mid << " " << isValid << endl; 
      if (isValid) hi = mid;
      else lo = mid + 1;
    }
    return hi - 1;
  }
};
