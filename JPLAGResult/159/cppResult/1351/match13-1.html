class Solution {
public:
    int balancedString(string s) {
        int cq=0,cw=0,ce=0,cr=0;
        int n = s.size();
        for (int i = 0; i < s.size(); ++ i) {

            switch (s[i]) {
            case 'Q': ++ cq; break;
            case 'W': ++ cw; break;
            case 'E': ++ ce; break;
            case 'R': ++ cr; break;
            }
        }
        int nq = (cq > n/4) ? (cq - n/4) : 0;
        int nw = (cw > n/4) ? (cw - n/4) : 0;
        int ne = (ce > n/4) ? (ce - n/4) : 0;
        int nr = (cr > n/4) ? (cr - n/4) : 0;
        if (nq+nw+ne+nr == 0) return 0;
        // printf("%d %d %d %d\n", nq, nw, ne, nr);
        int sq = 0, sw = 0, se = 0, sr = 0;
        int i = 0;
        int j = 0;
        int ans = s.size();
        while ( i < s.size()) {
            // printf("%d %d\n", i,j);
            for (;i < s.size() && !(sq >= nq && sw >= nw && se >= ne && sr >= nr); ++ i) {

                switch (s[i]) {
                case 'Q': ++ sq; break;
                case 'W': ++ sw; break;
                case 'E': ++ se; break;
                case 'R': ++ sr; break;
                }
            }
            
            for (; j < s.size() && sq >= nq && sw >= nw && se >= ne && sr >= nr; ++ j) {

                switch (s[j]) {
                case 'Q': -- sq; break;
                case 'W': -- sw; break;
                case 'E': -- se; break;
                case 'R': -- sr; break;
                }
            }
            
            printf("%d %d\n", i,j);
            ans = min(ans, i - j + 1);
        }
        return ans;
    }
};
