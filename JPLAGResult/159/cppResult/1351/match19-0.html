class Solution {
public:
    int idx(char c) {
        if(c == 'Q') return 0;
        else if(c == 'W') return 1;
        else if(c == 'E') return 2;
        else return 3;
    }
    
    bool decision(int len, vector<int>& need, string& s) {
        int cnt[4] = {0,0,0,0};
        for(int i=0;i<len;i++) {
            cnt[idx(s[i])]++;
        }
        bool can = true;
        for(int j=0;j<4;j++)
            if(cnt[j] < need[j]) {
                can = false;
                break;
            }
        if(can) return true;
        for(int i=len;i<s.size();i++) {
            cnt[idx(s[i-len])]--;
            cnt[idx(s[i])]++;
            bool can = true;
            for(int j=0;j<4;j++)
                if(cnt[j] < need[j]) {
                    can = false;
                    break;
                }
            if(can) return true;
        }
        return false;
    }
    
    int balancedString(string s) {
        int eachLen = s.size()/4;
        int cnt[4] = {0,0,0,0};
        for(char c: s)
            cnt[idx(c)]++;
        
        
        vector<int> need(4, 0);
        for(int i=0;i<4;i++)
            if(cnt[i] > eachLen) need[i] = cnt[i]-eachLen;
        
        int lo = 0, hi = s.size()-1;
        while(lo < hi) {
            int mid = (lo+hi)/2;
            if(decision(mid, need, s)) {
                hi = mid;
            } else {
                lo = mid+1;
            }
        }
        return hi;
        
        
    }
};
