class Solution {
public:
    int balancedString(string s) {
        unordered_map<char, int> count;
        for(auto c:s)
            count[c]++;
        unordered_map<char, int> target;
        int val = s.size()/4;
        int res = 0;
        for(auto x:count) {
            if(x.second>val)
                target[x.first] = x.second-val;
        }
        if(target.size()==0)
            return 0;
        int start_idx = -1;
        int end_idx = -1;
        int num = target.size();
        int j=0;
        for(int i=0; i<s.size(); i++) {
            if(target.count(s[i])) {
                target[s[i]]--;
                if(target[s[i]]==0)
                    num--;
            }
            while(num==0&&j<=i) {
                if(start_idx==-1||i-j+1<end_idx-start_idx+1) {
                    start_idx = j;
                    end_idx = i;
                }
                if(target.count(s[j])) {
                    target[s[j]]++;
                    if(target[s[j]]==1)
                        num++;
                }
                j++;
            }
        }
        return end_idx-start_idx+1;
    }
};
