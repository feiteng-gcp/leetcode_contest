class Solution {
public:
    int balancedString(string s) {
        int n = s.length();
        vector<vector<int>> pref(n+1,vector<int>(4));
        for (int i = 0; i < n; i++){
            pref[i+1][0]=pref[i][0]+(s[i]=='Q');
            pref[i+1][1]=pref[i][1]+(s[i]=='W');
            pref[i+1][2]=pref[i][2]+(s[i]=='E');
            pref[i+1][3]=pref[i][3]+(s[i]=='R');
        }
        int res = n;
        for (int i = 0; i <= n; i++){
            // keep < i

            int l = i-2, r = n;
            // throw [i,r]
            while(l + 1 < r){
                int mid = (l + r) / 2;
                bool good = true;
                for (int j = 0; j < 4; j++) {
                    if (pref[i][j] + pref[n][j] - pref[mid+1][j] <= n / 4) {
                        // cout << i << " " << mid << j <<" "
                        good &=true;
                    }
                    else {
                        // if (i == 7) {
                        // cout << i << " " << mid << " " << j << endl;
                        // cout << (pref[i][j] + pref[n][j] - pref[mid+1][j]) << endl;
                        // cout << (n - (mid - i + 1)) << " " << (n - (mid - i + 1) + 3) << endl;}
                        good &=false;
                    }
                }
                if (good) {
                    r = mid;
                }
                else l = mid;
            }
            if(r<n)
            res = min(res,r - i + 1);
        }
        return res;
    }
};
