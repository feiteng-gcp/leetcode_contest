class Solution {
public:
  int cnt[100010][4];
  int diff[4];
  
  int ord(char ch) {
    switch (ch) {
      case 'Q': return 0;
      case 'W': return 1;
      case 'E': return 2;
      case 'R': return 3;
    }
    assert(false);
  }
  
  bool good(int i, int j) {
    for (int c = 0; c < 4; ++c) {
      int d = cnt[j][c] - cnt[i-1][c];
      if (d < diff[c]) return false;
    }
    return true;
  }
  
  int balancedString(string s) {
    memset(cnt, 0, sizeof(cnt));
    
    s.insert(s.begin(), ' ');
    int n = s.length();
    

    for (int i = 1; i < n; ++i) {
      for (int j = 0; j < 4; ++j) {
        cnt[i][j] = cnt[i-1][j] + (j == ord(s[i]));
      }

    }
    
    /*for (int i = 0; i < n; ++i) {
      for (int j = 0; j < 4; ++j) cout << cnt[i][j] << " ";
      cout << endl;
    }
    cout << endl;*/
    
    int avg = s.length() / 4;
    for (int j = 0; j < 4; ++j) {
      if (cnt[n-1][j] >= avg) diff[j] = cnt[n-1][j] - avg;
    }
    
    //for (int i = 0; i < 4; ++i) cout << diff[i] << " ";
    //cout << endl;
    
    int j = 1, ret = 1<<30;
    for (int i = 1; i < n; ++i) {
      while (j < n && !good(i, j)) ++j;
      if (j < n && good(i, j)) {
        ret = min(ret, j-i+1);
      }
    }
    
    if (ret == 1<<30) ret = -1;
    return ret;
  }
};
