class Solution {
public:
    int balancedString(string s) {
        int n = s.length();
        map<char,int> cnt;
        for(char c : s) cnt[c]++;
        int l = 0, r = 0, ans = n;
        auto ng = [&]()->bool{
            for(auto p : cnt)   if(p.second > n/4)  return true;
            return false;
        };
        while(l < n){
            while(r < n && ng()){
                cnt[s[r]]--;
                r++;
            }
            if(r == n && ng())  break;
            ans = min(ans, r-l);
            cnt[s[l]]++;
            l++;
        }
        return ans;
    }
};
