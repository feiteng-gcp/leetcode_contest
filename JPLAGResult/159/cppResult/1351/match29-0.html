class Solution {
public:
    bool check(int val, int n, int a[], int g[], int c[], int t[]){
        for(int i=1; i<=n; i++){
            int flag=0;
            if(i+val-1>n)break;
            if(a[i-1]+a[n]-a[i+val-1]<=n/4)flag++;
            if(c[i-1]+c[n]-c[i+val-1]<=n/4)flag++;
            if(t[i-1]+t[n]-t[i+val-1]<=n/4)flag++;
            if(g[i-1]+g[n]-g[i+val-1]<=n/4)flag++;
            if(flag==4){return 1;}
        }
        return 0;
    }
    int balancedString(string s) {
        int n=s.length();
        s='$'+s;
        int MAXN=100010;
        int a[MAXN],g[MAXN],c[MAXN],t[MAXN];
        for(int i=1; i<=n; i++){
            a[i]=a[i-1];
            g[i]=g[i-1];
            c[i]=c[i-1];
            t[i]=t[i-1];
            if(s[i]=='Q')a[i]++;
            if(s[i]=='W')g[i]++;
            if(s[i]=='E')c[i]++;
            if(s[i]=='R')t[i]++;
        }
        int l=-1,r=n;
        while(r-l>1){
            int mid=(r+l)>>1;
            if(check(mid,n,a,g,c,t)) r=mid;
            else l=mid;
        }
        return r;
    }
};
