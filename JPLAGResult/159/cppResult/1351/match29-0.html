class Solution {
 public:
  int balancedString(string s) {
    vector<int> cnt(256);
    for (char c : s) ++cnt[c];
    int lb = -1;
    int ub = s.size() - 1;
    while (ub - lb > 1) {
      int mb = lb + ub >> 1;
      vector<int> mcnt(256);
      for (int i = 0; i < mb; ++i) ++mcnt[s[i]];
      bool ok = false;
      for (int i = 0; i + mb <= s.size(); ++i) {
        int qx = s.size() / 4 - (cnt['Q'] - mcnt['Q']);
        int wx = s.size() / 4 - (cnt['W'] - mcnt['W']);
        int ex = s.size() / 4 - (cnt['E'] - mcnt['E']);
        int rx = s.size() / 4 - (cnt['R'] - mcnt['R']);
        if (qx >= 0 && wx >= 0 && ex >= 0 && rx >= 0) ok = true;
        --mcnt[s[i]];
        if (i + mb < s.size()) ++mcnt[s[i + mb]];
      }
      (ok ? ub : lb) = mb;
    }
    return ub;
  }
};
