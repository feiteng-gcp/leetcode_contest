class Solution {
public:
    int balancedString(string s) {
        vector<int> ss;
        for (char c: s) {
            int n = 0;
            if (c == 'Q') {
                n = 0;
            }
            else if (c == 'W') {
                n = 1;
            }
            else if (c == 'E') {
                n = 2;
            }
            else if (c == 'R') {
                n = 3;
            }
            ss.push_back(n);
        }
        
        vector<int> cnt(4,0);
        for (int n: ss) {
            cnt[n]++;
        }
        const int len = s.size();
        const int balanced = len / 4;
        int matched = 0;
        
        vector<vector<int>> csum(4, vector<int>(len+1));
        for (int j = 0; j < s.size(); ++j) {
            for (int i = 0; i < 4; ++i) {
                csum[i][j+1] = csum[i][j];
            }
            csum[ss[j]][j+1]++;
        }
        int mindiff = 0;
        for (int i = 0; i < 4; ++i) {
            mindiff = max(mindiff, cnt[i] - balanced);
        }
        for (int i = mindiff; i <= len; ++i) {
            for (int j = 0; j + i <= len; ++j) {
                int lack = 0;
                bool ok = true;
                for (int k = 0; k < 4; ++k) {
                    int d = cnt[k] - (csum[k][j+i] - csum[k][j]);
                    if (d > balanced) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    return i;
                }
            }
        }
        return len;
    }
};
