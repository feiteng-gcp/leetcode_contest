class Solution {
public:
    int balancedString(string s) {
        int pre[100005][4];
        memset(pre,0,sizeof(pre));
        for(int i=s.size()-1;i>=0;i--){
            for(int j=0;j<4;j++)pre[i][j]=pre[i+1][j];
            if(s[i]=='Q')pre[i][0]++;
            if(s[i]=='W')pre[i][1]++;
            if(s[i]=='E')pre[i][2]++;
            if(s[i]=='R')pre[i][3]++;
        }
        if(pre[0][0]==pre[0][1]&&pre[0][0]==pre[0][2]&&pre[0][0]==pre[0][3])return 0;
        int ans=1e9;
        int now[4]={0,0,0,0};
        int tot=s.size()/4;
        for(int i = 0;i<s.size();i++){
            int Min=i,Max=s.size()+1;
            while(Max>Min+1){
                int mid=(Max+Min)/2;
                int ok=1;
                for(int i = 0;i<4;i++){
                    if(now[i]+pre[mid][i]>tot)ok=0;
                }
                if(ok)Max=mid;
                else Min=mid;
            }
            if(Max==s.size()+1)break;
            ans=min(ans,Max-i);
            if(s[i]=='Q')now[0]++;
            if(s[i]=='W')now[1]++;
            if(s[i]=='E')now[2]++;
            if(s[i]=='R')now[3]++;
        }
        return ans;
    }
};
