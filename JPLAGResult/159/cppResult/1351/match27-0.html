class Solution {
public:
    int balancedString(string s) {
        int n = s.size();
        vector<int> fact(4,- n / 4);
        for(char x:s){
            if(x == 'Q')fact[0]++;
            else if(x == 'W')fact[1]++;
            else if(x == 'R')fact[2]++;
            else if(x == 'E')fact[3]++;
        }
        int sum = 0;
        vector<int> cnt(26,0);
        if(fact[0] > 0)cnt['Q' - 'A'] += fact[0];
        if(fact[1] > 0)cnt['W' - 'A'] += fact[1];
        if(fact[2] > 0)cnt['R' - 'A'] += fact[2];
        if(fact[3] > 0)cnt['E' - 'A'] += fact[3];
        int k = 0;
        for(int x:fact)if(x > 0)k++;
        if(k == 0)return 0;
        // cout<<"k ="<<k<<endl;
        // for(int c:cnt)cout<<c<<" ";cout<<endl;
        int begin = 0,len = INT_MAX;
        for(int i = 0;i < s.size();i++){
            if(--cnt[s[i] - 'A'] == 0){
                k--;
            }
            while(k == 0){
                len = min(len,i - begin + 1);
                // cout<<begin<<","<<i<<endl;
                // cout<<s.substr(begin,i - begin + 1)<<endl;
                if(++cnt[s[begin++] - 'A'] == 1)k++;
            }

        }
        return len;   
    }
};
