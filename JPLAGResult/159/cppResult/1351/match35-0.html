class Solution {
private:
    int ct[4];
    int cur[4];
    bool check() {
        for (int i=0; i<4; i++) {
            if (ct[i] > 0 && ct[i] > cur[i]) return false;
        }
        return true;
    }
    int f(char c) {
        if (c == 'Q') {
            return 0;
        }
        else if (c == 'W') {
            return 1;
        }
        else if (c == 'E') {
            return 2;
        }
        else {
            return 3;
        }
    }
public:
    int balancedString(string s) {
        for (int i=0; i<4; i++) {
            ct[i] = 0;
            cur[i] = 0;
        }
        for (int i=0; i<s.size(); i++) {
            ct[f(s[i])]++;
        }
        bool bal = true;
        for (int i=0; i<4; i++) {
            ct[i] -= int(s.size()) / 4;
            if (ct[i] != 0) bal = false;
        }
        if (bal) return 0;
        
        int best = s.size();
        int start = 0;
        for (int i=0; i<s.size(); i++) {
            cur[f(s[i])]++;
            while (start <= i && check()) {
                best = min(best, i-start+1);
                cur[f(s[start])]--;
                start++;
            }
            // cout << start << " " << i << endl;
        }
        
        return best;
    }
};
