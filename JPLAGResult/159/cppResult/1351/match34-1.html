class Solution {
public:
    int no(map<char, int>& m, int k) {
        for (auto it : m) {
            if (it.second > k) {
                return 1;
            }
        }
        return 0;
    }
    

    int balancedString(string s) {
        map<char, int> m;
        for (char c : s) {
            m[c]++;
        }
        int k = s.length()/4;
        int ret = s.length();
        int cur = 0;
        for (int i = 0;i < s.length();i++) {
            int ok = 1;
            while (no(m, k)) {
                if (cur < s.length()) {
                    m[s[cur]]--;
                    cur++;
                }
                else {
                    ok = 0;
                    break;
                }
            }
            if (!ok)
                break;
            ret = min(ret, cur - i);
            m[s[i]]++;
        }
        return ret;
    }
};
