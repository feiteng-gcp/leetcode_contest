class Solution {
public:
    int balancedString(string s) {
        const int n = s.size();
        int cnt[128] = {0};
        for (int i = 0; i < n; ++i)
            ++cnt[s[i]];
        int ans = n - 1;
        for (int i = -1, j = 0; i < n; ++i) {

            if (i >= 0) ++cnt[s[i]];
            while (j < n && 
                   (cnt['Q'] > n / 4 ||
                    cnt['W'] > n / 4 ||
                    cnt['R'] > n / 4 ||
                    cnt['E'] > n / 4)) {
                --cnt[s[j++]];
            }
            if (cnt['Q'] <= n / 4 &&
                    cnt['W'] <= n / 4 &&
                    cnt['R'] <= n / 4 &&
                    cnt['E'] <= n / 4)
            {
                ans = min(ans, j - i - 1);
            }
        }
        return ans;
    }
};
