class Solution {
public:
    int get_idx(char p){
        if(p == 'Q') return 0;
        if(p == 'W') return 1;
        if(p == 'E') return 2;
        if(p == 'R') return 3;
        return -1;
    }
    bool judge(int now, const string & s){
        int tsum1[4];memset(tsum1,0,sizeof(tsum1));
        for(int i = 0;i < s.size(); i++) tsum1[ get_idx(s[i]) ] ++;
        int tsum2[4];memset(tsum2,0,sizeof(tsum2));
        int flag = 0;
        for(int i = 0;i <s.size() ; i++){
            tsum2[ get_idx(s[i])] ++;
            if(i >= now ){
                tsum2[get_idx(s[i-now])] --;
            }
            if(i >= now-1){
                int flag = 1;
                for(int j = 0; j < 4; j++){
                    if(tsum1[j] - tsum2[j] > (s.size() / 4)) flag = 0;
                }
                if(flag) return true;
            }
        }
        return false;
    }
    int binary_search(int l, int r, const string & s){
        
        int ans = r;
        while(l < r){
            int mid = (l+r) / 2;
            //cout<< l <<" "<<r<<" " << judge(mid, s)<<endl;
            if(judge(mid, s)){
                r = mid;
                ans = mid;
            }
            else{
                l = mid + 1;
            }
        }
        return ans;
    }
    int balancedString(string s) {
        int l, r;
        l = 0;r = s.size();
        return binary_search(l, r,s);
    }
};
