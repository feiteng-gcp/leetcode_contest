class Solution {
public:
    int balancedString(string s) {
        unordered_map<char,int> cnt;
        for(char c:s)
            cnt[c]++;
        int n=s.length()/4;
        
        int m=0;
        for(auto &p:cnt)
        {
            //cout<<n-p.second<<" ";
            if(n-p.second<0)
            {
                p.second=p.second-n;
                m++;
            }
            else p.second=0;
        }
        int l=0;
        n=s.length();
        int ans=n;
        //cout<<m<<endl;
        for(int r=0;r<n;r++)
        {
            if(--cnt[s[r]]==0) m--;
            while(l<=r&&m<=0)
            {
                if(cnt[s[l]]+1>0) break;
                cnt[s[l]]++;
                l++;
            }
            //cout<<l<<" "<<r<<" "<<m<<endl;
            if(m<=0) ans=min(ans,r-l+1);
        }
        //cout<<endl;
        return ans;
    }
};
