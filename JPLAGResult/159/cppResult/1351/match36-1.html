class Solution {
public:
    int balancedString(string s) {
        std::vector<int> count(4);

        for (int i = 0; i < s.length(); i++) {
            switch (s[i]) {
                case 'Q':
                    s[i] = 0;
                    break;
                case 'W':
                    s[i] = 1;
                    break;
                case 'E':
                    s[i] = 2;
                    break;
                case 'R':
                    s[i] = 3;
                    break;
            }
            count[s[i]]++;
        }
        int l = 0;
        int target = s.length() >> 2;
        int minLen = s.length();
        for (int r = 0; r <= s.length(); r++) {
            if (count[0] <= target && count[1] <= target && count[2] <= target && count[3] <= target) {
                if (r - l < minLen) {
                    minLen = r - l;
                }
                if (r == s.length()) {
                    break;
                }
            }
            count[s[r]]--;
            while (count[s[l]] < target) {
                count[s[l]]++;
                l++;
            }
        }
        return minLen;
    }
};
