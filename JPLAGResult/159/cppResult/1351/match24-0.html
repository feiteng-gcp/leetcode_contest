class Solution {
public:
    int balancedString(string s) {
        int L = s.length();
        map<int,int> cur, count;
        for (char c : s) {
            count[c]++;
        }
        int ans = L;
        for (int left = 0, right = 0; left < L; left++) {
            if (left > 0) {
                cur[s[left - 1]]--;
            }
            right = max(left, right);
            // left ~ right - 1
            while (right < L && ((count['Q'] - cur['Q'] > L / 4) || (count['W'] - cur['W'] > L / 4) || (count['E'] - cur['E'] > L / 4) || (count['R'] - cur['R'] > L / 4))) {
                cur[s[right]]++;
                right++;
            }
            if ((count['Q'] - cur['Q'] <= L / 4) && (count['W'] - cur['W'] <= L / 4) && (count['E'] - cur['E'] <= L / 4) && (count['R'] - cur['R'] <= L / 4)) {
                ans = min(ans, right - left);
            }
        }
        return ans;
    }
};
