#define FOR(i, begin, end) for(int i=(begin),i##_end_=(end);i<i##_end_;i++)
#define IFOR(i, begin, end) for(int i=(end)-1,i##_begin_=(begin);i>=i##_begin_;i--)
#define REP(i, n) FOR(i,0,n)
#define IREP(i, n) IFOR(i,0,n)
#define ALL(x) (x).begin(), (x).end()

class Solution {
public:
    int c2i(char c)
    {
        if (c == 'Q') return 0;
        if (c == 'W') return 1;
        if (c == 'E') return 2;
        if (c == 'R') return 3;
        return -1;
    }
    int balancedString(string s) {
        vector<int> v(s.length());
        REP(i, s.length()) v[i] = c2i(s[i]);
        
        int N = s.length();
        vector<int> count_(4);
        int ret = N;

        int r = N - 1;
        REP(l, N)
        {
            while (*max_element(ALL(count_)) <= N / 4 and r >= 0)
            {
                count_[v[r]]++;
                r--;
            }
            while (*max_element(ALL(count_)) > N / 4 and r < N - 1)
            {
                r++;
                count_[v[r]]--;
            }
            if (*max_element(ALL(count_)) > N / 4) break;
            ret = min(ret, r - l + 1);
            count_[v[l]]++;
        }
        return ret;
    }
};
