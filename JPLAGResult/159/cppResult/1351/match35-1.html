class Solution {
public:
    int sum[100];
    int cnt[100];
    
    int balancedString(string s) {
        for (int i = 0; i < s.size(); i++)
            sum[s[i]]++;
        int mid = s.size()/4;
        for (int i = 0; i < 100; i++)
            sum[i] -= mid;
        

        int sta = 0;
        int ans = s.size();
        for (int i = 0; i < s.size(); i++) {
            cnt[s[i]]++;
            while (sta < s.size() && isok()) {
                ans = min(ans, i - sta + 1);
                cnt[s[sta]]--;
                sta++;
            }
        }
        return ans;
    }
    
    bool isok() {
        for (int i = 0; i < 100; i++)
            if (sum[i] - cnt[i] > 0)
                return false;
        return true;
    }
};
