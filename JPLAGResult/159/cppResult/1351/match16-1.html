class Solution {
    
    const unordered_map<char, int> char_map{{'Q', 0}, {'W', 1}, {'E', 2}, {'R', 3}};
    array<int, 4> a;
    
    
    bool good(array<int, 4>& a, int n) {
        assert(n % 4 == 0);
        return *max_element(a.begin(), a.end()) <= n / 4;
    }
    
    bool can(string& s, int len) {
        int n = s.length();
        
        array<int, 4> b = a;
        for (int i = 0; i < len; i++) {
            b[char_map.at(s[i])]--;
        }
        

        if (good(b, n))
            return true;
        
        for (int i = 1; i + len - 1 < n; i++) {
            b[char_map.at(s[i - 1])]++;
            b[char_map.at(s[i + len - 1])]--;
            if (good(b, n))
                return true;
        }
        
        return false;
    }
public:
    
    Solution() {
        a = {};
    }
    int balancedString(string s) {
        int n = s.length();
        
        for (auto const& c : s) {
            a[char_map.at(c)]++;
        }
        
        if (good(a, n))
            return 0;
        

        int l = 1, r = n;
        
        while (l <= r) {
            int mid = (l + r) / 2;
            
            if (can(s, mid))
                r = mid - 1;
            else 
                l = mid + 1;
        }
        
        return l;
    }
};
