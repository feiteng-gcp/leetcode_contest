class Solution {
public:
	bool ok(int x, string s) {
		map<char, int> m;
		for (auto c : s) {
			m[c]++;
		}
		for (int i = 0; i < x; i++) {
			m[s[i]]--;
		}
		int n = s.length();
		if (m['Q'] <= n / 4 && m['W'] <= n / 4 && m['E'] <= n / 4 && m['R'] <= n / 4) {
			return true;
		}
		for (int i = 1; i + x <= n; i++) {
			m[s[i + x - 1]]--;
			m[s[i - 1]]++;
			if (m['Q'] <= n / 4 && m['W'] <= n / 4 && m['E'] <= n / 4 && m['R'] <= n / 4) {
				return true;
			}
		}
		return false;
	}
	int balancedString(string s) {
		int l = 0;
		int r = s.length();
		while (l < r) {
			int mid = (l + r ) / 2;
			if (ok(mid, s)) {
				r = mid;
			}
			else {
				l = mid + 1;
			}
		}
		return l;
	}
};
