class Solution {
public:
    inline int abs(int x){return x>0?x:-x;}
    int balancedString(string s) {
        int q=0,w=0,e=0,r=0,l=s.length(),n=l/4;
        for(char c:s)switch(c)
        {
            case 'Q':q++;break;
            case 'W':w++;break;
            case 'E':e++;break;
            case 'R':r++;break;
        }
        int i=0,j=0,Q=q,W=w,E=e,R=r,ans=9999999;
        while(j<=l)
        {
            while((i<=j)&&(q<=n)&&(w<=n)&&(e<=n)&&(r<=n))
            {
                ans=min(ans,j-i);
                switch(s[i])
                {
                    case 'Q':q++;break;
                    case 'W':w++;break;
                    case 'E':e++;break;
                    case 'R':r++;break;
                }
                i++;
            }
            switch(s[j])
            {
                case 'Q':q--;break;
                case 'W':w--;break;
                case 'E':e--;break;
                case 'R':r--;break;
            }
            j++;
        }
        return ans;
    }
};
