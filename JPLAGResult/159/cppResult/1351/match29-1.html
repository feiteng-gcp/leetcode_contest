class Solution {
public:
  int balancedString(string s) {
    int n=s.size();
    vector<int> Q(n+1,0);
    vector<int> W(n+1,0);
    vector<int> E(n+1,0);
    vector<int> R(n+1,0);
    for(int i=0;i<n;i++){
      Q[i+1]=Q[i]+(s[i]=='Q');
      W[i+1]=W[i]+(s[i]=='W');
      E[i+1]=E[i]+(s[i]=='E');
      R[i+1]=R[i]+(s[i]=='R');
    }

    int l=-1,r=n+1;
    while(l+1<r){
      int m=(l+r)>>1;

      int flg=0;
      for(int i=0;i+m<=n;i++){
        int qq=Q[n]-(Q[i+m]-Q[i]);
        int ww=W[n]-(W[i+m]-W[i]);
        int ee=E[n]-(E[i+m]-E[i]);
        int rr=R[n]-(R[i+m]-R[i]);
        if(max({qq,ww,ee,rr})<=n/4) flg=1;
      }
      if(flg) r=m;
      else l=m;
    }
    return r;
  }
};
