class Solution {
public:
    int Q[100100],W[100100],E[100010],R[100010],am,n;
    bool judge(int x){
        for(int l=1;l+x-1<=n;l++){
            int RR=l+x-1;
            int q=Q[l-1]+Q[n]-Q[RR];
            int w=W[l-1]+W[n]-W[RR];
            int e=E[l-1]+E[n]-E[RR];
            int r=R[l-1]+R[n]-R[RR];
            q=am-q;w=am-w;e=am-e;r=am-r;
            if(q>=0&&w>=0&&e>=0&&r>=0&&q+w+e+r==x)return true;
        }
        return false;
    }
    int balancedString(string s) {
        n=s.size();
        am=n/4;

        for(int i=1;i<=n;i++){
            Q[i]=Q[i-1];
            W[i]=W[i-1];
            E[i]=E[i-1];
            R[i]=R[i-1];
            if(s[i-1]=='Q')Q[i]++;
            if(s[i-1]=='W')W[i]++;
            if(s[i-1]=='E')E[i]++;
            if(s[i-1]=='R')R[i]++;
        }
        int l=0,r=n,mid;
        while(l<r){
            mid=l+(r-l)/2;
            if(judge(mid))r=mid;
            else l=mid+1;
        }
        return l;
    }
};
