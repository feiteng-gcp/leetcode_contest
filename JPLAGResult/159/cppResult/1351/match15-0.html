class Solution {
public:
    int balancedString(string s) {
        int n = s.size();
        vector<vector<int>> sum(n+1, vector<int>(4));
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < 4; j++) {
                sum[i+1][j] = sum[i][j];
            }
            if(s[i] == 'Q')sum[i+1][0]++;
            if(s[i] == 'W')sum[i+1][1]++;
            if(s[i] == 'E')sum[i+1][2]++;
            if(s[i] == 'R')sum[i+1][3]++;
        }
        int need = 0;
        for(int i = 0; i < 4; i++) {
            if(sum[n][i] > n/4) need += sum[n][i]-n/4;
        }
        int ans = INT_MAX;
        for(int i = 0; i < n; i++) {
            int l = i, r = n, ret = -1;
            while(l <= r) {
                int m = (l+r)/2;
                int cur = 0;
                for(int j = 0; j < 4; j++) {
                    if(sum[n][j] > n/4) {
                        cur += min(sum[m][j]-sum[i][j], sum[n][j]-n/4);
                    }
                }
                if(cur >= need) {
                    ret = m;
                    r = m-1;
                }
                else l = m+1;
            }
            if(ret != -1) {
                ans = min(ans, ret-i);
            }
        }
        return ans;
    }
};
