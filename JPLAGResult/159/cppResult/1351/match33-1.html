class Solution {
public:
    int balancedString(string s) {
        vector<int> a(4,0);

        for(int i=0;i<s.size();i++){
            if(s[i]=='Q') a[0]++;
            else if(s[i]=='W') a[1]++;
            else if(s[i]=='E') a[2]++;
            else if(s[i]=='R') a[3]++;
        }
        vector<int> b(4,0);
        int rmin=0;
        for(int i=0;i<4;i++){
            if(a[i]>s.size()/4){
                rmin+=(a[i]-s.size()/4);
                b[i]=a[i]-s.size()/4;
            }
        }
        vector<int> c(4,0);

        int rmax=0;
        for(int i=0;i<s.size();i++){
            if(s[s.size()-1-i]=='Q') c[0]++;
            else if(s[s.size()-1-i]=='W') c[1]++;
            else if(s[s.size()-1-i]=='E') c[2]++;
            else if(s[s.size()-1-i]=='R') c[3]++;
            int sign=0;
            for(int i=0;i<4;i++){
                if(c[i]>=b[i])
                    sign++;
            }
            if(sign==4){
                rmax=i+1;
                break;
            }
        }
        while(rmax>rmin){
            int m=(rmax+rmin)/2;
            for(int i=0;i<4;i++){
            c[i]=0;
        }

            for(int i=0;i<m;i++){
                if(s[i]=='Q') c[0]++;
                else if(s[i]=='W') c[1]++;
                else if(s[i]=='E') c[2]++;
                else if(s[i]=='R') c[3]++;
            }
            int sign=0;
            for(int i=0;i<4;i++){
                if(c[i]>=b[i])
                    sign++;
            }
            if(sign==4){
                rmax=m;
                continue;
            }
            int i=m;
            while(i<s.size()){
                if(s[i-m]=='Q') {c[0]--;if(c[0]==b[0]-1) sign--;}
                else if(s[i-m]=='W') {c[1]--;if(c[1]==b[1]-1) sign--;}
                else if(s[i-m]=='E') {c[2]--;if(c[2]==b[2]-1) sign--;}
                else if(s[i-m]=='R') {c[3]--;if(c[3]==b[3]-1) sign--;}
                if(s[i]=='Q') {c[0]++;if(c[0]==b[0]) sign++;}
                else if(s[i]=='W') {c[1]++;if(c[1]==b[1]) sign++;}
                else if(s[i]=='E') {c[2]++;if(c[2]==b[2]) sign++;}
                else if(s[i]=='R') {c[3]++;if(c[3]==b[3]) sign++;}
                if(sign==4){
                    rmax=m;
                    break;
                }
                i++;
            }
            if(i==s.size()){
                rmin=m+1;
            }
        }
        return rmin;
    }
};
