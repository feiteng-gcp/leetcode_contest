class Solution {
public:
bool curIsLarger(unordered_map<char,int> & target, unordered_map<char,int> & cur) {
    return cur['Q'] >= target['Q'] && cur['W'] >= target['W'] && cur['E'] >= target['E'] && cur['R'] >= target['R'];
}

int balancedString(string s) {
    int n = s.size();
    int m = n / 4;
    int Q = 0;
    int W = 0;
    int E = 0;
    int R = 0;

    for (int i = 0; i < s.size(); i++) {
        if (s[i] == 'Q') {
            Q++;
        } else if (s[i] == 'W') {
            W++;
        } else if (s[i] == 'E') {
            E++;
        } else {
            R++;
        }
    }
    if (Q == m && W == m && E == m && R == m)
        return 0;
    unordered_map<char,int> target;
    target['Q'] = (max(Q-m,0));
    target['W']=(max(W-m,0));
    target['E']=(max(E-m,0));
    target['R']=(max(R-m,0));
    
    int result = INT_MAX;
    int p1 = 0;
    int p2 = 0;
    unordered_map<char,int> cur;
    while (p1 < s.size()) {
        cur[s[p1]]++;
        if (curIsLarger(target, cur)) {
            while (curIsLarger(target, cur) && p2 <= p1) {
                cur[s[p2]]--;
                p2++;
            }
            result = min(result, p1 - p2 + 2);
        }
        p1++;
    }
    return result;
}
};
