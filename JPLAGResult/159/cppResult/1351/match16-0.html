class Solution {
public:
    string S;
    int N;
    map<char,int> D;
    bool ok(map<char,int> &M, int x) {
        int t = 0;
        //printf("M : "); for (auto pr : M) printf("(%c: %d)", pr.first, pr.second); printf("\n");
        //printf("D : "); for (auto pr : D) printf("(%c: %d)", pr.first, pr.second); printf("\n");
        for (auto pr : M) {
            char c = pr.first;
            if (M[c] + D[c] >= 0) t += M[c] + D[c];
            else return false;
        }
        //printf("x: %d, t: %d\n", x, t);
        return t == x;
    }
    
    bool can(int x) {
        map<char,int> M;
        M['Q'] = M['W'] = M['E'] = M['R'] = 0;
        for (int i=0;i<x;++i) M[S[i]]++;
        if (ok(M, x)) return true;
        for (int i=x;i<N;++i) {
            M[S[i-x]]--;
            M[S[i]]++;
            if (ok(M, x)) return true;
        }
        return false;
    }
    
    int balancedString(string _s) {
        D['Q'] = D['W'] = D['E'] = D['R'] = 0;
        S = _s;
        N = S.size();
        for (auto c : S) D[c]--;
        for (auto &pr : D) pr.second += N/4;
        //printf("D : "); for (auto pr : D) printf("(%c: %d)", pr.first, pr.second); printf("\n");
        int lo = 0, hi = N, mid;
        while (hi > lo) {
            int mid = (hi + lo) / 2;
            if (can(mid)) hi = mid;
            else lo = mid+1;
        }
        return hi;
    }
};
