class Solution {
public:
  int sidx(char c) {
    if (c == 'Q') return 0;
    if (c == 'W') return 1;
    if (c == 'E') return 2;
    if (c == 'R') return 3;
    assert(false);
    return -1;
  }
  int balancedString(string s) {
    int n = s.size();
    int ans = n;
    vector<int> cnt[4];
    for (int i = 0; i < 4; i++) cnt[i].push_back(0);
    for (int i = 0; i < s.size(); i++) {
      for (int j = 0; j < 4; j++) {
        cnt[j].push_back(cnt[j][i]);
      }
      cnt[sidx(s[i])][i+1]++;
    }
    
    for (int i = 0; i < s.size(); i++) {
      int cm = 0;
      for (int j = 0; j < 4; j++) {
        int l = i;
        int r = n;
        while (l < r) {
          int mid = (l + r) / 2;
          if (cnt[j][i] + cnt[j][n] - cnt[j][mid] > n/4) {
            l = mid + 1;
          } else {
            r = mid;
          }
        }
        if (cnt[j][i] + cnt[j][n] - cnt[j][l] <= n/4) {
          cm = max(cm, l-i);
        } else {
          cm = n;
        }
      }
      ans = min(ans, cm);
    }
    return ans;
  }
};
