class Solution {
public:
    int convert(char ch){
        if (ch == 'Q') return 0;
        if (ch == 'W') return 1;
        if (ch == 'E') return 2;
        if (ch == 'R') return 3;
        return 0;
    }
    
    int check(vector<int>count, vector<int>nums){
        for (int t = 0; t < 4; t++)
            if (nums[t] > 0){
                if (count[t] < nums[t])
                    return false;
            }
        return true;
    }
    int balancedString(string s) {
        vector<int> nums(4, 0);
        for (int i = 0;i < s.length();i++)

            nums[convert(s[i])]++;
        vector<int> count(4, 0);
        for (int t = 0; t < 4;t++)
            nums[t] -= s.length() / 4;
        int left = 0;
        int ans = s.length();
        for (int i = 0; i < s.length();i++){
            count[convert(s[i])]++;
            count[convert(s[left])]--;
            while(left < i && check(count, nums)){
                left++;
                count[convert(s[left])]--;
            }
            if (check(count, nums))
                return 0;
            count[convert(s[left])]++;
            if (check(count, nums))
                ans = min(ans, i - left + 1);
        }
        return ans;
    }
};
