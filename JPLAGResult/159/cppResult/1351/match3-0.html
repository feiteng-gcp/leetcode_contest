char cmap[4] = {'Q', 'W', 'E', 'R'};
class Solution {
public:
    bool test(unordered_map<char, int>& extra, unordered_map<char, int>& m) {
        for (auto& p : extra) {
            if (m[p.first] < p.second) return false;
        }
        return true;
    }
    int balancedString(string s) {
        int cnt[4] = {0, 0, 0, 0};
        for (auto c : s) {
            if (c == 'Q') ++cnt[0];
            if (c == 'W') ++cnt[1];
            if (c == 'E') ++cnt[2];
            if (c == 'R') ++cnt[3];
        }
        int expect = s.size() / 4;
        int diff[4];
        int min_mod = 0;
        unordered_map<char, int> extra;
        for (int i = 0; i < 4; ++i) {
            diff[i] = cnt[i] - expect;
            if (diff[i] > 0) {
                extra[cmap[i]] = diff[i];
            }
        }
        if (extra.size() == 0) return 0;
        int l = -1, r = 0;
        unordered_map<char, int> m;
        int res = INT_MAX;
        while (l < (int)s.size()) {
            while (!test(extra, m) && r < s.size()) {
                ++m[s[r]];
                ++r;
            }
            if (!test(extra, m)) break;
            res = min(res, r - l - 1);
            ++l;
            --m[s[l]];
        }
        return res;
    }
};
