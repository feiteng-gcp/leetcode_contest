class Solution {
public:
    int balancedString(string s) {
        int pre[s.size()+1][4];
        int ans = 1e9;
        for(int i=0;i<4;i++){
            pre[0][i] = 0;
        }
        for(int i=0;i<s.size();i++){
            for(int j=0;j<4;j++){
                pre[i+1][j] = pre[i][j];
            }
            if(s[i] == 'Q')pre[i+1][0]++;
            if(s[i] == 'W')pre[i+1][1]++;
            if(s[i] == 'E')pre[i+1][2]++;
            if(s[i] == 'R')pre[i+1][3]++;
        }
        int l =0;
        int r = s.size();
        while(l<=r){
            int m = (l+r)/2;
            int valid = 0;
            for(int i=0;i<s.size()-m+1;i++){
                int ma = 0;
                for(int j=0;j<4;j++){
                    ma = max(ma, pre[s.size()][j]+pre[i][j]-pre[i+m][j]);
                }
                int mov = 0;
                for(int j=0;j<4;j++){
                    mov += ma - (pre[s.size()][j]+pre[i][j]-pre[i+m][j]);
                }
                // cout<<ma<<":"<<mov<<"\n";
                
                if(m >= mov && (m-mov) % 4==0){
                    valid = 1;
                    break;
                }
            }
            // cout<<m<<" "<<valid<<"\n";
            if(valid){
                r = m-1;
            }else{
                l = m+1;
            }
        }
        return l;
    }
};
