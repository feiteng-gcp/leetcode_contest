class Solution {
public:
    int balancedString(string s) {
        int n = s.size();
        vector<int> a(n);
        for (int i = 0; i < n; ++ i)
        {
            if (s[i] == 'Q') a[i] = 0;
            if (s[i] == 'W') a[i] = 1;
            if (s[i] == 'E') a[i] = 2;
            if (s[i] == 'R') a[i] = 3;
        }
        
        vector<vector<int>> b(n+1, vector<int>(4));
        for (int i = 1; i <= n; ++ i)
        {
            b[i] = b[i-1];
            b[i][a[i-1]] ++;
        }
        
        auto check = [&](int m)
        {
            for (int i = 1; i+m-1 <= n; ++ i)
            {
                int j = i+m-1;
                vector<int> c = b[n];
                for (int k = 0; k < 4; ++ k)
                    c[k] -= b[j][k]-b[i-1][k];
                if (max({c[0], c[1], c[2], c[3]}) <= n/4) return true;
            }
            return false;
        };
        
        int L = 0, R = n-1;
        int ret = n;
        while (L <= R)
        {
            int m = (L+R)/2;
            //printf("%d %d %d\n", L, R, m);
            if (check(m))
            {
                ret = m;
                R = m-1;
            }
            else
                L = m+1;
        }
        return ret;
    }
};
