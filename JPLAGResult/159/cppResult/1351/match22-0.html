class Solution {
public:
    bool possible(string &s, int n, int q, int w, int e, int r, int l){
        int q1=0, w1=0, e1=0, r1=0;
        for(int i=0; i<l; i++){
            if(s[i]=='Q') q1++;
            if(s[i]=='E') e1++;
            if(s[i]=='W') w1++;
            if(s[i]=='R') r1++;
        }
        int m=n>>2;
        if(m+q1>=q && m+w1>=w && m+e1>=e && m+r1>=r) return true;
        for(int i=l, j=0; i<n; i++,j++){
            if(s[i]=='Q') q1++;
            if(s[i]=='E') e1++;
            if(s[i]=='W') w1++;
            if(s[i]=='R') r1++;
            
            if(s[j]=='Q') q1--;
            if(s[j]=='E') e1--;
            if(s[j]=='W') w1--;
            if(s[j]=='R') r1--;
            
            if(m+q1>=q && m+w1>=w && m+e1>=e && m+r1>=r) return true;
        }
        
        return false;
    }

    int balancedString(string s) {
        int n=s.length();
        int q=0, w=0, e=0, r=0;
        for(int i=0; i<n; i++){
            if(s[i]=='Q') q++;
            if(s[i]=='E') e++;
            if(s[i]=='W') w++;
            if(s[i]=='R') r++;
        }
        
        int l=0, h=n-1;
        while(l<h){
            int m=(l+h)>>1;
            if(possible(s,n,q,w,e,r,m)){
                h=m;
            } else l=m+1;
        }
        if(!possible(s,n,q,w,e,r,l)) l++;
        return l;
    }
};
