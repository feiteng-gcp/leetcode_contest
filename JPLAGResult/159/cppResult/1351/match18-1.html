class Solution {
public:
    int convert(char c) {
        if (c == 'Q') return 0;
        if (c == 'W') return 1;
        if (c == 'E') return 2;
        return 3;
    }
    

    bool check(int *a, int *b) {
        for (int i = 0; i < 4; i++) {
            if (a[i] > b[i]) {
                return false;
            }
        }
        return true;
    }
    
    int balancedString(string s) {
        int stat[4] = {0};
        int cur[4] = {0};
        int l = 0, r = 0;
        int ans = INT_MAX;
        for (char c : s) {
            stat[convert(c)]++;
        }
        for (int i = 0; i < 4; i++) {
            if (stat[i] >= s.length() >> 2) {
                stat[i] -= (s.length() >> 2);
            } else {
                stat[i] = 0;
            }
        }

        while (r < s.length()) {
            while (!check(stat, cur) && r < s.length()) {
                cur[convert(s[r++])]++;
            }
            
            while (check(stat, cur) && r >= l) {
                ans = min(ans, r - l);
                cur[convert(s[l++])]--;
            }
        }
        return ans;
    }
};
