class Solution {
public:
	int q[100010],w[100010],
		e[100010],r[100010];
    int n;
	bool test(int m){//cout<<"test "<<m<<":"<<endl;
		for(int i=0;i<=n-m;i++){
			int j=i+m;
			if(q[n]-q[j]+q[i]<=n/4
			&& w[n]-w[j]+w[i]<=n/4
			&& e[n]-e[j]+e[i]<=n/4
			&& r[n]-r[j]+r[i]<=n/4)
				return true;
		}
		return false;
	}
	int balancedString(string s){
		n=s.size();
		q[0]=w[0]=e[0]=r[0]=0;
		for(int i=1;i<=n;i++){
			q[i]=q[i-1]+(s[i-1]=='Q');
			w[i]=w[i-1]+(s[i-1]=='W');
			e[i]=e[i-1]+(s[i-1]=='E');
			r[i]=r[i-1]+(s[i-1]=='R');
		}
		int l=0,r=n,m;
		while(l<=r){
			m=(l+r)/2;
			if(test(m))
				r=m-1;
			else
				l=m+1;
		}
		//cout<<w[n]-w[6]+w[4]<<endl;
		return l;
	}
};
