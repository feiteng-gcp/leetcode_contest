class Solution {
public:
    int balancedString(string s) {
        int n = s.size();
        vector<int> q(n+1),w(n+1),e(n+1),r(n+1);
        
        for (int i = 0; i < n; ++i) {
            q[i+1] = q[i] + (s[i] == 'Q');
            w[i+1] = w[i] + (s[i] == 'W');
            e[i+1] = e[i] + (s[i] == 'E');
            r[i+1] = r[i] + (s[i] == 'R');
        }
        
        int best = n;
        
        for (int i = 0; i < n; ++i) {
            for (int j = min(n, i + best); j >= i; --j) {
            
                auto Q = q[n] - q[j] + q[i] - q[0];
                auto W = w[n] - w[j] + w[i] - w[0];
                auto E = e[n] - e[j] + e[i] - e[0];
                auto R = r[n] - r[j] + r[i] - r[0];
                if (Q > n/4)break;
                if (W > n/4)break;
                if (E > n/4)break;
                if (R > n/4)break;
                
                if (Q <= n / 4  && R <= n /4 && E <= n/4 && W <= n/ 4) {
                    best = min(best, j-i);
                }
            }
        }
        
        return best;
        
    }
};
