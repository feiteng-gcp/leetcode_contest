class Solution {
public:
    // q, w, e, r
    int sumq[100010], sumw[100010], sume[100010], sumr[100010];
    int n;
    bool ok(int m){
        int k = n/4;
        for(int i = 1; i <= n - m + 1; ++i){
            int t = sumq[i+m-1]-sumq[i-1]+sume[i+m-1]-sume[i-1]+sumw[i+m-1]-sumw[i-1]+sumr[i+m-1]-sumr[i-1];
            if(sumq[n] - (sumq[i+m-1]-sumq[i-1]) > k) continue;
            if(sumw[n] - (sumw[i+m-1]-sumw[i-1]) > k) continue;
            if(sume[n] - (sume[i+m-1]-sume[i-1]) > k) continue;
            if(sumr[n] - (sumr[i+m-1]-sumr[i-1]) > k) continue;
            if(t <= m) return true;
        }
        return false;
    }
    int balancedString(string s) {
        n = s.length();
        for(int i = 0; i < s.length(); ++i){
            sumq[i+1] = sumq[i] + (s[i] == 'Q');
            sumw[i+1] = sumw[i] + (s[i] == 'W');
            sume[i+1] = sume[i] + (s[i] == 'E');
            sumr[i+1] = sumr[i] + (s[i] == 'R');
        }
        int l = 0, r = s.length();
        int ans = 0;
        while(l <= r){
            int m = l + r >> 1;
            if(ok(m)){
                ans = m;
                r = m - 1;
            }
            else l = m + 1;
        }
        return ans;
    }
};
