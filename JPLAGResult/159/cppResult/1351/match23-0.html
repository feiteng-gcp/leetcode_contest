bool ok(vector<int>& arr){
    if(arr[0]<=0 && arr[1]<=0 && arr[2]<=0 &&arr[3]<=0){
        return true;
    }
    return false;
}
class Solution {
public:
    int balancedString(string s) {
        int balanced = s.length()/4;
        vector<int> arr(4,0);
        map<char,int> mp;
        mp['Q'] = 0;mp['W'] = 1;mp['E'] = 2;mp['R'] = 3;
        
        for(auto i:s){
            arr[mp[i]]++;
        }
        
        for(int i=0;i<4;++i){arr[i] = arr[i]-balanced;}
        
        int l = 0;
        int r = 0;
        int res = s.length();
        
        while(r<s.length()){
            if(ok(arr)){
                res = min(res,r-l);
                arr[mp[s[l]]]++;
                l++;
            }else{
                arr[mp[s[r]]]--;
                r++;
            }  
        }
        while(ok(arr)){
            res = min(res,r-l);
            arr[mp[s[l]]]++;
            l++;            
        }
        
        return res;

    }
};
