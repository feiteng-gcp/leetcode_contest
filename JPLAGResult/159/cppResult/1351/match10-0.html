class Solution {
public:
    int balancedString(string s) {
        int c = s.size()/4;
        int c0 = 0;
        int c1= 0;
        int c2= 0;
        int c3 = 0;
        for(char ns: s){
            if( ns == 'Q')
                c0++;
            else if (ns == 'W')
                c1++;
            else if (ns == 'E')
                c2++;
            else
                c3++;
        } 
        c0 = max(0, c0 - c);
        c1 = max(0, c1 - c);
        c2 = max(0, c2 - c);
        c3 = max(0, c3 - c);
        if(check(c0,c1,c2,c3)) return 0;
        int end = 0;
        int res = INT_MAX;
        for(int start = 0; start < s.size(); start++){
            if(start >0){
                char ps = s[start - 1];
                if( ps == 'Q')
                    c0++;
                else if (ps == 'W')
                    c1++;
                else if (ps == 'E')
                    c2++;
                else
                    c3++;    
            }
            while(end < s.size() && !check(c0,c1,c2,c3)){
                char ns = s[end];
                if( ns == 'Q')
                    c0--;
                else if (ns == 'W')
                    c1--;
                else if (ns == 'E')
                    c2--;
                else
                    c3--;
                end++;
            }
            if(check(c0,c1,c2,c3))
                res = min(res, end - start);
        }
        return res;
        
    }
    bool check(int c0, int c1, int c2, int c3){
        if(c0 <= 0 && c1 <= 0 && c2<=0 && c3<=0)
            return true;
        else 
            return false;
    }
};
