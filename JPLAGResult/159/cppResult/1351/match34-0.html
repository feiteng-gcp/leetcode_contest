class Solution {
public:
    int balancedString(string s) {
        unordered_map<char, int> count;
        for (char c:s) {
            count[c]++;
        }
        int ret=INT_MAX;
        int j=0;
        for (int i=0; i<s.size(); i++) {
            while(j<s.size() && max(max(count['Q'], count['W']), max(count['E'], count['R']))>s.size()/4) {
                count[s[j++]]--;
            }
            if (max(max(count['Q'], count['W']), max(count['E'], count['R']))==s.size()/4) {
                ret=min(ret, j-i);
            }
            count[s[i]]++;
        }
        return ret;
    }
};
