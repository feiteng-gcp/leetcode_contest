class Solution {
public:
    int balancedString(string s) {
        int N = s.size() / 4;
        unordered_map<char, int> count = {};
        for (char c : s) {
            count[c]++;
        }
        unordered_map<char, int> to_remove = {};
        int total_to_remove = 0;
        for (auto& entry: count) {
            to_remove[entry.first] = max(0, entry.second - N);
            total_to_remove += to_remove[entry.first];
        }
        if (total_to_remove == 0) {
            return 0;
        }

        int start = 0, res_len = s.size();
        for (int i = 0; i < s.size(); ++i) {
            if (to_remove[s[i]]-- > 0) {
                total_to_remove--;
            }
            while (total_to_remove == 0) {
                res_len = min(res_len, i - start + 1);
                if (to_remove[s[start++]]++ == 0) {
                    total_to_remove ++;
                }
            }
        }
        return res_len;
    }
};
