class Solution {
public:
    int cnt[100010][4];
    int n;
    int check(int len) {
        for (int st = 1 ; st+len-1 <= n ; ++st) {
            // printf("len:%d st:%d\n", len, st);
            int in[4] = {0};
            int tmp = 0;
            bool flg = true;
            for (int j = 0 ; j < 4 ; ++j) {
                in[j] = cnt[st+len-1][j] - cnt[st-1][j];
                int out = cnt[n][j] - in[j];
                if (out > n / 4) {flg = false; break;}
                tmp += n / 4 - out;
            }
            if (!flg) continue;
            if (tmp == len) return true;
        }
        return false;
    }
    int balancedString(string s) {
        int avg = s.size() / 4;
        n = s.size();
        memset(cnt, 0, sizeof(cnt));

        for (int i = 0 ; i < s.size() ; ++i) {
            if (s[i] == 'Q') cnt[i+1][0] = 1;
            if (s[i] == 'W') cnt[i+1][1] = 1;
            if (s[i] == 'E') cnt[i+1][2] = 1;
            if (s[i] == 'R') cnt[i+1][3] = 1;
        }
        for (int i = 1 ; i <= s.size() ; ++i) {
            for (int j = 0 ; j < 4 ; ++j)
                cnt[i][j] += cnt[i-1][j];
        }
        int tmp = 0;
        for (int j = 0 ; j < 4 ; ++j) {
            if (cnt[n][j] == avg) ++tmp;
        }
        if (tmp == 4) return 0;

        int ans = s.size();

        int lo = 1, hi = s.size();
        while (lo <= hi) {
            int mid = (lo + hi) / 2;
            if (check(mid)) {
                hi = mid - 1;
                ans = min(ans, mid);
            } else lo = mid + 1;
        }
        return ans;
    }
};
