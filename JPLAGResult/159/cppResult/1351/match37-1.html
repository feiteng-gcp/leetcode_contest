class Solution {
public:
    int ord(const char &c) {
        switch (c) {
            case 'Q': return 0;
            case 'W': return 1;
            case 'E': return 2;
            case 'R': return 3;
        }
        return -1;
    }
    
    bool is_replaceable(const array<int, 4> &count, const array<int, 4> slice, const int &desired) {
        for (int i = 0; i < 4; i++) {
            if (count[i] - slice[i] > desired) {
                return false;
            }
        }
        return true;
    }
    
    bool is_possible(const string &s, const int &k) {
        array<int, 4> count = {0, 0, 0, 0};
        for (const char &c : s) {
            count[ord(c)]++;
        }
        int desired = s.size() / 4;
        array<int, 4> slice = {0, 0, 0, 0};
        for (int i = 0; i < k; i++) {
            slice[ord(s[i])]++;
        }

        if (is_replaceable(count, slice, desired)) {
            return true;
        }
        for (int i = k; i < s.size(); i++) {
            slice[ord(s[i - k])]--;
            slice[ord(s[i])]++;
            if (is_replaceable(count, slice, desired)) {
                return true;
            }
        }
        return false;
    }
    
    int balancedString(string s) {
        int L = 0, R = s.size(), res;
        while (L <= R) {
            int M = L + (R - L) / 2;
            if (is_possible(s, M)) {
                res = M;
                R = M - 1;
            } else {
                L = M + 1;
            }
        }
        return res;
    }
};
