class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        sort(begin(folder), end(folder));
        vector<string> res;
        unordered_set<string> se;
        for(auto &s:folder){
            int n=s.size();
            bool rep=false;
            for(int i=0;i<n;i++){
                if(s[i]=='/' && i>0){
                    if(se.find(s.substr(0,i)) != se.end()){
                        rep=true;break;
                    }
                }
            }
            if(!rep){
                se.insert(s);
                res.push_back(s);
            }
        }
        return res;
    }
};
