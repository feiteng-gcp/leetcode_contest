class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        sort(folder.begin(), folder.end());
        vector<string> res;
        for (string& f : folder) {
            if (res.empty()) {
                res.push_back(f);
                continue;
            }
            string& l = res.back();
            if (l.size() >= f.size()) {
                res.push_back(f);
                continue;
            }
            if (l != f.substr(0, l.size()) || f[l.size()] != '/') {
                res.push_back(f);
                continue;
            }
        }
        return res;
    }
};
