class Solution {
private:
    inline bool starts(const string& a, const string& b) {
        for (int i=0; i<a.size(); i++) {
            if (a[i] != b[i]) {
                return false;
            }
        }
        
        return b[a.size()] == '/';
    }
    static bool by_sz(const string& a, const string& b) {
        return a.size() < b.size();
    }

public:
    vector<string> removeSubfolders(vector<string>& folder) {
        sort(folder.begin(), folder.end(), by_sz);
        vector<string> ans;
        for (int i=0; i<folder.size(); i++) {
            bool a = true;
            for (int j=0; j<ans.size(); j++) {
                if (starts(ans[j], folder[i])) {
                    a = false;
                    break;
                }
            }
            if (a) {
                ans.push_back(folder[i]);
            }
        }
        
        return ans;
    }
};
