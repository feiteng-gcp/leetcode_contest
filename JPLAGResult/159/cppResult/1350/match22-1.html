class Solution {
public:
	vector<string> removeSubfolders(vector<string>& folder) {
		int n = folder.size(), i, j, k;
		sort(folder.begin(), folder.end());
		vector<string> ans;
		unordered_set<string> ss;

		for (i = 0; i < n; i++) {
			bool valid = true;

			for (j = 0; j < folder[i].length(); j++) {
				if ((folder[i][j] == '/') && (ss.find(folder[i].substr(0, j + 1)) != ss.end())) {
					valid = false;
					break;
				}
			}
			if (valid) {
				ans.push_back(folder[i]);
				ss.insert(folder[i] + '/');
			}
		}
		return ans;
	}
};
