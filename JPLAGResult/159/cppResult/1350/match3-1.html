class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        vector<string> ans;
        unordered_set<string> visited;
        sort(folder.begin(), folder.end());
        
        for (auto f : folder) {
            string curr = "/";
            bool found = false;
            for (int i = 1; i < f.length(); i++) {

                if (f[i] == '/') {
                    if (visited.find(curr) != visited.end()) {
                        found = true;
                        break;
                    }
                }
                curr.push_back(f[i]);
            }
            if (!found) {
                visited.insert(curr);
                ans.push_back(curr);
            }
        }
        return ans;
    }
};
