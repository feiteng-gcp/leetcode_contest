class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        unordered_set<string> ss;
        sort(folder.begin(),folder.end());
        vector<string> ans;
        for(string s:folder){
            bool removed=false;
            for(int i=1;i<s.size();i++){
                if(s[i]=='/'){
                    if(ss.find(s.substr(0,i))!=ss.end()){
                        removed=true;
                        break;
                    };
                }
            }
            if(!removed){
                ss.insert(s);
                ans.push_back(s);
            }
        }
        return ans;
    }
};
