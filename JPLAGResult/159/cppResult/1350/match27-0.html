class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        vector<string> ans;
        sort(folder.begin(), folder.end());
        int n = folder.size();
        ans.push_back(folder[0]);
        for(int i = 1; i < n; ++i){
            if(folder[i].size() >= ans.back().size()){
                if(folder[i].substr(0, ans.back().size()) == ans.back() && folder[i][ans.back().size()] == '/') continue;
                else ans.push_back(folder[i]);
            }
            else {
                ans.push_back(folder[i]);
            }
        }
        return ans;
    }
};
