class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        map<string,int>mp;
        vector<string>ans;
        sort(folder.begin(),folder.end());
        for(int i=0; i<folder.size(); i++){
            string str="";
            int flag=1;
            for(int j=0; j<folder[i].length(); j++){
                if(folder[i][j]=='/' && mp[str]==1){
                    flag=0;
                    break;
                }
                str+=folder[i][j];
            }
            if(flag==1){
                ans.push_back(folder[i]);
                mp[folder[i]]=1;
            }
        }
        return ans;
    }
};
