class Solution {
public:
    vector<string> split(const string& s) {
        vector<string> ret;
        int n = s.size();
        for (int i = 0, j; i < n; i = j) {
            ++i;
            for (j = i; j < n && s[j] != '/'; ++j);
            ret.push_back(s.substr(i, j - i));
        }
        return ret;
    }
    vector<string> removeSubfolders(vector<string>& folder) {
        sort(folder.begin(), folder.end(), [](const string& s, const string& t) {
            return s.size() < t.size();
        });
        vector<string> ret;
        set<vector<string>> A;
        for (auto& s : folder) {
            auto cur = split(s);
            int m = cur.size();
            vector<string> nxt;

            bool found = false;
            for (int k = 0; k < m - 1; ++k) {
                nxt.push_back(cur[k]);
                if (A.count(nxt)) {
                    found = true;
                    break;
                }
            }
            if (!found) {
                ret.push_back(s);
                A.insert(cur);
            }
        }
        return ret;
    }
};
