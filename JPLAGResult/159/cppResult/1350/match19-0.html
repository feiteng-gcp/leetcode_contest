class Solution {
public:
    vector<string> removeSubfolders(vector<string>& a) {
        sort(a.begin(),a.end());
        vector<string>ret;
        unordered_set<string>st;
        for(int i=0;i<a.size();i++){
            bool f = false;
            for(int j=0;j+1<a[i].size();j++){
                if(a[i][j+1]=='/'&&st.find(a[i].substr(0,j+1))!=st.end()){
                    f=true;
                    break;
                }
            }
            if(!f){
                st.insert(a[i]);
                ret.push_back(a[i]);
            }
        }
        return ret;
    }
};
