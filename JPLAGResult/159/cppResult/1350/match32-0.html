class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        unordered_set<string> result;
        sort(folder.begin(),folder.end());
        for(int i=0;i<folder.size();i++){
            bool folderExist=false;
            for(int j=1;j<folder[i].size();j++){
                if(folder[i][j]=='/'){
                    if(result.find(folder[i].substr(0,j))!=result.end()){
                        folderExist=true;
                        break;
                    }
                }
            }
            if(!folderExist){
                result.insert(folder[i]);
            }
        }
        vector<string> res;
        for(auto p:result){
            res.push_back(p);
        }
        return res;
    }
};
