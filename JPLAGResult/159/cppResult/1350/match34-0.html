class Trie
{
public:
    bool flag=0;
    unordered_map<string,Trie*> next;
};

class Solution {
public:
    
    vector<string> separate(string folder)
    {
        vector<string> ret;
        string cur = "";
        for(int i=0;i<folder.size();i++)
        {
            if(folder[i] == '/')
            {
                if(cur.size()) ret.push_back(cur);
                cur = "";
                continue;
            }
            cur += folder[i];
        }
        if(cur.size()) ret.push_back(cur);
        return ret;
    }
    
    void build(Trie* root, vector<string>& sub)
    {
        for(int i=0;i<sub.size();i++)
        {
            if(root->next.find(sub[i]) != root->next.end())
                root = root->next[sub[i]];
            else
            {
                root->next[sub[i]] = new Trie();
                root = root->next[sub[i]];
            }
        }
        root->flag=true;
        root->next.clear();
        return;
    }
    
    void solve(Trie* root, string cur, vector<string> &ret)
    {
        if(root->flag)
        {
            ret.push_back(cur);
            return;
        }
        for(auto it: root->next)
        {
            string tmp = cur + "/" + it.first;
            solve(it.second,tmp,ret);
        }
        return;
    }
    
    vector<string> removeSubfolders(vector<string>& f) {
        
        Trie *root = new Trie();
        
        for(int i=0;i<f.size();i++)
        {
            vector<string> sub = separate(f[i]);
            build(root, sub);
        }
        vector<string> ret;
        solve(root,"",ret);
        return ret;
    }
};
