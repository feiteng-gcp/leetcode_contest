class Solution {
public:
    vector<string> removeSubfolders(vector<string>& f) {
        map<string,int> m;m.clear();
        vector<string> ans;ans.clear();
        sort(f.begin(),f.end());
        for(auto s:f){
            int n = s.size();
            int subf = 0;
            for(int i=1;i<n;i++){
                if(s[i]=='/'){
                    string t=s.substr(0,i);
                    if(m[t]) {
                        subf = 1;
                        break;
                    }
                }
            }
            if(!subf){
                m[s] = 1;
                ans.pb(s);
            }
        }
        return ans;
    }
};
