bool cmp(const string& a, const string& b) {
  if (a.length() != b.length()) return a.length() < b.length();
  return a < b;
}

class Solution {
public:
  vector<string> removeSubfolders(vector<string>& folders) {
    sort(folders.begin(), folders.end(), cmp);
    
    set<string> parents;
    
    for (string& f : folders) {
      bool found = false;
      
      string tmp;

      for (char ch : f) {
        if (ch == '/') {
          if (parents.find(tmp) != parents.end()) {
            found = true;
            break;
          }
        }
        tmp.push_back(ch);
      }
      
      if (!found) {
        parents.insert(f);
      }
    }
    
    return vector<string>(parents.begin(), parents.end());
  }
};
