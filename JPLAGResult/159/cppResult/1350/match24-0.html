class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        sort(folder.begin(), folder.end());
        set<string> res;
        for (auto &f: folder) {
            bool isSubfolder = false;
            for (int i = 0; i < f.length(); i++) {
                if (f[i] == '/' && res.count(f.substr(0, i)) > 0) {
                    isSubfolder = true;
                    break;
                }
            }
            if (!isSubfolder)
                res.insert(f);
        }
        vector<string> ans(res.begin(), res.end());
        return ans;
    }
};
