class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        sort(folder.begin(), folder.end());
        vector<string> ans;
        ans.push_back(folder[0]);
        int last = 0;
        for (int i = 1; i < folder.size(); ++i)
        {
            if (folder[i].find(folder[last]) == 0 && folder[i].size() > folder[last].size() && folder[i][folder[last].size()] == '/')
            {
                continue;
            }
            else
            {
                ans.push_back(folder[i]);
                last = i;
            }
        }
        return ans;
    }
};
