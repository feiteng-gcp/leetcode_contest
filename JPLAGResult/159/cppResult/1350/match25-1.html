class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        sort(folder.begin(), folder.end());
        
        vector<string> res;
        for (auto& f : folder) {

            if (res.empty()) {
                res.emplace_back(f);
                continue;
            }
            if (f.size() > res.back().size() && f.substr(0, res.back().size()) == res.back() && f[res.back().size()] == '/') {
                continue;
            }
            res.emplace_back(f);
        }
        
        return res;
    }
};
