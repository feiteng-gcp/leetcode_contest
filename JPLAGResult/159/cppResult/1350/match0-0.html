class Solution {
    struct Dir {
        string name;
        unordered_map<string, Dir*> subs;
        bool is_end = false;
        void add(const vector<string>& layer, int idx) {
            if (idx == layer.size()) {
                is_end = true;
                return;
            }
            if (!subs.count(layer[idx])) {
                subs[layer[idx]] = new Dir;
                subs[layer[idx]]->is_end = false;
                subs[layer[idx]]->name = layer[idx];
            }
            subs[layer[idx]]->add(layer, idx + 1);
        }
        void traverse(vector<string>& all, string total) {
            if (is_end) {
                all.push_back(total + '/' + name);
            } else {
                for (auto& [k, v] : subs) {
                    v->traverse(all, total + '/' + name);
                }
            }
        }
    };
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        Dir root;
        for (auto& s : folder) {
            string acc = "";
            vector<string> names;
            for (auto c : s) {
                if (c != '/') {
                    acc.push_back(c);
                } else {
                    if (acc != "") {
                        names.push_back(acc);
                    }
                    acc = "";
                }
            }
            if (acc != "") names.push_back(acc);
            root.add(names, 0);
        }
        vector<string> res;
        for (auto& [k, v] : root.subs) {
            v->traverse(res, "");
        }
        return res;
    }
};
