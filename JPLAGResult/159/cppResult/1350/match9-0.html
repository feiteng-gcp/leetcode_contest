class Solution {
public:
    static bool cmp(string &s1, string &s2) {
        return s1.size() < s2.size();
    }
    vector<string> removeSubfolders(vector<string>& f) {
        sort(f.begin(), f.end(), cmp);
        vector<string> res;
        unordered_map<string, int> h;
        for(int i = 0; i < f.size(); i++) {
            string s = f[i];
            bool in = true;
            for(int j = 1; j < s.size(); j++) {
                if(s[j] == '/') {
                    if(h[s.substr(0, j)]) {
                        in = false;
                        break;
                    }
                }
            }
            if(in) {
                res.push_back(s);
                h[s] = 1;
            }
        }
        return res;
    }
};
