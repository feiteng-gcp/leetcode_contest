class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        vector<string> res;
        
        set<string> st;
        sort(folder.begin(), folder.end());
        
        for (auto& s : folder) {
            int n = s.length();
            string t = "/";
            bool valid = true;
            for (int i = 1; i <= n; ++i) {
                if (i == n || s[i] == '/') {
                    if (st.count(t) > 0) {
                        valid = false;
                        break;
                    }
                }
                if (i < n) {
                    t += s[i];
                }
            }
            if (valid) {
                st.insert(s);
                res.push_back(s);
            }
        }
        
        return res;
    }
};
