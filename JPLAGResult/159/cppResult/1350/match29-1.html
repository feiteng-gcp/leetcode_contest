class Solution {
public:
    
    vector<string> removeSubfolders(vector<string>& A) {
        set<string> st;
        int n = A.size();
        sort(A.begin(), A.end());
        vector<string> ans;

        for(auto &s: A){
            // cout<<"New string "<<s<<endl;
            int m = s.length();
            string folder = "/";
            bool found = false;
            for(int i=1; i<m; i++){
                // cout<<folder<<endl;
                if(s[i] == '/'){
                    if(st.find(folder) != st.end()){
                        found = true;
                        break;
                    }  
                }
                folder+=s[i];
            }
            if(!found){
                if(st.find(folder) == st.end()){
                    st.insert(folder);
                    ans.push_back(folder);
                }
            }
        }
        return ans;
    }
};
