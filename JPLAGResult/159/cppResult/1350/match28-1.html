class Solution {
public:
    vector<string> removeSubfolders(vector<string>& a) {
        sort(a.begin(),a.end());
        set<string>s;
        vector<string>ret;

        for(int i=0;i<a.size();i++){
            string tmp="";
            bool f=true;
            for(int j=0;j<a[i].size();j++){
                tmp+=a[i][j];
                if(j+1==a[i].size()||a[i][j+1]=='/'){
                    if(s.find(tmp)!=s.end()){
                        f=false;
                        break;
                    }
                }   
            }
            if(!f)continue;
            s.insert(a[i]);
            ret.push_back(a[i]);
        }
        return ret;
    }
};
