class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        map<string,int> mp;
        for(auto &z:folder)mp[z]=1;
        
        for(auto &z:mp)
        {
            string s=z.first;
            string t;
            bool ok=1;
            for(int i=0;i<s.size();i++)
            {
                if(s[i]=='/')
                {
                    if(mp[t]!=0)
                    {
                        ok=0;break;
                    }
                }
                t.push_back(s[i]);
                
            }
            if(!ok)mp[s]=-1;
            s.clear();t.clear();
        }
        vector<string> v;
        for(auto &z:mp)if(z.second==1)v.push_back(z.first);
        return v;
    }
};
