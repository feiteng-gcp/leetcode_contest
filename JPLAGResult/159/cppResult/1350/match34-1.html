class Solution {
public:
vector<string> removeSubfolders(vector<string>& folder) {
    unordered_set<string> ss;
    vector<string> folder2;
    for (auto & s : folder) {
        bool canRemove = false;
        for (int i = 1; i < s.size(); i++) {
            if (s[i] == '/') {
                if (ss.count(s.substr(0, i))) {
                    canRemove = true;
                    break;
                }
            }
        }
        if (!canRemove) {
            ss.insert(s);
            folder2.push_back(s);
        }
    }
    unordered_set<string> sss;
    vector<string> folder3;
    reverse(folder2.begin(), folder2.end());
    for (auto & s : folder2) {

        bool canRemove = false;
        for (int i = 1; i < s.size(); i++) {
            if (s[i] == '/') {
                if (sss.count(s.substr(0, i))) {
                    canRemove = true;
                    break;
                }
            }
        }
        if (!canRemove) {
            sss.insert(s);
            folder3.push_back(s);
        }
    }
    return folder3;
}
};
