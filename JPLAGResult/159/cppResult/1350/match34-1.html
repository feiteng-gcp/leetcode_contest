class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        sort(folder.begin(), folder.end());
        vector<string> res;
        string s = folder[0];
        res.push_back(s);
        for (int i = 1; i < folder.size(); i++) {
            if (s.length() < folder[i].length() && folder[i].substr(0, s.length()) == s && folder[i][s.length()] == '/') {
                continue;
            }
            res.push_back(folder[i]);
            s = folder[i];
        }
        return res;
    }
};
