bool comp(string a,string b){
    return a.length() < b.length();
}
class Solution {
public:
    vector<string> removeSubfolders(vector<string>& folder) {
        vector<string> res;
        sort(folder.begin(),folder.end(),comp);
        
        set<string> st;
        for(auto i:folder){
            bool f = true;
            string tmp = "";
            for(int j=0;j<i.size();++j){
                tmp += i[j];
                if(i[j] != '/'){
                    if(st.find(tmp)!=st.end() && j<i.size()-1 && i[j+1]=='/'){
                        f = false;
                        break;
                    }
                }
                
            }
            if(f){
                st.insert(i);
            }
        }
        //["/a","/a/b","/c/d","/c/d/e","/c/f"]
        for(auto i:st){
            res.push_back(i);
        }
        
        return res;
    }
};
