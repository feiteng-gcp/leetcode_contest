struct Trie{
    unordered_map<string, Trie *> children;
    bool is_path = false;
    void insert(vector<string> & path) {
        Trie * cur = this;
        for(auto const &  x:path) {
            if(!cur->children.count(x)) {
                cur->children[x] = new Trie();
            }
            cur = cur->children[x];
        }
        cur->is_path = true;
    }
    
    bool find(vector<string> & path) {
        Trie * cur = this;
        for(auto const &  x:path) {
            if(cur->children.count(x)==0) {
                return cur->is_path;
            }
            cur = cur->children[x];
        }
        return is_path;
    }
    
    
};

class Solution {
public:
    vector<string> split (const string &s, char delim) {
        vector<string> result;
        stringstream ss (s.substr(1));
        string item;
        while (getline (ss, item, delim)) {
            result.push_back (item);
        }

        return result;
    }
    vector<string> removeSubfolders(vector<string>& folder) {
        Trie trie;
        sort(folder.begin(), folder.end());
        vector<string> res;
        for(auto & x:folder) {
            auto path = split(x, '/');
            if(trie.find(path))
                continue;
            trie.insert(path);
            res.push_back(x);
        }
        return res;
    }
};
