class Solution {
public:
    map<string,bool>mp;
    vector<string> removeSubfolders(vector<string>& folder) {
        sort(folder.begin(),folder.end());
        vector<string>ans;
        for(int i=0;i<folder.size();i++){
            int j,flag=1;

            for(j=1;j<folder[i].size();j++){
                if(folder[i][j]=='/'){
                    string tmp=folder[i].substr(0,j);
                    cout<<tmp<<"\n";
                    if(mp.count(tmp)){
                        flag=0;
                        break;
                    }
                }
            }
            if(flag){
                ans.push_back(folder[i]);
                mp[folder[i]]=1;
            }
        }
        return ans;
    }
};
