class Solution {
public:
  int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
    vector<int> allTimes;
    for (int x : startTime) allTimes.push_back(x);
    for (int x : endTime) allTimes.push_back(x);
    sort(allTimes.begin(), allTimes.end());
    allTimes.resize(unique(allTimes.begin(), allTimes.end()) - allTimes.begin());
    
    int n = startTime.size();
    int tn = allTimes.size();
    vector<vector<int>> jobs(tn);
    for (int i = 0; i < n; i++) {
      jobs[lower_bound(allTimes.begin(), allTimes.end(), endTime[i])-allTimes.begin()].push_back(i);
    }
    
    vector<int> ans(tn);
    for (int i = 0; i < tn; i++) {
      if (i) ans[i] = max(ans[i], ans[i-1]);
      for (int j : jobs[i]) {
        int sj = lower_bound(allTimes.begin(), allTimes.end(), startTime[j])-allTimes.begin();
        ans[i] = max(ans[i], ans[sj] + profit[j]);
      }
    }
    return ans[tn-1];
  }
};
