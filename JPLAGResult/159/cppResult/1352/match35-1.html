class Solution {
public:
    
    

    int NonConflict(vector<vector<int>>& jobs, int i)
    {
        for (int j = i - 1; j >= 0; --j)
        {
            if (jobs[j][0] <= jobs[i][1]) return j;
        }
        return -1;
    }
    
    int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
        
        vector< vector<int> > jobs;
        for (int i = 0; i < endTime.size(); ++i)
        {
            jobs.push_back( {endTime[i], startTime[i], profit[i]} );
        }
        sort(jobs.begin(), jobs.end());
        
        int n = jobs.size();

        vector<int> dp(n, 0);
        dp[0] = jobs[0][2];
        
        for (int i = 1; i < n; ++i)
        {
            int prof = jobs[i][2];
            int index = NonConflict(jobs, i);
            if (index != -1)
                prof += dp[index];
            
            dp[i] = max(prof, dp[i - 1]);
        }
                        
        return dp[n - 1];
    }
};
