class Solution {
private:
    vector<int>c;
    void init(int sz){
        c = vector<int>(sz+1, 0);
    }
    void upd(int x, int val){
        for (; x < c.size(); x += x & -x) c[x]= max(c[x], val);

    }
    int query(int x){
        int ret = 0;
        while(x>0){
            ret=max(ret, c[x]);
            x -= x & -x;
        }
        return ret;
    }
    struct P{
        int l,r,v;
        P(){}

        P(int l,int r,int v):l(l),r(r),v(v){}
        bool operator<(const P& a)const {
            return r<a.r;
        }
    };
public:

    int jobScheduling(vector<int>& s, vector<int>& e, vector<int>& p) {
        vector<P> a;
        int n = s.size();
        vector<int>tt;
        for(int i=0;i<n;i++){
            tt.push_back(s[i]);
            tt.push_back(e[i]);
            a.push_back(P(s[i],e[i],p[i]));
        }
        sort(tt.begin(),tt.end());
        tt.erase(unique(tt.begin(),tt.end()),tt.end());
        init(tt.size());
        for(int i=0;i<n;i++){
            a[i].l = lower_bound(tt.begin(),tt.end(),a[i].l) - tt.begin()+1;
            a[i].r = lower_bound(tt.begin(),tt.end(),a[i].r) - tt.begin()+1;
        }
        int ret = 0;
        sort(a.begin(),a.end());
        for(int i=0;i<a.size();i++){
            int pre = query(a[i].l);
            int cur = pre + a[i].v;
            ret=max(ret,cur);
            upd(a[i].r,cur);
        }
        return ret;
    }
};
