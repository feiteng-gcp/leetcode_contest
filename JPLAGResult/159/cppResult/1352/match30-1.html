class Solution {
public:
    int jobScheduling(vector<int>& startTime,
                      vector<int>& endTime,
                      vector<int>& profit) {
      vector<int> t;
      t.insert(t.begin(), startTime.begin(), startTime.end());
      t.insert(t.begin(), endTime.begin(), endTime.end());
      sort(t.begin(), t.end());
      t.resize(unique(t.begin(), t.end()) - t.begin());
      
      unordered_map<int, vector<int>> end_time_to_jobs;
      for (int i = 0; i < endTime.size(); ++i) {
        end_time_to_jobs[endTime[i]].push_back(i);
      }
      
      unordered_map<int, int> f;
      f[t[0]] = 0;
      for (int i = 1; i < t.size(); ++i) {
        f[t[i]] = f[t[i-1]];
        
        for (int j : end_time_to_jobs[t[i]]) {
          f[t[i]] = max(f[t[i]], f[startTime[j]] + profit[j]);
        }
      }
      
      int ret = 0;
      for (auto& p : f) ret = max(ret, p.second);
      return ret;
    }
};
