class Solution {
public:
    int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
        int n = startTime.size();
        vector<vector<int>> data(n);
        for(int i=0;i<n;i++) data[i]={startTime[i],endTime[i],profit[i]};
        sort(data.begin(),data.end(),[](auto&a, auto&b){ return a[1]<b[1]; });//by end time
        vector<int> dp(n);
        dp[0]=data[0][2];
        for(int i=1;i<n;i++){
            //find last index end <= cur start
            int j = i-1;
            while(j>=0 && data[j][1]>data[i][0]) --j;
            dp[i] = max(dp[i-1], (j>=0 ? dp[j] : 0)+data[i][2]);
        }
        return dp[n-1];
    }
};
