class Solution {
public:
    int jobScheduling(vector<int>& start, vector<int>& end, vector<int>& p) {
        int n = p.size();
        vector<vector<int>> jobs;
        
        for(int i = 0; i < n; ++i) {
            jobs.push_back({start[i], end[i], p[i]});
        }
        
        sort(jobs.begin(), jobs.end(), [] (vector<int>& v1, vector<int>& v2) {

            return v1[1] < v2[1];
        });
        
        vector<int> dp(n);
        dp[0] = jobs[0][2];
        for(int i = 1; i < n; ++i) {
            int prof = jobs[i][2];
            int l = binarySearch(jobs, i);
            if(l != -1)
                prof += dp[l];
            
            dp[i] = max(dp[i-1], prof);
        }
        return dp[n-1];
    }
    
    int binarySearch(vector<vector<int>>& jobs, int index) {
        int lo = 0, hi = index -1;
        while(lo <= hi) {
            int mid = (lo + hi) / 2;
            if(jobs[mid][1] <= jobs[index][0]) {
                if(jobs[mid+1][1] <= jobs[index][0])
                    lo = mid + 1;
                else
                    return mid;
            }
            else
                hi = mid -1;
        }
        return -1;
    }
};
