class Solution {
public:
    int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
        set<int> ts;
        for(int t:startTime)ts.insert(t);
        for(int t:endTime)ts.insert(t);
        unordered_map<int,int> t2i;
        int tn=0;
        for(int t:ts){
            t2i[t]=tn;
            tn++;
        }
        unordered_map<int,vector<pair<int,int>>> e2ps;
        int n=startTime.size();

        for(int i=0;i<n;i++)e2ps[t2i[endTime[i]]].push_back({profit[i],t2i[startTime[i]]});
        vector<int> dp(tn);
        for(int i=1;i<tn;i++){
            dp[i]=dp[i-1];
            for(auto &p:e2ps[i])
                dp[i]=max(dp[i],dp[p.second]+p.first);
        }
        return dp[tn-1];
    }
};
