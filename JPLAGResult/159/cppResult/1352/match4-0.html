class Solution {
public:
    int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
        vector<tuple<int,int,int>> intervals;
        for(int i=0; i<startTime.size(); i++) {
            intervals.push_back(make_tuple(endTime[i], startTime[i], profit[i]));
        }
        sort(intervals.begin(), intervals.end());
        vector<int> dp(startTime.size());
        int res = 0;
        for(int i=0; i<startTime.size(); i++) {
            int left_most = bs(intervals, get<1>(intervals[i]));
            if(left_most == -1)
                dp[i] = get<2>(intervals[i]);
            else
                dp[i] = get<2>(intervals[i]) + dp[left_most];
            if(i>0)
                dp[i] = max(dp[i-1], dp[i]);
            res = max(res, dp[i]);
        }
        return res;
    }
    
    int bs(vector<tuple<int,int,int>> const & array, int target) {
        int l = 0;
        int r = array.size();
        while(l<r) {
            int mid = (l+r)/2;
            if(get<0>(array[mid])<=target)
                l = mid+1;
            else
                r = mid;
        }
        return l-1;
    }
};
