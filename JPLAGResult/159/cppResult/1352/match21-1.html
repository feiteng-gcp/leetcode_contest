class Solution {
public:
    struct Job { 
        int start, finish, profit; 
    }; 
  

    static bool comp(Job J1, Job J2)  { 
        return (J1.finish < J2.finish); 
    } 
  
    int search(vector<Job>& J, int idx)  { 
        int low = 0, hi = idx - 1; 
        while (low <= hi) { 
            int mid = low + (hi - low) / 2; 
            if(J[mid].finish <= J[idx].start) { 
                if(J[mid + 1].finish <= J[idx].start) {
                    low = mid + 1; 
                } else {
                    return mid;
                }
            } else {
                hi = mid - 1; 
            }
        } 
        return -1; 
    } 
  
    int maxProfit(vector<Job>& J) { 
        sort(J.begin(), J.end(), comp); 
            
        int N = J.size();

        int dp[N]; 
        dp[0] = J[0].profit; 

        for (int i=1;i<N;i++) { 
            int incude = J[i].profit; 
            int latest = search(J,i); //  latest job that doesn't conflict with current job
            if (latest != -1) 
                incude += dp[latest]; 

            dp[i] = max(incude, dp[i-1]); 
        } 
        return dp[N-1]; 
    } 
    
    

    int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
        vector<Job> jobs;
        int N = startTime.size();
        
        for(int i=0;i<N;i++) {
            jobs.push_back({startTime[i],endTime[i],profit[i]});
        }
        
        return maxProfit(jobs);
    }
};
