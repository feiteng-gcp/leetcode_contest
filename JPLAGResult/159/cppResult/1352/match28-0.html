class Solution {
public:
    struct Job { 
        int start;
        int finish;
        int profit; 
    };
    int binary(vector<Job>& jobs, int index) { 
        int lo = 0, hi = index - 1; 

        while (lo <= hi) { 
            int mid = (lo + hi) / 2; 
            if (jobs[mid].finish <= jobs[index].start) { 
            if (jobs[mid + 1].finish <= jobs[index].start) 
                lo = mid + 1; 
            else return mid; 
            } else hi = mid - 1; 
        } 
  
        return -1; 
    }
    static bool cmp(Job j1, Job j2) { 
        return j1.finish < j2.finish; 
    } 
    int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
        int dp[50005];
        memset(dp, 0, sizeof(dp));
        vector<Job> jobs;
        
        for (int i = 0; i < startTime.size(); i++) {
            Job j;
            j.start = startTime[i];
            j.finish = endTime[i];
            j.profit = profit[i];
            jobs.push_back(j);
        }
        sort(jobs.begin(), jobs.end(), cmp);
        dp[0] = jobs[0].profit;
        
        
        for (int i = 1; i < jobs.size(); i++) { 
            int include = jobs[i].profit; 
            int l = binary(jobs, i); 
            if (l != -1) include += dp[l];
            
            dp[i] = max(include, dp[i-1]); 
        } 
        
        return dp[jobs.size() - 1];
    }
};
