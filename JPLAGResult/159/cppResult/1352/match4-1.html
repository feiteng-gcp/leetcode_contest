class Solution {
public:
    int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
        int n = startTime.size();
        vector<pair<pair<int, int> , int> > a;
        for (int i = 0; i < n; i++) a.emplace_back(make_pair(startTime[i], endTime[i]), profit[i]);
        sort(a.begin(), a.end());
        vector<int> dp(n + 1);
        for (int i = n - 1; i >= 0; i--) {
            dp[i] = ((i + 1) < n ? dp[i + 1] : 0);
            int idx = lower_bound(a.begin(), a.end(), make_pair(make_pair(a[i].first.second, -1), -1)) - a.begin();
            dp[i] = max(dp[i], a[i].second + dp[idx]);
        }
        return dp[0];
    }
};
