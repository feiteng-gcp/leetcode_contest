class Solution {
public:
    int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
        int n = endTime.size();
        vector<int> idx(n, 0);
        for (int i = 0; i < n; ++i)
            idx[i] = i;
        sort(idx.begin(), idx.end(), [&endTime](int i, int j){
            return endTime[i] < endTime[j];
        });
        auto latestNonConflict = [&idx, &startTime, &endTime](int i){ 
            for (int j = i - 1; j >= 0; j--) { 
                if (endTime[idx[j]] <= startTime[idx[i]]) 
                    return j; 
            } 

            return -1;
        };
        
        vector<int> dp(n, 0);
        dp[0] = profit[idx[0]];
        for (int i = 1; i < n; ++i)
        {
            int curr = profit[idx[i]];
            int last = latestNonConflict(i);
            if (last != -1)
                curr += dp[last];
            dp[i] = max(curr, dp[i-1]);
        }
        return dp[n-1];
    }
};
