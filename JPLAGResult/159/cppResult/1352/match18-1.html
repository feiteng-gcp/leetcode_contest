class Solution {
public:
    int jobScheduling( vector<int>& startTime, vector<int>& endTime, vector<int>& profit ) 
    {
        vector<vector<int>> jobs;
        for( int i =  0; i < startTime.size(); i++ )
        {
            vector<int> temp;
            temp.push_back( startTime[i] );
            temp.push_back( endTime[i] );
            temp.push_back( profit[i] );
            jobs.push_back( temp );
        }
        
        sort( jobs.begin(), jobs.end(), []( vector<int> &a, vector<int>& b ){
            return a[1] < b[1];
        } );
        
        int n = jobs.size();
        
        return helper( jobs, n ); 
    }
    
    int helper( vector<vector<int>>& jobs, int n )
    {

        vector<int>dp( n, 0); 
        dp[0] = jobs[0][2]; 

        for (int i=1; i<n; i++) 
        { 
            int pro = jobs[i][2]; 
            int l = last(jobs, i); 
            if (l != -1) 
                pro += dp[l]; 

            dp[i] = max(pro, dp[i-1]); 
        } 

        int result = dp[n-1]; 


        return result;  
    }
        
    int last( vector<vector<int>>& jobs, int i )
    {
        for (int j = i-1; j>=0; j--) 
        { 
            if ( jobs[j][1] <= jobs[i][0] ) 
                return j; 
        } 
        return -1; 
    } 
};
