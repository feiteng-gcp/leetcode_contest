class Solution {
public:
    int jobScheduling(vector<int>& startTime, vector<int>& endTime, vector<int>& profit) {
        int n = startTime.size();
        
        vector<vector<int>> arr;
        for (int i = 0; i < n; i++) arr.push_back({startTime[i], endTime[i], profit[i]});
        
        sort(arr.begin(), arr.end(), [](vector<int>& a, vector<int>& b) {
            return a[1] < b[1];
        });
        
        vector<int> dp(n, 0);
        dp[0] = arr[0][2];
        
        for (int i = 1; i < n; i++) {
            int p = arr[i][2];
            int j = helper(arr, i);
            if (j != -1) p += dp[j];
            dp[i] = max(p, dp[i-1]);
        }
        
        return dp[n-1];
    }
    
    
    int helper(vector<vector<int>>& arr, int i) {
        for (int j = i - 1; j >= 0; j--) {
            if (arr[j][1] <= arr[i][0]) return j;
        }
        return i;
    }
};
