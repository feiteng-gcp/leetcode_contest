class Solution {
public:
    bool checkStraightLine(vector<vector<int>>& coordinates) {
        double slope = numeric_limits<double>::quiet_NaN();
        sort(coordinates.begin(), coordinates.end(), [](const auto& l, const auto& r){
            return tie(l[0], l[1]) < tie(r[0], r[1]);
        });
        int x1 = coordinates[0][0], y1 = coordinates[0][1];
        int x2 = coordinates[1][0], y2 = coordinates[1][1];
        if (x2 != x1)
            slope = 1.0 * (y2 - y1) / (x2 - x1);
        for (int i = 2; i < coordinates.size(); ++i) {
           int x2 = coordinates[i][0], y2 = coordinates[i][1];
           if (x2 != x1 && isnan(slope)) return false;
           if (x2 != x1) {
               if (1.0 * (y2 - y1) / (x2 - x1) != slope) {
                   return false;
               }
           }
        }
        return true;
    }
};
