class Solution {
public:
    bool checkStraightLine(vector<vector<int>>& points) {
        sort(points.begin(), points.end());
        if(points.size()<=2)
            return true;
        int res = 0;
        int i=0;
        map<pair<int,int>, int> nums;
        int overlap = 1;
        int localmax = 0;
            
        for(int j=1; j<points.size(); j++) {
                if(points[i][0]==points[j][0]&&points[i][1]==points[j][1]) {
                    overlap++;
                    continue;
                }
                auto pr = cal_pr(points[i], points[j]);
                nums[pr]++;
                localmax = max(localmax, nums[pr]);
            }
            res = max(res, localmax+overlap);
        return res==points.size();
    }
    
    pair<int,int> cal_pr(vector<int> & left, vector<int> & right) {
        if(left[0]==right[0])
            return make_pair(left[0], 0);
        if(left[1]==right[1])
            return make_pair(0, left[1]);
        int x = left[0]-right[0];
        int y = left[1]-right[1];
        int gcf = gcd(x,y);
        x/=gcf;
        y/=gcf;
        return make_pair(x,y);

    }

};
