class Solution {
public:
    bool checkStraightLine(vector<vector<int>>& c) {
        int x1 = c[1][0] - c[0][0];
        int y1 = c[1][1] - c[0][1];
        for (int i = 2;i < c.size();i++) {
            int x2 = c[i][0] - c[0][0];
            int y2 = c[i][1] - c[0][1];
            if (y1 * x2 != x1 * y2)
                return false;
        }
        return true;
    }
};
