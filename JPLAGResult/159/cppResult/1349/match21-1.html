class Solution {
public:
bool checkStraightLine(vector<vector<int>>& coordinates) {
    for (int i = 2; i < coordinates.size(); i++) {
        int a = coordinates[i][0] - coordinates[0][0];
        int b = coordinates[i][0] - coordinates[1][0];
        int c = coordinates[i][1] - coordinates[0][1];
        int d = coordinates[i][1] - coordinates[1][1];
        if (a * d != b * c)
            return false;
    }
    return true;
}
};
