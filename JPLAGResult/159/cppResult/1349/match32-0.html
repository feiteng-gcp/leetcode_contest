string to_string(string s){return s;}

template<class T>
string to_string(vector<T>v){
	string res;
	res+="{";
	rep(i,v.size()){
		if(i)res+=",";
		res+=to_string(v[i]);
	}
	res+="}";
	return res;
}

template<class T>
ostream&operator<<(ostream&os,vector<T>v){
	cout<<to_string(v);
	return os;
}

ostream&operator<<(ostream&os,TreeNode*t){
	map<int,int>mp;
	function<void(TreeNode*,int)>dfs=[&](TreeNode*t,int k){
		if(!t)return;
		mp[k]=t->val;
		dfs(t->left,k*2+1);
		dfs(t->right,k*2+2);
	};
	dfs(t,0);
	int Max=0;
	for(auto p:mp)Max=max(Max,p.first);
	vector<string>ans(Max+1,"null");
	for(auto p:mp)ans[p.first]=to_string(p.second);
	os<<ans;
	return os;
}

class Solution {
public:
    bool checkStraightLine(vector<vector<int>> coordinates) {
        sort(coordinates.begin(),coordinates.end());
        for(int i=1;i+1<coordinates.size();i++){
			int x1=coordinates[i][0]-coordinates[i-1][0];
			int y1=coordinates[i][1]-coordinates[i-1][1];

			int x2=coordinates[i+1][0]-coordinates[i][0];
			int y2=coordinates[i+1][1]-coordinates[i][1];

			if(x1*y2-x2*y1!=0)return false;
		}
		return true;
    }
};
