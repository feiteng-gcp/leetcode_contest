class Solution {
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
        vector<string> result;
        for (auto& w : words) {
            if (check(w, pattern)) result.push_back(w);
        }
        return result;
    }
    
    bool check(string w, string p) {
        int len = w.length();
        unordered_map<char, char> s;
        unordered_map<char, char> t;
        for (int i = 0; i < len; i++) {
            char a = w[i];
            char b = p[i];
            if (s.find(a) == s.end() && t.find(b) == t.end()) {
                s[a] = b;
                t[b] = a;
            } else if (s.find(a) != s.end() && t.find(b) != t.end()) {
                if (s[a] != b || t[b] != a) return false;
            } else {
                return false;
            }
        }
        return true;
    }
};
