class Solution {
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
        vector<string> ret;
        for (const auto word : words)
            if (fun(word, pattern))
                ret.push_back(word);
        return ret;
    }
    
    bool fun(const string &word, const string &pattern)
    {
        if (word.size() != pattern.size())
            return false;
        map<char, char> m1, m2;
        for (auto i = 0; i < word.size(); i++)
            if (m1.find(pattern[i]) != m1.end())
            {
                if (m1[pattern[i]] != word[i])
                    return false;
            }
            else
            {
                if (m2.find(word[i]) != m2.end())
                    return false;
                m1[pattern[i]] = word[i];
                m2[word[i]] = pattern[i];
            }
                
        return true;
    }
};
