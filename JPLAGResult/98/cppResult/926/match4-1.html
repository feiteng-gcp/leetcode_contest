#define SZ(a) int((a).size())

class Solution {
public:
   vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
      vector<string> res;
      for (string w : words) {

         map<char,char> P;
         set<char> mapped_to;
         bool ok = true;
         for (int i = 0; i < SZ(w); ++i) {
            if (P.count(w[i])) {
               if (P[ w[i] ] != pattern[i]) {
                  ok = false;
                  break;
               }
            }
            else {
               if (mapped_to.count(pattern[i])) {
                  ok = false;
                  break;
               }
               mapped_to.insert( pattern[i] );
               P[ w[i] ] = pattern[i];
            }
         }
         if (ok)
            res.push_back(w);
      }
      return res;
   }
};
