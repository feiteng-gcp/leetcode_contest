class Solution {
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
        vector<string> result;

        for(int i = 0; i < words.size(); i++){
            map<char,char> flex;
            set<char> used;
            bool flag = true;
            for(int j = 0; j < words[i].size(); j++){
                if(flex.find(words[i][j]) != flex.end()){
                   if(flex[words[i][j]] != pattern[j]){
                      flag = false;
                      break;
                   }
                }
                else if(used.find(pattern[j]) == used.end()){
                   flex[words[i][j]] = pattern[j];
                   used.insert(pattern[j]);
                }
                else{
                   flag = false;
                   break;
                }
            }
            if(flag){
               result.push_back(words[i]);
            }
            flex.clear();
        }
        return result;
    }
};
