class Solution {
    bool is_match(string &s, string &p) {
        if (s.size() != p.size()) return false;
        unordered_map<char, char> p_to_s, s_to_p;

        for (int i = 0; i < p.size(); ++i) {
            if (p_to_s.count(p[i]) && p_to_s[p[i]] != s[i]) return false;
            if (s_to_p.count(s[i]) && s_to_p[s[i]] != p[i]) return false;
            p_to_s[p[i]] = s[i], s_to_p[s[i]] = p[i];
        }
        return true;
    }
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
        vector<string> res;
        for (string &word : words) {
            if (is_match(word, pattern)) res.push_back(word);
        }
        return res;
    }
};
