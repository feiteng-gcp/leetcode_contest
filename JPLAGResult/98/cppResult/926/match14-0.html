class Solution {
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
        vector<string> res;
        for(auto& str : words){
            map<char, char> reflect;
            map<char, char> reflect2;
            bool isgood = true;
            for(int i = 0; i < str.size(); i++){
                if(reflect.count(str[i]) == 1){
                    if(reflect[str[i]] != pattern[i]){
                        isgood = false;
                    }
                }else{
                    reflect[str[i]] = pattern[i];
                }
                
                if(reflect2.count(pattern[i]) == 1){
                    if(reflect2[pattern[i]] != str[i]){
                        isgood = false;
                    }
                }else{
                    reflect2[pattern[i]] = str[i];
                }
                
            }
            
            if(isgood){
                res.push_back(str);
            }
        }
        return res;
    }
};
