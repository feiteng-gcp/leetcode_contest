class Solution {
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) 
    {
        unordered_map<char,char> ab;
        unordered_map<char,char> ba;
        
        vector<string> ret;
        for(auto word : words)
        {
            ab.clear();
            ba.clear();
            if(word.size()!=pattern.size()) continue;
            int len=word.size();
            int i;

            for(i=0;i<len;i++)
            {
                char a=word[i];
                char b=pattern[i];
                if(ab.find(a)==ab.end()&&ba.find(b)==ba.end())
                {
                    ab[a]=b;
                    ba[b]=a;
                }
                else if(ab.find(a)!=ab.end()&&ba.find(b)!=ba.end()&&ab[a]==b&&ba[b]==a) {}
                else break;
            }
            if(i==len) ret.push_back(word);
        }
        return ret;
    }
};
