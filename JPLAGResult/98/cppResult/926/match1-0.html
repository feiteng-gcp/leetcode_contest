class Solution {
    bool check(string a, string b) {
        vector<int> mp(256, -1);
        vector<int> used(256, 0);
        
        for (int i = 0; i < a.length(); ++i) {
            if (mp[a[i]] != -1) {
                if (mp[a[i]] == b[i]) continue;
                return false;
            } else if (used[b[i]]) {
                return false;
            } else {
                mp[a[i]] = b[i];
                used[b[i]] = 1;
            }
        }
        
        return true;
    }
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
        vector<string> res;
        for (string s : words) {
            if (check(s, pattern)) res.push_back(s);
        }
        
        return res;
    }
};
