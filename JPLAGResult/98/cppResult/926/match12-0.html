class Solution {
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) 
    {
        vector<string> res;
        int n = words.size();
        for (int i = 0; i < n; i++)
        {
            vector<int> temp(26, -1);
            vector<int> dic(26, -1);
            bool find = true;
            for (int j = 0; j < words[i].size(); j++)
            {

                if (temp[pattern[j] - 'a'] == -1)
                {
                    temp[pattern[j] - 'a'] = words[i][j] - 'a';
                }
                else if (temp[pattern[j] - 'a'] != words[i][j] - 'a')
                {
                    find = false;
                    break;
                }
                if (dic[words[i][j] - 'a'] == -1)
                {
                    dic[words[i][j] - 'a'] = pattern[j] - 'a';
                }
                else if (dic[words[i][j] - 'a'] != pattern[j] - 'a')
                         {
                             find = false;
                             break;
                         }
                         
            }
            if (find)
                res.push_back(words[i]);

        }
        return res;
    }
};
