class Solution {
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
      vector<string> answer;
      answer.reserve(words.size());
      for (string& word : words) {
        unordered_map<char, char> c;
        unordered_map<char, char> cc;
        bool mismatch = false;
        if (word.size() != pattern.size()) continue;
        for(int i = 0; i < word.size(); i++) {
          if (c.count(word[i])) {
            if (c[word[i]] != pattern[i]) {
              mismatch = true;
              break;
            }
          } else {
            c[word[i]] = pattern[i];
          }
          
          if (cc.count(pattern[i])) {
            if (cc[pattern[i]] != word[i]) {
              mismatch = true;
              break;
            }
          } else {
            cc[pattern[i]] = word[i];
          }
        }
        if (!mismatch) {
          answer.push_back(word);
        }
      }
      return answer;
    }
};
