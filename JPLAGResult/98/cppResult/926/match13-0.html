class Solution {
public:
    vector<string> findAndReplacePattern(vector<string> & words, string pattern) {
        vector<string> res;
        for (auto str : words) {
            if (str.length() == pattern.length()) {
                bool flag = true;
                unordered_map<char, char> wtp, ptw;
                for (int i = 0; i < str.length(); i ++) {
                    auto iter1 = wtp.find(str[i]);
                    auto iter2 = ptw.find(pattern[i]);
                    if (iter1 == wtp.end() && iter2 == ptw.end()) {
                        wtp[str[i]] = pattern[i];
                        ptw[pattern[i]] = str[i];
                    }
                    else if (iter1 == wtp.end() || iter2 == ptw.end()) {
                        flag = false;
                        break;
                    }
                    else {
                        if (iter1->second != pattern[i] || iter2->second != str[i]) {
                            flag = false;
                            break;
                        }
                    }
                }
                if (flag) res.push_back(str);
            }
        }
        return res;
    }
};
