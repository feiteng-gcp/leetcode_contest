class Solution {
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
        vector<string> ans;
        for(auto emt: words) {
            int mp[300] = {0};
            bool occ[300] = {0};
            if(emt.length() != pattern.length()) continue;
            bool get = true;
            for(int i = 0 ; i < emt.length() ; i++) {
                if(mp[emt[i]] == 0 && occ[pattern[i]] == 0) {
                    mp[emt[i]] = pattern[i];
                    occ[pattern[i]] = 1;
                } else if(mp[emt[i]] == 0) {
                    get = false;
                    break;

                } else if(pattern[i] != mp[emt[i]]){
                    get = false;
                    break;
                }
            }
            if(get) ans.push_back(emt);
        }
        return ans;
    }
};
