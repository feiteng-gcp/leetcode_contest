class Solution {
    vector<vector<bool> > signature(string s) {
        vector<vector<bool> > sig(s.size(), vector<bool>(s.size(), false));
        for (int i = 0; i < s.size(); ++i) {
            for (int j = 0; j < s.size(); ++j) {
                if (s[i] == s[j]) sig[i][j] = true;
            }
        }
        return sig;
    }
public:
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
        vector<vector<bool> > sigPattern = signature(pattern);
        vector<string> results;
        for (auto word: words) {
            vector<vector<bool> > sigWord = signature(word);
            if (sigWord == sigPattern) results.push_back(word);
        }
        return results;
    }
};
