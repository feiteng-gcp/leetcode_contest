class Solution {
public:
    vector<string> findAndReplacePattern(vector<string>& w, string p) {
        int n = w.size();
        vector<string> res;
        for (auto& a : w) {
            if (check(a, p) && check(p, a)) {
                res.push_back(a);
            }
        }
        return res;
    }
    
    bool check(string& a, string&b) {
        unordered_map<char, char> mp;
        for (int i = 0; i < a.size(); ++i) {
            if (mp.find(a[i]) != mp.end()) {
                if (b[i] != mp[a[i]]) return false;
            } else {
                mp[a[i]] = b[i];
            }
        }
        return true;
    }
};
