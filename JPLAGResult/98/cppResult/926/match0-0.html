class Solution {
public:
    
    bool match(string s, string t) {
        if (s.size()!=t.size()) return false;
        vector<int> lf(128), rt(128);
        for (int i=0; i<s.size(); ++i) {
            if (lf[s[i]] && lf[s[i]]!=t[i]) return false;
            if (rt[t[i]] && rt[t[i]]!=s[i]) return false;
            lf[s[i]] = t[i];
            rt[t[i]] = s[i];
        }
        return true;
    }
    
    vector<string> findAndReplacePattern(vector<string>& words, string pattern) {
        vector<string> ans;
        for (auto s : words)
            if (match(s, pattern))
                ans.push_back(s);
        return ans;
    }
};
