typedef long long llong;
typedef vector<int> VI;
typedef vector<VI> VVI;

const int MAXN = 20000;
const int MAXA = 20000;
llong POW2[MAXN+1];

const int MOD = 1e9 + 7;
class Solution {
public:
   int sumSubseqWidths(vector<int>& A) {
      VI freq(MAXA+1);
      for (int x : A)
         ++freq[x];

      VI PS(MAXA+2);
      for (int i = 1; i <= MAXA; ++i)
         PS[i] = PS[i-1] + freq[i];

      VI SS(MAXA+2);
      for (int i = MAXA; i >= 1; --i)
         SS[i] = SS[i+1] + freq[i];


      POW2[0] = 1;
      for (int n = 1; n <= MAXN; ++n)
         POW2[n] = (POW2[n-1] * 2) % MOD;

      llong res = 0;
      for (int x = 1; x <= MAXA; ++x) {
         if (freq[x] == 0) continue;
         llong cnt = ( POW2[ PS[x-1] ] * ( POW2[ freq[x] ] - 1 ) ) % MOD;
         res += cnt * x;
         res %= MOD;
      }

      for (int x = 1; x <= MAXA; ++x) {
         if (freq[x] == 0) continue;
         llong cnt = ( POW2[ SS[x+1] ] * ( POW2[ freq[x] ] - 1 ) ) % MOD;
         res -= cnt * x;
         res %= MOD;
         if (res < 0)
            res += MOD;
      }
      
      return res;
   }
};
