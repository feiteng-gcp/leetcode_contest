#define LL long long
class Solution {
public:
    LL multi(int n)
    {
        LL mod = 1000000007;
        LL r = 1;
        LL mul = 2;
        while (n > 0)
        {
            if (n % 2)
                r = (r * mul) % mod;
            mul *= mul;
            mul %= mod;
            n /= 2;
        }
        return r;
    }
    int sumSubseqWidths(vector<int>& A) {
        int n = A.size();
        LL *a = new LL[n];
        LL mod = 1000000007;
        for (int i = 0; i < n; ++i)
            a[i] = A[i];
        sort(a, a + n);
        LL ans = 0;
        for (int i = n - 1; i >= 0; --i)
        {
            ans += (multi(i) * a[i]) % mod;
            ans -= (multi(n - 1 - i) * a[i]) % mod;
            ans %= mod;
        }
        if (ans < 0) ans += mod;
        return (int)(ans);
    }
};
