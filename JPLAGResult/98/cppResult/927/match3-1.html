class Solution {
    const int BIG_INT = 1000000007;
public:
    int sumSubseqWidths(vector<int>& A) {
        long sum = 0;
        int alen = A.size();
        sort(A.begin(), A.end());
        long pre_sum = A[0];
        long factor = 2;

        for (int i = 1; i < A.size(); ++i) {
            // for (int j = i-1; j >= 0; --j) {
            //     sum = (sum + ((A[i] - A[j]) % BIG_INT * factor)) % BIG_INT;
            //     factor = factor * 2 % BIG_INT;
            // }
            sum = (sum + ((factor - 1) * A[i]) % BIG_INT - pre_sum + BIG_INT) % BIG_INT;
            factor = (factor * 2) % BIG_INT;
            pre_sum = (pre_sum * 2 + A[i]) % BIG_INT;
        }
        return sum % BIG_INT;
    }
};
