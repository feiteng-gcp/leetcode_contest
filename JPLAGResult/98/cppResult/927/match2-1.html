class Solution {
public:
    int sumSubseqWidths(vector<int>& A) {
        sort(A.begin(),A.end());
        long long int sum = 0;
        const int bigmod = 1000000007;
        vector<long long int> mod(A.size());
        mod[0] = 1;
        for(int i = 1 ; i < A.size(); i++){
            mod[i] = (mod[i-1]*2)%bigmod;

        }
        for(int i = 0 ; i < A.size(); i++){
            sum += ((mod[i] + (bigmod - mod[A.size()-1-i]))*A[i]%bigmod);
            sum %= bigmod;
        }
        return sum;
    }
};
