class Solution {
public:
    int sumSubseqWidths(vector<int>& A) {
      vector<long long> deg2;
      long long mod = 1000000007;
      long long deg = 1;
      for (int i = 0; i!=A.size(); i++) {
        deg2.push_back(deg);
        deg=deg*2%mod;
      }
      sort(A.begin(), A.end());
      long long answer = 0;

      long long sum = 0;
      
      for (int i = 0; i<A.size(); i++) {
        answer += (A[i]*(deg2[i]-1) - sum)%mod; 
        sum = (2*sum+A[i]) % mod;
      }
      return ((answer%mod)+mod)%mod;
    }
};
