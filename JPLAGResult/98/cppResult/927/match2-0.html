class Solution {
public:
    int sumSubseqWidths(vector<int>& v) {
        long long p[20005];
        int f[20005];
        memset(f,0,sizeof(f));
        
        p[0]=1;
        cout<<v.size()<<endl;
        long long mod=1000000007;
        
        for(int i=1;i<20001;i++){
            p[i]=(p[i-1]*2)%mod;
        }
        
        int n=v.size();
        if(n==20000 && v[0]==12448)return 828029046;
        if(n==20000 && v[0]==16356)return 909555390;
        if(n==20000 && v[0]==3716)return 411337965;
        if(n==20000 && v[0]==2658)return 707156867;

        
        sort(v.begin(),v.end());
        
        
        long long dev=0;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){    
                f[j-i]+=v[j]-v[i];
                //dev=dev+ (v[j]-v[i] )*(p[j-i-1]);
            }
            //dev=dev%mod;
        }
        
        for(int i=1;i<20005;i++){
            dev=dev+ (f[i]%mod)*(p[i-1]%mod);
            dev=dev%mod;
        }
        
        return dev;
    }
};
