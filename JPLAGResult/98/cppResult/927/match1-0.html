class Solution {
public:
    long long pow(long long x, long long y, int MOD) {
        long long res = 1;
        while (y > 0) {
            if (y % 2 == 1) {
                res = (res * x) % MOD;
            } 
            y /= 2;
            x = (x * x) % MOD;
        }
        return res;
    }
    int sumSubseqWidths(vector<int>& A) {
        sort(A.begin(), A.end());
        int MOD = 1000000000 + 7;
        int n = A.size();
        vector<long long> pre(n);
        vector<long long> post(n);
        long long total = 0;
        for (int i = 0; i < n; i ++) {
            total += A[i];
            pre[i] = total;
        }
        total = 0;
        long long x = 0;
        for (int i = n - 1; i >= 0; i --) {
            total += A[i];
            post[n - 1 - i] = total;
        }
        for (int i = 1; i < n; i ++) {
            total = post[i - 1] - pre[i - 1];
            x += total * pow(2, i - 1, MOD);
            x %= MOD;
        }
        return x;
     }
};
