/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        return  cfpp(pre, post, 0, 0, pre.size());
    }
    TreeNode* cfpp(vector<int>& pre, vector<int>& post, int pre_start, int post_start, int size) {
        if (size == 0) return nullptr;
        TreeNode* head = new TreeNode(pre[pre_start]);
        if (size == 1) return head;
        int left_child = pre[pre_start + 1];
        
        int pos = -1;
        for (int i = post_start; i <= post_start + size; i++) if (post[i] == left_child) {
            pos = i; break;
        }
        
        int left_size = pos - post_start + 1;
        int right_size = size - left_size - 1;
        head->left = cfpp(pre, post, pre_start+1, post_start, left_size);
        head->right = cfpp(pre, post, pre_start+1+left_size, post_start + left_size, right_size);
        return head;
    }
};
