/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        return build(pre, post, 0, pre.size()-1, 0, post.size()-1);
    }
    TreeNode* build(vector<int>& pre, vector<int>& post, int s1, int e1, int s2, int e2)
    {
        if(s1>e1)
            return NULL;
        if(s1==e1)
        {
            return new TreeNode(pre[s1]);
        }
        TreeNode* r = new TreeNode(pre[s1]);
        int i, len;
        for(i = s2; i <= e2; i++)
        {
            if(post[i]==pre[s1+1])
            {
                len = i-s2+1;
                TreeNode* cl = build(pre, post, s1+1, s1+1+len-1, s2, i);
                r->left = cl;
                break;
            }
        }
        TreeNode* cr = build(pre, post, s1+len+1, e1, i+1, e2-1);
        r->right = cr;
        return r;
    }
};
