/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    
    TreeNode* BuildTree( vector<int>& pre, int i, int j, vector<int>& post, int p, int q )
    {
        if ( i > j || p > q )
            return NULL;
        if ( pre[i] != post[q] )
            return NULL;
        
        TreeNode *root = new TreeNode( pre[i] );
        //cout<< "i:" << i << ",j:" << j << ",p:" << p << ",q:" << q << endl;
        
        for ( int k = p; k < q; ++k )
        {
            if ( pre[i+1] == post[k])
            {
                int offset = k-p;
                root->left = BuildTree( pre, i+1, i+1+offset, post, p, k);
                root->right = BuildTree( pre, i+offset+2, j, post, k+1, q-1);
                return root;
            }
        }
        
        return root;
    }
    
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        return BuildTree( pre, 0, pre.size()-1, post, 0, post.size()-1);
    }
};
