/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    int pos2[32];
    int n;


    TreeNode* build(vector<int>& pre, vector<int>& post, int l1, int r1, int l2, int r2) {
        if (l1 > r1) return nullptr;
        TreeNode* rt = new TreeNode(pre[l1]);
        if (l1 == r1) {
            return rt;
        } else {
            int mid2 = pos2[pre[l1 + 1]];
            int len = mid2 - l2 + 1;
            rt->left = build(pre, post, l1 + 1, l1 + len, l2, mid2);
            rt->right = build(pre, post, l1 + len + 1, r1, mid2 + 1, r2 - 1);
            return rt;
        }
    }
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        n = pre.size();
        for (int i = 0; i < n; ++i) {
            pos2[post[i]] = i;
        }

        return build(pre, post, 0, n - 1, 0, n - 1);
    }
};
