/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    TreeNode* recursive(vector<int>& pre, int preIdxStart, int preIdxEnd, vector<int>& post, int postIdxStart, int postIdxEnd) {
        if (preIdxStart >= preIdxEnd) return NULL;
        TreeNode *root = new TreeNode(pre[preIdxStart]);
        TreeNode *left = NULL, *right = NULL;
        int sizeLeft = 0, sizeRight = 0;
        int k = -1, l = -1;
        
        if (preIdxStart + 1 < preIdxEnd) {
            int target = pre[preIdxStart + 1];
            for (int i = postIdxStart; i < postIdxEnd; ++i) {
                if (post[i] == target) {
                    k = i;
                    sizeLeft = i - postIdxStart + 1;
                    break;
                }
            } // size = k + 1
            cout << "left: " << k << ", " << sizeLeft << endl;
            left = recursive(pre, preIdxStart + 1, preIdxStart + 1 + sizeLeft, post, postIdxStart, k + 1);
        }
        
        if (postIdxEnd - 2 >= postIdxStart && postIdxEnd - 2 > k) {
            int target = post[postIdxEnd - 2];
            for (int i = preIdxEnd - 1; i >= preIdxStart; --i) {
                if (pre[i] == target) {
                    l = i;
                    sizeRight = preIdxEnd - i;
                    break;
                }
            }
            cout << "right: " << l << ", " << sizeRight << endl;
            right = recursive(pre, l, preIdxEnd, post, postIdxEnd - 1 - sizeRight, postIdxEnd - 1);
        }
        root->left = left;
        root->right = right;
        return root;
    }
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        return recursive(pre, 0, pre.size(), post, 0, post.size());
    }
};
