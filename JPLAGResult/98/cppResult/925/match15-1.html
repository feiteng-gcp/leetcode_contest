/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        int n = pre.size();

        return construct(pre, 0, n-1, post, 0, n-1);
    }
    
    TreeNode* construct(vector<int>& pre, int s1, int e1, vector<int>& post, int s2, int e2) {
        if (s1 > e1 || s2 > e2) {
            return NULL;
        }
        TreeNode *root = new TreeNode(pre[s1]);
        if (s1 == e1) {
            return root;
        }
        
        int i = s2;
        for (; i <= e2; i++) {
            if (pre[s1+1] == post[i]) {
                break;
            }
        }
        int leftNum = i-s2;
        if (post[i] == post[e2-1]) {
            TreeNode* node = construct(pre, s1+1, e1, post, s2, e2-1);
            root->left = node;
        } else {
            TreeNode* node1 = construct(pre, s1+1, s1+leftNum+1, post, s2, s2+leftNum);
            TreeNode* node2 = construct(pre, s1+leftNum+2, e1, post, s2+leftNum+1, e2-1);
            root->left = node1;
            root->right = node2;
        }
        return root;
    }
};
