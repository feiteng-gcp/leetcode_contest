/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        return helper(pre, 0, pre.size() - 1, post, 0, post.size() - 1);
    }
    
    TreeNode* helper(vector<int>& pre, int prS, int prE, vector<int>& post, int poS, int poE) {
        if (prS > prE || poE < poS) return nullptr;
        TreeNode* root = new TreeNode(pre[prS]);
        prS++;
        poE--;
        if (prS <= prE) {
            int idx = poE;
            while (post[idx] != pre[prS] && idx > poS) {
                idx--;
            }
            root->left = helper(pre, prS, prS + idx - poS, post, poS, idx);
            root->right = helper(pre, prS + idx - poS + 1, prE, post, idx + 1, poE);
        }
        return root;
    }
};
