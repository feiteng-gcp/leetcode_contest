/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        if(pre.size() == 0 && post.size() == 0){
           return NULL;
        }
        if(pre.size() == 1 && post.size() == 1){
           TreeNode *root = new TreeNode(pre[0]);
           return root;
        }
        else{
           TreeNode *root = new TreeNode(pre[0]);
           int lnode = pre[1];
           int len;
           for(len = 0; len < post.size() - 1; len++){
               if(post[len] == lnode){
                  break;
               }
           }
           vector<int> lpre(pre.begin() + 1, pre.begin() + 2 + len);
           vector<int> lpos(post.begin(), post.begin() + len + 1);
           TreeNode *lroot = constructFromPrePost(lpre,lpos);
           vector<int> rpre(pre.begin() + 2 + len,pre.end());
           vector<int> rpos(post.begin() + len + 1,post.end() - 1);
           TreeNode *rroot = constructFromPrePost(rpre,rpos);
           root->left = lroot;
           root->right = rroot;
           return root;
         }
    }
};
