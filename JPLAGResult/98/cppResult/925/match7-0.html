/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        return helper(pre, post, 0, pre.size()-1, 0, post.size()-1);
    }
    TreeNode* helper(vector<int>& pre, vector<int>& post, int preL, int preR, int postL, int postR) {
        if (preL > preR || postL > postR)   return nullptr;
        //cout << preL <<"," << preR << "," << postL <<"," << postR << endl;
        TreeNode* cur = new TreeNode(pre[preL]);
        if (preL == preR)   return cur;
        int leftLen=0;
        while (post[postL+leftLen] != pre[preL+1])  ++leftLen;
        cur->left = helper(pre, post, preL+1, preL+1+leftLen, postL, postL+leftLen);
        cur->right = helper(pre, post, preL+2+leftLen, preR, postL+leftLen+1, postR-1);
        return cur;
    }
};
