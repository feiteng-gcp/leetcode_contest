/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        if (pre.size() == 0) {
            return NULL;
        }
        if (pre.size() == 1) {
            return new TreeNode(pre[0]);
        }
        else {
            TreeNode* root = new TreeNode(pre[0]);
            int n_left = 0;
            while(post[n_left] != pre[1]) {
                n_left++;
            }

            ++n_left;
            vector<int> pre_left = vector<int>(pre.begin() + 1, pre.begin() + 1 + n_left);
            vector<int> post_left = vector<int>(post.begin(), post.begin() + n_left);
            vector<int> pre_right = vector<int>(pre.begin() + 1 + n_left, pre.end());
            vector<int> post_right = vector<int>(post.begin() + n_left, post.end() - 1);
            root->left = constructFromPrePost(pre_left, post_left);
            root->right = constructFromPrePost(pre_right, post_right);
            return root;
        }
    }
};
