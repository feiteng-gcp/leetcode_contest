/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* dfsHelper(vector<int>& pre, int pre_begin, int pre_end, vector<int>& post, int post_begin, int post_end) {
        if (pre_begin > pre_end || pre_begin >= pre.size()) return nullptr;
        TreeNode *root = new TreeNode(pre[pre_begin]);
        if (pre_begin == pre_end) return root;
        int pos = 1;
        set<int> tmp_pre, tmp_post;
        for (; pos <= pre_end - pre_begin; pos++) {
            tmp_pre.insert(pre[pre_begin + pos]);
            tmp_post.insert(post[post_begin + pos - 1]);
            if (tmp_pre == tmp_post) break;
        }
        root->left = dfsHelper(pre, pre_begin + 1, pre_begin + pos, post, post_begin, post_begin + pos - 1);
        root->right = dfsHelper(pre, pre_begin + pos + 1, pre_end, post, post_begin + pos, post_end - 1);
        return root;
    }
    
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        return dfsHelper(pre, 0, pre.size() - 1, post, 0, post.size() - 1);
    }
};
