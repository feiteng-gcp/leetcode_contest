/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) 
    {
        int len=pre.size();
        return go(pre,0,len-1,post,0,len-1);
    }
    
    TreeNode* go(vector<int>& pre, int a,int b, vector<int>& post, int c ,int d)
    {
        if(a>b&&c>d) return NULL;
        TreeNode* ret=new TreeNode(pre[a]);
        if(a==b&&c==d) return ret;
        int f=post[d-1];
        int aa;
        for(aa=a;aa<=b;aa++)
        {
            if(pre[aa]==f) break;
        }
        int ll=aa-a-1;
        ret->left=go(pre,a+1,aa-1,post,c,c+ll-1);
        ret->right=go(pre,aa,b,post,c+ll,d-1);
        return ret;
    }
};
