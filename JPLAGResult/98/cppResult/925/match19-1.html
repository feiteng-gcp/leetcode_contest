/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* helper(vector<int>& pre, int l1, int r1, vector<int>& post, int l2, int r2) {
        if(l1 > r1 || l2 > r2) return NULL;
        TreeNode *result = new TreeNode(pre[l1]);
        if(l1 == r1) return result;
        if(pre[l1+1] == post[r2-1]) {
            result->left = helper(pre, l1+1, r1, post, l2, r2-1);
            result->right = NULL;
            return result;
        }
        int sppre = -100, sppost = -100;
        for(int i = l2 ; i <= r2 ; i++) {
            if(post[i] == pre[l1+1]) {
                sppost = i;
                break;
            }
        }
        for(int i = l1 ; i <= r1 ; i++) {
            if(pre[i] == post[r2-1]) {
                sppre = i;
                break;
            }
        }
        result -> left = helper(pre, l1+1, sppre-1, post, l2, sppost);
        result -> right = helper(pre, sppre, r1, post, sppost+1, r2-1);
        return result;
    }
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        if(pre.size() == 0) return NULL;
        return helper(pre, 0, pre.size()-1, post, 0, post.size() - 1);
    }
};
