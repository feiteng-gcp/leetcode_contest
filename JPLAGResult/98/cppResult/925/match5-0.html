/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        return construct(pre, post, 0, pre.size()-1, 0, post.size()-1);
    }
    TreeNode * construct(vector<int> & pre, vector<int> & post, int prei, int prej, int posti, int postj) {
        if (prei == prej) {
            return new TreeNode(pre[prei]);
        }
        else if (prei < prej) {
            TreeNode * root = new TreeNode(pre[prei]);
            prei ++;
            postj --;
            int posttmp = posti;
            while (post[posttmp] != pre[prei]) posttmp ++;
            int pretmp = prei + posttmp - posti;
            root->left = construct(pre, post, prei, pretmp, posti, posttmp);
            root->right = construct(pre, post, pretmp+1, prej, posttmp+1, postj);
            return root; 
        }
        else return nullptr;
    }
};
