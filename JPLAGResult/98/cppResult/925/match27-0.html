/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        if(pre.size() != post.size())
            return NULL;
        int n = pre.size();
        TreeNode* ans = constructImpl(pre, 0, n-1, post, 0, n-1);
        return ans;
    }
    
private:
    TreeNode* constructImpl(vector<int>& pre, int prel, int prer, vector<int>& post, int postl, int postr) {
        if(prel > prer || postl > postr)
            return NULL;
        if(prel == prer)
        {
            return new TreeNode(pre[prel]);
        }

        int rootVal = pre[prel];
        TreeNode* root = new TreeNode(rootVal);
        int leftVal = pre[prel+1];
        int mid = postl;
        for(int i = mid; i <= postr; i++)
        {
            if(post[i] == leftVal)
            {
                mid = i;
                break;
            }
        }
        int leftCnt = mid - postl + 1;
        int rightCnt = postr - 1 - mid;
        root->left = constructImpl(pre, prel+1, prel+leftCnt, post, postl, postl+leftCnt-1);
        root->right = constructImpl(pre, prel+leftCnt+1, prer, post, mid+1, postr-1);
        return root;
    }
};
