/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
int n;

struct TreeNode* solution (vector<int> pre, vector<int> post, int* index,
                                int l, int h){
    if (*index >= n || l > h)return NULL;
    TreeNode* root = new TreeNode( pre[*index] );
    ++*index;
 
    if(l == h)return root;
 
    int id=-1;
    for(id = l; id <= h; ++id)
        if(pre[*index] == post[id])break;
 
    if (id <= h){
        root->left = solution(pre, post, index, l, id);
        root->right = solution(pre, post, index, id + 1, h-1);
    }
 
    return root;
}
 
class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
        n=pre.size();
        int index = 0;
        return solution(pre,post,&index,0,n-1);
    }
};
