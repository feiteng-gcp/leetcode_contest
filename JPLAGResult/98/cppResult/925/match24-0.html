/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* constructFromPrePost(vector<int>& pre, vector<int>& post) {
	if (pre.empty()) return nullptr;
	const size_t len = pre.size();
	if (len == 1) return new TreeNode(pre[0]);

	TreeNode* root = new TreeNode(pre[0]);
	
	if (pre[1] == post[len - 2]) { // root has only one child, doesn't matter left or right
		vector<int> lNewPre = vector<int>(pre.begin() + 1, pre.end());
		vector<int> lNewPost = vector<int>(post.begin(), post.end() - 1);
		root->left = constructFromPrePost(lNewPre, lNewPost);
	}
	else {
		int lVal = pre[1];
		int rVal = post[len - 2];
		vector<int> lNewPre = vector<int>(pre.begin() + 1, find(pre.begin(), pre.end(), rVal));
		vector<int> lNewPost = vector<int>(post.begin(), find(post.begin(), post.end(), lVal) + 1);
		vector<int> rNewPre = vector<int>(find(pre.begin(), pre.end(), rVal), pre.end());
		vector<int> rNewPost = vector<int>(find(post.begin(), post.end(), lVal) + 1, post.end() - 1);
		root->left = constructFromPrePost(
			lNewPre, lNewPost
		);
		root->right = constructFromPrePost(
			rNewPre, rNewPost
		);
	}

	return root;
    }
};
