class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        sort(A.begin(), A.end());
        sort(B.begin(), B.end());
        int sumA=0, sumB=0;
        for(int i=0;i<A.size();i++){
            sumA += A[i];
        }
        for(int i=0;i<B.size();i++){
            sumB += B[i];
        }
        int i=0, j=0;
        vector<int> ans;
        while(true){
            if(2*(A[i]-B[j])==sumA-sumB){
                ans.push_back(A[i]);
                ans.push_back(B[j]);
                return ans;
            }
            else if(2*(A[i]-B[j])<sumA-sumB){
                i++;
            }
            else{
                j++;
            }
        }
    }
};
