class Solution {
public:
  vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
    sort(B.begin(), B.end());
    int sa = 0, sb = 0;
    for (int a: A) sa += a;
    for (int b: B) sb += b;
    int d = (sa - sb) / 2;
    for (int a: A) {
      if (binary_search(B.begin(), B.end(), a-d)) {
        return {a, a-d};
      }
    }
    abort();
  }
};
