class Solution {
    unordered_set<int> s;
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {

        int suma = 0, sumb = 0;
        s.clear();
        for (int i = 0; i < A.size(); ++i) suma += A[i];
        for (int i = 0; i < B.size(); ++i) sumb += B[i];
        int sum = (suma + sumb) >> 1;
        for (int i = 0; i < B.size(); ++i) s.insert(B[i]);
        for (int i = 0; i < A.size(); ++i) {
            if (s.find(sum - (suma - A[i])) != s.end()) {
                vector<int> res;
                res.push_back(A[i]);
                res.push_back(sum - (suma - A[i]));
                return res;
            }
        }
        return vector<int> ();
    }
};
