class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        
        unordered_set<int> ms;
        int sum_A =0 , sum_B = 0;
        

        for(int i=0; i<B.size(); i++) {
            sum_B += B[i];
            ms.insert(B[i]);
        }
        
        for(int i=0; i<A.size(); i++)
            sum_A += A[i];
        
        int dif = sum_B - sum_A;
        for(int i=0; i<A.size(); i++) {
            
            int val = dif + 2*A[i];
            if(val%2==0) {
                val /= 2;
                if(ms.find(val)!=ms.end()) {
                    vector<int> ans = {A[i], val};
                    return ans;
                }
            }
        }
        
        
    }
};
