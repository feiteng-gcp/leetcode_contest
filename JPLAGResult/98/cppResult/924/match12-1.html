class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int sumA = 0, sumB = 0;
        unordered_set<int> numsA, numsB;
        for (auto i:A){
            sumA += i;
            numsA.insert(i);
        }
        for (auto i:B){

            sumB += i;
            numsB.insert(i);
        }
        int diff = sumA - sumB;
        for (auto i:numsA){
            if (numsB.count(i-diff/2))   return {i, i-diff/2};
        }
        return {0,0};
    }
};
