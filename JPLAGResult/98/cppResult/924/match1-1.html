class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        unordered_set<int> s(B.begin(), B.end());
        vector<int> ans(2);
        int suma = 0;
        for (auto &num : A) {
            suma += num;
        }
        int sumb = 0;
        for (auto &num : B) {
            sumb += num;
        }
        int sub = (suma-sumb) / 2;
        for (auto &num : A) {
            if (s.find(num-sub) != s.end()) {
                ans[0] = num;
                ans[1] = num-sub;
                break;
            }
        }
        return ans;
    }
};
