class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int m = A.size();
        int n = B.size();
        int sum = 0;
        int suma = 0;
        int sumb = 0;
        for (auto& a : A) {
            suma+=a;
        }
        for (auto& b : B) {
            sumb+=b;
        }
        sum = (suma+sumb)/2;
        if (sum > suma) {
            return find(B, A, sum-suma, true);
        } else {
            return find(A, B, sum-sumb, false);
        }
    }
    
    vector<int> find(vector<int>& A, vector<int>& B, int k, bool ok) {
        vector<int> res;
        unordered_set<int> seta;
        for (auto& e:A) {
            seta.insert(e);
        }
        for(auto& e:B) {
            if (seta.find(e+k) != seta.end()) {
                if (ok) {
                    res.push_back(e);
                    res.push_back(e+k);
                } else {
                    res.push_back(e+k);
                    res.push_back(e);
                }
                return res;
            }
        }
    }
};
