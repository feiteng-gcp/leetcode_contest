class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int sa = 0;
        for (auto x: A) {
            sa += x;
        }
        int sb = 0;
        for (auto x: B) {
            sb += x;
        }
        int diff = sa - sb;
        unordered_set<int> bs;
        for (auto x: B) {
            bs.insert(x);
        }
        vector<int> res(2);
        for (auto x: A) {
            if (bs.find(x - diff / 2) != bs.end()) {
                res[0] = x;
                res[1] = x - diff / 2;
                break;
            }
        }
        return res;
    }
};
