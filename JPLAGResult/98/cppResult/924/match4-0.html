class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int sum1=0, sum2=0;
        for (int x:A) sum1+=x;
        for (int x:B) sum2+=x;
        
        vector<int> res(2,0);
        for (int i=0; i<A.size(); ++i){
            for (int j=0; j<B.size(); ++j) {
                if (sum1-A[i]+B[j] == sum2-B[j]+A[i]) {
                    res[0]=A[i];
                    res[1]=B[j];
                    return res;
                }
            }
        }
    }
};
