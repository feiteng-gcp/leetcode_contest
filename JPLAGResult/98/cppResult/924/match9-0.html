class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        // sort(A.begin(), A.end());
        // sort(B.begin(), B.end());
        // int pointer_A = 0, pointer_B = 0;
        int sum_A = 0, sum_B = 0;
        for(int i = 0; i < A.size(); i++)
        {
            sum_A += A[i];
        }
        for(int i = 0; i < B.size(); i++)
        {
            sum_B += B[i];
        }
        int result_candy = (sum_A + sum_B) / 2;
        int result_A = sum_A;
        int result_B = sum_B;
        for(int i = 0; i < A.size();i++)
        {
            for(int j = 0; j < B.size(); j++)
            {
                result_A = sum_A - A[i] + B[j];
                if (result_A == result_candy)
                    return {A[i], B[j]};
            }
        }
        return {0,0};
    }
};
