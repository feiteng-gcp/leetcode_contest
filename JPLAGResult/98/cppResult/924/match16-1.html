class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        unordered_set<int> mp;
        vector<int> res;
        int suma = 0, sumb = 0;
        for (int i = 0; i < A.size(); i++) mp.insert(A[i]), suma += A[i];

        for (int i = 0; i < B.size(); i++) sumb += B[i];
        int k = (suma - sumb) / 2;
        for (int i = 0; i < B.size(); i++) {
            int tmp = B[i] + k;
            if (mp.count(tmp) > 0) {
                res.push_back(tmp);
                res.push_back(B[i]);
                return res;
            }
        }
    }
};
