class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int sa = 0;
        int sb = 0;
        for(int i : A) sa += i;
        for(int i : B) sb += i;
        unordered_set<int> mb(B.begin(), B.end());
        vector<int> res;
        for(int i : A) {
 
            int j = sb - sa + 2 * i;
            if(j % 2 != 0) continue;
            j /= 2;
            if(mb.find(j) != mb.end()) {
                res = {i, j};
                return res;
            }
        }
        return res;
    }
};
