typedef long long ll;
class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) 
    {

        unordered_map<int,int> ua;
        unordered_map<int,int> ub;
        ll sa=0;
        ll sb=0;
        
        for(int n : A)
        {
            ua[n]++;
            sa+=n;
        }
        for(int n : B)
        {
            ub[n]++;
            sb+=n;
        }
        
        ll mid=(sa+sb)/2;
        int gap=mid-sa;
        
        vector<int> ret;
        for(int n : A)
        {
            if(ub.find(n+gap)!=ub.end())
            {
                ret.push_back(n);
                ret.push_back(n+gap);
                return ret;
            }
        }
        
        return ret;
    }
};
