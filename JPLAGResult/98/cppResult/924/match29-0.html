class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        vector<int> res;
        int s1 = accumulate(A.begin(), A.end(), 0);
        int s2 = accumulate(B.begin(), B.end(), 0);
        int s = (s1 + s2) / 2;
        int need = s - s1;
        unordered_set<int> m1, m2;
        for (auto& a: A)
            m1.insert(a);
        for(auto& b: B)
            m2.insert(b);
        for (int i = 0; i < A.size(); ++i) {
            if (m2.find(A[i] + need) != m2.end()) {
                res = {A[i], A[i] + need};
                return res;
            }
        }
    }
};
