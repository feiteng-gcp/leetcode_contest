class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int sumA = 0, sumB = 0;
        unordered_set<int> sB;
        for (auto a: A) sumA += a;
        for (auto b: B) {
            sumB += b;
            sB.insert(b);
        }
        

        int sum = (sumA + sumB)/2;
        int diff = sum - sumA;
        
        for (auto a: A) {
            int change = a + diff;
            if (sB.find(change) != sB.end()) {
                return {a, change};
            }
        }
        
        return vector<int>();
    }
};
