class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        map<int,int> am, bm;
        int sa = 0, sb = 0;
        for(int& a : A) {
            am[a]++;
            sa += a;
        }
        for(int& b : B) {
            bm[b]++;
            sb += b;
        }
        int diff = (sb - sa) / 2;
        for(auto it : am) {
           if (bm.find(it.first + diff) != bm.end())
               return vector<int>{it.first, it.first + diff};
        }
        
    }
};
