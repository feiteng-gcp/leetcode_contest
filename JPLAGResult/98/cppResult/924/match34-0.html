class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int sumA = 0;
        int sumB = 0;
        set<int> alice;
        set<int> bob;
        for (int a : A) {
            sumA += a;
            alice.insert(a);
        }
        for (int b : B) {
            sumB += b;
            bob.insert(b);
        }
        vector<int> ans;
        int diff = sumA - sumB;
        for (int a : A) {
            int b = a - diff / 2;
            if (bob.find(b) != bob.end()) {
                ans = {a, b};
                break;
            }
        }
        return ans;
    }
};
