class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int sumA = 0, sumB = 0;
        for (int a : A) sumA += a;
        for (int b : B) sumB += b;
        int diff = (sumB - sumA)/2;
        
        unordered_set<int> sb;
        for (int b : B) sb.insert(b);
        for (int a : A) if (sb.find(a+diff) != sb.end()) {
            return vector<int>{a, a+diff};
        }
    }
};
