class Solution {
public:
    vector<int> fairCandySwap(vector<int>& a, vector<int>& b) {
        int sa = 0, sb = 0;
        for (int x : a) sa += x;
        for (int x : b) sb += x;
        int d = (sa - sb) / 2;
        set<int> tb(b.begin(), b.end());

        for (int x : a) {
            if (tb.find(x - d) != tb.end()) {
                return {x, x - d};
            }
        }
        return {};
    }
};
