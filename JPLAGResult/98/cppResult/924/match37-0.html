class Solution {
public:
vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
	int aSum = 0, bSum = 0;
	for (int num : A) aSum += num;
	for (int num : B) bSum += num;
	int targetDiff = (bSum - aSum) / 2;
	vector<int> ret(2,0);
	if (A.size() == 1 && B.size() == 1) {
		return vector<int>{A[0], B[0]};
	}
	else if (A.size() == 1) {
		ret[0] = A[0];
		for (int num : B) {
			if (num == A[0] + targetDiff) {
				ret[1] = num;
				return ret;
			}
		}
	}
	else if (B.size() == 1) {
		ret[1] = B[0];
		for (int num : A) {
			if (B[0] == num + targetDiff) {
				ret[0] = num;
				return ret;
			}
		}
	}
	else {
		for (int aNum : A) {
			for (int bNum : B) {
				if (aNum + targetDiff == bNum) {
					ret = { aNum, bNum };
					return ret;
				}
			}
		}
	}
}
};
