class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        sort(A.begin(), A.end());
        sort(B.begin(), B.end());
        int sum_A = accumulate(A.cbegin(), A.cend(), 0);
        int sum_B = accumulate(B.cbegin(), B.cend(), 0);
        int diff = (sum_B - sum_A) / 2;
        int pos_A(0);
        int pos_B(0);
        while (true)
        {
            if (A[pos_A] + diff < B[pos_B])
                pos_A++;
            else if (A[pos_A] + diff > B[pos_B])
                pos_B++;
            else
                return {A[pos_A], B[pos_B]};
        }
    }
};
