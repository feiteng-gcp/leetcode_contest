class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        unordered_map<int, int> dictA;
        int sumA = 0;
        for (auto x : A) {
            sumA += x;
            dictA[x] ++;
        }
        int sumB = accumulate(B.begin(), B.end(), 0);
        for (auto x : B) {
            if (dictA.count((sumA - sumB)/2 + x)) return {(sumA - sumB)/2 + x, x};
        }
        return {0, 0};
    }
};
