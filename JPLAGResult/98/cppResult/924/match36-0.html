class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int sum_a = 0;
        int sum_b = 0;
        unordered_set<int> set_b;
        for(int &x : A) {
            sum_a +=x;
        }
        for(int &x : B) {
            sum_b += x;
            set_b.insert(x);
        }
        int target = (sum_a + sum_b) / 2;
        int delta = target - sum_a;
        vector<int> ret(2);
        for(int &x : A) {
            if(set_b.find(x + delta) != set_b.end()) {
                ret[0] = x;
                ret[1] = x + delta;
            }
        }
        return ret;
    }
};
