class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        unordered_set<int> sA, sB;

        int sumA = 0, sumB = 0;
        for(int i = 0; i < A.size(); i++){
            sA.insert(A[i]);
            sumA += A[i];
        }
        for(int i = 0; i < B.size(); i++){
            sB.insert(B[i]);
            sumB += B[i];
        }
        for(int i = 0; i < A.size(); i++){
            if(sB.count(A[i] + (sumB - sumA) / 2)){
                return {A[i], A[i] + (sumB - sumA) / 2};
            }
        }
        return {};
    }
};
