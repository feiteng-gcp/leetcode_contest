class Solution {
public:
vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
	vector<int> result;
	int countA = accumulate(A.begin(), A.end(), 0);
	int countB = accumulate(B.begin(), B.end(), 0);
	int count = (countA + countB) / 2;
	for (int i = 0; i < A.size(); ++i)
	{
		int need = count - countA + A[i];
		for (int j = 0; j < B.size(); ++j)
		{
			if (B[j] == need)
			{
				result.push_back(A[i]);
				result.push_back(B[j]);
				return result;
			}
		}
	}
	return result;
}
};
