class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int s1=0, s2=0;
        unordered_set<int> sA;
        for(auto k: A) {
            s1 += k;
            sA.insert(k);
        }
        for(auto k: B) s2 += k;
        assert((s1-s2)%2 == 0);
        for(auto k: B){
            if(sA.count(k+(s1-s2)/2)) return vector<int>{k+(s1-s2)/2, k};
        }
        return {-1, -1};
    }
};
