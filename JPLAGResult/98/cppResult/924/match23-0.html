class Solution {
public:
    vector<int> fairCandySwap(vector<int>& A, vector<int>& B) {
        int n=A.size(),m=B.size();
	int suma=0,sumb=0;
	for(int i=0;i<n;i++)
	{
		suma+=A[i];
	}
	for(int i=0;i<m;i++)
	{
		sumb+=B[i];
	}
	sort(B.begin(),B.end());
	for(int i=0;i<n;i++)
	{
		int l=0,r=m-1;
		while(l<=r)
		{
			int mi=(l+r)/2;
			if(B[mi]==(sumb-suma+2*A[i])/2)
			{
				vector<int> ans;
				ans.push_back(A[i]);
				ans.push_back(B[mi]);
				return ans;			}
			else if(B[mi]>(sumb-suma+2*A[i])/2)r=mi-1;
			else l=mi+1;
		}
	}
    }
};
