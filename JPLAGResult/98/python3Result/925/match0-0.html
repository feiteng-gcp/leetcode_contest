# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def constructFromPrePost(self, pre, post):
        """
        :type pre: List[int]
        :type post: List[int]
        :rtype: TreeNode
        """
        self.pre_index = 0
        def construct_tree(l, h):
            if l>h or self.pre_index == len(pre):
                return None
            node = TreeNode(pre[self.pre_index])
            self.pre_index += 1
            if self.pre_index >= len(pre) or l == h:
                return node
            i = l 
            while i <= h:
                if post[i] == pre[self.pre_index]:
                    break
                i+=1
            if i <= h:
                node.left = construct_tree(l, i)
                node.right = construct_tree(i+1, h-1)
            return node
        return construct_tree(0, len(post)-1)
