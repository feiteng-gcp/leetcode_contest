# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def constructFromPrePost(self, pre, post):
        """
        :type pre: List[int]
        :type post: List[int]
        :rtype: TreeNode
        """
        if len(pre) == 0:
            return None
        root = TreeNode(pre[0])
        if len(pre) == 1:
            return root
        elif len(pre) == 2:
            root.left = TreeNode(pre[1])
            return root 
        else:
            n_left = post.index(pre[1])
            lefts_pre 
= pre[1:n_left+2]
            lefts_post = post[:n_left+1]
            rights_pre = pre[n_left+2:]
            rights_post = post[n_left+1:]
            
            root.left = self.constructFromPrePost(lefts_pre, lefts_post)
            root.right = self.constructFromPrePost(rights_pre, rights_post)
            return root
