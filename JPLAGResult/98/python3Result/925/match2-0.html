# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def constructFromPrePost(self, pre, post):
        """
        :type pre: List[int]
        :type post: List[int]
        :rtype: TreeNode
        """
        if len(pre) == 0:
            return None
        if len(pre) == 1:
            return TreeNode(pre[0])
        i = post.index(pre[1])
        left = self.constructFromPrePost(pre[1:i+2], post[:i+1])
        right = self.constructFromPrePost(pre[i+2:], post[i+1:-1])
        t = TreeNode(pre[0])
        t.left = left
        t.right = right
        return t
