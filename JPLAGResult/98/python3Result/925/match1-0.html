# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def constructFromPrePost(self, pre, post):
        """
        :type pre: List[int]
        :type post: List[int]
        :rtype: TreeNode
        """
        if not pre:
            return None
           
        root = TreeNode(pre[0])
        
        if len(pre) == 1:
            return root
        
        if len(pre) == 2:
            root.right = None
            root.left= TreeNode(pre[1])
            return root
        
        for i in range(1, len(pre)):
            if set(pre[1:i+1]) == set(post[:i]):
                break
                
        pre_leftSubTree = pre[1:i+1]
        pre_rightSubTree = pre[i+1:]
        
        post_leftSubTree = post[:i]
        post_rightSubTree= post[i:-1]
        # print(pre_leftSubTree, post_leftSubTree)
        # print(pre_rightSubTree, post_rightSubTree)
        root.left = self.constructFromPrePost(pre_leftSubTree, post_leftSubTree)
        root.right = self.constructFromPrePost(pre_rightSubTree, post_rightSubTree)
        
        return root
