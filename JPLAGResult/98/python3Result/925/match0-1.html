# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def constructFromPrePost(self, pre, post):
        """
        :type pre: List[int]
        :type post: List[int]
        :rtype: TreeNode
        """
        self.index = 0
        
        def helper(pre, post, low, high):
            if self.index >= len(pre) or low > high:
                return
            
            n = TreeNode(pre[self.index])
            self.index += 1
            
            if low == high or self.index >= len(pre):
                return n
            
            i = low
            
            while i <= high:
                if post[i] == pre[self.index]:
                    break
                i += 1
            
            if i <= high:
                n.left = helper(pre, post, low, i)
                n.right = helper(pre, post, i + 1, high - 1)
            
            return n

        return helper(pre, post, 0, len(pre) - 1)
