# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def constructFromPrePost(self, pre, post):
        """
        :type pre: List[int]
        :type post: List[int]
        :rtype: TreeNode
        """
        if len(pre) == 0:
            return None
        if len(pre) == 1:
            return TreeNode(pre[0])
        root = TreeNode(pre[0])
        for i in range(1, len(pre)):
            if set(pre[1:i+1]) == set(post[0:i]):
                root.left = self.constructFromPrePost(pre[1:i+1], post[0:i])
                root.right = self.constructFromPrePost(pre[i+1:], post[i:-1])
                break
        return root
