# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def constructFromPrePost(self, pre, post):
        """
        :type pre: List[int]
        :type post: List[int]
        :rtype: TreeNode
        """
        # print(pre,post)
        # if not pre or not post: return None
        if len(pre)==1: return TreeNode(pre[0])
        m=pre[0]
        curr=TreeNode(m)
        Lm=pre[1]
        Rm=post[-2]
        i=post.index(Lm)
        j=pre.index(Rm)
        if pre
[j]==Lm:
            curr.left=self.constructFromPrePost(pre[j:],post[:-1])
        else:
            curr.left=self.constructFromPrePost(pre[1:j],post[:i+1])
            curr.right=self.constructFromPrePost(pre[j:],post[i+1:-1])
        return curr
