# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def constructFromPrePost(self, pre, post):
        """
        :type pre: List[int]
        :type post: List[int]
        :rtype: TreeNode
        """
        if post:
            #print(pre)
            #print(post)
            val = post.pop(-1)
            pre.pop(0)
            root = TreeNode(val)
            if len(post):
                preset = set()
                postset = set()
                for i in range(len(post)):
                    preset.add(pre[i])
                    postset.add(post[i])
                    #print(preset)
                    #print(postset)
                    #print("---")
                    if preset == postset:
                        #print("Break",i)
                        break
            
                root
.left = self.constructFromPrePost(pre[:i+1], post[:i+1])
                root.right = self.constructFromPrePost(pre[i+1:], post[i+1:])
        
            return root
