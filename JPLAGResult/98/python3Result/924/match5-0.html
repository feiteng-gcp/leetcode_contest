class Solution:
    def fairCandySwap(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        A.sort()
        B.sort()
        target = (sum(B) - sum(A)) / 2
        idxa = idxb = 0
        
        while True:
            if B[idxb] - A[idxa] == target:
                return [A[idxa], B[idxb]]
            elif B[idxb] - A[idxa] > target:
                idxa += 1
            else:
                idxb += 1
