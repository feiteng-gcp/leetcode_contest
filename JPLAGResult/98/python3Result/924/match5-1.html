class Solution:
    def fairCandySwap(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        dif = sum(A)-sum(B)
        dif //= 2
        A.sort()
        B.sort
()
        i = 0
        j = 0
        while True:
            if A[i]-B[j]==dif:
                return [A[i],B[j]]
            elif A[i]-B[j] > dif:
                j += 1
            else:
                i += 1
