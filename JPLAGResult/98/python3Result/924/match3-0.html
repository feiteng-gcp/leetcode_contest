class Solution:
    def fairCandySwap(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        a = sum(A)
        b = sum(B)
        diff = (b-a)//2
        sa = set(A)
        for v in B:
            if v-diff in sa:
                return [v-diff, v]
