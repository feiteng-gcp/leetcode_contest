class Solution:
    def fairCandySwap(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        d=(sum(A)-sum(B))//2
        D1=set(A)
        D2=set(B)
        for x in D1:
            if x-d in D2:
                return [x,x-d]
