class Solution:
    def fairCandySwap(self, A, B):
        """
        :type A: List[int]
        :type B: List[int]
        :rtype: List[int]
        """
        
        sA = sum(A)
        sB = sum
(B)
        dif = sA - sB
        
        setB = set(B)
        for a in A:
            if(a - dif // 2 in setB):
                return [a, a-dif//2]
