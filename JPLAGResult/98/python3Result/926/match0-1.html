class Solution:
    def findAndReplacePattern(self, words, pat):
        """
        :type words: List[str]
        :type pattern: str
        :rtype: List[str]
        """
        ans=[]
        for wd in words:
            if len(wd)!=len(pat): continue
            pat2w={}
            w2pat
={}
            invalid=False
            for i in range(len(pat)):
                if pat[i] not in pat2w and wd[i] not in w2pat:
                    pat2w[pat[i]]=wd[i]
                    w2pat[wd[i]]=pat[i]
                else:
                    if (pat[i] in pat2w and pat2w[pat[i]]!=wd[i]) or (wd[i] in w2pat and w2pat[wd[i]]!=pat[i]):
                        invalid=True
                        break
            if not invalid: ans.append(wd)
        return ans
