/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
    	return makeTree(pre, post);
    }

	private TreeNode makeTree(int[] pre, int[] post) {
		if (pre.length == 0)
			return null;
		
		if (pre.length == 1)
			return new TreeNode(pre[0]);
		
		TreeNode newNode = new TreeNode(pre[0]);
		int preL, preR;
		int postL, postR = 0;
		int pMidPost = -1;
		
		int count;
		for (int i = 0 ; i < post.length ; i++)
		{
			if (pre[1] == post[i])
			{
				pMidPost = i;
				break;
			}
		}
		postL = 0;
		postR = pMidPost;	//must have
		count = postR - postL + 1;
		preL = 1;
		preR = preL + count - 1;
		int newPre[] = Arrays.copyOfRange(pre, preL, preR + 1);
		int newPost[] = Arrays.copyOfRange(post, postL, postR + 1);
		TreeNode leftNode = makeTree(newPre, newPost);

		preL = preR + 1;
		preR = pre.length - 1;
		postL = pMidPost + 1;
		postR = post.length - 1;
		newPre = Arrays.copyOfRange(pre, preL, preR + 1);
		newPost = Arrays.copyOfR
ange(post, postL, postR + 1);
		
		TreeNode rightNode = makeTree(newPre, newPost);
		newNode.left = leftNode;
		newNode.right = rightNode;

		return newNode;
	}
}
