/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        
        return constructFromPrePost(pre, post, 0, pre.length-1, 0, post.length-1); //includes start,  end 
        
    }
    
    public TreeNode constructFromPrePost(int[] pre, int[] post, int preS, int preE, int postS, int postE) {
        if(preS>preE) return null;
        
        TreeNode head = new TreeNode(pre[preS]);
        if(preS==preE) return head;
        
        if(preS==preE-1) {
            TreeNode left = new TreeNode(pre[preE]);
            
            head.left=left;
            return head;
        }
        
        int leftHead = -1;
        for(int i=postS; i<=postE; i++) {
            if(post[i] == pre[preS+1]) {
                leftHead = i;
            }
        }
        
        int leftSize = leftHead+1 -postS;
        
        TreeNode left = constructFromPrePost(pre, post, preS + 1 , leftSize+preS , postS, postS+leftSize-1);
        TreeNode right = constructFromPrePost(pre, post, leftSize+preS +1 , preE , postS+leftSize , postE-1);
        
        head.left=left;
        head.right=right;
        
        return head;
    }
}
