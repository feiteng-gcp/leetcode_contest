/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        return sol(pre, post, 0, 0, pre.length);
    }
    
    public TreeNode sol(int[] pre, int[] post, int preS, int postS, int len) 
{
        if(len == 0) return null;
        TreeNode root = new TreeNode(pre[preS]);
        if(len == 1) return root;
        int right = post[postS + len - 2];
        if(len == 2) {
            root.right = new TreeNode(right);
            return root;
        }
        
        int lenLeft = 0;
        while(pre[preS + lenLeft] != right) lenLeft++;
        root.left = sol(pre, post, preS + 1, postS, lenLeft - 1);
        int lenRight = len - lenLeft;
        root.right = sol(pre, post, preS + lenLeft, postS + lenLeft - 1, lenRight);
        return root;
    }
}
