/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        return c(pre,post,0,pre.length-1,0,post.length-1);
    }
    
    TreeNode c(int[] pre, int[] post,int i1,int j1,int i2,int j2){
        //if(pre[i1]==post[j2]){
        if(i1>j1||i2>j2){
            return null;
        }
        
            TreeNode  r = new TreeNode(pre[i1]);
            for(int i=i2;i<=j2-1;i++){
                if(post[i]==pre[i1+1]){
                   // d - (i1+1) == i - i2
                    
                    r.left = c(pre,post,i1+1,i-i2+i1+1,i2,i);
                    r.right = c(pre,post,i-i2+i1+2,j1,i+1,j2-1);
                    return r;
                }
            }
            
       // }
        return r;
    }
    
    
}
