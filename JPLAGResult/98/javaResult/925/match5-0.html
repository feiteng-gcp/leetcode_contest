/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        int len = pre.length;
        return construct(pre, post, 0, len-1, 0, len-1);
    }
    
    private TreeNode construct(int[] pre, int[] post, int s1, int e1, int s2, int e2) {
        if (s1 > e1) return null;
        TreeNode root = new TreeNode(pre[s1]);
        if (s1 + 1 > e1) return root;
        int i = s2;
        for (; i < e2; i++) {
            if (post[i] == pre[s1+1]) break;
        }
        int len = i - s2 + 1;
        TreeNode left = construct(pre, post, s1+1, s1+len, s2, i);
        TreeNode right = construct(pre, post, s1+len+1, e1, i+1, e2-1);
        root.left = left;
        root.right = right;
        return root;
    }
}
