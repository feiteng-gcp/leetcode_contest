/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        return helper(pre, 0, pre.length - 1, post, 0, post.length - 1);
    }
    
    public TreeNode helper(int[] pre, int l1, int r1, int[] post, int l2, int r2) {
        
        
        if (l1 > r1) return null;
        
        TreeNode root = new TreeNode(pre[l1]);
        if (l1 == r1) return root;
        
        int idx1 = l1 + 1;
        int idx2 = l2;
        Set<Integer> set = new HashSet<>();
        while (idx1 <= r1) {
            set.add(pre[idx1]);
            set.add(post[idx2]);
            if (set.size() == idx2 - l2 + 1) break;
            idx1++;
            idx2++;
        }
        
        
TreeNode left = helper(pre, l1 + 1, idx1, post, l2, idx2);
        TreeNode right = helper(pre, idx1 + 1, r1, post, idx2 + 1, r2 - 1);
        
        root.left = left;
        root.right = right;
        
        return root;
    }
    
    
}
