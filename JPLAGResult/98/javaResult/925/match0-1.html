/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
    	int n = pre.length;
    	if (n == 0) return null;
    	TreeNode res = new TreeNode(pre[0]);
    	if (n == 1) return res;
    	for (int i = 0; i + 1 < post.length; ++i) {
    		if (post[i] == pre[1]) {
    			int[] lpre = new 
int[i + 1];
    			int[] lpost = new int[i + 1];
    			int[] rpre = new int[n - i - 2];
    			int[] rpost = new int[n - i - 2];
    			for (int k = 0; k < i + 1; ++k) {
    				lpre[k] = pre[1 + k];
    				lpost[k] = post[k];
    			}
    			for (int k = 0; k < n - i - 2; ++k) {
    				rpre[k] = pre[i + 2 + k];
    				rpost[k] = post[i + 1 + k];
    			}
    			res.left = constructFromPrePost(lpre, lpost);
    			res.right = constructFromPrePost(rpre, rpost);
                return res;
    		}
    	}
    	return null;
    }
}
