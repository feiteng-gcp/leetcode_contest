/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    int i = 0;
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        i=0;
        return help(pre,post,0,pre.length-1);
    }
    public TreeNode help(int[] pre,int[] post,int s,int e){
        if(i>=pre.length || s>e){
            return null;
        }
        TreeNode root = new TreeNode(pre[i]);
        i++;
        if(s==e || i>=pre.length){
            return root;
        }
        int j=0;
        for(j=s;j<=e;j++){
            if(post[j]==pre[i]){
                break;
            
}
        }
        if(j<=e){
            root.left = help(pre,post,s,j);
            root.right = help(pre,post,j+1,e-1);
        }
        return root;
    }
}
