/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        
        int len = pre.length;
        if(len<1) return null;
        TreeNode root = new TreeNode(pre[0]);
        if(len==1) return root;
        
        int k=0;
        for(;k<len-1;k++){
            if(pre[1]==post[k]) break;
        }
        
        int[] lpre = new int[k+1];
        int[] lpost = new int[k+1];
        int[] rpre = new int[len-k-2];
        int[] rpost = new int[len-k-2];
        for(int i=0;i<=k;i++){
            lpre[i] = pre[i+1];
            lpost[i] = post[i]; 
        }
        
        for(int j=k+1;j<len-1;j++){
            rpost[j-k-1] = post[j];
            rpre[j-k-1] = pre[j+1];
        }
        
        root.left = constructFromPrePost(lpre,lpost);
        root.right = constructFromPrePost(rpre,rpost);
        return root;
        
    }
}
