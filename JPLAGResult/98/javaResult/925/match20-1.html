/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        return help(0, pre.length - 1, 0, post.length - 1, pre, post);
    }
    
    public TreeNode help(int s1, int e1, int s2, int e2, int[] pre, int[] post) {
        if (s1 > e1 || s2 > e2) {
            return null;
        }
        TreeNode node = new TreeNode(pre[s1]);
        if (s1 == e1) {
            return node;
        }
        if (s1 + 1 == e1) {
            node.left = new TreeNode(pre[e1]);
            return node;
        }
        if (pre[s1 + 1] == post[e2 - 1]) {
            node.left = help(s1 + 1, e1, s2, e2 - 1, pre, post);
            return node;
        }
        int left = pre[s1 + 1];
        int right = post[e2 - 1];
        int idx1 = -1;
        int idx2 = -1;
        for (int i = s1 + 1; i <= e1; i++) {
            if (pre[i] == right) {
                idx1 = i;
                break;
            }
        }
        for (int i = s2; i <= e2 - 1; i++) {
            if (post[i] == left) {
                idx2 = i;
                
break;
            }
        }
        // if (idx1 == -1 || idx2 == -1) {
        //     System.out.println(idx1);
        //     System.out.println(idx2);
        //     System.out.println(s1);
        //     System.out.println(e1);
        //     System.out.println(s2);
        //     System.out.println(e2);
        // }
        node.left = help(s1 + 1, idx1 - 1, s2, idx2, pre, post);
        node.right = help(idx1, e1, idx2 + 1, e2 - 1, pre, post);
        return node;
    }
}
