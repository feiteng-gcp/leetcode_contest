/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        if (pre == null || pre.length == 0) return null;
        TreeNode root = new TreeNode(pre[0]);
        if (pre.length == 1) return root;
        int i = 0;
        for (; i < post.length; i++) {
            if (post[i] == pre[1]) break;
        }
        int[] newpre = new int[i+1];
        int[] newpost = new int[i+1];
        System.arraycopy(pre, 1, newpre, 0, i+1);
        System.arraycopy(post, 0, newpost, 0, i+1);       
        root.left = constructFromPrePost(newpre, newpost);
        int len = pre.length;
        newpre = new int[len-i-2];
        newpost = new int[len-i-2];
        System.arraycopy(pre, i+2, newpre, 0, len-i-2);
        System.arraycopy(post, i+1, newpost, 0, len-i-2);
        root.right = constructFromPrePost(newpre, newpost);
        return root;
    }
}
