/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    TreeNode helper(int[] pre,int prei, int prej, int[] post, int posti, int postj){
        if(prei>prej||posti>postj) return null;
        //System.out.println(prei+" "+prej+" "+posti+" "+postj);
        TreeNode root=new TreeNode(pre[prei]);
        if(prei==prej) return root;
        int i=prei+1;
        int j=posti;
        for(;j<=postj;j++){
            if(post[j]==pre[i])
                break;
        }
       // System.out.println("left:"+i+" "+String.valueOf(i+(postj-j))+" "+posti+" "+j);
        root.left=helper(pre,i,i+(j-posti),post,posti,j);
        //System.out.println("right:"+i+(j-postj)+1+" "+prej+(j-postj)+" "+posti+" "+j);
       
        root.right=helper(pre, i+(j-posti)+1,prej,post,j+1,postj-1);
        return root;
    }
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
         return helper(pre,0,pre.length-1,post,0,post.length-1);
    }
}
