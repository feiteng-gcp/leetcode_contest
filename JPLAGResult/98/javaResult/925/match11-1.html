/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        if(pre == null || pre.length == 0) return null;
        
return dfs(pre, post, 0, pre.length-1, 0, pre.length-1);
    }
    
    public TreeNode dfs(int[] pre, int[] post, int a, int b, int x, int y){
        if(a > b || x > y) return null;
        TreeNode root = new TreeNode(pre[a]);
        if(a == b) return root;
        int leftH = pre[a+1], i = x;
        for(; i <= y; i++) if(post[i] == leftH) break;
        int leftL = i - x;
        root.left = dfs(pre, post, a+1, a+1+leftL, x, i);
        root.right = dfs(pre, post, a+2+leftL, b, i+1, y-1);
        return root;
    }
}
