/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        int len  = pre.length;
        
return imp(pre, post, 0, len-1, 0, len-1);        
    }
    
    public TreeNode imp(int[] pre, int[] post, int prestart, int preend, int poststart, int postend){
        if (prestart>preend || poststart > postend) return null;
        
        TreeNode node = new TreeNode(pre[prestart]);
        
        if (prestart==preend){            
            return node;
        }else{
            //look for prestart+1
            
for (int i=poststart; i<=postend; i++){
                if (post[i]==pre[prestart+1]){
                    node.left = imp(pre, post, prestart+1, prestart + i - poststart + 1, poststart, i);
                    node.right = imp(pre, post, prestart + i - poststart + 2, preend, i+1, postend);                    
                }
            }            
        }
        return node;
    }
        
}
