/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        return construct(pre, 0, pre.length - 1, post, 0, post.length - 1);
    }
    private TreeNode construct(int[] pre, int prel, int prer, int[] post, int postl, int postr) {
        if (prel > prer) {
            return null;
        }
        else if (prel == prer) {
            return new TreeNode(pre[prel]);
        }
        TreeNode node = new TreeNode(pre[prel]);
        prel++;
        int size = 0;
        for (int i = postl; i <= postr; i++) {
            if (post[i] == pre[prel]) {
                break;
            }
            size++;
        }
        node.left = construct(pre, prel, prel + size, post, postl, postl + size);
        node.right = construct(pre, prel + size + 1, prer, post, postl + size + 1, postr);
        return node;
    }
}
