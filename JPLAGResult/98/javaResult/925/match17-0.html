class Solution {
    public TreeNode constructFromPrePost(int[] pre, int[] post) {
        return construct(pre, 0, pre.length - 1, post, 0, post.length - 1);
    }
    
    public TreeNode construct(int[] pre, int left1, int right1, int[] post, int left2, int right2) {
        if(left1 > right1) {
            return null;
        }
        if(left1 == right1) {
            return new TreeNode(pre[left1]);
        }
        
        TreeNode root = new TreeNode(pre[left1]);
        int index = left2;
        while(index < right2 && post[index] != pre[left1 + 1]) {
            index++;
        }
        root.left = construct(pre, left1 + 1, left1 + 1 + index - left2, post, left2, index);
        root.right = construct(pre, left1 + 2 + index - left2, right1, post, index + 1, right2 - 1);
        return root;
    }
}
