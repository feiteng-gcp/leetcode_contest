class Solution {
    public int sumSubseqWidths(int[] A) {
        
        int mod = 1000000007;
        
        long[] count = new long[20001];
        long value = 1;
        for (int i=0; i<20001; i++){
            count[i] = value; 
            value = value * 2 % mod;
        }
        
        Arrays.sort(A);
        
        
Long sumRight = 0L;
        Long sumLeft = 0L;
        
        int n = A.length;
        
        for (int i=0; i<n; i++){
            sumLeft += A[i] * (count[n-i-1] );
            sumLeft = sumLeft % mod;
            
            sumRight += A[i] * count[i];
            sumRight = sumRight % mod;
        }
        
        return (sumRight.intValue()- sumLeft.intValue() + mod) % mod ;
    }
}
