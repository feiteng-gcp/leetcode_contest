class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        
        int diff = 0;
        Set<Integer> set = new HashSet<>();
        for (int num : A) {
            diff += num;
            set.add(num);
        }
        
        for (int num : B) {
            diff -= num;
        }
        for (int num : B) {
            if (set.contains(num + diff / 2)) {
                return new int[]{num + diff / 2, num};
            }
        }
        
        return new int[]{0, 0};
    }
}
