class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int ca = 0, ba = 0;
        HashSet<Integer> hashSet = new HashSet();
        for (int i = 0; i < A.length; i++) {
            ca += A[i];
            hashSet.add(A[i]);
        }
        for (int j = 0; j < B.length; j++) {
            ba += B[j];
        }
        int diff = (ca - ba) / 2;
        for (int j = 0; j < B.length; j++) {
            
if (hashSet.contains(B[j] + diff) == true) {
                return new int[] {B[j] + diff, B[j]};
            }
        }
        throw null;
    }
}
