class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        HashSet<Long> a=new HashSet<>();
        HashSet<Long> b=new HashSet<>();
        long sum=0;
        
for(int i=0;i<A.length;i++){
            a.add((long)A[i]);
            sum-=A[i];
        }
        for(int i=0;i<B.length;i++){
            b.add((long)B[i]);
            sum+=B[i];
        
}
        for(long c:a){
            if((sum+2*c)%2==0 && b.contains((sum+2*c)/2)){
                return new int[]{(int)c,(int)((sum+2*c)/2)};
            }   
        }
        return new int[]{0,0};
    }
}
