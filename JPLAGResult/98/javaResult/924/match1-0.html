class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int sumA = 0, sumB = 0;
        Set<Integer> set = new HashSet<>();
        for (int num : A) sumA += num;
        for (int num : B) {
            set.add(num);
            sumB += num;
        }
        int half = (sumA + sumB) / 2;
        for (int num : A) {
            int target = (half - sumA) + num;
            if (set.contains(target)) return new int[]{num, target};
        }
        return new int[]{0, 0};
    }
}
