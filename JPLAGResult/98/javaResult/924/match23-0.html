class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int sumA = 0;
        for(int a : A) {
            sumA += a;
        }
        int sumB = 0;
        for(int b : B) {
            sumB += b;
        }
        Arrays.sort(B);
        for(int i = 0; i < A.length; i++) {
            if(Arrays.binarySearch(B, A[i]-(sumA - sumB)/2) >= 0) {
                return new int[] {A[i], A[i]-(sumA - sumB)/2};
            }
        }
        return new int[]{0, 0};
    }
}
