class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int sumA = 0;
        int sumB = 0;
        Set<Integer> visited = new HashSet<>();
        for (int ele : A) {
            sumA += ele;
            visited.add(ele);
        }
        for (int ele : B) {
            sumB += ele;
        }
        
for (int ele : B) {
            int target = (sumA-sumB+2*ele)/2;
            if (visited.contains(target))
                return new int[]{target,ele};
        }
        return new int[]{0,0};
    }
}
