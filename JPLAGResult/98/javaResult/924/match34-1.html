class Solution {
  public int[] fairCandySwap(int[] A, int[] B) {
    int suma = 0, sumb = 0;
    for (int i = 0; i < A.length; ++i) {
      suma += A[i];
    }
    for (int j = 0; j < B.length; ++j) {
      sumb += B[j];
    
}
    for (int i = 0; i < A.length; ++i) {
      for (int j = 0; j < B.length; ++j) {
        if (suma - A[i] + B[j] == sumb - B[j] + A[i]) {
          return new int[] { A[i], B[j] };
        }
      }
    }
    return null;
  }
}
