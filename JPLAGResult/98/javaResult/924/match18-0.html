class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int sumA = 0;
        Set<Integer> setA = new HashSet<>();
        int sumB = 0;
        for (int a:A) {
            sumA += a;
            setA.add(a);
        }
        for (int b:B) {
            sumB += b;
        }
        int avg = (sumA+sumB)/2;
        int diff = avg-sumA;
        for (int b:B) {
            if (setA.contains(b-diff)) {
                return new int[]{b-diff,b};
            }
        }
        return new int[2];
    }
}
