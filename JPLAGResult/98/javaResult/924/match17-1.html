class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int diff = 0;
        Set<Integer> set = new HashSet<>();
        for(int value: A){
            set.add(value);
            diff += value;
        }
        for(int value: B){
            diff -= value;
        }
        
for(int value: B){
            if(set.contains(value + (diff / 2))){
                return new int[]{value + (diff / 2), value};
            }
        }
        return new int[2];
    }
}
