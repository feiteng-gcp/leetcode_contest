class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int sumA = 0;
        for (int num : A) {
            sumA += num;
        }
        int sumB = 0;
        HashSet<Integer> set = new HashSet<>();
        for (int num : B) {
            sumB += num;
            set.add(num);
        }
        int d = (sumB - sumA) / 2;
        
for (int num : A) {
            if (set.contains(num + d)) {
                return new int[] {num, num + d};
            }
        }
        return null;
    }
}
