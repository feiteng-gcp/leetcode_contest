class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int suma = 0, sumb = 0;
        Set<Integer> seta = new HashSet<Integer>();
        for(int a: A) {
            seta.add(a);
            suma +=  a;
        }
        Set<Integer> setb = new HashSet<>();
        for(int b: B) {
            setb.add(b);
            sumb += b;
        }
        int target = (suma + sumb)/2;
        for(int i = 0; i < A.length; i++){
            int leftA = suma - A[i];
            if(setb.contains(target - leftA))
                return new int[]{A[i], target - leftA};
        }
        return new int[]{};
    }
}
