class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int sa = 0;
        int sb = 0;
        Set<Integer> va = new HashSet<>();
        for (int i = 0; i < A.length; ++i) {
        	sa += A[i];
        	va.add(A[i]);
        }
        for (int i = 0; i < B.length; ++i) {
        	sb += B[i];
        }
        int avg = (int)(((long)sa + sb) / 2);
        int db = sb - avg;
        for (int i = 0; i < B.length; ++i) {
        	int cur = B[i];
        	int aa = B[i] - db;
        	if (va.contains(aa)) {
        		return new int[]{aa, cur};
        	}
        }
        throw new RuntimeException("WA");
    }
}
