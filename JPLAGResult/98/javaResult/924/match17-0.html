class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int sum = 0;
        HashSet<Integer> set = new HashSet();
        for (int a: A) {
            sum += a;
            set.add(a);
        }
        int sumA = sum;
        for (int b: B) sum += b;
        int diff = sumA - sum/2;
        for (int b: B) {
            if (set.contains(b+diff)) {
                return new int[]{b+diff, b}; 
            }
        }
        return new int[]{0,0};
    }
}
