	class Solution {
	    public int[] fairCandySwap(int[] A, int[] B) {
	        int bal = 0;
	        for(int v : A)bal -= v;
	        for(int v : B)bal += v;
	        for(int v : A){
	        	for(int w : B){
	        		if(w - v == bal/2){
	        			return new int[]{v, w};
	        		}
	        	}
	        }
	        return null;
	    }
	}
