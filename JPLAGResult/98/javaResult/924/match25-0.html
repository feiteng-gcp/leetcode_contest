class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int n = A.length;
        int m = B.length;
        int sumA = 0, sumB = 0;
        for (int a : A) {
            sumA += a;
        }
        for (int b : B) {
            sumB += b;
        }
        int[] ans = new int[2];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (sumA - A[i] + B[j] == sumB - B[j] + A[i]) {
                    ans[0] = A[i];
                    ans[1] = B[j];
                    return ans;
                }
            }
        }
        return ans;
    }
}
