class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int[] result = new int[2];
        
int sumA = 0, sumB = 0;
        List<Integer> AL = new LinkedList<>();
        List<Integer> BL = new LinkedList<>();
        for(int i:A){
            sumA += i;
            AL.add(i);
        }
        for(int i:B){
            sumB += i;
            BL.add(i);
        }
        if(sumA < (sumA+sumB)/2){
            int diff = (sumB-sumA) / 2;
            
for(int i=0; i<A.length; i++){
                if(BL.contains(A[i]+diff)){
                    result[0] = A[i];
                    result[1] = A[i]+diff;
                    break;
                }
            }
        }else{
            int diff = (sumA-sumB) / 2;
            for(int i=0; i<B.length; i++){
                if(AL.contains(B[i]+diff)){
                    result[0] = B[i]+diff;
                    result[1] = B[i];
                    break;
                }
            }
        }
        return result;
    }
}
