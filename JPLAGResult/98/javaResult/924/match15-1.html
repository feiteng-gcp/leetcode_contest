class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int sum1 = 0;
        int sum2 = 0;
        Set<Integer> set = new HashSet<>();
        for(int num : A) {
            sum1 += num;
            set.add(num);
        }
        for(int num : B) {
            sum2 += num;
        }
        int diff = (sum1 - sum2) / 2;
        for(int num : B) {
            if(set.contains(num + diff)) return new int[]{num + diff, num};
        }
        return new int[]{-1, -1};
    }
}
