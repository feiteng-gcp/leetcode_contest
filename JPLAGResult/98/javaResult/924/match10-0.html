class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int sum1 = 0;
        for (int i = 0; i < A.length; i++) {
            sum1 += A[i];
        }
        int sum2 = 0;
        for (int j = 0; j < B.length; j++) {
            sum2 += B[j];
        }
        for (int i = 0; i < A.length; i++) {
            for (int j = 0; j < B.length; j++) {
                if (sum1 - A[i] + B[j] == sum2 - B[j] + A[i]) {
                    return new int[]{A[i], B[j]};
                }
            }
        }
        return new int[]{-1, -1};
    }
}
