class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int ans[] = new int[2];

        int nA = A.length;
        int nB = B.length;
        int sumA = 0;
        int sumB = 0;

        for (int i = 0; i < nA; i++) {
            sumA += A[i];
        }
        for (int i = 0; i < nB; i++) {
            sumB += B[i];
        }

        int dif = sumA - sumB;
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < nB; i++) {
            set.add(B[i]);
        }

        for (int i = 0; i < nA; i++) {
            if (set.contains(A[i] - dif / 2)) {
                ans[0] = A[i];
                ans[1] = A[i] - dif / 2;
                break;
            }
        }

        return ans;
    }
}
