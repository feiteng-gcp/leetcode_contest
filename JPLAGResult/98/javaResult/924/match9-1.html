class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int s1 = 0;
        int s2 = 0;
        for(int a:A){
            s1 += a;
        }
        for(int b:B){
            s2 += b;
        }
        
       // int d = s1-s2;
        
        for(int i=0;i<A.length;i++){
            for(int j=0;j<B.length;j++){
                if(s1-A[i]+B[j]==s2-B[j]+A[i]){
                    return new int[]{A[i],B[j]};
                }
            }
        }
        return new int[]{0,0};
        
        
        
        
    }
}
