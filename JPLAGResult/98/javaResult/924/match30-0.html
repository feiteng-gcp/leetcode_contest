class Solution {

    public int[] fairCandySwap(int[] A, int[] B) {

    	int sum1 = 0 , sum2 = 0;
    	Set<Integer> set = new HashSet<>();
    	for (int value : A) {
    		sum1 += value;
    	}
    	for (int value : B) {
    		sum2 += value;
    	}
    	for (int value : B) {
    		set.add(value);
    	}
    	for (int value : A) {
    		int need = (sum2 - sum1) / 2 + value;
    		if (set.contains(need)) {
    			return new int[] {value , need};
    		}
    	}
    	return new int[] {};
    	
    }
	
}
