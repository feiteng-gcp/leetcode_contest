class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int[] ret = new int[2];
        int sum1 = 0,sum2 = 0;
        Set<Integer> set =new HashSet<>();
        for(int v:A) {
            sum1+=v;
            set.add(v);
        }
        for(int v:B) sum2+=v;
        int diff = (sum1-sum2)/2;
        
for(int v:B){
            if(set.contains(v+diff)){
                ret[0]=v+diff;
                ret[1]=v;
                break;
            }
        }
        return ret;
    }
}
