class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        int sum=0,sumA=0,sumB=0;
        Set<Integer> setA=new HashSet<>();
        Set<Integer> setB=new HashSet<>();
        for(int i=0;i<A.length;i++)
        {
            sumA+=A[i];
            setA.add(A[i]);
        }
        for(int i=0;i<B.length;i++)
        {
            sumB+=B[i];
            setB.add(B[i]);
        }
        sum=sumA+sumB;
        int even=sum/2;
        int[] res=new int[2];
        for(int i=0;i<A.length;i++)
        {
            if(setB.contains(A[i]-sumA+even))
            {
               
                res[0]=A[i];
                res[1]=A[i]-sumA+even;
                break;
            }
           
        }
        return res;
        
    }
}
