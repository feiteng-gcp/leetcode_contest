class Solution {
    public int[] fairCandySwap(int[] A, int[] B) {
        Set<Long> hs = new HashSet<>();
        for (int v : B) {
            hs.add(v * 1L);
        }

        long sum = 0;
        long sum1 = 0;
        for (int v : A) {
            sum += v;
            sum1 += v;
        
}
        for (int v : B) {
            sum += v;
        }

        sum /= 2;
        for (int v : A) {
            long need = v - (sum1 - sum);
            
if (hs.contains(need)) {
                return new int[]{v, (int) need};
            }
        }

        return new int[]{};
    }
}
