class Solution {
	
    public List<String> findAndReplacePattern(String[] words, String pattern) {
    	List<String> ans  = new ArrayList<String>();
    	char patternArray[] = pattern.toCharArray();
    	Map<Character, Character> relation = new HashMap<Character, Character>();
    	Map<Character, Character> rRelation = new HashMap<Character, Character>();
        for (String word : words)
        {
        	relation.clear();
        	rRelation.clear();
        	
        	char wArray[] = word.toCharArray();
        	
        	boolean isAccepted = true;
        	for (int i = 0 ; i < word.length() ; i++)
        	{
        		char a = wArray[i];
        		char b = patternArray[i];
        		char expA, expB;
        		
        		if (relation.containsKey(b))
        		{
        			expA = relation.get(b);
        			if (expA != a)
        			{
        				isAccepted = false;
        				break;
        			}
        			
        			expB = rRelation.get(a);
        			if (expB != b)
        			{
        				isAccepted = false;
        				break;        				
        			}
        		}
        		else
        		{
        			if (rRelation.containsKey(a))
        			{
        				isAccepted = false;
        				break;
        			}
        			relation.put(b, a);
        			rRelation.put(a, b);
        		}
        	}
        	
        	if (isAccepted)
        		ans.add(word);
        }
        
        return ans;
    }
}
