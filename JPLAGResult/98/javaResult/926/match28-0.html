class Solution {
    public List<String> findAndReplacePattern(String[] words, String pattern) {
        List<String> res = new ArrayList<>();
        int[] arr1 = new int[26];
        int[] arr2 = new int[26];
        for (String s : words) {
            Arrays.fill(arr1, -1);    
            Arrays.fill(arr2, -1);    
            if (match(s, pattern, arr1, arr2)) res.add(s);
        }
        return res;
    }
    
    private boolean match(String s, String t, int[] arr1, int[] arr2) {
        
        // System.out.println(s + " " + t);
        if (s.length() != t.length()) return false;
        int len = s.length();
        for (int i = 0; i < len; i++) {
            char c = s.charAt(i);
            
            int idx = c - 'a';
            int val = t.charAt(i) - 'a';
            // System.out.println(idx + " " + val);
            //never occur
            if (arr1[idx] == -1) {
                if (arr2[val] != -1) return false;
                arr1[idx] = val;
                arr2[val] = idx;
            } else {
                if (arr1[idx] != val) return false;
            }
        }
        
        return true;
    }
    
}
