class Solution {
    public List<String> findAndReplacePattern(String[] w, String p) {
        
        int wl = w.length;
        
        List<Integer> ml = cal(p);
        
        
        List<String> ans = new ArrayList<>();
        
        
for(int i = 0; i < wl; i++)
        {
            List<Integer> cl = cal(w[i]);
            
            if(ism(cl, ml))
            {
                ans.add(w[i]);
            }
        }
        
        return ans;
    }
    
    boolean ism(List<Integer> a, List<Integer> b)
    {
        if(a.size() != b.size())
        {
            return false;
        }
        
        for(int i = 0; i < a.size(); i++)
        {
            if(!a.get(i).equals(b.get(i)))
            {
                return false;
            }
        
}
        
        return true;
    }

    List<Integer> cal(String s)
    {
        Map<Character, Integer> map = new HashMap<>();
        
        List<Integer> l = new ArrayList<Integer>();
        
        int cnum = 1;
        
for(int i = 0; i < s.length(); i++)
        {
            Character ch = s.charAt(i);
            
            if(map.containsKey(ch))
            {
                l.add(map.get(ch));
            }
            else
            {
                
map.put(ch, cnum);
                l.add(cnum);
                cnum++;
            }
        }
        
        return l;
    }
