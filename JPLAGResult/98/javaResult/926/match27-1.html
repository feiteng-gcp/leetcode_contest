class Solution {
    boolean ok(String w,String p){
        if(w.length()!=p.length()){
            return false;
        }
        Map<Character,Character> mp = new HashMap<>();
        Set<Character> set = new HashSet<>();
        for(int i=0;i<w.length();i++){
            if(mp.containsKey(p.charAt(i))){
                char c = mp.get(p.charAt(i));
                if(c!=w.charAt(i)){
                    return false;
                }
            }else{
                if(set.contains(w.charAt(i))){
                    return false;
                }
                mp.put(p.charAt(i),w.charAt(i));
                set.add(w.charAt(i));
            }
        }
        return true;
        
    
}
    
    
    
    public List<String> findAndReplacePattern(String[] words, String p) {
        List<String> l = new ArrayList<>();
        for(String w:words){
            if(ok(w,p)){
                l.add(w);
            }
        }
        return l;
    }
}
