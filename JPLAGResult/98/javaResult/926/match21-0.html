class Solution {
    public List<String> findAndReplacePattern(String[] words, String pattern) {
        List<String> list = new ArrayList<>();
        for(String w:words)
            if(match(w.toLowerCase(),pattern.toLowerCase()))
                list.add(w);
        return list;
        
    }
    
    private boolean match(String w, String p){
        if(w.length()!=p.length()) return false;
        int[] ps = new int[26];
        int[] ws = new int[26];
        Arrays.fill(ps,-1);
        Arrays.fill(ws,-1);
        for(int i=0;i<w.length();i++){
            char chw = w.charAt(i);
            char chp = p.charAt(i);
            if(ps[chp-'a']==-1)
                ps[chp-'a']=chw-'a';
            else if(ps[chp-'a']!=(chw-'a')) return false;
            
            if(ws[chw-'a']==-1)
                ws[chw-'a']=chp-'a';
            else if(ws[chw-'a']!=(chp-'a')) return false;
        }
        return true;
    }
}
