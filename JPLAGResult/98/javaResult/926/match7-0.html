class Solution {
    public List<String> findAndReplacePattern(String[] words, String pattern) {
        List<String> result = new ArrayList<String>();
        for (int i = 0; i < words.length; i++) {
            if (isMatch(words[i], pattern) == true) {
                result.add(words[i]);
            }
        }
        return result;
    }
    private boolean isMatch(String word, String pattern) {
        if (word.length() != pattern.length()) {
            return false;
        }
        HashMap<Character, Character> wtop = new HashMap();
        HashMap<Character, Character> ptow = new HashMap();
        for (int i = 0; i < word.length(); i++) {
            char w = word.charAt(i);
            char p = pattern.charAt(i);
            if (wtop.containsKey(w) == true) {
                if (wtop.get(w) != p || ptow.containsKey(p) == false || ptow.get(p) != w) {
                    return false;
                }
            }
            else {
                if (ptow.containsKey(p) == true) {
                    return false;
                }
                wtop.put(w, p);
                ptow.put(p, w);
            }
        }
        return true;
    }
}
