class Solution {
    public List<String> findAndReplacePattern(String[] words, String pattern) {
        List<String> ans = new ArrayList<>();
        if (words.length == 0) return ans;
        for (int i = 0; i < words.length; i++) {
            
if (isMatch(words[i], pattern)) ans.add(words[i]);
        }
        return ans;
    }
    
    private boolean isMatch(String w, String p) {
        if (w.length() != p.length()) return false;
        char[] cp = p.toCharArray();
        char[] cw = w.toCharArray();
        int len = w.length();
        HashMap<Character, Character> map = new HashMap<>();
        Set<Character> set = new HashSet<>();
        for (int i = 0; i < len; i++) {
            if (!map.containsKey(cp[i])) {
                if (set.contains(cw[i])) return false;
                else {
                    set.add(cw[i]);
                    map.put(cp[i], cw[i]);
                }
            } else {
                if (map.get(cp[i]) != cw[i]) 
return false;
            }
        }
        return true;
    }
}
