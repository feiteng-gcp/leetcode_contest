class Solution {
	
	private boolean check(String string , String pattern) {
		
		if (string.length() == pattern.length()) {
			Map<Character , Chara
cter> map1 = new HashMap<>();
			Map<Character , Character> map2 = new HashMap<>();
			for (int i = 0;i < string.length();i ++) {
				char ch1 = string.charAt(i) , ch2 = pattern.charAt(i);
				if (map1.containsKey(ch1) && map2.containsKey(ch2)) {
					if (map1.get(ch1) == ch2 && map2.get(ch2) == ch1) {
						continue;
					} else {
						return false;
					}
				} else if (!map1.containsKey(ch1) && !map2.containsKey(ch2)) {
					map1.put(ch1 , ch2);
					map2.put(ch2 , ch1);
				} else {
					return false;
				}
			}
			return true;
		} else {
			return false;
		}
		
	}
	
    
public List<String> findAndReplacePattern(String[] words, String pattern) {
    
    	List<String> ans = new ArrayList<>();
    	for (String string : words) {
    		if (check(string , pattern)) {
    			ans.add(string);
    		}
    	}
    	return ans;
    	
    }
    
}
