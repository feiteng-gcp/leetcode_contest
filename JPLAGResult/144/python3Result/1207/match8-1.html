# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        ans = []
        dset = set(to_delete)
        
        def dfs(root, del_fa):
            if not root:
                return
            del_cur = root
.val in dset
            dfs(root.left, del_cur)
            dfs(root.right, del_cur)
            if root.left and root.left.val in dset:
                root.left = None
            if root.right and root.right.val in dset:
                root.right = None
            if del_fa and not del_cur:
                ans.append(root)
        
        dfs(root, True)
        return ans
