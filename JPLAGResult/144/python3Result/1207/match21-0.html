# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        ans = []
        dels = set(to_delete)
        def helper(node, isRoot):
            if not node: return
            if node.left:
                helper(node.left, node.val in dels)
                if node.left.val in dels: node.left = None
            if node.right:
                helper(node.right, node.val in dels)
                if node.right.val in dels: node.right = None
            if isRoot and node.val not in dels:
                ans.append(node)
        helper(root, True)
        return ans
