# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        
        def dn(root: TreeNode, td: set, parent: TreeNode) -> set:
            if root is None:
                return set()
            if len(td) == 0:
                return set([root])
            lset = dn(root.left, td, root)
            rset = dn(root.right, td, root)
            if root.val in td:                
                if parent:
                    if root == parent.left:
                        parent.left = None
                    else:
                        parent.right = None
                return lset.union(rset)
            if root.left in lset:
                lset.remove(root.left)
            if root.right in rset:
                rset.remove(root.right)
            lset.add(root)
            return lset.union(rset)
        
        res = dn(root, to_delete, None)
        return [i for i in res]
