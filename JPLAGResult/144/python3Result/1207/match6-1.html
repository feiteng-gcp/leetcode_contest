class Solution:
    def delNodes(self, root: TreeNode, to_del: List[int]) -> List[TreeNode]:
        out = []

        def aux(node, root=False):
            if not node:
                return
            if node
.val in to_del:
                aux(node.left, True)
                aux(node.right, True)
                return None
            else:
                if root:
                    out.append(node)
                node.left = aux(node.left)
                node.right = aux(node.right)
                return node

        to_del = set(to_del)
        aux(root, root=True)
        return out
