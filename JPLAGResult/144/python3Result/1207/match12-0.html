# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        delete = set(to_delete)
        ans = []
        def dfs(root, to_add):
            if root:
                if root.val in delete:
                    dfs(root.left, True)
                    dfs(root.right, True)
                else:
                    dfs(root.left, False)
                    dfs(root.right, False)
                    if root.left and root.left.val in delete:
                        root.left = None
                    if root.right and root.right.val in delete:
                        root.right = None
                    if to_add:
                        ans.append(root)
        dfs(root, True)
        return ans
