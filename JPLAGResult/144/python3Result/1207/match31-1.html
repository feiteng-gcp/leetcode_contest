# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        if not to_delete or not root:
            return [root]
        s = set(to_delete)
        res = []
        q = [[root, True]]
        while q:
            nq = []
            for n, add in q:
                if add and n.val not in s:
                    res.append(n)
                if n.left:
                    nq.append([n.left, n.val in s])
                    if n.left.val in s:
                        n.left = None
                if n.right:
                    nq.append([n.right, n.val in s])
                    if n.right.val in s:
                        n.right = None
            q = nq
        return res
