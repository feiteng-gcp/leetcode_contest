# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        result = []
        def recSol(node, result):
            if node is None:
                return None
            node.left = recSol(node.left, result)
            node.right = recSol(node.right, result)
            if node.val in to_delete:
                
if node.left is not None:
                    result.append(node.left)
                if node.right is not None:
                    result.append(node.right)
                return None
            return node
        
        root = recSol(root, result)
        if root is not None:
            result.append(root)
        return result
