# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        to_delete = set(to_delete)
        m = { }
        
        def dfs(c, p):
            m[c.val] = c, p
            if c.left: dfs(c.left, c)
            if c.right: dfs(c.right, c)
        dfs(root, None)
        
        ans = [root]
        for d in to_delete:
            c, p = m[d]
            if c.left: ans.append(c.left)
            if c.right: ans.append(c.right)
            if p and p.left == c: p.left = None
            if p and p.right == c: p.right = None
        
        ans = [c for c in ans if c.val not in to_delete]
        return ans
