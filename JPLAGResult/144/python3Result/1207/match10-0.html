# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        tds = set(to_delete)
        ret = []
        def helper(tds, ret, r, ir=False):
            if not r:
                return
            if r.val in tds:
                helper(tds, ret, r.left, True)
                helper(tds, ret, r.right, True)
            else:
                if ir:
                    ret.append(r)
                helper(tds, ret, r.left, False)
                helper(tds, ret, r.right, False)
            if r.left and r.left.val in tds:
                r.left = None
            if r.right and r.right.val in tds:
                r.right = None
        helper(tds, ret, root, True)
        return ret
