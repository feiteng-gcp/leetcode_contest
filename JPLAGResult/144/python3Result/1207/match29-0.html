# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def delNodescheck(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        ans=[]
        if root==None:
            return []
        if root.val not in to_delete:
            ans.append(root)
        if not root.right==None:
            ans=ans+self.delNodescheck(root.right,to_delete)
            if root.right.val in to_delete:
                root.right=None
        if not root.left==None:
            ans=ans+self.delNodescheck(root.left,to_delete)
            if root.left.val in to_delete:
                root.left=None
        return ans
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        ans=self.delNodescheck(root,to_delete)
        pd=[True for i in range(len(ans))]
        dic=[-1 for i in range(1001)]
        
        for i in range(len(ans)):
            dic[ans[i].val]=i
            
        for i in range(len(ans)):
            r=ans[i]
            if not r.right==None:
                pd[dic[r.right.val]]=False
            if not r.left==None:
                pd[dic[r.left.val]]=False
        reans=[]
        
        for i in range(len(ans)):
            if pd[i]==True:
                reans.append(ans[i])
        return reans
