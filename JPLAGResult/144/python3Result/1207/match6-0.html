# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        forest = []
        to_del = set(to_delete)
        def delete(n, is_root):
            if n is None:
                return None
            elif n.val in to_del:
                delete(n.left, True)
                delete(n.right, True)
                return None
            else:
                if is_root:
                    forest.append(n)
                n.left = delete(n.left, False) 
                n.right = delete(n.right, False)
                return n
        delete(root, True)
        return forest
