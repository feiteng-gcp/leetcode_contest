# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def delNodes(self, root: TreeNode, to_delete: List[int]) -> List[TreeNode]:
        
        l1 = []
        
        def dfs(r):
            if r == None:
                return
            l1.append(r.val)
            dfs(r.left)
            dfs(r.right)
            
        dfs(root)
        
        ch = set(to_delete)
        l2 = []
        
        for i in l1[::-1]:
            if i in ch:
                l2.append(i)
                
        res = [root]
        
        def rm(r, v):
            
            if r == None:
                return False
            
            if r.val == v:
                
                if r.left:
                    res.append(r.left)
                if r.right:
                    res.append(r.right)
                    
                # r = None
                return True
            
            if rm(r.left, v):
                r.left = None
            if rm(r.right, v):
                r.right = None
            
        for i in l2:
            rm(root, i)
            
        if l2[-1] == root.val:
            return res[1:]
        else:
            return res
