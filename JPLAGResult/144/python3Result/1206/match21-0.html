class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        events = {i:0 for i in range (1, n+2)}
        for [i, j, count] in bookings:
            events.setdefault (i, 0)
            events.setdefault (j+1, 0)
            events[i] += count
            events[j+1] -= count
        ans = []
        current = 0
        for flight in range (1, n+1):
            current += events[flight]
            ans.append (current)
        return ans
