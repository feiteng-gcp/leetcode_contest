class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        l=[]
        for i, j, k in bookings:
            l.append((i-1, k))
            l.append((j,-k))
        l.sort()
        ans=[]
        cur=0
        li=0
        for i in range(0, n):
            while li<len(l) and l[li][0]<=i:
                cur+=l[li][1]
                li+=1
            ans.append(cur)
        return ans
