class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        ends = []
        for i, j, k in bookings:
            ends.append((i - 1, False, k))
            ends.append((j, True, k))
        ends.sort()
        ans = [0] * n
        curr = 0
        seats = 0
        for i, end, k in ends:
            if curr < i:
                ans[curr:i] = [seats] * (i - curr)
                curr = i
            if not end:
                seats += k
            else:
                seats -= k
        return ans
