class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        f = []
        for i, j, k in bookings:
            f.append([i, k])
            f.append([j+1, -k])
        
        f.sort()
        ans = [0] * n
        c = 0
        j = 0
        for i in range(1, n + 1):
            while j < len(f) and f[j][0] <= i:
                c += f[j][1]
                j += 1
            ans[i-1] = c
        return ans
