class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        arr = [0]*(n+1)
        for i,j,k in bookings:
            arr[i-1] += k
            arr[j] -= k
        # print(arr)
        cur = 0
        for i in range(n):
            cur += arr[i]
            arr[i] = cur
        
        return arr[:-1]
