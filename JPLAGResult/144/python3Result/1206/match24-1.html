class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        data=[]
        for b in bookings:
            data.append([b[0], b[2]])
            data.append([b[1]+1, -b
[2]])
        data.sort()
        curr=0
        ret=[]
        j=0
        for i in range(1, n+1, 1):
            while j<len(data) and data[j][0]==i:
                curr+=data[j][1]
                j+=1
            ret.append(curr)
        return ret
