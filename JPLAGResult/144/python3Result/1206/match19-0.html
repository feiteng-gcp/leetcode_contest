class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        dt = [0]*(n+2)
        for i, j, k in bookings:
            dt[i] += k
            dt[j+1] -= k
        ans = [0]*n
        count = 0
        for i in range(n):
            count += dt[i+1]
            ans[i] = count
        return ans
