from collections import defaultdict
class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        d = defaultdict(int)
        for i, j, k in bookings:
            d[i] += k
            d[j + 1] -= k
        ret = []
        cur = 0
        for i in range(1, n + 1):
            cur += d[i]
            ret.append(cur)
        return ret
