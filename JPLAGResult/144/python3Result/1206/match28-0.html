class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        event = [0] * (n + 1)
        for i, j, k in bookings:
            event[i-1] += k
            event[j] -= k
        ans = []
        s = 0
        for idx, num in enumerate(event):
            if idx == len(event) - 1: break
            s += num
            ans.append(s)
        return ans
