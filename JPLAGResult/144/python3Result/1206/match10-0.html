class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        
        d = [0] * (n + 2)
        
        for b in bookings:
            d[b[0]] += b[2]
            d[b[1] + 1] -= b[2]
            
        res = []
        v = 0
        for i in d[1:-1]:
            v += i
            res.append(v)
            
        return res
