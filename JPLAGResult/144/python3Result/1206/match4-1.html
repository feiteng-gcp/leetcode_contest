class Solution:
    def corpFlightBookings(self, bookings: List[List[int]], n: int) -> List[int]:
        h = []
        for s, e, v in bookings:
            h.append([s - 1, v])
            h.append([e, -v])
        heapq.heapify(h)
        res = [0] * n
        c = 0
        i = 0
        while h:
            j, dt = heapq.heappop(h)
            while i < j:
                res[i] = c
                i += 1
            c += dt
        while i < len(res):
            res[i] = c
            i += 1
        return res
