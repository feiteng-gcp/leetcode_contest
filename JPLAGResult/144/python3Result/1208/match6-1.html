class Solution:
    def maxDepthAfterSplit(self, seq: str) -> List[int]:
        # idea: max(depth(A), depth(B)) can only be as low as 1/2 * depth(seq),
        #       because otherwise A and B are either (a) not actually VPS's, or
        #       (b) do not actually constitute the smallest possible max(depth(A),
        #       depth(B)). So, just choose any A and B where depth(A) = depth(B)
        #       or depth(A) = depth(B) + 1. They'll be equal if depth(seq) is
        #       an even number, otherwise they'll be off by one.
        
        depthSeq = 0
        curDepth = 0
        for el in seq:
            if el == '(':
                curDepth += 1
            else:
                curDepth -= 1
            
            depthSeq = max(depthSeq, curDepth)

        ret = []
        curDepth = 0
        for el in seq:
            if el == '(':
                curDepth += 1
            else:
                curDepth -= 1
            
            if curDepth > depthSeq//2 or (curDepth == depthSeq//2 and el != '('):
                ret.append(1)
            else:
                ret.append(0)
                
        return ret
