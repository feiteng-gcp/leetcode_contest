class Solution:
    def maxDepthAfterSplit(self, seq: str) -> List[int]:
        depa = 0
        depb = 0
        ret = []
        for c in seq:
            if c == '(':
                if depa > depb:
                    ret.append(0)
                    depb += 1
                else:
                    ret.append(1)
                    depa += 1
            else:
                if depa >= depb:
                    ret.append(1)
                    depa -= 1
                else:
                    ret.append(0)
                    depb -= 1
        return ret
