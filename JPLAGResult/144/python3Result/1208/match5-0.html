class Solution:
    def maxDepthAfterSplit(self, seq: str) -> List[int]:
        d1 = d2 = 0
        out = []
        for ch in seq:
            if ch == '(':
                if d1 < d2:
                    out.append(0)
                    d1 += 1
                else:
                    out.append(1)
                    d2 += 1
            else:
                if d1 > d2:
                    out.append(0)
                    d1 -= 1
                else:
                    out.append(1)
                    d2 -= 1
        return out
