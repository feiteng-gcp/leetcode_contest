class Solution:
    def maxDepthAfterSplit(self, seq: str) -> List[int]:
        if not seq: return []
        A,B = [],[]
        dA,dB = 0,0
        for i,c in enumerate(seq):
            
if c == '(':
                if dA<=dB:
                    A.append(i)
                    dA += 1
                else:
                    B.append(i)
                    dB += 1
            else:
                if dA>=dB:
                    A.append(i)
                    dA -= 1
                else:
                    B.append(i)
                    dB -= 1
        A = set(A)
        return [0 if i in A else 1 for i in range(len(seq))]
