class Solution:
    def maxDepthAfterSplit(self, seq: str) -> List[int]:
        
        maxd = 0
        stack = []
        
        c = 0
        for i in seq:
            if i == '(':
                stack.append(i)
                c += 1
                maxd = max(maxd, c)
            else:
                stack.pop()
                c -= 1
                
        mid = maxd // 2
        
        # print (maxd, mid)
        
        res = [0] * len(seq)
        c = 0
        # stack = []
        for i, vi in enumerate(list
(seq)):
            if vi == '(':
                # stack.append(vi)
                c += 1
                if c > mid:
                    res[i] = 1
            else:
                if c > mid:
                    res[i] = 1
                c -= 1
                
        return res
