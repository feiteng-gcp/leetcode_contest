class Solution:
    def maxDepthAfterSplit(self, seq: str) -> List[int]:
        ma = 0
        cur = 0
        for c in seq:
            if c == '(':
                cur += 1
            else:
                cur -= 1
            ma = max(cur, ma)
        lev = ma // 2
        #print(ma, lev)
        ans = [0]*len(seq)
        for i, c in enumerate(seq):
            if c == '(':
                cur += 1
                if cur > lev:
                    ans[i] = 1
            else:
                if cur > lev:
                    ans[i] = 1
                cur -= 1
        return ans
