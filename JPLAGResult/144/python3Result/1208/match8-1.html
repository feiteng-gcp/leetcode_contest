class Solution:

    def isValid(self, maxD):
        sol = []
        d0 = 0
        d1 = 0
        for c in self.seq:
            if c == "(":
                if d0 < maxD:
                    d0 += 1
                    sol.append(0)
                else:
                    if d1 < maxD:
                        d1 += 1
                        sol.append(1)
                    else:
                        return False, []
            else:
                if d0 > 0:
                    d0 -= 1
                    sol.append(0)
                else:
                    if d1 > 0:
                        d1 -= 1
                        sol.append(1)
                    else:
                        return False, []
        return True, sol

    def maxDepthAfterSplit(self, seq):
        if not seq:
            return 0
        self.seq = seq
        maxD 
= 0
        d = 0
        for c in seq:
            if c == "(":
                d += 1
            else:
                d -= 1
            maxD = max(maxD, d)
        l = 1
        r = maxD
        ansD = maxD
        ans = [0]*len(seq)
        while l<=r:
            mid = (l+r)//2
            ok, sol = self.isValid(mid)
            # print (l, r, mid, ok)
            if ok:
                if mid < ansD:
                    ansD = mid
                    ans = sol
                r = mid - 1
            else:
                l = mid + 1
        # print (ansD)
        return ans
