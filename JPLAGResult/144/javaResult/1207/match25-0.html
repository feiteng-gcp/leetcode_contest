/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        Set<Integer> set = new HashSet<>();
        for(int i: to_delete)
            set.add(i);
        List<TreeNode> res = new ArrayList<>();
        if(root == null)
            return res;
        if(!set.contains(root.val))
            res.add(root);
        travel(res, root, set);
        return res;
    }
    private void travel(List<TreeNode> res, TreeNode root, Set<Integer> set){
        if(set.contains(root.val)){
            if(root.left != null){
                travel(res, root.left, set);
                if (!set.contains(root.left.val))
                    res.add(root.left);
            }
            if(root.right != null){
                travel(res, root.right, set);
                if (!set.contains(root.right.val))
                    res.add(root.right);
            }        
        }else{
            if(root.left != null){
                travel(res, root.left, set);
                if(set.contains(root.left.val))
                    root.left = null;
            } 
            if(root.right != null){
                travel(res, root.right, set);
                if(set.contains(root.right.val))
                    root.right = null;
            } 
        }
    }
}
