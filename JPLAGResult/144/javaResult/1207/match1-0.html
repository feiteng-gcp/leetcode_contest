/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> res = new ArrayList<>();
        Set<Integer> set = new HashSet<>();
        TreeNode dummyHead = new TreeNode(0);
        dummyHead.left = root;
        set.add(0);
        for (int d : to_delete) {
            set.add(d);
        }
        dfs(dummyHead, set, res);
        return res;
    }
    
    private void dfs(TreeNode node, Set<Integer> set, List<TreeNode> res) {
        if (node == null) {
            return;
        }
        TreeNode leftCopy = node.left;
        TreeNode rightCopy = node.right;
        if (set.contains(node.val)) {
            if (leftCopy != null && !set.contains(leftCopy.val)) {
                res.add(leftCopy);
            }
            if (rightCopy != null && !set.contains(rightCopy.val)) {
                res.add(rightCopy);
            }
        }
        dfs(leftCopy, set, res);
        dfs(rightCopy, set, res);
        if (node.left != null && set.contains(node.left.val)) {
            node.left = null;
        }
        if (node.right != null && set.contains(node.right.val)) {
            node.right = null;
        }
    }
}
