/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> list = new ArrayList<>();
        Set<Integer> set = new HashSet<>();
        for(int d : to_delete)
            set.add(d);
        if(!set.contains(root.val))
            list.add(root);
        recursion(root, set, list);
        return list;
    }
    
    public void recursion(TreeNode root, Set<Integer> set, List<TreeNode> list){
        if(root == null)
            return;
        if(set.contains(root.val)){
            if(root.left != null && !set.contains(root.left.val))
                list.add(root.left);
            if(root.right != null && !set.contains(root.right.val))
                list.add(root.right);
        }
        if(root.left != null){
            recursion(root.left, set, list);
            if(set.contains(root.left.val))
                root.left = null;
        }
        if(root.right != null){
            recursion(root.right, set, list);
            if(set.contains(root.right.val))
                root.right = null;
        }
        
    }
}
