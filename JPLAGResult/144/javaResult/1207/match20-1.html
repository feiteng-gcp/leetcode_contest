/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    List<TreeNode> ret = new ArrayList<>();
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        Set<Integer> d = new HashSet<>(); 
        for (int x : to_delete) {
            d.add(x); 
        }
        root = 
dfs(root, d);
        if (root != null) {
            ret.add(root);
        }
        return ret;
    }
    
    private TreeNode dfs(TreeNode root, Set<Integer> d) {
        
        if (root == null) {
            return null; 
        
}
        TreeNode left = dfs(root.left, d);
        TreeNode right = dfs(root.right, d);
        if (d.contains(root.val)) {
            if (left != null) {
                ret.add(left);
            }
            if (right != null) {
                ret.add(right);
            }
            return null; 
        }
        root.left = left; 
        root.right = right; 
        return root;
        
    }
}
