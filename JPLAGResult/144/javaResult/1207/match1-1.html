/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> trees = new ArrayList<>();
        HashSet<Integer> delete = new HashSet<>();
        for (int i : to_delete) {
            delete.add(i);
        }
        if (!delete.contains(root.val)) {
            trees.add(root);
        }
        
delNodes(root, delete, trees);
        return trees;
    }
    public void delNodes(TreeNode root, HashSet<Integer> delete, List<TreeNode> trees) {
        if (root == null) return;
        delNodes(root.left, delete, trees);
        delNodes(root.right, delete, trees);
        
if (delete.contains(root.val)) {
            if (root.left != null && !delete.contains(root.left.val)) {
                trees.add(root.left);
            }
            if (root.right != null && !delete.contains(root.right.val)) {
                trees.add(root.right);
            }
        }
        
if (root.left != null && delete.contains(root.left.val)) {
            root.left = null;
        }
        if (root.right != null && delete.contains(root.right.val)) {
            root.right = null;
        }
    }
}
