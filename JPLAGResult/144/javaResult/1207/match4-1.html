/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> result = new ArrayList<>();
        Set<Integer> set = new HashSet<>();
        for (int d : to_delete) {
            set.add(d);
        }
        
TreeNode node = helper(root, set, result);
        if (node != null) {
            result.add(node);
        }
        return result;
    }
    private TreeNode helper(TreeNode node, Set<Integer> set, List<TreeNode> result) {
        if (node == null) {
            return null;
        
}
        TreeNode left = helper(node.left, set, result);
        TreeNode right = helper(node.right, set, result);
        if (set.contains(node.val)) {
            if (left != null) {
                result.add(left);
            }
            if (right != null) {
                result.add(right);
            }
            return null;
        }
        node.left = left;
        node.right = right;
        return node;
    }
}
