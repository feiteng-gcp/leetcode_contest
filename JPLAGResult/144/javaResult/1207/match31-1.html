/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Set<Integer> del = new HashSet<>();
    List<TreeNode> res = new ArrayList<>();

    TreeNode dfs(
TreeNode root, boolean r) {
        if (null == root) {
            return null;
        }
        if (del.contains(root.val)) {
            dfs(root.left, true);
            dfs(root.right, true);
            
return null;
        } else {
            if (r) {
                res.add(root);
            }
            root.left = dfs(root.left, false);
            root.right = dfs(root.right, false);
            return root;
        }
    }

    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        
for (int d : to_delete) {
            del.add(d);
        }
        dfs(root, true);
        return res;
    }
}
