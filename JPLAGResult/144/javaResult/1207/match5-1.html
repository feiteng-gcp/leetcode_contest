/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public void helper(TreeNode root, 
HashSet<Integer> set, ArrayList<TreeNode> ans, boolean hasParent) {
        if(root == null) {
            return;
        }
        
System.out.println("Call " + root.val);
        if(set.contains(root.val)) {
            if(root.left!=null) {
                helper(root.left, set, ans, false);
            }
            if(root.right!=null) {
                helper(root.right, set, ans, false);
            }
        } else {
            if(root.left!=null) {
                helper(root.left, set, ans, true);
                if(set.contains(root.left.val)){
                    root.left = null;
                }
            }
            if(root.right!=null) {
                helper(root.right, set, ans, true);
                if(set.contains(root.right.val)){
                    root.right = null;
                }
            }
            if(!hasParent){
                ans.add(root);   
            }
        }
    }
    
    
public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        ArrayList<TreeNode> list = new ArrayList<>();
        HashSet<Integer> set = new HashSet<>();
        for(int d : to_delete) {
            set.add(d);
        }
        helper(root, set, list, false);
        return list;
    }
}
