/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        Set<Integer> set = new HashSet<>();
        for (int del : to_delete) {
            set.add(del);
        }
        List<TreeNode> res = new ArrayList<>();
        dfs(root, set, res);
        return res;
    }

    private void dfs(TreeNode root, Set<Integer> set, List<TreeNode> res) {
        if (root == null) {
            return;
        }
        if (set.contains(root.val)) {
            dfs(root.left, set, res);
            dfs(root.right, set, res);
        } else {
            res.add(root);
            dfs2(root.left, root, true, set, res);
            dfs2(root.right, root, false, set, res);
        }
    }

    private void dfs2(TreeNode node, TreeNode parent, boolean isLeft, Set<Integer> set, List<TreeNode> res) {
        if (node == null) {
            return;
        }
        if (set.contains(node.val)) {
            if (isLeft) {
                parent.left = null;
            } else {
                parent.right = null;
            }
            dfs(node.left, set, res);
            dfs(node.right, set, res);
        } else {
            dfs2(node.left, node, true, set, res);
            dfs2(node.right, node, false, set, res);
        }
    }
}
