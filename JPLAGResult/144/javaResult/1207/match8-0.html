/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> ans = new LinkedList<>();
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i < to_delete.length; i++) {
            set.add(to_delete[i]);
        }
        
        dfs(ans, set, root, null, 1, 0);
        return ans;
    }
    
    private void dfs(List<TreeNode> ans, Set<Integer> set, TreeNode node, TreeNode parent, int f, int added) {
        if (node == null) return;
        if (set.contains(node.val)) {
            if (parent != null) {
                if (f == 0) parent.left = null;
                else parent.right = null;
            }
            
            dfs(ans, set, node.left, node, 0, 0);
            dfs(ans, set, node.right, node, 1, 0);
        }
        
        else {
            if (added == 0) ans.add(node);
            dfs(ans, set, node.left, node, 0, 1);
            dfs(ans, set, node.right, node, 1, 1);
        }
    }
}
