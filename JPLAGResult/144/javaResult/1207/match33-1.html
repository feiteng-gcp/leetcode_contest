/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> result = new ArrayList<TreeNode>();
        if(root == null) return result;
        boolean [] chk = new boolean [1001];
        for(int i = 0; i < to_delete.length; i++) chk[to_delete[i]] = true;
        
delNodesRecursive(root, null, result, chk, false);
        return result;
    }
    
    public void delNodesRecursive(TreeNode tmp, TreeNode father, List<TreeNode> result, boolean[] chk, boolean leftFlg){
        if(father == null && !chk[tmp.val]) result.add(tmp);
        if(chk[tmp.val]){
            
if(father != null){
                if(leftFlg) father.left = null;
                else father.right = null;
            }
            if(tmp.left != null && !chk[tmp.left.val]) result.add(tmp.left);
            if(tmp.right != null && !chk[tmp.right.val]) result.add(tmp.right);
        }
        father = tmp;
        if(tmp.left != null) delNodesRecursive(tmp.left, tmp, result, chk, true);
        if(tmp.right != null) delNodesRecursive(tmp.right, tmp, result, chk, false);
    } 
}
