/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        if (root == null) return new ArrayList<>();
        List<TreeNode> res = new ArrayList<>();
        
Set<Integer> set = new HashSet<>();
        for (int i : to_delete) {
            set.add(i);
        }
        helper(root, res, null, set);
        return res;
    }
    
    private void helper(TreeNode root, List<TreeNode> forest, TreeNode parent, Set<Integer> set) {
        if (root == null) return;
        helper(root.left, forest, root, set);
        helper(root.right, forest, root, set);
        
if (set.contains(root.val)) {
            if (parent != null) {
                if (parent.left == root) {
                    parent.left = null;
                } else {
                    parent.right = null;
                }
            }
        } else if (parent == null || set.contains(parent.val)) {
            forest.add(root);
        }
    }
}
