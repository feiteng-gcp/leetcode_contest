/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> res = new ArrayList<>();
        if (root == null) return res; 
        Set<Integer> targets = new HashSet<>();
        for (int x : to_delete) {
            targets.add(x);
        }
        Set<TreeNode> resSet = new HashSet<TreeNode>();
        dfs(resSet, targets, root, null, true);
        res.addAll(resSet); 
        return res;
    }
    
    private void dfs(Set<TreeNode> res, Set<Integer> targets, TreeNode node, TreeNode father, boolean isLeft) {
        if (node == null) {
            return;
        }
        if (father == null && !targets.contains(node.val)) {
            res.add(node);
        }
        if (targets.isEmpty()) {
            return;
        }
        TreeNode nextfather;
        if (targets.contains(node.val)) {
            targets.remove(node.val);
            nextfather = null; 
            if (father != null) {
                if (isLeft) father.left = null;
                else father.right = null;
            }
        } else {
            nextfather = node;
        }
        
        dfs(res, targets, node.left, nextfather, true);
        dfs(res, targets, node.right, nextfather, false);
    }
}
