/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> ret = new ArrayList<>();
        Set<Integer> s = new HashSet<>();
        for(int i : to_delete){
            s.add(i);
        }
        helper(root, s, ret, null);
        return ret;
    }
    
    private TreeNode helper(TreeNode n, Set<Integer> s, List<TreeNode> ret, TreeNode par){
        if(n == null) return n;
        if(s.size() == 0){
            if(par == null) ret.add(n);
            return n;
        }
        TreeNode cur = n;
        if(s.contains(n.val)){
            s.remove(n.val);
            cur = null;
        }
        if(par == null && cur != null){
            ret.add(cur);
        }
        TreeNode l = n.left;
        TreeNode r = n.right;
        n.left = helper(l, s, ret, cur);
        n.right = helper(r, s, ret, cur);
        return cur;
    }
    
}
