/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    List<TreeNode> r = new ArrayList<>();
    public List<TreeNode> delNodes(TreeNode root, int[] t) {
        
        Set<Integer> st = new HashSet<>();
        
        for(int dd:t){
            st.add(dd);
        }
        if(!st.contains(root.val)){
            r.add(root);
        }
        
        d(root,st,null);
        return r;
        
    }
    
    public void d(TreeNode rr,Set<Integer> st,TreeNode par){
        if(rr==null) return;
        
        if(st.contains(rr.val)){
            if(par!=null){
                if(par.left==rr){
                    par.left=null;
                }else{
                    par.right=null;
                }
            }
            if(rr.left!=null){
                if(!st.contains(rr.left.val)){
            r.add(rr.left);
                }
                d(rr.left,st,rr);
                }
            
            if(rr.right!=null){
            if(!st.contains(rr.right.val)){
            r.add(rr.right);
                }
                d(rr.right,st,rr);
                }
            
            
            
        }else{
            if(rr.left!=null){
          //  r.add(rr.left);
                d(rr.left,st,rr);
                }
            
            if(rr.right!=null){
           // r.add(rr.right);
                d(rr.right,st,rr);
                }
            
            
            
        }
        
    }
}
