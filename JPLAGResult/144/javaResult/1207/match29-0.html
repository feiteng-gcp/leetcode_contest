/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    List<TreeNode> ret = new ArrayList<>();
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        if (to_delete == null || to_delete.length == 0) {
            return ret;
        }
        Set<Integer> set = new HashSet<>();
        for (int d: to_delete) {
            set.add(d);
        }
        helper(root, null, set);
        return ret;
    }
    
    public void helper(TreeNode n, TreeNode parent, Set<Integer> set) {
        if (n == null) {
            return;
        }
        if (set.contains(n.val)) {
            // disconnect n from parent
            if (parent != null) {
                if (parent.left == n) {
                    parent.left = null;
                } else {
                    parent.right = null;
                }
            }
            helper(n.left, null, set);
            helper(n.right, null, set);
        } else {
            if (parent == null) {
                ret.add(n);
            } 
            helper(n.left, n, set);
            helper(n.right, n, set);
        }
    }
    
}
