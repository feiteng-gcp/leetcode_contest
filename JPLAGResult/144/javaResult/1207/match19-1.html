/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    Set<Integer> set = new HashSet();
    List<TreeNode> res = new ArrayList();
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        for(int d: to_delete)
            set.add(d);
        dfs(root, true);
        return res;
    }
    
    private boolean dfs(TreeNode t, boolean isRoot) {
        if(t == null) return true;
        if(set.contains(t.val)) {
            dfs(t.left, true);
            dfs(t.right, true);
            return true;
        }
        if(isRoot) {
            res.add(t);
        }
        if(dfs(t.left, false))
            t.left = null;
        if(dfs(t.right, false))
            t.right = null;
        return false;
    }
}
