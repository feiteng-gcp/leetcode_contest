/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        Set<Integer> s = new HashSet<Integer>();
        for (int val : to_delete) {
            s.add(val);
        }
        List<TreeNode> res = new ArrayList<TreeNode>();
        if (root == null) {
            return res;
        }
        helper(root, res, s, null, false, true);
        // if (!s.contains(root.val)) {
        //     res.add(root);
        // }
        return res;
    }
    
    private void helper(TreeNode root, 
List<TreeNode> res, Set<Integer> s, TreeNode parent, boolean isLeft, boolean addToResult) {
        if (root == null) {
            return;
        }
        
        boolean needToDelete = s.contains(root.val);
        if (!needToDelete && addToResult) {
            res.add(root);
        }
        if (needToDelete && parent != null) 
{
            // System.out.println("to delete");
            if (isLeft) {
                // System.out.println("to delete left");
                parent.left = null;
            } else {
                // System.out.println("to delete right");
                parent.right = null;
            }
        }
        
        helper(root.left, res, s, root, true, needToDelete);
        helper(root.right, res, s, root, false, needToDelete);
    }
}
