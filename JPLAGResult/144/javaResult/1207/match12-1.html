/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> result = new ArrayList<>();
        
Set<Integer> delete = new HashSet<>();
        for (int d : to_delete) delete.add(d);
        delNodes(root, delete, result, null);
        return result;
    }
    
    private TreeNode delNodes(TreeNode root, Set<Integer> delete, List<TreeNode> result, TreeNode parent) {
        if (root == null) return null;
        if (delete.contains(root.val)) {
            delNodes(root.left, delete, result, null);
            delNodes(root.right, delete, result, null);
            return null;
        } else {
            if (parent == null) {
                result.add(root);
            }
            root.left = delNodes(root.left, delete, result, root);
            root.right = delNodes(root.right, delete, result, root);
        }
        return root;
    }
}
