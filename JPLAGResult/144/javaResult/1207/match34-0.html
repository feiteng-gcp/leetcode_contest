/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        LinkedList<TreeNode> res = new LinkedList<>();
        if (root == null) {
            return res;
        }
        HashSet<Integer> set = new HashSet<>();
        for (int i = 0;i < to_delete.length;i++) {
            set.add(to_delete[i]);
        }
        delete(root, res, set, true);
        return res;
    }
    public boolean delete(TreeNode root, LinkedList<TreeNode> res, HashSet<Integer> set, boolean removed) {
        if (set.contains(root.val)) {
            if (root.left != null) {
                delete(root.left, res, set, true);
            }
            if (root.right != null) {
                delete(root.right, res, set, true);
            }
            return true;
        } else {
            if (root.left != null) {
                boolean deleteLeft = delete(root.left, res, set, false);
                if (deleteLeft) {
                    root.left = null;
                }
            }
            if (root.right != null) {
                boolean deleteRight = delete(root.right, res, set, false);
                if (deleteRight) {
                    root.right = null;
                }
            }
            if (removed) {
                res.add(root);
            }
            return false;
        }
    }
}
