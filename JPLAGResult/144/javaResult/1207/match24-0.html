/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        Set<Integer> td = new HashSet<>();
        for (int d: to_delete) {
            td.add(d);
        }
        
        List<TreeNode> res = new ArrayList<>();
        dfs(root, res, td, true);
        return res;
    }
    
    private void dfs(TreeNode root, List<TreeNode> res, Set<Integer> del, boolean isRoot) {
        if (root == null) return;
        
        if (isRoot && !del.contains(root.val)) {
            res.add(root);
        }
        
        if (root.left != null) {
            dfs(root.left, res, del, del.contains(root.val));
            if (del.contains(root.left.val)) root.left = null;
        }
        if (root.right != null) {
            dfs(root.right, res, del, del.contains(root.val));
            if (del.contains(root.right.val)) root.right = null;
        }
    }
}
