/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] A) {
        Set<Integer> vals = new HashSet<>();
        for (int v : A) vals.add(v);
        
        
        List<TreeNode> res = new ArrayList<>();
        TreeNode r = solve(res, vals, root);
        if (r != null) {
            res.add(r);
        }
        return res;
    }
    
    private TreeNode solve(List<TreeNode> res, Set<Integer> vals, TreeNode root) {
        if (root == null) return null;
        TreeNode l = solve(res, vals, root.left);
        TreeNode r = solve(res, vals, root.right);
        if (vals.contains(root.val)) {
            if (l != null) {
                res.add(l);
            }
            if (r != null) {
                res.add(r);
            }
            return null;
        } else {
            root.left = l;
            root.right = r;
            return root;
        }
    }
}
