/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> list = new ArrayList<>();
        Set<Integer> toDelete = new HashSet<>();
        for(int i : to_delete) toDelete.add(i);
        if(!toDelete.contains(root.val)) list.add(root);
        help(root,toDelete,list);
        return list;
    }
    
    Boolean help(TreeNode treeNode, Set<Integer> toDelete, List<TreeNode> list) {
        if(treeNode == null) return true;
        if(toDelete.contains(treeNode.val)) {
            if(!help(treeNode.left,toDelete,list)) list.add(treeNode.left);
            if(!help(treeNode.right, toDelete,list)) list.add(treeNode.right);
            return true;
        }else {
            if(help(treeNode.left,toDelete,list)) treeNode.left = null;
            if(help(treeNode.right,toDelete,list)) treeNode.right = null;
            return false;
        }
    }
}
