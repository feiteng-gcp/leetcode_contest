/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        Set<Integer> toDelete = new HashSet<>();
        for(int val : to_delete) {
            toDelete.add(val);
        }
        List<TreeNode> output = new ArrayList<>();
        if(!toDelete.contains(root.val)) {
            output.add(root);
        }
        traverse(root, toDelete, output, toDelete.contains(root.val), null);
        return output;
    
}
    
    void traverse(TreeNode root, Set<Integer> toDelete, List<TreeNode> output, boolean deleteParent, TreeNode parent){
        if(root == null) {
            return;
        } else {
            if(deleteParent) {
                if(!toDelete.contains(root.val)) {
                    output.add(root);
                }
            }
        }
        if(toDelete.contains(root.val)) {
            TreeNode left = root.left;
            TreeNode right = root.right;
            if(parent != null) {
                if(parent.right == root) {
                    parent.right = null;
                } else if(parent.left == root) {
                    parent.left = null;
                }                        
            
}
            traverse(left, toDelete, output, true, root);
            traverse(right, toDelete, output, true, root);
        } else {
            traverse(root.left, toDelete, output, false, root);
            traverse(root.right, toDelete, output, false, root);
        }
    }
    
}
