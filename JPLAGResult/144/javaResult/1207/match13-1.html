/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> res = new ArrayList<>();
        boolean[] deletes = new boolean[1001];
        for (int k : to_delete) deletes[k] = true
;
        delNodes(root, true, res, deletes);
        return res;
    }
    private TreeNode delNodes(TreeNode nd, boolean isRoot, List<TreeNode> res, boolean[] deletes) {
        if (nd == null) return null;
        if (!deletes[nd.val]) 
{
            if (isRoot) res.add(nd);
            nd.left = delNodes(nd.left, false, res, deletes);
            nd.right = delNodes(nd.right, false, res, deletes);
            return nd;
        }
        else {
            delNodes(nd.left, true, res, deletes);
            delNodes(nd.right, true, res, deletes);
            return null;
        }
    }
}
