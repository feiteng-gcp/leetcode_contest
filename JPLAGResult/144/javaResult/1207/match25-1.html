/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
	public List<TreeNode> res = new ArrayList<>();

	public Set<Integer> set = new HashSet<>();

	public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
		for (int i : to_delete)
			set.add(i);
		if (!set.conta
ins(root.val))
			res.add(root);
		dfs(root);
		return res;
	}

	public void dfs(TreeNode node) {
		if (set.contains(node.val)) {
			if (node.left != null) {
				if (!set.contains(node.left.val))
					res.add(node.left);
				dfs(node.left);
			}
			if (node.right != null) {
				if (!set.contains(node.right.val))
					res.add(node.right);
				dfs(node.right);
			}

		} else {
			if (node.left != null) {
				dfs(node.left);
				if (set.contains(node.left.val))
					node.left = null;
			}
			if (node.right != null) {
				dfs(node.right);
				if (set.contains(node.right.val))
					node.right = null;
			}
		}
	}
}
