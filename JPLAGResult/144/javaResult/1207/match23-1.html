/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        Set<Integer> del = new HashSet<>();
        for (int i : to_delete) del.add(i)
;
        List<TreeNode> res = new ArrayList<>();
        delNodes(root, del, res, null);
        return res;
    }
    
    public void delNodes(TreeNode root, Set<Integer> del, List<TreeNode> res, TreeNode parent) {
        if (root == null) return;
        if (!del.contains(root.val)) {
            if (parent == null) res.add(root);
            parent = root;
        } else if (parent != null) {
            if (root == parent.left) parent.left = null;
            else parent.right = null;
            parent = null;
        }
        delNodes(root.left, del, res, parent);
        delNodes(root.right, del, res, parent);
    }
}
