/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> res = new ArrayList<>();
        Set<Integer> set=  new HashSet<>();
        for (int d : to_delete) set.add(d);
        helper(res, null, root, set);
        return res;
    }
    
    private void helper(
List<TreeNode> res, TreeNode par, TreeNode cur, Set<Integer> set) {
        if (cur == null) return;
        if (set.contains(cur.val)) {
            if (par != null) {
                if (par.left == cur) par.left = null;
                else par.right = null;
            }
            helper(res, null, cur.left, set);
            helper(res, null, cur.right, set);
            return;
        }
        if (par == null) res.add(cur);
        helper(res, cur, cur.left, set);
        helper(res, cur, cur.right, set);
    }
}
