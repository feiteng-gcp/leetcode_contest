/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> forest = new ArrayList<TreeNode>();
        List<Integer> toDeleteIndicesList = new ArrayList<Integer>();
        for (int index : to_delete)
            toDeleteIndicesList.add(index);
        if (!toDeleteIndicesList.contains(root.val))
            forest.add(root);
        List<TreeNode> toDeleteNodesList = new ArrayList<TreeNode>();
        Queue<TreeNode> queue = new LinkedList<TreeNode>();
        queue.offer(root);
        while (!queue.isEmpty()) {
            TreeNode node = queue.poll();
            int nodeVal = node.val;
            TreeNode left = node.left;
            TreeNode right = node.right;
            if (left != null) {
                queue.offer(left);
                int leftVal = left.val;
                if (toDeleteIndicesList.contains(leftVal)) {
                    toDeleteNodesList.add(left);
                    node.left = null;
                }
                if (toDeleteIndicesList.contains(nodeVal) && !toDeleteIndicesList.contains(leftVal))
                    forest.add(left);
            }
            if (right != null) {
                queue.offer(right);
                int rightVal = right.val;
                if (toDeleteIndicesList.contains(rightVal)) {
                    toDeleteNodesList.add(right);
                    node.right = null;
                }
                if (toDeleteIndicesList.contains(nodeVal) && !toDeleteIndicesList.contains(rightVal))
                    forest.add(right);
            }
        }
        return forest;
    }
}
