/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
	public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
		HashSet<Integer> toDelete = new HashSet<>();
		for (int i : to_delete) {
			toDelete.add(i);
		}

		ArrayList<TreeNode> res = new ArrayList<>();
		if (!toDelete.contains(root.val)) {
			res.add(root);
		}
		return dfs(root, toDelete, res, toDelete.contains(root.val));
	}


	private List<TreeNode> dfs(TreeNode root, HashSet<Integer> toDelete, ArrayList<TreeNode> res, boolean includeChilds) {
		if (includeChilds) {
			if (root.left != null && !toDelete.contains(root.left.val)) {
				res.add(root.left);
			}
			if (root.right != null && !toDelete.contains(root.right.val)) {
				res.add(root.right);
			}

		}

		if (root.left != null) {
			dfs(root.left, toDelete, res, toDelete.contains(root.left.val));
			if (toDelete.contains(root.left.val)) {
				root.left = null;
			}
		}
		if (root.right != null) {
			dfs(root.right, toDelete, res, toDelete.contains(root.right.val));
			if (toDelete.contains(root.right.val)) {
				root.right = null;
			}
		}

		return res;
	}
}
