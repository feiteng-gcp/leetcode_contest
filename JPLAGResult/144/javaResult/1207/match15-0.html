/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    List<TreeNode> ans = new ArrayList<>();
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        Set<Integer> dS = new HashSet<>();
        for (int to : to_delete)
            dS.add(to);
        helper(dS, ans, root, true);
        return ans;
    }
    private TreeNode helper(Set<Integer> dS, List<TreeNode> ans, TreeNode r, boolean isRoot) {
        if (r == null) return null;
        if (dS.contains(r.val)) {
            helper(dS, ans, r.left, true);
            helper(dS, ans, r.right, true);
            return null;
        } else {
            if (isRoot) ans.add(r);
            r.left = helper(dS, ans, r.left, false);
            r.right = helper(dS, ans, r.right, false);
            return r;
        }
    }
}
