/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        Queue<TreeNode> queue = new LinkedList<TreeNode>();
        List<TreeNode> list = new ArrayList();
        HashSet<Integer> hs=new HashSet<Integer>();
        for(int i = 0; i < to_delete.length; i++)
            hs.add(to_delete[i]);
        queue.offer(root);
        list.add(root);
        while(!queue.isEmpty() && !hs.isEmpty()){
            TreeNode cur = queue.poll();
            if(hs.contains(cur.val)){
                hs.remove(cur.val);
                if(list.contains(cur)) list.remove(cur);
                if(cur.left != null) list.add(cur.left);
                if(cur.right != null) list.add(cur.right);
            }
            if(cur.left != null){
                queue.offer(cur.left);
                if(hs.contains(cur.left.val)) cur.left = null;
            }
            if(cur.right != null){
                queue.offer(cur.right);
                if(hs.contains(cur.right.val)) cur.right = null;
            }
        }
        return list;
    }
}
