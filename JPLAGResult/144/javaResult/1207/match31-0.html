/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<TreeNode> result = new ArrayList<>();
        Set<Integer> set = new HashSet<>();
        for (int i : to_delete) {
            set.add(i);
        }
        cal(root, set, result, true);
        return result;
    }
    
    private boolean cal(TreeNode root, Set<Integer> set, List<TreeNode> result, boolean isRoot) {
        if (root == null) {
            return false;
        }
        if (set.contains(root.val)) {
            if (cal(root.left, set, result, true)) {
                root.left = null;
            }
            if (cal(root.right, set, result, true)) {
                root.right = null;
            }
            return true;
        } else {
            if (isRoot) {
                result.add(root);
            }
            if (cal(root.left, set, result, false)) {
                root.left = null;
            }
            if (cal(root.right, set, result, false)) {
                root.right = null;
            }
            return false;
        }
    }
}
