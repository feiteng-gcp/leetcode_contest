/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public List<TreeNode> delNodes(TreeNode root, int[] to_delete) {
        List<Integer> v = new ArrayList<>();
        for (int d : to_delete) v.add(d);
        List<TreeNode> res = new ArrayList<>();
        dfs(root, true, res, v);
        return res;
    }

    private TreeNode dfs(TreeNode curr, 
boolean parentIsErasable, List<TreeNode> res, List<Integer> list) {
        if (curr == null) return null;
        boolean currIsErasable = list.contains(curr.val);
        if (parentIsErasable && !currIsErasable) res.add(curr);
        curr.left = dfs(curr.left, currIsErasable, res, list);
        curr.right = dfs(curr.right, currIsErasable, res, list);
        return currIsErasable ? null : curr;
    }
}
