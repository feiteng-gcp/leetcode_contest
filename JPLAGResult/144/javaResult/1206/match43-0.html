class Solution {
    public int[] corpFlightBookings(int[][] bookings, int n) {
        int[] ret = new int[n];
        for (int[] booking : bookings) {
            int begin = booking[0], end = booking[1];
            int seats = booking[2];
            for (int i = begin; i <= end; i++)
                ret[i - 1] += seats;
        }
        return ret;
    }
}
