class Solution {
    public int[] corpFlightBookings(int[][] bb, int n) {
        int[] ans = new int[n];
        for(int i = 0; i < bb.length; i++) {
            int a = bb[i][0];
            int b = bb[i][1];
            int c = bb[i][2];
            for(int j = a - 1; j < b; j++)
                ans[j] += c;
        }
        return ans;
    }
}
