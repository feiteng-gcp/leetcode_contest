class Solution {
    public int[] corpFlightBookings(int[][] bookings, int n) {
        int[] ret = new int[n];
        for (int[] booking: bookings) {
            int i = booking[0];
            int j = booking[1];
            int k = booking[2];
            for (int t = i ; t <= j; t++) {
                ret[t-1] += k;
            }
        }
        return ret;
    }
}
