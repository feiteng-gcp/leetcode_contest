class Solution {
    public int[] corpFlightBookings(int[][] bookings, int n) {
        int[] res = new int[n];
        for (int[] booking : bookings) {
            int i = booking[0], j = booking[1], k = booking[2];
            for (int z = i - 1; z <= j - 1; z++) {
                res[z] += k;
            }
        }
        return res;
    }
}
