class Solution {
    public int[] corpFlightBookings(int[][] arr, int n) {
        
        int[] ans=new int[n];
        
        for(int i=0;i<arr.length;i++) {
            
            int x=arr[i][0];
            int y=arr[i][1];
            int z=arr[i][2];
            
            for(int j=x;j<=y;j++) {
                ans[j-1]+=z;
            }
            
        }
        return ans;
    }
}
