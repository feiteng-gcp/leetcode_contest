class Solution {
    public int[] corpFlightBookings(int[][] bookings, int n) {
        int[] seats = new int[n];

        int[] sums = new int[n+2];
        for (int[] booking : bookings) {
            sums[booking[0]] += booking[2];
            sums[booking[1] + 1] -= booking[2];
        }

        int cumSum = 0;
        for (int i = 0; i < n; i++) {
            cumSum += sums[i+1];
            seats[i] = cumSum;
        }
        return seats;
    }
}
