class Solution {
    public int[] corpFlightBookings(int[][] bookings, int n) {
        int[] ans = new int[n];
        for (int[] booking : bookings) {
            int from = booking[0] - 1;
            int to = booking[1] - 1;
            int count = booking[2];
            for (int i = from; i <= to; i++) {
                ans[i] += count;
            }
        }
        return ans;
    }
}
