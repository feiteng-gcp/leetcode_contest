class Solution {
    public int[] corpFlightBookings(int[][] bookings, int n) {
        int[] res = new int[n];
        int[] oper = new int[n + 1];
        
        for(int[] booking : bookings){
            oper[booking[0] - 1] += booking[2];
            oper[booking[1]] -= booking[2];
        }
        
        int cnt = 0;
        for(int i = 0; i < n; i++){
            cnt += oper[i];
            res[i] = cnt;
        }
        
        return res;
    }
}
