class Solution {
    public int[] corpFlightBookings(int[][] bookings, int n) {
        int[] table = new int[n + 2];
        for (int i = 0; i < bookings.length; i++) {
            table[bookings[i][0]] += bookings[i][2];
            table[bookings[i][1] + 1] -= bookings[i][2];
        }
        int[] res = new int[n];
        res[0] = table[1];
        for (int i = 1; i < n; i++) {
            res[i] = res[i - 1] + table[i + 1];
        }
        return res;
    }
}
