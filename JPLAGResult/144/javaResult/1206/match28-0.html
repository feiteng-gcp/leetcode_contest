class Solution {
    public int[] corpFlightBookings(int[][] bookings, int n) {
        int[] res = new int[n];
        
        for (int[] booking : bookings) {
            int start = booking[0];
            int end = booking[1];
            int num = booking[2];
            for (int i = start; i <= end; i++) {
                res[i - 1] += num;
            }
        }
        return res;
    }
}
