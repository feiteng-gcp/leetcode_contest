class Solution {
    public int[] corpFlightBookings(int[][] bookings, int n) {
        int[] map = new int[n];
        int[] res = new int[n];
        for(int[] booking : bookings) {
            int start = booking[0] - 1, end = booking[1];
            int num = booking[2];
            map[start] += num;
            if(end < n) map[end] -= num;
        }
        
        int count = 0;
        for(int i = 0; i < n; i++) {
            // System.out.print(map[i] + " ");
            count += map[i];
            res[i] = count;
        }
        return res;
    }
}
