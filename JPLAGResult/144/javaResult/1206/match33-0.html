class Solution {
    public int[] corpFlightBookings(int[][] bb, int n) {
        
        int res[] = new int[n];
        
        for(int[] s:bb){
            int st = s[0];
            int ed = s[1];
            int k = s[2];
            
            res[st-1]+=k;
            if(ed!=n){
            res[ed]-=k;
            }
        }
        
        for(int i=1;i<n;++i){
            res[i]+=res[i-1];
        }
        return res;
    }
}
