class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int[] result = new int[seq.length()];
        int list1num = 0;
        int list2num = 0;
        for (int i = 0; i < seq.length(); i++) {
            if (seq.charAt(i) == '(') {
                if (list1num < list2num) {
                    result[i] = 0;
                    list1num++;
                } else {
                    result[i] = 1;
                    list2num++;
                }
            }else {
                if (list1num < list2num) {
                    result[i] = 1;
                    list2num--;
                } else {
                    result[i] = 0;
                    list1num--;
                }
            }
        }
        return result;
    }
}
