class Solution {
    // ()(())()
    /*
    1 0    a
    0 0    
    1 0    a
    1 1    ab
    1 0    a
    0 0    
    1 0    a
    0 0    0
    
    
    */
    public int[] maxDepthAfterSplit(String seq) {
        int N = seq.length();
        int depa = 0;
        int depb = 0;
        int[] ret = new int[N];
        Stack<Character> stack = new Stack<>();
        for (int i = 0 ; i < N; i++) {
            char c = seq.charAt(i);
            if (c == '(') {
                if (depa < depb || (depa == depb &&  (stack.isEmpty() || stack.peek() ==  'b'))) {
                    stack.push('a');
                    ret[i] = 0;
                    depa++;
                } else {
                    stack.push('b');
                    ret[i] = 1;
                    depb++;
                }
            } else {
                if (stack.peek() == 'a') {
                    stack.pop();
                    depa--;
                    ret[i] = 0;
                } else {
                    stack.pop();
                    depb--;
                    ret[i] = 1;
                }
            }
            
        }
        return ret;
    }
}
