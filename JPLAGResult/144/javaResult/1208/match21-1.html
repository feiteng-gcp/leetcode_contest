class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        
char[] cs = seq.toCharArray();
        int maxDepth = 0;
        int depth = 0;
        for (char c : cs) {
            if (c == '(') {
                ++depth;
                maxDepth = Math.max(maxDepth, depth);
            
} else {
                --depth;
            }
        }
        int halfDepth = maxDepth + 1 >> 1;
        depth = 0;
        int[] res = new int[cs.length];
        
for (int i = 0; i < cs.length; ++i) {
            if (cs[i] == '(') {
                ++depth;
                if (depth <= halfDepth) {
                    res[i] = 0;
                } else {
                    res[i] = 1;
                }
            } else {
                if (depth <= halfDepth) {
                    res[i] = 0;
                } else 
{
                    res[i] = 1;
                }
                --depth;
            }
        }
        return res;
    }

}
