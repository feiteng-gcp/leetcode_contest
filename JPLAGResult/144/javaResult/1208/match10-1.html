class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int[] ans = new int[seq.length()];
        for (int i = 0, cur = 0; i < seq.length(); i++) 
{
            char c = seq.charAt(i);
            if (c == '(') {
                cur ++;
                if (cur == 1) {
                    ans[i] = 0;
                } else {
                    ans[i] = 1;
                }
            } else {
                cur --;
                if (cur == 0) {
                    ans[i] = 0;
                } else {
                    ans[i] = 1;
                }
            }
        }
        return ans;
    }
}
