class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int[] res = new int[seq.length()];
        Arrays.fill(res, 0);
        Stack<Character> st1 = new Stack<>();
        Stack<Character> st2 = new Stack<>();
        for (int i = 0;i < seq.length();i++) {
            if (seq.charAt(i) == '(') 
{
                if (st1.size() > st2.size()) {
                    st2.push('(');
                    res[i] = 1;
                } else {
                    st1.push('(');
                    res[i] = 0;
                }
            } else {
                if (st1.size() > st2.size()) 
{
                    st1.pop();
                    res[i] = 0;
                } else {
                    st2.pop();
                    res[i] = 1;
                }
            }
        }
        return res;
    }
}
