class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        char[] cha = seq.toCharArray();
        int[] res = new int[cha.length];
        Stack<Integer> st = new Stack<>();
        for(int i = 0; i < cha.length; i++){
            if(cha[i] == '(')
                st.push(i);
            else{
                int idx = st.pop();
                if(st.isEmpty())
{
                    res[i] = 0;
                    res[idx] = 0;
                }else{
                    res[i] = 1;
                    res[idx] = 1;
                }
            }
        }
        return res;
    }
}
