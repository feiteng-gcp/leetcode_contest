class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int curDep = 0, dep = 0;
        for (char c : seq.toCharArray()) {
            if (c == '(') ++ curDep;
            else --curDep;
            dep = Math.max(dep, curDep);
        }
        dep = (dep + 1) / 2;
        int curDep1 = 0, curDep2 = 0;
        int[] res = new int[seq.length()];
        for (int i = 0; i < res.length; ++i) {
            char c = seq.charAt(i);
            if (c == '(') 
{
                if (curDep1 < dep) {
                    res[i] = 0;
                    ++curDep1;
                } else {
                    res[i] = 1;
                    ++curDep2;
                }
            } else {
                if (curDep1 > 0) {
                    res[i] = 0;
                    --curDep1;
                } else {
                    res[i] = 1;
                    --curDep2;
                }
            }
        }
        return res;
    }
}
