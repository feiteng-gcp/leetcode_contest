class Solution {
	public int[] maxDepthAfterSplit(String seq) {
		int[] res = new int[seq.length()];

		// { start, end, depth }
		ArrayList<int[]> vpss = new ArrayList<>();

		int count = 0;
		int curDepth = 0;
		int maxDepth = 0;
		int start = 0;
		for (int i = 0; i < seq.length(); i++) {
			if (seq.charAt(i) == '(') {
				count++;
				curDepth++;
				maxDepth = Math.max(maxDepth, curDepth);
				start = i;
			}
			if (seq.charAt(i) == ')') {
				count--;
				curDepth--;
			}
			if (count == 0) {
				vpss.add(new int[] { start, i, maxDepth });
				maxDepth = 0;
			}
		}

		maxDepth = Integer.MIN_VALUE;
		for (int[] vps : vpss) {
			maxDepth = Math.max(maxDepth, vps[2]);
		}
		maxDepth /= 2;

		int countA = 0
;
		for (int i = 0; i < seq.length(); i++) {
			if (seq.charAt(i) == '(') {
				if (countA < maxDepth) {
					countA++;
					res[i] = 0;
				} else {
					res[i] = 1;
				}
			}
			if (seq.charAt(i) == ')') {

				if (countA > 0) {
					countA--;
					res[i] = 0;
				} else {
					res[i] = 1;
				}
			}
		}

		return res;
	}
}
