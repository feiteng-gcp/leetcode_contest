class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int count = 0;
        int maxDepth = 0;
        for (int i = 0; i < seq.length(); i++) {
            if (seq.charAt(i) == '(') {
                count++;
                maxDepth = Math.max(maxDepth, count);
            } else {
                count--;
            }
        }
        int mid = maxDepth/2;
        // System.out.println("maxDepth: " + maxDepth + "  mid: " + mid);
        int[] res = new int[seq.length()];
        count = 0;
        
for (int i = 0; i < seq.length(); i++) {
            if (seq.charAt(i) == '(') {
                count++;
                if (count <= mid) {
                    res[i] = 0;
                } else {
                    res[i] = 1;
                }
            } else {
                if (count <= mid) {
                    res[i] = 0;
                } else {
                    res[i] = 1;
                }
                count--;
            }
        }
        return res;
    }
}
