class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int [] ans = new int[seq.length()];
        char[] array = seq.toCharArray();
        int depth = 0, maxDepth = 0;
        for(char c : array) {
            if(c=='(') {
                depth++;
                if(depth > maxDepth) {
                    maxDepth = depth;
                }
            } else {
                depth--;
            }
        }
        int target = maxDepth/2;
        depth = 0;
        for(int i=0; i<ans.length; i++) {
            if(array[i]=='(') {
                depth++;
                if(depth > target) {
                    ans[i] = 1;
                }
            } else {
                if(depth > target) {
                    ans[i] = 1;
                }
                depth--;
            }
        }
        return ans;
    }
}
