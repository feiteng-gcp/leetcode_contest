class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int[] result = new int[seq.length()];
        int left = 0;
        for (int i = 0; i < seq.length(); i++) {
            char c = seq.charAt(i);
            if (c == '(') 
{
                left++;
                if (left % 2 == 1) {
                    result[i] = 0;
                } else {
                    result[i] = 1;
                }
            } else {
                left--;
                if (left % 2 == 0) {
                    result[i] = 0;
                } else {
                    result[i] = 1;
                }
            }
        }
        return result;
    }
}
