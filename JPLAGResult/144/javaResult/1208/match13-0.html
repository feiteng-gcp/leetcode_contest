class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        // (((())))
        // 01010101
        
        int[] ret = new int[seq.length()];
        
        int a = 0, b = 0;
        for (int i = 0; i < seq.length(); i++) {
            if (seq.charAt(i) == '(') {
                if (a <= b) {
                    ret[i] = 0;
                    a++;
                } else {
                    ret[i] = 1;
                    b++;
                }
                
            }
            else {
                if (a <= b) {
                    ret[i] = 1;
                    b--;
                } else {
                    ret[i] = 0;
                    a--;
                }
            }
        }
        
        return ret;
    }
}
