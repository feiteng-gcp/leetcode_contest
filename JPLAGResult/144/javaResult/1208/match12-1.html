class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int n = seq.length();
        int size1 = 0;
        int size2 = 0;
        
        int[] res = new int[n];
        int loc = 0;
        
        for(char c : seq.toCharArray()){
            
if(c == '('){
                if(size1 > size2){
                    res[loc++] = 1;
                    size2++;
                }
                else{
                    res[loc++] = 0;
                    size1++;
                }
            }
            else{
                if(size1 > size2){
                    res[loc++] = 0;
                    size1--;
                }
                else{
                    res[loc++] = 1;
                    size2--;
                }
            }
        }
        
        return res;
    }
}
