class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int max = 0;
        int curr = 0;
        char[] arr = seq.toCharArray();        
        for(char c : arr)
        {
            if(c == '(') curr ++;
            else curr --;
            if(max < curr) max = curr;
        }
        int half = max / 2;
        int[] ans = new int[arr.length];
        for(int i = 0; i < ans.length; i ++)
        {
            if(arr[i] == '('){
                curr ++;                
                if(curr > half)
                    ans[i] = 1;
            }
            else {
                curr --;                
                if(curr >= half)
                    ans[i] = 1;
            }
        }
        return ans;
    }
}
