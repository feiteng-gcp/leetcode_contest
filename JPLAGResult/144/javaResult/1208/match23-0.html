class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int[] ret = new int[seq.length()];
        int diff = 0;
        for (int idx = 0; idx < seq.length(); idx++) {
            char ch = seq.charAt(idx);
            if (ch == '(') {
                diff++;
                if (0 == (diff & 0x1)) {
                    ret[idx] = 1;
                }
            } else {
                if (0 == (diff & 0x1)) {
                    ret[idx] = 1;
                }
                diff--;
            }
        }
        return ret;
    }
}
