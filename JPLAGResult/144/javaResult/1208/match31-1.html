class Solution {
	
    public int[] maxDepthAfterSplit(String seq) {
    	int cnt = 0;
    	int i , length = seq.length();
    	int[] ans = new int[length];
    	for (i = 0;i < length;i ++) {
    		if (seq.charAt(i) == '(') {
    			cnt ++;
    		} else {
    			cnt --;    			
    		}
    		if (seq.ch
arAt(i) == '(') {
	    		if (cnt % 2 == 0) {
	    			ans[i] = 1;
	    		} else {
	    			ans[i] = 0;    			
	    		}
    		} else {
    			if (cnt % 2 == 0) {
    				ans[i] = 0;
    			} else {
    				ans[i] = 1;
    			}
    		}
    	}
    	return ans;
    }
	
}
