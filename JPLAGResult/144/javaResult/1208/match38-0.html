class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int len = seq.length();
        int[] ans = new int[len];
        Stack<Integer> positions = new Stack<>();
        Stack<Boolean> isA = new Stack<>();
        int depth1 = 0;
        int depth2 = 0;
        for (int i = 0; i < len; i++) {
            char ch = seq.charAt(i);
            if (ch == '(') {
                positions.push(i);
                if (isA.isEmpty()) {
                    isA.push(true);
                } else {
                    isA.push(!isA.peek());
                }
            }
            if (ch == ')') {
                int pos = positions.pop();
                boolean status = isA.pop();
                if (status) {
                    ans[pos] = 0;
                    ans[i] = 0;
                } else {
                    ans[pos] = 1;
                    ans[i] = 1;
                }
            }
        }
        return ans;
    }
}
