class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        if (seq == null) {
            return new int[0];
        }
        
int len = seq.length();
        int[] result = new int[len];
        int a = 0;
        int b = 0;
        for (int i = 0; i < len; i++) {
            char c = seq.charAt(i);
            if (c == '(') {
                if (a <= b) {
                    a++;
                    result[i] = 0;
                
} else {
                    b++;
                    result[i] = 1;
                }
            } else {
                if (a > b) {
                    a--;
                    result[i] = 0;
                
} else {
                    b--;
                    result[i] = 1;
                }
            }
        }
        return result;
    }
}
