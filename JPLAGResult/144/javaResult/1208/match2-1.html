class Solution {
    public int[] maxDepthAfterSplit(String seq) {
        int[] res = new int[seq.length()];
        int max = 0;
        int now = 0;
        char[] cs = seq.toCharArray();
        for (char c : cs) {
            if (c == '(') {
                now++;
                max = Math.max(max, now);
            }else {
                now--;
            }
        }
        max /= 2;
        for (int i=0;i<cs.length;i++) {
            if (cs[i] == '(') 
{
                now++;
                if (now <= max) {
                    res[i] = 0;
                }else {
                    res[i] = 1;
                }
            }else {
                now--;
                if (now < max) {
                    res[i] = 0;
                }else {
                    res[i] = 1;
                }
            }
            
        }
        return res;
    }
}
