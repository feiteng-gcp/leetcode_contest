class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> cnt(n+2,0);
        for(auto p:bookings)
        {
            cnt[p[0]]+=p[2];
            cnt[p[1]+1]-=p[2];
        }
        for(int i=1;i<=n;i++)
            cnt[i]+=cnt[i-1];
        vector<int> ans;
        for(int i=1;i<=n;i++)
            ans.push_back(cnt[i]);
        return ans;
    }
};
