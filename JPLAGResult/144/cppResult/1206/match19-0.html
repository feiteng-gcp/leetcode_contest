class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>> &bookings, int n) {
        vector<int> a(n);
        int len = bookings.size();
        for (int i = 0; i < len; i++) {
            int l = bookings[i][0];
            int r = bookings[i][1];
            int k = bookings[i][2];
            a[l - 1] += k;
            if(r<n)
            a[r] -= k;
        }
        for(auto s:a){
            cout<<s<<" ";
        }
        cout<<endl;
        len = a.size();
        for (int i = 1; i < len; i++) {
            a[i] += a[i - 1];
        }
        return a;
    }
};
