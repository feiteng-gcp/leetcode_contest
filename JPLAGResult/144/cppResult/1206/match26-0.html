class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> ret;
        ret.resize(n, 0);
        vector<pair<int, int>> v;
        for(int i=0;i<bookings.size();i++){
            pair<int, int> p =make_pair(bookings[i][0]-1, bookings[i][2]);
            v.push_back(p);
            p=make_pair(bookings[i][1], -bookings[i][2]);
            v.push_back(p);
        }
        sort(v.begin(), v.end());
        int cur=0;
        int idx=0;
        for(int i=0;i<n;i++){
            while(idx<v.size()&&v[idx].first==i){
                cur+=v[idx].second;
                idx++;
            }
            ret[i]=cur;
        }
        return ret;
    }
};
