class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& data, int n) {
        vector<int> res(n);
        for (int i = 0; i < data.size(); ++i) {
            int st = data[i][0] - 1, en = data[i][1] - 1;

            res[st] += data[i][2];
            if (en+1 < n)
                res[en+1] -= data[i][2];
        }
        
        for (int i = 1; i < n; ++i) {
            res[i] += res[i-1];
        }
        return res;
    }
};
