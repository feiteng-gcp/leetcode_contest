class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> record(n+1);
        for (auto &v : bookings) {
            record[v[0]-1] += v[2];
            record[v[1]] -= v[2];
        }
        
        vector<int> seats(n);
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            cnt += record[i];
            seats[i] = cnt;
        }
        return seats;
    }
};
