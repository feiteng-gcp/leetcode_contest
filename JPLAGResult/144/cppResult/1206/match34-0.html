class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
		vector<int> c(n, 0);	
		for (auto v : bookings) {
			int l = v[0] - 1;
			int r = v[1] - 1;
			int a = v[2];
			c[l] += a;
			if (r + 1 < n)
				c[r + 1] -= a;
		}
		for (int i = 1; i < n; ++i)
			c[i] += c[i - 1];
		return c;
    }
};
