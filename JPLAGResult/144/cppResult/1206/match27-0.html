class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> res;
        vector<int> dynamic(n+1);
        for(int i=0; i<bookings.size(); i++) {
            dynamic[bookings[i][0]-1] += bookings[i][2];
            dynamic[bookings[i][1]]-=bookings[i][2];
        }
        int cur = dynamic[0];
        for(int i=1; i<dynamic.size(); i++) {
            res.push_back(cur);
            cur+=dynamic[i];
            }
        return res;
    }
};
