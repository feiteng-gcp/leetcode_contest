class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> s(n);
        for (auto &x : bookings) {
            s[x[0] - 1] += x[2];
            if (x[1] < n)
                s[x[1]] -= x[2];
        }
        for (int i = 1; i < n; ++i)
            s[i] += s[i - 1];
        return s;
    }
};
