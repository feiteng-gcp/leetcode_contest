class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> dp(n+2, 0);
        for(auto book: bookings) {
            dp[book[0]]+=book[2];
            dp[book[1]+1]-=book[2];
        }
        vector<int> res;
        int ans = 0;
        for(int i=1;i<=n;i++) {
            ans+=dp[i];
            res.push_back(ans);
        }
        return res;
    }
};
