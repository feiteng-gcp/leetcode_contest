class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& b, int n) {
        vector<int> ans(n,0);
        vector<int> v(n+1,0);
        for(int i = 0;i<b.size();i++){
            v[b[i][0]-1]+=b[i][2];
            v[b[i][1]]-=b[i][2];
        }
        int cnt = 0;
        for(int i = 0;i<n;i++){
            cnt+=v[i];
            ans[i]=cnt;
        }
        return ans;
    }
};
