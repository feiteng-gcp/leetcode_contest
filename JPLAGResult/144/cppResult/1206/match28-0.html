class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int>vec;
        for (int i=0;i<n;++i) vec.push_back(0);
        for (int i=0;i<(int)bookings.size();++i){
            int l=bookings[i][0],r=bookings[i][1],num=bookings[i][2];
            l-=1;
            vec[l]+=num;
            if (r<n) vec[r]-=num;
        }
        for (int i=1;i<n;++i) vec[i]+=vec[i-1];
        return vec;
    }
};
