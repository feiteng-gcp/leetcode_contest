class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> ans;
        for (int i=0; i<n; i++) ans.push_back(0);
        
        int m=bookings.size();
        for (int i=0; i<m; i++){
            int s=bookings[i][0], e=bookings[i][1], k=bookings[i][2];
            ans[s-1]+=k; 
            if (e<n) ans[e]-=k;
        }
        
        for (int i=1; i<n; i++) ans[i]=ans[i-1]+ans[i];
        
        return ans;
    }
};
