class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> ans(n);
        for (auto &e : bookings)
        {
            int i = e[0], j = e[1], k = e[2];

            ans[i - 1] += k;
            if (j < n) ans[j] -= k;
        }
        for (int i = 1; i < n; ++i) ans[i] += ans[i - 1];
        return ans;
    }
};
