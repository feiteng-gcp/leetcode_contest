class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> scan(n+2, 0);
        for(auto& v: bookings)
        {
            scan[v[0]] += v[2];
            scan[v[1]+1] -= v[2];
        }
        
        vector<int> res;
        int pre = 0;
        
        for(int i = 0; i < n; i ++)
        {
            pre += (scan[i+1]);
            res.push_back(pre);
        }
        
        return res;
    }
};
