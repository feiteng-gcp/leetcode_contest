class Solution {
public:
    bool check(string& s,int x){
        int cnt1=0,cnt2=0;
        for(int i=0;i<s.size();i++){
            if(s[i]=='('){
                if(cnt1<x)cnt1++;
                else cnt2++;
                if(cnt2>x)return false;
            }else{
                if(cnt1>0)cnt1--;
                else cnt2--;
            }
        }
        return true;
    }
    vector<int> maxDepthAfterSplit(string s) {
        int ret=s.size();
        int l=0,r=ret;
        while(l<=r){
            int mid=(l+r)/2;
            if(check(s,mid)){
                ret=mid;
                r=mid-1;
            }else{
                l=mid+1;
            }
        }
        vector<int>ans;
        int cnt1=0,cnt2=0;

        for(int i=0;i<s.size();i++){
            if(s[i]=='('){
                if(cnt1<ret){cnt1++;ans.push_back(0);}
                else {cnt2++;ans.push_back(1);}
            }else{
                if(cnt1>0){cnt1--;ans.push_back(0);}
                else {cnt2--;ans.push_back(1);}
            }
        }
        return ans;
    }
};
