class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        int i,mx,sum,lmt,sum1,sum2;
        vector<int> ans;
        
        sum = 0;
        mx = 0;
        for (char c : seq) {
            if (c == '(')
                sum++;
            else
                sum--;
            
            mx = max(mx,sum);
        }
        
        lmt = (mx + 1) >> 1;
        sum1 = 0;
        sum2 = 0;
        for (char c : seq) {
            if (c == '(') {
                if (sum1 < lmt) {
                    sum1++;
                    ans.push_back(0);
                }
                else {
                    sum2++;
                    ans.push_back(1);
                }
            }
            else {
                if (sum2 > 0) {
                    sum2--;
                    ans.push_back(1);
                }
                else {
                    sum1--;
                    ans.push_back(0);
                }
            }
        }
        
        return ans;
    }
};
