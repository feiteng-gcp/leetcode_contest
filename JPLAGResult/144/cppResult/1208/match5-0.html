class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        
        int N=seq.size();
        
        
        
        int ao=0;
        int bo=0;
        vector<int> ret;
        for(int i=0;i<N;i++) {
            if (seq[i]=='(') {
                if (ao<bo) {
                    ret.push_back(0);
                    ao++;
                } else {
                    ret.push_back(1);
                    bo++;
                }
            } else {
                if (ao > bo) {
                    ret.push_back(0);
                    ao--;
                } else {
                    ret.push_back(1);
                    bo--;
                }
            }
        }
        return ret;
        
    }
};
