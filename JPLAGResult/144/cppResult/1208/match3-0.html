class Solution {
public:
	vector<int> maxDepthAfterSplit(string seq) {
		vector<int> ret;
		int a = 0, b = 0;

		for (char c : seq) {
			if(c == '('){
				if (a <= b) {
					ret.push_back(0);
					a++;
				}
				else {
					ret.push_back(1);
					b++;
				}
			}
			else {
				if (a <= b) {
					ret.push_back(1);
					b--;
				}
				else {
					ret.push_back(0);
					a--;
				}
			}
		}


		return ret;
	}
};
