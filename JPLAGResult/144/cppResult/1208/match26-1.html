class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        vector<int> left(seq.size()/2,0);
        vector<int> right(seq.size()/2,0);
        vector<int> ans(seq.size());
        int cnt0=0;
        int cnt1=0;
        int cnt2=0;
        int cnt3=0;

        for(int i = 0;i<seq.size();i++){
            if(seq[i]=='('){
                if(cnt0>cnt2){
                    ans[i]=1;
                    cnt2++;
                }else{
                    ans[i]=0;
                    cnt0++;
                }
            }else{
                if(cnt1>cnt3){
                    ans[i]=1;
                    cnt3++;
                }else{
                    ans[i]=0;
                    cnt1++;
                }
            }
        }
        return ans;
    }
};
