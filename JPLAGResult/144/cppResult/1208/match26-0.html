class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        int n = seq.size();
        int cura = 0, curb = 0;
        vector<int> ans(n, 0);
        for(int i = 0; i < n; i++) {
            if(seq[i] == '(') {
                if(cura > curb) {
                    ans[i] = 1;
                    curb++;
                }
                else{
                    ans[i] = 0;
                    cura++;
                }
            }
            else{
                if(cura > curb) {
                    ans[i] = 0;
                    cura--;
                }
                else {
                    ans[i] = 1;
                    curb--;
                }
            }
        }
        return ans;
    }
};
