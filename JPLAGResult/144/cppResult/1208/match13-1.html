class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        vector<int> ret;
        ret.resize(seq.length(), -1);

        int curA=0;
        int curB=0;
        for(int i=0;i<seq.length();i++){
            if(seq[i]=='('){
                if(curA>curB){
                    curB++;
                    ret[i]=0;
                }
                else{
                    curA++;
                    ret[i]=1;
                }

            }
            else{
                if(curA>curB){
                    curA--;
                    ret[i]=1;
                }
                else{
                    curB--;
                    ret[i]=0;
                }
            }
        }
        return ret;
    }
};
