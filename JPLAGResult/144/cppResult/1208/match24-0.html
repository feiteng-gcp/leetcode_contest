class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        vector<int> res;
        int c =  0;
        for ( int i = 0 ; i < seq.size() ; ++i ) {
            if ( seq[i] == '(') ++c;
            
            res.push_back(c%2);
            if ( seq[i] == ')') --c;
        }
        return res;
    }
};
