class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        int N = seq.length();

        vector<int> ans;
        int d = 0;
        for (int i = 0; i < N; ++i) {
            if (seq[i] == '(')
                ++d;
            ans.push_back(d % 2);
            if (seq[i] == ')')
                --d;
        }
        return ans;
    }
};
