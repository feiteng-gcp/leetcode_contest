class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        int n = seq.size();
        vector<int> ret;
        int da = 0, db = 0;
        for (auto c : seq)

        {
            if (c == '(')
            {
                if (da <= db)
                {
                    da ++;
                    ret.push_back(0);
                }
                else
                {
                    db ++;
                    ret.push_back(1);
                }
            }
            else
            {
                if (da >= db)
                {
                    da --;
                    ret.push_back(0);
                }
                else
                {
                    db --;
                    ret.push_back(1);
                }
            }
        }
        return ret;
    }
};
