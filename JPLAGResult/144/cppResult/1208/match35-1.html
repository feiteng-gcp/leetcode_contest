template <typename S, typename T>
void fmtosseq(S& s, const T& xs) {
  s << "[ ";
  for (const auto& x : xs) {
    fmtos(s, x);
    s << " ";
  }
  s << "]";
}

template <size_t K, size_t N>
struct FmtTup {
  template <typename S, typename... Ts>
  static void fmt(S& s, const tuple<Ts...>& xs) {
    fmtos(s, get<K>(xs));
    if (K + 1 < N) {
      fmtos(s, ", ");
    }
    FmtTup<K + 1, N>::fmt(s, xs);
  }
};

template <size_t N>
struct FmtTup<N, N> {
  template <typename S, typename... Ts>
  static void fmt(S& s, const tuple<Ts...>& xs) {}
};

template <typename S, typename T>
void fmtos(S& s, const T& x) {
  s << x;
}

template <typename S, typename Lhs, typename Rhs>
void fmtos(S& s, const pair<Lhs, Rhs>& x) {
  s << "(" << x.first << ", " << x.second << ")";
}

template <typename S, typename... Ts>
void fmtos(S& s, const tuple<Ts...>& x) {
  fmtos(s, "(");
  FmtTup<0, tuple_size<tuple<Ts...>>::value>::fmt(s, x);
  fmtos(s, ")");
}

template <typename S, typename T>
void fmtos(S& s, const vector<T>& xs) {
  fmtosseq(s, xs);
}

template <typename S, typename T>
void fmtos(S& s, const set<T>& xs) {

template <typename S, typename K, typename V>

template <typename S, typename K, typename V>
void fmtos(S& s, const unordered_map<K, V>& xs) {
  fmtosseq(s, xs);
}

template <typename S, typename T, typename... Ts>
void fmtos(S& s, const T& x, const Ts&... xs) {
  fmtos(s, x);
  fmtos(s, xs...);
}

template <typename... Ts>
string fmts(const Ts&... xs) {
  ostringstream iss;
  fmtos(iss, xs...);
  return iss.str();
}

class Solution {
 public:
  vector<int> maxDepthAfterSplit(string seq) {
    int n = seq.size();
    vector<int> ans(n, -1);

    int x = 0;
    int y = 0;

    for (int j = 0; j < n; j++) {
      if (seq[j] == '(') {
        if (x <= y) {
          ans[j] = 0;
          x++;
        } else {
          ans[j] = 1;
          y++;
        }
      } else {
        if (x > 0) {
          ans[j] = 0;
          x--;
        } else {
          ans[j] = 1;
          y--;
        }
      }
    }

    return ans;
  }
};
