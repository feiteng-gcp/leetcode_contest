class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {

        vector<int> ret(seq.size(), 0);
        int a = 0, b = 0;
        for (int i = 0;i < seq.size();i++) {
            if (seq[i] == '(') {
                if (a < b) {
                    a++;
                    ret[i] = 0;
                } else {
                    b++;
                    ret[i] = 1;

                }
            } else {
                if (a > b) {
                    a--;
                    ret[i] = 0;
                } else {
                    b--;
                    ret[i] = 1;
                }
            }
        }
        return ret;
    }
};
