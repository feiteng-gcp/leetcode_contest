class Solution {
public:
    vector<int> maxDepthAfterSplit(string s) {
        
        int n=s.length();
        vector<int>v;
        
        int fio=0,fie=0,sio=0,sie=0;
        stack<int>last;
        
        for(int i=0;i<n;i++){
            
            if(s[i]=='('){
                
                if(fio<=sio){
                    fio++;
                    last.push(0);
                    v.push_back(0);
                }else{
                    sio++;
                    last.push(1);
                    v.push_back(1);
                }
            }else if(s[i]==')'){
                
                if(last.top()==1){
                    v.push_back(1);
                    sio--;
                    last.pop();
                }else{
                    v.push_back(0);
                    fio--;
                    last.pop();
                }
            }
        }
        
        return v;
    }
};
