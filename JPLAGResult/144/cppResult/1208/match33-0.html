class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        int da=0;
        int db=0;
        vector<int>ans;
        for(int i = 0 ; i < seq.size(); i++){
            if(seq[i]=='('){
                if(da<=db){
                    ans.push_back(0);
                    da++;
                }
                else{
                    ans.push_back(1);
                    db++;
                }
            }
            else{
                if(da>=db){
                    ans.push_back(0);
                    da--;
                }
                else{
                    ans.push_back(1);
                    db--;
                }
            }
        }
        return ans;
    }
};
