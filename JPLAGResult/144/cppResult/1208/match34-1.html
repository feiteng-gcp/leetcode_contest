class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        int a = 0, b = 0;
        vector<int> res;
        

        for (char c : seq) {
            if (c == '(') {
                if (a <= b) {
                    a++;
                    res.push_back(0);
                } else {
                    b++;
                    res.push_back(1);
                }
            } else {
                if (a > b) {
                    a--;
                    res.push_back(0);
                } else {
                    b--;
                    res.push_back(1);
                }
            }
        }
        
        return res;
    }
};
