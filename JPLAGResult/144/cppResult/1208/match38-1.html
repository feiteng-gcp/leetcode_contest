class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        vector<int> ans;
        ans.assign(seq.length(), 0);
        int low = 0;
        int high = seq.length();
        int mid;
        while(high - low > 1) {
            mid = (high + low) >> 1;
            if(judge(mid, ans, seq)) {
                high = mid;
            } else {
                low = mid;
            }
        }
        judge(high, ans, seq);
        return ans;
    }

private:
    bool judge(int mid, vector<int> &ans, const string &str) {
        int tmp = 0;

        int len = ans.size();
        for(int i = 0; i < len; ++i) {
            if(str[i] == '(') {
                if(tmp < mid) {
                    ++tmp;
                    ans[i] = 1;
                } else {
                    ans[i] = 0;
                }
            } else {
                if(tmp > 0) {
                    --tmp;
                    ans[i] = 1;
                } else {
                    ans[i] = 0;
                }
            }
        }
        return getLen(ans, str) <= mid;
    }

    int getLen(const vector<int> &ans, const string &str) {
        int tmp = 0;
        int Max = 0;
        int len = str.length();
        for(int i = 0; i < len; ++i) {
            if(ans[i] == 1) {
                continue;
            }
            if(str[i] == '(') {
                ++tmp;
            } else {
                --tmp;
            }
            Max = max(Max, tmp);
        }
        return Max;
    }
};
