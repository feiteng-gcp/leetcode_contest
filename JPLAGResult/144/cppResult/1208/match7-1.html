#include<bits/stdc++.h>
using namespace std;
class Solution{
  public:
  vector<int> maxDepthAfterSplit(string seq){
    int N=seq.size(), i, j, k1, k2;
    vector<int> res;
    k1 = k2 = 0;

    for(i=0;i<N;i++){
      if(seq[i]=='('){
        if(k1 <= k2){
          res.push_back(0);
          k1++;
        } else {
          res.push_back(1);
          k2++;
        }
      } else if(seq[i]==')'){

        if(k1 >= k2){
          res.push_back(0);
          k1--;
        }
        else{
          res.push_back(1);
          k2--;
        }
      }
    }
    return res;
  }
}
;
