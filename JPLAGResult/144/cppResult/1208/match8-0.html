class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        vector<int> ans;
        
        int n=seq.length();
        for (int i=0; i<n; i++) ans.push_back(0);
        
        int cur=0;
        for (int i=0; i<n; i++){
            if (seq[i]=='('){
                ++cur;
                ans[i]=cur%2;
            }else{
                ans[i]=cur%2;
                --cur;
            }
        }
        
        /*int mx=0; cur=0;
        for (int i=0; i<n; i++){
            if (ans[i]!=0) continue;
            if (seq[i]=='('){
                ++cur;
            }else --cur;
            mx=max(mx, cur);
        }
        cur=0;
        for (int i=0; i<n; i++){
            if (ans[i]!=1) continue;
            if (seq[i]=='('){
                ++cur;
            }else --cur;
            mx=max(mx, cur);
        }*/
        
        return ans;
        
    }
};
