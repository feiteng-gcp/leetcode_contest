class Solution {
public:
    vector<int> maxDepthAfterSplit(string s) {
        const int n = s.size();
        vector<int> ans(n);
        int sa = 0, sb = 0;
        for (int i = 0; i < n; ++i) {
            if (s[i] == '(') {
                if (sa < sb) {
                    sa++;
                    ans[i] = 1;
                } else {
                    sb++;
                }
            } else {
                if (sa > sb) {
                    sa--;
                    ans[i] = 1;
                } else {
                    sb--;
                }
            }
        }
        return ans;
    }
};
