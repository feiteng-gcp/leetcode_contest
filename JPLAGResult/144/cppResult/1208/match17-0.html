class Solution {
public:
  vector<int> maxDepthAfterSplit(string s) {
    int n=s.size();
    int op=0,dep=0;
    for(int i=0;i<n;i++){
      if(s[i]=='(') op++;
      if(s[i]==')') op--;
      dep=max(dep,op);
    }
    int cap=(dep+1)/2;
    vector<int> res(n,-1);
    int op1=0,op2=0;
    for(int i=0;i<n;i++){
      if(s[i]=='('){
        if(op1==cap){
          op2++;
          res[i]=1;
        }else{
          op1++;
          res[i]=0;
        }
      }else{
        if(op2!=0){
          op2--;
          res[i]=1;
        }else{
          op1--;
          res[i]=0;
        }
      }
    }
    return res;
  }
};
