class Solution {
public:
    vector<int> maxDepthAfterSplit(string seq) {
        vector<int> ans;    ans.resize(seq.size());
        int a = 0, b = 0;
        int ma = 0;
        for(int i = 0; i < seq.size(); ++ i) {
               if(seq[i] == '(') {
                   if(a < b){
                       ++ a;
                       ans[i] = 0;
                   }
                   else{
                       ++ b;
                       ans[i] = 1;
                   }
                   ma = max(ma, max(a, b) );
                       
               }
                else{
                    if(a > b){
                       -- a;
                       ans[i] = 0;
                   }
                   else{
                       -- b;
                       ans[i] = 1;
                   }
                   ma = max(ma, max(a, b) );
                }
        }
        return ans;
    }
};
