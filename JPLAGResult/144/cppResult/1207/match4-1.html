class Solution {
	vector<TreeNode*> ans;
	set<int> dele;


	TreeNode* dfs(TreeNode* cu, bool top = true) {
		if (cu == NULL) return NULL;

		if (dele.count(cu->val)) {
			dfs(cu->left, true);
			dfs(cu->right, true);
			return NULL;
		}

		if (top) ans.push_back(cu);

		cu->left = dfs(cu->left, false);
		cu->right = dfs(cu->right, false);

		return cu;
	}

public:
	vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
		ans.clear();
		dele.clear();
		fore(d, to_delete) dele.insert(d);

		dfs(root);

		return ans;
	}
};
