/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        vector<TreeNode*> res;
        unordered_set<int> s(to_delete.begin(), to_delete.end());
        dfs(root, s, true, res);
        return res;
    }
    void dfs(TreeNode* root, unordered_set<int>& s, bool b, vector<TreeNode*>& res) {
        if (!root) return;
        if (s.find(root->val) == s.end()) {
            if (b) res.push_back(root);
            dfs(root->left, s, false, res);
            dfs(root->right, s, false, res);
            if (root->left && s.find(root->left->val) != s.end()) root->left = NULL;
            if (root->right && s.find(root->right->val) != s.end()) root->right = NULL;
        } else {
            dfs(root->left, s, true, res);
            dfs(root->right, s, true, res);
        }
    }
};
