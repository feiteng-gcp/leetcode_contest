class Solution {
public:
	vector<TreeNode*>ret;
	set<int>d;

	void solve(TreeNode* root,bool add) {

		if (root->left && d.find(root->left->val) != d.end()) {
			solve(root->left, true);
			root->left = NULL;
		}
		if (root->right && d.find(root->right->val) != d.end()) {
			solve(root->right, true);
			root->right = NULL;
		}
		if (d.find(root->val) != d.end()) {
			if (root->left)solve(root->left,true);
			if (root->right)solve(root->right,true);
		}
		else {
			if (root->left)solve(root->left, false);
			if (root->right)solve(root->right, false);
			if (add)ret.push_back(root);
		}
	}
	vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
		ret.clear();
		for (auto it : to_delete)d.insert(it);
		solve(root, true);
		return ret;
	}
};
