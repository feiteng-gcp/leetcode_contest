/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
  
int search(vector<int>&to_delete,int l,int h,int x){
    
    if(l>h){
        return -1;
    }
    
    int mid=(l+h)/2;
    
    if(to_delete[mid]==x){
        return mid;
    }else if(to_delete[mid]>x){
        return search(to_delete,l,mid-1,x);
    }else{
        return search(to_delete,mid+1,h,x);
    }
    return -1;
}
    
TreeNode* solve(TreeNode *root,vector<int>&to_delete,vector<TreeNode *>&v,int n){
    
    if(root==NULL){
        return root;
    }
    
    root->left=solve(root->left,to_delete,v,n);
    root->right=solve(root->right,to_delete,v,n);
    
    TreeNode *temp=root;
    
    int x=search(to_delete,0,n-1,root->val);
    

    if(x!=-1){
    
    if( root->left!=NULL){
        v.push_back(root->left);
    }if( root->right!=NULL){
        v.push_back(root->right);
    }
    
        root=NULL;
    }
    return root;
}    
    
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        
        vector<TreeNode *>v;
        if(to_delete.size()==0){
            v.push_back(root);
            return v;
        }
        
        sort(to_delete.begin(),to_delete.end());
        int n=to_delete.size();
        
        TreeNode *root1=solve(root,to_delete,v,n);
        if(root1!=NULL){
            v.push_back(root1);
        }
        return v;
    }
};
