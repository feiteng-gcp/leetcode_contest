/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    set<int> del;
    vector<TreeNode*> seq;
    set<int> vis;
    void dfs(TreeNode *root){
        if(root){
            seq.push_back(root);   
            dfs(root->left);
            dfs(root->right);
        }
    }
    
    void ddfs(TreeNode* root, TreeNode *fa){

        if(del.find(root->val) != del.end()){
            if(fa->left && fa->left->val == root->val){
                fa->left = NULL;
            }
            if(fa->right && fa->right->val == root->val){
                fa->right = NULL;
            }
        }    
        else{
            vis.insert(root->val);
            if(root->left) ddfs(root->left, root);
            if(root->right) ddfs(root->right, root);
        }
    }
    
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        for(auto &x: to_delete) del.insert(x);
        seq.clear();
        dfs(root);
        vector<TreeNode *> ans;
        for(auto x: seq) if(del.find(x->val) == del.end() && vis.find(x->val) == vis.end()){
            ans.push_back(x);
            ddfs(x, NULL);
        }
        return ans;
    }
};
