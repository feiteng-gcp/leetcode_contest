/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> ret;
    set<int> st;
    void traverse(TreeNode* parrent, bool isLeft, TreeNode* current)
    {
        int val=current->val;
        if(st.find(val)!=st.end()){
            if(current->left!=NULL){
                if(st.find(current->left->val)==st.end()) ret.push_back(current->left);
            }
            if(current->right!=NULL){
                if(st.find(current->right->val)==st.end()) ret.push_back(current->right);
            }
        }
        if(current->left!=NULL) traverse(current, true, current->left);

        if(current->right!=NULL) traverse(current, false, current->right);
        if(parrent!=NULL&&st.find(val)!=st.end()){
            if(isLeft) parrent->left=NULL;
            else parrent->right=NULL;
        }
    }
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        ret.clear();
        for(int i=0;i<to_delete.size();i++) st.insert(to_delete[i]);
        if(st.find(root->val)==st.end()) ret.push_back(root);
        traverse(NULL, false, root);
        return ret;
    }
};
