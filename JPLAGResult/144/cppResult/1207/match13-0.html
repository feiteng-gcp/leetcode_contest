/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void traverse(TreeNode* root, unordered_set<int>& del, vector<TreeNode*>& res) {

        if (root->left) {
            traverse(root->left, del, res);
            if (del.count(root->left->val))
                root->left = NULL;
        }
        if (root->right) {
            traverse(root->right, del, res);
            if (del.count(root->right->val))
                root->right = NULL;
        }

        if (del.count(root->val)) {
            if (root->left) {
                res.push_back(root->left);
                root->left = NULL;
            }
            if (root->right) {
                res.push_back(root->right);
                root->right = NULL;
            }
        }
    }

    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        unordered_set<int> del(to_delete.begin(), to_delete.end());
        vector<TreeNode*> res;
        if (root == NULL)
            return res;
        if (!del.count(root->val))
            res.push_back(root);
        traverse(root, del, res);
        return res;
    }

};
