/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        vector<TreeNode *> ans;
        unordered_set<int> tod;
        for(int i=0;i<to_delete.size();++i){
            tod.insert(to_delete[i]);
        }
        // TreeNode * ghost=new TreeNode(INT_MAX);
        // ghost->right=root;
        trav(root,ans,tod,false);
        return ans;
    }
    void trav(TreeNode * & root,vector<TreeNode *> & ans,unordered_set<int> & tod,bool s){
        if(!root)
            return;
        if(!s){
            if(!tod.count(root->val)){
                ans.push_back(root);
                s=true;
            }
        }
        if(s && tod.count(root->val))
            s=false;
        trav(root->left,ans, tod,s);
        trav(root->right,ans,tod,s);
        if(tod.count(root->val))
            root=NULL;
    }
};
