/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        unordered_set<int> hs;
        for (int i = 0; i < to_delete.size(); i++) {
            hs.insert(to_delete[i]);
        }
        vector<TreeNode*> res = helper(root, hs, NULL, NULL);
        return res;
    }
private:
    vector<TreeNode*> helper(TreeNode* root, unordered_set<int>& hs, TreeNode *pl, TreeNode *pr) {
        vector<TreeNode*> res;
        if (!root) return res;
        if (hs.count(root->val)) {
            if (pl) {
                pl->left = NULL;
            } else if (pr) {
                pr->right = NULL;
            }
            vector<TreeNode*> l = helper(root->left, hs, NULL, NULL);
            vector<TreeNode*> r = helper(root->right, hs, NULL, NULL);
            for (int i = 0; i < l.size(); i++) res.push_back(l[i]);
            for (int i = 0; i < r.size(); i++) res.push_back(r[i]);
        } else {
            if (!pl && !pr) {
                res.push_back(root);
            }
            vector<TreeNode*> l = helper(root->left, hs, root, NULL);
            vector<TreeNode*> r = helper(root->right, hs, NULL, root);
            for (int i = 0; i < l.size(); i++) res.push_back(l[i]);
            for (int i = 0; i < r.size(); i++) res.push_back(r[i]);            
        }
        return res;
    }
};
