/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    
    void dfs(bool par_deleted, TreeNode* node, const set<int>& to_delete, vector<TreeNode*>* roots) {
        
        bool curr_deleted = (to_delete.find(node->val) != to_delete.end());
        
        if (par_deleted && !curr_deleted) {
            roots->push_back(node);
        }
        
        if (node->left)
            dfs(curr_deleted, node->left, to_delete, roots);
        if (node->right)
            dfs(curr_deleted, node->right, to_delete, roots);
        
    }
    
    void dfs2(TreeNode** node, const set<int>& to_delete) {
        
        
        if ((*node)->left) {
            if (to_delete.find((*node)->left->val) != to_delete.end()) {
                (*node)->left = nullptr;
            } else {
                dfs2(&(*node)->left, to_delete);
            }
        }
        
        
        if ((*node)->right) {
            if (to_delete.find((*node)->right->val) != to_delete.end()) {
                (*node)->right = nullptr;
            } else {
                dfs2(&(*node)->right, to_delete);
            }
        }
        
        
    }
    
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        set<int> to_delete_set(to_delete.begin(),to_delete.end());
        vector<TreeNode*> roots;
        dfs(true, root, to_delete_set, &roots);
        
        for(int j=0;j<roots.size();j++) {
            dfs2(&roots[j], to_delete_set);
        }
        
        
        return roots;
    }
};
