/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    unordered_set<int> s;
    vector<TreeNode*> ans;
    TreeNode* dfs(TreeNode *x) {
        if (x == nullptr)
            return x;
        x->left = dfs(x->left);
        x->right = dfs(x->right);
        if (s.count(x->val)) {
            if (x->left != nullptr) ans.push_back(x->left);
            if (x->right != nullptr) ans.push_back(x->right);
            return nullptr;
        } else {
            return x;
        }
    }
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        s = unordered_set<int>(to_delete.begin(), to_delete.end());
        ans.clear();
        root = dfs(root);
        if (root != nullptr) {
            ans.push_back(root);
        }
        return ans;
    }
};
