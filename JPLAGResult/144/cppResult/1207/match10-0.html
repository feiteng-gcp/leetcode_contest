/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> all;
    map<TreeNode*, TreeNode*> parent;
    set<TreeNode*> toDelete;
    void dfs(TreeNode* root, TreeNode* par, set<int>&se){
        if(root==NULL)
            return;
        all.push_back(root);
        parent[root]=par;
        if(se.count(root->val))
            toDelete.insert(root);
        dfs(root->left, root, se);
        dfs(root->right, root, se);

    }
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        set<int> se;
        for(auto x:to_delete){
            se.insert(x);
        }
        dfs(root, NULL, se);
        vector<TreeNode*> ret;
        if(se.count(root->val)==0)
            ret.push_back(root);
        for(auto x:all){
            if(x->left){
                if(se.count(x->left->val)){
                    x->left = NULL;
                }
            }
            if(x->right){
                if(se.count(x->right->val)){
                    x->right = NULL;
                }
            }
        }
        for(auto x:all){
            if(toDelete.count(parent[x])&&toDelete.count(x)==0)
                ret.push_back(x);
        }
        return ret;
    }
};
