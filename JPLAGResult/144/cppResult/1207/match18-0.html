/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool delNodes(TreeNode*& root, vector<bool>& to_delete, vector<TreeNode*>& res){
        if(root == NULL){
            return true;
        }else{
            if(delNodes(root->left, to_delete, res)){
                root->left = NULL;
            }
            if(delNodes(root->right, to_delete, res)){
                root->right = NULL;
            }
            if(to_delete[root->val] == true){
                if(root->left != NULL){
                    res.push_back(root->left);
                }
                if(root->right != NULL){
                    res.push_back(root->right);
                }
                root = NULL;
                return true;
            }else{
                return false;
            }
        }
    }
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        vector<TreeNode*> res;
        if(root == NULL){
            return res;
        }else{
            vector<bool> flag(1001, false);
            for(int val : to_delete){
                flag[val] = true;
            }
            if(delNodes(root, flag, res)){
                if(root != NULL && root->left != NULL){
                    res.push_back(root->left);
                }
                if(root != NULL && root->right != NULL){
                    res.push_back(root->right);
                }
            }else{
                res.push_back(root);
            }
            return res;
        }
    }
};
