/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    vector<TreeNode *> res;
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        if(root == nullptr)
            return {};
        unordered_set<int> delete_set(to_delete.begin(), to_delete.end());
        TreeNode * rt = dfs(root, delete_set);
        if(rt)
            res.push_back(rt);
        return res;
    }
    
    TreeNode * dfs(TreeNode * root, unordered_set<int> & delete_set) {
        if(root == nullptr)
            return nullptr;
        TreeNode * ori_l = root->left;
        root->left = dfs(root->left, delete_set);
        TreeNode * ori_r = root->right;
        root->right = dfs(root->right, delete_set);
        if(delete_set.count(root->val)){
            if(root->left)
                res.push_back(root->left);
            if(root->right)
                res.push_back(root->right);
            return nullptr;
        }
        else
            return root;
    }
};
