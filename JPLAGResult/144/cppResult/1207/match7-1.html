/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        vector<TreeNode*> ans;
        if(!root) return ans;
        ans.push_back(root);
        set<int> to_del;
        for(int i: to_delete) to_del.insert(i);
        dfs(root, to_del, ans);
        vector<TreeNode*> res;
        for(TreeNode* node: ans) {
            if(to_del.count(node->val)==0) res.push_back(node);
            else {
                if(node->left) res.push_back(node->left);
                if(node->right) res.push_back(node->right);
            }
        }
        return res;
    }
    void dfs(TreeNode* root, set<int>& to_del, vector<TreeNode*>& res) {
        if(!root) return;

        if(root->left) {
            dfs(root->left, to_del, res);
            if(to_del.count(root->left->val)){
                res.push_back(root->left);
                root->left = NULL;
            }
        }
        if(root->right) {
            dfs(root->right, to_del, res);
            if(to_del.count(root->right->val)) {
                res.push_back(root->right);
                root->right = NULL;
            }
        }
    }
};
