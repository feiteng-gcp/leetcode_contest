class Solution {
public:
  vector<TreeNode*> ans;
  set<int> ds;
  TreeNode* dfs(TreeNode* v,int top){
    if(!v) return v;
    if(ds.count(v->val)){
      dfs(v->left,1);
      dfs(v->right,1);
      return nullptr;
    }
    if(top) ans.emplace_back(v);
    v->left=dfs(v->left,0);
    v->right=dfs(v->right,0);
    return v;
  }
  vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_del) {
    ans.clear();
    ds.clear();
    for(int x:to_del) ds.emplace(x);
    dfs(root,1);
    return ans;
  }
};
