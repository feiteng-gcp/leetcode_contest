/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
map<int,bool> vis;
vector<TreeNode*> ans;
void dfs(TreeNode* root, int is_root){
    int is_del = vis[root->val];
    if(is_root&&!is_del){
        ans.push_back(root);
    }
    if(root->left!=NULL){
        if(is_del)
            dfs(root->left, 1);
        else
            dfs(root->left, 0);
        if(vis[root->left->val]){
            root->left=NULL;
        }
    }
    if(root->right!=NULL){
        if(is_del)
            dfs(root->right, 1);
        else
            dfs(root->right, 0);
        if(vis[root->right->val]){
            root->right=NULL;
        }
    }
}
class Solution {
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        ans.clear();
        vis.clear();
        for(auto a: to_delete){
            vis[a]=1;
        }
        dfs(root, 1);
        return ans;
    }
};
