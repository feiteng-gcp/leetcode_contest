class Solution {
public:
    vector<TreeNode*> ret;
    vector<int> del;
    void dfs(TreeNode* rt,int p)
    {
        int k = 0;
        if(find(del.begin(), del.end(), rt->val) != del.end()) k=1;
        if(p && find(del.begin(), del.end(), rt->val) == del.end()) ret.push_back(rt);

        if(rt->left)
        {
            dfs(rt->left, k);
            if(find(del.begin(), del.end(), rt->left->val)!=del.end()) rt->left=NULL;
        }
        if(rt->right)
        {
            dfs(rt->right, k);
            if(find(del.begin(), del.end(), rt->right->val)!=del.end()) rt->right=NULL;
        }
    }
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        del = to_delete;
        
        dfs(root, 1);
        return ret;
    }
};
