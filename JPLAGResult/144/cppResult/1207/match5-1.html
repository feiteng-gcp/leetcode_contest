/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        if (root == nullptr)
            return {};

        std::unordered_set<int> delCache;
        for (int v : to_delete)
            delCache.insert(v);

        std::vector<TreeNode*> res;
        worker(root, delCache, res, true);
        return res;
    }
    
    TreeNode* worker(TreeNode* root, const std::unordered_set<int>& to_delete, std::vector<TreeNode*>& res, bool parentDeleted)

    {
        if (root == nullptr)
            return nullptr;
        
        if (to_delete.find(root->val) != to_delete.end())
        {
            worker(root->left, to_delete, res, true);
            worker(root->right, to_delete, res, true);
            //delete root;
            return nullptr;
        }

        if (parentDeleted)
            res.push_back(root);
        
        root->left = worker(root->left, to_delete, res, false);
        root->right = worker(root->right, to_delete, res, false);
        return root;
    }
};
