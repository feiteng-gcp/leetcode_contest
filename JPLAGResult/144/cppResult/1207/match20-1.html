/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

vector<TreeNode*> ret;

void postorder(TreeNode* root,set<int> s)
{
    if(root==NULL)
        return;
    
    postorder(root->left,s);
    postorder(root->right,s);
    if(root->left!=NULL)
    {
       if(s.find(root->left->val)!=s.end())
       {
            ret.push_back(root->left->left);
            ret.push_back(root->left->right);
            root->left=NULL;
       }
        
    }
    if(root->right!=NULL)
    {

       if(s.find(root->right->val)!=s.end())
       {
            ret.push_back(root->right->left);
            ret.push_back(root->right->right);
            root->right=NULL;
       }
        
    }
    
}
class Solution {
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        
        set<int> s;
        
        for(int i=0;i<to_delete.size();i++)
        {
            s.insert(to_delete[i]);
        }
        ret.clear();
        
        postorder(root,s);
        
        if(s.find(root->val)!=s.end())
        {
            ret.push_back(root->left);
            ret.push_back(root->right);
            
        }
        else
            ret.push_back(root);
        
        vector<TreeNode*> t;
        for(int i=0;i<ret.size();i++)
        {
            if(ret[i]!=NULL)
                t.push_back(ret[i]);
        }
        return t;
        
    }
};
