/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        unordered_set<int> to_delete_set;
        auto is_to_delete = [&](int i) { return to_delete_set.find(i) != to_delete_set.end(); };
            
        for (int i: to_delete) to_delete_set.insert(i);
        
        vector<TreeNode*> roots;
        roots.push_back(root);
        queue<TreeNode*> q;
        q.push(root);
        while (q.size()) {
            TreeNode *n = q.front();
            q.pop();
            
            if (is_to_delete(n->val)) {
                if (n->left && !is_to_delete(n->left->val) ) {
                    roots.push_back(n->left);
                }
                if (n->right && !is_to_delete(n->right->val)) {
                    roots.push_back(n->right);
                }
            }
            
            if (n->left) q.push(n->left);
            if (n->right) q.push(n->right);
        }
        
        if (is_to_delete(root->val)) {
            roots.erase(roots.begin());
        }
        
        for (TreeNode* root: roots) {
            queue<TreeNode*> q;
            q.push(root);
            while (q.size()) {
                TreeNode *n = q.front();
                q.pop();
                
                if (n->left) {
                    if (is_to_delete(n->left->val)) {
                        n->left = NULL;
                    } else {
                        q.push(n->left);
                    }
                }
                
                if (n->right) {
                    if (is_to_delete(n->right->val)) {
                        n->right = NULL;
                    } else {
                        q.push(n -> right);
                    }
                }
            }
        }
        
        return roots;
    }
};
