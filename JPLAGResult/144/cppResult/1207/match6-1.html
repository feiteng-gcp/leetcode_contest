/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int to_delete2[1005];
    bool vis[1005];
    TreeNode *to[1005];

    void dfs(TreeNode *root) {
        to[root->val] = root;
        if (root->left) {
            if (to_delete2[root->left->val]) {
                dfs(root->left);
                root->left = NULL;
            } else {
                dfs(root->left);
            }
        }
        if (root->right) {
            if (to_delete2[root->right->val]) {
                dfs(root->right);
                root->right = NULL;
            } else {
                dfs(root->right);
            }
        }
    }
    void dfs2(TreeNode *root) {
        vis[root->val] = true;
        cout<<"--> vis "<<root->val<<endl;
        cout<<root->left<<" "<<root->right<<endl;
        if (root->left) {
            dfs2(root->left);
        }
        if (root->right) {
            dfs2(root->right);
        }
    }
    vector<TreeNode *> delNodes(TreeNode *root, vector<int> &to_delete) {
        memset(to_delete2, 0, sizeof(to_delete2));
        memset(vis, 0, sizeof(vis));
        for (auto s : to_delete) {
            to_delete2[s] = 1;
        }
        vector<TreeNode *> ans;
        if (root == NULL)
            return ans;
        dfs(root);
        for (int i = 1; i <= 1000; i++) {
            if (to[i] && !to_delete2[i] && !vis[i]) {
                for (int j = 1; j <= 10; j++) {
                    cout << vis[j] << " ";
                }
                cout << endl;
                dfs2(to[i]);
                ans.push_back(to[i]);
            }
        }
        return ans;
    }
};
