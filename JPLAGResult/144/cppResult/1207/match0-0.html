/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        unordered_set<int> deleted(to_delete.begin(), to_delete.end());
        vector<TreeNode*> res;
        helper(root, res, deleted, false);
        return res;
    }
    
    void helper (TreeNode* root, vector<TreeNode*>& res, unordered_set<int>& deleted, bool has_father) {
        if (root == nullptr) return;
        if (deleted.find(root->val) != deleted.end()) {
            helper(root->left, res, deleted, false);
            helper(root->right, res, deleted, false);
        } else {
            if (!has_father) {
                res.push_back(root);
            }
            helper(root->left, res, deleted, true);
            helper(root->right, res, deleted, true);
            if (root->left && deleted.find(root->left->val) != deleted.end()) root->left = nullptr;
            if (root->right && deleted.find(root->right->val) != deleted.end()) root->right = nullptr;
        }
    }
};
