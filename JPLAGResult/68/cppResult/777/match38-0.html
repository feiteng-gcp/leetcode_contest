class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = (matrix[0]).size();
        for (int i = 0; i < m; i++)
            for (int j = 1; i + j < m && j < n; j++)
                if (matrix[j][i + j] != matrix[0][i])
                    return false;
        for (int i = 1; i < n; i++)
            for (int j = 1; i + j < n && j < m; j++)
                if (matrix[i + j][j] != matrix[i][0])
                    return false;
        return true;
    }
};
