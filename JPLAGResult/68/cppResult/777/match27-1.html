class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        
        int n = matrix.size();

        int m = matrix[0].size();
        for(int i = 0; i < n; ++i) {
            int x = matrix[i][0];
            for(int j = i+1, k = 1; j < n && j < m && k < n && k < m; ++j, ++k)
                if(matrix[j][k] != x) return false;

        }
        
        for(int i = 1; i < m; ++i) {
            int x = matrix[0][i];
            for(int j = 1, k = i+1; j < n && j < m && k < n && k < m; ++j, ++k)
                if(matrix[j][k] != x) return false;
        }
        
        return true;
    }
};
