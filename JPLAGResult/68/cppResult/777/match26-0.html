class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        
        int m = matrix.size();
        int n = matrix[0].size();
        int c = 0 ;
        int nd = m+n-1;
        int i = 0 ; 
        int j = n-1;
        while(nd > 0)
        {
            int ii = i;
            int jj = j;
            
            int element = matrix[ii][jj];
            
            while(ii >=0 && ii < m && jj< n && jj >=0)
            {
                if(matrix[ii][jj] != element)
                    return false;
                ii++;
                jj++;
            }
            if(j >= 0)
                j--;
            if( j< 0 )
            {
                j = 0;
                i++;
            }
            
            nd--;
        }
        return true;
        
    }
};
