class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int m=matrix.size(),n=matrix[0].size();
        int l=max(n,m);
        for(int i=0;i<m;++i)
        {
            int pre=matrix[i][0];
            for(int j=1;j+i<m && j<n ;++j)
            {
                if(pre!=matrix[i+j][j])
                    return false;
            }
        }
        for(int i=1;i<n;++i)
        {
            int pre=matrix[0][i];
            for(int j=1;j+i<n && j<m;++j)
            {
                if(pre!=matrix[j][j+i])
                    return false;
            }
        }
        return true;
    }
};
