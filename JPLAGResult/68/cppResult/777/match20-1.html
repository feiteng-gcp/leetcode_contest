class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int row = matrix.size(), col = matrix[0].size();
        
        for(int i = 0; i < row; i++)
        {

            int x = i + 1, y = 1;
            while(x < row && y < col)
            {
                if(matrix[x][y] != matrix[x - 1][y - 1]) return false;
                ++x;
                ++y;
            }
        }
        
        for(int j = 1; j < col; j++)
        {
            int x = 1, y = j + 1;

            while(x < row && y < col)
            {
                if(matrix[x][y] != matrix[x - 1][y - 1]) return false;
                ++x;
                ++y;
            }
        }
        return true;
    }
};
