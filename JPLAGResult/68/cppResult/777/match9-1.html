class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int n = matrix.size(), m = matrix[0].size();
        for(int i = 0; i < n; i++){
            int now = matrix[i][0];
            for(int j = 0; j + i < n && j < m; j++){
                if(matrix[i+j][j] != now)return false;
            }
        }
        for(int i = 0; i < m; i++){
            int now = matrix[0][i];
            for(int j = 0; j + i < m && j < n; j++){
                if(matrix[j][i+j] != now)return false;
            }
        }
        return true;
    }
};
