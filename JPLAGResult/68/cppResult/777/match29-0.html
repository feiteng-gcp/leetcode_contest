class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        for (int i = 0; i < matrix.size(); ++i)
        {
            int diff = matrix[i][0];
                
            for (int j = 0; j < matrix[0].size(); ++j)
            {
                if (i + j >= matrix.size())
                    break;
                
                if (matrix[i+j][j] != diff)
                    return false;
            }
        }
        
        for (int j = 0; j < matrix[0].size(); ++j)
        {
            int diff = matrix[0][j];
            
            for (int i = 0; i < matrix.size(); ++i)
            {
                if (i + j >= matrix[0].size())
                    break;
                
                if (matrix[i][i+j] != diff)
                    return false;
            }
        }
        
        return true;
    }
};
