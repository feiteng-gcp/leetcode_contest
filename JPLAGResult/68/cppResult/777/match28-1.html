class Solution {
public:
bool check(int x, int y, int n, int m) {
    if(x >= 0 && x < n && y >= 0 && y < m) return 1;
    return 0;
}
bool isToeplitzMatrix(vector<vector<int>>& a) {
    int n = a.size(), m = a[0].size();
    for (int i = 0; i < n; i++) {

        int x = i, y = 0;
        int s = a[x][y];
        while(check(x, y, n, m)) {
            if(a[x][y] != s) return 0;
            x++; y++;
        }
    }
    for (int i = 0; i < n; i++) {
        int x = 0, y = i;

        int s = a[x][y];
        while(check(x, y, n, m)) {
            if(a[x][y] != s) return 0;
            x++; y++;
        }
    }
    return 1;
}
};
