class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        if(matrix.empty()) return false;
        int i = 0,j = 0;
        for(int y = 0;y<matrix[0].size();++y) {
            int l = i,m = y;
            auto val = matrix[l][m];
            while(l<matrix.size()&&m<matrix[0].size()) {
                if(matrix[l][m]!=val) return false;
                ++l;
                ++m;
            }
        }
        for(int x = 0;x<matrix.size();++x) {
            int l = j,m = x;
            auto val = matrix[m][l];
            while(m<matrix.size()&&l<matrix[0].size()) {
                if(matrix[m][l]!=val) return false;
                ++l;
                ++m;
            }
        }
        return true;
    }
};
