class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        for (int i = 0; i < n; ++i) {
            for (int k = 0; i+k < n and k < m; ++k) {
                if (a[i+k][k] != a[i][0]) {
                    return false;
                }
            }
        }
        for (int j = 1; j < m; ++j) {
            for (int k = 0; j+k < m and k < n; ++k) {
                if (a[k][j+k] != a[0][j]) {
                    return false;
                }
            }
        }
        return true;
    }
};
