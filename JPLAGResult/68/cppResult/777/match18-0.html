class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        
        int row = matrix.size();
        int col = matrix[0].size();
        
        for(int i=0; i<row; i++) {
            int value = matrix[i][0];
            int x = i+1;
            int y = 1;
            while (x < row && y <col) {
                if (matrix[x][y] != value) return false;
                x++;
                y++;
            }
        }
        
        for (int j=1; j<col; j++) {
            int value = matrix[0][j];
            int x = 1;
            int y = j+1;
            while (x < row && y <col) {
                if (matrix[x][y] != value) return false;
                x++;
                y++;
            }            
        }
        
        return true;      
    }
};
