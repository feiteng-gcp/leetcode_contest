class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int n = matrix.size();
        if (n==0) return true;
        int m = matrix[0].size();    
        for(int i=0; i<n; i++) {

            for(int ii=i+1, j=1; ii<n and j<m; ii++, j++) {
                if (matrix[ii][j] != matrix[i][0]) return false;
            }
        }
        for(int j=1; j<m; j++) {
            for(int i=1, jj=j+1; i<n and jj<m; i++, jj++) {
                if (matrix[i][jj] != matrix[0][j]) return false;
            }
        }
        return true;
    }
};
