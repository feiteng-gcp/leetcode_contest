class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int rows = matrix.size();
        if(rows == 0)
        {
            return true;
        }
        int cols = matrix[0].size();
        if(cols == 0)
        {
            return true;
        }
        
        for(int i = 0; i < rows; ++i)
        {
            int row = i;
            int col = 0;

            int val = matrix[row][col];
            while(row < rows && col < cols)
            {
                if(matrix[row][col] != val)
                {
                    return false;
                }
                ++row;
                ++col;
            }
        }
        
        for(int j = 0; j < cols; ++j)
        {
            int row = 0;
            int col = j;

            int val = matrix[row][col];
            while(row < rows && col < cols)
            {
                if(matrix[row][col] != val)
                {
                    return false;
                }
                ++row;
                ++col;
            }
        }
        
        return true;
    }
};
