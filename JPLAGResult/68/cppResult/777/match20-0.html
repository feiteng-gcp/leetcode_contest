class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int m = matrix.size(), n = matrix[0].size();
        for (int i = 0; i < m; i++) {
            int cmp = matrix[i][0];
            int row = i, col = 0;
            while (row < m && col < n) {
                if (matrix[row][col] != cmp)
                    return false;
                row++;
                col++;
            }
        }
        for (int i = 1; i < n; i++) {
            int cmp = matrix[0][i];
            int row = 0, col = i;
            while (row < m && col < n) {
                if (matrix[row][col] != cmp)
                    return false;
                row++;
                col++;
            }
        }
        return true;
    }
};
