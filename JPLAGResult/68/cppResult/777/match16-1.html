class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        if (matrix.size() == 0 || matrix[0].size() == 0) return true;
        int rowNum = matrix.size(); 
        int colNum = matrix[0].size();
        for (int ii = 0; ii < rowNum; ++ii) {
            for (int jj = 0; jj < colNum; ++jj) {
                int iicheck = ii + 1; 
                int jjcheck = jj + 1;
                if (iicheck >= rowNum || jjcheck >= colNum || matrix[iicheck][jjcheck] == matrix[ii][jj]) continue;
                else return false;
            }
        }
        return true;
    }
};
