class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) 
    {
        int m=matrix.size();
        int n=matrix[0].size();
        
        int i0=m;
        int j0=0;
        for(int k=0;k<m+n-1;k++)
        {
            if(i0>0)
            {
                i0--;
                j0=0;
            }
            else
            {
                j0++;
            }
            

            int i=i0;
            int j=j0;
            int t=matrix[i0][j0];
            while(i<m&&j<n)
            {
                if(matrix[i][j]!=t) return false;
                i++;
                j++;
            }
            
        }
        
        return true;
    }
};
