class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int n = matrix.size();
        if (n == 0) { return true; }
        int m = matrix[0].size();
        if (m == 0) { return true; }
        for (int i = 0; i < n; i++) {
            int x = i, y = 0;
            while (y < m && x < n) {
                if (matrix[x][y] != matrix[i][0]) { return false; }
                x++;
                y++;
            }
        }
        for (int i = 0; i < m; i++) {
            int x = 0, y = i;
            while (y < m && x < n) {
                if (matrix[x][y] != matrix[0][i]) { return false; }
                x++;
                y++;
            }
        }
        return true;
    }
};
