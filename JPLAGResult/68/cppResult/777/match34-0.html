class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        for(int i=0;i<matrix.size();i++){
            for(int j=0;j<matrix[i].size();j++){
                if(j==0||i==0){
                    if(check(matrix,i,j)==false)
                        return false;
                }
            }
        }
        return true;
    }
    
    int check(vector<vector<int>>& matrix,int m,int n){
        int i=m,j=n,x;
        while(i<matrix.size()&&j<matrix[0].size()){
            if(i==m&&j==n){
                x=matrix[i++][j++];
            }
            else if(x!=matrix[i++][j++])
                return 0;
        }
        return 1;
    }
    
};
