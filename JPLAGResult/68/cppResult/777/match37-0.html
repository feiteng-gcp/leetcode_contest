class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& mat) {
        int n=mat.size();
        int m=mat[0].size();
        for (int i=0;i<n;i++){
            int t=mat[i][0];
            int ff=1;
            for (int k=1;k<m;k++) if (i+k<n){
                if (mat[i+k][k]!=t) {
                    ff=0; break;
                }
            } else break;
            if (!ff) return false;
        }
        for (int i=0;i<m;i++){
            int t=mat[0][i];
            int ff=1;
            for (int k=1;k<n;k++) if (i+k<m){
                if (mat[k][i+k]!=t) {
                    ff=0; break;
                }
            } else break;
            if (!ff) return false;
        }
        return true;
    }
};
