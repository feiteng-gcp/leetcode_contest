class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int n = matrix.size();
        if (n <= 1) return true;
        int m = matrix[0].size();
        if (m <= 1) return true;
        
        for (int r = 0; r < n; ++r) {
            int i = r, j = 0;
            while (i+1 < n && j+1 < m) {
                if (matrix[i][j] != matrix[i+1][j+1])
                    return false;
                i++,j++;
            }
        }
        for (int c = 0; c < m; ++c) {
            int i = 0, j = c;
            while (i+1 < n && j+1 < m) {
                if (matrix[i][j] != matrix[i+1][j+1])
                    return false;
                i++,j++;
            }
        }
        return true;
    }
};
