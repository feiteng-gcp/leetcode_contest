class Solution {
public:

    bool isToeplitzMatrix(vector<vector<int>>& matrix)
    {
        int m = matrix.size();
        if (m == 0) return false;
        int n = matrix[0].size();
        if (n == 0) return false;
        for (int j = 0; j < n; ++j)
        {
            int x = 0, y = j;

            int val = matrix[x][y];
            while (x < m && y < n)
            {
                if (matrix[x][y] != val) return false;
                ++x; 
                 ++y;
            }
            
        }
        for (int i = 0; i < m; ++i)
        {
            int x = i, y = 0;

            int val = matrix[x][y];
            while (x < m && y < n)
            {
                if (matrix[x][y] != val) return false;
                ++x; 
                ++y;
            }
            
        }
        
        return true;
    }
};
