class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
            for (int i=n-1; i>=0; --i) {
                int t = matrix[i][0];
                for (int k=0; k<min(n-i,m); ++k)
                    if (matrix[i+k][k] != t)
                        return false;
            }
            for (int i=1; i<m; ++i) {
                int t = matrix[0][i];
                for (int k=0; k<min(m-i, n); ++k)
                    if (matrix[k][i+k] != t)
                        return false;
            }            
        return true;
    }
};
