class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        if(matrix.empty()) return false;
        int m = matrix.size();
        int n = matrix[0].size();
        //vector<vector<bool>> dp(m, vector<bool>(n, true));
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                int px = i-1;
                int py = j-1;
                if(px>=0&&py>=0){
                    if(matrix[i][j]!=matrix[px][py]) return false;
                }
            }
        }
        return true;
        
    }
};
