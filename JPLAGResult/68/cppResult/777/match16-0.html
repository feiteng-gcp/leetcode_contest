class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int m = matrix.size();
        if (m < 2) return true;
        int n = matrix[0].size();
        if (n <2 ) return true;
        for (int i=0; i<m; i++){
            int y = i+1, x = 1;
            while(y< m && x < n) {
                if (matrix[y][x] != matrix[y-1][x-1]) return false;
                y++, x++;
            }
        }
        
        for (int i=1; i<n; i++) {
            int x = i+1, y = 1;
            while(y< m && x < n) {
                if (matrix[y][x] != matrix[y-1][x-1]) return false;
                y++, x++;
            }
        }
        
        return true;
    }
};
