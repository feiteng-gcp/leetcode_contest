class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int m=matrix.size();
        int n=matrix[0].size();
        for(int i=0; i<n; ++i) {
            int ini = matrix[0][i];
            for(int j=1; j<m; ++j) {
                if(i+j>=n) break;
                if(matrix[j][i+j] != ini) return false;
            }
        }
        for(int i=0; i<m; ++i) {
            int ini = matrix[i][0];
            for(int j=1; j<n; ++j) {
                if(i+j>=m) break;
                if(matrix[i+j][j] != ini) return false;
            }
        }
        return true;
    }
};
