class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int r = matrix.size();
        if (r <= 1)
            return true;
        
        int c = matrix[0].size();
        if (c <= 1)
            return true;
        
        for (int j = 0; j < c; j++)
        {
            int a = matrix[0][j];

            int i = 1;
            int k = j + 1;
            while (i < r && k < c)
            {
                if (matrix[i][k] != a)
                    return false;
                i++;
                k++;
            }
        }
        return true;
    }
};
