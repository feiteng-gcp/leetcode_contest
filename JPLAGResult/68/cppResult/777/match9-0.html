class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& ma) {
        int M=ma.size(),N=ma[0].size();
        for(int i=0;i<M-1;++i){
            int tmp=ma[i][0];
            for(int j=0;j<M-i&&j<N;++j){
                if(ma[i+j][j]!=tmp)     return 0;
            }
        }
        for(int i=1;i<N-1;++i){
            int tmp=ma[0][i];
            for(int j=0;j<M&&j<N-i;++j){
                if(ma[j][i+j]!=tmp)     return 0;
            }
        }
        return 1;
    }
};
