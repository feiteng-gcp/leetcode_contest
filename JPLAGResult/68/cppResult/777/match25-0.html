class Solution {
public:
    bool legal( int x , int y , int row , int col ){
        return x>=0 && y >=0 && x < row && y<col ;
    }
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int n = matrix.size() ; 
        if( n == 0 )
            return false ;
        int m = matrix[0].size() ; 
        if( m == 0 )
            return false ;
        for( int i=0 ; i<m ; i++){
            int x = 0 ;
            int y = i ;
            int num = matrix[x][y];
            while( legal(x,y,n,m) ){
                if( matrix[x][y]!=num )
                    return false ;
                x++;y++;
            }
        }
        for( int i=0 ; i<n ; i++){
            int x = i ;
            int y = 0 ; 
            int num = matrix[x][y];
            while( legal(x , y , n , m ) ){
                if( matrix[x][y]!=num)
                    return false;
                x++;y++;
            }
        }
        return true ;
    }
};
