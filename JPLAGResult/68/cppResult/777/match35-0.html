class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int m = matrix[0].size();
        
        for (int j = 0; j < m; ++j) {
            int y = 0;
            int x = j;
            auto  c = matrix[y][x];
            while (y < n && x < m) {
                if (matrix[y][x] != c) return false;
                y++;
                x++;
            };
        };
        
        for (int i = 0; i < n; ++i) {
            int y = i;
            int x = 0;
            auto c = matrix[y][x];
            while (y < n && x < m) {
                if (matrix[y][x] != c) return false;
                y++;
                x++;
            };            
        };
        
        return true;
        

    }
};
