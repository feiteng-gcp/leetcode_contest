class Solution {
public:
    bool isToeplitzMatrix(vector<vector<int>>& matrix) {
        if(matrix.size() <= 1 || matrix[0].size() <= 1) return true;
        int m = matrix.size();
        int n = matrix[0].size();
        int i = 0, j = 0;
        for(; j < n; j++)
        {
            int curr = matrix[i][j];
            for(int k = 1; i + k < m && j + k < n; k++)
            {
                if(matrix[i + k][j + k] != curr) return false;
            }
        }
        j = 0;
        for(int i = 1; i < m; i++)
        {
            int curr = matrix[i][j];
            for(int k = 1; i + k < m && j + k < n; k++)
            {
                if(matrix[i + k][j + k] != curr) return false;
            }
        }
        return true;
    }
};
