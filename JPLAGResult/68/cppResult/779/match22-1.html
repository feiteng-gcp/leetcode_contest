class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<int> Buff = arr;
		unordered_map<int, int> Map;
		unordered_map<int, int> Count;
		int i;
		int Temp;
		int Result = 0;
		
		sort(Buff.begin(), Buff.end());
		for(i = 0; i < Buff.size(); i++)
		{
			Map[Buff[i]] = i;
			Count[Buff[i]]++;
		}
		
		i = 0;
		while(i < arr.size())
		{
			Temp = Map[arr[i]] - Count[arr[i]] + 1;
			
			while(i <= Temp)
			{
				Count[arr[i]]--;
				Temp = max(Temp, (Map[arr[i]] - Count[arr[i]]));
				i++;
			}
			
			Result++;
		}
		
		return Result;
    }
};
