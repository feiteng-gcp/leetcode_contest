class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int N = arr.size();
        int min_after[N + 1];
        min_after[N] = 100000001;
        for (int i = arr.size() - 1;i >= 0;--i)
            min_after[i] = min(min_after[i + 1], arr[i]);
        int max_now = -1;
        int res = 0;
        for (int i = 0;i < arr.size();++i) {
            max_now = max(max_now, arr[i]);
            if (max_now <= min_after[i + 1]) ++ res;
        }
        return res;
            
    }
};
