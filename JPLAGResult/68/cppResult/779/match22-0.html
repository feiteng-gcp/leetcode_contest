class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<int> tempa=arr;
        vector<int> finalpos;// ith element has final position finalpos[i]
        unordered_map<int,pair<int,int>> m; // value-> final position[begin,end]
        int i,j;
        int rightend;
        int ans = 0;
        
        sort(tempa.begin(), tempa.end());
        
        i=0;
        while (i<tempa.size()){
            j = i;
            while (j+1<tempa.size() && tempa[j+1]==tempa[i]) j++;
            //[i,j]
            m[tempa[i]] = pair<int,int>(i,j);
          //  cout<<"val:"<<tempa[i]<<"["<<i<<","<<j<<"]"<<endl;
            i= j+1;
        }
        
        i =0;
        while (i<arr.size()){
            rightend = max(i,m[arr[i]].first);
            m[arr[i]].first++;
           // cout<<"chunk:"<<i<<"("<<arr[i]<<")->";
            while (i<arr.size() && i!=rightend){
                i++;
                if (i<arr.size()){
                    rightend = max(rightend, m[arr[i]].first);
                    m[arr[i]].first++;
                }
            }
           // cout<<rightend<<"("<<arr[rightend]<<")"<<endl;
            i++;
            ans++;
        }
       // cout<<endl;
        return ans;
        
        
        
    }
};
