class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int n = arr.size();
        vector<int> maxval = arr;
        for (int i = 1; i < n; i++)
            maxval[i] = max(maxval[i], maxval[i - 1]);
        vector<int> dp(n);
        for (int i = 0; i < n; i++) {
            dp[i] = 1;
            int minval = arr[i];
            for (int j = i; j > 0; j--) {
                minval = min(minval, arr[j]);
                if (minval >= maxval[j - 1]) {
                    dp[i] = max(dp[i], dp[j - 1] + 1);
                }
            }
        }
        return dp[n - 1];
    }
};
