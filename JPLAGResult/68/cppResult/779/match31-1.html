class Solution {
public:
    int maxChunksToSorted(vector<int>& a) {
        int n = a.size();
        vector<pair<int, int>> tmp(n);
        for (int i = 0; i < n; ++i) {
            tmp[i] = {a[i],i};
        }
        sort(tmp.begin(), tmp.end());
        vector<int> b(n);
        for (int i = 0; i < n; ++i) {
            b[tmp[i].second] = i;
        }

        int res = 0;
        vector<bool> seen(n);
        int i = 0;
        while (i < n) {
            int maxSeen = -1;
            for (; i < n; ++i) {
                seen[i] = 1;

                maxSeen = max(maxSeen, b[i]);
                bool good = true;
                for (int j = 0; j <= maxSeen; ++j) {
                    if (!seen[j]) {
                        good = false;
                        break;
                    }
                }
                if (good) {
                    break;
                }
            }
            ++res;
            i++;
        }
        return res;
    }
};
