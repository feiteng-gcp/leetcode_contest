class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int left = 0, right = 0,result= 0;
        while(left<arr.size()){
            int i = left+1;
            int m = arr[left], mm = arr[left];
            while(i<arr.size()){
                if(arr[i]<mm){
                    right = i;
                    mm = m;
                }
                if(arr[i]>m){
                    m = arr[i];
                }
                i++;
            }
            left = right + 1;
            right = left;
            result++;
        }
        return result;
    }
};
