int sar[2005], st[2005], rk[2005];
bool cmp(int u, int v) {return sar[u] != sar[v]? sar[u] < sar[v] : u < v;}
class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int n = arr.size(), maxi;
        for (int i = 0; i < n; i++) sar[i] = arr[i], st[i] = i;
        sort(st, st + n, cmp);
        for (int i = 0; i < n; i++) rk[st[i]] = i;
        maxi = -1;

        int cnt = 0;
        for (int i = 0; i < n; i++)
        {
            maxi = max(maxi, rk[i]);
            if (maxi == i) cnt++;
        }
        return cnt;
    }
};
