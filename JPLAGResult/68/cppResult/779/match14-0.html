class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<int> arr2 = arr;
        sort(arr2.begin(), arr2.end());
        unordered_map<int, int> table;
        int count = 0;
        for (int i = 0; i < arr.size(); i++) {
            table[arr2[i]]++;
            table[arr[i]]--;
            if (table[arr2[i]] == 0)
                table.erase(arr2[i]);
            if (table[arr[i]] == 0)
                table.erase(arr[i]);
            if (table.empty()) {
                count++;
            }
        }
        return count;
    }
};
