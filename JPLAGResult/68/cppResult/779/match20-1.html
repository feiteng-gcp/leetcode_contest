class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int ret = 0;
        vector<int> vmin(arr);
        for (int i = (int)vmin.size() - 2; i >= 0; -- i) {
            vmin[i] = min(vmin[i + 1], vmin[i]);
        }
        vmin.push_back(INT_MAX);
        int vmax = INT_MIN;

        for (int i = 0; i < arr.size(); ++ i) {
            vmax = max(vmax, arr[i]);
            if (vmax <= vmin[i + 1]) {
                ret ++;
                vmax = INT_MIN;
            }
        }
        return ret;
    }
};
