class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int n=arr.size(),s=INT_MAX;
        vector<int> small(n);
        for(int i=n-1;i>=0;i--)
        {
            small[i]=s;
            s=min(s,arr[i]);
        }
        int count=0,large=0;
        for(int i=0;i<n;i++)
        {
            large=max(large,arr[i]);
            if(large<=small[i])
                count++;
        }
        return count;
    }
};
