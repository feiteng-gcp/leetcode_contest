bool myfcn(pair<int,int> i,pair<int,int> j) {
    if(i.first==j.first) return i.second < j.second;
    return i.first < j.first;
}
class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        
        vector<pair<int,int>> v;
        for(int i=0;i<(int) arr.size();i++) {
            v.push_back({arr[i],i});
        }
        sort(v.begin(),v.end(),myfcn);
        
        vector<int> narr;
        for(int i=0;i<(int) v.size();i++) {
            narr.push_back(v[i].second);
        }
        
        return maxChunksToSorted2(narr);
    }
    
    int maxChunksToSorted2(vector<int>& arr) {
        
        if(arr.size()==0) return 0;
        int ma = INT_MIN;
        int mi = INT_MAX;
        int div = 0;
        int st = 0;
        int n = arr.size();
        for(int i=0;i<n;i++){
            
            if(arr[i]>ma) ma = arr[i];
            if(arr[i]<mi) mi = arr[i];
            
            if(mi == st && ma == i) {
                div++;
                st = i+1;
                ma = INT_MIN;
                mi = INT_MAX;
            }
        }
        return div;
    }
    
};
