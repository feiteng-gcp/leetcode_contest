class Solution {
public:
    vector<int> eles;
    
    int maxChunksToSorted(vector<int>& arr) {
        int n = arr.size();
        unordered_map<int, int> count, cur_count;
        for (auto cur : arr) count[cur]++;
        for (auto cur : count) eles.push_back(cur.first);
        sort(eles.begin(), eles.end());

        int result = 0;
        for (int i = 0; i < n; i++) {
            cur_count[arr[i]]++;
            if (valid(count, cur_count, i + 1)) result++;
        }
        return result;
    }
    
    bool valid(unordered_map<int, int> &count, unordered_map<int, int> &cur_count, int num) {
        int n = eles.size();
        int sum = 0;
        for (int i = 0; i < n; i++) {
            int cur = eles[i];
            int target = count[cur];
            if (sum + count[cur] > num) target = num - sum;
            if (cur_count[cur] != target) return false;
            sum += count[cur];
            if (sum >= num) break;
        }
        return true;
    }
};
