class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<pair<int, int>> x;
        for(int i = 0; i < arr.size(); ++i)
        {
            x.push_back({arr[i], i});
        }
        
        sort(x.begin(), x.end());
        map<pair<int, int>, int> mp;
        for(int i = 0; i < x.size(); ++i)
        {
            mp[x[i]] = i;
        }
        
        int a = 0, i = 0, end, j;
        while(i < arr.size())
        {
            end = mp[{arr[i], i}];
            for(j = i + 1; j <= end; ++j)
            {
                int z = mp[{arr[j], j}];
                if (z > end)
                    end = z;
            }
            
            ++a;
            i = end + 1;
        }
        
        return a;
    }
};
