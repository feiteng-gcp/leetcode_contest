class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        if (arr.empty())
            return 0;
        vector<int> maxFromFront(arr.size()), minFromBack(arr.size());
        int mmax = INT_MIN;
        for (int i = 0; i < arr.size(); i++) {
            mmax = max(mmax, arr[i]);
            maxFromFront[i] = mmax;
        }

        int mmin = INT_MAX;
        for (int i = arr.size() - 1; i >= 0; i--) {
            mmin = min(mmin, arr[i]);

            minFromBack[i] = mmin;
        }

        int ans = 0;
        for (int i = 0; i < arr.size() - 1; i++) {
            if (maxFromFront[i] <= minFromBack[i + 1])
                ans++;
        }
        return ans + 1;
    }
};
