class Solution {
public:
    int N;
    int dp[2002];
    int max_val[2002];
    
    int maxChunksToSorted(vector<int>& arr) {
        N = arr.size();
        
        vector<int> A = {0};
        for (auto x: arr)
            A.push_back(x);
        
        dp[0] = 0;
        max_val[0] = 0;
        
        for (int i = 1; i <= N; i++) {
            max_val[i] = max(max_val[i-1], A[i]);
            

            int min_val = A[i];
            for (int j = i; j >= 1; j--) {
                min_val = min(min_val, A[j]);
                
                if (min_val >= max_val[j-1]) {
                    // can do
                    //cout << i << " -> " << j - 1 << "\n";
                    dp[i] = max(dp[i], dp[j-1] + 1);
                }
            }
            
            //cout << "dp[" << i << "] = " << dp[i] << "\n";
        }
        
        return dp[N];
    }
};
