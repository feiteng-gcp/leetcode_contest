typedef long long ll;
typedef vector<int> VI;
typedef pair<int,int> PII;

#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))

const int INF = (int)1E9;
#define MAXN 100005

int cntX[MAXN], cntY[MAXN];
int xs[MAXN], X;
class Solution {
public:
  int maxChunksToSorted(vector<int>& arr) {
    int n = arr.size(), s = 0;
    X = 0;
    REP(i,0,n) xs[X++] = arr[i];
    sort(xs, xs+X);
    X = unique(xs,xs+X) - xs;
    REP(i,0,n) arr[i] = lower_bound(xs,xs+X,arr[i]) - xs;
    VI ref = arr;
    sort(ref.begin(), ref.end());

    int ans = 0;
    while (s < n) {
      FILL(cntX, 0); FILL(cntY, 0);
      REP(i,s,n) {

        cntY[ref[i]]++; cntX[arr[i]]++;
        bool ok = true;
        REP(v,0,X) {
          if (cntX[v] != cntY[v]) {
            ok = false;
            break;
          }
        }
        if (ok) {
          s = i + 1;
          ans++;
          break;
        }
      }
    }
    return ans;
  }
};
