class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<int> sarr = arr;
        sort(sarr.begin(), sarr.end());
        int ret = 0;
        map<int, int> ma;
        map<int, int> sma;
        set<int> diff;
        for(int i = 0; i < arr.size(); ++i)
        {
            ma[arr[i]]++;
            sma[sarr[i]]++;
            if(ma[arr[i]] == sma[arr[i]])
            {
                diff.erase(arr[i]);
            }
            else 
            {
                diff.insert(arr[i]);
            }
            
            if(ma[sarr[i]] == sma[sarr[i]])
            {
                diff.erase(sarr[i]);
            }
            else
            {
                diff.insert(sarr[i]);
            }
            
            if(diff.empty())
            {
                ret++;
            }
        }
        
        return ret;
    }
};
