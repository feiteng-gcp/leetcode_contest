class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<int> copy = arr;
        
        map<int, int> count;
        for (auto x: arr) 
            count[x]++;
        
        int n = arr.size();        
        int curmin = INT_MAX;
        int curmax = INT_MIN;
            
        int left = count.begin()->first;
        int right = left;
        int sum = 0;

                
        int i = 0;

        int ans = 0;
        int num = 0;


        for (int i = 0; i < n; ++i) {        
            curmax = max(curmax, arr[i]);
            bool valid = true;
            for (int j = i+1; j < n; ++j) {
                if (arr[j] < curmax) {
                    valid = false;
                    break;
                };
            };

            if (valid) {
                curmax = INT_MIN;
                ans++;                
            };            
        };
        return ans;
        
                
    }
};
