class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        map<int, vector<int> > mp;
        for (int i = 0; i < arr.size(); ++i) mp[arr[i]].push_back(i);
        int end = -1, ans = 0, cnt = 0;
        for (auto& kv : mp) {
            for (auto idx : kv.second) {
                end = max(end, idx);
                if (end + 1 == ++cnt) ++ans;
            }
        }
        return ans;
    }
};
