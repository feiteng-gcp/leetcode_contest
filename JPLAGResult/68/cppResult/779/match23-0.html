class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int j,right,max_num,temp_max;
        int res=0;
        for(int i=0;i<arr.size();)
        {
            right = i;
            max_num = arr[i];
            temp_max = arr[i];
            for(int j=i+1;j<arr.size();j++)
            {
                temp_max=max(temp_max, arr[j]);
                if(arr[j]<max_num)
                {
                    max_num=temp_max;
                    right=j;
                }
            }
            res++;
            //printf("%d %d\n",i,right);
            i=right+1;
        }    
        return res;
            
    }
};
