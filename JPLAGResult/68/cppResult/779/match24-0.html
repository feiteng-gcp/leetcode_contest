class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        if (arr.size() == 0) return 0;
        vector<int> leftmax(arr.size(), -1);
        vector<int> rightmin(arr.size(), INT_MAX);
        for (int i = 0, j = arr.size() - 1; i < arr.size(); ++i, --j) {
            if (i - 1 >= 0) leftmax[i] = max(leftmax[i - 1], arr[i]);
            else leftmax[i] = arr[0];
            if (j + 1 < arr.size()) rightmin[j] = min(arr[j], rightmin[j + 1]);
            else rightmin[j] = arr.back();
        }
        int cnt = 1;
        for (int i = 0; i < arr.size() - 1; ++i) {
            if (leftmax[i] <= rightmin[i + 1]) ++cnt;
        }
        return cnt;
    }
};
