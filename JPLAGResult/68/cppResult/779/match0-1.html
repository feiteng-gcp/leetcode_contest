class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<int> copyarr = arr;
        sort(copyarr.begin(), copyarr.end());
        
        map<int, queue<int> > ordering;
        
        for (int i = 0; i < copyarr.size(); ++i)
            ordering[copyarr[i]].push(i);
        
        int currMin = 0;
        int currMax = -1;
        int ans = 0;
        
        for (int i = 0; i < arr.size();){
            bool minFound = 0;
            
            int j = i;
            for (; j < arr.size() && (minFound == 0 || (j-i)!=(currMax-currMin+1)); ++j){
                
                
                int id = ordering[arr[j]].front();
                ordering[arr[j]].pop();
                
                if (id == currMin){
                    minFound = 1;
                }
                currMax = max(currMax, id);
            }
            
            ++ans;
            currMin = currMax + 1;
            i = j;
        }
        
        return ans;
    }
};
