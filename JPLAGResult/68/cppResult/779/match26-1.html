class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<pair<int, int>>a;
        for(int i=0; i<arr.size(); i++) {
            a.push_back(make_pair(arr[i], i));
        }
        sort(a.begin(), a.end());
        int chunks = 0;
        for(int i=0, j; i<a.size(); i=j) {
            int maxm = INT_MIN;
            for(j=i; j<a.size(); j++) {
                maxm = max(maxm, a[j].second);
                if (maxm == j) {
                    chunks++;
                    break;
                }
            }
            j++;
        }
        return chunks;
    }
};
