class Solution {
public:
    string reorganizeString(string S) {
        string s = S;
        if(s.size() < 2) return s;
        string ans = "";
        vector<int> cnt(256, 0);
        for(int i = 0; i < s.size(); i++) cnt[s[i]]++;
        for(int k = 0; k < s.size(); k++) {
            int mx = -1, cc = -1;
            for(int c = 'a'; c <= 'z'; c++) {
                if(ans.size() > 0 && ans.back() == c) continue;
                if(cnt[c] == 0) continue;
                if(cnt[c] > mx) {
                    mx = cnt[c];
                    cc = c;
                }
            }
            
            if(mx == -1) return "";
            ans += char(cc);
            cnt[cc]--;
        }
        
        return ans;
    }
};
