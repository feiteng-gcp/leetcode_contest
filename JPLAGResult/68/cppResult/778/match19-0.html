class Solution {
public:
    string reorganizeString(string S) {
        vector<int> count(26, 0);
        for (char c : S) {
            count[c - 'a']++;
        }
        vector<int> pos(26, 0);
        int i = 0;
        int hi = 0;
        string str = "";
        while (i < S.size()) {
            int ind = getNextElement(count, pos, hi);
            if (ind == -1) {
                return "";
            }
            else {
                pos[ind] = hi + 2;
                count[ind]--;
                char c = 'a' + ind;
                str += c;
                i++;
            }
            hi++;
        }
        return str;
    }
private:
    int getNextElement(vector<int>& count, vector<int>& pos, int hi) {
        int res = -1, max = 0;
        for (int i = 0; i < 26; i++) {
            if (count[i] > max && pos[i] <= hi) {
                max = count[i];
                res = i;
            }
        }
        return res;
    }
};
