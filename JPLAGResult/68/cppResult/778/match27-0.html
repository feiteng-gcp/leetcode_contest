class Solution {
public:
    string reorganizeString(string S) {
        unordered_map<char, int> mp;
        for(auto & c : S)
            mp[c]++;
        
        priority_queue<pair<int, char>> q;
        for(auto & p : mp)
            q.push({p.second, p.first});
        
        string ans;
        while(!q.empty())
        {
            vector<pair<int,char>> v;
            for(int i = 0; !q.empty() && i < 2; ++i)
            {
                auto x = q.top();q.pop();
                if (!ans.empty() && ans.back() == x.second) return "";
                ans.push_back(x.second);
                x.first--;
                if (x.first)
                    v.push_back(x);
            }
            
            for(auto & x : v)
                q.push(x);
        }
        return ans;        
    }
};
