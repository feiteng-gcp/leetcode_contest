class Solution {
public:
    string reorganizeString(string S)
    {
        int len = S.size();
        int cnt[26];
        memset(cnt, 0, sizeof cnt);
        for (auto ch: S)
            ++cnt[ch - 'a'];
        string ans;

        int last = -1;
        for (int c = 0; c < len; ++c)
        {
            int maxnum = 0;
            int cur = 0;
            for (int i = 0; i < 26; ++i)
            {
                if (cnt[i] > maxnum && i != last)
                {
                    maxnum = cnt[i];
                    cur = i;
                }
            }
            if (maxnum == 0)
                return string("");
            ans += char('a' + cur);
            --cnt[cur];
            last = cur;
        }
        return ans;
    }
};
