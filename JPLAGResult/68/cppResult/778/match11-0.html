class Solution {
public:
    string reorganizeString(string S) {
        int count[26];
        memset(count, 0, sizeof(count));
        
        for (auto ch : S) {
            count[ch - 'a']++;
        }
        
        int start = -1;
        for (int i = 0; i < 26; i++) {
            if (start == -1 || count[i] > count[start]) {
                start = i;
            }
        }
        
        string ans;
        int last = -1;
        for (int i = 0; i < S.size(); i++) {
            int cur = -1;
            for (int i = 0; i < 26; i++) {
                if (i != last && count[i] > 0 && (cur == -1 || count[i] > count[cur])) {
                    cur = i;
                }
            }
            
            if (cur == -1) {
                return "";
            } else {
                count[cur]--;
                ans.push_back('a' + cur);
                last = cur;
            }
        }
        
        return ans;
    }
};
