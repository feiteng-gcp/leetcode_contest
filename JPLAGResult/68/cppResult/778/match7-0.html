class Solution {
public:
    string reorganizeString(string S) {
        int n = S.length();
        unordered_map<char, int> mp;
        for(auto c : S)
            mp[c]++;
        priority_queue<pair<int, char>> q;
        string res;
        char last = '-';
        for(auto m :mp)
            q.push(make_pair(m.second, m.first));
        while(!q.empty()){
            auto p1 = q.top();
            q.pop();
            if(p1.second != last){
                res = res + p1.second;
                last = p1.second;
                p1.first--;
                if(p1.first > 0)
                    q.push(p1);
                else
                    continue;
            }
            else{
                if(q.empty())
                    return "";
                else{
                    auto p2 = q.top();
                    q.pop();
                    res = res + p2.second;
                    last = p2.second;
                    p2.first--;
                    q.push(p1);
                    if(p2.first > 0)
                        q.push(p2);
                    else
                        continue;
                }
            }
        }
        return res;
    }
};
