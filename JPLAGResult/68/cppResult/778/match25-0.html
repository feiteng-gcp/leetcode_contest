class Solution {
public:
    string reorganizeString(string S) {
        int count[128] = {0};
        for (int i = 0; i < S.size(); i++) {
            count[S[i]]++;
        }
        priority_queue<pair<int, char>> q;
        for (auto c = 'a'; c <= 'z'; c++) {
            if (count[c]) {
                q.push({count[c], c});
            }
        }
        string ans;
        char last = 0;
        while (q.size() > 1) {
            auto a = q.top();
            q.pop();
            auto b = q.top();
            q.pop();
            if (a.second == last) {
                swap(a, b);
            }
            ans.push_back(a.second);
            a.first--;
            if (a.first > 0) {
                q.push(a);
            }
            q.push(b);
            last = a.second;
        }
        if (q.size()) {
            if (q.top().first > 1 || q.top().second == last) {
                return "";
            }
            ans.push_back(q.top().second);
        }
        return ans;
    }
};
