class Solution {
public:
    string reorganizeString(string S) {
        if (S.size() == 0) return "";
        vector<int> charCount(26, 0);
        for (int index = 0; index < S.size(); ++index) {
            charCount[S[index] - 'a']++;
        }
        int maxCount = 0;
        int totalLength = S.size();
        for (int index = 0; index < 26; ++index) maxCount = max(maxCount, charCount[index]);
        if ((totalLength % 2 == 0 && maxCount > totalLength /2) || (totalLength % 2 == 1 && maxCount > totalLength /2 + 1)) return "";
        
        vector<pair<int, char>> charList;
        for (int index = 0; index < 26; ++index) {
            if (charCount[index] != 0) {
                charList.push_back(make_pair(charCount[index], 'a' + index));
            }
        }
        sort(charList.rbegin(), charList.rend());
        
        int fpos = 0;
        for (int ii = 0; ii < charList.size(); ++ii) {
            for (int jj = 0; jj < charList[ii].first; ++jj) {
                S[fpos] = charList[ii].second;
                fpos += 2;
                if (fpos >= totalLength) fpos = 1;
            }
        }
        return S;
        
    }
};
