class Solution {
public:
    string reorganizeString(string S) {
        int cnt[26];
	memset(cnt,0,sizeof(cnt));
	int len=S.size();
	string ans="";
	for(int i=0;i<len;i++)
	{
		cnt[S[i]-'a']++;
	}
	for(int i=0;i<26;i++)
	{
		if(cnt[i]>(len+1)/2)return ans;
	}
	for(int i=0;i<len;i+=2)
	{
		int cur=-1;
		for(int j=0;j<26;j++)
		{
			if(cnt[j]&&(cur==-1||cnt[j]>cnt[cur]))cur=j;
		}
		ans=ans+(char)(cur+'a');
		cnt[cur]--;
		if(i<len-1)
		{
			int curr=-1;
			for(int j=0;j<26;j++)
			{
				if(j!=cur&&cnt[j]&&(curr==-1||cnt[j]>cnt[curr]))curr=j;
			}
			ans=ans+(char)(curr+'a');
			cnt[curr]--;
		}
	}
	return ans;
    }
};
