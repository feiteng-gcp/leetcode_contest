class Solution {
public:
    string reorganizeString(string S) {
        int counts[256] = {0};
        for (int i = 0; i < S.size(); ++i) {
            counts[S[i]]++;
        }
        map<int, char> m;
        for (int i = 0; i < 255; ++i) if(counts[i]) m[counts[i]] = i;
        //if (m[m.size()-1] > (S.size()+1)/2) return "";
        string ans;
        char prev = 0;
        while (1) {
            int maxi = 0;
            char next = 0;
            for (int i = 0; i < 255; ++i) {
                if (counts[i] > maxi && i != prev && counts[i]) {
                    maxi = counts[i];
                    next = i;
                }
            }
            if (next == 0) break; 
            counts[next]--;
            prev = next;
            ans += next;
            if (ans.size() == S.size()) break;
        }
        if (ans.size() != S.size()) return "";
        return ans;
    }
};
