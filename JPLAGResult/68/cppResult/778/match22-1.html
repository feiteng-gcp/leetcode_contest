class Solution {
public:
    string reorganizeString(string S) {
   //     S = "aaab";
        vector<int> table(26, 0);
        for (int i = 0; i < S.size(); ++i) {
            table[S[i] - 'a']++;
        }
        
 //       for (int i = 0; i < 26; ++i) {
 //           if (S.size() + 1 < table[i] * 2) return "";
 //       }
        
        int count = S.size();
        string ret = "";
        int lastC = -1;

        while (count > 0) {
            int maxC = 0, maxI = -1;
            for (int i = 0; i < 26; ++i) {
                if (table[i] > maxC && i != lastC) {
                    maxC = table[i];
                    maxI = i;
                }
            }
            
            if (maxI == -1) return "";
            
            ret += (char)(maxI + 'a');
            table[maxI]--;
            lastC = maxI;
            count--;
        }
        return ret;
    }
};
