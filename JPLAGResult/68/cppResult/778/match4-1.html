class Solution {
public:
    string reorganizeString(string S) {
        int cnt[666] = {};
        for (char c : S) cnt[c]++;
        string ans = "";

        const int n = S.size();
        int l = -1;
        for (int i = 0; i < n; ++i) {
            int ch = 0;
            for (int j = 0; j < 128; ++j)
                if ((i == 0 || j != l) && cnt[j] > cnt[ch])
                    ch = j;
            if (ch == 0)
                return "";
            cnt[ch]--;
            ans += char(ch);
            l = ch;
        }
        return ans;
    }
};
