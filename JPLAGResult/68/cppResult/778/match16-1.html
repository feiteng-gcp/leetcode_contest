class Solution {
public:
    string reorganizeString(string S) {
        unordered_map<char, int> m;
        for(auto i: S) {
            m[i] += 1;
        }
        unordered_map<char, int> prev;
        string res;
        while(res.size() < S.size()) {
            char c; int max_freq = -1;

            for(auto i: m) {
                if (i.second > 0 && i.second > max_freq && (prev.find(i.first) == prev.end() || res.size() > prev[i.first] + 1))                {
                    max_freq = i.second;
                    c = i.first;
                }
            }
            if (max_freq == -1) return "";
            prev[c] = res.size();
            m[c] -= 1;
            res += c;
        }
        return res;
    }
};
