class Solution {
public:
    string reorganizeString(string S) {
        vector<int> l(26, 0);
        int cur = 0;
        for (int i = 0; i < S.length(); i++) {
            if (++l[S[i] - 'a'] > l[cur]) {
                cur = S[i] - 'a';
            }
        }
        string ans;
        ans += char('a' + cur);

        l[cur]--;
        for (int i = 1; i < S.length(); i++) {
            int temp = -1;
            for (int j = 0; j < 26; j++) {
                if (l[j] > 0 && (j != ans[i - 1] - 'a') && (temp == -1 || l[temp] < l[j])) {
                    temp = j;
                }
            }
            if (temp == -1) { return ""; }
            l[temp]--;
            ans += char('a' + temp);
        }
        return ans;
    }
};
