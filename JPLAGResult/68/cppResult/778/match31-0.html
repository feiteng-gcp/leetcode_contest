class Solution {
	int cnt[256];
	pair<int, int> maxele;
	void gene() {
		int res = -1, idx = -1;
		for (int i = 'a'; i <= 'z'; ++i) {
			if (res < cnt[i]) {
				res = cnt[i];
				idx = i;
			}
		}
		maxele.first = res;
		maxele.second = idx;
	}
	int geneone(int pre) {
		for (int i = 'a'; i <= 'z'; ++i) {
			if (cnt[i] && i != pre)
				return i;
		}
		return -1;
	}
public:
	string reorganizeString(string S) {
		memset(cnt, 0, sizeof(cnt));
		int len = S.length();
		for (int i = 0; i < len; ++i)
			++cnt[S[i]];
		bool flag = true;
		int tmp;
		string res;
		for (int i = 0; i < len; ++i) {
			gene();
			if (maxele.first > (len - i + 1) / 2) {
				flag = false;
				break;
			}
			else if (maxele.first == (len - i + 1) / 2 && (len - i) % 2 == 1) {
				res.push_back(maxele.second);
				--cnt[maxele.second];
			}
			else {
				int tmp = geneone(i == 0 ? -1 : res.back());
				res.push_back(tmp);
				--cnt[tmp];
			}
		}
		if (!flag)
			return "";
		else
			return res;
	}
};
