class Solution {
public:
    string reorganizeString(string S) {
        
         vector<int> nums(26);
         string sb = "";
		 for(char c: S){
			 int num = c-'a';
			 nums[num]++;
		 }
		 for(int i:nums){
			 if(i>(S.size()+1)/2)
				 return sb;
		 }
		 
		 int last = -1; int curr = -1;
		 for(int i=0; i<S.size(); i++){
			 int max = 0;
			 for(int j=0; j<nums.size(); j++){
				 if(j == last)
					 continue;
				 if(nums[j] > max){
					 max = nums[j];
					 curr = j;
				 }
			 }
			 sb+=((char)('a'+curr));
			 nums[curr]--;
			 last = curr;
		 }
		 return sb;
    }
    
};
