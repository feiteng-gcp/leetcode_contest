class Solution {
public:
    
    string solve(string s) {
        int k = 2;        
        unordered_map<char,int> count;
        for (auto c: s) count[c]++;
        
        
        
        vector<pair<int, char>> freq;
        for (auto p: count) 
            freq.push_back({-p.second, p.first});        
        sort(freq.begin(), freq.end());                
        
        string ans;
        
        while (true) {
            vector<char> round;
            bool last = true;
            for (auto x: freq)
                if (x.first != -1) {
                    last =false;
                    break;
                };
            if (last) {
                for (auto x: freq)
                    ans.push_back(x.second);
                break;
                
            }
                
            if (freq.size() < k) return "";
            
            for (int i = 0; i < k; ++i) 
                round.push_back(freq[i].second);

            for (auto x: round) {
                ans.push_back(x);                            
                count[x]--;                
            };
            
            freq.clear();            
            for (auto p: count)
                if (p.second > 0) 
                    freq.push_back({-p.second, p.first});
            sort(freq.begin(), freq.end());            
        };
        
        return ans;                
    }
    string reorganizeString(string S) {
        return solve(S);
        
    }
};
