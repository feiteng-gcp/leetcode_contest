class Solution {
  public:
    string reorganizeString(string s) {
        const int n = s.length();
        vector<int> c(26, 0);
        for (char ch : s) c[ch - 'a']++;
        int r = n;
        int pre = -1;
        string a;  // answer
        while (r) {
            int j = -1;
            for (int i = 0; i < 26; i++) {
                if (pre == i || c[i] == 0) continue;
                if (j == -1 || c[i] > c[j]) j = i;
            }
            if (j == -1) return "";
            pre = j;
            c[j]--;
            a += char('a' + j);
            r--;
        }
        return a;
    }
};
