class Solution {
public:
    string reorganizeString(string S) {
        int counts[26] = {0};
        for (char c : S) {
            counts[c - 'a']++;
        }
        int last = -1;
        string result;
        for (int i = 0; i < S.length(); i++) {
            int c = last == 0 ? 1 : 0;
            for (int letter = 0; letter < 26; letter++) {
                if (letter != last && counts[letter] > counts[c]) {
                    c = letter;
                }
            }
            if (counts[c] < 1) {
                return "";
            }
            last = c;
            counts[c]--;
            result.push_back('a' + c);
        }
        return result;
    }
};
