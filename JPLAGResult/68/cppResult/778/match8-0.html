class Solution {
public:
    string reorganizeString(string S) {
        int n = S.length();
        int counts[26] = {0};
        for(char c: S){
            ++counts[c-'a'];
        }
        int idx[26];
        iota(&idx[0], &idx[26], 0);
        sort(&idx[0], &idx[26], [&counts](auto one, auto two){
            return counts[one] > counts[two];
        });
        if(n%2 == 0 && counts[idx[0]]>n/2) return "";
        if(n%2 ==1 && counts[idx[0]]>n/2+1) return "";
        string tmp;
        for(int i: idx){
            if(counts[i]==0) break;
            while(counts[i]--){
                tmp += static_cast<char>('a'+i);
            }
        }
        string result=S;
        int j=0;
        for(int i=0; i<n; i+=2){
            result[i]=tmp[j++];
        }
        for(int i=1; i<n; i+=2){
            result[i]=tmp[j++];
        }
        return result;
    }
};
