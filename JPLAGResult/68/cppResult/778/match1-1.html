class Solution {
public:
    string reorganizeString(string S) {
        vector<int> cnt(26, 0);

        int len = S.length();
        for (int i = 0; i < len; i++) {
            cnt[S[i] - 'a']++;
        }
        for (int i = 0; i < 26; i++) {
            if (cnt[i] > (len + 1) / 2) return "";
        }
        string ans = "";
        int last = -1;

        for (int i = 0; i < len; i++) {
            int c = -1;
            for (int j = 0; j < 26; j++) {
                if (j == last) continue;
                if (c == -1 || cnt[c] < cnt[j]) c = j;
            }
            last = c;
            cnt[c]--;
            ans += ((char) c + 'a');
        }
        return ans;
    }
};
