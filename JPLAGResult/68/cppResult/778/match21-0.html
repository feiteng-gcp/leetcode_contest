class Solution {
public:
    string reorganizeString(string S) {
        string res = "";
        vector<int> store (26, 0);
        for (auto &i : S){
            int idx = i - 'a';
            store[idx] ++;
        }
        int ct = 0;
        int ch = findMax(store, -1);
        while (ch != -1){
            ct ++;
            res += ('a' + ch);
            if (ct == S.size()){
                return res;
            }
            ch = findMax(store, ch);
            if (ch == -1){
                return "";
            }
        }
        return "";
    }
    
    int findMax(vector<int>&store, int last){
        int mi = -1;
        int mv = 0;
        for (int i = 0; i < 26; ++i){
            if (store[i] > mv && i != last){
                mv = store[i];
                mi = i;
            }
        }
        if (mi == -1) return -1;
        store[mi]--;
        return mi;
    }
};
