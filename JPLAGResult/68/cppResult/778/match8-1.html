class Solution {
public:
    string reorganizeString(string S) {
        int sz=S.size();
        int cnts[26]={0};
        for(char c: S)
            ++cnts[c-'a'];
        int maxCnt=0;
        for(int i: cnts)
            maxCnt=max(maxCnt, i);
        if(maxCnt>sz/2 && sz%2==0 || maxCnt>sz/2+1 && sz%2) {
            return "";
        }
        
        string res;
        int prev=26;
        for(int i=0; i<sz; ++i) {
            int mxIdx = (prev ? 0 : 1);
            int mx=cnts[mxIdx];
            for(int j=0; j<26; ++j)
                if(j!=prev && cnts[j]>cnts[mxIdx])
                    mxIdx=j,mx=cnts[j];
            
            res.push_back('a'+mxIdx);
            --cnts[mxIdx];
            
            prev=mxIdx;
        }
        
        return res;
        
    }
};
