class Solution {
public:
	string reorganizeString(string s) {
		int n = s.length(), i, j, k, best, id, last = 0;
		vector<int> ct(26, 0); 
		string s2 = "";
		for (i = 0; i < n; i++) ct[s[i] - 'a']++;

		for (i = 0; i < n; i++) {
			best = 0;
			for (j = 0; j < 26; j++) {
				if ((ct[j] > best) && ((i == 0) || (j != last))) {
					best = ct[j];
					id = j;
				}
			}
			if (best == 0)
				return "";
			s2 = s2 + (char)('a' + id);
			ct[id]--;
			last = id;
		}
		return s2;
	}
};
