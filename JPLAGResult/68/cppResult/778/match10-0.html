class Solution {
public:
    string reorganizeString(string S) {
        vector<int>f(26,0);
        for(int i=0;i<S.size();i++)
            f[S[i]-'a']++;
        int n=S.size();
        for(int i=0;i<26;i++)
            if(f[i]>(n+1)/2)return "";
        
        int last=-1;
        string dev="";
        for(int i=0;i<n;i++){
            int maxi=0;
            int id=0;
            for(int j=0;j<26;j++)
                if(f[j]>maxi &&last!=j){
                    maxi=f[j];
                    id=j;
                }
            
            f[id]--;
            last=id;
            dev+=('a'+id);
        }
        return dev;
    }
};
