class Solution {
public:
	struct letter
	{
		int freq;
		char c;
		bool operator<(const letter &k) const{
			return freq < k.freq;
		}
	};
	string reorganizeString(string S) {

		int n = S.length();
		vector<int>ct(26, 0);
		for (int i = 0; i < n; i++)
			ct[S[i] - 'a']++;

		priority_queue< letter > pq;
		for (int  i = 0; i < 26;i++) {
			int x = i + 'a';
			if (ct[x - 'a'])pq.push(letter{ ct[x - 'a'], x });
		}
		S = "";
		letter p{ -1, '#' };
		bool first = true;
		while (!pq.empty()){
			letter t = pq.top();
			pq.pop();
			S = S + t.c;
			if (p.freq > 0)pq.push(p);
			t.freq--;
			p = t;
		}
		if (n != S.length())return "";
		else return S;
	}
};
