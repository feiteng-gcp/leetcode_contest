class Solution {
public:
    string reorganizeString(string s) {
        vector<bool> avail(26, true);
        string res = "";
        vector<int> map(26, 0);
        for(int i = 0; i < s.size(); i ++){
            map[s[i]-'a'] ++;
        }
        for(int i=0; i< s.size(); i++){
            int maxc = 0, idx = -1;
            for(int j = 0; j < 26; j++){
                if(avail[j] && map[j] > maxc){
                    maxc = map[j];
                    idx = j;
                }
            }
            if(idx == -1) return "";
            res += ('a' + idx);
            map[idx] --;
            avail[idx] = false;
            if(i-1 >= 0)
                avail[res[i-1] - 'a'] = true;
        }
        return res;

    }
};
