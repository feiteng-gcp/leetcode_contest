struct Node {
    char ch;
    int count;
    Node(char c, int cnt) : ch(c), count(cnt) {}
    bool operator<(const Node &node) const {
        return count < node.count;
    }
};

class Solution {
public:
    string reorganizeString(string S) {
        vector<int> A(128, 0);
        
        for (char ch : S) {
            A[ch]++;
        }
        
        priority_queue<Node> q;
        
        for (int i = 0; i < 128; i++) {
            if (A[i] != 0) {
                q.push(Node(i, A[i]));
            }
        }
        
        int n = S.size(), i = 0;
        string res = S;
        
        while(!q.empty()) {
            auto t = q.top();
            q.pop();
            
            if (t.count > (n+1) / 2) return "";
            

            for (int j = 0; j < t.count; j++) {
                res[i] = t.ch;
                i += 2;
                if (i >= n) i = 1;
            }
        }
        
        return res;
    }
};
