class Solution {
public:
  int maxChunksToSorted(vector<int>& a) {
    int n = a.size();
    vector<int> d(n + 1, 0), mx(n + 1);
    for (int i = 0; i < n; ++i) {
      mx[i + 1] = max(mx[i], a[i]);
    }
    for (int i = 0; i < n; ++i) {
      int mn = a[i];
      for (int j = i; j >= 0; --j) {
        if (mn >= mx[j]) {
          d[i + 1] = max(d[i + 1], 1 + d[j]);
        }
        if (j > 0) {
          mn = min(mn, a[j - 1]);
        }
      }
    }
    return d[n];
  }
};
