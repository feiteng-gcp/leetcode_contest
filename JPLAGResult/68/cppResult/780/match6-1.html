class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<int> sorted = arr;
        sort(sorted.begin(), sorted.end());

        int res = 0, curr_max = INT_MIN;
        for (int i = 0; i < arr.size(); i++) {
            curr_max = max(curr_max, arr[i]);
            if (curr_max == sorted[i]) res++;
        }
        return res;
    }
};
