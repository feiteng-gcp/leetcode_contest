class Solution {
public:
    int maxChunksToSorted(vector<int>& a) {
        vector<int> next;
        for (int i = 0; i < a.size(); i++) {
            int pos = i;
            for (int j = i + 1; j < a.size(); j++)
                if (a[j] < a[i])
                    pos = j;
            next.push_back(pos);
        }
        int res = 0, p = 0;
        while (p < a.size()) {
            int q = next[p];
            while (p <= q) {
                q = max(q, next[p]);
                p++;
            }
            res++;
            p = q + 1;
        }
        return res;
    }
};
