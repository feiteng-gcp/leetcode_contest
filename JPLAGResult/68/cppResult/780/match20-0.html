class Solution {
public:
    bool check(vector<int> &arr, vector<int>& sortedarr, int begin, int end) {
        vector<int> sortedarr2(arr.begin(), arr.end());
        sort(sortedarr2.begin()+begin, sortedarr2.begin()+end);
        
        for (int i = begin; i < end; i++) {
            if (sortedarr2[i] != sortedarr[i]) return false;
        }
        return true;
    }
    
    bool check2(vector<int> &arr, vector<int>& sortedarr, int begin, int end) {
        
        for (int i = begin; i < end; i++) {
            if (arr[i] != sortedarr[i]) return false;
        }
        return true;
    }
    
    int myMaxChunksToSorted(vector<int>& arr, vector<int>& sortedarr, int begin, int end) {
        if (end == begin) return 0;
        if (check2(arr,sortedarr,begin,end)) return (end-begin);
        if (!check(arr,sortedarr,begin,end)) return 0;
        
        int max_chunk = 1;
        for (int i = begin+1; i < end; i++) {
            int nchunk = myMaxChunksToSorted(arr, sortedarr, begin, i) + myMaxChunksToSorted(arr, sortedarr, i, end);
            if (nchunk > max_chunk) max_chunk = nchunk;
        }
        
        return max_chunk;
    }
    
    int maxChunksToSorted(vector<int>& arr) {
        vector<int> sortedarr(arr.begin(), arr.end());
        sort(sortedarr.begin(), sortedarr.end());
        
        return myMaxChunksToSorted(arr, sortedarr, 0, arr.size());
    }
};
