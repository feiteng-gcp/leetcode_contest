class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<int> a = arr;
        vector<int> b = a;
        int n = a.size();
        sort(b.begin(), b.end());
        int ptr = 0;
        int ans = 0;
        while(ptr < n) {
            multiset<int> s1, s2;
            int nptr;
            for(int i = ptr; i < n; i++) {
                s1.insert(a[i]);
                s2.insert(b[i]);
                if(s1 == s2) {
                    nptr = i;
                    break;
                }
            }
            
            ans++;
            ptr = nptr + 1;
        }
        
        return ans;
    }
};
