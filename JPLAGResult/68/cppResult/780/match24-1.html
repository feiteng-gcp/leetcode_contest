class Solution {
public:
    int maxChunksToSorted(vector<int>& arr)
    {
        vector<int>v = arr;
        sort(v.begin(),v.end());
        map<int,vector<int>>hax;
        for(int i=0;i<v.size();i++)
        {
            hax[v[i]].push_back(i);
        }

        vector<pair<int,int>>val;

        for(int i=0;i<arr.size();i++)
        {
            int pos1 = hax[arr[i]][0];
            int pos2 = hax[arr[i]].back();
            if(abs(i-pos1)<abs(i-pos2))
            {
                val.push_back(mp(min(i,pos1),max(i,pos1)));
                hax[arr[i]].erase(hax[arr[i]].begin());
            }
            else
            {
                val.push_back(mp(min(i,pos2),max(i,pos2)));
                hax[arr[i]].pop_back();
            }
        }

        int ans = 1;
        sort(val.begin(),val.end());
        int st = val[0].second;
        for(int i=1;i<val.size();i++)
        {
            if(val[i].first < st)

            {
                st = max(st,val[i].second);
            }
            else
            {
                st = val[i].second;
                ans++;
            }
        }
        DE(ans)
        return ans;
    }
