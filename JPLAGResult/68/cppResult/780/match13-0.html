class Solution {
private:
    vector<int> root;
    int find_root(int key) {
        while(root[key] != key) {
            key = root[key];
        }
        return key;
    }
    void merge(int a, int b) {
        int root_a = find_root(a);
        int root_b = find_root(b);
        
        root[root_a] = root_b;
    }
public:
    int maxChunksToSorted(vector<int>& arr) {
        root = vector<int>(arr.size());
        for(int i = 0; i < arr.size(); ++i) {
            root[i] = i;
        }
        for(int i = 0; i < arr.size(); ++i) {
            for(int j = i; j <= arr[i]; ++j) {
                merge(j, i);
            }
        }
        
        set<int> ans;
        for(int i = 0; i < arr.size(); ++i) {
            ans.insert(find_root(i));
        }
        return ans.size();
    }
};
