class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        vector<int> flag(arr.size()), minInd, maxInd;
        
        for (int i = 0; i < arr.size(); i++) {
            int minIndex = arr.size() - 1, maxIndex = 0;
            
            while (flag[i] == 0) {
                if (minIndex > i)
                    minIndex = i;
                
                if (maxIndex < i)
                    maxIndex = i;
                
                flag[i] = 1;
                i = arr[i];
            }
            
            if (minIndex <= maxIndex) {
                minInd.push_back(minIndex);
                maxInd.push_back(maxIndex);
            }
        }
        
        int count = 1;
        
        for (int i = 1; i < maxInd.size(); i++) {
            if (minInd[i] > maxInd[i-1])
                count++;
            
            if (maxInd[i] < maxInd[i-1])
                maxInd[i] = maxInd[i-1];
        }
            
        
        return count;
    }
};
