class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int ret = 0, n = arr.size(), cur = 0, i = 0;
        while (i < n) {
            ++ret;
            while (i <= cur) {
                if (arr[i] != i) {
                    cur = max(cur, arr[i]);
                }
                ++i;
            }
            ++cur;
        }
        return ret;
    }
};
