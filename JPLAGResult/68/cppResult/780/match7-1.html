class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int n = arr.size();
        vector<int> idx(n);
        iota(idx.begin(), idx.end(), 0);
        int max_i = 0;

        int chunks=0;
        for(int i=0; i<n; ++i){
            max_i=max(arr[i], max_i);
            if(max_i <= i) ++chunks;
        }
        return chunks;
    }
};
