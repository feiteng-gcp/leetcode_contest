class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        //got it
        // 2-0-1-3
        if(arr.empty()) return 0;
        int ret = 0;
        int findVal = 0;
        auto maxVal = 0;
        set<int> mSet;
        for(int i = 0;i<arr.size();++i) {
//             if(arr[i]!=findVal||arr[i]!=maxVal) {
                
//                 mSet.insert(arr[i]);
//             }
            maxVal = max(maxVal,arr[i]);
            mSet.insert(arr[i]);
            auto vec = vector<int>(mSet.begin(),mSet.end());
            if(vec[vec.size()-1]== (int)(vec.size()-1)) {
                ret++;
            }
            else {
                findVal = vec.size()-1;
                maxVal = vec[vec.size()-1];
                
            }
        }
        return ret;
    }
};
