class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        
        if(arr.size()==0) return 0;
        int ma = INT_MIN;
        int mi = INT_MAX;
        int div = 0;
        int st = 0;
        int n = arr.size();
        for(int i=0;i<n;i++){
            
            if(arr[i]>ma) ma = arr[i];
            if(arr[i]<mi) mi = arr[i];
            
            if(mi == st && ma == i) {
                div++;
                st = i+1;
                ma = INT_MIN;
                mi = INT_MAX;
            }
        }
        return div;
    }
};
