class Solution {
public:
    bool feasible(vector<int>& arr, int from, int to) {
        for (int i = from; i <= to; i++) {
            if (arr[i] < from || arr[i] > to) return false;
        }
        return true;
    }

    int maxChunksToSorted(vector<int>& arr) {
        int answer = 0;
        int n = arr.size();
        int idx = 0;
        while (idx < n) {
            int right = idx;
            while (true) {
                if (feasible(arr, idx, right)) {
                    answer++;
                    idx = right + 1;
                    break;
                }
                right++;
            }
        }
        return answer;
    }
};
