class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
  		int l=arr.size();
  		int ch=0,start=0;
  		unordered_map<int,int> mp;
  		bool b;
  		for (int i=0;i<l;i++)
  		{
  			b=false;
  			mp[arr[i]]=1;
  			for (int j=start;j<=i;j++)
  			{
  				if (mp[j]==0)
  				{
  					b=true;
  					break;
  				}
  			}
  			if (b==false)
  			{
  					start=i+1;
  					ch++;
  			}
  		}      
  		return ch;
    }
};
