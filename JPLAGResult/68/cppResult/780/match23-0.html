class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int start = 0, end = -1;
        int cnt = 0;
        for(int i = 0; i < arr.size(); ++i){
            if(i == end + 1){
                //save the old segment
                if(i > 0) cnt++;
                start = i;
                end = i;
            }
            end = max(end, arr[i]);
        }
        cnt++;
        return cnt;
    }
};
