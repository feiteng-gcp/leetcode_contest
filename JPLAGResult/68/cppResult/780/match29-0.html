class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        
        int count = 0, max_index = arr[0];
        for(int i=0; i< arr.size(); ++i)
        {
            max_index = max(max_index, arr[i]);
            if(max_index == i)
            {
                count += 1;
            }
            
        }
        
        return count == 0 ? 1 : count;        
    }
};
