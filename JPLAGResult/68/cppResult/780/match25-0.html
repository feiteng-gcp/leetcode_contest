class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        if(arr.size() == 1) return 1;
        int curr_min = arr[0];
        int curr_max = arr[0];
        int curr_len = 0;
        int start_p = 0;
        int chunk = 0;
        for(int i = 0; i < arr.size(); i++)
        {
            if(curr_len == 0)
            {
                curr_min = curr_max = arr[i];
                curr_len = 1;
            }
            else
            {
                curr_min = min(curr_min, arr[i]);
                curr_max = max(curr_max, arr[i]);
                curr_len++;
            }
            if(curr_min == start_p && curr_max == start_p + curr_len - 1)
            {
                chunk++;
                start_p = curr_max + 1;
                curr_len = 0;
            }
        }
        return chunk;
    }
};
