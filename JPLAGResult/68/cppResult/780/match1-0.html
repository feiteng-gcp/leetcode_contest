class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int n = arr.size();
        vector<int> last(10, -1);
        for (int i = 0; i < n; ++i) {
            last[arr[i]] = i;
        }
        int j = 0;
        int res = 0;
        for (int i = 0; i < n; ++i) {
            j = max(j, last[i]);
            if (i == j) {
                j = i+1;
                res++;
            }
        }
        return res;
    }
};
