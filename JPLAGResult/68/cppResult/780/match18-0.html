class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {
        int look = 0;
        int mx = 0;
        int chunk = 0;
        for (int i =0; i<arr.size(); i++) {
            if (arr[i] != look) {
                mx = max(mx, arr[i]);
            }
            else {
                chunk++;
                while(mx > i ) {
                    for(; i <= mx; i++)
                        mx = max(mx, arr[i]);
                    i--;
                }
                look = ++mx;
            }
        }
        return chunk;
    }
};
