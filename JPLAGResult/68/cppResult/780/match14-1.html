class Solution {
public:
    int maxChunksToSorted(vector<int>& arr) {        
        int n = arr.size();
        if (n == 0) return 0;
        
        int i = 0;
        int curmin = INT_MAX;
        int curmax = INT_MIN;
        
        int left = 0;
        int ans = 0;
        
        while (i < n) {
            curmin = min(curmin, arr[i]);
            curmax = max(curmax, arr[i]);
            //printf("%d %d\n", curmin, curmax);
            
            if (curmin == left && curmax == i) {
                left = i+1;
                curmin = INT_MAX;
                curmax = INT_MIN;
                ans++;                                
            };    
            ++i;
        };
        
        return ans;
        
        
    
    }
};
