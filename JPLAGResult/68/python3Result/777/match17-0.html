class Solution:
    def isToeplitzMatrix(self, matrix):
        """
        :type matrix: List[List[int]]
        :rtype: bool
        """
        if not matrix: return True
        rn, cn = len(matrix), len(matrix[0])
        
        def allSame(i, j):
            v = matrix[i][j]
            while i < rn and j < cn:
                if matrix[i][j] != v:
                    return False
                i += 1
                j += 1
            return True
            
        ret = True
        for j in range(cn):
            ret = ret and allSame(0,j)
        for i in range(1, rn):
            ret = ret and allSame(i,0)
        return ret
