class Solution:
    def isToeplitzMatrix(self, matrix):
        #print(matrix[:][0], matrix[0][:])
        WideSize = len(matrix[0][:])
        HighSize = len(matrix)
        for i in range(0, WideSize):
            if i <= WideSize - HighSize:
                JMax = HighSize
            else:
                JMax = WideSize - i
            for j in range(1, JMax):
               # print("Wide", i+j,i, JMax  )
                if matrix[0][i] != matrix[j][i+j]:
                    #print("[j][i+j]",matrix[j][i+j], matrix[0][i])
                    return False
        #print("finish wide")
        for i in range(0, HighSize):
            if i <= HighSize - WideSize:
                JMax = WideSize
            else:
                JMax = HighSize - i
            for j in range(1, JMax):
                #print("High",i+j,i, JMax  )
                if matrix[i+j][j] != matrix[i][0]:
                    #print("[i+j][j]",matrix[i+j][j], matrix[i][0])
                    return False
        return True
