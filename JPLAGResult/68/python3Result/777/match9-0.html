class Solution:
    def isToeplitzMatrix(self, matrix):
        n=len(matrix)
        m=len(matrix[0])
        for k in range(m):
            t=matrix[0][k]
            i=1
            j=k+1
            while i<n and j<m:
                if t!=matrix[i][j]:
                    return False
                i+=1
                j+=1
        for k in range(n)[1:]:
            t=matrix[k][0]
            i=k+1
            j=1
            while i<n and j<m:
                if t!=matrix[i][j]:
                    return False
                i+=1
                j+=1
        return True
