class Solution:
    def isToeplitzMatrix(self, matrix):
        """
        :type matrix: List[List[int]]
        :rtype: bool
        """        
        M = len(matrix)
        N = len(matrix[0])
        for i in range(M):
            temp = matrix[i][0]
            dm, dn = 1, 1
            while i + dm < M and dn < N:
                if matrix[i + dm][dn] != temp:
                    break
                dm += 1
                dn += 1
            else:
                continue
            return False
        for i in range(1, N):
            temp = matrix[0][i]
            dm, dn = 1, 1
            while dm < M and i + dn < N:
                if matrix[dm][i + dn] != temp:
                    break
                dm += 1
                dn += 1
            else:
                continue
            return False
        return True
