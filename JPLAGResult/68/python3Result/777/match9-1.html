class Solution:
    def isToeplitzMatrix(self, matrix):
        """
        :type matrix: List[List[int]]
        :rtype: bool
        """
        if matrix is None:
            return False
        m = len(matrix)
        if m == 0:
            return True
        n = len(matrix[0])
        if n == 0:
            return True
        
for p in range(n):
            check = matrix[0][p]
            i = 0
            j = p
            while i < m and j < n:
                if matrix[i][j] != check:
                    return False
                i += 1
                j += 1
        for p in range(1, m):
            check 
= matrix[p][0]
            i = p
            j = 0
            while i < m and j < n:
                if matrix[i][j] != check:
                    return False
                i += 1
                j += 1
        return True
