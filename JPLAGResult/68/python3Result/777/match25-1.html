class Solution:
    def isToeplitzMatrix(self, matrix):
        """
        :type matrix: List[List[int]]
        :rtype: bool
        """
        n, m = len(matrix), len(matrix[0])
        
        # Check from the top row
        for i in range(m):
            el = matrix[0][i]
            for j in range(1, n):
                nxt_i = i + j
                if self.can_go(matrix, j, nxt_i) and matrix
[j][nxt_i] != el:
                    return False
            
        # Check from the left column
        for i in range(n):
            el = matrix[i][0]
            for j in range(1, m):
                nxt_i = i + j
                if self.can_go(matrix, nxt_i, j) and matrix[nxt_i][j] != el:
                    return False
                
        return True
            
    def can_go(self, matrix, i, j):
        return i < len(matrix) and j < len(matrix[0]) and i >= 0 and j >= 0
