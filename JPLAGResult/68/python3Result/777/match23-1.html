class Solution:
    def isToeplitzMatrix(self, matrix):
        """
        :type matrix: List[List[int]]
        :rtype: bool
        """
        row = len(matrix)
        col = len(matrix[0])
        if row <= col:
            for i in range(col - row + 1):
                tmp = matrix[0][i]
                for j in range(1, row):
                    if matrix[j][i + j] != tmp:
                        return False
            if row != col:
                for i in range(1, row - 1):
                    tmp1 = matrix[i][0]
                    tmp2 = matrix[row - i - 1][-1]
                    for j in range(1, row - i):
                        if matrix[i + j][j] != tmp1 or matrix[row - i - j - 1][-1 - j] != tmp2:
                            return False
        else:
            for i in range(row - col + 1):
                tmp = matrix[i][0]
                for j in range(1, col):
                    
if matrix[i + j][j] != tmp:
                        return False
            for i in range(1, col - 1):
                tmp1 = matrix[0][i]
                tmp2 = matrix[-1] [col - i - 1]
                for j in range(1, col - j):
                    if matrix[j][i + j] != tmp1 or matrix[-1 - j][col - i - j - 1] != tmp2:
                        return False
        
        return True
