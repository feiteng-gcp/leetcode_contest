class Solution:
    
def isToeplitz(self, matrix, r_num, c_num, px, py):
        target = matrix[px][py]
        while px < r_num and py < c_num:
            if target != matrix[px][py]:
                return False
            px += 1
            py += 1
        return True

    def isToeplitzMatrix(self, matrix):
        """
        :type matrix: List[List[int]]
        :rtype: bool
        """
        r_num, c_num = len(matrix), len(matrix[0])
        for i in range(c_num):
            if self.isToeplitz(matrix, r_num, c_num, 0, i):
                pass
            else:
                return False

        for i in range(1, r_num):
            if self.isToeplitz(matrix, r_num, c_num, i, 0):
                pass
            else:
                return False

        return True
