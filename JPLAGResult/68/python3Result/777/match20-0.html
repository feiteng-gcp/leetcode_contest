class Solution:
    def isToeplitzMatrix(self, matrix):
        """
        :type matrix: List[List[int]]
        :rtype: bool
        """
        row = len(matrix)
        col = len(matrix[0])
        
        for i in range(col):
            start = matrix[0][i]
            for j in range(1, row):
                if i+j >= col:
                    break
                if matrix[j][i+j] != start:
                    return False
        
        for i in range(1, row):
            start = matrix[i][0]
            for j in range(1, col):
                if i+j >=row:
                    break
                if matrix[i+j][j] != start:
                    return False
        return True
