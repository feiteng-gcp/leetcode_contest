class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int row = matrix.length;
        int col = matrix[0].length;
        
        for (int i = 0; i < col; i++){
            if (!same(matrix, 0, i))
                return false;
        }
        
        for (int i = 0; i < row; i++){
            if (!same(matrix, i, 0))
                return false;
        }
        return true;
    }
    
    public boolean same(int[][] matrix, int r, int c){
        int e = matrix[r][c];
        while(r < matrix.length && c < matrix[0].length){
            if (matrix[r][c] != e)
                return false;
            r++;
            c++;
        }
        return true;
    }
}
