class Solution {
  public boolean isToeplitzMatrix(int[][] matrix) {
    int M = matrix.length;
    int N = matrix[0].length;

    for (int idxCol = 0; idxCol < N; idxCol++) {
      for (int idxRow = 0; idxRow < 1; idxRow++) {
        int v = matrix[idxRow][idxCol];
        for (int i = 1; i + idxRow < M && i + idxCol < N; i++) {
          if (v != matrix[i + idxRow][i + idxCol]) {
            return false;
          }
        }
      }
    }

    for (int idxCol = 0; idxCol < 1; idxCol++) {
      for (int idxRow = 1; idxRow < M; idxRow++) {
        int v = matrix[idxRow][idxCol];
        for (
int i = 1; i + idxRow < M && i + idxCol < N; i++) {
          if (v != matrix[i + idxRow][i + idxCol]) {
            return false;
          }
        }
      }
    }

    return true;
  }
}
