class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        for (int i = 0; i < matrix[0].length; i++) {
            int toComp = matrix[matrix.length - 1][i];
            int j = 1;
            while (matrix.length - 1 - j >= 0 && i - j >= 0) {
                if (toComp != matrix[matrix.length - 1 - j][i - j]) return false;
                j++;
            }
        }
        for (int i = 0; i < matrix[0].length; i++) {
            int toComp = matrix[0][i];
            int j = 1;
            while (j < matrix.length && i + j < matrix[0].length) {
                if (toComp != matrix[j][i + j]) return false;
                j++;
            }
        }
        return true;
    }
}
