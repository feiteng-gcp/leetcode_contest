class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int N = matrix.length;
        int M = matrix[0].length;
        for (int i = 1; i < N; i++) {
            for (int j = 0; j < M-1; j++) {
                if (matrix[i-1][j] != matrix[i][j+1]) {
                    return false;
                
}
            }
        }
        return true;
    }
}
