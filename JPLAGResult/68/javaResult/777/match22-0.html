class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int n = matrix.length, m = matrix[0].length;
        for(int d = n-1; d >= 1-m; d--) {
            for(int x = Math.max(1+d,1), y = x-d; x < n && y < m; x++, y++) {
                if (matrix[x][y] != matrix[x-1][y-1])
                    return false;
            }
        }
        return true;
    }
}
