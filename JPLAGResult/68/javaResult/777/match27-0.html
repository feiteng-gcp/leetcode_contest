class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
         int R = matrix.length;
        int C = matrix[0].length;
        for (int i=C-2; i>=0;i--)
        {
            int c = matrix[0][i];
            for (int j= 1; j<R & i+j<C; j++)
            {
                if (matrix[j][i+j] != c)
                    return false;
            }
        }
       for (int i=1; i<R-1;i++)
        {
            int c = matrix[i][0];
            for (int j= 1; i+j<R & j<C; j++)
            {
                if (matrix[i+j][j] != c)
                    return false;
            }
        }
       return true;
    }
}
