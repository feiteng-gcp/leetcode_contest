class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int rows = matrix.length;
        int col = matrix[0].length;
        
        for (int i = 0; i < rows; i ++)
        {
            for(int j = 0; j < col; j ++)
            {
                int current = matrix[i][j];
                for(int k = 1 + i, l = 1 + j; k < rows && l < col; k ++, l ++)
                {
                    
                        if(current != matrix[k][l])
                        {
                            return false;
                        }
                    
                }
            }
        }
        return true;
    }
    
}
