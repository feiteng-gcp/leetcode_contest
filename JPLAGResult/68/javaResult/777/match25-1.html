class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int numRows = matrix.length;
        int numCols = matrix[0].length;
        for(int i = 0 ; i < numCols ; ++i) {
            int j = 0;
            int k = i;
            int num = matrix[j][i];
            while(j < numRows && k < numCols) {
                if(matrix[j][k] != num) {
                    return false;
                }
                j++;
                k++;
            } 
        }
        
        for(int i = 0 ; i < numRows ; ++i) {
            int j = 0; // this time col
            int k = i; // k is row
            int num = matrix[k][j];
            while(k < numRows && j < numCols) {
                if(matrix[k][j] != num) {
                    return false;
                }
                j++;
                k++;
            }
        }
        return true;
    }
}
