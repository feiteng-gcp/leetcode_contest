class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        
        for(int i = 0;i<matrix.length;i++){
            for(int j = 0;j<matrix[0].length;j++){
                if(i<1 || j < 1) continue;
                if(matrix[i-1][j-1]!=matrix[i][j]) 
return false;
            }
        }
        return true;
    }
}
