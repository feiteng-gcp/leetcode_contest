class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        if (matrix.length == 0 || matrix[0].length == 0) {
            return true;
        }
        int m = matrix.length, n = matrix[0].length;
        for (int j = 0; j < n; j++) {
            int num = matrix[0][j];
            for (int i = 1, k = j + 1; i < m && k < n; i++, k++) {
                if (num != matrix[i][k]) {
                    return false;
                }
            }
        }
        
        for (int row = 1; row < m; row++) {
            int num = matrix[row][0];
            for (int i = row, k = 0; i < m && k < n; i++, k++) {
                if (num != matrix[i][k]) {
                    return false;
                }
            }
        }
        
        return true;
    }
}
