class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int m = matrix.length, n = matrix[0].length;
        
        for (int i = 0; i < m; i++) {
            int v = matrix[i][0];
            for (int j = 1; j < n && (i + j) < m; j++) {
                if (matrix[i + j][j] != v) return false;
            }
        }
        
        for (int j = 0; j < n; j++) {
            int v = matrix[0][j];
            for (int i = 1; i < m && (i + j) < n; i++) {
                if (matrix[i][i+j] != v) return false;
            }
        }
        
        return true;
    }
}
