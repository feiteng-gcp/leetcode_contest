class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int h = matrix.length;
        int w = matrix[0].length;
        for(int i = 0; i < w; i ++)
        {
            int val = matrix[0][i];
            for(int j = 1; j < h && i + j < w; j ++) { 
                if(matrix[j][i+j] != val) return false;
            }            
        }
        for(int i = 1; i < h; i ++)
        {
            int val = matrix[i][0];
            int index = 1;
            for(int j = i + 1; j < h && index < w; j ++)
            {
                if(matrix[j][index ++] != val) return false; 
            }
        }
        return true;
    }
}
