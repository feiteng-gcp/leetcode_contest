class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        if (matrix.length == 0 || matrix[0].length == 0) {return true;}
        
for (int i = 0; i < matrix[0].length; ++i) {
            int n = matrix[0][i];
            for(int j = 0, k = i; j < matrix.length && k < matrix[0].length; ++k, ++j) {
                if (matrix[j][k] != n) { return false;}
            }
        }
        for (int i = 0; i < matrix.length; ++i) {
            int n = matrix[i][0];
            for(int j = i, k = 0; j < matrix.length && k < matrix[0].length; ++k, ++j) {
                if (matrix[j][k] != n) { return false;}
            }
        }
        return true;
    }
}
