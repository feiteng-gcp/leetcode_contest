class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int n = matrix.length;
        int m = matrix[0].length;
        for (int k = n - 1; k >= 1 - m; k--) {

            for (int j = Math.max(1, 1 - k); j < m && j + k < n; j++) {
                if (matrix[j + k][j] != matrix[j + k - 1][j - 1]) {
                    return false;
                
}
            }

        }
        return true;
    }
}
