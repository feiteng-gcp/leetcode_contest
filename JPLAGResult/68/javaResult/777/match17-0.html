class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        // row number
        for (int i = 0; i < matrix.length; i++) {
            int curr_x = i;
            int curr_y = 0;
            while (curr_x + 1 < matrix.length && curr_y + 1 < matrix[0].length) {
                if (matrix[curr_x][curr_y] != matrix[curr_x+1][curr_y+1])
                    return false;
                curr_x++;
                curr_y++;
            }
        }
        
        // col number
        for (int i = 0; i < matrix[0].length; i++) {
            int curr_x = 0;
            int curr_y = i;
            while (curr_x + 1 < matrix.length && curr_y + 1 < matrix[0].length) {
                if (matrix[curr_x][curr_y] != matrix[curr_x+1][curr_y+1])
                    return false;
                curr_x++;
                curr_y++;
            }
        }
        
        return true;
    }
}
