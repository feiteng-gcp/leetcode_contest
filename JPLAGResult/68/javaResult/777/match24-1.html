class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int sz1 = matrix.length;
        int sz2 = matrix[0].length;
        for(int i = 0; i < sz2; i++) {
            if(!check(0, i, matrix)) return false;
        }
        for(int i = 0; i < sz1; i++) {
            if(!check(i, 0, matrix)) return false;
        }
        return true;
    }
    private boolean check(int row, int col, int[][] matrix) {
        int num = matrix[row][col];
        while(row < matrix.length && col < matrix[0].length) {
            if(num != matrix[row][col]) return false;
            row++;
            col++;
        }
        return true;
    }
}
