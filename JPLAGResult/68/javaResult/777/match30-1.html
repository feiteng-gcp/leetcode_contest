class Solution {
    public boolean isToeplitzMatrix(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        for (int i = 0; i < n; ++i) {
            int temp = matrix[0][i];
            for (int d = 0; d + i < n && d < m; ++d) {
                if (matrix[d][i + d] != temp) {
                    return false;
                }
            }
        }
        for (int j = 1; j < m; ++j) {
            int temp = matrix[j][0];
            
for (int d = 0; d < n && d + j < m; ++d) {
                if (matrix[j + d][d] != temp) {
                    return false;
                }
            }
        }
        return true;
    }
}
