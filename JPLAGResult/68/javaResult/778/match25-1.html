class Solution {
    public String reorganizeString(String S) {
        int[] map = new int[26];
        int[] dist = new int[26];
        for (int i = 0; i < S.length(); i++)
            map[S.charAt(i) - 'a']++;
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < S.length(); i++) {
            int idx = next(map, dist, i);
            if (idx == -1)
                return "";
            map[idx]--;
            dist[idx] += 2;
            sb.append((char)(idx + 'a'));
        }
        return sb.toString();
    }
    
    public int next(int[] map, int[] dist, int pos) 
{
        int ans = 0, idx = -1;
        for (int i = 0; i < 26; i++) {
            if (map[i] > ans && dist[i] <= pos) {
                ans = map[i];
                idx = i;
            }
        }
        return idx;
    }
}
