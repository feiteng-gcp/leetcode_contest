class Solution {
    public String reorganizeString(String S) {
        int[] rs = new int[26];
        int maxnum = 0;
        for(int i = 0;i<S.length();i++){
            rs[S.charAt(i)-'a'] ++;
            maxnum = Math.max(rs[S.charAt(i)-'a'],maxnum);
        }
        if(maxnum > (S.length()+1)/2) return "";
        StringBuilder s = new StringBuilder();
        int lasti = -1,maxi=-1;
        while(s.length()!=S.length()){
            int n = 0;
            for(int i = 0;i<26;i++){
                if(rs[i]==0 || i==lasti) continue;
                if(n<=rs[i]) {n=rs[i];maxi = i;}
            }
            lasti=maxi;
            char aa = (char)('a'+maxi);
            s.append(Character.toString(aa));
            rs[maxi]--;
        }
        return s.toString();
    }
}
