class Solution {
      class Node {
        char c;
        int count;
        int avail;
        Node(char c, int cnt, int a) {
            count = cnt; avail = a; this.c = c;
        }
    }

    public String reorganizeString(String S) {
        if (S == null || S.length() <= 1) return S;
        Node[] counter = new Node[26];
        for (int i = 0; i < 26; i++) counter[i] = new Node((char)(i + 'a'), 0, 0);

        for (char c: S.toCharArray()) {
            counter[c - 'a'].count += 1;
        
}

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < S.length(); i++) {
            int idx = getChar(counter, i);
            if (idx == -1) return "";

            counter[idx].count -= 1;
            counter[idx].avail = i + 2;
            
sb.append(counter[idx].c);
        }

        return sb.toString();
    }

    private int getChar(Node[] counter, int cur) {
        Arrays.sort(counter, (c1, c2) -> c2.count - c1.count);
        for (int i = 0; i < 26; i++) {
            if (counter[i].count > 0 && counter[i].avail <= cur) {
                return i;
            }
        }
        return -1;
    }
}
