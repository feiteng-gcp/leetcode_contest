class Solution {
    public String reorganizeString(String S) {
        HashMap<Character,Integer> mp = new HashMap<>();
        for(char c: S.toCharArray())
        {
            mp.put(c,mp.getOrDefault(c,0)+1);
        }
        if(mp.keySet().size()==1)return "";
        StringBuilder sb = new StringBuilder();
        for(;;)
        {
            int len = sb.length();
            int t=0;
            
            //find bigtwo
            char c1 = '0';
            char c2 = '0';
            int min1=0;
            int min2=0;
            for(Character c:mp.keySet())
            {
                int count = mp.get(c);
                if(count>0){
                if(count>min1){c2=c1;c1=c;min2=min1;min1=count;}
                else if(count>min2){c2=c;min2=count;}
                }
            }
            if(c1!='0'){sb.append(c1);                
            mp.put(c1,mp.get(c1)-1); }
            if(c2!='0'){sb.append(c2);                
            mp.put(c2,mp.get(c2)-1); }
            
           /* for(Character c: mp.keySet())
            {
                int count = mp.get(c);
                if(count>0)
                {
                sb.append(c);                
                mp.put(c,count-1);
                t++;
                if(t==2)break;
                }
            }*/
            if(len==sb.length())return sb.toString();
            if(sb.charAt(sb.length()-2)==sb.charAt(sb.length()-1))break;
        }
        return "";
    }
}
