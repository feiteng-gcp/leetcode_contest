class Solution {
    
    class Key {
        int freq; // store frequency of character
        char ch;

        // function for priority_queue to store Key
        // according to freq
        public boolean compare(Key k) {
            return (this.freq < k.freq);
        }
        public Key(int freq, char ch) {
            this.freq = freq;
            this.ch = ch;
        }
    };
    
    public String reorganizeString(String S) {
        int n = S.length();
        
        int[] count = new int[26];
        
        for (int i = 0; i < n; i++) {
            count[S.charAt(i) - 'a']++;
        }
        
        PriorityQueue<Key> pq = new PriorityQueue<>(100, new Comparator<Key>(){
            public int compare(Key k1, Key k2) {
                return k2.freq - k1.freq;
            }
        });
        for (char c = 'a'; c <= 'z'; c++) {
            if (count[c - 'a'] > 0) {
                pq.offer(new Key(count[c - 'a'], c));
            }
        }
        
        StringBuilder sb = new StringBuilder();
        
        Key prev = new Key(-1, '#');
        while (!pq.isEmpty()) {
            Key k = pq.poll();
            sb.append(k.ch);
            
            if (prev.freq > 0) {
                pq.offer(prev);
            }
            
            k.freq--;
            prev = k;
        }
        //System.out.println(sb.toString());
        if (sb.length() != n) {
            return "";
        } else {
            return sb.toString();
        }
    }
}
