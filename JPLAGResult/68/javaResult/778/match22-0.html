class Solution {
    public String reorganizeString(String S) {
        // heap
        int[] map = new int[26];
        for (char c : S.toCharArray()) {
            map[c - 'a']++;
        }
        StringBuilder sb = new StringBuilder();
        PriorityQueue<Character> pq = new PriorityQueue<>(new Comparator<Character>(){
            public int compare(Character a, Character b) {
                return map[b - 'a'] - map[a - 'a'];
            }
        });
        for (int i = 0; i < 26; i++) {
            if (map[i] != 0) {
                pq.add((char) ('a' + i));
            }
        }
        Character last = null;
        while (sb.length() < S.length()) {
            if (last == null || last != pq.peek()) {
                char c = pq.poll();
                map[c - 'a']--;
                if (map[c - 'a'] != 0) {
                    pq.add(c);
                }
                sb.append(c);
            } else {
                char large = pq.poll();
                if (pq.peek() == null) {
                    return "";
                }
                char c = pq.poll();
                pq.add(large);
                map[c - 'a']--;
                if (map[c - 'a'] != 0) {
                    pq.add(c);
                }
                sb.append(c);
            }
            last = sb.charAt(sb.length() - 1);
        }
        return sb.toString();
    }
}
