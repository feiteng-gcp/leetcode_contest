class Solution {
    public String reorganizeString(String S) {
        int[] letters = new int[26];
        for(int i = 0; i < S.length(); i++) {
            letters[S.charAt(i)- 'a']++;
        }
        
        int max = 0;
        char maxChar = 'a';
        for(int i = 0; i < 26; i++) {
            
if(letters[i] > max) {
                max = letters[i];
                maxChar = (char)('a' + i);
            }
        }
        
        if(max - 1> S.length() - max) return "";
        
        List<StringBuilder> sbs = new ArrayList<>();
        
        for(int i = 0; i < max; i++) {
            sbs.add(new StringBuilder().append(maxChar));
        }
        
        letters[maxChar - 'a'] = 0;
        
        int rest = S.length() - max;
        
        int curIndex = 0;
        
        for(;rest > 0; rest--) {
            for(int i = 0; i < 26; i++) {
                if(letters[i] != 0) {
                    sbs.get(curIndex).append((char)('a' + i));
                    curIndex = (curIndex + 1) % max;
                    letters[i]--;
                    break;
                }
            }
        }
        
        StringBuilder result = new StringBuilder();
        
        for(int i = 0; i < sbs.size(); i++) {
            result.append(sbs.get(i).toString());
        }
                              
        return result.toString();
    }
}
