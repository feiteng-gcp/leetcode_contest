class Solution {
    public String reorganizeString(String S) {
        int[] counts = new int[26];
        
        for (char ch : S.toCharArray()) {
            counts[ch - 'a']++;
        }
        
        StringBuilder sb = new StringBuilder();
        char last = '#';
        for (int i = 0; i < S.length(); i++) {
            int max = 0;
            char best = '#';
            for (char ch = 'a'; ch <= 'z'; ch++) {
                if (ch == last) continue;
                if (counts[ch - 'a'] > max) {
                    max = counts[ch - 'a'];
                    best = ch;
                }
            }
            if (max == 0) {
                return "";
            }
            counts[best - 'a']--;
            sb.append(best);
            last = best;
        }
        
        return sb.toString();
    }
}
