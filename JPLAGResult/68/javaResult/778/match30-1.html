class Solution {
    public String reorganizeString(String S) {
        if (S.length() < 2) {
            return S;
        }
        if (S.length() == 2) {
            if (S.charAt(0) == S.charAt(1)) {
                return "";
            } else if (S.charAt(0) < S.charAt(1)) {
                return S;
            } else {
                return S.charAt(1) + "" + S.charAt(0);
            }
        }
        //String s = reorganizeString.substring(1);  
        
int[] cnt = new int[256];
        for (char c : S.toCharArray()) {
            cnt[c]++;
        }
        char mc = 'a';
        for (char c = 'a'; c <= 'z'; c++) {
            if (cnt[c] > cnt[mc]) {
                mc = c;
            }
        }
        List<List<Character>> lists = new ArrayList<>();
        for (int i = 0; i < cnt[mc]; i++) {
            lists.add(new ArrayList<Character>());
            lists.get(lists.size() - 1).add(mc);
            //cnt[mc]--;
        }
        int r = 0;
        for (char c = 'a'; c <= 'z'; c++) {
            while (c != mc && cnt[c] > 0) {
                lists.get(r).add(c);                
                cnt[c]--;
                r = (r + 1) % lists.size();
            }
        }
        if (lists.size() > 1 && lists.get(lists.size() - 2).size() == 1) {
            return "";
        }
        StringBuilder sb = new StringBuilder();
        for (List<Character> l : lists) {
            for (Character c : l) {
                sb.append(c);
            }
        }
        return sb.toString();
    }
}
