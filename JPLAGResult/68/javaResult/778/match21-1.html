class Solution {
    int[] chs=new int[26];
    char prev='@';
    StringBuilder sb=new StringBuilder();
    public String reorganizeString(String S) {
        for(char x:S.toCharArray()) {
            ++chs[x-'a'];
        }
        helper();
        return sb.length()==S.length()?sb.toString():"";
    }
    void helper() {
        char next='#';
        int cnt=0;
        for(char x='a';x<='z';++x) {
            if(x==prev) continue;
            
if(chs[x-'a']>=cnt) {
                cnt=chs[x-'a'];
                next=x;
            }
        }
        if(cnt==0) return;
        --chs[next-'a'];
        sb.append(next);
        prev=next;
        helper();
    }
}
