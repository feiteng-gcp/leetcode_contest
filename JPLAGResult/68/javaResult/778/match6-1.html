class Solution {
    class Node{
        public char c;
        public int count;
        public Node(char c, int count){
            this.c=c;
            this.count=count;
        }
    }
    
public String reorganizeString(String S) {
          if(S==null||S.length()<2){
	            return S;
	        }       
	        PriorityQueue<Node> heap=new PriorityQueue<Node>((a, b)->(b.count-a.count));
	        int[] m
ap=new int[26];
	        for(char c:S.toCharArray()){
	            map[c-'a']++;
	        }
	        int len=S.length();
	        for(int i=0;i<26;i++){
	            if(map[i]==0){
	            	continue;
	            }
	        	if((len%2==1&&map[i]>(len/2+1))||(len%2==0&&map[i]>len/2)){
	                return "";
	            }
	            if(map[i]>0){
	                heap.offer(new Node((char)('a'+i), map[i]));
	            }
	        }
	        Map<Integer, Node> dist=new HashMap<>();
	        int i=0;
	        StringBuilder sb=new StringBuilder();
	        while(!heap.isEmpty()||!dist.isEmpty()){
	            if(dist.containsKey(i-2)){
	                heap.offer(dist.get(i-2));
	                dist.remove(i-2);
	            }
	            Node node=heap.poll();
	            sb.append(node.c);
	            node.count--;
	            if(node.count>0){
	                dist.put(i, node);
	            }
	            i++;
	        }
	        return sb.toString();
    }
}
