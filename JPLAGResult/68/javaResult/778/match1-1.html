class Solution {
    public String reorganizeString(String S) {
        if(S == null || S.length() == 0){
            return S;
        }
        
        int[] map = new int[26];
        
int[] id = new int[26];
        
        for(char ch : S.toCharArray()){
            map[ch - 'a']++;
        }
        StringBuilder sb = new StringBuilder();
        
for(int i = 0; i < S.length(); i++){
            
            int pick = -1;
            int max = 0;
            
            for(int j = 0; j < 26; j++){
                
if(map[j] > max && id[j] <= i){
                    pick = j;
                    max = map[j];
                }
            }
            if(pick == -1){
                return "";
            }
            else{
                sb.append((char)('a' + pick));
                map[pick]--;
                id[pick] = i + 2;
            }
        }
        
        return sb.toString();
    }
}
