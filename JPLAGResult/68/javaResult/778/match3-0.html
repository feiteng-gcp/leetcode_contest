class Solution {
    public String reorganizeString(String S) {
        int[] count = new int[26];
        for (char c : S.toCharArray()) {
            count[c - 'a']++;
        }
        Queue<int[]> pq = new PriorityQueue<>((a1, a2) -> a2[1] - a1[1]);
        for (int i = 0; i < 26; i++) {
            if (count[i] > 0) {
                pq.offer(new int[] {i, count[i]});
            }
        }
        int noGood = -1;
        StringBuilder res = new StringBuilder();
        while (!pq.isEmpty()) {
            int[] cur = pq.poll();
            if (cur[0] == noGood) {
                if (pq.isEmpty()) return "";
                int[] next = pq.poll();
                pq.offer(cur);
                cur = next;
            }
            noGood = cur[0];
            res.append((char) (cur[0] + 'a'));
            if (cur[1] > 1)
                pq.offer(new int[] {cur[0], cur[1] - 1});
        }
        return res.toString();
    }
}
