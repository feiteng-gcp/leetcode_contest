class Solution {
    public int maxChunksToSorted(int[] arr) {
        int n = arr.length;
        int i = 0;
        int res = 0;
        while (i < n) {
            int j = i;
            
            while (i <= j) {
                if (arr[i] > j) {
                    j = arr[i];
                }
                i++;
            }
            res++;
        }
        
        return res;
    }
}
