class Solution {
    public int maxChunksToSorted(int[] arr) {
        int count=0,high=arr[0];
        for(int i=0;i<arr.length;i++){
            high=Math.max(high,arr[i]);
            if(i==high){
                count++;
            }
            
        }
        return count;
    }
}
