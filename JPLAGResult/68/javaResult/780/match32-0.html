class Solution {
    public int maxChunksToSorted(int[] arr) {
        int[] dp = new int[arr.length];
        Arrays.fill(dp, -1);
        return dfs(arr, 0, dp);
    }
    
    private int dfs(int[] arr, int firstIndex, int[] dp) {
        if (firstIndex == arr.length) {
            return 0;
        }
        if (dp[firstIndex] != -1) {
            return dp[firstIndex];
        }
        int ret = 1;
        int max = -1;
        for (int i = firstIndex; i < arr.length; i++) {
            max = Math.max(arr[i], max);
            if (max == i) {
                ret = Math.max(ret, 1 + dfs(arr, i + 1, dp));
            }
        }
        return dp[firstIndex] = ret;
    }
}
