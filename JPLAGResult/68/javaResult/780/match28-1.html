class Solution {
    public int maxChunksToSorted(int[] arr) {
        if(arr.length < 2)  return arr.length;
        int ans = 0;
        int max = -1;
        for(int i = 0; i < arr.length; ++i) {
            if(max == -1 || max < arr[i]) {
                max = arr[i];
            }
            
if(i == max) {
                ++ans;
                max = -1;
            }
        }
        return ans;
    }
}
