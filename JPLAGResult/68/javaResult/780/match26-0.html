class Solution {
    public int maxChunksToSorted(int[] arr) {
        int result = 0;
        
        int i = 0, j = -1;
        
        while(i < arr.length) {
            j = Math.max(j, arr[i]);
            if(i == j) result++;
            i++;
        }
        
        return result;
    }
}
