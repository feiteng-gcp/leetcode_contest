class Solution {
    public int maxChunksToSorted(int[] arr) {
        
int res = 0, upper = 0;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] > upper)
                upper = arr[i];
            if (i == upper)
                res++;
        }
        return res;
    }
}
