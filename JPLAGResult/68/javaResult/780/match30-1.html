class Solution {
    public int maxChunksToSorted(int[] arr) {
        int chunks = 1;
        
        int N = arr.length;
        int[] maxArr = new int[N];
        int[] minArr = new int[N];
        
        int max = -1;
        for (int i = 0; i < N; i++) {
            max = Math.max(max, arr[i]);
            maxArr[i] = max;
        }
        
        int min = N + 1;
        for (int i = N - 1; i >=0; i--) {
            min = Math.min(min, arr[i]);
            minArr[i] = min;
        }
        
        for (int i = 0; i < N - 1; i++) {
            
if (maxArr[i] < minArr[i+1]) {
                chunks++;
            }
        }
        
        return chunks;
    }
}
