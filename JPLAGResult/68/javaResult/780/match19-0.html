class Solution {
    public int maxChunksToSorted(int[] arr) {
        int result = 0;
        Set<Integer> value = new HashSet();
        Set<Integer> index = new HashSet();
        for (int i = 0; i < arr.length; ++i) {
            value.add(arr[i]);
            index.add(i);
            if(value.containsAll(index) && index.containsAll(value)) {
                result ++;
                value = new HashSet();
                index = new HashSet();
            }
        }
        return result;
    }
}
