class Solution {
    public int maxChunksToSorted(int[] arr) {
        
        int N = arr.length;
        if (N == 0 || N == 1) return N;
        
        int max = arr[0];
        int count = 0;
        
        for (int i = 0; i < N; i++)
        {
            if (arr[i] > max) max = arr[i];
            if (max <= i) {count++; max = 0; }
            
        }
        return count;
        
    }
}
