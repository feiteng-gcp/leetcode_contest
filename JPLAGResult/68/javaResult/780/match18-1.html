class Solution {
    public int maxChunksToSorted(int[] arr) {
        if (arr == null || arr.length == 0) {
            return 1;
        }
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < arr.length; i++) {
            map.put(arr[i], i);
        }
        
        
int count = 0, index = 0;
        
        while (index < arr.length) {
            int len = index;
            for (int i = 0; i <= len; i++) {
                len = Math.max(len, map.get(i));
            
}
            count++;
            index = len + 1;
        }
        
        return count;
    }
}
