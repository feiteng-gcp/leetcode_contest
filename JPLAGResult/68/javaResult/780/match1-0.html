class Solution {
    public int maxChunksToSorted(int[] arr) {
        int chunks = 0;
        int chunkIndex = 0;
        
        for (int i = 0; i < arr.length; i++) {
            chunkIndex = Math.max(chunkIndex, arr[i]);
            if (i == chunkIndex)
                chunks++;
        }
        
        return chunks;
    }
}
