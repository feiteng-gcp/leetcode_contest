class Solution {
    public int maxChunksToSorted(int[] arr) {
        if(arr==null||arr.length==0){
            return 0;
        }       
        int len=arr.length;
        int[] dp=new int[len];
        int[] temp=new int[len];
        for(int i=0;i<arr.length;i++){
            temp[i]=arr[i];
        }
        Arrays.sort(temp);
        Map<Integer, List<Integer>> map=new HashMap<>();
        for(int i=0;i<temp.length;i++){
            if(!map.containsKey(temp[i])){
                map.put(temp[i], new ArrayList<Integer>());
            }
            map.get(temp[i]).add(i);
        }
        int index=0;
        for(int i=0;i<arr.length;i++){
            List<Integer> list=map.get(arr[i]);
            dp[index++]=
list.get(0)-i;
            list.remove(0);
        }
        
        int res=0;
        int count=0;
        int i=0;
        while(i<arr.length){
            if(dp[i]==0){
                if(count==0){
                    res++;
                }
            }else{
                count+=dp[i];
                if(count==0){
                    res++;
                }
            }
            i++;
        }
        return res;
    }
}
