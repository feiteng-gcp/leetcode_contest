class Solution {
    public int maxChunksToSorted(int[] arr) {
        if (arr == null || arr.length == 0) {
            return 0;
        }
        int n = arr.length;
        int[] helper = new int[n];
        
        for (int i = 0; i < n; i++) {
            helper[i] = arr[i];
        }
        Arrays.sort(helper);
        
        Map<Integer, Queue<Integer>> map = new HashMap<>();
        
        for (int i = 0; i < n; i++) {
            map.putIfAbsent(arr[i], new LinkedList<>());
            map.get(arr[i]).offer(i);
        }
        
        int count = 0, index = 0;
        
        while (index < n) {
            int len = index;
            for (int i = index; i <= len; i++) {
                len = Math.max(len, map.get(helper[i]).poll());
                
            }
            count++;
            index = len + 1;
        }
        
        return count;
    }
}
