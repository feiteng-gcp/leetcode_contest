class Solution {
    public int maxChunksToSorted(int[] arr) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        int res = 0, max = -1;
        for (int num : arr)
            map.put(num, map.getOrDefault(num, 0) + 1);
        for (int num : arr) {
            max = Math.max(max, num);
            int count = map.get(num);
            if (count > 1)
                map.put(num, count - 1);
            else
                map.remove(num);
            if (map.size() == 0 || map.firstKey() >= max) 
{
                res++;
                max = -1;
            }
        }
        return res;
    }
}
