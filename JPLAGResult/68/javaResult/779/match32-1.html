class Solution {
    public int maxChunksToSorted(int[] a) {
        int n = a.length;
        int[] leftmostPos = new int[n];
        
        leftmostPos[0] = -1;
        for (int i = 1; i < n; ++i) {
            leftmostPos[i] = -1;
            for (int j = 0; j < i; ++j) {
                if (a[i] < a[j]) {
                    leftmostPos[i] = j;
                    break;
                }
            }
        }
        
        //System.out.println(Arrays.toString(leftmostPos));
        
        List<Interval> intervals = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            if (leftmostPos[i] == -1) {
                intervals.add(new Interval(i, i));
            } else {
                intervals.add(new Interval(leftmostPos[i], i));
            }
        }
        
        Collections.sort(intervals);
        int cnt = 1;
        
        Interval cur = new Interval(intervals.get(0).l, intervals.get(0).r);
        for (int i = 1; i < n; ++i) {
            Interval interval = intervals.get(i);
            if (interval.l <= cur.r) {
                //System.out.println("1: " + interval.l + " " + interval.r);
                cur.r = Math.max(interval.r, cur.r);
            } else {
                ++cnt;
                //System.out.println("2: " + interval.l + " " + interval.r);
                cur = new Interval(interval.l, interval.r);
            }
        }
        
        return cnt;
    }
    
    static class Interval implements Comparable<Interval> {
        public int l, r;
        
        public Interval(int l, int r) {
            this.l = l;
            this.r = r;
        }
        
        @Override
        public int compareTo(Interval i2) {
            if (this.l < i2.l)
                return -1;
            if (this.l > i2.l)
                return 1;
            return 0;
        }
    }
}
