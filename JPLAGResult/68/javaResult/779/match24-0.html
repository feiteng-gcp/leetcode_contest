class Solution {
    public int maxChunksToSorted(int[] arr) {
        int sol = 1;
        int n = arr.length;
        int[] min = new int[n];
        int[] max = new int[n];
        max[0] = arr[0];
        min[n-1] = arr[n-1];
        for(int i=1; i<n; i++ ){
            max[i] = Math.max(max[i-1], arr[i]);
            min[n-i-1] = Math.min(min[n-i], arr[n-i-1]);
        }
        int i = 0;
        while(i < n-1){
            if(max[i] <= min[i+1]){
                sol++;
            }
            i++;
        }
        
        return sol;
    }
}
