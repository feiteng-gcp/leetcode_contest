class Solution {
    public int maxChunksToSorted(int[] arr) {
        int[] preMax = new int[arr.length];
        int prev = 0;
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] < arr[prev]) {
                preMax[i] = prev;
            } else {
                preMax[i] = i;
            }
            prev = preMax[i];
        
}
        int count = 0;
        int i = arr.length - 1;
        while (i >= 0) {
            int index = preMax[i];
            while (index > 0 && arr[preMax[index - 1]] > arr[i]) {
                index = preMax[index - 1];
            }
            int min = arr[i];
            while (i >= index) {
                if (arr[i] < min) {
                    min = arr[i];
                    while (index > 0 && arr[preMax[index - 1]] > min) {
                        index = preMax[index - 1];
                    }
                
}
                i--;
            }
            count++;
        }
        return count;
    }
}
