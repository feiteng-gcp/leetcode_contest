class Solution {
    class Node{
        int freq;
        int val;
        Node(int val){
            this.val = val;
            freq = 1;
        }
    }
    public int maxChunksToSorted(int[] arr) {
        if(arr == null || arr.length == 0){
            return 0;
        }
        int len = arr.length;
        Map<Integer, Node> map = new HashMap<>();
        PriorityQueue<Node> pq = new PriorityQueue<Node>(1, new Comparator<Node>(){
           public int compare(Node n1, Node n2){
               return Integer.compare(n1.val, n2.val);
           } 
        });
        
        for(int num : arr){
            if(map.containsKey(num)){
                Node node = map.get(num);
                node.freq++;
            }    
            else{
                Node node = new Node(num);
                map.put(num, node);
                pq.offer(node);
            }
        
}
        
        int max = -1;
        int res = 0;
        for(int num : arr){
            // System.out.println(pq.size());
            max = Math.max(max, num);
            Node node = map.get(num);
            node.freq--;
            if(node.freq == 0){
                pq.remove(node);
            }
            if(pq.isEmpty() || pq.peek().val >= max){
                res++;
            }
        }
        return res;
    }
}
