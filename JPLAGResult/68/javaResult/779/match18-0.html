class Solution {
    public int maxChunksToSorted(int[] arr) {
        if(arr == null || arr.length == 0) {
            return 0;
        }
        
        int n = arr.length;
        int[] copy = new int[n];
        HashMap<Integer, Queue<Integer>> map = new HashMap<>();
        for(int i =0; i < n; i++) {
            copy[i] = arr[i];
            if(map.get(arr[i]) == null) {
                Queue<Integer> queue = new LinkedList<Integer>();
                map.put(arr[i], queue);
            }
        }
        
        Arrays.sort(copy);
        for(int i = 0; i < n; i++) {
            map.get(copy[i]).offer(i);
            //System.out.format("copy[i] = %d", copy[i]);
            //System.out.println(map.get(copy[i]).toString());
        }
        
        int max = -1;
        int i = 0;
        int count = 0;
        while(i < n) {
            //System.out.format("arr[i] = %d, i = %d", arr[i], i);
            Queue<Integer> queue = map.get(arr[i]);
            int pos = queue.poll();
            //System.out.format("pos = %d\n", pos);
            if(pos == i) {
                count++;
                i++;
            }
            else { // pos > i
                max =  pos;
                for(int j = i+1; j <= max; j++) {
                    int jpos = map.get(arr[j]).poll();
                    if(jpos > max) {
                        max = jpos;
                    }
                }
                
                count++;
                i = max+1;
                max = -1;
            }
        }
        
        return count;
    }
}
