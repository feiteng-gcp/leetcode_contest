class Solution {
    public int maxChunksToSorted(int[] arr) {
        int c = 0;
        int end = 0;
        int i;
        for (i = 0; i < arr.length; i++) {
            int aim = arr[i];
            boolean small = false;
            for (int x = i + 1; x < arr.length; x++) {
                if (aim > arr[x]) {
                    small = true;
                    end = Math.max(x,end);
                }
            }
            if (!small) {
                end = Math.max(i,end);
            }
            if (i >= end) {
                c++;
            }
        }
        return c;
    }

}
