class Solution {
    public int maxChunksToSorted(int[] arr) {
        int b[] = arr.clone();
        Arrays.sort(b);
        int n = arr.length;
        Map<Integer, ArrayDeque<Integer>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (!map.containsKey(b[i])) {
                map.put(b[i], new ArrayDeque<>());
            }
            map.get(b[i]).offer(i);
        }
        //System.out.println(map);
        int i = 0;
        int res = 0;
        while (i < n) {
            int j = i;
            
            while (i <= j) {
                int pos = map.get(arr[i]).poll();
                if (pos > j) {
                    j = pos;
                }
                i++;
            }
            res++;
        }
        
        return res;
    }
}
