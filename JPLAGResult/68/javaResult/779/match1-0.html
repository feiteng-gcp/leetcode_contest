class Solution {
    public int maxChunksToSorted(int[] arr) {
        int[] minToRight = new int[arr.length];
        minToRight[arr.length - 1] = arr[arr.length - 1];
        for (int i = arr.length - 2; i >= 0; i--) {
            minToRight[i] = Math.min(arr[i], minToRight[i + 1]);
        }
        
        int[] dp = new int[arr.length];
        Arrays.fill(dp, -1);
        return dfs(arr, 0, minToRight, dp);
    }
    
    private int dfs(int[] arr, int firstIndex, int[] minToRight, int[] dp) {
        if (firstIndex == arr.length) {
            return 0;
        }
        if (dp[firstIndex] != -1) {
            return dp[firstIndex];
        }
        
        int ret = 1;
        int maxSeen = arr[firstIndex];
        
        for (int i = firstIndex; i < arr.length; i++) {
            maxSeen = Math.max(arr[i], maxSeen);
            
            if (i == arr.length - 1 || maxSeen <= minToRight[i + 1]) {
                ret = Math.max(ret, 1 + dfs(arr, i + 1, minToRight, dp));
            }
        }
        
        return dp[firstIndex] = ret;
    }
}
