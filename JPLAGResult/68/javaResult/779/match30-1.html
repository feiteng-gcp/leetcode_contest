import java.util.*;

class Solution {
    public int maxChunksToSorted(int[] arr) {
        int n = arr.length;
        int[] sorted = arr.clone();
        Arrays.sort(sorted);
        int[] order = new int[n];
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int k = arr[i];
            int last = map.getOrDefault(k, -1);
            if (last < 0) {
                for (int j = 0; j < n; j++) {
                    if (k == sorted[j]) {
                        last = j;
                        break;
                    }
                }
            } else {
                last++;
            }
            order[i] = last;
            map.put(k, last);
        }
        return maxChunksToSorted0(order);
    }

    private int maxChunksToSorted0(int[] arr) {
        int res = 0;
        int n = arr.length;
        int min = n - 1;
        
int max = n - 1;
        int count = 0;
        for (int i = n - 1; i >= 0; i--) {
            min = Math.min(min, arr[i]);
            if (count++ == max - min) {
                res++;
                max = min - 1;
                count = 0;
            } else if (arr[i] != max) {
            }
        }
        return res;        
    }
}
