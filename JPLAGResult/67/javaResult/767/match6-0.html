class Solution {
    public int countPrimeSetBits(int L, int R) {
        int total = 0;

        for (int i = L; i <= R; i++) {
            int bitsCount = countBits(i);
            if (isPrime(bitsCount)) total++;
        }

        return total;
    }

    private boolean isPrime(int bitsCount) {
        if(bitsCount == 1) return false;

        for (int i = 2; i < bitsCount; i++) {
            if(bitsCount%i == 0) return false;
        }
        return true;
    }

    private int countBits(int i) {
        int count = 0;
        while (i > 0) {
            if (i % 2 == 1) count++;
            i /= 2;
        }
        return count;
    }
}
