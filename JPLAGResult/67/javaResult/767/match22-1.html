class Solution {
    public int countPrimeSetBits(int L, int R) {
        Set<Integer> set = new HashSet<>();
        set.add(2);
        set.add(3);
        set.add(5);
        set.add(7);
        set.add(11);
        set.add(13);
        // just in case
        set.add(17);
        set.add(19);

        int res = 0;
        for (int i = L; i <= R; i++) {
            int ones = count(i);
            if (set.contains(ones)) {
                res++;
            
}
        }
        return res;
    }

    private int count(int val) {
        int res = 0;
        while (val > 0) {
            res += (val & 1);
            val >>>= 1;
        }
        return res;
    }
}
