class Solution {
    public int countPrimeSetBits(int L, int R) {
        int countP = 0;
        HashSet<Integer> primeSet = new HashSet<>(Arrays.asList(2, 3, 5, 7, 11, 13, 17, 19));
        
        for(int i = L; i <= R; i++) {
            int temp = countOnes(i);
            if(primeSet.contains(temp)) {
                countP += 1;
            }
        }
        
        return countP;
    }
    
    private int countOnes(int num) {
        int count = 0;
        
        while(num != 0) {
            count += num % 2;
            num /= 2;
        }
        
        return count;
    }
}
