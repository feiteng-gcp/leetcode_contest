class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count=0;
        
        for(int num=L;num<=R;num++){
            int temp=0;
            for(int i=0;;i++){
                int set=1<<i;
                if(set>num){
                    break;
                }
                if((set & num)!=0){
                    temp++;
                }
            }
            
if(isPrime(temp)){
                count++;
            }
        }
        return count;
    }
    private boolean isPrime(int num){
         if(num==2){
             return true;
         }
         if(num>2 && num%2==0 || num<2 ){
             return false;
         }
         for(int i=3;i<=Math.sqrt(num);i++)
{
             if(num%i==0){
                 return false;
             }
         }
        return true;
     }
}
