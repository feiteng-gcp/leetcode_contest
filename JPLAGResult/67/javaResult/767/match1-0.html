class Solution {
    public int countPrimeSetBits(int L, int R) {
		 	int result = 0;
	        for(int i = L;i<=R;i++)
	        {
	        	String str = Integer.toBinaryString(i);
	        	char[] ch = str.toCharArray();
	        	int num = 0;
	        	for(int j = 0;j<ch.length;j++)
	        	{ 		
	        		if(ch[j] == '1')
	        			num++;
	        	}
	        	if(isPrime(num))
	        		result++;
	        }
	        
	        return result;
	    }
	 public static boolean isPrime(int n)
	 {
		 if(n == 1)
			 return false;
		 for(int i = 2;i<=Math.sqrt(n);i++)
		 {
			 if(n%i == 0)
				 return false;
		 }
		 return true;
	 }
}
