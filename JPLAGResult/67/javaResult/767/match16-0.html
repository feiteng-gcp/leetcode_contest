class Solution {
    public int countPrimeSetBits(int L, int R) {
        int c=0;
        for(int i=L;i<=R;i++){
            if(isPrime(binarys(i))){
                c++;
            }
        }
        return c;
    }
    
    public static int binarys(int m){
        int r=0;
        while(m>0){
            if(m%2!=0){
                r++;
            }
            m = m/2;
        }
        return r;
    }
    public static boolean isPrime(int m) {
        if(m==1){
            return false;
        }
        for(int i=2;i<=m/2;i++) {
            if(m%i == 0){
                return false;
            }
        }
        return true;
    }
}
