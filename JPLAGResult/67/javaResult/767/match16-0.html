class Solution {
    public int countPrimeSetBits(int L, int R) {
        int result = 0;
        for(int i = L; i<=R; i++) {
            if(isPrime(i)) {
                result++;
            }
        }
        return result;
    }
    
    public boolean isPrime(int number) {
        int count = 0;
        while(number>0) {
            if(number%2==1) {
                count++;
            }
            number /= 2;
        }
        if(count==1) {
            return false;
        }
        for(int i = 2; i < count; i++) {
            if(count%i==0) {
                return false;
            }
        }
        return true;
    }
}
