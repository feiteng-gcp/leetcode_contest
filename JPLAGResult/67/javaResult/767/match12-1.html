class Solution {
    public int countPrimeSetBits(int L, int R) {
        int res = 0;
        for (int i = L; i <= R; i++) {
            int bits = Integer.bitCount(i);
            if (isPrime(bits)) {
                res++;
            }
        }
        return res;
    }
    
    private boolean isPrime(int n) {
        if (n <= 1 ) {
            return false;
        }
        if (n == 2) {
            return true;
        }
        if (n % 2 == 0) {
            return false;
        }
        for (int i = 3; i <= Math.sqrt(n); i += 2) 
{
            if (n % i == 0) {
                return false;
            }
        }
        return true;
    }
}
