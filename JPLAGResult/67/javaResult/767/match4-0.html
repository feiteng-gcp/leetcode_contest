class Solution {
       public int countPrimeSetBits(int L, int R) {
        int sum = 0;
        for (int i = L; i <= R; i++) {
            int counter = 0;
            String binary = Integer.toBinaryString(i);
            for (int j = 0; j < binary.length(); j++) {
                if (binary.charAt(j) == '1') {
                    counter++;
                }
            }
            if ((counter == 2 || isPrime(counter)) && counter !=1) {
                sum++;
            }
        }
        return sum;

    }

    private static boolean isPrime(int numer){
        for (int i = 2; i < numer ; i++) {
            if (numer % i == 0){
                return false;
            }
        }
        return true;
    }
}
