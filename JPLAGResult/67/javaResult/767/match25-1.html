class Solution {
    public int countPrimeSetBits(int L, int R) {
        int result = 0;
            // System.out.println("--------------");
        for (int num = L; num <= R; num++){
            // System.out.println("num = " + num);
            int count = countSetbits(num);
            // System.out.println("count = " + count);
            boolean isP = isPrime(count);
            // System.out.println("isPrime = " + isP);
            if (isP) result++
;
        }
        return result;
    }
    
    private int countSetbits(int num){
        int count = 0;
        while (num > 0){
            if (num % 2 == 1) count++;
            num /= 2;
        
}
        return count;
    }
    
    private boolean isPrime(int count){
        if (count == 1 || count == 0) return false;
        if (count == 2) return true;
        int limit = (int)Math.sqrt(count);
        for (int i = 2; i <= limit; i ++)
{
            if (count % i == 0) return false;
        }
        return true;
    }
}
