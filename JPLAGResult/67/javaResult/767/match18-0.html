class Solution {
    public int countPrimeSetBits(int L, int R) {

        int counter = 0;
        for(int i=L; i<=R; i++){
            int numberOf1 = numOf1(i);
            boolean isPr = isPrime(numberOf1);

            if(isPr){
                counter++;
            }

//            System.out.println(i + ":" + numberOf1 + ":" + isPr);
        }

        return counter;

    }

    private int numOf1(int num){
        int count = 0;
        while(num!=0){
            count++;
            num&=(num -1);
        }
        return count;
    }


    private boolean isPrime(int i){
        if(i <= 1) return false;
        for(int j=2; j*j<=i; j++){
            if(i%j==0){
                return false;
            }
        }
        return true;
    }
}
