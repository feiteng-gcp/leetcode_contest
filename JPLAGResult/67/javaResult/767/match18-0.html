import java.util.Set;

class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;
        for(int i = L; i <= R; i++){
            if(isPrime(countOnes(i))){
                count++;
            }
        }
        
        return count;
    }

    private int countOnes(int n){
        int count = 0;
        int tmp = n;
        while (tmp > 0){
            if((tmp & 1) == 1){
                count++;
            }
            
            tmp >>>= 1;
        }
        
        return count;
    }

    private boolean isPrime(int n){
        if(n < 2){
            return false;
        } else {
            for( int i = 2; i * i <= n; i++){
                if(n % i == 0){
                    return false;
                }
            }

            return true;
        }
    }
}
