class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;
        for(int i=L; i<=R; i++) {
            if(isPrime(
numSetBits(i))) {
                count++;
            }
        }
        return count;
    }
    private int numSetBits(int num) {
        int count = 0;
        while(num!=0) {
            count = count+(num&1);
            num = num/2;
        }
        return count;
    }
    private boolean isPrime(int num) {
        if(num<2) {
            return false;
        }
        double upper = Math.sqrt(num);
        
for(int c=2; c<=upper; c++) {
            if(num%c==0) {
                return false;
            }
        }
        return true;
    }
}
