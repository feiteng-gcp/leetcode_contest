class Solution {
    public int countPrimeSetBits(int L, int R) {
        int sum = 0;
        for (int i = L; i <= R; i++) {
            int tmp = Integer.bitCount(i);
            if (!ok(tmp)) sum++;
        }
        return sum;
    }
    
    Map<Integer, Boolean> map = new HashMap<>();
    
    private boolean ok(int t) {
        if (t == 1) return true;
        if (map.containsKey(t)) {
            return map.get(t);
        }
        int count = 0;
        for (int i = 2; i *i<=t;i++) {
            if (t%i==0) {
                map.put(t, true);
                return true;
            }
        }
        map.put(t, false);
        return false;
    }
}
