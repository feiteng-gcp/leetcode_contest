class Solution {
    public int countPrimeSetBits(int L, int R) {
        int ans = 0;
        for (int x = L; x <= R; x++) if (isPrime(setBits(x))) ans++;
        return ans;
    }
    
    private static boolean isPrime(int x) {
        if (x == 1) return false;
        if (x == 2 || x == 3) return true;
        
for (int n = 2; n * n <= x; n++) if (x % n == 0) return false;
        return true;
    }
    
    private static int setBits(int x) {
        int count = 0;
        while (x != 0) {
            count += x % 2 == 1 ? 1 : 0;
            x /= 2;
        }
        return count;
    }
}
