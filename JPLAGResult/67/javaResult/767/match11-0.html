class Solution {
    
    public int countPrimeSetBits(int L, int R) {

        int result = 0;

        for (int i = L; i <= R; i++) {
            int bitCount = Integer.bitCount(i);
            if (isPrime(bitCount)) {
                result++;
            }
        }

        return result;
    }

    private boolean isPrime(int n) {

        if (n < 2) {
            return false;
        }

        if (n == 2 || n == 3) {
            return true;
        }

        int root = (int)Math.sqrt(n);

        for (int i = 2; i <= root; i++) {
            if (n % i == 0) {
                return false;
            }
        }

        return true;
    }
}
