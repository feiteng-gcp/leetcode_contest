class Solution {
    public int countPrimeSetBits(int L, int R) {
        int result = 0;
        for(int i = L; i <= R; i++) {
            if(isSetBitCountPrime(i)) {
                result++;
            }
        }
        return result;
    }
    
    public static boolean isSetBitCountPrime(int value) {
        int setCount = 0;
        for(int i = 0; i < 32; i++) {
            if((value & (1 << i)) > 0) {
                setCount++;
            }
        }
        return isPrime(setCount);
    }
    
    public static boolean isPrime(int value) {
        if(value <= 1) return false;
        
        for(double i = 2; i <= value - 1; i++) {
            if((value / i) == Math.floor((value / i))) {
                return false;
            }
        }
        return true;
    }
    
}
