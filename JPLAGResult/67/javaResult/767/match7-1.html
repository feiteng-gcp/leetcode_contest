class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;
        for (int i = L; i <= R; i++) {
            if (isPrime(numSetBit(i))) count++
;
        }
        return count;
    }

    private boolean isPrime(int a) {
        if (a <= 1) return false;
        for (int i = 2; i < a; i++) {
            if (a % i == 0) return false;
        
}
        return true;
    }

    private int numSetBit(int a) {
        int count = 0;
        while (a > 0) {
            if ((a & 1) == 1) count++;
            a >>= 1;
        }
        return count;
    }
}
