class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;
        for(int i = L; i <= R; i++){
            if(hasPrimeSetBits(i)) count++;
        }
        return count;
    }
    private boolean hasPrimeSetBits(int n){
        int numSetBits = countSetBits(n);
        return isPrime(numSetBits);
    }
    private boolean isPrime(int n){
        if(n==1) return false;
        if(n==2) return true;
        for(int i = 2; i<n; i++){
            if(n%i == 0) return false;
        }
        return true;
    }
    private int countSetBits(int n){
        int count = 0;
        while(n > 0){
            n&=(n-1);
            count++;
        }
        return count;
    }
}
