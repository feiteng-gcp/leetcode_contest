class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;

        for(int n = L; n <= R; n++){
            if(setBitsIsPrime(n))
                count++
;
        }
        return count;

    }

    boolean setBitsIsPrime(int n){
        if(isPrime(getNumberOfSetBit(n)))
            return true;
        else
            return false;
    }

    int getNumberOfSetBit(int n){
        int count = 0;

        while(n > 0){
            int bit = n - ((n >> 1)<<1);
            if(bit == 1)
                count ++;
            n = n >> 1;
        }
        return count;
    }

    boolean isPrime(int n){
        if(n <= 1)
            return false;
        
if(n == 2)
            return true;

        for(int i = 2; i < n; i++){
            if(n % i == 0)
                return false;
        }

        return true;
    }
}
