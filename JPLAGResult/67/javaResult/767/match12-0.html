class Solution {
    public int countPrimeSetBits(int L, int R) {
        
        Set<Integer> prime = new HashSet<>();
        prime.add(2);
        prime.add(3);
        prime.add(5);
        prime.add(7);
        prime.add(11);
        prime.add(13);
        prime.add(17);
        prime.add(19);
        prime.add(23);
        prime.add(29);
        prime.add(31);
        prime.add(37);
        
        int count=0;
        for(int i = L; i <=R; i++) {
            
            int ones = 0;
            
            int no=i;
            while(no>0) {
                ones += no & 1;
                no = no>>1;
            }
            if(prime.contains(ones)) {
                count++;
            }
        }
        return count;
        
        
    }
}
