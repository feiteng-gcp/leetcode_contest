class Solution {
    public int countPrimeSetBits(int L, int R) {
        int res = 0;
        for(int i = L; i <= R; i++){
            int count = CountOne(i);
            if(isPrimeNumber(count))
                res += 1;
        }
        return res;
    }
    public int CountOne(int n) {  
        int count = 0;  
        while(n != 0){  
            count++;  
            n = n & (n - 1);  
         }  
        return count;  
    }  
    
    public boolean isPrimeNumber(int num){  
        if(num == 2)
            return true; 
        if(num < 2 || num % 2 == 0)
            return false; 
        for(int i = 3; i <= Math.sqrt(num); i += 2){  
            if(num % i == 0)
                return false;   
        }   
        return true;  
    }
}
