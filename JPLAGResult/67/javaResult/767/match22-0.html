class Solution {
    public int countPrimeSetBits(int L, int R) {
        HashSet<Integer> contain=new HashSet<>();
        contain.add(7);
        contain.add(11);
        contain.add(13);
        contain.add(17);
        contain.add(19);
        contain.add(2);
        contain.add(3);
        contain.add(5);

        int n=0;
        for(int i=L;i<=R;i++){
            if(contain.contains(CC(i))==true) n++;
        }
        return n;
    }
    public int CC(int num){
        int i=0;
        while(num!=0){
            if((num&1)==1) i++;
            num=num>>1;
        }
        return i;
    }
}
