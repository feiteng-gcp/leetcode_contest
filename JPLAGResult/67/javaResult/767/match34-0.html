class Solution {
    public int countPrimeSetBits(int L, int R) {
        int result = 0;
        for(int i=L; i<=R; i++) {
            if(isPrime(getOnes(Integer.toBinaryString(i))))
                result++;
        }
        
        return result;
    }
    
    boolean isPrime(int n) {
        if(n==1) return false;
        
        for(int i=2; i<=Math.sqrt(n); i++) {
            if(n%i==0) return false;
        }
        return true;
    }
    
    int getOnes(String s) {
        int count = 0;
        for(char ch : s.toCharArray())
            if(ch=='1') count++;
        
        return count;
    }
}
