class Solution {
    public int countBits(int v) {
        int ct = 0;
        while (v > 0) {
            ct += v%2;
            v /= 2;
        }
        return ct;
    }
    
    public int countPrimeSetBits(int L, int R) {
        int ct = 0;
        Set<Integer> primes = new HashSet<Integer>(Arrays.asList(2, 3, 5, 7, 11, 13, 17, 19));
        for (int i = L; i <= R; i++) {
            int ctb = countBits(i);
            if (primes.contains(ctb)) {
                ct++;
            }
        }
        return ct;
    }
}
