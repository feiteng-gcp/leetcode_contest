class Solution {
    public int countPrimeSetBits(int L, int R) {
		int res = 0;
        for (int num = L; num <= R; ++num) {
        		int setbits = countNumberOfOne(num);
        		boolean prime = isPrime(setbits);
        		//System.out.println(num + " " + setbits + " " + prime);
        		if (prime) ++res;
        }
        return res;
    }
	
	public int countNumberOfOne(int num) {
		int cnt = 0;
		while (num != 0) {
			if ((num&1) == 1) ++cnt;
			num >>= 1;
		}
		return cnt;
	}
	
	public boolean isPrime(int num) {
		if (num == 2) return true;
		if (num == 1) return false;
		for (int k = 2; k <= (int)Math.sqrt(num) + 1; ++k) {
			if (num % k == 0) return false;
		}
		return true;
	}
}
