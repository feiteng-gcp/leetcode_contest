class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;
        for(int i = L; i<=R; i++){
            if(isPrime(
Integer.bitCount(i))){
                count++;
            }
        }
        return count;
    }
    private static boolean isPrime(int num) {
        if (num < 2) return false;
        if (num == 2) return true;
        if (num % 2 == 0) return false;
        for (int i = 3; i * i <= num; i += 2)
            if (num % i == 0) return false;
        return true;
}

}
