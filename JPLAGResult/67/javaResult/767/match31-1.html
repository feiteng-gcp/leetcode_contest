class Solution {
    public int countPrimeSetBits(int L, int R) {
        int res = 0;
        for(int i = L; i <= R; i++)
        {
            int sum = 0;
            for(int ii = 0; ii < 32; ii++)
            {
                sum += (i >>> ii) & 1;
            }
            
            res += check(sum) ? 1 : 0;
        
}
        
        return res;
    }
    
    public boolean check(int sum)
    {
        if(sum <= 1)
            return false;
        
        
for(int i = 2; i < sum; i++)
        {
            if(sum % i == 0)
                return false;
        }
        return true;
    }
}
