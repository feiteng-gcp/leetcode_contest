class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;
        for(int i=L; i<=R; i++){
            if(isPrime(countBinaryRep(i)))count++;
        }
        return count;
    }
    
    public int countBinaryRep(int num){
        int count = 0;
        for(int i = 0; i < 32; i++){
            if(((num >> i) & 1) == 1)count++
;
        }
        return count;
    }
    
    public boolean isPrime(int num) {
        if (num < 2) return false;
        if (num == 2) return true;
        if (num % 2 == 0) return false;
        for (int i = 3; i * i <= num; i += 2)
            if (num % i == 0) return false;
        return true;
    }
}
