class Solution {
    public int countPrimeSetBits(int L, int R) {
        int res = 0;
        for (int i = L; i <= R; i++) {
            String str = Integer.toBinaryString(i);
            int count = 0;
            for (int j = 0; j < str.length(); j++) {
                if (str.charAt(j) == '1') {
                    count++;
                }
            }
            boolean flag = isPrime(count);
            if (flag) 
{
                res++;
            }
        }
        return res;
    }

    public boolean isPrime(int val) {
        if (val == 1) {
            return false;
        }
        for (int i = 2; i < val; i++) {
            if (val % i == 0) {
                return false;
            }
        }
        return true;
    }
}
