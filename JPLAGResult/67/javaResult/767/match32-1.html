class Solution {
    public int countPrimeSetBits(int L, int R) {
        int res = 0;
        for (int i = L; i <= R; i++) {
            int n = Integer.bitCount(i);
            if (n == 2 || n == 3 || n == 5 || n == 7 || n == 11 || n == 13
                || n == 17 || n == 19 || n == 23 || n == 29 || n == 31)
                res++;
        }
        return res;
    }
}
