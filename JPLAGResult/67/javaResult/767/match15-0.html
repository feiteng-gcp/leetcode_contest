class Solution {
    public int countPrimeSetBits(int L, int R) {
        int[] primes = {2, 3, 5, 7, 11, 13, 17, 19};
        Set<Integer> primeSet = new HashSet<>();
        for(int n : primes) {
            primeSet.add(n);
        }
        int count = 0;
        for(int i = L; i <= R; ++ i) {
            int bits = 0;
            for(int n = i; n != 0; n >>= 1) {
                if((n & 1) == 1) {
                    ++ bits;
                }
            }
            if(primeSet.contains(bits)) {
                ++ count;
            }
        }
        return count;
    }
}
