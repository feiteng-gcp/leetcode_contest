class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;
        for (int i = L; i <= R; i++) {
            int num = Integer.bitCount(i);
            if (isPrime(num)) count++;
        }
        return count;
    }
    private boolean isPrime(int n) {
        if (n == 1) return false;
        if(n > 2 && (n & 1) == 0)
            return false;
        // only odd factors need to be tested up to n^0.5
        for(int i = 3; i * i <= n; i += 2) {
            if (n % i == 0) 
                return false;
        }
        return true;
    }
}
