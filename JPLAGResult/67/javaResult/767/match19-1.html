class Solution {
    public int countPrimeSetBits(int L, int R) {
        int ans = 0;
        for (int i = L; i <= R; i++) {
            int count = numOfOne(i);
            if (isPrime(count)) ans++
;
        }
        return ans;
    }
    
    private int numOfOne(int n) {
        int ans = 0;
        while (n > 0) {
            if (n % 2 == 1) ans++;
            n /= 2;
        }
        return ans;
    
}
    
    private boolean isPrime(int n) {
        if (n == 1) return false;
        for (int i = 2; i <= (int) Math.sqrt(n); i++) {
            if (n % i == 0) return false;
        }
        return true;
    }
}
