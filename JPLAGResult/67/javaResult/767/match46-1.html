class Solution {
    static boolean[] isPrime = new boolean[100];
    {
       
        Arrays.fill(isPrime, true);
        isPrime[0] = isPrime[1] = false;
        for(int i = 2; i * i < 100; i++){
            if(isPrime[i]){
                for(int j = 2 * i; j < 100; j += i){
                    isPrime[j] = false;
                }
            }
        }
    }
    public int count1(int n){
        int sum = 0;
        while(n > 0){
            if((n & 1) != 0) sum++;
            n >>= 1;
        }
        return sum;
    }
    
public int countPrimeSetBits(int L, int R) {
        int rs = 0;
        for(int i = L; i <= R; i++){
            int c1 = count1(i);
            if(isPrime[c1]) rs++;
        }
        return rs;
    }
}
