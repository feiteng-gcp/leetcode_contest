class Solution {
    public int countPrimeSetBits(int L, int R) {
        int c=0;
        for(int i=L;i<=R;i++)
        {
            String x = Integer.toBinaryString(i);
            int c1=0;
            for(int j=0;j<x.length();j++)
            {
                if(x.charAt(j)=='1')
                    c1++
;
            }
            if(isPrime(c1))
                c++;
        }
        return c;
    }
    public static boolean isPrime(int a)
    {
        if(a==1)
            return false;
        for(int i=2;i<=a/2;i++)
        
{
            if(a%i==0)
                return false;
        }
        return true;
    }
}
