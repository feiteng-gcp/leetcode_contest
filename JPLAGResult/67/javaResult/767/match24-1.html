class Solution {
    public int countPrimeSetBits(int L, int R) {
        int cnt = 0;
        for (int i = L; i <= R; i++) {
            if (isPrime(numOfBits(i))) {
                ++cnt;
            }
        }
        return cnt;
    }
    
    private int numOfBits(int num) {
        int cnt = 0;
        while (num != 0) {
            if ((num & 0x1) != 0) {
                ++cnt;
            }
            
            num /= 2;
        
}
        return cnt;
    }
    
    private boolean isPrime(int cnt) {
        if (cnt == 2 || cnt == 3 || cnt == 5 || cnt == 7 || cnt == 11 | cnt == 13 | cnt == 17 || cnt == 19 || cnt == 23 || cnt == 31) {
            return true;
        } else {
            return false;
        }
    }
}
