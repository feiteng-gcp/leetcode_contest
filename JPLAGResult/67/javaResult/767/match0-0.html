class Solution {
    public int countPrimeSetBits(int L, int R) {
     
        
		int count = 0;
		for(int i=L; i<=R; i++){
			String binary = Integer.toBinaryString(i);
			int count1s = 0;
			for(int j=0; j<binary.length(); j++){
				if(binary.charAt(j) == '1')
					count1s ++;
			}
			if(isPrime(count1s))
				count ++;
		}
		
		return count;
	
    }
    
    static boolean isPrime(int no) {
		if (no == 1)
			return false;

		for (int i = 2; i < no; i++) {
			if (no % i == 0)
				return false;
		}
		return true;
	}
}
