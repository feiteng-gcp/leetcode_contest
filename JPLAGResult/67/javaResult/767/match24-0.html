class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;
        for (int i = L; i <= R; i++)
            if (isPrime(Integer.bitCount(i))) count++;
        return count;
    }
    
    private boolean isPrime(int num) {
        if (num == 1) return false;
        if (num == 2 || num == 3) return true;
        long n = num;
        if ((n * n - 1) % 24 == 0) return true;
        else return false;
    }
}
