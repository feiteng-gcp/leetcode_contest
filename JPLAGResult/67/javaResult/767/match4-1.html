class Solution {
    public int countPrimeSetBits(int L, int R) {
        int result = 0;
        for(int i = L; i <= R; i++) {
            String s = Integer.toBinaryString(i);
            int countI = 0;
            
for(int j = 0; j < s.length(); j++) {
                if(s.charAt(j) == '1') {
                    countI++;
                }
            }
            if(isPrime(countI)) {
                result++;
            }
        }
        return result;
    }
    
    private boolean isPrime(int n) {
        if(n == 0 || n == 1) {
            return false;
        }
        
for(int i = 2; i < n; i++) {
            if(n%i == 0) {
                return false;
            }
        }
        return true;
    }
}
