class Solution {
	private boolean isPrime(int value) {
		if (value <= 1) {
			return false;			
		} else if (value == 2 || value == 3) {
			return true;
		} else {			
			for (int i = 2;i * i 
<= value;i ++) {
				if (value % i == 0) {
					return false;
				}
			}
			return true;
		}
	}
	
    public int countPrimeSetBits(int L, int R) {
     
    	int ans = 0;
    	for (int
 i = L;i <= R;i ++) {
    		int total = 0;
    		for (int j = 0;j < 32;j ++) {
    			if ((i & (1 << j)) != 0) {
    				total ++;
    			}
    		}
    		if (isPrim
e(total)) {
    			ans ++;
    		}
    	}
    	return ans;
    	
    }
}
