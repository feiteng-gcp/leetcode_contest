class Solution {
    public int countPrimeSetBits(int L, int R) {
        Set<Integer> set = new HashSet<>();
        set.add(2);
        set.add(3);
        set.add(5);
        set.add(7);
        set.add(11);
        set.add(13);
        set.add(17);
        set.add(19);
        int res = 0;
        for(int i = L; i <= R; i++){
            int count = 0;
            int num = i;
            while(num > 0){
                if((num & 1) > 0){
                    count++;
                }
                num >>= 1;
            }
            if(set.contains(count)){
                res++;
            }
        }
        return res;
    }
}
