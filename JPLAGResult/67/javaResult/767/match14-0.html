class Solution {
    public int countPrimeSetBits(int L, int R) {
        int res = 0;
        for (int i = L; i <= R; i++) {
            if (isValid(i)) {
                res++;
            }
        }
        return res;
    }
    
    public boolean isValid(int n) {
        int count = 0;
        while (n != 0) {
            count = count + (n & 1);
            n = n >>> 1;
        }
        if (count < 2) {
            return false;
        }
        
        int sqrt = (int)Math.sqrt(count);
        for (int i = 2; i <= sqrt; i++) {
            if (count % i == 0) {
                return false;
            }
        }
        return true; 
    }
}
