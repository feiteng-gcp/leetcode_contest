class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;
        Set<Integer> primeSet = new HashSet<>();
        primeSet.add(2);
        primeSet.add(3);
        primeSet.add(5);
        primeSet.add(7);
        primeSet.add(11);
        primeSet.add(13);
        primeSet.add(17);
        primeSet.add(19);
        primeSet.add(23);
        primeSet.add(29);
        primeSet.add(31);
        for (int i = L; i <= R; i++) {
            int curr = i;
            int num = 0;
            while (curr != 0) {
                num++;
                curr &= curr - 1;
            }
            
if (primeSet.contains(num)) {
                count++;
            }
        }
        return count;
    }
}
