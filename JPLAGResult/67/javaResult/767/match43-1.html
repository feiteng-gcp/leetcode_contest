class Solution {
    public int countPrimeSetBits(int L, int R) {
        int res = 0;
        for (int i = L; i <= R; i++) {
            if(
isCountPrime(i)) {
                res++;
            }
        }
        return res;
    }
    
    private boolean isCountPrime(int num) {
        int res = 0;
        if ((num & (num - 1)) == 0) {
            return false;
        }
        for (int i = 0; i < 32; i++) {
            if (((num >> i) & 1) == 1) {
                res++;
            }
        }
        
        
if (res == 1) {
            return false;
        }
        if (res == 2) {
            return true;
        }
        for (int i = 2; i < res; i++) 
{
            if (res % i == 0) {
                return false;
            }
        }
        return true;
    }
}
