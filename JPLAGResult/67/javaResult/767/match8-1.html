class Solution {
    public int countPrimeSetBits(int L, int R) {
        int count = 0;
        for (int i = L; i <= R; i++) {
            if (isPrime(countBits(i))) {
                count++;
            
}
        }
        return count;
    }
    public int countBits(int num) {
        int count = 0;
        for (int i = 0; i < 32; i++) {
            if ((num & 1) == 1) count++;
            num = num >> 1;
        }
        return count;
    }
    public boolean isPrime(int num) {
        
if (num == 1) return false;
        for (int i = 2; i < num; i++) {
            if (num % i == 0) return false;
        }
        return true;
    }
}
