class Solution {
    HashSet<Integer> primes;
    public int countPrimeSetBits(int L, int R) {
        primes = new HashSet<Integer>();
        primes.add(2);
        primes.add(3);
        primes.add(5);
        primes.add(7);
        primes.add(11);
        primes.add(13);
        primes.add(17);
        primes.add(19);
        primes.add(23);
        primes.add(29);
        primes.add(31);
        primes.add(37);
        
        int ret = 0;
        for(int i=L;i<=R;i++) {
            if(primes.contains(countSetBits(i))) {
                ret += 1;
            }
        }
        return ret;
    }
    
    private int countSetBits(int i) {
        if(i<=0) throw new RuntimeException("Cannot count set bit of " + i);
        int ret = 0;
        while(i>0) {
            ret += (i%2);
            i = i/2;
        }
        return ret;
    }
}
