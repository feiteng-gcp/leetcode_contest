class Solution {
    public int countPrimeSetBits(int L, int R) {
        int res = 0;
        for(int i = L; i <= R; i++){
            if(isPrime(
countBits(i))){
                res++;
            }
        }
        return res;
    }
    private int countBits(int n){
        int res = 0;
        while(n != 0){
            
if((1 & n) == 1){
                res ++;
            }
            n = n >> 1;
        }
        return res;
    }
    private boolean isPrime(int n){
        
if(n == 0 || n == 1) return false;
        for(int i = 2; i <= (int)Math.sqrt(n); i++){
            if(n % i == 0){
                return false;
            }
        }
        return true;
    }
}
