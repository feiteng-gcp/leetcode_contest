class Solution {
    public int countPrimeSetBits(int L, int R) {
        int total = 0;
        for (int i = L; i <= R; i++) {
            int sum = 0;
            String binary = Integer.toBinaryString(i);
            char[] arr = binary.toCharArray();
            for (char c : arr) {
                if (c == '1') {
                    sum++;
                }
            }
            if (isPrime(sum)) {
                total++;
            }
        }
        return total;
    }
    
    private boolean isPrime(int sum) {
        if (sum < 2) {
            return false;
        }
        for (int i = 2; i < sum; i++) {
            if (sum % i == 0) {
                return false;
            }
        }
        return true;
    }
}
