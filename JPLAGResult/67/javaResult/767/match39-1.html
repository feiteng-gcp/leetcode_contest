class Solution {
    public int countPrimeSetBits(int L, int R) {
        int cnt = 0;
        for (int i = L; i <= R; i++) {
            if (isPrime(Integer.bitCount(i))) cnt += 1
;
        }
        return cnt;
    }

    private boolean isPrime(int n) {
        
if (n == 1) return false;
        for (int i = 2; 2 * i <= n; i++) {
            if (n % i == 0) return false;
        }
        return true;
    }
}
