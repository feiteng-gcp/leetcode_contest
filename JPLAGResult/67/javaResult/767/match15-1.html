class Solution {
    public int countPrimeSetBits(int L, int R) {
        Set<Integer> prime=new HashSet<Integer>();
        prime.add(2);prime.add(3);prime.add(5);prime.add(7);prime.add(11);prime.add(13);prime.add(17);prime.add(19);
        prime.add(23);prime.add(29);prime.add(31);
        
int count=0;
        for(int i=L;i<=R;i++){
            int countBit=0;
            for(int j=0;j<32;j++){
                if((i&(1<<j))!=0){
                    countBit++;
                }
            }
            if(prime.contains(countBit)){
                count++;
            }
        }
        return count;
    }
}
