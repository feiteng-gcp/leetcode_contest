class Solution {
    public int countPrimeSetBits(int L, int R) {
        int res = 0;
        for(int i = L; i <= R; i++){
            int c = Integer.bitCount(i);
            if (c == 2 || c == 3 || c==5 || c==7 || c==11 || c== 13 || c== 17 || c== 19 || c== 23)
                res++;
        }
        return res;
    }
}
