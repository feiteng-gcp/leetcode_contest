/*
    分成多組, 每個letter只能出現在其中一組
*/
class Solution {

    public List<Integer> partitionLabels(String S) {
        Map<Integer, List<Integer>> start = new HashMap<>();
        Map<Integer, List<Integer>> end = new HashMap<>();
        List<Integer> result = new ArrayList<>();
        int charlen = 26;
        if(S == null || S.length() == 0) return result;        
        int len = S.length();
        int[][] arr = new int[charlen][2];
        
        for(int i = 0 ; i < charlen ; i++){
            arr[i][0] = -1;
            arr[i][1] = -1;
        }
        
        for(int i = 0 ; i < len ; i++){            
            int ind = S.charAt(i) - 'a';
            if(arr[ind][0] == -1) arr[ind][0] = i;
            arr[ind][1] = i;
        }
        
        for(int i = 0 ; i < charlen ; i++){
            if(arr[i][0]!= -1){
                int ind = arr[i][0];                
                int ind2 = arr[i][1];
                
                if(start.containsKey(ind)){
                    start.get(ind).add(i);
                }else{
                    List<Integer> list = new ArrayList<>();
                    list.add(i);
                    start.put(ind, list);
                }
                
                if(end.containsKey(ind2)){
                    end.get(ind2).add(i);                    
                }else{
                    List<Integer> list = new ArrayList<>();
                    list.add(i);
                    end.put(ind2, list);
                }                
            }
        }
        
        Set<Integer> set = new HashSet<>();
        int last = -1;

        for(int i = 0; i < len ; i++){
            if(start.containsKey(i)){
                List<Integer> templist = start.get(i);
                for(Integer c : templist){
                    set.add(c);
                }
            }
            if(end.containsKey(i)){
                List<Integer> templist = end.get(i);
                for(Integer c : templist){
                    set.remove(c);
                }                
            
}
            if(set.size() == 0){
                result.add(i-last);
                last = i;
            }
        }
        
        return result;
        
        
    }
}
