class Solution {
    public List<Integer> partitionLabels(String S) {
        int[] freq = new int[26];
        for (char c : S.toCharArray()) freq[c - 'a']++;
        List<Integer> res = new ArrayList<>();
        int[] curFreq = new int[26];
        int len = 0;
        for (char c : S.toCharArray()) {
            curFreq[c - 'a']++;
            len++;
            if (isPartition(curFreq, freq)) {
                res.add(len);
                len = 0;
            }
        }
        return res;
    }
    
    private static boolean isPartition(int[] curFreq, int[] freq) {
        for (int i = 0; i < 26; i++) {
            if (curFreq[i] != 0 && curFreq[i] != freq[i]) return false;
        }
        return true;
    }
}
