class Solution {
    public List<Integer> partitionLabels(String S) {
        
        Map<Character, List<Integer>> map = new HashMap<>();
        for (int i = 0; i <S.length(); i++) {
            char c = S.charAt(i);
            if (!map.containsKey(c))
            map.put(c, new ArrayList<Integer>());
            map.get(c).add(i);
        }
        List<Interset> ret = new ArrayList<>();
        for (Character c : map.keySet()) {
            List<Integer> list = map.get(c);
            if (list.size() >= 1)
            ret.add(new Interset(list.get(0), list.get(list.size() - 1)));
        }
        Collections.sort(ret, new Comparator<Interset>() {
            public int compare(Interset i1, Interset i2) {
                if (i1.st == i2.st) {
                    return i1.ed - i2.ed;
                }
                return i1.st - i2.st;
            }
        });
        int led = -1, lst = -1;
        List<Integer> res = new ArrayList<>();
        for (Interset i : ret) {
            if (i.ed > led) {
                if (i.st >= led) {
                    res.add(i.ed - i.st + 1);
                    lst = i.st;
                    led = i.ed;
                }
                else {
                    res.remove(res.size() - 1);
                    res.add(i.ed - lst + 1);
                    led = i.ed;
                }
            }
        
}
        return res;
        
    }
    
    private class Interset {
        int st;
        int ed;
        
        public Interset(int st, int ed) {
            this.st = st;
            this.ed = ed;
        }
    }
    
}
