class Solution {
    public List<Integer> partitionLabels(String S) {
       
        Map<Character, Integer> map = new HashMap<>();
        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            map.put(c, i);
        }
        List<Integer> res = new LinkedList<>();
        int len = S.length();
        
        int lastCount = -1, tmpCount = 0;
        for (int i = 0; i < len; i++) {
            tmpCount = Math.max(tmpCount, map.get(S.charAt(i)));
            if (i == tmpCount) 
{
                res.add(tmpCount - lastCount);
                lastCount = tmpCount;
                tmpCount++;
            }
        }
        return res;
    }
}
