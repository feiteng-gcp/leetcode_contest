class Solution {
    public List<Integer> partitionLabels(String S) {
        HashSet<Character> set = new HashSet<>();
        List<Integer> ans = new ArrayList<>();
        int lastIndex = 0;
        for (int i = 0; i < S.length(); i++) {
            set.add(S.charAt(i));
            boolean isValid = true;
            if (i != S.length() - 1) {
                for (char ch : set) {
                    if (S.substring(i + 1, S.length()).indexOf(ch) != -1) isValid = false;
                }
            }
            if (isValid) {
                ans.add(i - lastIndex + 1);
                
set.clear();
                lastIndex = i + 1;
            }
        }
        return ans;
    }
}
