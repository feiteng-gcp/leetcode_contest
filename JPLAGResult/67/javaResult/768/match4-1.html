public class Solution {
    public List<Integer> partitionLabels(String s) {
        int[][] records = new int[26][2];
        for (int[] record : records)
            record[0] = record[1] = -1;
        for (int i = 0; i < s.length(); ++i) {
            char c = s.charAt(i);
            int index = c - 'a';
            if (records[index][0] == -1)
                records[index][0] = i;
            else
                records[index][1] = i;
        }
        Map<Integer, Integer> map = new TreeMap<>();
        for (int[] record : records)
            if (record[0] != -1)
                if (record[1] != -1)
                    map.put(record[0], record[1]);
                else
                    map.put(record[0], record[0])
;
        List<Integer> list = new ArrayList<>();
        int start = -1, end = 0;
        for (int i : map.keySet()) {
            int j = map.get(i);
            if (start == -1) {
                start = i;
                end = j;
            } else if (i > end) {
                list.add(end-start+1);
                start = i;
                end = j;
            } else if (j > end)
                end = j;
        }                
        list.add(end-start+1);
        return list;
    }
}
