class Solution {
    public List<Integer> partitionLabels(String S) {
        int[] map = new int[26];
        for (int i = 0; i < S.length(); i++) {
            map[S.charAt(i) - 'a']++;
        }
        Set<Integer> set = new HashSet<Integer>();
        List<Integer> res = new ArrayList<>();
        int sum = 0;
        for (int i = 0; i < S.length(); i++) {
            sum++;
            map[S.charAt(i) - 'a']--;
            if (map[S.charAt(i) - 'a'] == 0) {
                set.remove(S.charAt(i) - 'a');
            } else {
                set.add(S.charAt(i) - 'a');
            }
            if (set.size() == 0) 
{
                res.add(sum);
                sum = 0;
            }
        }
        return res;
    }
}
