class Solution {
    public List<Integer> partitionLabels(String S) {
        int[][] a=new int[26][2];
        List<Integer> ans=new ArrayList<>();
        for (int i=0;i<26;i++) a[i][0]=10000;
        for (int i=0;i<S.length();i++)
        {
            int c=S.charAt(i)-97;
            if (a[c][0]==10000) a[c][0]=i;
            a[c][1]=i;
        }
        //if (1==1){
        //ans.add(a[1][0]); ans.add(a[1][1]); return ans;}
        Arrays.sort(a,(c,d)->(c[0]-d[0]));
        int l=a[0][0],r=a[0][1];
        for (int i=1;i<26;i++)
        {
            if (a[i][0]==10000) break;
            if (a[i][0]>r)
            {
                ans.add(r-l+1);
                l=a[i][0];
                
r=a[i][1];
            }
            r=Math.max(r,a[i][1]);
        }
        ans.add(r-l+1);
        return ans;
    }
}
