class Solution {
    public List<Integer> partitionLabels(String S) {
        List<Integer> res = new ArrayList<>();
        Queue<int[]> queue = new LinkedList<>();
        for (char c : S.toCharArray()) {
        	int n = queue.size();
        	boolean mergeRest = false;
        	int i = 0;
        	int[] tmpSet = new int[26];
        	for (i = 0; i < n; i++) {
        		int[] hash = queue.poll();
        		if (hash[c - 'a'] > 0) {
        			mergeRest = true;
        			tmpSet = hash;
        			break;
        		}
        		queue.offer(hash);
        	}
        	if (mergeRest) {
        		for (int j = i + 1; j < n; j++) {
        			int[] hash = queue.poll();
        			for (int k = 0; k < 26; k++) {
        				tmpSet[k] += hash[k];
        			}
        		}
        		tmpSet[c - 'a'] += 1;
        		queue.offer(tmpSet);
        	}
        	else {
        		int[] newSet = new int[26];
        		newSet[c - 'a'] = 1;
        		queue.offer(newSet);
        	}
        }
        int n = queue.size();
        for (int i = 0; i < n; i++) {
        	int[] hash = queue.poll();
        	int size = 0;
        	for (int k = 0; k < 26; k++) {
        		size += hash[k];
        	}
        	res.add(size);
        }
        return res;
    }
}
