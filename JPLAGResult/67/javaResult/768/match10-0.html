class Solution {
    public List<Integer> partitionLabels(String S) {
          Stack<Integer> stk = new Stack<>();
        int[] last = new int[128];
        Arrays.fill(last, -1);
        for (int i = 0; i < S.length(); i++) {
            int c = S.charAt(i);
            if (last[c] < 0) {
                stk.push(i);
            } else {
                while (!stk.isEmpty()) {
                    int p = stk.peek();
                    if (p > last[c]) {
                        stk.pop();
                        continue;
                    } else break;
                }
            }
            last[c] = i;
        }
        List<Integer> ans = new ArrayList<>();
        int p = S.length();
        while (!stk.isEmpty()) {
            int t = stk.pop();
            ans.add(p - t);
            p = t;
        }
        Collections.reverse(ans);
        return ans;
    }
}
