class Solution {
    public List<Integer> partitionLabels(String S) {
        int[] last = new int[26];
        for (int i = S.length() - 1; i >= 0; --i) {
            if (last[S.charAt(i) - 'a'] < i) {
                last[S.charAt(i) - 'a'] = i;
            }
        }
        List<Integer> rst = new ArrayList<>();
        int count = 0;
        int end = 0;
        for (int i = 0; i < S.length(); ++i) {
            int idx = S.charAt(i) - 'a';
            end = Math.max(end, last[idx]);
            count++;
            if (i >= end) {
                rst.add(count);
                count = 0;
            }
        }
        return rst;
    }
}
