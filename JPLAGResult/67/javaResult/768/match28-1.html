class Solution {
    public List<Integer> partitionLabels(String S) {
        List<Integer> res = new ArrayList<>();
        List<int[]> parts = new ArrayList<>();
        // add into parts
        for (char c = 'a'; c <= 'z'; c++) {
            int index = S.indexOf(c);
            if (index >= 0)
                parts.add(new int[]{index, S.lastIndexOf(c)});
        }
        // merge
        parts.sort(new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[0] - o2[0];
            }
        });
        Integer preStart = null, preEnd = null;
        for (int[] temp : parts) {
            
if (preStart == null) {
                preStart = temp[0];
                preEnd = temp[1];
            } else if (preEnd < temp[0]) {
                res.add(preEnd - preStart + 1);
                preStart = temp[0];
                preEnd = temp[1];
            } else {
                preEnd = Math.max(preEnd, temp[1]);
            }
        }
        if (preStart != null) res.add(preEnd - preStart + 1);
        return res;
    }
}
