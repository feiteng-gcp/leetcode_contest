class Solution {
    public List<Integer> partitionLabels(String S) {
        Map<Character, Integer> charCount = new HashMap<>();
        for (int i = 0; i < S.length(); i++) {
          Integer count = charCount.get(S.charAt(i));
          if (count == null) {
            charCount.put(S.charAt(i), 1);
          } else {
            charCount.put(S.charAt(i), count + 1);
          }
        }

        List<Integer> result = new ArrayList<>();
        Set<Character> currentPartChars = new HashSet<>();
        int currentPartSize = 0;

        for (int i = 0; i < S.length(); i++) {
          currentPartSize++;
          currentPartChars.add(S.charAt(i));
          charCount.put(S.charAt(i), charCount.get(S.charAt(i)) - 1);

          boolean uniquePartChars = true;
          for (Character c: currentPartChars) {
            if (charCount.get(c) > 0) {
              uniquePartChars = false;
              break;
            }
          }

          if (uniquePartChars) {
            result.add(currentPartSize);
            currentPartChars.clear();
            currentPartSize = 0;
          }
        }

        return result;
    }
}
