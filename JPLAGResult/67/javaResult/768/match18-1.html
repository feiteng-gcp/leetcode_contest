class Solution {
    public List<Integer> partitionLabels(String S) {
        List<Integer> res = new ArrayList<>();
        
        int[] idx = new int[26];
        Arrays.fill(idx, -1);
        
        for (int i = S.length() - 1; i >= 0; i--) {
            int k = S.charAt(i) - 'a';
            if (idx[k] == -1) idx[k] = i;
        }
        
        int i = 0, j = i, end = idx[S.charAt(i) - 'a'];
        
        for (; i < S.length(); i++) {
            if (i > end) {
                res.add(i - j);
                j = i;
                end = idx[S.charAt(i) - 'a'];
            }
            
            end = Math.max(end, idx[S.charAt(i) - 'a']);
        }
        
        res.add(i - j);
        
        return res;
    }
}
