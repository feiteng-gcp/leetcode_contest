class Solution {
    public List<Integer> partitionLabels(String s) {
        Map<Character, Integer> map = new HashMap();
        preprocess(s, map);
        int last = 0;
        int pre = 0;
        List<Integer> list  = new ArrayList();
        for (int i = 0; i < s.length(); i++) {
            last = Math.max(map.get(s.charAt(i)), last);
            if (last <= i) {
                if (list.size() == 0) {
                    list.add(i+1);
                    pre = i+1;
                } else {
                    list.add(i+1-pre);
                    pre += (i+1-pre); 
                }
                
            }
        }
        // if (list.size() == 0) {
        //     list.add
        // }
        return list;
    }
    
    private void preprocess(String s, Map<Character, Integer> map) {
        for (int i = 0; i < s.length(); i++) {
            map.put(s.charAt(i) , i);
        }
    }
}
