class Solution {
    public List<Integer> partitionLabels(String s) {
        List<Integer> res = new ArrayList<>();
        if(s == null || s.length() == 0){
            return res;
        }
        
        int len = s.length();
        int[] map = new int[26];
        Set<Character> set = new HashSet<>();
        
        for(char ch : s.toCharArray()){
            map[ch - 'a']++;
        }
        
        for(int i = 0, j = -1; i < len; i++){
            char ch = s.charAt(i);
            set.add(ch);
            if(--map[ch - 'a'] == 0){
                set.remove(ch);
                if(set.isEmpty()){
                    res.add(i - j);
                    j = i;
                }
            }
        }
        
        return res;
    }
}
