	class Solution {
	    public List<Integer> partitionLabels(String S) {
	        char[] s = S.toCharArray();
	        int[] first = new int[26];
	        int[] last = new int[26];
	        Arrays.fill(first, -1);
	        for(int i = 0;i < s.length;i++){
	        	if(first[s[i]-'a'] == -1){
	        		first[s[i]-'a'] = i;
	        	}
	        	last[s[i]-'a'] = i;
	        }
	        int[][] as = new int[26][];
	        int p = 0;
	        for(int i = 0;i < 26;i++){
	        	if(first[i] != -1){
	        		as[p++] = new int[]{first[i], last[i]};
	        	}
	        }
	        Arrays.sort(as, 0, p, new Comparator<int[]>() {
				public int compare(int[] a, int[] b) {
					return a[0] - b[0];
				}
			});
	        List<Integer> ret = new ArrayList<>();
	        int pos = -1;
	        int reach = -1;
	        for(int i = 0;i < p;i++){
	        	if(as[i][0] > reach){
	        		if(pos < reach){
	        			ret.add(reach - pos);
	        		}
	        		pos = reach;
	        	}
	        	reach = Math.max(reach, as[i][1]);
	        }
	        ret.add(s.length - pos - 1);
	        return ret;
	    }
	}
