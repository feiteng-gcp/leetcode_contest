class Solution {
	private int[] left = new int[26];
	private int[] right = new int[26];
	
    public List<Integer> partitionLabels(String S) {
        
    	List<Integer> ans = new ArrayList<>();
    	int i , j , length = S.length();
    	Arrays.fill(left , - 1);
    	Arrays.fill(right , - 1); 
    	for (i = 0;i < length;i ++) {
    		int index = S.charAt(i) - 'a';
    		if (left[index] < 0) {
    			left[index] = i;
    			right[index] = i;
    		} else {    			
    			right[index] = i;
    		}
    	}
    	for (i = 0;i < length;i ++) {
    		int index = S.charAt(i) - 'a';
    		int from = i , to = right[index];
    		while (i <= to) {
    			index = S.charAt(i) - 'a'
;
    			if (right[index] > to) {
    				to = right[index];
    			}
    			i ++;
    		}
    		i --;
    		ans.add(to - from + 1);
    	}
    	return ans;
    	
    }
	
}
