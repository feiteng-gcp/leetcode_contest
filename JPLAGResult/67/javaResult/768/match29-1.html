class Solution {
    public List<Integer> partitionLabels(String S) {
        int[][] f = new int[26][2];

        for (int i = 0; i < 26; i++) {
            f[i][0] = S.length();
            f[i][1] = -1;
        }

        for (int i = 0; i < S.length(); i++) {
            int j = (int) (S.charAt(i) - 'a');
            f[j][0] = Math.min(f[j][0], i);
            f[j][1] = Math.max(f[j][1], i);
        
}


        Arrays.sort(f, new Comparator<int[]>(){
            public int compare(int[] a, int[] b) {
                return a[0] - b[0];
            }
        });

        
        int i = 0;
        List<Integer> ans = new ArrayList<>();
        int left = 0, right = -1;
        while (i < 26) {
            if (f[i][1] != -1) {
                if (right == -1 || f[i][0] < right) {
                    right = Math.max(right, f[i][1]);
                } else {
                    ans.add(right - left + 1);
                    left = f[i][0];
                    right = f[i][1];
                }
            }
            i++;
        }
        ans.add(right - left + 1);

        return ans;
    }
}
