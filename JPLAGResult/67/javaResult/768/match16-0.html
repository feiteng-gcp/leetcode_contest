class Solution {
    
    public static final class Range implements Comparable {
        
        private final int start;
        private final int end;
        
        public Range(int start, int end) {
            this.start = start;
            this.end = end;
        }
        
        public int compareTo(Object o) {
            return Integer.compare(start, ((Range)o).start);
        }
        
    }
    
    public List<Integer> partitionLabels(String input) {
        Map<Character, Range> rangesMap = new HashMap<>();
        for(int i = 0; i < input.length(); i++) {
            char character = input.charAt(i);
            if(!rangesMap.containsKey(character)) {
                rangesMap.put(character, new Range(i, i));
            } else {
                rangesMap.put(character, new Range(rangesMap.get(character).start, i));
            }
        }
        
        List<Range> ranges = new ArrayList(rangesMap.values());
        Collections.sort(ranges);
        
        List<Integer> result = new ArrayList<>();
        
        int maxForThisRange = -1;
        int lowForThisRange = -1;
        
        for(Range range : ranges) {
            if(lowForThisRange == -1) {
                lowForThisRange = range.start;
                maxForThisRange = range.end;
            }
            
            if(range.start > maxForThisRange) {
                result.add(maxForThisRange - lowForThisRange + 1);
                maxForThisRange = range.end;
                lowForThisRange = range.start;
            }
            
            if(range.end > maxForThisRange) {
                maxForThisRange = range.end;
            }
        }
        
        result.add(maxForThisRange - lowForThisRange + 1);
        
        return result;
    }
    
}
