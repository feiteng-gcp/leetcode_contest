class Solution {
    public List<Integer> partitionLabels(String S) {
        int[] map = new int[26];
        for(int x = 0; x < S.length(); x++) {
            int maxIndex = map[S.charAt(x)-97];
            if(maxIndex < x) {
                map[S.charAt(x)-97] = x;
            }
        }
        // "ababcbaca defegd ehijhklij"
        List<Integer> list = new ArrayList<>();
        for(int x = 0; x < S.length(); x++) {
            int maxIndex = map[S.charAt(x)-97];
            
            int counter = 1;
            for(int i = x+1; i <= maxIndex; i++) {
                int nextMax = map[S.charAt(i)-97];
                if(nextMax > maxIndex) {
                    maxIndex = nextMax;
                }
                counter++;
            }
            list.add(counter);
            x = maxIndex;
        }
        return list;
    }
}
