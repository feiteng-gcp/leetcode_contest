class Solution {
    public List<Integer> partitionLabels(String S) {
        List<Integer> res = new ArrayList<>();
        if (S == null || S.length() == 0) return res;
        
        Map<Character, Integer> map = new HashMap<>();
        for (char c : S.toCharArray()) {
            map.put(c, map.getOrDefault(c, 0) + 1);
        }
        
        int start = 0;
        //i is included in the part, i is the end
        for (int i = 0; i < S.length(); i++) {
            if (isValidPart(S, start, i, map)) {
                res.add(i - start + 1); //length = i - start + 1
                start = i + 1;
            }
        }
        return res;
    }
    
    boolean isValidPart(String S, int start, int end, Map<Character, Integer> map) {
        Map<Character, Integer> currMap = new HashMap<>();
        for (int i = start; i <= end; i++) {
            char c = S.charAt(i);
            currMap.put(c, currMap.getOrDefault(c, 0) + 1);
        }
        
        for (char c : currMap.keySet()) {
            int freq = currMap.get(c);
            if (freq != map.get(c)) {
                return false;
            }
        }
        return true;
    }
}
