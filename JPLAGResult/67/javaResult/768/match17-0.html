class Solution {
    public List<Integer> partitionLabels(String a) {
        HashMap<Character,Range> hm = new HashMap<>();
        for(int i=0;i<a.length();i++){
            char ch = a.charAt(i);
            if(!hm.containsKey(ch)){
                hm.put(ch,new Range(i,i));
            }else{
                Range r = hm.get(ch);
                if(i>r.getE()){
                    r.setE(i);
                }
            }
        }
        List<Integer> list = new ArrayList<>();
        int oldendp = 0, endp = retEnd(a,0,hm);
        list.add(endp+1);
        while(endp!=a.length()-1){
            oldendp = endp+1;
            endp = retEnd(a,endp+1,hm);
//            System.out.println(oldendp+" "+endp);
            list.add(endp-oldendp+1);
        }
        return list;
    }
    public static int retEnd(String a,int st,HashMap<Character,Range> hm){
        int endp = hm.get(a.charAt(st)).getE();
        for(int i=st+1;i<=endp;i++){
            endp = Math.max(hm.get(a.charAt(i)).getE(),endp);
        }
        return endp;
    }
    
    static class Range{
        int s,e;

        public Range(int s, int e) {
            this.s = s;
            this.e = e;
        }

        public int getS() {
            return s;
        }

        public void setS(int s) {
            this.s = s;
        }

        public int getE() {
            return e;
        }

        public void setE(int e) {
            this.e = e;
        }
    }
}
