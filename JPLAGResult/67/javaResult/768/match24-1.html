class Solution {
    public List<Integer> partitionLabels(String S) {
        List<Integer> result = new ArrayList<Integer>();
        HashMap<Character, Integer> charToMaxIndex = new HashMap<Character, Integer>();
        for(int index=0; index < S.length(); index++) {
            charToMaxIndex.put(S.charAt(index), index);
        }
        int startIndex=0;
        int maxIndex=0;
        for(int index=0; index < S.length(); index++) {
            int endIndex = charToMaxIndex.get(S.charAt(index));
            if(maxIndex < endIndex) {
                maxIndex = endIndex;
            }
            if(maxIndex == index) {
                result.add(maxIndex - startIndex + 1);
                startIndex = maxIndex + 1;
            }
            
        }
        return result;
    }
}
