class Solution {
    static int[][] DIRS = new int[][] {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        int[][] maxOnes = new int[N][N];
        boolean[][] ms = new boolean[N][N];
        for (int[] mine : mines) ms[mine[0]][mine[1]] = true;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) maxOnes[i][j] = Integer.MAX_VALUE
;
        }
        for (int[] dir : DIRS) {
            for (int unchanging = 0; unchanging < N; unchanging++) {
                int x = getStart(dir[0], unchanging, N);
                int y = getStart(dir[1], unchanging, N);
                int ones = 0;
                while (x >= 0 && x < N && y >= 0 && y < N) {
                    if (!ms[x][y]) ones++;
                    else ones = 0;
                    maxOnes[x][y] = Math.min(maxOnes[x][y], ones);
                    x += dir[0];
                    y += dir[1];
                }
            
}
        }
        int ans = 0;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) ans = Math.max(maxOnes[i][j], ans);
        }
        return ans;
    }
    
    private static int getStart(int dir, int unchanging, int N) {
        if (dir == 1) return 0;
        else if (dir == -1) return N-1;
        else return unchanging;
    }
}
