class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        int[][] matrix = new int[N][N];
        for(int i = 0; i < mines.length; i++){
            matrix[mines[i][0]][mines[i][1]] = 1;
        }
        int max = 0;
        for(int j = 0; j < N; j++){
            for(int k = 0; k < N; k++){
                max = Math.max(max,align(N,matrix,j,k));
            }
        }
        return max;
    }
    private int align(int N, int[][] matrix, int j, int k){
        int count = 0;
        while((k+count<N && matrix[j][k+count]==0)
             && (j+count<N && matrix[j+count][k]==0)
             && (k-count>=0 && matrix[j][k-count]==0)
             && (j-count>=0 && matrix[j-count][k]==0)){
            count++;
        }
        return count;
    }
}
