class Solution {
    
public int orderOfLargestPlusSign(int N, int[][] mines) {
        int[][] board = new int[N][N];
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                board[i][j] = 1;
            }
        }
        for (int[] cell: mines) {
            board[cell[0]][cell[1]] = 0;
        }
        int maxOrder = 0;
        int maxPoss = (N - 1) / 2 + 1;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                maxOrder = Math.max(maxOrder, orderOfCell(N, board, i, j));
                if (maxOrder == maxPoss) return maxOrder;
            }
        }
        return maxOrder;
    }

    private int orderOfCell(int N, int[][] board, int i, int j) {
        int order = 0;
        int offset = 0;
        while (true) {
            if ((i - offset >= 0 && board[i - offset][j] == 1)
               && (i + offset < N && board[i + offset][j] == 1)
               && (j - offset >= 0 && board[i][j - offset] == 1)
               && (j + offset < N && board[i][j + offset] == 1)) {
                order++;
                offset++;
            } else {
                break;
            }
        }
        return order;
    }
}
