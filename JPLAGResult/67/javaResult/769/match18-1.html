class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
		int [][][] dp = new int[N][N][4];
		int [][] field = new int [N][N];
		// -1 for mine, 0 for cell
		for (int[] mine: mines) {
			field[mine[0]][mine[1]] = -1;
		}
		int cnt = 0;
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				dp[i][j][0] = cnt;
				cnt = (field[i][j] == 0) ? cnt + 1 : 0;
			}
			cnt = 0;
			for (int j = N - 1; j >= 0; j--) {
				dp[i][j][1] = cnt;
				cnt = (field[i][j] == 0) ? cnt + 1 : 0;
			}
			cnt = 0;

		}
		for (int j = 0; j < N; j++) {
			for (int i = 0; i < N; i++) {
				dp[i][j][2] = cnt;
				cnt = (field[i][j] == 0) ? cnt + 1 : 0;
			}
			cnt = 0;
			for (int i = N - 1; i >= 0; i--) {
				dp[i][j][3] = cnt;
				cnt = (field[i][j] == 0) ? cnt + 1 : 0;
			}
			cnt = 0;
		}

		int max = 0;
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if (field[i][j] == 0) {
					int localMax = 1 + Math.min(Math.min(dp[i][j][0], dp[i][j][1]), Math.min(dp[i][j][2], dp[i][j][3]));
					max = Math.max(max, localMax);

				}
			}
		}
		return max;
    }
}
