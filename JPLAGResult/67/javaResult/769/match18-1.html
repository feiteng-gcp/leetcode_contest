class Solution {
	private static int[][] left = new int[510][510];
	private static int[][] right = new int[510][510];
	private static int[][] up = new int[510][510];
	private static int[][] down = new int[510][510];
	private static int[][] matrix = new int[510][510];
	
    public int orderOfLargestPlusSign(int N, int[][] mines) {
     
    	int i , j;
    	for (i = 0;i < N;i ++) {
    		for (j = 0;j < N;j ++) {
    			left[i][j] = right[i][j] = up[i][j] = down[i][j] = matrix[i][j] = 0;

    		}
    	}
    	for (int[] array : mines) {
    		matrix[array[0]][array[1]] = 1;
    	}
    	for (i = 0;i < N;i ++) {
    		for (j = 0;j < N;j ++) {
    			if (matrix[i][j] == 0) {
    				int from = j , to = j , total = 0;
    				while (j < N && matrix[i][j] == 0) {
    					total ++;
    					to = j;
    					j ++;
    				}
    				int value1 = 0 , value2 = total - 1;
    				for (int k = from;k <= to;k ++) {
    					left[i][k] = value1;
    					right[i][k] = value2;
    					value1 ++;
    					value2 --;
    				}
    			}
    		}
    	}
    	for (j = 0;j < N;j ++) {
    		for (i = 0;i < N;i ++) {
    			if (matrix[i][j] == 0) {
    				int from = i , to = i , total = 0;
    				while (i < N && matrix[i][j] == 0) {
    					total ++;
    					to = i;
    					i ++;
    				}
    				int value1 = 0 , value2 = total - 1;
    				for (int k = from;k <= to;k ++) {
    					up[k][j] = value1;
    					down[k][j] = value2;
    					value1 ++;
    					value2 --;
    				}
    			}
    		}
    	}
    	int ans = 0;
    	for (i = 0;i < N;i ++) {
    		for (j = 0;j < N;j ++) {
    			if (matrix[i][j] == 0) {
    				int temp = Math.min(Math.min(up[i][j] , down[i][j]) , Math.min(left[i][j] , right[i][j]));
    				temp ++;
    				if (temp > ans) {

    					ans = temp;
    				}
    			}
    		}
    	}
    	return ans;
    	
    }
}
