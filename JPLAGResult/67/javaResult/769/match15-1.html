import java.util.Arrays;

class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        
int[][] grid = new int[N][N];
        for (int[] d : grid) {
        	Arrays.fill(d, 1);
        }
        for (int[] mine : mines) {
        	grid[mine[0]][mine[1]] = 0;
        }
        int[][] U = new int[N][N];
        int[][] D = new int[N][N];
        int[][] L = new int[N][N];
        int[][] R = new int[N][N];
        
for (int row = 0; row < N; ++row) {
        	for (int col = 0; col < N; ++col) {
        		if (grid[row][col] == 0) {
        			L[row][col] = 0;
        		} else {
        			L[row][col] = 1;
        			if (col - 1 >= 0) {
        				L[row][col] += L[row][col - 1];
        			}
        		}

        	}
        }
        for (int row = 0; row < N; ++row) {
        	for (int col = N - 1; col >= 0; --col) {
        		if (grid[row][col] == 0) {
        			R[row][col] = 0;
        		} else {
        			R[row][col] = 1;
        			if (col + 1 < N) {
        				R[row][col] += R[row][col + 1];
        			}
        		}
        	}
        }
        for (int col = 0; col < N; ++col) {
        	for (int row = 0; row < N; ++row) {
        		if (grid[row][col] == 0) {
        			U[row][col] = 0;
        		} else {
        			U[row][col] = 1;
        			if (row - 1 >= 0) {
        				U[row][col] += U[row - 1][col];
        			}
        		}
        	}
        }
        for (int col = 0; col < N; ++col) {
        	for (int row = N - 1; row >= 0; --row) {
        		if (grid[row][col] == 0) {
        			D[row][col] = 0;
        		} else {
        			D[row][col] = 1;
        			if (row + 1 < N) {
        				D[row][col] += D[row + 1][col];
        			}
        		}

        	}
        }
        int res = 0;
        for (int row = 0; row < N; ++row) {
        	for (int col = 0; col < N; ++col) {
        		int d = Integer.MAX_VALUE;
        		d = Math.min(d, U[row][col]);
        		d = Math.min(d, D[row][col]);
        		d = Math.min(d, L[row][col]);
        		d = Math.min(d, R[row][col]);
        		res = Math.max(res, d);
        	}
        }
        return res;
    }
}
