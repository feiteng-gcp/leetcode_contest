class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        // generate the table
        boolean[][] table = new boolean[N][N];
        for (int[] m: mines)
            table[m[0]][m[1]] = true;
        int res = 0;
        int[] heights = new int[N];
        for (int i = 0; i < N - res; i++) {
            for (int j = res; j < N - res; j++) {
                if (table[i][j]) {
                    heights[j] = 0;
                    continue;
                }
                int order = ++heights[j];

                // go down
                int x = i, y = j, temp = 0;
                while (x < N && x - i < order && !table[x++][y])
                    temp++;
                order = Math.min(order, temp);

                // go left
                x = i;
                y = j;
                temp = 0;
                while (y >= 0 && j - y < order && !table[x][y--])
                    temp++;
                order = Math.min(order, temp);

                // go right
                x = i;
                y = j;
                temp = 0;
                while (y < N && y - j < order && !table[x][y++])
                    temp++;
                order = Math.min(order, temp);

                res = Math.max(res, order);
            }
        }
        return res;
    }
}
