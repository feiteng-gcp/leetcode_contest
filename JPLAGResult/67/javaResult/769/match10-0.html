class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        int[][] grid = new int[N][N];
        
        for (int i = 0; i < N; i++) {
            Arrays.fill(grid[i], 1);
        }
        
        for (int[] m : mines) {
            grid[m[0]][m[1]] = 0;
        }
        
        int[][] dpL = new int[N][N];
        int[][] dpR = new int[N][N];
        int[][] dpU = new int[N][N];
        int[][] dpD = new int[N][N];
        
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                if (grid[i][j] != 0) dpL[i][j] = 1 + (j == 0 ? 0 : dpL[i][j - 1]);
            }
            
            for (int j = N - 1; j >= 0; j--) {
                if (grid[i][j] != 0) dpR[i][j] = 1 + (j == N - 1 ? 0 : dpR[i][j + 1]);
            }
        }
        
        for (int j = 0; j < N; j++) {
            for (int i = 0; i < N; i++) {
                if (grid[i][j] != 0) dpU[i][j] = 1 + (i == 0 ? 0 : dpU[i - 1][j]);
            }
            
            for (int i = N - 1; i >= 0; i--) {
                if (grid[i][j] != 0) dpD[i][j] = 1 + (i == N - 1 ? 0 : dpD[i + 1][j]);
            }
        }
        
        int res = 0;
        
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                if (grid[i][j] == 0) continue;
                int LR = Math.min(dpL[i][j], dpR[i][j]);
                int UD = Math.min(dpU[i][j], dpD[i][j]);
                res = Math.max(res, Math.min(LR, UD));
            }
        }
        
        return res;
    }
}
