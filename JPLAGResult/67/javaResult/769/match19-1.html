class Solution {
    int N;
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        this.N = N;
        int[][] grid = new int[N][N];
        for(int[] mine : mines){
            grid[mine[0]][mine[1]] = 1;
        }
        int[][] left = new int[N][N];
        int[][] right = new int[N][N];
        int[][] upper = new int[N][N];
        int[][] down = new int[N][N];
        
int res = 0;
        for(int i = 0; i < N; i++){
            for(int j = 0; j < N; j++){
                if(grid[i][j] == 0){
                    if(left[i][j] == 0){
                        dfs(grid, i, j, left, new int[]{-1, 0});
                    }
                    if(right[i][j] == 0){
                        dfs(grid, i, j, right, new int[]{1, 0});
                    }
                    if(upper[i][j] == 0){
                        dfs(grid, i, j, upper, new int[]{0, 1});
                    }
                    if(down[i][j] == 0){
                        dfs(grid, i, j, down, new int[]{0, -1});
                    }
                    res = Math.max(res, Math.min(Math.min(left[i][j], right[i][j]), 
Math.min(upper[i][j], down[i][j])));
                }
            }
        }
        return res;
    }
    
    public int dfs(int[][] grid, int x, int y, 
int[][] dp, int[] move){
        if(dp[x][y] != 0){
            return dp[x][y];
        }
        
        int xx = x + move[0];
        int yy = y + move[1];
        int res = 1;
        if(xx >= 0 && xx < N && yy >= 0 && yy < N && grid[xx][yy] == 0){
            res += dfs(grid, xx, yy, dp, move);
        }
        dp[x][y] = res;
        return res;
    }
}
