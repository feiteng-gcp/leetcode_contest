class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        if(N==0){
            return 0;
        }
        //0up 1left 2down 3 right
        int[][] matrix=new int[N][N];
        int[][][] count=new int[N][N][4];
        for(int i=0;i<N;i++){
            Arrays.fill(matrix[i],1);
        }
        for(int i=0;i<mines.length;i++){
            matrix[mines[i][0]][mines[i][1]]=0;
        }
        for(int i=0;i<matrix.length;i++){
            for(int j=0;j<matrix[0].length;j++){
                if(matrix[i][j]==1){
                    count[i][j][0]=i==0?1:(count[i-1][j][0]+1);
                    count[i][j][1]=j==0?1:(count[i][j-1][1]+1);
                }
            }
        }
        for(int i=matrix.length-1;i>=0;i--){
            for(int j=matrix[0].length-1;j>=0;j--){
                if(matrix[i][j]==1){
                    count[i][j][2]=i==matrix.length-1?1:(count[i+1][j][2]+1);
                    count[i][j][3]=j==matrix[0].length-1?1:(count[i][j+1][3]+1);
                }
            }
        }
        int res=0;
        for(int i=0;i<count.length;i++){
            for(int j=0;j<count[0].length;j++){
                if(matrix[i][j]==1){
                    int max=Integer.MAX_VALUE;
                    for(int k=0;k<4;k++){
                        max=Math.min(max,count[i][j][k]);
                    }
                    res=Math.max(res,max);
                }
            }
        }
        return res;
    }
}
