class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        int[][] grid = new int[N][N];   
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                
grid[i][j] = 1;
            }
        }

        for (int i = 0; i < mines.length; i++) {
            
                grid[mines[i][0]][mines[i][1]] = 0;
            
        }
        int max = 0;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                int plus = grid[i][j] == 0 ? 0 : getPlus(grid, i, j);
                if (plus > max) {
                    max = plus;
                
}
            }
        }
        return max;
        
    }
    
    int getPlus(int[][] grid, int i, int j) {
        int plus = 1;
        for (int k = 1; k < grid.length; k++) {
            if (i - k < 0 || grid[i-k][j] != 1 || j - k < 0 || grid[i][j-k] != 1) {
                break;
            }
            if (i + k >= grid.length || grid[i+k][j] != 1 || j + k >= grid.length || grid[i][j+k] != 1) {
                break;
            }
            plus = k + 1;
            
        }
        return plus;
    }
}
