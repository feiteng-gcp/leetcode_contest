class Solution {
	void getLeft(int[] grid, int[] res) {
		for (int i = 0; i < grid.length; i++) {
			if (grid[i] == 0) {
				res[i] = 0;
			} else {
				if (i == 0) {
					res[i] = 1;
				} else {
					res[i] = res[i - 1] + 1;
				}
			}
		}
	}
	
	void getRight(int[] grid, int[] res) {
		for (int i = grid.length - 1; i >= 0; i--) {
			if (grid[i] == 0) {
				res[i] = 0;
			} else {
				if (i == grid.length  - 1) {
					res[i] = 1;
				} else {
					res[i] = res[i + 1] + 1;
				}
			}
		}
	}
	
	

	public int orderOfLargestPlusSign(int N, int[][] mines) {
		int[][] grid =
 new int[N][N];
		
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				grid[i][j] = 1;
			}
		}
		
		for (int i = 0; i < mines.length; i++) {
			grid[mines[i][0]][mines[i][1]] = 0;
		}
		
		int[][] rowL = new int[N][N];
		int[][] rowR = new int[N][N];
		
	
		for (int i = 0; i < N; i++) {
			getLeft(grid[i], rowL[i]);
			getRight(grid[i], rowR[i]);
			
			
		}
		
		int[][] colL = new int[N][N];
		int[][] colR = new int[N][N];
		
		
		for (int j = 0; j < N; j++) {
			int[] col = new int[N];
			for (int i = 0; i < N; i++) {
				col[i] = grid[i][j];
			}
			getLeft(col, colL[j]);
			getRight(col, colR[j]);
		}
		
//		Util.print2Array(rowL);
//		Util.print2Array(rowR);
//		Util.print2Array(colL);
//		Util.print2Array(colR);
		int ans = 0;
		
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				int t = Math.min(rowL[i][j], rowR[i][j]);
				t = Math.min(t, Math.min(colL[j][i], colR[j][i]));
				ans = Math.max(ans, t);

				
			}
		}
		return ans;
	}
}
