class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        boolean[][] m = new boolean[N][N];
        for (int i = 0; i < mines.length; ++i) {
            m[mines[i][0]][mines[i][1]] = true;
        }
        int[][] us = new int[N][N];
        int[][] ds = new int[N][N];
        int[][] ls = new int[N][N];
        int[][] rs = new int[N][N];
        int rst = -1;
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < N; ++j) {
                if (m[i][j]) {
                    continue;
                }
                int v = left(i, j, ls, m);
                v = v == 0 ? v : Math.min(v, right(i, j, rs, m));
                v = v == 0 ? v : Math.min(v, up(i, j, us, m)); 
                v = v == 0 ? v : Math.min(v, down(i, j, ds, m));
                rst = Math.max(v, rst);
            }
        }
        return rst + 1;
    }
    
    private int left(int i, int j, int[][] ss, boolean[][] m) {
        if (ss[i][j] > 0) {
            return ss[i][j];
        }
        int ni = i;
        int nj = j - 1;
        if (ni < 0 || ni >= ss.length || nj < 0 || nj >= ss.length) {
            return 0;
        }
        if (m[ni][nj]) {
            return 0;
        }
        int r = left(ni, nj, ss, m);
        ss[i][j] = r + 1;
        return ss[i][j];
    }
    
    private int right(int i, int j, int[][] ss, boolean[][] m) {
        if (ss[i][j] > 0) {
            return ss[i][j];
        }
        int ni = i;
        int nj = j + 1;
        if (ni < 0 || ni >= ss.length || nj < 0 || nj >= ss.length) {
            return 0;
        }
        if (m[ni][nj]) {
            return 0;
        }
        int r = right(ni, nj, ss, m);
        ss[i][j] = r + 1;
        return ss[i][j];
    }
    
    private int up(int i, int j, int[][] ss, boolean[][] m) {
        if (ss[i][j] > 0) {
            return ss[i][j];
        }
        int ni = i - 1;
        int nj = j;
        if (ni < 0 || ni >= ss.length || nj < 0 || nj >= ss.length) {
            return 0;
        }
        if (m[ni][nj]) {
            return 0;
        }
        int r = up(ni, nj, ss, m);
        ss[i][j] = r + 1;
        return ss[i][j];
    }
    
    private int down(int i, int j, int[][] ss, boolean[][] m) {
        if (ss[i][j] > 0) {
            return ss[i][j];
        }
        int ni = i + 1;
        int nj = j;
        if (ni < 0 || ni >= ss.length || nj < 0 || nj >= ss.length) {
            return 0;
        }
        if (m[ni][nj]) {
            return 0;
        }
        int r = down(ni, nj, ss, m);
        ss[i][j] = r + 1;
        return ss[i][j];
    }
}
