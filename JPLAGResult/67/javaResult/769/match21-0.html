class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        boolean[][] bo=new boolean[N][N];
        int[][] up=new int[N][N];
        int[][] down=new int[N][N];
        int[][] left=new int[N][N];
        int[][] right=new int[N][N];
        for (int[] m:mines) bo[m[0]][m[1]]=true;
        int ans=-1;
        for (int i=0;i<N;i++)
            for (int j=0;j<N;j++)
            {
                if (bo[i][j]) left[i][j]=-1;
                else left[i][j]=(j==0)?0:left[i][j-1]+1;
            }
        for (int i=0;i<N;i++)
            for (int j=0;j<N;j++)
            {
                if (bo[i][j]) up[i][j]=-1;
                else up[i][j]=(i==0)?0:up[i-1][j]+1;
            }
        for (int i=0;i<N;i++)
            for (int j=N-1;j>=0;j--)
            {
                if (bo[i][j]) right[i][j]=-1;
                else right[i][j]=(j==N-1)?0:right[i][j+1]+1;
            }
        for (int i=N-1;i>=0;i--)
            for (int j=0;j<N;j++)
            {
                if (bo[i][j]) down[i][j]=-1;
                else down[i][j]=(i==N-1)?0:down[i+1][j]+1;
            }
        for (int i=0;i<N;i++)
            for (int j=0;j<N;j++)
            {
                int o=Math.min(up[i][j],down[i][j]);
                o=Math.min(o,left[i][j]);
                o=Math.min(o,right[i][j]);
                ans=Math.max(ans,o);
            }
        return ans+1;
    }
}
