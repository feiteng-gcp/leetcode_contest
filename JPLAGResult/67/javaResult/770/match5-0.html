class Solution {
    public int minSwapsCouples(int[] index) {
    	int N = index.length;
    	int res = 0;
    	boolean[] mark = new boolean[N];
    	int[] row = new int[N];
    	for (int i = 0, val = 0; i < N; i += 2, val++) {
    		row[index[i]] = val;
    		row[index[i + 1]] = val;
    	}
    	for (int i = 0; i < N; i += 2) {
    		if (row[i] == row[i + 1]) {
    			mark[i] = mark[i + 1] = true;
    		}
    	}
        for (int i = 0; i < N; i += 2) {
        	if (mark[i] && mark[i + 1]) {
        		continue;
        	}
        	for (int j = i + 2; j < N; j += 2) {
        		if (mark[i] && mark[i + 1]) {
        			continue;
        		}
        		if (row[i] == row[j] && row[i + 1] == row[j + 1]
        		|| row[i] == row[j + 1] && row[i + 1] == row[j]) {
        			mark[i] = mark[i + 1] = true;
        			mark[j] = mark[j + 1] = true;
        			++res;
        		}
        	}
        }
        for (int i = 0; i < N; i += 2) {
        	if (mark[i] && mark[i + 1]) {
        		continue;
        	}
        	for (int j = i + 2; j < N; j += 2) {
            	if (mark[j] && mark[j + 1]) {
            		continue;
            	}
            	if (row[i] == row[j]) {
            		++res;
            		swap(row, i + 1, j);
            	} else if (row[i] == row[j + 1]) {
            		++res;
            		swap(row, i + 1, j + 1);
            	} else if (row[i + 1] == row[j]) {
            		++res;
            		swap(row, i, j);
            	} else if (row[i + 1] == row[j + 1]) {
            		++res;
            		swap(row, i, j + 1);
            	} 
        	}
        }
        return res;
    }
    
    void swap(int[] row, int i, int j) {
    	int t = row[i];
    	row[i] = row[j];
    	row[j] = t;
    }
}
