class Solution {
    public int minSwapsCouples(int[] row) {
		List<Set<Integer>> group = new ArrayList<>();
		for (int i = 0; i < row.length; i+=2) {
			int cpNum1 = row[i]/2;
			int cpNum2 = row[i+1]/2;
			if (cpNum1 != cpNum2) {
				boolean found = false;
				Set<Integer> combineSet = null;
				for (Set<Integer> set : group) {
					if (set.contains(cpNum1) || set.contains(cpNum2)) {
						set.add(cpNum1);
						set.add(cpNum2);
						combineSet = set;
						found = true;
						break;
					}
				}
				if (!found) {
					Set<Integer> set = new HashSet<>();
					set.add(cpNum1);
					set.add(cpNum2);
					group.add(set);
				} else {
					for (int j = group.size() - 1; j >=0; j--) {
						Set<Integer> set = group.get(j);
						if (set.contains(cpNum1) || set.contains(cpNum2)) {
							if (set != combineSet) {
								combineSet.addAll(set);
								group.remove(j);
							}
						}

					}
				}
			}
		}
        int total = 0;
        for (Set<Integer> set : group) {
        	total += set.size() - 1;
        }
        return total;
    }
}
