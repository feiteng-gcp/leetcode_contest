class Solution {
    public int minSwapsCouples(int[] row) {
        int res = 0, n = row.length;
        
        int[] idx = new int[n];
        
        for (int i = 0; i < n; i++) {
			idx[row[i]] = i;
		}
        
        for (int k = 0; k < n; k += 2) {
			for (int i = k, j = partner(i), p = partner(row[i]); p != row[j]; 
                                            i = idx[p], j = partner(i), p = partner(row[i])) {
				swap(row, j, idx[p]);
				res++;
			}
		}
        
        return res;
    }
    
    private int partner(int k) {
		return (k & 1) == 0 ? k + 1 : k - 1;
	}
                 
    private void swap(int[] nums, int i, int j) {
		int t = nums[i];
		nums[i] = nums[j];
		nums[j] = t;
	}
}
