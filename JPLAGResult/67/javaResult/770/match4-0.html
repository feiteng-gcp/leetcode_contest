class Solution {
    public int minSwapsCouples(int[] row) {
        int count = 0;
        for(int i = 0; i < row.length; i+=2){
            if(row[i]%2==0){
                if(row[i+1]!=row[i]+1){
                    for(int j = i+2; j < row.length; j++){
                        if(row[j]==row[i]+1){
                            int temp = row[j];
                            row[j] = row[i+1];
                            row[i+1] = temp;
                            count++;
                            //System.out.println(i);
                        }
                    }
                }
            } else{
                if(row[i+1]!=row[i]-1){
                    for(int j = i+2; j < row.length; j++){
                        if(row[j]==row[i]-1){
                            int temp = row[j];
                            row[j] = row[i+1];
                            row[i+1] = temp;
                            count++;
                            //System.out.println(i);
                        }
                    }
                }
            }
        }
        return count;
    }
}
