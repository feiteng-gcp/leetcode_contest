class Solution {
    public int minSwapsCouples(int[] row) {
        return minSwapsCouples(row, 0);
    }

    private int minSwapsCouples(int[] row, int start) {
        if (start == row.length) return 0;
        int result = 0;
        int currVal = row[start];
        int pair = currVal ^ 1;
        if (row[start + 1] != pair) {
            for (int i = start + 2; i < row.length; i++) {
                if (row[i] == pair) {
                    swap(row, i, start + 1);
                    result = 1;
                    break;
                }
            }
        }
        return result + minSwapsCouples(row, start + 2);
    }

    private void swap(int[] row, int i, int j) {
        int tmp = row[i];
        row[i] = row[j];
        row[j] = tmp;
    }
}
