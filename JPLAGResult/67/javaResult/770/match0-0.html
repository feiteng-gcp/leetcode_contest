class Solution {
    public int minSwapsCouples(int[] row) {
        int[] indices = new int[row.length];
        for (int i = 0; i < row.length; i++)
            indices[row[i]] = i;
        
        int swap = 0;
        for (int i = 1; i < row.length; i += 2) {
            int one = row[i - 1];
            int other = one % 2 == 0 ? one + 1 : one - 1;
            if (row[i] != other) {
                int j = indices[other];
                row[j] = row[i];
                row[i] = other;
                swap++;
                
                indices[other] = i;
                indices[row[j]] = j;
            }
        }
        
        // System.out.println(joinIntArray(indices));
        // System.out.println(joinIntArray(row));
        
        return swap;
    }

    }

}
