class Solution {
    public int minSwapsCouples(int[] row) {
        int n=row.length/2;
        
int[] pos=new int[n*2];
        for (int i=0;i<2*n;i++) pos[row[i]]=i;
        int ans=0;
        for (int i=0;i<n;i++)
        {
            int x=row[i*2];
            int z=row[i*2+1];
            int y=x^1;
            int p=pos[y];
            if (p==i*2+1) continue;
            int t=row[p]; row[p]=row[i*2+1]; row[i*2+1]=t;
            pos[z]=p; pos[y]=i*2+1;
            ans++;
        }
        return ans;
    }
}
