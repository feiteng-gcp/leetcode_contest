class Solution {
    int[] id;
    int count;
    
    public int minSwapsCouples(int[] row) {
        int len = row.length;
        id = new int[len];
        int res = 0;
        for(int i = 0; i < len; i++){
            id[i] = i;
        }    
        for(int i = 0; i < len; i += 2){
            union(i, i + 1);    
        }
        
        for(int i = 0; i < len; i += 2){
            if(union(row[i], row[i + 1])){
                res++;
            }    
        }
        return res;
    }
    
    public int find(int p){
        while(id[p] != p){
            p = id[p];
            id[p] = id[id[p]];
        }
        return p;
    }
    
    
    public boolean union(int p, int q){
        int pRoot = find(p);
        int qRoot = find(q);
        if(pRoot != qRoot){
            id[pRoot] = qRoot;
            return true;
        }
        else{
            return false;
        }
    }
}
