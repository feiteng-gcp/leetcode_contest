class Solution {
    public int minSwapsCouples(int[] row) {
        Map<Integer,Integer> map=new HashMap<Integer,Integer>();
        for(int i=0;i<row.length;i++){
            map.put(row[i],i);
        }
        int[] res={Integer.MAX_VALUE};
        findMin(res,row,0,map,0);
        return res[0];
    }
    public void findMin(int[] res,int[] row,int index,Map<Integer,Integer> map,int currStep){
        if(currStep>=res[0]){
            return;
        }
        if(index==row.length){
            res[0]=Math.min(res[0],currStep);
            return ;
        }
        if((row[index]%2==0&&row[index+1]==row[index]+1)||(row[index]%2==1&&row[index+1]==row[index]-1)){
            findMin(res,row,index+2,map,currStep);
        }
        else{
            int n1=row[index];
            int n2=row[index+1];
            if(n1%2==0){
                int t1Pos=map.get(n1+1);
                row[t1Pos]=n2;
                row[index+1]=n1+1;
                map.put(n1+1,index+1);
                map.put(n2,t1Pos);
                findMin(res,row,index+2,map,currStep+1);
                map.put(n2,index+1);
                map.put(n1+1,t1Pos);
                row[t1Pos]=n1+1;
                row[index+1]=n2;
            }
            else{
                int t1Pos=map.get(n1-1);
                row[t1Pos]=n2;
                row[index+1]=n1-1;
                map.put(n1-1,index+1);
                map.put(n2,t1Pos);
                findMin(res,row,index+2,map,currStep+1);
                map.put(n2,index+1);
                map.put(n1-1,t1Pos);
                row[t1Pos]=n1-1;
                row[index+1]=n2;
            }/*
            if(n2%2==0){
                int t1Pos=map.get(n2+1);
                row[t1Pos]=n1;
                row[index]=n2+1;
                map.put(n2+1,index);
                map.put(n1,t1Pos);
                findMin(res,row,index+2,map,currStep+1);
                map.put(n1,index);
                map.put(n2+1,t1Pos);
                row[t1Pos]=n2+1;
                row[index]=n1;
            }
            else{
                int t1Pos=map.get(n2-1);

                row[index]=n2-1;

                map.put(n1,t1Pos);
                findMin(res,row,index+2,map,currStep+1);
                map.put(n1,index);
                map.put(n2-1,t1Pos);
                row[t1Pos]=n2-1;
                row[index]=n1;
            }*/
        }
    }
}
