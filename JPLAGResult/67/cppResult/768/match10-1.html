class Solution {
public:
    vector<int> partitionLabels(string S) {

        map<char, int> beginindex, endindex;
        for (int i = 0; i<S.size(); i++) {
            if (beginindex.count(S[i]) == 0) {
                beginindex[S[i]] = i;
            }
            endindex[S[i]] = i;
        }
        auto it1 = beginindex.begin();
        auto it2 = endindex.begin();
        vector<pair<int, int>> ranges;
        while (it1 != beginindex.end() && it2 != endindex.end()) {
            ranges.push_back(make_pair(it1->second, it2->second));
            it1++;
            it2++;
        }
        sort(ranges.begin(), ranges.end(), [](const pair<int, int>& a, const pair<int, int>& b) {
            return (a.first<b.first || (a.first == b.first && a.second<b.second));
        });
        vector<int> res;
        int leftside = 0, rightside = 0;
        int i = 0;
        for (; i<ranges.size(); i++) {
            if (rightside<ranges[i].first) {
                res.push_back(rightside - leftside + 1);
                leftside = ranges[i].first;
                rightside = ranges[i].second;
            }
            else {
                rightside = max(rightside, ranges[i].second);
            }
        }
        if (ranges.size() > 0) {
            res.push_back(S.size()- leftside);
        }

        return res;
    }
};
