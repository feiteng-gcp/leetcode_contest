class Solution {
public:
    vector<int> partitionLabels(string S) {
        unordered_map<char, int> charPos;
        for (int i = S.length() - 1; i >= 0; i--) {
            if (charPos.count(S[i]) == 0) {
                charPos[S[i]] = i;
            }
        }
        vector<int> ans;
        int prev_cut = 0;
        int tentative = 1;
        int pos = 0;
        while (pos < S.length()) {
            while(pos < tentative) {
               if (charPos[S[pos]] + 1 > tentative) {
                   tentative = charPos[S[pos]] + 1;
               } else {
                   
               }   
                pos++;
            }
            ans.push_back(tentative - prev_cut);
            prev_cut = tentative;
            tentative++;
        }
        return ans;
    }
    
};
