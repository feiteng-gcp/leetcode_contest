class Solution {
public:
    vector<int> partitionLabels(string s) {
        int n = s.size(); 
        int last[26];
        
        // get the last index; 
        for(int i=0; i<n; ++i){
            char c= s[i]; 
            last[c - 'a'] = i; 
        }
        
        vector<int> a(n); 
        vector<int> res;
        
        int startInd = 0;
        int maxInd = -1;
        for(int i=0; i<n; ++i){
            char c = s[i]; 
            a[i] = last[c-'a']; 
            
            maxInd = max(maxInd, a[i]);
            if(a[i] == i && maxInd == a[i]){
                int len = maxInd - startInd + 1; 
                res.push_back(len);
                startInd = maxInd + 1;
                cout << " i = " << i <<endl;
            }
        }
        
        
        
        return res;
        
    }
};
