class Solution {
public:
    vector<int> partitionLabels(string S) {
        /* ababcbacadefegdehijhklij
           [    ]
            [      ]
             [       ]
        
        
        */
        unordered_map<char, int> last;
        vector<int> ans;
        
        for (int i = 0; i < S.length(); i++) {
            last[S[i]] = i;
        }
        
        int tail = -1, lastPos = -1;
        for (int i = 0; i < S.length(); i++) {
            if (last[S[i]] > tail)
                tail = last[S[i]];
            if (tail == i) {
                ans.push_back(i - lastPos);
                lastPos = i;
            }
        }
        
        return ans;
    }
};
