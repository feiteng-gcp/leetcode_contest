class Solution {
public:
    vector<int> partitionLabels(string S) {
        vector<int> hash(26, -1);
        for (int i = 0; i < S.length(); ++i) hash[S[i]-'a'] = i;
        
        vector<int> res;
        // [i, next]
        int i = 0;
        int next = hash[S[0] - 'a'];
        if (next <= 0) { res.push_back(1); next = -1; ++i; }
        
        int j = 1;
        while (j < S.length()) {
            next = max(next, hash[S[j] - 'a']);
            if (j >= next) {
                res.push_back(j-i+1);
                ++j; i = j; next = -1;
            }
            else {
                ++j;
            }
        }
        return res;
    }
};
