class Solution {
public:
    vector<int> partitionLabels(string a) {
        int n = a.length();
        map<char,int> mp;
        for(int i=0;i<n;i++) mp[a[i]] = i;
        
        int i = 0;
        vector<int> ans;

        while(i < n) {
            int limit = mp[a[i]];
            int j = i;
            while(j <= limit) {
                limit = max(limit,mp[a[j]]);
                j++;
            }
            ans.push_back(j-i);
            i = j;
        }
        return ans;
    }
};
