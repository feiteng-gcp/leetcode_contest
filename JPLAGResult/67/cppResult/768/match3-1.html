class Solution {
public:
  vector<int> partitionLabels(string S) {
    int n = S.size();
    set<char> in;

    map<char, int> last;
    for (int i = 0; i < n; ++i) {
      last[S[i]] = i;
    }
    vector<int> res;
    for (int i = 0; i < n; ++i) {
      in.insert(S[i]);
      int r = last[S[i]];
      for (int j = i; j <= r; ++j) {
        r = max(r, last[S[j]]);
      }
      res.push_back(r - i + 1);
      i = r;
    }
    return res;
  }
};
