class Solution {
public:
    vector<int> partitionLabels(string S) {
        vector<int> ret;
        vector<pair<int, int>> arr;
        for (char c = 'a'; c <= 'z'; ++ c) {
            int p = INT_MAX, q = INT_MIN;
            for (int i = 0; i < S.size(); ++ i)
                if (S[i] == c) {
                    p = min(i, p);
                    q = max(i, q);
                }
            if (p <= q) arr.push_back(make_pair(p, q));
        }
        sort(arr.begin(), arr.end(), [](pair<int, int>& a, pair<int, int>& b){ return a.first < b.first; });
        int i = 0;
        while (i < arr.size()) {

            int j = i;
            while (j + 1 < arr.size() && arr[j + 1].first <= arr[i].second) {
                arr[i].second = max(arr[i].second, arr[j + 1].second);
                j ++;
            }
            ret.push_back(arr[i].second - arr[i].first + 1);
            i = j + 1;
        }
        return ret;
    }
};
