class Solution {
public:
    struct myclass{
        bool operator() (pair<int, int> a, pair<int, int> b){
            return a.first<b.first;
        }
    } compare;
    
    vector<int> partitionLabels(string S) {
        vector<int> result;
        if(S.length() == 0){
            return result;
        }
        map<char, pair<int, int>> m;
        for(int i = 0; i<S.length(); i++){
            if(m.find(S[i]) == m.end()){
                m[S[i]] = make_pair(i,i);
            } else{
                m[S[i]].second = i;
            }
        }
        vector<pair<int, int>> v;
        for(auto entry:m){
            //cout << entry.second.first << " " << entry.second.second << endl;
            v.push_back(make_pair(entry.second.first,entry.second.second));
        }
        sort(v.begin(), v.end(), compare);
        /*for(auto x:v){
            cout<<x.first<<" "<<x.second<<endl;
        }*/
        int start = v[0].first, end = v[0].second;
        for(int i = 1; i<v.size(); i++){
            int s = v[i].first;
            int e = v[i].second;
            if(s>=start && e<=end){
                continue;
            } else if(s>end){
                result.push_back(end-start+1);
                start = s;
                end = e;
            } else if(s<end && s>start && e>end){
                end = e;
            }
        }
        result.push_back(end-start+1);
        return result;
    }
};
