class Solution {
public:
    vector<int> partitionLabels(string S) {
        const int n = S.size();
        if (n == 0) return {};
        unordered_map<char, pair<int, int>> hash;
        for (int i = 0; i < n; i++) {
            if (hash.find(S[i]) == hash.end()) {
                hash[S[i]] = {i,i};
            }
            else {
                hash[S[i]].second = i;
            }
        }

        int p = 0;
        vector<int> res;
        while (p < n) {
            int start = p;
            int end = hash[S[p]].second;
            while (p < end) {
                p++;
                end = max(end, hash[S[p]].second);
            }
            res.push_back(end-start+1);
            p++;
            
            
        }
        return res;
    }
};
