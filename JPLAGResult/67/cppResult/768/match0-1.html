class Solution {
public:
    vector<int> partitionLabels(string S) {
        vector<int> last(26, -1);
        for(int i=0; i<S.size(); ++i) {
            last[S[i] - 'a'] = i;
        }
        vector<int> res;

        for(int i=0; i<S.size(); ) {
            int cur_end = last[S[i] - 'a'];
            int j = i+1;
            while(j<= cur_end) {
                cur_end = max(last[S[j] - 'a'], cur_end);
                ++j;
            }
            res.push_back(j - i);
            i = j;
        }
        return res;
    }
};
