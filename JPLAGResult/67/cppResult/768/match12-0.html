class Solution {
public:
    struct slot {
        int start;
        int end;
    };
    static bool cmp(slot& a, slot& b) {
        return a.start < b.start;
    }
    vector<int> partitionLabels(string S) {
        unordered_map<char, slot> char_map;
        
        for(int i = 0; i < S.size(); ++i){
            if(char_map.find(S[i]) == char_map.end()){
                char_map[S[i]].start = i;
                char_map[S[i]].end = i;
            } else {
                char_map[S[i]].end = i;
            }
        }
        vector<slot> slots;
        
        for(auto element: char_map){
            char c = element.first;
            slots.push_back(element.second);
        }
        sort(slots.begin(), slots.end(), cmp);
        int start = 0;
        int end = 0;
        vector<int> res;
        bool isbegin = true;
        if(S.empty()){return res;}
        for(slot s: slots) {
            if(isbegin){
                end = s.end;
                isbegin = false;
            } 
            if(s.start > end) {
                res.push_back(end - start + 1);
                start = s.start;
                end = s.end;
            } 
            start = min(s.start, start);
            end = max(end, s.end);
        }
        res.push_back(end-start+1);
        return res;
    }
};
