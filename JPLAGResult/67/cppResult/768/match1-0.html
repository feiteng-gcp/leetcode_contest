int a[26];
class Solution {
public:
    vector<int> partitionLabels(string S) {
        memset(a, 0, sizeof(a));
        for (int i = 0; i < S.length(); i++) {
            a[S[i]-'a'] = i;
        }
        vector<int> ans;
        int i = 0;
        while (i < S.length()) {
            int j = i;
            int last = a[S[i]-'a'];
            while (j <= last) {
                last = max(last, a[S[j]-'a']);
                j++;
            }
            ans.push_back(j-i);
            i = j;
        }
        return ans;
    }
};
