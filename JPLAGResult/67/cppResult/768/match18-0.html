class Solution {
public:
    vector<int> partitionLabels(string s) {
        // if(s.size() == 0) return 0;
        map <char, int > last;
        for(int i = 0; i < s.size(); i++) {
            last[s[i]] = i;
        }
        int parts = 0;
        int st = 0, en = last[s[0]];
        int ptr = 0;
        vector <int> ret;
        while(ptr != s.size()) {
            int co = 0;
            while(ptr <= en) {
                en = max(en, last[s[ptr]]);
                co++;
                ptr++;
            }
            ret.push_back(co);
            en = last[s[ptr]];
        }
        return ret;
    }
};
