class Solution {
public:
    vector<int> partitionLabels(string S) {
        vector<int> res;
        
        int len = S.length();
        if(len == 0) return res;
        
        vector<int> first(26, len+1);
        vector<int> last(26, -1);
        
        for(int i = 0; i < len; i++){
            int idx = S[i] - 'a';
            first[idx] = min(first[idx], i);
            last[idx] = max(last[idx], i);
        }
        

        int s = 0, e = 0;
        for(int i = 0; i < len; i++){
            int idx = S[i] - 'a';
            if(last[idx] > e){
                e = last[idx];
            }
            if(i == e){
                int l = e - s + 1;
                res.push_back(l);
                s = i+1;
                e = i+1;
            }
        }
        return res;
    }
};
