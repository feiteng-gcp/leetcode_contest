class Solution {
public:
    vector<int> partitionLabels(string S) {
        vector<pair<int, int> > positions(26, make_pair(-1, -1));
        for (int i = 0; i < S.size(); ++i) {
            auto& pos = positions[S[i] - 'a'];
            if (pos.first == -1) {
                pos.first = i;

                pos.second = i;
            } else {
                pos.second = i;
            }
        }
        /*for (int i = 0; i < 26; ++i) {
            if (positions[i].first != -1) {
                cout << ('a' + i) << ": " << positions[i].first << ", " << positions[i].second << endl;
            }
        }*/
        map<int, int> counters;
        for (auto& p : positions) {
            if (p.first == -1) {
                continue;
            }
            ++counters[p.first];
            --counters[p.second];
        }
        int start = 0;
        int overlap = 0;
        vector<int> result;
        for (auto& kv : counters) {
            overlap += kv.second;
            if (overlap == 0) {
                result.push_back(kv.first - start + 1);
                start = kv.first + 1;
            }
        }
        return result;
    }
};
