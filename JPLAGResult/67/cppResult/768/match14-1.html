class Solution {
  public:
    vector<int> partitionLabels(string S) {
      if( S.empty() ) return vi{};
      vii v(26,{-1,-1});
      int n = S.size();
      forall(i,0,n){
        int idx = S[i] - 'a';
        if( v[idx].first == -1 ){ 
          v[idx].first = i;
          v[idx].second = i;
        }else{
          v[idx].second = i;
        }
      }
      sort(all(v));
      vi ans;
      int l = 0, r=-1;
      for( auto p : v ){
        if( p.first == -1 ) continue;

        if( p.first > r ){
          ans.pb(r-l+1);
          l = p.first;
          r = p.second;
        }else{ // p.first < r
          r = max( p.second, r );
        }
      }
      ans.pb(r-l+1);
      return vi(ans.begin()+1, ans.end());
    }
};
