int Min[1000], Max[1000];
class Solution {
public:
    vector<int> partitionLabels(string S) {
        vector<int> ans;
        for (int c = 'a'; c <= 'z'; c++) {
            Min[c] = 111111;
            Max[c] = -1;
        }
        for (int i = 0; i < S.size(); i++) {
            Min[S[i]] = min(Min[S[i]], i);
            Max[S[i]] = max(Max[S[i]], i);
        }
                
        int l = 0;
        while (l < S.size()) {
            int r = l;
            int i = l;
            while (i <= r) {
                r = max(r, Max[S[i]]);
                i++;
            }
            ans.push_back(r - l + 1);
            l = r + 1;
        }
        return ans;
    }
};
