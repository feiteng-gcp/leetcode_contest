class Solution {
public:
    vector<int> partitionLabels(string S) {
        int startPos[26];
        int endPos[26];
        for(int i=0;i<26;i++){
            startPos[i]=-1;
            endPos[i]=-1;
        }
        
        for(int i=0;i<S.length();i++){
            if(startPos[S[i]-'a']==-1)
                startPos[S[i]-'a'] = i;
            endPos[S[i]-'a'] = i;
        }
        
        int now = 0;
        
        int startingPartPos = -1;
        int endingPartPos = -1;
        
        vector<int> sizeOfPart;
        
        while(now<S.length()){
            if(now>endingPartPos){
                if(startingPartPos>=0)
                    sizeOfPart.push_back(now-startingPartPos);
                startingPartPos = now;
                endingPartPos = endPos[S[now]-'a'];
            }else{
                endingPartPos = max(endingPartPos, endPos[S[now]-'a']);
            }
            now++;
        }
        sizeOfPart.push_back(now-startingPartPos);
        return sizeOfPart;
    }
};
