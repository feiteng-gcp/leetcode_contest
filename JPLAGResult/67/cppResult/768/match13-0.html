class Solution {
public:
    vector<int> partitionLabels(string S) {
        unordered_map<char, pair<int, int>> mmap;
        for (int i=0; i<S.size(); i++) {
            if (mmap.find(S[i]) == mmap.end()) {
                mmap[S[i]] = make_pair(i, i);
            }
            else {
                mmap[S[i]].second = max(mmap[S[i]].second, i);
            }
        }
        
        vector<int> ans;
        int start = -1, end = -1;
        for (int i=0; i<S.size(); i++) {
            if (start == -1) {
                start = mmap[S[i]].first;
                end = mmap[S[i]].second;
            }
            if (i == end) {
                ans.push_back(end - start + 1);
                start = -1, end = -1;
            }
            else {
                end = max(end, mmap[S[i]].second);
            }
        }

        return ans;
    }
};
