class Solution {
public:
    vector<int> partitionLabels(string S) {
        typedef pair<int,int> pii;
        vector<pii> ints;
        for (char c='a'; c <='z'; c++) {
            int i;
            for (i =0; i< S.length();i++)
                if (S[i]==c)
                    break;
            if (i<S.length())
                ints.push_back(make_pair(i,-1));
            for (i =S.length()-1; i>=0;i--)
                if (S[i]==c)
                    break;
            if (i>=0)
                ints[ints.size()-1].second = i;
        }
        
        sort(ints.begin(),ints.end(),[](const pii& a, const pii& b) {return a.first < b.first;});
        vector<int> res;
        int st = 0;
        int max = 0;
        for (int i = 0; i < ints.size();i++) {
            bool fl = true;
            max = ints[i].second;
            for (int j = 0; j < i; j++) {
                if (ints[j].second > max)
                    max = ints[j].second;
                if (ints[i].first < ints[j].second)
                    fl = false;
            }
            if (fl && i > 0) {
                res.push_back(ints[i].first-st);
                st = ints[i].first;
            }
        }
        res.push_back(max+1-st);
        return res;
    }
};
