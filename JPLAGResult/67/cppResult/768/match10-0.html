class Solution {
public:
    vector<int> partitionLabels(string S) {
        unordered_map<char, int> last;
        for(int i=0; i<S.size(); i++){
            last[S[i]]=i;
        }
        vector<int> ret;
        int start=0;
        while(start<S.size()){
            int end=start;
            for(int i=start; i<=end; i++){
                end=max(end, last[S[i]]);
            }
            ret.push_back(end+1-start);
            start=end+1;
        }
        return ret;
    }
};
