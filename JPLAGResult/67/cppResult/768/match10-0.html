class Solution {
public:
    vector<int> partitionLabels(string S) {
    
        unordered_map<char, int> overall;
        
        for(auto c : S)
            overall[c]++;
        
        int start = -1;
        vector<int> result;
        unordered_map<char,int> curr;
        for(int i = 0; i < S.length(); i++)
        {
            if(curr.count(S.at(i)) == 0) {
                curr[S.at(i)] = overall[S.at(i)];
                overall.erase(S.at(i));
            }
                        
            curr[S.at(i)]--;
            if(curr[S.at(i)] == 0)
            {
                curr.erase(S.at(i));
            }
            
            if(curr.size() == 0)
            {
                result.push_back(i - start);
                start = i;
            }
        }
        return result;
    }
};
