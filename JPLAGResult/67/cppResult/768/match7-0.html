class Solution {
public:
    deque<int> v[26];
    vector<int> ans;
    vector<int> partitionLabels(string S) {
        for(int i=0;i<S.size();i++){
            v[S[i]-'a'].push_back(i);
        }
        int mx=0;
        int lef=0;
        for(int i=0;i<S.size();i++){
          int id=S[i]-'a';
            v[id].pop_front();
            if(v[id].size()){
                mx=max(mx,v[id].back());
            }
            if(mx==i){
                ans.push_back(mx-lef+1);
            lef=i+1;
            mx=i+1;
            }
        }
        return ans;
    }
};
