class Solution {
public:
    struct Rang {
        char ch;
        int L;
        int R;
        Rang(){L = -1; R = -1;}
    };
    vector<int> partitionLabels(string S) {
        vector<Rang> range_of_char(26);
        for(int i=0;i<26;i++){
            range_of_char[i].ch = 'a' + i;
        }
        for(int i=0;i<S.size();i++){
            if(range_of_char[S[i]-'a'].L == -1){
                range_of_char[S[i]-'a'].L = i;
                range_of_char[S[i]-'a'].R = i;
            }
            else{
                range_of_char[S[i]-'a'].R = i;
            }
        }
        sort(range_of_char.begin(), range_of_char.end(), [](Rang& a, Rang& b){
            if(a.L == b.L) return a.R < b.R;
            return a.L < b.L;
        });
        
        vector<int> ret;
        int l=0, r=0;
        for(auto& range : range_of_char){
            if(range.L == -1) continue;
            if(range.L <= r){
                r = max(r, range.R);
            }
            else {
                ret.push_back(r-l+1);
                l = range.L;
                r = range.R;
            }
        }
        ret.push_back(r - l + 1);
        
        return ret;
    }
};
