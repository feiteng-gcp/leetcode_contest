class Solution {
public:
  int minSwapsCouples(vector<int>& row) {
    int n = row.size();
    int ans = 0;
    for (int i = 1; i < n; i += 2) {
      int p = (row[i - 1] >> 1);
      if ((row[i] >> 1) != p) {
        for (int j = i + 1; j < n; ++j) {
          if ((row[j] >> 1) == p) {
            int t = row[i];
            row[i] = row[j];
            row[j] = t;
            ++ans;
            break;
          }
        }
      }
    }
    return ans;
  }
};
