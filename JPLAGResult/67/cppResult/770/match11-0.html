class Solution {
public:
    bool isOK(int a, int b){
        if(a > b){
            int tmp = a;
            a = b;
            b = tmp;
        }
        if(a % 2 != 0) return false;
        if(b-a != 1) return false;
        return true;
    }
    
    void swap(vector<int>& row, int i, int j){
        int tmp = row[i];
        row[i] = row[j];
        row[j] = tmp;
    }
    
    int minSwapsCouples(vector<int>& row) {
        int len = row.size();
        
        int res = 0;
        for(int i = 0; i < len; i=i+2){
            if(isOK(row[i], row[i+1])) continue;
            for(int j = i+2; j < len; j++){
                if(isOK(row[j], row[i+1])){
                    swap(row, i, j);
                    res++;
                    break;
                }
            }
        }
        
        return res;
    }
};
