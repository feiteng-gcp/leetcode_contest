class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int P[100], n = row.size();
        for (int i = 0; i < n; i++) P[row[i]] = i;
        int vis[50] = {0};
        int cnt = n / 2;
        for (int i = 0; i < n; i += 2) if (!vis[i / 2]) {
            cnt--;
            for (int j = i; !vis[j / 2]; j = P[row[j ^ 1] ^ 1]) {
                vis[j / 2] = true;
            }
        }
        return cnt;
    }
};
