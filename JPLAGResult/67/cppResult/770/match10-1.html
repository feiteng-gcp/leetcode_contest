class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int n = row.size(), t=0;
        for(int i=0;i<n;i+=2){

            if(row[i] & 1){
                int pair = row[i]-1;
                if(row[i+1] != pair){
                    for(int j=i+2;j<n;j++){
                        if(row[j] == pair){
                            swap(row[j], row[i+1]);
                            break;
                        }
                    }
                    t++;
                }
            }
            else {
                int pair = row[i]+1;
                if(row[i+1] != pair){
                    for(int j=i+2;j<n;j++){
                        if(row[j] == pair){
                            swap(row[j], row[i+1]);
                            break;
                        }
                    }
                    t++;
                }
            }
        }
        return t;
    }
};
