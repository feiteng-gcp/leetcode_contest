class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int sz=row.size();
        vector<int> pos(sz);
        for(int i=0; i<sz; ++i)
            pos[row[i]]=i;
        
        int ans=0;
        for(int i=0; i<sz; i+=2) {
            int mn=min(row[i],row[i+1]);
            int mx=max(row[i],row[i+1]);
            
            if(mn%2==0 && mx==mn+1) continue;
            
            ++ans;
            if(mn%2) {
                int target=mn-1;
                row[pos[target]]=mx;
                row[pos[mx]]=target;
                swap(pos[target],pos[mx]);
            } else {
                int target=mn+1;
                row[pos[target]]=mx;
                row[pos[mx]]=target;
                swap(pos[target],pos[mx]);
            }
        }
        
        return ans;
    }
};
