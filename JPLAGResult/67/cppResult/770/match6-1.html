class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int n = row.size();
        vector<int> pos(n, 0);
        for (int i=0; i<n; ++i)
            pos[row[i]] = i;
        int ans = 0;
        for (int i=0; i<n; i+=2) 
            if (pos[row[i]^1]==i+1) 
                continue;
            else {
                int& p = pos[row[i]^1];
                int& s = pos[row[i+1]];
                s = p;
                p = i+1;
                swap(row[p], row[s]);
                ++ans;
            }
        
        return ans;
    }
};
