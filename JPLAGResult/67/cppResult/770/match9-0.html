class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int n=row.size();
	int ans=0;
	for(int i=0;i<n;i+=2)
	{
		int des;
		if(row[i]%2)des=row[i]-1;
		else des=row[i]+1;
		if(row[i+1]==des)continue;
		ans++;
		for(int j=i+2;j<n;j++)
		{
			if(row[j]==des)
			{
				int tmp=row[i+1];
				row[i+1]=row[j];
				row[j]=tmp;
			}
		}
	}
	return ans;

    }
};
