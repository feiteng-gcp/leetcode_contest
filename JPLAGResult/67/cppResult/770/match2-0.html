class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int sz = row.size();
        vector<int> mp(sz, 0);
        for (int i = 0; i < sz; i++) {
            mp[row[i]] = i;
        }
        int cnt = 0;
        for (int i = 0; i < sz; i += 2) {
            if (row[i] / 2 != row[i + 1] / 2) {
                cnt++;
                int idx = mp[row[i] % 2 ? row[i] - 1 : row[i] + 1];
                row[idx] = row[i + 1];
                mp[row[i + 1]] = idx;
            }
        }
        return cnt;
    }
};
