class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int N = row.size();
        vector<int> pos(N);
        for (int i = 0; i < N; ++i)
            pos[row[i]] = i;
        int res = 0;
        for (int i = 0; i < N; i += 2)
        {
            int peer = row[i] + ((row[i] & 1) ? -1 : 1);
            int peer_pos = pos[peer];
            if (peer_pos != i + 1)
            {
                ++res;
                pos[row[i + 1]] = peer_pos;
                swap(row[i + 1], row[peer_pos]);
            }
        }
        return res;
    }
};
