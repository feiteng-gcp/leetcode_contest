class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int n=row.size();
        vector<int>arr(n);
        int ans=0;
        for (int i=0;i<n;++i)
            arr[row[i]]=i;
        for (int i=0;i<n;i+=2){
            int k=arr[i];
            k=(k&1)?k-1:k+1;
            if (arr[i+1]==k)continue;
            int pos=row[k];
            int swaped=arr[i+1];
            swap(arr[pos],arr[i+1]);
            row[k]=i+1;
            row[swaped]=pos;
            ans++;
        }
        return ans;
    }
};
