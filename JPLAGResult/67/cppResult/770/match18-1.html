class Solution {
public:
    int minSwapsCouples(vector<int>& row) {

        unordered_map<int, int> seats;
        for(int i = 0; i < row.size(); i++)
            seats[row[i]] = i;
        vector<bool> visited(row.size(), 0);
        int ans = 0;
        for(int i = 0; i < row.size(); i += 2)
        {
            if (!visited[i])
            {
                int current = i;
                int length = 0;
                while(!visited[current])
                {
                    visited[current] = 1;
                    int next = row[current] % 2 ? row[current] - 1 : row[current] + 1;
                    next = seats[next];
                    visited[next] = 1;
                    current = next % 2 ? next - 1 : next + 1;
                    length++;
                }
                ans += length - 1;
            }
        }
        
        return ans;
    }
};
