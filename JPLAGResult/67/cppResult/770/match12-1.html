class Solution {
public:
  int minSwapsCouples(vector<int> &row) {
    int n = row.size();
    vector<int> at(n);
    for (int i = 0; i < n; ++i) {
      at[row[i]] = i;
    }
    set<int> vis;
    int ans = 0;
    for (int i = 0; i < n; ++i) {
      int x = row[i];
      if (vis.count(x)) {
        continue;
      }
      int cnt = 0;
      for (int y = x; !vis.count(y);) {
        ++cnt;
        vis.insert(y);
        // cerr << y << ",";
        y = p(y);
        vis.insert(y);
        // cerr << y << ",";
        y = row[at[y] ^ 1];
      }
      // cerr << endl;
      // cerr << cnt << endl;
      ans += cnt - 1;
    }
    return ans;
  }
  inline int p(int x) { return x ^ 1; }
};
