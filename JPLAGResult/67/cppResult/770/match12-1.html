class Solution {
public:
    int minSwapsCouples(vector<int>& row)
{

    int len = (int)row.size(), ans = 0;
    vector<int> pos(len, 0);
    for(int i = 0; i < len; i++) pos[row[i]] = i;
    for(int i = 0; i < len / 2; i++)
    {
        int L = i * 2, R = i * 2 + 1;
        if(pos[L] / 2 == pos[R] / 2) continue;
        int oR = pos[R] ^ 1;
        
        ans++;
        int temP = pos[L];
        swap(pos[L], pos[row[oR]]);
        swap(row[temP], row[oR]);
    }
    return ans;
}
};
