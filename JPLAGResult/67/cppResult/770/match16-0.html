class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int ans = 0;
        for (int i = 0; i < row.size(); i++) if (row[i]%2) row[i]--;
        for (int i = 0; i < row.size(); i+=2) {
            if (row[i] == row[i+1]) continue;
            for (int j = i+1; j < row.size(); j++) {
                if (row[i] == row[j]) {
                    swap(row[i+1],row[j]);
                    ans++;
                }
            }
        }
        return ans;
    }
};
