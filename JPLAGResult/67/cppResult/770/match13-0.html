class Solution {
public:
    int minSwapsCouples(vector<int>& row) 
    {
        int count = 0;
        map<int,int> m;
        for(int i = 0; i < row.size();i+=2)
        {
            if(row[i]%2==0)
            {//+1
                if(row[i]+1 == row[i+1])
                    continue;
                auto it = find(row.begin(), row.end(), row[i]+1);
                swap(*it, row[i+1]);
                count++;
            }else{
                //-1
                if(row[i]-1 == row[i+1])
                    continue;
                auto it = find(row.begin(), row.end(), row[i]-1);
                swap(*it, row[i+1]);
                count++;
            }
        }
        return count;
    }
};
