class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int ans=0;
        for(int i=L;i<=R;++i){
            int k=i;
            int co=0;
            while(k){
                if(k&1) ++co;
                k>>=1;
            }
            bool ju=true;
            if(co<2) ju=false;
            for(int j=2;j<co;j++) if(co%j==0) ju=false;
            if(ju) ++ans;
        }
        return ans;
    }
};
