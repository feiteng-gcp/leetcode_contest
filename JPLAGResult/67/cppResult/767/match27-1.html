class Solution {
    

    int numSetBits(int i) {
        int count = 0;
        while(i > 0) {
            
            i = (i & (i-1));
            count ++;
        }
        return count;
    }
    
public:
    int countPrimeSetBits(int L, int R) {
        
        vector<bool> primes(R + 1, true);
        primes[0] = primes[1] = false;
        for(int i = 2; i <= R; i ++) {
            for(int j = i + i; j <= R; j += i) {
                primes[j] = false;
            }

        }
        
        int count = 0;
        for(int i = L; i <= R; i ++) {
            int bit_set = numSetBits(i);
            if(primes[bit_set]) {
                count ++;
            }
        }
        return count;
    }
};
