class Solution {
public:
    bool isprime(long long int val){
	if (val <= 1)return false;
	for (long long int i = 2; i*i <= val; i++){
		if (val%i == 0LL)return false;
	}
	return true;
}
    int countPrimeSetBits(int L, int R) {
        int ans=0;
        for(int i=L;i<=R;i++){
            int cnt=0;
            int j=i;
            while(j){
                if((j&1)){
                    cnt++;
                }
                j>>=1;
            }
            if(isprime(cnt)){
             ans++;   
            }
        }
        return ans;
    }
};
