class Solution {
    int countset(int d){
        d = (d&0x55555555) + ((d>>1)&0x55555555);
        d = (d&0x33333333) + ((d>>2)&0x33333333);
        d = (d&0x0f0f0f0f) + ((d>>4)&0x0f0f0f0f);
        d = d+(d>>8);
        d = d+(d>>16);
        return d&0x3f;
    }
public:
    int countPrimeSetBits(int L, int R) {

        vector<bool>res(33,true);
        res[1] = false;
        for(int i = 2; i < 33; i++){
            if(res[i]){
                int j = i+i;
                while(j<33) {res[j] = false; j+=i;}
            }
        }

        int out = 0;
        for(int  i = L; i <= R; i++){
            int tmp = countset(i);
            if(res[tmp]){
                out++;
            }
        }
        return out;
    }
};
