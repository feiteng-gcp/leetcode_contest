class Solution {
public:
    int countPrimeBits(int n) {
        int result = 0;
        while(n) {
            result++;
            n = n & (n-1);
        }
        return result;
    }
    int countPrimeSetBits(int L, int R) {
        int result = 0;
        for (int i = L; i <= R; i++) {
            int k = countPrimeBits(i);
            if (k == 2 || k == 3 || k == 5 || k == 7 || k == 11 || k == 13 || k == 17 || k == 19) {
                result++;
            }
        }
        return result;
    }
};
