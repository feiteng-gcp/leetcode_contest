class Solution {
public:
    int countBits(int a) {
        int b = 1;
        int sum = 0;
        while (a > 0) {
            if ((a & b) != 0){ 
                sum++;
                a -= b;
            }
            b <<= 1;
        }
        return sum;
    }
    int isPrime(int a) {
        if (a==1) return false;
        if (a==2) return true;
        static set<int> hash = {2};
        if (hash.find(a) != hash.end()) return true;
        for (int i=2; i <= ceil(sqrt(a));i++) {
            if (a % i == 0) return false;
        }
        hash.insert(a);
        return true;
    }
    int countPrimeSetBits(int L, int R) {
        int sum = 0;
        for (int i=L; i <= R; i++) {
            int count = countBits(i);
            if (isPrime(count)) sum ++;
        }
        return sum;
    }
};
