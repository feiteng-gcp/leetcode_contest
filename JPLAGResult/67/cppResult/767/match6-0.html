class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int result = 0;
        for (int i = L; i <= R; i++){
            int temp = i;
            int setBit = 0;
            while(temp) {
                setBit += temp % 2;
                temp /= 2;
            }
            if (isPrime(setBit)){
                result++;
            }
        }
        return result;
    }
    
    bool isPrime(int i) {
        return (i == 2 || i == 3 || i == 5 || i == 7 || i == 11 || i == 13 || i == 17 || i == 19 );
    }
};
