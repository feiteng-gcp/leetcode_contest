class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int ans = 0;
        for (int i = L; i <= R; i++) {
            int c = 0, j = i;
            while (j) {
                c += j % 2;
                j /= 2;
            }
            if (c == 2 || c == 3 || c == 5 || c == 7 || c == 11 || c == 13 || c == 17 || c == 19 || c == 23)  {
                ans++;
            }
        }
        return ans;
    }
};
