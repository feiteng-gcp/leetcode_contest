class Solution {
public:
    
    map<int,bool> m;
    
    int get_count(int n) {
        int cnt = 0;
        while (n) {
            n = n & (n -1);
            cnt++;
        }
        
        //cout << cnt << endl;
        return cnt;
    }
    
    bool is_prime(int n) {
        if (n <= 1) return false;
        if (n <= 3) return true;
        
        if (n % 2 == 0 || n % 3 == 0) return false;
        for (int i = 5; i*i <= n; i = i + 6)
            if (n%i == 0 || n%(i + 2) == 0) return false;
       return true;
    }
    
    int countPrimeSetBits(int L, int R) {
        int cnt = 0;
        for (int i = L; i <= R; ++i) {
            if (is_prime(get_count(i)))
                cnt++;
        }
        return cnt;
    }
};
