class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        vector<bool> isPrime(32, false);
        isPrime[2] = isPrime[3] = isPrime[5] = isPrime[7] = true;
        isPrime[11] = isPrime[13] = isPrime[17] = isPrime[19] = true;
        isPrime[23] = isPrime[29] = isPrime[31] = true;
        

        int ret = 0;
        for(int i=L;i<=R;i++){
            int j = i;
            int c = 0;
            while(j > 0){
                c += (j & 1);
                j >>= 1;
            }
            if(isPrime[c])
                ret++;
        }
        
        return ret;
    }
};
