class Solution {
public:
    bool isPrime(int x) {
        if(x == 1 || x == 0)
         return false;
        
        for(int i = 2; i <= sqrt(x);i++){
            if(x % i == 0)
                return false;
        }
        return true;
    }
    
    int bitFinder(int n){
        int count = 0;
        while(n) {
            n = n & (n-1);
            count += 1;

        }
        return count;
    }
    
    int countPrimeSetBits(int L, int R) {
        int counter = 0;
        for(int i = L; i <= R;i++ ){
            int set_bit_count = bitFinder(i);
            if(isPrime(set_bit_count))
                counter++;
        }
       return counter;
    }
};
