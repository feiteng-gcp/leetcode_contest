class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int cnt = 0;
        for (int m = L; m <= R; ++m) {
            int one = 0;
            int k = m;
            while (k) {
                ++one;
                k = k&(k-1);
            }
            if (isPrim(one)) ++cnt;
        }
        return cnt;
        int n = 0;
        int M = L > R ? L : R;
        while (M) {
            ++n;
            M = M&(M-1);
        }
        int ret = isPrim(n) ? 1 : 0;
        
        int k = 31;
        int prev = 0;
        while (k >= 0) {
            int kk = 1 << k;
            if ((L&kk) != (R&kk)) break;
            if (L&&kk) ++prev;
            --k;
        }
        if (k < 0) return ret;
        
        for (int i = prev; i <= prev+k; ++i) {
            if (isPrim(i)) ret += C(i-prev, k);
        }
        return ret;
    }
    
    int isPrim(int k) {
        if (k < 2) return false;
        for (int i = 2; i * i <= k; ++i) {
            if (k%i == 0) return false;
        }
        return true;
    }
    
    int C(int m, int n) {
        long k = 1;
        for (int i = 0; i < m; ++i) {
            k *= n-i;
        }
        for (int i = 0; i < m; ++i) {
            k /= i+1;
        }
        return k;
    }
};
