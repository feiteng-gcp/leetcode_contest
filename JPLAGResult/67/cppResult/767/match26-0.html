class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int res = 0, i, t, s;
        for (i = L; i <= R; i++) {
            t = i;
            s = 0;
            while (t) {
                if (t & 1 == 1) {
                    s++;
                }
                t = t >> 1;
            }
            if (s == 2 || s == 3 || s == 5 || s == 7 || s == 11 || s == 13 || s == 17 || s == 19) {
                res++;
            }
        }
        return res;
    }
};
