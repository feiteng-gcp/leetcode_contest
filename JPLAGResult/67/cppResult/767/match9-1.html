bool flag[35];
void pre(){
    flag[2]=true;
    flag[3]=true;
    flag[5]=true;
    flag[7]=true;
    flag[11]=true;
    flag[13]=true;
    flag[17]=true;
    flag[19]=true;
    flag[23]=true;
    flag[29]=true;
    flag[31]=true;

}

bool isBitNumberPrime(int num){
    int bitNumber=0;
    while(num > 0){
        if(num & 1)
            bitNumber++;
        num>>=1;
    }
    return flag[bitNumber];
}

class Solution {

public:
    int countPrimeSetBits(int l, int r) {
        pre();
        int ans=0;
        for(int i=l;i<=r;i++){
            ans+= isBitNumberPrime(i) ? 1 : 0;
        }
        return ans;
    }
};
