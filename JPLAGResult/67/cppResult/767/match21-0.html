class Solution {
public:
    int countbit(int i) {
        int numbit = 0;
        while (i) {
            numbit += (i & 1);
            i >>= 1;
        }
        return numbit;
    }
    
    bool isprime(int x) {
        if (x == 1) return false;
        for (int i = 2; i * i <= x; i++) {
            if (x % i == 0) {
                return false;
            }
        }
        return true;
    }
    
    int countPrimeSetBits(int L, int R) {
        int ans = 0;
        for (int i = L; i <= R; i++) {
            int numbits = countbit(i);
            
            if (isprime(numbits)) {
                ans++;
            }
        }
        
        return ans;
    }
};
