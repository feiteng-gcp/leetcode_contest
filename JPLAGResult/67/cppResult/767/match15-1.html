class Solution {
public:
   bool isPrere(int num)
{
	int count = 0; 
	while (num != 0)
	{
		if (num & 1 == 1) count++;
		num >>= 1;
	}

	if (count == 1) return false;
	for (int i = 2; i <= sqrt(count); i++)
	{
		if (count%i == 0) return false;
	}
	return true;
}


int countPrimeSetBits(int L, int R) {

	int count = 0;
	for (int i = L; i <= R; i++)
	{
		if (isPrere(i)) count++;
	}
	return count;
}
};
