class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int res = 0;
        for (int i = L;i <= R;++i) {
            int cnt = 0;
            for (int x = i;x > 0;x /= 2) {
                if (x % 2 == 1) ++ cnt;
            }
            if (prime(cnt)) {
                ++ res;
            }
        }
        return res;
    }
    
    bool prime(int x) {
        if (x == 1) return false;
        for (int i = 2;i * i <= x;++i)
            if (x % i == 0) return false;
        return true;
    }
};
