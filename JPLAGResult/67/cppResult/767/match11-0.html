class Solution {
public:
    // 10^6<2^20
    int gets(int i){
        int ret=0;
        int curr=i;
        while(curr>0){
            ret+=curr%2;
            curr=curr>>1;
        }
        return ret;
    }
    int countPrimeSetBits(int L, int R) {
        int ret=0;
        for(int i=L; i<=R; i++){
            int res=gets(i);
            //cout<<res<<" ";
            if(res==2 || res==3 || res==5 || res==7 || res==11 || res==13 || res==17 || res==19){
                ret++;
            }
        }
        return ret;
    }
};
