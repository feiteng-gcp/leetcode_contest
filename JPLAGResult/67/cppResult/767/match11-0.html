class Solution {
public:
    int count_one(int n) {
        int ans = 0;
        while (n) {
            ans += n & 1;
            n >>= 1;
        }
        return ans;
    }
    bool is_prime(int n) {
        if (n == 2 || n == 3 || n == 5 || n == 7 || n == 11 || n == 13 || n == 17 || n == 19 || n == 23 || n == 29 || n == 31) 
            return true;
        else
            return false;
    } 
    int countPrimeSetBits(int L, int R) {
        int ans = 0;
        for (int i = L; i <= R; i++) {
            if (is_prime(count_one(i))) {
                ans += 1;
            }
        }
        return ans;
    }
};
