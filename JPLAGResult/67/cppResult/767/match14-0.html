class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int result = 0;
        for (int i = L; i <= R; i++) {
            if (isPrime(getSetBit(i))) result++;
        }
        return result;
    }
    
    int getSetBit(int a) {
        int result = 0;
        while(a > 0) {
            if (a % 2) result++;
            a = a / 2;
        }
        return result;
    }
    
    bool isPrime(int a) {
        if (a == 1) return false;
        int n = sqrt(a);
        for (int i = 2; i <= n; i++) {
            if ((a % i) == 0) return false;
        }
        return true;
    }
};
