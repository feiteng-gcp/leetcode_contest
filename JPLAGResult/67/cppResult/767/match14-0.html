class Solution {
public:
    int bitSetCount(int num) {
        int count = 0;
        while(num) {
            count++;
            num = num & (num-1);
        }
        return count;
    }
    
    bool isPrime(int num) {
      if (num <= 1) return false;
      int maxmult = num/2;
      for (int i = 2; i <= maxmult; i++) {
          if (num%i == 0) return false;
      }
        return true;
    }
    
    int countPrimeSetBits(int L, int R) {
        int primecount = 0;
        for (int i = L; i <= R; i++) {
            int bcount = bitSetCount(i);
            if (isPrime(bcount)) primecount++;
        }
        return primecount;
    }
};
