class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int cnt=0;
        for(int i=L;i<=R;++i){
            int n=bitset<32>(i).count();
            if(isPrime(n)) cnt++;
        }
        return cnt;
    }
    bool isPrime(int n){
        if(n<2) return false;
        for(int i=2;i<(int)sqrt(n)+1;++i)
            if(n%i==0) return false;
        return true;
    }
};
