class Solution {
    int countBits(int x) {
        int res = 0;
        while (x > 0) {
            res += (x & 1);
            x >>= 1;
        }
        
        return res;
    }
    
    bool isPrime(int x) {
        if (x == 1)
            return false;
        
        for (int d = 2; d < x; d++) {
            if (x % d == 0)
                return false;
        }
        
        return true;
    }
    
public:
    
    
    int countPrimeSetBits(int L, int R) {
        int ans = 0;
        
        for (int x = L; x <= R; x++) {
            if (isPrime(countBits(x)))
                ans++;
        }
        
        return ans;
    }
};
