class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int cnt = 0;
    
        for (int i = L; i <= R; i++) {
         //   cout << i <<  endl;
            int num = i, bits = 0;
            while (num) {
                num = num & (num - 1);
                bits++;
            }
        //    cout << bits << " " << i << endl;
            if (isprime(bits)) {
                cnt++;
            }
        }
        return cnt;
    }
    
    bool isprime(int num) {
        if (num < 2) return false;
        if (num == 2) return true;
        for (int i = 2; i * i <= num; i++) if (num % i == 0) return false;
        return true;
    }
};
