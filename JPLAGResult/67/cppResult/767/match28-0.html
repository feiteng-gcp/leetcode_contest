class Solution {
    bool isPrime(int n)
    {
        if(n==1)
            return false;
        
        if(n<=2)
            return true;
        
        int l=sqrt(n);
        for(int i=2;i<=l;++i)
        {
            if(n%i==0)
                return false;
        }
        
        return true;
    }
    
    int countBits(int n)
    {
        int count=0;
        while(n)
        {
            count++;
            n=n&(n-1);
        }
        
        return count;
    }
public:
    int countPrimeSetBits(int L, int R) {
        
        int count=0;
        
        for(int i=L;i<=R;++i)
        {
            if(isPrime(countBits(i)))   
                count++;
                
        }
        
        return count;
    }
};
