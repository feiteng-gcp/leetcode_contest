class Solution {
    int cnt(int x) {
        int res = 0;
        while (x) {
            ++res;
            x &= x - 1;
        }
        return res;
    }
    bool prime(int x) {
        if (x == 1) return false;
        for (int i = 2; i <= sqrt(x); ++i) {
            if (x % i == 0) return false;
        }
        return true;
    }
public:
    int countPrimeSetBits(int L, int R) {
        int res = 0;
        for (int i = L; i <= R; ++i) {
            if (prime(cnt(i))) ++res;
        }
        return res;
    }
};
