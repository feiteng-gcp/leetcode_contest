class Solution {
public:

    unordered_set<int> primes{2, 3, 5, 7, 11, 13, 17, 19};
    int bit(int n){
        int res = 0;
        while(n){
            if(n&1) ++res;
            n >>=1;
        }
        return res;
    }
    int countPrimeSetBits(int L, int R) {
        int res = 0;
        for(int i = L; i <= R; ++i){
            int b = bit(i);
            if(primes.find(b) != primes.end()) ++res;
        }
        return res;
    }
};
