class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int count=0;
        for(int x=L; x<=R; x++)
        {
            if (isSetBitsPrime(x))
                count++;
        }
        return count;
        
    }
    
    bool isSetBitsPrime (int x)
    {
        bitset<32> bSet(x);
        int n= bSet.count();
        return isPrime(n);
    }
    
    bool isPrime (int n)
    {
        if(n<=1)
            return false;
        
        for(int i=2; i<=n/2; i++)
        {
            if(n%i==0)
                return false;
        }
        
        return true;
        
    }
};
