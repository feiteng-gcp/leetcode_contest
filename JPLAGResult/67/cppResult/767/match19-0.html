class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int t[] = {2,3,5,7,11,13,17,19};
        int ans = 0;
        for (int i = L; i <= R; ++i) {
            int numbit = calcNumBit(i);
            for (int j = 0; j < 8; ++j) {
                if (numbit == t[j]) {
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
    
    int calcNumBit(int x) {
        int ret = 0;
        while (x) {
            x = (x & (x-1));
            ret++;
        }
        return ret;
    }
};
