class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int count = 0;
        
        for (int i = L; i <= R; ++i) {
            int countOfOnes = 0, cur = i;
            
            while (cur) {
                if (cur & 1) countOfOnes++;
                cur >>= 1;
            }
            
            if (isPrime(countOfOnes)) count++;
        }
        
        return count;
    }
    
private:
    bool isPrime(int n) {
        if (n < 2) return false;
            

        bool prime = true;
        for (int j = 2; j <= n / 2; ++j)
        {
            if(n % j == 0)
            {
                prime = false;
                break;
            }
        }
        
        return prime;
    }
};
