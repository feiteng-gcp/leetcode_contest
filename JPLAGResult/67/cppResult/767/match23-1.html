class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int p[]={2,3,5,7,11,13,17,19};
        int sum = 0;
        for (int i=L; i<=R; ++i) {
            int count = 0;
            for (int x=i; x; x>>=1) {
                if (x&1) {
                    ++count;
                }
            }
            for (int i=0; i<8; ++i) {
                if (count == p[i]) {
                    ++sum;
                }
            }
        }
        return sum;
    }
};
