class Solution {

    bool IsValid(int x) {
        int s = 0;
        while (x > 0) {
            s += x%2;
            x >>= 1;
        }
        return (s == 2 || s == 3 || s == 5 || s == 7 || s == 11 || s == 13 || s == 17 || s == 19);
    }
public:
    int countPrimeSetBits(int L, int R) {
        int ans = 0;
        for (int i = L; i <= R; ++i) {
            if (IsValid(i)) ++ans;
        }
        return ans;
    }
};
