class Solution {
public:
    int nrBits(int n) {
        int ret = 0;
        for (int i = 0; i < 32; i++)
            ret += !!(n & (1 << i));

        return ret;
    }

    int countPrimeSetBits(int L, int R) {
        vector<bool> primes(32, true);

        primes[0] = primes[1] = false;

        for (int i = 0; i < primes.size(); i++) {
            if (!primes[i]) continue;

            for (int j = i * 2; j < primes.size(); j += i)
                primes[j] = false;
        }

        int ret = 0;

        for (int n = L; n <= R; n++) {
            int bits = nrBits(n);

            if (primes[bits]) ret++;
        }

        return ret;
    }
};
