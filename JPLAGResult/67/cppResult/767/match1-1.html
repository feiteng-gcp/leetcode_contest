class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        vector<bool> isPrime(33, false);
        for(int i = 2; i < isPrime.size(); ++i)
        {

            bool res = true;
            for(int k = 2; k * k <= i; ++k)
            {
                if(i % k == 0)
                {
                    res = false;
                    break;
                }
            }
            isPrime[i] = res;
        }
        

        int ret = 0;
        for(int i = L; i <= R; ++i)
        {
            int t = i;
            int count = 0;
            while(t > 0)
            {
                count += t & 1;
                t = (t >> 1);
            }
            
            if(isPrime[count])
            {
                ret++;
            }
        }
        
        return ret;
    }
};
