class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int prime = 0;
        for (int i=L; i<=R; i++) {
            int n = i;
            int count = 0;
            while (n)
            {
                count += n & 1;
                n >>= 1;
            }
            if(count == 2 || count == 3 || count == 5 || count == 7 || count == 11 || count == 13 || count == 17 || count == 19)
                prime++;
        }
        return prime;
    }
};
