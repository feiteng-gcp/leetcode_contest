class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        vector<bool> prime(33, false);
        for (int i = 2; i < 33; ++i) {
            bool valid = true;
            for (int j = 2; j < i; ++j)
                if (i % j == 0) {
                    valid = false;
                    break;
                };
            if (valid) {
                cout << i << endl;
                prime[i] = true;
            };
            
        };
        
        int ans = 0;
        
        for (int i = L; i <= R; ++i) {
            int k = i;
            int count = 0 ;
            while (k) {
                if (k & 1) count++;
                k >>= 1;
            };
            
            if (prime[count]) ans++;
        };
        
        return ans;
        
        
        
    }
};
