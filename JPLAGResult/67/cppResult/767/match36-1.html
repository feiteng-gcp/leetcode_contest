class Solution {
public:
    bool isOK(int n){
        if(n == 2 || n == 3 || n == 5 || n == 7 || n == 11 || n == 13 || n == 17 || n == 19 || n == 23 || n == 29 || n == 31 || n == 37)
            return true;
        else return false;
    }
    
    int cntBit(int n){
        int res = 0;
        while(n){
            n = n & (n-1);
            res++;
        }
        return res;
    }
    
    int countPrimeSetBits(int L, int R) {

        int cnt = 0;
        for(int i = L; i <= R; i++){
            int bit = cntBit(i);
            //cout << bit << endl;
            if(isOK(bit)){
                cnt++;
            }
        }
        return cnt;
    }
};
