class Solution {
public:
    int countPrimeSetBits(int L, int R) {

        int res = 0;
        for (int i = L; i <= R; ++i) {
            if (isPrime(countBits(i))) res++;
        }
        return res;
    }

    int countBits(int i) {
        int res = 0;
        while (i) {
            i &= (i - 1);
            res++;
        }
        return res;
    }
    bool isPrime(int num) {
        if (num == 1) return false;
        if (num == 2 || num == 3 || num == 5) return true;
        for (int i = 2; i <= num / 2; ++i) {
            if (num % i == 0) return false;
        }
        return true;
    }
};
