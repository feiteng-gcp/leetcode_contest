class Solution {
public:

    int countPrimeSetBits(int L, int R) {
        vector<bool> flag(50);
        flag[0] = flag[1] = true;
        for (int i = 0; i < 50; ++i) {
            if (flag[i]) continue;
            for (int j = i + i; j < 50; j += i) {
                flag[j] = true;
            }

        }
        int ret = 0;
        for (int i = L; i <= R; ++i) {
            int x = __builtin_popcount(i);
            if (!flag[x]) ++ret;
        }
        return ret;
        
    }
};
