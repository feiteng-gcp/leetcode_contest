class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int ans=0;
        for(int i=L; i<=R; ++i) {
            int cnt=0;
            for(int j=0; j<31; ++j)
                cnt += (i>>j)%2;
            ans += (cnt==2 || cnt==3 || cnt==5 || cnt==7 || cnt==11 || cnt==13 || cnt==17 || cnt==19 || cnt==23 || cnt==29);
            
            //cout << i << " " << cnt << endl;
        }
        
        return ans;
    }
};
