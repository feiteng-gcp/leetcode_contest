class Solution {
public:
    int countSetBits(int n) {
        int count = 0;
        while (n) {
          n &= (n-1) ;
          count++;
        }
        return count;
    }
    
    bool is_prime(int n) {
        if (n == 1) return false;
        for (int i = 2; i <= sqrt(n); ++i) {
            if (n % i == 0) return false;
        }
        return true;
    }
    
    int countPrimeSetBits(int L, int R) {
        int res = 0;
        for (int i = L; i <= R; ++i) {
            int count = countSetBits(i);
            res += is_prime(count);
        }
        return res;
    }
};
