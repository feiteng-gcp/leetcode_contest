class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int res = 0;
        
        for(int i = L; i <= R; ++i){
           // cout << i << ":" << countBit(i) << endl;
            if(isPrime(countBit(i)))
                ++res;
        }
        return res;
    }

private:
    bool isPrime(int num){
        if(num <= 1)
            return false;
        
        for(int i = 2; i <= sqrt(num); ++i){
            if (num % i == 0)
                return false;
        }
        return true;
    }
    
    int countBit(int num){
        int count = 0;
        while(num !=0){
            if(num % 2)
                count++;
            num /= 2;
        }
        return count;
    }
    
};
