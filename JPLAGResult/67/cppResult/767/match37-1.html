class Solution {
public:
	int countPrimeSetBits(int L, int R) {

		int total = 0;
		for (int i = L; i <= R; i++) {
			int b = countBits(i);
			if (isPrime(b)) {
				total++;
			}
		}
		return total;
	}

	int countBits(int n) {
		int total = 0;
		while (n > 0) {
			int r = n % 2;
			if (r == 1) {
				total++;
			}
			n /= 2;
		}
		return total;
	}

	bool isPrime(int n) {
		return n == 2 || n == 3 || n == 5 || n == 7 || n == 11 || n == 13 || n == 17 || n == 19;
	}
};
