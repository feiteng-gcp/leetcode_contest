class Solution {
public:
    int bits(int x){
        if (x==0) return 0;
        return x%2+bits(x/2);
    }
    
    int prime(int x){
        if (x<2) return 0;

        if (x==2 || x==3) return 1;
        for (int i=2; i*i<=x; i++)
            if (x%i==0) return 0;
        return 1;
    }
    
    int countPrimeSetBits(int L, int R) {
        int ans=0;
        for (int i=L; i<=R; i++) {
            ans+=prime(bits(i));
            //cout << i << " " << bits(i) << " " << prime(bits(i)) << endl;
        }
        return ans;
    }
};
