class Solution {
public:
    bool isPrime(int n) {
        if(n == 1) return false;
        int nr = sqrt(n);
        for(int i = 2; i <= nr; i++) {
            if(n % i == 0) return false;
        }
        return true;
    }

    int cntSetBits(int n) {
        int res = 0;
        while(n) {
            if(n % 2 == 1) res++;
            n /=2;
        }
        return res;
    }

    int countPrimeSetBits(int L, int R) {
        int res = 0;
        for(int n = L; n <= R; n++) {
            if(isPrime(cntSetBits(n))) res++;
        }
        return res;
    }
};
