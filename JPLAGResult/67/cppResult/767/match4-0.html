class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int count=0;
        for(int i=L;i<=R;i++){
            if(isValid(i)) count++;
        }
        return count;
    }
private:
    bool isValid(int n){
        int count = 0;
        while(n){
            n = n & (n-1);
            count++;
        }
        if(count==1) return false;
        for (int j = 2; j < count; j++) {  
            if (count % j == 0)  return false;
        }  
        return true;
    }
};
