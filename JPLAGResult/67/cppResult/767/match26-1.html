class Solution {
public:
	int countPrimeSetBits(int L, int R) {
		int i, cnt = 0;
		for (i = L; i < R + 1; i++) {
			bitset<32> b(i);
			int bs = b.count();
			if (bs == 2 || bs == 3 || bs == 5 || bs == 7 || bs == 11 || bs == 13 || bs == 17 || bs == 19)
				cnt++;
		}
		return cnt;
	}
};
