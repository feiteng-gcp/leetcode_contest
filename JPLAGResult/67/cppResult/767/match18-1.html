class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int res = 0;
        for(int i = L; i <= R; ++i) {
            if(isP(getB(i))) ++res;
        }
        return res;
    }
    
    int getB(int x) {
        int res = 0;
        while(x > 0) {
            if(x % 2) ++res;
            x /= 2;
        }

        return res;
    }
    
    bool isP(int x) {
        if(x == 1) return false;
        for(int i = 2; i * i <= x; ++i) {
            if(x % i == 0) return false;
        }
        return true;
    }
};
