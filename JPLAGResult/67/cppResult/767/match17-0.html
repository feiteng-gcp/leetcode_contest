class Solution {
public:
    bool is[32];
    bool isp(int num)
    {
        int cnt = 0;
        while (num != 0)
        {
            num &= (num - 1);
            ++cnt;
        }
        return is[cnt];
    }
    int countPrimeSetBits(int L, int R)
    {
        for (int i = 0; i < 32; ++i)
            is[i] = false;
        is[2] = true;
        is[3] = true;
        is[5] = true;
        is[7] = true;
        is[11] = true;
        is[13] = true;
        is[17] = true;
        is[19] = true;
        is[23] = true;
        is[29] = true;
        is[31] = true;
        
        int ans = 0;
        for (int num = L; num <= R; ++num)
        {
            if (isp(num))
                ++ans;
        }
        return ans;
    }
};
