class Solution {
public:
    bool prime(int n){
        if(n <= 1)
            return false;
        for(int i = 2; i < n; i++){
            if(n % i == 0)
                return false;
        }
        return true;
    }
    int countPrimeSetBits(int L, int R) {
        int res = 0;
        for(int i = L; i <= R; i++){
            int x = __builtin_popcount(i);
            if(prime(x))
                res++;
        }
        return res;
    }
};
