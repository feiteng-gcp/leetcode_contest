class Solution {
public:
    
    bool isPrime(int n) {
        
        if (n == 1) return false;
        if (n == 2) return true;
        
        int sq = sqrt(n);
        

        for (int i = 2; i <= sq; i++) {
            if (n%i == 0) return false;
        }
        
        return true;        
    }
    
    int countPrimeSetBits(int L, int R) {
        
        int res = 0;
        
        for (int i = L; i <= R; i++) {
            int j = 1;
            int cnt = 0;
            while (j <= 1000000) {
                if (j&i) cnt++;
                j = j << 1;
            }
            
            if (isPrime(cnt)) {
                res++;
            }
        }
        
        return res;
    }
};
