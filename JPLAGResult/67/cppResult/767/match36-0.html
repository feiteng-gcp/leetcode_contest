class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int res = 0;
        for (int i = L; i <= R; i++){
            int tmp = i;
            int cnt = 0;
            while (tmp > 0){
                // cout<<tmp<<" "<<cnt<<endl;
                cnt = cnt + (tmp & 1);
                tmp = tmp >> 1;
            }
            // cout<<cnt<<endl;
            if (cnt == 2 || cnt == 3 || cnt == 5 || cnt == 7 || cnt == 11 || cnt == 13 || cnt == 17 || cnt == 19)
                res++;
            // cout<<res<<endl;
        }
        return res;
    }
};
