class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int ret = 0;
        for (int i = L; i <= R; ++ i) {
            int cnt = 0, tmp = i;
            while (tmp != 0) {
                cnt += (tmp % 2 == 1);
                tmp /= 2;
            }
            if (cnt < 2) continue;
            bool h = true;
            for (int j = 2; j <= sqrt(cnt); ++ j)
                if (cnt % j == 0) {
                    h = false;
                    break;
                }
            if (h) ret ++;
        }
        return ret;
    }
};
