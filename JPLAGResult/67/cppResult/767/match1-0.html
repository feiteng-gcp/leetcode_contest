class Solution {
public:
    int countPrimeSetBits(int L, int R) {
        int count = 0;
        for (int i = L; i <= R; i++) {
            int cur = i;
            int tmp = 0;
            while (cur) {
                tmp += cur % 2;
                cur /= 2;
            }
            
            bool find = tmp <= 1;
            for (int i = 2; i * i <= tmp; i++) {
                if (tmp % i == 0) {
                    find = true;
                    break;
                }
            }
            
            if (!find) {
                count++;
            }
        }
        
        return count;
    }
};
