class Solution {
public:
    int countOne (int i) {

        int count = 0;
        while (i) {
            count++;
            i = ((i - 1) & i);
        }
        return count;
    }
    int countPrimeSetBits(int L, int R) {
        int count = 0;
        int prime[33] = {0};

        prime[2] = 1;
        prime[3] = 1;
        prime[5] = 1;
        prime[7] = 1;
        prime[11] = 1;
        prime[13] = 1;
        prime[17] = 1;
        prime[19] = 1;
        prime[23] = 1;
        prime[29] = 1;
        prime[31] = 1;
        for (int i = L; i <= R; i++) {
           if (prime[countOne(i)])
               count++;
        }
        return count;
    }
};
