class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        set<int> X[510],Y[510];
        for (auto i : mines) {
            X[i[0]].insert(i[1]);
            Y[i[1]].insert(i[0]);

        }
        int ans = 0;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                auto it = X[i].lower_bound(j);
                int x2 = N-1, x1 = 0;
                if (it != X[i].end()) x2 = *it-1;
                if (it != X[i].begin()) {
                    it--;
                    x1 = *it+1;
                }
                it = Y[j].lower_bound(i);
                int y1 = 0, y2 = N-1;
                if (it != Y[j].end()) y2 = *it-1;
                if (it != Y[j].begin()) {
                    it--;
                    y1 = *it+1;
                }
                ans = max(ans,1+min(min(j-x1,x2-j),min(i-y1,y2-i)));
            }
        }
        return ans;
    }
};
