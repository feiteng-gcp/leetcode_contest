class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        vector<vector<int>> res(N, vector<int>(N, 0));
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                res[i][j] = 1 + min(N-1-i, min(i, min(j,N-1-j)));
            }
        }
        
//         for (auto &z : mines) {
//             res[z[0]][z[1]] = 0;
//         }
        
        
        
        for (auto &z : mines) {
            int x = z[0];
            int y = z[1];
            for (int j = 0; j < N; j++) {
                res[x][j] = min(res[x][j], abs(j-y));
            }
            for (int i = 0; i < N; i++) {
                res[i][y] = min(res[i][y], abs(i-x));
            }
        }
        

        int r = 0;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                r = max(r, res[i][j]);
            }
        }
        return r;
    }
};
