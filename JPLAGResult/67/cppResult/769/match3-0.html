class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        vector<vector<int>> matrix(N, vector<int>(N, 1));
        for (const auto& mine: mines)
            matrix[mine[0]][mine[1]]=0;
        int ans=0;
        for (int i=0; i<N; i++) if (ans<=i && i+ans<N)
            for (int j=0; j<N; j++) if (ans<=j && j+ans<N) {
                int k=0;
                while (i-k>=0 && i+k<N && j-k>=0 && j-k<N && matrix[i-k][j]==1 && matrix[i][j-k]==1 && matrix[i][j+k]==1 && matrix[i+k][j]==1) k++;
                if (ans<k) ans=k;
            }
        return ans;
    }
};
