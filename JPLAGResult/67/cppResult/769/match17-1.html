int cache[4][500][500];
char mm[500][500];
class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        memset(mm, 1, sizeof(mm));
        for(auto i: mines) mm[i[0]][i[1]] = 0;
        memset(cache, 0, sizeof(cache));
        for(int i=0; i<N; ++i) {
            for(int j=0; j<N; ++j) {
                if (mm[i][j]) {
                    cache[0][i][j] = 1;
                    cache[2][i][j] = 1;
                    if (j>0) cache[0][i][j] += cache[0][i][j-1];
                    if (i>0) cache[2][i][j] += cache[2][i-1][j];
                }
            }
            for(int j=N-1; j>=0; --j) {
                if (mm[i][j]) {
                    cache[1][i][j] = 1;
                    if (j+1<N) cache[1][i][j] += cache[1][i][j+1];
                }

            }
        }
        for(int i=N-1; i>=0; --i) {
            for(int j=N-1; j>=0; --j) {
                if (mm[i][j]) {
                    cache[3][i][j] = 1;
                    if (i+1<N) cache[3][i][j] += cache[3][i+1][j];
                }

            }
        }
        int res = 0;
        for(int i=0; i<N; ++i) {
            for(int j=0; j<N; ++j) {
                int cur = INT_MAX;
                for(int k=0; k<4; ++k) {
                    cur = min(cur, cache[k][i][j]);
                }
                res = max(res, cur);
            }
        }
        return res;
    }
};
