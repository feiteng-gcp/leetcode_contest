class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        vector<vector<int>> field(N, vector<int>(N, 1));
        for(auto& v: mines)
            field[v[0]][v[1]]=0;
        
        vector<vector<int>> dpLeft(N, vector<int>(N));
        vector<vector<int>> dpRight(N, vector<int>(N));
        vector<vector<int>> dpUp(N, vector<int>(N));
        vector<vector<int>> dpDown(N, vector<int>(N));
        
        for(int i=0; i<N; ++i) {
            int cnt=0;
            for(int j=0; j<N; ++j) {
                if(field[i][j]==1)
                    dpLeft[i][j]=++cnt;
                else
                    dpLeft[i][j]=cnt=0;
            }
        }
        
        for(int i=0; i<N; ++i) {
            int cnt=0;
            for(int j=N-1; j>=0; --j) {
                if(field[i][j]==1)
                    dpRight[i][j]=++cnt;
                else
                    dpRight[i][j]=cnt=0;
            }
        }
        
        for(int i=0; i<N; ++i) {
            int cnt=0;
            for(int j=0; j<N; ++j) {
                if(field[j][i]==1)
                    dpDown[j][i]=++cnt;
                else
                    dpDown[j][i]=cnt=0;
            }
        }
        
        for(int i=0; i<N; ++i) {
            int cnt=0;
            for(int j=N-1; j>=0; --j) {
                if(field[j][i]==1)
                    dpUp[j][i]=++cnt;
                else
                    dpUp[j][i]=cnt=0;
            }
        }
        
        int ans=0;
        for(int i=0; i<N; ++i)
        for(int j=0; j<N; ++j) {
            int tmp = min(min(dpLeft[i][j],dpRight[i][j]),min(dpDown[i][j],dpUp[i][j]));
            ans = max(ans,tmp);
        }
        
        return ans;
        
    }
};
