class Solution {
public:
    int a[502][502];
    int b[502][502];
    int c[502][502];
    int d[502][502];
    vector<vector<int> > v;
    
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        v.assign(N,vector<int>(N,1));
        for(int i=0;i<mines.size();i++){
            v[mines[i][0]][mines[i][1]]=0;
        }
        memset(a,0,sizeof(a));
        memset(b,0,sizeof(b));
        memset(c,0,sizeof(c));
        memset(d,0,sizeof(d));
        for(int i=0;i<v.size();i++){
            int cc=0;
            for(int j=0;j<v[i].size();j++){
                if(v[i][j]==0){
                    cc=0;
                }
                else{
                    cc++;
                }
                a[i][j]=cc;
            }
        }
        for(int i=0;i<v.size();i++){
            int cc=0;
            for(int j=v[i].size()-1;j>=0;j--){
                if(v[i][j]==0){
                    cc=0;
                }
                else{
                    cc++;
                }
                b[i][j]=cc;
            }
        }
        for(int j=0;j<v[0].size();j++){
            int cc=0;
            for(int i=0;i<v.size();i++){
                if(v[i][j]==0){
                    cc=0;
                }
                else{
                    cc++;
                }
                c[i][j]=cc;
            }
        }
         for(int j=0;j<v[0].size();j++){
            int cc=0;
            for(int i=v.size()-1;i>=0;i--){
                if(v[i][j]==0){
                    cc=0;
                }
                else{
                    cc++;
                }
                d[i][j]=cc;
            }
        }
        int ans=0;
        for(int i=0;i<v.size();i++){
            for(int j=0;j<v[i].size();j++){
                int k=a[i][j];
                k=min(k,b[i][j]);
                k=min(k,c[i][j]);
                k=min(k,d[i][j]);
                ans=max(ans,k);
            }
        }
        return ans;
    }
};
