bool mp[505][505];
int l[505][505], r[505][505], t[505][505], b[505][505];

class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        memset(mp, 1, sizeof(mp));
        memset(l, 0, sizeof(l));
        memset(r, 0, sizeof(r));
        memset(t, 0, sizeof(t));
        memset(b, 0, sizeof(b));
        for (const auto& mine : mines)
            mp[mine[0]][mine[1]] = 0;
        for (int i = 0; i < N; ++i)
        for (int j = 0; j < N; ++j)
        {
            if (mp[i][j])
            {
                l[i + 1][j + 1] = l[i + 1][j] + 1;
                t[i + 1][j + 1] = t[i][j + 1] + 1;
            }
        }
        for (int i = N - 1; i >= 0; --i)
        for (int j = N - 1; j >= 0; --j)
        {
            if (mp[i][j])
            {
                r[i + 1][j + 1] = r[i + 1][j + 2] + 1;
                b[i + 1][j + 1] = b[i + 2][j + 1] + 1;

            }
        }
        int res = 0;
        for (int i = 0; i < N; ++i)
        for (int j = 0; j < N; ++j)
        {
            if (!mp[i][j])
                continue;
            res = max(res, min(min(l[i + 1][j + 1], r[i + 1][j + 1]), min(t[i + 1][j + 1], b[i + 1][j + 1])));
        }
        return res;
    }
};
