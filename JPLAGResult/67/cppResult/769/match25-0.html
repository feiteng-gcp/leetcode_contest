class Solution {
public:
    int g[505][505];
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        for(int i=0;i<N;i++)
	{
		for(int j=0;j<N;j++)
		{
			g[i][j]=1;
		}
	}
	int cnt=mines.size();
	for(int i=0;i<cnt;i++)
	{
		g[mines[i][0]][mines[i][1]]=0;
	}
	int ans=0;
	for(int i=0;i<N;i++)
	{
		for(int j=0;j<N;j++)
		{
			for(int c=0;;c++)
			{
				if(i-c>=0&&j-c>=0&&i+c<N&&j+c<N&&g[i-c][j]&&g[i][j-c]&&g[i+c][j]&&g[i][j+c])continue;
				ans=max(ans,c);
				break;
			}
		}
	}
	return ans;
    }
};
