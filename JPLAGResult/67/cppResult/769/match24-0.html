class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        vector<vector<int> >  g(N, vector<int>(N, 1));
        vector<vector<int> >  top(N, vector<int>(N, 0));
        vector<vector<int> >  down(N, vector<int>(N, 0));
        vector<vector<int> >  left(N, vector<int>(N, 0));
        vector<vector<int> >  right(N, vector<int>(N, 0));
        
        for(int i = 0; i < mines.size(); i++){
            int r = mines[i][0];
            int c = mines[i][1];
            g[r][c] = 0;
        }
        
        for(int i = 0; i < N; i++){
            for(int j = 0; j < N; j++){
                if(g[i][j] == 0) continue;
                
                top[i][j] = 1;
                left[i][j] = 1;
                
                if(i > 0){
                    top[i][j] += top[i-1][j];
                }
                if(j > 0){
                    left[i][j] += left[i][j-1];
                }
            }
        }
        
        for(int i = N-1; i >= 0; i--){
            for(int j = N-1; j >= 0; j--){
                if(g[i][j] == 0) continue;
                
                down[i][j] = 1;
                right[i][j] = 1;
                
                if(i < N-1){
                    down[i][j] += down[i+1][j];
                }
                if(j < N-1){
                    right[i][j] += right[i][j+1];
                }
            }
        }
        
        int res = 0;
        for(int i = 0; i < N; i++){
            for(int j = 0; j < N; j++){
                int l = min(top[i][j], down[i][j]);
                l = min(l, left[i][j]);
                l = min(l, right[i][j]);
                res = max(res, l);
            }
        }
        
        return res;
    }
};
