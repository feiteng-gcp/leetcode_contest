class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        unordered_set<int> st;
        for (auto& m : mines) {
            st.insert(m[0] * N + m[1]);
        }
        vector<vector<int>> left(N + 2, vector<int>(N + 2, 0)), up(N + 2, vector<int>(N + 2, 0));
        for (int i = 1; i <= N; i++) {
            for (int j = 1; j <= N; j++) {
                if (!st.count((i - 1) * N + (j - 1))) {
                    left[i][j] = left[i][j - 1] + 1;
                    up[i][j] = up[i - 1][j] + 1;
                }
            }

        }
        int res = 0;
        for (int i = 1; i <= N; i++) {
            for (int j = 1; j <= N; j++) {
                for (int k = 0; ; k++) {
                    if (left[i][j + k] > 2 * k && up[i + k][j] > 2 * k) {
                        res = max(res, k + 1);
                    }
                    else {
                        break;
                    }
                }
            }
        }
        return res;
    }
};
