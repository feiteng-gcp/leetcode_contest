class Solution {
public:
	int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
		vector<vector<bool>> g(N, vector<bool>(N, true));
		for (vector<int> mine : mines) {
			g[mine[0]][mine[1]] = false;
		}

		vector<vector<int>> h(N, vector<int>(N, 0));
		for (int i = 0; i < N; i++) {
			int len = 0;
			for (int j = 0; j < N; j++) {
				if (g[i][j]) {
					len++;
					if (j == N - 1) {
						markH(h, i, j, len);
					}
				}

				if (!g[i][j]) {
					if (len != 0) {
						markH(h, i, j - 1, len);
					}
					len = 0;
				}
			}
		}

		vector<vector<int>> v(N, vector<int>(N, 0));
		for (int j = 0; j < N; j++) {		
			int len = 0;
			for (int i = 0; i < N; i++) {
				if (g[i][j]) {
					len++;
					if (i == N - 1) {
						markV(v, j, i, len);
					}
				}

				if (!g[i][j]) {
					if (len != 0) {
						markV(v, j, i - 1, len);
					}
					len = 0;
				}
			}
		}

		int maxOrder = 0;
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				int order = min(h[i][j], v[i][j]);
				maxOrder = max(maxOrder, order);
			}
		}
		return maxOrder;
	}
private:
	void markH(vector<vector<int>>& h, int i, int endJ, int len) {
		int startJ = endJ - len + 1;
		for (int j = startJ; j <= endJ; j++) {
			h[i][j] = min(j - startJ + 1, endJ - j + 1);
		}
	}

	void markV(vector<vector<int>>& v, int j, int endI, int len) {
		int startI = endI - len + 1;
		for (int i = startI; i <= endI; i++) {
			v[i][j] = min(i - startI + 1, endI - i + 1);
		}
	}
};
