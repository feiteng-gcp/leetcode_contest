class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        //unordered_map<int, unordered_set<int>> data;
        //vector<vector<int>> L=vector<vector<int>>(N, vector<int>(N, 0));
        //vector<vector<int>> R=vector<vector<int>>(N, vector<int>(N, 0));
        //vector<vector<int>> U=vector<vector<int>>(N, vector<int>(N, 0));
        //vector<vector<int>> D=vector<vector<int>>(N, vector<int>(N, 0));
        bool** data=new bool*[N];
        int** L=new int*[N];
        int** R=new int*[N];
        int** U=new int*[N];
        int** D=new int*[N];
        for(int i=0; i<N; i++){
            data[i]=new bool[N];
            for(int j=0; j<N; j++){
                data[i][j]=true;
            }
            L[i]=new int[N];
            R[i]=new int[N];
            U[i]=new int[N];
            D[i]=new int[N];
        }
        for(int i=0; i<mines.size(); i++){
            //data[mines[i][0]].insert(mines[i][1]);
            data[mines[i][0]][mines[i][1]]=false;
        }
        for(int i=0; i<N; i++){
            int curr=0;
            for(int j=0; j<N; j++){
                // 0
                //if(data.find(i)!=data.end() && data[i].find(j)!=data[i].end()){
                if(!data[i][j]){
                    curr=0;
                }else{
                    curr++;
                }
                L[i][j]=curr;
            }
        }
        for(int i=0; i<N; i++){
            int curr=0;
            for(int j=N-1; j>=0; j--){
                // 0
                //if(data.find(i)!=data.end() && data[i].find(j)!=data[i].end()){
                if(!data[i][j]){
                    curr=0;
                }else{
                    curr++;
                }
                R[i][j]=curr;
            }
        }
        for(int j=0; j<N; j++){
            int curr=0;
            for(int i=0; i<N; i++){
                // 0
                //if(data.find(i)!=data.end() && data[i].find(j)!=data[i].end()){
                if(!data[i][j]){
                    curr=0;
                }else{
                    curr++;
                }
                U[i][j]=curr;
            }
        }
        for(int j=0; j<N; j++){
            int curr=0;
            for(int i=N-1; i>=0; i--){
                // 0
                //if(data.find(i)!=data.end() && data[i].find(j)!=data[i].end()){
                if(!data[i][j]){
                    curr=0;
                }else{
                    curr++;
                }
                D[i][j]=curr;
            }
        }
        int ret=0;
        for(int i=0; i<N; i++){
            for(int j=0; j<N; j++){
                int res=min(L[i][j], min(R[i][j], min(U[i][j], D[i][j])));
                ret=max(ret, res);
            }
        }
        return ret;
    }
};
