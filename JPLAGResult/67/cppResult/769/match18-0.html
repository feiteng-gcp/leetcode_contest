class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        vector<vector<int>> mp(N, vector<int>(N, 1));
        for (auto p : mines) mp[p[0]][p[1]] = 0;
        vector<vector<int>> L(N, vector<int>(N, 0));
        auto R = L, U = L, D = L;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                if (mp[i][j]) L[i][j] = (j > 0 ? L[i][j - 1] : 0) + 1;
                if (mp[i][j]) U[i][j] = (i > 0 ? U[i - 1][j] : 0) + 1;
            }
        }
        for (int i = N - 1; i >= 0; i--) {
            for (int j = N - 1; j >= 0; j--) {
                if (mp[i][j]) R[i][j] = (j < N - 1 ? R[i][j + 1] : 0) + 1;
                if (mp[i][j]) D[i][j] = (i < N - 1 ? D[i + 1][j] : 0) + 1;
            }
        }
        int ret = 0;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                ret = max(ret, min(min(L[i][j], R[i][j]), min(U[i][j], D[i][j])));
            }
        }
                
        return ret;
    }
};
