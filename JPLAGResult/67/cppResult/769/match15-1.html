class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        vector<vector<int>> grids(N, vector<int>(N, 1));
        for(auto& m : mines)
            grids[m[0]][m[1]] = 0;
        
        int n = N;

        int ret = 0;
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(grids[i][j] == 0) continue;
                int d = min(i, j);
                d = min(d, n-i-1);
                d = min(d, n-j-1);
                int s = 1;
                for(int k=1;k<=d;k++){
                    if(grids[i-k][j] && grids[i+k][j] && grids[i][j-k] && grids[i][j+k]){
                        s++;
                    }
                    else break;
                }
                ret = max(s, ret);
            }
        }
        
        return ret;
    }
};
