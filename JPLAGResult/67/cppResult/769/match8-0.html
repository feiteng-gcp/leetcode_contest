class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        int n = N;
        if (n == 0) return 0;
        
        vector<vector<int>> board(n, vector<int>(n, 1));
        for (auto mine: mines) {
            board[mine[0]][mine[1]] = 0;    
        };
        
        vector<vector<int>> sl(n, vector<int>(n, 0));
        auto sr = sl;
        auto su = sl;
        auto sd = sl;
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (board[i][j] == 0)
                    continue;
                
                su[i][j] = 1;
                sl[i][j] = 1;
                
                if (i > 0) 
                    su[i][j] += su[i-1][j];
                if (j > 0)
                    sl[i][j] += sl[i][j-1];
            };            
        };
        
        for (int i = n-1; i >= 0; --i) {
            for (int j = n-1; j >= 0; --j) {
                if (board[i][j] == 0) 
                    continue;
                sd[i][j] = 1;
                sr[i][j] = 1;
                
                if (i < n-1)
                    sd[i][j] += sd[i+1][j];
                if (j < n-1)
                    sr[i][j] += sr[i][j+1];
            };
        };               
        
        
        int ans = 0;
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n; ++j) 
                if (board[i][j] == 1) {
                    int l = sl[i][j];
                    int r = sr[i][j];
                    int u = su[i][j];
                    int d = sd[i][j];
                    int value = min(min(l,r), min(u,d));
                    ans = max(ans, value);
                    //cout << i << " " << j << " " << value << endl;
                        
                };
        
        return ans;
        
                
                
          
        
    }
};
