class Solution {
public:
  int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
    vector<vector<bool>> m = vector<vector<bool>>(N, vector<bool>(N, true));
    vector<vector<int>> top(N, vector<int>(N, 0));
    vector<vector<int>> bottom(N, vector<int>(N, 0));
    vector<vector<int>> left(N, vector<int>(N, 0));
    vector<vector<int>> right(N, vector<int>(N, 0));
    for (auto& mine: mines) {
      m[mine[0]][mine[1]] = false;
    }
    for (int i = 0; i < N; ++i) {
      for (int j = 0; j < N; ++j) {
        if (m[i][j]) {
          left[i][j] = (j == 0 ? 0 : left[i][j - 1]) + 1;
          top[i][j] = (i == 0 ? 0 : top[i - 1][j]) + 1;
        }
      }
    }
    for (int i = N - 1; i >= 0; --i) {
      for (int j = N - 1; j >= 0; --j) {
        if (m[i][j]) {
          right[i][j] = (j == N - 1 ? 0 : right[i][j + 1]) + 1;
          bottom[i][j] = (i == N - 1 ? 0 : bottom[i + 1][j]) + 1;
        }
      }
    }
    int ans = 0;
    for (int i = 0; i < N; ++i) {
      for (int j = 0; j < N; ++j) {
        if (m[i][j]) {
          ans = max(ans, min(min(min(left[i][j], right[i][j]), top[i][j]), bottom[i][j]));
        }
      }
    }
    return ans;
  }
};
