class Solution {
public:
  int orderOfLargestPlusSign(int N, vector<vector<int>> &mines) {
    vector<vector<int>> a(N, vector<int>(N, 1));
    for (const auto &p : mines) {
      a[p[0]][p[1]] = 0;
    }
    int n = N;
    auto l = a, r = a, u = a, d = a;
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < n; ++j) {
        if (j && a[i][j]) {
          l[i][j] = max(l[i][j], l[i][j - 1] + 1);
        }
        if (i && a[i][j]) {
          u[i][j] = max(u[i][j], u[i - 1][j] + 1);
        }
      }
    }
    for (int i = n - 1; i >= 0; --i) {
      for (int j = n - 1; j >= 0; --j) {
        if (j + 1 < n && a[i][j]) {
          r[i][j] = max(r[i][j], r[i][j + 1] + 1);
        }
        if (i + 1 < n && a[i][j]) {
          d[i][j] = max(d[i][j], d[i + 1][j] + 1);
        }

      }
    }
    int ans = 0;
    for (int i = 0; i < n; ++i) {
      for (int j = 0; j < n; ++j) {
        int tmp = l[i][j];
        tmp = min(tmp, r[i][j]);
        tmp = min(tmp, u[i][j]);
        tmp = min(tmp, d[i][j]);
        ans = max(ans, tmp);
      }
    }
    return ans;
  }
};
