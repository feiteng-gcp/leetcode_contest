#include <bits/stdc++.h>
using namespace std;
#define REP(i, n) for (int i = 0; i < int(n); ++i)
#define REPE(i, a, b) for (int i = (a); i <= int(b); ++i)
#define SZ(x) ((int)(x).size())
#define ALL(x) x.begin(), x.end()
#define PB push_back
#define EB emplace_back
using LL = long long;
using PII = pair<int, int>;
#define F first
#define S second
void R(int &x) { scanf("%d", &x); }
void R(LL &x) { scanf("%lld", &x); }
void R(double &x) { scanf("%lf", &x); }
template <typename T> void R(T &t) { cin >> t; }
template <typename T> void R(vector<T> &ar) {
  for (auto &it : ar)
    R(it);
}
template <typename T, typename... Args> void R(T &t, Args &... args) {
  R(t);
  R(args...);
}

int a[555][555];
int d1[555][555];
int d2[555][555];
int d3[555][555];
int d4[555][555];
class Solution {
public:
  int orderOfLargestPlusSign(int n, vector<vector<int>> &mines) {
    memset(a, 0, sizeof a);
      memset(d1, 0, sizeof d1);
      memset(d2, 0, sizeof d2);
      memset(d3, 0, sizeof d3);
      memset(d4, 0, sizeof d4);
    REPE(i, 1, n) REPE(j, 1, n) a[i][j] = 1;
    for (auto x : mines)
      a[x[0]+1][x[1]+1] = 0;
    REPE(i, 1, n) REPE(j, 1, n) {
      d1[i][j] = a[i][j] == 0 ? 0 : 1 + d1[i - 1][j];
      d2[i][j] = a[i][j] == 0 ? 0 : 1 + d2[i][j - 1];
    }
    for (int i = n; i >= 1; --i) for (int j = n; j >= 1; --j) {
      d3[i][j] = a[i][j] == 0 ? 0 : 1 + d3[i + 1][j];
      d4[i][j] = a[i][j] == 0 ? 0 : 1 + d4[i][j + 1];
    }
    int ans = 0;
    REPE(i, 1, n) REPE(j, 1, n) {
      int x = min(d1[i][j], d2[i][j]);
      int y = min(d3[i][j], d4[i][j]);
      ans = max(ans, min(x, y));
    }
    return ans;
  }
};
