class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        vector<set<int> > row(N), col(N);
    int len = (int)mines.size();
    for(int i = 0; i < len; i++)
    {
        row[mines[i][0]].insert(mines[i][1]);
        col[mines[i][1]].insert(mines[i][0]);
    }
    for(int i = 0; i < N; i++)
    {
        row[i].insert(-1); row[i].insert(N);
        col[i].insert(-1); col[i].insert(N);
    }
    int ans = 0;
    for(int i = 0; i < N; i++)
        for(int j = 0; j < N; j++)
        {
            int L = *(prev(row[i].lower_bound(j)));
            int R = *(row[i].lower_bound(j));
            int U = *(prev(col[j].lower_bound(i)));
            int D = *(col[j].lower_bound(i));
            ans = max(ans, min(min(j - L, R - j), min(i - U, D - i)));
        }
    return ans;
    }
};
