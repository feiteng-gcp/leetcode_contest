class Solution {
public:
    int orderOfLargestPlusSign(int N, vector<vector<int>>& mines) {
        vector<vector<int>> field(N, vector<int>(N, 1));
        for(const auto& v : mines) field[v[0]][v[1]] = 0;
        
        vector<vector<int>> left(N, vector<int>(N, 0));
        auto right = left;
        auto up = left;
        auto down = left;
        for(int i = 0; i < N; i++)
        {
            for(int j = 1; j < N; j++)
            {
                left[i][j] = (field[i][j - 1] == 1 ? 1 + left[i][j - 1] : 0);
                right[i][N - j - 1] = (field[i][N - j] == 1 ? 1 + right[i][N - j] : 0);
                up[j][i] = (field[j - 1][i] == 1 ? 1 + up[j - 1][i] : 0);
                down[N - j - 1][i] = (field[N - j][i] == 1 ? 1 + down[N - j][i] : 0);
            }
        }
        int ans = 0;
        for(int i = 0; i < N; i++)
            for(int j = 0; j < N; j++)
            {
                if (field[i][j] == 1)
                {
                    ans = max(ans, 1 + min(min(up[i][j], down[i][j]), min(left[i][j], right[i][j])));
                }
            }
        return ans;
    }
};
