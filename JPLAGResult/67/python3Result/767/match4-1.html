class Solution:
    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        
def is_prime(n):
            if n == 1: return False
            for i in range(2, n):
                if n % i == 0:
                    return False
            return True

        ans = 0
        for i in range(L, R + 1):
            s = "{0:b}".format(i)
            cnt = 0
            for c in s:
                if c == '1': cnt += 1
            if is_prime(cnt):
                ans += 1
        return ans
