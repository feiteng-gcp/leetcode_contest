class Solution:
    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        prime = [i for i in range(2, 33)]

        for i in range(len(prime)):
            if prime[i] is None:
                continue
            for j in range(i+1, len(prime)):
                if prime[j] is None:
                    continue
                if str(prime[j] / prime[i])[-1] == '0':
                    prime[j] = None

        num 
= 0
        for i in range(L, R+1):
            if bin(i)[2:].count('1') in prime:
                num += 1

        return num
