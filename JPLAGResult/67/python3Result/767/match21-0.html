class Solution:
    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        def isPrime(n):
            if n < 2: return False
            if n == 2: return True
            if n % 2 == 0: return False
            maxFactor = round(n**0.5)
            for factor in range(3, maxFactor+1, 2):
                if (n % factor == 0): return False
            return True
        
        result = 0
        for num in range(L, R+1):
            binaryString = str(bin(num))
            setBitCount = 0
            for char in binaryString:
                if char == "1": setBitCount += 1
            if isPrime(setBitCount): result += 1
        return result
