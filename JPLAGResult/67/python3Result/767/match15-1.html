class Solution:
    
def isp(self,num):
        if num ==1:
            return False
        for i in range(2,num):
            if num%i==0:
                return False
        return True
    
    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        a=[]
        for i in range(L,R+1):
            tb = bin(i)[2:]
            temp = tb.count("1")
            if(self.isp(temp)):
                a.append(i)
        return len(a)
