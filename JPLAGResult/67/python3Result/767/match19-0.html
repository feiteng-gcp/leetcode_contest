import math

class Solution:
    def countPrimeSetBits(self, L, R):
        def countBits(x):
            return bin(x)[2:].count('1')
        
        def isPrime(x):
            if x==2:
                return True
            if x<2 or x%2==0:
                return False
            
            for i in range(3, int(math.sqrt(x))+1):
                if x % i == 0:
                    return False
            return True
        
        count = 0
        for i in range(L, R+1):
            if isPrime(countBits(i)):
                count+=1
        
        return count
