def is_prime(n):
    return n in [2, 3, 5, 7, 11, 13, 17, 19, 23]

def count_bit(n):
    c = 0
    while n > 0:
        n = n & (n - 1)
        c += 1
    return c


class Solution:

    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        ret 
= 0
        for n in range(L, R + 1):
            if is_prime(count_bit(n)):
                ret += 1

        return ret
