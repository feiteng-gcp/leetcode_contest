class Solution:
    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        count = 0
        for i in range(L, R+1):
            if self.isPrime(self.countSetBits(i)):
                count += 1
        return count
        
    def isPrime(self, num):
        """
        tells me if num is prime in boolean
        """
        if num == 1:
            return False
        for i in range(2, num):
            if num % i == 0:
                return False
        return True
    def  countSetBits(self, num):
        count = 0
        while (num):
            count += num & 1
            num >>= 1
        return count
