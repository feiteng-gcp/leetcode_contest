class Solution:
    prime_check = {}
    import math

        """

        return count
    
    def countBits(self,x):
        number_of_bits = 0
        while x >= 1:
            number_of_bits += x & 1
            x = x >> 1
        return number_of_bits
    
    def isPrime(self,y):
        if y in self.prime_check:
            if self.prime_check[y] == 1:
                return True
            else:
                return False
            
        if y == 1:
            return False
        if y == 2:
            return True
        
        for i in range(2,math.floor(math.sqrt(y))+1):
            if y % i == 0:
                self.prime_check[y] = 0
                return False   
        self.prime_check[y] = 1
        return True
