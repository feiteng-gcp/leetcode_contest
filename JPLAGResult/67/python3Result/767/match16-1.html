class Solution:
    
def isPrime(self, num):
        if num == 1: return False
        for i in range(2, num):
            if num % i == 0:
                return False
        return True
    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        A = {}
        total = 0
        for i in range(L,R+1):
            num =  "{0:b}".format(i).count('1')
            #print(i, "{0:b}".format(i), num )
            if num in A and A[num]:
                total += 1
            else:
                if self.isPrime(num):
                    A[num] = True
                    total += 1
                else:
                    A[num] = False
        return total
