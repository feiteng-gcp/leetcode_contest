class Solution:
    def countOfOne(self,n):
        count=0
        while n > 0:
            n = (n & (n-1)) # remove lowest set bit
            count += 1
        return count
    
    def isPrime(self, n):
        
if n <= 2:

        while div <= math.sqrt(n):
            if n % div  == 0:
                return False
            div +=1
        return True
    
    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        ans=0
        for n in range(L,R+1):
            count = self.countOfOne(n)
            #print("count of 1 in {0} is {1}".format(n, count))
            if self.isPrime(count):
                ans+=1
        return ans
