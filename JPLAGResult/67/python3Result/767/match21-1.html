import math


class Solution:
    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        def isPrime(num):
            if(num==1):
                return False
            if(num==2):
                return True
            if(num%2==0):
                return False

            i = 3
            while(i<math.sqrt(num)+1):
                if num%i==0:
                    return False
                i += 2
            return True
        
        total = 0
        for i in range(L, R+1):
            one_bits = bin(i).count("1")
            if isPrime(one_bits):
                total += 1
        return total
