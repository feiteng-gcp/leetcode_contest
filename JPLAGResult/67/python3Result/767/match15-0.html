def is_prime(a):
    if a == 1 or a == 0:
        return False
    for i in range(2, a):
        if a % i == 0:
            return False
    return True

class Solution:
    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        total_count = 0
        for i in range(L, R+1):
            binary = bin(i)
            count = binary.count('1')
            #print(i, count, is_prime(count))
            if (is_prime(count) == True):
                total_count = total_count + 1
        
        return total_count
