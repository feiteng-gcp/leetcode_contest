from collections import deque
class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        a = deque()
        b = deque()
        r = 1
        p = -1
        for i in range(len(nums)):
            while a and nums[a[-1]]<=nums[i]:
                a.pop()
            a.append(i)
            while b and nums[b[-1]]>=nums[i]:
                b.pop()
            b.append(i)
            while nums[a[0]]-nums[b[0]]>limit:
                if a[0]<b[0]:
                    p = a[0]
                    a.popleft()
                else:
                    p = b[0]
                    b.popleft()
            r = max(r,i-p)
        return r
