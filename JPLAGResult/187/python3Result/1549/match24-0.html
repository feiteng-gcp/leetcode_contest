class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        h1,h2 = [[-nums[0],0]],[[nums[0],0]]
        s,ans = 0,1
        for e in range(1,len(nums)):
            heappush(h1, [-nums[e],e])
            heappush(h2, [nums[e], e])
            while abs(h1[0][0]+h2[0][0]) > limit:
                s += 1
                while h1[0][1] < s:
                    heappop(h1)
                while h2[0][1] < s:
                    heappop(h2)
            ans = max(ans, e-s+1)
        return ans
