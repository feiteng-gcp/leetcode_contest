class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        iq, dq = collections.deque(), collections.deque()
        l, r = 0, 0
        ret = 0
        while r < len
(nums):
            while iq and nums[iq[-1]] > nums[r]:
                iq.pop()
            iq.append(r)
            while dq and nums[dq[-1]] < nums[r]:
                dq.pop()
            dq.append(r)
            r += 1
            
while dq and iq and nums[dq[0]]-nums[iq[0]] > limit:
                if dq[0] == l:
                    dq.popleft()
                if iq[0] == l:
                    iq.popleft()
                l += 1
            ret = max(ret, r-l)
        return ret
