class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        max_pq = []
        min_pq = []
        ans = 0
        cur_idx = 0
        for idx, num in enumerate(nums):
            heappush(max_pq, (-num, idx))
            heappush(min_pq, (
num, idx))
            while abs(-max_pq[0][0]-min_pq[0][0]) > limit:
                cur_idx += 1
                while max_pq[0][1] < cur_idx:
                    heappop(max_pq)
                while min_pq[0][1] < cur_idx:
                    heappop(min_pq)
            ans = max(idx-cur_idx, ans)
        return ans+1
