class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        queuemax = collections.deque()
        queuemin = collections.deque()
        def window_add(queue, num, isMax):
            if len(queue) == 0:
                queue.append(num)
            else:
                if isMax:
                    while queue and queue[-1] < num:
                        queue.pop()
                else:
                    while queue and queue[-1] > num:
                        queue.pop()
                queue.append(num)
        j = 0
        res = 0
        for i in range(len(nums)):
            window_add(queuemax, nums[i], True)
            window_add(queuemin, nums[i], False)
            while queuemax and queuemax[0]-queuemin[0] > limit:
                if queuemax[0] == nums[j]:
                    queuemax.popleft()
                if queuemin[0] == nums[j]:
                    queuemin.popleft()
                j += 1
            res = max(res, i-j+1)
        return res
