class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        
        left = 0
        right = 0
        
        n = len(nums)
        mx = []
        mn = []
        res = 1
        while right < n:
            
            while mx and mx[-1] < nums[right]:
                mx.pop()
            mx += [nums[right]]
            
            while mn and mn[-1] > nums[right]:
                mn.pop()
                
            mn += [nums[right]]
            
            
            while abs(mn
[0] - mx[0]) > limit:
                if nums[left] == mx[0]:
                    mx.pop(0)
                
                if nums[left] == mn[0]:
                    mn.pop(0)
                
                left+= 1
            
            
            res = max(res, right - left + 1)
            right += 1
        
        
        return res
