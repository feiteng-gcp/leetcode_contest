import sortedcontainers

class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        s = sortedcontainers.SortedList()
        ans = 0
        last = 0
        for i in range(len(nums)):
            s.add(nums[i])
            while s[-1] - s[0] > limit:
                s.remove(nums[last])
                last += 1
            ans = max(ans, len(s))
        return ans
