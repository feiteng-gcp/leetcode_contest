class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        def can_find(k):
            
            deque1 = collections.deque()
            deque2 = collections.deque()
            res = []

            for i in range(k):
                while deque1 and nums[deque1[-1]] <= nums[i]:
                    deque1.pop()
                deque1.append(i)
            res.append(nums[deque1[0]])
            
            for i in range(k):
                while deque2 and nums[deque2[-1]] >= nums[i]:
                    deque2.pop()
                deque2.append(i)
                res.append(nums[deque2[0]])
            
            if nums[deque1[0]] - nums[deque2[0]] <= limit:
                return True
            
            for i in range(k, len(nums)):
                if deque1[0] <= i - k:
                    deque1.popleft()
                if deque2[0] <= i - k:
                    deque2.popleft()
                while deque1 and nums[deque1[-1]] <= nums[i]:
                    deque1.pop()
                while deque2 and nums[deque2[-1]] >= nums[i]:
                    deque2.pop()
                deque1.append(i)
                deque2.append(i)
                
                if nums[deque1[0]] - nums[deque2[0]] <= limit:
                    return True
                
            return False
        
        low, high = 0, len(nums)
        while low < high:
            mid = (low + high + 1) // 2
            if can_find(mid):
                low = mid
            else:
                high = mid - 1
        
        return low
