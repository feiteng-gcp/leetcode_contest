class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        if len(nums) < 2:
            return len(nums)
        minstack, maxstack = [0], [0]
        pre = 0
        res = 0
        for i in range(1, len(nums)):
            while minstack and nums[minstack[-1]] > nums[i]:
                minstack.pop()
            minstack.append(i)
            
            while maxstack and nums[maxstack[-1]] < nums[i]:
                maxstack.pop()
            maxstack.append(i)
            
            while maxstack and minstack and nums[maxstack[0]] - nums[minstack[0]] > limit:
                if pre == minstack[0]:
                    minstack.pop(0)
                if pre == maxstack[0]:
                    maxstack.pop(0)
            
                pre += 1
            res = max(res, i-pre+1)
        return res
