class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        mxq = collections.deque()
        miq = collections.deque()
        n = len(nums)
        ans = 0
        j = 0
        ans = 0
        # mxq (v, i)
        for i in range(n):
            while j < n and (len(mxq) == 0 or (max(mxq[0][0], nums[j]) - min(miq[0][0], nums[j]) <= limit)):
                # print(i, ": ")
                # print(j, "max:", max(mxq[0][0], nums[j]) if mxq else nums[j])
                # print(j, "min:", min(miq[0][0], nums[j]) if miq else nums[j])
                nj = nums[j]
                while mxq and nj > mxq[-1][0]:
                    mxq.pop()
                mxq.append((nj, j))
                
                while miq and nj < miq[-1][0]:
                    miq.pop()
                miq.append((nj, j))
                
                j += 1
            
            ans = max(ans, j - i)
            # print(i, j)
            
if mxq[0][1] == i:
                mxq.popleft()
            if miq[0][1] == i:
                miq.popleft()
        
        # print("------")
        return ans
