class Solution:
    def destCity(self, paths: List[List[str]]) -> str:
        s = set()
        aa = set()
        for a, b in paths:
            aa.add(a)
            s.add(a)
            s.add(b)
            
        for ss in s:
            if ss not in aa:
                return ss
        
        return ""
