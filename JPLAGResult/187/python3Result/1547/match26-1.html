class Solution:
    def destCity(self, paths: List[List[str]]) -> str:
        if not paths:
            return ""
        
        cur = paths
[0][0]
        start = set()
        end = set()
        for s, e in paths:
            start.add(s)
            end.add(e)
        
        for e in end:
            if e not in start:
                return e
        return -1
