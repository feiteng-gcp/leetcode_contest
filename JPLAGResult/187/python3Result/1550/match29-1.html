class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        R, C = len(mat), len(mat[0])
        pq = [(sum(row[0] for row in mat), [0] * R)]
        cnt = 0
        seen = set()
        while pq:
            # print(pq)
            v, cols = heapq.heappop
(pq)
            t = tuple(cols)
            if t in seen:
                continue
            seen.add(t)
            cnt += 1
            if cnt == k:
                # print()
                return v
            for r in range(R):
                # can increment here
                if cols[r] + 1 < C:
                    cols[r] += 1
                    nv = sum(mat[i][cols[i]] for i in range(R))
                    heapq.heappush(pq, (nv, cols[:]))
                    cols[r] -= 1
        return -1
