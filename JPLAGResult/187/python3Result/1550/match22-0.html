from heapq import *
from collections import deque
class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        h,w=len(mat),len(mat[0])
        seen=set()
        
        hp=[(sum(row[0] for row in mat), [0] * h)]
        seen.add(tuple(hp[0][1]))
        for i in range(k-1):
            #print("i:%d,hp:%s"%(i,hp))
            s,ixs=hp[0]
            
            heappop(hp)
            for y, x in enumerate(ixs):
                if x+1<w:
                    nixs=ixs[:]
                    nixs[y]=x+1
                    if tuple(nixs) in seen:
                        continue
                    seen.add(tuple(nixs))
                    heappush(hp,(s+mat[y][x+1]-mat[y][x], nixs))
        return hp[0][0]
