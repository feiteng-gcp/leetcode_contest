class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        q = [(sum([x[0] for x in mat]), [0 for i in range(len(mat))])]
        
        touched = {0: True}
        
        def convert_to_code(choice, m, n):
            cnt = 0
            exp = 1
            for i in range(len(choice)):
                cnt += exp * choice[i]
                exp *= n
            return cnt
        
        m = len(mat)
        n = len(mat[0])
        
        while k > 0:
            k -= 1;
            if k == 0:
                return q[0][0]
            
            for i in range(m):
                if q[0]
[1][i] < n - 1:
                    sum_next = q[0][0] + mat[i][q[0][1][i] + 1] - mat[i][q[0][1][i]]
                    seq_next = [i for i in q[0][1]]
                    seq_next[i] += 1
                    code = convert_to_code(seq_next, m, n)
                    if not (code in touched):
                        q.append((sum_next, seq_next))
                        touched[code] = True
            del q[0]
            q = sorted(q, key = lambda x: x[0])
            if len(q) > k:
                q = q[:k]
