class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        indices = [ (sum([mat[i][0] for i in range(m)]), tuple([0] * m)) ]
        res = None
        visited = set()
        while k > 0:
            total, cur = heapq.heappop(indices)
            if cur in visited: continue
            visited.add(cur)
            k -= 1
            res = total
            for i in range(m):
                if cur[i] + 1 >= n: continue
                key = cur[:i] + (cur[i]+1,) + cur[i+1:]
                t = total - mat[i][cur[i]] + mat[i][key[i]]
                heapq.heappush(indices, (t, key))
        return res
