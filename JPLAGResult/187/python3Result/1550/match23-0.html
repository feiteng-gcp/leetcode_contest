class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        heap = [(sum(row[0] for row in mat), (0,) * m)]
        seen = set()
        for _ in range(k - 1):
            val, idx = heapq.heappop(heap)
            # print(val, idx)
            idx = list(idx)
            for i, x in enumerate(idx):
                if x < n - 1:
                    idx[i] += 1
                    tp = tuple(idx)
                    if tp not in seen:
                        heapq.heappush(heap, (sum(mat[i][x] for i, x in enumerate(idx)), tp))
                        seen.add(tp)
                    idx[i] -= 1
        return heap[0][0]
