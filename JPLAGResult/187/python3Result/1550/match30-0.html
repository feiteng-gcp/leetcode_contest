class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        pq = [(sum(mat[i][0] for i in range(m)), tuple([0]*m))]
        seen = set(tuple([0]*m))
        while k > 0:
            v, idx = heapq.heappop(pq)
            for i, j in enumerate(idx):
                if j+1 < n:
                    new_idx = list(idx)
                    new_idx[i] += 1
                    if tuple(new_idx) not in seen:
                        seen.add(tuple(new_idx))
                        heapq.heappush(pq, (v-mat[i][j]+mat[i][j+1], tuple(new_idx)))
            for i, j in enumerate(idx):
                d = 2
                while j+d < n and v-mat[i][j]+mat[i][j+d] <= pq[0][0]:
                    new_idx = list(idx)
                    new_idx[i] += d
                    if tuple(new_idx) not in seen:
                        seen.add(tuple(new_idx))
                        heapq.heappush(pq, (v-mat[i][j]+mat[i][j+d], tuple(new_idx)))
                    d += 1
            k -= 1
        return v
