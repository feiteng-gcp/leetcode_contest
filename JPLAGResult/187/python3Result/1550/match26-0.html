class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        hp = [(sum(row[0] for row in mat), (0,) * m)]
        reach = {(0,) * m}
        for _ in range(k - 1):
            bs = list(heapq.heappop(hp)[1])
            for i in range(m):
                if bs[i] + 1 >= n: continue
                bs[i] += 1
                bs2 = tuple(bs)
                if bs2 not in reach:
                    heapq.heappush(hp, (sum(mat[i][bs2[i]] for i in range(m)), bs2))
                    reach.add(bs2)
                bs[i] -= 1
        return hp[0][0]
