class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        n, m = len(mat), len(mat[0])
        h = [(sum(i[0] for i in mat), [0,] * n)]
        seen = set([(0,) * n])
        for i in range(k - 1):
            v, p = heapq.heappop(h)
            for j in range(n):
                nlp = p[:]
                nlp[j] += 1
                if nlp[j] >= m:
                    continue
                tnlp = tuple(nlp)
                if tnlp not in seen:
                    seen.add(tnlp)
                    heapq.heappush(h, (v + mat[j][nlp[j]] - mat[j][nlp[j] - 1], nlp))                    
        return heapq.heappop(h)[0]
