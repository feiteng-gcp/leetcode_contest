class Solution:
	def kthSmallest(self, mat: List[List[int]], k: int) -> int:
		from heapq import heappush, heappop
		nr, nc = len(mat), len(mat[0])

		cands = []
		seen = set([tuple([0]*nr)])
		heappush(cands,(sum(row[0] for row in mat),[0]*nr))
		
		for _ in range(k-1):
			s, idx = heappop(cands)
			for r,c in enumerate(idx):
				if c < nc-1:
					idx2 = idx[:r] + [c+1] + idx[r+1:]
					
if tuple(idx2) in seen: continue
					seen.add(tuple(idx2))
					heappush(cands, (s+mat[r][c+1]-mat[r][c], idx2))
		s, _ = heappop(cands)
		return s
