import heapq

class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        used = set()
        if n == 1:
            return sum(row[0] for row in mat) 

        col = [0] * m
        h = []
        
        tot = sum(row[0] for row in mat)
        for i in range(m):
            col[i] += 1
            t = tuple(col)
            if t not in used:
                used.add(t)
                heapq.heappush(h, (tot-mat[i][col[i]-1]+mat[i][col[i]], t))
            col[i] -= 1
            
        print(h)

        k -= 1
        while k:
            k -= 1
            res, col = heapq.heappop(h)
            # print(res, col)
            col = list(col)
            for i in range(m):
                if col[i] + 1 < n:
                    col[i] += 1
                    t = tuple(col)
                    if t not in used:
                        used.add(t)
                        heapq.heappush(h, (res-mat[i][col[i]-1]+mat[i][col[i]], t))
                    col[i] -= 1

        return res
