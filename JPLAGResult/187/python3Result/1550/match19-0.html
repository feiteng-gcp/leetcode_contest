class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        pre_list = [0]
        m = len(mat)
        n = len(mat[0])
        for i in range(m):
            cur_list = list()
            for j in range(n):
                for num in pre_list:
                    cur_list.append(num + mat[i][j])
            cur_list.sort()
            pre_list = cur_list[:k]
        return pre_list[k - 1]
