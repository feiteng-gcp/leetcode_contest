class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        
        m = len(mat)
        n = len(mat[0])
        
        pq = [-num for num in mat[0]]
        
        heapq.heapify(pq)
        while len(pq) > k:
            heapq.heappop(pq)
        
        for i in range(1, m):
            val = [num for num in pq]
            npq = []
            for j in range(len(val)):
                for q in range(n):
                    heapq.heappush(npq, val[j] - mat[i][q])
            while len(npq) > k:
                heapq.heappop(npq)
            pq = npq
            
        return -pq[0]
