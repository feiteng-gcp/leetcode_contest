import queue
class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        
        pq = queue.PriorityQueue()
        pq.put([sum([t[0] for t in mat]), tuple([0]*len(mat))])
        visited = set()
        visited.add(tuple([0]*len(mat)))
        
        for i in range(k-1) :
            _, p = pq.get()
            # print(_, p, [mat[i][t] for i, t in enumerate(p)])
            for i in range(m) :
                if p[i] == n-1 :
                    continue
                n_tuple = tuple([p[j] if not j == i else p[j]+1 for j in range(m)])
                if n_tuple in visited :
                    continue
                visited.add(n_tuple)
                pq.put([sum([mat[i][t] for i, t in enumerate(n_tuple)]), n_tuple])
        to_ret, _ = pq.get()
        return to_ret
