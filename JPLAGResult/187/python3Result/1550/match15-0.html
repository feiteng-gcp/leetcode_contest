class Solution:
    def kthSmallest(self, mat, k):
        m, n = len(mat), len(mat[0])
        seen = {(0,) * m}
        init = sum(row[0] for row in mat)
        q = [[init, [0] * m]]

        for _ in range(k - 1):
            val, state = heapq.heappop(q)
            for i in range(m):
                if state[i] < n - 1:
                    state[i] += 1
                    t = tuple(state)
                    if t not in seen:
                        seen.add(t)
                        cur = val - mat[i][state[i] - 1] + mat[i][state[i]]
                        heapq.heappush(q, [cur, state[::]])
                    state[i] -= 1
        return q[0][0]
