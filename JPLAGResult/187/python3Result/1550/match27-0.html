class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        row, col = len(mat), len(mat[0])
        queue = []
        curr, val = [], 0
        for i in range(row):
            val += mat[i][0]
            curr.append([i, 0])
        queue.append([val, curr])
        visit = set()
        while k > 1:
            k -= 1
            val, arr = heapq.heappop(queue)
            for i in range(len(arr)):
                if arr[i][1] < col -1:
                    v = val - mat[arr[i][0]][arr[i][1]] + mat[arr[i][0]][arr[i][1] + 1]
                    a = [aaa[:] for aaa in arr]
                    a[i][1] += 1
                    s = ""
                    for x,y in a:
                        s += str(x) + "," + str(y) + ","
                    if s not in visit:
                        visit.add(s)
                        heapq.heappush(queue, [v, a])
        v, a = heapq.heappop(queue)
        return v
