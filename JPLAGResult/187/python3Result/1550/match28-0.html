class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        visited = set()
        heap = [(sum(mat[i][0] for i in range(len(mat))), tuple([0] * len(mat)))]
        res = 0
        for i in range(k):
            res, code = heapq.heappop(heap)
            for i in range(len(code)):
                if code[i] + 1 < len(mat[i]):
                    nxt = code[:i] + tuple([code[i] + 1]) + code[i+1:]
                    if nxt not in visited:
                        visited.add(nxt)
                        heapq.heappush(heap, (res + mat[i][code[i]+1] - mat[i][code[i]], nxt))
        return res
