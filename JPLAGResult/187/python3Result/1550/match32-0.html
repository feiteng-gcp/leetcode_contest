class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        N = len(mat)
        Cols = len(mat[0])
        q = [(sum(x[0] for x in mat), [0] * N)]
        seen = set()
        topScore = 0
        
        def h(l):
            return " ".join(str(x) for x in l)
        
        while True:
            if k == 0:
                return score
            score, top = q[0]

            heapq.heappop(q)
            topScore = max(score, topScore)
            
            for x in range(N):
                index = top[x]

                if index + 1 < Cols:
                    nextTotal = score + mat[x][index+1] - mat[x][index]
                    
                    if len(q) <= 40000 or topScore > nextTotal:
                        nx = top[:]
                        nx[x] = index + 1
                        ha = h(nx)
                        if ha not in seen:
                            seen.add(ha)
                            heapq.heappush(q,(nextTotal, nx))

            #print(k, score)
            k -= 1
