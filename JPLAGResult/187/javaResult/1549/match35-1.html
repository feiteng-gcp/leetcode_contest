class Solution {
	private boolean check(TreeMap<Integer , Integer> map , int limit) {
		int first = map.firstKey() , last = map.lastKey();
		return last - first <= limit;
	}
	
    
public int longestSubarray(int[] nums, int limit) {
    	TreeMap<Integer , Integer> map = new TreeMap<>();
    	int i , j = 0 , n = nums.length , ans = 0;
    	for (i = 0;i < n;i ++) {
    		map.put(nums[i] , map.getOrDefault(nums[i] , 0) + 1);
    		while (!check(map , limit)) {
    			int value = nums[j];
    			map.put(value , map.get(value) - 1);
    			if (map.get(value).equals(0)) {
    				map.remove(value);
    			}

    			j ++;
    		}
    		ans = Math.max(ans , i - j + 1);
    	}
        return ans;
    }
}
