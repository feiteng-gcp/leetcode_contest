class Solution {
    public int longestSubarray(int[] nums, int limit) {
        // store index
        Deque<Integer> min = new ArrayDeque<>(); // monotonic inc
        Deque<Integer> max = new ArrayDeque<>(); // monotonic dec
        int diff = 0;
        int lastIdx = -1;
        
for (int i = 0; i < nums.length; i++) {
            // push min
            while (!min.isEmpty() && nums[min.peekLast()] >= nums[i]) {
                min.pollLast();
            }
            min.offerLast(i);
            // push max
            while (!max.isEmpty() && nums[max.peekLast()] <= nums[i]) {
                max.pollLast();
            }
            max.offerLast(i);
            // check diff
            
            while (nums[
max.peekFirst()] - nums[min.peekFirst()] > limit) {
                if (max.peekFirst() < min.peekFirst()) {
                    lastIdx = max.pollFirst();
                } else {
                    lastIdx = min.pollFirst();
                }
            }
            
            diff = Math.max(diff, i - lastIdx);
            
            // System.out.print("min: ");
            // System.out.println(min);
            // System.out.print("max: ");
            // System.out.println(max);
            // System.out.println("diff: " + diff);
        }
        
        return diff;
    }
}
