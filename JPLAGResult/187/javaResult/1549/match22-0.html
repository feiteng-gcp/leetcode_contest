class Solution {
    public int longestSubarray(int[] nums, int limit) {
    	TreeMap<Integer, Integer> map = new TreeMap<>();
    	int s = 0;
    	int e = 0;
    	int ans = 0;
    	int curlen = 0;
    	for(; e < nums.length; e++) {
    		map.put(nums[e], map.getOrDefault(nums[e], 0) + 1);
    		curlen++;
    		//System.out.println("0:" + s + "~" + e + ":" + map);
    		while(map.lastKey() - map.firstKey() > limit) {
    			int cnt = map.get(nums[s]);
    			cnt--;
    			if(cnt == 0) map.remove(nums[s]);
    			else map.put(nums[s], cnt);
    			s++;
    			curlen--;
    			//System.out.println("IN:" + s + "~" + e + ":" + map);
    		}
    		//System.out.println(s + "~" + e + ":" + map);
    		ans = Math.max(ans, curlen);
    	}
    	return ans;
    }
}
