class Solution {
    public int longestSubarray(int[] nums, int limit) {
        int n = nums.length, slow = 0;
        TreeMap<Integer, Integer> freq = new TreeMap();
        int res = 0;
        for(int f = 0; f < n; f++) {
            int v = nums[f];
            freq.put(v, freq.getOrDefault(v, 0) + 1);
            while(freq.lastKey() - freq.firstKey() > limit) {
                int count = freq.get(nums[slow]);
                if (count == 1) {
                    freq.remove(nums[slow]);
                } else {
                    freq.put(nums[slow], count - 1);
                }
                slow++;
            }
            res = Math.max(res, f - slow + 1);
        }
        return res;
    }
}
