class Solution {
    public int longestSubarray(int[] nums, int limit) {
        int l = 0, ans = 1;
        Deque<Integer> subMax = new ArrayDeque<>(), subMin = new ArrayDeque<>();
        subMax.addFirst(0);
        subMin.addFirst(0);
        for (int i = 1; i < nums.length; ++i) {
            while (!subMax.isEmpty() && nums[i] > nums[subMax.peekLast()]) {
                subMax.pollLast();
            }
            subMax.offerLast(i);
            while (!subMin.isEmpty() && nums[i] < nums[subMin.peekLast()]) {
                subMin.pollLast();
            }
            subMin.offerLast(i);
            if (nums[subMax.peekFirst()] - nums[subMin.peekFirst()] <= limit) {
                ans = Math.max(i - l + 1, ans);
            } else {
                while (!subMax.isEmpty() && !subMin.isEmpty() && nums[subMax.peekFirst()] - nums[subMin.peekFirst()] > limit) {
                    if (subMax.peekFirst() < subMin.peekFirst()) {
                        l = 1 + subMax.pollFirst();
                    } else {
                        l = 1 + subMin.pollFirst();
                    }
                }
            }
        }
        return ans;
    }
}
