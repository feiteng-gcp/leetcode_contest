class Solution {
    public int longestSubarray(int[] nums, int limit) {
        int low = 0;
        TreeMap<Integer, Integer> window = new TreeMap<>();
        
int ans = 0;
        for (int high = 0; high < nums.length; high++)
        {
            window.put(nums[high], window.getOrDefault(nums[high], 0) + 1);
            while(window.lastKey() - window.firstKey() > limit)
            {
                window.put(nums[low], window.get(nums[low]) - 1);
                if (window.get(nums[low]) == 0)
                {
                    window.remove(nums[low]);
                
}
                
                low++;
            }
            
            ans = Math.max(high - low + 1, ans);
        }
        
        return ans;
    }
}
