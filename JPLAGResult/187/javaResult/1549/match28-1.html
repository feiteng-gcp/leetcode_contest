class Solution {
    public int longestSubarray(int[] nums, int limit) {
        int ret = 0;
        TreeMap<Integer, Integer> bst = new TreeMap<>();
        
        int l = 0;
        for (int r = 0; r < nums.length; r++) {
            bst.compute(nums[r], (k, v) -> v == null ? 1 : v + 1);
            
            
while(l <= r && bst.lastKey() - bst.firstKey() > limit) {
                int freq = bst.get(nums[l]);
                if (freq == 1) {
                    bst.remove(nums[l]);
                } else {
                    bst.put(nums[l], freq - 1);
                }
                l++;
            }
            ret = Math.max(ret, r - l + 1);
        }
        return ret;
    }
}
