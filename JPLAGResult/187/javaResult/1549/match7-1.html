class Solution {
    public int longestSubarray(int[] nums, int limit) {
        Deque<Integer> min = new ArrayDeque<>();
        Deque<Integer> max = new ArrayDeque<>();
        int ans = 0;
        int last = 0;
        for (int i = 0; i < nums.length; i++) {
            while (!max.isEmpty() && nums[i] >= nums[max.getLast()]) max.removeLast();
            max.addLast(i);
            
            while (!min.isEmpty() && nums[i] <= nums[min.getLast()]) min.removeLast();
            min.addLast(i);
            
            while (nums[max.getFirst()] - nums[min.getFirst()] > limit) {
                last++;
                
if (max.getFirst() < last) max.removeFirst();
                if (min.getFirst() < last) min.removeFirst();
            }
            
            ans = Math.max(ans, i - last + 1);
        }
        return ans;
    }
}
