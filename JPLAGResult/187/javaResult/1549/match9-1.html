class Solution {
    public int longestSubarray(int[] nums, int limit) {
        int res = 0;
        
        PriorityQueue<Integer> max = new PriorityQueue<>( (a,
b) -> b - a);
        PriorityQueue<Integer> min = new PriorityQueue<>();
        
        int end = 0;
        for (int i = 0; i < nums.length; i++) {
            min.add(nums[i]);
            max.add(nums[i]);
            while (max.peek() - min.peek() > limit) {
                max.remove(nums[end]);
                min.remove(nums[end]);
                end++;
            
}
            res = Math.max(res, i - end + 1);
        }
        
        return res;
    }
}
