class Solution {
    public int kthSmallest(int[][] mat, int k) {
        return divide(0, mat.length - 1, mat, k)[k - 1];
    }
    public int[] divide(int start, int end, int[][] mat, int k) {
        int[] res;
        if (start == end) {
            return mat[start];
        } else if (start + 1 == end) {
            res = merge2(mat[start], mat[end], k);
        } else {
            int mid = (start + end) / 2;
            int[] arr1 = divide(start, mid, mat, k);
            int[] arr2 = divide(mid + 1, end, mat, k);
            res = merge2(arr1, arr2, k);
        }
        return res;
    
}
    public int[] merge2(int[] arr1, int[] arr2, int k) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[0]+a[1]-b[0]-b[1]);
        int len = Math.min(arr1.length * arr2.length, k);
        int[] res = new int[len];
        
for (int i = 0; i < arr1.length && i < k; i++) {
            pq.offer(new int[]{arr1[i], arr2[0], 0});
        }
        for (int i = 0; i < len; i++) {
            int[] cur = pq.poll();
            res[i] = cur[0] + cur[1];
            if (cur[2] == arr2.length - 1)
                continue;
            pq.offer(new int[]{cur[0], arr2[cur[2] + 1], cur[2] + 1});
        }
        return res;
    }
}
