class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length, n = mat[0].length;
        
        HashSet<Node> visited = new HashSet<>();
        PriorityQueue<Node> minHeap = new PriorityQueue<>();
        minHeap.offer(new Node(new int[m], mat));
        
        Node minNode = null;
        while(k > 0) {
            minNode = minHeap.poll();
            
            for (int i = 0; i < m; i++) {
                if (minNode.indexes[i] < n - 1) {
                    minNode.indexes[i]++;
                    Node newNode = new Node(minNode.indexes, mat);
                    if (visited.contains(newNode) == false) {
                        minHeap.offer(newNode);
                        visited.add(newNode);
                    }
                    minNode.indexes[i]--;
                }
            }
            
            k--;
        }
        return minNode.sum;
    }
    
    private class Node implements Comparable<Node>{
        int[] indexes;
        int sum = 0;
        
        Node(int[] indexes, int[][] mat) {
            this.indexes = indexes.clone();
            for (int i = 0; i < mat.length; i++) {
                sum += mat[i][indexes[i]];
            }
        }
        @Override
        public int compareTo(Node other) {
            return this.sum - other.sum;
        }
        @Override
        public boolean equals(Object other) {
            return Arrays.equals(this.indexes, ((Node)other).indexes);
        }
        @Override
        public int hashCode() {
            return Arrays.hashCode(this.indexes);
        }
    }
}
