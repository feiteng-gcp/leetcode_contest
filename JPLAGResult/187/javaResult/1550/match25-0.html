class Solution {
    public int kthSmallest(int[][] mat, int k) {
        return aux(mat, k, 0, mat.length-1)[k-1];
    }
    
    private int[] aux(int[][] mat, int k, int start, int end) {
        if (start == end) return mat[start];
        int mid = (start+end)/2;
        int[] top = aux(mat, k, start, mid);
        int[] bottom = aux(mat, k, mid+1, end);
        return kSmallestPairs(top, bottom, k);
    }
    
    private int[] kSmallestPairs(int[] nums1, int[] nums2, int k) {
        // if (nums1.length == 0 || nums2.length == 0) return new ArrayList();
        
        List<List<Integer>> res = new ArrayList();
        Queue<int[]> queue = new PriorityQueue<int[]>(k, (a,b) -> a[0]+a[1]-(b[0]+b[1]));
        
        for (int i = 0; i < nums1.length && i < k; i++) {
            queue.offer(new int[]{nums1[i], nums2[0], 0});
        }
        
        while (res.size() < k && !queue.isEmpty()) {
            int[] smallest = queue.poll();
            res.add(new ArrayList(Arrays.asList(smallest[0], smallest[1])));
            if (smallest[2] == nums2.length-1) continue;
            queue.offer(new int[]{ smallest[0], nums2[smallest[2]+1], smallest[2]+1 });
        }
        
        int[] ans = new int[res.size()];
        for (int i = 0; i < ans.length; i++) {
            ans[i] = res.get(i).get(0) + res.get(i).get(1);
        }
        
        return ans;
    }
}
