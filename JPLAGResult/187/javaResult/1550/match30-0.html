class Solution {
    public int kthSmallest(int[][] mat, int k) {
        PriorityQueue<int[][]> minHeap = new PriorityQueue<>((a, b) -> {
            // a is int[][], b is int[][]
            int sumA = getSum(a, mat);
            int sumB = getSum(b, mat);
            return sumA - sumB;
        });
        int r = mat.length;
        int c = mat[0].length;
        int[][] init = new int[r][2];
        for (int i = 0; i < r; i++) {
            init[i][0] = i;
            init[i][1] = 0;
        }
        HashSet<List<List<Integer>>> set = new HashSet();
        minHeap.offer(init);
        set.add(convertToList(init));
        for (int i = 0; i < k - 1; i++) {
            int[][] cur = minHeap.poll();
            // printHelper(cur, mat);
            // System.out.println(getSum(cur, mat));
            
            for (int j = 0; j < r; j++) {
                int[][] copy = copyArray(cur);
                copy[j][1] += 1;
                if (copy[j][1] < c && !set.contains(convertToList(copy))) {
                    set.add(convertToList(copy));
                    minHeap.offer(copy);
                }
            }
        }
        return getSum(minHeap.peek(), mat);
    }
    
    private List<List<Integer>> convertToList(int[][] copy) {
        List<List<Integer>> res = new ArrayList();
        for (int[] c : copy) {
            List<Integer> tmp = new ArrayList();
            tmp.add(c[0]);
            tmp.add(c[1]);
            res.add(tmp);    
        }
        return res;
    }
    
    private void printHelper(int[][] pos, int[][] mat) {
        for (int[] p : pos) {
            System.out.println(p[0] + "," + p[1] + ": " + mat[p[0]][p[1]]);
        }
    }
    
    private int getSum(int[][] pos, int[][] mat) {
        int res = 0;
        for (int[] p : pos) {
            res += mat[p[0]][p[1]];
        }
        return res;
    }
    
    private int[][] copyArray(int[][] arr) {
        int[][] res = new int[arr.length][arr[0].length];
        for (int i = 0; i < arr.length; i++) {
            for (int j = 0; j < arr[0].length; j++) {
                res[i][j] = arr[i][j];
            }
        }
        return res;
    }
}
