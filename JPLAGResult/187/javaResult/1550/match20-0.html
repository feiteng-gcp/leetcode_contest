class Solution {
    
    
    static class Vector implements Comparable<Vector> {
        int[] v;
        int sum;
        Vector(int[] a, int[][] mat) {
            v = a;
            
            sum = 0;
            for (int i = 0; i < v.length; i++) {
                sum += mat[i][v[i]];
            }
        }
        @Override
        public int compareTo(Vector that) {
            return sum - that.sum;
        }
    }
    
    
    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length, n = mat[0].length;
        PriorityQueue<Vector> pq = new PriorityQueue<>();
        Set<String> seen = new HashSet<>();
        
        int[] start = new int[m];
        pq.offer(new Vector(start, mat));
        
        while (!pq.isEmpty()) {
            Vector cur = pq.poll();
            k--;
            if (k == 0) return cur.sum;
            
            for (int i = 0; i < m; i++) {
                if (cur.v[i] == n - 1) continue;
                int[] next = Arrays.copyOf(cur.v, m);
                next[i]++;
                
                if (seen.contains(Arrays.toString(next)))
                    continue;
                
                pq.offer(new Vector(next, mat));
                seen.add(Arrays.toString(next));
            }
        }
        
        return -1;
    }
}
