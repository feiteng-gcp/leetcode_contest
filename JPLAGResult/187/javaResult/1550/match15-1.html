class Arr {
    public static int len = 0;
    public int[] arr;
    public Arr() {
        arr = new int[len];
    }
    public Arr(Arr a) {
        arr = Arrays.copyOf(a.arr, len);
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Arr arr1 = (Arr) o;
        return Arrays.equals(arr, arr1.arr);
    }

    @Override
    public int hashCode() {
        return Arrays.hashCode(arr);
    }
}

class Solution {
    public int kthSmallest(int[][] mat, int k) {
        PriorityQueue<Arr> heap = new PriorityQueue<>(Comparator.comparingInt(a -> arraySum(a.arr, mat)));
        Set<Arr> set = new HashSet<>();
        int n = mat.length, m = mat[0].length;
        Arr.len = n;
        Arr start = new Arr();
        heap.add(start);
        set.add(start);
        for (int i = 1; i < k; i++) 
{
            Arr a = heap.poll();
            // System.out.println(arraySum(a.arr, mat));
            for (int j = 0; j < n; j++) {
                if (a.arr[j] + 1 == m) continue;
                Arr newArr = new Arr(a);
                newArr.arr[j]++;
                
if (!set.contains(newArr)) {
                    heap.add(newArr);
                    set.add(newArr);
                }
            }
        }
        return arraySum(heap.peek().arr, mat);
    }

    public static int arraySum(int[] arr, int[][] mat) {
        int res = 0;
        
for (int i = 0; i < mat.length; i++) {
            res += mat[i][arr[i]];
        }
        return res;
    }
}
