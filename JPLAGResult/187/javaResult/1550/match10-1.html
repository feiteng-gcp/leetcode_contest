class Solution {
    public int kthSmallest(
int[][] mat, int k) {
        if (mat == null || mat.length == 0) {
            return -1;
        }
        int len = mat.length, sum = 0;
        for (int i = 0; i < len; ++i) {
            sum += mat[i][0];
        }
        PriorityQueue<Integer> queue = new PriorityQueue<>((a, b) -> (b.compareTo(a)));
        queue.add(0);
        for (int[] m : mat) {
            PriorityQueue<Integer> next = new PriorityQueue<>((a, b) -> (b.compareTo(a)));
            for (int i : queue) {
                next.add(i);
            }
            for (int i = 1; i < m.length; ++i) {
                
for (int j : queue) {
                    next.add(j + m[i] - m[0]);
                    if (next.size() > k) {
                        next.poll();
                    }
                }
            }
            queue = next;
        }
        return sum + queue.poll();
    }
}
