class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int row = mat.length;
        int col = mat[0].length;
        Map<String, Integer> map = new HashMap<>();
        PriorityQueue<String> queue = new PriorityQueue<>((a, b)->{
            int p1 = map.get(a);
            int p2 = map.get(b);
            return p1 - p2;
        });
        
        int[] arr = new int[row];
        String key = createKey(arr);
        int sum = 0;
        for (int i = 0; i < row; i++) {
            int c = arr[i];
            sum += mat[i][c];
        }
        map.put(key, sum);
        queue.offer(key);
        while (!queue.isEmpty()) {
            String temp = queue.poll();
            k--;
            if (k == 0) return map.get(temp);
            int[] ar = decode(temp);
            for (int i = 0; i < row; i++) {
                if (ar[i] == col - 1) continue;
                int org = ar[i];
                ar[i]++;
                String tk = createKey(ar);
                if (!map.containsKey(tk)) {
                    int val = comput(ar, mat);
                    map.put(tk, val);
                    queue.offer(tk);
                }
                ar[i] = org;
            
}
        }
        return -1;
    }
    
    public String createKey(int[] arr) {
        StringBuilder sb = new StringBuilder();
        for (int x : arr) sb.append(x).append("+");
        return sb.substring(0, 
sb.length()-1).toString();
    }
    
    public int comput(int[] arr, int[][] mat) {
        int sum = 0;
        for (int i = 0; i < arr.length; i++) {
            int c = arr[i];
            sum += mat[i][c];
        }
        return sum;
    }
    
    public int[] decode(String key) {
        String[] parts = key.split("\\+");
        int[] res = new int[parts.length];
        for (int i = 0; i < parts.length; i++) {
            int val = Integer.valueOf(parts[i]);
            res[i] = val;
        }
        return res;
    }
}
