class Solution {
    public int kthSmallest(int[][] mat, int k) {
        List<Integer> ret = new ArrayList<>();
        ret.add(0);
        
for (int i = 0;i < mat.length;i++) {
            List<Integer> temp = new ArrayList<>();
            for (int x: ret) {
                for (int y: mat[i]) {
                    temp.add(x + y);
                }
            }
            Collections.sort(temp);
            ret = temp.subList(0, Math.min(k, temp.size()));
        }
        return ret.get(k - 1);
    }
}
