class Solution {

    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length;
        if (m == 0) {
            return 0;
        }
        int n = mat[0].length;
        HashSet<String> set = new HashSet<>();
        int[] start = new int[m];
        Arrays.fill(start, 0);
        PriorityQueue<int[]> pq = new PriorityQueue<>(new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                int s1 = 0;
                
int s2 = 0;
                for (int i = 0; i < m; i++) {
                    s1 += mat[i][o1[i]];
                    s2 += mat[i][o2[i]];
                }
                return s1 - s2;
            }
        });
        pq.offer(start);
        set.add(Arrays.toString(start));
        while (k-- > 1) {
            int[] cur = pq.poll();

            for (int i = 0; i < m; i++) {
                int[] temp = cur.clone();
                if (temp[i] + 1 < n) {
                    temp[i] += 1;
                    
if (!set.contains(Arrays.toString(temp))){
                        pq.offer(temp);
                        set.add(Arrays.toString(temp));
                    }
                }
            }
        }
        int[] last = pq.poll();
        
int res = 0;
        for (int i = 0; i < last.length; i++) {
            res += mat[i][last[i]];
        }
        return res;
    }
}
