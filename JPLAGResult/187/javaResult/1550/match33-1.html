	class S
olution {
	    public int kthSmallest(int[][] a, int k) {
	        PriorityQueue<int[]> pq = new PriorityQueue<>((x, y) -> x[0] - y[0]);
	        int n = a.length;
	        int[] ini = new int[n+1];
	        for(int[] row : a)ini[0] += row[0];
	        pq.add(ini);
	        
	        int m = a[0].length;
	        Set<Long> cache = new HashSet<>();
	        for(int rep = 1;rep < k;rep++){
	        	int[] cur = pq.poll();
	        	for(int i = 0;i < n;i++){
	        		int[] nex = Arrays.copyOf(cur, n+1);
	        		if(++nex[i+1] == m)continue;
	        		nex[0] += a[i][nex[i+1]] - a[i][nex[i+1]-1];
	        		long h = 0;

	        		for(int u : nex){
	        			h = h * 1000000009 + u;
	        		}
	        		if(cache.add(h)){
	        			pq.add(nex);
	        		}
	        	}
	        }
	        return pq.poll()[0];
	    }
	}
