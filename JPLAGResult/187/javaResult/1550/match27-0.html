class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int rows = mat.length, cols = mat[0].length;
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>((i1, i2)-> {
            int sum1 = 0, sum2 = 0;
            for(int row = 0; row<rows; row++) {
                sum1+= mat[row][i1[row]];
                sum2 += mat[row][i2[row]];
            }
            return sum1-sum2;
        });
        
        int[] start = new int[rows];
        
        pq.add(start);
        
        Set<String> set = new HashSet();
        set.add(Arrays.toString(start));
        while(!pq.isEmpty() && k>1) {
            k--;
            int[] cur = pq.poll();
            
            int test = 0;
            for(int row = 0; row<rows; row++) {
                test+= mat[row][cur[row]];
            }
            //System.out.println(Arrays.toString(cur));
            //System.out.println(test);
            for(int row=0; row<rows; row++) {
                int[] next = new int[rows];
                for(int i=0; i<rows; i++) next[i] = cur[i];
                if(next[row]+1<cols) {
                    next[row] += 1;
                    if(set.contains(Arrays.toString(next))) continue;
                    set.add(Arrays.toString(next));
                    pq.add(next);
                }
            }
        }
        
        int[] idxes = pq.poll();
        int res = 0;
        for(int row = 0; row<rows; row++) {
            res+= mat[row][idxes[row]];
        }
        return res;
    }
}
