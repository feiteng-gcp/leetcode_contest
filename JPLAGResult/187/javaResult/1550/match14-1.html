class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int[] current = new int[mat.length];
        
PriorityQueue<int[]> queue = new PriorityQueue<int[]>(
            (i1 , i2) ->
            {                                          
                   int sum1 = 0, sum2 = 0;
                   for (int i = 0; i < mat.length; i++)
                   {
                       sum1 += mat[i][i1[i]];
                       sum2 += mat[i][i2[i]];
                   }
                
                return sum1 - sum2;
            }
            );
        
        HashSet<String> visited = new HashSet<String>();
        queue.offer(current);
        
        while (k > 1)
        {
            k--;
            int[] top = 
queue.poll();
            //System.out.println(Arrays.toString(top));
            for (int i = 0; i < mat.length; i++)
            {
                if (top[i] == mat[0].length - 1) continue;
                int[] copy = Arrays.copyOf(top, top.length);
                copy[i]++;
                String c = Arrays.toString(copy);
                if (!visited.contains(c))
                {
                    //System.out.println("offering " + Arrays.toString(copy));
                    queue.offer(copy);
                    visited.add(c);
                }
                
            }
        }
        
        int ans = 0;
        current = queue.poll();
        
for (int i = 0; i < current.length; i++)
        {
            ans += mat[i][current[i]];
        }
        
        return ans;
    }
}
