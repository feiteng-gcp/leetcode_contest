class Solution {
   public int kthSmallest(int[][] mat, int k) {
    	PriorityQueue<Integer> Q = new PriorityQueue<>((a,b) -> b-a);
    	int smallest = 0;
    	for(int i = 0; i < mat.length; i++)
    		smallest += mat[i][0];
    	//Q.add(smallest);
    	//System.out.println(Q);
    	recursion(mat, 0, smallest, Q, k);
    	while(Q.size() > k) {
    		Q.poll();
    	}
    	//System.out.println(Q);
    	return Q.poll();
    }
    
    private void recursion(int[][] mat, int curX, int ttl, PriorityQueue<Integer> Q, int k) {
    	while(Q.size() > k) {
    		Q.poll();
    	}
    	if(curX == mat.length) {
    		Q.add(ttl);
    		return;
    	}
    	if(Q.size() >= k && ttl >= Q.peek()) return;
    	
    	recursion(mat, curX+1, ttl, Q, k);
    	for(int i = 1; i < mat[0].length; i++) {
    		ttl -= mat[curX][i-1];
    		ttl += mat[curX][i];
    		recursion(mat, curX+1, ttl, Q, k);
    	}
    }
}
