class Data implements Comparable<Data>{
    int sum;
    int[] array;
    Data(int[] array, int sum){
        this.array = array;
        this.sum = sum;
    }
    
    @Override
    public int compareTo(Data that){
        return Integer.compare(sum, that.sum);
    }
}
class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length;
        int n = mat[0].length;
        HashSet<String> set = new HashSet<>();
        PriorityQueue<Data> pq = new PriorityQueue<>();
        int[] start = new int[m];
        pq.add(new Data(start, helper(mat, start)));
        Data top = null; 
        while(k > 0){
            top = pq.poll();
            int[] cur =  top.array;
            for(int i=0;i<m;i++){
                if(cur[i]<n-1){
                    int[] next = cur.clone();
                    next[i] = cur[i] + 1;
                    String hashVal = hashArray(next);
                    if(!set.contains(hashVal)){
                        int sum = top.sum + mat[i][next[i]] - mat[i][cur[i]];
                        pq.add(new Data(next, sum));
                        set.add(hashVal);
                    }
                }
            }
            k--;
        }
        return top.sum;
    }
    
    private String hashArray(int[] arr){
        StringBuilder sb = new StringBuilder();
        for(int num : arr){
            sb.append(num + ",");
        }
        return sb.toString();
    }
    private int helper(int[][] mat, int[] start){
        int m = mat.length;
        int n = mat[0].length;
        int ans = 0;
        for(int i=0;i<m;i++){
            ans += mat[i][start[i]];
        }
        return ans;
    }
}
