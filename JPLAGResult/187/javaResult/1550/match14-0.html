class Solution {
    public int kthSmallest(int[][] mat, int k) {
        TreeSet<int[]> pq = new TreeSet<>((x, y) -> {
            int sum1 = 0, sum2 = 0;
            for (int i = 0; i < x.length; ++i) {
                sum1 += mat[i][x[i]];
                sum2 += mat[i][y[i]];
            }
            if (sum1 != sum2) return sum1 - sum2;
            for (int i = 0; i < x.length; ++i) {
                if (x[i] != y[i]) return x[i] - y[i];
            }
            return 0;
        });
        int n = mat.length, m = mat[0].length;
        int[] first = new int[n];
        Arrays.fill(first, 0);
        pq.add(first);
        int[] ans = new int[n];
        while (k > 0) {
            ans = pq.pollFirst();
            for (int i = 0; i < n; ++i) {
                if (ans[i] == m - 1) continue;
                int[] tmp = Arrays.copyOf(ans, n);
                tmp[i]++;
                pq.add(tmp);
            }
            --k;
        }
        int sum = 0;
        for (int i = 0; i < n; ++i) {
            sum += mat[i][ans[i]];
        }
        return sum;
    }
}
