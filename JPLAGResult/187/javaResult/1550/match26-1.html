class Solution {
    public int kthSmallest(int[][] mat, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> Integer.compare(b, a));
        pq.offer(0);
        pq = 
collectSums(mat, 0, k, pq);
        while (pq.size() > k) {
            //System.out.print(pq.peek() + " ");
            pq.poll();
        }
        //System.out.println();
        
        return pq.peek();
    }
    
    PriorityQueue<Integer> collectSums(int[][] mat, int row, int k, PriorityQueue<Integer> pq) {
        if (row == mat.length) {
            return pq;
        }
        
        PriorityQueue<Integer> next = new PriorityQueue<>((a, b) -> Integer.compare(b, a));
        while (!pq.isEmpty()) {
            int sum = pq.poll();
            for (int j = 0; j < mat[0].length; ++j) {
                next.offer(sum + mat[row][j]);
                if (next.size() > k) {
                    next.poll();
                }
            }
        }
        
        return collectSums(mat, row + 1, k, next);
    }
}
