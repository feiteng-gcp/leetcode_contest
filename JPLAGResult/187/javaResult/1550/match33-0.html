class Solution {
    public int kthSmallest(int[][] mat, int k) {
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>((A, B) -> A[A.length - 1] - B[B.length - 1]);
        int m = mat.length, n = mat[0].length, sum = 0;
        for(int i = 0; i < mat.length; i++) {
            sum += mat[i][0];
        }
        for(int i = 0; i < m; i++) {
            int[] arr = new int[m + 1];
            arr[m] = sum + mat[i][1] - mat[i][0];
            arr[i] = 1;
            pq.add(arr);
        }
        if(k == 1) {
            return sum;
        }
        k--;
        HashSet<String> set = new HashSet<String>();
        while(k > 0) {
            int[] curr = pq.remove();
            // System.out.print(k + " " + curr[m] + " ");
            if(k == 1) {
                return curr[m];
            }
            for(int i = 0; i < m; i++) {
                if(curr[i] < n - 1) {
                    int[] arr = new int[m + 1];
                    String str = "";
                    for(int j = 0; j <= m; j++) {
                        arr[j] = curr[j];
                    }
                    if(set.contains(str)) {
                        continue;
                    }
                    arr[i]++;
                    arr[m] += mat[i][arr[i]] - mat[i][curr[i]];
                    for(int j = 0; j < m; j++) {
                        str += arr[j] + "-";
                    }
                    if(set.contains(str)) {
                        continue;
                    }
                    pq.add(arr);
                    set.add(str);
                }
            }
            k--;
        }
        return pq.remove()[m];
    }
}
