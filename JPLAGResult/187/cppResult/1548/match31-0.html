class Solution {
public:
    bool kLengthApart(vector<int>& nums, int k) {
        vector<int> vIndex;
        int N=nums.size();
        for(int i=0;i<N;i++){
            if(nums[i]==1){
                vIndex.push_back(i);
            }
        }
        for(int i=1;i<vIndex.size();i++){
            if(vIndex[i]<=vIndex[i-1]+k){
                return false;
            }
        }
        
        return true;
    }
};
