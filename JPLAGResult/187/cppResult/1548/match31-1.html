class Solution {
public:
    bool kLengthApart(vector<int>& nums, int k) {
        vector<int> pos;
        int n = nums.size();
        for (int i = 0; i < n; ++i) {
            if (nums[i] == 1) {
                pos.push_back(i);
            }
        }
        
        int m = pos.size();

        for (int i = 1; i < m; ++i) {
            // cout << pos[i] << " " << pos[i - 1] << " " << k << endl;
            if (pos[i] - pos[i - 1] - 1 < k) {
                return false;
            }
        }
        return true;
    }
};
