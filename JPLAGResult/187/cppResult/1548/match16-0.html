class Solution {
public:
    bool kLengthApart(vector<int>& nums, int k) {
        int n = nums.size();
        int lst = -100000;
        for (int i = 0; i < n; ++i){
            if (nums[i] == 1){
                if (i - lst - 1 < k) return false;
                lst = i;
            }
            
        }
        return true;
    }
};
