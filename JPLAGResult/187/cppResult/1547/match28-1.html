class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        map<string, bool> tab;
        set<string> S;
        for(auto& v : paths) {
            S.insert(v[0]);
            S.insert(v[1]);
            tab[v[0]] = true;
        }
        
        for(auto& s : S)
            if(!tab.count(s)) return s;
        return "";
    }
};
