class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int N = nums.size();
        deque<int> dmax;
        deque<int> dmin;
        int b = 0;
        int res = 0;
        for (int i = 0; i < N; ++i) {
            while (!dmax.empty() && dmax.back() < nums[i]) {
                dmax.pop_back();
            }
            while (!dmin.empty() && dmin.back() > nums[i]) {
                dmin.pop_back();
            }
            dmax.push_back(nums[i]);
            dmin.push_back(nums[i]);
            while (1) {
                if (dmax.front() - dmin.front() <= limit) {
                    break;
                }
                if (nums[b] == dmax.front()) {
                    dmax.pop_front();
                }
                if (nums[b] == dmin.front()) {
                    dmin.pop_front();
                }
                ++b;
            }
            res = max(res, i - b + 1);
        }
        return res;
    }
};
