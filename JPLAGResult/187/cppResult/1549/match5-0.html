class Solution {
public:
    int n;
    int Mn[20][100005];
    int Mx[20][100005];
    int getMin(int u, int v) {
        int k = log2(v-u+1);
        return min(Mn[k][u], Mn[k][v-(1<<k)+1]);
    }
    int getMax(int u, int v) {
        int k = log2(v-u+1);
        return max(Mx[k][u], Mx[k][v-(1<<k)+1]);
    }
    int longestSubarray(vector<int>& nums, int limit) {
        n = nums.size();
        for (int i = 1; i <= n; i++) Mn[0][i] = Mx[0][i] = nums[i-1];
        for (int k = 1; (1 << k) <= n; k++)
            for (int i = 1; i + (1 << k) - 1 <= n; i++) {
                Mx[k][i] = max(Mx[k-1][i], Mx[k-1][i+(1 << (k-1))]);
                Mn[k][i] = min(Mn[k-1][i], Mn[k-1][i+(1 << (k-1))]);   
            }
        int pointer = 1;
        int res = 0;
        for (int i = 1; i <= n; i++) {
            while (pointer <= i && (getMax(pointer, i) - getMin(pointer, i) > limit)) 
                pointer++;
            res = max(res, i - pointer+1);
        }
        return res;
    }
};
