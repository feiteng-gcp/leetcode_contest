class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int ret = 1;
        map<int, int> r;
        r[nums[0]]++;
        int count = 1;
        for (int i = 0; i < nums.size(); ++i) {
            if (!i) continue;
            if (nums[i] >= r.rbegin()->first - limit && nums[i] <= r.begin()->first + limit) {
                r[nums[i]]++;
                ++count;
                ret = max(ret, count);
                            // cout << i << " " << r.size() << " " << r.rbegin()->first << " " << r.begin()->first<<endl;

                continue;
            }
            while (count && (nums[i] < r.rbegin()->first - limit || nums[i] > r.begin()->first + limit) ) {
                int last = i - count;
                r[nums[last]]--;
                --count;
                if (r[nums[last]] == 0) {
                    r.erase(nums[last]);
                }
            }
            r[nums[i]]++;
            ++count;
        }

            // cout << beg  << " " << end<<endl;
        return ret;
    }
};
