class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        deque<int> mx, mn;
        int ans = 0;
        for (int i = 0, j = 0; j < nums.size(); j++) {
            while (mx.size() && nums[j] > mx.back()) mx.pop_back();
            mx.push_back(nums[j]);
            
            while (mn.size() && nums[j] < mn.back()) mn.pop_back();
            mn.push_back(nums[j]);
            
            while (mx.front() - mn.front() > limit) {
                if (mx.front() == nums[i]) mx.pop_front();
                if (mn.front() == nums[i]) mn.pop_front();
                i++;
            }
            ans = max(ans, j - i + 1);
        }
        return ans;
    }
};
