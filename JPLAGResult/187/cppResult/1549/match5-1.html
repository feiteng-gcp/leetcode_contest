class Solution {
public:
    int l[100001], st[100001][18][2];
    vector<int> a;
    
    void build(int N) {
        l[1] = 0;
        for (int i = 2; i <= 100000; i++)
            l[i] = l[i/2] + 1;

        for (int i = 0; i < N; i++)
            st[i][0][0]=st[i][0][1] = a[i];

        for (int j = 1; j <= 17; j++)
            for (int i = 0; i + (1 << j) <= N; i++) {
                st[i][j][0] = min(st[i][j-1][0], st[i + (1 << (j - 1))][j - 1][0]);
                st[i][j][1] = max(st[i][j-1][1], st[i + (1 << (j - 1))][j - 1][1]);
            }
    }
    

    int qryMin(int L, int R) {
        int j = l[R - L + 1];
        return min(st[L][j][0], st[R - (1 << j) + 1][j][0]);
    }
    
    int qryMax(int L, int R) {
        int j = l[R - L + 1];
        return max(st[L][j][1], st[R - (1 << j) + 1][j][1]);
    }
    
    int longestSubarray(vector<int>& nums, int limit) {
        a=nums;
        int n=nums.size();

        int last=0, ans=1;
        build(n);
        for (int i=1; i<n; ++i) {
            while (qryMax(last, i)-qryMin(last, i)>limit)
                last++;
            cout << qryMax(last, i) << ' ' << qryMin(last, i) << '\n';
            //cout << i << ' ' << last << '\n';
            ans=max(ans, i-last+1);
        }
        return ans;
    }
};
