class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int ans = 1;

        multiset<int> S;
        int l = 0;
        for(int i = 0; i < nums.size(); i++) {
            S.insert(nums[i]);
            while(!S.empty() && (*prev(S.end()) - *S.begin()) > limit)
                S.erase(S.find(nums[l++]));

            ans = max(ans, i-l+1);
        }
        
        return ans;
    }
};
