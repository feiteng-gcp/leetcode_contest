class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int N=nums.size();
        list<int> lbig;
        list<int> lsmall;
        int l=0;
        int ans=0;
        for(int r=0;r<N;r++){
            while(!lbig.empty()&&lbig.back()<nums[r]){
                lbig.pop_back();
            }
            lbig.push_back(nums[r]);
            while(!lsmall.empty()&&lsmall.back()>nums[r]){
                lsmall.pop_back();
            }
            lsmall.push_back(nums[r]);
            while(lbig.front()-lsmall.front()>limit){
                if(nums[l]==lbig.front()){
                    lbig.pop_front();
                }
                if(nums[l]==lsmall.front()){
                    lsmall.pop_front();
                }
                l++;
            }
            ans=max(ans,r-l+1);
        }
        return ans;
        
        
    }
};
