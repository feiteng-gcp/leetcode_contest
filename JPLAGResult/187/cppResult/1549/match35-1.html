class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int> ms;
        int i=0;
        int res=0;
        for (int j=0; j<nums.size(); ++j) {
            ms.insert(nums[j]);
            int ma=*(ms.rbegin()), mi=*(ms.begin());
            if (ma-mi<=limit) {
                res = max(res, j-i+1);
            } else {
                while(ma-mi>limit) {
                    ms.erase(ms.find(nums[i]));
                    i++;
                    ma=*(ms.rbegin()), mi=*(ms.begin());
                }
            }
        }
        return res;
    }
};
