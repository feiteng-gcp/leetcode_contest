class Solution {
public:
    struct state
    {
        int n;
        int pos[50];
        int cur;
        bool operator<(const state& rhs) const
        {
            if (cur != rhs.cur)
                return cur > rhs.cur;
            else
                for (int i = 0; i < n; ++i)
                    if (pos[i] != rhs.pos[i])
                        return pos[i] < rhs.pos[i];
            return false;
        }
    };
    int kthSmallest(vector<vector<int>>& mat, int k) {
        priority_queue<state> pq;
        set<state> used;
        state s;
        s.n = mat.size();
        int n = mat.size();
        int m = mat[0].size();
        for (int i = 0; i < n; ++i)
        {
            s.pos[i] = 0;
            s.cur += mat[i][0];
        }
        pq.push(s);
        used.insert(s);
        for (int i = 1; i < k; ++i)
        {

            s = pq.top(); pq.pop();
            for (int j = 0; j < n; ++j)
            {
                if (s.pos[j] + 1 < m)
                {
                    s.cur -= mat[j][s.pos[j]];
                    ++s.pos[j];
                    s.cur += mat[j][s.pos[j]];
                    if (used.find(s) == used.end())
                    {
                        pq.push(s);
                        used.insert(s);
                    }
                    s.cur -= mat[j][s.pos[j]];
                    --s.pos[j];
                    s.cur += mat[j][s.pos[j]];                    
                }
            }
        }
        return pq.top().cur;
    }
};
