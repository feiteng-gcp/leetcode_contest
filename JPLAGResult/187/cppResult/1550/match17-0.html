struct Node{
    int sum = 0;
    vector<int> index;
};

struct cmp {
	bool operator() (const Node& a, const Node& b) {
		return a.sum>b.sum;
	}
};

class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        priority_queue<Node, vector<Node>, cmp> pq;
        set<vector<int>> visited;
        
        Node node;
        int n = mat.size();
        int m = mat[0].size();
        
        node.index = vector<int>(n);
        node.sum = calc(mat, node.index);
        
        visited.insert(node.index);
        pq.push(node);
        
        
        
        Node res;
        while (k--) {
            res = pq.top();
            pq.pop();
            vector<int> index = res.index;
            for (int i = 0; i<n; ++i) {
                if (index[i]<m-1) {
                    ++index[i];
                    Node node;
                    node.index = index;
                    node.sum = calc(mat, node.index);
                    if (!visited.count(node.index)) {
                        pq.push(node);
                        visited.insert(node.index);
                    }                    
                    --index[i];
                }
            }
        }
        
        return res.sum;
    }
    
    int calc(vector<vector<int>>& mat, vector<int>& index) {
        int res = 0;
        for (int i = 0; i<mat.size(); ++i) {
            res += mat[i][index[i]];
        }
        
        return res;
    }
};
