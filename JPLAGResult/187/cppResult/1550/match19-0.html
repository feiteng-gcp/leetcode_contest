class Solution {
public:
    int kthSmallest(vector<vector<int>>& a, int k) {
        int n = a.size(), m = a[0].size();
        vector<int> cur(n, 0);
        int sum = 0;
        for (int i = 0; i < n; ++i) sum += a[i][0];
        priority_queue<pair<int, vector<int>>> Q;
        set<vector<int>> visit;
        Q.push({-sum, cur});
        visit.insert(cur);
        while (!Q.empty()) {
            auto [sum, cur] = Q.top();
            Q.pop();
            // cout << -sum << endl;
            // for (auto& v : cur) cout << v << " ";
            // cout << endl;
            if (--k == 0) return -sum;
            for (int i = 0; i < n; ++i) {
                if (cur[i] + 1 < m) {
                    vector<int> nxt = cur;
                    nxt[i] += 1;
                    if (visit.find(nxt) != visit.end()) continue;
                    int nxt_sum = -sum - a[i][cur[i]] + a[i][nxt[i]];
                    Q.push({-nxt_sum, nxt});
                    visit.insert(nxt);
                }
            }
        }
        return -1;
    }
};
