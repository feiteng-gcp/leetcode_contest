vector<vector<int>> m;
int sum_m(const vector<int> &v) {
  int sum = 0;
  for (int i = 0; i < v.size(); i++) {
    sum += m[i][v[i]];
  }
  return sum;
}
struct Cmp {
 public:
  bool operator()(const vector<int> &v1, const vector<int> &v2) {
    return sum_m(v1) > sum_m(v2);
  }
};
priority_queue<vector<int>, vector<vector<int>>, Cmp> pq;
class Solution {
 public:
  int kthSmallest(vector<vector<int>> &mat, int k) {
    while(!pq.empty()) pq.pop();
    m = mat;
    vector<int> z(mat.size(), 0);
    pq.push(z);
    set<vector<int>> in;
    in.insert(z);
    for (int i = 0; i < k - 1; i++) {
      auto head = pq.top();
      pq.pop();
      for (int i = 0; i < mat.size(); i++) {
        if (head[i] + 1 < mat[i].size()) {
          vector<int> tmp = head;
          tmp[i]++;
          if (in.find(tmp) == in.end()) {
            pq.push(tmp);
            in.insert(tmp);
          }
        }
      }
    }
    return sum_m(pq.top());
  }
};
