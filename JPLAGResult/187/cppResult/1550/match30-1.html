class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n = mat.size();
        int m = mat[0].size();
        
        vector<int> f;
        f.push_back(0);
        
        for (int i = 0; i < n; i++) {
            vector<int> g;
            for (int a : f)
                for (int j = 0; j < m; j++)
                    g.push_back(a + mat[i][j]);
            sort(g.begin(), g.end());
            
            f.clear();
            for (int j = 0; j < k; j++) {
                if (j >= g.size()) break;
                f.push_back(g[j]);
            }
        }
        
        return f[k-1];
    }
};
