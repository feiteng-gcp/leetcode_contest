class Solution {
public:
    static constexpr int maxn = 2e5+10;
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n = mat.size(), m = mat[0].size();
        vector<int> sum;
        sum.push_back(0);

        for(int i = 0; i < n; i++) {
            vector<int> nxt;
            for(int j = 0; j < m; j++) {
                for(auto v: sum) {
                    nxt.push_back(v+mat[i][j]);
                }
            }
            sort(nxt.begin(), nxt.end());
            if(nxt.size() > k) {
                nxt.resize(k);
            }
            sum = nxt;
        }
        return sum[k-1];
    }
};
