class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
      int n = mat.size();
      int m = mat[0].size();

      int w = 0;
      for (int i = 0; i < n; i++) w += mat[i][0];
      vector<int> a = vector<int>(n, 0);

      set<vector<int>> st;
      multimap<int, vector<int>> mp;

      mp.insert({w, a});

      while(true) {
        auto iter = mp.begin();

        int w = iter->first;
        auto v = iter->second;
        mp.erase(iter);
        if (st.find(v) != st.end()) {
          continue;
        }
        st.insert(v);
        k--;
        if (k == 0) {
          return w;
        }

        for (int j = 0; j < n; j++) {
          if (v[j] + 1 < m) {
            int nw = w - mat[j][v[j]] + mat[j][v[j] + 1];
            v[j] += 1;
            mp.insert({nw, v}); 
            v[j] -= 1;
          }
        }
      }
      return 0;
    }
};
