class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        // vector<int> sum(200005, 0);
        int base = 0;
        int m = mat.size(), n = mat[0].size();
        vector<int> s{0};
        for(int i=0;i<m;i++){
            vector<int> ns;
            for(int ii:s){
                for(int j:mat[i]){
                    ns.push_back(ii + j);
                }
            }
            sort(begin(ns), end(ns));
            if(ns.size()>200) ns.resize(200);
            swap(s, ns);
        }
        // cout<<s.size();
        return s[k-1];
    }
};
