class Solution {
public:
    int f[44][200] = {0};
    int res[205] = {0}, p[205] = {0};
    int kthSmallest(vector<vector<int>>& mat, int K) {
        int n = mat.size();
        int m = mat[0].size();
        
        for (int i = 0; i < n; ++i) 
            for (int j = 0; j < m; ++j)
                res[j] += !i ? mat[i][j] : mat[i][0];

        
        for (int i = m; i < K; i++)
            res[i] = 1e9;
        
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < m; j++)
                for (int k = 0; k < K; k++)
                    f[j][k] = res[k] - mat[i][0] + mat[i][j];
            fill(p, p + K, 0);
            for (int j = 0; j < K; j++) {
                int tp = -1;
                for (int k = 0; k < m; k++) {
                    if (p[k] == K) continue;
                    if (tp == -1 || f[tp][p[tp]] > f[k][p[k]])
                        tp = k;
                }
                assert(tp != -1);
                res[j] = f[tp][p[tp]++];
            }
        }

        
        for (int i = 0; i < K; i++)
            cout << res[i] << " ";

        return res[K - 1];
            
        
    }
};
