class Solution {
public:
    int d[40][40], dp[40][200001];
    
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n=mat.size();
        int m=mat[0].size();
        int sum=0;
        for (auto x : mat)
            sum+=x[0];
        
        for (int i=0; i<n; ++i)
            for (int j=0; j<m; ++j)
                d[i][j]=mat[i][j]-mat[i][0];
        
        for (int i=0; i<m; ++i)
            ++dp[0][d[0][i]];
        
        int MX=d[0][m-1], MX2=1e9;
        
        for (int i=1; i<n; ++i) {
            MX+=d[i][m-1];
            for (int j=0; j<=min(MX, MX2)+2; ++j) {
                for (int k=0; k<m; ++k) {
                    if (dp[i][j]>=205) {
                        MX2=j;
                        break;
                    }
                    if (j-d[i][k]<0)
                        continue;
                    dp[i][j]+=dp[i-1][j-d[i][k]];
                }
            }
        }
        for (int i=0; i<=200000; ++i) {
            if (k<=dp[n-1][i])
                return i+sum;
            k-=dp[n-1][i];
        }
        return -1;
    }
};
