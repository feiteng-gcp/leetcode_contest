vector<int> lst,crr;int tmp[100];
class Solution {
public:
    

    int kthSmallest(vector<vector<int>>& mat, int k) {
        int  m = mat.size();
        int n = mat[0].size();
        lst.clear();
        for(int i=0;i<n;i++)
            lst.push_back(mat[0][i]);
        
        
        
        for(int i=1;i<m;i++)
        {
            int lsts = lst.size();

            int news = min(lsts*n,k);
            
            crr.resize(news);
            
            for(int j=0;j<n;j++)tmp[j]=0;
            
            for(int l=0;l<news;l++)
            {
                int tmps = 100000000; int j1 = -1;
                for(int j=0;j<n;j++)
                {
                    if(tmp[j]==lsts)
                        continue;
                    if(mat[i][j]+lst[tmp[j]]<tmps)
                    {
                        tmps = mat[i][j]+lst[tmp[j]];
                        j1 = j;
                    }
                }
                crr[l]=tmps;tmp[j1]++;
            }
            
            lst.resize(news);
            for(int l=0;l<news;l++)
                lst[l]=crr[l]; 
        }
        
        return lst[k-1];
        
        
    }
};
