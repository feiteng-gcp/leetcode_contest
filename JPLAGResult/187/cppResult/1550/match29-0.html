class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n=mat.size();
        int m = mat[0].size();
        vector<int> old;
        for(int i=0;i<m;i++)old.push_back(mat[0][i]);
        for(int i=1;i<n;i++){
            vector<int> res;
            for(int j=0;j<m;j++){
                int lim;
                if(old.size()<k)lim = old.size();
                else lim = k;
                for(int x=0;x<lim;x++){
                    res.push_back(old[x]+mat[i][j]);
                }
            }
            sort(res.begin(), res.end());
            old = res;
        }
        return old[k-1];
    }
};
