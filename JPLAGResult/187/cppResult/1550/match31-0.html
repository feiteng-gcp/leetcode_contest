int n, m;
priority_queue<int> que[2];
vector<int> vct;

void Push(priority_queue<int> &que, int x, int k) {
    que.push(x);
    if((int)que.size() > k) {
        que.pop();
    }
}

void Clear(priority_queue<int> &que) {
    while(!que.empty()) {
        que.pop();
    }
}

void Change(priority_queue<int> &que, vector<int> &vct) {
    vct.clear();
    while(!que.empty()) {
        int tmp = que.top();
        vct.push_back(tmp);
        que.pop();
    }
}

class Solution {
public:
    int kthSmallest(vector<vector<int>>& num, int k) {
        for(int i = 0; i < 2; ++i) {
            Clear(que[i]);
        }
        n = num.size();
        m = num[0].size();
        for(int j = 0; j < m; ++j) {
            Push(que[0], num[0][j], k);
        }
        for(int i = 1; i < n; ++i) {
            int ii = i % 2;
            int iii = 1 - ii;
            Clear(que[ii]);
            Change(que[iii], vct);
            int len = vct.size();
            for(int j = 0; j < m; ++j) {
                for(int kk = 0; kk < len; ++kk) {
                    Push(que[ii], num[i][j] + vct[kk], k);
                }
            }
        }
        return que[(n - 1) % 2].top();
    }
};
