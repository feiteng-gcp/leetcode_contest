class Solution {
public:
    struct hyc
 {
      int sum;
      vector<int> index;
      hyc(int sum1, vector<int> i) {
          sum = sum1;
          index = i;

      }
    bool operator<(const hyc& a) const
    {
         return sum > a.sum; 
     }
 };
    int kthSmallest(vector<vector<int>>& mat, int K) {
        priority_queue<hyc> d;
        int n = mat.size();
        int m = mat[0].size();
        vector<int> A;
        int sum = 0;
        for (int i = 0; i < n; i++)
        {
            sum += mat[i][0];
            A.push_back(0);
        }
        d.push(hyc(sum, A));
        set<vector<int>> S;
        int ans=0;
        while (K--) {
            hyc x(d.top());
            while (1) {
             x = d.top();
            d.pop();
                if (S.count(x.index) == 0) break;
            }
            S.insert(x.index);
            vector<int> new_index;
            int sum = x.sum;
            ans = sum;
           // cout << sum << endl;
        //    for (int i = 0; i < n; i++)
        //        cout << x.index[i] << " ";
        //    cout << endl;
            for (int i = 0; i < n; i++) 
            if (x.index[i] < m - 1)
            {
                new_index = x.index;
                new_index[i] ++;
                int new_sum = sum + mat[i][new_index[i]] - mat[i][new_index[i]-1];
                d.push(hyc( new_sum, new_index));
            }
        }
        return ans;
    }
};
