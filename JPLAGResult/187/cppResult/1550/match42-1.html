class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int m=mat.size(), n=mat[0].size();
        set<vector<int>> visited;
        int sum = 0;
        for (int i=0; i<m; ++i) {
            sum += mat[i][0];
        }
        vector<int> cur(m, 0);
        visited.insert(cur);
        int cnt=0;
        priority_queue<pair<int, vector<int>>, vector<pair<int, vector<int>>>, greater<pair<int, vector<int>>>> pq;
        pq.push({sum, cur});
        while (true) {
            auto p=pq.top();
            pq.pop();
            cnt++;
            if (cnt==k) {
                return p.first;
            }
            auto v=p.second;
            for (int i=0; i<m; ++i) {
                if (v[i]<n-1) {
                    vector<int>nxt(v.begin(), v.end());
                    nxt[i]++;
                    if (visited.find(nxt) == visited.end()) {
                        visited.insert(nxt);
                        int s=p.first-mat[i][v[i]]+mat[i][v[i]+1];
                        pq.push({s, nxt});
                    }
                }
            }
        }
        return -1;
    }
};
