struct cmp
{
    bool operator()(vector<int> &a,vector<int> &b)
    {
        return a.back() > b.back();
    }
};


class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        priority_queue<vector<int>,vector<vector<int>>,cmp> q;
        set<vector<int>> sett;
        int h =  mat.size();
        int w = mat[0].size();
        int s0 = 0;
        
        vector<int> buffer;
        for(int i=0;i < h;i++){
            s0 += mat[i][0];
            buffer.push_back(0);
        }
        buffer.push_back(s0);
        sett.insert(buffer);
        q.push(buffer);
        cout << "a"<<endl;
        for(int i=1;i <= k;i++){
            // cout << "==="<<i<<endl;
            if(i == k){
                vector<int> ret = q.top();
                return ret.back();
            }

            vector<int> temp = q.top();
            // cout << "select ";
            // for(auto x : temp){
            //     cout << x<<" ";
            // }
            // cout << endl;
            q.pop();
            for(int j=0;j< h;j++){
                vector<int> tempp = temp;
                tempp[j]++;
                if(tempp[j] >= w){
                    continue;
                }
                int backval = tempp.back();
                backval -= mat[j][tempp[j]-1];
                backval += mat[j][tempp[j]];
                tempp.back() = backval;
                if(sett.find(tempp) != sett.end()){
                    continue;
                }
                sett.insert(tempp);
                q.push(tempp);            
            
            }
        }
        return -1;
    }
};
