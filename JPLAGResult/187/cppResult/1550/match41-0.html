class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int m = mat.size(), n = mat[0].size();
        priority_queue<vector<int>> pq;
        
        int sum = 0;
        for(int i = 0; i < m; ++i) {
            sum += mat[i][0];
        }
        vector<int> cur;
        cur.push_back(-sum);
        for(int i = 0; i < m; ++i) cur.push_back(0);
        set<vector<int> > vis;
        pq.push(cur);
        vis.insert(cur);
        int res= 0;
        while(k--) {
            vector<int> cur = pq.top(); pq.pop();
            res = -cur[0];
            for(int i = 0; i < m; ++i) {
                int idx = cur[i + 1];
                if(idx + 1 < mat[i].size()) {
                    vector<int> tmp = cur;
                    tmp[0] += mat[i][idx] - mat[i][idx + 1];
                    tmp[i + 1] = idx + 1;
                    if(vis.find(tmp) != vis.end()) continue;
                    vis.insert(tmp);
                    pq.push(tmp);
                }
            }
        }

        return res;
    }
};
