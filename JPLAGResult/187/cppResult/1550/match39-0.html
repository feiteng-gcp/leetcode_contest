class Solution {
public:
	class Mycomp {
	public:
		bool operator()(const vector<int>& v1, const vector<int>& v2) {
			return v1[0] > v2[0];
		}
	};
	string getKey(vector<int>& tmp) {
		string key;
		for (int i = 1; i < tmp.size(); i++) {
			key += to_string(tmp[i]);
		}
		return key;
	}
	int kthSmallest(vector<vector<int>>& mat, int k) {
		int m = mat.size(), n = mat[0].size();
		int sum = 0;

		for (int i = 0; i < m; i++) {
			sum += mat[i][0];
		}
		if (n == 1)
			return sum;
		vector<int> col(m + 1, 0);
		col[0] = sum;
		priority_queue<vector<int>, vector<vector<int> >, Mycomp> pq;
		pq.push(col);
		int cnt = 0;
		string key(m, '0');
		unordered_set<string> visited;
		visited.insert(key);
		while (cnt < k) {
			vector<int> cur = pq.top();
			pq.pop();
			sum = cur[0];
			cnt++;
			
			for (int j = 1; j < cur.size(); j++) {
				int c = cur[j];
				if (c + 1 < n) {
					vector<int> nxt = cur;
					nxt[j] = c + 1;
					string ckey = getKey(nxt);
					if (visited.find(ckey) != visited.end())
						continue;
					visited.insert(ckey);
					nxt[0] += mat[j - 1][c + 1] - mat[j - 1][c];
					pq.push(nxt);
				}
			}
		}
		return sum;
	}
};
