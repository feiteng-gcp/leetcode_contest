class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n = mat.size();
        int m = mat[0].size();
        vector<multiset<int>> dp(n);
        for (int i = 0; i < m; ++i)
        {
            auto& s = dp[0];
            if (s.size() < k)
            {
                s.insert(mat[0][i]);
            }
            else
            {
                break;
            }
        }
        for (int i = 1; i < n; ++i)
        {
            const auto& s = dp[i - 1];
            auto& t = dp[i];
            for (int j = 0; j < m; ++j)
            {
                int v = mat[i][j];
                for (auto ii : s)
                {
                    t.insert(v + ii);
                }
            }
            while (t.size() > k)
            {
                t.erase(--t.end());
            }
        }
        return *(--dp[n - 1].end());
    }
};
