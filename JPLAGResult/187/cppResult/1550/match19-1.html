class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int m = mat.size(), n = mat[0].size();
        
        vector<int> ps(m, 0);
        int sum = 0;
        for (int i = 0; i < m; ++i) sum += mat[i][0];
        
        priority_queue<pair<int, vector<int>>> pq;
        pq.emplace(-sum, ps);
        
        set<vector<int>> seen;
        int last = -1;
        
        while (seen.size() != k) {
            auto [s, is] = pq.top();
            pq.pop();
            
            last = s;
            
            for (int r = 0; r < m; ++r) {
                if (is[r] + 1 < n) {
                    auto bak = s;
                    
                    s -= mat[r][is[r] + 1] - mat[r][is[r]];
                    ++is[r];
                    pq.emplace(s, is);
                    --is[r];
                    s = bak;

                }
            }
            
            seen.emplace(is);
        }
        
        return -last;
    }
};
