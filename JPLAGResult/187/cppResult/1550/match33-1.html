class Solution {
public:
    int dp[41][200010];
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int m=mat.size();
        int n=mat[0].size();
        memset(dp,0,sizeof(dp));
        int minVal=0,maxVal=0;
        for(int i=0;i<m;++i)
            minVal+=mat[i][0];
        for(int i=0;i<m;++i)
            maxVal+=mat[i][n-1];

        for(int i=0;i<n;++i)
            dp[m-1][mat[m-1][i]]+=1;
        for(int j=1;j<=maxVal;++j)
        {
            for(int i=m-2;i>=0;--i)
            {
                for(int k=0;k<n&&j>mat[i][k];++k)
                    dp[i][j]+=dp[i+1][j-mat[i][k]];
            }
            k-=dp[0][j];
            if(k<=0) return j;
        }
        return maxVal;
    }
};
