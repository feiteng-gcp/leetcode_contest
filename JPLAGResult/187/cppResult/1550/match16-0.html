class Solution {
 public:
  int kthSmallest(vector<vector<int>>& mat, int k) {
    int m = mat.size(), n = mat[0].size();
    auto cmp = [](pair<int, vector<int>> &p, pair<int, vector<int>> &q) {
      return p.first > q.first;
    };
    priority_queue<pair<int, vector<int>>, vector<pair<int, vector<int>>>, decltype(cmp)> pq(cmp);
    vector<int> idx(m, 0);
    int sum = 0;
    for (int i = 0; i < m; i++) sum += mat[i][0];
    pq.push({sum, idx});
    set<vector<int>> visited;
    for (int i = 0; i < k - 1; i++) {
      auto [sum, idx] = pq.top();
      pq.pop();
      for (int i = 0; i < m; i++) {
        if (idx[i] < n - 1) {
          int diff = mat[i][idx[i] + 1] - mat[i][idx[i]];
          sum += diff;
          idx[i]++;
          if (visited.find(idx) == visited.end()) {
            pq.push({sum, idx});  
            visited.insert(idx);
          }
          idx[i]--;
          sum -= diff;
        }
      }
    }
    return pq.top().first;
  }
};
