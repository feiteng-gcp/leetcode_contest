class Solution {
public:
    struct node {
        vector<int> vec;
        int sum;
        node(){}
        node(vector<int> vec,int sum):vec(vec),sum(sum){}
        bool operator<(const node &rhs)const{
            return sum>rhs.sum;
        }
    };
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n=mat.size();
        int m=mat[0].size();
        vector<int> vec;
        int sum=0;
        for (int i=0;i<n;++i) {
            vec.push_back(0);
            sum+=mat[i][0];
        }
        priority_queue<node> que;
        que.push(node(vec,sum));
        set<vector<int>> vis;
        vis.insert(vec);
        int cnt=0;
        while (!que.empty()) {
            ++cnt;
            vector<int> vec=que.top().vec;
            int sum=que.top().sum;
            que.pop();
            if (cnt==k) return sum;
            vector<int> nxt=vec;
            for (int i=0;i<n;++i) {
                int x=vec[i];
                if (x+1<m) {
                    ++nxt[i];
                    if (vis.find(nxt)==vis.end()) {
                        int nxtsum=sum+mat[i][x+1]-mat[i][x];
                        que.push(node(nxt,nxtsum));
                        vis.insert(nxt);
                    }
                    --nxt[i];
                }
            }
        }
        return -1;
    }
};
