class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int rows = mat.size();
        int cols = mat[0].size();
        map<int, vector<string>> omap;
        unordered_set<string> record;
        int init = 0;
        for(int i=0; i<rows; i++){
            init += mat[i][0];
        }
        omap[init] = {string(rows, '0')};
        record.insert(string(rows, '0'));
        int idx = 0;
        while(idx<k){
            auto itor = omap.begin();
            auto vecs = itor->second;
            auto val = itor->first;
            int size = itor->second.size();
            if(size+idx>=k){
                return val;
            }
            idx += size;
            omap.erase(omap.begin());

            for(auto vec: vecs){
                for(int r=0; r<rows; r++){
                    int c = vec[r]-'0';
                    if(c==cols-1) continue;
                    auto tempvec = vec;
                    tempvec[r]++;
                    if(record.count(tempvec)) continue;
                    record.insert(tempvec);
                    int delta = mat[r][c+1] - mat[r][c];
                    omap[val+delta].push_back(tempvec);
                }
            }
        }
        return 0;
    }
};
