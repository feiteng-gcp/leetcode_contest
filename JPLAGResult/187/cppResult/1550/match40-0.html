class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>>> pq;
        int sum = 0;
        int m = mat.size();
        int n = mat[0].size();
        vector<int> cur(m, 0);
        for (int i = 0; i < m; i++) {
            sum += mat[i][0];
        }
        cur.insert(cur.begin(), sum);
        set<vector<int>> visited = {cur};
        pq.push(cur);
        int cnt = 1;
        while (cnt < k) {
            cur = pq.top();
            pq.pop();
            cnt++;
            for (int i = 1; i < cur.size(); i++) {
                auto next = cur;
                next[i]++;
                if (next[i] >= mat[i - 1].size()) continue;
                //cout << i << "," << next[i] <<endl;
                next[0] += mat[i - 1][next[i]] - mat[i - 1][next[i] - 1];
                if (visited.find(next) == visited.end()) {
                    visited.insert(next);
                    pq.push(next);
                }
            }
        }
        return pq.top()[0];
    }
};
