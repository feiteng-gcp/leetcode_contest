class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        vector<int> idx;
        idx.resize(mat.size(), 0);

        int sum=0;
        for(int i=0;i<mat.size();i++){
            sum+=mat[i][0];
        }
        if(k==1) return sum;
        priority_queue<pair<int, vector<int>>, vector<pair<int, vector<int>>>, greater<pair<int, vector<int>>>> pq;
        pq.push(make_pair(sum, idx));
        set<vector<int>> used;
        while(k--){
            //printf("k=%d\n", k);
            pair<int, vector<int>> cur = pq.top();
            pq.pop();
            if(used.find(cur.second)!=used.end()){
                k++;
                continue;
            }
            used.insert(cur.second);

            sum = cur.first;
            //printf("sum %d\n", sum);
            for(int i=0;i<cur.second.size();i++){
                int j = cur.second[i];
                if(j<mat[i].size()-1){
                    int tmp = sum+mat[i][j+1]-mat[i][j];
                    cur.second[i]++;
                    pq.push(make_pair(tmp, cur.second));
                    cur.second[i]--;
                }
            }
        }
        return sum;
    }
};
