class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int m = mat.size(), n = mat[0].size();
        vector<int> s(1,0);
        // s.insert(0);
        for (int i = m-1; i >=0; --i) {
            vector<int> ss;
            for (int a: mat[i])
                for (int b: s)
                    ss.push_back(a+b);
            sort(ss.begin(), ss.end());
            s.clear();
            for (int j = 0; j < k && j < ss.size(); ++j) {
                s.push_back(ss[j]);
            }
        }
        return s[k-1];
    }
};
