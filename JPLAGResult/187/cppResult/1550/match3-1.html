class Solution {
public:
    vector<int> sz;
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n = mat[0].size();
        for (int i = 0; i < mat.size(); ++i) {
            if (sz.empty()) {
                sz = mat[i];
            } else {
                vector<int> summed;
                for (int j = 0; j < min<int>(k, n); ++j) {
                    for (auto&& v : sz) {
                        summed.emplace_back(v + mat[i][j]);
                    }
                }
                sort(summed.begin(), summed.end());
                //summed.resize(max<int>(summed.size(), k));
                sz.swap(summed);
            }
            sz.resize(min<int>(sz.size(), k));
        }
        return sz.back();
    }
};
