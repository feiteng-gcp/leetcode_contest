int dp[41][205];

class Solution {
public:
    int kthSmallest(vector<vector<int>>& a, int k) {
        int n=a.size(),m=a[0].size();
        priority_queue<int> pq[2];
        for(int i=0;i<min(k,m);i++){
            pq[0].push(a[0][i]);
        }
        for(int i=1;i<n;i++){
            while(pq[1^(i%2)].size()){
                for(int j=0;j<m;j++){
                    pq[i%2].push(pq[1^(i%2)].top()+a[i][j]);
                }
                pq[1^(i%2)].pop();
            }
            while(pq[i%2].size()>k)
                pq[i%2].pop();
        }
        return pq[(n-1)%2].top();
    }
};
