class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int m = mat.size();
        int n = mat[0].size();
        unordered_set<string> searched;
        auto comp = [](pair<int,vector<int>>& a, pair<int,vector<int>>& b) {
            return a.first > b.first;
        };
        priority_queue<pair<int,vector<int>>, vector<pair<int,vector<int>>>, decltype(comp)> pq(comp);
        vector<int> curr(m, 0);
        int s = getSum(mat, curr);
        pq.push(make_pair(s, curr));
        searched.insert(encodeVector(curr));
        int id = 1;
        while (pq.size() && id < k) {

            curr = pq.top().second;
            pq.pop();
            // for (int i = 0; i < m; i++)
            //     cout << curr[i] << " ";
            // cout << endl;
            for (int i = 0; i < m; i++) {
                if (curr[i] < n-1) {
                    vector<int> nbr(curr);
                    nbr[i]++;
                    string enc = encodeVector(nbr);
                    if (!searched.count(enc)) {
                        searched.insert(enc);
                        pq.push(make_pair(getSum(mat, nbr), nbr));
                    }
                }
            }
            id++;
        }
        return pq.top().first;
    }
    
    string encodeVector(vector<int>& choose) {
        string res;
        for (int i : choose) {
            res += to_string(i);
            res += ',';
        }
        return res;
    }
    

    int getSum(vector<vector<int>>& mat, vector<int>& idx) {
        int sum = 0;
        for (int i = 0; i < idx.size(); i++) {
            sum += mat[i][idx[i]];
        }
        return sum;
    }
};
