class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        int v5, v10, v20;
        v5 = v10 = v20 = 0;
        int n = bills.size();
        for( int i = 0 ; i < n ; ++i ){
            if( bills[i] == 5 ){
                v5++;
            }else if( bills[i] == 10 ){
                if( v5 == 0 ) return false;
                v5--;
                v10++;
            }else{
                v20++;
                if( v10 > 0 && v5 > 0 ){
                    v10--; v5--;
                }else if( v5 >= 3 ){
                    v5 -= 3;
                }else return false;
            }
        }
        return true;
    }
};
