class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        unordered_map<int,int> c;
        for (int bill : bills) {
            if (bill == 5) {
                c[5]++;
            } else if (bill == 10) {
                if (c[5] > 0) {
                    c[5]--;
                    c[10]++;
                } else {
                    return false;
                }
            } else {
                if (c[10] > 0 && c[5] > 0) {
                    c[10]--;
                    c[5]--;
                    c[20]++;
                } else if (c[5] >= 3) {
                    c[5] -= 3;
                    c[20]++;
                } else {
                    return false;
                }
            }
        }
        return true;
    }
};
