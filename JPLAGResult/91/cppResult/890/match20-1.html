class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        int five = 0, ten = 0, fif = 0;

        for (int v: bills) {
            if (v == 5) {
                ++ five;
            } else if (v == 10) {
                if (five == 0) return false;
                -- five;
                ++ ten;
            } else {
                if (fif >= 1) {
                    -- fif;

                } else if (ten >= 1 && five >= 1) {
                    -- ten;
                    -- five;
                } else if (five >= 3) {
                    five -= 3;
                } else {
                    return false;
                }
            } 
        }
        return true;
    }
};
