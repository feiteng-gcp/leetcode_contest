class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        int bill5 = 0, bill10 = 0;
        for (int bill : bills) {
            if (bill == 5) {
                bill5++;
            } else if (bill == 10) {
                if (bill5 == 0) {
                    return false;
                }
                bill10++;
                bill5--;
            } else if (bill == 20) {
                if (bill5 == 0 || (bill5 < 3 && bill10 == 0)) {
                    return false;
                }
                if (bill10 != 0) {
                    bill10--;
                    bill5--;
                } else {
                    bill5 -= 3;
                }
            }
        }
        return true;
    }
};
