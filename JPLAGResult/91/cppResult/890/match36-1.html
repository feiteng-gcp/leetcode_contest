const ll MOD=1e9+7;
class Solution {
public:
    bool lemonadeChange(vector<int>& b) {
        int x[3]{};

        for(int y:b){
            if(y==5){
                x[0]++;
            }else if(y==10){
                if(x[0]==0){
                    return 0;
                }
                x[0]--;
                x[1]++;

            }else{
                if(x[0]>=1 && x[1]>=1){
                    x[0]--;x[1]--;x[2]++;
                }else if(x[0]>=3){
                    x[0]-=3;x[2]++;
                }else{
                    return 0;
                }
            }
        }
        return 1;
    }
};
