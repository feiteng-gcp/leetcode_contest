class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        int cnt[3] = {0, 0};

        for (int i : bills) {
            if (i == 5) {
                cnt[0]++;
            } else if (i == 10) {
                if (cnt[0] == 0) return false;
                --cnt[0];
                ++cnt[1];
            } else {

                if (cnt[0] && cnt[1]) {
                    --cnt[0];
                    --cnt[1];
                } else if (cnt[0] >= 3) {
                    cnt[0] -= 3;
                } else return false;
            }
        }
        return true;
    }
};
