class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        vector<int> rec(2,0);
        for(auto b:bills){
            if(b==5){
                rec[0]++;
            }
            else if(b==10){
                if(rec[0]==0) return false;
                rec[0]--;
                rec[1]++;
            }
            else if(b==20){
                if(rec[1]>0){
                    if(rec[0]==0) return false;
                    rec[1]--;
                    rec[0]--;
                }
                else{
                    if(rec[0]<3) return false;
                    rec[0]-=3;
                }
            }
        }
        return true;
    }
};
