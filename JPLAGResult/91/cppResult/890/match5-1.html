class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        vector<int> num(2, 0);
        for (int i = 0; i < bills.size(); ++i) {
            if (bills[i] == 5) {
                ++num[0];
            } else if (bills[i] == 10) {
                if (num[0] == 0) {
                    return false;
                }
                ++num[1];
                --num[0];
            } else if (num[1] && num[0]) {

                --num[1];
                --num[0];
            } else if (num[0] >= 3) {
                num[0] -= 3;
            } else {
                return false;
            }
            
        }
        return true;
        
    }
};
