class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        int count5 = 0, count10 = 0, count20 = 0;
        
        for (int &b : bills) {
            if (b == 5) count5++;
            else if (b == 10) {
                if (!count5) return false;
                --count5;
                ++count10;
            }
            else {
                if (count10 > 0) {
                    --count10;
                    if (!count5 ) return false;
                    --count5;
                }
                else {
                    if (count5 < 3) return false;
                    count5 -=3;
                }
            }
        }
        return true;
    }
};
