class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        map<int,int> p;
        for(auto v:bills)
        {
            if(v==5) p[5]++;
            else
            if(v==10)
            {
                if(p[5]==0) return 0;
                p[5]--;
                p[10]++;
            }else
            {
                if(p[10]>0&&p[5]>0)
                {
                    p[10]--;
                    p[5]--;
                    continue;
                }
                if(p[5]>=3)
                {
                    p[5]-=3;
                    p[20]++;
                    continue;
                }
                return 0;
            }
        }
        return 1;
    }
};
