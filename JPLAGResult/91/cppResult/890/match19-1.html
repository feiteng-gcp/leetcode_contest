class Solution {
public:
    bool lemonadeChange(std::vector<int> &bills) {
        int d5count = 0;
        int d10count = 0;

        for (const auto &payment : bills)
        {
            if (payment == 5)
            {
                ++d5count;
            }
            else if (payment == 10)
            {
                // Must give back $5;
                if (d5count == 0) return false;
                --d5count;
                ++d10count;
            }
            else if (payment == 20)
            {
                // Either $5 + $10 or 3 x $5
                if (d10count > 0)
                {
                    if (d5count == 0) return false;

                    --d10count;
                    --d5count;
                }
                else if (d5count >= 3)
                {
                    d5count -= 3;
                }
                else
                {
                    return false;
                }
            }
        }
        return true;
    }
};
