class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        int n5=0,n10=0;

        for(int c:bills){
            if(c==5)n5++;
            else if(c==10){
                if(n5==0)return false;
                n5--;
                n10++;
            }
            else{
                if(n10>0){
                    if(n5==0)return false;

                    n10--;
                    n5--;
                }
                else{
                    if(n5<3)return false;
                    n5-=3;
                }
            }
        }
        return true;
    }
};
