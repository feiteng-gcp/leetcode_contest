class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {

        int a = 0, b = 0, c = 0;
        for(auto i: bills) {
            if (i == 5) a += 1;
            else if (i == 10) {
                if (a == 0) return false;
                a -= 1;
                b += 1;
            } else {
                if (b > 0) {
                    b-= 1;
                    if (a == 0) return false;

                    a -= 1;
                } else {
                    if (a < 3) return false;
                    a -= 3;
                }
            }
        }
        return true;
    }
};
