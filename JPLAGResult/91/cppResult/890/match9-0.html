class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        vector<int> dp(3, 0);
        for(int x: bills){
            if(x == 5)
                ++dp[0];
            else if(x == 10){
                if(dp[0] <= 0)
                    return false;
                --dp[0];
                ++dp[1];
            }
            else{
                ++dp[2];
                if(dp[1] > 0 && dp[0] > 0){
                    --dp[1];
                    --dp[0];
                }
                else if(dp[0] >= 3){
                    dp[0] -= 3;
                }
                else
                    return false;
            }
        }
        return true;
    }
};
