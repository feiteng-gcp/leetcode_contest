class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
      int ar[2] = {0,0};
      for (auto c : bills) {
        if (c == 5) {
          ar[0]++;
        } else if (c==10) {
          if (ar[0] == 0) return false;
          ar[0]--;
          ar[1]++;
        } else if (c==20) {
          if (ar[1] >= 1 && ar[0] >= 1) {
            ar[1]--;
            ar[0]--;
          } else if (ar[0] >= 3) {
            ar[0] -= 3;
          } else {
            return false;
          }
        }
      }
      return true;
    }
};
