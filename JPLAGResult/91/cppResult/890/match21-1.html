class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        int l = bills.size();
        int c5 = 0;
        int c0 = 0;
        for (int i = 0; i < l; ++i) {
            if (bills[i] == 5) c5 += 1;
            else if (bills[i] == 10) {
                c0 += 1;
                if (c5) {
                    c5 -= 1;
                } else return false;

            } else {
                if (c0 > 0 && c5 > 0) {
                    c0 -= 1;
                    c5 -= 1;
                } else if (c5 > 2) {
                    c5 -= 3;
                } else return false;
            }
        }
        
        return true;
    }
};
