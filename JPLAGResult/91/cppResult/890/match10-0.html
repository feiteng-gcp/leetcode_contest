class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        int five = 0, ten = 0;
        for (auto v : bills) {
            if (v == 5) {
                five++;
            } else if (v == 10) {
                if (five) {
                    ten++;
                    five--;
                } else {
                    return false;
                }
            } else {
                if (five && ten) {
                    five--;
                    ten--;
                } else if (five >= 3) {
                    five -= 3;
                } else {
                    return false;
                }
            }
        }
        return true;
    }
};
