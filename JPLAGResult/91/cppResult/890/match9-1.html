class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        vector<int> carsh(2, 0);
        for(int b:bills){
            if(b == 5) carsh[0]++;
            else if(b == 10)
                if(carsh[0]>0){
                    carsh[1]++;
                    carsh[0]--;
                }else
                    return false;
            else if(b == 20){

                if(carsh[1]>=1 && carsh[0]>=1){
                    carsh[1]--;
                    carsh[0]--;
                }else if(carsh[0]>=3){
                    carsh[0] -= 3;
                }
                else return false;
            }
        }
        return true;
    }
};
