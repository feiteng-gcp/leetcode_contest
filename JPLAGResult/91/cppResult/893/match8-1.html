/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
unordered_map<TreeNode*, unordered_set<TreeNode*> > g;
class Solution {
public:

    void dfs(TreeNode* root) {
        if (!root) return;
        if (root->left) {
            g[root->left].insert(root);
            g[root].insert(root->left);
            dfs(root->left);
        }
        if (root->right) {
            g[root].insert(root->right);
            g[root->right].insert(root);
            dfs(root->right);
        }
    }
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        if (!root || !target) return {};
        if (K == 0) return {target->val};
        g.clear();
        dfs(root);
        queue<TreeNode*> q; q.push(target); unordered_set<TreeNode*> s; s.insert(target);
        for(int i=0; i<K; ++i) {
            int sz = q.size();
            for(int j=0; j<sz; ++j) {
                auto node = q.front(); q.pop();
                for(auto nn: g[node]) {
                    if (s.find(nn) == s.end()) {
                        q.push(nn);
                        s.insert(nn);
                    }
                }
            }
        }
        vector<int> res;
        while(!q.empty()) {res.push_back(q.front()->val); q.pop();}
        return res;
    }
};
