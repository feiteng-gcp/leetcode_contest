/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> ans;
    void f1 (TreeNode* root, int K) {
        if (root == NULL)
            return;
        if (K == 0) {
            ans.push_back(root->val);
            return;
        }
        f1(root->left, K-1);
        f1(root->right, K-1);
    }
    int recur(TreeNode* root, TreeNode* target, int K) {
        if (root == target) {
            f1(root, K);
            return 1;
        }
        if (root->left == NULL && root->right == NULL)
            return -1;
        int left = -1, right = -1;
        if (root->left != NULL)
            left = recur(root->left, target, K);
        if (root->right != NULL)
            right = recur(root->right, target, K);
        if (left >= 0) {
            if (K - left < 0)
                return left + 1;
            if (K - left > 0)
                f1(root->right, K-left-1);
            else if (K-left == 0) {
                ans.push_back(root->val);
            }
            return left + 1;
        } else if (right >= 0) {
            if (K - right < 0)
                return right + 1;
            if (K - right > 0)
                f1(root->left, K-right-1);
            else if (K - right == 0) {
                ans.push_back(root->val);
            }
            return right + 1;
        }
        return -1;
    }
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        int temp = recur(root, target, K);
        return ans;
    }
};
