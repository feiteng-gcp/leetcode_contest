/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
#ifdef PYONPOI
 struct TreeNode {
     int val;
     TreeNode *left;
     TreeNode *right;
     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 };
#endif

class Solution {
public:
  vector<TreeNode*> par;
  void dfs(TreeNode* root, TreeNode* u, TreeNode* p, int K, VI& res){
	if(!u) return;
	if(K == 0){
	  res.PB(u->val);
	  return;
	}

	if(u->left && u->left != p)
	  dfs(root, u->left, u, K-1, res);
	if(u->right && u->right != p)
	  dfs(root, u->right, u, K-1, res);
	if(par[u->val] && par[u->val] != p)
	  dfs(root, par[u->val], u, K-1, res);
  }

  void init(TreeNode* u){
	if(u->left){
	  par[u->left->val] = u;
	  init(u->left);
	}
	if(u->right){
	  par[u->right->val] = u;
	  init(u->right);
	}
  }

  vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
	par.assign(600, nullptr);
	init(root);
	//REP(i,10) cout << i <<": "<<(par[i]?par[i]->val:-1)<<endl;
	
	VI ans;
	dfs(root, target, nullptr, K, ans);
	return ans;
  }
};
