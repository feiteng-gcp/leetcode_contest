/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    map<TreeNode*,TreeNode*> fa;
    void dfs(TreeNode* r,TreeNode* f)
    {
        if(r==NULL) return;
       
        if(r!=f)
        {
            fa[r]=f;
        }
        dfs(r->left,r);
        dfs(r->right,r);
    }
    vector<int> ans;
    void dfs2(TreeNode* r,TreeNode* f,int d)
    {
        if(r==NULL) return;
         //cout<<(r->val)<<"fa="<<(f->val)<<endl;
        if(d==0) 
        {
            ans.push_back(r->val);
            return;
        }
        if(r->left!=f) dfs2(r->left,r,d-1);
        if(r->right!=f) dfs2(r->right,r,d-1);
        
        if(fa.find(r)!=fa.end()&&fa[r]!=f) dfs2(fa[r],r,d-1);
    }
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        dfs(root,root);
        dfs2(target,target,K);
        return ans;
    }
};
