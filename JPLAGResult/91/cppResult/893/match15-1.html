/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

map<TreeNode*, TreeNode*> parent;
VI ans;
void dfs(TreeNode *x) {
    if (x->left) {
	parent[x->left] = x;
	dfs(x->left);
    }
    if (x->right) {
	parent[x->right] = x;
	dfs(x->right);
    }
}
void dfs2(TreeNode *x, TreeNode *p, int d) {
    if (d == 0) {
	ans.push_back(x->val);
	return;
    }
    if (x->left && p != x->left) dfs2(x->left, x, d-1);
    if (x->right && p != x->right) dfs2(x->right, x, d-1);
    auto it = parent.find(x);
    if (it != parent.end() && p != it->second) dfs2(it->second, x, d-1);
}
class Solution {
public:
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
	parent.clear();
	ans.clear();
	dfs(root);
	dfs2(target, target, K);
	return ans;
    }
};
