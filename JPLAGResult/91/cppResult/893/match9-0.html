/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void dfs(TreeNode *root, map<int, vector<int> > &a) {
        if (root == NULL) {
            return;
        }
        if (root->left != NULL) {
            a[root->val].push_back(root->left->val);
            a[root->left->val].push_back(root->val);
            dfs(root->left, a);
        }
        if (root->right != NULL) {
            a[root->val].push_back(root->right->val);
            a[root->right->val].push_back(root->val);
            dfs(root->right, a);
        }
    }
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        map<int, vector<int> > a;
        dfs(root, a);
        queue<pair<int, int> > que;
        que.push(make_pair(target->val, 0));
        map<int, vector<int> > res;
        set<int> st;
        st.insert(target->val);
        while (!que.empty()) {
            auto now = que.front();
            res[now.second].push_back(now.first);
            que.pop();
            for (int i = 0; i < a[now.first].size(); i++) {
                int nxt = a[now.first][i];
                if (st.find(nxt) == st.end()) {
                    st.insert(nxt);
                    que.push(make_pair(nxt, now.second + 1));
                }
            }
        }
        return res[K];
    }
};
