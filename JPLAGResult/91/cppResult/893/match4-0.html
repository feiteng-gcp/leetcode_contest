/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
const int INF=1000000000;
class Solution {
    unordered_map<int,vector<int>> e;
    void setedge(TreeNode* root){
        if(root->left){
            e[root->val].push_back(root->left->val);
            e[root->left->val].push_back(root->val);
            setedge(root->left);
        }
        if(root->right){
            e[root->val].push_back(root->right->val);
            e[root->right->val].push_back(root->val);
            setedge(root->right);
        }
    }
    vector<int> bfs(int init,int k){
        unordered_set<int> vis;
        queue<pair<int,int>> q;
        vector<int> ret;
        vis.insert(init);
        q.push({init,0});
        while(!q.empty()){
            int u=q.front().first,d=q.front().second;
            q.pop();
            if(d>k)break;
            if(d==k)ret.push_back(u);
            for(int v:e[u]){
                if(!vis.count(v)){
                    vis.insert(v);
                    q.push({v,d+1});
                }
            }
        }
        return ret;
    }
public:
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        setedge(root);
        return bfs(target->val,K);
    }
};
