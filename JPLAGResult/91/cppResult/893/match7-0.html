/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        
        set<TreeNode*> ss;
        stack<TreeNode*> pa;
        vector<int> res;
        
        if (searchNode(root, target , pa) == false) return {};
        
        cout<<pa.size()<<endl;
        
        int t = 0;
        while (!pa.empty()) {
            TreeNode* top = pa.top();
            pa.pop();
            
            if (K-t < 0) break;
            
            getK(top, K - t, res, ss);
            
            ss.insert(top);
            t ++;          
        }
        
        return res;
    }
    
    void getK(TreeNode* start, int K, vector<int>& res, set<TreeNode*>& ss) {
        
        if (ss.find(start) != ss.end()) {
            return;
        }
        
        if (start == NULL) return ;
        
        if (K == 0) {
            res.push_back(start->val);
            return ;
        }
        
        getK(start->left, K - 1, res, ss);
        getK(start->right, K - 1, res, ss);        
    }
    
    bool searchNode(TreeNode* root, TreeNode* target, stack<TreeNode*>& parents) {
        
        if (root == NULL) return false;
        
        parents.push(root);
        
        if (root == target) {
            return true;
        }
        
        bool f = searchNode(root->left, target, parents);
        
        if (f) return true;
        
        f = searchNode(root->right, target, parents);
        
        if (!f) parents.pop();
 
        return f;
    }
    
};
