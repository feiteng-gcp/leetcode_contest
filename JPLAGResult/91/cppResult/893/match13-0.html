/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    unordered_map<int, unordered_set<int>> g;
    unordered_set<int> visit;

    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        if(!root || !target || K<0)
        {
            return {};
        }

        if(K == 0)
        {
            return {target->val};
        }

        g = unordered_map<int, unordered_set<int>>();
        visit = unordered_set<int>();

        genG(root, NULL);

        queue<int> q;
        q.push(target->val);
        visit.insert(target->val);

        while(q.size()>0 && K>0)
        {
            int size = q.size();

            while(size--)
            {
                auto cur = q.front(); q.pop();

                for(auto next : g[cur])
                {
                    if(visit.find(next) != visit.end())
                    {
                        continue;
                    }

                    q.push(next);
                    visit.insert(next);
                }
                
            }

            K--;

            if(K==0)
            {
                vector<int> result;

                while(q.size() > 0)
                {
                    result.push_back(q.front());
                    q.pop();
                }

                return  result;
            }
        }

        return {};
    }

    void genG(TreeNode* root, TreeNode * pre)
    {
        if(!root)
        {
            return;
        }

        if(pre)
        {
            g[pre->val].insert(root->val);
            g[root->val].insert(pre->val);
        }

        genG(root->left, root);
        genG(root->right, root);
    }
};
