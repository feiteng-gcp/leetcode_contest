/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

vector<int> X;

void check1(TreeNode *root,int k) {
    if(root==NULL) return;
    if(k==0) {
        X.push_back(root->val); return;
    }
    if(root->left!=NULL) check1(root->left,k-1);
    if(root->right!=NULL) check1(root->right,k-1);
} 

int check(TreeNode *root,TreeNode *x,int k) {
    if(root==NULL) return -1;
    if(root==x)  {
        check1(x,k);  return k-1;
    }
    int l,r;  l=check(root->left,x,k);  r=check(root->right,x,k);
    if(l==0||r==0) {
        X.push_back(root->val);  return -1;
    }
    if(l>0) {
        check1(root->right,l-1);
    }
    else check1(root->left,r-1);
    return max(l,r)-1;
}

class Solution {
public:
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        X.clear();
        if(root==NULL) return X;
        check(root,target,K);
        return X;
    }
};
