class Solution
{
public:
    void dfs_fill(TreeNode *node, std::map<int, std::set<int>> &graph)
    {
        if (node == nullptr)
        {
            return;
        }

        if (node->left != nullptr)
        {
            graph[node->val].insert(node->left->val);
            graph[node->left->val].insert(node->val);
            dfs_fill(node->left, graph);
        }

        if (node->right != nullptr)
        {
            graph[node->val].insert(node->right->val);
            graph[node->right->val].insert(node->val);
            dfs_fill(node->right, graph);
        }
    }

    std::vector<int> distanceK(TreeNode *root, TreeNode *target, int K)
    {
        if (root == nullptr || target == nullptr)
        {
            return {};
        }

        std::map<int, std::set<int>> graph;
        dfs_fill(root, graph);
        std::set<int> result;

        // (node, level)
        std::queue<std::pair<int, int>> queue;
        queue.push(std::make_pair(target->val, 0));
        std::set<int> visited;
        visited.insert(target->val);

        while (!queue.empty())
        {
            int node;
            int level;
            std::tie(node, level) = queue.front();
            queue.pop();

            if (level == K)
            {
                result.insert(node);
            }

            for (auto neigh : graph[node])
            {
                if (visited.count(neigh) == 0)
                {
                    queue.push(std::make_pair(neigh, level + 1));
                    visited.insert(neigh);
                }
            }
        }

        return {result.begin(), result.end()};
    }
};
