  // struct TreeNode {
  //     int val;
  //     TreeNode *left;
  //     TreeNode *right;
  //     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
  // };
class Solution {
public:
    vector<int> res;
    vector<TreeNode*> st;
    void dn(TreeNode* x, TreeNode* prv,int K){

        if(x==NULL || x==prv || K<0) return;
        if(K==0){
            res.pb(x->val);
            return;
        }
        dn(x->left,prv,K-1);
        dn(x->right,prv,K-1);
    }

    bool dfs(TreeNode* x,TreeNode* tar){
        if(x==NULL) return 0;
        st.pb(x);
        if(x==tar) return 1;
        if(dfs(x->left,tar)) return 1;
        if(dfs(x->right,tar)) return 1;
        st.pop_back();
        return 0;
    }

    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        dfs(root,target);
        reverse(st.begin(), st.end());
        int k=K;
        TreeNode* prv=NULL;
        for(auto px:st){
            dn(px,prv,k);
            prv=px;
            k--;
        }
        return res;
    }
};
