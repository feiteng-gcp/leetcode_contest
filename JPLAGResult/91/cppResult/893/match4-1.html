/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
    unordered_map<int, vector<int>> mp;
    void traverse(TreeNode *x) {
        if (x->left) {
            mp[x->val].push_back(x->left->val);
            mp[x->left->val].push_back(x->val);
            traverse(x->left);
        }
        if (x->right) {
            mp[x->val].push_back(x->right->val);
            mp[x->right->val].push_back(x->val);
            traverse(x->right);
        }
    }
public:
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        traverse(root);
        unordered_map<int, int> visited;
        queue<int> q;
        q.push(target->val);
        visited[target->val] = 0;
        vector<int> result;
        while (!q.empty()) {
            int x = q.front(); q.pop();
            vector<int> &v = mp[x]; int step = visited[x];
            if (step == K)
                result.push_back(x);
            else if (step == K + 1)
                return result;
            for (int y : v) {
                if (visited.find(y) == visited.end()) {
                    visited[y] = step + 1;
                    q.push(y);
                }
            }
        }
        return result;
    }
};
