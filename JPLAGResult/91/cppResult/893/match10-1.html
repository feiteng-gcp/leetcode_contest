/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {

    void collectNode(TreeNode* root, unordered_map<TreeNode*, vector<TreeNode*>>& memo) {
        if (root == nullptr) {
            return;
        }
        
        collectNode(root->left, memo);
        if (root->left) {
            memo[root].push_back(root->left);
            memo[root->left].push_back(root);
        }
        if (root->right) {
            memo[root].push_back(root->right);
            memo[root->right].push_back(root);
        }
        collectNode(root->right, memo);
    }
    void dfs(TreeNode* node, unordered_map<TreeNode*, vector<TreeNode*>>& memo,
             unordered_map<TreeNode*, bool>& used,
             int K, vector<int>& res) {
        used[node] = true;
        if (K == 0) {
            res.push_back(node->val);
            return;
        }
        const auto& neighbors = memo[node];
        
        for (auto n:neighbors) {
            if (!used[n]) {
                dfs(n, memo, used, K-1, res);
            }
        }
    }
public:
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        vector<int> res;
        
        unordered_map<TreeNode*, vector<TreeNode*>> imap;
        collectNode(root, imap);
        
        unordered_map<TreeNode*, bool> used;
        for (const auto& item: imap) {
            used[item.first] = false;
        }
        dfs(target, imap, used, K, res);
        return res;
    }
};
