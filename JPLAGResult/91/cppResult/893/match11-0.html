class Solution {
	public:
		set<TreeNode *> seen;
		map<TreeNode *, TreeNode *> parents;
		vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
			calcParents(root);
			return dK(root, target, K);
		}

		void calcParents(TreeNode* root) {
			if (root->left) {
				parents[root->left] = root;
				calcParents(root->left);
			}

			if (root->right) {
				parents[root->right] = root;
				calcParents(root->right);
			}
		}

		vector<int> dK(TreeNode* root, TreeNode* target, int K) {						
			vector<int> res;
			if (seen.find(target) != seen.end()) {
				return res;
			}

			seen.insert(target);

			if (K == 0) {
				res.push_back(target->val);
				return res;
			}

			if (target->left) {
				auto r = dK(root, target->left, K - 1);
				res.insert(res.end(), r.begin(), r.end());
			}

			if (target->right) {
				auto r = dK(root, target->right, K - 1);
				res.insert(res.end(), r.begin(), r.end());
			}

			if (parents.find(target) != parents.end()) {
				auto r = dK(root, parents[target], K - 1);
				res.insert(res.end(), r.begin(), r.end());
			}

			return res;
		}
};
