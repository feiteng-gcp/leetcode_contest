/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    #define MAXN 2000
    vector<int> G[MAXN];
    bool visited[MAXN];
    vector<int> ans;
    int maximaPr;
    
    void go(TreeNode* root){
        if(root == NULL) return;
        if(root->left != NULL){
            G[root->val].push_back(root->left->val);
            G[root->left->val].push_back(root->val);
        }
        if(root->right != NULL){
            G[root->right->val].push_back(root->val);
            G[root->val].push_back(root->right->val);
        }
        go(root->left);
        go(root->right);
    }
    
    void DFS(int inicio,int lvl){
        if(visited[inicio]) return;
        if(lvl > maximaPr) return;
        if(lvl == maximaPr){
            // introducir ans
            ans.push_back(inicio);
        }
        visited[inicio] = true;
        for(int i=0;i<G[inicio].size();i++){
            DFS(G[inicio][i],lvl+1);
        }
    }
    
    vector<int> distanceK(TreeNode* root, TreeNode* target, int K) {
        maximaPr = K;
        go(root);
        // recorrer el grafo.
        memset(visited,0,sizeof(visited));
        DFS(target->val,0);
        return ans;
    }
};
