class Solution {
public:
    int a[100][100];
    int matrixScore(vector<vector<int>>& A) {
        int n=A.size();
        if (!n) return 0;

        int m=A[0].size();
        for(int i=0; i<n; i++) {
            for(int j=0; j<m; j++) {
                a[i][j]=A[i][j];
            }
        }

        for(int i=0; i<n; i++) {
            if (a[i][0]==0) {
                for(int j=0; j<m; j++) {
                    a[i][j]=1-a[i][j];
                }

            }
        }
        for(int j=0; j<m; j++) {
            int cur=0;
            for(int i=0; i<n; i++) {
                cur+=a[i][j];
            }
            if (cur<n-cur) {
                for(int i=0; i<n; i++) {
                    a[i][j]=1-a[i][j];
                }
            }
        }
        int sum=0;
        for(int i=0; i<n; i++) {
            int row=0;
            for(int j=0; j<m; j++) {
               row=row*2+a[i][j]; 
            }
            sum=sum+row;
        }
        return sum;    
    }
};
