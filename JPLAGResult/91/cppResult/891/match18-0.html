class Solution {
public:
    int matrixScore(vector<vector<int>>& a) {
        if (a.empty() || a[0].empty())
            return 0;
        int n = a.size();
        int m = a[0].size();
        for (auto & row : a) {
            if (row[0] == 0) {
                for (auto & x : row) x = 1 - x;
            }
        }
        int ans = 0;
        for (int i = 0; i < m; ++i) {
            int c1 = 0;
            for (int j = 0; j < n; ++j) {
                c1 += a[j][i];
            }
            c1 = max(c1, n - c1);
            ans += c1 * (1 << (m - i - 1));
        }
        return ans;
    }
};
