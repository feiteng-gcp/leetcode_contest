class Solution {
public:
    int matrixScore(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        for (int i = 0; i < n; i++) {
            if (a[i][0] == 0) {
                for (int j = 0; j < m; j++) {
                    a[i][j] = 1 - a[i][j];
                }
            }
        }
        int res = (1 << (m - 1)) * n;
        for (int i = 1; i < m; i++) {
            int sum = 0;
            for (int j = 0; j < n; j++) {
                sum += a[j][i];
            }
            sum = max(sum, n - sum);
            res += (1 << (m - i - 1)) * sum;
        }
        return res;
    }
};
