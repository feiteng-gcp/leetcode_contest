class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int ret=0;
        while(true){
            bool change=false;
            for(int i=0; i<A.size(); i++){
                if(A[i][0]==0){
                    for(int j=0; j<A[i].size(); j++){
                        A[i][j]=(A[i][j]==1 ? 0 : 1);
                    }
                    change=true;
                }
            }
            for(int j=0; j<A[0].size(); j++){
                int sum=0;
                for(int i=0; i<A.size(); i++){
                    sum+=A[i][j];
                }
                if(sum*2<A.size()){
                    for(int i=0; i<A.size(); i++){
                        A[i][j]=(A[i][j]==1 ? 0 : 1);
                    }
                    change=true;
                }
            }
            if(!change){
                break;
            }
        }
        int base=1;
        for(int j=A[0].size()-1; j>=0; j--){
            for(int i=0; i<A.size(); i++){
                ret+=A[i][j]*base;
            }
            base=base<<1;
        }
        return ret;
    }
};
