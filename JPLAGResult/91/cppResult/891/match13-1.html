class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int m = A.size(), n = A[0].size();

        for (int i = 0; i < m; i++) {
            if (A[i][0] == 0) rowToggle(A, i);
        }
        for (int j = 0; j < n; j++) {
            int cnt = 0;
            for (int i = 0; i < m; i++) {

                cnt += A[i][j];
            }
            if (cnt < m - cnt) colToggle(A, j);
        }
        int res = 0;
        for (int i = 0; i < m; i++) {
            int temp = 0;
            for (int j = n - 1; j >= 0; j--) {

                temp += A[i][j] << (n - j - 1);
            }
            res += temp;
        }
        return res;
    }
    
    void rowToggle(vector<vector<int>>& A, int row) {
        for (int col = 0; col < A[0].size(); col++) {
            A[row][col] ^= 1;
        }
    }
    
    void colToggle(vector<vector<int>>& A, int col) {
        for (int row = 0; row < A.size(); row++) {
            A[row][col] ^= 1;
        }
    }
};
