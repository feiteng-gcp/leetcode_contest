class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int row=A.size();
        int col=A[0].size();
         for(int i=0;i<row;i++){
            if(A[i][0]==1)
                continue;
            else{
                for(int j=0;j<col;j++){
                    if(A[i][j]==0)
                        A[i][j]=1;
                    else
                        A[i][j]=0;
                }
            }
        }
        for(int i=0;i<col;i++){
            int sum=0;
            for(int j=0;j<row;j++){
                if(A[j][i]==0)
                    sum++;
            }
            if(sum>row-sum){
                for(int j=0;j<row;j++){
                    if(A[j][i]==0)
                        A[j][i]=1;
                    else
                        A[j][i]=0;
                }
            }
        }
       /* for(int i=0;i<row;i++){
            for(int j=0;j<col;j++)
                cout << A[i][j]<<' ';
            cout <<endl;
        }*/
        long long ans=0;
        for(int i=0;i<row;i++){
            int mul=1;
            for(int j=col-1;j>=0;j--){
                ans+=mul*A[i][j];
                mul*=2;
            }
        }
        return ans;
    }
};
