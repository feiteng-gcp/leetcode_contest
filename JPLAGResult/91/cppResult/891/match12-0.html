class Solution {
public:
    int matrixScore(vector<vector<int>>& a) {
        int n=a.size(),m=a[0].size(),mask=(1<<m)-1,head=1<<(m-1);
        vector<int> b(n);
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
                b[i]+=a[i][j]<<(m-j-1);
        int ans=0;
        for(int op=0;op<(1<<m);op++){
            int sm=0;
            for(int i=0;i<n;i++){
                int c=b[i]^op;
                if(!(c&head))c^=mask;
                sm+=c;
            }
            ans=max(ans,sm);
        }
        return ans;
    }
};
