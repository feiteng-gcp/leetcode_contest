class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int m = A.size();
        if (m == 0) return 0;
        int n = A[0].size();
        
        for (int i = 0; i < m ; i++) {
            if (A[i][0] != 1) {
                for (int j = 0; j < n ; j++) {
                    if (A[i][j] == 0) A[i][j] = 1;
                    else A[i][j] = 0;
                }
            }
        }
        
        int res = m * (1 << (n-1));
        
        for (int i = 1; i < n; i++) {
            int cnt = 0;
            for (int j = 0; j < m ; j++) {
                if (A[j][i] == 1) cnt++;
            }
            res += max(cnt, (m-cnt)) * (1 << (n - i - 1));
        }
        
        return res;
    }
};
