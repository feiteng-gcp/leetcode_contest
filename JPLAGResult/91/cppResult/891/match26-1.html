class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int n=A.size(),m=A[0].size();
	int ans=0;
	int yi=0;
	for(int i=0;i<n;i++)
	{
		if(!A[i][0])yi=yi^(1<<i);
	}
	//for(int i=0;i<(1<<n);i++)
	//{
		//printf("i:%d\n",i);
		int now=0;

		for(int j=0;j<m;j++)
		{
			int cnt=0;
			for(int k=0;k<n;k++)
			{
				if(A[k][j]^(((1<<k)&yi)!=0))cnt++;
			}
			//printf("j:%d cnt:%d\n",j,cnt);
			cnt=max(cnt,n-cnt);
			now=now+cnt*(1<<(m-1-j));
		}
		//printf("now:%d\n",now);
		ans=max(ans,now);
	//}
	yi=yi^((1<<n)-1);
	//for(int i=0;i<(1<<n);i++)
	//{
		//printf("i:%d\n",i);
		now=0;

		for(int j=0;j<m;j++)
		{
			int cnt=0;
			for(int k=0;k<n;k++)
			{
				if(A[k][j]^(((1<<k)&yi)!=0))cnt++;
			}
			//printf("j:%d cnt:%d\n",j,cnt);
			cnt=max(cnt,n-cnt);
			now=now+cnt*(1<<(m-1-j));
		}
		//printf("now:%d\n",now);
		ans=max(ans,now);
	//}
	return ans;
    }
};
