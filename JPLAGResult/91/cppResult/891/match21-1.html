class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int R = A.size();
        int C = A[0].size();

        vector<int> B(R);
        for (int r = 0; r < R; ++r) {
            for (int c = 0; c < C; ++c)
                if (A[r][c])
                    B[r] |= 1<<(C-1-c);
        }
        
        int best = 0;

        for (int mask = (1<<C)-1; mask >= 0; --mask) {
            int sum = 0;
            for (int r = 0; r < R; ++r) {
                int x = B[r] ^ mask;
                if (x > (((1<<C)-1) ^ x))
                    sum += x;
                else
                    sum += (((1<<C)-1) ^ x);
            }
            best = max(best, sum);
        }
        return best;
    }
};
