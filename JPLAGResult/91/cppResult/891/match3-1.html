class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int m = A.size(), n = A[0].size();
        for (int i = 0; i < m; ++i) {
            if (A[i][0] == 0) {
                for (int j = 0; j < n; ++j) {
                    A[i][j] = (A[i][j]) ? 0 : 1;
                }
            }
        }
        
        for (int j = 1; j < n; ++j) {
            int zerocount = 0;
            for (int i = 0; i < m; ++i) {
                if (!A[i][j]) zerocount++;
            }
            if (zerocount >= m / 2 + 1) {
                for (int i = 0; i < m; ++i) {
                    A[i][j] = (A[i][j]) ? 0 : 1;

                }
            }
        }
        
        int res = 0;
        for (int i = 0; i < m; ++i) {
            int levelres = 0;
            for (int j = 0; j < n; ++j) {
                // cout << A[i][j] << endl;
                if (A[i][j]) {
                    levelres += (int)pow(2, n - j - 1);
                }
            }
            cout << levelres << endl;
            res += levelres;
        }
        
        return res;
    }
};
