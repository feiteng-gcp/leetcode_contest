class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int m = A.size(), n = A[0].size();
        // For rows
        for (int i = 0; i < m; i++){
            if (A[i][0] == 0){
                for (int j = 0; j < n; j++){
                    A[i][j] = 1 - A[i][j];
                }
            }
        }
        
        // For cols
        for (int j = 0; j < n; j++){
            int cnt1 = 0, cnt0 = 0;
            for (int i = 0; i < m; i++)
                if (A[i][j] == 0) cnt0++;
                else cnt1++;
            if (cnt0 > cnt1){
                for (int i = 0; i < m; i++)

                    A[i][j] = 1 - A[i][j];
            }
        }
        
        // sum up
        int res = 0;
        for (int i = 0; i < m; i++){
            int row_val = 0;
            for (int j = 0; j < n; j++)
                row_val = 2 * row_val + A[i][j];
            res += row_val;
        }
        return res;
    }
};
