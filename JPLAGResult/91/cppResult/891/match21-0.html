class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
int L = A[0].size();
int res = 0;
vector<int> B(A.size());
for(int i = 0; i < A.size(); i++) {
    for(int j = 0; j < L; j++) {
        if (A[i][j]) {
            B[i] |= (1 << (L-1-j));
        }
    }
}
int mask = 0;
for(int i = 0; i < L; i++) {
    mask |= (1 << i);
}
for(int s = 0; s < (1 << L); s++) {
    int local = 0;
    for(int i = 0; i < B.size(); i++) {
        int X = B[i] ^ s;
        int Y = (~X) & mask;
        local += max(X, Y);
    }
    res = max(res, local);
}
return res;


        
    }
};
