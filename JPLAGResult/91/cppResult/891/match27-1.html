class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        for(int i=0;i<A.size();i++){
            if(A[i][0]!=1){
                for(int j=0;j<A[i].size();j++){

                    A[i][j]= 1-A[i][j];
                }
            }
        }
        
        for(int i=0;i<A[0].size();i++){
            int nroOnes = 0;
            for(int j=0;j<A.size();j++){
                nroOnes += A[j][i];
            }
            if((A.size()-nroOnes) > nroOnes){
                // se invierte el valor
                for(int j=0;j<A.size();j++){
                    A[j][i] = 1 - A[j][i];

                }   
            }
        }
        
        int ans = 0;
        for(int i=0;i<A.size();i++){
            int nro = 0;
            int potencia = 1;
            int pos=0;
            for(int j=A[i].size()-1; j >= 0; j--){
                if(A[i][j]==1){
                    nro += (1<<(pos));
                }
                pos++;
            }
            ans+=nro;
        }
        return ans;
    }
};
