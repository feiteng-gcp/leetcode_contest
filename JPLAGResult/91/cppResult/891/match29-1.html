class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        // convert first row to 1
        for (int i = 0; i < A.size(); i++) {

            if (A[i][0] == 0) { // toggling
                for (int j = 0; j < A[i].size(); j++) {
                    if (A[i][j])
                        A[i][j] = 0;
                    else
                        A[i][j] = 1;
                }
            }
        }
        
        for (int i = 0; i < A[0].size(); i++) {
            int count = 0;
            for (int j = 0; j < A.size(); j++) {
                if (A[j][i]) {
                    count++;
                }
            }
            if (count < (A.size() + 1) / 2) {
                for (int j = 0; j < A.size(); j++) {
                    if (A[j][i]) {
                        A[j][i] = 0;
                    } else {
                        A[j][i] = 1;
                    }
                } 
            }
        }
               
        // binary sum

        int sum = 0;
        for (int i = 0; i < A.size(); i++) {
            int index = 1;
            for (int j = A[i].size() - 1; j >= 0; j--) {
                sum += A[i][j] * index;
                index *= 2;
            }
        }
        return sum;
    }
};
