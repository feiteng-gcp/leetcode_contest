class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {

        int ans = 0;
        for(auto& row: A){
            if(row[0] == 0){
                for(auto& c: row) c = 1 - c;
            }
        }
        
        int h = A.size(), w = A[0].size();

        ans += pow(2, w - 1) * h;
        for(int i=1;i<w;i++){
            int cnt = 0;
            for(int j=0;j<h;j++){
                cnt += A[j][i];
            }
            ans += pow(2, w - i - 1) * max(cnt, h - cnt);
        }
        
        return ans;
    }
};
