class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int n = A.size(), m = A[0].size();
        vector<int> row(n);
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                row[i] |= A[i][j] << (m - 1 - j);
            }
        }
        int tot = (1 << m) - 1;

        int ret = 0;
        for (int k = 0; k < (1 << m); ++k) {
            int sum = 0;
            for (int i = 0; i < n; ++i) {
                int cur = row[i] ^ k;
                sum += max(cur, tot - cur);
            }
            ret = max(ret, sum);
        }
        return ret;
    }
};
