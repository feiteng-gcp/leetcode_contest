class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int m = (int)A.size(), n = (int)A[0].size();
    for(int i = 0; i < m; i++) if(A[i][0] == 0)
        for(int j = 0; j < n; j++) A[i][j] = 1 - A[i][j];
    int w = 1, ans = 0;
    for(int j = n - 1; j >= 0; j--)
    {
        int cnt1 = 0;
        for(int i = 0; i < m; i++) cnt1 += A[i][j];
        ans += w * max(cnt1, m - cnt1);
        w <<= 1;
    }
    return ans;
    }
};
