class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int ylen = A.size(), xlen = A[0].size();
        for (int y = 0; y < ylen; ++y) {
            if (A[y][0] == 0) {
                for (int x = 0; x < xlen; ++x) {
                    A[y][x] = 1 - A[y][x];
                }
            }
        }
        int ans = 0, factor = 1;
        for (int x = xlen-1; x >= 0; --x) {
            int num_of_1 = 0;
            for (int y = 0; y < ylen; ++y) {
                num_of_1 += A[y][x];
            }
            ans += factor * max(num_of_1, ylen - num_of_1);
            factor *= 2;
        }
        return ans;
    }
};
