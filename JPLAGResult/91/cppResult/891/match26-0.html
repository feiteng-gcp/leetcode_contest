class Solution {
public:
    int matrixScore(vector<vector<int>>& A) {
        int m = A.size(), n = A[0].size();
        int res = 0;
        vector<int> vals(m);
        for(int i=0; i<m; ++i) {
            int val = 0;
            for(int j=0; j<n; ++j) {
                val += A[i][j] ? (1 << (n-1-j)) : 0;
            }
            vals[i] = val;
        }
        for(int mask = 0; mask < (1 << n); ++mask) {
            int sum = 0;
            for(int i=0; i<m; ++i) {
                int val = vals[i];
                int o = val & mask;
                int t = val | (mask);
               // cout << val <<" "<< (t) << endl;
                int cur = o + ((~t) & ((1 << n) - 1));
                sum += max(cur, (1 << n) - 1 - cur);
            }
            res = max(sum, res);
        }
        return res;
    }
};
