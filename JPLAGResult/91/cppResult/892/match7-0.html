class Solution {
public:
    #define ll long long
    bool check(vector<ll> &a, int K, int l) {
        multiset<int> s;
        // x - a[i] >= K
        for(int i= (int) a.size()-1; i>=0; --i) {
            if (s.size() > l) s.erase(s.find(a[i + l+1]));
            if (s.lower_bound(a[i] + K) != s.end()) {
            //    cout << 111 << endl;
                return true;
            }
            s.insert(a[i]);
         //   cout << s.size() << " " << a[i] << " " << i << l<< endl;
        }
        return false;
    }
    int shortestSubarray(vector<int>& A, int K) {
        if (A.empty()) return -1;
        vector<ll> a = {0};
        ll val = 0;
        for(auto i: A) {
            val += i;
            a.push_back(val);
        }
        int xx = *max_element(A.begin(), A.end());
        if (xx >= K) return 1;
        if (!check(a, K, a.size())) return -1;
        int L = 1, R = a.size();
        while(L < R) {
            int mid = (L + R) / 2;
            if (check(a, K, mid)) {
                R = mid;
            } else L = mid + 1;
        }
        return L;
    }// 0 2 1 3
};
