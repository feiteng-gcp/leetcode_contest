class Solution {
private:
    int n, a[50005];
    long long b[50005], k;
    bool judge(int lim) {
        deque<int> q;
        q.push_back(0);
        for (int i = 1; i <= n; ++i) {
            while (!q.empty() && q.front() < i - lim)
                q.pop_front();
            if (!q.empty() && b[i] - b[q.front()] >= k)
                return true;
            while (!q.empty() && b[q.back()] >= b[i])
                q.pop_back();
            q.push_back(i);
        }
        return false;
    }
public:
    int shortestSubarray(vector<int>& A, int K) {
        n = A.size();
        k = K;
        for (int i = 0; i < n; ++i)
            a[i + 1] = A[i];
        for (int i = 1; i <= n; ++i)
            b[i] = b[i - 1] + a[i];
        int l = 1, r = n, mid, ans = -1;
        while (l <= r) {
            mid = (l + r) / 2;
            if (judge(mid))
                ans = mid, r = mid - 1;
            else
                l = mid + 1;
        }
        return ans;
    }
};
