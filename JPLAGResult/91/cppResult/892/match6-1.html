class Solution {
public:
    int shortestSubarray(vector<int>& a, int k) {
        int n = a.size();
        vector<long long> s(n + 1);
        for (int i = 0; i < n; ++i) s[i + 1] = s[i] + a[i];
        vector<long long> ss(s.begin(), s.end());
        sort(ss.begin(), ss.end());
        int ns = unique(ss.begin(), ss.end()) - ss.begin();
        // for (long long x : s) cout << x << " "; cout << endl;
        // for (int i = 0; i < ns; ++i) cout << ss[i] << " "; cout << endl;
        vector<int> best(ns + 10, -1);
        {
            int i = lower_bound(ss.begin(), ss.begin() + ns, 0) - ss.begin();
            for (int u = i + 1; u <= ns; u += u & -u) best[u] = max(best[u], 0);
        }
        int ans = 1000010000;
        for (int i = 1; i <= n; ++i) {
            int j = upper_bound(ss.begin(), ss.begin() + ns, s[i] - k) - ss.begin();
            // cout << i << " " << s[i] << " " << j << endl;
            if (j > 0) {
                int t = -1;
                for (int u = j; u > 0; u -= u & -u) t = max(t, best[u]);
                // cout << s[i] << " " << j << " " << t << endl;
                if (t >= 0) {
                    ans = min(ans, i - t);
                }
            }
            {
                int x = lower_bound(ss.begin(), ss.begin() + ns, s[i]) - ss.begin();
                // cout << "updating " << x + 1 << " with " << i << endl;
                for (int u = x + 1; u <= ns; u += u & -u) best[u] = max(best[u], i);
            }
        }
        return (ans > n ? -1 : ans);
    }
};
