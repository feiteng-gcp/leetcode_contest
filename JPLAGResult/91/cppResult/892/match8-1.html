typedef long long int64;
typedef pair<int64, int> ii;
const int INF = 1 << 30;

const int N = 5e4 + 10;
int64 sum[N];
int64 d[20][N];
int lg2[N];

int64 get_max(int x, int y) {
    int k = lg2[y - x + 1];
    return max(d[k][x], d[k][y - (1 << k) + 1]);
}
class Solution {
public:
    int shortestSubarray(vector<int>& A, int K) {
        int n = A.size();
        vector<int64> sum(n + 1);
        lg2[1] = 0;
        for (int i = 2; i < N; ++i) lg2[i] = lg2[i / 2] + 1;
        for (int i = 0; i < n; ++i) sum[i + 1] = sum[i] + A[i];
        for (int i = 0; i <= n; ++i) d[0][i] = sum[i];
        for (int k = 1; k < 20; ++k) {
            for (int i = 0; i + (1 << k) <= n + 1; ++i) {
                d[k][i] = max(d[k - 1][i], d[k - 1][i + (1 << (k - 1))]);
            }
        }
        int ret = INF;
        for (int i = 0; i < n; ++i) {
            if (A[i] <= 0) continue;
            int low = i + 1, high = n + 1;
            while (low != high) {
                int mid = (low + high) / 2;

                int64 cur = get_max(i + 1, mid) - sum[i];
                if (cur < K) {
                    low = mid + 1;
                } else {
                    high = mid;
                }
            }
            if (high <= n) {
                ret = min(ret, high - i);
            }
        }
        if (ret == INF) ret = -1;
        return ret;
    }
};
