const int inf=0x3f3f3f3f;
class Solution {
public:
    int shortestSubarray(vector<int>& A, int K) {
        #define ll long long
        int n=A.size();
        vector<ll> sum(n+1,0);
        for(int i=1;i<=n;i++) sum[i]=sum[i-1]+A[i-1];
        
        vector<pair<ll,int> > ar;
        ar.push_back({sum[0],0});
        int ans=inf;
        for(int i=1;i<=n;i++)
        {
            int s=0,t=(int)ar.size()-1;
            while(s<=t)
            {
                int mid=(s+t)>>1;
                if(ar[mid].first<=sum[i]-K)
                    s=mid+1;
                else
                    t=mid-1;
            }
            int p=t;
            if(p>=0)
            {
                ans=min(ans,i-ar[p].second);
            }
            while(ar.size() &&ar.back().first>=sum[i]) ar.pop_back();
            ar.push_back({sum[i],i});
        }
        return ans==inf?-1:ans;
    }
};
