typedef long long llong;
typedef vector<int> VI;
typedef vector<VI> VVI;

const int INF = 1000000010;

class BinaryIndexedTree {
   int N;
   VI tree;
public:
   BinaryIndexedTree(int _N) : N(_N), tree(_N+1, -INF) {}

   int query(int idx) {
      int res = -INF;
      for (int i = idx; i > 0; i -= (i & -i))
         res = max(res, tree[i]);
      return res;
   }

   void update(int idx, int val = 1) {
      for (int i = idx; i <= N; i += (i & -i))
         tree[i] = max(tree[i], val);
   }
};

class Solution {

public:
   int shortestSubarray(vector<int>& A, int K) {
      int N = A.size();
      vector<llong> PS(N+1);
      for (int i = 1; i <= N; ++i)
         PS[i] = PS[i-1] + A[i-1];

      vector<llong> SPS(PS);
      SPS.push_back(INF * -1LL * INF);
      sort(SPS.begin(), SPS.end());
      SPS.erase( unique(SPS.begin(), SPS.end()), SPS.end() );

      // for (llong x : SPS) cerr << x << ' '; cerr << endl;

      int best = INF;
      BinaryIndexedTree bit(SPS.size()+1);

      for (int i = 0; i <= N; ++i) {
         int q = upper_bound(SPS.begin(), SPS.end(), PS[i] - K) - SPS.begin() -1;
         int r = bit.query(q);
         
      // cerr << "i: " << i << "  q: " << q << "  r: " << r << endl;
         
         best = min(best, i - r);

         int p = lower_bound(SPS.begin(), SPS.end(), PS[i]) - SPS.begin();
         bit.update(p, i);
      }
      return best >= INF ? -1 : best;
   }
};
