typedef long long ll;
typedef pair<int,int> pii;
template<typename T> using min_queue=priority_queue<T,vector<T>,greater<T>>;

const ll MOD=1e9+7;

class Solution {
public:
    int N;
    vector<ll> s;
    vector<pair<ll,int>> st;
    int shortestSubarray(vector<int>& a, int K) {
        N=a.size();
        if(N==0) return -1;

        s.resize(N+1);
        for(int i=1;i<=N;i++) s[i]=s[i-1]+a[i-1];

        int res=1000000;
        for(int i=N;i>=1;i--){
            while(!st.empty() && st.back().fst<=s[i]){
                st.pop_back();
            }
            st.push_back({s[i],i});


            int M=st.size()-1;

            int l=-1,r=M;
            // cout<<N<<' '<<i<<' '<<endl;
            // for(auto pr:st){
            //     cout<<pr.fst<<' '<<pr.snd<<endl;
            // }
            // cout<<endl;
            while(l<r){
                int mid=(l+r+1)/2;
                if(st[mid].fst>=s[i-1]+K){
                    l=mid;
                }else{
                    r=mid-1;
                }
            }
            if(l>=0)
                res=min(res,st[l].snd-i+1);
        }
        if(res>N) res=-1;
        return res;
        
    }
};
