	class Solution {
	    public int shortestSubarray(int[] a, int K) {
	        int n = a.length;
	        ans = Integer.MAX_VALUE;
	        dfs(0, n, a, K);
	        if(ans == Integer.MAX_VALUE)return -1;
	        return ans;
	    }
	    
	    int ans;
	    long O = 1L<<46;
	    
	    void dfs(int l, int r, int[] a, int K)
	    {
	    	if(r-l <= 0)return;
	    	int h = l+r>>1;
	    	// include h
	    	long s = 0;
	    	long[] b = new long[h-l+1];
	    	int p = 0;
	    	for(int i = h, j = 1;i >= l;i--,j++){
	    		s += a[i];
	    		b[p++] = s+O<<16|j;
	    	}
	    	assert p == b.length;
	    	Arrays.sort(b);
	    	
	    	long[] c = new long[r-h];
	    	int q = 0;
	    	s = 0;
	    	for(int i = h+1, j = 0;i <= r;i++,j++){
	    		c[q++] = s+O<<16|j;
	    		if(i < r)s += a[i];
	    	}
	    	assert q == r-h;
	    	Arrays.sort(c);
	    	p = b.length-1;
	    	int minlen = Integer.MAX_VALUE / 2;
	    	for(int i = 0;i < r-h;i++){
	    		long val = (c[i]>>>16)-O;
	    		int len = (int)(c[i]&65535);
	    		long need = K-val;
	    		while(p >= 0 && (b[p]>>>16)-O >= need){
	    			minlen = Math.min(minlen, (int)(b
[p]&65535));
	    			p--;
	    		}
	    		if(minlen != Integer.MAX_VALUE / 2){
	    			ans = Math.min(ans, minlen + len);
	    		}
	    	}
	    	dfs(l, h, a, K);
	    	dfs(h+1, r, a, K);
	    }
	    
	}
