class Solution {
    class Index{
        int[] rank;
        int n;
        public Index(int n){
            this.n = n;
            rank = new int[n+1];
            Arrays.fill(rank, -1);
        }
        
        public int get(int x){
            int ans = -1;
            while(x > 0){
                ans = Math.max(ans, rank[x]);
                x-=lastBit(x);
            }
            return ans;
        }
        
        public void add(int x, int v){
            while(x <= n){
                rank[x] = Math.max(rank[x], v);
                x+=lastBit(x);
            }
        }
        
        public int lastBit(int x){
            return x&(-x);
        }
    }
    
    public int shortestSubarray(int[] a, int K) {
        int n = a.length;
        int[] sum = new int[n+1];
        
        int ans = n+1;
        
        TreeSet<Integer> set = new TreeSet<Integer>();
        set.add(0);
        for(int i=0;i<n;i++){
            sum[i+1] = sum[i] + a[i];
            set.add(sum[i+1]);
        }
        
        TreeMap<Integer,Integer> map = new TreeMap<Integer,Integer>();
        
        int cnt = 1;
        Iterator<Integer> it = set.iterator();
        while(it.hasNext()){
            map.put(it.next(), cnt++);
        }
        
        Index index = new Index(cnt);
        index.add(map.get(0),0);
        for(int i=1;i<=n;i++){
            int gap = sum[i]-K;
            
            Map.Entry<Integer,Integer> entry = map.floorEntry(gap);
            if(entry != null){
                int j = index.get(entry.getValue());
                ans = Math.min(ans,i-j);
            }
            
            index.add(map.get(sum[i]),i);
        }
        
        
        if(ans == n+1) return -1;
        return ans;
    }
}
