class Solution {
    private static final int INF = 1234567890;
    
    public int shortestSubarray(int[] A, int K) {
        int N = A.length;
        long[] sum = new long[N + 1];
        for (int i = 0; i < N; ++i) {
            sum[i + 1] = sum[i] + A[i];
        }
        return can(A, sum, K);

    public static int can(int[] A, long[] sum, int K) {
        int N = A.length;
        TreeMap<Long, Integer> tm = new TreeMap<>();
        tm.put(0L, 0);
        int best = INF;
        
for (int i = 0; i < N; ++i) {
            long want = sum[i + 1] - K + 1;
            Long key = tm.lowerKey(want);
            if (key != null) {
                int idx = tm.get(key);
                best = Math.min(best, i - idx + 1);
            }
            Long k = tm.higherKey(sum[i + 1] - 1);
            while (k != null) {
                tm.remove(k);
                k = tm.higherKey(k);
            }
            tm.put(sum[i + 1], i + 1);
        }
        if (best == INF) {
            return -1;
        }
        return +best;
    }
}
