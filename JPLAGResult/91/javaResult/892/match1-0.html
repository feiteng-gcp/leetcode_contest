class Solution {

    public int shortestSubarray(int[] A, int K) {
        int len=Integer.MAX_VALUE;
        int n=A.length;
        if(A[0]>=K) return 1;
        int[] sum=new int[n+1];
        sum[0]=0;
        for(int i=1; i<=n; i++){
            sum[i]=A[i-1]+sum[i-1];
        }
        
        PriorityQueue<int[]> pq=new PriorityQueue<>((a, b)->(a[1]==b[1]?a[0]-b[0]:a[1]-b[1]));
        
        for(int i=1; i<=n; i++){
            int j=-1;
            if(sum[i]>=K) j=0;
            
            while(!pq.isEmpty()&&sum[i]-sum[pq.peek()[0]]>=K){
                j=Math.max(pq.poll()[0], j);
            }
            
            if(j>=0){
                len=Math.min(len, i-j);
            }
            pq.add(new int[]{i, sum[i]});
        }
        
        return len==Integer.MAX_VALUE?-1:len;
    }
}
