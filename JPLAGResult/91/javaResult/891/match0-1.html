class Solution {
    public int matrixScore(int[][] A) {
        if(A.length == 0 || A[0].length == 0) return 0;
        
int m = A.length, n = A[0].length;
        for(int i = 0; i < m; ++i) {
            if(A[i][0] == 0) {
                for(int j = 0; j < n; ++j) A[i][j] = 1- A[i][j]
;
            }
        }
        int ans = m;
        for(int j = 1; j < n; ++j) {
            int count = 0;
            for(int i = 0; i < m; ++i) {
                if(A[i][j] == 1) ++count;
            }
            ans = ans*2 + Math.max(count, m - count);
        }
        return ans;
    }
}
