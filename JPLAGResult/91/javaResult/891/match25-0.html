class Solution {
    public int matrixScore(int[][] A) {
        
        int m = A.length, n = A[0].length;
        int[][] valCount = new int[2][2];
        
        /*
        for(int i=0;i<m;i++){
            for(int j=1;j<n;j++){
                int first = (A[i][0]==1?1:0);
                if(A[i][j]==0)
                    valCount[first][0] += Math.pow(2,n-j-1);
                else
                    valCount[first][1] += Math.pow(2,n-j-1);
            }
        }
        
        if(valCount[0][1]+valCount[1][0] > valCount[0][0]+valCount[1][1]){
            System.out.println("true");
            flipCol(A,0);
        }
         */  
        
        for(int i=0;i<m;i++){
            if(A[i][0]==0)
                flipRow(A,i);
        }
        
        for(int j=1;j<n;j++){
            int colCount = 0;
            for(int i=0;i<m;i++){
                if(A[i][j]==1)
                    colCount++;
            }
            if(colCount<(m*1.0)/2)
                flipCol(A,j);
        }
        
        int res = 0;
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                if(A[i][j]==1)
                    res += Math.pow(2,n-j-1);
            }
        }
        
        return res;
    }
    
    public void flipRow(int[][] A , int r){
        for(int i=0;i<A[0].length;i++)
            A[r][i] ^=1;
    }
    
    public void flipCol(int[][] A , int c){
        for(int i=0;i<A.length;i++)
            A[i][c] ^=1;
    }
}
