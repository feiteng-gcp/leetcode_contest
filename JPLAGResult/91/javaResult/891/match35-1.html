class Solution {
    public int matrixScore(int[][] A) {
        for (int i = 0; i < A.length; i++) {
            if (A[i][0] == 1) continue;
            else 
{
                for (int j = 0; j < A[0].length; j++) {
                    A[i][j] = 1 - A[i][j];
                }
            }
        }
        for (int j = 0; j < A[0].length; j++) {
            int sum = 0;
            for (int i = 0; i < A.length; i++) {
                sum += A[i][j];
            }
            if (sum <= A.length / 2) {
                for (int i = 0; i < A.length; i++) A[i][j] = 1 - A[i][j]
;
            }
        }
        int res = 0;
        for (int i = 0; i < A.length; i++) {
            int sum = 0;
            for (int j = 0; j < A[0].length; j++) {
                sum = sum * 2 + A[i][j];
            }
            res += sum;
        }
        return res;
    }
}
