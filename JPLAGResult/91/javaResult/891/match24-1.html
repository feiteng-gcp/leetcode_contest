class Solution {
    public int matrixScore(int[][] A) {
        int[][] X = copy(A);
        int[][] Y = copy(A);
        flipCol(Y, 0);
        return Math.max(maximize(X), maximize(Y));
    }
    
    public static int maximize(int[][] X) {
        
for (int i = 0; i < X.length; ++i) {
            if (X[i][0] == 0) {
                flip(X, i);
            }
        }
        for (int i = 1; i < X[0].length; ++i) {
            int cnt1 = 0;
            for (int r = 0; r < X.length; ++r) {
                cnt1 += X[r][i];
            }
            
if (cnt1 <= X.length / 2) {
                flipCol(X, i);
            }
        }
        int sum = 0;
        for (int r = 0; r < X.length; ++r) {
            for (int c = 0; c < X[r].length; ++c) {
                int b = X[r].length - c - 1;
                sum += X[r][c] * (1 << b);
            }
        
}
        return sum;
    }
    
    public static void flip(int[][] X, int r) {
        for (int i = 0; i < X[r].length; ++i) {
            X[r][i] = 1 - X[r][i];
        }
    }
    
    public static void flipCol(int[][] X, int c) {
        for (int i = 0; i < X.length; ++i) {
            X[i][c] = 1 - X[i][c];
        }
    }
    
    public static int[][] copy(int[][] A) {
        int[][] B = new int[A.length][];
        for (int i = 0; i < A.length; ++i) {
            B[i] = new int[A[i].length];
            for (int j = 0; j < A[i].length; ++j) {
                B[i][j] = A[i][j];
            }
        }
        return B;
    }
}
