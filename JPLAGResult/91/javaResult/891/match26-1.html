class Solution {
    public int matrixScore(int[][] A) {
        int a = 0;
        int n = A.length, m = A[0].length;
        for (int[] row : A) {
            if (row[0] == 0) {
                for (int i = 0; i < m; i++) row[i] = 1 - row[i]
;
            }
        }
        
        for (int i = 0; i < m; i++) {
            int n1 = 0;
            for (int j = 0; j < n; j++) n1 += A[j][i];
            
            if (n1 < n - n1) {
                for (int j = 0; j < n; j++) A[j][i] = 1 - A[j][i];
            }
        }
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                //System.out.println(i +  " " + j + " " + A[i][j]);
                a += (A[i][j] << (m-1-j));
            }
        }
        return a;
    }
}
