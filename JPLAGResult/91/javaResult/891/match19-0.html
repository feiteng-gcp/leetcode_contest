class Solution {
    public int matrixScore(int[][] A) {
        // adj 1st col => all 1
        for (int i = 0; i < A.length; i++) {
            if (A[i][0] == 0) {
                // flip all row
                for (int j = 0; j < A[i].length; j++) {
                    if(A[i][j] == 0) {
                        A[i][j] = 1;
                    } else {
                        A[i][j] = 0;
                    }
                }
            }
        }
        // for (int i = 0; i < A.length; i++) {
        //     for (int j = 0; j < A[i].length; j++) {
        //         System.out.print(A[i][j]+" ");
        //     }
        //     System.out.print("\n");
        // }
        // check all other col
        for (int j = 1; j < A[0].length; j++) {
            int count = 0;
            for (int i = 0; i < A.length; i++) {
                count += A[i][j];
            }
            if (count * 2 < A.length) {
                // flip col j
                for (int i = 0; i < A.length; i++) {
                    if(A[i][j] == 0) {
                        A[i][j] = 1;
                    } else {
                        A[i][j] = 0;
                    }
                }
            }
            // for (int s = 0; s < A.length; s++) {
            //     for (int t = 0; t < A[s].length; t++) {
            //         System.out.print(A[s][t]+" ");
            //     }
            //     System.out.print("\n");
            // }
        }
        // get result
        int res = 0;
        for (int i = 0; i < A.length; i++) {
            int curr = 0;
            for (int j = 0; j < A[i].length; j++) {
                curr = curr << 1;
                //System.out.println(curr);
                if (A[i][j] > 0) {
                    curr++;
                    //System.out.println(curr);
                }
            }
            res += curr;
        }
        return res;
    }
}
