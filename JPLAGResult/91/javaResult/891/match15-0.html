class Solution {
    public int matrixScore(int[][] A) {
    	int m = A.length;
    	int n = A[0].length;
    	for (int i = 0; i < m; ++i) {
    		int[] num = A[i];
    		if (num[0] == 0) for (int j = 0; j < n; ++j) num[j] = num[j] == 0 ? 1 : 0;
    	}
    	for (int i = 0; i < n; ++i) {
    		int zero = 0;
    		for (int j = 0; j < m; ++j) {
    			if (A[j][i] == 0) zero++;
    		}
    		int half = m % 2 == 0 ? m >> 1 : (m >> 1) + 1;
    		if (zero >= half) {
    			for (int j = 0; j < m; ++j) {
    				A[j][i] = A[j][i] == 0 ? 1 : 0;
        		}
    		}
    	}
    	int total = 0;
    	for (int i = 0; i < m; ++i) {
    		for (int j = n - 1,mask = 1; j >= 0; --j,mask <<= 1) {
    			total += A[i][j] * mask;
    		}
    	}
    	return total;
    }
}
