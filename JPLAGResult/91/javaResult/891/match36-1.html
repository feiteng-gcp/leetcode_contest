class Solution {
    public int matrixScore(int[][] A) {
        int r = A.length;
        int c = A[0].length;
        
for (int i = 0; i < r; i++) {
            if (A[i][0] != 1) {
                for (int j = 0; j < c; j++) {
                    A[i][j] = 1 - A[i][j];
                }
            }
        }
        int o = r * (int) Math.pow(2, c - 1);
        
for (int j = 1; j < c; j++) {
            int one = 0;
            for (int i = 0; i < r; i++) {
                one += A[i][j];
            }
            o += 
Math.max(one, r - one) * Math.pow(2, c - 1 - j);
        }
        return o;
    }

    public static void main(String[] args) {
        Solution s = new Solution();
//        System.out.println(s.lemonadeChange(new int[] {5,5,5,10,20}));
//        System.out.println(s.lemonadeChange(new int[] {5,5,10}));
//        System.out.println(s.lemonadeChange(new int[] {10,10}));
//        System.out.println(s.lemonadeChange(new int[] {5,5,10,10,20}));
        System.out.println(s.matrixScore(new int[][] {{0,0,1,1},{1,0,1,0},{1,1,0,0}}));
        System.out.println(s);
        System.out.println(s);
        System.out.println(s);
    }
}
