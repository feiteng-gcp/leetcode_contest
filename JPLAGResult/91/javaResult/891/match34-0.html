class Solution {
    
    public int matrixScore(int[][] A) {
        for(int[] a : A) {
            if(a[0] == 0) {
                for(int i = 0; i < a.length; ++ i) {
                    a[i] = 1 - a[i];
                }
            }
        }
        for(int j = 0; j < A[0].length; ++ j) {
            int ones = 0;
            for(int[] a : A) {
                ones += a[j];
            }
            if(ones * 2 < A.length) {
                for(int[] a : A) {
                    a[j] = 1 - a[j];
                }
            }
        }
        int value = 0;
        for(int[] a : A) {
            value += getValue(a);
        }
        return value;
    }
    
    public int matrixScore0(int[][] A) {
        int max = 0;
        for(int i = 0; i < (1 << A[0].length); ++ i) {
            int n = i;
            for(int j = 0; n > 0; ++ j, n /= 2) {
                if(n % 2 == 1) {
                    flipc(A, j);
                }
            }
            max = Math.max(max, getMax(A));
            n = i;
            for(int j = 0; n > 0; ++ j, n /= 2) {
                if(n % 2 == 1) {
                    flipc(A, j);
                }
            }
        }
        return max;
    }
    
    int getMax(int[][] A) {
        int max = 0;
        for(int[] a : A) {
            max += Math.max(getValue(a), getFlipValue(a));
        }
        return max;
    }
    
    int getValue(int[] a) {
        int n = 0;
        for(int i : a) {
            n = n * 2 + i;
        }
        return n;
    }
    
    int getFlipValue(int[] a) {
        int n = 0;
        for(int i : a) {
            n = n * 2 + (1 - i);
        }
        return n;
    }
    
    void flipc(int[][] A, int j) {
        for(int[] r : A) {
            r[j] = 1 - r[j];
        }
    }
}
