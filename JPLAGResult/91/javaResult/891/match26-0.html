class Solution {
    public int matrixScore(int[][] a) {
        int m = a.length;
        int n = a[0].length;
        
        int sum = 0;
        for(int i=0;i<m;i++){
            if(a[i][0] == 0){
                for(int j=0;j<n;j++){
                    a[i][j] = 1^a[i][j];
                }
            }
        }
        
        for(int j=0;j<n;j++){
            int c = 0;
            for(int i=0;i<m;i++){
                if(a[i][j] == 1) c++;
            }
            c = Math.max(c, m-c);
            sum = sum * 2 + c;
        }
        return sum;
    }
}
