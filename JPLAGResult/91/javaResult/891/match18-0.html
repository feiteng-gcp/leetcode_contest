class Solution {
    public int matrixScore(int[][] A) {
        int len1=A.length;
        int len2=A[0].length;
        for(int i=0;i<len1;i++){
            if(A[i][0]==1){
                continue;
            }
            for(int j=0;j<len2;j++){
                A[i][j]=Math.abs(A[i][j]-1);
            }
        }
        int maxBit = 1 << (len2-1);
        int sum = maxBit*len1;
        for(int i=1;i<len2;i++){
            int count = 0;
            for(int j=0;j<len1;j++){
                if(A[j][i]==1){
                    count++;
                }
            }
            maxBit=maxBit>>1;
            sum+=maxBit*Math.max(count,len1-count);
        }
        return sum;
    }
}
