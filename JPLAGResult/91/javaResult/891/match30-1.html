class Solution {
    public int matrixScore(int[][] A) {
        //change leading number to 1
        for( int[] B : A )
        {
            if( B[0] == 0 )
            {
                
for( int i = 0 ; i < B.length ; i++ )
                {
                    if( B[i] == 0 )
                    {
                        B[i] = 1;
                    }
                    else
                    {
                        B[i] = 0;
                    }
                }
            }
        }
        //change column, making them have most 1
        
int totalRow = A.length;
        
        for( int col = 1 ; col < A[0].length ; col++ )
        {
            //count this col 1
            int num_1 = 0;
            
            for( int i = 0 ; i < totalRow ; i++ )
            {
                if(A[i][col] == 1)
                {
                    num_1++;
                }
            }
            
            if( num_1 < (totalRow - num_1) )
            {
                
for( int i = 0 ; i < totalRow ; i++ )
                {
                    if(A[i][col] == 1)
                    {
                        A[i][col] = 0;
                    }
                    else
                    {
                        A[i][col] = 1;
                    }
                }
            }
            
        }
        
        //cal result
        
        int result = 0;
        
        for( int[] B : A )
        {
            result += calBinaryVal(B);
        
}
        return result;
    }
    
    private int calBinaryVal(int[] array)
    {
        int res = 0 ;
        
        for( int i = array.length - 1 ; i >= 0 ; i-- )
        {
            if( array[i] == 1 )
            {
                res = res + (int)Math.pow( 2 , array.length - 1 - i );
            }
        }
        
        return res;
    }
}
