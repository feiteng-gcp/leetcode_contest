class Solution {
    public int matrixScore(int[][] A) {
        int m = A.length;
        if(m == 0) return 0;
        
int n = A[0].length;
        int sum = 0;
        for(int i=0; i<m; ++i){
            if(A[i][0] == 0){
                for(int j=0; j<n; ++j){
                    if(A[i][j] == 0) A[i][j] = 1;
                    else if(A[i][j] == 1) A[i][j] = 0;
                
}
            }
        }
        for(int j=0; j<n; ++j){
            int count=0;
            for(int i=0; i<m; ++i){
                count += A[i][j];
            }
            count = Math.max(count, m-count);
            sum += (int)Math.pow(2, n-1-j) * count;
        }
        return sum;
    }
}
