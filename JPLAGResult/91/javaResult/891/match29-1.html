class Solution {
    public int matrixScore(int[][] A) {
        
int n=A[0].length;
        int m=A.length;
        for(int i=0;i<m;++i) {
            if(A[i][0]==0) {
                for(int j=0;j<n;++j) A[i][j]^=1
;
            }
        }
        int res=0;
        for(int j=0;j<n;++j) {
            int base=1<<(n-j-1);
            int cnt=0;
            for(int i=0;i<m;++i) {
                cnt+=A[i][j];
            }
            res+=base*Math.max(cnt,m-cnt);
        }
        return res;
    }
}
