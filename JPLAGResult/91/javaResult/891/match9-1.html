class Solution {
    public int matrixScore(int[][] A) {
        int m = A.length;
        int n = A[0].length; System.out.println(cal(A));
        
for (int r = 0; r < m; r++) {
            if (A[r][0] == 0) {
                flipRow(A, r);
            }
        }
        for (int c = 1; c < n; c++) {
            int cnt = 0;
            for (int r = 0; r < m; r++) {
                cnt += A[r][c];
            }
            if (cnt > m - cnt) continue;
            else flipCol(A, c);
        }
        return cal(A);
    }
    
    public void flipRow(int[][] A, int r) {
        int len = A[0].length;
        
for (int i = 0; i < len; i++) {
            A[r][i] = A[r][i] == 1 ? 0 : 1;
        }
    }
    
    public void flipCol(int[][] A, int c) {
        int len = A.length;
        
for (int i = 0; i < len; i++) {
            A[i][c] = A[i][c] == 1 ? 0 : 1;
        }
    }
    
    public int cal(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        int ans = 0;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (A[i][j] == 1) {
                    ans += 1 << (n - j - 1);
                }
            }
        }
        return ans;
    }
}
