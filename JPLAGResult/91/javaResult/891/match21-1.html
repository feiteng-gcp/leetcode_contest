class Solution {
    public int matrixScore(int[][] A) {
        int m = A.length;
        int n = A[0].length;
        boolean[] rowFlip = new boolean[m];
        boolean[] colFlip = new boolean[n];
        for (int i = 0; i < m; ++i) {
            if (A[i][0] == 0) {
                for (int j = 0; j < n; ++j) {
                    A[i][j] = 1 - A[i][j];
                
}
            }
        }
        int res = 0;
        for (int j = 0; j < n; ++j) {
            int count = 0;
            for (int i = 0; i < m; ++i) {
                if (A[i][j] == 1) {
                    count++;
                
}
            }
            count = Math.max(count, m - count);
            res += count * (1 << (n - j - 1));
        }
        return res;
    }
}
