class Solution {
    public int matrixScore(int[][] A) {
        int N = A.length;
        int M = A[0].length;
        changeRow(A);
        changeCol(A);
        return cal(A) / 2;
    }
    private int cal(int [][] A){
        int M = A.length;
        int N = A[0].length;
        int sum = 0;
        for(int i = 0; i < M; i++){
            
            for(int j = N-1; j >= 0 ; j--){
                if(A[i][j] == 1) sum += (2 << (N-1 - j));
            }
        }
        return sum;
    }
    private void changeCol(int [][] A){
        int M = A.length;
        int N = A[0].length;
        for(int i = 0; i < N; i++){
            int count = 0;
            for(int j = 0; j < M; j++){
                if(A[j][i] == 0){
                    count--;
                }else{
                    count++;
                }
            }
            if(count < 0){
                flipCol(A, i);
            }
        }
        
    }
    private void flipCol(int [][] A, int j){
        int M = A.length;
        int N = A[0].length;
        for(int i = 0; i < M; i++){
            if(A[i][j] == 0){
                A[i][j] = 1;
            }else{
                A[i][j] = 0;
            }
        }
    }
    private void changeRow(int [][] A){
        int N = A.length;
        int M = A[0].length;
        for(int i = 0; i < N; i++){
            if(A[i][0] == 0){
                flip(A, i);
            }
        }
    }
    private void flip(int [][] A, int i){
        int M = A[0].length;
        for(int j = 0; j < M; j++){
            if(A[i][j] == 0){
                A[i][j] = 1;
            }else{
                A[i][j] = 0;
            }
        } 
    }
    
}
