class Solution {
    public int matrixScore(int[][] A) {
        int answer = 0;
        int rows = A.length;
        int cols = A[0].length;
        for(int x=0;x<rows;x++){
            if(A[x][0] != 1){
                for(int y=0;y<cols;y++){
                    A[x][y] = A[x][y]^1;
                }
            }
        }
        answer += rows*Math.pow(2, cols-1);
        for(int x=1;x<cols;x++){
            int sum = 0;
            for(int y=0;y<rows;y++){
                if(A[y][x] == 1){
                    sum++;
                }
            }
            if(sum < (rows+1)/2){
                for(int y=0;y<rows;y++){
                    A[y][x] = A[y][x]^1;
                }
                sum = rows-sum;
            }
            answer+=sum*Math.pow(2, cols-1-x);
        }
        //for(int x=0;x<rows;x++){
        //    for(int y=0;y<cols;y++){
        //        System.out.print(A[x][y] + " ");
        //    }
        //    System.out.println();
        //}
        return answer;
    }
    
}
