class Solution {
    public int matrixScore(int[][] A) {
        int r = A.length;
        int c = A[0].length;
        for (int i = 0; i < r; i++) {
            if (A[i][0] == 0) {
                switchRow(A, i);
            }
        }
        
        for (int j = 1; j < c; j++) {
            int num1 = 0;
            for(int i = 0; i < r; i++) {
                if (A[i][j] == 1) {
                    num1++;
                }
            }
            if (num1 <= r/2) {
                switchColumn(A, j);
            }
        }
        
        int sum = 0;
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < c; j++) {
                sum += A[i][j] * Math.pow(2, (c - j - 1));
            }
        }
        return sum;
    }
    
    private void switchRow(int[][] A, int i) {
        int c = A[0].length;
        for (int j = 0; j < c; j++) {
            if (A[i][j] == 0) {
                A[i][j] = 1;
            } else {
                A[i][j] = 0;
            }
        }
    }
    
    private void switchColumn(int[][] A, int j) {
        int r = A.length;
        for (int i = 0; i < r; i++) {
            if (A[i][j] == 0) {
                A[i][j] = 1;
            } else {
                A[i][j] = 0;
            }
        }
    }
}
