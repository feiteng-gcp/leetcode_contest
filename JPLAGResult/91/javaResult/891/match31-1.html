class Solution {
    public int matrixScore(int[][] A) {
        int wid = A.length, len = A[0].length;
        
for (int j = 0; j < wid; j++) {
            if (A[j][0] == 0) {
                for (int i = 0; i < len; i++) {
                    A[j][i] = 1 - A[j][i];
                }
            }
        }
        int base = 1;
        int res = 0;
        
for (int i = len - 1; i >= 0; i--) {
            int sum = 0;
            for (int j = 0; j < wid; j++) {
                sum += A[j][i];
            }
            if (sum <= wid / 2) {
                sum = wid - sum;
            }
            res += sum * base;
            base *= 2;
        }
        return res;
    }
}
