class Solution {
    public int matrixScore(int[][] matrix) {
        int n = matrix.length;
        int m = matrix[0].length;
        if(n==0 || m==0) return 0;
        for(int i = 0; i < n; ++ i){
            if(matrix[i][0] == 0){
                toggleRow(matrix, i);
            }
        }
        // for(int[] num : matrix){
        //     System.out.println(Arrays.toString(num));
        // }
        for(int j = 1; j < m; ++ j){
            int setCnt = getColSetNum(matrix, j);
            if(setCnt*2 < n){
                toggleCol(matrix, j);
            }
        }
        // for(int[] num : matrix){
        //     System.out.println(Arrays.toString(num));
        // }
        int sum = 0;
        for(int i = 0; i < n; ++ i){
            sum += getRowInteger(matrix[i]);
        }
        return sum;
    }
    public void toggleRow(int[][] matrix, int row){
        for(int j = 0; j < matrix[0].length; ++ j){
            matrix[row][j] = 1-matrix[row][j];
        }
    }
    public void toggleCol(int[][] matrix, int col){
        for(int i = 0; i < matrix.length; ++ i){
            matrix[i][col] = 1-matrix[i][col];
        }
    }
    public int getColSetNum(int[][] matrix, int col){
        int cnt = 0;
        for(int i = 0; i < matrix.length; ++ i){
            if(matrix[i][col] == 1) cnt += 1;
        }
        return cnt;
    }
    public int getRowInteger(int[] nums){
        int res = 0;
        int cnt = 1;
        for(int i = nums.length-1; i >= 0; -- i){
            int num = nums[i];
            if(num == 1) res += cnt;
            cnt = cnt << 1;
        }
        return res;
    }
}
