class Solution {
    public int matrixScore(int[][] A) {
        
int N = A.length;
        int M = A[0].length;
        
        for (int i = 0; i < N; i++) {
            if (A[i][0] == 0) {
                for (int j = 0; j < M; j++) {
                    A[i][j] ^= 1;
                
}
            }
        }
        
        for (int j = 1; j < M; j++) {
            int count = 0;
            for (int i = 0; i < N; i++) {
                count += A[i][j];
            }
            if (count <= N / 2) {
                for (int i = 0; i < N; i++) {
                    A[i][j] ^= 1;
                }
            }
        }
        
        int sum = 0;
        int power = 1;
        for (int j = M - 1; j >= 0; j--) {
            for (int i = 0; i < N; i++) {
                sum += A[i][j] * power;
            }
            power *= 2;
        }
        
        return sum;
    }
}
