import java.util.ArrayList;
import java.util.List;
class Solution {
	List<Integer> result;
	int val
ue;
	int k;
	public List<Integer> distanceK(TreeNode root, TreeNode target, int k) {
        this.result = new ArrayList<>();
        this.value = target.val;
        this.k = k;
        
distance(root, -1);
        return result;
    }
	public int distance(TreeNode node, int distance) {
		if (node != null) {
			if (distance >= 0) distance ++;
			if (node.val == value) distance = 0;
			int left = distance(node.left, distance);
			distance = distance == -1 ? left >= 0 ? left + 1 : -1 : distance;
			int right = distance(node.right, distance);
			if (distance == -1) {
				distance = right >= 0 ? right + 1 : -1;
				distance(node.left, distance);
			}
			if (distance == k) result.add(node.val);
			return distance;
		} else {
			return -1;
		}
	}
}
