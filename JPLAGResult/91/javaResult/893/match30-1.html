/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private void children(TreeNode node, int K, Set<Integer> kids) {
        if (node == null) return;
        if (K == 0) kids.add(node.val);
        else {
            children(node.left, K - 1, kids);
            children(node.right, K - 1, kids);
        }
    }
    
    private int backtrack(TreeNode node, TreeNode target, int K, Set<Integer> kids) {
        
if (node == null) return -1;
        if (node == target) {
            children(node, K, kids);
            return 1;
        }
        int r = backtrack(node.left, target, K, kids);
        if (r != -1) {
            if (K > r) children(node.right, K - r - 1, kids);
            if (K == r) kids.add(node.val);
            return r + 1;
        }
        r = backtrack(node.right, target, K, kids);
        if (r != -1) {
            if (K > r) children(node.left, K - r - 1, kids);
            if (K == r) kids.add(node.val);
            return r + 1;
        
}
        return -1;
    }
    
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        Set<Integer> kids = new HashSet<>();
        if (K != 0) {
            backtrack(root, target, K, kids);
            kids.remove(target.val);
        } else {
            kids.add(target.val);
        }
        return new ArrayList<>(kids);
    }
}
