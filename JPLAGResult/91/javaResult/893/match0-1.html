/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    // find grandchildren grandparent and sibling
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        List<Integer> result = new ArrayList<Integer>();
        recurse(root, new int[]{K}, result,  target);
        
return result;
    }
    
    public boolean recurse(TreeNode root, int[] K, List<Integer> list, TreeNode target) {
        if(root == null) return false;
        if(target == root) {
            //found = true;
            distanceK(root, K[0], list);
            return true;
        }
        boolean result = recurse(root.left, K, list, target);
        if(result) {
            //System.out.println(K);
            
if(K[0] >= 2) distanceK(root.right, K[0]-2, list);
            if(K[0] == 1) list.add(root.val);
            K[0]--;
            return true;
        }
        result = recurse(root.right, K, list , target);
        if(result) {
            
if(K[0] >= 2) distanceK(root.left, K[0]-2, list);
            if(K[0] == 1) list.add(root.val);
            K[0]--;
            return true;
        }
        return false;
    }
    
    public void distanceK(TreeNode root, Integer K, List<Integer> list) {
        if(root == null) return;
        if(K == 0) {
            list.add(root.val);
            return;
        }
        distanceK(root.left, K-1, list);
        distanceK(root.right, K-1, list);
    }
}
