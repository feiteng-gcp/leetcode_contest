/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        List<Integer> res = new ArrayList<Integer>();
        if(K==0) { res.add(target.val); return res;}
        
        HashMap<Integer, TreeNode> map = new HashMap<>();
        HashMap<Integer, Integer> parents = new HashMap<>();
        Stack<TreeNode> st = new Stack<>();
        st.push(root);
        
        while(!st.isEmpty()){
            TreeNode cur = st.pop();
            map.put(cur.val, cur);
            if(cur.val==target.val) break;
            if(cur.left!=null){   
               parents.put(cur.left.val, cur.val);
               st.push(cur.left); 
            }
            
            if(cur.right!=null){
                parents.put(cur.right.val, cur.val);
                st.push(cur.right);
            }
            
        }
        
        // find all children
        HashSet<Integer> visited = new HashSet<>();
        bsf(target, K,0, res, visited);
        visited.add(target.val);
        
        int child = target.val;
        for(int count =1; count<=K; count++){
            if(parents.containsKey(child)){
                int p = parents.get(child);
                TreeNode pnode = map.get(p);
                bsf(pnode, K,count, res, visited);
                visited.add(pnode.val);
                child = p;
            }else{
                break;
            }
        }
        
        return res;
    }
    
    private void bsf(TreeNode root, 
int K, int step, List<Integer> res, HashSet<Integer> visited ){
        if(root==null) return;
        if(visited.contains(root.val)) 
return;
        if(step==K){
            res.add(root.val); return;
        }
        
        bsf(root.left, K, step+1, res, visited);
        bsf(root.right, K, step+1, res, visited);
    }
    
    
    
}
