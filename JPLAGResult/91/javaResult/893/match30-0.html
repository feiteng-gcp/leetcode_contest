/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

import java.util.ArrayList;
import java.util.HashMap;

class Solution {
    


    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        List<Integer> result = new ArrayList<>();
        
        if( root == null || target == null )
            return result;
        
        if( K < 0 )
        {
            return result;
        }
        else if( K == 0 )
        {
            result.add(target.val);
            return result;
        }        
        
        HashMap<TreeNode,TreeNode> map = new HashMap<>();
        
        DFS(root,map);
        
        List<Integer> candidate1 = distanceK_sub_down(target,K,map);
        List<Integer> candidate2 = distanceK_sub_up(target,K,map);
        
        result.addAll(candidate1);
        result.addAll(candidate2);
        
        return result;
    }
    
    private List<Integer> distanceK_sub_down(TreeNode target, int K, HashMap<TreeNode,TreeNode> map) {
        List<Integer> result = new ArrayList<>();
        
        if( K < 0 )
            return result;
        
        if( target == null )
            return result;
        
        if( K == 0 )
        {
            result.add(target.val);
            return result;
        }
        
        if( target.left != null )
        {
            result.addAll( distanceK_sub_down(target.left,K-1,map) );
        }
        
        if( target.right != null )
        {
            result.addAll( distanceK_sub_down(target.right,K-1,map) );
        }
        
        return result;
    }
    
    private List<Integer> distanceK_sub_up(TreeNode target, int K, HashMap<TreeNode,TreeNode> map) {
        List<Integer> result = new ArrayList<>();
        
        if( K < 0 )
            return result;
        
        if( target == null )
            return result;
        
        if( K == 0 )
        {
            result.add(target.val);
            return result;
        }
        else if( K == 1 )
        {
            if( map.containsKey(target) )
            {
                TreeNode parent = map.get(target);

                result.add(parent.val);
                
                return result;
            }
            else            
            {
                return result;
            }
        }
        else
        {
            if( map.containsKey(target) )
            {
                TreeNode parent = map.get(target);

                if( parent.left != target )
                {
                    result.addAll(distanceK_sub_down(parent.left,K-2, map));
                }
                if( parent.right != target )
                {
                    result.addAll(distanceK_sub_down(parent.right,K-2, map));
                }
                
                result.addAll(distanceK_sub_up(parent, K - 1 , map));

                return result;
            }
            else
            {
                return result;
            }
        }
    }
    
    private void DFS(TreeNode root,HashMap<TreeNode,TreeNode> map)
    {
        if( root == null )
            return;
        
        if( root.left != null )
        {
            if( map.containsKey(root.left) == false )
            {
                map.put(root.left,root);
            }
            
            DFS(root.left,map);
        }
        
        if( root.right != null )
        {
            if( map.containsKey(root.right) == false )
            {
                map.put(root.right,root);
            }
            
            DFS(root.right,map);
        }
    }
    
    
   
}
