/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        List<Integer> answer = new ArrayList<Integer>();
        List<TreeNode> parents = new ArrayList<TreeNode>();
        getParents(root, target, parents);
        find(root, target, K, answer, parents);
        return answer;
    }
    public void find(TreeNode root, TreeNode target, int K, List<Integer> answer, List<TreeNode> parents){
        if(root == null){
            return;
        }
        if(root.val == target.val){
            //for(int i=0;i<parents.size();i++){
            //    System.out.print(parents.get(i).val + " ");
           // }
            //System.out.println();
            if(K==0){
                answer.add(root.val);
            }
            addChildren(root.left, K-1, answer, root.val, target.val);
            addChildren(root.right, K-1, answer, root.val, target.val);
            int temp = root.val;
            int size = parents.size();
            for(int x=0;K>0 && x<size;x++){
                //System.out.println(parents.get(0).val + " " + K + " " + temp + "hello");
                addChildren(parents.get(0), K-1, answer, temp, target.val);
                temp = parents.remove(0).val;
                K--;
            }
        }
        else{
            find(root.left, target, K, answer, parents);
            find(root.right, target, K, answer, parents);
        }
        
    }
    public void addChildren(TreeNode root, int K, List<Integer> answer, int val, int target){
        if(root== null || root.val == val || root.val == target){
            return;
        }
        if(K==0){
            answer.add(root.val);
            return;
        }
        else{
            addChildren(root.left, K-1, answer, val, target);
            addChildren(root.right, K-1, answer, val, target);
        }
    }
    public boolean getParents(TreeNode root, TreeNode target, List<TreeNode> parents){
        if(root==null){
            return false;
        }
        if(root.val == target.val){
            return true;
        }
        if(getParents(root.left, target, parents)){
            parents.add(root);
            return true;
        }
        else if(getParents(root.right, target, parents)){
            parents.add(root);
            return true;
        }
        return false;
    }
}
