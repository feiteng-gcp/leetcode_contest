/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        HashMap<Integer, HashSet<Integer>> graph = new HashMap();
        buildGraph(graph, root);
        List<Integer> result = new ArrayList<Integer>();
        dfs(result, graph, new HashSet(), target.val, K);
        return result;
    }
    
    private void buildGraph(HashMap<Integer, HashSet<Integer>> graph, TreeNode node) {
        if (node == null) {
            return;
        }
        if (node.left != null) {
            addEdge(graph, node.val, node.left.val);
            buildGraph(graph, node.left);
        }
        if (node.right != null) {
            addEdge(graph, node.val, node.right.val);
            buildGraph(graph, node.right);
        }
    }
    
    private void addEdge(HashMap<Integer, HashSet<Integer>> graph, int source, int target) {
        HashSet<Integer> set = graph.getOrDefault(source, new HashSet());
        set.add(target);
        graph.put(source, set);
        set = graph.getOrDefault(target, new HashSet());
        set.add(source);
        graph.put(target, set);
    }
    
    private void dfs(List<Integer> result, HashMap<Integer, HashSet<Integer>> graph, HashSet<Integer> visited, int value, int k) {
        if (k == 0) {
            result.add(value);
            return;
        }
        visited.add(value);
        if (graph.containsKey(value) == true) {
            HashSet<Integer> set = graph.get(value);
            for (int next : set) {
                if (visited.contains(next) == false) {
                    dfs(result, graph, visited, next, k - 1);
                }
            }
        }
    }
}
