/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        
        List<Integer> resultList = new ArrayList<>();
        if(root==null)
            return resultList;
        
        findNodeK(root,target,resultList,K);        
        return resultList;
    }
    
    public int findNodeK(TreeNode root, TreeNode target, List<Integer>resultList, int K){
        if(root==null)
            return -1;
        
        if(root==target){
            findKthNodes(root,resultList,K);
            return 1;
        }
        else{
            int lPath = findNodeK(root.left,target,resultList,K);
            int rPath = findNodeK(root.right,target,resultList,K);
            
            if(lPath==K || rPath==K){
                resultList.add(root.val);
                return -1;
            }
            else if(lPath>0){
                findKthNodes(root.right, resultList, K-lPath-1);
                return lPath+1;
            }
            else if(rPath>0){
                findKthNodes(root.left, resultList, K-rPath-1);
                return rPath+1;
            }
        }
        
        return -1;
    }
    
    public void findKthNodes(TreeNode root, List<Integer>resultList, int level){
        if(root==null || level<0)
            return;
        
        if(level==0)
            resultList.add(root.val);
        else if(level>0){
            findKthNodes(root.left, resultList, level-1);
            findKthNodes(root.right, resultList, level-1);
        }
    }
    
    
}
