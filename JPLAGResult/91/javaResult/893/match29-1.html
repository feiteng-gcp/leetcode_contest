/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        List<Integer> res = new LinkedList<>();
        findK(target, res, K);
        dfs(root, res, target, K);
        return res;
    }
    
    private int dfs(TreeNode root, List<Integer> res, 
TreeNode target, int K) {
        if (root == null) {
            return -1;
        }
        if (root == target) {
            return 0;
        }
        int left = dfs(root.left, res, target, K);
        int right = dfs(root.right, res, target, K);
        if (left == -1 && right == -1) {
            return -1;
        }
        if (left >= 0) {
            left = left + 1;
            if (K - left == 0) {
                res.add(root.val);
            } else {
                findK(root.right, res, K - left - 1);
            }
            return left;
        } else {
            right = right + 1;
            if (K - right == 0) {
                res.add(root.val);
            } else {
                findK(root.left, res, K - right - 1);
            }
            return right;
        
}
    }
    
    private void findK(TreeNode target, List<Integer> res, int K) {
        if(target == null || K < 0) {
            return;
        }
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(target);
        for (int i = 0; i < K; ++i) {
            Queue<TreeNode> temp = new LinkedList<>();
            while (!queue.isEmpty()) {
                TreeNode node = queue.poll();
                if (node.left != null) {
                    temp.add(node.left);
                }
                if (node.right != null) {
                    temp.add(node.right);
                }
            }
            queue = temp;
            if (queue.isEmpty()) {
                return;
            }
        }
        while (!queue.isEmpty()) {
            res.add(queue.poll().val);
        }
    }
}
