/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    List<Integer> ans = new ArrayList<Integer>();
    int k;
    int targetDepth = -1;
    
    public void add(TreeNode root, int k){
        if(root == null) return;
        if(k == 0){
            ans.add(root.val);
        }
        else{
            add(root.left, k-1);
            add(root.right, k-1);
        }
    }
    
    public boolean dfs(TreeNode root, TreeNode target, int k, int depth){
        if(root == null) return false;
        
        if(root == target){
            targetDepth = depth;
            add(root, k);
            return true;
        }
        else{
            boolean left = dfs(root.left, target, k, depth+1);
            if(left && targetDepth-depth<=k){
                if(targetDepth-depth == k) add(root, 0);
                else add(root.right, k - (targetDepth-depth) -1 );
                return true;
            }
            boolean right = dfs(root.right, target, k, depth+1);
            if(right){
                if(targetDepth-depth == k) add(root, 0);
                else add(root.left, k - (targetDepth-depth) -1);
                return true;
            }
        }
        return false;
    }
    
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        dfs(root, target, K, 0);
        return ans;
    }
}
