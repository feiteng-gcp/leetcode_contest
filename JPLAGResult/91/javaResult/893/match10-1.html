/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    Stack<TreeNode> st=new Stack();
    List<Integer> res=new ArrayList();
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        if(K==0) {
            res.add(target.val);
            return res;
        }
        find(root,target);
        int d=st.size()-1;
        while(st.size()>1) {
            if(d>K) st.pop();
            else if(d==K) {
                res.add(st.pop().val);
            } else {
                TreeNode cur=st.pop();
                TreeNode next=st.peek();
                if(cur.left==next) {
                    dfs(cur.right,K-d-1);
                } else {
                    dfs(cur.left,K-d-1);
                }
            }
            --d;
        }
        dfs(target.left,K-1);
        dfs(target.right,K-1);
        return res;
    }
    void dfs(
TreeNode root,int k) {
        if(root==null) return;
        if(k==0) {
            res.add(root.val);
            return;
        }
        dfs(root.left,k-1);
        dfs(root.right,k-1);
    }
    boolean find(
TreeNode root,TreeNode t) {
        if(root==null) return false;
        if(root==t) {
            st.push(root);
            return true;
        }
        if(find(root.left,t)) {
            st.push(root);
            return true;
        }
        if(find(root.right,t)) {
            st.push(root);
            return true;
        }
        return false;
    }
}
