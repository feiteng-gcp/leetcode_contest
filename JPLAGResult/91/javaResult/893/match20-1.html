/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private Stack<TreeNode> findNode(TreeNode target, TreeNode curr,Stack<TreeNode> stacker){
        if(curr == null)
            return null
;
        if(curr == target){
            stacker.push(curr);
            return stacker;
            
        }
        else{
            stacker.push(curr);
            Stack<TreeNode> left = findNode(target,curr.left, (Stack<TreeNode>)stacker.clone());
            if(left != null)
                return left;
            Stack<TreeNode> right = findNode(target, curr.right, (Stack<TreeNode>)stacker.clone());
            if(right != null)
                return right;
            return null;
        
}
    }
    private List<Integer> findLeaves(TreeNode curr, int dist, int k, Set<TreeNode> visited){
        List<Integer> ans = new ArrayList<>();

        if(curr == null|| visited.contains(curr))
            return ans;
        if(dist == k){
            ans.add(curr.val);
            return ans;
        }
        visited.add(curr);
        ans.addAll(findLeaves(curr.left, dist + 1, k, visited));
        
ans.addAll(findLeaves(curr.right, dist + 1, k, visited));
        return ans;
    }
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        Stack<TreeNode> path = findNode(target, root, new Stack<>());   
        // System.out.println(path.size());
        List<Integer> ans = new ArrayList<>();
        Set<TreeNode> visited = new HashSet<>();
        int count = K;
        while(!path.isEmpty()){
            // System.out.println();
            ans.addAll(findLeaves(path.pop(),0,count--, visited));
        }
        return ans;
    }
}
