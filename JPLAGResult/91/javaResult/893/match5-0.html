/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        List<Integer> res = new ArrayList<>();
        
        searchTarget(root, target, res, K);
        
        return res;
    }
    
    int searchTarget(TreeNode node, TreeNode target, List<Integer> res, int K) {
        if (node == null) return -1;
        if (node == target) {
            addNodes(node, res, K);
            return 0;
        }
        int left = searchTarget(node.left, target, res, K);
        if (left >= 0) {
            int curr = left + 1;
            if (curr == K) res.add(node.val);
            if (curr < K) addNodes(node.right, res, K - curr - 1);
            return curr;
        }
        int right = searchTarget(node.right, target, res, K);
        if (right >= 0) {
            int curr = right + 1;
            if (curr == K) res.add(node.val);
            if (curr < K) addNodes(node.left, res, K - curr - 1);
            return curr;
        }
        return -1;
    }
    
    void addNodes(TreeNode node, List<Integer> res, int remaining) {
        if (node == null) return;
        if (remaining == 0) {
            res.add(node.val);
            return;
        }
        addNodes(node.left, res, remaining - 1);
        addNodes(node.right, res, remaining - 1);
    }
}
