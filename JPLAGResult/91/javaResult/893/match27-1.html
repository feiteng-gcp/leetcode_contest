/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    private static final int MAX = 1000;
    
public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        ArrayList<Integer> ans = new ArrayList<>();
        if (K == 0) {
            ans.add(target.val);
            return ans;
        }
        Node[] nodes = new Node[MAX];
        for (int i = 0; i < nodes.length; ++i) {
            nodes[i] = new Node(i);
        }
        dfs(root, null, nodes);
        Queue<Node> qn = new ArrayDeque<>();
        Queue<Integer> qd = new ArrayDeque<>();
        boolean[] seen = new boolean[MAX];
        seen[target.val] = true;
        qn.offer(nodes[target.val]);
        qd.offer(0);
        while (!qn.isEmpty()) {
            Node curr = qn.poll();
            int dist = qd.poll();
            if (dist < K) {
                for (Node v : curr.next) {
                    if (!seen[v.id]) {
                        seen[v.id] = true;
                        qn.add(v);
                        qd.add(dist + 1);
                    }
                }
            } else if (dist == K) {
                ans.add(curr.id);
            }
        }
        return ans;
    }
    
    public static void dfs(TreeNode curr, TreeNode parent, Node[] nodes) {
        if (curr == null) {
            return;
        }
        if (parent != null) {
            nodes[curr.val].next.add(nodes[parent.val]);
        
}
        if (curr.left != null) {
            nodes[curr.val].next.add(nodes[curr.left.val]);
            dfs(curr.left, curr, nodes);
        }
        if (curr.right != null) {
            nodes[curr.val].next.add(nodes[curr.right.val]);
            dfs(curr.right, curr, nodes);
        }
    }
    
    public static class Node {
        public int id;
        public HashSet<Node> next = new HashSet<Node>();
        
        public Node(int i) {
            this.id = i;
        }
    }
}
