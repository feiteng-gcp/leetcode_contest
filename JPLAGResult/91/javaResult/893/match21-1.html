public class Solution {

    private final Map<Integer, Set<Integer>> neighbor = new HashMap<>();

    private void addEdge(int u, int v) {
        neighbor.putIfAbsent(u, new HashSet<>());
        neighbor.get(u).add(v);
    }

    private void traverse(TreeNode current) {
        if (current.left != null) {
            addEdge(current.val, current.left.val);
            addEdge(current.left.val, current.val);
            traverse(current.left);
        }
        if (current.right != null) {
            addEdge(current.val, current.right.val);
            
addEdge(current.right.val, current.val);
            traverse(current.right);
        }
    }

    private void search(int current, int left, Set<Integer> result, Set<Integer> seen) {
        if (seen.contains(current)) {
            return;
        }
        seen.add(current);
        if (left == 0) {
            result.add(current);
            return;
        }
        if (neighbor.containsKey(current)) {
            for (int v : neighbor.get(current)) {
                search(v, left - 1, result, seen);
            }
        }
    }

    
public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        Set<Integer> result = new HashSet<>();
        Set<Integer> seen = new HashSet<>();
        traverse(root);
        search(target.val, K, result, seen);
        return new ArrayList<>(result);
    }
}
