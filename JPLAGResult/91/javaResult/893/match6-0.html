/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        List<TreeNode> path = new ArrayList<>();
        findPath(root, target, path);
        List<Integer> ans = new ArrayList<>();
        int i = path.size() - 1;
        ans.addAll(solve(path.get(i), K));
        //System.out.println(path.get(i).val + ", " + K);
        K -= 2;
        --i;
        while(K >= 0 && i >= 0) {
            TreeNode curr = path.get(i), pre = path.get(i+1);
            if(curr.left == pre)  {
                ans.addAll(solve(curr.right, K));
            }
            else {
                ans.addAll(solve(curr.left, K));
            }
            --i;
            --K;
        }
        if(i >= 0 && K == -1) ans.add(path.get(i).val);
        return ans;
    }
    
    private boolean findPath(TreeNode root, TreeNode target, List<TreeNode> path) {
        if(root == null) return false;
        path.add(root);
        if(root == target) return true;
        if(findPath(root.left, target, path) || findPath(root.right, target, path)) return true;
        path.remove(path.size()-1);
        return false;
    }
    
    private List<Integer> solve(TreeNode root, int i) {
        if(root == null || i < 0) return new ArrayList<>();
        List<Integer> ans = new ArrayList<>();
        if(i == 0) {
            ans.add(root.val);
            return ans;
        }
        else {
            ans.addAll(solve(root.left, i-1));
            ans.addAll(solve(root.right, i-1));
            return ans;
        }
    }
}
