/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public List<Integer> distanceK(TreeNode root, TreeNode target, int K) {
        List<TreeNode> path = new ArrayList<>();
        getPath(root, target, path);
        List<TreeNode> tp = new ArrayList<>(path);
        List<Integer> result = new ArrayList<>();
        path.clear();
        dfs(root, K, path, tp, result);
        return result;
    }
    
    void dfs(TreeNode root, int K, List<TreeNode> path, List<TreeNode> tp, List<Integer> result) {
        if(root == null) {
            return;
        }
        path.add(root);
        if(getDistance(path, tp) == K) {
            result.add(root.val);
        }
        dfs(root.left, K, path, tp, result);
        dfs(root.right, K, path, tp, result);
        path.remove(path.size() - 1);
    }
    
    int getDistance(List<TreeNode> p1, List<TreeNode> p2) {
        int same = 0;
        for(int i = 0; i < p1.size() && i < p2.size(); ++ i) {
            if(p1.get(i) == p2.get(i)) {
                ++ same;
            }
        }
        return p1.size() + p2.size() - same * 2;
    }
    
    boolean getPath(TreeNode root, TreeNode target, List<TreeNode> path) {
        if(root == null) {
            return false;
        }
        path.add(root);
        if(root == target) {
            return true;
        }
        if(getPath(root.left, target, path)) {
            return true;
        }
        if(getPath(root.right, target, path)) {
            return true;
        }
        path.remove(path.size() - 1);
        return false;
    }
}
