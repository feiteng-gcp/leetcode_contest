class Solution {
    public boolean lemonadeChange(int[] bills) {
        int n5 = 0, n10 = 0;
        for(int b : bills) {
            if(b == 5) {
                n5++;
            } else if(b == 10) {
                if(n5 == 0) return false;
                n5--;
                n10++;
            } else if(b == 20) {
                if(n10 > 0) {
                    n10--;
                    if(n5 > 0) {
                        n5--;
                    } else return false;
                } else if(n5 >= 3) {
                    n5 -=3;
                } else return false;
            }
        }
        return true;
    }
}
