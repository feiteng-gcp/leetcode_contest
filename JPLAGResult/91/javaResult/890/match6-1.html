class Solution {
    public boolean lemonadeChange(int[] bills) {
        int fs = 0;
        int ts = 0;
        
        
for (int i = 0; i < bills.length; i++) {
            if (bills[i] == 5) fs++;
            else if (bills[i] == 10) {
                fs--;
                ts++;
            }
            else {
                if (ts > 0) ts--;
                else fs -= 2;
                fs--;
            }
            if (fs < 0 || ts < 0) return false;
        }
        
        return true;
    }
}
