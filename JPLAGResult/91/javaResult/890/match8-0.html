class Solution {
    public boolean lemonadeChange(int[] bills) {
        int fives = 0;
        int tens = 0;
        for(int i=0;i<bills.length;i++) {
            if(bills[i] == 5) {
                fives++;
            } else if(bills[i] == 10) {
                if(fives <= 0) return false;
                fives--;
                tens++;
            } else if(bills[i] == 20) {
                if(tens > 0) {
                    tens--;
                    if(fives > 0) {
                        fives--;
                    } else {
                        return false;
                    }
                } else if(fives > 2) {
                    fives -= 3;
                } else {
                    return false;
                }
            }
        }
        return true;
    }
}
