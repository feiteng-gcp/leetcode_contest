class Solution {
    public boolean lemonadeChange(int[] bills) {
        int[] cnt = new int[3];
        for (int x : bills) {
            if (x == 20) {
                if (cnt[0] > 0 && cnt[1] > 0) {
                    cnt[0]--;
                    cnt[1]--;
                } else if (cnt[0] > 2) {
                    cnt[0] -= 3;
                } else {
                    return false;
                }
                cnt[2]++;
            } else if (x == 10) {
                if (cnt[0] > 0) {
                    cnt[0]--;
                } else {
                    return false;
                }
                cnt[1]++;
            } else {
                cnt[0]++;
            }
        }
        return true;
    }
}
