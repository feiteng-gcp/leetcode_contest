class Solution {
    public boolean lemonadeChange(int[] bills) {
        int[] arr = new int[2];
        
for (int x : bills) {
            if (x == 5) {
                arr[0]++;
            } else if (x == 10) {
                if (arr[0] == 0) {
                    return false;
                }
                arr[0]--;
                arr[1]++;
            } else {
                if (arr[1] == 0) {
                    if (arr[0] < 3) {
                        return false;
                    
}
                    arr[0] -= 3;
                } else {
                    arr[1]--;
                    if (arr[0] == 0) {
                        return false;
                    }
                    arr[0]--;
                }
            }
        }
        return true;
    }
}
