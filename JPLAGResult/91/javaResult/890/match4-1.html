class Solution {
    public boolean lemonadeChange(int[] bills) {
        int[] have = new int[2];
        for (int b : bills) {
            if (b == 5) have[0]++;
            else if (b == 10) {
                have[0]--;
                have[1]++;
            }
            else {
                if (have[1] > 0) {
                    have[1]--;
                    have[0]--;
                }
                else 
have[0] -= 3;
            }
            if (have[0] < 0 || have[1] < 0) return false;
        }
        return true;
    }
}
