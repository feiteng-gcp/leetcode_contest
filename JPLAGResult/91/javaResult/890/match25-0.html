	class Solution {
	    public boolean lemonadeChange(int[] bills) {
	        int n = bills.length;
	        int[] f = new int[11];
	        for(int v : bills){
	        	if(v == 5){
	        		f[v]++;
	        	}else if(v == 10){
	        		if(f[5] == 0)return false;
	        		f[5]--;
	        		f[10]++;
	        	}else if(v == 20){
	        		if(f[10] > 0 && f[5] > 0){
	        			f[10]--;
	        			f[5]--;
	        		}else if(f[5] >= 3){
	        			f[5] -= 3;
	        		}else{
	        			return false;
	        		}
	        	}
	        }
	        return true;
	    }
	}
