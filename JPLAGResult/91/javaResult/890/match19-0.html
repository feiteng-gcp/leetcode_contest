class Solution {
    public boolean lemonadeChange(int[] bills) {
        int[] b = {0, 0}; // 5 10
        for(int i = 0; i<bills.length; i++){
            if(bills[i] == 5){
                b[0] ++;
            } else if(bills[i] == 10){
                if(b[0] < 1){
                    return false;
                }
                b[0] --;
                b[1] ++;
            } else {
                if(b[1] > 0){
                    if(b[0] < 1){
                        return false;
                    } else {
                        b[0] --;
                        b[1] --;
                    }
                    
                    
                } else {
                    if(b[0] < 3){
                        return false;
                    }
                    b[0] -= 3;
                }
            }
        }
        return true;
    }
}
