class Solution {
    public boolean lemonadeChange(int[] bills) {
    	int[] billnum = new int[2];
    	
    	for(int i = 0; i < bills.length; i++) {
    		if(bills[i] == 5) billnum[0]++;
    		else if(bills[i] == 10) {
    			billnum[0]--;
    			billnum[1]++;
    		}
    		else if(billnum[1] > 0){
    			billnum[0]--;
    			billnum[1]--;
    		}
    		else {
    			billnum[0]-=3;
    		}
    		if(billnum[0] < 0 || billnum[1] < 0) return false;
    	}
    	return true;
    }
}
