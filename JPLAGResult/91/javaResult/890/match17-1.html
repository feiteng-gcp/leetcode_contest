class Solution {
    public boolean lemonadeChange(int[] bills) {
        int f = 0, t = 0;
        
for (int b : bills) {
            if (b == 20) {
                if (f >= 1 && t >= 1) {
                    t -= 1;
                    f -= 1;
                } else if (f >= 3) {
                    f -= 3;
                } else {
                    return false;
                }
            } else if (b == 10) {
                if (f >= 1) {
                    f -= 1;
                    t += 1;
                } else {
                    return false;
                }
            } else {
                f += 1;
            }
      
        }
        return true;
    }
}
