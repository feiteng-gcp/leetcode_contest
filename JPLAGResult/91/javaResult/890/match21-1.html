class Solution {
    public boolean lemonadeChange(int[] bills) {
        int a = 0;
        int b = 0;
        int n = bills.length;
        
        for (int i = 0; i < n; i++) {
            if (bills[i] == 5) {
                a++;
            } else if (bills[i] == 10) {
                b++;
                a--;
                if (a < 0) {
                    
return false;
                }
            } else {
                if (b >= 1) {
                    b--;
                    a--;
                } else {
                    a -= 3;
                }
                if (a < 0) {
                    return false;
                
}
            }
        }
        return true;
    }
}
