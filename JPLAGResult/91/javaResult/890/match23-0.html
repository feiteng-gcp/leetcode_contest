//import java.util.HashMap;

class Solution {
    public boolean lemonadeChange(int[] bills) {
        
        //HashMap<Integer,Integer> map = new HashMap<>();
        int numFive = 0;
        int numTen = 0;
        
        for( int i = 0 ; i < bills.length ; i++ )
        {
            int curBill = bills[i];
            
            if( curBill == 5 )
            {
                numFive++;
            }
            else if( curBill == 10 )
            {
                if( numFive > 0 )
                {
                    numFive--;
                    numTen++;                    
                }
                else
                {
                    return false;
                }
            }
            else
            {
                if( numTen > 0 && numFive > 0 )
                {
                    numTen--;
                    numFive--;
                }
                else if( numFive >= 3 )
                {
                    numFive = numFive - 3;
                }
                else
                {
                    return false;
                }
            }
            
        }
        
        return true;
    }
}
