class Solution:
    def lemonadeChange(self, bills):
        """
        :type bills: List[int]
        :rtype: bool
        """
        n = [0] * 2
        for b in bills:
            if b == 5:
                n[0] += 1
            elif b == 10:
                if n[0] == 0:
                    return False
                n[0] -= 1
                n[1] += 1
            else:
                
if n[1] >= 1 and n[0] >= 1:
                    n[0] -= 1
                    n[1] -= 1
                elif n[0] >= 3:
                    n[0] -= 3
                else:
                    return False
        return True
