import collections

class Solution:
    def lemonadeChange(self, bills):
        """
        :type bills: List[int]
        :rtype: bool
        """
        counter 
= collections.defaultdict(int)
        for b in bills:
            if b == 5:
                counter[b] += 1
            elif b == 10:
                if counter[5] == 0:
                    return False
                else:
                    counter[5] -= 1
                    counter[10] += 1
            elif b == 20:
                if counter[5] == 0:
                    return False
                if counter[10] >= 1:
                    counter[10] -= 1
                    counter[5] -= 1
                else:
                    if counter[5] < 3:
                        return False
                    else:
                        counter[5] -= 3
        return True
