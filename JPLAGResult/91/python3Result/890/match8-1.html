from collections import defaultdict
class Solution:
    def lemonadeChange(self, bills):
        """
        :type bills: List[int]
        :rtype: bool
        """
        bill = defaultdict(int)
        # print('here')
        for cu in bills:
            # print(cu)
            # print(bills)
            if cu == 5:
                bill[5] += 1
            elif cu == 10:
                bill[10] += 1

                if bill[5] > 0:
                    # print('give 5')
                    bill
[5] -= 1 #give 5 back
                else:
                    # print('cannot give 5')
                    return False
            else:
                if bill[10] > 0 and bill[5] > 0:
                    bill[10] -= 1
                    bill[5] -= 1
                elif bill[5] >= 3:
                    bill[5] -= 3
                else:
                    return False
        return True
