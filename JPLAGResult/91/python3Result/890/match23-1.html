class Solution:
    def lemonadeChange(self, bills):
        """
        :type bills: List[int]
        :rtype: bool
        """
        cnts5, cnts10 
= 0, 0
        for bill in bills:
            if bill == 5:
                cnts5 += 1
            elif bill == 10:
                if cnts5 == 0:
                    return False
                else:
                    cnts5 -= 1
                cnts10 += 1
            else:
                if cnts10 > 0:
                    cnts10 -= 1
                    cnts5 -= 1
                else:
                    cnts5 -= 3
                if cnts5 < 0:
                    return False
        return True
