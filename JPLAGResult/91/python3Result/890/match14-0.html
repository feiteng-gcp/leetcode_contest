from collections import defaultdict
class Solution:
    def lemonadeChange(self, bills):
        """
        :type bills: List[int]
        :rtype: bool
        """
        d = defaultdict(int)
        for bill in bills:
            if bill == 10:
                if d[5] > 0:
                    d[5] -= 1
                else:
                    return False
            elif bill == 20:
                if d[10] > 0 and d[5] > 0:
                    d[10] -= 1
                    d[5] -= 1
                elif d[5] >= 3:
                    d[5] -= 3
                else:
                    return False
            d[bill] += 1
        return True
