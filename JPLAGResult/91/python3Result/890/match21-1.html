class Solution:
    def lemonadeChange(self, bills):
        changes = [0]*2
        for b in bills:
            if b==5:
                changes[0] += 1
            elif b==10:
                if changes[0]==0:
                    return False
                else:
                    changes[0] -= 1
                changes[1] += 1
            elif b==20:
                
if changes[0]>=1 and changes[1]>=1:
                    changes[0] -= 1
                    changes[1] -= 1
                elif changes[0]>=3:
                    changes[0] -= 3
                else:
                    return False
        return True
