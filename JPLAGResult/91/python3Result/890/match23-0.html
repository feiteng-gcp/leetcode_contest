class Solution:
    def lemonadeChange(self, bills):
        """
        :type bills: List[int]
        :rtype: bool
        """
        five = ten = 0
        for coin in bills:
            if coin == 5:
                five += 1
            elif coin == 10:
                if five == 0:
                    return False
                else:
                    five -= 1
                ten += 1
            elif coin == 20:
                if ten > 0 and five > 0:
                    ten -= 1
                    five -= 1
                elif five >= 3:
                    five -= 3
                else:
                    return False
        return True
