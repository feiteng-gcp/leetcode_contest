class Solution:
    def lemonadeChange(self, bills):
        """
        :type bills: List[int]
        :rtype: bool
        """
        h = {5: 0, 10: 0, 20: 0}
        
        # Count the bills
        for bill in bills:
            #print('bill', bill)
            
            if bill == 5:
                h[5] += 1
            if bill == 10:
                if h[5] > 0:
                    h[5] -= 1
                else:
                    return False # no change available
                h[10] += 1
            if bill == 20:
                if h[5] > 0 and h[10] > 0:
                    h[5] -= 1
                    h[10] -= 1
                elif h[5] >= 3:
                    h[5] -= 3
                else:
                    return False
            #print(h)
                
        return True
