class Solution:
    def lemonadeChange(self, bills):
        """
        :type bills: List[int]
        :rtype: bool
        """

        dic = {}
        dic[5] = 0
        dic[10] = 0

        for b in bills:
            if b == 5:
                dic[5] += 1
            elif b == 10:
                if dic[5] <= 0:
                    return False
                dic[5] -= 1
                dic[10] += 1
            else:
                if dic[10] > 0 and dic[5] > 0:
                    dic[10] -= 1
                    dic[5] -= 1
                elif dic[5] >= 3:
                    dic[5] -= 3
                else:
                    return False
        return True
