class Solution:
    def calcScore(self, A):
        score = 0
        for ls in A:
            binStr = ''.join(map(str, ls))
            score += int(binStr, 2)
        return score

    def matrixScore(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        if not A or len(A) <= 0:
            return 0
        li = len(A)
        lj = len(A[0])
        # A = self.revert(A)
        print(A)
        for ls in A:
            if ls[0] == 0:
                for i in range(len(ls)):
                    ls[i] = (ls[i] + 1) % 2
        # print(A)
        for i in range(1, lj):
            cnt = 0
            for ls in A:
                if ls[i] == 1:
                    cnt += 1
            # print(cnt, lj)
            if cnt < li / 2:
                for ls in A:
                    ls[i] = (ls[i] + 1) % 2
        # print(A)
        return self.calcScore(A)
