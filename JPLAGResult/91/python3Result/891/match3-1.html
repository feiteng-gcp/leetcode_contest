class Solution:
    def matrixScore(self, A):
        if not A or not A[0]:
            return 0
        m, n = len(A), len
(A[0])
        for i in range(m):
            if A[i][0]==0:
                for j in range(n):
                    A[i][j] = 1-A[i][j]
        ret = m*(1<<(n-1))
        for j in range(1, n):
            cnt = 0
            for i in range(m):
                if A[i][j]==1:
                    cnt += 1
            ret += max(cnt, m-cnt)*(1<<(n-j-1))
        return ret
