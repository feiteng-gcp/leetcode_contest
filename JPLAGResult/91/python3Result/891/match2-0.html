class Solution:
    def matrixScore(self, A):
        """
        :type A: List[List[int]]
        :rtype: int
        """
        if not any(A):
            return 0
        rows, cols = len(A), len(A[0])
        # Make all row with leading 1
        for i in range(rows):
            if A[i][0] == 0:
                for j in range(cols):
                    A[i][j] = 1 - A[i][j] 
        # Add values by column            
        ans = 0            
        for j in range(cols):
            ones = sum(A[i][j] for i in range(len(A)))
            ans += max(ones, rows - ones) * (2 ** (cols - j - 1))
        print(A)
        return ans
