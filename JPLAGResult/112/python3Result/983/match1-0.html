class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        popped = [*reversed(popped)]

        stack = []
        for x in pushed:
            stack.append(x)
            while stack and popped and stack[-1] == popped[-1]: 
                stack.pop()
                popped.pop()

        return not popped
