class Solution:
    def validateStackSequences(self, pushed, popped):
        pop = 0
        pup = 0
        a = []
        while pop!=len(popped):
            if len(a) == 0:
                a.append(pushed[pup])
                pup+=1
            else:
                if popped[pop] == a[len(a)-1]:
                    a.pop()
                    pop+=1
                elif pup == len
(pushed):
                    return False
                else:
                    a.append(pushed[pup])
                    pup+=1
        return True
