class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        i = 0
        j = 0
        if len(pushed) != len(popped):
            return False
        stack = []
        while True:
            if i == len(pushed) and j == len(popped):
                break

            if len(stack) == 0:
                if i == len(pushed):
                    return False
                else:
                    stack.append(pushed[i])
                    i += 1
            elif stack[-1] == popped[j]:
                stack.pop()
                j += 1
            else:
                if i == len(pushed):
                    return False
                else:
                    stack.append(pushed[i])
                    i += 1
        return True
