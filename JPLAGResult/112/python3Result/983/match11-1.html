class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        if not pushed:
            return True
        
        push_i = 0
        pop_i = 0
        
        stack = []
        while push_i != len(pushed):
            pushing = pushed[push_i]
            stack.append(pushing)
            
            while stack and stack[-1] == popped[pop_i]:
                stack.pop()
                pop_i += 1
                
            push_i += 1
        
        while pop_i != len
(popped):
            if not stack or stack[-1] != popped[pop_i]:
                return False
            stack.pop()
            pop_i += 1
            
        return True
