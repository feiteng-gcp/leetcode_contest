class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        n=len(pushed)
        stack=[]
        i=0
        j=0
        while j<n:
            while (not stack or stack[-1]!=popped[j]) and i<n:
                stack.append(pushed[i])
                i+=1
            
if i==n and (not stack or stack[-1]!=popped[j]):
                return False
            stack.pop()
            j+=1
        return True
