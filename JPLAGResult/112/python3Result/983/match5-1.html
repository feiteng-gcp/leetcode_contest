class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        if not pushed:
            return True
        i=1
        imax = len(pushed)
        j=0
        jmax = len(popped)
        l = [pushed[0]]
        while True:
            if not l:
                if i < imax:
                    l.append(pushed[i])
                    i += 1
                elif j == jmax:
                    return True
                else:
                    return False
            elif l
[-1] == popped[j]:
                l.pop(-1)
                j += 1
            elif i < imax:
                l.append(pushed[i])
                i += 1
            else:
                return False
