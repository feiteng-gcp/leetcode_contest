class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        if(len(pushed)!=len(popped)):
            return False
        if(len(pushed)==0):
            return True
        st=[pushed[0]]
        s=0
        for i in range(1,len(pushed)):
            while(len(st)>0 and st[-1]==popped[s]):
                st.pop()
                s+=1
            st.append(pushed[i])
        while(len(st)>0 and s<len(popped)):
            if(st[-1]!=popped[s]):
                return False
            st.pop()
            s+=1
        return True
