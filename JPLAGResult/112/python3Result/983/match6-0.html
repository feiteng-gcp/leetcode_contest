class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        if pushed == [] and popped == []:
            return True
        
        pushed = list(reversed(pushed))
        popped = list(reversed(popped))
        
        validation_stack = [pushed.pop()]
        
        while pushed or popped:
            if popped and validation_stack and (popped[-1] == validation_stack[-1]):
                validation_stack.pop()
                popped.pop()
            elif pushed:
                validation_stack.append(pushed.pop())
            else:
                break
        if pushed or popped:
            return False
        else:
            return True
