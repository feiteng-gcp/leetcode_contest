class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        s = []
        i = 0
        j = 0
        n = len(pushed)
        for a in range(n * 2):
            if s and j < n and popped[j] == s[-1]:
                s.pop()
                j += 1
            else:
                if i == n:
                    return False
                s.append(pushed[i])
                i += 1
        return i == n and j == n
