class Solution:
    def validateStackSequences(self, pushed, popped):
        i=0
        stack=[]
        for pop in popped:
            while(len(stack)==0 or stack[-1]!=pop):
                if(i<len(pushed)):
                    stack.append(pushed[i])
                    i+=1
                else:
                    return False
            stack=stack[:-1]
        return True
