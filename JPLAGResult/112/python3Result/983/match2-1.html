class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        
        if not pushed:
            return True
        
        stack = []
        iPo 
= 0
        for pu in pushed:
            stack.append(pu)
            while stack and stack[-1] == popped[iPo]:
                iPo += 1
                stack.pop()
        return True if not stack else False
