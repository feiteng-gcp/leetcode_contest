class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        N = len(pushed)
        i, j = 0, 0
        st = []
        while j < N:
            if st and st[-1] == popped[j]:
                st.pop()
                j += 1
            elif i < N:
                st.append(pushed[i])
                i += 1
            else:
                return False
        return True
