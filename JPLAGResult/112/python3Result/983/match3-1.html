class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        stack = []
        i, j = 0, 0
        n, m = len(pushed), len(popped)
        while i < n or j < m:
            if len(stack) and stack
[-1] == popped[j]:
                stack.pop()
                j += 1
            else:
                if i == n and j < m: return False
                stack.append(pushed[i])
                i += 1
        return True
