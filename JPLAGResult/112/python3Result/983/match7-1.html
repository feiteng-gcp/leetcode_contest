class Solution:
    def validateStackSequences(self, pushed, popped):
        """
        :type pushed: List[int]
        :type popped: List[int]
        :rtype: bool
        """
        stack = []
        n = len(pushed)
        i = j = 0
        while i < n:
            stack
.append(pushed[i])
            i += 1
            while stack and stack[-1] == popped[j] and j < n:
                stack.pop()
                j += 1
        return len(stack) == 0
