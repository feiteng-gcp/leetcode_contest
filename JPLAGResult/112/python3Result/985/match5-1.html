class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens.sort()
        
        pt, ma, i, j = 0, 0, 0, len(tokens) - 1
        while i <= j:
            if P >= tokens[i]:
                P -= tokens[i]
                i += 1
                pt += 1
                ma = max(pt, ma)
                continue
            if pt >= 1:
                P += tokens[j]
                j -= 1
                pt -= 1
            else:
                break
        
        return ma
