class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens.sort()
        i = 0
        j = len(tokens)-1
        ret = 0
        while i <= j:
            if P >= tokens[i]:
                P -= tokens[i]
                i += 1
                ret += 1
            elif ret == 0:
                break
            else:
                P += tokens[j]
                j -= 1
                ret -= 1
                if j < i or tokens[i] > P:
                    ret += 1
                    break
        return ret
