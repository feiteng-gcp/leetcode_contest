class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """

        n = len(tokens)
        tokens = list(sorted(tokens))
        res = 0
        cur_power 
= P
        left, right = 0, n - 1
        temp_power = P
        temp_res = 0
        while left <= right:
            flag = False
            while left <= right and temp_power >= tokens[left]:
                temp_power -= tokens[left]
                temp_res += 1
                left += 1
                flag = True
            res = max(res, temp_res)
            if temp_res > 0 and right > left:
                temp_power += tokens[right]
                right -= 1
                temp_res -= 1
                flag = True
            if not flag:
                break
        return res
