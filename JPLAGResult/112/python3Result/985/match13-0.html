class Solution:
    def bagOfTokensScore(self, tokens, p):
        """
        :type tokens: List[int]
        :type p: int
        :rtype: int
        """
        if not tokens or p < min(tokens):
            return 0
        tokens.sort()
        scores = 1
        p = p - tokens[0]
        i, j = 1, len(tokens) - 1
        while i < j:
            if p >= tokens[i]:
                p -= tokens[i]
                scores += 1
                i += 1
            else:
                scores -= 1
                p += tokens[j]
                j -= 1
        if i == j and p >= tokens[i]:
            return scores + 1
        else:
            return scores
