class Solution:
    def bagOfTokensScore(self, tokens, P):
        tokens.sort()
        i,j=0,len(tokens)-1
        points=0
        power=P
        m_p=0
        while(i<=j):
            if(power>=tokens[i]):
                points+=1
                power-=tokens[i]
                i+=1
                m_p=max(m_p,points)
            elif(points>0):
                points-=1
                power+=tokens[j]
                j-=1
            else:
                break
        return m_p
