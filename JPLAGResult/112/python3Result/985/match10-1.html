class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        ans = 0
        tokens
.sort()
        n = len(tokens)
        i = 0

        while i < n:
            if P >= tokens[i]:
                ans += 1
                P -= tokens[i]
                i += 1
            elif n - 1 > i and ans > 0:
                P += tokens[n-1] - tokens[i]
                i += 1
                n -= 1
            else:
                break

        return ans
