class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens = sorted(tokens)
        cumsum = [0] * len(tokens)
        for i, x in enumerate(tokens):
            cumsum[i] = x if i == 0 else cumsum[i - 1] + x
        
        l, r = 0, len(tokens) - 1
        pts, curr, ans = 0, P, 0
        while l < r:            
            remain = cumsum[r] - (cumsum[l] if l > 0 else 0)
            # print(l, r, pts, curr, remain)
            
            if pts >= remain:
                pts += r - l + 1
                curr -= remain
                ans = max(ans, pts)
                break
            elif curr >= tokens[l]:
                pts += 1
                ans = max(ans, pts)
                curr -= tokens[l]
                l += 1
            elif pts > 0:
                pts -= 1
                curr += tokens[r]
                r -= 1
            else:
                break
        
        # print(l, r, pts, curr)
        if l == r and curr >= tokens[l]:
            pts += 1
            ans = max(ans, pts)
                
        return ans
