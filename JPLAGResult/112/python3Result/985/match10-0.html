class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        if len(tokens) == 0:
            return 0
        
        tokens.sort()
        
        a,b = 0, len(tokens)-1
        score = 0
        while a<=b:
            if tokens[a] <= P:
                score += 1
                P -= tokens[a]
                a+=1     
            elif score>0:
                P += tokens[b]
                b -= 1
                P -= tokens[a]
                a += 1
            else:
                break
                
        return score
