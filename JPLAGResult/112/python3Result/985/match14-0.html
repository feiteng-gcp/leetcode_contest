class Solution:
    def bagOfTokensScore(self, tokens, P):
        tokens = sorted(tokens)
        l = 0
        r = len(tokens) - 1
        
        res = 0
        while l <= r:
            while l <= r and P >= tokens[l]:
                P -= tokens[l]
                l += 1
                res += 1
            if l < r and res:
                P += tokens[r]
                r -= 1
                res -= 1
            else:
                break
        return res
