class Solution:
    def bagOfTokensScore(self, t, p):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        t.sort()
        i = 0
        j = len(t) - 1
        res = 0
        ct = 0
        
        while i <= j:
            if p >= t[i]:
                p -= t[i]
                ct += 1
                res = max(res, ct)
                i += 1
            elif ct > 0:
                ct -= 1
                p += t[j]
                j -= 1
            else:
                return res
        return res
