class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens.sort()
        point = 0
        i, j = 0, len(tokens) - 1
        result = 0
        while i <= j:
            flag = 0
            if P >= tokens[i]:
                P -= tokens[i]
                point += 1
                result = max(result, point)
                i += 1
                flag += 1
            elif point >= 1:
                point -= 1
                P += tokens[j]
                j -= 1
                flag += 1
            if not flag:
                break
        return result
