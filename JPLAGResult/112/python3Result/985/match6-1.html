class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        i = 0
        j = len(tokens)-1
        tokens.sort()
        ret = 0
        
        cur = 0
        
while i <= j:
            if P >= tokens[i]:
                P -= tokens[i]
                i += 1
                cur += 1
            elif cur:
                P += tokens[j]
                j -= 1
                cur -= 1
            else:
                break
                
            ret = max(cur, ret)
            
        return max(cur, ret)
