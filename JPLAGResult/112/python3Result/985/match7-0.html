from collections import deque

class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens.sort()
        token_deque = deque(tokens)
        best_score = 0
        current_score = 0
        current_power = P
        while token_deque:
            if token_deque[0] <= current_power:
                current_score += 1
                current_power -= token_deque.popleft()
            elif current_score > 0:
                current_score -= 1
                current_power += token_deque.pop()
            else:
                break
            best_score = max(best_score, current_score)
        return best_score
