class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens.sort()
        start = 0
        end = len(tokens) - 1
        points = 0
        maxpoints = 0
        while(start <= end):
            if(tokens[start] <= P): # always gain point when you can
                P = P - tokens[start]
                points = points + 1
                start = start + 1
            elif points == 0:
                break
            else:
                P = P + tokens[end]
                points = points - 1
                end = end - 1
            if points > maxpoints:
                maxpoints = points
        return maxpoints
