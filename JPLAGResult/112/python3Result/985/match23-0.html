class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        N=len(tokens)
        tokens.sort()
        if N==0:
            return 0
        if N==1:
            if P>=tokens[0]:
                return 1
            else:
                return 0
        start=0
        end=N
        t=0
        maxp=0
        while(start<end):
            while(start<end and P>=tokens[start]):
                P=P-tokens[start]
                t=t+1
                start=start+1
            maxp=max(maxp,t)
            if t>=1:
                t=t-1
                end=end-1
                P=P+tokens[end]
            else:
                break
        return maxp
