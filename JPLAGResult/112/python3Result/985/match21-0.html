class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens.sort()
        n = len(tokens)
        i = 0
        j = n - 1
        res = 0
        cur = 0
        while i <= j:
            ok = False
            while i <= j and P >= tokens[i]:
                cur += 1
                ok = True
                P -= tokens[i]
                i += 1
            res = max(res, cur)
            if i <= j and cur > 0:
                cur -= 1
                ok = True
                P += tokens[j]
                j -= 1
            if not ok:
                break
        return res
