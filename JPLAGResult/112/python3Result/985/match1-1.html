class Solution:
    def bagOfTokensScore(self, tokens, P):
        maxscore = score = 0
        tokens.sort()
        l, r = 0, len(tokens)-1
        while l <= r:
            if P >= tokens
[l]:
                score += 1
                maxscore = max(maxscore, score)
                P -= tokens[l]
                l += 1
            elif score > 0:
                score -= 1
                P += tokens[r]
                r -= 1
            else:
                break
        return maxscore
