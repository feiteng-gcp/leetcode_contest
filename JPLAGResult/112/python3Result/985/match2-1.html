class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        n = len(tokens)
        if n==0: return 0
        
        power = P
        points = 0
        ans = 0
        tokens.sort()
        
        i,j
=0,n-1
        while i<=j:
            if tokens[i]<=power:
                power -= tokens[i]
                points += 1
                ans = max(ans,points)
                i+=1
            elif points > 0:
                points -= 1
                power += tokens[j]
                j-=1
            else:
                return ans # out of moves
        return ans
