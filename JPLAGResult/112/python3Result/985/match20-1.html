class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens.sort()
        N = len(tokens)
        i, j  = 0, N-1
        res = 0
        while i < j:
            if tokens[i] <= P:
                res += 1
                P 
-= tokens[i]
                i += 1
            elif res > 0:
                res -= 1
                P += tokens[j]
                j -= 1
            else:
                break
        
        if N>0 and tokens[i] <= P:
            return res+1
        return res
