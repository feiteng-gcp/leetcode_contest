class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        if not tokens:
            return 0
        tokens = list(sorted(tokens))
        i = 0
        j = len(tokens)
        count = 0
        maxcount = 0
        while i < j:
            
if P - tokens[i] >= 0:
                P = P - tokens[i]
                count += 1
                maxcount = max(maxcount, count)
                i += 1
            elif count > 0:
                count -= 1
                j -= 1
                P += tokens[j]
            else:
                return maxcount
        return maxcount
