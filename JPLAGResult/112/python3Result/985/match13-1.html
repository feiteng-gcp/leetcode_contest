class Solution:
    def bagOfTokensScore(self, tokens, P):
        l = 0
        r = len(tokens)-1
        res = 0
        cur = 0
        tokens 
= sorted(tokens)
        while l<=r:
            if tokens[l]<=P:
                P-=tokens[l]
                cur+=1
                l+=1
            elif cur>0:
                cur-=1
                P+=tokens[r]
                r-=1
            else:
                break
            res = max(res,cur)
        return res
