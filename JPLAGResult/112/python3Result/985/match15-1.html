class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        
        if not tokens:
            return 0
        
        tokens
.sort()
        pos, neg, res = 0, 0, 0
        i, j = 0, len(tokens) - 1
        cur = P
        while i <= j:
            if cur >= tokens[i]:
                cur -= tokens[i]
                i += 1
            elif i - (len(tokens) - 1 - j) > 0:
                cur += tokens[j]
                j -= 1
            else:
                break           
            res = max(res, i - (len(tokens) - 1 - j))
            #print(i, j, cur, res)
        
        return res
