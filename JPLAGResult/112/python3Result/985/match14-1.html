class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens.sort()
        i = 0
        j = len(tokens) - 1
        score = 0
        res = 0
        while i <= j:
            while i < len(tokens) and P >= tokens[i] and i <= j:
                P -= tokens[i]
                i += 1
                score += 1
                res = max(
res, score)
            if score > 0 and i <= j:
                P += tokens[j]
                score -= 1
                j -= 1
            else:
                break
        return res
