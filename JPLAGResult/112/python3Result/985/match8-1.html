class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        max_score = 0
        score = 0
        tokens.sort()

        i = 0
        j = len(tokens)-1
        while i <= j:
            if P >= tokens[i]:
                score += 1
                P -= tokens[i]
                max_score 
= max(score, max_score)
                i += 1
            else:
                if score >= 1:
                    score -= 1
                    P += tokens[j]
                    j -= 1
                else:
                    break
        
        return max_score
