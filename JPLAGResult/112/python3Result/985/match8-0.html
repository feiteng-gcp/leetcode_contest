class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens = sorted(tokens)
        c=0
        start = 0
        stop  = len(tokens)-1
        local = 0
        while start <=stop:
            if P>=tokens[start]:
                P-=tokens[start]
                local+=1
                c= max(c,local)
                start+=1
            else:
                if (local>0):
                    local-=1
                    P+=tokens[stop]
                    stop-=1
                else:
                    break
        return c
