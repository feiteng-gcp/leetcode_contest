class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens = sorted(tokens)
        i = 0
        j = len(tokens) - 1
        max_score = 0
        cur_score = 0
        cur_power = P
        
while i <= j:
            if cur_power >= tokens[i]:
                cur_score += 1
                max_score = max(max_score, cur_score)
                cur_power -= tokens[i]
                i += 1
            elif cur_score >= 1:
                cur_score -= 1
                cur_power += tokens[j]
                j -= 1
            else:
                break
        
        return max_score
