class Solution:
    def bagOfTokensScore(self, tokens, P):
        """
        :type tokens: List[int]
        :type P: int
        :rtype: int
        """
        tokens = sorted(tokens)
        a = 0
        b = len(tokens)-1
        res = 0
        bar = 0
        while b >= a:
            # print(tokens[a], tokens[b], P, bar)
            if P >= tokens[a]:
                bar += 1
                P -= tokens[a]
                a += 1
                if bar > res:
                    res = bar
            else:
                
if bar > 0:
                    bar -= 1
                    P += tokens[b]
                    b -= 1
                else:
                    break
        return res
