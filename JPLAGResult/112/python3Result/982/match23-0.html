class Solution:
    def minIncrementForUnique(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        
        if not A:
            return 0
        
        A.sort()
        aMin = min(A)
        res, cur = 0, aMin
        
        for num in A:
            if num <= cur:
                res += cur - num
            else:
                cur = num
            cur += 1
        
        return res
