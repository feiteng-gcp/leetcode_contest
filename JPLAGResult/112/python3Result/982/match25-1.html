class Solution:
    def minIncrementForUnique(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if not A:
            return 0
        A1 = list(sorted
(A))
        count = 0
        last = A1[0]
        for i in range(1, len(A1)):
            if A1[i] <= last:
                count += last + 1 - A1[i]
                last += 1
            else:
                last = A1[i]
        return count
