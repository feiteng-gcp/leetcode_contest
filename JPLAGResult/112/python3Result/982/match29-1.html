class Solution:
    def minIncrementForUnique(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if not A: return 0
        A.sort()
        last = A[0] - 1
        ans = 0
        for a in A:
            if a <= last:
                ans += last + 1 - a
                last += 1
            else:
                last = a
        return ans
