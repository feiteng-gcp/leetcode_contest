class Solution:
    def minIncrementForUnique(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if not A:
            return 0
        A.sort()
        c = A[0]
        res = 0
        for a in A:
            if c < a:
                c = a
            res += (c-a)
            c += 1
        return res
