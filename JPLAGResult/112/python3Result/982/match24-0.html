class Solution:
    def minIncrementForUnique(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        if not A: return 0
        A.sort()
        cap = float('-inf')
        R = 0
        for a in A:
            if a <= cap:
                R += (cap + 1 - a)
                cap += 1
            else:
                cap = a
        return R
