class Solution:
    def minIncrementForUnique(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        ans = 0
        
        A.sort()
        seen = set()
        
        max_v = -1
        
        for num in A:
            if num > max_v:
                # seen.add(num)
                max_v = num
            else:
                ans = ans + (max_v + 1 - num)
                max_v = max_v + 1
                # seen.add(max_v)
        
        return ans
