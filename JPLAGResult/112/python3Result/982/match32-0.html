class Solution:
    def minIncrementForUnique(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        N = len(A)
        if N == 1:
            return 0
        A.sort()
        #print(A)
        res = 0
        for i in range(1,N):
            if A[i] <= A[i-1]:
                res += A[i-1] - A[i] + 1
                A[i] = A[i-1] + 1
        #print(A)
        return res
