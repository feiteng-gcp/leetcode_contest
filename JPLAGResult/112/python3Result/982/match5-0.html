class Solution:
    def minIncrementForUnique(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        A.sort()
        steps = 0
        for i in range(1, len(A)):
            if A[i] <= A[i - 1]:
                steps += A[i - 1] - A[i] + 1
                A[i] = A[i - 1] + 1
        return steps
