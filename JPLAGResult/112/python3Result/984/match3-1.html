class Solution:
    def removeStones(self, stones):
        """
        :type stones: List[List[int]]
        :rtype: int
        """
        n = len(stones)
        if n<=1: return 0
        
        byrow 
= {}
        bycol = {}
        for r,c in stones:
            if r not in byrow: byrow[r]=set()
            if c not in bycol: bycol[c]=set()
            byrow[r].add((r,c))
            bycol[c].add((r,c))
            
            
        def dfs(r,c,viz):
            cnt = 1
            for nr,nc in byrow[r]:
                if (nr,nc) in viz: continue
                viz.add((nr,nc))
                cnt += dfs(nr,nc,viz)
            for nr,nc in bycol[c]:
                if (nr,nc) in viz: continue
                viz.add((nr,nc))
                cnt += dfs(nr,nc,viz)
            return cnt
        
        
        count = 0
        ans = 0
        for r,c in stones:
            if (r,c) not in byrow[r]: continue
            count += 1
            viz=set()
            viz.add((r,c))
            ans += dfs(r,c,viz)
            for vr,vc in viz:
                byrow[vr].remove((vr,vc))
                bycol[vc].remove((vr,vc))
                
        return ans - count
