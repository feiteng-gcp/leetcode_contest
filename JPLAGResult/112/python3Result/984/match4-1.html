class Solution:
    def removeStones(self, stones):
        """
        :type stones: List[List[int]]
        :rtype: int
        """

        for i,s in enumerate(stones):
            rows[s[0]].append(i)
            cols[s[1]].append(i)
        v=[False]*n
        left=0
        for i in range(n):
            if not v[i]:
                v[i]=True
                left+=1
                b=[i]
                j=0
                while j<len(b):
                    x,y=stones[b[j]]
                    for p in rows[x]:
                        if not v[p]:
                            v[p]=True
                            b.append(p)
                    for p in cols[y]:
                        if not v[p]:
                            v[p]=True
                            b.append(p)                    
                    j+=1
        return n-left
