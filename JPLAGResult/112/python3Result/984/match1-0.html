class Solution:
    def removeStones(self, stones):
        """
        :type stones: List[List[int]]
        :rtype: int
        """
        def remove(x, y):
            dicx[x].remove(y)
            if len(dicx[x]) == 0:
                dicx.pop(x)
            dicy[y].remove(x)
            if len(dicy[y]) == 0:
                dicy.pop(y)
        def dfs(x, y, v):
            v.add((x, y))
            for nexty in dicx[x]:
                if (x, nexty) not in v:
                    dfs(x, nexty, v)
            for nextx in dicy[y]:
                if (nextx, y) not in v:
                    dfs(nextx, y, v)

        dicx = {}
        dicy = {}
        for x, y in stones:
            if x not in dicx:
                dicx[x] = set()
            if y not in dicy:
                dicy[y] = set()
            dicx[x].add(y)
            dicy[y].add(x)
        visited = 0
        count = 0
        while visited < len(stones):
            for x in dicx:
                for y in dicx[x]:
                    break
                break
            visited_set = set()
            dfs(x, y, visited_set)
            visited += len(visited_set)
            for x, y in visited_set:
                remove(x, y)
            count += 1
        return len(stones) - count
