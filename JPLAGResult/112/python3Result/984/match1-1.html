from collections import defaultdict

class Solution:
    def removeStones(self, stones):
        """
        :type stones: List[List[int]]
        :rtype: int
        """

        
def helper(stones, visit, s0, s1, mp0, mp1):
            visit.add((s0, s1))
            for s_1 in mp0[s0]:
                if (s0, s_1) not in visit:
                    helper(stones, visit, s0, s_1, mp0, mp1)
            for s_0 in mp1[s1]:
                if (s_0, s1) not in visit:
                    helper(stones, visit, s_0, s1, mp0, mp1)

        mp0, mp1 = defaultdict(set), defaultdict(set)
        for s in stones:
            mp0[s[0]].add(s[1])
            mp1[s[1]].add(s[0])
        visit = set()
        res = 0
        for s in stones:
            if (s[0], s[1]) not in visit:
                helper(stones, visit, s[0], s[1], mp0, mp1)
                res += 1
        return res
