class Solution:
    def removeStones(self, stones):
        """
        :type stones: List[List[int]]
        :rtype: int
        """
        rows = collections.defaultdict(list)
        cols = collections.defaultdict(list)
        for r, c in stones:
            rows[r].append(c)
            cols[c].append(r)

        ans = 0
        v = set()
        
        def dfs(r, c):
            v
.add((r, c))
            for nc in rows[r]:
                if (r, nc) in v: continue
                dfs(r, nc)
            for nr in cols[c]:
                if (nr, c) in v: continue
                dfs(nr, c) 
        
        for r, c in stones:
            if (r, c) in v: continue
            dfs(r, c)
            ans += 1
        return len(stones) - ans
