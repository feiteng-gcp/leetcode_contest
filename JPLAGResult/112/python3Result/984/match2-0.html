class Solution:
    def removeStones(self, stones):
        import collections
        Dict_x=collections.defaultdict(list)
        Dict_y=collections.defaultdict(list)
        for stone in stones:
            Dict_x[stone[0]].append(stone[1])
            Dict_y[stone[1]].append(stone[0])
        s=0
        def dfs(stone):
            x,y=stone
            for tar in Dict_x[x]:
                Dict_x[x].remove(tar)
                dfs([x,tar])
            for tar in Dict_y[y]:
                Dict_y[y].remove(tar)
                dfs([tar,y])
        for stone in stones:
            if(stone[1] in Dict_x[stone[0]]):
                s+=1
                dfs(stone)
        return len(stones)-s
