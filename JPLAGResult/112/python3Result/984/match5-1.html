from collections import defaultdict

class Solution:
    def removeStones(self, stones):
        """
        :type stones: List[List[int]]
        :rtype: int
        """

        xP = defaultdict(list)
        yP = defaultdict(list)
        for i in range(len(stones)):
            stones[i] = tuple(stones[i])


        for s in stones:
            xP
[s[0]].append(s)
            yP[s[1]].append(s)

        cC = 0
        seen = {}

        for s in stones:
            if s not in seen:
                cC += 1
                self.dfs(s, seen, cC, xP, yP)

        return len(stones) - cC

    def dfs(self, s, seen, cC, xP, yP):
        if s in seen:
            return

        seen[s] = cC

        for p in xP[s[0]]:
            if p not in seen:
                self.dfs(p, seen, cC, xP, yP)

        for p in yP[s[1]]:
            if p not in seen:
                self.dfs(p, seen, cC, xP, yP)
