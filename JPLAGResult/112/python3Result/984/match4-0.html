class Solution:
    def removeStones(self, stones):
        """
        :type stones: List[List[int]]
        :rtype: int
        """
        n = len(stones)
        byx = collections.defaultdict(list)
        byy = collections.defaultdict(list)
        for i, v in enumerate(stones):
            byx[v[0]].append(i)
            byy[v[1]].append(i)
        group = [0] * n
        ng = 0
        px = [False] * 10000
        py = [False] * 10000
        
        def dfs(node):
            q = [node]
            while q:
                cur = q.pop()
                group[cur] = ng
                cx = stones[cur][0]
                cy = stones[cur][1]
                if not px[cx]:
                    for j in byx[cx]:
                        if j != cur:
                            q.append(j)
                    px[cx] = True
                if not py[cy]:
                    for j in byy[cy]:
                        if j != cur:
                            q.append(j)
                    py[cy] = True
                    
        for i in range(n):
            if not group[i]:
                ng += 1
                dfs(i)
        return n - ng
