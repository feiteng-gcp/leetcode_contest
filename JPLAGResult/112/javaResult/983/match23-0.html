class Solution {
    public boolean validateStackSequences(int[] pushed, int[] popped) {
        Stack<Integer> stack = new Stack<Integer>();
        int length = pushed.length;
        int pushIndex = 0, popIndex = 0;
        while (pushIndex < length) {
            while (popIndex < length && !stack.empty() && stack.peek().intValue() == popped[popIndex]) {
                stack.pop();
                popIndex++;
            }
            int pushElement = pushed[pushIndex++];
            stack.push(pushElement);
        }
        while (popIndex < length && !stack.empty() && stack.peek().intValue() == popped[popIndex]) {
            stack.pop();
            popIndex++;
        }
        return stack.empty();
    }
}
