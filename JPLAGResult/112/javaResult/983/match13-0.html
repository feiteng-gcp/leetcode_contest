class Solution {
    public boolean validateStackSequences(int[] pushed, int[] popped) {
        int popi = 0;
        Stack<Integer> stack = new Stack<Integer>();
        
        for (int i : pushed) {
            stack.push(i);
            
            while (!stack.empty() && stack.peek() == popped[popi]) {
                
                //if (!stack.empty())
                    stack.pop();
                //else
                //    return false;
                popi++;
            }
        }
        
        return stack.empty();        
    }
}
