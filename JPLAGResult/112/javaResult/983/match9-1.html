class Solution {
    public boolean validateStackSequences(int[] pushed, int[] popped) {
        LinkedList<Integer> list = new LinkedList<>();
        int i = 0;
        for(int el: pushed) {
            list.push(el);
            while(i < popped.length && list.size()>0 && list.peek() == popped[i]) {
                ++i;
                list.pop();
            }
        }
        return list.isEmpty();
    }
}
