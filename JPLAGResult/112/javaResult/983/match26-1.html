class Solution {
	public boolean validateStackSequences(int[] pushed, int[] popped) {
		Deque<Integer> stack = new ArrayDeque<>();
		int n= pushed.length;
		int j = 0;
		for (int i = 0; i < n; ++i) {

			stack.push(pushed[i]);
			while (!stack.isEmpty() && popped[j] == stack.peek()) {
				stack.pop();
				j++;
			}
		}
		return j == n;
	}
}
