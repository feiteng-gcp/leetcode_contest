class Solution {
    public boolean validateStackSequences(int[] pushed, int[] popped) {
        Deque<Integer> stack = new LinkedList();
        int start = 0;
        for (int push : pushed) {
            stack.addFirst(push);
            while (!stack.isEmpty() && stack.peek() == popped[start]) {
                stack.poll();
                start++;
            }
        }
        return stack.isEmpty();
    }
}
