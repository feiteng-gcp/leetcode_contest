class Solution {
    public boolean validateStackSequences(int[] pushed, int[] popped) {
        Stack<Integer> stk = new Stack<>();
        int i = 0;
        for (int x1 : pushed) {
            stk.push(x1);
            while (i < popped.length && !stk.isEmpty() && popped[i] == stk.peek()) {
                stk.pop();
                i++;
            }
        }
        return i >= popped.length;
    }
}
