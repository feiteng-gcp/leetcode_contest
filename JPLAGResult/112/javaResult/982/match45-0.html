class Solution {
    public int minIncrementForUnique(int[] A) {
        Arrays.sort(A);
        int ans = 0;
        int cur = -1;
        for (int n : A) {
            if (n < cur) {
                ans += (cur - n);
                cur++;
            } else {
                cur = n + 1;
            }
        }
        return ans;
    }
}
