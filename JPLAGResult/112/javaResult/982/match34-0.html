class Solution {
    public int minIncrementForUnique(int[] A) {
        // find duplicate numbers
        // game of numbers?
        if(A.length <= 1) return 0;
        
        Arrays.sort(A);
        
        int moves = 0;
        int prev = A[0];
        for(int i = 1; i< A.length; i++) {
            if(prev >= A[i]){
                moves += prev- A[i] + 1;
                prev++;
            }
            else {
                prev = A[i];
            }
        }
        
        return moves;
    }
}
