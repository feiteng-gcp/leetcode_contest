import java.util.Arrays;


class Solution {
  public int minIncrementForUnique(int[] A) {
    Arrays.sort(A);

    int moves = 0;

    for (int i = 1; i < A.length; ++i) {
      int value = A[i];

      
if (A[i - 1] >= A[i]) {
        int move = A[i - 1] - A[i] + 1;
        moves += move;
        A[i] += move;
      }
    }

    return moves;
  }
}
