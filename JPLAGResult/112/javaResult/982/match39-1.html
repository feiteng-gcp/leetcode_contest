class Solution {
    public int minIncrementForUnique(int[] A) {
        Arrays.sort(A);
        int ans=0;
        int n=A.length;
        for (int i=1;i<n;++i) {
            if (A[i]<=A[i-1]) {
                int old=A[i];
                
A[i]=A[i-1]+1;
                ans+=A[i]-old;
            }
        }
        return ans;
    }
}
