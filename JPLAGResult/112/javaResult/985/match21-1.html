class Solution {
public int bagOfTokensScore(int[] tokens, int P) {
        int n = tokens.length;
        if (n == 0) return 0;
        
Arrays.sort(tokens);
        int l = 0;
        int r = n - 1;
        int point = 0;
        int curP = P;
        int ans = 0;
        while (l <= r) {
            boolean changed = false;
            while (l <= r && curP >= tokens[l]) {
                point++;
                curP -= tokens[l];
                l++;
                changed = true;
            }
            ans = Math.max(ans, point);
            if (l <= r && point > 0) {
                point--;
                curP += tokens[r];
                r--;
                changed = true;
            }
            if (!changed) break;
        }
        return ans;
    }
}
