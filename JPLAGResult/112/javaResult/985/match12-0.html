class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        if (tokens.length == 0) return 0;
        Arrays.sort(tokens);
        int lo = 0;
        int hi = tokens.length - 1;
        int res = 0;
        if (P < tokens[0])
            return 0;
        while (lo <= hi){
            if (P >= tokens[lo]){
                P -= tokens[lo];
                lo++;
                res++;
            }
            else{
                if (res > 0 && lo != hi){
                    res--;
                    P += tokens[hi];
                    hi--;
                }
                else{
                    break;
                }
            }
        }
        return res;
    }
}
