class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int left = 0;
        int right = tokens.length-1;
        int score = 0;
        int result = 0;
        int energy = P;
        while (left <= right) {
            boolean flag = false;
            while (left <= right && energy >= tokens[left]) {
                energy -= tokens[left];
                score += 1;
                left++;
                flag = true;
            }
            result = Math.max(result, score);
            if (score > 0) {
                energy += tokens[right];
                score -= 1;
                right--;
                flag = true;
            }
            if (!flag) break;
        }
        return result;
    }
}
