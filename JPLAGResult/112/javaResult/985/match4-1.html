class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        
int res = 0, l=0, r=tokens.length-1, power=P, point=0;;
        while(l<=r){
            if(power>=tokens[l]){
                power -= tokens[l];
                l++;
                point++;
            }else if(res>=1){
                power += tokens[r];
                r--;
                point--;
            }else{
                break;
            }
            res = Math.max(point, res);
        }
        return res;
    }
}
