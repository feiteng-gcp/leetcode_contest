class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int n = tokens.length;
        int point = 0;
        int max = 0;
        int h = 0, t = n - 1;
        while(h < t + 1) {
            if (P >= tokens[h]) {
                point += 1;
                P -= tokens[h];
                h++;
                if (point > max) {
                    max = point;
                }
            }else if (point > 0) {
                point -= 1;
                P += tokens[t];
                t--;
            }else {
                break;
            }
        }
        return max;
    }
}
