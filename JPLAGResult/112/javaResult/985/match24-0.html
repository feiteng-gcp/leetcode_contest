class Solution {
        public int bagOfTokensScore(int[] tokens, int P) {
            Arrays.sort(tokens);
            int i = 0, j = tokens.length - 1;
            int powers = P;
            int maxP = 0, points = 0;
            while (i <= j) {
                if (powers >= tokens[i]) {
                    powers -= tokens[i ++];
                    points ++;
                    if (points > maxP) {
                        maxP = points;
                    }
                } else if (points >= 1) {
                    powers += tokens[j--];
                    points --;
                } else {
                    break;
                }
            }
            return maxP;
        }
    }
