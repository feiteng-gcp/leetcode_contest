class Solution {
    int[] tokens;
    int maxPoint = 0;
    
public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        // this.tokens = tokens;
        int n = tokens.length;
        // dfs(0, n - 1, P, 0);
        // return maxPoint;
        int i = 0, j = n - 1;
        int points = 0;
        while(i < j) {
            while(i < j && P >= tokens[i]) {
                P -= tokens[i];
                i ++;
                points ++; 
            } 
            if(i == j) break;
            
if(points >= 1) {
                P += tokens[j];
                j --;
                points --;
            } else {
                break;
            }
        }
        if(i == j) {
            if(P >= tokens[i]) points ++;
        }
        return points;
    }
