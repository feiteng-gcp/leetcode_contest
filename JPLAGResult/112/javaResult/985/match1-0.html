class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int point = 0;
        int coin = tokens.length - 1;
        int maxScore = 0;
        int score = 0;
        while (point <= coin) {
            if (P >= tokens[point]) {
                P-= tokens[point];
                score += 1;
                point ++;
            } else if (point > 0) {
                P += tokens[coin];
                coin --;
                score -= 1;
            } else {
                break;
            }
            maxScore = Math.max(maxScore, score);
        }
        return maxScore;
    }
}
