class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        int power = P;
        int score = 0;
        int maxScore = 0;
        Arrays.sort(tokens);
        if (tokens == null || tokens.length == 0 || power < tokens[0])
            return maxScore;
        List<Integer> tokensList = new ArrayList<Integer>();
        int length = tokens.length;
        for (int i = 0; i < length; i++)
            tokensList.add(tokens[i]);
        while (!tokensList.isEmpty()) {
            if (!tokensList.isEmpty() && score > 0) {
                power += tokensList.remove(tokensList.size() - 1);
                score--;
            }
            while (!tokensList.isEmpty() && power >= tokensList.get(0)) {
                power -= tokensList.remove(0);
                score++;
                maxScore = Math.max(maxScore, score);
            }
        }
        return maxScore;
    }
}
