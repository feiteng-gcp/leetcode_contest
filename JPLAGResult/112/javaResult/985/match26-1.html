class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int n = tokens.length;
        if(n == 0) {
            return 0;
        }
        
int i = 0, j = n - 1;
        int res = 0;
        
        while(i < j) {
            if(tokens[i] <= P) {
                P -= tokens[i];
                i++;
                res++;
            } else if(res > 0) {
                // gain power, lose points
                P += tokens[j];
                res--;
                j--;
            } else {
                return res;
            }
        }
        
        if(tokens[i] <= P) {
            res++;
            P -= tokens[i];
        }
        
        return res;
        
    }
}
