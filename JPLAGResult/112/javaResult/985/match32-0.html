class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        
        Arrays.sort(tokens);
        
        int score = 0, maxScore = 0;
        int lo = 0, hi = tokens.length - 1;
        
        while(lo <= hi){
            while(lo <= hi && P >= tokens[lo]){
                P -= tokens[lo];
                ++score;
                ++lo;
            }
            
            maxScore = Math.max(maxScore, score);
            
            if(lo <= hi && score >= 1){
                P += tokens[hi];
                --score;
                --hi;
            }else{
                break;
            }
        }
        
        return maxScore;
    }
}
