class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int p1 = 0, p2 = tokens.length-1;
        int point = 0;
        int max = 0;
        while(p1 <= p2){
            if(P >= tokens[p1]){
                P -= tokens[p1];
                point++;
                p1++;
            
}else if(point > 0){
                P += tokens[p2];
                point--;
                p2--;
            }else{
                break;
            }
            max = Math.max(max, point);
        }
        return max;
    }
}
