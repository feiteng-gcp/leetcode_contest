class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        if (tokens.length == 0)
            return 0;
        Arrays.sort(tokens);
        int score = 0;
        int i = 0, j = tokens.length - 1;
        while (i < j) {
            if (tokens[i] <= P) {
                P -= tokens[i++];
                score++;
            } else if (score == 0) {
                return 0;
            } else {
                P += tokens[j--];
                score--;
            }
        }
        if (tokens[i] <= P)
            score++;
        return score;
    }
}
