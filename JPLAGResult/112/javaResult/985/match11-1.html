class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {        
        if(tokens.length == 0) return 0;
        Arrays.sort(tokens);  
        int i = 0;
        int j = tokens.length - 1;
        int maxPoints = 0;
        int points = 0;       
        while(i <= j){
            if(P >= tokens[i]){
                P -= tokens[i];
                points++;
                maxPoints = Math.max(maxPoints, points);
                i++;
            } else {
                if(points > 0){
                    P += tokens[j];
                    points--;
                } else {
                    break;
                }
                j--;
            }
        }
        return maxPoints;
    }
}
