class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int i=0, j = tokens.length - 1;
        int score = 0, best = 0;
        while(i <= j) {
            if(tokens[i] <= P) {
                P-=tokens[i++];
                ++score;
                if(score > best)
                    best = score;
            
} else if(score > 0) {
                P+=tokens[j--];
                --score;                
            } else {
                break;
            }
        }
        return best;
    }
}
