class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int max = 0;
        int point = 0;
        int front = 0;
        int end = tokens.length - 1;
        while (front < tokens.length && front <= end) {
            while (front < tokens.length && tokens[front] <= P) {
                P -= tokens[front];
                front++;
                point++;
                max = Math.max(point, max);
            }
            if (front > end) break;
            if (point != 0) {
                point--;
                P += tokens[end];
                end--;
                max = Math.max(point, max);
            
} else break;
        }
        return max;
    }
}
