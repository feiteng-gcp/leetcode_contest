class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int i = 0;
        int j = tokens.length-1;
        int res = 0;
        int pow = P;
        int points = 0;
        while(i <= j) {
        	if(pow >= tokens[i]) {
        		pow -= tokens[i++];
        		points++;
        	}else if(points > 0){
        		pow += tokens[j--];
        		points--;
        	}else {
        		break;
        	}
        	res = Math.max(res, points);
        }
        return res;
    }
}
