class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int ret = 0, score = 0;
        int n = tokens.length;
        int i = 0, j = n - 1;
        while(j >= i) {
            if (tokens[i] <= P) {
                P -= tokens[i];
                score += 1;
                i += 1;
                if (score > ret) {
                    ret = score;
                }
            } else if (score > 0) {
                score -= 1;
                P += tokens[j];
                j -= 1;
            } else {
                break;
            }
        }
        return ret;
    }
}
