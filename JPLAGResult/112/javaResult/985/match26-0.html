class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int power = P;
        int points = 0;
        int a = 0;
        int b = tokens.length - 1;
        
        while (a < b) {
            if (tokens[a] < power) {
                power -= tokens[a];
                points++;
                a++;
            } else if (points > 0) {
                points--;
                power += tokens[b];
                b--;
            } else {
                return points;
            }
        }
        if (a == b && tokens[a] <= power) {
            points++;
        }
        return points;
    }
}
