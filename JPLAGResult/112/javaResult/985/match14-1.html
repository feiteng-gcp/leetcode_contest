class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        
int left = 0, right = tokens.length;
        int res = 0, point = 0;
        while (left < right) {
            if (P >= tokens[left]) {
                point++;
                P -= tokens[left++];
            } else if (res != 0) {
                point--;
                P += tokens[--right];
            } else {
                break;
            }
            res = Math.max(res, point);
        }
        return res;
    }
}
