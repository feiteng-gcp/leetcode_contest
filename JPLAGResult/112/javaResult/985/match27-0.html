class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        if(tokens == null || tokens.length == 0) return 0;
        int res = 0;
        Arrays.sort(tokens);
        int i = 0;
        int j = tokens.length-1;
        while(i<=j){
            if(P >= tokens[i]){
                P -= tokens[i];
                i++;
                res++;
            }else{
                if(res > 0 && i < j){
                    P += tokens[j];
                    res--;
                    j--;
                }else{
                    break;
                }
            }
        }
        return res;
    }
}
