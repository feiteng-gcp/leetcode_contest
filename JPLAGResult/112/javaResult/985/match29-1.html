class Solution {
    public int bagOfTokensScore(int[] tokens, int P) {
        Arrays.sort(tokens);
        int N = tokens.length;
        int start = 0;
        int end = N - 1;
        int count = 0;
        int max = 0;
        while (start <= end) {
            if (P < tokens[start]) {
                if (count == 0) {
                    return 0;
                } else {
                    count--;
                    P += tokens[end--];
                
}
            } else {
                count++;
                max = Math.max(max, count);
                P -= tokens[start++];
            }
        }
        return max;
    }
}
