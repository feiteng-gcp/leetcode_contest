class Solution {
    class UnionFind {
        int[] ids;
        int[] szs;
        int size;
        public UnionFind(int size) {
            this.size = size;
            ids = new int[size];
            szs = new int[size];
            for (int i = 0; i < size; i++) {
                ids[i] = i;
                szs[i] = 1;
            }
        }
        
        public boolean find(int a, int b) {
            return getRoot(a) == getRoot(b);
        }
        
        private int getRoot(int a) {
            int cur = a;
            while(cur != ids[cur]) {
                ids[cur] = ids[ids[cur]];
                cur = ids[cur];
            }
            ids[a] = cur;
            return cur;
        }
        
        public void union(int a, int b) {
            int A = getRoot(a);
            int B = getRoot(b);
            if (szs[A] > szs[B]) {
                ids[B] = A;
                szs[A] += szs[B];
            } else {
                ids[A] = B;
                szs[B] += szs[A];
            }
            this.size--;
        }
    }
    public int removeStones(int[][] stones) {
        int row = stones.length;
        int col = stones[0].length;
        UnionFind uf = new UnionFind(stones.length);
        HashMap<Integer, Set<Integer>> xMap = new HashMap<>();
        HashMap<Integer, Set<Integer>> yMap = new HashMap<>();
        for (int i = 0; i < stones.length; i++) {
            int x = stones[i][0];
            int y = stones[i][1];
            if (!xMap.containsKey(x)) xMap.put(x, new HashSet<>());
            if (!yMap.containsKey(y)) yMap.put(y, new HashSet<>());
            xMap.get(x).add(i);
            yMap.get(y).add(i);
            for (int idx:xMap.get(x)) {
                if (!uf.find(idx,i)) uf.union(idx,i);
            }
            for (int idx:yMap.get(y)) {
                if (!uf.find(idx,i)) uf.union(idx,i);
            }
        }
        return stones.length - uf.size;
    }
}
