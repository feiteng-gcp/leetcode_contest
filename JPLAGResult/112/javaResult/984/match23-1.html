class Solution {
    public int removeStones(int[][] stones) {
        Map<Integer, List<Integer>> rowToCol = new HashMap<>();
        Map<Integer, List<Integer>> colToRow = new HashMap<>();
        for (int[] stone: stones) {
            
if (!rowToCol.containsKey(stone[0])) rowToCol.put(stone[0], new ArrayList<>());
            if (!colToRow.containsKey(stone[1])) colToRow.put(stone[1], new ArrayList<>());
            rowToCol.get(stone[0]).add(stone[1]);
            colToRow.get(stone[1]).add(stone[0]);
        }
        int count = 0;
        Set<Integer> row = new HashSet<>();
        Set<Integer> col = new HashSet<>();
        for (int[] stone: stones) {
            if (!row.contains(stone[0]) && !col.contains(stone[1])) {
                count++;
                dfs(rowToCol, stone[0], colToRow, true, row, col);
                dfs(colToRow, stone[1], colToRow, false, row, col);
            }
        }
        return stones.length-count;
    }
    
    private void dfs(Map<Integer, List<Integer>> rowToCol, int i, Map<Integer, List<Integer>> colToRow, boolean isRow, Set<Integer> row, Set<Integer> col) {
        if (isRow) {
            row.add(i);
            
for (int j: rowToCol.get(i)) {
                if (!col.contains(j)) dfs(rowToCol, j, colToRow, false, row, col);
            }
        } else {
            col.add(i);
            
for (int j: colToRow.get(i)) {
                if (!row.contains(j)) dfs(rowToCol, j, colToRow, true, row, col);
            }
        }
    }
}
