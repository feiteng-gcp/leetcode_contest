class Solution {
    public int removeStones(int[][] stones) {
        int g = 1;
        int[] v = new int[stones.length];
        for (int i = 0; i < stones.length; i++) {
            if (v[i] != 0) continue;
            dfs(stones, v, stones[i], g);
            g++;
        
}
        //for (int x : v)System.out.print(x +",  ");
        return stones.length - (g - 1);
    }
    private void dfs(int[][] stones, int[] v, int[] p, int g) {
        for (int i = 0; i < stones.length; i++) {
            
if (v[i] == 0 && (stones[i][0] == p[0] || stones[i][1] == p[1])) {
                v[i] = g;
                dfs(stones, v, stones[i], g);
            }
        }
    }
}
