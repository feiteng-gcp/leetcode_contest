class Solution {
    public int removeStones(int[][] stones) {
        Map<Integer,int[]> col=new HashMap<Integer,int[]>();
        Map<Integer,int[]> row=new HashMap<Integer,int[]>();
        Map<int[],int[]> parent=new HashMap<int[],int[]>();
        Map<int[],Integer> rank=new HashMap<int[],Integer>();
        for(int i=0;i<stones.length;i++){
            int[] curr=stones[i];
            int r=curr[0];
            int c=curr[1];
            parent.put(curr,curr);
            rank.put(curr,0);
            if(row.containsKey(r)){
                merge(row.get(r),curr,parent,rank);
            }
            else{
                row.put(r,curr);
            }
            if(col.containsKey(c)){
                merge(col.get(c),curr,parent,rank);
            }
            else{
                col.put(c,curr);
            }
        }
        int count=0;
        for(int[] key:parent.keySet()){
            if(key==parent.get(key)){
                count++;
            }
        }
        return stones.length-count;
    }
    public int[] findParent(int[] s,Map<int[],int[]> parent){
        if(parent.get(s)!=s){
            int[] p=parent.get(s);
            p=findParent(p,parent);
            parent.put(s,p);
        }
        return parent.get(s);
    }
    public void merge(int[] s1,int[] s2,Map<int[],int[]> parent,Map<int[],Integer> rank){
        int[] p1=findParent(s1,parent);
        int[] p2=findParent(s2,parent);
        if(p1==p2){
            return ;
        }
        int r1=rank.get(p1);
        int r2=rank.get(p2);
        if(r1==r2){
            parent.put(p1,p2);
            rank.put(p2,r2+1);
        }
        else if(r1<r2){
            parent.put(p1,p2);
        }
        else{
            parent.put(p2,p1);
        }
    }
}
