class Solution {
    class UnionFind {
    private int[] parent;
    private int[] sz;

    // 0 ... n - 1
    public UnionFind(int n) {
        
parent = new int[n];
        sz = new int[n];
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            sz[i] = 1;
        }
    }

    public int find(int x) {
        return parent[x] == x ? x : (parent[x] = find(parent[x]));
    
}

    public void union(int x, int y) {
        int px = find(x);
        int py = find(y);
        if (px == py) return;
        parent[py] = px;
        sz[px] += sz[py];
    }

    public int size(int x) {
        return sz[find(x)];
    
}
}
    
public int removeStones(int[][] stones) {
        int n = stones.length;

        Map<Integer, Set<Integer>> row = new HashMap<>();
        Map<Integer, Set<Integer>> col = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int x = stones[i][0];
            int y = stones[i][1];
            row.computeIfAbsent(x, k -> new HashSet<>()).add(i);
            col.computeIfAbsent(y, k -> new HashSet<>()).add(i);
        }

        //boolean[] vst = new boolean[n];
        UnionFind uf = new UnionFind(n);
        for (int i = 0; i < n; i++) {
//            if (vst[i]) continue;
//            vst[i] = true;
            int x = stones[i][0];
            int y = stones[i][1];

            Iterator<Integer> it = row.get(x).iterator();
            while (it.hasNext()) {
                int k = it.next();
                uf.union(i, k);
                it.remove();
            }

            it = col.get(y).iterator();
            while (it.hasNext()) {
                int k = it.next();
                uf.union(i, k);
                it.remove();
            }
        }

        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (uf.find(i) == i) {
                if (uf.size(i) > 1) {
                    ans += uf.size(i) - 1;
                }
            }
        }
        return ans;
    }

}
