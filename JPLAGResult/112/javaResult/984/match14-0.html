class Solution {
    public int removeStones(int[][] stones) {
        if (stones.length == 1)
            return 0;
        HashSet<Integer> rowSet = new HashSet(), colSet = new HashSet();
        HashSet<int[]> set = new HashSet();
        int islands = 0;
        for (int i = 0; i < stones.length; i++ ) {
            int[] a = stones[i];
            if (!set.contains(a)) {
                islands++;
                dfs(a, set, stones);
            }
            
        }
        
        return stones.length - islands;
    }
    
    private void dfs(int[] a, HashSet<int[]> set, int[][] stones) {
        if (set.contains(a)) {
            return;
        }
        
        set.add(a);
        
        for (int i = 0; i < stones.length; i++ ) {
            if (stones[i][0] == a[0] || stones[i][1] == a[1]) {
                dfs(stones[i], set, stones);
            }
        }
    }
}
