class Solution {
    public int removeStones(int[][] stones) {
        ArrayList<Integer>[] x = new ArrayList[10001];
        ArrayList<Integer>[] y = new ArrayList[10001];
        boolean[][] graph = new boolean[stones.length][stones.length];
        for(int i = 0; i < stones.length; i++){
            if(x[stones[i][0]] == null)
                x[stones[i][0]] = new ArrayList<>();
            if(y[stones[i][1]] == null)
                y[stones[i][1]] = new ArrayList<>();
            for(int j = 0; j < x[stones[i][0]].size(); j++){
                graph[i][x[stones[i][0]].get(j)] = true;
                graph[x[stones[i][0]].get(j)][i] = true;
            }
            x[stones[i][0]].add(i);
            for(int j = 0; j < y[stones[i][1]].size(); j++){
                graph[i][y[stones[i][1]].get(j)] = true;
                graph[y[stones[i][1]].get(j)][i] = true;
            }
            y[stones[i][1]].add(i);
        }
        /*for(int i = 0; i < stones.length; i++)
            for(int j = 0; j < stones.length; j++)
                System.out.println(i + " " + j + " " + graph[i][j]);*/
        int cc = findConnectComponents(graph);
        return stones.length - cc;
    }
    int findConnectComponents(boolean[][] graph){
        boolean[] visited = new boolean[graph.length];
        int cc = 0;
        for(int i = 0; i < graph.length; i++){
            if(!visited[i]){
                cc++;
                DFSUtil(graph, i, visited);
            }
        }
        return cc;
    }
    void DFSUtil(boolean[][] graph, int v, boolean visited[]){ 
        visited[v] = true; 
        for(int i = 0; i < graph.length; i++)
            if(graph[v][i] && !visited[i]) 
                DFSUtil(graph, i, visited);
    } 
}
