class Solution {
    int[] sts;
    int[] g;
    int find(int x) {
        if (g[x]<0) {
            return x;
        
}
        return g[x]=find(g[x]);
    }
    int join(int x, int y) {
        x=find(x);
        y=find(y);
        if (x==y) return x;
        if (g[x]<g[y]) {
            int t=x;
            x=y;
            y=t;
        }
        g[y]+=g[x];
        g[x]=y;
        return y;
    }
    public int removeStones(int[][] stones) {
        int n=stones.length;
        g= new int[n];
        Arrays.fill(g, -1);
        
for (int i=0;i<n;++i) {
            for (int j=i+1;j<n;++j) {
                if (stones[i][0]==stones[j][0]||stones[i][1]==stones[j][1]) {
                    join(i, j);
                }
            }
        }
        int ans=0;
        for (int i=0;i<n;++i) {
            if (g[i]<-1) {
                ans+=-g[i]-1;
            }
        }
        return ans;
    }

    public int removeStones1(int[][] stones) {
        int n=stones.length;
        int[] rn=new int[10010];
        int[] cn=new int[10010];
        for (int[] s: stones){
            ++rn[s[0]];
            ++cn[s[1]];
        }
        boolean[] vis = new boolean[10000];
        int ans=0;
        for (int i=0;i<n;++i) {
            int k=-1;
            int deg=20020;
            for (int j=0;j<n;++j) {
                if (vis[j]) continue;
                int jdeg = rn[stones[j][0]]+cn[stones[j][1]];
                if (jdeg<deg&&jdeg>=3) {
                    deg=jdeg;
                    k=j;
                }
            }
            if (k==-1) {
                break;
            }
            ++ans;
            vis[k]=true;
            --rn[stones[k][0]];
            --cn[stones[k][1]];
        }
       
        return ans;
    }
}
