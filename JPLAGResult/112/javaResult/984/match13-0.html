class Solution {
    // x x
    // x   x
    //   x x
    
    // x   x
    //   x
    // x   x
    // int[][] dirs = {{1, 0}, {-1, 0} , {0, 1}, {0, -1}};
    // int m = 0;
    // int n = 0;
    
    public int removeStones(int[][] stones) {
        Map<Integer, Set<Integer>> rowMap = new HashMap<>();
        Map<Integer, Set<Integer>> colMap = new HashMap<>();
        Set<String> visited = new HashSet<>();
        int clusters = 0;
        for(int[] stone : stones) {
            int r = stone[0];
            // m = Math.max(m, r);
            int c = stone[1];
            // n = Math.max(n, c);
            
            if(!rowMap.containsKey(r)) {
                rowMap.put(r, new HashSet<>());
            }
            rowMap.get(r).add(c);
            
            if(!colMap.containsKey(c)) {
                colMap.put(c, new HashSet<>());
            }
            colMap.get(c).add(r);
        }
        
        for(int r : rowMap.keySet()) {
            for(int c : rowMap.get(r)) {
                // start from r, c
                
                if(visited.contains(r + " " + c)) {
                    continue;
                }
                
                clusters++;
                dfs(rowMap, colMap, r, c, visited);
            }
        }
        
        return stones.length -  clusters;
    }
    
    private void dfs(Map<Integer, Set<Integer>> rowMap, Map<Integer, Set<Integer>> colMap, int r, int c, Set<String> visited) {
        if(visited.contains(r + " " + c)) {
            return;
        }


        visited.add(r + " " + c);

        for(int nextc : rowMap.get(r)) {
            dfs(rowMap, colMap, r, nextc, visited);
        }
        for(int nextr : colMap.get(c)) {
            dfs(rowMap, colMap, nextr, c, visited);
        }
    }
}
