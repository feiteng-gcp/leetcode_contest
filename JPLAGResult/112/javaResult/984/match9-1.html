class Solution {
    public int removeStones(int[][] stones) {
        int N = stones.length;
        if (N == 1) {
            return 0;
        }
        int[] root = 
new int[N];
        for (int i = 0; i < N; i++) {
            root[i] = i;
        }
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < i; j++) {
                if (stones[i][0] == stones[j][0] || stones[i][1] == stones[j][1]) {
                    union(i, j, root);
                }
            }
        }
        Set<Integer> rs = new HashSet<>();
        for (int i = 0; i < N; i++) {
            rs.add(getRoot(i, root));
        }
        return N - rs.size();
    }

    private void union(int i, 
int j, int[] root) {
        int ri = getRoot(i, root);
        int rj = getRoot(j, root);
        if (ri < rj) {
            root[rj] = ri;
        } else {
            root[ri] = rj;
        }
    }

    private int getRoot(int i, 
int[] root) {
        List<Integer> list = new ArrayList<>();
        while (root[i] != i) {
            list.add(i);
            i = root[i];
        }
        for (int ii : list) {
            root[ii] = i;
        }
        return i;
    }
}
