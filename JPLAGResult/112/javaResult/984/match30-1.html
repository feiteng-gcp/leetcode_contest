class Solution {
    public int removeStones(int[][] stones) {
        Map<Integer, List<Integer>> xmap = new HashMap();
        Map<Integer, List<Integer>> ymap = new HashMap();
        
        for (int[] stone : stones) {
            if (!xmap.containsKey(stone[0]))
                xmap.put(stone[0], new ArrayList());
            xmap.get(stone[0]).add(stone[1]);
            if (!ymap.containsKey(stone[1]))
                ymap.put(stone[1], new ArrayList());
            ymap.get(stone[1]).add(stone[0]);
        }
        
        
Set<String> set = new HashSet();
        int res = 0;
        for (int[] stone : stones) {
            String next = stone[0] + "#" + stone[1];
            if (!set.contains(next)) {
                setRemove(xmap, ymap, set, stone[0], stone[1]);
                res++;
            }
        }
        
        return stones.length - res;
    }
    
    private void setRemove(Map<Integer, List<Integer>> xmap, Map<Integer, List<Integer>> ymap, Set<String> set, int r, int c) {
        for (int y : xmap.get(r)) {
            String next = r + "#" + y;
            if (!set.contains(next)) {
                set.add(next);
                setRemove(xmap, ymap, set, r, y);
            }
        }
        
        for (int x : ymap.get(c)) {
            String next = x + "#" + c;
            if (!set.contains(next)) {
                set.add(next);
                setRemove(xmap, ymap, set, x, c);
            }
        }
    }
}
