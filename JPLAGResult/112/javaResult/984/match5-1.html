class Solution {
    
        int m = 10000;
    private int toX(int a, int b) {
        return a * m + b;
    }
    
    public int removeStones(int[][] stones) {
        int k = stones.length;
        UnionFind uf = new UnionFind(k);

        Map<Integer, Integer> map = new HashMap<>();
        int d = 0;
        for (int i = 0; i < k; i++) {
            map.put(toX(stones[i][0], stones[i][1]), d++);
        }

        for (int i = 0; i < k; i++) {
            for (int j = i + 1; j < k; j++) {
                if (stones[i][0] == stones[j][0] || stones[i][1] == stones[j][1]) {
                    uf.union(map.get(toX(stones[i][0], stones[i][1])), map.get(
toX(stones[j][0], stones[j][1])));
                }
            }
        }
        return k - uf.count;
    }

    class UnionFind{
        int[] root;
        
int[] size;
        int n;
        int count;

        public UnionFind(int n) {
            this.n = n;
            root = new int[n];
            size = new int[n];
            for (int i = 0; i < n; i++) {
                root[i] = i;
            }
            count = n;
            Arrays.fill(size, 1);
        }

        public int find(int a) {
            if (a == root[a]) {
                return a;
            }
            root[a] = find(root[a]);
            
return root[a];
        }

        public void union(int a, int b) {
            int rootA = find(a);
            int rootB = find(b);
            if (rootA != rootB) {
                root[rootA] = rootB;
                size[rootB] += size[rootA];
                count--;
            }
        }

        public int size(int a) {
            return size[find(a)];
        }
    }
}
