class Solution {
    public int removeStones(int[][] stones) {
        if (stones == null || stones.length == 0) {
            return 0;
        }
        int n = stones.length;
        Map<Integer, List<Integer>> rows = new HashMap<>();
        Map<Integer, List<Integer>> cols = new HashMap<>();
        for (int[] s: stones) {
            List<Integer> rList = rows.getOrDefault(s[0], new ArrayList<>());
            rList.add(s[1]);
            rows.put(s[0], rList);
            List<Integer> cList = cols.getOrDefault(s[1], new ArrayList<>());
            cList.add(s[0]);
            cols.put(s[1], cList);
        }
        int count = 0;
        Set<String> set = new HashSet<>();
        for (int[] s: stones) {
            if (!set.contains(s[0] + "," + s[1])) {
                dfs(s, rows, cols, set);
                count++;
            }
        }
        return count;
        
    }
    
    private void dfs(int[] start, Map<Integer, List<Integer>> rows, Map<Integer, List<Integer>> cols, Set<String> set) {
        if (set.contains(start[0] + "," + start[1])) return;
        set.add(start[0] + "," + start[1]);
        for (int col : rows.get(start[0])) {
            dfs(new int[]{start[0], col}, rows, cols, set);
        }
        for (int row : cols.get(start[1])) {
            dfs(new int[]{row, start[1]}, rows, cols, set);
        }
    }
}
