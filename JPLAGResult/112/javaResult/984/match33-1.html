class Solution {
    public int removeStones(int[][] stones) {
        int num = stones.length;
        if (num == 0) return 0;
        int rows = 0, cols = 0;
        for (int[] coord: stones) {
            rows = Math.max(rows, coord[0]);
            cols = Math.max(cols, coord[1]);
        }
        ++rows; ++cols;
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int i = 0; i < num; ++i) {
            for (int j = i+1; j < num; ++j) {
                int[] st1 = stones[i], st2 = stones[j];
                if (st1[0] == st2[0] || st1[1] == st2[1]) {
                    map.computeIfAbsent(i, x -> new HashSet<Integer>()).add(j);
                    map.computeIfAbsent(j, x -> new HashSet<Integer>()).add(i);
                }
            }
        }
        
        
int cnt = 0;
        Set<Integer> vis = new HashSet<>();
        for (int i = 0; i < num; ++i) {
            if (!vis.contains(i)) {
                cnt += 
dfs(i, stones, vis, map) - 1;
            }
        }
        
        return cnt;
    }
    
    int dfs(int id, int[][] stones, Set<Integer> vis, Map<Integer, Set<Integer>> map) {
        if (vis.contains(id)) return 0;
        int cnt = 1;
        Set<Integer> nSet = map.getOrDefault(id, new HashSet<Integer>());
        vis.add(id);
        for (int x: nSet) {
            cnt += dfs(x, stones, vis, map);
        }
        
        return cnt;
    }
}
