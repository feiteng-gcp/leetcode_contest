class Solution {
    public int removeStones(int[][] stones) {
        if(stones.length <= 1) return 0;
        
        int n = stones.length;
        
int k = 0;
        Set<int[]> visited = new HashSet();
        for(int[] s: stones){
            // find connected
            if(!visited.contains(s)){
                k++;
                visited.add(s);
                
dfs(s, stones, visited);
            }
        }
        
        return n - k;
    }
    
    void dfs(int[] curr, int[][] stones, Set<int[]> visited){
        for(int[] next: stones){
            // find connected
            if(curr[0] == next[0] || curr[1] == next[1]){
                if(!visited.contains(next)){
                    visited.add(next);
                    dfs(next, stones, visited);
                }
            }
        }
    }
}
