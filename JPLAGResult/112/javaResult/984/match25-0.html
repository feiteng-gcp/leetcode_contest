class Solution {
    public int removeStones(int[][] stones) {
        int res = 0;
        Map<Integer, Set<Integer>> rows = new HashMap<>();
        Map<Integer, Set<Integer>> columns = new HashMap<>();
        for (int[] stone : stones) {
            rows.putIfAbsent(stone[0], new HashSet<>());
            rows.get(stone[0]).add(stone[1]);
            columns.putIfAbsent(stone[1], new HashSet<>());
            columns.get(stone[1]).add(stone[0]);
        }
        
        for (int[] stone: stones) {
            if (rows.get(stone[0]).size() != 0 && columns.get(stone[1]).size() != 0) {
                res++;
                dFS(rows, columns, stone[0], stone[1]);
            }
        }
        
        return stones.length - res;
    }
    
    public void dFS(Map<Integer, Set<Integer>> rows, Map<Integer, Set<Integer>> columns, int i, int j) {
        rows.get(i).remove(j);
        columns.get(j).remove(i);
        Set<Integer> s1 = new HashSet<>(rows.get(i));
        Set<Integer> s2 = new HashSet<>(columns.get(j));
        for (int c : s1) {
            dFS(rows, columns, i, c);
        }
        
        for (int r : s2) {
            dFS(rows, columns, r, j);
        }
        
        return;
    }
}
