class Solution {
    private class UF {
        private int[] root;
        private int[] size;
        private int N;
        public UF(int n) {
            N = n;
            root = new int[N];
            size = new int[N];
            for (int i = 0; i < N; i++) {
                root[i] = i;
                size[i] = 1;
            }
        }
        public int find(int p) {
            while (p != root[p]) p = root[p];
            return p;
        }
        public void union(int p, int q) {
            int rp = find(p);
            int rq = find(q);
            if (rp == rq) return;
            N--;
            if (size[rp] < size[rq]) {
                size[rq] += size[rp];
                root[rp] = rq;
            }
            else {
                size[rp] += size[rq];
                root[rq] = rp;
            }
        }
    }
    public int removeStones(int[][] stones) {
        int n = stones.length;
        UF uf = new UF(n);
        for (int i = 0; i < n; i++) {
            int x = stones[i][0], y = stones[i][1];
            for (int j = 0; j < i; j++) {
                if (stones[j][0] == x || stones[j][1] == y) {
                    uf.union(i, j);
                }
            }
        }
        return n - uf.N;
    }
}
