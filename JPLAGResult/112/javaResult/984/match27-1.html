class Solution {
    List<Integer> adj[];
    boolean[] vis;
    
    public int removeStones(int[][] stones) {
        int n = stones.length; 
        adj = new ArrayList[n];
        vis = new boolean[n];
        for (int i = 0; i < n; i++) {
            adj[i] = new ArrayList<>();
        }
        
        for (
int i = 0; i < stones.length; i++) {
            for (int j = 0; j < stones.length; j++) {
                if (j != i) {
                    if (intersects(stones[i], stones[j])) {
                        adj[i].add(j);
                    }
                }
            }
        }
        
        int cmpts = 0;
        for (int i = 0; i < n; i++) {
            if (!vis[i]) {
                cmpts++;
                
dfs(i);
            }
        }
        return n - cmpts;
    }
    
    void dfs(int u) {
        vis[u] = true;
        for (int v : adj[u]) {
            if (!vis[v])
                dfs(v);
        }
    }
    
    boolean intersects(int[] a, int[] b) {
        return (a[0] == b[0]) || (a[1] == b[1]);
    }
}
