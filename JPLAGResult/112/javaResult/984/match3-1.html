class Solution {
    int[] size;
    int[] parent;
    int count = 0;
    public void init(int n) {
        count = n;
        size = new int[n];
        parent = new int[n];
        for (int i = 0; i < n; i++) {
            size[i] = 1;
            parent[i] = i;
        }
    }
    public int removeStones(int[][] stones) {
        Map<Integer, Integer> map1 = new HashMap<>();
        Map<Integer,Integer> map2 = new HashMap<>();
        int n = stones.length;
        init(n);
        for (int i = 0; i < n; i++) {
            int row = stones[i][0];
            int col = stones[i][1];

            if (!map1.containsKey(row) && !map2.containsKey(col)) {
                map1.put(row, i);
                map2.put(col, i);
            }else {
                Integer n1 = map1.get(row);
                Integer n2 = map2.get(col);
                if (n1 != null) {
                    union(n1, i);
                }
                if (n2 != null) {
                    union(n2, i);
                }
                map1.put(row, find(i));
                map2.put(col, find(i));
            }
        }
        
return count;
    }
    public void union(int i, int j) {
        int rootA = find(i);
        int rootB = find(j);
        if (rootA == rootB) {
            return;
        
}
        count--;
        if (size[rootA] >= size[rootB]) {
            size[rootA] += size[rootB];
            parent[rootB] = rootA;
        } else {
            size[rootB] += size[rootA];
            parent[rootA] = rootB;
        }
        return;
    }
    public int find(int i) {
        if (parent[i] != i) {
            int root = find(parent[i]);
            parent[i] = root;
        }
        return parent[i];
    }

}
