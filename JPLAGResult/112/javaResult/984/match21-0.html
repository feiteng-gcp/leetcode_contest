class Solution {
    public int removeStones(int[][] stones) {
        int n = stones.length;
        Map<int[], Integer> map = new HashMap<>();
        Map<int[], List<int[]>> graph = new HashMap<>();
        int i = 0;
        
        for (int[] s1 : stones) {
            map.put(s1, i++);
            if (!graph.containsKey(s1)) {
                graph.put(s1, new ArrayList<>());
            }
            for (int[] s2 : stones) {
                if (s1 != s2 && (s1[0] == s2[0] || s1[1] == s2[1])) {    
                    graph.get(s1).add(s2);
                }
            }
        }
        
        int count = 0;
        int[] visited = new int[n];
        
        for (i = 0; i < n; i++) {
            if (visited[i] == 0) {
                count++;
                visited[i] = 1;
                dfsHelper(graph, stones, i, map, visited);
            }
        }
        
        return count;
    }
    
    private void dfsHelper(Map<int[], List<int[]>> graph, int[][] stones, int cur, Map<int[], Integer> map, int[] visited) {
        for (int[] next : graph.get(stones[cur])) {
            int index = map.get(next);
            if (visited[index] == 0) {
                visited[index] = 1;
                dfsHelper(graph, stones, index, map, visited);
            }
        }
    }
}
