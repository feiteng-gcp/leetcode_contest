class Solution {
    
public int removeStones(int[][] stones) {
        boolean[] visited = new boolean[stones.length];
        int count = 0;
        for (int i = 0; i < stones.length; i++) {
            int[] stone = stones[i];

            if (!visited[i]) {
                visited[i] = true;
                count += 
dfs(stone, visited, stones);
            }
        }
        return count;
    }
    
    int dfs(int[] ori, boolean[] visited, int[][] stones) {

        int count = 0;
        for (int i = 0; i < stones.length; i++) {
            if (visited[i]) continue;
            if (stones[i][0] == ori[0] && stones[i][1] == ori[1]) {
                continue;
            } else if (stones[i][0] == ori[0] || stones[i][1] == ori[1]) {
                visited[i] = true;
                count += dfs(stones[i], visited, stones) + 1;
            }
        }
        return count;
    }
}
