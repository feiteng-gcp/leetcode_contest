class Solution {
    
    private int[] parent;
    private int cnt;
    private int m;
    private int n;
    
    public int removeStones(int[][] stones) {
        // [[0,0],[0,1],[1,0],[1,2],[2,1],[2,2]]
        // 1 1 0
        // 1 0 1
        // 0 1 1
        // --> 1
        
        // [[0,0],[0,2],[1,1],[2,0],[2,2]]
        // 1 0 1
        // 0 1 0
        // 1 0 1
        // --> 2
        
        // union find
        m = 0;
        n = 0;
        
        int p = stones.length;
        cnt = p;
        
        for (int i = 0; i < p; i++) {
            m = Math.max(stones[i][0], m);
            n = Math.max(stones[i][1], n);
        }
        
        m += 1;
        n += 1;
        
        parent = new int[p];
        for (int i = 0; i < parent.length; i++) {
            parent[i] = i;
        }
        
        for (int i = 0; i < p; i++) {
            for (int j = i + 1; j < p; j++) {
                if (stones[i][0] == stones[j][0] || stones[i][1] == stones[j][1]) {
                    //System.out.println(stones[i][0] + " " +  stones[i][1] + " "+ stones[j][0] + " "+ stones[j][1]);
                    union(i, j);
                }
            }
        }
        
        return p - cnt;
    }
    
    private int find(int x) {
        
        List<Integer> l = new LinkedList<>();
        while (parent[x] != x) {
            l.add(x);
            x = parent[x];
        }
        
        for (Integer i : l) {
            parent[i] = x;
        }
        
        return x;
    }
    
    private void union(int x1, int x2) {
        int f1 = find(x1);
        int f2 = find(x2);
        if (f1 == f2) {
            return;
        }
        
        parent[f1] = f2;
        cnt--;
    }
}
