class Solution {
    int[] root, size, rank;
    int total;
    public int removeStones(int[][] stones) {
        root = new int[stones.length];
        size = new int[stones.length];
        rank = new int[stones.length];
        for (int i = 0; i < root.length; i++) {
            root[i] = i;
            size[i] = 1;
        }
        total = stones.length;
        
for (int i = 0; i < stones.length; i++) {
            for (int j = i+1; j < stones.length; j++) {
                if (stones[i][0] == stones[j][0] || stones[i][1] == stones[j][1]) {
                    union(i, j);
                }
            }
        }
        
return total;
    }
    
    public void union(int i, int j) {
        int id1 = find(i), id2 = find(j);
        if (id1 != id2) {
            total--;
            if (rank[id1] <= rank[id2]) {
                root[id1] = id2;
                if (rank[id1] == rank[id2])
                    rank[id2]++;
            } else {
                root[id2] = id1;
            }
        }
    }
    
    public int find(int i) {
        if (i != root[i])
            root[i] = find(root[i]);
        return root[i];
    }
}
