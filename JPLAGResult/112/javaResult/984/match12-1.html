public class Solution {
    public int removeStones(int[][] stones) {
        Map<Integer, Integer> setR = new HashMap<>(), setC = new HashMap<>();
        int n = stones.length;
        int[] parents = new int[n];
        for (int i = 0;i < n; ++i) parents[i] = i;
        for (int i = 0; i < n; ++i) {
            int[] s = stones[i];
            
if (setR.containsKey(s[0])) {
                merge(parents, i, setR.get(s[0]));
            } else {
                setR.put(s[0], i);
            }
            if (setC.containsKey(s[1])) {
                merge(parents, i, setC.get(s[1]));
            } else {
                setC.put(s[1], i);
            }
        }
        int cnt = 0;
        for (int i = 0; i < n; ++i) if (find(parents, i) != i) cnt++;
        return cnt;
    }

    private void merge(
int[] parents, int i1, int i2) {
        int p1 = find(parents, i1), p2 = find(parents, i2);
        if (p1 != p2) parents[p2] = p1;
    }

    private int find(int[] parents, int i) {
        if (parents[i] == i) return i;
        return parents[i] = find(parents, parents[i]);
    }
}
