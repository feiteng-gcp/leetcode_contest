class Solution {
    final int CONVERT = 10000;
    
    public int removeStones(int[][] stones) {
        Set<Integer> visited = new HashSet<>();
        
Map<Integer, Set<Integer>> rowCords = new HashMap<>();
        Map<Integer, Set<Integer>> colCords = new HashMap<>();
        for (int[] stone : stones) {
            int row = stone[0], col = stone[1];
            int cord = row * CONVERT + col;
            if (!rowCords.containsKey(row)) {
                rowCords.put(row, new HashSet<>());
            }
            if (!colCords.containsKey(col)) {
                colCords.put(col, new HashSet<>());
            
}
            rowCords.get(row).add(cord);
            colCords.get(col).add(cord);
        }
        int count = 0;
        for (int[] stone : stones) {
            int cord = stone[0] * CONVERT + stone[1];
            if (!visited.contains(cord)) {
                dfs(cord, rowCords, colCords, visited);
                count++;
            }
        }
        return stones.length - count;
    }
    
    private void dfs(int cord, Map<Integer, Set<Integer>> rowCords, Map<Integer, Set<Integer>> colCords, Set<Integer> visited) {
        visited.add(cord);
        int row = cord / CONVERT, col = cord % CONVERT;
        for (int rowNext : rowCords.get(row)) {
            if (!visited.contains(rowNext)) {
                dfs(rowNext, rowCords, colCords, visited);
            
}
        }
        for (int colNext : colCords.get(col)) {
            if (!visited.contains(colNext)) {
                dfs(colNext, rowCords, colCords, visited);
            }
        }
    }
}
