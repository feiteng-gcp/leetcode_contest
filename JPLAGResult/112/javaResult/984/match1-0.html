class Solution {
    public int removeStones(int[][] stones) {
        int[] parent = new int[stones.length];
        for (int i = 0; i < parent.length; i++) {
            parent[i] = i;
        }
        HashMap<Integer, List<Integer>> col = new HashMap<>();
        HashMap<Integer, List<Integer>> row = new HashMap<>();
        for (int i = 0; i < stones.length; i++) {
            int[] stone = stones[i];
            int r = stone[0];
            int c = stone[1];
            if (!col.containsKey(c)) col.put(c, new ArrayList<>());
            if (!row.containsKey(r)) row.put(r, new ArrayList<>());
            col.get(c).add(i);
            row.get(r).add(i);
        }
        int index = 0;
        for (int[] stone : stones) {
            
            int r = stone[0];
            int c = stone[1];
            int neigh = -1;
            for (int e : row.get(r)) {
                if (e != index) {
                    int x = find(parent, e);
                    int y = find(parent, index);
                    if (x != y) parent[x] = y;
                }
            }

            for (int e : col.get(c)) {
                if (e != index) {
                    int x = find(parent, e);
                    int y = find(parent, index);
                    if (x != y) parent[x] = y;
                }
            }
            index++;
        }
        Set<Integer> set = new HashSet<>();
        for (int e = 0; e < stones.length; e++) {
            set.add(find(parent, e));
        }
        return stones.length-set.size();
    }
    
    public int find(int[] parent, int f) {
        if (f != parent[f]) {
            parent[f] = find(parent, parent[f]);
        }
        return parent[f];
    }
    
}
