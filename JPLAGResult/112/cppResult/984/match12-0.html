class Solution {
public:
    vector<int> father;
    int getFather(int i) {
        return i == father[i] ? i : father[i] = getFather(father[i]);
    }
    int removeStones(vector<vector<int>>& stones) {
        int n = stones.size();
        father = vector<int>(n);
        for (int i = 0; i < n; i++) {
            father[i] = i;
        }
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (stones[i][0] == stones[j][0] || stones[i][1] == stones[j][1]) {
                    int fi = getFather(i), fj = getFather(j);
                    father[fi] = fj;
                }
            }
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            int fi = getFather(i);
            if (i == fi) {
                res++;
            }
        }
        return res;
    }
};
