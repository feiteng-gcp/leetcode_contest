class Solution {
public:
  int N;
  bool graph[1005][1005];
  bool visited[1005];
  
  void dfs(int u) {
    if (visited[u]) return;
    visited[u] = true;
    for (int i = 0; i < N; i++) {
      if (graph[u][i]) {
        dfs(i);
      }
    }
  }
  
  int removeStones(vector<vector<int>>& stones) {
    N = (int)stones.size();

    for (int i = 0; i < N; i++) {
      for (int j = i + 1; j < N; j++) {
        if (stones[i][0] == stones[j][0] || stones[i][1] == stones[j][1]) {
          graph[i][j] = graph[j][i] = true;
        }
      }
    }
    int res = 0;
    for (int i = 0; i < N; i++) {
      if (visited[i]) continue;
      res++;
      dfs(i);
    }
    return res;
  }
};
