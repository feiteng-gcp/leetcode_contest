class Solution {
    int getpa(int a, vector<int>&pa){
        if(a == pa[a])
            return a;
        return pa[a] = getpa(pa[a],pa);
    }
    void unf(int i, int j, vector<int>&pa){
        int fa = getpa(i,pa);
        int fb = getpa(j,pa);
        if(fa==fb)
            return;
        pa[fb]=fa;
        pa[j] = fa;
    }
public:
    int removeStones(vector<vector<int>>& st) {
        int n = st.size();
        vector<int>pa(n,0);
        for(int i = 1; i < n; i++)

            pa[i]=i;
        for(int i = 0; i < n ; i++){
            for(int j = 0; j < i; j++){
                if(st[i][0] == st[j][0]||st[i][1] == st[j][1]){
                    unf(i,j,pa);

                }
            }
        }
        int res = 0;
        for(int i = 0; i < n; i++){
            if(pa[i] == i)
                res++;
        }
        return n-res;
    }
};
