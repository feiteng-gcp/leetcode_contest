class Solution {
public:
    vector<int> v[1123];
    int n;
    int vis[1123];
    
    void dfs(int u) {
        vis[u] = 1;
        for (int i = 0; i < v[u].size(); i++) {
            if (vis[ v[u][i] ] == 0) dfs(v[u][i]);
        }
    }
    
    int removeStones(vector<vector<int>>& s) {
        n = s.size();
        for (int i = 0; i < n; i++)
            v[i].clear(), vis[i] = 0;
        
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int r1 = s[i][0];
                int c1 = s[i][1];
                int r2 = s[j][0];
                int c2 = s[j][1];
                
                if (r1 == r2 || c1 == c2) {
                    v[i].push_back(j);
                    v[j].push_back(i);
                }
            }
        }
        
        int components = 0;
        for (int i = 0; i < n; i++) {
            if (vis[i] == 0) {
                dfs(i);
                components++;
            }
        }
        return n - components;
    }
};
