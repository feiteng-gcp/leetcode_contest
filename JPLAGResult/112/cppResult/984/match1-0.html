class Solution {
public:
    int find( int u , vector< int > & uf ) {
        if( uf[u] == u )
            return u ; 
        return uf[u] = find( uf[u] , uf ) ; 
    }
    void Union( int u , int v  , vector< int > & uf ){
        int fu = find( u , uf ) ; 
        int fv = find( v , uf ) ; 
        if( fu == fv  )
            return ;
        uf[fv] = fu ; 
    }
    
    int removeStones(vector<vector<int>>& stones) {
        int size = stones.size() ; 
        vector< int > uf ( size, 0 ) ;
        for( int i=0 ; i< size ; i ++)
            uf[i] = i ; 
        
        for( int i=0 ; i< stones.size() ; i++){
            for( int j=i+1 ; j<stones.size() ; j++){
                if(  stones[i][0] == stones[j][0] ||  stones[i][1] == stones[j][1]) {
                    Union( i , j ,uf) ; 
                }
            }
        }
        map< int ,int > check ;  
        for( int i=0 ; i< size ; i++){
            check[find( i , uf ) ] ++ ; 
        }
        int ans = 0 ; 
        for( auto &x : check ) {
            ans +=( x.second -1 ) ; 
        }
        return ans ;
        
    }
};
