class Solution {
public:
    int ds[1005], cnt[1005];
    
    int p(int x){
        while(ds[x] != x){
            x = ds[x];
        }
        return x;
    }
    
    void u(int a, int b){
        int tmp = (cnt[p(a)] + cnt[p(b)]);
        cnt[p(a)] = cnt[p(b)] = tmp;
        ds[p(a)] = p(b);
        return;
    }
    
    
    int removeStones(vector<vector<int>>& stones){
        for(int i = 0;i < 1005;i++){
            ds[i] = i;
            cnt[i] = 1;
        }

        for(int i = 0;i < stones.size();i++){
            for(int j = i+1;j < stones.size();j++){
                if(stones[i][0] == stones[j][0] || stones[i][1] == stones[j][1]){
                    if(p(i) != p(j))
                        u(i, j);

                }
            }
        }
        set<int> s;
        int ret = 0;
        for(int i = 0;i < stones.size();i++){
            if(!s.count(p(i))){
                s.insert(p(i));
                ret += cnt[p(i)] - 1;
            }
        }
        /*
        for(int i = 0;i < stones.size();i++){
            cout << ds[i] << " ";
        }
        cout << endl;
        for(int i = 0;i < stones.size();i++){
            cout << cnt[i] << " ";
        }
        cout << endl;
        */
        return ret;
    }
};
