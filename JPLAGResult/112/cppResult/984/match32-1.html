class Solution {
public:
    int removeStones(vector<vector<int>>& stones) {
        if (stones.size() <= 1) return 0;
        int L = stones.size();

        for (int i = 0; i < L; i++) {
            father[i] = i;
        }
        for (int i = 0; i < L; i++) {
            for (int j = i + 1; j < L; j++) {
                if (in_sameline(stones, i, j)) {
                    union_father(i, j);
                }
            }
        }
        int res = 0;
        unordered_map<int, int> child_count;
        for (int i = 0; i < L; i++) {
            find_father(i);
            child_count[father[i]]++;
        }
        for (auto i : child_count) {
            res += i.second - 1;
        }
        return res;
        
    }
    

    int find_father(int i) {
        if (father[i] == i) return i;
        return father[i] = find_father(father[i]);
    }
    
    void union_father(int left, int right) {
        int left_father = find_father(left);
        int right_father = find_father(right);
        if (left_father != right_father) {
            father[right_father] = left_father;
        }
    }
    
    bool in_sameline(vector<vector<int>>& stones, int i, int j) {
        return stones[i][0] == stones[j][0] || stones[i][1] == stones[j][1];
    }
private:
    unordered_map<int, int> father;
};
