class Solution {
public:
    void dfs(vector<vector<int>> &data,vector<int>&vis,int i){
        if(vis[i])
            return;
        vis[i]=1;
        for(auto&k:data[i])
            dfs(data,vis,k);
        return;
    }

    int removeStones(vector<vector<int>>& s) {
        int n=s.size();
        vector<vector<int>> data(n);
        for(int i=0;i<n;i++){
            for(int j=i;j<n;j++){
                if(s[i][0]==s[j][0]||s[i][1]==s[j][1]){
                    data[i].push_back(j);
                    data[j].push_back(i);
                }
            }
        }
        vector<int>vis(n,0);
        int cnt=0;

        for(int i=0;i<n;i++){
            if(vis[i]==0){
                cnt++;
                dfs(data,vis,i);
            }
        }
        return n-cnt;
    }
};
