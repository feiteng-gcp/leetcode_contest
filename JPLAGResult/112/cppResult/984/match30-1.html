class Solution {
public:
    int removeStones(vector<vector<int>>& stones) {
        int n = stones.size();
        vector<vector<int>> graph(n + 1);
        
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                if (stones[i][0] == stones[j][0] || stones[i][1] == stones[j][1]) {
                    graph[i].push_back(j);
                    graph[j].push_back(i);
                    // cout << i << " " << j << endl;
                }
            }
        }
        
        vector<vector<int>> components;
        vector<bool> visited(n, false);
        
        auto bfs0 = [&](int u) {
            queue<int> q;
            q.push(u);
            visited[u] = true;
            
            vector<int> res;
            res.push_back(u);
            while (!q.empty()) {
                auto u = q.front();
                q.pop();
                
                for (auto v : graph[u]) {
                    if (!visited[v]) {
                        visited[v] = true;
                        q.push(v);
                        res.push_back(v);
                    }
                }
            }
            
            components.emplace_back(res);
        };
        
        for (int i = 0; i < n; ++i) {
            if (visited[i]) {
                continue;
            }
            bfs0(i);
        }
        
        
        auto bfs = [&](int s) {
            queue<int> q;
            visited[s] = true;
            q.push(s);
            vector<bool> visited(n + 1, false);    
            while (!q.empty()) {
                auto u = q.front();
                q.pop();
                
                for (auto v : graph[u]) {
                    if (!visited[v]) {
                        // cout << s << " " << v << endl;
                        visited[v] = true;
                        q.push(v);
                    }
                }
            }
            
            int res = 0;
            
            for (auto b : visited) {
                if (b) {
                    ++res;
                }
            }
            
            return res;
        };
        
        
        int res = 0;
        for (auto& comp : components) {
            int ans = 0;
            for (auto u : comp) {
                ans = max(ans, bfs(u) - 1);
            }
            res += ans;
        }
        
        return res;
    }
};
