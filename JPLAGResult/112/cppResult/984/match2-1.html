class Solution {
public:

    int removeStones(vector<vector<int>>& stones) {
        int n=stones.size();
        vector<int> gr[n];
        for(int i=0; i<n; i++){
            for(int j=i+1; j<n; j++){
                if(stones[i][1]==stones[j][1] || stones[i][0]==stones[j][0]){
                    gr[i].push_back(j);
                    gr[j].push_back(i);
                }
            }
        }
        
        vector<bool> vst(n,false);
        queue<int> q;
        int cnt=0;
        for(int i=0; i<n; i++){
            if(!vst[i]){
                q.push(i);
                vst[i]=true;
                while(!q.empty()){
                    int j=q.front();
                    q.pop();
                    for(auto x : gr[j]){
                        if(!vst[x]){
                            vst[x]=true;
                            q.push(x);
                        }
                    }
                }
                cnt++;
            }
        }
        
        return n-cnt;
    }
};
