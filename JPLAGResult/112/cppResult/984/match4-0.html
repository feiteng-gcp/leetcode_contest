class Solution {
public:
    int removeStones(vector<vector<int>>& stones) {
        int n = stones.size();
        vector<vector<int>> g(n);
        for(int i = 0; i < n; ++i)
        {
            for(int j = i + 1; j < n; ++j)
            {
                auto u = stones[i], v = stones[j];
                if(adjacent(u ,v))
                {
                    g[i].push_back(j);
                    g[j].push_back(i);
                }
            }
        }
        
        int cnt = 0;
        unordered_set<int> visited;
        for(int i = 0; i < n; ++i)
        {
            if(visited.find(i) == visited.end())
            {
                ++cnt;
                dfs(g, i, visited);
            }
        }
        return n - cnt;
    }
    
private:
    bool adjacent(const vector<int>& node1, const vector<int>& node2)
    {
        return node1[0] == node2[0] || node1[1] == node2[1];
    }
    
    void dfs(const vector<vector<int>>& g, int curr, unordered_set<int>& visited)
    {
        if(visited.find(curr) != visited.end())return;
        visited.insert(curr);
        for(const auto& adj : g[curr])
            dfs(g, adj, visited);
    }
};
