class Solution {
public:
    int n;
    int a[1010][1010];
    int vis[1010];
    
    int dfs(int cur) {
        vis[cur]=1;
        int ret=1;
        for(int i=0; i<n; i++) {
            if (vis[i]) continue;
            if (!a[cur][i]) continue;
            ret+=dfs(i);
        }
        return ret;
    } 
    
    int removeStones(vector<vector<int>>& stones) {
        memset(a, 0, sizeof(a));
        memset(vis, 0, sizeof(vis));
        n=stones.size();
        for(int i=0; i<n; i++) {
            for(int j=i+1; j<n; j++) {
                if ( stones[i][0]==stones[j][0] || stones[i][1]==stones[j][1] ) {
                    a[i][j]=a[j][i]=1;
                }
            }
        }
        int ans=0;
        for(int i=0; i<n; i++) {
            if (!vis[i]) ans+=dfs(i)-1;
        }
        return ans;
    }
};
