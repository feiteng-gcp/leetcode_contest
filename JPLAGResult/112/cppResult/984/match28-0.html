bool visited[1000+5];
vector<int> graph[1000+5];
int compo = 0;

void dfs(int u)
{
    compo++;
    visited[u] = true;

    for (auto v: graph[u])
    {
        if (!visited[v])
        {
            dfs(v);
        }
    }
}

class Solution {
public:
    int removeStones(vector<vector<int>>& stones) {
        int n = stones.size();

        for (int i = 0; i <= n; i++)
        {
            visited[i] = false;
            graph[i].clear();
        }

        for (int i = 0; i < n; i++)
        {
            for (int j = i+1; j < n; j++)
            {
                if ((stones[i][0] == stones[j][0]) || (stones[i][1] == stones[j][1]))
                {
                    graph[i+1].push_back(j+1);
                    graph[j+1].push_back(i+1);
                }
            }
        }

        int ans = 0;

        for (int i = 1; i <= n; i++)
        {
            if (!visited[i])
            {
                compo = 0;
                dfs(i);
                ans += compo-1;
            }
        }

        return ans;
    }
};
