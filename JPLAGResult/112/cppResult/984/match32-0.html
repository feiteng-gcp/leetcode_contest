class Solution {
public:
	vector<int>p;
	int findSet(int x) {
		if (p[x] == x)return x;
		return p[x] = findSet(p[x]);
	}
	void join(int x, int y) {
		int a = findSet(x), b = findSet(y);
		if (a == b)return;
		p[a] = b;
	}
	int removeStones(vector<vector<int>>& stones) {
		p.resize(20000);
		for (int i = 0; i < p.size(); i++) {
			p[i] = i;
		}
		for (int i = 0; i < stones.size(); i++) {
			join(stones[i][0], stones[i][1] + 10000);
		}
		map<int, int>ct;
		for (int i = 0; i < p.size(); i++) {
			ct[findSet(i)]++;
		}
		int p = 0;
		for (auto it : ct) {
			if (it.second > 1)p++;
		}
		return p;
	}
};
