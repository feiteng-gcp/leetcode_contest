class Solution {
public:
    int n;
    vector < int > vis;
    vector < vector < int > > A;
    
    int dfs(int node) {
        vis[node] = true;

        int ret = 1;
        for(int i = 0;i < A[node].size();++i) {
            int to = A[node][i];
            if(!vis[to]) {
                ret += dfs(to);
            }
        }
        return ret;
    }
    
    int removeStones(vector<vector<int>>& stones) {
        int n = stones.size();
        A.resize(n);
        vis.resize(n);

        for(int i = 0;i < n;++i) {
            for(int j = i + 1;j < n;++j) {
                if(stones[i][0] == stones[j][0] or stones[i][1] == stones[j][1]) {
                    A[i].push_back(j);
                    A[j].push_back(i);
                }
            }
        }
        
        int ans = 0;
        for(int i = 0;i < n;++i) {
            if(!vis[i]) {
                int x = dfs(i);
                ans += x - 1;
            }
        }
        return ans;
    }
};
