class Solution {
    vector<int> ss;
    int find(int i){
        if(ss[i]!=i) ss[i]=find(ss[i]);
        return ss[i];
    }
    
    void merge(int i,int j){
        ss[find(i)]=find(j);
    }
public:
    int removeStones(vector<vector<int>>& stones) {
        int N=stones.size();
        ss=vector<int>(N);

        for(int i=0;i<N;i++) ss[i]=i;
        for(int i=0;i<N;i++)
            for(int j=i+1;j<N;j++){
                if(stones[i][0]==stones[j][0]||stones[i][1]==stones[j][1]){
                    merge(i,j);  
                }
            }
        int ans=0;
        for(int i=0;i<N;i++){
            if(find(i)!=i) ans++;
        }
        return ans;
    }
};
