class Solution {
public:
    vector<int> parent, depth, children;
    
    int removeStones(vector<vector<int>>& stones) {
        //stones = {{0,0},{0,2},{1,1},{2,0},{2,2}};
        //stones = {{0, 0}};
        parent = vector<int>(stones.size(), 0);
        depth = vector<int>(stones.size(), 1);
        children = vector<int>(stones.size(), 1);

        for(int i = 0; i < parent.size(); i++) parent[i] = i;
        for(int i = 0; i < stones.size(); i++)
        {
            for(int j = i + 1; j < stones.size(); j++)
            {
                if(stones[i][0] == stones[j][0] || stones[i][1] == stones[j][1]) unite(i, j);

            }
        }
        int ret = 0;
        for(int i = 0; i < stones.size(); i++)
        {
            if(parent[i] == i)
            {
                ret += children[i] - 1;
            }
        }
        return ret;
    }
    
    int root(int node)
    {
        while(parent[node] != node) node = parent[node];
        return node;
    }
    
    void unite(int lhs, int rhs)
    {
        int rootLHS = root(lhs);
        int rootRHS = root(rhs);
        if(rootLHS == rootRHS) return;
        if(depth[rootLHS] < depth[rootRHS])
        {
            parent[rootLHS] = rootRHS;
            depth[rootRHS] = max(depth[rootRHS], depth[rootLHS] + 1);
            children[rootRHS] += children[rootLHS];
        }
        else
        {
            parent[rootRHS] = rootLHS;
            depth[rootLHS] = max(depth[rootLHS], depth[rootRHS] + 1);
            children[rootLHS] += children[rootRHS];
        }
    }
};
