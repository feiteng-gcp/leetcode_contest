class Solution {
public:
    bool validateStackSequences(vector<int>& pushed, vector<int>& popped) {
        stack<int> st;
        int i = 0;
        for(int n2 : popped){
            while(i < pushed.size() && (st.empty() || st.top() != n2)){
                st.push(pushed[i++]);
            }
            if(st.top() == n2){
                st.pop();
            }else{
                return false;
            }
        }
        return true;
    }
};
