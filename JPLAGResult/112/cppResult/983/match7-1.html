class Solution{
public:
	bool validateStackSequences(vector<int>& pushed, vector<int>& popped) {
		stack<int> st;
		int b = 0;

		for (int a : pushed) {
			st.push(a);

			while (!st.empty() && b < popped.size() && popped[b] == st.top()) {
				st.pop();
				b++;
			}
		}


		return b == popped.size();
	}
};
