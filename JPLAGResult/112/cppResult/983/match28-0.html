class Solution {
public:
    bool validateStackSequences(vector<int>& pushed, vector<int>& popped) {
        stack<int> s;
        int idx = 0;
        for (auto it: pushed) {
            s.push(it);
            while (!s.empty() && idx < popped.size() && s.top() == popped[idx]) {
                idx++;
                s.pop();
            }
        }
        return s.empty();
    }
};
