class Solution {
public:
    bool validateStackSequences(vector<int>& pushed, vector<int>& popped) {
        stack<int> s;
        int pos = 0;
        for (auto& x : popped) {
            if (!s.empty() && s.top() == x) {
                s.pop();
            }
            else {
                while (pos < pushed.size() && (s.empty() || s.top() != x)) {
                    s.push(pushed[pos]);
                    ++pos;
                }
                if (s.top() == x) {
                    s.pop();
                }
                else {
                    break;
                }
            }
        }
        
        return s.empty() && pos == pushed.size();
    }
};
