class Solution {
public:
    bool validateStackSequences(vector<int>& pushed, vector<int>& popped) {
        vector<int> st;
        int pop_idx = 0;
        for(int push_value: pushed) {
            st.push_back(push_value);
            while (!st.empty() && st.back() == popped[pop_idx]) {
                pop_idx++;
                st.pop_back();
            }            
        }
        return pop_idx == popped.size();
    }
};
