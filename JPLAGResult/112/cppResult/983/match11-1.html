class Solution {
public:
    bool validateStackSequences(vector<int> &pushed, vector<int> &popped) {
        vector<int> stack;
        while (!pushed.empty()) {
            while (stack.empty() || stack.back() != pushed.back()) {
                if (popped.empty()) return false;
                stack.push_back(popped.back());
                popped.pop_back();
            }
            
            if (stack.empty() || stack.back() != pushed.back()) 
                return false;
            
            stack.pop_back();
            pushed.pop_back();
        }
        
        return pushed.empty();
    }
};
