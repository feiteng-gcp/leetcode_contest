class Solution {
public:
    bool validateStackSequences(vector<int>& pushed, vector<int>& popped) {
        stack<int> myStack;
        int curIndex = 0;
        for(int cur : popped) {
            while((myStack.empty() || cur != myStack.top()) && curIndex < pushed.size()) {
                myStack.push(pushed[curIndex++]);
            }
            if(cur == myStack.top())
                myStack.pop();
        }
        
        return curIndex == pushed.size() && myStack.empty();
    }
};
