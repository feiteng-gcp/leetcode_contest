class Solution {
public:
	bool validateStackSequences(vector<int>& pushed, vector<int>& popped) {
		stack<int> gogo;
		int now_itr = 0;
		for (int i = 0; i < pushed.size(); ++i) {
			gogo.push(pushed[i]);
			while (gogo.empty() == false && gogo.top() == popped[now_itr]) {
				gogo.pop();
				now_itr++;
			}
		}
		if (now_itr == popped.size()) {
			return true;
		}
		return false;
	}
};
