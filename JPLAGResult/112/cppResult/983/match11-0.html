class Solution {
public:
    bool validateStackSequences(vector<int>& pushed, vector<int>& popped) {
        vector<int> s;
        reverse(pushed.begin(), pushed.end());
        reverse(popped.begin(), popped.end());
        while(popped.size()) {
            while(s.empty() || s.back() != popped.back()) {
                if(pushed.empty()) return false;
                s.emplace_back(pushed.back());
                pushed.pop_back();
            }
            if(s.empty() || s.back() != popped.back()) return false;
            s.pop_back();
            popped.pop_back();
        }
        return true;
    }
};
