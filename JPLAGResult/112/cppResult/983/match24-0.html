class Solution {
public:
    bool validateStackSequences(vector<int>& push, vector<int>& pop) {
        vector<int>a;
        int j=0;
        for(int i=0;i<push.size();i++){
            a.push_back(push[i]);
            while(a.size()>0&&j<pop.size()&&a.back()==pop[j]){
                a.pop_back();
                j++;
            }
        }
        while(a.size()>0&&j<pop.size()&&a.back()==pop[j]){
            a.pop_back();
            j++;
        }
        if(a.size()!=0||j<pop.size())
            return false;
        return true;
    }
};
