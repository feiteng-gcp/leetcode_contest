class Solution {
public:
    int minIncrementForUnique(vector<int>& A) {
        if (A.empty())
        {
            return 0;
        }
        sort(A.begin(), A.end());
        int last = A[0];

        int ans = 0;

        for (int i = 1; i < A.size(); i++)
        {
            if (A[i] <= last)
            {
                last++;
                ans += (last - A[i]);
            }
            else
            {
                last = A[i];
            }
        }

        return ans;
    }
};
