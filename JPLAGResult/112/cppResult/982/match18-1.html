class Solution {
public:
    int minIncrementForUnique(vector<int>& A) {
        if (A.size() <= 1)
        {
            return 0;
        }
        
        sort(A.begin(), A.end());
        
        int last = A[0];
        int moves = 0;
        for (int i = 1; i < A.size(); i++)
        {
            if (A[i] <= last)
            {
                last++;
                moves += (last - A[i]);
            }
            else
            {
                last = A[i];
            }
        }
        
        return moves;
    }
};
