class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        sort(tokens.begin(), tokens.end());
        int i = 0, j = tokens.size()-1;
        int cur = 0, maxx = 0;
        while (i <= j) {
            if (tokens[i] <= P) {
                cur++;
                maxx = max(maxx, cur);
                P -= tokens[i];
                i++;
            }
            else if (cur > 0) {
                P += tokens[j];
                cur--;
                j--;
            }
            else break;
        }
        return maxx;
    }
};
