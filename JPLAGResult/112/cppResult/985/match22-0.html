class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        
        if (tokens.empty()) return 0;
        
        sort(tokens.begin(), tokens.end());
        
        int left = 0, right = tokens.size() - 1;
        
        int curp = 0;
        int resp = 0;
        
        while (left <= right) {
            
            if (P >= tokens[left]) {
                P -= tokens[left];
                curp ++;
                left ++;
            }
            else if (curp > 0) {
                P += tokens[right];
                --right;
                curp --;
            }
            else {
                break;
            }
            
            resp = max(resp, curp);            
        }
        
        return resp;
    }
};
