class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        vector<int> t = tokens;
        int i = 0, j = (int)t.size()-1;
        int ans = 0;
        int temp = 0;
        int point = 0;
        
        sort(t.begin(), t.end());
        
        while (i <= j){
            if (P >= t[i]){
                P -= t[i];
                ++i;
                ++point;
                ans = max(ans, point);
            }
            else if (point > 0)
            {
                P += t[j];
                --j;
                --point;
            }
            else
                break;
        }
        
        return ans;
    }
};
