class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        sort(tokens.begin(), tokens.end());
        int n=tokens.size();
        int s=0, e=n-1, ans=0, cnt=0;

        while(s<=e){
            if(tokens[s]<=P){
                // cout<<s<<"-i ";
                cnt++;
                P-=tokens[s];
                ans=max(ans,cnt);
                s++;
            } else if(cnt){
                // cout<<e<<"-r ";
                P+=tokens[e];
                e--;
                cnt--;
            } else break;
        }
        return ans;
    }
};
