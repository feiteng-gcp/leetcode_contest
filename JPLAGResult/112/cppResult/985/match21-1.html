class Solution {
public:
    int bagOfTokensScore(vector<int>& v, int P) {
        int res = 0;
        sort(v.begin(), v.end());
        
        int start = 0;
        int end = v.size() - 1;
        

        int points = 0;
        while (start <= end) {
            if (P >= v[start]) {
                points++;
                P -= v[start];
                start++;
            } else if (points > 0) {
                points--;
                P += v[end];
                end--;
            } else {
                break;
            }
            res = max(res, points);
        }
        return res;
        
    }
};
