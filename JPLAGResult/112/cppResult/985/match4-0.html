class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        int point = 0;
        int power = P;        
        sort(tokens.begin(), tokens.end());
        int L = 0;
        int R = tokens.size() - 1;
        int ans = 0;
        while (L <= R) {
            if (tokens[L] <= power) {
                point += 1;
                ans = max(ans, point);
                power -= tokens[L++];
            } else {
                if (point > 0) {
                    point -= 1;
                    power += tokens[R--];
                } else {
                    break;
                }                
            }
        }
        return ans;
        
    }
};
