class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int p) {
        std::sort(tokens.begin(), tokens.end());
        
        int x = 0, y = tokens.size();
        int sc = 0;
        int ans = 0;
        while (x < y)
        {
            while (p >= tokens[x] && x < y)
            {
                p -= tokens[x];
                sc ++;
                x ++;
                ans = std::max(ans, sc);
            }
            if (y > x && sc > 0)
            {
                sc --;
                y --;
                p += tokens[y];                
            }
            else
            {
                break;
            }
        }
        return ans;
    }
};
