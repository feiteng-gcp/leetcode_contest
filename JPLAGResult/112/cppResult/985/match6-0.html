class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        sort(tokens.begin(), tokens.end());
        
        int b = 0;
        int e = tokens.size() - 1;
        int p = 0;
        int sol = 0;
        while (b <= e) {
            if (P >= tokens[b]) {
                P -= tokens[b];
                p++;
                b++;
                sol = max(sol, p);
            } else if (p) {
                P += tokens[e];
                p--;
                --e;
            } else {
                break;
            }
        }
        return sol;
    }
};
