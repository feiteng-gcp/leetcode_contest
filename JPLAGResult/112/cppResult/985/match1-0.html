class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        sort(tokens.begin(), tokens.end());
        int points = 0;
        int lo = 0;
        int hi = tokens.size() - 1;
        while(lo <= hi) {
            if (lo == hi) {
                if (P >= tokens[lo]) {
                    ++points;
                }
                ++lo;
            }
            else {
                if (P >= tokens[lo]) {
                    ++points;
                    P -= tokens[lo];
                    ++lo;
                }
                else {
                    if (points > 0) {
                        --points;
                        P += tokens[hi];
                    }
                    --hi;
                }
            }
        }
        return points;
    }
};
