class Solution {
public:
    int bagOfTokensScore(vector<int>& t, int p) {
        if (t.size()==0) return 0;
        int i=0,j,ans=0,cnt=0;
        int n=t.size()-1;
        sort(t.begin(),t.end());
        if (p<t[0]) return 0;

        while (i<=n)
        {
            if (p>=t[i]){
                p-=t[i];
                i++;
                cnt++;
                ans=max(ans,cnt);
            } else if (cnt>0) {
                p+=t[n];
                n--;
                cnt--;
            }
        }
        return ans;
    }
};
