class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
      std::sort(tokens.begin(), tokens.end());
            

      int i = 0, j = tokens.size() - 1;
      int points = 0;
      int max_points = 0;
      while (i <= j) {
        if (P >= tokens[i]) {
          P -= tokens[i];
          ++i;
          points += 1;
          max_points = max(points, max_points);
        } else if (points > 0) {
          --points;
          P += tokens[j];
          --j;
        } else {
          break;
        }
      }
      
      return max_points;
    }
};
