class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        sort(tokens.begin(), tokens.end());
        int score = 0;
        int left = 0, right = tokens.size() - 1;
        int largest = score;
        while (left <= right) {
            if (tokens[left] <= P) {
                ++score;
                P -= tokens[left++];
                largest = max(largest, score);
            }
            else if (score > 0) {
                --score;
                P += tokens[right--];
            }
            else break;
        }
        return largest;
    }
};
