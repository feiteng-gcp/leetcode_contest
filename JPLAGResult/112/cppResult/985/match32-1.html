#define REP(i,n) for(int i = 0; i < n; i++)
#define ALL(v) v.begin(), v.end()
#define SZ(v) (int)v.size()

class Solution {
public:
  int bagOfTokensScore(vector<int>& tokens, int P) {
		sort(ALL(tokens));
		int ans = 0;
		REP (len, SZ(tokens)) {
			ans = max(ans, f(tokens, P, len));
		}
		return ans;
  }
  

  int f(vInt tokens, int power, int len) {
  	int score = 0;
  	int firstPtr = 0;
  	int secondPtr = SZ(tokens) - 1;
  	int secondGroupBegin = SZ(tokens) - len;
  	while (true) {
  		bool canAdd = (firstPtr < secondGroupBegin);
  		bool canDelete = (secondPtr >= secondGroupBegin);
			if (canDelete && score > 0) {

				score --;
				power += tokens[secondPtr];
				secondPtr --;
			} else {
				if (canAdd && power >= tokens[firstPtr]) {
					score ++;
					power -= tokens[firstPtr];
					firstPtr ++;
				} else {
					break;
				}
			}
  	}
  	return score;
  }
};
