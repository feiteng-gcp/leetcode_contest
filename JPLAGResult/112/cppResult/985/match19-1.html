class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        sort(tokens.begin(), tokens.end());
        int l = 0, r = tokens.size()-1;
        int cur = 0, ret = 0;
        
        while(l <= r){
            if(P >= tokens[l]){
                P -= tokens[l];
                l++;
                cur++;
                ret = max(ret, cur);
            }
            else if(cur >= 1){
                P += tokens[r];
                r--;
                cur--;
            }
            else{
                r--;
            }
        }
        return ret;
    }
};
