class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        sort(begin(tokens), end(tokens));
        int n=tokens.size();
        int i=0, j=n;
        int res=0;
        while (i<j) {
            if (P>=tokens[i]) {
                P-=tokens[i];
                ++i;
                ++res;

            }
            else if (res>0&&j-i>1) {
                P+=tokens[j-1];
                --j;
                --res;
            }
            else break;
        }
        return res;
    }
};
