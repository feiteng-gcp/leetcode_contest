    int bagOfTokensScore(vector<int>& tokens, int P) {
        int n = tokens.size();
        sort(tokens.begin(),tokens.end());
        int low = 0, high = n-1 , p=P;
        if(n == 1){
            if(p >= tokens[0])return 1;
            return 0;
        }
        int ans = 0 , count =0;
        while(low <= high){
            if(p >=tokens[low]){
                count++;
                ans = max(ans , count);
               // low++;
                p-= tokens[low];
                low++;
                continue;
            }else{
                if(count >=1){
                count-=1;
                p+=tokens[high];
                high--;
                } else {
                    break;
                }
                //high--;
            }
        }
        
        return ans;
        
    }
};
