class Solution {
public:
    int bagOfTokensScore(vector<int>& tokens, int P) {
        sort(tokens.begin(),tokens.end());
        
        int points = 0 , ans = 0;
        
        int s = 0, e = tokens.size()-1;
        
        while(s <= e && P >= tokens[s]){
            while(P >= tokens[s] && s <= e){
                points++;
                P -= tokens[s];
                ++s;
            }
            
            ans = points;
            
            while(P < tokens[s] && e > s){
                points--;
                P += tokens[e];
                --e;
            }
        }
        
        return points;
        
    }
};
