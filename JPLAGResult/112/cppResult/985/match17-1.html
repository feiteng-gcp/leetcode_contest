class Solution {
public:
int bagOfTokensScore(vector<int>& vec, int power) {
    sort(vec.begin(), vec.end());
    
    int n = vec.size();
    int i = 0, j = n - 1;
    
    int ret = 0;
    int cnt = 0;
    while (i <= j) {
        if (power >= vec[i]) {
            power -= vec[i];
            cnt++;
            ret = max(cnt, ret);
            i++;
            continue;

        } else {
            if (cnt > 0) {
                cnt--;
                power += vec[j];
                j--;
            } else {
                break;
            }

        }
    }

    return ret;
}
};
