class Solution {
public:
    int bagOfTokensScore(vector<int>& token, int p) {
        if (token.size() == 0)
            return 0;
        if (token.size() == 1 && token[0] <= p)
            return 1;
        sort(token.begin(), token.end());
        int start = 0, end = token.size() - 1;

        int ret = 0;
        while (p >= token[start] || (ret > 0 && token[start] < token[end]))
        {
            if (p >= token[start])
            {
                ret++;
                p -= token[start];
                start++;
            }
            else if (ret > 0 && token[start] < token[end])
            {
                p += (token[end] - token[start]);
                start++;
                end--;
            }
        }
        return ret;
    }
};
