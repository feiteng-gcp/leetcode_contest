class Solution {
public:
int bagOfTokensScore(vector<int>& tokens, int P) {
    sort(tokens.begin(), tokens.end());
    int count = 0;
    int result = 0;
    int fule = P;
    
    int i = 0;
    int j = tokens.size() - 1;
    
    while (i <= j && (fule >= tokens[i] || count)) {
        
        if (fule >= tokens[i]) {
            fule -=  tokens[i];

            i++;
            count++;
        } else {
            count--;
            fule +=  tokens[j];
            j--;
        }
        
        result = max(result,count);
    }
    
    return result;
}
};
