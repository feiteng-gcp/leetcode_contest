class Solution {
public:
    double mincostToHireWorkers(vector<int>& q, vector<int>& w, int K) {
        int n = q.size();
        if(K == 0) return 0.0;
        vector<pair<double, int>> a;
        for(int i = 0; i < n; ++i) {
            a.emplace_back((double)w[i] / (double)q[i], i);
        }
        
        sort(a.begin(), a.end());
        int ps = 0;
        priority_queue<int> que;

        for(int i = 0; i < K; ++i) {
            ps += q[a[i].second];
            que.push(q[a[i].second]);
        }
        double res = (ps + 0.0) * a[K - 1].first;
        
        for(int i = K; i < n; ++i) {
            int cur = que.top();
            ps += q[a[i].second] - cur;
            que.pop();
            que.push(q[a[i].second]);
            res = min(res, (ps + 0.0) * a[i].first);
        }
        return res;
    }
};
