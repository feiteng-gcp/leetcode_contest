class Solution {
public:
    double mincostToHireWorkers(vector<int>& q, vector<int>& w, int k) {
        int n = q.size();
        vector<pair<double, int>> a;
        for (int i=0; i<n; ++i) a.push_back({1.0*w[i]/q[i], q[i]});
        sort(a.begin(), a.end());
        // for (int i=0; i<n; ++i) printf("%f %d\n", a[i].first, a[i].second);
        multiset<int> st;
        int sum = 0;
        for (int i=0; i<k-1; ++i) {
            st.insert(a[i].second);
            sum += a[i].second;
        }
        
        double ans = INT_MAX;
        for (int i=k-1; i<n; ++i) {
            // printf("%f\n", a[i].first*(sum+a[i].second));
            ans = min(ans, a[i].first*(sum+a[i].second));
            sum += a[i].second;
            st.insert(a[i].second);
            if (st.size()>k-1) {
                sum -= *prev(st.end(), 1);
                st.erase(prev(st.end(), 1));
            }
        }
        return ans;
    }
};
