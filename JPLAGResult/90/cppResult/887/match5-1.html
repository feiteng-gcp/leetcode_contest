struct Node {
  int x, y, idx;
  bool operator <(const Node& rhs) const {
    return x < rhs.x || (x == rhs.x && idx < rhs.idx);
  }
};

const int N = 1e4 + 10;
int c[N];
double d[N];

int query(int x) {
  int ret = 0;
  for (; x; x -= x & -x) {
    ret += c[x];
  }
  return ret;
}
void add(int x) {
  for (; x < N; x += x & -x) {
    c[x]++;
  }
}

double query2(int x) {
  double ret = 0;
  for (; x; x -= x & -x) {
    ret += d[x];
  }
  return ret;
}

void add2(int x, double v) {
  for (; x < N; x += x & -x) {
    d[x] += v;
  }
}

class Solution {

public:
    double mincostToHireWorkers(vector<int>& a, vector<int>& b, int m) {
        int n = a.size();
        vector<Node> A(n);
        for (int i = 0; i < n; ++i) A[i] = {a[i], b[i], i};
        sort(A.begin(), A.end(), [](const Node& u, const Node& v) {
            return u.y * v.x > v.y * u.x;
        });
        // for (int i = 0; i < n; ++i) {
        //   trace(A[i].x, A[i].y);
        // }
        double ret = 1e300;
        vector<Node> B(A.begin(), A.end());
        sort(B.begin(), B.end());
        vector<double> sum(n + 1);
        for (int i = 0; i < n; ++i) {
          sum[i + 1] = sum[i] + B[i].x;
        }
        // puts("");
        // for (int i = 0; i < n; ++i) {
        //   trace(B[i].x, B[i].y);
        // }
        memset(c, 0, sizeof(c));
        memset(d, 0, sizeof(d));
        for (int i = 0; i + m <= n; ++i) {
          int idx = lower_bound(B.begin(), B.end(), A[i]) - B.begin();
          add(idx + 1);
          add2(idx + 1, B[idx].x);
          int low = 1, high = n + 1;
          while (low != high) {
            int mid = (low + high + 1) / 2;
            if (mid - query(mid) > m - 1) {
              high = mid - 1;
            } else {
              low = mid;
            }
          }
          // auto t = high - query(high);
          // trace(high, t);
          double p = (double)A[i].y / A[i].x;
          double cur = (sum[high] - query2(high) + A[i].x) * p;
          // trace(i, high, cur);
          ret = min(ret, cur);
        }
        return ret;
    }
};
