struct Elem {
    int q, w;
    bool operator<(const Elem&b) const {
        return w*b.q < q*b.w;
    }
    Elem(int _q, int _w) {
        q = _q;
        w = _w;
    }
} ;
class Solution {
public:
    double mincostToHireWorkers(vector<int>& quality, vector<int>& wage, int K) {
        vector<Elem> lst;
        int N = quality.size();
        for(int i=0; i<N; i++) {
            lst.push_back(Elem(quality[i], wage[i]));
        }
        sort(lst.begin(), lst.end());
        priority_queue<int> heap;
        int heap_sum = 0;
        double ans;
        for(int i=0; i<K; i++) {
            heap_sum += lst[i].q;
            heap.push(lst[i].q);
        }
        ans = (double)lst[K-1].w / lst[K-1].q * heap_sum;
        //printf("i=%d nans=%f push %d %f\n", K-1, ans, lst[K-1].q, (double)lst[K-1].w / lst[K-1].q);
        for(int i=K; i<N; i++) {
            heap_sum += lst[i].q;
            heap.push(lst[i].q);
            heap_sum -= heap.top();
            heap.pop();
            double nans = (double)lst[i].w / lst[i].q * heap_sum;
            //printf("i=%d nans=%f push %d pop %d %f\n", i, nans, lst[i].q, to_del, (double)lst[i].w / lst[i].q);
            if(ans > nans) ans = nans;
        }
        return ans;
    }
};
