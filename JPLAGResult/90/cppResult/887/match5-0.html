class Solution {
    struct Data {
        int q, w;
        double r() const { return (double)w/q;}
        bool operator < ( const Data &t) const { return r()<t.r();}
    } d[10010];
    int h1[10010];
    int h2[10010];
public:
    double mincostToHireWorkers(vector<int>& quality, vector<int>& wage, int K) {
        int n = wage.size();
        for (int i=0;i<wage.size();i++) d[i].w= wage[i], d[i].q=quality[i];
        sort(d,d+wage.size());
        int t1=0,t2=0;
        int now = 0;
        double score= 1e10;
        for (int i=0;i<K-1;i++) {
            cout << d[i].w << " " << d[i].q;
            h2[t2++]=d[i].q;
            push_heap(h2,h2+t2,less<int>());
            now += d[i].q;
        }
        for (int i=K-1;i<n;i++) {
            if (score>(now+d[i].q)*d[i].r()) score = (now+d[i].q)*d[i].r();
            h2[t2++]=d[i].q;
            now += d[i].q;
            push_heap(h2,h2+t2,less<int>());
            //for (int i=0;i<t2;i++) cout << h2[i] << " ";cout <<endl;
            pop_heap(h2,h2+t2,less<int>());
            //for (int i=0;i<t2;i++) cout << h2[i] << " ";cout <<endl;
            --t2;
            now -= h2[t2];
            
        }
        return score;
    }
};
