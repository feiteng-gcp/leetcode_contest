class Solution {
    int dcp(int a, int b) {
        if(a < b) {
            return dcp(b, a);
        }
        if(b == 0)
            return a;
        else
            return dcp(b, a % b);

    }
public:
    int mirrorReflection(int p, int q) {
        int d = dcp(p, q);
        int n = p / d;
        if(n % 2 == 0)
            return 2;
        int m = q / d;
        if(m % 2 == 1)
            return 1;
        else 
            return 0;
    }
};
