class Solution {
public:
    int gcd(int a, int b) {
        if (!b) return a;
        else return gcd(b, a%b);
    }


    int mirrorReflection(int p, int q) {
        int lcm=p*q/gcd(p, q);
        int s=lcm/p;
        cout<<s<<endl;
        int t=lcm/q;
        cout<<t<<endl;
        swap(t, s);
        if (s%2==1&&t%2==0) return 0;
        if (s%2==1&&t%2==1) return 1;
        if (s%2==0&&t%2==1) return 2;
    }
};
