class Solution {
public:
    int mirrorReflection(int p, int q) {
        if(q==0) return 0;
        if(p==q) return 1;
        if(p%2==1){
            if(q%2==0) return 0;
            else return 1;
        }
        else{
            if(q%2==1) return 2;
            else return mirrorReflection(p/2,q/2);
        }
    }
};
