class Solution {
public:
    int mirrorReflection(int p, int q) {
        int times = 1, len = p;
        while (len % q != 0) {
            len += p;
            times++;
        }
        if (times % 2 == 0) return 0;
        if (len / q % 2 == 0) return 2;
        return 1;
    }
};
