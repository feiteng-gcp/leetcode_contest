class Solution {
    int gcd(int a,int b)  
{   int c;  
    c=(a>b)?b:a;  
    while(a%c!=0||b%c!=0)  
    {   c--;  
    }  
    return c;  
} 
public:
    int mirrorReflection(int p, int q) {
        int h = gcd(p,q);
        int x = p/h;
        int y = q/h;
        if(x%2==0&&y%2==1)return 2;
        if(x%2==1&&y%2==1)return 1;
        if(x%2==1&&y%2==0)return 0;
    }
};
