class Solution {
private:
    int gcd(int a, int b) {
        return (b==0) ? a : gcd(b, a % b); 
    }

public:
    int mirrorReflection(int p, int q) {
        if (q == 0) return 0;
        if (q == p) return 1;
        int g = gcd(p, q);
        int x = p / g;
        int y = q / g;
        if (((x + y) % 2) == 0) return 1;
        if ((x % 2) == 0) return 2;
        return 0;
        
    }
};
