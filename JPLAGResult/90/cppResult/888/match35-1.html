#include <iostream>
#include <cassert>

using namespace std;


class Solution {
public:
    int mirrorReflection(int p, int q) {

        int g = gcd(p, q);
        int k = p / g;

        int x = p * k;
        int y = q * k;
        assert(y % p == 0);

        if(isEven(y / p))
            return 0;

        if(isEven(x / p))
            return 2;
        return 1;
    }

private:
    bool isEven(int x){
        return x % 2 == 0;
    }

    int gcd(int a, int b){
        if(a > b)
            swap(a, b);

        if(b % a == 0)
            return a;
        return gcd(b % a, a);
    }
};
