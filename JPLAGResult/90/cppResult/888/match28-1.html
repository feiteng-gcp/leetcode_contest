class Solution {
public:
    int mirrorReflection(int p, int q) {
        
        // int ans = 0;
        int l = 0;
        int dir = 1;
        int lft = 0;
        while(true) {
            lft ++;
            l += dir * q;
            // ans = (ans + 1) % 3;
            if(l == p) return (lft % 2 == 0) ? 2 : 1;
            if(l > p) {
                l = p - (l - p);
                dir = -1;
            }
            if(l == 0) return 0;
            if(l < 0) {
                l = -(l);
                dir = 1;
            }
            // ans = (ans + 1) % 3;
        }
    }
};
