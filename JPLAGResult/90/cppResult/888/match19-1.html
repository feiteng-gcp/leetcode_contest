class Solution {
public:
    int mirrorReflection(int p, int q) {
        if (q == 0) return 0;
        if (q == p) return 1;
        int g = gcd(p, q);
        int y = q / g, x = p / g;
        if (y % 2 == 0) return 0;
        else {
            if (x % 2) return 1;
            else return 2;
        }
    }
    
    int gcd(int a, int b)
    {
        // Everything divides 0 
        if (a == 0 || b == 0)
           return 0;

        // base case
        if (a == b)
            return a;

        // a is greater
        if (a > b)
            return gcd(a-b, b);
        return gcd(a, b-a);
    }
};
