class Solution {
public:
    int mirrorReflection(int p, int q) {
        if (q == 0)
            return 0;
        int dx = p, dy = q;
        double x = 0, y = 0;
        int lb = -1, rb = 0, lt = 2, rt = 1, tmp;
        while (1) {
        	// printf("%g %g (%d %d %d %d)\n", x, y, lb, rb, rt, lt);
        	// sleep(1);
            double diffX = p - x;
            double diffY = diffX / dx * dy;
            if (y + diffY <= p) {
                x = p, y += diffY;
                tmp = lb, lb = rb, rb = rt, rt = lt, lt = tmp;
                x = y, y = 0;
            	swap(dx, dy);
            } else {
                x += double(p) / dy * dx;
                assert(x <= p);
                y = p;
                tmp = lb, lb = lt, lt = tmp;
                tmp = rb, rb = rt, rt = tmp;
                y = 0;
            }
            if (fabs(x) < 1e-8 && fabs(y) < 1e-8)
                return lb;
            if (fabs(x - p) < 1e-8 && fabs(y) < 1e-8)
                return rb;
            if (fabs(x - p) < 1e-8 && fabs(y - p) < 1e-8)
                return rt;
            if (fabs(x) < 1e-8 && fabs(y - p) < 1e-8)
                return lt;
        }
    }
};
