#include <algorithm>


class Solution {
public:
    int mirrorReflection(int p, int q) {
        // int gcd = p*q/(std::__gcd(p,q));
        int x= p, y = q;
        while (! (x%p == 0 && y%p == 0)) {
            x+=p;
            y+=q;
        }
        
        int h = x / p;
        int v = y / p;
        //cout << h << " " << v << endl;
        
        int a = h%2;
        int b = v%2;
        if (a==1 && b==0) return 0;
        if (a==1 && b==1) return 1;
        if (a==0 && b==1) return 2;
        assert(false);
        return 0;
    }
};
