class Solution {
public:
    int mirrorReflection(int p, int q) {
        int y = q, iter = 1, num = 0;
        while(y % p != 0) {
            iter++;
            y += q;
            if(y > p) {
                y = y % p;
                num++;
            }
        }
        if(iter % 2 == 0) {
            return 2;
        } else {
           if(num % 2 == 0)  {
               return 1;
           } else {
               return 0;
           }
        }
    }
};
