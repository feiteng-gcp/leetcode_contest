class Solution {
public:
    int mirrorReflection(int p, int q) {
        if(q == 0) return 0;
        
        int flip = 0, inc = p;
        while(true)
        {
            // cout << p << endl;
            int refTimes = p / q;
            if(p % q == 0)
            {
                if(flip % 2 == 0)
                {
                    if(refTimes % 2 == 0) return 2;
                    else return 1;
                }
                else
                {
                    if(refTimes % 2 == 0) p += inc;
                    else return 0;
                }
            }
            else 
            {
                p += inc;
                flip++;
            }
        }
        return -1;
    }
};
