class Solution {
public:
    int gcd(int p,int q) {
        int x = p, y = q, k = 1;
        while (k>0) {
            k = x%y;
            x = y;
            y = k;
        }
        return p*q/x;
    }
    int mirrorReflection(int p, int q) {
        if (q == 0) return 0;
        int len = gcd(p,q);
        int v = len/p;
        int h = len/q;
        if (v%2==0 && h%2==1) return 0;
        if (v%2==1 && h%2==1) return 1;
        if (v%2==1 && h%2==0) return 2;
    }
};
