class Solution {
public:
    int mirrorReflection(int p, int q) {
        for (int i=1; i<=q; i++) {
            if ((i*p % q)==0) {
                int j=i*p/q;
                if (j & 1) {
                    if (i & 1) return 1;
                    else return 0;
                } else return 2;
            }
        }
    }
};
