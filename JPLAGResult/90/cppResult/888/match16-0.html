class Solution {
public:
    int mirrorReflection(int p, int q) {
        if(q==0)
            return 0;
        else{
            int sum = q;
            int total = 1;
            while(sum%p!=0){
                sum += q;
                total++;
            }
            int res = sum/p;
            if(res%2==0){
                return 0;
            }
            else{
                if(total%2==1)
                    return 1;
                else
                    return 2;
            }
        }
    }
};
