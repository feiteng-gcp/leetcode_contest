class Solution {
public:
    int gcd(int a, int b) {
        if(!b) return a;
        return gcd(b, a%b);
    }
    int mirrorReflection(int p, int q) {
        if(!q) return 0;
        int g = gcd(p, q);
        int lcm = p * q / g;
        int ret1 = (lcm / p)%2;
        int ret2 = (lcm / q)%2;
        if(ret1 == 1 && ret2 == 0) return 2;
        if(ret1 == 0 && ret2 == 1) return 0;
        return 1;
       
    }
};
