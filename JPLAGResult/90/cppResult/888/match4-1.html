class Solution {
public:

    int mirrorReflection(int p, int q) {
        int div = divisor(p,q);
        int x = p/div, y = q/div;
        if(x % 2 == 0) return 2;
        if(y % 2 == 1) return 1;
        return 0;
    }
    int divisor(int p, int q) {
        if(p < q) return divisor(q,p);
        if(q == 0) return p;
        int temp = p % q;
        return divisor(q,temp);
    }
};
