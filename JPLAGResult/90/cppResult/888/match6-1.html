    int gcd(int a, int b)
{

    if (a == 0)
        return b;
    return gcd(b%a, a);
}
        
    int mirrorReflection(int p, int q) {
        
        int g = gcd(q,p);
        int n1=q/g,n2=p/g;
        
        if(n1%2==0) return 0;
        
        if(n2%2==0) return 2;
        
        return 1;
        
    }
};
