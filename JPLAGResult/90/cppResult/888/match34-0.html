class Solution {
    int gcd( int a ,int b ) {
        if (a<b) swap(a,b);
        return b?gcd(b,a%b):a;
    }
public:
    int mirrorReflection(int p, int q) {
        //p = 3, q = 2;
        int row = q/gcd(p,q);
        int col = p/gcd(p,q);
        if (row%2) return col%2?1:2;
        return 0;
    }
};
