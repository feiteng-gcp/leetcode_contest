class Solution {
public:
    int mirrorReflection(int p, int q) {
        if (p == q) return 1;
        if (q == 0) return 0;
        // x, y
        // x % p = 0, y % p = 0;
        // x1, y1
        // x1/ y1 = p / q
        // x1 % p = 0, y1 % p = 0
        // x1 = y1 * p / q
        int d = __gcd(p, q);
        
        int y1 =  p* q / __gcd(p, q);
        int x1 = y1 * p / q;
        cout << d << " " << p << " " << q << " "<< y1 << " " << x1  << endl;
        x1 /= p;
        y1 /= p;
        cout << x1 << " " << y1 << endl;
        if (x1 % 2 == 0) return 2;
        if (y1 % 2 == 0) return 0;
        return 1;
    }
};
