class Solution {
public:
    int gcd(int a, int b)
    {
        // base case
        if (a == b)
            return a;

        // a is greater
        if (a > b)
            return gcd(a-b, b);
        return gcd(a, b-a);
    }
    
    int lcm(int a, int b)
    {
        return (a*b)/gcd(a, b);
    }
    
    int mirrorReflection(int p, int q) {
        int pq_lcm = lcm(p,q);
        
        int p_num = pq_lcm / p;
        int q_steps = pq_lcm / q;
        
        if(q_steps % 2 == 0)
            return 2;
        else{
            if(p_num % 2 == 1)
                return 1;
            else
                return 0;
        }
        
    }
};
