class Solution {
public:
    int mirrorReflection(int p, int q) {
        int m = gcd(p, q);
        p = p / m;
        q = q / m;
        
        int x = p * p / gcd(p * p, q);
        int y = q * x / p;
        if (x / p % 2 == 0) {
            return 2;
        } else {
            if (y / p % 2 == 1) {
                return 1;
            } else {
                return 0;
            }
        }
    }
    int gcd(int a, int b) {
        return b == 0 ? a : gcd(b, a % b);
    }
};
