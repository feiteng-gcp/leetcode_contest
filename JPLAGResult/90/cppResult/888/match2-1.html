class Solution {
public:
    int mirrorReflection(int p, int q) {
                
        int sum = 0;
        
        while (true)
        {
            sum += q;
            
            if (sum % p == 0)
            {
                int round = sum / q;
                

                if (round % 2 == 0)
                {
                    return 2;
                }
                
                int k = sum / p;
                
                if (k % 2 == 0)
                {
                    return 0;
                }
                else
                {
                    return 1;
                }
            }
        }
        
    }
};
