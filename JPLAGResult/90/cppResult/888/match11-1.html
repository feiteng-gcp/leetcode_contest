int gcd(int a,int b)
{
    if(b==0) return a;
    return gcd(b,a%b);
}

int lcm(int a,int b)
{
    return a/gcd(a,b)*b;
}
class Solution {
public:
    int mirrorReflection(int p, int q)
    {
        if(q==0) return 0;


        int l=lcm(p,q);
        int x=l/q;
        int y=l/p;

        if(x%2==0&&y%2==1) return 2;
        if(x%2==1&&y%2==0) return 0;
        if(x%2==1&&y%2==1) return 1;
    }
};
