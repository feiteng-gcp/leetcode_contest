class Solution {
public:
    int gcd(int a, int b)
    {
        // Everything divides 0 
        if (a == 0 || b == 0)
           return 0;

        // base case
        if (a == b)
            return a;

        // a is greater
        if (a > b)
            return gcd(a-b, b);
        return gcd(a, b-a);
    }
    int mirrorReflection(int p, int q) {
        int g = gcd(p, q);
        int x = p/g, y = q/g;
        if(x%2 == 1)
            if(y%2 == 1)
                return 1;
            else
                return 0;
        else
            if(y%2 == 1)
                return 2;
            else
                return 1;
        return 0;
    }
};
