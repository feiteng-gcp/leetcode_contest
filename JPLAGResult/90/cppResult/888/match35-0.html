class Solution {
public:
    int mirrorReflection(int p, int q) {
        
        int g = __gcd( p , q );
        int m = p * q / g;
        
        int c = m / p;
        
        if( c % 2 == 1 ){
            int h = m / q;
            if( h % 2 == 1 )
                return 1;
            else return 2;
        }
        else{
            return 0;
        }
        
    }
};
