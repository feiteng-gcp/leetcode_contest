class Solution {
public:

    int mirrorReflection(int p, int q) {
    int times=1;
    int tp=p;
    while(tp%q!=0){
        tp+=p;
        ++times;
    }
    int ref=tp/q;
    if(times%2){
        if(ref%2)
            return 1;
        else
            return 2;
    }else{
        return 0;
    }
}
};
