class Solution {
public:
    int mirrorReflection(int p, int q) {
        if (q == 0) return 0;
        int g = __gcd(p, q);
        int n = p / g, m = q / g;
        n %= 2;
        m %= 2;
        if (n == 1 && m == 0) return 0;
        if (n == 1 && m == 1) return 1;
        if (n == 0 && m == 1) return 2;
        return -1;
    }
};
