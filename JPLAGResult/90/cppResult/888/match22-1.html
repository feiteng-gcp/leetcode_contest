class Solution {
public:
    int mirrorReflection(int p, int q) {
        int k = 1;
        for (; k <= 1000; k++) {
            if ((k*p) % q == 0) break;
        }

        int t = (k*p) / q;
        if (t & 1) {
            if (k & 1) return 1;
            else return 0;
        }
        else return 2;
    }
};
