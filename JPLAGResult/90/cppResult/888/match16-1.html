class Solution {
public:
    int mirrorReflection(int p, int q) {
        if (q == 0) return 0;
        int tgcd = __gcd(p, q);
        p /= tgcd;

        q /= tgcd;
        if (p % 2 == 0) {
            return 2;
        } else {
            if (q % 2) return 1;
            else return 0;
        }
    }
};
