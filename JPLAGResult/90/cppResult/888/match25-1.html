class Solution {
public:
    int mirrorReflection(int p, int q) {
        int target = gcd(p,q);
        target = p*q/target;

        if ((target/q)%2 == 0) {
            return 2;
        } else {
            if ((target/p)%2 == 0) {
                return 0;
            } else {
                return 1;
            }
        }
    }
private:
    int gcd(int a, int b){
        if (a%b == 0) {
            return b;
        }
        return gcd(b, a%b);
    }
};
