class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.length() != B.length()) {
            return false;
        }
        bool have = false;
        int state = 0;
        vector<int> v;
        for (int i = 0; i < A.length(); ++i) {
            if (A[i] != B[i]) {
                v.push_back(i);
            } else {
                int x = A[i] - 'a';
                if (state & (1 << x)) {
                    have = true;
                } else {
                    state |= 1 << x;
                }
            }
        }
        if (v.size() > 2) {
            return false;
        }
        if (v.size() == 1) {
            return false;
        }
        if (v.size() == 0) {
            return have;
        }
        return A[v[0]] == B[v[1]] && A[v[1]] == B[v[0]];
        
    }
};
