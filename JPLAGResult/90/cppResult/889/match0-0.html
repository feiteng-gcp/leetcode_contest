class Solution {
public:
    bool buddyStrings(string A, string B) {
        if(A.size()!=B.size()) return false;
        unordered_set<int> a, b;
        unordered_set<int> diff;
        for(int i = 0; i<A.size(); ++i){
            if(A[i]!=B[i]){
                a.insert(A[i]);
                b.insert(B[i]);
            }else{
                diff.insert(A[i]);
            }
        }
        if(a.size()==2 || (a.size()==0 && diff.size()<A.size()))  return a==b;
        return false;
    }
};
