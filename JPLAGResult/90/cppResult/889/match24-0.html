class Solution {
public:
	bool buddyStrings(string A, string B) {
		if (A.size() != B.size() || A.size()<2)
		{
			return false;
		}
		//记录不同的，如果两者可以交换，返回true，完全相同，返回true
		int counter = 0;
		string A1="";
		string A2="";
		for (int i = 0; i < A.size(); i++)
		{
			if (A[i] != B[i])
			{
				counter++;
				A1 += A[i];
				A2 += B[i];
			}
		}
		if (counter == 0)
		{
			//完全一样的两个字符串 //？如果只有一个字符，比如a 和 a算是什么？
			//需要有两个完全一样的字符
			map<char, int> tmp;
			for (int i = 0; i < A.size(); i++)
			{
				if (tmp.count(A[i])==1)
				{
					tmp[A[i]]++;
				}
				else
				{
					tmp[A[i]] = 1;
				}
			}
			map<char, int>::iterator it;
			for (it = tmp.begin(); it != tmp.end(); it++)
			{
				if (it->second >= 2)
				{
					return true;
				}
			}
			return false;
		}
		if (A1.size()!=2)
		{
			return false;
		}
		else
		{
			if (A1[0] == A2[1] && A1[1] == A2[0])
			{
				return true;
			}
            else
			{
				return false;
			}
		}
	}
};
