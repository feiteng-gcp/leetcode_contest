class Solution {
public:
    bool buddyStrings(string a, string b) {
        int m=a.length(), n=b.length();
        
        if (m!=n) return false;
        
        int p1, p2, found=0;
        int aux[26]={0};
        for (int i=0; i<m; i++) {
            if (a[i]==b[i]) {aux[a[i] - 'a']++; continue; }
            switch (found) {
                case 0:
                    p1=i;
                    break;
                case 1:
                    p2=i;
                    break;
                default:
                    return false;
                    break;
            }
            found++;
        }
        if ((found==2) && (a[p1]==b[p2]) && (a[p2]==b[p1])) return true;
        if (found==0) {
            if (m>26) return true;
            for (int i=0; i<26; i++) {
                if (aux[i]>=2) return true;
            }
        }
        return false;
    }
};
