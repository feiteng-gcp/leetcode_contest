class Solution {
public:
    bool buddyStrings(string A, string B) {
        int len=A.length();
        int len2=B.length();
        int cnt[30]={0}, cnt2[30]={0};
        if (len!=len2) return false;
        for(int i=0; i<len; i++) {
            cnt[A[i]-'a']++;
            cnt2[B[i]-'a']++;
        }
        int ok=0;
        for(int i=0; i<26; i++) {
            if (cnt[i]>=2) ok=1;
            if (cnt[i]!=cnt2[i]) return false;
        }

        int pos=-1, pos2=-1;
        for(int i=0; i<len; i++) {
            if (A[i]==B[i]) continue;
            if (pos==-1) pos=i;
            else if (pos2==-1) pos2=i;
            else return false;
        }
        if (pos==-1) return ok;
        if (pos2==-1) return false;
        swap(A[pos], A[pos2]);
        return A==B;
    }
};
