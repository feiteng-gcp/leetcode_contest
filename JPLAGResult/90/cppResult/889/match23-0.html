class Solution {
public:
    bool buddyStrings(string a, string b) 
    {
        if(a.size()!=b.size())
            return false;
       int x=-1,y=-1; 
        int cnt=0;
        int gone=1;
        vector<int>t(26,0);
        for(int i=0;i<a.size();i++)
        {
            t[int(a[i]-'a')]++;
            if(t[int(a[i]-'a')]>1)
            {
                gone=0;
            }
            if(a[i]!=b[i])
            {
                cnt++;
                if(x==-1)
                    x=i;
                else
                    y=i;
            }
        }
        if(cnt==0&&gone==0)
            return true;
        if(cnt!=2)
            return false;
        if(a[x]==b[y]&&a[y]==b[x])
            return true;
        return false;
    }
};
