class Solution {
public:
    bool buddyStrings(string A, string B) {
        if(A.size() != B.size()) return false;
        if(A.size() < 2) return false;
        
        unordered_set<char> dict;
        vector<int> cands;
        int cnt = 0;

        for(int i = 0; i < A.size(); i++)
        {
            dict.insert(A[i]);
            if(A[i] != B[i])
            {
                cnt++;
                cands.push_back(i);
            }
        }
        if(cnt == 0) return dict.size() < A.size();
        else if(cnt == 2)
        {
            swap(A[cands[0]], A[cands[1]]);
            return A == B;
        }
        return false;
    }
};
