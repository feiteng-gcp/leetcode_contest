class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.size() < 2 || A.size() != B.size()) return false;
        vector<int> diff_pos;
        vector<int> char_count1(26, 0), char_count2(26, 0);
        for (int i = 0; i < A.size(); ++i) {
            ++char_count1[A[i]-'a'];
            ++char_count2[B[i]-'a'];
            if (A[i] != B[i]) diff_pos.push_back(i);
        }
        bool has_multi = false;
        for (int tchar=0; tchar<26; ++tchar) {
            if (char_count1[tchar] != char_count2[tchar]) return false;
            if (char_count1[tchar] > 1) has_multi = true;
        }
        return ((has_multi && diff_pos.size() == 0) || diff_pos.size() == 2);
    }
};
