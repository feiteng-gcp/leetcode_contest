class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.size() != B.size()) return false;
        bool has[26];
        for (int i = 0; i < 26; i++) has[i] = false;
        int l = -1, r = -1;
        for (int i = 0; i < A.size(); i++) {
            if (has[A[i]-'a']) return true;
            else has[A[i]-'a'] = true;
            if (A[i] != B[i]) {
                if (l == -1) l = i;
                else if (r == -1) {
                    if (A[l] == B[i] && A[i] == B[l]) r = i;

                    else {
                        cout<<"not equal";
                        return false;
                    }
                }
                else {
                    cout<<"more than two differences";
                    return false;
                }
            }
        }
        if (l == -1 || r == -1) {
            cout<<"less than two differences l = "<<l<<", r = "<< r;
            return false;
        }
        return true;
    }
};
