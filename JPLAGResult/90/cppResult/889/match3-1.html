class Solution {
public:
    bool buddyStrings(string A, string B) {
        if(A.size()!=B.size()) return false;
        
        int count[26]={0};
        
        for(int i=0;i<A.size();i++){
            if(A[i]==B[i]){
                count[A[i]-'a']++;
                A.erase(A.begin()+i);
                B.erase(B.begin()+i);
                i--;
            }

        }
        
        if(A.size()==0){
            for(int i=0;i<26;i++){
                if(count[i]>=2) return true;
            }
        }
        
        if(A.size()!=2) return false;
        
        reverse(A.begin(),A.end());
        if(A==B) return true;
        return false;
        
    }
};
