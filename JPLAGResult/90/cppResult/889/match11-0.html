class Solution {
public:
    bool buddyStrings(string A, string B) {
        int hash[128];
        memset( hash , 0 , sizeof( hash ) ) ;
        bool swapsame = false ; 
        for( int i=0 ; i<A.size() ; i++){
            hash[A[i]]++;
            if( hash[A[i]] >=2 )
                swapsame = true ; 
        }
        for( int i=0 ; i<B.size() ; i++){
            hash[B[i]]--;
        }
        for( int i=0 ; i<128;i++){
            if( hash[i])
                return false; 
        }
        int left =-1;
        int right = -1 ; 
        for( int i=0 ; i<A.size() ; i++){
            if( A[i] != B[i] ) {
                if( left == -1 ) 
                    left = i ; 
                else if( right ==-1 )
                    right =i ;
                else
                    return false ; 
            }
        }
        if( left ==-1 && right ==-1 && swapsame )
            return true  ; 
        if( left !=-1 && right !=-1 ){
            swap( A[left], A[right] ) ;
            return A==B ;
        }
        return false ; 
    }
};
