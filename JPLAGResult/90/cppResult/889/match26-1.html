class Solution {
public:
    bool buddyStrings(string A, string B) {
        int l = A.length();
        if(l!=B.length())return 0;
        int p = 0;
        char diff[4][2];
        for(int i=0;i<l;i++){
            if(A[i]!=B[i]){
                diff[p][0] = A[i];
                diff[p++][1] = B[i];
                if(p==3)return 0;
            }
            
        }
        if(p==2 && diff[0][0] == diff[1][1] && diff[0][1] == diff[1][0]){
            return 1;
        }

        else if(p==0){
            for(int i=0;i<l;i++){
                for(int j=0;j<l;j++){
                    if(A[i]==B[j] && i!=j){
                        return 1;
                    }
                }
            }
        }
        return 0;
    }
};
