class Solution {
public:
    bool buddyStrings(string A, string B) {
        int na = A.size();
        int nb = B.size();
        
        if (na != nb) return false;
        vector<int> pos;
        for (int i = 0; i < na; i += 1) {
            if (A[i] != B[i]) pos.push_back(i);
        }
        
        if (pos.size() > 2) return false;
        else if (pos.size() == 2) {
            if (A[pos[0]] == B[pos[1]] && A[pos[1]] == B[pos[0]])
                return true;
            else
                return false;
        }
        else {
            vector<int> cnt(128, 0);
            for (int i = 0; i < na; i += 1) {
                if (++cnt[A[i]] >= 2) return true;
            }
            return false;
        }
    }
};
