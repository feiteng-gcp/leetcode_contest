class Solution {
public:
    bool buddyStrings(string A, string B) {
        if(A.size()!=B.size() || A.size()<2 || B.size()<2) return false;
        vector<int> v(26,0);
        bool has_same=false;
        int diff=0,first=-1,second=-1;
        for(int i=0;i<A.size();++i){
            if(++v[A[i]-'a']>1) has_same=true;
            if(A[i]!=B[i]){
                ++diff;

                if(diff==1) first=i;
                else if(diff==2) second=i;
                else return false;
            }
        }
        if(first==-1 && second==-1) return has_same;
        if(first!=-1 && second!=-1) return A[first]==B[second] && A[second]==B[first];
        return false;
    }
};
