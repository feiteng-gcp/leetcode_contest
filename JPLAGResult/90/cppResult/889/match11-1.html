class Solution {
public:
    bool buddyStrings(string A, string B) {
        int p = -1, p2 = -1;
        if (A.size() != B.size()) return false;
        if (A == B) {
            map<char, bool> mp;
            for (auto &ch: A) {
                if (mp[ch] == 0) mp[ch] = 1;
                else return true;
            }
            return false;
        }

        for (int i = 0; i < A.size(); i++) {
            if (A[i] != B[i]) {
                if (p == -1) p = i;
                else if(p2 == -1) p2 = i;
                else return false;
            }
        }
        if (A[p] == B[p2] && A[p2] == B[p]) return true;
        return false;
    }
};
