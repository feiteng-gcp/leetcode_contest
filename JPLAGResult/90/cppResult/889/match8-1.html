class Solution {
public:
	bool buddyStrings(string A, string B) {
		string C = A;
		int count = 0;
		vector<int>res;
		sort(C.begin(), C.end());
        if (A.length() == 1)return false;
		if (A==B)
		{
			
			for (int i = 1; i<C.length(); i++)
			{
				if (C[i] == C[i - 1])return true;
			}
            return false;
		}
       
		for (int i = 0; i < A.length(); i++)
		{

			if (A[i] != B[i])
            {count++;
			res.push_back(i);
            }
		}
		if (count != 2)return false;
		else
		{
			if (A[res[0]] == B[res[1]] && A[res[1]] == B[res[0]])
				return true;
            return false;
		}
	}
};
