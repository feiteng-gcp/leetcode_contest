class Solution {
public:
    bool buddyStrings(string A, string B) {
        if(A.size()!=B.size()){
            return false;
        }
        if(A==B){
            unordered_map<char, int> cc;
            for(char c:A){
                cc[c]++;
                if(cc[c]>=2){
                    return true;
                }
            }
        }
        vector<int> diff;
        for(int i=0; i<A.size(); i++) {
            if(A[i]==B[i]){
                continue;
            }else{
                diff.push_back(i);
            }

        }
        if(diff.size()!=2){
            return false;
        }else{
            if(A[diff[0]]==B[diff[1]] && A[diff[1]]==B[diff[0]]){
                return true;
            }else{
                return false;
            }
        }
    }
};
