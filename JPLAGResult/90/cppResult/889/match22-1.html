class Solution {
public:
    bool buddyStrings(string A, string B) {
        if(A.length() != B.length())
            return false;
        int p1 = -1, p2 = -1;
        map<char, int> cnt;
        bool flag = 0;
        for(int i = 0; i < A.length(); i++){
            cnt[A[i]]++;
            if(cnt[A[i]] > 1){
                flag = 1;
            }
            if(A[i] != B[i]){
                if(p2 >= 0)
                    return false;
                if(p1 >= 0){
                    p2 = i;
                }
                else {
                    p1 = i;
                }
            }
        }
        if(p1 < 0){
            return flag;
        }
        return A[p1] == B[p2] && A[p2] == B[p1];
    }
};
