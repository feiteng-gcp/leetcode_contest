class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.length() != B.length()) {
            return false;
        }
        char mp[26]{0};
        char c1 = -1;
        char c2 = -2;
        int cnt = 0;
        bool hasSame = false;
        bool ret = false;
        for (int i = 0; i < A.length(); ++i) {
            if (A[i] == B[i]) {
                mp[A[i] - 'a']++;
                if (mp[A[i] - 'a'] >= 2) {
                    hasSame = true;
                }
            } else {
                ++cnt;
                if (cnt == 1) {
                    c1 = A[i];
                    c2 = B[i];
                } else if (cnt == 2) {
                    if (B[i] == c1 && A[i] == c2) {
                        ret = true;
                    } else {
                        return false;
                    }
                } else {
                    return false;
                }
            }
        }
        if (cnt == 2 && ret) {
            return true;
        }
        if (cnt == 0 && hasSame) {
            return true;
        }
        return false;
    }
};
