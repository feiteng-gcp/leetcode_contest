class Solution {
public:
    bool buddyStrings(string A, string B) {
        const int sizeA = A.length();
        const int sizeB = B.length();
        if (sizeA != sizeB) return false;

        int cnt[26] = {0};
        vector<int> diffs;
        bool havedup = false;
        for (int i=0; i<sizeA; ++i)
        {
            if (A[i] != B[i])   {
                diffs.push_back(i);
                if (diffs.size() > 2)    return false;
            }
            else if (++cnt[A[i]-'a'] >= 2) havedup = true;
        }
        if (diffs.empty() and havedup)  return true;
        if (diffs.size() == 2 and A[diffs[0]] == B[diffs[1]] and A[diffs[1]] == B[diffs[0]])    return true;
        return false;
    }
};
