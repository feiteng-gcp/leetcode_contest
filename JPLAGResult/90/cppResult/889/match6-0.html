class Solution {
public:
    bool buddyStrings(string A, string B) {
        int lena = A.size();
        int lenb = B.size();
        if (lena != lenb) return false;
        if (lena == 0) return false;
        
        int diffindex = -1;
        vector<int> count(26, 0);
        bool dup = false;
        for (int i = 0; i < lena; i++)
        {
            if (A[i] != B[i])
            {
                if (diffindex == -1)
                {
                    diffindex = i;
                }
                else if (diffindex == -2) return false;
                else if (A[i] == B[diffindex] && B[i] == A[diffindex])
                {
                    diffindex = -2;
                }
                else return false;
            }
            
            count[A[i] - 'a'] ++;
            if (count[A[i] - 'a'] > 1) dup = true;
        }
        
        cout << diffindex << " " << dup << endl;
        return (diffindex == -1 && dup) || diffindex == -2;
    }
};
