class Solution {
public:
    bool buddyStrings(string A, string B) {
        int n = A.size(), m = B.size();
        if (n != m) return false;
        vector<pair<char, char>> a;
        for (int i = 0; i < n; ++i) {
            if (A[i] == B[i]) continue;
            a.push_back({A[i], B[i]});
        }
        if (a.size() == 0) {
            vector<int> cnt(26);
            for (int i = 0; i < n; ++i) {
                cnt[A[i] - 'a']++;
            }
            for (int k = 0; k < 26; ++k) {
                if (cnt[k] >= 2) return true;
            }
            return false;
        }
        if (a.size() != 2) return false;
        swap(a[1].first, a[1].second);
        return a[0] == a[1];
    }
};
