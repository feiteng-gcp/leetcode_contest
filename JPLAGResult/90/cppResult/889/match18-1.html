class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.length() != B.length()) {
            return false;
        }
        int diffcnt = 0;
        char fsta, snda, fstb, sndb;
        unordered_set<char> charset;
        bool dup = false;
        for (int i = 0; i < A.length(); i++) {
            if (charset.count(A[i])) {
                dup = true;
            }
            charset.insert(A[i]);
            if (A[i] != B[i]) {
                if (diffcnt == 0) {
                    fsta = A[i];
                    fstb = B[i];
                } else if (diffcnt == 1) {
                    snda = A[i];
                    sndb = B[i];
                } else {
                    return false;
                }
                diffcnt++;

            }
        }
        if (diffcnt == 1) { return false; }
        if (diffcnt == 2 && fsta == sndb && fstb == snda) {
            return true;
        }
        if (diffcnt == 0 && dup) {
            return true;
        }
        return false;
    }
};
