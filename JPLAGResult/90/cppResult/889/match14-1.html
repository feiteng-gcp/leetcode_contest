class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.length() != B.length())
            return false;
        
        int len = A.length();
        vector<int> diffPos;
        int letterMap[26] = {0};
        
        for (int i = 0; i < len; i++) {
            if (A[i] != B[i]) {
                diffPos.push_back(i);
            }
            letterMap[A[i] - 'a']++;
        }
        
        if (diffPos.size() == 2)
            if (A[diffPos[0]] == B[diffPos[1]] && A[diffPos[1]] == B[diffPos[0]])
                return true;
        
        if (diffPos.size() == 0) {

            for (int let = 0; let < 26; let++) {
                if (letterMap[let] > 1)
                    return true;
            }
        }
        
        return false;
    }
};
