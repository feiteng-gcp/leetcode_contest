class Solution {
public:
    bool buddyStrings(string A, string B) {
        if(A.length() != B.length())
            return false;
        
        int count = 0;
        int sum = 0;
        char A1, A2, B1, B2;
        bool repeated = false;
        set<char> C;
        

        for(int i = 0; i < A.length(); ++i){
            
            if(A[i] != B[i]){
                count++;
                if(count > 2)
                    return false;
                if(count == 1){
                    A1 = A[i];
                    B1 = B[i];
                }else{
                    A2 = A[i];
                    B2 = B[i];
                }
                //sum += A[i] - B[i];
            }
            if(C.find(A[i]) != C.end()){
                repeated = true;
            }
            C.insert(A[i]);
        }
        cout << count << "  " << repeated << "\n";
        cout << A1 << " " << A2 << " " << B1 << " " << B2 << "\n";
        if(count == 2 && A1 == B2 && A2 == B1)
            return true;
        else{
            if(count == 0 && repeated)
                return true;
            return false;
        }
    }
};
