class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.size() != B.size()) return false;
        int i1 = -1;
        int i2 = -1;
        
        for (int i = 0; i < A.size(); i++)
        {
            if (A[i] == B[i]) continue;
            if (i1 == -1) i1 = i;
            else if (i2 == -1) i2 = i;
            else return false;
        }
        
        if (i1 != -1 && i2 == -1) return false;
        if (i1 == -1)
        {
            unordered_set<int> hash;
            for (char ch : A)
            {
                auto iter = hash.find(ch);
                if (iter == hash.end())
                {
                    hash.insert(ch);
                }
                else return true;
            }
            return false;
        }
        
        return A[i1] == B[i2] && A[i2] == B[i1];
    }
};
