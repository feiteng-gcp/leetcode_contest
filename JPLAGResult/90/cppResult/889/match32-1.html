class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.length() != B.length()) {
            return false;
        }
        int difs = 0;
        int d1 = -1;
        int d2 = -1;
        for (int i = 0; i < A.size(); i++) {
            if (A[i] != B[i]) {
                difs++;
                if (d1 == -1) d1 = i;
                else d2 = i;
            }
        }
        
        if (difs == 0) {
            int alpha[26] = {0};

            for (int i = 0; i < A.size(); i++) {
                alpha[A[i] - 'a']++;
            }
            for (int i = 0; i <26; i++) {
                if (alpha[i] > 1) {
                    return true;
                }
            }
        } else if (difs == 2) {
            swap(A[d1], A[d2]);
            if (A == B) {
                return true;
            }
        }
        return false;
    }
};
