class Solution {
public:
    bool buddyStrings(string A, string B) {
        if(A.size()!=B.size()) return false;
        
        int diff=0;
        bool uniq = true;
        set<char> s;
        int a=-1,b=-1;
        for(int i=0;i<A.size();i++){
            if(A[i]!=B[i]){
                if(a==-1) a=i;
                else b=i;
                diff++;
            }
            if(s.count(A[i])) uniq = false;
            s.insert(A[i]);
        }
        if(diff==0&&!uniq) return true;
        if(diff!=2) return false;
        
        swap(A[a],A[b]);
        return A==B;
    }
};
