class Solution {
public:
    bool buddyStrings(string A, string B) {
        if(A.size()  != B.size()) return false;
        
        vector<int> cnts(26, 0);
        for(int i=0;i<A.size();i++){
            ++cnts[A[i]-'a'];
            if(A[i] != B[i]){
                bool found = false;
                int j;
                for(j=i+1;j<A.size();j++){
                    if(B[j] != A[i]) continue;
                    if(A[j] == B[i]) { found = true; break; }
                }
                if(!found) return false;
                swap(A[i], A[j]);
                for(j=i+1;j<A.size();j++) if(A[j] != B[j]) return false;
                return true;
            }
        }
        
        for(int i=0;i<26;i++) if(cnts[i] > 1) return true;
        
        return false;
    }
};
