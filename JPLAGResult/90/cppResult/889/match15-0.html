class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.length() != B.length()) {
            return false;    
        }
        int diff[2] = {0};
        int index = 0;
        for (int i = 0; i < A.length(); i++) {
            if (A.at(i) != B.at(i)) {
                if (index == 2)
                    return false;
                diff[index] = i;
                index++; 
            }       
        }
        if (index == 2 && A.at(diff[0]) == B.at(diff[1]) && A.at(diff[1]) == B.at(diff[0])) {
            return true;
        }
        if (index == 0) {
            for (int i = 0; i < A.length(); i++)
                for (int j = i + 1; j < A.length(); j++) {
                    if (A.at(i) == B.at(j))
                        return true;
                }
        }
        return false;
    }
};
