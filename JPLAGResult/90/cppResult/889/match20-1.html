class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.size()!=B.size()) return false;

        char a,b;
        int diffs=0;
        bool dup=false;
        vector<int> count(26,0);

        for (int i=0; i<A.size(); i++) {
            if (++count[A[i]-'a']>1) dup=true;
            if (A[i]!=B[i]) {
                diffs++;
                if (diffs>2) return false;
                if (diffs==1) a=A[i], b=B[i];
                else if (A[i]!=b || B[i]!=a) return false;
            }
        }
        if (diffs==2 || (diffs==0 && dup)) return true;
        else return false;
    }
};
