class Solution {
public:
    bool buddyStrings(string A, string B) {
        if (A.size() != B.size())
            return false;
        if (A.size() < 2 || B.size() < 2)
            return false;
        map<int,char> mpa,mpb;
        for (int i = 0;i != A.size();i ++)
        {
            if (A[i] != B[i])
            {
                mpa[i] = A[i];
                mpb[i] = B[i];
            }
            if (mpa.size() > 2)
            	return false;
        }
        if (mpa.size() == 2)
        {
        char a[2],b[2];
        int i = 0;
        for (auto ita = mpa.begin();ita != mpa.end();ita++)
            a[i++] = ita->second;
        i = 0;
        for (auto itb = mpb.begin();itb != mpb.end();itb++)
            b[i++] = itb->second;
        if (a[0] == b[1] && a[1] == b[0])
            return true;
    	}
    	else if (mpa.size() == 0)
    	{
    		sort(A.begin(),A.end());
    		for (int i = 1;i != A.size();i ++)
    		{
    			if (A[i] == A[i-1])
    				return true;
    		}
    	}
        return false;
    }
};
