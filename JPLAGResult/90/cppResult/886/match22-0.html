class Solution {
public:
	int scoreOfParentheses(string S) {
		return getScore(S, 0, S.length());
	}

	int getScore(const string& S, int l, int r)
	{
		if (l == r)return 0;
		if (l == r - 2)return 1;

		int res = 0;
		int cnt = 0;
		for (int i = l; i<r; ++i)
		{
			if (S[i] == '(')
			{
				++cnt;
			}
			else
			{
				--cnt;
				if (cnt == 0)
				{
					int tmp = getScore(S, l + 1, i);
					if (tmp == 0)res += 1;
					else res += 2 * tmp;
					l = i + 1;
				}
			}
		}
		return res;
	}
};
