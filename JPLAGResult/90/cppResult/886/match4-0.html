class Solution {
public:
    int scoreOfParentheses(string S) {
        stack<int> stk;
        for (auto c : S) {
            if (c == '(') {
                stk.push(-1);
            } else {
                if (stk.top() == -1) {
                    stk.pop();
                    stk.push(1);
                } else {
                    int sum = 0;
                    while (stk.size() && stk.top() != -1) {
                        sum += stk.top();
                        stk.pop();
                    }
                    stk.pop();
                    stk.push(sum * 2);
                }
            }
        }
        int res = 0;
        while (stk.size()) {
            res += stk.top();
            stk.pop();
        }
        return res;
    }
};
