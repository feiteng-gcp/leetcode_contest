class Solution {
public:
    int scoreOfParentheses(string S) {
		int n = S.length();
		if (n == 2)
			return 1;
		int count = 1;
		for (int i = 1; i < n; i++)
		{
			if (S[i] == '(')
				count++;
			else
				count--;
			if (count == 0)
			{
				if (i < n - 1)
					return scoreOfParentheses(S.substr(0, i + 1)) + scoreOfParentheses(S.substr(i + 1));
				else
					return 2 * scoreOfParentheses(S.substr(1, n - 2));
			}
		}
		return 0;
	}
};
