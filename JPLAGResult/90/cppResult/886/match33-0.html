class Solution {
public:
    int scoreOfParentheses(string S) {
        int score = 0;
        if(S == "()") return 1;
        
        int p = 0, c = 1;
        for(int i=1; i<S.size(); ++i) {
            if(S[i] == '(')
                ++c;
            else
                --c;
            
            if(c == 0) {
                if(i == S.size()-1 && p == 0)
                    score = 2 * scoreOfParentheses(S.substr(1, S.size()-2));
                else {
                    score += scoreOfParentheses(S.substr(p, i-p+1));
                    p = i+1;
                }
            }
        }
        
        return score;
    }
};
