class Solution {
public:
    int scoreOfParentheses(string S) {
        S = '(' + S + ')';
        stack<int> s;
        int l = S.length();

        for(int i = 0;i<l;i++){
            if(S[i]=='('){
                s.push(-1);
            }
            else{
                if(s.top()==-1){
                    s.pop();
                    s.push(1);
                }
                else{
                    int temp=0;
                    while(s.top()!=-1){
                        temp+=s.top();
                        s.pop();
                    }
                    s.pop();
                    s.push(temp*2);
                }
                
                
            }
        }
        return s.top()/2;
    }
};
