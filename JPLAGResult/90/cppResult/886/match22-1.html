class Solution {
public:
    int scoreOfParentheses(string S) {
        int start=0,end=0;
        int n = S.size();
        int count = 0;
        int result = 0;

        while(end<n){
            if(S[end]=='(')
            {
                count++;
            }else{
                count--;
                if(count==0)
                {
                    int sz = end-start+1;
                    if(sz==2) result += 1;
                    else result += 2 * scoreOfParentheses(S.substr(start+1,sz-2));
                    start = end+1;
                }
            }
            end++;
        }
        return result;
    }
};
