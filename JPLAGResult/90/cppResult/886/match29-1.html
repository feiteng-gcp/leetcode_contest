class Solution {
public:
    int scoreOfParentheses(string S) {
        int score = 0, N = S.size();
        stack<int> st;

        vector<int> record(N, 0);
        for (int i = 0; i < N; ++i){
            if (S[i] == '(')
                st.push(i);
            else {
                int prev = st.top(); st.pop();
                if (prev == i-1)
                    record[i] = 1;
                else {
                    int sum = 0;
                    for (int j = prev; j < i; ++j){
                        sum += record[j];
                        record[j] = 0;
                    }
                    record[i] = sum*2;
                }
            }
        }
        return accumulate(record.begin(), record.end(), 0);
    }
};
