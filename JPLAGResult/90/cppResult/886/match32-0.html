class Solution {
public:
  int scoreOfParentheses(string S) {
    vector<int> v(S.size(), 0);
    int pos = 0;
    for (int i = 0; i < S.size(); ++i) {
      if (S[i] == '(') {
        ++pos;
        v[pos] = 0;
      } else {
        v[pos-1] += v[pos] == 0 ? 1 : 2*v[pos];
        --pos;
      }
    }
    return v[0];
  }
};
