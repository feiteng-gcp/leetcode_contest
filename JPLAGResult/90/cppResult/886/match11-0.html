class Solution {
public:
    int scoreOfParentheses(string S) {
        cout << S << endl;
        if (S == "()"){
            return 1;
        }
        if (S == ""){
            return 0;
        }
        bool closed = false;
        int i = 1; 
        int numOpen = 1;
        while (numOpen > 0){
            if (S[i] == ')'){
                numOpen--;
            }
            else{
                numOpen++;
            }
            i++;
        }
        if (i == 2){
            return 1 + scoreOfParentheses(S.substr(i, S.length() - i));
        }
        else{
            return 2 * scoreOfParentheses(S.substr(1, i - 2)) + scoreOfParentheses(S.substr(i, S.length() - i));
        }
        
    }
};
