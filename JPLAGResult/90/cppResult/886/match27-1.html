class Solution {
public:
    int scoreOfParentheses(string S) {
        std::stack<int> ps;
        std::map<int, int> ones;
        char lp = '(', rp = ')';

        for(int i = 0; i < S.length(); i++) {
            if(S[i] == lp) {
                ps.push(i);
            } else {
                if(ps.top() == i-1) {
                    ones[i] = 1;
                } else {
                    auto itlo = ones.lower_bound(ps.top());
                    auto itup = ones.upper_bound(i);
                    for(auto it = itlo; it != itup; ++it) {
                        it->second = it->second * 2;
                    }                    
                }
                ps.pop();
            }
        }
        int score = 0;
        for(auto it = ones.begin(); it != ones.end(); ++it) {
            score += it->second;
        }
        return score;
    }
};
