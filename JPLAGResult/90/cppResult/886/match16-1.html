class Solution {
public:
    int scoreOfParentheses(string S) {
        vector<int> stk;
        int res = 0, temp, temp2 = 0;
        for (int i = 0; i<S.size(); i++){
            if (S[i] == '('){
                stk.push_back(-1);
                continue;
            }
            temp = stk.back();
            stk.pop_back();
            if (temp == -1)
                stk.push_back(1);
            else {
                stk.pop_back();
                stk.push_back(temp*2);
            }
            if (stk.size() >= 2 && stk[stk.size()-2] != -1){
                stk[stk.size()-2] += stk[stk.size()-1];
                stk.pop_back();
            }
        }
        return stk.back();
    }
};
