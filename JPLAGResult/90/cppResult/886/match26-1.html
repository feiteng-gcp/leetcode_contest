class Solution {
public:
    int scoreOfParentheses(string S) {
        stack<int> st;
        st.push(0);
        for(int i=0; i<S.size(); i++) {
            if(S[i] == '(') st.push(0);
            else {
                int front = st.top();
                st.pop();
                if(front == 0) front = 1;
                else front *= 2;
                st.top() += front;
            }
        }
        return st.top();
    }
};
