class Solution {
public:
    int scoreOfParentheses(string s) {
        int res = 0;
        int left = 0;
        int open = 0;
        for(int i=0;i<s.size();i++){
            if(s[i]=='(') open++;
            else{
                open--;
                if(open==0){
                    int len = i-left + 1;
                    if(len==2){
                        res++;
                    }
                    else{
                        string cur = s.substr(left+1,i-1-(left+1)+1);
                        res += 2* scoreOfParentheses(cur);
                    }
                    left = i+1;
                }
            }
        }
        return res;
        
    }
    
};
