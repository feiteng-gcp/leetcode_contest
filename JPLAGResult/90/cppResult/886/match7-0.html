class Solution {
public:
    int scoreOfParentheses(string S) {
        stack<int> s;
        for (char c: S) {
            if (c=='(') s.push(0);
            else {
                int sum=0;
                while (s.top()>0) {
                    sum+=s.top();
                    s.pop();
                }
                s.pop(); // pop 0
                if (sum) s.push(sum*2);
                else s.push(1);
            }
        }
        int score=0;
        while (s.size()) {
            score += s.top();
            s.pop();
        }
        return score;
    }
};
