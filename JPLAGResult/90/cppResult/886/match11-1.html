class Solution {

public:
    int scoreOfParentheses(string S) {
        //S = "(()(()))";
        return tmp(S);
    }
private:
    int tmp(string S){    
        int len = S.length();
        if (len == 2) {
            return 1;
        }

        int cnt = 1;
        int pos = 1;
        while(pos < len){
            if (S[pos] == '(') {
                cnt++;
            } else {
                cnt--;
                if (cnt == 0) {
                    break;
                }

            }
            pos++;
        }
        //cout<<"pos: "<<pos<<endl;
        if (pos == len-1) {
            return 2*tmp(S.substr(1,len-2));
        } else {
            return tmp(S.substr(0, pos+1))+tmp(S.substr(pos+1));
        }
    }
};
