class Solution {
public:
    int scoreOfParentheses(string S) {
        stack<int> stk;
        int result = 0;
        for(int i=0;i<S.size();++i){
            if(S[i] == '(')
                stk.push(-1);
            else{
                vector<int> tmp;
                while(stk.top()!=-1){
                    tmp.push_back(stk.top());
                    stk.pop();
                }
                stk.pop();
                if(tmp.size()==0){
                    stk.push(1);
                }
                else if(tmp.size()==1){
                    stk.push(2*tmp[0]);
                }
                else {
                    int res = 0;
                    for(int i=0;i<tmp.size();++i)
                        res += tmp[i];
                    stk.push(2*res);
                }
            }
        }
        while(stk.size() > 0){
            result += stk.top();
            stk.pop();
        }
        return result;
    }
};
