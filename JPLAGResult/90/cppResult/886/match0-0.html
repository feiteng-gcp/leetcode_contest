class Solution {
public:
    map< int ,int > check ; 
    int cal ( string &  S , int left , int right ){
        if( left > right )
            return 0 ; 
        if( left +1 == right )
            return 1 ; 
        int res = 0 ; 
        for( int i=left+1 ;i<right ; i++){
            int t = check[i];
            int temp = cal( S , i , check[i] ) ; 
            i = check[i] ;
            res += temp ; 
        }
        return 2*res ; 
    }
    int scoreOfParentheses(string S) {
        if( S.size() == 0  )
            return 0 ;
        stack< pair <char , int >  > slist ;
        slist.push( {S[0] , 0 }) ; 
        for( int i=1 ; i< S.size() ; i++){
            if (slist.size() ==0 ){
                slist.push( {S[i] , i } ) ; 
            } 
            else{
                char c = slist.top().first ; 
                if( c =='('&&S[i] ==')' ){
                    check[slist.top().second ] = i ; 
                    slist.pop();
                }
                else{
                    slist.push({S[i] , i} ) ; 
                }
            }
        }
        int res = 0 ; 
        for( int i=0 ; i< S.size() ; i++){
            res += cal( S, i , check[i] ) ; 
            i = check[i] ; 
        }
        return res ; 
    }
};
