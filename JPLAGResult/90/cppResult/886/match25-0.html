class Solution {
public:
    int scoreOfParentheses(string S) {
        stack<int> st;
        int ans = 0, prev;
        for(int i=0; i<S.length(); i++)
        {
            if(S[i] == '(')st.push(i);
            else if(st.size() == 1) {
                prev = st.top();
                st.pop();
                int len = i - prev - 1;
                if(len == 0)ans++;
                else
                {
                    string s2 = S.substr(prev+1,len);
                    ans = ans + 2*scoreOfParentheses(s2);
                }
                
            }
            else st.pop();
        }
            
        return ans;
        
    }
};
