class Solution {
public:
    int scoreOfParentheses(string S) {
        stack<int> sta;
        int res = 0, last = -1;
        for(int i = 0; i < S.size(); ++i) {
            char c = S[i];
            if(c == '(') sta.push(i);
            else {
                int num = sta.top();
                sta.pop();
                if(num > last) res += (1 << sta.size());
                last = num;
            }
        }
        return res;
    }
};
