class Solution {
public:
    int to[60];
    int dfs(int l, int r) {
        if (l >= r) return 0;
        if (l + 1 == r) return 1;
        if (to[l] == r) return 2 * dfs(l+1, r-1);
        int ans = 0;
        for (int i = l; i <= r; i = to[i] + 1) {
            ans += dfs(i, to[i]);
        }
        return ans;
    }

    int scoreOfParentheses(string S) {
        stack<int> stk;
        for (int i = 0; i < S.size(); i++) {
            if (S[i] == '(') stk.push(i);
            else {
                int t = stk.top(); stk.pop();
                to[t] = i; to[i] = t;
            }
        }
        return dfs(0, S.size() - 1);
    }
};
