class Solution {
public:
    string s;
    
    int f(int x,int y){
        if( y <= x )
            return 1;
        if( x + 1 == y )
            return 1;
        int t = 0;
        for(int i = x ; i <= y ; i++){
            if( s[i] == '(' ) t++;
            else t--;
            if( t == 0 ){
                if( i == y ) return 2 * f( x + 1 , y - 1 );
                else return f( x , i ) + f( i + 1 , y );
            }
        }
        
    }
    
    int scoreOfParentheses(string S) {
        s = S;
        return f( 0 , s.size() - 1 );
    }
};
