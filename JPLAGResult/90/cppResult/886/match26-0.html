class Solution {
public:
    int scoreOfParentheses(string S) {
        vector<int> left(1, 0);
        for (int i = 0; i < S.length(); i++) {
            if (S[i] == '(') {
                left.push_back(0);
            } else {
                if (S[i - 1] == '(') {
                    left.pop_back();
                    left.back()++;
                } else {
                    auto v = left.back();
                    left.pop_back();
                    left.back() += 2 * v;
                }
            }
        }
        return left[0];
    }
};
