class Solution {
public:
    int scoreOfParentheses(string S) {
        
        stack<string> sol;
        int cont=0,c;
        for(int i=0;i<S.size();i++){
            if(S[i]=='('){
                sol.push("(");
            }
            else{
                if(sol.top()=="("){
                    sol.pop();
                    sol.push("1");
                }
                else{
                    int count=0;
                    while(sol.top()!="("){
                        count+=stoi(sol.top());
                        sol.pop();
                    }
                    sol.pop();
                    sol.push(to_string(count*2));
                }
            }
        }
        int ans=0;
        while(!sol.empty()){
            ans+=stoi(sol.top());
            sol.pop();
        }
               return ans; 
    }
};
