class Solution {
public:
    int scoreOfParentheses(string S) {
        if(S=="")return 0;

        if(S=="()")return 1;
        int lvl=0;
        for(int i=0;i<S.size();i++){
            if(S[i]=='(')lvl++;
            else lvl--;
            if(lvl==0){
                int l=1,r=0;
                if(i>1)l=scoreOfParentheses(S.substr(1,i-1))*2;
                if(i<S.size()-1)r=scoreOfParentheses(S.substr(i+1,S.size()-i-1));
                return l+r;
            }
        }
    }
};
