class Solution {
public:
    int scoreOfParentheses(string S) {
        stack<int> za;
        for(auto i:S){
            if(i=='('){
                za.push(-1);
            }
            else{
                int tmp=0,tpa=za.top();
                while(tpa!=-1){
                    za.pop();
                    if(za.empty())      return -1;
                    tmp+=tpa;
                    tpa=za.top();
                }
                za.pop();
                if(!tmp){
                    za.push(1);
                }
                else        za.push(2*tmp);
            }
        }
        int rz=0;
        while(!za.empty()){
            rz+=za.top();
            za.pop();
        }
        return rz;
    }
};
