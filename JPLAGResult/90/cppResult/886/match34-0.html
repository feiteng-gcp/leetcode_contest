class Solution {
public:
    int scoreOfParentheses(string S) {
        if(S == "") return 0;
        if(S == "()") return 1;
        int len = sz(S);
        
        int cnt = 0;
        bool found = false;
        int ind = 0;
        //cout<<S<<endl;
        while(ind < len && (cnt || !found)) {
            //cout<<"ind: "<<ind<<" "<<S[ind]<<endl;
            if(S[ind] == '(') {
                cnt++;
                found = true;
            } else if (S[ind] == ')') {
                cnt--;
                found = true;
            }
            ind++;
        }
        if(ind == len) {
            if(S[len-1] == ')' && S[0] == '(') return 2*scoreOfParentheses(S.substr(1, len-2));

            return 0;
        }

        //cout<<S.substr(0, ind)<<"###"<<S.substr(ind, len-ind)<<endl;


        return scoreOfParentheses(S.substr(0, ind)) + scoreOfParentheses(S.substr(ind, len-ind));
    }
};
