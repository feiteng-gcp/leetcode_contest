class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A == null || B == null) {
            return A == B;
        }
        if (A.length() != B.length()) {
            return false;
        
}
        if (A.equals(B)) {
            Set<Character> set = new HashSet<>();
            for (int i = 0; i < A.length(); i ++) {
                if (set.contains(A.charAt(i))) {
                    return true;
                }
                set.add(A.charAt(i));
            }
            
return false;
        }
        List<Integer> diff = new ArrayList<>();
        for (int i = 0; i < A.length(); i ++) {
            if (A.charAt(i) != B.charAt(i)) {
                diff.add(i);
            }
        }
        if (diff.size() != 2) {
            return false;
        }
        return A.charAt(diff.get(0)) == B.charAt(diff.get(1)) &&  B.charAt(diff.get(0)) == A.charAt(diff.get(1));
    }
}
