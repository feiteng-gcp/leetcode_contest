class Solution {
 public boolean buddyStrings(String A, String B) {
        if (A.length() != B.length()) return false;
        int cnt = 0;
        int[] pos = new int[5];
        for (int i = 0; i < A.length(); i++) {
            if (A.charAt(i) != B.charAt(i)) {
                pos[cnt++] = i;
                if (cnt > 2) return false;
            }
        }
        if (cnt == 0) {
            Set<Character> chs = new HashSet<Character>();
            for (char ch : A.toCharArray()) chs.add(ch);
            if (chs.size() != A.length()) return true;
        }
        return (cnt == 2) && (A.charAt(pos[0]) == B.charAt(pos[1]) && A.charAt(pos[1]) == B.charAt(pos[0]));
    }
}
