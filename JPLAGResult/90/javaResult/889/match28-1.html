class Solution {
    public boolean buddyStrings(String A, String B) {
        if(A.length() != B.length()) return false;
        
        char[] ac = A.toCharArray(), bc = B.toCharArray();
        int len = ac.length, idx = 0;
        int[] cnt = new int[26];
        
        while(idx < len && ac[idx] == bc[idx]){
            ++cnt[ac[idx]-'a'];
            ++idx;
        }
        if(idx == len){
            
for(int i = 0; i < 26; ++i){
                if(cnt[i] > 1) return true;
            }
            
            return false;
        }
        
        int first = idx;
        ++idx;
        
        while(idx < len && ac[idx] == bc[idx]) ++idx;
        
        int second = idx;
        if(second == len || ac[first] != bc[second] || bc[first] != ac[second]) return false;
        
        ++idx;
        while(idx < len){
            if(ac[idx] != bc[idx]) return false;
            ++idx;
        }
        
        return true;
    }
}
