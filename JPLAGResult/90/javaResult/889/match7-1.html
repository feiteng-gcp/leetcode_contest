class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.length()!=B.length()) return false;
        int count=0;
        int[] map=
new int[26];
        char[][] diff = new char[2][2];
        for (int i=0;i<A.length();i++) {
            if (A.charAt(i)!=B.charAt(i)) {
                count++;
                if (count>2) return false;
                diff[count-1][0]=A.charAt(i);
                diff[count-1][1]=B.charAt(i);
            }
            map[A.charAt(i)-'a']++;
        }
        if (count ==2) {
            if (diff[0][0]==diff[1][1] && diff[0][1]==diff[1][0])
                return true;
        }
        if (count ==0) {
            for (int n:map) {
                if (n>=2) return true;
            }
        }
        return false;
    }
}
