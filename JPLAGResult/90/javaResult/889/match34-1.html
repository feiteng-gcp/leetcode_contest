class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.length() != B.length() || A.length() < 2) {
            return false;
        }
        int i = 0;
        int j = 0;
        int count = 0;
        int[] counts = new int[26];
        int distinctNum = 0;
        int first = 0;
        int second = 1;
        while (i < A.length()) {
            if (A.charAt(i) != B.charAt(j)) {
                count++;
                if (count == 1) {
                    first = i;
                }
                else 
if (count == 2) {
                    second = i;
                }
                else {
                    return false;
                }
            }
            if (counts[A.charAt(i)-'a'] == 0) {
                distinctNum++;
                counts[A.charAt(i)-'a']++;
            }
            i++;
            j++;
        }
        if (count == 1) {
            return false;
        }
        else if (count == 0) {
            if (distinctNum == A.length()) {
                return false;
            }
            else {
                return true;
            }
        }
        if (A.charAt(first) == B.charAt(second) && A.charAt(second) == B.charAt(first)) {
            return true;
        }
        return false;
    }
}
