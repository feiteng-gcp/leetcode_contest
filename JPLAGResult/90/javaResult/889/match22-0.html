class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.length() != B.length()) {
            return false;
        }
        int len = A.length();
        boolean ans = false;
        char swap1 = '0';
        char swap2 = '0';
        Set<Character> hash = new HashSet<>();
        boolean hasDup = false;
        for (int i = 0; i < len; ++i) {
            if (!hash.add(A.charAt(i))) {
                hasDup = true;
            }
            if (A.charAt(i) != B.charAt(i)) {
                if (!ans) {
                    ans = true;
                    swap1 = A.charAt(i);
                    swap2 = B.charAt(i);
                } else {
                    if (A.charAt(i) != swap2 || B.charAt(i) != swap1) {
                        return false;
                    }
                }
            }
        }
        return ans || hasDup;
    }
}
