class Solution {
    public boolean buddyStrings(String A, String B) {
       if (A.length() != B.length())
            return false;
        if (A.length() == 0)
            return false;
        int diffC = 0;
        int[] indx = new int[3];
        boolean[] chars = new boolean[26];
        boolean duplicate = false;
        
for (int i = 0; i < A.length(); i++) {
            if (A.charAt(i) != B.charAt(i)) {
                indx[diffC] = i;
                diffC++;
                if (diffC > 2)
                    return false;
            }

            if (!duplicate) {
                if (chars[A.charAt(i) - 'a'])
                    duplicate = true;
                else
                    chars[A.charAt(i) - 'a'] = true;
            }

        }
        return (diffC == 2 &&
                (A.charAt(indx[0]) == B.charAt(indx[1]) && A.charAt(indx[1]) == B.charAt(indx[0])) ||
                        (A.length() > 1 && duplicate && diffC == 0));
    }
}
