class Solution {
    public boolean buddyStrings(String A, String B) {
        char[] arrayA = A.toCharArray();
        char[] arrayB = B.toCharArray();
        int len = A.length();
        if (A.length() != B.length()) {
            return false;
        }
        
        int diff1 = Integer.MAX_VALUE, diff2 = Integer.MAX_VALUE;
        int count = 0;
        for (int i = 0; i < len; i++) {
            int diff = arrayA[i] - arrayB[i];
            if (diff != 0) {
                count++;
                if (count == 1) {
                    diff1 = diff;
                } else if (count == 2) {
                    diff2 = diff;
                }
            }
        }
        
        if (count == 0) {
            int[] hash = new int[26];
            for (int i = 0; i < len; i++) {
                hash[arrayA[i] - 'a']++;
                if (hash[arrayA[i] - 'a'] >= 2) {
                    return true;
                }
            }
            return false;
        } else if (count == 2) {
            return diff1 + diff2 == 0;
        } else {
            return false;
        }
    }
    
}
