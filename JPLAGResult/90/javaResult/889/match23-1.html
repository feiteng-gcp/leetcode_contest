class Solution {
    public boolean buddyStrings(String A, String B) {
        if(A.length()!=B.length()) return false;
        int[] map = new int[26];
        boolean charRepeat = false;
        
int first = -1,second = -1;
        for(int i=0;i<A.length();i++){
            char c1 = A.charAt(i);
            char c2 = B.charAt(i);
            map[c1-'a']++;
            if(map[c1-'a']>1) charRepeat = true;
            if(c1==c2) continue;
            if(first==-1) first = i;
            else if(second!=-1) return false;
            else{
            second = i;
            if(c1!=B.charAt(first)||c2!=A.charAt(first)) return false;
            }
        }
        
        return second!=-1||charRepeat;
    }
}
