class Solution {
    public boolean buddyStrings(String A, String B) {
        int len = A.length();
        if(len < 2 || len != B.length()) return false;
        int count = 0;
        int[] statistic = new int[26];
        char[] temp = new char[4];
        for(int i = 0; i < len; i++){
            if(A.charAt(i) != B.charAt(i)){
                if(count >= 2) return false;
                temp[count * 2] = A.charAt(i);
                temp[count * 2 + 1] = B.charAt(i);
                count++;
            }else{
                statistic[A.charAt(i) - 'a'] ++;
            }
        }
        if(count == 0) {
            for(int i=0; i < 26; i++){
                if(statistic[i] > 1) return true;
            }
            return false;
        }
        if(count != 2) return false;
        return temp[0] == temp[3] && temp[1] == temp[2];
    }
}
