class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.length() != B.length())
            return false;
        int[] c = new int[26];
        int count = 0;
        char a = 'a', b = 'b';
        boolean swap = false;
        for (int i = 0; i < A.length(); i++) {
            if (A.charAt(i) != B.charAt(i)) {
                count++;
                if (count == 1) {
                    a = A.charAt(i);
                    b = B.charAt(i);
                } else if (count == 2) {
                    if (B.charAt(i) != a || A.charAt(i) != b)
                        return false;
                } else {
                    return false;
                }
            } else {
                if (++c[A.charAt(i) - 'a'] >= 2){
                    swap = true;
                }
                
            }
        }
        return (count == 0 && swap) || count == 2;
    }
}
