class Solution {
    public boolean buddyStrings(String A, String B) {
        if ( A==null || B==null || A.length()!= B.length()){
            return false;
        }
        int count = 0; 
        Set<Character> hashset_A = new HashSet<>();
        Set<Character> hashset_B = new HashSet<>();
        Set<Character> unique = new HashSet<>();
        boolean duplicate = false;
        
for (int i = 0; i < A.length();i++) {
            if (A.charAt(i) == B.charAt(i)){
                if(!unique.add(A.charAt(i))){
                    duplicate = true;
                }
                
                continue;
                
            }
            count++;
            if (count > 2){
                return false;
            }
            hashset_A.add(A.charAt(i));
            hashset_B.add(B.charAt(i));
            
        }
        for (char c : hashset_A) 
{
            if(hashset_B.add(c)){
                return false;
            }
        }
        if (hashset_A.size() == 0 && !duplicate){
            return false;
        }
        return true;
    }
}
