class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.length() != B.length()) return false;
		
		int[] alphaA = new int[26];
		int[] alphaB = new int[26];
		int count = 0;
		boolean dup = false;
		
		for (int i = 0; i < A.length(); i++) {
			char a = A.charAt(i);
			char b = B.charAt(i);
            if (a != b) count++;
			if (count > 2) return false;
			alphaA[a - 'a']++;
			alphaB[b - 'a']++;
		}
		for (int i = 0; i < 26; i++) {
			if (alphaA[i] != alphaB[i]) return false;
			if (alphaA[i] > 1) dup = true;
		}
		if (count == 0) {
			if (dup) return true;
			else return false;
		}
		return true;
    }
}
