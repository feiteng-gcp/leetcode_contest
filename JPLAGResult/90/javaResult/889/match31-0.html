class Solution {
    public boolean buddyStrings(String A, String B) {
        if(A == null || B == null || A.length() < 2 || B.length() < 2)
            return false;
        
        int n = A.length();
        int m = B.length();
        if(n != m) {
            return false;
        }
        int[] map = new int[26];
        int cnt = 0;
        int index1 = -1;
        int index2 = -1;
        boolean mark = false;
        
        for(int i = 0; i < n; i++) {
            
            map[A.charAt(i) - 'a']++;
            
            if(map[A.charAt(i) - 'a'] >= 2) {
                mark = true;
            }      
            
            if(A.charAt(i) == B.charAt(i)) {
                continue;
            }
            

            cnt++;
            if(cnt == 1) {
                index1 = i;
            } else if(cnt == 2) {
                index2 = i;
            } else if(cnt > 2){
                return false;
            }
        }
        
         if(cnt == 0) {
             return mark;
         } else if(cnt == 1) {
             return false;
         }
        
        
        if((A.charAt(index1) == B.charAt(index2)) && 
            A.charAt(index2) == B.charAt(index1)) {
            return true;
        }
        return false;
        
    }
}
