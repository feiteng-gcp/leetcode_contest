class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.length() != B.length()) {
            return false;
        }
        
        if (A.equals(B)) 
{
            int[] cnt = new int[26];
            
            for (char c : A.toCharArray()) {
                if (++cnt[c - 'a'] > 1) {
                    return true;
                }
            }
            
            
return false;
        }
        
        int cnt = 0, l = -1;
        
        for (int i = 0; i < A.length(); i++) {
            if (A.charAt(i) != B.charAt(i)) {
                if (++cnt > 2) {
                    return false;
                }
                
                if (l == -1) {
                    l = i;
                } else if (A.charAt(l) != B.charAt(i) || A.charAt(i) != B.charAt(l)) 
{
                    return false;
                }
            }
        }
        
        return (cnt == 2);
    }
}
