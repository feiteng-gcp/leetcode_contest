class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.equals(B)) {
            boolean[] letter = new boolean[26];
            Arrays.fill(letter, false);
            for (int i = 0; i < A.length(); i++) {
                if (letter[A.charAt(i)-'a'])
                    return true;
                else
                    letter[A.charAt(i)-'a'] = true;
            }
            return false;
        
}
        if (A.length() != B.length())
            return false;
        int diff = 0;
        char a1 = '0';
        char b1 = '0';
        char a2 = '0';
        char b2 = '0';
        
for (int i = 0; i < A.length(); i++) {
            if (A.charAt(i) != B.charAt(i)) {
                diff++;
                if (diff == 1) {
                    a1 = A.charAt(i);
                    b1 = B.charAt(i);
                
} else if (diff == 2) {
                    a2 = A.charAt(i);
                    b2 = B.charAt(i);
                } else if (diff >2) {
                    return false;
                }
            }
        }
        if ((diff == 2) && (a1 == b2) && (a2 == b1))
            return true;
        else
            return false;
    }
}
