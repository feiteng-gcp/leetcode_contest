class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.length() != B.length()) return false;
        if (A.length() <=1 || B.length() <=1) return false;
        if (A.equals(B) && !unique(A)) return true;
        List<Character> listA = new LinkedList<>();
        List<Character> listB = new LinkedList<>();
        for (int i = 0; i < A.length(); i++) {
            if (A.charAt(i) != B.charAt(i)) {
                listA.add(A.charAt(i));
                listB.add(B.charAt(i));
            }
        }
        if (listA.size() == 2 && listA.get(0) == listB.get(1) && listA.get(1) == listB.get(0)) {
            return true;
        }
        return false;
        
    }
    
    private boolean unique(String s) {
        Set<Character> set = new HashSet<>();
        for (char c: s.toCharArray()) {
            if (!set.add(c))
                return false;
        }
        return true;
    }
}
