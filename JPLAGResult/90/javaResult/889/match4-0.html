class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.length() != B.length()) {
            return false;
        }
        Integer index1 = null;
        Integer index2 = null;
        for (int i = 0; i < A.length(); i++) {
            if (A.charAt(i) != B.charAt(i)) {
                if (index1 == null) {
                    index1 = i;
                } else if (index2 == null) {
                    index2 = i;
                } else {
                    return false;
                }
            }
        }
        if (index1 != null && index2 != null) {
            return A.charAt(index2) == B.charAt(index1) && A.charAt(index1) == B.charAt(index2);
        }
        if (index1 == null && index2 == null) {
            int[] count = new int[26];
            for (char aa : A.toCharArray()) {
                count[aa - 'a']++;
                if (count[aa - 'a'] >= 2) {
                    return true;
                }
            }
        }
        return false;
    }
}
