class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.length() != B.length()) {
            return false;
        }
        int n = A.length();
        if (n == 0) {
            return false;
        }
        
        
        int[] a = count(A);
        int[] b = count(B);
        int i = 0;
        int j = n - 1;
        
        List<Integer> l = new ArrayList<>();
        
        while (i < n) {
            if (A.charAt(i) != B.charAt(i)) {
                l.add(i);
            }
            i++;
        }
        
        return l.size() == 0 && hasDuplicate(a) > 0 || l.size() == 2 &&  A.charAt(l.get(0)) == B.charAt(l.get(1)) && A.charAt(l.get(1)) == B.charAt(l.get(0));
    }
    
    int[] count(String s) {
        int[] a = new int[256];
        for (char c : s.toCharArray()) {
            a[c]++;
        }
        
        return a;
    }
    
    int hasDuplicate(int[] a){
        int count = 0;
        for (int x : a) {
            if (x > 1) {
                count++;
            }
        }
        return count;
    }
}
