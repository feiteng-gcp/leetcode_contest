class Solution {
    public boolean buddyStrings(String A, String B) {
        if(A.length() != B.length()) return false;
        int diff = 0;
        boolean hasSame = false;
        int[] ca = new int[26];
        
char tmpA = ' ', tmpB = ' ';
        for(int i = 0 ; i < A.length() ; i++){
            if(A.charAt(i) != B.charAt(i)){
                if(diff == 0){
                    tmpA = A.charAt(i);
                    tmpB = B.charAt(i);
                    diff++;
                }else if(diff == 1){
                    if(B.charAt(i) == tmpA && A.charAt(i) == tmpB) diff++;
                    else return false;
                }else return false;
            }else{
                ca[A.charAt(i) - 'a']++;
                if(ca[A.charAt(i) - 'a'] >= 2) hasSame = true
;
            }
        }
        return diff == 2 || (diff == 0 && hasSame);
    }
}
