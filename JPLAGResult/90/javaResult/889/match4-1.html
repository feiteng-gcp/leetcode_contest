class Solution {
    public boolean buddyStrings(String A, String B) {
        if (A.length() != B.length()) return false;
        if (A.length() < 2) return false
;
        int i1= -1;
        int i2= -1;
        for (int i = 0; i < A.length(); i++) {
            if (A.charAt(i) != B.charAt(i)) {
                if (i1 < 0) {
                    i1 = i;
                } else if (i2 < 0) {
                    i2 = i;
                } else {
                    return false;
                }
            }
        }
        if (i1 >= 0 && i2 >= 0) {
            return A.charAt(i1) == B.charAt(i2) && A.charAt(i2) == B.charAt(i1);
        } else if (i1 < 0 && i2 < 0) {
            // A has dup characters
            Map<Character, Integer> cm = new HashMap<Character, Integer>();
            for (int i = 0; i < A.length(); i++) {
                char c = A.charAt(i);
                cm.put(c, cm.getOrDefault(c,0) + 1);
                if (cm.get(c) >= 2) return true;
            }
            return false;
        }
        return false;
    }
}
