class Solution {
    public boolean buddyStrings(String a, String b) {
        if (a.length() != b.length()) {
            return false;
        } else if (a.length() <= 1) {
            return false;
        }
        
        int aI = -1, bI = -1;
        int[] alpha = new int[26];

        for (int i = 0; i < a.length(); i++) {
            if (a.charAt(i) != b.charAt(i)) {
                if (aI != -1 && bI != -1) {
                    return false;
                } else if (aI != -1) {
                    if (b.charAt(i) != a.charAt(aI) || b.charAt(aI) != a.charAt(i)) {
                        return false;
                    } else {
                        bI = i;
                    }
                } else {
                    aI = i;
                }
            }
            
            alpha[a.charAt(i) - 'a'] += 1;
        }
        
        if (aI == -1 && bI == -1) {
            for (int i = 0; i < 26; i++) {
                if (alpha[i] > 1) {
                    return true;
                }
            }
            
            return false;
        } else if (bI == -1) {
            return false;
        } else {
            return true;
        }
        
    }
}
