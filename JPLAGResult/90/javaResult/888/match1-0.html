class Solution {
    public int mirrorReflection(int p, int q) {
        int a = q;
        int time = 1;
        
        while (a % p != 0) {
            a += q;
            time++;
        }
        
        if (time % 2 == 0) return 2;
        if ((a / p) % 2 == 1) return 1;
        return 0;
    }
}
