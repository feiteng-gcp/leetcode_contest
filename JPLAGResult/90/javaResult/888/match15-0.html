class Solution {
    public int mirrorReflection(int p, int q) {
        int lcm = this.lcm(p, q);
        int num1 = lcm / q;
        int num2 = lcm / p;
        if (num1 % 2 == 0) {
            return 2;
        } else {
            if (num2 % 2 == 0) {
                return 0;
            } else {
                return 1;
            }
        }
    }
    private int lcm(int a, int b){  
        int m = a, n = b; 
        int t = m % n;  
        while (t != 0){  
            m = n;  
            n = t;  
            t = m % n;  
        }
        int l = a * b / n;  
        return l;  
    }  
}
