class Solution {
    public int mirrorReflection(int p, int q) {
        if (p % q == 0 && p / q % 2 == 0) {
            return 2;
        }
        if (p % q == 0 && p / q % 2 == 1) {
            return 1;
        }
        int times = p * q / gcd(p, q);
        System.out.println(times);
        int move = times  / p;
        if (move % 2 == 0) {
            return 0;
        }
        if (times / q % 2 == 0) {
            return 2;
        } else {
            return 1;
        }
        
    }
    public int gcd(int p, int q) {
        if (p < q) {
            return gcd(q, p);
        }
        if (q == 0) {
            return p;
        }
        return gcd(q, p%q);
    }
}
