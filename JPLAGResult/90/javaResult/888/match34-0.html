    public int mirrorReflection(int p, int q) {
        if (p <= 0) {
            return -1;
        }
        boolean fromBottom = true;
        boolean fromLeft = true;
        int h = p;
        while (h % q != 0) {
            if ((h / q) % 2 != 0) {
                fromLeft = !fromLeft;
            }
            h = p + h % q;
            fromBottom = !fromBottom;
        }
        return checkPoint(h, q, fromBottom, fromLeft);

    private int checkPoint(int h, int q, boolean fromBottom, boolean fromLeft) {
        int turn = h / q;
        if (turn % 2 == 0) {
            if (fromBottom && fromLeft) {
                return 2;
            } else if (fromBottom) {
                return 1;
            } else {
                return 0;
            }
        } else {
            if (fromBottom && fromLeft) {
                return 1;
            } else if (fromBottom) {
                return 2;
            } else {
                return 0;
            }
        }
