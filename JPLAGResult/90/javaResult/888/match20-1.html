class Solution {
    public int mirrorReflection(int p, int q) {
        long hor = p * q;
        long ver = p * p;
        long lcm = lcm(hor, ver);

        long x = lcm / hor, y = lcm / ver;

        if (x % 2 == 1 && y % 2 == 1) {
            return 1;
        }
        
if (x % 2 == 0 && y % 2 == 1) {
            return 2;
        }
        if (x % 2 == 1 && y % 2 == 0) {
            return 0;
        }

        return -1;
    }

    private static long gcd(long a, long b) {
        
while (b > 0) {
            long temp = b;
            b = a % b; // % is remainder
            a = temp;
        }
        return a;
    }

    private static long lcm(long a, long b) {
        return a * (b / gcd(a, b));
    }
}
