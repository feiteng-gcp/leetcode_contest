class Solution {
    public int mirrorReflection(int p, int q) {
        int com = p * q / gcd(p, q);
        int kp = com / p;
        int kq = com / q;
        if(kp % 2 == 1 && kq % 2 == 1) return 1;
        else 
if(kp % 2 == 0 && kq % 2 == 1) return 0;
        else if(kp % 2 == 1 && kq % 2 == 0) return 2;
        else return -1;
    }
    
    private int gcd(int a, int b){
        
if(a % b == 0) return b;
        else return gcd(b, a % b);
    }
}
