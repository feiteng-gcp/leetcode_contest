class Solution {
    public int mirrorReflection(int p, int q) {
        int g = gcd(p, q);
        System.out.println(g);
        boolean up = q / g % 2 == 1;
        boolean right = p / g % 2 == 1;
        if(up){
            if(right){
                return 1;
            } else {
                return 2;
            }
        } else {
            return 0;
        }
    }
    
    private int gcd(int a, int b){
        return b == 0 ? a : gcd(b, a % b);
    }
}
