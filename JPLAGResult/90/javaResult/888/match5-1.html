class Solution {
    public int mirrorReflection(int p, int q) {
        int curr = q;
        int count = 1;
        while(curr%p != 0) {
            count++;
            curr += q;
        }
        if ((count % 2) == 0)
            return 2;
        else 
{
            if ((curr/p)%2 == 0)
                return 0;
            else
                return 1;
        }
    }
}
