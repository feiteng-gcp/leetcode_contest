class Solution {
    public int mirrorReflection(int p, int q) {
    	int left = 0;
    	int right = q;
    	while (true) {
    		if (right % p == 0) {
    			if ((right / p) % 2 == 0) return 0;
    			else return 1;
    		}

    		right += 2 * q;
    		
    		left += 2 * q;
    		if (left > 0 && left % p == 0) return 2;
    	}
    }
}
