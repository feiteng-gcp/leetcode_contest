class Solution {
    public int mirrorReflection(int p, int q) {
        if (q == 0)
            return 0;
        if (q == p)
            return 1;

        double x=0, y =0;
        double dX = 1, dY = (double)q/p;

        while(true){
            if (dX>0 && dY > 0){
                double sY = Math.abs((p-y)/dY);
                double sX = Math.abs((p-x)/dX);
                x += (sY>sX ? sX:sY)*dX;
                y += (sY>sX ? sX:sY)*dY;
                if (sY<sX){
                    dY =-dY;
                }else
                    dX = -dX;

            }else
            if (dX<0 && dY > 0){
                double sY = Math.abs((p-y)/dY);
                double sX = Math.abs((x)/dX);
                x += (sY>sX ? sX:sY)*dX;
                y += (sY>sX ? sX:sY)*dY;
                if (sY<sX){
                    dY =-dY;
                }else
                    dX = -dX;
            }
            else
            if (dX<0 && dY < 0){
                double sY = Math.abs((y)/dY);
                double sX = Math.abs((x)/dX);
                x += (sY>sX ? sX:sY)*dX;
                y += (sY>sX ? sX:sY)*dY;
                if (sY<sX){
                    dY =-dY;
                }else
                    dX = -dX;
            }
            else
            if (dX>0 && dY < 0){
                double sY = Math.abs((y)/dY);
                double sX = Math.abs((p-x)/dX);
                x += (sY>sX ? sX:sY)*dX;
                y += (sY>sX ? sX:sY)*dY;
                if (sY<sX){
                    dY =-dY;
                }else
                    dX = -dX;
            }

            if (equals(x,p) && equals(y,0))
                return 0;
            else
            if (equals(x,p) && equals(y,p))
                return 1;
            else
            if (equals(x,0) && equals(y,p))
                return 2;
        }
    }
    static double EPSILON = 0.0001d;
    public static boolean equals (final double a, final double b) {
        if (a==b) return true;
        return Math.abs(a - b) < EPSILON; //EPSILON = 0.0000001d
    }
}
