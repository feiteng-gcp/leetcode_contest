class Solution {
    public int mirrorReflection(int p, int q) {
        int count = 0;
        int i = 1;
        int p2 = p * 2;
        System.out.println((q * i) % p == 0);
        while (!((q * i) % p == 0 && !((q * i) % p2 == 0 && i % 2 == 0))) {
            i++;
        }
        
if (i % 2 == 1) {
            if ((q * i) % p2 == 0) {
                return 0;
            } else {
                return 1;
            }
        } else {
            return 2;
        }
    }
}
