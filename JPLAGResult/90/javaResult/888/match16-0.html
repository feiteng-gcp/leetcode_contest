class Solution {
  public int mirrorReflection(int p, int q) {
        int x = cal(p, q);
        int count = x / p;
        int count2 = x / q;
        if (count2 % 2 == 0) {
            return 2;
        }
        if (count % 2 == 1) {
            return 1;
        }
        return 0;
    }

    private int cal(int a, int b) {
        int c = a * b;
        if (a < b) {
            int r = 0;
            r = a;
            a = b;
            b = r;
        }
        while (true) {
            int r = a % b;
            if (r == 0) {
                return c / b;
            } else {
                a = b;
                b = r;
            }
        }
    }
}
