class Solution {
    public int mirrorReflection(int p, int q) {
        int cur = 0;
        int height = 0;
        boolean left = true;
        while (true) {
        	if (height == 0 && !left) {
        		return 0;
        	} else if (height == p && !left) {
        		return 1;
        	} else if (height == p && left) {
        		return 2;
        	}
        	cur = (cur + q) % (p * 2);
        	height = Math.abs(p - Math.abs(p - cur));
        	left = !left;
        }
    }
}
