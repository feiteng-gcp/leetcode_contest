class Solution {
    public int mirrorReflection(int p, int q) {
        if(q == 0) return 0;
        int g = gcd(p, q);
        int p2 = p / g, q2 = q / g;
        if(p2 % 2 == 0) return 2;
        if(q2 % 2 != 0) return 1;
        return 0;
    }
    
    int gcd(int a, int b)
    {
        if(a == 0) return b;
        if(b == 0) return a;
        return gcd(b, a % b);
    }
}
