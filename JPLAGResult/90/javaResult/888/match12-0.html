class Solution {
    public int mirrorReflection(int p, int q) {
        if (q == 0)
            return 0;
        if (q == p)
            return 1;
        int gy = gcd(p, q);
        int gb = 0;
        if (gy == 0)
            gb = p * q;
        else
            gb = p * q / gy;
        int sidenum = gb / p;
        int hitnum = gb / q;
        
        if (sidenum % 2 == 0)
            return 0;
        if (hitnum % 2 == 0)
            return 2;
        else
            return 1;
    }
    
    public int gcd(int p,int q){
         if(q == 0)    return p;
         return gcd(q, p % q);
    }
}
