class Solution {
    public int mirrorReflection(int p, int q) {
        int lcm = lcm(p, q);
        int x = lcm / p;
        int y = lcm / q;
        if (y % 2 == 0) {
            return 2;
        }
        if (x % 2 == 1) {
            return 1;
        }
        return 0;
    
}
    private int lcm(int m, int n) {
        int a = m;
        int b = n;
        int c = 0;
        while(b != 0) { 
            c = a % b; 
            a = b;  
            b = c;
        }
        return m*n/a;
    }
    
}
