class Solution {
    public int mirrorReflection(int p, int q) {
        // 求p q最小公倍数
        int temp =  p * q / maxCommonDivisor(p, q);  
        if (temp / q % 2 == 0) {
            return 2;
        } else if (temp / p % 2 == 0) {
            return 0;
        } else {
            return 1;
        }
    }
    
    public static int maxCommonDivisor(int m, int n) {  
        if (m < n) {     // 保证被除数大于除数
            int temp = m;  
            m = n;  
            n = temp;  
        }  
        while (m % n != 0) {  // 在余数不能为0时,进行循环  
            int temp = m % n;  
            m = n;  
            n = temp;  
        }  
        return n;    // 返回最大公约数  
    }
}
