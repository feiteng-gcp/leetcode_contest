class Solution {
    public int mirrorReflection(int p, int q) {
        if(q == 0)
            return 0;
        int gcd = GCD(p,q);
        p = p/gcd;
        q = q/gcd;
        
if(q%2 == 0){
            return 0;
        }
        else if(p%2 == 0){
            return 2;
        }
        else{
            return 1;
        }
    }
    
    public int GCD(int a, int b) {
        if (b==0) return a;
        return GCD(b,a%b);
    }
}
