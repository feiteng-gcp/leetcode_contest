class Solution {
    public int mirrorReflection(int p, int q) {
        int g = gcd(p, q);
        p = p / g;
        q = q / g;
        if (p % 2 == 0) {
            if (q % 2 == 0) {
                return -1;
            }
            else {
                return 2;
            }
        }
        else {
            
if (q % 2 == 0) {
                return 0;
            }
            else {
                return 1;
            }
        }
    }

    
int gcd(int a, int b) {
        while(a != 0 && b != 0) {
            int c = b;
            b = a % b;
            a = c;
        }
        return a + b;
    }
}
