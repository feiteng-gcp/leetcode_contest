class Solution {
    public int mirrorReflection(int p, int q) {
    	boolean up = true;
    	boolean right = true;
    	int left = p;
        while((left % q) != 0) {
        	int temp = left / q;
        	up = !up;
        	left = p - (q - (left - temp * q));
        	if (temp % 2 == 1) {
        		continue;
        	}else {
        		right = !right;
        	}
        }
        
if (!up) {
        	return 0;
        }else if (right && (left / q % 2) == 0) {
        	return 2;
        }else if (!right && (left / q % 2) == 1) {
        	return 2;
    	}else {
        	return 1;
        }
    }
}
