class Solution {
    private static int gcd(int a, int b){
        while (b > 0){
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
    public int mirrorReflection(int p, int q) {
        int LCM = p*q/gcd(p, q);
        //Theory: if LCM is odd, it will hit the top, if even, the bottom
        if((LCM/p & 1) == 0){
            return 0;
        }
        else{
            p = p/gcd(p, q);
            q = q/gcd(p, q);
            if((p & 1) == 0){
                return 2; 
            }
            else{
                return 1;
            }
        }
    }
}
