class Solution {
    private int findGCD(int a, int b) {
        if (a % b == 0) {
            return b;
        }
        else {
            return findGCD(b, a % b);
        
}
    }
    public int mirrorReflection(int p, int q) {
        if (q == 0) {
            return 0;
        }
        int gcd = findGCD(p, q);
        p /= gcd;
        q /= gcd;
        if (p % 2 == 0) return 2;
        return q % 2;
    }
}
