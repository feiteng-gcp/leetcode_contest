class Solution {
    public int mirrorReflection(int p, int q) {
        int mcd = maxCommonDivisor(p, q);
        int reflect = p / mcd;
        int rotate = q / mcd;
        
        if (rotate % 2 == 1) {
            if (reflect % 2 == 1) return 1;
            else return 2;
        }
        else return 0
;
    }
    
    private int maxCommonDivisor(int m, int n) {
        if (m < n) {
            int tmp = m;
            m = n;
            n = tmp;
        }
        
        while (m % n != 0) {
            int tmp = m % n;
            m = n;
            n = tmp;
        }
        
        return n;
    }
}
