class Solution {
    public int mirrorReflection(int p, int q) {
        if (q == 0) {
            return 0;
        }
        int len = p;
        int step = q;
        while (true){
            if(len == step){
                if( (len / q) % 2 == 0){
                    return 2;
                }else if((len / p) % 2 == 1){
                    return 1;
                }else{
                    return 0;
                }
            }else if(len < step){
                len+=p;
            }else{
                step+=q;
            }
        }

    }
}
