class Solution {
    public int mirrorReflection(int p, int q) {
        boolean left=false;
        int y=0;
        while (true){
            y+=q;
            if(y%p==0){
                if (left) return 2;
                else{
                    if(y%(2*p)==0) return 0;
                    else return 1;
                }
            }
            left=!left;
        }
    }

}
