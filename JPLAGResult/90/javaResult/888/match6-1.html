class Solution {
    public int mirrorReflection(int p, int q) {
        int i = 1;
        int y = q;
        while(true){
            if (y%p == 0){
                if (i%2 == 0){
                    return 2;
                }else if ((y/p)%2 == 0){
                    return 0;
                }else
{
                    return 1;
                }
            }
            y += q;
            i++;
            
        }
    }
}
