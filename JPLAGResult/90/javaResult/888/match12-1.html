class Solution {
    public int mirrorReflection(int p, int q) {
        if (q == 0) return 0;
        if (q == p) return 1;
        int k = gcd(p, q);
        int n = q / k;
        int m = p / k;
        int output = (m % 2 == 0)? 2 : ((n % 2 == 0)? 0 : 1);
        System.out.println(String.format("p:%d q:%d mirror:%d", p, q, output));
        return output;
    
}
    
    public static int gcd(int a, int b) {
        if (b==0) return a;
        return gcd(b, a%b);
    }
}
