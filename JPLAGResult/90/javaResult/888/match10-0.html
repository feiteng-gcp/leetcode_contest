class Solution {
    public int mirrorReflection(int p, int q) {
        int gcd = findGCD(p, q);
        int m = p / gcd;
        int n = q / gcd;
        m = m % 2;
        n = n % 2;
        if (m == 1 && n == 0) {
            return 0;
        } else if (m == 0 && n == 1) {
            return 2;
        } else {
            return 1;
        }
    }
    
    private static int findGCD(int number1, int number2) { 
        if(number2 == 0) { 
            return number1; 
        }     
        return findGCD(number2, number1 % number2); 
    }
}
