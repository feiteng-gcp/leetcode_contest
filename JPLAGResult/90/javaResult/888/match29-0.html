class Solution { 
  public int mirrorReflection(int p, int q) {
    int gcp = gcp(p, q);
    p /= gcp;
    q /= gcp;

    if ((p & 1) == 1 && (q & 1) == 1) return 1;
    else if ((p & 1) == 0 && (q & 1) == 1) return 2;
    else return 0;
  }

  public int gcp(int p, int q) {
    int rest = p % q;
    if (rest == 0) return q;
    else return gcp(q, rest);
  }
}
