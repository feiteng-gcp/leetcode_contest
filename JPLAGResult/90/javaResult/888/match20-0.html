class Solution {
    private int lcm(int a, int b) {
        int mul = a * b;
        while (b != 0) {
            int r = a % b;
            a = b;
            b = r;
        }
        return mul / a;
    }

    public int mirrorReflection(int p, int q) {
        int lcm = lcm(p, q);
        int size = lcm / q;
        if (size % 2 == 0) {
            return 2;
        }
        if (lcm / p % 2 == 1) {
            return 1;
        }
        return 0;
    }
}
