class Solution {
    public int mirrorReflection(int p, int q) {
        int cd = gcd(p, q);
        int numOfP = p / cd;
        int numOfQ = q / cd;
        if (numOfP % 2 == 0) {
            return 2;
        }
        
        if (numOfQ % 2  == 0) {
            return 0;
        }
        
        if (numOfQ % 2 == 1) {
            return 1;
        }
        
        return 3;
    }
    
    private int gcd(int x, int y) {
        if (y == 0) {
            return x;
        } else {
            return gcd(y, x%y);
        }
    }
}
