class Solution {
    public double mincostToHireWorkers(int[] quality, int[] wage, int K) {
        int N = quality.length;
        int[][] pairs = new int[N][2];
        int[][] index = new int[N][2];
        for (int i = 0; i < N; i++) {
            pairs[i] = new int[]{wage[i], quality[i]};
        }
        Arrays.sort(pairs, (a, b) -> Double.compare((double)b[0] / b[1], (double)a[0] / a[1]));
        for (int i = 0; i < N; i++) {
            index[i] = new int[]{i, pairs[i][1]};
        }
        Arrays.sort(index, (a, b) -> (a[1] - b[1]));
        boolean[] visited = new boolean[N];
        double result = Double.MAX_VALUE;
        for (int i = 0; i <= N - K; i++) {
            visited[i] = true;
            double base = (double)pairs[i][0] / pairs[i][1];
            result = Math.min(result, (double)pairs[i][0] + topK(index, visited, base, K - 1));
        }
        return result;
    }
    private double topK(int[][] index, boolean[] visited, double base, int K) {
        double result = 0;
        int count = 0, i = 0;
        while (count < K) {
            if (!visited[index[i][0]]) {
                result += base * index[i][1];
                count++;
            }
            i++;
        }
        return result;
    }
}
