class Solution {
    public double mincostToHireWorkers(int[] quality, int[] wage, int K) {
        double res = Double.MAX_VALUE;
        double[][] index = new double[wage.length][2];
        for (int i = 0; i < quality.length; i++) {
            index[i][0] = (double)wage[i] / quality[i];
            index[i][1] = i;
        }
        Arrays.sort(index, (a, 
b) -> (Double.compare(a[0], b[0])));
        PriorityQueue<Integer> pq = new PriorityQueue(new Comparator() {
            @Override
            public int compare(Object o1, Object o2) {
                return (int)o2 - (int)o1;
            }
        });
        int curAll = 0;
        for (int i = 0; i < K - 1; i++) {
            curAll += quality[(int)index[i][1]];
            pq.add(quality[(int)index[i][1]]);
        }
        for (int i = K - 1; i < wage.length; i++) {

            curAll += quality[(int)index[i][1]];
            pq.add(quality[(int)index[i][1]]);
            double curRatio = index[i][0];
            if (curRatio * curAll < res) {
                res = curRatio * curAll;
            }
            curAll -= pq.poll();
        }
        return res;
    }
}
