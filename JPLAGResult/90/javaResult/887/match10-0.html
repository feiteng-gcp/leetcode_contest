import java.util.ArrayList;
import java.util.Collections;
import java.util.PriorityQueue;
class Solution {
    public double mincostToHireWorkers(int[] quality, int[] wage, int K) {
        ArrayList<Worker> list = new ArrayList<>();
        for (int i = 0; i < wage.length; i++) {
            list.add(new Worker(wage[i], quality[i]));
        }
        Collections.sort(list, new Comparator<Worker>() {
            @Override
            public int compare(Worker w1, Worker w2) {
                if (w1.ratio < w2.ratio)
                    return -1;
                if (w1.ratio > w2.ratio)
                    return 1;
                return 0;
            }
        });
        double minRatio = list.get(K - 1).ratio;
        PriorityQueue<Integer> queue = new PriorityQueue<>(K, new Comparator<Integer>() {
            @Override
            public int compare(Integer i1, Integer i2) {
                return i2 - i1;
            }
        });
        int qSum = 0;
        for (int i = 0; i < K; i++) {
            int q = list.get(i).quality;
            queue.add(q);
            qSum += q;
        }
        double minSum = minRatio * qSum;
        for (int i = K; i < wage.length; i++) {
            int q = list.get(i).quality;
            int maxQ = queue.peek();
            if (q > maxQ)
                continue;
            queue.poll();
            queue.add(q);
            minRatio = list.get(i).ratio;
            qSum = 0;
            Iterator<Integer> it = queue.iterator();
            for (int j = 0; j < K; j++) {
                qSum += it.next();
            }
            minSum = Math.min(minSum, qSum * minRatio);
        }
        return minSum;
    }
    
    class Worker {
        int quality;
        double ratio;
        public Worker(int w, int q) {quality = q; ratio = 1.0 * w / q;}
    }
}
