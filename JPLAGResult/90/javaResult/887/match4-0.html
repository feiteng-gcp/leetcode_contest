class Solution {
    public double mincostToHireWorkers(int[] quality, int[] wage, int K) {
        int n = quality.length;
        Queue<Integer> pq = new PriorityQueue<>(K, (a, b) -> Integer.compare(b, a));
        int[][] arr = new int[n][2];
        for (int i = 0; i < n; i++) {
            arr[i][0] = wage[i];
            arr[i][1] = quality[i];
        }
        Arrays.sort(arr, (a, b) -> Integer.compare(a[0] * b[1] , b[0] * a[1]));
        // for (int[] brr : arr) System.out.println(Arrays.asList(brr[0], brr[1]));
        int sum = 0;
        for (int i = 0; i < K; i++)
        {
            pq.offer(arr[i][1]);
            sum += arr[i][1];
        }
        double ans = sum * (arr[K - 1][0] * 1.0 / arr[K - 1][1]);
        for (int i = K; i < arr.length; i++) {
            sum -= pq.poll();
            pq.offer(arr[i][1]);
            sum += arr[i][1];
            ans = Math.min(ans, sum * (arr[i][0] * 1.0 / arr[i][1]));
        }
        return ans;
    }
}
