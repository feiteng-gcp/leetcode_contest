class Solution {
    class Person implements Comparable<Person> {
        double quality;
        double minWage;
        double cost;
        Person(int quality, int wage) {
            this.quality = quality;
            this.minWage = wage;
            this.cost = minWage / quality;
        }
        @Override 
        public int compareTo(Person o) {
            return Double.compare(cost, o.cost);
        }
        @Override 
        public String toString() {
            return quality + " " + minWage + " " + cost;
        }
    }
    public double mincostToHireWorkers(int[] quality, int[] wage, int K) {
        Person[] people = new Person[quality.length];
        Double result = null;
        for (int i = 0; i < people.length; i++) {
            people[i] = new Person(quality[i], wage[i]);
        }
        //System.out.println(Arrays.toString(people));
        Arrays.sort(people);
        //System.out.println(Arrays.toString(people));
        
        double sum = 0.0;
        PriorityQueue<Double> q = new PriorityQueue<>();
        for (int i = 0; i < people.length; i++) {
            q.add(-people[i].quality);
            sum += people[i].quality;
            while (q.size() > K) {
                sum -= -q.poll();
            }
            //System.out.println(i + " " + sum + " " + people[i].cost);
            if (q.size() == K) {
                if (result == null) {
                    result =  1.0 * people[i].cost * sum;
                } else {
                    result = 
Math.min(result, 1.0 * people[i].cost * sum);
                }
                //System.out.println(result);
            }
        }
        return result;
    }
}
