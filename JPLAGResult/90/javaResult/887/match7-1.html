class Solution {
    public double mincostToHireWorkers(int[] quality, int[] wage, int K) {
        int n = wage.length;
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            list.add(i);
        }
        
        Collections.sort(list, (i, j) -> {
            if (wage[i] * 1.0 / quality[i] < wage[j] * 1.0 / quality[j]) {
                return -1;
            } else if (wage[i] * 1.0 / quality[i] > wage[j] * 1.0 / quality[j]) {
                return 1;
            } else {
                return 0;
            }
        });
        PriorityQueue<Integer> heap = new PriorityQueue<>(Comparator.reverseOrder());
        double res = -1;
        for (int i = 0, sum = 0; i < n; i++) {
            int index = list.get(i);
            heap.add(quality[index]);
            sum += quality[index];
            if (heap.size() > K) {
                sum -= heap.poll();
            }
            
            if (heap.size() == K) {
                double ratio = wage[index] * 1.0 / quality[index];
                
                if (res == -1 || ratio * sum < res) {
                    res = ratio * sum;
                
}
            }
            
            
        }
        
        return res;
    }
}
