import java.util.*;
import java.math.*;
import java.util.stream.*;

public class Solution {


    public double mincostToHireWorkers(int[] quality, int[] wage, int K) {
        if (K == 1) {
            double ans = 1E300;
            for (int i : wage) {
                ans = Math.min(ans, i);
            }
            return ans;
        }

        int n = quality.length;
        Integer[] rank = new Integer[n];
        for (int i = 0; i < n; i++) {
            rank[i] = i;
        }


        Arrays.sort(rank, Comparator.<Integer>comparingDouble(i -> 1.0 * wage[i] / quality[i]));
//        System.out.println(Arrays.toString(rank));

        PriorityQueue<Integer> queue = new PriorityQueue<>(Comparator.<Integer>comparingInt(i -> -quality[i]));
        double totalQuality = 0;
        double ans = 1E300;

        for (int i : rank) {
//            System.out.println(queue);
            if (queue.size() > K - 1) {
                totalQuality -= quality[queue.poll()];
            }

            if (queue.size() == K - 1) {
                ans = Math.min(ans, 1.0 * (quality[i] + totalQuality) * wage[i] / quality[i]);
            }

            queue.add(i);
            totalQuality += quality[i];
        }

        return ans;
    }


    public static void main(String[] args) {
        Solution sol = new Solution();
        System.out.println(sol.mincostToHireWorkers(new int[]{10, 20, 5}, new int[]{70, 50, 30}, 2));
        System.out.println(sol.mincostToHireWorkers(new int[]{3, 1, 10, 10, 1}, new int[]{4, 8, 2, 2, 7}, 3));
        System.out.println(sol.mincostToHireWorkers(new int[]{3, 4, 3}, new int[]{13, 8, 20}, 1));

    }
}
