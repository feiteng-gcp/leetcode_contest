class Solution {
    public int scoreOfParentheses(String S) {
        return helper(S, 0, S.length() - 1);
    }
    
    private int helper(String s, int st, int ed) {
        if (st >= ed) {
            return 0;
        }

        if (st + 1 == ed) {
            return 1;
        }

        if (s.charAt(st + 1) == ')') {
            return 1 + helper(s, st + 2, ed);
        }
        
        int left = 0, right = 0;

        for (int i = st; i <= ed; ++i) {
            if (s.charAt(i) == '(') {
                ++left;
            } else {
                ++right;
            }

            if (left == right) {
                return 2 * helper(s, st + 1, i - 1) + helper(s, i + 1, ed);
            }
        }

        return -1;
    }
}
