import java.util.Stack;

class Solution {
    public int scoreOfParentheses(String S) {
        
        Stack<Integer> stack = new Stack<>();

        for (int i = 0, size = S.length(); i < size; ++i) {
          char ch = S.charAt(i);
          switch (ch) {
            case '(':
              stack.push((int)ch);
              break;
            case ')':
              
if (stack.peek() == '(') {
                stack.pop();
                stack.push(1);
              } else {
                int sum = 0;
                while (stack.peek() != '(') {
                  sum += stack.pop();
                }
                stack.pop();
                stack.push(sum * 2);
              }
              break;
          }
        
}
        
        int total = 0;
        while (!stack.isEmpty()) {
            total += stack.pop();
        }

        return total;
    }
}
