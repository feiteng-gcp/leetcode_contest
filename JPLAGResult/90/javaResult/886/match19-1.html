class Solution {
    public int scoreOfParentheses(String S) {
        return dfs(S, 0, S.length() - 1);
    }
    public int dfs(String S, int start, int end) {
        if (start >= end) {
            return 0;
        }
        if (start == end - 1) {
            return 1;
        }
        int ct = 0;
        
int temp = start;
        int sum = 0;
        for (int i = start; i <= end; ++i) {
            if (S.charAt(i) == '(') {
                ct++;
            } else {
                ct--;
            }
            if (ct == 0) {
                if (temp == start && i == end) {
                    sum = 2 * dfs(S, start + 1, end - 1);
                } else {
                    sum += dfs(S, temp, i);
                    temp = i + 1;
                }
            }
        }
        return sum;
    }
}
