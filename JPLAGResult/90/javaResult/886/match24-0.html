class Solution {
    public int scoreOfParentheses(String S) {
        Stack<Integer> stack = new Stack();
        int res = 0;
        
        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            if (c == '(') {
                stack.push((int) Math.pow(2, stack.size()));
            } else {
                int temp = stack.pop();
                if (S.charAt(i - 1) == '(') {
                    res += temp;
                }
            }
        }
        
        return res;
    }
}
