class Solution {
    public int scoreOfParentheses(String S) {
        Deque<Integer> stac = new ArrayDeque();
        
        for (char aa : S.toCharArray()) {
            if (aa == '(') {
                stac.push(-1);
            }
            if (aa == ')') {
                int cur = 0;
                while (stac.peek() > 0) {
                    cur += stac.poll();
                }
                if (cur == 0) {
                    cur ++;
                } else {
                    cur *= 2;
                }
                stac.pop();
                stac.push(cur);
            }
        }
        int res = 0;
        while (!stac.isEmpty()) {
            res += stac.pop();
        }
        return res;
    }
}
