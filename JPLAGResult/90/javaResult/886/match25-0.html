class Solution {
    public int scoreOfParentheses(String S) {
        int[] sum = new int[S.length() * 2];
        int[] open = new int[S.length() * 2];
        int top1 = 0, top2 = -1;
        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            if (c == '(') {
                open[++top2] = i;
                sum[++top1] = 0;
            } else {
                if (open[top2] == i - 1) {
                    top2--;
                    sum[--top1]++;
                } else {
                    top2--;
                    sum[top1 - 1] += 2 * sum[top1];
                    top1--;
                }
            }
        }
        return sum[0];
    }
}
