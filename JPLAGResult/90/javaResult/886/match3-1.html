class Solution {
    public int scoreOfParentheses(String S) {
        Stack<Integer> stack = new Stack<Integer>();
        for(int i = 0; i < S.length(); i++) {
        	//System.out.println(stack);
        	if(S.charAt(i) == '(') {
        		stack.push(-1);
        	}
        	else {
        		int Q = -100;
        		Stack<Integer> numStack = new Stack<Integer>();
        		while(!stack.isEmpty()) {
        			Q = stack.pop();
        			if(Q == -1 && numStack.isEmpty()) {
        				stack.push(1);
        				break;
        			}
        			else if(Q != -1) {
        				numStack.push(Q);
        			}
        			else if(Q == -1 && !numStack.isEmpty()) {

        				//System.out.println("numStack: " + numStack);
        				int sum = 0;
        				while(!numStack.isEmpty()) {
        					sum += numStack.pop();
        				}
        				stack.push(sum * 2);
        				break;
        			}
        		}
        	}

        }
        //System.out.println("===");
        //System.out.println(stack);
        int ans = 0;
		while(!stack.isEmpty()) {
			ans += stack.pop();
		}
		return ans;
    }
}
