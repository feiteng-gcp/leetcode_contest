class Solution {
    public int scoreOfParentheses(String S) {
                Stack<Integer> stk = new Stack<>();
        for(int i = 0; i < S.length(); i++) {
        	char c = S.charAt(i);
        	if(c == '(') {
        		stk.add(0);
        	}
        	if(c == ')') {
        		
        		if(stk.peek() == 0) {
        			stk.pop();
        			stk.add(1);
        		}
        		else {
        			int cur = 0;
        			while(stk.peek() != 0) {
        				cur += stk.pop();
        			}
            		stk.pop();
            		cur *= 2;
            		stk.add(cur);
        		}
        	}
        }
        int res = 0;
        while(!stk.empty()) {
        	res += stk.pop();
        }
        return res;
    }
}
