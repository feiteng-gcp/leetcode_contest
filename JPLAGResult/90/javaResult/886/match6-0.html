class Solution {
    public int scoreOfParentheses(String S) {
        Stack<Character> stack = new Stack<>();
        for (char c : S.toCharArray()) {
            if (c == '(') {
                stack.push(c);
            } else {
                char cur = stack.pop();
                if (cur == '(') {
                    stack.push('1');
                } else {
                    int tmp = cur - '0';
                    if (stack.peek() == '(') {
                        stack.pop();
                        stack.push((char) ('0' + tmp * 2));
                    } else {
                        int sum = tmp;
                        while (stack.peek() != '(') {
                            sum += stack.pop() - '0';
                        }
                        stack.pop();
                        stack.push((char) ('0' + sum * 2));
                    }
                }
            }
            // System.out.println(stack);
        }
        int res = 0;
        while (!stack.isEmpty()) {
            res += stack.pop() - '0';
        }
        return res;
    }
}
