class Solution {
    public int scoreOfParentheses(String S) {
        Stack<Integer> stack = new Stack<>();
        int count = 0;
        for(char c: S.toCharArray()) {
            if(c == '(') {
                stack.add(-1);    
            } 
            else {
                int tmp;
                if(stack.peek() == -1) {
                    stack.pop();
                    tmp = 1;
                } else {
                    tmp = stack.peek() * 2;
                    stack.pop();
                    stack.pop();
                }
                
                if(stack.isEmpty()){
                    count += tmp;
                } else if(stack.peek() == -1) {
                    stack.add(tmp);
                } else {
                    tmp += stack.pop();
                    stack.add(tmp);
                
}
            }
        }
        return count;
    }
}
