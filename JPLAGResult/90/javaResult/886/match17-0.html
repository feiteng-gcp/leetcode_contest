class Solution {
    public int scoreOfParentheses(String S) {
        
        Stack<Integer> myStack = new Stack<>();
        
        
        for(int i = 0; i < S.length(); i++) {
            
            char ch = S.charAt(i);
            if(ch == '(') {
                myStack.push(-1);
            } else {
                //if ')'
                int pop = myStack.pop();
                if(pop == -1) {
                    myStack.push(1);   
                } else {
                    
                    int get = pop;
                    while(!myStack.isEmpty() && myStack.peek() != -1) {
                        get += myStack.pop();
                    }
                    
                    if(!myStack.isEmpty() && myStack.peek() == -1) {
                        myStack.pop();
                        myStack.push(get * 2);
                    } 
                }
            }
        }
        
        int res = 0;
        while(!myStack.isEmpty()) {
            int get = myStack.pop();
            res += get;
        }
        return res;
    }
}
