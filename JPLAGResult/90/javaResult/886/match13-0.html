class Solution {
    public int scoreOfParentheses(String S) {
        Stack<Integer> stack = new Stack<>();
        if(S == null || S.length() == 0)
        	return 0;
        
        for(int i = 0;i<S.length();i++){
        	if(S.charAt(i) == '('){
        		stack.push(0);
        	} else {
        		int k = 0;
        		while(!stack.isEmpty() && stack.peek() != 0){
        			k += stack.pop();
        		}
        		stack.pop();
        		if(k>0){
        			stack.add(k*2);
        		} else{
        			stack.add(1);
        		}
        	}
        }
        int res = 0;
        while(!stack.isEmpty()){
        	res+=stack.pop();
        }
        return res;
    }
}
