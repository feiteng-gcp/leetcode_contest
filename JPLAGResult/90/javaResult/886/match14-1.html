class Solution {
    public int scoreOfParentheses(String S) {
        Stack<Integer> stack = new Stack<Integer>();
        int res = 0;
        
for (int i=0; i<S.length(); i++) {
            char c = S.charAt(i);
            if (c == '(') {
                stack.push(-1);
            }
            else {
                int top = stack.pop();
                int sum = 0;
                while (top != -1) {
                    sum += top;
                    top = stack.pop();
                }
                
if (sum == 0) {
                    stack.push(1);
                }
                else {
                    stack.push(sum * 2);
                }
            }
        }
        
while (stack.size() != 0) {
            res += stack.pop();
        }
        return res;
    }
}
