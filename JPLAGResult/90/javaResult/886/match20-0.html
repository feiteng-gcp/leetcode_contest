class Solution {
    public int scoreOfParentheses(String S) {
        return score(S, 0, S.length());
    }

    private int score(String s, int start, int end) {
        if (start == end) {
            return 0;
        }

        if (end - start == 2) {
            return 1;
        }

        int result = 0;
        int i = start;
        int currentStart = start;
        int count = 0;
        while (i <= end) {
            if (i < s.length() && s.charAt(i) == '(') {
                count++;
            } else {
                count--;
            }

            if (count == 0) {
                if (i - currentStart > 2) {
                    result += 2 * score(s, currentStart + 1, i - 1);
                } else {
                    result += 1;
                }
                currentStart = i + 1;
            }

            i++;
        }

        return result;
    }
}
