class Solution {
    public int scoreOfParentheses(String S) {
        Stack<Integer> stack = new Stack<>();
        for(char c : S.toCharArray()){
            if(stack.isEmpty()||c=='(') 
                stack.push(0);
            else{
                int c1 = stack.pop();
                int c2 = 1;
                if(c1!=0) {
                    stack.pop();
                    c2=2*c1;
                }
                while(!stack.isEmpty()&&stack.peek()!=0){
                    c2 += stack.pop();
                }
                stack.push(c2);
            
}
        }
        return stack.peek();
    }
}
