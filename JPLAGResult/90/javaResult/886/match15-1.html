class Solution {
    public int scoreOfParentheses(String S) {
        
        
Stack<Integer> stack = new Stack<>();
        
        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            if (c == '(') {
                stack.push(Integer.MIN_VALUE);
            } else {
                int val = 0;
                while (!stack.isEmpty() && stack.peek() != Integer.MIN_VALUE) {
                    val += stack.pop();
                
}
                stack.pop();
                if (val == 0) {
                    stack.push(1);
                } else {
                    stack.push(2 * val);
                }
            }
        
}
        
        int result = 0;
        
        while (!stack.isEmpty()) {
            result += stack.pop();
        }
        return result;
    }
}
