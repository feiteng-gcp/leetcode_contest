class Solution:
    def scoreOfParentheses(self, S):
        """
        :type S: str
        :rtype: int
        """
        if len(S) == 2:
            return 1
        l = 0
        r = 0
        ans = 0
        for i in range(len(S)):
            if S[i] == '(':
                l += 1
            else:
                r += 1
            if l == r:
                if l == 1:
                    ans += 1
                else:
                    ans += 2* self.scoreOfParentheses(S[i-2*l + 2:i])
                l = 0
                r = 0
        return ans
