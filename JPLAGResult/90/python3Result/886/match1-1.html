class Solution:
    def scoreOfParentheses(self, S):
        """
        :type S: str
        :rtype: int
        """
        if S == '()':
            return 1
        else:
            l 
= []
            balance = 1
            start = 0
            for i in range(1, len(S)):
                if S[i] == '(':
                    balance += 1
                else:
                    balance -= 1
                if balance == 0:
                    l += [S[start:i+1]]
                    start = i + 1
            if len(l) == 1:
                return 2 * self.scoreOfParentheses(S[1:-1])
            else:
                return sum(list(map(self.scoreOfParentheses, l)))
