class Solution:
    def scoreOfParentheses(self, S):
        """
        :type S: str
        :rtype: int
        """
        left, right = 0, 0
        if len(S) == 0:
            return 0
        
for i in range(len(S)):
            if S[i] == '(':
                left += 1
            else:
                right += 1
            if left == right:
                if i == 1:
                    cur = 1
                else:
                    cur = 2 * self.scoreOfParentheses(S[1:i])
