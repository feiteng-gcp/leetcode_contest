class Solution:
    def scoreOfParentheses(self, S):
        """
        :type S: str
        :rtype: int
        """
        print(S)
        if len(S) == 2:
            return 1
        cnt = 1
        right = 0
        for i in range(1, len(S)):
            if S[i] == '(':
                cnt += 1
            else:
                cnt -= 1
            if cnt == 0:
                right = i
                break
        if right == 1:
            return 1 + self.scoreOfParentheses(S[2:])
        elif right == len(S) - 1:
            return 2* self.scoreOfParentheses(S[1:len(S)-1])
        else:
            return 2* self.scoreOfParentheses(S[1:right]) +  self.scoreOfParentheses(S[right + 1:])
