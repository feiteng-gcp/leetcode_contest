class Solution:
    def mirrorReflection(self, p, q):
        """
        :type p: int
        :type q: int
        :rtype: int
        """
        if p==q:
            return 1
        if q==0:
            return 0

        dist = q
        n=1
        while True:
            dist += q
            n+=1
            if dist%p ==0:
                break
        # here n is horizontal np

        # now get vertical m
        m = n*q/p

        # now try to get top right corner
        n_flag = n%2
        m_flag = m%2
        if (n_flag == 1) and (m_flag == 1):
            return 1
        if (n_flag == 0) and (m_flag == 1):
            return 2
        if (n_flag == 1) and (m_flag == 0):
            return 0
