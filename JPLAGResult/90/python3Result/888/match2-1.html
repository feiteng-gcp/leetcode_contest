class Solution:
    def power_of_2_divisor(self,num):
        res = 1
        while num % 2 == 0:
            res = res * 2
            num = num // 2
        return res
    
    def mirrorReflection(self, p, q):
        """
        :type p: int
        :type q: int
        :rtype: int
        """
        if q == 0:
            return 0
        pw_2_p 
= self.power_of_2_divisor(p)
        pw_2_q = self.power_of_2_divisor(q)
        if pw_2_p > pw_2_q:
            return 2
        elif pw_2_p == pw_2_q:
            return 1
        else:
            return 0
