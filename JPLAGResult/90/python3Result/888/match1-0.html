class Solution:
    def mirrorReflection(self, p, q):
        """
        :type p: int
        :type q: int
        :rtype: int
        """
        if q == 0:
            return 0
        d = 1
        for i in range(1,p+1):
            if q*i%p==0:
                d = i
                break
        x = d
        y = q*d//p
        if x%2 == 1 and y%2 == 1:
            return 1
        if x%2 == 1:
            return 0
        if y%2 == 1:
            return 2
