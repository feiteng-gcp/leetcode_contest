class Solution:
    def mirrorReflection(self, p, q):
        """
        :type p: int
        :type q: int
        :rtype: int
        """
        
        
        v = self.lcm(p ,q)// q
        h = self.lcm(p, q)// p
        if v % 2 == 1 and h % 2 == 1:
            return 1
        elif v %2 == 0 and h % 2 == 1:
            return 2
        elif v % 2 == 1 and h % 2 == 0:
            return 0
    def lcm(self, x, y):
        if x > y:
            greater = x
        else:
            greater = y
     
        while(True):
            if((greater % x == 0) and (greater % y == 0)):
                lcm = greater
                break
            greater += 1
     
        return lcm
