class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        m, n, counts, diff = len(A), len(B), [0] * 26, []
        if not m or m != n:
            return False
        for a, b in zip(A, B):
            counts[ord(a) - 97] += 1
            if a != b:
                diff.append((a, b))
                if len(diff) > 2:
                    return False
        if not diff:
            if max(counts) > 1:
                return True
            return False
        elif len(diff) == 2:
            return diff[0][0] == diff[1][1] and diff[0][1] == diff[1][0]
        return False
