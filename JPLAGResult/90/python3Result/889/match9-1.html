class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        if len(A) != len(B):
            return False
        if set(A) != set(B):
            return False
        if A == B and len(set(A)) == len(A):
            return False
        templist = [(A[i], B[i]) for i in range(len(A)) if A[i] != B[i]]
        if len(templist) == 2:
            if templist[0][0] == templist[1][1] and templist[1][0] == templist[0][1]:
                return True
        if len(templist) == 0:
            return True
        return False
