class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        if len(A) !=len(B):
            return False
        t = 0 
        temp1, temp2 = "",""
        for i in range(len(A)):
            if A[i] != B[i]:
                t = t+1
                temp1 = temp1 + A[i]
                temp2 = temp2 + B[i]
            if t > 2:
                return False
        if t ==2 :
            if temp1[::-1] == temp2:
                return True
            else:
                return False
        elif t== 0:
            for item in set(A):
                if A.count(item) >=2:
                    return True
            return False
        else:
            return False
