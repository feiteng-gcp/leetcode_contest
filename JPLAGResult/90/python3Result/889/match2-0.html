class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        len_A = len(A)
        len_B = len(B)
        if len_A != len_B or len_A < 2 or len_B < 2:
            return False
        
        diff_pos = []
        nums = []
        has_repeat_num = False
        for i in range(len_A):
            if A[i] != B[i]:
                diff_pos.append(i)
            if len(diff_pos) > 2:
                return False
            if has_repeat_num:
                continue
            if A[i] in nums:
                has_repeat_num = True
            else:
                nums.append(A[i])
            
        if len(diff_pos) == 0 and has_repeat_num:
            return True
        
        if len(diff_pos) != 2:
            return False
        
        # has two different pos
        if A[diff_pos[0]] == B[diff_pos[1]] and A[diff_pos[1]] == B[diff_pos[0]]:
            return True
        return False
