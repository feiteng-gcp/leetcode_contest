class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        if len(A) != len(B):
            return False
        if A==B:
            count={}
            for i in A:
                count[i]=count.get(i,0)+1
                if count[i]>1:
                    return True
            return False
        else:
        
            d={}
            for i in range(len(A)):
                if A[i]!=B[i]:
                    d[A[i]]=B[i]

            if len(d)!=2:
                return False
            else:
                keys = [j for j in d]
                if d[keys[0]]==keys[1] and d[keys[1]]==keys[0]:
                    return True
                else:
                    return False
