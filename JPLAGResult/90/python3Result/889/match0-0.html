class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        na,nb=len(A),len(B)
        if na!=nb or na<2 or nb<2:
            return False
        if A==B and len(set(A))<na:
            return True
        difa=[]
        difb=[]
        for i in range(na):
            if A[i]!=B[i]:
                difa.append(A[i])
                difb.append(B[i])
        if len(difa)!=2:
            return False
        if difa[0]==difb[1] and difb[0]==difa[1]:
            return True
        else:
            return False
