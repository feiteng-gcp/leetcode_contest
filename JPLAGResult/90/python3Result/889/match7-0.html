class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        if len(A) != len(B):
            return False
        diff = []
        for i in range(len(A)):
            a = A[i]
            b = B[i]
            if a != b:
                diff.append(i)
        if not diff:
            dict = {}
            for c in A:
                if c in dict:
                    return True
                dict[c] = 1
            return False
        if len(diff) != 2:
            return False
        return A[diff[0]] == B[diff[1]] and A[diff[1]] == B[diff[0]]
