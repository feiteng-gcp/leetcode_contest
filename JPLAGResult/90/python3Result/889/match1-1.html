class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        na, nb = len(A), len(B)
        if na != nb:
            return False
        if A == B:
            for chr in list('abcdefghijklmnopqrstuvwxyz'):
                if A.count(chr) > 1:
                    return True
            return False
        mis_a = []
        mis_b = []
        count = 0
        for i in range
(na):
            if A[i] != B[i]:
                mis_a.append(A[i])
                mis_b.append(B[i])
                count += 1
            if count > 2:
                return False
        if count == 1:
            return False
        if (mis_a[0] == mis_b[1]) and (mis_a[1] == mis_b[0]):
            return True
        else:
            return False
