class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        diffs = []
        if len(A) != len(B):
            return False
        for ch1, ch2 in zip(A, B):
            if ch1 != ch2:
                diffs.append([ch1, ch2])
                
        if len(diffs) == 1 or len(diffs) > 2:
            return False
    
        if len
(diffs) == 2:
            return diffs[0][0] == diffs[1][1] and diffs[0][1] == diffs[1][0]
        
        return len(set(A)) != len(A)
