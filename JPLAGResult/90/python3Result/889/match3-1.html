class Solution:
    def buddyStrings(self,A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        if len(A) != len(B):
            return False
        elif A == B:
            if len(A) == len(set(A)):
                return False
            else:
                return True
        else:
            # same length, but not same string
            # A and B need to have 2 and only 2 different letters/locations
            count = 0
            a 
= ''
            b = ''
            for i in range(len(A)):
                if A[i] != B[i]:
                    count += 1
                    a += A[i]
                    b += B[i]
            if (count ==2) and set(a)==set(b):
                return True
            else:
                return False
