class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        if not A or not B:
            return False
        if len(A) != len(B):
            return False
        e_index = []
        for i in range(len(A)):
            if A[i] != B[i]:
                e_index.append(i)
            if len(e_index) > 2:
                return False
        if len(e_index) == 2:
            if A[e_index[0]] == B[e_index[1]] and A[e_index[1]] == B[e_index[0]]:
                return True
            else:
                return False
        elif not e_index:
            A = list(A)
            A.sort()
            for i in range(len(A)-1):
                if A[i] == A[i+1]:
                    return True
            return False
        else:
            return False
