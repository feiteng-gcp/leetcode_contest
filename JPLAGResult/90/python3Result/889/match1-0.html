class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        aq = []
        bq = []
        count = 0
        if len(A) != len(B):
            return False
        else:
            for i in range(0, len(A)):
                if A[i] != B[i]:
                    aq.append(A[i])
                    bq.append(B[i])
                    count += 1
        if count == 0 and len(A) > len(set(A)):
            return True
        elif count != 2:
            return False
        else:
            if aq[0] == bq[1] and bq[0] == aq[1]:
                return True
            else:
                return False
