class Solution:
    def buddyStrings(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: bool
        """
        if len(A)==0 or len(B)==0:
            return False
        if len(A)!=len(B):
            return False
        i=-1
        j=-1
        k=0
        dic={}
        while(k<len(A)):
            if A[k]!=B[k]:
                if i==-1:
                    i=k
                elif j==-1:
                    j=k
                else:
                    return False
            else:
                if A[k] in dic:
                    dic[A[k]]+=1
                else:
                    dic[A[k]]=1
            k=k+1
        if i==-1:
            for x in dic:
                if dic[x]>=2:
                    return True
            return False
        elif A[i]==B[j] and A[j]==B[i]:
            return True
        else:
            return False
