class DinnerPlates {
    set<int> validStack;
    int capacity = 0;
    map<int, stack<int>> m;
public:
    DinnerPlates(int capacity) {
        this->capacity = capacity;
    }
    
    void push(int val) {
        int pos;
        if (validStack.empty()) {
            pos = m.size();
            validStack.insert(pos);
        }
        else { 
            pos = *validStack.begin();
        }
        m[pos].push(val);
        if (m[pos].size() == capacity) {
            validStack.erase(pos);
        }
    }
    
    int pop() {
        if (m.empty())
            return -1;
        int pos = m.rbegin()->first;
        int val = m[pos].top();
        m[pos].pop();
        validStack.insert(pos);
        if (m[pos].size() == 0)
            m.erase(pos);
        return val;
    }
    
    int popAtStack(int index) {
        if (m.find(index) == m.end())
            return -1;
        int val = m[index].top();
        m[index].pop();
        validStack.insert(index);
        if (m[index].size() == 0)
            m.erase(index);
        return val;
    }
};
