class DinnerPlates {
private:
  set<int> nonFull;
  set<int> nonEmpty;
  vector<stack<int> > v;
  
  int limit;
  
public:
    DinnerPlates(int capacity) {
      nonFull.clear();
      nonEmpty.clear();
      v.clear();
      limit = capacity;
    }
    
    void push(int val) {
      
      if(nonFull.empty()) {
        stack<int> cur;
        cur.push(val);
        v.push_back(cur);
        nonEmpty.insert(-(v.size() -1));
        if(limit > 1) nonFull.insert(v.size() -1);
      } else {
        set<int>::iterator it = nonFull.begin();
        int index = (*it);
        
        v[index].push(val);
        
        if(v[index].size() == 1) nonEmpty.insert(-index);
        if(v[index].size() == limit) nonFull.erase(index);
      }
      
    
    }
    
    int pop() {
        if(nonEmpty.empty()) {
          return -1;
        } else {
          set<int>::iterator it = nonEmpty.begin();
          
          int index = -(*it);
          
          int ans = v[index].top();
          v[index].pop();
          
          nonFull.insert(index);
          
          if(v[index].size() == 0) nonEmpty.erase(-index);
          
          return ans;
        }
    }
    
    int popAtStack(int index) {
        if(index >= v.size() || v[index].size() == 0) {
          return -1;
        }

      
      int ans = v[index].top();
      v[index].pop();
      
      nonFull.insert(index);
          
          if(v[index].size() == 0) nonEmpty.erase(-index);
          
          return ans;
      
    }
};

 * DinnerPlates* obj = new DinnerPlates(capacity);
 * obj->push(val);
 * int param_2 = obj->pop();
 * int param_3 = obj->popAtStack(index);
