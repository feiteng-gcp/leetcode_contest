class DinnerPlates {
public:

    DinnerPlates(int capacity) {
        cap = capacity;
    }
    
    void push(int val) {
        int k = -1;
        if (nofull.empty()) {
            stacks.push_back(vector<int>());
            //stop.push_back(0);
            nofull.insert(stacks.size() - 1);
            k = stacks.size() - 1;
        }
        else {
            k = *(nofull.begin());
        }
        
        //++stop[k];
        //if (stacks[k].size() >= stop[k]) {
        //    stacks[k][stop[k] - 1] = val; 
        //} else 
        stacks[k].push_back(val);

        if (stacks[k].size() == cap) {
            nofull.erase(k);
        }
        noempty.insert(k);
    }
    
    int pop() {
        if (noempty.empty())
            return -1;
        
        int k = *(noempty.rbegin());
        if (stacks[k].size() == cap)
            nofull.insert(k);
        
        int val = stacks[k].back();
        stacks[k].pop_back();
        //--stop[k];
        //if (stop[k] == 0)
        if (stacks[k].size() == 0)
            noempty.erase(k);
        return val;
    }
    
    int popAtStack(int index) {
        if (index >= stacks.size()) return -1;
        if (stacks[index].empty()) return -1;
        
        if (stacks[index].size() == cap)
            nofull.insert(index);
        int val = stacks[index].back();
        stacks[index].pop_back();
        //--stop[index];
        //if (stop[index] == 0) 
        if (stacks[index].empty())
            noempty.erase(index);
        return val;
    }
private:
    set<int> nofull, noempty;
    vector<vector<int>> stacks;
    //vector<int> stop;
    int cap;
};

 * obj->push(val);
 * int param_2 = obj->pop();
 * int param_3 = obj->popAtStack(index);
