class DinnerPlates {
public:
    DinnerPlates(int capacity) {
        cap = capacity;
    }
    
    void push(int val) {
        if (hash.size() == 0)        
        {
            stack<int> stk;            
            vecStk.push_back(stk);            
            vecStk.back().push(val);
            if (vecStk.back().size() < cap) hash[vecStk.size() - 1] = 0;        
            lastIndex = max(lastIndex, int(vecStk.size()) - 1);
            return;
        }
        
        int index = hash.begin()->first;
        vecStk[index].push(val);
        if (vecStk[index].size() >= cap) hash.erase(index);        
        lastIndex = max(lastIndex ,index);        
    }
    
    int pop() {    
        
        for (int i = lastIndex; i >= 0; --i)
            if (vecStk[i].size() != 0)
            {
                int ret = vecStk[i].top();
                vecStk[i].pop();
                hash[i] = 0;
                lastIndex = i;
                return ret;
            }
        
        return -1;
    }
    
    int popAtStack(int index) {
        //if (lastIndex < 0) return - 1;
        if (index >= vecStk.size()) return -1;
        if (vecStk[index].size() == 0) return -1;
        
        int ret = vecStk[index].top();
        vecStk[index].pop();
        hash[index] = 0;
        
        return ret;
    }
    
    map<int, int> hash;    
    vector<stack<int>> vecStk;
    int cap;
    int lastIndex;
};

 * DinnerPlates* obj = new DinnerPlates(capacity);
 * obj->push(val);
 * int param_2 = obj->pop();
 * int param_3 = obj->popAtStack(index);
 */
