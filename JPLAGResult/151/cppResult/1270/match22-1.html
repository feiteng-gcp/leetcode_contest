class DinnerPlates {
private:
    int cp;
    set<int> less;
    vector<stack<int>> v;
public:
    DinnerPlates(int capacity) {
        cp = capacity;
    }
    
    void push(int val) {
        if (less.empty()) {
            v.push_back(stack<int>());
            v.back().push(val);
            if (cp > 1) less.insert(v.size() - 1);
        } else {
            v[*less.begin()].push(val);
            if (v[*less.begin()].size() == cp)
                less.erase(less.begin());
        }
    }
    
    int pop() {
        int ans = popAtStack((int)(v.size()) - 1);
        while (!v.empty() && v.back().empty()) {
            if (less.count((int)v.size() - 1))
                less.erase((int)v.size() - 1);
            v.pop_back();
        }
        // cout << "popping " << ans << endl;
        return ans;
    }
    
    int popAtStack(int index) {
        if (index < 0 || index >= v.size()) return -1;
        if (v[index].empty()) return -1;
        int ans = v[index].top();
        v[index].pop();
        less.insert(index);
        // cout << "popping at " << index << ": " << ans << endl;
        return ans;
    }
};

 * int param_3 = obj->popAtStack(index);
