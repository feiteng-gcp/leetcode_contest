class DinnerPlates {
  private:
    vector<stack<int> > stacks;
    // indices of free stacks, could be empty.
    set<int> freelist;
    int cap;
  public:
    DinnerPlates(int capacity) {
      cap = capacity;
    }

    void push(int val) {
      if (freelist.empty()) {
        // Have to add into the stack.
        stack<int> new_stack;
        stacks.push_back(new_stack);
        stacks.back().push(val);
        if (cap > 1) freelist.insert(stacks.size() - 1);
      } else {
        // Insert into the first stack in freelist.
        int to_insert = *freelist.begin();
        push(to_insert, val);
      }
    }

    void push(int idx, int val) {
      stacks[idx].push(val);
      if (stacks[idx].size() == cap) {
        freelist.erase(idx);
      }
    }

    int pop() {
      if (stacks.empty()) {
        return -1;
      }
      int rtn = stacks.back().top();
      int idx = stacks.size() - 1;
      stacks.back().pop();
      if (stacks.back().empty()) {
        stacks.pop_back();
        if (freelist.find(idx) != freelist.end()) freelist.erase(idx);
      } else {
        freelist.insert(idx);
      }
      return rtn;
    }

    int popAtStack(int index) {
      if (index < 0 || index >= stacks.size()) return -1;
      if (stacks[index].empty()) return -1;
      if (index == stacks.size() - 1) {
        return pop();
      }
      int rtn = stacks[index].top();
      stacks[index].pop();
      freelist.insert(index);
      return rtn;
    }
};
