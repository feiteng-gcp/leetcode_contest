class DinnerPlates {
public:
    int cap;

    unordered_map<int, stack<int>> stks;
    set<int> notfull;
    set<int> notempty;
    DinnerPlates(int capacity) {
        cap = capacity;
        for (int i = 0; i <= 200000; i++) {
            notfull.insert(i);
        }
    }
    
    void push(int val) {
        int i = *notfull.begin();
        stks[i].push(val);
        notempty.insert(i);
        if (stks[i].size() == cap) {
            notfull.erase(i);
        }
    }
    
    int pop() {
        if (notempty.empty()) {
            return (-1);
        }

        int i = *notempty.rbegin();
        int ans = stks[i].top();
        stks[i].pop();
        if (stks[i].empty()) {
            notempty.erase(i);
        }
        return (ans);
    }
    
    int popAtStack(int index) {
        if (stks[index].empty()) {
            return (-1);
        }
        int ans = stks[index].top();
        stks[index].pop();
        if (stks[index].empty()) {
            notempty.erase(index);
        }
        if (stks[index].size() == cap - 1) {
            notfull.insert(index);
        }
        return (ans);
    }
};

 * int param_3 = obj->popAtStack(index);
