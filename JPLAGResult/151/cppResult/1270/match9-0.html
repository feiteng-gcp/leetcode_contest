class DinnerPlates {
 public:
  DinnerPlates(int capacity) : cap_(capacity) {}

  void push(int val) {
    if (cand_.empty()) {
      cand_.insert(plates_.size());
      plates_.emplace_back();
    }
    int i = *cand_.begin();
    plates_[i].push(val);
    if (plates_[i].size() == cap_) {
      cand_.erase(i);
    }
    wtf_.insert(i);
  }

  int pop() {
    if (wtf_.empty()) {
      return -1;
    }
    int i = *wtf_.rbegin();
    int ans = plates_[i].top();
    plates_[i].pop();
    cand_.insert(i);
    if (plates_[i].empty()) {
      wtf_.erase(i);
    }
    return ans;
  }

  int popAtStack(int index) {
    if (index >= plates_.size() || plates_[index].empty()) {
      return -1;
    }
    int ans = plates_[index].top();
    plates_[index].pop();
    cand_.insert(index);
    if (plates_[index].empty()) {
      wtf_.erase(index);
    }
    return ans;
  }

 private:
  int cap_;
  set<int> cand_, wtf_;
  vector<stack<int>> plates_;
};
