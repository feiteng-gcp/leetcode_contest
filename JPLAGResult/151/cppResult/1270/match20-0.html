class DinnerPlates {
public:
    vector<stack<int> > a;
    int n;
    set<int> avail, nonem;
    
    DinnerPlates(int capacity) {
        a.clear();
        avail.clear();
        nonem.clear();
        n = capacity;
    }
    
    void push(int val) {
        int target = -1;
        if (SZ(avail) == 0) {
            stack<int> s;
            a.emplace_back(s);
            avail.insert(SZ(a)-1);
        }
        target = *avail.begin();
        a[target].push(val);
        if (SZ(a[target]) == n) avail.erase(target);
        nonem.insert(target);
    }
    
    int pop() {
        if (SZ(nonem) == 0) return -1;
        int target = *nonem.rbegin();
        int res = a[target].top(); a[target].pop();
        if (SZ(a[target]) < n) avail.insert(target);
        if (SZ(a[target]) == 0) nonem.erase(target);
        return res;
    }
    
    int popAtStack(int index) {
        if (index >= SZ(a)) return -1;
        if (SZ(a[index]) == 0) return -1;
        int target = index;
        int res = a[target].top(); a[target].pop();
        if (SZ(a[target]) < n) avail.insert(target);
        if (SZ(a[target]) == 0) nonem.erase(target);
        return res;
    }
};
