class DinnerPlates {
public:
    int capa;
    // vector<int> tmp;
    vector<vector<int>> element;
    set<int> notempty; 
    set<int> notfull;
    DinnerPlates(int capacity) {

        capa=capacity;
        // tmp.resize(200000, 0);
        element.resize(200000);
        for(int i=0;i<200000;i++){
            notfull.insert(i);
        }
        
    }
    
    void push(int val) {
        int i;
        i = *notfull.begin();
        element[i].push_back(val);
        
        notempty.insert(i);
        
        if(element[i].size()==capa){
            notfull.erase(i);
        }
    }
    
    int pop() {
        if(notempty.empty()) return -1;
        int i = *notempty.rbegin();
        int res = element[i].back();
        element[i].pop_back();
        notfull.insert(i);
        if(element[i].size()==0){
            notempty.erase(i);
        }
        return res;
    }
    
    int popAtStack(int index) {
        if(element[index].empty()) return -1;
        notfull.insert(index);
        int res = element[index].back();
        element[index].pop_back();
        return res;
    }
};

 * DinnerPlates* obj = new DinnerPlates(capacity);
 * obj->push(val);
 * int param_2 = obj->pop();
 * int param_3 = obj->popAtStack(index);
