class DinnerPlates {
private:
    vector<vector<int>> stacks;
    int cap;
    set<int> non_full;
    set<int> non_empty;

public:
    DinnerPlates(int capacity) {
        cap = capacity;
    }
    
    void incrStack() {
        vector<int> s;
        stacks.push_back(s);

        int idx = stacks.size() - 1;
        change(stacks[idx], idx);
    }
    
    void change(vector<int>& stk, int idx) {
        int size = stk.size();
        if (size == cap) {
            non_full.erase(idx);
            non_empty.insert(idx); 
        } else if (size == 0) {
            non_full.insert(idx);
            non_empty.erase(idx);
        } else {
            non_full.insert(idx);
            non_empty.insert(idx);
        }
    }
    
    void push(int val) {
        if (non_full.empty()) {
            incrStack();
        } 
        
        int idx = *non_full.begin();
        stacks[idx].push_back(val);
        change(stacks[idx], idx);
    }
    
    int pop() {
        if (!non_empty.empty()) {
            int idx = *non_empty.rbegin();
            
            int val = stacks[idx][stacks[idx].size() - 1];
            stacks[idx].pop_back();
            change(stacks[idx], idx);
            return val;
        } else {

            return -1;
        }
    }
    
    int popAtStack(int index) {
        if (index >= stacks.size() || stacks[index].size() == 0) {
            return -1;
        }
        
        int val = stacks[index][stacks[index].size() - 1];
        stacks[index].pop_back();
        change(stacks[index], index);
        return val;
    }
};

 * DinnerPlates* obj = new DinnerPlates(capacity);
 * obj->push(val);
 * int param_2 = obj->pop();
 * int param_3 = obj->popAtStack(index);
