class DinnerPlates {
public:
    DinnerPlates(int capacity) : capacity(capacity) {}
    
    void push(int val) {
        int stack_index = 0;
        if (stack_with_space.empty()) {
            if (!stacks.empty()) {
                stack_index = stacks.rbegin()->first + 1;
            }
        } else {
            stack_index = *(stack_with_space.begin());
        }
        stacks[stack_index].push(val);
        if (stacks[stack_index].size() == capacity) {
            stack_with_space.erase(stack_index);
        } else if (stacks[stack_index].size() == 1) {
            stack_with_space.insert(stack_index);
        }
    }

    int pop() {
        if (stacks.empty()) {
            return -1;
        }
        return popAtStack(stacks.rbegin()->first);
    }
    
    int popAtStack(int index) {
        if (stacks.find(index) == stacks.end() || stacks[index].empty()) {
            return -1;
        }
        int val = stacks[index].top();
        stacks[index].pop();
        stack_with_space.insert(index);
        if (stacks[index].empty()) {
            stacks.erase(index);
        }
        return val;
    }
    
private:
    map<int, stack<int>> stacks;
    set<int> stack_with_space;
    int capacity;
};
