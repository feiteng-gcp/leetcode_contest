class DinnerPlates {
public:
  int C; 
  vector<stack<int>> V;
  set<int> S;
  DinnerPlates(int capacity) {
    C = capacity;
  }

  void push(int val) {
    int t = -1; 
    if (S.empty()) {
      V.push_back(stack<int>());
      t = V.size() - 1;
      S.insert(t);
    } else {
      t = *S.begin(); 
    }
    //cout << "push " << t << " " << val << endl;
    V[t].push(val);
    if (V[t].size() == C) S.erase(t);
  }

  int pop() {
    if (V.size() == 0) return -1;
    return popAtStack(V.size() - 1);
  }

  int popAtStack(int index) {
    if (index >= V.size()) return -1;
    if (V[index].empty()) return -1;
    int t = V[index].top();
    V[index].pop();
    if (V[index].size() == 0 && index == V.size() - 1) {
      V.pop_back();
      S.erase(index);
    }
    else S.insert(index); 
    //cout << "pop " << index << " " << t << endl;
    return t;
  }
};
