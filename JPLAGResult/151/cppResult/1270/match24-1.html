class DinnerPlates {
public:
    vector<vector<int>> a;
    set<int> alive, full;
    int C;
    DinnerPlates(int capacity) {
        a.clear();
        alive.clear();
        full.clear();
        C = capacity;
    }
    
    void push(int val) {
        if (alive.empty()) {            
            a.push_back({val});
            int k = a.size() - 1;
            if (a[k].size() == C) {
                full.insert(k);
            } else {
                alive.insert(k);
            }
        } else {
            int k = *alive.begin();
            a[k].push_back(val);
            if (a[k].size() == C) {
                alive.erase(k);
                full.insert(k);
            }

        }
        // cout << "push: " << val << endl;
        // cout << "==alive==" << endl;
        // for (auto& it : alive) cout << it << " ";
        // cout << endl;
        // cout << "==full==" << endl;
        // for (auto& it : full) cout << it << " ";
        // cout << endl;
        // cout << "===" << endl;
    }
    
    int pop() {
        int k = -1;
        if (!alive.empty()) {
            k = max(k, *alive.rbegin());
        }
        if (!full.empty()) {
            k = max(k, *full.rbegin());
        }
        if (k < 0) return -1;
        int ret = a[k].back();
        a[k].pop_back();
        if (a[k].size() == 0) {
            if (alive.count(k)) alive.erase(k);
            if (full.count(k)) full.erase(k);
        }
        if (a[k].size() == C - 1) {
            full.erase(k);
            if (C - 1 > 0) alive.insert(k);

        }
        // cout << "pop" << endl;
        // cout << "==alive==" << endl;
        // for (auto& it : alive) cout << it << " ";
        // cout << endl;
        // cout << "==full==" << endl;
        // for (auto& it : full) cout << it << " ";
        // cout << endl;
        // cout << "===" << endl;
        return ret;
    }
    
    int popAtStack(int k) {
        if (a[k].size() == 0) return -1;
        int ret = a[k].back();
        a[k].pop_back();
        if (a[k].size() == 0) {
            if (alive.count(k)) alive.erase(k);
            if (full.count(k)) full.erase(k);
        }
        if (a[k].size() == C - 1) {
            full.erase(k);
            if (C - 1 > 0) alive.insert(k);
        }
        
        // cout << "popAtStack " << k << endl;
        // cout << "==alive==" << endl;
        // for (auto& it : alive) cout << it << " ";
        // cout << endl;
        // cout << "==full==" << endl;
        // for (auto& it : full) cout << it << " ";
        // cout << endl;
        // cout << "===" << endl;
        
        return ret;
    }
};
