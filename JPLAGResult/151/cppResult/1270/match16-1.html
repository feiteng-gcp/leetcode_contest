class DinnerPlates {
public:
    vector<vector<int>> data;
    int size;
    set<int> todo;
    set<int> ne;
    DinnerPlates(int capacity) {
        size=capacity;
    }
    
    void push(int val) {
        if (todo.empty()) {
            data.push_back(vector<int>());
            todo.insert(data.size()-1);
        }
        data[*todo.begin()].push_back(val);
        ne.insert(*todo.begin());

        if (data[*todo.begin()].size()==size) {
            todo.erase(*todo.begin());
        }
    }
    
    int pop() {
        if (ne.empty()) {
            return -1;
        }
        int idx=*prev(ne.end());

        return popAtStack(idx);
    }
    
    int popAtStack(int index) {
        if (ne.find(index)==ne.end()) {
            return -1;
        }
        int ret=data[index].back();
        data[index].pop_back();
        todo.insert(index);
        if (data[index].empty()) {
            ne.erase(index);
        }
        return ret;
    }
};
