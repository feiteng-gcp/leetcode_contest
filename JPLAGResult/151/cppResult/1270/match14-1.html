int mn = 0, mx = 0;
stack<int> s[200500];
int c;

class DinnerPlates {
public:
    DinnerPlates(int capacity) {
        c = capacity;
        mn = mx = 0;
        for (int i = 0; i <= 200000; ++i) while (!s[i].empty()) s[i].pop();

    }
    
    void push(int val) {
        for (int i = mn; ; ++i) {
            if (s[i].size() < c) {
                s[i].push(val);
                mn = i;
                mx = max(mx, i);
                break;
            }
        }

    }
    
    int pop() {
        for (int i = mx; i >= 0; --i) {
            if (s[i].size() > 0) {
                int x = s[i].top(); s[i].pop();
                mn = min(mn, i);
                mx = i;
                return x;               
            }
        }
        mn = mx = 0;
        return -1;
    }
    
    int popAtStack(int index) {
        if (s[index].size() == 0) return -1;
        int x = s[index].top(); s[index].pop();
        mn = min(mn, index);
        return x;
    }
};

 * int param_3 = obj->popAtStack(index);
