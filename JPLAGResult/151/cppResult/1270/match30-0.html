class Segments {
 public:
   void insert(int x) {
       auto itr = tree.lower_bound(x);
       if (itr != tree.begin() && prev(itr, 1)->second >= x - 1) {
           --itr;
           itr->second = max(itr->second, x);
       } else {
           tree[x] = x;
           itr = tree.lower_bound(x);
       }
       auto next_itr = next(itr, 1);
       if (next_itr != tree.end() && next_itr->first == x + 1) {
           itr->second = next_itr->second;
           tree.erase(next_itr);
       }
   }
   
    void erase(int x) {
        auto itr = tree.upper_bound(x);
        if (itr == tree.begin()) {
            return;
        }
        --itr;
        if (itr->first == x) {
            if (itr->second == x) {
                tree.erase(itr);
            } else {
                tree[x + 1] = itr->second;
                tree.erase(itr);
            }
        } else {
            if (itr->second > x) {
                tree[x + 1] = itr->second;
            }
            itr->second = x - 1;
        }
    }
    
    int firstNonNegative() const {
        if (tree.begin() == tree.end() || tree.begin()->first > 0) {
            return 0;
        }
        return tree.begin()->second + 1;
    }
    
 private:
   map<int, int> tree; 
};

class DinnerPlates {
public:
    DinnerPlates(int capacity_) : capacity(capacity_) {
        
    }
    
    void push(int val) {
        int index = full_stacks.firstNonNegative();
        auto &stack = stacks[index];
        stack.emplace(val);
        if (stack.size() == capacity) {
            full_stacks.insert(index);
        }
    }
    
    int pop() {
        if (stacks.empty()) {
            return -1;
        }
        auto &stack = stacks.rbegin()->second;
        int result = stack.top();
        stack.pop();
        if (stack.size() + 1 == capacity) {
            full_stacks.erase(stacks.rbegin()->first);
        }
        if (stack.empty()) {
            stacks.erase(stacks.rbegin()->first);
        }
        return result;
    }
    
    int popAtStack(int index) {
        auto &stack = stacks[index];
        if (stack.empty()) {
            return -1;
        }
        int result = stack.top();
        stack.pop();
        if (stack.size() + 1 == capacity) {
            full_stacks.erase(index);
        }
        if (stack.empty()) {
            stacks.erase(index);
        }
        return result;
    }
private:
    const int capacity;
    map<int, stack<int>> stacks;
    Segments full_stacks;
};
