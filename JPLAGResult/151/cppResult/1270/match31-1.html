class DinnerPlates {
public:
    priority_queue<int, vector<int>, greater<int> > to_fill;
    vector <stack <int> > plates;
    int c;
    DinnerPlates(int capacity) {
        c = capacity;        
    }
    
    void push(int val) {
        if (to_fill.size() == 0) {
            plates.push_back(stack<int>());
            to_fill.push(plates.size()-1);
        }
        
        int s = to_fill.top();
        cout << "Pushed at " << s << endl;
        plates[s].push(val);
        if (plates[s].size() == c)
            to_fill.pop();
    }
    
    int pop() {
        int s = plates.size() - 1;
        while (s >= 0 && plates[s].size() == 0) {
            plates.pop_back();

            s--;
        }
        
        if (s < 0) return -1;
        
        int val = plates[s].top();
        if (plates[s].size() == c)
            to_fill.push(s);
        
        plates[s].pop();
        return val;
    }
    
    int popAtStack(int index) {
        if (plates[index].size() == 0) return -1;
        
        int val = plates[index].top();
        if (plates[index].size() == c) to_fill.push(index);
        
        plates[index].pop();
        
        return val;
    }
};

 * int param_3 = obj->popAtStack(index);
