class DinnerPlates {

private:
    int C;
    vector<stack<int>> vec;
    set<int> okStack, nonEmptyStack;

public:
    DinnerPlates(int capacity): C(capacity){}

    void push(int val) {
        if(okStack.size()){
            int index = *okStack.begin();
            vec[index].push(val);
            if(vec[index].size() == C) okStack.erase(index);
            nonEmptyStack.insert(index);
        }
        else{
            if(vec.empty() || vec.back().size() == C){
                stack<int> newstack;
                vec.push_back(newstack);
            }
            vec.back().push(val);
            nonEmptyStack.insert(vec.size() - 1);
        }
    }

    int pop() {

        if(nonEmptyStack.size() == 0) return -1;
        int index = *nonEmptyStack.rbegin();
        int ret = vec[index].top();
        vec[index].pop();
        okStack.insert(index);
        if(vec[index].size() == 0) nonEmptyStack.erase(index);
        return ret;
    }

    int popAtStack(int index) {
        if(index < 0 || index >= vec.size()) return -1;
        if(vec[index].size() == 0) return -1;
        int ret = vec[index].top();
        vec[index].pop();
        okStack.insert(index);
        if(vec[index].size() == 0) nonEmptyStack.erase(index);
        return ret;
    }
};
