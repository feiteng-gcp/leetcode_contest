class DinnerPlates {
public:
    vector<stack<int>> data;
    int cap;
    int l, r;
    DinnerPlates(int capacity) : cap(capacity) {
        data.reserve(100000);
        l = r = 0;
    }
    
    void push(int val) {
        for (int i = l; i < data.size(); ++i) {
            if (data[i].size() < cap) {
                data[i].push(val);
                l = i;
                r = max(r, i);
                return;
            }
        }
        data.push_back(stack<int>());
        data.back().push(val);
        l = r = data.size() - 1;
    }
    
    int pop() {
        for (int i = r; i >= 0; --i) {
            if (data[i].size() > 0) {
                int ret = data[i].top();
                data[i].pop();
                r = i;
                l = min(l, i);
                return ret;
            }
        }
        r = 0;
        return -1;
    }
    
    int popAtStack(int index) {
        if (data.size() <= index) return -1;
        if (data[index].size() <= 0) return -1;
        int ret = data[index].top();
        data[index].pop();
        l = min(l, index);
        return ret;
    }
};

 * int param_3 = obj->popAtStack(index);
