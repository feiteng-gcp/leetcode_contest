class DinnerPlates {
    set<int> nonEmptyList;
    set<int> nonFullList;
    vector<stack<int>> stackList;
    int capacity;
public:
    DinnerPlates(int capacity) {
        this->capacity = capacity;
    }
    
    void push(int val) {
        if (nonFullList.empty()) {
            nonFullList.insert(stackList.size());
            stackList.push_back(stack<int>());

        }
        int idx = *nonFullList.begin();
        stackList[idx].push(val);
        nonEmptyList.insert(idx);
        if (stackList[idx].size() == capacity) {
            nonFullList.erase(idx);
        }
    }
    
    int pop() {
        if (nonEmptyList.empty()) {
            return -1;
        }
        int idx = *nonEmptyList.rbegin();
        int res = stackList[idx].top();
        stackList[idx].pop();
        nonFullList.insert(idx);
        if (stackList[idx].empty()) {
            nonEmptyList.erase(idx);
        }
        return res;
    }
    
    int popAtStack(int index) {
        if (index >= stackList.size() || stackList[index].empty()) {
            return -1;
        }
        int res = stackList[index].top();
        stackList[index].pop();
        nonFullList.insert(index);
        if (stackList[index].empty()) {
            nonEmptyList.erase(index);
        }
        return res;
    }
};
