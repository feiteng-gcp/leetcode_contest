class DinnerPlates {
public:
    DinnerPlates(int capacity) {
        cap = capacity;
    }
    
    void push(int val) {
        if (not_full.empty() && empty.empty()) {
            stk.push_back(stack<int>());
            empty.insert(stk.size() - 1);
        }
        
        int pos = 0x3f3f3f3f;
        if (!not_full.empty()) pos = min(pos, *(not_full.begin()));
        if (!empty.empty()) pos = min(pos, *(empty.begin()));
        stk[pos].push(val);
        
        if (stk[pos].size() == 1) {
            remove(empty, pos);
            not_full.insert(pos);
        }
        
        if (stk[pos].size() == cap) {
            not_full.erase(pos);
            full.insert(pos);
        }
    }
    
    int pop() {
        int v = -1;
        if (!full.empty()) {
            v = max(v, *(full.rbegin()));
        }
        if (!not_full.empty()) {
            v = max(v, *(not_full.rbegin()));
        }
        
        if (v == -1) return -1;
        int val = stk[v].top();
        stk[v].pop();
        
        if (stk[v].empty()) {
            empty.insert(v);
            remove(full, v);
            remove(not_full, v);
        } else {    
            if (full.count(v) != 0) {
                full.erase(v);
                not_full.insert(v);
            }
        }
        return val;
    }
    
    int popAtStack(int index) {
        if (full.count(index) == 0 && not_full.count(index) == 0) return -1;
        int val = stk[index].top();
        stk[index].pop();
        
        if (stk[index].empty()) {
            empty.insert(index);
            remove(full, index);
            remove(not_full, index);
        } else {    
            if (full.count(index) != 0) {
                full.erase(index);
                not_full.insert(index);
            }
        }
    
        return val;
    }
    
    void remove(set<int>& s, int val) {
        if (s.count(val) != 0) s.erase(val);
    }
    
    int cap;
    set<int> full;
    set<int> not_full;
    set<int> empty;
    vector<stack<int>> stk;
};

 * int param_3 = obj->popAtStack(index);
