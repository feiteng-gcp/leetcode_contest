class DinnerPlates {
public:
    int cap;
    vector<int> stk[100004];
    set<int> emp, non_emp;
    DinnerPlates(int _c) {
        cap = _c;
        for(int i=0;i<=100000;i++){
            stk[i].clear();
            emp.insert(i);
        }
       // cout << "ok\n";
    }
    
    void push(int val) {

        int id = *emp.begin();
        stk[id].push_back(val);
        non_emp.insert(id);
        if(stk[id].size()==cap) emp.erase(id);
        //cout << "push " << val << "\n";
    }
    
    int pop() {
        //cout << "pop\n";
        if(non_emp.size()==0) return -1;
        int id = *non_emp.rbegin();
        int ret = stk[id].back();
        stk[id].pop_back();
        if(stk[id].size()==cap-1) emp.insert(id);
        if(stk[id].size()==0) non_emp.erase(id);
       // cout << "  popped\n";
        return ret;
    }
    
    int popAtStack(int index) {
       // cout << "popIdx" << index << "\n";
        if(stk[index].size()==0) return -1;
        int ret = stk[index].back();
        stk[index].pop_back();
        if(stk[index].size()==cap-1) emp.insert(index);
         //cout << "   poppedIdx\n";
        return ret;
    }
};

 * obj->push(val);

 * int param_3 = obj->popAtStack(index);
