class DinnerPlates {
public:
    vector < int > st[100100];
    int cap;
    set < int > S; //set of available
    set < int > M;
    
    
    DinnerPlates(int cap) {
        this->cap = cap;
        for (int i=0;i<100100;i++) {
            S.insert(i);
        }
    }
    
    void push(int val) {
        int ind = *S.begin();
        st[ind].pb(val);
        M.insert(ind);
        if (sz(st[ind]) == cap) {
            S.erase(S.begin());
        }
    }
    
    int pop() {
        if (M.empty()) {
            return -1;
        }        

        int ind = *(--M.end());
        int val = st[ind].back();
        st[ind].pop_back();
        if (st[ind].empty()) {
            M.erase(ind);
        }
        if (sz(st[ind]) < cap) {
            S.insert(ind);
        }
        return val;
    }
    
    int popAtStack(int index) {

        if (st[index].empty()) {
            return -1;
        }   
        int val = st[index].back();
        st[index].pop_back();
        if (sz(st[index]) <= cap) {
            S.insert(index);
        }
        if (st[index].empty()) {
            M.erase(index);
        }
        return val;
    }
};
