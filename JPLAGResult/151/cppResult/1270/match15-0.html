class DinnerPlates {
public:
	vector<int> Data[100001];
	int Num[100001] = { 0 };
	int N,max = 0,min = 0;
	DinnerPlates(int capacity) {
		N = capacity;
	}

	void push(int val) {
		for (int i = min; i < 100001; ++i)
		{
			if (Data[i].size() < N)
			{
				Data[i].push_back(val);
				Num[i]++;
				if (i > max)
					max = i;
                if(i > min)
                    min  =i;
				return;
			}
		}
	}

	int pop() {
		for (int i = max; i >= 0; --i)
		{
			if (Data[i].size() > 0)
			{
				int res = Data[i][Num[i] - 1];
				Num[i] --;
				Data[i].pop_back();
                if(max > i)
                    max = i;
				return res;
			}
		}
		return -1;
	}

	int popAtStack(int index) {
		if (Data[index].size() > 0)
		{
			int res = Data[index][Num[index] - 1];
			Num[index] --;
			Data[index].pop_back();
            if(index < min)
                min = index;
			return res;
		}
		else
			return -1;
	}
};

 * obj->push(val);
 * int param_2 = obj->pop();
 * int param_3 = obj->popAtStack(index);
 */
