class DinnerPlates {
public:
    DinnerPlates(int capacity): capacity(capacity), rightmostId(-1) {}
    
    void push(int val) {
        if (nonFullIds.empty())
            nonFullIds.insert(rightmostId + 1);
        int id = *(nonFullIds.begin());
        while (stacks.size() <= id) stacks.push_back(stack<int>());
        stacks[id].push(val);
        if (stacks[id].size() < capacity) {
            nonFullIds.insert(id);
        } else {
            nonFullIds.erase(id);
        }
        rightmostId = max(rightmostId, id);

    }
    
    int pop() {
        if (rightmostId < 0)
            return -1;
        return popAtStack(rightmostId);
    }
    
    int popAtStack(int index) {
        if (stacks.size() <= index)
            return -1;
        if (stacks[index].empty())
            return -1;
        int ans = stacks[index].top();
        stacks[index].pop();
        nonFullIds.insert(index);
        if (index == rightmostId) {
            while (rightmostId >= 0 && stacks[rightmostId].empty())
                --rightmostId;
        }
        return ans;
    }
    
    vector<stack<int>> stacks;
    int capacity;
    set<int> nonFullIds;
    int rightmostId;
};
