class DinnerPlates {
    int capacity;
    vector<stack<int>> ss;
    set<int> notempty;
    set<int> notfull;
public:
    DinnerPlates(int capacity) {
        this->capacity = capacity;
        ss=vector<stack<int>>();
    }
    
    void push(int val) {
        if(notfull.size()==0){
            ss.push_back(stack<int>());
            notfull.insert(ss.size()-1);
        }
        int t = *notfull.begin();
        ss[t].push(val);
        if(ss[t].size()==capacity) notfull.erase(t);

        if(ss[t].size()==1)
            notempty.insert(t);
    }
    
    int pop() {
        if(notempty.size()==0) return -1;
        int t=*notempty.rbegin();
        int res=ss[t].top();
        ss[t].pop();
        if(ss[t].size()==0) notempty.erase(t);
        if(ss[t].size()+1==capacity)
            notfull.insert(t);
        return res;
    }
    
    int popAtStack(int index) {
        if(index>=ss.size()||ss[index].size()==0) return -1;
        int res=ss[index].top();
        ss[index].pop();
        if(ss[index].size()==0) notempty.erase(index);
        if(ss[index].size()+1==capacity)
            notfull.insert(index);
        return res;
    }
};

 * int param_3 = obj->popAtStack(index);
