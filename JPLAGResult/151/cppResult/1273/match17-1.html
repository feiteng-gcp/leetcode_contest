class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> counts;
        for(string &w: words){
            counts.push_back(helper(w));
        }
        vector<int> res;
        for(string &q: queries){
            int tmp = helper(q);

            int tmpRes = 0;
            for(int i = 0; i < counts.size(); i++){
                if(counts[i] > tmp) tmpRes++;
            }
            res.push_back(tmpRes);
        }
        return res;
    }
    
    int helper(string &s){
        char mn = 'z';
        int cnt = 0;

        for(char c: s){
            if(c < mn){
                mn = c;
                cnt = 1;
            }else if(c == mn){
                cnt++;
            }
        }
        //cout<<cnt<<endl;
        return cnt;
    }
};
