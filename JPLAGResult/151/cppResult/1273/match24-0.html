class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> p;
        for( auto w : words ) p.push_back(calc(w));
        sort(p.begin(), p.end());
        vector<int> r;
        for( auto q : queries ){
            r.push_back((int)(p.end() - upper_bound(p.begin(), p.end(), calc(q))));
        }
        return r;
    }
    
    int calc(string w){
        int memo[128] = {};
        for( auto c : w )
            memo[c]++;
        for(int i = 0 ; i < 128 ; i++){
            if( memo[i]) return memo[i];
        }
        return 0;
    }
};
