class Solution {
public:
    char leastChar(string s){
        char minC = s[0];
        for (char c : s) {
            if (c<minC) {
                minC = c;
            }
        }
        return minC;
    }
    
    char countChar(string s, char c){
        int count = 0;
        for (char nc : s) {
            if (nc==c) count++;
        }
        return count;
    }
    
    int f(string s) {
        char c = leastChar(s);
        return countChar(s,c);
    }
    
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {

        vector<int> answers;
        for (auto query : queries) {
            int count = f(query);
            int answer = 0;
            for (auto word : words) {
                if (f(word) > count) answer++;
            }
            answers.push_back(answer);
        }
        return answers;
    }
};
