class Solution {
public:
  int calc(string &s) {
    char c = *min_element(s.begin(), s.end());
    return count(s.begin(), s.end(), c);
  }
  vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
    vector<int> cnt(16);
    for (auto t: words) cnt[calc(t)]++;
    for (int i = 14; i >= 0; i--) cnt[i] += cnt[i+1];
    vector<int> ans;
    for (auto t: queries) ans.push_back(cnt[calc(t)+1]);
    return ans;
  }
};
