class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> vv;
        for(auto s : words)
            vv.push_back( freq(s) );
        
        sort(vv.begin(), vv.end() );
        
        vector<int> ret;
        for(auto s : queries){
            int x = freq(s);
            auto it = upper_bound(vv.begin(), vv.end(), x);
            if(it == vv.end()) ret.push_back(0);
            else if( it == vv.begin() ) ret.push_back(vv.size());
            else{
                int d = distance(vv.begin(), it);
                ret.push_back(vv.size() - d);
            }
        }
        return ret;
    }
    
    int freq(string s){
        map<char, int> mmp;
        for(auto c : s)
            mmp[c]++;
        
        return mmp.begin()->second;
    }
};
