class Solution {
public:
    int comp(string &s) {
        int ans = 0;
        vector<int> d(26);
        for (auto c : s) {
            d[c-'a']++;
        }
        for (int j = 0; j < 26; j++) {
            if (d[j]) {
                ans = d[j];
                break;
            }
        }
        return (ans);
    }
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        int n = words.size();
        vector<int> f(n);

        for (int i = 0; i < n; i++) {
            f[i] = comp(words[i]);
        }
        sort(f.begin(), f.end());
        int q = queries.size();
        vector<int> ans(q);
        for (int i = 0; i < q; i++) {
            int w = comp(queries[i]);
            ans[i] = distance(upper_bound(f.begin(), f.end(), w), f.end());
        }
        return (ans);
    }
};
