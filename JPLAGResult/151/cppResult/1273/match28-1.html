class Solution {
public:
    int f(const string& s) {
        char cc = s[0];
        int cnt = 0;

        for (auto c : s) {
            if (c < cc) {
                cnt = 1, cc = c;
            } else if (c == cc) {
                ++cnt;
            }
        }
        return cnt;
    }
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> fw;
        for (auto& s : words) {
            fw.push_back(f(s));
        }
        sort(fw.begin(), fw.end());
        int N = fw.size();
        vector<int> res;
        for (auto& q : queries) {
            int v = f(q);
            auto it = upper_bound(fw.begin(), fw.end(), v);
            res.push_back(N - (it - fw.begin()));
        }
        return res;
    }
};
