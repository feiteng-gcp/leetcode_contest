class Solution {
public:
    
    int getCount(string s) {
        char c = 'z';
        for (int i=0;i<s.length();i++) {
            c = min(c, s[i]);
        }
        
        int cnt = 0;
        for (int i=0;i<s.length();i++) {
            cnt += (s[i] == c);
        }            
        return cnt;
    }
    
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector < int > S;
        for (string ww: words) {
            S.push_back(getCount(ww));
        }
        vector < int > ans;
        for (string qq: queries) {
            int cnt = getCount(qq);
            int res = 0;
            for (int i=0;i<S.size();i++) {
                if (cnt < S[i]) res++;
            }
            ans.push_back(res);
        }
        return ans;
    }
};
