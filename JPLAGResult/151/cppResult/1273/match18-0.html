class Solution {
    int f(string s){
        vector<int> idx(26,0);
        for(char c:s) idx[c-'a']++;
        for(int i=0;i<26;i++) if(idx[i]>0) return idx[i];
        return 0;
    }
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> ww;
        for(string s:words) ww.push_back(f(s));
        sort(ww.begin(),ww.end());
        vector<int> ans;
        for(string s:queries){
            int a=f(s);
            auto it=upper_bound(ww.begin(),ww.end(),a);
            ans.push_back(ww.end()-it);
        }
        return ans;
    }
};
