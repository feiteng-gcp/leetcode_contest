class Solution {
private:
  int f(string& x) {

  map<int, int> mp;
          for(int j = 0; j < x.length(); j ++) {
            mp[x[j]] ++;
          }
          
          for(int i = 'a'; i <= 'z'; i ++) {
            if(mp[i] != 0) {
              return mp[i];
            }
          }
  return 0;
}
  
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        map<int, int> c;

        for(int i = 0; i < words.size(); i ++) {
          c[f(words[i])] ++;
        }
      
      vector<int> ans;
      for(int i = 0; i < queries.size(); i ++) {
        int y = f(queries[i]);
        int sum = 0;
        for(int  j = y + 1; j <= 10; j ++) {
          sum = sum + c[j];
        }
        ans.push_back(sum);
      }
      return ans;
    }
};
