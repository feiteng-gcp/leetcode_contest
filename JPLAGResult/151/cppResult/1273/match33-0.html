class Solution {
    int getFreq(string& s) {
        vector<int> cnt(26, 0);
        for (char c : s) {
            cnt[c - 'a']++;
        }
        int smallest = s[0] - 'a';
        for (int i = 0; i < 26; i++) {
            if (cnt[i] > 0 && i < smallest) {
                smallest = i;
            }
        }
        return cnt[smallest];
    }
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> freq;
        for (auto& s : words) {
            freq.push_back(getFreq(s));
        }
        sort(freq.begin(), freq.end());
        vector<int> ans;
        for (auto& s : queries) {
            auto itr = upper_bound(freq.begin(), freq.end(), getFreq(s));
            ans.push_back(distance(itr, freq.end()));
        }
        return ans;
    }
};
