class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> qrs;
        vector<int> ans;
        for (int i = 0; i < words.size(); i++) {
            qrs.push_back(freq(words[i]));
        }
        sort(qrs.begin(), qrs.end());
        for (int i = 0; i < queries.size(); i++) {
            int fr = freq(queries[i]);
            ans.push_back(fnd(qrs, fr));
        }
        return ans;
    }
    
    int fnd(vector<int>& nmbrs, int tar) {
        int l = 0, r = nmbrs.size()-1;
        while (l < r) {
            int m = (l+r)/2;
            if (nmbrs[m] > tar) {
                r = m;
            } else {
                l = m+1;
            }
        }
        if (nmbrs[l] <= tar) l++;
        return (int)nmbrs.size()-l;
    }
    
    int freq(string& word) {
        if (word.length() == 0) return 0;

        char mn = 'z';
        for (int i = 0; i < word.length(); i++) {
            if (word[i] < mn) mn = word[i];
        }
        int ret = 0;
        for (int i = 0; i < word.length(); i++) {
            if (word[i] == mn) ret++;
        }
        return ret;
    }
};
