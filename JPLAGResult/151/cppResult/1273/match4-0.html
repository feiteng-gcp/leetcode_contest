class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        int n = queries.size(), m = words.size();
        vector<int> a(n), b(m);
        for (int i = 0; i < n; ++i) {
            vector<int> cnt(26);
            for (auto& c : queries[i]) {
                cnt[c - 'a']++;
            }
            for (int k = 0; k < 26; ++k) {
                if (cnt[k]) {
                    a[i] = cnt[k];
                    break;
                }
            }
        }
        for (int i = 0; i < m; ++i) {
            vector<int> cnt(26);
            for (auto& c : words[i]) {
                cnt[c - 'a']++;
            }
            for (int k = 0; k < 26; ++k) {
                if (cnt[k]) {
                    b[i] = cnt[k];
                    break;
                }
            }
        }
        vector<int> ret(n);
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (b[j] > a[i]) ++ret[i];
            }
        }
        return ret;
    }
};
