class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> freq;
        for (auto& w : words) {
            freq.push_back(f(w));
        }
        sort(freq.begin(), freq.end());
        
        vector<int> result;
        for (auto& q : queries) {
            int count = f(q);
            int index = binarySearch(freq, count);
            if (index == -1) {
                result.push_back(0);
            } else {
                result.push_back(freq.size() - index);
            }
        }
        return result;
    }
private:
    int binarySearch(vector<int>& freq, int target) {
        int left = 0, right = freq.size()-1;
        
        while (left + 1 < right) {
            int mid = (right - left) / 2 + left;
            if (freq[mid] > target) {
                right = mid;
            } else {
                left = mid;
            }
        }
        if (freq[left] > target) {
            return left;
        }
        
        if (freq[right] > target) {
            return right;
        }
        return -1;
    }
    int f(const string& s) {
        char smallest = 'z';
        for (auto c : s) {
            if (c < smallest) {
                smallest = c;
            }
        }
        int count = 0;
        for (auto c : s) {
            if (c == smallest) {
                count++;
            }
        }
        return count;
    }
};
