int f(const string &s) {
    int freq[26];
    fill(freq, freq + 26, 0);

    for (char c : s)
        freq[c - 'a']++;

    for (int i = 0; i < 26; i++)
        if (freq[i] > 0)
            return freq[i];

    return 0;
}

class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> word_f;

        for (const string &word : words)
            word_f.push_back(f(word));

        sort(word_f.begin(), word_f.end());
        vector<int> answer;

        for (const string &query : queries) {
            int query_f = f(query);
            answer.push_back(word_f.end() - upper_bound(word_f.begin(), word_f.end(), query_f));
        }

        return answer;
    }
};
