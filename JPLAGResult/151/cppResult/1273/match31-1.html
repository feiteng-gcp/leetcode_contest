class Solution {
public:
    int gao(string  s)
    {
        map<char,int> mp;
        for(int i = 0;i<s.size();i++)
            mp[s[i]]++;
        
        return (*mp.begin()).second;
    }
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        
        int n = queries.size();
        int m = words.size();
        vector<int> _querie(n,0);
        vector<int> _word(m,0);
        
        for(int i = 0;i<n;i++)
            _querie[i] = gao(queries[i]);
        
        for(int i = 0;i<m;i++)
            _word[i] = gao(words[i]);
        

        vector<int> ans(n,0);
        for(int i =0 ;i<n;i++)
        {
            for(int j = 0;j<m;j++)
            {
                if(_querie[i] < _word[j])
                    ans[i]++;
            }
        }
        
        return ans;
    }
};
