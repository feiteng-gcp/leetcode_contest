class Solution {
public:
    int f(string s)
    {
        int alphabet[26] = {0};
        for (char ch : s)
            alphabet[ch-'a']++;
        int res = 0;
        for (int i = 0; i < 26; i++)
            if (alphabet[i] > 0) return alphabet[i];
        return res;
    }
    
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> word(words.size(), 0);
        for (int i = 0; i < words.size(); i++)
            word[i] = f(words[i]);
        vector<int> res;
        for (int i = 0; i < queries.size(); i++)
        {
            int target = f(queries[i]);
            int sum = 0;
            for (int j = 0; j < words.size(); j++)
                if (word[j] > target) sum++;
            res.push_back(sum);
        }
        return res;
    }
};
