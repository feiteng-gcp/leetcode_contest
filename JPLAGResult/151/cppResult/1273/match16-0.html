class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        int nq = queries.size();
        int nw = words.size();
        int arr1[nq];
        int arr2[nw];
        
        for ( int i = 0; i < nq; ++i ) { 
            arr1[i] = getVal(queries[i]);
            // cout << arr1[i] << " ";
        }
        // cout << endl;
        
        for ( int i = 0; i < nw; ++i ) { 
            arr2[i] = getVal(words[i]);
            // cout << arr2[i] << " ";
        }
        // cout << endl;
        sort(arr2, arr2 + nw);
        vector<int> res;
        for ( int i = 0; i < nq; ++i ) {
            int cnt = 0;
            for ( int j = nw - 1; j >=0 ; --j ) {
                if ( arr2[j] <= arr1[i] ) break;
                ++cnt;
            }
            res.push_back(cnt);
        }
        return res;
    }
    
    int getVal(string& s) {
        char c = s[0];
        int freq = 1;
        for ( int i = 1; i < s.length(); ++i ) {
            if ( s[i] > c ) continue;
            if ( s[i] == c ) freq++;
            else {
                c = s[i];
                freq = 1;
            }
        }
        return freq;
    }
};
