class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> cc(11, 0);
        vector<int> wc;
        for (auto & w : words) wc.push_back(fun(w));
        for (int i=0; i<wc.size(); ++i) {
            cc[wc[i]]++;
        }
        
        int n = queries.size();

        vector<int> res(n, 0);
        for(int i=0; i<n; ++i) {
            int funq = fun(queries[i]);
            for (int j = funq+1; j<=10; ++j) res[i] += cc[j];

        }
        return res;
    }
    
    int fun(string word) {
        char c = *min_element(word.begin(), word.end());
        int res = 0;
        for (int i=0; i<word.size(); ++i) if (word[i]==c) res++;
        return res;
    }
};
