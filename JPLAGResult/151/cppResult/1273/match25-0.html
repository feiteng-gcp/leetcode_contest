class Solution {
public:
    int helper(std::string & s){
        std::vector<int> lut(26, 0);
        int minv = 128;
        for(auto each : s){
            if(each < minv){
                minv = each;
            }
            ++lut[each - 'a'];
        }
        return lut[minv - 'a'];
    }
    
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        std::vector<int> lut;
        for(auto & each : words){
            lut.push_back(helper(each));
        }
        std::vector<int> r;
        for(auto & each : queries){
            int cnt = 0;
            int tmp = helper(each);
            for(auto each_w : lut){
                if(tmp < each_w) ++cnt;
            }
            r.push_back(cnt);
        }
        return r;
    }
    
};
