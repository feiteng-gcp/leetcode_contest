class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> ans;
        for (auto s: queries) {
            int t = 0;
            int fs = f(s);
            for (auto w: words)
                t += (fs < f(w));
            ans.push_back(t);
        }
        return ans;
    }
    int f(const string &A) {
        char minC = 'z';
        int L = A.length();
        for (int i = 0; i < L; ++i) {
            minC = min(minC, A[i]);
        }
        int ans = 0;
        for (int i = 0; i < L; ++i) {
            ans += (A[i] == minC);
        }
        return ans;
    }
};
