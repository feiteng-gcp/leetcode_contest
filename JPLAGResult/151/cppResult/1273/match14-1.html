class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) 
    {
        int sz1 = queries.size();
        int sz2 = words.size();
        vector<int> fwords(sz2, 0);

        vector<int> fquery(sz1, 0);
        for(int i = 0; i != sz2; ++i)
        {
            fwords[i] = f(words[i]);
        }
        for(int i = 0; i != sz1; ++i)
        {
            fquery[i] = f(queries[i]);
        }
        sort(fwords.begin(), fwords.end(), [](const int &l, const int &r){return l > r;});
        vector<int> ans(sz1, 0);
        for(int i = 0; i != sz1; ++i)
        {
            int cnt(0);
            for(int j = 0; j != sz2; ++j)
            {
                if(fwords[j] > fquery[i])
                    ++cnt;
                else
                    break;
            }
            ans[i] = cnt;
        }
        return ans;
    }
    
    int f(const string &s)
    {
        int cnt[26];
        for(int i = 0; i != 26; ++i)
            cnt[i] = 0;
        for(const auto c: s)

        {
            ++cnt[c-'a'];
        }
        for(int i =0; i != 26; ++i)
        {
            if(cnt[i] != 0)
                return cnt[i];
        }
        return  0;
    }
};
