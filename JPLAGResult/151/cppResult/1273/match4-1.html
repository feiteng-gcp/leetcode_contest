class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector <int> qs(queries.size(), 0); 
        vector <int> ws(words.size(), 0); 
        map <char, int> m;
        
        for (int i = 0; i < queries.size(); i++) {
            m.clear();
            for (char c : queries[i]) m[c]++;

            for (int j = 0; j < 26; j++) {
                if (m['a' + j] > 0) { 
                    qs[i] = m['a' + j];
                    break;
                }
            }
        }
        
        for (int i = 0; i < words.size(); i++) {
            m.clear();
            for (char c : words[i]) m[c]++;

            for (int j = 0; j < 26; j++) {
                if (m['a' + j] > 0) { 
                    ws[i] = m['a' + j];
                    break;
                }
            }
        }
        
        vector <int> out(queries.size(), 0);
        
        for (int i = 0; i < out.size(); i++) {
            for (int j = 0; j < ws.size(); j++) {
                if (qs[i] < ws[j]) out[i]++;
            }    
        }
        return out;
    }
};
