class Solution {
public:
	int Judge[2001];
	vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
		vector<int>room;
		for (int i = 0; i < words.size(); ++i)
		{
			int count = 0; 
			char min = 'z';
			for (int j = 0; j < words[i].size(); ++j)
			{
                //cout << words[i][j] <<endl;
				if (words[i][j] < min)
				{
					min = words[i][j];
					count = 1;
				}
				else if (words[i][j] == min)
				{
					count++;
				}
			}
			Judge[i] = count;
            //cout << i << ' ' << count <<endl;
		}
		for (int i = 0; i < queries.size(); ++i)
		{
			int count = 0;
			char min = 'z';
			for (int j = 0; j < queries[i].size(); ++j)
			{
				if (queries[i][j] < min)
				{
					min = queries[i][j];
					count = 1;
				}
				else if (queries[i][j] == min)
				{
					count++;
				}
                //cout << count <<endl;
			}
			int num = 0;
			for (int j = 0; j < words.size(); ++j)
			{
				//int num = 0;
				if (count < Judge[j])
					num++;
			}
			room.push_back(num);
		}
        return room;
	}
};
