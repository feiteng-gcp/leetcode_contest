class Solution {
public:
	vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
		vector<int>q(queries.size(), 0), result(queries.size(), 0);
		vector<int>w(words.size(), 0);
		for (int i = 0; i < queries.size(); i++)
		{
			int num[26] = { 0 };
			for (int j = 0; j < queries[i].size(); j++) num[queries[i][j] - 'a']++;

			for (int j = 0; j < 26; j++)
			{
				if (num[j])
				{
					q[i] = num[j];
					break;
				}
			}
		}
		for (int i = 0; i < words.size(); i++)
		{
			int num[26] = { 0 };
			for (int j = 0; j < words[i].size(); j++) num[words[i][j] - 'a']++;

			for (int j = 0; j < 26; j++)
			{
				if (num[j])
				{
					w[i] = num[j];
					break;
				}
			}
		}
		for (int i = 0; i < queries.size(); i++)
		{
			int sum = 0;
			for (int j = 0; j < words.size(); j++)
			{
				if (q[i] < w[j]) sum++;
			}
			result[i] = sum;
		}
		return result;
	}
};
