class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
      int qs = queries.size();
      int ws = words.size();
      vector<int> q_value(qs), w_value(ws);
      for(int i = 0; i < qs; ++i)
        q_value[i] = func(queries[i]);
      for(int i = 0; i < ws; ++i)
        w_value[i] = func(words[i]);
      vector<int> ret;
      for(int i = 0; i < qs; ++i) {
        int cnt = 0;
        for(int j = 0; j < ws; ++j) {
          if(q_value[i] < w_value[j])
            cnt++;
        }
        ret.push_back(cnt);
      }

      return ret;
    }
    int func(string& s) {
      vector<int> count(26, 0);
      for(auto& c : s)
        count[c - 'a']++;
      for(int i = 0; i < 26; ++i)
        if(count[i] > 0)
          return count[i];
      return 0;
    }
};
