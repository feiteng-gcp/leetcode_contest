int s[2002];
class Solution {
public:
    int func(string s) {
        int i, n = s.size();
        int t = 1;
        char x = s[0];
        for (i = 1; i < n; i++) {
            if (s[i] < x) {
                x = s[i];
                t = 1;
            } else if (s[i] == x) {
                t++;
            }
        }
        return t;
    }
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        int n = queries.size(), m = words.size();
        memset(s, 0, sizeof(s));
        int i;
        for (i = 0; i < m; i++) {
            s[func(words[i])]++;
        }
        for (i = 1999; i >= 0; i--)
            s[i] += s[i + 1];
        vector<int>res;
        for (i = 0; i < n; i++) {
            res.push_back(s[func(queries[i]) + 1]);
        }
        return res;
    }
};
