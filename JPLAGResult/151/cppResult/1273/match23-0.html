class Solution {
public:
    int fun(string& s) {
        char c = s[0];
        int cnt = 0;
        int n = s.size();
        for (int i = 0; i < n; ++i) {
            if (s[i] < c) c = s[i], cnt = 0;
            if (s[i] == c) ++cnt;
        }
        return cnt;
    }
    vector<int> numSmallerByFrequency(vector<string>& qs, vector<string>& ws) {
        int n = qs.size(), m = ws.size();
        vector<int> ret(n), fws(m);
        for (int i = 0; i < m; ++i) fws[i] = fun(ws[i]);
        sort(fws.begin(), fws.end());
        for (int i = 0; i < n; ++i) {
            int q = fun(qs[i]);
            int l = lower_bound(fws.begin(), fws.end(), q+1) - fws.begin();
            ret[i] = m - l;
        }
        return ret;
    }
};
