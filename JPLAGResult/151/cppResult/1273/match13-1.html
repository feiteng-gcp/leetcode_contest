class Solution {
public:
    int cal(string s){
        vector<int> v(26,0);
        for(int i = 0; i < s.length(); ++i){
            v[s[i]-'a']++;
        }
        for(int i = 0; i < 26; ++i){
            if(v[i]) return v[i];
        }
        return 0;
    }
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        int n = queries.size();
        int m = words.size();

        vector<int> vec(n+m);
        for(int i = 0; i < n; ++i){
            vec[i] = cal(queries[i]);
        }
        for(int i = 0; i < m; ++i){
            vec[i+n] = cal(words[i]);
        }
        vector<int> ans;
        for(int i = 0; i < n; ++i){
            int cnt = 0;
            for(int j = 0; j < m; ++j){
                if(vec[i] < vec[j+n]){
                    cnt++;
                }
            }
            ans.push_back(cnt);
        }
        return ans;
    }
};
