class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& q, vector<string>& ar) {

        auto f = [&](string s){
            map<char,int> hm;
            for(auto i : s) hm[i]++;
            return (*hm.begin()).second;
        };
        
        vector<int> fq(q.size()) , fw(ar.size()) , ans(q.size());
        int n = q.size(); int m = ar.size();
        for(int i =0;i<n ;i++){
            fq[i] = f(q[i]);
        }
        for(int j = 0;j<m;j++){
            fw[j] = f(ar[j]);
        }
    
        


        for(int i =0 ;i<n;i++){
            for(int j = 0;j<m;j++){
                if(fq[i]<fw[j]) ans[i]++;
            }
        }
        return ans;
    }
};
