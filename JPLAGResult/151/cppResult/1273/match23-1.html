class Solution {
public:
    int f(string s) {
        if (s.empty())
            return 0;
        
        vector<int> cnt(128, 0);
        for (auto ch : s)
            cnt[ch]++;
        for (int i = 'a'; i <= 'z'; i++) {
            if (cnt[i] != 0) {
                return cnt[i];
            }

        }
        return 0;
    }
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        int m = queries.size(), n = words.size();
        vector<int> q(m, 0), w(n, 0);
        for (int i = 0; i < m; i++)
            q[i] = f(queries[i]);
        for (int i = 0; i < n; i++) 
            w[i] = f(words[i]);
        
        sort(w.begin(), w.end());
        
        vector<int> res(m, 0);
        for (int i = 0; i < m; i++) {
            int cnt = 0;
            auto it = upper_bound(w.begin(), w.end(), q[i]);
            res[i] = distance(it, w.end());
        }
        return res;
    }
};
