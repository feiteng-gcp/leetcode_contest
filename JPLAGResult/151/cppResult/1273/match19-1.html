int F(string s) {
    int len = s.size();
    int v[26];
    memset(v, 0, sizeof v);

    for (int i = 0; i < len; ++i) v[s[i]-'a']++;
    for (int i = 0; i < 26; ++i) if (v[i]) return v[i];
    return 0;
}

class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        int b[3000];
        int n = queries.size();
        int m = words.size();
        
        for (int i = 0; i < m; ++i) b[i] = F(words[i]);
        sort(b, b+m);
        
        vector<int> ret;
        for (int i = 0; i < n; ++i) {
            int x = F(queries[i]);
            int ans = (b+m) - upper_bound(b, b+m, x);
            ret.push_back(ans);
        }
        return ret;
    }
};
