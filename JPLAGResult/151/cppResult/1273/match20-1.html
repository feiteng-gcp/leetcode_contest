class Solution {
public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> fws;
        for (const string& w : words) fws.push_back(f(w));
        vector<int> ans;
        for (const string& q : queries) {
            int cnt = 0;
            int fq = f(q);
            for (int fw : fws) if (fq < fw) ++cnt;
            ans.push_back(cnt);
        }
        return ans;
    }
    
    int f(const string& s) {
        char min_char = 'z';
        for (const char& c : s) {
            min_char = min(min_char, c);
        }
        int cnt = 0;
        for (const char& c : s) if (c == min_char) ++cnt;
        return cnt;
    }
};
