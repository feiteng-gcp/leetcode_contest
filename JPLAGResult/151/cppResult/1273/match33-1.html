int f(string& s) {
    map<char, int> m;
    for (auto i : s)
        m[i]++;
    return m.begin()->second;
}

class Solution {

public:
    vector<int> numSmallerByFrequency(vector<string>& queries, vector<string>& words) {
        vector<int> v;
        for (auto& i : words) {
            v.push_back(f(i));
        }
        vector<int> ans;
        for (auto& i : queries) {
            int tmp = f(i);
            ans.push_back(0);
            for (auto j : v)
                if (j > tmp)
                        ans.back()++;
        }
        return ans;
    }
};
