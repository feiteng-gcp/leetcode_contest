struct Trans {
    string name;
    int time;
    int amount;
    string city;
    bool invalid = false;
    Trans(string& s) {
        int i = 0, n = s.size(), st = i;
        while(s[i] != ',') i++;
        name = s.substr(st, i - st);
        st = ++i;
        while(s[i] != ',') i++;
        time = stoi(s.substr(st, i - st));
        st = ++i;
        while(s[i] != ',') i++;
        amount = stoi(s.substr(st, i - st));
        city = s.substr(i+1);
    }
};

class Solution {
public:
    vector<string> invalidTransactions(vector<string>& transs) {
        vector<Trans> tmp;
        vector<string> res;
        int n = transs.size();
        for(int i = 0; i < n; i++) {
            Trans t = Trans(transs[i]);
            if(t.amount > 1000) t.invalid = true;
            for(int j = 0; j < i; j++) {
                auto& t1 = tmp[j];
                if(t1.name == t.name && t1.city != t.city && abs(t1.time - t.time) <= 60) {
                    t.invalid = true;
                    if(!t1.invalid) {
                        t1.invalid = true;
                        res.push_back(transs[j]);
                    }
                }
            }
            if(t.invalid) res.push_back(transs[i]);
            tmp.push_back(t);
        }
        return res;
    }
};
