class Node{
public:
    string name;
    int time;
    int money;
    string city;
    Node(string _name,int _time,int _money,string _city){
        name = _name;
        time = _time;
        money = _money;
        city = _city;
    }
    Node(){
    }
    void show(){
        cout << name << " " << time << " " << money << " " << city << endl;
    }
};
class Solution {
public:
    int ToInt(string str){
        int res = 0;
        for(int i=0;i<str.length();++i)
            res = res*10 + (str[i]-'0');
        return res;
    }
    Node GetNode(string str){
        Node res;
        string temp = "";
        int Count = 0;
        for(int i=0;i<str.length();++i){
            if(str[i] == ','){
                Count++;
                if(Count == 1)
                    res.name = temp;
                else if(Count == 2)
                    res.time = ToInt(temp);
                else if(Count == 3)
                    res.money = ToInt(temp);
                temp = "";
            }
            else
                temp += str[i];
        }
        res.city = temp;
        return res;
    }
    vector<string> invalidTransactions(vector<string>& transactions) {
        for(string str:transactions){
            vec.push_back(GetNode(str));
        }
        vector<string> res;
        set<int> st;
        for(int i=0;i<vec.size();++i){
            Node &now = vec[i];
            if(now.money > 1000)
                st.insert(i);
            bool have = false;
            for(int j=i-1;j>=0;--j){
                Node &v = vec[j];
                if(now.name == v.name && now.city != v.city && abs(now.time-v.time) <= 60){
                    have = true;
                    st.insert(j);
                }
            }
            if(have)
                st.insert(i);
        }
        for(int v:st)
            res.push_back(transactions[v]);
        return res;
    }
private:
    vector<Node> vec;
};
