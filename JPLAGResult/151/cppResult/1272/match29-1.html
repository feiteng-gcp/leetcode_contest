struct Trans {
    string name;
    int time;
    int amount;
    string city;
    Trans (const string& _name, int _time, int _amount, const string& _city) : name(_name), time(_time), amount(_amount), city(_city) {}
    Trans(const string& s) {
        time = 0;
        amount = 0;
        
        int i = 0;
        while (s[i] != ',') {

            name += s[i++];
        }
        i++;
        while (s[i] != ',') {
            time = time * 10 + s[i++] - '0';
        }
        i++;
        while (s[i] != ',') {
            amount = amount * 10 + s[i++] - '0';
        }
        i++;
        while (i < s.length()) {
            city += s[i++];
        }
    }
    
    string toString() {
        return name + "," + to_string(time) + "," + to_string(amount) + "," + city;
    }
};
class Solution {
public:
    vector<string> invalidTransactions(vector<string>& transactions) {
        vector<Trans> trans;
        unordered_map<string, vector<Trans>> nameToTrans;
        
        unordered_set<string> result;
        
        for (auto& s : transactions) {
            auto t = Trans(s);
            
            if (t.amount > 1000) {
                result.insert(t.toString());
            }
            
            for (auto& p : nameToTrans[t.name]) {
                if (abs(t.time - p.time) <= 60 && t.city != p.city) {
                    result.insert(p.toString());
                    result.insert(t.toString());
                }
            }
            nameToTrans[t.name].push_back(t);
        }
        
        vector<string> res;
        for (auto& s : result) {
            res.push_back(s);
        }
        return res;
    }
};
