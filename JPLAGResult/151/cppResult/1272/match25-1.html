class Solution {
public:
    vector<string> split(string &s) {
        int i = 0;
        vector<string> ans;
        int n = s.size();
        while (i < n) {
            int j = s.find(',', i);
            if (j == string::npos) {
                j = n;
            }
            ans.push_back(s.substr(i, j-i));
            i = j+1;
        }
        return (ans);
    }
    vector<string> invalidTransactions(vector<string>& t) {
        int n = t.size();
        vector<string> ans;
        vector<vector<string>> tok(n);
        for (int i = 0; i < n; i++) {
            tok[i] = split(t[i]);

        }
        for (int i = 0; i < n; i++) {
            if (stoi(tok[i][2]) > 1000) {
                ans.push_back(t[i]);
                continue;
            }
            int tm1 = stoi(tok[i][1]);
            for (int j = 0; j < n; j++) {
                if (j == i) {
                    continue;
                }
                int tm2 = stoi(tok[j][1]);
                if (abs(tm1 - tm2) <= 60 && tok[i][0] == tok[j][0] && tok[i][3] != tok[j][3]) {
                    ans.push_back(t[i]);
                    break;
                }
            }
        }
        return (ans);
    }
};
