class Solution {
public:
    vector<string> invalidTransactions(vector<string>& transactions) {
        vector<vector<string>> toks(transactions.size());
        unordered_set<string> results;
        for (int i = 0; i < transactions.size(); ++i) {
            tokenize(transactions[i], toks[i]);
            if (stoi(toks[i][2]) > 1000) {
                results.emplace(transactions[i]);
            }
            for (int j = 0; j < i; ++j) {
                if (toks[j][0] == toks[i][0] && toks[j][3] != toks[i][3] && abs(stoi(toks[j][1]) - stoi(toks[i][1])) <= 60) {
                    results.emplace(transactions[j]);
                    results.emplace(transactions[i]);
                }
            }
        }
        return vector<string>(results.begin(), results.end());
    }
private:
    void tokenize(const string &transaction, vector<string> &toks) const {
        for (int i = 0, j = 0; j <= transaction.size(); ++j) {
            if (j == transaction.size() || transaction[j] == ',') {
                toks.emplace_back(transaction.substr(i, j - i));
                i = j + 1;
            }
        }
    }
};
