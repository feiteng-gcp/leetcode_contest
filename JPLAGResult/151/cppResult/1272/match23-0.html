class Solution {

	struct node
	{
		int time;
		int amount;
		string name;
		string city;
	};
public:

	node gao(string s)
	{
		int num = 0;
		string tmp;

		node ans;
		for (int i = 0; i < s.size(); i++)
		{
			if (s[i] == ',')
			{
				if (num == 0)
				{
					ans.name = tmp;
				}
				else if (num == 1)
				{
					ans.time = stoi(tmp.c_str());
				}
				else if (num == 2)
				{
					ans.amount = stoi(tmp.c_str());
				}
				tmp = "";
				num++;
			}
			else
				tmp += s[i];
		}
		ans.city = tmp;
		return ans;
	}
	vector<string> invalidTransactions(vector<string>& transactions) {
		vector<string> ans;

		int n = transactions.size();

		vector<node> transaction;
		for (int i = 0; i < n; i++)
		{
			transaction.push_back(gao(transactions[i]));
		}
		for (int i = 0; i<n; i++)
		{
			//{name},{time},{amount},{city}
			if (transaction[i].amount > 1000)
			{
				ans.push_back(transactions[i]);
				continue;
			}
			for (int j = 0; j<n; j++)
			{
				if (i != j && transaction[i].city != transaction[j].city && transaction[i].name == transaction[j].name  && abs(transaction[i].time - transaction[j].time) <= 60)
				{
					ans.push_back(transactions[i]);
					break;
				}
			}
		}
		return ans;
	}

};
