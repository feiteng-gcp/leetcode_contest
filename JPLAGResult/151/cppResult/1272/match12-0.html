class Solution {
public:
    vector<string> invalidTransactions(vector<string>& transactions) {
        int N = transactions.size();
        vector<Transaction> v;
        for (int i = 0; i < N; ++i) {
            const char *s = transactions[i].c_str();
            v.push_back(Transaction());
            char name[15];
            char city[15];
            sscanf(s, "%[^,],%d,%d,%[^,]", name, &v[i].time, &v[i].amount, city);
            v[i].name = string(name);
            v[i].city = string(city);
        }
        vector<string> ans;
        for (int i = 0; i < N; ++i) {
            bool ok = true;
            if (v[i].amount > 1000)
                ok = false;
            for (int j = 0; j < N && ok; ++j) {
                if (j == i)
                    continue;
                if (abs(v[i].time - v[j].time) <= 60 && v[i].name == v[j].name && v[i].city != v[j].city)
                    ok = false;
            }
            if (!ok)
                ans.push_back(transactions[i]);
        }
        return ans;
    }
    struct Transaction {
        string name;
        int time;
        int amount;
        string city;
    };
};
