class Solution {
public:
  struct Tran {
    string name;
    int time;
    int amount;
    string city;
    bool isValid;
  };
  vector<string> invalidTransactions(vector<string>& transactions) {
    vector<Tran> A;
    for (auto s: transactions) {
      for (auto &c: s) if (c == ',') c=' ';
      stringstream ss(s);
      Tran t; ss >> t.name >> t.time >> t.amount >> t.city;
      t.isValid = true;
      //cout << t.name << " " << t.time << " " << t.amount << " " << t.city << endl;
      A.push_back(t);
    }
    for (int i = 0; i < A.size(); i++) if (A[i].amount > 1000) A[i].isValid = false;
    for (int i = 0; i < A.size(); i++) {
      for (int j = i + 1; j < A.size(); j++) {
        if (A[i].name == A[j].name && A[i].city != A[j].city && abs(A[i].time - A[j].time) <= 60) {
          A[i].isValid = A[j].isValid = false;
        }
      }
    }
    vector<string> ans;
    for (int i = 0; i < A.size(); i++) if (!A[i].isValid) ans.push_back(transactions[i]);
    return ans;
  }
};
