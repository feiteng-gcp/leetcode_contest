class Solution {
public:
    vector <string> get_segs(string s) {
        vector <string> out;
        int i = 0;
        int prev = 0;
        while (i < s.size()) {
            if (s[i] == ',') {
                out.push_back(s.substr(prev,i-prev));
                prev = i+1;
            }
            i++;
        }
        out.push_back(s.substr(prev,i-prev));
        return out;
        
    }
    vector<string> invalidTransactions(vector<string>& t) {
        vector <vector <string> > segs;
        vector <string> out;

        for (int i = 0; i < t.size(); i++) {
            segs.push_back(get_segs(t[i]));            
        }
        
        for (int i = 0; i < segs.size(); i++) {
            if (stoi(segs[i][2]) > 1000) {
                out.push_back(t[i]);
                continue;                
            }
            
            for (int j = 0; j < segs.size(); j++) {
                if (j == i) continue;
                // cout << segs[i][0] << " " << segs[j][0] << ", " << segs[i][1] << " " << segs[j][1] << ", "  << segs[i][3] << " " << segs[j][3] << endl;
                if (segs[i][0] == segs[j][0] && abs(stoi(segs[i][1]) - stoi(segs[j][1])) <= 60 && segs[i][3] != segs[j][3]) {
                    out.push_back(t[i]);
                    break;
                }
            }
        }
         return out;
    }
};
