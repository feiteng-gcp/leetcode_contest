class Solution {
public:
    struct tran{
        string name, city, transaction;
        int time, amount;
        //  tran(string n, int t, int a, string c, string tr): name(n), time(t), amount(a), city(c), transaction(tr){}
    };
    
    vector<string> invalidTransactions(vector<string>& transactions) {
        vector<string> res;
        if (transactions.empty()) return res;
        tran trans[transactions.size()]; 
        for (int i = 0; i < transactions.size(); i++)
        {
            string n = "";
            string c= "";
            int t = 0, a = 0;
            int j = 0;
            while (j < transactions[i].size())
            {
                if (transactions[i][j] != ',')
                    n += transactions[i][j];
                else break;
                j++;
            }
            j++;
            while (j < transactions[i].size() && transactions[i][j] != ',')
            {
                t = t*10+(transactions[i][j]-'0');
                j++;
            }
            j++;
            while (j < transactions[i].size() && transactions[i][j] != ',')
            {
                a = a*10+(transactions[i][j]-'0');
                j++;
            }
            j++;
            while (j < transactions[i].size() && transactions[i][j] != ',')
            {
                c += transactions[i][j];
                j++;
            }
            //tran tmp = new tran(n,t,a,c,transactions[i]);
            //trans.push_back(tmp);
            trans[i].name = n;
            trans[i].time = t;
            trans[i].amount = a;
            trans[i].city = c;
            trans[i].transaction = transactions[i];
        }
        for (int i = 0; i < transactions.size(); i++)
        {
            int flag = true;
            if (trans[i].amount > 1000)
            {
                res.push_back(trans[i].transaction);
                continue;
            }
            for (int j = 0; j < transactions.size(); j++)
                if (i == j) continue;
                else
                {
                    if (trans[i].name == trans[j].name && trans[i].city != trans[j].city && abs(trans[i].time-trans[j].time) <= 60)
                    {
                        flag = false;
                        break;
                    }
                }
            if (!flag)
                res.push_back(trans[i].transaction);
        }
        return res;
    }
};
