class Solution {
public:
    vector<string> invalidTransactions(vector<string>& transactions) {
        vector<string> name, city;
        vector<int> time, amount;
        unordered_map<string, vector<int>> dict;
        for (int i = 0; i < transactions.size(); ++i) {
            int pos1 = -1, pos2 = -1, pos3 = -1;
            for (int j = 0; j < transactions[i].size(); ++j) {
                if (transactions[i][j] == ',') {
                    if (pos1 == -1) pos1 = j;
                    else if (pos2 == -1) pos2 = j;
                    else pos3 = j;
                }
            }
            name.push_back(transactions[i].substr(0, pos1));
            time.push_back(stoi(transactions[i].substr(pos1 + 1, pos2 - pos1 - 1)));
            amount.push_back(stoi(transactions[i].substr(pos2 + 1, pos3 - pos2 - 1)));
            city.push_back(transactions[i].substr(pos3 + 1));
            dict[name.back()].push_back(i);

        }
        vector<string> ans;
        for (int i = 0; i < transactions.size(); ++i) {
            if (amount[i] > 1000) {
                ans.push_back(transactions[i]);
                continue;
            }
            for (int j = 0; j < dict[name[i]].size(); ++j) {
                const int k = dict[name[i]][j];
                if (i == k) continue;
                if (abs(time[i] - time[k]) <= 60 && city[i] != city[k]) {
                    ans.push_back(transactions[i]);
                    break;
                }
            }
        }
        return ans;
    }
};
