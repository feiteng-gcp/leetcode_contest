struct X {
  string name;
  int time;
  int amount;
  string city;
}s[1010];

class Solution {
private:
  int cnt;
  
  void work(string& x) {
    int i;
    
    s[cnt].name = "";
    s[cnt].time = 0;
    s[cnt].amount = 0;
    s[cnt].city = "";
    
    for(i = 0; ; i ++) {
      if(x[i] == ',') {i ++; break;}
      s[cnt].name = s[cnt].name + x[i];
    }
    
    for(; ; i ++) {
      if(x[i] == ',') {i ++; break;}
      s[cnt].time = s[cnt].time * 10 + x[i] - '0';
    }
    

    for(; ; i ++) {
      if(x[i] == ',') {i ++; break;}
      s[cnt].amount = s[cnt].amount * 10 + x[i] - '0';
    }
    

    for(; i < x.length(); i ++) {
      if(x[i] == ',') {i ++; break;}
      s[cnt].city = s[cnt].city + x[i];
    }
    
  
    cnt ++;
  }
    
public:
    vector<string> invalidTransactions(vector<string>& transactions) {
      
      cnt = 0;
      
      
      
      
      for(int i = 0; i < transactions.size(); i ++) {
        work(transactions[i]);
      
      }
      
      vector<string> ans;
      for(int i = 0; i < cnt; i ++) {
        if(s[i].amount > 1000) {
          ans.push_back(transactions[i]);
          continue;
        }
        
        int fail = 0;
        
        for(int j = 0; j < cnt; j ++) {
          if(s[j].name == s[i].name && s[j].city != s[i].city && abs(s[j].time - s[i].time) <= 60) {
            fail = 1;
            break;
          }
        }
        
        if(fail) {
          ans.push_back(transactions[i]);
        }
      }
      
      return ans;
    }
};
