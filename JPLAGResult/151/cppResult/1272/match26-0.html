class Solution {
    struct Trans {
        string name;
        int time;
        int amount;
        string city;
        int index;
    };
public:
    vector<string> invalidTransactions(vector<string>& transactions) {
        map<string, vector<Trans>> m;
        set<string> ans;
        int idx = 0;
        for (auto& trans : transactions) {
            auto t = from_str(trans, idx++);
            if (t.amount > 1000) {
                ans.insert(trans);
            }
            for (auto& i : m[t.name]) {
                if (abs(i.time - t.time) <= 60 && i.city != t.city) {
                    ans.insert(transactions[i.index]);
                    ans.insert(trans);
                } 
            }
            m[t.name].push_back(t);
        }
        return vector<string>(ans.begin(), ans.end());
    }
    
    Trans from_str(const string& s, int idx) {
        Trans t;
        t.index = idx;
        stringstream ss(s);
        string item;
        getline(ss, item, ',');
        t.name = item;
        getline(ss, item, ',');
        t.time = stoi(item);
        getline(ss, item, ',');
        t.amount = stoi(item);
        getline(ss, item, ',');
        t.city = item;
        // printf("%s %d %d %s\n", t.name.c_str(), t.time, t.amount, t.city.c_str());
        return t;
    }
    
};
