class Solution {
public:
    vector<string> invalidTransactions(vector<string>& transactions_) {
        vector< tuple<string, int, int, string, string> > transactions;
        for( auto t : transactions_){
            string w = t;
            for( auto &c : t ) if( c == ',') c = ' ';
            stringstream ss(t);
            string a;
            int b,c;
            string d;
            ss >> a >> b >> c >> d;
            transactions.push_back({a, b, c, d, w});
        }
        vector<string> res;
        map<string, int> value;

        for( auto t : transactions ){
            
            bool chk = true;

            for( auto u : transactions ){
                if( get<3>(t) != get<3>(u) && get<0>(t) == get<0>(u) && abs(get<1>(t)-get<1>(u)) <= 60){
                    chk = false;
                }
            }
            if( !(get<2>(t) <= 1000 && chk)  ){
                res.push_back(get<4>(t));
            }
        }
        return res;
    }
    
    
};
