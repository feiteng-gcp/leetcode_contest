class Solution {
public:
    vector<string> invalidTransactions(vector<string>& transactions) {
      vector<string> ret;
      int size = transactions.size();
      vector<vector<string>> splits;
      for(int i = 0; i < size; ++i) {
        splits.push_back(parse(transactions[i]));
      }
      
      for(int i = 0; i < size; ++i) {
        if(stoi(splits[i][2]) > 1000) {
          ret.push_back(transactions[i]);
          continue;
        }
        int flag = 0;
        for(int j = 0; j < size; ++j) {
          if(j == i) continue;
          if(splits[i][0] == splits[j][0] && splits[i][3] != splits[j][3] && abs(stoi(splits[i][1]) - stoi(splits[j][1])) <= 60) {
            flag = 1;
            break;
          }
        }
        if(flag == 1)
          ret.push_back(transactions[i]);
      }
      
      return ret;
    }
  
    vector<string> parse(string& s) {
      vector<string> ret;
      int start = 0, end =0;
      int size = s.size();
      while(end <= size) {
        while(end <= size && s[end] != ',')
          end++;
        ret.push_back(s.substr(start, end - start));
        start = end + 1;
        end = start;
      }
      return ret;
    }
};
