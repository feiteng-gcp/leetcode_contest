class Solution {
public:
    vector<string> invalidTransactions(vector<string>& transactions) {
        std::unordered_map<std::string, std::vector<std::pair<std::pair<std::pair<int, int>, std::string>, std::string>>> lut;
        std::vector<std::string> r;
        for(auto & each : transactions){
            int pos = 0, next = 0;
            std::string name;
            int time, amount;
            next = each.find(",", 0);
            name = each.substr(0, next);
            pos = next + 1;
            next = each.find(",", pos);
            time = std::stoi(each.substr(pos, next - pos));
            pos = next + 1;
            next = each.find(",", pos);
            amount = std::stoi(each.substr(pos, next - pos));
            pos = next + 1;
            lut[name].push_back({{{time, amount}, each.substr(pos, each.size() - pos)}, each});
        }
        for(auto & each : lut){
            auto & vec = each.second;
            std::vector<bool> invalids(vec.size(), true);
            for(int i = 0; i < vec.size(); ++i){
                if(vec[i].first.first.second > 1000){
                    invalids[i] = false;
                }
                for(int j = 0; j < vec.size(); ++j){
                    if(i == j) continue;
                    if(std::abs(vec[i].first.first.first - vec[j].first.first.first) <= 60 && vec[i].first.second != vec[j].first.second){
                        invalids[i] = false;
                        invalids[j] = false;
                    }
                }
            }
            for(int i = 0; i < vec.size(); ++i){
                if(!invalids[i]) r.push_back(vec[i].second);
            }
        }
        return r;
    }
};
