class Solution {
public:
    vector<string> invalidTransactions(vector<string>& transactions) {
        unordered_set<string> invalid;
        unordered_map<string, unordered_map<int, vector<string>>> mp;
        for (auto tran : transactions) {
            int i = 0;
            int j = tran.find(',');
            string name = tran.substr(i, j-i);
            i = j + 1;
            j = tran.find(',', i);
            int minute = stoi(tran.substr(i, j - i));
            i = j + 1;
            j = tran.find(',', i);
            int amount = stoi(tran.substr(i, j - i));

            i = j + 1;
            string city = tran.substr(i);
            
            if (amount > 1000) {
                invalid.insert(tran);
            }
            for (auto m : mp[name]) {
                if (abs(m.first - minute) <= 60 && m.second[0] != city) {
                    invalid.insert(tran);
                    invalid.insert(name + ',' + to_string(m.first) + ',' + m.second[1] + ',' + m.second[0]);
                }
            }
            mp[name][minute] = { city, to_string(amount) };
        }
        
        return vector<string> (invalid.begin(), invalid.end());
    }
};
