class Solution {
public:
    typedef struct transS
    {
        string name;
        string city;
        int amount;
        int time;
    }trans;
    
    vector<string> invalidTransactions(vector<string>& transactions) 
    {
        // bf
        int n = transactions.size();
        vector<trans> t(n);
        trans temp;
        for(int i = 0; i < n; ++i)
        {
            string div = ",";
            auto t1 = transactions[i].find(div);
            temp.name = transactions[i].substr(0, t1);

            auto t2 = transactions[i].find(div, t1 + 1);
            //cout << transactions[i].substr(t1 + 1, t2 - t1 - 1) << endl;
            temp.time = stoi(transactions[i].substr(t1 + 1, t2 - t1 - 1));
            //cout << temp.time << endl;
            t1 = t2;
            t2 = transactions[i].find(div, t1 + 1);
            temp.amount = stoi(transactions[i].substr(t1 + 1, t2 - t1 - 1));
            temp.city = transactions[i].substr(t2 + 1, transactions[i].size() - t2 - 1);
            t[i] = temp;
        }
        
        vector<string> ans;
        for(int i = 0; i < n; ++i)
        {
            if(t[i].amount > 1000)
            {
                ans.push_back(transactions[i]);
                continue;
            }
                
            for(int j = 0; j < n; ++j)
            {
                if(i == j)
                    continue;
                
                if(t[i].name == t[j].name && t[i].city != t[j].city && abs(t[i].time - t[j].time) <= 60)
                {
                    ans.push_back(transactions[i]);
                    break;
                }
            }
        }
        
        return ans;
    }
};
