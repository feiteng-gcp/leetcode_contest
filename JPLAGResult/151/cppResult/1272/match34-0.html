class Solution {
public:
    struct Transaction {
        string city;
        int time;
        string full_name;
        
        Transaction(string c, int t, string n) : city(c), time(t), full_name(n) {};
    };
    
    vector<string> invalidTransactions(vector<string>& transactions) {
        unordered_map<string, vector<Transaction>> map = {};
        unordered_set<string> res_set = {};
        for (string t : transactions) {
            vector<string> parts = {};
            parse(t, parts);
            string name = parts[0], city = parts[3];
            int time = stoi(parts[1]), amount = stoi(parts[2]);
            Transaction trans(city, time, t);
            bool valid = true;
            if (amount > 1000) {
                valid = false;
            }
            for (Transaction& tr : map[name]) {
                if (tr.city != city && abs(tr.time - time) <= 60) {
                    valid = false;
                    res_set.insert(tr.full_name);
                }
            }
            if (!valid) {
                res_set.insert(t);
            }
            map[name].push_back(trans);
        }
        vector<string> res(res_set.begin(), res_set.end());
        return res;
    }
    
private:
    void parse(string transaction, vector<string>& parts) {
        char str[transaction.size() + 1];
        strcpy(str, transaction.c_str());
        char* token = strtok(str, ",");
        while (token != NULL) {
            parts.push_back(token);
            token = strtok(NULL, ",");
        }
    }
};
