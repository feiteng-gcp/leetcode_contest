class Solution {
public:
    class Node{
        public:
        string s;
        string name;
        int m;
        int val;
        string city;
        Node(string s){
            this->s = s;
            int pre = 0;
            int cnt = 0;
            for(int i = 0; i < s.length(); ++i){
                if(s[i] == ',' && cnt == 0){
                    name = s.substr(pre, i);
                    pre = i+1;
                    cnt++;
                }
                else if(s[i] == ',' && cnt == 1){
                    string t = s.substr(pre, i-pre);
                    int sum = 0;
                    for(int j = 0; j < t.length(); ++j){
                        sum = sum * 10 + t[j] - '0';
                    }
                    m = sum;
                    pre = i+1;
                    cnt++;
                }
                else  if(s[i] == ',' && cnt == 2){
                    string t = s.substr(pre, i-pre);
                    int sum = 0;
                    for(int j = 0; j < t.length(); ++j){
                        sum = sum * 10 + t[j] - '0';
                    }
                    val = sum;
                    pre = i+1;
                    cnt++;
                }
            }
            city = s.substr(pre);
        }
    };
    vector<string> invalidTransactions(vector<string>& transactions) {
        vector<Node> vec;
        for(int i = 0; i < transactions.size(); ++i){
            string s = transactions[i];
            vec.push_back(Node(s));
        }
        vector<string> ans;
        for(int i = 0; i < vec.size(); ++i){
          //  cout << vec[i].s << " " << vec[i].name << " " << vec[i].city << endl;
            if(vec[i].val > 1000){
                ans.push_back(vec[i].s);
                continue;
            }
            for(int j = 0; j < vec.size(); ++j){
                if(i == j) continue;
                if(vec[i].name == vec[j].name && vec[i].city != vec[j].city && abs(vec[i].m - vec[j].m) <= 60){
                    ans.push_back(vec[i].s);
                    break;
                }
            }
        }
        return ans;
    }
};
