class Solution {
public:
    struct node{
        string name,city;
        int time,amount;
    };
    vector<node> tr;
    int isIn[1010];
    vector<string> answer;
    
    vector<string> invalidTransactions(vector<string>& transactions) {
        memset(isIn, 0, sizeof(isIn));
        tr.clear();
        for (int i=0; i < transactions.size(); ++i){
            string s = transactions[i];
            int p = 0, t = 0;
            node no;
            for (int j = 0; j <s.length(); ++j ){
                if (s[j] != ',') continue;
                if (p == 0){
                    no.name="";
                    for (int k = t; k < j; ++k){
                        no.name +=s[k];
                    }
                    p++;
                    t = j +1;
                    // answer.push_back(no.name); 
                    continue;
                }
                if (p == 1){
                    no.time = 0;
                    for (int k = t; k < j; ++k){
                        no.time += s[k] - '0';
                        no.time *=10;
                    }
                    no.time /=10;
                    p++;
                    
                    t = j+1;
                     continue;
                }
                if (p == 2){
                    
                    no.amount = 0;
                    for (int k = t; k <j; ++k){
                        no.amount += s[k] - '0';
                        no.amount *=10;
                    }
                    no.amount /=10;
                    p++;
                    t = j+1;
                    // answer.push_back(to_string(no.amount));
                     continue;
                }
            }
            no.city="";
            for (int k = t; k <s.length(); ++k)
               no.city +=s[k];
            tr.push_back(no);
//             answer.push_back(no.city);
            
        }
        // return answer;

        for (int i=0; i < tr.size();++i){
            if (tr[i].amount > 1000) {
                isIn[i] = 1;
            }
            for (int j = i+1; j < tr.size(); ++j){
                if (tr[i].name == tr[j].name && tr[i].city != tr[j].city && abs(tr[i].time - tr[j].time)<=60){

                    isIn[i] = 1; isIn[j] = 1;
                    // answer.push_back(to_string(i));
                    // answer.push_back(to_string(j));
                    // return answer;
                }
                
            }
        }
        answer.clear();
        for (int i=0; i < transactions.size(); ++i){
            if (isIn[i]) answer.push_back(transactions[i]);
        }
        return answer;
        
        
    }
};
