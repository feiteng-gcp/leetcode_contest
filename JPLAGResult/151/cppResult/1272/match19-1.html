class Solution {
public:

    struct Transaction {
        Transaction(string name, int time, int amount, string city, string input_str):
            name(name), time(time), amount(amount), city(city), input_str(input_str), valid(true) {}
        string name;
        int time;
        int amount;
        string city;
        string input_str;
        bool valid;
    };
    
    vector<string> invalidTransactions(vector<string>& transactions) {
        std::unordered_map<string, pair<int, string>> M;
        vector<string> ret;
        vector<Transaction> trans;
        for (const auto& transaction: transactions) {
            int pos = transaction.find(",");
            string name = transaction.substr(0, pos);

            string rest = transaction.substr(pos + 1);
            pos = rest.find(",");
            int time_t = std::stoi(rest.substr(0, pos));
            
            rest = rest.substr(pos + 1);
            pos = rest.find(",");
            int amount = std::stoi(rest.substr(0, pos));
            
            rest = rest.substr(pos + 1);
            pos = rest.find(",");
            string city = rest.substr(0, pos);
            trans.push_back(Transaction(name, time_t, amount, city, transaction));
        }
        
        for (int i = 0; i < trans.size(); i++) {
            if (trans[i].amount > 1000) {
                trans[i].valid = false;
            }

            for (int j = i + 1; j < trans.size(); j++) {
                if (trans[i].name == trans[j].name &&
                   abs(trans[i].time - trans[j].time) <= 60 &&
                   trans[i].city != trans[j].city) {
                    trans[i].valid = false;
                    trans[j].valid = false;
                }
            }
        }
        
        for (int i = 0; i < trans.size(); i++)
            if (trans[i].valid == false)
                ret.push_back(trans[i].input_str);
        return ret;      
    }
};
