class Solution {
public:
    vector<string> invalidTransactions(vector<string>& transactions) {
        vector<string> name;
        vector<int> time;
        vector<int> amount;
        vector<string> city;
        for(string s:transactions){
            stringstream ss(s);
            string t;
            vector<string> res;
            while(getline(ss,t,',')){
                res.push_back(t);
            }
            name.push_back(res[0]);
            time.push_back(stoi(res[1]));
            amount.push_back(stoi(res[2]));
            city.push_back(res[3]);
        }
        
        vector<string> ans;

        for(int i=0;i<transactions.size();i++){
            bool flag=false;
            if(amount[i]>1000) flag=true;
            for(int j=0;j<transactions.size();j++){
                if(flag) break;
                if(i==j) continue;
                if(abs(time[i]-time[j])<=60&&city[i]!=city[j]&&name[i]==name[j]) flag=true;
            }
            
            if(flag) ans.push_back(transactions[i]);
        }
        return ans;
    }
};
