class Solution {
public:
    vector<string> invalidTransactions(vector<string>& tr) {
        int n = tr.size();
        vector<string> res, name(n), city(n);
        vector<int> time(n), amount(n);
        
        for (int i = 0; i < n; ++i) {
            string s = tr[i];
            int p = s.find(',');
            name[i] = s.substr(0, p);
            int q = s.find(',', p+1);
            
            string tmp = s.substr(p+1, q-p-1);
            time[i] = stoi(tmp);
            
            p = q;
            q = s.find(',', p+1);
            amount[i] = stoi(s.substr(p+1, q-p-1));
            city[i] = s.substr(q+1);
        }
        
        for (int i = 0; i < n; ++i) {
            if (amount[i] > 1000)
                res.push_back(tr[i]);
            else {
                for (int j = 0; j < n; ++j)
                    if (j != i && name[j] == name[i] && abs(time[i] - time[j]) <= 60 && city[i] != city[j]) {
                        res.push_back(tr[i]);
                        break;
                    }
            }
        }
        
        return res;
    }
};
