/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        vector<int> nums;
        auto* p=head;
        while(p!=NULL){
            nums.push_back(p->val);
            p=p->next;
        };
        
        while(true){
            unordered_map<int,int> dd;
            dd[0]=0;
            int s=0;
            bool flag=false;
            for(int i=0;i<nums.size();i++){
                s+=nums[i];
                if(dd.find(s)!=dd.end()){
                    int last=dd[s];
                    flag=true;
                    vector<int> new_nums;
                    for(int j=0;j<last;j++){
                        new_nums.push_back(nums[j]);
                    }
                    for(int j=i+1;j<nums.size();j++){
                        new_nums.push_back(nums[j]);
                    }
                    nums=new_nums;
                    break;
                }else dd[s]=i+1;
            }
            
            if(!flag) break;
        }
        
        if(nums.size()==0) return NULL;
        ListNode* hh=new ListNode(nums[0]);
        p=hh;
        for(int i=1;i<nums.size();i++){
            p->next=new ListNode(nums[i]);
            p=p->next;
        }
        return hh;
    }
};
