class Solution {
public:
	ListNode* removeZeroSumSublists(ListNode* head) {

		vector<int> tmp;
		ListNode* now = head;
		while (now != nullptr)
		{
			tmp.push_back(now->val);
			now = now->next;
		}
		while (1)
		{
			if (tmp.size() == 0)
				break;
			vector<int> num = tmp;
			vector<int> sum(num.size());
			map<int, int>mp;


			bool flag = false;
			for (int i = 0; i<num.size(); i++)
			{
				if (i == 0)
					sum[i] = num[i];
				else
					sum[i] = sum[i - 1] + num[i];
				if (sum[i] == 0)
				{
					tmp.clear();
					for (int j = i + 1; j<num.size(); j++)
						tmp.push_back(num[j]);
					flag = true;
					break;
				}
				if (mp.find(sum[i]) != mp.end())
				{
					tmp.clear();
					for (int j = 0; j <= mp[sum[i]]; j++)
						tmp.push_back(num[j]);
					for (int j = i + 1; j<num.size(); j++)
						tmp.push_back(num[j]);
					flag = true;
					break;
				}

				mp[sum[i]] = i;
			}

			if (flag == false)
				break;

		}

		ListNode* new_head = nullptr;

		for (int i = 0; i<tmp.size(); i++)
		{
			if (new_head == nullptr)
			{
				new_head = new ListNode(tmp[i]);
				now = new_head;
			}
			else
			{
				now->next = new ListNode(tmp[i]);
				now = now->next;
			}
		}

		return new_head;

	}

	vector<int> gao(vector<int> &tmp)
	{
		while (1)
		{
			if (tmp.size() == 0)
				break;
			vector<int> num = tmp;
			vector<int> sum(num.size());
			map<int, int>mp;

			
			bool flag = false;
			for (int i = 0; i<num.size(); i++)
			{
				if (i == 0)
					sum[i] = num[i];
				else
					sum[i] = sum[i-1] + num[i];
				if (sum[i] == 0)
				{
					tmp.clear();
					for (int j = i + 1; j<num.size(); j++)
						tmp.push_back(num[j]);
					flag = true;
					break;
				}
				if (mp.find(sum[i])!=mp.end())
				{
					tmp.clear();
					for (int j = 0; j<=mp[sum[i]]; j++)
						tmp.push_back(num[j]);
					for (int j = i + 1; j<num.size(); j++)
						tmp.push_back(num[j]);
					flag = true;
					break;
				}
				
				mp[sum[i]] = i;
			}
			
			if (flag == false)
				break;

		}
		return tmp;
	}
};
