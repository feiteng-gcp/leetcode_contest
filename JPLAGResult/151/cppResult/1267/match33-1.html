/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        ListNode preHead(0);
        preHead.next = head;
        vector<int> nums;
        auto it = head;
        while (it) {
            nums.push_back(it->val);
            it = it->next;
        }
        int len = nums.size();
        vector<bool> remove(len, false);
        unordered_map<int, int> sumToPos;
        sumToPos[0] = -1;
        unordered_map<int, int> posToSum;
        posToSum[-1] = 0;
        int sum = 0;
        for (int i = 0; i < len; ++i) {
            sum += nums[i];
            auto it = sumToPos.find(sum);
            if (it != sumToPos.end()) {
                int start = it->second + 1;
                int end = i;
                for (int j = start; j <= end; ++j) {
                    auto itt = posToSum.find(j);
                    remove[j] = true;
                    if (itt != posToSum.end()) {
                        sumToPos.erase(itt->second);
                        posToSum.erase(itt->first);
                    }
                }
            }
            sumToPos[sum] = i;
            posToSum[i] = sum;
        }
        it = &preHead;
        int i = 0;
        for (int i = 0; i < remove.size(); ++i) {

            if (remove[i]) {
                it->next = it->next->next;
            }
            else {
                it = it->next;
            }
        }
        return preHead.next;
    }
};
