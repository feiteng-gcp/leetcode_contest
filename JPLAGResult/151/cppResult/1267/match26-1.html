/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        ListNode dummy(0);
        dummy.next = head;
        ListNode *head_prev = &dummy;
        
        while (head_prev->next)
        {
            ListNode *node = head_prev->next;
            int sum = 0;
            
            while (node)
            {
                sum += node->val;                
                node = node->next;
                
                if (sum == 0) break;
            }
            

            if (sum == 0)
            {
                head_prev->next = node;
            }
            else
            {
                head_prev = head_prev->next;
            }
        }
        
        return dummy.next;        
    }
};
