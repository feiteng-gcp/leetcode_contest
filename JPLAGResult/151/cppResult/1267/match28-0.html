/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
ListNode* removeZeroSumSublists(ListNode* head) {
    ListNode* start = head;
    ListNode* end;
    ListNode *t=NULL;
    ListNode* beg = NULL;
    while (start)
    {
        bool mod = false;
        int sum = 0;
        end = start;
        while (end)
        {
            sum += end->val;
            if ( sum == 0 )
            {
                start = end;
                mod = true;
                break;
            }
            end = end->next;
        }
        if ( mod == false ) {
            if(beg==NULL)
                beg = start,t=beg;
            else{
                beg->next = start;
                beg = start;
            }
        }
        else{
            if(beg==NULL)
            {
                
            }
            else
            beg->next = NULL;
        }
       
        start = start->next;
    }
        return t;
}
};
