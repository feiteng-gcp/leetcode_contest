/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> r;
    ListNode* removeZeroSumSublists(ListNode* head) {
        f(head);
        
        while(true){
            map<int, int> memo;
            memo[0] = -1;
            int sum = 0;
            bool ok = false;
            for(int i = 0 ; i < r.size() ; i++){
                sum += r[i];
                if( memo.count(sum) ){
                    int l = memo[sum] + 1;
                    int R = i + 1;
                    vector<int> tr;
                    for(int k = 0 ; k < r.size() ; k++){
                        if( l <= k && k < R ) continue;
                        tr.push_back(r[k]);
                    }
                    r = tr;
                    ok = true;
                    break;
                }
                memo[sum] = i;
            }
            
            if(!ok) break;
        }
        
        if( r.size() == 0 ) return NULL;
        
        
        ListNode *root = new ListNode(r[0]);
        ListNode *tail = root;
        for(int i = 1 ; i < r.size() ; i++){
            tail->next = new ListNode(r[i]);
            tail = tail->next;
        }
        return root;
    }
    
    void f(ListNode *x){
        if(!x) return;
        r.push_back(x->val);
        f(x->next);
    }
    
    
};
