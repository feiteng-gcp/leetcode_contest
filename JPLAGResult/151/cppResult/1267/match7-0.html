/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        ListNode *newHead = new ListNode(0);
        newHead->next = head;
        unordered_map<int, ListNode*> m;
        m[0] = newHead;
        int sum = 0;
        ListNode *p = head;
        while (p) {
            sum += p->val;
            if (m.find(sum) != m.end()) {
                p = p->next;
                m[sum]->next = p;
            }
            else {
                m[sum] = p;
                p = p->next;
            }
        }
        return newHead->next;
    }
};
