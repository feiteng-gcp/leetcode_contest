/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        ListNode * iter = head;
        map<int, ListNode*> vis;
        ListNode * sentinel = new ListNode(0);
        sentinel->next = head;
        vis[0] = sentinel;

        int sum = 0;
        while (iter!= nullptr) {
            sum += iter->val;
            if (vis.count(sum)) {
                vis[sum]->next = iter->next;
                iter = vis[sum];
            } else {
                vis[sum] = iter;
            }
            iter = iter->next;
        }
        return sentinel->next;
    }
};
