/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        int len = 0;
        ListNode* p = head;
        vector<ListNode*> istack;
        while(p) {

            len += 1;
            istack.push_back(p);
            p = p->next;
            
            int sum = 0;
            for(int i = len - 1; i >= 0; --i) {
                sum += istack[i]->val;
                if(sum == 0) {
                    len = i;
                    break;
                }
            }
            while(len != istack.size()) {
                istack.pop_back();
            }
        }
        for(int i = 0; i < istack.size(); ++i) {
            if(i != istack.size() - 1) {
                istack[i]->next = istack[i + 1];
            } else {
                istack[i]->next = NULL;
            }
        }
        return istack.size() == 0 ? NULL : istack[0];
    }
};
