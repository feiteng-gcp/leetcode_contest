/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
#define ll int 
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        map<ll,ListNode*>pk;
        ListNode *cur=head;
        
        ll sum=0;
        while(cur!=NULL){
            sum+=cur->val;
            if(sum == 0) head=cur->next,cur=head;
            else {
                ll tsum=0;
                ListNode *tp=head;
                while(tp!=cur){
                    tsum+=tp->val;
                    if(tsum==sum) break;
                    tp=tp->next;
                }
                if(tsum==sum){
                    tp->next=cur->next;
                    sum=tsum;
                    cur=cur->next;
                }
                else {
                    cur=cur->next;
                }
            }
        }
        return head;
    }
};
