/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        unordered_map<int, ListNode*> hm; // prefix sum -> node
        auto dummy = new ListNode(0);
        dummy->next = head;
        hm[0] = dummy;
        int sum = 0;
        while(head) {
            sum += head->val;
            if(sum == 0) {
                dummy->next = head->next;
                hm.clear();
                hm[0] = dummy;
                head = head->next;
            } else if(hm.count(sum)) {
                auto p = hm[sum];
                auto tmp = p->next;
                while(tmp != head) {
                    sum += tmp->val;
                    hm.erase(sum);
                    tmp = tmp->next;
                }
                sum += head->val;
                p->next = head->next;
                head = head->next;
            } else {
                hm[sum] = head;
                head = head->next;
            }
        }
        return dummy->next;
    }
};
