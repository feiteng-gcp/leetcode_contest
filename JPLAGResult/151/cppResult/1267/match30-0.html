/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        vector<int> a;
        for (ListNode* p = head; p; p = p->next) a.push_back(p->val);
        while (true) {
            int n = a.size();
            vector<int> sum(n + 1);
            for (int i = 1; i <= n; ++i) sum[i] = sum[i - 1] + a[i - 1];
            unordered_map<int, int> A;
            A[0] = 0;
            int x = -1, y = -1;
            for (int i = 1; i <= n; ++i) {
                if (A.count(sum[i])) {
                    x = A[sum[i]];
                    y = i;
                    break;
                }
                A[sum[i]] = i;
            }
            
            if (x < 0) break;
            vector<int> b;
            for (int i = 0; i < n; ++i) {
                if (i >= x && i < y) continue;
                b.push_back(a[i]);
            }
            a = b;
            // cout << "==" << x << " " << y << endl;
            // for (auto& it : a) cout << it << " ";
            // cout << endl;
        }
        ListNode* ret = nullptr;
        ListNode* last = ret;
        for (auto& it : a) {
            ListNode* p = new ListNode(it);
            if (!last) {
                ret = last = p;
            } else {
                last->next = p;
                last = p;
            }
        }
        return ret;
    }
};
