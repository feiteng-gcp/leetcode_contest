/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        
        ListNode hh(0);
        hh.next = head;
        while(true)
        {
            ListNode* cur = hh.next;
            map<int, ListNode*> dp;
            dp[0] = &hh;
            int sum = 0;
            bool flag = false;
            while(cur)
            {
                sum += cur->val;
                if(dp.find(sum)!=dp.end())
                {
                    dp[sum]->next = cur->next;
                    flag = true;
                    break;
                }
                else
                {
                    dp[sum] = cur;
                }
                cur = cur->next;
                
            }
            if(!flag) break;
        }
        return hh.next;
        
    }
};
