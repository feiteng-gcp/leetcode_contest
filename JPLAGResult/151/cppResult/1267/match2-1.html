/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        int cur = 0;

        ListNode *ret = new ListNode(0);
        ret->next = head;
        map<int, ListNode *> m;
        m[0] = ret;

        int ps = 0;
        while (head) {
            ps += head->val;
            if (m.count(ps)) {
                m[ps]->next = head->next;
            } else {
                m[ps] = head;
            }
            head = head->next;
        }

        return ret->next;
    }
};
