class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {

        ListNode* dummyHead = new ListNode(-1);
        dummyHead->next = head;

        bool del = false;
        do{
            del = false;
            ListNode* from = NULL;
            ListNode* to = NULL;
            unordered_map<ListNode*, ListNode*> pre;
            for(ListNode* node = dummyHead; node->next; node = node->next)
                pre[node->next] = node;
            
            del = detect(dummyHead, from, to);
            if(del){
                assert(from && to);
                pre[from]->next = to->next;
            }
        }while(del);
        return dummyHead->next;
    }

private:
    bool detect(ListNode* dummyHead, ListNode*& from, ListNode*& to){

        vector<int> presum;
        unordered_map<int, ListNode*> map;
        for(ListNode* node = dummyHead; node; node = node->next){
            presum.push_back(presum.size() == 0 ? 0 : presum.back() + node->val);
            if(map[presum.back()]){
                from = map[presum.back()]->next;
                to = node;
                return true;
            }
            map[presum.back()] = node;
        }
        return false;
    }
};
