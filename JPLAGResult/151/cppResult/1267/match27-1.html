/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        vector<int> v;
        for (auto it = head; it; it = it->next)
            v.push_back(it->val);
        //for (auto x :v) printf("%d ", x); puts("");
        while (1)
        {
            int flag = 0;
            int n = v.size();
            vector<int> s(n+1);
            unordered_map<int, int> F;
            F[0] = 0;
            for (int i = 1; i <= n; ++ i)
            {
                s[i] = s[i-1]+v[i-1];
                if (F.count(s[i]))
                {
                    flag = 1;
                    int x = F[s[i]]+1, y = i;
                    //printf("%d %d\n", x, y);
                    vector<int> w;
                    for (int i = 1; i < x; ++ i)
                        w.push_back(v[i-1]);

                    for (int i = y+1; i <= n; ++ i)
                        w.push_back(v[i-1]);
                    v = w;
                    //for (auto x :v) printf("%d ", x); puts("");
                    break;
                }
                F[s[i]] = i;
            }
            if (!flag) break;
        }
        //for (auto x :v) printf("%d ", x); puts("");
        if (v.empty())
            return NULL;
        ListNode *ret = new ListNode(v[0]);
        int n = v.size();

        auto cur = ret;
        for (int i = 1; i < n; ++ i)
        {
            cur->next = new ListNode(v[i]);
            cur = cur->next;
        }
        return ret;
    }
};
