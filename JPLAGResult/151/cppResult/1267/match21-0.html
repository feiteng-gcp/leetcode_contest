/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        
        ListNode dummy(0);
        dummy.next = head;
        while (dummy.next) {
            bool flag = true;
            int sum = 0;
            unordered_map<int, ListNode*> presum;
            ListNode* cur = dummy.next;
            presum[0] = &dummy;
            while (cur) {
                sum += cur->val;
                if (presum.find(sum) == presum.end()) {
                    presum[sum] = cur;
                }
                else {
                    presum[sum]->next = cur->next;
                    flag = false;
                    break;
                }
                cur = cur->next;
            }
            if (flag) break;
        }
        return dummy.next;
