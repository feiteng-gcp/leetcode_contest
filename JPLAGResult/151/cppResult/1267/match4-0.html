/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        vector<int> psum;
        psum.push_back(0);
        map<int, vector<ListNode*>> m;
        ListNode p{-1};
        p.next = head;
        m[0].push_back(&p);
        int v = 0;
        while (head) {
            v = head->val + v;
            if (!m[v].empty()) {
                m[v].back()->next = head->next;
                // m[v].pop_back();
            } else {
                m[v].push_back(head);
            }
            head = head->next;
        }
        return p.next;
    }
};
