/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* num[2000010];
    ListNode* removeZeroSumSublists(ListNode* head) {
        memset(num, 0, sizeof(num));
        ListNode* p = head;
        int add = 0;

        while (p!=NULL){
            add += p->val;
            if (add == 0){
                head = p->next;
            }

            else if (num[add+1000000]!=0){
                ListNode* q = num[add+1000000];
                q->next = p ->next;
            }
            else {
                num[add+1000000] = p;
            }
            p = p -> next;
            
        }
        return head;
        
    }
};
