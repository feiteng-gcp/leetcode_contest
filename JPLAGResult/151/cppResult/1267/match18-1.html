/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        int sum = 0;
        unordered_map<int, ListNode*> sumToNode;
        sumToNode[0] = new ListNode(INT_MAX);
        sumToNode[0]->next = head;
        //while (head && head->val == 0) {
        //    head = head->next;
        //}
        
        auto tmp = head;
        

        while (tmp) {
            sum += tmp->val;
            if (sumToNode.find(sum) != sumToNode.end()) {
                sumToNode[sum]->next = tmp->next;
            } else {
                sumToNode[sum] = tmp;
            }
            tmp = tmp->next;
        }
        return sumToNode[0]->next;
    }
};
