/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        vector<int> a;
        ListNode* cur = head;
        while(cur != NULL){
            a.push_back(cur->val);
            cur = cur->next;
        }
        vector<int> ret;
        for(int i=0;i<a.size();i++){

            if(a[i]==0)continue;
            int sum = a[i];
            int pos = -1;
            for(int j=ret.size()-1;j>=0;j--){
                sum+=ret[j];
                if(sum==0){
                    pos = j;
                    break;
                }
            }
            if(pos==-1)ret.push_back(a[i]);
            else {
                while(ret.size()>pos)ret.pop_back();

            }
        }
        if(ret.size()==0)return NULL;
        ListNode* rethead = new ListNode(ret[0]);
        ListNode* now = rethead;
        for(int i=1;i<ret.size();i++){
            now->next = new ListNode(ret[i]);
            now = now->next;
        }
        return rethead;
    }
};
