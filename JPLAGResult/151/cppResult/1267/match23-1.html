/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        ListNode h(0);
        h.next = head;
        ListNode* t=&h, *tmp;
        while(t){
            int sum = 0;
            tmp = t->next;
            while(tmp){
                sum+=tmp->val;
                if(sum==0) break;
                tmp = tmp->next;
            }

            if(tmp!=nullptr){
                t->next = tmp->next;
            }
            else{
                t=t->next;
            }
        }
        return h.next;
    }
};
