/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        ListNode* ret = new ListNode(0);
        ret->next = head;
        
        map<int, ListNode*> data = map<int, ListNode*>();
        int sum = 0;
        ListNode *p = ret;
        while (p != NULL) {
            sum += p->val;
            if (data.find(sum) == data.end()) {
                data[sum] = p;
            }
            else {
                data[sum]->next = p->next;                
            }
            
            p = p->next;
        }
        
        return ret->next;
    }
};
