/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:

    bool remove(vector<int> &V) {
        unordered_map<int,int> index;
        index[0] = -1;
        int prefix_sum = 0, N = V.size();
        for (int i = 0; i < N; i++) {
            prefix_sum += V[i];
            if (index.count(prefix_sum)) {
                int L = index[prefix_sum] + 1, R = i;
                vector<int> temp;
                for (int j = 0; j < L; j++) {
                    temp.push_back(V[j]);
                }
                for (int j = R + 1; j < N; j++) {
                    temp.push_back(V[j]);
                }
                V.swap(temp);
                return true;
            }
            index[prefix_sum] = i;
        }
        return false;
    }
    ListNode* removeZeroSumSublists(ListNode* head) {
        vector<int> V;
        ListNode *cur = head;
        while (cur != NULL) {
            V.push_back(cur->val);
            cur = cur->next;
        }
        bool go = true;
        while (go) {
            go = remove(V);
        }
        int N = V.size();
        if (N == 0) {
            return NULL;
        }
        ListNode *new_head = new ListNode(V[0]);
        ListNode *prev = new_head;
        for (int i = 1; i < N; i++) {
            ListNode *cur = new ListNode(V[i]);
            prev->next = cur;
            prev = cur;
        }
        return new_head;
    }
};
