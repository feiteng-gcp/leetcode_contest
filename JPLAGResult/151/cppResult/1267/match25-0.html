/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
  ListNode* removeZeroSumSublists(ListNode* head) {
    ListNode* tail=NULL;
    ListNode** cur = &tail;
    while (head != NULL) {
      int tot = 0;
      ListNode *i = head;
      while (i!=NULL) {
        tot += i->val;
        if (tot == 0) break;
        i = i->next;
      }
      if (tot == 0) {
        head = i->next;
      } else {
        auto next = head->next;
        head->next = NULL;
        *cur = head;
        cur = &head->next;
        head = next;
      }
    }
    return tail;
  }
};
