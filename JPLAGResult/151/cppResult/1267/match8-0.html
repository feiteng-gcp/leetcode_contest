/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        unordered_map<int, ListNode*> preSumToNode;
        unordered_map<ListNode*, int> NodeToPreSum;
        ListNode* dummy = new ListNode(0);
        dummy->next = head;
        ListNode* ptr = head;
        int currSum = 0;
        preSumToNode[0] = dummy;
        while(ptr){
            currSum += ptr->val;
            //cout<<currSum<<endl;
            if(preSumToNode.count(currSum) == 0){
                preSumToNode[currSum] = ptr;
                NodeToPreSum[ptr] = currSum;
            }else{
                ListNode* tmp = preSumToNode[currSum]->next;
                while(tmp != ptr){
                    preSumToNode.erase(NodeToPreSum[tmp]);
                    tmp = tmp->next;
                }
                preSumToNode[currSum]->next = ptr->next;
                ptr = preSumToNode[currSum];
            }
            ptr = ptr->next;
            
        }
        return dummy->next;
    }
};
