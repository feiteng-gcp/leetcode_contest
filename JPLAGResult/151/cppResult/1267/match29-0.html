/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeZeroSumSublists(ListNode* head) {
        vector<int> v;
        
        while (head) {
            int val = head->val;
            head = head->next;
            
            bool erased = false;
            int sum = 0;
            for (int i = v.size() - 1; i >= 0; i--) {
                sum += v[i];
                if (sum + val == 0) {
                    v.resize(i);
                    erased = true;
                    break;
                }
            }
            
            if (!erased && val != 0) {
                v.push_back(val);
            }
        }
        
        ListNode* phead = new ListNode(-1);
        ListNode* ans = phead;
        for (int e : v) {
            ListNode* node = new ListNode(e);
            phead->next = node;
            phead = phead->next;
        }
        
        return ans->next;
    }
};
