class DinnerPlates:

    def __init__(self, capacity: int):
        self.s = []
        self.l = 0
        self.c = capacity
        

    def push(self, val: int) -> None:
        while self.l < len(self.s) and len(self.s[self.l]) == self.c:
            self.l += 1
        if self.l == len(self.s):
            self.s.append([])
        self.s[self.l].append(val)
        

    def pop(self) -> int:
        if not self.s:
            return -1
        r = self.s[-1].pop()
        while self.s and not self.s[-1]:
            self.s.pop()
        if self.l > len(self.s):
            self.l = len(self.s)
        return r
        

    def popAtStack(self, index: int) -> int:
        if index == len(self.s) - 1:
            return self.pop()
        if index >= len(self.s) or not self.s[index]:
            return -1
        r = self.s[index].pop()
        if index < self.l:
            self.l = index
        return r
