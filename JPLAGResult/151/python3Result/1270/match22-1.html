    def push(self, val: int) -> None:
        if len(self.pq) == 0:
            self.stacks.append([])
            self.pq.append(len(self.stacks) - 1)
            self.rightmost += 1
        ind = self.pq[0]
        self.stacks[ind].append(val)
        self.rightmost = max(self.rightmost, ind)
        if len(self.stacks[ind]) == self.capacity:
            heapq.heappop(self.pq)
