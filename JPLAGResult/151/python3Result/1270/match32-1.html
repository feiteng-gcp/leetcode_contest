class DinnerPlates:

    def __init__(self, capacity: int):
        self.M = 100001
        self.cap = capacity
        self.din = [[] for _ in range(self.M)]
        self
.pushInd = 0
        self.popInd = 0
        

    def push(self, val: int) -> None:
        self.din[self.pushInd].append(val)
        while len(self.din[self.pushInd]) == self.cap:
            self.pushInd += 1
            if self.popInd < self.pushInd:
                self.popInd = self.pushInd

    def pop(self) -> int:
        while self.popInd >= 0 and not self.din[self.popInd]:
            self.popInd -= 1
        if self.popInd == -1:
            return -1
        res = self.din[self.popInd].pop()
        if not self.din[self.popInd]:
            while self.popInd >= 0 and (not self.din[self.popInd]):
                self.popInd -= 1
        if self.popInd < self.pushInd:
            self.pushInd = self.popInd
        return res

    def popAtStack(self, index: int) -> int:
        res = -1
        if not self.din[index]:
            return res
        res = self.din[index].pop()
        if self.pushInd > index:
            self.pushInd = index
        if not self.din[index]:
            if self.popInd == index:
                while self.popInd >= 0 and (not self.din[self.popInd]):
                    # print("popAtStack", self.popInd)
                    self.popInd -= 1
        
        if self.popInd < self.pushInd:
            self.pushInd = self.popInd
            if self.pushInd == -1:
                self.pushInd += 1
        return res
