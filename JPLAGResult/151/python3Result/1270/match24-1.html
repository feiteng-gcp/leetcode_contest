class DinnerPlates:

    def __init__(self, capacity: int):
        self.cnt = capacity
        self.stack = [[]]
        self.nxtpush = collections.deque()
    def push(self, val: int) -> None:
        while self.nxtpush:
            idx = self.nxtpush.popleft()
            if len
(self.stack[idx])<self.cnt:
                self.stack[idx].append(val)
                return
        if len(self.stack
[-1])<self.cnt:
            self.stack[-1].append(val)
        else:
            self.stack.append([val])
        
    def pop(self) -> int:
        while self.stack and self.stack[-1]==[]:
            self.stack.pop()
        if self.stack and self.stack[-1]:
            return self.stack[-1].pop()        
        return -1
            
                

    def popAtStack(self, index: int) -> int:
        if not self.stack[index]:
            return -1
        bisect.insort(self.nxtpush, index)
        return self.stack[index].pop()
