import collections
import bisect
class DinnerPlates:

    def __init__(self, capacity: int):
        self.capacity = capacity
        self.stacks = []
        self.blanks = []
        

    def push(self, val: int) -> None:
        if not self.stacks:
            self.stacks.append([val])
            if self.capacity > 1:
                self.blanks = [0]
        else:
            if not self.blanks:
                self.stacks.append([val])
                if self.capacity > 1:
                    self.blanks = [len(self.stacks) - 1]
            else:
                #print(self.blanks, self.stacks)
                self.stacks[self.blanks[0]].append(val)
                #self.blanks[0] += 1
                if len(self.stacks[self.blanks[0]]) >= self.capacity:
                    self.blanks.pop(0)

    def pop(self) -> int:
        #print(self.stacks)
        while self.stacks and not self.stacks[-1]:
            self.stacks.pop()
        
        while self.blanks and self.blanks[-1] >= len(self.stacks):
            self.blanks.pop()
        
        if not self.stacks:
            return -1
        if self.blanks and self.blanks[-1] < len(self.stacks) - 1:
            self.blanks.append(len(self.stacks) - 1)
        return self.stacks[-1].pop()

    def popAtStack(self, index: int) -> int:
        #print(self.blanks, self.stacks)
        if index >= len(self.stacks):
            return -1
        if index not in self.blanks:
            bisect.insort(self.blanks, index)
        
        if not self.stacks[index]:
            return -1
        else:
            return self.stacks[index].pop()
