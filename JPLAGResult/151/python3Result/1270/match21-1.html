class DinnerPlates:

    def __init__(self, capacity: int):
        self.capacity = capacity
        self
.left = 0
        self.right = -1
        self.stack_list = [[]]

    def push(self, val: int) -> None:
        self.stack_list[self.left].append(val)
        if self.left > self.right:
            self.right = self.left

        while self.left < len(self.stack_list) and len(self.stack_list[self.left]) == self.capacity:
            self.left += 1
        if self.left == len(self.stack_list):
            self.stack_list.append([])

    def pop(self) -> int:
        if self.right == -1:
            return -1

        ans = self.stack_list[self.right].pop()
        while self.right >= 0 and len(self.stack_list[self.right]) == 0:
            self.right -= 1

        if self.right == -1:
            self.left = 0
        elif self.right < self.left and len(self.stack_list[self.right]) < self.capacity:
            self.left = self.right
        elif self.right + 1 < self.left and len(self.stack_list[self.right]) == self.capacity:
            self.left = self.right + 1

        return ans


    def popAtStack(self, index: int) -> int:
        if index > self.right:
            return -1
        elif index == self.right:
            return self.pop()
        
        
if len(self.stack_list[index]) == 0:
            return -1
        ans = self.stack_list[index].pop()
        if index < self.left:
            self.left = index
        return ans
