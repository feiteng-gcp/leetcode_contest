        self.empty = deque([0])
        self.cap = capacity

    def push(self, val: int) -> None:
        
        idx = self.empty.popleft()
        while self.empty and self.empty[0] == idx:
            self.empty.popleft()
            
        self.stacks[idx].append(val)
        
        if len(self.stacks[idx]) < self.cap:
            self.empty.appendleft(idx)
        else:
            if not self.empty:
                self.empty.append(len(self.stacks))
                self.stacks.append([])
