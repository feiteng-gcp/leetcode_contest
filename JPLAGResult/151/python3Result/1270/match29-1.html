class DinnerPlates:

    def __init__(self, capacity: int):
        self._capacity = capacity
        self._stacks_size = 0
        self._stacks = []
        self._next_push_idx = 0
        

    def push(self, val: int) -> None:
        while self._next_push_idx < self._stacks_size:
            if len(self._stacks[self._next_push_idx]) < self._capacity:
                self._stacks[self._next_push_idx].append(val)
                return
            
            self._next_push_idx += 1
        
        self._stacks.append([val])
        self._stacks_size += 1

    def pop(self) -> int:
        if not self._stacks:
            return -1
        
        val = self._stacks[-1].pop()
            
        if not self._stacks[-1]:
            self._stacks.pop()
            self._stacks_size -= 1
            
        if self._next_push_idx >= self._stacks_size:
            self._next_push_idx = self._stacks_size - 1
            
        return val

    def popAtStack(self, index: int) -> int:
        if len(self._stacks) <= index:
            return -1
        
        if not self._stacks[index]:
            return -1
        
        val = self
._stacks[index].pop()
        if index == len(self._stacks) - 1 and not self._stacks[-1]:
            self._stacks.pop()
            self._stacks_size -= 1
        
        if self._next_push_idx > index:
            self._next_push_idx = index
            
        return val
