from collections import defaultdict
class DinnerPlates:

    def __init__(self, capacity: int):
        self.stacks = defaultdict(list)
        self.left = 0
        self
.right = 0
        self.size = capacity

    def push(self, val: int) -> None:
        self.stacks[self.left].append(val)
        while len(self.stacks[self.left]) == self.size:
            self.left += 1
        if self.left > self.right:
            self.right = self.left
            if len(self.stacks[self.left]) == 0:
                self.right -= 1
        # print(self.stacks)

    def pop(self) -> int:
        if self.right == -1:
            return -1
        if self.right == 0 and len(self.stacks[0]) == 0:
            return -1
        p = self.stacks[self.right].pop()
        while len(self.stacks[self.right]) == 0 and self.right > -1:
            self.right -= 1
        # print(self.stacks)
        return p

    def popAtStack(self, index: int) -> int:
        if len(self
.stacks[index]) == 0:

        if index < self.left:
