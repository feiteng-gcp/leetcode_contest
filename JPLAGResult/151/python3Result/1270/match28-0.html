import bisect
class DinnerPlates:

    def __init__(self, capacity: int):
        self.cap = capacity
        self.queue = []
        self.cur = [(0,0)] # idx, @stack
        

    def push(self, val: int) -> None:
        idx, i = self.cur.pop(0)
        if len(self.queue)<idx+1:
            self.queue.append([])
        self.queue[idx].append(val)
        if not self.cur:
            if i+1<self.cap:
                self.cur.append((idx,i+1))
            else:
                self.cur.append((idx+1,0))

    def pop(self) -> int:
        if not self.queue:
            return -1
        val = self.queue[-1].pop()
        idx,i = self.cur.pop()
        if not self.queue[-1]:
            self.queue.pop()
        if i:
            self.cur.append((idx,i-1))
        else:
            self.cur.append((idx-1,self.cap-1))
        return val

    def popAtStack(self, index: int) -> int:
        if len(self.queue)<=index or not self.queue[index]:
            return -1
        val = self.queue[index].pop()
        bisect.insort(self.cur,(index,len(self.queue[index])))
        return val
