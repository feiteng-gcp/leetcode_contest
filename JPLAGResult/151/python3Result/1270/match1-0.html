from heapq import heappush, heappop
class DinnerPlates:

    def __init__(self, capacity: int):
        self.stacks = []
        self.K = capacity
        self.avail = []

    def push(self, val: int) -> None:
        if not self.avail:
            heappush(self.avail, len(self.stacks))
            self.stacks.append([val])
        else:
            i = heappop(self.avail)
            self.stacks[i].append(val)
            if len(self.stacks[i]) < self.K:
                heappush(self.avail, i)
        # print(f'push {val}', self.stacks, self.avail)

    def pop(self) -> int:
        if not self.stacks: return -1
        return self.popAtStack(len(self.stacks) - 1)
        
    def popAtStack(self, index: int) -> int:
        if index >= len(self.stacks) or not self.stacks[index]:
            return -1
        else:
            stack = self.stacks[index]
            val = stack.pop()
            if index == len(self.stacks) - 1 and not stack:
                if self.avail:
                    self.avail.pop()
                self.stacks.pop()
            elif len(stack) + 1 == self.K:
                heappush(self.avail, index)
        # print(f'popAtStack {index}', self.stacks, self.avail)
        return val
