from heapq import *
class DinnerPlates:

    def __init__(self, capacity: int):
        self.maxC = capacity
        self.stacks = collections.deque([[]])
        self
.nextAvail = [0]
        self.nextPop = []
    
    def push(self, val: int) -> None:
        if not self.nextAvail:
            heappush(self.nextAvail, len(self.stacks))
            heappush(self.nextPop, -len(self.stacks))
            self.stacks.append([val])
        else:
            idx = heappop(self.nextAvail)
            if idx >= len(self.stacks):
                heappush(self.nextAvail, len(self.stacks))
                heappush(self.nextPop, -len(self.stacks))
                self.stacks.append([val])
            else:
                self
.stacks[idx].append(val)
                if len(self.stacks[idx]) < self.maxC:
                    heappush(self.nextAvail, idx)
                heappush(self.nextPop, -idx)
            

    def pop(self) -> int:
        while self.nextPop:
            idx = -heappop(self.nextPop)
            if self.stacks[idx]:
                if len(self.stacks[idx]) == self.maxC:
                    heappush(self.nextAvail, idx)
                return self.stacks[idx].pop()
        return -1 
        

    def popAtStack(self, index: int) -> int:
        if index >= len(self.stacks):
            return -1
        if self.stacks[index]:
            if len(self.stacks[index]) == self.maxC:
                heappush(self.nextAvail, index)
            return self.stacks[index].pop()
        else:
            return -1
