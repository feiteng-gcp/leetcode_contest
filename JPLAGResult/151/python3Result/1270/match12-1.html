class DinnerPlates:

    def __init__(self, capacity: int):
        self.stacks = []
        for i in range(200000):
            self.stacks.append([])
        
        self.leftMostNonFull = 0
        self.rightMostNonEmpty = -1
        self.capacity = capacity

    def push(self, val: int) -> None:
        self
.stacks[self.leftMostNonFull].append(val)
        if self.leftMostNonFull > self.rightMostNonEmpty:
            self.rightMostNonEmpty = self.leftMostNonFull
        while len
(self.stacks[self.leftMostNonFull]) == self.capacity:
            self.leftMostNonFull += 1    
        
        
    def pop(self) -> int:
        if self.rightMostNonEmpty == -1:
            return -1
        else:
            res = self
.stacks[self.rightMostNonEmpty].pop()
            if self.rightMostNonEmpty < self.leftMostNonFull:
                self.leftMostNonFull = self.rightMostNonEmpty
            while self.rightMostNonEmpty >= 0 and not self.stacks[self.rightMostNonEmpty]:
                self.rightMostNonEmpty -= 1
            return res

    def popAtStack(self, index: int) -> int:    
        if not self.stacks[index]:
            return -1
        else:
            res = self.stacks[index].pop()
            if index < self.leftMostNonFull:
                self.leftMostNonFull = index
            if not self.stacks[index] and index == self.rightMostNonEmpty:
                while self.rightMostNonEmpty >= 0 and not self.stacks[self.rightMostNonEmpty]:
                    self.rightMostNonEmpty -= 1
            return res
