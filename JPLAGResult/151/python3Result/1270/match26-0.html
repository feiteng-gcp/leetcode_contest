from collections import deque


class DinnerPlates:

    def __init__(self, capacity: int):
        self.capacity = capacity
        self.stacks = []
        self.push_idx = deque()

    def push(self, val: int) -> None:
        if not self.push_idx:
            self.stacks.append([val])
            if self.capacity > 1:
                self.push_idx.append(len(self.stacks) - 1)
        else:
            idx = self.push_idx[0]
            self.stacks[idx].append(val)
            if len(self.stacks[idx]) == self.capacity:
                self.push_idx.popleft()

    def pop(self) -> int:
        while self.stacks and not self.stacks[-1]:
            self.stacks.pop()
        if not self.stacks:
            self.push_idx.clear()
            return -1
        ans = self.stacks[-1][-1]
        if len(self.stacks[-1]) == 1:
            self.stacks.pop()
        else:
            self.stacks[-1].pop()
        self.push_idx = [i for i in self.push_idx if i < len(self.stacks)]
        return ans

    def popAtStack(self, index: int) -> int:
        if index >= len(self.stacks) or not self.stacks[index]:
            return -1
        ans = self.stacks[index].pop()
        insert = -1
        for i, idx in enumerate(self.push_idx):
            if idx == index:
                break
            if idx > index:
                insert = i
                break
        if insert != -1:
            self.push_idx.insert(insert, index)
        return ans
