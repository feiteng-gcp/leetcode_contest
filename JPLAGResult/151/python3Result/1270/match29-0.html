class DinnerPlates:

    def __init__(self, capacity: int):
        self.capacity = capacity
        self.not_full = []
        self.full = []
        self.stacks = []

    def push(self, val: int) -> None:
        if len(self.not_full) == 0:
            if self.full:
                self.not_full.append(self.full[-1] + 1)
            else:
                self.not_full.append(0)
            self.stacks.append([])
        i = self.not_full[0]
        self.stacks[i].append(val)
        if len(self.stacks[i]) == self.capacity:
            self.not_full.pop(0)
            bisect.insort(self.full, i)
        #print("push {}, notfull:{}, full:{}, stacks:{}".format(val, self.not_full, self.full, self.stacks))
        return 

    def pop(self) -> int:
        i = 0
        if self.not_full and self.not_full[-1] > i:
            i = self.not_full[-1]
        if self.full and self.full[-1] > i:
            i = self.full[-1]
        return self.popAtStack(i)

    def popAtStack(self, index: int) -> int:
        if index >= len(self.stacks):
            return -1
        if len(self.stacks[index]) == 0:
            return -1
        if len(self.stacks[index]) == self.capacity:
            i = bisect.bisect_left(self.full, index)
            self.full.pop(i)
            bisect.insort(self.not_full, index)
        
        val = self.stacks[index].pop()
        if index == len(self.stacks)-1 and len(self.stacks[index]) == 0:
            self.not_full.pop()
            self.stacks.pop()
        #print("pop at stack {},val:{}, notfull:{}, full:{}, stacks:{}".format(index,val, self.not_full, self.full, self.stacks))
        return val
