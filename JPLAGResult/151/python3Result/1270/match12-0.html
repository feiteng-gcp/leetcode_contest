class DinnerPlates:

    def __init__(self, capacity: int):
        self.ind = 100001
        self.li = 0
        self.ri = -1
        self.stacks = []
        self.cap = capacity
        for _ in range(self.ind):
            self.stacks.append([])
        

    def push(self, val: int) -> None:
        if self.li >= self.ind:
            return
        self.stacks[self.li].append(val)
        if self.ri < self.li:
            self.ri = self.li 
        while self.li < self.ind and len(self.stacks[self.li]) == self.cap:
            self.li += 1
        

    def pop(self) -> int:
        if self.ri < 0:
            return -1
        val = self.stacks[self.ri].pop()
        if self.ri < self.li:
            self.li = self.ri 
        while self.ri >= 0 and len(self.stacks[self.ri]) == 0:
            self.ri -= 1
        return val
        

    def popAtStack(self, index: int) -> int:
        if len(self.stacks[index]) == 0:
            return -1
        val = self.stacks[index].pop()
        if index < self.li:
            self.li = index 
        return val
