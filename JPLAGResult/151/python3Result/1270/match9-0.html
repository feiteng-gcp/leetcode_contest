import collections

class DinnerPlates:
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.stacks = collections.defaultdict(list)
        self.push_idx = 0
        self.pop_idx = -1

    def push(self, val: int) -> None:
        self.stacks[self.push_idx].append(val)
        if self.pop_idx < self.push_idx:
            self.pop_idx = self.push_idx
        while len(self.stacks[self.push_idx]) == self.capacity:
            self.push_idx += 1

    def pop(self) -> int:
        if self.pop_idx < 0:
            return -1
        val = self.stacks[self.pop_idx].pop()
        if self.push_idx > self.pop_idx:
            self.push_idx = self.pop_idx
        while self.pop_idx >= 0 and not self.stacks[self.pop_idx]:
            self.pop_idx -= 1
        return val

    def popAtStack(self, index: int) -> int:
        if not self.stacks[index]:
            return -1
        val = self.stacks[index].pop()
        if self.push_idx > index:
            self.push_idx = index
        if self.pop_idx == index:
            while self.pop_idx >= 0 and not self.stacks[self.pop_idx]:
                self.pop_idx -= 1
        return val
