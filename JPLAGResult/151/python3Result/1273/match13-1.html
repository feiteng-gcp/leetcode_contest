from collections import Counter
from bisect import bisect

class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def f(s):
            s = Counter(s)
            return s[min(s)]
        wc = [f(w) for w in words]
        wc.sort()
        res = [len(wc) - bisect(wc, f(q)) for q in queries]
        return res
