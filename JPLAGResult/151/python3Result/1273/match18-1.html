class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        f1 = []
        f2 = []
        for q in queries:
            m = '{'
            app 
= 0
            for i in range(len(q)):
                if q[i]<m:
                    m = q[i]
                    app = 1
                elif q[i]==m:
                    app+=1
            f1.append(app)
        for q in words:
            m = '{'
            app = 0
            for i in range(len(q)):
                if q[i]<m:
                    m = q[i]
                    app = 1
                elif q[i]==m:
                    app+=1
            f2.append(app)
        answer = []
        
for i in f1:
            a = 0
            for j in f2:
                if i<j:
                    a+=1
            answer.append(a)
        return answer
