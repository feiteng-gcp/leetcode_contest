class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def f(s):
            freq = 26 * [0]
            for c in s:
                freq[ord(c) - ord('a')] += 1
            for i in range(26):
                if freq[i] > 0:
                    return freq[i]
        
        f_words = [f(w) for w in words]
        answer = len(queries) * [0]
        for i in range(len(queries)):
            f_q = f(queries[i])
            for f_w in f_words:
                if f_q < f_w:
                    answer[i] += 1
        return answer
