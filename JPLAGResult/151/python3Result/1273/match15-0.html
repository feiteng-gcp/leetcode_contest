class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def f(str1):
            smallest = min(str1)
            
            cnt = 0
            for c in str1:
                if smallest == c:
                    cnt += 1
            
            return cnt
        
        f_queries = [f(x) for x in queries]
        f_words = [f(x) for x in words]
        
        f_words.sort()

        #print(f_words)

            res.append(len(f_words) - insert_pos)
        
        return res
