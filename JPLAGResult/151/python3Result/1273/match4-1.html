from collections import Counter
class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def f(s):
            cnt = Counter(s)
            return cnt[min(cnt)]
        aux = Counter([f(s) for s in words])
        return [sum(v for k, v in aux.items() if f(q) < k) for q in queries]
