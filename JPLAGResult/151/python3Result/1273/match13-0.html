class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def f(w):
            c = min(w)
            return w.count(c)
        q = [f(w) for w in queries]
        ws = [f(w) for w in words]
        return [sum([qn < wn for wn in ws]) for qn in q]
