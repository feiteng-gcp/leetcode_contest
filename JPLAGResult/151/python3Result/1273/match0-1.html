class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        
        def F(s):
            if not s: return 0
            ch = min(s)
            return s.count(ch)
        
        fwords = []
        for word in words:
            fwords.append(F(word))
        
        res = []
        for q in queries:
            f_q = F(q)
            res.append(sum([f_q < fword for fword in fwords]))    
            
        return res
