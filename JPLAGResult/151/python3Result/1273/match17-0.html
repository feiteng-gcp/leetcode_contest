class Solution:
    def calcFreq(self, word):
        m_l = min(word)
        c = 0
        for i in word:
            c += int(m_l == i)
        return c
        
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        freq_q = []
        freq_w = []
        
        for i in queries:
            freq_q.append(self.calcFreq(i))
        
        for i in words:
            freq_w.append(self.calcFreq(i))
        
        ans = []
        for i in freq_q:
            c = 0
            for j in freq_w:
                c += int(i < j)
            
            ans.append(c)
        return ans
