from collections import Counter
class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        w_aux = []
        for w in words:
            c = Counter(w)
            t = sorted(list(c.keys()))[0]
            w_aux.append(c[t])
        q_aux = []
        for q in queries:
            c = Counter(q)
            t = sorted(list(c.keys()))[0]
            q_aux.append(c[t])
        ans = []
        for q in q_aux:
            tmp 
= 0
            for w in w_aux:
                if q < w:
                    tmp += 1
            ans.append(tmp)
        return ans
