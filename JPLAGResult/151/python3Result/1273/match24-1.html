class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def f(word):
            return word.count(min(word))
        check = []
        for w in words:
            bisect.insort(check, f(w))
        res = []
        for q in queries:
            idx = bisect.bisect_left(check, f(q)+1)
            res.append(len(check)-idx)
        return res
