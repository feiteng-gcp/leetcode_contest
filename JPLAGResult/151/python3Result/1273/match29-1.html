class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        
        def f(word):
            return word.count(min(word))
        
        queries = [f(q) for q in queries]
        words = sorted([f(w) for w in words])
        print(queries)
        print(words)
        n = len(words)
        ans = [n - bisect.bisect_right(words, t) for t in queries]
        return ans
