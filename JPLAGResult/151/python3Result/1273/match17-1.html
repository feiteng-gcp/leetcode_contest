class Solution:
    def f(self, s):
        count = collections.Counter(s)
        a = 'a'
        while a <= 'z':
            if count[a] != 0:
                return count[a]
            a = chr(ord(a) + 1)
        
return 0
    
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        res = []
        x, m = [], []
        for q in queries:
            x.append(self.f(q))
        for w in words:
            m.append(self.f(w))
        for q in x:
            r = 0
            for w in m:
                r += 1 if q < w else 0
            res.append(r)
        return res
