class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def f(s):
            freq = [0]*26
            for c in s:
                freq[ord(c)-ord('a')]+=1
            for i in range(26):
                if freq[i]:
                    return freq[i]
        
        queries_int = list(map(f, queries))
        words_int = list(map(f, words))
        ans = [0]*len(queries)
        for i in range(len(queries)):
            for j in range(len(words_int)):
                if words_int[j] > queries_int[i]:
                    ans[i] += 1
        return ans
