class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        
        def f(w):
            c = min(w)
            return w.count(c)
        
        fs = sorted([f(w) for w in words])
        n = len(words)
        return [n - bisect.bisect_right(fs, f(w)) for w in queries]
