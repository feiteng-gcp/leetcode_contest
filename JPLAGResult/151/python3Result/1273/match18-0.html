class Solution:
    def numSmallerByFrequency(self, queries, words):
        def fun(s):
            _min = s[0]
            count = 1
            for i in range(1,len(s)):
                if _min > s[i]:
                    _min = s[i]
                    count = 1
                elif _min == s[i]:
                    count += 1
            return count
        res = []
        records_q = [fun(q) for q in queries]
        records_w = [fun(w) for w in words]
        for q in records_q:
            tmp = 0
            for w in records_w:
                if q < w:
                    tmp += 1
            res.append(tmp)
        return res
