class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def helper(w):
            c = min(w)
            return w.count(c)
        temp = sorted([helper(w) for w in words])
        n = len(words)
        ans = []
        
        for q in queries:
            ans.append(n - bisect.bisect_right(temp, helper(q)))
        #print(temp)
        return ans
