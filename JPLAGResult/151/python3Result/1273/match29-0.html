import bisect

class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def func(string):
            return string.count(min(string))
        
        f_queries = list(map(func, queries))
        f_words = sorted(list(map(func, words)))
        n = len(words)
        
        res = []
        # print(f_queries, f_words)
        
        for fq in f_queries:
            idx = bisect.bisect_left(f_words, fq+1)
            # print(fq, idx)
            res.append(n-idx)
            
        return res
