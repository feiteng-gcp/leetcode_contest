class Solution:
    def freqSmallest(self, s):
        m = min(s)
        n = 0
        for ch in s:
            n += ch == m
        return n
    def numSmallerByFrequency(self, queries, words) :
        q_len = []
        w_len = []
        ans = []
        for q in queries:
            q_len.append(self.freqSmallest(q))
        for w in words:
            w_len.append(self.freqSmallest(w))
        for q in q_len:
            tmp = 0
            for w in w_len:
                tmp += (q<w)
            ans.append(tmp)
            
        return ans
