from bisect import bisect_right
class Solution:
    def numSmallerByFrequency(self, queries: List[str], words: List[str]) -> List[int]:
        def calcMFreq(x):
            c = min(x)
            return x.count(c)
        
        wcnt = [calcMFreq(x) for x in words]
        qcnt = [calcMFreq(x) for x in queries]
        wcnt.sort()
        #print (wcnt, qcnt)
        ans, n = [], len(wcnt)
        for i, x in enumerate(qcnt):
            k = bisect_right(wcnt, x)
            ans.append(n - k)
        return ans
