class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        
        table = collections.defaultdict(list)
        
        res = set()
        
        for tran in transactions:
            
            name,time,amount,city = tran.split(",")
            
            if int(amount) > 1000:
                res.add(tran)
                
            if name in table:
                
                for t,c,ans in table[name]:
                
                    if c != city and abs(int(time) -int(t)) <= 60:
                        res.add(tran)
                        if ans not in res:
                            res.add(ans)
                
            table[name].append((time,city,tran))
            
        return list(res)
