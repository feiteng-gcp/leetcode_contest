class Solution:
    def invalidTransactions(self, transactions):
        transactions = [x.split(',') for x in transactions]
        S = set()
        for x in transactions:
            if float(x[2]) > 1000:
                S.add(','.join(x))
            for y in transactions:
                if x[0] == y[0] and x[3]!=y[3] and abs(float(x[1])-float(y[1]))<=60:
                    S.add(','.join(x))
                    S.add(','.join(x))

        res = list(S)
        return res
