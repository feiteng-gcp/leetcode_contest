class Solution:
    def invalidTransactions(self, ar: List[str]) -> List[str]:
        n = len(ar)
        ans = []
        for i in range(n):
            x = ar[i].split(",")
            x[1] = int(x[1])
            x[2] = int(x[2])
            
            if(x[2]>1000):
                ans.append(ar[i])
                continue
                
            for j in range(n):
                y = ar[j].split(",")
                y[1] = int(y[1])
                y[2] = int(y[2])
                
if y[0]==x[0] and x[3]!=y[3] and abs(x[1]-y[1])<=60:
                    ans.append(ar[i])
                    break
        return ans
