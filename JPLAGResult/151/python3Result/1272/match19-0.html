class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        flag = [0] * len(transactions)
        records = []
        
        for transaction in transactions:
            record = transaction.split(',')
            records.append(record)
        
        for i in range(len(records)):
            if int(records[i][2]) > 1000:
                flag[i] = 1
            else:
                for j in range(len(records)):
                    if records[i][0] == records[j][0] and records[i][3] != records[j][3] and \
                    abs(int(records[i][1]) - int(records[j][1])) <= 60:
                        flag[i] = 1
                        flag[j] = 1
                        
        return [transactions[i] for i in range(len(transactions)) if flag[i]]
