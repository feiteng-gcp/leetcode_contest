class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        name_map = collections.defaultdict(list)
        bad = []
        for transaction in transactions:
            name, time, amount, city = transaction.split(',')
            name_map[name].append(transaction)
        for transaction in transactions:
            name, time, amount, city = transaction.split(',')
            amount = int(amount)
            if amount > 1000:
                bad.append(transaction)
                continue
            time = int(time)
            ctransacitons = name_map[name]
            for ct in ctransacitons:
                if ct == transaction:
                    continue
                _, ctime, camount, ccity = ct.split(',')
                camount = int(camount)
                ctime = int(ctime)
                if city != ccity and abs(time-ctime) <= 60:
                    bad.append(transaction)
                    break
        return bad
