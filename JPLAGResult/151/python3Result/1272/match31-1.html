class Solution:
    def invalidTransactions(self, t: List[str]) -> List[str]:
        n = len(t)
        invalid = []
        
        for i in range(n):
            name,time,amount,city = t[i].split(",")
            if int(amount) > 1000:
                invalid.append(t[i])
                continue
                
            for j in range(n):
                if i == j:
                    continue
                n2,t2,a2,c2 = t[j].split(",")
                if name==n2 and city!=c2 and abs(int(time)-int(t2)) <= 60:
                    invalid.append(t[i])
                    break
        return invalid
