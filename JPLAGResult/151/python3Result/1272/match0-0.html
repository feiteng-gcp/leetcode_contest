class Solution:
    def invalidTransactions(self, tra: List[str]) -> List[str]:
        res = set()
        idx = collections.defaultdict(list)
        for t in tra:
            a, b, c, d = t.split(",")
            idx[a].append([b, c, d])
            if int(c) > 1000:
                res.add(t)
        
        for key in idx:
            for i, s1 in enumerate(idx[key]):
                for j, s2 in enumerate(idx[key]):
                    if i == j or s1[2] == s2[2]:
                        continue
                    if abs(int(s1[0]) - int(s2[0])) <= 60:
                        res.add(",".join([key] + s1))
                        res.add(",".join([key] + s2))
                        
        return res
