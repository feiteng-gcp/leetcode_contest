class Solution:
    def invalidTransactions(self, transactions):
        n = len(transactions)
        trans_list = []
        ans = []
        for i,trans in enumerate(transactions):
            l = trans.split(",")
            #print(l)
            if int(l[2])>1000:
                ans.append(i)
            trans_list.append(l)
        for i in range(n):
            tmp1 = trans_list[i]
            for j in range(n):
                tmp2 = trans_list[j]
                if tmp1[0] == tmp2[0] and abs(int(tmp1[1])-int(tmp2[1]))<=60 and tmp1[3] != tmp2[3]:
                    ans += [i,j]
        ans = set(ans)
        return [transactions[i] for i in ans]
