class Solution:
    def invalidTransactions(self, t: List[str]) -> List[str]:
        ans = []
        for ii in range(len(t)):
            i = t[ii]
            n,tim,amount,c=i.split(',')
            g = False
            if int(amount) > 1000:
                g = True
            tim = int(tim)
            
for jj in range(len(t)):
                if ii == jj:
                    continue
                j = t[jj]
                nn,ti,_,cc=j.split(',')
                if 0 <= abs(tim - int(ti)) <= 60 and n == nn and c != cc:
                    g = True
                    break
            if g:
                ans.append(i)
        return ans
