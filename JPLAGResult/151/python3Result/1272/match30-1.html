class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        ans = []
        transactions = [t.split(",") for t in transactions]
        for name, t, amt, city in transactions:
            if int(amt) > 1000:
                ans.append(','.join([name, t, amt, city]))
                continue
            for name2, t2, amt2, city2 in transactions:
                    if name == name2 and city != city2 and abs(int(t) - int(t2)) <= 60:
                        ans
.append(','.join([name, t, amt, city]))
                        break
        return ans
