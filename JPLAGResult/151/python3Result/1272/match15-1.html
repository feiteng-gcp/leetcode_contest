class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        ret = []
        n = len(transactions)
        t = [x.split(',') for x in transactions]
        for i in range(n):
            if int(t[i][2]) > 1000:
                ret.append(transactions[i])
                continue
            for j in range
(n):
                if i == j:
                    continue
                if t[i][0] == t[j][0] and t[i][3] != t[j][3] and abs(int(t[i][1])-int(t
[j][1])) <= 60:
                    ret.append(transactions[i])
                    break
        return ret
