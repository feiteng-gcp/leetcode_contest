class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        data = {}
        
        transactions.sort(key=lambda x: int(x.split(",")[1]))
        result = []
        res = set()
        for i, transaction in enumerate(transactions):
            tranaction_value = transaction.split(",")
            if int(tranaction_value[2]) > 1000:
                res.add(i)
            key = tranaction_value[0]
            data.setdefault(key, [])
            data[key].append((int(tranaction_value[1]), i, tranaction_value[3]))
            
        for k, v in data.items():
            v_len = len(v)
            if v_len == 1:
                continue
            for i in range(v_len-1):
                for j in range(i+1, v_len):
                    if (v[j][0] - v[i][0]) <= 60 and  v[j][2] != v[i][2]:
                        res.add(v[j][1])
                        res.add(v[i][1])
        
        return [transactions[i] for i in res]
