class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        res = set()
        box = []
        for t in transactions:
            tt 
= t.split(',')
            if int(tt[2]) > 1000:
                res.add(t)
            box.append(tt)
        for a in box:
            for b in box:
                if a[0] == b[0] and abs(int(a[1]) - int(b[1])) <= 60 and a[3] != b[3]:
                    res.add(','.join(a))
                    break
        return res
