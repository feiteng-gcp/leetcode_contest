from collections import defaultdict

class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        ans=[]
        for i in range(len(transactions)):
            t1=transactions[i]
            a1,b1,c1,d1=t1.split(',')
            if int(c1)>1000:
                ans.append(t1)
                continue
            for j in range(len(transactions)):
                if i==j:
                    continue
                t2=transactions[j]
                a2,b2,c2,d2=t2.split(',')
                if a1==a2 and d1!=d2 and abs(int(b1)-int(b2))<=60:
                    ans.append(t1)
                    break
        return ans
