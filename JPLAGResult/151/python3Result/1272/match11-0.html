class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        if not transactions:
            return []
        res = set()
        trans = [tran.split(',') for tran in transactions]
        trans.sort(key = lambda x: x[1])
        for i in range(len(trans)):
            if int(trans[i][2]) > 1000:
                res.add(','.join(trans[i]))
            for j in range(i):
                if trans[i][0] == trans[j][0] and trans[i][-1] != trans[j][-1] and abs(int(trans[i][1]) - int(trans[j][1])) <= 60:
                    res.add(','.join(trans[i]))
                    res.add(','.join(trans[j]))
        return list(res)
