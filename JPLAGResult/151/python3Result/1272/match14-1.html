from collections import defaultdict

class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        trans = defaultdict(list)
        invalids = set()
        for log in transactions:
            name, time, amount, city = log.split(',')
            amount = int(amount)
            time = int(time)
            
            if amount > 1000:
                invalids.add(log)
            
            trans[name].append((time, city, log))
        
        for name in trans:
            translist = trans[name]
            n = len(translist)
            for i in range(n):
                for j in range(i+1, n):
                    if abs(translist
[i][0] - translist[j][0]) <= 60 and translist[i][1] != translist[j][1]:
                        invalids.add(translist[i][2])
                        invalids.add(translist[j][2])
            
                
        return invalids
