class Solution:
    def invalid(self, t1, t2):
        if t1[0] == t2[0] and t1[3] != t2[3] and abs(int(t1[1]) - int(t2[1])) <= 60: return True
        return False
    
    def invalidTransactions(self, ts: List[str]) -> List[str]:
        trans = [t.split(',') for t in ts]
        ret = []
        for i in range(len(trans)):
            if int(trans[i][2]) > 1000:
                ret.append(",".join(trans[i]))
                continue
            for j in range(len(trans)):
                if i != j and self.invalid(trans[i], trans[j]):
                        ret.append(",".join(trans[i]))
                        break
        return ret
