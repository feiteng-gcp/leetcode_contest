class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        s = transactions
        t = []
        r = []
        for i in range(len(s)):
            a,b,c,d = s[i].split(',')
            if int(c) > 1000:
                r.append(i)
            for j in range(i+1,len(s)):
                a1,b1,c1,d1 = s[j].split(',')
                if abs(int(b1)-int(b)) <= 60 and a == a1 and d != d1:
                    r.append(i)
                    r.append(j)
        r = list(set(r))
        for i in range(len(r)):
            t.append(s[r[i]])
        return t
