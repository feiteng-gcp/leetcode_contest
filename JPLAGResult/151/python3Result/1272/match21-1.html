class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        hash_map = collections.defaultdict(list)
        res = set()
        for t in transactions:
            name, time, amount, city = t.split(',')
            hash_map[name].append((time, amount, city))
            for record in hash_map[name]:
                rec_time, rec_amount, rec_city = record
                if int(amount) >= 1000:
                    res.add(t)
                if abs(int(time) - int(rec_time)) <= 60 and city != rec_city:
                    res.add(t)
                    res.add(','.join([name, rec_time, rec_amount, rec_city]))
        return list(res)
