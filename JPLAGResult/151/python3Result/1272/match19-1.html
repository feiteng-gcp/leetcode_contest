class Solution:
    def invalidTransactions(self, transactions: 'List[str]') -> 'List[str]':
        "{name},{time},{amount},{city}"
        l=len(transactions)
        arr=[];ans=[]
        for i in range(0,l):
            arr.append(transactions[i].split(","))
        for i,t in enumerate(arr):
            if int(t[2])>1000:
                ans.append(transactions[i])
            else:
                
for j in range(len(arr)):
                    if i!=j and arr[j][0]==arr[i][0] and arr[j][3]!=arr[i][3] and abs(int(arr[j][1])-int(arr[i][1]))<=60:
                        ans.append(transactions[i])
                        break
        return ans
