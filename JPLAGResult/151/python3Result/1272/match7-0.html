class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        res = set()
        dic = collections.defaultdict(list)
        for transac in transactions:
            name, time, amount, city = transac.split(',')
            time, amount = int(time), int(amount)
            if amount > 1000:
                res.add(transac)
            dic[name].append([time, amount, city])
        for name in dic:
            lst = dic[name]
            for i in range(len(lst) - 1):
                for j in range(i + 1, len(lst)):
                    if lst[i][2] != lst[j][2] and abs(lst[i][0] - lst[j][0]) <= 60:
                        res.add(name + ',' + str(lst[i][0]) + ',' + str(lst[i][1]) + ',' + lst[i][2])
                        res.add(name + ',' + str(lst[j][0]) + ',' + str(lst[j][1]) + ',' + lst[j][2])
        return list(res)
