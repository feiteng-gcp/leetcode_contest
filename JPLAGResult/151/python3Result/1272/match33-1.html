class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        res = []
        for i,tx in enumerate(transactions):
            x,t,n,c = tx.split(',')
            t, n = int(t), int(n)
            if n > 1000:
                res.append(tx)
                continue
            
            for j,TX in enumerate(transactions):
                X,T,N,C = TX.split(',')
                T,N = int(T), int
(N)
                if X==x and C!=c and abs(T-t) <= 60:
                    res.append(tx)
                    break
                
        return res
