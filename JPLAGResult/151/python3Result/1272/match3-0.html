class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        res = set()
        l = len(transactions)
        for i in range(l):
            t = transactions[i]
            name, time, money, city = t.split(',')
            if int(money) > 1000:
                res.add(t)
                continue
            for j in range(l):
                na,ti,mo,ci = transactions[j].split(',')
                if name == na and abs(int(time) - int(ti)) <= 60 and city != ci:
                    res.add(t)
                    res.add(transactions[j])
                    break
        return list(res)
