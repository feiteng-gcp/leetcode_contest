class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        newt = []
        for t in transactions:
            t = t.split(",")
            newt.append([t[0], int(t[1]), int(t[2]), t[3]])
        ans = []
        for i in range(len(transactions)):
            if newt[i][2] > 1000:
                ans.append(transactions[i])
                continue
            for j in range(len(transactions)):
                if i == j:
                    continue
                if newt[i][0] == newt[j][0] and newt[i][3] != newt[j][3] and abs(newt[i][1] - newt[j][1]) <= 60:
                    ans.append(transactions[i])
                    break
        return ans
