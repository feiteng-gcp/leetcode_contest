class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        hashmap = collections.defaultdict(list)
        ans = set
()
        for tran in transactions:
            temp = tran.split(',')
            if int(temp[2]) > 1000:
                ans.add(tran)
            hashmap[temp[0]].append(temp[1:])
        for name in hashmap:
            temp = hashmap[name]
            for i in range(len(temp)):
                for j in range(i+1, len(temp)):
                    if temp[i][-1] != temp[j][-1] and abs(int(temp[i][0])-int(temp
[j][0])) <= 60:
                        ans.add(name + ',' + ','.join(temp[i]))
                        ans.add(name + ',' + ','.join(temp[j]))
        return list(ans)
