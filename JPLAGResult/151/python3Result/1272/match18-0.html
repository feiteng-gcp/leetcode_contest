class Solution:
    def invalidTransactions(self, a: List[str]) -> List[str]:
        n = len(a)
        b = []
        ret = []
        for i in range(n):
            name, time, amount, city = a[i].split(',')
            b.append([name, int(time), int(amount), city])
        for i in range(n):
            name, time, amount, city = b[i]
            if int(amount) > 1000:
                ret.append(a[i])
                continue
            for j in range(n):
                if name == b[j][0] and city != b[j][3] and abs(time-b[j][1]) <= 60:
                    ret.append(a[i])
                    break
        return ret
