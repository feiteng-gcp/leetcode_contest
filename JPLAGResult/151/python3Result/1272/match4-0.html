class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        invalid = set([])
        name_tr = collections.defaultdict(list)
        
        for i in transactions:
            temp = i.split(",")
            if int(temp[2]) > 1000:
                invalid.add(i)
            name_tr[temp[0]].append(temp)
        
        
        for i in name_tr.keys():
            temp = name_tr[i]
            #temp = sorted(temp, temp_cmp)
            #print(temp)
            temp.sort(key=lambda x: int(x[1]))
            for j in range(len(temp)):
                for k in range(j+1, len(temp)):
                    if (int(temp[k][1]) - int(temp[j][1]) <= 60) and (temp[j][3] != temp[k][3]):
                        invalid.add(",".join(temp[j]))
                        invalid.add(",".join(temp[k]))
        
        return invalid
