class Solution:
    def invalidTransactions(self, transactions: List[str]) -> List[str]:
        n=len(transactions)
        names={}
        for i in range(n):
            tokens=transactions[i].split(',')
            if tokens[0] not in names:
                names[tokens[0]]=[]
            names[tokens[0]].append([int(tokens[1]),tokens[3]])                        
        ret=[]
        for i in range(n):
            tokens=transactions[i].split(',')
            valid=True
            if int(tokens[2])>1000:
                valid=False
            if valid and tokens[0] in names:
                for (time,city) in names[tokens[0]]:
                    if abs(time-int(tokens[1]))<=60 and city!=tokens[3]:
                        valid=False
                        break
            if not valid:
                ret.append(transactions[i])
        return ret
