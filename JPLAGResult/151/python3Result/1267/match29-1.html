# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        cum_vals = []
        nodes = []
        
        cum_val = 0
        current = head
        while current is not None:
            cum_val += current.val
            # print(cum_val)
            # print(cum_vals)
            if cum_val == 0:
                cum_vals = []
                nodes = []
            elif cum_val in cum_vals:
                idx = cum_vals.index(cum_val)
                cum_vals = cum_vals[:idx+1] 
                nodes = nodes[:idx+1]
            
            else:
                cum_vals.append(cum_val)
                nodes.append(
current)
            current = current.next
        
        if not nodes:
            return None
        
        for i in range(len(nodes)- 1):
            nodes[i].next = nodes[i+1]
        
        nodes[-1].next = None
        return nodes[0]
