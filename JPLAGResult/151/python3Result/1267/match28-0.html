# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        s = []
        h = head
        while h:
            s.append(h.val)
            h = h.next
        res = []
        while 1:
            x = -1
            y = -1
            for i in range(len(s)):
                t = 0
                for j in range(i,len(s)):
                    t = t+s[j]
                    if t == 0:
                        x = i
                        y = j
                        break
                if x != -1:
                    break
            if x == -1:
                break
            for k in range(j-i+1):
                del s[i]
        q = ListNode(0)
        w = q
        for i in range(len(s)):
            q.next = ListNode(s[i])
            q = q.next
        return w.next
