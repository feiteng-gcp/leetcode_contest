# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        stack = []
        if not head:
            return None
        node = head
        while node != None:
            if node.val == 0:
                node = node.next
                continue
            if not stack:
                stack.append(node)
            else:
                sums = node.val
                i = len(stack) - 1
                while i >= 0 and sums != 0:
                    sums += stack[i].val
                    i -= 1
                if sums == 0:
                    #print(i, node.val)
                    while len(stack) > i + 1:
                        stack.pop()
                else:
                    stack.append(node)
            node = node.next
        #print(len(stack))
        if not stack:
            return None
        for i in range(len(stack) - 1):
            stack[i].next = stack[i+1]
        stack[-1].next = None
        return stack[0]
