# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None


class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        dummy = ListNode(sys.maxsize)
        dummy.next = head
        
        while True:
            remFlag = False
            idx = collections.defaultdict(ListNode)
            pre = dummy
            p = dummy.next
            cursum = 0
            while p:
                curval = p.val
                if curval == 0:
                    pre.next = p.next
                    remFlag = True
                    break
                
                if -p.val in idx:
                    # remove middle part
                    preNode = idx[-p.val]
                    preNode.next = p.next
                    remFlag = True
                    break
                
                cursum += p.val
                newidx = collections.defaultdict(ListNode)
                for key in idx:
                    newidx[key + curval] = idx[key]
                idx = newidx.copy()
                idx[curval] = pre
                
                pre = pre.next
                p = p.next
                
            if not remFlag:
                break
            
        return dummy.next
