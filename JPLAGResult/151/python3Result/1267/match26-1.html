# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        #tsum, node
        sumDic = {}       
        tmpHead = ListNode(0)
        tmpHead.next = head

        flag = 1
        while flag == 1:
            tsum 
= 0
            sumDic[0] = tmpHead
            flag = 0
            while head != None:
                tsum += head.val
                if tsum in sumDic.keys():
                    sumDic[tsum].next = head.next
                    head = head.next
                    flag = 1
                else:
                    sumDic[tsum] = head
                    head = head.next
        
        return tmpHead.next
