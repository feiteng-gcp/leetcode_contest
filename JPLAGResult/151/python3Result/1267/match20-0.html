# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        '''
        leverage cumsum.
        [1,2,3,-3,4]
        [1,3,6,3,7]
        
        [1,2,3,-3,-2]
        [1,3,6,3,1]
        
        when we spot two equal numbers, that means the segment in between can be cancelled (sum to 0)
        '''
        
        cumsum = {}
        front = ListNode(0)
        front.next = head
        cumsum[0] = front
        cur = 0
        change = False
        while head:
            cur += head.val
            if cur in cumsum:
                a = cumsum[cur]
                a.next = head.next
                del cumsum[cur]
                change = True
            else:
                cumsum[cur] = head
            head = head.next
        
        head = front.next
        
        if change:
            return self.removeZeroSumSublists(head)
        else:
            return head
