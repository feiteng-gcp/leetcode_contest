# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

from collections import OrderedDict

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        dummy_head = ListNode(0)
        dummy_head.next = head
        
        cumsum_to_node = OrderedDict()
        cumsum = 0
        node = dummy_head
        while node:
            cumsum += node.val
            if cumsum in cumsum_to_node:
                my_cumsum, my_node = cumsum_to_node.popitem(last=True)
                while my_cumsum != cumsum:
                    my_cumsum, my_node = cumsum_to_node.popitem(last=True)
                cumsum_to_node[cumsum] = my_node
                my_node.next = node.next
            else:
                cumsum_to_node[cumsum] = node
            node = node.next
        return dummy_head.next
