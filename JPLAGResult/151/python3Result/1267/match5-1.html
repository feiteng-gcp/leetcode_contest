# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        dummy = ListNode(0)
        dummy.next = head
        pos_map = {0:dummy}
        cur, cur_sum = head, 0
        while cur:
            cur_sum += cur.val
            if cur_sum in pos_map:
                pos_map
[cur_sum].next = cur.next
            else:
                pos_map[cur_sum] = cur
            cur = cur.next
        return dummy.next
