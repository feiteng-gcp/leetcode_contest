# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None


class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        
        nums = []
        temp = head
        while temp:
            nums.append(temp.val)
            temp = temp.next
            
        s = []
        for num in nums:
            s.append(num)
            total = 0
            for j in range(len(s)-1,-1,-1):
                total += s[j]
                if total == 0:
                    s = s[:j]
        if not s:
            return None
        head = ListNode(s[0])
        temp 
= head
        for i in range(1, len(s)):
            temp.next = ListNode(s[i])
            temp = temp.next
        return head
