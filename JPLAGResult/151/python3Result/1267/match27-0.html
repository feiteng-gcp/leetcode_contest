# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        helper = ListNode(0)
        helper.next = head
        s = {0: helper}
        m = 0
        cur = head
        while cur:
            m += cur.val
            if m in s:
                s[m].next = cur.next
            else:
                s[m] = cur
            cur = cur.next
        return helper.next
