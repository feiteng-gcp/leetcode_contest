class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        sum = 0
        dp = {}
        nav = ListNode(-1)
        nav.next = head
        dp[0] = nav

        cur = head
        while cur != None:
            sum += cur.val
            if sum in dp.keys():
                dp[sum].next = cur.next
            else:
                dp[sum] = cur
            cur = cur.next

        return nav.next
