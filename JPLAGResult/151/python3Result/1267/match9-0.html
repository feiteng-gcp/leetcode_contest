# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        dummy = ListNode(0)
        dummy.next = head
        dic = {0: dummy}
        presum = 0
        while head:
            presum += head.val
            if presum in dic:
                dic[presum].next = head.next
            else:
                dic[presum] = head
            head = head.next
        return dummy.next
