# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        nhead = ListNode(-1)
        nhead.next = head
        
        d = {0: nhead}
        
        nvalue = 0
        while head:
            nvalue +=  head.val
            if nvalue in d:
                d
[nvalue].next = head.next
            else:
                d[nvalue] = head
            head = head.next
        return nhead.next
