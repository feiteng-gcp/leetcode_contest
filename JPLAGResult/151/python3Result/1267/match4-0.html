# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        sum_dict = {}
        dummy = ListNode(0)
        dummy.next = head
        sum_dict[0] = dummy
        
        node = head
        cur_sum = 0
        while node != None:
            cur_sum += node.val
            if cur_sum in sum_dict:
                sum_dict[cur_sum].next = node.next
            else:
                sum_dict[cur_sum] = node 
            node = node.next 
            
        return dummy.next
