# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None


class Solution:
    def removeZeroSumSublists(self, head: ListNode) -> ListNode:
        l = []
        cur = head
        while cur:
            l += [cur.val]
            cur = cur.next
        res = []
        for i in l:
            res += [i]
            tmp = 0
            for j in range(len(res)-1,-1,-1):
                tmp += res[j]
                if tmp == 0:
                    res = res[:j]
                    break
        ans = []
        if not res:
            return None
        for i in res:
            ans += [ListNode(i)]
        for i in range(len(ans)-1):
            ans[i].next = ans[i+1]
        return ans[0]
