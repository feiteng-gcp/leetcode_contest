class Solution {
  public List<String> invalidTransactions(String[] transactions) {
        int n = transactions.length;
        boolean[] invalid = new boolean[n];

        List<String> ans = new ArrayList<>();
//        for (int i = 0; i < n; i++) {
//            String[] s = transactions[i].split(",");
//            int amount = Integer.parseInt(s[2]);
//            if (amount > 1000) {
//                invalid[i] = true;
//                ans.add(transactions[i]);
//            }
//        }

        for (int i = 0; i < n; i++) {
            if (invalid[i]) continue;
            String[] a = transactions[i].split(",");

            String name1 = a[0];
            int time1 = Integer.parseInt(a[1]);
            int amouunt1 = Integer.parseInt(a[2]);
            String city1 = a[3];

            
if (amouunt1 > 1000) {
                ans.add(transactions[i]);
                continue;
            }

            for (int j = 0; j < n; j++) {
                if (invalid[j] || i == j) continue;

                String[] b = transactions[j].split(",");
                String name2 = b[0];
                int time2 = Integer.parseInt(b[1]);
                int amouunt2 = Integer.parseInt(b[2]);
                String city2 = b[3];

                
if (name1.equals(name2) && Math.abs(time1 - time2) <= 60 && !city1.equals(city2)) {
                    ans.add(transactions[i]);
                    break;
                }
            }
        }
        return ans;
    }

}
