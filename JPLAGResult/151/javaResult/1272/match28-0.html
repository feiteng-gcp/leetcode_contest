	class Solution {
	    public List<String> invalidTransactions(String[] transactions) {
	    	int n = transactions.length;
	    	boolean[] ng = new boolean[n];
	    	
	    	String[] names = new String[n];
	    	int[] times = new int[n];
	    	int[] ams = new int[n];
	    	String[] poss = new String[n];
	    	for(int i = 0;i < n;i++){
	    		String t = transactions[i];
	        	String[] s = t.split(",");
	        	names[i] = s[0];
	        	times[i] = Integer.parseInt(s[1]);
	        	ams[i] = Integer.parseInt(s[2]);
	        	poss[i] = s[3];
	        }
	    	for(int i = 0;i < n;i++){
	    		if(ams[i] > 1000){
	    			ng[i] = true;
	    		}
	    		for(int j = i+1;j < n;j++){
	    			if(names[i].equals(names[j]) && 
	    					!poss[i].equals(poss[j]) &&
	    					Math.abs(times[i] - times[j]) <= 60){
	    				ng[i] = ng[j] = true;
	    			}
	    		}
	    	}
	    	
	    	List<String> ret = new ArrayList<>();
	    	for(int i = 0;i < n;i++){
	    		if(ng[i]){
	    			ret.add(transactions[i]);
	    		}
	    	}
	    	return ret;
	    }
	}
