class Solution {
    
    class Trans {
        String name;
        int time;
        int amount;
        String city;
        int idx;
        public Trans(String name, int time, int amount, String city, int idx) {
            this.name = name;
            this.time = time;
            this.amount = amount;
            this.city = city;
            this.idx = idx;
        }
    }
    
    public List<String> invalidTransactions(String[] transactions) {
        int N = transactions.length;
        HashMap<String, List<Trans>> map = new HashMap<>();
        List<String> res = new ArrayList<String>();
        for (int i = 0; i < N; i++) {
            String txn = transactions[i];
            String[] t = txn.split(",");
            String name = t[0];
            int time = Integer.valueOf(t[1]);
            int amount = Integer.valueOf(t[2]);
            String city = t[3];
            map.computeIfAbsent(name, a -> new ArrayList<>()).add(new Trans(name, time, amount, city, i));
        }
        for (String name : map.keySet()) {
            List<Trans> l = map.get(name);
            for (Trans t : l) {
                if (t.amount > 1000) {
                    res.add(transactions[t.idx]);
                } else {
                    for (Trans tt : l) {
                        if (Math.abs(tt.time - t.time) <= 60 && !tt.city.equals(t.city)) {
                            res.add(transactions[t.idx]);
                            break;
                        }
                    }
                } 
            }
        }
        return res;
    }
}
