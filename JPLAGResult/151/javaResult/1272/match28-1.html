class Tran {
    public String name;
    public int time;
    public int amount;
    public String city;
    
    public Tran(String s) {
        
String[] parts = s.split(",");
        this.name = parts[0];
        this.time = Integer.valueOf(parts[1]);
        this.amount = Integer.valueOf(parts[2]);
        this.city = parts[3];
    }
}

class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        int n = transactions.length;
        Set<String> set = new HashSet<>();
        List<Tran> trans = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            trans.add(new Tran(transactions[i]));
        }
        
        boolean[] invalid = new boolean[n];
        for (int i = 0; i < n; ++i) {
            if (trans.get(i).amount > 1000) {
                invalid[i] = true;
            }
            
            Tran a = trans.get(i);
            for (int j = i + 1; j < n; ++j) {
                Tran b = trans.get(j);
                
if (a.name.equals(b.name) && Math.abs(a.time - b.time) <= 60 && !a.city.equals(b.city)) {
                    invalid[i] = invalid[j] = true;
                }
            }
        }
        
        List<String> ret = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            if (invalid[i]) {
                ret.add(transactions[i]);
            }
        }
        return ret;
    }
}
