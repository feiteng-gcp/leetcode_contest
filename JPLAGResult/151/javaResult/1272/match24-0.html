class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        HashMap<String, List<Transaction>> records = new HashMap<>();
        
        List<String> ans = new ArrayList<String>();
        for(int i=0;i<transactions.length;i++) {
            String[] tokens = transactions[i].split(",");
            String name = tokens[0];
            int time = Integer.valueOf(tokens[1]);
            int amount = Integer.valueOf(tokens[2]);
            String city = tokens[3];
            String newName = name; //+ "ywdltql" + city;
            Transaction t = new Transaction(name,time,amount,city);
            if(!records.containsKey(newName)) {
                List<Transaction> list = new ArrayList<Transaction>();
                records.put(newName, list);
            }
            List<Transaction> list = records.get(newName);
            list.add(t);
            records.put(newName, list);
        }
        
        for (Map.Entry<String, List<Transaction>> entry : records.entrySet()) {
            String key = entry.getKey();
            List<Transaction> list = entry.getValue();
            ArrayList<Transaction> array = (ArrayList<Transaction>) list;
            Collections.sort(array);
            HashSet<Integer> invalid = new HashSet<>();
            for(int i=0;i<array.size();i++) {
                if(invalid.contains(i)) continue;
                Transaction t = array.get(i);
                if(t.amount > 1000) {
                    ans.add(t.toString());
                    invalid.add(i);
                    continue;
                }
                
                for(int j=0;j<array.size();j++) {
                    Transaction other = array.get(j);
                    if(other.time - t.time > 60) break;
                    if(!t.city.equals(other.city) && Math.abs(other.time-t.time) <= 60 ) {
                        if(!invalid.contains(i)) {
                            ans.add(t.toString());
                            invalid.add(i);
                        }
                        if(!invalid.contains(j)) {
                            ans.add(other.toString());
                            invalid.add(j);
                        }
                    }
                }
            }
        }
        
        return ans;
    }
    
    private class Transaction implements Comparable<Transaction> {
        String name;
        int time;
        int amount;
        String city;
        
        public Transaction(String n,int t,int a,String c) {
            this.name = n;
            this.time = t;
            this.amount = a;
            this.city = c;
        }
        
        public int compareTo(Transaction other) {
            return Integer.valueOf(this.time).compareTo(other.time);
        }
        
        public String toString() {
            return this.name + "," + this.time + "," + this.amount + "," + this.city;
        }
    }
}
