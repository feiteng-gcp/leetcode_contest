class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        
         ArrayList<String> res = new ArrayList<>();
        
        for(int i=0;i<transactions.length;i++)
        {
            String arr[] = transactions[i].split(",");
            if(Integer.parseInt(arr[2])>1000)
            {
                res.add(transactions[i]);
                continue;
            }
                
            for(int j=0;j<transactions.length;j++)
            {
                if(j!=i)
                {
                    
String arr1[] = transactions[j].split(",");
                    if(arr1[0].equals(arr[0]) && !arr1[3].equals(arr[3]) && Math.abs(Integer.parseInt(arr[1])-Integer.parseInt(arr1[1]))<=60)
                    {res.add(transactions[i]);break;}
                }
            }
        }
        
        
        return res;
    }
}
