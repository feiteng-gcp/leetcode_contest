class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        List<String> ret = new ArrayList<>();
        Set<String> set = new HashSet<>();
        int len = transactions.length;
        
        for (int i = 0; i < len; i++) {
            for (int j = i + 1; j < len; j++) {
                String[] temp1 = transactions[i].split(",");
                String[] temp2 = transactions[j].split(",");
                if (Integer.valueOf(temp1[2]) > 1000) set.add(transactions[i]);
                if (Integer.valueOf(temp2[2]) > 1000) set.add(transactions[j]);
                
                if (temp1[0].equals(temp2[0]) && !temp1[3].equals(temp2[3])) {
                    if (Math.abs(Integer.valueOf(temp1[1]) - Integer.valueOf(temp2[1])) <= 60) {
                        set.add(transactions[i]);
                        set.add(transactions[j]);
                    }
                }
            }
        }
        
        for (String tran: set) {
            ret.add(tran);
        }
        
        return ret;
    }
}
