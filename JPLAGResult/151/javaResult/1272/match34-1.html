class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        
Set<String> invalidTxns = new HashSet<>();
        for (int i = 0; i < transactions.length; ++i) {
            String[] parts1 = transactions[i].split(",");
            String name1 = parts1[0];
            int amount1 = Integer.parseInt(parts1[2]);
            int time1 = Integer.parseInt(parts1[1]);
            String cityName1 = parts1[3];
            if (amount1 > 1000) {
                invalidTxns.add(transactions[i]);
            }
            for (int j = i + 1; j < transactions.length; ++j) {
                String[] parts2 = transactions[j].split(",");
                String name2 = parts2[0];
                int amount2 = Integer.parseInt(parts2[2]);
                int time2 = Integer.parseInt(parts2[1]);
                String cityName2 = parts2[3];
                if (name1.equals(name2) && !cityName1.equals(cityName2) && 
Math.abs(time1 - time2) <= 60) {
                    invalidTxns.add(transactions[i]);
                    invalidTxns.add(transactions[j]);
                }
            }
        }
        return new ArrayList<>(invalidTxns);
    }
}
