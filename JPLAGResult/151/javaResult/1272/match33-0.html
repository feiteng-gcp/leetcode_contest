class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        Set<String> res = new HashSet<>();
        TreeMap<Integer, Set<String>> map = new TreeMap<>();
        for(String trans : transactions){
            String[] first = trans.split(",");
            int time = Integer.parseInt(first[1]), cost = Integer.parseInt(first[2]);
            String city = first[3];
            if(cost>1000){
                res.add(trans);
            }
            Map<Integer, Set<String>> subMap = map.subMap(time-60, time+61);
            for(Set<String> ltrans: subMap.values()){
                for(String tran: ltrans){
                    String[] s = tran.split(",");
                    if(!first[3].equals(s[3])&&first[0].equals(s[0])){
                        res.add(tran);
                        res.add(trans);
                    }
                }
            }
            map.putIfAbsent(time, new HashSet<>());
            map.get(time).add(trans);
        }
        return new ArrayList<>(res);
    }
}
