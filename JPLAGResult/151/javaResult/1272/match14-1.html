class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        List<String> result = new ArrayList<>();
        Set<String> set = new HashSet<>();
        Map<String, Set<Integer>> map = new HashMap<>();
        for (int i = 0; i < transactions.length; i++) {
            String[] ss = 
transactions[i].split(",");
            if (Integer.parseInt(ss[2]) > 1000) {
                set.add(transactions[i]);
            }
            
            if (map.containsKey(ss[0])) {
                for (int j : map.get(ss[0])) {
                    //System.out.println(j);
                    String[] old = transactions[j].split(",");
                    if (Math.abs(Integer.parseInt(ss[1]) - Integer.parseInt(old[1])) <= 60 && !ss[3].equals(old[3])) {
                        
set.add(transactions[i]);
                        set.add(transactions[j]);
                    }
                }
            } else {
                map.put(ss[0], new HashSet<Integer>());
            }
            
            
            map.get(ss[0]).add(i);
        }
        
        return new ArrayList<String>(set);
    }
}
