class Solution {
    public List<String> invalidTransactions(String[] arr) {
        int n = arr.length;
        if(n == 0) return new ArrayList();
        
        Set<Integer> set = new HashSet();
        HashMap<String, List<String[]>> map = new HashMap();
        for(int i = 0; i < n; i++) {
            String[] info = arr[i].split(",");
            map.putIfAbsent(info[0], new LinkedList());
            map.get(info[0]).add(new String[]{info[1], info[3], Integer.toString(i)});
            if(Integer.valueOf(info[2]) > 1000) set.add(i);
            for(String[] s_arr: map.get(info[0])) {
                if(s_arr[1].equals(info[3])) continue;
                if(Math.abs(Integer.valueOf(s_arr[0]) - Integer.valueOf(info[1])) <= 60) {
                    set.add(i);
                    set.add(Integer.valueOf(s_arr[2]));
                }
            }
        }
        
        List<String> res = new ArrayList();
        for(Integer id: set) {
            res.add(arr[id]);
        }
        return res;
    }
}
