class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        List<String> ans = new ArrayList<>();
        
for (int i = 0; i < transactions.length; i++) {
            String[] tokens = transactions[i].split(",");
            String name = tokens[0];
            int time = Integer.parseInt(tokens[1]);
            int amount = Integer.parseInt(tokens[2]);
            String city = tokens[3];
            
            if (amount > 1000) {
                ans.add(transactions[i]);
                continue;
            }
            
            for (String x : transactions) {
                String[] prevTokens = x.split(",");
                String prevName = prevTokens[0];
                
int prevTime = Integer.parseInt(prevTokens[1]);
                String prevCity = prevTokens[3];
                if (name.equals(prevName) && Math.abs(prevTime - time) <= 60 && !city.equals(prevCity)) {
                    ans.add(transactions[i]);
                    break;
                }
            }
        }
        return ans;
    }
}
