class Solution {
   public List<String> invalidTransactions(String[] transactions) {
       Map<String, List<String>> finder = new HashMap<>();
       Set<String> set = new HashSet<>();
       List<String> result = new ArrayList<>();
       for(String tran : transactions) {
           String[] con = tran.split(",");
           if(Integer.parseInt(con[2])>1000){
               set.add(tran);
           }
           
           if(finder.get(con[0])!=null) {
             for(String x : finder.get(con[0])){
                 String[] pre = x.split(",");
                 int time = Integer.parseInt(pre[1]);
                 String city = pre[3];
                 if(
Math.abs(Integer.parseInt(con[1]) - time) <=60 && !city.equals(con[3])) {
                     set.add(tran);
                     set.add(x);
                 }
             }   
           }
           else {
               finder.put(con[0], new ArrayList<>());
           }
           finder.get(con[0]).add(tran);
           
       }
       for(String x : set){
           result.add(x);
       }
       return result;
   }
}
