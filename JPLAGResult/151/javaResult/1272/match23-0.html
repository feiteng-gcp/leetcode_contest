class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        List<String> list = new ArrayList();
        Map<String,ArrayList<Trans>> map = new HashMap();
        for(String s:transactions){
            String[] arr = s.split(",");
            String name = arr[0];
            String time = arr[1];
            String amount = arr[2];
            String city = arr[3];
            if(map.containsKey(name)){
                map.get(name).add(new Trans(time,amount,city,s));
            }
            else{
                ArrayList<Trans> tmp = new ArrayList();
                tmp.add(new Trans(time,amount,city,s));
                map.put(name,tmp);
            }
        }
        
        for(String name:map.keySet()){
            ArrayList<Trans> tmp = map.get(name);
            int n = tmp.size();
            for(int i = 0;i < n;i++){
                if(Integer.parseInt(tmp.get(i).amount) > 1000){
                    list.add(tmp.get(i).content);
                    continue;
                }
                for(int j = 0;j < n;j++){
                    if(i == j){
                        continue;
                    }
                    if(!tmp.get(i).city.equals(tmp.get(j).city) && Math.abs(Integer.parseInt(tmp.get(i).time) - Integer.parseInt(tmp.get(j).time)) <= 60){
                        list.add(tmp.get(i).content);
                        break;
                    }
                }
            }
        }
        
        return list;
    }
    
    public class Trans{
        String time;
        String amount;
        String city;
        String content;
        
        Trans(String t,String a,String c,String con){
            time = t;
            amount = a;
            city = c;
            content = con;
        }
    }
}
