class Solution {
    public List<String> invalidTransactions(String[] t) {
        
        Set<String> li = new HashSet<>();
        int len  =t.length;
        for(int i=0;i<len;++i){
            String s = t[i];
            String y[]= s.split(",");
            int am = Integer.parseInt(y[2]);
            if(am>1000){
                li.add(s);
            }
            int tm = Integer.parseInt(y[1]);
            for(int j=0;j<len;++j){
                if(i==j) continue;
                String ss = t[j];
                String ys[]= ss.split(",");
                if(ys[0].equals(y[0])){
                   
                    int tm1 = Integer.parseInt(ys[1]);
            
                    if(Math.abs(tm1-tm)<=60&&!ys[3].equals(y[3])){
                        li.add(s);
                        break;
                    }
                }
            }
            
            
        }
        List<String> li1 = new ArrayList<>(li);
        return li1;
    }
}
