class Solution {

  public List<String> invalidTransactions(String[] transactions) {
    Map<String, List<String[]>> map = new HashMap<>();
    Set<String> invalid = new HashSet<>();
    String[] tx;
    for(int i = 0; i < transactions.length; ++i) {
      tx = transactions[i].split(",");
      if(Integer.parseInt(tx[2]) > 1000) {
        invalid.add(transactions[i]);
      }
      if(map.containsKey(tx[0])) {
        for(String[] t : map.get(tx[0])) {
          if(Math.abs(Integer.parseInt(t[1]) - Integer.parseInt(tx[1])) <= 60 && !t[3].equals(tx[3])) {
            invalid.add(String.join(",", t));
            invalid.add(transactions[i]);
          }
        }
      } else {
        map.put(tx[0], new ArrayList<>());
      }
      map.get(tx[0]).add(tx);
    }
    return new ArrayList<>(invalid);
  }
}
