class Solution {
 	privat
e static class Transaction {
		public String name;
		public int time;
		public int amount;
		public String city;
		public String string;		
		public Transaction(String string , String name , int time , int amount , String city) {
			this.string = string;
			this.name = name;
			this.time = time;
			this.amount = amount;
			this.city = city;			
		}
	}
	
	private boolean check(List<Transaction> list , int index) {
		if (list.get(index).amount > 1000) {
			return true;
		}
		for (int i = 0;i < list.size();i ++) {
			if (i != index) {
				int timeDiff = Math.abs(list.get(i).time - list.get(index).time);
				if (timeDiff <= 60 && list.get(i).name.equals(list.get(index).name) &&
						list.get(i).city.equals(list.get(index).city) == false) {
					return true;					
				}
			}
		}
		return false;
	}
	
    
public List<String> invalidTransactions(String[] transactions) {
    	List<Transaction> list = new ArrayList<>();
    	List<String> ans = new ArrayList<>();    	
    	for (String tran : transactions) {
    		String[] strs = tran.split(",");
    		String name = strs[0];
    		int time = Integer.parseInt(strs[1]);
    		int amount = Integer.parseInt(strs[2]);
    		String city = strs[3];
    		list.add(new Transaction(tran , name , time , amount , city));    		
    	}
    	for
 (int i = 0;i < list.size();i ++) {
    		if (check(list , i)) {
    			ans.add(list.get(i).string);
    		}
    	}
    	return ans;
    }
    
}
