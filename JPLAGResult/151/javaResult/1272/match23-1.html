class Solution {
    
class Transaction {
    	final int time;
    	final String city;
    	final int amount;
    	final String rep;
    	
    	Transaction(int time, String city, int amount, String rep) {
    		this.time = time;
    		this.city = city;
    		this.amount = amount;
    		this.rep = rep;
    	}
    }
    
public List<String> invalidTransactions(String[] transactions) {
        List<String> result = new ArrayList<>();
        Map<String, List<Transaction>> map = new HashMap<>();
        for (String t : transactions) {
        	String[] vals = t.split(",");
        	String name = vals[0];
        	Transaction tr = new Transaction(Integer.valueOf(vals[1]), vals[3], Integer.valueOf(vals[2]), t);
        	if (!map.containsKey(name)) {
        		map.put(name, new ArrayList<>());
        	}
        	map.get(name).add(tr);
        }
        for (Map.Entry<String, List<Transaction>> e : map.entrySet()) {
        	List<Transaction> list = e.getValue();
        	for (Transaction t : list) {
        		if (t.amount > 1000) {
        			result.add(t.rep);
        			continue;
        		}
        		for (Transaction t1 : list) {
        			if (!t1.city.equals(t.city) && Math.abs(t1.time - t.time) <= 60) {
        				result.add(t.rep);

        				break;
        			}
        		}
        	}
        }
        return result;
    }
}
