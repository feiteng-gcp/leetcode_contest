class Solution {
    class Info{
        int time;
        String name;
        String city;
        int amount;
        
        public Info(String name, int time, String city, int amount){
            this.time = time;
            this.name = name;
            this.city = city;
            this.amount = amount;
        }
        
        public String toString(){
            return name+","+time+","+amount+","+city;
        }
    }
    
    public List<String> invalidTransactions(String[] transactions) {
        Set<String> res = new HashSet<>();
        int n = transactions.length;
        
        List<Info> trans_info = new LinkedList<>();
        for(String trans: transactions){
            String[] info = trans.split(",");
            String name = info[0], city = info[3];
            int time = Integer.parseInt(info[1]), amount = Integer.parseInt(info[2]);
            trans_info.add(new Info(name, time, city, amount));
        }
        
        for(int i = 0; i < n; i++){
            Info info1 = trans_info.get(i);
            if(info1.amount > 1000) res.add(info1.toString());
            for(int j = i+1; j < n; j++){
                Info info2 = trans_info.get(j);
                if(info1.name.equals(info2.name)){
                    if(!info1.city.equals(info2.city) && Math.abs(info1.time - info2.time) <= 60){
                        res.add(info1.toString());
                        res.add(info2.toString());
                    }
                }
            }
        }
        return new LinkedList<>(res);
    }
}
