class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        Map<String, List<String[]>> map = new HashMap<>();
        for (String rawTransaction : transactions) {
            String[] transaction = rawTransaction.split(",");
            List<String[]> occurences = map.getOrDefault(transaction[0], new ArrayList<>());
            occurences.add(new String[]{transaction[1], transaction[3]});
            map.put(transaction[0], occurences);
        }
        List<String> ans = new ArrayList<>();
        for (String rawTransaction : transactions) {
            String[] transaction = rawTransaction.split(",");
            int curTime = Integer.parseInt(transaction[1]);
            if (Integer.parseInt(transaction[2]) > 1000) {
                ans.add(rawTransaction);
            } else {
                List<String[]> occurences = map.getOrDefault(transaction[0], new ArrayList<>());
                for (String[] occurence : occurences) {
                    int occTime = Integer.parseInt(occurence[0]);
                    if (!occurence[1].equals(transaction[3]) && Math.abs(curTime - occTime) <= 60) {
                        ans.add(rawTransaction);
                        break;
                    }
                }
            }
        }
        return ans;
    }
}
