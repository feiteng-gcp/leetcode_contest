class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        Map<String, Map<String, List<Integer[]>>> map = new HashMap<>();
        Set<Integer> answer = new HashSet<>();
        for(int i = 0; i < transactions.length; i++){
            String t = transactions[i];
            String[] res = t.split(",");
            String name = res[0];
            int time = Integer.parseInt(res[1]);
            int amount = Integer.parseInt(res[2]);
            String city = res[3];
            if(!map.containsKey(name)){
                map.put(name, new HashMap<>());
            }
            for(String city2 : map.get(name).keySet()){
                for(Integer[] nums : map.get(name).get(city2)){
                    int time2 = nums[0];
                    if(Math.abs(time2 - time) <= 60 && !city.equals(city2)){
                        answer.add(nums[1]);
                        answer.add(i);
                    }
                }
            }
            if(amount > 1000){
                answer.add(i);
            }
            if(!map.get(name).containsKey(city)) map.get(name).put(city, new ArrayList<>());
            map.get(name).get(city).add(new Integer[]{time, i});
        }
        List<String> r = new ArrayList<>();
        for(int i : answer){
            r.add(transactions[i]);
        }
        return r;
    }
}
