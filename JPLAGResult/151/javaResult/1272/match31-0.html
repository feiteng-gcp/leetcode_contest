class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        List<String> res = new ArrayList<>();
        Map<String, List<String[]>> map = new HashMap<>();
        
        for(String trans : transactions) {
            String[] strs = trans.split(",");

            String name = strs[0];
            
            if(!map.containsKey(name)) {
                map.put(name, new ArrayList<>());
            }
            
            map.get(name).add(strs);
        }
        
        for(String key : map.keySet()) {
            List<String[]> list = map.get(key);
            Collections.sort(list, (a, b) -> (Integer.valueOf(a[1]) - Integer.valueOf(b[1])));
            
            for(int i = 0; i < list.size(); i++) {
                if( Integer.valueOf(list.get(i)[2]) > 1000) {
                    res.add(String.join(",", list.get(i)));
                    continue;
                }
                
                boolean flag = false;
                
                for(int j = i - 1; j >= 0; j--) {
                    if(Integer.valueOf(list.get(i)[1]) - Integer.valueOf(list.get(j)[1]) <= 60 ) {
                        if(!list.get(i)[3].equals(list.get(j)[3])) {
                            flag = true;
                            res.add(String.join(",", list.get(i)));
                            break;
                        }
                    } else {
                        break;
                    }
                }
                
                if(flag == true) {
                    continue;
                }
                
                for(int j = i + 1; j < list.size() ; j++) {
                    if(Integer.valueOf(list.get(j)[1]) - Integer.valueOf(list.get(i)[1]) <= 60 ) {
                        if(!list.get(i)[3].equals(list.get(j)[3])) {
                            flag = true;
                            res.add(String.join(",", list.get(i)));
                            break;
                        }
                    } else {
                        break;
                    }
                }
            }
        }
        
        return res;
    }
}
