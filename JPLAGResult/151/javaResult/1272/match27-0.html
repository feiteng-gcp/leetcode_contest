class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        Map<String, TreeMap<Integer, Set<Node>>> map = new HashMap<>();
        Set<String> set = new HashSet<>();
        
        for (String transaction : transactions) {
            String[] strArr = transaction.split(",");
            String name = strArr[0];
            int time = Integer.parseInt(strArr[1]);
            int amount = Integer.parseInt(strArr[2]);
            String city = strArr[3];
            
            if (amount > 1000) {
                set.add(transaction);
            }
            
            if (map.get(name) != null) {
                TreeMap<Integer, Set<Node>> treeMap = map.get(name);

                // if (treeMap.get(time) != null) {
                //     for (Node node : treeMap.get(time)) {
                //         if (!node.city.equals(city)) {
                //             set.add(transaction);
                //             set.add(node.transaction);
                //         }
                //     }
                // }
                
                Map<Integer, Set<Node>> subMap = treeMap.subMap(time - 60, true, time + 60, true);
                for (int key : subMap.keySet()) {
                    for (Node node : subMap.get(key)) {
                        if (!node.city.equals(city)) {
                            set.add(transaction);
                            set.add(node.transaction);
                        }
                    }
                }
                
//                 if (treeMap.floorKey(time) != null && time - treeMap.floorKey(time) <= 60) {
//                     for (Node node : treeMap.get(treeMap.floorKey(time))) {
                        
//                     }
//                 }
            
//                 if (treeMap.ceilingKey(time) != null && treeMap.ceilingKey(time) - time <= 60) {
//                     for (Node node : treeMap.get(treeMap.ceilingKey(time))) {
//                         if (!node.city.equals(city)) {
//                             set.add(transaction);
//                             set.add(node.transaction);
//                         }
//                     }
//                 }
            }
            
            if (map.get(name) == null) {
                map.put(name, new TreeMap<>());
            }
            if (map.get(name).get(time) == null) {
                map.get(name).put(time, new HashSet<>());
            }
            map.get(name).get(time).add(new Node(name, time, amount, city, transaction));
        }
        
        List<String> res = new ArrayList<>();
        res.addAll(set);
        return res;
    }
}

class Node {
    public String name;
    public int time;
    public int amount;
    public String city;
    public String transaction;
    
    public Node(String name, int time, int amount, String city, String transaction) {
        this.name = name;
        this.time = time;
        this.amount = amount;
        this.city = city;
        this.transaction = transaction;
    }
}
