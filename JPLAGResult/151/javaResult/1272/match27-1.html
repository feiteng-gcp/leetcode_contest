class Solution {
    class Item {
        String name;
        
int time;
        int amount;
        String city;
        boolean valid;
        String tran;
        Item(String name, int time, int amount, String city, String tran) {
            this.name = name;
            this.time = time;
            this.amount = amount;
            this.city = city;
            valid = true;
            this.tran = tran;
        }
    }
    public List<String> invalidTransactions(String[] t) {
        Set<String> res = new HashSet<>();
        
Map<String, List<Item>> map = new HashMap<>();
        List<Item> trans = new ArrayList<>();
        for (String tran : t) {
            String[] info = tran.split(",");
            trans.add(new Item(info[0], Integer.valueOf(info[1]), Integer.valueOf(info[2]), info[3], tran));
        }
        Collections.sort(trans, (a, b) -> (a.time - b.time));
        for (Item i : trans) {
            
if (i.amount > 1000) {
                res.add(i.tran);
            }
            if (map.containsKey(i.name)) {
                List<Item> last = map.get(i.name);
                for (Item j : last) 
{
                    if (!i.city.equals(j.city) && i.time - j.time <= 60) {
                        res.add(i.tran);
                        res.add(j.tran);
                    }
                }
                last.add(i);
            } else {
                map.put(i.name, new ArrayList<>());
                map.get(i.name).add(i);
            }
        }
        List<String> resList = new ArrayList<>(res);
        return resList;
    }
}
