class Solution {
    public List<String> invalidTransactions(String[] tr) {
        int n = tr.length;//{name},{time},{amount},{city}
        Map<String, Integer> map = new HashMap<>();
        Set<Integer> set = new HashSet<>();
        for(int i = 0; i < n; i++){
            String[] res1 = tr[i].split(",");
            String name1 = res1[0];
            int time1 = Integer.parseInt(res1[1]);
            int amount1 = Integer.parseInt(res1[2]);
            String city1 = res1[3];
            if(amount1 > 1000){
                set.add(i);
            }
            for(int j = i + 1; j < n; j++){
                
String[] res2 = tr[j].split(",");
                String name2 = res2[0];
                int time2 = Integer.parseInt(res2[1]);
                int amount2 = Integer.parseInt(res2[2]);
                String city2 = res2[3];
                if(name1.equals(name2) && !
city1.equals(city2) && Math.abs(time1 - time2) <= 60){
                    set.add(i);
                    set.add(j);
                }
            }
        
}
        List<String> list = new ArrayList<>();
        for(int i : set){
            list.add(tr[i]);
        }
        return list;
    }
}
