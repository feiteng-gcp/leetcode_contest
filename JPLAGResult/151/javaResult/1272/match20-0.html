class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        List<String> ans = new ArrayList<>();

        int N = transactions.length;
       

        for (int i = 0; i < N; i++) {
        	String[] aa =  transactions[i].split(",");
        	String nameA = aa[0];
        	int timeA = Integer.parseInt(aa[1]);
        	int accontA = Integer.parseInt(aa[2]);
        	String cityA = aa[3];



        	for (int j = 0; j < N; j++) {
        		if (j == i) continue;


        		String[] bb =  transactions[j].split(",");
        		String nameB = bb[0];
        		int timeB = Integer.parseInt(bb[1]);
        		int accontB = Integer.parseInt(bb[2]);
        		String cityB = bb[3];

        		if (accontA > 1000 || (nameA.compareTo(nameB) == 0 && cityA.compareTo(cityB) != 0 && (int)Math.abs(timeA-timeB) <= 60)) {
        			ans.add(transactions[i]);
        			break;
        		}

        	}
        }
        return ans;
    }
}
