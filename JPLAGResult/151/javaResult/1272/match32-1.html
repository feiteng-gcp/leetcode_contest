class Solution {
    public List<String> invalidTransactions(String[] transactions) {
        Set<String> res = new HashSet<>();
        if(transactions == null || transactions.length == 0) return new ArrayList<>();
        
for(String t : transactions) {
            String[] data = t.split(",");
            if(Integer.valueOf(data[2]) > 1000) res.add(t);
            for(String k : transactions) {
                String[] data2 = k.split(",");
                if(data[0].equals(data2[0]) && Math.abs(Integer.valueOf(data[1]) - Integer.valueOf(data2[1])) <= 60 && !data[3].equals(data2[3])) {
                    res.add(t);
                    res.add(k);
                
}
            }
        }
        return new ArrayList<>(res);
    }
}
