class Solution {
    public int[] numSmallerByFrequency(String[] q, String[] words) {
        int[] qArr = new int[q.length];
        for (int i = 0; i < q.length; i++) {
            qArr[i] = func(q[i]);
        }
        
        int[] wCount = new int[12];
        for (int i = 0; i < words.length; i++) {
            int temp = func(words[i]);
            wCount[temp]++;
        }
        for (int i = 10; i >= 0; i--) {
            wCount[i] += wCount[i + 1];
        }
        
        int[] ans = new int[q.length];
        for (int i = 0; i < q.length; i++) {
            ans[i] = wCount[qArr[i] + 1];
        }
        return ans;
    }
    
    int func(String s) {
        char[] arr = s.toCharArray();
        int n = arr.length;
        char minChar = 'z';
        for (char c : arr) {
            if (c < minChar) {
                minChar = c;
            }
        }
        int count = 0;
        for (char c : arr) {
            if (c == minChar) {
                count++;
            }
        }
        return count;
    }
}
