class Solution {
    public int[] numSmallerByFrequency(String[] queries, String[] words) {
        int len1 = queries.length;
        int len2 = words.length;
        
        int[] arr1 = new int[len1];
        int[] arr2 = new int[len2];
        
int[] ret = new int[len1];
        
        for (int i = 0; i < len1; i++) {
            arr1[i] = help(queries[i]);
        }
        
        for (int i = 0; i < len2; i++) {
            arr2[i] = help(words[i]);
        }
        
        Map<Integer,Integer> map = new HashMap<>();
        
        for (int i = 0; i < len1; i++) {
            if (map.containsKey(arr1[i])) {
                ret[i] = map.get(arr1[i]);
            } else {
                
int count = 0;
                for (int j = 0; j < len2; j++) {
                    if (arr1[i] < arr2[j]) count++;
                }
                ret[i] = count;
                map.put(arr1[i], count);
            }
        }
        
        return ret;
     }
    
    
public int help(String str) {
        int[] counts = new int[26];
        for (Character cha: str.toCharArray()) {
            counts[cha - 'a']++;
        }
        
        for (int i = 0; i < 26; i++) {
            if (counts[i] != 0) return counts[i];
        }
        
        return 0;
    }
}
