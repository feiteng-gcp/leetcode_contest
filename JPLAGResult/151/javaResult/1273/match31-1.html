class Solution {
    public int[] numSmallerByFrequency(String[] queries, String[] words) {
        int ql = queries.length, wl=words.length;
        int[] fq = new int[ql];
        int[] wq = new int[wl];
        int[] res =  new int[ql];
        for(int i=0;i<ql;++i){
            fq[i]=getF(queries[i]);
        }
        for(int i=0;i<wl;++i){
            wq[i]=getF(words[i]);
        }
        
        for(int i=0;i<ql;++i){
            for(int j=0;j<wl;++j){
                if(fq[i]<wq[j]){
                    ++res[i];
                }
            }
            
        
}
        
        return res;
    }
    
    public int getF(String w){
        int[] counts = new int[26];
        for(char c:w.toCharArray()){
            ++counts[c-'a'];
        }
        for(int c:counts){
            if (c>0){
                return c;
            }
        }
        
        throw null;
    }
}
