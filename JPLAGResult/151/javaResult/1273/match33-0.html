class Solution {
    public int[] numSmallerByFrequency(String[] queries, String[] words) {
        int len = queries.length;
        int[] res = new int[len];
        for (int i = 0; i < len; i++) {
            res[i] = count(queries[i], words);
        }
        return res;
    }
    
    private int count(String q, String[] words) {
        int sum = 0;
        for (String w : words) {
            if (getf(q) < getf(w)) {
                sum++;
            }
        }
        return sum;
    }
    
    private int getf(String str) {
        int[] letters = new int[26];
        for (char c : str.toCharArray()) {
            letters[c - 'a']++;
        }
        for (int i = 0; i < 26; i++) {
            if (letters[i] > 0) {
                return letters[i];
            }
        }
        return 0;
    }
}
