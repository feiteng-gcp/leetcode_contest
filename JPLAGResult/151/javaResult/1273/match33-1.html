class Solution {
    public int[] numSmallerByFrequency(String[] queries, String[] words) {
        int n = words.length;
        int[] x = new int[n];
        for (int i = 0; i < n; ++i) {
            x[i] = p(words[i]);
        }
        
        Arrays.sort(x);
        int m = queries.length;
        int[] res = new int[m];
        
        for (int i = 0; i < m; ++i) {
            int q = p(queries[i]);
            int gg = bs(x, q);
            res[i] = gg;
        
}
        return res;
    }
    
    private int p(String s) {
        int[] cnt = new int[26];
        for (char c: s.toCharArray()) {
            cnt[c - 'a']++;
        }
        for (int i = 0; i < 26; ++i) {
            if (cnt[i] != 0) {
                return cnt[i];
            }
        }
        return -1;
    }
    
    private int bs(int[] nums, int target) {
        int l = 0 , r = nums.length - 1;
        while (l < r) {
            int mid = l + r >> 1;
            if (nums[mid] <= target) l = mid + 1;
            else r = mid;
        }
        if (nums[l] > target) return nums.length - l;
        return 0;
    }
}
