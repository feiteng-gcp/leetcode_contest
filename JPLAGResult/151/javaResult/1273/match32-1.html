class Solution {
    public int[] numSmallerByFrequency(String[] queries, String[] words) {
        int n = words.length;
        
int[] fList = new int[n];
        for (int i = 0; i < n; i++) {
            fList[i] = f(words[i]);
        }
        
        int[] ans = new int[queries.length];
        for (int i = 0; i < queries.length; i++) {
            int fValue = f(queries[i]);
            
for (int j = 0; j < n; j++) {
                if (fValue < fList[j]) {
                    ans[i]++;
                }
            }
        
}
        return ans;
    }

    int f(String s) {
        char[] chars = s.toCharArray();
        Arrays.sort(chars);

        int i = 0;
        while (i < chars.length && chars[i] == chars[0]) i++;
        return i;
    }
}
