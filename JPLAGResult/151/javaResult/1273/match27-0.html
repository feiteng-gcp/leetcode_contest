	class Solution {
	    public int[] numSmallerByFrequency(String[] queries, String[] words) {
	    	int n =queries.length;
	    	int m = words.length;
	        int[] ws = new int[m];
	        for(int i = 0;i < m;i++)ws[i] = f(words[i]);
	        
	        int[] ret = new int[n];
	        for(int i = 0;i < n;i++){
	        	int u = f(queries[i]);
	        	for(int j = 0;j < m;j++){
	        		if(u < ws[j]){
	        			ret[i]++;
	        		}
	        	}
	        }
	        return ret;
	    }
	    
	    public int f(String s)
	    {
	    	int[] f = new int[26];
	    	for(char c : s.toCharArray()){
	    		f[c-'a']++;
	    	}
	    	for(int i = 0;i < 26;i++){
	    		if(f[i] > 0)return f[i];
	    	}
	    	return 0;
	    }
	}
