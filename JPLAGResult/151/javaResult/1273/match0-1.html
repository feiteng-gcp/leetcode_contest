class Solution {
    
    public int f(String s){
        int d[] = new int[26];
        for(char c:s.toCharArray()){
            d[c-'a']++;
        }
        for(int i=0;i<26;++i){
            if(d[i]>0){
                return d[i];
            }
        }
        return -1;
        
    }
    
    public int[] numSmallerByFrequency(String[] qs, String[] ws) {
        int res[] = new int[qs.length];
        int x =0;
        for(String q: qs){
            int fq = f(q);
            int r = 0;
            for(String w:ws){
                int fw = f(w);
                if(fq<fw){
                    r++;
                }
            }
            res[x++] =r;
            
            
        }
        return res;
    }
}
