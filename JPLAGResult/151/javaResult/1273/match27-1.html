class Solution {
    public int[] numSmallerByFrequency(String[] queries, String[] words) {
        int[] sum = new int[15];
        for(String w : words)
        {
            int ff = f(w);
            // map.put(ff, map.getOrDefault(ff, 0) + 1);
            sum[ff]++;
        }
        int n = queries.length;
        
int[] re = new int[n];
        for(int i = 0; i < n; i++)
        {
            int q = f(queries[i]);
            for(int j = q + 1; j <= 10; j++) re[i] += sum[j]
;            
        }
        return re;
    }
    
    int f(String str)
    {
        int[] g = new int[26];
        for(char ch : str.toCharArray())
        {
            g[ch - 'a']++;
        }
        for(int i = 0; i < 26; i++)
        {
            if(g[i] > 0) return g[i];
        }
        return 0;
    }
}
