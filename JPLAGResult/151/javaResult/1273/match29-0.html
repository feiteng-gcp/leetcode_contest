class Solution {
    public int[] numSmallerByFrequency(String[] queries, String[] words) {
        int N = queries.length;
        int M = words.length;

        int[] len = new int[M];
        for (int i = 0; i < M; i++) {
        	len[i] = f(words[i]);

        }
        int[] ans = new int[N];
        Arrays.sort(len);
        for (int i = 0; i < N; i++) {
        	int cnt = 0;
        	int val = f(queries[i]);
        	for (int j = M-1; j >= 0; j--) {
        		if (val < len[j]) cnt++;
        		else break;
        	}
      		ans[i] = cnt;
        }
        return ans;
    }
    private int f(String word) {
    	int[] cnt = new int[26];
    	for (char ch : word.toCharArray()) {
    		cnt[ch - 'a']++;
    	}
    	for (int i = 0; i < 26; i++) {
    		if (cnt[i] > 0) return cnt[i];

    	}
    	return 0;
    }
}
