class DinnerPlates {
    
List<Deque<Integer>> list;
    int capacity;
    int leftMost;
    public DinnerPlates(int capacity) {
        this.capacity = capacity;
        list = new ArrayList<>();
        
leftMost = 0;
    }
    
    public void push(int val) {
        int index = -1;
        for (int i = leftMost; i < list.size(); i++) {
            Deque<Integer> curStack = list.get(i);
            
if (curStack.size() < capacity) {
                index = i;
                break;
            }
        }
        if (index == -1) {
            Deque<Integer> temp = new ArrayDeque<>();
            temp.push(val);
            list.add(temp);
            leftMost = list.size() - 1;
        } else {
            list.get(index).push(val);
            leftMost = index;
        }
    }
    
    public int pop() {
        while (list.size() > 0) {
            Deque<Integer> temp = list.get(list.size() - 1);
            if (!temp.isEmpty()) {
                return temp.pop();
            } else {
                list.remove(list.size() - 1);
            }
        }
        return -1;
    }
    
    public int popAtStack(int index) {
        if (index < list.size()) {
            Deque<Integer> temp = list.get(index);
            leftMost = Math.min(leftMost, index);
            if (!temp.isEmpty()) {
                return temp.pop();
            }
        }
        return -1;
    }
}
