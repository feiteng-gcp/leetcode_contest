import java.math.*;
import java.util.*;
import java.io.*;

class DinnerPlates {
    
    int total = 0, cap = 0;
    TreeMap<Integer, Stack<Integer>> map = new TreeMap<>();
    // Map<Integer, Integer> idx = new HashMap<>();
    TreeSet<Integer> toPop = new TreeSet<>(), toPush = new TreeSet<>();
    
    public DinnerPlates(int capacity) {
        cap = capacity;
    }
    
    public void push(int val) {
        int index = -1;
        if(toPush.size() < 1)    // add next one
        {
            map.put(total, new Stack<>());
            // idx.put(total, 0);
            index = total;
            total++;
        }
        else
        {
            // left most
            index = toPush.first();
        }
        map.get(index).push(val);
        if(map.get(index).size() == cap)
        {
            toPush.remove(index);
        }
        else toPush.add(index);
        toPop.add(index);
    }
    
    public int pop() {
        // pop from last
        if(toPop.isEmpty()) return -1;
        int idx = toPop.last();

        return popAtStack(idx);
    }
    
    public int popAtStack(int index) {
        if(map.get(index).isEmpty()) return -1;
        int re = map.get(index).pop();
        if(map.get(index).isEmpty())
        {
            toPop.remove(index); 
        }
        toPush.add(index);
        return re;
    }
}
