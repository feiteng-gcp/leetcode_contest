class DinnerPlates {

    private int cap;
    private List<Integer> size;
    private List<int[]> stack;
    private TreeSet<Integer> avail;
    public DinnerPlates(int capacity) {
        size = new ArrayList<>();
        stack = new ArrayList<>();
        avail = new TreeSet<>();
        cap = capacity;
    
}
    
    public void push(int val) {
        if (avail.size() > 0) {
            int first = avail.first();
            int k = size.get(first);
            stack.get(first)[k] = val;
            if (++k == cap) {
                avail.remove(first);
            }
            size.set(first, k);
        }
        else if (cap == 1) {
            stack.add(new int[]{val});
            size.add(1);
        }
        else {
            avail.add(stack.size());
            int[] arr = new int[cap];
            int k = 0;
            arr[k++] = val;
            stack.add(arr);
            size.add(k);
        
}
    }
    
    public int pop() {
        if (stack.size() == 0) {
            return -1;
        }
        int[] arr = stack.get(stack.size() - 1);
        int k = size.get(size.size() - 1);
        if (k == 1) {
            stack.remove(stack.size() - 1);
            size.remove(size.size() - 1);
            avail.remove(stack.size());
        }
        else 
{
            size.set(size.size() - 1, k - 1);
            avail.add(stack.size() - 1);
        }
        return arr[k - 1];
    }
    
    public int popAtStack(int index) {
        if (stack.size() <= index) return -1;
        int k = size.get(index);
        if (k == 0) return -1;
        int[] arr = stack.get(index);
        int val = arr[k - 1];
        if (index == stack.size() - 1 && k == 1) {
            stack.remove(stack.size() - 1);
            size.remove(size.size() - 1);
            avail.remove(stack.size());
        }
        else {
            size.set(index, k - 1);
            avail.add(index);
        }
        return val;
    }
}
