class DinnerPlates {
    int count;
    int capacity;
    List<Stack<Integer>> list;
    TreeSet<Integer> nonEmptySet;
    TreeSet<Integer> nonFullSet;
    
    public DinnerPlates(int capacity) {
        count = 0;
        this.capacity = capacity;
        list = new ArrayList<>();
        nonEmptySet = new TreeSet<>();
        
nonFullSet = new TreeSet<>();
    }
    
    public void push(int val) {
        if (nonFullSet.isEmpty()) {
            list.add(new Stack<>());
            nonFullSet.add(list.size() - 1);
        }
        int first = nonFullSet.first();
        Stack<Integer> stack = list.get(first);
        count++;
        if (stack.size() == 0) {
            nonEmptySet.add(first);
        }
        if (stack.size() == capacity - 1) {
            nonFullSet.remove(first);
        }
        stack.push(val);
    }
    
    public int pop() {
        if (count == 0) {
            return -1;
        }
        int last = nonEmptySet.last();
        return popAtStack(last);
    
}
    
    public int popAtStack(int index) {
        if (index >= list.size()) {
            return -1;
        }
        Stack<Integer> stack = list.get(index);
        if (stack.size() == 0) {
            return -1;
        }
        count--;
        if (stack.size() == capacity) {
            nonFullSet.add(index);
        }
        
if (stack.size() == 1) {
            nonEmptySet.remove(index);
        }
        return stack.pop();
    }
}
