class DinnerPlates {
    TreeSet<Integer> notFull = new TreeSet<>();
    TreeSet<Integer> notEmpty = new TreeSet<>();
    int cap = 0;
    Map<Integer, Stack<Integer>> map = new HashMap<>();

    public DinnerPlates(int capacity) {
        cap = capacity;
    }
    
    public void push(int val) {
        if (notFull.isEmpty()) {
            int index = map.size();
            map.put(index, new Stack<>());
            
notFull.add(index);
        }
        int index = notFull.first();
        Stack<Integer> stk = map.get(index);
        stk.push(val);
        if (stk.size() == cap) {
            notFull.remove(index);
        }
        notEmpty.add(index);
    }
    
    public int pop() {
        if (notEmpty.isEmpty()) return -1;
        int index = notEmpty.last();
        Stack<Integer> stk = map.get(index);
        int v = stk.pop();
        if (stk.isEmpty()) notEmpty.remove(index);
        notFull.add(index);
        return v;
    }
    
    public int popAtStack(int index) {
        if (!map.containsKey(index) || !notEmpty.contains(index)) return -1;
        Stack<Integer> stk = map.get(index);
        int v = stk.pop();
        if (stk.isEmpty()) notEmpty.remove(index);
        notFull.add(index);
        return v;
    }
}
