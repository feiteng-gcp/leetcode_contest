import java.util.BitSet;
import java.util.Stack;
import java.util.TreeMap;

class DinnerPlates {
    int capacity;
    TreeMap<Integer, Stack<Integer>> map = new TreeMap<>();
    BitSet bitSet = new BitSet();

    public DinnerPlates(int capacity) {
        this.capacity = capacity;

    }

    public void push(int val) {
        int index = this.bitSet.nextClearBit(0);
        if(!this.map.containsKey(index)) {
            this.map.put(index, new Stack<>());
        }
        Stack<Integer> stack = this.map.get(index);
        stack.push(val);
        if(stack.size() == this.capacity) {
            this.bitSet.set(index);
        }
    }

    public int pop() {
        if(this.map.isEmpty()) {
            return -1;
        }
        int index = this.map.lastKey();
        return this.popAtStack(index);
    }

    public int popAtStack(int index) {
        if(!this.map.containsKey(index)) {
            return -1;
        }
        Stack<Integer> stack = this.map.get(index);
        int ret = stack.pop();
        if(stack.empty()) {
            this.map.remove(index);
        }
        this.bitSet.clear(index);
        return ret;
    }
}
