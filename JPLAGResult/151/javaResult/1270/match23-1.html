class DinnerPlates {
    int cap;
    List<List<Integer>> stacks;
    PriorityQueue<Integer> queue;
    public DinnerPlates(int capacity) {
        this.cap = capacity;
        this.stacks = new ArrayList<>();
        this.queue = new PriorityQueue<>();
    }
    
    public void push(int val) {
        if(stacks.isEmpty()) {
            stacks.add(new ArrayList<Integer>());
            stacks.get(0).add(val);
        } else {
            int index = findSlot(stacks, cap);
            List<Integer> stack = stacks.get(index);
            stack.add(val);
            if(stack.size() < cap) queue.offer(index);
        }
        return;
    }
    private int findSlot(List<List<Integer>> stacks, int cap) {
        if(queue.isEmpty()) {
            if(stacks.get(stacks.size() - 1).size() == cap) {
                stacks.add(new ArrayList<Integer>());
                return stacks.size() - 1;
            } else {
                return stacks.size() - 1;
            }
        } else {
            int res = queue.poll();
            while(!queue.isEmpty() && res == queue.peek()) queue.poll();
            return res;
        
}
        
        
    }
    public int pop() {
        if(stacks.isEmpty()) return -1;
        List<Integer> stack = stacks.get(stacks.size() - 1);
        int res = stack.get(stack.size() - 1);
        if(stack.size() <= 1) {
            stacks.remove(stacks.size() - 1);
        } else {
            stack.remove(stack.size() - 1);
        
}
        return res;
    }
    
    public int popAtStack(int index) {
        if(stacks.size() <= index) {
            return -1;
        } else {
            List<Integer> stack = stacks.get(index);
            if(stack.isEmpty())return -1;
            int res = stack.get(stack.size() - 1);
            stack.remove(stack.size() - 1);
            queue.add(index);
            return res;
        }
    }
}
