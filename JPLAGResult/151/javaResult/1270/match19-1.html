class DinnerPlates {

    TreeSet<Integer> notfull = new TreeSet<>();
    TreeSet<Integer> notempty = new TreeSet<>();
    
    Stack li[] = new Stack[200001];
    int cp = 0;
    int lst = 0;
    
    
    public DinnerPlates(int capacity) {
        cp  = capacity;
        for(int i=0;i<200001;++i){
            li[i] = new Stack();
        }
       // notfull.add(0);
    
}
    
    public void push(int val) {
        if(notfull.size()>0){
        int index = notfull.first();
        
li[index].push(val);
        notempty.add(index);
        if(li[index].size()==cp){
            notfull.remove(index);
        }
   
        }else{
          //  System.out.println(lst);
            li[lst].push(val);
            notempty.add(lst);
            if(li[lst].size()<cp){
                notfull.add(lst);
            }
            lst++;
            
        
}
    }
    
    public int pop() {
        if(notempty.size()>0){
            int index = notempty.last();
  
            int res = (int)li[index].pop();
            if(li[index].size()==0){
                notempty.remove(index);
            }
            
            notfull.add(index);
            return res;
        }else{
            return -1;
        }
        
    
}
    
    public int popAtStack(int index) {
        if(li[index].size()==0){
            return -1;
        }else{
            int res = (int)li[index].pop();
            notfull.add(index);
            return res;
        }
    }
}
