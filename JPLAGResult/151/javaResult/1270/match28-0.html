class DinnerPlates {

    int capacity;
    List<Stack<Integer>> list;
    List<Integer> size;
    int startidx;
    int endidx;
    public DinnerPlates(int capacity) {
        this.capacity = capacity;
        list = new ArrayList<>();
        size = new ArrayList<>();
        list.add(new Stack<>());
        size.add(0);
        startidx = 0;
        endidx = 0;
    }
    
    public void push(int val) {
        //System.out.println(startidx);
        //find position
        int idx = -1;
        for (int i = startidx; i < list.size(); i++){
            if (size.get(i) < capacity){
                idx =i;
                break;
            }
        }
        //need new one
        if (idx == -1){
            Stack<Integer> stack = new Stack<>();
            stack.push(val);
            list.add(stack);
            size.add(1);
            startidx = list.size()-1;
            endidx = list.size()-1;
            return;
        }
        
        Stack<Integer> stack = list.get(idx);
        stack.push(val);
        int temp = size.get(idx);
        size.set(idx, temp+1);
        startidx = idx;
        endidx = Math.max(endidx, idx);
        //System.out.println(size);
    }
    
    public int pop() {
        int idx = endidx;
        while(idx >= 0 && size.get(idx) == 0)
            idx--;
        if (idx == -1)
            return -1;
        
        if (idx < startidx)
            startidx = idx;
        endidx = idx;
        //System.out.println(endidx);
        
        Stack<Integer> stack = list.get(idx);
        int ans = stack.pop();
        int temp = size.get(idx);
        size.set(idx, temp-1);
        //System.out.println("pop" + size);
        
        return ans;
        
    }
    
    public int popAtStack(int index) {
        if (index >= list.size())
            return -1;
        Stack<Integer> stack = list.get(index);
        if (size.get(index) == 0)
            return -1;
        int ans = stack.pop();
        int temp = size.get(index);
        size.set(index, temp-1);
        //System.out.println("popat" + size);
        if (index < startidx)
            startidx = index;
        return ans;
    }
}

 * int param_3 = obj.popAtStack(index);
