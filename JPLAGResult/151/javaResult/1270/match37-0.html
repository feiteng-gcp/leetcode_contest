class DinnerPlates {
    int capacity;
    TreeSet<Integer> notFull;
    List<Stack<Integer>> st;
    int size = 0;
    
    public DinnerPlates(int capacity) {
        this.capacity = capacity;
        notFull = new TreeSet<>();
        st = new ArrayList<>();
    }
    
    public void push(int val) {
        if (notFull.isEmpty()) {
            if (st.isEmpty()) st.add(new Stack<>());
            Stack<Integer> cur = st.get(st.size() - 1);
            if (cur.size() == capacity) {
                cur = new Stack<>();
                st.add(cur);
            }
            cur.push(val);
        } else {
            if (st.get(st.size() - 1).size() == capacity) st.add(new Stack<>());
            int p = notFull.first();
            if (p >= st.size()) {
                p = st.size() - 1;
                notFull = new TreeSet<>();
            }
            st.get(p).push(val);
            if (st.get(p).size() == capacity) notFull.remove(p);
        }
        size++;
    }
    
    public int pop() {
        if (size == 0) return -1;
        Stack<Integer> last = st.get(st.size() - 1);
        int res = last.pop();
        while (st.size() != 0 && st.get(st.size() - 1).size() == 0) st.remove(st.size() - 1);
        size--;
        return res;
    }
    
    public int popAtStack(int index) {
        if (index >= st.size()) return -1;
        if (st.get(index).size() == 0) return -1;
        int res = st.get(index).pop();
        while (st.size() != 0 && st.get(st.size() - 1).size() == 0) st.remove(st.size() - 1);
        if (index < st.size() - 1) notFull.add(index);
        size--;
        return res;
    }
}
