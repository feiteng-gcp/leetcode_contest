class DinnerPlates {

    private TreeMap<Integer , Integer> map = new TreeMap<>();
    private TreeSet<Integer> freeSet = new TreeSet<>();		
    private int capacity;

    public DinnerPlates(int capacity) {
        this.capacity = capacity;	        
    }

    public void push(int val) {
        if (map.isEmpty()) {
            map.put(0 , val);	    		
        } else {	    		
            if (freeSet.isEmpty()) {
                int next = map.lastKey() + 1;
                map.put(next , val);	    			
            } else {
                int index = freeSet.first();
                freeSet.remove(index);
                
map.put(index , val);	    		
            }
        }
    }

    public int pop() {
        if (map.isEmpty()) {
            return - 1;
        } else {
            int index = map.lastKey();
            int ans = map.get(index);
            map.remove(index);
            freeSet.add(index);
            return ans;	        	
        }
    }

    public int popAtStack(int index) {
        int start = index * this.capacity , end = (index + 1) * this.capacity - 1;
        Integer pos = map.floorKey(end);
        if (pos != null && pos >= start) {
            int ans = map.get(pos);
            map.remove(pos);
            freeSet.add(pos);
            return ans;
        } else {
            return - 1;
        }
    }

}
