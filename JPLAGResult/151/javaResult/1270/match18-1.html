class DinnerPlates {
    TreeSet<Integer> empty = new TreeSet<>();
    TreeSet<Integer> fill = new TreeSet<>();
    HashMap<Integer, Stack<Integer>> stacks = new HashMap<>();
    
    int capacity;

    public DinnerPlates(int capacity) {
        this.capacity = capacity;
        for(int i = 0; i <= 100000; i++){
            empty.add(i);
        }
    
}
    
    public void push(int val) {
        //System.out.println("push");
        if(empty.isEmpty()){
            return;
        }
        int loc = empty.first();
        int size = safePush(loc, val);
        
        fill.add(loc);
        if(size == capacity){
            
empty.remove(loc);
        }
    }
    
    public int pop() {
        //System.out.println("pop");
        if(fill.isEmpty()){
            return -1;
        }
        
        return popAtStack(fill.last());
    }
    
    public int popAtStack(int index) {
        //System.out.println("pop at stack");
        Stack<Integer> cur = stacks.get(index);
        //System.out.println(cur);
        
        if(cur == null || cur.isEmpty()){
            return -1;
        }
        
        if(cur.size() == capacity){
            empty.add(index);
        }
        
if(cur.size() == 1){
            fill.remove(index);
        }
        
        return cur.pop();
    }
    
    private int safePush(int index, int val){
        if(stacks.containsKey(index)){
            Stack<Integer> s = stacks.get(index);
            s.push(val);
            return s.size();
        }
        
        Stack<Integer> s = new Stack<>();
        s.push(val);
        stacks.put(index, s);
        return 1;
    }
}
