class DinnerPlates {
    
private List<List<Integer>> stack;
    private int capacity;
    private PriorityQueue<Integer> q1;
    private PriorityQueue<Integer> q2;

    public DinnerPlates(int capacity) {
        this.capacity = capacity;
        
        stack = new ArrayList<>();
        q1 = new PriorityQueue<>();
        q2 = new PriorityQueue<>(Collections.reverseOrder());
        for (int i = 0; i < 200005; ++i) {
            stack.add(new ArrayList<>());
            q1.add(i);
        }
    }
    
    public void push(int val) {
        int k = q1.peek();
        List<Integer> list = stack.get(k);
        if (list.isEmpty()) {
            q2.add(k);
        }
        
        list.add(val);
        
        if (list.size() >= capacity) {
            q1.remove(k);
        }
    }
    
    public int pop() {
        if (q2.isEmpty()) {
            return -1;
        
}
        
        int k = q2.peek();
        List<Integer> list = stack.get(k);
        if (list.size() == capacity) {
            q1.add(k);
        }
        
        int val = list.get(list.size() - 1);
        list.remove(list.size() - 1);
        
        if (list.isEmpty()) {
            q2.remove(k);
        }
        return val;
    }
    
    public int popAtStack(int index) {
        
List<Integer> list = stack.get(index);
        if (list.isEmpty()) {
            return -1;
        }
        if (list.size() == capacity) {
            q1.add(index);
        }
        int val = list.get(list.size() - 1);
        list.remove(list.size() - 1);
        if (list.isEmpty()) {
            q2.remove(index);
        }
        return val;
    }
}
