class DinnerPlates
{
    private List<Stack<Integer>> stacks;
    private TreeSet<Integer> available;
    private TreeSet<Integer> hasData;
    private int capacity;

    public DinnerPlates(int capacity)
    {
        stacks = new ArrayList<>();
        available = new TreeSet<>();
        for (int i = 0; i < 100010; i++)
        {
            stacks.add(new Stack<>());
            available.add(i);
        }
        hasData = new TreeSet<>();
        this.capacity = capacity;
    }

    public void push(int val)
    {
        int h = available.pollFirst();
        stacks.get(h).push(val);
        hasData.add(h);
        if (stacks.get(h).size() < capacity) available.add(h);
    }

    public int pop()
    {
        if (hasData.size() == 0) return -1;
        int tail = hasData.pollLast();
        int ret = stacks.get(tail).pop();
        available.add(tail);
        if (stacks.get(tail).size() > 0) hasData.add(tail);
        return ret;
    }

    public int popAtStack(int index)
    {
        if (stacks.get(index).size() == 0) return -1;
        int ret = stacks.get(index).pop();
        available.add(index);
        if (stacks.get(index).size() == 0) hasData.remove(index);
        return ret;
    }
}
