class DinnerPlates {

    public Map<Integer, Stack<Integer>> map;
    public int capacity;
    public int minIndex;
    public int maxIndex;
    
    public DinnerPlates(int capacity) {
        map = new HashMap<>();
        this.capacity = capacity;
        minIndex = 0;
        maxIndex = -1;
    }
    
    public void push(int val) {
        if (map.get(minIndex) == null) {
            map.put(minIndex, new Stack<>());
        }
        map.get(minIndex).push(val);
        
        maxIndex = Math.max(maxIndex, minIndex);
        
        while (true) {
            if (map.get(minIndex) == null || map.get(minIndex).size() < capacity) {
                break;
            } else {
                minIndex++;
            }
        }
    }
    
    public int pop() {
        if (maxIndex == -1) {
            return -1;
        }
            
        int temp = map.get(maxIndex).pop();
        
        minIndex = Math.min(minIndex, maxIndex);
        
        while (maxIndex >= 0) {
            if (map.get(maxIndex) != null && map.get(maxIndex).size() > 0) {
                break;
            } else {
                maxIndex--;
            }
        }
        
        return temp;
    }
    
    public int popAtStack(int index) {
        if (map.get(index) == null || map.get(index).size() == 0) {
            return -1;
        }
        
        int temp = map.get(index).pop();
        
        minIndex = Math.min(minIndex, index);
        
        while (maxIndex >= 0) {
            if (map.get(maxIndex) != null && map.get(maxIndex).size() > 0) {
                break;
            } else {
                maxIndex--;
            }
        }
        
        return temp;
    }
}
