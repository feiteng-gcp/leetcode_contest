class DinnerPlates {
    
    ArrayList<ArrayDeque<Integer>> stacks = new ArrayList<>();
    PriorityQueue<Integer> pq = new PriorityQueue<>();
    int cap;
    int n;

    public DinnerPlates(int capacity) {
        cap = capacity;
    }
    
    public void push(int val) {
        if (pq.isEmpty())
        {
            if (n == 0 || stacks.get(n-1).size() == cap)
            {
                stacks.add(new ArrayDeque<>());
                n++;
            }
            stacks.get(n-1).push(val);
        }
        else
        {
            int index = pq.remove();
            if (index >= n)
            {
                pq.clear();
                push(val);
                return;
            }
            stacks.get(index).push(val);
        }
    }
    
    public int pop() {
        if (n == 0)
        {
            return -1;
        }
        int retVal = stacks.get(n-1).pop();
        while (n > 0 && stacks.get(n-1).isEmpty())
        {
            n--;
        }
        return retVal;
    }
    
    public int popAtStack(int index) {
        if (index >= stacks.size())
        {
            return -1;
        }
        if (stacks.get(index).isEmpty())
        {
            return -1;
        }
        pq.add(index);
        int retVal = stacks.get(index).pop();
        while (n > 0 && stacks.get(n-1).isEmpty())
        {
            n--;
        }
        return retVal;
    }
}
