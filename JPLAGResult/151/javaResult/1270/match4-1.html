class DinnerPlates {

    
    
private int maxStackSize;
    private TreeSet<Integer> nonEmptyStacks;
    private TreeSet<Integer> nonFullStacks;
    private List<Stack<Integer>> stacks;
    
    public DinnerPlates(int capacity) {
        
        this.maxStackSize = capacity;
        
        // The stacks that have items on them.
        this.nonEmptyStacks = new TreeSet<>();
        
        // The stacks that have space on them. 
        this.nonFullStacks = new TreeSet<>();
        this.nonFullStacks.add(0);
        
        // The stacks themselves.
        
this.stacks = new ArrayList<>();
        this.stacks.add(new Stack<>());
    }
    
    
    public void push(int val) {
        // if there are no stacks with space, we need to make a new one.
        if (nonFullStacks.size() == 0) {
            stacks.add(new Stack<>());
            nonFullStacks.add(stacks.size() - 1);
        }
        int stackID = nonFullStacks.first();
        Stack<Integer> stack = stacks.get(stackID);
        stack.push(val);
        if (stack.size() == maxStackSize) {
            nonFullStacks.remove(stackID);
        }
        nonEmptyStacks.add(stackID);
    }
    
    
    public int pop() {
        // If there are no non-empty stacks, we can't do anything.
        if (nonEmptyStacks.size() == 0) return -1;
        // Otherwise, we need to find the rightmost stack.
        int stackID = nonEmptyStacks.last();
        Stack<Integer> stack = stacks.get(stackID);
        int returnVal = stack.pop();
        nonFullStacks.add(stackID);
        if (stack.size() == 0) {
            nonEmptyStacks.remove(stackID);
        
}
        return returnVal;
    }
    
    public int popAtStack(int index) {
        if (index >= stacks.size()) {
            return -1;
        }
        Stack<Integer> stack = stacks.get(index);
        if (
stack.size() == 0) {
            return -1;
        }
        int returnVal = stack.pop();
        nonFullStacks.add(index);
        if (stack.size() == 0) {
            nonEmptyStacks.remove(index);
        }
        return returnVal;
    }
}
