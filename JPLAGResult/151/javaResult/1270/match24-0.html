class DinnerPlates {

    int capacity = 0;
    TreeMap<Integer, Stack<Integer>> map;
    PriorityQueue<Integer> q;
    
    public DinnerPlates(int capacity) {
        this.capacity = capacity;
        map = new TreeMap<>();
        q = new PriorityQueue<>();
    }
    
    public void push(int val) {
        if(q.isEmpty()) {
            
            if(map.size()==0) map.put(0, new Stack<>());
            
            Map.Entry<Integer, Stack<Integer>> entry = map.lastEntry(); 
            if(entry.getValue().size()<capacity) entry.getValue().push(val);
            else { //at the cap
                Stack s = new Stack<>();
                map.put(entry.getKey()+1, s);
                s.push(val);
            }
        }
        else { //get from queue
            int i = q.poll();
            if(map.get(i) == null) {
                map.put(i, new Stack<>());
            }
            
            map.get(i).push(val);
        }
    }
    
    public int pop() {
        
        if(map.size()==0) return -1;
        
        Map.Entry<Integer, Stack<Integer>> entry = map.lastEntry(); 
        
        int val = entry.getValue().pop();
        
        if(entry.getValue().size()==0) map.remove(entry.getKey());
        
        return val;
    }
    
    public int popAtStack(int index) {
        if(map.get(index) == null) return -1;
        
        int val = map.get(index).pop();
        
        if(map.get(index).size()==0) map.remove(index);
        
        q.add(index);
        
        return val;
    }
}

 * int param_3 = obj.popAtStack(index);
