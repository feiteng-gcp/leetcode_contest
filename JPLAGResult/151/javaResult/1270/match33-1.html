class DinnerPlates {
    int[][] plates;
    int[] sz;
    
int n, cap;
    TreeSet<Integer> set, contains;
    public DinnerPlates(int capacity) {
        n = (200000+capacity-1)/capacity;
        cap = capacity;
        sz = new int[n];
        plates = new int[n][cap];
        set = new TreeSet<>();contains = new TreeSet<>();
        for(int i = 0; i< n; i++)set.add(i);
    }
    
    public void push(int val) {
        int ind = set.first();
        plates[ind][sz[ind]++] = val;
        contains.add(ind);
        if(sz[ind] == cap)set.remove(ind)
;
    }
    
    public int pop() {
        // System.out.println("POP "+" ");
        if(contains.size() == 0)return -1;
        int ind = contains.last();
        if(sz[ind] == 1)contains.remove(ind);
        return plates[ind][--sz[ind]];
    }
    
    public int popAtStack(int index) {
        if(sz[index] == 0)return -1;
        if(sz[index] == 1)contains.remove(index);
        if(sz[index] == cap)
set.add(index);
        return plates[index][--sz[index]];
    }
}

class Pair implements Comparable<Pair>{
    int ind, pos;
    public Pair(int i, int p){
        ind = i;pos = p;
    }
    public int compareTo(Pair p){
        if(ind != p.ind)return Integer.compare(ind, p.ind);
        return Integer.compare(pos, p.pos);
    }
}
