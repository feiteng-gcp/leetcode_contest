class DinnerPlates {

    int cap = 0;
    List<Stack<Integer>> list;
    PriorityQueue<Integer> que = new PriorityQueue<>((Integer a, Integer b) -> a - b);
    public DinnerPlates(int capacity) {
        this.cap = capacity;
        list = new ArrayList<>();
    }
    
    public void push(int val) {
        if(!que.isEmpty()){
            int idx = que.poll();
            Stack<Integer> stack = list.get(idx);
            stack.push(val);
            if(stack.size() < cap){
                que.add(idx);
            }
        }else{
            Stack<Integer> stack = new Stack<>();
            stack.push(val);
            if(stack.size() < cap){
                que.add(list.size());
            }
            list.add(stack);
        }
    }
    
    public int pop() {
        if(list.size() == 0){
            return -1;
        }
        Stack<Integer> stack = list.get(list.size()-1);
        int res = stack.pop();
        if(stack.isEmpty()){
            que.remove(list.size()-1);
            list.remove(list.size()-1);
        }
        return res;
        
    }
    
    public int popAtStack(int index) {
        if(index >= list.size()){
            return -1;
        }
        Stack<Integer> stack = list.get(index);
        if(stack.isEmpty()){
            return -1;
        }else {
            if(stack.size() == cap){
                que.add(index);
            }
            return stack.pop();
        }
        
        
    }
}
