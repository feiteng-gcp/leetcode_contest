class DinnerPlates {
    
    private int capacity;
    private int size;
    private int currIdx, nextIdx;
    private List<Deque<Integer>> plates;

    public DinnerPlates(int capacity) {
        this.capacity = capacity;
        this.currIdx = 0;
        this.nextIdx = 1;
        this.plates = new ArrayList<>();
        plates.add(new ArrayDeque<>());
    }
    
    public void push(int val) {
        if (plates.get(currIdx).size() >= capacity) {
            currIdx = nextIdx++;
            if (currIdx >= plates.size()) {
                plates.add(new ArrayDeque<>());
            }
        }
        plates.get(currIdx).push(val);
    }
    
    public int pop() {
        if (currIdx == 0 && plates.get(currIdx).isEmpty()) {
            return -1;
        }
        int result = plates.get(currIdx).pop();
        if (plates.get(currIdx).size() == 0) {
            currIdx = Math.max(currIdx - 1, 0);
            nextIdx = currIdx + 1;
        }
        return result;
    }
    
    public int popAtStack(int index) {
        if (index >= plates.size()) {
            return -1;
        }
        if (plates.get(index).size() == 0) {
            return -1;
        }
        int result = plates.get(index).pop();
        nextIdx = currIdx;
        currIdx = index;
        return result;
    }
}
