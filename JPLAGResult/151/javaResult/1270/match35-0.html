class DinnerPlates {
    List<List<Integer>> room = new ArrayList<>();
    List<Data> available = new ArrayList<>();
    int capacity = 0;
    public DinnerPlates(int capacity) {
        this.capacity = capacity;
    }
    
    public void push(int val) {
        if(available.size() == 0){
            int ind = room.size();
            List<Integer> l = new ArrayList<>();
            Data d = new Data(ind, l);
            l.add(val);
            room.add(l);
            if(capacity > 1){
                available.add(d);
            }
        }else{
            List<Integer> l = available.get(0).list;
            int ind = available.get(0).ind;
            l.add(val);
            if(l.size() == capacity){
                available.remove(0);
            }
            if(ind >= room.size()){
                room.add(l);
            }
        }
    }
    
    public int pop() {
        if(room.size() == 0){
            return -1;
        }
        List<Integer> l = room.get(room.size()-1);
        if(l.size() == 0){
            return -1;
        }
        int ind = l.size()-1;
        int val = l.get(ind);
        l.remove(ind);
        
        if(l.size() == capacity-1){
            available.add(new Data(room.size()-1, l));
        }
        if(l.size()==0){
            room.remove(room.size()-1);
        }
        return val;
    }
    
    public int popAtStack(int index) {
        if(room.size() <= index){
            return -1;
        }
        List<Integer> l = room.get(index);
        if(l.size() == 0){
            return -1;
        }
        int ret = l.get(l.size()-1);
        l.remove(l.size()-1);
        if(l.size() == capacity -1){
            //find pos
            int pos = find(index, available);
            available.add(pos, new Data(index, l));
        }
        return ret;
        
    }
    //find first element that are bigger than index;
    private int find(int index, List<Data> lists){
        if(lists.size() == 0){return 0;}
        int l = 0, h = lists.size()-1;
        while(l < h){
            int mid = (l+h)/2;
            if(lists.get(mid).ind < index){
                l = mid +1;
            }else{
                h = mid;
            }
        }
        return lists.get(l).ind > index?l:l+1;
    }
}
public class Data{
    int ind;
    List<Integer> list;
    public Data(int ind, List<Integer> list){
        this.ind = ind;
        this.list = list;
    }
}
