class DinnerPlates {
    
    int capacity;
    List<Deque<Integer>> stacks;
    
    TreeSet<Integer> hasElems;
    TreeSet<Interval> atMax; // interval of maximum capacity
    
    public DinnerPlates(int capacity) {
        this.capacity = capacity;
        stacks = new ArrayList<>();
        for (int i = 0; i <= 100000; i++) {
            stacks.add(null);
        }
        hasElems = new TreeSet<>();
        atMax = 
new TreeSet<>();
    }
    
    public void push(int val) {
        if (atMax.isEmpty()) {
            if (stacks.get(0) == null) stacks.set(0, new ArrayDeque<>());
            stacks.get(0).addLast(val);
            if (stacks.get(0).size() == capacity) {
                atMax.add(new Interval(0, 0));
            }
            hasElems.add(0);
            return;
        }
        Interval maxInterval = atMax.first();
        if (maxInterval.s != 0) {
            hasElems.add(0);
            if (stacks.get(0) == null) stacks.set(0, new ArrayDeque<>());
            stacks.get(0).addLast(val);
            if (stacks.get(0).size() == capacity) {
                if (maxInterval.s == 1) {
                    atMax.remove(maxInterval);
                    maxInterval.s = 0;
                    atMax.add(maxInterval);
                    return;
                }
                atMax.add(new Interval(0, 0));
                return;
            }
        } else {
            int insIdx = maxInterval.e + 1;
            hasElems.add(insIdx);
            if (stacks.get(insIdx) == null) stacks.set(insIdx, new ArrayDeque<>());
            stacks.get(insIdx).addLast(val);
            if (stacks.get(insIdx).size() == capacity) {
                Interval nextInterval = atMax.higher(maxInterval);
                if (nextInterval != null && nextInterval.s == insIdx + 1) {
                    atMax.remove(maxInterval);
                    atMax.remove(nextInterval);
                    maxInterval.e = nextInterval.e;
                    atMax.add(maxInterval);
                    return;
                } else {
                    atMax.remove(maxInterval);
                    maxInterval.e++;
                    atMax.add(maxInterval);
                    return;
                }
            }
        }
    
}
    
    public int pop() {
        if (hasElems.isEmpty()) return -1;
        int idx = hasElems.last();
        if (stacks.get(idx).size() == capacity) {
            // fix intervals
            Interval contains = atMax.floor(new Interval(idx, idx));
            if (contains.s == idx && contains.e == idx) {
                atMax.remove(contains);
            } else if (contains.s == idx) {
                atMax.remove(contains);
                contains.s++;
                atMax.add(contains);
            } else if (contains.e == idx) {
                atMax.remove(contains);
                contains.e--;
                atMax.add(contains);
            } else {
                atMax.remove(contains);
                Interval contains2 = new Interval(idx + 1, contains.e);
                contains.e = idx - 1;
                atMax.add(contains);
                atMax.add(contains2);
            }
        }
        
        int val = stacks.get(idx).removeLast();
        if (stacks.get(idx).isEmpty()) hasElems.remove(idx)
;
        return val;
    }
    
    public int popAtStack(int index) {
        if (stacks.get(index) == null || stacks.get(index).isEmpty()) return -1;
        int idx = index;
        if (stacks.get(idx).size() == capacity) {
            // fix intervals
            Interval contains = atMax.floor(new Interval(idx, idx));
            if (contains.s == idx && contains.e == idx) {
                atMax.remove(contains);
            } else if (contains.s == idx) {
                atMax.remove(contains);
                contains.s++;
                atMax.add(contains);
            } else if (contains.e == idx) {
                atMax.remove(contains);
                contains.e--;
                atMax.add(contains);
            } else {
                atMax.remove(contains);
                Interval contains2 = new Interval(idx + 1, contains.e);
                contains.e = idx - 1;
                atMax.add(contains);
                atMax.add(contains2);
            }
        }
        
        int val = stacks.get(idx).removeLast();
        if (stacks.get(idx).isEmpty()) hasElems.remove(idx);
        return val;
    }
}

class Interval implements Comparable<Interval> {
    int s;
    int e;
    public Interval(int s, int e) {
        this.s = s;
        this.e = e;
    }
    @Override
    public int compareTo(Interval o) {
        return Integer.compare(this.s, o.s);
    }
    @Override
    public boolean equals(Object o) {
        if (!(o instanceof Interval)) return false;
        Interval i = (Interval) o;
        return i.s == this.s && i.e == this.e;
    }
}
