/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */

class Solution {
    public ListNode removeZeroSumSublists(ListNode head) {
        
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        ListNode prev = dummy;
        ListNode p = prev.next;
        
        while(p != null) {            
            ListNode t = p.next;
            int sum = p.val;
            
            ListNode tprev = prev;
            while(t != null && sum != 0) {
                sum += t.val;
                tprev = tprev.next;
                t = t.next;
            }
            if(sum == 0) {     
                prev.next = t;
            }else {
                prev = prev.next;
                
            
}
            
            
            p = prev.next;
            
            
        }
        
        return dummy.next;
    }
}
