/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode removeZeroSumSublists(ListNode head) {
        boolean exist = true;
        ListNode superHead = new ListNode(0);
        superHead.next = head;
        while (exist) {
            Map<Integer, ListNode> sum = new HashMap<>();
            
ListNode p = superHead;
            ListNode from = null;
            int t = 0;
            while (p != null) {
                t += p.val;
                if (sum.containsKey(t)) {
                    from = sum.get(t);
                    break;
                
} else {
                    sum.put(t, p);   
                }
                p = p.next;
            }
            
            if (from != null) {
                from.next = p.next;
            } else {
                exist = false;
            }
        }
        return superHead.next;
    }
}
