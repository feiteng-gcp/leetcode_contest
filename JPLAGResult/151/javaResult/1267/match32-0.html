/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode removeZeroSumSublists(ListNode head) {
        while (head != null && head.val == 0) {
            head = head.next;
        }
        ListNode node = head;
        ListNode p = null;
        while (node != null) {
            if (node.val == 0) {
                p.next = node.next;
            }
            p = node;
            node = node.next;
        }
        hn = head;
        while(true && hn != null) {
       //     print(hn);
            if (!helper(hn)) {
                break;
            }
        }
        return hn;
    }
    
    void print(ListNode node) {
        while (node != null) {
            System.out.print(node.val+",");
            node = node.next;
        }
        System.out.println();
    }
    
    ListNode hn;
    boolean helper(ListNode head) {
        HashMap<Integer, ListNode> psum = new HashMap<Integer, ListNode>();
        psum.put(0, null);
        ListNode cur = head;
        int sum = 0;
        while (cur != null) {
            sum += cur.val;
        //    System.out.println(psum);
            if (psum.containsKey(sum)) {
                ListNode node = psum.get(sum);
          //      System.out.println(node);
                if (node == null) {
                    hn = cur.next;
                    return true;
                } else {
                    node.next = cur.next;
                    return true;
                }
            }
            psum.put(sum, cur);
            cur = cur.next;
        }
        return false;
    }
}
