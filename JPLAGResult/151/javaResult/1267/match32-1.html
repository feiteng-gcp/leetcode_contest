/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    
private ListNode conduct(ListNode dummy) {
        HashMap<Integer, ListNode> map = new HashMap();
        map.put(0, dummy);
        int sum = 0;
        ListNode cur = dummy.next;
        while(cur != null) {
            sum += cur.val;
            if(map.containsKey(sum)) {
                // found sum to 0
                ListNode prev = map.get(sum);
                prev.next = cur.next;
                return cur.next;
            
}
            map.put(sum, cur);
            cur = cur.next;
        }
        return null;
    }
    
    public ListNode removeZeroSumSublists(ListNode head) {
        ListNode dummy = new ListNode(0);
        dummy.next = head;
        
        ListNode tail = head;
        while(tail != null) {
            tail = conduct(dummy);
        }
        return dummy.next;
    }
}
