class Solution {
    public int kthGrammar(int N, int K) {
        // Map<String, String> map = new HashMap<>();
        // map.put("0", "01");
        // map.put("1", "10");
        
        return helper(N, K - 1);
    }
    
    
    int helper(int N, int K) {
        if (N == 1) return 0;
        int res = helper(N - 1, K / 2);
        if (res == 0 && K % 2 == 0 || res == 1 && K % 2 == 1) {
            return 0;
        } else {
            return 1;
        }
    }
