class Solution {
    public int kthGrammar(int n, int k) {
        // int res = 0;
        // for(int i = 0; i < k; i++){
        //     res ^= j;
        //     j >>= 1;
        // }
        // return res&1;
        if(n == 1){
            return 0;
        }
        int prek = (k+1)>>1 ;
        int prer = kthGrammar(n-1,prek);
        if(k%2==0){
            return 1-prer;
        }else{
            return prer;
        }
    }
}
