class Solution {
    public int kthGrammar(int N, int K) {
        return query(K-1,0,(int)Math.pow(2,N-1)-1,0);
    }
    public static int query(int ith, 
int left, int right, int bit){
		if(left == right){
			return bit;
		}
		else{
			int mid = (left+right)>>>1;
			if(left <= ith && mid >= ith){
				return query(ith,left,mid,bit);
			}
			else return query(ith,mid+1,right,1-bit);
		}
	}
}
