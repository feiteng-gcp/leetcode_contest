class Solution {
  public int kthGrammar(int N, int K) {
     
    	if (N % 2 == 0) {
    		return dfs(K , (1 << (N - 1)) , true);
    	} else {
    		return dfs(K , (1 << (N - 1)) , false);
    	}
    	
    }
    
    private int dfs(int k , int total , boolean flag) {
    	
    	if (total == 1) {
    		return 0;
    	} else {
    		int mid = total / 2;
    		if (k > mid) {
    			int result = dfs(total - k + 1 , total / 2 , !flag);
    			if (flag) {
    				result = 1 - result;
    			}
    			return result;
    		} else {
    			return dfs(k , total / 2 , !flag);
    		}
    	}
    	
    }
}
