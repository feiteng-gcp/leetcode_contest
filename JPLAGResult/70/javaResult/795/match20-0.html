class Solution {
public int kthGrammar(int N, int K) {
        String pre = "0110";
        K --;

        return solve(pre, K);

    }

    private int solve(String pre, int K) {
        if (K < pre.length()) {
            return pre.charAt(K) - '0';
        } else {
            int p = solve(pre, K / 2);
            if (K % 2 == 0) {
                if (p == 0) return 0;
                else return 1;
            } else {
                if (p == 0) return 1;
                else return 0;
            }
        }
    }
}
