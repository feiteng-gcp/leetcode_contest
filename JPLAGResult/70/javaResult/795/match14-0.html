class Solution {
    public int kthGrammar(int N, int K) {
    	if (N == 1) return 0;
    	int lastK = (K + 1) / 2;
        int val = kthGrammar(N - 1, lastK);
        if (val == 0) {
        	if (K % 2 == 0) return 1; else return 0;
        } else {
        	if (K % 2 == 0) return 0; else return 1;
        }
    }
}
