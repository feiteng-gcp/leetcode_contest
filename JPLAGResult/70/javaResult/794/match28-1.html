class Solution {
    
class Point implements Comparable<Point>{
        int value;
        int row;
        int col;
        public Point(int val, int x, int y){
            value = val;
            row = x;
            col = y;
        }
        public int compareTo(Point a){
            return value - a.value;
        }
    }
    int min = Integer.MIN_VALUE;
    public int swimInWater(int[][] grid) {
        PriorityQueue<Point> queue = new PriorityQueue<>();
        queue.add(new Point(grid[0][0],0,0));
        search(grid,queue);
        return min;
    }
    public void search(int[][] grid, PriorityQueue<Point> queue){
        while(queue.size() > 0){
            Point p = queue.poll();
            min = Math.max(min,p.value);
            grid[p.row][p.col] = Integer.MAX_VALUE;
            
            if(p.row == grid.length-1 && p.col == grid[0].length-1)return;
            
            if(p.row+1 < grid.length && grid[p.row+1][p.col] < Integer.MAX_VALUE){
                queue.add(new Point(grid[p.row+1][p.col],p.row+1,p.col));
            }
            if(p.row-1 >=0 && grid[p.row-1][p.col] < Integer.MAX_VALUE){
                queue.add(new Point(grid[p.row-1][p.col],p.row-1,p.col));
            }
            if(p.col+1 < grid[0].length && grid[p.row][p.col+1] < Integer.MAX_VALUE){
                queue.add(new Point(grid[p.row][p.col+1],p.row,p.col+1));
            }
            if(p.col-1 >=0 && grid[p.row][p.col-1] < Integer.MAX_VALUE){
                queue.add(new Point(grid[p.row][p.col-1],p.row,p.col-1));
            }
        }
    }
}
