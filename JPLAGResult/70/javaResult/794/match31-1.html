class Solution {
    
    int n,mask = 63;
    boolean valid( int x, int y ) { return 0 <= x && x < n && 0 <= y && y < n; }
    int e( int x, int y, int t, int [][]g ) {
        return Math.max(t,g[x][y]);   
    
}
    boolean canReach( int t, int sx, int sy, int tx, int ty, int [][]g ) {
        Queue<Integer> q = new LinkedList<>();
        Set<Integer> set = new HashSet<>();
        int u,v,x,y,nx,ny,i,j,k;
        for ( u = (sx|(sy<<6)), set.add(u), q.add(u); !q.isEmpty(); ) {
            u = q.poll();
            x = (u&63);
            y = (u>>6);
            if ( x == tx && y == ty )
                return true ;
            for ( nx = x-1; nx <= x+1; ++nx )
                for ( ny = y-1; ny <= y+1; ++ny )
                    if (Math.abs(x-nx) + Math.abs(ny-y) == 1 )
                        if ( valid(nx,ny) && e(x,y,t,g) <= t && e(nx,ny,t,g) <= t ) {
                            v = (nx|(ny<<6));
                            if ( !set.contains(v)) {
                                set.add(v);
                                q.add(v);
                            }
                        }
        }
        return false;
    }
    public int swimInWater(int[][] grid) {
        int i,j,k,good,bad,mid;
        n = grid.length;
        if ( canReach(0,0,0,n-1,n-1,grid))
            return 0;
        for ( good = (1<<29), bad = 0; bad+1 < good; ) {
            mid = (good+bad)/2;
            if ( canReach(mid,0,0,n-1,n-1,grid) )
                good = mid;
            else bad = mid;
        }
        return good;
    }

}
