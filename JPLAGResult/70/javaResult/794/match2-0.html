class Solution {
    public int swimInWater(int[][] grid) {
        
        Comparator<Pair> cmp = new Comparator<Pair>(){
            public int compare(Pair a, Pair that) {
                return grid[a.x][a.y] - grid[that.x][that.y];
            }
        };
        PriorityQueue<Pair> heap = new PriorityQueue<>(cmp);
        int n = grid.length;
        boolean[][] visited = new boolean[n][n];
        int res = 0;
        heap.offer(new Pair(0, 0));
        visited[0][0] = true;
        while (!heap.isEmpty()) {
            Pair pair = heap.poll();
            res = Math.max(res, grid[pair.x][pair.y]);
            
            if (pair.x == n - 1 && pair.y == n - 1) {
                return res;
            }
            
            if (pair.x > 0 && !visited[pair.x - 1][pair.y]) {
                visited[pair.x - 1][pair.y] = true;
                heap.offer(new Pair(pair.x - 1, pair.y));
            }
            
            if (pair.x < n - 1 && !visited[pair.x + 1][pair.y]) {
                visited[pair.x + 1][pair.y] = true;
                heap.offer(new Pair(pair.x + 1, pair.y));
            }
            
            if (pair.y > 0 && !visited[pair.x][pair.y - 1]) {
                visited[pair.x][pair.y - 1] = true;
                heap.offer(new Pair(pair.x , pair.y - 1));
            }
            
            if (pair.y < n - 1 && !visited[pair.x ][pair.y + 1]) {
                visited[pair.x ][pair.y + 1] = true;
                heap.offer(new Pair(pair.x, pair.y + 1));
            }
        }
        return res;
    }
    
    static class Pair {
        int x;
        int y;
        
        Pair(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }
}
