class Solution {
int n;
    int[][] dir = new int[][] {{0,1},{1,0},{0,-1},{-1,0}};

    public int swimInWater(int[][] grid) {
        n = grid.length;
        int l = 0;
        int r = n * n;

        while (l < r) {
            int m = l + (r - l) / 2;
            boolean ok = solve(grid, m);
            if (ok) {
                r = m;
            } else {
                l = m + 1;
            }
        }
        return r;
    }

    private boolean solve(int[][] grid, int m) {
        boolean[][] visited = new boolean[n][n];
        Queue<int[]> q = new LinkedList<>();
        if (grid[0][0] > m) return false;
        q.add(new int[]{0, 0});
        visited[0][0] = true;
        while (!q.isEmpty()) {
            int[] c = q.poll();
            int i = c[0];
            int j = c[1];
            if (i == n-1 && j == n-1) return true;
            for (int k = 0; k < 4; k++) {
                int ni = i + dir[k][0];
                int nj = j + dir[k][1];

                if (ni < 0 || ni >= n || nj < 0 || nj >= n || visited[ni][nj]) continue;
                if (grid[ni][nj] > m) continue;
                if (ni == n-1 && nj == n-1) return true;
                q.add(new int[]{ni, nj});
                visited[ni][nj] = true;
            }
        }
        return false;
    }
}
