class Solution {
    public int swimInWater(int[][] grid) {
        int max = 0;
        Set<String> visited = new HashSet<>();
        PriorityQueue<int[]> queue = new PriorityQueue<>(new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return grid[o1[0]][o1[1]] - grid[o2[0]][o2[1]];
            }
        });
        visited.add(0 + " " + 0);
        queue.add(new int[]{0, 0});

        while (!queue.isEmpty()) {

            int[] curr = queue.poll();
            max = Math.max(max, grid[curr[0]][curr[1]]);
            if (curr[0] == grid.length - 1 && curr[1] == grid[0].length - 1) return max;
            if (curr[0] != 0 && !visited.contains(curr[0] - 1 + " " + curr[1])) {
                queue.add(new int[]{curr[0] - 1, curr[1]});
                visited.add(curr[0] - 1 + " " + curr[1]);
            }
            if (curr[0] != grid.length - 1 && !visited.contains(curr[0] + 1 + " " + curr[1])) {
                queue.add(new int[]{curr[0] + 1, curr[1]});
                visited.add(curr[0] + 1 + " " + curr[1]);
            }
            if (curr[1] != 0 && !visited.contains(curr[0] + " " + (curr[1] - 1))) {
                queue.add(new int[]{curr[0], curr[1] - 1});
                visited.add(curr[0] + " " + (curr[1] - 1));
            }
            if (curr[1] != grid[0].length - 1 && !visited.contains(curr[0] + " " + (curr[1] + 1))) {
                queue.add(new int[]{curr[0], curr[1] + 1});
                visited.add(curr[0] + " " + (curr[1] + 1));
            }
        }
        return -1;
    }
}
