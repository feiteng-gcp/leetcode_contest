class Solution {
    
    class Pair {
        int i, j;
        public Pair(int x, int y) {i=x; j=y;}
        public boolean equals(Object o) {
            if (o instanceof Pair) return equals( (Pair) o);
            else return false;
        }
        public boolean equals(Pair p) {
            return (i==p.i && j==p.j);
        }
        
        public int hashCode() {
            return (i<<8) + j;
        }
    }
    
    public int swimInWater(int[][] grid) {
        int left = 0, right = 0, n = grid.length;
        for(int i=0;i<n;i++) for(int j=0;j<n;j++) {
            if(grid[i][j]>right) right = grid[i][j];
        }
        while(right-left>1) {
            int middle = left + (right-left) / 2;
            if (canSwim(grid, middle)) {
                right = middle;
            } else {
                left = middle;
            }
        }
        if (canSwim(grid, left)) return left;
        else return right;
    }
    
    public boolean canSwim(int[][] grid, int bar) {
        //if grid[i][j] <= bar, then can pass (i,j)
        if(grid[0][0]>bar) return false;
        int n = grid.length;
        LinkedList<Pair> bag = new LinkedList<Pair>();
        HashSet<Pair> dedupe = new HashSet<Pair>();
        bag.add(new Pair(0,0));
        while(bag.size()>0) {
            Pair p = bag.removeFirst();
            if (p.i==n-1 && p.j==n-1) return true;
            dedupe.add(p);
            for (Pair next : getNext(p, n)) {
                if(dedupe.contains(next)) continue;
                int i=next.i, j=next.j;
                if(grid[i][j]<=bar) bag.addFirst(next);
            }
        }
        return false;
    }
    
    public List<Pair> getNext(Pair p, int n) {
        LinkedList<Pair> ret = new LinkedList<Pair>();
        int i=p.i, j=p.j;
        if (i>0) {
            ret.add(new Pair(i-1,j));
        }
        if (i<n-1) {
            ret.add(new Pair(i+1,j));
        }
        if (j>0) {
            ret.add(new Pair(i,j-1));
        }
        if (j<n-1) {
            ret.add(new Pair(i,j+1));
        }
        return ret;
    }
}
