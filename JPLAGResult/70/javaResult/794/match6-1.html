class Solution {
    public int swimInWater(int[][] grid) {
        int lo = 0, hi = 0, len = grid.length;
        for (int i = 0; i < len; i++) {
            for (int j = 0; j < len; j++) {
                if (grid[i][j] > hi)
                    hi = grid[i][j];
            }
        }
        while (lo < hi) {
            int mid = (hi - lo) / 2 + lo;
            if (isValid(grid, mid, len)) {
                hi = mid;
            } else {
                lo = mid + 1;
            }
        }
        return lo;
    }
    private int[][] dirs = new int[][]{{0,1},{1,0},{-1,0},{0,-1}};
    private boolean isValid(int[][] grid, int t, int len) {
        if (grid[0][0] > t) return false;
        Queue<int[]> q = new LinkedList<>();
        Set<Integer> visited = new HashSet<>();
        
visited.add(0);
        q.add(new int[]{0, 0});
        while (q.size() > 0) {
            int[] temp = q.poll();
            if (temp[0] == len - 1 && temp[1] == len - 1) return true;
            for (int[] d: dirs) {
                int x = temp[0] + d[0], y = temp[1] + d[1];
                
if (x >= 0 && y >= 0 && x < len && y < len && grid[x][y] <= t
                        && ! visited.contains(x * len + y)) {
                    q.add(new int[]{x, y});
                    visited.add(x * len + y);
                }
            }
        }
        return false;
    }
}
