class Solution {
    int[][] d={{0,1},{1,0},{0,-1},{-1,0}};
    public int swimInWater(int[][] grid) {
        int n=grid.length;
        int[][] dis=new int[n][n];
        for (int i=0;i<n;i++)
            for (int j=0;j<n;j++)
                dis[i][j]=10000000;
        dis[0][0]=grid[0][0];
        Queue<int[]> q=new ArrayDeque<>();
        q.offer(new int[]{0,0});
        boolean[][] bo=new boolean[n][n];
        bo[0][0]=true;
        while (!q.isEmpty())
        {
            int[] now=q.poll();
            bo[now[0]][now[1]]=false;
            for (int i=0;i<4;i++)
            {
                int x=now[0]+d[i][0],y=now[1]+d[i][1];
                if (x>=0 && y>=0 && x<n && y<n)
                {
                    int o=Math.max(dis[now[0]][now[1]],grid[x][y]);
                    if (o<dis[x][y])
                    {
                        dis[x][y]=o;
                        if (!bo[x][y]) q.offer(new int[]{x,y});
                        bo[x][y]=true;
                    }
                }
            }
        }
        return dis[n-1][n-1];
    }
}
