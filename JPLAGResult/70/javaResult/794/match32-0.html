class Solution {
    public int swimInWater(int[][] grid) {
        int N = grid.length;
        int NN = N * N;
        boolean[][] visited = new boolean[N][N];
        boolean[][] used = new boolean[N][N];
        visited[0][0] = true;
        for (int steps = 1; steps < NN; ++steps) {
        	for (int r = 0; r < N; ++r) {
        		for (int c = 0; c < N; ++c) {
        			used[r][c] = false;
         		}
        	}
        	for (int r = 0; r < N; ++r) {
        		for (int c = 0; c < N; ++c) {
        			if (!used[r][c] && visited[r][c]) {
        				DFS(used, visited, grid, r, c, steps);
        			}
        		}
        	}
        	if (visited[N - 1][N - 1]) return steps;
        }
        return -1;
    }
    
    void DFS(boolean[][] used, boolean[][] visited, int[][] grid, int r, int c, int E) {
    	int N = used.length;
    	if (r < 0 || c < 0 || r >= N || c >= N || used[r][c] || grid[r][c] > E) return;
    	used[r][c] = true;
    	visited[r][c] = true;
    	DFS(used, visited, grid, r + 1, c, E);
    	DFS(used, visited, grid, r - 1, c, E);
    	DFS(used, visited, grid, r, c + 1, E);
    	DFS(used, visited, grid, r, c - 1, E);
    }
}
