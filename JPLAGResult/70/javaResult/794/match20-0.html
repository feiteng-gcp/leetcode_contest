class Solution {
private int[] set;
	private int[] dx = new int[] {- 1 , 0 , 1, 0};
	private int[] dy = new int[] {0 , 1 , 0 , - 1};
	
	private int findSet(int x) {
		if (x == set[x]) {
			return x;
		} else {
			return set[x] = findSet(set[x]);
		}
	}
	
	private boolean check(int[][] grid , int value) {
		
		int i , j , n = grid.length;
		set = new int[n * n];
		for (i = 0;i < n * n;i ++) {
			set[i] = i;
		}
		for (i = 0;i < n;i ++) {
			for (j = 0;j < n;j ++) {
				if (grid[i][j] <= value) {
					for (int d = 0;d < 4;d ++) {
						int tx = i + dx[d];
						int ty = j + dy[d];
						if (tx >= 0 && tx < n && ty >= 0 && ty < n && grid[tx][ty] <= value) {
							int index1 = i * n + j;
							int index2 = tx * n + ty;
							int f1 = findSet(index1) , f2 = findSet(index2);
							if (f1 != f2) {
								set[f1] = f2;
							}
						}
					}
				}
			}
		}
		return findSet(0) == findSet(n * n - 1);
		
	}
	
    public int swimInWater(int[][] grid) {
     
    	int n = grid.length;
    	int low = 0 , high = n * n;
    	while (low < high) {
    		int mid = (low + high) / 2;
    		if (check(grid , mid)) {
    			high = mid;
    		} else {
    			low = mid + 1;
    		}
    	}
    	return low;
    	
    }
}
