class Solution {
    public int swimInWater(int[][] grid) {
        int[][] count=new int[grid.length][grid[0].length];
        for(int i=0;i<count.length;i++){
            Arrays.fill(count[i],Integer.MAX_VALUE);
        }
        // row,col,cost
        PriorityQueue<int[]> pq=new PriorityQueue<int[]>(new Comparator<int[]>(){
            public int compare(int[] i1,int[] i2){
                return i1[2]-i2[2];
            }
        });
        
pq.offer(new int[]{0,0,grid[0][0]});
        while(!pq.isEmpty()){
            int[] top=pq.poll();
            int r=top[0];
            int c=top[1];
            if(top[2]<count[r][c]){
                count[r][c]=top[2];
                if(r-1>=0){
                    pq.offer(new int[]{r-1,c,Math.max(top[2],grid[r-1][c])});
                }
                if(r+1<grid.length){
                    pq.offer(new int[]{r+1,c,Math.max(top[2],grid[r+1][c])});
                }
                if(c-1>=0){
                    pq.offer(new int[]{r,c-1,Math.max(top[2],grid[r][c-1])});
                }
                if(c+1<grid[0].length)
{
                    pq.offer(new int[]{r,c+1,Math.max(top[2],grid[r][c+1])});
                }
            }
        }
        return count[count.length-1][count[0].length-1];
    }
}
