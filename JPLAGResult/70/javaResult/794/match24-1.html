class Solution {
    private static int[][] DIRS = new int[][] {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

    
private static class Point {
        int x;
        int y;

        public Point(int x, int y) {
            this.x = x;
            this.y = y;
        }

        @Override
        public boolean equals(Object o) {
            if (this == o) return true;
            if (o == null || getClass() != o.getClass()) return false;

            Point point = (Point) o;

            if (x != point.x) return false;
            if (y != point.y) return false;

            return true;
        }

        @Override
        public int hashCode() {
            int result = x;
            result = 127 * result + y;
            return result;
        }
    }

    public int swimInWater(int[][] grid) {
        for (int t = 0; t < 3000; t++) {
            if (isPath(grid, t)) return t;
        }
        return -1;
    }

    private boolean isPath(int[][] grid, int t) {
        
Queue<Point> bfs = new LinkedList<>();
        Set<Point> visited = new HashSet<>();
        bfs.add(new Point(0, 0));
        while (!bfs.isEmpty()) {
            Point p = bfs.poll();
            if (p.x < 0 || p.x >= grid.length || p.y < 0 || p.y >= grid[0].length) continue;
            if (grid[p.x][p.y] > t) continue;
            if (p.x == grid.length - 1 && p.y == grid[0].length - 1) return true;
            if (visited.contains(p)) continue;
            visited.add(p);
            for (int[] dir : DIRS) bfs.add(new Point(p.x + dir[0], p.y + dir[1]));
        }
        return false;
    }
}
