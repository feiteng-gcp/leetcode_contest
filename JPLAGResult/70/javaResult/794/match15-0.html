class Solution {
    public int swimInWater(int[][] grid) {
        if(grid.length == 0){return 0;}
        int min = grid[0][0];
        int n = grid.length;
        int max = Integer.MIN_VALUE;
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                max = Math.max(max, grid[i][j]);
            }
        }
        if(win(grid, min)){
            return min;
        }
        while(min < max - 1){
            int mid = (min+max)/2;
            if(win(grid, mid)){
                max = mid;
            }else{
                min = mid;
            }
        }
        return max;
        
    }
    public boolean win(int[][] grid, int t){
        int n = grid.length;
        Set<String> set = new HashSet<>();
        Queue<String> queue = new LinkedList<>();
        if(grid[0][0] > t || grid[n-1][n-1] >t){
            return false;
        }
        queue.add("0,0");
        set.add("0,0");
        int[] dx = new int[]{0,1,0,-1};
        int[] dy = new int[]{1,0,-1,0};
        while(queue.size() > 0){
            String[] idx = queue.poll().split(",");
            int i = Integer.parseInt(idx[0]);
            int j = Integer.parseInt(idx[1]);
            for(int k = 0; k < 4; k++){
                int x = i + dx[k];
                int y = j + dy[k];
                if (x == n-1 && y == n-1 ){
                    return true;
                }
                if(!set.contains(x+","+y) && 0 <= x && x < n && 0<= y && y < n && grid[x][y] <=t){
                    queue.add(x+","+y);
                    set.add(x+","+y);
                }
            }
        }
        return false;
    }
}
