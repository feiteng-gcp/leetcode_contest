class Solution {
    public int swimInWater(int[][] grid) {
        int[] dx = {1,-1,0,0};
        int[] dy = {0,0,1,-1};
        PriorityQueue<H> queue = new PriorityQueue<>((H x, H y)->(x.depth-y.depth));
        queue.offer(new H(0,0,grid[0][0]));
        grid[0][0] = -1;
        int res = -1;
        while(!queue.isEmpty()){
            H tmp = queue.poll();
            if(tmp.depth > res) res = tmp.depth;
            if((tmp.x == grid[0].length-1) && (tmp.y == grid.length-1)){
                return res;
            }
            for(int i=0;i<4;i++){
                int x = tmp.x+dx[i];
                int y = tmp.y+dy[i];
                if(boundry(x,y,grid[0].length,grid.length)){
                    if(grid[x][y]!=-1){
                        queue.offer(new H(x,y,grid[x][y]));
                        
grid[x][y] = -1;
                    }
                }
            }
        }
        return res;
    }
    
    private class H{
        public int x;
        public int y;
        public int depth;
        public H(int x, int y, int depth){
            this.x = x;
            this.y = y;
            this.depth = depth;
        }
    }
    private boolean boundry(int x,int y, int m, int n){
        if(x<0||x>=m||y<0||y>=n) return false;
        return true;
    }
}
