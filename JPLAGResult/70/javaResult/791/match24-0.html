/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        // small, large
        TreeNode small = null;
        TreeNode large = null;
        TreeNode[] tree = new TreeNode[2];
        tree[0] = small;
        tree[1] = large;
        if(root == null) {
            return tree;
        }
        TreeNode[] res;
        if(root.val > V) {
            res = splitBST(root.left, V);
            root.left = res[1];
            tree[0] = res[0];
            tree[1] = root;
        } else {
            res = splitBST(root.right, V);
            root.right = res[0];
            tree[0] = root;
            tree[1] = res[1];
        }
        return tree;
    }
}
