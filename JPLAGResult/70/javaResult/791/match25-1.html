/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode[] splitBST(TreeNode root, int k) {
        TreeNode root2 = null;
        
TreeNode tmp = root;
        TreeNode parent1 = null;
        TreeNode parent2 = null;
        // Determine at which side of the root we will travel
        boolean toLeft = root != null && k < root.val;

        while (root != null) {
            while (root != null && (k < root.val) == toLeft) {
                parent1 = root;
                root = toLeft ? root.left : root.right;
            }
            setChild(parent1, toLeft, null); // Cut out the edge. root is now detached
            toLeft = !toLeft; // toggle direction
            if (root2 == null) {
                root2 = root; // This is the root of the other tree.
            } else if (parent2 != null) {
                setChild(parent2, toLeft, root); // re-attach the detached subtree
            }
            parent2 = parent1;
            parent1 = null;
        }
        TreeNode[] res = new TreeNode[2];
        res[0] = tmp;
        res[1] = root2;
        
return res;
    }
    
    public void setChild(TreeNode node, boolean toLeft, TreeNode child){
        // Assign child node to the indicated direction: left or right
        if (toLeft) {
            node.left = child;
        } else {
            node.right = child;
        }
    }
}
