/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        TreeNode[] result = new TreeNode[2];
        if (root == null) return result;
        TreeNode target = null, parent = null;
        if (root.val == V) {
            // root
            result[0] = root;
            result[1] = root.right;
            root.right = null;
            return result;
        } else {
            TreeNode[] findResults = findTarget(root, V, parent);
            if (findResults == null) {
                result[0] = root;
                result[1] = null;
                return result;
            }
            target = findResults[0];
            parent = findResults[1];
        }
        result[0] = target;
        result[1] = root;
        if (parent.val > target.val) {
            // left
            parent.left = target.right;
            target.right = null;
        } else {
            // right
            parent.right = target.left;
            target.left = null;
        }
        return result;
    }
    
    TreeNode[] findTarget(TreeNode node, int V, TreeNode parent) {
        if (V == node.val) {
            return new TreeNode[] {node, parent};
        } else if (node.left != null && V < node.val) {
            return findTarget(node.left, V, node);
        } else if (node.right != null) {
            return findTarget(node.right, V, node);
        } else {
            return null;
        }
    }
}
