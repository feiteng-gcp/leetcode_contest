/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        TreeNode node = root;
        TreeNode preNode = null;
        TreeNode[] re = new TreeNode[2];
        boolean flag = true;
        
        if (root == null) {
            re[0] = null;
            re[1] = null;
            
            return re;
        }
        
        if (root.val == V) {
            re[0] = root;
            re[1] = root.right;
            root.right = null;
            return re;
        }
        
        if (root.val <= V) {
            TreeNode[] subTree = splitBST(root.right, V);
            root.right = subTree[0];
            re[0] = root;
            re[1] = subTree[1];
        }
        
        else {
            TreeNode[] subTree = splitBST(root.left, V);
            re[0] = subTree[0];
            root.left = subTree[1];
            re[1] = root;
        }
        
        return re;
    }
}
