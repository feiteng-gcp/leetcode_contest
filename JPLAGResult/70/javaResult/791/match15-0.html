/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        if (root == null || (root.left == null && root.right == null)) return new TreeNode[]{root, null};
        TreeNode parent = helper(root, V);
        if (parent == null) {
            return new TreeNode[]{root, null};
        }
        if (parent.left.val == V) {
            TreeNode small = parent.left;
            parent.left = small.right;
            small.right = null;
            return new TreeNode[]{small, root};
        } else {
            TreeNode large = parent.right;
            parent.right = large.left;
            large.left = null;
            return new TreeNode[]{root, large};
        }
    }
    
    //helper function
    public TreeNode helper(TreeNode root, int V) {
        if (root == null) return null;
        if (root.left != null && root.left.val == V || root.right != null && root.right.val == V) return root;
        TreeNode left = helper(root.left, V);
        TreeNode right = helper(root.right, V);
        if (left != null) return left;
        else return right;
    }
}
