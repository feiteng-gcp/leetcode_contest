/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        if (root==null) return new TreeNode[] {null, null};
        if (root.val<=V) {
            TreeNode subtree = root.right;
            root.right = null;
            TreeNode[] subsplit = splitBST(subtree, V);
            
root.right = subsplit[0];
            return new TreeNode[] {root, subsplit[1]};
        } else {
            TreeNode subtree = root.left;
            root.left = null;
            TreeNode[] subsplit = splitBST(subtree, V);
            
root.left = subsplit[1];
            return new TreeNode[] {subsplit[0], root};
        }
    }
}
