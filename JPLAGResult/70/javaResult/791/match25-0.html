/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        TreeNode obj[] = new TreeNode[2];
        obj[0] = root;
        obj[1] = splitTree(root,V);
        
        return obj;
    }
    
    static void x(TreeNode node, boolean l, TreeNode c){
    
        if (l) {
            node.left = c;
        } else {
            node.right = c;
        }
    }

    static TreeNode splitTree(TreeNode root, int k){
        TreeNode root2 = null;
        TreeNode p1 = null;
        TreeNode p2 = null;
        
        boolean l = root != null && k < root.val;

        while (root != null) {
            while (root != null && (k < root.val) == l) {
                p1 = root;
                root = l ? root.left : root.right;
            }
            //x(p1, l , null);
            
            if(l) {
                p1.left = null;
            }
            else {
                p1.right = null;
            }
            
            l = !l; 
            if (root2 == null) {
                root2 = root; 
            } 
            else if (p2 != null) {
                
                if(l) {
                    p2.left = root;
                }
                else {
                    p2.right = root;
                }
                
                
            }
            p2 = p1;
            p1 = null;
        }
        return root2;
    }
    
}
