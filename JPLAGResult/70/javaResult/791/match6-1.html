/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        TreeNode[] nodes = new TreeNode[2];
        if (root == null) {
            return nodes;
        }
        if (root.val == V) {
            nodes[0] = root;
            nodes[1] = root.right;
            root.right = null;
        } else if (root.val < V) {
            nodes[0] = root;
            while (root.right != null && root.right.val <= V) {
                root = root.right;
            }
            if (root.right == null || root.right.val == V)  {
                
nodes[1] = root.right;
                root.right = null;
            } else {
                TreeNode[] tmp = splitBST(root.right, V);
                root.right = tmp[0];
                nodes[1] = tmp[1];
            }
        } else {
            nodes[1] = root;
            while (root.left != null && root.left.val > V) {
                root = root.left;
            }
            if (root.left != null) {
                TreeNode[] tmp = splitBST(root.left, V);
                root.left = tmp[1];
                nodes[0] = tmp[0];
            }
        }
        return nodes;
    }
}
