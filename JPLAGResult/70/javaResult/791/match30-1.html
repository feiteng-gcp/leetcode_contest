/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        if (root == null) {
            return new TreeNode[2];
        }
        TreeNode[] ans = new TreeNode[2];
        TreeNode parent = findSplit(root, V);
        
        
if (parent == null) {
            ans[0] = root;
            ans[1] = null;
        } else {
            if (parent.val > V) {
                ans[0] = parent.left;
                ans[1] = root;
                parent.left = ans[0].right;
                ans[0].right = null;
            } else {
                
ans[0] = root;
                ans[1] = parent.right;
                parent.right = null;
            }
        }
        
        return ans;
    }
    
    private TreeNode findSplit(TreeNode cur, int V) {
        TreeNode pre = null;
        while (cur != null) {
            if (cur.val > V) {
                pre = cur;
                cur = cur.left;
            } else {
                pre = cur;
                cur = cur.right;
            }
            if (cur != null && ((pre.val > V && cur.val <= V) || (pre.val <= V && cur.val > V))) {
                return pre;
            }  
        }
        
        return cur;
    }
}
