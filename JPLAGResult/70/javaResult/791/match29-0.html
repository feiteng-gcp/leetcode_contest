/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        TreeNode[] ret = new TreeNode[2];
        if (root != null) {
            if (root.val > V) {
                TreeNode[] res = splitBST(root.left, V);
                ret[0] = res[0];
                root.left = res[1];
                ret[1] = root;
             } else if (root.val == V) { 
                ret[0] = root;
                ret[1] = root.right;
                ret[0].right = null;
             } else {
                TreeNode[] res = splitBST(root.right, V);
                ret[0] = res[0];
                root.right = res[1];
                ret[1] = root;
            }
        } else {
            ret[0] = ret[1] = null;
        }
        return ret; 
    }
}
