/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        TreeNode[] nodes = new TreeNode[2];
        if (root == null) {
            nodes[0] = nodes[1] = null;
        } else if (root.val > V) {
            TreeNode[] subNodes = splitBST(root.left, V);
            root.left = subNodes[1];
            nodes[0] = subNodes[0];
            nodes[1] = root;
        } else {
            TreeNode[] subNodes = splitBST(root.right, V);
            root.right = subNodes[0];
            nodes[0] = root;
            nodes[1] = subNodes[1];
        }
        return nodes;
    }
}
