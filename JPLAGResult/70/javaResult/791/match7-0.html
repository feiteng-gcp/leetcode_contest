/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
// how to solve this?   


class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        // return 小于的部分  return 大于的部分    我感觉就是这样操作的...  return 小于3的部分 return大于3的部分.
        // 就是这样操作？  感觉是个迭代的过程 how to solve this problem???
        if (root == null) return new TreeNode[2];
        TreeNode[] res = new TreeNode[2];
        if (root.val > V) {
            TreeNode[] left = splitBST(root.left, V);
            root.left = left[1];
            res[1] = root;
            res[0] = left[0];
        }
        else {
            TreeNode[] right = splitBST(root.right, V);
            root.right = right[0];
            res[0] = root;
            res[1] = right[1];
        }
        
        return res;
    }
}
