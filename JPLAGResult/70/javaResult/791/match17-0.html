/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        if(root == null) return new TreeNode[2];
        TreeNode[] res = new TreeNode[2];
        if(root.val == V){
            res[0] = root.left;
            res[1] = root;
            root.left = null;
            return res;
        }
        if(root.left == null && root.right == null){
            if(root.val > V){
                res[1] = root;
                res[0] = null;
            }else{
                res[0] = root;
                res[1] = null;
            }
            return res;
        }
        if(root.val > V){
            res[1] = root;
            TreeNode[] temp = splitBST(root.left,V);
            root.left = temp[1];
            res[0] = temp[0];
        }else{
            //root.val < V
            res[0] = root;
            TreeNode[] temp = splitBST(root.right,V);
            root.right = temp[0];
            res[1] = temp[1];
        }
        return res;
    }
}
