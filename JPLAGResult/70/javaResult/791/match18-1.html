/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
    public TreeNode[] splitBST(TreeNode root, int V) {
        if (root == null) {
            return new TreeNode[2];
        }
        return dfs(root, V);
    
    }
    public TreeNode[] dfs(TreeNode root, int V) {
        TreeNode[] res = new TreeNode[2];
        if (root == null) {
            return res;
        }
        if (root.val == V) {
            TreeNode p = root.right;
            
root.right = null;
            res[0] = root;
            res[1] = p;
            return res;
        } else if (root.val > V) {
            TreeNode[] h = dfs(root.left, V);
            root.left = h[1];
            res[0] = h[0];
            res[1] = root;
            return res;
        } else {
            TreeNode[] h = dfs(root.right, V);
            root.right = h[0];
            res[0] = root;
            res[1] = h[1];
            return res;
        }
    }
}
