/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public TreeNode[] splitBST(TreeNode root, int V) {
        TreeNode[] ans = new TreeNode[2];
        if (root == null){
            return ans;
        }
        if (root.val == V){
            ans[0] = root;
            ans[1] = root.right;
            root.right = null;
            return ans;
        }
        if (root.left == null && root.right == null){
            if (root.val <= V){
                ans[0] = root;
            } else {
                ans[1] = root;
            }
            return ans;
        }
        if (root.val <= V){
            ans[0] = root;
            TreeNode[] right = splitBST(root.right, V);
            
root.right = right[0];
            ans[1] = right[1];
        } else {
            TreeNode[] left = splitBST(root.left, V);
            root.left = left[1];
            ans[0] = left[0];
            ans[1] = root;
        }
        return ans;
    }
}
