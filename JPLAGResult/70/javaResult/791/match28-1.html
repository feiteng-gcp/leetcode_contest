/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    
public TreeNode[] splitBST(TreeNode root, int V) {
        //这个数是二分查找树
        TreeNode[] res = new TreeNode[2];
        if(root == null){
            return res;
        }
        
        //第一个节点是 小于等于， 第二个数大于
        if(root.val < V ){
            //如果 ，左子树全部都是
            
            TreeNode[] nodes = splitBST(root.right,V);
            res[0] = root;
            res[0].right = nodes[0];
            res[1] = nodes[1];
            
        
}else if(root.val > V){
            //如果大于
            
                        
            TreeNode[] nodes = splitBST(root.left,V);
            res[0] = nodes[0];
            res[1] = root;
            res[1].left = nodes[1];
        }else{
            res[0] = root;
            res[1] = root.right;
            res[0].right = null;
        }
        return res;
    }
}
