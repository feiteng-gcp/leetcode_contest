class Solution {
    public boolean canTransform(String start, String end) {
        if (start.length() != end.length()) {
            return false;
        }
        
        int s_ptr = 0, e_ptr = 0;
        while (true) {
            // get next non 'X' character and compare.
            s_ptr = nextIndex(start, s_ptr);
            e_ptr = nextIndex(end, e_ptr);
            if (s_ptr == start.length() && e_ptr == end.length()) {
                return true;
            } else if (s_ptr == start.length() && e_ptr != end.length()) {
                return false;
            } else if (s_ptr != start.length() && e_ptr == end.length()) {
                return false;
            } else {
                if (start.charAt(s_ptr) != end.charAt(e_ptr)) {
                    return false;
                } else {
                    if (start.charAt(s_ptr) == 'R' && s_ptr > e_ptr) {
                        return false;
                    } else if (start.charAt(s_ptr) == 'L' && s_ptr < e_ptr) {
                        return false;
                    } else {
                        s_ptr++;
                        e_ptr++;                        
                    }
                }
            }
        }
    }
    
    private int nextIndex(String str, int i) {
        for (; i < str.length(); ++i) {
            if (str.charAt(i) != 'X') {
                break;
            }
        }
        return i;
    }
}
