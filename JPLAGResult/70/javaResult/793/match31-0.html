class Solution {
private String getString(String string) {
		
		StringBuilder builder = new StringBuilder();
		int i , length = string.length();
		for (i = 0;i < length;i ++) {
			if (string.charAt(i) != 'X') {
				builder.append(string.charAt(i));
			}
		}
		return builder.toString();
		
	}
	
	private List<Integer> getList(String string) {
		
		List<Integer> list = new ArrayList<>();
		int i , length = string.length() , cnt = 0;
		for (i = 0;i < length;i ++) {
			if (string.charAt(i) != 'X') {				
				list.add(cnt);
				cnt = 0;
			} else {
				cnt ++;
			}
		}
		list.add(cnt);
		return list;
		
	}
	
    public boolean canTransform(String start, String end) {
    
    	if (getString(start).equals(getString(end))) {
    		String string = getString(start);
    		List<Integer> list1 = getList(start);
    		List<Integer> list2 = getList(end);
    		for (int i = 0;i < list1.size();i ++) {
    			if (list1.get(i).equals(list2.get(i))) {
    				continue;
    			} else {
    				if (list1.get(i) > list2.get(i)) {
    					if (string.charAt(i) == 'L') {
    						int temp = list1.get(i) - list2.get(i) + list1.get(i + 1);
    						list1.set(i + 1 , temp);					
    					} else {
    						return false;
    					}
    				} else {
    					if (string.charAt(i) == 'R') {
    						int temp = list1.get(i + 1) - (list2.get(i) - list1.get(i));
    						list1.set(i + 1 , temp);
    					} else {
    						return false;
    					}
    				}
    			}
    		}
    		return true;
    	} else {
    		return false;
    	}
    	
    }
	
}
