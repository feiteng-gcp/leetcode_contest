class Solution {
    public boolean canTransform(String start, String end) {
        int index1=0;
        int index2=0;
        while(index1<start.length()&&start.charAt(index1)=='X'){
            index1++;
        }
        while(index2<start.length()&&end.charAt(index2)=='X'){
            index2++;
        }
        while(index1<
start.length()&&index2<end.length()){
            if(start.charAt(index1)!=end.charAt(index2)){
                return false;
            }
            if(start.charAt(index1)=='L'){
                if(index1<index2){
                    return false;
                }
            }
            else{
                if(index1>index2){
                    return false;
                }
            }
            index1++;
            index2++;
            while(index1<start.length()&&start.charAt(index1)=='X'){
                index1++;
            }
            while(index2<start.length()&&end.charAt(index2)=='X'){
                index2++;
            }
        }
        return index1==start.length()&&index2==end.length();
    }
}
