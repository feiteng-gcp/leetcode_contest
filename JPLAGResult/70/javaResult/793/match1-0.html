class Solution {
    public boolean canTransform(String s, String e) {
        
        char[] ss = s.toCharArray();
        char[] es = e.toCharArray();
        if(s.length()!=e.length()) return false;
        List<Integer> sl = new ArrayList<>();
        List<Integer> sr = new ArrayList<>();
        List<Integer> esl = new ArrayList<>();
        List<Integer> esr = new ArrayList<>();
        String ns = "";
        String ens = "";
        
        for(int i=0;i<ss.length;i++){
            if(ss[i]=='L'){
                sl.add(i);
                ns += ss[i];
            }else if(ss[i]=='R'){
                sr.add(i);
                ns += ss[i];
            }
            
            if(es[i]=='L'){
                esl.add(i);
                ens += es[i];
            }else if(es[i]=='R'){
                esr.add(i);
                ens += es[i];
            }
            
            
        }
        if(!ns.equals(ens)) return false;
        if(sl.size()!=esl.size()) return false;
        if(sr.size()!=esr.size()) return false;
        
        for(int i=0;i<sl.size();i++){
            if(sl.get(i)<esl.get(i)){
                return false;
            }
        }
        for(int i=0;i<sr.size();i++){
            if(sr.get(i)>esr.get(i)){
                return false;
            }
        }
        return true;
        
    }
}
