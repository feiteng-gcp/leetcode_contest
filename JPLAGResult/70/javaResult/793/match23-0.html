class Solution {
    public boolean canTransform(String start, String end) {
        int j = 0;
        for (int i = 0; i < start.length(); i++) if (start.charAt(i) != 'X'){
            while (j < end.length() && end.charAt(j) == 'X') j++;
            if (j >= end.length() || start.charAt(i) != end.charAt(j)) return false;

            if (i > j && start.charAt(i) == 'R') return false;
            if (i < j && start.charAt(i) == 'L') return false;
            j++;
        }
        while (j < end.length() && end.charAt(j) == 'X') j++;
        return j == end.length();
    }
}
