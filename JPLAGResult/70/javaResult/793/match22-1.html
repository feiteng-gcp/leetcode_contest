class Solution {
    public boolean canTransform(String start, String end) {
        if(start.equals(end))return true;
        String a = "";
        String b= "";
        
int countX1 = 0;
        int countX2 = 0;
        List<Integer> RP = new ArrayList<>();
        List<Integer> LP = new ArrayList<>();
        for(int i = 0; i < start.length(); i++){
            char x = start.charAt(i);
            if(x == 'R'){a+="R";RP.add(i);}
            else if(x == 'L'){a+="L";LP.add(i);}
            else if(x == 'X')countX1++;
        }
        int rindex = 0;
        int lindex = 0;
        for(int i = 0; i < end.length(); i++){
            char x = end.charAt(i);
            if(x == 'R'){
                b+="R";
                if(rindex < RP.size()) {
                    if(RP.get(rindex) > i){
                        return false;
                    }
                    else{
                        rindex++;
                    }
                }
                else{
                    return false;
                }
            }
            else if(x == 'L'){
                b+="L";
                if(lindex < LP.size()){
                    if(LP.get(lindex) < i){
                        return false;
                    }
                    else {
                        lindex++;
                    }
                }
                else{
                    return false;
                }
            }
            else if(x == 'X')countX2++;
        }
        return a.equals(b) && countX1 == countX2;
    }
}
