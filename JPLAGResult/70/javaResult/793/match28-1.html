class Solution {
    public boolean canTransform(String start, String end) {
        char[] sa = start.toCharArray(), ea = end.toCharArray();
        for (int i = 0; i < sa.length; i++) {
            if (sa[i] == ea[i]) continue;
            if (sa[i] == 'L') return false;
            if (sa[i] == 'R' && ea[i] == 'L') return false;
            if (sa[i] == 'R') {
                int j = i;
                while (j < sa.length && sa[j] == 'R') j++;
                if (j == sa.length || sa[j] != 'X') return false;
                sa[j] = 'R';
            } else {
                int j = i;
                while (j < sa.length && ea[j] == 'L') j++;
                if (j == sa.length || ea[j] != 'X') return false;
                ea[j] = 'L';
            }
        }
        return true;
    }
}
