class Solution {
  
  class Item {
    char c;
    int p;
    Item(char c, int p) {
      this.c = c;
      this.p = p;
    }
  }
  
  
public boolean canTransform(String start, String end) {
    if (start.length() != end.length()) return false;
    int n = start.length();
    Stack<Item> stk = new Stack<Item>();      
    for (int i = 0; i != n; ++i) {
      char c = start.charAt(i);
      if (c == 'X') continue;
      stk.push(new Item(c, i));
    }
    for (int i = n - 1; i >= 0; --i) {
      char c = end.charAt(i);
      if (c == 'X') continue;
      if (stk.empty()) return false;
      if (c != stk.peek().c) return false;
      if (c == 'L' && i > stk.peek().p) return false;
      if (c == 'R' && i < stk.peek().p) return false;
      stk.pop();
    }
    return stk.empty();
  }
}
