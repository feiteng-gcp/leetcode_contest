class Solution {
    public boolean canTransform(String start, String end) {
        if(start.equals(end))
            return true;
        
        for(int i = 0; i < end.length(); i++){
            if(start.charAt(i)==end.charAt(i))
                continue;
            else{
                if(end.charAt(i)=='R')
                    return false;
                
                else if(end.charAt(i) == 'X'){
                    if(start.charAt(i) == 'L')
                        return false;
                    else{
                        int j = i;
                        while(j < start.length()){
                            if(start.charAt(j)=='R'){
                                j++;
                                if(j==start.length())
                                    return false;
                                continue;
                            }
                            else if(start.charAt(j)=='L'){
                                return false;
                            }
                            else{
                                start = start.substring(0,i)+"X"+start.substring(i+1,j)+"R"+start.substring(j+1);
                                break;
                            }
                        }
                    }
                }
                
                
                else{
                    if(start.charAt(i) == 'R')
                        return false;
                    else{
                        int j = i;
                        while(j < start.length()){
                            if(start.charAt(j)=='X'){
                                j++;
                                if(j==start.length())
                                    return false;
                                continue;
                            }
                            else if(start.charAt(j)=='R'){
                                return false;
                            }
                            else{
                                start = start.substring(0,i)+"L"+start.substring(i+1,j)+"X"+start.substring(j+1);
                                break;
                            }
                        }
                    }
                }
            }
        }
        return true;
        
    }
}
