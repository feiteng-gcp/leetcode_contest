class Solution {
    public boolean canTransform(String src, String trg) {
        if(src.length() != trg.length()){return false;}
        int i = 0;
        int j = 0;
        int len = src.length();
        while(i < len && j < len){
            while(i < len && src.charAt(i)=='X'){i++;}
            while(j < len && trg.charAt(j)=='X'){j++;}
            if(i < len && j < len){
                char a = src.charAt(i);
                char b = trg.charAt(j);
                if(a != b){return false;}
                if(a == 'L'){
                    if(i < j){return false;}
                    else{
                        src = swap(src, i, j);
                        i=j+1;
                        j++;
                    }
                }else if(a == 'R'){
                    if(i>j){return false;}
                    else{
                        trg = swap(trg,i,j);
                        j = i+1;
                        i++;
                    }
                }
            }else if(i < len && j== len || i==len && j < len){
                return false;
            }
        }
        return true;
    }
    public String swap(String str, int i , int j){
        StringBuffer sb = new StringBuffer(str);
        char tmp = str.charAt(i);
        sb.setCharAt(i,str.charAt(j));
        sb.setCharAt(j,tmp);
        return sb.toString();
    }
}
