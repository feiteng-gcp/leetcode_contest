	class Solution {
	    public boolean canTransform(String start, String end) {
	    	if(!start.replace("X", "").equals(end.replace("X", "")))return false;
	    	int n = start.length();
	    	int p = 0;
	    	for(int i = 0;i < n;i++){
	    		if(start.charAt(i) != 'X'){
	    			while(p < n && end.charAt(p) == 'X'){
	    				p++;
	    			}
	    			if(start.charAt(i) == 'R'){

	    				if(i > p)return false;
	    			}else{
	    				if(i < p)return false;
	    			}
	    			p++;
	    		}
	    	}
	    	return true;
	    }
	}
