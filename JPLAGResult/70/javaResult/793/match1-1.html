class Solution {
    public boolean canTransform(String start, String end) {
        String S = "";
        String E = "";
        char[] St = start.toCharArray();
        char[] En = end.toCharArray();
        int l1 = start.length();
        int l2 = end.length();
        if (l1 != l2) {
            return false;
        }

        for (int i = 0; i < l1; i++) {
            if (St[i] != 'X') {
                S = S + St[i];
            }
            if (En[i] != 'X') {
                E = E + En[i];
            }
        }
        if (!E.equals(S)) {
            return false;
        }

        Stack<Integer> ss = new Stack<>();
        Stack<Integer> es = new Stack<>();

        for (int i = 0; i < l1; i++) {
            if (St[i] == 'R') {
                ss.push(i);
            }
            if (En[i] == 'R') {
                es.push(i);
            }

        }

        while (!
ss.isEmpty()) {
            if (ss.pop() > es.pop()) {
                return false;
            }
        }


        for (int i = 0; i < l1; i++) {
            if (St[i] == 'L') {
                ss.push(i);
            }
            if (En[i] == 'L') {
                es.push(i);
            }

        }

        while (!
ss.isEmpty()) {
            if (ss.pop() < es.pop()) {
                return false;
            }
        }


        return true;
    }
}
