class Solution {
    public boolean canTransform(String start, String end) {
        List<int[]> sL = countL(start);
        List<int[]> sR = countR(start);
        List<int[]> eL = countL(end);
        List<int[]> eR = countR(end);

        if (sL.size() != eL.size() || sR.size() != eR.size()) {
            return false;
        }

        for (int i = 0; i < sL.size(); i++) {
            int[] a = sL.get(i);
            int[] b = eL.get(i);
            if (a[0] != b[0]) return false;
            if (a[1] < b[1]) return false;
        }

        for (int i = 0; i < sR.size(); i++) {
            int[] a = sR.get(i);
            int[] b = eR.get(i);
            if (a[0] != b[0]) return false;
            if (a[1] > b[1]) return false;
        }
        return true;
    }

    public List<int[]> countR(String s) {
        int L = 0;
        int X = 0;
        List<int[]> ans = new ArrayList<>();
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == 'L') L++;
            if (s.charAt(i) == 'X') X++;
            if (s.charAt(i) == 'R') ans.add(new int[]{L, X});
        }
        return ans;
    }

    public List<int[]> countL(String s) {
        int R = 0;
        int X = 0;
        List<int[]> ans = new ArrayList<>();
        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == 'R') R++;
            if (s.charAt(i) == 'X') X++;
            if (s.charAt(i) == 'L') ans.add(new int[]{R, X});
        }
        return ans;
    }
}
