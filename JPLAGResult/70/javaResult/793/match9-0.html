class Solution {
    public boolean canTransform(String start, String end) {
        if(start.length()!=end.length()) return false;
        int countR = 0;
        int countL = 0;
        for(int i = 0;i < start.length();i++){
            if(countR<0||countL<0) return false;
            if(start.charAt(i)=='R') countR++;
            if(end.charAt(i)=='R') countR--;
            if(start.charAt(i)=='L') countL--;
            if(end.charAt(i)=='L') countL++;
        }
        return (countR==0&&countL==0);
    }
}
