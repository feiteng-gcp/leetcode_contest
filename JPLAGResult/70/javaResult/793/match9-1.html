class Solution {
    public boolean canTransform(String start, String end) {
        if (start.length() != end.length()) return false;
        int r1s = 0;
        int l1s = 0;
        int r2s = 0;
        int l2s = 0;
        for (int i = 0; i < start.length(); i++) {
            char c1 = start.charAt(i);
            char c2 = end.charAt(i);
            if (c1 == 'R') r1s++;
            else if (c1 == 'L') l1s++;
            if (c2 == 'R') r2s++;
            else if (c2 == 'L') l2s++;
            if (r1s < r2s) return false;
            if (l1s > l2s) return false;
            if (r1s > r2s && l1s < l2s) return false;
        }
        return r1s == r2s && l1s == l2s;
    }
}
