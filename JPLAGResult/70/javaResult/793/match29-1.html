class Solution {
    public boolean canTransform(String start, String end) {
        
ArrayList<Integer> startR = new ArrayList();
        ArrayList<Integer> startL = new ArrayList();
        ArrayList<Integer> endR = new ArrayList();
        ArrayList<Integer> endL = new ArrayList();
        
        int countSL = 0;
        int countSR = 0;
        
        for(int i=0; i<start.length(); i++) {
            if(start.charAt(i)=='L') {
                countSL++;
            } else if(start.charAt(i)=='R') {
                countSR++;
            } else if(start.charAt(i)=='X') {
                startL.add(countSL);
                startR.add(countSR);
            }
        }
        int countEL = 0;
        int countER = 0;
        for(int i=0; i<end.length(); i++) {
            if(end.charAt(i)=='L') {
                countEL++;
            } else if(end.charAt(i)=='R') {
                countER++;
            } else if(end.charAt(i)=='X') {
                endL.add(countEL);
                endR.add(countER);
            }
        }
        
        if(startL.size()!=endL.size() || countSL!=countEL || countSR!=countER) {
            return false;
        }
        for(int i=0; i<startL.size(); i++) {
            if(startL.get(i).intValue()>endL.get(i).intValue()) {
                return false;
            } else if(startR.get(i).intValue()<endR.get(i).intValue()) {
                return false;
            }
        }
        return true;
    }
}
