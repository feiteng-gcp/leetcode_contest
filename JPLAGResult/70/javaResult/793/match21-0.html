class Solution {
    public boolean canTransform(String start, String end) {

        if (start.length() != end.length()) return false;

        int l = 0, r = 0;

        Queue<Character> sq = new LinkedList<>();
        Queue<Character> eq = new LinkedList<>();

        for (int i = 0; i < start.length(); i++) {
            if (start.charAt(i) == 'L') l++;
            else if (start.charAt(i) == 'R') r++;
            if (end.charAt(i) == 'L') l--;
            else if (end.charAt(i) == 'R') r--;

            if (start.charAt(i) != 'X') sq.offer(start.charAt(i));
            if (end.charAt(i) != 'X') eq.offer(end.charAt(i));

            if (!sq.isEmpty() && !eq.isEmpty()) {
                if (sq.poll() != eq.poll()) {
                    return false;
                }
            }
            
            if (l > 0) return false;
            else if (r < 0) return false;
        }

        return l==0 && r==0;

    }
}
