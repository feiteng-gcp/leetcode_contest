class Solution {
public:
    bool canTransform(string start, string end) {
        int i;
        int ind_start=0;
        int ind_end=0;
        int scntl=0,scntr=0,ecntl=0,ecntr=0;
        if (start == end) return true;
        if (start.length() != end.length()) return false;
        for (i=0; i<start.length(); i++)
            if (start[i] == 'L')
                scntl++;
            else if (start[i] == 'R')
                scntr++;
        
        for (i=0; i<end.length(); i++)
            if (end[i] == 'L')
                ecntl++;
            else if (end[i] == 'R')
                ecntr++;
        
        if (scntl != ecntl || scntr != ecntr) return false;
        

        for (i=0; i<end.length(); i++) {
            while (ind_start < start.length() && start[ind_start] == 'X') ind_start++;
            while (ind_end < end.length() && end[ind_end] == 'X') ind_end++;
        
            if (ind_start == start.length() && ind_end == end.length()) return true;
            if (ind_start == start.length() || ind_end == end.length()) return false;
            
            if (start[ind_start] != end[ind_end]) return false;
            if (start[ind_start] == 'R') {
                if (ind_start>ind_end) return false;
            } else {
                if (ind_start<ind_end) return false;
            }
            ind_start++;
            ind_end++;
        }
        return true;
    }
};
