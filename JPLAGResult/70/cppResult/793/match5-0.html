class Solution {
public:
bool canTransform(string a, string b) {
    string sa = a, sb = b;
    sort(sa.begin(), sa.end());
    sort(sb.begin(), sb.end());
    if(sa != sb) return 0;
    int n = a.size();
    int x = 0, y = 0;
    while(x < n && y < n) {
        while(a[x] == 'X') x++;
        while(b[y] == 'X') y++;
        if(!(x < n && y < n)) break;
        if(a[x] != b[y]) return 0;
        if(a[x] == 'L') {
            if(x < y) return 0;
        } else {
            if(x > y) return 0;
        }
        x++;
        y++;
    }
    while(x < n) {
        if(a[x] != 'X') return 0;
        x++;
    }
    while(y < n) {
        if(b[y] != 'X') return 0;
        y++;
    }
    return 1;
}
};
