class Solution {
public:
    bool canTransform(string start, string end) {
    if(start.size()!=end.size())
        return false;
    if(start.size()==0)
        return true;
    vector<int> count(128,0);
    for(char c:start)
        count[c]++;
    for(char c:end)
        count[c]--;
    for(int a:count)
        if(a!=0)
            return false;
    vector<int> sp,ep;
    for(int i=0;i<start.size();++i){
        if(start[i]=='R')
            sp.push_back(i+1);
        else if(start[i]=='L')
            sp.push_back(-(i+1));
    }
    for(int i=0;i<end.size();++i){
        if(end[i]=='R')
            ep.push_back(i+1);
        else if(end[i]=='L')
            ep.push_back(-(i+1));

    }
    if(sp.size()!=ep.size())
        return false;
    for(int i=0;i<sp.size();++i){
        if(sp[i]*ep[i]<0)
            return false;
        if(sp[i]>ep[i])
            return false;
    }
    return true;
}
};
