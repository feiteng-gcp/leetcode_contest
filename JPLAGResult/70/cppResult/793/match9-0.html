class Solution {
public:
    bool canTransform(string start, string end) {
        int cntsl=0, cntsr=0, cntel=0, cnter=0;
        vector<char> temp1, temp2;
        if(start.size() != end.size())
            return false;
        for(int i=0; i<start.size(); i++){
            if(start[i] == 'L'){
                cntsl++;
                temp1.push_back(start[i]);
            }else if(start[i] == 'R'){
                cntsr++;
                temp1.push_back(start[i]);
            }
            if(end[i] == 'L'){
                cntel++;
                temp2.push_back(end[i]);
            }else if(end[i] == 'R'){
                cnter++;
                temp2.push_back(end[i]);
            }

            if((cntsl>cntel) || (cntsr<cnter))
                return false;
        }
        if(temp1.size() != temp2.size())
            return false;
        for(int i=0; i<temp1.size(); i++){
            if(temp1[i] != temp2[i])
                return  false;
        }

        return true;
    }
};
