class Solution {
public:
    bool canTransform(string start, string end) {
        string a, b;
        for (auto c: start) {
            if (c == 'L' || c == 'R') {
                a.push_back(c);
            }
        }
        for (auto c: end) {
            if (c == 'L' || c == 'R') {
                b.push_back(c);
            }
        }
        if (a != b) {
            return false;
        }
        int x = 0, y = 0;

        for (int i = 0; i < start.size(); i++) {
            if (start[i] == 'L') {
                x++;
            }
            if (end[i] == 'L') {
                y++;
            }
            if (x > y) {
                return false;
            }
        }
        x = 0; y = 0;

        for (int i = start.size() - 1; i >= 0; i--) {
            if (start[i] == 'R') {
                x++;
            }
            if (end[i] == 'R') {
                y++;
            }
            if (x > y) {
                return false;
            }
        }
        return true;
    }
};
