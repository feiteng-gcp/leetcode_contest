class Solution {
public:
    bool canTransform(string start, string end) {
      string s1,s2;
      vector<int> v1,v2;
      int c1,c2,i,j;
      c1=0;c2=0;s1="";s2="";
      v1.clear();
      v2.clear();
      for (i=0;i<start.length();i++)
        if (start[i]=='X') c1++;
        else
        {
          s1.push_back(start[i]);
          v1.push_back(i);
        }
      for (i=0;i<end.length();i++)
        if (end[i]=='X') c2++;
        else
        {
          s2.push_back(end[i]);
          v2.push_back(i);
        }
      if ((s1!=s2)||(c1!=c2)) return false;
      for (i=0;i<v1.size();i++)
      {
        if ((s1[i]=='L')&&(v1[i]<v2[i])) return false;
        if ((s1[i]=='R')&&(v1[i]>v2[i])) return false;
      }
      return true;
    }
};
