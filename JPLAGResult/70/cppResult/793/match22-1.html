class Solution {
  unordered_map<string, string> mt = {
      {"XL", "LX"},
      {"LX", "XL"},
      {"RX", "XR"},
      {"XR", "RX"},

  };

 public:
  bool canTransform(string start, string end) {
    const int n = start.size();
    string s1;
    string s2;
    for (auto c : start)
      if (c != 'X') s1 += c;
    for (auto c : end)
      if (c != 'X') s2 += c;
    if (s1 != s2) return false;

    vector<int> r1;
    vector<int> r2;
    vector<int> l1;
    vector<int> l2;

    for (int i = 0; i < n; ++i) {
      if (start[i] == 'L')
        l1.push_back(i);
      else if (start[i] == 'R')
        r1.push_back(i);
    }
    for (int i = 0; i < n; ++i) {
      if (end[i] == 'L')
        l2.push_back(i);
      else if (end[i] == 'R')
        r2.push_back(i);
    }


    for (int i = 0; i < l1.size(); ++i) {
      if (l1[i] < l2[i]) return false;
    }

    for (int i = 0; i < r1.size(); ++i) {
      if (r1[i] > r2[i]) return false;
    }
    return true;
  }
};
