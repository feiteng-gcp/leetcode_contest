class Solution {
public:
    bool canTransform(string start, string end) {
        // observation: RL can never swap, so two conditions must be met:
        // 1. LR sequence to be the same
        // 2. number of X are the same
        int l1 = start.size();
        int l2 = end.size();
        if ( l1 != l2 ) {
            return false;
        }
        vector< char > r1;
        vector< char > r2;
        vector< int > lidx1;
        vector< int > lidx2;
        vector< int > ridx1;
        vector< int > ridx2;
        
        for ( int i = 0; i < l1; ++i ) {
            if ( start[ i ] != 'X' ) {
                r1.push_back( start[ i ] );
                if ( start[ i ] == 'L' ) {
                    lidx1.push_back(i);
                } else {
                    ridx1.push_back( i );
                }
            }
        }
        for ( int i = 0; i < l1; ++i ) {
            if ( end[ i ] != 'X' ) {
                r2.push_back( end[ i ] );
                if ( end[ i ] == 'L' ) {
                    lidx2.push_back(i);
                } else {
                    ridx2.push_back( i );
                }
            }
        }
        if ( r1.size() != r2.size() ) {
            return false;
        }
        
        for ( size_t i = 0; i < r1.size(); ++i ) {
            if ( r1[ i ] != r2[ i ] ) {
                return false;
            }
        }
        
        // L can only move to left, R can only move to right
        for ( size_t i = 0; i < lidx1.size(); ++i ) {
            if ( lidx1[i] <lidx2[i] ) {
                return false;
            }
        }
        for ( size_t i = 0; i < ridx2.size(); ++i ) {
            if ( ridx1[i] >ridx2[i] ) {
                return false;
            }
        }
        
        return true;
    }
};
