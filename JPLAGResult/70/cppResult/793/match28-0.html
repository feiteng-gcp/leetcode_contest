class Solution {
public:
    bool canTransform(string start, string end) {
        if (start.size() != end.size()) return false;
        int sz = start.size();
        vector<int> s;
        vector<int> t;
        for (int i = 0; i < sz; i++) {
            if (start[i] != 'X') s.push_back(i);
        }
        for (int i = 0; i < sz; i++) {
            if (end[i] != 'X') t.push_back(i);
        }
        if (s.size() != t.size()) return false;
        sz = s.size();
        for (int i = 0; i< sz; i++){
            int sx = s[i], tx = t[i];
            if(start[sx] != end[tx]) return false;
            if(start[sx] == 'L' && sx < tx) return false;
            if(start[sx] == 'R' && sx > tx) return false;
        }
        return true;
    }
};
