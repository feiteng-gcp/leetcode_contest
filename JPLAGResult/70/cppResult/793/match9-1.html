class Solution {
public:
    bool canTransform(string start, string end) {
        if(start.size()!=end.size()) return false;
        int len = start.size();

        int i=0, j=0;
        while(i<len&&j<len){
            while(i<len&&start[i] == 'X') i++;
            while(j<len&&end[j] == 'X') j++;
            if(start[i]!=end[j]) return false;
            if(start[i] == 'L'&&i<j){
                return false;
            }
            if(start[i] == 'R'&&i>j){
                return false;
            }
            i++;
            j++;
        }
        return true;
    }
};
