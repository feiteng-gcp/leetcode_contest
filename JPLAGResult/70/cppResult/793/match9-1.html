class Solution {
public:
  bool canTransform(string start, string end) {

    if (start.size() != end.size()) return false;
    init(start, end);
    for (int i = 0; i < start.size(); ++i) {
      if (!matchIndex(i)) return false;
    }
    return true;
  }

  bool matchIndex(int i) {
    if (current_[i] == end_[i]) return true;
    // now current_[i] != end_[i]
    if (current_[i] == 'L') return false;
    if (end_[i] == 'R') return false;
    // left can be {X, R}, right can be {X, L}
    // note that R -> L not possible
    if (current_[i] == 'R' && end_[i] == 'L') return false;
    // only possible combination:
    // X -> L, R -> X
    // 1. X->L
    if (current_[i] == 'X') {
      if (end_[i] != 'L') return false;
      return bubbleSwapL(i);
    // 2. R->X
    } else {
      assert(current_[i] == 'R');
      if (end_[i] != 'X') return false;
      return bubbleSwapR(i);
    }
  }

  bool bubbleSwapL(int idx) {
    assert(current_[idx] == 'X');
    for (int i = idx + 1; i < size_; ++i) {
      switch(current_[i]) {
        case 'R':
          return false;
        case 'X':
          break;
        case 'L':
          swap(current_[i], current_[idx]);
          return true;
      }
    }
    return false;
  }

  bool bubbleSwapR(int idx) {
    assert(current_[idx] == 'R');
    for (int i = idx + 1; i < size_; ++i) {
      switch(current_[i]) {
        case 'L':
          return false;
        case 'R':
          break;
        case 'X':
          swap(current_[i], current_[idx]);
          return true;
      }
    }
    return false;
  }
  

private:
  void init(string start, string end) {
    current_ = start;
    end_ = end;
    size_ = start.size();
  }
  string current_;
  string end_;
  int size_;
};
