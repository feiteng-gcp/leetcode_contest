class Solution {
public:
    bool canTransform(string start, string end) {
        int n = start.length();
        vector<int> left_start (0);
        vector<int> right_start (0);
        vector<int> left_end (0);
        vector<int> right_end (0);
        for (int i = 0; i < n; ++i) {
            if (start[i] == 'L') {
                left_start.push_back(i);
            }
            if (start[i] == 'R') {
                right_start.push_back(i);
            }
            if (end[i] == 'L') {
                left_end.push_back(i);
            }
            if (end[i] == 'R') {
                right_end.push_back(i);
            }
        }
        if (left_start.size() != left_end.size()) {
            return false;
        }
        if (right_start.size() != right_end.size()) {
            return false;
        }
        
        int l = left_start.size();
        int r = right_start.size();
        
        int overlap_left[n]={0};
        int overlap_right[n]={0};
        
        for (int i = 0; i < l; ++i) {
            if (left_start[i] < left_end[i]) return false;
            for (int j = left_end[i]; j <= left_start[i]; ++j) {
                overlap_left[j] = 1;
            }
        }
        
        for (int i = 0; i < r; ++i) {
            if (right_start[i] > right_end[i]) return false;
            for (int j = right_start[i]; j <= right_end[i]; ++j) {
                overlap_right[j] = 1;
            }
        }
        
        for (int i = 0; i < n; ++i) {
            if (overlap_left[i] == 1 && overlap_right[i] == 1) return false;
        }
        return true;
    }
};
