class Solution {
public:
    string clear(string s)
    {
        string ans;
        for(auto i:s) 
            if(i!='X') ans+=i;
        return ans;
    }
    vector<int> trans(string s)
    {
        vector<int> ans;
        for(int i=0;s[i];i++)
            if(s[i]!='X') ans.push_back(i),cout<<i<<" ";
        cout<<endl;
        return ans;
    }
    bool canTransform(string s, string t) {
        if(s.length()!=t.length()) return 0;
        if(clear(s)!=clear(t)) return 0;       
        string ss=clear(s);
        vector<int> ps=trans(s),pt=trans(t);
       // cout<<"H"<<endl;
      //  cout<<ss<<endl;
        
        for(int i=0;ss[i];i++)
        {
            //cout<<ss[i]<<' '<<ps[i]<<" "<<pt[i]<<endl;
            if(ss[i]=='L')
        {
            if(ps[i]<pt[i]) return 0;
        }else 
        {
            if(ps[i]>pt[i]) return 0;
        }
        }
        return 1;
        
    }
};
