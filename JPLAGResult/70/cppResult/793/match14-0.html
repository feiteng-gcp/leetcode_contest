class Solution {
public:
    bool canTransform(string start, string end) {
        string a1="", a2="";
        vector<int> S, E;
        for(int i=0;i<start.length();i++){
            if(start[i]!='X') {a1+=start[i]; S.push_back(i);}
            if(end[i]!='X'){ a2+=end[i]; E.push_back(i);}
        }
        
    
        if (a1!=a2) return false;
        for(int i =0;i<S.size();i++){
            if(a1[i]=='L' && S[i] < E[i]) return false;
            if(a1[i]=='R' && S[i] > E[i]) return false;

        }
        return true;
