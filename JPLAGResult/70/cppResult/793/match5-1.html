class Solution {
public:
    bool canTransform(string s, string e) {
        string s0, e0;
        for(char c : s) if(c != 'X') s0.push_back(c);
        for(char c : e) if(c != 'X') e0.push_back(c);
        if(s0 != e0) return false;
        vector<int> sl, sr, el, er;
        for(int i = 0; i < s.size(); ++i) {
            if(s[i] == 'L') sl.push_back(i);
            if(s[i] == 'R') sr.push_back(i);
            if(e[i] == 'L') el.push_back(i);

            if(e[i] == 'R') er.push_back(i);
        }
        
        for(int i = 0; i < sl.size(); ++i) {
            if(sl[i] < el[i]) return false;
        }
        
        for(int i = 0; i < sr.size(); ++i) {
            if(sr[i] > er[i]) return false;
        }  
        
        return true;
    }
};
