class Solution {
public:
    bool canTransform(string start, string end) {
        string L, R;
        int cnt = 0;
        vector<int> a, b;
        for (auto& c : start) {
            if (c != 'X') L += c;
            else ++cnt;
        }
        for (auto& c : end) {
            if (c != 'X') R += c;
            else --cnt;
        }
        if (!(L == R && cnt == 0)) return false;
        for (int k = 0; k < start.size(); ++k) {
            if (start[k] == 'L') a.push_back(k);
        }
        for (int k = 0; k < end.size(); ++k) {
            if (end[k] == 'L') b.push_back(k);
        }
        for (int i = 0; i < a.size(); ++i) {
            if (a[i] < b[i]) return false;
        }
        a.clear();
        b.clear();
        for (int k = 0; k < start.size(); ++k) {
            if (start[k] == 'R') a.push_back(k);
        }
        for (int k = 0; k < end.size(); ++k) {
            if (end[k] == 'R') b.push_back(k);
        }
        for (int i = 0; i < a.size(); ++i) {
            if (a[i] > b[i]) return false;
        }
        return true;
    }
};
