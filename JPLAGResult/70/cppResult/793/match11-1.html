class Solution {
public:
    bool canTransform(string start, string end) {
        int a[10005],b[10005],c[10005],d[10005],i,n=0,m=0;
        for(i=0;i<start.size();i++)if(start[i]!='X')
        {
            a[++n]=i;
            b[n]=start[i];
        }
        for(i=0;i<end.size();i++)if(end[i]!='X')
        {
            c[++m]=i;
            d[m]=end[i];

        }
        if(n!=m)return 0;
        for(i=1;i<=n;i++)
        {
            if(b[i]!=d[i])return 0;
            if(b[i]=='L'&&a[i]<c[i]||b[i]=='R'&&a[i]>c[i])return 0;
        }
        return 1;
    }
};
