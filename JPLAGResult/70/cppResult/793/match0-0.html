class Solution {
public:
    bool canTransform(string start, string end) {
        unordered_set<string> exist;
        vector<string> curr, next;
        curr.push_back(start);
        int n = start.size();
        if (start.size() != end.size())
            return false;
        
        if (invariance(start, end) == 0)
            return false;
        
        string s1, e1;
        for (int i = 0; i < n; i++) {
            if (start[i] == 'R' || start[i] == 'L') {
                s1.push_back(start[i]);
            }
            if (end[i] == 'R' || end[i] == 'L') {
                e1.push_back(end[i]);
            }
        }
        
        if (s1 == e1)
            return true;
        else
            return false;
        
        
        while (!curr.empty()) {
            for (auto str : curr) {
                if (str == end)
                    return true;
                exist.insert(str);
                // flip R
                for (int i = 0; i < n - 1; i++) {
                    if (str[i] == 'R' && str[i + 1] == 'X') {
                        swap(str[i], str[i + 1]);
                        if (exist.find(str) == exist.end() && invariance(str, end)) {
                            next.push_back(str);
                        }
                        swap(str[i], str[i + 1]);
                    }
                }
                
                for (int i = 1; i < n; i++) {
                    if (str[i] == 'L' && str[i - 1] == 'X') {
                        swap(str[i], str[i - 1]);
                        if (exist.find(str) == exist.end() && invariance(str, end)) {
                            next.push_back(str);
                        }
                        swap(str[i], str[i - 1]);
                    }
                }
                
            }
            
            curr = next;
            next.clear();
        }
        return false;
    }
    
    bool invariance(string a, string b) {
        int suma = 0, sumb = 0;
        for (int i = 0; i < a.size(); i++) {
            if (a[i] == 'R')
                suma++;
            if (b[i] == 'R')
                sumb++;
            if (suma < sumb)
                return false;
        }
        suma = 0; sumb = 0;
        for (int i = a.size() - 1; i >= 0; i--) {
            if (a[i] == 'L')
                suma++;
            if (b[i] == 'L')
                sumb++;
            if (suma < sumb)
                return false;
        }
        return true;
        
    }
};
