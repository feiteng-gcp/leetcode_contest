class Solution {
public:
    bool canTransform(string start, string end) {
        string rs;
        vector<int> is;
        int i = 0;
        for (char c : start) {
            if (c == 'L' || c == 'R') {
                rs += c;
                is.push_back(i);
            }
            i++;
        }
        string re;
        vector<int> ie;
        i = 0;
        for (char c : end) {
            if (c == 'L' || c == 'R') {
                re += c;
                ie.push_back(i);
            }
            i++;
        }
        if (rs == re) {

            for (i = 0; i < rs.length(); i++) {
                // cout << rs[i] << is[i] << ie[i] << endl;
                if (rs[i] == 'L') {
                    if (is[i] < ie[i]) {
                        return false;
                    }
                } else {
                    if (is[i] > ie[i]) {
                        return false;
                    }
                }
            }
            return true;
        }
        return false;
    }
};
