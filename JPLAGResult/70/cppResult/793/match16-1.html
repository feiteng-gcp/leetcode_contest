class Solution {
public:
    bool canTransform(string start, string end) {
        if (start.length() != end.length()) {
            return false;
        }
        string ss, ee;
        vector<int> rs, re, ls, le;
        for (int i = 0; i < start.length(); i++) {
            if (start[i] != 'X') {
                ss += start[i];

                if (start[i] == 'L') {
                    ls.push_back(i);
                } else {
                    rs.push_back(i);
                }
            }
        }
        for (int i = 0; i < end.length(); i++) {
            if (end[i] != 'X') {
                ee += end[i];

                if (end[i] == 'L') {
                    le.push_back(i);
                } else {
                    re.push_back(i);
                }
            }
        }
        if (ss.compare(ee) != 0) {
            return false;
        }
        for (int i = 0; i < ls.size(); i++) {
            if (ls[i] < le[i]) {
                return false;
            }
        }
        for (int i = 0; i < rs.size(); i++) {
            if (rs[i] > re[i]) {
                return false;
            }
        }
        return true;
    }
};
