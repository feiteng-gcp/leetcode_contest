class Solution {
public:
    bool canTransform(string start, string end) {
        if(start.length()!=end.length())return false;
        vector<int> l1;
        vector<int> r1;
        vector<int> l2;
        vector<int> r2;
        string s1;
        string s2;
        int len = start.length();
        for(int i = 0;i<len;++i){
            if(start[i] == 'L'){
                l1.push_back(i);
                s1.push_back('L');
            }
            else if(start[i] == 'R'){
                r1.push_back(i);
                s1.push_back('R');
            }
            if(end[i] == 'L'){
                l2.push_back(i);
                s2.push_back('L');
            }
            else if(end[i] == 'R'){
                r2.push_back(i);
                s2.push_back('R');
            }
        }
        if(s1 != s2)return false;
        for(int i = 0;i<l1.size();++i){
            if(l1[i] < l2[i])return false;
        }
        for(int i = 0;i<r1.size();++i){
            if(r1[i] > r2[i])return false;
        }
        return true;
    }
};
