class Solution {
public:
    string process(string now, vector< pair<char, int> >& list_char) {
        string ret = "";
        for (int i = 0; i < now.size(); i++) {
            if (now[i] == 'R')
                ret += "R", list_char.push_back(make_pair('R', i));
            if (now[i] == 'L')
                ret += "L", list_char.push_back(make_pair('L', i));
        }
        return ret;
    }
    
    
    
    bool canTransform(string start, string end) {
        if (start.size() != end.size())
            return false;        
        vector< pair<char, int> > vpci_st, vpci_nd;
        string st = process(start, vpci_st), nd = process(end, vpci_nd);
                

        if (st != nd || vpci_st.size() != vpci_nd.size())
            return false;
        
        for (int i = 0; i < vpci_st.size(); i++) {
            if ( vpci_st[i].first == 'R' && vpci_st[i].second > vpci_nd[i].second)
                return false;
            if ( vpci_st[i].first == 'L' && vpci_st[i].second < vpci_nd[i].second)
                return false;
        }
        return true;
    }
};
