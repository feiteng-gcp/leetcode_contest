class Solution {
public:
    bool canTransform(string start, string end) {
        int n = start.size();
        vector<pair<char,int>> pos1, pos2;      
        for (int i = 0; i < n; ++i) {
            if (start[i] != 'X') {
                pos1.push_back(make_pair(start[i], i));
            };
            if (end[i] != 'X') {
                pos2.push_back(make_pair(end[i], i));
            };
        };
        if (pos1.size() != pos2.size()) return false;
        
        for (int i = 0; i < pos1.size(); ++i) {
            if (pos1[i].first != pos2[i].first) return false;

            if (pos1[i].first == 'R') {
                if (pos1[i].second > pos2[i].second) return false;
            }
            else {
                if (pos1[i].second < pos2[i].second) return false;
            };
        };
        return true;
        
        
    
    }
};
