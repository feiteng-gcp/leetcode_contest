        char t = a;

        vector<int> R;
        vector<int> L;
        string RL;

            if(start[i]=='R') {
                R.push_back(i);
                RL+='R';
            }
            if(start[i]=='L') {
                L.push_back(i);
                RL+='L';
            }
            
        }
        vector<int> R2;
        vector<int> L2;
        string RL2;
        for(int i=0;i<end.size();i++){
            if(end[i]=='R') {
                R2.push_back(i);
                RL2+='R';
            }
            if(end[i]=='L') {
                L2.push_back(i);
                RL2+='L';
            }
        }
        if(R.size()!=R2.size()) return false;
        if(L.size()!=L2.size()) return false;
        if(RL!=RL2) return false;
        for(int i=0;i<R.size();i++){
            if(R[i]>R2[i]) return false;
        }
        for(int i=0;i<L.size();i++){
            if(L[i]<L2[i]) return false;
        }
        return true;
