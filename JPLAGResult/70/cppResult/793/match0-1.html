class Solution {
public:
    bool canTransform(string start, string end) {

        int L = 0, R = 0;
        for (int i = 0; i < end.size(); i++) {
            if (start[i] == 'L') {
                L++;
            }
            if (end[i] == 'L') {
                L--;
            } else if (end[i] == 'R') {
                if (L != 0) return false;
            }
            if (L > 0) return false;
        }

        for (int i = end.size() - 1; i >= 0; i--) {
            if (start[i] == 'R') {
                R++;
            }
            if (end[i] == 'R') {
                R--;
            } else if (end[i] == 'L') {
                if (R != 0) return false;
            }
            if (R > 0) return false;
        }
        return L == R;
    }
};
