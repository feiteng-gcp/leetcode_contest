class Solution {
public:
    bool canTransform(string start, string end) {
        int len=start.size();
	string s="",e="";
	for(int i=0;i<len;i++)
	{
		if(start[i]!='X')s+=start[i];
		if(end[i]!='X')e+=end[i];
	}
	if(s!=e)return false;

	int dif=0;
	for(int i=0;i<len;i++)
	{
		if(start[i]=='L')dif--;
		if(end[i]=='L')dif++;
		if(dif<0)return false;
	}

	dif=0;
	for(int i=0;i<len;i++)
	{
		if(start[i]=='R')dif++;
		if(end[i]=='R')dif--;
		if(dif<0)return false;
	}
	return true;
    }
};
