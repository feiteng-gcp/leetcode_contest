class Solution {
public:
    bool canTransform(string start, string end) {
        string rlstart, rlend;
        int len = start.size();
        if (len == 0) {
            return true;
        }
        
        vector<int> Lstart, Lend, Rstart, Rend;
        
        for (int i = 0; i < len; i++) {
            if (start[i] != 'X') {
                rlstart += start[i];
                if (start[i] == 'L') {
                    Lstart.push_back(i);
                } else {
                    Rstart.push_back(i);
                }
            } 
            if (end[i] != 'X') {
                rlend += end[i];
                if (end[i] == 'L') {
                    Lend.push_back(i);
                } else {
                    Rend.push_back(i);
                }
                
            } 
        }
        

        if (rlstart != rlend) {
            return false;
        }
        
        
        for (int i = 0; i < Lstart.size(); i++) {
            if (Lstart[i] < Lend[i]) {
                return false;
            }
        }
        
        for (int i = 0; i < Rstart.size(); i++) {
            if (Rstart[i] > Rend[i]) {
                return false;
            }
        }
        
     /*   int xlen = xpos_start.size();
        for (int i = 0; i < xlen; i++) {
            if (xpos_start[i] == xpos_end[i]) {
                continue;
            } else if (xpos_start[i] < xpos_end[i]) {
                int istart = xpos_start[i] + 1;
                for (int j = istart; j <= xpos_end[i]; i++) {
                    if (start[j] != 'L') {
                        return false;
                    }
                }
            } else {
                int istart = xpos_start[i] - 1;
                for (int j = istart; j >= xpos_end[i]; j--) {
                    if (start[j] != 'R') {
                        return false;
                    }
                }
            }
        }*/
        
        return true;
        
        
    }
};
