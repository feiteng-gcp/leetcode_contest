class Solution {
public:
    bool canTransform(string start, string end) {
        int sz = start.size();
        vector<pair<char, int>> v1;
        vector<pair<char, int>> v2;
        for(int i=0; i<sz; i++){
            if(start[i] == 'R'){
                v1.push_back({'R', i});
            }else if(start[i] == 'L'){
                v1.push_back({'L', i});
            }
            if(end[i] == 'R'){
                v2.push_back({'R', i});
            }else if(end[i] == 'L'){
                v2.push_back({'L', i});
            }
        }
        int v_sz1 = v1.size();
        int v_sz2 = v2.size();
        if(v_sz1 != v_sz2){
            return false;
        }
        for(int i=0; i<v_sz1; i++){
            if(v1[i].first != v2[i].first){
                return false;

            }else{
                if(v1[i].first == 'L'){
                    if(v1[i].second < v2[i].second){
                        return false;
                    }
                }else{
                    if(v1[i].second > v2[i].second){
                        return false;
                    }
                }
            }
        }
        return true;
    }
};
