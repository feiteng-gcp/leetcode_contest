class Solution {
public:
    bool canTransform(string start, string end) {
        vector<int> l1, l2, r1, r2;
        string s1 = "", s2 = "";
        for (int i = 0; i < start.size(); ++ i) {
            if (start[i] == 'L') l1.push_back(i);
            if (start[i] == 'R') r1.push_back(i);
            if (start[i] != 'X') s1 += start[i];
        }
        for (int i = 0; i < end.size(); ++ i) {
            if (end[i] == 'L') l2.push_back(i);
            if (end[i] == 'R') r2.push_back(i);
            if (end[i] != 'X') s2 += end[i];
        }
        if (s1 != s2) return false;
        if (l1.size() != l2.size() || r1.size() != r2.size()) return false;
        for (int i = 0; i < l1.size(); ++ i)
            if (l2[i] > l1[i]) return false;
        for (int i = 0; i < r1.size(); ++ i)
            if (r2[i] < r1[i]) return false;
        return true;
    }
};
