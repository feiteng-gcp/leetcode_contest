const int maxn = 50;
int fa[maxn * maxn];

int find(int x) {
    return fa[x] == x ? x : fa[x] = find(fa[x]);
}

bool merge(int u, int v) {
    u = find(u);
    v = find(v);
    if (u == v) {
        return false;
    }
    fa[u] = v;
    return true;
}

struct Square {
    int r, c, h;

    bool operator<(const Square &rhs) const {
        return h < rhs.h;
    }
};


int dir[4][2] = {{0,  1},
                 {0,  -1},
                 {-1, 0},
                 {1,  0}};

class Solution {
public:
    int swimInWater(vector<vector<int>> &grid) {
        int n = grid.size();
        vector<Square> squares;
        squares.reserve(n * n);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                fa[i * n + j] = i * n + j;
                squares.push_back(Square{i, j, grid[i][j]});
            }
        }
        sort(squares.begin(), squares.end());
        for (int t = 0; t < squares.size(); t++) {
            int r = squares[t].r, c = squares[t].c;
            for (int d = 0; d < 4; d++) {
                int nr = r + dir[d][0], nc = c + dir[d][1];
                if (nr >= 0 && nr < n && nc >= 0 && nc < n && grid[nr][nc] <= t) {
                    merge(r * n + c, nr * n + nc);
                }
            }
            if (find(0 * n + 0) == find((n - 1) * n + n - 1)) {
                return t;
            }
        }
        return -1;
    }
};
