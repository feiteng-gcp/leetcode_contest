class Solution {
public:
    int genK32(int x, int y) {
        return (x << 16) | y;
    }
    bool check(vector< vector<int> > & vv, int x) {
        if(vv[0][0] > x) return false;
        int N = vv.size();
        queue<int> qi;
        vector< vector<bool> > visited(N, vector<bool>(N, false));
        qi.push(genK32(0, 0));
        visited[0][0] = true;
        const int move[][2] = {
            {-1, 0}, {1, 0}, {0, -1}, {0, 1}
        };
        while(!qi.empty()) {
            int cur = qi.front();
            qi.pop();
            int px = (cur >> 16) & 0xFFFF;
            int py = cur & 0xFFFF;
            for(int i = 0; i < 4; i ++) {
                int nx = px + move[i][0];
                int ny = py + move[i][1];
                if(nx < 0 || nx >= N || ny < 0 || ny >= N || visited[nx][ny]) continue;
                if(vv[nx][ny] > x) continue;
                if(nx == N - 1 && ny == N - 1) return true;
                qi.push(genK32(nx, ny));
                visited[nx][ny] = true;
            }
        }
        return false;
    }
    int swimInWater(vector< vector<int> > & grid) {
        int N = grid.size();

        int lo = 0, hi = N * N - 1;
        while(lo < hi) {
            int mid = lo + (hi - lo) / 2;
            if(!check(grid, mid)) lo = mid + 1;
            else hi = mid;
        }
        return lo;
    }
};
