#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

class Solution {

private:
    int N, M;
    int d[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};

public:
    int swimInWater(vector<vector<int>>& grid) {

        this->N = grid.size();
        this->M = grid[0].size();

        vector<int> heights;
        for(int i = 0 ; i < N ; i ++)
            for(int j = 0 ; j < M ; j ++)
                heights.push_back(grid[i][j]);

        sort(heights.begin(), heights.end());

        int l = 0, r = heights.size();
        while(l != r){
            int mid = (l + r) / 2;
            if(ok(grid, heights[mid]))
                r = mid;
            else
                l = mid + 1;
        }

        return heights[r];
    }

private:
    bool ok(const vector<vector<int>>& grid, int h){

        if(grid[0][0] > h)
            return false;
        
        vector<vector<bool>> g;
        for(int i = 0 ; i < N ; i ++){
            vector<bool> row;
            for(int j = 0 ; j < M ; j ++)
                row.push_back(grid[i][j] <= h ? true : false);
            g.push_back(row);
        }

        return connected(g);
    }

    bool connected(const vector<vector<bool>>& g){

        vector<vector<bool>> visited(N, vector<bool>(M, false));
        return go(g, 0, 0, visited);
    }

    bool go(const vector<vector<bool>>& g, int x, int y,
            vector<vector<bool>>& visited){

        visited[x][y] = true;

        if(x == N-1 && y == M-1)
            return true;

        for(int i = 0 ; i < 4 ; i ++){
            int newX = x + d[i][0];
            int newY = y + d[i][1];
            if(inArea(newX, newY) && !visited[newX][newY] && g[newX][newY])
                if(go(g, newX, newY, visited))
                    return true;
        }

        return false;
    }

    bool inArea(int x, int y){
        return x >= 0 && x < N && y >= 0 && y < M;
    }
};
