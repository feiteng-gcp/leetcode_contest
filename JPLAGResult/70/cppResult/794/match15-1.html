int l1[1000000], l2[1000000];
int dist[55][55];
bool visit[55][55];
const int dx[4] = {0, 1, 0, -1};
const int dy[4] = {1, 0, -1, 0};
bool inq[55][55];

class Solution {
public:
    int swimInWater(vector<vector<int>>& grid) {
        int n = (int)grid.size();
        memset(visit, 0, sizeof(visit));
        memset(inq, 0, sizeof(inq));
        int s = 0, t = 1;
        l1[1] = l2[1] = 0;
        dist[0][0] = grid[0][0];
        inq[0][0] = true;
        while (s < t) {
            ++s;

            int x = l1[s], y = l2[s];
            inq[x][y] = false;
            for (int i = 0; i < 4; i++) {
                int nx = x + dx[i], ny = y + dy[i];
                if (nx < 0 || nx >= n || ny < 0 || ny >= n) {
                    continue;
                }
                if (!visit[nx][ny]) {
                    visit[nx][ny] = true;
                    dist[nx][ny] = max(dist[x][y], grid[nx][ny]);
                    l1[++t] = nx, l2[t] = ny;
                    inq[nx][ny] = true;
                } else if (dist[nx][ny] > max(dist[x][y], grid[nx][ny])) {
                    dist[nx][ny] = max(dist[x][y], grid[nx][ny]);
                    if (!inq[nx][ny]) {
                        inq[nx][ny] = true;
                        l1[++t] = nx, l2[t] = ny;
                    }
                }
            }
        }
        return dist[n - 1][n - 1];
    }
};
