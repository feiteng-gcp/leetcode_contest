class Solution {
public:
    struct Node{
        int x, y, h;
        bool operator<(const Node& a) const {
            return (h > a.h);
        }
    };
    
    int swimInWater(vector<vector<int>>& grid) {
        int n = grid.size();
        if (n < 2) { return 0; }
        int ans = 0;
        priority_queue<Node> heap;
        Node temp;
        temp.x = temp.y = 0;
        temp.h = grid[0][0];
        heap.push(temp);
        int dx[4] = {0, 0, 1, -1};
        int dy[4] = {1, -1, 0, 0};
        vector<vector<bool>> used(n, vector<bool>(n, false));
        while (true) {
            Node cur = heap.top();
            heap.pop();
            ans = max(ans, cur.h);
            if (cur.x == n - 1 && cur.y == n - 1) { return ans; }
            for (int i = 0; i < 4; i++) {
                int x = cur.x + dx[i], y = cur.y + dy[i];
                if (x < 0 || x >= n || y < 0 || y >= n || used[x][y]) { continue; }
                Node temp;
                temp.x = x; temp.y = y;
                temp.h = grid[x][y];
                used[x][y] = true;
                heap.push(temp);
            }
        }
        return grid[n - 1][n - 1];
    }
};
