class Solution {
public:
    bool visited[101][101];
    
    vector<vector<int> >v;
    int di[4]={-1,1,0,0};
    int dj[4]={0,0,1,-1};
    int R,C;
    
    void dfs(int x,int y,int level){
        visited[x][y]=1;
        
        for(int i=0;i<4;i++){
            int X=x+di[i];
            int Y=y+dj[i];
            if(X>=0 && X<R && Y>=0 && Y<C){
                if(!visited[X][Y] && v[X][Y]<=level)
                    dfs(X,Y,level);
            }
        }
    }
    
    int swimInWater(vector<vector<int>>& grid) {
        v=grid;
        int lo=v[0][0];int hi=2500;
        R=grid.size();
        C=grid[0].size();
        
        for(int caso=0;caso<15;caso++){
            memset(visited,0,sizeof(visited));
            int me=(lo+hi)/2;
            dfs(0,0,me);
            if(visited[R-1][C-1]){
                hi=me;
            }else{
                lo=me+1;
            }
            
        }
        
        return lo;
    }
};
