class Solution {
public:
    int swimInWater(vector<vector<int>>& grid) {
        if(grid.empty()) return 0;
        vector<vector<int>> dirs = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        int m = grid.size(), n = grid[0].size();
        vector<vector<bool>> visited(m, vector<bool>(n, false));
        map<int, pair<int, int>> heap;
        heap[grid[0][0]] = {0, 0};
        visited[0][0] = true;
        int res = 0;
        while(!heap.empty()){
            pair<int, int> tmp = (*heap.begin()).second;
            int time = (*heap.begin()).first;
            res = max(res, time);
            heap.erase(heap.begin());
            if(tmp.first == m-1&&tmp.second == m-1){
                return res;
            }
            for(auto dir:dirs){
                int x = tmp.first + dir[0];
                int y = tmp.second + dir[1];
                if(x>=0&&x<m&&y>=0&&y<n&&!visited[x][y]){
                    heap[grid[x][y]] = {x, y};
                    visited[x][y] = true;
                }
            }
        }
        return m*m-1;
    }
};
