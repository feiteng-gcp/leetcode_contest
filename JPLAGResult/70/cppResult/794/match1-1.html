class Solution {
public:
    int swimInWater(vector<vector<int>>& grid) {
        int n = grid.size();
        int height[n][n];
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                height[i][j]=3000;
            }
        }
        bool changed;
        height[0][0]=grid[0][0];
        do
        {
            changed=false;

            for(int i=0;i<n;i++)
            {
                for(int j=0;j<n;j++)
                {
                    if(i>0&&height[i-1][j]>height[i][j])
                    {
                        if(grid[i-1][j]<grid[i][j])
                        {
                            height[i-1][j]=height[i][j];
                            changed=true;
                        }
                        else if(height[i-1][j]>max(height[i][j],grid[i-1][j]))
                        {
                            height[i-1][j]=max(height[i][j],grid[i-1][j]);
                            changed=true;
                        }
                    }
                    if(i+1<n&&height[i+1][j]>height[i][j])
                    {
                        if(grid[i+1][j]<grid[i][j])
                        {
                            height[i+1][j]=height[i][j];
                            changed=true;
                        }
                        else if(height[i+1][j]>max(height[i][j],grid[i+1][j]))
                        {
                            height[i+1][j]=max(height[i][j],grid[i+1][j]);
                            changed=true;
                        }
                    }
                    if(j>0&&height[i][j-1]>height[i][j])
                    {
                        if(grid[i][j-1]<grid[i][j])
                        {
                            height[i][j-1]=height[i][j];
                            changed=true;
                        }
                        else if(height[i][j-1]>max(height[i][j],grid[i][j-1]))
                        {
                            height[i][j-1]=max(height[i][j],grid[i][j-1]);
                            changed=true;
                        }
                    }
                    if(j+1<n&&height[i][j+1]>height[i][j])
                    {
                        if(grid[i][j+1]<grid[i][j])
                        {
                            height[i][j+1]=height[i][j];
                            changed=true;
                        }
                        else if(height[i][j+1]>max(height[i][j],grid[i][j+1]))
                        {
                            height[i][j+1]=max(height[i][j],grid[i][j+1]);
                            changed=true;
                        }
                    }
                }
            }
        }while(changed);
        return height[n-1][n-1];
    }
};
