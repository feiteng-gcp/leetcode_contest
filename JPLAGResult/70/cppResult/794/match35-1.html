const int N = 2500;

const int d[4][2] = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
int f[N];
int find(int x) {
    if (f[x] != f[f[x]]) f[x] = find(f[x]);
    return f[x];
}
class Solution {
public:
    int swimInWater(vector<vector<int>>& a) {
        int n = a.size();
        int m = n * n;
        for (int ret = 1; ret <= m; ++ret) {
            for (int i = 0; i < m; ++i) f[i] = i;
            for (int i = 0; i < n; ++i) {
                for (int j = 0; j < n; ++j) {
                    if (a[i][j] > ret) continue;
                    for (int k = 0; k < 4; ++k) {
                        int ni = i + d[k][0];
                        int nj = j + d[k][1];
                        if (ni < 0 || ni >= n || nj < 0 || nj >= n || a[ni][nj] > ret) continue;
                        int u = i * n + j;
                        int v = ni * n + nj;
                        int ru = find(u), rv = find(v);
                        if (ru != rv) {
                            f[rv] = ru;
                        }
                    }
                }
            }
            if (find(0) == find(n * n - 1)) return ret;
        }
        return -1;
    }
};
