class Solution {
public:
    int n;
    int mx[4]={-1,0,1,0};
    int my[4]={0,-1,0,1};
    
    bool ok(vector<vector<int>>&a,int v)
    {
        vector<bool> vis(n*n,0);
        if(a[0][0]>v) return 0;
        queue<int> q;
        q.push(0);
        vis[0]=1;
        while(q.size())
        {
            int x=q.front()/n,y=q.front()%n;q.pop();
            for(int i=0;i<4;i++)
            {
                int nx=x+mx[i];
                int ny=y+my[i];
                if(nx<0||nx>=n||ny<0||ny>=n||vis[nx*n+ny]||a[nx][ny]>v) continue;
                vis[nx*n+ny]=1;
                q.push(nx*n+ny);
                if(nx*n+ny==n*n-1) return 1;
            }
        }
        return 0;
    }
    int swimInWater(vector<vector<int>>& grid) {

        int s=0,t=50*50;
        n=grid.size();
        while(s<=t)
        {
            int mid=(s+t)>>1;
            if(ok(grid,mid)) t=mid-1;
            else s=mid+1;
        }
        return s;
    }
};
