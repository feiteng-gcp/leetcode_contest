class Solution {
public:
    const int dx[4] = {1, -1, 0, 0}, dy[4] = {0, 0, 1, -1};

bool bfs(vector<vector<int> >& grid, int N, int key)
{
    queue<pair<int, int> > q;
    vector<vector<bool> > vis(N, vector<bool>(N, false));
    q.push({0, 0});
    vis[0][0] = true;
    while(!q.empty())
    {
        pair<int, int> cur = q.front(); q.pop();
        if(cur.first == N - 1 && cur.second == N - 1) return true;
        for(int i = 0; i < 4; i++)
        {
            int nx = cur.first + dx[i], ny = cur.second + dy[i];
            if(nx >= 0 && nx < N && ny >= 0 && ny < N && !vis[nx][ny] && key >= grid[nx][ny])
            {
                q.push({nx, ny});
                vis[nx][ny] = true;
            }
        }
    }
    return false;
}

int swimInWater(vector<vector<int> >& grid)
{
    int N = (int)grid.size();
    int l = grid[0][0], r = N * N;
    while(l < r)
    {
        int m = l + (r - l) / 2;
        if(!bfs(grid, N, m)) l = m + 1;
        else r = m;
    }
    return l;
}
};
