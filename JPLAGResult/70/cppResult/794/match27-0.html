class Solution {
public:
    struct edge
    {
        int s,t,d;
        bool operator<(const edge& y)const
        {
            return d<y.d;
        }
    }e[10005];
    int f[2505];
    int get(int x)
    {
        if(f[x]==x)return x;
        return f[x]=get(f[x]);
    }
    int swimInWater(vector<vector<int>>& grid) {
        int n=grid.size(),m=0,i,j;
        for(i=0;i<n;i++)for(j=0;j+1<n;j++)
        {
            e[++m].s=i*n+j;
            e[m].t=i*n+j+1;
            e[m].d=max(grid[i][j],grid[i][j+1]);
        }
        for(i=0;i+1<n;i++)for(j=0;j<n;j++)
        {
            e[++m].s=i*n+j;
            e[m].t=i*n+j+n;
            e[m].d=max(grid[i][j],grid[i+1][j]);
        }
        sort(e+1,e+m+1);
        n*=n;
        for(i=0;i<n;i++)f[i]=i;
        for(i=1;i<=m;i++)if(get(e[i].s)!=get(e[i].t))
        {
            f[get(e[i].s)]=get(e[i].t);
            if(get(0)==get(n-1))return e[i].d;
        }
    }
};
