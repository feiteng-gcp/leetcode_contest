class Solution {
public:
    int swimInWater(vector<vector<int>>& grid) {
        int N = grid.size();
        bool visited[N][N];
        memset(visited, false, sizeof(bool)*N*N);
        vector<pair<int, int>> pos(N*N, {0, 0});
        
        for(int i=0; i<N; i++){
            for(int j=0; j<N; j++){
                pos[grid[i][j]] = {i, j};
            }
        }
        priority_queue<int, vector<int>, greater<int> > q;
        int res = grid[0][0];
        q.push(res);
        visited[0][0] = true;
        while(true){
            int cur = q.top();
            //cout<<cur<<endl;
            if(cur > res){
                res = cur;
            }
            q.pop();
            pair<int, int> cur_pos = pos[cur];
            int i=cur_pos.first;
            int j = cur_pos.second;
            if(i == N-1 && j == N-1){
                return res;
            }

            if(i > 0 && visited[i-1][j] == false){
                q.push(grid[i-1][j]);
                visited[i-1][j] = true;
            }
            if(i < N - 1 && visited[i+1][j] == false){
                q.push(grid[i+1][j]);
                visited[i+1][j] = true;
            }
            if(j > 0 && visited[i][j-1] == false){
                q.push(grid[i][j-1]);
                visited[i][j-1] = true;
            }
            if(j < N - 1 && visited[i][j+1] == false){
                q.push(grid[i][j+1]);
                visited[i][j+1] = true;
            }
        }
    }
};
