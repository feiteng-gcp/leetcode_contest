class Solution {
public:
    void explore(vector<vector<int>>& reach, vector<vector<int>>& grid, int ii, int jj, int res){
        if (ii>0 && grid[ii-1][jj] <= res && reach[ii-1][jj] == 0){
            reach[ii-1][jj] = 1;
            explore(reach, grid, ii-1, jj, res);
        }
        if (ii<grid.size()-1 && grid[ii+1][jj] <= res && reach[ii+1][jj] == 0){
            reach[ii+1][jj] = 1;
            explore(reach, grid, ii+1, jj, res);
        }
        if (jj>0 && grid[ii][jj-1] <= res && reach[ii][jj-1] == 0){
            reach[ii][jj-1] = 1;
            explore(reach, grid, ii, jj-1, res);
        }
        if (jj<grid.size()-1 && grid[ii][jj+1] <= res && reach[ii][jj+1] == 0){
            reach[ii][jj+1] = 1;
            explore(reach, grid, ii, jj+1, res);
        }
        return;
    }
    int swimInWater(vector<vector<int>>& grid) {
        int res = 0;
        int n = grid.size();
        bool found = false;
        vector<vector<int>> reach(n, vector<int>(n,0));
        reach[0][0] = 1;
        res = grid[0][0];
        // cout<<"okay"<<endl;
        while(!found){
            for (int i = 0; i < n; i++){
                for (int j = 0; j < n; j++){
                    if (reach[i][j] == 1 && grid[i][j] <= res){
                        // cout<<res<<" "<<i<<" "<<j<<" okay"<<endl;
                        explore(reach, grid, i, j, res);
                    }
                }
            }
            if (reach[n-1][n-1] == 1){
                found = true;
                break;
            }
            res++;
        }
        return res;
    }
};
