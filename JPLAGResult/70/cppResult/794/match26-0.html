int neigh[4][2]={{0,1},{1,0},{0,-1},{-1,0}};
vector<vector<int> > a;
int m,n;
int used[100][100];

int doit(int d)
{
  int i,j,x,y,xx,yy,k;
  vector<pair<int,int> > q;
  memset(used,0,sizeof(used));
  q.clear();
  if (a[0][0]<=d)
  {
    q.push_back(make_pair(0,0));
    used[0][0]=1;
  }
  for (i=0;i<q.size();i++)
  {
    x=q[i].first;
    y=q[i].second;
    for (k=0;k<4;k++)
    {
      xx=x+neigh[k][0];
      yy=y+neigh[k][1];
      if ((xx>=0)&&(xx<m)&&(yy>=0)&&(yy<n)&&(used[xx][yy]==0)&&(a[xx][yy]<=d))
      {
        q.push_back(make_pair(xx,yy));
        used[xx][yy]=1;
      }
    }
  }
  if (used[m-1][n-1]==1) return 1;
  else return 0;
}

class Solution {
public:
    int swimInWater(vector<vector<int> >& grid) {
      int left,right,mid;
      m=grid.size();
      n=grid[0].size();
      a=grid;
      left=0;right=10000;
      while (right-left>1)
      {
        mid=(left+right)/2;
        if (doit(mid)==1) right=mid;
        else left=mid;
      }
      return right;
    }
};
