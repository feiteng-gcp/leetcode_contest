class Solution {
public:
    int getf(vector<int>& f, int p) {
        return f[p]==p?p:f[p]=getf(f, f[p]);
    }
    int swimInWater(vector<vector<int>>& grid) {

        const int dir[4][2]={{-1, 0}, {0, -1}, {0, 1}, {1, 0}};
        int N=grid.size();
        if (grid[N-1][N-1]==N*N-1) return N*N-1;
        vector<int> pos(N*N);
        for (int i=0; i<N; i++)
            for (int j=0; j<N; j++) pos[grid[i][j]]=i*N+j;
        vector<int> f(N*N);
        for (int i=0; i<N*N; i++) f[i]=i;
        for (int i=0; i<N*N; i++) {
            int r=pos[i]/N, c=pos[i]%N;
            for (int k=0; k<4; k++) {
                int nextr=r+dir[k][0], nextc=c+dir[k][1];
                if (0<=nextr && nextr<N && 0<=nextc && nextc<N && grid[nextr][nextc]<i) {
                    f[getf(f, grid[nextr][nextc])]=i;
                }
                if (getf(f, grid[0][0])==getf(f, grid[N-1][N-1])) return i;
            }
        }
        return -1;
    }
};
