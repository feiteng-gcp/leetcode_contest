class Solution {
public:
    bool check(int height, const vector<vector<int> > &grid, int n) {
        int dx[] = {0, 1, 0, -1};
        int dy[] = {-1, 0, 1, 0};
        queue<pair<int, int> > que;
        que.push(make_pair(0, 0));
        vector<vector<bool> > vis(n, vector<bool>(n));
        vis[0][0] = true;
        while (!que.empty()) {
            int x = que.front().first, y = que.front().second;
            que.pop();
            for (int dir = 0; dir < 4; dir++) {
                int nx = x + dx[dir], ny = y + dy[dir];
                if (nx < 0 || nx >= n || ny < 0 || ny >= n || vis[nx][ny] || grid[nx][ny] > height || grid[x][y] > height) {
                    continue;
                }
                vis[nx][ny] = true;
                que.push(make_pair(nx, ny));
            }
        }
        return vis[n - 1][n - 1];
    }
    int swimInWater(vector<vector<int>>& grid) {
        int n = grid.size();
        int l = 0, r = n * n;
        while (l < r) {
            int mid = (l + r) / 2;
            // cout << "mid = " << mid << endl;
            if (check(mid, grid, n)) {
                r = mid;
            } else {
                l = mid + 1;
            }
        }
        return l;
    }
};
