class Solution {
  vector<vector<int>> a;
  int n;
  bool vis[55][55];
  void dfs(int x, int y, int v) {
    if (x < 0 || y < 0 || x == n || y == n || a[x][y] > v || vis[x][y]) {
      return;
    }
    vis[x][y] = true;
    dfs(x-1,y,v);
    dfs(x+1,y,v);
    dfs(x,y-1,v);
    dfs(x,y+1,v);
  }
  bool ok(int v) {
    memset(vis, 0, sizeof vis);
    dfs(0, 0, v);
    return vis[n-1][n-1];
  }
public:
  int swimInWater(vector<vector<int>>& grid) {
    a = grid;
    n = a.size();
    int l = 0, r = n * n - 1;
    while (l < r) {
      int m = (l + r) >> 1;
      if (ok(m)) r = m;
      else l = m + 1;
    }
    return l;
  }
};
