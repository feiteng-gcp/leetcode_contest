class Solution {
public:
    int n;
    vector<vector<int>> g;
    bool f(int t) {

        int dir[4][2] = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        queue<pair<int, int> > q;
        set<pair<int, int> > h;
        q.push(pair<int, int>(0, 0));
        h.insert(pair<int, int>(0, 0));
        while (!q.empty()) {
            pair<int, int> node = q.front();
            q.pop();
            int r = node.first, c = node.second;
            if (r == n - 1 && c == n - 1)
                return true;
            for (int d = 0; d < 4; d++) {
                int tr = r + dir[d][0], tc = c + dir[d][1];
                if (tr < 0 || tr >= n || tc < 0 || tc >= n)
                    continue;
                if (h.find(pair<int, int>(tr, tc)) != h.end())
                    continue;
                if (g[tr][tc] <= t) {
                    q.push(pair<int, int>(tr, tc));
                    h.insert(pair<int, int>(tr, tc));
                }
            }
        }
        return false;
    }
    int swimInWater(vector<vector<int>>& grid) {
        n = grid.size();
        g = grid;
        for (int i = grid[0][0]; i <= n * n; i++)
            if (f(i))
                return i;
        return 0;
    }
};
