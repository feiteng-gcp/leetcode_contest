class Solution {
public:
  bool check(int x) {
    set<pair<int, int>> vis;
    queue<pair<int, int>> q;
    q.push({0, 0});
    vis.insert({0, 0});
    int dx[] = {0, 1, 0, -1}, dy[] = {1, 0, -1, 0};
    while (!q.empty()) {
      auto s = q.front();
      q.pop();
      for (int i = 0; i < 4; ++i) {
        auto v = s;
        v.first += dx[i];
        v.second += dy[i];
        if (0 <= v.first && v.first < n && 0 <= v.second && v.second < n &&
            !vis.count(v) && a[v.first][v.second] <= x &&
            a[s.first][s.second] <= x) {
          vis.insert(v);
          q.push(v);
        }
      }
    }
    return vis.count({n - 1, n - 1});
  }
  int swimInWater(vector<vector<int>> &grid) {
    n = grid.size();

    a = grid;
    int s = 0, e = n * n;
    while (s <= e) {
      int mid = (s + e) >> 1;
      if (check(mid)) {
        e = mid - 1;
      } else {
        s = mid + 1;
      }
    }
    return s;
  }
  int n;
  vector<vector<int>> a;
};
