class Solution {
public:
    int swimInWater(vector<vector<int>>& grid) {
        int n = grid.size();
        auto comp = [&](int a, int b){ return grid[a/n][a%n] > grid[b/n][b%n]; };
        priority_queue<int, vector<int>, decltype(comp)> pq(comp);
        
        pq.push(n*n-1);
        unordered_set<int> visited;
        visited.insert(n*n-1);
        int maxE = 0;
        vector<vector<int>> dir({{0,1},{1,0},{0,-1},{-1,0}});
        while(!pq.empty()){
            auto cur = pq.top(); pq.pop();
            visited.insert(cur);
            
            int i=cur/n, j=cur%n;
            // cout << grid[i][j] << ' ';
            maxE = max(maxE, grid[i][j]);
            if(cur == 0) return maxE;
            for(auto d : dir){
                int i2 = i + d[0], j2 = j+d[1];
                if(i2 < 0 || i2 >= n || j2 < 0 || j2 >= n || visited.find(i2*n+j2) != visited.end()) continue;
                pq.push(i2*n+j2);
            }
        }
        
        assert(false);
        return -1;
    }
};
