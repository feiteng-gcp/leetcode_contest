class Solution {
public:
    int kthGrammar(int N, int K) {
        return dfs(N, K, 0);
    }
    int dfs(int N, int K, int cur) {
        if (N == 0) {
            return cur;
        }
        int halfsize = 1 << (N - 1);
        if (cur == 0) {
            if ( K <= halfsize) {
                return dfs(N - 1, K, 0);
            } else {
                return dfs(N - 1, K - halfsize, 1);
            }
        } else {
            if ( K <= halfsize) {
                return dfs(N - 1, K, 1);
            } else {
                return dfs(N - 1, K - halfsize, 0);
            }
        }
    }
};
