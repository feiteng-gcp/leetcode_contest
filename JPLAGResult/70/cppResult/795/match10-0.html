class Solution {
public:
    int kthGrammar(int N, int K) {
        if (N  == 1) {
            return 0;
        } else {
            int half = (1 << (N - 2));
            if (K <= half) {
                return kthGrammar(N - 1, K);
            } else {
                return 1 - kthGrammar(N - 1, K - half);
            }
        }
    }
};
