class Solution {
public:
    int kthGrammar(int N, int K) {
        if(N == 1)
        {
            return 0;
        }
        if(N == 2)
        {
            return (K == 1)? 0 : 1;
        }
        
        if(K <= (1 << (N - 2)))
        {
            return kthGrammar(N - 1, K);
        }
        else
        {
            if(K <= (1 << (N - 2)) + (1 << (N - 3)))
            {
                return kthGrammar(N - 1, K - (1 << (N - 2)) + (1 << (N - 3)));
            }
            else
            {
                return kthGrammar(N - 1, K - (1 << (N - 2)) - (1 << (N - 3)));
            }
        }
    }
};
