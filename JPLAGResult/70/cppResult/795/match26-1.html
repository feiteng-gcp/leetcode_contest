class Solution {
public:
    int kthGrammar(int N, int K) {
        if (N == 1) { return 0; }
        if (N == 2) { return K - 1; }
        int len = (1 << (N - 2));
        if (K <= len) { return kthGrammar(N - 1, K); }
        K -= len;
        if (K > len / 2) {
            return kthGrammar(N - 1, K - len / 2);
        } else {
            return kthGrammar(N - 1, K + len / 2);
        }
    }
};
