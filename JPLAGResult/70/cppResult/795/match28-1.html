class Solution {
public:
  int solve(int n, int k) {
    if (n == 1) return 0;
    int b = solve(n - 1, k/2);
    if (k % 2 == 0) {
      return b == 0 ? 0 : 1;
    } else {
      return b == 0 ? 1 : 0;
    }
  }
  int kthGrammar(int N, int K) {
    K--;
    return solve(N, K);
  }
};
