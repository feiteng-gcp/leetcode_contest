class Solution {
public:
    int kthGrammar(int n, int k) {
      if (n==1) return 0;
      int tmp;
      tmp=kthGrammar(n-1,(k+1)/2);
      if ((tmp==0)&&(k%2!=0)) return 0;
      if ((tmp==0)&&(k%2==0)) return 1;
      if ((tmp==1)&&(k%2!=0)) return 1;
      if ((tmp==1)&&(k%2==0)) return 0;
      return 0;
    }
};
