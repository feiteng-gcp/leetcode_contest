/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* find(TreeNode* root, TreeNode* father, int V, bool& finded) {
        if (root == NULL) {
            return NULL;
        }
        else if (root->val == V) {
            finded = true;
            return father;
        }
        else if (root->val < V) {
            return find(root->right, root, V, finded);
        }
        else if (root->val > V) {
            return find(root->left, root, V, finded);
        }
    }
    vector<TreeNode*> splitBST(TreeNode* root, int V) {
        vector<TreeNode*> res;
        TreeNode * father, *r, *l;
        if (root == NULL) {
            res.push_back(root);
            return res;
        }
        else if (root->val == V) {
            r = root->right;
            root->right = NULL;
            res.push_back(root);
            if (r != NULL) {
                res.push_back(r);
            }
            return res;
        }
        father = NULL;
        bool finded = false;
        father = find(root, father, V, finded);
        if (finded == false) {
            res.push_back(root);
            return res;
        }
        else {
            if (father->left != NULL && father->left->val == V) {
                l = father->left;
                father->left = l->right;
                l->right = NULL;
                res.push_back(l);
                res.push_back(root);
                return res;
            }
            else if (father->right != NULL && father->right->val == V) {
                r = father->right;
                father->right = r->right;
                r->right = NULL;
                res.push_back(r);
                res.push_back(root);
                return res;
            }
        }
        return res;
    }
};
