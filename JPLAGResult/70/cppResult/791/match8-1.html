/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> splitBST(TreeNode* root, int V) {
        vector<TreeNode*> res(2);
        if (!root) {
            return res;
        }
        if (root->val == V) {
            TreeNode* right = root->right;
            root->right = NULL;
            res[0] = root;
            res[1] = right;
        }
        else if (root->val < V) {
            vector<TreeNode*> subtree = splitBST(root->right, V);
            root->right = subtree[0];

            res[0] = root;
            res[1] = subtree[1];
        }
        else {
            vector<TreeNode*> subtree = splitBST(root->left, V);
            root->left = subtree[1];
            res[0] = subtree[0];
            res[1] = root;
        }
        return res;
    }
};
