class Solution {
public:
	vector<TreeNode*> splitBST(TreeNode* root, int V) {
		TreeNode *it = root;
		TreeNode *prev = root;
		bool side = true;
        vector<TreeNode *> result;

		while (it != nullptr) {
			if (it->val == V)
				break;

			if (V>it->val) {
				prev = it;
				it = it->right;
				side = false;
			}
			else {
				prev = it;
				it = it->left;
				side = true;
			}
		}
         if(it==nullptr){
            result.push_back(root);
            return result;
        }
        if(it->val!=V){
            result.push_back(root);
            return result;
        }
        if(root->val==it->val){
            result.push_back(root);
            return result;
        }
		if (side) {
			prev->left = it->right;
			it->right = nullptr;
		}
		else {
			prev->right = it->right;
			it->right = nullptr;
		}
		result.push_back(it);
		result.push_back(root);
		return result;
	}
};
