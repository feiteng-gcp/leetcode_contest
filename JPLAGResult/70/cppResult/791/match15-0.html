/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> splitBST(TreeNode* root, int V) {
        if (!root) return vector<TreeNode*> {NULL, NULL};
        if (root->val == V) {
            vector<TreeNode*> rtn;
            rtn.push_back(root);
            rtn.push_back(root->right);
            root->right = NULL;
            return rtn;
        }
        if (root->val < V) {
            vector<TreeNode*> split_right = splitBST(root->right, V);
            root->right = split_right[0];
            return vector<TreeNode*> {root, split_right[1]};
        }
        vector<TreeNode*> split_left = splitBST(root->left, V);
        root->left = split_left[1];
        return vector<TreeNode*> {split_left[0], root};
    }
};
