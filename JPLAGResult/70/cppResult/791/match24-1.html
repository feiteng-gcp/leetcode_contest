/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> ans;

    TreeNode* searchGreater(TreeNode *now, int V)
    {
        TreeNode *parent = NULL;
        while (now != NULL && now->val <= V) {
            parent = now;
            now = now->right;
        }
        if (now == NULL)
            return NULL;
        return parent;
    }
    TreeNode* searchLessEqual(TreeNode *now, int V)
    {
        TreeNode *parent = NULL;
        while (now != NULL && now->val > V) {
            parent = now;
            now = now->left;
        }
        if (now == NULL)
            return NULL;
        return parent;
    }


    vector<TreeNode*> splitBST(TreeNode* root, int V)
    {
        if (root == NULL) {
            ans.push_back(NULL);
            ans.push_back(NULL);
            return ans;
        }
        bool groot = false;
        TreeNode* parent = NULL, *now = root;
        if (now->val > V)
            groot = true;
        if (groot) {
            while (now != NULL && now->val > V) {
                parent = now;
                now = now->left;
            }
            TreeNode *cut = searchGreater(now, V);
            if (cut != NULL) {
                parent->left = cut->right;
                cut->right = NULL;
            } else {
                parent->left = NULL;
            }
            ans.push_back(now);
            ans.push_back(root);
        } else {
            while (now != NULL && now->val <= V) {
                parent = now;
                now = now->right;
            }
            TreeNode *cut = searchLessEqual(now, V);
            if (cut != NULL) {
                parent->right = cut->left;
                cut->left = NULL;
            } else {
                parent->right = NULL;
            }
            ans.push_back(root);
            ans.push_back(now);
        }
        return ans;
    }

};
