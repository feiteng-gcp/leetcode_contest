/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    void solve (TreeNode*root, int v, TreeNode* & root1, TreeNode* & root2) {
        root1 = NULL;
        root2 = NULL;
        if (root == NULL) return;

        if (root->val <= v) {
            root1 = root;
            TreeNode *tmp1, *tmp2;
            solve(root->right, v, tmp1, tmp2);
            root2 = tmp2;
            root->right = tmp1;
        }
        else {
            root2 = root;
            TreeNode *tmp1, *tmp2;
            solve(root->left, v, tmp1, tmp2);
            root1 = tmp1;
            root->left = tmp2;
        }
        
    }
public:
    vector<TreeNode*> splitBST(TreeNode* root, int V) {
        TreeNode* ret1, *ret2;
        solve(root, V, ret1, ret2);
        vector<TreeNode*> ret;
        ret.push_back(ret1); ret.push_back(ret2);
        return ret;
        
    }
};
