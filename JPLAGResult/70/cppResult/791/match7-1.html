class Solution {
 public:
  vector<TreeNode *> splitBST(TreeNode *root, const int V) {
    if (!root) return {NULL, NULL};
    const int c = root->val;

    if (c <= V) {
      auto rr = splitBST(root->right, V);
      root->right = rr[0];
      return {root, rr[1]};
    } else {
      auto rr = splitBST(root->left, V);
      root->left = rr[1];
      return {rr[0], root};
    }
  }
};
