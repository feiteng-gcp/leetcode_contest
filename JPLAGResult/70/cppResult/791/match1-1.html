/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> splitBST(TreeNode* root, int v) {
        if (root == nullptr) {
            return {nullptr, nullptr};

        } else if (root->val <= v) {
            auto a = splitBST(root->right, v);
            root->right = a[0];
            return {root, a[1]};
        } else {
            auto a = splitBST(root->left, v);
            root->left = a[1];
            return {a[0], root};
        }
    }
};
