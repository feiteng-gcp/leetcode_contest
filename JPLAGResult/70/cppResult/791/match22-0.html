/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> splitBST(TreeNode* root, int V) {
        if(root==NULL) return vector<TreeNode*>{NULL, NULL};
        stack<pair<TreeNode *, bool>> trav;
        TreeNode *rootl = NULL, *rootr = NULL, *cur = root;
        trav.push(make_pair(root, true));
        while(cur!=NULL){
            if(cur->val<=V){
                if(cur->right==NULL) break;
                trav.push(make_pair(cur->right, false));
                cur = cur->right;
            }else{
                if(cur->left==NULL) break;
                trav.push(make_pair(cur->left, true));
                cur = cur->left;
            }
        }
        if(cur->val<=V){
            rootl = cur;
        }else{
            rootr = cur;
        }
        while(trav.size()>1){
            auto tmp = trav.top();
            trav.pop();
            if(tmp.second){
                auto parent = trav.top();
                parent.first->left = rootr;
                rootr = parent.first;
            }else{
                auto parent = trav.top();
                parent.first->right = rootl;
                rootl = parent.first;
            }
        }
        return vector<TreeNode*>{rootl, rootr};
    }
};
