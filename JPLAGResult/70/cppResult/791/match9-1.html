/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
    TreeNode* target;
    TreeNode* parent;
public:
    void find(TreeNode* tmp,TreeNode* p ,int V){
        if(!tmp) return;
        if(tmp->val<=V){
            target = tmp;
            parent = p;
            return;
        }
        find(tmp->left,tmp,V);
        find(tmp->right,tmp,V);
    }

    vector<TreeNode*> splitBST(TreeNode* root, int V) {
        vector<TreeNode*> result(2,nullptr);
        find(root, nullptr,V);
        if(!target) {
            result[0]=root;
            result[1]=nullptr;
        }
        else if(!parent){
            result[0]=nullptr;
            result[1]=root;
        
        }else{
            parent->left = target->right;
            target->right=nullptr;
            result[0]=root;
            result[1]=target;
        }
        return result;
    }
};
