TreeNode* another;
int v;
int r;

TreeNode* doit(TreeNode* p)
{
  if (p==NULL) return NULL;
  TreeNode* tmp;
  if (p->val<=v)
  {
    tmp=doit(p->right);
    if (((tmp!=NULL)&&(tmp->val<=v))||((p->right!=NULL)&&(p->right->val>v)))
    {
      swap(tmp,p->right);
    }
  }
  else
  {
    tmp=doit(p->left);
    if (((tmp!=NULL)&&(tmp->val>v))||((p->left!=NULL)&&(p->left->val<=v)))
    {
      swap(tmp,p->left);
    }
  }
  return tmp;
}
 
class Solution {
public:
    vector<TreeNode*> splitBST(TreeNode* root, int vv) {
      vector<TreeNode*> ret;
      if (root==NULL)
      {
        ret.clear();
        ret.push_back(NULL);
        ret.push_back(NULL);
        return ret;
      }
      v=vv;
      r=root->val;
      ret.clear();
      another=NULL;
      TreeNode* tmp=doit(root);
      if (r<=v)
      {
        ret.push_back(root);
        ret.push_back(tmp);
      }
      else
      {
        ret.push_back(tmp);
        ret.push_back(root);
      }
      return ret;
    }
};
