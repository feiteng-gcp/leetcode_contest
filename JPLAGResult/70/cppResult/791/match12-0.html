class Solution {
	vector<TreeNode*> ans;
	bool solved;

	void solve(TreeNode* root, int V, TreeNode* t1, TreeNode* t2) {
		if (root == NULL) return;
		struct TreeNode* p = new struct TreeNode(root->val);
		if (root->val <= V) {
			// insert to n1
			if (t1->val > root->val) {
				t1->left = p;
				solve(root->left, V, p, t2);
				solve(root->right, V, p, t2);
			}
			else {
				t1->right = p;
				solve(root->left, V, p, t2);
				solve(root->right, V, p, t2);
			}
		}
		else {
			if (t2->val > root->val) {
				t2->left = p;
				solve(root->left, V, t1, p);
				solve(root->right, V, t1, p);
			}
			else {
				t2->right = p;
				solve(root->left, V, t1, p);
				solve(root->right, V, t1, p);
			}
		}
	}

public:
	vector<TreeNode*> splitBST(TreeNode* root, int V) {
		struct TreeNode* n1 = new struct TreeNode(V + 1);
		struct TreeNode* n2 = new struct TreeNode(V - 1);
		solve(root, V, n1, n2);
		
		ans.clear();
		ans.push_back(n1->left);
		ans.push_back(n2->right);
		return ans;
	}
};
