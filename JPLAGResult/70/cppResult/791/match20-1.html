/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    vector<TreeNode*> splitBST(TreeNode* root, int V) {
        if(root==NULL){
            vector<TreeNode*> ans;
            ans.push_back(NULL);
            ans.push_back(NULL);
            return ans;
        }
            
        TreeNode* split_node=root;
        TreeNode* smaller_node=NULL;
        while(split_node!=NULL){
            //cout<<"root:"<<split_node->val<<endl;
            if(split_node->val<V){
                smaller_node=split_node;
                split_node=split_node->right;
            }
            else if(split_node->val>V){
                split_node=split_node->left;
            }
            else if(split_node->val==V){
                smaller_node=split_node;
                break;
            }
        }
        split_node=smaller_node;
        if(split_node==NULL){
            vector<TreeNode*> ans;
            ans.push_back(root);
            ans.push_back(NULL);
            return ans;
        }
            
        //cout<<split_node->val<<endl;
        
        int split_val=split_node->val;
        vector<TreeNode*> parents;
        TreeNode* temp=root;
        while(temp!=NULL){
            parents.push_back(temp);
            if(temp->val==split_val)
                break;
            else if(temp->val<split_val)
                temp=temp->right;
            else
                temp=temp->left;
        }
        
        TreeNode* greater_child=split_node->right;
        split_node->right=NULL;
        int parent_num=parents.size()-1;
        TreeNode* parent;
        TreeNode* child;
        while(parent_num!=0){
            parent_num--;
            parent=parents[parent_num];
            child=parents[parent_num+1];
            if(parent->left==child)
                break;
        }
       
        vector<TreeNode*> ans;
        if(parent->left==child){
           // cout<<"found it"<<endl;
            parent->left=greater_child;
          
            ans.push_back(child);
              ans.push_back(root);
            return ans;
            
        }
        else{
          
            ans.push_back(root);
             ans.push_back(greater_child);
            return ans;
        }
        
    
    }
};
