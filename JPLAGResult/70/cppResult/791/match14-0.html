/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> splitBST(TreeNode* root, int V) {
        if (!root)
            return vector<TreeNode *>{NULL, NULL};
        if (root->val == V) {
            TreeNode *tmp = root->right;
            root->right = NULL;
            return vector<TreeNode *>{root, tmp};
        }
        if (root->val > V) {
            vector<TreeNode *> sub_ans = splitBST(root->left, V);
            root->left = sub_ans[1];
            return vector<TreeNode *>{sub_ans[0], root};
        }
        else {
            vector<TreeNode *> sub_ans = splitBST(root->right, V);
            root->right = sub_ans[0];
            return vector<TreeNode *>{root, sub_ans[1]};
        }
    }
};
