# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    
def splitBST(self, root, V):
        if root==None:
            return [None,None]
        if root.val==V:
            right=root.right
            root.right=None
            return (root,right)
        tmp=root
        while True:
            if V<tmp.val:
                if tmp.left==None:
                    return [root,None]
                else:
                    if tmp.left.val<V:
                        cur=tmp.left.right
                        while cur and cur.val<=V:
                            cur=cur.right
                        tmpLeft=tmp.left
                        tmp.left=cur
                        cur2=tmpLeft
                        while cur2.right!=cur:
                            cur2=cur2.right
                        cur2.right=None
                        return [root,tmpLeft]
                    elif tmp.left.val==V:
                        tmpLeft=tmp.left
                        tmp.left=tmpLeft.right
                        tmpLeft.right=None
                        return [root,tmpLeft]
                    else:
                        tmp=tmp.left
            else:
                if tmp.right==None:
                    return [root,None]
                else:
                    if tmp.right.val>V:
                        cur=tmp.right.left
                        while cur and cur.val>V:
                            cur=cur.left
                        tmpRight=tmp.right
                        tmp.right=cur
                        cur2=tmpRight
                        while cur2.left!=cur:
                            cur2=cur2.left
                        cur2.left=None
                        return [root,tmpRight]
                    elif tmp.right.val==V:
                        resRight=tmp.right.right
                        tmp.right.right=None
                        return [root,resRight]
                    else:
                        tmp=tmp.right
        
        """
        :type root: TreeNode
        :type V: int
        :rtype: List[TreeNode]
        """
