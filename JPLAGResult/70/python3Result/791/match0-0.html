# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def splitBST(self, root, V):
        """
        :type root: TreeNode
        :type V: int
        :rtype: List[TreeNode]
        """
        def rec(T, v):
            if T == None:
                return (None, None)
            if v == T.val:
                lg = T.right
                T.right = None
                return (T, lg)
            if v < T.val:
                sm, lg = rec(T.left, v)
                T.left = lg
                return (sm, T)
            else:
                # v > T.val
                sm, lg = rec(T.right, v)
                T.right = sm
                return (T, lg)
        return rec(root, V)
