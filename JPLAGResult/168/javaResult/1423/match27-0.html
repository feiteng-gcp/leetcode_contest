class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        int res = 0;
        for(int i = minSize; i <= maxSize; i++) {
            res = Math.max(res, curMaxFreq(s, maxLetters, i));
        }
        return res;
    }
    
    private int curMaxFreq(String s, int maxLetters, int size) {
        int len = s.length();
        if(size > len) {
            return 0;
        }
        
        int i = 0;
        Map<Character, Integer> charMap = new HashMap<>();
        while(i < size) {
            charMap.put(s.charAt(i), charMap.getOrDefault(s.charAt(i), 0) + 1);
            i++;
        }
        
        Map<String, Integer> map = new HashMap<>();
    
        int max = 0;
        if(charMap.size() <= maxLetters) {
            map.put(s.substring(0, size), 1);
            max = 1;
        }
        
        i = 0;
        int j = size;
        while(j < len) {
            char remove = s.charAt(i);
            char add = s.charAt(j);
            charMap.put(remove, charMap.get(remove) - 1);
            if(charMap.get(remove) == 0) {
                charMap.remove(remove);
            }
            
            charMap.put(add, charMap.getOrDefault(add, 0) + 1);
            
            if(charMap.size() <= maxLetters) {
                String sub = s.substring(i + 1, j + 1);
                map.put(sub, map.getOrDefault(sub, 0) + 1);
                max = Math.max(max, map.get(sub));
            }
            i++;
            j++;
        }
        
        return max;
    }
}
