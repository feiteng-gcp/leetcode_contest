class Solution {
    public int maxFreq(String s, 
int maxLetters, int minSize, int maxSize) {
        Map<Character, Integer> cnts = new HashMap();
        Map<String, Integer> strCnts = new HashMap();
        for (int i = 0; i < s.length(); i++) {
            cnts.put(s.charAt(i), cnts.getOrDefault(s.charAt(i), 0) + 1);
            
            
            if (i >= minSize) {
                
cnts.put(s.charAt(i - minSize), cnts.getOrDefault(s.charAt(i - minSize), 0) - 1);
                if (cnts.get(s.charAt(i - minSize)) == 0) cnts.remove(s.charAt(i - minSize));
            }
            
            
            
if (i >= minSize - 1 && cnts.size() <= maxLetters) {
                String section = s.substring(i - minSize + 1, i + 1);
                strCnts.put(section, strCnts.getOrDefault(section, 0) + 1);
            }
        }
        

        int res = 0;
        for (int freq : strCnts.values()) res = Math.max(res, freq);
        return res;
    }
}
