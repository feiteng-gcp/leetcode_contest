import java.util.HashMap;
import java.util.Map;

class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        int ans = 0;
        for (int i = minSize; i <= maxSize; i++) {
            ans = Math.max(ans, maxCount(s, maxLetters, i));
        }
        return ans;
    }

    public int maxCount(String s, int ml, int size) {
        Map<Character, Integer> map = new HashMap<>();
        Map<String, Integer> counter = new HashMap<>();
        for (int i = 0; i < size - 1; i++) map.put(s.charAt(i), map.getOrDefault(s.charAt(i), 0) + 1);
        for (int i = size - 1; i < s.length(); i++) {
            map.put(s.charAt(i), map.getOrDefault(s.charAt(i), 0) + 1);
            if (map.size() <= ml) {
                String key = s.substring(i - size + 1, i + 1);
                counter.put(key, counter.getOrDefault(key, 0) + 1);
            }
            map.put(s.charAt(i - size + 1), map.get(s.charAt(i - size + 1)) - 1);
            if (map.get(s.charAt(i - size + 1)) == 0) map.remove(s.charAt(i - size + 1));
        }
        int ans = 0;
        for (String k : counter.keySet()) {
            ans = Math.max(ans, counter.get(k));
        }
        return ans;
    }
}
