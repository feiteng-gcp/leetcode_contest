class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        char[] cs = s.toCharArray();
        HashMap<String,Integer> map = new HashMap<>();
        for (int i = 0; i < cs.length; i++) {
            for (int j = i+minSize; j <= i+maxSize && j<=cs.length ; j++) {
                if (help(cs,i,j,maxLetters)){
                    map.put(s.substring(i,j),map.getOrDefault(s.substring(i,j),0)+1);
                    break;
                }
            }
        }
        int res = 0;
        for (String x : map.keySet()){
            if (map.get(x)>res){
                res = map.get(x);
            }
        }
        return res;
    }

    private boolean help(char[] cs, int i, int j, int maxLetters) {
        int[] map = new int[26];
        for (int k = i; k < j; k++) {
            if (map[cs[k]-'a'] == 0){
                map[cs[k]-'a'] = 1;
                maxLetters--;
            }
            if (maxLetters<0)return false;
        }
        return true;
    }
}
