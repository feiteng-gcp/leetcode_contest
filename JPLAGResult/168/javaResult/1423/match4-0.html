class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        Map<String, Integer> map = new HashMap<>();
        int len = s.length(), res = 0;
        int[] cnt = new int[26];
        for(int step = minSize; step <= maxSize; step++){
            map.clear();
            for(int i = 0; i + step <= len; i++){
                Arrays.fill(cnt, 0);
                int let = 0;
                for(int j = 0; j < step; j++){
                   int index = s.charAt(j + i) - 'a';
                    if(cnt[index] == 0)
                        let ++;
                    cnt[index] ++; 
                }
                if(let <= maxLetters){
                    String str = s.substring(i, i + step);
                    // System.out.printf("st:%d, ed:%d, str:%s\n", i, i + step + 1, str);
                    int val = map.getOrDefault(str, 0);
                    map.put(str, val + 1);
                    res = Math.max(val + 1, res);
                }
            }
        }
        return res;
        
    }
}
