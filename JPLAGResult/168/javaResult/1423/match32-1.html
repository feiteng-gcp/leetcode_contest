class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        Map<String, Integer> ss = new HashMap<>();
        int n = s.length();
        for (int i = 0; i < n - minSize + 1; i++) {
            Set<Character> cnt = new HashSet<>();
            for (int j = i; j < i + minSize; j++) {
                cnt.add(s.charAt(j));
            }
            if (cnt.size() <= maxLetters) {
                String cd = s.substring(i, i + minSize);
                int a = (ss.containsKey(cd) ? ss.get(cd) : 0) + 1;
                ss.put(cd, a);
            }
            for (int j = i + minSize; j < i + maxSize; j++) {
                if (j >= n) {
                    break;
                }
                cnt.add(s.charAt(j));
                if (cnt.size() <= maxLetters) {
                    String cd = s.substring(i, j + 1);
                    int a = (ss.containsKey(cd) ? ss.get(cd) : 0) + 1;
                    ss.put(cd, a);
                } else {
                    break;
                
}
            }
        }
        int ans = 0;
        for (Map.Entry me : ss.entrySet()) {
            //System.out.println(me.getKey() + " " + me.getValue());
            ans = Math.max(ans, (int)me.getValue());
        }
        return ans;
    }
}
