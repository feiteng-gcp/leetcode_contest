class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        Map<String, Integer> subToFreq = new HashMap<>();
        int res = 0;
        Map<Character, Integer> charToFreq = new HashMap<>();
        int distinct = 0;
        int slow = 0;
        
        for (int fast = 0; fast < s.length(); fast++) {
            // 1. add fast
            char chf = s.charAt(fast);
            charToFreq.put(chf,  charToFreq.getOrDefault(chf,0)+1);
            if (charToFreq.get(chf) == 1) {
                distinct++; 
            }
            
            // 2 .move slow
            while (distinct > maxLetters || fast-slow+1 > minSize) {
                char chs = s.charAt(slow++);
                charToFreq.put(chs,  charToFreq.get(chs)-1);
                if (charToFreq.get(chs) == 0) {
                    distinct--; 
                }
            }
            
            // 3. check current substring
            if (minSize <= fast-slow+1 && fast-slow+1 <= maxSize) {
                String sub = s.substring(slow, fast+1);
                subToFreq.put(sub, subToFreq.getOrDefault(sub, 0) + 1);
                res = Math.max(res, subToFreq.get(sub));
            }
        } 
        
        return res;
    }
}
