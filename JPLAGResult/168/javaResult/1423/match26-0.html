class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        int max = 0;
        for(int size = minSize; size <= maxSize; size++){
            max = Math.max(max, maxFreq(s, maxLetters, size));
        }
        return max;
    }
    
    private int maxFreq(String s, int maxLetters, int size){
        int[] cnt = new int[26];
        int pre = 0, cur = 0;
        Map<String, Integer> map = new HashMap<>();
        int max = 0;
        StringBuilder sb = new StringBuilder();
        int diff = 0;
        while(cur < s.length()){
            if(cnt[s.charAt(cur) - 'a']++ == 0){
                diff++;
            }
            sb.append(s.charAt(cur));
            if(sb.length() > size){
                if(--cnt[s.charAt(pre++) - 'a'] == 0){
                    diff--;
                }
                sb.deleteCharAt(0);
            }
            if(sb.length() == size && diff <= maxLetters){
                String tmp = sb.toString();
                map.put(tmp, map.getOrDefault(tmp, 0) + 1);
                max = Math.max(max, map.get(tmp));
            }
            cur++;
        }
        return max;
    }
}
