class Solution {
        public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
            int freq = 0;

            HashMap<String, Integer> count = new HashMap<>();
            for (int i = 0; i < s.length()-minSize+1; i++) {
                String sub = s.substring(i, i+minSize);
                if (letters(sub, maxLetters)) {
                    if (!count.containsKey(sub)) {
                        count.put(sub, 1);
                        freq = Math.max(1, freq);
                    } else {
                        count.put(sub, count.get(sub) + 1);
                        freq = Math.max(freq, count.get(sub));
                    }
                }
                
            }
            
            return freq;
        }
        public boolean letters(String sub, int maxLetters) {
            TreeSet<Character> all = new TreeSet<>();
            for (char x : sub.toCharArray()) {
                all.add(x);
            }
            return all.size() <= maxLetters;
        }
    }
