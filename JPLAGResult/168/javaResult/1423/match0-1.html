class Solution {

  public int maxFreq(String s, int a, int b, int d) {
    Map<String, Integer> m = new HashMap<>();
    for (int i = 0; i < s.length(); i++) {
      for (int j = i + b; j <= Math.min(s.length(), i + d); j++) {
        String t = s.substring(i, j);
        boolean[] set = new boolean[26];
        int count = 0;
        for (char c : t.toCharArray()) {
          if (!set[c - 'a']) {
            set[c - 'a'] = true;
            count++;
            if (count > a) {
              break;
            }
          }
        }
        if (count <= a) {
          m.put(t, m.getOrDefault(t, 0) + 1);
          break;
        }
      }
    }
    int x = 0;
    for (Map.Entry<String, Integer> k : m.entrySet()) {
      if (k.getValue() > x) {
        x = k.getValue();
        if (x == d) {
          return x;
        }
      }
    }
    return x;
  }
}
