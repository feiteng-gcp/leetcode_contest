class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        Map<String, Integer> counts = new HashMap<>();
        int best = 0;
        for (int i=0; i+minSize<=s.length(); i++) {
            StringBuilder sb = new StringBuilder();
            boolean visited[] = new boolean[26];
            int uniq = 0;
            for (int j=0; j<maxSize && (i+j<s.length()); j++) {
                if (!visited[s.charAt(i+j)-'a']) {
                    visited[s.charAt(i+j)-'a'] = true;
                    uniq++;
                }
                if (uniq > maxLetters) break;
                sb.append(s.charAt(i+j));
                if (sb.length() >= minSize && sb.length() <= maxSize) {
                    int count = counts.getOrDefault(sb.toString(), 0) + 1;
                    best = Math.max(best, count);
                    counts.put(sb.toString(), count);
                }
            }
        }
        return best;
    }
}
