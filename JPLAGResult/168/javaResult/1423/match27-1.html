class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        Map<String, Integer> subs = new HashMap<>();
        
        
int out = 0;
        
        for(int size = minSize; size <= maxSize; size++) {
            out = Math.max(out, maxFreq(s, maxLetters, size));
        }
        
        return out;
    }
    
    private int maxFreq(String s, int maxLetters, int size) {
        int out = 0;
        
        Map<String, Integer> subs = new HashMap<>();
        
        Map<Character, Integer> chars = new HashMap<>();
        Set<Character> uniqs = new HashSet<>();
        
        for(int i = 0; i < size; i++) {
            char ch = s.charAt(i);
            
            uniqs.add(ch);
            chars.put(ch, chars.getOrDefault(ch, 0) + 1);
        }
        
        if(uniqs.size() <= maxLetters) {
            out = 1;
            subs.put(s.substring(0, size), 1);
        }
        
        for(int i = size; i < s.length(); i++) {
            char newCh = s.charAt(i);
            
            uniqs.add(newCh);
            chars.put(newCh, 
chars.getOrDefault(newCh, 0) + 1);
            
            char oldCh = s.charAt(i-size);
            chars.put(oldCh, chars.get(oldCh) - 1);
            if(chars.get(oldCh) == 0) {
                uniqs.remove(oldCh);
            }
            
            
if(uniqs.size() <= maxLetters) {
                String newSub = s.substring(i-size + 1, i+1);
                
                //System.out.println(i + " " + newSub);
                
                subs.put(newSub, subs.getOrDefault(newSub, 0) + 1);
                out = Math.max(out, subs.get(newSub));
            }
        }
        
        //System.out.println(size + " " + subs);
        
        return out;
    }
}
