class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        Map<Character, Integer> count = new HashMap<>();
        char[] arr = s.toCharArray();
        Map<String, Integer> substrCount = new HashMap<>();
        int n = s.length();
        int answer = 0;
       for (int size = minSize; size <= maxSize; size++) {
           count = new HashMap<>();
           int left = 0; 
           int right = size;
           for (int i = 0; i < right; i++) {
               count.put(arr[i], count.getOrDefault(arr[i], 0) + 1);
           }
           if (count.size() <= maxLetters) {
               String substr = s.substring(left, right);
               substrCount.put(substr, substrCount.getOrDefault(substr, 0) + 1);
           }
           while (right < n) {
               count.put(arr[right], 
count.getOrDefault(arr[right], 0) + 1);
               count.put(arr[left], count.get(arr[left]) - 1);
               if (count.get(arr[left]) == 0) {
                   count.remove(arr[left]);
               }
               left++;
               right++;
               if (count.size() <= maxLetters) {
                String substr = s.substring(left, right);
                substrCount.put(substr, substrCount.getOrDefault(substr, 0) + 1);
                }
           }
       }
        
for (String substr: substrCount.keySet()) {
            //System.out.println(substr + "," + substrCount.get(substr));
            answer = Math.max(answer, substrCount.get(substr));
        }
        return answer;
    }
}
