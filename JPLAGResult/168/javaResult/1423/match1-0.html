class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        int left = 0;
        int right = 0;
        int num = 0;
        int[] count = new int[26];
        Map<String, Integer> map = new HashMap<>();
        
        while (right < s.length()) {
            char c = s.charAt(right);
            if (count[c - 'a'] == 0) {
                num++;
            }
            count[c - 'a']++;
            
            while (num > maxLetters || right - left + 1 > maxSize) {
                char l = s.charAt(left);
                count[l - 'a']--;
                if (count[l - 'a'] == 0) {
                    num--;
                }
                left++;
            }
            
            int start = left;
            while (right - start + 1 >= minSize) {
                map.put(s.substring(start, right + 1), map.getOrDefault(s.substring(start, right + 1), 0) + 1);
                start++;
            }
            right++;
        }
        
        int res = 0;
        for (String str : map.keySet()) {
            res = Math.max(res, map.get(str));
        }
        return res;
    }
}
