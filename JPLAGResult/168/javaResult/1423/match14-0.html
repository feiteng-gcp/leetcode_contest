class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        int l = 0;
        int r = 0;
        int[] count = new int[26];
        int letter = 0;
        HashMap<String, Integer> res = new HashMap<>();
        
        while(r < s.length()){
            char c = s.charAt(r++);
            count[c - 'a']++;
            if(count[c - 'a'] == 1){
                letter++;
            }
            
            while(r - l > maxSize || letter > maxLetters){
                char before = s.charAt(l++);
                count[before - 'a']--;
                if(count[before - 'a'] == 0){
                    letter--;
                }
            }
            
            
            for(int i = l; r - i - minSize >= 0; i++){
                String sub = s.substring(i, r);
                res.put(sub, res.getOrDefault(sub, 0) + 1);
            }
        }
        
        int max = 0;
        for(int i : res.values()){
            max = Math.max(max, i);
        }
        
        return max;
    }
}
