class Solution {
    public static int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        Map<String, Integer> substringToCount = new HashMap<>();
        int n = s.length();
        for (int i=0;i<n;i++) {
            Set<Character> uniqueCharacters = new HashSet<>();
            for (int j=i;j<n;j++) {
                uniqueCharacters.add(s.charAt(j));
                if (uniqueCharacters.size() > maxLetters || (j-i+1) > maxSize) {
                    break;
                }
                if (j-i+1 >= minSize) {
                    String substring = s.substring(i,j+1);
//                    System.out.println("i: " + i + " j: " + j);
//                    System.out.println(substring);
                    substringToCount.put(substring ,  substringToCount.getOrDefault(substring, 0) + 1);
                }
            }
        }

        int maxCount = 0;
        for (Map.Entry<String, Integer> entry : substringToCount.entrySet()) {
            if (entry.getValue() > maxCount) {
                maxCount = entry.getValue();
            }
        }
        return maxCount;

    }
}
