class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        Map<String, Integer> count = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            boolean[] unique = new boolean[26];
            for (int len = minSize; len <= maxSize && i + len - 1 < s.length(); len++) {
                String sub = s.substring(i, i + len);
                if (count.containsKey(sub)) {
                    count.put(sub, count.get(sub) + 1);
                    continue;
                }
                if (numUnique(sub) <= maxLetters) {
                    count.put(sub, 1);
                }
            }
        }
        int ans = 0;
        for (Map.Entry<String, Integer> e : count.entrySet()) {
            ans = Math.max(ans, e.getValue());
        }
        return ans;
    }
    
    public int numUnique(String s) {
        boolean[] unique = new boolean[26];
        int count = 0;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (!unique[c-'a']) {
                unique[c-'a'] = true;
                count++;
            }
        }
        return count;
    }
}
