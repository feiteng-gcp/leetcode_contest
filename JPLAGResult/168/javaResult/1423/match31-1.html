class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        int max = 0;
        for (int l = minSize; l <= maxSize; l++) {
            Map<String, Integer> countMap = new HashMap<>();
            Map<Character, Integer> map = new HashMap<>();
            for (int i = 0; i < l; i++) {
                map.put(s.charAt(i), map.getOrDefault(s.charAt(i), 0) + 1);
            }
            if (map.keySet().size() <= maxLetters) {
                String ss = s.substring(0, l);
                countMap.put(ss, countMap.getOrDefault(ss, 0) + 1);
                max = Math.max(countMap.get(ss), max);
            }
            // System.out.println(countMap);
            for (int i = l; i < s.length(); i++) {
                map.put(s.charAt(i - l), map.get(s.charAt(i - l)) - 1);
                if (map.get(s.charAt(i - l)) == 0) {
                    map.remove(s.charAt(i - l));
                }
                map.put(s.charAt(i), map.getOrDefault(s.charAt(i), 0) + 1);
                if (
map.keySet().size() <= maxLetters) {
                    String ss = s.substring(i - l + 1, i + 1);
                    countMap.put(ss, countMap.getOrDefault(ss, 0) + 1);
                    max = Math.max(countMap.get(ss), max);
                }
            }
            // System.out.println(countMap);
        }
        return max;
        
    }
}
