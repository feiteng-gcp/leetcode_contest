class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        Map<String, Integer> countMap = new HashMap<>();
        Map<Character, Integer> charCount = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            charCount.put(c, charCount.getOrDefault(c, 0) + 1);
            
            if (i >= minSize - 1) {
                if (charCount.keySet().size() <= maxLetters) {
                    String subStr = s.substring(i + 1 - minSize, i + 1);
                    countMap.put(subStr, countMap.getOrDefault(subStr, 0) + 1);
                }
                
                char prevC = s.charAt(i - minSize + 1);
                charCount.put(prevC, charCount.get(prevC) - 1);
                if (charCount.get(prevC) == 0) {
                    charCount.remove(prevC);
                }
            }            
        }
        
        int max = 0;
        for (String str : countMap.keySet()) {
            max = Math.max(max, countMap.get(str));
        }
        
        return max;
    }
}
