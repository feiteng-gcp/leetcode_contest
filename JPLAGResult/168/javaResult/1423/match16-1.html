public class Solution {
    public int maxFreq(String s, int maxLetters, int minSize, int maxSize) {
        int ret = 0;
        for(int len = minSize; len <= maxSize; len ++){
            Map<String,Integer> ctr = new HashMap<>();
            for(int i = 0; i < s.length(); i++){
                if (i + len > s.length()) break;
                String ss = s.substring(i, i + len);
                if (nChar(ss) > maxLetters) continue;

                if (!ctr.containsKey(ss)){
                    ctr.put(ss,0);
                }
                int newct = ctr.get(ss)+1;
                ctr.put(ss,newct);
                
ret = Math.max(ret,newct);
            }
        }
        return ret;
    }

    int nChar(String s){
        Set<Character> x = new HashSet<>();
        for (char c : s.toCharArray()) {
            x.add(c);
        }
        return x.size();
    }
}
