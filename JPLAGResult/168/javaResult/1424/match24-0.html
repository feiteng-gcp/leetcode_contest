class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        Set<Integer> opens = new HashSet<>();
        Set<Integer> closed = new HashSet<>();
        Set<Integer> ks = new HashSet<>();
        
        for (int v : initialBoxes) {
            if (status[v] == 1) {
                opens.add(v);
            } else {
                closed.add(v);
            }
        }
        
        int max = 0;
        while (!opens.isEmpty()) {
            int i = opens.iterator().next();
            max += candies[i];
            
            opens.remove(i);
            
            for (int v : containedBoxes[i]) {
                if (status[v] == 1 || ks.contains(v)) {
                    opens.add(v);
                } else {
                    closed.add(v);
                }
            }
            
            for (int k : keys[i]) {
                ks.add(k);
                if (closed.contains(k)) {
                    closed.remove(k);
                    opens.add(k);
                }
            }
        }
        
        return max;
    }
}
