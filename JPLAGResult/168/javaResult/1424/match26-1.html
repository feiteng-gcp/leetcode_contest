class Solution {

  public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
    
Queue<Integer> q = new ArrayDeque<>();
    Set<Integer> s = new HashSet<>();
    for (int i : initialBoxes) {
      q.add(i);
    }
    int sum = 0;
    
while (!q.isEmpty()) {
      boolean flag = false;
      for (int i = q.size(); i > 0; i--) {
        int h = q.remove();
        if (!s.contains(h) && status[h] == 1) {
          flag = true;
          s.add(h);
          sum += candies[h];
          for (int j : containedBoxes[h]) {
            q.add(j);
          }
          for (int j : keys[h]) {
            status[j] = 1;
          }
        } else if (!s.contains(h) && status[h] == 0) {
          q.add(h);
        }
      }
      if (!flag) {
        break;
      }
    }
    return sum;
  }
}
