class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        int out = 0;
        
        Set<Integer> keysInHand = new HashSet<>();
        
        Queue<Integer> openBoxes = new LinkedList<>();
        Set<Integer> closedBoxes = new HashSet<>();
        
        for(int box : initialBoxes) {
            if(status[box] == 1) {
                openBoxes.add(box);
            } else {
                closedBoxes.add(box);
            }
        }
        
        //System.out.println(openBoxes);
        
        while(!openBoxes.isEmpty()) {
            //System.out.println(openBoxes + " " + out);

            int box = openBoxes.poll();

            //System.out.println(openBoxes + " " + out);
            
            out += candies[box];
            
            for(int newBox : containedBoxes[box]) {
                if(status[newBox] == 1 || keysInHand.contains(newBox)) {
                    openBoxes.add(newBox);
                } else {
                    closedBoxes.add(newBox);
                }
            }
            
            for(int key : keys[box]) {
                keysInHand.add(key);
                
                if(closedBoxes.contains(key)) {
                    closedBoxes.remove(key);
                    openBoxes.add(key);
                }
            }
        }
        
        return out;
    }
}
