class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        Queue<Integer> toVisit = new LinkedList<>();
        Set<Integer> visited = new HashSet<>();
        Set<Integer> accessibleBoxes = new HashSet<>();
        int res = 0;
        for (int i : initialBoxes) {
            if (status[i] == 0) {
                accessibleBoxes.add(i);
            } else {
                toVisit.add(i);
                visited.add(i);
            }
        }

        while(!toVisit.isEmpty()) {
            int curr = toVisit.poll();
            res += candies[curr];
            for (int i : containedBoxes[curr]) {
                if (status[i] == 0) {
                    accessibleBoxes.add(i);
                } else {
                    toVisit.add(i);
                    visited.add(i);
                }
            }
            for (int i : keys[curr]) {
                status[i] = 1;
                if (accessibleBoxes.contains(i) && !visited.contains(i)) {
                    toVisit.add(i);
                    visited.add(i);
                }
            }
        }
        return res;
    }
}
