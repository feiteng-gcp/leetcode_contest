class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        Queue<Integer> openBox = new LinkedList<>();
        HashSet<Integer> closeBox = new HashSet<>();
        for(int i : initialBoxes){
            if(status[i] == 0){
                closeBox.add(i);
            }
            else{
                openBox.add(i);
            }
        }
        
        int res = 0;
        while(!openBox.isEmpty()){
            int cur = openBox.poll();
            res += candies[cur];
            for(int key : keys[cur]){
                status[key] = 1;
                if(closeBox.remove(key)){
                    openBox.add(key);
                }
            }
            
            for(int box : containedBoxes[cur]){
                if(status[box] == 0){
                    closeBox.add(box);
                }
                else{
                    openBox.add(box);
                }
            }
        }
        
        return res;
    }
}
