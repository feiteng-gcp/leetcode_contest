public class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        Set<Integer> used = new HashSet<>(), opened = new HashSet<>(), closed = new HashSet<>();
        int N = status.length;
        for (int b : initialBoxes) {
            if (status[b] == 1) {
                opened.add(b);
            } else {
                closed.add(b);
            }
        }
        int ans = 0;
        Set<Integer> obtainedKeys = new HashSet<>();
        while (!opened.isEmpty()) {
            int cur = opened.iterator().next();
            opened.remove(cur);
            if (used.contains(cur)) {
                continue;
            }
            used.add(cur);

            ans += candies[cur];
            for (int neighbor : containedBoxes[cur]) {
                if (obtainedKeys.contains(neighbor) || status[neighbor] == 1) {
                    opened.add(neighbor);
                } else {
                    closed.add(neighbor);
                }
            }
            for (int key : keys[cur]) {
                obtainedKeys.add(key);
                if (closed.contains(key)) {
                    closed.remove(key);
                    opened.add(key);
                }
            }
        }
        return ans;
    }
}
