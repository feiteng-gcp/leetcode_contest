class Solution {

	public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
		int sum = 0;
		ArrayDeque<Integer> queue = new ArrayDeque<>();
		HashSet<Integer> set = new HashSet<>();
		for (int initialBox : initialBoxes) {
			queue.add(initialBox);
		}
		while (!queue.isEmpty()) {
			boolean flag = false;
			for (int i = queue.size(); i > 0; i--) {
				int head = queue.remove();
				if (!set.contains(head) && status[head] == 1) {
					flag = true;
					set.add(head);
					sum += candies[head];
					for (int containedBox : containedBoxes[head]) {
						queue.add(containedBox);
					}
					for (int key : keys[head]) {
						status[key] = 1;
					}
				} else if (!set.contains(head) && status[head] == 0) {
					queue.add(head);
				}
			}
			if (!flag) {
				break;
			}
		}
		return sum;
	}
}
