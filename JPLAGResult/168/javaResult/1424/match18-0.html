class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        int ans = 0;
        Deque<Integer> q = new ArrayDeque<>();  // contains boxes ready to open
        Set<Integer> haveButLocked = new HashSet<>();  // boxes we have but are locked
        for (int i : initialBoxes) {
            if (status[i] == 1) {
                q.addLast(i);
            } else {
                haveButLocked.add(i);
            }
        }
        while (!q.isEmpty()) {
            int box = q.removeFirst();
            ans += candies[box];
            for (int b : containedBoxes[box]) {
                if (status[b] == 1) {
                    q.addLast(b);
                } else {
                    haveButLocked.add(b);
                }
            }
            for (int unlockable : keys[box]) {
                status[unlockable] = 1;
                if (haveButLocked.contains(unlockable)) {
                    haveButLocked.remove(unlockable);
                    q.addLast(unlockable);
                }
            }
        }
        return ans;
    }
}
