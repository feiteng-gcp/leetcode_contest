class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        int res = 0;
        
Queue<Integer> q = new LinkedList<>();
        Set<Integer> have = new HashSet<>();
        Set<Integer> key = new HashSet<>();
        
        for (int box : initialBoxes) {
            if (status[box] == 1) {
                q.offer(box);
            }
            else {
                have.add(box);
            }
        }
        
        while (!q.isEmpty()) {
            int cur = q.poll();
            res += candies[cur];
            
            
for (int b : containedBoxes[cur]) {
                if (status[b] == 1) {
                    q.offer(b);
                }
                else {
                    have.add(b); 
                }
            }
            for (int k : keys[cur]) {
                key.add(k);
            
}
            
            for (int k : key) {
                if (have.contains(k)) {
                    have.remove(k);
                    q.offer(k);
                }
            }
        }
        return res;
    }
}
