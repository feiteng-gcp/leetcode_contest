class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        int n = status.length;
        boolean[] hasKey = new boolean[n];
        
        
Queue<Integer> candidates = new LinkedList<>();
        Set<Integer> accessible = new HashSet<>();
        
        for (int i : initialBoxes) {
            if (status[i] == 1) {
                candidates.offer(i);
            } else {
                accessible.add(i);
            }
        }
        
        int res = 0;
        
        while (!candidates.isEmpty()) {
            int curr = candidates.poll();
            res += candies[curr];
            
            
for (int k : keys[curr]) {
                hasKey[k] = true;
                if (accessible.contains(k)) {
                    candidates.offer(k);
                    accessible.remove(k);
                }
            }
            
            
for (int c : containedBoxes[curr]) {
                if (hasKey[c] || status[c] == 1) {
                    candidates.offer(c);
                } else {
                    accessible.add(c);
                }
            }
        }
        
        return res;
    }
}
