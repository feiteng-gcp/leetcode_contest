class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        int len = status.length, res = 0;
        Queue<Integer> que = new LinkedList<>();
        Set<Integer> key = new HashSet<>();
        Set<Integer> box = new HashSet<>();
        Set<Integer> open = new HashSet<>();
        for(int i = 0; i < len; i++){
            if(status[i] == 1)
                key.add(i);
        }
        
        for(int i : initialBoxes){
            que.add(i);
        }
        
        while(!que.isEmpty()){
            int b = que.poll();
            if(open.contains(b))
                continue;
            open.add(b);
            for(int next: containedBoxes[b]){
                box.add(next);
                if(key.contains(next)){
                    if(!open.contains(next))
                        que.add(next);
                }
            }
            for(int next: keys[b]){
                key.add(next);
                if(box.contains(next)){
                    if(!open.contains(next))
                        que.add(next);
                }
            }
            res += candies[b];
            
        }
        return res;
        
    }
}
