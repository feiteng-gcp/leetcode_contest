class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        if (initialBoxes.length == 0) {
            return 0;
        }
        int n = status.length;
        int sum = 0;
        Queue<Integer> queue = new ArrayDeque<>();
        Set<Integer> boxFound = new HashSet<>(), keyFound = new HashSet<>();
        for (int box : initialBoxes) {
            if (status[box] == 1) {
                queue.add(box);    
            } else {
                boxFound.add(box);
            }
        }
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int box = queue.remove();
                sum += candies[box];
                for (int containedBox : containedBoxes[box]) {
                    if (status[containedBox] == 1) {
                        queue.add(containedBox);
                    } else {
                        boxFound.add(containedBox);    
                    }
                }
                for (int key : keys[box]) {
                    keyFound.add(key);
                }
            }
            for (Iterator<Integer> iterator = boxFound.iterator(); iterator.hasNext(); ) {
                int box = iterator.next();
                if (keyFound.contains(box)) {
                    queue.add(box);
                    iterator.remove();
                } 
            }
        }
        return sum;
    }
}
