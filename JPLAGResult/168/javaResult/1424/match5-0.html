class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        final int N = status.length;
        Queue<Integer> queue = new ArrayDeque<>(N);
        Set<Integer> visited = new HashSet<>();
        Set<Integer> keySet = new HashSet<>();
        Set<Integer> boxSet = new HashSet<>();

        for (int box : initialBoxes) {
            if (status[box] == 1) {
                queue.offer(box);
            } else {
                boxSet.add(box);
            }
        }

        int count = 0;
        while (!queue.isEmpty()) {
            int curr = queue.poll();

            count += candies[curr];

            for (int key : keys[curr]) {
                keySet.add(key);
            }
            for (int box : containedBoxes[curr]) {
                if (status[box] == 1) {
                    queue.offer(box);
                } else {
                    boxSet.add(box);
                }
            }
            for (int key : keySet) {
                if (boxSet.contains(key)) {
                    queue.offer(key);
                    boxSet.remove(key);
                }
            }
        }

        return count;
    }
}
