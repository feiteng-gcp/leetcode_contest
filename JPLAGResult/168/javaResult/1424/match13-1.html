class Solution {
    public int maxCandies(int[] status, int[] candies, int[][] keys, int[][] containedBoxes, int[] initialBoxes) {
        int n = status.length;
        int res = 0;
        
HashSet<Integer> closedBoxes = new HashSet<>();
        LinkedList<Integer> openBoxes = new LinkedList<>();
        for (int x : initialBoxes) {
            if (status[x] == 1) {
                openBoxes.add(x);
            } else {
                closedBoxes.add(x);
            }
        }
        while (!openBoxes.isEmpty()) {
            int b = openBoxes.remove();
            
for (int x : containedBoxes[b]) {
                if (status[x] == 1) {
                    openBoxes.add(x);
                } else {
                    closedBoxes.add(x);
                }
            }
            for (int k : keys[b]) {
                status[k] = 1;
                
if (closedBoxes.contains(k)) {
                    openBoxes.add(k);
                    closedBoxes.remove(k);
                }
            }
            res += candies[b];
        }
        return res;
    }
}
