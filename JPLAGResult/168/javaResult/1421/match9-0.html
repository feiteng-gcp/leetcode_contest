class Solution {
    public int findNumbers(int[] nums) {
        int res = 0;
        for(int n: nums) {
            int digits = 0;
            while(n != 0) {
                n /= 10;
                digits++;
            }
            if(digits % 2 == 0) {
                res++;
            }
        }
        return res;
    }
}
