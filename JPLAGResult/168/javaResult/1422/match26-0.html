class Solution {
    public boolean isPossibleDivide(int[] nums, int k) {
        int n = nums.length;
        if (n % k != 0) return false;
        Arrays.sort(nums);
        TreeMap<Integer, Integer> pq = new TreeMap<>();
        for (int i : nums) {
            pq.put(i, pq.getOrDefault(i, 0) + 1);
        }
        for (int i = 0; i < n / k; i++) {
            int cur = pq.firstKey();
            for (int j = 0; j < k; j++) {
                if (pq.getOrDefault(cur, 0) <= 0) return false;
                pq.put(cur, pq.get(cur) - 1);
                if (pq.get(cur) == 0) pq.remove(cur);
                cur++;
            }
        }
        return true;
    }
}
