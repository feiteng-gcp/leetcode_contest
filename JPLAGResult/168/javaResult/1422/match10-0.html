class Solution {
    public boolean isPossibleDivide(int[] nums, int k) {
        int n = nums.length;
        if(n%k!=0) return false;
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for(int i: nums) {
            if(!map.containsKey(i)) map.put(i, 1);
            else map.put(i, map.get(i)+1);
        }
        while(map.size()!=0) {
            int a = map.firstKey();
            
            for(int i=0; i<k; i++) {
                if(!map.containsKey(a)) return false;
                int x = map.get(a);
                if(x==1) map.remove(a);
                else map.put(a, x-1);
                a++;
            }
            
        }
        return true;
    }
}
