class Solution {
    public boolean isPossibleDivide(int[] nums, int k) {
        TreeMap<Integer,Integer> map = new TreeMap<>();
        for(int num :nums){
            map.put(num, map.getOrDefault(num, 0) + 1);
        }
        while(!map.isEmpty()) {
            Integer first = map.firstKey();
            int number = map.get(first);
            for(int i = 0; i< k; ++i) {
                if(!map.containsKey(first+i) || map.get(first+i) < number)
                    return false;
                map.put(first+i, map.get(first+i) - number);
                if(map.get(first+i) == 0)
                    map.remove(first+i);
            }
        }
        return true;
    }
}
