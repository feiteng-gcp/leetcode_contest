class Solution {
    public boolean isPossibleDivide(int[] nums, int k) {
        // 1. bucket sort
        TreeMap<Integer, Integer> numToFreq = new TreeMap<>();
        for (int num : nums) {
            numToFreq.put(num, numToFreq.getOrDefault(num, 0) + 1);
        }
        
        // 2. check each straight
        for (Map.Entry<Integer, Integer> entry : numToFreq.entrySet()) {
            int num = entry.getKey();
            int freq = entry.getValue();
            if (freq == 0) {
                continue;
            }
            for (int i = 0; i < k; i++) {
                if (numToFreq.getOrDefault(num + i, 0) < freq) {
                    return false;
                }
                numToFreq.put(num + i, numToFreq.get(num + i) - freq);
            }
        } 
        
        return true;
    }
}
