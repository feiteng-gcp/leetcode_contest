class Solution {
    public boolean isPossibleDivide(int[] nums, int k) {
        if(nums.length%k != 0) {
            return false;
        }
        
        TreeMap<Integer, Integer> map = new TreeMap<>();
        
        for(Integer x : nums) {
            if(map.get(x) == null){
                map.put(x,0);
            }
            map.put(x, map.get(x)+1);
        }
        
        while(map.firstEntry() !=null){
            Integer s = map.firstKey();
            map.put(s, map.get(s)-1);
            if(map.get(s)==0){
                map.remove(s);
            }
            for(int i=1; i<k; i++){
                int next = s+i;
                //System.out.println("next is " + next);
                if(map.get(next) == null){
                    return false;
                }
                else{
                    map.put(next, map.get(next)-1);
                    if(map.get(next)==0){
                        map.remove(next);
                    }
                }
            }
            
        }
        return true;
    }
}
