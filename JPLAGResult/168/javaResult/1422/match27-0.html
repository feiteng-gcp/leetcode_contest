class Solution {
    public boolean isPossibleDivide(int[] nums, int k) {
      if(nums.length%k != 0) return false;
      TreeMap<Integer, Integer> tm = new TreeMap<>();
      for(int n : nums) {
        tm.put(n, tm.getOrDefault(n, 0)+1);
      }
      int rep = nums.length/k;
      for(int i=0; i<rep; i++) {
        int v = tm.firstKey();
        for(int j=0; j<k; j++) {
          if(tm.containsKey(v)) {
            tm.put(v, tm.get(v) - 1);
            if(tm.get(v) == 0) tm.remove(v);
            v++;
          } else {
            return false;
          }
        }
      }
      return true;
    }
  }
