class Solution {
    public boolean isPossibleDivide(int[] nums, int k) {
        if(nums.length % k != 0) return false;
        TreeMap<Integer, Integer> map = new TreeMap();
        for(int num : nums)        
            map.put(num, map.getOrDefault(num, 0) + 1);
        while(map.size() != 0)
        {
            int tmp = map.firstKey();
            for(int i = 0; i < k; i ++)
            {
                if(map.get(tmp + i) == null) return false;
                int freq = map.get(tmp + i);
                if(freq == 1)
                    map.remove(tmp + i);
                else
                    map.put(tmp + i, freq - 1);
            }
        }
        return true;
    }
}
