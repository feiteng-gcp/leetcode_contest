class Solution {
    public boolean isPossibleDivide(int[] nums, int k) {
        if (nums.length % k != 0) return false;
        
        
TreeMap<Integer, Integer> cnts = new TreeMap();
        for (int n : nums) {
            cnts.put(n, cnts.getOrDefault(n, 0) + 1);
        }
        
        
        for (int i = 0; i < nums.length / k; i++) {
            int start = cnts.ceilingKey(Integer.MIN_VALUE);
            
            for (int j = start; j < start + k; j++) {
                if (!cnts.containsKey(j)) return false;
                cnts.put(j, cnts.get(j) - 1);
                if (cnts.get(j) == 0) cnts.remove(j);
            }
        }
        return true;
        
    }
}
