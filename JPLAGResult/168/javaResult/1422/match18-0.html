class Solution {
    public boolean isPossibleDivide(int[] nums, int k) {
        TreeMap<Integer, Integer> map = new TreeMap<>();
        
        for(int num : nums) {
            map.put(num, map.getOrDefault(num, 0)+1);
        }
        
        while(!map.isEmpty()) {
            int num = map.firstKey();
            for(int i=0;i<k;i++) {
                if(!map.containsKey(num)) {
                    
                    return false;
                }
                map.put(num, map.get(num)-1);
                if(map.get(num)==0) {
                    map.remove(num);
                }
                num++;
            }
        }
        
        return true;
        
    }
}
