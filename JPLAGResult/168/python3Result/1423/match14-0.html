class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        checker, cnt = {}, {}
        for i in range(minSize):
            cnt[s[i]] = cnt.get(s[i], 0) + 1
        if len(cnt) <= maxLetters:
            checker[s[:minSize]] = 1
        for i in range(1, len(s) - minSize + 1):
            cnt[s[i - 1]] -= 1
            cnt[s[i + minSize - 1]] = cnt.get(s[i + minSize - 1], 0) + 1
            if cnt[s[i - 1]] == 0:
                del cnt[s[i - 1]]
            if len(cnt) <= maxLetters:
                checker[s[i: i + minSize]] = checker.get(s[i: i + minSize], 0) + 1
        return max(checker.values()) if checker.values() else 0
