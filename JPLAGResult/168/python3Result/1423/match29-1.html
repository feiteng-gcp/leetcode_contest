from collections import Counter
class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        if len(s)<minSize:
            return 0
        cnt = Counter()
        c = Counter()
        ans = 0
        cc = 0
        for i in range(minSize):
            v = s[i]
            c[v]+=1
            if c[v]==1:
                cc+=1
        if cc<=maxLetters:
            cnt[s[:minSize]]+=1
        for l in range(len(s)-minSize):
            r = l+minSize
            val 
= s[l+1:r+1]
            c[s[l]]-=1
            if c[s[l]]==0:
                cc-=1
            if c[s[r]]==0:
                cc+=1
            c[s[r]]+=1
            if cc<=maxLetters:
                cnt[val]+=1
                ans = max(ans, cnt[val])
        print(cnt)
        return ans
