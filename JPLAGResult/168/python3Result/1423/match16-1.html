class Solution:
    def maxFreq(self, S: str, MU: int, m: int, M: int) -> int:
        L, D, K = len(S), collections.defaultdict(int), set()
        for i in range(m,M+1):
            for j in range(L):
                if j in K: continue
                if j + i > L: break
                s 
= S[j:j+i]
                if len(set(s)) <= MU: D[s] += 1
                else: K.add(j)
        return max(D.values()) if D else 0
