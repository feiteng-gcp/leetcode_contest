from collections import Counter,defaultdict


class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:

        for k in range(minSize,maxSize+1):
            dic 
= defaultdict(int)
            for i in range(len(s)-k+1):
                dic[s[i:i+k]] += 1
            res = 0
            for m in dic:
                if len(set(m)) <= maxLetters:
                    res = max(res, dic[m])
            
            if res:
                return res
        return 0
