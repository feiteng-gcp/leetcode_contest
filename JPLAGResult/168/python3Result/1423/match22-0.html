import collections

class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        counter = collections.Counter()
        for i in range(len(s) - minSize + 1):
            ss = s[i:i + minSize]
            if len(set(ss)) <= maxLetters:
                counter[ss] += 1
        if len(counter.values()) == 0:
            return 0
        else:
            return max(counter.values())
