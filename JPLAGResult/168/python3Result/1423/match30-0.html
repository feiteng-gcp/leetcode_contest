class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        # we don't care about maxsize
        bigolddatastructure = {0:0} #the 0 is so we have a base case
        for index in range(0,len(s)+1-minSize):
            substring = s[index:index+minSize]
            unique = set()
            for char in substring:
                unique.add(char)
            if len(unique)<=maxLetters:
                bigolddatastructure[substring] = bigolddatastructure.get(substring,0)+1
        
        #print(bigolddatastructure)
        return max(bigolddatastructure.values())
