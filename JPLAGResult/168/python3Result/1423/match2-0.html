from collections import defaultdict


class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        l = len(s)
        d = defaultdict(int)
        for size in range(minSize, maxSize + 1):
            for i in range(0, l - size + 1):
                ss = s[i:i + size]
                if len(set(ss)) <= maxLetters:
                    d[ss] += 1
        if not d:
            return 0
        return max(d.values())
