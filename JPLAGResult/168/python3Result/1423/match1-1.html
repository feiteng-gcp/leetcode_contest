class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        n = len(s)
        wc = collections.defaultdict(int)
        for i in range
(n-minSize+1):
            w = s[i: i+minSize]
            if len(set(w)) <= maxLetters:
                wc[w] += 1

        return max(wc.values() or [0])
