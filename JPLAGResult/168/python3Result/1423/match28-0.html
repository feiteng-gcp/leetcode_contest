class Solution:
    def maxFreq(self, s, maxLetters, minSize, maxSize):
        counts = {}
        for start in range(0, len(s) - minSize + 1):
            for end in range(start + minSize, min(start + maxSize + 1, len(s) + 1)):
                temp = s[start:end]
                #print(start, end, temp)
                uniques = len(set(temp))
                if uniques <= maxLetters:
                    counts[temp] = counts.get(temp, 0) + 1
        if not counts:
            return 0
        return max(counts.values())
