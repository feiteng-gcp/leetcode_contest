class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        d = collections.defaultdict(int)
        seen = set()
        for size in range(minSize, maxSize + 1):
            for i in range(size, len(s) + 1):
                sub = s[i - size:i]
                if len(set(sub)) <= maxLetters:
                    d[sub] += 1
                    
        m = 0
        for key in d:
            if d[key] > m:
                m = d[key]
        return m
