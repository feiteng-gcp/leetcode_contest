import collections

class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        ans = 0
        for l in range(minSize, maxSize + 1):
            C = [0] * 26
            M = collections.defaultdict(int)
            V = 0
            for i in range(len(s)):
                c = ord(s[i]) - ord('a')
                C[c] += 1
                if C[c] == 1:
                    V += 1
                if i >= l:
                    c = ord(s[i - l]) - ord('a')
                    C[c] -= 1
                    if C[c] == 0:
                        V -= 1
                if i >= l - 1:
                    if V <= maxLetters:
                        M[s[i - l + 1:i + 1]] += 1
            # print(l, M.items())
            if len(M) > 0:
                ans = max(ans, sorted(M.values())[-1])
        return ans
