class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        count = {}
        bal = 0
        letter_freq = [0] * 26
        for i in range(minSize - 1):
            cur_char = ord(s[i]) - ord('a')
            letter_freq[cur_char] += 1
            if letter_freq[cur_char] == 1:
                bal += 1
            
        
for end in range(minSize - 1, len(s)):
            cur_char = ord(s[end]) - ord('a')
            letter_freq[cur_char] += 1
            if letter_freq[cur_char] == 1:
                bal += 1
            if end >= minSize:
                prev_char = ord(s[end - minSize]) - ord('a')
                letter_freq[prev_char] -= 1
                if letter_freq[prev_char] == 0:
                    bal -= 1
            if bal <= maxLetters:
                cur = s[end - minSize + 1:end + 1]
                count.setdefault(cur, 0)
                count[cur] += 1
                
        ans = sorted([(freq, substring) for substring, freq in count.items()], reverse=True)
        return 0 if not ans else ans[0][0]
