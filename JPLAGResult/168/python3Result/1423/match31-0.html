class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        i, j = 0, 0
        d, letters = collections.defaultdict(int), 0
        
        valid = collections.defaultdict(int)
        for i in range(len(s)-minSize+1):
            cnt = collections.Counter(s[i:i+minSize])
            if len(cnt.keys()) > maxLetters:
                continue
            valid[s[i:i+minSize]] += 1
            for j in range(i+minSize, len(s)):
                cnt[s[j]] += 1
                if len(cnt.keys()) > maxLetters:
                    break
                valid[s[i:j+1]] += 1
        
        res = 0
        for _, x in valid.items():
            res = max(res, x)
        
        return res
