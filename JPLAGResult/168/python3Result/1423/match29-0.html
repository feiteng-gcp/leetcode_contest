class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        count = collections.defaultdict(int)
        ret = 0
        n = len(s)
        nums = [0] * n
        con = ord('a')
        for i in range(n):
            nums[i] = ord(s[i]) - con + 1
        
        for i in range(minSize, maxSize + 1):
            cnt = [0] * 27
            tot = 0
            mod = 1
            code = 0
            for j in range(i):
                code = code * 27 + nums[j]
                mod *= 27
                if cnt[nums[j]] == 0: tot += 1
                cnt[nums[j]] += 1
            if tot <= maxLetters: count[code] += 1
            ret = max(ret, count[code])
            for j in range(i, n):
                code = (code * 27 + nums[j]) % mod
                cnt[nums[j - i]] -= 1
                if cnt[nums[j - i]] == 0: tot -= 1
                if cnt[nums[j]] == 0: tot += 1
                cnt[nums[j]] += 1
                if tot <= maxLetters:
                    count[code] += 1
                    ret = max(ret, count[code])
        return ret
