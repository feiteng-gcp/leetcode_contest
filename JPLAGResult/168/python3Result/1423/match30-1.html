class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        # a=[1,2,3,4,5]
        # print(a[2:4])
        # thought: brute force all combinations
        from collections import defaultdict
        freq=defaultdict(int)
        for i in range (len(s)-minSize+1): #+1?
            for j in range (minSize,minSize+1):#max doesn't mmatter
                
                # for k in range (i+1,i+j):
                if i+j>len(s):
                    continue
                substring
=s[i:i+j] #+1?
                #print(substring)
                letters=set()
                for l in substring:
                    letters.add(l)
                if len(letters)<=maxLetters:
                    freq[substring]+=1
        #print(freq)
        if bool(freq):
            return max(list(freq.values()))
        return 0
