class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        
        n = len(s)
        
        curmax = 0
        ctr = collections.Counter()
        
        for i in range(0, n - minSize + 1):
            sub = s[i: i + minSize]
            lset = set(sub)
            if len(lset) <= maxLetters:
                ctr[sub] += 1
                if ctr[sub] > curmax:
                    curmax = ctr[sub]
        return curmax
