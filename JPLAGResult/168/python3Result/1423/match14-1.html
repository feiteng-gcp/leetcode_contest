class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        count = collections.Counter()
        letters = collections.Counter(s[:minSize])
        #print(letters, len(letters))
        
if len(letters) <= maxLetters:
            count[s[:minSize]] = 1
            
        for i in range(1, len(s) - minSize+1):
            letters[s[i+minSize-1]] += 1
            letters[s[i-1]] -= 1
            
            
if letters[s[i-1]] == 0:
                del letters[s[i-1]]
            if len(letters) <= maxLetters:
                count[s[i:i+minSize]] += 1
        
        # print(count)
                
        return max([count[k] for k in count]) if count else 0
