class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        cc = collections.Counter()
        n = len(s)
        for i in range(n-minSize+1):
            for j in range(i+minSize, 1+min(n, i+maxSize)):
                sc = s[i:j]
                if len(set(sc)) <= maxLetters:
                    cc[sc] += 1
        if len(cc) == 0:
            return 0
        return max(cc.values())
