class Solution:
    def maxFreq(self, s: str, maxLetters: int, minSize: int, maxSize: int) -> int:
        occur 
= collections.defaultdict(int)
        for i in range(len(s)-minSize+1):
            count = collections.Counter(s[i:i+minSize])
            # print(s[i:i+minSize])
            if len(count.keys()) <= maxLetters:
                occur[s[i:i+minSize]] += 1
        max_occur = 0
        for o in occur:
            if occur[o] > max_occur:
                max_occur = occur[o]
                
        return max_occur
