class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], contain: List[List[int]], q: List[int]) -> int:
        res,l = 0, len(status) 
        used,flag = {},True
        while flag:
            flag = False
            for x in q:
                if used.get(x,True) == True and status[x] == 1:
                    used[x] = False
                    res += candies[x]
                    for k in keys[x]:
                        status[k] = 1
                    for k in contain[x]:
                        q.append(k)
                    flag = True
        return res
