class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], cb: List[List[int]], init: List[int]) -> int:
        if not init:
            return 0
        ks = set()
        boxes = set()
        res = 0
        for i in init:
            res += candies[i]
            for b in cb[i]:
                boxes.add(b)
            for k in keys[i]:
                ks.add(k)
        
        while True:
            nboxes = set()
            flag = False
            for b in boxes:
                if status[b] or b in ks:
                    res += candies[b]
                    for bb in cb[b]:
                        nboxes.add(bb)
                    for k in keys[b]:
                        ks.add(k)
                    flag = True
                else:
                    nboxes.add(b)
            boxes = nboxes
            if not flag:
                break
        
        return res
