class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        n = len(status)
        todo = []
        waiting = []
        availableKeys = [False] * n
        for box in initialBoxes:
            if status[box]:
                todo.append(box)
            else:
                waiting.append(box)
        res = 0        
        while todo:
            for box in todo:
                res += candies[box]
                for key in keys[box]:
                    availableKeys[key] = True
                for newBox in containedBoxes[box]:
                    waiting.append(newBox)
            todo = []
            nextWaiting = []
            # print(waiting)
            for box in waiting:
                if availableKeys[box] or status[box]:
                    todo.append(box)
                else:
                    nextWaiting.append(box)
            waiting = nextWaiting
        
        return res
