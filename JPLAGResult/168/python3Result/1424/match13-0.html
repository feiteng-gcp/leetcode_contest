class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]],
                   initialBoxes: List[int]) -> int:
        n = len(status)
        available_closed = set([i for i in initialBoxes if status[i] == 0])
        available_open = set([i for i in initialBoxes if status[i] == 1])
        sol = 0
        while available_open:
            i = available_open.pop()
            sol += candies[i]
            for b in containedBoxes[i]:
                if status[b]:
                    available_open.add(b)
                else:
                    available_closed.add(b)
            for k in keys[i]:
                status[k] = 1
                if k in available_closed:
                    available_closed.remove(k)
                    available_open.add(k)
        return sol
