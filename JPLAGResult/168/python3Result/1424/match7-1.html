class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        # found_keys = set()
        found_closed_boxes = set()
        # found_opened_boxes = list(initialBoxes)
        box_to_check = list(initialBoxes)
        found_candy_count = 0
        opened_boxes = set()
        
        while box_to_check:
            bi = box_to_check.pop()
            
            # get keys and closed boxes
            if status[bi] == 0:
                found_closed_boxes.add(bi)
            # if is opened
            else:
                if bi not in opened_boxes:
                    opened_boxes.add(bi)
                    found_candy_count += candies[bi]
                    box_to_check.extend(containedBoxes
[bi])
                    for ki in keys[bi]:
                        status[ki] = 1
                        if ki in found_closed_boxes:
                            found_closed_boxes.remove(ki)
                            box_to_check.append(ki)
        return found_candy_count
