class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        availableBoxes = set(initialBoxes)
        openBoxes = set()
        for i,stat in enumerate(status):
            if stat:
                openBoxes.add(i)
        total = 0
        
        while openBoxes and availableBoxes:
            newavail = set()
            for box in availableBoxes:
                if box in openBoxes:
                    openBoxes.remove(box)
                    total += candies[box]
                    for newbox in keys[box]:
                        openBoxes.add(newbox)
                    for newbox in containedBoxes[box]:
                        newavail.add(newbox)
                else:
                    newavail.add(box)
            availableBoxes = newavail
        return total
