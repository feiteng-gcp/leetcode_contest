class Solution:
    def maxCandies(self, s: List[int], candies: List[int], keys: List[List[int]], new: List[List[int]], have: List[int]) -> int:
        n = len(s)
        #print(n)
        haveopen = []
        haveclosed = set()
        havek = set()
        ans = 0
        for b in have:
            if s[b]:
                haveopen.append(b)
            else:
                haveclosed.add(b)
        check = set(haveopen)
        while haveopen:
            #print(haveopen)
            check |= set(haveopen)
            for b in haveopen:
                ans += candies[b]
                havek |= set(keys[b])
                for x in new[b]:
                    if s[x]:
                        haveopen.append(x)
                    else:
                        haveclosed.add(x)
            use = haveclosed & havek
            havek -= use
            haveopen = list(use - check)

        
        return ans
