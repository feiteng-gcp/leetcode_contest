class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        #ends when no open box. key to change status
        box=initialBoxes
        candy=0
        while box:
            #print(candy)
            dead=True
            for b in box:
                if status[b]:
                    dead=False
                    box.extend(containedBoxes[b])
                    for k in keys[b]:
                        status[k]=1
                    candy+=candies[b]
                    box.remove(b)
                    #status[b]=0#maybe pop
            if dead:
                return candy
        return candy
