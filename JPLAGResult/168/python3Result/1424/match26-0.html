from collections import deque
class Solution:
    def maxCandies(self, opened: List[int], candies: List[int], keys: List[List[int]], boxes: List[List[int]], initial: List[int]) -> int:
        
        result = 0
        
        q = deque()
        closed = set()
        for box in initial:
            if opened[box]:
                q.append(box)
            else:
                closed.add(box)
        
        while q:
            cur = q.popleft()
            
            result += candies[cur]
            
            for k in keys[cur]:
                opened[k] = 1
                if k in closed:
                    q.append(k)
                    closed.remove(k)
                    
            for b in boxes[cur]:
                if opened[b]:
                    q.append(b)
                else:
                    closed.add(b)
                
        return result
