class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        n = len(status)
        c = 0
        o = set() # old
        s = set(initialBoxes) # seen
        k = set() # keys
        u = set(b for b in range(n) if status[b]) # unlocked
        q = []
        while True:
            new = s.intersection(u)
            if len(new) == 0:
                break
            for b in new:
                k.update(keys[b])
                s.update(containedBoxes[b])
                o.add(b)
                s.remove(b)
                c += candies[b]
            u.update(k) # unlock all
        return c
