class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        found_keys = set()
        res = 0
        while initialBoxes:
            tmp = []
            for ib in initialBoxes:
                if status[ib] == 1 or ib in found_keys:
                    # get candy
                    res += candies[ib]
                    for key in keys[ib]:
                        found_keys.add(key)
                    tmp += containedBoxes[ib]
                else:
                    tmp.append(ib)
            if initialBoxes == tmp:
                break
            initialBoxes = tmp
        return res
