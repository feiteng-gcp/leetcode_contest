class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        k = set()
        boxes = set(initialBoxes)
        to_open = collections.deque
()
        seen = set()
        ret = 0
        
        for b in boxes:
            if b not in seen and status[b] == 1 or b in k:
                to_open.append(b)
                seen.add(b)
                
        while to_open:
            #print(to_open, boxes, seen, k)
            b = to_open.popleft()
            
            ret += candies[b]
            k |= set(keys[b])
            boxes |= set(containedBoxes[b])
            
            #print(b, k, keys[b])
            for b in boxes:
                if b in seen:
                    continue
                    
                if status[b] == 1 or b in k:
                    to_open.append(b)
                    seen.add(b)
                    
            #print(to_open, boxes, seen, k)
            
        return ret
