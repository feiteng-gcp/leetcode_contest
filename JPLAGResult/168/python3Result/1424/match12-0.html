class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        n = len(status)
        seen_locked = [False] * n
        ans = 0
        box_layer = initialBoxes
        keys_layer = []
        while box_layer or keys_layer:
            next_box_layer = []
            next_keys_layer = []
            for key in keys_layer:
                if status[key] == 0:
                    status[key] = 1
                    if seen_locked[key]:
                        ans += candies[key]
                        box_layer.extend(containedBoxes[key])
                        next_keys_layer.extend(keys[key])
            for box in box_layer:
                if status[box] == 0:
                    seen_locked[box] = True
                else:
                    ans += candies[box]
                    next_box_layer.extend(containedBoxes[box])
                    next_keys_layer.extend(keys[box])
            box_layer, keys_layer = next_box_layer, next_keys_layer
        return ans
