class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        
        available = set(initialBoxes)
        visited = set(initialBoxes)
        queue = collections.deque(initialBoxes)
        result = 0
        
        while queue:
            box = queue.popleft()
            result += candies[box]
            
            for cont in containedBoxes[box]:
                if cont in visited:
                    continue
                    
                available.add(cont)
                if status[cont]:
                    queue.append(cont)
                    visited.add(cont)
            
            for key in keys[box]:
                if key in visited:
                    continue
                
                status
[key] = 1
                if key in available:
                    queue.append(key)
                    visited.add(key)
        
        return result
