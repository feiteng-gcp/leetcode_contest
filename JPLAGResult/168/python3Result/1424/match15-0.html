class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        have_key = set()
        have_box = set()
        opened = set()
        stk = []
        total = 0
        for idx, is_open in enumerate(status):
            if is_open:
                have_key.add(idx)
        for i in initialBoxes:
            if i in have_key:
                stk.append(i)
                opened.add(i)
            else:
                have_box.add(i)
        while (len(stk) > 0):
            idx = stk.pop()
            total += candies[idx]
            for i in keys[idx]:
                if i in have_box and i not in opened:
                    opened.add(i)
                    stk.append(i)
                else:
                    have_key.add(i)
            for i in containedBoxes[idx]:
                if i in have_key and i not in opened:
                    opened.add(i)
                    stk.append(i)
                else:
                    have_box.add(i)
        return total
