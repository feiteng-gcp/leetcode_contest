class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        opened=set()
        key = set()
        tt = 0
        while initialBoxes:
            new = set()
            r = []
            for i in initialBoxes:
                if (status[i]==1 or i in key) and (i not in opened):
                    opened.add(i)
                    new.add(i)
                    tt
+=candies[i]
                    for j in containedBoxes[i]:
                        r.append(j)
                    for j in keys[i]:
                        key.add(j)
                else:
                    r
.append(i)
            initialBoxes = r
            if not new:
                break
        return tt
