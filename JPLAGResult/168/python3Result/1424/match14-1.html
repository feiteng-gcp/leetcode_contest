class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        n = len(status)
        status = [bool(box) for box in status]
        ctr = 0
        opened = {box for box in initialBoxes if status[box]}
        closed = {box for box in initialBoxes if not status[box]}
        while opened:
            v = opened.pop()
            
            ctr 
+= candies[v]
            
            for to in containedBoxes[v]:
                if status[to]:
                    opened.add(to)
                else:
                    closed.add(to)
            
            for k in keys[v]:
                if k in closed:
                    opened.add(k)
                    closed.remove(k)
                else:
                    status[k] = True
        return ctr
