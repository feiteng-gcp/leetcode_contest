class Solution:
    def maxCandies(self, status: List[int], candies: List[int],
                   keys: List[List[int]], containedBoxes: List[List[int]],
                   initialBoxes: List[int]) -> int:
        closed_boxes = set()
        opened_boxes = set()
        for box in initialBoxes:
            if status[box] == 0:
                closed_boxes.add(box)
            else:
                opened_boxes.add(box)
        res = 0
        while opened_boxes:
            new_opened_boxes = set()
            for box in opened_boxes:
                res += candies[box]
                for key in keys[box]:
                    status[key] = 1
                    if key in closed_boxes:
                        new_opened_boxes.add(key)
                        closed_boxes.remove(key)
                for box in containedBoxes[box]:
                    if status[box] == 0:
                        closed_boxes.add(box)
                    else:
                        new_opened_boxes.add(box)
            opened_boxes = new_opened_boxes

        return res
