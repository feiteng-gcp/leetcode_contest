from typing import List

class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        boxCount = len(status)
        myCandy = 0
        myKey = set()
        myboxes = set(initialBoxes)

        mb = set()
        mk = set()

        while mb != myboxes:
            mb = set(myboxes)
            for box in list(myboxes):
                if status[box] == 1 or box in myKey:
                    myCandy += candies[box]
                    myKey |= set(keys[box])
                    myboxes |= set(containedBoxes[box])
                    myboxes.remove(box)
               
        return myCandy
