class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        ans = 0
        ck = [0] * len(status)
        q = set()
        visited = set()
        hold = set()
        for i in set(initialBoxes):
            if status[i] or ck
[i]:
                q.add(i)
            else:
                hold.add(i)
        while q:
            u = q.pop()
            ans += candies[u]
            visited.add(u)
            for v in keys[u]:
                if v not in visited:
                    if v in hold:
                        q.add(v)
                    else:
                        ck[v] = 1
            for v in containedBoxes[u]:
                if v not in visited:
                    if status[v] or ck[v]:
                        q.add(v)
                    else:
                        hold.add(v)
        return ans
