class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        dic = {}
        ans = 0
        while initialBoxes:
            temp = []
            for i in range(len(initialBoxes)):
                box = initialBoxes[i]
                if status[box] or box in dic:
                    ans += candies
[box]
                    for key in keys[box]:
                        dic[key] = 1
                    initialBoxes += containedBoxes[box]
                    temp.append(i)
            if temp:
                for i in temp[::-1]:
                    initialBoxes.pop(i)
            else:
                break
        return ans
