class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        n = len(status)
        
        findKeys = [0] * n
        findBoxes = [0] * n
        for i in initialBoxes:
            findBoxes[i] = 1
        
        visited = [0] * n
        res = 0
        while True:
            find = False
            i = 0
            while not find and i < n:
                if visited[i] == 0 and findBoxes[i] == 1 and status[i] + findKeys[i] > 0:
                    find = True
                else:
                    i += 1
            if not find:
                break
            visited[i] = 1
            res += candies[i]
            for k in keys[i]:
                findKeys[k] = 1
            for b in containedBoxes[i]:
                findBoxes[b] = 1
        
        return res
