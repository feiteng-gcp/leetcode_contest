class Solution:
    def maxCandies(self, sta: List[int], cd: List[int], keys: List[List[int]], cb: List[List[int]], ib: List[int]) -> int:
        n = len(sta)
        hak = [0] * n
        res = 0
        hb = ib
        f = 1
        while hb and f:
            leb = []
            f = 0
            for b in hb:
                if sta[b] == 1 or hak
[b] == 1:
                    f = 1
                    res += cd[b]
                    for newkey in keys[b]:
                        hak[newkey] = 1
                    for newb in cb[b]:
                        leb.append(newb)
                else:
                    leb.append(b)
            hb = leb[:]
        return res
