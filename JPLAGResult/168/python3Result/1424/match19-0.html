class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        n = len(status)
        # graph = [set() for _ in range(n)]
        # in_degree = [0]*n
        # for u, contained in enumerate(containedBoxes):
        #     for v in contained:
        #         graph[u].add(v)
        #         in_degree[v] += 1
        # queue = []
        # for i, degree in enumerate(in_degree):
        #     if degree == 0:
        #         queue.append(i)
        queue = []
        for box in initialBoxes:
            queue.append(box)
        newKeys = set()
        res = 0
        while queue:
            length = len(queue)
            #try to get all keys
            newBoxes = []
            opened = set()
            while True:
                addedKeys = set()
                for curr in queue:
                    if (status[curr] == 1 or curr in newKeys)  and curr not in opened:
                        res += candies[curr]
                        for key in keys[curr]:
                            newKeys.add(key)
                            addedKeys.add(key)
                        for box in containedBoxes[curr]:
                            newBoxes.append(box)
                        opened.add(curr)
                if len(addedKeys) == 0:
                    break
            queue = newBoxes
        return res
