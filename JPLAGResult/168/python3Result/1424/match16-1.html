class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        found_keys = set()
        boxes = set(initialBoxes)
        ret = 0
        opened = True
        while boxes and opened:
            new_boxes = set()
            opened = False
            
for bid in list(boxes):
                if status[bid] == 1 or bid in found_keys:
                    ret += candies[bid]
                    found_keys |= set(keys[bid])
                    new_boxes |= set(containedBoxes[bid])
                    boxes.remove(bid)
                    opened = True
                    
            boxes |= new_boxes
        return ret
