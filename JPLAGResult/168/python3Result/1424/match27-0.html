class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]],
                   initialBoxes: List[int]) -> int:
        queue = initialBoxes
        myKeys = set()
        myCandies = 0
        while queue:
            new_queue = []
            flag = False
            for box in queue:
                # print(box)
                if status[box] or box in myKeys:
                    myCandies += candies[box]
                    myKeys |= set(keys[box])
                    new_queue += containedBoxes[box]
                    flag = True
                else:
                    new_queue.append(box)
            if not flag:break
            queue = new_queue
        return myCandies
