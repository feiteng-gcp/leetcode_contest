class Solution:
    def maxCandies(self, status: List[int], candies: List[int], keys: List[List[int]], containedBoxes: List[List[int]], initialBoxes: List[int]) -> int:
        n = len(status)
        found = [0]*n
        seen = [0]*n
        
        res = 0
        q = collections.deque(initialBoxes)
        while q:
            b = q.popleft()
            #print(res)
            res
+=candies[b]
            for nxtb in containedBoxes[b]:
                found[nxtb]=1
            for nxtb in keys[b]:
                status[nxtb]= 1
            for i in range(n):
                if found[i]==1 and seen[i]==0 and status[i]==1:
                    seen[i] = 1
                    q.append(i)
        return res
