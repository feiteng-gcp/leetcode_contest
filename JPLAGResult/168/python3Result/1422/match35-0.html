class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k != 0:
            return False
        nums.sort()
        dic = collections.Counter(nums)
        l = 0
        while l < len(nums):
            start = nums[l]
            if start in dic and dic[start] > 0:
                for i in range(start, start+k):
                    if i in dic and dic[i] > 0:
                        dic[i] -= 1
                    else:
                        return False
            else:
                l += 1
        return True
