class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        from collections import defaultdict
        count = defaultdict(int)
        for n in nums:
            count[n] += 1
        
        unique_nums = list(set(nums))
        unique_nums.sort()
        
        for u in unique_nums:
            while count[u] > 0:
                count[u] -= 1
                current, target = u, k
                while target > 1:
                    if count[current + 1] > 0:
                        count[current + 1] -= 1
                        current += 1
                        target -= 1
                    else:
                        return False
        
        return True
