from collections import Counter

class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums)%k:
            return False
        
        counter = Counter(nums)
        while counter:
            start = min(counter.keys())
            for n in range(start,start+k):
                if n in counter:
                    counter
[n] -= 1
                    if counter[n] == 0:
                        del counter[n]
                else:
                    return False
                    
        return True
