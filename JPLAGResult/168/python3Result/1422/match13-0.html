class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        count = collections.Counter(nums)
        nums.sort()
        for i,c in enumerate(nums):
            if count[c] > 0:
                for j in range(k):
                    if count[c+j] == 0:
                        return False
                    else:
                        count[c+j] -= 1
        return True
