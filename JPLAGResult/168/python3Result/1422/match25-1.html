class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k:
            return False
        count = collections.Counter(nums)
        while count:
            start 
= min(count.keys())
            ct = count[start]
            for i in range(k):
                if start + i not in count or count[start+i] < ct:
                    return False
                count[start+i] -= ct
                if count[start+i] == 0:
                    count.pop(start+i)
        return True
