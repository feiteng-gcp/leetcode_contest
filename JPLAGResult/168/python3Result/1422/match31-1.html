class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        d = collections.defaultdict(lambda:0)
        
        for n in nums:
            d[n] += 1
        
        s = list(d.keys())
        s = sorted(s)        
        
        while len(s) > 0:
            #print(s, d)
            n = s.pop(0)
            
            
if d[n] == 0:
                continue
            
            f = d[n]
            
            for i in range(0, k):
                if d[n+i] < f:
                    return False
                d[n+i] = d[n+i] - f
        
        return True
