from collections import Counter
class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        cnt = len(nums)
        if cnt%k != 0:
            return False
        sig = cnt // k
        cter = Counter(nums)
        while len(cter) > 0:
            # key = sorterd(cter.keys())
            mn = min(cter.keys())
            for i in range(k):
                tmp = mn + i
                if cter[tmp] > 0:
                    cter[tmp] -= 1
                    if cter[tmp] == 0:
                        del cter[tmp]
                else:
                    return False
        return True
