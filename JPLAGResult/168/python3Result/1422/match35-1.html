class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums)%k!=0:
            return False
        nums.sort()
        s=list(set(nums))
        s.sort()
        s.reverse()
        print(s)
        d=dict()
        for n in nums:
            if n not in d:
                d[n]=1
            else:
                d[n]+=1
        
while len(s)>0:
            a=s[-1]
            if a in d and d[a]>0:
                for v in range(a,a+k):
                    if v in d and d[v]>0:
                        d[v]-=1
                    else:
                        return False
            else:
                s.pop()
        return True
