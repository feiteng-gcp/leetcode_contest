class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        d = {}
        for x in nums:
            if x in d:
                d[x] += 1
            else:
                d[x] = 1
        ke = sorted(d.keys())
        
        for index,val in enumerate(ke):
            if d[val] != 0:
                u = d
[val]
                for i in range(k):
                    if i+val not in d or d[i+val] < u:
                        return False
                    else:
                        d[i+val] -= u
        return True
