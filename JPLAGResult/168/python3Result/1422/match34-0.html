from heapq import heappop, heappush
class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k: return False
        nums.sort()
        buf = []
        for x in nums:
            if buf and buf[0][0] == x:
                n = heappop(buf)[1] + 1
                if n != k: heappush(buf, (x + 1, n))
            else: heappush(buf, (x + 1, 1))
        return not buf
