class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k != 0:
            return False
        
        cts = collections.Counter(nums)
        m = sorted( cts.keys() )
        
        i = 0
        while i < len(m): 
            x = m[i]
            if cts[x] > 0:
                for j in range(k):
                    y = x + j
                    if y in cts and cts[y] > 0:
                        cts[y] -= 1
                    else:
                        return False
            else:
                i += 1
        
        return True
