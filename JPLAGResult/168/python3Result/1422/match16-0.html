class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        cnt = collections.Counter(nums)
        for x in sorted(cnt):
            n = cnt[x]
            if n == 0:
                continue
            for y in range(x, x + k):
                if cnt[y] < n:
                    return False
                cnt[y] -= n
        return True
