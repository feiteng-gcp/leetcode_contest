from collections import OrderedDict
class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k != 0:
            return False
        m = OrderedDict()
        nums.sort()
        for num in nums:
            if num in m:
                m[num] += 1
            else:
                m[num] = 1
        while m:
            start = list(m.keys())[0]
            
for i in range(k):
                if start + i in m:
                    m[start + i] -= 1
                    if m[start + i] == 0:
                        del m[start + i]
                else:
                    return False
        
        return True
