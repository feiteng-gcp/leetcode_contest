class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        nums_c = collections.Counter(nums)
        nums = list(set(nums))
        nums.sort()
        for num in nums:
            if not nums_c[num]:
                continue
            total = nums_c[num]
            for dx in range(k):
                if nums_c[num+dx] < total:
                    return False
                nums_c[num+dx] -= total
        return True
