class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        count = collections.Counter(nums)
        for x in sorted(count.keys()):
            if count[x] == 0:
                continue
            temp = count[x]
            for i in range(k):
                count[x + i] -= temp
                if count[x + i] < 0:
                    return False
        return True
