class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k != 0:
            return False
        count = collections.Counter(nums)
        keys = sorted(list(count.keys()))
        #print(count, keys)
        for p in keys:
            if count[p] != 0:
                #print(p)
                n = count[p]
                for i in range(k):
                    
                    #print(p, p+i, count[p], count[p+i])
                    if count[p+i] < n:
                        return False
                    count[p+i] -= n
                #print(count)
        return True
