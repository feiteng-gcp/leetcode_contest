from collections import Counter

class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k != 0:
            return False
        nums = Counter(nums)
        while nums:
            n = min(nums)
            #print(nums)
            for x in range(n, n+k):
                if nums[x] == 0:
                    #print("can't find " + str(x) + " in " + str(nums))
                    return False
                nums[x] -= 1
                if nums[x] == 0:
                    del nums[x]
        return True
