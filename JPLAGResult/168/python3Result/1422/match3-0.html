import collections


class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        n = len(nums)
        if n % k != 0:
            return False
        counter = collections.Counter(nums)
        keys = sorted(counter.keys())
        for key in keys:
            x = counter[key]
            if x > 0:
                for i in range(k):
                    counter[key + i] -= x
                    if counter[key + i] < 0:
                        return False
        for key in keys:
            if counter[key] != 0:
                return False
        return True
