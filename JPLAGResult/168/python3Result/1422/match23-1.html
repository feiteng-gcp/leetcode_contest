from typing import List

class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        from collections import defaultdict

        d = {}

        if len(nums) % k != 0:
            return False

        for x in nums:
            if x not in d:
                d[x] = 1
            else:
                d[x] += 1
        # print(d)

        while d.keys
():
            e = min(d.keys())

            for x in range(e, e+k):
                if x not in d:
                    return False
                d[x] -= 1
                if d[x] == 0:
                    del d[x]
                elif d[x] < 0:
                    # print(d)
                    return False
            # print(e)
            # print(d)
        return True
