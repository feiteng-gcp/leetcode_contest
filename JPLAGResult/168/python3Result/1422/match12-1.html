from collections import Counter

class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        c = Counter(nums)
        keys = list(sorted(set(nums)))
        idx = 0

        if len(nums) % k != 0:
            return False

        for _ in range(len(nums) // k):
            while c[keys[idx]] == 0:
                idx += 1
            start_num = keys[idx]
            
for i in range(k):
                if c[start_num + i] == 0:
                    return False
                c[start_num + i] -= 1

        return True
