class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        cnt = collections.Counter(nums)
        
        while cnt:
            #print(cnt)
            mn = min(cnt.keys())
            n = cnt[mn]
            #print(mn, n)
            for i in range(k):
                if cnt.get(mn + i, 0) < n:
                    
return False
                else:
                    cnt[mn + i] -= n
                    if cnt[mn + i] == 0:
                        cnt.pop(mn + i)
        return True
