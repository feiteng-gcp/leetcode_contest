from collections import Counter
from heapq import heapify, heappop

class Solution:
  def isPossibleDivide(self, nums: List[int], k: int) -> bool:
    if len(nums) % k:
      return False
    
    cnt = Counter(nums)
    heapify(nums)
    
    while nums:
      el = heappop(nums)
      
if el not in cnt or cnt[el] == 0:
        continue

      for v in range(el, el + k):
        if v not in cnt or cnt[v] == 0:
          return False
        cnt[v] -= 1

    return True
