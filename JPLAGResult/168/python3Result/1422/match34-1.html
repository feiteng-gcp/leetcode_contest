class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k != 0:
            return False
        nums.sort()
        cnt = {}
        for n in nums:
            if n not in cnt:
                cnt[n] = 0
            cnt[n] += 1
        found = 0
        for n in nums:
            if n not in cnt:
                continue
            for m in range(n, n + k):
                if m not in cnt:
                    return False
                cnt[m] -= 1
                if cnt[m] == 0:
                    del cnt[m]
            found += 1
        return found * k == len(nums)
