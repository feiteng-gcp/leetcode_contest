class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        nums = collections.Counter(nums)
        
        
        for n in sorted(nums.keys()):
            if nums[n] == 0: continue
            for i in range(1,k):
                if n+i not in nums or nums[n+i] < nums[n]: return False
                nums[n+i] -= nums[n]
            nums[n] = 0
            #print(nums)
        return all(n == 0 for n in nums.values())
