class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        from collections import defaultdict
        freq=defaultdict(int)
        #print(bool(freq))
        # freq[1]=3
        # print(bool(freq))
        for n in nums:
            freq[n]+=1
            
        while bool(freq): #when not empty
            start=min(freq.keys())
            for i in range (k):
                if start+i not in freq:
                    
return False
                freq[start+i]-=1
                if freq[start+i]==0:
                    del freq[start+i]
                #print(freq)
        return True
