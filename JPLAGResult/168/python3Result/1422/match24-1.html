class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        c = collections.Counter(nums)
        for x in sorted(c.keys()):
            if (c[x] > 0):
                d = c[x]
                c[x] -= d
                
for j in range(1, k):
                    if x + j not in c or c[x + j] < d:
                        return False
                    c[x + j] -= d
        return True
