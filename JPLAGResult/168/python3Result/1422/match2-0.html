class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k != 0:
            return False
        
        cnt = collections.Counter(nums)
        seen = set()
        nums.sort()
        
        for item in nums:
            if item in seen or cnt[item] == 0:
                continue
            else:
                seen.add(item)
                
            prev_count = cnt[item]
            for i in range(1, k):
                if cnt[item + i] < prev_count:
                    return False
                else:
                    cnt[item + i] -= prev_count
        return True
