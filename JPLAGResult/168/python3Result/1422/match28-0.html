class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        ans = len(nums)
        if ans%k != 0:
            return False
        dd = {}
        nums.sort()
        for n in nums:
            dd[n] = dd.get(n,0)+1
        while len(dd)!=0:
            a = min(dd.keys())
            for i in range(a,a+k):
                if i in dd:
                    dd[i]-=1
                    if dd[i] == 0:
                        del dd[i]
                else:
                    return False
        return True
