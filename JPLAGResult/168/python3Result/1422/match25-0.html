class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k != 0:
            return False
        ngroups = len(nums)//k
        cnt = collections.Counter(nums)
        for _ in range(ngroups):
            sorted_key = sorted(cnt.keys())
            start = sorted_key[0]
            for _ in range(k):
                if cnt[start] <= 0:
                    return False
                
                cnt[start] -= 1
                if cnt[start] == 0:
                    del cnt[start]
                start += 1
                
        return True
