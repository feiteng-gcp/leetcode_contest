import collections

class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        c = collections.Counter(nums)
        for n in sorted(c.keys()):
            count 
= c[n]
            if count > 0:
                for i in range(k):
                    c[n + i] -= count
                    if c[n + i] < 0:
                        return False
        return all(not v for v in c.values())
