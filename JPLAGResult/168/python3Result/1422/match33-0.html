class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k != 0:
            return False
        cnts = dict()
        for i in nums:
            if i not in cnts:
                cnts[i] = 0
            cnts[i] += 1
        ss = sorted(list(set(cnts.keys())))
        for i in ss:
            if i not in cnts:
                continue
            cs = cnts[i]
            for j in range(k):
                if i + j not in cnts or cnts[i + j] < cs:
                    return False
                cnts[i + j] -= cs
                if cnts[i + j] == 0:
                    del cnts[i + j]
        return True
