from collections import Counter
class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        c = Counter(nums)
        nums.sort()
        for n in nums:
            if c[n] == 0:
                continue
            for i in range(n, n+k):
                if c[i] == 0:
                    return False
                c[i] -= 1
        return True
