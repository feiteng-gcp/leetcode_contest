class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        counts = collections.Counter(nums)
        while counts:
            left = [x for x in counts]
            minElement = min(left)
            for i in range(k):
                if minElement+i not in counts:
                    return False
                else:
                    counts[minElement+i] -= 1
                    if counts[minElement+i]==0:
                        counts.pop(minElement+i)
        return True
