class Solution:
    def isPossibleDivide(self, nums: List[int], k: int) -> bool:
        if len(nums) % k != 0:
            return False
        count = collections.Counter(nums)
        keys = list(count.keys())
        keys.sort()
        j = 0
        possible = True
        while j < len(keys) and possible:
            key = keys[j]
            # print(count)
            for i in range(k):
                if count[key+i] > 0:
                    count[key+i] -= 1
                else:
                    if i != 0:
                        possible = False
                        break
                    else:
                        break
            if possible and count[key] == 0:
                j += 1
        
        return possible
