class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        
        int ret = 0;
        queue<int> boxes_to_open;
        queue<int> boxes_cant_open_yet;
        
        for (int i = 0; i < initialBoxes.size(); i++){
            if (status[initialBoxes[i]] == 1){
                boxes_to_open.push(initialBoxes[i]);
            } else {
                boxes_cant_open_yet.push(initialBoxes[i]);
            }
        }
        
        while (!boxes_to_open.empty()){
            int box_num = boxes_to_open.front();
            boxes_to_open.pop();
            
            ret += candies[box_num];
            
            for (int i = 0; i < keys[box_num].size(); i++){
                status[keys[box_num][i]] = 1;
            }
            
            for (int i = 0; i < containedBoxes[box_num].size(); i++){
                boxes_cant_open_yet.push(containedBoxes[box_num][i]);
            }
            
            int t_size = boxes_cant_open_yet.size();
            for (int i = 0; i < t_size; i++){
                int b_num = boxes_cant_open_yet.front();
                boxes_cant_open_yet.pop();
                if (status[b_num] == 1){
                    boxes_to_open.push(b_num);
                } else {
                    boxes_cant_open_yet.push(b_num);
                }
            }
            
        }
        
        return ret;
        
    }
};
