#define BOX_IGNORE (1000000)
#define BOX_CLOSE (0)
#define BOX_OPEN (1)
#define BOX_TRAV (2)
#define BOX_READY (BOX_OPEN | BOX_TRAV)
class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int sol = 0;
        queue<int> q;

        for (int box : initialBoxes) {
            // 0: close
            // 1: open
            status[box] |= BOX_TRAV;
            if (status[box] == BOX_READY) {
                status[box] = BOX_IGNORE;
                q.push(box);
            }
        }
        
        while (!q.empty()) {
            // for each box traversed
            int now = q.front();
            q.pop();
            
            // get candies
            sol += candies[now];
            
            // For each box contained in it
            for (int box : containedBoxes[now]) {
                status[box] |= BOX_TRAV;
                if (status[box] == BOX_READY) {
                    status[box] = BOX_IGNORE;
                    q.push(box);
                }
            }
            
            // For each keys contained in it
            for (int box : keys[now]) {
                status[box] |= BOX_OPEN;
                if (status[box] == BOX_READY) {
                    status[box] = BOX_IGNORE;
                    q.push(box);
                }
            }
        }
        
        return sol;
    }
};
