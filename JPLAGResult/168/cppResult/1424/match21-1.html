class Solution {
public:
    int maxCandies(vector<int> &status, vector<int> &candies,
                   vector<vector<int>> &keys, vector<vector<int>> &containedBoxes,
                   vector<int> &initialBoxes) {
        unordered_map<int, int> myKeys;
        unordered_map<int, int> obtain;
        queue<int> Q;
        int ans = 0;

        for (int i:initialBoxes) {
            Q.push(i);
            ++obtain[i];
        }
        while (!Q.empty()) {
            int tmp = Q.front();
            Q.pop();
            if (obtain[tmp] == 0 || (status[tmp] == 0 && myKeys[tmp] == 0) || candies[tmp] == 0) continue;
            ans += candies[tmp];
            candies[tmp] = 0;
            for (int i:containedBoxes[tmp]) {
                ++obtain[i];
                Q.push(i);
            }
            for (int i:keys[tmp]) {
                ++myKeys[i];
                Q.push(i);
            }
        }
        return ans;
    }
};
