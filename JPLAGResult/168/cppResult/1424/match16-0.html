class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int ans = 0;
        queue<int> bfs;
        int n = status.size();
        vector<bool> found(n);
        vector<bool> explored(n);
        for (int i : initialBoxes) {
            found[i] = true;
            if (status[i] == 1) {
                bfs.push(i);
                explored[i] = true;                
            }
        }
       
        while (!bfs.empty()) {
            int f = bfs.front();
            bfs.pop();
            ans += candies[f];
            for (int i : keys[f])
                status[i] = 1;
            for (int i : containedBoxes[f]) {
                found[i] = true;
            }
            for (int i : keys[f])
                if (status[i] == 1 && found[i] && !explored[i]) {
                    bfs.push(i);
                    explored[i] = true;
                }
            for (int i : containedBoxes[f])
                if (status[i] == 1 && found[i] && !explored[i]) {
                    bfs.push(i);
                    explored[i] = true;
                }
        }
        return ans;
    }
};
