class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int ans = 0;
        
        queue<int> Q;
        for (int b : initialBoxes) {
            status[b] |= 2;
            if (status[b] == 3) {
                Q.push(b);
                status[b] = -1;
            }
        }
        
        while (!Q.empty()) {
            int p = Q.front(); Q.pop();
            ans += candies[p];
            for (int b : containedBoxes[p]) {
                status[b] |= 2;
                if (status[b] == 3) { Q.push(b); status[b] = -1; }
            }
            for (int b : keys[p]) {
                status[b] |= 1;
                if (status[b] == 3) { Q.push(b); status[b] = -1; }
            }
        }
        
        return ans;
    }
};
