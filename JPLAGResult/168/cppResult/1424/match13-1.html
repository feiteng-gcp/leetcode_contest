class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int n = static_cast<int>(status.size());
        queue<int> q;
        vector<bool> vKey(n), vCon(n);
        auto take = [&candies](int index) {
            int num = candies[index];
            candies[index] = 0;
            return num;
        };

        for (auto box : initialBoxes) {
            q.push(box);
            vCon[box] = true;
        }
        int total = 0;
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            if (candies[u] && (status[u] || vKey[u])) {

                total += take(u);
                for (auto con : containedBoxes[u]) {
                    q.push(con);
                    vCon[con] = true;
                }
                for (auto key : keys[u]) {
                    vKey[key] = true;
                    if (vCon[key]) {
                        q.push(key);
                    }
                }
            }
        }
        return total;
    }
};
