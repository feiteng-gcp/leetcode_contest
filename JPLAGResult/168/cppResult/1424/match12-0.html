class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int ret = 0;
        //queue<int> owned;
        vector<bool> owned(1024, false);
        vector<bool> keyed(1024, false);
        vector<bool> opened(1024, false);
        queue<int> open;
        for (auto &i: initialBoxes) {
            owned[i] = true;
            if (status[i] == 1) {open.push(i); opened[i] = true;}
        }
        while(!open.empty()) {
            int a = open.front();
            open.pop();
            ret += candies[a];
            for (auto &i: containedBoxes[a]) {
                owned[i] = true;
                if (keyed[i] == true || status[i] == true) {open.push(i); opened[i] = true;}
            } 
            for (auto &i: keys[a]) {
                keyed[i] = true;
                if (owned[i] == true && opened[i] == false) {open.push(i); opened[i] = true;}
            } 
            
        }
        return ret;
        
    }
};
