class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        set<int> keysSet;
        set<int> boxes;
        
        set<int> openedBoxes;
        int ans = 0;
        
        for(int box : initialBoxes) {
            boxes.insert(box);
        }
        
        while(!boxes.empty()) {
            bool flag = true;
            set<int> next;
            int size = boxes.size();
            for(int box : boxes) {
                if(openedBoxes.count(box) > 0) {
                    continue;
                }

                if(status[box] == 1 || keysSet.count(box) > 0) {
                    ans += candies[box];
                    for(int key : keys[box]) {
                        keysSet.insert(key);
                    }
                    for(int containedBox : containedBoxes[box]) {
                        next.insert(containedBox);
                    }
                    flag = false;
                    openedBoxes.insert(box);
                } else {
                    next.insert(box);
                }
            }
            
            if(flag) {
                break;
            }
            
            boxes = next;
        }
        
        return ans;
        
    }
};
