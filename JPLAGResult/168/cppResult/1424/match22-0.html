class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int n = status.size();
        vector<bool> boxed(n, true);
        vector<bool> empty(n, false);
        int totalCandies = 0;
        queue<int> queue;
        for (int i : initialBoxes) {
            boxed[i] = false;
            if (status[i]) queue.push(i);
        }
        while (queue.size() > 0) {
            int box = queue.front(); queue.pop();
            if (empty[box]) continue;
            empty[box] = true;
            totalCandies += candies[box];
            for (int key : keys[box]) {
                status[key] = 1;
                if (!boxed[key]) queue.push(key);
            }
            for (int b : containedBoxes[box]) {
                boxed[b] = false;
                if (status[b]) queue.push(b);
            }
        }
        return totalCandies;
    }
};
