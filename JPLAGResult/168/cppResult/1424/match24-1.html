class Solution {
public:
    int maxCandies(vector<int> &status, vector<int> &candies, vector<vector<int>> &keys, vector<vector<int>> &containedBoxes, vector<int> &initialBoxes) {
        int ans = 0;
        queue<int> boxes, staging;
        unordered_set<int> k;
        
        for (int b : initialBoxes) boxes.push(b);
        
        while (!boxes.empty()) {
            int b = boxes.front();
            boxes.pop();
            
            if (status[b] || k.count(b)) {
                ans += candies[b];
                for (int key : keys[b]) k.insert(key);
                for (int contained : containedBoxes[b]) boxes.push(contained);
            } else {
                staging.push(b);
            }
            
            if (boxes.empty()) {

                int n = staging.size();
                for (int i = 0; i < n; i++) {
                    int s = staging.front();
                    staging.pop();
                    if (k.count(s)) {
                        boxes.push(s);
                    } else {
                        staging.push(s);
                    }
                }
            }
        }
        
        return ans;
    }
    
    
};
