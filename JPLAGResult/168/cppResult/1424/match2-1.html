class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int n = status.size();
        if(n==0) return 0;
        
        queue<int> q;
        for(int b : initialBoxes) {
            q.push(b);
        }
        
        vector<bool> unsolved(n, false);
        
        int ret = 0;

        while(!q.empty()) {
            int b = q.front();
            q.pop();
            ret += candies[b];
            for(int k : keys[b]) {
                status[k] = 1;
                if(unsolved[k]) {
                    q.push(k);
                    unsolved[k] = false;
                }
            }
            
            for(int a : containedBoxes[b]) {
                if(status[a] == 1) {
                    q.push(a);
                } else {
                    unsolved[a] = true;
                }
            }
        }
        return ret;
    }
};
