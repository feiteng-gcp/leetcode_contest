class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int n = status.size();
        queue<int> q;
        vector<int> indeg(n, 0);
        for(int i = 0; i < n; i++){
            for(int j : containedBoxes[i])  indeg[j]++;
        }

        vector<bool> there(n, 0);
        int ret = 0;
        vector<bool> done(n, 0);
        for(int i : initialBoxes){
            there[i] = true;
            if(indeg[i] > 0 || status[i] == 0)    continue;
            q.push(i), done[i] = true;
        }

        while(!q.empty()){
            int x = q.front();  q.pop();
            ret += candies[x];
            for(int j : containedBoxes[x]){
                indeg[j]--;
                if(indeg[j] == 0)   there[j] = true;
            }
            for(int j : keys[x]){
                status[j] = 1;
            }
            for(int j : containedBoxes[x]){

                if(indeg[j] == 0 && there[j] && status[j] == 1 && !done[j]){
                    done[j] = true;
                    q.push(j);
                }
            }
            for(int j : keys[x]){
                if(indeg[j] == 0 && there[j] && status[j] == 1 && !done[j]){
                    done[j] = true;
                    q.push(j);
                }
            }
        }
        
        return ret;
    }
};
