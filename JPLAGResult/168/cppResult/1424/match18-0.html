class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int numCand = 0;
        queue<int> q;
        unordered_set<int> gainedKeys, gainedBoxes;
        for(auto& box: initialBoxes){
            if(status[box] == 1)
                q.push(box);
            else
                gainedBoxes.insert(box);
        }
        
        while(!q.empty()){
            int curBox = q.front();
            q.pop();
            numCand += candies[curBox];
            for(auto& key: keys[curBox]){
                gainedKeys.insert(key);
                if(gainedBoxes.count(key)){
                    q.push(key);
                    gainedKeys.erase(key);
                    gainedBoxes.erase(key);
                }
            }
            
            for(auto& box: containedBoxes[curBox]){
                gainedBoxes.insert(box);
                if(gainedKeys.count(box) || status[box] == 1){
                    q.push(box);
                    gainedKeys.erase(box);
                    gainedBoxes.erase(box);
                }
            }
        }
        return numCand;
    }
};
