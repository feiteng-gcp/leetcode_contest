bool got[1005];
bool open[1005];
bool vis[1005];


class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) 
    {
        int ret=0;
        memset(got, false, sizeof got);
        memset(open,false, sizeof open);
        memset(vis,false,sizeof vis);
        
        int n=status.size();
        for(int i=0;i<n;i++)
        {
            if(status[i]==1)
            {
                open[i]=true;
            }
        }
        for(int i: initialBoxes) got[i]=true;
        
        bool f=true;
        while(f)
        {
            f=false;
            for(int i=0;i<n;i++)
            {
                if(!vis[i]&&got[i]&&open[i])
                {
                    f=true;

                    vis[i]=true;
                    ret+=candies[i];
                    for(int b:containedBoxes[i])
                    {
                        got[b]=true;
                    }
                    for(int k:keys[i])
                    {
                        open[k]=true;
                    }
                }
            }
        }
        
        return ret;
    }
};
