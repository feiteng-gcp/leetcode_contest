class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int n = status.size();
        int ret = 0;
        vector<bool> visited(n), found(n);
        queue<int> Q;
        for (auto& x : initialBoxes) {
            found[x] = true;
            if (status[x]) {
                Q.push(x);
                visited[x] = true;
            }
        }
        while (!Q.empty()) {
            int x = Q.front();
            Q.pop();
            ret += candies[x];
            for (auto& y : keys[x]) {
                status[y] = 1;
                if (!visited[y] && found[y]) {
                    visited[y] = true;
                    Q.push(y);
                }
            }   
            for (auto& y : containedBoxes[x]) {
                found[y] = true;
                if (!visited[y] && status[y]) {
                    visited[y] = true;
                    Q.push(y);
                }
            }
        }
        return ret;
    }
};
