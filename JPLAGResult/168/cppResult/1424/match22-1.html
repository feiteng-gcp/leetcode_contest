class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int total = 0;
        bool visited[status.size()], unlocked[status.size()];
        memset(visited, false, sizeof(visited));
        memset(unlocked, false, sizeof(unlocked));

        
        stack<int> s;
        for(int x : initialBoxes)
        {
            s.push(x);
            unlocked[x] = true;

        }
        
        while(!s.empty())
        {
            int x = s.top();
            s.pop();
            if(status[x] == 0 || visited[x] || !unlocked[x])continue;
            visited[x] = true;
            total += candies[x];
            for(int y : keys[x])
            {
                status[y] = 1;
                s.push(y);
            }
            for(int y : containedBoxes[x])
            {
                s.push(y);
                unlocked[y] = true;
            }               
        }
        
        return total;
    }
};
