class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        queue<int> boxs;
        unordered_set<int> allkeys, st;
        
        for(int initbox : initialBoxes) boxs.push(initbox);
        int res = 0;
        
        while(boxs.size())
        {
            int box = boxs.front(); boxs.pop();
            
            if(st.count(box)) continue;
            st.insert(box);
            
            if(allkeys.count(box) || status[box] == 1)
            {
                res += candies[box];
                for(int key : keys[box]) allkeys.insert(key);
                for(int containbox : containedBoxes[box]) boxs.push(containbox);
            }
        }
        
        
        for(int initbox : initialBoxes) boxs.push(initbox);
        st.clear();
        res = 0;
        
        while(boxs.size())
        {
            int box = boxs.front(); boxs.pop();
            
            if(st.count(box)) continue;
            st.insert(box);
            
            if(allkeys.count(box) || status[box] == 1)
            {
                res += candies[box];
                for(int key : keys[box]) allkeys.insert(key);
                for(int containbox : containedBoxes[box]) boxs.push(containbox);
            }
        }
        
        return res;
    }
};
