class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int N = status.size();
        vector<int> K(N, 0);
        vector<int> B(N, 0);
        queue<int> q;
        int res = 0;
        for (auto i : initialBoxes) {
            q.push(i);
        }
        unordered_set<int> s;
        while (!q.empty()) {
            int k = q.front();
            q.pop();

            res += candies[k];
            for (auto i : containedBoxes[k]) {
                B[i] = 1;
                s.insert(i);
            }
            for (auto i : keys[k]) {
                K[i] = 1;
            }
            for (int i = 0; i < N; ++i) {
                if (B[i] == 1 && (K[i] + status[i] > 0)) {
                    q.push(i);
                    B[i] = K[i] = 0;
                }
            }
        }
        return res;
    }
};
