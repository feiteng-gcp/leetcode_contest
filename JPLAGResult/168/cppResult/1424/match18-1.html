class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int res = 0;
        unordered_set<int> usk;
        unordered_set<int> usb;
        unordered_set<int> vb;
        vector<int> v1, v2;
        // v1 = initialBoxes;
        for (auto b : initialBoxes) {
            if (status[b] == 1) {
                vb.insert(b);
                v1.push_back(b);
            }
            usb.insert(b);
        }
        while (v1.size() > 0) {
            v2.clear();
            for (auto b : v1) {

                res += candies[b];
                for (auto k : keys[b]) {
                    usk.insert(k);
                    if (usb.find(k) != usb.end() && vb.find(k) == vb.end()) {
                        v2.push_back(k);
                        vb.insert(k);
                    }
                }
                for (auto next : containedBoxes[b]) {
                    if ((usk.find(next) != usk.end() || status[next] == 1) && vb.find(next) == vb.end()) {
                        v2.push_back(next);
                        vb.insert(next);
                    }
                    usb.insert(next);
                }
                
            }
            v1 = v2;
        }
        return res;
    }
};
