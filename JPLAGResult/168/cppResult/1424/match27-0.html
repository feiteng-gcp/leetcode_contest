class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int ans = 0;
        unordered_set<int> have, keyss;
        queue<int> q;
        for(int i : initialBoxes) {
            have.insert(i);
            if(status[i] == 1) {
                q.push(i);
                status[i] = 2;
            }
        }
        while(!q.empty()) {
            int cur = q.front();
            q.pop();
            ans += candies[cur];
            for(int i : containedBoxes[cur]) {
                have.insert(i);
                if(status[i] == 1) {
                    q.push(i);
                    status[i] = 2;
                }
                if(keyss.count(i) && status[i] == 0) {
                    q.push(i);
                    status[i] = 2;
                }
            }
            for(int i : keys[cur]) {
                keyss.insert(i);
                if(have.count(i) && status[i] == 0) {
                    q.push(i);
                    status[i] = 2;
                }
            }
        }
        return ans;
    }
};
