class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        set<int> lockedBoxes,availKeys;
        queue<int> boxesToOpen;
        for(int i=0;i<initialBoxes.size();i++) {
            if(status[initialBoxes[i]]==1)
                boxesToOpen.push(initialBoxes[i]);
            else {
                lockedBoxes.insert(initialBoxes[i]);
            }
        }
        int cnt=0;

        while(!boxesToOpen.empty()) {
            int tp = boxesToOpen.front();
            boxesToOpen.pop();
            cnt += candies[tp];
            for(int i=0;i<keys[tp].size();i++) {
                if(lockedBoxes.find(keys[tp][i])!=lockedBoxes.end()) {
                    lockedBoxes.erase(keys[tp][i]);
                    boxesToOpen.push(keys[tp][i]);
                } else {
                    availKeys.insert(keys[tp][i]);

                }
            }
            for(int i=0;i<containedBoxes[tp].size();i++) {
                if(status[containedBoxes[tp][i]]==1) {
                    boxesToOpen.push(containedBoxes[tp][i]);
                } else {
                    if(availKeys.find(containedBoxes[tp][i])!=availKeys.end()) {
                        availKeys.erase(containedBoxes[tp][i]);
                        boxesToOpen.push(containedBoxes[tp][i]);
                    } else {
                        lockedBoxes.insert(containedBoxes[tp][i]);
                    }
                }
            }
        }
        return cnt;
    }
};
