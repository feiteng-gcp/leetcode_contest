class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int n = status.size();
        int res = 0;
        queue<int> q;
        vector<int> has(n, 0), vis(n, 0);
        for (int i : initialBoxes) {
            has[i] = 1;
            if (status[i] && !vis[i]) q.push(i), vis[i] = 1;
        }
        while (q.size()) {
            int cur = q.front();
            q.pop();
            res += candies[cur];
            for (int i : keys[cur]) {
                status[i] = 1;
                if (has[i] && !vis[i]) q.push(i), vis[i] = 1;
            }
            for (int i : containedBoxes[cur]) {
                has[i] = 1;
                if (status[i] && !vis[i]) q.push(i), vis[i] = 1;
            }
        }
        return res;
    }
};
