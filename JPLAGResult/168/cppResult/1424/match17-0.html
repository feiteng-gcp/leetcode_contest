class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int cnt=0;
        queue<int>q;
        set<int>box;
        for(int tmp:initialBoxes)
        {
            if(status[tmp])
                 q.push(tmp);
            else
                box.insert(tmp);
        }
        while(!q.empty())
        {
            int u=q.front();
            q.pop();
            cnt+=candies[u];
            for(int tmp:keys[u])
            {
                status[tmp]=1;
                if(box.find(tmp)!=box.end())
                {
                    q.push(tmp);
                    box.erase(tmp);
                }
            }
            for(int tmp:containedBoxes[u])
            {
                if(status[tmp])
                    q.push(tmp);
                else
                    box.insert(tmp);
            }
        }
        return cnt;
    }
};
