class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int N=status.size();
        vector<bool> findKey(N,false);
        queue<int> q;
        int res=0;
        for(int i:initialBoxes){
            q.push(i);
        }
        vector<bool> visited(N,false);
        unordered_set<int> wait;
        while(!q.empty()){
            int cur=q.front();
            q.pop();
            if(status[cur] || findKey[cur]){
                
                res+=candies[cur];
                vector<int> contains=containedBoxes[cur];
                for(auto i:contains){
                    q.push(i);
                }
                for(auto i:keys[cur]){
                    findKey[i]=true;
                    if(wait.find(i)!=wait.end()){
                        q.push(i);
                        wait.erase(i);
                    }
                }
            }else{
                wait.insert(cur);
            }
        }
        for(int i:wait){
            if(findKey[i]) res+=candies[i];
        }
        return res;
    }
};
