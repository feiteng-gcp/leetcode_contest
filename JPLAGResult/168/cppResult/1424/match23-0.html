class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        unordered_set<int> curKey;
        int res = 0;
        while (initialBoxes.size() != 0) {
            int opened = 0;
            vector<int> nxtBoxes;
            for (int i = 0; i < initialBoxes.size(); i++) {
                int curBox = initialBoxes[i];
                if (status[curBox] == 1 || curKey.find(curBox) != curKey.end()) {
                    res += candies[curBox];
                    for (int child : containedBoxes[curBox]) {
                        nxtBoxes.push_back(child);
                    }
                    // curKey.insert(keys[curBox]);
                    for (int keyss : keys[curBox]) {
                        curKey.insert(keyss);
                    }
                    opened++;
                } else {
                    nxtBoxes.push_back(curBox);
                }
            }
            if (opened == 0) return res;
            initialBoxes = nxtBoxes;
        }
        return res;
    }
};
