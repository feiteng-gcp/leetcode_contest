class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int res = 0;
        int n = status.size();
        set<int> ready, remain;
        vector<int> picked(n, 0), hasKey(n, 0);
        
        for (int i : initialBoxes) {
            if (status[i]) {
                ready.insert(i);
            } else {
                remain.insert(i);
            }
        }
        while (!ready.empty()) {
            int x = *ready.begin();
            ready.erase(x);
            res += candies[x];
            picked[x] = 1;
            for (int k : keys[x]) {
                hasKey[k] = 1;
                if (remain.find(k) != remain.end() && ready.find(k) == ready.end()) {
                    remain.erase(k);
                    ready.insert(k);
                }
            }
            for (int b : containedBoxes[x]) {
                if (status[b] || hasKey[b]) {
                    ready.insert(b);
                } else {
                    remain.insert(b);
                }
            }
        }
        return res;
    }
};
