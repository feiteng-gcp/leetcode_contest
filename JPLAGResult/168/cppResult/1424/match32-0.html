class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        map<int, set<int>> m;
        for(auto it:initialBoxes) {
            if(status[it]==1) {
                if(m.find(1)==m.end())
                    m.insert(make_pair(1, set<int>()));
                m[1].insert(it);
            } else {
                if(m.find(2)==m.end())
                    m.insert(make_pair(2, set<int>()));
                m[2].insert(it);
            }
        }
        for(int i=0;i<status.size();i++) {
            if(m[1].find(i)==m[1].end()&&m[2].find(i)==m[2].end()) {
                if(status[i]==1) {
                    if(m.find(3)==m.end())
                        m.insert(make_pair(3, set<int>()));
                    m[3].insert(i);
                } else {
                    if(m.find(4)==m.end())
                        m.insert(make_pair(4, set<int>()));
                    m[4].insert(i);
                }
            }
        }
        int out=0;
        set<int> k;
        while(m[1].size()!=0) {
            int b = *m[1].begin();
            
            m[1].erase(b);
            out+=candies[b];
            for(int i=0;i<containedBoxes[b].size();i++) {
                if(status[containedBoxes[b][i]]==1) {
                    if(m[3].find(containedBoxes[b][i])!=m[3].end()) {
                        m[3].erase(containedBoxes[b][i]);
                        m[1].insert(containedBoxes[b][i]);
                    }
                } else {
                    if(m[4].find(containedBoxes[b][i])!=m[3].end()) {
                        m[4].erase(containedBoxes[b][i]);
                        if(k.find(containedBoxes[b][i])!=k.end()) {
                            m[1].insert(containedBoxes[b][i]);
                            k.erase(containedBoxes[b][i]);
                        } else {
                            m[2].insert(containedBoxes[b][i]);
                        }
                    }}
                
            }
                
            for(int i=0;i<keys[b].size();i++) {
                if(m[2].find(keys[b][i])!=m[2].end()) {
                    m[2].erase(keys[b][i]);
                    m[1].insert(keys[b][i]);
                } else {
                    k.insert(keys[b][i]);
                }
            }
        }
        return out;
    }
};
