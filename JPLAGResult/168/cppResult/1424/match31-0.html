class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int res =0;
        unordered_set<int> si_key;
        unordered_set<int> si_box;
        unordered_set<int> si_visited;
        for(int i: initialBoxes){
            si_box.insert(i);
            for(int j: keys[i]){
                si_key.insert(j);
            }
        }
        while(1){
            int cnt =0;
            for(int i: si_box){
                if(si_visited.count(i)) continue;
                if(status[i]==1||si_key.count(i)){
                    cnt++;
                    res+=candies[i];
                    si_visited.insert(i);
                    for(int j: containedBoxes[i]){
                        if(si_visited.count(j)!=0) continue;
                        si_box.insert(j);
                    }
                    for(int j: keys[i]){
                        if(si_visited.count(j)!=0) continue;
                        si_key.insert(j);
                    }
                }
            }
            if(cnt ==0) break;
        }
        return res;
    }
};
