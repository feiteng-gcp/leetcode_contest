class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        vector<int> retrieved(status.size());
        vector<int> have_keys(status.size());
        vector<int> is_out(status.size());
        queue<int> q;
        for(auto c:initialBoxes) {
            is_out[c] = 1;
            if(status[c]) {
                q.push(c);
                retrieved[c] = 1;

            }
        }
        int res = 0;

        while(!q.empty()) {
            int cur = q.front();
            q.pop();
            res+=candies[cur];
            for(auto c:containedBoxes[cur]) {
                is_out[c] = 1;
                if((have_keys[c] || status[c]) && retrieved[c]==0) {
                    q.push(c);
                    retrieved[c] = 1;
                }
            }
            for(auto c:keys[cur]) {
                have_keys[c] = 1;
                if(is_out[c] && retrieved[c]==0) {
                    status[c] = 1;
                    q.push(c);
                    retrieved[c] = 1;
                }
            }

        }
        return res;

    }
};
