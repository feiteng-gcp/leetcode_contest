class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int n = status.size();

        bool inhand[n];
        memset(inhand, false, sizeof(inhand));
        for (int a : initialBoxes) {
            inhand[a] = true;
        }

        bool v[n];
        memset(v, false, sizeof(v));
        
        queue<int> q;
        for (int a : initialBoxes) {
            if (status[a]) {
                v[a] = true;
                q.push(a);
            }
        }

        int ret = 0;
        while (!q.empty()) {
            int x = q.front();
            q.pop();


            v[x] = true;
            ret += candies[x];

            for (int y : keys[x]) {
                status[y] = 1;
                if (!v[y] && inhand[y]) {
                    v[y] = true;
                    q.push(y);
                }
            }

            for (int y : containedBoxes[x]) {
                inhand[y] = 1;
                if (!v[y] && status[y]) {
                    v[y] = true;
                    q.push(y);
                }
            }
        }

        return ret;
    }
};
