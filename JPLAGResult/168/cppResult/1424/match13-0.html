class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int size = status.size();
        vector<bool> opened(size, false);
        vector<bool> has_keys(size, false);
        vector<bool> found(size, false);
        
        queue<int> q;
        for (auto i : initialBoxes) {
            q.push(i);
            found[i] = true;
        }
        int res = 0;
        while (!q.empty()) {
            int f = q.front();
            q.pop();
            if (!has_keys[f] && status[f] == 0) {
                continue;
            }
            if (status[f] == 1 || has_keys[f]) {
                if (opened[f]) continue;
                res += candies[f];
                opened[f] = true;
            }
            
            vector<int>& box = containedBoxes[f];
            vector<int>& key = keys[f];
            for (auto next: box) {
                q.push(next);
                found[next] = true;
            }
            for (auto next: key) {
                has_keys[next] = true;
                if (found[next]) {
                    q.push(next);
                }
            }
            
        }
        return res;
        
    }
};
