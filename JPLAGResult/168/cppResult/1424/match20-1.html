class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int ans = 0;
        unordered_set<int> nokeyset;
        unordered_set<int> noboxset;
        queue<int> waitopen;
        for(auto i : initialBoxes){
            waitopen.push(i);
        }
        while(!waitopen.empty()){
            int tmp = waitopen.front();
            waitopen.pop();
            if(status[tmp] == 1){

                ans += candies[tmp];
                for(auto i : containedBoxes[tmp]){
                    waitopen.push(i);
                }
                for(auto i : keys[tmp]){
                    status[i] = 1;
                    if(nokeyset.find(i) != nokeyset.end()){
                        nokeyset.erase(i);
                        waitopen.push(i);
                    }
                }
            } else {
                nokeyset.insert(tmp);
            }
        }
        return ans;
    }
};
