class Solution {
public:
    int maxCandies(vector<int>& status, vector<int>& candies, vector<vector<int>>& keys, vector<vector<int>>& containedBoxes, vector<int>& initialBoxes) {
        int n = status.size(), res = 0, i, tmp;
        unordered_set<int> locked_set, key_set, visited_boxes;
        queue<int> q;
        for (auto b:initialBoxes) {
            if (status[b] == 1) {
                q.push(b);
                visited_boxes.insert(b);
            }
            else locked_set.insert(b);
        }
        while (!q.empty()) {
            tmp = q.front();
            q.pop();
            res += candies[tmp];
            // cout << candies[tmp] << " " << res << endl;
            for (auto key_ind:keys[tmp]) {
                key_set.insert(key_ind);
                if (locked_set.count(key_ind) && !visited_boxes.count(key_ind)) {
                    q.push(key_ind); 
                    visited_boxes.insert(key_ind);
                }
            }
            for (auto b:containedBoxes[tmp]) {
                if (status[b] == 1 || key_set.count(b)) {
                    q.push(b);
                    visited_boxes.insert(b);
                }
                else locked_set.insert(b);
            }
        }
        return res;
    }
};
