class Solution {
public:
  bool isPossibleDivide(vector<int>& nums, int k) {
    if (nums.size() % k != 0) return false;
    map<int, int> M;
    for (auto n: nums) M[n]++;
    while (M.size()) {
      int t = M.begin()->first;
      for (int i = t; i < t + k; i++) {
        if (M.count(i) == 0) return false;
        if (--M[i] == 0) M.erase(i);
      }
    }
    return true;
  }
};
