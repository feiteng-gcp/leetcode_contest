class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        map<int, int> m;
        vector<int> vec;
        for (auto a: nums) {
            m[a]++;
            vec.push_back(a);
        }
        sort(vec.begin(), vec.end());
        for (auto a:vec) {
            if (!m[a]) continue;
            for (int i = 0; i < k; i++) {
                if (!m[a+i]) return false;
                m[a+i]--;
            }
        }
        return true;
    }
};
