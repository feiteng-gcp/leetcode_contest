class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {

        int n=nums.size();
        if (n%k != 0)
            return false;
        
        
        map<int,int> counts;
        for(int x : nums) {
            counts[x]++;
        }
        
        while(!counts.empty()) {
            
            int a = counts.begin()->first;
            
            for(int i=a;i<=a+k-1;i++) {
                if (counts.find(i) == counts.end())
                    return false;
                counts[i]--;
                if (counts[i]==0)
                    counts.erase(i);
            }
        }
        return true;
    }
};
