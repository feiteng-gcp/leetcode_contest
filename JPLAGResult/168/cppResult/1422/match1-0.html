class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        if(nums.size()%k)
            return false;
        map<int,int> app;
        for(auto c:nums)
            app[c]+=1;
        int total = nums.size()/k;
        for(int i=0; i<total; i++) {
            int cur = app.begin()->first;
            for(int j=0; j<k; j++) {
                if(app.count(j+cur)==0) {
                    cout<<j<<" "<<cur<<endl;
                    return false;
                }
                app[cur+j]--;
                if(app[cur+j]==0)
                    app.erase(cur+j);
            }
        }
        return true;
    }
};
