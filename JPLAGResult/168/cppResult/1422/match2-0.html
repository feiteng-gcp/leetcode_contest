class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        int n = nums.size();
        if(n % k) return false;
        
        vector<pair<int, int>> numst;
        for(int i : nums) numst.push_back({i, 1});
        
        sort(numst.begin(), numst.end());
        
        for(int i = 0; i < numst.size(); i++)
        {
            if(numst[i].second == 0) continue;
            
            int pre = numst[i].first;
            cout << pre << " ->";
            for(int j = 1; j < k; j++)
            {
                int num = pre + 1;

                auto it = lower_bound(numst.begin() + i, numst.end(), make_pair(num, 1));

                if(it == numst.end() || it->second == 0) return false;
                if(it->first != num) return false;
                cout << it->first << ' ';
                it->second = 0;
                pre = num;
            }
            cout << endl;
        }
        return true;
    }
};
