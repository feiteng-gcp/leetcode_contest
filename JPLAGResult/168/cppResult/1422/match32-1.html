class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        map<int, int> m;
        for(auto v: nums){
            m[v]++;
        }
        
        while(m.size() > 0){
            if(m.size() < k){
                return false;
            }
            auto it = m.begin();

            int val = it->first;
            for(int i = 0; i < k; i++){
                if(it->first != val){
                    return false;
                }
                it->second--;
                val++;
                if(it->second == 0){
                    m.erase(it++);
                }else{
                    it++;
                }
            }
            
        }
        
        return true;
    }
};
