class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        int n = nums.size();
        if(n % k) return false;
        sort(nums.begin(), nums.end());
        unordered_map<int,int> store;
        for(auto a : nums)
            store[a]++;
        
        for(auto &a : nums)
        {
            if(store[a] == 0) continue;

            for(int i = 0; i < k; ++i)
            {
                //cout << a + k << endl;
                if(store[a + i] == 0) return false;
                store[a + i]--;
            }
        }
        return true;
    }
};
