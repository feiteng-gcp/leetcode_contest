class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        if (nums.size() % k != 0) {
            return false;
        }

        map<int, int> cnt;
        for (int a : nums) {
            cnt[a]++;
        }

        while (!cnt.empty()) {

            int x = cnt.begin()->first;
            for (int i = 0; i < k; i++) {
                if (!cnt.count(x)) {
                    return false;
                }
                cnt[x]--;
                if (cnt[x] == 0) {
                    cnt.erase(x);
                }
                x++;
            }
        }

        return true;
    }
};
