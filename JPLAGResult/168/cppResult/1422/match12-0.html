class Solution {
public:
    bool isPossibleDivide(vector<int>& a, int k) {
        multiset<int> s(a.begin(), a.end());
        if (a.size() % k != 0) return false;
        while(!s.empty()) {
            int val = *s.begin();
            for(int i=0; i<k; ++i) {
                if (s.find(val+i) == s.end()) return false;
                s.erase(s.find(val+i));
            }
        }
        return true;
    }
};
