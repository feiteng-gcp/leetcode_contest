class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        if(nums.size() % k != 0)
            return false;
        sort(nums.begin(),nums.end());
        int i = 0;
        multiset <int> m(nums.begin(),nums.end());
        while(!m.empty()){
            int t = *m.begin();
            for(int nk = 0; nk < k; ++nk){
                auto tmp = m.find(t);
                if(tmp == m.end())
                    return false;
                m.erase(tmp);
                t = t + 1;
            }
        }
        
        return m.empty();
    }
};
