class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        if((nums.size() % k) || nums.size() < k )return false;

        map<int, int> counter;
        for(int x : nums)
            ++counter[x];
        
        while(!counter.empty())
        {
            auto it = counter.begin();
            int x = it->first;
            for(int i = 0; i < k; ++i)
            {
                if(counter[x+i] == 0)return false;
                --counter[x+i];
                if(counter[x+i] == 0)
                    counter.erase(x+i);
            }
        }
        
        return true;
    }
};
