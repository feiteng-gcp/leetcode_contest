class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        map<int,int> m;
        for(int i : nums){
            m[i]++;
        }
        while(!m.empty()){
            int x = m.begin()->first;
           // if(x==3)return true;
            for(int j = x ; j < x + k ; j++){
                m[j]--;
                if(m[j]<0)return false;
                if(m[j]==0)m.erase(j);
            }
            
        }
        return true;
    }
};
