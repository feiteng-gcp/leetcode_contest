class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        if (nums.size() % k != 0) {
            return false;
        }
        map<int, int> m;
        for (int i = 0; i < nums.size(); i++) {
            m[nums[i]]++;
        }
        for(int i = 0; i < nums.size()/k; i++) {
            int start = m.begin()->first;
            for (int j = start; j < start + k; j++) {
                //cout << j << "\n";
                if (m[j] == 0) {
                    return false;
                }
                m[j]--;
                if (m[j] == 0) {
                    m.erase(j);
                }
            }
        }
        return true;
    }
};
