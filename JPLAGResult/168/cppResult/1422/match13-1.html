class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        if(0 != (nums.size() % k)) return false;
        if(1 == k) return true;

        unordered_map<int, int> um;
        sort(nums.begin(), nums.end());
        for(int i = 0; i < nums.size(); i ++) {
            um[nums[i]] ++;
        }
        for(int i = 0; i < nums.size(); i ++) {
            int one = nums[i];
            if(0 == um[one]) continue;
            for(int x = one; x < one + k; x ++) {
                if(0 == um[x]) return false;
                um[x] --;
            }
        }
        return true;
    }
};
