class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        if (nums.size() % k != 0) return false;

        multiset<int> s(nums.begin(), nums.end());
        while(not s.empty()) {
            int a = *s.begin();
            for (int i = 0; i < k; ++i) {
                if (s.find(a + i) == s.end())
                    return false;
                s.erase(s.find(a + i));
            }
        }
        return true;
    }
};
