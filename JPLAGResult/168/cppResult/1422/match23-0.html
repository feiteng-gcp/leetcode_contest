class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        int len = nums.size();
        if (len % k != 0) return false;
        map<int, int> cnt;
        for (int i : nums) {
            cnt[i]++;
        }
        while (cnt.size() > 0) {
            int first = cnt.begin()->first;
            for (int i = 0; i < k; i++) {
                if (cnt.find(first + i) == cnt.end()) {
                    return false;
                }
                if (--cnt[first + i] == 0) {
                    cnt.erase(first + i);
                }
            }
        }
        return true;
    }
};
