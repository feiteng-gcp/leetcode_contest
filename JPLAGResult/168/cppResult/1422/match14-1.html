class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        if (nums.size() % k)
            return false;
        map<int, int> mem;
        for (auto n: nums)
            mem[n] ++;
        while (!mem.empty()) {
            int val = mem.begin()->first;
            for (int i = val; i < val+k; i++) {
                if (mem.count(i) == 0)
                    return false;
                mem[i] --;
                if (mem[i] == 0)
                    mem.erase(i);
            }
        }
        return true;
    }
};
