class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        int n = nums.size();
        if (n % k) return false;
        
        map<int, int> m;
        for (auto it: nums) {
            m[it]++;    
        }
        
        sort(nums.begin(), nums.end());
        for (auto it: nums) {
            if (!m[it]) continue;
            // get first num

            int f = it;
            for (int i = f; i < f + k; ++i) {
                if (!m[i]) return false;
                m[i]--;
            }
        }
        return true;
    }
};
