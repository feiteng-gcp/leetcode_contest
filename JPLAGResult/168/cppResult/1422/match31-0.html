class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        map<int, int> m;
        for (auto i : nums) {
            m[i]++;
        }
        while (!m.empty()) {
            auto it = m.begin();
            int f = it->first;
            //cout << "m.size()=" << m.size() << " f=" << f << endl;
            for (int i=1; i < k; ++i) {
                if (!m.count(f+i)) {
                    //cout << "first=" << f << " n=" << f+i << endl;
                    return false;
                }
                m[f+i]--;
                if (m[f+i] == 0) m.erase(f+i);
            }
            m[f]--;
            if (m[f] == 0) m.erase(f);
        }
        return true;
        
    }
};
