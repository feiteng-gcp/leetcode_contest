class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {

		int cnt = 0;
		map<int, int> freq;
		for (int num: nums)
			freq[num]++;
		for (auto it: freq) {
			if (it.second == 0)
				continue;

			for (int i = it.first + 1; i - it.first < k; i++) {
				if (freq[i] < it.second)
					return false;
				freq[i] -= it.second;
			}
		}
		return true;
    }
};
