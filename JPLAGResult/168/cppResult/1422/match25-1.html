class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {

        multiset<int> remaining(nums.begin(), nums.end());
        
        while (!remaining.empty()) {
            int start = *remaining.begin();
            for (int i = 0; i < k; ++i) {
                auto it = remaining.find(start);
                if (it == remaining.end()) return false;
                remaining.erase(it);
            //    cout << start << " " ;
                ++start;
            }
           // cout << endl;
        }
        
        return true;
    }
};
