class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        multiset<int> s(nums.begin(), nums.end());
        while (!s.empty()) {
            int ta = *s.begin();
            s.erase(s.begin());
            for (int i = 1; i < k; i++) {
                if (s.empty())
                    return 0;
                auto it = s.find(++ta);
                if (it == s.end())
                        return 0;
                s.erase(it);
            }
        }
        return 1;
    }
};
