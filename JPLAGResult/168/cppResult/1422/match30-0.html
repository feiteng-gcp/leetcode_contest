class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        map<int,int> m;
        for( auto & iter : nums ) {
            m[ iter ]++;
        }
        while( !m.empty() ) {
            int n = m.begin()->first;
            for( int i = 0; i < k; i++ ) {
                if( m.count( n ) <= 0 ) {
                    return false;
                }
                m[ n ]--;
                if( m[ n ] == 0 ) {
                    m.erase( n );
                }
                n++;
            }
        }
        return true;
    }
};
