class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        map<int, int> m;
        for(auto& e: nums) ++m[e];

        while (m.size()) {
            int x=begin(m)->first;
            for(int i=1; i<k; ++i) {
                if (m[x+i]<m[x]) return false;
                m[x+i]-=m[x];
                if (m[x+i]==0) m.erase(x+i);
            }
            m.erase(x);
        }
        return true;
    }
};
