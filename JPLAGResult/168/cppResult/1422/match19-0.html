class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        if (nums.size() % k != 0) {
            return false;
        }
        map<int, int> cnt;
        for (int n : nums) {
            cnt[n] ++;
        }
        int N = nums.size();
        while (N != 0) {
            auto it = cnt.begin();
            int v = it->first;
            for (int i=0;i<k;++i) {
                if (cnt.count(v+i) == 0) {
                    return false;
                }
                cnt[v+i] --;
                if (cnt[v+i] == 0) {
                    cnt.erase(v+i);
                }
            }
            N -= k;
        }
        return true;
    }
};
