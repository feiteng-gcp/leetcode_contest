class Solution {
public:
    bool isPossibleDivide(vector<int>& nums, int k) {
        map<int, int> count;
        for (auto& x: nums) {
            count[x]++;
        }
        while (count.size() > 0) {
            int x = count.begin()->first;
            for (int i = 0;i < k;i++) {
                if (count.find(x) == count.end()) {
                    return false;
                }
                count[x]--;
                if (count[x] == 0) {
                    count.erase(count.find(x));
                }
                x++;
            }
        }
        return true;
    }
};
