class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        unordered_map<string, int> m;
        for(int i =0 ;i <s.size(); i++){
            string curr = "";
            set<char> in;
            for(int j =i, c = 0; j< s.size() && c < maxSize; c++,j++){
                curr += s[j];
                in.insert(s[j]);

                if(curr.size() >= minSize && curr.size() <= maxSize && in.size() <= maxLetters) m[curr]++;
            }
        }
        int mx = 0;
        for(auto &it : m){
            mx = max(mx, it.second);
        }
        return mx;
        
    }
};
