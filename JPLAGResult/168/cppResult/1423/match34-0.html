class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        int n=s.size();
        map<string, int> m;
        for(int i=0; i<n; ++i) {
            unordered_map<char, int> a;
            bool ok=true;
            string t;
            for(int j=0; i+j<n and j<minSize; ++j) {
                ++a[s[i+j]];
                if (a.size()>maxLetters) ok=false;
                t+=s[i+j];
            }
            if (ok and t.size()==minSize) m[t]++;
        }
        int res=0;
        for(auto& e: m) {
            res=max(res, e.second);
        }
        return res;
    }
};
