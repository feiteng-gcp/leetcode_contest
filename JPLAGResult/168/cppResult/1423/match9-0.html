class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        int l = s.size();
        unordered_map<string, int> r;
        for (int i = minSize; i <= maxSize; ++i) {
            vector<int> st(26, 0);
            for (int j = 0; j < i; ++j) {
                st[s[j] - 'a']++;
            }
            for (int j = i - 1; j < l; ++j) {
                if (j != i - 1) {
                    st[s[j-i] - 'a']--;
                    st[s[j] - 'a']++;
                }
                int count = 0;
                for (auto &k : st) {
                    if (k > 0) ++count;
                }
                if (count > maxLetters)
                    continue;
                string ss = s.substr(j - i + 1, i);
                if (r.find(ss) == r.end()) {
                    r[ss] = 1;
                } else {
                    r[ss]++;
                }
            }
        }
        int result = 0;
        for (auto &p : r) {
            result = max(result, p.second);
        }
        return result;
    }
};
