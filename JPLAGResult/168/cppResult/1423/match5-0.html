class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        int ans = 0;
        for(int len = minSize; len <= maxSize; ++len){
            unordered_map<string, int> strcnt;
            vector<int> lettercnt(26, 0);
            int cnt = 0;
            if(len >= s.size()) return ans;
            for(int i = 0; i < len; ++i){
                if(lettercnt[s[i]-'a']++ == 0) cnt++;
            }
            if(cnt <= maxLetters){
                strcnt[s.substr(0, len)]++;
                ans = max(ans, strcnt[s.substr(0, len)]);
            }
            for(int i = 1; i <= s.size() - len; ++i){
                if(lettercnt[s[i-1]-'a']-- == 1) cnt--;
                if(lettercnt[s[i+len-1]-'a']++ == 0) cnt++;
                if(cnt <= maxLetters){
                    strcnt[s.substr(i, len)]++;
                    ans = max(ans, strcnt[s.substr(i, len)]);
                }
            }
        }
        return ans;
    }
};
