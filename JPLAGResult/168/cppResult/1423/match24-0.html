class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
            
        map<string,int> M;
        int res=0;
        for(int i=0;i+minSize-1<s.length();i++){
            string sub =s.substr(i,minSize);
            set<char> Set;
            for(char c:sub)
                Set.insert(c);
            if(Set.size()<=maxLetters)
            {
                if(++M[sub]>res)
                    res++;
            }

        }
        return res;
            
        
    }
};
