typedef long long int64;
class Solution {
 public:
  int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
    int n = s.size();
    int best = 0;
    maxSize = minSize;
    for (int len = minSize; len <= maxSize; ++len) {
      //trace(len);
      vector<int> cnt(26);
      map<string, int> frequent;
      int sz = 0;
      for (int i = 0; i < len; ++i) {
        if (++cnt[s[i] - 'a'] == 1) sz++;
      }
      if (sz <= maxLetters) frequent[s.substr(0, len)]++;
      for (int i = len; i < n; ++i) {
        if (--cnt[s[i - len] - 'a'] == 0) sz--;
        if (++cnt[s[i] - 'a'] == 1) sz++;
        if (sz <= maxLetters) frequent[s.substr(i - len + 1, len)]++;
      }
      for (auto& it : frequent) {
        best = max(best, it.second);
      }
    }
    return best;
  }
};
