class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        unordered_map<string,int> cnt;
        for(int i=0;i<=s.size()-minSize;++i){
            string tmp=s.substr(i,minSize);
            if(isvalid(tmp,maxLetters)){
                cnt[tmp]++;
            }
        }
        int res=0;
        for(auto m:cnt){
            res=max(res,m.second);
        }
        return res;
    }
    
    bool isvalid(string &s, int maxLetters){
        vector<int> cnt(26,0);
        for(char c:s){
            cnt[c-'a']++;
        }
        int c=0;
        for(auto m:cnt){
            if(m>0) c++;
        }
        return c<=maxLetters;
    }
};
