class Solution {
public:
	int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
		unordered_map<string, int>ct;
		for (int i = 0; i < s.size(); i++) {
			if (i + minSize - 1 >= s.size())continue;
			string curr = s.substr(i, minSize);
			unordered_set<char>p(curr.begin(), curr.end());
			if (p.size() > maxLetters)continue;
			ct[curr]++;
			for (int j = i + minSize; j < s.size() && j - i + 1 <= maxSize;j++) {
				curr += s[j];
				p.insert(s[j]);
				if (p.size() > maxLetters)break;
				ct[curr]++;
			}
		}
		int maxx = 0;
		for (auto it : ct) {
			maxx = max(maxx, it.second);
		}
		return maxx;
	}
};
