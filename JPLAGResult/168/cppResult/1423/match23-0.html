class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        unordered_map<string, int> dp;
        int ls = s.size();
        
        for (int i = minSize-1; i < ls; i++) {
            int cnt[26] = {};
            int has = 0;
            
            for (int j = i, len = 1; j >= 0 && len <= maxSize; j--, len++) {
                if (cnt[s[j]-'a']++ == 0) has++;
                if (has > maxLetters) break;
                
                if (len >= minSize) {
                    dp[s.substr(j, len)]++;
                }
            }
        }
        int ans = 0;
        for (auto& ir : dp) {
            ans = max(ans, ir.second);
        }
        
        
        return ans;
        
    }
};
