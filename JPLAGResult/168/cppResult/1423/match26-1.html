class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        unordered_map<string,int> m;
        int size=s.size();
        for(int len=minSize;len<=maxSize;len++){
            int a[26]={0},count=0;
            for(int i=0;i<len;i++){
                if(a[s[i]-'a']==0) count++;
                a[s[i]-'a']++;
            }
            for(int j=0;j<=size-len;j++){
                if(count<=maxLetters) m[s.substr(j,len)]++;
                if(j==size-len) break;
                if(a[s[j+len]-'a']==0) count++;

                a[s[j+len]-'a']++;
                if(a[s[j]-'a']==1) count--;
                a[s[j]-'a']--;
            }
        }
        int res=0;
        for(auto item : m) res=max(res,item.second);
        return res;
    }
};
