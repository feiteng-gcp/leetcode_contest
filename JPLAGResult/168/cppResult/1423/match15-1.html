class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        map<string, int> mp;
        for(int i = 0; i+minSize <= s.size(); i++) {
            string t = s.substr(i, minSize);
            vector<bool> vis(26, 0);
            int num = 0;
            for(auto c: s.substr(i, minSize)) {
                if(!vis[c-'a']) {
                    vis[c-'a'] = 1;
                    num++;
                }
            }

            if(num <= maxLetters) {
                mp[t]++;
            }
        }
        int ans = 0;
        for(auto p: mp) {
            ans = max(ans, p.second);
        }
        return ans;
    }
};
