typedef unsigned long long ull;
int cnt[30];
const int P = 1331;
class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        unordered_map<ull, int> mp;
        int sz = s.length();
        for (int i = 0; i <= sz - minSize; ++i) {
            memset(cnt, 0, sizeof(cnt));
            int lts = 0;
            ull val = 0;
            for (int j = 0; j < minSize; ++j) {
                int cur = s[i + j] - 'a' + 1;
                if (cnt[cur] == 0) { ++lts; }
                ++cnt[cur];
                if (lts > maxLetters) { break; }
                else {
                    val = val * P + cur;
                }
            }
            if (lts <= maxLetters) {
                ++mp[val];
                for (int j = minSize; j < maxSize && i + j < sz; ++j) {
                    int cur = s[i + j] - 'a' + 1;
                    if (cnt[cur] == 0) { ++lts; }
                    ++cnt[cur];
                    if (lts > maxLetters) { break; }
                    else {
                        val = val * P + cur;
                        ++mp[val];
                    }
                }
            }
        }
        int ans = 0;
        for (auto p : mp) {
            ans = max(ans, p.second);
        }
        return ans;
    }
};
