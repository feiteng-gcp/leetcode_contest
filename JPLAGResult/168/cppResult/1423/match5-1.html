class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        int ans = 0;
        for (int size = minSize; size <= maxSize; size++) {
            vector<int> charCnt(26, 0);
            unordered_map<string, int> sCnt;
            int curL = 0;

            for (int i=0; i<size; i++) {
                if (++charCnt[s[i] - 'a'] == 1) curL++;
            }
            if (curL <= maxLetters) {
                ans = max(ans, ++sCnt[s.substr(0, size)]);
            }
            for (int i=0; i<s.size()-size; i++) {
                if (--charCnt[s[i] - 'a'] == 0) curL--;
                if (++charCnt[s[i+size] - 'a'] == 1) curL++;
                if (curL <= maxLetters) {
                    ans = max(ans, ++sCnt[s.substr(i+1, size)]);
                }
            }
        }
        return ans;
    }
};
