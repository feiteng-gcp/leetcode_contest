class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        
        int L=s.size();
        map<string,int> counts;
        for(int len=minSize;len<=minSize;len++) {
            for(int i=0;i+len-1<=L-1;i++) {
                // i to i+len-1
                int mask=0;
                for(int j=i;j<=i+len-1;j++) {
                    mask |= (1<<(s[j]-'a'));
                }
                if (__builtin_popcount(mask) <= maxLetters) {
                    counts[s.substr(i,len)]++;
                }
            }
        }
        
        int ans=0;
        
        for(auto x : counts) {
            ans = max(ans, x.second);
        }
        return ans;
        
    }
};
