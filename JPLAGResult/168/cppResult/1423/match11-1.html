class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        int size = s.size();
        map<string,int> freq;
        for( int i = 0; i + minSize <= size; i++ ) {
            map<int,int> m;
            for( int j = i; j < i + maxSize && j < size; j++ ) {
                m[ s[j] ]++;
                if( m.size() > maxLetters ) {
                    break;

                }
                if( j - i + 1 >= minSize ) {
                    //cout << s.substr( i, j - i + 1) << endl;
                    freq[ s.substr( i, j - i + 1) ]++;
                }
            }
        }
        int res = 0;
        for( auto &iter : freq ) {
            res = max( res, iter.second );
        }
        return res;
    }
};
