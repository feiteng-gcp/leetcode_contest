class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        map<string,int> ct;
        int n=s.size();
        for (int i=0; i<=n-minSize; i++) {
            string cur=s.substr(i,minSize);
            set<char> st(cur.begin(),cur.end());
            if(st.size()<=maxLetters)
                ct[cur]++;
        }
        int ret=0;
        for (auto it : ct)
            ret=max(ret,it.second);
        return ret;
    }
};
