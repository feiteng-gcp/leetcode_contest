class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        map<string,int> count;
        for(int i = 0;i<s.size();i++){
            string tmp;
            int cnt[26];
            int vis=0;
            fill(cnt,cnt+26,0);
            //fill(num,num+26,0);
            //int Max=0;
            for(int j = i;j<i+maxSize;j++){
                if(j==s.size())break;
                tmp+=s[j];
                if(cnt[s[j]-'a']==0)vis++;
                cnt[s[j]-'a']++;
                if(vis<=maxLetters&&j-i+1>=minSize){
                    count[tmp]++;
                }
            }
        }
        int ans=0;
        for(auto it:count){
            ans = max(ans, it.second);
        }
        return ans;
    }
};
