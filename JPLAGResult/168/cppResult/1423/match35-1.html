class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        int n = s.size();
        unordered_map<string, int> all;
        for (int l = minSize; l <= maxSize; ++l) {
            unordered_map<char, int> m;
            for (int i = 0; i < l - 1; ++i) ++m[s[i]];
            for (int i = l - 1; i < n; ++i) {
                ++m[s[i]];
                if ((int)m.size() <= maxLetters) {
                    ++all[s.substr(i - l + 1, l)];
                }

                if (--m[s[i - l + 1]] == 0) m.erase(s[i - l + 1]);
            }
        }
        int res = 0;
        for (auto& p : all) {
            res = max(p.second, res);
        }
        return res;
    }
};
