class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
       // s = string(100000, 'a');maxLetters = 26;minSize = 1;maxSize = 26;
        unordered_map<string, int> m;
        int ret = 0;
        
        for (int len = minSize; len <= maxSize; ++len) {
            vector<int> c(26, 0);

            int count = 0;
            for (int j = 0; j < len; ++j) {
                if (c[s[j] - 'a'] == 0) ++count;
                c[s[j] - 'a']++;
            }
            for (int i = 0; i <= s.size() - len; ++i) {
                if (i > 0) {
                    c[s[i - 1] - 'a']--;
                    if (c[s[i - 1] - 'a'] == 0) count--;
                    if (c[s[i + len - 1] - 'a'] == 0) count++;
                    c[s[i + len - 1] - 'a']++;
                }
                if (count > maxLetters) continue;
                string sub = s.substr(i, len);
                m[sub]++;
            }
        }
        
        for (auto &e: m) 
            ret = max(ret, e.second);
        
        return ret;
    }
};
