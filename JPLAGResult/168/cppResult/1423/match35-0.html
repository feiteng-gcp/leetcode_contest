class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        int N=s.size();
        unordered_map<string, int> M;
        for(int i=0; i<N; i++)
        {
            unordered_set<char> SS;
            for(int j=i; j<=i+maxSize-1 && j<N; j++)
            {
               SS.insert(s[j]);
                
                if(i+minSize-1<=j && SS.size()<=maxLetters) M[s.substr(i, j-i+1)]++;
            }
        }
        
        int result=0;
        for(auto [a, b]: M)
        {
            result=max(result, b);
        }
        return result;
    }
};
