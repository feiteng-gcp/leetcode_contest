class Solution {
    unordered_map<string,int>Map;
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) 
    {
        for (int len = minSize; len<=maxSize; len++)
        {
            helper(s, len, maxLetters);
        }
        int ret = 0;
        for (auto x: Map)
        {
            ret = max(ret,x.second);
            // if (x.second==1) cout<<x.first<<endl;
        }
            
        return ret;
    }
    
    void helper(string &s, int len, int maxLetters)
    {
        vector<int>count(26,0);
        int letters = 0;
        for (int i=0; i<len; i++)
        {
            count[s[i]-'a']++;
            if (count[s[i]-'a']==1) letters++;
        }
        if (letters<=maxLetters)
            Map[s.substr(0,len)]++;
        
        for (int i=len; i<s.size(); i++)
        {
            count[s[i]-'a']++;
            if (count[s[i]-'a']==1) letters++;
            count[s[i-len]-'a']--;
            if (count[s[i-len]-'a']==0) letters--;
            if (letters<=maxLetters)
            {
                Map[s.substr(i-len+1,len)]++;
                // cout<<s.substr(i-len+1,len)<<" "<<letters<<endl;
            }
                
        }
    }
};
