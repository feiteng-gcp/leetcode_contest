class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        unordered_map<string, int> mp;
        int n = s.length(), res = 0, left = 0;
        map<char, int> cnt;
        int sum = 0;
        for (int i = 0; left < n;) {
            while (i < n) {
               if (!cnt.count(s[i]) && sum == maxLetters) break;
               if (++cnt[s[i++]] == 1) ++sum; 
            }
            //cout << left << " " << i << " " << sum << endl;
            for (int j = minSize+left; j <= maxSize+left && j <= i; ++j) {
                //cout << s.substr(left, j-left) << "\n";
                res = max(res, ++mp[s.substr(left, j-left)]);
            }
            if (--cnt[s[left]] == 0) {
                cnt.erase(s[left]);
                --sum;
            }
            left++;
        }
        return res;
    }
};
