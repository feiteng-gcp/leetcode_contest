class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        map<string, int> ctr;
        int n = s.size();
        for (int i = 0; i + minSize - 1 <= n - 1; i++) {
            set<char> st;
            for (int j = i; j <= i + minSize - 1; j++) {
                st.insert(s[j]);
            }
            if (st.size() <= maxLetters) {
                ctr[s.substr(i, minSize)]++;

            }
        }
        int ret = 0;
        for (auto p : ctr) {
            ret = max(ret, p.second);
        }
        return ret;
    }
};
