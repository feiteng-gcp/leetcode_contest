class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        int numLetters = 0;
        vector<int> letters(26);
        for (int i = 0; i < minSize; i++) {
            if (letters[s[i]-'a'] == 0) {
                numLetters++;
            }
            letters[s[i]-'a']++;
        }
        unordered_map<string, int> m;
        for (int i = minSize; i <= s.length(); i++) {
            if (numLetters <= maxLetters) {
                string s1 = s.substr(i - minSize, minSize);
                m[s1]++;
            }
            if (i == s.length()) {
                break;
            }
            if (letters[s[i - minSize]-'a'] == 1) {
                numLetters--;
            }
            letters[s[i - minSize]-'a']--;
            if (letters[s[i]-'a'] == 0) {
                numLetters++;
            }

            letters[s[i]-'a']++;
        }
        int best = 0;
        for (auto it:m) {
            best = max(best, it.second);
        }
        return best;
    }
};
