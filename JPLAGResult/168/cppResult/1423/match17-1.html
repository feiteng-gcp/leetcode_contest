class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        map<string, int> cnt;
        for (int i = 0; i <= s.length() - minSize; i++) {
            cnt[s.substr(i, minSize)]++;
        }

        int ret = 0;
        for (auto p : cnt) {
            string s = p.first;
            bool v[26] = {false};
            int cnt = 0;

            for (char c : s) {
                if (!v[c-'a']) {
                    v[c-'a'] = true;
                    cnt++;
                }
            }

            if (cnt <= maxLetters) {
                ret = max(ret, p.second);
            }
        }

        return ret;
    }
};
