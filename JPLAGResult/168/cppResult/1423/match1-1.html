class Solution {
public:
    int maxFreq(string s, int maxLetters, int minSize, int maxSize) {
        if(s.size()<minSize)
            return 0;
        unordered_map<string, int> strs;
        unordered_map<int, int> diff;
        for(int i=0; i<minSize; i++)
            diff[s[i]]++;
        if(diff.size()<=maxLetters)
            strs[s.substr(0, minSize)]+=1;
        for(int i=minSize; i<s.size(); i++) {
            diff[s[i-minSize]]--;
            if(diff[s[i-minSize]]==0)
                diff.erase(s[i-minSize]);
            diff[s[i]]++;
            if(diff.size()<=maxLetters)
                strs[s.substr(i-minSize+1, minSize)]+=1;
        }
        int res = 0;
        for(auto & x:strs)
            res = max(x.second, res);
        return res;
    }
};
