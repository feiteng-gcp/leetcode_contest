class Solution {
public:
    int findNumbers(vector<int>& nums) {
        int res = 0;
        for (int num : nums) {
            int cur = 0;
            while (num != 0) {
                num /= 10;
                cur++;
            }
            if (cur % 2 == 0) res++;
        }
        return res;
    }
};
