class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        import collections
        cnt = collections.Counter(A.split(' '))
        cnt2 = collections.Counter(B.split(' '))
        
        res = []
        for k, v in cnt.items():
            if v == 1 and k not in cnt2:
                res.append(k)
        for k, v in cnt2.items():
            if v == 1and k not in cnt:
                res.append(k)
        return res
