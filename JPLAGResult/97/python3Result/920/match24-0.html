import collections
class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        re = set()
        counter = collections.Counter()
        for w in A.split():
            counter[w] += 1

        for w in B.split():
            counter[w] += 1

        return [k for k in counter if counter[k] == 1]
