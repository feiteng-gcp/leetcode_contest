class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        
        from collections import Counter
        
        s1 = Counter(A.split())
        s2 = Counter(B.split())
        
        res = []
        
        
for k, v in s1.items():
            if v < 2 and k not in s2:
                res.append(k)
                
        for k, v in s2.items():
            if v < 2 and k not in s1:
                res.append(k)
                
        return res
