from collections import defaultdict

class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        a = defaultdict(int)
        b = defaultdict(int)
        for s in A.split():
            a[s] += 1
        for s in B.split():
            b[s] += 1
        c = set()
        
for k, v in a.items():
            if v == 1 and k not in b:
                c.add(k)
        for k, v in b.items():
            if v == 1 and k not in a:
                c.add(k)
        return list(c)
