from collections import defaultdict
class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        d_a = defaultdict(int)
        d_b = defaultdict(int)
        A_words = []
        B_words = []
        if len(A):
            A_words = A.split(' ')
        for word in A_words:
            d_a[word] += 1
        if len(B):
            B_words = B.split(' ')
        for word in B_words:
            d_b[word] += 1
        res 
= []
        for word in A_words:
            if d_a[word] == 1 and d_b[word] == 0:
                res.append(word)
        for word in B_words:
            if d_b[word] == 1 and d_a[word] == 0:
                res.append(word)
        return res
