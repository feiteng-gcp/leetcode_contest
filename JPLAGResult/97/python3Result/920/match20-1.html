import collections
import queue
import bisect
import random
import math
class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        dicA = collections.Counter()
        dicB = collections.Counter()
        
        for a in A.split():
            dicA[a] += 1
        
        for b in B.split():
            dicB[b] += 1
        
        result 
= []
        
        for a in A.split():
            if dicA[a] == 1 and a not in dicB:
                result.append(a)
        
        for b in B.split():
            if dicB[b] == 1 and b not in dicA:
                result.append(b)
    
        return result
