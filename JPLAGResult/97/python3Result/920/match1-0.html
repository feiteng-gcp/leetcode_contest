class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        dA=collections.defaultdict(lambda:0)
        dB=collections.defaultdict(lambda:0)

        wordsA=[word for word in A.split(' ') if word!='']
        wordsB=[word for word in B.split(' ') if word!='']
        for word in wordsA:
            dA[word]+=1
        for word in wordsB:
            dB[word]+=1
        
        res=[]
        for word in dA:
            if dA[word]==1 and (word not in dB):
                res.append(word)
        for word in dB:
            if dB[word]==1 and (word not in dA):
                res.append(word)
        return res
