class Solution:
    def uncommonFromSentences(self, A, B):
        A = A.split()
        B = B.split()
        resA = [w for w in A if (w not in B) and w and A.count(w)==1]
        resB = [w for w in B if (w not in A) and w and B.count(w)==1]
        return list(set(resA + resB))
