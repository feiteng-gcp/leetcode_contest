class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        occur = {}
        A = A.split()
        B = B.split()
        for word in A:
            occur[word] = occur.get(word, 0) + 1
        for word in B:
            occur[word] = occur.get(word, 0) + 1
        ret = []
        for word, count in occur.items():
            if count == 1:
                ret.append(word)
        return ret
