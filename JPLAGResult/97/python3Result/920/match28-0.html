class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        ta=A.split(' ');
        sd=dict()
        
        for one in ta:
            if one not in sd:
                sd[one] = 0;
            sd[one]  = sd[one] + 1
        
        sd2=dict()
        tb = B.split(' ')
        for one in tb:
            if one not in sd2:
                sd2[one] = 0;
            sd2[one] = sd2[one] + 1
        
                
        
        ans = set()
        for k,v in sd.items():
            if v==1 and k not in sd2:
                ans.add(k)
        for k,v in sd2.items():
            if v==1 and k not in sd:
                ans.add(k)
        
        return list(ans)
