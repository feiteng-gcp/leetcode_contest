class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        wordmap = {}
        for word in A.split():
            if word in wordmap:
                wordmap[word] += 1
            else:
                wordmap[word] = 1
        for word in B.split():
            if word in wordmap:
                wordmap[word] += 1
            else:
                wordmap[word] = 1
        res = []
        for word in wordmap.keys():
            if wordmap[word] == 1:
                res.append(word)
        return res
