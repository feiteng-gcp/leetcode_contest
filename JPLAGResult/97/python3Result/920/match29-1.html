from collections import Counter


class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        A = A.split()
        B = B.split()
        cnt_a = Counter(A)
        cnt_b = Counter(B)
        res = []
        for k, v in cnt_a.items():
            if v == 1 and cnt_b[k] == 0:
                res.append(k)
        for k, v in cnt_b.items():
            if v == 1 and cnt_a[k] == 0:
                res.append(k)
        return res
