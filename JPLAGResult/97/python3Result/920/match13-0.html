class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        
#         setA = set(A.split())

#         return list(setA ^ setB)

        colA = collections.Counter(A.split())
        colB = collections.Counter(B.split())
        
        ans = []
        for w in colA:
            if colA[w] == 1 and colB[w] == 0:
                ans.append(w)
        
        for w in colB:
            if colB[w] == 1 and colA[w] == 0:
                ans.append(w)
        
        return ans
