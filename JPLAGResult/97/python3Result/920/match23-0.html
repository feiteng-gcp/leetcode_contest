class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        a_map = {}
        for word in A.split(' '):
            a_map[word] = a_map.get(word, 0) + 1
        b_map = {}
        for word in B.split(' '):
            b_map[word] = b_map.get(word, 0) + 1
        result = []
        for word, count in a_map.items():
            if count == 1 and word not in b_map:
                result.append(word)
        for word, count in b_map.items():
            if count == 1 and word not in a_map:
                result.append(word)
        return result
