from collections import Counter
class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        A_words = A.split(' ') if len(A) > 0 else []
        B_words = B.split(' ') if len(B) > 0 else []
        A_freqs = Counter(A_words)
        B_freqs = Counter(B_words)
        ans = []
        for word in A_freqs:
            if A_freqs[word] == 1 and word not in B_freqs:
                ans.append(word)
        for word in B_freqs:
            if B_freqs[word] == 1 and word not in A_freqs:
                ans.append(word)
        return ans
