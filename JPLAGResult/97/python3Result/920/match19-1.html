class Solution:
    def uncommonFromSentences(self, A, B):
        """
        :type A: str
        :type B: str
        :rtype: List[str]
        """
        return self.visit(A,B)
    
    
    def visit(self, A, B):
        uncommon = []
        wordDict 
= {}
        for word in A.split():
            if word not in wordDict:
                wordDict[word] = 0
            else:
                wordDict[word] += 1
        for word in B.split():
            if word in wordDict:
                wordDict[word] += 1
            else:
                wordDict[word] = 0
        for word in wordDict:
            if wordDict[word] == 0:
                uncommon.append(word)
        return uncommon
