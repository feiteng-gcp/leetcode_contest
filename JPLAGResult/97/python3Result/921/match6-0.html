class Solution:
    def spiralMatrixIII(self, R, C, r0, c0):
        """
        :type R: int
        :type C: int
        :type r0: int
        :type c0: int
        :rtype: List[List[int]]
        """
        direct, res, n = [(-1, 0), (0, 1), (1, 0), (0, -1)], [[r0, c0]], R * C
        def dfs(i, j, ind, l):
            if len(res) == n: return
            for _ in range(l):
                i += direct[ind][0]
                j += direct[ind][1]
                if 0 <= i < R and 0 <= j < C:
                    res.append([i, j])
            ind = (ind + 1) % 4
            for _ in range(l):
                i += direct[ind][0]
                j += direct[ind][1]
                if 0 <= i < R and 0 <= j < C:
                    res.append([i, j])
            ind = (ind + 1) % 4
            dfs(i, j, ind, l + 1)
        dfs(r0, c0, 1, 1)
        return res
