class Solution:
    def spiralMatrixIII(self, R, C, r0, c0):
        """
        :type R: int
        :type C: int
        :type r0: int
        :type c0: int
        :rtype: List[List[int]]
        """

        dir = [[0, 1], [1, 0], [0, -1], [-1, 0]]
        k = 0
        r, c = r0, c0
        res = [[r0, c0]]
        cnt, cntc = 1, 1
        while len
(res) < R*C:
            r += dir[k][0]
            c += dir[k][1]
            if 0<=r<R and 0<=c<C and cnt:
                res.append([r, c])
            cnt -= 1
            if not cnt:
                k += 1
                k %= 4
                if k == 2 or k == 0:
                    cntc += 1
                cnt = cntc
        return res
