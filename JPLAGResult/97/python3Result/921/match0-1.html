class Solution:
    def spiralMatrixIII(self, R, C, r0, c0):
        """
        :type R: int
        :type C: int
        :type r0: int
        :type c0: int
        :rtype: List[List[int]]
        """
        dirR,dirC=[-1,0,1,0],[0,1,0,-1]
        dirs=1
        length=1
        realGo=1
        res=[[r0,c0]]
        r,c=r0,c0
        while realGo<R*C:
            for i in range(length):
                r,c=r+dirR[dirs],c+dirC[dirs]
                if 0<=r<R and 0<=c<C:
                    realGo+=1
                    res.append([r,c])
            dirs=(dirs+1)%4
            for i in range(length):
                r,c=r+dirR[dirs],c+dirC[dirs]
                if 0<=r<R and 0<=c<C:
                    realGo+=1
                    res.append([r,c])
            dirs=(dirs+1)%4
            length+=1
        return res
