class Solution:
    def spiralMatrixIII(self, R, C, r0, c0):
        """
        :type R: int
        :type C: int
        :type r0: int
        :type c0: int
        :rtype: List[List[int]]
        """
        directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        direction = 0
        step = 1
        same_step = False
        current_step = 1
        result = [[r0, c0]]
        space_count = R * C
        while len
(result) < space_count:
            r0 += directions[direction][0]
            c0 += directions[direction][1]
            if 0 <= r0 < R and 0 <= c0 < C:
                result.append([r0, c0])
            step -= 1
            if step == 0:
                if same_step:
                    same_step = False
                    current_step += 1
                else:
                    same_step = True
                step = current_step
                direction = (direction + 1) % 4
        return result
