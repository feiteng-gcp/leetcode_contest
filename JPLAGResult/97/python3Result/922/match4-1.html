import collections, bisect

import numpy as np

MOD = 1000000007

class Solution:
    def possibleBipartition(self, N, dislikes):
        """
        :type N: int
        :type dislikes: List[List[int]]
        :rtype: bool
        """

        status = [0] * (N + 1)
        dis = collections.defaultdict
(set)

        for d in dislikes:
            dis[d[0]].add(d[1])
            dis[d[1]].add(d[0])

        cand = collections.deque()

        status[1] = -1
        cand.append(1)
        while len(cand) > 0:
            n = cand.pop()
            g = status[n]
            for p in dis[n]:
                if status[p] == g:
                    return False
                if status[p] == 0:
                    status[p] = -g
                    cand.append(p)

        return True
