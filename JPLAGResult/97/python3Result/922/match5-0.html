class Solution:
    def possibleBipartition(self, N, dislikes):
        dic = collections.defaultdict(set)
        for a, b in dislikes:
            dic[a].add(b)
            dic[b].add(a)
            
        tag = [0]*(N+1)
        
        for i in range(1, N+1):
            if tag[i]!=0: continue
            q = [i]
            t = 1
            visited = {i}
            while q:
                new_q = []
                for v in q:
                    if tag[v]==0:
                        tag[v] = t
                        for vv in dic[v]:
                            new_q.append(vv)
                    else:
                        if tag[v]!=t:
                            return False
                t = -t
                q = new_q
        
        return True
