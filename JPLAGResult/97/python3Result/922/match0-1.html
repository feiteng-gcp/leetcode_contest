class Solution:
    def possibleBipartition(self, N, dislikes):
        """
        :type N: int
        :type dislikes: List[List[int]]
        :rtype: bool
        """
        if N<2:
            return True
        graph=collections.defaultdict(list)
        for u,v in dislikes:
            graph[u].append(v)
            graph[v].append(u)
        setP={dislikes[0][0]}
        q=collections.deque()
        q.append(dislikes[0][0])
        d={}
        d[dislikes[0][0]]=0
        while q:
            u=q.popleft()
            #ulike=d[u]
            for v in graph[u]:
                if v not in d:
                    d[v]=1-d[u]
                    q.append(v)
                else:
                    if d[u]==d[v]:
                        return False
        return True
