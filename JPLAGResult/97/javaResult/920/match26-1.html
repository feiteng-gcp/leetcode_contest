class Solution {
    public String[] uncommonFromSentences(String A, String B) {
		List<String> uncommon = new ArrayList<>();
		Map<String, Integer> aMap = new HashMap<>();
		Map<String, Integer> bMap = new HashMap<>();
		for (String a : A.split("\\s+")) {
			aMap.put(a, aMap.getOrDefault(a, 0) + 1);
		}
		for (String b : B.split("\\s+")) {
			bMap.put(b, bMap.getOrDefault(b, 0) + 1);
		}
		for (String s 
: aMap.keySet()) {
			if (aMap.get(s) == 1 && !bMap.containsKey(s)) {
				uncommon.add(s);
			}
		}
		for (String s : bMap.keySet()) {
			if (bMap.get(s) == 1 && !aMap.containsKey(s)) {
				uncommon.add(s);
			}
		}
		String[] res = new String[uncommon.size()];
		for (int i = 0; i < uncommon.size(); i++) {

			res[i] = uncommon.get(i);
		}
		return res;
    }
}
