class Solution {
    public String[] uncommonFromSentences(String A, String B) {
        Map<String,Integer> A_record = new HashMap<>();
        Map<String,Integer> B_record = new HashMap<>();
        String[] words_A = A.split(" ");
        String[] words_B = B.split(" ");
        for (String word: words_A) {
            A_record.put(word,A_record.getOrDefault(word,0)+1);
        }
        for (String word: words_B) {
            B_record.put(word,B_record.getOrDefault(word,0)+1);
        }        
        List<String> result = new ArrayList<>();
        for (String word:A_record.keySet()) {
            if (A_record.get(word) == 1 && !B_record.containsKey(word)) {
                result.add(word);
            }
        }
        for (String word:B_record.keySet()) {
            if (B_record.get(word) == 1 && !A_record.containsKey(word)) {
                result.add(word);
            }
        }
        String[] result_s = new String[result.size()];
        for (int i=0; i<result_s.length; i++) {
            result_s[i] = result.get(i);
        }
        return result_s;
    }
}
