class Solution {
    public String[] uncommonFromSentences(String A, String B) {
        HashMap<String,Integer> map = new HashMap<>();
        String[] strs = A.split("\\s+");
        for(String str : strs)
        {     
            if(map.containsKey(str)) 
                map.put(str,2);
            else
                map.put(str,1);
                
        }
        strs = B.split("\\s+");
        for(String str : strs)
        {
            if(map.containsKey(str)) 
                map.put(str,2);
            else
                map.put(str,1);
        
}
        List<String> anslist = new ArrayList();
        for(Map.Entry<String, Integer> entry : map.entrySet())
        {
            if(entry.getValue() < 2)
                anslist.add(entry.getKey())
;
        }
        String[] ansstr = new String[anslist.size()];
        int count = 0;
        for(String str : anslist)
        {   
            ansstr[count++] = str;
        }
        return ansstr;
    }
}
