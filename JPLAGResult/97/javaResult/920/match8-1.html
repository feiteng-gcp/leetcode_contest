class Solution {
    public String[] uncommonFromSentences(String A, String B) {
         Map<String, Integer> words1 = new HashMap<>();
        String[] As = A.split(" ");
        for (String w : As) {
            words1.put(w, words1.getOrDefault(w, 0) + 1);
        }
        String[] Bs = B.split(" ");
        for (String w : Bs) {
            words1.put(w, words1.getOrDefault(w, 0) + 1);
        }
        List<String> result = new ArrayList<>();
        for (Map.Entry<String, Integer> e : words1.entrySet()) {
            if (e.getValue() == 1) {
                result.add(e.getKey());
            
}
        }
        String[] res = new String[result.size()];
        int i = 0;
        for (String w : result) {
            res[i] = w;
            i++;

        }
        return res;
    }
}
