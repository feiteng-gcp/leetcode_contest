class Solution {
    public String[] uncommonFromSentences(String A, String B) {
        String[] A_words = A.split("\\s+");
        String[] B_words = B.split("\\s+");
        
        Map<String,Integer> A_count = new HashMap<>();
        Map<String,Integer> B_count = new HashMap<>();

        for (String word : A_words) {
            A_count.put(word,A_count.getOrDefault(word,0) + 1);
        }

        for (String word : B_words) {
            B_count.put(word,B_count.getOrDefault(word,0) + 1);
        }
        
        List<String> ls = new ArrayList<>();
        for (String word:A_count.keySet()
             ) {
            if (A_count.get(word) == 1 && !B_count.containsKey(word)) {
                ls.add(word);
            }           
        }

        for (String word:B_count.keySet()
                ) {
            if (B_count.get(word) == 1 && !A_count.containsKey(word)) {
                ls.add(word);
            }
        }
        
        String[] ans = new String[ls.size()];

        for (int i = 0; i < ans.length; i++) {
            ans[i] = ls.get(i);
        }
        
        return ans;
    }
}
