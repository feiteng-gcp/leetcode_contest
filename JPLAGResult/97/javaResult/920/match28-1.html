class Solution {
    public String[] uncommonFromSentences(String A, String B) {
        String[] tokensA = A.split(" ");
        String[] tokensB = B.split(" ");
        Map<String, Integer> cntA = new HashMap<>();
        Map<String, Integer> cntB = new HashMap<>();
        for (String str: tokensA) {
            cntA.compute(str, (k,v) -> v==null? 1:v+1);
        }
        for (String str: tokensB) {
            cntB.compute(str, (k,v) -> v==null? 1:v+1);
        
}
        StringBuilder sb = new StringBuilder();
        for(String keyA: cntA.keySet()) {
            if (cntA.get(keyA) == 1 && !cntB.containsKey(keyA)) {
                sb.append(keyA+" ");
            }
        }
        for(String keyB: cntB.keySet()) {
            if (cntB.get(keyB) == 1 && !cntA.containsKey(keyB)) {
                sb.append(keyB+" ");
            }
        }
        if (sb.length() == 0) {
            return new String[]{};
        }
        return sb.toString().trim().split(" ");
    }
}
