class Solution {
    public boolean possibleBipartition(int N, int[][] dislikes) {
        Set<Integer> setA = new HashSet<>();
        Set<Integer> setB = new HashSet<>();
        List<Integer>[] adjList = new ArrayList[N + 1];
        for (int i = 1; i <= N; i++)
            adjList[i] = new ArrayList<>()
;
        for (int[] pair : dislikes) {
            adjList[pair[0]].add(pair[1]);
            adjList[pair[1]].add(pair[0]);
        }
        for (int i = 1; i < N; i++) {
            if (!setA.contains(i) && !setB.contains(i)) {
                setA.add(i);
                if (!dfs(setA, setB, i, adjList))
                    return false;
            }
        }
        return true;
    }
    
    public boolean dfs(Set<Integer> setA, Set<Integer> setB, int cur, List<Integer>[] adjList) {
        for (int next : adjList[cur]) {
            if (setA.contains(cur)) {
                if (setA.contains(next))
                    return false;
                if (setB.contains(next))
                    continue;
                setB.add(next);
                if (!dfs(setA, setB, next, adjList))
                    return false;
            } else {
                if (setB.contains(next))
                    return false;
                if (setA.contains(next))
                    continue;
                setA.add(next);
                if (!dfs(setA, setB, next, adjList))
                    return false;
            }
        }
        return true;
    } 
}
