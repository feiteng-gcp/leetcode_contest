class Solution {
    public boolean possibleBipartition(int N, int[][] dislikes) {
        List<Integer>[] graph = new List[N];
        for (int i = 0; i < N; i++) {
            graph[i] = new ArrayList<>();
        }
        for (int[] dislike : dislikes) {
            graph[dislike[0]-1].add(dislike[1]-1);
            graph[dislike[1]-1].add(dislike[0]-1);

        }
        int[] color = new int[N];
        
for (int i = 0; i < N; i++) {
            if (color[i] == 0) {
                color[i] = 1;
                if (!dfs(i, graph, color)) {
                    return false;
                }
            }
        }
        return true;
    }

    boolean dfs(int c, List<Integer>[] graph, int[] color) {
        int cc = color[c];
        for (int n : graph[c]) {
            if (color[n] == cc) {
                return false;
            }
            
if (color[n] == 0) {
                color[n] = 3 - cc;
                if (!dfs(n, graph, color)) {
                    return false;
                }
            }
        }
        return true;
    }
}
