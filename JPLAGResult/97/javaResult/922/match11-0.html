class Solution {
    public boolean possibleBipartition(int n, int[][] d) {
        LinkedList<Integer> a[]=new LinkedList[n];
        for(int i=0;i<n;i++){
            a[i]=new LinkedList<>();
        }
        for(int c[]:d){
            a[c[0]-1].add(c[1]-1);
            a[c[1]-1].add(c[0]-1);
        }
        boolean g[]=new boolean[n];
        boolean v[]=new boolean[n];
        for(int i=0;i<n;i++){
            if(!v[i]){
                if(!dfs(i,g,v,false,a)){
                    return false;
                }
            }
        }
        return true;
    }
    boolean dfs(int c,boolean g[],boolean v[],boolean cb,LinkedList<Integer> a[]){
        v[c]=true;g[c]=cb;
        for(int nt:a[c]){
            if(!v[nt]){
                if(!dfs(nt,g,v,!cb,a)){
                    return false;
                }
            }
            else if(g[nt]==cb){
                return false;
            }
        }
        return true;
    }
}
