class Solution {
    public boolean possibleBipartition(int N, int[][] dislikes) {
        int[] partition = new int[N+1];
        //int[] visited = new int[N+1];
        Deque<Integer> queue = new LinkedList<>();
        Set<Integer>[] record = new Set[N+1];
        for (int i=1; i<N+1;i++) {
            record[i]  = new HashSet<Integer>();
        }
        for (int[] dislike:dislikes){
            record[dislike[0]].add(dislike[1]);
            record[dislike[1]].add(dislike[0]);            
        }
        for (int i=1; i<=N;i++){
            if (partition[i] == 0) {
                partition[i] = 1;
                queue.add(i);
                //visited[i] = 1;
                while (!queue.isEmpty()){
                    int curr = queue.poll();
                    int curr_group = partition[curr];
                    for (int candidate:record[curr]){
                        if (partition[candidate] == 0) {
                            partition[candidate] = 0- curr_group;
                            queue.add(candidate);
                        }
                        else {
                            if (partition[candidate] == curr_group) {
                                /*System.out.println("curr: " +curr+ " curr_group: "+curr_group+" candidate: "+candidate);
                                for (int j=1; j<=N;j++) {
                                    System.out.print(partition[j]+" ");
                                }*/
                                return false;
                            }
                        }
                    }
                }
            }
        }
        return true;
    }
}
