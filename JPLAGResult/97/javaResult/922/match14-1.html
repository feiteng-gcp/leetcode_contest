class Solution {
    public boolean possibleBipartition(int N, int[][] dislikes) {
        int[] color = new int[N + 1];
        Map<Integer, boolean[]> map = new HashMap();
        for (int i = 1; i <= N; i++) {
            map.put(i, new boolean[N + 1]);
        }
        for (int[] d : dislikes) {
            map.get(d[0])[d[1]] = true;
            map.get(d[1])[d[0]] = true;
        }
        for (int i = 1; i <= N; i++) {
            if (color[i] != 0) {
                continue;
            }
            if (!dfs(map, color, i, 1)) {
                return false;
            
}
        }
        return true;
    }
    
    public boolean dfs(Map<Integer, boolean[]> map, int[] color, int person, int currColor) {
        if (color[person] != 0) {
            return color[person] == currColor;
        }
        color[person] = currColor;
        boolean[] dislike = map.get(person);
        for (int i = 1; i < dislike.length; i++) {
            if (!dislike[i]) {
                continue;
            }
            if (!dfs(map, color, i, -currColor)) {
                return false;
            }
        }
        return true;
    }
}
