class Solution {
public boolean possibleBipartition(int N, int[][] dislikes) {
            int[] colors = new int[N];

            Map<Integer, List<Integer>> graph = new HashMap<>();
            for(int[] d : dislikes)
            {
                graph.computeIfAbsent(d[0]-1, k->new ArrayList<>()).add(d[1]-1);
                graph.computeIfAbsent(d[1]-1, k->new ArrayList<>()).add(d[0]-1);
            }

            for(int i=0; i<N; i++)
            {
                if (colors[i] == 0)
                {

                    if (!colorGraph(graph, colors, 1, i))
                    {
                        // graph is not bipartite
                        // two adjacent nodes have same color
                        return false;
                    }

                }
            }

            // graph is bipartite
            return true;
        }

        // return true if adjacent nodes have different color
        private boolean colorGraph(Map<Integer, List<Integer>> graph, int[] colors, int c, int i) {
            if (colors[i] != 0)
            {
                return colors[i] == c;
            }

            colors[i] = c;
            if (graph.containsKey(i))
            {
                for (Integer adj : graph.get(i))
                {
                    if (!colorGraph(graph, colors, c*-1, adj))
                    {
                        return false;
                    }
                }
            }
            return true;
        }
}
