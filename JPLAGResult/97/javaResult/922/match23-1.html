class Solution {
    public boolean possibleBipartition(int N, int[][] dislikes) {
        List<List<Integer>> graph = new ArrayList<>();
        
for (int i = 0; i <= N; ++i) { graph.add(new ArrayList<>()); } // ignore index 0
        for (int i = 0; i < dislikes.length; ++i) {
            int p1 = dislikes[i][0];
            int p2 = dislikes[i][1];
            graph.get(p1).add(p2);
            graph.get(p2).add(p1);
        }
        
        boolean[] group1 = new boolean[N + 1];  // ignore index 0
        boolean[] group2 = new boolean[N + 1];
//        boolean[] placed = new boolean[N + 1];
        boolean[] processed = new boolean[N + 1];
        for (int i = 1; i <= N; ++i) {
            if (!processed[i]) {
//                placed[i] = true;
                processed[i] = true;
                Deque<Integer> queue = new ArrayDeque<>();
                queue.add(i);
                while (!queue.isEmpty()) {
                    int curr = queue.remove();
                    int currgroup;
                    if (!group1[curr] && !group2[curr]) {
                        group1[curr] = true;    // new CC, arbitrarily place
                        currgroup = 1;
                    } else if (group1[curr]) {
                        currgroup = 1;
                    } else {
                        currgroup = 2;
                    }
                    for (int neighbor : graph.get(curr)) {
                    	// check
                    	if (currgroup == 1 && group1[neighbor]) { return false; }
                        if (currgroup == 2 && group2[neighbor]) { return false; }
                        if (!group1[neighbor] && !group2[neighbor]) {
                        	if (currgroup == 1) { group2[neighbor] = true; }
                            else { group1[neighbor] = true; }
                        }
                        // reaching here means no error and all neighbors were placed in group
                        // now add them to the queue
                        if (!processed[neighbor]) {
                            queue.add(neighbor);
                            processed[neighbor] = true;
                        }
                    }
                }
            }
        }
        return true;
    }
}
