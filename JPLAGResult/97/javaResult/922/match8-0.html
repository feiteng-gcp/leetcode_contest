class Solution {
    
    class Node {
        public int id;
        public List<Node> adj = new ArrayList<Node>();
        public Node(int idd) {
            this.id = idd;
        }
        
    }
    public boolean possibleBipartition(int N, int[][] dislikes) {
        Node [] nodes = new Node[N];
        
        for (int i = 0; i < N; i++)
        {
            nodes[i] = new Node(i + 1);
        }
        for (int [] dislike: dislikes)
        {
            int d1 = dislike[0];
            int d2 = dislike[1];
            
            nodes[d1 - 1].adj.add(nodes[d2 - 1]);
            nodes[d2 - 1].adj.add(nodes[d1 - 1]);
        }
        
        int [] mark = new int [N];
        
        for (int i = 0; i < N; i++)
        {
            if (mark[i] != 0) continue;
            if (!exploreGraph(nodes[i], mark, 1)) return false;
        }
        return true;        
    }
    
    private boolean exploreGraph(Node n, int [] mark, int color)
    {
        int mcolor = mark[n.id - 1];
        if (mcolor != 0 && mcolor != color) return false;
        if (mcolor != 0 && mcolor == color) return true;
        mark[n.id - 1] = color;
        for (Node m : n.adj)
        {
            if (!exploreGraph(m, mark, -1 * color)) return false;
        }
        return true;
    }
}
