class Solution {
    public boolean possibleBipartition(int N, int[][] graph) {
        int[] visited = new int[N+1];
        
       Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int[] g: graph) {
            if (!map.containsKey(g[0])) {
                map.put(g[0], new HashSet<>());
            }
            if (!map.containsKey(g[1])) {
                map.put(g[1], new HashSet<>());
            }
            map.get(g[0]).add(g[1]);
            map.get(g[1]).add(g[0]);
        }
        
        for (int i = 1; i <= N; i++) {
            if (map.containsKey(i) && visited[i] == 0) {
                visited[i] = 1;
                Queue<Integer> q = new LinkedList<>();
                q.offer(i);
                while(!q.isEmpty()) {
                    int current = q.poll();
                    
                    for (int next: map.get(current)) {
                        if (visited[next] == 0) {
                            visited[next] = (visited[current] == 1) ? 2 : 1;
                            q.offer(next);
                        } else {
                            if (visited[current] == visited[next]) return false;
                        }
                    }
                }
            }
        }
        return true;
    }
}
