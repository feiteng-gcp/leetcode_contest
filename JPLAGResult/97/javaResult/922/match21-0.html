class Solution {
    public boolean possibleBipartition(int N, int[][] dislikes) {
    	HashMap<Integer, Set<Integer>> graph = new HashMap<>();
    	if (dislikes.length == 0 || dislikes[0].length == 0) return true;
    	for (int i = 0; i < dislikes.length; i++) {
    		int u = dislikes[i][0] - 1;
    		int v = dislikes[i][1] - 1;
    		if (!graph.containsKey(u)) graph.put(u, new HashSet<Integer>());
    		if (!graph.containsKey(v)) graph.put(v, new HashSet<Integer>());
    		graph.get(u).add(v);
    		graph.get(v).add(u);
    	}
    	return isBipartite(N, graph);
    }
    
    public boolean isBipartite(int n, HashMap<Integer, Set<Integer>> graph) {
        int[] color = new int[n];
        Arrays.fill(color, -1);

        for (int start = 0; start < n; ++start) {
            if (color[start] == -1) {
                Stack<Integer> stack = new Stack<>();
                stack.push(start);
                color[start] = 0;

                while (!stack.empty()) {
                    Integer node = stack.pop();
                    if (!graph.containsKey(node)) continue;
                    for (int nei: graph.get(node)) {
                        if (color[nei] == -1) {
                            stack.push(nei);
                            color[nei] = color[node] ^ 1;
                        } else if (color[nei] == color[node]) {
                            return false;
                        }
                    }
                }
            }
        }

        return true;
    }
}
