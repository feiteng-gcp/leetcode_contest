class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int[][] dirs = new int[][] {{0,1},{1,0},{0,-1},{-1,0}};
        int dir = 0;
        int index = 0;
        int steps_count = 0;
        int dir_count = 0;
        int steps = 1;
        int total = R*C;
        int[][] result = new int[total][2];
        while (index<R*C) {
            if (r0>=0 && r0<R && c0 >=0 && c0<C) {
                result[index][0] = r0;
                result[index][1] = c0;
                index++;
            }
            r0 = r0+dirs[dir][0];
            c0 = c0+dirs[dir][1];
            steps_count++;
            if (steps_count == steps){
                steps_count = 0;
                dir = (dir + 1) %4;
                dir_count++;
                if (dir_count == 2) {
                    dir_count = 0;
                    steps++;
                }
            }
        }
        return result;
    }
}
