class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
    	int[][] dir = new int[][] {{0, 1}, {1, 0},{0, -1},{-1, 0}};
    	
    	int current = 0;
    	int x = r0;
    	int y = c0;
    	int d = 0;
    	int[][] ret = new int[R*C][2];
    	int len = 1;
    	while (current < R * C) {
    		for (int i = 0; i < 2; i++) {
        		for (int j = 0; j < len; j++) {
            		if (x >= 0 && x < R && y >= 0 && y < C) {
            			ret[current][0] = x;
            			ret[current][1] = y;
            			current++;
            		}
        			x += dir[d][0];
            		y += dir[d][1];
        		}        		
        		d = (d + 1) % 4;    			
    		}
    		len++;
    	}
        return ret;
    }
}
