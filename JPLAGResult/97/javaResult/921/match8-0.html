class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int[] dirs = {0, 1, 0, -1, 0};
        int step = 1;
        int d = 0;
        int count = 1;
        int i = r0, j = c0;
        int[][] res = new int[R*C][2];
        res[0][0] = r0;
        res[0][1] = c0;
        while (count < R * C) {
            for (int s = 0; s < step; ++s) {
                i += dirs[d];
                j += dirs[d + 1];
                // System.out.println(i + "," + j);
                if (i >= 0 && i < R && j >= 0 && j < C) {
                    res[count][0] = i;
                    res[count++][1] = j;
                }
            }
            for (int s = 0; s < step; ++s) {
                i += dirs[d + 1];
                j += dirs[d + 2];
                // System.out.println(i + "," + j);
                if (i >= 0 && i < R && j >= 0 && j < C) {
                    res[count][0] = i;
                    res[count++][1] = j;
                }
            }
            d = 2 - d;
            ++step;
        }
        return res;
    }
}
