class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int size = R * C;
        int[][] matrix = new int[size][2];
        int count = 0;
        int len = 1;
        while (count < size) {
            for (int i = 0; i < len; i++) {
                if (isValid(r0, c0, R, C)) {
                    matrix[count][0] = r0;
                    matrix[count][1] = c0;
                    count++;
                }
                c0 += 1;
            }
            for (int i = 0; i < len; i++) {
                if (isValid(r0, c0, R, C)) {
                    matrix[count][0] = r0;
                    matrix[count][1] = c0;
                    count++;
                }
                r0 += 1;
            }
            len++;
            for (int i = 0; i < len; i++) {
                if (isValid(r0, c0, R, C)) {
                    matrix[count][0] = r0;
                    matrix[count][1] = c0;
                    count++;
                }
                c0 -= 1;
            }
            for (int i = 0; i < len; i++) {
                if (isValid(r0, c0, R, C)) {
                    matrix[count][0] = r0;
                    matrix[count][1] = c0;
                    count++;
                }
                r0 -= 1;
            }
            len++;
        }
        return matrix;
    }
    
    public boolean isValid(int r0, int c0, int R, int C) {
        if (r0 < 0 || c0 < 0 || r0 >= R || c0 >= C)
            return false;
        return true;
    }
}
