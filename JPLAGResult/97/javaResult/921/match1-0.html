class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int[][] dirArray = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int dir = 0;
        int[][] ret = new int[R*C][2];
        ret[0] = new int[]{r0, c0};
        int i = 1;
        for (int step = 1; step <= Math.max(R, C) * 2 + 2; ++step) {
            for (int j = 0; j < 2; ++j) {
                for (int k = 0; k < step; ++k) {
                    r0 += dirArray[dir][0];
                    c0 += dirArray[dir][1];
                    if (r0 >= 0 && r0 < R && c0 >= 0 && c0 < C) {
                        ret[i] = new int[]{r0, c0};
                        i += 1;
                    }
                }
                dir = (dir + 1) % 4;
            }
        }
        return ret;
    }
}
