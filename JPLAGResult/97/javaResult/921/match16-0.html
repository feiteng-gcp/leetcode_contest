class Solution {
    public int[][] spiralMatrixIII(int r, int c, int r0, int c0) {
        int a[][]=new int[r*c][2];
        int cr=r0,cc=c0,tct=0,spl=1,cs=1,i=cc,j=cr;
        while(tct<r*c){
            for(i=cc;i<cc+spl && i>cc-spl;i+=cs){
                if(i<c && i>=0 && j<r && j>=0){
                    a[tct][0]=j;a[tct][1]=i;
                    tct++;
                }
            }
            cc=i;
            if(tct<r*c){
                for(j=cr;j<cr+spl && j>cr-spl;j+=cs){
                    if(i<c && i>=0 && j<r && j>=0){
                        a[tct][0]=j;a[tct][1]=i;
                        tct++;
                    }
                }
            }
            cr=j;
            spl++;
            cs=cs*-1;
        }
        return a;
    }
}
