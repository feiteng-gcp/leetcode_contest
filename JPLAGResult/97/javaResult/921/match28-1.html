class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int N = R * C;
        int[][] ans = new int[N][];
        int curr = 1;   // 1-indexed
        int row = r0;
        int col = c0;
        int dir = 0;    // 0=right, 1=down, 2=left, 3=up
        int stepsize = 1;
        int numsteps = 0;   // num steps in current direction
        while (curr <= N) {
            if (inBounds(row, col, R, C)) {
                ans[curr - 1] = new int[]{row, col};
                ++curr;
            }
            
if (dir == 0) { ++col; }
            else if (dir == 1) { ++row; }
            else if (dir == 2) { --col; }
            else { --row; }
            ++numsteps;
            if (numsteps == stepsize) {
                if (dir == 0) {
                    dir = 1;
                } else if (dir == 1) {
                    dir = 2;
                    ++stepsize;
                } else if (dir == 2) {
                    dir = 3;
                } else {
                    dir = 0;
                    ++stepsize;
                }
                numsteps = 0;
            }
        }
        return ans;
    }
    
    public boolean inBounds(int row, int col, int R, int C) {
        return row >= 0 && row < R && col >= 0 && col < C;
    }
}
