class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int n = R * C;
        int[][] res = new int[n][2];
        res[0][0] = r0; 
        res[0][1] = c0;
        int cnt = 1;
        int len = 1;
        while (cnt < n) {
            if (r0 >= 0 && r0 < R) {
                for (int i = 0; i < len; ++i) {++c0; if (c0 >= 0 && c0 < C) {res[cnt][0] = r0; res[cnt][1] = c0; ++cnt; } }
            } else c0 += len;
            if (c0 >= 0 && c0 < C) {
                for (int i = 0; i < len; ++i) {++r0; if (r0 >= 0 && r0 < R) {res[cnt][0] = r0; res[cnt][1] = c0; ++cnt; } }
            } else r0 += len;
            //System.out.printf("cnt:%d. (%d, %d)\n", cnt, r0, c0);
            ++len;
            if (r0 >= 0 && r0 < R) {
                for (int i = 0; i < len; ++i) {--c0; if (c0 >= 0 && c0 < C) {res[cnt][0] = r0; res[cnt][1] = c0; ++cnt; } }
            } else c0 -= len;
            if (c0 >= 0 && c0 < C) {
                for (int i = 0; i < len; ++i) {--r0; if (r0 >= 0 && r0 < R) {res[cnt][0] = r0; res[cnt][1] = c0; ++cnt; } }
            } else r0 -= len;
            ++len;
            //System.out.printf("cnt:%d. (%d, %d)\n", cnt, r0, c0);
            //break;
        }
        return res;
    }
