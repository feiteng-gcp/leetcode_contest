class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int[][] res = new int[R*C][2];
        int i = 0;
        
        
int[][] dir = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int di = 0;
        
        int rows = 1;
        int cols = 1;
        boolean isRow = true;
        
        res[i][0] = r0;
        res[i][1] = c0;
        i++;
        
        while(i < R*C) {
            int curDir = rows;
            if(!isRow) {
                curDir = cols;
            }
            
for(int j=0; j<curDir; j++) {
                r0 += dir[di][0];
                c0 += dir[di][1];
                if(r0 >= 0 && r0 < R && c0 >=0 && c0<C) {
                    res[i][0] = r0;
                    res[i][1] = c0;
                    i++;
                }
            }
            if(isRow) {
                rows++;
            } else {
                cols++;
            }
            di = (di + 1)%4;
            isRow = !isRow;
        }
        
        return res;
    }
}
