class Solution {
    private int nextRow;
    private int nextCol;
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int[][] res = new int[R*C][2];
        int index = 0;
        if (r0 >= 0 && r0 < R && c0 >= 0 && c0 < C) {
            res[0][0] = r0;
            res[0][1] = c0;
            ++index;
        }
        nextRow = r0;
        nextCol = c0;
        int steps = 1;
        while (index < R*C) {
            index = rightAndDown(R,C,res,index,steps);
            ++steps;
            index = leftAndUp(R,C,res,index,steps);
            ++steps;
        }
        return res;
    }
    private int rightAndDown(int R, int C, int[][] res, int index, int steps) 
{
        for (int i = 1; i <= steps; ++i) {
            ++nextCol;
            if (nextRow >= 0 && nextRow < R && nextCol >= 0 && nextCol < C) {
                res[index][0] = nextRow;
                res[index][1] = nextCol;
                ++index;
            }
        }
        
for (int i = 1; i <= steps; ++i) {
            ++nextRow;
            if (nextRow >= 0 && nextRow < R && nextCol >= 0 && nextCol < C) { 
                res[index][0] = nextRow;
                res[index][1] = nextCol;
                ++index;
            }
        }
        return index;
    }
    private int leftAndUp(int R, int C, int[][] res, int index, int steps) 
{
        for (int i = 1; i <= steps; ++i) {
            --nextCol;
            if (nextRow >= 0 && nextRow < R && nextCol >= 0 && nextCol < C) {
                res[index][0] = nextRow;
                res[index][1] = nextCol;
                ++index;
            }
        }
        
for (int i = 1; i <= steps; ++i) {
            --nextRow;
            if (nextRow >= 0 && nextRow < R && nextCol >= 0 && nextCol < C) { 
                res[index][0] = nextRow;
                res[index][1] = nextCol;
                ++index;
            }
        }
        return index;
    }
}
