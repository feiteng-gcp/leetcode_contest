class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int[] xd = new int[] {0, 1, 0, -1};
        int[] yd = new int[] {1, 0, -1, 0};
        int[] step = new int[]{1, 1, 1, 1};
        int total = R * C;
        int k = 0;
        int dir = 0;
        int x = r0, y = c0;
        int[][] ans = new int[total][2];
        ans[0] = new int[]{r0, c0};
        k++;
        while (k < total) {
            for (int i = 0; i < step[dir]; i++) {
                x = x + xd[dir];
                y = y + yd[dir];
                
if (x >= 0 && x < R && y >= 0 && y < C) {
                    ans[k++] = new int[]{x, y};
                }
            }
            step[dir]++;
            step[(dir + 2) % 4]++;
            dir = (dir + 1) % 4;
        }
        return ans;
    }
}
