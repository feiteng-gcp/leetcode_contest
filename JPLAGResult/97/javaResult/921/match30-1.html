class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int[][] dirs = new int[][] {{0,1},{1,0},{0,-1},{-1,0}};
        int[][] index = new int[R * C][2];
        boolean[][] used = new boolean[R][C];
        dfs(dirs,0,2,r0,c0,R,C,0,index,used);
        return index;
    }
    
    private void dfs (int[][] dirs, int dirIndex, int twoSteps,
                      int i, int j,int R, int C,int order, int[][] index ,boolean[][] used) {
        if (order == R * C) {
            return;
        }
        else {
//            System.out.println("i = " + i);
//            System.out.println("j = " + j);
//            System.out.println("order = " + order);
//            System.out.println("R * C = " + R * C);
//            if (i >= 0 && i < R && j >= 0 && j < C) {
//                index[order++] = new int[]{i,j};
//            }
            int[] dir = dirs[dirIndex];
            dirIndex = (dirIndex + 1) % 4;
            int step = twoSteps / 2;
            twoSteps++;
            int ci = i + dir[0] * step;
            int cj = j + dir[1] * step;
            if (ci == i && i >= 0 && i < R) {
                if (cj > j) {
                    for (int k = Math.max(j,0); k <= Math.min(cj, C - 1); k++) {
                    if (!used[i][k]) {
                        index[order++] = new int[]{i,k};
                        used[i][k] = true;
                        }
                    }
                }
                else {
                    for (int k = Math.min(j,C - 1); k >= Math.max(cj, 0); k--) {
                        if (!used[i][k]) {
                            index[order++] = new int[]{i,k};
                            used[i][k] = true;
                        }
                    }
                }
            }
            else if (cj == j && j>=0 && j < C){
                if (ci > i) {
                    for (int k = Math.max(i,0); k <= Math.min(ci, R - 1); k++) {
                        if (!used[k][j]) {
                            index[order++] = new int[]{k,j};
                            used[k][j] = true;
                        }
                    }
                }
                else {
                    for (int k = Math.min(i,R - 1); k >= Math.max(ci, 0); k--) {
                        if (!used[k][j]) {
                            index[order++] = new int[]{k,j};
                            used[k][j] = true;
                        }
                    }
                }

            }
            dfs(dirs,dirIndex,twoSteps,ci,cj,R,C,order,index,used);
        }
    }
}
