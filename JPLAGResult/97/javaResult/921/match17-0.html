class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
        int[][] visited = new int[R*C][2];
        visited[0] = pair(r0, c0);
        int[][] mv = {{0,1},{1,0},{0,-1},{-1,0}};
        int len = 0;
        int index = 1;
        while (index < R * C) {
            for (int m = 0; m < 4; m++) {
                if (m == 0 || m == 2) {
                    len++;
                }
                for (int i = 0; i < len; i++) {
                    r0 += mv[m][0];
                    c0 += mv[m][1];
                    if (r0 >= R || r0 < 0 || c0 >= C || c0 < 0) {
                        continue;
                    }
                    visited[index] = pair(r0, c0);
                    index++;
                    if (index >= R * C) {
                        break;
                    }
                }
                if (index >= R * C) {
                    break;
                }
            }
        }
        return visited;
    }
    
    public int[] pair(int r, int c) {
        int[] rc = {r, c};
        return rc;
    }
}
