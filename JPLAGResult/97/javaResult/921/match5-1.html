class Solution {
    public int[][] spiralMatrixIII(int R, int C, int r0, int c0) {
         int[][] NESW = new int[][]{{-1, 0}, {0, 1}, {1, 0}, {0, -1}};
        int d = 1;
        int N = R * C;
        boolean[][] visited = new boolean[R][C];
        int[][] result = new int[N][2];
        result[0][0] = r0;
        result[0][1] = c0;
        int[] pos = {r0, c0};
        visited[pos[0]][pos[1]] = true;
        int step = 1;
        while (step < N) {
            while (step < N) {
                pos[0] = pos[0] + NESW[d][0];
                pos[1] = pos[1] + NESW[d][1];
                if (pos[0] >= 0 && pos[0] < R && pos[1] >= 0 && pos[1] < C && !visited[pos[0]][pos[1]]) {
                    result[step][0] = pos[0];
                    result[step][1] = pos[1];
                    visited[pos[0]][pos[1]] = true;
                    step++;
                }
                int[] next = new int[2];
                int turn = (d + 1) % 4;
                next[0] = pos[0] + NESW[turn][0];
                next[1] = pos[1] + NESW[turn][1];
                if (next[0] < 0 || next[0] == R || next[1] < 0 || next[1] == C) {
                    d = turn;
                } else if (!visited[next[0]][next[1]]) {
                    d = turn;
                }
                continue;
            }
        }
        return result;
    }
}
