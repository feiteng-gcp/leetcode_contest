class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        int radius = max(max(r0, R - r0 - 1), max(c0, C - c0 - 1));
        int x = r0;
        int y = c0;
        vector<vector<int>> steps = {{x, y}};
        for (int r = 1; r <= radius; r++) {
            y++;
            if (0 <= x && x < R && 0 <= y && y < C) {
                steps.push_back({x, y});
            }
            for (int i = 0; i < r * 2 - 1; i++) {
                x++;
                if (0 <= x && x < R && 0 <= y && y < C) {
                    steps.push_back({x, y});
                }
            }
            for (int i = 0; i < r * 2; i++) {
                y--;
                if (0 <= x && x < R && 0 <= y && y < C) {
                    steps.push_back({x, y});
                }
            }
            for (int i = 0; i < r * 2; i++) {
                x--;
                if (0 <= x && x < R && 0 <= y && y < C) {
                    steps.push_back({x, y});
                }
            }
            for (int i = 0; i < r * 2; i++) {
                y++;
                if (0 <= x && x < R && 0 <= y && y < C) {
                    steps.push_back({x, y});
                }
            }
            
        }
        return steps;
    }
};
