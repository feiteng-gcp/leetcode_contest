class Solution {
public:
    bool in(int R, int C, int r, int c){
        return 0 <= r && r < R && 0 <= c && c < C;
    }
    
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector<int>> ans;
        
        int dir = 0, step = 1, si = 0, r = r0, c = c0;
        ans.push_back({r0, c0});
        while(ans.size() < R * C){
            if(dir == 0){

                for(int i = 0; i < step; ++i){
                    c++;
                    if(in(R, C, r, c)){
                        ans.push_back({r, c});
                    }
                }
            }else if(dir == 1){

                for(int i = 0; i < step; ++i){
                    r++;
                    if(in(R, C, r, c)){
                        ans.push_back({r, c});
                    }
                }
            }else if(dir == 2){

                for(int i = 0; i < step; ++i){
                    c--;
                    if(in(R, C, r, c)){
                        ans.push_back({r, c});
                    }
                }
            }else{

                for(int i = 0; i < step; ++i){
                    r--;
                    if(in(R, C, r, c)){
                        ans.push_back({r, c});
                    }
                }
            }
            
            dir = (dir + 1) % 4;
            si++;
            if(si == 2){
                si = 0;
                step++;
            }
        }
        
        return ans;
    }
};
