class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        int n = R*C;

        vector<vector<int>> ans, dir{{0,1},{1,0},{0,-1},{-1,0}};
        ans.reserve(n);
        int curDir = 0, m = 1, t = 0;
        int x = r0, y = c0;
        ans.push_back({x,y});

        while((int)ans.size() < n) {
            //cout << x << " " << y << endl;
            for(int i = 0; i < m; i++) {
                x+=dir[curDir][0];
                y+=dir[curDir][1];
                if(x >=0 && x < R && y >= 0 && y < C) {
                    ans.push_back({x,y});
                }
            }
            if(t==0) {
                t++;
            } else {
                m++;
                t=0;
            }
            curDir++;
            curDir%=4;
        }
        return ans;
    }
};
