class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector<int>> dirs={{0,1},{1,0},{0,-1},{-1,0}};
        int DIR=0;
        int num=R*C;
        int now=1;
        vector<int> steps{1,1};//横轴和竖轴的位移初始都是1和1
        vector<vector<int>> ret;
        while(now<=num){
            for(int j=0;j<steps[DIR%2];j++){
                if(r0>=0&&r0<=R-1&c0>=0&&c0<=C-1){
                    ret.push_back(vector<int>{r0,c0});
                    now+=1;
                }
                r0+=dirs[DIR][0];
                c0+=dirs[DIR][1];
                
            }
            steps[DIR%2]++;
            DIR=(DIR+1)%4;
            
            
            
        }
        return ret;
    }
};
