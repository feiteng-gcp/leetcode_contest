class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector<int>> result;
        vector<vector<int>> dir = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int step = 1;
        int dir_idx = 0;
        int spiral_count = 0;
        int r = r0;
        int c = c0;
        result.push_back({r0, c0});
        while(result.size() < R * C) {
            int dr = dir[dir_idx][0];
            int dc = dir[dir_idx][1];
            
            for(int i = 0; i < step; i++) {
                r = r + dr;
                c = c + dc;
                if (inGrid(R, C, r, c)) {
                    result.push_back({r, c});
                }
            }
            dir_idx = (dir_idx + 1) % 4;
            spiral_count++;
            if (spiral_count % 2 == 0) {
                step++;
            }
        }
        return result;
    }
private:
    bool inGrid(int R, int C, int r, int c) {
        return r >= 0 && r < R && c >=0 && c < C;
    }
};
