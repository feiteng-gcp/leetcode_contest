class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector<int>> ans;
        
        // 2次1步，2次2步，……
        int mx[4] = {0, 1, 0, -1}, my[4] = {1, 0, -1, 0};
        
        int direction = 0;
        int step = 1;
        int x = r0, y = c0;

        while (ans.size() < R * C) {
            for (int i = 0; i < 2; i++) {
                for (int j = 0; j < step; j++) {
                    // cout << x << ' ' << y << endl;
                    if (0 <= x && x < R && 0 <= y && y < C) {
                        vector<int> pos;
                        pos.push_back(x);
                        pos.push_back(y);
                        ans.push_back(pos);
                        if (ans.size() >= R * C)
                            break;
                    }
                    x += mx[direction];
                    y += my[direction];
                }
                if (ans.size() >= R * C)
                    break;
                direction = (direction + 1) % 4;
            }
            step++;
        }
        
        return ans;
    }
};
