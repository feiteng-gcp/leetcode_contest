class Solution {
public:
   vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<pair<int,int>> direct = {make_pair(0,1),make_pair(1,0),make_pair(0,-1),make_pair(-1,0)};        int rest = R*C;
        int x = r0,y = c0;
        vector<vector<int>> result;
        int step = 0;
        int lenth = 1;
        int times = 0;
        int dis = 0;
        while(rest > 0){
           if(x >= 0 && x < R && y >= 0 && y < C){
              vector<int> local;
              local.push_back(x);
              local.push_back(y);
              result.push_back(local);
              rest--;
           }
           x += direct[dis].first;
           y += direct[dis].second;
           step++;
           if(step == lenth){
              step = 0;
              dis = (dis + 1)%4;
              times++;
              if(times == 2){
                 times = 0;
                 lenth += 1;
              }
           }
        }
        return result;
    }
};
