class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector<int>> ans;
        vector<int> lastVisited = {r0, c0};
        
        ans.push_back(vector<int>{r0, c0});
        int step = 0;
        while (ans.size() < R * C) {
            int r = lastVisited[0], c = lastVisited[1];
            
            // Right, down
            step++;
            for (int i = 1; i <= step; i++) {
                if (r < R && r >= 0 && (c+i) < C && (c+i) >= 0)
                    ans.push_back(vector<int>{r, c+i});
            }
            c += step;
            for (int i = 1; i <= step; i++) {
                if ((r+i) < R && (r+i) >= 0 && c < C && c >= 0)
                    ans.push_back(vector<int>{r+i, c});
            }
            r += step;
            
            // Left, up
            step++;
            for (int i = 1; i <= step; i++) {
                if (r < R && r >= 0 && (c-i) < C && (c-i) >= 0)
                    ans.push_back(vector<int>{r, c-i});
            }
            c -= step;
            for (int i = 1; i <= step; i++) {
                if ((r-i) < R && (r-i) >= 0 && c < C && c >= 0)
                    ans.push_back(vector<int>{r-i, c});
            }
            r -= step;
            
            lastVisited[0] = r; lastVisited[1] = c;
        }
        
        return ans;
    }
};
