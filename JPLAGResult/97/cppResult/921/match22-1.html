class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector<int>> ans;
        
        int currentX = c0;
        int currentY = r0;
        
        int nth = 0;
        
        int dir = 3; // 0:right, 1:bottom, 2:left, 3:top
        
        while (true)
        {
            ++nth;
            for (int i = 0; i < nth / 2; ++i)
            {
                //cout << currentX << ' ' << currentY << endl;
                
                if (currentX >= 0 && currentY >= 0 && currentX < C && currentY < R)
                {
                    vector<int> t;
                    t.push_back(currentY);
                    t.push_back(currentX);
                    ans.push_back(t);
                }
                
                switch (dir)
                {
                    case 0: ++currentX; break;
                    case 1: ++currentY; break;
                    case 2: --currentX; break;
                    case 3: --currentY; break;
                }
                
                if (ans.size() == R*C)
                    return ans;
            }
            
            ++dir;
            if (dir == 4)
                dir = 0;
        }
        
        return ans;
    }
};
