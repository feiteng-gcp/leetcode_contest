class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector<int>> res;
        res.push_back({r0, c0});
        int n = R * C - 1;

        int ds[4][2] = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int d = 0, s = 1, r = r0, c = c0;

        while(n > 0) {
            for(int i = 1; i <= s; ++i) {
                r += ds[d][0];
                c += ds[d][1];
                if(r >= 0 && r < R && c >= 0 && c < C) {
                    res.push_back({r, c});
                    --n;
                    if(n == 0) break;
                }
            }            
            ++d;
            if(d % 2 == 0) ++s;
            if(d == 4) d = 0;
        }
        return res;
    }
};
