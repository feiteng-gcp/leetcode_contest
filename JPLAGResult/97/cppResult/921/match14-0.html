class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        int rLowLimit = r0;
        int rHighLimit = r0 + 1;
        int cLowLimit = c0;
        int cHighLimit = c0 + 1;
        
        int prop[4][2] = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int propIndex = 0;
        vector<vector<int>> re;
        while (re.size() < R * C) {
            if (r0 < R && r0 >= 0 && c0 < C && c0 >= 0) {
                re.push_back(vector<int>{r0, c0});
            }
            r0 += prop[propIndex % 4][0];
            c0 += prop[propIndex % 4][1];
            if (r0 >= rHighLimit) {
                propIndex++;
                rHighLimit++;
            } else if (c0 >= cHighLimit) {
                propIndex++;
                cHighLimit++;
            } else if (c0 < cLowLimit) {
                propIndex++;
                cLowLimit--;
            } else if (r0 < rLowLimit) {
                propIndex++;
                rLowLimit--;
            }
        }
        return re;
    }
};
