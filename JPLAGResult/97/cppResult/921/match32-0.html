class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector<int>> res;
        int dx[4] = {0, 1, 0, -1};
        int dy[4] = {1, 0, -1, 0};
        int i = 0;
        res.push_back({r0, c0});
        while(res.size() != R*C)
        {
            for(int j = 0; j <= i/2 ; j++)
            {
                
                r0 += dx[i%4];
                c0 += dy[i%4];
                if(r0 >= 0 && r0 < R && c0 >= 0 && c0 < C)
                    res.push_back({r0, c0});
            }
            i++;
        }
        
            
        return res;
    }
};
