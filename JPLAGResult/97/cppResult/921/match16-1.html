class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        int visit = R*C - 1;
        vector<vector<int>>ans;
        int step = 0;
        int dx[4] = {0, 1, 0, -1};
        int dy[4] = {1, 0, -1, 0};
        ans.push_back(vector<int> {r0, c0});

        int length = 1;
        while(visit) {
            for(int i = 0; i < (step+2)/2; ++i) {
                r0 += dx[step%4];
                c0 += dy[step%4];
                if(r0 >= 0 && r0 < R && c0 >=0 && c0 < C){
                    ans.push_back(vector<int>{r0,c0});
                    visit--;
                }
            }
            step += 1;
        }
        return ans;
    }
};
