class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        int dirs[4][2] = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
        int cur_dir = 0, cur_len = 1, flag = 0, x = r0, y = c0, len = 1;
        vector<vector<int>> res;
        while(res.size() < R*C) {
            if(isValid(R, C, x, y)) {
                res.push_back(vector<int>{x, y});
            }
            x += dirs[cur_dir][0];
            y += dirs[cur_dir][1];
            if(--len == 0) {
                cur_dir = (cur_dir+1)%4; 
                if(flag == 1) {
                    cur_len++;
                    flag = 0;
                }
                else
                    flag = 1;
                len = cur_len;
            }
        }
        return res;
    }
    
    inline bool isValid(int R, int C, int x, int y) {
        return x >= 0 && x < R && y >= 0 && y < C;
    }
};
