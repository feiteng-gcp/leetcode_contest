int dx[] = {0, 1, 0, -1};
int dy[] = {1, 0, -1, 0};
class Solution {
public:

bool check(int x, int y, int n, int m) {
	if(x >= 0 && x < n && y >= 0 && y < m) return 1;
	return 0;
}
vector<vector<int>> spiralMatrixIII(int n, int m, int x, int y) {
    int c = 0;
    int s = n * m;
    int d = 0;
    int p0 = 0, p1 = 1;
    vector<vector<int>> res;
    res.push_back({x, y});c++;

    while(c < s) {
    	for (int i = 0; i < p1; i++) {
    		x = x + dx[d];
    		y = y + dy[d];
    		if(check(x, y, n, m)) {
    			//cout << x << " " << y << endl;
    			res.push_back({x, y});
    			c++;
    		}
    	}

    	d = (d + 1) % 4;
    	for (int i = 0; i < p1; i++) {
    		x = x + dx[d];
    		y = y + dy[d];
    		if(check(x, y, n, m)) {
    			//cout << x << " " << y << endl;
    			res.push_back({x, y});
    			c++;
    		}
    	}
    	d = (d + 1) % 4;
    	p1++;

    }
    return res;
}

};
