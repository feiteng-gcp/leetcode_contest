class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        int cnt = 1;
        vector<vector<int>> ans;
        ans.push_back({r0, c0});
        int dx[4] = {1, 0, -1, 0};
        int dy[4] = {0, 1, 0, -1};
        int cur = 1;
        int d = 0;
        while (cnt < R * C) {
            //cout << r0 << " " << c0 << "\n";
            for (int i = 1; i <= cur; i++) {
                r0 += dy[d%4];
                c0 += dx[d%4];
                if (r0 < R && r0 >= 0 && c0 < C && c0 >= 0) {
                    ans.push_back({r0, c0});
                    cnt++;
                }
                if (cnt >= R * C) break;
            }           
            d++;
            if (cnt >= R * C) break;
            for (int i = 1; i <= cur; i++) {
                r0 += dy[d%4];
                c0 += dx[d%4];
                if (r0 < R && r0 >= 0 && c0 < C && c0 >= 0) {
                    ans.push_back({r0, c0});
                    cnt++;
                }
                if (cnt >= R * C) break;
            }
            cur++;
            d++;
        }
        return ans;
    }
};
