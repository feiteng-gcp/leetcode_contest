#define inside(i, j) (i >= 0 && i < R && j >= 0 && j < C)
#define add(i, j) { re.push_back(vector<int>{ i, j }); ++cnt; }

int gd[] = { -1, 0, 1, 0 };
int gc[] = { 0, 1, 0, -1 };

class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector<int>> re;
        
        int i = r0, j = c0, d = 1, x = 1, cnt = 0;
        if (inside(i, j)) add(i, j);
        
        while (cnt < R * C) {
            for (int k = 0; k < 2; ++k) {
                for (int p = 0; p < x; ++p) {
                    //cout << i << " " << j << " " << d << " " << x << endl;
                    
                    i += gd[d];
                    j += gc[d];
                    
                    if (inside(i, j)) add(i, j);
                }
                d = (d + 1) % 4;
            }
            x += 1;
        }
        
        return re;
    }
};
