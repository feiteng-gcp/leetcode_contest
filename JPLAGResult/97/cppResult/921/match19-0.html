class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<pair<int, int>> dirs = {{0,1}, {1,0}, {0,-1}, {-1,0}};
        int dir_idx = 3;
        set<pair<int, int>> visited;
        int found = 0;
        vector<vector<int>> spiral;
        while(found < R * C) {
            if (r0 < R and r0 >= 0 and c0 < C and c0 >= 0) {
                found ++;
                spiral.push_back({r0, c0});
                visited.emplace(r0, c0);
            }
            
            int new_idx = (dir_idx + 1) % 4;
            int new_r0 = r0 + dirs[new_idx].first;
            int new_c0 = c0 + dirs[new_idx].second;
            if (visited.count({new_r0, new_c0}) == 0) {
                r0 = new_r0;
                c0 = new_c0;
                dir_idx = new_idx;
            } else {
                r0 += dirs[dir_idx].first;
                c0 += dirs[dir_idx].second;
            }
        }
        return spiral;
    }
};
