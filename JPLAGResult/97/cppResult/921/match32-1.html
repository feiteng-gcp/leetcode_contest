class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
        vector<vector<int>> ans;
        int d=max(max(r0,R-1-r0),max(c0,C-1-c0));
        ans.push_back({r0,c0});
        for(int i=1;i<=d;i++){
            int x=r0-i+1,y=c0+i;
            for(int j=0;j<2*i;j++){
                ans.push_back({x+j,y});
            }
            x=x+2*i-1,y=y-1;
            for(int j=0;j<2*i;j++){
                ans.push_back({x,y-j});
            }
            x=x-1,y=y+1-2*i;
            for(int j=0;j<2*i;j++){
                ans.push_back({x-j,y});
            }
            x=x+1-2*i,y=y+1;
            for(int j=0;j<2*i;j++){
                ans.push_back({x,y+j});
            }
        }
        vector<vector<int>> r;
        for(auto t:ans){
            if(t[0]>=0&&t[0]<R&&t[1]>=0&&t[1]<C) r.push_back(t);
        }
        return r;
    }
};
