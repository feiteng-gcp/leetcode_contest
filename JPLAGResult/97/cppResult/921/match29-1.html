class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r, int c) {
        vector<vector<int>> dirs { {0,1}, {1,0}, {0,-1}, {-1,0} };
        vector<int> steps {1,1};
        int id = 0;
        int is = 0;
        int cnt = 1;
        int total = R * C;
        vector<vector<int>> ans;
        ans.push_back( { r, c } );

        while( cnt < total )
        {
            for( int i = 0; i < steps[is]; i++ )
            {
                r += dirs[id][0];
                c += dirs[id][1];
                if( r >= 0 && r < R && c >= 0 && c < C )
                {
                    ans.push_back( { r, c } );
                    cnt++;
                    
                    if( cnt >= total )
                        break;
                }
            }
            steps[is]++;
            id = (id + 1) % 4;
            is = (is + 1) % 2;
        }
        return ans;
    }
};
