class Solution {
public:
	bool ok(int x, int y, int R, int C)
	{
		return 0 <= x && x < R && 0 <= y && y < C;
	}
	vector<vector<int>> spiralMatrixIII(int R, int C, int r0, int c0) {
		vector<vector<int>> ans;
		vector<vector<int>> dirs = { {0,1},{1,0},{0,-1},{-1,0} };
		int d = 0;
		int tot = 1;
		int len = 1;

		int x, y;
		x = r0;
		y = c0;
		ans.push_back({ x,y });
		int turn = 0;
		while (tot < R * C)
		{
			for (int i = 0; i < len; ++i)
			{
				x = x + dirs[d][0];
				y = y + dirs[d][1];
				//cout << x << " " << y << endl;
				if (ok(x, y, R, C)) {
					ans.push_back({ x,y });
					tot++;
				}
			}
			turn++;
			d = (d + 1) % 4;
			if (turn % 2 == 0)
			{
				len++;
			}
		}
		return ans;
    }
};
