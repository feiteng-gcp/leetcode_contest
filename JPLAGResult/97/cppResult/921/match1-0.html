class Solution {
public:
    vector<vector<int>> spiralMatrixIII(int R, int C, int r, int c) {
        vector<vector<int>> ans;
        ans.push_back({r, c});
        for(int i = 1; i <= max(R, C); ++i) {
            for(int j = 0; j < i * 2 - 1; ++j) { // right
                ++c;
                if(r >= 0 && r < R && c >= 0 && c < C) {
                    ans.push_back({r, c});
                }
            }
            for(int j = 0; j < i * 2 - 1; ++j) { // down
                ++r;
                if(r >= 0 && r < R && c >= 0 && c < C) {
                    ans.push_back({r, c});
                }
            }
            for(int j = 0; j < i * 2; ++j) { // left
                --c;
                if(r >= 0 && r < R && c >= 0 && c < C) {
                    ans.push_back({r, c});
                }
            }
            for(int j = 0; j < i * 2; ++j) { // up
                --r;
                if(r >= 0 && r < R && c >= 0 && c < C) {
                    ans.push_back({r, c});
                }
            }
        }
        return ans;
    }
};
