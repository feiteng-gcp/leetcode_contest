int mat[105][10005];

class Solution {
public:

    Solution() {
        for (int i = 1; i < 10005; ++i) {
            mat[1][i] = i;
        }
        for (int i = 2; i < 105; ++i) {
            for (int j = 1; j < 10005; ++j) {
                int t = mat[i - 1][j] - 1;
                int l = 1, r = j - 1;
                while (l <= r) {
                    int k = (l + r) >> 1;
                    int a = mat[i][k], b = mat[i - 1][j - k - 1];
                    t = min(t, max(a, b));
                    if (a == b) break;
                    if (a > b) {
                        r = k - 1;
                    } else {
                        l = k + 1;
                    }
                }
                mat[i][j] = t + 1;
            }
        }
    }

    int superEggDrop(int K, int N) {
        return mat[K][N];
    }
};
