class Solution {
public:
    int superEggDrop(int K, int N) {
        vector<vector<int>> dp(K+1, vector<int>(N+1, 0));
        for(int i = 1; i <= N; i++)
            dp[1][i] = i;

        for(int i = 2; i <= K; i++)
        {
            
            for(int sum = 0, j = 1; j <= N && sum < N; j++)
            {
                sum += dp[i-1][j-1]+1;
                dp[i][j] = sum;
            }
        }
        for(int i = N; i >= 1; i--)
            if(dp[K][i])
                return i;
        return -1;
    }
};
