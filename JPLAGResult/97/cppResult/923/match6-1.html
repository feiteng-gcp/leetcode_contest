class Solution {
public:
    int dfs2(vector<vector<int>> &dp, int K, int N, int k, int n){
        if(dp[k][n]>=0) return dp[k][n];
	int cur=-1;
	if(n%2==0){
	    cur=max(dfs(dp, K, N, k-1, n/2 - 1), dfs(dp, K, N, k, n/2));
	    cur=min(cur, max(dfs(dp, K, N, k, n/2 - 1), dfs(dp, K, N, k-1, n/2)));

	}else{
	    cur=max(dfs(dp, K, N, k, n/2), dfs(dp, K, N, k-1, n/2));
	    cur=min(cur, max(dfs(dp, K, N, k, n/2-1), dfs(dp, K, N, k-1, n/2+1)));
	    cur=min(cur, max(dfs(dp, K, N, k-1, n/2-1), dfs(dp, K, N, k, n/2+1)));
	    
	}
	dp[k][n]=cur + 1;
        return dp[k][n];
    }


    int dfs(vector<vector<int>> &dp, int K, int N, int k, int n){
        if(dp[k][n]>=0) return dp[k][n];
	int cur=INT_MAX;
	

	int l=1, r=n;
	while(l<=r){
	    int m=l + (r-l)/2;
	    if(dfs(dp,K,N, k-1, m-1)>dfs(dp,K,N,k,n-m)){
		r = m -1;
	    }else{
		l = m + 1;
	    }
	}

	//for(int i=1; i<=n; i++){
	for(int i=max(l-5,1); i<=l+5 && i<=n; i++){ 
	    cur=min(cur, max(dfs(dp, K, N, k-1, i-1), dfs(dp, K, N, k, n-i)));
	}	
	
	dp[k][n]=cur+1;

        return dp[k][n];
    }
    int superEggDrop(int K, int N) {
        vector<vector<int>> dp(K+1, vector<int>(N+1, -1));

        for(int i=0; i<=N; i++){
            dp[1][i]=i;
        }
        for(int i=0; i<=K; i++){
            dp[i][1]=1;
            dp[i][0]=0;
        }
        return dfs(dp, K, N, K, N);
        
    }
};
