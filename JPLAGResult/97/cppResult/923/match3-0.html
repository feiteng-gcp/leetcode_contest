class Solution {
public:
    int superEggDrop(int k, int n) {
        vector<vector<int>> opt(k + 1, vector<int>(n + 1, 0));
        for (int i = 1; i <= n; i++) {
            opt[1][i] = i;
        }
        for (int i = 2; i <= k; i++) {
            opt[i][0] = 0;
            opt[i][1] = 1;
            for (int j = 2; j <= n; j++) {
                int lo = 1;
                int hi = j;
                while (hi > lo + 1) {
                    int mid = lo + (hi - lo) / 2;
                    if (opt[i][j - mid] >= opt[i - 1][mid - 1]) {
                        lo = mid;
                    } else {
                        hi = mid;
                    }
                }
                opt[i][j] = max(opt[i][j - lo], opt[i - 1][lo - 1]) + 1;
            }
        }
        return opt[k][n];
    }
};
