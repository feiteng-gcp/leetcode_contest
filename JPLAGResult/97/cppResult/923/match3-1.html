class Solution {
    vector<vector<int>> ans;
public:
    int superEggDrop(int K, int N) {
        ans=vector<vector<int>>(K+1,vector<int>(N+1,-1));
        return dp(K,N);
    }
    
    int dp(int k,int n){
        if(k==1) return n;
        if(n<=1) return n;
        if(ans[k][n]!=-1) return ans[k][n];
        

        int s=1,e=n;
        while(s<e){
            int mid=(s+e)/2;
            if(dp(k-1,mid-1)<=dp(k,n-mid)){
                s=mid+1;
            }else{
                e=mid;
            } 
        }
        
        int mid=s;
        int r=max(dp(k-1,mid-1),dp(k,n-mid))+1;
        r=min(r,max(dp(k-1,mid-2),dp(k,n-mid+1))+1);
        ans[k][n]=r;
        return r;
    }
};
