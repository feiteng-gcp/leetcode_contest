class Solution {
public:
    int superEggDrop(int K, int N) {
        int f[105][10005];
        int g[105][10005];
        memset(f, 0, sizeof f);
        for (int i = 1; i <= N; i++) 
            f[1][i] = i;
        for (int i = 2; i <= K; i++) {
            f[i][1] = 1;
            g[i][1] = 1;
            for (int j = 2; j <= N; j++) {
                f[i][j] = f[i - 1][j];
                g[i][j] = g[i][j - 1];
                for (int l = g[i][j - 1]; l <= j; l++) {
                    int tmp = max(f[i][j - l], f[i - 1][l - 1]) + 1;
                    if (tmp <= f[i][j]) {
                        f[i][j] = tmp;
                        g[i][j] = l;
                    }
                    else break;
                }
            }
        }
        return f[K][N];
    }
};
