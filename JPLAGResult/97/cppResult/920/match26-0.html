class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        unordered_map<string, int> htb;
        string tmp = "";
        for (char c : A) {
            if (c == ' ') {
                htb[tmp]++;
                tmp = "";
            }
            else tmp += c;
        }
        htb[tmp]++;
        tmp = "";
        for (char c : B) {
            if (c == ' ') {
                htb[tmp]++;
                tmp = "";
            }
            else tmp += c;
        }
        htb[tmp]++;
        vector<string> ans;
        for (unordered_map<string, int>::iterator it = htb.begin(); it != htb.end(); it++) {
            if (it->second == 1) 
                ans.push_back(it->first);
        }
        return ans;
    }
};
