class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        stringstream ss(A);
        string tmp;
        map<string, int> dp;
        while(ss>>tmp){
            dp[tmp]++;
        }
        stringstream sb(B);
        while(sb>>tmp){
            dp[tmp]++;
        }
        vector<string> res;
        for(auto it:dp){
            if(it.second == 1) res.push_back(it.first);
        }
        return res;
    }
};
