class Solution {
private:
    void count(string str, unordered_map<string, int>& cnt, unordered_set<string>& cache)
    {
        string tmp = "";
        for (int i = 0; i < str.size(); i++)
        {
            if (str[i] == ' ')
            {
                cnt[tmp]++;
                cache.insert(tmp);
                tmp = "";
            }
            else
            {
                tmp += str[i];
            }
        }
        cnt[tmp]++;
        cache.insert(tmp);
    }
public:
    vector<string> uncommonFromSentences(string A, string B) {
        vector<string> result;
        unordered_map<string, int> cnt;
        unordered_set<string> setA, setB;
        count(A, cnt, setA);
        count(B, cnt, setB);
        for (auto it : setA)
        {
            if (setB.find(it) == setB.end() && cnt.find(it) != cnt.end() && cnt[it] == 1)
                result.push_back(it);
        }
        for (auto it : setB)
        {
            if (setA.find(it) == setA.end() && cnt.find(it) != cnt.end() && cnt[it] == 1)
                result.push_back(it);
        }

        return result;
    }
};
