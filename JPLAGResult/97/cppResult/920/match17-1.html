class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        istringstream inA(A);
        

        map<string,int> m;
        string s;
        while(inA>>s) {
            m[s]++;
        }
        
        istringstream inB(B);
        while(inB>>s) {
            m[s]++;
        }
        
        vector<string> result;
        for(auto &el:m) {
            if(el.second==1) result.push_back(el.first);
        }
        
        return result;
    }
};
