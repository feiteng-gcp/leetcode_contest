class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        unordered_map<string, int> freqM;
        stringstream ssA(A);
        stringstream ssB(B);
        string token;
        while (!ssA.eof()) {
            getline(ssA, token, ' ');
            ++freqM[token];
            // cout << token << "\n";
        }
        while (!ssB.eof()) {
            getline(ssB, token, ' ');
            ++freqM[token];
            // cout << token << "\n";
        } 
        
        vector<string> res;
        for (auto& p : freqM) {
            if (p.second == 1) res.push_back(p.first);
        }
        return res;
    }
};
