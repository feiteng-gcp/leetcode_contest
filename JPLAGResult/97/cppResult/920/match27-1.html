class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        unordered_map<string, int> mA, mB;
        vector<string> res;
        string word = "";

        for (char x : A) {
            if (x == ' ') {
                mA[word] += 1;
                word = "";
            }
            else {
                word += x;
            }
        }
        mA[word] += 1;
        word = "";
        for (char x : B) {
            if (x == ' ') {
                mB[word] += 1;
                word = "";
            }
            else {
                word += x;
            }
        }
        mB[word] += 1;
        

        for (auto x : mA) {
            if (x.second == 1 && mB.find(x.first) == mB.end())
                res.push_back(x.first);
        }
        for (auto x : mB) {
            if (x.second == 1 && mA.find(x.first) == mA.end())
                res.push_back(x.first);
        }
        
        return res;
    }
};
