class Solution {
private:
    map<string, int> m;
public:
    vector<string> uncommonFromSentences(string A, string B) {
        vector<string> ans;
        if (A.length() == 0 && B.length() == 0) return ans;
        stringstream ss, ss2;
        ss << A;
        string tmp;

        while (ss >> tmp) {
            cout << tmp << endl;
            m[tmp]++;
        }
        ss2 << B;
        while (ss2 >> tmp) {
            cout << tmp << endl;
            m[tmp]++;
        }
        for (auto it : m) {
            if (it.second == 1) ans.push_back(it.first);
        }
        return ans;
    }
};
