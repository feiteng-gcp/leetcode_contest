class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        stringstream ss1(A), ss2(B);
        unordered_map<string, int> s1, s2;
        string tmp;
        while(ss1 >> tmp)
            s1[tmp]++;
        while(ss2 >> tmp)
            s2[tmp]++;
        
        vector<string> res;
        for(auto p: s1) {
            if(p.second == 1 && !s2.count(p.first))
                res.push_back(p.first);
        }
        for(auto p: s2) {
            if(p.second == 1 && !s1.count(p.first))
                res.push_back(p.first);
        }
        return res;
    }
};
