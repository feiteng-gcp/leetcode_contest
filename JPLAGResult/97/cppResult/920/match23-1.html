class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        map<string,int> a;
        map<string,int> b;
        vector<string> ans;
        string tmp;
        stringstream ssa(A);
        stringstream ssb(B);
        while(getline(ssa,tmp,' ')){
            a[tmp]++;
        }
        while(getline(ssb,tmp,' ')){
            b[tmp]++;
        }
        for(auto t:a){
            if(a[t.first]==1&&b.find(t.first)==b.end()) ans.push_back(t.first);
        }
        for(auto t:b){
            if(b[t.first]==1&&a.find(t.first)==a.end()) ans.push_back(t.first);
        }
        return ans;
    }
};
