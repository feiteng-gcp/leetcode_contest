class Solution {
private:
    unordered_map<string, int> str2mp(string s) {
        istringstream ss(s);
        string str;
        unordered_map<string, int> re;
        
        while(std::getline(ss, str, ' ')) {
            re[str]++;
        }
        
        return re;
    }
public:
    vector<string> uncommonFromSentences(string A, string B) {
        auto mpa = str2mp(A);
        auto mpb = str2mp(B);
        
        vector<string> re;
        
        for(auto p : mpa) {
            if(p.second == 1 && !mpb.count(p.first)) re.push_back(p.first);
        }
        
        for(auto p : mpb) {
            if(p.second == 1 && !mpa.count(p.first)) re.push_back(p.first);
        }
        
        return re;
    }
};
