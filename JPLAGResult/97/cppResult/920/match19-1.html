class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        unordered_map<string, int> a;
        stringstream ss;
        ss<<A;
        string cur;
        while(ss>>cur) {
            a[cur]++;
        }
        ss.clear();
        ss<<B;
        while(ss>>cur) {
            a[cur]++;
        }
        vector<string> ans;
        for(auto it = a.begin(); it != a.end(); it++) {
            if(it->second == 1) {
                ans.push_back(it->first);
            }
            
        }
        return ans;
    }
};
