class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        map<string,int> a;
        map<string,int> b;
        
        stringstream ss(A);
        stringstream bb(B);
        string input;
        while (ss >> input) {
            a[input]++;
        }
        
        while (bb >> input) {
            b[input]++;
        }
        vector<string> out;
        for (auto &p : a) {
            if (p.second == 1 && !b.count(p.first)) {
                out.push_back(p.first);
            }
        }
        
        swap(a, b);
        for (auto &p : a) {
            if (p.second == 1 && !b.count(p.first)) {
                out.push_back(p.first);
            }
        }
        
        return out;
    }
};
