class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        vector<string> res;
        
        unordered_map<string, int> ma, mb;
        stringstream sa(A);
        stringstream sb(B);
        string one;
        while(getline(sa, one, ' ')) ++ma[one];
        

        while(getline(sb, one, ' ')) ++mb[one];
        
        for(auto& s : ma) {
            if(s.second != 1) continue;
            if(mb.count(s.first) == 0) res.push_back(s.first);
        }
        
        for(auto& s : mb) {
            if(s.second != 1) continue;
            if(ma.count(s.first) == 0) res.push_back(s.first);
        }
        return res;
    }
};
