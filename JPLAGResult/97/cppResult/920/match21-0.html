class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        static regex re("\\s+");
        unordered_map<string, int> a, b;
        for (sregex_token_iterator p(A.begin(), A.end(), re, -1), e; p != e; ++p) ++a[p->str()];
        for (sregex_token_iterator p(B.begin(), B.end(), re, -1), e; p != e; ++p) ++b[p->str()];
        vector<string> res;
        for (auto& p : a) {
            if (p.second != 1) continue;
            if (b.count(p.first) == 0) res.push_back(p.first);
        }
        for (auto& p : b) {
            if (p.second != 1) continue;
            if (a.count(p.first) == 0) res.push_back(p.first);
        }
        return res;
    }
};
