class Solution {
public:
    vector<string> tok(string s) {
        istringstream in(s);
        vector<string> v;
        string word;
        while(in >> word) {
            v.push_back(word);
        }
        return v;
    }
    vector<string> uncommonFromSentences(string A, string B) {
        auto v1 = tok(A);
        auto v2 = tok(B);
        vector<string> res;
        unordered_map<string, int> m1, m2;
        for(auto i: v1) m1[i] += 1;

        for(auto i: v2) m2[i] += 1;
        for(auto i: m1) {
            if (i.second == 1 && m2.find(i.first) == m2.end()) res.push_back(i.first);
        }
        for(auto i: m2) {
            if (i.second == 1 && m1.find(i.first) == m1.end()) res.push_back(i.first);
        }
        return res;
    }
};
