class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        stringstream ssa(A), ssb(B);
        string tmp;
        map<string,int> m;
        while(ssa >> tmp){
            m[tmp] +=1;
        }
        while(ssb >> tmp){
            m[tmp] +=1;
        }
        vector<string> res;
        for(auto it: m){
            if(it.second == 1) res.push_back(it.first);
        }
        return res;
        
    }
};
