class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        map<string, int> cntA, cntB;
        int len = A.size();
        for (int i = 0, j; i < len; i = j) {
            for (; i < len && A[i] == ' '; ++i);
            if (i == len) break;
            for (j = i + 1; j < len && A[j] != ' '; ++j);
            cntA[A.substr(i, j - i)]++;
        }
        len = B.size();
        for (int i = 0, j; i < len; i = j) {
            for (; i < len && B[i] == ' '; ++i);
            if (i == len) break;
            for (j = i + 1; j < len && B[j] != ' '; ++j);
            cntB[B.substr(i, j - i)]++;
        }
        vector<string> ret;

        for (auto& it : cntA) {
            if (it.second > 1) continue;
            if (!cntB.count(it.first)) ret.push_back(it.first);
        }
        for (auto& it : cntB) {
            if (it.second > 1) continue;
            if (!cntA.count(it.first)) ret.push_back(it.first);
        }
        return ret;
    }
};
