class Solution {
public:
    vector<string> uncommonFromSentences(string A, string B) {
        istringstream iss(A + " " + B);
        map<string, int> count;
        string s;
        while(iss >> s) {
            ++count[s];
        }
        vector<string> ans;
        for(map<string,int>::iterator it = count.begin(); it != count.end(); ++it) {
            if(it->second == 1) {
                ans.push_back(it->first);
            }
        }
        return ans;
    }
};
