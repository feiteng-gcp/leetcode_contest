class Solution {
public:
	bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
		vector<vector<int>> graph(N, vector<int>());
		for (auto it : dislikes) {
			graph[it[0]-1].push_back(it[1]-1);
			graph[it[1]-1].push_back(it[0]-1);
		}
		return isBipartite(graph);
	}

	bool isBipartite(vector<vector<int>>& graph) {
		int i = 0, n = graph.size();
		vector<int> color(n, -1);
		vector<pair<int, int>> vec;
		vector<unordered_set<int>> dom(n, { 0, 1 });
		for (auto v : graph)
			vec.push_back({ -(int)v.size(), i }), i++;
		sort(vec.begin(), vec.end());
		dom[vec[0].second].erase(1);
		return canColor(0, n, vec, graph, color, dom);
	}
private:
	bool canColor(int idx, int n, vector<pair<int, int>>& vec, vector<vector<int>>& graph, vector<int>& color, vector<unordered_set<int>>& dom) {
		if (idx == n) return true;
		int idx2 = vec[idx].second;
		if (graph[idx2].empty()) return true;
		for (int i : dom[idx2]) {
			color[idx2] = i;
			bool flag = true;
			unordered_set<int> st;
			for (int j : graph[idx2]) {
				if (dom[j].count(i)) {
					dom[j].erase(i);
					st.insert(j);
					if (dom[j].empty()) {
						flag = false;
						break;
					}
				}
			}
			if (flag && canColor(idx + 1, n, vec, graph, color, dom))
				return true;
			color[idx2] = -1;
			for (int j : st)
				dom[j].insert(i);
		}
		return false;
	}
};
