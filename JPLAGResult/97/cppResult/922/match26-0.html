class Solution {
    vector<vector<int>> g;
    vector<int> color;
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        g=vector<vector<int>>(N+1);
        color=vector<int>(N+1,-1);
        for(auto item:dislikes){
            g[item[0]].push_back(item[1]);
            g[item[1]].push_back(item[0]);
        }
        
        for(int i=0;i<dislikes.size();i++){
            int a=dislikes[i][0];
            int b=dislikes[i][1];
            if(color[a]==-1&&color[b]==-1){
                color[a]=0;
                color[b]=1;
                if(!dfs(a)) return false;
                if(!dfs(b)) return false;
            }
        }
        return true;
    }
    
    bool dfs(int n){
        queue<int> qu;
        qu.push(n);
        while(!qu.empty()){
            int a=qu.front();
            qu.pop();
            for(int b:g[a]){
                if(color[b]==-1){
                    color[b]=1-color[a];
                    qu.push(b);
                }else{
                    if(color[b]==color[a]) return false;
                }
            }
        }
        return true;
    }
};
