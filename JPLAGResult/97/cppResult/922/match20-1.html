class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        vector<vector<int> > g(N+1);
        int color[2001] = {0};
        for(auto e : dislikes) {
            int u = e[0], v = e[1];
            g[u].push_back(v);
            g[v].push_back(u);
        }
        // bfs with colors!
        queue<pair<int,int>> q;
        color[1] = 2;
        q.push({1,1});

        while(!q.empty()) {
            int u = q.front().first;
            int c = q.front().second;
            q.pop();
            cout << "u " << u << endl;
            for(auto v : g[u])  {
                if(color[v] == 0) {
                    color[v] = c;
                    //cout << "c " << v << " " << c << endl;
                    q.push({v, (c==1) ? 2:1});
                } else if (color[v] != c) {
                    //cout << v <<" "<< color[v] << " " << c << endl;
                    return false;
                }
            }
        }
        // if visited already, return false
        // return true
        return true;
    }
};
