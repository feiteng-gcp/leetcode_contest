class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        vector<vector<int>> graph( N );
        vector<int> groups( N, -1 );
        for( auto& dislike : dislikes )
        {
            int p1 = dislike[0]-1;
            int p2 = dislike[1]-1;
            graph[p1].push_back(p2);
            graph[p2].push_back(p1);
        }
        

        for( int i = 0; i < N; i++ )
        {
            if( groups[i] == -1 )
            {
                if( !dfs( graph, groups, i, 0 ) )
                    return false;
            }
        }
        
        return true;
    }
    
    bool dfs( vector<vector<int>>& graph, vector<int>& groups, int i, int g )
    {
        if( groups[i] == g )
            return true;

        if( groups[i] != -1 && groups[i] != g )
            return false;
        
        groups[i] = g;
        for( auto& n : graph[i] )
            if( !dfs(graph, groups, n, 1 - g ) )
                return false;
        
        return true;
    }
};
