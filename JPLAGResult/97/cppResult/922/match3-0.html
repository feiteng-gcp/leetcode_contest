class Solution {
    bool test(int x, vector<vector<int>>& g, vector<int>& grp, int gid) {
        // cout << x << " " << gid << " " << grp[x] << endl;
        if (grp[x] >= 0) {
            return grp[x] == gid;
        }
        grp[x] = gid;
        for (int y : g[x]) {
            if (x == y) continue;
            if (!test(y, g, grp, 1 - gid)) return false;
        }
        return true;
    }
public:
    bool possibleBipartition(int n, vector<vector<int>>& dislikes) {
        vector<vector<int>> g(n + 1);
        for (auto p : dislikes) {
            g[p[0]].push_back(p[1]);
            g[p[1]].push_back(p[0]);
        }
        // for (int x = 1; x <= n; ++x, cout << endl) for (int y : dislikes[x]) cout << x << "- " << y << endl;
        vector<int> grp(n + 10, -1);
        for (int x = 1; x <= n; ++x) {
            if (grp[x] < 0) {
                // cout << "testing " << x << endl;
                if (!test(x, g, grp, 0)) return false;
            }
        }
        return true;
    }
};
