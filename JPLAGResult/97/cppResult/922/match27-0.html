class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        if (dislikes.size() == 0)
            return true;
        vector<int> group(N+1, -1);
        unordered_map<int, vector<int>> adj;
        for(int i = 0; i < dislikes.size(); i++) {
            adj[dislikes[i][0]].push_back(dislikes[i][1]);
            adj[dislikes[i][1]].push_back(dislikes[i][0]);
        }
        queue<int> q;
        group[adj.begin()->first] = 0;
        q.push(adj.begin()->first);
        while(!q.empty()){
            int n = q.size();
            for(int i = 0; i < n; i++) {
                int k = q.front();
                q.pop();
                if (adj.find(k) != adj.end()) {
                    for(int j : adj[k]) {
                        if (group[j] == -1) {
                            group[j] = 1 - group[k];
                            q.push(j);
                        }
                        else if (group[j] == group[k]) {
                        return false;
                    }
                }
                }
            }
        }
        return true;
    }
};
