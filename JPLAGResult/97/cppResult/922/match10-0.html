class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        unordered_map<int, unordered_set<int>> graph;
        for (auto dislike : dislikes) {
            graph[dislike[0]].insert(dislike[1]);
            graph[dislike[1]].insert(dislike[0]);
        }
        set<int> candidates;
        for (int i = 1; i <= N; ++i) {
            candidates.insert(i);
        }
        vector<int> color(N + 1, -1);
        while (!candidates.empty()) {
            int i = *candidates.begin();
            color[i] = 1;
            queue<int> q;
            q.push(i);
            while (!q.empty()) {
                int cur = q.front();
                q.pop();
                candidates.erase(cur);
                for (auto j : graph[cur]) {
                    if (color[j] == color[cur]) return false;
                    else if (color[j] == -1) {
                        color[j] = 1 - color[cur];
                        q.push(j);
                    }
                }
            }
        }
        return true;
    }
};
