class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        unordered_set<int> done;
        unordered_map<int, bool> assigned;
        unordered_map<int, unordered_set<int>> g;
        for (const auto& d : dislikes) {
            g[d[0]].insert(d[1]);
            g[d[1]].insert(d[0]);
            done.insert(d[0]);
            done.insert(d[1]);
        }
        while (!done.empty()) {
            auto sn = *done.begin();
            done.erase(done.begin());
            queue<int> q;
            q.push(sn);
            assigned[sn] = true;
            while(!q.empty()) {
                sn = q.front();
                q.pop();
                for (const auto& nn : g[sn]) {
                    const auto it = done.find(nn);

                    if (it != done.end()) {
                        done.erase(nn);
                        assigned[nn] = !assigned[sn]; 
                        q.push(nn);
                    }
                    else if (assigned[nn] != !assigned[sn]) {
                        return false;
                    }
                }
            }
        }
        return true;
    }
};
