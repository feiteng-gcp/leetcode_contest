class Solution {
public:
	bool dfs(int u, vector<int> &color, vector<vector<int>> &g)
	{
		vector<int> nx;
		for (int v : g[u])
		{
			if (color[v] == 0)
			{
				color[v] = -color[u];
				nx.push_back(v);
			}
			else
				if (color[v] == color[u])
					return false;
		}
		for (auto v : nx)
		{
			if (!dfs(v, color, g))
				return false;
		}
		return true;
	}

	bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
		vector<vector<int>> g(N + 1);
		for (auto &e : dislikes)
		{
			int a = e[0], b = e[1];
			g[a].push_back(b);
			g[b].push_back(a);
		}
		vector<int> color(N + 1);

		for (int i = 1; i <= N; ++i)
		{
			if (color[i] == 0)
			{
				color[i] = 1;
				if (!dfs(i, color, g)) return false;
			}
		}
		return true;
	}
};
