class Solution {
public:
    int n;
    vector < int > vis;
    vector < vector < int > > A;
    set < int > S[2];
    
    void dfs(int node, bool flip) {
        vis[node] = true;
        for(int i = 0;i < A[node].size();++i) {
            int to = A[node][i];
            S[flip ^ 1].insert(to);
            if(!vis[to]) {
                dfs(to, flip ^ 1);
            }
        }
    }
    
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        n = N;
        if(dislikes.size() < 2) {
            return true;
        }
        A.resize(n + 5);
        vis.resize(n + 5);
        for(int i = 0;i < dislikes.size();++i) {
            int from = dislikes[i][0] - 1;
            int to = dislikes[i][1] - 1;
            A[from].push_back(to);
            A[to].push_back(from);
        }
        for(int i = 0;i < n;++i) {
            if(!vis[i]) {
                S[0].insert(i);   
                dfs(i, 0);
            }
        }
        
        for(auto it = S[0].begin();it != S[0].end();++it) {
            if(S[1].find(*it) != S[1].end()) {
                return false;
            }
        }
        return true;
    }
};
