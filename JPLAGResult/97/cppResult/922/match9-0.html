class Solution {
public:
    bool tryColor(vector<vector<int>>& adj_list, vector<int>& color, int x) {
        for (const auto& adj : adj_list[x]) {
            if (color[x] == color[adj]) {
                return false;
            }
            if (color[adj] == 0) {
                color[adj] = -color[x];
                if (!tryColor(adj_list, color, adj)) {
                    return false;
                }
            }
        }
        return true;
    }
    
    bool possibleBipartition(int n, vector<vector<int>>& dislikes) {
        vector<vector<int>> adj_list(n);
        for (const auto& edge : dislikes) {
            adj_list[edge[0] - 1].push_back(edge[1] - 1);
            adj_list[edge[1] - 1].push_back(edge[0] - 1);
        }
        vector<int> color(n, 0);
        for (int i = 0; i < n; i++) {
            if (color[i] == 0) {
                color[i] = 1;
                if (!tryColor(adj_list, color, 0)) {
                    return false;
                }
            }
        }
        return true;
    }
};
