class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        graph = vector<vector<int>>(N + 1, vector<int>{} ); // adjacency list

        tags = vector<int>(N + 1, -1); // tag all nodes as "untagged"
        for(auto &p : dislikes) {
            graph[p[0]].push_back(p[1]);
            graph[p[1]].push_back(p[0]);
        }
        for(int i = 1; i <= N; ++i) { // graph may be disconnected
            if(tags[i] == -1 && !addTag(i, 0)) { // tag each untagged node as group A
                return false; // false if fail to tag some node
            }
        }
        return true;
    }
    bool addTag(int i, int tag) {
        if(tags[i] == -1) { // untagged
            tags[i] = tag;
            for(auto &j : graph[i]) { // for each i's neighbor
                if(!addTag(j, 1 - tag)) {
                    return false;
                }
            }
        } else { // tagged, check if tag align
            return tags[i] == tag;
        }
        return true;
    }
private:
    vector<vector<int>> graph;
    vector<int> tags;
};
