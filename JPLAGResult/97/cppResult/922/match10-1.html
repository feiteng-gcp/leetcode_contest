class Solution {
public:
    bool dfs(int k, vector<int>& color, vector<vector<int>>& list)
    {
        int c = color[k];
        
        vector<int>& others = list[k];
        for (int o : others)
        {
        //    cout << k << " " << c << " " << o << " " << color[o] << endl;
            if (color[o] == c) return false;
            if (color[o] == 0)
            {
                color[o] = ((c == 1) ? 2 : 1);
                if (!dfs(o, color, list)) return false;
            }
        }
        
        return true;
    }
    

    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        vector<vector<int>> list(N, vector<int>());
        for (auto& dis : dislikes)
        {
            list[dis[0]-1].push_back(dis[1]-1);
            list[dis[1]-1].push_back(dis[0]-1);
        }
        
        vector<int> color(N, 0);
        
        for (int i = 0; i < N; i++)
        {
            if (color[i] == 0) 
            {
                color[i] = 1;
                if (!dfs(i, color, list))
                {
                    return false;
                }
            }
        }
        
        return true;
    }
};
