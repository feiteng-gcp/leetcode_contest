class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        vector<int> visited(N + 1, 0);
        unordered_map<int, unordered_set<int>> m;
        for(const auto& v : dislikes)
        {
            m[v[0]].insert(v[1]);
            m[v[1]].insert(v[0]);
        }


        for(int i = 1; i <= N; ++i)
        {
            if(visited[i] == 0)
            {
                if(!dfs(visited, i, m, 1)) return false;
            }
        }
        return true;
    }

    bool dfs(vector<int>& visited, int i, unordered_map<int, unordered_set<int>>& m, int value)
    {
        if(visited[i] * value == -1) return false;
        if(visited[i] == value) return true;
        visited[i] = value;

        for(const auto& dis : m[i])
        {
            if(!dfs(visited, dis, m, -value)) return false;
        }
        return true;
    }
};
