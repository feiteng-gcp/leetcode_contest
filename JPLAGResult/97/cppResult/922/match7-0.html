class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        if (N <= 2 || dislikes.size() == 0) return true;
        vector<int> groupNum(N + 1, 0);
        vector<unordered_set<int>> v(N + 1, unordered_set<int>());
        for (auto dislike : dislikes) {
            int p1 = dislike[0];
            int p2 = dislike[1];
            v[p1].insert(p2);
            v[p2].insert(p1);
        }
        queue<int> q;
        // cout << "triggered1" << endl;
        for (int index = 0; index < N + 1; ++index) {
            if (v[index].size() == 0 || groupNum[index] != 0) continue;
            unordered_set<int> toBeChecked = v[index];
            groupNum[index] = 1;
            for (auto mem : toBeChecked) {
                if (groupNum[mem] == 0) {
                    groupNum[mem] = -1;
                    q.push(mem);
                } else if (groupNum[mem] == 1) {
                    return false;
                }
            }
            // cout << "triggered2" << endl;
            while (!q.empty()) {
                int indexChecking = q.front();
                // cout << "checking" << indexChecking << endl;
                toBeChecked = v[indexChecking];
                q.pop();
                for (auto mem : toBeChecked) {
                    if (groupNum[mem] == 0) {
                        groupNum[mem] = -groupNum[indexChecking];
                        q.push(mem);
                    } else if (groupNum[mem] == groupNum[indexChecking]) {
                        return false;
                    }
                }
            }
        }
        return true;
    }
};
