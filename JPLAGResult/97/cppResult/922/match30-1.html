class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        vector<int> color(N + 1, -1);
        vector<vector<int>> g(N + 1);
        for (auto & edge : dislikes) {
            g[edge[0]].push_back(edge[1]);
            g[edge[1]].push_back(edge[0]);
        }

        int s = 0;
        for (int i = 1; i <= N; ++i) {
            if (g[i].empty() == false) {
                s = i;
                break;
            }
        }
        
        queue<int> q;
        color[s] = 0;
        q.push(s);
        while (q.empty() == false) {
            int u = q.front(); q.pop();
            int uc = color[u];
            for (auto v : g[u]) {
                if (color[v] == -1) {
                    q.push(v);
                    color[v] = 1 - uc;
                } else if (color[v] != 1 - uc) {
                    return false;
                }
            }
        }
        return true;
    }
};
