class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& ds) {
        int s[N + 1] = {};    
        vector<unordered_set<int>> adj(N + 1);
        for(auto& d : ds) {
            adj[d[0]].insert(d[1]);
            adj[d[1]].insert(d[0]);
        }
        
        for(int i = 1; i <= N; ++i) {
            if(s[i] == 0 && !dfs(adj, i, s, 1)) return false;
        }
        return true;
    }
    
    bool dfs(const vector<unordered_set<int>>& g, int from, int s[], int c) {
        if(s[from] != 0) return s[from] == c;
        s[from] = c;
        
        for(int to : g[from]) {
            if(!dfs(g, to, s, 3 - c)) return false;
        }
        return true;
    }
};
