class Solution {
public:
    bool possibleBipartition(int n, vector<vector<int>>& dislikes) {
        vector<vector<int>> adj(n);
        for(auto &d : dislikes){
            d[0]--;
            d[1]--;
            adj[d[0]].push_back(d[1]);
            adj[d[1]].push_back(d[0]);
        }
        vector<bool> visited(n, false);
        vector<int> group(n, -1);
        for(int i=0;i<n;i++){
            if(group[i] != -1) continue;
            int g = 0;
            group[i] = 1 - g;
            queue<int> que; que.push(i);
            while(!que.empty()){
                int sz = que.size();
                while(sz -- > 0){
                    auto cur = que.front();
                    que.pop();
                    for(int i : adj[cur]){
                        if(group[i] != -1 && group[i] != g) return false;
                        if(group[i] != -1) continue;
                        group[i] = g;
                        que.push(i);
                    }
                }
                g = 1 - g;
            }
        }
        
        return true;
    }
};
