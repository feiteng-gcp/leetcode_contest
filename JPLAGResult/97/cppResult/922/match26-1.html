class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        // check if a graph is bipartite by 2-coloring;
        vector<int> color(N+1, -1);
        vector<unordered_set<int>> dis(N+1);
        for(auto v: dislikes) {
            dis[v[0]].insert(v[1]);
            dis[v[1]].insert(v[0]);
            if(v[0] == v[1])
                return false;
        }
        queue<int> q; q.push(1);
        color[1] = 1;

        while(!q.empty()) {
            int cur = q.front(); q.pop();
            for(int next: dis[cur]) {
                if(color[next] == -1) {
                    color[next] = 1-color[cur];
                    q.push(next);
                }
                else if(color[next] == color[cur])
                    return false;
            }
        }
        return true;
    }
};
