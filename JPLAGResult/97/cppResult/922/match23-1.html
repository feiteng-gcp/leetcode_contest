class Solution {
private:
    vector<int> pos;
    vector<vector<int>> dis;
    int n;
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        n = N;
        dis = vector<vector<int> >(N);
        pos = vector<int>(N, -1);
        for(auto& d: dislikes) {
            dis[d[0] - 1].push_back(d[1] - 1);
            dis[d[1] - 1].push_back(d[0] - 1);
        }
        for(int i = 0; i < n; ++i) {
            if(pos[i] != -1) {
                continue;
            }
            pos[i] = 0;
            queue<pair<int, int>> q;
            q.push(make_pair(i, 0));
            while(!q.empty()) {
                auto top = q.front();
                q.pop();
                int next_color = 1 - top.second;

                for(auto next: dis[top.first]) {
                    if(pos[next] == -1) {
                        pos[next] = next_color;
                        q.push(make_pair(next, next_color));
                    } else if(pos[next] != next_color) {
                        return false;
                    }
                }
            }
        }
        return true;
    }
};
