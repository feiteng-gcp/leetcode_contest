class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
         if(dislikes.size() == 0){
            return true;
         }
         set<int>A,B;
         map<int,vector<int>> csr;
         for(int i = 0; i < dislikes.size(); i++){
             csr[dislikes[i][0]].push_back(dislikes[i][1]);
             csr[dislikes[i][1]].push_back(dislikes[i][0]);
         }
         while(1){
            int notq = -1;
            for(int i = 1; i <= N; i++){
                if(A.find(i) == A.end() && B.find(i) == B.end() && csr.find(i) != csr.end()){
                   notq = i;
                   break;
                }
            }
            vector<int> nnode;
            if(notq == -1){
               return true;
            }
            A.insert(notq);
            for(vector<int>::iterator it = csr[notq].begin(); it != csr[notq].end(); it++){
                B.insert(*it);
                nnode.push_back(*it);
            }
            swap(A,B);
            while(!nnode.empty()){
                vector<int> frontier;
                for(vector<int>::iterator it = nnode.begin(); it != nnode.end(); it++){
                    if(csr.find(*it) != csr.end()){
                       for(vector<int>::iterator itv = csr[*it].begin(); itv != csr[*it].end(); itv++){
                           if(A.find(*itv) != A.end()){
                              return false;
                           }
                           else if(B.find(*itv) == B.end()){
                              B.insert(*itv);
                              frontier.push_back(*itv);
                           }
                       }
                    }
                }
                swap(frontier,nnode);
                frontier.clear();
                swap(A,B);
            }
         }
    }
};
