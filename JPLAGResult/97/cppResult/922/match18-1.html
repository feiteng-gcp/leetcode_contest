class Solution {
public:
    bool possibleBipartition(int N, vector<vector<int>>& dislikes) {
        vector<int> g[N+1];
        for (auto &elem : dislikes) {
            int a= elem[0];
            int b = elem[1];
            g[a].push_back(b);
            g[b].push_back(a);
        }
        queue<int> q;

        vector<int> color(N+1, -1);
        q.push(1);
        color[1] = 0;
        while (!q.empty()) {
            int top = q.front();
            q.pop();
            int c = color[top];
            for (auto &nei : g[top]) {
                if (color[nei] == c) {
                    return false;
                }
                if (color[nei] == -1) {
                    q.push(nei);
                }
                color[nei] = 1 - c;
            }
        }
        return true;
    }
};
