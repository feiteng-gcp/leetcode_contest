typedef pair<int, pair<int, int>> Node;
class Solution {
public:
    int minCost(vector<vector<int>>& g) {
      int m = g.size();
      int n = g[0].size();

      int inf = 1e9;
      vector<vector<int>> dp(m, vector<int> (n, inf));
      priority_queue<Node, std::vector<Node>, std::greater<Node>> q;
      q.push({0, {0, 0}});
      dp[0][0] = 0;

      vector<int> stepx= {0, 0, 1, -1};
      vector<int> stepy= {1, -1, 0, 0};
      while (!q.empty()) {
        auto c = q.top();
        q.pop();
        int v = c.first;
        int x = c.second.first;
        int y = c.second.second;
        if (dp[x][y] < v)
          continue;
        for (int i = 0; i < 4; ++i) {
          int nx = x + stepx[i];
          int ny = y + stepy[i];
          if (nx < 0 || nx >= m || ny < 0 || ny >= n)
            continue;
          int nv = v + (i + 1 == g[x][y] ? 0 : 1);
          if (nv < dp[nx][ny]) {
            dp[nx][ny] = nv;
            q.push({nv, {nx, ny}});
          }
        }
      }
      return dp[m - 1][n - 1];
    }
};
