class Solution {
public:
    vector<vector<int>> direct = {{0, 1},{0, -1},{1, 0},{-1, 0}};

    int minCost(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        queue<vector<int>> q;
        vector<vector<int>> vis(m, vector<int>(n, -1));
        q.push({0, 0});
        vis[0][0] = 0;
        while(!q.empty()) {
            auto v = q.front(); q.pop();
            // cout<<v[0]<<" "<<v[1]<<" "<<vis[v[0]][v[1]]<<endl;

            int tmp = grid[v[0]][v[1]];
            for(int i=0;i<4;i++) {
                int x = v[0]+direct[i][0], y=v[1]+direct[i][1];
                if(x<0||x>=m||y<0||y>=n) continue;
                int ans = vis[v[0]][v[1]]+(i+1==tmp?0:1);
                if(vis[x][y]==-1||vis[x][y]>ans) {
                    vis[x][y]=ans; q.push({x, y});
                }
            }
        }
        return vis[m-1][n-1];
    }
};
