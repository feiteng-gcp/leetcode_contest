class Solution {
public:
    int dx[5] = {0, 0, 0, 1, -1};
    int dy[5] = {0, 1, -1, 0, 0};
    int minCost(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<int> dist(n*m, INT_MAX);
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
        dist[0] = 0;
        pq.push(make_pair(0, 0));
        while(!pq.empty()) {
            int d = pq.top().first;
            int pos = pq.top().second;
            pq.pop();
            if(dist[pos] < d) continue;
            int x = pos/m, y = pos%m;
            for(int i = 1; i <= 4; i++) {
                int nx = x+dx[i], ny = y+dy[i];
                if(nx >= 0 && nx < n && ny >= 0 && ny < m) {
                    int cost = (i!=grid[x][y]);
                    if(dist[nx*m+ny] > dist[pos]+cost) {
                        dist[nx*m+ny] = dist[pos]+cost;
                        pq.push(make_pair(dist[nx*m+ny], nx*m+ny));
                    }
                }
            }
        }
        return dist[(n-1)*m+m-1];
    }
};
