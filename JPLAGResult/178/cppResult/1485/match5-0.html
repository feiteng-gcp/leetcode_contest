int dx[5]={0,0,0,1,-1};
int dy[5]={0,1,-1,0,0};

class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        vector<vector<int>> dist(0);
        int m=grid.size();
        if (m==0) return 0;
        int n=grid[0].size();
        if (n<=1 && m<=1) return 0;
        
        for (int i=0;i<m;i++)
        {
            dist.push_back(vector<int>(0));
            for (int j=0;j<n;j++)
            {
                dist[i].push_back(100000);
            }
        }
        dist[0][0]=0;
        deque<int>qx(0);
        deque<int>qy(0);
        qx.push_back(0);
        qy.push_back(0);
        while (!qx.empty())
        {
            int x=qx.front();
            int y=qy.front();
            //cout<<x<<' '<<y<<' '<<dist[x][y]<<endl;
            qx.pop_front();
            qy.pop_front();
            for (int i=1;i<=4;i++)
            {
                int X=x+dx[i];
                int Y=y+dy[i];
                if (X<0 || X>=m || Y<0 || Y>=n) continue;
                //cout<<"safe"<<' '<<i<<endl;
                int cost = dist[x][y];
                
                if (i!=grid[x][y]) cost++;
                if (cost<dist[X][Y])
                {
                    dist[X][Y]=cost;
                    qx.push_back(X);
                    qy.push_back(Y);
                }
                //cout<<"safe"<<' '<<i<<endl;
            }
        }
        return dist[m-1][n-1];

    }
};
