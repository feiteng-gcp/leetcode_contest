class Solution {
public:
    int minCost(vector<vector<int>>& grid) {

            // 1 0,1   1,2
            // 2 0,-1  1,0
            // 3 1,0   2,1
            // 4 -1,0  0,1

        vector<vector<int>> next({{0,0},{0,1},{0,-1},{1,0},{-1,0}});

        int m = grid.size(), n = grid[0].size();
        vector<vector<int>> f(grid.size(),vector<int>(grid[0].size(),INT_MAX));
        deque<pair<int,int>> dq;
        f[0][0] = 0;
        dq.push_back({0,0});
        while (!dq.empty()){
            auto p = dq.front();
            dq.pop_front();
            int x = p.first;
            int y = p.second;
            for (int k = 1; k <= 4;k++){
                int dx = next[k][0];
                int dy = next[k][1];
                int c = (k == grid[x][y]) ? f[x][y] : f[x][y] + 1;
                int nx = x + dx;

                int ny = y + dy;
                if (nx < 0 || nx >= m || ny < 0 || ny >= n)
                    continue;
                if (f[nx][ny] > c){
                    f[nx][ny] = c;
                    dq.push_back({nx,ny});
                }
            }
        }
        return f[m-1][n-1];
    }
};
