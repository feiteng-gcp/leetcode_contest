class Solution {
public:
    int dp[101][101];
    int minCost(vector<vector<int>>& grid) {
        for (int i = 0; i < 101; i++) for (int j = 0; j < 101; j++) dp[i][j] = INT_MAX;
        dp[0][0] = 0;

        set<vector<int>> pq;
        
        pq.insert({0, 0, 0});
        
        while (!pq.empty()) {
            auto x = *pq.begin();pq.erase(x);
            int dir[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
            for (int i = 0; i < 4; i++) {
                int nx = dir[i][0] + x[1];
                int ny = dir[i][1] + x[2];
                if (nx < 0 || ny < 0 || nx >= grid.size() || ny >= grid[0].size()) continue;
                if (dp[nx][ny] > dp[x[1]][x[2]] + (grid[x[1]][x[2]] != i + 1)) {
                    dp[nx][ny] = dp[x[1]][x[2]] + (grid[x[1]][x[2]] != i + 1);
                    pq.insert({dp[nx][ny], nx, ny});
                }
            }
        }
        return dp[grid.size() - 1][grid[0].size() - 1];
    }
};
