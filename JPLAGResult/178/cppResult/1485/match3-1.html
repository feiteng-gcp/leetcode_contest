struct Cell
{
  int x, y;
  int cost;
};

bool operator <(Cell a, Cell b)
{
  return a.cost > b.cost;
}

class Solution
{
  private:

  bool valid(int x, int y, int n, int m)
  {
    return x >= 0 && x < n && y >= 0 && y < m;
  }

  int solve(vector<vector<int>> &a)
  {
    priority_queue<Cell> Q;
    Cell cell, next;

    int n = a.size();
    int m = a[0].size();
    bool visited[105][105][105] = {0};

    cell.x = 0;
    cell.y = 0;
    cell.cost = 0;
    Q.push(cell);
    while(!Q.empty())
    {
      cell = Q.top();
      Q.pop();

      //cout << cell.x << " " << cell.y << " " << cell.cost << endl;
      if(cell.x == n - 1 && cell.y == m - 1)
      {
        return cell.cost;
      }

      if(a[cell.x][cell.y] == 1)
      {      
        next.x = cell.x;
        next.y = cell.y + 1;
        next.cost = cell.cost;
        if(valid(next.x, next.y, n, m))
        {
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x, cell.y - 1, n, m))
        {
          next.x = cell.x;
          next.y = cell.y - 1;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x - 1, cell.y, n, m))
        {
          next.x = cell.x - 1;
          next.y = cell.y;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x + 1, cell.y, n, m))
        {
          next.x = cell.x + 1;
          next.y = cell.y;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
      }
      else if(a[cell.x][cell.y] == 2)
      {      
        next.x = cell.x;
        next.y = cell.y - 1;
        next.cost = cell.cost;
        if(valid(next.x, next.y, n, m))
        {
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x, cell.y + 1, n, m))
        {
          next.x = cell.x;
          next.y = cell.y + 1;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x - 1, cell.y, n, m))
        {
          next.x = cell.x - 1;
          next.y = cell.y;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x + 1, cell.y, n, m))
        {
          next.x = cell.x + 1;
          next.y = cell.y;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
      }
      else if(a[cell.x][cell.y] == 3)
      {      
        next.x = cell.x + 1;
        next.y = cell.y;
        next.cost = cell.cost;
        if(valid(next.x, next.y, n, m))
        {
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x - 1, cell.y, n, m))
        {
          next.x = cell.x - 1;
          next.y = cell.y;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x, cell.y - 1, n, m))
        {
          next.x = cell.x;
          next.y = cell.y - 1;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x, cell.y + 1, n, m))
        {
          next.x = cell.x;
          next.y = cell.y + 1;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
      }
      else
      {      
        next.x = cell.x - 1;
        next.y = cell.y;
        next.cost = cell.cost;
        if(valid(next.x, next.y, n, m))
        {
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x + 1, cell.y, n, m))
        {
          next.x = cell.x + 1;
          next.y = cell.y;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x, cell.y - 1, n, m))
        {
          next.x = cell.x;
          next.y = cell.y - 1;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
        if(valid(cell.x, cell.y + 1, n, m))
        {
          next.x = cell.x;
          next.y = cell.y + 1;
          next.cost = cell.cost + 1;
          if(!visited[next.x][next.y][next.cost])
          {
            visited[next.x][next.y][next.cost] = true;
            Q.push(next);
          }
        }
      }
    }

    return -1;
  }

  public:

  int minCost(vector<vector<int>> &a)
  {
    return solve(a);
  }
};
