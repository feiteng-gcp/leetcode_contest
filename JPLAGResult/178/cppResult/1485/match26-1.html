class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        int u[] = {0, 0, 1, -1};
        int v[] = {1, -1, 0, 0};
        
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<int>> cost(n, vector<int>(m, INT_MAX / 2));
        vector<vector<int>> vis(n, vector<int>(m, false));
        deque<vector<int>> que;
        cost[n-1][m-1] = 0;
        que.push_back({n-1, m-1});
        vis[n-1][m-1] = true;
        while (!que.empty()) {
            int r = que.front()[0];
            int c = que.front()[1];
            que.pop_front();

            vis[r][c] = false;
            for (int i=0; i<4; ++i) {
                int tr = r + u[i];
                int tc = c + v[i];
                if (0 <= tr && tr < n && 0 <= tc && tc < m) {
                    int t = grid[tr][tc] - 1;
                    int tmp_cost = cost[r][c];
                    if (tr + u[t] != r || tc + v[t] != c) ++tmp_cost;
                    if (tmp_cost < cost[tr][tc]) {
                        cost[tr][tc] = tmp_cost;
                        if (!vis[tr][tc]) {
                            vis[tr][tc] = true;
                            que.push_back({tr, tc});
                        }
                    }
                }
            }
        }
        return cost[0][0];
    }
};
