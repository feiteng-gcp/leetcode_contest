bool inq[110][110];
const int cx[] = {0, 0, 0, 1, -1};
const int cy[] = {0, 1, -1, 0, 0};
int res[110][110];

class Solution {
public:
    bool isNext(int x, int y, int val, int a, int b) {
        int h = x + cx[val];
        int k = y + cy[val];
        return (h == a) && (k == b);
    }
    int minCost(vector<vector<int>>& grid) {
        vector<pair<int, int> > q;
        memset(inq, false, sizeof(inq));
        q.push_back(make_pair(0, 0));
        for (int i = 0; i < grid.size(); ++i) {
            for (int j = 0; j < grid[0].size(); ++j) {
                res[i][j] = 1000000000;
            }
        }
        inq[0][0] = true;
        res[0][0] = 0;
        int fir = 0;
        while (fir < q.size()) {
            pair<int, int> e = q[fir];
            ++fir;
            int x = e.first;
            int y = e.second;
            inq[x][y] = false;
            for (int i = 1; i <= 4; ++i) {
                int h = x + cx[i];
                int k = y + cy[i];
                if (h >= 0 && h < grid.size() && k >= 0 && k < grid[0].size()) {
                    int c = 1;
                    if (isNext(x, y, grid[x][y], h, k)) c = 0;
                    
                    if (res[h][k] > res[x][y] + c) {
                        res[h][k] = res[x][y] + c;
                        if (!inq[h][k]) {
                            q.push_back(make_pair(h, k));
                            inq[h][k] = true;
                        }
                    }
                }
            }
        }
        return res[grid.size() - 1][grid[0].size() - 1];
    }
};
