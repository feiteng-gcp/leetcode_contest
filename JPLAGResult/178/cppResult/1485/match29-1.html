class Solution {
public:
    int dx[4] = {0, 0, 1, -1};
    int dy[4] = {1, -1, 0, 0};
    
    int minCost(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        vector <pair <int, int> > v[10005];
        vector <int> dist(n * m + 1, 1e9);
        priority_queue <pair <int, int> > q;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int d = 0; d < 4; d++) {
                    int x = i + dx[d], y = j + dy[d];
                    if (x < 0 || x >= n || y < 0 || y >= m) {
                        continue;
                    }
                    int aa = i * m + j, bb = x * m + y;
                    if (a[i][j] == d + 1) {
                        v[bb].push_back(make_pair(aa, 0));
                    } else {
                        v[bb].push_back(make_pair(aa, 1));
                    }
                }
            }
        }
        dist[n * m - 1] = 0;
        q.push(make_pair(0, n * m - 1));
        while (q.size() > 0) {
            int cur = q.top().second;
            if (dist[cur] != -q.top().first) {
                q.pop();
                continue;
            }
            
            q.pop();
            for (int i = 0; i < v[cur].size(); i++) {
                int nxt = v[cur][i].first;
                if (dist[cur] + v[cur][i].second < dist[nxt]) {
                    dist[nxt] = dist[cur] + v[cur][i].second;
                    q.push(make_pair(-dist[nxt], nxt));
                }
            }
        }
        return dist[0];
    }
};
