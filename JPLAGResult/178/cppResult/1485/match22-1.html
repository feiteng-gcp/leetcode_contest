using pii = pair<int, int>;
int dp[110][110];

int dx[] = {0, 0, 1, -1};
int dy[] = {1, -1, 0, 0};
class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        memset(dp, 0x3f, sizeof(dp));
        priority_queue<pii, vector<pii>, greater<pii>> pq;
        pq.push(pii(0, 0));
        
        while (!pq.empty()) {
            auto cur = pq.top(); pq.pop();
            int cost = cur.first, x = cur.second / m, y = cur.second % m;
            if (x == n - 1 && y == m - 1) return cost;
            

            if (cost >= dp[x][y]) continue;
            dp[x][y] = cost;
            
            for (int i = 0; i < 4; ++ i) {
                int xi = x + dx[i];
                int yi = y + dy[i];
                if (xi < 0 || yi < 0 || xi >= n || yi >= m) continue;

                pq.push(pii((cost + (grid[x][y] != i + 1)), xi * m + yi));
            }
        }
        return -1;
    }
};
