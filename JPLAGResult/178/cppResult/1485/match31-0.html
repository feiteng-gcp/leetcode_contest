#define ar array
const int di[4]={0, 0, 1, -1}, dj[4]={1, -1, 0, 0};
class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        int n=grid.size(), m=grid[0].size();
        vector<vector<int>> d(n, vector<int>(m, 1e9));
        vector<vector<bool>> vis(n, vector<bool>(m));
        deque<ar<int, 2>> dq;
        dq.push_back({0, 0});
        d[0][0]=0;
        while(dq.size()) {
            ar<int, 2> u=dq.front();
            dq.pop_front();
            if(vis[u[0]][u[1]])
                continue;
            vis[u[0]][u[1]]=1;
            
            for(int k=0; k<4; ++k) {
                int ni=u[0]+di[k], nj=u[1]+dj[k];
                if(ni<0||ni>=n||nj<0||nj>=m)
                    continue;
                int nd=d[u[0]][u[1]];
                if(k+1!=grid[u[0]][u[1]])
                    ++nd;
                if(nd<d[ni][nj]) {
                    d[ni][nj]=nd;
                    if(k+1==grid[u[0]][u[1]])
                        dq.push_front({ni, nj});
                    else
                        dq.push_back({ni, nj});
                }
            }
        }
        return d[n-1][m-1];
    }
};
