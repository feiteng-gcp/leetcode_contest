class Solution {
    int ans = 0;
public:
    int minCost(vector<vector<int>>& grid) {
        int R=grid.size(), C=grid[0].size();
        vector<vector<int>> visit(R, vector<int>(C));
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> Q;

        Q.push({0, 0});
        int dy[] = {0, 0, 1, -1}; // 1:R 2: L 3: D : 4 U
        int dx[] = {1,-1, 0,  0}; 
        while(!Q.empty()) {
            auto P = Q.top(); Q.pop();
            int cost = P.first;
            int r = P.second/C;

            int c = P.second%C;
            if(r==R-1 && c==C-1) return cost;
            if(visit[r][c]) continue;
            visit[r][c] = 1;
            //cout<< r << "," << c << endl;
            for(int j=0; j<4; j++) {
                int nr = r+dy[j];
                int nc = c+dx[j];
                if(nr>=0 && nr <R && nc >= 0 && nc <C && visit[nr][nc] == 0) {
                    if(j+1 == grid[r][c])
                        Q.push({cost, nr*C+nc});
                    else
                        Q.push({cost+1, nr*C+nc});
                }
            }
        }
        return -1;
    }
};
