class Solution {
public:
    
    const int inf = 1e9 + 10;
    
    vector<int> dx = {0, 0, 1, -1};
    vector<int> dy = {1, -1, 0, 0};
    
    int dp[110][110];
    int n, m;
    
    bool checkInside(int x, int y) {
        return (x >= 0 && x < n && y >= 0 && y < m);
    }

    int minCost(vector<vector<int>>& grid) {
        
        for (int i = 0; i < 110; i++) {
            for (int j = 0; j < 110; j++) {
                dp[i][j] = inf;
            }
        }
        
        n = grid.size();
        m = grid[0].size();
        
        priority_queue<pair<int, pair<int, int>>> pq;
        dp[0][0] = 0;
        pq.push({0, {0, 0}});
        while (!pq.empty()) {

            int dis = -pq.top().first;
            int r = pq.top().second.first;
            int c = pq.top().second.second;
            pq.pop();
            if (dp[r][c] < dis) continue;
            for (int i = 0; i < 4; i++) {
                int rr = r + dx[i];
                int cc = c + dy[i];
                if (!checkInside(rr, cc)) continue;
                int cost = (i + 1 != grid[r][c]) ? 1 : 0;
                if (cost + dp[r][c] < dp[rr][cc]) {
                    dp[rr][cc] = cost + dp[r][c];
                    pq.push({-dp[rr][cc], {rr, cc}});
                }
            }
        }
        
        
        return dp[n - 1][m - 1];
        
    }
};
