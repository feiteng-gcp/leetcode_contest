class Solution {
public:
        struct PtCost{
            int i, j, cost;
            
            PtCost(int _i, int _j, int _c) : i(_i), j(_j), cost(_c){;};
            
            bool operator>(const PtCost & b) const{
                return cost > b.cost;
            };
        };
        
    
    int minCost(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<vector<int>> COST(m, vector<int>(n, INT_MAX));
        

        priority_queue<PtCost, vector<PtCost>, greater<PtCost>> minq;
        minq.push(PtCost(0, 0, 0)); COST[0][0] = 0;
        
        int dx[] = { 0,  0,  +1, -1};
        int dy[] = {+1, -1,   0,  0};
        
        while (!minq.empty()){
            auto curr = minq.top(); minq.pop();
            
            int i = curr.i, j = curr.j, cost = curr.cost;
            for (int d = 0; d < 4; ++d){
                int x = i + dx[d], y = j + dy[d];
                if ((x < 0) || (x >= m) || (y < 0) || (y >= n)){continue;};
                
                int c2 = 0;
                if (d == (grid[i][j] - 1)){
                    c2 = cost;
                }
                else{
                    c2 = cost + 1;
                }
                
                if (c2 < COST[x][y]){
                    COST[x][y] = c2;
                    minq.push(PtCost(x, y, c2));
                }
            } 
        }
        
        return COST[m-1][n-1];
    }
};
