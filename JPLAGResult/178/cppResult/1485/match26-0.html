class Solution {
public:
    int bfs(vector<vector<int>>& G)
    {
         int dx[]={0,0,0,1,-1};
         int dy[]={0,1,-1,0,0};
        int n = G.size();
        int m = G[0].size();
        vector<vector<int> > dp(n);
        const int INF= 100000;
        for(int i=0;i<n;i++){
            dp[i].resize(m);
            for(int j=0;j<m;j++){
                dp[i][j]=INF;
            }
        }
        dp[0][0]=0;
        
        queue<pair<int,int> > q;
        q.push({0,0});
        while(!q.empty()){
            pair<int,int> a = q.front();
            q.pop();
            int x = a.first;
            int y = a.second;
            for(int i=1;i<=4;i++){
                int tx = x + dx[i];
                int ty = y + dy[i];
                if(tx>=0&&tx<n&&ty>=0&&ty<m&&dp[x][y]+(G[x][y]!=i)<dp[tx][ty]){
                    dp[tx][ty]=dp[x][y]+(G[x][y]!=i);
                    q.push({tx,ty});
                }
            }
        }
        return dp[n-1][m-1];
        
    }
    int minCost(vector<vector<int>>& G) {
       
        int ans = bfs(G);
        
        
        
        return ans;

    }
};
