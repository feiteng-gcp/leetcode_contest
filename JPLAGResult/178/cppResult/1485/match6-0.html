class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        int x = grid.size(), y = grid[0].size();
        
        vector<vector<int>> move = {{0, 0}, {0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        
        vector<vector<int>> cost(x, vector<int>(y, INT_MAX));
        
        vector<vector<bool>> visited(x, vector<bool>(y, false));
        
        multimap<int, pair<int, int>> m;
        
        m.insert({0, {0, 0}});
        cost[0][0] = 0;
        
        while (!m.empty()) {
            auto e = *m.begin();
            m.erase(m.begin());
            int cx = e.second.first, cy = e.second.second;
            if (visited[cx][cy]) continue;
            visited[cx][cy] = true;
            //printf("%d %d -> %d\n", cx, cy, cost[cx][cy]);
            for (int i = 1; i <= 4; ++i) {
                int nx = cx + move[i][0], ny = cy + move[i][1];
                if (nx < 0 || nx >= x || ny < 0 || ny >= y) continue;
                int add = (grid[cx][cy] == i)?0:1;
                if (cost[nx][ny] > cost[cx][cy] + add) {
                    cost[nx][ny] = cost[cx][cy] + add;
                    m.insert({cost[nx][ny], {nx, ny}});
                }
                
            }
        }
        
        return cost[x - 1][y - 1];
    }
};
