class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        
        priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>>> pq;
        pq.emplace(vector<int>{0, 0, 0});
        
        const int INF = 9999999;
        vector<vector<int>> dist(m, vector<int>(n, INF));

        int dir[4][2] = {{0,1}, {0,-1}, {1,0},{-1,0}};
        
        while(!pq.empty()) {
            auto st = pq.top(); pq.pop();
            if(st[1] == m-1 && st[2] == n-1) {
                return st[0];
            }
            

            for(int i = 0; i < 4; i++) {
                int nx = st[1] + dir[i][0];
                int ny = st[2] + dir[i][1];
                
                if(nx < 0 || ny < 0 || nx >= m || ny >= n) continue;
                int nc = st[0] + (grid[st[1]][st[2]] - 1 != i);
                if(dist[nx][ny] > nc) {
                    dist[nx][ny] = nc;
                    pq.push(vector<int>{nc, nx, ny});
                }
            }
        }
        
        return -1;
    }
};
