const int maxn = 105;
int f[maxn][maxn];
bool in[maxn][maxn];

class Solution {
private:
    vector<vector<int>> dir = {{0,1}, {0,-1}, {1, 0}, {-1,0}};
public:
    int minCost(vector<vector<int>>& a) {
        int n = a.size();
        int m = a[0].size();
        memset(f,-1,sizeof(f));
        memset(in, 0,sizeof(in));
        queue<int>qx,qy;
        qx.push(0);
        qy.push(0);
        in[0][0] = 1;
        f[0][0] = 0;
        while(!qx.empty()){
            int x = qx.front();
            qx.pop();
            int y = qy.front();
            qy.pop();
            in[x][y] = 0;
            for(int i=0;i<4;i++){
                int nx = x + dir[i][0];
                int ny = y + dir[i][1];
                if(nx<0||nx>=n||ny<0||ny>=m)continue;
                int tt = f[x][y] + (i+1!=a[x][y]);
                if(f[nx][ny]==-1||f[nx][ny]>tt){
                    f[nx][ny] = tt;
                    if(!in[nx][ny]){
                        in[nx][ny] = 1;
                        qx.push(nx);
                        qy.push(ny);
                    }
                }
            }
        }
        return f[n-1][m-1];
    }
};
