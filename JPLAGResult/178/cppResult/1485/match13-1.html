class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        vector<pair<int, int>> dir(5);
        dir[1] = make_pair(0, 1);
        dir[2] = make_pair(0, -1);
        dir[3] = make_pair(1, 0);
        dir[4] = make_pair(-1, 0);
        int n = grid.size();
        int m = grid[0].size();
        vector<vector<int>> dp(n, vector<int>(m, -1));
        dp[0][0] = 0;
        queue<pair<int, int>> q;
        vector<vector<int>> inq(n, vector<int>(m, 0));
        inq[0][0] = 1;
        q.push(make_pair(0, 0));
        while (q.size()) {

            int x = q.front().first;
            int y = q.front().second;
            // cout << x << " " << y << endl;
            q.pop();
            inq[x][y] = 0;
            for (int i = 1; i <= 4; i++) {
                int nx = x + dir[i].first;
                int ny = y + dir[i].second;
                if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;

                int c = grid[x][y] == i ? 0 : 1;
                if (dp[nx][ny] == -1 || dp[nx][ny] > dp[x][y] + c) {
                    dp[nx][ny] = dp[x][y] + c;
                    if (!inq[nx][ny]) {
                        q.push(make_pair(nx, ny));
                        inq[nx][ny] = 1;
                    }
                }
            }
        }
        return dp[n - 1][m - 1];
    }
};
