class Solution {
public:
    int minCost(vector<vector<int>>& a) {
        int m=a.size();
        int n=a[0].size();
        int d[105][105];
        for(int i=0;i<m;i++)
            for(int j=0;j<n;j++)
                d[i][j]=1e9;
        d[0][0]=0;

        priority_queue<pair<int,pair<int,int>>>q;
        q.push(make_pair(0,make_pair(0,0)));
        int dir[4][2]={{0,1},{0,-1},{1,0},{-1,0}};
        while(!q.empty()){

            int t=q.top().first;
            int x=q.top().second.first;
            int y=q.top().second.second;
            q.pop();
            for(int i=0;i<4;i++){
                int xx=x+dir[i][0];
                int yy=y+dir[i][1];
                int c=1;
                if(i==a[x][y]-1)c=0;
                if(xx<0||xx>=m||yy<0||yy>=n||d[xx][yy]<=d[x][y]+c)continue;
                d[xx][yy]=d[x][y]+c;
                q.push(make_pair(-d[xx][yy],make_pair(xx,yy)));
            }
        }
        return d[m-1][n-1];
    }
};
