class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        
        vector<vector<int>> f(m, vector<int>(n, n * m));
        f[0][0] = 0;
        
        vector<vector<bool>> inQ(m, vector<bool>(n));
        
        queue<pair<int, int>> q;
        q.push(make_pair(0, 0));
        inQ[0][0] = true;
        
        while (!q.empty()) {
            auto cur = q.front();
            q.pop();
            
            int i = cur.first;
            int j = cur.second;
            
            inQ[i][j] = false;
            
            for (int d = 1; d <= 4; ++d) {
                int x = i + di[d];
                int y = j + dj[d];
                if (x >= 0 && x < m && y >= 0 && y < n) {
                    int cost = d == grid[i][j] ? 0 : 1;
                    if (f[i][j] + cost < f[x][y]) {
                        f[x][y] = f[i][j] + cost;
                        if (!inQ[x][y]) {
                            q.push(make_pair(x, y));
                            inQ[x][y] = true;
                        }
                    }
                }
            }
        }
        
        return f[m - 1][n - 1];
    }
    
private:
    vector<int> di = {0, 0, 0, 1, -1};
    vector<int> dj = {0, 1, -1, 0, 0};
    
};
