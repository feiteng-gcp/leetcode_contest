class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        vector<int> dx = {0,0,1,-1};
        vector<int> dy = {1,-1,0,0};

        int m = grid.size();
        int n = grid[0].size();
        int dist[m][n];
        for(int i=0; i < m; i++)for(int j = 0; j < n; j++)dist[i][j] = 10000;
        dist[0][0] = 0;
        queue<vector<int>> q;
        q.push({0,0});
        while(!q.empty()){

            auto t = q.front();
            //cout << t[0] << t[1] << endl;
            q.pop();
            for(int i =0; i < 4; i++){
                int nx = t[0] + dx[i];
                int ny = t[1] + dy[i];
                int right = (grid[t[0]][t[1]]-1 == i) ? 0 : 1;
                if(0 <= nx && nx < m && 0 <= ny && ny < n){
                    if(dist[nx][ny] > dist[t[0]][t[1]] + right){
                        dist[nx][ny] = dist[t[0]][t[1]] + right;
                        q.push({nx,ny});
                    }
                }
            }
        }
        return dist[m-1][n-1];
    }
};
