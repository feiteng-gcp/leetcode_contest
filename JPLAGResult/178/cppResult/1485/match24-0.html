class Solution {
private:
    int n, m;
    int dp[105][105];
    vector<vector<int>> mp;
    
    int dir[4][2] = { {0, 1}, {0, -1}, {1, 0}, {-1, 0} };
    void bfs(int x, int y) {
        //
        queue<pair<int, int> > Q;
        
        Q.push({x, y});
        
        while(!Q.empty()) {
            auto tt = Q.front(); Q.pop();
            int x = tt.first; int y = tt.second;
             // printf("%d %d %d\n", x, y, dp[x][y]);
            for(int i = 0; i < 4; ++i) {
                int xx = x + dir[i][0];
                int yy = y + dir[i][1];
                if(xx >= 0 && xx < n && yy >= 0 && yy < m && dp[xx][yy] > dp[x][y] + ((i+1) != mp[x][y]) ) {
                    dp[xx][yy] = dp[x][y] + ((i+1) != mp[x][y]);
                    // printf("%d %d\n", xx, yy, dp[xx][yy]);
                    // dfs(xx, yy);
                    // if(dp[xx][yy] + n - 1 - xx + m - 1 - yy >= dp[n-1][m-1]) continue;
                    
                    Q.push({xx, yy});
                }
            }
        }
    }
public:
    int minCost(vector<vector<int>>& grid) {
        const int INF = 0x3f3f3f3f;
        mp = grid;
        n = grid.size();
        m = grid[0].size();
        memset(dp, INF, sizeof(dp));
        
        dp[0][0] = 0;

        bfs(0, 0);
        
        return dp[n - 1][m - 1];
    }
};
