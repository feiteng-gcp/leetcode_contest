# define PR pair<int, int>


class Solution {
public:
    
    int minCost(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        vector<vector<int>> dp(m, vector<int>(n, 0));
        vector<vector<bool>> visited(m, vector<bool>(n, false));
        priority_queue<PR, vector<PR>, greater<PR> > stk;
        stk.push({0, 0});   // val, pos
        while(!stk.empty()){
            PR pr = stk.top();
            int x = pr.second/n;
            int y = pr.second%n;
            stk.pop();
            int _x = x;
            int _y = y;
            cout<<x<<", "<<y<<", "<<pr.first<<endl;
            if(x == m-1 && y == n-1){
                dp[x][y] = pr.first;
                break;
            }
            if(visited[x][y]){
                continue;
            }else{
                visited[x][y] = true;
            }
            
            if(grid[x][y] == 1){
                _y++;
            }else if(grid[x][y] == 2){
                _y--;
            }else if(grid[x][y] == 3){
                _x++;
            }else{
                _x--;

            }
            
            vector<int> delta = {0, 1, 0, -1, 0};
            for(int i = 0; i < 4; i++){
                int dx = delta[i] + x;
                int dy = delta[i+1] + y;
                if(dx >= 0 && dx < m && dy >= 0 && dy < n && visited[dx][dy] == false){
                    dp[dx][dy] = pr.first+1;
                    stk.push({pr.first+1, dx*n + dy});
                }
            }
            if(_x >= 0 && _x < m && _y >= 0 && _y < n && visited[_x][_y] == false){
                dp[_x][_y] = pr.first;
                stk.push({pr.first, _x*n + _y});
            }
        }
        return dp[m-1][n-1];
    }
};
