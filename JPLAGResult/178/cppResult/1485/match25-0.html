class Solution {
  int dx[5]={0,0,0,1,-1};
  int dy[5]={0,1,-1,0,0};
  int m,n;
  struct pos{
    pos(int _x,int _y,int _cost):x(_x),y(_y),cost(_cost){}
    int x;
    int y;
    int cost;
  };
  bool valid(int x,int y){
    if(x<0||x>=m) return false;
    if(y<0||y>=n) return false;
    return true;
  }
public:
    int minCost(vector<vector<int>>& grid) {
      m=grid.size();n=grid[0].size();
      bool vis[m][n];
      memset(vis,false,sizeof(vis));
      int mincost[m][n];
      memset(mincost,0x3f,sizeof(mincost));
      queue<pos>q;
      int ans=m+n;
      vis[0][0]=true;mincost[0][0]=0;
      q.push(pos(0,0,0));
      while(!q.empty()){
        pos p=q.front();q.pop();
        if(p.x==m-1&&p.y==n-1) {ans=min(ans,p.cost);}
        else{
          int origin=grid[p.x][p.y];
          for(int i=1;i<=4;++i){
            int newx=p.x+dx[i],newy=p.y+dy[i];
            if(!valid(newx,newy)) continue;
            if(i==origin){
              if(p.cost<mincost[newx][newy]){
                q.push(pos(newx,newy,p.cost));mincost[newx][newy]=p.cost;
              }
            }
            else{
              if(p.cost+1<mincost[newx][newy]){
                q.push(pos(newx,newy,p.cost+1));mincost[newx][newy]=p.cost+1;

              }
            }
          }
        }
      }
      return ans;
    }
};
