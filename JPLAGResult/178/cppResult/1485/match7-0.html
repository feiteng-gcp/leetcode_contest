struct node {
    int cost;
    int x;
    int y;
    node(int cc = 0, int xx = 0, int yy = 0):cost(cc), x(xx), y(yy) {}
};

struct cmp {
    bool operator()(const node &a, const node &b) {
        return a.cost>b.cost;   
    }
};
class Solution {
public:
    
    int minCost(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        priority_queue<node, vector<node>, cmp> qe;
        vector<vector<int> > visit(n, vector<int>(m, -1));
        qe.push({0,0,0});
        visit[0][0] = 0;
        int d[][2] = {{0,0}, {0,1}, {0,-1}, {1,0}, {-1,0}};
        while(!qe.empty()) {
            auto p = qe.top(); qe.pop();
            if (p.x == n-1 && p.y == m-1) return p.cost;
            for(int i = 1; i<=4; i++) {
                int x = p.x+d[i][0], y = p.y + d[i][1];
                if (x<0 || x>=n || y<0 || y>=m) continue;
                int cost = p.cost + (i!=grid[p.x][p.y]?1:0);
                if (visit[x][y] == -1 || visit[x][y]>cost) {
                    visit[x][y] = cost;
                    qe.push({cost,x,y});
                }
            }
        }
        return 0;
        
        
    }
};
