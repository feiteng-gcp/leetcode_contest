class Solution {
public:
    int N, M;
    int getid(int x,int y) {
        return x * M + y;
    }
    int minCost(vector<vector<int>>& grid) {
        N = grid.size();
        M = grid[0].size();
        vector<pair<int,int> >e[N * M];
        int di[4] = {0, 0, 1, -1}, dj[4] = {1, -1, 0, 0};
        for (int i=0;i<N;i++) {
            for (int j=0;j<M;j++) {
                for (int dir = 0; dir < 4; dir++) {
                    int ni = i + di[dir], nj = j + dj[dir];
                    if (ni < 0 || ni >= N || nj < 0 || nj >= M) {
                        continue;
                    }
                    int id = getid(i, j), nid = getid(ni, nj);
                    if (dir == grid[i][j] - 1) {
                        e[id].push_back(make_pair(nid, 0));
                    } else {
                        e[id].push_back(make_pair(nid, 1));
                    }
                }
            }
        }
        vector<int> d(N * M, N + M + 1);
        vector<bool> inq(N * M, false);
        queue<int> q;
        d[0] = 0;
        q.push(0);
        inq[0] = true;
        while(!q.empty()) {
            int now = q.front();
            for (auto t:e[now]) {
                if (d[now] + t.second < d[t.first]) {
                    d[t.first] = d[now] + t.second;
                    if (!inq[t.first]) {
                        inq[t.first] = true;
                        q.push(t.first);
                    }
                }
            }
            inq[now] = false;
            q.pop();
        }
        return d[N * M - 1];
    }
};
