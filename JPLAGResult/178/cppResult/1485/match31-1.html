class Solution {
public:
    vector<int> h;
    vector<int> e;
    vector<int> ne;
    vector<int> w;
    vector<int> dist;
    vector<bool> st;
    int idx;
    void init(){
        h = vector<int> (10010, -1);
        e = vector<int> (100100, 0);
        idx = 0;
        ne = vector<int> (100100, 0);
        w = vector<int> (100100, 0);
        dist = vector<int> (10010, 0x3f3f3f);
        st = vector<bool> (10010, 0);
    }
    void add(int a, int b, int c){
        e[idx] = b;
        ne[idx] = h[a];
        w[idx] = c;
        h[a] = idx ++;
    }
    int bfs(int src, int des){
        dist[src] = 0;
        deque<int> dq;
        dq.push_back(src);

        while(!dq.empty()){
            int curr = dq.front();
            dq.pop_front();
            if(st[curr])continue;
            st[curr] = true;
            for(int i = h[curr]; ~i; i = ne[i]){
                int node = e[i];
                if(st[node]) continue;
                if(dist[node] > dist[curr] + w[i]){
                    if(w[i] == 0) dq.push_front(node);
                    else dq.push_back(node);
                    dist[node] = dist[curr] + w[i];
                }
            }
        }
        return dist[des];
    }
    int minCost(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        init();
        int dirx[5] = {0, 0, 0, 1, -1};
        int diry[5] = {0, 1, -1, 0, 0};
        for(int i = 0; i < m; i ++){
            for(int j = 0; j < n; j ++){
                int curr = i * n + j;
                for(int k = 1; k <= 4; k ++){
                    int new_x = i + dirx[k];
                    int new_y = j + diry[k];
                    if(new_x < 0 || new_y < 0 || new_x >= m || new_y >= n) continue;

                    int curr_next = new_x * n + new_y;
                    if(grid[i][j] == k){
                        add(curr, curr_next, 0);
                    } else {
                        add(curr, curr_next, 1);
                    }
                }
            }
        }
        return bfs(0, m*n - 1);
    }
};
