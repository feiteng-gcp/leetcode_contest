int f[100][100];
class Solution {
public:
    const int MAX_INF = 99999999;
    int m, n;
    void runIt(vector<vector<int>>& grid, int x, int y, int change)
    {
        if (x < 0 || x == n || y < 0 || y == m) return ;
        if (f[x][y] <= change) return ;
        if (change > m + n) return;
        f[x][y] = change;
        if (grid[x][y] == 1) 
        {
            runIt(grid, x, y + 1, change);
            runIt(grid, x, y - 1, change + 1);
            runIt(grid, x + 1, y, change + 1);
            runIt(grid, x - 1, y, change + 1);
        }
        else if (grid[x][y] == 2) 
        {
            runIt(grid, x, y - 1, change);
            runIt(grid, x, y + 1, change + 1);
            runIt(grid, x + 1, y, change + 1);
            runIt(grid, x - 1, y, change + 1);
        }
        else if (grid[x][y] == 3) 
        {
            runIt(grid, x + 1, y, change);
            runIt(grid, x, y + 1, change + 1);
            runIt(grid, x, y - 1, change + 1);
            runIt(grid, x - 1, y, change + 1);
        }
        else if (grid[x][y] == 4) 
        {
            runIt(grid, x - 1, y, change);
            runIt(grid, x, y + 1, change + 1);
            runIt(grid, x, y - 1, change + 1);
            runIt(grid, x + 1, y, change + 1);
        }
    }
    int minCost(vector<vector<int>>& grid) {
        n = grid.size();
        m = grid[0].size();
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < m; ++j)
                f[i][j] = MAX_INF;
        runIt(grid, 0, 0, 0);
        return f[n - 1][m - 1];
    }
};
