int fx[]={0,0,+1,-1};
int fy[]={+1,-1,0,0};
class Solution {
public:
    int dist[102][102];    
    bool is_valid(int i, int j, int r, int c)
    {
        return (i>=0 && i<r && j>=0 && j<c);
    }
    
    int minCost(vector<vector<int>>& grid) {
        for(int i=0; i<=100; i++)
        {
            for(int j=0; j<=100; j++)
            {
                dist[i][j]=1e9;
            }
        }
        deque<pair<int,int>> dq;
        dq.push_front({0,0});
        dist[0][0]=0;
        while(!dq.empty())
        {
            pair<int,int> u=dq.front();
            dq.pop_front();
            
            for(int i=0; i<4; i++)
            {
                int tx=fx[i]+u.first;
                int ty=fy[i]+u.second;
                
                if(is_valid(tx,ty,grid.size(),grid[0].size()))
                {

                    if(i+1==grid[u.first][u.second])
                    {
                        if(dist[tx][ty]>dist[u.first][u.second])
                        {
                            dist[tx][ty]=dist[u.first][u.second];
                            dq.push_front({tx,ty});
                        }
                    }
                    else
                    {
                        if(dist[tx][ty]>dist[u.first][u.second]+1)
                        {
                            dist[tx][ty]=dist[u.first][u.second]+1;
                            dq.push_back({tx,ty});
                        }
                    }
                }
            }
        }
        return dist[(int)grid.size()-1][(int)grid[0].size()-1];

        
    }
};
