class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        
        vector<vector<int>> cost(m, vector<int>(n, 10000));
        // vector<vector<bool>> seen(m, vector<bool>(n, false));
        queue<pair<int,int>> q;
        q.push(make_pair(m-1, n-1));
        cost[m-1][n-1]=0;
        while(!q.empty()) {
            pair<int,int> p = q.front();
            q.pop();
            int x = p.first;
            int y = p.second;
            
            int dx[] = {0, 0, -1, 1};
            int dy[] = {-1, 1, 0, 0};
            for(int i = 0; i < 4; i++) {
                int nx = x+dx[i];
                int ny = y+dy[i];
                if(nx >= 0 && nx < m && ny >= 0 && ny < n) {
                    int newCost = cost[x][y];
                    if(i != grid[nx][ny]-1) newCost++;
                    if(newCost < cost[nx][ny]) {
                        cost[nx][ny] = newCost;
                        q.push(make_pair(nx,ny));
                    }
                }
            }
        }
        return cost[0][0];
    }
};
