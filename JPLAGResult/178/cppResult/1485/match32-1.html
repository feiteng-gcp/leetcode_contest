class Solution {
    void pushCanReach(vector<pair<int, int>> & current, map<pair<int, int>, bool> & vis, vector<vector<int>> & grid, int x, int y) {
        stack<pair<int, int>> data;
        int m = grid.size();
        int n = grid[0].size();
        data.push({x, y});
        
        while (data.size() > 0) {
            auto p = data.top();
            data.pop();
            
            int i = p.first;
            int j = p.second;
            if (grid[i][j] == 1) {
                j++;
            }
            else if (grid[i][j] == 2) {
                j--;
            }
            else if (grid[i][j] == 3) {
                i++;
            }
            else if (grid[i][j] == 4) {
                i--;
            }
            
            if (i < 0 || i >= m || j < 0 || j >= n) continue;
            if (vis[{i, j}]) continue;
                    
            vis[{i, j}] = true;
            current.push_back({i, j});
            data.push({i, j});
        }
    }
    
public:
    int minCost(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        
        vector<pair<int, int>> current;
        vector<pair<int, int>> next;
        map<pair<int, int>, bool> vis;
        int step = 0;
        current.push_back({0, 0});

        vis[{0, 0}] = true;
        pushCanReach(current, vis, grid, 0, 0);
        
        int diff[4][2] = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        while (1) {
            next.clear();
            for (auto p : current) {
                if (p.first == m-1 && p.second == n-1) return step;
                

                for (int k=0; k<4; k++) {
                    int i = p.first + diff[k][0];
                    int j = p.second + diff[k][1];
                    
                    if (i < 0 || i >= m || j < 0 || j >= n) continue;
                    if (vis[{i, j}]) continue;
                            
                    vis[{i, j}] = true;
                    next.push_back({i, j});
                    pushCanReach(next, vis, grid, i, j);
                }
            }
            
            current.clear();
            swap(next, current);
            step++;
        }
        
        return 0;
    }
};
