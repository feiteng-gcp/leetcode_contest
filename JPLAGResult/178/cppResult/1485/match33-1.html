class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        vector<vector<int>> cost(grid.size(), vector<int>(grid[0].size(), INT_MAX));
        dfs(grid, cost, 0, 0, 0);
        return cost[cost.size() - 1][cost[0].size() - 1];
    }
    
    void dfs(vector<vector<int>>& grid, vector<vector<int>>& cost, int i, int j, int x) {
        if (i >= 0 && i < grid.size() && j >= 0 && j < grid[0].size() && cost[i][j] > x) {

            cost[i][j] = x;
            if (grid[i][j] == 1) {
                dfs(grid, cost, i, j + 1, x);
                dfs(grid, cost, i, j - 1, x + 1);
                dfs(grid, cost, i + 1, j, x + 1);
                dfs(grid, cost, i - 1, j, x + 1);
            } else if (grid[i][j] == 2) {
                dfs(grid, cost, i, j - 1, x);
                dfs(grid, cost, i, j + 1, x + 1);
                dfs(grid, cost, i + 1, j, x + 1);
                dfs(grid, cost, i - 1, j, x + 1);
            } else if (grid[i][j] == 3) {
                dfs(grid, cost, i + 1, j, x);
                dfs(grid, cost, i, j + 1, x + 1);
                dfs(grid, cost, i, j - 1, x + 1);
                dfs(grid, cost, i - 1, j, x + 1);
            } else if (grid[i][j] == 4) {
                dfs(grid, cost, i - 1, j, x);
                dfs(grid, cost, i, j + 1, x + 1);
                dfs(grid, cost, i, j - 1, x + 1);
                dfs(grid, cost, i + 1, j, x + 1);
            }
        }
    }
};
