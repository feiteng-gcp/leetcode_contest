class Solution {
public:
    int hash(int x, int y) {
        return (x <<16) | y;
    }
    void dehash(int v, int &x, int &y) {
        x = (v >> 16);
        y = (v & 0xff);
    }
    int minCost(vector<vector<int>>& grid) {
        int dx[4] = {0, 0, 1, -1};
        int dy[4] = {1, -1, 0, 0};
        unordered_map<int, int> seen;
        queue<int> q;
        q.push(0);
        seen[0] = 0;
        int ans = 0;
        int lh = hash(grid.size() - 1, grid[0].size() - 1);
        while(q.size()) {
            int x, y;
            auto h = q.front();q.pop();
            dehash(h, x, y);
            for(int i = 0; i <4; ++i) {
                int ny = y + dy[i];
                int nx = x + dx[i];
                if(nx <0  || ny <0 || nx >= grid.size() || ny >= grid[0].size())
                    continue;
                int nh = hash(nx, ny);
                int am = grid[x][y] != i + 1;
                if(seen.count(nh) && seen[nh] <= seen[h] + am) continue;
                seen[nh] = seen[h] + am;
                q.push(nh);
            }
        }
        return seen[lh];
    }
};
