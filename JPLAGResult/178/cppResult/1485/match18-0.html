class Solution {
public:
    int minCost(vector<vector<int>>& grid) {
        int m=grid.size(),n=grid[0].size();
        vector<vector<int> > dp(m,vector<int>(n,INT_MAX));
        dp[m-1][n-1]=0;
        queue<int> q;
        q.push((m-1)*n+n-1);
        int dr[5]={0,0,0,1,-1},dc[5]={0,1,-1,0,0};
        while(!q.empty())
        {
            int cur=q.front();
            q.pop();
            int r=cur/n,c=cur%n;
            for(int i=1;i<=4;i++)
            {
                int new_r=r+dr[i],new_c=c+dc[i];
                if(new_r>=0&new_r<m&&new_c>=0&&new_c<n)
                {
                    int temp;
                    if(new_r+dr[grid[new_r][new_c]]==r&&new_c+dc[grid[new_r][new_c]]==c) temp=dp[r][c];
                    else temp=dp[r][c]+1;
                    if(temp<dp[new_r][new_c])
                    {
                        dp[new_r][new_c]=temp;
                        q.push(new_r*n+new_c);
                    }
                }
            }
        }
        return dp[0][0];
    }
};
