struct is
{
    int x;
    int y;
    int step;
    bool operator < (const is& A)const
    {
        return step > A.step;
    }
};
class Solution {
public:
    bool check(int x, int y, int n, int m)
    {
        if(x < 0 || y < 0 || x >= n || y >= m)
            return false;
        return true;
    }
    int minCost(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        priority_queue<is> Q;
        int vis[110][110];
        memset(vis, -1, sizeof(vis));
        Q.push(is{n - 1, m - 1, 0});
        int dx[4] = {0, 0, -1, 1};
        int dy[4] = {-1, 1, 0, 0};
        while(!Q.empty())
        {
            is now = Q.top();
            Q.pop();
            if(now.x == 0 && now.y == 0) return now.step;
            if(vis[now.x][now.y] != -1)continue;
            vis[now.x][now.y] = now.step;
            cout<<now.x<<" " <<now.y <<" " << now.step<<endl;
            for(int i = 0; i < 4; i++)
            {
                int nexx = now.x + dx[i];
                int nexy = now.y + dy[i];
                if(check(nexx, nexy, n, m))
                {
                    if(grid[nexx][nexy] == i + 1)
                    {
                        if(vis[nexx][nexy] == -1)
                        Q.push(is{nexx,nexy,now.step});
                    }
                    else
                    {
                        if(vis[nexx][nexy] == -1)
                        Q.push(is{nexx,nexy,now.step + 1});
                    }
                }
            }
        }
        return -1;
    }
};
