class Solution {
public:
    int dx[5] = {0, 0, 0, 1, -1};
    int dy[5] = {0, 1, -1, 0, 0};
    int minCost(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        auto in = [&](int i, int j) { return 0 <= i && i < m && 0 <= j && j < n; };
        deque<int> x, y;
        x.emplace_back(0), y.emplace_back(0);
        vector<vector<int>> d(m, vector<int>(n, 1e9));
        d[0][0] = 0;
        while (!x.empty()) {

            int i = x.front(), j = y.front();
            x.pop_front(), y.pop_front();
            for (int t = 1; t <= 4; t++) {
                int ni = i + dx[t], nj = j + dy[t];
                if (!in(ni, nj)) continue;
                if (t == grid[i][j]) {
                    if (d[ni][nj] > d[i][j]) {
                        d[ni][nj] = d[i][j];
                        x.emplace_front(ni), y.emplace_front(nj);
                    }
                } else {
                    if (d[ni][nj] > d[i][j] + 1) {
                        d[ni][nj] = d[i][j] + 1;
                        x.emplace_back(ni), y.emplace_back(nj);
                    }
                }
            }
        }
        return d[m-1][n-1];
    }
};
