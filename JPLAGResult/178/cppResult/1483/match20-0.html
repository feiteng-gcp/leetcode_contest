class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int len = votes[0].length();
        vector<vector<int>> cnts(26, vector<int>(len, 0));
        for (const auto& vote : votes) {
            for (int i = 0; i < len; ++i) {
                ++cnts[vote[i] - 'A'][i];
            }
        }
        string res = votes[0];
        auto cmp = [&cnts, len](char x, char y){
            for (int i = 0; i < len; ++i) {
                if (cnts[x - 'A'][i] > cnts[y - 'A'][i]) {
                    return true;
                } else if (cnts[x - 'A'][i] < cnts[y - 'A'][i]) {
                    return false;
                }
            }
            return x < y;
        };
        sort(res.begin(), res.end(), cmp);
        return res;
    }
};
