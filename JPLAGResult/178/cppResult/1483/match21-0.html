class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n=votes.size(), m=26;
        vector<vector<int>> t(m, vector<int>(m+1));
        set<char> s;
        for(int i=0; i<n; ++i) {
            for(int j=0; j<votes[i].size(); ++j) {
                ++t[votes[i][j]-'A'][j];
                s.insert(votes[i][j]);
            }
        }
        for(int j=0; j<m; ++j)
            t[j][m]=j;
        sort(t.begin(), t.end(), [&](vector<int> &a, vector<int> &b) {
            for(int i=0; i<m; ++i) {
                if(a[i]^b[i]) {
                    return a[i]>b[i];
                }
            }
            return a[m]<b[m];
        });
        string ans;
        for(int i=0; i<m; ++i)
            if(s.find('A'+t[i][m])!=s.end())
                ans+='A'+t[i][m];
        return ans;
    }
};
