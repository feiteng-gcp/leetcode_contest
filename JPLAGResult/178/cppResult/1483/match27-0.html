class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n_votes = votes[0].size();
        vector<vector<int>> freq(26, vector<int>(n_votes+1, 0));
        vector<int> ok(26, false);
        for(auto s: votes){
            for(int i=0;i<s.size();++i){
                freq[s[i]-'A'][i]++;
                ok[s[i]-'A']=true;
            }
        }
        for(int i=0;i<26;++i) freq[i].back()=-i;
        sort(begin(freq), end(freq));
        string ret;
        for(int i=25;i>=0;--i){
            if(ok[-freq[i].back()]) ret+='A'-freq[i].back();
        }
        return ret;
    }
};
