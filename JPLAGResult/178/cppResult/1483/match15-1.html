class Solution {
public:
    string rankTeams(vector<string>& votes) {
        using vi = vector<int>;
        using pcv = pair<vi, int>;

        pcv p[26];
        for (int i = 0; i < 26; i++) {
            p[i].first.resize(votes[0].length());
            p[i].second = -i;
        }


        for (string s : votes) {
            for (int i = 0; i < s.length(); i++) {
                p[s[i]-'A'].first[i]++;
            }
        }

        sort(p, p+26);
        reverse(p, p+26);

        string ret;
        for (int i = 0; i < votes[0].size(); i++) {
            ret += (char)('A' - p[i].second);
        }

        return ret;
    }
};
