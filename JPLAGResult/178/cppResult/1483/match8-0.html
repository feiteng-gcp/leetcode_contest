class Solution {
public:
    string rankTeams(vector<string>& votes) {
        unordered_map<char, vector<int>> cnts;
        for (char ch : votes[0]) {
            cnts.emplace(ch, vector<int>(votes[0].size()));
        }
        for (const auto &vote : votes) {
            for (int i = 0; i < vote.size(); ++i) {
                cnts[vote[i]][i] += 1;
            }
        }
        string candidates = votes[0];
        sort(candidates.begin(), candidates.end(), [&](char a, char b) {
            for (int i = 0; i < votes[0].size(); ++i) {
                int diff = cnts[a][i] - cnts[b][i];
                if (diff > 0) {
                    return true;
                } else if (diff < 0) {
                    return false;
                }
            }
            return a < b;
        });
        return candidates;
    }
};
