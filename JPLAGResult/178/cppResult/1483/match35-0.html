class Node{
public:
    vector<int> vHave;
    char C;
    int N;
    Node(char c,int n){
        C=c;
        vHave=vector<int>(n);
        N=n;
    }
    bool operator<(const Node& other)const{
        for(int i=0;i<N;i++){
            if(vHave[i]!=other.vHave[i]){
                return vHave[i]>other.vHave[i];
            }
        }
        return C<other.C;
    }
};
class Solution {
public:
    int N;  ///
    int S;
    string rankTeams(vector<string>& votes) {
        N=votes.size();
        S=votes[0].size();
        vector<Node> vNode=vector<Node>(26,Node('A',S));
        for(int i=0;i<26;i++){
            vNode[i].C='A'+i;
        }
        string ans;
        for(auto &s:votes){
            for(int i=0;i<S;i++){
                vNode[s[i]-'A'].vHave[i]++;  
            }
        }
        vector<Node> sNode;
        for(auto c:votes[0]){
            sNode.push_back(vNode[c-'A']);
        }
        sort(sNode.begin(),sNode.end());
        for(auto x:sNode){
            ans.push_back(x.C);
        }
        return ans;
    }
};
