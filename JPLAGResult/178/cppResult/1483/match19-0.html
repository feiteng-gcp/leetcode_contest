class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n = votes[0].length();
        vector<vector<int>> rating(26, vector<int>(n));
        for (int i = 0; i < votes.size(); i++) {
            for (int j = 0; j < n; j++) {
                rating[votes[i][j] - 'A'][j]++;
            }
        }
        string s = votes[0];
        sort(s.begin(), s.end());
        sort(s.begin(), s.end(),[rating, n](const char& a, const char&b) -> bool {
           for (int i = 0; i < n; i++) {
               if (rating[a-'A'][i] != rating[b-'A'][i]) {
                   return rating[a-'A'][i] > rating[b-'A'][i];
               }   
           } 
            return a < b;
        });
        return s;
    }
};
