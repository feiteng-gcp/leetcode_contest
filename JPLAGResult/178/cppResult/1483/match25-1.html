class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n = 26;
        vector<pair<vector<int>, int>> lst (n);
        vector<bool> used(n, false);
        for (int i = 0; i < n; i++) {

            lst[i].second = i;
            lst[i].first.assign(n, 0);
        }
        
        for (int i = 0; i < votes.size(); i++) {
            for (int j = 0; j < votes[i].size(); j++) {
                int team = votes[i][j]-'A';
                used[team] = true;
                lst[team].first[j]--;
            }
        }

        sort(lst.begin(), lst.end());
        
        string ans = "";
        for (int i = 0; i < lst.size(); i++) {
            if (used[lst[i].second])
                ans += 'A'+lst[i].second;
        }
        return ans;
    }
};
