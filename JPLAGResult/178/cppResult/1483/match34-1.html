class Solution {
public:
  string rankTeams(vector<string>& a) {
    const int n = a.size();
    const int m = a[0].size();
    vector<vector<int>> cnt(26, vector<int>(m));
    vector<bool> used(26);

    for (int i = 0; i < n; i++) {
      for (int j = 0; j < m; j++) {
        cnt[a[i][j] - 'A'][j]++;
        used[a[i][j] - 'A'] = true;
      }
    }
    vector<int> ord;
    for (int i = 0; i < 26; i++) {
      if (used[i]) {
        ord.push_back(i);
      }
    }
    sort(ord.begin(), ord.end(), [&](int i, int j) {
      return cnt[i] > cnt[j];
    });
    string ans;
    for (int i : ord) {
      ans += i + 'A';
    }
    return ans;
  }
};
