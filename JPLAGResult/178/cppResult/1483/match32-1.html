class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n = votes.size();
        if (n == 0) {
            return "";
        }
        int m = votes[0].length();
        if (m == 0) {
            return "";
        }
        

        const int A = 128;
        vector<vector<int>> cnts(A, vector<int>(m, 0));
        
        vector<bool> used(A, false);
        for (auto& s : votes) {
            for (int i = 0; i < s.length(); ++i) {
                cnts[s[i]][i] += 1;
                used[s[i]] = true;
            }
        }
        
        string res;
        for (int i = 0; i < A; ++i) {
            if (used[i]) {
                res += (char) i;
            }
        }
        
        auto cmp = [&](char l, char r) {
            for (int i = 0; i < m; ++i) {
                if (cnts[l][i] > cnts[r][i]) {
                    return true;
                } else if (cnts[l][i] < cnts[r][i]) {
                    return false;
                }
            }  
            return l < r;
        };
        
        sort(res.begin(), res.end(), cmp);
        
        return res;
    }
};
