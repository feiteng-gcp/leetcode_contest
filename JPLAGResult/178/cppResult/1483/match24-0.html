class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int m = votes.size(), n = votes[0].size();
        unordered_map<char, vector<int>> cnt;
        for (string &s : votes) {
            for (int i = 0; i < n; ++i) {
                if (cnt.find(s[i]) == cnt.end()) {
                    cnt[s[i]] = vector<int>(n, 0);
                }
                cnt[s[i]][i]++;
            }
        }
        auto cmp = [&](char a, char b) -> bool {
            for (int i = 0; i < n; ++i) {
                if (cnt[a][i] > cnt[b][i]) {
                    return true;
                }else if (cnt[a][i] < cnt[b][i]) {
                    return false;
                }
            }
            return a < b;
        };
        string ans = votes[0];
        sort(ans.begin(), ans.end(), cmp);
        return ans;
    }
};
