class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n=votes[0].size();
        map<int, vector<char>> m;
        map<char, vector<int>> score;
        for (int i=0;i<n;i++)
            score[votes[0][i]].resize(n, 0);
        
        for (auto& s:votes) 
            for (int i=0;i<n;i++)
                score[s[i]][i]++;
        
        // for (auto& t:score) {
        //     cout<<t.first<<' ';
        //     for (int i=0;i<n;i++)
        //         cout<<t.second[i]<<' ';
        //     cout<<endl;
        // }
        
        vector<bool> used(26, true);
        for (int i=0;i<n;i++)
            used[votes[0][i]-'A']=false;
        
        string ans;
        
        for (int i=0;i<n;i++) {
            char k='A';
            while (k<='Z' && used[k-'A'])
                k++;
            // cout<<k<<endl;
            
            for (auto& t:score)
                if (!used[t.first-'A'] && t.first!=k) {
                    vector<int> &a=score[k], &b=t.second;
                    int j=0;
                    while (j<n && a[j]==b[j])
                        j++;
                    if (j!=n && b[j]>a[j])
                        k=t.first;
                    if (j==n && t.first<k)
                        k=t.first;
                 }
            
            used[k-'A']=true;
            ans+=k;
        }
        
        
        return ans;
    }
};
