class Solution {
public:
	string rankTeams(vector<string>& votes) {
		int n = votes.size();
        set<char> st;
		vector<pair<vector<int>, int>> v(26);
		for (int i = 0; i < 26; i++) {
			v[i].first = vector<int>(votes[0].size());
			v[i].second = -i;
		}
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < votes[i].size(); j++) {
				v[votes[i][j] - 'A'].first[j]++;
                st.insert(votes[i][j]);
			}
		}
		sort(v.rbegin(), v.rend());
		string res;
		for (int i = 0; i < st.size(); i++) {
			res += (char)('A' - v[i].second);
		}
		return res;
	}
};
