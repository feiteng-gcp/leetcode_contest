class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n=votes[0].length();
        unordered_map<char, vector<int>> mp;
        
        //init map
        for(char c: votes[0])
        {
            mp[c].resize(n+1,0);
            mp[c][n]=c;
        }
        
        for(auto v: votes)
        {

            for(int i=0; i<v.length(); i++)
            {
               mp[v[i]][i]++;     
            }
        }
        
        string opt=votes[0];
        sort(opt.begin(), opt.end(), [&](char &a, char &b){
            for(int i=0; i<n; i++)
            {
               if(mp[a][i]!=mp[b][i])
                   return mp[a][i]>mp[b][i];
            }
            return a<b;
        });
        return opt;
    }
};
