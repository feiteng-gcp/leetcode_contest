class Solution {
public:
    
    struct cmp{
        int l;
        vector< vector<int> > c;
        cmp(int ll, const vector< vector<int> > &cc): l(ll),c(26, vector<int>(26)) {
            for(int i = 0; i < 26; ++i) {
                for(int j = 0; j < 26; ++j) {
                    c[i][j] = cc[i][j];
                }
            }
        }

       bool operator()(const char& a, const char& b) const {
           for(int i = 0; i < l; ++i) {
               if(c[a-'A'][i] != c[b-'A'][i]) {
                   return c[a-'A'][i] > c[b-'A'][i];
               }
           }
           return a < b;
       } 
    };
    string rankTeams(vector<string>& votes) {

        vector< vector<int> > c(26, vector<int>(26));
        for(int i = 0; i < votes.size(); ++i) {
            for(int j = 0; j < votes[i].length(); ++j) {
                c[ votes[i][j] - 'A' ][j] ++;
            }
        }
        sort(votes[0].begin(), votes[0].end(), cmp(votes[0].length(), c));
        return votes[0];
    }
};
