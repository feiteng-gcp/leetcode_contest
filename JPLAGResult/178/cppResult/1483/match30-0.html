class Solution {
public:
    unordered_map<char,int> id ;
    
    struct node {
        vector<int> count ;
        char c ;
    } ;
    
    static bool cmp(const node &A , const node &B) {
        for (int i = 0 ; i < A.count.size() ; ++i) {
            if (A.count[i] != B.count[i]) {
                return A.count[i] > B.count[i] ;
            }
        }
        return A.c < B.c ;
    }
    
    string rankTeams(vector<string>& votes) {
        int n = votes.size() , m = votes[0].length() ;
        string s = votes[0] ;
        sort(s.begin() , s.end()) ;
        for (int i = 0 ; i < m ; ++i)
            id[s[i]] = i ;
        vector<node> chars(m) ;
        for (int i = 0 ; i < m ; ++i) {
            chars[i].c = s[i] ;
            chars[i].count = vector<int> (m , 0) ;
        }
        for (int i = 0 ; i < n ; ++i) {
            for (int j = 0 ; j < m ; ++j) {
                chars[id[votes[i][j]]].count[j]++ ;
            }
        }
        sort(chars.begin() , chars.end() , cmp) ;
        string res = "" ;
        for (int i = 0 ; i < m ; ++i) {
            res += chars[i].c ;
        }
        return res ;
    }
};
