class Solution {
    set<char> visit;
public:
    string rankTeams(vector<string>& votes) {
        vector<vector<int>> dp(26, vector<int>(27));
        for(int i=0; i<26; i++)
            dp[i][26] = 25-i;
        for(auto &vote: votes) {
            for(int i=0; i<vote.size(); i++) {
                visit.insert(vote[i]);
                dp[vote[i]-'A'][i]++;
            }
        }
        sort(dp.rbegin(), dp.rend());
        string ans;
        for(int i=0; i<26; i++) {
            char key = 'A' + 25 - dp[i][26];
            if(visit.count(key) > 0) {
                ans += key;
            }
        }
        return ans;
    }
};
