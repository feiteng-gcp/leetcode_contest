class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int cnt[26][26];
        for (int i=0;i<26;i++) {
            for (int j=0;j<26;j++) {
                cnt[i][j] = 0;
            }
        }
        int N = votes.size(), M = votes[0].length();
        for (int i=0;i<N;i++) {
            for (int j=0;j<M;j++) {
                cnt[votes[i][j] - 'A'][j]++;
            }
        }
        vector<char>ans;
        for (int i=0;i<votes[0].size();i++) {
            ans.push_back(votes[0][i]);
        }
        sort(ans.begin(), ans.end(), [&](char c1, char c2) {
            int v1 = c1 - 'A', v2 = c2 - 'A';
            for (int i=0;i<M;i++) {
                if (cnt[v1][i] != cnt[v2][i]) {
                    return cnt[v1][i] > cnt[v2][i];
                }
            }  
            return c1 < c2;
        });
        string ans_str = "";
        for (int i=0;i<M;i++) {
            ans_str += ans[i];
        }
        return ans_str;
    }
};
