struct Node {
    char id;
    int total = 0;
    int fre[26] = {0};
};


bool cmp (const Node &a, const Node &b) {
    for (int i = 0; i < 26; i++) {
        if (a.fre[i] != b.fre[i]) return a.fre[i] > b.fre[i];
    }
    return a.id < b.id;
}

class Solution {
public:
    string rankTeams(vector<string>& v) {
        vector<Node> nodes;
        for (int i = 0; i < 26; i++) {
            Node node;
            node.id = i + 'A';
            node.total = 0;
            for (int i = 0; i < 26; i++) node.fre[i] = 0;
            nodes.push_back(node);
        }
        
        for (int i = 0; i < v.size(); i++) {
            for (int j = 0; j < v[i].size(); j++) {
                int id = v[i][j] - 'A';
                nodes[id].total++;
                nodes[id].fre[j]++;
            }
        }
        
        sort(nodes.begin(), nodes.end(), cmp);
        string res = "";
        for (int i = 0; i < nodes.size(); i++) {
            if (nodes[i].total != 0) res += nodes[i].id;
        }
        return res;
    }
};
