class Solution {
public:
    string rankTeams(vector<string>& votes) {
        string ans = votes[0];
        int ts = ans.size();
        int v[26][26] = {};

        for (const auto &vote : votes) {
            for (int i = 0; i < ts; ++i) {
                v[vote[i] - 'A'][i]++;
            }
        }
        sort(ans.begin(), ans.end(), [ts, v] (char a, char b) {
            for (int i = 0; i < ts; ++i) {
                int comp = v[a - 'A'][i] - v[b - 'A'][i];

                if (comp > 0) {
                    return true;
                } else if (comp < 0) {
                    return false;
                }
            }
            return a < b;
        });
        return ans;
    }
};
