struct Team
{
 char name;
 vector<int> counts;
};

bool compare(Team a, Team b)
{
 int n = a.counts.size();
 for(int i = 0; i < n; ++i)
 {
   if(a.counts[i] != b.counts[i]) return a.counts[i] > b.counts[i];
 }

 return a.name < b.name;
}

class Solution
{
 public:

 string rankTeams(vector<string> &votes)
 {
   int n = votes.size();
   int m = votes[0].size();

   string names = votes[0];
   sort(names.begin(), names.end());

   vector<Team> teams;
   map<char, Team> M;
   for(int i = 0; i < names.length(); ++i)
   {
     Team team;
     team.name = names[i];
     vector<int> counts;
     for(int j = 0; j < m; ++j)
     {
       counts.push_back(0);
     }

     team.counts = counts;
     M[names[i]] = team;
   }

   for(int i = 0; i < n; ++i)
   {
     for(int j = 0; j < votes[i].size(); ++j)
     {
       char name = votes[i][j];
       Team team = M[name];
       team.counts[j]++;
       M[name] = team;
     }
   }

   for(auto t : M) teams.push_back(t.second);

   string result = "";
   sort(teams.begin(), teams.end(), compare);
   for(int i = 0; i < teams.size(); ++i) result.push_back(teams[i].name);

   return result;
 }
};
