class Solution {
public:
    string rankTeams(vector<string>& votes) {
    int n,cnt[200][200];
        memset(cnt,0,sizeof(cnt));
        for(auto s : votes)
        {
            n=s.length();
            for(int i=0;i<n;++i)
            {
                ++cnt[s[i]][i];
            }
        }
        string ans=votes[0];
        sort(ans.begin(),ans.end(),[n,cnt](char a,char b){
            for(int i=0;i<n;++i)
            {
                if(cnt[a][i]!=cnt[b][i])
                    return cnt[a][i]>cnt[b][i];
            }
            return a<b;
        });
        return ans;
    }
};
