class Solution {
public:
    string rankTeams(vector<string>& votes) {
        vector<array<int, 27>> dp(26);
        
        for (int i = 0; i < 26; i++) 
            dp[i][26] = -i;
        
        int lv = votes[0].size();
        for (string& v : votes) {
            for (int i = 0; i < lv; i++) {
                dp[v[i]-'A'][i]++;
            }
        }
        sort(dp.rbegin(), dp.rend());
        
        string ans;
        for (int i = 0; i < lv; i++) {
            ans.push_back(-dp[i][26]+'A');
        }
        return ans;
    }
};
