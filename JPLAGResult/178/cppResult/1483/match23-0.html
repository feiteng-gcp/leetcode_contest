class Team {
public:
  char name;
  vector<int> rank;
  Team() {};
  Team(char name_, int n_team) {
    name = name_;
    rank = vector<int>(n_team, 0);
  }
  bool operator< (const Team& another) const { // who goes first
    for (int i = 0; i < rank.size(); ++i) {
      if (rank[i] > another.rank[i]) {
        return true;
      }
      if (rank[i] < another.rank[i]) {
        return false;
      }
    }
    return name < another.name;
  }
};


class Solution {
public:
  string rankTeams(vector<string>& votes) {
    unordered_map<char, Team> teams_map;
    for (auto t: votes[0]) {
      teams_map.emplace(t, Team(t, votes[0].size()));
    }
    for (auto& v: votes) {
      for (int i = 0; i < v.size(); ++i) {
        teams_map[v[i]].rank[i] += 1;
      }
    }
    vector<Team> teams_arr;
    for (auto t: votes[0]) {
      teams_arr.push_back(move(teams_map[t]));
    }
    sort(teams_arr.begin(), teams_arr.end());
    string ans;
    for (auto& t: teams_arr) {
      ans.push_back(t.name);
    }
    return ans;
  }
};
