class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n = votes[0].length();
        vector<vector<int>> r(26, vector<int>(n + 1, 0));
        for (auto s: votes) {
            for (int i = 0; i < n; ++i) {
                // cout << n - i << " ";
                r[s[i] - 'A'][i] -= 1;
                r[s[i] - 'A'][n] = s[i];
            }
        }
        sort(r.begin(), r.end());
        // for (int i = 0; i < r.size(); ++i) {
        //     cout << r[i][n] << " ";
        //     for (int j = 0; j < n; ++j) {
        //         cout << (int)r[i][j] << " ";
        //     }
        //     cout<<endl;
        // }
        string ret;
        for (int i = 0; i < r.size(); ++i) {
                            // cout << i << " " <<  r[i]<<endl;

            if (r[i][n] == 0) continue;
            ret += (char)r[i][n];
        }
        // reverse(ret.begin(), ret.end());
        return ret;
    }
};
