class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int cnt[26][26] = {0};
        int l = votes[0].size();
        set<int> vs;
        for(auto c:votes[0]) vs.insert(c-'A');
        for(auto &s : votes){
            for(int i=0;i<l;i++){
                cnt[s[i]-'A'][i]++;
            }
        }
        vector<int> v(26);
        for(int i=0;i<26;i++) v[i] = i;
        sort(v.begin(), v.end(), [&](int a, int b){
            for(int i=0;i<26;i++){
                if(cnt[a][i] != cnt[b][i]){
                    return cnt[a][i] > cnt[b][i];
                }
            }
            return a < b;
        });
        string s;
        for(int i=0;i<26;i++){
            if(vs.count(v[i]) > 0){
                s.push_back(v[i]+'A');
            } 
        }
        return s;
    }
};
