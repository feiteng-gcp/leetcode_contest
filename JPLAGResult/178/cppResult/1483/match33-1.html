class Solution {
public:
    struct nod{
        char s;
        vector<int> v;

        friend bool operator<(nod a,nod b){
            for(int i=0;i<a.v.size();i++){
                if(a.v[i]<b.v[i])return false;
                else if(a.v[i]>b.v[i])return true;
                
            }
            return a.s<b.s;
        }
    };
    string rankTeams(vector<string>& v) {
        int n = v.size();
        vector<nod> g;
        int m = v[0].size();
        for(int i=0;i<m;i++){
            nod t;t.s =v[0][i];
            t.v.resize(m);
            fill(t.v.begin(),t.v.end(),0);
            t.v[i]++;
            g.push_back(t);
        }
        for(int i=1;i<n;i++){

            for(int j=0;j<m;j++){
                for(int k=0;k<m;k++){
                    if(g[k].s == v[i][j]){
                        g[k].v[j]++;
                    }
                }
            }
        }
        sort(g.begin(),g.end());
        string s ="";
        for(int i=0;i<g.size();i++){
            s+=g[i].s;
        }
        return s;

    }
};
