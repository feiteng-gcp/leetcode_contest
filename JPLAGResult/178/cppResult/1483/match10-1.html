class Solution {
public:
    string rankTeams(vector<string>& votes) {

        vector<vector<int>> v(26, vector<int> (27, 0));
        for (int i = 0; i < 26; i++) v[i][26] = -i;
        int n = votes[0].size();
        for (string vote : votes) {
            for (int i = 0; i < n; i++) {
                v[vote[i]-'A'][i]++;
            }
        }
        sort(v.rbegin(), v.rend());
        string res;
        res.reserve(n);
        for (int i = 0; i < n; i++) {
            res += (char) ('A'-v[i][26]);
        }
        return res;
    }
};
