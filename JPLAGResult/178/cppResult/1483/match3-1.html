typedef struct node
{
    char c;
    int e[26];
} Node;

bool cmp(Node a, Node b)
{
    for (int i=0;i<26;i++)
    {
        if (a.e[i]>b.e[i]) return true;
        if (a.e[i]<b.e[i]) return false;
    }

    if (b.c<a.c) return false;
    return true;
}

class Solution {
public:
    string rankTeams(vector<string>& votes) {
        string ret="";
        vector<Node> s(0);
        vector<bool> visit(0);
        for (int i=0;i<26;i++) visit.push_back(false);
        for (char c='A';c<='Z';c++)
        {
            Node temp;
            temp.c=c;
            for (int i=0;i<26;i++) temp.e[i]=0;
            s.push_back(temp);
        }
        for (auto &str:votes)
        {
            for(int i=0;i<str.length();i++)
            {
                s[str[i]-'A'].e[i]++;
            }
            for (auto c:str) visit[c-'A']=true;
        }
        
        sort(s.begin(),s.end(),cmp);
        //for (auto a:)
        for (int i=0;i<26;i++)
        {
            if (!visit[s[i].c-'A']) continue;
            ret=ret+char(s[i].c);
        }
        return ret;
    }
};
