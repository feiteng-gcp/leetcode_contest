class Solution {
public:
    string rankTeams(vector<string>& votes) {
        string ans ="";
        vector<int> flag(26);
        vector<vector<int> >sum(26, vector<int>(26));
        for(int i = 0; i < votes.size(); i++)
        {
            for(int j = 0; j < votes[i].size(); j++)
            {
                if(!flag[votes[i][j] - 'A'])
                {
                    flag[votes[i][j] - 'A'] = 1;
                    ans += votes[i][j];
                }
                sum[votes[i][j] - 'A'][j]++;
            }
        }
        cout<<ans<<endl;
        sort(ans.begin(), ans.end(), [&](char a, char b)->bool{
             for(int i = 0; i < 26; i++)
             {
                 if(sum[a - 'A'][i] < sum[b - 'A'][i])
                    return false;
                 else if(sum[a - 'A'][i] > sum[b - 'A'][i])
                     return true;
             }
             return a < b;
             });
        return ans;
    }
};
