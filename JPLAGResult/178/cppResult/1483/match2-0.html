class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n = votes[0].size();
        vector<vector<int> > rank(26, vector<int>(26, 0));
        for (auto s: votes) {
            for (int i = 0; i < n; ++i) rank[s[i]-'A'][i]++;
        }
        
        vector<pair<char, vector<int> > > ret;
        for (int i = 0; i < votes[0].size(); ++i) {
            ret.push_back({votes[0][i], rank[votes[0][i]-'A']});
        }
        
        sort(ret.begin(), ret.end(), [](auto a, auto b) {
            for (int i = 0; i < 26; ++i) if (a.second[i] != b.second[i]) {
                return a.second[i] > b.second[i];
            }
            return a.first < b.first;
        });
        
        string ans;
        for (auto x: ret) ans += x.first;
        return ans;
    }
};
