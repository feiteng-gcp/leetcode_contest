class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n = static_cast<int>(votes.size());
        int m = static_cast<int>(votes[0].length());
        vector<vector<int>> count(26, vector<int>(m));

        for (auto vote : votes) {
            for (int i = 0; i < m; ++i) {
                ++count[vote[i] - 'A'][i];
            }
        }
        vector<int> teams(26);
        for (int i = 0; i < 26; ++i) {
            teams[i] = i;
        }

        sort(teams.begin(), teams.end(), [&count, &m](int a, int b) {
            for (int i = 0; i < m; ++i) {
                if (count[a][i] != count[b][i]) {
                    return count[a][i] > count[b][i];
                }
            }
            return a < b;
        });
        string rank;
        for (int i = 0; i < m; ++i) {
            rank += static_cast<char>(teams[i] + 'A');
        }
        return rank;
    }
};
