class Solution {
public:
    string rankTeams(vector<string>& votes) {
        vector<pair<vector<int>, char>> teams(26);

        for (int i = 0; i < teams.size(); ++i) {
            teams[i].first.resize(votes[0].size());
            teams[i].second = 'A' + i;
        }
        for (auto& v : votes) {
            for (int i = 0; i < v.size(); ++i) {
                teams[v[i] - 'A'].first[i] -= 1;
            }
        }
        sort(teams.begin(), teams.end());
        string res;
        for (auto& t : teams) {
            if (std::accumulate(t.first.begin(), t.first.end(), 0)) {
                res += t.second;
            }
        }
        return res;
    }
};
