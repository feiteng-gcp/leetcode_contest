class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n = votes.size(), m = votes[0].size(), K = max(m, n);
        vector<vector<int>> teams(256, vector<int>(m+1, 0));
        for (auto & s : votes){
            for (int i = 0; i < s.size(); ++i){
                char a = s[i];
                
                teams[a][i] += 1;
                teams[a][m]  = 'Z' - (a - 'A');
                
                cout << a << ":" << i << ", " << teams[a][i] << endl;
            }
        }
        vector<vector<int>> tt;
        for (int a = 'A'; a <= 'Z'; ++a){
            if (teams[a][m] != 0){
                tt.push_back(teams[a]);
            }
        }
        
        sort(tt.begin(), tt.end());
        
        string res; int L = tt.size();
        for (int i = 0; i < L; ++i){
            res = string(1, 'Z' - (tt[i][m] - 'A')) + res;
        }
        return res;
    }
};
