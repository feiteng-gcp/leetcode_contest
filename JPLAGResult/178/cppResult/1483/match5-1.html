class Solution {
public:
    string rankTeams(vector<string>& votes) {
        map<char, vector<int>> m;
        
        for (int i = 0; i < votes.size(); ++i) {
            for (int j = 0; j < votes[i].size(); ++j) {
                auto it = m.find(votes[i][j]);
                if (it == m.end()) {
                    m[votes[i][j]] = vector<int>(votes[i].size(), 0);
                }
                m[votes[i][j]][j]++;
            }
        }
        
        vector<pair<char, vector<int>>> p;
        for (auto &e: m) {
            p.push_back(e);
        }
        sort(p.begin(), p.end(), [&](pair<char, vector<int>> &p1, pair<char, vector<int>> &p2) {
            for (int i = 0; i < p1.second.size(); ++i) {
                if (p1.second[i] > p2.second[i]) return true;
                if (p1.second[i] < p2.second[i]) return false;
            }    
            return p1.first < p2.first;
        });
        
        string ret;
        for (int i = 0; i < p.size(); ++i) ret += p[i].first;
        return ret;
    }
};
