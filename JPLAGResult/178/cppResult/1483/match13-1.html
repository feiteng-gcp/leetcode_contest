class Solution {

public:
    string rankTeams(vector<string>& votes) {
        
        vector<vector<int>> a(26, vector<int>(27, 0));
        for (int i = 0; i < 26; ++ i) a[i][26] = i;
        
        for (auto vote: votes) {
            for (int i = 0; i < vote.length(); ++ i) {
                -- a[vote[i] - 'A'][i];
            }
        }
        
        sort(begin(a), end(a));
        string ans;
        for (int i = 0; i < 26; ++ i) {
            if (accumulate(begin(a[i]), end(a[i]), 0) == a[i].back()) continue;
            char c = 'A' + a[i].back();
            ans.push_back(c);
        }
        
        return ans;
    }
};
