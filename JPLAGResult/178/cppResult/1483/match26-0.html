class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n = votes[0].length();
        vector<vector<int>> t(26, vector<int>(n, 0)); //team_letter:position:number_of_votes
        
        
        for (auto vote: votes) {
            for (int i = 0; i < n; i++) {
                int tl = vote[i] - 'A';
                t[tl][i]++;
            }
        }
        
        string ans = votes[0];
        
        sort(ans.begin(), ans.end(), [&](const char& e, const char& v) {
            int tl_e = e - 'A', tl_v = v - 'A';
            for (int i = 0; i < n; i++) {
                if (t[tl_e][i] > t[tl_v][i])
                    return true;
                else if (t[tl_e][i] < t[tl_v][i])
                    return false;
            }
            return e < v;
        });
        
        return ans;
        
    }
};
