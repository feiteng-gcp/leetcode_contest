#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    string rankTeams(vector<string>& votes) {
        int n = votes.size();
        int m = votes[0].size();
        vector<pair<vector<int>, int>> v(26);
        for (int i = 0; i < 26; i++) {
            v[i].first = vector<int>(m, 0);
            v[i].second = -i;
        }
        for (int i = 0; i < n; i++) {
            auto& w = votes[i];
            for (int j = 0; j < m; j++) {
                int c = votes[i][j] - 'A';
                v[c].first[j]++;
            }
        }
        sort(v.rbegin(), v.rend());
        string s;
        for (int i = 0; i < m; i++) {
            s += (char) ('A' - v[i].second); 
        }
        return s;
    }
};
