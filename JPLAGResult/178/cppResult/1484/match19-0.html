/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isSubPath(ListNode* head, TreeNode* root) {
        /*
        ListNode * prev = NULL, * next;
        while (head){
            next = head->next;
            head->next = prev;
            
            prev = head;
            head = next;
        }
        */
        
        vector<int> hhh;
        while (head){
            hhh.push_back(head->val);
            head = head->next;
        }
        
        cout << hhh.size() << endl;
        
        vector<int> lll; bool bFound = false;
        function<void(TreeNode *)> dfs;
        dfs = [&](TreeNode * node){
            if ((node == NULL) || (bFound)){return;}
            
            lll.push_back(node->val);
            int m = lll.size(), n = hhh.size();
            
            if (m >= n){
                bool found = true;
                for (int i = m - 1, j = n - 1; j >= 0; --i, --j){
                    if (lll[i] != hhh[j]){
                        found = false;
                        break;
                    }
                }
                if (found){bFound = true;}
            }
            
            dfs(node->left);
            dfs(node->right);
            
            lll.pop_back();
        };
        
        
        
        dfs(root);
        return bFound;
    }
};
