/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    
    bool isValid(ListNode* head, TreeNode* root) {
        if (!head) return true;
        if (!root) return false;
        // cout << head->val << " " << root->val << endl;
        if (head->val != root->val) return false;
        return isValid(head->next, root->left) || isValid(head->next, root->right);
    }
    
    bool dfs(ListNode* head, TreeNode* root) {
        if (isValid(head, root)) return true;
        if (!root) return false;
        // if (root->val == 4)
            // cout << isValid(head, root) << " " << root->val << endl;
        // cout << "))" << endl;
        return dfs(head, root->left) || dfs(head, root->right);
    }
    
    bool isSubPath(ListNode* head, TreeNode* root) {
        return dfs(head, root);
    }
};
