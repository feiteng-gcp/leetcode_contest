/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
vector<int> path(0);
vector<int> aim(0);

class Solution {
public:
    bool can()
    {
        //for (auto a:path) cout<<a<<' ';
        //cout<<endl;
        //cout<<"========="<<endl;
        if (path.size()<aim.size()) return false;
        int d=path.size()-aim.size();
        for (int i=0;i<aim.size();i++)
        {
            if (aim[i]!=path[i+d]) 
            {
                //cout<<"false"<<endl;
                return false;
            }
        }
        //cout<<"true"<<endl;
        return true;
    }
    
    bool travel(TreeNode* root)
    {
        if (root==NULL) return false;
        path.push_back(root->val);
        if (can()) return true;
        if (travel(root->left)) return true;
        if (travel(root->right)) return true;
        path.pop_back();
        return false;
    }
    bool isSubPath(ListNode* head, TreeNode* root) {
        aim.clear();
        path.clear();
        if (root==NULL) return false;
        if (head==NULL) return true;
        for (auto p=head;p!=NULL;p=p->next)
        {
            aim.push_back(p->val);
        }
        return travel(root);
    }
};
