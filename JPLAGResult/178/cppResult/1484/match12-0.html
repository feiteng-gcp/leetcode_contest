/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool check(TreeNode* t_node, ListNode* l_node) {
        if (!l_node) return true;
        if (!t_node) return false;
        if (t_node->val != l_node->val) return false;
        return check(t_node->left, l_node->next) || check(t_node->right, l_node->next);
    }
    
    bool dfs(TreeNode* node, ListNode* head) {
        if (!node) return false;
        if (check(node, head)) return true;
        return dfs(node->left, head) || dfs(node->right, head);
    }
    
    bool isSubPath(ListNode* head, TreeNode* root) {
        return dfs(root, head);
    }
};
