class Solution {
public:
    vector<int> smallerNumbersThanCurrent(vector<int>& v) {
        vector<int> res;
        for (int i = 0; i < v.size(); i++) {
            int cnt = 0;
            for (int j = 0; j < v.size(); j++) {
                if (v[j] < v[i]) cnt++;
            }
            res.push_back(cnt);
        }
        return res;
    }
};
