class Solution {
public:
    vector<int> smallerNumbersThanCurrent(vector<int>& nums) {
        vector<int> c = nums;
        sort(c.begin(), c.end());
        
        vector<int> ret(nums.size(), 0);
        for (int i = 0; i < nums.size(); ++i) {
            auto it = lower_bound(c.begin(), c.end(), nums[i]);
            ret[i] = distance(c.begin(), it);
        }
        return ret;
    }
};
