# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        self.expectedList = []
        ptr = head
        while ptr is not None:
            self.expectedList.append(ptr.val)
            ptr = ptr.next
        self.expectedListLen = len(self.expectedList)
        return self.dfsVisit(root, [])
    def dfsVisit(self, ptr, existedList):
        if ptr is None:
            return False
        thisList = existedList + [ptr.val]
        if len(thisList) >= self.expectedListLen:
            sublist = thisList[-self.expectedListLen:]
            if self.identicalList(sublist, self.expectedList):
                return True
        if self.dfsVisit(ptr.left, thisList):
            return True
        elif self.dfsVisit(ptr.right, thisList):
            return True
        else:
            return False
    def identicalList(self, list1, list2):
        if len(list1) != len(list2):
            return False
        return sum([1 for i, j in zip(list1, list2) if i != j]) == 0
