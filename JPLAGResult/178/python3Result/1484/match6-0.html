# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        
        from functools import lru_cache
        @lru_cache(None)
        def dfs(lis, node):
            if not lis: return True
            if not node: return False
            res = False
            res |= dfs(head, node.left)
            res |= dfs(head, node.right)
            if lis.val == node.val:
                res |= dfs(lis.next, node.left)
                res |= dfs(lis.next, node.right)
            return res
            
            
        return dfs(head, root)
