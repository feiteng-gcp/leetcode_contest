# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        def helper(node, treeNode):
            if not node: return True
            if not treeNode: return False
            if node.val != treeNode.val: return False
            left = helper(node.next, treeNode.left)
            right = helper(node.next, treeNode.right)
            return left or right
        
        queue = collections.deque([root])
        while queue:
            node = queue.popleft()
            if node.val == head.val:
                if helper(head, node): 
                    return True
            if node.left:
                queue.append(node.left)
            if node.right:
                queue.append(node.right)
        
        return False
