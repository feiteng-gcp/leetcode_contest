# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
from collections import deque
class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        node_list = []
        def inorder(r):
            if not r:
                return 
            inorder(r.left)
            if r.val == head.val:
                node_list.append(r)
            inorder(r.right)
        
def varify(node, h):
            
            if not h:
                return True
            if not node:
                return False
            if node.val != h.val:
                return False
            return varify(node.right, h.next) or varify(node.left, h.next)
        inorder(root)
        for n in node_list:
            if varify(n, head):
                return True
        return False
