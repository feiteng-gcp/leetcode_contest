# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
from collections import deque
class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        
        if not head:
            return True
        
        def dfs(lnode, tnode):
            if not lnode:
                return True
            if not tnode:
                return False
            
            if lnode.val == tnode.val:
                return dfs(lnode.next, tnode.left) or dfs(lnode.next, tnode.right)
            return False
            
        q = deque([root])
        while q:
            node = q.popleft()
            if node.val == head.val and dfs(head, node):
                return True
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        return False
