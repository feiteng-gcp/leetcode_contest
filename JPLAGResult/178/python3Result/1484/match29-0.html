# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        def search_path(tree_node, search_node):
            if not search_node:
                return True
            if not tree_node:
                return False
            if (tree_node, search_node) in memo:
                return memo[(tree_node, search_node)]
            
            if tree_node.val == search_node.val:
                if search_path(tree_node.left, search_node.next) or search_path(tree_node.right, search_node.next):
                    memo[(tree_node, search_node)] = True
                    return True
               
            if search_path(tree_node.left, head) or search_path(tree_node.right, head):
                memo[(tree_node, search_node)] = True
                return True
            
            memo[(tree_node, search_node)] = False
            return False
        
        
        memo = {}
        return search_path(root, head)
