# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        dp = {}
        
def dfs(tn, ln):
            if not ln:
                return True
            if not tn:
                return False
            if (tn, ln) in dp:
                return dp[(tn, ln)]
            dp[(tn, ln)] = False
            
if tn.val == ln.val:
                if dfs(tn.left, ln.next) or dfs(tn.right, ln.next):                    
                    dp[(tn, ln)] = True
            else:
                dp[(tn, ln)] = False
            return dp[(tn, ln)] 
        def has(node):
            if not node:
                return head == None
            return dfs(node, head) or has(node.left) or has(node.right)
        return has(root)
