# Definition for singly-linked list.

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        '''
        Algo: Find head, and for each head, enumerate
        
        2500 x 100 (?)
        
        may not be 100. there may be multiple paths matching a prefix of the linked list.
        Exponential.
        
        worst case: 
        - full binary tree, with all 1's expcet for a leaf being 2
        - list is all 1's except last element being 2
        
        There is still an upperbound. 
        Starting from each head, we at most tranverse the depth of the tree
        
        so:
        2500 x max(100, log(2500))
        bounded.
        
        
        '''
        
        def tranverse(root, head):
            if head is None:
                return True
            if root is None:
                return False
            if root.val != head.val:
                return False
            re = tranverse(root.left, head.next)
            if re:
                return True
            re = tranverse(root.right, head.next)
            if re:
                return True
        
        def search_head(root, head):
            if root is None or head is None:
                return False
            
            if root.val == head.val:
                # print(root.val)
                re = tranverse(root, head)
                if re:
                    
return True
                
            re = search_head(root.left, head)
            if re:
                return True
            
            re = search_head(root.right, head)
            if re:
                return True
            
            return False
        
        return search_head(root, head)
