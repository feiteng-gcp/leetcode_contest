# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        found=False
        def height(node):
            if not node:
                return 0
            return 1 + max(height(node.left), height(node.right))
        ht=height(root)
        ln=0
        cur=head
        while cur:
            ln+=1
            cur=cur.next
        oldhead=head
        def dfs(node, ht, head, ln):
            if not head:
                return True
            
if not node:
                return False
            if ht<ln:
                return False
            if node.val == head.val:
                #print("test %d, %d at height %d, ln %d"%(node.val, head.val, ht, ln))
                if dfs(node.left, ht-1, head.next, ln-1) or dfs(node.right, ht-1, head.next, ln-1):
                    return True
            if head==oldhead:
                if dfs(node.left, ht-1, head, ln) or dfs(node.right, ht-1, head, ln):
                    return True
            return False
        return dfs(root, ht, head, ln)
