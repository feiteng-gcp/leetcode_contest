# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        if not root and head: return False
        
        
def isSubPathFromRoot(head, root):
            if head is None: return True
            if root is None: return False
            if (head.val != root.val): return False
            return isSubPathFromRoot(head.next, root.left) or isSubPathFromRoot(head.next, root.right)
        
        return isSubPathFromRoot(head, root) or (root and (self.isSubPath(head, root.left) or self.isSubPath(head, root.right)))
