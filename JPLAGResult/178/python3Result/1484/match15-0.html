# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def check(self, p, root):
        if not p: return True
        if not root: return False
        if p.val != root.val: return False
        return self.check(p.next, root.left) or self.check(p.next, root.right)
    
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        if not head: return True
        if not root: return False
        res = False
        if head.val == root.val:
            res = self.check(head, root)
        if res: return True
        res = self.isSubPath(head, root.left)
        if res: return True
        return self.isSubPath(head, root.right)
