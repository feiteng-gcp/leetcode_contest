# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode, is_check=False) -> bool:
        if head is None:
            return True
        if root is None:
            return False
        if head.val == root.val:
            if self.isSubPath(head.next, root.left, True):
                return True
            if self.isSubPath(head.next, root.right, True):
                return True
        if not is_check:
            if self.isSubPath(head, root.left, False):
                return True
            if self.isSubPath(head, root.right, False):
                return True
        return False
