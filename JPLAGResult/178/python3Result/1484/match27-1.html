# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isSubPath(self, head: ListNode, root: TreeNode) -> bool:
        def is_sub_path(head, root):
            if head is None:
                return True
            if root is None:
                return False
            return (
                is_sub_path_from_here(head, root)
                or is_sub_path(head, root.left) or is_sub_path(head, root.right)
            )
            
        
def is_sub_path_from_here(head, root):
            if head is None:
                return True
            if root is None:
                return False
            if head.val != root.val:
                return False
            return is_sub_path_from_here(head.next, root.left) or is_sub_path_from_here(head.next, root.right)
        
        return is_sub_path(head, root)
