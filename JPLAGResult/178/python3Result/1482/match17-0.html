class Solution:
    def smallerNumbersThanCurrent(self, nums: List[int]) -> List[int]:
        r = [0] * len(nums)
        for i in range(len(nums)):
            for j in range(len(nums)):
                if j == i:
                    continue
                if nums[j] < nums[i]:
                    r[i] += 1
        return r
