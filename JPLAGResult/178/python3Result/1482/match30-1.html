class Solution:
    def smallerNumbersThanCurrent(self, nums: List[int]) -> List[int]:
        n = len(nums)
        ret = [0] * n

        for i in range(n):
            for j in range(n):
                if i == j : continue
                if nums[j] < nums[i] :
                    ret[i] += 1
        
        return ret
