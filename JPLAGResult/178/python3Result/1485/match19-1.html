class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        MAX=1000
        m=len(grid)
        n=len(grid[0])
        a=[[MAX for _ in range(n)] for i in range
(m)]
        a[0][0]=0
        L=[(0,0)]
        i=0
        while i<len(L):
            p=L[i]
            d=a[p[0]][p[1]]
            #print(d,a[0][1])
            if grid[p[0]][p
[1]]==1 and p[1]+1<n:
                #print('test1')
                if a[p[0]][p[1]+1]>d:
                    #print('test2',p[0],p[1]+1,d)
                    a[p[0]][p[1]+1]=d
                    #print(a[0][1])
                    L.append((p[0],p[1]+1))
            if grid[p[0]][p
[1]]==2 and p[1]-1>=0:
                if a[p[0]][p[1]-1]>d:
                    a[p[0]][p[1]-1]=d
                    L.append((p[0],p[1]-1))                
            if grid[p[0]][p[1]]==3 and p[0]+1<m:
                if a[p[0]+1][p[1]]>d:
                    a[p[0]+1][p[1]]=d
                    L.append((p[0]+1,p[1]))
            if grid[p[0]][p[1]]==4 and p[0]-1>=0:
                if a[p[0]-1][p[1]]>d:
                    a[p[0]-1][p[1]]=d
                    L.append((p[0]-1,p[1]))
            if p[1]+1<n:
                if a[p[0]][p[1]+1]>d+1:
                    a[p[0]][p[1]+1]=d+1
                    L.append((p[0],p[1]+1))
            if p[1]-1>=0:
                if a[p[0]][p[1]-1]>d+1:
                    a[p[0]][p[1]-1]=d+1
                    L.append((p[0],p[1]-1))                
            if p[0]+1<m:
                if a[p[0]+1][p[1]]>d+1:
                    a[p[0]+1][p[1]]=d+1
                    L.append((p[0]+1,p[1]))
            if p[0]-1>=0:
                if a[p[0]-1][p[1]]>d+1:
                    a[p[0]-1][p[1]]=d+1
                    L.append((p[0]-1,p[1]))
            i+=1
        #print(a)
        return a[m-1][n-1]
