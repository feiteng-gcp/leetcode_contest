class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        n = len(grid)
        m = len(grid[0])
        dirs = [None, (0, 1), (0, -1), (1, 0), (-1, 0)]
        dist = [[None] * m for _ in range(n)]
        done = [[False] * m for _ in range(n)]
        q = [(0, 0, 0)]
        
        for l in grid:
            print(["><V^"[x - 1] for x in l])
        
        def dfs(i, j, d):
            if done[i][j]:
                return
            done[i][j] = True
            dist[i][j] = d
            for xx, xy in [(i - 1, j), (i, j - 1), (i + 1, j), (i, j + 1)]:
                if xx < 0 or xy < 0 or xx >= n or xy >= m:
                    continue
                q.append((xx, xy, d + 1))
            di, dj = dirs[grid[i][j]]
            x, y = i + di, j + dj
            if x < 0 or y < 0 or x >= n or y >= m:
                return
            dfs(x, y, d)
        for x, y, d in q:
            if done[x][y] or dist[x][y] is not None and dist[x][y] < d:
                continue
            dfs(x, y, d)
        for l in dist:
            print(l)
        return dist[-1][-1]
