from functools import lru_cache
from heapq import heappush, heappop

class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        def cnv(x):
            if x == 1:
                return (0, 1)
            if x == 2:
                return (0, -1)
            if x == 3:
                return (1, 0)
            if x == 4:
                return (-1, 0)
            
        vis = set()
        q = []
        heappush(q, (0, 0, 0))
        while q:
            cost, r, c = heappop(q)
            if r == len(grid) - 1 and c == len(grid[0]) - 1:
                return cost
            
            if (r, c) in vis:
                continue
            
            vis.add((r, c))
            
            for i in range(1, 5):
                ccnv = cnv(i)
                nr = r + ccnv[0]
                nc = c + ccnv[1]
                if (nr, nc) in vis:
                    continue
                if nr < 0 or nc < 0 or nr >= len(grid) or nc >= len(grid[0]):
                    continue
                heappush(q, (cost + (1 if i != grid[r][c] else 0), nr, nc))
            
        return 1e9
