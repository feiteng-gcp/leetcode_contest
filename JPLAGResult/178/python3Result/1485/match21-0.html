from heapq import heappush, heappop
class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        steps = [[10001] * n for i in range(m)]
        
        def find_connected(i, j, visited):
            if (i, j) in visited:
                return
            visited.add((i, j))
            if i > 0 and grid[i-1][j] == 3:
                find_connected(i-1, j, visited)
            if i < m - 1 and grid[i+1][j] == 4:
                find_connected(i+1, j, visited)
            if j > 0 and grid[i][j-1] == 1:
                 find_connected(i, j-1, visited)
            if j < n - 1 and grid[i][j+1] == 2:
                 find_connected(i, j+1, visited)
                    
        def get_nearby(i, j, step, res):
            if i > 0 and steps[i-1][j] > step:
                res.add((3, i-1, j))
            if i < m - 1 and steps[i+1][j] > step:
                res.add((4, i+1, j))
            if j > 0 and steps[i][j-1] > step:
                res.add((1, i, j-1))
            if j < n - 1 and steps[i][j+1] > step:
                res.add((2, i, j+1))
        
        # count = 1000
        pq = [(0, grid[m - 1][n - 1], m - 1, n - 1)]
        while len(pq) > 0:
            # print(pq)
            step, pointer, i, j = heappop(pq)
            if step >= steps[i][j]:
                continue
            grid[i][j] = pointer
            visited = set()
            find_connected(i, j, visited)
            for i, j in visited:
                if i == 0 and j == 0:
                    return step
                steps[i][j] = step
            # print("visited: ", visited)
            res = set()
            for i, j in visited:
                get_nearby(i, j, step, res)
            for pointer, p, q in res:
                heappush(pq, (step + 1, pointer, p, q))
            # print("res: ", res)
            # count -= 1
        return m + n - 2
