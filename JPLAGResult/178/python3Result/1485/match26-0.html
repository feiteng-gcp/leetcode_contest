class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        h = len(grid)
        w = len(grid[0])
        
        dist = [[float("inf")]*w for _ in range(h)]
        dist[0][0] = 0
        q = []
        heapq.heappush(q, (0, 0, 0))

        while q:
            prov_cost, i, j = heapq.heappop(q)
            # print(i, j)
            
            if dist[i][j] < prov_cost:
                continue

            # 他の頂点の探索
            for idx, dest in enumerate([(0,1), (0,-1), (1,0), (-1,0)]):
                di, dj = dest
                if 0<=i+di<h and 0<=j+dj<w:
                    cost = 0 if grid[i][j] == idx+1 else 1
                    if dist[i+di][j+dj] > dist[i][j] + cost:
                        dist[i+di][j+dj] = dist[i][j] + cost
                        heapq.heappush(q, (dist[i+di][j+dj], i+di, j+dj))
            # print(dist)
                        
        return dist[-1][-1]
