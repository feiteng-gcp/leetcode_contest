class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        M, N = len(grid), len(grid[0])
        if M == 1: return 0
        dirs = ((0, 1), (0, -1), (1, 0), (-1, 0))
        heap = [(0, 0, 0)]
        dis = [[float('inf') for _ in range(N)] for _ in range(M)]
        while heap:
            cost, x, y = heapq.heappop(heap)
            # if cost == M + N - 3: return cost
            for i, (dx, dy) in enumerate(dirs):
                nx, ny = x + dx, y + dy
                ncost = cost + (1 if i + 1 != grid[x][y] else 0)
                # if nx == M -1 and ny == N-1: return ncost
                if 0 <= nx < M and 0 <= ny < N and dis[nx][ny] > ncost :
                    dis[nx][ny] = ncost
                    heapq.heappush(heap, (ncost, nx, ny))
        return dis[-1][-1]
