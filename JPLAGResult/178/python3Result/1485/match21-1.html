class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        n,m=len(grid),len(grid[0])
        dp=[[-1]*m for _ in range(n)]
        def fill(i,j,val):
            if 0<=i<n and 0<=j<m and dp[i][j]==-1:
                dp[i][j]=val
                
if j>0 and grid[i][j-1]==1:
                    fill(i,j-1,val)
                if j<m-1 and grid[i][j+1]==2:
                    fill(i,j+1,val)
                if i>0 and grid[i-1][j]==3:
                    fill(i-1,j,val)
                if i<n-1 and grid[i+1][j]==4:
                    fill(i+1,j,val)
        def findBorder(i,j,res):
            if 0<=i<n and 0<=j<m:
                if dp[i][j]<0:
                    if dp[i][j]==-1:
                        res.add((i,j))
                    return
                dp[i][j]=-dp[i][j]
                findBorder(i+1,j,res)
                findBorder(i-1,j,res)
                findBorder(i,j+1,res)
                findBorder(i,j-1,res)
        que={(n-1,m-1)}
        v=2
        while que:
            tmp=set()
            for i,j in que:
                fill(i,j,v)
            for i,j in que:
                findBorder(i,j,tmp)
            que=tmp
            v+=1
        return (-dp[0][0])-2
