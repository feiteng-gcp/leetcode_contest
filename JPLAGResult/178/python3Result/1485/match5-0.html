class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        heap = [(0, 0, 0)]
        n = len(grid)
        m = len(grid[0])
        def check(i, j):
            return 0 <= i < n and 0 <= j < m
        
        d = [(0, 1), (0, - 1), (1, 0), (-1, 0)]
        mincost = [[sys.maxsize] * m for _ in range(n)]
        
        while heap:
            curcost, i, j = heapq.heappop(heap)
            if curcost >= mincost[i][j]: continue
            mincost[i][j] = min(mincost[i][j], curcost)
            if i == n - 1 and j == m - 1: return curcost
            val = grid[i][j]
            for k in range(4):
                nxti, nxtj = i + d[k][0], j + d[k][1]
                if check(nxti, nxtj):
                    if val == k + 1:
                        heapq.heappush(heap, (curcost, nxti, nxtj))
                    else:
                        heapq.heappush(heap, (curcost + 1, nxti, nxtj))
