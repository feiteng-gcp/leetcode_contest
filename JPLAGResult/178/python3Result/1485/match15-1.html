class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
              
        m = len(grid)
        n = len
(grid[0])
        if m == n == 1: return 0
        
        memo = [[float('inf') for j in range(n)] for i in range(m)]
        
        heap = [(0, m-1, n-1)]
        heapq.heapify(heap)
        
        while heap:
            step, i, j = heapq.heappop(heap)
            # print(step, i, j)
            if i == 0 and j == 0: 
                return step
            x1 = i - 1
            y1 = j
            if x1 >= 0:
                
if grid[x1][y1] == 3:
                    if memo[x1][y1] > step:
                        memo[x1][y1] = step
                        heapq.heappush(heap, (step, x1, y1))
                else:
                    if memo[x1][y1] > step + 1:
                        memo[x1][y1] = step + 1
                        heapq.heappush(heap, (step+1, x1, y1))
            x2 = i
            y2 = j - 1
            if y2 >= 0:
                if grid[x2][y2] == 1:
                    if memo[x2][y2] > step:
                        memo[x2][y2] = step
                        heapq.heappush(heap, (step, x2, y2))
                else:
                    if memo[x2][y2] > step + 1:
                        memo[x2][y2] = step + 1
                        heapq.heappush(heap, (step+1, x2, y2))
            x3 = i + 1
            y3 = j
            if x3 < m:
                if grid[x3][y3] == 4:
                    if memo[x3][y3] > step:
                        memo[x3][y3] = step
                        heapq.heappush(heap, (step, x3, y3))
                else:
                    if memo[x3][y3] > step + 1:
                        memo[x3][y3] = step + 1
                        heapq.heappush(heap, (step+1, x3, y3))
            x4 = i
            y4 = j + 1
            if y4 < n:
                if grid[x4][y4] == 2:
                    if memo[x4][y4] > step:
                        memo[x4][y4] = step
                        heapq.heappush(heap, (step, x4, y4))
                else:
                    if memo[x4][y4] > step + 1:
                        memo[x4][y4] = step + 1
                        heapq.heappush(heap, (step+1, x4, y4))
