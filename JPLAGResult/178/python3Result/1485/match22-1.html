"""
1 which means go to the cell to the right. (i.e go from grid[i][j] to grid[i][j + 1])
2 which means go to the cell to the left. (i.e go from grid[i][j] to grid[i][j - 1])
3 which means go to the lower cell. (i.e go from grid[i][j] to grid[i + 1][j])
4 which means go to the upper cell. (i.e go from grid[i][j] to grid[i - 1][j])
"""
class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        costs = [[float('inf')]*n for _ in range(m)]
        
        def nextIdx(x, y, dire):
            if dire == 1:
                y += 1
            elif dire == 2:
                y -= 1
            elif dire == 3:
                x += 1
            else:
                x -= 1
            return x, y
        
        costs[0][0] = 0
        queue = [[0, 0]]
        changed = set()
        while queue:
            x, y = queue.pop(0)
            nx, ny = nextIdx(x, y, grid[x][y])
                
            
if 0 <= nx < m and 0 <= ny < n and costs[nx][ny] > costs[x][y]:
                costs[nx][ny] = costs[x][y]
                queue.insert(0, [nx, ny])
                
            if (x, y) not in changed:
                for i in range(1, 5):
                    nx, ny = nextIdx(x, y, i)            
                    
if 0 <= nx < m and 0 <= ny < n and costs[nx][ny] > costs[x][y]+1:
                        costs[nx][ny] = costs[x][y] + 1
                        queue.append([nx, ny])
        return costs[-1][-1]
