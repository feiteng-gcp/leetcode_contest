class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        
        
        dirs = {1:(0, 1),
                2:(0,-1),
                3:(1, 0),
                4:(-1,0)}
        m, n = len(grid), len(grid[0])
        cost = [(0, 0, 0)]  # 到达(0,0) 的最小费用是 0
        heapq.heapify(cost)
        visited = {(0,0): 0}
        while cost:
            pcost, pi, pj = heapq.heappop(cost)
            if (pi,pj) == (m-1, n-1):
                return pcost
            d = grid[pi][pj]
            if d in dirs:
                ci, cj = pi + dirs[d][0], pj + dirs[d][1]
                if 0 <= ci < m and 0 <= cj < n:
                    if (ci,cj) not in visited or pcost < visited[ci,cj]:
                        heapq.heappush(cost, (pcost, ci, cj))
                        visited[ci,cj] = pcost
            for new_d in [1,2,3,4]:
                if new_d != d:
                    ci, cj = pi + dirs[new_d][0], pj + dirs[new_d][1]
                    if 0 <= ci < m and 0 <= cj < n:
                        if (ci,cj) not in visited or pcost+1 < visited[ci,cj]:
                            heapq.heappush(cost, (pcost+1, ci, cj))
                            visited[ci,cj] = pcost+1
        return m+n-2
