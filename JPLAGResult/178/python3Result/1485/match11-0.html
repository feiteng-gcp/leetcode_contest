class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        rows, cols = len(grid), len(grid[0])
        dp = [[float('inf')]*cols for _ in range(rows)]
        dp[0][0] = 0
        
        to_check = {(0, 0)}
        def update(i, j):
            for ni, nj, sign in [(i, j+1,1), (i, j-1,2), (i+1, j,3), (i-1, j, 4)]:
                if 0 <= ni < rows and 0 <= nj < cols:
                    new_val = dp[i][j] + (grid[i][j] != sign)
                    if new_val < dp[ni][nj]:
                        dp[ni][nj] = new_val
                        #update(ni, nj)
                        to_check.add((ni, nj))
        
        while to_check:
            i, j = to_check.pop()
            update(i, j)
        
        return dp[-1][-1]
