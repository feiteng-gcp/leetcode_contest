class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        def minDist(grid1, grid2):
            return abs(grid1[1]-grid2[1])+abs(grid1[0]-grid2[0])
        m, n = len(grid), len(grid[0])

        dp = [[0] * n for _ in range(m)]
        
        for i in range(m):
            for j in range(n):
                dp[i][j] = minDist((0,0), (i,j))
        
        def recurse(i, j, dp):
            
if j+1 < n and dp[i][j] + (0 if grid[i][j] == 1 else 1) < dp[i][j+1]:
                dp[i][j+1] = dp[i][j] + (0 if grid[i][j] == 1 else 1)
                recurse(i, j+1, dp)
            
if j > 0 and dp[i][j] + (0 if grid[i][j] == 2 else 1) < dp[i][j-1]:
                dp[i][j-1] = dp[i][j] + (0 if grid[i][j] == 2 else 1)
                recurse(i, j-1, dp)
            if i+1 < m and dp[i][j] + (0 if grid[i][j] == 3 else 1) < dp[i+1][j]:
                dp[i+1][j] = dp[i][j] + (0 if grid[i][j] == 3 else 1)
                recurse(i+1, j, dp)
            if i > 0 and dp[i][j] + (0 if grid[i][j] == 4 else 1) < dp[i-1][j]:
                dp[i-1][j] = dp[i][j] + (0 if grid[i][j] == 4 else 1)
                recurse(i-1, j, dp)

        for i in range(m):
            for j in range(n):
                recurse(i, j, dp)
                
        return dp[m-1][n-1]
