class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        M = len(grid)-1
        N = len(grid[0])-1
        direct = [[],[0,1],[0,-1],[1,0],[-1,0]]
        queue = []
        heapq.heappush(queue, (0, 0, 0)) # cost, x, y
        visit = set()
        visit.add((0,0))
        while(queue):
            cost, x, y = heapq.heappop(queue)
            visit.add((x,y))
            print(cost, x, y)
            if x==M and y==N:
                return cost
            tmp = direct[grid[x][y]]
            nx, ny = x+tmp[0], y+tmp[1]
            #print('new',nx,ny, nx >= 0 and nx <=M and ny>=0 and ny<=N and (nx,ny) not in visit)
            if nx >= 0 and nx <=M and ny>=0 and ny<=N and (nx,ny) not in visit:
                visit.add((nx,ny))
                heapq.heappush(queue, (cost, nx, ny))
            for _direct in direct[1:]:
                nx,ny = x+_direct[0],y+_direct[1]
                if nx >= 0 and nx <=M and ny>=0 and ny<=N and (nx,ny) not in visit:
                    #visit.add((nx,ny))
                    heapq.heappush(queue, (cost+1, nx, ny))
        return -1
