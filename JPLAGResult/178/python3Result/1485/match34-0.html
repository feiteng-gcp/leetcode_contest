from collections import deque
class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        direction = [(0,1),(0,-1),(1,0),(-1,0)]
        cost = [[-1] * n for _ in range(m)]
        vis = [[False] * n for _ in range(m)]
        cur = 0
        cur_stk = deque()
        cur_stk.append((0,0))
        cost[0][0] = 0
        vis[0][0] = True
        while cur_stk:
            p = cur_stk.popleft()
            cur_x, cur_y = p[0],p[1]
            circur_visit = []
            while True:
                d = direction[grid[cur_x][cur_y] - 1]
                next_x = cur_x + d[0]
                next_y = cur_y + d[1]
                if 0 <= next_x < m and 0 <= next_y < n and (next_x,next_y) not in circur_visit:
                    if cost[next_x][next_y] == -1 or cost[next_x][next_y] > cost[cur_x][cur_y]:
                        cost[next_x][next_y] = cost[cur_x][cur_y]
                    if not vis[next_x][next_y]:
                        vis[next_x][next_y] = True
                    circur_visit.append((next_x,next_y))
                    cur_x, cur_y = next_x, next_y
                else:
                    break
            
            cur_x, cur_y = p[0],p[1]
            circur_visit = []
            while True:     
                for ds in direction:
                    next_x = cur_x + ds[0]
                    next_y = cur_y + ds[1]
                    if 0 <= next_x < m and 0 <= next_y < n:
                        if cost[next_x][next_y] == -1 or cost[next_x][next_y] > cost[cur_x][cur_y] + 1:
                            cost[next_x][next_y] = cost[cur_x][cur_y] + 1
                        if not vis[next_x][next_y]:
                            vis[next_x][next_y] = True
                            cur_stk.append((next_x,next_y))
                
                d = direction[grid[cur_x][cur_y] - 1]
                next_x = cur_x + d[0]
                next_y = cur_y + d[1]
                if 0 <= next_x < m and 0 <= next_y < n and (next_x,next_y) not in circur_visit:
                    circur_visit.append((next_x,next_y))
                    cur_x, cur_y = next_x, next_y
                else:
                    break

        
        return cost[m-1][n-1]
