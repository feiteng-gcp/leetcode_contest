class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        if m == n == 1: return 0
        memo = [[float('inf') for j in range(n)] for i in range(m)]
        heap = [(0, m-1, n-1)]
        heapq.heapify(heap)
        
        def helper(num, x, y):
            if grid[x][y] == num:
                if memo[x][y] > step:
                    memo[x][y] = step
                    heapq.heappush(heap, (step, x, y))
            else:
                if memo[x][y] > step + 1:
                    memo[x][y] = step + 1
                    heapq.heappush(heap, (step+1, x, y))
                    
        while heap:
            step, i, j = heapq.heappop(heap)
            if i == 0 and j == 0: 
                return step
            x1, y1 = i - 1, j
            if x1 >= 0: helper(3, x1, y1)
            x2, y2 = i, j - 1
            if y2 >= 0: helper(1, x2, y2)
            x3, y3 = i + 1, j
            if x3 < m: helper(4, x3, y3)
            x4, y4 = i, j + 1
            if y4 < n: helper(2, x4, y4)
