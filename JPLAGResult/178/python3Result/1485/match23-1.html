class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        visited = [[float('inf')] * n for _ in range(m)]
        queue = [(0, m-1, n-1)]
        while queue:
            cost, i, j = heapq.heappop(queue)
            if cost >= visited[i][j]:
                continue
            else:
                visited[i][j] = cost
            if i > 0:
                queue.append((cost + (0 if grid[i-1][j] == 3 else 1), i-1, j))
            if j > 0:
                queue.append((cost + (0 if grid[i][j-1] == 1 else 1), i, j-1))
            if i < m-1:
                queue.append((cost + (0 if grid[i+1][j] == 4 else 1), i+1, j))
            if j < n-1:
                queue.append((cost + (0 if grid[i][j+1] == 2 else 1), i, j+1))
        return visited[0][0]
