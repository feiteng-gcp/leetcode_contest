class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        temp = [[201]*n for _ in range
(m)]
        temp[-1][-1] = 0
        di = [(0,1), (0,-1), (1,0), (-1,0)]
        h = [(0, m- 1, n -1 )]
        
        while h:
            d, i, j= heapq.heappop(h)
            if d > temp[i][j]:
                continue
            #print(i, j, d)
            if  (i, j) == (0, 0):
                return d
            for a, b in di:
                if 0 <= i + a < m and 0 <= j + b < n:
                    if di[grid[i+a][j+b] - 1] == (-a, -b) and d < temp[i+a][j+b]:
                        #print('case', i + a, j + b, (-a, -b))
                        heapq.heappush(h, (d, i +a ,j +b))
                        temp[i+a][j+b] = d
                    elif d + 1 < temp[i+a][j+b]:
                        heapq.heappush(h, (d  +1, i + a, j + b))
                        temp[i+a][j+b] = d + 1

        for row in temp:
            print(row)
