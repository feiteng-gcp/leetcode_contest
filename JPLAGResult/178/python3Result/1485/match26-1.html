from collections import deque

class Solution:
    def minCost(self, grid):
        m = len(grid)
        n = len(grid[0])
        dp = []
        for temp in range(m):
            dp.append([float('inf'),] * n)
        dp[0][0] = 0

        queue = deque()
        queue.append((0,0))

        while queue:
            i, j = queue.popleft()

            for dir, dx, dy in [(1,0,1),(2,0,-1),(3,1,0),(4,-1,0)]:
                s = i+dx
                t 
= j+dy
                if s >= 0 and s < m and \
                   t >= 0 and t < n:
                    cost = 0 if grid[i][j] == dir else 1
                    #print('cost', i, j, '->', s, t, cost)
                    if dp[s][t] > dp[i][j] + cost:
                        dp[s][t] = dp[i][j] + cost
                        queue.append((s,t))
        #print('\n'.join([''.join(map(lambda x: '%2d' % x, row)) for row in dp]))
        return dp[m-1][n-1]
