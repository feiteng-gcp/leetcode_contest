from collections import deque
class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        
        d = {1: (0, 1),
            2: (0, -1),
            3: (1, 0),
            4: (-1, 0)}
        
        m, n = len(grid), len(grid[0])
        
        i, j = 0, 0
        q = deque()
        seen = set()
        while (i, j) not in seen and 0 <= i < m and 0 <= j < n:
            if i == m-1 and j == n-1:
                return 0
            
            seen.add((i, j))
            q.append((0, i, j))
            x, y = d[grid[i][j]]
            i += x
            j += y
            
        # print(q)
        
        while q:
            step, i, j = q.popleft()
            for direct in range(1, 5):
                if direct != grid[i][j]:
                    
                    x, y = d[direct]
                    ni, nj = i + x, j + y
                    while (ni, nj) not in seen and 0 <= ni < m and 0 <= nj < n:
                        if ni == m-1 and nj == n-1:
                            return step + 1
                        
                        seen.add((ni, nj))
                        q.append((step + 1, ni, nj))
                        x, y = d[grid[ni][nj]]
                        ni += x
                        nj += y
            # print(q)
