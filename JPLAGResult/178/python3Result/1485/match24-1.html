class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        W, H = len(grid[0]), len(grid)
        
        seen = [[float('inf')] * W for _ in range(H)]
        
        prio = [(0, 0, 0)]
        while prio:
            cost, x, y = heapq.heappop(prio)
            
            if x == H - 1 and y == W - 1:
                return cost
            
            if seen[x][y] < cost:
                continue
            seen[x][y] = cost
            
            for direction, (dx, dy) in enumerate([(0, 1), (0, -1), (1, 0), (-1, 0)]):
                direction += 1
                nx, ny 
= x + dx, y + dy
                new_cost = cost + (direction != grid[x][y])
                if 0 <= nx < H and 0 <= ny < W and seen[nx][ny] > new_cost:
                    seen[nx][ny] = new_cost
                    heapq.heappush(prio, (new_cost, nx, ny))
        return -1
