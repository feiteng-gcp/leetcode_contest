class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
                
        m, n = len(grid), len(grid[0])
                
        arr = [ [float('inf')] * n for _ in range(m) ]
        
        arr[0][0] = 0
        
        q = [(0, 0)]
        
        const_d = [ (), (0, 1), (0, -1), (1, 0), (-1, 0) ]
        
        for u, v in q:
            
            d = grid[u][v]
                                
            for i in range(1, 5):
                
                if i == d:
                    change = 0
                else:
                    change = 1
                
                x = u + const_d[i][0]
                y = v + const_d[i][1]

                cost = arr[u][v] + change

                if 0 <= x < m and 0 <= y < n:

                    
if cost < arr[x][y]:
                        arr[x][y] = cost

                        q.append((x, y))
                        
        return arr[m - 1][n - 1]
