class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        dp = [[float('inf')] * n for i in range(m)] # dp[i][j]: min cost to reach (m - 1, n - 1) from (i, j)
        dp[m-1][n-1] = 0
        dirs = [(0, -1), (0, 1), (-1, 0), (1, 0)]
        
        queue = [(m - 1, n - 1)]
        while queue:
            i, j = queue.pop(0)
            for num, (ki, kj) in enumerate(dirs):
                ni, nj = i + ki, j + kj
                if 0 <= ni < m and 0 <= nj < n:
                    newcost = dp[i][j] + (0 if num + 1 == grid[ni][nj] else 1)
                    if newcost < dp[ni][nj]:
                        dp[ni][nj] = newcost
                        queue.append((ni, nj))
        return dp[0][0]
