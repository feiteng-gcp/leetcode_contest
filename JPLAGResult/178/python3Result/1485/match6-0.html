from collections import deque
class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        dir_map = {1: (0, 1), 2: (0, -1), 3: (1, 0), 4: (-1, 0)}
        m, n = len(grid), len(grid[0])
        
        i, j = 0, 0
        queue, visited = deque(), set()
        while (i, j) not in visited and 0 <= i < m and 0 <= j < n:
            if i == m-1 and j == n-1:
                return 0
            queue.append((0, i, j))
            visited.add((i, j))
            x, y = dir_map[grid[i][j]]
            i, j = i + x, j+ y
            
        while queue:
            cnt, i, j = queue.popleft()
            for d in range(1, 5):
                if d != grid[i][j]:
                    
                    x, y = dir_map[d]
                    ni = i + x
                    nj = j + y
                    while 0 <= ni < m and 0 <= nj < n and (ni, nj) not in visited:
                        if ni == m-1 and nj == n-1:
                            return cnt + 1
                        queue.append((cnt + 1, ni, nj))
                        visited.add((ni, nj))
                        x, y = dir_map[grid[ni][nj]]
                        ni, nj = ni + x, nj + y
