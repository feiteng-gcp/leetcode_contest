class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        DIRECTIONS = [(0,-1), (0,1), (-1,0), (1,0)]
        m, n = len(grid), len(grid[0])
        
        hqueue = [(0, (m-1,n-1))]
        finalized = set()
        
        while hqueue:
            currcost, node = heapq.heappop(hqueue)
            if node == (0,0):
                return currcost
            if node in finalized:
                continue
            finalized.add(node)
            
            for i, (dx,dy) in enumerate(DIRECTIONS):
                x = node[0] + dx
                y = node[1] + dy
                if not self.within(m, n, x, y):
                    continue
                if (x, y) in finalized:
                    continue
                if i + 1 == grid[x][y]:
                    heapq.heappush(hqueue, (currcost, (x,y)))
                else:
                    heapq.heappush(hqueue, (currcost+1, (x,y)))
                    
    def within(self, m, n, x, y):
        return 0<=x<m and 0<=y<n
