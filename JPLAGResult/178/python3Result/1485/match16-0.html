class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        R, C = len(grid), len(grid[0])
        dp = [[float("Inf") for _ in range(C)] for _ in range(R)]
        dp[R-1][C-1] = 0
        queue = collections.deque([(R-1, C-1)])
        while queue:
            i, j = queue.popleft()
            # upper
            if i - 1 >= 0:
                cost = min(dp[i-1][j], dp[i][j] + (grid[i-1][j] != 3))
                if cost < dp[i-1][j]:
                    dp[i-1][j] = cost
                    queue.append((i-1, j))
            # left
            if j - 1 >= 0:
                cost = min(dp[i][j-1], dp[i][j] + (grid[i][j-1] != 1))
                if cost < dp[i][j-1]:
                    dp[i][j-1] = cost
                    queue.append((i, j-1))
            # right
            if j + 1 < C:
                cost = min(dp[i][j+1], dp[i][j] + (grid[i][j+1] != 2))
                if cost < dp[i][j+1]:
                    dp[i][j+1] = cost
                    queue.append((i, j+1))
            # lower
            if i + 1 < R:
                cost = min(dp[i+1][j], dp[i][j] + (grid[i+1][j] != 4))
                if cost < dp[i+1][j]:
                    dp[i+1][j] = cost
                    queue.append((i+1, j))
        return dp[0][0]
