from copy import deepcopy
from queue import PriorityQueue
class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        self.m, self.n = len(grid), len(grid[0])
        targetX, targetY = self.m - 1, self.n - 1
        self.grid = grid
        minCost = deepcopy(grid)
        for i in range(self.m):
            for j in range(self.n):
                minCost[i][j] = 0x3f3f3f3f
        pq = PriorityQueue()
        # cost, locationSortKey, location
        pq.put((0, 0, (0, 0)))
        while not pq.empty():
            cost, _, (x, y) = pq.get()
            if minCost[x][y] <= cost:
                continue
            else:
                minCost[x][y] = cost
            l = [
                self.getMoveQueueItem(x, y, cost, 1),
                self.getMoveQueueItem(x, y, cost, 2),
                self.getMoveQueueItem(x, y, cost, 3),
                self.getMoveQueueItem(x, y, cost, 4)
            ]
            for x in l:
                if x is not None:
                    pq.put(x)
        return minCost[targetX][targetY]
    def getMoveQueueItem(self, x, y, cost, direction):
        nx, ny, nc = x, y, cost
        if direction == 1:
            ny += 1
        elif direction == 2:
            ny -= 1
        elif direction == 3:
            nx += 1
        elif direction == 4:
            nx -= 1
        if direction != self.grid[x][y]:
            nc += 1
        if nx < 0 or ny < 0 or nx >= self.m or ny >= self.n:
            return None
        return (nc, -(nx + ny), (nx, ny))
