class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        dirs = [(0,1), (0,-1),(1,0),(-1,0)]
        visited = set()
        ranks = {0:[(0,0)]}
        rank = 0
        while rank in ranks:
            for i,j in ranks[rank]:
                if (i,j) not in visited:
                    visited.add((i,j))
                    if i == m-1 and j == n-1:
                        return rank
                    for val, d in enumerate(dirs):
                        x = i + d[0]
                        y = j + d[1]
                        if 0 <= x<m and 0<=j<n:
                            next_rank = rank + (0 if val+1 == grid[i][j] else 1)
                            if next_rank not in ranks:
                                ranks[next_rank] = []
                            ranks[next_rank].append((x,y))
            rank += 1
        return 0
