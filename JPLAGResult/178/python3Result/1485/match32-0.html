class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        que = collections.deque()
        que.append((0, 0))
        dp = [[math.inf] * n for _ in range(m)]
        dp[0][0] = 0
        while que:
            row, col = que.popleft()
            if col + 1 < n:
                if grid[row][col] == 1 and dp[row][col] < dp[row][col + 1]:
                    dp[row][col + 1] = dp[row][col]
                    que.append((row, col + 1))
                if grid[row][col] != 1 and dp[row][col] + 1 < dp[row][col + 1]:
                    dp[row][col + 1] = dp[row][col] + 1
                    que.append((row, col + 1))
            if col - 1 >= 0:
                if grid[row][col] == 2 and dp[row][col] < dp[row][col - 1]:
                    dp[row][col - 1] = dp[row][col]
                    que.append((row, col - 1))
                if grid[row][col] != 2 and dp[row][col] + 1 < dp[row][col - 1]:
                    dp[row][col - 1] = dp[row][col] + 1
                    que.append((row, col - 1))
            if row + 1 < m:
                if grid[row][col] == 3 and dp[row][col] < dp[row + 1][col]:
                    dp[row + 1][col] = dp[row][col]
                    que.append((row + 1, col))
                if grid[row][col] != 3 and dp[row][col] + 1 < dp[row + 1][col]:
                    dp[row + 1][col] = dp[row][col] + 1
                    que.append((row + 1, col))
            if row - 1 >= 0:
                if grid[row][col] == 4 and dp[row][col] < dp[row - 1][col]:
                    dp[row - 1][col] = dp[row][col]
                    que.append((row - 1, col))
                if grid[row][col] != 4 and dp[row][col] + 1 < dp[row - 1][col]:
                    dp[row - 1][col] = dp[row][col] + 1
                    que.append((row - 1, col))
        return dp[m - 1][n - 1]
