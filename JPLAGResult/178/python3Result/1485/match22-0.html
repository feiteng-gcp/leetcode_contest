class Solution:
    def minCost(self, grid: List[List[int]]) -> int:
        
        m = len(grid)
        n = len(grid[0])
        
        moves = {
            1: (0, 1),
            2: (0, -1),
            3: (1, 0),
            4: (-1, 0)
        }
        
        dist = collections.defaultdict(lambda: float("inf"))
        dist[(0, 0)] = 0
        target = (m - 1, n - 1)
        
        cur_step = [(0, 0)]  # all within this cur distance
        while cur_step:
            next_step = set()
            while cur_step:
                cx, cy = cur_step.pop(0)  # current x, current y

                move = grid[cx][cy]
                dx, dy = moves[move]
                nx, ny = cx + dx, cy + dy

                if 0 <= nx < m and 0 <= ny < n:
                    if dist[(nx, ny)] > dist[(cx, cy)]:
                        dist[(nx, ny)] = dist[(cx, cy)]  # same distance as current
                        cur_step.append((nx, ny))
                
                # save reachable neighbors with unset distances
                for dx, dy in moves.values():
                    nx, ny = cx + dx, cy + dy
                    
                    if 0 <= nx < m and 0 <= ny < n:
                        if dist[(nx, ny)] > dist[(cx, cy)] + 1:
                            dist[(nx, ny)] = dist[(cx, cy)] + 1  # distance one greater than
                            next_step.add((nx, ny))
            
            if dist[target] < float("inf"):
                # print(dict(dist))
                return dist[target]
            
            cur_step = list(next_step)
        
        # print(dict(dist))
        return dist[target]
