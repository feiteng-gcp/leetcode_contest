class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        s = set(votes[0])
        n = len(votes)
        m = len(votes[0])
        score = collections.defaultdict(int)
        for vote in votes:
            for i, ch in enumerate(vote):
                score[ch] += (n + 1) ** (m - i)
                
        return ''.join(sorted(s, key = lambda v:(-score[v], v)))
