class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        m = len(votes[0])
        count = collections.defaultdict(lambda: [0]*m)
        #cand = set(votes[0])
        for s in votes:
            for i, c in enumerate(s):
                count[c][i] -= 1
        a = sorted(count.keys(), key = lambda x: (count[x], x))

        return ''.join(a)
