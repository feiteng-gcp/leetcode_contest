from copy import deepcopy
class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        if len(votes) == 1:
            return votes[0]
        contestors = len(votes[0])
        if contestors == 1:
            return votes[0]
        d = {}
        for x in list('ABCDEFGHIJKLMNOPQRSTUVWXYZ'):
            d[x] = deepcopy([0] * contestors)
        for vote in votes:
            for i, v in enumerate(list(vote)):
                d[v][i] += 1
        l = list(sorted(sorted(d.items(), key=lambda x: -ord(x[0])), key=lambda x: x[1]))[::-1]
        print(l)
        l = list(map(lambda x: x[0], filter(lambda x: sum(x[1]) != 0, l)))
        return ''.join(l)
