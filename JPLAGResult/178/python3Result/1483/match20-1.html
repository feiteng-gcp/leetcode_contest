class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        m = len(votes[0])
        cnt = collections.defaultdict(lambda: [0] * m)
        for v in votes:
            for i, c in enumerate(v):
                cnt[c][i] += 1
        result = sorted(cnt, key=lambda x: tuple(-a for a in cnt[x]) + (x,))
        return "".join(result)
