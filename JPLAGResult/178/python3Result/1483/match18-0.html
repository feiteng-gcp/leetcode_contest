class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        ranks = collections.defaultdict(lambda: [0] * 26)
        for vote in votes:
            for i, letter in enumerate(vote):
                ranks[letter][i] -= 1
        return ''.join(sorted(ranks.keys(), key = lambda x: [ranks[x], x]))
