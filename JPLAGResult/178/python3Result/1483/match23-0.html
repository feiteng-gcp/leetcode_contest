class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        
        l = len(votes[0])
        d = {c: [0] * l for c in votes[0]}
        
        for vote in votes:
            for i, c in enumerate(vote):
                d[c][i] -= 1
                
        nd = collections.defaultdict(list)
        for c, l in d.items():
            nd[tuple(l)].append(c)
            
        res = ''
        for key in sorted(nd):
            l = nd[key]
            res += ''.join(sorted(l))
        return res
