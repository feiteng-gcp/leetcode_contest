import collections

class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        d = collections.defaultdict(lambda: [0] * len(votes[0]))
        for vote in votes:
            for i, letter in enumerate(vote):
                d[letter][i] -= 1
        order = [(count, letter) for letter, count in d.items()]
        order.sort()
        return ''.join([letter for count, letter in order])
