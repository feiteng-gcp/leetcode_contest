class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        p = {}
        for v in votes:
            for i, c in enumerate(v):
                if c not in p:
                    p[c] = [0] * 26
                p[c][i] += 1
        pp = []
        for k, v in p.items():
            pp.append(v + [-ord(k), k])
        pp.sort(reverse=True)
        return ''.join([p[-1] for p in pp])
