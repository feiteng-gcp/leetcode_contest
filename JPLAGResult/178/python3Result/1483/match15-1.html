class Solution:
    def rankTeams(self, votes):
        count = {}

        l = len(votes[0])
        
for vote in votes:
            for rank, p in enumerate(vote):
                if p not in count:
                    count[p] = [0 ] *l
                count[p][rank] -= 1

        res = sorted(count, key=lambda key: (count[key], key))
        return ''.join(res)
