class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        n = len(votes[0])
        teams = {x:[0]*n for x in votes[0]}
        for v in votes:
            for i, t in enumerate(v):
                teams[t][i] -= 1
                
        res = sorted([(tuple(v),k) for k, v in teams.items()])
        return ''.join(v for _,v in res)
