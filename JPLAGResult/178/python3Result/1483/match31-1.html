class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        
        
        
        cnt = len(votes[0])
        freq = {ch:[0]*cnt for ch in votes[0]}  # freq[team] = [3,1,8,4,..] 表示team或者3次第一，1次第二，8次第三，4次第四，。。。
        for vote in votes:
            for i,team in enumerate(vote):
                freq[team][i] += 1
                
        freqs = collections.defaultdict(list)
        for team,f in freq.items():
            freqs[tuple(f)].append(team)
        for tup in freqs:
            freqs[tup].sort()
        res = list(freqs.keys())
        res.sort(reverse = True)
        result = ''
        for tup in res:
            result += ''.join(freqs[tup])
        return result
