class Solution:
  def rankTeams(self, votes: List[str]) -> str:
    if not votes:
      return ''
    n 
= len(votes[0])
    c = {}
    for x in votes[0]:
      c[x] = [0] * n
    for v in votes:
      for i, x in enumerate(v):
        c[x][i] -= 1
    d = sorted([(v, k) for k, v in c.items()])
    ret = ''
    for a in d:
      ret += a[1]
    return ret
