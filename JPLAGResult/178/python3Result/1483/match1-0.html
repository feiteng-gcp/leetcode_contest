class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        ranks = {}
        teams = [c for c in votes[0]]
        for string in votes:
            for i, v in enumerate(string):
                if v not in ranks:
                    ranks[v] = [1010] * len(string) + [v]
                ranks[v][i] -= 1
        teams.sort(key = lambda x:ranks[x])
        return "".join(teams)
