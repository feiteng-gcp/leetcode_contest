class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        if len(votes) == 0:
            return ""
        l = len(votes[0])
        rank = {ch: [0]*(l) + [ch] for ch in votes[0]}
        
for v in votes:
            for p, ch in enumerate(v):
                rank[ch][p] -= 1
        return "".join([it[-1] for it in sorted(rank.values())])
