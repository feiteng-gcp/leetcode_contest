class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        n_teams = len(votes[0])
        team_votes = {}
        for t in votes[0]:
            team_votes[t] = [0] * n_teams
        for vote in votes:
            for i, t in enumerate(vote):
                team_votes[t][i] += 1
        for t in team_votes:
            team_votes[t].append(-ord(t))
        res = [t for t in votes[0]]
        res.sort(key=lambda t : team_votes[t], reverse=True)
        return ''.join(res)
