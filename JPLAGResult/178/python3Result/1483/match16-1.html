class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        L = len(votes[0])
        count = {}
        for s in votes:
            for i, c in enumerate(s):
                if c not in count:
                    count[c] = [0] * L
                count[c][i] -= 1
        ans = []
        for c, val in count.items():
            ans.append(tuple(val+[c]))
        ans.sort()
        return ''.join([x[-1] for x in ans])
