class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        result = collections.defaultdict(lambda:[0]*len(votes[0]))
        for s in votes:
            for i,c in enumerate(s):
                result[c][i]+=1
        return ''.join(sorted(result, key = lambda x: result[x]+[-ord(x)], reverse = True))
