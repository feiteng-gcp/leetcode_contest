class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        if len(votes) == 1: return votes[0]
        
        N 
= len(votes[0])
        
        count = {}
        
        for vote in votes:
            for i, char in enumerate(vote):
                if char not in count:
                    count[char] = [0] * N
                count[char][i] -= 1
        
        res = []
        for char, val in count.items():
            res.append(val + [char])
        res.sort()

        return ''.join([x[-1] for x in res])
