class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        m = len(votes[0])
        cnt = {}
        for vote in votes:
            for i, c in enumerate(vote):
                if c not in cnt:
                    cnt[c] = [0] * m
                cnt[c][i] -= 1
        res = sorted(cnt, key = lambda x: (cnt[x], x)) 
        return ''.join(res)
