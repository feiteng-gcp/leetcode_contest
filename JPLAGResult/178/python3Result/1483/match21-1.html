class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        N = len(votes[0])
        ans = collections.defaultdict(lambda :[0]*N)
        for vote in votes:
            for no, name in enumerate(vote):
                ans[name][no]-=1
        vote_name = [(tuple(j), i) for i,j in ans.items()]
        vote_name.sort(reverse=False)
        ans = [i[-1] for i in vote_name]
        return ''.join(ans)
