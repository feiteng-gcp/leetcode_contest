class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        n = len(votes[0])
        info = {}
        for c in votes[0]:
            info[c] = [0] * n
        for vote in votes:
            for i, c in enumerate(vote):
                info[c][i] -= 1
        words = list(info.keys())
        words.sort(key = lambda x : info[x] + [x])
        return ''.join(words)
