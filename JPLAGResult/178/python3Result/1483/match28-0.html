class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        counter = collections.defaultdict(lambda: [0] * 26)
        for vote in votes:
            for i, c in enumerate(vote):
                counter[c][i] -= 1
        
        ranking = sorted([val + [key] for key, val in counter.items()])
        return ''.join(r[-1] for r in ranking)
