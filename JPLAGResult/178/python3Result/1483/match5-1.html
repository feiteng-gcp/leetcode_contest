from functools import cmp_to_key

class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        def cmp(a, b):
            for i in range(len(a) - 1):
                if a[i] > b[i]:
                    return -1
                if b[i] > a[i]:
                    return 1
            return -1 if a[-1] < b[-1] else 1
        numPos = len(votes[0])
        comp = {}
        for vote in votes:
            for idx, c in enumerate(vote):
                if c not in comp:
                    comp[c] = [0 for i in range(numPos)]
                comp[c][idx] += 1
                
        x = []
        for key, val in comp.items():
            x.append(val + [key])
        x.sort(key=cmp_to_key(cmp))
        
        r = []
        for i in x:
            r.append(i[-1])
        return ''.join(r)
