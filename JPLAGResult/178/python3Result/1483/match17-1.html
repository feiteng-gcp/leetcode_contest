class Solution:
    def rankTeams(self, votes: List[str]) -> str:
        score = collections.defaultdict(lambda : [0] * len(votes[0]))
        for vote in votes:
            for i, can in enumerate(vote):
                score[can][i] -= 1
        return ''.join(sorted(score.keys(), key=lambda x:score[x]+[x]))
