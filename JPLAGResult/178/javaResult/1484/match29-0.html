/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isSubPath(ListNode head, TreeNode root) {
        if (head == null || root == null) {
            return false;
        }
        return rootPath(head, root)
            || nonRootPath(head, root);
    }
    
    boolean rootPath(ListNode head, TreeNode root) {
        if (head == null || root == null) {
            return false;
        }
        
        if (head.val != root.val) {
            return false;
        }
        if (head.next == null) {
            return true;
        } else {
            return rootPath(head.next, root.left)
                || rootPath(head.next, root.right);
        }
    }
    
    boolean nonRootPath(ListNode head, TreeNode root) {
        if (root == null) {
            return false;
        }
        return isSubPath(head, root.left)
            || isSubPath(head, root.right);
    }
}
