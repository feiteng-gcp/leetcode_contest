/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public boolean isSubPath(ListNode head, TreeNode root) {
        if(head == null) {
            return true;
        }
        if(root== null) {
            return false;
        }
        if(head.val == root.val) {
            if(checkDownword(head, root)) {
                return true;
            }
        
}
        return isSubPath(head, root.left) || isSubPath(head, root.right);
    }
    
    private boolean checkDownword(ListNode head, TreeNode root) {
        if(head == null) {
            return true;
        }
        if(root== null || root.val != head.val) {
            return false;
        }
        return checkDownword(head.next, root.left) || checkDownword(head.next, root.right);
    }
}
