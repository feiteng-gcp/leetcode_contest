/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    public boolean isSubPath(ListNode head, TreeNode root) {
        return dfs(head, root);
    }
    
    public boolean dfs(ListNode head, TreeNode node){
        return isMatch(head, node) || node!=null&&dfs(head, node.left) || node!=null&&dfs(head, node.right);
    }
    
    public boolean isMatch(ListNode lNode, TreeNode tNode){
        if(lNode==null){
            return true;
        }
        
        if(tNode==null || lNode.val!=tNode.val){
            return false;
        }
        
        return isMatch(lNode.next, tNode.left) || isMatch(lNode.next, tNode.right);
    }
}
