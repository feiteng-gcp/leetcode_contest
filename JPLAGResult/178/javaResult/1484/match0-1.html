/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    
    Map<TreeNode, Map<ListNode, Boolean>> map;
    public boolean isSubPath(ListNode head, TreeNode root) {
        map = new HashMap<>();
        return helper(head, root, head);
    }
    
    private boolean helper(ListNode head, TreeNode root, ListNode head2) {
        if (head == null) {
            return true;
        }
        
        if (root == null) {
            return false;
        }
        
        if (map.containsKey(root) && map.get(root).containsKey(head)) {
            return map.get(root).get(head);
        }
        
        map.putIfAbsent(root, new HashMap<>());
        
        if (head.val == root.val) {
            if (helper(head.next, root.left, head2) || helper(head.next, root.right, head2)) {
                map.get(root).put(head, true);
                return true;
            }
        }
        if (helper(head2, root.right, head2) || helper(head2, root.left, head2)) {
            map.get(root).put(head, true);
            return true;
        }
        map.get(root).put(head, false);
        return false;
    }
}
