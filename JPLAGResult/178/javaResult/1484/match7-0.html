/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    List<Integer> li =new ArrayList<>();
    public boolean isSubPath(ListNode head, TreeNode root) {
        for(ListNode p = head;p!=null;p=p.next){
            li.add(p.val);
        }
        return s(root);
    }
    
    class Pair{
        TreeNode node;
        int pos;
        Pair(TreeNode n,int p){
            pos =p;
            node =n;
        }
        
    }
    
    boolean s(TreeNode r){
        
        Queue<Pair> q = new ArrayDeque<>();
        if(r.val==li.get(0)){
            q.offer(new Pair(r,0));
        }
        while(q.size()>0){
            Pair cur = q.poll();
            TreeNode nn = cur.node;
            int pp = cur.pos+1;
            if(pp==li.size()) return true;
            if(nn.left!=null&&nn.left.val==li.get(pp)){
                q.offer(new Pair(nn.left,pp));
            }
            if(nn.right!=null&&nn.right.val==li.get(pp)){
                q.offer(new Pair(nn.right,pp));
            }
            
        }
        
        
        
        
        
        
        
        
        if(r.left!=null&&s(r.left)){
            return true;
        }
        if(r.right!=null&&s(r.right)){
            return true;
        }
        return false;
    }
    
}
