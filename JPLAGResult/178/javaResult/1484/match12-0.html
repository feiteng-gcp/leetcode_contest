/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    boolean res=false;
    ListNode gHead=null;
    public boolean isSubPath(ListNode head, TreeNode root) {
        if(head==null && root==null)
            return true;
        if(head==null || root==null)
            return false;
        gHead=head;
        dfs1(root);
        return res;
    }
    private void dfs1(TreeNode root) {
        if(root==null)
            return;
        dfs(root, gHead, 0, 0);
        dfs1(root.left);
        dfs1(root.right);
    }
    private void dfs(TreeNode root, ListNode head, int currLevel, int level) {
        if(head==null)
        {
            res=true;
            return;
        }
        if(root==null)
            return;
        if(head.val==root.val) {   
            if(currLevel==0 || level-currLevel==1) {
                currLevel=level;
                head=head.next;
            } 
        }
        dfs(root.left, head, currLevel, level+1);
        dfs(root.right, head, currLevel, level+1);
    }            
}
