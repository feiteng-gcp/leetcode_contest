/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */

class Solution {
    public boolean isSubPath(ListNode head, TreeNode root) {
        return helper(head, root, true);
    }
    
    private boolean helper(ListNode head, TreeNode root, boolean isHead) {
        if (root == null && head != null) {
            return false;
        }
        
        if (head == null) {
            return true;
        }
        
        if (head.val != root.val && !isHead) {
            return false;
        }
        
        boolean result = false; 
        if (head.val == root.val) {
            result |= completePath(head.next, root.left) || completePath(head.next, root.right);
        }
        
        return result || 
helper(head, root.left, isHead) || helper(head, root.right, isHead);
    }
    
    private boolean completePath(ListNode head, TreeNode root) {
        if (head == null) {
            return true;
        }
        
        if (root == null) {
            return false;
        }
        
        return head.val == root.val && (completePath(head.next, root.left) || completePath(head.next, root.right));
    }
}
