/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
class Solution {
    // HashSet<Node> mark = new HashSet<>();
    boolean ans = false;
    public boolean isSubPath(ListNode head, TreeNode root) {
        ans = false;
        solve2(root, head);
        
        return ans;
    }
    
    private void solve2(TreeNode node, ListNode root) {
        if (node==null) return;
        if (ans) return;
        
        if (solve(node, root)) {
            ans = true;
            return;
        }
        
        solve2(node.left, root);
        solve2(node.right, root);
    }
    
    private boolean solve(TreeNode node1, ListNode node2) {
        if (node2==null) return true;
        if (node1==null) return false;
        
        if (node1.val!=node2.val) return false;
        return solve(node1.left, node2.next) || solve(node1.right, node2.next);
        
    }
}
