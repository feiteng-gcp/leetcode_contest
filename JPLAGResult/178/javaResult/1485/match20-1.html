	class S
olution {
	    public int minCost(int[][] a) {
	        int n = a.length, m = a[0].length;
	        int[][] dp = new int[n][m];
	        for(int i = 0;i < n;i++){
	        	Arrays.fill(dp[i], 99999999);
	        }
	        dp[0][0] = 0;
	        Deque<int[]> q = new ArrayDeque<>();
	        q.add(new int[]{0, 0});
	        int[] dr = { 1, 0, -1, 0 };
			int[] dc = { 0, 1, 0, -1 };
			int[] to = {-1, 1, 3, 0, 2};
	        while(!q.isEmpty()){
	        	int[] cur = q.poll();
	        	int r = cur[0]
, c = cur[1];
	        	for(int k = 0;k < 4;k++){
	        		int nr = r + dr[k], nc = c + dc[k];
	        		if(nr >= 0 && nr < n && nc >= 0 && nc < m){
	        			if(to[a[r][c]] == k){
		        			if(dp[nr][nc] > dp[r][c]){
		        				dp[nr][nc] = dp[r][c];
		        				q.addFirst(new int[]{nr, nc});
		        			}
	        			}else{
		        			if(dp[nr][nc] > dp[r][c] + 1){

		        				dp[nr][nc] = dp[r][c] + 1;
		        				q.addLast(new int[]{nr, nc});
		        			}
	        			}
	        		}
	        	}
	        }
	        return (dp[n-1][m-1]);
	    }
	}
