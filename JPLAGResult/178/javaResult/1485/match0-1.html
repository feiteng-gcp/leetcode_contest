import java.awt.Point;

class Solution {
    
    int[] dr = new int[]{Integer.MIN_VALUE, 0, 0, 1, -1};
    int[] dc = new int[]{Integer.MIN_VALUE, 1, -1, 0, 0};
    
    public int minCost(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        boolean[][] seen = new boolean[m][n];
        Queue<Point> q = new ArrayDeque<>();
        traversal(grid, 0, 0, 0, seen, q);
        while (!q.isEmpty()) {
            Point p = q.poll();
            if (p.x / 100 == m - 1 && p.x % 100 == n - 1) return p.y;
            int r = p.x / 100, c = p.x % 100;
            for (int k = 1; k <= 4; k++) {
                int nr = r + dr[k], nc = c + dc[k];
                if (nr >= 0 && nc >= 0 && nr < m && nc < n && !seen[nr][nc]) {
                    traversal(grid, nr, nc, p.y + 1, seen, q);
                }
            }
        }
        return -1;
    }
    
    private void traversal(int[][] grid, int sr, int sc, int cost, boolean[][] seen, Queue<Point> q) {
        while (sr >= 0 && sc >= 0 
               && sr < seen.length && sc < seen[0].length
               && !seen[sr][sc]) {
            seen[sr][sc] = true;
            q.offer(new Point(sr * 100 + sc, cost));
            int direction = grid[sr][sc];
            sr += dr[direction];
            sc += dc[direction];
        }
    }
}
