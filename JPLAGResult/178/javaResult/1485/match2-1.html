class Solution {
    public int minCost(int[][] grid) {
        
int[][] cost = new int[grid.length][grid[0].length];
        for (int i = 0; i < cost.length; i++)
        {
            Arrays.fill(cost[i], grid.length * grid[0].length + 1);
        }
        
        int[][] dirs = 
new int[][] {
          null, {0, 1}, {0, -1}, {1, 0}, {-1, 0}  
        };
        
        cost[0][0] = 0;
        
        PriorityQueue<int[]> q = new PriorityQueue<int[]>((x, y) -> x[0] - y[0]);
        q.offer(new int[] {0, 0, 0});
        
        while(q.size() > 0)
        {
            int[] entry = q.poll();
            
            System.out.println(entry[0] + "===" + entry[1] + "=====" + entry[2]);
            if (entry[1] == grid.length - 1 && entry[2] == grid[0].length - 1)
            {
                return entry[0];
            
}
            
            for (int i = 1; i <= 4; i++)
            {
                int ni = entry[1] + dirs[i][0];
                int nj = entry[2] + dirs[i][1];
                
                if (ni < 0 || nj < 0 || ni >= grid.length || nj >= grid[0].length)
                {
                    continue;
                }
                
                int ncost = grid[entry[1]][entry[2]] == i ? entry[0] : entry[0] + 1;
                
if (ncost < cost[ni][nj])
                {
                    cost[ni][nj] = ncost;
                    q.offer(new int[] {ncost, ni, nj});
                }
                
            }
        }
        
        return -1;
    }
}
