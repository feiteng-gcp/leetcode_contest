class Solution {
    public int minCost(int[][] g) {
        int l = g.length;
        int ll = g[0].length;
        
        int dp[][] =new int[l][ll];
        
        for(int v[]:dp){
            Arrays.fill(v,100000000);
        }
        dp[0][0] = 0;
        Queue<int[]> q = new ArrayDeque<>();
        q.offer(new int[]{0,0});
        
        int dx[] = {0,0,1,-1};
        int dy[] = {1,-1,0,0};
        
        while(q.size()>0){
            int cur[] = q.poll();
            int x = cur[0]; int y = cur[1];
            int go = g[x][y]-1;
            
            for(int j=0;j<=3;++j){
                int nx = x+dx[j];
                int ny = y+dy[j];
                
                if(nx>=0&&nx<l&&ny>=0&&ny<ll){
                    
                    int add = (j==go)?0:1;
                    
if(dp[nx][ny]>dp[x][y]+add){
                        dp[nx][ny] = dp[x][y]+add;
                        
                    
                        
                        q.offer(new int[]{nx,ny});
                    }
                    
                }
                
            }
            
            
            
            
        }
        return dp[l-1][ll-1];
        
        
        
    }
}
