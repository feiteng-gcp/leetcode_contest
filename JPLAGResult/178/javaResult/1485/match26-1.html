class Solution {
    
private class State {
        int i, j, cost;
        public State(int i, int j, int cost) {
            this.i = i;
            this.j = j;
            this.cost = cost;
        }
    }
    
    
public int minCost(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] costs = new int[m][n];
        for (int[] arr : costs) Arrays.fill(arr, Integer.MAX_VALUE);
        PriorityQueue<State> pq = new PriorityQueue<State>(new Comparator<State>(){
            @Override
            public int compare(State s1, State s2) {
                return s1.cost - s2.cost;
            }
        });
        
        pq.offer(new State(0, 0, 0));
        costs[0][0] = 0;
        
int[][] dirs = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
        while (!pq.isEmpty()) {
            State cur = pq.poll();
            int i = cur.i, j = cur.j, c = cur.cost;
            // System.out.println(i + "," + j + "," + c);
            if (i == m - 1 && j == n - 1) return c;
            int[] dir = dirs[grid[i][j] - 1];
            // do not change dir
            int x = i + dir[0], y = j + dir[1];
            if (x >= 0 && x < m && y >= 0 && y < n && costs[x][y] > c) {
                costs[x][y] = c;
                pq.offer(new State(x, y, c));
            }
            
            // change dir
            for (int k = 0; k < dirs.length; k++) {
                if (k == grid[i][j] - 1) continue;
                int[] curDir = dirs[k];
                int xx = i + curDir[0], yy = j + curDir[1];
                if (xx >= 0 && xx < m && yy >= 0 && yy < n && costs[xx][yy] > c + 1) {
                    costs[xx][yy] = c + 1;
                    pq.offer(new State(xx, yy, c + 1));
                
}
            }
        }
        
        return -1;
    }
}
