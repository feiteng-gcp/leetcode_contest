class Solution {
    
    
int[][] ds = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};

    int[] add(int[] v1, int[] v2) {
        return new int[]{v1[0] + v2[0], v1[1] + v2[1], v1[2]};
    }
    
public int minCost(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[][] table = new int[m][n];
        for (int[] row : table) {
            Arrays.fill(row, m * n);
        }
        Deque<int[]> dq = new LinkedList<>();
        dq.offerLast(new int[]{0, 0, 0});
        table[0][0] = 0;
        
while (!dq.isEmpty()) {
            int[] cur = dq.pollFirst();
            for (int i = 0; i < 4; i++) {
                int[] next = add(cur, ds[i]);
                if (next[0] == -1 || next[0] == m || next[1] == -1 || next[1] == n) {
                    continue;
                }
                if (i + 1 != grid[cur[0]][cur[1]]) {
                    next[2]++;
                }
                if (next[2] < table[next[0]][next[1]]) {
                    dq.offerLast(next);                    
                    table[next[0]][next[1]] = next[2];
                
}
            }
        }
        return table[m - 1][n - 1];
    }
}
