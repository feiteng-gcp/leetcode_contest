class Solution {
    int R, C;
    
int[][] dirs = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public int minCost(int[][] grid) {
        R = grid.length;
        C = grid[0].length;
        int[][] minCost = new int[R][C];
        for (int[] m : minCost) {
            Arrays.fill(m, Integer.MAX_VALUE);
        }
        PriorityQueue<int[]> q = new PriorityQueue<int[]>((a, b) -> (a[2] - b[2]));
        minCost[0][0] = 0;
        
q.offer(new int[]{0, 0, 0});
        while (!q.isEmpty()) {
            int[] cur = q.poll();
            int r0 = cur[0], c0 = cur[1], cost = cur[2];
            if (minCost[r0][c0] < cost) {
                continue;
            }
            for (int i = 1; i <= 4; i++) {
                int[] d = dirs[i - 1];
                int r = r0 + d[0], c = c0 + d[1];
                if (r >= 0 && c >= 0 && r < R && c < C) {
                    int curCost = cost;
                    if (i != grid[r0][c0]) {
                        curCost += 1;
                    }
                    if (curCost < minCost[r][c]) {
                        q.offer(new int[]{r, c, curCost});
                        minCost[r][c] = curCost;
                    
}
                }
            }
        }
        return minCost[R - 1][C - 1];
    }
}
