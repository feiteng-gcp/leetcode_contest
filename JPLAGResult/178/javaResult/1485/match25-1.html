class Node {
    int i, j, cost;
    Node(int i, int j, int cost) {
        this.i=i;
        this.j=j;
        this.cost=cost;
    }
}


class Solution {
    
    int dir[][]={{1,0,1}, {2,0,-1}, {3,+1,0}, {4,-1,0}};
    
    public int minCost(int[][] grid) {
        
        int n=grid.length,
            m=grid[0].length;
        
        int dp[][]=new int[n][m];
        for(int i=0; i<n; i++)
            Arrays.fill(dp[i], Integer.MAX_VALUE);
        
        PriorityQueue<Node> q=new PriorityQueue<>((n1, n2)->{
            return n1.cost-n2.cost;
        });
        
q.add(new Node(0, 0, 0));
        
        while(!q.isEmpty()) {
            
            Node item=q.poll();
            
            if(dp[item.i][item.j]<=item.cost)
                continue;
            
            dp[item.i][item.j]=item.cost;
            
            for(int k=0; k<4; k++) {
                int tempi=item.i+dir[k][1];
                int tempj=item.j+dir[k][2];
                if(tempi>=0 && tempi<n && tempj>=0 && tempj<m)
                    q.add(new Node(tempi, tempj, item.cost+(dir[k][0]==grid[item.i][item.j]?0:1)))
;
            }
        }
        
        return dp[n-1][m-1];
    }
}
