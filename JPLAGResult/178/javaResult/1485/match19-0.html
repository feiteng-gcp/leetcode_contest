class Solution {
    int[][] dirs = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public int minCost(int[][] grid) {
        int M = grid.length;
        int N = grid[0].length;
        int[][] dp = new int[M][N];
        for(int i = 0; i < M; i++)
            Arrays.fill(dp[i], Integer.MAX_VALUE);
        dp[0][0] = 0;
        boolean[][][] visited = new boolean[M][N][4];
        Queue<int[]> que = new LinkedList<>();
        que.add(new int[]{0, 0, 0});
        while(!que.isEmpty()) {
            int[] arr = que.remove();
            int r = arr[0];
            int c = arr[1];
            int step = arr[2];
            for(int k = 0; k < 4; k++) {
                int rr = r + dirs[k][0];
                int cc = c + dirs[k][1];
                // if(r == 1 && c == 1)
                //     System.out.println(rr + "\t" + cc + "\t" + visited[rr][cc][k]);
                if(rr >= 0 && rr < M && cc >= 0 && cc < N && !visited[rr][cc][k]) {
                    // visited[rr][cc][k] = true;
                    int next = k == grid[r][c] - 1 ? step : step + 1;
                    if(next < dp[rr][cc]) {
                        dp[rr][cc] = next;
                        que.add(new int[]{rr, cc, next});
        //                 for(int i = 0; i < M; i++)  {
        //     System.out.println();
        //     for(int j = 0; j < N; j++)
        //         System.out.print(dp[i][j] + " \t");
        // }
                    }
                    
                }
            }
        }
        
        return dp[M - 1][N - 1];
    }
}
