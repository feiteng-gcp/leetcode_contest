import java.util.Arrays;


class Solution {
    public int minCost(int[][] grid) {

        int n = grid.length, m = grid[0].length;
        int[][] dp = new int[n][m];

        for (int[] row : dp) {
            Arrays.fill(row, m*n);
        }

        dp[0][0] = 0;

        /*for (int[] row : dp) {
            for (int num : row) {
                System.out.print(num);
                System.out.print(" ");
            }
            System.out.println();
        }
        System.out.println();*/


        for (int k = 0; k <= n * m; k++) {
            boolean updated = false;

            for (int i = 0; i < n; i++) {

                for (int j = 0; j < m; j++) {

                    int prev = dp[i][j];

                    if (i - 1 >= 0) {
                        if (pointsTo(grid, i - 1, j, i, j)) {
                            dp[i][j] = Math.min(dp[i][j], dp[i - 1][j]);
                        } else {
                            dp[i][j] = Math.min(dp[i][j], dp[i - 1][j] + 1);
                        }
                    }

                    if (i + 1 < n) {
                        if (pointsTo(grid, i + 1, j, i, j)) {
                            dp[i][j] = Math.min(dp[i][j], dp[i + 1][j]);
                        } else {
                            dp[i][j] = Math.min(dp[i][j], dp[i + 1][j] + 1);
                        }
                    }


                    if (j > 0) {
                        if (pointsTo(grid, i, j - 1, i, j)) {
                            dp[i][j] = Math.min(dp[i][j], dp[i][j - 1]);
                        } else {
                            dp[i][j] = Math.min(dp[i][j], dp[i][j - 1] + 1);
                        }
                    }


                    if (j + 1 < m) {
                        if (pointsTo(grid, i, j + 1, i, j)) {
                            dp[i][j] = Math.min(dp[i][j], dp[i][j + 1]);
                        } else {
                            dp[i][j] = Math.min(dp[i][j], dp[i][j + 1] + 1);
                        }
                    }


                    if (prev != dp[i][j]) {
                        updated = true;
                    }

                    /*for (int[] row : dp) {
                        for (int num : row) {
                            System.out.print(num);
                            System.out.print(" ");
                        }
                        System.out.println();
                    }
                    System.out.println();*/

                }
            }
            if (!updated) {
                break;
            }
        }

        return dp[n - 1][m - 1];


    }

    private boolean pointsTo(
int[][] grid, int i1, int j1, int i2, int j2) {

        int n = grid.length, m = grid[0].length;
        if (i1 < 0 || i1 >= n || i2 < 0 || i2 >= n) {
            return false;
        }
        if (j1 < 0 || j1 >= m || j2 < 0 || j2 >= m) {
            return false;
        }

        if (i2 - i1 == 1) {
            return grid[i1][j1] == 3;
        }
        if (i2 - i1 == -1) {
            return grid[i1][j1] == 4;
        }
        if (j2 - j1 == 1) {
            return grid[i1][j1] == 1;
        }
        if (j2 - j1 == -1) {
            return grid[i1][j1] == 2;
        }
        return false;
    }

    public static void main(String[] a) {
        Solution sol = new Solution();
        
int[][] grid = {{1, 1, 1, 1}, {2, 2, 2, 2}, {1, 1, 1, 1}, {2, 2, 2, 2}};
        System.out.println(sol.minCost(grid));
    }
}
