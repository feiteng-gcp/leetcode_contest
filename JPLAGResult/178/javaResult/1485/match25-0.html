class Solution {
    
    private static int[][] dirs = {{0,0}, {0 ,1}, {0, -1},{1, 0}, {-1, 0}};
    
    static class Node {
        int x;
        int y;
        int cost;
        
        public Node (int x, int y, int cost) {
            this.x = x;
            this.y = y;
            this.cost = cost;
        }
    }
    
    public int minCost(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        if (m == 1 && n == 1)
            return 0;
        
        int[][] dp = new int[m][n];
        for (int[] d : dp)
            Arrays.fill(d, Integer.MAX_VALUE);
        
        Queue<Node> queue = new LinkedList<>();
        queue.offer(new Node(0,0,0));
        
        while (!queue.isEmpty()) {
            Node curr = queue.poll();
            if (curr.cost >= dp[m-1][n-1])   continue;
            
            int orient = grid[curr.x][curr.y];
            for (int i = 1; i <= 4; i++) {
                int[] dir = dirs[i];
                int px = curr.x + dir[0], py = curr.y + dir[1];
                if (px >= 0 && px < m && py >= 0 && py < n) {
                    int cost_n = curr.cost + (i == orient ? 0 : 1);
                    if (dp[px][py] <= cost_n)   continue;
                    
                    dp[px][py] = cost_n;
                    if (px == m-1 && py == n-1) continue;
                    
                    queue.offer(new Node(px, py, cost_n));
                }
            }
        }
        
        return dp[m-1][n-1];
    }
}
