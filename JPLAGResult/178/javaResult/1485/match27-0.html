class Solution {
    public int minCost(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        boolean[][] visited = new boolean[n][m];
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> {
            return a[2] - b[2];
        });
        pq.add(new int[]{0, 0, 0});
        while (!pq.isEmpty()) {
            int[] curr = pq.remove();
            if (curr[0] < 0 || curr[0] > n-1 || curr[1] < 0 || curr[1] > m-1 || visited[curr[0]][curr[1]]) {
                continue;
            }
            visited[curr[0]][curr[1]] = true;
            if (curr[0] == n-1 && curr[1] == m-1) {
                return curr[2];
            }
            int[] dr = {0, 0, 1, -1};
            int[] dc = {1, -1, 0, 0};
            for (int d = 0; d < 4; d++) {
                if (grid[curr[0]][curr[1]] == d+1) {
                    pq.add(new int[]{curr[0]+dr[d], curr[1]+dc[d], curr[2]});
                } else {
                    pq.add(new int[]{curr[0]+dr[d], curr[1]+dc[d], curr[2]+1});
                }
            }
        }
        return 1000;
    }
}
