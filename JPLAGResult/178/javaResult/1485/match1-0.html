class Solution {
    
    private int[][] dirs = {{}, {0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    
    public int minCost(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        int[][] dis = new int[n][m];
        for(int i = 0 ; i < n ; i++){
            Arrays.fill(dis[i], 100000);
        }
        PriorityQueue<int[]> queue = new PriorityQueue<>(new Comparator<int[]>(){
            @Override
            public int compare(int[] a, int[] b){
                return a[0] - b[0];
            }
        });
        queue.add(new int[]{0, 0, 0});
        dis[0][0] = 0;
        int res = 100000;
        while(!queue.isEmpty()){
            int[] cur = queue.poll();
            int r = cur[1], c = cur[2], d = cur[0];
            if(r == n - 1 && c == m - 1){
                res = Math.min(res, d);
            }
            if(d > res){
                break;
            }
            if(d > dis[r][c]){
                continue;
            }
            for(int i = 1; i < 5; i++){
                int nr = r + dirs[i][0];
                int nc = c + dirs[i][1];
                if(nr < 0 || nc < 0 || nr >= n || nc >= m){
                    continue;
                }
                int tmp = d;
                if(grid[r][c] != i){
                    tmp++;
                }
                if(dis[nr][nc] > tmp){
                    dis[nr][nc] = tmp;
                    queue.add(new int[]{tmp, nr, nc});
                }
            }
        }
        return res;
    }
}
