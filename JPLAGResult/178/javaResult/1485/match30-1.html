class Solution {
    int[][] dp;
    int[][] g;
    
int[][] gos = {{0,0},{0, 1},{0, -1}, {1, 0},{-1, 0}};
    public int minCost(int[][] grid) {
        int m = grid.length;
        if(m ==0) return 0;
        int n = grid[0].length;
        if(n == 0) return 0;
        dp = new int[m][n];
        g = grid;
        for(int i = 0;i<m;i++) {
            for(int j=0;j<n;j++) {
                dp[i][j] = Integer.MAX_VALUE;
            }
        }
        dp[0][0] = 0;
        
        PriorityQueue<int[]> pg = new PriorityQueue<>((o1, o2) -> {
            return dp[o1[0]][o1[1]] - dp[o2[0]][o2[1]];
        });
        pg.add(new int[]{0,0});
        
        
while(!pg.isEmpty()) {
            int[] xy = pg.poll();
            for(int i = 1;i<=4;i++) {
                int[] go = gos[i];
                int x = xy[0] + go[0];
                int y = xy[1] + go[1];
                if(x>=0 && x<m && y>=0 && y<n) {
                    int tmp = dp[xy[0]][xy[1]];
                    if(grid[xy[0]][xy[1]]!= i) {
                        tmp++;
                    }
                    
                    
if(tmp<dp[x][y]) {
                        dp[x][y] = tmp;
                        pg.add(new int[]{x, y});
                    }
                }
            }
        }
        return dp[m-1][n-1];
    }
}
