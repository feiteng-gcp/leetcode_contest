class Solution {
    
    /*
    
    1 which means go to the cell to the right. (i.e go from grid[i][j] to grid[i][j + 1])
    2 which means go to the cell to the left. (i.e go from grid[i][j] to grid[i][j - 1])
    3 which means go to the lower cell. (i.e go from grid[i][j] to grid[i + 1][j])
    4 which means go to the upper cell. (i.e go from grid[i][j] to grid[i - 1][j])
    */
    
    
private static final int[][] Directions = {
        {0,1},
        {0,-1},
        {1,0},
        {-1,0}
    };
    
    private static final int[] Rev = {
        2,
        1,
        4,
        3
    };
    
    
public int minCost(int[][] grid) {
        final int r = grid.length, c = grid[0].length;
        final int[][] ans = new int[r][c];
        for (final int[] tmp : ans) {
            Arrays.fill(tmp, Integer.MAX_VALUE);
        }
        final PriorityQueue<int[]> q = new PriorityQueue(r * c, new Comparator<int[]>(){
            @Override
            public int compare(final int[] f, final int[] s) {
                return Integer.compare(f[0], s[0]);
            }
        });
        
        q.add(new int[]{0,r-1,c-1});
        ans[r-1][c-1] = 0;
        
        
while(!q.isEmpty()) {
            final int[] current = q.poll();
            for (int i = 0; i < Directions.length; i++) {
                final int newR = current[1] + Directions[i][0], newC = current[2] + Directions[i][1];
                if (0 <= newR && newR < r && 0 <= newC && newC < c) {
                    final int newPrice = current[0] + (grid[newR][newC] == Rev[i] ? 0 : 1);
                    
if (newPrice < ans[newR][newC]) {
                        ans[newR][newC] = newPrice;
                        q.offer(new int[]{newPrice, newR, newC});
                    }
                }
            }
        }
        
        return ans[0][0];
    }
}
