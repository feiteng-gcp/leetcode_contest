class Solution {
  public int minCost(int[][] grid) {
    int m = grid.length;
    int n = grid[0].length;
    int[][] dp = new int[m][n];
    for(int i = 0; i < m; i++){
      Arrays.fill(dp[i], m*n+1);
    }
    dfs(grid, dp, new boolean[m][n], 0, 0, 0);
    return dp[m-1][n-1];
  }
  
  int[][] dirs = new int[][]{{0,0},{0,1},{0,-1},{1,0},{-1,0}};
  public void dfs(int[][] grid, int[][] dp, boolean[][] visited, int i, int j, int cost){
    int m = grid.length;
    int n = grid[0].length;
    if(i < 0 || i >= m || j < 0 || j >= n){
      return;
    }
    if(visited[i][j]){
      return;
    }
    if(cost >= dp[i][j]){
      return;
    }
    
    dp[i][j] = Math.min(dp[i][j], cost);
    visited[i][j] = true;
    int gridValue = grid[i][j];
    dfs(grid, dp, visited, i + dirs[gridValue][0], j + dirs[gridValue][1], cost);
    for(int k = 1; k < dirs.length; k++){
      if(gridValue == k){
        continue;
      }
      // int c = gridValue == k ? 0 : 1;
      int nextI = i + dirs[k][0];
      int nextJ = j + dirs[k][1];
      dfs(grid, dp, visited, nextI, nextJ, cost + 1);
    }
    visited[i][j] = false;
  }
}
