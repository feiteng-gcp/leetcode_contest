class Solution {
    public int minCost(int[][] grid) {
        int m=grid.length;
        int n=grid[0].length;
        int[][] dp=new int[m][n];
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                dp[i][j]=Integer.MAX_VALUE;
            }
        }
        dp[0][0]=0;
        PriorityQueue<int[]> queue=new PriorityQueue<>(Comparator.comparing(i->i[2]));
        
queue.add(new int[]{0,0,0});
        int[][] dir=new int[][]{{0,1},{0,-1},{1,0},{-1,0}};
        while(!queue.isEmpty()){
            int[] pos=queue.poll();
            int x=pos[0];
            int y=pos[1];
            int c=pos[2];
            // System.out.format("x is %d, y is %d, c is %d\n",x,y,c);
            if(x==m-1&&y==n-1) break;
            int arrow=grid[x][y];
            for(int i=0;i<4;i++){
                int [] d=dir[i];
                int nx=x+d[0];
                int ny=y+d[1];
                int nc=dp[x][y]+(i+1==arrow?0:1);
                
if(nx>=0&&nx<m&&ny>=0&&ny<n&&dp[nx][ny]>nc){
                    dp[nx][ny]=nc;
                    queue.add(new int[]{nx,ny,nc});
                }
            }
        }
        return dp[m-1][n-1];
    }
}
