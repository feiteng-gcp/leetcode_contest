class Solution {
    public String rankTeams(String[] votes) {
        Candidate[] candidates = new Candidate[26];
        for (String v : votes) {
            for (int i = 0; i < v.length(); i++) {
                char c = v.charAt(i);
                if (candidates[c - 'A'] == null) candidates[c - 'A'] = new Candidate(c, v.length());
                candidates[c - 'A'].rank[i]++;
            }
        }
        
        List<Candidate> list = new ArrayList<>();
        for (Candidate can : candidates) {
            if (can == null) continue;
            list.add(can);
        }
        
        Collections.sort(list, new Comparator<Candidate>(){
            @Override
            public int compare(Candidate a, Candidate b) {
                int ret = compareRank(a.rank, b.rank);
                if (ret == 0) return a.c - b.c;
                return ret;
            }
        });
        
        StringBuilder sb = new StringBuilder();
        for (Candidate can : list) {
            sb.append(can.c);
        }
        
        return sb.toString();
    }
    
    private int compareRank(int[] a, int[] b) {
        for (int i = 0; i < a.length; i++) {
            if (a[i] != b[i]) return b[i] - a[i];
        }
        return 0;
    }
}

class Candidate{
    char c;
    int[] rank;
    
    Candidate(char c, int l) {
        this.c = c;
        rank = new int[l];
    }
}
