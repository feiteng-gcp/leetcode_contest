class Solution {
    public String rankTeams(String[] votes) {
int[][] arr = new int[26][votes[0].length() +1];
		for (int i = 0; i < 26; ++i) {
			arr[i][votes[0].length()] = 'A' + i;
		}
		for (String s : votes) {
			char[] chars = s.toCharArray();
			for (int i = 0; i < chars.length; ++i) {
				arr[chars[i] - 'A'][i] += 1;
			}
		}
		Arrays.sort(arr, new Comparator<int[]>() {
			@Override
			public int compare(int[] o1, int[] o2) {
				for (int i = 0; i < o1.length-1; ++i) {
					if (o1[i] != o2[i]) {
						return Integer.compare(o2[i], o1[i]);
					}
				}
				return Integer.compare(o1[ o1.length-1], o2[ o1.length-1]);
			}
		});
		StringBuilder stringBuilder = new StringBuilder();
		for(int i = 0; i < votes[0].length(); ++i) {
			stringBuilder.append((char)(arr[i][votes[0].length()]));
		}
		return stringBuilder.toString();
    }
}
