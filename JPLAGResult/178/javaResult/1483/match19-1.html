class Solution {
     public String rankTeams(String[] votes) {
        
Map<Character, TreeMap<Integer, Integer>> m = new HashMap<>();
        for (String v : votes) {
            for (int i = 0; i < v.length(); i++) {
                char c = 
v.charAt(i);
                TreeMap<Integer, Integer> cur = m.getOrDefault(c, new TreeMap<>());
                cur.put(i, cur.getOrDefault(i, 0) + 1);
                m.put(c, cur);
            }
        }
        Character[] r = new Character[m.keySet().size()];
        int i = 0;
        for (char k : m.keySet()) {
            r[i++] = k;
        }
        Arrays.sort(r, new Comparator<Character>() {
            @Override
            public int compare(Character o1, Character o2) {
                TreeMap<Integer, Integer> v1 = m.getOrDefault(o1, new TreeMap<>());
                TreeMap<Integer, Integer> v2 = m.getOrDefault(o2, new TreeMap<>());
                
Iterator<Map.Entry<Integer, Integer>> it1 = v1.entrySet().iterator();
                Iterator<Map.Entry<Integer, Integer>> it2 = v2.entrySet().iterator();
                while (it1.hasNext() && it2.hasNext()) {
                    Map.Entry<Integer, Integer> k1 = it1.next();
                    Map.Entry<Integer, Integer> k2 = it2.next();
                    if (k1.getKey() < k2.getKey()) {
                        return -1;
                    } else if (k1.getKey() > k2.getKey()) {
                        return 1;
                    } else {
                        if (k1.getValue() > k2.getValue()) {
                            return -1;
                        } else if (k1.getValue() < k2.getValue()) {
                            return 1;
                        }
                    }
                }
                if (it1.hasNext()) {
                    return -1;
                } else if (it2.hasNext()) {
                    return 1;
                } else {
                    return o1.compareTo(o2);
                }
            }

        });
        
StringBuilder sb = new StringBuilder();
        for (int j = 0; j < r.length; j++) {
            sb.append(r[j]);
        }
        return sb.toString();
    }
}
