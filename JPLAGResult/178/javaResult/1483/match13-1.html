class Solution {
    public String rankTeams(String[] votes) {
        StringBuilder sb = new StringBuilder();
        
int l = votes.length;
        int all = votes[0].length();
        int[][] count = new int[26][all];
        //count[i][j] is character i at jth votes have
        
        for (String v : votes) {
            for (int i = 0; i < all; i++) {
                char c = 
v.charAt(i);
                count[c - 'A'][i]++;
            }
        }
        
        PriorityQueue<Character> pq = new PriorityQueue<>(new Comparator<Character>() {
            public int compare(
Character c1, Character c2) {
                for (int i = 0; i < all; i++) {
                    if (count[c1 - 'A'][i] != count[c2 - 'A'][i]) {
                        return count[c2 - 'A'][i] - count[c1 - 'A'][i];
                    }
                }
                return c1 - c2;
            }
        });
        
        for (char c : votes[0].toCharArray()) {
            pq.offer(c);    
        }
        
        
while (!pq.isEmpty()) {
            sb.append(pq.poll());
        }
        return sb.toString();
    }
}
