class Solution {
    public String rankTeams(String[] vs) {
        
        int p = vs[0].length();
        
        int d[][] = new int[26][p];
        
        boolean hs[] = new boolean[26];
        for(String s:vs){
            
            for(int i=0;i<s.length();++i){
                int f = s.charAt(i)-'A';
                hs[f] = true;
                d[f][i]++;
            }
        }
        
        List<Integer> li = new ArrayList<>();
        for(int i=0;i<26;++i){
            if(hs[i]){
                li.add(i);
            }
        }
        
        Collections.sort(li,(x,y)->{
            for(int j=0;j<p;++j){
                if(d[x][j]>d[y][j]){
                    return -1;
                }else if(d[x][j]<d[y][j]){
                    return 1;
                }
            }
            return x-y;
        });
        StringBuilder sb = new StringBuilder();
        for(int u:li){
            sb.append((char)('A'+u));
        }        
        return sb.toString();
        
    }
}
