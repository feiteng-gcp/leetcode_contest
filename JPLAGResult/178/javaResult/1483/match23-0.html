class Solution {
    public String rankTeams(String[] votes) {
        boolean[]existed=new boolean[26];
        //最后一位表示当前字母
        int[][]score=new int[26][27];
        for(int i=0;i<26;i++){
            score[i][26]=i;
        }
        for(String str:votes){
            for(int i=0;i<str.length();i++){
                char c=str.charAt(i);
                existed[c-'A']=true;
                score[c-'A'][i]++;
            }
        }
        Arrays.sort(score,(a,b)->{
            for(int i=0;i<26;i++){
                if(a[i]!=b[i]){
                     return b[i]-a[i];
                }
            }
            return 0;
        });
        StringBuilder res=new StringBuilder();
        for(int i=0;i<26;i++){
            if(existed[score[i][26]]){
                res.append((char)(score[i][26]+'A'));
            }
        }
        return res.toString();
    }
}
