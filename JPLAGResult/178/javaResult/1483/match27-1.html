class Solution {
    public String rankTeams(String[] votes) {
        int[][] rank = new int[26][26];
        for (String vote : votes) {
            for (int i = 0; i < vote.length(); i++) {
                char c = vote.charAt(i);
                rank[c - 'A'][i]++;
            }
        }

        List<Node> list = new ArrayList<>();
        for (char c : votes[0].toCharArray()) {
            Node node = new Node(c, rank[c - 'A']);
            
list.add(node);
        }
        Collections.sort(list, new Comparator<Node>() {
            @Override
            public int compare(Node o1, Node o2) {
                int[] votes1 = o1.votes;
                int[] votes2 = o2.votes;
                for (int i = 0; i < 26; i++) {
                    if (votes1[i] != votes2[i])
                        return votes2[i] - votes1[i];
                }
                
return o1.letter - o2.letter;
            }
        });

        StringBuilder sb = new StringBuilder();
        for (Node node : list) {
            sb.append(node.letter);
        }
        return sb.toString();
    }

    
class Node {
        char letter;
        int[] votes;

        public Node(char letter, int[] votes) {
            this.letter = letter;
            this.votes = votes;
        }
    }
}
