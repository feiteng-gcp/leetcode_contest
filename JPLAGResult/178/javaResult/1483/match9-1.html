class Solution {
    public String rankTeams(String[] votes) {
        List<Character> list = new ArrayList<>();
        for (char c : votes[0].toCharArray()) {
            list.add(c);
        }
        
int[][] rank = new int[26][26];
        for (String v : votes) {
            for (int i = 0; i < v.length(); i++) {
                rank[v.charAt(i) - 'A'][i]++;
            }
        
}
        list.sort((c1, c2) -> {
            for (int i = 0; i < 26; i++) {
                if (rank[c1-'A'][i] > rank[c2-'A'][i]) {
                    return -1;
                } else if (rank[c1-'A'][i] < rank[c2-'A'][i]) {
                    return 1;
                }
            }
            return c1 - c2;
        });
        StringBuilder sb = new StringBuilder();
        for (char c : list) {
            sb.append(c);
        }
        return sb.toString();
    }
}
