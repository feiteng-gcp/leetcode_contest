class Solution {
    public String rankTeams(String[] votes) {
        int V = votes.length; // number of voters
        int T = votes[0].length(); // number of teams
        int[][] count = new int[T][26];
        for (String vote : votes) {
            for (int i = 0; i < T; i++) {
                char c = vote.charAt(i);
                count[i][c - 'A']++;
            }
        }
        List<Character> list = new ArrayList();
        for (char c : votes[0].toCharArray()) {
            list.add(c);
        }
        Collections.sort(list, (a, b) -> {
            for (int i = 0; i < T; i++) {
               if (count[i][a - 'A'] != count[i][b - 'A']) {
                   return count[i][b - 'A'] - count[i][a - 'A'];
               }
            }
            return a - b;
        });
        StringBuilder sb = new StringBuilder();
        for (char c : list) {
            sb.append(c);
        }
        return sb.toString();
    }
}
