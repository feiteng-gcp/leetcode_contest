class Solution {
    public String rankTeams(String[] votes) {
        int[][] rank = new int[26][votes[0].length() + 2];
        
for (int i = 0; i < 26; i++) {
            rank[i][rank[0].length - 1] = i;
        }
        for (String s : votes) {
            for (int i = 0; i < s.length(); i++) {
                char c = s.charAt(i);
                rank[c - 'A'][i]++;
                rank[c - 'A'][rank[0].length - 2]++;
            }
        }
        Arrays.sort(rank, new Comparator<int[]>() {
            @Override
            public int compare(
int[] o1, int[] o2) {
                for (int i = 0; i < o1.length - 1; i++) {
                    if (o1[i] != o2[i]) {
                        return Integer.compare(o2[i], o1[i]);
                    }
                }
                return Integer.compare(o1[o1.length - 1], o2[o2.length - 1]);
            }
        });
        
StringBuilder sb = new StringBuilder();
        for (int[] r : rank) {
            if (r[r.length - 2] > 0) {
                sb.append((char)('A' + r[r.length - 1]));
            }
        }
        return sb.toString();
    }
}
