class Solution {
    class Vote {
        char id;
        int[] freq;
        public Vote(char id) {
            this.id = id;
            this.freq = new int[26];
        }
    }
    public String rankTeams(String[] votes) {
        
Map<Character, Vote> map = new HashMap<>();
        for (String v : votes) {
            for (int i = 0; i < v.length(); i++) {
                char c = v.charAt(i);
                map.putIfAbsent(c, new Vote(c));
                map.get(c).freq[i]++;
            }
        }
        List<Vote> list = new ArrayList<>(map.values());
        
Collections.sort(list, new Comparator<Vote>(){
            @Override
            public int compare(Vote a, Vote b) {
                for (int i = 0; i < 26; i++) {
                    if (a.freq[i] != b.freq[i])
                        return b.freq[i] - a.freq[i];
                }
                return a.id - b.id;
            }
        });
        StringBuilder sb = new StringBuilder();
        for (Vote c : list)
            sb.append(c.id);
        return sb.toString();
    }
}
