class Solution {
    public String rankTeams(String[] votes) {
        int l = votes[0].length();
        int[][] counts = new int[26][votes[0].length()+1];
        for (int i = 0; i < 26; i++) {
            counts[i][0] = i;
        }
        for (String s : votes) {
            for (int i = 0; i < s.length(); i++) {
                counts[s.charAt(i)-'A'][i+1]++;
            }
        }
        Arrays.sort(counts, (a, b) -> {
            for (int i = 1; i < l+1; i++) {
                if (a[i] != b[i]) {
                    return -Integer.compare(a[i], b[i]);
                }
            }
            return a[0] - b[0];
        });
        for (int[] arr : counts) {
            System.out.println(Arrays.toString(arr));
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 26; i++) {
            boolean is = false;
            for (int e = 1; e < l+1; e++) {
                if (counts[i][e] > 0) {
                    is = true;
                    break;
                }
            }
            if (is) {
                sb.append((char)(counts[i][0]+'A'));
            }
        }
        return sb.toString();
    }
}
