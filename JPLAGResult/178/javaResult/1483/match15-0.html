class Solution {
    public String rankTeams(String[] votes) {
        int n = votes.length;
        int len = votes[0].length();
        int[][] dp = new int[26][26];
        for (String vote : votes) {
            for (int i = 0; i < vote.length(); i++) {
                dp[vote.charAt(i) - 'A'][i]++;
            }
        }
        // heap
        Queue<Character> q = new PriorityQueue<>(26, new Comparator<>(){
            @Override
            public int compare(Character o1, Character o2) {
                for (int i = 0; i < 26; i++) {
                    if (dp[o1- 'A'][i] != dp[o2-'A'][i]) {
                        // reverse
                        return dp[o2-'A'][i] - dp[o1-'A'][i];
                    }
                }     
                // alphabet
                return o1 - o2;
            }
        });
        String vote = votes[0];
        for (char c : vote.toCharArray()) {
            q.add(c);
        }
        StringBuilder sb = new StringBuilder();
        while (!q.isEmpty()) {
            sb.append(q.poll());
        }
        return sb.toString();
    }
}
