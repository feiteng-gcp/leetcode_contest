class Solution {
    public String rankTeams(String[] votes) {
        int[][] map = new int[26][26];
        Set<Character> set = new HashSet<>();
        for(String vote : votes) {
            for(int i = 0; i<vote.length(); i++) {
                map[vote.charAt(i)-'A'][i]++;
                set.add(vote.charAt(i));
            }
        }
        List<Character> teams = new ArrayList<>(set);
        Collections.sort(teams, new Comparator<Character>() {
           @Override
            public int compare(Character a, Character b) {
                for(int i = 0; i<26; i++) {
                    if(map[a-'A'][i]!=map[b-'A'][i]) return map[a-'A'][i]-map[b-'A'][i];
                }
                return b-a;
            }
        });
        StringBuilder ans = new StringBuilder();
        for(char team : teams) ans.append(team);
        return ans.reverse().toString();
    }
}
