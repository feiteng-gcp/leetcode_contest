class Solution {
    public String rankTeams(String[] votes) {
        
Map<Character, int[]> map = new HashMap<>();
        for (String v : votes) {
            for (int i = 0; i < v.length(); i++) {
                int[] cnt = map.get(v.charAt(i));
                if (cnt == null) cnt = new int[26];
                cnt[i]++;
                map.put(v.charAt(i), cnt);
            
}
        }
        
        List<Character> list = new ArrayList<>();
        for (char c : map.keySet()) list.add(c);
        Collections.sort(list, new Comparator<Character>(){
            @Override
            public int compare(
Character c1, Character c2) {
                int[] cnt1 = map.get(c1);
                int[] cnt2 = map.get(c2);
                for (int i = 0; i < cnt1.length; i++) {
                    if (cnt1[i] < cnt2[i]) return 1;
                    else if (cnt1[i] > cnt2[i]) return -1
;
                }
                
                return c1 - c2;
            }
        });
        StringBuilder sb = new StringBuilder();
        for (char c : list) sb.append(c);
        
        return sb.toString();
    }
}
