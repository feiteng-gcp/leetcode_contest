class Solution {
    public String rankTeams(String[] votes) {
        if (votes.length == 0) {
            return "";
        }

        final int n = votes[0].length();

        int[][] st = new int[26][n+1];
        for (int i = 0; i < 26; i++) {
            st[i][n] = i;
        }

        for (String vote : votes) {
            for (int r = 0; r < n; r++) {
                char c = vote.charAt(r);
                st[c-'A'][r]++;
            }
        }

        Arrays.sort(st, new Comparator<int[]>() {
            public int compare(int[] s1, int[] s2) {
                for (int r = 0; r < n; r++) {
                    int c = Integer.compare(s1[r], s2[r]);
                    if (c != 0) {
                        return -c;
                    }
                }
                return 0;
            }
        });

        char[] res = new char[n];
        for (int r = 0; r < n; r++) {
            res[r] = (char) ('A' + st[r][n]);
        }
        return String.valueOf(res);
    }
}
