class Solution {
    public String rankTeams(String[] votes) {
        int n=votes[0].length();
        
char[] chs1=votes[0].toCharArray();
        Character[] chs=new Character[n];
        for(int i=0;i<n;i++){
            chs[i]=chs1[i];
        }
        int[][] count=new int[26][n];
        
for(String str:votes){
            for(int i=0;i<n;i++){
                char c=str.charAt(i);
                count[c-'A'][i]++;
            }
        }
        Arrays.sort(chs,new Comparator<Character>(){
            public int compare(Character a,Character b){
                for(int i=0;i<n;i++){
                    if(count[a-'A'][i]>count[b-'A'][i]){
                        return -1;
                    }else if(count[a-'A'][i]<count[b-'A'][i]){
                        return 1;
                    
}
                }
                return a-b;
            }
        });
        char[] chs2=new char[n];
        for(int i=0;i<n;i++)chs2[i]=chs[i];
        return new String(chs2);
    }
}
