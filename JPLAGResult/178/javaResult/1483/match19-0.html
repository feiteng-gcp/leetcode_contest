class Solution {
    public String rankTeams(String[] votes) {
        HashSet<Character> visited = new HashSet<>();
        HashMap<Character, TreeMap<Integer, Integer>> charToRank = new HashMap<>();
        
        for(String v : votes){
            for(int i = 0; i < v.length(); i++){
                char c = v.charAt(i);
                visited.add(c);
                TreeMap<Integer, Integer> tm = charToRank.getOrDefault(c, new TreeMap<Integer, Integer>());
                tm.put(i, tm.getOrDefault(i, 0) + 1);
                charToRank.put(c, tm);
            }
        }
        
        ArrayList<Character> list = new ArrayList<>(visited);
        Collections.sort(list, (a, b) -> {
            TreeMap<Integer, Integer> t1 = charToRank.get(a);
            TreeMap<Integer, Integer> t2 = charToRank.get(b);
            Iterator<Map.Entry<Integer, Integer>> it1 = t1.entrySet().iterator();
            Iterator<Map.Entry<Integer, Integer>> it2 = t2.entrySet().iterator();
            while(it1.hasNext() && it2.hasNext()){
                Map.Entry<Integer, Integer> e1 = it1.next();
                Map.Entry<Integer, Integer> e2 = it2.next();
                if(e1.getKey() < e2.getKey()){
                    return -1;
                }else if(e1.getKey() > e2.getKey()){
                    return 1;
                }else{
                    if(e1.getValue() > e2.getValue()){
                        return -1;
                    }else if(e1.getValue() < e2.getValue()){
                        return 1;
                    }
                }
            }
            
            if(it1.hasNext()){
                return -1;
            }
            
            if(it2.hasNext()){
                return 1;
            }
            
            return (int)(a - b);
        });
        
        StringBuilder sb = new StringBuilder();
        for(char c : list){
            sb.append(c);
        }
        
        return sb.toString();
    }
}
