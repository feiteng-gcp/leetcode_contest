class Solution {
    public String rankTeams(String[] votes) {
        
int[][] score = new int[26][26];
        boolean[] has = new boolean[26];
        for(String vote : votes){
            for(int i = 0; i < vote.length(); i++){
                char cur = vote.charAt(i);
                has[cur - 'A'] = true;
                score[cur - 'A'][i]++;
            }
        }
        
        List<Character> res = new ArrayList<>();
        for(int i = 0; i < 26; i++){
            if(has[i]){
                res.add((char)('A' + i));
            }
        }
        
        Collections.sort(res, new Comparator<>(){
           public int compare(
Character c1, Character c2){
               for(int i = 0; i < 26; i++){
                   if(score[c1 - 'A'][i] > score[c2 - 'A'][i]){
                       return -1;
                   }
                   else if(score[c1 - 'A'][i] < score[c2 - 'A'][i]){
                       return 1;
                   }
               }
               
               return c1 - c2;
           } 
        });
        
        
StringBuilder out = new StringBuilder();
        for(char c : res){
            out.append(c);
        }
        
        return out.toString();
    }
}
