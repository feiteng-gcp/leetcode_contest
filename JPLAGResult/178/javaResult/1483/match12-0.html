import java.util.Arrays;

class Solution {
    public String rankTeams(String[] votes) {
        int m = votes.length, n = votes[0].length();
        int[][] cnt = new int[n][26];

        for (String vote : votes) {
            for(int i = 0; i < n; i++) {
                cnt[i][vote.charAt(i) - 'A']++;
            }
                
        }
        Character[] chars = new Character[n];

        for (int i = 0; i < n; i++) {
            chars[i] = votes[0].charAt(i);
        }

        Arrays.sort(chars, (o1, o2) -> {
            for (int i = 0; i < n; i++) {
                if (cnt[i][o1  - 'A'] != cnt[i][o2  - 'A']) {
                    return cnt[i][o2  - 'A'] - cnt[i][o1  - 'A'];
                }
            }
            return o1.compareTo(o2);
        });
        
        StringBuilder sb = new StringBuilder();
        for (char c : chars) {
            sb.append(c);
        }
        return sb.toString();

    }
}
