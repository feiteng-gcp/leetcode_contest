	class Solution {
	    public int[] smallerNumbersThanCurrent(int[] nums) {
	    	int[] ans = new int[nums.length];
	    	for (int idx=0;idx<nums.length;idx++) {
	    		ans[idx] = 0;
	    		for (int pos=0;pos<nums.length;pos++) {
	    			if (nums[pos] < nums[idx]) {
	    				ans[idx] ++;
	    			}
	    		}
	    	}
	    	return ans;
	    }
	}
