class Solution {
    String s = "croak";
    public int numOfArrays(int n, int m, int k) {
        long dp[][][] = new long[55][105][55];
        dp[0][0][0] = 1;
        // 第几个 最大值 k 数量
        for (int i = 1;i <= n;i ++) {
            for (int j = 1; j <= m;j ++) {
                for (int h = 1;h <= k;h ++) {
                    long sum = 0;
                    for (int l = 0;l < j;l ++) {
                        sum += dp[i - 1][l][h - 1];
                        sum %= 1000000007;
                    }
                    sum += (dp[i - 1][j][h] * (j) * 1L) % 1000000007;
                    sum %= 1000000007;
                    dp[i][j][h] = (int)sum;
                }
            }
        }
        long ret = 0;
        for (int j = 1;j <= m;j ++) {
            ret += dp[n][j][k];
            ret %= 1000000007;
        }
        return (int)ret;
    }
}
