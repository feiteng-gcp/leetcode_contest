class Solution {
    private static final int MOD = (int) 1e9 + 7;
    public int numOfArrays(int n, int m, int k) {
        int[][][] dp = new int[n][m][k];
        
        for (int i = 0; i < m; i++) {
            dp[0][i][0] = 1;
        }
        
        for (int i = 1; i < n; i++) {
            for (int cm = 0; cm < m; cm++) {
                for (int l = 0; l < k; l++) {
                    for (int j = 0; j <= cm; j++) {
                        dp[i][cm][l] += dp[i - 1][cm][l];
                        dp[i][cm][l] %= MOD;
                    }
                    if (l > 0) {
                        for (int j = 0; j < cm; j++) {
                            dp[i][cm][l] += dp[i - 1][j][l - 1];
                            dp[i][cm][l] %= MOD;
                        }
                    }
                
}
            }
            // System.out.println(Arrays.deepToString(dp[i]));
        }
        int res = 0;
        for (int i = 0; i < m; i++) {
            res = (res + dp[n - 1][i][k - 1]) % MOD;
        }
        return res;
    }
}
