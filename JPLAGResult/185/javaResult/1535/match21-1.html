class Solution {
    private static final long MOD = 1000000007L;
    
    public int numOfArrays(int n, int m, int K) {
        
long[][][] dp = new long[n + 2][n + 2][m + 1];
        for (int v = 1; v <= m; ++v) {
            dp[1][1][v] = 1;
        }
        
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= n; ++j) {
                for (int k = 1; k <= m; ++k) {
                    dp[i + 1][j][k] += dp[i][j][k] * k;
                    dp[i + 1][j][k] %= MOD;
                    for (int l = k + 1; l <= m; ++l) {
                        dp[i + 1][j + 1][l] += dp[i][j][k];
                        dp[i + 1][j + 1][k] %= MOD;
                    } 
                }
            }
        }
        
        long total = 0;
        for (int k = 1; k <= m; ++k) {
            total += dp[n][K][k];
            total %= MOD;
        }
        return (int)total;
    }
}
