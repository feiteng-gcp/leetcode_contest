class Solution {
    private final static long MOD = 1000L * 1000L * 1000L + 7;
    public int numOfArrays(int n, int m, int k) {
        long[][][] dp = new long[1 + n][1 + m][1 + k];
        dp[0][0][0] = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= m; j++) {
                for (int p = 0; p <= k; p++) {
                    long cur = dp[i][j][p];
                    if (cur == 0) {
                        continue;
                    }
                    for (int nj = 1; nj <= m; nj++) {
                        if (nj > j) {
                            if (p < k) {
                                dp[i + 1][nj][p + 1] += cur;
                                dp[i + 1][nj][p + 1] %= MOD;
                            }
                        
} else {
                            dp[i + 1][j][p] += cur;
                            dp[i + 1][j][p] %= MOD;
                        }
                    }
                }
            }
        }

        long res = 0;
        for (int j = 1; j <= m; j++) {
            res += dp[n][j][k];
            res %= MOD;
        }
        return (int) res;
    }
}
