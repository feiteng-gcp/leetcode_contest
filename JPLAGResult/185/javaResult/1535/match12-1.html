class Solution {
    
public int numOfArrays(int n, int m, int k) {
        long[][][] dp = new long[n + 1][m + 1][k + 1];
        
        // dp[a][b][c] means:
        // number of elements so far is a
        // current max is b
        // number of comparisons so far is c
        for(int i = 1; i <= m; i++) {
            dp[1][i][1] = 1;
        }
        for(int a = 2; a <= n; a++) {
            for(int c = 1; c <= k; c++) {
                for(int b = 1; b <= m; b++) {
                    dp[a][b][c] = dp[a - 1][b][c] * b;
                    dp[a][b][c] %= 1000000007;
                    
for(int d = 1; d < b; d++) {
                        dp[a][b][c] += dp[a - 1][d][c - 1];
                        dp[a][b][c] %= 1000000007;
                    }
                }
            }
        }
        
        long t = 0;
        for(int a = 1; a <= m; a++) {
            t += dp[n][a][k];
            t %= 1000000007;
        }
        
        return (int)t;
    }
}
