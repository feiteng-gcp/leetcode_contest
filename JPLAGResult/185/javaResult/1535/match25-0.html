class Solution {
    int MOD = 1000000007;
    public int numOfArrays(int n, int m, int k) {
        long[][][] dp = new long[n+1][m][k];
        for (int i = 0; i < m; i++) {
            dp[n][i][0] = 1;
        }
        for (int i = n-1; i >= 0; i--) {
            for (int e = 0; e < m; e++) {
                for (int w = 0; w < k; w++) {
                    for (int q = 0; q < m; q++) {
                        if (q > e && w > 0) {
                            dp[i][e][w] += dp[i+1][q][w-1];
                        } else if (q <= e) {
                            dp[i][e][w] += dp[i+1][e][w];
                        }
                        dp[i][e][w] %= MOD;
                    }
                }
            }
        }
        // for (long[][] mat : dp) {
        //     for (long[] arr : mat) {
        //         System.out.print(Arrays.toString(arr)+ " ");
        //     }
        //     System.out.println();
        // }
        long sum = 0;
        for (int i = 0; i < m; i++) {
            sum += dp[1][i][k-1];
            sum %= MOD;
        }
        return (int)sum;
        // return (int)dp[0][0][k-1];
    }
}
