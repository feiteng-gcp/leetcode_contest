class Solution {
    public int numOfArrays(int n, int m, int k) {
        int M = 1000000007;
        long[][][] dp = new long[n][m+1][k+1];
        for(int i = 1;i < m+1;++i) {
            dp[0][i][1] = 1;
        }
        for(int i = 1;i < n;++i) {
            for(int j = 1;j < m+1;++j) {
                for(int t = 1;t < k+1;++t) {
                    for(int s = 1;s < j;++s) {
                        dp[i][j][t]+=dp[i-1][s][t-1];
                        dp[i][j][t]%=M;
                        dp[i][j][t]+=dp[i-1][j][t];
                        dp[i][j][t]%=M;
                    }
                    dp[i][j][t]+=dp[i-1][j][t];
                    dp[i][j][t]%=M;
                }
            }
        }
        long res = 0;
        for(int i = 1;i < m+1;++i) {
            res+=dp[n-1][i][k];
            res%=M;
        }
        return (int)res;
    }
}
