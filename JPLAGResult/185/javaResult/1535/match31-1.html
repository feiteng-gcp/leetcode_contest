class Solution {
    public int numOfArrays(int n, int m, int k) {
        if(m<k){
            return 0;
        }
        
long[][][] dp=new long[n][k+1][m+1];
        for(int i=1;i<=m;i++){
            dp[0][1][i]=1;
        }

        for(int i=1;i<n;i++){
            for(int mi=1;mi<=m;mi++){
                for(int cm=1;cm<mi;cm++){
                    for(int ki=2;ki<=k;ki++){
                        dp[i][ki][mi]+=dp[i-1][ki-1][cm];
                        dp[i][ki][mi]%=1000000007L;
                    }
                }
                for(int cm=mi;cm<=m;cm++){
                    for(int ki=1;ki<=k;ki++){
                        dp[i][ki][cm]+=dp[i-1][ki][cm];
                        dp[i][ki][cm]%=1000000007L;
                    }
                }
            
}
        }
        long ans=0;
        for(int i=1;i<=m;i++){
            ans+=dp[n-1][k][i];
        }
        return (int)(ans%1000000007);
    }
}
