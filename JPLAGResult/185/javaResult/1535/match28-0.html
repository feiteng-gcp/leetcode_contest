 class Solution {
		 long mod = 1000000007;
		 public int numOfArrays(int n, int m, int k) {
	        long[][][] dp = new long[n+1][k+1][m+1];
	        dp[0][0][0] = 1;
	        for(int i = 1; i <= n; i++) {
	        	for(int c = 1; c <= k; c++) {
	        		for(int u = 0; u <= m; u++) {
	        			for(int v = 1; v <= m; v++) {
	        				if(v > u) {
	        					dp[i][c][v] += dp[i-1][c-1][u]; dp[i][c][v] %= mod;
	        				}
	        				else {
	        					dp[i][c][u] += dp[i-1][c][u]; dp[i][c][u] %= mod;
	        				}
	        			}
	        		}
	        	}
	        }
	        long res = 0;
	        for(int j = 1; j <= m; j++) {
	        	res += dp[n][k][j]; res %= mod;
	        }
	        return (int)res;
	    }
	}
