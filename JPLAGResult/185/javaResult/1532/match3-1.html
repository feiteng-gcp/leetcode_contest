class Solution {
    public String reformat(String s) {
        Queue<Character> alpha = new ArrayDeque<>();
        Queue<Character> numeric = new ArrayDeque<>();
        
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            
if (c >= '0' && c <= '9') {
                numeric.add(c);
            } else {
                alpha.add(c);
            }
        }
        
        Queue<Character> first = alpha;
        Queue<Character> second = numeric;
        
        if (numeric.size() > alpha.size()) {
            first = numeric;
            second = alpha;
        }
        
        if (first.size() - second.size() > 1) {
            return "";
        
}
        
        StringBuilder result = new StringBuilder();
        
        while (!first.isEmpty()) {
            result.append(first.remove());
            
            if (!second.isEmpty()) {
                result.append(second.remove());
            }
        }
        
        return result.toString();
    }
}
