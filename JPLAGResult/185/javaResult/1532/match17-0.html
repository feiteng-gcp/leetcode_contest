import java.util.Stack;

class Solution {
  public String reformat(String s) {
    Stack<Character> alpha = new Stack<>();
    Stack<Character> number = new Stack<>();
    
    for (char ch : s.toCharArray()) {
      if (ch >= 'a' && ch <= 'z') {
        alpha.add(ch);
      } else {
        number.add(ch);
      }
    }
    
    if (Math.abs(alpha.size() - number.size()) > 1) {
      return "";
    }
    StringBuilder sb = new StringBuilder();
    Stack<Character> greater = alpha.size() > number.size() ? alpha : number;
    Stack<Character> lesser = alpha.size() > number.size() ? number : alpha;
    
    while (greater.size() > 0) {
      sb.append(greater.pop());
      if (!lesser.isEmpty()) {
        sb.append(lesser.pop());
      }
    }
    
    return sb.toString();
  }
}
