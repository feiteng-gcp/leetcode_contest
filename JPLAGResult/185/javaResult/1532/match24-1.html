import java.util.ArrayList;
import java.util.List;


class Solution {
    public String reformat(String s) {
        List<Character> ab = new ArrayList<>();
        List<Character> num = new ArrayList<>();

        for (char c : s.toCharArray()) {
            if (c >= '0' && c <= '9') {
                num.add(c);
            } else {
                ab.add(c);
            }
        }

        if (num.size() > ab.size() + 1 || num.size() < ab.size() - 1) {
            return "";
        } else 
if (num.size() >= ab.size()) {
            return merge(num, ab);
        } else {
            return merge(ab, num);
        
}
    }

    private String merge(List<Character> a, List<Character> b) {

        StringBuilder sb = new StringBuilder();

        for (int i = 0; i < b.size(); i++) {
            sb.append(a.get(i));
            sb.append(b.get(i));
        }
        if (a.size() > b.size()) {
            sb.append(a.get(a.size() - 1));
        }
        return sb.toString();
    }
}
