class Solution {
    public String reformat(String s) {
        Queue<Character> digit = new LinkedList<>();
        Queue<Character> letter = new LinkedList<>();
        for(int i = 0; i < s.length(); i++) {
            if(Character.isDigit(s.charAt(i))) {
                digit.add(s.charAt(i));
            } else {
                letter.add(
s.charAt(i));
            }
        }
        if(Math.abs(digit.size() - letter.size()) >= 2) return "";
        StringBuilder sb = new StringBuilder();
        boolean isLetter = digit.size() <= letter.size();
        
while(digit.size() > 0 || letter.size() > 0) {
            if(isLetter) {
                sb.append(letter.poll());
            } else {
                sb.append(digit.poll());
            }
            isLetter = !isLetter;
        }
        return sb.toString();
    }
}
