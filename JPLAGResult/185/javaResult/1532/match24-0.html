class Solution {
    public String reformat(String s) {
        List<Character> chars = new ArrayList<>();
        List<Character> digits = new ArrayList<>();
        for (char ch: s.toCharArray()) {
            if (Character.isDigit(ch)) digits.add(ch);
            else chars.add(ch);
        }
        if (Math.abs(digits.size() - chars.size()) > 1) return "";
        if (chars.size() > digits.size()) {
            return concat(chars, digits);
        } else {
            return concat(digits, chars);
        }
    }
    
    private String concat(List<Character> l1, List<Character> l2) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < l2.size(); i++) {
            sb.append(l1.get(i));
            sb.append(l2.get(i));
        }
        if (l1.size() > l2.size()) sb.append(l1.get(l1.size() - 1));
        return sb.toString();
    }
}
