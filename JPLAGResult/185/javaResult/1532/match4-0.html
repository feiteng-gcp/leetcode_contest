class Solution {
    public String reformat(String s) {
        Deque<Character> t1 = new ArrayDeque();
        Deque<Character> t2 = new ArrayDeque();
        for(char c : s.toCharArray()){
            if(c >= '0' && c <= '9'){
                t1.addLast(c);
            }else{
                t2.addLast(c);
            }
        }
        if(Math.abs(t1.size() - t2.size()) > 1){
            return "";
        }
        if(t1.size() > t2.size()){
            Deque<Character> tmp = t1;
            t1 = t2;
            t2 = tmp;
        }
        StringBuilder ans = new StringBuilder();
        while(t2.size() > 0){
            ans.append(t2.removeFirst());
            if(t1.size() > 0){
                ans.append(t1.removeFirst());
            }
        }
        return ans.toString();
    }
}
