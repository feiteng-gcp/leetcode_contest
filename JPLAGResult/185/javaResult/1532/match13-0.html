class Solution {
    public String reformat(String s) {
        Stack<Character> letters = new Stack<>();
        Stack<Character> numbers = new Stack<>();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c >= '0' && c <= '9') {
                numbers.add(c);
            } else {
                letters.add(c);
            }
        }
        if (Math.abs(letters.size() - numbers.size()) > 1) {
            return "";
        }
        boolean letter = letters.size() > numbers.size();
        StringBuilder sb = new StringBuilder();
        while (letters.size() > 0 || numbers.size() > 0) {
            if (letter) {
                sb.append(letters.pop());
            } else {
                sb.append(numbers.pop());
            }
            letter = !letter;
        }
        return sb.toString();
    }
}
