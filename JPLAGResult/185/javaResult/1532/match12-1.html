class Solution {
    public String reformat(String s) {
        Stack<Character> alphabet = new Stack<>();
        Stack<Character> digits = new Stack<>();
        
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            
            if (ch >= '0' && ch <= '9') {
                digits.push(ch);
            } else {
                alphabet.push(ch);
            }
        }
        
        StringBuilder sb = new StringBuilder();
        while (!alphabet.isEmpty() && !digits.isEmpty()) {
            sb.append(alphabet.pop()).append(digits.pop());
        }
        
        if (alphabet.size() > 1 || digits.size() > 1) return "";
        
        if (alphabet.size() == 1) {
            sb.append(alphabet.pop());
        } else if (digits.size() == 1) {
            sb.insert(0, digits.pop());
        }
        
        return sb.toString();
    }
}
