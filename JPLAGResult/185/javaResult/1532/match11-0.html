class Solution {
    public String reformat(String s) {
        List<Character> chs = new ArrayList<>();
        List<Character> dig = new ArrayList<>();
        for(char ch : s.toCharArray()) {
            if(ch>='0' && ch<='9') dig.add(ch);
            else chs.add(ch);
        }
        if(Math.abs(chs.size()-dig.size())>1) return "";
        if(dig.size()>chs.size()) {
            List<Character> tmp = dig;
            dig=chs;
            chs=tmp;
        }
        int p = 0;
        int q = 0;
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i<s.length(); i++) {
            if(i%2==0 && p<chs.size()) sb.append(chs.get(p++));
            else sb.append(dig.get(q++));
        }
        return sb.toString();
    }
}
