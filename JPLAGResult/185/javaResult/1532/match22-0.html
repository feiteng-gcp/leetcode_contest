class Solution {
    public String reformat(String s) {
        List<Character> word = new ArrayList<>(), num = new ArrayList<>();
        for(char c : s.toCharArray()) {
            if(c >='0' && c <= '9') num.add(c);
            else word.add(c);
        }
        StringBuilder sb = new StringBuilder();
        if(Math.abs(word.size() - num.size()) > 1) return "";
        else if(word.size() >= num.size()) {
            int i= 0, j = 0 ;
            while(i < word.size() || j < num.size()) {
                if(i < word.size()) {
                    sb.append(word.get(i++));
                }
                if(j < num.size()) {
                    sb.append(num.get(j++));
                }
            }
        }else {
            int i = 0, j = 0;
            while(i < word.size() || j < num.size()) {
                if(j < num.size()) {
                    sb.append(num.get(j++));
                }
                if(i < word.size()) {
                    sb.append(word.get(i++));
                }
            }
            
        }
        return sb.toString();
    }
}
