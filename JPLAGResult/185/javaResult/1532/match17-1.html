class Solution {
    public String reformat(String s) {
        ArrayList<Character> clist = new ArrayList<>();
        ArrayList<Character> nlist = new ArrayList<>();
        for (char c : s.toCharArray()) {
            if (c >= '0' && c <= '9') {
                nlist.add(c);
            } else {
                clist.add(c);
            }
        }
        if (Math.abs(clist.size() - nlist.size()) > 1) {
            return "";
        }
        if (clist.size() > nlist.size()) {
            ArrayList<Character> temp = clist;
            clist = nlist;
            nlist = temp;
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 
nlist.size(); i++) {
            sb.append(nlist.get(i));
            if (i < clist.size()) {
                sb.append(clist.get(i));   
            }
        }
        return sb.toString();
    }
}
