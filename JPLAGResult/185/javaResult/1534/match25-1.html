class Solution {
    public int minNumberOfFrogs(String s) {
        char[] chars = s.toCharArray();
        String ss = "croak";
        int n = chars.length;
        int[] count = new int[5];
        int cc = 0, max = 0;
        for (
char c : chars) {
            int index = ss.indexOf(c);
            if (index == -1) {
                return -1;
            }
            
if (index > 0 && count[index-1] == 0) {
                return -1;
            }
            if (index > 0) {
                count[index-1]--;
            }
            count[index]++;
            
            if (index == 0) {
                cc++;
            }
            if (index == 4) {
                cc--;
            }
            max = Math.max(max, cc);
        }
        if (count[0] + count[1] + count[2] + count[3] > 0) {
            return -1;
        }
        return max;
    }
}
