class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int ret = 0;
        int numCroaks = 0;
        int[] chars = new int[5];
        for (int i = 0; i < croakOfFrogs.length(); i++) {
            int index = croakMap(croakOfFrogs, i);
            if (index > 0 && chars[index-1] == 0) {
                return -1;
            }
            if (index == 0) {
                numCroaks++;
            }
            if (index > 0) {
                chars[index-1]--;
            }
            if (index == 4) {
                numCroaks--;
            }
            ret = Math.max(ret, numCroaks);
            chars[index]++;
        }
        if (chars[0] == 0 && chars[1] == 0 && chars[2] == 0 && chars[3] == 0) {
            return ret;
        } else {
            return -1;
        }
    }
    
    public int croakMap(String s, int index) {
        char c = s.charAt(index);
        if (c == 'c') {
            return 0;
        }
        if (c== 'r') {
            return 1;
        }
        if (c=='o') {
            return 2;
        }
        if (c=='a') {
            return 3;
        }
        return 4;
    }
}
