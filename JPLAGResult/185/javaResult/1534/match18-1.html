class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int c = 0,o = 0, r= 0, a= 0, k = 0;
        int max = 0;
        for(char ch: croakOfFrogs.toCharArray()){
            if(ch == 'c'){
                c++;
                if(c > max)max = c;
            
} else if(ch == 'r'){
                r++;
                if(r > c)return -1;
            } else if(ch == 'o'){
                o++;
                if(o > r) return -1;
            } else if(ch == 'a'){
                a++;
                if(a > o)return -1;
            } else if(ch == 'k'){
                k ++;
                if(k > a)return -1;
                c--;
                r--;
                o--;
                a--;
                k--;
            } else {
                return -1;
            }
        }
        if(c > 0)return -1;
        return max;
    }
}
