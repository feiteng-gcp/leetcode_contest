class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        
int arr[]=new int[200];
        for(int i=0;i<croakOfFrogs.length();i++) {
            char ch = croakOfFrogs.charAt(i);
            if (ch=='c') {
                if (arr['k']>0) {arr['k']--;}
                arr['c']++;
            }
            if (ch=='r') 
{
                if (arr['c']>0) {arr['c']--;} else return -1;
                arr['r']++;
            }
            if (ch=='o') {
                if (arr['r']>0) {arr['r']--;} else return -1;
                arr['o']++;
            }
            if (ch=='a') {
                if (arr['o']>0) {arr['o']--;} else return -1;
                arr['a']++;
            }
            if (ch=='k') {
                if (arr['a']>0) {arr['a']--;} else return -1;
                arr['k']++;
            }
        
}
        for(int i=0;i<arr.length;i++)if (i!='k'&&arr[i]>0)return -1;
        return arr['k'];
    }
}
