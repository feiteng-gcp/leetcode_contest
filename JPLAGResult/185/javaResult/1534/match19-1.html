class Solution {
    public int minNumberOfFrogs(String cof) {
        int ans = 0;
        int[] nf = new int[5];
        for (char c : cof.toCharArray()) {
            
if (c=='c') {
                if (nf[4]>0)
                    nf[4]--;
                nf[0]++;
            }
            if (c=='r') {
                nf[0]--;
                nf[1]++;
                if (nf[0]<0)
                    return -1;
            }
            if (c=='o') {
                nf[1]--;
                nf[2]++;
                if (nf[1]<0)
                    return -1;
            }
            if (c=='a') {
                nf[2]--;
                nf[3]++;
                if (nf[2]<0)
                    return -1;
            }
            if (c=='k') {
                nf[3]--;
                nf[4]++;
                if (nf[3]<0)
                    return -1;
            }
        }
        if (nf[0]>0||nf[1]>0||nf[2]>0||nf[3]>0)
            return -1;
        return nf[4];
    }
}
