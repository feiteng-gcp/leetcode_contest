class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int[] cnt = new int[5];
        for(int i = 0;i < croakOfFrogs.length();++i) {
            char c = croakOfFrogs.charAt(i);
            if(c=='c') {
                if(cnt[4]>0) cnt[4]--;
                cnt[0]++;
            }
            if(c=='r') {
                if(cnt[0]==0) return -1;
                cnt[0]--;
                cnt[1]++;
            }
            if(c=='o') {
                if(cnt[1]==0) return -1;
                cnt[1]--;
                cnt[2]++;
            }
            if(c=='a') {
                if(cnt[2]==0) return -1;
                cnt[2]--;
                cnt[3]++;
            }
            if(c=='k') {
                if(cnt[3]==0) return -1;
                cnt[3]--;
                cnt[4]++;
            
}
        }
        for(int i = 0;i < 4;++i) {
            if(cnt[i]!=0) return -1;
        }
        return cnt[4];
    }
}
