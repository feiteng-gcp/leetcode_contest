class Solution {
    //每只青蛙都按顺序 c r o a k, 求最少几只青蛙能完成string，注意每只青蛙要发完5个音
    //看着就像是greedy，音都不一样，保证前面的音总是多于后面的音
    public int minNumberOfFrogs(String croakOfFrogs) {
        if(croakOfFrogs.length() % 5 !=0) return -1;
        int c = 0;
        int r = 0;
        int o = 0;
        int a = 0;
        //int k = 0;
        int min = 0;
        for(int i=0;i<croakOfFrogs.length();i++){
            if(croakOfFrogs.charAt(i) == 'c'){
                c++;
            } else if(croakOfFrogs.charAt(i) == 'r'){
                if(c < 1) return -1;
                c--;
                r++;
            } else if(croakOfFrogs.charAt(i) == 'o'){
                if(r < 1) return -1;
                r--;
                o++;
            } else if(croakOfFrogs.charAt(i) == 'a'){
                if(o < 1) return -1;
                o--;
                a++;
            } else {
                if(a < 1) return -1;
                a--;
                //k++;
                //min = Math.max(min, k);
            }
            min = Math.max(min, c + r + o + a);
        }
        if(c > 0 || r > 0 || o > 0 || a > 0) return -1;
        return min;
    }
}
