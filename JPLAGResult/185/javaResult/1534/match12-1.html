class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int[] cnt = new int[5];
        int frogs = 0;
        for (char ch: croakOfFrogs.toCharArray()) {
            if (ch == 'c') {
                cnt[0]++;
            
} else if (ch == 'r') {
                if (cnt[1] >= cnt[0]) return -1;
                cnt[1]++;
            } else if (ch == 'o') {
                if (cnt[2] >= cnt[1]) return -1;
                cnt[2]++;
            } else if (ch == 'a') {
                if (cnt[3] >= cnt[2]) return -1;
                cnt[3]++;
            } else if (ch == 'k') {
                if (cnt[4] >= cnt[3]) return -1;
                frogs = Math.max(frogs, cnt[0] - cnt[4]);
                cnt[4]++;
            }
        }
        
        if (cnt[0] == cnt[1] && cnt[1] == cnt[2] && cnt[2] == cnt[3] && cnt[3] == cnt[4]) return frogs;
        return -1;
    }
}
