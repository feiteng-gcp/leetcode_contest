class Solution {
    public int minNumberOfFrogs(String cc) {
        char ck[] = {'c','r','o','a','k'};
        int d[] =new int[256];
        
        for(char c:cc.toCharArray()){
            d[c]++;
        }
        int w = -1;
        for(char me:ck){
            if(w==-1){
                w = d[me];
            }else if(w!=d[me]){
                return -1;
            }
        }
        d = new int[256];
        int mx = 0;
        for(char c:cc.toCharArray()){
            if(c=='c'){
                d[c]++;
                
            }else if(c=='r')
{
                d['r']++;
                if(d['r']<=d['c']){
                   
                    
                }else{
                    return -1;
                }
            }else if(c=='o')
{
                d['o']++;
                if(d['o']<=d['r']){
                    
                    
                }else{
                    return -1;
                }
            }else if(c=='a')
{
                d['a']++;
                if(d['a']<=d['o']){
                    
                    
                }else{
                    return -1;
                }
            } else if(c=='k')
{
                d['k']++;
                if(d['k']<=d['a']){
                    d['c']--;
                    d['r']--;
                    d['o']--;
                    d['a']--;
                   d['k']--;
                }else{
                    return -1;
                }
            }  
            mx = Math.max(mx,d['c']);
          //  System.out.println(mx);
        }
       // System.out.println(w);
        
        return mx;
        
        
    }
}
