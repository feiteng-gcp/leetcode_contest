class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        int[] index = new int[128];
        index['c'] = 1;
        index['r'] = 2;
        index['o'] = 3;
        index['a'] = 4;
        index['k'] = 5;
        int[] cnt = new int[10];
        cnt[0] = 0;
        for(char c : croakOfFrogs.toCharArray()){
            if(cnt[index[c] - 1] == 0)
            {
                if(index[c] > 1){
                    return -1;
                }else{
                    cnt[0]++;
                }
            }
            cnt[index[c] - 1]--;
            if(index[c] == 5){
                
            }
            cnt[index[c] % 5]++;
        }
        
        for(int i = 1; i <= 5; i++){
            if(cnt[i] > 0){
                return -1;
            }
        }
        return cnt[0];
    }
}
