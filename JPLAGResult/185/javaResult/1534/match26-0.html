class Solution {
    public int minNumberOfFrogs(String c) {
        int[] count = new int[5];
        int num = 0;
        
        for(int i = 0; i < c.length(); i++) {
        	if(c.charAt(i) == 'c') {
        		count[0]++;
        		num = Math.max(num, count[0]);
        	}
        	else if(c.charAt(i) == 'r' && count[0] > count[1]){
        		count[1]++;
        	}
        	else if(c.charAt(i) == 'o' && count[1] > count[2]){
        		count[2]++;
        	}
        	else if(c.charAt(i) == 'a' && count[2] > count[3]){
        		count[3]++;
        	}
        	else if(c.charAt(i) == 'k' && count[3] > count[4]){
        		count[4]++;
        		count[4]--;
        		count[3]--;
        		count[2]--;
        		count[1]--;
        		count[0]--;
        	}
        	else {
        		return -1;
        	}
        }
        int v = count[0];
        for(int i = 0; i < 5; i++) {
        	if(v != count[i]) return -1;
        }
        return num;
    }
}
