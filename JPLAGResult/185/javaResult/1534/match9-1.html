class Solution {
    public int minNumberOfFrogs(String croakOfFrogs) {
        String cc = "croak";
        
int[] cnt = new int[5];
        for (char ch : croakOfFrogs.toCharArray()) {
            int idx = cc.indexOf(ch);
            if (idx == 0) {
                if (cnt[0] == 0) {
                    cnt[0]++;
                }
                cnt[0]--;
                cnt[1]++;
            } else {
                if (cnt[idx] == 0) return -1;
                cnt[idx]--;
                cnt[(idx + 1) % 5]++;
            
}
        }
        for (int i = 1; i < 5; ++i) {
            if (cnt[i] > 0) return -1;
        }
        return cnt[0];
    }
}
