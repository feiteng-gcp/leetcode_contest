class Solution {
    public int minNumberOfFrogs(String s) {
        int[] croak = new int[5];
        int started = 0;
        
int answer = 0;
        for (int i=0; i < s.length(); ++i) {
            if (s.charAt(i) == 'c') {
                started++;
                answer = Math.max(answer, started);
                croak[0]++;
            } else if (s.charAt(i) == 'r') {
                if (croak[0] > 0) {
                    croak[0]--;
                    croak[1]++;
                } else {
                    return -1;
                }
            } else if (s.charAt(i) == 'o') {
                if (croak[1] > 0) {
                    croak[1]--;
                    croak[2]++;
                } else {
                    return -1;
                }
            } else if (s.charAt(i) == 'a') {
                
                if (croak[2] > 0) {
                    croak[2]--;
                    croak[3]++;
                } else {
                    return -1;
                }
            } else if (s.charAt(i) == 'k') {
                started--;
                if (croak[3] > 0) {
                    croak[3]--;
                    croak[4]++;
                } else {
                    return -1;
                }
            
} else {
                return -1;
            }
        }
        
for (int i=0; i <4; ++i) {
            if (croak[i] != 0) return -1;
        }
        return answer;
    }
}
