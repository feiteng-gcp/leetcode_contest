class Solution {
    public int minNumberOfFrogs(String s) {
        int c = 0, r = 0, o = 0, a = 0, k = 0;
        int sum = 0;
        
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            
            if (ch == 'c') {
                c++;
                if (k > 0) k--;
            } else if (ch == 'r') {
                r++;
                if (c == 0) return -1;
                c--;
            } else if (ch == 'o') {
                o++;
                if (r == 0) return -1;
                r--;
            } else if (ch == 'a') {
                a++;
                if (o == 0) return -1;
                o--;
            } else if (ch == 'k') {
                k++;
                if (a == 0) return -1;
                a--;
            }
            
            sum = Math.max(sum, c + r + o + a + k);
        }
        
        return c + r + o + a == 0 ? sum : -1;
    }
}
