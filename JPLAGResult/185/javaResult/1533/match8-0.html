class Solution {
    
    public List<List<String>> displayTable(List<List<String>> orders) {
        Set<String> menuSet = new TreeSet<>();
        Map<Integer, Map<String, Integer>> output = new TreeMap<>();
        for(List<String> order : orders) {
            int table = Integer.parseInt(order.get(1));
            String menu = order.get(2);
            output.putIfAbsent(table, new TreeMap<>());
            output.get(table).putIfAbsent(menu, 0);
            output.get(table).put(menu, output.get(table).get(menu) + 1);
            menuSet.add(menu);
        }
        List<List<String>> outputTable = new ArrayList<>();
        List<String> header = new ArrayList<>();
        header.add("Table");
        for(String menu : menuSet) {
            header.add(menu);
        }
        outputTable.add(header);
        for(Map.Entry<Integer, Map<String, Integer>> e : output.entrySet()) {
            List<String> currTable = new ArrayList<>();
            currTable.add(e.getKey()+"");
            for(String menu : menuSet) {
                currTable.add(e.getValue().getOrDefault(menu, 0)+"");
            }
            outputTable.add(currTable);
        }
        return outputTable;
    }
}
