class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeMap<Integer, Map<String, Integer>> tables = new TreeMap<>();
        TreeSet<String> items = new TreeSet<>();
        
        for(List<String> order : orders){
            Map<String, Integer> map = tables.computeIfAbsent(Integer.valueOf(order.get(1)), k->new HashMap<>());
            items.add(order.get(2));
            map.put(order.get(2), map.getOrDefault(order.get(2), 0)+1);
        }
        
        List<List<String>> result = new ArrayList<>();
        
        List<String> header = new ArrayList<>();
        header.add("Table");
        header.addAll(items);
        result.add(header);
        
        
        for(Map.Entry<Integer, Map<String, Integer>> e : tables.entrySet()){
            List<String> list = new ArrayList<>();
            list.add(String.valueOf(e.getKey()));
            
            for(String item : items){
                list.add(String.valueOf(e.getValue().getOrDefault(item, 0)));
            }
            result.add(list);
        }
        return result;
        
    }
}
