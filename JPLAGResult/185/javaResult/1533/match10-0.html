class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        Set<String> foods = new HashSet<>();
        TreeMap<Integer, Map<String, Integer>> tables = new TreeMap<>();
        
        for (List<String> order: orders) {
            Map<String, Integer> map = tables.computeIfAbsent(Integer.parseInt(order.get(1)), k -> new HashMap<>());
            map.put(order.get(2), map.getOrDefault(order.get(2), 0) + 1);
            foods.add(order.get(2));
        }
        
        int n = foods.size();
        List<String> first = new ArrayList<>(foods);
        Collections.sort(first);
        first.add(0, "Table");
        
        List<List<String>> res = new LinkedList<>();
        res.add(first);
        
        for (int num: tables.keySet()) {
            List<String> line = new ArrayList<>();
            line.add(String.valueOf(num));
            for (int i = 1; i <= n; i++) {
                line.add(String.valueOf(tables.get(num).getOrDefault(first.get(i), 0)));    
            }
            res.add(line);
        }
        
        return res;
    }
}
