import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map.Entry;
import java.util.TreeMap;
import java.util.TreeSet;

class Solution {
  public List<List<String>> displayTable(List<List<String>> input) {
    TreeSet<String> items = new TreeSet<>();
    TreeMap<Integer, HashMap<String, Integer>> orders = new TreeMap<>();
    
    for (List<String> order : input) {
      int table = Integer.valueOf(order.get(1));
      String food = order.get(2);
      items.add(food);
      
      orders.put(table, orders.getOrDefault(table, new HashMap<>()));
      orders.get(table).put(food, orders.get(table).getOrDefault(food, 0) + 1);
    }
    
    List<List<String>> ret = new ArrayList<>();
    List<String> header = new ArrayList<>();
    header.add("Table");
    header.addAll(items);
    ret.add(header);
    for (Entry<Integer, HashMap<String, Integer>> entry : orders.entrySet()) {
      ArrayList<String> row = new ArrayList<>();
      row.add(entry.getKey() + "");
      for (String item : items) {
        row.add(entry.getValue().getOrDefault(item, 0) + "");
      }
      ret.add(row);
    }
    return ret;
  }
}
