class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        Map<Integer,  Map<String, Integer>> tables = new TreeMap<>();
        SortedSet<String> items = new TreeSet<>();
        
        for (List<String> order : orders) {
            Integer tNum = Integer.valueOf(order.get(1));
            String item = order.get(2);
            
            items.add(item);
            
            Map<String, Integer> table = tables.computeIfAbsent(tNum, k -> new HashMap<>());
            table.compute(item, (k, v) -> (v == null) ? 1 : v + 1);
        }
        
        System.out.println(items);
        System.out.println(tables);
        
        List<List<String>> result = new ArrayList<>();
        
        List<String> header = new ArrayList<>();
        header.add("Table");
        for (String item : items) {
            header.add(item);
        }
        
        result.add(header);
        
        for (Map.Entry<Integer, Map<String, Integer>> tEntry : tables.entrySet()) {
            Map<String, Integer> table = tEntry.getValue();
            
            List<String> tResult = new ArrayList<>();
            tResult.add(tEntry.getKey().toString());

            for (String item : items) {
                tResult.add(table.getOrDefault(item, 0).toString());
            }
            
            result.add(tResult);
        }
        
        return result;
    }
}
