class Solution {
    //题目好难读
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeMap<Integer, TreeMap<String, Integer>> display = new TreeMap<>();
        TreeSet<String> set = new TreeSet<>();
        for(List<String> order : orders){
            String name = order.get(0);
            int table = Integer.parseInt(order.get(1));
            String food = order.get(2);
            
            set.add(food);
            TreeMap<String, Integer> map = display.getOrDefault(table, new TreeMap<>());
            map.put(food, map.getOrDefault(food, 0) + 1);
            display.put(table, map);
        }
        
        List<List<String>> res = new ArrayList<>();
        //first row
        List<String> title = new ArrayList<>();
        title.add("Table");
        for(String item : set) title.add(item);
        res.add(title);
        
        //data row
        for(Map.Entry<Integer, TreeMap<String, Integer>> entry : display.entrySet()){
            int table = entry.getKey();
            TreeMap<String, Integer> items = entry.getValue();
            List<String> row = new ArrayList<>();
           // System.out.println("1");
            row.add(String.valueOf(table));
            for(String item : set){
                row.add(String.valueOf(items.getOrDefault(item, 0)));
            }
            res.add(row);
        }
        return res;
    }
}
