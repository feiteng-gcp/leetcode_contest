class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        Map<Integer, Map<String, Integer>> dishes = new HashMap<>();
        Set<Integer> ts = new HashSet<>();
        Set<String> ds = new HashSet<>();
        
        for (List<String> o : orders) {
            int t = Integer.parseInt(o.get(1));
            String d = o.get(2);
            ds.add(d);
            ts.add(t);
            
            dishes.putIfAbsent(t, new HashMap<>());
            
int c = dishes.get(t).getOrDefault(d, 0) + 1;
            dishes.get(t).put(d, c);
        }
        
        List<String> hs = new ArrayList<>();
        List<String> tmp = new ArrayList<>(ds);
        Collections.sort(tmp);
        hs.add("Table");
        hs.addAll(tmp);
        
        List<List<String>> res = new ArrayList<>();
        res.add(hs);
        
        List<Integer> list = new ArrayList<>(ts);
        Collections.sort(list);
        
for (int t : list) {
            List<String> row = new ArrayList<>();
            row.add(String.valueOf(t));
            for (int i = 1; i < hs.size(); i++) {
                String d = hs.get(i);
                int c = 
dishes.get(t).getOrDefault(d, 0);
                row.add(String.valueOf(c));
            }
            res.add(row);
        }
        return res;
    }
}
