class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeSet<Integer> tables = new TreeSet<>();
        TreeSet<String> foods = new TreeSet<>();
        HashMap<String,Integer> cnts = new HashMap<>();

        for(List<String> o : orders){
            tables.add(Integer.parseInt(o.get(1)));
            foods.add(o.get(2));
            String key = o.get(1) + "#" + o.get(2);
            cnts.put(key,cnts.getOrDefault(key,0) + 1);
        }

        List<List<String>> ret = new ArrayList<>();
        List<String> head = new ArrayList<>();
        head.add("Table");
        head.addAll(foods);
        ret.add(head);

        for(int t : tables){
            List<String> row = new ArrayList<>();
            row.add(Integer.toString(t));
            for(String f : foods){
                row.add(Integer.toString(cnts.getOrDefault(t + "#" + f,0)));
            }
            ret.add(row);
        }

        return ret;
    }
}
