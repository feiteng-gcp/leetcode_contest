class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        Map<Integer, Map<String, Integer>> to = new HashMap<>(); 
        Set<Integer> tableSet = new TreeSet<>();
        Set<String> foodSet = new TreeSet<>();
        
        for (List<String> order : orders) {
            String c = order.get(0);
            int t = Integer.parseInt(order.get(1));
            String f = order.get(2);
            
            tableSet.add(t);
            foodSet.add(f);
            
            Map<String, Integer> map = to.get(t);
            if (map == null) {
                map = new HashMap<>();
                to.put(t, map);
            }
            
            int count = map.getOrDefault(f, 0);
            map.put(f, count + 1);
        }
        
        List<List<String>> result = new ArrayList<>();
        List<String> header = new ArrayList<>();
        header.add("Table");
        header.addAll(foodSet);
        result.add(header);
        
        for (int t : tableSet) {
            List<String> row = new ArrayList<>();
            row.add(String.valueOf(t));
            Map<String, Integer> fc = to.get(t);
            for (String f : foodSet) {
                int count = fc.getOrDefault(f, 0);
                row.add(String.valueOf(count));
            }
            result.add(row);
        }
        return result;
    }
}
