class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeSet<String> items = new TreeSet<>();
        TreeMap<Integer, int[]> map = new TreeMap<>();
        
        for (List<String> list : orders) {
            items.add(list.get(2));
        }
        
        ArrayList<String> itemsArr = new ArrayList<>();
        for (String item : items) {
            itemsArr.add(item);
        }
        
        HashMap<String, Integer> itemMap = new HashMap<>();
        
        for (int i = 0;i < itemsArr.size(); i++) {
            itemMap.put(itemsArr.get(i), i);
        }
        
        int size = items.size();
        
        for (List<String> list : orders) {
            int table = Integer.parseInt(list.get(1));
            String item = list.get(2);
            if (!map.containsKey(table)) {
                map.put(table, new int[size]);
            }
            map.get(table)[itemMap.get(item)]++;
        }
        List<List<String>> ret = new LinkedList<>();
        List<String> firstRow = new LinkedList<>();
        firstRow.add("Table");
        for (String item : items) {
            firstRow.add(item);
        }
        ret.add(firstRow);
        for (Map.Entry<Integer, int[]> entry : map.entrySet()) {
            List<String> row = new LinkedList<>();
            row.add("" + entry.getKey());
            for (int i = 0; i < entry.getValue().length; i++) {
                row.add("" + entry.getValue()[i]);
            }
            ret.add(row);
        }
        return ret;
    }
}
