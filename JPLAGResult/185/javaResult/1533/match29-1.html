class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeMap<Integer, HashMap<String, Integer>> map = new TreeMap<>();
        TreeSet<String> all = new TreeSet<>();
        for (List<String> order : orders) {
            
int table = Integer.parseInt(order.get(1));
            String item = order.get(2);
            if (!map.containsKey(table)) {
                map.put(table, new HashMap<>());
            }
            HashMap<String, Integer> asdf = map.get(table);
            asdf.put(item, asdf.getOrDefault(item, 0) + 1);
            all.add(item);
        }
        List<List<String>> res = new ArrayList<>();
        List<String> list = new ArrayList<>();
        list.add("Table");
        for (String s : all) {
            list.add(s);
        }
        res.add(list);
        for (int table : map.keySet()) {
            list = new ArrayList<>();
            list.add(table + "");
            HashMap<String, Integer> asdf = map.get(table);
            for (String item : all) {
                list.add(asdf.getOrDefault(item, 0) + "");
            
}
            res.add(list);
        }
        
        return res;
    }
}
