class Solution {
    public List<List<String>> displayTable(List<List<String>> orders) {
        TreeMap<Integer, HashMap<String, Integer>> map = new TreeMap<>();
        TreeSet<String> name = new TreeSet<>();
        
        for(List<String> order : orders){
            HashMap<String, Integer> inner = map.computeIfAbsent(Integer.parseInt(order.get(1)), (id) -> new HashMap<>());
            inner.put(order.get(2), inner.getOrDefault(order.get(2), 0) + 1);
            name.add(order.get(2));
        }
        
        List<List<String>> res = new ArrayList<>(map.size());
        List<String> headers = new ArrayList<>(1 + name.size());
        headers.add("Table");
        for(String n : name){
            headers.add(n);
        }
        res.add(headers);
        
        for(Integer key : map.keySet()){
            HashMap<String, Integer> order = map.get(key);
            List<String> row = new ArrayList<>(1 + name.size());
            row.add(String.valueOf(key));
            for(String n : name){
                row.add(String.valueOf(order.getOrDefault(n, 0)));
            }
            res.add(row);
        }
        
        return res;
    }
}
