class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        vector<string> header;
        header.push_back("Table");
        
        set<string> s;
        map<int, unordered_map<string, int>> table;
        for (auto &o : orders) {
            s.insert(o[2]);
            table[stoi(o[1])][o[2]]++;
        }
        
        for (auto &x : s) {
            header.push_back(x);
        }
        
        vector<vector<string>> ret;
        ret.emplace_back(header);
        
        for (auto &kv : table) {
            vector<string> tmp;
            tmp.push_back(to_string(kv.first));
            for (int i = 1; i < header.size(); ++i) {
                tmp.push_back(to_string(kv.second[header[i]]));
            }
            ret.emplace_back(tmp);
        }
        
        return ret;
    }
};
