class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {

        map<string, map<int,int> > m;
        set<int> s;
        for(auto k : orders) {
            s.insert(stoi(k[1]));
            m[k[2]][stoi(k[1])]+=1;
        }
        vector<vector<string> > ans;
        vector<string> v;
        v.push_back("Table");
        for(auto &k : m){ v.push_back(k.first);}
        ans.push_back(v);
        
        for(auto k : s){
            vector<string> p;
            p.push_back(to_string(k));
            for(auto &kk : m){
                p.push_back(to_string(kk.second[k]));
            }
            ans.push_back(p);
        }
        return ans;
    }
};
