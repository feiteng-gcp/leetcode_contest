class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> dishes;
        set<int> tables;
        unordered_map<string, unordered_map<string, int>> table_dish_map;
        for (auto order : orders) {
            dishes.insert(order[2]);
            tables.insert(stoi(order[1]));
            ++table_dish_map[order[1]][order[2]];
        }
        vector<vector<string>> res;
        vector<string> row;
        row.push_back("Table");
        for (auto dish : dishes) {
            row.push_back(dish);
        }
        res.push_back(row);
        
        for (auto table : tables) {
            vector<string> cur;
            cur.push_back(to_string(table));
            for (auto dish : dishes) {
                cur.push_back(to_string(table_dish_map[to_string(table)][dish]));
            }
            res.push_back(cur);
        }
        return res;
    }
};
