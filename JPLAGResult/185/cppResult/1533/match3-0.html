class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> items;
        map<int, unordered_map<string, int>> tables;
        
        for(auto order:orders){
            items.insert(order[2]);
            tables[atoi(order[1].c_str())][order[2]]++;
        }
        
        vector<vector<string>> ans;
        vector<string> row;
        row.push_back("Table");
        for(auto y:items){
            row.push_back(y);
        }
        ans.push_back(row);
        for(auto x:tables) {
            vector<string> row;
            row.push_back(to_string(x.first));
            for(auto y:items) {
                row.push_back(to_string(tables[x.first][y]));
            }
            ans.push_back(row);
        }
        return ans;
    }
};
