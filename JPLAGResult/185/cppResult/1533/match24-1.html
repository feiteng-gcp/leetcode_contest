class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<int, map<string, int>> mp;
        set<string> food;
        for (auto o: orders) {
            mp[stoi(o[1])][o[2]]++;
            food.insert(o[2]);
        }
        vector<vector<string>> an;
        an.push_back({"Table"});
        for (auto &f: food) an.back().push_back(f);
        for (auto &kv: mp) {
            an.push_back({});
            an.back().push_back(to_string(kv.first));
            for (auto &f: food) {
                an.back().push_back(to_string(kv.second[f]));
            }
        }
        
        return an;
    }
};
