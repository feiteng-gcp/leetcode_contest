class Solution {
    #define pb push_back
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        map<int,map<string,int>> tab;
        set<string> food;
        for(auto v:orders)
        {
            food.insert(v[2]);
            tab[stoi(v[1])][v[2]]++;
        }
        vector<vector<string>> ans(1);
        ans[0].pb("Table");
        for(auto s:food)
            ans[0].pb(s);
        for(auto v:tab)
        {
            vector<string> cur;
            cur.push_back(to_string(v.first));
            for(auto s:food)
                cur.push_back(to_string(v.second[s]));
            ans.pb(cur);
        }
        return ans;
    }
};
