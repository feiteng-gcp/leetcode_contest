class Solution {
public:
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        set<string> vegnam;
        map<int,unordered_map<string,int>> tab;
        for(auto &i:orders){
            vegnam.insert(i[2]);
            ++tab[stoi(i[1])][i[2]];
        }
        vector<vector<string>> res;
        res.reserve(100);
        res.push_back({"Table"});
        for(auto &i:vegnam)res.back().push_back(i);
        for(auto &i:tab){
            res.push_back({to_string(i.first)});
            for(auto &j:vegnam)res.back().push_back(to_string(i.second[j]));
        }
        return res;
    }
};
