class Solution {
public:
    int getNum(string s) {
        int i, x = 0;
        for (i = 0; i < s.size(); i++) {
            x = x * 10 + s[i] - '0';
        }
        return x;
    }
    string int2String(int x) {
        if (!x)
            return "0";
        string s = "";
        while (x > 0) {
            s.push_back(x % 10 + '0');
            x /= 10;
        }
        string s2 = "";
        for (int i = s.size() - 1; i >= 0; i--)
            s2.push_back(s[i]);
        return s2;
    }
    vector<vector<string>> displayTable(vector<vector<string>>& orders) {
        int i, n = orders.size();
        map<int, map<string, int> >mp;
        set<string>S;
        for (i = 0; i < n; i++) {
            int tbn = getNum(orders[i][1]);
            if (!mp.count(tbn))
                mp[tbn] = map<string, int>();
            mp[tbn][orders[i][2]]++;
            S.insert(orders[i][2]);
        }
        vector<vector<string>>res;
        res.push_back(vector<string>());
        res[0].push_back("Table");
        for(auto it: S) {
            res[0].push_back(it);
        }
        for (auto it: mp) {
            vector<string>cur;
            cur.push_back(int2String(it.first));
            for (auto it2: S) {
                cur.push_back(int2String(mp[it.first][it2]));
            }
            res.push_back(cur);
        }
        return res;
    }
};
