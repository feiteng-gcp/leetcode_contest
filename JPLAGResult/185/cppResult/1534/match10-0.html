class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        vector<int> dp(5);
        int count = 0;
        int ans = 0;
        for (auto c : croakOfFrogs) {
            if (c == 'c') {
                count++;
                dp[0]++;
            } else if (c == 'r') {
                if (dp[0] == 0) return -1;
                dp[0]--;
                dp[1]++;
            } else if (c == 'o') {
                if (dp[1] == 0) return -1;
                dp[1]--;
                dp[2]++;
            } else if (c == 'a') {
                if (dp[2] == 0) return -1;
                dp[2]--;
                dp[3]++;
            } else {
                if (dp[3] == 0) return -1;
                dp[3]--;
                dp[4]++;
                ans = max(ans, count);
                count--;
            }
        }
        for (int i=0; i<4; i++) {
            if (dp[i] > 0) return -1;
        }
        return ans;
    }
};
