class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        map<string,int> store;
        if(croakOfFrogs.length()%5!=0){
            return -1;
        }
        store["c"]=0;
        store["cr"]=0;
        store["cro"]=0;
        store["croa"]=0;
        int pos = 0;
        int res = 1;
        int t = 0;
        t=max(t,res);
        while(pos<croakOfFrogs.length()){
            if(croakOfFrogs[pos]=='c'){
                if(store["c"]==0 && store["cr"]==0 && store["cro"]==0 && store["croa"]==0){
                    
                }else{
                    res++;
                    t=max(t,res);
                }

                store["c"]+=1;
            }else if(croakOfFrogs[pos]=='r'){
                if(store["c"]==0){
                    return -1;
                }
                store["c"]-=1;
                store["cr"]+=1;
            }else if(croakOfFrogs[pos]=='o'){
                if(store["cr"]==0){
                    return -1;
                }
                store["cr"]-=1;
                store["cro"]+=1;
            }else if(croakOfFrogs[pos]=='a'){
                if(store["cro"]==0){
                    return -1;
                }
                store["cro"]-=1;
                store["croa"]+=1;
            }else{
                if(store["croa"]==0){
                    return -1;
                }
                store["croa"]-=1;
                res--;
            }
            pos++;
        }
        if(store["c"]==0 && store["cr"]==0 && store["cro"]==0&&
          store["croa"]==0){
            return t;
        }
        return -1;
    }
};
