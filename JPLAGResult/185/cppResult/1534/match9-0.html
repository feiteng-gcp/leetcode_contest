class Solution {
public:
    int c[10];
    int minNumberOfFrogs(string s) {
        int n = s.length();
        int mx = 0;
        for (int i = 0; i < n; ++i) {
            if (s[i] == 'c') ++c[0];
            else if (s[i] == 'r') {
                if (c[0] == 0) return -1;
                --c[0]; ++c[1];
            } else if (s[i] == 'o') {
                if (c[1] == 0) return -1;
                --c[1]; ++c[2];
            } else if (s[i] == 'a') {
                if (c[2] == 0) return -1;
                --c[2]; ++c[3];
            } else if (s[i] == 'k') {
                if (c[3] == 0) return -1;
                --c[3]; ++c[4];
            }
            int t = 0;
            for (int j = 0; j < 4; ++j) t += c[j];
            mx = max(mx, t);
        }
        if (c[4] * 5 != n) return -1;
        return mx;
    }
};
