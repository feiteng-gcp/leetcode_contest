class Solution {
public:
  int minNumberOfFrogs(string croakOfFrogs) {
    int a = 0, b = 0;
    int c[128] = {0};
    for (const char ch : croakOfFrogs) {
      if (ch == 'c') {
        if (a) {
          --a;
        }
        ++b;
        ++c['c'];
      } else if (ch == 'r') {
        if (c['c']) {
          ++c[ch];
          --c['c'];
        } else {
          return -1;
        }
      } else if (ch == 'o') {
        if (c['r']) {
          ++c[ch];
          --c['r'];
        } else {
          return -1;
        }
      } else if (ch == 'a') {
        if (c['o']) {
          ++c[ch];
          --c['o'];
        } else {
          return -1;
        }
      } else if (ch == 'k') {
        if (c['a']) {
          ++a;
          --b;
          ++c[ch];
          --c['a'];
        } else {
          return -1;
        }
      } else {
        return -1;
      }
    }
    if (c['c'] || c['r'] || c['o'] || c['a']) return -1;
    return a;
  }
};
