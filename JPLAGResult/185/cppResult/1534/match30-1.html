class Solution {
public:
  int minNumberOfFrogs(string s) {
    int c[4],ans=0;
    mem(c);

    rep(i,s.size()) {
      if(s[i]=='c') {
        c[0]++;
      } else if(s[i]=='r') {
        if(!c[0]) return -1;
        c[0]--,c[1]++;
      } else if(s[i]=='o') {
        if(!c[1]) return -1;
        c[1]--,c[2]++;
      } else if(s[i]=='a') {
        if(!c[2]) return -1;
        c[2]--,c[3]++;
      } else {
        if(!c[3]) return -1;
        c[3]--;
      }
      ans=max(ans,c[0]+c[1]+c[2]+c[3]);
    }
    if(c[0]||c[1]||c[2]||c[3]) return -1;
    return ans;
  }
};
