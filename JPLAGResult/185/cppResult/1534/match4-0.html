class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int n = croakOfFrogs.length();
        string &s = croakOfFrogs;
        int c = 0, r = 0, o = 0, a = 0, k = 0, ans = 0;;
        for(int i = 0; i < n; i++) {
            if(s[i] == 'c') {
                c++;
            }
            else if(s[i] == 'r') {
                if(!c) return -1;
                c--; r++;
            }
            else if(s[i] == 'o'){
                if(!r) return -1;
                r--; o++;
            }
            else if(s[i] == 'a') {
                if(!o) return -1;
                o--; a++;
            }
            else if(s[i] == 'k') {
                if(!a) return -1;
                a--; k++;
            }
            else {
                return -1;
            }
            ans = max(ans, c+r+o+a);
        }
        if(c || r || o || a) return -1;
        return ans;
    }
};
