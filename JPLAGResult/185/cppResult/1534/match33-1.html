class Solution {
public:
    int minNumberOfFrogs(string s) {
        int ans = 0;
        if (s.size()%5 != 0) {
            return -1;
        }
        int c = 0, r = 0, o = 0, a = 0, k = 0;
        int mini = 0;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == 'c') {

                c++;
            } else if (s[i] == 'r') {
                if (c > r) {
                    r++;
                } else {
                    return -1;
                }
            } else if (s[i] == 'o') {
                if (r > o) {
                    o++;
                } else {
                    return -1;
                }
            } else if (s[i] == 'a') {
                if (o > a) {
                    a++;
                } else {
                    return -1;
                }
            } else if (s[i] == 'k') {
                if (a > k) {
                    a--, c--, r--, o--;
                } else {
                    return -1;
                }
            } else {
                return -1;
            }
            mini = max(mini, c);
        }
        return c == 0 && r == 0 && o == 0 && a == 0 ? mini : -1;
    }
};
