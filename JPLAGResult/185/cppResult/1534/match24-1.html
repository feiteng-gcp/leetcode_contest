class Solution {
public:
    int minNumberOfFrogs(string s) {
        int cur = 0;
        int ans = 0;
        vector<int> cnt(4, 0);
        for (int i = 0; i < s.size(); i++)

            if (s[i] == 'c') {
                cur++;
                ans = max(ans, cur);
                cnt[0]++;
            }
            else if (s[i] == 'r') {
                if (cnt[0] == 0) return -1;
                cnt[0]--;
                cnt[1]++;
            }
            else if (s[i] == 'o') {
                if (cnt[1] == 0) return -1;
                cnt[1]--;
                cnt[2]++;
            }
            else if (s[i] == 'a') {
                if (cnt[2] == 0) return -1;
                cnt[2]--;
                cnt[3]++;
            }
            else if (s[i] == 'k') {
                if (cnt[3] == 0) return -1;
                cnt[3]--;
                cur--;
            }
        for (int i = 0; i < 4; i++)
            if (cnt[i] != 0) return -1;
        return ans;
    }
};
