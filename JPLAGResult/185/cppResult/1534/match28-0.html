class Solution {
public:
    int minNumberOfFrogs(string s) {
        int bc=0,br=0,bo=0,ba=0,bk=0;
        int maxx=0;
        for(char c:s){
            if(c=='c') ++bc;
            if(c=='r'){
                if(bc==0) return -1;
                --bc;
                ++br;
            }
            if(c=='o'){
                if(br==0) return -1;
                --br;
                ++bo;
            }
            if(c=='a'){
                if(bo==0) return -1;
                --bo;
                ++ba;
            }
            if(c=='k'){
                if(ba==0) return -1;
                --ba;
                maxx=max(maxx,ba+bo+br+bc+1);
            }
        }
        if(bc||br||bo||ba) return -1;
        return maxx;
    }
};
