class Solution {
public:
    int minNumberOfFrogs(string s) {
        int n = s.length();
        if(n%5 != 0) {
            return -1;
        }
        
        //c -> counts[1], a->counts[4]
        vector<int> counts(5);
        int ans = 0;
        int frogs = 0;
        for(int i = 0; i < n; i++) {
            if(s[i] == 'c') {
                counts[1]++;
                frogs++;
            }
            else if(s[i] == 'r') {
                if(counts[1] == 0) {
                    return -1;
                }
                counts[2]++;
                counts[1]--;
            }
            else if(s[i] == 'o') {
                if(counts[2] == 0) {
                    return -1;
                }
                counts[3]++;
                counts[2]--;
            }
            else if(s[i] == 'a') {
                if(counts[3] == 0) {
                    return -1;
                }
                counts[4]++;
                counts[3]--;
            }
            else {
                if(counts[4] == 0) {
                    return -1;
                }
                counts[4]--;
                frogs--;
            }
            
            ans = max(ans, frogs);
        }
        return ans;
    }
};
