class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int a=0,b=0,c=0,d=0,ans=0;
        for (int i=0;i<croakOfFrogs.size();i++)

        {
            if (croakOfFrogs[i]=='c')
                a++;
            if (croakOfFrogs[i]=='r')
            {
                if (a==0)
                    return -1;
                a--;
                b++;
            }
            if (croakOfFrogs[i]=='o')
            {
                if (b==0)
                    return -1;
                b--;
                c++;
            }
            if (croakOfFrogs[i]=='a')
            {
                if (c==0)
                    return -1;
                c--;
                d++;
            }
            if (croakOfFrogs[i]=='k')
            {
                if (d==0)
                    return -1;
                d--;
            }
            ans=max(ans,a+b+c+d);
        }
        if (a+b+c+d>0)
            return -1;
        return ans;
    }
};
