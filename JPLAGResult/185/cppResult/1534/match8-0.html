class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int c=0,r=0,o=0,a=0,cur=0,res=0;
        for (char tmp:croakOfFrogs)
        {
            if (tmp=='c')
                c++;
            else if (tmp=='r')
            {
                if (c==0)
                    return -1;
                r++;
                c--;
            }
            else if (tmp=='o')
            {
                if (r==0)
                    return -1;
                o++;
                r--;
            }
            else if (tmp=='a')
            {
                if (o==0)
                    return -1;
                a++;
                o--;
            }
            else if (tmp=='k')
            {
                if (a==0)
                    return -1;
                res++;
                a--;
            }
            cur=max(cur,c+o+a+r);
        }
        if (c!=0 || r!=0 || o!=0 || a!=0)
            return -1;
        return min(cur,res);
    }
};
