class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        map<char, int> MAP;
        for (auto it : croakOfFrogs) {
            MAP[it]++;
        }
        int n = MAP['c'];
        if (n != MAP['r'])
            return -1;
        if (n != MAP['o'])
            return -1;
        if (n != MAP['a'])
            return -1;
        if (n != MAP['k'])
            return -1;
        MAP.clear();
        int ans = 0, temp = 0;
        for (auto it : croakOfFrogs) {
            if (it == 'c') {
                MAP['c']++;
                temp ++;
                ans = max(ans, temp);
            } else if (it == 'r') {
                if (MAP['c'] == 0)
                    return -1;
                MAP['c']--;
                MAP['r']++;
            } else if (it == 'o') {
                if (MAP['r'] == 0)
                    return -1;
                MAP['r']--;
                MAP['o']++;
            } else if (it == 'a') {
                if (MAP['o'] == 0)
                    return -1;
                MAP['o']--;
                MAP['a']++;
            } else if (it == 'k') {
                if (MAP['a'] == 0)
                    return -1;
                MAP['a']--;
                MAP['k']++;
                temp--;
            }
        }
        return ans;
    }
};
