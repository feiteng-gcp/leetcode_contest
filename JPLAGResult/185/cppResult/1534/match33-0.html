class Solution {
public:
    int minNumberOfFrogs(string croakOfFrogs) {
        int ans = 0;
        
        vector<int> cnt(4, 0);
        for (auto c : croakOfFrogs) {
            if (c == 'c') {
                ans = max(ans, ++cnt[0]);
            }
            else if (c == 'r') {
                if (cnt[1] != cnt[0]) {
                    ++cnt[1];
                }
                else {
                    return -1;
                }
            }
            else if (c == 'o') {
                if (cnt[2] != cnt[1]) {
                    ++cnt[2];
                }
                else {
                    return -1;
                }
            }
            else if (c == 'a') {
                if (cnt[3] != cnt[2]) {
                    ++cnt[3];
                }
                else {
                    return -1;
                }
            }
            else {
                --cnt[0];
                --cnt[1];
                --cnt[2];
                --cnt[3];
            }
        }
        
        return cnt[0] == 0 && cnt[1] == 0 && cnt[2] == 0 && cnt[3] == 0 ? ans : -1;
    }
};
