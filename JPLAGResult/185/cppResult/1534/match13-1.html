class Solution {
public:
    int minNumberOfFrogs(string c) {
        vector<int> v(5,0);
        int ans = 0;

        for(int i = 0; i< c.size();i++){
            if(c[i] == 'c'){
                v[0]++;
            }else if(c[i] == 'r'){
                if(v[0] == 0){
                    return -1;
                }
                v[0]--;
                v[1]++;
            }else if(c[i] == 'o'){
                if(v[1] == 0){
                    return -1;
                }
                v[1]--;
                v[2]++;
            }else if(c[i] == 'a'){
                if(v[2] == 0){
                    return -1;
                }
                v[2]--;
                v[3]++;
            }else if(c[i] == 'k'){
                if(v[3] == 0){
                    return -1;
                }
                v[3]--;
                v[4]++;
            }else{
                return -1;
            }
            ans = max(ans,v[0]+v[1]+v[2]+v[3]);
        }
        return v[0]==0 && v[1]==0 && v[2]==0 && v[3]==0?ans:-1;
    }
};
