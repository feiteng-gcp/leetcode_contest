class Solution {
public:
    int minNumberOfFrogs(string cof) {
        if (cof.length() % 5) return -1;
        vector<int> croak(5);
        //  c r o a k
        int ans(0);
        for (const char& c: cof) {
            if (c == 'c') {

                croak[0]++;
                //if (croak[1] || )
                ans = max(ans, croak[0] + croak[1] + croak[2] + croak[3]);
            } else if (c == 'r') {
                if (croak[0]) {
                    croak[0]--;
                    croak[1]++;
                    //ans = max(ans, croak[1]);
                } else {
                    return -1;
                }
            } else if (c == 'o') {
                if (croak[1]) {
                    croak[1]--;
                    croak[2]++;
                    //ans = max(ans, croak[2]);
                } else {
                    return -1;
                }
            } else if (c == 'a') {
                if (croak[2]) {
                    croak[2]--;
                    croak[3]++;
                    //ans = max(ans, croak[3]);
                } else {
                    return -1;
                }
            } else if (c == 'k') {
                if (croak[3]) {
                    croak[3]--;
                    croak[4]++;
                    //ans = max(ans, croak[4]);
                } else {
                    return -1;
                }
            }
        }
        for (int i = 0; i < 4; ++i) if (croak[i]) return -1;
        return ans;
    }
};
