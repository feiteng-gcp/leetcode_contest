class Solution {
public:
    const int mod = 1e9+7;
    int recur(int n, int m, int k, int mx, vector<vector<vector<int>>>& dp) {
        if(n <= 0) return (k == 0)? 1: 0;
        if(k < 0 || k > n) return 0;
        if(dp[n][mx][k] != -1) return dp[n][mx][k];
        long long res = 0;
        for(int i = 1; i <= m; i++) {
            if(mx < i) res += recur(n-1, m, k-1, i, dp);
            else res += recur(n-1, m, k, mx, dp);
            res %= mod;
        }
        dp[n][mx][k] = res;
        return res;
        
    }
    int numOfArrays(int n, int m, int k) {
        if(k > m || k == 0) return 0;
        vector<vector<vector<int>>> dp(n+1, vector<vector<int>> (m+1, vector<int>(k+1, -1)));
        return recur(n, m, k, 0, dp);
        
    }
};
