class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        long long dp[n + 1][m + 1][k + 1];
        memset(dp, 0, sizeof(dp));
        

        for (int mm = 1; mm <= m; ++mm) {
            dp[1][mm][1] = 1;
        }
        
        for (int nn = 2; nn <= n; ++nn) {
            for (int mm = 1; mm <= m; ++mm) {
                for (int kk = 1; kk <= k; ++kk) {

                    dp[nn][mm][kk] = dp[nn - 1][mm][kk] * mm;
                    // cout << "nn: " << nn << ", mm: " << mm << ", kk: " << kk << ": " << dp[nn][mm][kk] << endl;
                    for (int mmm = 1; mmm < mm; ++mmm) {
                        dp[nn][mm][kk] += dp[nn - 1][mmm][kk - 1];
                    }
                    dp[nn][mm][kk] = dp[nn][mm][kk] % 1000000007;
                    // cout << "nn: " << nn << ", mm: " << mm << ", kk: " << kk << ": " << dp[nn][mm][kk] << endl;
                }
            }
        }
        
        long long res = 0;
        for (int mm = 1; mm <= m; ++mm) {
            res += dp[n][mm][k];
            res = res % 1000000007;
        }
        return res % 1000000007;
    }
};
