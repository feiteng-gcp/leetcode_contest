class Solution {
private:
    static constexpr int mod = 1000000007;
    int f[51][101][51];
    
public:
    int numOfArrays(int n, int m, int k) {
        if (!k) {
            return 0;
        }
        memset(f, 0, sizeof(f));
        for (int i = 1; i <= m; ++i) {
            f[1][i][1] = 1;
        }
        for (int i = 2; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                for (int k = 1; k <= j && k <= n; ++k) {
                    // 前 i 个数 最大为 j 阶梯为 k
                    f[i][j][k] = (long long)f[i - 1][j][k] * j % mod;
                    for (int j0 = 1; j0 < j; ++j0) {
                        f[i][j][k] += f[i - 1][j0][k - 1];
                        f[i][j][k] %= mod;
                    }
                }
            }
        }
        int ans = 0;
        for (int i = k; i <= m; ++i) {
            ans += f[n][i][k];
            ans %= mod;
        }
        return ans;
    }
};
