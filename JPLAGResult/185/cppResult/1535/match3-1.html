class Solution {
public:
    const int MOD = 1e9+7;

    int numOfArrays(int n, int m, int k) {
       // if(k == 0) return 0;
        //dp[i][j][k] 表示i结尾的最大是j的cost为k的合法数量
        //dp[i][j][k] = dp[i - 1][j][k] * j + dp[i - 1][j - 1][k - 1] + dp[i - 1][j - 2][k - 1]
        vector<vector<vector<long long> > > dp(n + 5, vector<vector<long long>>(m + 5, vector<long long>(k + 5, 0)));
        for(int i = 1; i <= m; i++) dp[0][i][1] = 1;

        for(int i = 1; i < n; i++){
            for(int j = 1; j <= m; j++){
                for(int kk = 1; kk <= k; kk++){
            //        cout<<i<<' '<<j<<' '<<kk<<endl;
                    dp[i][j][kk] = (dp[i - 1][j][kk]*j)%MOD;
                    for(int jj = j - 1; jj >= 1; jj--)
                        dp[i][j][kk] = (dp[i][j][kk] + dp[i - 1][jj][kk - 1]) %MOD;

                }
            }
        }
        long long ans = 0;
        for(int i = 1; i <= m; i++) ans = (ans + dp[n - 1][i][k])%MOD;
        return ans;
    }
};
