class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        const int mod=1e9+7;
        //dp[i][j][k]: number of constructions from 0,...,i where max is j and cost is k
        long long *** dp=new long long**[n];
        for(int i=0;i<n;i++){
            dp[i]=new long long*[m+1];
            for(int j=0;j<m+1;j++){
                dp[i][j]=new long long[n+1];
                for(int t=0;t<k+1;t++){
                    dp[i][j][t]=0;
                }
            }
        }
        for(int j=1;j<=m;j++) dp[0][j][1]=1;
        for(int i=1;i<n;i++){
            for(int j=1;j<=m;j++){
                for(int oldj=1;oldj<j;oldj++){
                    for(int t=1;t<k;t++){
                        dp[i][j][t+1]+=dp[i-1][oldj][t];
                        dp[i][j][t+1]%=mod;
                    }
                }
                for(int oldj=j;oldj<=m;oldj++){
                    for(int t=1;t<=k;t++){
                        dp[i][oldj][t]+=dp[i-1][oldj][t];
                        dp[i][oldj][t]%=mod;
                    }
                }
            }
        }
        
        long long ans=0;
        for(int j=1;j<=m;j++){
            ans+=dp[n-1][j][k];
            ans%=mod;
        }
        return ans;
    }
};
