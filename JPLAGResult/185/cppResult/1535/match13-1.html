class Solution {
public:
    static constexpr int maxn = 1e2+10;
    static constexpr int mod = 1e9+7;
    int dp[maxn][maxn][maxn];

    int numOfArrays(int n, int m, int k) {
        memset(dp, 0, sizeof(dp));
        dp[0][0][0] = 1;
        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= m; j++) {
                for(int k = 0; k <= m; k++) {
                    for(int cost = 0; cost <= k; cost++) {
                        dp[i][max(j, k)][cost+(j>k)] = (dp[i][max(j, k)][cost+(j>k)]+dp[i-1][k][cost])%mod;

                    }
                }
            }
        }
        int ans = 0;
        for(int i = 1; i <= m; i++) {
            ans += dp[n][i][k];
            ans %= mod;
        }
        return ans;
    }
};
