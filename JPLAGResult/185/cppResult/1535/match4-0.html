const int MAX_INF = 1000000007;
class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        vector<vector<vector<long long>>> f(n + 1, vector<vector<long long>>(m + 1, vector<long long>(k + 1, 0)));
        f[0][0][0] = 1;
        for (int i = 1; i <= n; ++i)
        {
            for (int j = 1; j <= m; ++j)
            {
                for (int z = 1; z <= k; ++z)
                {
                    f[i][j][z] = f[i - 1][j][z] * j % MAX_INF;
                    for (int x = 0; x <= j - 1; ++x)
                    {
                        f[i][j][z] = (f[i][j][z] + f[i - 1][x][z - 1]) % MAX_INF;
                    }
                    
                    // f[i][j][z] = (f[i][j - 1][z] + f[i - 1][j - 1][z - 1] + ((f[i - 1][j][z] - f[i - 1][j - 1][z] + MAX_INF)%MAX_INF) * j) % MAX_INF;
                }
            }
        }
        int ans = 0;
        for (int i = 0; i <= m; ++i)
        {
            ans = (ans + f[n][i][k]) % MAX_INF;
        }
        return ans;
        // return f[n][m][k];
    }
};
