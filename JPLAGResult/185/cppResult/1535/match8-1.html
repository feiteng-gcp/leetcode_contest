class Solution {
public:
    const int MOD = 1e9 + 7;
    int dp[55][110][55];
    
    int solve(int n, int m, int k, int mx = 0) {
        if (n <= 0) return k == 0;
        if (k < 0) return 0;
        

        if (k > n) return 0;

        if (dp[n][mx][k] != -1) return dp[n][mx][k];
        
        long long ret = 0;
        for (int i = 1; i <= m; ++i) {
            if (mx < i) ret += solve(n - 1, m, k - 1, max(mx, i));
            else ret += solve(n - 1, m, k, mx);
            ret %= MOD;
        }
        
        dp[n][mx][k] = ret;
        
        return ret;
    }
    
    int numOfArrays(int n, int m, int k) {
        memset(dp, -1, sizeof(dp));
        return solve(n, m, k);
    }
};
