#define LL long long 
const LL MOD = 1e9 + 7;

LL dp[55][55][150];

class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        
        memset(dp, 0, sizeof(dp));
        dp[0][0][0] = 1;
        
        for (int i = 1; i <= n; i++){
            for (int p = 0; p <= k; p++){
                for (int l = 0; l <= m; l++){
                    if (dp[i-1][p][l] == -1) continue;
                    for (int v = 1; v <= m; v++){
                        int np = p, nv = max(v, l); if (v > l) np++;
                        dp[i][np][nv] = (dp[i][np][nv] + dp[i-1][p][l]) % MOD;
                    }
                }
            }
        }
        
        LL ans = 0;
        for (int v = 1; v <= m; v++){
            ans = (ans + dp[n][k][v]) % MOD; 
        }
        return ans;
    }
};
