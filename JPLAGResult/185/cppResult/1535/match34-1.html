class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        const int M = 1000000007;
        long long dp[51][101][101];
        memset(dp, 0, sizeof(dp));
        for(int i = 1; i <= m; ++i){
            dp[1][1][i] = 1;
        }

        for(int i = 1; i <= k; ++i){
            for(int j = 1; j <= n; ++j){
                for(int l = 1; l <= m; ++l){
                    for(int t = 1; t < l; ++t){
                        dp[i][j][l] += dp[i - 1][j - 1][t];
                        dp[i][j][l] %= M;
                    }
                    dp[i][j][l] += dp[i][j - 1][l] * l;
                    dp[i][j][l] %= M;
                }
            }
        }

        int rtn = 0;
        for(int i = 1; i <= m; ++i){
            rtn += dp[k][n][i];
            rtn %= M;
        }
        return rtn;
    }
};
