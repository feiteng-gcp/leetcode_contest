typedef long long ll;
ll const mod = 1000000007;
ll f[55][101][55];

class Solution {
public:
    void add(ll &x, int y) {
        x += y;
        if (x >= mod) x-=mod;
    }
    int numOfArrays(int n, int m, int k) {
        for (int i = 0; i <= n; ++i)
            for (int j = 0; j <= m; ++j)
                for (int w = 0; w <= k + 1; ++w) {
                    f[i][j][w] = 0;
                }
        f[0][0][0] = 1;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j <= m; ++j) {
                for (int w = 0; w <= k; ++w) {
                    if (f[i][j][w] != 0) {
                        for (int t = 1; t <= m; ++t) {
                            if (t <= j) add(f[i + 1][j][w], f[i][j][w]);
                            else add(f[i + 1][t][w + 1], f[i][j][w]);
                        }
                    }
                }
            }
        }
        ll ans = 0;
        for (int i = 0; i <= m; ++i) add(ans, f[n][i][k]);
        return ans;
    }
};
