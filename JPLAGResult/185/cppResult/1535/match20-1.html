typedef long long ll;

const ll mod = 1e9 + 7;

class Solution {
public:
    ll dp[101][51][101];
    int numOfArrays(int n, int m, int k) {
        memset(dp,0,sizeof(dp));
        
        // for (int x=0; x<=m; x++) {
        //     dp[0][0][x] = 1;
        // }

        dp[0][0][0] = 1;
        
        for (int i=1; i<=n; i++) {
            for (int j=1; j<=k; j++) {
                for (int x=1; x<=m; x++) {
                    dp[i][j][x] = 1LL*x*dp[i-1][j][x]%mod;//saw x previously, we are 1..x
                    for (int p=0; p<x; p++) {

                        dp[i][j][x] += dp[i-1][j-1][p];//we are x, incr the cost
                        dp[i][j][x] %= mod;
                    }
                    
                 
                    // cout<<i<<" "<<j<<" "<<x<<": "<<dp[i][j][x]<<endl;
                }
                
            }
        }
        
        ll ans = 0;
        for (int x=1; x<=m; x++) {
            ans += dp[n][k][x];
            // cout<<x<<": "<<dp[n][k][x]<<endl;
            ans %= mod;
        }
        
        ans %= mod;
        ans += mod;
        ans %= mod;
        return ans;
            
    }
};
