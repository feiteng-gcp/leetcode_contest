class Solution {
    const long long MOD = 1000000007;
    long long dp[100][110][100];
public:
    int numOfArrays(int n, int m, int k) {
      if(k == 0)return 0;
      // clean

      for(int i =0;i<=n;i++){
        for(int j=0;j<=m;j++){
          for(int kk =0;kk<=k;kk++){
            dp[i][j][kk] = 0;
          }
        }
      }
        
        

      for(int mm = 1;mm<=m;mm++){
        dp[0][mm][1] = 1;
      }
      for(int i =1;i<n;i++){
        for(int kk = 1;kk <= min(k,i+1);kk++){
          for(int mm = 1;mm<=m;mm++){
            (dp[i][mm][kk] = dp[i-1][mm][kk] * mm)%=MOD; // 最大不变
            for(int mmm = 1;mmm < mm;mmm++){
              (dp[i][mm][kk] += dp[i-1][mmm][kk-1])%=MOD; // 最大变
            }

          }
        }
      }
      long long ans = 0;
      for(int mm=1;mm<=m;mm++){
          (ans+=dp[n-1][mm][k])%=MOD;
      }
      return ans;
    }

};
