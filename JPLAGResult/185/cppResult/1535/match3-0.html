class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        vector<vector<vector<long long> > > dp(n, vector<vector<long long> >(m+1, vector<long long>(k+1, 0)));
        for(int i = 1; i<=m; i++) {
            dp[0][i][1] = 1;
        }
        long long M = 1e9+7;
        for(int i = 1; i<n; i++) {
            for(int j = 1; j<=m; j++) {
                for(int t = 1; t<=k; t++) {
                    dp[i][j][t] = (dp[i-1][j][t]*j)%M;
                    for(int u = 1; u<j; u++){
                        dp[i][j][t] = (dp[i][j][t] + dp[i-1][u][t-1])%M;
                    }
                }
            }
        }
        long long ans = 0;
        for(int i = 1; i<=m; i++)ans = (ans + dp[n-1][i][k])%M;
        return ans;
    }
};
