class Solution {
public:
    long long dp[55][105][55];
    const int MOD = 1e9 + 7;
    int numOfArrays(int n, int m, int k) {
        if (k == 0) return 0;
        for (int i = 1; i <= m; i++){
            dp[1][i][1] = 1;
        }
        for (int i = 1; i <= n; i++){
            for (int j = 1; j <= m; j++){
                for (int t = 1; t <= k; t++){
                    dp[i][j][t] += dp[i-1][j][t]*j ;
                    dp[i][j][t] %= MOD;
                    for (int q = 1; q < j; q++){
                        dp[i][j][t] += dp[i-1][q][t-1];
                        dp[i][j][t] %= MOD;
                    }
                }
            }
        }
        long long  res = 0;
        for (int i = 1; i <= m; i++){
            res += dp[n][i][k];
            res %= MOD;
            // cout <<i <<' '<< dp[n][i][k] <<' '<< endl;
        }
        return res;
    }
};
