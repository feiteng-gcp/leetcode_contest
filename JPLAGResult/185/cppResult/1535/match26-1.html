class Solution {
#define MOD 1000000007
public:
	long long dp[50][51][101];

	int numOfArrays(int n, int m, int K) {
		if (K == 0)return 0;
		
		for (int i = 0; i < 50; i++) {
			for (int j = 0; j <= 50; j++) {
				for (int k = 0; k <= 100; k++) {
					dp[i][j][k] = 0;
				}
			}
		}
		for (int i = 1; i <= m; i++) {
			dp[0][1][i] = 1;
		}
		vector<vector<long long>>p(K+1, vector<long long>(m + 1, 0));
		for (int i = 0; i <= K; i++) {
			for (int j = 1; j <= m; j++) {
				p[i][j] = p[i][j - 1] + dp[0][i][j];
			}
		}

		for (int i = 1; i < n; i++) {
			for (int j = 0; j <= K; j++) {
				for (int k = 1; k <= m; k++) {
					dp[i][j][k] = (dp[i][j][k] + dp[i - 1][j][k] * k) % MOD;
					if (j != K) {
						dp[i][j + 1][k] = (dp[i][j + 1][k] + p[j][k - 1]) % MOD;
					}
				}
			}
			for (int j = 0; j <= K; j++) {
				for (int k = 1; k <= m; k++) {
					p[j][k] = p[j][k - 1] + dp[i][j][k];

				}
			}
		}
		long long ret = 0;
		for (int i = 1; i <= m; i++) {
			ret += dp[n-1][K][i];
			ret %= MOD;
		}
		return ret;
	}
};
