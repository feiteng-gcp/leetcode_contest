class Solution {
public:
    const int M = 1000000007;
    int f[51][101][51];

    int numOfArrays(int n, int m, int k) { 
        memset(f, 0, sizeof f);
        for (int j = 1; j <= m; ++j) {
            f[1][j][1] = 1;
        }

        for (int i = 2; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                for (int k = 1; k <= i; ++k) {
                    // = j
                    for (int l = 1; l < j; ++l) {
                        f[i][j][k] = (f[i][j][k] + f[i - 1][l][k - 1]) % M;    
                    }
                    f[i][j][k] = (f[i][j][k] + f[i - 1][j][k]) % M;
                    // < j
                    f[i][j][k] = (f[i][j][k] + 1ll * f[i - 1][j][k] * (j - 1) % M) % M;
                    

                }
            }
        }
        int ans = 0;
        for (int i = 1; i <= m; ++i) {
            ans = (ans + f[n][i][k]) % M;
        } 
        return ans;
    }
};
