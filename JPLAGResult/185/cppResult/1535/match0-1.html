class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        if (k == 0) {
            return 0;
        }
        constexpr long long kPrime = 1e9 + 7;
        vector<vector<vector<int>>> dp(n + 1, vector<vector<int>>(k + 1, vector<int>(m + 1)));

        for (int v = 1; v <= m; ++v) {
            dp[1][1][v] = 1;
        }
        for (int i = 2; i <= n; ++i) {
            for (int j = 1; j <= k && j <= i; ++j) {
                for (int v = j; v <= m; ++v) {
                    dp[i][j][v] = static_cast<int>(1LL * dp[i - 1][j][v] * v % kPrime);

                    for (int p = j - 1; p < v; ++p) {
                        dp[i][j][v] = (dp[i][j][v] + dp[i - 1][j - 1][p]) % kPrime;
                    }
                }
            }
            
        }
        int result = 0;
        for (int v = 1; v <= m; ++v) {
            result = (result + dp[n][k][v]) % kPrime;
        }
        return result;
    }
};
