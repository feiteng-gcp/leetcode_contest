class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        const int mod = 1e9 + 7;
        vector<vector<vector<int> > >dp(n + 1, vector<vector<int> >(m + 1, vector<int>(k + 1)));
        dp[0][0][0] = 1;
        for(int i = 1; i <= n; i++)
        {
            for(int j = 1; j <= m; j++)
            {
                for(int pk = 1; pk <= k; pk++)
                {
                    for(int prej = 0; prej < j; prej++)
                    dp[i][j][pk] += dp[i - 1][prej][pk - 1], dp[i][j][pk] %= mod;
                    for(int prej = 1; prej <= j; prej++)
                    dp[i][j][pk] += dp[i - 1][j][pk], dp[i][j][pk] %= mod;

                }
            }
        }
        int ans = 0;
        for(int i = 0; i <= m ;i++)
            ans += dp[n][i][k], ans %= mod;
        return ans;
    }
};
