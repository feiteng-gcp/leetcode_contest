class Solution {
  int dp[51][101][51];
public:

  int numOfArrays(int n, int M, int k) {
    memset(dp, 0, sizeof(dp));
    for (int x = 0; x < M; ++x) {
      dp[0][x][0] = 1;
    }
    int mod = 1e9 + 7;
    

    for (int i = 1; i < n; ++i) {
      for (int c = 0; c <= k; ++c) {
        for (int x = 0; x < M; ++x) {       
          for (int x2 = 0; x2 < M; ++x2) {
            dp[i][x][c] = (dp[i][x][c] + (x > x2 ? (c ? dp[i-1][x2][c-1]:0) : dp[i-1][x][c])) % mod;

          }
        }
      }
    }

    int ans = 0;
    for (int x = k-1; x < M; ++x) {
      ans = (ans + dp[n-1][x][k-1]) % mod;
    }
    return ans;
  }
};
