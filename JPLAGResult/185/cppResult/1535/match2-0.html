class Solution {
public:
    int f[105][105][105];
    int numOfArrays(int n, int m, int k) {
        int mo = 1000000007;
        for (int i = 0; i <= 100; i++) {
            for (int j = 0; j <= 100; j++) {
                for (int l = 0; l <= 100; l++) {
                    f[i][j][l] = 0;
                }
            }
        }
        f[0][0][0] = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j <= m; j++) {
                for (int l = 0; l <= k; l++) {
                    for (int nxt = 1; nxt <= m; nxt++) {
                        if (nxt <= j) {
                            (f[i + 1][j][l] += f[i][j][l]) %= mo;
                        } else {
                            (f[i + 1][nxt][l + 1] += f[i][j][l]) %= mo;
                        }
                    }
                }
            }
        }
        int ans = 0;
        for (int j = 0; j <= m; j++) {
            (ans += f[n][j][k]) %= mo;
        }
        printf("\n");
        return ans;
    }
};
