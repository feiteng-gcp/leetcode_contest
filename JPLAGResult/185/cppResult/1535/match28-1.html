class Solution {
public:
    const long long MOD = 1e9 + 7;
    long long dp[51][51][101];
    int numOfArrays(int n, int m, int k) {

        memset(dp, 0, sizeof(dp));
        for (int s = 1; s <= m; ++ s) dp[1][1][s] = 1;

        for (int i = 2; i <= n; ++ i) {
            for (int j = 1; j <= i && j <= k; ++ j) {
                long long prevSum = 0;
                for (int s = 1; s <= m; ++ s) {
                    dp[i][j][s] = (dp[i][j][s] + dp[i - 1][j][s] * s) % MOD;
                    if (j > 1) {
                        dp[i][j][s] = (dp[i][j][s] + prevSum) % MOD;
                        prevSum = (prevSum + dp[i - 1][j - 1][s]) % MOD;
                    }

                }
            }
        }
        long long ans = 0;
        for (int s = 1; s <= m; ++ s) ans += dp[n][k][s];
        return ans % MOD;
    }
};
