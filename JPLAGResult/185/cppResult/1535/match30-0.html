class Solution {
    vector<vector<vector<int>>> dp;
    int mod = 1e9 + 7;

    int mpow(int a, int n){
        int res = 1;
        while(n--) res = res*a%mod;
        return res;
    }

    int res(int n, int m, int k){
        if (dp[n][m][k] != -1) return dp[n][m][k];
        int ans = 0;
        for (int start = 1; start <= m; ++start){
            for (int nextid = 1; nextid < n; ++nextid){
                ans += mpow(start, nextid-1) * res(n-start, 0,0);
            }
        }
        return 0;
    }
public:
    int numOfArrays(int n, int m, int k) {
        dp = vector<vector<vector<int>>>(n+1, vector<vector<int>>(m+1, vector<int>(k+1, 0)));

        for (int cur =1; cur<=m; ++cur){
            dp[1][cur][1] = 1;
        }

        for (int len = 1; len<=n-1; ++len){
            for (int cnt =1; cnt<=k; ++cnt){
                for (int curm = 1; curm<=m; ++curm){
                    int d = dp[len][curm][cnt];
                    for (int next = 1; next <=m; ++next){
                        if (next<=curm){
                            dp[len+1][curm][cnt] += d;
                            dp[len+1][curm][cnt] = dp[len+1][curm][cnt]%mod;
                        }
                        else{
                            if (cnt + 1 <=k)
                            {
                                dp[len+1][next][cnt+1] += d;
                                dp[len+1][next][cnt+1] = dp[len+1][next][cnt+1]%mod;
                            }
                        }
                    }

                }
            }
        }

        int res = 0;

        for (int ma=1; ma<=m; ++ma) {
            res += dp[n][ma][k];
            res = res%mod;
        }
        return res;


    }
};
