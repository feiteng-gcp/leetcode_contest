class Solution {
public:
  int numOfArrays(int n, int m, int k) {
    const int M = 1000000007;
    vector<vector<vector<int>>> a(n + 1, vector<vector<int>>(m + 2, vector<int>(k + 1)));
    a[0][0][0] = 1;
    for (int i = 1; i <= n; ++i) {
      for (int j = 0; j <= m; ++j) {
        for (int x = 0; x <= k; ++x) {
          for (int y = 1; y <= m; ++y) {
            int j1 = max(j, y);
            int x1 = (j < y) ? (x + 1) : x;
            if (x1 <= k) (a[i][j1][x1] += a[i - 1][j][x]) %= M;
          }
        }
      }
    }
    int ret = 0;
    for (int i = 1; i <= m; ++i) (ret += a[n][i][k]) %= M;
    return ret;
  }
};
