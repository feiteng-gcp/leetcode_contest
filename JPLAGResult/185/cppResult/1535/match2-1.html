const int mod=1e9+7;
int dp[52][102][52];
class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        //dp[i][max][cost]
        memset(dp, 0, sizeof(dp));

        dp[0][0][0]=1;
        for(int i=0; i<n; i++) {
            for(int nxt=1; nxt<=m; nxt++) {
                for(int cur=0; cur<=m; cur++) {
                    for(int cost=0; cost<=k; cost++) {
                        if (nxt>cur) {
                            dp[i+1][nxt][cost+1]+=dp[i][cur][cost];
                            dp[i+1][nxt][cost+1]%=mod;
                        }
                        else {
                            dp[i+1][cur][cost]+=dp[i][cur][cost];
                            dp[i+1][cur][cost]%=mod;
                        }
                        //printf("dp[%d][%d][%d]=%d\n", i, cur, cost, dp[i][cur][cost]);
                    }
                }
            }
        } 
        int ans=0;
        for(int i=1; i<=m; i++) {
            //cout<<dp[n][i][k]<<' ';
            ans+=dp[n][i][k];
            ans%=mod;
        }
        //cout<<endl;
        return ans;
    }
};
