#define LL long long
const LL mod = 1e9 + 7;

int dp[100][200][100];
class Solution {
public:
    int numOfArrays(int n, int m, int k) {

      for (int i = 0; i <= n; i++) {
        for (int j = 0; j <= m; j++) {
          for (int l = 0; l <= k; l++) {
            dp[i][j][l] = 0;
          }
        }
      }
      dp[0][0][0] = 1;

      for (int i = 1; i <= n; i++) {
        for (int j = 0; j <= m; j++) {
          for (int l = 1; l <= m; l++) {
            for (int r = 0; r <= k; r++) {
              if (l <= j) {
                dp[i][j][r] += dp[i - 1][j][r];
                dp[i][j][r] %= mod;
              }
              else {
                dp[i][l][r + 1] += dp[i - 1][j][r];
                dp[i][l][r + 1] %= mod;
              }
            }
          }
        }
      }

      LL ans = 0;
      for (int i = 1; i <= m; i++) {
        ans += dp[n][i][k];
        ans %= mod;
      }
      return ans;
    }
};
