class Solution {
public:
    int numOfArrays(int n, int m, int k) {
        long long  mem[51][51][101];
        int Mod =1e9+7;
        memset(mem,0,sizeof(mem));
        for(int j=1;j<=m;j++)
            mem[0][1][j]=1;
        for(int i=1;i<n;i++){
            for(int kk=1;kk<=k;kk++){
                long long sum=0;
                for(int j=1;j<=m;j++){
                    sum+=mem[i-1][kk-1][j-1];
                    mem[i][kk][j]= (mem[i-1][kk][j]*(long long)(j)+sum)%Mod;
                    
                    // mem[i][kk][j]+=sum;
                    
                    // cout<<mem[i][kk][j]<<endl;
                }
            }
            
        }
        long long res=0;
        for(int j=1;j<=m;j++)
            res+=mem[n-1][k][j];
        return res%Mod;
    }
};
