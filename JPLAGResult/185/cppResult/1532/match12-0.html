class Solution {
public:
    string reformat(string s) {
        string t1, t2;
        for(char c : s) {
            if(isdigit(c)) t1 += c;
            else t2 += c;
        }
        if(abs((int)t1.size() - (int)t2.size()) > 1) return "";
        if(t1.size() > t2.size()) swap(t1, t2);
        string res;
        for(int i=0,j=0;i<t1.size() || j < t2.size();) {
            if(j < t2.size()) res.push_back(t2[j++]);
            if(i < t1.size()) res.push_back(t1[i++]);
        }
        return res;
    }
};
