class Solution {
public:
  string reformat(string s) {
    string s1, s2;
    for (auto ch: s) {
      if (isdigit(ch)) s1.push_back(ch); 
      else s2.push_back(ch);
    }
    if (s1.size() < s2.size()) swap(s1, s2);
    if (s1.size() - s2.size() > 1) return "";
    string t;
    for (int i = 0; i < s1.size(); i++) {
      t.push_back(s1[i]);
      if (i < s2.size()) t.push_back(s2[i]);
    }
    return t;
  }
  
};
