class Solution {
public:
    string reformat(string s) {
        string a="",b="";
        for (int i=0;i<s.size();i++)
            if (s[i]>='0'&&s[i]<='9')
                a+=s[i];
            else
                b+=s[i];
        if (abs(a.size()-b.size())>1)
            return "";
        if (a.size()<b.size())
        {
            string c=a;
            a=b;
            b=c;
        }
        string ans;
        for (int i=0;i<b.size();i++)
        {
            ans+=a[i];
            ans+=b[i];
        }
        if (a.size()>b.size())
            ans+=a[a.size()-1];
        return ans;
    }
    int abs(int x)
    {
        if (x>=0)
            return x;
        return -x;
    }
};
