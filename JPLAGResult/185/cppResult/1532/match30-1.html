class Solution {
public:
    string reformat(string s) {
        string alphas, digits;
        for(int i = 0; i < s.size(); i++) {
            if(isalpha(s[i])) alphas.push_back(s[i]);
            else digits.push_back(s[i]);
        }
        int na = alphas.size();
        int nd = digits.size();
        if(abs(na - nd) > 1) return "";
        string ans;
        if(na >= nd) {
            bool a = true;

            int j = 0, k = 0;
            for(int i = 0; i < s.size(); i++) {
                if(a) {
                    ans.push_back(alphas[j++]);
                } else {
                    ans.push_back(digits[k++]);
                }
                a ^= 1;
            }
        } else {
            bool d = true;

            int j = 0, k = 0;
            for(int i = 0; i < s.size(); i++) {
                if(d) {
                    ans.push_back(digits[j++]);
                } else {
                    ans.push_back(alphas[k++]);
                }
                d ^= 1;
            }
        }
        return ans;
    }
};
