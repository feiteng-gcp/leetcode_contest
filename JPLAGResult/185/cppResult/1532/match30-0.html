class Solution {
public:
    string reformat(string s) {
        string l = "", n = "";
        for (int i = 0; i < s.length(); i++) {
            if (s[i] >= 'a' && s[i] <= 'z') {
                l += s[i];
            } else {
                n += s[i];
            }
        }
        
        if (l.length() < n.length()) {
            swap(l, n);
        }
        
        if (l.length() - n.length() > 1) {
            return "";
        }
        
        string res = "";
        for (int i = 0; i < s.length(); i++) {
            if (i % 2) {
                res += n[i / 2];
            } else {
                res += l[i / 2];
            }                                     
        }
        
        return res;
    }
};
