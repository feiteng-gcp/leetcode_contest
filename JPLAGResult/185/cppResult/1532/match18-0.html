class Solution {
public:
    string reformat(string s) {
        int n = s.length(), a = 0, b = 0;
        string x = "", y = "";
        for(int i=0;i<n;i++)
        {
            if(s[i] <= '9' and s[i] >='0')
            {
                x+= s[i];
                a++;
            }
            else
            {
                y += s[i];
                b++;
            }
        }
        if(abs(a-b) <= 1)
        {
            string final = "";
            int l = 0, r = 0;
            if(a > b)
            {
                final +=x[l];
                l++;
                while(l < a)
                {
                    final += y[r];
                    final += x[l];
                    
                    l++;
                    r++;
                }
            }
            else if(b > a)
            {
                final += y[r];
                r++;
                while(r < b)
                {
                    final += x[l];
                    final += y[r];
                    l++;
                    r++;
                }
            }
            else
            {
                while(l < a)
                {
                    final += x[l];
                    final += y[r];
                    l++;
                    r++;
                }
            }
            return final;
        }
        else return "";
        
    }
};
