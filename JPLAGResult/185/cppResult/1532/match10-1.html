class Solution {
public:
    vector<char> num, let;
    string reformat(string s) {
        string ans="";
        for (char c:s) {
            if ('0'<=c && c<='9')
                num.push_back(c);
            else
                let.push_back(c);
        }
        if (abs((int)num.size()-(int)let.size())>1)
            return "";

        if (num.size()>let.size()) {
            ans+=num.back(); num.pop_back();
            for (int i=0; i<let.size(); ++i) {
                ans += let[i];
                ans += num[i];
            }
        }
        else if (num.size()<let.size()) {
            ans+=let.back(); let.pop_back();
            for (int i=0; i<num.size(); ++i) {
                ans += num[i];
                ans += let[i];
            }
        }
        else {
            for (int i=0; i<num.size(); ++i) {
                ans += num[i];
                ans += let[i];
            }
        }
        return ans;
    }
};
