class Solution {
public:
    string reformat(string s) {
        vector<int> v1,v2;

        for(char c: s) {
            if (c>='0'&&c<='9') {
                v1.push_back(c);
            } else {
                v2.push_back(c);
            }
        }
        int n = v1.size();
        int m = v2.size();
        if (abs(n-m)>1) return "";
        if (v1.size() < v2.size()) swap(v1,v2);
        string ans;
        for (int i=0; i<s.length(); i++) {
            if (i%2==0) {
                ans += v1[i/2];
            } else {
                ans += v2[i/2];
            }
        }
        return ans;
    }
};
