#define LL long long


class Solution {
public:
    string reformat(string s) {
      string a, b;
      for (char ch : s) {
        if (ch <= '9' && ch >= '0') a += ch;
        else b += ch;
      }
      if (abs((int)a.size() - (int)b.size()) > 1) return "";

      if (a.size() < b.size()) swap(a, b);

      string ans;
      for (int i = 0; i < a.size(); i++) {
        ans += a[i];
        if (i < b.size()) {
          ans += b[i];
        }
      }
      return ans;
    }
};
