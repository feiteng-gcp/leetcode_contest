class Solution {
public:
    string reformat(string s) {
        string a = "", b = "";
        for (int i = 0; i < s.size(); i++)
            if (isalpha(s[i])) a += s[i];
            else b += s[i];
        if (abs((int) (a.size() - b.size())) > 1) return "";
        if (a.size() < b.size()) swap(a, b);
        string ans = "";

        for (int i = 0; i < b.size(); i++) { ans += a[i]; ans += b[i]; }
        if (a.size() > b.size()) ans += a[a.size() - 1];
        return ans;
    }
};
