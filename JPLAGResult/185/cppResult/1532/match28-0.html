class Solution {
public:
    string reformat(string s) {
        vector<char> a[2];
        for(int i=0;i<s.length();++i)
        {
            if(isdigit(s[i]))a[0].push_back(s[i]);
            else a[1].push_back(s[i]);
        }
        if(abs((int)a[0].size()-(int)a[1].size())>1)return "";
        int now=a[0].size()>a[1].size()?0:1;
        string ans;
        for(int i=0;i<s.length();++i)
        {
            ans.push_back(a[now][a[now].size()-1]);
            a[now].pop_back();
            now=1-now;
        }
        return ans;
    }
};
