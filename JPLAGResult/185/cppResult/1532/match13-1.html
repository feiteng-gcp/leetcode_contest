class Solution {
public:
  string reformat(string s) {
    string x,y;
    for(char c:s){
      if(isdigit(c)) x+=c;
      else y+=c;
    }
    if(abs((int)x.size()-(int)y.size())>1) return "";
    if(x.size()<y.size()) swap(x,y);
    string res;
    for(int i=0;i<(int)x.size()+10;i++){
      if(i<(int)x.size()) res+=x[i];
      if(i<(int)y.size()) res+=y[i];
    }
    return res;
  }
};
