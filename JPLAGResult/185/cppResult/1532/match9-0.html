class Solution {
public:
    string reformat(string s) {
        string letters, numbers;
        for (auto& c : s) {
            if (c >= '0' && c <= '9') {
                numbers += c;
            } else {
                letters += c;
            }
        }
        if (abs((int)letters.size() - (int)numbers.size()) > 1) return "";
        string ret;
        if (letters.size() > numbers.size()) {
            ret += letters[0];
            for (int i = 0; i < numbers.size(); ++i) {
                ret += numbers[i];
                ret += letters[i + 1];
            }
        } else if (letters.size() < numbers.size()) {
            ret += numbers[0];
            for (int i = 0; i < letters.size(); ++i) {
                ret += letters[i];
                ret += numbers[i + 1];
            }
        } else {
            for (int i = 0; i < letters.size(); ++i) {
                ret += letters[i];
                ret += numbers[i];
            }
        }
        return ret;
    }
};
