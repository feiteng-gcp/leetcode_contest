class Solution {
public:
    string reformat(string s) {
        string num;
        string chr;
        for (char c: s) {
            if (c >= '0' && c <= '9') num.push_back(c);
            else chr.push_back(c);
        }
        string res;

        if (num.size()-chr.size()==1) {
            for (int i = 0; i < chr.size(); i++) {
                res += num[i];
                res += chr[i];
            }
            res += num.back();
        } else if (chr.size()-num.size() ==1) {
             for (int i = 0; i < num.size(); i++) {
                res += chr[i];
                res += num[i];
            }
            res += chr.back();
        } else if (chr.size()==num.size()) {
            for (int i = 0; i < chr.size(); i++) {
                res += chr[i];
                res += num[i];
            }
        }
        return res;
    }
};
