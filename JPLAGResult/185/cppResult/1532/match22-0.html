class Solution {
public:
    string reformat(string s) {
        vector<char> A, B;
        for (char c: s) {
            if (c >= '0' && c <= '9') {
                A.push_back(c);
            } else {
                B.push_back(c);
            }
        }
        int n = A.size();
        int m = B.size();
        if (abs(n - m) > 1) return "";
        if (n < m) swap(A, B), swap(n, m);
        string ans;
        for (int i = 0; i < n; i++) {
            ans.push_back(A[i]);
            if (i < m) ans.push_back(B[i]);
        }
        return ans;
    }
};
