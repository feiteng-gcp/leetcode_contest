class Solution {
public:
    string reformat(string s) {
        string alp, dig, res;
        for (char &c : s) {
            if (c >= '0' && c <= '9') dig += c;
            else alp += c;
        }
        if (abs((int)alp.size()-(int)dig.size()) > 1) return "";
        if (alp.size() < dig.size()) swap(alp, dig);
        for (int i = 0; i < alp.size(); ++i) {
            res += alp[i];
            if (i < dig.size()) res += dig[i];
        }
        return res;
    }
};
