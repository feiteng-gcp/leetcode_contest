class Solution {
public:
    string reformat(string s) {
        int d = 0, a = 0;
        vector<char> dig, alp;

        for (int i = 0; i < s.size(); i++) {
            if (isdigit(s[i])) {
                d++;
                dig.push_back(s[i]);
            } else {
                a++;                
                alp.push_back(s[i]);                
            }
        }
        if (abs(a-d) > 1) {
            return "";
        }
        if (a > d) {
            swap(d, a);
            swap(dig, alp);
        } 
        string t = "";

        for (int i = 0; i < a; i++) {
            t += dig[i];
            t += alp[i];
        }
        if (a != d) {
          t += dig[d-1];
        }
        return t;
    }
};
