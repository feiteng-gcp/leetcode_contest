class Solution {
public:
    string reformat(string s) {
        string ch = "", num = "";
        for(int i = 0; i < s.size(); i++)
        {
            if(s[i] >= 'a' && s[i] <= 'z')
                ch.push_back(s[i]);
            else num.push_back(s[i]);
        }
        string ans = "";
        if(abs((int)num.size() - (int)ch.size()) <= 1)
        {
            if(num.size() > ch.size())
            {
                ans += num.back();
                num.pop_back();
                for(int i = 0; i < num.size(); i++)
                {
                    ans += ch[i];
                    ans += num[i];
                }
            }
            else if(num.size() < ch.size())
            {
                ans += ch.back();
                ch.pop_back();
                for(int i = 0; i < num.size(); i++)
                {
                    ans += num[i];
                    ans += ch[i];
                }
            }
            else
            {
                for(int i = 0; i < num.size(); i++)
                {
                    ans += ch[i];
                    ans += num[i];
                }
            }
            return ans;
        }
        return "";
    }
};
