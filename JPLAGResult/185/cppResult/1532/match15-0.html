class Solution {
public:
    string reformat(string s) {
        string chars, ds;
        for (auto& ch : s) {
            if (isalpha(ch)) {
                chars.push_back(ch);
            } else {
                ds.push_back(ch);
            }
        }
        if (chars.size() == ds.size()) {
            string res;
            for (int i = 0; i < chars.size(); ++i) {
                res.push_back(chars[i]);
                res.push_back(ds[i]);
            }
            return res;
        } else if (chars.size() + 1 == ds.size()) {
            string res;
            res.push_back(ds.back());
            for (int i = 0; i < chars.size(); ++i) {
                res.push_back(chars[i]);
                res.push_back(ds[i]);
            }
            return res;
        } else if (chars.size() == ds.size() + 1) {
            string res;
            res.push_back(chars.back());
            for (int i = 0; i < ds.size(); ++i) {
                res.push_back(ds[i]);
                res.push_back(chars[i]);
            }
            return res;
        }
        return "";
    }
};
