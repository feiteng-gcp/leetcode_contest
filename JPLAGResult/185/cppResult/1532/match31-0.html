class Solution {
public:
    string reformat(string s) {
        vector<char> v1,v2;
        for(auto x:s)
        {
            if(isdigit(x))
            {
                v1.push_back(x);
            }
            else
                v2.push_back(x);
        }
        string ans="";
        int n=v1.size(),m=v2.size(),i,j,k;
        if(abs(n-m)>1)
            return ans;
        if(n>m)
            k=0;
        else
            k=1;
        i=0;j=0;
        while(n||m)
        {
            if(k&1)
            {
                ans.push_back(v2[j++]);
                m--;
            }
            else
            {
                ans.push_back(v1[i++]);
                n--;
            }
            k++;
            
        }
        return ans;
            
    }
};
