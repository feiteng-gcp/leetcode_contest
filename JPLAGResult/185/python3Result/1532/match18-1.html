class Solution:
    def reformat(self, s: str) -> str:
        alpha = []
        digit = []
        for c in s:
            if 'a' <= c <= 'z':
                alpha.append(c)
            else:
                digit.append(c)
        
        if abs(len(alpha) - len(digit)) > 1:
            return ''
        
        res = []
        for a, b in zip(alpha, digit):
            res.append(a)
            res.append(
b)
        
        if len(alpha) > len(digit):
            res.append(alpha[-1])
        elif len(alpha) < len(digit):
            res.insert(0, digit[-1])
        
        return ''.join(res)
