class Solution:
	def reformat(self, s: str) -> str:
		A, B = [], []
		for c in s:
			if '0' <= c <= '9':
				B.append(c)
			else:
				A.append(c)

		if len(A) > len(B): A, B = B, A
		if len(B) - len
(A) > 2: return ''
		if len(B) - len(A) == 1: A += ' '
		res = ''
		for i in range(len(A)):
			res += B[i] + A[i]
		return res.strip(' ')
