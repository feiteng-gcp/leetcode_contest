class Solution:
    def reformat(self, s: str) -> str:
        cs, ns = [], []
        for c in s :
            if c in '0123456789' :
                ns.append(c)
            else :
                cs.append(c)
        if abs(len(cs) - len(ns)) > 1 :
            return ''
        if len(cs) > len(ns) :
            cs, ns = ns, cs
        to_ret = []
        for i in range(len(cs)) :
            to_ret.append(ns[i])
            to_ret.append(cs[i])
        if len(ns) > len(cs) :
            to_ret.append(ns[-1])
        return ''.join(to_ret)
