class Solution:
    def reformat(self, s: str) -> str:
        num=[]
        let=[]
        for c in s:
            if ord
(c)>=97:
                num.append(c)
            else:
                let.append(c)
        if abs(len(num)-len(let))>=2:
            return ""
        arr=[]
        for i in range(min(len(let),len(num))):
            arr.append(let[i])
            arr.append(num[i])
        ans
=''.join(arr)
        if len(let)>len(num):
            ans=ans+let[-1]
        elif len(let)<len(num):
            ans=num[-1]+ans
        return ans
