class Solution:
    def reformat(self, s: str) -> str:
        if len(s) <= 1:
            return s
        
        digits 
= []
        letters = []
        for x in s:
            if x.isdigit():
                digits.append(x)
            else:
                letters.append(x)
                
        if abs(len(digits) - len(letters)) >=2 :
            return ''
        
        if len(digits) > len(letters):
            lt1 = digits
            lt2 = letters
        else:
            lt2 = digits
            lt1 
= letters
            
        res = []
        for i in range(len(lt2)):
            res.append(lt1[i])
            res.append(lt2[i])
        if len(lt1) > len(lt2):
            res.append(lt1[-1])
            
        return ''.join(res)
