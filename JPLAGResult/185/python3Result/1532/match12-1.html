class Solution:
    def reformat(self, s: str) -> str:
        char = []
        num 
= []
            
        for ch in s:
            if '0' <= ch <= '9':
                num.append(ch)
            else:
                char.append(ch)
        
        if abs(len(num) - len(char)) > 1:
            return ""
        else:
            n = min(len(char), len(num))
            res = []
            for i in range
(n):
                res.append(char[i])
                res.append(num[i])
            if len(char) > len(num):
                res.append(char[-1])
            elif len(num) > len(char):
                res.insert(0, num[-1])
        
        return ''.join(res)
