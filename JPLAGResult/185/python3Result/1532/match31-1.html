class Solution:
    def reformat(self, s: str) -> str:
        num = []
        alpha = []
        for c in s:
            if c.isnumeric():
                num.append(c)
            else:
                alpha.append(c)
        if abs(len(num)-len(alpha)) > 1:
            return ''
        res = []
        if len(num) > len(alpha):
            num, alpha = alpha, num
        for _ in range(len(num)+len(alpha)):
            if len(num) >= len(alpha):
                res.append(num.pop())
            else:
                res.append(alpha.pop())
        return ''.join(res)
