class Solution:
    def reformat(self, s: str) -> str:
        digit = []
        alpha = []
        for i in range(len(s)):
            if s[i].isdigit():
                digit.append(s[i])
            else:
                alpha.append(s[i])
        
        if abs(len(digit) - len(alpha)) > 1:
            return ""
        
        ans = ''
        if len(digit) >= len(alpha):
            for i in range(len(alpha)):
                ans = ans + digit[i]
                ans = ans + alpha[i]
            if len(digit) > len(alpha):
                ans = ans + digit[-1]
            return ans
        
        if len(digit) <= len(alpha):
            for i in range(len(digit)):
                ans = ans + alpha[i]
                ans = ans + digit[i]
                
            if len(digit) < len(alpha):
                ans = ans + alpha[-1]
        
            return ans
