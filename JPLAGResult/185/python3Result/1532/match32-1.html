class Solution:
    def reformat(self, s: str) -> str:
        l, d = [], []
        for si in s:
            if si.isalpha():
                l.append(si)
            else:
                d.append(si)
        
        if abs(len(l) - len(d)) > 1:
            return ''
        
        r = ''
        if len(l) > len(d):
            for i in range(len(d)):
                r += l[i]
                r += d[i]
            r += l[-1]
        else:
            for i in range(len(l)):
                r += d[i]
                r += l[i]
            if len(d) > len(l):
                r += d[-1]
                
        return r
