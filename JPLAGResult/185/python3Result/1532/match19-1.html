class Solution:
    def reformat(self, s: str) -> str:
        l, d = [], []
        for c in s:
            if c.isalpha():
                l
.append(c)
            else:
                d.append(c)
        if abs(len(l) - len(d)) > 1:
            return ''
        ret = []
        if len(l) > len(d):
            ret.append(l[-1])
        
for i, j in zip(d, l):
            ret.append(i)
            ret.append(j)
        if len(d) > len(l):
            ret.append(d[-1])
        return ''.join(ret)
