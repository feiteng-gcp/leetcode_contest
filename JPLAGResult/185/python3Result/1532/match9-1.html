class Solution:
    def reformat(self, s: str) -> str:
        digits = list('0123456789')
        numbers = []
        chars 
= []
        for ch in s:
            if ch in digits:
                numbers.append(ch)
            else:
                chars.append(ch)
                
        if abs(len(numbers) - len(chars)) <= 1:
            if len(numbers) >= len(chars):
                a, b = numbers, chars
            else:
                a, b 
= chars, numbers
            
            ans = []
            for i in range(len(b)):
                ans.append(a[i])
                ans.append(b[i])
            if len(a) > len(b):
                ans.append(a[-1])
            return ''.join(ans)
        return ''
