class Solution:
    def reformat(self, s: str) -> str:
        ans = ''
        aux1 = ''
        aux2 = ''
        for c in s:
            if c.isdigit():
                aux1 += c
            else:
                aux2 += c
        if abs(len(aux1) - len(aux2)) > 1:
            return ""
        if len(aux1) > len(aux2):
            for i in range(len(aux1)):
                ans += aux1[i]
                if i < len(aux2):
                    ans += aux2[i]
        else:
            for i in range(len(aux2)):
                ans += aux2[i]
                if i < len(aux1):
                    ans += aux1[i]
        return ans
