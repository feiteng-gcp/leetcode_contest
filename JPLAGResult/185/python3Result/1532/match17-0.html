class Solution:
    def reformat(self, s: str) -> str:
        res = []
        digit = []
        chars = []
        
        for c in s:
            if c.isalpha():
                chars.append(c)
            else:
                digit.append(c)
        if abs(len(digit) - len(chars)) > 1:
            return ""
        
        res = [i+j for i,j in zip(digit, chars)]
        res = "".join(res)
        
        if len(digit) > len(chars):
            res += digit[-1]
        elif len(digit) < len(chars):
            res = chars[-1] + res
            
        return res
