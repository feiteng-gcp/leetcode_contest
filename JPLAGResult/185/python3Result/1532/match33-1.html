class Solution:
    def reformat(self, s: str) -> str:
        if len(s) <= 1:
            return s
        
        nums = []
        alphas = []
        for c in s:
            if 0 <= ord
(c) - ord('0') <= 9:
                nums.append(c)
            else:
                alphas.append(c)
                
        if abs(len(nums) - len(alphas)) > 1:
            return ""
        
        result = []
        if len(nums) > len(alphas):
            
for i in range(len(alphas)):
                result.append(nums[i])
                result.append(alphas[i])
            result.append(nums[-1])
        elif len(nums) < len(alphas):
            for i in range(len(nums)):
                result.append(alphas[i])
                result.append(nums[i])
                
            result.append(alphas[-1])
        else:
            for i in range(len(alphas)):
                result.append(alphas[i])
                result.append(nums[i])
        return ''.join(result)
