class Solution:
    def reformat(self, s: str) -> str:
        d = []
        a = []
        for c in s:
            if c.isdigit():
                d.append(c)
            else:
                a.append(c)
        if abs(len(d) - len(a)) > 1:
            return ""
        res = ""
        if len(d) > len(a):
            first = d
            second = a
        else:
            first = a
            second = d
        
        for i, c in enumerate(first):
            res += c
            if i < len(second):
                res += second[i]
        
        return res
