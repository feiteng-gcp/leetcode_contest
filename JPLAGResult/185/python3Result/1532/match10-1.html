class Solution:
    def reformat(self, s: str) -> str:
        digits = []
        chars = []
        for c in s:
            if ord('a') <= ord(c) <= ord('z'):
                chars.append(c)
            if ord('0') <= ord(c) <= ord
('9'):
                digits.append(c)
        # print(len(chars), len(digits))
        if abs(len(chars) - len(digits)) > 1:
            return ''
        if len(chars) >  len(digits):
            chars, digits = digits, chars
        # len(chars) <= len(digits)
        
        ans = []
        for i, ch in enumerate(chars):
            ans.append(digits[i])
            ans.append(
ch)
        if len(digits) > len(chars):
            ans.append(digits[-1])
        return ''.join(ans)
