class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        mod = 10**9+7

        @functools.lru_cache(None)
        def dfs(n1, k1, c):
            if k1 < 0:
                return 0
            if n1 == 0 and k1 == 0:
                return 1
            if n1 == 0 and k1 > 0:
                
return 0

            ans = 0
            for i in range(1, m + 1):
                if i <= c:
                    ans += dfs(n1 - 1, k1, c)
                elif k1:
                    ans += dfs(n1 - 1, k1 - 1, i)
            return ans
        return dfs(n, k, -1) % mod
