from functools import lru_cache
class Solution:
    def numOfArrays(self, N: int, M: int, K: int) -> int:
        MOD = 10 ** 9 + 7
        mkToCount = defaultdict(int)
        for m in range(1, M + 1):
            mkToCount[(m, 1)] = 1
        for i in range(N - 1):
            temp = defaultdict(int)
            for newM in range(1, M + 1):
                for (m, k), c in mkToCount.items():
                    if newM > m:
                        if k + 1 <= K:
                            temp[(newM, k + 1)] += c
                    else:
                        temp[(m, k)] += c
            mkToCount = temp
            
            for (m, k), c in mkToCount.items():
                mkToCount[(m, k)] = c % MOD
            
        return sum(mkToCount[(m, K)] for m in range(1, M + 1)) % MOD
