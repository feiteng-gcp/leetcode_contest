from functools import lru_cache
class Solution:
    def numOfArrays(self, N: int, M: int, K: int) -> int:
        @lru_cache(None)
        def dfs(m,n,k):
            if n == 0:
                if k == 0:
                    return 1
                else:
                    return 0
            if m < 1 or k < 0:
                return 0
            sums = 0
            for x in range(1,m+1):
                sums += dfs(x-1,n-1,k-1)
                sums += dfs(m,n-1,k)-dfs(x-1,n-1,k)
            return sums%(10**9+7)
        return dfs(M,N,K)
