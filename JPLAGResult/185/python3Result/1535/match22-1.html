from functools import lru_cache


class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        mod = 1000000007
        @lru_cache(None)
        def dfs(running_max, remain, remain_k):
            if remain_k < 0:
                return 0
            if remain == 0:
                return remain_k == 0
            if running_max == m and remain_k > 0:
                r
eturn 0

            ans = 0
            for i in range(1, running_max + 1):
                ans += dfs(running_max, remain - 1, remain_k)
            for i in range(running_max + 1, m + 1):
                ans += dfs(i, remain - 1, remain_k - 1)
            return ans % mod
        return dfs(0, n, k)
