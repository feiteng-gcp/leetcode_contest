class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        dp = [[[0 for cost in range(k+1)] for i in range(m+1)] for nn in range(n+1)]
        modu = 10 ** 9 + 7
            
        
for lastmax in range(1, m+1):
            dp[1][lastmax][1] = 1

        for nn in range(2, n+1): # current has nn number
            for lastmax in range(1, m+1):
                for cost in range(1, min
(nn+1, k+1)):
                    dp[nn][lastmax][cost] += dp[nn-1][lastmax][cost] * lastmax
                    for prevmax in range(1, lastmax):
                        dp[nn][lastmax][cost] += dp[nn-1][prevmax][cost-1]
                    dp[nn][lastmax][cost] = dp[nn][lastmax][
cost] % modu
        return sum([dp[n][mm][k] for mm in range(1, m+1)]) % modu
