class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        # num ways to build a length n array with values between 1 and m with k ramps
        if k == 0: return 0
        
        # dp on num ways to build length n-1 array with k more ramps
        # lenght, ramps, max element
        
        from functools import lru_cache
        @lru_cache(None)
        def solve(n, ramps, ma):
            if n == 1:
                if ramps == 1:
                    return (m - ma)
                elif ramps == 0:
                    return ma
                else:
                    return 0
            num = 0
            # make ramp
            for next_ma in range(ma+1, m+1):
                num += solve(n-1, ramps-1, next_ma)
            # don't make ramp
            for _ in range(1, ma+1):
                num += solve(n-1, ramps, ma)
            # print(n, ramps, ma, num)
            return num
            
        return solve(n, k, 0) % (10**9+7)
