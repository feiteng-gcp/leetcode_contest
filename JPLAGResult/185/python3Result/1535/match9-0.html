class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        
        target_cost = k
        module = 1e9 + 7
        
        # f[i][j][k] Step i, Value j, Cost k
        f = [[[0 for k in range(n + 2)] for j in range(m + 2)] for i in range(n + 2)]
        
        for i in range(1, m + 1):
            f[0][i][1] = 1
        
        for i in range(1, n): # step
            for j in range(1, m + 1): # current max value
                for k in range(1, target_cost + 1): # current cost
                    for l in range(1, j): # last max value
                        f[i][j][k] = (f[i][j][k] + f[i - 1][l][k - 1]) % module
                    f[i][j][k] = (f[i][j][k] + f[i - 1][j][k] * j) % module
        
        ans = 0
        for i in range(1, m + 1):
            ans = (ans + f[n - 1][i][target_cost]) % module
            
        return int(ans)
