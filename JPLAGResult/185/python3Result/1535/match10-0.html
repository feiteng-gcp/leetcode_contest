class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        @functools.lru_cache(None)
        def solve(n, mx, cost):            
            if n == 1 and cost == 1:
                return 1
            if n == 1 and cost != 1:
                return 0
                
            res = 0
            for p_mx in range(1, mx):
                res = (res + solve(n-1, p_mx, cost-1)) % MOD
            for cur in range(1, mx+1):
                res = (res + solve(n-1, mx, cost)) % MOD
            return res
        
        MOD = 10**9 + 7
        res = 0
        for i in range(m+1):
            res = (res + solve(n, i, k)) % MOD
            
        return res
