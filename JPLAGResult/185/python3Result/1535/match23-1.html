class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        # dp[i][j][c]: 第i位, sc = j, max(r[:i])=c
        # dp[i][j][c] = sum (dp[i-1][j][d](if c<=d)) + sum(dp[i-1][j-1][d] if c>d)
        
@lru_cache(None)
        def dp(i, j, c):
            if not 1<=c<=m:
                return 0
            if j > k or j > i or j == 0:
                return 0
            if i == 1 and j == 1:
                return 1
            res = 0
            for d in range(1, c):
                res += dp(i-1, j-1, d)
            res += dp(i-1, j, c) * c
            return res % (10**9+7)
        return sum(dp(n, k, c) for c in range(1, m+1)) % (10**9+7)
