from functools import lru_cache

class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        @lru_cache(None)
        def dfs(n, m, k, pos, high_count, high_pre, mod):
            if high_count < 0 : return 0
            if high_count > n - pos + 1: return 0
            
if high_count > m - high_pre: return 0
            if pos >= n + 1: return 1
            
            r1, r2 = 0, 0
            
            for high_new in range(high_pre+1, m + 1):
                r1 = (r1 + dfs(n, m, k, pos+1, high_count-1, high_new, mod)) % mod
            
            r2 = (high_pre * dfs(n, m, k, pos+1, high_count, high_pre, mod)) % mod
            
            #print(pos, high_count, high_pre, (r1+r2) % mod)
            
return (r1 + r2) % mod

        res = 0
        
        for h1 in range(1, m + 1):
            res = (res + dfs(n, m, k, 2, k - 1, h1, mod)) % mod
        
        return res
