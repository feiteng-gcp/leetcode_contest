class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        if k == 0: return 0
        dp = [[[0 for _ in range(m+1)] for __ in range
(k+1)] for ___ in range(n+1)]
        for i in range(1, m+1):
            dp[1][1][i] = 1
        for n_ in range(2, n+1):
            for k_ in range(1, k+1):
                for m_ in range(1, m+1):
                    dp[n_][k_][m_] = dp[n_-1][k_][m_] * m_ + sum(dp[n_-1][k_-1][i] for i in range(m_))
                    dp[n_][k_][m_] %= 10**9 + 7
        return sum(dp[n][k]) % (10**9 + 7)
