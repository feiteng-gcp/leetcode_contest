class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        mod = 10 ** 9 + 7
        if k > m: return 0
        
        from functools import lru_cache as lc
        @lc(None)
        def dfs(n, k, s):
            if k == 0:
                return s ** n
            if n == 0:
                if k == 0:
                    return 1
                else:
                    return 0
            ans = 0
            for i in range(s+1, m+1):
                ans += dfs(n-1, k-1, i)
            
            if s > 0:
                ans += dfs(n-1, k, s) * s
            return ans
        
        return dfs(n, k, 0) % mod
