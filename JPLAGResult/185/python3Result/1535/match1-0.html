class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        
        @lru_cache(None)
        def dfs(n,mv,k): 
            if n == 0 and k == 0: return 1
            if n == 0 or k < 0: return 0               
                
            ans = 0
            for num in range(1,m+1):
                if num > mv: ans += dfs(n-1,num,k-1)
                else: ans += dfs(n-1,mv,k)
            return ans
            
            
        return dfs(n,-1,k) % (10**9+7)
