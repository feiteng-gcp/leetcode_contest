from functools import lru_cache

class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        
        @lru_cache(None)
        def dp(arr_len, lrg_end, inc_times):
            # base case
            if arr_len == 1:
                if inc_times == 1:
                    return 1
                else:
                    return 0
            if inc_times == 0:
                return 0
            
            ans = 0
            for end in range(1, m + 1):
                if end <= lrg_end:
                    ans += dp(arr_len - 1, lrg_end, inc_times)
                else:
                    ans += dp(arr_len - 1, end, inc_times - 1)
                
            return ans % 1000000007
        
        return sum([dp(n, i, k) for i in range(1, m + 1)]) % 1000000007
