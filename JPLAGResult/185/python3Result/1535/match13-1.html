from collections import Counter
class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        mod = 10 ** 9 + 7
        dp = Counter()
        for i in range(1, m + 1):
            dp[(i, 1)] = 1
        for _ in range(n - 1):
            tmp = Counter()
            for i in range(1, m + 1):
                for (max_, k_), v in dp.items():
                    if i > max_:
                        if k_ < k:
                            tmp[(i, k_ + 1)] += v
                    else:
                        tmp[(max_, k_)] += v
            dp = Counter()
            for key, v in tmp.items():
                dp[key] = v % mod
        return sum([v for (max_, k_), v in dp.items() if k_ == k]) % mod
