class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        @lru_cache(None)
        def dp(k, m, n, pre_m):
            if  k<0 :
                return 0
            if n == 0 and k == 0:
                return 1
            elif n == 0 and k>0:
                return 0
            cur_ans = 0
            for i in range(1, m+1):
                if i<= pre_m:
                    cur_ans += dp(k, m, n-1, pre_m) % (10**9+7)
                else:
                    cur_ans += dp(k-1, m, n-1, i) % (10**9+7)
            return cur_ans  % (10**9+7)
        return dp(k, m, n, -1)
