class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        MOD = 10**9 + 7
        def find_num_combos(n, m, k, best, dp):
            if n == 0:
                if k != 0:
                    return 0
                else:
                    return 1 
            
            if (n, k, best) in dp:
                return dp[(n, k, best)]
            
            total = 0
            for i in range(1, m + 1):
                if i <= best:
                    total = (total + find_num_combos(n - 1, m, k, best, dp)) % MOD
                elif k != 0:
                    total = (total + find_num_combos(n - 1, m, k - 1, i, dp)) % MOD
                
            dp[(n, k, best)] = total 
            return total
        
        return find_num_combos(n, m, k, 0, {})
