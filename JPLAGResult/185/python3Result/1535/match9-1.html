class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        mod 
= 10 ** 9 + 7
        f = [[[0 for _ in range(k + 1)] for _ in range(m + 1)] for _ in range(n)]
        for y in range(1, m + 1):
            f[0][y][1] = 1
        #import pdb; pdb.set_trace()
        for x in range(1, n):
            for y in range(1, m + 1):
                for z in range(1, k + 1):
                    f[x][y][z] = (y * f[x - 1][y][z]) % mod
                    
for yy in range(1, y):
                        f[x][y][z] = (f[x][y][z] + f[x - 1][yy][z - 1]) % mod
        return sum([f[n - 1][p][k] for p in range(1, m + 1)]) % mod
