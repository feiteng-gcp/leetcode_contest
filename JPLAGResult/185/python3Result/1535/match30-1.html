class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        mod = 10**9+7
        dp = {}
        def dfs(idx, cost, maxnum):
            if (idx, cost, maxnum) in dp:
                return dp[(idx, cost, maxnum)]
            if cost > k:
                return 0
            
if idx == n:
                if cost == k:
                    return 1
                else:
                    return 0
            if n-idx < k-cost:
                return 0
            ret = 0
            for cand in range(1, m+1):
                if cand <= maxnum:
                    ret += dfs(idx+1, cost, maxnum)%mod
                else:
                    ret += dfs(idx+1, cost+1, cand)%mod
            dp[(idx, cost, maxnum)] = ret%mod
            return ret%mod
        
        return dfs(0, 0, -1)
