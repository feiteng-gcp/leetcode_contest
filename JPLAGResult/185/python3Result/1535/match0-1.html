MOD = 10 ** 9 + 7

class Solution:
    def numOfArrays(self, n: int, m: int, k: int) -> int:
        
        
@lru_cache(None)
        def D(n, l, k):
            if k < 0:
                return 0
            if n == 0:
                return k == 0
            return (sum(D(n - 1, l, k) for i in range(1, l + 1)) + sum(D(n - 1, i, k - 1) for i in range(l + 1, m + 1))) % MOD
            
            
        return D(n, 0, k)
