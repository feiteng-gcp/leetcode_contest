MOD = 10**9 + 7

class Solution:
	def numOfArrays(self, N: int, M: int, K: int) -> int:
		if K <= 0: return 0
		dp 
= [[[0]*(M+1) for i in range(K+1)] for j in range(N+1)]
		# num ways to construct arr of length n having k peaks where max num so far == m

		for m in range(1,M+1):
			dp[1][1][m] = 1


		for n in range(2,N+1):
			for k in range(1,K+1):
				for m in range(1,M+1):
					dp[n][k][m] += dp[n-1][k][m] * m
					dp[n][k][m] %= MOD
					for mm in range(1,m):
						dp[n][k][m] += dp[n-1][k-1][mm]
						dp[n][k][m] %=
 MOD


		res = 0
		for m in range(M+1):
			res += dp[N][K][m]
		return res % MOD
