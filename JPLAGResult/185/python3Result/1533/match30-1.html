class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        names = set()
        tables  = set()
        for o in orders:
            names.add(o[2])
            tables.add(o[1])
        names = list(names)
        names.sort()
        tables = list(tables)
        tables.sort(key = lambda x: int(x))
        mp = {}
        mp2 = {}
        for i in range(len(tables)):
            mp[tables[i]] = i + 1
        for i in range(len(names)):
            mp2[names[i]] = i + 1
            
        res = [None] * (len(tables) + 1)
        res[0] = ['Table'] + names
        
        for i in range(len(tables)):
            res[i+1] = [tables[i]] + [0]* len(names)
        
        for o in orders:
            x = mp[o[1]]
            y = mp2[o[2]]
            res
[x][y] += 1 
        for i in range(1, len(res)):
            for j in range(1, len(res[0])):
                res[i][j] = str(res[i][j])
        return res
