from collections import defaultdict
class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        
        itemsDict = set()
        tableDict = dict()
        
        for order in orders:
            customer, table, item = order
            table = int(table)
            itemsDict.add(item)
            if table not in tableDict:
                tableDict[table] = defaultdict(lambda: 0)
            tableDict[table][item] += 1
        
        headers = ["Table"]
        
        sortedItems = list(sorted(itemsDict))
        headers += sortedItems
        
        ans = []
        ans.append(headers)
        
        for table in sorted(tableDict.keys()):
            row = [str(table)]
            
            for item in sortedItems:
                row.append(str(tableDict[table][item]))
            
            ans.append(row)
        
        return ans
