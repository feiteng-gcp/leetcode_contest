class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        dic = {}
        food = set()
        for c,t,f in orders:
            t = int(t)
            if t not in dic:
                dic[t] = {}
            if f not in dic[t]:
                dic[t][f] = 0
            dic[t][f] += 1
            food.add(f)

        food = list(food)
        food.sort()
        index = sorted(dic)
        re = [[] for i in range(len(dic) + 1)]
        re[0] = ["Table"] + food

        for i in range(1,len(re)):
            re[i] = ["" for i in range(len(re[0]))]
            re[i][0] = str(index[i-1])
            j = 1
            for k in food:
                if k not in dic[index[i-1]]:
                    re[i][j] += "0"
                else:
                    re[i][j] += str(dic[index[i-1]][k])
                j += 1
        return re
