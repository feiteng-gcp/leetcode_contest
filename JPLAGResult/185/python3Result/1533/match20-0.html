class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        table_food_count = {}
        food_set = set()
        for [person,table,food] in orders:
            food_set.add(food)
            
            if table not in table_food_count:
                table_food_count[table] = {}
            if food not in table_food_count[table]:
                table_food_count[table][food] = 0
                
            table_food_count[table][food] += 1
        ans = []
        foods = sorted(list(food_set))
    
        header = ['Table'] + foods
        ans.append(header)
        for table in sorted([x for x in table_food_count], key=lambda x: int(x)):
            row = []
            row.append(table)
            for food in foods:
                if food in table_food_count[table]:
                    row.append(str(table_food_count[table][food]))
                else:
                    row.append("0")
            ans.append(row)
        return ans
