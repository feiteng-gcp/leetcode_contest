class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        d = {}
        fooditem = set()
        tableitem = set()
        for name,table,food in orders:
            if table not in d:
                d[table] = {}
            if food not in d[table]:
                d[table][food] = 1
            else:
                d[table][food] += 1
            fooditem.add(food)
            tableitem.add(int(table))
        food2ix = {}
        table2ix = {}
        ans = [['Table']]
        for i,f in enumerate(sorted(list(fooditem))):
            ans[0].append(f)
            food2ix[f] = i+1
        for i,t in enumerate(sorted(list(tableitem))):
            ans.append([str(t)]+['0' for _ in range(len(fooditem))])
            table2ix[str(t)] = i+1
        # print(ans)
        # print(table2ix)
        # print(food2ix)
        for table in d:
            for food in d[table]:
                ans[table2ix[table]][food2ix[food]] = str(d[table][food])
        return ans
