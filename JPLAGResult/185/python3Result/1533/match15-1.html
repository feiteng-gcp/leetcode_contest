class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        d = {}
        all_fis = set()
        for cn, tn, fi in orders:
            tn = int(tn)
            if tn not in d:
                d[tn] = {}
            if fi not in d[tn]:
                d[tn][fi] = 0
            d[tn][fi] += 1
            all_fis.add(fi)
            
        all_fis = sorted(all_fis)
        ans = [["Table"] + all_fis]
        for tn in sorted(d.keys()):
            row = [str(tn)]
            for fi in all_fis:
                if fi in d[tn]:
                    row += [str(d[tn][fi])]
                else:
                    row += ["0"]
            ans += [row]
        return ans
