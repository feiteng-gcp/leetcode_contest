class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        item = set()
        count = dict()
        for order in orders:
            table = order[1]
            food = order[2]
            if table not in count:
                count[table] = {}
            if food not in item:
                item.add(
food)
            if food not in count[table]:
                count[table][food] = 0
            count[table][food] += 1
        # output
        item =  sorted(item)
        result = []
        for table in count:
            order = [table]
            
for food in item:
                if food in count[table]:
                    order.append(str(count[table][food]))
                else:
                    order.append("0")
            result.append(order)
        result.sort(key=lambda x:int(x[0]))
        result = [["Table"] +item] + result
        return result
