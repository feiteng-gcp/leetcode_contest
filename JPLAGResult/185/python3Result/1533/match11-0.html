class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        dishes = set()
        tableNum = collections.defaultdict(lambda : collections.defaultdict(int))
        for name, t, dish in orders:
            tableNum[int(t)][dish] += 1
            dishes.add(dish)
        
        #print(tableNum)
        dishes = sorted(list(dishes))
        #print(dishes)
        dishMap = {d: i for i, d in enumerate(dishes)}
        #print(dishMap)
        cols = len(dishes)
        headRow = ["Table"] + dishes
        res = [headRow]
        for t in sorted(tableNum.keys()):
            row = ["0" for i in range(cols)]
            row = [str(t)] + row
            for d in tableNum[t]:
                idx = dishMap[d]
                row[idx + 1] = str(tableNum[t][d])
            res.append(row)
        
        return res
