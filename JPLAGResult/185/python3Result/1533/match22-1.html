class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        foods = set()
        d = {}
        for name, table, food in orders:
            table = int(table)
            if food not in foods:
                foods.add(food)
            if table not in d:
                d[table] = {}
            if food not in d[table]:
                d[table][food] = 0
            d[table][food] += 1
        foods = sorted(list(foods))
        
        ret = []
        ret
.append(['Table'] + foods)
        for table in sorted(d.keys()):
            row = [str(table)]
            for f in foods:
                row.append(str(d[table].get(f, 0)))
            ret.append(row)
        return ret
