class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        cols = set()
        table = defaultdict(lambda: defaultdict(int))
        for c, t, f in orders:
            table[int(t)][f] += 1
            cols.add(f)
        
        cols = sorted(cols)
        rows = [["Table"] + cols]
        for t, counts in sorted(table.items()):
            row = [str(t)]
            for c in cols:
                row.append(str(counts[c]))
            rows.append(row)
        return rows
