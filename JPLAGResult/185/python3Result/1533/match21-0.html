class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        dishes = set()        
        for cus, tab, dish in orders:
            dishes.add(dish)
        dishes = sorted(list(dishes))
        num_dish = len(dishes)
        dtoi = {v: i for i, v in enumerate(dishes)}
        tables = collections.defaultdict(lambda: [0]*num_dish)
        for cus, tab, dish in orders:
            tables[int(tab)][dtoi[dish]] += 1
        res = [["Table"] + dishes]
        for table_num in sorted(tables.keys()):
            res.append(list(map(str, [table_num] + tables[table_num])))
        return res
