from collections import defaultdict, Counter
class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        
        foods = set()
        d = defaultdict(Counter)
        for _, t, f in orders:
            foods.add(f)
            d[int(t)][f] += 1
            
        foods = sorted(list(foods))
        d_food = {food: i for i, food in enumerate(foods)}
        res = [['Table'] + foods]
        
        for t in sorted(d):
            cur = [str(t)]
            for f in foods:
                cur.append(str(d[t][f]))
            res.append(cur)
        return res
