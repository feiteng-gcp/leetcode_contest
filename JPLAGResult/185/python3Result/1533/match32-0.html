from collections import defaultdict
class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        dishes = set()
        row = defaultdict(lambda: defaultdict(int))
        for o in orders:
            row[int(o[1])][o[2]] += 1
            dishes.add(o[2])
        ans = [['Table']]
        for i in sorted(row.keys()):
            ans.append([str(i)])
        # print(dishes)
        for d in sorted(dishes):
            i = 1
            ans[0].append(d)
            for t_num in ans[1:]:
                ans[i].append(str(row[int(t_num[0])][d]))
                i += 1
        return ans
