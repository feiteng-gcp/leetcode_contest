class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        # maps str(tableNumber) to another map
        # of the items that table ordered mapped 
        # to the number of times that item was ordered
        tableMap = {}
        ordered 
= set()
        
        for _, t, o in orders:
            if t not in tableMap:
                tableMap[t] = {}
                
            if o not in tableMap[t]:
                tableMap[t][o] = 0
            
            tableMap[t][o] += 1
            ordered.add(o)
            
        ret 
= [["Table"]]
        orderedIndexMap = {}
        for i,el in enumerate(sorted(list(ordered))):
            ret[-1].append(el)
            orderedIndexMap[el] = i
        
        for t in sorted(tableMap.keys(), key = int):
            ret.append([t])
            for _ in range(len(ordered)):
                ret[-1].append("0")
            
            for el in orderedIndexMap:
                if el in tableMap[t]:
                    ret[-1][1 + orderedIndexMap[el]] = str(tableMap[t][el])
                    
        return ret
