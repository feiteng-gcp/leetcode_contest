class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        tables = collections.defaultdict(dict)
        food_set = set()
        
        for customer, table, food in orders:
            table = int(table)
            if food not in food_set:
                food_set.add(food)
            tables[table][food] = tables[table].get(food, 0) + 1
        
        food_arr = sorted(list(food_set))
        
        ans = [['Table']]
        ans[0].extend(food_arr)
        
        for table in sorted(tables.keys()):
            row = [str(table)]
            for food in food_arr:
                row.append(str(tables[table].get(food, 0)))
            ans.append(row)
        
        return ans
