class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        n = len(orders)
        food_list = []
        for i in range(n):
            food_list.append(orders[i][2])
        food_list = list(set(food_list))
        food_list.sort()
        table_list = []
        for i in range(n):
            table_list.append(int(orders[i][1]))
        table_list = list(set(table_list))
        table_list.sort()
        ans = [[]]
        ans[0].append('Table')
        for i in range(len(food_list)):
            ans[0].append(food_list[i])
        for i in range(len(table_list)):
            ans.append([0] * (len(food_list) + 1))
            ans[i + 1][0] = table_list[i]
        for i in range(n):
            x = int(orders[i][1])
            y = orders[i][2]
            pos_y = -1
            for j in range(len(food_list)):
                if y == food_list[j]:
                    pos_y = j
            pos_x = -1
            for j in range(len(table_list)):
                if x == table_list[j]:
                    pos_x = j
            ans
[pos_x + 1][pos_y + 1] += 1
        for i in range(len(ans)):
            for j in range(len(ans[0])):
                ans[i][j] = str(ans[i][j])
        return ans
