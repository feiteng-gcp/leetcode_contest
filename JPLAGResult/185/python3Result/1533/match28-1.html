class Solution:
    
def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        tables = sorted(set(int(t) for _, t, _ in orders))
        foods = sorted(set(food for _, _, food in orders))
        m = {f:i for i, f in enumerate(foods)}
        
        res = collections.OrderedDict([(t, [0] * len(foods)) for t in tables])
        for _, t, f in orders:
            res[int(t)][m[f]] += 1
        return [['Table'] + foods] + [[str(k)] + [*map(str, (v))] for k, v in res.items()]
