from collections import Counter, defaultdict
class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        foodNames = set
()
        tableD = defaultdict(Counter)
        for _, table, food in orders:
            table = int(table)
            tableD[table][food] += 1
            foodNames.add(food)
        foodNames = sorted(foodNames)
        header = ["Table"] + foodNames
        ans = [header]
        tableList = sorted(tableD.keys())
        for tableI in tableList:
            row = []
            row.append(str
(tableI))
            for food in foodNames:
                row.append(str(tableD[tableI][food]))
            ans.append(row)
        return ans
