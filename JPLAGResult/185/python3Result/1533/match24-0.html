class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        table_set = sorted(set([int(t[1]) for t in orders]))
        food_set  = sorted(set([t[2] for t in orders]))
        
        t2i = {t:i for i, t in enumerate(table_set)}
        f2i = {t:i for i, t in enumerate(food_set)}
        to_ret = []
        to_ret.append(['Table']+food_set)
        for t in table_set :
            to_ret.append([str(t)]+[0 for i in range(len(food_set))])
        
        for _, t, f in orders :
            to_ret[t2i[int(t)]+1][f2i[f]+1]+=1
        
        for i in range(1, len(to_ret)) :
            for j in range(1, len(to_ret[0])) :
                to_ret[i][j] = str(to_ret[i][j])
        
        
        return to_ret
