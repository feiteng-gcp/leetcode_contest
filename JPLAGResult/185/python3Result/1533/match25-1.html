class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        df = collections.defaultdict(collections.Counter)
        items = set()
        for name, table, item in orders:
            df[table][item] += 1
            items.add(item)
        
        items = sorted(list(items))
        item_id = {}
        for i, item in enumerate(items):
            item_id[item] = i
        
        result = []
        
        for table in df:
            row = [int(table)] + [0] * len(item_id)
            for item in df[table]:
                row[item_id[item] + 1] = df[table][item]
            result.append(row)
        result.sort()
        
        final_result = [["Table"] + items]
        for row in result:
            final_result.append([str(x) for x in row])
        return final_result
