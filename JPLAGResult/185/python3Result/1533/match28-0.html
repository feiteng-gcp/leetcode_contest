class Solution:
    res = {}
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        ts = sorted(set(int(t) for _, t, _ in orders))
        fs = sorted(set(food for _, _, food in orders))
        m = {f:i for i, f in enumerate(fs)}
        
        self.res = collections.OrderedDict([(t, [0] * len(fs)) for t in ts])
        for _, t, f in orders:
            self.res[int(t)][m[f]] += 1
        return [['Table'] + fs] + [[str(k)] + [str(x) for x in v] for k, v in self.res.items()]
