class Solution:
    def displayTable(self, orders: List[List[str]]) -> List[List[str]]:
        dishes = sorted(list(set([o[2] for o in orders])))
        i2dish = dict(zip(range(len(dishes)), dishes))
        dish2i = dict(zip(dishes, range(len(dishes)) ))

        tables = sorted(list(set([int(o[1]) for o in orders])))
        i2table = dict(zip(range(len(tables)), tables))
        table2i = dict(zip(tables, range(len(tables)) ))


        ordering = [ [str(i2table[t])] + [0] * len(dishes) for t in range(len(tables))]

        for o in orders:
            table = table2i[int(o[1])]
            dish = dish2i[o[2]]
            ordering
[table][dish+1] += 1

        for t in range(len(tables)):
            for d in range(1, 1+len(dishes)):
                ordering[t][d] = str(ordering[t][d])

        return [['Table'] + dishes] + ordering
