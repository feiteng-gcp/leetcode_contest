class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        ans = collections.defaultdict(int)
        for c in croakOfFrogs:
            if c not in "croak":
                return - 1
            if c == 'c':
                if ans['k'] > 0:
                    ans['k'] -= 1
                ans['c'] += 1
            elif c == 'r':
                if ans['c'] > 0:
                    ans['c'] -= 1
                    ans['r'] += 1
                else:
                    return - 1
            elif c == 'o':
                if ans['r'] > 0:
                    ans['r'] -= 1
                    ans['o'] += 1
                else:
                    return - 1
            elif c == 'a':
                if ans['o'] > 0:
                    ans['o'] -= 1
                    ans['a'] += 1
                else:
                    return - 1
            elif c == 'k':
                if ans['a'] > 0:
                    ans['a'] -= 1
                    ans['k'] += 1
                else:
                    return -1
        for c in "croa":
            if ans[c]:
                return - 1
        else:
            return ans['k']
