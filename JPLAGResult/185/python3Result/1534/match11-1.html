from collections import Counter, defaultdict
class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        
        n = len(croakOfFrogs)
        if n % 5 != 0:
            return -1
        
        cnter = Counter(croakOfFrogs)
        if not cnter['c'] == cnter['r'] == cnter['o'] == cnter['a'] == cnter['k'] == n // 5:
            return -1
        
        d, cnt, res = defaultdict(int), 0, 0
        for c in croakOfFrogs:
            if c == 'c':
                cnt += 1
                d
['c'] += 1
            elif c == 'r':
                if d['c'] == 0: return -1
                d['c'] -= 1
                d['r'] += 1
            elif c == 'o':
                if d['r'] == 0: return -1
                d['r'] -= 1
                d['o'] += 1
            elif c == 'a':
                if d['o'] == 0: return -1
                d['o'] -= 1
                d['a'] += 1
            elif c == 'k':
                if d['a'] == 0: return -1
                d['a'] -= 1
                cnt -= 1
            res = max(res, cnt)
        return res
