class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        nums = [0] * 5
        ans = 0
        tmp = 0
        for ch in croakOfFrogs:
            if ch == 'c':
                nums[0] += 1
                tmp += 1 
                ans = max(tmp, ans)
            if ch == 'r':
                if nums[0] == 0:
                    return -1 
                nums[0] -= 1 
                nums[1] 
+= 1 
            if ch == 'o':
                if nums[1] == 0:
                    return -1 
                nums[1] -= 1 
                nums[2] += 1 
                
            if ch == 'a':
                if nums[2] == 0:
                    return -1 
                nums[2] -= 1 
                nums[3] += 1 
                
            if ch == 'k':
                if nums[3] == 0:
                    return -1 
                nums[3] -= 1 
                nums[4] += 1 
                tmp -= 1 
        for i in range(4):
            if nums[i] != 0:
                return -1 
        return ans
