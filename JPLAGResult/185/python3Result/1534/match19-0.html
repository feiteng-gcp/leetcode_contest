class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        ns = {c:0 for c in 'croak'}
        to_ret = 0
        for c in croakOfFrogs :
            if c == 'c' :
                ns['c'] += 1
            elif c == 'r' :
                if ns['c'] == 0 :
                    return -1
                ns['c'] -= 1
                ns['r'] += 1
            elif c == 'o' :
                if ns['r'] == 0 :
                    return -1
                ns['r'] -= 1
                ns['o'] += 1
            elif c == 'a' :
                if ns['o'] == 0 :
                    return -1
                ns['o'] -= 1
                ns['a'] += 1
            elif c == 'k' :
                if ns['a'] == 0 :
                    return -1
                ns['a'] -= 1
                ns['k'] += 1
            else :
                return -1
            to_ret = max(to_ret, sum([ns[c] for c in 'croa']))
        if ns['c']>0 or ns['r']>0 or ns['o']>0 or ns['a']>0 :
            return -1
        return to_ret
