class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        croak=list("croak")
        locs={croak[i]:i for i in range(5)}
        numOnLetter=[0,0,0,0,0]
        res=0
        for i in range(len(croakOfFrogs)):
            if croakOfFrogs[i] not in locs:
                return -1
            loc=locs[croakOfFrogs[i]]
            if loc>0 and numOnLetter[loc-1]==0:
                return -1
            elif loc==0:
                numOnLetter[0]+=1
            else:
                numOnLetter
[loc-1]-=1
                numOnLetter[loc]+=1
                if loc==4:
                    numOnLetter[loc]=0
            res=max(res,sum(numOnLetter))
        if sum(numOnLetter)>0:
            return -1
        return res
