class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        counter = collections.Counter(croakOfFrogs)
        
        cnt = -1
        for c in list('croak'):
            if c not in counter:
                return -1
            if cnt == -1:
                cnt = counter[c]
            
            if counter[c] != cnt:
                return -1
        
        res = 0
        cur = 0
        counter2 = collections.Counter()
        for c in croakOfFrogs:
            #print(counter2)
            if c == 'c':
                counter2['c'] += 1
                cur += 1
            elif c == 'r':
                if counter2['c'] > 0:
                    counter2['c'] -= 1
                    counter2['cr'] += 1
                else:
                    return -1
            elif c == 'o':
                if counter2['cr'] > 0:
                    counter2['cr'] -= 1
                    counter2['cro'] += 1
                else:
                    return -1
            elif c == 'a':
                if counter2['cro'] > 0:
                    counter2['cro'] -= 1
                    counter2['croa'] += 1
                else:
                    return -1
            elif c == 'k':
                if counter2['croa'] > 0:
                    counter2['croa'] -= 1
                    cur -= 1
                else:
                    return -1
            else:
                return -1
            res = max(res, cur)
        for x, val in counter2.items():
            if val > 0:
                return -1
        return res
