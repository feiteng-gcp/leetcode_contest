class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        m = {"c": 0, "r": 1, "o": 2, "a": 3, "k": 4}
        cnt = [0,0,0,0,0]
        max_frogs = 0
        for c in croakOfFrogs:
            if c not in m:
                return -1
            cnt[m[c]] += 1
            if not (cnt[0] >= cnt[1] and cnt[1] >= cnt[2] and cnt[2] >= cnt[3] and cnt[3] >= cnt[4]):
                return -1
            if cnt[0] >= 1 and cnt[1] >= 1 and cnt[2] >= 1 and cnt[3] >= 1 and cnt[4] >= 1:
                if max_frogs < cnt[0]:
                    max_frogs = cnt[0]
                for i in range(5):
                    cnt[i] -= 1
        if cnt[0] >= 1 or cnt[1] >= 1 or cnt[2] >= 1 or cnt[3] >= 1 or cnt[4] >= 1:
            return -1
        return max_frogs
