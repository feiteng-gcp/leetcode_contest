class Solution:
    m = {c:i for i, c in enumerate('croak')}
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        pos = [0] * 5
        
        for c in croakOfFrogs:
            i = self.m[c]
            if i != 0 and pos[i - 1] == 0:
                return -1
            pos[(i - 1) % 5] = max(pos[(i - 1) % 5] - 1, 0)
            pos[i] += 1
        if sum(pos) != pos[4]: return -1            
        return sum(pos)
