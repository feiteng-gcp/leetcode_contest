class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        ans = {'c': 0, 'r': 0, 'o': 0, 'a': 0, 'k': 0}
        for c in croakOfFrogs:
            if c == 'c':
                if ans['k']:
                    ans['k'] -= 1
                    ans['c'] += 1
                else:
                    ans['c'] += 1
            elif c == 'r':
                if ans['c']:
                    ans['c'] -= 1
                    ans['r'] += 1
                else:
                    return -1
            elif c == 'o':
                if ans['r']:
                    ans['r'] -= 1
                    ans['o'] += 1
                else:
                    return -1
            elif c == 'a':
                if ans['o']:
                    ans['o'] -= 1
                    ans['a'] += 1
                else:
                    return -1
            elif c == 'k':
                if ans['a']:
                    ans['a'] -= 1
                    ans['k'] += 1
                else:
                    return -1
            else:
                return -1
        if sum(ans.values()) - ans['k']:
            return -1
        return ans['k']
