class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        cnt = 0
        cur = 0
        o = collections.defaultdict(int)
        for c in croakOfFrogs:
            if c == 'c':
                o['c'] += 1
                cur += 1
            if c == 'r':
                if not o['c']:
                    return -1
                
                o['c'] -= 1
                o['r'] += 1
                
            if c == 'o':
                if not o['r']: return -1
                o['r'] -= 1
                o['o'] += 1
                
                
            if c == 'k':
                if not o['o']: return -1
                o['o'] -= 1
                o['k'] += 1
                cur -= 1
            
            cnt = max(cur, cnt)
        
        
        
        return cnt if cur == 0 else -1
