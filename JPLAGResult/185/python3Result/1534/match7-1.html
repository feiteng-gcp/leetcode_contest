class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        ret=0
        cur=0
        counts = {'c': 0, 'r': 0, 'o': 0, 'a': 0, 'k': 0}
        for x in croakOfFrogs:
            if x=='c':
                counts[x]+=1
                cur
+=1
            elif x=='r':
                if counts['c']<1:
                    return -1
                counts['c']-=1
                counts[x]+=1
            elif x=='o':
                if counts['r']<1:
                    return -1
                counts['r']-=1
                counts[x]+=1
            elif x=='a':
                if counts['o']<1:
                    return -1
                counts['o']-=1
                counts[x]+=1
            elif x=='k':
                if counts['a']<1:
                    return -1
                counts['a']-=1
                cur-=1
            
            ret=max(ret,cur)
            
        for x in counts.values():
            if x>0:
                return -1
            
        return ret
