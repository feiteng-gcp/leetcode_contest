class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        L = [0, 0, 0, 0]
        m = 0
        for c in croakOfFrogs:
            if c == "c":
                L[0] += 1
            elif c == "r":
                L[0] -= 1
                L[1] += 1
            elif c == "o":
                L[1] -= 1
                L[2] += 1
            elif c == "a":
                L[2] -= 1
                L[3] += 1
            elif c == "k":
                L
[3] -= 1
            if any(l < 0 for l in L):
                return -1
            m = max(m, sum(L))
        if any(L):
            return -1
        return m
