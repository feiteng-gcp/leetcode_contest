class Solution:
    def minNumberOfFrogs(self, croakOfFrogs: str) -> int:
        croak = [0, 0, 0, 0]
        frogs = 0
        for c in croakOfFrogs:
            if c == 'c':
                croak[0] 
+= 1
            elif c == 'r':
                if croak[0] > 0:
                    croak[0] -= 1
                    croak[1] += 1
                else:
                    return -1
            elif c == 'o':
                if croak[1] > 0:
                    croak[1] -= 1
                    croak[2] += 1
                else:
                    return -1
            elif c == 'a':
                if croak[2] > 0:
                    croak[2] -= 1
                    croak[3] += 1
                else:
                    return -1
            elif c == 'k':
                if croak[3] > 0:
                    croak[3] -= 1
                else:
                    return -1
                
            frogs = max(frogs, sum(croak))

            
        for c in croak:
            if c != 0:
                return -1
        
        return frogs
