class Solution {
    int[] dist;
    int[] segments;
    public double minmaxGasDist(int[] stations, int K) {
        Arrays.sort(stations);
        dist = new int[stations.length - 1];
        segments = new int[stations.length - 1];
        double l = 0;
        double r = (double) Integer.MIN_VALUE;
        for (int i = 0; i < stations.length - 1; ++i) {
            dist[i] = stations[i + 1] - stations[i];
            r = Math.max(dist[i], r);
        }
        double m = r;
        while (r - l > 0.000001) {
            m = l + (r - l) / 2;
            
int temp = count(m);
            // System.out.println(m + ": " + temp);
            if (temp > K) {
                l = m;
            } else {
                r = m;
            }
        }
        return m;
        // double result = 0;
        // for (int i = 0; i < dist.length; ++i) {
        //     result = Math.max(result, 1.0 * dist[i] / segments[i]);
        // }
        // return result;
    }
    int count(double m) {
        int ret = 0;
        for (int i = 0; i < dist.length; ++i) {
            segments[i] = (int) Math.ceil(dist[i] / m);
            ret += segments[i] - 1;
        }
        return ret;
    }
}
