class Solution {
    public double minmaxGasDist(int[] stations, int K) {
        double l = 0;
        double r = 0;
        for (int i = 1; i < stations.length; i++) {
            r = Math.max(r, stations[i]-stations[i-1]);
        }

        while (r-l>0.00000001) {
            double m = (r+l)/2;

            int remain = K;
            int i = 0;
            double pos = stations[i];
            while (i < stations.length-1) {
                double dis = stations[i+1]-pos;
                if (dis < m) {
                    i++;
                    pos = stations[i];
                } else {
                    int need = (int) ((dis-m)/m+1);
                    if (remain < need) break;
                    remain -= need;
                    i++;
                    pos = stations[i];
                }
            }

            if (i == stations.length-1) {
                r = m;
            } else {
                l = m;
            }
        }
        return l;
    }
}
