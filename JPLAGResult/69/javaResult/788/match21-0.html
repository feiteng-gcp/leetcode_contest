class Solution {
    public double minmaxGasDist(int[] stations, int K) {
        //Arrays.sort(stations);
        int n = stations.length;
        int[] gap = new int[n-1];
        int maxgap = 0;
        for(int i = 0; i< n-1; i++){
            gap[i] = stations[i+1] - stations[i];
            maxgap = Math.max(maxgap, gap[i]);
        }
        
        
       
        
        double l = 0.0, r = (double) maxgap;

        
        while(r-l> 0.000001){
            double mid = (r+l)/2;
            
            if(count(gap, mid)<= K){
                r = mid;
            }
            else{
                l = mid;
            }
        }
        
        return l;
        
        
    }
    
    public int count(int[] gap, double val){
        int res = 0;
        
        for(int i = 0; i< gap.length; i++){
            res += Math.ceil( gap[i]/val) -1;
        }
        
        return res;
    }
}
