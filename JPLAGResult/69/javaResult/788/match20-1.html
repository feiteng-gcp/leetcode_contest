class Solution {
    public double minmaxGasDist(int[] stations, int K) {
        int[] bt = new int[stations.length - 1];
        for (int i = 0; i < bt.length; i++) {
            bt[i] = stations[i + 1] - stations[i];
        }
        int maxinum = 0;
        for (int i: bt) {
            
maxinum = Math.max(maxinum, i);
        }
        
        double l = 0, r = maxinum;
        while (r - l > 0.0000001) {
            double mid = (l + r) / 2;
            if (canAccept(bt, K, mid)) {
                r = mid;
            }else {
                l = mid;
            }
        }
        return (l + r) / 2;
    }

    public boolean canAccept(int[] bt, int K, double number) {
        for (int i: bt) {
            double tmp = i / number;
            tmp = Math.ceil(tmp);
            K -= tmp - 1;
            if (K < 0) return false;
        }
        return true;
    }
}
