class Solution {
    public double minmaxGasDist(int[] stations, int K) {
        Arrays.sort(stations);
        
double min=0;
        double max=0;
        for(int i=0;i<stations.length-1;i++){
            max=Math.max(max,stations[i+1]-stations[i]);
        }
        while(max-min>0.000001){
            double mid=min+(max-min)/2.0;
            int left=K;
            for(int i=0;i<stations.length-1&&left>=0;i++){
                double dis=stations[i+1]-stations[i];
                int need=(int)(dis/mid);
                left=left-need;
            }
            
if(left<0){
                min=mid;
            }
            else{
                max=mid;
            }
        }
        return min;
    }
}
