public class Solution {
	double EPS = 1e-12;
	public double minmaxGasDist(int[] stations, int K) {
		// use binary search
		int n = stations.length;
		Arrays.sort(stations);
		int[] segs = n
ew int[n - 1];
		double maxSeg = 0;
		for (int i = 0; i < n - 1; ++i) {
			segs[i] = stations[i + 1] - stations[i];
			maxSeg = Math.max(maxSeg, segs[i]);
		}
		int[] res = new int[n - 1];
		double left = maxSeg / (K + 1), right = maxSeg;
		while (left + EPS < right) {
			int cnt = 0;
			int res1[] = new int[n - 1];
			double mid = left + (right - left) / 2;
			for (int i = 0; i < n - 1; ++i) {
				res1[i] = (int)Math.ceil(segs[i] / mid);
				cnt += res1[i];
			}
			
			if (cnt <= K + n - 1) {
//				res = res1;
				right = mid;
			}else
				left = mid;
		}
		return right;
	}
}
