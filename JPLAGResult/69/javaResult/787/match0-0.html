class Solution {
    public int slidingPuzzle(int[][] board) {
        Set<String> set = new HashSet<>();
        set.add(encode(board));
        Queue<int[][]> q = new LinkedList<>();
        int count = 0;
        q.offer(board);
        while (!q.isEmpty()) {
            int size = q.size();
            for (int i = 0; i < size; i++) {
                int[][] cur = q.poll();
                if (encode(cur).equals("123450")) {
                    return count;
                }
                for (int[][] next : getNextState(cur)) {
                    String str = encode(next);
                    if (set.contains(str)) {
                        continue;
                    }
                    set.add(str);
                    q.offer(next);
                }
            }
            count++;
        }
        return -1;
    }
    
    public String encode(int[][] board) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                sb.append(board[i][j]);
            }
        }
        return sb.toString();
    }
    
    public List<int[][]> getNextState(int[][] board) {
        List<int[][]> res = new ArrayList<>();
        int x = 0, y = 0;
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                if (board[i][j] == 0) {
                    x = i;
                    y = j;
                }
            }
        }
        int[][] dirs = new int[][]{{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        for (int[] dir : dirs) {
            int nx = x + dir[0];
            int ny = y + dir[1];
            if (nx < 0 || nx >= 2 || ny < 0 || ny >= 3) {
                continue;
            }
            int[][] copy = clone(board);
            int temp = copy[x][y];
            copy[x][y] = copy[nx][ny];
            copy[nx][ny] = temp;
            res.add(copy);
        }
        return res;
    }
    
    public int[][] clone(int[][] board) {
        int[][] copy = new int[2][3];
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                copy[i][j] = board[i][j];
            }
        }
        return copy;
    }
}
