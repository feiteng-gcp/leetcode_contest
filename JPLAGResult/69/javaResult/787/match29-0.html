class Solution {
    public boolean isVisited(int[] arr, Set<String> set){
       
       if(set.contains(getStr(arr))) return true;
        
        return false;
    }
    public int[] getArr(int[][] board){
        int arr[] = new int[6];
        for(int i=0;i<3;i++){
            arr[i] = board[0][i];
        }
        for(int i=0;i<3;i++){
            arr[i+3] = board[1][i];
        }
        return arr;
    }
    public List<int[]> getAdj(int[] board){
        int i=0;
        for(int j=0;j<6;j++){
            if(board[j] == 0) {
                i=j;
                break;
            }
        }
        List<int[]> list = new ArrayList<>();
        switch(i){
            case 0:  
                list.add(new int[]{board[1],board[0], board[2], board[3],board[4],board[5]});
                list.add(new int[]{board[3],board[1], board[2], board[0],board[4],board[5]});
                     
                break;
            case 1:
                list.add(new int[]{board[1],board[0], board[2], board[3],board[4],board[5]});
                list.add(new int[]{board[0],board[4], board[2], board[3],board[1],board[5]});
                list.add(new int[]{board[0],board[2], board[1], board[3],board[4],board[5]});
                
                break;
            case 2:
                list.add(new int[]{board[0],board[2], board[1], board[3],board[4],board[5]});
                list.add(new int[]{board[0],board[1], board[5], board[3],board[4],board[2]});
                
                break;
            case 3:
                list.add(new int[]{board[3],board[1], board[2], board[0],board[4],board[5]});
                list.add(new int[]{board[0],board[1], board[2], board[4],board[3],board[5]});
                
                break;
            case 4:
                list.add(new int[]{board[0],board[4], board[2], board[3],board[1],board[5]});
                list.add(new int[]{board[0],board[1], board[2], board[4],board[3],board[5]});
                list.add(new int[]{board[0],board[1], board[2], board[3],board[5],board[4]});
                
                break;
            case 5:
                list.add(new int[]{board[0],board[1], board[2], board[3],board[5],board[4]});
                list.add(new int[]{board[0],board[1], board[5], board[3],board[4],board[2]});
                
                break;
            
        }
        return list;
        
    }
    public boolean matches(int[] tarr){
        for(int i=0;i<5;i++){
            if(tarr[i]!=i+1){
                return false;
            }
        }
        if(tarr[5] == 0){
            return true;
        }
        return false;
    }
    public String getStr(int[] arr){
        StringBuilder str = new StringBuilder();
        for(int i=0;i<arr.length;i++){
            str.append(arr[i]);
        }
        return str.toString();
    }
    public int slidingPuzzle(int[][] board) {
        Set<String> visited = new HashSet<>();
        int[] arr = getArr(board);
        Queue<int[]> q = new LinkedList<>();
        q.offer(arr);
        visited.add(getStr(arr));
        int level = 0;
        while(!q.isEmpty()){
            int n = q.size();
            //System.out.println(n);
            for(int i=0;i<n;i++){
                int tarr[] = q.poll();
                //for(int y=0;y<tarr.length;y++){
                //    System.out.print(" " + tarr[y]);
                //}
                //System.out.println();
                if(matches(tarr)){
                    return level;
                }
                visited.add(getStr(tarr));
                for(int[] aff : getAdj(tarr)){
                    if(!isVisited(aff,visited)){
                        q.add(aff);
                    }
                }
            }
            level++;
        }
        return -1;
    }
}
