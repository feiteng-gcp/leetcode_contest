class Solution {
    int[] state = new int[46657];
     
int[][] dirs = {{1,0},{0,1},{-1,0},{0,-1}};
    int steps = -1;
    public  int toState(int[][] board) {
        return 1*board[0][0] +     6 *board[0][1] + 36*board[0][2]
                +216*board[1][0] + 1296*board[1][1] + 7776*board[1][2];
    }
    public  List<int[][]> next (int[][] board) {
        
int x = 0;
        int y = 0;
        for(int i = 0; i <  2; i++) {
            for(int j = 0; j < 3; j++) {
                if(board[i][j] == 0) {
                    x = i;
                    y = j;
                }
            }
        }

        List<int[][]> ans = new ArrayList<>();
        for(int[] dir:dirs) {
            int i = x + dir[0];
            int j = y + dir[1];
            if(i >= 0 && j >= 0 && i < 2 && j < 3) 
{
                int[][] n = new int[2][3];
                for(int a = 0; a < 2; a++) {
                    for(int b = 0; b < 3; b++) {
                        n[a][b] = board[a][b];
                    }
                }
                n[i][j] = 0;
                n[x][y] = board[i][j];
                if(state[toState(n)] == 0)
                ans.add(n);
            }
        }
        return ans;
    }
    public void backtracking(int[][] board, int cur) {
        if(steps != -1 && cur > steps) {
            return;
        }
        int s = toState(board);
        if(state[s] != 0) {
            return;
        }

        if(s == 7465) {
            if(steps == -1) {
                steps = cur;
            } else {
                steps = Math.min(steps, cur);
            }
            return;
        }
        for(int[][] n : next(board)) {
            backtracking(n, cur + 1);
        }

        state[s] = 1;
    }
    public int slidingPuzzle(int[][] board) {
        
LinkedList<int[][]> lst = new LinkedList<>();
        lst.add(board);
        int steps = 0;
        while(lst.size()  > 0) {
            int size = lst.size();
            for(int i = 0; i < size; i++) {
                int[][] cur = lst.poll();
                int s = toState(cur);
                state[s] = 1;
                if(7465 == s) {
                    return steps;
                }
                for(int[][] n : next(cur)) {
                    if(state[toState(n)] == 0) {
                        
lst.add(n);
                    }
                }
            }
            steps++;
        }
        return -1;
    }
}
