class Solution {
     public int slidingPuzzle(int[][] board) {
    Queue<int[][]> q = new LinkedList<>();
    q.offer(board);
    Set<String> visited = new HashSet<>();
    visited.add(
mkString(board));
    int step = 0;
    while(!q.isEmpty()) {
      int qsize = q.size();
      for(int i=0; i<qsize; i++) {
        int[][] curr = q.poll();
        if(mkString(curr).equals("123450")) return step;
        for(int[][] next: mkNext(curr)) {
          if(!visited.contains(mkString(next))) {
            q.offer(next);
            visited.add(
mkString(next));
          }
        }
      }
      step++;
    }

    return -1;
  }
  String mkString(int[][] board) {
    StringBuilder sb = new StringBuilder();
    for(int[] l: board) {
      for(int i: l) sb.append(i);
    }
    return sb.toString();
  }

  List<int[][]> mkNext(int[][] b) {
    List<int[][]> res = new ArrayList<>();
    int[] idx = new int[]{0,0};
    
for(int i=0; i<2; i++) {
      for(int j=0; j<3; j++) {
        if(b[i][j]==0) {
          idx = new int[]{i, j};
          break;
        }
      }
    }
    
int[][] dirs = {{-1,0},{0,-1},{0,1},{1,0}};
    for(int[] d: dirs) {
      int ni = idx[0] + d[0];
      int nj = idx[1] + d[1];
      if(ni>=0 && ni<2 && nj>=0 && nj < 3) {
        int[][] next = new int[2][3];
        next[0] = b[0].clone();
        next[1] = b[1].clone();
        next[ni][nj] = 0;
        next[idx[0]][idx[1]] = b[ni][nj];
        res.add(next);
      }
    }
    return res;
  }
}
