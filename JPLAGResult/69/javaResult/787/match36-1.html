class Solution {
      
class E {
        String s;
        int d;
        E(String ss, int dd) {
            s = ss;
            d = dd;
        }
    }
    Set<String> seen = new HashSet<>();
    
public int slidingPuzzle(int[][] board) {
        StringBuilder b = new StringBuilder();
        for (int i = 0; i < 2; i++)
            for (int j = 0; j < 3; j++) {
                int v = board[i][j];
                b.append(v);
            }
        String t = "123450";
        Queue<E> q = new LinkedList<>();
        q.offer(new E(b.toString(), 0));
        int[] delta = {1, -1, 3, -3};
        while (!q.isEmpty()) {
            E e = q.poll();
            if (e.s.equals(t)) return e.d;
            if (seen.contains(e.s)) continue;
            seen.add(e.s);
            int i = e.s.indexOf('0');
            char[] cs = e.s.toCharArray();
            
for (int d : delta) {
                int x = i + d;
                if ((d == 1 || d == -1) && x /3 != i / 3) continue;
                if (x < 0 || x > 5) continue;                
                cs[i] = cs[x];
                cs[x] = '0';
                q.offer(new E(new String(cs), e.d + 1));
                cs[x] = cs[i];
            }
        }
        return -1;
    }
}
