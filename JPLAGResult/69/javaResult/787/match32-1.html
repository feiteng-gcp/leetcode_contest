class Solution {
    public int slidingPuzzle(int[][] board) {
        
StringBuilder sb = new StringBuilder();
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                sb.append((char)(board[i][j] + '0'));
            }
        }
        String start = sb.toString();
        Queue<String> queue = new LinkedList<>();
        Queue<String> next = new LinkedList<>();
        queue.add(start);
        Set<String> set = new HashSet<>();
        set.add(start);
        int ans = 0;
        while (!queue.isEmpty()) {
            String curr = queue.poll();
            if (curr.equals("123450")) return ans;
            String[] tmp = gene(curr);
            for (String s: tmp) {
                if (!set.contains(s)) {
                    set.add(s);
                    next.add(s);
                }
            }
            if (queue.isEmpty()) {
                queue = next;
                next = new LinkedList();
                ans++;
            }
        }
        return -1;
    }

    public String[] gene(String start) {
        String[] ans = null;
        char[] arr = start.toCharArray();
        int idx = start.indexOf('0');
        switch (idx) {
            case 0:
                ans = new String[2];
                ans[0] = exchange(arr, 0, 3);
                ans[1] = exchange(arr, 0, 1);
                break;
            case 1:
                ans = new String[3];
                ans[0] = exchange(arr, 1, 0);
                ans[1] = exchange(arr, 1, 2);
                ans[2] = exchange(arr, 1, 4);
                break;
            case 2:
                ans = new String[2];
                ans[0] = exchange(arr, 2, 1);
                ans[1] = exchange(arr, 2, 5);
                break;
            case 3:
                ans = new String[2];
                ans[0] = exchange(arr, 3, 0);
                ans[1] = exchange(arr, 3, 4);
                break;
            case 4:
                ans = new String[3];
                ans[0] = exchange(arr, 4, 3);
                ans[1] = exchange(arr, 4, 1);
                ans[2] = exchange(arr, 4, 5);
                break;
            case 5:
                ans = new String[2];
                ans[0] = exchange(arr, 5, 2);
                ans[1] = exchange(arr, 5, 4);
        }
        return ans;
    }

    public String exchange(char[] arr, int a, int b) {
        char tmp = arr[a];
        arr[a] = arr[b];
        arr[b] = tmp;
        String s = new String(arr);
        tmp = arr[a];
        arr[a] = arr[b];
        arr[b] = tmp;
        return s;
    }
}
