class Solution {
    private Set<String> visited;
    public int slidingPuzzle(int[][] board) {
        StringBuffer start=new StringBuffer();
        for (int j=0;j<board.length;j++)
            for (int i=0;i<board[0].length;i++)
                start.append(board[j][i]);
        String b=start.toString();
        visited=new HashSet<String>();
        Queue<PReturn> v=new LinkedList<PReturn>();
        Queue<Integer> steps=new LinkedList<Integer>();
        v.offer(new PReturn(b,b.indexOf("0")));
        steps.offer(0);
        if (b.equals("123450"))
            return 0;
        while (v.size()>0)
        {
            int step=steps.poll();
            PReturn cur=v.poll();
            visited.add(cur.b);
            //System.out.println(cur.b+" "+step);
            Set<PReturn> nexts=findNext(cur.b,cur.position);
            for (PReturn next : nexts)
            {
                if (!visited.contains(next.b))
                {
                    if (next.b.equals("123450"))
                        return step+1;
                    else
                    {
                        v.offer(new  PReturn(next.b,next.position));
                        steps.offer(step+1);
                    }
                }
            }
        }
        return -1;
        //return stepUses(b,b.indexOf("0"));
    }
    
    private int stepUses(String s, int pos)
    {
        System.out.println(s);
        visited.add(s);
        if (s.equals("123450"))
        {
            return 0;
        }
        else
        {
            Set<PReturn> nexts=findNext(s,pos);
            for (PReturn next : nexts)
            {
                if (!visited.contains(next.b))
                {
                    int ns=stepUses(next.b,next.position);
                    if (ns!=-1)
                        return ns+1;
                }
            }
            return -1;
        }
    }
    
    private Set<PReturn> findNext(String s, int pos)
    {
        int[] dis={-1,1,3};
        if (pos>=3)
            dis[2]=-3;
        Set<PReturn> nexts=new HashSet<PReturn>();
        for (int i=0;i<dis.length;i++)
        {
            int newpos=pos+dis[i];
            if ((pos==2)&&(newpos==3))
                continue;
            if ((pos==3)&&(newpos==2))
                continue;
            if ((newpos>=0)&&(newpos<=5)){
                char[] ss=s.toCharArray();
                char temp=ss[pos];
                ss[pos]=ss[newpos];
                ss[newpos]=temp;
                nexts.add(new PReturn(new String(ss),newpos));
            }
        }
        return nexts;
    }
    
    class PReturn{
        public String b;
        public int position;
        public PReturn(String s, int p)
        {
            b=s;
            position=p;
        }
    }
}
