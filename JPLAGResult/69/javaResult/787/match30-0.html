import java.util.LinkedList;
import java.util.Set;

class Solution {
    private static final int[][] DIRS = {
        { 1, 0 }, { -1, 0 }, { 0, 1 }, { 0, -1 } 
    };
    public int slidingPuzzle(int[][] board) {
        if (getKey(board).equals("123450")) {
            return 0;
        }
        Set<String> visited = new HashSet<>();
        LinkedList<int[][]> queue = new LinkedList<>();
        queue.addLast(board);
        visited.add(getKey(board));
        int steps = 1;
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int[][] curr = queue.removeFirst();
                        
                // System.out.println(getKey(curr));

                int[] coor = findZero(curr);
                for (int[] d : DIRS) {
                    int ni = coor[0] + d[0];
                    int nj = coor[1] + d[1];
                    if (swap(curr, coor[0], coor[1], ni, nj)) {
                        String key = getKey(curr);
                        if (visited.contains(key)) {
                            swap(curr, coor[0], coor[1], ni, nj);
                            continue;
                        }
                        if (key.equals("123450")) {
                            // System.out.println("end");
                            return steps;
                        }

                        visited.add(key);
                        queue.addLast(clone(curr));
                        swap(curr, coor[0], coor[1], ni, nj);
                    }
                }
            }
            steps++;
        }
        return -1;
    }

    private int[] findZero(int[][] board) {
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                if (board[i][j] == 0) {
                    return new int[] { i, j };
                }
            }
        }
        return null;
    }

    private boolean swap(int[][] curr, int i1, int j1, int i2, int j2) {
        if (i1 < 0 || i1 == 2 || i2 < 0 || i2 == 2) {
            return false;
        }
        if (j1 < 0 || j1 == 3 || j2 < 0 || j2 == 3) {
            return false;
        }

        int tmp = curr[i1][j1];
        curr[i1][j1] = curr[i2][j2];
        curr[i2][j2] = tmp;
        return true;
    }

    private String getKey(int[][] board) {
        StringBuilder b = new StringBuilder();
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                b.append(board[i][j]);
            }
        }
        return b.toString();
    }

    private int[][] clone(int[][] board) {
        int[][] b = new int[2][3];
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                b[i][j] = board[i][j];
            }
        }
        return b;
    }
}
