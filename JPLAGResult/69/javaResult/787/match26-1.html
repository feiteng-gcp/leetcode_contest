class Solution {
       private boolean isFinal(int[][] board) {
        for (int i = 0; i < 6; ++i) {
            int y = i / 3;
            int x = i % 3;
            if (board[y][x] != (i+1)%6) return false;
        }
        return true;
    }
    private int hash(int[][] board) {
        int sum = 0;
        for (int i = 0; i < 6; ++i) {
            int y = i / 3;
            int x = i % 3;
            sum = 6 * sum + board[y][x];
        
}
        return sum;
    }
    private int[][] clone(int[][] board) {
        int[][] ans = new int[2][3];
        for (int y = 0; y < 2; y++) {
            for (int x = 0; x < 3; x++) {
                ans[y][x] = board[y][x];
            }
        }
        return ans;
    }
    public int slidingPuzzle(int[][] board) {
        int[] best = new int[50000];
        boolean[] visited = new boolean[50000];
        Queue<int[][]> q = new LinkedList<>();
        Queue<Integer> qLevel = new LinkedList<>();
        q.add(board);
        qLevel.add(0);
        visited[hash(board)] = true;
        int[] dirs = {1, 0, -1, 0, 1};
        int depth = 0;
        while (!q.isEmpty()) {
            board = q.remove();
            int level = qLevel.remove();
            int hv = hash(board);
            if (isFinal(board)) {
                return level;
            }
            for (int i = 0; i < 6; ++i) {
                int y = i / 3;
                int x = i % 3;
                if (board[y][x] == 0) {
                    for (int k = 0; k < 4; ++k) {
                        int ny = y + dirs[k], nx = x + dirs[k+1];
                        if (ny >= 0 && ny < 2 && nx>=0 && nx<3) {
                            int[][] next = clone(board);
                            swap(next, y, x, ny, nx);
                            int nextHv = hash(next);
                            if (!visited[nextHv]) {
                                q.add(next);
                                qLevel.add(level+1);
                                visited[nextHv] = true;
                            }
                        }
                    }
                    break;
                
}
            }
        }
        return -1;
    }
    
void swap(int[][]board, int y1, int x1, int y2, int x2) {
        int t=board[y1][x1];
        board[y1][x1] = board[y2][x2];
        board[y2][x2] = t;
    }
}
