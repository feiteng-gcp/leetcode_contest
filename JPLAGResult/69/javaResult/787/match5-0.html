class Solution {
    public int slidingPuzzle(int[][] board) {
        int[][] dirs = new int[][] {{0,1},{0,-1},{1,0},{-1,0}};
        String res = "1,2,3,4,5,0,";
        Queue<Node> queue = new LinkedList<>();
        Set<String> visited = new HashSet<String>();
        
        for(int i = 0; i < board.length; i++) {
            for(int j = 0; j < board[i].length; j++) {
                if(board[i][j] == 0) {
                    queue.offer(new Node(board, 0, i, j));
                    break;
                }
            }
        }
        
        while(!queue.isEmpty()) {
            Node current = queue.poll();
            String str = stringify(current.board);
            visited.add(str);
            if(str.equals(res)) {
                return current.count;
            }
            
            for(int[] dir : dirs) {
                int x = current.i + dir[0], y = current.j + dir[1];
                if(isValid(x, y, board)) {
                    int[][] swapped = swap(current.i, current.j, x, y, current.board);
                    String key = stringify(swapped);
                    if(!visited.contains(key)) {
                        queue.offer(new Node(swapped, current.count+1, x, y));
                    }
                }
            }
        }
        return -1;
    }
    
    private String stringify(int[][] board) {
        StringBuilder builder = new StringBuilder();
        for(int i = 0; i < board.length; i++) {
            for(int j = 0; j < board[i].length; j++) {
                builder.append(board[i][j]).append(",");
            }
        }
        return builder.toString();
    }
    
    private boolean isValid(int x, int y, int[][] board) {
        return x >= 0 && x < board.length && y >= 0 && y < board[x].length;
    }
    
    private int[][] swap(int a, int b, int x, int y, int[][] board) {
        int[][] res = new int[board.length][board[0].length];
        for(int i = 0; i < board.length; i++) {
            for(int j = 0; j < board[i].length; j++) {
                res[i][j] = board[i][j];
            }
        }
        int temp = res[x][y];
        res[x][y] = res[a][b];
        res[a][b] = temp;
        return res;
    }
    
    class Node {
        int[][] board;
        int count = 0, i = 0, j = 0;
        public Node(int[][] board, int count, int i, int j) {
            this.board = board;
            this.count = count;
            this.i = i;
            this.j = j;
        }
    }
}
