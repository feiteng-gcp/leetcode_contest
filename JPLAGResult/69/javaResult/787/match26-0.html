class Solution {
    private final static int[] DIRS = new int[]{0, 1, 0, -1, 0};
    public int slidingPuzzle(int[][] board) {
        String solution = "1,2,3,4,5,0";

        int[] start = null;
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                sb.append(board[i][j]).append(",");
                if (board[i][j] == 0) start = new int[]{i, j};
            }
        }

        Queue<String> q = new LinkedList<>();
        Queue<int[]> pos = new LinkedList<>();

        q.add(sb.substring(0, sb.length() - 1));
        pos.add(start);

        Set<String> visited = new HashSet<>();

        int steps = 0;
        while (!q.isEmpty()) {
            int size = q.size();

            while (size > 0) {
                String b = q.poll();
                int[] loc = pos.poll();

                size -= 1;

                if (b.equals(solution)) return steps;
                if (visited.contains(b)) continue;
                visited.add(b);

                for (int d = 0; d < 4; d++) {
                    int i = loc[0] + DIRS[d], j = loc[1] + DIRS[d + 1];
                    if (i < 0 || j < 0 || i >= 2 || j >= 3)
                        continue;

                    String[] ss = b.split(",");
                    int p = i * 3 + j;
                    int ori = loc[0] * 3 + loc[1];
                    ss[ori] = ss[p];
                    ss[p] = "0";
                    q.add(String.join(",", ss));
                    pos.add(new int[]{i, j});
                }

            }
            steps += 1;
        }

        return -1;
    }
}
