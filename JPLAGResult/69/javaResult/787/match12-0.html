class Solution {
    public int slidingPuzzle(int[][] board) {
        int[] visited = new int[46656];
        int target = encode(endBoard);
        int start = encode(board);
        if (start == target) {
            return 0;
        }
        //visited.add(start);
        visited[start] = 1;
        LinkedList<Integer> queue = new LinkedList<>();
        queue.offer(start);
        int count = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            count++;
            for (int k = 0; k < size; k++) {            
                int code = queue.poll();
                int[][] a = decode(code);
                for (int next : move(a)) {
                    if (next == target) {
                        return count;
                    }
                    if (visited[next] == 0) {
                        visited[next] = 1;
                        queue.offer(next);
                    }
                }
            }
            
        }
        return -1;
    }
    
    final int[][] endBoard = {{1,2,3}, {4,5,0}};
    
    List<Integer> move(int[][] a) {
        List<Integer> list = new ArrayList<>();
        int x0 = -1;
        int y0 = -1;
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                if (a[i][j] == 0) {
                    x0 = i;
                    y0 = j;
                }
            }
        }
        for (int i = 0; i < dx.length; i++) {
            int x = x0 + dx[i];
            int y = y0 + dy[i];
            if (x < 0 || x >= 2 || y < 0 || y >= 3) {
                continue;
            }
            int temp = a[x0][y0];
            a[x0][y0] = a[x][y];
            a[x][y] = temp;
            list.add(encode(a));
            a[x][y] = a[x0][y0];
            a[x0][y0] = temp;
        }
        return list;
    }
    
    int[] dx = {1, -1, 0, 0};
    int[] dy = {0, 0, 1, -1};
    
    int encode(int[][] board) {
        int x = 1;
        int y = 0;
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                y += x * board[i][j];
                x *= 6;
            }
        }
        return y;
    }
    
    int[][] decode(int code) {
        int[][] a = new int[2][3];
        for (int i = 0; i < 2; i++) {
            for (int j = 0; j < 3; j++) {
                a[i][j] = code % 6;
                code /= 6;
            }
        }
        return a;
    }
}
