class Solution {

    class Vertex {

        int[][] board;
        int value;

        Vertex(int[][] board) {
            this.board = board;
            this.value = getValue();
        }

        // [[1,2,3],[4,5,0]]
        Set<Vertex> getAdjacentVertices() {
            Set<Vertex> vertices = new HashSet<>();
            if (board[0][0] == 0) {
                int[][] copy = getCopy(board);
                swap(copy, 0, 0, 0, 1);
                vertices.add(new Vertex(copy));

                copy = getCopy(board);
                swap(copy, 0, 0, 1, 0);
                vertices.add(new Vertex(copy));
            } else if (board[0][1] == 0) {
                int[][] copy = getCopy(board);
                swap(copy, 0, 1, 0, 0);
                vertices.add(new Vertex(copy));

                copy = getCopy(board);
                swap(copy, 0, 1, 1, 1);
                vertices.add(new Vertex(copy));

                copy = getCopy(board);
                swap(copy, 0, 1, 0, 2);
                vertices.add(new Vertex(copy));
            } else if (board[0][2] == 0) {
                int[][] copy = getCopy(board);
                swap(copy, 0, 2, 0, 1);
                vertices.add(new Vertex(copy));

                copy = getCopy(board);
                swap(copy, 0, 2, 1, 2);
                vertices.add(new Vertex(copy));
            } else if (board[1][0] == 0) {
                int[][] copy = getCopy(board);
                swap(copy, 1, 0, 0, 0);
                vertices.add(new Vertex(copy));

                copy = getCopy(board);
                swap(copy, 1, 0, 1, 1);
                vertices.add(new Vertex(copy));
            } else if (board[1][1] == 0) {
                int[][] copy = getCopy(board);
                swap(copy, 1, 1, 1, 0);
                vertices.add(new Vertex(copy));

                copy = getCopy(board);
                swap(copy, 1, 1, 1, 2);
                vertices.add(new Vertex(copy));

                copy = getCopy(board);
                swap(copy, 1, 1, 0, 1);
                vertices.add(new Vertex(copy));
            } else if (board[1][2] == 0) {
                int[][] copy = getCopy(board);
                swap(copy, 1, 2, 0, 2);
                vertices.add(new Vertex(copy));

                copy = getCopy(board);
                swap(copy, 1, 2, 1, 1);
                vertices.add(new Vertex(copy));
            
}

            return vertices;
        }

        private int[][] getCopy(int[][] board) {
            int[][] copy = {{0, 0, 0}, {0, 0, 0}};
            copy[0][0] = board[0][0];
            copy[0][1] = board[0][1];
            copy[0][2] = board[0][2];
            copy[1][0] = board[1][0];
            copy[1][1] = board[1][1];
            copy[1][2] = board[1][2];
            return copy;
        }

        private void swap(int[][] b, int ax, int ay, int bx, int by) {
            int temp = b[ax][ay];
            b[ax][ay] = b[bx][by];
            b[bx][by] = temp;
        }

        // [[1,2,3],[4,5,0]]
        boolean isGoal() {
            if (board[0][0] != 1) {
                return false;
            }

            if (board[0][1] != 2) {
                return false;
            }

            if (board[0][2] != 3) {
                return false;
            }

            if (board[1][0] != 4) {
                return false;
            }

            if (board[1][1] != 5) {
                return false;
            }

            
if (board[1][2] != 0) {
                return false;
            }

            return true;
        }

        int getValue() {
            int sum = 0;
            sum += board[0][0] * 100000;
            sum += board[0][1] * 10000;
            sum += board[0][2] * 1000;
            sum += board[1][0] * 100;
            sum += board[1][1] * 10;
            sum += board[1][2];
            return sum;
        }

    }

    public int slidingPuzzle(int[][] board) {
        Vertex startVertex = new Vertex(board);
        Queue<FrontierItem> frontier = new LinkedList<>();
        Set<Integer> seen = new HashSet<>();
        frontier.add(new FrontierItem(0, startVertex));
        seen.add(startVertex.value);

        while (!frontier.isEmpty()) {
            FrontierItem item = frontier.remove();
            Vertex vertex = item.vertex;
            if (vertex.isGoal()) {
                return item.depth;
            }

            for (Vertex adjacentVertex : vertex.getAdjacentVertices()) {
                if (!seen.contains(adjacentVertex.value)) {
                    seen.add(adjacentVertex.value);
                    frontier.add(new FrontierItem(item.depth + 1, adjacentVertex));
                }
            }
        }

        return -1;
    }

    class FrontierItem {

        int depth;
        Vertex vertex;

        FrontierItem(int depth, Vertex vertex) {
            this.depth = depth;
            this.vertex = vertex;
        }

    }

}
