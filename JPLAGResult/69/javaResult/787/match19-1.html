class Solution {
    public int slidingPuzzle(int[][] board) {
        int target = 123450;
        Queue<Integer> q = new LinkedList();
        int start = toHash(board);
        
steps.put(start, 0);
        q.add(start);
        while(q.size() > 0) {
            int current = q.poll();
            int currentStep = steps.get(current);
            if (current == target) return steps.get(current);
            int[][] bd = toBoard(current);
            int zerox = -1;
            int zeroy = -1;
            for (int i = 0; i < 2; ++i)
                for (int j = 0; j < 3; ++j) {
                    if (bd[i][j] == 0) {
                        zerox = i;
                        zeroy = j;
                        break;
                    }
                }
            for (int i = 0; i < 4; ++i) {
                int newx = zerox + dx[i];
                int newy = zeroy + dy[i];
                if (newx >= 0 && newx < 2 && newy >= 0 && newy < 3) {
                    
swap(bd, zerox, zeroy, newx, newy);
                    int s = toHash(bd);
                    if (!steps.containsKey(s)) {
                        steps.put(s, currentStep + 1);
                        q.add(s);
                    }
                    swap(bd, newx, newy, zerox, zeroy);
                
}
            }
        }
        return -1;
    }
    
    int[] dx = new int[] {0, 1, 0, -1};
    int[] dy = new int[] {1, 0, -1, 0};
    
    
    
    void swap(int[][] bd, int x1, 
int y1, int x2, int y2) {
        int t = bd[x1][y1];
        bd[x1][y1] = bd[x2][y2];
        bd[x2][y2] = t;
    }
    
    HashMap<Integer, Integer> steps = new HashMap();
    
    int toHash(int[][] board) {
        int result = 0;
        for (int i = 0; i < board.length; ++i)
            for (int j = 0; j < board[i].length; ++j) {
                result = 10 * result + board[i][j];
            }
        return result;
    }
    
    int[][] toBoard(int value) {
        int[][] result = new int[2][3];
        for (int i = 1; i >= 0; --i) {
          for (int j = 2; j >= 0; --j) {
                result[i][j] = value % 10;
                value /= 10;
          }
        }
        return result;
    }
}
