class Solution {

    public int slidingPuzzle(int[][] board) {
        if (trans(board) == 123450) {
            return 0;
        }

        Set<Integer> set = new HashSet<>();
        Queue<int[][]> queue = new LinkedList<>();
        int[] xDir = new int[]{1,-1,0,0};
        int[] yDir = new int[]{0,0,1,-1};
        int target = 123450;
        set.add(trans(board));
        queue.offer(board);
        int steps = 0;

        while (!queue.isEmpty()) {
            int size = queue.size();
            steps++;
            for (int i = 0; i < size; i++) {
                int idx = 0;
                board = queue.poll();
                for (int j = 0; j < 6; j++) {
                    if (board[j / 3 ] [j % 3] == 0) {
                        idx = j;
                        break;
                    }
                }

                int x = idx / 3;
                int y = idx % 3;

                for (int j = 0; j < 4; j++) {
                    int xnew = x + xDir[j];
                    int ynew = y + yDir[j];

                    if (xnew >= 0 && xnew < 2 && ynew >= 0 && ynew < 3) {
                        int[][] arr = new int[2][3];
                        for (int k = 0; k < 6; k++) {
                            arr[k / 3][k % 3] = board[k / 3][k % 3];
                        }

                        int temp = arr[xnew][ynew];
                        arr[xnew][ynew] = arr[x][y];
                        arr[x][y] = temp;

                        int mark = trans(arr);
//                        System.out.println(mark + "...");
                        if (mark == target) {
                            return steps;
                        }
                        if (set.contains(mark)) {
                            continue;
                        }
                        set.add(mark);
                        queue.offer(arr);
                    }
                }
            }
        }


        return -1;
    }

    public int trans (int[][] board) {
        return board[0][0] * 100000 + board[0][1] * 10000 + board[0][2] * 1000 + board[1][0]
                * 100 + board[1][1] * 10 + board[1][2];
    }
}
