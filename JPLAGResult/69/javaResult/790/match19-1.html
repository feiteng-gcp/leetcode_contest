class Solution {
    public boolean isIdealPermutation(int[] A) {
        if (A.length < 2) return true;
        int []dp = new int[A.length];
        dp[0] = A[0];
        dp[1] = Math.max(A[0], A[1]);
        
for (int i = 2; i < A.length; i++) {
            if (A[i] < dp[i-2]) return false;
            dp[i] = Math.max(dp[i-1], A[i]);
        }
        return true;
    }
}
