public class Solution {
	private int mergeSort(int[] arr) {
		int n = arr.length;
		int tmp[] = new int[n];
		return mergeSort(arr, tmp, 0, n - 1);
	}
	
	private int mergeSort(int[] arr, int[] tmp, int left, int right) {
		int mid, inv_count = 0;
		if (right > left) {
			mid = left + (right - left) /2 ;
			inv_count = mergeSort(arr, tmp, left, mid);
			inv_count += mergeSort(arr, tmp, mid + 1, right);
			
			inv_count += merge(arr, tmp, left, mid + 1, right);
		}
		return inv_count;
	}

	private int merge(int[] arr, int[] tmp, int left, int mid, int right) {
		int i = left, j = mid, k = left, inv_count = 0;
		while (i <= mid - 1 && j <= right) {
			if (arr[i] <= arr[j]) {
				tmp[k++] = arr[i++];
			} else {
				tmp[k++] = arr[j++];
				inv_count += (mid - i);
			}
		}
		while (j <= right)
			tmp[k++] = arr[j++];
		while (i <= mid - 1)
			tmp[k++] = arr[i++];
		
		System.arraycopy(tmp, left, arr, left, right - left + 1);
		return inv_count;
	}

boolean isIdealPermutation(int[] A) {
		int localInvs = 0;
		for (int i = 0; i < A.length - 1; ++i) {
			if (A[i] > A[i + 1])
				localInvs++;
		}
		int globalInvs = mergeSort(A);
		return globalInvs == localInvs;
	}
}
