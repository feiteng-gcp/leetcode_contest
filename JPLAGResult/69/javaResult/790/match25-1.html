class Solution {
    public boolean isIdealPermutation(int[] A) {
        int n = A.length;
        if (n < 3) {
            return true;
        }
        int min = A[n - 1];
        for (int i = n - 3; i >= 0; i--) 
{
            if (A[i] > min) {
                return false;
            }
            if (A[i + 1] < min) {
                min = A[i + 1];
            }
        }
        return true;
    }
}
