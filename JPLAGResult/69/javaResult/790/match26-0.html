class Solution {
    public boolean isIdealPermutation(int[] A) {
        int[] rightmin = new int[A.length];
        
        for(int i = A.length-1; i>= 0; i--){
            if(i == A.length-1){
                rightmin[i] = A[i];
                continue;
            }
            
            rightmin[i] = Math.min(A[i], rightmin[i+1]);
        }
        int i = 0, j = 1;
        
        while(i< A.length){
            if(j< A.length && A[i]> rightmin[j]){
                j++;
            }
            else{
                if(j-i>2){
                    return false;
                }
                i++;
            }
        }
        
        return true;
        
    }
}
