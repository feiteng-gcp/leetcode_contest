class Solution {
    public boolean isIdealPermutation(int[] A) {
        int n = A.length;
        if(n <= 1) return true;
        int B[] = new int[n];
        System.arraycopy(A, 0, B, 0, n);
        Arrays.sort(B);
        int i=0;
        while(i < n-1) {
            if(B[i] == A[i]) {
                i++;
                continue;
            
}
            else {
                if(!(B[i] == A[i+1] && A[i] == B[i+1])) {
                    return false;
                }
                i += 2;
            }
        }
        return true;
    }
}
