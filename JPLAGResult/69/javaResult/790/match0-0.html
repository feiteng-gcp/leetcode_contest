class Solution {
    public boolean isIdealPermutation(int[] A) {
        if(A.length < 3) {
            return true;
        }
        int k = A.length;
        int min = A[k-1];
        for(int i=k-3; i>=0; i--) {
            if(A[i] > min) {
                return false;
            }
            min = Math.min(A[i+1], min);
        }
        return true;
    }
}
