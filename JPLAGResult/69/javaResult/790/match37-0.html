class Solution {
    public boolean isIdealPermutation(int[] A) {
        int ans = 0;
        for(int i = 0; i < A.length - 1; ++i) {
            if(A[i] > A[i + 1]) {
                ++ans;
            }
        }
        PriorityQueue<Integer> min = new PriorityQueue<>(), max = new PriorityQueue<>(Collections.reverseOrder());
        for(int i = 0; i < A.length - 1; ++i) {
            int t = helper(min, max, A[i], A.length);
            ans -= t;
            if(ans < 0) return false;
        }
        return ans == 0;
    }
    private int helper(PriorityQueue<Integer> min, PriorityQueue<Integer> max, int x, int n) {
        if(min.size() == 0 && max.size() == 0) {
            min.add(x);
            return x;
        } else if(min.size() == 0) {
            while(max.size() > 0 && max.peek() > x) {
                min.add(max.poll());
            }
            min.add(x);
            return x - max.size();
        } else if(max.size() == 0) {
            while(min.size() > 0 && min.peek() < x) {
                max.add(min.poll());
            }
            min.add(x);
            return x - max.size();
        } else {
            while(min.size() > 0 && min.peek() < x) {
                max.add(min.poll());
            }
            while(max.size() > 0 && max.peek() > x) {
                min.add(max.poll());
            }
            min.add(x);
            return x - max.size();
        }
    }
}
