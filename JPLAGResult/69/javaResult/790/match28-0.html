class Solution {
    private int[] pos;
    public boolean isIdealPermutation(int[] A) {
        pos = new int[A.length];
        int max=0;
        for(int i=0;i<A.length;i++){
            max = Math.max(max,A[i]);
            if(max-A[i]>1){
                return false;
            }
        }
        return true;
    }
}
