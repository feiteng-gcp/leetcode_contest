class Solution {
    public boolean isIdealPermutation(int[] A) {
        
        int local = 0;
        for(int i = 0; i < A.length - 1; i++) {
            if(A[i] > A[i+1]) {
                local++;
            }
        }
       // System.out.println("local: " + local);
        int global = 0;
        for(int i = 0; i < A.length - 1; i++) {
            for(int j = i + 1; j < A.length; j++) {
                if(A[i] > A[j]) {
                    global++;
                }
            }
        }
        // System.out.println("global: " + global);
        return local == global;
        
    }
}
