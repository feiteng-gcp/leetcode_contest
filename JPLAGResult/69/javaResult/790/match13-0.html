class Solution {
    public boolean isIdealPermutation(int[] A) {
        if (A.length <= 2) {
            return true;
        }
        int max = A[0];
        int N = A.length;
        for (int i = 2; i < N; i++) {
            if (A[i] < max) {
                return false;
            }
            max = Math.max(max, A[i - 1]);
        }
        return true;
    }
}
