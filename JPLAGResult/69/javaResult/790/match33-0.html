class Solution {
    public boolean isIdealPermutation(int[] A) {
        int local = 0;
        for (int i = 0 ; i < A.length-1; i++) {
            if (A[i] > A[i+1]) local++;
        }

        return local == globalInversion(A, 0, A.length-1);
    }

    int globalInversion(int[] v, int s, int e) {
        if (e-s<1) return 0;
        if (e == s+1) {

            if (v[e] < v[s]) {
                int tmp = v[s];
                v[s] = v[e];
                v[e] = tmp;
                return 1;
            } else {
                return 0;
            }
        }

        int mid = (s+e)/2;
        int l = globalInversion(v, s, mid);
        int r = globalInversion(v, mid+1, e);
        int ans = l+r;

        int j = mid+1;
        for (int i = s; i <= mid; i++) {
            while (j <= e && v[j] < v[i]) {
                j++;
            }
            ans += j-mid-1;
        }

        Arrays.sort(v, s, e+1);
        return ans;
    }
}
