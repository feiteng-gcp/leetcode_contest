class Solution {
    
public int local (int[] A){
        int count = 0;
        for(int i=0;i<A.length-1;i++){
            if(A[i]> A[i+1]){
                count++;
            }
        }
        return count;
    }
    public int global(int[] A, int i, int j){
        if(i>=j){
            return 0;
        }
        int n = A.length;
        int mid = (i+j)/2;
        int inv1 = global(A,i,mid);
        int inv2 = global(A,mid+1,j);
        int a = i;
        int b = mid+1;
        int tempArr[] = new int[j-i+1];
        int k=0;
        int count =0;
        while(a<=mid && b<=j)
{
            if(A[a]<=A[b]){
                tempArr[k++] = A[a];
                a++;
            }
            else {
                count+= mid-a+1;
                tempArr[k++] = A[b];
                b++;
            }
        }
        while(a<=mid){
            tempArr[k++] = A[a++];
        }
        while(b<=j){
            tempArr[k++] = A[b++];
        }
        int jp = 0;
        
for(int y=i;y<=j;y++){
            A[y] = tempArr[jp++];
        }
        return count + inv1 + inv2;
    }
    public boolean isIdealPermutation(int[] A) {
        if(local(A) == global(A,0,A.length-1)){
            return true;
        }
        return false;
    }
}
